   789)   subroutine ncd_defvar_bynf(ncid, varname, xtype, ndims, dimid, varid, &
   790)        long_name, units, cell_method, missing_value, fill_value, &
   791)        imissing_value, ifill_value, comment, flag_meanings, &
   792)        flag_values, nvalid_range )
   793)     !
   794)     ! !DESCRIPTION:
   795)     !  Define a netcdf variable
   796)     !
   797)     ! !ARGUMENTS:
   798)     class(file_desc_t) , intent(inout) :: ncid                  ! netcdf file id
   799)     character(len=*)   , intent(in)  :: varname                 ! variable name
   800)     integer            , intent(in)  :: xtype                   ! external type
   801)     integer            , intent(in)  :: ndims                   ! number of dims
   802)     integer            , intent(inout) :: varid                 ! returned var id
   803)     integer            , intent(in), optional :: dimid(:)       ! dimids
   804)     character(len=*)   , intent(in), optional :: long_name      ! attribute
   805)     character(len=*)   , intent(in), optional :: units          ! attribute
   806)     character(len=*)   , intent(in), optional :: cell_method    ! attribute
   807)     character(len=*)   , intent(in), optional :: comment        ! attribute
   808)     character(len=*)   , intent(in), optional :: flag_meanings(:) ! attribute
   809)     real(r8)           , intent(in), optional :: missing_value  ! attribute for real
   810)     real(r8)           , intent(in), optional :: fill_value     ! attribute for real
   811)     integer            , intent(in), optional :: imissing_value ! attribute for int
   812)     integer            , intent(in), optional :: ifill_value    ! attribute for int
   813)     integer            , intent(in), optional :: flag_values(:)  ! attribute for int
   814)     integer            , intent(in), optional :: nvalid_range(2)  ! attribute for int
   815)     !
   816)     ! !LOCAL VARIABLES:
   817)     integer :: n                   ! indices
   818)     integer :: ldimid(4)           ! local dimid
   819)     integer :: dimid0(1)           ! local dimid
   820)     integer :: status              ! error status 
   821)     integer :: lxtype              ! local external type (in case logical variable)
   822)     type(var_desc_t)   :: vardesc  ! local vardesc
   823)     character(len=128) :: dimname  ! temporary
   824)     character(len=256) :: str      ! temporary
   825)     character(len=*),parameter :: subname='ncd_defvar_bynf' ! subroutine name
   826)     !-----------------------------------------------------------------------
   827) 
   828)     varid = -1
   829) 
   830)     dimid0 = 0
   831)     ldimid = 0
   832)     if (present(dimid)) then
   833)        ldimid(1:ndims) = dimid(1:ndims)
   834)     else   ! ndims must be zero if dimid not present
   835)        if (ndims /= 0) then
   836)           write(iulog,*) subname//' ERROR: dimid not supplied and ndims ne 0 ',trim(varname),ndims
   837)           call shr_sys_abort(errMsg(__FILE__, __LINE__))
   838)        endif
   839)     endif
   840) 
   841)     if ( xtype == ncd_log )then
   842)        lxtype = ncd_int
   843)     else
   844)        lxtype = xtype
   845)     end if
   846)     if (masterproc .and. debug > 1) then
   847)        write(iulog,*) 'Error in defining variable = ', trim(varname)
   848)        write(iulog,*) subname//' ',trim(varname),lxtype,ndims,ldimid(1:ndims)
   849)     endif
   850) 
   851)     if (ndims >  0) then 
   852)        status = pio_inq_dimname(ncid,ldimid(ndims),dimname)
   853)     end if
   854) 
   855)     ! Define variable

   857)        status = PIO_def_var(ncid,trim(varname),lxtype,dimid(1:ndims),vardesc)
   858)     else
   859)        status = PIO_def_var(ncid,trim(varname),lxtype,dimid0        ,vardesc)
   860)     endif

   862)

   866)     if (present(long_name)) then
   867)        call ncd_putatt(ncid, varid, 'long_name', trim(long_name))
   868)     end if

   870)        status = PIO_put_att(ncid,varid,'flag_values',flag_values)
   871)        if ( .not. present(flag_meanings)) then
   872)           write(iulog,*) 'Error in defining variable = ', trim(varname)
   873)           call shr_sys_abort(" ERROR:: flag_values set -- but not flag_meanings"//errMsg(__FILE__, __LINE__))

   875)     end if

   878)           write(iulog,*) 'Error in defining variable = ', trim(varname)
   879)           call shr_sys_abort(" ERROR:: flag_meanings set -- but not flag_values"//errMsg(__FILE__, __LINE__) )

   882)           write(iulog,*) 'Error in defining variable = ', trim(varname)
   883)           call shr_sys_abort(" ERROR:: flag_meanings and flag_values dimension different"//errMsg(__FILE__, __LINE__))
   884)        end if
   885)        str = flag_meanings(1)
   886)        do n = 1, size(flag_meanings)
   887)           if ( index(flag_meanings(n), ' ') /= 0 )then
   888)              write(iulog,*) 'Error in defining variable = ', trim(varname)
   889)              call shr_sys_abort(" ERROR:: flag_meanings has an invalid space in it"//errMsg(__FILE__, __LINE__) )
   890)           end if
   891)           if ( n > 1 ) str = trim(str)//" "//flag_meanings(n)
   892)        end do
   893)        status = PIO_put_att(ncid,varid,'flag_meanings', trim(str) )
   894)     end if
   895)     if (present(comment)) then
   896)        call ncd_putatt(ncid, varid, 'comment', trim(comment))
   897)     end if
   898)     if (present(units)) then
   899)        call ncd_putatt(ncid, varid, 'units', trim(units))

   908)     if (present(missing_value)) then

   910)     end if
   911)     if (present(ifill_value)) then
   912)        call ncd_putatt(ncid, varid, '_FillValue', ifill_value, lxtype)
   913)     end if
   914)     if (present(imissing_value)) then
   915)        call ncd_putatt(ncid, varid, 'missing_value', imissing_value, lxtype)
   916)     end if
   917)     if (present(nvalid_range)) then
   918)        status = PIO_put_att(ncid,varid,'valid_range', nvalid_range )
   919)     end if
   920)     if ( xtype == ncd_log )then
   921)        status = PIO_put_att(ncid,varid,'flag_values',     (/0, 1/) )
   922)        status = PIO_put_att(ncid,varid,'flag_meanings',  "FALSE TRUE" )
   923)        status = PIO_put_att(ncid,varid,'valid_range',    (/0, 1/) )
