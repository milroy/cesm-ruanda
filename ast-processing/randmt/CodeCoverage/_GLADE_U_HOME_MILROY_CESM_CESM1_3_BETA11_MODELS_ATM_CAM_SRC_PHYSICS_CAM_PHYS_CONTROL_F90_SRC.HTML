    90) subroutine phys_ctl_readnl(nlfile)
    91) 
    92)    use namelist_utils,  only: find_group_name
    93)    use units,           only: getunit, freeunit
    94)    use mpishorthand
    95) 
    96)    character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
    97) 
    98)    ! Local variables
    99)    integer :: unitn, ierr
   100)    character(len=*), parameter :: subname = 'phys_ctl_readnl'
   101) 
   102)    namelist /phys_ctl_nl/ cam_physpkg, cam_chempkg, waccmx_opt, deep_scheme, shallow_scheme, &
   103)       eddy_scheme, microp_scheme,  macrop_scheme, radiation_scheme, srf_flux_avg, &
   104)       use_subcol_microp, atm_dep_flux, history_amwg, history_vdiag, history_aerosol, history_aero_optics, &
   105)       history_eddy, history_budget,  history_budget_histfile_num, history_waccm, &
   106)       history_waccmx, history_chemistry, history_carma, history_clubb, &
   107)       do_clubb_sgs, do_tms, state_debug_checks, use_gw_oro, use_gw_front, &
   108)       use_gw_convect
   109)    !-----------------------------------------------------------------------------
   110) 
   111)    if (masterproc) then
   112)       unitn = getunit()
   113)       open( unitn, file=trim(nlfile), status='old' )
   114)       call find_group_name(unitn, 'phys_ctl_nl', status=ierr)
   115)       if (ierr == 0) then
   116)          read(unitn, phys_ctl_nl, iostat=ierr)
   117)          if (ierr /= 0) then
   118)             call endrun(subname // ':: ERROR reading namelist')
   119)          end if
   120)       end if
   121)       close(unitn)
   122)       call freeunit(unitn)
   123)    end if
   124) 
   125) #ifdef SPMD
   126)    ! Broadcast namelist variables
   127)    call mpibcast(deep_scheme,      len(deep_scheme)      , mpichar, 0, mpicom)
   128)    call mpibcast(cam_physpkg,      len(cam_physpkg)      , mpichar, 0, mpicom)
   129)    call mpibcast(cam_chempkg,      len(cam_chempkg)      , mpichar, 0, mpicom)
   130)    call mpibcast(waccmx_opt,       len(waccmx_opt)       , mpichar, 0, mpicom)
   131)    call mpibcast(shallow_scheme,   len(shallow_scheme)   , mpichar, 0, mpicom)
   132)    call mpibcast(eddy_scheme,      len(eddy_scheme)      , mpichar, 0, mpicom)
   133)    call mpibcast(microp_scheme,    len(microp_scheme)    , mpichar, 0, mpicom)
   134)    call mpibcast(radiation_scheme, len(radiation_scheme) , mpichar, 0, mpicom)
   135)    call mpibcast(macrop_scheme,    len(macrop_scheme)    , mpichar, 0, mpicom)
   136)    call mpibcast(srf_flux_avg,                    1 , mpiint,  0, mpicom)
   137)    call mpibcast(use_subcol_microp,               1 , mpilog,  0, mpicom)
   138)    call mpibcast(atm_dep_flux,                    1 , mpilog,  0, mpicom)
   139)    call mpibcast(history_amwg,                    1 , mpilog,  0, mpicom)
   140)    call mpibcast(history_vdiag,                   1 , mpilog,  0, mpicom)
   141)    call mpibcast(history_eddy,                    1 , mpilog,  0, mpicom)
   142)    call mpibcast(history_aerosol,                 1 , mpilog,  0, mpicom)
   143)    call mpibcast(history_aero_optics,             1 , mpilog,  0, mpicom)
   144)    call mpibcast(history_budget,                  1 , mpilog,  0, mpicom)
   145)    call mpibcast(history_budget_histfile_num,     1 , mpiint,  0, mpicom)
   146)    call mpibcast(history_waccm,                   1 , mpilog,  0, mpicom)
   147)    call mpibcast(history_waccmx,                  1 , mpilog,  0, mpicom)
   148)    call mpibcast(history_chemistry,               1 , mpilog,  0, mpicom)
   149)    call mpibcast(history_carma,                   1 , mpilog,  0, mpicom)
   150)    call mpibcast(history_clubb,                   1 , mpilog,  0, mpicom)
   151)    call mpibcast(do_clubb_sgs,                    1 , mpilog,  0, mpicom)
   152)    call mpibcast(do_tms,                          1 , mpilog,  0, mpicom)
   153)    call mpibcast(state_debug_checks,              1 , mpilog,  0, mpicom)
   154)    call mpibcast(use_gw_oro,                      1 , mpilog,  0, mpicom)
   155)    call mpibcast(use_gw_front,                    1 , mpilog,  0, mpicom)
   156)    call mpibcast(use_gw_convect,                  1 , mpilog,  0, mpicom)
   157) #endif
   158) 
   159)    ! Error checking:
   160) 
   161)    ! Check compatibility of eddy & shallow schemes
   162)    if (( shallow_scheme .eq. 'UW' ) .and. ( eddy_scheme .ne. 'diag_TKE' )) then
   163)       write(iulog,*)'Do you really want to run UW shallow scheme without diagnostic TKE eddy scheme? Quiting'
   164)       call endrun('shallow convection and eddy scheme may be incompatible')
   165)    endif
   166) 
   167)    if (( shallow_scheme .eq. 'Hack' ) .and. ( ( eddy_scheme .ne. 'HB' ) .and. ( eddy_scheme .ne. 'HBR' ))) then
   168)       write(iulog,*)'Do you really want to run Hack shallow scheme with a non-standard eddy scheme? Quiting.'
   169)       call endrun('shallow convection and eddy scheme may be incompatible')
   170)    endif
   171) 
   172)    ! Check compatibility of PBL and Microphysics schemes
   173)    if (( eddy_scheme .eq. 'diag_TKE' ) .and. ( microp_scheme .eq. 'RK' )) then
   174)       write(iulog,*)'UW PBL is not compatible with RK microphysics.  Quiting'
   175)       call endrun('PBL and Microphysics schemes incompatible')
   176)    endif
   177)    
   178)    ! Add a check to make sure CLUBB and MG are used together
   179)    if ( do_clubb_sgs .and. ( microp_scheme .ne. 'MG')) then
   180)       write(iulog,*)'CLUBB is only compatible with MG microphysics.  Quiting'
   181)       call endrun('CLUBB and microphysics schemes incompatible')
   182)    endif
   183) 
   184)    ! Check that eddy_scheme, macrop_scheme, shallow_scheme are all set to CLUBB_SGS if do_clubb_sgs is true
   185)    if (do_clubb_sgs) then
   186)       if (eddy_scheme .ne. 'CLUBB_SGS' .or. macrop_scheme .ne. 'CLUBB_SGS' .or. shallow_scheme .ne. 'CLUBB_SGS') then
   187)          write(iulog,*)'eddy_scheme, macrop_scheme and shallow_scheme must all be CLUBB_SGS.  Quiting'
   188)          call endrun('CLUBB and eddy, macrop or shallow schemes incompatible')
   189)       endif
   190)    endif
   191)       
   192) 
   193) 
   194)    ! prog_modal_aero determines whether prognostic modal aerosols are present in the run.
   195)    prog_modal_aero = (     cam_chempkg_is('trop_mam3') &
   196)                       .or. cam_chempkg_is('trop_mam7') &
   197)                       .or. cam_chempkg_is('super_fast_llnl_mam3') &
   198)                       .or. cam_chempkg_is('trop_mozart_mam3') &
   199)                       .or. cam_chempkg_is('trop_strat_mam3') &
   200)                       .or. cam_chempkg_is('trop_strat_mam7') &
   201)                       .or. cam_chempkg_is('waccm_mozart_mam3'))
   202) end subroutine phys_ctl_readnl
