   139) 
   140)       integer (kind=int_kind), dimension (nx_block*ny_block), &
   141)          intent(in) :: &
   142)          indxi, indxj     ! compressed indices for cells with ice
   143) 
   144)       real (kind=dbl_kind), intent(in) :: &
   145)          dt      ! time step
   146) 
   147)       ! ice state variables
   148)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   149)          intent(inout) :: &
   150)          aicen , & ! concentration of ice
   151)          vicen , & ! volume per unit area of ice          (m)

   159)          intent(inout) :: &
   160)          eicen     ! energy of melting for each ice layer (J/m^2)
   161) 
   162)       real (kind=dbl_kind), dimension(nx_block,ny_block,nslyr), &
   163)          intent(inout) :: &
   164)          esnon     ! energy of melting for each snow layer (J/m^2)
   165) 
   166)       ! input from atmosphere
   167)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   168)          intent(in) :: &
   169)          flw     , & ! incoming longwave radiation (W/m^2)
   170)          potT    , & ! air potential temperature  (K) 
   171)          Qa      , & ! specific humidity (kg/kg) 
   172)          rhoa    , & ! air density (kg/m^3) 
   173)          fsnow   , & ! snowfall rate (kg m-2 s-1)
   174)          shcoef  , & ! transfer coefficient for sensible heat
   175)          lhcoef      ! transfer coefficient for latent heat
   176) 
   177)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   178)          intent(inout) :: &
   179)          fswsfc  , & ! SW absorbed at ice/snow surface (W m-2)
   180)          fswint  , & ! SW absorbed in ice interior, below surface (W m-2)
   181)          fswthrun    ! SW through ice to ocean         (W/m^2)
   182) 
   183)       real (kind=dbl_kind), dimension (nx_block,ny_block,nslyr), &
   184)          intent(inout) :: &
   185)          Sswabs      ! SW radiation absorbed in snow layers (W m-2)
   186) 
   187)       real (kind=dbl_kind), dimension (nx_block,ny_block,nilyr), &
   188)          intent(inout) :: &
   189)          Iswabs      ! SW radiation absorbed in ice layers (W m-2)
   190) 
   191)       ! input from ocean
   192)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(in) :: &
   193)          fbot    , & ! ice-ocean heat flux at bottom surface (W/m^2)
   194)          Tbot        ! ice bottom surface temperature (deg C)
   195) 
   196)       ! coupler fluxes to atmosphere
   197)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(out):: &
   198)          fsensn  , & ! sensible heat flux (W/m^2) 
   199)          fswabsn , & ! shortwave flux absorbed in ice and ocean (W/m^2) 
   200)          flwoutn , & ! outgoing longwave radiation (W/m^2) 
   201)          evapn       ! evaporative water flux (kg/m^2/s) 
   202) 
   203)       ! Note: these are intent out if calc_Tsfc = T, otherwise intent in
   204)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(inout):: &
   205)          flatn    , & ! latent heat flux   (W/m^2) 
   206)          fsurfn   , & ! net flux to top surface, excluding fcondtopn
   207)          fcondtopn    ! downward cond flux at top surface (W m-2)
   208) 
   209)       ! coupler fluxes to ocean
   210)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(out):: &
   211)          freshn  , & ! fresh water flux to ocean (kg/m^2/s)
   212)          fsaltn  , & ! salt flux to ocean (kg/m^2/s)
   213)          fhocnn      ! net heat flux to ocean (W/m^2) 
   214) 
   215)       ! diagnostic fields
   216)       real (kind=dbl_kind), dimension(nx_block,ny_block), &
   217)          intent(inout):: &
   218)          meltt    , & ! top ice melt             (m/step-->cm/day) 
   219)          melts    , & ! snow melt                (m/step-->cm/day) 
   220)          meltb    , & ! basal ice melt           (m/step-->cm/day) 
   221)          congel   , & ! basal ice growth         (m/step-->cm/day) 
   222)          snoice   , & ! snow-ice formation       (m/step-->cm/day) 
   223)          mlt_onset, & ! day of year that sfc melting begins 
   224)          frz_onset    ! day of year that freezing begins (congel or frazil) 
   225) 
   226)       real (kind=dbl_kind), intent(in) :: &
   227)          yday      ! day of year
   228) 
   229)       logical (kind=log_kind), intent(out) :: &

   231) 
   232)       integer (kind=int_kind), intent(out) :: &
   233)          istop, jstop    ! indices of grid cell where code aborts

   236) !
   237)       integer (kind=int_kind) :: &
   238)          i, j        , & ! horizontal indices
   239)          ij          , & ! horizontal index, combines i and j loops
   240)          ilo,ihi,jlo,jhi, & ! beginning and end of physical domain
   241)          k           , & ! ice layer index
   242)          il1, il2    , & ! ice layer indices for eice
   243)          sl1, sl2        ! snow layer indices for esno
   244) 
   245)       real (kind=dbl_kind) :: &
   246)          dhi         , & ! change in ice thickness
   247)          dhs             ! change in snow thickness
   248) 
   249) ! 2D state variables (thickness, temperature, enthalpy)
   250) 
   251)       real (kind=dbl_kind), dimension (icells) :: &
   252)          hilyr       , & ! ice layer thickness
   253)          hslyr       , & ! snow layer thickness
   254)          Tsf         , & ! ice/snow top surface temp, same as Tsfcn (deg C)
   255)          hin         , & ! ice thickness (m)
   256)          hsn         , & ! snow thickness (m)
   257)          hsn_new     , & ! thickness of new snow (m)
   258)          worki       , & ! local work array
   259)          works           ! local work array
   260) 
   261)       real (kind=dbl_kind), dimension (icells,nilyr) :: &
   262)          qin         , & ! ice layer enthalpy, qin < 0 (J m-3)
   263)          Tin             ! internal ice layer temperatures
   264) 
   265)       real (kind=dbl_kind), dimension (icells,nslyr) :: &
   266)          qsn         , & ! snow layer enthalpy, qsn < 0 (J m-3)
   267)          Tsn             ! internal snow layer temperatures
   268) 
   269) ! other 2D flux and energy variables
   270) 
   271)       real (kind=dbl_kind), dimension (icells) :: &
   272)          fcondbot    , & ! downward cond flux at bottom surface (W m-2)
   273)          einit       , & ! initial energy of melting (J m-2)
   274)          efinal          ! final energy of melting (J m-2)
   275) 
   276) ! ech: the size of these arrays should be reduced to icells
   277)       real (kind=dbl_kind), dimension (nx_block,ny_block) :: &
   278)          Tsfcn, & ! temperature of ice/snow top surface  (C)
   279)          iage     ! ice age (s)
   280) 
   281)       !-----------------------------------------------------------------
   282)       ! Initialize
   283)       !-----------------------------------------------------------------
   284) 
   285)       l_stop = .false.
   286)       istop = 0
   287)       jstop = 0
   288) 
   289)       do j=1, ny_block
   290)       do i=1, nx_block
   291)          fsensn (i,j) = c0
   292)          fswabsn(i,j) = c0
   293)          flwoutn(i,j) = c0
   294)          evapn  (i,j) = c0
   295) 
   296)          freshn (i,j) = c0
   297)          fsaltn (i,j) = c0
   298)          fhocnn (i,j) = c0
   299) 
   300)          meltt  (i,j) = c0
   301)          meltb  (i,j) = c0
   302)          melts  (i,j) = c0
   303)          congel (i,j) = c0
   304)          snoice (i,j) = c0
   305) 
   306)          Tsfcn(i,j) = trcrn(i,j,nt_Tsfc)
   307)          if (tr_iage) iage(i,j) = trcrn(i,j,nt_iage)
   308)       enddo
   309)       enddo
   310) 
   311)       if (calc_Tsfc) then
   312)          do j=1, ny_block
   313)          do i=1, nx_block
   314)             flatn    (i,j) = c0
   315)             fsurfn   (i,j) = c0
   316)             fcondtopn(i,j) = c0
   317)          enddo
   318)          enddo
   319)       endif
   320) 
   321)       !-----------------------------------------------------------------
   322)       ! Compute variables needed for vertical thermo calculation
   323)       !-----------------------------------------------------------------
   324) 
   325)       call init_vertical_profile (nx_block,     ny_block,     &
   326)                                   my_task,      istep1,       &
   327)                                   icells,                     &
   328)                                   indxi,        indxj,        &
   329)                                   aicen(:,:),                 &
   330)                                   vicen(:,:),   vsnon(:,:),   &
   331)                                   Tsfcn(:,:),                 &
   332)                                   eicen(:,:,:), esnon(:,:,:), &
   333)                                   hin,          hilyr,        &
   334)                                   hsn,          hslyr,        &
   335)                                   qin,          Tin,          &
   336)                                   qsn,          Tsn,          &
   337)                                   Tsf,          einit,        &
   338)                                   l_stop,                     &
   339)                                   istop,        jstop)
   340) 
   341)       if (l_stop) return
   342) 
   343)       do ij = 1, icells
   344)          ! Save initial ice and snow thickness (for fresh and fsalt)
   345)          worki(ij) = hin(ij)
   346)          works(ij) = hsn(ij)
   347)       enddo
   348)    
   349)       !-----------------------------------------------------------------
   350)       ! Compute new surface temperature and internal ice and snow
   351)       !  temperatures.
   352)       !-----------------------------------------------------------------
   353) 
   354)       if (heat_capacity) then   ! usual case
   355) 
   356)          call temperature_changes(nx_block,      ny_block, &
   357)                                   my_task,       istep1,   &
   358)                                   dt,            icells,   & 
   359)                                   indxi,         indxj,    &
   360)                                   rhoa,          flw,      &
   361)                                   potT,          Qa,       &
   362)                                   shcoef,        lhcoef,   &
   363)                                   fswsfc,        fswint,   &
   364)                                   fswthrun,      Sswabs(:,:,:),&
   365)                                   Iswabs,                  &
   366)                                   hilyr,         hslyr,    &
   367)                                   qin,           Tin,      &
   368)                                   qsn,           Tsn,      &
   369)                                   Tsf,           Tbot,     &
   370)                                   fsensn,        flatn,    &
   371)                                   fswabsn,       flwoutn,  &
   372)                                   fsurfn,                  &
   373)                                   fcondtopn,     fcondbot, &
   374)                                   einit,         l_stop,   &
   375)                                   istop,         jstop)
   376) 
   377)       else
   378) 
   379)          if (calc_Tsfc) then       
   380) 
   381)             call zerolayer_temperature(nx_block,      ny_block, &
   382)                                        my_task,       istep1,   &
   383)                                        dt,            icells,   & 
   384)                                        indxi,         indxj,    &
   385)                                        rhoa,          flw,      &
   386)                                        potT,          Qa,       &
   387)                                        shcoef,        lhcoef,   &
   388)                                        fswsfc,        fswthrun, &
   389)                                        hilyr,         hslyr,    &
   390)                                        Tsf,           Tbot,     &
   391)                                        fsensn,        flatn,    &
   392)                                        fswabsn,       flwoutn,  &
   393)                                        fsurfn,                  &
   394)                                        fcondtopn,     fcondbot, &
   395)                                        l_stop,                  &
   396)                                        istop,         jstop)
   397) 
   398)          else
   399) 
   400)             !------------------------------------------------------------
   401)             ! Set fcondbot = fcondtop for zero layer thermodynamics
   402)             ! fcondtop is set in call to set_sfcflux in step_therm1
   403)             !------------------------------------------------------------
   404) 
   405)             do ij = 1, icells
   406)                i = indxi(ij)
   407)                j = indxj(ij)
   408)                fcondbot(ij)  = fcondtopn(i,j)   ! zero layer         
   409)             enddo
   410)       
   411)          endif      ! calc_Tsfc
   412) 
   413)       endif         ! heat_capacity
   414) 
   415)       if (l_stop) return
   416) 
   417)       !-----------------------------------------------------------------
   418)       ! Compute growth and/or melting at the top and bottom surfaces.
   419)       ! Add new snowfall.
   420)       ! Repartition ice into equal-thickness layers, conserving energy.
   421)       !-----------------------------------------------------------------
   422) 
   423)       call thickness_changes(nx_block,     ny_block, &
   424)                              dt,                     &
   425)                              yday,         icells,   &
   426)                              indxi,        indxj,    &
   427)                              efinal,                 &
   428)                              hin,          hilyr,    &
   429)                              hsn,          hslyr,    &
   430)                              qin,          qsn,      &
   431)                              fbot,         Tbot,     &
   432)                              flatn,        fsurfn,   &
   433)                              fcondtopn,    fcondbot, &
   434)                              fsnow,        hsn_new,  &
   435)                              fhocnn,       evapn,    &
   436)                              meltt,        melts,    &
   437)                              meltb,        iage,     &
   438)                              congel,       snoice,   &
   439)                              mlt_onset,    frz_onset)
   440) 
   441)       !-----------------------------------------------------------------
   442)       ! Check for energy conservation by comparing the change in energy
   443)       ! to the net energy input
   444)       !-----------------------------------------------------------------
   445) 
   446)       call conservation_check_vthermo(nx_block, ny_block, &
   447)                                       my_task,  istep1,   &
   448)                                       dt,       icells,   &
   449)                                       indxi,    indxj,    &
   450)                                       fsurfn,   flatn,    &
   451)                                       fhocnn,   fswint,   &
   452)                                       fsnow,              &
   453)                                       einit,    efinal,   &
   454)                                       l_stop,             &
   455)                                       istop,    jstop)
   456) 
   457)       if (l_stop) return
   458) 
   459)       !-----------------------------------------------------------------
   460)       ! If prescribed ice, set hi back to old values
   461)       !-----------------------------------------------------------------
   462) 
   463)       if (prescribed_ice) then
   464)          do ij = 1, icells
   465)             i = indxi(ij)
   466)             j = indxj(ij)
   467)             hin(ij) = worki(ij)
   468)             fhocnn(i,j) = c0             ! for diagnostics
   469)          enddo                  ! ij
   470)       endif
   471) 
   472)       !-----------------------------------------------------------------
   473)       ! Compute fluxes of water and salt from ice to ocean.
   474)       ! evapn < 0 => sublimation, evapn > 0 => condensation
   475)       !-----------------------------------------------------------------
   476) 
   477)       do ij = 1, icells
   478)          i = indxi(ij)
   479)          j = indxj(ij)
   480)             
   481)          dhi = hin(ij) - worki(ij)
   482)          dhs = hsn(ij) - works(ij)
   483)                
   484)          freshn(i,j) = evapn(i,j) - &
   485)                        (rhoi*dhi + rhos*(dhs-hsn_new(ij))) / dt
   486)          fsaltn(i,j) = -rhoi*dhi*ice_ref_salinity*p001/dt
   487) 
   488)       enddo                     ! ij
   489) 
   490)       !-----------------------------------------------------------------
   491)       !  Given the vertical thermo state variables (hin, hsn, Tsf,
   492)       !   qin, qsn,), compute the new ice state variables (vicen, vsnon,
   493)       !   Tsfcn, eicen, esnon).
   494)       !-----------------------------------------------------------------
   495) 
   496)       call update_state_vthermo(nx_block,     ny_block,   &
   497)                                 icells,                   &
   498)                                 indxi,        indxj,      &
   499)                                 Tbot,         Tsf,        &     
   500)                                 hin,          hsn,        &
   501)                                 qin,          qsn,        &
   502)                                 aicen(:,:),               &
   503)                                 vicen(:,:),   vsnon(:,:), &
   504)                                 Tsfcn(:,:),               &
   505)                                 eicen(:,:,:), esnon(:,:,:))
   506) 
   507)       !-----------------------------------------------------------------
   508)       ! Reload tracer array
   509)       !-----------------------------------------------------------------
   510) 
   511)       do j = 1, ny_block
   512)       do i = 1, nx_block
   513)          trcrn(i,j,nt_Tsfc) = Tsfcn(i,j)
   514)          if (tr_iage) trcrn(i,j,nt_iage) = iage(i,j)
   515)       enddo
   516)       enddo
   517) 
   518)       end subroutine thermo_vertical

   550)       real (kind=dbl_kind)    :: zn       ! normalized ice thickness

   552)       !-----------------------------------------------------------------
   553)       ! Determine l_brine based on saltmax.
   554)       ! Thermodynamic solver will not converge if l_brine is true and
   555)       !  saltmax is close to zero.
   556)       ! Set l_brine to false for zero layer thermodynamics
   557)       !-----------------------------------------------------------------
   558) 
   559)       if (saltmax > min_salin .and. heat_capacity) then
   560)          l_brine = .true.
   561)       else
   562)          l_brine = .false.
   563)       endif
   564) 
   565)       !-----------------------------------------------------------------
   566)       ! Prescibe vertical profile of salinity and melting temperature.
   567)       !-----------------------------------------------------------------
   568) 
   569)       if (l_brine) then
   570)          do k = 1, nilyr
   571)             zn = (real(k,kind=dbl_kind)-p5) /  &
   572)                   real(nilyr,kind=dbl_kind)
   573)             salin(k)=(saltmax/c2)*(c1-cos(pi*zn**(nsal/(msal+zn))))
   574) !            salin(k)=saltmax ! for isosaline ice
   575)             Tmlt(k) = -salin(k)*depressT
   576)          enddo
   577)          salin(nilyr+1) = saltmax
   578)          Tmlt(nilyr+1) = -salin(nilyr+1)*depressT
   579)       else
   580)          do k = 1, nilyr+1
   581)             salin(k) = c0
   582)             Tmlt(k) = c0
   583)          enddo
   584)       endif
   585) 
   586)       ustar_scale = c1           ! for nonzero currents
   587) 
   588)       end subroutine init_thermo_vertical

   593) ! !ROUTINE: frzmlt_bottom_lateral - bottom and lateral heat fluxes
   594) !
   595) ! !DESCRIPTION:
   596) !
   597) ! Adjust frzmlt to account for changes in fhocn since from_coupler.
   598) ! Compute heat flux to bottom surface.
   599) ! Compute fraction of ice that melts laterally.
   600) !
   601) ! !REVISION HISTORY:
   602) !
   603) ! authors C. M. Bitz, UW
   604) !         William H. Lipscomb, LANL
   605) !         Elizabeth C. Hunke, LANL
   606) !
   607) ! !INTERFACE:
   608) !
   609)       subroutine frzmlt_bottom_lateral (nx_block, ny_block, &
   610)                                         ilo, ihi, jlo, jhi, &
   611)                                         dt,                 &
   612)                                         aice,     frzmlt,   &
   613)                                         eicen,    esnon,    &
   614)                                         sst,      Tf,       &
   615)                                         strocnxT, strocnyT, &
   616)                                         Tbot,     fbot,     &
   617)                                         rside)
   618) !
   619) ! !USES:
   620) !
   621)       use ice_itd, only: ilyr1, slyr1
   622) 
   623) ! !INPUT/OUTPUT PARAMETERS:
   624) 
   625)       integer (kind=int_kind), intent(in) :: &
   626)          nx_block, ny_block, & ! block dimensions
   627)          ilo,ihi,jlo,jhi       ! beginning and end of physical domain
   628) 
   629)       real (kind=dbl_kind), intent(in) :: &
   630)          dt                  ! time step
   631) 
   632)       real (kind=dbl_kind), dimension(nx_block,ny_block), intent(in) :: &
   633)          aice    , & ! ice concentration
   634)          frzmlt  , & ! freezing/melting potential (W/m^2)
   635)          sst     , & ! sea surface temperature (C)
   636)          Tf      , & ! freezing temperature (C)
   637)          strocnxT, & ! ice-ocean stress, x-direction
   638)          strocnyT    ! ice-ocean stress, y-direction
   639) 
   640)       real (kind=dbl_kind), dimension(nx_block,ny_block,ntilyr), &
   641)          intent(in) :: &
   642)          eicen       ! energy of melting for each ice layer (J/m^2)
   643) 
   644)       real (kind=dbl_kind), dimension(nx_block,ny_block,ntslyr), &
   645)          intent(in) :: &
   646)          esnon       ! energy of melting for each snow layer (J/m^2)
   647) 
   648)       real (kind=dbl_kind), dimension(nx_block,ny_block), &
   649)          intent(out) :: &
   650)          Tbot    , & ! ice bottom surface temperature (deg C)
   651)          fbot    , & ! heat flux to ice bottom  (W/m^2)
   652)          rside       ! fraction of ice that melts laterally
   653) !
   654) !EOP
   655) !
   656)       integer (kind=int_kind) :: &
   657)          i, j           , & ! horizontal indices
   658)          n              , & ! thickness category index
   659)          k              , & ! layer index
   660)          ij             , & ! horizontal index, combines i and j loops
   661)          imelt              ! number of cells with ice melting
   662) 
   663)       integer (kind=int_kind), dimension (nx_block*ny_block) :: &
   664)          indxi, indxj     ! compressed indices for cells with ice melting
   665) 
   666)       real (kind=dbl_kind), dimension (:), allocatable :: &
   667)          etot    , & ! total energy in column
   668)          fside       ! lateral heat flux (W/m^2)
   669) 
   670)       real (kind=dbl_kind) :: &
   671)          deltaT    , & ! SST - Tbot >= 0
   672)          ustar     , & ! skin friction velocity for fbot (m/s)
   673)          wlat      , & ! lateral melt rate (m/s)
   674)          xtmp          ! temporary variable
   675) 
   676)       ! Parameters for bottom melting
   677) 
   678)       ! 0.006 = unitless param for basal heat flx ala McPhee and Maykut
   679) 
   680)       real (kind=dbl_kind), parameter :: &
   681)          cpchr = -cp_ocn*rhow*0.006_dbl_kind, &
   682)          ustar_min = 5.e-3_dbl_kind
   683) 
   684) 
   685)       ! Parameters for lateral melting
   686) 
   687)       real (kind=dbl_kind), parameter :: &
   688)          floediam = 300.0_dbl_kind, & ! effective floe diameter (m)
   689)          alpha    = 0.66_dbl_kind , & ! constant from Steele (unitless)
   690)          m1 = 1.6e-6_dbl_kind     , & ! constant from Maykut & Perovich
   691)                                       ! (m/s/deg^(-m2))
   692)          m2 = 1.36_dbl_kind           ! constant from Maykut & Perovich
   693)                                       ! (unitless)
   694) 
   695)       do j = 1, ny_block
   696)       do i = 1, nx_block
   697)          rside(i,j) = c0
   698)          Tbot (i,j) = Tf(i,j)
   699)          fbot (i,j) = c0
   700)       enddo
   701)       enddo
   702) 
   703)       !-----------------------------------------------------------------
   704)       ! Identify grid cells where ice can melt.
   705)       !-----------------------------------------------------------------
   706) 
   707)       imelt = 0
   708)       do j = jlo, jhi
   709)       do i = ilo, ihi
   710)          if (aice(i,j) > puny .and. frzmlt(i,j) < c0) then ! ice can melt
   711)             imelt = imelt + 1
   712)             indxi(imelt) = i
   713)             indxj(imelt) = j
   714)          endif
   715)       enddo                     ! i
   716)       enddo                     ! j
   717) 
   718)       allocate(etot (imelt))
   719)       allocate(fside(imelt))
   720) 
   721)       do ij = 1, imelt  ! cells where ice can melt
   722)          i = indxi(ij)
   723)          j = indxj(ij)
   724) 
   725)          fside(ij) = c0
   726) 
   727)       !-----------------------------------------------------------------
   728)       ! Use boundary layer theory for fbot.
   729)       ! See Maykut and McPhee (1995): JGR, 100, 24,691-24,703.
   730)       !-----------------------------------------------------------------
   731) 
   732)          deltaT = max((sst(i,j)-Tbot(i,j)),c0)
   733) 
   734)          ! strocnx has units N/m^2 so strocnx/rho has units m^2/s^2
   735)          ustar = sqrt (sqrt(strocnxT(i,j)**2+strocnyT(i,j)**2)/rhow)
   736)          ustar = max (ustar,ustar_min*ustar_scale)
   737) 
   738)          fbot(i,j) = cpchr * deltaT * ustar ! < 0
   739)          fbot(i,j) = max (fbot(i,j), frzmlt(i,j)) ! frzmlt < fbot < 0
   740) 
   741) !!! uncomment to use all frzmlt for standalone runs
   742) !!!         fbot(i,j) = min (c0, frzmlt(i,j))
   743) 
   744)       !-----------------------------------------------------------------
   745)       ! Compute rside.  See these references:
   746)       !    Maykut and Perovich (1987): JGR, 92, 7032-7044
   747)       !    Steele (1992): JGR, 97, 17,729-17,738
   748)       !-----------------------------------------------------------------
   749) 
   750)          wlat = m1 * deltaT**m2 ! Maykut & Perovich
   751)          rside(i,j) = wlat*dt*pi/(alpha*floediam) ! Steele
   752)          rside(i,j) = max(c0,min(rside(i,j),c1))
   753) 
   754)       enddo                     ! ij
   755) 
   756)       !-----------------------------------------------------------------
   757)       ! Compute heat flux associated with this value of rside.
   758)       !-----------------------------------------------------------------
   759) 
   760)       do n = 1, ncat
   761) 
   762)          do ij = 1, imelt
   763)             etot(ij) = c0
   764)          enddo
   765) 
   766)          ! melting energy/unit area in each column, etot < 0
   767) 
   768)          do k = 1, nslyr
   769) !DIR$ CONCURRENT !Cray
   770) !cdir nodep      !NEC
   771) !ocl novrec      !Fujitsu
   772)             do ij = 1, imelt
   773)                i = indxi(ij)
   774)                j = indxj(ij)
   775)                etot(ij) = etot(ij) + esnon(i,j,slyr1(n)+k-1)
   776)             enddo               ! ij
   777)          enddo
   778) 
   779)          do k = 1, nilyr
   780) !DIR$ CONCURRENT !Cray
   781) !cdir nodep      !NEC
   782) !ocl novrec      !Fujitsu
   783)             do ij = 1, imelt
   784)                i = indxi(ij)
   785)                j = indxj(ij)
   786)                etot(ij) = etot(ij) + eicen(i,j,ilyr1(n)+k-1)
   787)             enddo               ! ij
   788)          enddo                  ! nilyr
   789) 
   790) !DIR$ CONCURRENT !Cray
   791) !cdir nodep      !NEC
   792) !ocl novrec      !Fujitsu
   793)          do ij = 1, imelt
   794)             i = indxi(ij)
   795)             j = indxj(ij)
   796)             ! lateral heat flux
   797)             fside(ij) = fside(ij) + rside(i,j)*etot(ij)/dt ! fside < 0
   798)          enddo                  ! ij
   799) 
   800)       enddo                     ! n
   801) 
   802)       !-----------------------------------------------------------------
   803)       ! Limit bottom and lateral heat fluxes if necessary.
   804)       !-----------------------------------------------------------------
   805) 
   806) !DIR$ CONCURRENT !Cray
   807) !cdir nodep      !NEC
   808) !ocl novrec      !Fujitsu
   809) 
   810) 
   811)       do ij = 1, imelt
   812)          i = indxi(ij)
   813)          j = indxj(ij)
   814) 
   815)          xtmp = frzmlt(i,j)/(fbot(i,j) + fside(ij) + puny) 
   816)          xtmp = min(xtmp, c1)
   817)          fbot(i,j)  = fbot(i,j)  * xtmp
   818)          rside(i,j) = rside(i,j) * xtmp
   819)       enddo                     ! ij
   820) 
   821)       deallocate(etot)
   822)       deallocate(fside)
   823) 
   824)       end subroutine frzmlt_bottom_lateral

   829) ! !ROUTINE: init_vertical_profile - initial thickness, enthalpy, temperature
   830) !
   831) ! !DESCRIPTION:
   832) !
   833) ! Given the state variables (vicen, vsnon, eicen, esnon, Tsfcn),
   834) ! compute variables needed for the vertical thermodynamics
   835) ! (hin, hsn, qin, qsn, Tin, Tsn, Tsf).
   836) !
   837) ! !REVISION HISTORY:
   838) !
   839) ! authors William H. Lipscomb, LANL
   840) !         C. M. Bitz, UW
   841) !
   842) ! !INTERFACE:
   843) !
   844)       subroutine init_vertical_profile(nx_block, ny_block, &
   845)                                        my_task,  istep1,   &
   846)                                        icells,             &
   847)                                        indxi,    indxj,    &
   848)                                        aicen,    vicen,    &
   849)                                        vsnon,    Tsfcn,    &
   850)                                        eicen,    esnon,    &
   851)                                        hin,      hilyr,    &
   852)                                        hsn,      hslyr,    &
   853)                                        qin,      Tin,      &
   854)                                        qsn,      Tsn,      &
   855)                                        Tsf,      einit,    &
   856)                                        l_stop,             &
   857)                                        istop,    jstop)
   858) !
   859) ! !USES:
   860) !
   861) ! !INPUT/OUTPUT PARAMETERS:
   862) !
   863)       integer (kind=int_kind), intent(in) :: &
   864)          nx_block, ny_block, & ! block dimensions
   865)          my_task           , & ! task number (diagnostic only)
   866)          istep1            , & ! time step index (diagnostic only)
   867)          icells                ! number of cells with aicen > puny
   868) 
   869)       integer (kind=int_kind), dimension(nx_block*ny_block), &
   870)          intent(in) :: &
   871)          indxi, indxj    ! compressed indices for cells with aicen > puny
   872) 
   873)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(in) :: &
   874)          aicen , & ! concentration of ice
   875)          vicen , & ! volume per unit area of ice          (m)
   876)          vsnon , & ! volume per unit area of snow         (m)
   877)          Tsfcn     ! temperature of ice/snow top surface  (C)
   878) 
   879)       real (kind=dbl_kind), dimension(nx_block,ny_block,nilyr), &
   880)          intent(in) :: &
   881)          eicen     ! energy of melting for each ice layer (J/m^2)
   882) 
   883)       real (kind=dbl_kind), dimension(nx_block,ny_block,nslyr), &
   884)          intent(in) :: &
   885)          esnon     ! energy of melting for each snow layer (J/m^2)
   886) 
   887)       real (kind=dbl_kind), dimension(icells), intent(out):: &
   888)          hilyr       , & ! ice layer thickness
   889)          hslyr       , & ! snow layer thickness
   890)          Tsf         , & ! ice/snow surface temperature, Tsfcn
   891)          einit           ! initial energy of melting (J m-2)
   892) 
   893)       real (kind=dbl_kind), dimension(icells), intent(out):: &
   894)          hin         , & ! ice thickness (m)
   895)          hsn             ! snow thickness (m)
   896) 
   897)       real (kind=dbl_kind), dimension (icells,nilyr), &
   898)          intent(out) :: &
   899)          qin         , & ! ice layer enthalpy (J m-3)
   900)          Tin             ! internal ice layer temperatures
   901) 
   902)       real (kind=dbl_kind), dimension (icells,nslyr), &
   903)          intent(out) :: &
   904)          qsn         , & ! snow enthalpy
   905)          Tsn             ! snow temperature
   906) 
   907)       logical (kind=log_kind), intent(inout) :: &
   908)          l_stop          ! if true, print diagnostics and abort model
   909) 
   910)       integer (kind=int_kind), intent(inout) :: &
   911)          istop, jstop    ! i and j indices of cell where model fails
   912) !
   913) !EOP
   914) !
   915)       real (kind=dbl_kind), parameter :: &
   916)          Tmin = -100._dbl_kind ! min allowed internal temperature (deg C)
   917) 
   918)       integer (kind=int_kind) :: &
   919)          i, j        , & ! horizontal indices
   920)          ij          , & ! horizontal index, combines i and j loops
   921)          k               ! ice layer index
   922) 
   923)       real (kind=dbl_kind) :: &
   924)          rnslyr,        & ! real(nslyr)
   925)          aa1, bb1, cc1, & ! terms in quadratic formula
   926)          Tmax             ! maximum allowed snow/ice temperature (deg C)
   927) 
   928)       logical (kind=log_kind) :: &   ! for vector-friendly error checks
   929)          tsno_high   , & ! flag for Tsn > Tmax
   930)          tice_high   , & ! flag for Tin > Tmlt
   931)          tsno_low    , & ! flag for Tsn < Tmin
   932)          tice_low        ! flag for Tin < Tmin
   933) 
   934)       !-----------------------------------------------------------------
   935)       ! Initialize
   936)       !-----------------------------------------------------------------
   937) 
   938)       rnslyr = real(nslyr,kind=dbl_kind)
   939) 
   940)       do ij = 1, icells
   941)          einit(ij) = c0
   942)       enddo
   943) 
   944)       tsno_high = .false.
   945)       tice_high = .false.
   946)       tsno_low  = .false.
   947)       tice_low  = .false.
   948) 
   949)       !-----------------------------------------------------------------
   950)       ! Load arrays for vertical thermo calculation.
   951)       !-----------------------------------------------------------------
   952) !DIR$ CONCURRENT !Cray
   953) !cdir nodep      !NEC
   954) !ocl novrec      !Fujitsu
   955)       do ij = 1, icells
   956)          i = indxi(ij)
   957)          j = indxj(ij)
   958) 
   959)       !-----------------------------------------------------------------
   960)       ! Surface temperature, ice and snow thickness
   961)       ! Initialize internal energy
   962)       !-----------------------------------------------------------------
   963) 
   964)          Tsf(ij)    = Tsfcn(i,j)
   965)          hin(ij)    = vicen(i,j) / aicen(i,j)
   966)          hsn(ij)    = vsnon(i,j) / aicen(i,j)
   967)          hilyr(ij)    = hin(ij) / real(nilyr,kind=dbl_kind)
   968)          hslyr(ij)    = hsn(ij) / rnslyr
   969) 
   970)       enddo                     ! ij
   971) 
   972)       !-----------------------------------------------------------------
   973)       ! Snow enthalpy and maximum allowed snow temperature
   974)       ! If heat_capacity = F, qsn and Tsn are never used.
   975)       !-----------------------------------------------------------------
   976) 
   977)       do k = 1, nslyr
   978) !DIR$ CONCURRENT !Cray
   979) !cdir nodep      !NEC
   980) !ocl novrec      !Fujitsu
   981)          do ij = 1, icells
   982)             i = indxi(ij)
   983)             j = indxj(ij)
   984) 
   985)       !-----------------------------------------------------------------
   986)       !
   987)       ! Tmax based on the idea that dT ~ dq / (rhos*cp_ice)
   988)       !                             dq ~ q dv / v
   989)       !                             dv ~ puny = eps11
   990)       ! where 'd' denotes an error due to roundoff.
   991)       !-----------------------------------------------------------------
   992) 
   993)             if (hslyr(ij) > hs_min/rnslyr .and. heat_capacity) then
   994)                ! qsn, esnon < 0              
   995)                qsn  (ij,k) = esnon(i,j,k)*rnslyr/vsnon(i,j) 
   996)                Tmax = -qsn(ij,k)*puny*rnslyr / &
   997)                        (rhos*cp_ice*vsnon(i,j))
   998)             else
   999)                qsn  (ij,k) = -rhos * Lfresh
  1000)                Tmax = puny
  1001)             endif
  1002) 
  1003)       !-----------------------------------------------------------------
  1004)       ! Compute snow temperatures from enthalpies.
  1005)       ! Note: qsn <= -rhos*Lfresh, so Tsn <= 0.
  1006)       !-----------------------------------------------------------------
  1007)             Tsn(ij,k) = (Lfresh + qsn(ij,k)/rhos)/cp_ice
  1008)   
  1009)       !-----------------------------------------------------------------
  1010)       ! Check for Tsn > Tmax (allowing for roundoff error) and Tsn < Tmin.
  1011)       !-----------------------------------------------------------------
  1012)             if (Tsn(ij,k) > Tmax) then
  1013)                tsno_high = .true.
  1014)             elseif (Tsn(ij,k) < Tmin) then
  1015)                tsno_low  = .true.
  1016)             endif
  1017) 
  1018)          enddo                  ! ij
  1019)       enddo                     ! nslyr
  1020) 
  1021)       !-----------------------------------------------------------------
  1022)       ! If Tsn is out of bounds, print diagnostics and exit.
  1023)       !-----------------------------------------------------------------
  1024) 
  1025)       if (tsno_high .and. heat_capacity) then
  1026)          do k = 1, nslyr
  1027)             do ij = 1, icells
  1028)                i = indxi(ij)
  1029)                j = indxj(ij)
  1030) 
  1031)                if (hslyr(ij) > hs_min/rnslyr) then
  1032)                   Tmax = -qsn(ij,k)*puny*rnslyr / &
  1033)                            (rhos*cp_ice*vsnon(i,j))
  1034)                else
  1035)                   Tmax = puny
  1036)                endif
  1037) 
  1038)                if (Tsn(ij,k) > Tmax) then
  1039)                   write(nu_diag,*) ' '
  1040)                   write(nu_diag,*) 'Starting thermo, Tsn > Tmax'
  1041)                   write(nu_diag,*) 'Tsn=',Tsn(ij,k)
  1042)                   write(nu_diag,*) 'Tmax=',Tmax
  1043)                   write(nu_diag,*) 'istep1, my_task, i, j:', &
  1044)                                     istep1, my_task, i, j
  1045)                   write(nu_diag,*) 'qsn',qsn(ij,k)
  1046)                   l_stop = .true.
  1047)                   istop = i
  1048)                   jstop = j
  1049)                   return
  1050)                endif
  1051) 
  1052)             enddo               ! ij
  1053)          enddo                  ! nslyr
  1054)       endif                     ! tsno_high
  1055) 
  1056)       if (tsno_low .and. heat_capacity) then
  1057)          do k = 1, nslyr
  1058)             do ij = 1, icells
  1059)                i = indxi(ij)
  1060)                j = indxj(ij)
  1061) 
  1062)                if (Tsn(ij,k) < Tmin) then ! allowing for roundoff error
  1063)                   write(nu_diag,*) ' '
  1064)                   write(nu_diag,*) 'Starting thermo, Tsn < Tmin'
  1065)                   write(nu_diag,*) 'Tsn=', Tsn(ij,k)
  1066)                   write(nu_diag,*) 'Tmin=', Tmin
  1067)                   write(nu_diag,*) 'istep1, my_task, i, j:', &
  1068)                                     istep1, my_task, i, j
  1069)                   write(nu_diag,*) 'qsn', qsn(ij,k)
  1070)                   l_stop = .true.
  1071)                   istop = i
  1072)                   jstop = j
  1073)                   return
  1074)                endif
  1075) 
  1076)             enddo               ! ij
  1077)          enddo                  ! nslyr
  1078)       endif                     ! tsno_low
  1079) 
  1080)       do k = 1, nslyr
  1081) !DIR$ CONCURRENT !Cray
  1082) !cdir nodep      !NEC
  1083) !ocl novrec      !Fujitsu
  1084)          do ij = 1, icells
  1085) 
  1086)             if (Tsn(ij,k) > c0) then   ! correct roundoff error
  1087)                Tsn(ij,k) = c0
  1088)                qsn(ij,k) = -rhos*Lfresh
  1089)             endif
  1090) 
  1091)       !-----------------------------------------------------------------
  1092)       ! initial energy per unit area of ice/snow, relative to 0 C
  1093)       !-----------------------------------------------------------------
  1094)             einit(ij) = einit(ij) + hslyr(ij)*qsn(ij,k)
  1095) 
  1096)          enddo                  ! ij
  1097)       enddo                     ! nslyr
  1098) 
  1099)       do k = 1, nilyr
  1100) !DIR$ CONCURRENT !Cray
  1101) !cdir nodep      !NEC
  1102) !ocl novrec      !Fujitsu
  1103)          do ij = 1, icells
  1104)             i = indxi(ij)
  1105)             j = indxj(ij)
  1106) 
  1107)       !-----------------------------------------------------------------
  1108)       ! Compute ice enthalpy
  1109)       ! If heat_capacity = F, qin and Tin are never used.
  1110)       !-----------------------------------------------------------------
  1111)             ! qin, eicen < 0
  1112)             qin(ij,k) = eicen(i,j,k)*real(nilyr,kind=dbl_kind) &
  1113)                         /vicen(i,j)  
  1114) 
  1115)       !-----------------------------------------------------------------
  1116)       ! Compute ice temperatures from enthalpies using quadratic formula
  1117)       !-----------------------------------------------------------------
  1118) 
  1119)             if (l_brine) then
  1120)                aa1 = cp_ice
  1121)                bb1 = (cp_ocn-cp_ice)*Tmlt(k) - qin(ij,k)/rhoi - Lfresh 
  1122)                cc1 = Lfresh * Tmlt(k)
  1123)                Tin(ij,k) =  (-bb1 - sqrt(bb1*bb1 - c4*aa1*cc1)) /  &
  1124)                              (c2*aa1)
  1125)                Tmax = Tmlt(k)
  1126) 
  1127)             else                ! fresh ice
  1128)                Tin(ij,k) = (Lfresh + qin(ij,k)/rhoi) / cp_ice
  1129)                Tmax = -qin(ij,k)*puny/(rhos*cp_ice*vicen(i,j))
  1130)                          ! as above for snow
  1131)             endif
  1132) 
  1133)       !-----------------------------------------------------------------
  1134)       ! Check for Tin > Tmax and Tin < Tmin
  1135)       !-----------------------------------------------------------------
  1136)             if (Tin(ij,k) > Tmax) then
  1137)                tice_high = .true.
  1138)             elseif (Tin(ij,k) < Tmin) then
  1139)                tice_low  = .true.
  1140)             endif
  1141) 
  1142)          enddo                  ! ij
  1143) 
  1144)       !-----------------------------------------------------------------
  1145)       ! If Tin is out of bounds, print diagnostics and exit.
  1146)       !-----------------------------------------------------------------
  1147) 
  1148)          if (tice_high .and. heat_capacity) then
  1149)             do ij = 1, icells
  1150)                i = indxi(ij)
  1151)                j = indxj(ij)
  1152) 
  1153)                if (l_brine) then
  1154)                   Tmax = Tmlt(k)
  1155)                else             ! fresh ice
  1156)                   Tmax = -qin(ij,k)*puny/(rhos*cp_ice*vicen(i,j))
  1157)                endif
  1158) 
  1159)                if (Tin(ij,k) > Tmax) then
  1160)                   write(nu_diag,*) ' '
  1161)                   write(nu_diag,*) 'Starting thermo, T > Tmax, layer', k
  1162)                   write(nu_diag,*) 'Tin=',Tin(ij,k),', Tmax=',Tmax
  1163)                   write(nu_diag,*) 'istep1, my_task, i, j:', &
  1164)                                     istep1, my_task, i, j
  1165)                   write(nu_diag,*) 'qin',qin(ij,k)
  1166)                   l_stop = .true.
  1167)                   istop = i
  1168)                   jstop = j
  1169)                   return
  1170)                endif
  1171)             enddo               ! ij
  1172)          endif                  ! tice_high
  1173) 
  1174)          if (tice_low .and. heat_capacity) then
  1175)             do ij = 1, icells
  1176)                i = indxi(ij)
  1177)                j = indxj(ij)
  1178) 
  1179)                if (Tin(ij,k) < Tmin) then
  1180)                   write(nu_diag,*) ' '
  1181)                   write(nu_diag,*) 'Starting thermo T < Tmin, layer', k
  1182)                   write(nu_diag,*) 'Tin =', Tin(ij,k)
  1183)                   write(nu_diag,*) 'Tmin =', Tmin
  1184)                   write(nu_diag,*) 'istep1, my_task, i, j:', &
  1185)                                     istep1, my_task, i, j
  1186)                   l_stop = .true.
  1187)                   istop = i
  1188)                   jstop = j
  1189)                   return
  1190)                endif
  1191)             enddo               ! ij
  1192)          endif                  ! tice_low
  1193) 
  1194)       !-----------------------------------------------------------------
  1195)       ! initial energy per unit area of ice/snow, relative to 0 C
  1196)       !-----------------------------------------------------------------
  1197) 
  1198) !DIR$ CONCURRENT !Cray
  1199) !cdir nodep      !NEC
  1200) !ocl novrec      !Fujitsu
  1201)          do ij = 1, icells
  1202) 
  1203)             if (Tin(ij,k) > c0) then ! correct roundoff error
  1204)                Tin(ij,k) = c0
  1205)                qin(ij,k) = -rhoi*Lfresh
  1206)             endif
  1207)             
  1208)             einit(ij) = einit(ij) + hilyr(ij)*qin(ij,k) 
  1209)             
  1210)          enddo                  ! ij
  1211) 
  1212) 
  1213)       enddo                     ! nilyr
  1214) 
  1215)       end subroutine init_vertical_profile

  1220) ! !ROUTINE: temperature_changes  - new vertical temperature profile
  1221) !
  1222) ! !DESCRIPTION:
  1223) !
  1224) ! Compute new surface temperature and internal ice and snow
  1225) ! temperatures.  Include effects of salinity on sea ice heat
  1226) ! capacity in a way that conserves energy (Bitz and Lipscomb, 1999).
  1227) !
  1228) ! New temperatures are computed iteratively by solving a tridiagonal
  1229) ! system of equations; heat capacity is updated with each iteration.
  1230) ! Finite differencing is backward implicit.
  1231) !
  1232) ! See Bitz, C.M., and W.H. Lipscomb, 1999:
  1233) ! An energy-conserving thermodynamic model of sea ice,
  1234) ! J. Geophys. Res., 104, 15,669-15,677.
  1235) !

  1237) !
  1238) ! authors William H. Lipscomb, LANL
  1239) !         C. M. Bitz, UW

  1288)          lhcoef      , & ! transfer coefficient for latent heat

  1303)          intent(inout) :: &

  1307)          intent(inout) :: &
  1308)          Iswabs          ! SW radiation absorbed in ice layers (W m-2)
  1309) 
  1310)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(inout):: &
  1311)          fsurfn      , & ! net flux to top surface, excluding fcondtopn
  1312)          fcondtopn   , & ! downward cond flux at top surface (W m-2)
  1313)          fsensn      , & ! surface downward sensible heat (W m-2)
  1314)          flatn       , & ! surface downward latent heat (W m-2)
  1315)          fswabsn     , & ! shortwave absorbed by ice (W m-2)
  1316)          flwoutn         ! upward LW at surface (W m-2)
  1317) 
  1318)       real (kind=dbl_kind), dimension (icells), intent(out):: &
  1319)          fcondbot        ! downward cond flux at bottom surface (W m-2)
  1320) 
  1321)       real (kind=dbl_kind), dimension (icells), &

  1324) 
  1325)       real (kind=dbl_kind), dimension (icells,nilyr), &
  1326)          intent(inout) :: &
  1327)          qin         , & ! ice layer enthalpy (J m-3)
  1328)          Tin             ! internal ice layer temperatures
  1329) 
  1330)       real (kind=dbl_kind), dimension (icells,nslyr), &

  1335)       logical (kind=log_kind), intent(inout) :: &

  1340) !
  1341) !EOP
  1342) !

  1344)          nitermax = 50, & ! max number of iterations in temperature solver
  1345)          nmat = nslyr + nilyr + 1  ! matrix dimension
  1346) 
  1347)       real (kind=dbl_kind), parameter :: &
  1348)          Tsf_errmax = 5.e-4_dbl_kind ! max allowed error in Tsf
  1349)                                      ! recommend Tsf_errmax < 0.01 K
  1350) 
  1351)       integer (kind=int_kind) :: &
  1352)          i, j        , & ! horizontal indices
  1353)          ij, m       , & ! horizontal indices, combine i and j loops
  1354)          k           , & ! ice layer index
  1355)          niter           ! iteration counter in temperature solver
  1356) 
  1357)       integer (kind=int_kind) :: &
  1358)          isolve          ! number of cells with temps not converged
  1359) 
  1360)       integer (kind=int_kind), dimension (icells) :: &
  1361)          indxii, indxjj  ! compressed indices for cells not converged
  1362) 
  1363)       integer (kind=int_kind), dimension (icells) :: &
  1364)          indxij          ! compressed 1D index for cells not converged

  1373)          dTi1        , & ! Ti1(1) - Tin_start(1)

  1376)          enew            ! new energy of melting after temp change (J m-2)
  1377)

  1379)          dTsf_prev   , & ! dTsf from previous iteration
  1380)          dTi1_prev   , & ! dTi1 from previous iteration

  1382)          dflat_dT    , & ! deriv of flat wrt Tsf (W m-2 deg-1)
  1383)          dflwout_dT  , & ! deriv of flwout wrt Tsf (W m-2 deg-1)
  1384)          dt_rhoi_hlyr    ! dt/(rhoi*hilyr)
  1385) 
  1386)       real (kind=dbl_kind), dimension (icells,nilyr) :: &
  1387)          Tin_init    , & ! Tin at beginning of time step
  1388)          Tin_start       ! Tin at start of iteration
  1389) 
  1390)       real (kind=dbl_kind), dimension (icells,nslyr) :: &
  1391)          Tsn_init    , & ! Tsn at beginning of time step
  1392)          Tsn_start   , & ! Tsn at start of iteration
  1393)          etas            ! dt / (rho * cp * h) for snow layers
  1394) 
  1395)       real (kind=dbl_kind), dimension (:,:), allocatable :: &
  1396)          etai        , & ! dt / (rho * cp * h) for ice layers
  1397)          sbdiag      , & ! sub-diagonal matrix elements
  1398)          diag        , & ! diagonal matrix elements
  1399)          spdiag      , & ! super-diagonal matrix elements
  1400)          rhs         , & ! rhs of tri-diagonal matrix equation
  1401)          Tmat            ! matrix output temperatures
  1402) 
  1403)       real (kind=dbl_kind), dimension(icells,nilyr+nslyr+1):: &
  1404)          kh              ! effective conductivity at interfaces (W m-2 deg-1)
  1405) 
  1406)       real (kind=dbl_kind) :: &
  1407)          ci          , & ! specific heat of sea ice (J kg-1 deg-1)
  1408)          avg_Tsf     , & ! = 1. if Tsf averaged w/Tsf_start, else = 0.
  1409)          ferr        , & ! energy conservation error (W m-2)
  1410)          Iswabs_tmp  , &
  1411)          Sswabs_tmp  , &
  1412)          etai_tmp
  1413)

  1417)       logical (kind=log_kind) :: &

  1419) 
  1420)       !-----------------------------------------------------------------

  1422)       !-----------------------------------------------------------------
  1423)

  1427) 
  1428)          converged (ij) = .false.
  1429)          l_snow    (ij) = .false.
  1430)          l_cold    (ij) = .true.

  1432)          dTsf_prev (ij) = c0
  1433)          dTi1_prev (ij) = c0
  1434)          dfsens_dT (ij) = c0
  1435)          dflat_dT  (ij) = c0
  1436)          dflwout_dT(ij) = c0

  1438)          if (hslyr(ij) > hs_min/real(nslyr,kind=dbl_kind)) &
  1439)             l_snow(ij) = .true.
  1440)       enddo                     ! ij
  1441) 
  1442)       do k = 1, nslyr
  1443)          do ij = 1, icells
  1444)             Tsn_init (ij,k) = Tsn(ij,k) ! beginning of time step
  1445)             Tsn_start(ij,k) = Tsn(ij,k) ! beginning of iteration
  1446)             if (l_snow(ij)) then
  1447)                etas(ij,k) = dt/(rhos*cp_ice*hslyr(ij))
  1448)             else
  1449)                etas(ij,k) = c0
  1450)             endif
  1451)          enddo                  ! ij
  1452)       enddo                     ! k
  1453) 
  1454)       do k = 1, nilyr
  1455)          do ij = 1, icells
  1456)             Tin_init (ij,k) = Tin(ij,k)   ! beginning of time step
  1457)             Tin_start(ij,k) = Tin(ij,k)   ! beginning of iteration
  1458)          enddo
  1459)       enddo
  1460)

  1465)       !  simplify the logic.

  1469)                          l_snow,   icells,           &
  1470)                          indxi,    indxj,    indxij, &
  1471)                          hilyr,    hslyr,            &
  1472)                          Tin,      kh )
  1473) 
  1474)       !-----------------------------------------------------------------

  1476)       ! temperature overshoots.  Switch that radiation to fswsfc if necessary. 
  1477)       ! NOTE: This option is not available if the atmosphere model
  1478)       !       has already computed fsurf.  (Unless we adjust fsurf here)
  1479)       !-----------------------------------------------------------------
  1480) !mclaren: Should there be an if calc_Tsfc statement here then?? 
  1481) 
  1482)       do k = 1, nilyr
  1483)          do ij = 1, icells
  1484)             i = indxi(ij)
  1485)             j = indxj(ij)
  1486) 
  1487)             if (Tin_init(ij,k) <= (Tmlt(k) - p01)) then
  1488) 
  1489)             if (l_brine) then
  1490)                ci = cp_ice - Lfresh*Tmlt(k) /  &
  1491)                   max( Tin_init(ij,k)**2, Tmlt(k)**2 )
  1492)                etai_tmp = dt_rhoi_hlyr(ij) / ci
  1493)                Iswabs_tmp = min(Iswabs(i,j,k), &
  1494)                                 (Tmlt(k)-Tin_init(ij,k))/etai_tmp)
  1495)             else
  1496)                ci = cp_ice
  1497)                etai_tmp = dt_rhoi_hlyr(ij) / ci
  1498)                Iswabs_tmp = min(Iswabs(i,j,k), &
  1499)                                 Tin_init(ij,k)/etai_tmp)
  1500)             endif
  1501) 
  1502)             fswsfc(i,j)   = fswsfc(i,j) + (Iswabs(i,j,k) - Iswabs_tmp)
  1503)             fswint(i,j)   = fswint(i,j) - (Iswabs(i,j,k) - Iswabs_tmp)
  1504)             Iswabs(i,j,k) = Iswabs_tmp
  1505) 
  1506)             else
  1507) 
  1508)             fswsfc(i,j)   = fswsfc(i,j) + Iswabs(i,j,k)
  1509)             fswint(i,j)   = fswint(i,j) - Iswabs(i,j,k)
  1510)             Iswabs(i,j,k) = c0
  1511) 
  1512)             endif
  1513)          enddo
  1514)       enddo
  1515) 
  1516)       do k = 1, nslyr
  1517)          do ij = 1, icells
  1518)             if (l_snow(ij)) then
  1519)                i = indxi(ij)
  1520)                j = indxj(ij)
  1521) 
  1522)                if (Tsn_init(ij,k) <= -p01) then
  1523) 
  1524)                Sswabs_tmp = min(Sswabs(i,j,k), &
  1525)                             -0.9_dbl_kind*Tsn_init(ij,k)/etas(ij,k))
  1526) 
  1527)                fswsfc(i,j)   = fswsfc(i,j) + (Sswabs(i,j,k) - Sswabs_tmp)
  1528)                fswint(i,j)   = fswint(i,j) - (Sswabs(i,j,k) - Sswabs_tmp)
  1529)                Sswabs(i,j,k) = Sswabs_tmp
  1530) 
  1531)                else
  1532) 
  1533)                fswsfc(i,j)   = fswsfc(i,j) + Sswabs(i,j,k)
  1534)                fswint(i,j)   = fswint(i,j) - Sswabs(i,j,k)
  1535)                Sswabs(i,j,k) = c0
  1536) 
  1537)                endif
  1538)             endif
  1539)          enddo
  1540)       enddo
  1541) 
  1542) !lipscomb - This could be done in the shortwave module instead.
  1543) !           (Change zerolayer routine also)
  1544)       ! absorbed shortwave flux for coupler
  1545) 
  1546)       do ij = 1, icells
  1547)          i = indxi(ij)
  1548)          j = indxj(ij)
  1549)          fswabsn(i,j) = fswsfc(i,j) + fswint(i,j) + fswthrun(i,j)
  1550)       enddo
  1551) 
  1552)       !-----------------------------------------------------------------
  1553)       ! Solve for new temperatures.
  1554)       ! Iterate until temperatures converge with minimal energy error.
  1555)       !-----------------------------------------------------------------
  1556) 
  1557)       do niter = 1, nitermax
  1558) 
  1559)       !-----------------------------------------------------------------
  1560)       ! Identify cells, if any, where calculation has not converged.
  1561)       !-----------------------------------------------------------------
  1562) 
  1563)          if (all_converged) then  ! thermo calculation is done
  1564)             exit
  1565)          else                     ! identify cells not yet converged
  1566)             isolve = 0
  1567)             do ij = 1, icells
  1568)                i = indxi(ij)
  1569)                j = indxj(ij)
  1570)                if (.not.converged(ij)) then
  1571)                   isolve = isolve + 1
  1572)                   indxii(isolve) = i
  1573)                   indxjj(isolve) = j
  1574)                   indxij(isolve) = ij
  1575)                endif
  1576)             enddo               ! ij
  1577)          endif
  1578) 
  1579)       !-----------------------------------------------------------------
  1580)       ! Allocate and initialize
  1581)       !-----------------------------------------------------------------
  1582) 
  1583)          allocate(   sbdiag(isolve,nilyr+nslyr+1))
  1584)          allocate(     diag(isolve,nilyr+nslyr+1))
  1585)          allocate(   spdiag(isolve,nilyr+nslyr+1))
  1586)          allocate(      rhs(isolve,nilyr+nslyr+1))
  1587)          allocate(     Tmat(isolve,nilyr+nslyr+1))
  1588)          allocate(     etai(isolve,nilyr))
  1589)          allocate(Tsf_start(isolve))
  1590)          allocate(     dTsf(isolve))
  1591)          allocate(dfsurf_dT(isolve))
  1592)          allocate(  avg_Tsi(isolve))
  1593)          allocate(     enew(isolve))
  1594)          allocate(     dTi1(isolve))
  1595) 
  1596)          all_converged = .true.
  1597) 
  1598)          do ij = 1, isolve
  1599)             m = indxij(ij)
  1600)             converged(m)  = .true.
  1601)             dfsurf_dT(ij) = c0
  1602)             avg_Tsi  (ij) = c0
  1603)             enew     (ij) = c0
  1604)          enddo
  1605) 
  1606)       !-----------------------------------------------------------------
  1607)       ! Update specific heat of ice layers.
  1608)       ! To ensure energy conservation, the specific heat is a function of
  1609)       ! both the starting temperature and the (latest guess for) the
  1610)       ! final temperature.
  1611)       !-----------------------------------------------------------------
  1612) 
  1613)          do k = 1, nilyr
  1614)             do ij = 1, isolve
  1615)                m = indxij(ij)
  1616)                i = indxii(ij)
  1617)                j = indxjj(ij)
  1618) 
  1619)                if (l_brine) then
  1620)                   ci = cp_ice - Lfresh*Tmlt(k) /  &
  1621)                                 (Tin(m,k)*Tin_init(m,k))
  1622)                else
  1623)                   ci = cp_ice
  1624)                endif
  1625)                etai(ij,k) = dt_rhoi_hlyr(m) / ci
  1626) 
  1627)             enddo
  1628)          enddo
  1629) 
  1630)          if (calc_Tsfc) then
  1631) 
  1632)       !-----------------------------------------------------------------
  1633)       ! Update radiative and turbulent fluxes and their derivatives
  1634)       ! with respect to Tsf.
  1635)       !-----------------------------------------------------------------
  1636) 
  1637)           call surface_fluxes(nx_block,    ny_block,          &
  1638)                               isolve,      icells,            &
  1639)                               indxii,      indxjj,    indxij, &
  1640)                               Tsf,         fswsfc,            &
  1641)                               rhoa,        flw,               &
  1642)                               potT,        Qa,                &
  1643)                               shcoef,      lhcoef,            &
  1644)                               flwoutn,     fsensn,            &
  1645)                               flatn,       fsurfn,            &
  1646)                               dflwout_dT,  dfsens_dT,         &
  1647)                               dflat_dT,    dfsurf_dT)
  1648) 
  1649) !DIR$ CONCURRENT !Cray
  1650) !cdir nodep      !NEC
  1651) !ocl novrec      !Fujitsu
  1652)           do ij = 1, isolve
  1653)             i = indxii(ij)
  1654)             j = indxjj(ij)
  1655)             m = indxij(ij)
  1656) 
  1657)       !-----------------------------------------------------------------
  1658)       ! Compute conductive flux at top surface, fcondtopn.
  1659)       ! If fsurfn < fcondtopn and Tsf = 0, then reset Tsf to slightly less
  1660)       !  than zero (but not less than -puny).
  1661)       !-----------------------------------------------------------------
  1662) 
  1663)             if (l_snow(m)) then
  1664)                fcondtopn(i,j) = kh(m,1) * (Tsf(m) - Tsn(m,1))
  1665)             else
  1666)                fcondtopn(i,j) = kh(m,1+nslyr) * (Tsf(m) - Tin(m,1))
  1667)             endif
  1668) 
  1669)             if (fsurfn(i,j) < fcondtopn(i,j)) &
  1670)                  Tsf(m) = min (Tsf(m), -puny)
  1671) 
  1672)       !-----------------------------------------------------------------
  1673)       ! Save surface temperature at start of iteration
  1674)       !-----------------------------------------------------------------
  1675) 
  1676)             Tsf_start(ij) = Tsf(m)
  1677) 
  1678)             if (Tsf(m) <= -puny) then
  1679)                l_cold(m) = .true.
  1680)             else
  1681)                l_cold(m) = .false.
  1682)             endif
  1683)           enddo                  ! ij
  1684) 
  1685)       !-----------------------------------------------------------------
  1686)       ! Compute elements of tridiagonal matrix.
  1687)       !-----------------------------------------------------------------
  1688) 
  1689)             call get_matrix_elements_calc_Tsfc &
  1690)                                   (nx_block, ny_block,         &
  1691)                                    isolve,   icells,           &
  1692)                                    indxii,   indxjj,   indxij, &
  1693)                                    l_snow,   l_cold,           &
  1694)                                    Tsf,      Tbot,             &
  1695)                                    fsurfn,   dfsurf_dT,        &
  1696)                                    Tin_init, Tsn_init,         &
  1697)                                    kh,       Sswabs(:,:,:),    &
  1698)                                    Iswabs,                     &
  1699)                                    etai,     etas,             &
  1700)                                    sbdiag,   diag,             &
  1701)                                    spdiag,   rhs)
  1702) 
  1703)          else
  1704)             call get_matrix_elements_know_Tsfc &
  1705)                                   (nx_block, ny_block,         &
  1706)                                    isolve,   icells,           &
  1707)                                    indxii,   indxjj,   indxij, &
  1708)                                    l_snow,   Tbot,             &
  1709)                                    Tin_init, Tsn_init,         &
  1710)                                    kh,       Sswabs(:,:,:),    &
  1711)                                    Iswabs,                     &
  1712)                                    etai,     etas,             &
  1713)                                    sbdiag,   diag,             &
  1714)                                    spdiag,   rhs,              &
  1715)                                    fcondtopn)
  1716)          endif  ! calc_Tsfc
  1717) 
  1718)       !-----------------------------------------------------------------
  1719)       ! Solve tridiagonal matrix to obtain the new temperatures.
  1720)       !-----------------------------------------------------------------
  1721) 
  1722)          call tridiag_solver (nx_block, ny_block, &
  1723)                               isolve,   icells,   &
  1724)                               indxii,   indxjj,   &
  1725)                               nmat,     sbdiag,   &
  1726)                               diag,     spdiag,   &
  1727)                               rhs,      Tmat)
  1728) 
  1729)       !-----------------------------------------------------------------
  1730)       ! Determine whether the computation has converged to an acceptable
  1731)       ! solution.  Five conditions must be satisfied:
  1732)       !
  1733)       !    (1) Tsf <= 0 C.
  1734)       !    (2) Tsf is not oscillating; i.e., if both dTsf(niter) and
  1735)       !        dTsf(niter-1) have magnitudes greater than puny, then
  1736)       !        dTsf(niter)/dTsf(niter-1) cannot be a negative number
  1737)       !        with magnitude greater than 0.5.  
  1738)       !    (3) abs(dTsf) < Tsf_errmax
  1739)       !    (4) If Tsf = 0 C, then the downward turbulent/radiative 
  1740)       !        flux, fsurfn, must be greater than or equal to the downward
  1741)       !        conductive flux, fcondtopn.
  1742)       !    (5) The net energy added to the ice per unit time must equal 
  1743)       !        the net change in internal ice energy per unit time,
  1744)       !        within the prescribed error ferrmax.
  1745)       !
  1746)       ! For briny ice (the standard case), Tsn and Tin are limited
  1747)       !  to prevent them from exceeding their melting temperatures.
  1748)       !  (Note that the specific heat formula for briny ice assumes
  1749)       !  that T < Tmlt.)  
  1750)       ! For fresh ice there is no limiting, since there are cases
  1751)       !  when the only convergent solution has Tsn > 0 and/or Tin > 0.
  1752)       !  Above-zero temperatures are then reset to zero (with melting 
  1753)       !  to conserve energy) in the thickness_changes subroutine.
  1754)       !-----------------------------------------------------------------
  1755) 
  1756)          if (calc_Tsfc) then
  1757) 
  1758) !DIR$ CONCURRENT !Cray
  1759) !cdir nodep      !NEC
  1760) !ocl novrec      !Fujitsu
  1761)           do ij = 1, isolve
  1762)             m = indxij(ij)
  1763) 
  1764)       !-----------------------------------------------------------------
  1765)       ! Reload Tsf from matrix solution
  1766)       !-----------------------------------------------------------------
  1767) 
  1768)             if (l_cold(m)) then
  1769)                if (l_snow(m)) then
  1770)                   Tsf(m) = Tmat(ij,1)
  1771)                else
  1772)                   Tsf(m) = Tmat(ij,1+nslyr)
  1773)                endif
  1774)             else                ! melting surface
  1775)                Tsf(m) = c0
  1776)             endif
  1777) 
  1778)       !-----------------------------------------------------------------
  1779)       ! Initialize convergence flag (true until proven false), dTsf,
  1780)       !  and temperature-averaging coefficients.
  1781)       ! Average only if test 1 or 2 fails.
  1782)       ! Initialize energy.
  1783)       !-----------------------------------------------------------------
  1784) 
  1785)             dTsf(ij) = Tsf(m) - Tsf_start(ij)
  1786)             avg_Tsf  = c0
  1787) 
  1788)       !-----------------------------------------------------------------
  1789)       ! Condition 1: check for Tsf > 0
  1790)       ! If Tsf > 0, set Tsf = 0, then average Tsn and Tin to force
  1791)       ! internal temps below their melting temps.
  1792)       !-----------------------------------------------------------------
  1793) 
  1794)             if (Tsf(m) > puny) then
  1795)                Tsf(m) = c0
  1796)                dTsf(ij) = -Tsf_start(ij)
  1797)                if (l_brine) avg_Tsi(ij) = c1   ! avg with starting temp
  1798)                converged(m) = .false.
  1799)                all_converged = .false.
  1800) 
  1801)       !-----------------------------------------------------------------
  1802)       ! Condition 2: check for oscillating Tsf
  1803)       ! If oscillating, average all temps to increase rate of convergence.
  1804)       !-----------------------------------------------------------------
  1805) 
  1806)             elseif (niter > 1 &                ! condition (2)
  1807)               .and. Tsf_start(ij) <= -puny &
  1808)               .and. abs(dTsf(ij)) > puny &
  1809)               .and. abs(dTsf_prev(m)) > puny &
  1810)               .and. -dTsf(ij)/(dTsf_prev(m)+puny*puny) > p5) then
  1811) 
  1812)                if (l_brine) then ! average with starting temp
  1813)                   avg_Tsf  = c1    
  1814)                   avg_Tsi(ij) = c1
  1815)                endif
  1816)                dTsf(ij) = p5 * dTsf(ij)
  1817)                converged(m) = .false.
  1818)                all_converged = .false.
  1819)             endif
  1820) 
  1821) !!!            dTsf_prev(m) = dTsf(ij)
  1822) 
  1823)       !-----------------------------------------------------------------
  1824)       ! If condition 2 failed, average new surface temperature with
  1825)       !  starting value.
  1826)       !-----------------------------------------------------------------
  1827)             Tsf(m)  = Tsf(m) &
  1828)                       + avg_Tsf * p5 * (Tsf_start(ij) - Tsf(m))
  1829) 
  1830)           enddo  ! ij
  1831) 
  1832)          endif   ! calc_Tsfc
  1833) 
  1834)          do k = 1, nslyr
  1835) !DIR$ CONCURRENT !Cray
  1836) !cdir nodep      !NEC
  1837) !ocl novrec      !Fujitsu
  1838)             do ij = 1, isolve
  1839)                m = indxij(ij)
  1840) 
  1841)       !-----------------------------------------------------------------
  1842)       ! Reload Tsn from matrix solution
  1843)       !-----------------------------------------------------------------
  1844) 
  1845)                if (l_snow(m)) then
  1846)                   Tsn(m,k) = Tmat(ij,k+1)
  1847)                else
  1848)                   Tsn(m,k) = c0
  1849)                endif
  1850)                if (l_brine) Tsn(m,k) = min(Tsn(m,k), c0)
  1851) 
  1852)       !-----------------------------------------------------------------
  1853)       ! If condition 1 or 2 failed, average new snow layer
  1854)       !  temperatures with their starting values.
  1855)       !-----------------------------------------------------------------
  1856)                Tsn(m,k) = Tsn(m,k) &
  1857)                          + avg_Tsi(ij)*p5*(Tsn_start(m,k)-Tsn(m,k))
  1858) 
  1859)       !-----------------------------------------------------------------
  1860)       ! Compute qsn and increment new energy.
  1861)       !-----------------------------------------------------------------
  1862)                qsn(m,k) = -rhos * (Lfresh - cp_ice*Tsn(m,k))
  1863)                enew(ij) = enew(ij) + hslyr(m) * qsn(m,k)
  1864) 
  1865)                Tsn_start(m,k) = Tsn(m,k) ! for next iteration
  1866) 
  1867)             enddo               ! ij
  1868)          enddo                  ! nslyr
  1869) 
  1870)          do k = 1, nilyr
  1871) !DIR$ CONCURRENT !Cray
  1872) !cdir nodep      !NEC
  1873) !ocl novrec      !Fujitsu
  1874)             do ij = 1, isolve
  1875)                m = indxij(ij)
  1876) 
  1877)       !-----------------------------------------------------------------
  1878)       ! Reload Tin from matrix solution
  1879)       !-----------------------------------------------------------------
  1880)                Tin(m,k) = Tmat(ij,k+1+nslyr)
  1881)                if (l_brine) Tin(m,k) = min(Tin(m,k), Tmlt(k))
  1882) 
  1883)       !-----------------------------------------------------------------
  1884)       ! Condition 2b: check for oscillating Tin(1)
  1885)       ! If oscillating, average all ice temps to increase rate of convergence.
  1886)       !-----------------------------------------------------------------
  1887) 
  1888)                if (k==1 .and. .not.calc_Tsfc) then
  1889)                   dTi1(ij) = Tin(m,k) - Tin_start(m,k)
  1890) 
  1891)                   if (niter > 1 &                    ! condition 2b    
  1892)                       .and. abs(dTi1(ij)) > puny &
  1893)                       .and. abs(dTi1_prev(m)) > puny &
  1894)                       .and. -dTi1(ij)/(dTi1_prev(m)+puny*puny) > p5) then
  1895) 
  1896)                      if (l_brine) avg_Tsi(ij) = c1
  1897)                      dTi1(ij) = p5 * dTi1(ij)
  1898)                      converged(m) = .false.
  1899)                      all_converged = .false.
  1900)                   endif
  1901)                   dTi1_prev(m) = dTi1(ij)
  1902)                endif   ! k = 1 .and. calc_Tsfc = F
  1903) 
  1904)       !-----------------------------------------------------------------
  1905)       ! If condition 1 or 2 failed, average new ice layer
  1906)       !  temperatures with their starting values.
  1907)       !-----------------------------------------------------------------
  1908)                Tin(m,k) = Tin(m,k) &
  1909)                          + avg_Tsi(ij)*p5*(Tin_start(m,k)-Tin(m,k))
  1910) 
  1911)       !-----------------------------------------------------------------
  1912)       ! Compute qin and increment new energy.
  1913)       !-----------------------------------------------------------------
  1914)                qin(m,k) = -rhoi * (cp_ice*(Tmlt(k)-Tin(m,k)) &
  1915)                                    + Lfresh*(c1-Tmlt(k)/Tin(m,k)) &
  1916)                                    - cp_ocn*Tmlt(k))
  1917)                enew(ij) = enew(ij) + hilyr(m) * qin(m,k)
  1918) 
  1919)                Tin_start(m,k) = Tin(m,k) ! for next iteration
  1920) 
  1921)             enddo               ! ij
  1922)          enddo                  ! nilyr
  1923) 
  1924)          if (calc_Tsfc) then
  1925) 
  1926) !DIR$ CONCURRENT !Cray
  1927) !cdir nodep      !NEC
  1928) !ocl novrec      !Fujitsu
  1929)           do ij = 1, isolve
  1930)             i = indxii(ij)
  1931)             j = indxjj(ij)
  1932)             m = indxij(ij)
  1933) 
  1934)       !-----------------------------------------------------------------
  1935)       ! Condition 3: check for large change in Tsf
  1936)       !-----------------------------------------------------------------
  1937) 
  1938)             if (abs(dTsf(ij)) > Tsf_errmax) then
  1939)                converged(m) = .false.
  1940)                all_converged = .false.
  1941)             endif
  1942) 
  1943)       !-----------------------------------------------------------------
  1944)       ! Condition 4: check for fsurfn < fcondtopn with Tsf > 0
  1945)       !-----------------------------------------------------------------
  1946) 
  1947)             fsurfn(i,j) = fsurfn(i,j) + dTsf(ij)*dfsurf_dT(ij)
  1948)             if (l_snow(m)) then
  1949)                fcondtopn(i,j) = kh(m,1) * (Tsf(m)-Tsn(m,1))
  1950)             else
  1951)                fcondtopn(i,j) = kh(m,1+nslyr) * (Tsf(m)-Tin(m,1))
  1952)             endif
  1953) 
  1954)             if (Tsf(m) > -puny .and. fsurfn(i,j) < fcondtopn(i,j)) then
  1955)                converged(m) = .false.
  1956)                all_converged = .false.
  1957)             endif
  1958) 
  1959)             dTsf_prev(m) = dTsf(ij)
  1960) 
  1961)           enddo                  ! ij
  1962)          endif                   ! calc_Tsfc
  1963) 
  1964)       !-----------------------------------------------------------------
  1965)       ! Condition 5: check for energy conservation error
  1966)       ! Change in internal ice energy should equal net energy input.
  1967)       !-----------------------------------------------------------------
  1968) 
  1969)          do ij = 1, isolve
  1970)             i = indxii(ij)
  1971)             j = indxjj(ij)
  1972)             m = indxij(ij)
  1973) 
  1974)             fcondbot(m) = kh(m,1+nslyr+nilyr) * &
  1975)                            (Tin(m,nilyr)   - Tbot(i,j))
  1976) 
  1977)             ferr = abs( (enew(ij)-einit(m))/dt &
  1978)                  - (fcondtopn(i,j) - fcondbot(m) + fswint(i,j)) )
  1979) 
  1980)             ! factor of 0.9 allows for roundoff errors later
  1981)             if (ferr > 0.9_dbl_kind*ferrmax) then         ! condition (5)
  1982)                converged(m) = .false.
  1983)                all_converged = .false.
  1984)             endif
  1985) 
  1986)          enddo                  ! ij
  1987) 
  1988)          deallocate(sbdiag)
  1989)          deallocate(diag)
  1990)          deallocate(spdiag)
  1991)          deallocate(rhs)
  1992)          deallocate(Tmat)
  1993)          deallocate(etai)
  1994)          deallocate(Tsf_start)
  1995)          deallocate(dTsf)
  1996)          deallocate(dfsurf_dT)
  1997)          deallocate(avg_Tsi)
  1998)          deallocate(enew)
  1999)          deallocate(dTi1)
  2000) 
  2001)       enddo                     ! temperature iteration niter
  2002) 
  2003)       if (.not.all_converged) then
  2004) 
  2005)          do ij = 1, icells
  2006)             i = indxi(ij)
  2007)             j = indxj(ij)
  2008) 
  2009)       !-----------------------------------------------------------------
  2010)       ! Check for convergence failures.
  2011)       !-----------------------------------------------------------------
  2012)             if (.not.converged(ij)) then
  2013)                write(nu_diag,*) 'Thermo iteration does not converge,', &
  2014)                                 'istep1, my_task, i, j:', &
  2015)                                  istep1, my_task, i, j
  2016)                write(nu_diag,*) 'Ice thickness:',  hilyr(ij)*nilyr
  2017)                write(nu_diag,*) 'Snow thickness:', hslyr(ij)*nslyr
  2018)                write(nu_diag,*) 'dTsf, Tsf_errmax:',dTsf_prev(ij), &
  2019)                                  Tsf_errmax
  2020)                write(nu_diag,*) 'Tsf:', Tsf(ij)
  2021)                write(nu_diag,*) 'fsurf:', fsurfn(i,j)
  2022)                write(nu_diag,*) 'fcondtop, fcondbot, fswint', &
  2023)                                  fcondtopn(i,j), fcondbot(ij), fswint(i,j)
  2024)                write(nu_diag,*) 'fswsfc, fswthrun', &
  2025)                                  fswsfc(i,j), fswthrun(i,j)
  2026)                write(nu_diag,*) 'Flux conservation error =', ferr
  2027)                write(nu_diag,*) 'Internal snow absorption:'
  2028)                write(nu_diag,*) (Sswabs(i,j,k),k=1,nslyr)
  2029)                write(nu_diag,*) 'Internal ice absorption:'
  2030)                write(nu_diag,*) (Iswabs(i,j,k),k=1,nilyr)
  2031)                write(nu_diag,*) 'Initial snow temperatures:'
  2032)                write(nu_diag,*) (Tsn_init(ij,k),k=1,nslyr)
  2033)                write(nu_diag,*) 'Initial ice temperatures:'
  2034)                write(nu_diag,*) (Tin_init(ij,k),k=1,nilyr)
  2035)                write(nu_diag,*) 'Final snow temperatures:'
  2036)                write(nu_diag,*) (Tsn(ij,k),k=1,nslyr)
  2037)                write(nu_diag,*) 'Final ice temperatures:'
  2038)                write(nu_diag,*) (Tin(ij,k),k=1,nilyr)
  2039)                l_stop = .true.
  2040)                istop = i
  2041)                jstop = j
  2042)                return
  2043)             endif
  2044)          enddo                  ! ij
  2045)       endif                     ! all_converged
  2046) 
  2047)       if (calc_Tsfc) then
  2048) !DIR$ CONCURRENT !Cray
  2049) !cdir nodep      !NEC
  2050) !ocl novrec      !Fujitsu
  2051)          do ij = 1, icells
  2052)             i = indxi(ij)
  2053)             j = indxj(ij)
  2054) 
  2055)             ! update fluxes that depend on Tsf
  2056)             flwoutn(i,j) = flwoutn(i,j) + dTsf_prev(ij) * dflwout_dT(ij)
  2057)             fsensn(i,j)  = fsensn(i,j)  + dTsf_prev(ij) * dfsens_dT(ij)
  2058)             flatn(i,j)   = flatn(i,j)   + dTsf_prev(ij) * dflat_dT(ij)
  2059) 
  2060)          enddo                     ! ij
  2061)       endif                        ! calc_Tsfc
  2062) 
  2063)       end subroutine temperature_changes
  2064)

  2071) !
  2072) ! Compute thermal conductivity at interfaces (held fixed during
  2073) !  the subsequent iteration).
  2074) !
  2075) ! NOTE: Ice conductivity must be >= kimin
  2076) !

  2078) !

  2080) !         C. M. Bitz, UW
  2081) !

  2083) !
  2084)       subroutine conductivity (nx_block, ny_block,         &
  2085)                                l_snow,   icells,           &
  2086)                                indxi,    indxj,    indxij, &
  2087)                                hilyr,    hslyr,            &
  2088)                                Tin,      kh)
  2089) !
  2090) ! !USES:
  2091) !
  2092) ! !INPUT/OUTPUT PARAMETERS:
  2093) !
  2094)       integer (kind=int_kind), intent(in) :: &
  2095)          nx_block, ny_block, & ! block dimensions
  2096)          icells          ! number of cells with aicen > puny
  2097) 
  2098)       logical (kind=log_kind), dimension(icells), &
  2099)          intent(in) :: &
  2100)          l_snow          ! true if snow temperatures are computed
  2101) 
  2102)       integer (kind=int_kind), dimension(nx_block*ny_block), &
  2103)          intent(in) :: &
  2104)          indxi, indxj    ! compressed indices for cells with aicen > puny
  2105) 
  2106)       integer (kind=int_kind), dimension (icells), &
  2107)          intent(in) :: &
  2108)          indxij          ! compressed 1D index for cells not converged
  2109) 
  2110)       real (kind=dbl_kind), dimension (icells), intent(in) :: &
  2111)          hilyr       , & ! ice layer thickness (same for all ice layers)
  2112)          hslyr           ! snow layer thickness (same for all snow layers)
  2113) 
  2114)       real (kind=dbl_kind), dimension (icells,nilyr), &
  2115)          intent(in) :: &

  2969)                                  isolve,   icells,   &
  2970)                                  indxii,   indxjj,   &
  2971)                                  nmat,     sbdiag,   &
  2972)                                  diag,     spdiag,   &
  2973)                                  rhs,      xout)
  2974) !
  2975) ! !USES:
  2976) !
  2977) ! !INPUT/OUTPUT PARAMETERS:
  2978) !
  2979)       integer (kind=int_kind), intent(in) :: &

  3003) !
  3004)       integer (kind=int_kind) :: &

  3017) !ocl novrec      !Fujitsu

  3166)          dfsurf_dT       ! derivative of fsurfn wrt Tsf

  3182)          kratio      , & ! ratio of ice and snow conductivies

  3199) 
  3200)          converged (ij) = .false.
  3201) 
  3202)          dTsf_prev (ij) = c0
  3203) 
  3204)       enddo                     ! ij
  3205)

  3207)       ! Solve for new temperatures.
  3208)       ! Iterate until temperatures converge with minimal temperature
  3209)       ! change.
  3210)       !-----------------------------------------------------------------
  3211) 
  3212)       do niter = 1, nitermax
  3213) 
  3214)          if (all_converged) then  ! thermo calculation is done
  3215)             exit
  3216)          else                     ! identify cells not yet converged
  3217)             isolve = 0
  3218)             do ij = 1, icells
  3219)                i = indxi(ij)
  3220)                j = indxj(ij)
  3221)                if (.not.converged(ij)) then
  3222)                   isolve = isolve + 1
  3223)                   indxii(isolve) = i
  3224)                   indxjj(isolve) = j
  3225)                   indxij(isolve) = ij
  3226)                endif
  3227)             enddo               ! ij
  3228)          endif
  3229)

  3231)          allocate(      rhs(isolve))
  3232)          allocate(       kh(isolve))
  3233)          allocate(     heff(isolve))
  3234)          allocate(Tsf_start(isolve))
  3235)          allocate(     dTsf(isolve))
  3236)          allocate(dfsurf_dT(isolve))

  3239)       ! Update radiative and turbulent fluxes and their derivatives
  3240)       ! with respect to Tsf.
  3241)       !-----------------------------------------------------------------
  3242) 
  3243)          call surface_fluxes (nx_block,    ny_block,          &
  3244)                               isolve,      icells,            &
  3245)                               indxii,      indxjj,    indxij, &
  3246)                               Tsf,         fswsfc,            &
  3247)                               rhoa,        flw,               &

  3249)                               shcoef,      lhcoef,            &

  3251)                               flatn,       fsurfn,            &

  3253)                               dflat_dT,    dfsurf_dT)

  3256)       ! Compute effective ice thickness (includes snow) and thermal 
  3257)       ! conductivity 
  3258)       !-----------------------------------------------------------------
  3259) 
  3260)          kratio = kseaice/ksno
  3261)  
  3262)          do ij = 1, isolve
  3263)              m = indxij(ij)
  3264)    
  3265)              heff(ij) = hilyr(m) + kratio * hslyr(m)
  3266)              kh(ij) = kseaice / heff(ij)        
  3267)          enddo                     ! ij
  3268)

  3271) !cdir nodep      !NEC
  3272) !ocl novrec      !Fujitsu
  3273)          do ij = 1, isolve
  3274)             i = indxii(ij)
  3275)             j = indxjj(ij)
  3276)             m = indxij(ij)
  3277) 
  3278)       !-----------------------------------------------------------------

  3280)       ! If fsurfn < fcondtopn and Tsf = 0, then reset Tsf to slightly less
  3281)       !  than zero (but not less than -puny).
  3282)       !-----------------------------------------------------------------
  3283) 
  3284)             fcondtopn(i,j) = kh(ij) * (Tsf(m) - Tbot(i,j))
  3285) 
  3286)             if (fsurfn(i,j) < fcondtopn(i,j)) &
  3287)                  Tsf(m) = min (Tsf(m), -puny)
  3288) 
  3289)       !-----------------------------------------------------------------
  3290)       ! Save surface temperature at start of iteration
  3291)       !-----------------------------------------------------------------
  3292) 
  3293)             Tsf_start(ij) = Tsf(m)
  3294) 
  3295)          enddo                  ! ij
  3296)

  3299)       !-----------------------------------------------------------------

  3303)             j = indxjj(ij)
  3304)             m = indxij(ij)
  3305) 
  3306)             diag(ij)  = dfsurf_dT(ij) - kh(ij)
  3307)             rhs(ij)   = dfsurf_dT(ij)*Tsf(m) - fsurfn(i,j)   &
  3308)                         - kh(ij)*Tbot(i,j)
  3309)             Tsf(m)  = rhs(ij) / diag(ij)
  3310) 
  3311)          enddo
  3312) 
  3313)       !-----------------------------------------------------------------
  3314)       ! Determine whether the computation has converged to an acceptable
  3315)       ! solution.  Four conditions must be satisfied:
  3316)       !
  3317)       !    (1) Tsf <= 0 C.
  3318)       !    (2) Tsf is not oscillating; i.e., if both dTsf(niter) and
  3319)       !        dTsf(niter-1) have magnitudes greater than puny, then
  3320)       !        dTsf(niter)/dTsf(niter-1) cannot be a negative number
  3321)       !        with magnitude greater than 0.5.  
  3322)       !    (3) abs(dTsf) < Tsf_errmax
  3323)       !    (4) If Tsf = 0 C, then the downward turbulent/radiative 
  3324)       !        flux, fsurfn, must be greater than or equal to the downward
  3325)       !        conductive flux, fcondtopn.
  3326)       !-----------------------------------------------------------------
  3327) 
  3328)          ! initialize global convergence flag
  3329)          all_converged = .true.
  3330) 
  3331) !DIR$ CONCURRENT !Cray
  3332) !cdir nodep      !NEC
  3333) !ocl novrec      !Fujitsu
  3334)          do ij = 1, isolve
  3335)             m = indxij(ij)
  3336) 
  3337)       !-----------------------------------------------------------------
  3338)       ! Initialize convergence flag (true until proven false), dTsf,
  3339)       !  and temperature-averaging coefficients.
  3340)       ! Average only if test 1 or 2 fails.
  3341)       ! Initialize energy.
  3342)       !-----------------------------------------------------------------
  3343) 
  3344)             converged(m) = .true.
  3345)             dTsf(ij) = Tsf(m) - Tsf_start(ij)
  3346)             avg_Tsf      = c0
  3347) 
  3348)       !-----------------------------------------------------------------
  3349)       ! Condition 1: check for Tsf > 0
  3350)       ! If Tsf > 0, set Tsf = 0 and leave converged=.true.
  3351)       !-----------------------------------------------------------------
  3352) 
  3353)             if (Tsf(m) > puny) then
  3354)                Tsf(m) = c0
  3355)                dTsf(ij) = -Tsf_start(ij)
  3356) 
  3357)       !-----------------------------------------------------------------
  3358)       ! Condition 2: check for oscillating Tsf
  3359)       ! If oscillating, average all temps to increase rate of convergence.
  3360)       ! It is possible that this may never occur.
  3361)       !-----------------------------------------------------------------
  3362) 
  3363)             elseif (niter > 1 &                ! condition (2)
  3364)               .and. Tsf_start(ij) <= -puny &
  3365)               .and. abs(dTsf(ij)) > puny &
  3366)               .and. abs(dTsf_prev(m)) > puny &
  3367)               .and. -dTsf(ij)/(dTsf_prev(m)+puny*puny) > p5) then
  3368) 
  3369)                avg_Tsf  = c1  ! average with starting temp  
  3370)                dTsf(ij) = p5 * dTsf(ij)
  3371)                converged(m) = .false.
  3372)                all_converged = .false.
  3373)             endif
  3374) 
  3375)       !-----------------------------------------------------------------
  3376)       ! If condition 2 failed, average new surface temperature with
  3377)       !  starting value.
  3378)       !-----------------------------------------------------------------
  3379)             Tsf(m)  = Tsf(m) &
  3380)                       + avg_Tsf * p5 * (Tsf_start(ij) - Tsf(m))
  3381) 
  3382)          enddo  ! ij
  3383) 
  3384) !DIR$ CONCURRENT !Cray
  3385) !cdir nodep      !NEC
  3386) !ocl novrec      !Fujitsu
  3387)          do ij = 1, isolve
  3388)             i = indxii(ij)
  3389)             j = indxjj(ij)
  3390)             m = indxij(ij)
  3391) 
  3392)       !-----------------------------------------------------------------
  3393)       ! Condition 3: check for large change in Tsf
  3394)       !-----------------------------------------------------------------
  3395) 
  3396)             if (abs(dTsf(ij)) > Tsf_errmax) then
  3397)                converged(m) = .false.
  3398)                all_converged = .false.
  3399)             endif
  3400) 
  3401)       !-----------------------------------------------------------------
  3402)       ! Condition 4: check for fsurfn < fcondtopn with Tsf > 0
  3403)       !-----------------------------------------------------------------
  3404) 
  3405)             fsurfn(i,j) = fsurfn(i,j) + dTsf(ij)*dfsurf_dT(ij)
  3406)             fcondtopn(i,j) = kh(ij) * (Tsf(m)-Tbot(i,j))
  3407) 
  3408)             if (Tsf(m) > -puny .and. fsurfn(i,j) < fcondtopn(i,j)) then
  3409)                converged(m) = .false.
  3410)                all_converged = .false.
  3411)             endif
  3412) 
  3413)             fcondbot(m) = fcondtopn(i,j)
  3414) 
  3415)             dTsf_prev(m) = dTsf(ij)
  3416) 
  3417)          enddo                  ! ij
  3418) 
  3419)          deallocate(diag)
  3420)          deallocate(rhs)
  3421)          deallocate(kh)
  3422)          deallocate(heff)
  3423)          deallocate(Tsf_start)
  3424)          deallocate(dTsf)
  3425)          deallocate(dfsurf_dT)
  3426) 
  3427)       enddo                     ! temperature iteration niter
  3428) 
  3429)       if (.not.all_converged) then
  3430) 
  3431)          do ij = 1, icells
  3432)             i = indxi(ij)
  3433)             j = indxj(ij)
  3434) 
  3435)       !-----------------------------------------------------------------
  3436)       ! Check for convergence failures.
  3437)       !-----------------------------------------------------------------
  3438)             if (.not.converged(ij)) then
  3439)                write(nu_diag,*) 'Thermo iteration does not converge,', &
  3440)                                 'istep1, my_task, i, j:', &
  3441)                                  istep1, my_task, i, j
  3442)                write(nu_diag,*) 'Ice thickness:',  hilyr(ij)*nilyr
  3443)                write(nu_diag,*) 'Snow thickness:', hslyr(ij)*nslyr
  3444)                write(nu_diag,*) 'dTsf, Tsf_errmax:',dTsf_prev(ij), &
  3445)                                  Tsf_errmax
  3446)                write(nu_diag,*) 'Tsf:', Tsf(ij)
  3447)                write(nu_diag,*) 'fsurfn:', fsurfn(i,j)
  3448)                write(nu_diag,*) 'fcondtopn, fcondbot', &
  3449)                                  fcondtopn(i,j), fcondbot(ij)
  3450)                l_stop = .true.
  3451)                istop = i
  3452)                jstop = j
  3453)                return
  3454)             endif
  3455)          enddo                  ! ij
  3456)       endif                     ! all_converged
  3457) 
  3458)       !-----------------------------------------------------------------
  3459)       ! Check that if Tsfc < 0, then fcondtopn = fsurfn
  3460)       !-----------------------------------------------------------------
  3461) 
  3462)       if (l_zerolayerchecks) then

  3464)             i = indxi(ij)
  3465)             j = indxj(ij)
  3466)             
  3467)             if (Tsf(ij) < c0 .and. & 
  3468)                   abs(fcondtopn(i,j)-fsurfn(i,j)) > puny) then

  3475)                write(nu_diag,*) 'fsurfn=',fsurfn(i,j)

  3477)                istop = i
  3478)                jstop = j
  3479)                return
  3480)             endif
  3481)          enddo                  ! ij
  3482)       endif                     ! l_zerolayerchecks
  3483) 
  3484) 
  3485) !DIR$ CONCURRENT !Cray
  3486) !cdir nodep      !NEC
  3487) !ocl novrec      !Fujitsu
  3488)       do ij = 1, icells
  3489)          i = indxi(ij)
  3490)          j = indxj(ij)
  3491) 
  3492)          ! update fluxes that depend on Tsf
  3493)          flwoutn(i,j) = flwoutn(i,j) + dTsf_prev(ij) * dflwout_dT(ij)
  3494)          fsensn(i,j)  = fsensn(i,j)  + dTsf_prev(ij) * dfsens_dT(ij)
  3495)          flatn(i,j)   = flatn(i,j)   + dTsf_prev(ij) * dflat_dT(ij)

  3498)          fswabsn(i,j) = fswsfc(i,j) + fswthrun(i,j)

  3500)       enddo                     ! ij
