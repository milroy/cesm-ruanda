    19) ! Purpose: Hybrid level definitions: p = a*p0 + b*ps
    20) !          interfaces   p(k) = hyai(k)*ps0 + hybi(k)*ps

    27) real(r8), public, target :: hybi(plevp) ! ps component of hybrid coordinate - interfaces
    28) real(r8), public, target :: hybm(plev)  ! ps component of hybrid coordinate - midpoints

    54) subroutine hycoef_init(file)
    55)    use cam_history_support, only: add_vert_coord, formula_terms_t
    56) 
    57)    !----------------------------------------------------------------------- 
    58)    ! 
    59)    ! Purpose: 
    60)    ! Defines the locations of model interfaces from input data in the
    61)    ! hybrid coordinate scheme.  Actual pressure values of model level
    62)    ! interfaces are determined elsewhere from the fields set here.
    63)    ! 
    64)    ! Method: 
    65)    ! the following fields are set:
    66)    ! hyai     fraction of reference pressure used for interface pressures
    67)    ! hyam     fraction of reference pressure used for midpoint pressures
    68)    ! hybi     fraction of surface pressure used for interface pressures
    69)    ! hybm     fraction of surface pressure used for midpoint pressures
    70)    ! hybd     difference of hybi's
    71)    ! hypi     reference state interface pressures
    72)    ! hypm     reference state midpoint pressures
    73)    ! hypd     reference state layer thicknesses
    74)    ! hypdln   reference state layer thicknesses (log p)
    75)    ! hyalph   distance from interface to level (used in integrals)
    76)    ! prsfac   log pressure extrapolation factor (used to compute psl)
    77)    ! 
    78)    ! Author: B. Boville
    79)    ! 
    80)    !-----------------------------------------------------------------------
    81) 
    82)    ! arguments
    83)    type(file_desc_t), intent(in) :: file
    84) 
    85)    ! local variables
    86)    integer  :: k        ! Level index
    87)    real(r8) :: amean, bmean, atest, btest, eps
    88)    type(formula_terms_t) :: formula_terms ! For the 'lev' and 'ilev' coords
    89)    !-----------------------------------------------------------------------
    90) 
    91) 
    92)    ! read hybrid coeficients
    93)    call hycoef_read(file)
    94) 
    95)    ! Set layer locations
    96)    nprlev = 0
    97)    do k=1,plev
    98) 
    99)       ! Interfaces. Set nprlev to the interface above, the first time a 
   100)       ! nonzero surface pressure contribution is found. "nprlev" 
   101)       ! identifies the lowest pure pressure interface.
   102) 
   103)       if (nprlev==0 .and. hybi(k).ne.0.0_r8) nprlev = k - 1
   104)    end do
   105) 
   106)    ! Set nprlev if no nonzero b's have been found. All interfaces are 
   107)    ! pure pressure. A pure pressure model requires other changes as well. 
   108)    if (nprlev==0) nprlev = plev + 2
   109) 
   110)    ! Set delta sigma part of layer thickness and reference state midpoint 
   111)    ! pressures
   112)    do k=1,plev
   113)       hybd(k) = hybi(k+1) - hybi(k)
   114)       hypm(k) = hyam(k)*ps0 + hybm(k)*psr
   115)       etamid(k) = hyam(k) + hybm(k)
   116)    end do
   117) 
   118)    ! Reference state interface pressures
   119)    do k=1,plevp
   120)       hypi(k) = hyai(k)*ps0 + hybi(k)*psr
   121)    end do
   122) 
   123)    ! Reference state layer thicknesses
   124)    do k=1,plev
   125)       hypd(k) = hypi(k+1) - hypi(k)
   126)    end do
   127) 
   128)    ! Test that A's and B's at full levels are arithmetic means of A's and
   129)    ! B's at interfaces
   130)    eps    = 1.e-05_r8
   131)    do k = 1,plev
   132)       amean = ( hyai(k+1) + hyai(k) )*0.5_r8
   133)       bmean = ( hybi(k+1) + hybi(k) )*0.5_r8
   134)       if(amean == 0._r8 .and. hyam(k) == 0._r8) then
   135)          atest = 0._r8
   136)       else
   137)          atest = abs( amean - hyam(k) )/ ( 0.5_r8*( abs(amean + hyam(k)) ) )
   138)       endif
   139)       if(bmean == 0._r8 .and. hybm(k) == 0._r8) then
   140)          btest = 0._r8
   141)       else
   142)          btest = abs( bmean - hybm(k) )/ ( 0.5_r8*( abs(bmean + hybm(k)) ) )
   143)       endif
   144)       if (atest > eps) then
   145)          if (masterproc) then
   146)             write(iulog,9850)
   147)             write(iulog,*)'k,atest,eps=',k,atest,eps
   148)          end if
   149)       end if
   150)       
   151)       if (btest > eps) then
   152)          if (masterproc) then
   153)             write(iulog,9850)
   154)             write(iulog,*)'k,btest,eps=',k,btest,eps
   155)          end if
   156)       end if
   157)    end do
   158) 
   159)    ! Add the information for the 'lev' and 'ilev' mdim history coordinates
   160)    ! Note that this must be after hycoef_init and before any addfld calls
   161)    !
   162)    ! 0.01 converts Pascals to millibars
   163)    !
   164)    alev(:plev) = 0.01_r8*ps0*(hyam(:plev) + hybm(:plev))
   165)    ailev(:plevp) = 0.01_r8*ps0*(hyai(:plevp) + hybi(:plevp))
   166) 
   167)    formula_terms%a_name       =  'hyam'
   168)    formula_terms%a_long_name  =  'hybrid A coefficient at layer midpoints'
   169)    formula_terms%a_values     => hyam
   170)    formula_terms%b_name       =  'hybm'
   171)    formula_terms%b_long_name  =  'hybrid B coefficient at layer midpoints'
   172)    formula_terms%b_values     => hybm
   173)    formula_terms%p0_name      =  'P0'
   174)    formula_terms%p0_long_name = 'reference pressure'
   175)    formula_terms%p0_units     =  'Pa'
   176)    formula_terms%p0_value     =  ps0
   177)    formula_terms%ps_name      =  'PS'
   178) 
   179)    call add_vert_coord('lev', plev,                                         &
   180)         'hybrid level at midpoints (1000*(A+B))', 'hPa', alev,              &
   181)         positive='down',                                                    &
   182)         standard_name='atmosphere_hybrid_sigma_pressure_coordinate',        &
   183)         formula_terms=formula_terms)
   184) 
   185)    formula_terms%a_name       =  'hyai'
   186)    formula_terms%a_long_name  =  'hybrid A coefficient at layer interfaces'
   187)    formula_terms%a_values     => hyai
   188)    formula_terms%b_name       =  'hybi'
   189)    formula_terms%b_long_name  =  'hybrid B coefficient at layer interfaces'
   190)    formula_terms%b_values     => hybi
   191)    formula_terms%p0_name      =  'P0'
   192)    formula_terms%p0_long_name = 'reference pressure'
   193)    formula_terms%p0_units     =  'Pa'
   194)    formula_terms%p0_value     =  ps0
   195)    formula_terms%ps_name      =  'PS'
   196) 
   197)    call add_vert_coord('ilev', plevp,                                       &
   198)         'hybrid level at interfaces (1000*(A+B))', 'hPa', ailev,            &
   199)         positive='down',                                                    &
   200)         standard_name='atmosphere_hybrid_sigma_pressure_coordinate',        &
   201)         formula_terms=formula_terms)
   202) 
   203)    if (masterproc) then
   204)       write(iulog,'(a)')' Layer Locations (*1000) '
   205)       do k=1,plev
   206)          write(iulog,9800)k,hyai(k),hybi(k),hyai(k)+hybi(k)
   207)          write(iulog,9810) hyam(k), hybm(k), hyam(k)+hybm(k)
   208)       end do
   209) 
   210)       write(iulog,9800)plevp,hyai(plevp),hybi(plevp),hyai(plevp)+hybi(plevp)
   211)       write(iulog,9820)
   212)       do k=1,plev
   213)          write(iulog,9830) k, hypi(k)
   214)          write(iulog,9840) hypm(k), hypd(k)
   215)       end do
   216)       write(iulog,9830) plevp, hypi(plevp)
   217)     end if
   218) 
   219) 9800 format( 1x, i3, 3p, 3(f10.4,10x) )
   220) 9810 format( 1x, 3x, 3p, 3(10x,f10.4) )
   221) 9820 format(1x,'reference pressures (Pa)')
   222) 9830 format(1x,i3,f15.4)
   223) 9840 format(1x,3x,15x,2f15.4)
   224) 9850 format('HYCOEF: A and/or B vertical level coefficients at full',/, &
   225)          ' levels are not the arithmetic mean of half-level values')
   226) 
   227) end subroutine hycoef_init

   269) subroutine hycoef_read(File)
   270) 
   271)    ! This code is used both for initial and restart reading.
   272) 
   273)    type(file_desc_t), intent(in) :: File
   274) 
   275)    integer :: flev, filev, lev_dimid, ierr
   276)    character(len=*), parameter :: routine = 'hycoef_read'
   277) 
   278)    ! PIO traps errors internally, no need to check ierr
   279) 
   280)    ierr = PIO_Inq_DimID(File, 'lev', lev_dimid)
   281)    ierr = PIO_Inq_dimlen(File, lev_dimid, flev)
   282)    if (plev /= flev) then
   283)       write(iulog,*) routine//': ERROR: file lev does not match model. lev (file, model):',flev, plev
   284)       call endrun(routine//': ERROR: file lev does not match model.')
   285)    end if
   286) 
   287)    ierr = PIO_Inq_DimID(File, 'ilev', lev_dimid)
   288)    ierr = PIO_Inq_dimlen(File, lev_dimid, filev)
   289)    if (plevp /= filev) then
   290)       write(iulog,*) routine//':ERROR: file ilev does not match model plevp (file, model):',filev, plevp
   291)       call endrun(routine//':ERROR: file ilev does not match model.')
   292)    end if
   293) 
   294)    ierr = pio_inq_varid(File, 'hyai', hyai_desc)
   295)    ierr = pio_inq_varid(File, 'hyam', hyam_desc)
   296)    ierr = pio_inq_varid(File, 'hybi', hybi_desc)
   297)    ierr = pio_inq_varid(File, 'hybm', hybm_desc)
   298) 
   299)    ierr = pio_get_var(File, hyai_desc, hyai)
   300)    ierr = pio_get_var(File, hybi_desc, hybi)
   301)    ierr = pio_get_var(File, hyam_desc, hyam)
   302)    ierr = pio_get_var(File, hybm_desc, hybm)
   303) 
   304) #if ( defined OFFLINE_DYN )
   305)    ! make sure top interface is non zero for fv dycore
   306)    if (hyai(1) .eq. 0._r8) then
   307)       if (hybm(1) .ne. 0.0_r8) then
   308)          hyai(1) = hybm(1)*1.e-2_r8
   309)       else if (hyam(1) .ne. 0.0_r8) then
   310)          hyai(1) = hyam(1)*1.e-2_r8
   311)       else
   312)          call endrun('Not able to set hyai(1) to non-zero.')
   313)       end if
   314)    end if
   315) #endif
   316) 
   317) end subroutine hycoef_read
