   189)   subroutine prescribed_ozone_adv( state, pbuf2d)
   190) 
   191)     use tracer_data,  only : advance_trcdata
   192)     use physics_types,only : physics_state
   193)     use ppgrid,       only : begchunk, endchunk
   194)     use ppgrid,       only : pcols, pver
   195)     use string_utils, only : to_lower, GLC
   196)     use cam_history,  only : outfld
   197)     use cam_control_mod, only: aqua_planet
   198)     use phys_control, only : cam_physpkg_is
   199)     use physconst,    only : mwdry                ! molecular weight dry air ~ kg/kmole
   200)     use physconst,    only : boltz                ! J/K/molecule
   201)     
   202)     use physics_buffer, only : physics_buffer_desc, pbuf_get_chunk, pbuf_get_field, pbuf_set_field
   203) 
   204)     implicit none
   205) 
   206)     type(physics_state), intent(in)    :: state(begchunk:endchunk)                 
   207)     
   208)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   209) 
   210)     type(physics_buffer_desc), pointer :: pbuf_chnk(:)
   211)     integer :: c,ncol
   212)     real(r8) :: to_mmr(pcols,pver)
   213)     real(r8) :: molmass
   214)     real(r8) :: amass
   215)     real(r8) :: outdata(pcols,pver)
   216)     real(r8),pointer :: tmpptr(:,:)
   217) 
   218)     character(len=32) :: units_str
   219) 
   220)     if( .not. has_prescribed_ozone ) return
   221) 
   222)     if( cam_physpkg_is('cam3') .and. aqua_planet ) then
   223)        molmass = 48._r8
   224)        amass   = 28.9644_r8
   225)     else
   226)        molmass = 47.9981995_r8
   227)        amass   = mwdry
   228)     end if
   229) 
   230)     call advance_trcdata( fields, file, state, pbuf2d )
   231) 
   232)     units_str = trim(to_lower(trim(fields(1)%units(:GLC(fields(1)%units)))))
   233) 
   234)     ! set the correct units and invoke history outfld
   235) !$OMP PARALLEL DO PRIVATE (C, NCOL, OUTDATA, TO_MMR, TMPPTR, PBUF_CHNK)
   236)     do c = begchunk,endchunk
   237)        ncol = state(c)%ncol
   238)      
   239)        select case ( units_str )
   240)        case ("molec/cm3","/cm3","molecules/cm3","cm^-3","cm**-3")
   241)           to_mmr(:ncol,:) = (molmass*1.e6_r8*boltz*state(c)%t(:ncol,:))/(amass*state(c)%pmiddry(:ncol,:))
   242)        case ('kg/kg','mmr')
   243)           to_mmr(:ncol,:) = 1._r8
   244)        case ('mol/mol','mole/mole','vmr','fraction')
   245)           to_mmr(:ncol,:) = molmass/amass
   246)        case default
   247)           write(iulog,*) 'prescribed_ozone_adv: units = ',trim(fields(1)%units) ,' are not recognized'
   248)           call endrun('prescribed_ozone_adv: units are not recognized')
   249)        end select
   250) 
   251)        pbuf_chnk => pbuf_get_chunk(pbuf2d, c)
   252)        call pbuf_get_field(pbuf_chnk, fields(1)%pbuf_ndx, tmpptr )
   253) 
   254)        tmpptr(:ncol,:) = tmpptr(:ncol,:)*to_mmr(:ncol,:)
   255) 
   256)        outdata(:ncol,:) = (amass/molmass)* tmpptr(:ncol,:) ! vmr
   257)        call outfld( fields(1)%fldnam, outdata(:ncol,:), ncol, state(c)%lchnk )
   258)     enddo
   259) 
   260)   end subroutine prescribed_ozone_adv
