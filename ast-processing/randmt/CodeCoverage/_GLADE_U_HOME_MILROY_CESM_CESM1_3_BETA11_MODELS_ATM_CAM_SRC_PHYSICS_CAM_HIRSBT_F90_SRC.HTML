    19)     subroutine hirsrtm(lchnk, ncol, pp, tt, rmix, co2mmr, o3mix, ts, oro, &
    20)                        tb_ir, britemp)
    21) 
    22) ! mji hirsrtm
    23) ! Code includes modifications for F90 formatting and for application
    24) ! to NCAR CAM based on the original code in CCM3.
    25) ! M. J. Iacono, AER Inc., May 2004

    29) !      SUBROUTINE HIRSRTM(PP, TT, RMIX, O3MIX, TS, ORO,
    30) !     $                   TB_IR,BRITEMP)
    31) ! -- ------------------------------------------------------------------2
    32) !
    33) !                  ***       VERSION 2.0        ***
    34) !
    35) ! This subroutine calculates brightness temperatures at the top of the
    36) ! atmosphere for 7 TOVS/HIRS channels (2,4,6,8,10,11,12) and 4 TOVS/MSU 
    37) ! channels (1,2,3,4).
    38) !
    39) ! On Input:
    40) !    pp      -  level pressure (hPa) at which fluxes are evaluated
    41) !               from top of the atmosphere to surface
    42) !    tt      -  layer temperatures (K)
    43) !    rmix    -  layer H2O mass mixing ratio in kg/kg
    44) !    co2mix  -  layer CO2 mass mixing ratio kg/kg 
    45) !    o3mix   -  layer ozone mass mixing ratio kg/kg 
    46) !    ts      -  surface temperature (K)
    47) !    oro     - land-sea flag (sea=0, land=1)
    48) !
    49) ! On Ouput:
    50) !    tb_ir   -  infrared brightness temperatures
    51) !    britemp -  microwave brightness temperatures
    52) !
    53) !
    54) !  A flag to include the 4 MSU channels can be switched on (msu_flag=1)
    55) !  and off (msu_flag=0). To decrease the amount of computer time, the
    56) !  microwave routine is changed to a lookup table with almost the same 
    57) !  accuracy as the original routine.
    58) !
    59) ! **  last revised 3/31/97 by Richard Engelen **
    60) !
    61) !   This version differs from original version:
    62) !
    63) !     1.  New NOAA10 coefficients
    64) !     2.  Continuum added
    65) !     3.  Any level exceeding 100% RH is changed to 100% RH
    66) !     4.  New channels (2,4,6,8,10,11,12)
    67) !
    68) !-----------------------------------------------------------------------
    69) 
    70) !------------------------------Arguments--------------------------------
    71) !
    72) ! Input arguments
    73) !
    74)    integer, intent(in) :: lchnk                 ! chunk identifier
    75)    integer, intent(in) :: ncol                  ! number of atmospheric columns
    76) 
    77)    real(r8), intent(in) :: pp(pcols,pverp)      ! Pressure for each column and layer (hPa)
    78)    real(r8), intent(in) :: tt(pcols,pver)       ! Temperature of each column and layer (K)
    79)    real(r8), intent(in) :: ts(pcols)            ! Surface temperature (K)
    80)    real(r8), intent(in) :: rmix(pcols,pver)     ! Water vapor mass mixing ratio (kg/kg)
    81)    real(r8), intent(in) :: co2mmr(pcols)        ! CO2 mass mixing ratio (kg/kg)
    82)    real(r8), intent(in) :: o3mix(pcols,pver)    ! Ozone mass mixing ratio (kg/kg)
    83)    real(r8), intent(in) :: oro(pcols)           ! Land surface flag, sea=0, land=1
    84) !
    85) ! Output arguments
    86) !
    87)    real(r8), intent(out) :: britemp(pcols,pnf_msu)   ! HIRS brightness temperatures
    88)    real(r8), intent(out) :: tb_ir(pcols,pnb_hirs)    ! MSU brightness temperatures
    89) 
    90) !------------------------------Local variables--------------------------
    91) 
    92) !      real(r8), parameter :: rmwco2 = mwco2/mwdry   ! ratio of molecular weights of co2 to dry air
    93) 
    94)       real(r8)  uh2o(pver)
    95)       real(r8)  uo3(pver)
    96)       real(r8)  a_hirs(pnb_hirs)
    97)       real(r8)  b_hirs(pnb_hirs)
    98)       real(r8)  xh2o(pnb_hirs)
    99)       real(r8)  yh2o(pnb_hirs)
   100)       real(r8)  xo3(pnb_hirs)
   101)       real(r8)  yo3(pnb_hirs)
   102)       real(r8)  xco2(pnb_hirs)
   103)       real(r8)  yco2(pnb_hirs)
   104)       real(r8)  b_ir(pnb_hirs)
   105)       real(r8)  otrans(pnb_hirs)
   106)       real(r8)  tband(pnb_hirs)
   107)       real(r8)  scoef10(pnb_hirs)
   108)       real(r8)  fcoef10(pnb_hirs)
   109)       real(r8)  cwn(pnb_hirs)
   110)       real(r8)  dtrans(pnb_hirs)
   111)       real(r8)  rad_lay(pnb_hirs)
   112)       real(r8)  radir(pnb_hirs)
   113)       real(r8)  rad(pnb_hirs)
   114)       real(r8)  rad2(pnb_hirs)
   115)       real(r8)  rad3(pnb_hirs)
   116)       real(r8)  refl(pnb_hirs)
   117)       real(r8)  otr_mw(pnf_msu)
   118)       real(r8)  tau(pnf_msu)
   119)       real(r8)  trans(pnf_msu)
   120)       real(r8)  otr_mw2(pnf_msu)
   121)       real(r8)  tau2(pnf_msu)
   122)       real(r8)  trans2(pnf_msu)
   123) !     real(r8)  britemp1(pcols,pnf_msu)
   124) !     real(r8)  britemp2(pcols,pnf_msu)
   125) !     real(r8)  britemp3(pcols,pnf_msu)
   126)       real(r8)  freq(pnf_msu)
   127)       real(r8)  upath_h2o
   128)       real(r8)  upath_co2
   129)       real(r8)  upath_o3
   130)       real(r8)  ucont1
   131)       real(r8)  ucont2
   132)       real(r8)  ppath_h2o
   133)       real(r8)  ppath_co2
   134)       real(r8)  ppath_o3
   135)       real(r8)  sfctemp                     ! surface temperature
   136)       real(r8)  dp                          ! pressure depth of layer
   137)       real(r8)  tlay                        ! layer temperature
   138)       real(r8)  play                        ! pressure
   139)       real(r8)  rlay                        ! water vapor mixing ratio
   140)       real(r8)  o3lay                       ! ozone mixing ratio.
   141)       real(r8)  rhoair                      ! layer density
   142)       real(r8)  e                           ! saturation pressure
   143)       real(r8)  delz                        ! height of layer
   144)       real(r8)  dp2                         ! pressure depth of layer below
   145)       real(r8)  tlay2
   146)       real(r8)  play2
   147)       real(r8)  rlay2
   148)       real(r8)  rhoair2
   149)       real(r8)  e2
   150)       real(r8)  delz2
   151)       real(r8)  b_mw
   152)       real(r8)  b_mw2
   153)       real(r8)  dtr_mw
   154)       real(r8)  uco2(pver)
   155) !      real(r8)  uco2
   156)       real(r8)  pw
   157)       real(r8)  psc_h2o                   ! partial pressure of h2o
   158)       real(r8)  psc_co2                   ! partial pressure of co2
   159)       real(r8)  psc_o3                    ! partial pressure of o3
   160)       real(r8)  t_cont
   161)       real(r8)  t_h2o
   162)       real(r8)  t_co2
   163)       real(r8)  t_o3
   164)       real(r8)  abs(pnf_msu)              ! absorption
   165)       real(r8)  abs2(pnf_msu)
   166)       real(r8)  dtr_mw2
   167) !
   168) !      real(r8) t_malk, plnck, btemp
   169) !      external t_malk, plnck, btemp
   170)       integer  i, ib, if, icol    ! loop control
   171) 
   172) !------------------------------data statments---------------------------
   173)       data scoef10/3.09_r8,2.64_r8,2.28_r8,1.004_r8,0.429_r8,1.945_r8,11.95_r8/
   174)       data fcoef10/.00434_r8,.00301_r8,.0018_r8,4.33e-5_r8,0.000393_r8,0.0738_r8,1.110_r8/
   175)       data a_hirs/.0183_r8,-.00203_r8,.0653_r8,0.21797_r8,0.29846_r8,0.04612_r8,0.06453_r8/
   176)       data b_hirs/.99992_r8,.99994_r8,.9998_r8,.99957_r8,.9996_r8,.99963_r8,1.0006_r8/
   177)       data cwn/680.23_r8,704.33_r8,733.13_r8,899.5_r8,1224.07_r8,1363.32_r8,1489.42_r8/
   178)       data freq/50.31_r8,53.73_r8,54.96_r8,57.95_r8/
   179) 
   180) 
   181)       data xh2o/0.41_r8,0.52_r8,0.17_r8,0.05_r8,0.68_r8,5.02_r8,17.24_r8/
   182)       data yh2o/0.70_r8,1.41_r8,0.14_r8,0.02_r8,1.09_r8,125.9_r8,1194.5_r8/
   183) 
   184)       data xco2/64.32_r8,13.29_r8,5.04_r8,0.05_r8,0.03_r8,0.25_r8,0.01_r8/
   185)       data yco2/1325.2_r8,121.01_r8,14.36_r8,0.01_r8,0.03_r8,0.02_r8,0.01_r8/
   186) 
   187)       data xo3/32.83_r8,28.9_r8,27.44_r8,0.01_r8,0.68_r8,0.37_r8,0.01_r8/
   188)       data yo3/77.44_r8,66.7_r8,67.44_r8,1.6_r8,4.17_r8,0.07_r8,0.01_r8/
   189) 
   190)       real(r8) grav, r
   191) !      real(r8) grav, rco2, r
   192) !      data grav,rco2,r/9.8,0.523e-3,287./
   193)       real(r8) eps
   194)       data eps/0.5_r8/
   195) 
   196) ! use values for constants consistent with radiation code.
   197)       grav = gravit
   198)       r = rair
   199) !      rco2 = co2vmr*rmwco2
   200) 
   201)       do icol=1,ncol
   202) 
   203)         sfctemp=ts(icol)
   204) 
   205)         do ib=1,pnb_hirs
   206)           radir(ib)=0._r8
   207)         enddo
   208) 
   209)         upath_h2o=0.0_r8
   210)         upath_co2=0.0_r8
   211)         upath_o3=0.0_r8
   212)         ucont1=0._r8
   213)         ucont2=0._r8
   214)         ppath_h2o=0.0_r8
   215)         ppath_co2=0.0_r8
   216)         ppath_o3=0._r8
   217) !       if(msu_flag.eq.1) then
   218)           do if=1,pnf_msu
   219)              tau(if)=0._r8
   220)              trans(if)=0._r8
   221)              rad(if)=0._r8
   222)              otr_mw(if)=1._r8
   223)              tau2(if)=0._r8
   224)              trans2(if)=0._r8
   225)              rad2(if)=0._r8
   226)              otr_mw2(if)=1._r8
   227)           end do
   228) !       endif
   229)         do ib=1,pnb_hirs
   230)            tband(ib)=1.0_r8
   231)            otrans(ib)=1._r8
   232)           end do
   233) 
   234) 
   235)         do i=1,pver
   236)       
   237)           dp=pp(icol,i+1)-pp(icol,i)
   238)           tlay=tt(icol,i)
   239)           play=sqrt(pp(icol,i)*pp(icol,i+1))
   240)           rlay=rmix(icol,i)
   241)           o3lay=o3mix(icol,i)
   242)           
   243)           rhoair=play*100._r8/(r*tt(icol,i))
   244)           e = play*rlay/(rlay+0.6220_r8)
   245)           delz=dp*0.1_r8/(rhoair*grav)
   246)           
   247) !         if(msu_flag.eq.1) then
   248)             dp2=pp(icol,pver+2-i)-pp(icol,pver+1-i)
   249)             tlay2=tt(icol,pver+1-i)
   250)             play2=sqrt(pp(icol,pver+1-i)*pp(icol,pver+2-i))
   251)             rlay2=rmix(icol,pver+1-i)
   252) 
   253)             rhoair2=play2*100._r8/(r*tt(icol,pver+1-i))
   254)             e2 = play2*rlay2/(rlay2+0.6220_r8)
   255)             delz2=dp2*0.1_r8/(rhoair2*grav)
   256)           
   257) !
   258) !         microwave transfer
   259) !
   260)             call lookup ( play-e, tlay, abs)
   261)             call lookup (play2-e2, tlay2, abs2)
   262)             do if=1,pnf_msu
   263)               tau(if)=tau(if)+abs(if)*delz
   264)               trans(if) = exp(-tau(if))
   265)               b_mw = 1.47445e-23_r8*freq(if)**3/(exp(0.047981_r8*freq(if) &
   266)                      /tlay)-1)
   267)               dtr_mw = otr_mw(if)-trans(if)
   268)               rad(if)=rad(if) + b_mw*dtr_mw
   269)               otr_mw(if)=trans(if)
   270)            
   271)               tau2(if)=tau2(if)+abs2(if)*delz2
   272)               trans2(if) = exp(-tau2(if))
   273)               b_mw2 = 1.47445e-23_r8*freq(if)**3/(exp(0.047981_r8*freq(if) &
   274)                    /tlay2)-1)
   275)               dtr_mw2 = otr_mw2(if)-trans2(if)
   276)               rad2(if)=rad2(if) + b_mw2*dtr_mw2
   277)               otr_mw2(if)=trans2(if)
   278)             end do
   279) !         endif
   280) 
   281) !
   282) !                ir transfer
   283) !
   284)           uh2o(i)=rlay*100._r8*dp/grav
   285)           uo3(i)=o3lay*100._r8*dp/grav
   286)           uco2(i)=co2mmr(icol)*100._r8*dp/grav
   287) !          uco2=rco2*100.*dp/grav
   288)           pw=play*rlay*28.97_r8/18._r8
   289)           ucont1=ucont1+uh2o(i)*((play-pw)/1013._r8)*(296._r8/tlay)
   290)           ucont2=ucont2+uh2o(i)*(pw/1013._r8)*(296._r8/tlay)
   291) 
   292)           upath_h2o=upath_h2o+uh2o(i)
   293)           ppath_h2o=ppath_h2o+uh2o(i)*play
   294)           psc_h2o=ppath_h2o/upath_h2o
   295) 
   296)           upath_co2=upath_co2+uco2(i)
   297)           ppath_co2=ppath_co2+uco2(i)*play
   298) !          upath_co2=upath_co2+uco2
   299) !          ppath_co2=ppath_co2+uco2*play
   300)           psc_co2=ppath_co2/upath_co2
   301) 
   302)           upath_o3=upath_o3+uo3(i)
   303)           ppath_o3=ppath_o3+uo3(i)*play
   304)           psc_o3=ppath_o3/upath_o3
   305) 
   306)           do ib=1,pnb_hirs
   307)             t_cont=exp(-scoef10(ib)*ucont2-fcoef10(ib)*ucont1)
   308) 
   309)             t_h2o=t_malk(upath_h2o,psc_h2o,xh2o(ib),yh2o(ib))
   310)             t_co2=t_malk(upath_co2,psc_co2,xco2(ib),yco2(ib))
   311)             t_o3=t_malk(upath_o3,psc_o3,xo3(ib),yo3(ib))
   312)             tband(ib)=t_co2*t_o3*t_h2o*t_cont
   313) 
   314) 
   315)             b_ir(ib)=plnck(cwn(ib),a_hirs(ib),b_hirs(ib),tlay)
   316)             dtrans(ib)=otrans(ib)-tband(ib)
   317)             rad_lay(ib)=  b_ir(ib)*dtrans(ib)
   318)             radir(ib)= radir(ib)+ rad_lay(ib)
   319)             otrans(ib) = tband(ib)
   320) 
   321)           end do
   322)         end do     ! end of loop over vertical levels
   323) 
   324) 
   325) !
   326) !    add in the surface contribution to the radiance, including
   327) !    reflection
   328) !
   329)         
   330) !       if(msu_flag.eq.1) then
   331)           if(oro(icol).eq.0)then
   332)             eps=0.65_r8      ! ocean
   333)           else
   334)             eps=0.9_r8    ! land or sea-ice
   335)           end if
   336)       
   337)           do if=1,pnf_msu
   338)             b_mw = 1.47445e-23_r8*freq(if)**3/(exp(0.047981_r8*freq(if)/ &
   339)                sfctemp)-1)
   340)             rad(if) = rad(if) + eps*trans(if)*b_mw
   341)             refl(if)=(1-eps)*rad2(if)*trans(if)
   342) !           britemp1(icol,if) = 0.047981*freq(if)/log(1.0 
   343) !    $                + 1.47445e-23*freq(if)**3/rad(if))
   344) !           britemp2(icol,if) = 0.047981*freq(if)/log(1.0 
   345) !    $                + 1.47445e-23*freq(if)**3/rad2(if))
   346) !           britemp3(icol,if) = 0.047981*freq(if)/log(1.0 
   347) !    $                + 1.47445e-23*freq(if)**3/refl(if))
   348)      
   349)             rad3(if) = rad(if) + refl(if)
   350)             britemp(icol,if) = 0.047981_r8*freq(if)/log(1.0_r8  &
   351)                       + 1.47445e-23_r8*freq(if)**3/rad3(if))
   352)           end do
   353) !       endif
   354)       
   355)         do ib=1,pnb_hirs
   356)           b_ir(ib)=plnck(cwn(ib),a_hirs(ib),b_hirs(ib),sfctemp)
   357)           radir(ib)=radir(ib)+b_ir(ib)*tband(ib)
   358)           tb_ir(icol,ib)=btemp(cwn(ib),a_hirs(ib),b_hirs(ib),radir(ib))
   359)         end do
   360) 
   361)       end do   ! end of loop over columns
   362) 
   363)     end subroutine hirsrtm

   367)     function btemp(wvn,a,b,chnrad)
   368) !
   369) !*  calculates the brightness temperature given the channel radiance.
   370) !   uses planck function tuned to tovs frequencies
   371) !
   372)       real(r8) btemp
   373) 
   374) !------------------------------arguments--------------------------------
   375)       real(r8) wvn
   376)       real(r8) chnrad
   377)       real(r8) a
   378)       real(r8) b
   379) !------------------------------local variables--------------------------
   380)       real(r8) c1
   381)       real(r8) c2
   382) !
   383) !   planck function parameters
   384) !   note: parameters a and b are temperature correction factors
   385) !   which are dependent on the channel and satellite.  these
   386) !   parameters were extracted from the rttovs model.
   387) !
   388)       parameter( c1=1.191066e-08_r8 )
   389)       parameter( c2=1.438833_r8 )
   390) 
   391)       btemp=(c2*wvn/log(c1*wvn**3/chnrad+1._r8)-a)/b
   392) 
   393)     end function btemp
   394) !
   395) !==================================================================
   396) !
   397)     function plnck(wvn,a,b,t)
   398) 
   399) !
   400) ! planck function
   401) !
   402)       real(r8) plnck
   403) 
   404)       real(r8) wvn,a,b,t,c1,c2
   405) 
   406)       parameter( c1=1.191066e-08_r8 )
   407)       parameter( c2=1.438833_r8 )
   408) 
   409)       plnck=c1*(wvn)**3/(exp(c2*wvn/(a+b*t))-1._r8)
   410) 
   411)     end function plnck
   412) !
   413) !===================================================================
   414) !
   415)     function t_malk(u,p,x,y)
   416) 
   417)       real(r8) t_malk
   418) 
   419)       real(r8) u, p, x, y
   420)       real(r8) p0, dnu, pi, b, bp
   421) 
   422)       parameter( p0=1013._r8 )
   423)       parameter( dnu=10._r8 )
   424) 
   425)       pi=acos(-1._r8)
   426)       b=4._r8*x**2/(pi*y*dnu)
   427)       bp=pi*b*p/p0
   428)       t_malk = exp(-0.5_r8*bp*(sqrt(1._r8+4._r8*y*u/(dnu*bp))-1._r8))
   429) 
   430)     end function t_malk

   434)     subroutine lookup(p,t,abs)
   435) 
   436)       real(r8) p, t, abs(pnf_msu)
   437) !
   438)       integer if, i, j, n, m
   439)       parameter( n=17 )
   440)       parameter( m=16 )
   441)       real(r8) xx(n), yy(m)
   442)       real(r8) zz1(n,m), zz2(n,m), zz3(n,m), zz4(n,m)
   443)       real(r8) t1, t2
   444)       data yy/5.0_r8,7.5_r8,10._r8,25._r8,50._r8,100._r8,200._r8,300._r8, &
   445)               400._r8,500._r8,600._r8,700._r8,800._r8,900._r8,1000,1050._r8/
   446)       data xx/160._r8,170._r8,180._r8,190._r8,200._r8,210._r8,220._r8,230._r8,240._r8,250._r8, &
   447)               260._r8,270._r8,280._r8,290._r8,300._r8,310._r8,320._r8/
