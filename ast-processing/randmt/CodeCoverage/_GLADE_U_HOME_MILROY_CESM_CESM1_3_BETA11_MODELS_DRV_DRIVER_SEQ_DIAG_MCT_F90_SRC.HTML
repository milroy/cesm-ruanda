   273) ! !INTERFACE: ------------------------------------------------------------------

   275) subroutine seq_diag_zero_mct(EClock,mode)
   276) 
   277) ! !INPUT/OUTPUT PARAMETERS:
   278) 
   279)    type(ESMF_Clock), intent(in),optional :: EClock
   280)    character(len=*), intent(in),optional :: mode
   281) 
   282) !EOP
   283) 
   284)    integer(IN) :: ip,yr,mon,day,sec
   285)    !----- formats -----
   286)    character(*),parameter :: subName = '(seq_diag_zero_mct) '
   287) 
   288) !-------------------------------------------------------------------------------
   289) !
   290) !-------------------------------------------------------------------------------
   291) 
   292)    if (.not. present(EClock) .and. .not. present(mode)) then
   293)       call shr_sys_abort(subName//' ERROR EClock or mode should be present')
   294)    endif
   295) 
   296)    if (present(EClock)) then
   297)       call seq_timemgr_EClockGetData(EClock,curr_yr=yr, &
   298)          curr_mon=mon,curr_day=day,curr_tod=sec)
   299) 
   300)       do ip = 1,p_size
   301)          if (ip == p_inst) then
   302)             budg_dataL(:,:,ip) = 0.0_r8
   303)             budg_dataG(:,:,ip) = 0.0_r8
   304)             budg_ns(:,:,ip) = 0.0_r8
   305)          endif
   306)          if (ip==p_day .and. sec==0) then
   307)             budg_dataL(:,:,ip) = 0.0_r8
   308)             budg_dataG(:,:,ip) = 0.0_r8
   309)             budg_ns(:,:,ip) = 0.0_r8
   310)          endif
   311)          if (ip==p_mon .and. day==1 .and. sec==0) then
   312)             budg_dataL(:,:,ip) = 0.0_r8
   313)             budg_dataG(:,:,ip) = 0.0_r8
   314)             budg_ns(:,:,ip) = 0.0_r8
   315)          endif
   316)          if (ip==p_ann .and. mon==1 .and. day==1 .and. sec==0) then
   317)             budg_dataL(:,:,ip) = 0.0_r8
   318)             budg_dataG(:,:,ip) = 0.0_r8
   319)             budg_ns(:,:,ip) = 0.0_r8
   320)          endif
   321)       enddo
   322)    endif
   323) 
   324)    if (present(mode)) then
   325)       if (trim(mode) == 'inst') then
   326)          budg_dataL(:,:,p_inst) = 0.0_r8
   327)          budg_dataG(:,:,p_inst) = 0.0_r8
   328)          budg_ns(:,:,p_inst) = 0.0_r8
   329)       elseif (trim(mode) == 'day') then
   330)          budg_dataL(:,:,p_day) = 0.0_r8
   331)          budg_dataG(:,:,p_day) = 0.0_r8
   332)          budg_ns(:,:,p_day) = 0.0_r8
   333)       elseif (trim(mode) == 'mon') then
   334)          budg_dataL(:,:,p_mon) = 0.0_r8
   335)          budg_dataG(:,:,p_mon) = 0.0_r8
   336)          budg_ns(:,:,p_mon) = 0.0_r8
   337)       elseif (trim(mode) == 'ann') then
   338)          budg_dataL(:,:,p_ann) = 0.0_r8
   339)          budg_dataG(:,:,p_ann) = 0.0_r8
   340)          budg_ns(:,:,p_ann) = 0.0_r8
   341)       elseif (trim(mode) == 'inf') then
   342)          budg_dataL(:,:,p_inf) = 0.0_r8
   343)          budg_dataG(:,:,p_inf) = 0.0_r8
   344)          budg_ns(:,:,p_inf) = 0.0_r8
   345)       elseif (trim(mode) == 'all') then
   346)          budg_dataL(:,:,:) = 0.0_r8
   347)          budg_dataG(:,:,:) = 0.0_r8
   348)          budg_ns(:,:,:) = 0.0_r8
   349)       else
   350)          call shr_sys_abort(subname//' ERROR in mode '//trim(mode))
