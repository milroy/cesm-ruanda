   168) subroutine gw_prof (ncol, cpair, t, pmid, pint, rhoi, ti, nm, ni)
   169)   !-----------------------------------------------------------------------
   170)   ! Compute profiles of background state quantities for the multiple
   171)   ! gravity wave drag parameterization.
   172)   !
   173)   ! The parameterization is assumed to operate only where water vapor
   174)   ! concentrations are negligible in determining the density.
   175)   !-----------------------------------------------------------------------
   176)   use gw_utils, only: midpoint_interp
   177)   !------------------------------Arguments--------------------------------
   178)   ! Column dimension.
   179)   integer, intent(in) :: ncol
   180) 
   181)   ! Specific heat of dry air, constant pressure.
   182)   real(r8), intent(in) :: cpair
   183)   ! Midpoint temperatures.
   184)   real(r8), intent(in) :: t(ncol,pver)
   185)   ! Midpoint and interface pressures.
   186)   real(r8), intent(in) :: pmid(ncol,pver), pint(ncol,0:pver)
   187) 
   188)   ! Interface density.
   189)   real(r8), intent(out) :: rhoi(ncol,0:pver)
   190)   ! Interface temperature.
   191)   real(r8), intent(out) :: ti(ncol,0:pver)
   192)   ! Midpoint and interface Brunt-Vaisalla frequencies.
   193)   real(r8), intent(out) :: nm(ncol,pver), ni(ncol,0:pver)
   194) 
   195)   !---------------------------Local Storage-------------------------------
   196)   ! Column and level indices.
   197)   integer :: i,k
   198) 
   199)   ! dt/dp
   200)   real(r8) :: dtdp
   201)   ! Brunt-Vaisalla frequency squared.
   202)   real(r8) :: n2
   203) 
   204)   ! Minimum value of Brunt-Vaisalla frequency squared.
   205)   real(r8), parameter :: n2min = 1.e-8_r8
   206) 
   207)   !------------------------------------------------------------------------
   208)   ! Determine the interface densities and Brunt-Vaisala frequencies.
   209)   !------------------------------------------------------------------------
   210) 
   211)   ! The top interface values are calculated assuming an isothermal
   212)   ! atmosphere above the top level.
   213)   k = 0
   214)   do i = 1, ncol
   215)      ti(i,k) = t(i,k+1)
   216)      rhoi(i,k) = pint(i,k) / (rair*ti(i,k))
   217)      ni(i,k) = sqrt(gravit*gravit / (cpair*ti(i,k)))
   218)   end do
   219) 
   220)   ! Interior points use centered differences.
   221)   ti(:,1:pver-1) = midpoint_interp(t)
   222)   do k = 1, pver-1
   223)      do i = 1, ncol
   224)         rhoi(i,k) = pint(i,k) / (rair*ti(i,k))
   225)         dtdp = (t(i,k+1)-t(i,k)) / (pmid(i,k+1)-pmid(i,k))
   226)         n2 = gravit*gravit/ti(i,k) * (1._r8/cpair - rhoi(i,k)*dtdp)
   227)         ni(i,k) = sqrt(max(n2min, n2))
   228)      end do
   229)   end do
   230) 
   231)   ! Bottom interface uses bottom level temperature, density; next interface
   232)   ! B-V frequency.
   233)   k = pver
   234)   do i = 1, ncol
   235)      ti(i,k) = t(i,k)
   236)      rhoi(i,k) = pint(i,k) / (rair*ti(i,k))
   237)      ni(i,k) = ni(i,k-1)
   238)   end do
   239) 
   240)   !------------------------------------------------------------------------
   241)   ! Determine the midpoint Brunt-Vaisala frequencies.
   242)   !------------------------------------------------------------------------
   243)   nm = midpoint_interp(ni)
   244) 
   245) end subroutine gw_prof

   249) subroutine momentum_energy_conservation(ncol, tend_level, dt, taucd, &
   250)      pint, pdel, u, v, dudt, dvdt, dsdt, utgw, vtgw, ttgw)
   251) 
   252)   ! C.-C. Chen, momentum & energy conservation
   253) 
   254)   integer, intent(in) :: ncol
   255)   integer, intent(in) :: tend_level(ncol)
   256)   real(r8), intent(in) :: dt
   257)   real(r8), intent(in) :: taucd(ncol,0:pver,4)
   258)   real(r8), intent(in) :: pint(ncol,pver)
   259)   real(r8), intent(in) :: pdel(ncol,pver)
   260)   real(r8), intent(in) :: u(ncol,pver)
   261)   real(r8), intent(in) :: v(ncol,pver)
   262)   ! The following are assumed-shape because, for now, CAM is passing them
   263)   ! with size pcols.
   264)   real(r8), intent(inout) :: dudt(:,:)
   265)   real(r8), intent(inout) :: dvdt(:,:)
   266)   real(r8), intent(inout) :: dsdt(:,:)
   267)   real(r8), intent(inout) :: utgw(ncol,pver)
   268)   real(r8), intent(inout) :: vtgw(ncol,pver)
   269)   real(r8), intent(inout) :: ttgw(ncol,pver)
   270)   ! Local variables.
   271)   integer :: i, k
   272)   real(r8) :: dz(ncol), dE(ncol), ut_dz(ncol), vt_dz(ncol)
   273) 
   274)   ! Total mass from ground to source level: rho*dz = dp/gravit
   275)   dz = 0.0_r8
   276)   do k = pver, minval(tend_level)+1, -1
   277)      where (k > tend_level)
   278)         dz = dz + pdel(:,k)/gravit
   279)      end where
   280)   end do
   281) 
   282)   ! Tendency for U & V below source level.
   283)   do i = 1, ncol
   284)      ut_dz(i) = -(taucd(i,tend_level(i), east) + &
   285)           taucd(i,tend_level(i), west))/dz(i)
   286)      vt_dz(i) = -(taucd(i,tend_level(i),north) + &
   287)           taucd(i,tend_level(i),south))/dz(i)
   288)   end do
   289) 
   290)   do k = minval(tend_level)+1, pver
   291)      where (k > tend_level)
   292)         dudt(:ncol,k) = dudt(:ncol,k)+ut_dz
   293)         dvdt(:ncol,k) = dvdt(:ncol,k)+vt_dz
   294)         utgw(:,k) = utgw(:,k)+ut_dz
   295)         vtgw(:,k) = vtgw(:,k)+vt_dz
   296)      end where
   297)   end do
   298) 
   299)   ! Net gain/loss of total energy in the column.
   300)   dE = 0.0_r8
   301)   do k = 1, pver
   302)      dE = dE + pdel(:,k) * (dsdt(:ncol,k) + &
   303)           dudt(:ncol,k)*(u(:,k)+dudt(:ncol,k)*0.5_r8*dt) + &
   304)           dvdt(:ncol,k)*(v(:,k)+dvdt(:ncol,k)*0.5_r8*dt) )
   305)   end do
   306) 
   307)   do i = 1, ncol
   308)      dE(i) = dE(i)/(pint(i,pver)-pint(i,tend_level(i)))
   309)   end do
   310) 
   311)   ! Subtract net gain/loss of total energy below source level.
   312)   do k = minval(tend_level)+1, pver
   313)      where (k > tend_level)
   314)         dsdt(:ncol,k) = dsdt(:ncol,k)-dE
   315)         ttgw(:,k) = ttgw(:,k)-dE
   316)      end where
   317)   end do
   318) 
   319) end subroutine momentum_energy_conservation

   323) subroutine gw_drag_prof(ncol, ngwv, src_level, tend_level, do_taper, dt, &
   324)      lat,           t,    ti,  pmid, pint, dpm,   rdpm, &
   325)      piln, rhoi,    nm,   ni,  ubm,  ubi,  xv,    yv,   &
   326)      effgw,      c, kvtt, q,   dse,  tau,  utgw,  vtgw, &
   327)      ttgw, qtgw, taucd,   egwdffi,   gwut, dttdf, dttke)
   328) 
   329)   !-----------------------------------------------------------------------
   330)   ! Solve for the drag profile from the multiple gravity wave drag
   331)   ! parameterization.
   332)   ! 1. scan up from the wave source to determine the stress profile
   333)   ! 2. scan down the stress profile to determine the tendencies
   334)   !     => apply bounds to the tendency
   335)   !          a. from wkb solution
   336)   !          b. from computational stability constraints
   337)   !     => adjust stress on interface below to reflect actual bounded
   338)   !        tendency
   339)   !-----------------------------------------------------------------------
   340) 
   341)   use gw_diffusion, only: gw_ediff, gw_diff_tend
   342)   use vdiff_lu_solver, only: lu_decomp
   343) 
   344)   !------------------------------Arguments--------------------------------
   345)   ! Column and gravity wave spectrum dimensions.
   346)   integer, intent(in) :: ncol, ngwv
   347)   ! Level from which gravity waves are propagated upward.
   348)   integer, intent(in) :: src_level(ncol)
   349)   ! Lowest level where wind tendencies are calculated.
   350)   integer, intent(in) :: tend_level(ncol)
   351)   ! Using tend_level > src_level allows the orographic waves to prescribe
   352)   ! wave propagation up to a certain level, but then allow wind tendencies
   353)   ! and adjustments to tau below that level.
   354) 
   355)   ! Whether or not to apply the polar taper.
   356)   logical, intent(in) :: do_taper
   357)   ! Time step.
   358)   real(r8), intent(in) :: dt
   359) 
   360)   ! Latitudes for each column.
   361)   real(r8), intent(in) :: lat(ncol)
   362)   ! Midpoint and interface temperatures.
   363)   real(r8), intent(in) :: t(ncol,pver), ti(ncol,0:pver)
   364)   ! Midpoint and interface pressures.
   365)   real(r8), intent(in) :: pmid(ncol,pver), pint(ncol,0:pver)
   366)   ! Layer thickness (delta pressure) and reciprocal of layer thickness.
   367)   real(r8), intent(in) :: dpm(ncol,pver), rdpm(ncol,pver)
   368)   ! Log of interface pressures.
   369)   real(r8), intent(in) :: piln(ncol,0:pver)
   370)   ! Interface densities.
   371)   real(r8), intent(in) :: rhoi(ncol,0:pver)
   372)   ! Midpoint and interface Brunt-Vaisalla frequencies.
   373)   real(r8), intent(in) :: nm(ncol,pver), ni(ncol,0:pver)
   374)   ! Projection of wind at midpoints and interfaces.
   375)   real(r8), intent(in) :: ubm(ncol,pver), ubi(ncol,0:pver)
   376)   ! Unit vectors of source wind (zonal and meridional components).
   377)   real(r8), intent(in) :: xv(ncol), yv(ncol)
   378)   ! Tendency efficiency.
   379)   real(r8), intent(in) :: effgw
   380)   ! Wave phase speeds for each column.
   381)   real(r8), intent(in) :: c(ncol,-pgwv:pgwv)
   382)   ! Molecular thermal diffusivity.
   383)   real(r8), intent(in) :: kvtt(ncol,0:pver)
   384)   ! Constituent array.
   385)   real(r8), intent(in) :: q(:,:,:)
   386)   ! Dry static energy.
   387)   real(r8), intent(in) :: dse(ncol,pver)
   388) 
   389)   ! Wave Reynolds stress.
   390)   real(r8), intent(inout) :: tau(ncol,-pgwv:pgwv,0:pver)
   391)   ! Zonal/meridional wind tendencies.
   392)   real(r8), intent(out) :: utgw(ncol,pver), vtgw(ncol,pver)
   393)   ! Gravity wave heating tendency.
   394)   real(r8), intent(out) :: ttgw(ncol,pver)
   395)   ! Gravity wave constituent tendency.
   396)   real(r8), intent(out) :: qtgw(:,:,:)
   397) 
   398)   ! Reynolds stress for waves propagating in each cardinal direction.
   399)   real(r8), intent(out) :: taucd(ncol,0:pver,4)
   400) 
   401)   ! Effective gravity wave diffusivity at interfaces.
   402)   real(r8), intent(out) :: egwdffi(ncol,0:pver)
   403) 
   404)   ! Gravity wave wind tendency for each wave.
   405)   real(r8), intent(out) :: gwut(ncol,pver,-ngwv:ngwv)
   406) 
   407)   ! Temperature tendencies from diffusion and kinetic energy.
   408)   real(r8), intent(out) :: dttdf(ncol,pver)
   409)   real(r8), intent(out) :: dttke(ncol,pver)
   410) 
   411)   !---------------------------Local storage-------------------------------
   412)   ! Column, level, wavenumber, and constituent loop indices.
   413)   integer :: i, k, l, m
   414) 
   415)   ! "Total" and saturation diffusivity.
   416)   real(r8) :: d(ncol), dsat(ncol)
   417)   ! Fraction of dsat to use.
   418)   real(r8) :: dscal(ncol)
   419)   ! Imaginary part of vertical wavenumber.
   420)   real(r8) :: mi(ncol)
   421)   ! Stress after damping.
   422)   real(r8) :: taudmp(ncol)
   423)   ! Saturation stress.
   424)   real(r8) :: tausat(ncol,-ngwv:ngwv)
   425)   ! ubi at tend_level.
   426)   real(r8) :: ubi_tend(ncol)
   427)   ! (ub-c) and (ub-c)**2
   428)   real(r8) :: ubmc(ncol,-ngwv:ngwv), ubmc2(ncol)
   429)   ! Temporary ubar tendencies (overall, at wave l, and at saturation).
   430)   real(r8) :: ubt(ncol,pver), ubtl(ncol), ubtlsat(ncol)
   431)   real(r8) :: wrk(ncol)
   432) 
   433)   ! Signed wave Reynolds stress.
   434)   real(r8) :: tausg(ncol)
   435) 
   436)   ! Reynolds stress for waves propagating behind and forward of the wind.
   437)   real(r8) :: taub(ncol)
   438)   real(r8) :: tauf(ncol)
   439) 
   440)   ! Polar taper.
   441)   real(r8) :: ptaper(ncol)
   442) 
   443)   ! Recalculated rhoi to preserve answers.
   444)   real(r8) :: rhoi_kludge(ncol,pver+1)
   445) 
   446)   ! LU decomposition.
   447)   type(lu_decomp) :: decomp
   448) 
   449)   !------------------------------------------------------------------------
   450) 
   451)   ! Initialize gravity wave drag tendencies to zero.
   452) 
   453)   utgw = 0._r8
   454)   vtgw = 0._r8
   455) 
   456)   taucd = 0._r8
   457) 
   458)   gwut = 0._r8
   459) 
   460)   dttke = 0._r8
   461)   ttgw = 0._r8
   462) 
   463)   ! Workaround floating point exception issues on Intel by initializing
   464)   ! everything that's first set in a where block.
   465)   dsat = 0._r8
   466)   dscal = 0._r8
   467)   mi = 0._r8
   468)   taudmp = 0._r8
   469)   tausat = 0._r8
   470)   tausg = 0._r8
   471)   ubmc = 0._r8
   472)   ubmc2 = 0._r8
   473)   wrk = 0._r8
   474) 
   475)   !------------------------------------------------------------------------
   476)   ! Compute the stress profiles and diffusivities
   477)   !------------------------------------------------------------------------
   478) 
   479)   ! Loop from bottom to top to get stress profiles.
   480)   do k = maxval(src_level)-1, ktop, -1
   481) 
   482)      ! Determine the absolute value of the saturation stress.
   483)      ! Define critical levels where the sign of (u-c) changes between
   484)      ! interfaces.
   485) 
   486)      do l = -ngwv, ngwv
   487)         where (src_level > k)
   488) 
   489)            ubmc(:,l) = ubi(:,k) - c(:,l)
   490) 
   491)            ! Test to see if u-c has the same sign here as the level below.
   492)            where (ubmc(:,l) * (ubi(:,k+1)-c(:,l)) > 0.0_r8)
   493)               tausat(:,l) = abs(effkwv * rhoi(:,k) * ubmc(:,l)**3 / &
   494)                    (2._r8*ni(:,k)))
   495)               where (tausat(:,l) <= taumin) tausat(:,l) = 0._r8
   496)            elsewhere
   497)               tausat(:,l) = 0.0_r8
   498)            end where
   499) 
   500)         end where
   501)      end do
   502) 
   503)      ! Determine the diffusivity for each column.
   504) 
   505)      d = dback
   506)      if (do_molec_diff) then
   507)         d = d + kvtt(:,k)
   508)      else
   509)         do l = -ngwv, ngwv
   510)            where (src_level > k)
   511)               dsat = (ubmc(:,l) / ni(:,k))**2 * &
   512)                    (effkwv * ubmc(:,l)**2 / (rog * ti(:,k) * ni(:,k)) - &
   513)                    alpha(k))
   514)               dscal = min(1.0_r8, tau(:,l,k+1) / (tausat(:,l)+taumin))
   515)               d = max(d, dscal * dsat)
   516)            end where
   517)         end do
   518)      end if
   519) 
   520)      ! Compute stress for each wave. The stress at this level is the min of
   521)      ! the saturation stress and the stress at the level below reduced by
   522)      ! damping. The sign of the stress must be the same as at the level
   523)      ! below.
   524) 
   525)      ! If molecular diffusion is on, only do this in levels with molecular
   526)      ! diffusion. Otherwise, do it everywhere.
   527)      if (k <= nbot_molec .or. .not. do_molec_diff) then
   528)         do l = -ngwv, ngwv
   529)            where (src_level > k)
   530) 
   531)               ubmc2 = max(ubmc(:,l)**2, ubmc2mn)
   532)               mi = ni(:,k) / (2._r8 * kwv * ubmc2) * &
   533)                    (alpha(k) + ni(:,k)**2/ubmc2 * d)

   536)               where (wrk >= -150._r8 .or. .not. do_molec_diff)
   537)                  taudmp = tau(:,l,k+1) * exp(wrk)
   538)               elsewhere
   539)                  taudmp = 0._r8
   540)               end where
   541)               where (taudmp <= taumin) taudmp = 0._r8
   542)               tau(:,l,k) = min(taudmp, tausat(:,l))
   543) 
   544)            end where
   545)         end do
   546)      else
   547)         do l = -ngwv, ngwv
   548)            where (src_level > k)
   549)               tau(:,l,k) = min(tau(:,l,k+1), tausat(:,l))
   550)            end where
   551)         end do
   552)      end if
   553) 
   554)   end do

   558)   ! conservation routine and for diagnostic output.
   559)   if ( ngwv > 0) then
   560) 
   561)      ubi_tend = (/ (ubi(i,tend_level(i)), i = 1, ncol) /)
   562) 
   563)      do k = ktop, maxval(tend_level)
   564) 
   565)         taub = 0._r8
   566)         tauf = 0._r8
   567) 
   568)         do l = -ngwv, ngwv
   569)            where (k <= tend_level)
   570) 
   571)               tausg = sign(tau(:,l,k), c(:,l)-ubi(:,k))

   575)               elsewhere (  c(:,l) > ubi_tend )

   577)               end where

   579)            end where
   580)         end do

   586)            elsewhere ( xv < 0._r8)
   587)               taucd(:,k,east) = taub * xv
   588)               taucd(:,k,west) = tauf * xv
   589)            end where

   591)            where ( yv > 0._r8)
   592)               taucd(:,k,north) = tauf * yv
   593)               taucd(:,k,south) = taub * yv
   594)            elsewhere ( yv < 0._r8)

   596)               taucd(:,k,south) = tauf * yv
   597)            end where
   598)         end where

   610)   else                  ! do not taper other cases

   612)   endif

   615)   if (tau_0_ubc) tau(:,:,0) = 0._r8

   631)            ! solution for a saturated region [proportional to (u-c)^3].

   633)                 (2._r8*rog*t(:,k)*nm(:,k))

   638)         ! 1. du/dt < |c-u|/dt  so u-c cannot change sign
   639)         !    (u^n+1 = u^n + du/dt * dt)
   640)         ! 2. du/dt < tndmax    so that ridicuously large tendencies are not
   641)         !    permitted

   668)            ! causing stress divergence in the next layer down. This
   669)            ! smoothes large stress divergences downward while conserving
   670)            ! total stress.

   672) 
   673)         end where

   678)      if (.not. orographic_only) then

   681)            vtgw(:,k) = ubt(:,k) * yv
   682)         end where

   685)            utgw(:,k) = ubt(:,k) * xv * effgw * ptaper
   686)            vtgw(:,k) = ubt(:,k) * yv * effgw * ptaper
   687)         end where
   688)      end if

   690)      ! End of level loop.
   691)   end do

   696)      ! recalculate it here to preserve answers. (Ideal gas law.)

   702)              (rair * (t(:,k) + t(:,k-1)))

   707)      ! vertical diffusion solver.

   709)           gwut, ubm, nm, rhoi_kludge, dt, gravit, pmid, rdpm, c, &
   710)           egwdffi, decomp)

   712)      ! Calculate tendency on each constituent.
   713)      do m = 1, size(q,3)
   714) 
   715)         call gw_diff_tend(ncol, pver, kbotbg, ktop, q(:,:,m), dt, &
   716)              decomp, qtgw(:,:,m))
   717) 
   718)      enddo

   721)      call gw_diff_tend(ncol, pver, kbotbg, ktop, dse, dt, decomp, dttdf)
   722)

   724)      ! energy into thermal.

   726)         do k = ktop+1, kbotbg
   727)            dttke(:,k) = dttke(:,k) + c(:,l) * gwut(:,k,l)
   728)         end do
   729)      end do

   731)      ttgw = dttke + dttdf
   732) 
   733)      ! Deallocate decomp.
   734)      call decomp%finalize()

   739)      ttgw = 0._r8
