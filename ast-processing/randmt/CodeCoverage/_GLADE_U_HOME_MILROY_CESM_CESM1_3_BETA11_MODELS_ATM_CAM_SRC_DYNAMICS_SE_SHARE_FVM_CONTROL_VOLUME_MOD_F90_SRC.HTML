    98) !SUBROUTINE fvm_MESH_ARI--------------------------------------------------CE-for FVM!
    99) ! AUTHOR: CHRISTOPH ERATH, 11.June 2011                                             !
   100) ! DESCRIPTION: Calculates the mesh for fvm, use the elem structure from HOMME       !
   101) !                                                                                   !
   102) ! CALLS: create_ari, create_interpolation_points                                    !
   103) ! INPUT: elem     ...  element structure from HOMME                                 !
   104) !        tl       ...  time level structure                                         !
   105) ! INTPUT/OUTPUT:                                                                    !
   106) !        fvm   ...  structure                                                       !
   107) !-----------------------------------------------------------------------------------!
   108) subroutine fvm_mesh_ari(elem, fvm, tl)
   109)   use time_mod, only : timelevel_t
   110)   implicit none
   111)   type (element_t), intent(in)      :: elem
   112)   type (fvm_struct), intent(inout)   :: fvm
   113)   type (timeLevel_t), intent(in)       :: tl              ! time level struct
   114)   
   115)   integer                              :: i,j
   116)   logical                              :: corner
   117)   integer                              :: mynbr_cnt, cnt, mystart, start
   118) 
   119)   fvm%faceno=elem%FaceNum
   120)   ! write the neighbors in the structure
   121)   fvm%cubeboundary=0
   122)   corner=.FALSE.
   123) 
   124) 
   125)   do j=1,8
   126)      mynbr_cnt = elem%vertex%nbrs_ptr(j+1) - elem%vertex%nbrs_ptr(j) !length of neighbor location  
   127)      mystart = elem%vertex%nbrs_ptr(j) 
   128)      !NOTE: assuming that we do not have multiple corner neighbors (so not a refined mesh)
   129)      if (mynbr_cnt > 0 ) then
   130)         fvm%nbrsface(j)=elem%vertex%nbrs_face(mystart)
   131)         ! note that if the element lies on a corner, it will be at j=5,6,7,8
   132)         if ((fvm%nbrsface(j) /= fvm%faceno) .AND. (j<5)) then
   133)            fvm%cubeboundary=j
   134)         endif
   135)     else   ! corner on the cube
   136)       if (.NOT. corner) then
   137)         fvm%nbrsface(j)=-1
   138)         fvm%cubeboundary=j
   139)         corner=.TRUE.
   140)       else
   141)         print *,'Error in fvm_CONTROL_VOLUME_MOD - Subroutine fvm_MESH_ARI: '
   142)         call abortmp('Do not allow one element per face for fvm, please increase ne!')
   143)       endif
   144)     end if
   145)   end do
   146) 
   147) 
   148)   call create_ari(elem,fvm)
   149)   call create_interpolation_points(elem,fvm)
   150) 
   151) end subroutine fvm_mesh_ari
   152) !END SUBROUTINE fvm_MESH_ARI----------------------------------------------CE-for FVM

   155) !SUBROUTINE CREATE_ARI----------------------------------------------------CE-for FVM!
   156) ! AUTHOR: CHRISTOPH ERATH, 7.November 2011                                          !
   157) ! DESCRIPTION: Calculates the mesh for fvm, including for the halo zone             !
   158) !                                                                                   !
   159) ! INPUT: elem     ...  element structure from HOMME                                 ! 
   160) ! INTPUT/OUTPUT:                                                                    !
   161) !        fvm   ...  structure                                                       !
   162) !-----------------------------------------------------------------------------------!
   163) subroutine create_ari(elem, fvm)
   164)   use coordinate_systems_mod, only : cart2spherical  
   165)   implicit none
   166)   type (element_t), intent(in)      :: elem
   167)   type (fvm_struct), intent(inout)   :: fvm
   168)   
   169)   integer                              :: i,j, jx, jy, jtmp
   170)   logical                              :: corner
   171)   integer                              :: jx_min, jx_max, jy_min, jy_max
   172)   integer                              :: jx_min1, jx_max1, jy_min1, jy_max1
   173)   integer                              :: jx_min2, jx_max2, jy_min2, jy_max2
   174)   integer                              :: invx1, invy1,invx2, invy2
   175)   logical                              :: swap1, swap2
   176)   real (kind=real_kind)                :: centerx,centery
   177)   
   178)   swap1=.FALSE.
   179)   swap2=.FALSE.
   180)   invx1=1 ! 1 order the same, -1 reorder in x direction
   181)   invy1=1 ! 1 order the same, -1 reorder in y direction
   182)   invx2=1 ! 1 order the same, -1 reorder in x direction
   183)   invy2=1 ! 1 order the same, -1 reorder in y direction
   184) ! fvm mesh is (alpha,beta) equidistant
   185)   fvm%dalpha=abs(elem%corners(1)%x-elem%corners(2)%x)/nc !in alpha 
   186)   fvm%dbeta=abs(elem%corners(1)%y-elem%corners(4)%y)/nc  !in beta
   187)   
   188)   do i=1,nc+1
   189)     fvm%acartx(i) = tan(elem%corners(1)%x+(i-1)*fvm%dalpha) 
   190)     fvm%acarty(i) = tan(elem%corners(1)%y+(i-1)*fvm%dbeta)
   191)   end do
   192) 
   193)   !
   194)   do j=1,nc
   195)     do i=1,nc
   196)       centerx = tan(elem%corners(1)%x+(i-0.5)*fvm%dalpha)  
   197)       centery = tan(elem%corners(1)%y+(j-0.5)*fvm%dbeta) 
   198)       fvm%centersphere(i,j) = &
   199)             cart2spherical(centerx,centery,fvm%faceno)
   200)            
   201)       fvm%asphere(i,j) = cart2spherical(fvm%acartx(i),fvm%acarty(j),fvm%faceno)
   202)     enddo
   203)     ! values for the nodes on the edges
   204)     fvm%asphere(nc+1,j) = cart2spherical(fvm%acartx(nc+1),fvm%acarty(j),fvm%faceno)
   205)     fvm%asphere(j,nc+1) = cart2spherical(fvm%acartx(j),fvm%acarty(nc+1),fvm%faceno)  
   206)   enddo  
   207)   ! value for the upper right corner
   208)   fvm%asphere(nc+1,nc+1) = cart2spherical(fvm%acartx(nc+1),fvm%acarty(nc+1),fvm%faceno) 
   209)   
   210)   select case (fvm%cubeboundary)
   211)   ! element is completly inside a face, note that this case will be the most one
   212)   ! INTERIOR ELEMENT
   213)     case (0) 
   214)       jx_min=1-nhe
   215)       jx_max=nc+1+nhe
   216)       jy_min=1-nhe
   217)       jy_max=nc+1+nhe
   218)       !fill in halo zone
   219)       do i=0,nhe-1
   220)         fvm%acartx(-i) = tan(elem%corners(1)%x-(i+1)*fvm%dalpha) 
   221)         fvm%acartx(nc+2+i) = tan(elem%corners(1)%x+(nc+1+i)*fvm%dalpha) 
   222)         fvm%acarty(-i) = tan(elem%corners(1)%y-(i+1)*fvm%dbeta)
   223)         fvm%acarty(nc+2+i) = tan(elem%corners(1)%y+(nc+1+i)*fvm%dbeta)
   224)       enddo
   225) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!        
   226)   !WEST EDGE ON THE CUBE SIDE
   227)     case (west)  
   228)       jx_min=1
   229)       jx_max=nc+1+nhe
   230)       jy_min=1-nhe
   231)       jy_max=nc+1+nhe
   232)       do i=0,nhe-1
   233)         fvm%acartx(nc+2+i) = tan(elem%corners(1)%x+(nc+1+i)*fvm%dalpha) 
   234)         fvm%acarty(-i) = tan(elem%corners(1)%y-(i+1)*fvm%dbeta)
   235)         fvm%acarty(nc+2+i) = tan(elem%corners(1)%y+(nc+1+i)*fvm%dbeta)
   236)       enddo
   237)       jx_min1=1-nhe
   238)       jx_max1=1
   239)       jy_min1=1-nhe
   240)       jy_max1=nc+1+nhe
   241)       ! use symmetry of the cubed-sphere on the edge
   242)       if (fvm%faceno<=4) then
   243)         fvm%acarty1(jy_min1:jy_max1)=fvm%acarty(jy_min1:jy_max1)
   244)         do jx=jx_min1,jx_max1
   245)           fvm%acartx1(jx)=-fvm%acartx(1+(jx_max1-jx))   
   246)         end do
   247)       end if
   248)       if (fvm%faceno==5) then
   249)         ! flip orientation
   250)         swap1=.TRUE.
   251)         invx1=-1
   252)         fvm%acarty1(jy_min1:jy_max1)=fvm%acarty(jy_min1:jy_max1)
   253)         do jx=jx_min1,jx_max1
   254)           fvm%acartx1(jx)=fvm%acartx(1+jx-jx_min1)   
   255)         end do
   256)       end if
   257)       ! use symmetry of the cubed-sphere on the edge
   258)       if (fvm%faceno==6) then
   259)         ! flip orientation
   260)         swap1=.TRUE.
   261)         invy1=-1
   262)         do jy=jy_min1,jy_max1
   263)           jtmp=jy_max1+jy_min1-jy
   264)           fvm%acarty1(jtmp)=-fvm%acarty(jy)
   265)         end do
   266)         do jx=jx_min1,jx_max1
   267)           fvm%acartx1(jx)=-fvm%acartx(1+jx_max1-jx)  
   268)         end do 
   269)       end if
   270)       
   271) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!        
   272)       
   273)   !EAST EDGE ON THE CUBE SIDE
   274)     case(east)
   275)       jx_min=1-nhe
   276)       jx_max=nc+1
   277)       jy_min=1-nhe
   278)       jy_max=nc+1+nhe 
   279)       do i=0,nhe-1
   280)         fvm%acartx(-i) = tan(elem%corners(1)%x-(i+1)*fvm%dalpha) 
   281)         fvm%acarty(-i) = tan(elem%corners(1)%y-(i+1)*fvm%dbeta)
   282)         fvm%acarty(nc+2+i) = tan(elem%corners(1)%y+(nc+1+i)*fvm%dbeta)
   283)       enddo
   284)       jx_min1=nc+1
   285)       jx_max1=nc+1+nhe
   286)       jy_min1=1-nhe
   287)       jy_max1=nc+1+nhe 
   288)       if (fvm%faceno<=4) then
   289)         fvm%acarty1(jy_min1:jy_max1)=fvm%acarty(jy_min1:jy_max1)
   290)         do jx=jx_min1,jx_max1
   291)           fvm%acartx1(jx)=-fvm%acartx(nc+1-(jx-jx_min1))   
   292)         end do
   293)         
   294)       end if
   295)       if (fvm%faceno==5) then
   296)         ! flip orientation
   297)         swap1=.TRUE.
   298)         invy1=-1
   299)         do jy=jy_min1,jy_max1
   300)           jtmp=jy_max1+jy_min1-jy
   301)           fvm%acarty1(jtmp)=-fvm%acarty(jy)
   302)         end do
   303)         do jx=jx_min1,jx_max1
   304)           fvm%acartx1(jx)=-fvm%acartx(nc+1-(jx-jx_min1))  
   305)         end do 
   306)       end if
   307)       ! use symmetry of the cubed-sphere on the edge
   308)       if (fvm%faceno==6) then
   309)         ! flip orientation
   310)         swap1=.TRUE.
   311)         invx1=-1
   312)         fvm%acarty1(jy_min1:jy_max1)=fvm%acarty(jy_min1:jy_max1)
   313)         do jx=jx_min1,jx_max1
   314)           fvm%acartx1(jx)=fvm%acartx(nc+1-(jx_max1-jx))  
   315)         end do 
   316)       end if
   317) 
   318)   !NORTH EDGE ON THE CUBE SIDE        
   319)     case(north)
   320)       jx_min=1-nhe
   321)       jx_max=nc+1+nhe
   322)       jy_min=1-nhe
   323)       jy_max=nc+1
   324)       do i=0,nhe-1
   325)         fvm%acartx(-i) = tan(elem%corners(1)%x-(i+1)*fvm%dalpha) 
   326)         fvm%acartx(nc+2+i) = tan(elem%corners(1)%x+(nc+1+i)*fvm%dalpha) 
   327)         fvm%acarty(-i) = tan(elem%corners(1)%y-(i+1)*fvm%dbeta)
   328)       enddo
   329)       jx_min1=1-nhe
   330)       jx_max1=nc+1+nhe
   331)       jy_min1=nc+1
   332)       jy_max1=nc+1+nhe
   333)       ! use symmetry of the cubed-sphere on the edge
   334)       if ((fvm%faceno==1) .OR. (fvm%faceno==5)) then
   335)         fvm%acartx1(jx_min1:jx_max1)=fvm%acartx(jx_min1:jx_max1)
   336)         do jy=jy_min1,jy_max1
   337)           fvm%acarty1(jy)=-fvm%acarty(nc+1-(jy-jy_min1))   
   338)         end do 
   339)       end if
   340)       ! use symmetry of the cubed-sphere on the edge
   341)       if ((fvm%faceno==3) .OR. (fvm%faceno==6)) then
   342)         invx1=-1
   343)         invy1=-1
   344)         do jx=jx_min1,jx_max1
   345)           jtmp=jx_max1+jx_min1-jx
   346)           fvm%acartx1(jtmp)=-fvm%acartx(jx)
   347)         end do
   348)         do jy=jy_min1,jy_max1
   349)           fvm%acarty1(jy)=fvm%acarty(nc+1-(jy_max1-jy))  
   350)         end do 
   351)       end if
   352)       if (fvm%faceno==2) then
   353)         ! flip orientation
   354)         swap1=.TRUE.
   355)         invy1=-1
   356)         fvm%acartx1(jx_min1:jx_max1)=fvm%acartx(jx_min1:jx_max1)
   357)         do jy=jy_min1,jy_max1
   358)           fvm%acarty1(jy)=fvm%acarty(nc+1-(jy_max1-jy))   
   359)         end do 
   360)       end if
   361)       if (fvm%faceno==4) then
   362)         ! flip orientation
   363)         swap1=.TRUE.
   364)         invx1=-1
   365)         do jy=jy_min1,jy_max1
   366)           fvm%acarty1(jy)=-fvm%acarty(nc+1-(jy-jy_min1))  
   367)         end do 
   368)         do jx=jx_min1,jx_max1
   369)           fvm%acartx1(jx)=-fvm%acartx(jx_max1-(jx-jx_min1))
   370)         end do
   371)       end if
   372)       
   373)   !SOUTH EDGE ON THE CUBE SIDE    
   374)     case(south)
   375)       jx_min=1-nhe
   376)       jx_max=nc+1+nhe
   377)       jy_min=1
   378)       jy_max=nc+1+nhe
   379)       do i=0,nhe-1
   380)         fvm%acartx(-i) = tan(elem%corners(1)%x-(i+1)*fvm%dalpha) 
   381)         fvm%acartx(nc+2+i) = tan(elem%corners(1)%x+(nc+1+i)*fvm%dalpha) 
   382)         fvm%acarty(nc+2+i) = tan(elem%corners(1)%y+(nc+1+i)*fvm%dbeta)
   383)       enddo
   384)       jx_min1=1-nhe
   385)       jx_max1=nc+1+nhe
   386)       jy_min1=1-nhe
   387)       jy_max1=1
   388)       ! use symmetry of the cubed-sphere on the edge
   389)       if ((fvm%faceno==1) .OR. (fvm%faceno==6)) then
   390)         fvm%acartx1(jx_min1:jx_max1)=fvm%acartx(jx_min1:jx_max1)
   391)         do jy=jy_min1,jy_max1
   392)           fvm%acarty1(jy)=-fvm%acarty(1+(jy_max1-jy))   
   393)         end do 
   394)       end if
   395)       if (fvm%faceno==2) then
   396)         ! flip orientation
   397)         swap1=.TRUE.
   398)         invx1=-1
   399)         do jx=jx_min1,jx_max1
   400)           fvm%acartx1(jx)=-fvm%acartx(jx_max1-(jx-jx_min1))
   401)         end do
   402)         do jy=jy_min1,jy_max1
   403)           fvm%acarty1(jy)=-fvm%acarty(1+(jy_max1-jy))   
   404)         end do 
   405)       end if
   406)       ! use symmetry of the cubed-sphere on the edge
   407)       if (fvm%faceno==3) then
   408)         invx1=-1
   409)         invy1=-1
   410)         do jx=jx_min1,jx_max1
   411)           jtmp=jx_max1+jx_min1-jx
   412)           fvm%acartx1(jtmp)=-fvm%acartx(jx)
   413)         end do
   414)         do jy=jy_min1,jy_max1
   415)           fvm%acarty1(jy)=fvm%acarty(1+(jy-jy_min1))  
   416)         end do 
   417)       end if
   418)       if (fvm%faceno==4) then
   419)         ! flip orientation
   420)         swap1=.TRUE.
   421)         invy1=-1
   422)         do jy=jy_min1,jy_max1
   423)           fvm%acarty1(jy)=fvm%acarty(1+(jy-jy_min1))   
   424)         end do 
   425)         do jx=jx_min1,jx_max1
   426)           fvm%acartx1(jx)=fvm%acartx(jx)
   427)         end do
   428)       end if
   429)       ! use symmetry of the cubed-sphere on the edge
   430)       if (fvm%faceno==5) then
