   173) /*
   174) ** GPTL_PAPIsetoption: enable or disable PAPI event defined by "counter". Called 
   175) **   from GPTLsetoption.  Since all events are off by default, val=false degenerates
   176) **   to a no-op.  Coded this way to be consistent with the rest of GPTL
   177) **
   178) ** Input args: 
   179) **   counter: PAPI counter
   180) **   val:     true or false for enable or disable
   181) **
   182) ** Return value: 0 (success) or GPTLerror (failure)
   183) */
   184) 
   185) int GPTL_PAPIsetoption (const int counter,  /* PAPI counter (or option) */
   186) 			const int val)      /* true or false for enable or disable */
   187) {
   188)   int n;       /* loop index */
   189)   int ret;     /* return code */
   190)   int numidx;  /* numerator index */
   191)   int idx;     /* derived counter index */
   192)   char eventname[PAPI_MAX_STR_LEN]; /* returned from PAPI_event_code_to_name */
   193) 
   194)   /*
   195)   ** First, check for option which is not an actual counter
   196)   */
   197) 
   198)   switch (counter) {
   199)   case GPTLverbose:
   200)   /* don't printf here--that'd duplicate what's in gptl.c */
   201)     verbose = (bool) val;
   202)     return 0;
   203)   case GPTLmultiplex:
   204)     enable_multiplexing = (bool) val;
   205)     if (verbose)
   206)       printf ("GPTL_PAPIsetoption: boolean enable_multiplexing = %d\n", val);
   207)     return 0;
   208)   case GPTLnarrowprint:
   209)     narrowprint = (bool) val;
   210)     if (verbose)
   211)       printf ("GPTL_PAPIsetoption: boolean narrowprint = %d\n", val);
   212)     return 0;
   213)   case GPTLpersec:
   214)     persec = (bool) val;
   215)     if (verbose)
   216)       printf ("GPTL_PAPIsetoption: boolean persec = %d\n", val);
   217)     return 0;
   218)   default:
   219)     break;
   220)   }
   221) 
   222)   /* 
   223)   ** If val is false, return an error if the event has already been enabled.
   224)   ** Otherwise just warn that attempting to disable a PAPI-based event
   225)   ** that has already been enabled doesn't work--for now it's just a no-op
   226)   */
   227) 
   228)   if (! val) {
   229)     if (already_enabled (counter))
   230)       return GPTLerror ("GPTL_PAPIsetoption: already enabled counter %d cannot be disabled\n",
   231) 			counter);
   232)     else
   233)       if (verbose)
   234) 	printf ("GPTL_PAPIsetoption: 'disable' %d currently is just a no-op\n", counter);
   235)     return 0;
   236)   }
   237) 
   238)   /* If the event has already been enabled for printing, exit */
   239) 
   240)   if (already_enabled (counter))
   241)     return GPTLerror ("GPTL_PAPIsetoption: counter %d has already been enabled\n", 
   242) 		      counter);
   243) 
   244)   /* 
   245)   ** Initialize PAPI if it hasn't already been done.
   246)   ** From here on down we can assume the intent is to enable (not disable) an option
   247)   */
   248) 
   249)   if (GPTL_PAPIlibraryinit () < 0)
   250)     return GPTLerror ("GPTL_PAPIsetoption: PAPI library init error\n");
   251) 
   252)   /* Ensure max nevents won't be exceeded */
   253) 
   254)   if (nevents+1 > MAX_AUX)
   255)     return GPTLerror ("GPTL_PAPIsetoption: %d is too many events. Can be increased in private.h\n",
   256) 		      nevents+1);
   257) 
   258)   /* Check derived events */
   259) 
   260)   switch (counter) {
   261)   case GPTL_IPC:
   262)     if ( ! canenable2 (PAPI_TOT_INS, PAPI_TOT_CYC))
   263)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_IPC unavailable\n");
   264) 
   265)     idx = getderivedidx (GPTL_IPC);
   266)     pr_event[nevents].event    = derivedtable[idx];
   267)     pr_event[nevents].numidx   = enable (PAPI_TOT_INS);
   268)     pr_event[nevents].denomidx = enable (PAPI_TOT_CYC);
   269)     if (verbose)
   270)       printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_TOT_INS / PAPI_TOT_CYC\n", 
   271) 	      pr_event[nevents].event.namestr);
   272)     ++nevents;
   273)     return 0;
   274)   case GPTL_CI:
   275)     idx = getderivedidx (GPTL_CI);
   276)     if (canenable2 (PAPI_FP_OPS, PAPI_LST_INS)) {
   277)       pr_event[nevents].event    = derivedtable[idx];
   278)       pr_event[nevents].numidx   = enable (PAPI_FP_OPS);
   279)       pr_event[nevents].denomidx = enable (PAPI_LST_INS);
   280)       if (verbose)
   281) 	printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_FP_OPS / PAPI_LST_INS\n", 
   282) 		pr_event[nevents].event.namestr);
   283)     } else if (canenable2 (PAPI_FP_OPS, PAPI_L1_DCA)) {
   284)       pr_event[nevents].event    = derivedtable[idx];
   285)       pr_event[nevents].numidx   = enable (PAPI_FP_OPS);
   286)       pr_event[nevents].denomidx = enable (PAPI_L1_DCA);
   287) #ifdef DEBUG
   288)       printf ("GPTL_PAPIsetoption: pr_event %d is derived and will be PAPI event %d / %d\n", 
   289) 	      nevents, pr_event[nevents].numidx, pr_event[nevents].denomidx);
   290) #endif
   291)       if (verbose)
   292) 	printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_FP_OPS / PAPI_L1_DCA\n", 
   293) 		pr_event[nevents].event.namestr);
   294)     } else {
   295)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_CI unavailable\n");
   296)     }
   297)     ++nevents;
   298)     return 0;
   299)   case GPTL_FPC:
   300)     if ( ! canenable2 (PAPI_FP_OPS, PAPI_TOT_CYC))
   301)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_FPC unavailable\n");
   302) 
   303)     idx = getderivedidx (GPTL_FPC);
   304)     pr_event[nevents].event    = derivedtable[idx];
   305)     pr_event[nevents].numidx   = enable (PAPI_FP_OPS);
   306)     pr_event[nevents].denomidx = enable (PAPI_TOT_CYC);
   307)     if (verbose)
   308)       printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_FP_OPS / PAPI_TOT_CYC\n", 
   309) 	      pr_event[nevents].event.namestr);
   310)     ++nevents;
   311)     return 0;
   312)   case GPTL_FPI:
   313)     if ( ! canenable2 (PAPI_FP_OPS, PAPI_TOT_INS))
   314)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_FPI unavailable\n");
   315) 
   316)     idx = getderivedidx (GPTL_FPI);
   317)     pr_event[nevents].event    = derivedtable[idx];
   318)     pr_event[nevents].numidx   = enable (PAPI_FP_OPS);
   319)     pr_event[nevents].denomidx = enable (PAPI_TOT_INS);
   320)     if (verbose)
   321)       printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_FP_OPS / PAPI_TOT_INS\n", 
   322) 	      pr_event[nevents].event.namestr);
   323)     ++nevents;
   324)     return 0;
   325)   case GPTL_LSTPI:
   326)     idx = getderivedidx (GPTL_LSTPI);
   327)     if (canenable2 (PAPI_LST_INS, PAPI_TOT_INS)) {
   328)       pr_event[nevents].event    = derivedtable[idx];
   329)       pr_event[nevents].numidx   = enable (PAPI_LST_INS);
   330)       pr_event[nevents].denomidx = enable (PAPI_TOT_INS);
   331)       if (verbose)
   332) 	printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_LST_INS / PAPI_TOT_INS\n", 
   333) 		pr_event[nevents].event.namestr);
   334)     } else if (canenable2 (PAPI_L1_DCA, PAPI_TOT_INS)) {
   335)       pr_event[nevents].event    = derivedtable[idx];
   336)       pr_event[nevents].numidx   = enable (PAPI_L1_DCA);
   337)       pr_event[nevents].denomidx = enable (PAPI_TOT_INS);
   338)       if (verbose)
   339) 	printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_L1_DCA / PAPI_TOT_INS\n", 
   340) 		pr_event[nevents].event.namestr);
   341)     } else {
   342)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_LSTPI unavailable\n");
   343)     }
   344)     ++nevents;
   345)     return 0;
   346)   case GPTL_DCMRT:
   347)     if ( ! canenable2 (PAPI_L1_DCM, PAPI_L1_DCA))
   348)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_DCMRT unavailable\n");
   349) 
   350)     idx = getderivedidx (GPTL_DCMRT);
   351)     pr_event[nevents].event    = derivedtable[idx];
   352)     pr_event[nevents].numidx   = enable (PAPI_L1_DCM);
   353)     pr_event[nevents].denomidx = enable (PAPI_L1_DCA);
   354)     if (verbose)
   355)       printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_L1_DCM / PAPI_L1_DCA\n", 
   356) 	      pr_event[nevents].event.namestr);
   357)     ++nevents;
   358)     return 0;
   359)   case GPTL_LSTPDCM:
   360)     idx = getderivedidx (GPTL_LSTPDCM);
   361)     if (canenable2 (PAPI_LST_INS, PAPI_L1_DCM)) {
   362)       pr_event[nevents].event    = derivedtable[idx];
   363)       pr_event[nevents].numidx   = enable (PAPI_LST_INS);
   364)       pr_event[nevents].denomidx = enable (PAPI_L1_DCM);
   365)       if (verbose)
   366) 	printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_LST_INS / PAPI_L1_DCM\n", 
   367) 		pr_event[nevents].event.namestr);
   368)     } else if (canenable2 (PAPI_L1_DCA, PAPI_L1_DCM)) {
   369)       pr_event[nevents].event    = derivedtable[idx];
   370)       pr_event[nevents].numidx   = enable (PAPI_L1_DCA);
   371)       pr_event[nevents].denomidx = enable (PAPI_L1_DCM);
   372)       if (verbose)
   373) 	printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_L1_DCA / PAPI_L1_DCM\n", 
   374) 		pr_event[nevents].event.namestr);
   375)     } else {
   376)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_LSTPDCM unavailable\n");
   377)     }
   378)     ++nevents;
   379)     return 0;
   380)     /*
   381)     ** For L2 counts, use TC* instead of DC* to avoid PAPI derived events
   382)     */
   383)   case GPTL_L2MRT:
   384)     if ( ! canenable2 (PAPI_L2_TCM, PAPI_L2_TCA))
   385)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_L2MRT unavailable\n");
   386) 
   387)     idx = getderivedidx (GPTL_L2MRT);
   388)     pr_event[nevents].event    = derivedtable[idx];
   389)     pr_event[nevents].numidx   = enable (PAPI_L2_TCM);
   390)     pr_event[nevents].denomidx = enable (PAPI_L2_TCA);
   391)     if (verbose)
   392)       printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_L2_TCM / PAPI_L2_TCA\n", 
   393) 	      pr_event[nevents].event.namestr);
   394)     ++nevents;
   395)     return 0;
   396)   case GPTL_LSTPL2M:
   397)     idx = getderivedidx (GPTL_LSTPL2M);
   398)     if (canenable2 (PAPI_LST_INS, PAPI_L2_TCM)) {
   399)       pr_event[nevents].event    = derivedtable[idx];
   400)       pr_event[nevents].numidx   = enable (PAPI_LST_INS);
   401)       pr_event[nevents].denomidx = enable (PAPI_L2_TCM);
   402)       if (verbose)
   403) 	printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_LST_INS / PAPI_L2_TCM\n", 
   404) 		pr_event[nevents].event.namestr);
   405)     } else if (canenable2 (PAPI_L1_DCA, PAPI_L2_TCM)) {
   406)       pr_event[nevents].event    = derivedtable[idx];
   407)       pr_event[nevents].numidx   = enable (PAPI_L1_DCA);
   408)       pr_event[nevents].denomidx = enable (PAPI_L2_TCM);
   409)       if (verbose)
   410) 	printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_L1_DCA / PAPI_L2_TCM\n", 
   411) 		pr_event[nevents].event.namestr);
   412)     } else {
   413)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_LSTPL2M unavailable\n");
   414)     }
   415)     ++nevents;
   416)     return 0;
   417)   case GPTL_L3MRT:
   418)     if ( ! canenable2 (PAPI_L3_TCM, PAPI_L3_TCR))
   419)       return GPTLerror ("GPTL_PAPIsetoption: GPTL_L3MRT unavailable\n");
   420) 
   421)     idx = getderivedidx (GPTL_L3MRT);
   422)     pr_event[nevents].event    = derivedtable[idx];
   423)     pr_event[nevents].numidx   = enable (PAPI_L3_TCM);
   424)     pr_event[nevents].denomidx = enable (PAPI_L3_TCR);
   425)     if (verbose)
   426)       printf ("GPTL_PAPIsetoption: enabling derived event %s = PAPI_L3_TCM / PAPI_L3_TCR\n", 
   427) 	      pr_event[nevents].event.namestr);
   428)     ++nevents;
   429)     return 0;
   430)   default:
   431)     break;
   432)   }
   433) 
   434)   /* Check PAPI presets */
   435) 
   436)   for (n = 0; n < npapientries; n++) {
   437)     if (counter == papitable[n].counter) {
   438)       if ((numidx = papievent_is_enabled (counter)) >= 0) {
   439) 	pr_event[nevents].event  = papitable[n];
   440) 	pr_event[nevents].numidx = numidx;
   441) 	pr_event[nevents].denomidx = -1;     /* flag says not derived (no denominator) */
   442)       } else if (canenable (counter)) {
   443) 	pr_event[nevents].event  = papitable[n];
   444) 	pr_event[nevents].numidx = enable (counter);
   445) 	pr_event[nevents].denomidx = -1;     /* flag says not derived (no denominator) */
   446)       } else {
   447) 	return GPTLerror ("GPTL_PAPIsetoption: Can't enable event \n", 
   448) 			  papitable[n].longstr);
   449)       }
   450)       if (verbose)
   451) 	printf ("GPTL_PAPIsetoption: enabling PAPI preset event %s\n", 
   452) 		pr_event[nevents].event.namestr);
   453)       ++nevents;
   454)       return 0;
   455)     }
   456)   }
   457) 
   458)   /*
   459)   ** Check native events last: If PAPI_event_code_to_name fails, give up
   460)   */
   461)   
   462)   if ((ret = PAPI_event_code_to_name (counter, eventname)) != PAPI_OK)
   463)     return GPTLerror ("GPTL_PAPIsetoption: name not found for counter %d: PAPI_strerror: %s\n", 
   464) 		      counter, PAPI_strerror (ret));
   465) 
   466)   /*
   467)   ** A table with predefined names of various lengths does not exist for
   468)   ** native events. Just truncate eventname.
   469)   */
   470) 
   471)   if ((numidx = papievent_is_enabled (counter)) >= 0) {
   472)     pr_event[nevents].event.counter = counter;
   473) 
   474)     pr_event[nevents].event.namestr = (char *) GPTLallocate (12+1);
   475)     strncpy (pr_event[nevents].event.namestr, eventname, 12);
   476)     pr_event[nevents].event.namestr[12] = '\0';
   477) 
   478)     pr_event[nevents].event.str16 = (char *) GPTLallocate (16+1);
   479)     strncpy (pr_event[nevents].event.str16, eventname, 16);
   480)     pr_event[nevents].event.str16[16] = '\0';
   481) 
   482)     pr_event[nevents].event.longstr = (char *) GPTLallocate (PAPI_MAX_STR_LEN);
   483)     strncpy (pr_event[nevents].event.longstr, eventname, PAPI_MAX_STR_LEN);
   484) 
   485)     pr_event[nevents].numidx = numidx;
   486)     pr_event[nevents].denomidx = -1;     /* flag says not derived (no denominator) */
   487)   } else if (canenable (counter)) {
   488)     pr_event[nevents].event.counter = counter;
   489) 
   490)     pr_event[nevents].event.namestr = (char *) GPTLallocate (12+1);
   491)     strncpy (pr_event[nevents].event.namestr, eventname, 12);
   492)     pr_event[nevents].event.namestr[12] = '\0';
   493) 
   494)     pr_event[nevents].event.str16 = (char *) GPTLallocate (16+1);
   495)     strncpy (pr_event[nevents].event.str16, eventname, 16);
   496)     pr_event[nevents].event.str16[16] = '\0';
   497) 
   498)     pr_event[nevents].event.longstr = (char *) GPTLallocate (PAPI_MAX_STR_LEN);
   499)     strncpy (pr_event[nevents].event.longstr, eventname, PAPI_MAX_STR_LEN);
   500) 
   501)     pr_event[nevents].numidx = enable (counter);
   502)     pr_event[nevents].denomidx = -1;     /* flag says not derived (no denominator) */
   503)   } else {
   504)     return GPTLerror ("GPTL_PAPIsetoption: Can't enable event %s\n", eventname);
   505)   }
   506) 
   507)   if (verbose)
   508)     printf ("GPTL_PAPIsetoption: enabling native event %s\n", pr_event[nevents].event.longstr);
   509) 
   510)   ++nevents;
   511)   return 0;
   512) }

   514) /*
   515) ** canenable: determine whether a PAPI counter can be enabled
   516) **
   517) ** Input args: 
   518) **   counter: PAPI counter
   519) **
   520) ** Return value: 0 (success) or non-zero (failure)
   521) */
   522)  
   523) int canenable (int counter)
   524) {
   525)   char eventname[PAPI_MAX_STR_LEN]; /* returned from PAPI_event_code_to_name */
   526) 
   527)   if (npapievents+1 > MAX_AUX)
   528)     return false;
   529) 
   530)   if (PAPI_query_event (counter) != PAPI_OK) {
   531)     (void) PAPI_event_code_to_name (counter, eventname);

   533)     return false;
   534)   }
   535) 
   536)   return true;
   537) }

   558)     return false;

   560)

   565) 
   566)   return true;
   567) }

   569) /*
   570) ** papievent_is_enabled: determine whether a PAPI counter has already been
   571) **   enabled. Used internally to keep track of PAPI counters enabled. A given
   572) **   PAPI counter may occur in the computation of multiple derived events, as
   573) **   well as output directly. E.g. PAPI_FP_OPS is used to compute
   574) **   computational intensity, and floating point ops per instruction.

   576) ** Input args: 
   577) **   counter: PAPI counter
   578) **
   579) ** Return value: index into papieventlist (success) or negative (not found)
   580) */
   581)  
   582) int papievent_is_enabled (int counter)
   583) {
   584)   int n;
   585) 
   586)   for (n = 0; n < npapievents; ++n)
   587)     if (papieventlist[n] == counter)
   588)       return n;
   589)   return -1;
   590) }

   592) /*
   593) ** already_enabled: determine whether a PAPI-based event has already been
   594) **   enabled for printing. 
   595) **
   596) ** Input args: 
   597) **   counter: PAPI or derived counter
   598) **
   599) ** Return value: 1 (true) or 0 (false)
   600) */
   601)  
   602) int already_enabled (int counter)
   603) {
   604)   int n;
   605) 
   606)   for (n = 0; n < nevents; ++n)
   607)     if (pr_event[n].event.counter == counter)
   608)       return 1;
   609)   return 0;
   610) }

   612) /*
   613) ** enable: enable a PAPI event. ASSUMES that canenable() has already determined
   614) **   that the event can be enabled.
   615) **
   616) ** Input args: 
   617) **   counter: PAPI counter
   618) **
   619) ** Return value: index into papieventlist
   620) */
   621)  
   622) int enable (int counter)
   623) {
   624)   int n;
   625) 
   626)   /* If the event is already enabled, return its index */
   627) 
   628)   for (n = 0; n < npapievents; ++n) {
   629)     if (papieventlist[n] == counter) {
   630) #ifdef DEBUG
   631)       printf ("enable: PAPI event %d is %d\n", n, counter);
   632) #endif
   633)       return n;
   634)     }
   635)   }
   636) 
   637)   /* New event */
   638) 
   639)   papieventlist[npapievents++] = counter;
   640)   return npapievents-1;
   641) }

   644) ** getderivedidx: find the table index of a derived counter
   645) **
   646) ** Input args: 
   647) **   counter: derived counter

   651)

   653) {

   655)

   658)       return n;
   659)   }

   663) /*
   664) ** GPTL_PAPIlibraryinit: Call PAPI_library_init if necessary
   665) **
   666) ** Return value: 0 (success) or GPTLerror (failure)
   667) */
   668) 
   669) int GPTL_PAPIlibraryinit ()
   670) {
   671)   int ret;
   672) 
   673)   if ((ret = PAPI_is_initialized ()) == PAPI_NOT_INITED) {
   674)     if ((ret = PAPI_library_init (PAPI_VER_CURRENT)) != PAPI_VER_CURRENT) {
   675)       fprintf (stderr, "GPTL_PAPIlibraryinit: ret=%d PAPI_VER_CURRENT=%d\n", 
   676) 	       ret, (int) PAPI_VER_CURRENT);
   677)       return GPTLerror ("GPTL_PAPIlibraryinit: PAPI_library_init failure:%s\n",
   678) 			PAPI_strerror (ret));

   681)   return 0;
   682) }

   684) /*
   685) ** GPTL_PAPIinitialize(): Initialize the PAPI interface. Called from GPTLinitialize.
   686) **   PAPI_library_init must be called before any other PAPI routines.  
   687) **   PAPI_thread_init is called subsequently if threading is enabled.
   688) **   Finally, allocate space for PAPI counters and start them.
   689) **
   690) ** Input args: 
   691) **   maxthreads: number of threads
   692) **
   693) ** Return value: 0 (success) or GPTLerror or -1 (failure)
   694) */
   695)  
   696) int GPTL_PAPIinitialize (const int maxthreads,     /* number of threads */
   697) 			 const bool verbose_flag,  /* output verbosity */
   698) 			 int *nevents_out,         /* nevents needed by gptl.c */
   699) 			 Entry *pr_event_out)      /* events needed by gptl.c */
   700) {
   701)   int ret;       /* return code */
   702)   int n;         /* loop index */
   703)   int t;         /* thread index */
   704) 
   705)   verbose = verbose_flag;
   706) 
   707)   if (maxthreads < 1)
   708)     return GPTLerror ("GPTL_PAPIinitialize: maxthreads = %d\n", maxthreads);
   709) 
   710)   /* Ensure that PAPI_library_init has already been called */
   711) 
   712)   if ((ret = GPTL_PAPIlibraryinit ()) < 0)
   713)     return GPTLerror ("GPTL_PAPIinitialize: GPTL_PAPIlibraryinit failure\n");
   714) 
   715)   /* PAPI_thread_init needs to be called if threading enabled */
   716) 
   717) #if ( defined THREADED_OMP )
   718)   if (PAPI_thread_init ((unsigned long (*)(void)) (omp_get_thread_num)) != PAPI_OK)
   719)     return GPTLerror ("GPTL_PAPIinitialize: PAPI_thread_init failure\n");
   720) #elif ( defined THREADED_PTHREADS )
   721)   if (PAPI_thread_init ((unsigned long (*)(void)) (pthread_self)) != PAPI_OK)
   722)     return GPTLerror ("GPTL_PAPIinitialize: PAPI_thread_init failure\n");
   723) #endif
   724) 
   725)   /* allocate and initialize static local space */
   726) 
   727)   EventSet     = (int *)        GPTLallocate (maxthreads * sizeof (int));
   728)   papicounters = (long_long **) GPTLallocate (maxthreads * sizeof (long_long *));
   729) 
   730)   for (t = 0; t < maxthreads; t++) {
   731)     EventSet[t] = PAPI_NULL;
   732)     papicounters[t] = (long_long *) GPTLallocate (MAX_AUX * sizeof (long_long));
   733)   }
   734) 
   735)   *nevents_out = nevents;
   736)   for (n = 0; n < nevents; ++n) {
   737)     pr_event_out[n].counter = pr_event[n].event.counter;
   738)     pr_event_out[n].namestr = pr_event[n].event.namestr;
   739)     pr_event_out[n].str8    = pr_event[n].event.str8;
   740)     pr_event_out[n].str16   = pr_event[n].event.str16;
   741)     pr_event_out[n].longstr = pr_event[n].event.longstr;
   742)   }
   743)   return 0;
   744) }

   746) /*
   747) ** GPTLcreate_and_start_events: Create and start the PAPI eventset.
   748) **   Threaded routine to create the "event set" (PAPI terminology) and start
   749) **   the counters. This is only done once, and is called from get_thread_num
   750) **   for the first time for the thread.
   751) ** 
   752) ** Input args: 
   753) **   t: thread number
   754) **
   755) ** Return value: 0 (success) or GPTLerror (failure)
   756) */
   757) 
   758) int GPTLcreate_and_start_events (const int t)  /* thread number */
   759) {
   760)   int ret; /* return code */
   761)   int n;   /* loop index over events */
   762)   char eventname[PAPI_MAX_STR_LEN]; /* returned from PAPI_event_code_to_name */
   763) 
   764)   /* Create the event set */
   765) 
   766)   if ((ret = PAPI_create_eventset (&EventSet[t])) != PAPI_OK)
   767)     return GPTLerror ("GPTLcreate_and_start_events: thread %d failure creating eventset: %s\n", 
   768) 		      t, PAPI_strerror (ret));
   769) 
   770)   if (verbose)
   771)     printf ("GPTLcreate_and_start_events: successfully created eventset for thread %d\n", t);
   772) 
   773)   /* Add requested events to the event set */
   774) 
   775)   for (n = 0; n < npapievents; n++) {
   776)     if ((ret = PAPI_add_event (EventSet[t], papieventlist[n])) != PAPI_OK) {
   777)       if (verbose) {
   778) 	fprintf (stderr, "%s\n", PAPI_strerror (ret));
   779) 	ret = PAPI_event_code_to_name (papieventlist[n], eventname);
   780) 	fprintf (stderr, "GPTLcreate_and_start_events: failure adding event:%s\n",
   781) 		 eventname);
   782)       }
   783) 
   784)       if (enable_multiplexing) {
   785)         if (verbose)
   786) 	  printf ("Trying multiplexing...\n");
   787) 	is_multiplexed = true;
   788) 	break;
   789)       } else
