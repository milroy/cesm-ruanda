   140) !-----------------------------------------------------------------------

   142)   subroutine SETSOX( &
   143)        ncol,   &
   144)        lchnk,  &
   145)        loffset,&
   146)        dtime,  &
   147)        press,  &
   148)        pdel,   &
   149)        tfld,   &
   150)        mbar,   &
   151)        lwc,    &
   152)        cldfrc, &
   153)        cldnum, &
   154)        xhnm,   &
   155)        invariants, &
   156)        qcw,    &
   157)        qin     &
   158)        )
   159) 
   160)     !-----------------------------------------------------------------------      
   161)     !          ... Compute heterogeneous reactions of SOX
   162)     !
   163)     !       (0) using initial PH to calculate PH
   164)     !           (a) HENRYs law constants
   165)     !           (b) PARTIONING
   166)     !           (c) PH values
   167)     !
   168)     !       (1) using new PH to repeat
   169)     !           (a) HENRYs law constants
   170)     !           (b) PARTIONING
   171)     !           (c) REACTION rates
   172)     !           (d) PREDICTION
   173)     !-----------------------------------------------------------------------      
   174)     !
   175)     use ppgrid,    only : pcols, pver
   176)     use chem_mods, only : gas_pcnst, nfs
   177)     use chem_mods,    only : adv_mass
   178)     use physconst,    only : mwdry, gravit
   179)     use mo_constants, only : pi
   180)     use cam_history,  only : outfld
   181)     use sox_cldaero_mod, only : sox_cldaero_update, sox_cldaero_create_obj, sox_cldaero_destroy_obj
   182)     use cldaero_mod,     only : cldaero_conc_t
   183) 
   184)     !
   185)     implicit none
   186)     !
   187)     !-----------------------------------------------------------------------      
   188)     !      ... Dummy arguments
   189)     !-----------------------------------------------------------------------      
   190)     integer,          intent(in)    :: ncol              ! num of columns in chunk
   191)     integer,          intent(in)    :: lchnk             ! chunk id
   192)     integer,          intent(in)    :: loffset           ! offset of chem tracers in the advected tracers array
   193)     real(r8),         intent(in)    :: dtime             ! time step (sec)
   194)     real(r8),         intent(in)    :: press(:,:)        ! midpoint pressure ( Pa )
   195)     real(r8),         intent(in)    :: pdel(:,:)         ! pressure thickness of levels (Pa)
   196)     real(r8),         intent(in)    :: tfld(:,:)         ! temperature
   197)     real(r8),         intent(in)    :: mbar(:,:)         ! mean wet atmospheric mass ( amu )
   198)     real(r8), target, intent(in)    :: lwc(:,:)          ! cloud liquid water content (kg/kg)
   199)     real(r8), target, intent(in)    :: cldfrc(:,:)       ! cloud fraction
   200)     real(r8),         intent(in)    :: cldnum(:,:)       ! droplet number concentration (#/kg)
   201)     real(r8),         intent(in)    :: xhnm(:,:)         ! total atms density ( /cm**3)
   202)     real(r8),         intent(in)    :: invariants(:,:,:)
   203)     real(r8), target, intent(inout) :: qcw(:,:,:)        ! cloud-borne aerosol (vmr)
   204)     real(r8),         intent(inout) :: qin(:,:,:)        ! transported species ( vmr )
   205) 
   206)     !-----------------------------------------------------------------------      
   207)     !      ... Local variables
   208)     !
   209)     !           xhno3 ... in mixing ratio
   210)     !-----------------------------------------------------------------------      
   211)     integer,  parameter :: itermax = 20
   212)     real(r8), parameter :: ph0 = 5.0_r8  ! INITIAL PH VALUES
   213)     real(r8), parameter :: const0 = 1.e3_r8/6.023e23_r8
   214)     real(r8), parameter :: xa0 = 11._r8
   215)     real(r8), parameter :: xb0 = -.1_r8
   216)     real(r8), parameter :: xa1 = 1.053_r8
   217)     real(r8), parameter :: xb1 = -4.368_r8
   218)     real(r8), parameter :: xa2 = 1.016_r8
   219)     real(r8), parameter :: xb2 = -2.54_r8
   220)     real(r8), parameter :: xa3 = .816e-32_r8
   221)     real(r8), parameter :: xb3 = .259_r8
   222) 
   223)     real(r8), parameter :: kh0 = 9.e3_r8            ! HO2(g)          -> Ho2(a)
   224)     real(r8), parameter :: kh1 = 2.05e-5_r8         ! HO2(a)          -> H+ + O2-
   225)     real(r8), parameter :: kh2 = 8.6e5_r8           ! HO2(a) + ho2(a) -> h2o2(a) + o2
   226)     real(r8), parameter :: kh3 = 1.e8_r8            ! HO2(a) + o2-    -> h2o2(a) + o2
   227)     real(r8), parameter :: Ra = 8314._r8/101325._r8 ! universal constant   (atm)/(M-K)
   228)     real(r8), parameter :: xkw = 1.e-14_r8          ! water acidity
   229) 
   230)     !
   231)     real(r8) :: xdelso4hp(ncol,pver)
   232)     real(r8) :: xphlwc(ncol,pver)
   233) 
   234)     integer  :: k, i, iter, file
   235)     real(r8) :: wrk, delta
   236)     real(r8) :: xph0, aden, xk, xe, x2
   237)     real(r8) :: tz, xl, px, qz, pz, es, qs, patm
   238)     real(r8) :: Eso2, Eso4, Ehno3, Eco2, Eh2o, Enh3
   239)     real(r8) :: so2g, h2o2g, co2g, o3g
   240)     real(r8) :: hno3a, nh3a, so2a, h2o2a, co2a, o3a
   241)     real(r8) :: rah2o2, rao3, pso4, ccc
   242)     real(r8) :: cnh3, chno3, com, com1, com2, xra
   243) 
   244)     real(r8) :: hno3g(ncol,pver), nh3g(ncol,pver)
   245)     !
   246)     !-----------------------------------------------------------------------      
   247)     !            for Ho2(g) -> H2o2(a) formation 
   248)     !            schwartz JGR, 1984, 11589
   249)     !-----------------------------------------------------------------------      
   250)     real(r8) :: kh4    ! kh2+kh3
   251)     real(r8) :: xam    ! air density /cm3
   252)     real(r8) :: ho2s   ! ho2s = ho2(a)+o2-
   253)     real(r8) :: r1h2o2 ! prod(h2o2) by ho2 in mole/L(w)/s
   254)     real(r8) :: r2h2o2 ! prod(h2o2) by ho2 in mix/s
   255) 
   256)     real(r8), dimension(ncol,pver)  ::             &
   257)          xhno3, xh2o2, xso2, xso4, xno3, &
   258)          xnh3, xnh4, xo3,         &
   259)          cfact, &
   260)          xph, xho2,         &
   261)          xh2so4, xmsa, xso4_init, &
   262)          hehno3, &            ! henry law const for hno3
   263)          heh2o2, &            ! henry law const for h2o2
   264)          heso2,  &            ! henry law const for so2
   265)          henh3,  &            ! henry law const for nh3
   266)          heo3              !!,   &            ! henry law const for o3
   267) 
   268)     real(r8) :: patm_x
   269) 
   270)     real(r8), dimension(ncol)  :: work1
   271)     logical :: converged
   272) 
   273)     real(r8), pointer :: xso4c(:,:)
   274)     real(r8), pointer :: xnh4c(:,:)
   275)     real(r8), pointer :: xno3c(:,:)
   276)     type(cldaero_conc_t), pointer :: cldconc
   277) 
   278)     real(r8) :: fact1_hno3, fact2_hno3, fact3_hno3
   279)     real(r8) :: fact1_so2, fact2_so2, fact3_so2, fact4_so2
   280)     real(r8) :: fact1_nh3, fact2_nh3, fact3_nh3
   281)     real(r8) :: tmp_hp, tmp_hso3, tmp_hco3, tmp_nh4, tmp_no3
   282)     real(r8) :: tmp_oh, tmp_so3, tmp_so4
   283)     real(r8) :: tmp_neg, tmp_pos
   284)     real(r8) :: yph, yph_lo, yph_hi
   285)     real(r8) :: ynetpos, ynetpos_lo, ynetpos_hi
   286) 
   287)     !-----------------------------------------------------------------
   288)     !       ... NOTE: The press array is in pascals and must be
   289)     !                 mutiplied by 10 to yield dynes/cm**2.
   290)     !-----------------------------------------------------------------
   291)     !==================================================================
   292)     !       ... First set the PH
   293)     !==================================================================
   294)     !      ... Initial values
   295)     !           The values of so2, so4 are after (1) SLT, and CHEM
   296)     !-----------------------------------------------------------------
   297)     xph0 = 10._r8**(-ph0)                      ! initial PH value
   298) 
   299)     do k = 1,pver
   300)        cfact(:,k) = xhnm(:,k)     &          ! /cm3(a)  
   301)             * 1.e6_r8             &          ! /m3(a)
   302)             * 1.38e-23_r8/287._r8 &          ! Kg(a)/m3(a)
   303)             * 1.e-3_r8                       ! Kg(a)/L(a)
   304)     end do
   305) 
   306)     cldconc => sox_cldaero_create_obj( cldfrc,qcw,lwc, cfact, ncol, loffset )
   307)     xso4c => cldconc%so4c
   308)     xnh4c => cldconc%nh4c
   309)     xno3c => cldconc%no3c
   310) 
   311)     xso4(:,:) = 0._r8
   312)     xno3(:,:) = 0._r8
   313)     xnh4(:,:) = 0._r8
   314) 
   315)     do k = 1,pver
   316)        xph(:,k) = xph0                                ! initial PH value
   317) 
   318)        if ( inv_so2 ) then
   319)           xso2 (:,k) = invariants(:,k,id_so2)/xhnm(:,k)  ! mixing ratio
   320)        else
   321)           xso2 (:,k) = qin(:,k,id_so2)                   ! mixing ratio
   322)        endif
   323) 
   324)        if (id_hno3 > 0) then
   325)           xhno3(:,k) = qin(:,k,id_hno3)
   326)        else
   327)           xhno3(:,k) = 0.0_r8
   328)        endif
   329) 
   330)        if ( inv_h2o2 ) then
   331)           xh2o2 (:,k) = invariants(:,k,id_h2o2)/xhnm(:,k)  ! mixing ratio
   332)        else
   333)           xh2o2 (:,k) = qin(:,k,id_h2o2)                   ! mixing ratio
   334)        endif
   335) 
   336)        if (id_nh3  > 0) then
   337)           xnh3 (:,k) = qin(:,k,id_nh3)
   338)        else
   339)           xnh3 (:,k) = 0.0_r8
   340)        endif
   341) 
   342)        if ( inv_o3 ) then
   343)           xo3  (:,k) = invariants(:,k,id_o3)/xhnm(:,k) ! mixing ratio
   344)        else
   345)           xo3  (:,k) = qin(:,k,id_o3)                  ! mixing ratio
   346)        endif
   347)        if ( inv_ho2 ) then
   348)           xho2 (:,k) = invariants(:,k,id_ho2)/xhnm(:,k)! mixing ratio
   349)        else
   350)           xho2 (:,k) = qin(:,k,id_ho2)                 ! mixing ratio
   351)        endif
   352) 
   353)        if (cloud_borne) then
   354)           xh2so4(:,k) = qin(:,k,id_h2so4)
   355)        else
   356)           xso4  (:,k) = qin(:,k,id_so4) ! mixing ratio
   357)        endif
   358)        if (id_msa > 0) xmsa (:,k) = qin(:,k,id_msa)
   359) 
   360)     end do
   361)     
   362)     !-----------------------------------------------------------------
   363)     !       ... Temperature dependent Henry constants
   364)     !-----------------------------------------------------------------
   365)     ver_loop0: do k = 1,pver                               !! pver loop for STEP 0
   366)        col_loop0: do i = 1,ncol
   367)           
   368)           if (cloud_borne .and. cldfrc(i,k)>0._r8) then
   369)              xso4(i,k) = xso4c(i,k) / cldfrc(i,k)
   370)              xnh4(i,k) = xnh4c(i,k) / cldfrc(i,k)
   371)              xno3(i,k) = xno3c(i,k) / cldfrc(i,k)
   372)           endif
   373)           xl = cldconc%xlwc(i,k)
   374) 
   375)           if( xl >= 1.e-8_r8 ) then
   376)              work1(i) = 1._r8 / tfld(i,k) - 1._r8 / 298._r8
   377) 
   378)              !-----------------------------------------------------------------
   379)              ! 21-mar-2011 changes by rce
   380)              ! ph calculation now uses bisection method to solve the electro-neutrality equation
   381)              ! 3-mode aerosols (where so4 is assumed to be nh4hso4)
   382)              !    old code set xnh4c = so4c
   383)              !    new code sets xnh4c = 0, then uses a -1 charge (instead of -2)
   384)              !       for so4 when solving the electro-neutrality equation
   385)              !-----------------------------------------------------------------
   386) 
   387)              !-----------------------------------------------------------------
   388)              !  calculations done before iterating
   389)              !-----------------------------------------------------------------
   390) 
   391)              !-----------------------------------------------------------------
   392)              pz = .01_r8*press(i,k)       !! pressure in mb
   393)              tz = tfld(i,k)
   394)              patm = pz/1013._r8
   395)              xam  = press(i,k)/(1.38e-23_r8*tz)  !air density /M3
   396) 
   397)              !-----------------------------------------------------------------
   398)              !        ... hno3
   399)              !-----------------------------------------------------------------
   400)              ! previous code
   401)              !    hehno3(i,k)  = xk*(1._r8 + xe/xph(i,k))
   402)              !    px = hehno3(i,k) * Ra * tz * xl
   403)              !    hno3g = xhno3(i,k)/(1._r8 + px)
   404)              !    Ehno3 = xk*xe*hno3g *patm
   405)              ! equivalent new code
   406)              !    hehno3 = xk + xk*xe/hplus
   407)              !    hno3g = xhno3/(1 + px)
   408)              !          = xhno3/(1 + hehno3*ra*tz*xl)
   409)              !          = xhno3/(1 + xk*ra*tz*xl*(1 + xe/hplus)
   410)              !    ehno3 = hno3g*xk*xe*patm
   411)              !          = xk*xe*patm*xhno3/(1 + xk*ra*tz*xl*(1 + xe/hplus)
   412)              !          = ( fact1_hno3    )/(1 + fact2_hno3 *(1 + fact3_hno3/hplus)
   413)              !    [hno3-] = ehno3/hplus
   414)              xk = 2.1e5_r8 *EXP( 8700._r8*work1(i) )
   415)              xe = 15.4_r8
   416)              fact1_hno3 = xk*xe*patm*xhno3(i,k)
   417)              fact2_hno3 = xk*ra*tz*xl
   418)              fact3_hno3 = xe
   419) 
   420)              !-----------------------------------------------------------------
   421)              !          ... so2
   422)              !-----------------------------------------------------------------
   423)              ! previous code
   424)              !    heso2(i,k)  = xk*(1._r8 + wrk*(1._r8 + x2/xph(i,k)))
   425)              !    px = heso2(i,k) * Ra * tz * xl
   426)              !    so2g =  xso2(i,k)/(1._r8+ px)
   427)              !    Eso2 = xk*xe*so2g *patm
   428)              ! equivalent new code
   429)              !    heso2 = xk + xk*xe/hplus * xk*xe*x2/hplus**2
   430)              !    so2g = xso2/(1 + px)
   431)              !         = xso2/(1 + heso2*ra*tz*xl)
   432)              !         = xso2/(1 + xk*ra*tz*xl*(1 + (xe/hplus)*(1 + x2/hplus))
   433)              !    eso2 = so2g*xk*xe*patm
   434)              !          = xk*xe*patm*xso2/(1 + xk*ra*tz*xl*(1 + (xe/hplus)*(1 + x2/hplus))
   435)              !          = ( fact1_so2    )/(1 + fact2_so2 *(1 + (fact3_so2/hplus)*(1 + fact4_so2/hplus)
   436)              !    [hso3-] + 2*[so3--] = (eso2/hplus)*(1 + 2*x2/hplus)
   437)              xk = 1.23_r8  *EXP( 3120._r8*work1(i) )
   438)              xe = 1.7e-2_r8*EXP( 2090._r8*work1(i) )
   439)              x2 = 6.0e-8_r8*EXP( 1120._r8*work1(i) )
   440)              fact1_so2 = xk*xe*patm*xso2(i,k)
   441)              fact2_so2 = xk*ra*tz*xl
   442)              fact3_so2 = xe
   443)              fact4_so2 = x2
   444) 
   445)              !-----------------------------------------------------------------
   446)              !          ... nh3
   447)              !-----------------------------------------------------------------
   448)              ! previous code
   449)              !    henh3(i,k)  = xk*(1._r8 + xe*xph(i,k)/xkw)
   450)              !    px = henh3(i,k) * Ra * tz * xl
   451)              !    nh3g = (xnh3(i,k)+xnh4(i,k))/(1._r8+ px)
   452)              !    Enh3 = xk*xe*nh3g/xkw *patm
   453)              ! equivalent new code
   454)              !    henh3 = xk + xk*xe*hplus/xkw
   455)              !    nh3g = xnh34/(1 + px)
   456)              !         = xnh34/(1 + henh3*ra*tz*xl)
   457)              !         = xnh34/(1 + xk*ra*tz*xl*(1 + xe*hplus/xkw)
   458)              !    enh3 = nh3g*xk*xe*patm/xkw
   459)              !          = ((xk*xe*patm/xkw)*xnh34)/(1 + xk*ra*tz*xl*(1 + xe*hplus/xkw)
   460)              !          = ( fact1_nh3            )/(1 + fact2_nh3  *(1 + fact3_nh3*hplus)
   461)              !    [nh4+] = enh3*hplus
   462)              xk = 58._r8   *EXP( 4085._r8*work1(i) )
   463)              xe = 1.7e-5_r8*EXP( -4325._r8*work1(i) )
   464) 
   465)              fact1_nh3 = (xk*xe*patm/xkw)*(xnh3(i,k)+xnh4(i,k))
   466)              fact2_nh3 = xk*ra*tz*xl
   467)              fact3_nh3 = xe/xkw
   468) 
   469)              !-----------------------------------------------------------------
   470)              !        ... h2o effects
   471)              !-----------------------------------------------------------------
   472)              Eh2o = xkw
   473) 
   474)              !-----------------------------------------------------------------
   475)              !        ... co2 effects
   476)              !-----------------------------------------------------------------
   477)              co2g = 330.e-6_r8                            !330 ppm = 330.e-6 atm
   478)              xk = 3.1e-2_r8*EXP( 2423._r8*work1(i) )
   479)              xe = 4.3e-7_r8*EXP(-913._r8 *work1(i) )
   480)              Eco2 = xk*xe*co2g  *patm
   481) 
   482)              !-----------------------------------------------------------------
   483)              !         ... so4 effect
   484)              !-----------------------------------------------------------------
   485)              Eso4 = xso4(i,k)*xhnm(i,k)   &         ! /cm3(a)
   486)                   *const0/xl
   487) 
   488) 
   489)              !-----------------------------------------------------------------
   490)              ! now use bisection method to solve electro-neutrality equation
   491)              !
   492)              ! during the iteration loop,
   493)              !    yph_lo = lower ph value that brackets the root (i.e., correct ph)
   494)              !    yph_hi = upper ph value that brackets the root (i.e., correct ph)
   495)              !    yph    = current ph value
   496)              !    yposnet_lo and yposnet_hi = net positive ions for
   497)              !       yph_lo and yph_hi
   498)              !-----------------------------------------------------------------
   499)              do iter = 1,itermax
   500) 
   501)                 if (iter == 1) then
   502)                    ! 1st iteration ph = lower bound value
   503)                    yph_lo = 2.0_r8
   504)                    yph_hi = yph_lo
   505)                    yph = yph_lo
   506)                 else if (iter == 2) then
   507)                    ! 2nd iteration ph = upper bound value
   508)                    yph_hi = 7.0_r8
   509)                    yph = yph_hi
   510)                 else
   511)                    ! later iteration ph = mean of the two bracketing values
   512)                    yph = 0.5_r8*(yph_lo + yph_hi)
   513)                 end if
   514)                 ! calc current [H+] from ph
   515)                 xph(i,k) = 10.0_r8**(-yph)
   516) 
   517)                 !-----------------------------------------------------------------
   518)                 !        ... hno3
   519)                 !-----------------------------------------------------------------
   520)                 Ehno3 = fact1_hno3/(1.0_r8 + fact2_hno3*(1.0_r8 + fact3_hno3/xph(i,k)))
   521) 
   522)                 !-----------------------------------------------------------------
   523)                 !          ... so2
   524)                 !-----------------------------------------------------------------
   525)                 Eso2 = fact1_so2/(1.0_r8 + fact2_so2*(1.0_r8 + (fact3_so2/xph(i,k)) &
   526)                      *(1.0_r8 +  fact4_so2/xph(i,k))))
   527) 
   528)                 !-----------------------------------------------------------------
   529)                 !          ... nh3
   530)                 !-----------------------------------------------------------------
   531)                 Enh3 = fact1_nh3/(1.0_r8 + fact2_nh3*(1.0_r8 + fact3_nh3*xph(i,k)))
   532) 
   533)                 tmp_nh4  = Enh3 * xph(i,k)
   534)                 tmp_hso3 = Eso2 / xph(i,k)
   535)                 tmp_so3  = tmp_hso3 * 2.0_r8*fact4_so2/xph(i,k)
   536)                 tmp_hco3 = Eco2 / xph(i,k)
   537)                 tmp_oh   = Eh2o / xph(i,k)
   538)                 tmp_no3  = Ehno3 / xph(i,k)
   539)                 tmp_so4 = cldconc%so4_fact*Eso4
   540)                 tmp_pos = xph(i,k) + tmp_nh4
   541)                 tmp_neg = tmp_oh + tmp_hco3 + tmp_no3 + tmp_hso3 + tmp_so3 + tmp_so4
   542) 
   543)                 ynetpos = tmp_pos - tmp_neg
   544) 
   545) 
   546)                 ! yposnet = net positive ions/charge
   547)                 ! if the correct ph is bracketed by yph_lo and yph_hi (with yph_lo < yph_hi),
   548)                 !    then you will have yposnet_lo > 0 and yposnet_hi < 0
   549)                 converged = .false.
   550)                 if (iter > 2) then
   551)                    if (ynetpos == 0.0_r8) then
   552)                       ! the exact solution was found (very unlikely)
   553)                       tmp_hp = xph(i,k)
   554)                       converged = .true.
   555)                       exit
   556)                    else if (ynetpos >= 0.0_r8) then
   557)                       ! net positive ions are >= 0 for both yph and yph_lo
   558)                       !    so replace yph_lo with yph
   559)                       yph_lo = yph
   560)                       ynetpos_lo = ynetpos
   561)                    else
   562)                       ! net positive ions are <= 0 for both yph and yph_hi
   563)                       !    so replace yph_hi with yph
   564)                       yph_hi = yph
   565)                       ynetpos_hi = ynetpos
   566)                    end if
   567) 
   568)                    if (abs(yph_hi - yph_lo) .le. 0.005_r8) then
   569)                       ! |yph_hi - yph_lo| <= convergence criterion, so set
   570)                       !    final ph to their midpoint and exit
   571)                       ! (.005 absolute error in pH gives .01 relative error in H+)
   572)                       tmp_hp = xph(i,k)
   573)                       yph = 0.5_r8*(yph_hi + yph_lo)
   574)                       xph(i,k) = 10.0_r8**(-yph)
   575)                       converged = .true.
   576)                       exit
   577)                    else 
   578)                       ! do another iteration
   579)                       converged = .false.
   580)                    end if
   581) 
   582)                 else if (iter == 1) then
   583)                    if (ynetpos <= 0.0_r8) then
   584)                       ! the lower and upper bound ph values (2.0 and 7.0) do not bracket
   585)                       !    the correct ph, so use the lower bound
   586)                       tmp_hp = xph(i,k)
   587)                       converged = .true.
   588)                       exit
   589)                    end if
   590)                    ynetpos_lo = ynetpos
   591) 
   592)                 else ! (iter == 2)
   593)                    if (ynetpos >= 0.0_r8) then
   594)                       ! the lower and upper bound ph values (2.0 and 7.0) do not bracket
   595)                       !    the correct ph, so use they upper bound
   596)                       tmp_hp = xph(i,k)
   597)                       converged = .true.
   598)                       exit
   599)                    end if
   600)                    ynetpos_hi = ynetpos
   601)                 end if
   602) 
   603)              end do ! iter
   604) 
   605)              if( .not. converged ) then
   606)                 write(iulog,*) 'SETSOX: pH failed to converge @ (',i,',',k,'), % change=', &
   607)                      100._r8*delta
   608)              end if
   609)           else
   610)              xph(i,k) =  1.e-7_r8
   611)           end if
   612)        end do col_loop0

   629)           !        ... hno3
   630)           !-----------------------------------------------------------------------

   636)           !        ... h2o2

   643)           !         ... so2
   644)           !-----------------------------------------------------------------

   647)           x2 = 6.0e-8_r8*EXP( 1120._r8*work1(i) )
   648) 
   649)           wrk = xe/xph(i,k)
   650)           heso2(i,k)  = xk*(1._r8 + wrk*(1._r8 + x2/xph(i,k)))
   651) 
   652)           !-----------------------------------------------------------------
   653)           !          ... nh3
   654)           !-----------------------------------------------------------------
   655)           xk = 58._r8   *EXP( 4085._r8*work1(i) )
   656)           xe = 1.7e-5_r8*EXP(-4325._r8*work1(i) )
   657)           henh3(i,k)  = xk*(1._r8 + xe*xph(i,k)/xkw)
   658) 
   659)           !-----------------------------------------------------------------
   660)           !        ... o3
   661)           !-----------------------------------------------------------------
   662)           xk = 1.15e-2_r8 *EXP( 2560._r8*work1(i) )
   663)           heo3(i,k) = xk
   664) 
   665)           !------------------------------------------------------------------------
   666)           !       ... for Ho2(g) -> H2o2(a) formation 
   667)           !           schwartz JGR, 1984, 11589
   668)           !------------------------------------------------------------------------
   669)           kh4 = (kh2 + kh3*kh1/xph(i,k)) / ((1._r8 + kh1/xph(i,k))**2)
   670)           ho2s = kh0*xho2(i,k)*patm*(1._r8 + kh1/xph(i,k))  ! ho2s = ho2(a)+o2-
   671)           r1h2o2 = kh4*ho2s*ho2s                         ! prod(h2o2) in mole/L(w)/s
   672) 
   673)           if ( cloud_borne ) then
   674)              r2h2o2 = r1h2o2*xl        &    ! mole/L(w)/s   * L(w)/fm3(a) = mole/fm3(a)/s
   675)                   / const0*1.e+6_r8  &    ! correct a bug here ????
   676)                   / xam
   677)           else
   678)              r2h2o2 = r1h2o2*xl  &          ! mole/L(w)/s   * L(w)/fm3(a) = mole/fm3(a)/s
   679)                   * const0     &          ! mole/fm3(a)/s * 1.e-3       = mole/cm3(a)/s
   680)                   / xam                   ! /cm3(a)/s    / air-den     = mix-ratio/s
   681)           endif
   682) 
   683)           if ( .not. modal_aerosols ) then
   684)              xh2o2(i,k) = xh2o2(i,k) + r2h2o2*dtime         ! updated h2o2 by het production
   685)           endif

   688)           !       ... Partioning 
   689)           !-----------------------------------------------
   690) 
   691)           !-----------------------------------------------------------------
   692)           !        ... hno3
   693)           !-----------------------------------------------------------------
   694)           px = hehno3(i,k) * Ra * tz * xl
   695)           hno3g(i,k) = (xhno3(i,k)+xno3(i,k))/(1._r8 + px)
   696) 
   697)           !------------------------------------------------------------------------
   698)           !        ... h2o2
   699)           !------------------------------------------------------------------------
   700)           px = heh2o2(i,k) * Ra * tz * xl
   701)           h2o2g =  xh2o2(i,k)/(1._r8+ px)
   702) 
   703)           !------------------------------------------------------------------------
   704)           !         ... so2
   705)           !------------------------------------------------------------------------
   706)           px = heso2(i,k) * Ra * tz * xl
   707)           so2g =  xso2(i,k)/(1._r8+ px)
   708) 
   709)           !------------------------------------------------------------------------
   710)           !         ... o3
   711)           !------------------------------------------------------------------------
   712)           px = heo3(i,k) * Ra * tz * xl
   713)           o3g =  xo3(i,k)/(1._r8+ px)
   714) 
   715)           !------------------------------------------------------------------------
   716)           !         ... nh3
   717)           !------------------------------------------------------------------------
   718)           px = henh3(i,k) * Ra * tz * xl
   719)           if (id_nh3>0) then
   720)              nh3g(i,k) = (xnh3(i,k)+xnh4(i,k))/(1._r8+ px)
   721)           else
   722)              nh3g(i,k) = 0._r8
   723)           endif
   724) 
   725)           !-----------------------------------------------
   726)           !       ... Aqueous phase reaction rates
   727)           !           SO2 + H2O2 -> SO4
   728)           !           SO2 + O3   -> SO4
   729)           !-----------------------------------------------
   730) 
   731)           !------------------------------------------------------------------------
   732)           !       ... S(IV) (HSO3) + H2O2
   733)           !------------------------------------------------------------------------
   734)           rah2o2 = 8.e4_r8 * EXP( -3650._r8*work1(i) )  &
   735)                / (.1_r8 + xph(i,k))
   736) 
   737)           !------------------------------------------------------------------------
   738)           !        ... S(IV)+ O3
   739)           !------------------------------------------------------------------------
   740)           rao3   = 4.39e11_r8 * EXP(-4131._r8/tz)  &
   741)                + 2.56e3_r8  * EXP(-996._r8 /tz) /xph(i,k)
   742) 
   743)           !-----------------------------------------------------------------
   744)           !       ... Prediction after aqueous phase
   745)           !       so4
   746)           !       When Cloud is present 
   747)           !   
   748)           !       S(IV) + H2O2 = S(VI)
   749)           !       S(IV) + O3   = S(VI)
   750)           !
   751)           !       reference:
   752)           !           (1) Seinfeld
   753)           !           (2) Benkovitz
   754)           !-----------------------------------------------------------------
   755)           
   756)           !............................
   757)           !       S(IV) + H2O2 = S(VI)
   758)           !............................
   759)           
   760)           IF (XL .ge. 1.e-8_r8) THEN    !! WHEN CLOUD IS PRESENTED

   764)              else

   766)              endif

   774)                      * heso2(i,k)  * so2g  * patm_x    ! [M/s]

   785)              ccc = max(ccc, 1.e-30_r8)

   809)                    xh2o2(i,k)=1.e-20_r8
   810)                 else

   823)
