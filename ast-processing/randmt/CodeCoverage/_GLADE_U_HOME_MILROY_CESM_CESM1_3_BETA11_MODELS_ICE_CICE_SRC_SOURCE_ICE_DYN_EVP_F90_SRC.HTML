  1431)          ! divergence  =  e_11 + e_22
  1432)          divune    = cyp(i,j)*uvel(i  ,j  ) - dyt(i,j)*uvel(i-1,j  ) &
  1433)                    + cxp(i,j)*vvel(i  ,j  ) - dxt(i,j)*vvel(i  ,j-1)
  1434)          divunw    = cym(i,j)*uvel(i-1,j  ) + dyt(i,j)*uvel(i  ,j  ) &
  1435)                    + cxp(i,j)*vvel(i-1,j  ) - dxt(i,j)*vvel(i-1,j-1)
  1436)          divusw    = cym(i,j)*uvel(i-1,j-1) + dyt(i,j)*uvel(i  ,j-1) &
  1437)                    + cxm(i,j)*vvel(i-1,j-1) + dxt(i,j)*vvel(i-1,j  )
  1438)          divuse    = cyp(i,j)*uvel(i  ,j-1) - dyt(i,j)*uvel(i-1,j-1) &
  1439)                    + cxm(i,j)*vvel(i  ,j-1) + dxt(i,j)*vvel(i  ,j  )
  1440) 
  1441)          ! tension strain rate  =  e_11 - e_22
  1442)          tensionne = -cym(i,j)*uvel(i  ,j  ) - dyt(i,j)*uvel(i-1,j  ) &
  1443)                    +  cxm(i,j)*vvel(i  ,j  ) + dxt(i,j)*vvel(i  ,j-1)
  1444)          tensionnw = -cyp(i,j)*uvel(i-1,j  ) + dyt(i,j)*uvel(i  ,j  ) &
  1445)                    +  cxm(i,j)*vvel(i-1,j  ) + dxt(i,j)*vvel(i-1,j-1)
  1446)          tensionsw = -cyp(i,j)*uvel(i-1,j-1) + dyt(i,j)*uvel(i  ,j-1) &
  1447)                    +  cxp(i,j)*vvel(i-1,j-1) - dxt(i,j)*vvel(i-1,j  )
  1448)          tensionse = -cym(i,j)*uvel(i  ,j-1) - dyt(i,j)*uvel(i-1,j-1) &
  1449)                    +  cxp(i,j)*vvel(i  ,j-1) - dxt(i,j)*vvel(i  ,j  )
  1450) 
  1451)          ! shearing strain rate  =  e_12
  1452)          shearne = -cym(i,j)*vvel(i  ,j  ) - dyt(i,j)*vvel(i-1,j  ) &
  1453)                  -  cxm(i,j)*uvel(i  ,j  ) - dxt(i,j)*uvel(i  ,j-1)
  1454)          shearnw = -cyp(i,j)*vvel(i-1,j  ) + dyt(i,j)*vvel(i  ,j  ) &
  1455)                  -  cxm(i,j)*uvel(i-1,j  ) - dxt(i,j)*uvel(i-1,j-1)
  1456)          shearsw = -cyp(i,j)*vvel(i-1,j-1) + dyt(i,j)*vvel(i  ,j-1) &
  1457)                  -  cxp(i,j)*uvel(i-1,j-1) + dxt(i,j)*uvel(i-1,j  )
  1458)          shearse = -cym(i,j)*vvel(i  ,j-1) - dyt(i,j)*vvel(i-1,j-1) &
  1459)                  -  cxp(i,j)*uvel(i  ,j-1) + dxt(i,j)*uvel(i  ,j  )
  1460) 
  1461)          ! Delta (in the denominator of zeta, eta)
  1462)          Deltane = sqrt(divune**2 + ecci*(tensionne**2 + shearne**2))
  1463)          Deltanw = sqrt(divunw**2 + ecci*(tensionnw**2 + shearnw**2))
  1464)          Deltase = sqrt(divuse**2 + ecci*(tensionse**2 + shearse**2))
  1465)          Deltasw = sqrt(divusw**2 + ecci*(tensionsw**2 + shearsw**2))
  1466) 
  1467)       !-----------------------------------------------------------------
  1468)       ! on last subcycle, save quantities for mechanical redistribution
  1469)       !-----------------------------------------------------------------
  1470)          if (ksub == ndte) then
  1471)             divu(i,j) = p25*(divune + divunw + divuse + divusw) * tarear(i,j)
  1472)             tmp = p25*(Deltane + Deltanw + Deltase + Deltasw)   * tarear(i,j)
  1473)             rdg_conv(i,j)  = -min(divu(i,j),c0)
  1474)             rdg_shear(i,j) = p5*(tmp-abs(divu(i,j))) 
  1475) 
  1476)             ! diagnostic only
  1477)             ! shear = sqrt(tension**2 + shearing**2)
  1478)             shear(i,j) = p25*tarear(i,j)*sqrt( &
  1479)                  (tensionne + tensionnw + tensionse + tensionsw)**2 &
  1480)                 +  (shearne +   shearnw +   shearse +   shearsw)**2)
  1481) 
  1482)          endif
  1483) 
  1484)       !-----------------------------------------------------------------
  1485)       ! replacement pressure/Delta                   ! kg/s
  1486)       ! save replacement pressure for principal stress calculation
  1487)       !-----------------------------------------------------------------
  1488)          if (evp_damping) then
  1489)             ! enforce damping criterion
  1490)             c0ne = min(strength(i,j)/max(Deltane,c4*tinyarea(i,j)),rcon)
  1491)             c0nw = min(strength(i,j)/max(Deltanw,c4*tinyarea(i,j)),rcon)
  1492)             c0sw = min(strength(i,j)/max(Deltasw,c4*tinyarea(i,j)),rcon)
  1493)             c0se = min(strength(i,j)/max(Deltase,c4*tinyarea(i,j)),rcon)
  1494)             prs_sig(i,j) = strength(i,j)* &
  1495)                            Deltane/max(Deltane,c4*tinyarea(i,j)) ! ne
  1496)          else
  1497)             ! original version
  1498)             c0ne = strength(i,j)/max(Deltane,tinyarea(i,j))
  1499)             c0nw = strength(i,j)/max(Deltanw,tinyarea(i,j))
  1500)             c0sw = strength(i,j)/max(Deltasw,tinyarea(i,j))
  1501)             c0se = strength(i,j)/max(Deltase,tinyarea(i,j))
  1502)             prs_sig(i,j) = c0ne*Deltane ! northeast
  1503)          endif
  1504) 
  1505)          c1ne = c0ne*dte2T
  1506)          c1nw = c0nw*dte2T
  1507)          c1sw = c0sw*dte2T
  1508)          c1se = c0se*dte2T
  1509) 
  1510)       !-----------------------------------------------------------------
  1511)       ! the stresses                            ! kg/s^2
  1512)       ! (1) northeast, (2) northwest, (3) southwest, (4) southeast
  1513)       !-----------------------------------------------------------------
  1514) 
  1515)          stressp_1(i,j) = (stressp_1(i,j) + c1ne*(divune - Deltane)) &
  1516)                           * denom1
  1517)          stressp_2(i,j) = (stressp_2(i,j) + c1nw*(divunw - Deltanw)) &
  1518)                           * denom1
  1519)          stressp_3(i,j) = (stressp_3(i,j) + c1sw*(divusw - Deltasw)) &
  1520)                           * denom1
  1521)          stressp_4(i,j) = (stressp_4(i,j) + c1se*(divuse - Deltase)) &
  1522)                           * denom1
  1523) 
  1524)          stressm_1(i,j) = (stressm_1(i,j) + c1ne*tensionne) * denom2
  1525)          stressm_2(i,j) = (stressm_2(i,j) + c1nw*tensionnw) * denom2
  1526)          stressm_3(i,j) = (stressm_3(i,j) + c1sw*tensionsw) * denom2
  1527)          stressm_4(i,j) = (stressm_4(i,j) + c1se*tensionse) * denom2
  1528) 
  1529)          stress12_1(i,j) = (stress12_1(i,j) + c1ne*shearne*p5) * denom2
  1530)          stress12_2(i,j) = (stress12_2(i,j) + c1nw*shearnw*p5) * denom2
  1531)          stress12_3(i,j) = (stress12_3(i,j) + c1sw*shearsw*p5) * denom2
  1532)          stress12_4(i,j) = (stress12_4(i,j) + c1se*shearse*p5) * denom2
  1533) 
  1534)       !-----------------------------------------------------------------
  1535)       ! Eliminate underflows.
  1536)       ! The following code is commented out because it is relatively 
  1537)       ! expensive and most compilers include a flag that accomplishes
  1538)       ! the same thing more efficiently.  This code is cheaper than
  1539)       ! handling underflows if the compiler lacks a flag; uncomment
  1540)       ! it in that case.  The compiler flag is often described with the 
  1541)       ! phrase "flush to zero".
  1542)       !-----------------------------------------------------------------
  1543) 
  1544) !      stressp_1(i,j) = sign(max(abs(stressp_1(i,j)),puny),stressp_1(i,j))
  1545) !      stressp_2(i,j) = sign(max(abs(stressp_2(i,j)),puny),stressp_2(i,j))
  1546) !      stressp_3(i,j) = sign(max(abs(stressp_3(i,j)),puny),stressp_3(i,j))
  1547) !      stressp_4(i,j) = sign(max(abs(stressp_4(i,j)),puny),stressp_4(i,j))
  1548) 
  1549) !      stressm_1(i,j) = sign(max(abs(stressm_1(i,j)),puny),stressm_1(i,j))
  1550) !      stressm_2(i,j) = sign(max(abs(stressm_2(i,j)),puny),stressm_2(i,j))
  1551) !      stressm_3(i,j) = sign(max(abs(stressm_3(i,j)),puny),stressm_3(i,j))
  1552) !      stressm_4(i,j) = sign(max(abs(stressm_4(i,j)),puny),stressm_4(i,j))
  1553) 
  1554) !      stress12_1(i,j) = sign(max(abs(stress12_1(i,j)),puny),stress12_1(i,j))
  1555) !      stress12_2(i,j) = sign(max(abs(stress12_2(i,j)),puny),stress12_2(i,j))
  1556) !      stress12_3(i,j) = sign(max(abs(stress12_3(i,j)),puny),stress12_3(i,j))
  1557) !      stress12_4(i,j) = sign(max(abs(stress12_4(i,j)),puny),stress12_4(i,j))
  1558) 
  1559)       !-----------------------------------------------------------------
  1560)       ! combinations of the stresses for the momentum equation ! kg/s^2
  1561)       !-----------------------------------------------------------------
  1562) 
  1563)          ssigpn  = stressp_1(i,j) + stressp_2(i,j)
  1564)          ssigps  = stressp_3(i,j) + stressp_4(i,j)
  1565)          ssigpe  = stressp_1(i,j) + stressp_4(i,j)
  1566)          ssigpw  = stressp_2(i,j) + stressp_3(i,j)
  1567)          ssigp1  =(stressp_1(i,j) + stressp_3(i,j))*p055
  1568)          ssigp2  =(stressp_2(i,j) + stressp_4(i,j))*p055
  1569) 
  1570)          ssigmn  = stressm_1(i,j) + stressm_2(i,j)
  1571)          ssigms  = stressm_3(i,j) + stressm_4(i,j)
  1572)          ssigme  = stressm_1(i,j) + stressm_4(i,j)
  1573)          ssigmw  = stressm_2(i,j) + stressm_3(i,j)
  1574)          ssigm1  =(stressm_1(i,j) + stressm_3(i,j))*p055
  1575)          ssigm2  =(stressm_2(i,j) + stressm_4(i,j))*p055
  1576) 
  1577)          ssig12n = stress12_1(i,j) + stress12_2(i,j)
  1578)          ssig12s = stress12_3(i,j) + stress12_4(i,j)
  1579)          ssig12e = stress12_1(i,j) + stress12_4(i,j)
  1580)          ssig12w = stress12_2(i,j) + stress12_3(i,j)
  1581)          ssig121 =(stress12_1(i,j) + stress12_3(i,j))*p111
  1582)          ssig122 =(stress12_2(i,j) + stress12_4(i,j))*p111
  1583) 
  1584)          csigpne = p111*stressp_1(i,j) + ssigp2 + p027*stressp_3(i,j)
  1585)          csigpnw = p111*stressp_2(i,j) + ssigp1 + p027*stressp_4(i,j)
  1586)          csigpsw = p111*stressp_3(i,j) + ssigp2 + p027*stressp_1(i,j)
  1587)          csigpse = p111*stressp_4(i,j) + ssigp1 + p027*stressp_2(i,j)
  1588)          
  1589)          csigmne = p111*stressm_1(i,j) + ssigm2 + p027*stressm_3(i,j)
  1590)          csigmnw = p111*stressm_2(i,j) + ssigm1 + p027*stressm_4(i,j)
  1591)          csigmsw = p111*stressm_3(i,j) + ssigm2 + p027*stressm_1(i,j)
  1592)          csigmse = p111*stressm_4(i,j) + ssigm1 + p027*stressm_2(i,j)
  1593)          
  1594)          csig12ne = p222*stress12_1(i,j) + ssig122 &
  1595)                   + p055*stress12_3(i,j)
  1596)          csig12nw = p222*stress12_2(i,j) + ssig121 &
  1597)                   + p055*stress12_4(i,j)
  1598)          csig12sw = p222*stress12_3(i,j) + ssig122 &
  1599)                   + p055*stress12_1(i,j)
  1600)          csig12se = p222*stress12_4(i,j) + ssig121 &
  1601)                   + p055*stress12_2(i,j)
  1602) 
  1603)          str12ew = p5*dxt(i,j)*(p333*ssig12e + p166*ssig12w)
  1604)          str12we = p5*dxt(i,j)*(p333*ssig12w + p166*ssig12e)
  1605)          str12ns = p5*dyt(i,j)*(p333*ssig12n + p166*ssig12s)
  1606)          str12sn = p5*dyt(i,j)*(p333*ssig12s + p166*ssig12n)
  1607) 
  1608)       !-----------------------------------------------------------------
  1609)       ! for dF/dx (u momentum)
  1610)       !-----------------------------------------------------------------
  1611)          strp_tmp  = p25*dyt(i,j)*(p333*ssigpn  + p166*ssigps)
  1612)          strm_tmp  = p25*dyt(i,j)*(p333*ssigmn  + p166*ssigms)
  1613) 
  1614)          ! northeast (i,j)
  1615)          str8(i,j,1) = -strp_tmp - strm_tmp - str12ew &
  1616)               + dxhy(i,j)*(-csigpne + csigmne) + dyhx(i,j)*csig12ne
  1617) 
  1618)          ! northwest (i+1,j)
  1619)          str8(i,j,2) = strp_tmp + strm_tmp - str12we &
  1620)               + dxhy(i,j)*(-csigpnw + csigmnw) + dyhx(i,j)*csig12nw
  1621) 
  1622)          strp_tmp  = p25*dyt(i,j)*(p333*ssigps  + p166*ssigpn)
  1623)          strm_tmp  = p25*dyt(i,j)*(p333*ssigms  + p166*ssigmn)
  1624) 
  1625)          ! southeast (i,j+1)
  1626)          str8(i,j,3) = -strp_tmp - strm_tmp + str12ew &
  1627)               + dxhy(i,j)*(-csigpse + csigmse) + dyhx(i,j)*csig12se
  1628) 
  1629)          ! southwest (i+1,j+1)
  1630)          str8(i,j,4) = strp_tmp + strm_tmp + str12we &
  1631)               + dxhy(i,j)*(-csigpsw + csigmsw) + dyhx(i,j)*csig12sw
  1632) 
  1633)       !-----------------------------------------------------------------
  1634)       ! for dF/dy (v momentum)
  1635)       !-----------------------------------------------------------------
  1636)          strp_tmp  = p25*dxt(i,j)*(p333*ssigpe  + p166*ssigpw)
  1637)          strm_tmp  = p25*dxt(i,j)*(p333*ssigme  + p166*ssigmw)
  1638) 
  1639)          ! northeast (i,j)
  1640)          str8(i,j,5) = -strp_tmp + strm_tmp - str12ns &
  1641)               - dyhx(i,j)*(csigpne + csigmne) + dxhy(i,j)*csig12ne
  1642) 
  1643)          ! southeast (i,j+1)
  1644)          str8(i,j,6) = strp_tmp - strm_tmp - str12sn &
  1645)               - dyhx(i,j)*(csigpse + csigmse) + dxhy(i,j)*csig12se
  1646) 
  1647)          strp_tmp  = p25*dxt(i,j)*(p333*ssigpw  + p166*ssigpe)
  1648)          strm_tmp  = p25*dxt(i,j)*(p333*ssigmw  + p166*ssigme)
  1649) 
  1650)          ! northwest (i+1,j)
  1651)          str8(i,j,7) = -strp_tmp + strm_tmp + str12ns &
  1652)               - dyhx(i,j)*(csigpnw + csigmnw) + dxhy(i,j)*csig12nw
  1653) 
  1654)          ! southwest (i+1,j+1)
  1655)          str8(i,j,8) = strp_tmp - strm_tmp + str12sn &
  1656)               - dyhx(i,j)*(csigpsw + csigmsw) + dxhy(i,j)*csig12sw
  1657) 
  1658)       endif                     ! phase
  1659)       enddo                     ! ij
  1660) 
  1661)       end subroutine stress

  1666) ! !IROUTINE: stepu - integrates mom eqn for u,v
  1667) !
  1668) ! !INTERFACE:
  1669) !
  1670)       subroutine stepu (nx_block,   ny_block, &
  1671)                         icellu,               &
  1672)                         indxui,     indxuj,   &
  1673)                         aiu,        str8,      &
  1674)                         uocn,       vocn,     &
  1675)                         waterx,     watery,   &
  1676)                         forcex,     forcey,   &
  1677)                         umassdtei,  fm,       &
  1678)                         uarear,               &
  1679)                         strocnx,    strocny,  &
  1680)                         strintx,    strinty,  &
  1681)                         uvel,       vvel)
  1682) !
  1683) ! !DESCRIPTION:
  1684) !
  1685) ! Calculation of the surface stresses
  1686) ! Integration of the momentum equation to find velocity (u,v)
  1687) !
  1688) ! !REVISION HISTORY:
  1689) !
  1690) ! author: Elizabeth C. Hunke, LANL
  1691) !
  1692) ! !USES:
  1693) !
  1694) ! !INPUT/OUTPUT PARAMETERS:
  1695) !
  1696)       integer (kind=int_kind), intent(in) :: &
  1697)          nx_block, ny_block, & ! block dimensions
  1698)          icellu                ! total count when iceumask is true
  1699) 
  1700)       integer (kind=int_kind), dimension (nx_block*ny_block), &
  1701)          intent(in) :: &
  1702)          indxui  , & ! compressed index in i-direction
  1703)          indxuj      ! compressed index in j-direction
  1704) 
  1705)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(in) :: &
  1706)          aiu     , & ! ice fraction on u-grid
  1707)          waterx  , & ! for ocean stress calculation, x (m/s)
  1708)          watery  , & ! for ocean stress calculation, y (m/s)
  1709)          forcex  , & ! work array: combined atm stress and ocn tilt, x
  1710)          forcey  , & ! work array: combined atm stress and ocn tilt, y
  1711)          umassdtei,& ! mass of U-cell/dte (kg/m^2 s)
  1712)          uocn    , & ! ocean current, x-direction (m/s)
  1713)          vocn    , & ! ocean current, y-direction (m/s)
  1714)          fm      , & ! Coriolis param. * mass in U-cell (kg/s)
  1715)          uarear      ! 1/uarea
  1716) 
  1717)       real (kind=dbl_kind), dimension(nx_block,ny_block,8), &
  1718)          intent(in) :: &
  1719)          str8
  1720) 
  1721)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
  1722)          intent(inout) :: &
  1723)          uvel    , & ! x-component of velocity (m/s)
  1724)          vvel        ! y-component of velocity (m/s)
  1725) 
  1726)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
  1727)          intent(inout) :: &
  1728)          strocnx , & ! ice-ocean stress, x-direction
  1729)          strocny , & ! ice-ocean stress, y-direction
  1730)          strintx , & ! divergence of internal ice stress, x (N/m^2)
  1731)          strinty     ! divergence of internal ice stress, y (N/m^2)
  1732) !
  1733) !EOP
  1734) !
  1735)       integer (kind=int_kind) :: &
  1736)          i, j, ij
  1737) 
  1738)       real (kind=dbl_kind) :: &
  1739)          uold, vold        , & ! old-time uvel, vvel
  1740)          vrel              , & ! relative ice-ocean velocity
  1741)          cca,ccb,ab2,cc1,cc2,& ! intermediate variables
  1742)          taux, tauy            ! part of ocean stress term          
  1743) 
  1744)       !-----------------------------------------------------------------
  1745)       ! integrate the momentum equation
  1746)       !-----------------------------------------------------------------
  1747) 
  1748)       do ij =1, icellu
  1749)          i = indxui(ij)
  1750)          j = indxuj(ij)
  1751) 
  1752)          uold = uvel(i,j)
  1753)          vold = vvel(i,j)
  1754) 
  1755)          ! (magnitude of relative ocean current)*rhow*drag*aice
  1756)          vrel = aiu(i,j)*dragw*sqrt((uocn(i,j) - uold)**2 + &
  1757)                                     (vocn(i,j) - vold)**2)  ! m/s
  1758)          ! ice/ocean stress
  1759)          taux = vrel*waterx(i,j) ! NOTE this is not the entire
  1760)          tauy = vrel*watery(i,j) ! ocn stress term
  1761) 
  1762)          ! alpha, beta are defined in Hunke and Dukowicz (1997), section 3.2
  1763)          cca = umassdtei(i,j) + vrel * cosw      ! alpha, kg/m^2 s
  1764)          ccb = fm(i,j)        + vrel * sinw      ! beta,  kg/m^2 s
  1765)          ab2 = cca**2 + ccb**2
  1766) 
  1767)          ! divergence of the internal stress tensor
  1768)          strintx(i,j) = uarear(i,j)* &
  1769)              (str8(i,j,1) + str8(i+1,j,2) + str8(i,j+1,3) + str8(i+1,j+1,4))
  1770)          strinty(i,j) = uarear(i,j)* &
  1771)              (str8(i,j,5) + str8(i,j+1,6) + str8(i+1,j,7) + str8(i+1,j+1,8))
  1772) 
  1773)          ! finally, the velocity components
  1774)          cc1 = strintx(i,j) + forcex(i,j) + taux &
  1775)              + umassdtei(i,j)*uold
  1776)          cc2 = strinty(i,j) + forcey(i,j) + tauy &
  1777)              + umassdtei(i,j)*vold
  1778) 
  1779)          uvel(i,j) = (cca*cc1 + ccb*cc2) / ab2 ! m/s
  1780)          vvel(i,j) = (cca*cc2 - ccb*cc1) / ab2
