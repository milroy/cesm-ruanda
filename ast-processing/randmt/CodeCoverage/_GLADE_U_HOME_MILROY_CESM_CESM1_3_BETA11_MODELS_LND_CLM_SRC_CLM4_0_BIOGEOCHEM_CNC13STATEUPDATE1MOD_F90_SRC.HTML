    34) ! !INTERFACE:
    35) subroutine C13StateUpdate0(num_soilp, filter_soilp)
    36) !
    37) ! !DESCRIPTION:
    38) ! On the radiation time step, update cpool carbon state
    39) !
    40) ! !USES:
    41)    use clmtype
    42)    use clm_time_manager, only: get_step_size
    43)    use clm_varctl, only : use_c13
    44) !
    45) ! !ARGUMENTS:
    46)    implicit none
    47)    integer, intent(in) :: num_soilp       ! number of soil pfts in filter
    48)    integer, intent(in) :: filter_soilp(:) ! filter for soil pfts
    49) !
    50) ! !CALLED FROM:
    51) ! subroutine CNEcosystemDyn
    52) !
    53) ! !REVISION HISTORY:
    54) ! 7/1/05: Created by Peter Thornton
    55) !
    56) ! !LOCAL VARIABLES:
    57) ! local pointers to implicit in arrays
    58)    real(r8), pointer :: psnshade_to_cpool(:)
    59)    real(r8), pointer :: psnsun_to_cpool(:)
    60) !
    61) ! local pointers to implicit in/out arrays
    62)    real(r8), pointer :: cpool(:)              ! (gC/m2) temporary photosynthate C pool
    63) ! !OTHER LOCAL VARIABLES:
    64)    integer :: p     ! indices
    65)    integer :: fp   ! lake filter indices
    66)    real(r8):: dt      ! radiation time step (seconds)
    67) !
    68) !EOP
    69) !-----------------------------------------------------------------------
    70) 
    71)    if (.not. use_c13) then
    72)       RETURN
    73)    end if
    74) 
    75)     ! assign local pointers at the pft level
    76)     cpool                          => pc13s%cpool
    77)     psnshade_to_cpool              => pc13f%psnshade_to_cpool
    78)     psnsun_to_cpool                => pc13f%psnsun_to_cpool
    79) 
    80)     ! set time steps
    81)     dt = real( get_step_size(), r8 )
    82) 
    83)     ! pft loop
    84)     do fp = 1,num_soilp
    85)        p = filter_soilp(fp)
    86)        ! gross photosynthesis fluxes
    87)        cpool(p) = cpool(p) + psnsun_to_cpool(p)*dt
    88)        cpool(p) = cpool(p) + psnshade_to_cpool(p)*dt
    89)     end do
    90) 
    91) end subroutine C13StateUpdate0
