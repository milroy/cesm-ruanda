   696)   subroutine het_diags( het_rates, mmr, pdel, lchnk, ncol )
   697) 
   698)     use cam_history,  only : outfld
   699)     use phys_grid,    only : get_wght_all_p
   700)     implicit none
   701) 
   702)     integer,  intent(in)  :: lchnk
   703)     integer,  intent(in)  :: ncol
   704)     real(r8), intent(in)  :: het_rates(ncol,pver,max(1,gas_pcnst))
   705)     real(r8), intent(in)  :: mmr(ncol,pver,gas_pcnst)
   706)     real(r8), intent(in)  :: pdel(ncol,pver)
   707) 
   708)     real(r8), dimension(ncol) :: noy_wk, sox_wk, nhx_wk, wrk_wd
   709)     integer :: m, k, j
   710)     integer :: plat
   711)     real(r8) :: wght(ncol)
   712)     !
   713)     ! output integrated wet deposition field
   714)     !
   715)     noy_wk(:) = 0._r8
   716)     sox_wk(:) = 0._r8
   717)     nhx_wk(:) = 0._r8
   718) 
   719)     call get_wght_all_p(lchnk, ncol, wght)
   720) 
   721)     do m = 1,gas_pcnst
   722)        !
   723)        ! compute vertical integral
   724)        !
   725)        wrk_wd(:ncol) = 0._r8
   726)        do k = 1,pver
   727)           wrk_wd(:ncol) = wrk_wd(:ncol) + het_rates(:ncol,k,m) * mmr(:ncol,k,m) * pdel(:ncol,k) 
   728)        end do
   729)        !
   730)        wrk_wd(:ncol) = wrk_wd(:ncol) * rgrav * wght(:ncol) * rearth**2
   731)        !
   732) 
   733)        call outfld( wetdep_name(m), wrk_wd(:ncol),               ncol, lchnk )
