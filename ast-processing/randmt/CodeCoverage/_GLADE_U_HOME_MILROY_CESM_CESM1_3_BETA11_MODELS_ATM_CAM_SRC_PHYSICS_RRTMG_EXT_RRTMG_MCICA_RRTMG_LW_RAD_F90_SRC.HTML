    94) ! -------- Description --------
    95) 
    96) ! This program is the driver subroutine for RRTMG_LW, the AER LW radiation 
    97) ! model for application to GCMs, that has been adapted from RRTM_LW for
    98) ! improved efficiency.
    99) !
   100) ! NOTE: The call to RRTMG_LW_INI should be moved to the GCM initialization
   101) !  area, since this has to be called only once. 
   102) !
   103) ! This routine:
   104) !    a) calls INATM to read in the atmospheric profile from GCM;
   105) !       all layering in RRTMG is ordered from surface to toa. 
   106) !    b) calls CLDPRMC to set cloud optical depth for McICA based 
   107) !       on input cloud properties 
   108) !    c) calls SETCOEF to calculate various quantities needed for 
   109) !       the radiative transfer algorithm
   110) !    d) calls TAUMOL to calculate gaseous optical depths for each 
   111) !       of the 16 spectral bands
   112) !    e) calls RTRNMC (for both clear and cloudy profiles) to perform the
   113) !       radiative transfer calculation using McICA, the Monte-Carlo 
   114) !       Independent Column Approximation, to represent sub-grid scale 
   115) !       cloud variability
   116) !    f) passes the necessary fluxes and cooling rates back to GCM
   117) !
   118) ! Two modes of operation are possible:
   119) !     The mode is chosen by using either rrtmg_lw.nomcica.f90 (to not use
   120) !     McICA) or rrtmg_lw.f90 (to use McICA) to interface with a GCM. 
   121) !
   122) !    1) Standard, single forward model calculation (imca = 0)
   123) !    2) Monte Carlo Independent Column Approximation (McICA, Pincus et al., 
   124) !       JC, 2003) method is applied to the forward model calculation (imca = 1)
   125) !
   126) ! This call to RRTMG_LW must be preceeded by a call to the module
   127) !     mcica_subcol_gen_lw.f90 to run the McICA sub-column cloud generator,
   128) !     which will provide the cloud physical or cloud optical properties
   129) !     on the RRTMG quadrature point (ngpt) dimension.
   130) !
   131) ! Two methods of cloud property input are possible:
   132) !     Cloud properties can be input in one of two ways (controlled by input 
   133) !     flags inflglw, iceflglw, and liqflglw; see text file rrtmg_lw_instructions
   134) !     and subroutine rrtmg_lw_cldprop.f90 for further details):
   135) !
   136) !    1) Input cloud fraction and cloud optical depth directly (inflglw = 0)
   137) !    2) Input cloud fraction and cloud physical properties (inflglw = 1 or 2);  
   138) !       cloud optical properties are calculated by cldprop or cldprmc based
   139) !       on input settings of iceflglw and liqflglw
   140) !
   141) ! One method of aerosol property input is possible:
   142) !     Aerosol properties can be input in only one way (controlled by input 
   143) !     flag iaer, see text file rrtmg_lw_instructions for further details):
   144) !
   145) !    1) Input aerosol optical depth directly by layer and spectral band (iaer=10);
   146) !       band average optical depth at the mid-point of each spectral band.
   147) !       RRTMG_LW currently treats only aerosol absorption;
   148) !       scattering capability is not presently available. 
   149) !
   150) !
   151) ! ------- Modifications -------
   152) !
   153) ! This version of RRTMG_LW has been modified from RRTM_LW to use a reduced 
   154) ! set of g-points for application to GCMs.

   403) ! icld = 0, clear only
   404) ! icld = 1, with clouds using random cloud overlap
   405) ! icld = 2, with clouds using maximum/random cloud overlap
   406) ! icld = 3, with clouds using maximum cloud overlap (McICA only)
   407)       if (icld.lt.0.or.icld.gt.3) icld = 2
   408) 
   409) ! Set iaer to select aerosol option
   410) ! iaer = 0, no aerosols
   411) ! iaer = 10, input total aerosol optical depth (tauaer) directly 
   412)       iaer = 10
   413) 
   414) ! Call model and data initialization, compute lookup tables, perform
   415) ! reduction of g-points from 256 to 140 for input absorption coefficient 
   416) ! data and other arrays.
   417) !
   418) ! In a GCM this call should be placed in the model initialization
   419) ! area, since this has to be called only once.  
   420) !      call rrtmg_lw_ini
   421) 
   422) !  This is the main longitude/column loop within RRTMG.
   423)       do iplon = 1, ncol
   424) 
   425) !  Prepare atmospheric profile from GCM for use in RRTMG, and define
   426) !  other input parameters.  
   427) 
   428)          call inatm (iplon, nlay, icld, iaer, &
   429)               play, plev, tlay, tlev, tsfc, h2ovmr, &
   430)               o3vmr, co2vmr, ch4vmr, o2vmr, n2ovmr, cfc11vmr, cfc12vmr, &
   431)               cfc22vmr, ccl4vmr, emis, inflglw, iceflglw, liqflglw, &
   432)               cldfmcl, taucmcl, ciwpmcl, clwpmcl, reicmcl, relqmcl, tauaer, &
   433)               pavel, pz, tavel, tz, tbound, semiss, coldry, &
   434)               wkl, wbrodl, wx, pwvcm, inflag, iceflag, liqflag, &
   435)               cldfmc, taucmc, ciwpmc, clwpmc, reicmc, dgesmc, relqmc, taua)
   436) 
   437) !  For cloudy atmosphere, use cldprop to set cloud optical properties based on
   438) !  input cloud physical properties.  Select method based on choices described
   439) !  in cldprop.  Cloud fraction, water path, liquid droplet and ice particle
   440) !  effective radius must be passed into cldprop.  Cloud fraction and cloud
   441) !  optical depth are transferred to rrtmg_lw arrays in cldprop.  
   442) 
   443)          call cldprmc(nlay, inflag, iceflag, liqflag, cldfmc, ciwpmc, &
   444)                       clwpmc, reicmc, dgesmc, relqmc, ncbands, taucmc)
   445) 
   446) ! Calculate information needed by the radiative transfer routine
   447) ! that is specific to this atmosphere, especially some of the 
   448) ! coefficients and indices needed to compute the optical depths
   449) ! by interpolating data from stored reference atmospheres. 
   450) 
   451)          call setcoef(nlay, istart, pavel, tavel, tz, tbound, semiss, &
   452)                       coldry, wkl, wbrodl, &
   453)                       laytrop, jp, jt, jt1, planklay, planklev, plankbnd, &
   454)                       colh2o, colco2, colo3, coln2o, colco, colch4, colo2, &
   455)                       colbrd, fac00, fac01, fac10, fac11, &
   456)                       rat_h2oco2, rat_h2oco2_1, rat_h2oo3, rat_h2oo3_1, &
   457)                       rat_h2on2o, rat_h2on2o_1, rat_h2och4, rat_h2och4_1, &
   458)                       rat_n2oco2, rat_n2oco2_1, rat_o3co2, rat_o3co2_1, &
   459)                       selffac, selffrac, indself, forfac, forfrac, indfor, &
   460)                       minorfrac, scaleminor, scaleminorn2, indminor)
   461) 
   462) !  Calculate the gaseous optical depths and Planck fractions for 
   463) !  each longwave spectral band.
   464) 
   465)          call taumol(nlay, pavel, wx, coldry, &
   466)                      laytrop, jp, jt, jt1, planklay, planklev, plankbnd, &
   467)                      colh2o, colco2, colo3, coln2o, colco, colch4, colo2, &
   468)                      colbrd, fac00, fac01, fac10, fac11, &
   469)                      rat_h2oco2, rat_h2oco2_1, rat_h2oo3, rat_h2oo3_1, &
   470)                      rat_h2on2o, rat_h2on2o_1, rat_h2och4, rat_h2och4_1, &
   471)                      rat_n2oco2, rat_n2oco2_1, rat_o3co2, rat_o3co2_1, &
   472)                      selffac, selffrac, indself, forfac, forfrac, indfor, &
   473)                      minorfrac, scaleminor, scaleminorn2, indminor, &
   474)                      fracs, taug)
   475) 
   476) 
   477) 
   478) ! Combine gaseous and aerosol optical depths, if aerosol active
   479)          if (iaer .eq. 0) then
   480)             do k = 1, nlay
   481)                do ig = 1, ngptlw 
   482)                   taut(k,ig) = taug(k,ig)
   483)                enddo
   484)             enddo
   485)          elseif (iaer .eq. 10) then
   486)             do k = 1, nlay
   487)                do ig = 1, ngptlw 
   488)                   taut(k,ig) = taug(k,ig) + taua(k,ngb(ig))
   489)                enddo
   490)             enddo
   491)          endif
   492) 
   493) ! Call the radiative transfer routine.
   494) ! Either routine can be called to do clear sky calculation.  If clouds
   495) ! are present, then select routine based on cloud overlap assumption
   496) ! to be used.  Clear sky calculation is done simultaneously.
   497) ! For McICA, RTRNMC is called for clear and cloudy calculations.
   498) 
   499)          call rtrnmc(nlay, istart, iend, iout, pz, semiss, ncbands, &
   500)                      cldfmc, taucmc, planklay, planklev, plankbnd, &
   501)                      pwvcm, fracs, taut, &
   502)                      totuflux, totdflux, fnet, htr, &
   503)                      totuclfl, totdclfl, fnetc, htrc, totufluxs, totdfluxs )
   504) 
   505) !  Transfer up and down fluxes and heating rate to output arrays.
   506) !  Vertical indexing goes from bottom to top
   507) 
   508)          do k = 0, nlay
   509)             uflx(iplon,k+1) = totuflux(k)
   510)             dflx(iplon,k+1) = totdflux(k)
   511)             uflxc(iplon,k+1) = totuclfl(k)
   512)             dflxc(iplon,k+1) = totdclfl(k)
   513)             uflxs(:,iplon,k+1) = totufluxs(:,k)
   514)             dflxs(:,iplon,k+1) = totdfluxs(:,k)
   515)          enddo
   516)          do k = 0, nlay-1
   517)             hr(iplon,k+1) = htr(k)
   518)             hrc(iplon,k+1) = htrc(k)
   519)          enddo
   520) 
   521)       enddo
   522) 
   523)       end subroutine rrtmg_lw
   524) 
   525) !***************************************************************************
   526)       subroutine inatm (iplon, nlay, icld, iaer, &
   527)               play, plev, tlay, tlev, tsfc, h2ovmr, &
   528)               o3vmr, co2vmr, ch4vmr, o2vmr, n2ovmr, cfc11vmr, cfc12vmr, &
   529)               cfc22vmr, ccl4vmr, emis, inflglw, iceflglw, liqflglw, &
   530)               cldfmcl, taucmcl, ciwpmcl, clwpmcl, reicmcl, relqmcl, tauaer, &
   531)               pavel, pz, tavel, tz, tbound, semiss, coldry, &
   532)               wkl, wbrodl, wx, pwvcm, inflag, iceflag, liqflag, &
   533)               cldfmc, taucmc, ciwpmc, clwpmc, reicmc, dgesmc, relqmc, taua)
   534) !***************************************************************************
   535) !
   536) !  Input atmospheric profile from GCM, and prepare it for use in RRTMG_LW.
   537) !  Set other RRTMG_LW input parameters.  
   538) !
   539) !***************************************************************************
   540) 
   541) ! --------- Modules ----------
   542) 
   543)       use parrrtm, only : nbndlw, ngptlw, nmol, maxxsec, mxmol
   544)       use rrlw_con, only: fluxfac, heatfac, oneminus, pi, grav, avogad
   545)       use rrlw_wvn, only: ng, nspa, nspb, wavenum1, wavenum2, delwave, ixindx
   546) 
   547) ! ------- Declarations -------
   548) 
   549) ! ----- Input -----
   550)       integer, intent(in) :: iplon                      ! column loop index
   551)       integer, intent(in) :: nlay                       ! Number of model layers
   552)       integer, intent(in) :: icld                       ! clear/cloud and cloud overlap flag
   553)       integer, intent(in) :: iaer                       ! aerosol option flag
   554) 
   555)       real(kind=r8), intent(in) :: play(:,:)            ! Layer pressures (hPa, mb)
   556)                                                         !    Dimensions: (ncol,nlay)
   557)       real(kind=r8), intent(in) :: plev(:,:)            ! Interface pressures (hPa, mb)
   558)                                                         !    Dimensions: (ncol,nlay+1)
   559)       real(kind=r8), intent(in) :: tlay(:,:)            ! Layer temperatures (K)
   560)                                                         !    Dimensions: (ncol,nlay)
   561)       real(kind=r8), intent(in) :: tlev(:,:)            ! Interface temperatures (K)
   562)                                                         !    Dimensions: (ncol,nlay+1)
   563)       real(kind=r8), intent(in) :: tsfc(:)              ! Surface temperature (K)
   564)                                                         !    Dimensions: (ncol)
   565)       real(kind=r8), intent(in) :: h2ovmr(:,:)          ! H2O volume mixing ratio
   566)                                                         !    Dimensions: (ncol,nlay)
   567)       real(kind=r8), intent(in) :: o3vmr(:,:)           ! O3 volume mixing ratio
   568)                                                         !    Dimensions: (ncol,nlay)
   569)       real(kind=r8), intent(in) :: co2vmr(:,:)          ! CO2 volume mixing ratio
   570)                                                         !    Dimensions: (ncol,nlay)
   571)       real(kind=r8), intent(in) :: ch4vmr(:,:)          ! Methane volume mixing ratio
   572)                                                         !    Dimensions: (ncol,nlay)
   573)       real(kind=r8), intent(in) :: o2vmr(:,:)           ! O2 volume mixing ratio
   574)                                                         !    Dimensions: (ncol,nlay)
   575)       real(kind=r8), intent(in) :: n2ovmr(:,:)          ! Nitrous oxide volume mixing ratio
   576)                                                         !    Dimensions: (ncol,nlay)
   577)       real(kind=r8), intent(in) :: cfc11vmr(:,:)        ! CFC11 volume mixing ratio
   578)                                                         !    Dimensions: (ncol,nlay)
   579)       real(kind=r8), intent(in) :: cfc12vmr(:,:)        ! CFC12 volume mixing ratio
   580)                                                         !    Dimensions: (ncol,nlay)
   581)       real(kind=r8), intent(in) :: cfc22vmr(:,:)        ! CFC22 volume mixing ratio
   582)                                                         !    Dimensions: (ncol,nlay)
   583)       real(kind=r8), intent(in) :: ccl4vmr(:,:)         ! CCL4 volume mixing ratio
   584)                                                         !    Dimensions: (ncol,nlay)
   585)       real(kind=r8), intent(in) :: emis(:,:)            ! Surface emissivity
   586)                                                         !    Dimensions: (ncol,nbndlw)
   587) 
   588)       integer, intent(in) :: inflglw                    ! Flag for cloud optical properties
   589)       integer, intent(in) :: iceflglw                   ! Flag for ice particle specification
   590)       integer, intent(in) :: liqflglw                   ! Flag for liquid droplet specification
   591) 
   592)       real(kind=r8), intent(in) :: cldfmcl(:,:,:)       ! Cloud fraction
   593)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   594)       real(kind=r8), intent(in) :: ciwpmcl(:,:,:)       ! Cloud ice water path (g/m2)
   595)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   596)       real(kind=r8), intent(in) :: clwpmcl(:,:,:)       ! Cloud liquid water path (g/m2)
   597)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   598)       real(kind=r8), intent(in) :: reicmcl(:,:)         ! Cloud ice effective radius (microns)
   599)                                                         !    Dimensions: (ncol,nlay)
   600)       real(kind=r8), intent(in) :: relqmcl(:,:)         ! Cloud water drop effective radius (microns)
   601)                                                         !    Dimensions: (ncol,nlay)
   602)       real(kind=r8), intent(in) :: taucmcl(:,:,:)       ! Cloud optical depth
   603)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   604)       real(kind=r8), intent(in) :: tauaer(:,:,:)        ! Aerosol optical depth
   605)                                                         !    Dimensions: (ncol,nlay,nbndlw)
   606) 
   607) ! ----- Output -----
   608) ! Atmosphere
   609)       real(kind=r8), intent(out) :: pavel(:)            ! layer pressures (mb) 
   610)                                                         !    Dimensions: (nlay)
   611)       real(kind=r8), intent(out) :: tavel(:)            ! layer temperatures (K)
   612)                                                         !    Dimensions: (nlay)
   613)       real(kind=r8), intent(out) :: pz(0:)              ! level (interface) pressures (hPa, mb)
   614)                                                         !    Dimensions: (0:nlay)
   615)       real(kind=r8), intent(out) :: tz(0:)              ! level (interface) temperatures (K)
   616)                                                         !    Dimensions: (0:nlay)
   617)       real(kind=r8), intent(out) :: tbound              ! surface temperature (K)
   618)       real(kind=r8), intent(out) :: coldry(:)           ! dry air column density (mol/cm2)
   619)                                                         !    Dimensions: (nlay)
   620)       real(kind=r8), intent(out) :: wbrodl(:)           ! broadening gas column density (mol/cm2)
   621)                                                         !    Dimensions: (nlay)
   622)       real(kind=r8), intent(out) :: wkl(:,:)            ! molecular amounts (mol/cm-2)
   623)                                                         !    Dimensions: (mxmol,nlay)
   624)       real(kind=r8), intent(out) :: wx(:,:)             ! cross-section amounts (mol/cm-2)
   625)                                                         !    Dimensions: (maxxsec,nlay)
   626)       real(kind=r8), intent(out) :: pwvcm               ! precipitable water vapor (cm)
   627)       real(kind=r8), intent(out) :: semiss(:)           ! lw surface emissivity
   628)                                                         !    Dimensions: (nbndlw)
   629) 
   630) ! Atmosphere/clouds - cldprop
   631)       integer, intent(out) :: inflag                    ! flag for cloud property method
   632)       integer, intent(out) :: iceflag                   ! flag for ice cloud properties
   633)       integer, intent(out) :: liqflag                   ! flag for liquid cloud properties
   634) 
   635)       real(kind=r8), intent(out) :: cldfmc(:,:)         ! cloud fraction [mcica]
   636)                                                         !    Dimensions: (ngptlw,nlay)
   637)       real(kind=r8), intent(out) :: ciwpmc(:,:)         ! cloud ice water path [mcica]
   638)                                                         !    Dimensions: (ngptlw,nlay)
   639)       real(kind=r8), intent(out) :: clwpmc(:,:)         ! cloud liquid water path [mcica]
   640)                                                         !    Dimensions: (ngptlw,nlay)
   641)       real(kind=r8), intent(out) :: relqmc(:)           ! liquid particle effective radius (microns)
   642)                                                         !    Dimensions: (nlay)
   643)       real(kind=r8), intent(out) :: reicmc(:)           ! ice particle effective radius (microns)
   644)                                                         !    Dimensions: (nlay)
   645)       real(kind=r8), intent(out) :: dgesmc(:)           ! ice particle generalized effective size (microns)
   646)                                                         !    Dimensions: (nlay)
   647)       real(kind=r8), intent(out) :: taucmc(:,:)         ! cloud optical depth [mcica]
   648)                                                         !    Dimensions: (ngptlw,nlay)
   649)       real(kind=r8), intent(out) :: taua(:,:)           ! Aerosol optical depth
   650)                                                         ! Dimensions: (nlay,nbndlw)
   651) 
   652) 
   653) ! ----- Local -----
   654)       real(kind=r8), parameter :: amd = 28.9660_r8      ! Effective molecular weight of dry air (g/mol)
   655)       real(kind=r8), parameter :: amw = 18.0160_r8      ! Molecular weight of water vapor (g/mol)
   656) !      real(kind=r8), parameter :: amc = 44.0098_r8      ! Molecular weight of carbon dioxide (g/mol)
   657) !      real(kind=r8), parameter :: amo = 47.9998_r8      ! Molecular weight of ozone (g/mol)
   658) !      real(kind=r8), parameter :: amo2 = 31.9999_r8     ! Molecular weight of oxygen (g/mol)
   659) !      real(kind=r8), parameter :: amch4 = 16.0430_r8    ! Molecular weight of methane (g/mol)
   660) !      real(kind=r8), parameter :: amn2o = 44.0128_r8    ! Molecular weight of nitrous oxide (g/mol)
   661) !      real(kind=r8), parameter :: amc11 = 137.3684_r8   ! Molecular weight of CFC11 (g/mol) - CCL3F
   662) !      real(kind=r8), parameter :: amc12 = 120.9138_r8   ! Molecular weight of CFC12 (g/mol) - CCL2F2
   663) !      real(kind=r8), parameter :: amc22 = 86.4688_r8    ! Molecular weight of CFC22 (g/mol) - CHCLF2
   664) !      real(kind=r8), parameter :: amcl4 = 153.823_r8    ! Molecular weight of CCL4 (g/mol) - CCL4
   665) 
   666) ! Set molecular weight ratios (for converting mmr to vmr)
   667) !  e.g. h2ovmr = h2ommr * amdw)
   668)       real(kind=r8), parameter :: amdw = 1.607793_r8    ! Molecular weight of dry air / water vapor
   669)       real(kind=r8), parameter :: amdc = 0.658114_r8    ! Molecular weight of dry air / carbon dioxide
   670)       real(kind=r8), parameter :: amdo = 0.603428_r8    ! Molecular weight of dry air / ozone
   671)       real(kind=r8), parameter :: amdm = 1.805423_r8    ! Molecular weight of dry air / methane
   672)       real(kind=r8), parameter :: amdn = 0.658090_r8    ! Molecular weight of dry air / nitrous oxide
   673)       real(kind=r8), parameter :: amdc1 = 0.210852_r8   ! Molecular weight of dry air / CFC11
   674)       real(kind=r8), parameter :: amdc2 = 0.239546_r8   ! Molecular weight of dry air / CFC12
   675) 
   676)       real(kind=r8), parameter :: sbc = 5.67e-08_r8     ! Stefan-Boltzmann constant (W/m2K4)
   677) 
   678)       integer :: isp, l, ix, n, imol, ib, ig            ! Loop indices
   679)       real(kind=r8) :: amm, amttl, wvttl, wvsh, summol  
   680) 
   681) !  Initialize all molecular amounts and cloud properties to zero here, then pass input amounts
   682) !  into RRTM arrays below.
   683) 
   684)       wkl(:,:) = 0.0_r8
   685)       wx(:,:) = 0.0_r8
   686)       cldfmc(:,:) = 0.0_r8
   687)       taucmc(:,:) = 0.0_r8
   688)       ciwpmc(:,:) = 0.0_r8
   689)       clwpmc(:,:) = 0.0_r8
   690)       reicmc(:) = 0.0_r8
   691)       dgesmc(:) = 0.0_r8
   692)       relqmc(:) = 0.0_r8
   693)       taua(:,:) = 0.0_r8
   694)       amttl = 0.0_r8
   695)       wvttl = 0.0_r8
   696)  
   697) !  Set surface temperature.
   698)       tbound = tsfc(iplon)
   699) 
   700) !  Install input GCM arrays into RRTMG_LW arrays for pressure, temperature,
   701) !  and molecular amounts.  
   702) !  Pressures are input in mb, or are converted to mb here.
   703) !  Molecular amounts are input in volume mixing ratio, or are converted from 
   704) !  mass mixing ratio (or specific humidity for h2o) to volume mixing ratio
   705) !  here. These are then converted to molecular amount (molec/cm2) below.  
   706) !  The dry air column COLDRY (in molec/cm2) is calculated from the level 
   707) !  pressures, pz (in mb), based on the hydrostatic equation and includes a 
   708) !  correction to account for h2o in the layer.  The molecular weight of moist 
   709) !  air (amm) is calculated for each layer.  
   710) !  Note: In RRTMG, layer indexing goes from bottom to top, and coding below
   711) !  assumes GCM input fields are also bottom to top. Input layer indexing
   712) !  from GCM fields should be reversed here if necessary.
   713) 
   714)       pz(0) = plev(iplon,nlay+1)
   715)       tz(0) = tlev(iplon,nlay+1)
   716)       do l = 1, nlay
   717)          pavel(l) = play(iplon,nlay-l+1)
   718)          tavel(l) = tlay(iplon,nlay-l+1)
   719)          pz(l) = plev(iplon,nlay-l+1)
   720)          tz(l) = tlev(iplon,nlay-l+1)
   721) ! For h2o input in vmr:
   722)          wkl(1,l) = h2ovmr(iplon,nlay-l+1)
   723) ! For h2o input in mmr:
   724) !         wkl(1,l) = h2o(iplon,nlay-l)*amdw
   725) ! For h2o input in specific humidity;
   726) !         wkl(1,l) = (h2o(iplon,nlay-l)/(1._r8 - h2o(iplon,nlay-l)))*amdw
   727)          wkl(2,l) = co2vmr(iplon,nlay-l+1)
   728)          wkl(3,l) = o3vmr(iplon,nlay-l+1)
   729)          wkl(4,l) = n2ovmr(iplon,nlay-l+1)
   730)          wkl(6,l) = ch4vmr(iplon,nlay-l+1)
   731)          wkl(7,l) = o2vmr(iplon,nlay-l+1)
   732) 
   733)          amm = (1._r8 - wkl(1,l)) * amd + wkl(1,l) * amw            
   734) 
   735)          coldry(l) = (pz(l-1)-pz(l)) * 1.e3_r8 * avogad / &
   736)                      (1.e2_r8 * grav * amm * (1._r8 + wkl(1,l)))
   737) 
   738) ! Set cross section molecule amounts from input; convert to vmr if necessary
   739)          wx(1,l) = ccl4vmr(iplon,nlay-l+1)
   740)          wx(2,l) = cfc11vmr(iplon,nlay-l+1)
   741)          wx(3,l) = cfc12vmr(iplon,nlay-l+1)
   742)          wx(4,l) = cfc22vmr(iplon,nlay-l+1)
   743) 
   744)       enddo
   745) 
   746)       coldry(nlay) = (pz(nlay-1)) * 1.e3_r8 * avogad / &
   747)                         (1.e2_r8 * grav * amm * (1._r8 + wkl(1,nlay-1)))
   748) 
   749) ! At this point all molecular amounts in wkl and wx are in volume mixing ratio; 
   750) ! convert to molec/cm2 based on coldry for use in rrtm.  also, compute precipitable
   751) ! water vapor for diffusivity angle adjustments in rtrn and rtrnmr.
   752) 
   753)       do l = 1, nlay
   754)          summol = 0.0_r8
   755)          do imol = 2, nmol
   756)             summol = summol + wkl(imol,l)
   757)          enddo
   758)          wbrodl(l) = coldry(l) * (1._r8 - summol)
   759)          do imol = 1, nmol
   760)             wkl(imol,l) = coldry(l) * wkl(imol,l)
   761)          enddo
   762)          amttl = amttl + coldry(l)+wkl(1,l)
   763)          wvttl = wvttl + wkl(1,l)
   764)          do ix = 1,maxxsec
   765)             if (ixindx(ix) .ne. 0) then
   766)                wx(ixindx(ix),l) = coldry(l) * wx(ix,l) * 1.e-20_r8
   767)             endif
   768)          enddo
   769)       enddo
   770) 
   771)       wvsh = (amw * wvttl) / (amd * amttl)
   772)       pwvcm = wvsh * (1.e3_r8 * pz(0)) / (1.e2_r8 * grav)
   773) 
   774) ! Set spectral surface emissivity for each longwave band.  
   775) 
   776)       do n=1,nbndlw
   777)          semiss(n) = emis(iplon,n)
   778) !          semiss(n) = 1.0_r8
