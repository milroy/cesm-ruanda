    57)   subroutine fstrat_inti( fstrat_file, fstrat_list )
    58)     !------------------------------------------------------------------
    59)     !	... initialize upper boundary values
    60)     !------------------------------------------------------------------
    61) 
    62)     use mo_constants,  only : d2r
    63)     use phys_grid,     only : get_ncols_p, get_rlat_all_p
    64) 
    65)     use time_manager,  only : get_step_size
    66)     use time_manager,  only : get_calday
    67)     use ioFileMod,     only : getfil
    68)     use spmd_utils,    only : masterproc
    69) #ifdef SPMD
    70)     use mpishorthand,  only : mpicom,mpiint,mpir8
    71) #endif
    72)     use mo_tracname,  only : solsym
    73)     use chem_mods,    only : gas_pcnst
    74)     use mo_chem_utls, only : get_spc_ndx, get_inv_ndx
    75)     use constituents, only : pcnst
    76)     use dyn_grid,     only : get_dyn_grid_parm
    77)     use interpolate_data, only : interp_type, lininterp_init, lininterp
    78)     implicit none
    79) 
    80)     character(len=*), intent(in) :: fstrat_file
    81)     character(len=*), intent(in) :: fstrat_list(:)
    82) 
    83)     !------------------------------------------------------------------
    84)     !	... local variables
    85)     !------------------------------------------------------------------
    86)     real(r8), parameter :: mb2pa = 100._r8
    87) 
    88)     integer :: i, j, nchar
    89)     integer :: spcno, lev, month, ierr
    90)     integer :: vid, ndims
    91)     type(file_desc_t) :: ncid
    92)     integer :: dimid_lat, dimid_lev, dimid_species, dimid_month
    93)     integer :: dimid(4)
    94)     integer :: start(4)
    95)     integer :: count(4)
    96)     integer, parameter :: dates(12) = (/ 116, 214, 316, 415,  516,  615, &
    97)          716, 816, 915, 1016, 1115, 1216 /)
    98)     integer :: ncols, c
    99)     real(r8), allocatable :: mr_ub_in(:,:,:,:)
   100)     real(r8), allocatable :: lat(:)
   101)     real(r8) :: to_lats(pcols)
   102)     character(len=80) :: attribute
   103)     character(len=8)  :: wrk_name
   104)     character(len=25), allocatable :: ub_species_names(:)
   105)     character(len=256) :: locfn
   106)     type(interp_type) :: lat_wgts
   107) 
   108) 
   109)     !-----------------------------------------------------------------------
   110)     !       ... get species indicies
   111)     !-----------------------------------------------------------------------    
   112)     no_ndx      = get_spc_ndx( 'NO' )
   113)     no2_ndx     = get_spc_ndx( 'NO2' )
   114)     sim_has_nox = no_ndx > 0 .or. no2_ndx > 0
   115)     ox_ndx      = get_spc_ndx( 'OX' )
   116)     if( ox_ndx < 1 ) then
   117)        ox_ndx = get_spc_ndx( 'O3' )
   118)     end if
   119)     o3s_ndx     = get_spc_ndx( 'O3S' )
   120)     o3inert_ndx = get_spc_ndx( 'O3INERT' )
   121)     o3rad_ndx   = get_spc_ndx( 'O3RAD' )
   122)     synoz_ndx   = get_spc_ndx( 'SYNOZ' )
   123)     o3a_ndx     = get_spc_ndx( 'O3A' )
   124)     xno_ndx      = get_spc_ndx( 'XNO' )
   125)     xno2_ndx      = get_spc_ndx( 'XNO2' )
   126) 
   127)     if (.not.((o3rad_ndx > 0) .eqv. (synoz_ndx > 0))) then
   128)        call endrun('fstrat_inti: Both SYNOZ and O3RAD are required in chemical mechanism.')
   129)     endif
   130)     if(masterproc) then
   131)        if (synoz_ndx > 0) then
   132)           if ( .not. any(fstrat_list == 'O3RAD') ) then
   133)              write(iulog,*) 'fstrat_inti: ***WARNING*** O3RAD is not include in fstrat_list namelist variable'
   134)           endif
   135)        else if (ox_ndx > 0) then
   136)           if ( .not. any(fstrat_list == 'O3') ) then
   137)              write(iulog,*) 'fstrat_inti: ***WARNING*** O3 is not include in fstrat_list namelist variable'
   138)           endif
   139)        endif
   140)     end if
   141)     h2o_ndx     = get_spc_ndx( 'H2O' )
   142)     if( h2o_ndx < 0 ) then
   143)        h2o_ndx  = get_inv_ndx( 'H2O' )
   144)     end if
   145) 
   146)     has_fstrat(:) = .false.
   147) 
   148)     do i = 1,pcnst
   149) 
   150)        if ( len_trim(fstrat_list(i))==0 ) exit
   151) 
   152)        j = get_spc_ndx(fstrat_list(i))
   153) 
   154)        if ( j > 0 ) then
   155)           has_fstrat(j) = .true.
   156)        else
   157)           write(iulog,*) 'fstrat_inti: '//trim(fstrat_list(i))//' is not included in species set'
   158)           call endrun('fstrat_inti: invalid fixed stratosphere species')
   159)        endif
   160) 
   161)     enddo
   162) 
   163)     if (.not. any(has_fstrat(:))) return
   164) 
   165)     !-----------------------------------------------------------------------
   166)     !       ... open netcdf file
   167)     !-----------------------------------------------------------------------
   168)     call getfil (fstrat_file, locfn, 0)
   169)     call cam_pio_openfile (ncid, trim(locfn), PIO_NOWRITE)
   170)     !-----------------------------------------------------------------------
   171)     !       ... get latitude
   172)     !-----------------------------------------------------------------------
   173)     ierr = pio_inq_dimid( ncid, 'lat', dimid_lat )
   174)     ierr = pio_inq_dimlen( ncid, dimid_lat, ub_nlat )
   175)     allocate( lat(ub_nlat), stat=ierr )
   176)     if( ierr /= 0 ) then
   177)        write(iulog,*) 'fstrat_inti: lat allocation error = ',ierr
   178)        call endrun
   179)     end if
   180)     ierr = pio_inq_varid( ncid, 'lat', vid )
   181)     ierr = pio_get_var( ncid, vid, lat )
   182)     lat(:ub_nlat) = lat(:ub_nlat) * d2r
   183) 
   184)     if( ierr /= 0 ) then
   185)        write(iulog,*) 'fstrat_inti: failed to deallocate lat; ierr = ',ierr
   186)        call endrun
   187)     end if
   188) 
   189)     !-----------------------------------------------------------------------
   190)     !       ... get vertical coordinate (if necessary, convert units to pa)
   191)     !-----------------------------------------------------------------------
   192)     ierr = pio_inq_dimid( ncid, 'lev', dimid_lev )
   193)     ierr = pio_inq_dimlen( ncid, dimid_lev, ub_nlevs )
   194)     allocate( ub_plevs(ub_nlevs), stat=ierr )
   195)     if( ierr /= 0 ) then
   196)        write(iulog,*) 'fstrat_inti: ub_plevs allocation error = ',ierr
   197)        call endrun
   198)     end if
   199)     ierr = pio_inq_varid( ncid, 'lev', vid )
   200)     ierr = pio_get_var( ncid, vid, ub_plevs )
   201)     attribute(:) = ' '
   202)     call pio_seterrorhandling(ncid, pio_BCAST_error)
   203)     ierr = pio_get_att( ncid, vid, 'units', attribute )
   204)     call pio_seterrorhandling(ncid, pio_INTERNAL_error)
   205)     if( ierr == PIO_noerr )then
   206)        if( trim(attribute) == 'mb' .or. trim(attribute) == 'hpa' )then
   207)           if (masterproc) write(iulog,*) 'fstrat_inti: units for lev = ',trim(attribute),'... converting to pa'
   208)           ub_plevs(:) = mb2pa * ub_plevs(:)
   209)        else if( trim(attribute) /= 'pa' .and. trim(attribute) /= 'pa' )then
   210)           write(iulog,*) 'fstrat_inti: unknown units for lev, units=*',trim(attribute),'*'
   211)           write(iulog,*) 'fstrat_inti: ',attribute=='mb',trim(attribute)=='mb',attribute(1:2)=='mb'
   212)           call endrun
   213)        end if
   214)     else
   215)        write(iulog,*) 'fstrat_inti: warning! units attribute for lev missing, assuming mb'
   216)        ub_plevs(:) = mb2pa * ub_plevs(:)
   217)     end if
   218)     !-----------------------------------------------------------------------
   219)     !       ... get time and species dimensions
   220)     !-----------------------------------------------------------------------
   221)     ierr = pio_inq_dimid( ncid, 'month', dimid_month )
   222)     ierr = pio_inq_dimlen( ncid, dimid_month, ub_nmonth )
   223)     if( ub_nmonth /= 12 )then
   224)        write(iulog,*) 'fstrat_inti: error! number of months = ',ub_nmonth,', expecting 12'
   225)        call endrun
   226)     end if
   227)     ierr = pio_inq_dimid( ncid, 'species', dimid_species )
   228)     ierr = pio_inq_dimlen( ncid, dimid_species, ub_nspecies )
   229) 
   230)     !------------------------------------------------------------------
   231)     !	... allocate arrays
   232)     !------------------------------------------------------------------
   233)     allocate( mr_ub_in(ub_nlat,ub_nspecies,ub_nmonth,ub_nlevs), stat=ierr )
   234)     if( ierr /= 0 ) then
   235)        write(iulog,*) 'fstrat_inti: mr_ub_in allocation error = ',ierr
   236)        call endrun
   237)     end if
   238)     allocate( mr_ub(pcols,ub_nspecies,ub_nmonth,ub_nlevs,begchunk:endchunk), stat=ierr )
   239)     if( ierr /= 0 ) then
   240)        write(iulog,*) 'fstrat_inti: mr_ub allocation error = ',ierr
   241)        call endrun
   242)     end if
   243) 
   244)     !------------------------------------------------------------------
   245)     !	... read in the species names
   246)     !------------------------------------------------------------------
   247) 
   248)     ierr = pio_inq_varid( ncid, 'specname', vid )
   249)     ierr = pio_inq_vardimid( ncid, vid, dimid )
   250)     ierr = pio_inq_dimlen( ncid, dimid(1), nchar )
   251)     allocate( ub_species_names(ub_nspecies), stat=ierr )
   252)     if( ierr /= 0 ) then
   253)        write(iulog,*) 'fstrat_inti: ub_species_names allocation error = ',ierr
   254)        call endrun
   255)     end if
   256)     allocate( map(ub_nspecies), stat=ierr )
   257)     if( ierr /= 0 ) then
   258)        write(iulog,*) 'fstrat_inti: map allocation error = ',ierr
   259)        call endrun
   260)     end if
   261) 
   262)     table_loop :  do i = 1,ub_nspecies
   263)        start(:2) = (/ 1, i /)
   264)        count(:2) = (/ nchar, 1 /)
   265)        ub_species_names(i)(:) = ' '
   266)        ierr = pio_get_var( ncid, vid, start(:2), count(:2), ub_species_names(i:i))
   267)        if( trim(ub_species_names(i)) == 'NOX' ) then
   268)           table_nox_ndx = i
   269)        else if( trim(ub_species_names(i)) == 'H2O' ) then
   270)           table_h2o_ndx = i
   271)        else if( trim(ub_species_names(i)) == 'OX' ) then
   272)           table_ox_ndx = i
   273)        else if( trim(ub_species_names(i)) == 'SYNOZ' ) then
   274)           table_synoz_ndx = i
   275)        end if
   276)        map(i) = 0
   277)        do j = 1,gas_pcnst 
   278)           if( trim(ub_species_names(i)) == trim(solsym(j)) ) then
   279)              if( has_fstrat(j) ) then 
   280)                 map(i) = j
   281)                 if( masterproc ) write(iulog,*) 'fstrat_inti: '//trim(solsym(j))//' is fixed in stratosphere'
   282)                 exit
   283)              end if
   284)           endif
   285)        end do
   286)        if( map(i) == 0 ) then
   287)           if( trim(ub_species_names(i)) == 'OX' ) then
   288)              if( o3rad_ndx > 0 ) then
   289)                 wrk_name = 'O3RAD'
   290)              else
   291)                 wrk_name = 'O3'
   292)              end if
   293)              do j = 1,gas_pcnst
   294)                 if( trim(wrk_name) == trim(solsym(j)) ) then
   295)                    if( has_fstrat(j) ) then 
   296)                       if( masterproc ) write(iulog,*) 'fstrat_inti: '//trim(solsym(j))//' is fixed in stratosphere'
   297)                       map(i) = j
   298)                       exit
   299)                    end if
   300)                 end if
   301)              end do
   302)              if( map(i) == 0 ) then
   303)                 write(iulog,*) 'fstrat_inti: ubc table species ',trim(ub_species_names(i)), ' not used'
   304)              end if
   305)           else if( (trim(ub_species_names(i)) /= 'NOX') .and. (trim(ub_species_names(i)) /= 'H2O') ) then
   306)              write(iulog,*) 'fstrat_inti: ubc table species ',trim(ub_species_names(i)), ' not used'
   307)           end if
   308)        end if
   309)     end do table_loop
   310) 
   311)     if( table_nox_ndx > 0 ) then
   312)        if ( any(fstrat_list(:) == 'NO') .or. any(fstrat_list(:) == 'NO2') ) then
   313)           map(table_nox_ndx) = gas_pcnst + 1
   314)        else
   315)           write(iulog,*) 'fstrat_inti: ubc table species ',trim(ub_species_names(table_nox_ndx)), ' not used'
   316)        endif
   317)     end if
   318)     if( table_h2o_ndx > 0 ) then
   319)        if ( h2o_ndx > 0 ) then
   320)           map(table_h2o_ndx) = gas_pcnst + 2
   321)        else
   322)           write(iulog,*) 'fstrat_inti: ubc table species ',trim(ub_species_names(table_h2o_ndx)), ' not used'
   323)        endif
   324)     end if
   325) 
   326)     if (masterproc) write(iulog,*) 'fstrat_inti: h2o_ndx, table_h2o_ndx  = ', h2o_ndx, table_h2o_ndx
   327) 
   328)     !------------------------------------------------------------------
   329)     !	... check dimensions for vmr variable
   330)     !------------------------------------------------------------------
   331)     ierr = pio_inq_varid( ncid, 'vmr', vid )
   332)     ierr = pio_inq_varndims( ncid, vid, ndims )
   333)     if( ndims /= 4 ) then
   334)        write(iulog,*) 'fstrat_inti: error! variable vmr has ndims = ',ndims,', expecting 4'
   335)        call endrun
   336)     end if
   337)     ierr = pio_inq_vardimid( ncid, vid, dimid )
   338)     if( dimid(1) /= dimid_lat .or. dimid(2) /= dimid_species .or. &
   339)          dimid(3) /= dimid_month .or. dimid(4) /= dimid_lev )then
   340)        write(iulog,*) 'fstrat_inti: error! dimensions in wrong order for variable vmr,'// &
   341)             'expecting (lat,species,month,lev)'
   342)        call endrun
   343)     end if
   344) 
   345)     !------------------------------------------------------------------
   346)     !	... read in the ub mixing ratio values
   347)     !------------------------------------------------------------------
   348)     start = (/ 1, 1, 1, 1 /)
   349)     count = (/ ub_nlat, ub_nspecies, ub_nmonth, ub_nlevs /)
   350) 
   351)     ierr = pio_get_var( ncid, vid, start, count, mr_ub_in )
   352)     call pio_closefile (ncid)
   353)     !--------------------------------------------------------------------
   354)     !	... regrid
   355)     !--------------------------------------------------------------------
   356)     do c=begchunk,endchunk
   357)        ncols = get_ncols_p(c)
   358)        call get_rlat_all_p(c, pcols, to_lats)
   359)        call lininterp_init(lat,ub_nlat, to_lats, ncols, 1, lat_wgts)
   360) 
   361)        do lev = 1,ub_nlevs
   362)           do month = 1,ub_nmonth
   363)              do spcno = 1,ub_nspecies
   364)                 if( map(spcno) > 0 ) then
   365) 
   366)                    call lininterp(mr_ub_in(:,spcno, month, lev), ub_nlat, mr_ub(:,spcno,month,lev,c), &
   367)                         ncols, lat_wgts)
   368) 
   369)                    !                call regrid_1d( mr_ub_in(:,spcno,month,lev), mr_ub(:,spcno,month,lev), gndx, &
   370)                    !                     do_lat=.true.,to_lat_min=1, to_lat_max=plat )
   371) #ifdef DEBUG
   372)                    if( lev == 25 .and. month == 1 .and. spcno == 1 ) then
   373)                       write(iulog,*) 'mr_ub_in='
   374)                       write(iulog,'(10f7.1)') mr_ub_in(:,spcno,month,lev)*1.e9_r8
   375)                       write(iulog,*) 'mr_ub='
   376)                       write(iulog,'(10f7.1)') mr_ub(:,spcno,month,lev)*1.e9_r8
   377)                    end if
   378) #endif
   379)                    !                mr_ub(1,spcno,month,lev) = mr_ub(2,spcno,month,lev)
   380)                    !                mr_ub(plat,spcno,month,lev) = mr_ub(plat-1,spcno,month,lev)
   381) 
   382)                 end if
   383) 
   384)              end do
   385)           end do
   386)        end do
   387)     end do
   388)     !--------------------------------------------------------
   389)     !	... initialize the monthly day of year times
   390)     !--------------------------------------------------------
   391)     do month = 1,12
   392)        days(month) = get_calday( dates(month), 0 )
   393)     end do
   394) 
   395)     !--------------------------------------------------------
   396)     !   	... set up the relaxation for lower stratosphere
   397)     !--------------------------------------------------------
   398)     ! 	... taurelax = relaxation timescale (in sec)
   399)     !           facrelax = fractional relaxation towards ubc
   400)     !            1 => use ubc
   401)     !            0 => ignore ubc, use model concentrations
   402)     !--------------------------------------------------------
   403)     dtime = get_step_size()
   404)     facrelax = 1._r8 - exp( -real(dtime)/taurelax )
   405) 
   406)     !--------------------------------------------------------
   407)     ! 	... setup conserving interp for OX
   408)     !--------------------------------------------------------
   409)     if( table_ox_ndx > 0 ) then
   410)        allocate( ub_plevse(ub_nlevs-1), stat=ierr )
   411)        if( ierr /= 0 ) then
   412)           write(iulog,*) 'fstrat_inti: ub_plevse allocation error = ',ierr
   413)           call endrun
   414)        end if
   415)        ub_plevse(1:ub_nlevs-1) = .5_r8*(ub_plevs(1:ub_nlevs-1) + ub_plevs(2:ub_nlevs))
   416)     end if
   417) 
   418)   end subroutine fstrat_inti

   420)   subroutine set_fstrat_vals( vmr, pmid, pint, ltrop, calday, ncol,lchnk )
   421) 
   422)     !--------------------------------------------------------------------
   423)     !	... set the upper boundary values for :
   424)     !           ox, nox, hno3, ch4, co, n2o, n2o5 & stratospheric o3
   425)     !--------------------------------------------------------------------
   426) 
   427)     use mo_synoz, only : synoz_region => po3
   428) 
   429)     implicit none
   430) 
   431)     !--------------------------------------------------------------------
   432)     !	... dummy args
   433)     !--------------------------------------------------------------------
   434)     integer,  intent(in)    :: lchnk             ! chunk number
   435)     integer,  intent(in)    :: ncol              ! columns in chunk
   436)     integer,  intent(in)    :: ltrop(pcols)      ! tropopause vertical index
   437)     real(r8), intent(in)    :: calday            ! day of year including fraction
   438)     real(r8), intent(in)    :: pmid(pcols,pver)  ! midpoint pressure (Pa)
   439)     real(r8), intent(in)    :: pint(pcols,pverp) ! interface pressure (Pa)
   440)     real(r8), intent(inout) :: vmr(ncol,pver,gas_pcnst) ! species concentrations (mol/mol)
   441) 
   442)     !--------------------------------------------------------------------
   443)     !	... local variables
   444)     !--------------------------------------------------------------------
   445)     integer, parameter :: zlower = pver
   446)     real(r8), parameter    :: synoz_thres = 100.e-9_r8      ! synoz threshold
   447)     real(r8), parameter    :: o3rad_relax = 0.5_r8*86400._r8 ! 1/2 day relaxation constant
   448)     real(r8), parameter    :: synoz_relax = 2._r8*86400._r8 ! 2 day relaxation constant
   449)     real(r8), parameter    :: synoz_strat_relax = 5._r8*86400._r8 ! 5 day relaxation constant
   450) 
   451)     integer  ::  m, last, next, i, k, k1, km
   452)     integer  ::  astat
   453)     integer  ::  kmax(ncol)
   454)     integer  ::  levrelax
   455)     integer  ::  kl(ncol,zlower)
   456)     integer  ::  ku(ncol,zlower)
   457)     real(r8) ::  vmrrelax
   458)     real(r8) ::  fac_relax
   459)     real(r8) ::  pinterp
   460)     real(r8) ::  nox_ubc, xno, xno2, rno
   461)     real(r8) ::  dels
   462)     real(r8) ::  delp(ncol,zlower)
   463)     real(r8) ::  pint_vals(2)
   464)     real(r8), allocatable :: table_ox(:)
   465)     logical  ::  found_trop
   466)     integer  :: lat
   467) 
   468)     if (.not. any(has_fstrat(:))) return
   469) 
   470)     !--------------------------------------------------------
   471)     !	... setup the time interpolation
   472)     !--------------------------------------------------------
   473)     if( calday < days(1) ) then
   474)        next = 1
   475)        last = 12
   476)        dels = (365._r8 + calday - days(12)) / (365._r8 + days(1) - days(12))
   477)     else if( calday >= days(12) ) then
   478)        next = 1
   479)        last = 12
   480)        dels = (calday - days(12)) / (365._r8 + days(1) - days(12))
   481)     else
   482)        do m = 11,1,-1
   483)           if( calday >= days(m) ) then
   484)              exit
   485)           end if
   486)        end do
   487)        last = m
   488)        next = m + 1
   489)        dels = (calday - days(m)) / (days(m+1) - days(m))
   490)     end if
   491)     dels = max( min( 1._r8,dels ),0._r8 )
   492) 
   493)     !--------------------------------------------------------
   494)     !	... setup the pressure interpolation
   495)     !--------------------------------------------------------
   496)     do k = 1,zlower
   497)        do i = 1,ncol
   498)           if( pmid(i,k) <= ub_plevs(1) ) then
   499)              kl(i,k) = 1
   500)              ku(i,k) = 1
   501)              delp(i,k) = 0._r8
   502)           else if( pmid(i,k) >= ub_plevs(ub_nlevs) ) then
   503)              kl(i,k) = ub_nlevs
   504)              ku(i,k) = ub_nlevs
   505)              delp(i,k) = 0._r8
   506)           else
   507)              pinterp = pmid(i,k)
   508)              do k1 = 2,ub_nlevs
   509)                 if( pinterp <= ub_plevs(k1) ) then
   510)                    ku(i,k) = k1
   511)                    kl(i,k) = k1 - 1
   512)                    delp(i,k) = log( pinterp/ub_plevs(kl(i,k)) ) &
   513)                         / log( ub_plevs(ku(i,k))/ub_plevs(kl(i,k)) )
   514)                    exit
   515)                 end if
   516)              end do
   517)           end if
   518)        end do
   519)     end do
   520) 
   521)     !--------------------------------------------------------
   522)     !	... find max level less than 50 mb
   523)     !           fix UB vals from top of model to this level
   524)     !--------------------------------------------------------
   525)     do i = 1,ncol
   526)        do k = 2,pver
   527)           if( pmid(i,k) > 50.e2_r8 ) then
   528)              kmax(i) = k
   529)              exit
   530)           end if
   531)        end do
   532)     end do
   533) 
   534)     !--------------------------------------------------------
   535)     !	... setup for ox conserving interp
   536)     !--------------------------------------------------------
   537)     if( table_ox_ndx > 0 ) then
   538)        if( map(table_ox_ndx) > 0 ) then
   539)           allocate( table_ox(ub_nlevs-2),stat=astat )
   540)           if( astat /= 0 ) then
   541)              write(iulog,*) 'set_fstrat_vals: table_ox allocation error = ',astat
   542)              call endrun
   543)           end if
   544) #ifdef UB_DEBUG
   545)           write(iulog,*) ' '
   546)           write(iulog,*) 'set_fstrat_vals: ub_nlevs = ',ub_nlevs
   547)           write(iulog,*) 'set_fstrat_vals: ub_plevse'
   548)           write(iulog,'(1p5g15.7)') ub_plevse(:)
   549)           write(iulog,*) ' '
   550) #endif
   551)        end if
   552)     end if
   553) 
   554)     !--------------------------------------------------------
   555)     !	... set the mixing ratios at upper boundary
   556)     !--------------------------------------------------------
   557)     species_loop : do m = 1,ub_nspecies
   558)        ub_overwrite : if( map(m) > 0 ) then
   559)           if( m == table_ox_ndx ) then
   560)              do i = 1,ncol
   561) 
   562)                 table_ox(1:ub_nlevs-2) = mr_ub(i,m,last,2:ub_nlevs-1,lchnk) &
   563)                      + dels*(mr_ub(i,m,next,2:ub_nlevs-1,lchnk) &
   564)                      - mr_ub(i,m,last,2:ub_nlevs-1,lchnk))
   565) #ifdef UB_DEBUG
   566)                 write(iulog,*) 'set_fstrat_vals: table_ox @ lat = ',lat
   567)                 write(iulog,'(1p5g15.7)') table_ox(:)
   568)                 write(iulog,*) ' '
   569) #endif
   570) 
   571)                 km = kmax(i)
   572) #ifdef UB_DEBUG
   573)                 write(iulog,*) 'set_fstrat_vals: pint with km = ',km
   574)                 write(iulog,'(1p5g15.7)') pint(i,:km+1)
   575)                 write(iulog,*) ' '
   576)                 write(iulog,*) 'set_fstrat_vals: pmid with km = ',km
   577)                 write(iulog,'(1p5g15.7)') pmid(i,:km)
   578)                 write(iulog,*) ' '
   579) #endif
   580)                 call rebin( ub_nlevs-2, km, ub_plevse, pint(i,:km+1), table_ox, vmr(i,:km,map(m)) )
   581) #ifdef UB_DEBUG
   582)                 write(iulog,*) 'set_fstrat_vals: ub o3 @ lat = ',lat
   583)                 write(iulog,'(1p5g15.7)') vmr(i,:km,map(m))
   584) #endif
   585)              end do
   586)              deallocate( table_ox )
   587)              cycle species_loop
   588)           end if
   589)           do i = 1,ncol
   590)              do k = 1,kmax(i)
   591)                 pint_vals(1) = mr_ub(i,m,last,kl(i,k),lchnk) &
   592)                      + delp(i,k) &
   593)                      * (mr_ub(i,m,last,ku(i,k),lchnk) &
   594)                      - mr_ub(i,m,last,kl(i,k),lchnk))
   595)                 pint_vals(2) = mr_ub(i,m,next,kl(i,k),lchnk) &
   596)                      + delp(i,k) &
   597)                      * (mr_ub(i,m,next,ku(i,k),lchnk) &
   598)                      - mr_ub(i,m,next,kl(i,k),lchnk))
   599)                 if( m /= table_nox_ndx .and. m /= table_h2o_ndx .and. m /= table_synoz_ndx ) then
   600)                    vmr(i,k,map(m)) = pint_vals(1) &
   601)                         + dels * (pint_vals(2) - pint_vals(1))
   602)                 else if( m == table_nox_ndx .and. sim_has_nox ) then
   603) 
   604)                    nox_ubc = pint_vals(1) + dels * (pint_vals(2) - pint_vals(1))
   605)                    if( no_ndx > 0 ) then
   606)                       xno  = vmr(i,k,no_ndx)
   607)                    else
   608)                       xno  = 0._r8
   609)                    end if
   610)                    if( no2_ndx > 0 ) then
   611)                       xno2 = vmr(i,k,no2_ndx)
   612)                    else
   613)                       xno2 = 0._r8
   614)                    end if
   615)                    rno  = xno / (xno + xno2)
   616)                    if( no_ndx > 0 ) then
   617)                       vmr(i,k,no_ndx)  = rno * nox_ubc
   618)                    end if
   619)                    if( no2_ndx > 0 ) then
   620)                       vmr(i,k,no2_ndx) = (1._r8 - rno) * nox_ubc
   621)                    end if
   622)                 end if
   623)              end do
   624)           end do
   625)        end if ub_overwrite
   626)     end do species_loop
   627) 
   628)     col_loop2 : do i = 1,ncol
   629)        !--------------------------------------------------------
   630)        ! 	... relax lower stratosphere to extended ubc
   631)        !           check to make sure ubc is not being imposed too low
   632)        !           levrelax = lowest model level (highest pressure)
   633)        !                      in which to relax to ubc
   634)        !--------------------------------------------------------
   635)        levrelax = ltrop(i)
   636)        do while( pmid(i,levrelax) > ub_plevs(ub_nlevs) )
   637)           levrelax = levrelax - 1
   638)        end do
   639) #ifdef DEBUG
   640)        if( levrelax /= ltrop(i) ) then
   641)           write(iulog,*) 'warning -- raised ubc: ',lat,i,
   642)           ltrop(i)-1,nint(pmid(i,ltrop(i)-1)/mb2pa),'mb -->',
   643)           levrelax,nint(pmid(i,levrelax)/mb2pa),'mb'
   644)        end if
   645) #endif
   646)        level_loop2 : do k = kmax(i)+1,levrelax
   647)           if( sim_has_nox ) then
   648)              if( no_ndx > 0 ) then
   649)                 xno  = vmr(i,k,no_ndx)
   650)              else
   651)                 xno  = 0._r8
   652)              end if
   653)              if( no2_ndx > 0 ) then
   654)                 xno2 = vmr(i,k,no2_ndx)
   655)              else
   656)                 xno2 = 0._r8
   657)              end if
   658)              rno     = xno / (xno + xno2)
   659)              nox_ubc = xno + xno2
   660)           end if
   661)           do m = 1,ub_nspecies
   662)              if( map(m) < 1 ) then
   663)                 cycle
   664)              end if
   665)              pint_vals(1) = mr_ub(i,m,last,kl(i,k),lchnk) &
   666)                   + delp(i,k) &
   667)                   * (mr_ub(i,m,last,ku(i,k),lchnk) &
   668)                   - mr_ub(i,m,last,kl(i,k),lchnk))
   669)              pint_vals(2) = mr_ub(i,m,next,kl(i,k),lchnk) &
   670)                   + delp(i,k) &
   671)                   * (mr_ub(i,m,next,ku(i,k),lchnk) &
   672)                   - mr_ub(i,m,next,kl(i,k),lchnk))
   673)              vmrrelax = pint_vals(1) &
   674)                   + dels * (pint_vals(2) - pint_vals(1))
   675)              if( m /= table_nox_ndx .and. m /= table_h2o_ndx  .and. m /= table_synoz_ndx ) then
   676)                 vmr(i,k,map(m)) = vmr(i,k,map(m)) &
   677)                      + (vmrrelax - vmr(i,k,map(m))) * facrelax
   678)              else if( m == table_nox_ndx .and. sim_has_nox) then
   679) 
   680)                 nox_ubc = nox_ubc + (vmrrelax - nox_ubc) * facrelax
   681)              end if
   682)           end do
   683)           if( sim_has_nox ) then
   684)              if( no_ndx > 0 ) then
   685)                 vmr(i,k,no_ndx)  = rno * nox_ubc
   686)              end if
   687)              if( no2_ndx > 0 ) then
   688)                 vmr(i,k,no2_ndx) = (1._r8 - rno) * nox_ubc
   689)              end if
   690)           end if
   691)        end do level_loop2
   692) 
   693)        has_synoz : if( synoz_ndx > 0 ) then
   694)           if ( synoz_ndx > 0 .and. table_synoz_ndx > 0 ) then
   695)              fac_relax = 1._r8 - exp( -real(dtime) / synoz_strat_relax )
   696)              do k = 1,pver
   697)                 m = table_synoz_ndx
   698)                 if ( synoz_region(i,k,lchnk) > 0._r8 ) then
   699)                    pint_vals(1) = mr_ub(i,m,last,kl(i,k),lchnk) &
   700)                         + delp(i,k) &
   701)                         * (mr_ub(i,m,last,ku(i,k),lchnk) &
   702)                         - mr_ub(i,m,last,kl(i,k),lchnk))
   703)                    pint_vals(2) = mr_ub(i,m,next,kl(i,k),lchnk) &
   704)                         + delp(i,k) &
   705)                         * (mr_ub(i,m,next,ku(i,k),lchnk) &
   706)                         - mr_ub(i,m,next,kl(i,k),lchnk))
   707)                    vmrrelax = pint_vals(1) &
   708)                         + dels * (pint_vals(2) - pint_vals(1))
   709)                    vmr(i,k,map(m)) = vmr(i,k,map(m)) &
   710)                         + (vmrrelax - vmr(i,k,map(m))) * fac_relax
   711)                 endif
   712)              enddo
   713)           endif
   714)  
   715)           !--------------------------------------------------------
   716)           ! 	... special assignments if synoz is present
   717)           !           update ox, o3s, o3inert in the stratosphere
   718)           !--------------------------------------------------------
   719)           if( ox_ndx > 0 ) then
   720)              do k = 1,levrelax
   721)                 if( vmr(i,k,synoz_ndx) >= synoz_thres ) then
   722)                    vmr(i,k,ox_ndx) = vmr(i,k,synoz_ndx)
   723)                 end if
   724)              end do
   725)           end if
   726)           if( o3s_ndx > 0 ) then
   727)              do k = 1,levrelax
   728)                 if( vmr(i,k,synoz_ndx) >= synoz_thres ) then
   729)                    vmr(i,k,o3s_ndx) = vmr(i,k,synoz_ndx)
   730)                 end if
   731)              end do
   732)           end if
   733)           if( o3rad_ndx > 0 .and. o3inert_ndx > 0 ) then
   734)              vmr(i,:ltrop(i),o3inert_ndx) = vmr(i,:ltrop(i),o3rad_ndx)
   735)           end if
   736)           !--------------------------------------------------------
   737)           ! 	... O3RAD is relaxed to climatology in the stratosphere
   738)           !           (done above) and OX in the troposphere
   739)           !--------------------------------------------------------
   740)           if( o3rad_ndx > 0 .and. ox_ndx > 0 ) then
   741)              fac_relax = 1._r8 - exp( -real(dtime) / o3rad_relax )
   742)              do k = levrelax+1,pver
   743)                 vmr(i,k,o3rad_ndx) = vmr(i,k,o3rad_ndx) &
   744)                      + (vmr(i,k,ox_ndx) - vmr(i,k,o3rad_ndx)) * fac_relax
   745)              end do
   746)           end if
   747)           !--------------------------------------------------------
   748)           ! 	... relax synoz to 25 ppbv in lower troposphere
   749)           !           (p > 500 hPa) with an e-fold time of 2 days
   750)           !           (Mc Linden et al., JGR, p14,660, 2000)
   751)           !--------------------------------------------------------
   752)           fac_relax = 1._r8 - exp( -real(dtime) / synoz_relax )
   753)           vmrrelax = 25.e-9_r8
   754)           do k = levrelax+2,pver
   755)              if( pmid(i,k) >= 50000._r8 ) then
   756)                 vmr(i,k,synoz_ndx) = vmr(i,k,synoz_ndx) &
   757)                      + (vmrrelax - vmr(i,k,synoz_ndx)) * fac_relax
   758)              end if
   759)           end do
   760)        else has_synoz
   761) 
   762)           !--------------------------------------------------------
   763)           !       ... set O3S and O3INERT to OX when no synoz
   764)           !--------------------------------------------------------
   765)           if( ox_ndx > 0 ) then
   766)              if( o3s_ndx > 0 ) then
   767)                 vmr(i,:ltrop(i),o3s_ndx)     = vmr(i,:ltrop(i),ox_ndx)
   768)              end if
   769)              if( o3inert_ndx > 0 ) then
   770)                 vmr(i,:ltrop(i),o3inert_ndx) = vmr(i,:ltrop(i),ox_ndx)
   771)              end if
   772)           end if
   773) 
   774)        end if has_synoz
   775) 
   776)        if ( o3a_ndx > 0 ) then
   777)           vmr(i,:ltrop(i),o3a_ndx) = (1._r8 - facrelax ) * vmr(i,:ltrop(i),o3a_ndx)
   778)        endif
   779)        if ( xno_ndx > 0 ) then
   780)           vmr(i,:ltrop(i),xno_ndx) = (1._r8 - facrelax ) * vmr(i,:ltrop(i),xno_ndx)
   781)        endif
   782)        if ( xno2_ndx > 0 ) then
   783)           vmr(i,:ltrop(i),xno2_ndx) = (1._r8 - facrelax ) * vmr(i,:ltrop(i),xno2_ndx)
   784)        endif
   785) 
   786)     end do col_loop2
   787) 
   788)   end subroutine set_fstrat_vals

   790)   subroutine set_fstrat_h2o( h2o, pmid, ltrop, calday, ncol, lchnk )
   791)     !--------------------------------------------------------------------

   795)

   797) 
   798)     !--------------------------------------------------------------------
   799)     !	... dummy args
   800)     !--------------------------------------------------------------------

   803)     integer,  intent(in)    :: lchnk
   804)     real(r8), intent(in)    :: calday            ! day of year including fraction

   819)     real(r8) ::  fac_relax
   820)     real(r8) ::  pinterp
   821)     real(r8) ::  dels
   822)     real(r8) ::  delp(ncol,zlower)
   823)     real(r8) ::  pint_vals(2)
   824)     logical  ::  found_trop
   825)     integer  ::  lat
   826) 
   827)     h2o_overwrite : if( h2o_ndx > 0 .and. table_h2o_ndx > 0 ) then
   828)        !--------------------------------------------------------
   829)        !	... setup the time interpolation
   830)        !--------------------------------------------------------

   833)           last = 12
   834)           dels = (365._r8 + calday - days(12)) / (365._r8 + days(1) - days(12))
   835)        else if( calday >= days(12) ) then
   836)           next = 1
   837)           last = 12
   838)           dels = (calday - days(12)) / (365._r8 + days(1) - days(12))
   839)        else
   840)           do m = 11,1,-1
   841)              if( calday >= days(m) ) then
   842)                 exit
   843)              end if
   844)           end do
   845)           last = m
   846)           next = m + 1
   847)           dels = (calday - days(m)) / (days(m+1) - days(m))
   848)        end if
   849)        dels = max( min( 1._r8,dels ),0._r8 )
   850) 
   851)        !--------------------------------------------------------
   852)        !	... setup the pressure interpolation
   853)        !--------------------------------------------------------
   854)        do k = 1,zlower
   855)           do i = 1,ncol
   856)              if( pmid(i,k) <= ub_plevs(1) ) then
   857)                 kl(i,k) = 1
   858)                 ku(i,k) = 1
   859)                 delp(i,k) = 0._r8
   860)              else if( pmid(i,k) >= ub_plevs(ub_nlevs) ) then
   861)                 kl(i,k) = ub_nlevs
   862)                 ku(i,k) = ub_nlevs
   863)                 delp(i,k) = 0._r8
   864)              else
   865)                 pinterp = pmid(i,k)
   866)                 do k1 = 2,ub_nlevs
   867)                    if( pinterp <= ub_plevs(k1) ) then
   868)                       ku(i,k) = k1
   869)                       kl(i,k) = k1 - 1
   870)                       delp(i,k) = log( pinterp/ub_plevs(kl(i,k)) ) &
   871)                            / log( ub_plevs(ku(i,k))/ub_plevs(kl(i,k)) )
   872)                       exit
   873)                    end if
   874)                 end do
   875)              end if
   876)           end do
   877)        end do
   878) 
   879)        !--------------------------------------------------------
   880)        !	... Find max level less than 50 mb
   881)        !           fix UB vals from top of model to this level
   882)        !--------------------------------------------------------

   884)           do k = 2,pver
   885)              if( pmid(i,k) > 50.e2_r8 ) then
   886)                 kmax(i) = k
   887)                 exit
   888)              end if
   889)           end do
   890)        end do
   891)        !--------------------------------------------------------
   892)        !	... set the mixing ratio at upper boundary
   893)        !--------------------------------------------------------
   894)        m = table_h2o_ndx
   895)        do i = 1,ncol
   896)           do k = 1,kmax(i)
   897)              pint_vals(1) = mr_ub(i,m,last,kl(i,k),lchnk) &
   898)                   + delp(i,k) &
   899)                   * (mr_ub(i,m,last,ku(i,k),lchnk) &
   900)                   - mr_ub(i,m,last,kl(i,k),lchnk))
   901)              pint_vals(2) = mr_ub(i,m,next,kl(i,k),lchnk) &
   902)                   + delp(i,k) &
   903)                   * (mr_ub(i,m,next,ku(i,k),lchnk) &
   904)                   - mr_ub(i,m,next,kl(i,k),lchnk))
   905)              h2o(i,k) = pint_vals(1) &
   906)                   + dels * (pint_vals(2) - pint_vals(1))
   907)           end do
   908)        end do
   909) 
   910)        col_loop2 : do i = 1,ncol
   911)           !--------------------------------------------------------
   912)           ! 	... relax lower stratosphere to extended ubc
   913)           !           check to make sure ubc is not being imposed too low
   914)           !           levrelax = lowest model level (highest pressure)
   915)           !                      in which to relax to ubc
   916)           !--------------------------------------------------------
   917)           levrelax = ltrop(i)
   918)           do while( pmid(i,levrelax) > ub_plevs(ub_nlevs) )
   919)              levrelax = levrelax - 1
   920)           end do
   921) #ifdef DEBUG
   922)           if( levrelax /= ltrop(i) ) then
   923)              write(iulog,*) 'warning -- raised ubc: ',lat,i,
   924)              ltrop(i)-1,nint(pmid(i,ltrop(i)-1)/100._r8),'mb -->',
   925)              levrelax,nint(pmid(i,levrelax)/100._r8),'mb'
   926)           end if
   927) #endif
   928)           do k = kmax(i)+1,levrelax
   929)              pint_vals(1) = mr_ub(i,m,last,kl(i,k),lchnk) &
   930)                   + delp(i,k) &
