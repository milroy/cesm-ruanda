     6)       subroutine sphers( z, zen, dsdh, nid )
     7) !-----------------------------------------------------------------------------
     8) !   purpose:
     9) !   calculate slant path over vertical depth ds/dh in spherical geometry.
    10) !   calculation is based on:  a.dahlback, and k.stamnes, a new spheric model
    11) !   for computing the radiation field available for photolysis and heating
    12) !   at twilight, planet.space sci., v39, n5, pp. 671-683, 1991 (appendix b)
    13) !-----------------------------------------------------------------------------
    14) !   parameters:
    15) !   nz      - integer, number of specified altitude levels in the working (i)
    16) !             grid
    17) !   z       - real, specified altitude working grid (km)                  (i)
    18) !   zen     - real, solar zenith angle (degrees)                          (i)
    19) !   dsdh    - real, slant path of direct beam through each layer crossed  (o)
    20) !             when travelling from the top of the atmosphere to layer i;
    21) !             dsdh(i,j), i = 0..nz-1, j = 1..nz-1
    22) !   nid     - integer, number of layers crossed by the direct beam when   (o)
    23) !             travelling from the top of the atmosphere to layer i;
    24) !             nid(i), i = 0..nz-1
    25) !-----------------------------------------------------------------------------
    26) 
    27)       use mo_params,    only : rearth
    28)       use mo_constants, only : d2r
    29)       use ppgrid,       only : pver, pverp
    30)       use shr_kind_mod, only : r8 => shr_kind_r8
    31) 
    32)       implicit none
    33) 
    34) !-----------------------------------------------------------------------------
    35) !	... dummy arguments
    36) !-----------------------------------------------------------------------------
    37)       real(r8), intent(in)     :: zen
    38)       real(r8), intent(in)     :: z(pverp)
    39)       integer, intent(out) :: nid(0:pver)
    40)       real(r8), intent(out)    :: dsdh(0:pver,pver)
    41) 
    42) !-----------------------------------------------------------------------------
    43) !	... local variables
    44) !-----------------------------------------------------------------------------
    45)       integer :: i, j, k
    46)       integer :: id
    47)       real(r8)    :: re, ze(pverp)
    48)       real(r8)    :: zd(0:pver)
    49)       real(r8)    :: zenrad, rpsinz, rj, rjp1, dsj, dhj, ga, gb, sm
    50)       real(r8)    :: radius
    51) 
    52)       radius = rearth*1.e-3_r8    ! rearth m -> km
    53)       zenrad = zen*d2r
    54) !-----------------------------------------------------------------------------
    55) ! 	... include the elevation above sea level to the radius of the earth:
    56) !-----------------------------------------------------------------------------
    57)       re = radius + z(1)
    58) !-----------------------------------------------------------------------------
    59) ! 	... correspondingly z changed to the elevation above earth surface:
    60) !-----------------------------------------------------------------------------
    61)       ze(1:pverp) = z(1:pverp) - z(1)
    62) 
    63) !-----------------------------------------------------------------------------
    64) ! 	... inverse coordinate of z
    65) !-----------------------------------------------------------------------------
    66)       zd(0) = ze(pverp)
    67)       do k = 1,pver
    68)         zd(k) = ze(pverp - k)
    69)       end do
    70) 
    71) !-----------------------------------------------------------------------------
    72) ! 	... initialize dsdh(i,j), nid(i)
    73) !-----------------------------------------------------------------------------
    74)       dsdh(0:pver,1:pver) = 0._r8
    75)       nid(0:pver)         = 0
    76) 
    77) !-----------------------------------------------------------------------------
    78) ! 	... calculate ds/dh of every layer
    79) !-----------------------------------------------------------------------------
    80)       do i = 0,pver
    81)         rpsinz = (re + zd(i)) * sin(zenrad)
    82)         if( zen > 90._r8 .and. rpsinz < re ) then
    83)            nid(i) = -1
    84)         else
    85) !-----------------------------------------------------------------------------
    86) ! 	... find index of layer in which the screening height lies
    87) !-----------------------------------------------------------------------------
    88)            id = i 
    89)            if( zen > 90._r8 ) then
    90)               do j = 1,pver
    91)                  if( (rpsinz < ( zd(j-1) + re ) ) .and. (rpsinz >= ( zd(j) + re )) ) then
    92) 		    id = j
    93) 		 end if
    94)               end do
    95)            end if
    96)  
    97)            do j = 1,id
    98)              if( j == id .and. id == i .and. zen > 90._r8) then
    99)                 sm = -1._r8
   100) 	     else
   101)                 sm = 1._r8
   102)              end if 
   103)              rj = re + zd(j-1)
   104)              rjp1 = re + zd(j)
   105)              dhj = zd(j-1) - zd(j)
   106)              ga = max( 0._r8,rj*rj - rpsinz*rpsinz )
   107)              gb = max( 0._r8,rjp1*rjp1 - rpsinz*rpsinz )
   108)              if( id > i .and. j == id ) then
   109)                 dsj = sqrt( ga )
   110)              else
   111)                 dsj = sqrt( ga ) - sm*sqrt( gb )
   112)              end if
   113)              dsdh(i,j) = dsj / dhj
   114)            end do
   115)            nid(i) = id
   116)         end if
   117)       end do
   118) 
   119)       end subroutine sphers
   120) 
   121)       end module mo_sphers
