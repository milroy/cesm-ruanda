     8)       subroutine airmas( z, zen, dsdh, nid, cz, &
     9)                          vcol, scol )
    10) !-----------------------------------------------------------------------------
    11) !   purpose:
    12) !   calculate vertical and slant air columns, in spherical geometry, as a
    13) !   function of altitude.
    14) !-----------------------------------------------------------------------------
    15) !   parameters:
    16) !   nz      - integer, number of specified altitude levels in the working (i)
    17) !             grid
    18) !   z       - real, specified altitude working grid (km)                  (i)
    19) !   zen     - real, solar zenith angle (degrees)                          (i)
    20) !   dsdh    - real, slant path of direct beam through each layer crossed  (o)
    21) !             when travelling from the top of the atmosphere to layer i;
    22) !             dsdh(i,j), i = 0..nz-1, j = 1..nz-1
    23) !   nid     - integer, number of layers crossed by the direct beam when   (o)
    24) !             travelling from the top of the atmosphere to layer i;
    25) !             nid(i), i = 0..nz-1
    26) !   vcol    - real, output, vertical air column, molec cm-2, above level iz
    27) !   scol    - real, output, slant air column in direction of sun, above iz
    28) !             also in molec cm-2
    29) !-----------------------------------------------------------------------------
    30) 
    31)       use mo_params, only : largest
    32)       use ppgrid,    only : pverp, pver
    33)       use shr_kind_mod, only : r8 => shr_kind_r8
    34) 
    35)       implicit none
    36) 
    37) !-----------------------------------------------------------------------------
    38) !	... dummy arguments
    39) !-----------------------------------------------------------------------------
    40)       integer, intent(in) :: nid(0:pver)
    41)       real(r8), intent(in)    :: z(pverp)
    42)       real(r8), intent(in)    :: zen
    43)       real(r8), intent(in)    :: dsdh(0:pver,pver)
    44)       real(r8), intent(in)    :: cz(pverp)
    45)       real(r8), intent(out)   :: vcol(pverp)
    46)       real(r8), intent(out)   :: scol(pverp)
    47) 
    48) !-----------------------------------------------------------------------------
    49) !	... local variables
    50) !-----------------------------------------------------------------------------
    51)       integer :: id, j
    52)       real(r8)    :: sum, ssum, vsum, ratio
    53) 
    54) !-----------------------------------------------------------------------------
    55) ! 	... calculate vertical and slant column from each level: work downward
    56) !-----------------------------------------------------------------------------
    57)       vsum = 0._r8
    58)       ssum = 0._r8
    59)       do id = 1,pver
    60)          vsum = vsum + cz(pverp-id)
    61)          vcol(pverp-id) = vsum
    62)          sum = 0._r8
    63)          if( nid(id) < 0 ) then
    64)             sum = largest
    65)          else
    66) !-----------------------------------------------------------------------------
    67) ! 	... single pass layers:
    68) !-----------------------------------------------------------------------------
    69)             do j = 1,min( nid(id),id )
    70)                sum = sum + cz(pverp-j)*dsdh(id,j)
    71)             end do
    72) !-----------------------------------------------------------------------------
    73) ! 	... double pass layers:
    74) !-----------------------------------------------------------------------------
    75)             do j = min( nid(id),id )+1,nid(id)
    76)                sum = sum + 2._r8*cz(pverp-j)*dsdh(id,j)
    77)             end do
    78)          end if
    79)          scol(pverp - id) = sum
    80)       end do
    81) 
    82) !-----------------------------------------------------------------------------
    83) ! 	... special section to set scol(pverp)
    84) !-----------------------------------------------------------------------------
    85)       if( scol(pver-1) /= 0._r8 ) then
    86)          ratio       = scol(pver)/scol(pver-1)
    87) 	 scol(pverp) = ratio * scol(pver)
    88)       else
    89)          scol(pverp) = 0._r8
    90)       end if
    91) 
    92)       end subroutine airmas
