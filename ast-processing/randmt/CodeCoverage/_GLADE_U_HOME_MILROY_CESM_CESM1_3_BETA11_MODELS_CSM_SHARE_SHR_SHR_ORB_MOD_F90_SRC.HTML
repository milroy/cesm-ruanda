    44) real(SHR_KIND_R8) FUNCTION shr_orb_cosz(jday,lat,lon,declin)
    45) 
    46)    !----------------------------------------------------------------------------
    47)    !
    48)    ! FUNCTION to return the cosine of the solar zenith angle.
    49)    ! Assumes 365.0 days/year.
    50)    !
    51)    !--------------- Code History -----------------------------------------------
    52)    !
    53)    ! Original Author: Brian Kauffman
    54)    ! Date:            Jan/98
    55)    ! History:         adapted from statement FUNCTION in share/orb_cosz.h
    56)    !
    57)    !----------------------------------------------------------------------------
    58) 
    59)    real   (SHR_KIND_R8),intent(in) :: jday   ! Julian cal day (1.xx to 365.xx)
    60)    real   (SHR_KIND_R8),intent(in) :: lat    ! Centered latitude (radians)
    61)    real   (SHR_KIND_R8),intent(in) :: lon    ! Centered longitude (radians)
    62)    real   (SHR_KIND_R8),intent(in) :: declin ! Solar declination (radians)
    63) 
    64)    !----------------------------------------------------------------------------
    65) 
    66)    shr_orb_cosz = sin(lat)*sin(declin) - &
    67)    &              cos(lat)*cos(declin)*cos(jday*2.0_SHR_KIND_R8*pi + lon)
    68) 
    69) END FUNCTION shr_orb_cosz

   283)    &            201.6651_SHR_KIND_R8, 294.6547_SHR_KIND_R8,  99.8233_SHR_KIND_R8,   &
   284)    &            213.5577_SHR_KIND_R8, 154.1631_SHR_KIND_R8, 232.7153_SHR_KIND_R8,   &
   285)    &            138.3034_SHR_KIND_R8, 204.6609_SHR_KIND_R8, 106.5938_SHR_KIND_R8,   &
   286)    &            250.4676_SHR_KIND_R8, 332.3345_SHR_KIND_R8,  27.3039_SHR_KIND_R8/)
   287)  
   288)    !---------------------------Local variables----------------------------------
   289)    integer(SHR_KIND_IN) :: i       ! Index for series summations
   290)    real   (SHR_KIND_R8) :: obsum   ! Obliquity series summation
   291)    real   (SHR_KIND_R8) :: cossum  ! Cos series summation for eccentricity/fvelp
   292)    real   (SHR_KIND_R8) :: sinsum  ! Sin series summation for eccentricity/fvelp
   293)    real   (SHR_KIND_R8) :: fvelp   ! Fixed vernal equinox long of perihelion
   294)    real   (SHR_KIND_R8) :: mvsum   ! mvelp series summation
   295)    real   (SHR_KIND_R8) :: beta    ! Intermediate argument for lambm0
   296)    real   (SHR_KIND_R8) :: years   ! Years to time of interest ( pos <=> future)
   297)    real   (SHR_KIND_R8) :: eccen2  ! eccentricity squared
   298)    real   (SHR_KIND_R8) :: eccen3  ! eccentricity cubed
   299) 
   300)    !-------------------------- Formats -----------------------------------------
   301)    character(len=*),parameter :: F00 = "('(shr_orb_params) ',4a)"
   302)    character(len=*),parameter :: F01 = "('(shr_orb_params) ',a,i9)"
   303)    character(len=*),parameter :: F02 = "('(shr_orb_params) ',a,f6.3)"
   304)    character(len=*),parameter :: F03 = "('(shr_orb_params) ',a,es14.6)"
   305) 
   306)    !----------------------------------------------------------------------------
   307)    ! radinp and algorithms below will need a degree to radian conversion factor
   308)  
   309)    if ( log_print .and. s_loglev > 0 ) then
   310)      write(s_logunit,F00) 'Calculate characteristics of the orbit:'
   311)    end if
   312)  
   313)    ! Check for flag to use input orbit parameters
   314)  
   315)    IF ( iyear_AD == SHR_ORB_UNDEF_INT ) THEN
   316) 
   317)       ! Check input obliq, eccen, and mvelp to ensure reasonable
   318)  
   319)       if( obliq == SHR_ORB_UNDEF_REAL )then
   320)          write(s_logunit,F00) trim(subname)//' Have to specify orbital parameters:'
   321)          write(s_logunit,F00) 'Either set: iyear_AD, OR [obliq, eccen, and mvelp]:'
   322)          write(s_logunit,F00) 'iyear_AD is the year to simulate orbit for (ie. 1950): '
   323)          write(s_logunit,F00) 'obliq, eccen, mvelp specify the orbit directly:'
   324)          write(s_logunit,F00) 'The AMIP II settings (for a 1995 orbit) are: '
   325)          write(s_logunit,F00) ' obliq =  23.4441'
   326)          write(s_logunit,F00) ' eccen =   0.016715'
   327)          write(s_logunit,F00) ' mvelp = 102.7'
   328)          call shr_sys_abort(subname//' ERROR: unreasonable obliq')

   330)          write(s_logunit,F00) 'Use input orbital parameters: '
   331)       end if

   333)          write(s_logunit,F03) 'Input obliquity unreasonable: ', obliq
   334)          call shr_sys_abort(subname//' ERROR: unreasonable obliq')
   335)       end if
   336)       if( (eccen < SHR_ORB_ECCEN_MIN).or.(eccen > SHR_ORB_ECCEN_MAX) ) then

   338)          call shr_sys_abort(subname//' ERROR: unreasonable eccen')
   339)       end if
   340)       if( (mvelp < SHR_ORB_MVELP_MIN).or.(mvelp > SHR_ORB_MVELP_MAX) ) then
   341)          write(s_logunit,F03) 'Input mvelp unreasonable: ' , mvelp
   342)          call shr_sys_abort(subname//' ERROR: unreasonable mvelp')
   343)       end if

   345)       eccen3 = eccen2*eccen
   346) 
   347)    ELSE  ! Otherwise calculate based on years before present
   348)  
   349)       if ( log_print .and. s_loglev > 0) then
   350)          write(s_logunit,F01) 'Calculate orbit for year: ' , iyear_AD
   351)       end if

   353)       if ( abs(yb4_1950AD) .gt. 1000000.0_SHR_KIND_R8 )then
   354)          write(s_logunit,F00) 'orbit only valid for years+-1000000'
   355)          write(s_logunit,F00) 'Relative to 1950 AD'
   356)          write(s_logunit,F03) '# of years before 1950: ',yb4_1950AD
   357)          write(s_logunit,F01) 'Year to simulate was  : ',iyear_AD
   358)          call shr_sys_abort(subname//' ERROR: unreasonable year')
   359)       end if
   360)  
   361)       ! The following calculates the earths obliquity, orbital eccentricity
   362)       ! (and various powers of it) and vernal equinox mean longitude of
   363)       ! perihelion for years in the past (future = negative of years past),
   364)       ! using constants (see parameter section) given in the program of:
   365)       !
   366)       ! Berger, Andre.  1978  A Simple Algorithm to Compute Long-Term Variations
   367)       ! of Daily Insolation.  Contribution 18, Institute of Astronomy and
   368)       ! Geophysics, Universite Catholique de Louvain, Louvain-la-Neuve, Belgium.
   369)       !
   370)       ! and formulas given in the paper (where less precise constants are also
   371)       ! given):
   372)       !
   373)       ! Berger, Andre.  1978.  Long-Term Variations of Daily Insolation and
   374)       ! Quaternary Climatic Changes.  J. of the Atmo. Sci. 35:2362-2367
   375)       !
   376)       ! The algorithm is valid only to 1,000,000 years past or hence.
   377)       ! For a solution valid to 5-10 million years past see the above author.
   378)       ! Algorithm below is better for years closer to present than is the
   379)       ! 5-10 million year solution.
   380)       !
   381)       ! Years to time of interest must be negative of years before present
   382)       ! (1950) in formulas that follow. 
   383)  
   384)       years = - yb4_1950AD
   385)  
   386)       ! In the summations below, cosine or sine arguments, which end up in
   387)       ! degrees, must be converted to radians via multiplication by degrad.
   388)       !
   389)       ! Summation of cosine series for obliquity (epsilon in Berger 1978) in
   390)       ! degrees. Convert the amplitudes and rates, which are in arc secs, into
   391)       ! degrees via multiplication by psecdeg (arc seconds to degrees conversion
   392)       ! factor).  For obliq, first term is Berger 1978 epsilon star; second
   393)       ! term is series summation in degrees.
   394)   
   395)       obsum = 0.0_SHR_KIND_R8

   397)          obsum = obsum + obamp(i)*psecdeg*cos((obrate(i)*psecdeg*years + &
   398)          &       obphas(i))*degrad)
   399)       end do
   400)       obliq = 23.320556_SHR_KIND_R8 + obsum
   401)  
   402)       ! Summation of cosine and sine series for computation of eccentricity 
   403)       ! (eccen; e in Berger 1978) and fixed vernal equinox longitude of 
   404)       ! perihelion (fvelp; pi in Berger 1978), which is used for computation 
   405)       ! of moving vernal equinox longitude of perihelion.  Convert the rates, 
   406)       ! which are in arc seconds, into degrees via multiplication by psecdeg.
   407)  
   408)       cossum = 0.0_SHR_KIND_R8
   409)       do i = 1, pecclen
   410)         cossum = cossum+ecamp(i)*cos((ecrate(i)*psecdeg*years+ecphas(i))*degrad)
   411)       end do
   412)  
   413)       sinsum = 0.0_SHR_KIND_R8
   414)       do i = 1, pecclen
   415)         sinsum = sinsum+ecamp(i)*sin((ecrate(i)*psecdeg*years+ecphas(i))*degrad)
   416)       end do
   417)  
   418)       ! Use summations to calculate eccentricity
   419)  
   420)       eccen2 = cossum*cossum + sinsum*sinsum
   421)       eccen  = sqrt(eccen2)
   422)       eccen3 = eccen2*eccen
   423)

   425)          
   426)       if (abs(cossum) .le. 1.0E-8_SHR_KIND_R8) then
   427)         if (sinsum .eq. 0.0_SHR_KIND_R8) then
   428)           fvelp = 0.0_SHR_KIND_R8
   429)         else if (sinsum .lt. 0.0_SHR_KIND_R8) then
   430)           fvelp = 1.5_SHR_KIND_R8*pi
   431)         else if (sinsum .gt. 0.0_SHR_KIND_R8) then
   432)           fvelp = .5_SHR_KIND_R8*pi
   433)         endif
   434)       else if (cossum .lt. 0.0_SHR_KIND_R8) then
   435)         fvelp = atan(sinsum/cossum) + pi
   436)       else if (cossum .gt. 0.0_SHR_KIND_R8) then
   437)         if (sinsum .lt. 0.0_SHR_KIND_R8) then
   438)           fvelp = atan(sinsum/cossum) + 2.0_SHR_KIND_R8*pi
   439)         else
   440)           fvelp = atan(sinsum/cossum)
   441)         endif
   442)       endif
   443)  
   444)       ! Summation of sin series for computation of moving vernal equinox long
   445)       ! of perihelion (mvelp; omega bar in Berger 1978) in degrees.  For mvelp,
   446)       ! first term is fvelp in degrees; second term is Berger 1978 psi bar 
   447)       ! times years and in degrees; third term is Berger 1978 zeta; fourth 
   448)       ! term is series summation in degrees.  Convert the amplitudes and rates,
   449)       ! which are in arc seconds, into degrees via multiplication by psecdeg.  
   450)       ! Series summation plus second and third terms constitute Berger 1978
   451)       ! psi, which is the general precession.
   452)  
   453)       mvsum = 0.0_SHR_KIND_R8
   454)       do i = 1, pmvelen
   455)         mvsum = mvsum + mvamp(i)*psecdeg*sin((mvrate(i)*psecdeg*years + &
   456)         &       mvphas(i))*degrad)
   457)       end do
   458)       mvelp = fvelp/degrad + 50.439273_SHR_KIND_R8*psecdeg*years + 3.392506_SHR_KIND_R8 + mvsum
   459)  
   460)       ! Cases to make sure mvelp is between 0 and 360.
   461)  
   462)       do while (mvelp .lt. 0.0_SHR_KIND_R8)
   463)         mvelp = mvelp + 360.0_SHR_KIND_R8
   464)       end do
   465)       do while (mvelp .ge. 360.0_SHR_KIND_R8)
   466)         mvelp = mvelp - 360.0_SHR_KIND_R8
   467)       end do
   468) 
   469)    END IF  ! end of test on whether to calculate or use input orbital params
   470)  
   471)    ! Orbit needs the obliquity in radians
   472)  
   473)    obliqr = obliq*degrad
   474)  
   475)    ! 180 degrees must be added to mvelp since observations are made from the
   476)    ! earth and the sun is considered (wrongly for the algorithm) to go around
   477)    ! the earth. For a more graphic explanation see Appendix B in:
   478)    !
   479)    ! A. Berger, M. Loutre and C. Tricot. 1993.  Insolation and Earth Orbital
   480)    ! Periods.  J. of Geophysical Research 98:10,341-10,362.
   481)    !
   482)    ! Additionally, orbit will need this value in radians. So mvelp becomes
   483)    ! mvelpp (mvelp plus pi)
   484)  
   485)    mvelpp = (mvelp + 180._SHR_KIND_R8)*degrad
   486)  
   487)    ! Set up an argument used several times in lambm0 calculation ahead.
   488)  
   489)    beta = sqrt(1._SHR_KIND_R8 - eccen2)
   490)  
   491)    ! The mean longitude at the vernal equinox (lambda m nought in Berger
   492)    ! 1978; in radians) is calculated from the following formula given in 
   493)    ! Berger 1978.  At the vernal equinox the true longitude (lambda in Berger
   494)    ! 1978) is 0.
   495) 
   496)    lambm0 = 2._SHR_KIND_R8*((.5_SHR_KIND_R8*eccen + .125_SHR_KIND_R8*eccen3)*(1._SHR_KIND_R8 + beta)*sin(mvelpp)  &
   497)    &      - .250_SHR_KIND_R8*eccen2*(.5_SHR_KIND_R8    + beta)*sin(2._SHR_KIND_R8*mvelpp)            &
   498)    &      + .125_SHR_KIND_R8*eccen3*(1._SHR_KIND_R8/3._SHR_KIND_R8 + beta)*sin(3._SHR_KIND_R8*mvelpp))
   499)  
   500)    if ( log_print ) then
   501)      write(s_logunit,F03) '------ Computed Orbital Parameters ------'
   502)      write(s_logunit,F03) 'Eccentricity      = ',eccen
   503)      write(s_logunit,F03) 'Obliquity (deg)   = ',obliq
   504)      write(s_logunit,F03) 'Obliquity (rad)   = ',obliqr
   505)      write(s_logunit,F03) 'Long of perh(deg) = ',mvelp
   506)      write(s_logunit,F03) 'Long of perh(rad) = ',mvelpp
   507)      write(s_logunit,F03) 'Long at v.e.(rad) = ',lambm0
   508)      write(s_logunit,F03) '-----------------------------------------'
   509)    end if
   510)  
   511) END SUBROUTINE shr_orb_params

   515) SUBROUTINE shr_orb_decl(calday ,eccen ,mvelpp ,lambm0 ,obliqr ,delta ,eccf)
   516) 
   517) !-------------------------------------------------------------------------------
   518) !
   519) ! Compute earth/orbit parameters using formula suggested by
   520) ! Duane Thresher.
   521) !
   522) !---------------------------Code history----------------------------------------
   523) !
   524) ! Original version:  Erik Kluzek
   525) ! Date:              Oct/1997
   526) !
   527) !-------------------------------------------------------------------------------
   528) 
   529)    !------------------------------Arguments--------------------------------
   530)    real   (SHR_KIND_R8),intent(in)  :: calday ! Calendar day, including fraction
   531)    real   (SHR_KIND_R8),intent(in)  :: eccen  ! Eccentricity
   532)    real   (SHR_KIND_R8),intent(in)  :: obliqr ! Earths obliquity in radians
   533)    real   (SHR_KIND_R8),intent(in)  :: lambm0 ! Mean long of perihelion at the 
   534)                                               ! vernal equinox (radians)
   535)    real   (SHR_KIND_R8),intent(in)  :: mvelpp ! moving vernal equinox longitude
   536)                                               ! of perihelion plus pi (radians)
   537)    real   (SHR_KIND_R8),intent(out) :: delta  ! Solar declination angle in rad
   538)    real   (SHR_KIND_R8),intent(out) :: eccf   ! Earth-sun distance factor (ie. (1/r)**2)
   539)  
   540)    !---------------------------Local variables-----------------------------
   541)    real   (SHR_KIND_R8),parameter :: dayspy = 365.0_SHR_KIND_R8  ! days per year
   542)    real   (SHR_KIND_R8),parameter :: ve     = 80.5_SHR_KIND_R8   ! Calday of vernal equinox
   543)                                                      ! assumes Jan 1 = calday 1
   544)  
   545)    real   (SHR_KIND_R8) ::   lambm  ! Lambda m, mean long of perihelion (rad)
   546)    real   (SHR_KIND_R8) ::   lmm    ! Intermediate argument involving lambm
   547)    real   (SHR_KIND_R8) ::   lamb   ! Lambda, the earths long of perihelion
   548)    real   (SHR_KIND_R8) ::   invrho ! Inverse normalized sun/earth distance
   549)    real   (SHR_KIND_R8) ::   sinl   ! Sine of lmm
   550)  
   551)    ! Compute eccentricity factor and solar declination using
   552)    ! day value where a round day (such as 213.0) refers to 0z at
   553)    ! Greenwich longitude.
   554)    !
   555)    ! Use formulas from Berger, Andre 1978: Long-Term Variations of Daily
   556)    ! Insolation and Quaternary Climatic Changes. J. of the Atmo. Sci.
   557)    ! 35:2362-2367.
   558)    !
   559)    ! To get the earths true longitude (position in orbit; lambda in Berger 
   560)    ! 1978) which is necessary to find the eccentricity factor and declination,
   561)    ! must first calculate the mean longitude (lambda m in Berger 1978) at
   562)    ! the present day.  This is done by adding to lambm0 (the mean longitude
   563)    ! at the vernal equinox, set as March 21 at noon, when lambda=0; in radians)
   564)    ! an increment (delta lambda m in Berger 1978) that is the number of
   565)    ! days past or before (a negative increment) the vernal equinox divided by
   566)    ! the days in a model year times the 2*pi radians in a complete orbit.
   567)  
   568)    lambm = lambm0 + (calday - ve)*2._SHR_KIND_R8*pi/dayspy
   569)    lmm   = lambm  - mvelpp
   570)  
   571)    ! The earths true longitude, in radians, is then found from
   572)    ! the formula in Berger 1978:
   573)  
   574)    sinl  = sin(lmm)
   575)    lamb  = lambm  + eccen*(2._SHR_KIND_R8*sinl + eccen*(1.25_SHR_KIND_R8*sin(2._SHR_KIND_R8*lmm)  &
   576)    &     + eccen*((13.0_SHR_KIND_R8/12.0_SHR_KIND_R8)*sin(3._SHR_KIND_R8*lmm) - 0.25_SHR_KIND_R8*sinl)))
   577)  
   578)    ! Using the obliquity, eccentricity, moving vernal equinox longitude of
   579)    ! perihelion (plus), and earths true longitude, the declination (delta)
   580)    ! and the normalized earth/sun distance (rho in Berger 1978; actually inverse
   581)    ! rho will be used), and thus the eccentricity factor (eccf), can be 
   582)    ! calculated from formulas given in Berger 1978.
   583)  
   584)    invrho = (1._SHR_KIND_R8 + eccen*cos(lamb - mvelpp)) / (1._SHR_KIND_R8 - eccen*eccen)
   585)  
   586)    ! Set solar declination and eccentricity factor
   587)  
   588)    delta  = asin(sin(obliqr)*sin(lamb))
   589)    eccf   = invrho*invrho
   590)  
   591)    return
   592)  
   593) END SUBROUTINE shr_orb_decl
