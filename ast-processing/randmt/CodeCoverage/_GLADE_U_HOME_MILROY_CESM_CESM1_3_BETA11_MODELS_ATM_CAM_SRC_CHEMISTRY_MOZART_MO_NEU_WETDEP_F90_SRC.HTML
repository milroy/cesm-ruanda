    49) subroutine neu_wetdep_init
    50) !
    51)   use constituents, only : cnst_get_ind,cnst_mw
    52)   use cam_history,  only : addfld, add_default, phys_decomp
    53)   use ppgrid,       only : pver
    54)   use phys_control, only : phys_getopts
    55) !
    56)   integer :: m,l
    57)   character*20 :: test_name
    58) 
    59)   logical :: history_chemistry
    60) 
    61)   call phys_getopts(history_chemistry_out=history_chemistry)
    62) 
    63)   do_neu_wetdep = gas_wetdep_method == 'NEU' .and. gas_wetdep_cnt>0
    64) 
    65)   if (.not.do_neu_wetdep) return
    66) 
    67)   allocate( mapping_to_heff(gas_wetdep_cnt) )
    68)   allocate( mapping_to_mmr(gas_wetdep_cnt) )
    69)   allocate( ice_uptake(gas_wetdep_cnt) )
    70)   allocate( mol_weight(gas_wetdep_cnt) )
    71) 
    72) !
    73) ! find mapping to heff table
    74) !
    75)   if ( debug ) then
    76)     print '(a,i4)','gas_wetdep_cnt=',gas_wetdep_cnt
    77)     print '(a,i4)','n_species_table=',n_species_table
    78)   end if
    79)   mapping_to_heff = -99
    80)   do m=1,gas_wetdep_cnt
    81) !
    82)     test_name = gas_wetdep_list(m)
    83)     if ( debug ) print '(i4,a)',m,trim(test_name)
    84) !
    85) ! mapping based on the MOZART4 wet removal subroutine;
    86) ! this might need to be redone (JFL: Sep 2010)
    87) !
    88)     select case( trim(test_name) )
    89) !
    90) ! CCMI: added SO2t and NH_50W
    91) !       
    92)       case( 'HYAC', 'CH3COOH' , 'HCOOH', 'EOOH' )
    93)          test_name = 'CH2O'
    94)       case ( 'SOGB','SOGI','SOGM','SOGT','SOGX' )
    95)          test_name = 'H2O2'
    96)       case ( 'SO2t' )
    97)          test_name = 'SO2'
    98)       case ( 'CLONO2','BRONO2','HCL','HOCL','HOBR','HBR', 'Pb', 'MACROOH', 'ISOPOOH', 'XOOH', 'H2SO4', 'HF', 'COF2', 'COFCL')
    99)          test_name = 'HNO3'
   100)       case ( 'NH_50W' ) 
   101)          test_name = 'HNO3'
   102)       case ( 'ALKOOH', 'MEKOOH', 'TOLOOH', 'TERPOOH' )
   103)          test_name = 'CH3OOH'        
   104) 
   105)     end select
   106) !
   107)     do l = 1,n_species_table
   108) !
   109) !      if ( debug ) print '(i4,a)',l,trim(species_name_table(l))
   110) !
   111)        if( trim(test_name) == trim( species_name_table(l) ) ) then
   112)           mapping_to_heff(m)  = l
   113)           if ( debug ) print '(a,a,i4)','mapping to heff of ',trim(species_name_table(l)),l
   114)           exit
   115)        end if
   116)     end do
   117)     if ( mapping_to_heff(m) == -99 ) then
   118)       if (masterproc) print *,'problem with mapping_to_heff of ',trim(test_name)
   119) !      call endrun()
   120)     end if
   121) !
   122) ! special cases for NH3 and CO2
   123) !
   124)     if ( trim(test_name) == 'NH3' ) then
   125)       nh3_ndx = m
   126)     end if
   127)     if ( trim(test_name) == 'CO2' ) then
   128)       co2_ndx = m
   129)     end if
   130)     if ( trim(test_name) == 'HNO3' ) then
   131)       hno3_ndx = m
   132)     end if
   133) !
   134)   end do
   135)    
   136)    if (any ( mapping_to_heff(:) == -99 ))  call endrun('mo_neu_wet->depwetdep_init: unmapped species error' )
   137) !
   138)   if ( debug ) then
   139)     print '(a,i4)','co2_ndx',co2_ndx
   140)     print '(a,i4)','nh3_ndx',nh3_ndx
   141)   end if
   142) !
   143) ! find mapping to species
   144) !
   145)   mapping_to_mmr = -99
   146)   do m=1,gas_wetdep_cnt
   147)     if ( debug ) print '(i4,a)',m,trim(gas_wetdep_list(m))
   148)     call cnst_get_ind(gas_wetdep_list(m), mapping_to_mmr(m), abort=.false. )
   149)     if ( debug ) print '(a,i4)','mapping_to_mmr ',mapping_to_mmr(m)
   150)     if ( mapping_to_mmr(m) <= 0 ) then
   151)       print *,'problem with mapping_to_mmr of ',gas_wetdep_list(m)
   152)       call endrun('problem with mapping_to_mmr of '//trim(gas_wetdep_list(m)))
   153)     end if
   154)   end do
   155) !
   156) ! define specie-dependent arrays
   157) !
   158)   do m=1,gas_wetdep_cnt
   159) !
   160)     mol_weight     (m) = cnst_mw(mapping_to_mmr(m))
   161)     if ( debug ) print '(i4,a,f8.4)',m,' mol_weight ',mol_weight(m)
   162)     ice_uptake(m) = .false.
   163)     if ( trim(gas_wetdep_list(m)) == 'HNO3' ) then
   164)       ice_uptake(m) = .true.
   165)     end if
   166) !
   167)   end do
   168) !
   169) ! indices for cloud quantities
   170) !
   171)   call cnst_get_ind( 'CLDICE', index_cldice )
   172)   call cnst_get_ind( 'CLDLIQ', index_cldliq )
   173) !
   174) ! define output
   175) !
   176)   do m=1,gas_wetdep_cnt
   177)     call addfld     ('DTWR_'//trim(gas_wetdep_list(m)),'mol/mol/s',pver, 'A','wet removal Neu scheme tendency',phys_decomp)
   178)     if (history_chemistry) then
   179)        call add_default('DTWR_'//trim(gas_wetdep_list(m)), 1, ' ')
   180)     end if
   181)   end do
   182) !
   183)   if ( do_diag ) then
   184)     call addfld     ('QT_RAIN_HNO3','mol/mol/s',pver, 'A','wet removal Neu scheme rain tendency',phys_decomp)
   185)     call addfld     ('QT_RIME_HNO3','mol/mol/s',pver, 'A','wet removal Neu scheme rain tendency',phys_decomp)
   186)     call addfld     ('QT_WASH_HNO3','mol/mol/s',pver, 'A','wet removal Neu scheme rain tendency',phys_decomp)
   187)     call addfld     ('QT_EVAP_HNO3','mol/mol/s',pver, 'A','wet removal Neu scheme rain tendency',phys_decomp)
   188)     if (history_chemistry) then
   189)        call add_default('QT_RAIN_HNO3',1,' ')
   190)        call add_default('QT_RIME_HNO3',1,' ')
   191)        call add_default('QT_WASH_HNO3',1,' ')
   192)        call add_default('QT_EVAP_HNO3',1,' ')
   193)     end if
   194)   end if
   195) !
   196)   return
   197) !
   198) end subroutine neu_wetdep_init
   199) !
   200) subroutine neu_wetdep_tend(lchnk,ncol,mmr,pmid,pdel,zint,tfld,delt, &
   201)      prain, nevapr, cld, cmfdqr, wd_tend)
   202) !
   203)   use ppgrid,           only : pcols, pver
   204) !!DEK  
   205)   use phys_grid,        only : get_area_all_p, get_rlat_all_p
   206)   use shr_const_mod,    only : SHR_CONST_REARTH,SHR_CONST_G
   207)   use cam_history,      only : outfld
   208) !
   209)   implicit none
   210) !
   211)   integer,        intent(in)    :: lchnk,ncol
   212)   real(r8),       intent(in)    :: mmr(pcols,pver,pcnst)    ! mass mixing ratio (kg/kg)
   213)   real(r8),       intent(in)    :: pmid(pcols,pver)         ! midpoint pressures (Pa)
   214)   real(r8),       intent(in)    :: pdel(pcols,pver)         ! pressure delta about midpoints (Pa)
   215)   real(r8),       intent(in)    :: zint(pcols,pver+1)       ! interface geopotential height above the surface (m)
   216)   real(r8),       intent(in)    :: tfld(pcols,pver)         ! midpoint temperature (K)
   217)   real(r8),       intent(in)    :: delt                     ! timestep (s)
   218) !  
   219) 
   220)   real(r8),       intent(in)    :: prain(ncol, pver)
   221)   real(r8),       intent(in)    :: nevapr(ncol, pver)
   222)   real(r8),       intent(in)    :: cld(ncol, pver)
   223)   real(r8),       intent(in)    :: cmfdqr(ncol, pver)
   224) 
   225)   real(r8),       intent(inout) :: wd_tend(pcols,pver,pcnst)
   226) 
   227) 
   228) 
   229) !
   230) ! local arrays and variables
   231) !
   232)   integer :: i,k,l,kk,m,id
   233)   real(r8), parameter                       :: rearth = SHR_CONST_REARTH    ! radius earth (m)
   234)   real(r8), parameter                       :: gravit = SHR_CONST_G         ! m/s^2
   235)   real(r8), dimension(ncol)                 :: area
   236)   real(r8), dimension(ncol,pver)            :: cldice,cldliq,cldfrc,totprec,totevap,delz,delp,p
   237)   real(r8), dimension(ncol,pver)            :: rls,evaprate,mass_in_layer,temp
   238)   real(r8), dimension(ncol,pver,gas_wetdep_cnt) :: trc_mass,heff,dtwr
   239)   real(r8), dimension(ncol,pver,gas_wetdep_cnt) :: wd_mmr
   240)   logical , dimension(gas_wetdep_cnt)           :: tckaqb
   241)   integer , dimension(ncol)                 :: test_flag
   242) !
   243) ! arrays for HNO3 diagnostics
   244) !
   245)   real(r8), dimension(ncol,pver)            :: qt_rain,qt_rime,qt_wash,qt_evap
   246) !
   247) ! for Henry's law calculations
   248) !
   249)   real(r8), parameter       :: t0     = 298._r8
   250)   real(r8), parameter       :: ph     = 1.e-5_r8
   251)   real(r8), parameter       :: ph_inv = 1._r8/ph
   252)   real(r8)                  :: e298, dhr
   253)   real(r8), dimension(ncol) :: dk1s,dk2s,wrk
   254) !!DEK
   255)   real(r8) :: pi
   256)   real(r8) :: lats(pcols)
   257) !
   258) ! from cam/src/physics/cam/stratiform.F90
   259) !
   260) !!DEK
   261)   pi = 4._r8*atan(1.0_r8)
   262) 
   263)   if (.not.do_neu_wetdep) return
   264) !
   265) ! don't do anything if there are no species to be removed
   266) !
   267)   if ( gas_wetdep_cnt == 0 ) return
   268) !
   269) ! reset output variables
   270) !
   271) !  wd_tend = 0._r8
   272) !
   273) ! get area (in radians square)
   274) !
   275)   call get_area_all_p(lchnk, ncol, area)
   276)   area = area * rearth**2                     ! in m^2
   277) !
   278) ! reverse order along the vertical before calling
   279) ! J. Neu's wet removal subroutine
   280) !
   281)   do k=1,pver
   282)     kk = pver - k + 1
   283)     do i=1,ncol
   284) !
   285)       mass_in_layer(i,k) = area(i) * pdel(i,kk)/gravit          ! kg
   286) !
   287)       cldice (i,k) = mmr(i,kk,index_cldice)                     ! kg/kg
   288)       cldliq (i,k) = mmr(i,kk,index_cldliq)                     ! kg/kg
   289)       cldfrc (i,k) = cld(i,kk)                                  ! unitless
   290) !
   291)       totprec(i,k) = (prain(i,kk)+cmfdqr(i,kk)) &
   292)                                   * mass_in_layer(i,k)          ! kg/s
   293)       totevap(i,k) = nevapr(i,kk) * mass_in_layer(i,k)          ! kg/s
   294) !
   295)       delz(i,k) = zint(i,kk) - zint(i,kk+1)                     ! in m
   296) !
   297)       temp(i,k) = tfld(i,kk)
   298) !
   299) ! convert tracer mass to kg
   300) !
   301)       trc_mass(i,k,:) = mmr(i,kk,mapping_to_mmr(:)) * mass_in_layer(i,k)
   302) !
   303)       delp(i,k) = pdel(i,kk) * 0.01_r8          ! in hPa
   304)       p   (i,k) = pmid(i,kk) * 0.01_r8          ! in hPa
   305) !
   306)     end do
   307)   end do
   308) !
   309) ! define array for tendency calculation (on model grid)
   310) !
   311)   dtwr(1:ncol,:,:) = mmr(1:ncol,:,mapping_to_mmr(:))
   312) !
   313) ! compute 1) integrated precipitation flux across the interfaces (rls)
   314) !         2) evaporation rate
   315) !
   316)   rls      (:,pver) = 0._r8
   317)   evaprate (:,pver) = 0._r8
   318)   do k=pver-1,1,-1
   319)     rls     (:,k) = max(0._r8,totprec(:,k)-totevap(:,k)+rls(:,k+1))
   320)     !evaprate(:,k) = min(1._r8,totevap(:,k)/(rls(:,k+1)+totprec(:,k)+1.e-36_r8))
   321)     evaprate(:,k) = min(1._r8,totevap(:,k)/(rls(:,k+1)+1.e-36_r8)) 
   322)   end do
   323) !
   324) ! compute effective Henry's law coefficients
   325) ! code taken from models/drv/shr/seq_drydep_mod.F90
   326) !
   327)   heff = 0._r8
   328)   do k=1,pver
   329) !
   330)     kk = pver - k + 1
   331) !
   332)     wrk(:) = (t0-tfld(1:ncol,kk))/(t0*tfld(1:ncol,kk))
   333) !
   334)     do m=1,gas_wetdep_cnt
   335) !
   336)       l    = mapping_to_heff(m)
   337)       id   = 6*(l - 1)
   338)       e298 = dheff(id+1)
   339)       dhr  = dheff(id+2)
   340)       heff(:,k,m) = e298*exp( dhr*wrk(:) )
   341)       test_flag = -99
   342)       if( dheff(id+3) /= 0._r8 .and. dheff(id+5) == 0._r8 ) then
   343)         e298 = dheff(id+3)
   344)         dhr  = dheff(id+4)
   345)         dk1s(:) = e298*exp( dhr*wrk(:) )
   346)         where( heff(:,k,m) /= 0._r8 )
   347)           heff(:,k,m) = heff(:,k,m)*(1._r8 + dk1s(:)*ph_inv)
   348)         elsewhere
   349)           test_flag = 1
   350)           heff(:,k,m) = dk1s(:)*ph_inv
   351)         endwhere
   352)       end if
   353) !
   354)       if (k.eq.1 .and. maxval(test_flag) > 0 .and. debug ) print '(a,i4)','heff for m=',m
   355) !
   356)       if( dheff(id+5) /= 0._r8 ) then
   357)         if( nh3_ndx > 0 .or. co2_ndx > 0 ) then
   358)           e298 = dheff(id+3)
   359)           dhr  = dheff(id+4)
   360)           dk1s(:) = e298*exp( dhr*wrk(:) )
   361)           e298 = dheff(id+5)
   362)           dhr  = dheff(id+6)
   363)           dk2s(:) = e298*exp( dhr*wrk(:) )
   364)           if( m == co2_ndx ) then
   365)              heff(:,k,m) = heff(:,k,m)*(1._r8 + dk1s(:)*ph_inv)*(1._r8 + dk2s(:)*ph_inv)
   366)           else if( m == nh3_ndx ) then
   367)              heff(:,k,m) = heff(:,k,m)*(1._r8 + dk1s(:)*ph/dk2s(:))
   368)           else
   369)              write(iulog,*) 'error in assigning henrys law coefficients'
   370)              write(iulog,*) 'species ',m
   371)           end if
   372)         end if
   373)       end if
   374) !
   375)     end do
   376)   end do
   377) !
   378) ! define flag for high effective Henry's law
   379) !
   380)   do m=1,gas_wetdep_cnt
   381)     if ( maxval(heff(:,:,m)) > 1.e4_r8 ) then
   382)       tckaqb(m) = .true.
   383)     else
   384)       tckaqb(m) = .false.
   385)     end if
   386)   end do
   387) !
   388)   if ( debug ) then
   389)     print '(a,50f8.2)','tckaqb     ',tckaqb
   390)     print '(a,50e12.4)','heff      ',heff(1,1,:)
   391)     print '(a,50i4)'  ,'ice_uptake ',ice_uptake
   392)     print '(a,50f8.2)','mol_weight ',mol_weight(:)
   393)     print '(a,50f8.2)','temp       ',temp(1,:)
   394)     print '(a,50f8.2)','p          ',p   (1,:)
   395)   end if
   396) !
   397) ! call J. Neu's subroutine
   398) !
   399)   do i=1,ncol
   400) !
   401)     call washo(pver,gas_wetdep_cnt,delt,trc_mass(i,:,:),mass_in_layer(i,:),p(i,:),delz(i,:) &
   402)               ,rls(i,:),cldliq(i,:),cldice(i,:),cldfrc(i,:),temp(i,:),evaprate(i,:) &
   403)               ,area(i),heff(i,:,:),mol_weight(:),tckaqb(:),ice_uptake(:) &
   404)               ,qt_rain(i,:),qt_rime(i,:),qt_wash(i,:),qt_evap(i,:) )
   405) !
   406)   end do
   407) !
   408) ! compute tendencies and convert back to mmr
   409) ! on original vertical grid
   410) !
   411)   do k=1,pver
   412)     kk = pver - k + 1
   413)     do i=1,ncol
   414) !
   415) ! convert tracer mass from kg
   416) !
   417)       wd_mmr(i,kk,:) = trc_mass(i,k,:) / mass_in_layer(i,k)
   418) !
   419)     end do
   420)   end do
   421) !
   422) ! tendency calculation (on model grid)
   423) !
   424)   dtwr(1:ncol,:,:) = wd_mmr(1:ncol,:,:) - dtwr(1:ncol,:,:)
   425)   dtwr(1:ncol,:,:) = dtwr(1:ncol,:,:) / delt 
   426) 
   427) !!DEK polarward of 60S, 60N and <200hPa set to zero!
   428)   call get_rlat_all_p(lchnk, pcols, lats )
   429)   do k = 1, pver
   430)     do i= 1, ncol
   431)       if ( abs( lats(i)*180._r8/pi ) > 60._r8 ) then
   432)         if ( pmid(i,k) < 20000._r8) then
   433)            dtwr(i,k,:) = 0._r8
   434)         endif 
   435)       endif
   436)     end do
   437)   end do
   438) !
   439) ! output tendencies
   440) !
   441)   do m=1,gas_wetdep_cnt
   442)     wd_tend(1:ncol,:,mapping_to_mmr(m)) = wd_tend(1:ncol,:,mapping_to_mmr(m)) + dtwr(1:ncol,:,m)
   443)     call outfld( 'DTWR_'//trim(gas_wetdep_list(m)),dtwr(:,:,m),ncol,lchnk )
   444)   end do
   445) !
   446)   if ( do_diag ) then
   447)     call outfld('QT_RAIN_HNO3', qt_rain, ncol, lchnk )
   448)     call outfld('QT_RIME_HNO3', qt_rime, ncol, lchnk )
   449)     call outfld('QT_WASH_HNO3', qt_wash, ncol, lchnk )
   450)     call outfld('QT_EVAP_HNO3', qt_evap, ncol, lchnk )
   451)   end if
   452) !
   453)   return
   454) end subroutine neu_wetdep_tend

   523)       real(r8), parameter  :: CWMIN=1.0e-5_r8
   524)       real(r8), parameter  :: DMIN=1.0e-1_r8       !mm
   525)       real(r8), parameter  :: VOLPOW=1._r8/3._r8

   531)       real(r8), parameter  :: COLEFFAER=0.05_r8

   534) !

   539)       real(r8) :: rcxa_wrk(lpar)
   540)       real(r8) :: fcxa_wrk(lpar)
   541)       real(r8) :: rcxb_wrk(lpar)

   544)       real(r8) :: fax_wrk(lpar,2)
   545)       real(r8) :: rama_wrk(lpar)

   547)       real(r8) :: deltarime_wrk(lpar)
   548)       real(r8) :: clwx_wrk(lpar)
   549)       real(r8) :: frc(lpar,3)
   550)       real(r8) :: rlsog(lpar)
   551) !

   554)       logical :: rnew_flag(lpar)
   555)       logical :: cf_trigger(lpar)

   557) !

   559)       real(r8), parameter :: adj_factor = one + 10._r8*epsilon( one )
   560) !
   561)       integer :: LWASHTYP,LICETYP
   562) !
   563)       if ( debug ) then
   564)         print '(a,50f8.2)','tckaqb     ',tckaqb
   565)         print '(a,50e12.4)','hstar     ',hstar(1,:)
   566)         print '(a,50i4)'  ,'ice_uptake ',TCNION
   567)         print '(a,50f8.2)','mol_weight ',TCMASS(:)
   568)         print '(a,50f8.2)','temp       ',tem(:)
   569)         print '(a,50f8.2)','p          ',pofl(:)
   570)       end if
   571) 
   572) !-----------------------------------------------------------------------
   573)       LE = LPAR-1   
   574) !
   575)       rls_flag(1:le) = rls(1:le) > zero 
   576)       freezing(1:le) = tem(1:le) < tice
   577)       rlsog(1:le) = rls(1:le)/garea
   578) !
   579) species_loop : &
   580)      do N = 1,NTRACE
   581)        QTT(:lpar)    = QTTJFL(:lpar,N)
   582)        QTTNEW(:lpar) = QTTJFL(:lpar,N)
   583)        is_hno3 = n == hno3_ndx
   584)        if( is_hno3 ) then
   585)          qt_rain(:lpar) = zero
   586)          qt_rime(:lpar) = zero
   587)          qt_wash(:lpar) = zero
   588)          qt_evap(:lpar) = zero
   589)          rca_wrk(:lpar) = zero
   590)          fca_wrk(:lpar) = zero
   591)          rcxa_wrk(:lpar) = zero
   592)          fcxa_wrk(:lpar) = zero
   593)          rcxb_wrk(:lpar) = zero
   594)          fcxb_wrk(:lpar) = zero
   595)          rls_wrk(:lpar) = zero
   596)          rnew_wrk(:lpar) = zero
   597)          cf_trigger(:lpar) = .false.
   598)          clwx_wrk(:lpar) = -9999._r8
   599)          deltarime_wrk(:lpar) = -9999._r8
   600)          rax_wrk(:lpar,:) = zero
   601)          fax_wrk(:lpar,:) = zero
   602)        endif
   603) !-----------------------------------------------------------------------
   604) !  calculate scavenging by large-scale stratiform precipitation
   605) !  check whether mass-limited or henry's law
   606) !-----------------------------------------------------------------------
   607)        if( TCKAQB(N) ) then
   608)          LWASHTYP = 1
   609)        else
   610)          LWASHTYP = 2
   611)        end if
   612) !-----------------------------------------------------------------------
   613) !  check whether soluble in ice
   614) !-----------------------------------------------------------------------
   615)        if( TCNION(N) ) then
   616)          LICETYP = 1
   617)        else
   618)          LICETYP = 2
   619)        end if
   620) 
   621) !-----------------------------------------------------------------------
   622) !  initialization
   623) !-----------------------------------------------------------------------
   624)        QTTOPAA = zero
   625)        QTTOPCA = zero
   626) 
   627)        RCA  = zero
   628)        FCA  = zero
   629)        DCA  = zero
   630)        RAMA = zero
   631)        FAMA = zero
   632)        DAMA = zero
   633) 
   634)        AMPCT      = zero
   635)        AMCLPCT    = zero
   636)        CLNEWPCT   = zero
   637)        CLNEWAMPCT = zero
   638)        CLOLDPCT   = zero
   639)        CLOLDAMPCT = zero
   640) !-----------------------------------------------------------------------
   641) !  Check whether precip in top layer - if so, require CF ge 0.2
   642) !-----------------------------------------------------------------------
   643)        if( RLS(LE) > zero ) then
   644)          CFXX(LE) = max( CFMIN,CFR(LE) )
   645)        else
   646)          CFXX(LE) = CFR(LE)
   647)        endif
   648) 
   649)        rnew_flag(1:le) = .false.
   650) 
   651) level_loop : &
   652)        do L = LE,1,-1
   653)          LM1  = L - 1
   654)          FAX  = zero
   655)          RAX  = zero
   656)          DAX  = zero
   657)          FCXA = zero
   658)          FCXB = zero
   659)          DCXA = zero
   660)          DCXB = zero
   661)          RCXA = zero
   662)          RCXB = zero
   663) 
   664)          QTDISCF   = zero
   665)          QTDISRIME = zero
   666)          QTDISCXA  = zero
   667) 
   668)          QTEVAPAXP = zero
   669)          QTEVAPAXW = zero
   670)          QTEVAPAX  = zero
   671)          QTWASHAX  = zero
   672) 
   673)          QTEVAPCXAP = zero
   674)          QTEVAPCXAW = zero
   675)          QTEVAPCXA  = zero
   676)          QTRIMECXA  = zero
   677)          QTWASHCXA  = zero
   678)          QTRAINCXA  = zero
   679)          QTRAINCXB  = zero
   680)          
   681)          RAMPCT = zero
   682)          RCXPCT = zero
   683) 
   684)          RCXALOC = zero
   685)          RCXBLOC = zero
   686)          RAXLOC  = zero
   687)          RAMALOC = zero
   688)          RCALOC  = zero
   689) 
   690)          RPRECIP       = zero
   691)          DELTARIMEMASS = zero
   692)          DELTARIME     = zero
   693)          DOR           = zero
   694)          DNEW          = zero
   695) 
   696)          QTTOPAAX = zero
   697)          QTTOPCAX = zero
   698) 
   699) has_rls : &
   700)          if( rls_flag(l) ) then
   701) !-----------------------------------------------------------------------
   702) !-----Evaporate ambient precip and decrease area-------------------------
   703) !-----If ice, diam=diam falling from above  If rain, diam=4mm (not used)
   704) !-----Evaporate tracer contained in evaporated precip
   705) !-----Can't evaporate more than we start with-----------------------------
   706) !-----Don't do washout until we adjust ambient precip to match Rbot if needed
   707) !------(after RNEW if statements)
   708) !-----------------------------------------------------------------------
   709)            FAX = max( zero,FAMA*(one - evaprate(l)) )
   710)            RAX = RAMA								     !kg/m2/s
   711)            if ( debug ) then
   712)              if( (l == 3 .or. l == 2) ) then
   713)                write(*,*) 'washout: l,rls,fax = ',l,rls(l),fax
   714)              endif
   715)            endif
   716)            if( FAMA > zero ) then
   717)              if( freezing(l) ) then
   718)                DAX = DAMA      !mm
   719)              else
   720)                DAX = four    !mm - not necessary
   721)              endif
   722)            else
   723)              DAX = zero
   724)            endif
   725) 
   726)            if( RAMA > zero ) then
   727)              QTEVAPAXP = min( QTTOPAA,EVAPRATE(L)*QTTOPAA )
   728)            else
   729)              QTEVAPAXP = zero
   730)            endif
   731)            if( is_hno3 ) then
   732)              rax_wrk(l,1) = rax
   733)              fax_wrk(l,1) = fax
   734)            endif
   735) 
   736) 
   737) !-----------------------------------------------------------------------
   738) !  Determine how much the in-cloud precip rate has increased------
   739) !-----------------------------------------------------------------------
   740)            WRK = RAX*FAX + RCA*FCA
   741)            if( WRK > 0._r8 ) then
   742)              RNEW_TST = RLS(L)/(GAREA * WRK)
   743)            else
   744)              RNEW_TST = 10._r8
   745)            endif
   746)            RNEW = RLSOG(L) - (RAX*FAX + RCA*FCA)     !GBA*CF
   747)            rnew_wrk(l) = rnew_tst
   748)            if ( debug ) then
   749)              if( is_hno3 .and. l == kdiag-1 ) then
   750)                write(*,*) ' '
   751)                write(*,*) 'washout: rls,rax,fax,rca,fca'
   752)                write(*,'(1p,5g15.7)') rls(l),rax,fax,rca,fca
   753)                write(*,*) ' '
   754)              endif
   755)            endif
   756) !-----------------------------------------------------------------------
   757) !  if RNEW>0, there is growth and/or new precip formation
   758) !-----------------------------------------------------------------------
   759) has_rnew:  if( rlsog(l) > adj_factor*(rax*fax + rca*fca) ) then
   760) !-----------------------------------------------------------------------
   761) !  Min cloudwater requirement for cloud with new precip
   762) !  Min CF is set at top for LE, at end for other levels
   763) !  CWMIN is only needed for new precip formation - do not need for RNEW<0
   764) !-----------------------------------------------------------------------
   765)              if( cfxx(l) == zero ) then
   766)                if ( do_diag ) then
   767)                  write(*,*) 'cfxx(l) == zero',l
   768)                  write(*,*) qttjfl(:,n)
   769)                  write(*,*) qm(:)
   770)                  write(*,*) pofl(:)
   771)                  write(*,*) delz(:)
   772)                  write(*,*) rls(:)
   773)                  write(*,*) clwc(:)
   774)                  write(*,*) ciwc(:)
   775)                  write(*,*) cfr(:)
   776)                  write(*,*) tem(:)
   777)                  write(*,*) evaprate(:)
   778)                  write(*,*) hstar(:,n)
   779)                end if
   780) !
   781) ! if we are here,, that means that there is
   782) ! a inconsistency and this will lead to a division
   783) ! by 0 later on! This column should then be skipped
   784) !
   785)                QTTJFL(:lpar,n) = QTT(:lpar)
   786)                cycle species_loop
   787) !
   788) !              call endrun()
   789) !
   790)              endif
   791)              rnew_flag(l) = .true.
   792)              CLWX = max( CLWC(L)+CIWC(L),CWMIN*CFXX(L) )
   793)              if( is_hno3 ) then
   794)                clwx_wrk(l) = clwx
   795)              endif
   796) !-----------------------------------------------------------------------
   797) !  Area of old cloud and new cloud
   798) !-----------------------------------------------------------------------
   799)              FCXA = FCA
   800)              FCXB = max( zero,CFXX(L)-FCXA )
   801) !-----------------------------------------------------------------------
   802) !                           ICE
   803) !  For ice and mixed phase, grow precip in old cloud by riming
   804) !  Use only portion of cloudwater in old cloud fraction
   805) !  and rain above old cloud fraction
   806) !  COLEFF from Lohmann and Roeckner (1996), Loss rate from Rotstayn (1997)
   807) !-----------------------------------------------------------------------
   808) is_freezing : &
   809)              if( freezing(l) ) then
   810)                COLEFFSNOW = exp( 2.5e-2_r8*(TEM(L) - TICE) )
   811)                if( TEM(L) <= TFROZ ) then
   812)                  RHOSNOW = RHOSNOWFIX
   813)                else
   814)                  RHOSNOW = 0.303_r8*(TEM(L) - TFROZ)*RHOSNOWFIX
   815)                endif
   816)                if( FCXA > zero ) then
   817)                  if( DCA > zero ) then
   818)                    DELTARIMEMASS = CLWX*QM(L)*(FCXA/CFXX(L))* &
   819)                      (one - exp( (-COLEFFSNOW/(DCA*1.e-3_r8))*((RCA)/(2._r8*RHOSNOW))*DTSCAV ))   !uses GBA R
   820)                  else
   821)                    DELTARIMEMASS = zero
   822)                  endif
   823)                else
   824)                  DELTARIMEMASS = zero
   825)                endif
   826) !-----------------------------------------------------------------------
   827) !  Increase in precip rate due to riming (kg/m2/s):
   828) !  Limit to total increase in R in cloud
   829) !-----------------------------------------------------------------------
   830)                if( FCXA > zero ) then
   831)                  DELTARIME = min( RNEW/FCXA,DELTARIMEMASS/(FCXA*GAREA*DTSCAV) ) !GBA
   832)                else
   833)                  DELTARIME = zero
   834)                endif
   835)                if( is_hno3 ) then
   836)                  deltarime_wrk(l) = deltarime
   837)                endif
   838) !-----------------------------------------------------------------------
   839) !  Find diameter of rimed precip, must be at least .1mm
   840) !-----------------------------------------------------------------------
   841)                if( RCA > zero ) then
   842)                  DOR = max( DMIN,(((RCA+DELTARIME)/RCA)**VOLPOW)*DCA )
   843)                else
   844)                  DOR = zero
   845)                endif
   846) !-----------------------------------------------------------------------
   847) !  If there is some in-cloud precip left, we have new precip formation
   848) !  Will be spread over whole cloud fraction 
   849) !-----------------------------------------------------------------------
   850) !  Calculate precip rate in old and new cloud fractions
   851) !-----------------------------------------------------------------------
   852)                RPRECIP = (RNEW-(DELTARIME*FCXA))/CFXX(L) !kg/m2/s    !GBA
   853) !-----------------------------------------------------------------------
   854) !  Calculate precip rate in old and new cloud fractions
   855) !-----------------------------------------------------------------------
   856)                RCXA = RCA + DELTARIME + RPRECIP          !kg/m2/s GBA
   857)                RCXB = RPRECIP                            !kg/m2/s GBA
   858) 
   859) !-----------------------------------------------------------------------
   860) !  Find diameter of new precip from empirical relation using Rprecip
   861) !  in given area of box- use density of water, not snow, to convert kg/s
   862) !  to mm/s -> as given in Field and Heymsfield
   863) !  Also calculate diameter of mixed precip,DCXA, from empirical relation
   864) !  using total R in FCXA - this will give larger particles than averaging DOR and
   865) !  DNEW in the next level
   866) !  DNEW and DCXA must be at least .1mm
   867) !-----------------------------------------------------------------------
   868)                if( RPRECIP > zero ) then
   869)                  WEMP = (CLWX*QM(L))/(GAREA*CFXX(L)*DELZ(L)) !kg/m3
   870)                  REMP = RPRECIP/((RHORAIN/1.e3_r8))             !mm/s local
   871)                  DNEW = DEMPIRICAL( WEMP, REMP )
   872)                  if ( debug ) then
   873)                    if( is_hno3 .and. l >= 15 ) then
   874)                      write(*,*) ' '
   875)                      write(*,*) 'washout: wemp,remp.dnew @ l = ',l
   876)                      write(*,'(1p,3g15.7)') wemp,remp,dnew
   877)                      write(*,*) ' '
   878)                    endif
   879)                  endif
   880)                  DNEW = max( DMIN,DNEW )
   881)                  if( FCXB > zero ) then
   882)                    DCXB = DNEW
   883)                  else
   884)                    DCXB = zero
   885)                  endif
   886)                else
   887)                  DCXB = zero
   888)                endif
   889) 
   890)                if( FCXA > zero ) then
   891)                  WEMP = (CLWX*QM(L)*(FCXA/CFXX(L)))/(GAREA*FCXA*DELZ(L)) !kg/m3
   892)                  REMP = RCXA/((RHORAIN/1.e3_r8))                         !mm/s local
   893)                  DEMP = DEMPIRICAL( WEMP, REMP )
   894)                  DCXA = ((RCA+DELTARIME)/RCXA)*DOR + (RPRECIP/RCXA)*DNEW
   895)                  DCXA = max( DEMP,DCXA )
   896)                  DCXA = max( DMIN,DCXA )
   897)                else
   898)                  DCXA = zero
   899)                endif
   900)                if ( debug ) then
   901)                  if( is_hno3 .and. l >= 15 ) then
   902)                    write(*,*) ' '
   903)                    write(*,*) 'washout: rca,rcxa,deltarime,dor,rprecip,dnew @ l = ',l
   904)                    write(*,'(1p,6g15.7)') rca,rcxa,deltarime,dor,rprecip,dnew 
   905)                    write(*,*) 'washout: dcxa,dcxb,wemp,remp,demp'
   906)                    write(*,'(1p,5g15.7)') dcxa,dcxb,wemp,remp,demp
   907)                    write(*,*) ' '
   908)                  end if
   909)                endif
   910) 
   911)                if( QTT(L) > zero ) then   
   912) !-----------------------------------------------------------------------
   913) !                       ICE SCAVENGING
   914) !-----------------------------------------------------------------------
   915) !  For ice, rainout only hno3/aerosols using new precip
   916) !  Tracer dissolved given by Kaercher and Voigt (2006) for T<258K
   917) !  For T>258K, use Henry's Law with Retention coefficient
   918) !  Rain out in whole CF
   919) !-----------------------------------------------------------------------
   920)                  if( RPRECIP > zero ) then
   921)                    if( LICETYP == 1 ) then
   922)                      RRAIN = RPRECIP*GAREA                                  !kg/s local
   923)                      call DISGAS( CLWX, CFXX(L), TCMASS(N), HSTAR(L,N), &
   924)                                   TEM(L),POFL(L),QM(L),                 &
   925)                                   QTT(L)*CFXX(L),QTDISCF )
   926)                      call RAINGAS( RRAIN, DTSCAV, CLWX, CFXX(L),        &
   927)                                    QM(L), QTT(L), QTDISCF, QTRAIN )
   928)                      WRK       = QTRAIN/CFXX(L)
   929)                      QTRAINCXA = FCXA*WRK
   930)                      QTRAINCXB = FCXB*WRK
   931)                    elseif( LICETYP == 2 ) then
   932)                      QTRAINCXA = zero
   933)                      QTRAINCXB = zero
   934)                    endif
   935)                    if( debug .and. is_hno3 .and. l == kdiag ) then
   936)                      write(*,*) ' '
   937)                      write(*,*) 'washout: Ice Scavenging'
   938)                      write(*,*) 'washout: qtraincxa, qtraincxb, fcxa, fcxb, qt_rain, cfxx(l), wrk @ level = ',l
   939)                      write(*,'(1p,7g15.7)') qtraincxa, qtraincxb, fcxa, fcxb, qt_rain(l), cfxx(l), wrk
   940)                      write(*,*) ' '
   941)                    endif
   942)                  endif
   943) !-----------------------------------------------------------------------
   944) !  For ice, accretion removal for hno3 and aerosols is propotional to riming, 
   945) !  no accretion removal for gases
   946) !  remove only in mixed portion of cloud
   947) !  Limit DELTARIMEMASS to RNEW*DTSCAV for ice - evaporation of rimed ice to match
   948) !  RNEW precip rate would result in HNO3 escaping from ice (no trapping) 
   949) !-----------------------------------------------------------------------
   950)                  if( DELTARIME > zero ) then
   951)                    if( LICETYP == 1 ) then
   952)                      if( TEM(L) <= TFROZ ) then
   953)                        RHOSNOW = RHOSNOWFIX
   954)                      else
   955)                        RHOSNOW = 0.303_r8*(TEM(L) - TFROZ)*RHOSNOWFIX
   956)                      endif
   957)                      QTCXA = QTT(L)*FCXA
   958)                      call DISGAS( CLWX*(FCXA/CFXX(L)), FCXA, TCMASS(N),   &
   959)                                   HSTAR(L,N), TEM(L), POFL(L),            &
   960)                                   QM(L), QTCXA, QTDISRIME )       
   961)                      QTDISSTAR = (QTDISRIME*QTCXA)/(QTDISRIME + QTCXA)
   962)                      if ( debug ) then
   963)                        if( is_hno3 .and. l >= 15 ) then
   964)                          write(*,*) ' '
   965)                          write(*,*) 'washout: fcxa,dca,rca,qtdisstar @ l = ',l
   966)                          write(*,'(1p,4g15.7)') fcxa,dca,rca,qtdisstar
   967)                          write(*,*) ' '
   968)                        endif
   969)                      endif
   970)                      QTRIMECXA = QTCXA*                             &
   971)                         (one - exp((-COLEFFSNOW/(DCA*1.e-3_r8))*       &
   972)                         (RCA/(2._r8*RHOSNOW))*                         &  !uses GBA R
