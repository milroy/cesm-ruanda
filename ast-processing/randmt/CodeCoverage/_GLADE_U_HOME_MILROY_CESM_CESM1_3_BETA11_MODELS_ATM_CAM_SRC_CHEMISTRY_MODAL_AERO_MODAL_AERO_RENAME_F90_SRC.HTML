    85) ! !PARAMETERS:
    86)    character(len=*), intent(in) :: fromwhere    ! identifies which module
    87)                                                 ! is making the call
    88)    integer,  intent(in)    :: lchnk                ! chunk identifier
    89)    integer,  intent(in)    :: ncol                 ! number of atmospheric column

    91)    integer,  intent(in)    :: loffset              ! offset applied to modal aero "ptrs"

    98)    real(r8), intent(in)    :: qqcw(ncol,pver,pcnstxx) ! like q but for cloud-borne species

   100)    real(r8), intent(inout) :: dqdt(ncol,pver,pcnstxx)  ! TMR tendency array;
   101)                               ! incoming dqdt = tendencies for the 
   102)                               !     "fromwhere" continuous growth process 
   103)                               ! the renaming tendencies are added on
   104)                               ! *** NOTE ncol and pcnstxx dimensions
   105)    real(r8), intent(inout) :: dqqcwdt(ncol,pver,pcnstxx)
   106)    real(r8), intent(in)    :: dqdt_other(ncol,pver,pcnstxx)  
   107)                               ! tendencies for "other" continuous growth process 
   108)                               ! currently in cam3
   109)                               !     dqdt is from gas (h2so4, nh3) condensation
   110)                               !     dqdt_other is from aqchem and soa
   111)                               ! *** NOTE ncol and pcnstxx dimensions
   112)    real(r8), intent(in)    :: dqqcwdt_other(ncol,pver,pcnstxx)  
   113)    logical,  intent(inout) :: dotendrn(pcnstxx) ! identifies the species for which
   114)                               !     renaming dqdt is computed
   115)    logical,  intent(inout) :: dotendqqcwrn(pcnstxx)
   116) 
   117)    logical,  intent(in)    :: is_dorename_atik          ! true if dorename_atik is provided
   118)    logical,  intent(in)    :: dorename_atik(ncol,pver) ! true if renaming should
   119)                                                         ! be done at i,k
   120)    integer,  intent(in)    :: jsrflx_rename        ! qsrflx index for renaming
   121)    integer,  intent(in)    :: nsrflx               ! last dimension of qsrflx
   122) 
   123)    real(r8), intent(inout) :: qsrflx(pcols,pcnstxx,nsrflx)
   124)                               ! process-specific column tracer tendencies 
   125)    real(r8), intent(inout) :: qqcwsrflx(pcols,pcnstxx,nsrflx)
   126) 
   127) ! !DESCRIPTION: 
   128) ! computes TMR (tracer mixing ratio) tendencies for "mode renaming"
   129) !    during a continuous growth process
   130) ! currently this transfers number and mass (and surface) from the aitken
   131) !    to accumulation mode after gas condensation or stratiform-cloud
   132) !    aqueous chemistry
   133) ! (convective cloud aqueous chemistry not yet implemented)
   134) !
   135) ! !REVISION HISTORY:
   136) !   RCE 07.04.13:  Adapted from MIRAGE2 code
   137) !
   138) !EOP
   139) !----------------------------------------------------------------------
   140) !BOC
   141) 
   142) ! local variables
   143)    integer, parameter :: ldiag1=-1
   144)    integer :: i, icol_diag, ipair, iq, j, k, l, l1, l2, la, lc, lunout
   145)    integer :: lsfrma, lsfrmc, lstooa, lstooc
   146)    integer :: mfrm, mtoo, n, n1, n2, ntot_msa_a
   147)    integer :: idomode(ntot_amode)
   148)    integer, save :: lun = -1  ! logical unit for diagnostics (6, or other
   149)                               ! if a special diagnostics file is opened)
   150) 
   151) 
   152)    real (r8) :: deldryvol_a(ncol,pver,ntot_amode)
   153)    real (r8) :: deldryvol_c(ncol,pver,ntot_amode)
   154)    real (r8) :: deltatinv
   155)    real (r8) :: dp_belowcut(maxpair_renamexf)
   156)    real (r8) :: dp_cut(maxpair_renamexf)
   157)    real (r8) :: dgn_aftr, dgn_xfer
   158)    real (r8) :: dgn_t_new, dgn_t_old
   159)    real (r8) :: dryvol_t_del, dryvol_t_new
   160)    real (r8) :: dryvol_t_old, dryvol_t_oldbnd
   161)    real (r8) :: dryvol_a(ncol,pver,ntot_amode)
   162)    real (r8) :: dryvol_c(ncol,pver,ntot_amode)
   163)    real (r8) :: dryvol_smallest(ntot_amode)
   164)    real (r8) :: dum
   165)    real (r8) :: dum3alnsg2(maxpair_renamexf)
   166)    real (r8) :: dum_m2v, dum_m2vdt
   167)    real (r8) :: factoraa(ntot_amode)
   168)    real (r8) :: factoryy(ntot_amode)
   169)    real (r8) :: frelax
   170)    real (r8) :: lndp_cut(maxpair_renamexf)
   171)    real (r8) :: lndgn_new, lndgn_old
   172)    real (r8) :: lndgv_new, lndgv_old
   173)    real (r8) :: num_t_old, num_t_oldbnd
   174)    real (r8) :: onethird
   175)    real (r8) :: pdel_fac
   176)    real (r8) :: tailfr_volnew, tailfr_volold
   177)    real (r8) :: tailfr_numnew, tailfr_numold
   178)    real (r8) :: v2nhirlx(ntot_amode), v2nlorlx(ntot_amode)
   179)    real (r8) :: xfercoef, xfertend
   180)    real (r8) :: xferfrac_vol, xferfrac_num, xferfrac_max
   181) 
   182)    real (r8) :: yn_tail, yv_tail
   183) 
   184) ! begin
   185) 	lunout = 6
   186) 
   187) !   get logical unit (for output to dumpconv, deactivate the "lun = 6")
   188)  	lun = 6
   189) 	if (lun < 1) then
   190) 	   lun = getunit()
   191)  	   open( unit=lun, file='dump.rename',   &
   192)  			status='unknown', form='formatted' )
   193) 	end if
   194) 
   195) 
   196) !
   197) !   calculations done once on initial entry
   198) !
   199) !   "init" is now done through chem_init (and things under it)
   200) !	if (npair_renamexf .eq. -123456789) then
   201) !	    npair_renamexf = 0
   202) !	    call modal_aero_rename_init
   203) !	end if
   204) 
   205) !
   206) !   check if any renaming pairs exist
   207) !
   208) 	if (npair_renamexf .le. 0) return
   209) ! 	if (ncol .ne. -123456789) return
   210) !	if (fromwhere .eq. 'aqchem') return
   211) 
   212) !
   213) !   compute aerosol dry-volume for the "from mode" of each renaming pair
   214) !   also compute dry-volume change during the continuous growth process
   215) !	using the incoming dqdt*deltat
   216) !
   217) 	deltatinv = 1.0_r8/(deltat*(1.0_r8 + 1.0e-15_r8))
   218) 	onethird = 1.0_r8/3.0_r8
   219) 	frelax = 27.0_r8
   220) 	xferfrac_max = 1.0_r8 - 10.0_r8*epsilon(1.0_r8)   ! 1-eps
   221) 
   222) 	do n = 1, ntot_amode
   223) 	    idomode(n) = 0
   224) 	end do
   225) 
   226) 	do ipair = 1, npair_renamexf
   227) 	    if (ipair .gt. 1) goto 8100
   228) 	    idomode(modefrm_renamexf(ipair)) = 1
   229) 
   230) 	    mfrm = modefrm_renamexf(ipair)
   231) 	    mtoo = modetoo_renamexf(ipair)
   232) 	    factoraa(mfrm) = (pi/6._r8)*exp(4.5_r8*(alnsg_amode(mfrm)**2))
   233) 	    factoraa(mtoo) = (pi/6._r8)*exp(4.5_r8*(alnsg_amode(mtoo)**2))
   234) 	    factoryy(mfrm) = sqrt( 0.5_r8 )/alnsg_amode(mfrm)
   235) !   dryvol_smallest is a very small volume mixing ratio (m3-AP/kmol-air)
   236) !   used for avoiding overflow.  it corresponds to dp = 1 nm
   237) !   and number = 1e-5 #/mg-air ~= 1e-5 #/cm3-air
   238) 	    dryvol_smallest(mfrm) = 1.0e-25_r8
   239) 	    v2nlorlx(mfrm) = voltonumblo_amode(mfrm)*frelax
   240) 	    v2nhirlx(mfrm) = voltonumbhi_amode(mfrm)/frelax
   241) 
   242) 	    dum3alnsg2(ipair) = 3.0_r8 * (alnsg_amode(mfrm)**2)
   243) 	    dp_cut(ipair) = sqrt(   &
   244) 		dgnum_amode(mfrm)*exp(1.5_r8*(alnsg_amode(mfrm)**2)) *   &
   245) 		dgnum_amode(mtoo)*exp(1.5_r8*(alnsg_amode(mtoo)**2)) )
   246) 	    lndp_cut(ipair) = log( dp_cut(ipair) )
   247) 	    dp_belowcut(ipair) = 0.99_r8*dp_cut(ipair)
   248) 	end do
   249) 
   250) 	do n = 1, ntot_amode
   251) 	    if (idomode(n) .gt. 0) then
   252) 		dryvol_a(1:ncol,:,n) = 0.0_r8
   253) 		dryvol_c(1:ncol,:,n) = 0.0_r8
   254) 		deldryvol_a(1:ncol,:,n) = 0.0_r8
   255) 		deldryvol_c(1:ncol,:,n) = 0.0_r8
   256) 		do l1 = 1, nspec_amode(n)
   257) 		    l2 = lspectype_amode(l1,n)
   258) !   dum_m2v converts (kmol-AP/kmol-air) to (m3-AP/kmol-air)
   259) !            [m3-AP/kmol-AP]= [kg-AP/kmol-AP]  / [kg-AP/m3-AP]
   260) 		    dum_m2v = specmw_amode(l2) / specdens_amode(l2)
   261) 		    dum_m2vdt = dum_m2v*deltat
   262) 		    la = lmassptr_amode(l1,n)-loffset
   263) 		    if (la > 0) then
   264) 		    dryvol_a(1:ncol,:,n) = dryvol_a(1:ncol,:,n)    &
   265) 			+ dum_m2v*max( 0.0_r8,   &
   266)                           q(1:ncol,:,la)-deltat*dqdt_other(1:ncol,:,la) )
   267) 		    deldryvol_a(1:ncol,:,n) = deldryvol_a(1:ncol,:,n)    &
   268) 			+ (dqdt_other(1:ncol,:,la) + dqdt(1:ncol,:,la))*dum_m2vdt
   269) 		    end if
   270) 
   271) 		    lc = lmassptrcw_amode(l1,n)-loffset
   272) 		    if (lc > 0) then
   273) 		    dryvol_c(1:ncol,:,n) = dryvol_c(1:ncol,:,n)    &
   274) 			+ dum_m2v*max( 0.0_r8,   &
   275)                           qqcw(1:ncol,:,lc)-deltat*dqqcwdt_other(1:ncol,:,lc) )
   276) 		    deldryvol_c(1:ncol,:,n) = deldryvol_c(1:ncol,:,n)    &
   277) 			+ (dqqcwdt_other(1:ncol,:,lc) +   &
   278) 			         dqqcwdt(1:ncol,:,lc))*dum_m2vdt
   279) 		    end if
   280) 		end do
   281) 	    end if
   282) 	end do
   283) 
   284) 
   285) 
   286) !
   287) !   loop over levels and columns to calc the renaming
   288) !
   289) mainloop1_k:  do k = 1, pver
   290) mainloop1_i:  do i = 1, ncol
   291) 
   292) !   if dorename_atik is provided, then check if renaming needed at this i,k
   293) 	if (is_dorename_atik) then
   294) 	    if (.not. dorename_atik(i,k)) cycle mainloop1_i
   295) 	end if
   296) 	pdel_fac = pdel(i,k)/gravit
   297) 
   298) !
   299) !   loop over renameing pairs
   300) !
   301) mainloop1_ipair:  do ipair = 1, npair_renamexf
   302) 
   303) 	mfrm = modefrm_renamexf(ipair)
   304) 	mtoo = modetoo_renamexf(ipair)
   305) 
   306) !   dryvol_t_old is the old total (a+c) dry-volume for the "from" mode 
   307) !	in m^3-AP/kmol-air
   308) !   dryvol_t_new is the new total dry-volume
   309) !	(old/new = before/after the continuous growth)
   310) 	dryvol_t_old = dryvol_a(i,k,mfrm) + dryvol_c(i,k,mfrm)
   311) 	dryvol_t_del = deldryvol_a(i,k,mfrm) + deldryvol_c(i,k,mfrm)
   312) 	dryvol_t_new = dryvol_t_old + dryvol_t_del
   313) 	dryvol_t_oldbnd = max( dryvol_t_old, dryvol_smallest(mfrm) )
   314) 
   315) !   no renaming if dryvol_t_new ~ 0 or dryvol_t_del ~ 0
   316) 	if (dryvol_t_new .le. dryvol_smallest(mfrm)) cycle mainloop1_ipair
   317) 	if (dryvol_t_del .le. 1.0e-6_r8*dryvol_t_oldbnd) cycle mainloop1_ipair
   318) 
   319) !   num_t_old is total number in particles/kmol-air
   320) 	num_t_old = q(i,k,numptr_amode(mfrm)-loffset)
   321) 	num_t_old = num_t_old + qqcw(i,k,numptrcw_amode(mfrm)-loffset)
   322) 	num_t_old = max( 0.0_r8, num_t_old )
   323) 	dryvol_t_oldbnd = max( dryvol_t_old, dryvol_smallest(mfrm) )
   324) 	num_t_oldbnd = min( dryvol_t_oldbnd*v2nlorlx(mfrm), num_t_old )
   325) 	num_t_oldbnd = max( dryvol_t_oldbnd*v2nhirlx(mfrm), num_t_oldbnd )
   326) 
   327) !   no renaming if dgnum < "base" dgnum, 
   328) 	dgn_t_new = (dryvol_t_new/(num_t_oldbnd*factoraa(mfrm)))**onethird
   329) 	if (dgn_t_new .le. dgnum_amode(mfrm)) cycle mainloop1_ipair
   330) 
   331) !   compute new fraction of number and mass in the tail (dp > dp_cut)
   332) 	lndgn_new = log( dgn_t_new )
   333) 	lndgv_new = lndgn_new + dum3alnsg2(ipair)
   334) 	yn_tail = (lndp_cut(ipair) - lndgn_new)*factoryy(mfrm)
   335) 	yv_tail = (lndp_cut(ipair) - lndgv_new)*factoryy(mfrm)
   336) 	tailfr_numnew = 0.5_r8*erfc( yn_tail )
   337) 	tailfr_volnew = 0.5_r8*erfc( yv_tail )
   338) 
   339) !   compute old fraction of number and mass in the tail (dp > dp_cut)
   340) 	dgn_t_old =   &
   341) 		(dryvol_t_oldbnd/(num_t_oldbnd*factoraa(mfrm)))**onethird
   342) !   if dgn_t_new exceeds dp_cut, use the minimum of dgn_t_old and 
   343) !   dp_belowcut to guarantee some transfer
   344) 	if (dgn_t_new .ge. dp_cut(ipair)) then
   345) 	    dgn_t_old = min( dgn_t_old, dp_belowcut(ipair) )
   346) 	end if
   347) 	lndgn_old = log( dgn_t_old )
   348) 	lndgv_old = lndgn_old + dum3alnsg2(ipair)
   349) 	yn_tail = (lndp_cut(ipair) - lndgn_old)*factoryy(mfrm)
   350) 	yv_tail = (lndp_cut(ipair) - lndgv_old)*factoryy(mfrm)
   351) 	tailfr_numold = 0.5_r8*erfc( yn_tail )
   352) 	tailfr_volold = 0.5_r8*erfc( yv_tail )
   353) 
   354) !   transfer fraction is difference between new and old tail-fractions
   355) !   transfer fraction for number cannot exceed that of mass
   356) 	dum = tailfr_volnew*dryvol_t_new - tailfr_volold*dryvol_t_old
   357) 	if (dum .le. 0.0_r8) cycle mainloop1_ipair
   358) 
   359) 	xferfrac_vol = min( dum, dryvol_t_new )/dryvol_t_new
   360) 	xferfrac_vol = min( xferfrac_vol, xferfrac_max ) 
   361) 	xferfrac_num = tailfr_numnew - tailfr_numold
   362) 	xferfrac_num = max( 0.0_r8, min( xferfrac_num, xferfrac_vol ) )
   363) 
   364) !   diagnostic output start ----------------------------------------
   365) !!$ 	if (ldiag1 > 0) then
   366) !!$ 	icol_diag = -1
   367) !!$ 	if ((lonndx(i) == 37) .and. (latndx(i) == 23)) icol_diag = i
   368) !!$ 	if ((i == icol_diag) .and. (mod(k-1,5) == 0)) then
   369) !!$ !	write(lun,97010) fromwhere, nstep, lchnk, i, k, ipair
   370) !!$ 	write(lun,97010) fromwhere, nstep, latndx(i), lonndx(i), k, ipair
   371) !!$ 	write(lun,97020) 'drv old/oldbnd/new/del     ',   &
   372) !!$ 		dryvol_t_old, dryvol_t_oldbnd, dryvol_t_new, dryvol_t_del
   373) !!$ 	write(lun,97020) 'num old/oldbnd, dgnold/new ',   &
   374) !!$ 		num_t_old, num_t_oldbnd, dgn_t_old, dgn_t_new
   375) !!$ 	write(lun,97020) 'tailfr v_old/new, n_old/new',   &
   376) !!$ 		tailfr_volold, tailfr_volnew, tailfr_numold, tailfr_numnew
   377) !!$ 	dum = max(1.0e-10_r8,xferfrac_vol) / max(1.0e-10_r8,xferfrac_num)
   378) !!$ 	dgn_xfer = dgn_t_new * dum**onethird
   379) !!$ 	dum = max(1.0e-10_r8,(1.0_r8-xferfrac_vol)) /   &
   380) !!$               max(1.0e-10_r8,(1.0_r8-xferfrac_num))
   381) !!$ 	dgn_aftr = dgn_t_new * dum**onethird
   382) !!$ 	write(lun,97020) 'xferfrac_v/n; dgn_xfer/aftr',   &
   383) !!$ 		xferfrac_vol, xferfrac_num, dgn_xfer, dgn_aftr
   384) !!$ !97010	format( / 'RENAME ', a, '  nx,lc,i,k,ip', i8, 4i4 )
   385) !!$ 97010	format( / 'RENAME ', a, '  nx,lat,lon,k,ip', i8, 4i4 )
   386) !!$ 97020	format( a, 6(1pe15.7) )
   387) !!$ 	end if
   388) !!$ 	end if
   389) !   diagnostic output end   ------------------------------------------
   390) 
   391) 
   392) !
   393) !   compute tendencies for the renaming transfer
   394) !
   395) 	j = jsrflx_rename
   396) 	do iq = 1, nspecfrm_renamexf(ipair)
   397) 	    xfercoef = xferfrac_vol*deltatinv
   398) 	    if (iq .eq. 1) xfercoef = xferfrac_num*deltatinv
   399) 
   400) 	    lsfrma = lspecfrma_renamexf(iq,ipair)-loffset
   401) 	    lsfrmc = lspecfrmc_renamexf(iq,ipair)-loffset
   402) 	    lstooa = lspectooa_renamexf(iq,ipair)-loffset
