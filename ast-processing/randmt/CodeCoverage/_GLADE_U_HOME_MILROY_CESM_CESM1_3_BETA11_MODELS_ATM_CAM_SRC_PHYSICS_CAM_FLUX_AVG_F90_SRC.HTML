   173) subroutine smooth(new, old, res, nstep, deltat, ncol)
   174) 
   175)    real(r8), intent(inout) :: new(pcols)
   176)    real(r8), intent(inout) :: old(pcols)
   177)    real(r8), intent(inout) :: res(pcols)
   178)    real(r8), intent(in)    :: deltat
   179)    integer,  intent(in)    :: nstep
   180)    integer,  intent(in)    :: ncol
   181) 
   182)    real(r8) :: temp(pcols)
   183)    integer i
   184) 
   185)    temp(1:ncol) = new(1:ncol)
   186)    if (nstep > 0) then
   187)       new(1:ncol) = 0.5_r8*(new(1:ncol)+old(1:ncol))
   188)    else
   189)       old(1:ncol) = new(1:ncol)
   190)       res(1:ncol) = 0._r8
   191)    endif
   192) 
   193)    ! storing the old value for smoothing on the next step
   194)    ! doesnt seem to be stable
   195)    ! old(1:ncol) = temp(1:ncol)
   196) 
   197)    ! storing the smoothed value for the next step
   198) 
   199)    ! first add the flux that the surface model wanted to provide less
   200)    ! the flux the atmosphere will actually see to the residual
   201)    res(1:ncol) = res(1:ncol) + temp(1:ncol)-new(1:ncol)
   202) 
   203)    ! now calculate the amount that we might increment the new flux
   204)    ! to include some of the residual
   205)    ! If the residual is small we will just add it all, 
   206)    ! but if it is large we will add it at the rate required to put
   207)    ! the residual back into the flux over a 2 hour period
   208)    do i = 1,ncol
   209)       if (abs(res(i)).lt.max(abs(new(i)),abs(old(i)))*0.05_r8) then
   210)          temp(i) = res(i)
   211)          res(i) = 0._r8
   212)       else
   213)          temp(i) = res(i)*deltat/7200._r8
   214)          !     temp(i) = res(i)*deltat*0.5/7200.
   215)          res(i) = res(i)-temp(i)
   216)       endif
   217)    end do
   218) 
   219)    ! dont do conservative smoothing for first 12 hours
   220)    if (nstep*deltat/86400._r8 < 0.5_r8) then
   221)       ! use this line if your dont want to use the residual
   222)       !if (.true.) then
   223)       temp = 0._r8
   224)       res = 0._r8
   225)    endif
   226) 
   227)    ! make the new flux the average of the sfc model and last timestep
   228)    ! plus some of the residual
   229)    new(1:ncol) = new(1:ncol) + temp(1:ncol)
   230)    old(1:ncol) = new(1:ncol)
   231) 
   232) end subroutine smooth
