   901)   function Factor(num) result(res)
   902) 
   903)     implicit none
   904)     integer,intent(in)  :: num
   905) 
   906)     type (factor_t)     :: res
   907)     integer             :: tmp,tmp2,tmp3,tmp5
   908)     integer             :: i,n
   909)     logical             :: found
   910) 
   911)     ! --------------------------------------
   912)     ! Allocate for max # of factors
   913)     ! --------------------------------------
   914)     tmp = num
   915)     tmp2 = log2(num)
   916)     allocate(res%factors(tmp2))
   917) 
   918)     n=0
   919)     !-----------------------
   920)     !  Look for factors of 2
   921)     !-----------------------
   922)     found=.TRUE.
   923)     do while (found)
   924)        found = .FALSE.
   925)        tmp2 = tmp/2
   926)        if( tmp2*2 == tmp ) then
   927)           n = n + 1
   928)           res%factors(n) = 2
   929)           found = .TRUE.
   930)           tmp = tmp2
   931)        endif
   932)     enddo
   933) 
   934)     !-----------------------
   935)     !  Look for factors of 3
   936)     !-----------------------
   937)     found=.TRUE.
   938)     do while (found)
   939)        found = .FALSE.
   940)        tmp3 = tmp/3
   941)        if( tmp3*3 == tmp ) then
   942)           n = n + 1
   943)           res%factors(n) = 3
   944)           found = .TRUE.
   945)           tmp = tmp3
   946)        endif
   947)     enddo
   948) 
   949)     !-----------------------
   950)     !  Look for factors of 5
   951)     !-----------------------
   952)     found=.TRUE.
   953)     do while (found)
   954)        found = .FALSE.
   955)        tmp5 = tmp/5
   956)        if( tmp5*5 == tmp ) then
   957)           n = n + 1
   958)           res%factors(n) = 5
   959)           found = .TRUE.
   960)           tmp = tmp5
   961)        endif
   962)     enddo
   963) 
   964)     tmp=1
   965)     do i=1,n
   966)        tmp = tmp * res%factors(i)
   967)     enddo
   968)     if(tmp == num) then
   969)        res%numfact = n
   970)     else
   971)        res%numfact = -1
   972)     endif
   973) 
   974)   end function Factor
