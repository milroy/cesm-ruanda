    90) 
    91) ! !INPUT/OUTPUT PARAMETERS:
    92) 
    93)    real (dbl_kind), intent(inout) :: &
    94)       scalar               ! scalar to be broadcast
    95) 
    96) !EOP

    99) !
   100) !  local variables
   101) !
   102) !-----------------------------------------------------------------------
   103) 
   104)    integer (int_kind) :: ierr  ! local MPI error flag
   105)

   108)    call MPI_BCAST(scalar, 1, mpiR8, root_pe, MPI_COMM_ICE, ierr)
   109)    call MPI_BARRIER(MPI_COMM_ICE, ierr)

   520) !-----------------------------------------------------------------------

   522) !  local variables

   524) !-----------------------------------------------------------------------
   525) 
   526)    integer (int_kind), dimension(:), allocatable :: &
   527)       array_int            ! temporary array for MPI bcast
   528) 
   529)    integer (int_kind) :: &
   530)       nelements,          &! size of array to be broadcast
   531)       ierr                 ! local MPI error flag
   532) 
   533) !-----------------------------------------------------------------------
   534) 
   535)    nelements = size(array)
   536)    allocate(array_int(nelements))
   537) 
   538)    where (array)
   539)      array_int = 1
   540)    elsewhere
   541)      array_int = 0
   542)    end where
   543) 
   544)    call MPI_BCAST(array_int, nelements, MPI_INTEGER, root_pe, &
   545)                   MPI_COMM_ICE, ierr)
   546)    call MPI_BARRIER(MPI_COMM_ICE, ierr)
   547) 
   548)    where (array_int == 1)
   549)      array = .true.
   550)    elsewhere
   551)      array = .false.
   552)    end where
   553) 
   554)    deallocate(array_int)
   555) 
   556) !-----------------------------------------------------------------------
   557) !EOC
   558) 
   559)  end subroutine broadcast_array_log_1d
