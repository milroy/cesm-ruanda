    15) 
    16)   subroutine lnd_import( bounds, x2l, a2l, a2l_not_downscaled_gcell, x2s)
    17) 
    18)     !---------------------------------------------------------------------------
    19)     ! !DESCRIPTION:
    20)     ! Convert the input data from the coupler to the land model 
    21)     !
    22)     ! !USES:
    23)     use clm_atmlnd      , only: atm2lnd_type, atm2lnd_downscaled_fields_type
    24)     use clm_glclnd      , only: glc2lnd_type
    25)     use clm_varctl      , only: co2_type, co2_ppmv, iulog, use_c13, create_glacier_mec_landunit
    26)     use clm_varcon      , only: rair, o2_molar_const, c13ratio
    27)     use shr_const_mod   , only: SHR_CONST_TKFRZ
    28)     use domainMod       , only: ldomain
    29)     implicit none
    30)     !
    31)     ! !ARGUMENTS:
    32)     type(bounds_type)                    , intent(in)    :: bounds                 ! bounds
    33)     real(r8)                             , intent(in)    :: x2l(:,:)               ! driver import state to land model
    34)     type(atm2lnd_type)                   , intent(inout) :: a2l                    ! clm internal input data type
    35)     type(atm2lnd_downscaled_fields_type) , intent(inout) :: a2l_not_downscaled_gcell ! clm internal input data type
    36)     type(glc2lnd_type)                   , intent(inout) :: x2s                    ! clm internal input data type
    37)     !
    38)     ! !LOCAL VARIABLES:
    39)     integer  :: g,i,nstep,ier        ! indices, number of steps, and error code
    40)     real(r8) :: forc_rainc           ! rainxy Atm flux mm/s
    41)     real(r8) :: e                    ! vapor pressure (Pa)
    42)     real(r8) :: qsat                 ! saturation specific humidity (kg/kg)
    43)     real(r8) :: forc_t               ! atmospheric temperature (Kelvin)
    44)     real(r8) :: forc_q               ! atmospheric specific humidity (kg/kg)
    45)     real(r8) :: forc_pbot            ! atmospheric pressure (Pa)
    46)     real(r8) :: forc_rainl           ! rainxy Atm flux mm/s
    47)     real(r8) :: forc_snowc           ! snowfxy Atm flux  mm/s
    48)     real(r8) :: forc_snowl           ! snowfxl Atm flux  mm/s
    49)     real(r8) :: co2_ppmv_diag        ! temporary
    50)     real(r8) :: co2_ppmv_prog        ! temporary
    51)     real(r8) :: co2_ppmv_val         ! temporary
    52)     integer  :: co2_type_idx         ! integer flag for co2_type options
    53)     real(r8) :: esatw                ! saturation vapor pressure over water (Pa)
    54)     real(r8) :: esati                ! saturation vapor pressure over ice (Pa)
    55)     real(r8) :: a0,a1,a2,a3,a4,a5,a6 ! coefficients for esat over water
    56)     real(r8) :: b0,b1,b2,b3,b4,b5,b6 ! coefficients for esat over ice
    57)     real(r8) :: tdc, t               ! Kelvins to Celcius function and its input
    58)     integer  :: num                  ! counter
    59)     character(len=32), parameter :: sub = 'lnd_import_mct'
    60) 
    61)     ! Constants to compute vapor pressure
    62)     parameter (a0=6.107799961_r8    , a1=4.436518521e-01_r8, &
    63)          a2=1.428945805e-02_r8, a3=2.650648471e-04_r8, &
    64)          a4=3.031240396e-06_r8, a5=2.034080948e-08_r8, &
    65)          a6=6.136820929e-11_r8)
    66) 
    67)     parameter (b0=6.109177956_r8    , b1=5.034698970e-01_r8, &
    68)          b2=1.886013408e-02_r8, b3=4.176223716e-04_r8, &
    69)          b4=5.824720280e-06_r8, b5=4.838803174e-08_r8, &
    70)          b6=1.838826904e-10_r8)
    71)     !
    72)     ! function declarations
    73)     !
    74)     tdc(t) = min( 50._r8, max(-50._r8,(t-SHR_CONST_TKFRZ)) )
    75)     esatw(t) = 100._r8*(a0+t*(a1+t*(a2+t*(a3+t*(a4+t*(a5+t*a6))))))
    76)     esati(t) = 100._r8*(b0+t*(b1+t*(b2+t*(b3+t*(b4+t*(b5+t*b6))))))
    77)     !---------------------------------------------------------------------------
    78) 
    79)     co2_type_idx = 0
    80)     if (co2_type == 'prognostic') then
    81)        co2_type_idx = 1
    82)     else if (co2_type == 'diagnostic') then
    83)        co2_type_idx = 2
    84)     end if
    85)     if (co2_type == 'prognostic' .and. index_x2l_Sa_co2prog == 0) then
    86)        call endrun( sub//' ERROR: must have nonzero index_x2l_Sa_co2prog for co2_type equal to prognostic' )
    87)     else if (co2_type == 'diagnostic' .and. index_x2l_Sa_co2diag == 0) then
    88)        call endrun( sub//' ERROR: must have nonzero index_x2l_Sa_co2diag for co2_type equal to diagnostic' )
    89)     end if
    90) 
    91)     ! Note that the precipitation fluxes received  from the coupler
    92)     ! are in units of kg/s/m^2. To convert these precipitation rates
    93)     ! in units of mm/sec, one must divide by 1000 kg/m^3 and multiply
    94)     ! by 1000 mm/m resulting in an overall factor of unity.
    95)     ! Below the units are therefore given in mm/s.
    96) 
    97) 
    98)     do g = bounds%begg,bounds%endg
    99)        i = 1 + (g - bounds%begg)
   100) 
   101)        ! Determine flooding input, sign convention is positive downward and
   102)        ! hierarchy is atm/glc/lnd/rof/ice/ocn.  so water sent from rof to land is negative,
   103)        ! change the sign to indicate addition of water to system.
   104) 
   105)        a2l%forc_flood(g)   = -x2l(index_x2l_Flrr_flood,i)  
   106) 
   107)        a2l%volr(g)   = x2l(index_x2l_Flrr_volr,i) * (ldomain%area(g) * 1.e6_r8)
   108) 
   109)        ! Determine required receive fields
   110) 
   111)        a2l%forc_hgt(g)     = x2l(index_x2l_Sa_z,i)         ! zgcmxy  Atm state m
   112)        a2l%forc_u(g)       = x2l(index_x2l_Sa_u,i)         ! forc_uxy  Atm state m/s
   113)        a2l%forc_v(g)       = x2l(index_x2l_Sa_v,i)         ! forc_vxy  Atm state m/s
   114)        a2l%forc_solad(g,2) = x2l(index_x2l_Faxa_swndr,i)   ! forc_sollxy  Atm flux  W/m^2
   115)        a2l%forc_solad(g,1) = x2l(index_x2l_Faxa_swvdr,i)   ! forc_solsxy  Atm flux  W/m^2
   116)        a2l%forc_solai(g,2) = x2l(index_x2l_Faxa_swndf,i)   ! forc_solldxy Atm flux  W/m^2
   117)        a2l%forc_solai(g,1) = x2l(index_x2l_Faxa_swvdf,i)   ! forc_solsdxy Atm flux  W/m^2
   118) 
   119)        a2l_not_downscaled_gcell%forc_th(g)      = x2l(index_x2l_Sa_ptem,i)      ! forc_thxy Atm state K
   120)        a2l_not_downscaled_gcell%forc_q(g)       = x2l(index_x2l_Sa_shum,i)      ! forc_qxy  Atm state kg/kg
   121)        a2l_not_downscaled_gcell%forc_pbot(g)    = x2l(index_x2l_Sa_pbot,i)      ! ptcmxy  Atm state Pa
   122)        a2l_not_downscaled_gcell%forc_t(g)       = x2l(index_x2l_Sa_tbot,i)      ! forc_txy  Atm state K
   123)        a2l_not_downscaled_gcell%forc_lwrad(g)   = x2l(index_x2l_Faxa_lwdn,i)    ! flwdsxy Atm flux  W/m^2
   124) 
   125)        forc_rainc          = x2l(index_x2l_Faxa_rainc,i)   ! mm/s
   126)        forc_rainl          = x2l(index_x2l_Faxa_rainl,i)   ! mm/s
   127)        forc_snowc          = x2l(index_x2l_Faxa_snowc,i)   ! mm/s
   128)        forc_snowl          = x2l(index_x2l_Faxa_snowl,i)   ! mm/s
   129) 
   130)        !
   131)        ! anomaly forcing code 
   132)        !
   133)        ! bias correct atmospheric input fields if streams exist
   134)        if (index_x2l_Sa_precsf /= 0) then
   135)           a2l%bc_precip(g)    = x2l(index_x2l_Sa_precsf,i)
   136)           a2l%bc_precip(g)    = min(1.e2_r8,a2l%bc_precip(g))
   137)           forc_rainc          = forc_rainc * a2l%bc_precip(g)
   138)           forc_rainl          = forc_rainl * a2l%bc_precip(g)
   139)           forc_snowc          = forc_snowc * a2l%bc_precip(g)
   140)           forc_snowl          = forc_snowl * a2l%bc_precip(g)
   141)        endif
   142) 
   143)        ! adjust atmospheric input fields if anomaly forcing streams exist
   144)        if (index_x2l_Sa_u_af /= 0) then
   145)           a2l%af_uwind(g)    = x2l(index_x2l_Sa_u_af,i)
   146)           a2l%forc_u(g) = a2l%forc_u(g) + a2l%af_uwind(g)
   147)        endif
   148) 
   149)        if (index_x2l_Sa_v_af /= 0) then
   150)           a2l%af_vwind(g)    = x2l(index_x2l_Sa_v_af,i)
   151)           a2l%forc_v(g) = a2l%forc_v(g) + a2l%af_vwind(g)
   152)        endif
   153) 
   154)        if (index_x2l_Sa_shum_af /= 0) then
   155)           a2l%af_shum(g)    = x2l(index_x2l_Sa_shum_af,i)
   156)           a2l_not_downscaled_gcell%forc_q(g) = a2l_not_downscaled_gcell%forc_q(g) + a2l%af_shum(g)
   157)           ! avoid possible negative q values
   158)           if(a2l_not_downscaled_gcell%forc_q(g) < 0._r8) then 
   159)              a2l_not_downscaled_gcell%forc_q(g) = 1.e-6_r8
   160)           endif
   161)        endif
   162) 
   163)        if (index_x2l_Sa_pbot_af /= 0) then
   164)           a2l%af_pbot(g)    = x2l(index_x2l_Sa_pbot_af,i)
   165)           a2l_not_downscaled_gcell%forc_pbot(g) =  a2l_not_downscaled_gcell%forc_pbot(g) + a2l%af_pbot(g)
   166)        endif
   167) 
   168)        if (index_x2l_Sa_tbot_af /= 0) then
   169)           a2l%af_tbot(g)    = x2l(index_x2l_Sa_tbot_af,i)
   170)           a2l_not_downscaled_gcell%forc_t(g) =  a2l_not_downscaled_gcell%forc_t(g) + a2l%af_tbot(g)
   171)        endif
   172) 
   173)        if (index_x2l_Sa_lwdn_af /= 0) then
   174)           a2l%af_lwdn(g)    = x2l(index_x2l_Sa_lwdn_af,i)
   175)           a2l_not_downscaled_gcell%forc_lwrad(g) = a2l_not_downscaled_gcell%forc_lwrad(g) * a2l%af_lwdn(g)
   176)        endif
   177) 
   178)        if (index_x2l_Sa_prec_af /= 0) then
   179)           a2l%af_precip(g)    = x2l(index_x2l_Sa_prec_af,i)
   180)           forc_rainc = forc_rainc * a2l%af_precip(g)
   181)           forc_rainl = forc_rainl * a2l%af_precip(g)
   182)           forc_snowc = forc_snowc * a2l%af_precip(g)
   183)           forc_snowl = forc_snowl * a2l%af_precip(g)
   184)        endif
   185) 
   186)        if (index_x2l_Sa_swdn_af /= 0) then
   187)           a2l%af_swdn(g)    = x2l(index_x2l_Sa_swdn_af,i)
   188)           a2l%forc_solad(g,2) = a2l%forc_solad(g,2) * a2l%af_swdn(g)
   189)           a2l%forc_solad(g,1) = a2l%forc_solad(g,1) * a2l%af_swdn(g)
   190)           a2l%forc_solai(g,2) = a2l%forc_solai(g,2) * a2l%af_swdn(g)
   191)           a2l%forc_solai(g,1) = a2l%forc_solai(g,1) * a2l%af_swdn(g)
   192)        endif
   193)        !
   194)        ! anomaly forcing code ends
   195)        !
   196) 
   197)        ! atmosphere coupling, for prognostic/prescribed aerosols
   198)        a2l%forc_aer(g,1)  =  x2l(index_x2l_Faxa_bcphidry,i)
   199)        a2l%forc_aer(g,2)  =  x2l(index_x2l_Faxa_bcphodry,i)
   200)        a2l%forc_aer(g,3)  =  x2l(index_x2l_Faxa_bcphiwet,i)
   201)        a2l%forc_aer(g,4)  =  x2l(index_x2l_Faxa_ocphidry,i)
   202)        a2l%forc_aer(g,5)  =  x2l(index_x2l_Faxa_ocphodry,i)
   203)        a2l%forc_aer(g,6)  =  x2l(index_x2l_Faxa_ocphiwet,i)
   204)        a2l%forc_aer(g,7)  =  x2l(index_x2l_Faxa_dstwet1,i)
   205)        a2l%forc_aer(g,8)  =  x2l(index_x2l_Faxa_dstdry1,i)
   206)        a2l%forc_aer(g,9)  =  x2l(index_x2l_Faxa_dstwet2,i)
   207)        a2l%forc_aer(g,10) =  x2l(index_x2l_Faxa_dstdry2,i)
   208)        a2l%forc_aer(g,11) =  x2l(index_x2l_Faxa_dstwet3,i)
   209)        a2l%forc_aer(g,12) =  x2l(index_x2l_Faxa_dstdry3,i)
   210)        a2l%forc_aer(g,13) =  x2l(index_x2l_Faxa_dstwet4,i)
   211)        a2l%forc_aer(g,14) =  x2l(index_x2l_Faxa_dstdry4,i)
   212) 
   213)        ! Determine optional receive fields
   214) 
   215)        if (index_x2l_Sa_co2prog /= 0) then
   216)           co2_ppmv_prog = x2l(index_x2l_Sa_co2prog,i)   ! co2 atm state prognostic
   217)        else
   218)           co2_ppmv_prog = co2_ppmv
   219)        end if
   220) 
   221)        if (index_x2l_Sa_co2diag /= 0) then
   222)           co2_ppmv_diag = x2l(index_x2l_Sa_co2diag,i)   ! co2 atm state diagnostic
   223)        else
   224)           co2_ppmv_diag = co2_ppmv
   225)        end if
   226) 
   227)        if (index_x2l_Sa_methane /= 0) then
   228)           a2l%forc_pch4(g) = x2l(index_x2l_Sa_methane,i)
   229)        endif
   230) 
   231)        ! Determine derived quantities for required fields
   232) 
   233)        forc_t = a2l_not_downscaled_gcell%forc_t(g)
   234)        forc_q = a2l_not_downscaled_gcell%forc_q(g)
   235)        forc_pbot = a2l_not_downscaled_gcell%forc_pbot(g)
   236)        
   237)        a2l%forc_hgt_u(g) = a2l%forc_hgt(g)    !observational height of wind [m]
   238)        a2l%forc_hgt_t(g) = a2l%forc_hgt(g)    !observational height of temperature [m]
   239)        a2l%forc_hgt_q(g) = a2l%forc_hgt(g)    !observational height of humidity [m]
   240)        a2l%forc_vp(g)    = forc_q * forc_pbot &
   241)             / (0.622_r8 + 0.378_r8 * forc_q)
   242)        a2l_not_downscaled_gcell%forc_rho(g)   = (forc_pbot - 0.378_r8 * a2l%forc_vp(g)) &
   243)             / (rair * forc_t)
   244)        a2l%forc_po2(g)   = o2_molar_const * forc_pbot
   245)        a2l%forc_wind(g)  = sqrt(a2l%forc_u(g)**2 + a2l%forc_v(g)**2)
   246)        a2l%forc_solar(g) = a2l%forc_solad(g,1) + a2l%forc_solai(g,1) + &
   247)             a2l%forc_solad(g,2) + a2l%forc_solai(g,2)
   248) 
   249)        a2l_not_downscaled_gcell%forc_rain(g)  = forc_rainc + forc_rainl
   250)        a2l_not_downscaled_gcell%forc_snow(g)  = forc_snowc + forc_snowl
   251) 
   252)        if (forc_t > SHR_CONST_TKFRZ) then
   253)           e = esatw(tdc(forc_t))
   254)        else
   255)           e = esati(tdc(forc_t))
   256)        end if
   257)        qsat           = 0.622_r8*e / (forc_pbot - 0.378_r8*e)
   258)        a2l%forc_rh(g) = 100.0_r8*(forc_q / qsat)
   259)        ! Make sure relative humidity is properly bounded
   260)        ! a2l%forc_rh(g) = min( 100.0_r8, a2l%forc_rh(g) )
   261)        ! a2l%forc_rh(g) = max(   0.0_r8, a2l%forc_rh(g) )
   262) 
   263)        ! Determine derived quantities for optional fields
   264)        ! Note that the following does unit conversions from ppmv to partial pressures (Pa)
   265)        ! Note that forc_pbot is in Pa
   266) 
   267)        if (co2_type_idx == 1) then
   268)           co2_ppmv_val = co2_ppmv_prog
   269)        else if (co2_type_idx == 2) then
   270)           co2_ppmv_val = co2_ppmv_diag 
   271)        else
   272)           co2_ppmv_val = co2_ppmv
   273)        end if
   274)        a2l%forc_pco2(g)   = co2_ppmv_val * 1.e-6_r8 * forc_pbot 
   275)        if (use_c13) then
   276)           a2l%forc_pc13o2(g) = co2_ppmv_val * c13ratio * 1.e-6_r8 * forc_pbot
   277)        end if
   278) 
   279)        ! glc coupling 
   280) 
   281)        if (create_glacier_mec_landunit) then
   282)           do num = 0,glc_nec
   283)              x2s%frac(g,num)  = x2l(index_x2l_Sg_frac(num),i)
   284)              x2s%topo(g,num)  = x2l(index_x2l_Sg_topo(num),i)
   285)              x2s%hflx(g,num)  = x2l(index_x2l_Flgg_hflx(num),i)
   286)           end do
   287)           x2s%icemask(g)  = x2l(index_x2l_Sg_icemask,i)
   288)        end if
   289) 
   290)     end do
   291) 
   292)   end subroutine lnd_import
   293)

   296)   subroutine lnd_export( bounds, clm_l2a, clm_s2x, l2x)
   297) 
   298)     !---------------------------------------------------------------------------
   299)     ! !DESCRIPTION:
   300)     ! Convert the data to be sent from the clm model to the coupler 
   301)     ! 
   302)     ! !USES:
   303)     use shr_kind_mod       , only : r8 => shr_kind_r8
   304)     use clm_varctl         , only : iulog, create_glacier_mec_landunit
   305)     use clm_time_manager   , only : get_nstep, get_step_size  
   306)     use seq_drydep_mod     , only : n_drydep
   307)     use shr_megan_mod      , only : shr_megan_mechcomps_n
   308)     !
   309)     ! !ARGUMENTS:
   310)     implicit none
   311)     type(bounds_type) , intent(in)    :: bounds  ! bounds
   312)     type(lnd2atm_type), intent(inout) :: clm_l2a ! clm land to atmosphere exchange data type
   313)     type(lnd2glc_type), intent(inout) :: clm_s2x ! clm land to atmosphere exchange data type
   314)     real(r8)          , intent(out)   :: l2x(:,:)! land to coupler export state on land grid
   315)     !
   316)     ! !LOCAL VARIABLES:
   317)     integer  :: g,i   ! indices
   318)     integer  :: ier   ! error status
   319)     integer  :: nstep ! time step index
   320)     integer  :: dtime ! time step   
   321)     integer  :: num   ! counter
   322)     !---------------------------------------------------------------------------
   323) 
   324)     ! cesm sign convention is that fluxes are positive downward
   325) 
   326)     l2x(:,:) = 0.0_r8
   327) 
   328)     do g = bounds%begg,bounds%endg
   329)        i = 1 + (g-bounds%begg)
   330)        l2x(index_l2x_Sl_t,i)        =  clm_l2a%t_rad(g)
   331)        l2x(index_l2x_Sl_snowh,i)    =  clm_l2a%h2osno(g)
   332)        l2x(index_l2x_Sl_avsdr,i)    =  clm_l2a%albd(g,1)
   333)        l2x(index_l2x_Sl_anidr,i)    =  clm_l2a%albd(g,2)
   334)        l2x(index_l2x_Sl_avsdf,i)    =  clm_l2a%albi(g,1)
   335)        l2x(index_l2x_Sl_anidf,i)    =  clm_l2a%albi(g,2)
   336)        l2x(index_l2x_Sl_tref,i)     =  clm_l2a%t_ref2m(g)
   337)        l2x(index_l2x_Sl_qref,i)     =  clm_l2a%q_ref2m(g)
