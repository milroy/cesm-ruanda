   570) #ifdef DEBUG

  1258)   subroutine cloud_mod( zen_angle, clouds, lwc, delp, srf_alb, &

  1260)     !-----------------------------------------------------------------------
  1261)     ! 	... cloud alteration factors for photorates and albedo
  1262)     !-----------------------------------------------------------------------
  1263) 
  1264)     implicit none
  1265) 
  1266)     !-----------------------------------------------------------------------
  1267)     ! 	... dummy arguments
  1268)     !-----------------------------------------------------------------------

  1277)     !-----------------------------------------------------------------------
  1278)     ! 	... local variables
  1279)     !-----------------------------------------------------------------------
  1280)     integer :: k
  1281)     real(r8)    :: coschi
  1282)     real(r8)    :: del_lwp(pver)
  1283)     real(r8)    :: del_tau(pver)
  1284)     real(r8)    :: above_tau(pver)
  1285)     real(r8)    :: below_tau(pver)
  1286)     real(r8)    :: above_cld(pver)
  1287)     real(r8)    :: below_cld(pver)
  1288)     real(r8)    :: above_tra(pver)

  1291)     real(r8)    :: fac2(pver)

  1293)     real(r8), parameter :: rgrav = 1._r8/9.80616_r8
  1294) 
  1295)     !---------------------------------------------------------
  1296)     !	... modify lwc for cloud fraction and form
  1297)     !	    liquid water path for each layer
  1298)     !---------------------------------------------------------
  1299)     where( clouds(:) /= 0._r8 )
  1300)        del_lwp(:) = rgrav * lwc(:) * delp(:) * 1.e3_r8 / clouds(:)
  1301)     elsewhere
  1302)        del_lwp(:) = 0._r8
  1303)     endwhere

  1306)     !---------------------------------------------------------
  1307)     where( clouds(:) /= 0._r8 )
  1308)        del_tau(:) = del_lwp(:) *.155_r8 * clouds(:)**1.5_r8
  1309)     elsewhere
  1310)        del_tau(:) = 0._r8
  1311)     end where
  1312)     !---------------------------------------------------------
  1313)     !    	... form integrated tau from top down
  1314)     !---------------------------------------------------------
  1315)     above_tau(1) = 0._r8
  1316)     do k = 1,pverm
  1317)        above_tau(k+1) = del_tau(k) + above_tau(k)
  1318)     end do
  1319)     !---------------------------------------------------------
  1320)     !    	... form integrated tau from bottom up
  1321)     !---------------------------------------------------------
  1322)     below_tau(pver) = 0._r8
  1323)     do k = pverm,1,-1

  1327)     !	... form vertically averaged cloud cover above and below
  1328)     !---------------------------------------------------------
  1329)     above_cld(1) = 0._r8
  1330)     do k = 1,pverm
  1331)        above_cld(k+1) = clouds(k) * del_tau(k) + above_cld(k)
  1332)     end do
  1333)     do k = 2,pver
  1334)        if( above_tau(k) /= 0._r8 ) then
  1335)           above_cld(k) = above_cld(k) / above_tau(k)
  1336)        else
  1337)           above_cld(k) = above_cld(k-1)
  1338)        end if
  1339)     end do
  1340)     below_cld(pver) = 0._r8
  1341)     do k = pverm,1,-1
  1342)        below_cld(k) = clouds(k+1) * del_tau(k+1) + below_cld(k+1)
  1343)     end do
  1344)     do k = pverm,1,-1
  1345)        if( below_tau(k) /= 0._r8 ) then
  1346)           below_cld(k) = below_cld(k) / below_tau(k)
  1347)        else
  1348)           below_cld(k) = below_cld(k+1)
  1349)        end if
  1350)     end do
  1351)     !---------------------------------------------------------
  1352)     !	... modify above_tau and below_tau via jfm
  1353)     !---------------------------------------------------------
  1354)     where( above_cld(2:pver) /= 0._r8 )
  1355)        above_tau(2:pver) = above_tau(2:pver) / above_cld(2:pver)
  1356)     end where
  1357)     where( below_cld(:pverm) /= 0._r8 )
  1358)        below_tau(:pverm) = below_tau(:pverm) / below_cld(:pverm)
  1359)     end where
  1360)     where( above_tau(2:pver) < 5._r8 )
  1361)        above_cld(2:pver) = 0._r8
  1362)     end where
  1363)     where( below_tau(:pverm) < 5._r8 )
  1364)        below_cld(:pverm) = 0._r8
  1365)     end where
  1366)     !---------------------------------------------------------
  1367)     !	... form transmission factors
  1368)     !---------------------------------------------------------
  1369)     above_tra(:) = 11.905_r8 / (9.524_r8 + above_tau(:))
  1370)     below_tra(:) = 11.905_r8 / (9.524_r8 + below_tau(:))
  1371)     !---------------------------------------------------------
  1372)     !	... form effective albedo
  1373)     !---------------------------------------------------------
  1374)     where( below_cld(:) /= 0._r8 )
  1375)        eff_alb(:) = srf_alb + below_cld(:) * (1._r8 - below_tra(:)) &
  1376)                   * (1._r8 - srf_alb)
  1377)     elsewhere
  1378)        eff_alb(:) = srf_alb
  1379)     end where
  1380)     coschi = max( cos( zen_angle ),.5_r8 )
  1381)     where( del_lwp(:)*.155_r8 < 5._r8 )
  1382)        fac1(:) = 0._r8
  1383)     elsewhere
  1384)        fac1(:) = 1.4_r8 * coschi - 1._r8
  1385)     end where
  1386)     fac2(:)     = min( 0._r8,1.6_r8*coschi*above_tra(:) - 1._r8 )
  1387)     cld_mult(:) = 1._r8 + fac1(:) * clouds(:) + fac2(:) * above_cld(:)
  1388)     cld_mult(:) = max( .05_r8,cld_mult(:) )
  1389)
