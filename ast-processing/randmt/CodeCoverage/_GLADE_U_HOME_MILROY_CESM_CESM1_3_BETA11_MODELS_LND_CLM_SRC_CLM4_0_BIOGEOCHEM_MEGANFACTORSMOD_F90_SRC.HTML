   267)   ! Purpose: Generate a hash key on the interval [0 .. tbl_hash_sz-1]
   268)   !          given a character string.
   269)   !
   270)   ! Algorithm is a variant of perl's internal hashing function.
   271)   !
   272)   !-----------------------------------------------------------------------
   273)   integer function gen_hashkey(string)
   274) 
   275)     implicit none
   276)     !
   277)     !  Arguments:
   278)     !
   279)     character(len=*), intent(in) :: string
   280)     !
   281)     !  Local vars
   282)     !
   283)     integer :: hash
   284)     integer :: i
   285) 
   286)     integer, parameter :: tbl_max_idx = 15  ! 2**N - 1
   287)     integer, parameter :: gen_hash_key_offset = z'000053db'
   288)     integer, dimension(0:tbl_max_idx) :: tbl_gen_hash_key =  (/61,59,53,47,43,41,37,31,29,23,17,13,11,7,3,1/)
   289) 
   290)     hash = gen_hash_key_offset
   291) 
   292)     if ( len(string) /= 19 ) then
   293)        !
   294)        ! Process arbitrary string length.
   295)        !
   296)        do i = 1, len(string)
   297)           hash = ieor(hash , (ichar(string(i:i)) * tbl_gen_hash_key(iand(i-1,tbl_max_idx))))
   298)        end do
   299)     else
   300)        !
   301)        ! Special case string length = 19
   302)        !
   303)        do i = 1, tbl_max_idx+1
   304)           hash = ieor(hash , ichar(string(i:i))   * tbl_gen_hash_key(i-1)) 
   305)        end do
   306)        do i = tbl_max_idx+2, len(string)
   307)           hash = ieor(hash , ichar(string(i:i))   * tbl_gen_hash_key(i-tbl_max_idx-2)) 
   308)        end do
   309)     end if
   310) 
   311)     gen_hashkey = iand(hash, tbl_hash_sz-1)
   312) 
   313)     return
