    43) !            (6) included algae absorption for sea ice lowest layer

   203) ! !ROUTINE: init_shortwave
   204) !
   205) ! !DESCRIPTION:
   206) !
   207) !  Initialize shortwave
   208) ! 
   209) ! !REVISION HISTORY: same as module
   210) !
   211) ! !INTERFACE:
   212) !
   213)       subroutine init_shortwave
   214) !
   215) ! !USES:
   216) !
   217)       use ice_domain_size
   218)       use ice_blocks
   219)       use ice_calendar
   220)       use ice_domain
   221)       use ice_flux
   222)       use ice_grid
   223)       use ice_itd
   224)       use ice_orbital
   225)       use ice_state
   226) !
   227) ! !INPUT/OUTPUT PARAMETERS:
   228) !
   229) !EOP
   230) !
   231) !     local temporary variables
   232) 
   233)       integer (kind=int_kind) :: &
   234)          icells          ! number of cells with aicen > puny
   235) 
   236)       integer (kind=int_kind), dimension(nx_block*ny_block) :: &
   237)          indxi, indxj    ! indirect indices for cells with aicen > puny
   238) 
   239)       integer (kind=int_kind) :: &
   240)          i, j, ij    , & ! horizontal indices
   241)          iblk        , & ! block index
   242)          ilo,ihi,jlo,jhi, & ! beginning and end of physical domain
   243)          n           , & ! thickness category index
   244)          il1, il2    , & ! ice layer indices for eice
   245)          sl1, sl2        ! snow layer indices for esno
   246) 
   247)       real (kind=dbl_kind) :: cszn ! counter for history averaging
   248) 
   249)       type (block) :: &
   250)          this_block      ! block information for current block
   251) 
   252)       do iblk=1,nblocks
   253)       do j = 1, ny_block
   254)       do i = 1, nx_block
   255)          alvdr_gbm(i,j,iblk) = c0
   256)          alidr_gbm(i,j,iblk) = c0
   257)          alvdf_gbm(i,j,iblk) = c0
   258)          alidf_gbm(i,j,iblk) = c0
   259)       enddo
   260)       enddo
   261)       enddo
   262) 
   263)       if (trim(shortwave) == 'dEdd') then ! delta Eddington
   264) 
   265) #ifndef CCSMCOUPLED
   266)          ! These come from the driver in the coupled model.
   267)          call init_orbit       ! initialize orbital parameters
   268) #endif
   269)          call init_dEdd        ! initialize delta Eddington
   270)  
   271)       else                     ! basic (ccsm3) shortwave
   272) 
   273)          coszen(:,:,:) = p5 ! sun above the horizon
   274) 
   275)          do iblk=1,nblocks
   276)             this_block = get_block(blocks_ice(iblk),iblk)         
   277)             ilo = this_block%ilo
   278)             ihi = this_block%ihi
   279)             jlo = this_block%jlo
   280)             jhi = this_block%jhi
   281) 
   282)             do n = 1, ncat
   283) 
   284)                icells = 0
   285)                do j = jlo, jhi
   286)                do i = ilo, ihi
   287)                   if (aicen(i,j,n,iblk) > puny) then
   288)                      icells = icells + 1
   289)                      indxi(icells) = i
   290)                      indxj(icells) = j
   291)                   endif
   292)                enddo               ! i
   293)                enddo               ! j
   294) 
   295)                il1 = ilyr1(n)
   296)                il2 = ilyrn(n)
   297)                sl1 = slyr1(n)
   298)                sl2 = slyrn(n)
   299) 
   300)                Sswabsn(:,:,sl1:sl2,iblk) = c0
   301) 
   302)                call shortwave_ccsm3(nx_block,     ny_block,           &
   303)                               icells,                                 &
   304)                               indxi,             indxj,               &
   305)                               aicen(:,:,n,iblk), vicen(:,:,n,iblk),   &
   306)                               vsnon(:,:,n,iblk),                      &
   307)                               trcrn(:,:,nt_Tsfc,n,iblk),              &
   308)                               swvdr(:,:,  iblk), swvdf(:,:,  iblk),   &
   309)                               swidr(:,:,  iblk), swidf(:,:,  iblk),   &
   310)                               alvdrn(:,:,n,iblk),alidrn(:,:,n,iblk),  &
   311)                               alvdfn(:,:,n,iblk),alidfn(:,:,n,iblk),  &
   312)                               fswsfcn(:,:,n,iblk),fswintn(:,:,n,iblk),&
   313)                               fswthrun(:,:,n,iblk),                   &
   314)                               Iswabsn(:,:,il1:il2,iblk),              &
   315)                               albicen(:,:,n,iblk),albsnon(:,:,n,iblk))
   316) 
   317) 
   318)             enddo  ! ncat
   319)          enddo     ! nblocks
   320) 
   321)       endif
   322) 
   323)       !-----------------------------------------------------------------
   324)       ! Aggregate albedos 
   325)       !-----------------------------------------------------------------
   326) 
   327)       do iblk=1,nblocks
   328)          this_block = get_block(blocks_ice(iblk),iblk)         
   329)          ilo = this_block%ilo
   330)          ihi = this_block%ihi
   331)          jlo = this_block%jlo
   332)          jhi = this_block%jhi
   333) 
   334)          do n = 1, ncat
   335) 
   336)             icells = 0
   337)             do j = jlo, jhi
   338)             do i = ilo, ihi
   339)                if (aicen(i,j,n,iblk) > puny) then
   340)                   icells = icells + 1
   341)                   indxi(icells) = i
   342)                   indxj(icells) = j
   343)                endif
   344)             enddo               ! i
   345)             enddo               ! j
   346) 
   347)             do ij = 1, icells
   348)                i = indxi(ij)
   349)                j = indxj(ij)
   350) 
   351)                alvdf(i,j,iblk) = alvdf(i,j,iblk) &
   352)                   + alvdfn(i,j,n,iblk)*aicen(i,j,n,iblk)
   353)                alidf(i,j,iblk) = alidf(i,j,iblk) &
   354)                   + alidfn(i,j,n,iblk)*aicen(i,j,n,iblk)
   355)                alvdr(i,j,iblk) = alvdr(i,j,iblk) &
   356)                   + alvdrn(i,j,n,iblk)*aicen(i,j,n,iblk)
   357)                alidr(i,j,iblk) = alidr(i,j,iblk) &
   358)                   + alidrn(i,j,n,iblk)*aicen(i,j,n,iblk)
   359) 
   360)                if (coszen(i,j,iblk) > puny) then ! sun above horizon
   361)                   albice(i,j,iblk) = albice(i,j,iblk) &
   362)                      + albicen(i,j,n,iblk)*aicen(i,j,n,iblk)
   363)                   albsno(i,j,iblk) = albsno(i,j,iblk) &
   364)                      + albsnon(i,j,n,iblk)*aicen(i,j,n,iblk)
   365)                   albpnd(i,j,iblk) = albpnd(i,j,iblk) &
   366)                      + albpndn(i,j,n,iblk)*aicen(i,j,n,iblk)
   367)                endif
   368) 
   369) #ifdef AEROFRC
   370)                dalvdf_noaero(i,j,iblk) = dalvdf_noaero(i,j,iblk) &
   371)                   + dalvdfn_noaero(i,j,n,iblk)*aicen(i,j,n,iblk)
   372)                dalidf_noaero(i,j,iblk) = dalidf_noaero(i,j,iblk) &
   373)                   + dalidfn_noaero(i,j,n,iblk)*aicen(i,j,n,iblk)
   374)                dalvdr_noaero(i,j,iblk) = dalvdr_noaero(i,j,iblk) &
   375)                   + dalvdrn_noaero(i,j,n,iblk)*aicen(i,j,n,iblk)
   376)                dalidr_noaero(i,j,iblk) = dalidr_noaero(i,j,iblk) &
   377)                   + dalidrn_noaero(i,j,n,iblk)*aicen(i,j,n,iblk)
   378) 
   379)                if (coszen(i,j,iblk) > puny) then ! sun above horizon
   380)                   dalbice_noaero(i,j,iblk) = dalbice_noaero(i,j,iblk) &
   381)                      + dalbicen_noaero(i,j,n,iblk)*aicen(i,j,n,iblk)
   382)                   dalbsno_noaero(i,j,iblk) = dalbsno_noaero(i,j,iblk) &
   383)                      + dalbsnon_noaero(i,j,n,iblk)*aicen(i,j,n,iblk)
   384)                   dalbpnd_noaero(i,j,iblk) = dalbpnd_noaero(i,j,iblk) &
   385)                      + dalbpndn_noaero(i,j,n,iblk)*aicen(i,j,n,iblk)
   386)                endif
   387) #endif
   388) #ifdef CCSM3FRC
   389)                dalvdf_ccsm3(i,j,iblk) = dalvdf_ccsm3(i,j,iblk) &
   390)                   + dalvdfn_ccsm3(i,j,n,iblk)*aicen(i,j,n,iblk)
   391)                dalidf_ccsm3(i,j,iblk) = dalidf_ccsm3(i,j,iblk) &
   392)                   + dalidfn_ccsm3(i,j,n,iblk)*aicen(i,j,n,iblk)
   393)                dalvdr_ccsm3(i,j,iblk) = dalvdr_ccsm3(i,j,iblk) &
   394)                   + dalvdrn_ccsm3(i,j,n,iblk)*aicen(i,j,n,iblk)
   395)                dalidr_ccsm3(i,j,iblk) = dalidr_ccsm3(i,j,iblk) &
   396)                   + dalidrn_ccsm3(i,j,n,iblk)*aicen(i,j,n,iblk)
   397) 
   398)                if (coszen(i,j,iblk) > puny) then ! sun above horizon
   399)                   dalbice_ccsm3(i,j,iblk) = dalbice_ccsm3(i,j,iblk) &
   400)                      + dalbicen_ccsm3(i,j,n,iblk)*aicen(i,j,n,iblk)
   401)                   dalbsno_ccsm3(i,j,iblk) = dalbsno_ccsm3(i,j,iblk) &
   402)                      + dalbsnon_ccsm3(i,j,n,iblk)*aicen(i,j,n,iblk)
   403)                endif
   404) #endif
   405) #ifdef PONDFRC
   406)                dalvdf_nopond(i,j,iblk) = dalvdf_nopond(i,j,iblk) &
   407)                   + dalvdfn_nopond(i,j,n,iblk)*aicen(i,j,n,iblk)
   408)                dalidf_nopond(i,j,iblk) = dalidf_nopond(i,j,iblk) &
   409)                   + dalidfn_nopond(i,j,n,iblk)*aicen(i,j,n,iblk)
   410)                dalvdr_nopond(i,j,iblk) = dalvdr_nopond(i,j,iblk) &
   411)                   + dalvdrn_nopond(i,j,n,iblk)*aicen(i,j,n,iblk)
   412)                dalidr_nopond(i,j,iblk) = dalidr_nopond(i,j,iblk) &
   413)                   + dalidrn_nopond(i,j,n,iblk)*aicen(i,j,n,iblk)
   414) 
   415)                if (coszen(i,j,iblk) > puny) then ! sun above horizon
   416)                   dalbice_nopond(i,j,iblk) = dalbice_nopond(i,j,iblk) &
   417)                      + dalbicen_nopond(i,j,n,iblk)*aicen(i,j,n,iblk)
   418)                   dalbsno_nopond(i,j,iblk) = dalbsno_nopond(i,j,iblk) &
   419)                      + dalbsnon_nopond(i,j,n,iblk)*aicen(i,j,n,iblk)
   420)                   dalbpnd_nopond(i,j,iblk) = dalbpnd_nopond(i,j,iblk) &
   421)                      + dalbpndn_nopond(i,j,n,iblk)*aicen(i,j,n,iblk)
   422)                endif
   423) #endif
   424)             enddo
   425) 
   426)          enddo  ! ncat
   427) 
   428)       !----------------------------------------------------------------
   429)       ! Store grid box mean albedos and fluxes before scaling by aice
   430)       !----------------------------------------------------------------
   431) 
   432)          do j = 1, ny_block
   433)          do i = 1, nx_block
   434)             alvdf_gbm  (i,j,iblk) = alvdf  (i,j,iblk)
   435)             alidf_gbm  (i,j,iblk) = alidf  (i,j,iblk)
   436)             alvdr_gbm  (i,j,iblk) = alvdr  (i,j,iblk)
   437)             alidr_gbm  (i,j,iblk) = alidr  (i,j,iblk)
   438) 
   439)             ! for history averaging
   440)             cszn = c0
   441)             if (coszen(i,j,iblk) > puny) cszn = c1
   442)             do n = 1, nstreams
   443)                albcnt(i,j,iblk,n) = albcnt(i,j,iblk,n) + cszn
   444)             enddo
   445)          enddo
   446)          enddo
   447) 
   448)       enddo     ! nblocks
   449) 
   450)       end subroutine init_shortwave

   607) !
   608) ! Compute albedos for each thickness category
   609) !
   610) ! !REVISION HISTORY:
   611) !
   612) ! authors:  same as module
   613) !
   614) ! !USES:
   615) !
   616) ! !INPUT/OUTPUT PARAMETERS:
   617) !
   618)       integer (kind=int_kind), intent(in) :: &
   619)          nx_block, ny_block, & ! block dimensions
   620)          icells              ! number of ice-covered grid cells
   621) 
   622)       integer (kind=int_kind), dimension (nx_block*ny_block), &
   623)          intent(in) :: &
   624)          indxi   , & ! compressed indices for ice-covered cells
   625)          indxj
   626) 
   627)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   628)          intent(in) :: &
   629)          aicen   , & ! concentration of ice per category
   630)          vicen   , & ! volume of ice per category
   631)          vsnon   , & ! volume of ice per category
   632)          Tsfcn       ! surface temperature
   633) 
   634)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   635)          intent(out) :: &
   636)          alvdrni  , & ! visible, direct, ice   (fraction)
   637)          alidrni  , & ! near-ir, direct, ice   (fraction)
   638)          alvdfni  , & ! visible, diffuse, ice  (fraction)
   639)          alidfni  , & ! near-ir, diffuse, ice  (fraction)
   640)          alvdrns  , & ! visible, direct, snow  (fraction)
   641)          alidrns  , & ! near-ir, direct, snow  (fraction)
   642)          alvdfns  , & ! visible, diffuse, snow (fraction)
   643)          alidfns  , & ! near-ir, diffuse, snow (fraction)
   644)          alvdrn   , & ! visible, direct, avg   (fraction)
   645)          alidrn   , & ! near-ir, direct, avg   (fraction)
   646)          alvdfn   , & ! visible, diffuse, avg  (fraction)
   647)          alidfn   , & ! near-ir, diffuse, avg  (fraction)
   648)          albin    , & ! bare ice 
   649)          albsn        ! snow 
   650) !
   651) !EOP
   652) !
   653)       real (kind=dbl_kind), parameter :: &
   654)          ahmax     = p5          , & ! thickness above which ice albedo 
   655)                                      ! is constant (m)
   656)          dT_mlt    = c1          , & ! change in temp to give dalb_mlt 
   657)                                      ! albedo change
   658)          dalb_mlt  = -0.075_dbl_kind, & ! albedo change per dT_mlt change
   659)                                      ! in temp for ice
   660)          dalb_mltv = -p1         , & ! albedo vis change per dT_mlt change
   661)                                      ! in temp for snow
   662)          dalb_mlti = -p15            ! albedo nir change per dT_mlt change
   663)                                      ! in temp for snow
   664) 
   665)       integer (kind=int_kind) :: &
   666)          i, j, n
   667) 
   668)       real (kind=dbl_kind) :: &
   669)          hi  , & ! ice thickness  (m)
   670)          hs  , & ! snow thickness  (m)
   671)          albo, & ! effective ocean albedo, function of ice thickness
   672)          fh  , & ! piecewise linear function of thickness
   673)          fT  , & ! piecewise linear function of surface temperature
   674)          dTs , & ! difference of Tsfc and Timelt
   675)          fhtan,& ! factor used in albedo dependence on ice thickness
   676)          asnow   ! fractional area of snow cover
   677) 
   678)       integer (kind=int_kind) :: &
   679)          ij      ! horizontal index, combines i and j loops
   680) 
   681)       fhtan = atan(ahmax*c4)
   682) 
   683)       do j = 1, ny_block
   684)       do i = 1, nx_block
   685)          alvdrni(i,j) = albocn
   686)          alidrni(i,j) = albocn
   687)          alvdfni(i,j) = albocn
   688)          alidfni(i,j) = albocn
   689) 
   690)          alvdrns(i,j) = albocn
   691)          alidrns(i,j) = albocn
   692)          alvdfns(i,j) = albocn
   693)          alidfns(i,j) = albocn
   694) 
   695)          alvdrn(i,j) = albocn
   696)          alidrn(i,j) = albocn
   697)          alvdfn(i,j) = albocn
   698)          alidfn(i,j) = albocn
   699) 
   700)          albin(i,j) = c0
   701)          albsn(i,j) = c0
   702)       enddo         
   703)       enddo         
   704) 
   705)       !-----------------------------------------------------------------
   706)       ! Compute albedo for each thickness category.
   707)       !-----------------------------------------------------------------
   708) 
   709) !DIR$ CONCURRENT !Cray
   710) !cdir nodep      !NEC
   711) !ocl novrec      !Fujitsu
   712)       do ij = 1, icells
   713)          i = indxi(ij)
   714)          j = indxj(ij)
   715) 
   716)          hi = vicen(i,j) / aicen(i,j)
   717)          hs = vsnon(i,j) / aicen(i,j)            
   718) 
   719)          ! bare ice, thickness dependence
   720)          fh = min(atan(hi*c4)/fhtan,c1)
   721)          albo = albocn*(c1-fh)
   722)          alvdfni(i,j) = albicev*fh + albo
   723)          alidfni(i,j) = albicei*fh + albo
   724) 
   725)          ! bare ice, temperature dependence
   726)          dTs = Timelt - Tsfcn(i,j)
   727)          fT = min(dTs/dT_mlt-c1,c0)
   728)          alvdfni(i,j) = alvdfni(i,j) - dalb_mlt*fT
   729)          alidfni(i,j) = alidfni(i,j) - dalb_mlt*fT
   730) 
   731)          ! avoid negative albedos for thin, bare, melting ice
   732)          alvdfni(i,j) = max (alvdfni(i,j), albocn)
   733)          alidfni(i,j) = max (alidfni(i,j), albocn)
   734) 
   735)          if (hs > puny) then
   736) 
   737)             alvdfns(i,j) = albsnowv
   738)             alidfns(i,j) = albsnowi
   739) 
   740)             ! snow on ice, temperature dependence
   741)             alvdfns(i,j) = alvdfns(i,j) - dalb_mltv*fT
   742)             alidfns(i,j) = alidfns(i,j) - dalb_mlti*fT
   743) 
   744)          endif                  ! hs > puny
   745) 
   746)          ! direct albedos (same as diffuse for now)
   747)          alvdrni(i,j) = alvdfni(i,j)
   748)          alidrni(i,j) = alidfni(i,j)
   749)          alvdrns(i,j) = alvdfns(i,j)
   750)          alidrns(i,j) = alidfns(i,j)
   751) 
   752) 
   753)          ! fractional area of snow cover
   754)          if (hs > puny) then
   755)             asnow = hs / (hs + snowpatch)
   756)          else
   757)             asnow = c0
   758)          endif
   759) 
   760)          ! combine ice and snow albedos (for coupler)
   761)          alvdfn(i,j) = alvdfni(i,j)*(c1-asnow) + &
   762)                        alvdfns(i,j)*asnow
   763)          alidfn(i,j) = alidfni(i,j)*(c1-asnow) + &
   764)                        alidfns(i,j)*asnow
   765)          alvdrn(i,j) = alvdrni(i,j)*(c1-asnow) + &
   766)                        alvdrns(i,j)*asnow
   767)          alidrn(i,j) = alidrni(i,j)*(c1-asnow) + &
   768)                        alidrns(i,j)*asnow
   769) 
   770)          ! save ice and snow albedos (for history)
   771)          albin(i,j) = awtvdr*alvdrni(i,j) + awtidr*alidrni(i,j) &
   772)                     + awtvdf*alvdfni(i,j) + awtidf*alidfni(i,j) 
   773)          albsn(i,j) = awtvdr*alvdrns(i,j) + awtidr*alidrns(i,j) &
   774)                     + awtvdf*alvdfns(i,j) + awtidf*alidfns(i,j) 
   775) 
   776)       enddo                     ! ij
   777) 
   778)       end subroutine compute_albedos

  1119) !cdir nodep      !NEC

  1123)          j = indxj(ij)
  1124) 
  1125)          ! SW penetrating thru ice into ocean
  1126)          fswthru(i,j) = fswpen(i,j) * tranbot(i,j)
  1127) 
  1128)          ! SW absorbed in ice interior
  1129)          fswint(i,j)  = fswpen(i,j) - fswthru(i,j)
  1130) 
  1131)       enddo                     ! ij
  1132)

  1135)       ! interior, so add to surface absorption
  1136)       !----------------------------------------------------------------
  1137) 
  1138)       if (.not. heat_capacity) then
  1139) 
  1140) !DIR$ CONCURRENT !Cray
  1141) !cdir nodep      !NEC
  1142) !ocl novrec      !Fujitsu

  1144)            i = indxi(ij)
  1145)            j = indxj(ij)
  1146) 
  1147)            ! SW absorbed at snow/ice surface
  1148)            fswsfc(i,j) = fswsfc(i,j) + fswint(i,j)
  1149) 
  1150)            ! SW absorbed in ice interior (nilyr = 1)
  1151)            fswint(i,j)   = c0
  1152)            Iswabs(i,j,1) = c0
  1153) 
  1154)         enddo                     ! ij
  1155) 
  1156)       endif                       ! heat_capacity

  2025) !EOP
  2026) !-----------------------------------------------------------------------
  2027) !
  2028) ! Set up optical property profiles, based on snow, sea ice and ponded 
  2029) ! ice IOPs from:
  2030) !
  2031) ! Briegleb, B. P., and B. Light (2007): A Delta-Eddington Multiple 
  2032) !    Scattering Parameterization for Solar Radiation in the Sea Ice 
  2033) !    Component of the Community Climate System Model, NCAR Technical 
  2034) !    Note  NCAR/TN-472+STR  February 2007
  2035) !
  2036) ! Computes column Delta-Eddington radiation solution for specific
  2037) ! surface type: either snow over sea ice, bare sea ice, or ponded sea ice.
  2038) !
  2039) ! Divides solar spectrum into 3 intervals: 0.2-0.7, 0.7-1.19, and
  2040) ! 1.19-5.0 micro-meters. The latter two are added (using an assumed
  2041) ! partition of incident shortwave in the 0.7-5.0 micro-meter band between
  2042) ! the 0.7-1.19 and 1.19-5.0 micro-meter band) to give the final output 
  2043) ! of 0.2-0.7 visible and 0.7-5.0 near-infrared albedos and fluxes.
  2044) !
  2045) ! Specifies vertical layer optical properties based on input snow depth,
  2046) ! density and grain radius, along with ice and pond depths, then computes
  2047) ! layer by layer Delta-Eddington reflectivity, transmissivity and combines
  2048) ! layers (done by calling routine solution_dEdd). Finally, surface albedos
  2049) ! and internal fluxes/flux divergences are evaluated.
  2050) !
  2051) !  Description of the level and layer index conventions. This is
  2052) !  for the standard case of one snow layer and four sea ice layers.
  2053) !
  2054) !  Please read the following; otherwise, there is 99.9% chance you
  2055) !  will be confused about indices at some point in time........ :)
  2056) !
  2057) !  CICE4.0 snow treatment has one snow layer above the sea ice. This 
  2058) !  snow layer has finite heat capacity, so that surface absorption must
  2059) !  be distinguished from internal. The Delta-Eddington solar radiation
  2060) !  thus adds extra surface scattering layers to both snow and sea ice.
  2061) !  Note that in the following, we assume a fixed vertical layer structure
  2062) !  for the radiation calculation. In other words, we always have the 
  2063) !  structure shown below for one snow and four sea ice layers, but for 
  2064) !  ponded ice the pond fills "snow" layer 1 over the sea ice, and for 
  2065) !  bare sea ice the top layers over sea ice are treated as transparent air.
  2066) !
  2067) !  SSL = surface scattering layer for either snow or sea ice
  2068) !  DL  = drained layer for sea ice immediately under sea ice SSL
  2069) !  INT = interior layers for sea ice below the drained layer.
  2070) !
  2071) !  Notice that the radiation level starts with 0 at the top. Thus,
  2072) !  the total number radiation layers is klev+1, where klev is the
  2073) !  sum of nslyr, the number of CCSM snow layers, and nilyr, the
  2074) !  number of CCSM sea ice layers, plus the sea ice SSL:
  2075) !  klev = 1 + nslyr + nilyr
  2076) !
  2077) !  For the standard case illustrated below, nslyr=1, nilyr=4,
  2078) !  and klev=6, with the number of layer interfaces klevp=klev+1.
  2079) !  Layer interfaces are the surfaces on which reflectivities,
  2080) !  transmissivities and fluxes are evaluated.
  2081) !
  2082) !  CCSM3 Sea Ice Model            Delta-Eddington Solar Radiation
  2083) !                                     Layers and Interfaces
  2084) !                             Layer Index             Interface Index
  2085) !    ---------------------            ---------------------  0
  2086) !                                  0  \\\   snow SSL    \\\
  2087) !       snow layer 1                  ---------------------  1
  2088) !                                  1    rest of snow layer
  2089) !    +++++++++++++++++++++            +++++++++++++++++++++  2
  2090) !                                  2  \\\ sea ice SSL   \\\
  2091) !      sea ice layer 1                ---------------------  3
  2092) !                                  3      sea ice  DL
  2093) !    ---------------------            ---------------------  4
  2094) !
  2095) !      sea ice layer 2             4      sea ice INT
  2096) !
  2097) !    ---------------------            ---------------------  5
  2098) !
  2099) !      sea ice layer 3             5      sea ice INT
  2100) !
  2101) !    ---------------------            ---------------------  6
  2102) !
  2103) !      sea ice layer 4             6      sea ice INT
  2104) !
  2105) !    ---------------------            ---------------------  7
  2106) !
  2107) ! When snow lies over sea ice, the radiation absorbed in the
  2108) ! snow SSL is used for surface heating, and that in the rest
  2109) ! of the snow layer for its internal heating. For sea ice in
  2110) ! this case, all of the radiant heat absorbed in both the
  2111) ! sea ice SSL and the DL are used for sea ice layer 1 heating.
  2112) !
  2113) ! When pond lies over sea ice, and for bare sea ice, all of the
  2114) ! radiant heat absorbed within and above the sea ice SSL is used
  2115) ! for surface heating, and that absorbed in the sea ice DL is
  2116) ! used for sea ice layer 1 heating.
  2117) !
  2118) ! Basically, vertical profiles of the layer extinction optical depth (tau), 
  2119) ! single scattering albedo (w0) and asymmetry parameter (g) are required over
  2120) ! the klev+1 layers, where klev+1 = 2 + nslyr + nilyr. All of the surface type
  2121) ! information and snow/ice iop properties are evaulated in this routine, so
  2122) ! the tau,w0,g profiles can be passed to solution_dEdd for multiple scattering
  2123) ! evaluation. Snow, bare ice and ponded ice iops are contained in data arrays
  2124) ! in this routine.
  2125) !

  3297) ! Delta-Eddington solution for snow/air/pond over sea ice
  3298) !
  3299) ! Generic solution for a snow/air/pond input column of klev+1 layers,
  3300) ! with srftyp determining at what interface fresnel refraction occurs.
  3301) !
  3302) ! Computes layer reflectivities and transmissivities, from the top down
  3303) ! to the lowest interface using the Delta-Eddington solutions for each
  3304) ! layer; combines layers from top down to lowest interface, and from the
  3305) ! lowest interface (underlying ocean) up to the top of the column.
  3306) !
  3307) ! Note that layer diffuse reflectivity and transmissivity are computed
  3308) ! by integrating the direct over several gaussian angles. This is
  3309) ! because the diffuse reflectivity expression sometimes is negative,
  3310) ! but the direct reflectivity is always well-behaved. We assume isotropic
  3311) ! radiation in the upward and downward hemispheres for this integration.
  3312) !
  3313) ! Assumes monochromatic (spectrally uniform) properties across a band
  3314) ! for the input optical parameters.
  3315) !
  3316) ! If total transmission of the direct beam to the interface above a particular 
  3317) ! layer is less than trmin, then no further Delta-Eddington solutions are
  3318) ! evaluated for layers below.
  3319) !
  3320) ! The following describes how refraction is handled in the calculation.
  3321) !
  3322) ! First, we assume that radiation is refracted when entering either
  3323) ! sea ice at the base of the surface scattering layer, or water (i.e. melt
  3324) ! pond); we assume that radiation does not refract when entering snow, nor 
  3325) ! upon entering sea ice from a melt pond, nor upon entering the underlying 
  3326) ! ocean from sea ice.
  3327) !
  3328) ! To handle refraction, we define a "fresnel" layer, which physically
  3329) ! is of neglible thickness and is non-absorbing, which can be combined to 
  3330) ! any sea ice layer or top of melt pond. The fresnel layer accounts for 
  3331) ! refraction of direct beam and associated reflection and transmission for
  3332) ! solar radiation. A fresnel layer is combined with the top of a melt pond 
  3333) ! or to the surface scattering layer of sea ice if no melt pond lies over it. 
  3334) !
  3335) ! Some caution must be exercised for the fresnel layer, because any layer
  3336) ! to which it is combined is no longer a homogeneous layer, as are all other
  3337) ! individual layers. For all other layers for example, the direct and diffuse
  3338) ! reflectivities/transmissivities (R/T) are the same for radiation above or
  3339) ! below the layer. This is the meaning of homogeneous! But for the fresnel
  3340) ! layer this is not so. Thus, the R/T for this layer must be distinguished
  3341) ! for radiation above from that from radiation below. For generality, we
  3342) ! treat all layers to be combined as inhomogeneous.
  3343) !

  3475)       ! Delta-Eddington solution expressions

  3483)       omgs(w,f)        = (c1 - f)*w/(c1 - w*f)

  3539)               ! the top of the current layer k:
  3540)               !
  3541)               !              layers       interface
  3542)               !         
  3543)               !       ---------------------  k-1 
  3544)               !                k-1
  3545)               !       ---------------------  k

  3622)               trnlay(k,ij) = max(exp_min, exp(-ts/mu0n))
  3623)               alp = alpha(ws,mu0n,gs,lm)
  3624)               gam = gamma(ws,mu0n,gs,lm)
  3625)               apg = alp + gam
  3626)               amg = alp - gam

  3628)                            apg*rdif_a(k,ij)

  3630)                           (amg*rdif_a(k,ij) - (apg-c1))*trnlay(k,ij)

  3646)                 alp = alpha(ws,mu,gs,lm)
  3647)                 gam = gamma(ws,mu,gs,lm)
  3648)                 apg = alp + gam
  3649)                 amg = alp - gam

  3651)                       apg*rdif_a(k,ij)
  3652)                 tdr = apg*tdif_a(k,ij) + &
  3653)                      (amg*rdif_a(k,ij)-(apg-c1))*trn

  3655)                 smt = smt + mu*tdr*gwt

  3658)               tdif_a(k,ij) = smt/swt

  3661)               rdif_b(k,ij) = rdif_a(k,ij)

  3707)                                      Rf_dif_b*rintfc*tdif_a(kfrsnl,ij)
