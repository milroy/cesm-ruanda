    77) !
    78) ! !DESCRIPTION:

    93) ! !CALLED FROM:
    94) ! subroutine CNEcosystemDyn in module CNEcosystemDynMod.F90
    95) !
    96) ! !REVISION HISTORY:
    97) ! 7/28/03: Created by Peter Thornton
    98) ! 9/05/03, Peter Thornton: moved from call with (p) to call with (c)
    99) ! 10/3/03, Peter Thornton: added subroutine calls for different phenology types

   101) !    routines, and moved onset, offset, background litfall routines into
   102) !    main phenology call.

   104) ! local pointers to implicit in arrays
   105) !
   106) ! local pointers to implicit in/out scalars
   107) !
   108) ! local pointers to implicit out scalars
   109) !

   111) !EOP
   112) !-----------------------------------------------------------------------
   113) 
   114)    ! each of the following phenology type routines includes a filter
   115)    ! to operate only on the relevant pfts
   116) 
   117)    call CNPhenologyClimate(num_soilp, filter_soilp, num_pcropp, filter_pcropp)
   118)    
   119)    call CNEvergreenPhenology(num_soilp, filter_soilp)

   126)

   132)    call CNOffsetLitterfall(num_soilp, filter_soilp)
   133) 
   134)    call CNBackgroundLitterfall(num_soilp, filter_soilp)
   135) 
   136)    call CNLivewoodTurnover(num_soilp, filter_soilp)
   137) 
   138)    ! gather all pft-level litterfall fluxes to the column

   154) !
   155) ! !DESCRIPTION:
   156) ! Initialization of CNPhenology. Must be called after time-manager is

   170) ! !REVISION HISTORY:
   171) ! 3/28/11: Created by Erik Kluzek

   182)

   184)     ! parameter file after testing
   185)

   197)     ndays_off = 15._r8
   198) 
   199)     ! transfer parameters
   200)     fstor2tran = 0.5_r8

   202)     ! Constants for CNStressDecidPhenology
   203)     ! -----------------------------------------
   204) 
   205)     ! onset parameters

   207)     ! critical onset gdd now being calculated as a function of annual
   208)     ! average 2m temp.
   209)     ! crit_onset_gdd = 150.0 ! c3 grass value
   210)     ! crit_onset_gdd = 1000.0   ! c4 grass value

   216)     crit_offset_swi = 15.0_r8

   226) 
   227)     ! -----------------------------------------
   228)     ! Call any subroutine specific initialization routines
   229)     ! -----------------------------------------
   230) 
   231)     if ( crop_prog ) call CropPhenologyInit( begp, endp )

   241) ! !INTERFACE:
   242) subroutine CNPhenologyClimate (num_soilp, filter_soilp, num_pcropp, filter_pcropp)
   243) !
   244) ! !DESCRIPTION:
   245) ! For coupled carbon-nitrogen code (CN).
   246) !
   247) ! !USES:
   248)    use clm_time_manager, only: get_days_per_year
   249)    use clm_time_manager, only: get_curr_date, is_first_step
   250)    use CropRestMod     , only: CropRestYear
   251) !
   252) ! !ARGUMENTS:
   253)    integer, intent(in) :: num_soilp       ! number of soil pfts in filter
   254)    integer, intent(in) :: filter_soilp(:) ! filter for soil pfts
   255)    integer, intent(in) :: num_pcropp      ! number of prognostic crops in filter
   256)    integer, intent(in) :: filter_pcropp(:)! filter for prognostic crop pfts
   257) !
   258) ! !CALLED FROM:
   259) ! subroutine CNPhenology
   260) !
   261) ! !REVISION HISTORY:
   262) ! 3/13/07: Created by Peter Thornton
   263) !
   264) ! !LOCAL VARIABLES:
   265) ! local pointers to implicit in scalars
   266) !
   267)    integer , pointer :: ivt(:)             ! pft vegetation type
   268)    ! ecophysiological constants
   269)    real(r8), pointer :: t_ref2m(:)         ! 2m air temperature (K)
   270)    real(r8), pointer :: tempavg_t2m(:)     ! temp. avg 2m air temperature (K)
   271)    real(r8), pointer :: gdd0(:)            ! growing deg. days base 0 deg C (ddays)
   272)    real(r8), pointer :: gdd8(:)            !    "     "    "    "   8  "  "    "
   273)    real(r8), pointer :: gdd10(:)           !    "     "    "    "  10  "  "    "
   274)    real(r8), pointer :: gdd020(:)          ! 20-yr mean of gdd0 (ddays)
   275)    real(r8), pointer :: gdd820(:)          ! 20-yr mean of gdd8 (ddays)
   276)    real(r8), pointer :: gdd1020(:)         ! 20-yr mean of gdd10 (ddays)
   277)    integer , pointer :: pgridcell(:)       ! pft's gridcell index
   278) !
   279) ! local pointers to implicit in/out scalars
   280) !
   281) !
   282) ! local pointers to implicit out scalars
   283) !
   284) ! !OTHER LOCAL VARIABLES:
   285)    integer :: p                    ! indices
   286)    integer :: fp                   ! lake filter pft index
   287)    integer :: nyrs                 ! number of years prognostic crop has run
   288)    real(r8):: dayspyr              ! days per year (days)
   289)    integer kyr                     ! current year
   290)    integer kmo                     !         month of year  (1, ..., 12)
   291)    integer kda                     !         day of month   (1, ..., 31)
   292)    integer mcsec                   !         seconds of day (0, ..., seconds/day)
   293)    real(r8), parameter :: yravg   = 20.0_r8      ! length of years to average for gdd
   294)    real(r8), parameter :: yravgm1 = yravg-1.0_r8 ! minus 1 of above
   295) !EOP
   296) !-----------------------------------------------------------------------
   297) 
   298)    ! assign local pointers to derived type arrays
   299)    ivt                           => pft%itype
   300)    t_ref2m                       => pes%t_ref2m
   301)    tempavg_t2m                   => pepv%tempavg_t2m
   302) 
   303)    gdd0                          => pps%gdd0
   304)    gdd8                          => pps%gdd8
   305)    gdd10                         => pps%gdd10
   306)    gdd020                        => pps%gdd020
   307)    gdd820                        => pps%gdd820
   308)    gdd1020                       => pps%gdd1020
   309)    pgridcell                     => pft%gridcell
   310) 
   311)    ! set time steps
   312) 
   313)    dayspyr = get_days_per_year()
   314) 
   315)    do fp = 1,num_soilp
   316)       p = filter_soilp(fp)
   317)       tempavg_t2m(p) = tempavg_t2m(p) + t_ref2m(p) * (fracday/dayspyr)
   318)    end do
   319) 
   320)    !
   321)    ! The following crop related steps are done here rather than CropPhenology
   322)    ! so that they will be completed each time-step rather than with doalb.
   323)    !
   324)    ! The following lines come from ibis's climate.f + stats.f
   325)    ! gdd SUMMATIONS ARE RELATIVE TO THE PLANTING DATE (see subr. updateAccFlds)
   326) 
   327)    if (num_pcropp > 0) then
   328)       ! get time-related info
   329)       call get_curr_date(kyr, kmo, kda, mcsec)
   330)       nyrs = CropRestYear()
   331)    end if
   332) 
   333)    do fp = 1,num_pcropp
   334)       p = filter_pcropp(fp)
   335)       if (kmo == 1 .and. kda == 1 .and. nyrs == 0) then ! YR 1:
   336)          gdd020(p)  = 0._r8                             ! set gdd..20 variables to 0
   337)          gdd820(p)  = 0._r8                             ! and crops will not be planted
   338)          gdd1020(p) = 0._r8
   339)       end if
   340)       if (kmo == 1 .and. kda == 1 .and. mcsec == 0) then        ! <-- END of EVERY YR:
   341)          if (nyrs  == 1) then                                   ! <-- END of YR 1
   342)             gdd020(p)  = gdd0(p)                                ! <-- END of YR 1
   343)             gdd820(p)  = gdd8(p)                                ! <-- END of YR 1
   344)             gdd1020(p) = gdd10(p)                               ! <-- END of YR 1
   345)          end if                                                 ! <-- END of YR 1
   346)          gdd020(p)  = (yravgm1* gdd020(p)  + gdd0(p))  / yravg  ! gdd..20 must be long term avgs
   347)          gdd820(p)  = (yravgm1* gdd820(p)  + gdd8(p))  / yravg  ! so ignore results for yrs 1 & 2
   348)          gdd1020(p) = (yravgm1* gdd1020(p) + gdd10(p)) / yravg 
   349)       end if
   350)    end do
   351) 
   352) end subroutine CNPhenologyClimate

   361) subroutine CNEvergreenPhenology (num_soilp, filter_soilp)
   362) !
   363) ! !DESCRIPTION:
   364) ! For coupled carbon-nitrogen code (CN).
   365) !
   366) ! !USES:
   367)    use clm_varcon      , only: secspday
   368)    use clm_time_manager, only: get_days_per_year
   369) !
   370) ! !ARGUMENTS:
   371)    integer, intent(in) :: num_soilp       ! number of soil pfts in filter
   372)    integer, intent(in) :: filter_soilp(:) ! filter for soil pfts
   373) !
   374) ! !CALLED FROM:
   375) ! subroutine CNPhenology
   376) !
   377) ! !REVISION HISTORY:
   378) ! 10/2/03: Created by Peter Thornton
   379) !
   380) ! !LOCAL VARIABLES:
   381) ! local pointers to implicit in scalars
   382) !
   383)    integer , pointer :: ivt(:)       ! pft vegetation type
   384)    ! ecophysiological constants
   385)    real(r8), pointer :: evergreen(:) ! binary flag for evergreen leaf habit (0 or 1)
   386)    real(r8), pointer :: leaf_long(:) ! leaf longevity (yrs)
   387) !
   388) ! local pointers to implicit in/out scalars
   389) !
   390)    real(r8), pointer :: bglfr(:)     ! background litterfall rate (1/s)
   391)    real(r8), pointer :: bgtr(:)      ! background transfer growth rate (1/s)
   392)    real(r8), pointer :: lgsf(:)      ! long growing season factor [0-1]
   393) !
   394) ! local pointers to implicit out scalars
   395) !
   396) ! !OTHER LOCAL VARIABLES:
   397)    real(r8):: dayspyr                ! Days per year
   398)    integer :: p                      ! indices
   399)    integer :: fp                     ! lake filter pft index
   400) !EOP
   401) !-----------------------------------------------------------------------
   402) 
   403)    ! assign local pointers to derived type arrays
   404)    ivt       => pft%itype
   405)    evergreen => pftcon%evergreen
   406)    leaf_long => pftcon%leaf_long
   407)    bglfr     => pepv%bglfr
   408)    bgtr      => pepv%bgtr
   409)    lgsf      => pepv%lgsf
   410)    dayspyr   = get_days_per_year()
   411) 
   412)    do fp = 1,num_soilp
   413)       p = filter_soilp(fp)
   414)       if (evergreen(ivt(p)) == 1._r8) then
   415)           bglfr(p) = 1._r8/(leaf_long(ivt(p))*dayspyr*secspday)
   416)           bgtr(p)  = 0._r8
   417)           lgsf(p)  = 0._r8
   418)       end if
   419)    end do
   420) 
   421) end subroutine CNEvergreenPhenology

   533) !

   564)    deadstemn_storage             => pns%deadstemn_storage
   565)    livecrootn_storage            => pns%livecrootn_storage
   566)    deadcrootn_storage            => pns%deadcrootn_storage

   568)    woody                         => pftcon%woody
   569) 
   570)    ! Assign local pointers to derived type arrays (out)
   571)    dormant_flag                  => pepv%dormant_flag

   573)    onset_flag                    => pepv%onset_flag
   574)    onset_counter                 => pepv%onset_counter

   578)    offset_counter                => pepv%offset_counter

   582)    prev_leafc_to_litter          => pepv%prev_leafc_to_litter
   583)    prev_frootc_to_litter         => pepv%prev_frootc_to_litter
   584)    bglfr                         => pepv%bglfr
   585)    bgtr                          => pepv%bgtr
   586)    lgsf                          => pepv%lgsf
   587)    leafc_xfer_to_leafc           => pcf%leafc_xfer_to_leafc
   588)    frootc_xfer_to_frootc         => pcf%frootc_xfer_to_frootc

   594)    frootn_xfer_to_frootn         => pnf%frootn_xfer_to_frootn
   595)    livestemn_xfer_to_livestemn   => pnf%livestemn_xfer_to_livestemn
   596)    deadstemn_xfer_to_deadstemn   => pnf%deadstemn_xfer_to_deadstemn
   597)    livecrootn_xfer_to_livecrootn => pnf%livecrootn_xfer_to_livecrootn
   598)    deadcrootn_xfer_to_deadcrootn => pnf%deadcrootn_xfer_to_deadcrootn

   600)    frootc_xfer                   => pcs%frootc_xfer
   601)    livestemc_xfer                => pcs%livestemc_xfer
   602)    deadstemc_xfer                => pcs%deadstemc_xfer
   603)    livecrootc_xfer               => pcs%livecrootc_xfer

   608)    deadstemn_xfer                => pns%deadstemn_xfer
   609)    livecrootn_xfer               => pns%livecrootn_xfer

   612)    frootc_storage_to_xfer        => pcf%frootc_storage_to_xfer

   614)    deadstemc_storage_to_xfer     => pcf%deadstemc_storage_to_xfer

   621)    deadstemn_storage_to_xfer     => pnf%deadstemn_storage_to_xfer

   628)       p = filter_soilp(fp)
   629)       c = pcolumn(p)
   630) 
   631)       if (season_decid(ivt(p)) == 1._r8) then
   632) 
   633)          ! set background litterfall rate, background transfer rate, and
   634)          ! long growing season factor to 0 for seasonal deciduous types
   635)          bglfr(p) = 0._r8
   636)          bgtr(p) = 0._r8
   637)          lgsf(p) = 0._r8
   638) 
   639)          ! onset gdd sum from Biome-BGC, v4.1.2
   640)          crit_onset_gdd = exp(4.8_r8 + 0.13_r8*(annavg_t2m(p) - SHR_CONST_TKFRZ))
   641) 
   642)          ! use solar declination information stored during Surface Albedo()
   643)          ! and latitude from gps to calcluate daylength (convert latitude from degrees to radians)
   644)          ! the constant 13750.9871 is the number of seconds per radian of hour-angle
   645) 
   646)          prev_dayl(p) = dayl(p)
   647)          lat = (SHR_CONST_PI/180._r8)*latdeg(pgridcell(p))
   648)          temp = -(sin(lat)*sin(decl(c)))/(cos(lat) * cos(decl(c)))
   649)          temp = min(1._r8,max(-1._r8,temp))
   650)          dayl(p) = 2.0_r8 * 13750.9871_r8 * acos(temp)
   651) 
   652)          ! set flag for solstice period (winter->summer = 1, summer->winter = 0)
   653)          if (dayl(p) >= prev_dayl(p)) then
   654)             ws_flag = 1._r8
   655)          else
   656)             ws_flag = 0._r8
   657)          end if
   658) 
   659)          ! update offset_counter and test for the end of the offset period
   660)          if (offset_flag(p) == 1.0_r8) then
   661)             ! decrement counter for offset period
   662)             offset_counter(p) = offset_counter(p) - dt
   663) 
   664)             ! if this is the end of the offset_period, reset phenology
   665)             ! flags and indices
   666)             if (offset_counter(p) == 0.0_r8) then
   667)                ! this code block was originally handled by call cn_offset_cleanup(p)
   668)                ! inlined during vectorization
   669) 
   670)                offset_flag(p) = 0._r8
   671)                offset_counter(p) = 0._r8

   675)                   pftmayexist(p) = .true.
   676)                end if

   679)                prev_leafc_to_litter(p) = 0._r8
   680)                prev_frootc_to_litter(p) = 0._r8
   681)             end if
   682)          end if
   683) 
   684)          ! update onset_counter and test for the end of the onset period
   685)          if (onset_flag(p) == 1.0_r8) then
   686)             ! decrement counter for onset period
   687)             onset_counter(p) = onset_counter(p) - dt
   688) 
   689)             ! if this is the end of the onset period, reset phenology
   690)             ! flags and indices
   691)             if (onset_counter(p) == 0.0_r8) then
   692)                ! this code block was originally handled by call cn_onset_cleanup(p)
   693)                ! inlined during vectorization
   694) 
   695)                onset_flag(p) = 0.0_r8
   696)                onset_counter(p) = 0.0_r8

   698)                leafc_xfer_to_leafc(p)   = 0.0_r8
   699)                frootc_xfer_to_frootc(p) = 0.0_r8
   700)                leafn_xfer_to_leafn(p)   = 0.0_r8
   701)                frootn_xfer_to_frootn(p) = 0.0_r8
   702)                if (woody(ivt(p)) == 1.0_r8) then
   703)                   livestemc_xfer_to_livestemc(p)   = 0.0_r8
   704)                   deadstemc_xfer_to_deadstemc(p)   = 0.0_r8
   705)                   livecrootc_xfer_to_livecrootc(p) = 0.0_r8
   706)                   deadcrootc_xfer_to_deadcrootc(p) = 0.0_r8
   707)                   livestemn_xfer_to_livestemn(p)   = 0.0_r8
   708)                   deadstemn_xfer_to_deadstemn(p)   = 0.0_r8
   709)                   livecrootn_xfer_to_livecrootn(p) = 0.0_r8
   710)                   deadcrootn_xfer_to_deadcrootn(p) = 0.0_r8
   711)                end if
   712)                ! set transfer pools to 0.0
   713)                leafc_xfer(p) = 0.0_r8
   714)                leafn_xfer(p) = 0.0_r8
   715)                frootc_xfer(p) = 0.0_r8
   716)                frootn_xfer(p) = 0.0_r8
   717)                if (woody(ivt(p)) == 1.0_r8) then
   718)                   livestemc_xfer(p) = 0.0_r8
   719)                   livestemn_xfer(p) = 0.0_r8
   720)                   deadstemc_xfer(p) = 0.0_r8
   721)                   deadstemn_xfer(p) = 0.0_r8
   722)                   livecrootc_xfer(p) = 0.0_r8
   723)                   livecrootn_xfer(p) = 0.0_r8
   724)                   deadcrootc_xfer(p) = 0.0_r8
   725)                   deadcrootn_xfer(p) = 0.0_r8
   726)                end if
   727)             end if
   728)          end if
   729) 
   730)          ! test for switching from dormant period to growth period
   731)          if (dormant_flag(p) == 1.0_r8) then
   732) 
   733)             ! Test to turn on growing degree-day sum, if off.
   734)             ! switch on the growing degree day sum on the winter solstice
   735) 
   736)             if (onset_gddflag(p) == 0._r8 .and. ws_flag == 1._r8) then
   737)                onset_gddflag(p) = 1._r8
   738)                onset_gdd(p) = 0._r8
   739)             end if
   740) 
   741)             ! Test to turn off growing degree-day sum, if on.
   742)             ! This test resets the growing degree day sum if it gets past
   743)             ! the summer solstice without reaching the threshold value.
   744)             ! In that case, it will take until the next winter solstice
   745)             ! before the growing degree-day summation starts again.
   746) 
   747)             if (onset_gddflag(p) == 1._r8 .and. ws_flag == 0._r8) then
   748)                onset_gddflag(p) = 0._r8
   749)                onset_gdd(p) = 0._r8
   750)             end if
   751) 
   752)             ! if the gdd flag is set, and if the soil is above freezing
   753)             ! then accumulate growing degree days for onset trigger
   754) 
   755)             soilt = t_soisno(c,3)
   756)             if (onset_gddflag(p) == 1.0_r8 .and. soilt > SHR_CONST_TKFRZ) then
   757)                onset_gdd(p) = onset_gdd(p) + (soilt-SHR_CONST_TKFRZ)*fracday
   758)             end if
   759) 
   760)             ! set onset_flag if critical growing degree-day sum is exceeded
   761)             if (onset_gdd(p) > crit_onset_gdd) then
   762)                onset_flag(p) = 1.0_r8
   763)                dormant_flag(p) = 0.0_r8
   764)                onset_gddflag(p) = 0.0_r8
   765)                onset_gdd(p) = 0.0_r8
   766)                onset_counter(p) = ndays_on * secspday
   767) 
   768)                ! move all the storage pools into transfer pools,
   769)                ! where they will be transfered to displayed growth over the onset period.
   770)                ! this code was originally handled with call cn_storage_to_xfer(p)
   771)                ! inlined during vectorization
   772) 
   773)                ! set carbon fluxes for shifting storage pools to transfer pools
   774)                leafc_storage_to_xfer(p)  = fstor2tran * leafc_storage(p)/dt
   775)                frootc_storage_to_xfer(p) = fstor2tran * frootc_storage(p)/dt
   776)                if (woody(ivt(p)) == 1.0_r8) then
   777)                   livestemc_storage_to_xfer(p)  = fstor2tran * livestemc_storage(p)/dt
   778)                   deadstemc_storage_to_xfer(p)  = fstor2tran * deadstemc_storage(p)/dt
   779)                   livecrootc_storage_to_xfer(p) = fstor2tran * livecrootc_storage(p)/dt
   780)                   deadcrootc_storage_to_xfer(p) = fstor2tran * deadcrootc_storage(p)/dt
   781)                   gresp_storage_to_xfer(p)      = fstor2tran * gresp_storage(p)/dt
   782)                end if
   783) 
   784)                ! set nitrogen fluxes for shifting storage pools to transfer pools
   785)                leafn_storage_to_xfer(p)  = fstor2tran * leafn_storage(p)/dt
   786)                frootn_storage_to_xfer(p) = fstor2tran * frootn_storage(p)/dt
   787)                if (woody(ivt(p)) == 1.0_r8) then
   788)                   livestemn_storage_to_xfer(p)  = fstor2tran * livestemn_storage(p)/dt
   789)                   deadstemn_storage_to_xfer(p)  = fstor2tran * deadstemn_storage(p)/dt
   790)                   livecrootn_storage_to_xfer(p) = fstor2tran * livecrootn_storage(p)/dt
   791)                   deadcrootn_storage_to_xfer(p) = fstor2tran * deadcrootn_storage(p)/dt
   792)                end if
   793)             end if
   794) 
   795)          ! test for switching from growth period to offset period
   796)          else if (offset_flag(p) == 0.0_r8) then
   797)             if (use_cndv) then
   798)                ! If days_active > 355, then remove pft in
   799)                ! CNDVEstablishment at the end of the year.
   800)                ! days_active > 355 is a symptom of seasonal decid. pfts occurring in
   801)                ! gridcells where dayl never drops below crit_dayl.
   802)                ! This results in TLAI>1e4 in a few gridcells.
   803)                days_active(p) = days_active(p) + fracday
   804)                if (days_active(p) > 355._r8) pftmayexist(p) = .false.
   805)             end if
   806) 
   807)             ! only begin to test for offset daylength once past the summer sol
   808)             if (ws_flag == 0._r8 .and. dayl(p) < crit_dayl) then
   809)                offset_flag(p) = 1._r8
   810)                offset_counter(p) = ndays_off * secspday
   811)                prev_leafc_to_litter(p) = 0._r8
   812)                prev_frootc_to_litter(p) = 0._r8
   813)             end if
   814)          end if
   815) 
   816)       end if ! end if seasonal deciduous
   817) 
   818)    end do ! end of pft loop
   819) 
   820) end subroutine CNSeasonDecidPhenology

   829) subroutine CNStressDecidPhenology (num_soilp, filter_soilp)
   830) !
   831) ! !DESCRIPTION:
   832) ! This routine handles phenology for vegetation types, such as grasses and
   833) ! tropical drought deciduous trees, that respond to cold and drought stress
   834) ! signals and that can have multiple growing seasons in a given year.
   835) ! This routine allows for the possibility that leaves might persist year-round
   836) ! in the absence of a suitable stress trigger, by switching to an essentially
   837) ! evergreen habit, but maintaining a deciduous leaf longevity, while waiting
   838) ! for the next stress trigger.  This is in contrast to the seasonal deciduous
   839) ! algorithm (for temperate deciduous trees) that forces a single growing season
   840) ! per year.
   841) !
   842) ! !USES:
   843)    use clm_time_manager, only: get_days_per_year
   844)    use clm_varcon      , only: secspday
   845)    use shr_const_mod   , only: SHR_CONST_TKFRZ, SHR_CONST_PI
   846) !
   847) ! !ARGUMENTS:
   848)    integer, intent(in) :: num_soilp       ! number of soil pfts in filter
   849)    integer, intent(in) :: filter_soilp(:) ! filter for soil pfts
   850) !
   851) ! !CALLED FROM:
   852) ! subroutine CNPhenology
   853) !
   854) ! !REVISION HISTORY:
   855) ! 10/27/03: Created by Peter Thornton
   856) ! 01/29/04: Made onset_gdd critical sum a function of temperature, as in
   857) !           seasonal deciduous algorithm.
   858) !
   859) ! !LOCAL VARIABLES:
   860) ! local pointers to implicit in scalars
   861) !
   862)    integer , pointer :: ivt(:)                ! pft vegetation type
   863)    integer , pointer :: pcolumn(:)            ! pft's column index
   864)    integer , pointer :: pgridcell(:)          ! pft's gridcell index
   865)    real(r8), pointer :: latdeg(:)             ! latitude (radians)
   866)    real(r8), pointer :: decl(:)               ! solar declination (radians)
   867)    real(r8), pointer :: leafc_storage(:)      ! (gC/m2) leaf C storage
   868)    real(r8), pointer :: frootc_storage(:)     ! (gC/m2) fine root C storage
   869)    real(r8), pointer :: livestemc_storage(:)  ! (gC/m2) live stem C storage
   870)    real(r8), pointer :: deadstemc_storage(:)  ! (gC/m2) dead stem C storage
   871)    real(r8), pointer :: livecrootc_storage(:) ! (gC/m2) live coarse root C storage
   872)    real(r8), pointer :: deadcrootc_storage(:) ! (gC/m2) dead coarse root C storage
   873)    real(r8), pointer :: gresp_storage(:)      ! (gC/m2) growth respiration storage
   874)    real(r8), pointer :: leafn_storage(:)      ! (gN/m2) leaf N storage
   875)    real(r8), pointer :: frootn_storage(:)     ! (gN/m2) fine root N storage
   876)    real(r8), pointer :: livestemn_storage(:)  ! (gN/m2) live stem N storage
   877)    real(r8), pointer :: deadstemn_storage(:)  ! (gN/m2) dead stem N storage
   878)    real(r8), pointer :: livecrootn_storage(:) ! (gN/m2) live coarse root N storage
   879)    real(r8), pointer :: deadcrootn_storage(:) ! (gN/m2) dead coarse root N storage
   880)    real(r8), pointer :: t_soisno(:,:)         ! soil temperature (Kelvin)  (-nlevsno+1:nlevgrnd)

  1254)             if (soilt <= SHR_CONST_TKFRZ) then

  1259)             end if
  1260)             
  1261)             ! force offset if daylength is < 6 hrs
  1262)             if (dayl(p) <= secspqtrday) then
  1263)                offset_flag(p) = 1._r8
  1264)             end if
  1265) 
  1266)             ! if this is the beginning of the offset period
  1267)             ! then reset flags and indices
  1268)             if (offset_flag(p) == 1._r8) then
  1269)                offset_fdd(p) = 0._r8
  1270)                offset_swi(p) = 0._r8
  1271)                offset_counter(p) = ndays_off * secspday
  1272)                prev_leafc_to_litter(p) = 0._r8
  1273)                prev_frootc_to_litter(p) = 0._r8
  1274)             end if
  1275)          end if

  1277)          ! keep track of number of days since last dormancy for control on
  1278)          ! fraction of new growth to send to storage for next growing season
  1279) 
  1280)          if (dormant_flag(p) == 0.0_r8) then
  1281)              days_active(p) = days_active(p) + fracday
  1282)          end if
  1283) 
  1284)          ! calculate long growing season factor (lgsf)
  1285)          ! only begin to calculate a lgsf greater than 0.0 once the number
  1286)          ! of days active exceeds days/year.
  1287)          lgsf(p) = max(min((days_active(p)-dayspyr)/dayspyr, 1._r8),0._r8)
  1288) 
  1289)          ! set background litterfall rate, when not in the phenological offset period
  1290)          if (offset_flag(p) == 1._r8) then
  1291)             bglfr(p) = 0._r8
  1292)          else
  1293)             ! calculate the background litterfall rate (bglfr)
  1294)             ! in units 1/s, based on leaf longevity (yrs) and correction for long growing season
  1295) 
  1296)             bglfr(p) = (1._r8/(leaf_long(ivt(p))*dayspyr*secspday))*lgsf(p)
  1297)          end if
  1298) 
  1299)          ! set background transfer rate when active but not in the phenological onset period
  1300)          if (onset_flag(p) == 1._r8) then
  1301)             bgtr(p) = 0._r8
  1302)          else
  1303)             ! the background transfer rate is calculated as the rate that would result
  1304)             ! in complete turnover of the storage pools in one year at steady state,
  1305)             ! once lgsf has reached 1.0 (after 730 days active).
  1306) 
  1307)             bgtr(p) = (1._r8/(dayspyr*secspday))*lgsf(p)
  1308) 
  1309)             ! set carbon fluxes for shifting storage pools to transfer pools
  1310) 
  1311)             leafc_storage_to_xfer(p)  = leafc_storage(p) * bgtr(p)
  1312)             frootc_storage_to_xfer(p) = frootc_storage(p) * bgtr(p)
  1313)             if (woody(ivt(p)) == 1.0_r8) then
  1314)                livestemc_storage_to_xfer(p)  = livestemc_storage(p) * bgtr(p)
  1315)                deadstemc_storage_to_xfer(p)  = deadstemc_storage(p) * bgtr(p)
  1316)                livecrootc_storage_to_xfer(p) = livecrootc_storage(p) * bgtr(p)
  1317)                deadcrootc_storage_to_xfer(p) = deadcrootc_storage(p) * bgtr(p)
  1318)                gresp_storage_to_xfer(p)      = gresp_storage(p) * bgtr(p)
  1319)             end if
  1320) 
  1321)             ! set nitrogen fluxes for shifting storage pools to transfer pools
  1322)             leafn_storage_to_xfer(p)  = leafn_storage(p) * bgtr(p)
  1323)             frootn_storage_to_xfer(p) = frootn_storage(p) * bgtr(p)
  1324)             if (woody(ivt(p)) == 1.0_r8) then
  1325)                livestemn_storage_to_xfer(p)  = livestemn_storage(p) * bgtr(p)
  1326)                deadstemn_storage_to_xfer(p)  = deadstemn_storage(p) * bgtr(p)
  1327)                livecrootn_storage_to_xfer(p) = livecrootn_storage(p) * bgtr(p)
  1328)                deadcrootn_storage_to_xfer(p) = deadcrootn_storage(p) * bgtr(p)
  1329)             end if
  1330)          end if
  1331) 
  1332)       end if ! end if stress deciduous
  1333) 
  1334)    end do ! end of pft loop
  1335) 
  1336) end subroutine CNStressDecidPhenology

  1663)             ! from leaf emergence to beginning of grain-fill period
  1664)             ! this hypothetically occurs at the end of tassling, not the beginning
  1665)             ! tassel initiation typically begins at 0.5-0.55 * gddmaturity
  1666) 
  1667)             ! calculate linear relationship between huigrain fraction and relative
  1668)             ! maturity rating for maize
  1669) 
  1670)             if (ivt(p) == ncorn) then
  1671)                ! the following estimation of crmcorn from gddmaturity is based on a linear
  1672)                ! regression using data from Pioneer-brand corn hybrids (Kucharik, 2003,
  1673)                ! Earth Interactions 7:1-33: fig. 2)
  1674)                crmcorn = max(73._r8, min(135._r8, (gddmaturity(p)+ 53.683_r8)/13.882_r8))
  1675) 
  1676)                ! the following adjustment of grnfill based on crmcorn is based on a tuning
  1677)                ! of Agro-IBIS to give reasonable results for max LAI and the seasonal
  1678)                ! progression of LAI growth (pers. comm. C. Kucharik June 10, 2010)
  1679)                huigrain(p) = -0.002_r8  * (crmcorn - 73._r8) + grnfill(ivt(p))
  1680) 
  1681)                huigrain(p) = min(max(huigrain(p), grnfill(ivt(p))-0.1_r8), grnfill(ivt(p)))
  1682)                huigrain(p) = huigrain(p) * gddmaturity(p)     ! Cabelguenne et
  1683)             else
  1684)                huigrain(p) = grnfill(ivt(p)) * gddmaturity(p) ! al. 1999
  1685)             end if
  1686) 
  1687)          end if ! crop not live nor planted
  1688) 
  1689)          ! ----------------------------------
  1690)          ! from AgroIBIS subroutine phenocrop
  1691)          ! ----------------------------------
  1692) 
  1693)          ! all of the phenology changes are based on the total number of gdd needed
  1694)          ! to change to the next phase - based on fractions of the total gdd typical
  1695)          ! for  that region based on the April 1 - Sept 30 window of development
  1696) 
  1697)          ! crop phenology (gdd thresholds) controlled by gdd needed for
  1698)          ! maturity (physiological) which is based on the average gdd
  1699)          ! accumulation and hybrids in United States from April 1 - Sept 30
  1700)          
  1701)          ! Phase 1: Planting to leaf emergence (now in CNAllocation)
  1702)          ! Phase 2: Leaf emergence to beginning of grain fill (general LAI accumulation)
  1703)          ! Phase 3: Grain fill to physiological maturity and harvest (LAI decline)
  1704)          ! Harvest: if gdd past grain fill initiation exceeds limit
  1705)          ! or number of days past planting reaches a maximum, the crop has
  1706)          ! reached physiological maturity and plant is harvested;
  1707)          ! crop could be live or dead at this stage - these limits
  1708)          ! could lead to reaching physiological maturity or determining
  1709)          ! a harvest date for a crop killed by an early frost (see next comments)
  1710)          ! --- --- ---
  1711)          ! keeping comments without the code (slevis):
  1712)          ! if minimum temperature, t_ref2m_min <= freeze kill threshold, tkill
  1713)          ! for 3 consecutive days and lai is above a minimum,
  1714)          ! plant will be damaged/killed. This function is more for spring freeze events
  1715)          ! or for early fall freeze events
  1716) 
  1717)          ! spring temperate cereal is affected by this, winter cereal kill function
  1718)          ! is determined in crops.f - is a more elaborate function of
  1719)          ! cold hardening of the plant
  1720) 
  1721)          ! currently simulates too many grid cells killed by freezing temperatures
  1722) 
  1723)          ! removed on March 12 2002 - C. Kucharik
  1724)          ! until it can be a bit more refined, or used at a smaller scale.
  1725)          ! we really have no way of validating this routine
  1726)          ! too difficult to implement on 0.5 degree scale grid cells
  1727)          ! --- --- ---
  1728) 
  1729)          onset_flag(p)  = 0._r8 ! CN terminology to trigger certain
  1730)          offset_flag(p) = 0._r8 ! carbon and nitrogen transfers
  1731) 
  1732)          if (croplive(p)) then
  1733) 
  1734)             ! call vernalization if winter temperate cereal planted, living, and the
  1735)             ! vernalization factor is not 1;
  1736)             ! vf affects the calculation of gddtsoi & gddplant
  1737) 
  1738)             if (t_ref2m_min(p) < 1.e30_r8 .and. vf(p) /= 1._r8 .and. ivt(p) == nwcereal) then
  1739)                call vernalization(p)
  1740)             end if
  1741) 
  1742)             ! days past planting may determine harvest
  1743) 
  1744)             if (jday >= idop(p)) then
  1745)                idpp = jday - idop(p)
  1746)             else
  1747)                idpp = int(dayspyr) + jday - idop(p)
  1748)             end if
  1749) 
  1750)             ! onset_counter initialized to zero when .not. croplive
  1751)             ! offset_counter relevant only at time step of harvest
  1752) 
  1753)             onset_counter(p) = onset_counter(p) - dt
  1754) 
  1755)             ! enter phase 2 onset for one time step:
  1756)             ! transfer seed carbon to leaf emergence
  1757) 
  1758)             if (leafout(p) >= huileaf(p) .and. hui(p) < huigrain(p) .and. idpp < mxmat(ivt(p))) then
  1759)                if (abs(onset_counter(p)) > 1.e-6_r8) then
  1760)                   onset_flag(p)    = 1._r8
  1761)                   onset_counter(p) = dt
  1762)                else
  1763)                   onset_counter(p) = dt ! ensure no re-entry to onset of phase2
  1764)                end if
  1765) 
  1766)             ! enter harvest for one time step:
  1767)             ! - transfer live biomass to litter and to crop yield
  1768)             ! - send xsmrpool to the atmosphere
  1769)             ! if onset and harvest needed to last longer than one timestep
  1770)             ! the onset_counter would change from dt and you'd need to make
  1771)             ! changes to the offset subroutine below
  1772) 
  1773)             else if (hui(p) >= gddmaturity(p) .or. idpp >= mxmat(ivt(p))) then
  1774)                if (harvdate(p) >= NOT_Harvested) harvdate(p) = jday
  1775)                croplive(p) = .false.     ! no re-entry in greater if-block
  1776)                if (tlai(p) > 0._r8) then ! plant had emerged before harvest
  1777)                   offset_flag(p) = 1._r8
  1778)                   offset_counter(p) = dt
  1779)                else                      ! plant never emerged from the ground
  1780)                   dwt_seedc_to_leaf(c) = dwt_seedc_to_leaf(c) - leafc_xfer(p)/dt
  1781)                   dwt_seedn_to_leaf(c) = dwt_seedn_to_leaf(c) - leafn_xfer(p)/dt
  1782)                   leafc_xfer(p) = 0._r8  ! revert planting transfers
  1783)                   leafn_xfer(p) = leafc_xfer(p) / leafcn(ivt(p))
  1784)                end if
  1785) 
  1786)             ! enter phase 3 while previous criteria fail and next is true;
  1787)             ! in terms of order, phase 3 occurs before harvest, but when
  1788)             ! harvest *can* occur, we want it to have first priority.
  1789)             ! AgroIBIS uses a complex formula for lai decline.
  1790)             ! Use CN's simple formula at least as a place holder (slevis)
  1791) 
  1792)             else if (hui(p) >= huigrain(p)) then
  1793)                bglfr(p) = 1._r8/(leaf_long(ivt(p))*dayspyr*secspday)
  1794)             end if
  1795) 
  1796)          else   ! crop not live
  1797)             onset_counter(p) = 0._r8
  1798)             leafc_xfer(p) = 0._r8
  1799)             leafn_xfer(p) = leafc_xfer(p) / leafcn(ivt(p))
  1800)          end if ! croplive
  1801) 
  1802)       end do ! prognostic crops loop
  1803) 
  1804) end subroutine CropPhenology

  1893) ! !INTERFACE:
  1894)   subroutine vernalization(p)
  1895) !
  1896) ! !DESCRIPTION:
  1897) !
  1898) ! * * * only call for winter temperate cereal * * *
  1899) !
  1900) ! subroutine calculates vernalization and photoperiod effects on
  1901) ! gdd accumulation in winter temperate cereal varieties. Thermal time accumulation
  1902) ! is reduced in 1st period until plant is fully vernalized. During this
  1903) ! time of emergence to spikelet formation, photoperiod can also have a
  1904) ! drastic effect on plant development.
  1905) !
  1906) ! !ARGUMENTS:
  1907)       implicit none
  1908)       integer, intent(in) :: p    ! PFT index running over
  1909) !
  1910) ! !REVISION HISTORY:
  1911) ! Created by Sam Levis from AGROIBIS
  1912) !
  1913) !EOP
  1914) 
  1915) ! LOCAL VARAIBLES:
  1916)       real(r8) tcrown                     ! ?
  1917)       real(r8) vd, vd1, vd2               ! vernalization dependence
  1918)       real(r8) tkil                       ! Freeze kill threshold
  1919)       integer  c,g                        ! indices
  1920) ! local pointers to implicit in scalars
  1921)       integer , pointer :: pcolumn(:)     ! pft's column index
  1922)       logical , pointer :: croplive(:)    ! Flag, true if planted, not harvested
  1923)       real(r8), pointer :: tlai(:)        ! one-sided leaf area index, no burying by snow
  1924)       real(r8), pointer :: t_ref2m(:)     ! 2 m height surface air temperature (K)
  1925)       real(r8), pointer :: t_ref2m_min(:) !daily minimum of average 2 m height surface air temperature (K)
  1926)       real(r8), pointer :: t_ref2m_max(:) !daily maximum of average 2 m height surface air temperature (K)
  1927)       real(r8), pointer :: snowdp(:)      ! snow height (m)
  1928) ! local pointers to implicit out scalars
  1929)       real(r8), pointer :: vf(:)          ! vernalization factor for cereal
  1930)       real(r8), pointer :: cumvd(:)       ! cumulative vernalization d?ependence?
  1931)       real(r8), pointer :: gddmaturity(:) ! gdd needed to harvest
  1932)       real(r8), pointer :: huigrain(:)    ! heat unit index needed to reach vegetative maturity
  1933)       real(r8), pointer :: hdidx(:)       ! cold hardening index?
  1934) !------------------------------------------------------------------------
  1935) 
  1936)         pcolumn     => pft%column
  1937)         croplive    => pps%croplive
  1938)         hdidx       => pps%hdidx
  1939)         cumvd       => pps%cumvd
  1940)         vf          => pps%vf
  1941)         gddmaturity => pps%gddmaturity
  1942)         huigrain    => pps%huigrain
  1943)         tlai        => pps%tlai
  1944)         t_ref2m     => pes%t_ref2m
  1945)         t_ref2m_min => pes%t_ref2m_min
  1946)         t_ref2m_max => pes%t_ref2m_max
  1947)         snowdp      => cps%snowdp
  1948) 
  1949)         c = pcolumn(p)
  1950) 
  1951)         ! for all equations - temperatures must be in degrees (C)
  1952)         ! calculate temperature of crown of crop (e.g., 3 cm soil temperature)
  1953)         ! snow depth in centimeters
  1954) 
  1955)         if (t_ref2m(p) < tfrz) then !slevis: t_ref2m inst of td=daily avg (K)
  1956)            tcrown = 2._r8 + (t_ref2m(p) - tfrz) * (0.4_r8 + 0.0018_r8 * &
  1957)                     (min(snowdp(c)*100._r8, 15._r8) - 15._r8)**2)
  1958)         else !slevis: snowdp inst of adsnod=daily average (m)
  1959)            tcrown = t_ref2m(p) - tfrz
  1960)         end if
  1961) 
  1962)         ! vernalization factor calculation
  1963)         ! if vf(p) = 1.  then plant is fully vernalized - and thermal time
  1964)         ! accumulation in phase 1 will be unaffected
  1965)         ! refers to gddtsoi & gddplant, defined in the accumulation routines (slevis)
  1966)         ! reset vf, cumvd, and hdidx to 0 at planting of crop (slevis)
  1967) 
  1968)         if (t_ref2m_max(p) > tfrz) then
  1969)            if (t_ref2m_min(p) <= tfrz+15._r8) then
  1970)              vd1      = 1.4_r8 - 0.0778_r8 * tcrown
  1971)              vd2      = 0.5_r8 + 13.44_r8 / ((t_ref2m_max(p)-t_ref2m_min(p)+3._r8)**2) * tcrown
  1972)              vd       = max(0._r8, min(1._r8, vd1, vd2))
  1973)              cumvd(p) = cumvd(p) + vd
  1974)            end if
  1975) 
  1976)            if (cumvd(p) < 10._r8 .and. t_ref2m_max(p) > tfrz+30._r8) then
  1977)              cumvd(p) = cumvd(p) - 0.5_r8 * (t_ref2m_max(p) - tfrz - 30._r8)
  1978)            end if
  1979)            cumvd(p) = max(0._r8, cumvd(p))       ! must be > 0
  1980) 
  1981)            vf(p) = 1._r8 - p1v * (50._r8 - cumvd(p))
  1982)            vf(p) = max(0._r8, min(vf(p), 1._r8)) ! must be between 0 - 1
  1983)         end if
  1984) 
  1985)         ! calculate cold hardening of plant
  1986)         ! determines for winter cereal varieties whether the plant has completed
  1987)         ! a period of cold hardening to protect it from freezing temperatures. If
  1988)         ! not, then exposure could result in death or killing of plants.
  1989) 
  1990)         ! there are two distinct phases of hardening
  1991) 
  1992)         if (t_ref2m_min(p) <= tfrz-3._r8 .or. hdidx(p) /= 0._r8) then
  1993)            if (hdidx(p) >= hti) then   ! done with phase 1
  1994)               hdidx(p) = hdidx(p) + 0.083_r8
  1995)               hdidx(p) = min(hdidx(p), hti*2._r8)
  1996)            end if
  1997) 
  1998)            if (t_ref2m_max(p) >= tbase + tfrz + 10._r8) then
  1999)               hdidx(p) = hdidx(p) - 0.02_r8 * (t_ref2m_max(p)-tbase-tfrz-10._r8)
  2000)               if (hdidx(p) > hti) hdidx(p) = hdidx(p) - 0.02_r8 * (t_ref2m_max(p)-tbase-tfrz-10._r8)
  2001)               hdidx(p) = max(0._r8, hdidx(p))
  2002)            end if
  2003) 
  2004)         else if (tcrown >= tbase-1._r8) then
  2005)            if (tcrown <= tbase+8._r8) then
  2006)               hdidx(p) = hdidx(p) + 0.1_r8 - (tcrown-tbase+3.5_r8)**2 / 506._r8
  2007)               if (hdidx(p) >= hti .and. tcrown <= tbase + 0._r8) then
  2008)                  hdidx(p) = hdidx(p) + 0.083_r8
  2009)                  hdidx(p) = min(hdidx(p), hti*2._r8)
  2010)               end if
  2011)            end if
  2012) 
  2013)            if (t_ref2m_max(p) >= tbase + tfrz + 10._r8) then
  2014)               hdidx(p) = hdidx(p) - 0.02_r8 * (t_ref2m_max(p)-tbase-tfrz-10._r8)
  2015)               if (hdidx(p) > hti) hdidx(p) = hdidx(p) - 0.02_r8 * (t_ref2m_max(p)-tbase-tfrz-10._r8)
  2016)               hdidx(p) = max(0._r8, hdidx(p))
  2017)            end if
  2018)         end if
  2019) 
  2020)         ! calculate what the cereal killing temperature
  2021)         ! there is a linear inverse relationship between
  2022)         ! hardening of the plant and the killing temperature or
  2023)         ! threshold that the plant can withstand
  2024)         ! when plant is fully-hardened (hdidx = 2), the killing threshold is -18 C
  2025) 
  2026)         ! will have to develop some type of relationship that reduces LAI and
  2027)         ! biomass pools in response to cold damaged crop
  2028) 
  2029)         if (t_ref2m_min(p) <= tfrz - 6._r8) then
  2030)            tkil = (tbase - 6._r8) - 6._r8 * hdidx(p)
  2031)            if (tkil >= tcrown) then
  2032)               if ((0.95_r8 - 0.02_r8 * (tcrown - tkil)**2) >= 0.02_r8) then
  2033)                  write (iulog,*)  'crop damaged by cold temperatures at p,c =', p,c
  2034)               else if (tlai(p) > 0._r8) then ! slevis: kill if past phase1
  2035)                  gddmaturity(p) = 0._r8      !         by forcing through
  2036)                  huigrain(p)    = 0._r8      !         harvest
  2037)                  write (iulog,*)  '95% of crop killed by cold temperatures at p,c =', p,c
  2038)               end if
  2039)            end if
  2040)         end if
  2041) 
  2042)   end subroutine vernalization
