     7) ! !MODULE: shr_pcdf_mod -- generic pio file reader and writer
     8) !
     9) ! !DESCRIPTION:
    10) !
    11) !    Reads & writes pio files
    12) !
    13) ! !REMARKS:
    14) !
    15) !    supports aVect, 1d real and integer, and scalar real and integer fields
    16) !    using a common decomp for all fields.  this is a heavily overloaded interface
    17) !    that supports read and write of multiple fields/type to a file using a single call.
    18) !
    19) ! !REVISION HISTORY:
    20) !     2009-Oct-15 - T. Craig - initial implementation
    21) !
    22) ! !INTERFACE: ------------------------------------------------------------------
    23) 
    24) module shr_pcdf_mod
    25) 
    26)   use shr_kind_mod,      only: R8 => SHR_KIND_R8, IN => SHR_KIND_IN
    27)   use shr_kind_mod,      only: CL => SHR_KIND_CL, CS => SHR_KIND_CS
    28)   use shr_sys_mod,       only: shr_sys_abort, shr_sys_flush
    29)   use shr_const_mod,     only: shr_const_spval
    30)   use shr_log_mod,       only: shr_log_unit, shr_log_level
    31)   use mct_mod
    32)   use pio
    33)  
    34)   implicit none
    35) 
    36)   private
    37) 
    38)  !PUBLIC TYPES:
    39) 
    40)   ! no public types
    41) 
    42) !!PUBLIC MEMBER FUNCTIONS
    43) 
    44)   public :: shr_pcdf_readwrite
    45) 
    46) !!PUBLIC DATA MEMBERS:
    47) 
    48)   ! no public data
    49) 
    50) !EOP
    51) 
    52)   character(len=*),parameter :: version   = 'shr_pcdf_v0_0_01'
    53)   real(r8)        ,parameter :: fillvalue = SHR_CONST_SPVAL
    54)   integer(in)     ,parameter :: ifillvalue = -999999
    55)   
    56) !===============================================================================
    57) contains
    58) !===============================================================================
    59) subroutine shr_pcdf_readwrite(type,iosystem,pio_iotype,filename,mpicom,gsmap,dof,clobber,cdf64, &
    60)                      id1,id1n,rs1,rs1n,is1,is1n,rf1,rf1n,if1,if1n,av1,av1n, &
    61)                      id2,id2n,rs2,rs2n,is2,is2n,rf2,rf2n,if2,if2n,av2,av2n, &
    62)                      id3,id3n,rs3,rs3n,is3,is3n,rf3,rf3n,if3,if3n,av3,av3n, &
    63)                      id4,id4n,rs4,rs4n,is4,is4n,rf4,rf4n,if4,if4n,av4,av4n  )
    64)   use pio, only : iosystem_desc_t
    65)   implicit none
    66) 
    67)   character(len=*) , intent(in)    :: type      ! 'read' or 'write'
    68)   type(iosystem_desc_t) :: iosystem
    69)   integer(IN), intent(in) :: pio_iotype
    70)   character(len=*) , intent(in)    :: filename  ! filename
    71)   integer(IN)      , intent(in)    :: mpicom    ! mpicom
    72) 
    73)   !--- one of these must be set ---
    74)   type(mct_gsmap)  , optional, intent(in)    :: gsmap    ! decomp for all data
    75)   integer(IN)      , optional, intent(in)    :: dof(:)   ! decomp for all data
    76) 
    77)   !--- optional settings ---
    78)   logical          , optional, intent(in)    :: clobber
    79)   logical          , optional, intent(in)    :: cdf64
    80)   ! add root, stride, ntasks, netcdf/pnetcdf, etc
    81) 
    82)   !--- data to write ---
    83) 
    84)   !--- single scalar dimensions, assumed valid on the io root pe ---
    85)   integer(IN)      , optional, intent(inout) :: id1      ! int field 1
    86)   character(len=*) , optional, intent(in)    :: id1n     ! if1 name
    87)   integer(IN)      , optional, intent(inout) :: id2      ! int field 2
    88)   character(len=*) , optional, intent(in)    :: id2n     ! if2 name
    89)   integer(IN)      , optional, intent(inout) :: id3      ! int field 3
    90)   character(len=*) , optional, intent(in)    :: id3n     ! if3 name
    91)   integer(IN)      , optional, intent(inout) :: id4      ! int field 4
    92)   character(len=*) , optional, intent(in)    :: id4n     ! if4 name
    93) 
    94)   !--- single scalar variables, assumed valid on the io root pe ---
    95)   real(R8)         , optional, intent(inout) :: rs1      ! real field 1
    96)   character(len=*) , optional, intent(in)    :: rs1n     ! rf1 name
    97)   real(R8)         , optional, intent(inout) :: rs2      ! real field 2
    98)   character(len=*) , optional, intent(in)    :: rs2n     ! rf2 name
    99)   real(R8)         , optional, intent(inout) :: rs3      ! real field 3
   100)   character(len=*) , optional, intent(in)    :: rs3n     ! rf3 name
   101)   real(R8)         , optional, intent(inout) :: rs4      ! real field 4
   102)   character(len=*) , optional, intent(in)    :: rs4n     ! rf4 name
   103)   integer(IN)      , optional, intent(inout) :: is1      ! int field 1
   104)   character(len=*) , optional, intent(in)    :: is1n     ! if1 name
   105)   integer(IN)      , optional, intent(inout) :: is2      ! int field 2
   106)   character(len=*) , optional, intent(in)    :: is2n     ! if2 name
   107)   integer(IN)      , optional, intent(inout) :: is3      ! int field 3
   108)   character(len=*) , optional, intent(in)    :: is3n     ! if3 name
   109)   integer(IN)      , optional, intent(inout) :: is4      ! int field 4
   110)   character(len=*) , optional, intent(in)    :: is4n     ! if4 name
   111) 
   112)   !--- single field, decomposed f90 data in 1d arrays ---
   113)   real(R8)         , optional, intent(inout) :: rf1(:)   ! real field 1
   114)   character(len=*) , optional, intent(in)    :: rf1n     ! rf1 name
   115)   real(R8)         , optional, intent(inout) :: rf2(:)   ! real field 2
   116)   character(len=*) , optional, intent(in)    :: rf2n     ! rf2 name
   117)   real(R8)         , optional, intent(inout) :: rf3(:)   ! real field 3
   118)   character(len=*) , optional, intent(in)    :: rf3n     ! rf3 name
   119)   real(R8)         , optional, intent(inout) :: rf4(:)   ! real field 4
   120)   character(len=*) , optional, intent(in)    :: rf4n     ! rf4 name
   121)   integer(IN)      , optional, intent(inout) :: if1(:)   ! int field 1
   122)   character(len=*) , optional, intent(in)    :: if1n     ! if1 name
   123)   integer(IN)      , optional, intent(inout) :: if2(:)   ! int field 2
   124)   character(len=*) , optional, intent(in)    :: if2n     ! if2 name
   125)   integer(IN)      , optional, intent(inout) :: if3(:)   ! int field 3
   126)   character(len=*) , optional, intent(in)    :: if3n     ! if3 name
   127)   integer(IN)      , optional, intent(inout) :: if4(:)   ! int field 4
   128)   character(len=*) , optional, intent(in)    :: if4n     ! if4 name
   129) 
   130)   !--- attr vect, decomposed f90 data in av datatype ---
   131)   type(mct_aVect)  , optional, intent(inout) :: av1      ! avect 1
   132)   character(len=*) , optional, intent(in)    :: av1n     ! av1 name
   133)   type(mct_aVect)  , optional, intent(inout) :: av2      ! avect 2
   134)   character(len=*) , optional, intent(in)    :: av2n     ! av2 name
   135)   type(mct_aVect)  , optional, intent(inout) :: av3      ! avect 3
   136)   character(len=*) , optional, intent(in)    :: av3n     ! av3 name
   137)   type(mct_aVect)  , optional, intent(inout) :: av4      ! avect 4
   138)   character(len=*) , optional, intent(in)    :: av4n     ! av4 name
   139) 
   140)   !--- local ---
   141)   integer(IN)   :: iam,ntasks
   142)   integer(IN)   :: ier,rcode
   143)   integer(IN)   :: loop,minloop,maxloop
   144)   integer(IN)   :: n,nf
   145)   logical       :: readtype
   146)   integer(IN)   :: lsize,gsize
   147)   logical       :: lclobber
   148)   logical       :: lcdf64
   149)   logical       :: exists
   150)   integer       :: nmode
   151)   character(CL) :: fname
   152)   character(CL) :: vname
   153)   type(mct_string) :: mstring     ! mct char type
   154)   integer(IN)   :: dimid1(1)
   155) 
   156) 
   157)   type(file_desc_t)     :: fid
   158)   type(var_desc_t)      :: varid
   159)   type(io_desc_t)       :: iodescd
   160)   type(io_desc_t)       :: iodesci
   161)   integer(IN), pointer  :: ldof(:)
   162) 
   163)   character(len=*),parameter :: subname = '(shr_pcdf_readwrite) '
   164)   
   165)   !-------------
   166) 
   167)   if (trim(type) == 'read') then
   168)      readtype = .true.
   169)   elseif (trim(type) == 'write') then
   170)      readtype = .false.
   171)   else
   172)      call shr_sys_abort(subname//' ERROR: read write type invalid')
   173)   endif
   174) 
   175)   lclobber = .false.
   176)   if (present(clobber)) lclobber=clobber
   177) 
   178)   lcdf64 = .false.
   179)   if (present(cdf64)) lcdf64=cdf64
   180) 
   181)   call mpi_comm_size(mpicom,ntasks,ier)
   182)   call mpi_comm_rank(mpicom,iam,ier)
   183) 
   184)   if (iam == 0) then
   185)      write(shr_log_unit,*) subname,' filename   = ',trim(filename)
   186)      write(shr_log_unit,*) subname,' type       = ',trim(type)
   187)      write(shr_log_unit,*) subname,' clobber    = ',lclobber
   188)      write(shr_log_unit,*) subname,' cdf64      = ',lcdf64
   189)      call shr_sys_flush(shr_log_unit)
   190)   endif
   191) 
   192)   if (present(gsmap) .and. present(dof)) then
   193)      call shr_sys_abort(trim(subname)//' ERROR: either gsmap OR dof must be an argument')
   194)   endif
   195)   if (present(gsmap)) then
   196)      lsize = mct_gsmap_lsize(gsmap,mpicom)
   197)      gsize = mct_gsmap_gsize(gsmap)
   198)      call mct_gsmap_OrderedPoints(gsmap, iam, ldof)
   199)      call pio_initdecomp(iosystem, pio_double, (/gsize/), ldof, iodescd)
   200)      call pio_initdecomp(iosystem, pio_int   , (/gsize/), ldof, iodesci)
   201)      deallocate(ldof)
   202)   elseif (present(dof)) then
   203)      lsize = size(dof)
   204)      call shr_mpi_sum(lsize,gsize,mpicom,string=trim(subname),all=.true.)
   205)      call pio_initdecomp(iosystem, pio_double, (/gsize/), ldof, iodescd)
   206)      call pio_initdecomp(iosystem, pio_int   , (/gsize/), ldof, iodesci)
   207)   else
   208)      call shr_sys_abort(trim(subname)//' ERROR: either gsmap OR dof must be an argument')
   209)   endif
   210) 
   211)   if (iam == 0) then
   212)      if (len_trim(filename) == 0) then
   213)         call shr_sys_abort(trim(subname)//' ERROR: filename is empty')
   214)      endif
   215)      inquire(file=trim(filename),exist=exists)
   216)   endif
   217)   call shr_mpi_bcast(exists,mpicom,trim(subname)//' exists')
   218) 
   219)   if (readtype) then
   220)      if (.not.exists) then
   221)         call shr_sys_abort(trim(subname)//' ERROR: '//trim(filename)//' doesnt exist')
   222)      endif
   223)      nmode = pio_nowrite
   224)      rcode = pio_openfile(iosystem, fid, pio_iotype, trim(filename), nmode)
   225)   else
   226)      if (.not.lclobber .and. exists) then
   227)         call shr_sys_abort(trim(subname)//' ERROR: '//trim(filename)//' exists, no clobber set')
   228)      endif
   229)      if (lclobber .or. .not.exists) then
   230)         nmode = pio_clobber
   231)         if (lcdf64) nmode = ior(nmode,PIO_64BIT_OFFSET)
   232)         rcode = pio_createfile(iosystem, fid, pio_iotype, trim(filename), nmode)
   233)      else
   234)         nmode = pio_write
   235)         if (lcdf64) nmode = ior(nmode,PIO_64BIT_OFFSET)
   236)         rcode = pio_openfile(iosystem, fid, pio_iotype, trim(filename), nmode)
   237)      endif
   238)      rcode = pio_put_att(fid,pio_global,"file_version",version)
   239)   endif
   240)   call pio_seterrorhandling(fid,PIO_INTERNAL_ERROR)
   241) 
   242)   if (readtype) then
   243)      minloop = 11
   244)      maxloop = 11
   245)   else
   246)      minloop = 21
   247)      maxloop = 22
   248)   endif
   249) 
   250)   ! loop = 11 is read
   251)   ! loop = 21 is define
   252)   ! loop = 22 is write
   253)   do loop = minloop,maxloop
   254) 
   255)      if (loop == 21) rcode = pio_def_dim(fid,'gsize',gsize,dimid1(1))
   256) 
   257)      if (present(id1)) then
   258)         fname = 'id1'
   259)         if (present(id1n)) fname = trim(id1n)
   260)         if (loop == 11) call shr_pcdf_readdim(fid,trim(fname),id1)
   261)         if (loop == 21) call shr_pcdf_writedim(fid,trim(fname),id1)
   262)      endif
   263) 
   264)      if (present(id2)) then
   265)         fname = 'id2'
   266)         if (present(id2n)) fname = trim(id2n)
   267)         if (loop == 11) call shr_pcdf_readdim(fid,trim(fname),id2)
   268)         if (loop == 21) call shr_pcdf_writedim(fid,trim(fname),id2)
   269)      endif
   270) 
   271)      if (present(id3)) then
   272)         fname = 'id3'
   273)         if (present(id3n)) fname = trim(id3n)
   274)         if (loop == 11) call shr_pcdf_readdim(fid,trim(fname),id3)
   275)         if (loop == 21) call shr_pcdf_writedim(fid,trim(fname),id3)
   276)      endif
   277) 
   278)      if (present(id4)) then
   279)         fname = 'id4'
   280)         if (present(id4n)) fname = trim(id4n)
   281)         if (loop == 11) call shr_pcdf_readdim(fid,trim(fname),id4)
   282)         if (loop == 21) call shr_pcdf_writedim(fid,trim(fname),id4)
   283)      endif
   284) 
   285)      if (present(rs1)) then
   286)         fname = 'rs1'
   287)         if (present(rs1n)) fname = trim(rs1n)
   288)         if (loop == 11) call shr_pcdf_readr0d(fid,trim(fname),rs1)
   289)         if (loop == 21) call shr_pcdf_defvar0d(fid,trim(fname),PIO_DOUBLE)
   290)         if (loop == 22) call shr_pcdf_writer0d(fid,trim(fname),rs1)
   291)      endif
   292) 
   293)      if (present(rs2)) then
   294)         fname = 'rs2'
   295)         if (present(rs2n)) fname = trim(rs2n)
   296)         if (loop == 11) call shr_pcdf_readr0d(fid,trim(fname),rs2)
   297)         if (loop == 21) call shr_pcdf_defvar0d(fid,trim(fname),PIO_DOUBLE)
   298)         if (loop == 22) call shr_pcdf_writer0d(fid,trim(fname),rs2)
   299)      endif
   300) 
   301)      if (present(rs3)) then
   302)         fname = 'rs3'
   303)         if (present(rs3n)) fname = trim(rs3n)
   304)         if (loop == 11) call shr_pcdf_readr0d(fid,trim(fname),rs3)
   305)         if (loop == 21) call shr_pcdf_defvar0d(fid,trim(fname),PIO_DOUBLE)
   306)         if (loop == 22) call shr_pcdf_writer0d(fid,trim(fname),rs3)
   307)      endif
   308) 
   309)      if (present(rs4)) then
   310)         fname = 'rs4'
   311)         if (present(rs4n)) fname = trim(rs4n)
   312)         if (loop == 11) call shr_pcdf_readr0d(fid,trim(fname),rs4)
   313)         if (loop == 21) call shr_pcdf_defvar0d(fid,trim(fname),PIO_DOUBLE)
   314)         if (loop == 22) call shr_pcdf_writer0d(fid,trim(fname),rs4)
   315)      endif
   316) 
   317)      if (present(is1)) then
   318)         fname = 'is1'
   319)         if (present(is1n)) fname = trim(is1n)
   320)         if (loop == 11) call shr_pcdf_readi0d(fid,trim(fname),is1)
   321)         if (loop == 21) call shr_pcdf_defvar0d(fid,trim(fname),PIO_INT)
   322)         if (loop == 22) call shr_pcdf_writei0d(fid,trim(fname),is1)
   323)      endif
   324) 
   325)      if (present(is2)) then
   326)         fname = 'is2'
   327)         if (present(is2n)) fname = trim(is2n)
   328)         if (loop == 11) call shr_pcdf_readi0d(fid,trim(fname),is2)
   329)         if (loop == 21) call shr_pcdf_defvar0d(fid,trim(fname),PIO_INT)
   330)         if (loop == 22) call shr_pcdf_writei0d(fid,trim(fname),is2)
   331)      endif
   332)  
   333)      if (present(is3)) then
   334)         fname = 'is3'
   335)         if (present(is3n)) fname = trim(is3n)
   336)         if (loop == 11) call shr_pcdf_readi0d(fid,trim(fname),is3)
   337)         if (loop == 21) call shr_pcdf_defvar0d(fid,trim(fname),PIO_INT)
   338)         if (loop == 22) call shr_pcdf_writei0d(fid,trim(fname),is3)
   339)      endif
   340) 
   341)      if (present(is4)) then
   342)         fname = 'is4'
   343)         if (present(is4n)) fname = trim(is4n)
   344)         if (loop == 11) call shr_pcdf_readi0d(fid,trim(fname),is4)
   345)         if (loop == 21) call shr_pcdf_defvar0d(fid,trim(fname),PIO_INT)
   346)         if (loop == 22) call shr_pcdf_writei0d(fid,trim(fname),is4)
   347)      endif
   348) 
   349)     if (present(rf1)) then
   350)         fname = 'rf1'
   351)         if (present(rf1n)) fname = trim(rf1n)
   352)         if (loop == 11) call shr_pcdf_readr1d(fid,trim(fname),iodescd,rf1)
   353)         if (loop == 21) call shr_pcdf_defvar1d(fid,trim(fname),PIO_DOUBLE,dimid1)
   354)         if (loop == 22) call shr_pcdf_writer1d(fid,trim(fname),iodescd,rf1)
   355)      endif
   356) 
   357)      if (present(rf2)) then
   358)         fname = 'rf2'
   359)         if (present(rf2n)) fname = trim(rf2n)
   360)         if (loop == 11) call shr_pcdf_readr1d(fid,trim(fname),iodescd,rf2)
   361)         if (loop == 21) call shr_pcdf_defvar1d(fid,trim(fname),PIO_DOUBLE,dimid1)
   362)         if (loop == 22) call shr_pcdf_writer1d(fid,trim(fname),iodescd,rf2)
   363)      endif
   364) 
   365)      if (present(rf3)) then
   366)         fname = 'rf3'
   367)         if (present(rf3n)) fname = trim(rf3n)
   368)         if (loop == 11) call shr_pcdf_readr1d(fid,trim(fname),iodescd,rf3)
   369)         if (loop == 21) call shr_pcdf_defvar1d(fid,trim(fname),PIO_DOUBLE,dimid1)
   370)         if (loop == 22) call shr_pcdf_writer1d(fid,trim(fname),iodescd,rf3)
   371)      endif
   372) 
   373)      if (present(rf4)) then
   374)         fname = 'rf4'
   375)         if (present(rf4n)) fname = trim(rf4n)
   376)         if (loop == 11) call shr_pcdf_readr1d(fid,trim(fname),iodescd,rf4)
   377)         if (loop == 21) call shr_pcdf_defvar1d(fid,trim(fname),PIO_DOUBLE,dimid1)
   378)         if (loop == 22) call shr_pcdf_writer1d(fid,trim(fname),iodescd,rf4)
   379)      endif
   380) 
   381)      if (present(if1)) then
   382)         fname = 'if1'
   383)         if (present(if1n)) fname = trim(if1n)
   384)         if (loop == 11) call shr_pcdf_readi1d(fid,trim(fname),iodesci,if1)
   385)         if (loop == 21) call shr_pcdf_defvar1d(fid,trim(fname),PIO_INT,dimid1)
   386)         if (loop == 22) call shr_pcdf_writei1d(fid,trim(fname),iodesci,if1)
   387)      endif
   388) 
   389)      if (present(if2)) then
   390)         fname = 'if2'
   391)         if (present(if2n)) fname = trim(if2n)
   392)         if (loop == 11) call shr_pcdf_readi1d(fid,trim(fname),iodesci,if2)
   393)         if (loop == 21) call shr_pcdf_defvar1d(fid,trim(fname),PIO_INT,dimid1)
   394)         if (loop == 22) call shr_pcdf_writei1d(fid,trim(fname),iodesci,if2)
   395)      endif
   396) 
   397)      if (present(if3)) then
   398)         fname = 'if3'
   399)         if (present(if3n)) fname = trim(if3n)
   400)         if (loop == 11) call shr_pcdf_readi1d(fid,trim(fname),iodesci,if3)
   401)         if (loop == 21) call shr_pcdf_defvar1d(fid,trim(fname),PIO_INT,dimid1)
   402)         if (loop == 22) call shr_pcdf_writei1d(fid,trim(fname),iodesci,if3)
   403)      endif
   404) 
   405)      if (present(if4)) then
   406)         fname = 'if4'
   407)         if (present(if4n)) fname = trim(if4n)
   408)         if (loop == 11) call shr_pcdf_readi1d(fid,trim(fname),iodesci,if4)
   409)         if (loop == 21) call shr_pcdf_defvar1d(fid,trim(fname),PIO_INT,dimid1)
   410)         if (loop == 22) call shr_pcdf_writei1d(fid,trim(fname),iodesci,if4)
   411)      endif
   412) 
   413)      if (present(av1)) then
   414)         fname = 'av1_'
