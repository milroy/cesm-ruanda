     7) ! !MODULE: shr_map_mod -- generic map data type and associated methods
     8) !
     9) ! !DESCRIPTION:
    10) !    Generic map data type and associated methods
    11) !    \newline
    12) !    This module supports mapping of fields from one grid to another.
    13) !    A general datatype, shr\_map\_mapType, stores the mapping information
    14) !    set in shr\_map\_mapSet.  shr\_map\_mapData then allows this mapping
    15) !    to be applied to an input array to generate the output array.
    16) !    \newline
    17) !    The mapType has several flags that give the user various options
    18) !    for setting the mapping
    19) !      type: [remap,fill]
    20) !         remap - mapping of data between different grids, primarily
    21) !            for the active grid area
    22) !         fill - mapping of data on the same grid, primarily to fill missing
    23) !            areas, copy data, or set the array to a spval.
    24) !      algo: [copy,bilinear,nn,nnoni,nnonj,spval]
    25) !         copy - copy data from one array to another using indexing
    26) !         bilinear - bilinear remapping using 4 corner points
    27) !         nn - nearest neighbor, set value to nn value
    28) !         nnoni - nearest neighbor using i, search for nearest neighbor in the
    29) !            i direction first, then j
    30) !         nnonj - nearest neighbor using j, search for nearest neighbor in the
    31) !            j direction first, then i
    32) !         spval - set values to the spval
    33) !      mask: [srcmask,dstmask,nomask,bothmask]
    34) !         srcmask - use only src points with mask = true in mapping
    35) !         dstmask - map only to dst points where mask = true
    36) !         nomask - ignore both src and dst mask in mapping
    37) !         bothmask - use both src and dst mask in mapping (srcmask and dstmask)
    38) !      vect: [scalar,vector]
    39) !         scalar - fields are scalar type (default)
    40) !         vector - fields are vector type, operates only on 2 fields to 2 fields
    41) !    NOTE: Not all combinatations are unique and not all combinations are valid
    42) !    \newline
    43) !    The above settings are put into the maptype using shr\_map\_put.  Public
    44) !    parameters are available to users to set the switches.  The first three
    45) !    switches must be set then the mapSet method can be called.  After the
    46) !    mapSet method is called, the mapData method can be used.
    47) !    \newline
    48) !    A Note on Subroutine Arguments:
    49) !    Lat, lon, and mask arguments in these routines are 2d (nx,ny) 
    50) !    Array arguments are 2d (nf,nxy), number of fields by grid point
    51) !    \newline
    52) !    General Usage:
    53) !       type(shr\_map\_mapType) :: mymap
    54) !       call shr\_map\_put(mymap,'type','remap')
    55) !       call shr\_map\_put(mymap,shr\_map\_fs\_algo,shr\_map\_fs\_bilinear)
    56) !       call shr\_map\_put(mymap,shr\_map\_fs\_mask,'bothmask')
    57) !       call shr\_map\_put(mymap,shr\_map\_fs\_vect,'scalar')
    58) !       call shr\_map\_mapSet(mymap,Xsrc,Ysrc,Msrc,Xdst,Ydst,Mdst,rc=rCode)
    59) !       call shr\_map\_mapData(Asrc,Adst,mymap)
    60) !       \newline
    61) !       call shr\_map\_mapSet(mymap,Xs,Ys,Ms,Xd,Yd,Md,name='fillnnoni',type='fill',algo='nnoni',mask='dstmask',rc=rc)
    62) !       call shr\_map\_mapData(Asrc,Adst,mymap)
    63) !       \newline
    64) !       call shr\_map\_mapData(Ain,Aout,Xs,Ys,Ms,Xd,Yd,Md,type='remap',algo='nn',mask='dstmask',rc)
    65) !
    66) ! !REMARKS:

  1569) 
  1570)   write(s_logunit,F00) 'ERROR this routine is not validated'

  1573)   lrc = 0
  1574)   if (present(rc)) rc = lrc
  1575)

  1607)               (Xdst(i) > rmax .and. Xdst(i)-cang >= rmin))
  1608)       if (Xdst(i) < rmin) then
  1609)         Xdst(i) = Xdst(i) + cang
  1610)       elseif (Xdst(i) > rmax) then
  1611)         Xdst(i) = Xdst(i) - cang
  1612)       else

  1614)       endif
  1615)     enddo
  1616)   enddo
  1617)   
  1618)   call shr_map_checkGrids_dest(Xsrc,Ysrc,Msrc,Xdst,Ydst,Mdst,map,lrc)

  1621)   map%nsrc = nis*njs
  1622)   map%ndst = ndst

  1624) ! deallocate(map%xsrc,stat=irc) ! this used to be a safe way to delloc when necessary,
  1625) ! deallocate(map%ysrc,stat=irc) ! but do nothing when pointers were undefined or
  1626) ! deallocate(map%xdst,stat=irc) ! un-associated, in Oct 2005, undefined ptrs started
  1627) ! deallocate(map%ydst,stat=irc) ! causing seg-faults on bluesky (B. Kauffman)
  1628)   allocate(map%xsrc(nis*njs))
  1629)   allocate(map%ysrc(nis*njs))

  1631)   allocate(map%ydst(nid*njd))
  1632)   do j=1,njs
  1633)   do i=1,nis
  1634)     call shr_map_2dto1d(n1,nis,njs,i,j)
  1635)     map%xsrc(n1) = Xsrc(i,j)*c2*pi/cang
  1636)     map%ysrc(n1) = Ysrc(i,j)*c2*pi/cang

  1638)   enddo
  1639)   do i=1,nid
  1640)     map%xdst(i) = Xdst(i)*c2*pi/cang

  1644)   masksrc = .false.
  1645)   maskdstbysrc = .false.
  1646)   maskdst = .false.
  1647)   renorm  = .true.

  1649)   if (trim(map%type) /= trim(shr_map_fs_fill) .and. &
  1650)       trim(map%type) /= trim(shr_map_fs_cfill)) then
  1651)     if (trim(map%mask) == trim(shr_map_fs_bothmask) .or. &
  1652)         trim(map%mask) == trim(shr_map_fs_srcmask)) masksrc = .true.

  1654)         trim(map%mask) == trim(shr_map_fs_dstmask)) maskdst = .true.

  1658)     renorm = .false.
  1659)   endif
  1660)

  1662)     if (s_loglev > 0) write(s_logunit,*) ' '
  1663)     call shr_map_print(map)
  1664)   endif
  1665) 
  1666)   if (lrc /= 0) then
  1667)     if (present(rc)) rc = lrc
  1668)     return
  1669)   endif
  1670) 
  1671)   if (trim(map%algo) == trim(shr_map_fs_bilinear)) then
  1672)     if (dopole) then
  1673)       pmax = nis+2   ! possible for high lat points
  1674)       ptot = 4*nid*njd  ! start with bilinear estimate
  1675)     else
  1676)       pmax = 4   ! bilinear with 4 wts/map
  1677)       ptot = 4*nid*njd
  1678)     endif
  1679)   else
  1680)     pmax = 1     ! nn with 1 wts/map
  1681)     ptot = 1*nid*njd
  1682)   endif

  1687)   allocate(ptj(pmax))
  1688)   allocate(ptw(pmax))

  1691)   nwts = nid*njd
  1692)   do n=1,nwts
  1693)     lid(n) = Idst(n)
  1694)     lis(n) = Idst(n)
  1695)     lwt(n) = c1
  1696)   enddo
  1697) 
  1698)   !--- index copy anytime algo = copy ---
  1699)   if (trim(map%algo) == trim(shr_map_fs_copy)) then
  1700)     map%init = initcopy
  1701)     ! just use copy default
  1702) 
  1703)   !--- for fill ---
  1704)   elseif (trim(map%type) == trim(shr_map_fs_fill) .or. &
  1705)           trim(map%type) == trim(shr_map_fs_cfill)) then

  1713)           call shr_map_2dto1d(lis(n),nis,njs,inn,jnn)
  1714)         endif
  1715)       enddo

  1717)       do n=1,nwts
  1718)         if (Mdst(n) == 0) then
  1719)           call shr_map_findnnon('i',Xdst(n),Ydst(n),Xsrc,Ysrc,Msrc,inn,jnn)
  1720)           call shr_map_2dto1d(lis(n),nis,njs,inn,jnn)
  1721)         endif
  1722)       enddo

  1724)       do n=1,nwts
  1725)         if (Mdst(n) == 0) then
  1726)           call shr_map_findnnon('j',Xdst(n),Ydst(n),Xsrc,Ysrc,Msrc,inn,jnn)

  1737)     if (trim(map%algo) == trim(shr_map_fs_spval)) then

  1739)     elseif (trim(map%algo) == trim(shr_map_fs_nn)) then

  1785)         enddo
  1786)       enddo
  1787)     else
  1788)       call shr_map_abort(subName//' ERROR: unsupported map option combo')
  1789)       if (present(rc)) rc = 1
  1790)       return
  1791)     endif
  1792)   else
  1793)     call shr_map_abort(subName//' ERROR: unsupported map option combo')

  1795)     return
  1796)   endif

  1800)   if (debug > 1 .and. s_loglev > 0) write(s_logunit,F01) 'init: ',map%init

  1802)       trim(map%type) /= trim(shr_map_fs_cfill)) then
  1803)     ncnt = 0
  1804)     do n=1,nwts
  1805)       if (lid(n) == lis(n) .and. abs(lwt(n)-c1) < eps) then
  1806)         ! skipit
  1807)       else
  1808)         ncnt = ncnt+1
  1809)         lid(ncnt) = lid(n)
  1810)         lis(ncnt) = lis(n)
  1811)         lwt(ncnt) = lwt(n)
  1812)       endif
  1813)     enddo
  1814)     if (debug > 1 .and. s_loglev > 0) write(s_logunit,F02) ' rm dst grid points, nwts old/new = ',nwts,ncnt
  1815)     nwts = ncnt
  1816)   endif

  1818)   !--- remove dst grid points ---

  1820)   if (maskdst) then
  1821)     ncnt = 0
  1822)     do n=1,nwts
  1823)       if (Mdst(n) /= 0) then
  1824)         ncnt = ncnt+1
  1825)         lid(ncnt) = lid(n)
  1826)         lis(ncnt) = lis(n)
  1827)         lwt(ncnt) = lwt(n)
  1828)       endif
  1829)     enddo
  1830)     if (debug > 1 .and. s_loglev > 0) write(s_logunit,F02) ' rm dst grid points, nwts old/new = ',nwts,ncnt
  1831)     nwts = ncnt
  1832)   endif
  1833) 
  1834)   !--- remove dst grid points based on src mask---
  1835)   if (debug > 1 .and. s_loglev > 0) write(s_logunit,F01) 'maskdstbysrc: ',maskdstbysrc
  1836)   if (maskdstbysrc) then
  1837)     ncnt = 0
  1838)     do n=1,nwts
  1839)       call shr_map_1dto2d(lid(n),nis,njs,i,j)
  1840)       if (Msrc(i,j) /= 0) then
  1841)         ncnt = ncnt+1
  1842)         lid(ncnt) = lid(n)
  1843)         lis(ncnt) = lis(n)
  1844)         lwt(ncnt) = lwt(n)
  1845)       endif
  1846)     enddo
  1847)     if (debug > 1 .and. s_loglev > 0) write(s_logunit,F02) ' rm dst grid points by src, nwts old/new = ',nwts,ncnt
  1848)     nwts = ncnt
  1849)   endif

  1852)   if (debug > 1 .and. s_loglev > 0) write(s_logunit,F01) 'masksrc: ',masksrc
  1853)   if (masksrc) then
  1854)     ncnt = 0
  1855)     do n=1,nwts
  1856)       call shr_map_1dto2d(lis(n),nis,njs,i,j)
  1857)       if (Msrc(i,j) /= 0) then
  1858)         ncnt = ncnt+1
  1859)         lid(ncnt) = lid(n)
  1860)         lis(ncnt) = lis(n)
  1861)         lwt(ncnt) = lwt(n)
  1862)       endif
  1863)     enddo
  1864)     if (debug > 1 .and. s_loglev > 0) write(s_logunit,F02) ' rm src grid points, nwts old/new = ',nwts,ncnt
  1865)     nwts = ncnt
  1866)   endif
  1867)

  1869)   allocate(sum(ndst))
  1870)   !--- sum weights for dst grid points ---
  1871)   sum(:) = c0
  1872)   do n=1,nwts
  1873)     sum(lid(n)) = sum(lid(n)) + lwt(n)
  1874)   enddo
  1875)   !--- print min/max sum ---
  1876)   rmin = maxval(sum)
  1877)   rmax = minval(sum)
  1878)   do n=1,ndst
  1879)     if (sum(n) > eps) then
  1880)       rmin = min(rmin,sum(n))
  1881)       rmax = max(rmax,sum(n))
  1882)     endif
  1883)   enddo
  1884)   if (debug > 1 .and. s_loglev > 0) write(s_logunit,F03) 'sum wts min/max ',rmin,rmax
  1885)   !--- renormalize so sum on destination is always 1.0 for active dst points
  1886)   if (debug > 1 .and. s_loglev > 0) write(s_logunit,F01) 'renorm: ',renorm
  1887)   if (renorm) then
  1888)     do n=1,nwts
  1889)       if (sum(lid(n)) > eps) then
  1890)         lwt(n) = lwt(n) / sum(lid(n))
  1891)       endif
  1892)     enddo

  2982)     ysl = -cpole

  2987)   dy  = (ysu-ysl)

  2989)   dy1 = ( yd-Ysl)

  3009)   dy1 = dy1 / dy

  3016)

  3018)     pti(2) = ir
  3019)     pti(3) = il
  3020)     pti(4) = ir

  3025)     ptj(4) = ju

  3035)     pti(2) = ir
  3036) 
  3037)     ptj(1) = jl
  3038)     ptj(2) = jl

  3053) 
  3054)     ptj(1) = ju
  3055)     ptj(2) = ju
  3056) 
  3057)     ptw(1) = (c1-dx1)*(   dy1)
  3058)     ptw(2) = (   dx1)*(   dy1)

  3060)     do n=1,isize
  3061)       pti(2+n) = n
  3062)       ptj(2+n) = jl
  3063)       ptw(2+n) = (c1-dy1)/real(isize,SHR_KIND_R8)
  3064)     enddo
  3065) 
  3066)   else
  3067) 
  3068)     write(s_logunit,F00) ' ERROR illegal pnum situation '
  3069)     call shr_map_abort(subName//' ERROR illegal pnum situation')
  3070) 
  3071)   endif
  3072) 
  3073) end subroutine shr_map_getWts

  3119)       enddo

  3160)       enddo
  3161)     endif

  3164)     if (Ydst < Ysrc(1,jsize)) then
  3165)       jl = jsize
  3166)       ju = jsize
  3167)     elseif (Ydst > Ysrc(1,1)) then
  3168)       jl = 1
  3169)       ju = 1
