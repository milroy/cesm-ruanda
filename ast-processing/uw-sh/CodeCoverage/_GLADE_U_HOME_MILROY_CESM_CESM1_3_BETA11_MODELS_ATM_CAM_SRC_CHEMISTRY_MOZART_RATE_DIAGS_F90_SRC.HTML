   146)   subroutine parse_rate_sums
   147) 
   148)     integer :: ndxs(512)
   149)     integer :: nelem, spc_len, i,j,k, rxt_ndx
   150)     character(len=CL) :: tmp_str, tmp_name
   151) 
   152)     character(len=8) :: xchr ! multipler
   153)     real(r8) :: xdbl
   154) 
   155)     character(len=CX) :: sum_string
   156) 
   157)     ! a group is  a sum of reaction rates 
   158) 
   159)     ! count the numger of sums (or groups)
   160)     sumcnt: do i = 1,maxsums
   161)        spc_len=len_trim(rxn_rate_sums(i))
   162)        if ( spc_len > 0 ) then
   163)           ngrps = ngrps+1
   164)        else
   165)           exit sumcnt
   166)        endif
   167)     enddo sumcnt
   168) 
   169)     ! parse the individual sum strings...  and form the groupings
   170)     has_grps: if (ngrps>0) then
   171) 
   172)        allocate( grps(ngrps) )
   173) 
   174)        ! from shr_megan_mod ... should be generalized and shared...
   175)        grploop: do i = 1,ngrps
   176) 
   177)           ! parse out the rxn names and multipliers
   178)           ! from first parsing out the terms in the summation equation ("+" separates the terms)
   179) 
   180)           sum_string = rxn_rate_sums(i)
   181)           j = scan( sum_string, '=' )
   182)           nelem = 1
   183)           ndxs(nelem) = j ! ndxs stores the index of each term of the equation
   184) 
   185)           ! find indices of all the terms in the equation
   186)           tmp_str = trim( sum_string(j+1:) )
   187)           j = scan( tmp_str, '+' )
   188)           do while(j>0)
   189)              nelem = nelem+1
   190)              ndxs(nelem) = ndxs(nelem-1) + j
   191)              tmp_str = tmp_str(j+1:)
   192)              j = scan( tmp_str, '+' )
   193)           enddo
   194)           ndxs(nelem+1) = len(sum_string)+1
   195) 
   196)           grps(i)%nmembers = nelem ! number of terms 
   197)           grps(i)%name =  trim(adjustl( sum_string(:ndxs(1)-1))) ! thing to the left of the "=" is used as the name of the group
   198) 
   199)           ! now that we have the number of terms in the summation allocate memory for the map (reaction indices) and multipliers
   200)           allocate(grps(i)%map(nelem)) 
   201)           allocate(grps(i)%multipler(nelem))
   202) 
   203)           ! now parse out the  rxn names and multiplers from the terms 
   204)           elmloop: do k = 1,nelem
   205)              grps(i)%multipler(k) = 1._r8
   206)              ! get the rxn name which follows the '*' operator if the is one
   207)              tmp_name = adjustl(sum_string(ndxs(k)+1:ndxs(k+1)-1))
   208)              j = scan( tmp_name, '*' )
   209)              if (j>0) then
   210)                 xchr = tmp_name(1:j-1) ! get the multipler (left of the '*')
   211)                 read( xchr, * ) xdbl   ! convert the string to a real
   212)                 grps(i)%multipler(k) = xdbl ! store the multiplier
   213)                 tmp_name = adjustl(tmp_name(j+1:)) ! get the rxn name (right of the '*')
   214)              endif
   215)              ! look up the corresponding reaction index ...
   216)              rxt_ndx = lookup_tag_ndx( tmp_name )
   217)              if ( rxt_ndx > 0 ) then
   218)                 grps(i)%map(k) = rxt_ndx
   219)              else
   220)                 call endrun('rate_diags::parse_rate_sums rate name not found : '//trim(tmp_name))
   221)              endif
   222)           enddo elmloop
   223)        enddo grploop
   224)     endif has_grps
   225) 
   226)   end subroutine parse_rate_sums
