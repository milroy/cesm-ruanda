    32) !----------------------------------------------------------------------------
    33)       subroutine setcoef(nlayers, istart, pavel, tavel, tz, tbound, semiss, &
    34)                          coldry, wkl, wbroad, &
    35)                          laytrop, jp, jt, jt1, planklay, planklev, plankbnd, &
    36)                          colh2o, colco2, colo3, coln2o, colco, colch4, colo2, &
    37)                          colbrd, fac00, fac01, fac10, fac11, &
    38)                          rat_h2oco2, rat_h2oco2_1, rat_h2oo3, rat_h2oo3_1, &
    39)                          rat_h2on2o, rat_h2on2o_1, rat_h2och4, rat_h2och4_1, &
    40)                          rat_n2oco2, rat_n2oco2_1, rat_o3co2, rat_o3co2_1, &
    41)                          selffac, selffrac, indself, forfac, forfrac, indfor, &
    42)                          minorfrac, scaleminor, scaleminorn2, indminor)
    43) !----------------------------------------------------------------------------
    44) !
    45) !  Purpose:  For a given atmosphere, calculate the indices and
    46) !  fractions related to the pressure and temperature interpolations.
    47) !  Also calculate the values of the integrated Planck functions 
    48) !  for each band at the level and layer temperatures.
    49) 
    50) ! ------- Declarations -------
    51) 
    52) ! ----- Input -----
    53)       integer, intent(in) :: nlayers         ! total number of layers
    54)       integer, intent(in) :: istart          ! beginning band of calculation
    55) 
    56)       real(kind=r8), intent(in) :: pavel(:)           ! layer pressures (mb) 
    57)                                                         !    Dimensions: (nlayers)
    58)       real(kind=r8), intent(in) :: tavel(:)           ! layer temperatures (K)
    59)                                                         !    Dimensions: (nlayers)
    60)       real(kind=r8), intent(in) :: tz(0:)             ! level (interface) temperatures (K)
    61)                                                         !    Dimensions: (0:nlayers)
    62)       real(kind=r8), intent(in) :: tbound             ! surface temperature (K)
    63)       real(kind=r8), intent(in) :: coldry(:)          ! dry air column density (mol/cm2)
    64)                                                         !    Dimensions: (nlayers)
    65)       real(kind=r8), intent(in) :: wbroad(:)          ! broadening gas column density (mol/cm2)
    66)                                                         !    Dimensions: (nlayers)
    67)       real(kind=r8), intent(in) :: wkl(:,:)           ! molecular amounts (mol/cm-2)
    68)                                                         !    Dimensions: (mxmol,nlayers)
    69)       real(kind=r8), intent(in) :: semiss(:)          ! lw surface emissivity
    70)                                                         !    Dimensions: (nbndlw)
    71) 
    72) ! ----- Output -----
    73)       integer, intent(out) :: laytrop        ! tropopause layer index
    74)       integer, intent(out) :: jp(:)          ! 
    75)                                                         !    Dimensions: (nlayers)
    76)       integer, intent(out) :: jt(:)          !
    77)                                                         !    Dimensions: (nlayers)
    78)       integer, intent(out) :: jt1(:)         !
    79)                                                         !    Dimensions: (nlayers)
    80)       real(kind=r8), intent(out) :: planklay(:,:)     ! 
    81)                                                         !    Dimensions: (nlayers,nbndlw)
    82)       real(kind=r8), intent(out) :: planklev(0:,:)    ! 
    83)                                                         !    Dimensions: (0:nlayers,nbndlw)
    84)       real(kind=r8), intent(out) :: plankbnd(:)       ! 
    85)                                                         !    Dimensions: (nbndlw)
    86) 
    87)       real(kind=r8), intent(out) :: colh2o(:)         ! column amount (h2o)
    88)                                                         !    Dimensions: (nlayers)
    89)       real(kind=r8), intent(out) :: colco2(:)         ! column amount (co2)
    90)                                                         !    Dimensions: (nlayers)
    91)       real(kind=r8), intent(out) :: colo3(:)          ! column amount (o3)
    92)                                                         !    Dimensions: (nlayers)
    93)       real(kind=r8), intent(out) :: coln2o(:)         ! column amount (n2o)
    94)                                                         !    Dimensions: (nlayers)
    95)       real(kind=r8), intent(out) :: colco(:)          ! column amount (co)
    96)                                                         !    Dimensions: (nlayers)
    97)       real(kind=r8), intent(out) :: colch4(:)         ! column amount (ch4)
    98)                                                         !    Dimensions: (nlayers)
    99)       real(kind=r8), intent(out) :: colo2(:)          ! column amount (o2)
   100)                                                         !    Dimensions: (nlayers)
   101)       real(kind=r8), intent(out) :: colbrd(:)         ! column amount (broadening gases)
   102)                                                         !    Dimensions: (nlayers)
   103) 
   104)       integer, intent(out) :: indself(:)
   105)                                                         !    Dimensions: (nlayers)
   106)       integer, intent(out) :: indfor(:)
   107)                                                         !    Dimensions: (nlayers)
   108)       real(kind=r8), intent(out) :: selffac(:)
   109)                                                         !    Dimensions: (nlayers)
   110)       real(kind=r8), intent(out) :: selffrac(:)
   111)                                                         !    Dimensions: (nlayers)
   112)       real(kind=r8), intent(out) :: forfac(:)
   113)                                                         !    Dimensions: (nlayers)
   114)       real(kind=r8), intent(out) :: forfrac(:)
   115)                                                         !    Dimensions: (nlayers)
   116) 
   117)       integer, intent(out) :: indminor(:)
   118)                                                         !    Dimensions: (nlayers)
   119)       real(kind=r8), intent(out) :: minorfrac(:)
   120)                                                         !    Dimensions: (nlayers)
   121)       real(kind=r8), intent(out) :: scaleminor(:)
   122)                                                         !    Dimensions: (nlayers)
   123)       real(kind=r8), intent(out) :: scaleminorn2(:)
   124)                                                         !    Dimensions: (nlayers)
   125) 
   126)       real(kind=r8), intent(out) :: &                 !
   127)                          fac00(:), fac01(:), &          !    Dimensions: (nlayers)
   128)                          fac10(:), fac11(:) 
   129)                                                         
   130)       real(kind=r8), intent(out) :: &                 !
   131)                          rat_h2oco2(:),rat_h2oco2_1(:), &
   132)                          rat_h2oo3(:),rat_h2oo3_1(:), & !    Dimensions: (nlayers)
   133)                          rat_h2on2o(:),rat_h2on2o_1(:), &
   134)                          rat_h2och4(:),rat_h2och4_1(:), &
   135)                          rat_n2oco2(:),rat_n2oco2_1(:), &
   136)                          rat_o3co2(:),rat_o3co2_1(:)
   137)                                                         
   138) 
   139) ! ----- Local -----
   140)       integer :: indbound, indlev0
   141)       integer :: lay, indlay, indlev, iband
   142)       integer :: jp1
   143)       real(kind=r8) :: stpfac, tbndfrac, t0frac, tlayfrac, tlevfrac
   144)       real(kind=r8) :: dbdtlev, dbdtlay
   145)       real(kind=r8) :: plog, fp, ft, ft1, water, scalefac, factor, compfp
   146) 
   147) 
   148)       hvrset = '$Revision: 1.2 $'
   149) 
   150)       stpfac = 296._r8/1013._r8
   151) 
   152)       indbound = tbound - 159._r8
   153)       if (indbound .lt. 1) then
   154)          indbound = 1
   155)       elseif (indbound .gt. 180) then
   156)          indbound = 180
   157)       endif
   158)       tbndfrac = tbound - 159._r8 - float(indbound)
   159)       indlev0 = tz(0) - 159._r8
   160)       if (indlev0 .lt. 1) then
   161)          indlev0 = 1
   162)       elseif (indlev0 .gt. 180) then
   163)          indlev0 = 180
   164)       endif
   165)       t0frac = tz(0) - 159._r8 - float(indlev0)
   166)       laytrop = 0
   167) 
   168) ! Begin layer loop 
   169) !  Calculate the integrated Planck functions for each band at the
   170) !  surface, level, and layer temperatures.
   171)       do lay = 1, nlayers
   172)          indlay = tavel(lay) - 159._r8
   173)          if (indlay .lt. 1) then
   174)             indlay = 1
   175)          elseif (indlay .gt. 180) then
   176)             indlay = 180
   177)          endif
   178)          tlayfrac = tavel(lay) - 159._r8 - float(indlay)
   179)          indlev = tz(lay) - 159._r8
   180)          if (indlev .lt. 1) then
   181)             indlev = 1
   182)          elseif (indlev .gt. 180) then
   183)             indlev = 180
   184)          endif
   185)          tlevfrac = tz(lay) - 159._r8 - float(indlev)
   186) 
   187) ! Begin spectral band loop 
   188)          do iband = 1, 15
   189)             if (lay.eq.1) then
   190)                dbdtlev = totplnk(indbound+1,iband) - totplnk(indbound,iband)
   191)                plankbnd(iband) = semiss(iband) * &
   192)                    (totplnk(indbound,iband) + tbndfrac * dbdtlev)
   193)                dbdtlev = totplnk(indlev0+1,iband)-totplnk(indlev0,iband)
   194)                planklev(0,iband) = totplnk(indlev0,iband) + t0frac * dbdtlev
   195)             endif
   196)             dbdtlev = totplnk(indlev+1,iband) - totplnk(indlev,iband)
   197)             dbdtlay = totplnk(indlay+1,iband) - totplnk(indlay,iband)
   198)             planklay(lay,iband) = totplnk(indlay,iband) + tlayfrac * dbdtlay
   199)             planklev(lay,iband) = totplnk(indlev,iband) + tlevfrac * dbdtlev
   200)          enddo
   201) 
   202) !  For band 16, if radiative transfer will be performed on just
   203) !  this band, use integrated Planck values up to 3250 cm-1.  
   204) !  If radiative transfer will be performed across all 16 bands,
   205) !  then include in the integrated Planck values for this band
   206) !  contributions from 2600 cm-1 to infinity.
   207)          iband = 16
   208)          if (istart .eq. 16) then
   209)             if (lay.eq.1) then
   210)                dbdtlev = totplk16(indbound+1) - totplk16(indbound)
   211)                plankbnd(iband) = semiss(iband) * &
   212)                     (totplk16(indbound) + tbndfrac * dbdtlev)
   213)                dbdtlev = totplnk(indlev0+1,iband)-totplnk(indlev0,iband)
   214)                planklev(0,iband) = totplk16(indlev0) + &
   215)                     t0frac * dbdtlev
   216)             endif
   217)             dbdtlev = totplk16(indlev+1) - totplk16(indlev)
   218)             dbdtlay = totplk16(indlay+1) - totplk16(indlay)
   219)             planklay(lay,iband) = totplk16(indlay) + tlayfrac * dbdtlay
   220)             planklev(lay,iband) = totplk16(indlev) + tlevfrac * dbdtlev
   221)          else
   222)             if (lay.eq.1) then
   223)                dbdtlev = totplnk(indbound+1,iband) - totplnk(indbound,iband)
   224)                plankbnd(iband) = semiss(iband) * &
   225)                     (totplnk(indbound,iband) + tbndfrac * dbdtlev)
   226)                dbdtlev = totplnk(indlev0+1,iband)-totplnk(indlev0,iband)
   227)                planklev(0,iband) = totplnk(indlev0,iband) + t0frac * dbdtlev
   228)             endif
   229)             dbdtlev = totplnk(indlev+1,iband) - totplnk(indlev,iband)
   230)             dbdtlay = totplnk(indlay+1,iband) - totplnk(indlay,iband)
   231)             planklay(lay,iband) = totplnk(indlay,iband) + tlayfrac * dbdtlay
   232)             planklev(lay,iband) = totplnk(indlev,iband) + tlevfrac * dbdtlev
   233)          endif
   234) 
   235) !  Find the two reference pressures on either side of the
   236) !  layer pressure.  Store them in JP and JP1.  Store in FP the
   237) !  fraction of the difference (in ln(pressure)) between these
   238) !  two values that the layer pressure lies.
   239) !         plog = alog(pavel(lay))
   240)          plog = dlog(pavel(lay))
   241)          jp(lay) = int(36._r8 - 5*(plog+0.04_r8))
   242)          if (jp(lay) .lt. 1) then
   243)             jp(lay) = 1
   244)          elseif (jp(lay) .gt. 58) then
   245)             jp(lay) = 58
   246)          endif
   247)          jp1 = jp(lay) + 1
   248)          fp = 5._r8 *(preflog(jp(lay)) - plog)
   249) 
   250) !  Determine, for each reference pressure (JP and JP1), which
   251) !  reference temperature (these are different for each  
   252) !  reference pressure) is nearest the layer temperature but does
   253) !  not exceed it.  Store these indices in JT and JT1, resp.
   254) !  Store in FT (resp. FT1) the fraction of the way between JT
   255) !  (JT1) and the next highest reference temperature that the 
   256) !  layer temperature falls.
   257)          jt(lay) = int(3._r8 + (tavel(lay)-tref(jp(lay)))/15._r8)
   258)          if (jt(lay) .lt. 1) then
   259)             jt(lay) = 1
   260)          elseif (jt(lay) .gt. 4) then
   261)             jt(lay) = 4
   262)          endif
   263)          ft = ((tavel(lay)-tref(jp(lay)))/15._r8) - float(jt(lay)-3)
   264)          jt1(lay) = int(3._r8 + (tavel(lay)-tref(jp1))/15._r8)
   265)          if (jt1(lay) .lt. 1) then
   266)             jt1(lay) = 1
   267)          elseif (jt1(lay) .gt. 4) then
   268)             jt1(lay) = 4
   269)          endif
   270)          ft1 = ((tavel(lay)-tref(jp1))/15._r8) - float(jt1(lay)-3)
   271)          water = wkl(1,lay)/coldry(lay)
   272)          scalefac = pavel(lay) * stpfac / tavel(lay)
   273) 
   274) !  If the pressure is less than ~100mb, perform a different
   275) !  set of species interpolations.
   276)          if (plog .le. 4.56_r8) go to 5300
   277)          laytrop =  laytrop + 1
   278) 
   279)          forfac(lay) = scalefac / (1.+water)
   280)          factor = (332.0_r8-tavel(lay))/36.0_r8
   281)          indfor(lay) = min(2, max(1, int(factor)))
   282)          forfrac(lay) = factor - float(indfor(lay))
   283) 
   284) !  Set up factors needed to separately include the water vapor
   285) !  self-continuum in the calculation of absorption coefficient.
   286)          selffac(lay) = water * forfac(lay)
   287)          factor = (tavel(lay)-188.0_r8)/7.2_r8
   288)          indself(lay) = min(9, max(1, int(factor)-7))
   289)          selffrac(lay) = factor - float(indself(lay) + 7)
   290) 
   291) !  Set up factors needed to separately include the minor gases
   292) !  in the calculation of absorption coefficient
   293)          scaleminor(lay) = pavel(lay)/tavel(lay)
   294)          scaleminorn2(lay) = (pavel(lay)/tavel(lay)) &
   295)              *(wbroad(lay)/(coldry(lay)+wkl(1,lay)))
   296)          factor = (tavel(lay)-180.8_r8)/7.2_r8
   297)          indminor(lay) = min(18, max(1, int(factor)))
   298)          minorfrac(lay) = factor - float(indminor(lay))
   299) 
   300) !  Setup reference ratio to be used in calculation of binary
   301) !  species parameter in lower atmosphere.
   302)          rat_h2oco2(lay)=chi_mls(1,jp(lay))/chi_mls(2,jp(lay))
   303)          rat_h2oco2_1(lay)=chi_mls(1,jp(lay)+1)/chi_mls(2,jp(lay)+1)
   304) 
   305)          rat_h2oo3(lay)=chi_mls(1,jp(lay))/chi_mls(3,jp(lay))
   306)          rat_h2oo3_1(lay)=chi_mls(1,jp(lay)+1)/chi_mls(3,jp(lay)+1)
   307) 
   308)          rat_h2on2o(lay)=chi_mls(1,jp(lay))/chi_mls(4,jp(lay))
   309)          rat_h2on2o_1(lay)=chi_mls(1,jp(lay)+1)/chi_mls(4,jp(lay)+1)
   310) 
   311)          rat_h2och4(lay)=chi_mls(1,jp(lay))/chi_mls(6,jp(lay))
   312)          rat_h2och4_1(lay)=chi_mls(1,jp(lay)+1)/chi_mls(6,jp(lay)+1)
   313) 
   314)          rat_n2oco2(lay)=chi_mls(4,jp(lay))/chi_mls(2,jp(lay))
   315)          rat_n2oco2_1(lay)=chi_mls(4,jp(lay)+1)/chi_mls(2,jp(lay)+1)
   316) 
   317) !  Calculate needed column amounts.
   318)          colh2o(lay) = 1.e-20_r8 * wkl(1,lay)
   319)          colco2(lay) = 1.e-20_r8 * wkl(2,lay)
   320)          colo3(lay) = 1.e-20_r8 * wkl(3,lay)
   321)          coln2o(lay) = 1.e-20_r8 * wkl(4,lay)
   322)          colco(lay) = 1.e-20_r8 * wkl(5,lay)
   323)          colch4(lay) = 1.e-20_r8 * wkl(6,lay)
   324)          colo2(lay) = 1.e-20_r8 * wkl(7,lay)
   325)          if (colco2(lay) .eq. 0._r8) colco2(lay) = 1.e-32_r8 * coldry(lay)
   326)          if (colo3(lay) .eq. 0._r8) colo3(lay) = 1.e-32_r8 * coldry(lay)
   327)          if (coln2o(lay) .eq. 0._r8) coln2o(lay) = 1.e-32_r8 * coldry(lay)
   328)          if (colco(lay) .eq. 0._r8) colco(lay) = 1.e-32_r8 * coldry(lay)
   329)          if (colch4(lay) .eq. 0._r8) colch4(lay) = 1.e-32_r8 * coldry(lay)
   330)          colbrd(lay) = 1.e-20_r8 * wbroad(lay)
   331)          go to 5400
   332) 
   333) !  Above laytrop.
   334)  5300    continue
   335) 
   336)          forfac(lay) = scalefac / (1.+water)
   337)          factor = (tavel(lay)-188.0_r8)/36.0_r8
   338)          indfor(lay) = 3
   339)          forfrac(lay) = factor - 1.0_r8
   340) 
   341) !  Set up factors needed to separately include the water vapor
   342) !  self-continuum in the calculation of absorption coefficient.
   343)          selffac(lay) = water * forfac(lay)
   344) 
   345) !  Set up factors needed to separately include the minor gases
   346) !  in the calculation of absorption coefficient
   347)          scaleminor(lay) = pavel(lay)/tavel(lay)         
   348)          scaleminorn2(lay) = (pavel(lay)/tavel(lay)) &
   349)              * (wbroad(lay)/(coldry(lay)+wkl(1,lay)))
   350)          factor = (tavel(lay)-180.8_r8)/7.2_r8
   351)          indminor(lay) = min(18, max(1, int(factor)))
   352)          minorfrac(lay) = factor - float(indminor(lay))
   353) 
   354) !  Setup reference ratio to be used in calculation of binary
   355) !  species parameter in upper atmosphere.
   356)          rat_h2oco2(lay)=chi_mls(1,jp(lay))/chi_mls(2,jp(lay))
   357)          rat_h2oco2_1(lay)=chi_mls(1,jp(lay)+1)/chi_mls(2,jp(lay)+1)         
   358) 
   359)          rat_o3co2(lay)=chi_mls(3,jp(lay))/chi_mls(2,jp(lay))
   360)          rat_o3co2_1(lay)=chi_mls(3,jp(lay)+1)/chi_mls(2,jp(lay)+1)         
   361) 
   362) !  Calculate needed column amounts.
   363)          colh2o(lay) = 1.e-20_r8 * wkl(1,lay)
   364)          colco2(lay) = 1.e-20_r8 * wkl(2,lay)
   365)          colo3(lay) = 1.e-20_r8 * wkl(3,lay)
   366)          coln2o(lay) = 1.e-20_r8 * wkl(4,lay)
   367)          colco(lay) = 1.e-20_r8 * wkl(5,lay)
   368)          colch4(lay) = 1.e-20_r8 * wkl(6,lay)
   369)          colo2(lay) = 1.e-20_r8 * wkl(7,lay)
   370)          if (colco2(lay) .eq. 0._r8) colco2(lay) = 1.e-32_r8 * coldry(lay)
   371)          if (colo3(lay) .eq. 0._r8) colo3(lay) = 1.e-32_r8 * coldry(lay)
   372)          if (coln2o(lay) .eq. 0._r8) coln2o(lay) = 1.e-32_r8 * coldry(lay)
   373)          if (colco(lay)  .eq. 0._r8) colco(lay) = 1.e-32_r8 * coldry(lay)
   374)          if (colch4(lay) .eq. 0._r8) colch4(lay) = 1.e-32_r8 * coldry(lay)
   375)          colbrd(lay) = 1.e-20_r8 * wbroad(lay)
   376)  5400    continue
   377) 
   378) !  We have now isolated the layer ln pressure and temperature,
   379) !  between two reference pressures and two reference temperatures 
   380) !  (for each reference pressure).  We multiply the pressure 
   381) !  fraction FP with the appropriate temperature fractions to get 
   382) !  the factors that will be needed for the interpolation that yields
   383) !  the optical depths (performed in routines TAUGBn for band n).`
   384) 
   385)          compfp = 1. - fp
   386)          fac10(lay) = compfp * ft
   387)          fac00(lay) = compfp * (1._r8 - ft)
   388)          fac11(lay) = fp * ft1
   389)          fac01(lay) = fp * (1._r8 - ft1)
   390) 
   391) !  Rescale selffac and forfac for use in taumol
   392)          selffac(lay) = colh2o(lay)*selffac(lay)
   393)          forfac(lay) = colh2o(lay)*forfac(lay)
   394) 
   395) ! End layer loop
   396)       enddo
   397) 
   398)       end subroutine setcoef
   399) 
   400) !***************************************************************************
   401)       subroutine lwatmref
   402) !***************************************************************************
   403) 
   404)       save
   405)  
   406) ! These pressures are chosen such that the ln of the first pressure
   407) ! has only a few non-zero digits (i.e. ln(PREF(1)) = 6.96000) and
   408) ! each subsequent ln(pressure) differs from the previous one by 0.2.
