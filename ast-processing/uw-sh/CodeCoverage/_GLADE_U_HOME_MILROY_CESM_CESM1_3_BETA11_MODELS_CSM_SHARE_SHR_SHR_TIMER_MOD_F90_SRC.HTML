     8)    !----------------------------------------------------------------------------
     9)    !
    10)    ! routines that support multiple CPU timers via F90 intrinsics
    11)    !
    12)    ! Note: 
    13)    ! o if   an operation is requested on an invalid timer number n
    14)    !   then nothing is done in a routine
    15)    ! o if   more than max_timers are requested, 
    16)    !   then timer n=max_timers is "overloaded" and becomes invalid/undefined
    17)    !
    18)    ! * cpp if-defs were introduced in 2005 to work-around a bug in the ORNL Cray
    19)    !   X1 F90 intrinsic system_clock() function -- ideally this Cray bug would be
    20)    !   fixed and cpp if-defs would be unnecessary and removed.
    21)    !
    22)    ! !REVISION HISTORY:

    78) subroutine shr_timer_init
    79) 
    80)    !----- local -----
    81)    integer(SHR_KIND_IN) :: cycles ! count rate return by system clock
    82) #if (defined UNICOSMP)
    83)    integer(kind=8) :: irtc_rate
    84) #endif
    85) 
    86)    !----- i/o formats -----
    87)    character(len=*),parameter :: F00 = "('(shr_timer_init) ',a,i5)"
    88) 
    89) !-------------------------------------------------------------------------------
    90) ! This routine initializes:
    91) ! 1) values in all timer array locations
    92) ! 2) machine parameters necessary for computing cpu time from F90 intrinsics.
    93) !    F90 intrinsic: system_clock(count_rate=cycles, count_max=cycles_max)
    94) !-------------------------------------------------------------------------------
    95) 
    96)    call shr_timer_free_all
    97) 
    98) #if (defined UNICOSMP)
    99)    cycles = irtc_rate()
   100) #else
   101)    call system_clock(count_rate=cycles, count_max=cycles_max)
   102) #endif
   103) 
   104)    if (cycles /= 0) then
   105)      clock_rate = 1.0_SHR_KIND_R8/real(cycles,SHR_KIND_R8)
   106)    else
   107)      clock_rate = 0._SHR_KIND_R8
   108)      if (s_loglev > 0) write(s_logunit,F00) 'ERROR: no system clock available' 
   109)    endif
   110)

   115) subroutine shr_timer_get(n, str)
   116)

   119)    character (*)       ,intent( in) :: str  ! text string with timer name
   120) 
   121)    !----- i/o formats -----
   122)    character(len=*),parameter :: F00 = "('(shr_timer_get) ',a,i5)"
   123) 
   124) !-----------------------------------------------------------------------
   125) !  search for next free timer
   126) !-----------------------------------------------------------------------
   127) 
   128)    do n=1,max_timers
   129)      if (status(n) == stat_free) then
   130)        status(n) = stat_inuse
   131)        name  (n) = str
   132)        calls (n) = 0
   133)        return
   134)      endif
   135)    end do

   147)    !----- arguments -----

   151) #if (defined UNICOSMP)

   158) !-----------------------------------------------------------------------
   159) !  This routine starts a given timer.
   160) !-----------------------------------------------------------------------
   161) 
   162)    if ( n>0 .and. n<=max_timers) then
   163)      if (status(n) == stat_started) call shr_timer_stop(n)

   171)    else

   173)    end if

   179) subroutine shr_timer_stop(n)
   180) 
   181)    !----- arguments -----
   182)    integer(SHR_KIND_IN), intent(in) :: n  ! timer number
   183) 
   184)    !----- local -----
   185)    real (SHR_KIND_R8) :: elapse      ! elapsed time returned by system counter
   186) #if (defined UNICOSMP)
   187)    integer(kind=8) :: irtc
   188) #endif
   189) 
   190)    !----- i/o formats -----
   191)    character(len=*),parameter :: F00 = "('(shr_timer_stop) ',a,i5)"
   192) 
   193) !-------------------------------------------------------------------------------
   194) !  This routine stops a given timer, checks for cycle wrapping, computes the 
   195) !  elapsed time, and accumulates the elapsed time in the dt(n) array
   196) !-------------------------------------------------------------------------------
   197) 
   198)    if ( n>0 .and. n<=max_timers) then
   199)      if ( status(n) == stat_started) then
   200) #if (defined UNICOSMP)
   201)        cycles2(n) = irtc()
   202) #else
   203)        call system_clock(count=cycles2(n))
   204) #endif
   205)        if (cycles2(n) >= cycles1(n)) then
   206)          dt(n) = dt(n) + clock_rate*(cycles2(n) - cycles1(n))
   207)        else
   208)          dt(n) = dt(n) + clock_rate*(cycles_max + cycles2(n) - cycles1(n))
   209)        endif
   210)        calls (n) = calls(n) + 1
   211)        status(n) = stat_stopped
   212)      end if
   213)    else
   214)      if (s_loglev > 0) write(s_logunit,F00) 'ERROR: invalid timer number: ',n
   215)    end if
   216) 
   217) end subroutine shr_timer_stop
