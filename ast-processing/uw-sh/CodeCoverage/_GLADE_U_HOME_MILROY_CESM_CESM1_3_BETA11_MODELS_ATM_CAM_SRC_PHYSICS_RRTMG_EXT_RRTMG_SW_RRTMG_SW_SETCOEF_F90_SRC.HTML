    32)       subroutine setcoef_sw(nlayers, pavel, tavel, pz, tz, tbound, coldry, wkl, &
    33)                             laytrop, layswtch, laylow, jp, jt, jt1, &
    34)                             co2mult, colch4, colco2, colh2o, colmol, coln2o, &
    35)                             colo2, colo3, fac00, fac01, fac10, fac11, &
    36)                             selffac, selffrac, indself, forfac, forfrac, indfor)
    37) !----------------------------------------------------------------------------
    38) !
    39) ! Purpose:  For a given atmosphere, calculate the indices and
    40) ! fractions related to the pressure and temperature interpolations.
    41) 
    42) ! Modifications:
    43) ! Original: J. Delamere, AER, Inc. (version 2.5, 02/04/01)
    44) ! Revised: Rewritten and adapted to ECMWF F90, JJMorcrette 030224
    45) ! Revised: For uniform rrtmg formatting, MJIacono, Jul 2006
    46) 
    47) ! ------ Declarations -------
    48) 
    49) ! ----- Input -----
    50)       integer, intent(in) :: nlayers         ! total number of layers
    51) 
    52)       real(kind=r8), intent(in) :: pavel(:)           ! layer pressures (mb) 
    53)                                                         !    Dimensions: (nlayers)
    54)       real(kind=r8), intent(in) :: tavel(:)           ! layer temperatures (K)
    55)                                                         !    Dimensions: (nlayers)
    56)       real(kind=r8), intent(in) :: pz(0:)             ! level (interface) pressures (hPa, mb)
    57)                                                         !    Dimensions: (0:nlayers)
    58)       real(kind=r8), intent(in) :: tz(0:)             ! level (interface) temperatures (K)
    59)                                                         !    Dimensions: (0:nlayers)
    60)       real(kind=r8), intent(in) :: tbound             ! surface temperature (K)
    61)       real(kind=r8), intent(in) :: coldry(:)          ! dry air column density (mol/cm2)
    62)                                                         !    Dimensions: (nlayers)
    63)       real(kind=r8), intent(in) :: wkl(:,:)           ! molecular amounts (mol/cm-2)
    64)                                                         !    Dimensions: (mxmol,nlayers)
    65) 
    66) ! ----- Output -----
    67)       integer, intent(out) :: laytrop        ! tropopause layer index
    68)       integer, intent(out) :: layswtch       ! 
    69)       integer, intent(out) :: laylow         ! 
    70) 
    71)       integer, intent(out) :: jp(:)          ! 
    72)                                                         !    Dimensions: (nlayers)
    73)       integer, intent(out) :: jt(:)          !
    74)                                                         !    Dimensions: (nlayers)
    75)       integer, intent(out) :: jt1(:)         !
    76)                                                         !    Dimensions: (nlayers)
    77) 
    78)       real(kind=r8), intent(out) :: colh2o(:)         ! column amount (h2o)
    79)                                                         !    Dimensions: (nlayers)
    80)       real(kind=r8), intent(out) :: colco2(:)         ! column amount (co2)
    81)                                                         !    Dimensions: (nlayers)
    82)       real(kind=r8), intent(out) :: colo3(:)          ! column amount (o3)
    83)                                                         !    Dimensions: (nlayers)
    84)       real(kind=r8), intent(out) :: coln2o(:)         ! column amount (n2o)
    85)                                                         !    Dimensions: (nlayers)
    86)       real(kind=r8), intent(out) :: colch4(:)         ! column amount (ch4)
    87)                                                         !    Dimensions: (nlayers)
    88)       real(kind=r8), intent(out) :: colo2(:)          ! column amount (o2)
    89)                                                         !    Dimensions: (nlayers)
    90)       real(kind=r8), intent(out) :: colmol(:)         ! 
    91)                                                         !    Dimensions: (nlayers)
    92)       real(kind=r8), intent(out) :: co2mult(:)        !
    93)                                                         !    Dimensions: (nlayers)
    94) 
    95)       integer, intent(out) :: indself(:)
    96)                                                         !    Dimensions: (nlayers)
    97)       integer, intent(out) :: indfor(:)
    98)                                                         !    Dimensions: (nlayers)
    99)       real(kind=r8), intent(out) :: selffac(:)
   100)                                                         !    Dimensions: (nlayers)
   101)       real(kind=r8), intent(out) :: selffrac(:)
   102)                                                         !    Dimensions: (nlayers)
   103)       real(kind=r8), intent(out) :: forfac(:)
   104)                                                         !    Dimensions: (nlayers)
   105)       real(kind=r8), intent(out) :: forfrac(:)
   106)                                                         !    Dimensions: (nlayers)
   107) 
   108)       real(kind=r8), intent(out) :: &                 !
   109)                          fac00(:), fac01(:), &          !    Dimensions: (nlayers)
   110)                          fac10(:), fac11(:) 
   111) 
   112) ! ----- Local -----
   113) 
   114)       integer :: indbound
   115)       integer :: indlev0
   116)       integer :: lay
   117)       integer :: jp1
   118) 
   119)       real(kind=r8) :: stpfac
   120)       real(kind=r8) :: tbndfrac
   121)       real(kind=r8) :: t0frac
   122)       real(kind=r8) :: plog
   123)       real(kind=r8) :: fp
   124)       real(kind=r8) :: ft
   125)       real(kind=r8) :: ft1
   126)       real(kind=r8) :: water
   127)       real(kind=r8) :: scalefac
   128)       real(kind=r8) :: factor
   129)       real(kind=r8) :: co2reg
   130)       real(kind=r8) :: compfp
   131) 
   132) 
   133) ! Initializations
   134)       stpfac = 296._r8/1013._r8
   135) 
   136)       indbound = tbound - 159._r8
   137)       tbndfrac = tbound - int(tbound)
   138)       indlev0  = tz(0) - 159._r8
   139)       t0frac   = tz(0) - int(tz(0))
   140) 
   141)       laytrop  = 0
   142)       layswtch = 0
   143)       laylow   = 0
   144) 
   145) ! Begin layer loop
   146)       do lay = 1, nlayers
   147) ! Find the two reference pressures on either side of the
   148) ! layer pressure.  Store them in JP and JP1.  Store in FP the
   149) ! fraction of the difference (in ln(pressure)) between these
   150) ! two values that the layer pressure lies.
   151) 
   152)          plog = log(pavel(lay))
   153)          jp(lay) = int(36._r8 - 5*(plog+0.04_r8))
   154)          if (jp(lay) .lt. 1) then
   155)             jp(lay) = 1
   156)          elseif (jp(lay) .gt. 58) then
   157)             jp(lay) = 58
   158)          endif
   159)          jp1 = jp(lay) + 1
   160)          fp = 5._r8 * (preflog(jp(lay)) - plog)
   161) 
   162) ! Determine, for each reference pressure (JP and JP1), which
   163) ! reference temperature (these are different for each  
   164) ! reference pressure) is nearest the layer temperature but does
   165) ! not exceed it.  Store these indices in JT and JT1, resp.
   166) ! Store in FT (resp. FT1) the fraction of the way between JT
   167) ! (JT1) and the next highest reference temperature that the 
   168) ! layer temperature falls.
   169) 
   170)          jt(lay) = int(3._r8 + (tavel(lay)-tref(jp(lay)))/15._r8)
   171)          if (jt(lay) .lt. 1) then
   172)             jt(lay) = 1
   173)          elseif (jt(lay) .gt. 4) then
   174)             jt(lay) = 4
   175)          endif
   176)          ft = ((tavel(lay)-tref(jp(lay)))/15._r8) - float(jt(lay)-3)
   177)          jt1(lay) = int(3._r8 + (tavel(lay)-tref(jp1))/15._r8)
   178)          if (jt1(lay) .lt. 1) then
   179)             jt1(lay) = 1
   180)          elseif (jt1(lay) .gt. 4) then
   181)             jt1(lay) = 4
   182)          endif
   183)          ft1 = ((tavel(lay)-tref(jp1))/15._r8) - float(jt1(lay)-3)
   184) 
   185)          water = wkl(1,lay)/coldry(lay)
   186)          scalefac = pavel(lay) * stpfac / tavel(lay)
   187) 
   188) ! If the pressure is less than ~100mb, perform a different
   189) ! set of species interpolations.
   190) 
   191)          if (plog .le. 4.56_r8) go to 5300
   192)          laytrop =  laytrop + 1
   193)          if (plog .ge. 6.62_r8) laylow = laylow + 1
   194) 
   195) ! Set up factors needed to separately include the water vapor
   196) ! foreign-continuum in the calculation of absorption coefficient.
   197) 
   198)          forfac(lay) = scalefac / (1.+water)
   199)          factor = (332.0_r8-tavel(lay))/36.0_r8
   200)          indfor(lay) = min(2, max(1, int(factor)))
   201)          forfrac(lay) = factor - float(indfor(lay))
   202) 
   203) ! Set up factors needed to separately include the water vapor
   204) ! self-continuum in the calculation of absorption coefficient.
   205) 
   206)          selffac(lay) = water * forfac(lay)
   207)          factor = (tavel(lay)-188.0_r8)/7.2_r8
   208)          indself(lay) = min(9, max(1, int(factor)-7))
   209)          selffrac(lay) = factor - float(indself(lay) + 7)
   210) 
   211) ! Calculate needed column amounts.
   212) 
   213)          colh2o(lay) = 1.e-20_r8 * wkl(1,lay)
   214)          colco2(lay) = 1.e-20_r8 * wkl(2,lay)
   215)          colo3(lay) = 1.e-20_r8 * wkl(3,lay)
   216) !           colo3(lay) = 0._r8
   217) !           colo3(lay) = colo3(lay)/1.16_r8
   218)          coln2o(lay) = 1.e-20_r8 * wkl(4,lay)
   219)          colch4(lay) = 1.e-20_r8 * wkl(6,lay)
   220)          colo2(lay) = 1.e-20_r8 * wkl(7,lay)
   221)          colmol(lay) = 1.e-20_r8 * coldry(lay) + colh2o(lay)
   222) !           colco2(lay) = 0._r8
   223) !           colo3(lay) = 0._r8
   224) !           coln2o(lay) = 0._r8
   225) !           colch4(lay) = 0._r8
   226) !           colo2(lay) = 0._r8
   227) !           colmol(lay) = 0._r8
   228)          if (colco2(lay) .eq. 0._r8) colco2(lay) = 1.e-32_r8 * coldry(lay)
   229)          if (coln2o(lay) .eq. 0._r8) coln2o(lay) = 1.e-32_r8 * coldry(lay)
   230)          if (colch4(lay) .eq. 0._r8) colch4(lay) = 1.e-32_r8 * coldry(lay)
   231)          if (colo2(lay) .eq. 0._r8) colo2(lay) = 1.e-32_r8 * coldry(lay)
   232) ! Using E = 1334.2 cm-1.
   233)          co2reg = 3.55e-24_r8 * coldry(lay)
   234)          co2mult(lay)= (colco2(lay) - co2reg) * &
   235)                272.63_r8*exp(-1919.4_r8/tavel(lay))/(8.7604e-4_r8*tavel(lay))
   236)          goto 5400
   237) 
   238) ! Above laytrop.
   239)  5300    continue
   240) 
   241) ! Set up factors needed to separately include the water vapor
   242) ! foreign-continuum in the calculation of absorption coefficient.
   243) 
   244)          forfac(lay) = scalefac / (1.+water)
   245)          factor = (tavel(lay)-188.0_r8)/36.0_r8
   246)          indfor(lay) = 3
   247)          forfrac(lay) = factor - 1.0_r8
   248) 
   249) ! Calculate needed column amounts.
   250) 
   251)          colh2o(lay) = 1.e-20_r8 * wkl(1,lay)
   252)          colco2(lay) = 1.e-20_r8 * wkl(2,lay)
   253)          colo3(lay)  = 1.e-20_r8 * wkl(3,lay)
   254)          coln2o(lay) = 1.e-20_r8 * wkl(4,lay)
   255)          colch4(lay) = 1.e-20_r8 * wkl(6,lay)
   256)          colo2(lay)  = 1.e-20_r8 * wkl(7,lay)
   257)          colmol(lay) = 1.e-20_r8 * coldry(lay) + colh2o(lay)
   258)          if (colco2(lay) .eq. 0._r8) colco2(lay) = 1.e-32_r8 * coldry(lay)
   259)          if (coln2o(lay) .eq. 0._r8) coln2o(lay) = 1.e-32_r8 * coldry(lay)
   260)          if (colch4(lay) .eq. 0._r8) colch4(lay) = 1.e-32_r8 * coldry(lay)
   261)          if (colo2(lay)  .eq. 0._r8) colo2(lay)  = 1.e-32_r8 * coldry(lay)
   262)          co2reg = 3.55e-24_r8 * coldry(lay)
   263)          co2mult(lay)= (colco2(lay) - co2reg) * &
   264)                272.63_r8*exp(-1919.4_r8/tavel(lay))/(8.7604e-4_r8*tavel(lay))
   265) 
   266)          selffac(lay) = 0._r8
   267)          selffrac(lay)= 0._r8
   268)          indself(lay) = 0
   269) 
   270)  5400    continue
   271) 
   272) ! We have now isolated the layer ln pressure and temperature,
   273) ! between two reference pressures and two reference temperatures 
   274) ! (for each reference pressure).  We multiply the pressure 
   275) ! fraction FP with the appropriate temperature fractions to get 
   276) ! the factors that will be needed for the interpolation that yields
   277) ! the optical depths (performed in routines TAUGBn for band n).
   278) 
   279)          compfp = 1._r8 - fp
   280)          fac10(lay) = compfp * ft
   281)          fac00(lay) = compfp * (1._r8 - ft)
   282)          fac11(lay) = fp * ft1
   283)          fac01(lay) = fp * (1._r8 - ft1)
   284) 
   285) ! End layer loop
   286)       enddo
   287) 
   288)       end subroutine setcoef_sw
