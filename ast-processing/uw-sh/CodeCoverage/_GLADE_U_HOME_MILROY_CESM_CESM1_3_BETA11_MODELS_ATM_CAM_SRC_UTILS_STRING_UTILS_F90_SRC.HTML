    91) integer function INCSTR( s, inc )
    92)   !-----------------------------------------------------------------------
    93)   ! 	... Increment a string whose ending characters are digits.
    94)   !           The incremented integer must be in the range [0 - (10**n)-1]
    95)   !           where n is the number of trailing digits.
    96)   !           Return values:
    97)   !
    98)   !            0 success
    99)   !           -1 error: no trailing digits in string
   100)   !           -2 error: incremented integer is out of range
   101)   !-----------------------------------------------------------------------
   102) 
   103)   implicit none
   104) 
   105)   !-----------------------------------------------------------------------
   106)   ! 	... Dummy variables
   107)   !-----------------------------------------------------------------------
   108)   integer, intent(in) :: &
   109)        inc                                       ! value to increment string (may be negative)
   110)   character(len=*), intent(inout) :: &
   111)        s                                         ! string with trailing digits
   112) 
   113) 
   114)   !-----------------------------------------------------------------------
   115)   ! 	... Local variables
   116)   !-----------------------------------------------------------------------
   117)   integer :: &
   118)        i, &                          ! index
   119)        lstr, &                       ! number of significant characters in string
   120)        lnd, &                        ! position of last non-digit
   121)        ndigit, &                     ! number of trailing digits
   122)        ival, &                       ! integer value of trailing digits
   123)        pow, &                        ! power of ten
   124)        digit                         ! integer value of a single digit
   125) 
   126)   lstr   = GLC( s )
   127)   lnd    = LASTND( s )
   128)   ndigit = lstr - lnd
   129) 
   130)   if( ndigit == 0 ) then
   131)      INCSTR = -1
   132)      return
   133)   end if
   134) 
   135)   !-----------------------------------------------------------------------
   136)   !     	... Calculate integer corresponding to trailing digits.
   137)   !-----------------------------------------------------------------------
   138)   ival = 0
   139)   pow  = 0
   140)   do i = lstr,lnd+1,-1
   141)      digit = ICHAR(s(i:i)) - ICHAR('0')
   142)      ival  = ival + digit * 10**pow
   143)      pow   = pow + 1
   144)   end do
   145) 
   146)   !-----------------------------------------------------------------------
   147)   !     	... Increment the integer
   148)   !-----------------------------------------------------------------------
   149)   ival = ival + inc
   150)   if( ival < 0 .or. ival > 10**ndigit-1 ) then
   151)      INCSTR = -2
   152)      return
   153)   end if
   154) 
   155)   !-----------------------------------------------------------------------
   156)   !     	... Overwrite trailing digits
   157)   !-----------------------------------------------------------------------
   158)   pow = ndigit
   159)   do i = lnd+1,lstr
   160)      digit  = MOD( ival,10**pow ) / 10**(pow-1)
   161)      s(i:i) = CHAR( ICHAR('0') + digit )
   162)      pow    = pow - 1
   163)   end do
   164) 
   165)   INCSTR = 0
   166) 
   167) end function INCSTR

   172)   ! 	    Return values:

   176)

   178) 
   179)   !-----------------------------------------------------------------------
   180)   ! 	... Dummy arguments

   182)   character(len=*), intent(in) :: cs       !  Input character string
   183)

   189)   n = GLC( cs )
   190)   if( n == 0 ) then     ! empty string
   191)      LASTND = 0
   192)      return
   193)   end if
   194) 
   195)   do nn = n,1,-1

   197)      if( digit < 0 .or. digit > 9 ) then
