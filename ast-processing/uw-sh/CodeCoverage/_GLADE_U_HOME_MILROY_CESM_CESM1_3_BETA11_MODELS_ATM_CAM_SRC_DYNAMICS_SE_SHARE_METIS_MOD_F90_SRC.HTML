    22)     use gridgraph_mod, only : GridVertex_t, GridEdge_t, freegraph, createsubgridgraph, printgridvertex

    31)

    33)     type (GridEdge_t),   intent(inout) :: GridEdge(:)
    34) 
    35)     integer , target, allocatable :: xadj(:),adjncy(:)    ! Adjacency structure for METIS
    36)     integer , target, allocatable :: vwgt(:),adjwgt(:)    ! Weights for the adj struct for METIS
    37) 
    38)     integer , target, allocatable :: xadj_nl(:),adjncy_nl(:)
    39)     integer , target, allocatable :: vwgt_nl(:),adjwgt_nl(:)

    47)     integer, allocatable          :: part_nl(:),local2global_nl(:)
    48)     integer, allocatable          :: part_fl(:),local2global_fl(:)
    49) 
    50) 
    51)     integer, allocatable          :: cnt(:),newnum(:),oldnum(:)
    52) 
    53)     integer                       :: nelem_edge,numflag,edgecut,wgtflag
    54)     integer                       :: head_part,tail_part
    55)     integer                       :: options(5)

    57)     integer                       :: nelem_nl,nelem_fl,newPartition

    59)     integer                       :: nodes_per_frame
    60)     logical , parameter           :: Debug = .FALSE.
    61)     real (kind=4)                 :: dummy(1)

    63) 
    64)     allocate(tpwgts(npart))
    65)     allocate(part(nelem))
    66)     allocate(xadj(nelem+1))
    67)     allocate(vwgt(nelem))
    68)     allocate(adjncy(nelem_edge))
    69)     allocate(adjwgt(nelem_edge))
    70) 
    71)     if(Debug) write(iulog,*)'genmetispart: point #1'
    72)     ! =============================================
    73)     !   Generate Graph for METIS
    74)     ! =============================================
    75)     !DBG  call PrintGridVertex(GridVertex)
    76)     call CreateMetisGraph(GridVertex,xadj,adjncy,adjwgt)
    77)     ! Add weights to all the vertices
    78)     vwgt(:)=VertexWeight
    79)     if(Debug) write(iulog,*)'genmetispart: point #2'
    80) 
    81)     ! ====================================================
    82)     ! Some cruff from the Weighted partitioning 
    83)     ! experimentation... Remove soon 
    84)     ! ====================================================
    85)     do i=1,npart
    86)        tpwgts(i) = i
    87)     enddo
    88)     tmp = SUM(tpwgts)
    89)     tpwgts = tpwgts/tmp
    90)     if(Debug) write(iulog,*)'genmetispart: point #3'
    91) 
    92)     !================================================
    93)     !  Setup some options for the METIS partitioning
    94)     !================================================
    95)     options(1) = 0     ! Use Default METIS options
    96)     !  wgtflag    = 1     ! Weights on the Edges only
    97)     wgtflag    = 3     ! Weights on the edges and vertices
    98)     numflag    = 1     ! Use Fortran based numbering
    99) 
   100)     if(npart > 1) then 
   101) 
   102)        !     PartitionForFrames=.FALSE.
   103)        if(PartitionForFrames) then 
   104)           numPartitions   = numframes
   105)           PartitionMethod = WRECURSIVE 
   106)        elseif (PartitionForNodes) then 
   107)           numPartitions   = nnodes
   108)           PartitionMethod = partmethod 
   109)        else
   110)           numPartitions   = npart
   111)           PartitionMethod = partmethod 
   112)        endif
   113)        !===========================================
   114)        !  Generate the METIS partitioning
   115)        !===========================================
   116)        if(iam .eq. 1) write(6,100) nelem,PartitionMethod,numPartitions
   117)        if(Debug) write(iulog,*)'genmetispart: point #4'
   118)        if (PartitionMethod==WRECURSIVE) then

   120)                vwgt,adjwgt,wgtflag,numflag, &
   121)                numPartitions,FrameWeight,options,edgecut,part)
   122)        else
   123)           ! FrameWeight has not been allocated in this case, so replace
   124)           ! with dummy argument: 
   125)           call PartitionGraph(PartitionMethod,nelem,xadj,adjncy, &
   126)                vwgt,adjwgt,wgtflag,numflag, &
   127)                numPartitions,dummy,options,edgecut,part)

   131) 
   132)           ! =================================================================

   134)           ! =================================================================

   136) 
   137)           allocate(cnt(numframes))
   138)           allocate(oldnum(numframes))
   139)           allocate(Newnum(numframes))
   140) 
   141)           ii = 1
   142)           do i=1,SIZE(FrameCount)
   143)              if(FrameCount(i) .ne. 0) then 
   144)                 cnt(ii) = FrameCount(i)
   145)                 oldnum(ii) = ii
   146)                 ii = ii+1
   147)              endif
   148)           enddo
   149) 
   150)           if(Debug) write(iulog,*)'genmetispart: point #6'
   151)           call NewPartitionNumber(newnum,oldnum,cnt)
   152)           if(iam .eq. 1) write(iulog,*)'genmetispart: After NewPartitionNumber cnt    : ',cnt
   153)           if(iam .eq. 1) write(iulog,*)'genmetispart: After NewPartitionNumber oldnum : ',oldnum
   154)           if(iam .eq. 1) write(iulog,*)'genmetispart: After NewPartitionNumber newnum : ',newnum

   157)              part(ip)  = NewNum(part(ip))
   158)           enddo
   159) 
   160)           if(iam .eq. 1) write(iulog,*)'genmetispart: ForFrames: After reassignment  : ',part 
   161) 
   162)           if(Debug) write(iulog,*)'genmetispart: point #7'
   163)           do if = 1,numframes
   164) 
   165) 
   166)              ! =======================================
   167)              !  Figure out the new partitioning number 
   168)              ! =======================================
   169)              newPartition = NewNum(if)
   170) 
   171)              nelem_fl = COUNT(part .eq. newPartition)
   172) 
   173)              ! ===================================================
   174)              ! Allocate all the memory for frame level partitioning 
   175)              ! ===================================================
   176) 
   177)              allocate(local2global_fl(nelem_fl))
   178)              allocate(part_fl(nelem_fl))

   180)              if(Debug) write(iulog,*)'genmetispart: point #8'
   181) 
   182)              ! =====================================
   183)              !   Setup the index translation arrays
   184)              ! =====================================
   185)              call genLocal2Global(local2global_fl,part,NewPartition)
   186)              if(Debug) write(iulog,*)'genmetispart: point #9'
   187) 
   188)              ! ======================================================
   189)              ! Create a set of the Vertices that represent a subgraph

   191)              call CreateSubGridGraph(GridVertex,SubVertex,local2global_fl)

   195)              !  Convert graph to Metis format

   199) 
   200)              ! =======================
   201)              ! Partition the subgraph 
   202)              ! =======================
   203)              nodes_per_frame = cnt(if)
   204)              if(iam .eq. 1) write(6,100) nelem_fl,partmethod,nodes_per_frame
   205)              if(nodes_per_frame .gt. 1) then

   207)                      vwgt,adjwgt,wgtflag,numflag, &
   208)                      nodes_per_frame,tpwgts,options,edgecut,part_fl)
   209)              else
   210)                 part_fl(:) = 1
   211)              endif
   212)              if(Debug) write(iulog,*)'genmetispart: point #12'
   213) 
   214)              ! ========================================================= 
   215)              ! Apply the Frame partitioning the the overall partitioning 
   216)              ! ========================================================= 
   217)              do i=1,nelem_fl
   218)                 ig = local2global_fl(i)
   219)                 part(ig) = part(ig) + (part_fl(i)-1)
   220)              enddo
   221) 
   222)              ! ======================================
   223)              ! Free up the temporaries that were used
   224)              ! ======================================
   225)              deallocate(local2global_fl)
   226)              deallocate(part_fl)
   227) 
   228) 
   229) 	     call FreeGraph(SubVertex)
   230)              deallocate(SubVertex)

   232)           enddo
   233)           if(Debug) write(iulog,*)'genmetispart: point #13'
   234) 
   235)           deallocate(newnum)
   236)           deallocate(oldnum)
   237)           deallocate(cnt)
   238) 
   239)           if(iam .eq. 1) write(iulog,*)'genmetispart: Partitioning after Frame partitioning: ',part
   240)           !JMD	call haltmp('genmetispart: After Frame based partitioning:')

   242)        ! ===============================
   243)        !  Do not partitiion for nodes 
   244)        ! ===============================
   245)        !     PartitionForNodes=.FALSE.
   246)        if(PartitionForNodes)  then 
   247)           ! =========================================
   248)           ! Partition the graph on each compute node

   250)           if(nmpi_per_node .ne. 1 ) then

   255) 
   256)              if(Debug) write(iulog,*)'genmetispart: point #14'

   261)              do in = 1,nnodes
   262) 
   263)                 newPartition = nmpi_per_node*(in -1) + 1
   264)                 nelem_nl = COUNT(part .eq. newPartition)

   266)                 ! ===================================================
   267)                 ! Allocate all the memory for node level partitioning 
   268)                 ! ===================================================
   269) 
   270)                 allocate(local2global_nl(nelem_nl))
   271)                 allocate(part_nl(nelem_nl))
   272) 
   273)                 if(Debug) write(iulog,*)'genmetispart: point #15'

   277)                 allocate(adjwgt_nl(8*nelem_nl))
   278)                 adjncy_nl(:) = 0
   279)                 adjwgt_nl(:) = 0
   280)                 part_nl(:)   = 0
   281) 
   282)                 if(Debug) write(iulog,*)'genmetispart: point #16'
   283)                 !Add vertex weights to the subgraphs
   284)                 vwgt_nl(:) = VertexWeight
   285)                 allocate(SubVertex(nelem_nl))
   286) 
   287)                 ! =====================================
   288)                 !   Setup the index translation arrays
   289)                 ! =====================================
   290)                 call genLocal2Global(local2global_nl,part,newPartition)
   291) 
   292)                 if(Debug) write(iulog,*)'genmetispart: point #17'
   293)                 ! ======================================================
   294)                 ! Create a set of the Vertices that represent a subgraph 
   295)                 ! ======================================================
   296)                 if(Debug) write(iulog,*)'genmetispart: local2global_nl',local2global_nl
   297)                 if(Debug) call PrintGridVertex(GridVertex)
   298)                 call CreateSubGridGraph(GridVertex,SubVertex,local2global_nl)
   299)                 !JMD call CheckGridNeighbors(SubVertex)
   300)                 if(Debug) call PrintGridVertex(SubVertex)
   301)                 if(Debug) write(iulog,*)'genmetispart: point #18'
   302) 
   303)                 ! ==============================
   304)                 !  Convert grep to Metis format
   305)                 ! ==============================
   306)                 call CreateMetisGraph(SubVertex,xadj_nl,adjncy_nl,adjwgt_nl)	
   307) 
   308)                 !debug	     call PrintMetisgraph(xadj_nl,adjncy_nl,adjwgt_nl)
   309)                 ! =======================
   310)                 ! Partition the subgraph 
   311)                 ! =======================
   312)                 if(iam .eq. 1) write(6,100) nelem_nl,partmethod,nmpi_per_node
   313)                 call PartitionGraph(partmethod,nelem_nl,xadj_nl,adjncy_nl, &
   314)                      vwgt_nl,adjwgt_nl,wgtflag,numflag, &
   315)                      nmpi_per_node,tpwgts,options,edgecut,part_nl)
   316)                 if(Debug) write(iulog,*)'genmetispart: point #19'
   317) 
   318)                 ! ========================================================= 
   319)                 ! Apply the node partitioning the the overall partitioning 
   320)                 ! ========================================================= 
   321)                 do i=1,nelem_nl
   322)                    ig = local2global_nl(i)
   323)                    part(ig) = part(ig) + (part_nl(i)-1)
   324)                 enddo
   325) 
   326)                 ! ======================================
   327)                 ! Free up the temporaries that were used
   328)                 ! ======================================
   329)                 deallocate(local2global_nl)
   330)                 deallocate(part_nl)
   331)                 deallocate(xadj_nl)
   332)                 deallocate(vwgt_nl)
   333)                 deallocate(adjncy_nl)
   334)                 deallocate(adjwgt_nl)
   335) 
   336) 
   337)                 if(Debug) write(iulog,*)'genmetispart: point #20'
   338)                 call FreeGraph(SubVertex)
   339)                 deallocate(SubVertex)
   340)                 if(Debug) write(iulog,*)'genmetispart: point #21'
   341) 
   342)              enddo
   343) 
   344) 
   345)           endif  ! if node based partitioning 
   346) 
   347)        endif   ! if multilevel 
   348) 
   349)     else   ! else no partitioning needed
   350)        !======================================================================
   351)        ! It appears that Metis will set part(:) = 2 if nnodes == 1, 
   352)        !	which messes stuff up, so set it myself
   353)        !======================================================================
   354)        part(:)=1 
   355)     endif
   356) 
   357)     if(Debug) write(iulog,*)'genmetispart: point #22'
   358)     ! ===========================================
   359)     !  Add the partitioning information into the 
   360)     !    Grid Vertex and Grid Edge structures
   361)     ! ===========================================

   364)     if(Debug) write(iulog,*)'genmetispart: point #23'
   365)     !=================================================
   366)     !  Output the partitioning information 
   367)     !=================================================
   368) #if 0
   369) 
   370)     write(iulog,*)'Metis Parititioning: '
   371)     write(iulog,*)part
   372) 
   373)     write(iulog,*)'Metis Parititioning: '
   374)     do k=1,nelem
   375)        write(iulog,*) k,part(k)
   376)     enddo
   377)     call abortmp(' at the end of genmetispart')
   378) 
   379)     if(OutputFiles) then
   380)        do k=1,nelem
   381)           write(10,*) GridVertex(k)%processor_number
   382)        end do
   383)        close(10)
   384)     end if
   385) #endif
   386) 
   387) 100 format("genmetispart:  Partitioning ",i4," elements, using method: ",i2," into ",i4," pieces")
   388) 
   389)   end subroutine genmetispart

   465)   subroutine CreateMetisGraph(GridVertex,xadj,adjncy,adjwgt)
   466)     use gridgraph_mod, only : GridVertex_t, num_neighbors
   467)     use kinds, only : real_kind, int_kind
   468)     type (GridVertex_t), intent(in) :: GridVertex(:)
   469)     integer,intent(inout)           :: xadj(:),adjncy(:),adjwgt(:)
   470) 
   471)     integer                         :: i,j,k,ii,jj
   472)     integer                         :: degree,nelem
   473)     integer(kind=int_kind),allocatable  :: neigh_list(:),index(:)
   474)     real(kind=REAL_KIND),allocatable    :: neigh_wgt(:)
   475)     integer                         :: max_neigh
   476) 
   477)     integer :: start, cnt
   478) 
   479)     nelem = SIZE(GridVertex)
   480) 
   481)     degree = 0
   482)     ii     = 1
   483) 
   484)     max_neigh = num_neighbors
   485)     allocate(index(max_neigh))
   486)     allocate(neigh_list(max_neigh))
   487)     allocate(neigh_wgt(max_neigh))
   488) 
   489)     do i=1,nelem
   490)        xadj(i)      = ii
   491)        jj=1
   492)        neigh_list=0
   493) 
   494) 
   495)        do j=1,num_neighbors
   496)           cnt = GridVertex(i)%nbrs_ptr(j+1) -  GridVertex(i)%nbrs_ptr(j) 
   497)           start =  GridVertex(i)%nbrs_ptr(j) 
   498)           do k=0, cnt-1
   499)              if(GridVertex(i)%nbrs_wgt(start+k) .gt. 0) then 
   500)                 adjncy(ii+jj-1)   = GridVertex(i)%nbrs(start+k)
   501) 	        neigh_list(jj)    = GridVertex(i)%nbrs(start+k)
   502)                 adjwgt(ii+jj-1)   = GridVertex(i)%nbrs_wgt(start+k)*EdgeWeight
   503)                 neigh_wgt(jj)     = GridVertex(i)%nbrs_wgt(start+k)*EdgeWeight
   504)                 jj=jj+1
   505)              endif
   506)           enddo
   507)        enddo
   508)        if (max_neigh < jj+1) call abortmp( "number or neighbors foudn exceeds expected max error")
   509) 
   510)        call sort(max_neigh,neigh_list,index)
   511)        degree       = COUNT(GridVertex(i)%nbrs_wgt(:) .gt. 0) 
   512)        ! Copy the sorted adjncy list in
   513)        do j=1,degree
   514)           adjncy(ii+j-1) = neigh_list(index(j))
   515)           adjwgt(ii+j-1) = neigh_wgt(index(j))
   516)        enddo
   517)        ii           = ii + degree
   518)     enddo
   519)     xadj(nelem+1)     = ii
   520) 
   521)     deallocate(neigh_list)
   522)     deallocate(neigh_wgt)
   523)     deallocate(index)
   524) 
   525)   end subroutine CreateMetisGraph
   526)   subroutine sort(n,list,index)
   527)     use kinds, only : int_kind
   528)     implicit none
   529)     integer, intent(in) :: n
   530)     integer(kind=int_kind), intent(in) :: list(n)
   531)     integer(kind=int_kind), intent(inout) :: index(n)
   532) 
   533)     ! Local variables
   534)     integer :: i,iloc,nct,ii,lmax,lmin
   535)     logical :: msk(n)
   536)     logical,parameter  :: Debug =.FALSE.
   537) 
   538)     msk=.TRUE.
   539)     if(Debug) write(iulog,*)'sort: point #1'
   540)     nct=0
   541)     do i=1,n
   542)        if(list(i) .eq. 0) then 
   543)           msk(i)=.FALSE.
   544)        else
   545)           nct = nct + 1
   546)        endif
   547)     enddo
   548)     if(Debug) write(iulog,*)'sort: point #2   list: ',list
   549)     if(Debug) write(iulog,*)'sort: point #2.1 msk: ',msk
   550) 
   551)     lmax=maxval(list)
   552)     do i=1,nct
   553)        if(Debug) write(iulog,*)'sort: point #3'
   554) 
   555)        !     pgf90 Rel 3.1-4i:  minloc() with mask is buggy   
   556)        !     iloc = minloc(list,dim=1,mask=msk)
   557)        lmin=lmax
