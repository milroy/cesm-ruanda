    76)   subroutine convect_shallow_register
    77) 
    78)   !-------------------------------------------------- !
    79)   ! Purpose : Register fields with the physics buffer !
    80)   !-------------------------------------------------- !
    81) 
    82)   use physics_buffer, only : pbuf_add_field, dtype_r8, dyn_time_lvls
    83)   use unicon_cam,     only: unicon_cam_register
    84)   
    85)   call phys_getopts( shallow_scheme_out = shallow_scheme, microp_scheme_out = microp_scheme)
    86)                      
    87) 
    88)   call pbuf_add_field('ICWMRSH',    'physpkg' ,dtype_r8,(/pcols,pver/),       icwmrsh_idx )
    89)   call pbuf_add_field('RPRDSH',     'physpkg' ,dtype_r8,(/pcols,pver/),       rprdsh_idx )
    90)   call pbuf_add_field('RPRDTOT',    'physpkg' ,dtype_r8,(/pcols,pver/),       rprdtot_idx )
    91)   call pbuf_add_field('CLDTOP',     'physpkg' ,dtype_r8,(/pcols,1/),          cldtop_idx )
    92)   call pbuf_add_field('CLDBOT',     'physpkg' ,dtype_r8,(/pcols,1/),          cldbot_idx )
    93)   call pbuf_add_field('cush',       'global'  ,dtype_r8,(/pcols,dyn_time_lvls/), cush_idx ) 	
    94)   call pbuf_add_field('NEVAPR_SHCU','physpkg' ,dtype_r8,(/pcols,pver/),       nevapr_shcu_idx )
    95)   call pbuf_add_field('PREC_SH',    'physpkg' ,dtype_r8,(/pcols/),            prec_sh_idx )
    96)   call pbuf_add_field('SNOW_SH',    'physpkg' ,dtype_r8,(/pcols/),            snow_sh_idx )
    97) 
    98)   if (shallow_scheme .eq. 'UW' .or. shallow_scheme .eq. 'UNICON') then
    99)      call pbuf_add_field('shfrc', 'physpkg', dtype_r8, (/pcols,pver/), shfrc_idx)
   100)   end if
   101) 
   102) ! shallow interface gbm flux_convective_cloud_rain+snow (kg/m2/s)
   103)   call pbuf_add_field('SH_FLXPRC','physpkg',dtype_r8,(/pcols,pverp/),sh_flxprc_idx)  
   104) 
   105) ! shallow interface gbm flux_convective_cloud_snow (kg/m2/s)
   106)   call pbuf_add_field('SH_FLXSNW','physpkg',dtype_r8,(/pcols,pverp/),sh_flxsnw_idx)  
   107) 
   108) ! shallow gbm cloud liquid water (kg/kg)
   109)   call pbuf_add_field('SH_CLDLIQ','physpkg',dtype_r8,(/pcols,pver/),sh_cldliq_idx)  
   110) 
   111) ! shallow gbm cloud ice water (kg/kg)
   112)   call pbuf_add_field('SH_CLDICE','physpkg',dtype_r8,(/pcols,pver/),sh_cldice_idx)  
   113) 
   114)   if (shallow_scheme .eq. 'UNICON') then
   115)      call unicon_cam_register()
   116)   end if
   117) 
   118)   end subroutine convect_shallow_register

   125)   subroutine convect_shallow_init(pref_edge, pbuf2d)
   126) 
   127)   !------------------------------------------------------------------------------- !
   128)   ! Purpose : Declare output fields, and initialize variables needed by convection !
   129)   !------------------------------------------------------------------------------- !
   130) 
   131)   use cam_history,       only : addfld, add_default, phys_decomp
   132)   use ppgrid,            only : pcols, pver
   133)   use hk_conv,           only : mfinti
   134)   use uwshcu,            only : init_uwshcu
   135)   use unicon_cam,        only : unicon_cam_init
   136)   use physconst,         only : rair, gravit, latvap, rhoh2o, zvir, &
   137)                                 cappa, latice, mwdry, mwh2o
   138)   use pmgrid,            only : plev, plevp
   139)   use spmd_utils,        only : masterproc
   140)   use abortutils,        only : endrun
   141)   use phys_control,      only : cam_physpkg_is
   142)   
   143)   use physics_buffer,    only : pbuf_get_index, physics_buffer_desc, pbuf_set_field
   144)   use time_manager,      only : is_first_step
   145) 
   146)   real(r8),                  intent(in) :: pref_edge(plevp)  ! Reference pressures at interfaces
   147)   type(physics_buffer_desc), pointer    :: pbuf2d(:,:)
   148) 
   149)   integer limcnv                                   ! Top interface level limit for convection
   150)   integer k
   151)   character(len=16)          :: eddy_scheme
   152)     
   153)   ! ------------------------------------------------- !
   154)   ! Variables for detailed abalysis of UW-ShCu scheme !
   155)   ! ------------------------------------------------- !
   156) 
   157)   call addfld( 'qt_pre_Cu    ', 'kg/kg'   ,  pver ,  'I' , 'qt_preCU'                                         ,  phys_decomp )
   158)   call addfld( 'sl_pre_Cu    ', 'J/kg'    ,  pver ,  'I' , 'sl_preCU'                                         ,  phys_decomp )
   159)   call addfld( 'slv_pre_Cu   ', 'J/kg'    ,  pver ,  'I' , 'slv_preCU'                                        ,  phys_decomp )
   160)   call addfld( 'u_pre_Cu     ', 'm/s'     ,  pver ,  'I' , 'u_preCU'                                          ,  phys_decomp )
   161)   call addfld( 'v_pre_Cu     ', 'm/s'     ,  pver ,  'I' , 'v_preCU'                                          ,  phys_decomp )
   162)   call addfld( 'qv_pre_Cu    ', 'kg/kg'   ,  pver ,  'I' , 'qv_preCU'                                         ,  phys_decomp )
   163)   call addfld( 'ql_pre_Cu    ', 'kg/kg'   ,  pver ,  'I' , 'ql_preCU'                                         ,  phys_decomp )
   164)   call addfld( 'qi_pre_Cu    ', 'kg/kg'   ,  pver ,  'I' , 'qi_preCU'                                         ,  phys_decomp )
   165)   call addfld( 't_pre_Cu     ', 'K'       ,  pver ,  'I' , 't_preCU'                                          ,  phys_decomp )
   166)   call addfld( 'rh_pre_Cu    ', '%'       ,  pver ,  'I' , 'rh_preCU'                                         ,  phys_decomp )
   167) 
   168)   call addfld( 'qt_aft_Cu    ', 'kg/kg'   ,  pver ,  'I' , 'qt_afterCU'                                       ,  phys_decomp )
   169)   call addfld( 'sl_aft_Cu    ', 'J/kg'    ,  pver ,  'I' , 'sl_afterCU'                                       ,  phys_decomp )
   170)   call addfld( 'slv_aft_Cu   ', 'J/kg'    ,  pver ,  'I' , 'slv_afterCU'                                      ,  phys_decomp )
   171)   call addfld( 'u_aft_Cu     ', 'm/s'     ,  pver ,  'I' , 'u_afterCU'                                        ,  phys_decomp )
   172)   call addfld( 'v_aft_Cu     ', 'm/s'     ,  pver ,  'I' , 'v_afterCU'                                        ,  phys_decomp )
   173)   call addfld( 'qv_aft_Cu    ', 'kg/kg'   ,  pver ,  'I' , 'qv_afterCU'                                       ,  phys_decomp )
   174)   call addfld( 'ql_aft_Cu    ', 'kg/kg'   ,  pver ,  'I' , 'ql_afterCU'                                       ,  phys_decomp )
   175)   call addfld( 'qi_aft_Cu    ', 'kg/kg'   ,  pver ,  'I' , 'qi_afterCU'                                       ,  phys_decomp )
   176)   call addfld( 't_aft_Cu     ', 'K'       ,  pver ,  'I' , 't_afterCU'                                        ,  phys_decomp )
   177)   call addfld( 'rh_aft_Cu    ', '%'       ,  pver ,  'I' , 'rh_afterCU'                                       ,  phys_decomp )
   178) 
   179)   call addfld( 'tten_Cu      ', 'K/s'     ,  pver ,  'I' , 'Temprtaure tendency by cumulus convection'        ,  phys_decomp )
   180)   call addfld( 'rhten_Cu     ', '%/s'     ,  pver ,  'I' , 'RH tendency by cumumus convection'                ,  phys_decomp )
   181) 
   182)   ! ------------------------------------------- !
   183)   ! Common Output for Shallow Convection Scheme !
   184)   ! ------------------------------------------- !
   185) 
   186)   call addfld( 'CMFDT   '     , 'K/s     ',  pver ,  'A' , &
   187)        'T tendency - shallow convection'                           ,  phys_decomp )
   188)   call addfld( 'CMFDQ   '     , 'kg/kg/s ',  pver ,  'A' , &
   189)        'QV tendency - shallow convection'                          ,  phys_decomp )
   190)   call addfld( 'CMFDLIQ '     , 'kg/kg/s ',  pver ,  'A' , &
   191)        'Cloud liq tendency - shallow convection'                   ,  phys_decomp )
   192)   call addfld( 'CMFDICE '     , 'kg/kg/s ',  pver ,  'A' , &
   193)        'Cloud ice tendency - shallow convection'                   ,  phys_decomp )
   194)   call addfld( 'CMFDQR  '     , 'kg/kg/s ',  pver ,  'A' , &
   195)        'Q tendency - shallow convection rainout'                   ,  phys_decomp )
   196)   call addfld( 'EVAPTCM '     , 'K/s     ',  pver ,  'A' , &
   197)        'T tendency - Evaporation/snow prod from Hack convection'   ,  phys_decomp )
   198)   call addfld( 'FZSNTCM '     , 'K/s     ',  pver ,  'A' , &
   199)        'T tendency - Rain to snow conversion from Hack convection' ,  phys_decomp )
   200)   call addfld( 'EVSNTCM '     , 'K/s     ',  pver ,  'A' , &
   201)        'T tendency - Snow to rain prod from Hack convection'       ,  phys_decomp )
   202)   call addfld( 'EVAPQCM '     , 'kg/kg/s ',  pver ,  'A' , &
   203)        'Q tendency - Evaporation from Hack convection'             ,  phys_decomp )
   204)   call addfld( 'QC      '     , 'kg/kg/s ',  pver ,  'A' , &
   205)        'Q tendency - shallow convection LW export'                 ,  phys_decomp )
   206)   call addfld( 'PRECSH  '     , 'm/s     ',  1,      'A' , &
   207)        'Shallow Convection precipitation rate'                     ,  phys_decomp )
   208)   call addfld( 'CMFMC   '     , 'kg/m2/s ',  pverp,  'A' , &
   209)        'Moist convection (deep+shallow) mass flux'                 ,  phys_decomp )
   210)   call addfld( 'CMFSL   '     , 'W/m2    ',  pverp,  'A' , &
   211)        'Moist shallow convection liquid water static energy flux'  ,  phys_decomp )
   212)   call addfld( 'CMFLQ   '     , 'W/m2    ',  pverp,  'A' , &
   213)        'Moist shallow convection total water flux'                 ,  phys_decomp )
   214)   call addfld( 'CIN     '     , 'J/kg    ',  1    ,  'A' , &
   215)        'Convective inhibition'                                     ,  phys_decomp )
   216)   call addfld( 'CBMF    '     , 'kg/m2/s ',  1    ,  'A' , &
   217)        'Cloud base mass flux'                                      ,  phys_decomp )
   218)   call addfld( 'CLDTOP  '     , '1       ',  1    ,  'I' , &
   219)        'Vertical index of cloud top'                               ,  phys_decomp )
   220)   call addfld( 'CLDBOT  '     , '1       ',  1    ,  'I' , &
   221)        'Vertical index of cloud base'                              ,  phys_decomp )
   222)   call addfld( 'PCLDTOP '     , '1       ',  1    ,  'A' , &
   223)        'Pressure of cloud top'                                     ,  phys_decomp )
   224)   call addfld( 'PCLDBOT '     , '1       ',  1    ,  'A' , &
   225)        'Pressure of cloud base'                                    ,  phys_decomp )
   226) 
   227)   call addfld( 'FREQSH '      , 'fraction',  1    ,  'A' , &
   228)        'Fractional occurance of shallow convection'                ,  phys_decomp )
   229)                                                                                                                     
   230)   call addfld( 'HKFLXPRC'     , 'kg/m2/s ',  pverp,  'A' , &
   231)        'Flux of precipitation from HK convection'                  ,  phys_decomp )
   232)   call addfld( 'HKFLXSNW'     , 'kg/m2/s ',  pverp,  'A' , &
   233)        'Flux of snow from HK convection'                           ,  phys_decomp )
   234)   call addfld( 'HKNTPRPD'     , 'kg/kg/s ',  pver ,  'A' , &
   235)        'Net precipitation production from HK convection'           ,  phys_decomp )
   236)   call addfld( 'HKNTSNPD'     , 'kg/kg/s ',  pver ,  'A' , &
   237)        'Net snow production from HK convection'                    ,  phys_decomp )
   238)   call addfld( 'HKEIHEAT'     , 'W/kg'    ,  pver ,  'A' , &
   239)        'Heating by ice and evaporation in HK convection'           ,  phys_decomp )
   240) 
   241)   call addfld ('ICWMRSH  '    , 'kg/kg   ',  pver,   'A' , &
   242)        'Shallow Convection in-cloud water mixing ratio '           ,  phys_decomp )
   243) 
   244)   if( shallow_scheme .eq. 'UW' ) then
   245)      call addfld( 'UWFLXPRC'     , 'kg/m2/s ',  pverp,  'A' , &
   246)           'Flux of precipitation from UW shallow convection'          ,  phys_decomp )
   247)      call addfld( 'UWFLXSNW'     , 'kg/m2/s ',  pverp,  'A' , &
   248)           'Flux of snow from UW shallow convection'                   ,  phys_decomp )
   249)   end if
   250) 
   251) 
   252) 
   253)   call phys_getopts( eddy_scheme_out = eddy_scheme      , &
   254)                      history_amwg_out = history_amwg    , &
   255)                      history_budget_out = history_budget, &
   256)                      history_budget_histfile_num_out = history_budget_histfile_num)
   257) 
   258) 
   259)   if( history_budget ) then
   260)       call add_default( 'CMFDLIQ  ', history_budget_histfile_num, ' ' )
   261)       call add_default( 'CMFDICE  ', history_budget_histfile_num, ' ' )
   262)       call add_default( 'CMFDT   ', history_budget_histfile_num, ' ' )
   263)       call add_default( 'CMFDQ   ', history_budget_histfile_num, ' ' )
   264)       if( cam_physpkg_is('cam3') .or. cam_physpkg_is('cam4') ) then
   265)          call add_default( 'EVAPQCM  ', history_budget_histfile_num, ' ' )
   266)          call add_default( 'EVAPTCM  ', history_budget_histfile_num, ' ' )
   267)       end if
   268)   end if
   269)   pblh_idx  = pbuf_get_index('pblh')
   270) 
   271) 
   272)   select case (shallow_scheme)
   273) 
   274)   case('off')  ! None
   275) 
   276)      if( masterproc ) write(iulog,*) 'convect_shallow_init: shallow convection OFF'
   277)      continue
   278) 
   279)   case('Hack') ! Hack scheme
   280) 
   281)      qpert_idx = pbuf_get_index('qpert')
   282) 
   283)      if( masterproc ) write(iulog,*) 'convect_shallow_init: Hack shallow convection'
   284)    ! Limit shallow convection to regions below 40 mb
   285)    ! Note this calculation is repeated in the deep convection interface
   286)      if( pref_edge(1) >= 4.e3_r8 ) then
   287)          limcnv = 1
   288)      else
   289)          do k = 1, plev
   290)             if( pref_edge(k) < 4.e3_r8 .and. pref_edge(k+1) >= 4.e3_r8 ) then
   291)                 limcnv = k
   292)                 goto 10
   293)             end if
   294)          end do
   295)          limcnv = plevp
   296)      end if
   297) 10   continue
   298) 
   299)      if( masterproc ) then
   300)          write(iulog,*) 'MFINTI: Convection will be capped at intfc ', limcnv, ' which is ', pref_edge(limcnv), ' pascals'
   301)      end if
   302)      
   303)      call mfinti( rair, cpair, gravit, latvap, rhoh2o, limcnv) ! Get args from inti.F90
   304) 
   305)   case('UW') ! Park and Bretherton shallow convection scheme
   306) 
   307)      if( masterproc ) write(iulog,*) 'convect_shallow_init: UW shallow convection scheme (McCaa)'
   308)      if( eddy_scheme .ne. 'diag_TKE' ) then
   309)          write(iulog,*) 'ERROR: shallow convection scheme ', shallow_scheme, ' is incompatible with eddy scheme ', eddy_scheme
   310)          call endrun( 'convect_shallow_init: shallow_scheme and eddy_scheme are incompatible' )
   311)      endif
   312)      call init_uwshcu( r8, latvap, cpair, latice, zvir, rair, gravit, mwh2o/mwdry )
   313) 
   314)      tke_idx = pbuf_get_index('tke')
   315) 
   316)   case('UNICON') ! Sungsu Park's General Convection Model
   317) 
   318)      if ( masterproc ) write(iulog,*) 'convect_shallow_init: General Convection Model by Sungsu Park'
   319)      if ( eddy_scheme .ne. 'diag_TKE' ) then
   320)           write(iulog,*)  eddy_scheme
   321)           write(iulog,*) 'ERROR: shallow convection scheme ',shallow_scheme,' is incompatible with eddy scheme ', eddy_scheme
   322)           call endrun( 'convect_shallow_init: shallow_scheme and eddy_scheme are incompatible' )
   323)      endif
   324)      call unicon_cam_init(pbuf2d)
   325) 
   326)   end select
   327) 
   328)   cld_idx      = pbuf_get_index('CLD')
   329)   concld_idx   = pbuf_get_index('CONCLD')
   330)   rprddp_idx   = pbuf_get_index('RPRDDP')
   331) 
   332)   end subroutine convect_shallow_init

   336) function convect_shallow_implements_cnst(name)
   337) 
   338)    ! Return true if specified constituent is implemented by a shallow convetion package
   339) 
   340)    use unicon_cam, only: unicon_implements_cnst
   341) 
   342)    character(len=*), intent(in) :: name          ! constituent name
   343)    logical :: convect_shallow_implements_cnst    ! return value
   344)    
   345)    integer :: m
   346)    !-----------------------------------------------------------------------
   347) 
   348)    select case (shallow_scheme)
   349) 
   350)    case('UNICON')
   351)       convect_shallow_implements_cnst = unicon_implements_cnst(name)
   352) 
   353)    case default
   354)       convect_shallow_implements_cnst = .false.
   355) 
   356)    end select
   357)

   386)   function convect_shallow_use_shfrc()
   387)   !-------------------------------------------------------------- !
   388)   ! Return true if cloud fraction should use shallow convection   !
   389)   !          calculated convective clouds.                        !
   390)   !-------------------------------------------------------------- !
   391)      implicit none
   392)      logical :: convect_shallow_use_shfrc     ! Return value
   393) 
   394)      if (shallow_scheme .eq. 'UW' .or. shallow_scheme .eq. 'UNICON') then
   395)           convect_shallow_use_shfrc = .true.
   396)      else
   397) 	  convect_shallow_use_shfrc = .false.
   398)      endif
   399) 
   400)      return
   401)

   507)    real(r8), dimension(pcols,pver) :: sl_preCu, qt_preCu, slv_preCu

   510)    type(physics_ptend) :: ptend_loc                                      ! Local tendency from processes, added up to return as ptend_all
   511)

   513)    real(r8), pointer, dimension(:,:) :: cld
   514)    real(r8), pointer, dimension(:,:) :: concld
   515)    real(r8), pointer, dimension(:,:) :: icwmr                            ! In cloud water + ice mixing ratio
   516)    real(r8), pointer, dimension(:,:) :: rprddp                           ! dq/dt due to deep convective rainout
   517)    real(r8), pointer, dimension(:,:) :: rprdsh                           ! dq/dt due to deep and shallow convective rainout
   518)    real(r8), pointer, dimension(:,:) :: evapcsh                          ! Evaporation of shallow convective precipitation >= 0.
   519)    real(r8), pointer, dimension(:)   :: cnt
   520)    real(r8), pointer, dimension(:)   :: cnb
   521)    real(r8), pointer, dimension(:)   :: cush
   522)    real(r8), pointer, dimension(:,:) :: tke
   523)    real(r8), pointer, dimension(:,:) :: shfrc
   524)    real(r8), pointer, dimension(:,:) :: flxprec                          ! Shallow convective-scale flux of precip (rain+snow) at interfaces [ kg/m2/s ]
   525)    real(r8), pointer, dimension(:,:) :: flxsnow                          ! Shallow convective-scale flux of snow at interfaces [ kg/m2/s ]
   526)    real(r8), pointer, dimension(:,:) :: sh_cldliq
   527)    real(r8), pointer, dimension(:,:) :: sh_cldice

   533)    ! ----------------------- !
   534)    ! Main Computation Begins ! 
   535)    ! ----------------------- !
   536) 
   537)    zero  = 0._r8
   538)    nstep = get_nstep()
   539)    lchnk = state%lchnk
   540)    ncol  = state%ncol
   541)   
   542)    call physics_state_copy( state, state1 )          ! Copy state to local state1.
   543) 
   544)    ! Associate pointers with physics buffer fields
   545) 
   546) 
   547)    itim_old   =  pbuf_old_tim_idx()
   548)    call pbuf_get_field(pbuf, cld_idx,         cld,    start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
   549)    call pbuf_get_field(pbuf, concld_idx,      concld, start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
   550) 
   551)    call pbuf_get_field(pbuf, icwmrsh_idx,     icwmr)
   552) 
   553)    call pbuf_get_field(pbuf, rprddp_idx,      rprddp )
   554) 
   555)    call pbuf_get_field(pbuf, rprdsh_idx,      rprdsh )
   556) 
   557)    call pbuf_get_field(pbuf, nevapr_shcu_idx, evapcsh  )
   558) 
   559)    call pbuf_get_field(pbuf, cldtop_idx,      cnt )
   560) 
   561)    call pbuf_get_field(pbuf, cldbot_idx,      cnb )
   562) 
   563)    call pbuf_get_field(pbuf, prec_sh_idx,   precc )
   564) 
   565)    call pbuf_get_field(pbuf, snow_sh_idx,    snow )
   566) 
   567)    if( convect_shallow_use_shfrc() ) then
   568)        call pbuf_get_field(pbuf, shfrc_idx,  shfrc  )
   569)    endif
   570)

   575)    call cnst_get_ind( 'CLDICE', ixcldice )

   579)    !  This field probably should reference the pbuf tpert field but it doesnt

   581) 
   582) 
   583)    select case (shallow_scheme)
   584) 
   585)    case('off', 'CLUBB_SGS') ! None
   586) 
   587)       lq(:) = .TRUE.
   588)       call physics_ptend_init( ptend_loc, state%psetcols, 'convect_shallow (off)', ls=.true., lq=lq ) ! Initialize local ptend type
   589)

   591)       ptend_loc%q = 0._r8
   592)       ptend_loc%s = 0._r8
   593)       rprdsh      = 0._r8
   594)       cmfdqs      = 0._r8
   595)       precc       = 0._r8
   596)       slflx       = 0._r8
   597)       qtflx       = 0._r8
   598)       icwmr       = 0._r8
   599)       rliq2       = 0._r8

   601)       cmfsl       = 0._r8
   602)       cmflq       = 0._r8
   603)       cnt2        = pver
   604)       cnb2        = 1._r8
   605)       evapcsh     = 0._r8
   606)       snow        = 0._r8
   607) 
   608)    case('Hack') ! Hack scheme
   609)                                    
   610)       lq(:) = .TRUE.

   612)

   617)                    nstep        ,  ztodt        ,  state%pmid ,  state%pdel  ,                  &
   618)                    state%rpdel  ,  state%zm     ,  tpert      ,  qpert       ,  state%phis  ,   &

   621)                    qc2          ,  cnt2         ,  cnb2       ,  icwmr       ,  rliq2       ,   & 
   622)                    state%pmiddry,  state%pdeldry,  state%rpdeldry )
   623) 
   624)    case('UW')   ! UW shallow convection scheme
   625) 
   626)       ! -------------------------------------- !
   627)       ! uwshcu does momentum transport as well !
   628)       ! -------------------------------------- !
   629) 
   630)       ! Initialize local ptend type
   631)       lq(:) = .TRUE.
   632)       call physics_ptend_init( ptend_loc, state%psetcols, 'UWSHCU', ls=.true., lu=.true., lv=.true., lq=lq  )

   635)       call pbuf_get_field(pbuf, tke_idx,  tke)

   637)

   639)       call pbuf_get_field(pbuf, sh_flxsnw_idx, flxsnow)

   641)       call compute_uwshcu_inv( pcols     , pver    , ncol           , pcnst         , ztodt         ,                   &
   642)                                state%pint, state%zi, state%pmid     , state%zm      , state%pdel    ,                   & 
   643)                                state%u   , state%v , state%q(:,:,1) , state%q(:,:,ixcldliq), state%q(:,:,ixcldice),     &
   644)                                state%t   , state%s , state%q(:,:,:) ,                                                   &
   645)                                tke       , cld     , concld         , pblh          , cush          ,                   &
   646)                                cmfmc2    , slflx   , qtflx          , 							&
   647) 			       flxprec, flxsnow, 			         					&
   648)                                ptend_loc%q(:,:,1)  , ptend_loc%q(:,:,ixcldliq), ptend_loc%q(:,:,ixcldice),              &
   649)                                ptend_loc%s         , ptend_loc%u    , ptend_loc%v   , ptend_tracer  ,                   &
   650)                                rprdsh              , cmfdqs         , precc         , snow          ,                   &
   651)                                evapcsh             , shfrc          , iccmr_UW      , icwmr_UW      ,                   &
   652)                                icimr_UW            , cbmf           , qc2           , rliq2         ,                   &
   653)                                cnt2                , cnb2           , lchnk         , state%pdeldry )
   654) 
   655)       ! --------------------------------------------------------------------- !
   656)       ! Here, 'rprdsh = qrten', 'cmfdqs = qsten' both in unit of [ kg/kg/s ]  !
   657)       ! In addition, define 'icwmr' which includes both liquid and ice.       !
   658)       ! --------------------------------------------------------------------- !
   659) 
   660)       icwmr(:ncol,:)  = iccmr_UW(:ncol,:) 
   661)       rprdsh(:ncol,:) = rprdsh(:ncol,:) + cmfdqs(:ncol,:)
   662)       do m = 4, pcnst
   663)          ptend_loc%q(:ncol,:pver,m) = ptend_tracer(:ncol,:pver,m)
   664)       enddo
   665) 
   666)       ! Conservation check
   667)       
   668)       !  do i = 1, ncol
   669)       !  do m = 1, pcnst
   670)       !     sum1 = 0._r8
   671)       !     sum2 = 0._r8
   672)       !     sum3 = 0._r8
   673)       !  do k = 1, pver
   674)       !       if(cnst_get_type_byind(m).eq.'wet') then
   675)       !          pdelx = state%pdel(i,k)
   676)       !       else
   677)       !          pdelx = state%pdeldry(i,k)
   678)       !       endif
   679)       !       sum1 = sum1 + state%q(i,k,m)*pdelx
   680)       !       sum2 = sum2 +(state%q(i,k,m)+ptend_loc%q(i,k,m)*ztodt)*pdelx  
   681)       !       sum3 = sum3 + ptend_loc%q(i,k,m)*pdelx 
   682)       !  enddo
   683)       !  if( m .gt. 3 .and. abs(sum1) .gt. 1.e-13_r8 .and. abs(sum2-sum1)/sum1 .gt. 1.e-12_r8 ) then
   684)       !! if( m .gt. 3 .and. abs(sum3) .gt. 1.e-13_r8 ) then
   685)       !      write(iulog,*) 'Sungsu : convect_shallow.F90 does not conserve tracers : ', m, sum1, sum2, abs(sum2-sum1)/sum1
   686)       !!     write(iulog,*) 'Sungsu : convect_shallow.F90 does not conserve tracers : ', m, sum3
   687)       !  endif
   688)       !  enddo
   689)       !  enddo
   690) 
   691)       ! ------------------------------------------------- !
   692)       ! Convective fluxes of 'sl' and 'qt' in energy unit !
   693)       ! ------------------------------------------------- !
   694) 
   695)       cmfsl(:ncol,:) = slflx(:ncol,:)
   696)       cmflq(:ncol,:) = qtflx(:ncol,:) * latvap
   697) 
   698)       call outfld( 'PRECSH' , precc  , pcols, lchnk )
   699) 
   700) 
   701)    case('UNICON')
   702) 
   703)       icwmr = 0.0_r8
   704) 
   705)       call unicon_cam_tend(ztodt, state, cam_in, sgh30, &
   706)                            pbuf, ptend_loc, unicon_out)
   707) 
   708)       cmfmc2(:ncol,:) = unicon_out%cmfmc(:ncol,:)
   709)       qc2(:ncol,:)    = unicon_out%rqc(:ncol,:)
   710)       rliq2(:ncol)    = unicon_out%rliq(:ncol)
   711)       cnt2(:ncol)     = unicon_out%cnt(:ncol)
   712)       cnb2(:ncol)     = unicon_out%cnb(:ncol)
   713) 
   714)       ! ------------------------------------------------- !
   715)       ! Convective fluxes of 'sl' and 'qt' in energy unit !
   716)       ! ------------------------------------------------- !
   717) 
   718)       cmfsl(:ncol,:) = unicon_out%slflx(:ncol,:)
   719)       cmflq(:ncol,:) = unicon_out%qtflx(:ncol,:) * latvap
   720) 
   721)       call outfld( 'PRECSH' , precc  , pcols, lchnk )
   722) 
   723)    end select
   724) 
   725)    ! --------------------------------------------------------!     
   726)    ! Calculate fractional occurance of shallow convection    !
   727)    ! --------------------------------------------------------!
   728) 
   729)  ! Modification : I should check whether below computation of freqsh is correct.

   731)    freqsh(:) = 0._r8
   732)    do i = 1, ncol
   733)       if( maxval(cmfmc2(i,:pver)) <= 0._r8 ) then
   734)           freqsh(i) = 1._r8
   735)       end if
   736)    end do
   737) 
   738)    ! ------------------------------------------------------------------------------ !
   739)    ! Merge shallow convection output with prior results from deep convection scheme !
   740)    ! ------------------------------------------------------------------------------ !
   741) 
   742)    ! ----------------------------------------------------------------------- !
   743)    ! Combine cumulus updraft mass flux : 'cmfmc2'(shallow) + 'cmfmc'(deep)   !
   744)    ! ----------------------------------------------------------------------- !
   745) 
   746)    cmfmc(:ncol,:) = cmfmc(:ncol,:) + cmfmc2(:ncol,:)
   747) 
   748)    ! -------------------------------------------------------------- !
   749)    ! 'cnt2' & 'cnb2' are from shallow, 'cnt' & 'cnb' are from deep  !
   750)    ! 'cnt2' & 'cnb2' are the interface indices of cloud top & base: ! 
   751)    !        cnt2 = float(kpen)                                      !

   753)    ! Note that indices decreases with height.                       !

   756)    do i = 1, ncol
   757)       if( cnt2(i) < cnt(i)) cnt(i) = cnt2(i)
   758)       if( cnb2(i) > cnb(i)) cnb(i) = cnb2(i)
   759)       pcnt(i) = state%pmid(i,int(cnt(i)))
   760)       pcnb(i) = state%pmid(i,int(cnb(i)))     
   761)    end do
   762)    
   763)    ! ----------------------------------------------- !
   764)    ! This quantity was previously known as CMFDQR.   !
   765)    ! Now CMFDQR is the shallow rain production only. !
   766)    ! ----------------------------------------------- !
   767) 
   768)    
   769)    call pbuf_set_field(pbuf, rprdtot_idx, rprdsh(:ncol,:pver) + rprddp(:ncol,:pver), start=(/1,1/), kount=(/ncol,pver/))
   770)  
   771)    ! ----------------------------------------------------------------------- ! 
   772)    ! Add shallow reserved cloud condensate to deep reserved cloud condensate !
   773)    !     qc [ kg/kg/s] , rliq [ m/s ]                                        !
   774)    ! ----------------------------------------------------------------------- !
   775) 
   776)    qc(:ncol,:pver) = qc(:ncol,:pver) + qc2(:ncol,:pver)
   777)    rliq(:ncol)     = rliq(:ncol) + rliq2(:ncol)    
   778) 
   779)    ! ---------------------------------------------------------------------------- !
   780)    ! Output new partition of cloud condensate variables, as well as precipitation !
   781)    ! ---------------------------------------------------------------------------- ! 
   782) 
   783)    if( microp_scheme == 'MG' ) then
   784)        call cnst_get_ind( 'NUMLIQ', ixnumliq )
   785)        call cnst_get_ind( 'NUMICE', ixnumice )
   786)    endif
   787) 
   788)    ftem(:ncol,:pver) = ptend_loc%s(:ncol,:pver)/cpair
   789) 
   790)    call outfld( 'ICWMRSH ', icwmr                    , pcols   , lchnk )
   791) 
   792)    call outfld( 'CMFDT  ', ftem                      , pcols   , lchnk )
   793)    call outfld( 'CMFDQ  ', ptend_loc%q(1,1,1)        , pcols   , lchnk )
   794)    call outfld( 'CMFDICE', ptend_loc%q(1,1,ixcldice) , pcols   , lchnk )
   795)    call outfld( 'CMFDLIQ', ptend_loc%q(1,1,ixcldliq) , pcols   , lchnk )
   796)    call outfld( 'CMFMC'  , cmfmc                     , pcols   , lchnk )
   797)    call outfld( 'QC'     , qc2                       , pcols   , lchnk )
   798)    call outfld( 'CMFDQR' , rprdsh                    , pcols   , lchnk )
   799)    call outfld( 'CMFSL'  , cmfsl                     , pcols   , lchnk )
   800)    call outfld( 'CMFLQ'  , cmflq                     , pcols   , lchnk )
   801)    call outfld( 'DQP'    , qc2                       , pcols   , lchnk )
   802)    call outfld( 'CLDTOP' , cnt                       , pcols   , lchnk )
   803)    call outfld( 'CLDBOT' , cnb                       , pcols   , lchnk )
   804)    call outfld( 'PCLDTOP', pcnt                      , pcols   , lchnk )
   805)    call outfld( 'PCLDBOT', pcnb                      , pcols   , lchnk )  
   806)    call outfld( 'FREQSH' , freqsh                    , pcols   , lchnk )
   807) 
   808)    if( shallow_scheme .eq. 'UW' ) then
   809)       call outfld( 'CBMF'   , cbmf                      , pcols   , lchnk )
   810)       call outfld( 'UWFLXPRC', flxprec                  , pcols   , lchnk )  
   811)       call outfld( 'UWFLXSNW' , flxsnow                 , pcols   , lchnk )
   812)    endif
   813) 
   814)    ! ---------------------------------------------------------------- !
   815)    ! Add tendency from this process to tend from other processes here !
