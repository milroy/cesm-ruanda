   144)       subroutine get_crs( xs_short_file, sht_indexer )
   145) !=============================================================================
   146) !   PURPOSE:
   147) !   Reads a NetCDF file that contains:
   148) !     Cross_sections*quantum yield data for species <200nm.
   149) !     Exoatmospheric flux on the wavelength grid of the crs
   150) !=============================================================================
   151) !   PARAMETERS:
   152) !     Input:
   153) !      xs_short_file.... NetCDF file that contains the crs*QY for wavenum species
   154) !
   155) !     Output:
   156) !      xs_species.. Cross Sections * QY data for each species
   157) !      etfphot..... Exoatmospheric flux in photons cm-2 s-1 nm-1
   158) !      etfphot_ms93.Minshwanner and Siskind JNO SRB etf (on MS93 grid)
   159) !      wc.......... wavelength center (nm)
   160) !      numwl ...... Number of wavelengths < 200nm in NetCDF input file
   161) !      wlintv ..... Wavelength inteval for grid, nm
   162) !=============================================================================
   163) !   EDIT HISTORY:
   164) !   Created by Doug Kinnison, 1/14/2002
   165) !   Modified by S. Walters, 4/2/2003
   166) !=============================================================================
   167) 
   168)       use chem_mods,      only : phtcnt, pht_alias_lst, rxt_tag_lst
   169)       use ioFileMod,      only : getfil
   170)       use error_messages, only : alloc_err
   171)       use pio,            only : file_desc_t, pio_get_var, pio_closefile, pio_noerr, &
   172)            pio_seterrorhandling, pio_bcast_error, pio_internal_error, pio_inq_varid, &
   173)            pio_inq_dimlen, pio_nowrite, pio_inq_dimid
   174)       use cam_pio_utils,  only : cam_pio_openfile
   175)       implicit none
   176) 
   177) !------------------------------------------------------------------------------
   178) !       ... dummy arguments
   179) !------------------------------------------------------------------------------
   180)       integer, intent(inout)       :: sht_indexer(:)
   181)       character(len=*), intent(in) :: xs_short_file
   182) 
   183) !------------------------------------------------------------------------------
   184) !       ... local variables
   185) !------------------------------------------------------------------------------
   186)       integer :: wn
   187)       type(file_desc_t) :: ncid
   188)       integer :: ierr
   189)       integer :: i, m, ndx
   190)       integer :: varid, dimid
   191)       integer :: wrk_ndx(phtcnt)
   192)       real(r8), allocatable :: xs_species(:)
   193)       character(len=256) :: locfn
   194) 
   195) !------------------------------------------------------------------------------
   196) !       ... open NetCDF File
   197) !------------------------------------------------------------------------------
   198)       call getfil(xs_short_file, locfn, 0)
   199)       call cam_pio_openfile(ncid, trim(locfn), PIO_NOWRITE)
   200) 
   201) !------------------------------------------------------------------------------
   202) !       ... get dimensions
   203) !------------------------------------------------------------------------------
   204)       ierr = pio_inq_dimid( ncid, 'numwl', dimid )
   205)       ierr = pio_inq_dimlen( ncid, dimid, nw )
   206) 
   207) !------------------------------------------------------------------------------
   208) !       ... check for cross section in dataset
   209) !------------------------------------------------------------------------------
   210)       call pio_seterrorhandling(ncid, pio_bcast_error)
   211)       do m = 1,phtcnt
   212)          if( pht_alias_lst(m,1) == ' ' ) then
   213)             ierr = pio_inq_varid( ncid, rxt_tag_lst(m), varid )
   214)             if( ierr == PIO_noerr ) then 
   215)                sht_indexer(m) = varid
   216)             end if
   217)          else if( pht_alias_lst(m,1) == 'userdefined' ) then
   218)             sht_indexer(m) = -1
   219)          else
   220)             ierr = pio_inq_varid( ncid, pht_alias_lst(m,1), varid )
   221)             if( ierr == PIO_noerr ) then 
   222)                sht_indexer(m) = varid
   223)             else
   224)                write(iulog,*) 'get_crs : ',rxt_tag_lst(m)(:len_trim(rxt_tag_lst(m))),' alias ', &
   225)                     pht_alias_lst(m,1)(:len_trim(pht_alias_lst(m,1))),' not in dataset'            
   226)                call endrun
   227)             end if
   228)          end if
   229)       end do
   230)       call pio_seterrorhandling(ncid, pio_internal_error)
   231) 
   232)       if (masterproc) then
   233)          write(iulog,*) ' '
   234)          write(iulog,*) '###############################################'
   235)          write(iulog,*) 'get_crs: sht_indexer'
   236)          write(iulog,'(10i6)') sht_indexer(:)
   237)          write(iulog,*) '###############################################'
   238)          write(iulog,*) ' '
   239)       endif
   240) 
   241)       nj = 0
   242)       do m = 1,phtcnt
   243)          if( sht_indexer(m) > 0 ) then
   244)             if( any( sht_indexer(:m-1) == sht_indexer(m) ) ) then
   245)                cycle
   246)             end if
   247)             nj = nj + 1
   248)          end if
   249)       end do
   250) 
   251) !------------------------------------------------------------------------------
   252) !       ... allocate arrays
   253) !------------------------------------------------------------------------------
   254)       allocate( wc(nw),stat=ierr )
   255)       if( ierr /= 0 ) then 
   256) 	 call alloc_err( ierr, 'get_crs', 'wc', nw )
   257)       end if
   258)       allocate( we(nw+1),stat=ierr )
   259)       if( ierr /= 0 ) then 
   260) 	 call alloc_err( ierr, 'get_crs', 'we', nw+1 )
   261)       end if
   262)       allocate( wlintv(nw),stat=ierr )
   263)       if( ierr /= 0 ) then 
   264) 	 call alloc_err( ierr, 'get_crs', 'wlintv', nw )
   265)       end if
   266)       allocate( etfphot(nw),stat=ierr )
   267)       if( ierr /= 0 ) then 
   268) 	 call alloc_err( ierr, 'get_crs', 'etfphot', nw )
   269)       end if
   270)       allocate( bde_o2_a(nw),bde_o2_b(nw),bde_o3_a(nw),bde_o3_b(nw),stat=ierr )
   271)       if( ierr /= 0 ) then 
   272) 	 call alloc_err( ierr, 'get_crs', 'bde_o2_a ... bde_o3_b', nw )
   273)       end if
   274)       allocate( etfphot_ms93(nw_ms93),stat=ierr )
   275)       if( ierr /= 0 ) then 
   276) 	 call alloc_err( ierr, 'get_crs', 'etfphot_ms93', nw_ms93 )
   277)       end if
   278)       allocate( xs_o2src(nw),stat=ierr )
   279)       if( ierr /= 0 ) then 
   280) 	 call alloc_err( ierr, 'get_crs', 'xs_o2src', nw )
   281)       end if
   282)       allocate( xs_o3a(nw),xs_o3b(nw),stat=ierr )
   283)       if( ierr /= 0 ) then 
   284)          call alloc_err( ierr, 'get_crs', 'xs_o3a,xs_o3b', nw )
   285)       end if
   286)       allocate( xs_species(nw),xs_wl(nw,nj),stat=ierr )
   287)       if( ierr /= 0 ) then 
   288)          call alloc_err( ierr, 'get_crs', 'xs_species,xs_wl', nw*nj )
   289)       end if
   290) 
   291) !------------------------------------------------------------------------------
   292) !       ... read arrays
   293) !------------------------------------------------------------------------------
   294)       ierr = pio_inq_varid( ncid, 'wc', varid )
   295)       ierr = pio_get_var( ncid, varid, wc )
   296)       ierr = pio_inq_varid( ncid, 'wlintv', varid )
   297)       ierr = pio_get_var( ncid, varid, wlintv )
   298)       ierr = pio_inq_varid( ncid, 'xs_o2src', varid )
   299)       ierr = pio_get_var( ncid, varid, xs_o2src )
   300)       ndx = 0
   301)       do m = 1,phtcnt
   302)          if( sht_indexer(m) > 0 ) then
   303)             if( any( sht_indexer(:m-1) == sht_indexer(m) ) ) then
   304)                cycle
   305)             end if
   306)             ierr = pio_get_var( ncid, sht_indexer(m), xs_species )
   307)             ndx = ndx + 1
   308)             xs_wl(:,ndx) = xs_species(:)*wlintv(:)
   309)          end if
   310)       end do
   311)       deallocate( xs_species )
   312)       if( ndx /= nj ) then
   313)          write(iulog,*) 'get_crs : ndx count /= cross section count'
   314)          call endrun
   315)       end if
   316)       !------------------------------------------------------------------------------
   317)       !       ... get jo3 cross sections
   318)       !------------------------------------------------------------------------------
   319)       ierr = pio_inq_varid( ncid, 'jo3_a', varid )
   320)       ierr = pio_get_var( ncid, varid, xs_o3a )
   321)       ierr = pio_inq_varid( ncid, 'jo3_b', varid )
   322)       ierr = pio_get_var( ncid, varid, xs_o3b )
   323)       !------------------------------------------------------------------------------
   324)       !       ... setup final sht_indexer
   325)       !------------------------------------------------------------------------------
   326)       ndx = 0
   327)       wrk_ndx(:) = sht_indexer(:)
   328)       do m = 1,phtcnt
   329)          if( wrk_ndx(m) > 0 ) then
   330)             ndx = ndx + 1
   331)             i = wrk_ndx(m)
   332)             where( wrk_ndx(:) == i )
   333)                sht_indexer(:) = ndx
   334)                wrk_ndx(:)     = -100000
   335)             end where
   336)          end if
   337)       end do
   338) 
   339)       call pio_closefile( ncid )
   340) 
   341) #ifdef USE_BDE
   342)       if (masterproc) write(iulog,*) 'Jshort using bdes'
   343) #else
   344)       if (masterproc) write(iulog,*) 'Jshort not using bdes'
   345) #endif
   346)       do wn = 1,nw
   347) #ifdef USE_BDE
   348)          bde_o2_a(wn) = max(0._r8, hc*(wc_o2_a - wc(wn))/(wc_o2_a*wc(wn)) )
   349)          bde_o2_b(wn) = max(0._r8, hc*(wc_o2_b - wc(wn))/(wc_o2_b*wc(wn)) )
   350)          bde_o3_a(wn) = max(0._r8, hc*(wc_o3_a - wc(wn))/(wc_o3_a*wc(wn)) )
   351)          bde_o3_b(wn) = max(0._r8, hc*(wc_o3_b - wc(wn))/(wc_o3_b*wc(wn)) )
   352) #else
   353)          bde_o2_a(wn) = hc/wc(wn)
   354)          bde_o2_b(wn) = hc/wc(wn)
   355)          bde_o3_a(wn) = hc/wc(wn)
   356)          bde_o3_b(wn) = hc/wc(wn)
   357) #endif
   358)       end do
   359) 
   360)       end subroutine get_crs

   362)       subroutine xs_init(xs_coef_file)
   363) !-------------------------------------------------------------
   364) !    	... Loads XS_COEFFS containing the Chebyshev
   365) !           polynomial coeffs necessary to calculate O2 effective
   366) !           cross-sections
   367) !-------------------------------------------------------------
   368) 
   369)       use ioFileMod,     only : getfil
   370)       use units,         only : getunit, freeunit
   371) 
   372) !------------------------------------------------------------------------------
   373) !    ... Dummy arguments
   374) !------------------------------------------------------------------------------            
   375)       character(len=*), intent(in) :: xs_coef_file
   376) 
   377) !-------------------------------------------------------------
   378) !     ... Local variables
   379) !-------------------------------------------------------------
   380)       integer :: unit	! file unit number
   381)       integer :: istat	! i/o status
   382)       integer :: i, j
   383)       character(len=256) :: locfn
   384) 
   385) !----------------------------------------------------------------------
   386) !	... Get first strato photo rate file
   387) !----------------------------------------------------------------------
   388)          call getfil(xs_coef_file, locfn, 0)
   389) !----------------------------------------------------------------------
   390) !	... open file
   391) !----------------------------------------------------------------------
   392)          unit = getunit()
   393)          open( unit   = unit, &
   394)                file   = trim(locfn), &
   395)                status = 'old', &
   396)                form   = 'formatted', &
   397)                iostat = istat )
   398)          if( istat /= 0 ) then
   399) !----------------------------------------------------------------------
   400) !	... Open error exit
   401) !----------------------------------------------------------------------
   402)             write(iulog,*) 'xs_init: error ',istat,' opening file ',trim(locfn)
   403)             call endrun
   404)          end if
   405) !----------------------------------------------------------------------
   406) !	... read file
   407) !----------------------------------------------------------------------
   408)          read(unit,901)
   409)          do i = 1,20
   410)             read(unit,903,iostat=istat) ac(i,:)
   411)             if( istat /= 0 ) then
   412)                write(iulog,*) 'xs_init: error ',istat,' reading ac'
   413)                call endrun
   414)             end if
   415)          end do
   416) 
   417)          read(unit,901)
   418)          do i = 1,20
   419)             read(unit,903,iostat=istat) bc(i,:)
   420)             if( istat /= 0 ) then
   421)                write(iulog,*) 'xs_init: error ',istat,' reading bc'
   422)                call endrun
   423)             end if
   424)          end do
   425)          close( unit )
   426)          call freeunit( unit )
   427) 
   428)       wave_num(17:1:-1) = (/ (48250._r8 + (500._r8*i),i=1,17) /)
   429) 
   430)  901  format( / )
   431)  903  format( 17(e23.14,1x))
   432) 
   433)       end subroutine xs_init

   511)       subroutine jshort_hrates( nlev, zen, o2_vmr, o3_vmr, o2cc, &
   512)                                 o3cc, tlev, zkm, mw, qrs, cparg, &
   513)                                 lchnk, long, co2cc, scco2, do_diag )
   514) !==============================================================================!
   515) !   Subroutine Jshort                                                          !
   516) !==============================================================================!
   517) !   Purpose:                                                                   !
   518) !     To calculate thermal heating rates for lamba < 200 nm
   519) !==============================================================================!
   520) !   This routine uses JO2 parameterizations based on:                          !
   521) !        Lyman alpha... Chabrillat and Kockarts, GRL, 25, 2659, 1998           !
   522) !        SRC .......... Brasseur and Solomon, 1986 (from TUV)                  !
   523) !        SRB .......... Koppers and Murtagh, Ann. Geophys., 14, 68-79, 1996    !
   524) !                        (supplied by Dan Marsh, NCAR ACD                      !
   525) !   and JNO:                                                                   !
   526) !        SRB .......... Minschwanner and Siskind, JGR< 98, 20401, 1993.        !
   527) !==============================================================================!
   528) !   Input:                                                                     !
   529) !       o2cc....... O2 concentration, molecule cm-3			       !
   530) !	o3cc....... O3 concentration, molecule cm-3			       !
   531) !     zen........ zenith angle, units = degrees                                !
   532) !     tlev....... Temperature Profile (K)                                      !
   533) !     zkm ....... Altitude, km                                                 !
   534) !                                                                              !
   535) !   Output:                                                                    !
   536) !     qrs ... short wavelength thermal heating rates
   537) !==============================================================================!
   538) !                                                                              !
   539) !   Approach:                                                                  !
   540) !                                                                              !
   541) !    1) Call sphers (taken from TUV)                                           !
   542) !         -> derives dsdh and nid used in slant column routines                !
   543) !	  -> zenith angle dependent                                            !
   544) !                                                                              !
   545) !    2) Call  slant_col (taken from TUV)                                       !
   546) !		-> derives the slant column for each species                   !
   547) !                                                                              !
   548) !    3) Calls get_crs                                                          !
   549) !		-> read a NetCDF file                                          !
   550) !		-> returns cross sections*quantum yields for all species that  !
   551) !		   have absorption below 200nm.                                !
   552) !                                                                              !
   553) !    4) Derives transmission and photolysis rates for selective species        !
   554) !                                                                              !
   555) !==============================================================================!
   556) !   EDIT HISTORY:                                                              !
   557) !   Created by Doug Kinnison, 3/14/2002                                        !
   558) !==============================================================================!
   559) 
   560)        use physconst,       only : avogad
   561)        use error_messages, only : alloc_err
   562) 
   563)        implicit none
   564) 
   565)        integer, parameter  :: branch = 2           ! two photolysis pathways for JO2
   566)        real(r8), parameter :: km2cm  = 1.e5_r8
   567) 
   568) !------------------------------------------------------------------------------
   569) !     ... dummy arguments
   570) !------------------------------------------------------------------------------
   571)        integer, intent(in)     :: nlev                 ! model vertical levels
   572)        integer, intent(in)     :: lchnk                ! chunk index
   573)        integer, intent(in)     :: long                 ! chunk index
   574)        real(r8), intent(in)    :: zen                  ! Zenith angle (degrees)
   575)        real(r8), intent(in)    :: o2_vmr(nlev)         ! o2 conc (mol/mol)
   576)        real(r8), intent(in)    :: o3_vmr(nlev)         ! o3 conc (mol/mol)
   577)        real(r8), intent(in)    :: o2cc(nlev)           ! o2 conc (mol/cm^3)
   578)        real(r8), intent(in)    :: co2cc(nlev)          ! co2 conc (mol/cm^3)
   579)        real(r8), intent(in)    :: o3cc(nlev)           ! o3 conc (mol/cm^3)
   580)        real(r8), intent(in)    :: tlev(nlev)           ! Temperature profile
   581)        real(r8), intent(in)    :: zkm(nlev)            ! Altitude, km
   582)        real(r8), intent(in)    :: mw(nlev)             ! atms molecular weight
   583)        real(r8), intent(in)    :: cparg(nlev)          ! column specific heat capacity
   584)        real(r8), intent(inout) :: qrs(:,:)             ! sw heating rates
   585)        real(r8), intent(out)   :: scco2(nlev)          ! co2 column concentration (molec/cm^2)
   586)        logical,  intent(in)    :: do_diag
   587) 
   588) !------------------------------------------------------------------------------
   589) !     ... local variables
   590) !------------------------------------------------------------------------------
   591)        integer :: k, k1                     ! Altitude indicies
   592)        integer :: wn                        ! Wavelength index
   593)        integer :: astat
   594)        integer :: nid(0:nlev)               ! Number of layers crossed by the direct
   595)                                             ! beam when travelling from the top of the
   596)                                             ! atmosphere to layer i; NID(i), i = 0..NZ-1
   597)        real(r8) :: hfactor
   598)        real(r8) :: dsdh(0:nlev,nlev)        ! Slant path of direct beam through each
   599)                                             ! layer crossed  when travelling from the top of
   600)                                             ! the atmosphere to layer i; DSDH(i,j), i = 0.
   601)                                             ! NZ-1, j = 1..NZ-1   (see sphers.f)
   602)        real(r8), allocatable :: fnorm(:,:)      ! Normalized ETF
   603)        real(r8), allocatable :: trans_o2(:,:)   ! Transmission o2 (total)
   604)        real(r8), allocatable :: trans_o3(:,:)   ! Transmission, ozone
   605)        real(r8), allocatable :: wrk(:)      ! wrk array
   606)        real(r8) :: jo2_lya(nlev)            ! Total photolytic rate constant for Ly alpha
   607)        real(r8) :: jo2_srb(nlev)            ! Total JO2  for SRB
   608)        real(r8) :: jo2_src(nlev)            ! Total JO2 for SRC
   609)        real(r8) :: delz(nlev)               ! layer thickness (cm)
   610)        real(r8) :: o2scol(nlev)             ! O2 Slant Column
   611)        real(r8) :: o3scol(nlev)             ! O3 Slant Column
   612)        real(r8) :: rmla(nlev)               ! Transmission, Lyman Alpha (other species)
   613)        real(r8) :: ro2la(nlev)              ! Transmission, Lyman Alpha (for JO2)
   614)        real(r8) :: tlevmin(nlev)
   615)        real(r8) :: abs_col(nlev)
   616)        real(r8) :: tsrb(nlev,nsrbtuv)       ! Transmission in the SRB
   617)        real(r8) :: xs_o2srb(nlev,nsrbtuv)   ! Cross section * QY for O2 in SRB
   618) 
   619)       allocate( fnorm(nlev,nw),stat=astat )
   620)       if( astat /= 0 ) then
   621)          call alloc_err( astat, 'jshort_hrates', 'fnorm', nw*nlev )
   622)       end if
   623)       allocate( trans_o2(nlev,nw),stat=astat )
   624)       if( astat /= 0 ) then
   625)          call alloc_err( astat, 'jshort_hrates', 'trans_o2', nw*nlev )
   626)       end if
   627)       allocate( trans_o3(nlev,nw),stat=astat )
   628)       if( astat /= 0 ) then
   629)          call alloc_err( astat, 'jshort_hrates', 'trans_o3', nw*nlev )
   630)       end if
   631)       allocate( wrk(nw),stat=astat )
   632)       if( astat /= 0 ) then
   633)          call alloc_err( astat, 'jshort_hrates', 'wrk', nw )
   634)       end if
   635) 
   636) !------------------------------------------------------------------------------
   637) !     ... derive Slant Path for Spherical Atmosphere
   638) !------------------------------------------------------------------------------
   639)       call sphers( nlev, zkm, zen, dsdh, nid )
   640) 
   641) !------------------------------------------------------------------------------
   642) !  	... derive O2, O3, and CO2 slant columns
   643) !------------------------------------------------------------------------------
   644)       delz(1:nlev-1) = km2cm*(zkm(1:nlev-1) - zkm(2:nlev))
   645)       call slant_col( nlev, delz, dsdh, nid, o2cc, o2scol )
   646)       call slant_col( nlev, delz, dsdh, nid, o3cc, o3scol )
   647)       call slant_col( nlev, delz, dsdh, nid, co2cc, scco2 )
   648) 
   649) !------------------------------------------------------------------------------
   650) !  	... transmission due to ozone
   651) !------------------------------------------------------------------------------
   652)       do wn = 1,nw
   653)          abs_col(:)     = min( (xs_o3a(wn) + xs_o3b(wn))*o3scol(:),100._r8 )
   654)          trans_o3(:,wn) = exp( -abs_col(:) )
   655)       end do
   656) 
   657) !------------------------------------------------------------------------------
   658) !	... derive the cross section and transmission for
   659) !           molecular oxygen Lya, SRC, and SRB's
   660) !------------------------------------------------------------------------------
   661) !  	... transmission due to molecular oxygen in the SRC
   662) !------------------------------------------------------------------------------
   663)       do wn = 1,nsrc_tot
   664)          abs_col(:) = min( xs_o2src(wn)*o2scol(:),100._r8 )
   665)          trans_o2(:,wn) = exp( -abs_col(:) )
   666)       end do
   667) 
   668) !------------------------------------------------------------------------------
   669) !     	... transmission and cross sections due to O2 at lyman alpha
   670) !------------------------------------------------------------------------------
   671)       call lymana( nlev, o2scol, rmla, ro2la )
   672) 
   673) !------------------------------------------------------------------------------
   674) !    	... place lya reduction faction in transmission array
   675) !           this must follow the SRC placement (above)
   676) !------------------------------------------------------------------------------
   677)       trans_o2(:,1) = rmla(:)
   678) 
   679) !------------------------------------------------------------------------------
   680) !     ... Molecular Oxygen, SRB
   681) !------------------------------------------------------------------------------
   682) !     ... Koppers Grid (see Koppers and Murtagh, Ann. Geophys., 14, 68-79, 1996)
   683) !        #    wl(i)       wl(i+1)
   684) !        1   174.4        177.0
   685) !        2   177.0        178.6
   686) !        3   178.6        180.2
   687) !        4   180.2        181.8
   688) !        5   181.8        183.5
   689) !        6   183.5        185.2
   690) !        7   185.2        186.9
   691) !        8   186.9        188.7
   692) !        9   188.7        190.5
   693) !        10  190.5        192.3
   694) !        11  192.3        194.2
   695) !        12  194.2        196.1
   696) !        13  196.1        198.0
   697) !        14  198.0        200.0  <<last wl bin for <200nm
   698) !        ----------------------
   699) !        15  200.0        202.0
   700) !        16  202.0        204.1
   701) !        17  204.1        205.8
   702) !------------------------------------------------------------------------------
   703) 
   704) !------------------------------------------------------------------------------
   705) !     ... Kopper SRB parameterization is used here
   706) !------------------------------------------------------------------------------
   707)       tlevmin(nlev:1:-1) = min( max( tlev(:),150._r8 ),350._r8 )
   708)       call calc_o2srb( nlev, nid, o2scol, tlevmin, tsrb, xs_o2srb )
   709) 
   710) !------------------------------------------------------------------------------
   711) !     ... Place Koppers SRB transmission (1-14) on user
   712) !         grid #20 (wc=175.7nm)
   713) !------------------------------------------------------------------------------
   714)       do wn = 1,nsrb_tot
   715)          trans_o2(:,wn+nsrc_tot) = tsrb(:,wn)
   716)       end do
   717) 
   718) !------------------------------------------------------------------------------
   719) !     ... Derive the normalize flux at each altitude,
   720) !         corrected for O2 and O3 absorption
   721) !------------------------------------------------------------------------------
   722)       do wn = 1,nw                                  ! nw = 33 (nsrb_tot+nsrc_tot)
   723)          fnorm(:,wn) = etfphot(wn)*trans_o2(:,wn)*trans_o3(:,wn)
   724)       end do
   725) 
   726) !------------------------------------------------------------------------------
   727) !     ... Derive the O2 rate constant and apply branching ratio (QY)
   728) !------------------------------------------------------------------------------
   729) !     ... SRC and SRB QY
   730) !         Longward  of 174.65 the product is O2 + hv => O(3P) + O(3P)
   731) !         Shortward of 174.65 the product is O2 + hv => O(3P) + O(1D)
   732) !         The QY is assumed to be unity in both wavelength ranges.
   733) !
   734) !     ... Lyman Alpha QY
   735) !         O2 + hv -> O(3P) + O(3P) at Lyman Alpha has a QY = 0.47
   736) !         O2 + hv -> O(3P) + O(1D) at Lyman Alpha has a QY = 0.53
   737) !         Lacoursiere et al., J. Chem. Phys. 110., 1949-1958, 1999.
   738) !------------------------------------------------------------------------------
   739) !     ... lyman alpha
   740) !------------------------------------------------------------------------------
   741)       jo2_lya(:) = etfphot(1)*ro2la(:)*wlintv(1)
   742) 
   743)       wrk(1:nsrc_tot) = xs_o2src(1:nsrc_tot)*wlintv(1:nsrc_tot) &
   744) 					    *bde_o2_a(1:nsrc_tot)
   745)       wrk(1)          = 0._r8
   746) !------------------------------------------------------------------------------
   747) !     ... o2 src heating
   748) !------------------------------------------------------------------------------
   749)       if( do_diag ) then
   750)       write(iulog,*) '-------------------------------------------------'
   751)       write(iulog,*) 'jshort_hrates: fnorm,wrk at long,lchnk = ',long,lchnk
   752)       write(iulog,'(1p,5g12.5)') fnorm(nlev,1:nsrc_tot)
   753)       write(iulog,*) ' '
   754)       write(iulog,'(1p,5g12.5)') wrk(1:nsrc_tot)
   755)       write(iulog,*) '-------------------------------------------------'
   756)       end if
   757) #ifdef USE_ESSL
   758)       call dgemm( 'N', 'N', nlev, 1, nsrc_tot, &
   759) 		  1._r8, fnorm, nlev, wrk, nw, &
   760) 		  0._r8, qrs, nlev )
   761) #else
   762)       qrs(:,1) = matmul( fnorm(:,1:nsrc_tot),wrk(1:nsrc_tot) )
   763) #endif
   764) !------------------------------------------------------------------------------
   765) !     ... o2 srb heating
   766) !------------------------------------------------------------------------------
   767)       do k = 1,nlev
   768)          wrk(1:nsrb_tot) = xs_o2srb(k,1:nsrb_tot)*wlintv(nsrc_tot+1:nsrc_tot+nsrb_tot) &
   769)                                                  *bde_o2_b(nsrc_tot+1:nsrc_tot+nsrb_tot)
   770) 	 qrs(k,2)        = dot_product( fnorm(k,nsrc_tot+1:nsrc_tot+nsrb_tot),wrk(1:nsrb_tot) )
   771)       end do
   772) 
   773)       if( do_diag ) then
   774)       write(iulog,*) '-------------------------------------------------'
   775)       write(iulog,*) 'jshort_hrates: lya,bde_o2_a,qrs(nlev) at long,lchnk = ',long,lchnk
   776)       write(iulog,'(1p,5g12.5)') jo2_lya(nlev),bde_o2_a(2),qrs(nlev,1)
   777)       write(iulog,*) '-------------------------------------------------'
   778)       end if
   779) !------------------------------------------------------------------------------
   780) !     ... total o2 heating
   781) !------------------------------------------------------------------------------
   782) !------------------------------------------------------------------------------
   783) !     ... Branch 1, O2 + hv => O(3P) + O(3P); wavelengths >175nm
   784) !------------------------------------------------------------------------------
   785)       qrs(:,2)  = qrs(:,2) + jo2_lya(:)*.47_r8*bde_o2_b(2)
   786) !------------------------------------------------------------------------------
   787) !     ... Branch 2, O2 + hv => O(3P) + O(1D);  wavelengths <175nm
   788) !------------------------------------------------------------------------------
   789)       qrs(:,1)  = qrs(:,1) + jo2_lya(:)*.53_r8*bde_o2_a(2)
   790)       if( do_diag ) then
   791)       write(iulog,*) '-------------------------------------------------'
   792)       write(iulog,*) 'jshort_hrates: o2(1),qrs(nlev) at long,lchnk = ',long,lchnk
   793)       write(iulog,'(1p,5g12.5)') o2_vmr(1),qrs(nlev,1)
   794)       write(iulog,*) '-------------------------------------------------'
   795)       end if
   796) 
   797) !------------------------------------------------------------------------------
   798) !	... the o3 heating rates
   799) !------------------------------------------------------------------------------
   800)       wrk(:) = xs_o3a(:)*wlintv(:)*bde_o3_a(:)
   801) #ifdef USE_ESSL
   802)       call dgemm( 'N', 'N', nlev, 1, nw, &
   803) 		  1._r8, fnorm, nlev, wrk, nw, &
   804) 		  0._r8, qrs(1,3), nlev )
   805) #else
   806)       qrs(:,3) = matmul( fnorm,wrk )
   807) #endif
   808)       wrk(:) = xs_o3b(:)*wlintv(:)*bde_o3_b(:)
   809) #ifdef USE_ESSL
   810)       call dgemm( 'N', 'N', nlev, 1, nw, &
   811) 		  1._r8, fnorm, nlev, wrk, nw, &
   812) 		  0._r8, qrs(1,4), nlev )
   813) #else
   814)       qrs(:,4) = matmul( fnorm,wrk )
   815) #endif
   816) 
   817) !------------------------------------------------------------------------------
   818) !	... form actual heating rates (k/s)
   819) !------------------------------------------------------------------------------
   820)       do k = 1,nlev
   821) 	 k1       = nlev - k + 1
   822)          hfactor  = avogad/(cparg(k1)*mw(k1))
   823)          qrs(k,1) = qrs(k,1)*hfactor*o2_vmr(k1)
   824)          qrs(k,2) = qrs(k,2)*hfactor*o2_vmr(k1)
   825)          qrs(k,3) = qrs(k,3)*hfactor*o3_vmr(k1)
   826)          qrs(k,4) = qrs(k,4)*hfactor*o3_vmr(k1)
   827)       end do
   828) 
   829)       deallocate( fnorm, trans_o2, trans_o3, wrk )
   830) 
   831)       end subroutine jshort_hrates

   896) 
   897) !------------------------------------------------------------------------------

   907) 	real(r8), intent(in)    :: zkm(nlev)		! Altitude, km
   908) 	real(r8), intent(out)   :: jo2_sht(nlev,branch) ! JO2, sec-1, <200nm
   909) 	real(r8), intent(out)   :: jno_sht(nlev)        ! JNO, sec-1, SRB
   910) 	real(r8), intent(out)   :: jsht(:,:)	        ! Additional J's
   911) 
   912) !------------------------------------------------------------------------------
   913) !     ... local variables
   914) !------------------------------------------------------------------------------
   915) 	integer :: k 			     ! Altitude index
   916) 	integer :: wn			     ! Wavelength index
   917) 	integer :: astat
   918)         integer :: nid(0:nlev)	             ! Number of layers crossed by the direct
   919)                                              ! beam when travelling from the top of the
   920) 			                     ! atmosphere to layer i; NID(i), i = 0..NZ-1
   921)         real(r8) :: hfactor
   922) 	real(r8) :: dsdh(0:nlev,nlev)        ! Slant path of direct beam through each
   923)                                              ! layer crossed  when travelling from the top of
   924) 				             ! the atmosphere to layer i; DSDH(i,j), i = 0.
   925) 					     ! NZ-1, j = 1..NZ-1   (see sphers.f)
   926)         real(r8), allocatable :: fnorm(:,:)      ! Normalized ETF
   927) 	real(r8), allocatable :: trans_o2(:,:)   ! Transmission o2 (total)
   928)         real(r8), allocatable :: trans_o3(:,:)   ! Transmission, ozone
   929) 	real(r8), allocatable :: wrk(:)	     ! wrk array
   930) 	real(r8) :: jo2_lya(nlev)	     ! Total photolytic rate constant for Ly alpha
   931) 	real(r8) :: jo2_srb(nlev)	     ! Total JO2  for SRB
   932) 	real(r8) :: jo2_src(nlev)            ! Total JO2 for SRC
   933) 	real(r8) :: delz(nlev)               ! layer thickness (cm)
   934) 	real(r8) :: o2scol(nlev)             ! O2 Slant Column
   935)         real(r8) :: o3scol(nlev)	     ! O3 Slant Column
   936)         real(r8) :: noscol(nlev)	     ! NO Slant Column
   937)         real(r8) :: rmla(nlev)		     ! Transmission, Lyman Alpha (other species)
   938) 	real(r8) :: ro2la(nlev)		     ! Transmission, Lyman Alpha (for JO2)
   939) 	real(r8) :: tlevmin(nlev)
   940) 	real(r8) :: abs_col(nlev)
   941)         real(r8) :: tsrb(nlev,nsrbtuv)       ! Transmission in the SRB
   942) 	real(r8) :: xs_o2srb(nlev,nsrbtuv)   ! Cross section * QY for O2 in SRB
   943) 
   944)       allocate( fnorm(nlev,nw),stat=astat )
   945)       if( astat /= 0 ) then
   946) 	 call alloc_err( astat, 'jshort_photo', 'fnorm', nw*nlev )
   947)       end if
   948)       allocate( trans_o2(nlev,nw),stat=astat )
   949)       if( astat /= 0 ) then
   950) 	 call alloc_err( astat, 'jshort_photo', 'trans_o2', nw*nlev )
   951)       end if
   952)       allocate( trans_o3(nlev,nw),stat=astat )
   953)       if( astat /= 0 ) then
   954) 	 call alloc_err( astat, 'jshort_photo', 'trans_o3', nw*nlev )
   955)       end if
   956)       allocate( wrk(nw),stat=astat )
   957)       if( astat /= 0 ) then
   958) 	 call alloc_err( astat, 'jshort_photo', 'wrk', nw )
   959)       end if
   960) 
   961) !------------------------------------------------------------------------------
   962) !     ... Derive Slant Path for Spherical Atmosphere
   963) !------------------------------------------------------------------------------
   964)       call sphers( nlev, zkm, zen, dsdh, nid )
   965) 
   966) !------------------------------------------------------------------------------
   967) !     ... Derive O2, O3, and NO Slant Column
   968) !------------------------------------------------------------------------------
   969)       delz(1:nlev-1) = km2cm*(zkm(1:nlev-1) - zkm(2:nlev))
   970)       call slant_col( nlev, delz, dsdh, nid, o2cc, o2scol )
   971)       call slant_col( nlev, delz, dsdh, nid, o3cc, o3scol )
   972)       call slant_col( nlev, delz, dsdh, nid, nocc, noscol )
   973) 
   974) !------------------------------------------------------------------------------
   975) !     ... Transmission due to ozone
   976) !------------------------------------------------------------------------------
   977)       do wn = 1,nw
   978)          abs_col(:)     = min( (xs_o3a(wn) + xs_o3b(wn))*o3scol(:),100._r8 )
   979) 	 trans_o3(:,wn) = exp( -abs_col(:) )
   980)       end do
   981) 
   982) !------------------------------------------------------------------------------
   983) !    ... Derive the cross section and transmission for
   984) !        molecular oxygen Lya, SRC, and SRB's
   985) !------------------------------------------------------------------------------
   986) !------------------------------------------------------------------------------
   987) !     ... Transmission due to molecular oxygen in the SRC
   988) !------------------------------------------------------------------------------
   989)       do wn = 1,nsrc_tot
   990)          abs_col(:) = min( xs_o2src(wn)*o2scol(:),100._r8 )
   991) 	 trans_o2(:,wn) = exp( -abs_col(:) )
   992)       end do
   993) 
   994) !------------------------------------------------------------------------------
   995) !     ... Transmission and cross sections due to O2 at lyman alpha
   996) !------------------------------------------------------------------------------
   997)       call lymana( nlev, o2scol, rmla, ro2la )
   998) 
   999) !------------------------------------------------------------------------------
  1000) !    ... Place lya reduction faction in transmission array
  1001) !    ... This must follow the SRC placement (above)
  1002) !------------------------------------------------------------------------------
  1003)       trans_o2(:,1) = rmla(:)
  1004) 
  1005) !------------------------------------------------------------------------------
  1006) !     ... Molecular Oxygen, SRB
  1007) !------------------------------------------------------------------------------
  1008) !     ... Koppers Grid (see Koppers and Murtagh, Ann. Geophys., 14, 68-79, 1996)
  1009) !        #    wl(i)       wl(i+1)
  1010) !        1   174.4        177.0
  1011) !        2   177.0        178.6
  1012) !        3   178.6        180.2
  1013) !        4   180.2        181.8
  1014) !        5   181.8        183.5
  1015) !        6   183.5        185.2
  1016) !        7   185.2        186.9
  1017) !        8   186.9        188.7
  1018) !        9   188.7        190.5
  1019) !        10  190.5        192.3
  1020) !        11  192.3        194.2
  1021) !        12  194.2        196.1
  1022) !        13  196.1        198.0
  1023) !        14  198.0        200.0  <<last wl bin for <200nm
  1024) !        ----------------------
  1025) !        15  200.0        202.0
  1026) !        16  202.0        204.1
  1027) !        17  204.1        205.8
  1028) !------------------------------------------------------------------------------
  1029) 
  1030) !------------------------------------------------------------------------------
  1031) !     ... Kopper SRB parameterization is used here
  1032) !------------------------------------------------------------------------------
  1033)       tlevmin(nlev:1:-1) = min( max( tlev(:),150._r8 ),350._r8 )
  1034)       call calc_o2srb( nlev, nid, o2scol, tlevmin, tsrb, xs_o2srb )
  1035) 
  1036) !------------------------------------------------------------------------------
  1037) !     ... Place Koppers SRB transmission (1-14) on user
  1038) !         grid #20 (wc=175.7nm)
  1039) !------------------------------------------------------------------------------
  1040)       do wn = 1,nsrb_tot
  1041)          trans_o2(:,wn+nsrc_tot) = tsrb(:,wn)
  1042)       end do
  1043) 
  1044) !------------------------------------------------------------------------------
  1045) !     ... Derive the normalize flux at each altitude,
  1046) !         corrected for O2 and O3 absorption
  1047) !------------------------------------------------------------------------------
  1048)       do wn = 1,nw                               ! nw = 33 (nsrb_tot+nsrc_tot)
  1049)          fnorm(:,wn) = etfphot(wn)*trans_o2(:,wn)*trans_o3(:,wn)
  1050)       end do
  1051) 
  1052) !------------------------------------------------------------------------------
  1053) !     ... Derive the O2 rate constant and apply branching ratio (QY)
  1054) !------------------------------------------------------------------------------
  1055) !     ... SRC and SRB QY
  1056) !         Longward  of 174.65 the product is O2 + hv => O(3P) + O(3P)
  1057) !         Shortward of 174.65 the product is O2 + hv => O(3P) + O(1D)
  1058) !         The QY is assumed to be unity in both wavelength ranges.
  1059) !
  1060) !     ... Lyman Alpha QY
  1061) !         O2 + hv -> O(3P) + O(3P) at Lyman Alpha has a QY = 0.47
  1062) !         O2 + hv -> O(3P) + O(1D) at Lyman Alpha has a QY = 0.53
  1063) !         Lacoursiere et al., J. Chem. Phys. 110., 1949-1958, 1999.
  1064) !------------------------------------------------------------------------------
  1065) !     ... Lyman Alpha
  1066) !------------------------------------------------------------------------------
  1067)       jo2_lya(:) = etfphot(1)*ro2la(:)*wlintv(1)
  1068) 
  1069)       wrk(1:nsrc_tot) = xs_o2src(1:nsrc_tot)*wlintv(1:nsrc_tot)
  1070)       wrk(1)          = 0._r8
  1071) !------------------------------------------------------------------------------
  1072) !     ... o2 src photolysis
  1073) !------------------------------------------------------------------------------
  1074) #ifdef USE_ESSL
  1075)       call dgemm( 'N', 'N', nlev, 1, nsrc_tot, &
  1076) 		  1._r8, fnorm, nlev, wrk, nw, &
  1077) 		  0._r8, jo2_src, nlev )
  1078) #else
  1079)       jo2_src(:) = matmul( fnorm(:,1:nsrc_tot),wrk(1:nsrc_tot) )
  1080) #endif
  1081) !------------------------------------------------------------------------------
  1082) !     ... o2 srb photolysis
  1083) !------------------------------------------------------------------------------
  1084)       do k = 1,nlev
  1085)          wrk(1:nsrb_tot) = xs_o2srb(k,1:nsrb_tot)*wlintv(nsrc_tot+1:nsrc_tot+nsrb_tot)
  1086) 	 jo2_srb(k)      = dot_product( fnorm(k,nsrc_tot+1:nsrc_tot+nsrb_tot),wrk(1:nsrb_tot) )
  1087)       end do
  1088) 
  1089) !------------------------------------------------------------------------------
  1090) !     ... total o2 photolysis
  1091) !------------------------------------------------------------------------------
  1092) !------------------------------------------------------------------------------
  1093) !     ... Branch 1, O2 + hv => O(3P) + O(3P); wavelengths >175nm
  1094) !------------------------------------------------------------------------------
  1095)       jo2_sht(:,1) = jo2_lya(:)*.47_r8 + jo2_srb(:)
  1096) !------------------------------------------------------------------------------
  1097) !     ... Branch 2, O2 + hv => O(3P) + O(1D);  wavelengths <175nm
  1098) !------------------------------------------------------------------------------
  1099)       jo2_sht(:,2) = jo2_lya(:)*.53_r8 + jo2_src(:)
  1100) 
  1101) !------------------------------------------------------------------------------
  1102) !     ... Derive the NO rate constant Minsch. and Siskind, JGR, 98, 20401, 1993
  1103) !------------------------------------------------------------------------------
  1104)       call calc_jno( nlev, etfphot_ms93, n2cc, o2scol, o3scol, &
  1105)                      noscol, jno_sht )
  1106) 
  1107) !------------------------------------------------------------------------------
  1108) !    ... Derive addtional rate constants for species with wl < 200 nm.!
  1109) !        Temperature dependence of the cross sections are not included in this
  1110) !        version.
  1111) !------------------------------------------------------------------------------
  1112) #if defined USE_ESSL
  1113)       call dgemm( 'N', 'N', nlev, nj, nw, &
  1114)  	          1._r8, fnorm, nlev, xs_wl, nw, &
  1115)  	          0._r8, jsht, nlev )
  1116) #else
  1117)       jsht(:,:) = matmul( fnorm,xs_wl )
  1118) #endif
  1119) 
  1120)       deallocate( fnorm, trans_o2, trans_o3, wrk )
  1121) 
  1122)       end subroutine jshort_photo

  1124)       subroutine sphers( nlev, z, zenith_angle, dsdh, nid )
  1125) !=============================================================================!
  1126) !   Subroutine sphers                                                         !
  1127) !=============================================================================!
  1128) !   PURPOSE:                                                                  !
  1129) !   Calculate slant path over vertical depth ds/dh in spherical geometry.     !
  1130) !   Calculation is based on:  A.Dahlback, and K.Stamnes, A new spheric model  !
  1131) !   for computing the radiation field available for photolysis and heating    !
  1132) !   at twilight, Planet.Space Sci., v39, n5, pp. 671-683, 1991 (Appendix B)   !
  1133) !=============================================================================!
  1134) !   PARAMETERS:                                                               !
  1135) !   NZ      - INTEGER, number of specified altitude levels in the working (I) !
  1136) !             grid                                                            !
  1137) !   Z       - REAL, specified altitude working grid (km)                  (I) !
  1138) !   ZEN     - REAL, solar zenith angle (degrees)                          (I) !
  1139) !   DSDH    - REAL, slant path of direct beam through each layer crossed  (O) !
  1140) !             when travelling from the top of the atmosphere to layer i;      !
  1141) !             DSDH(i,j), i = 0..NZ-1, j = 1..NZ-1                             !
  1142) !   NID     - INTEGER, number of layers crossed by the direct beam when   (O) !
  1143) !             travelling from the top of the atmosphere to layer i;           !
  1144) !             NID(i), i = 0..NZ-1                                             !
  1145) !=============================================================================!
  1146) !   EDIT HISTORY:                                                             !
  1147) !   Original: Taken By Doug Kinnison from Sasha Madronich, TUV Code, V4.1a,   !
  1148) !             on 1/1/02                                                       !
  1149) !=============================================================================!
  1150) 
  1151)         use physconst,    only : rearth
  1152) 
  1153)       implicit none
  1154) 
  1155) !------------------------------------------------------------------------------
  1156) !       ... Dummy arguments
  1157) !------------------------------------------------------------------------------
  1158)       integer, intent(in)  :: nlev              ! number model vertical levels
  1159)       integer, intent(out) :: nid(0:nlev)       ! see above
  1160)       real(r8), intent (in) :: zenith_angle		! zenith_angle
  1161)       real(r8), intent (in) :: z(nlev)		! geometric altitude (km)
  1162)       real(r8), intent (out) :: dsdh(0:nlev,nlev)   ! see above
  1163) 
  1164) 
  1165) !------------------------------------------------------------------------------
  1166) !       ... Local variables
  1167) !------------------------------------------------------------------------------
  1168)       real(r8) :: radius
  1169)       real(r8) :: re
  1170)       real(r8) :: zenrad
  1171)       real(r8) :: rpsinz
  1172)       real(r8) :: const0
  1173)       real(r8) :: rj
  1174)       real(r8) :: rjp1
  1175)       real(r8) :: dsj
  1176)       real(r8) :: dhj
  1177)       real(r8) :: ga
  1178)       real(r8) :: gb
  1179)       real(r8) :: sm
  1180)       real(r8) :: zd(0:nlev-1)
  1181) 
  1182)       integer :: i
  1183)       integer :: j
  1184)       integer :: k
  1185)       integer :: id
  1186)       integer :: nlayer
  1187) 
  1188) 
  1189)       radius = rearth*1.e-3_r8   ! radius earth (km)
  1190) 
  1191) !------------------------------------------------------------------------------
  1192) !       ... set zenith angle in radians
  1193) !------------------------------------------------------------------------------
  1194)       zenrad = zenith_angle*d2r
  1195)       const0 = sin( zenrad )
  1196) 
  1197) !------------------------------------------------------------------------------
  1198) !       ... set number of layers:
  1199) !------------------------------------------------------------------------------
  1200)       nlayer = nlev - 1
  1201) 
  1202) !------------------------------------------------------------------------------
  1203) !       ... include the elevation above sea level to the radius of the earth:
  1204) !------------------------------------------------------------------------------
  1205)       re = radius + z(nlev)
  1206) 
  1207) !------------------------------------------------------------------------------
  1208) !       ... inverse coordinate of z
  1209) !------------------------------------------------------------------------------
  1210)       do k = 0,nlayer
  1211)         zd(k) = z(k+1) - z(nlev)
  1212)       end do
  1213) 
  1214) !------------------------------------------------------------------------------
  1215) !       ... initialize dsdh(i,j), nid(i)
  1216) !------------------------------------------------------------------------------
  1217)       nid(:) = 0
  1218)       do j = 1,nlev
  1219)         dsdh(:,j) = 0._r8
  1220)       end do
  1221) 
  1222) !------------------------------------------------------------------------------
  1223) !       ... calculate ds/dh of every layer
  1224) !------------------------------------------------------------------------------
  1225)       do i = 0,nlayer
  1226)         rpsinz = (re + zd(i)) * const0
  1227)         if( zenith_angle <= 90._r8 .or. rpsinz >= re ) then
  1228) !------------------------------------------------------------------------------
  1229) ! Find index of layer in which the screening height lies
  1230) !------------------------------------------------------------------------------
  1231)            id = i 
  1232)            if( zenith_angle > 90._r8 ) then
  1233)               do j = 1,nlayer
  1234)                  if( rpsinz < (zd(j-1) + re) .and.  rpsinz >= (zd(j) + re) ) then
  1235) 		    id = j
  1236) 		    exit
  1237) 		 end if
  1238)               end do
  1239)            end if
  1240)  
  1241)            do j = 1,id
  1242)              sm = 1._r8
  1243)              if( j == id .and. id == i .and. zenith_angle > 90._r8 ) then
  1244)                 sm = -1._r8
  1245)              end if
  1246)              rj   = re + zd(j-1)
  1247)              rjp1 = re + zd(j)
  1248)              dhj  = zd(j-1) - zd(j)
  1249)              ga   = max( rj*rj - rpsinz*rpsinz,0._r8 )
  1250)              gb   = max( rjp1*rjp1 - rpsinz*rpsinz,0._r8 )
  1251)              if( id > i .and. j == id ) then
  1252)                 dsj = sqrt( ga )
  1253)              else
  1254)                 dsj = sqrt( ga ) - sm*sqrt( gb )
  1255)              end if
  1256)              dsdh(i,j) = dsj / dhj
  1257)            end do
  1258)            nid(i) = id
  1259)         else
  1260)            nid(i) = -1
  1261)         end if
  1262)       end do
  1263) 
  1264)       end subroutine sphers

  1266)       subroutine slant_col( nlev, delz, dsdh, nid, absden, scol )
  1267) !=============================================================================!
  1268) !   PURPOSE:                                                                  !
  1269) !   Derive Column
  1270) !=============================================================================!
  1271) !   PARAMETERS:                                                               !
  1272) !   NLEV   - INTEGER, number of specified altitude levels in the working  (I) !
  1273) !            grid                                                             !
  1274) !   DELZ   - REAL, specified altitude working grid (km)                   (I) !
  1275) !   DSDH   - REAL, slant path of direct beam through each layer crossed  (O)  !
  1276) !             when travelling from the top of the atmosphere to layer i;      !
  1277) !             DSDH(i,j), i = 0..NZ-1, j = 1..NZ-1                             !
  1278) !   NID    - INTEGER, number of layers crossed by the direct beam when   (O)  !
  1279) !             travelling from the top of the atmosphere to layer i;           !
  1280) !             NID(i), i = 0..NZ-1                                             !
  1281) !            specified altitude at each specified wavelength                  !
  1282) !   absden - REAL, absorber concentration, molecules cm-3                     !
  1283) !   SCOL   - REAL, absorber Slant Column, molecules cm-2                      !
  1284) !=============================================================================!
  1285) !   EDIT HISTORY:                                                             !
  1286) !   09/01  Read in profile from an input file, DEK                            !
  1287) !   01/02  Taken from Sasha Madronich's TUV code                              !
  1288) !=============================================================================!
  1289) 
  1290)       implicit none
  1291) 
  1292) !------------------------------------------------------------------------------
  1293) !       ... Dummy arguments
  1294) !------------------------------------------------------------------------------
  1295)       integer, intent(in) :: nlev
  1296)       integer, intent(in) :: nid(0:nlev)                ! see above
  1297)       real(r8), intent(in)    :: delz(nlev)	        ! layer thickness (cm)
  1298)       real(r8), intent(in)    :: dsdh(0:nlev,nlev)	! see above
  1299)       real(r8), intent(in)    :: absden(nlev)           ! absorber concentration (molec. cm-3)
  1300)       real(r8), intent(out)   :: scol(nlev)		! absorber Slant Column (molec. cm-2)
  1301) 
  1302) !------------------------------------------------------------------------------
  1303) !       ... Local variables
  1304) !------------------------------------------------------------------------------
  1305)       real(r8), parameter :: largest = 1.e+36_r8
  1306) 
  1307)       real(r8) :: sum
  1308)       real(r8) :: hscale
  1309)       real(r8) :: numer, denom
  1310)       real(r8) :: cz(nlev)
  1311) 
  1312)       integer :: id
  1313)       integer :: j
  1314)       integer :: k
  1315) 
  1316) !------------------------------------------------------------------------------
  1317) !     ... compute column increments (logarithmic integrals)
  1318) !------------------------------------------------------------------------------
  1319)       do k = 1,nlev-1
  1320) 	if( absden(k) /= 0._r8 .and. absden(k+1) /= 0._r8 ) then
  1321)            cz(nlev-k) = (absden(k) - absden(k+1))/log( absden(k)/absden(k+1) ) * delz(k)
  1322) 	else
  1323)            cz(nlev-k) = .5_r8*(absden(k) + absden(k+1)) * delz(k)
  1324) 	end if
  1325)       end do
  1326) 
  1327) !------------------------------------------------------------------------------
  1328) !     ... Include exponential tail integral from infinity to model top
  1329) !         specify scale height near top of data.For WACCM-X model, scale
  1330) !         height needs to be increased for higher model top
  1331) !------------------------------------------------------------------------------
  1332)       if (nlev==pver) then
  1333)          if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
  1334)            hscale     = 20.e5_r8
  1335)          else
  1336)            hscale     = 10.e5_r8
  1337)          endif
  1338)          cz(nlev-1) = cz(nlev-1) + hscale * absden(1)
  1339)       endif
  1340) 
  1341) !------------------------------------------------------------------------------
  1342) !       ...  Calculate vertical and slant column from each level:
  1343) !            work downward
  1344) !------------------------------------------------------------------------------
  1345)       do id = 0,nlev-1
  1346)          sum = 0._r8
  1347)          if( nid(id) >= 0 ) then
  1348) !------------------------------------------------------------------------------
  1349) !       ...  Single pass layers:
  1350) !------------------------------------------------------------------------------
  1351)             do j = 1, min(nid(id), id)
  1352)                sum = sum + cz(nlev-j)*dsdh(id,j)
  1353)             end do
  1354) !------------------------------------------------------------------------------
  1355) !       ...  Double pass layers:
  1356) !------------------------------------------------------------------------------
  1357)             do j = min(nid(id),id)+1, nid(id)
  1358)                sum = sum + 2._r8*cz(nlev-j)*dsdh(id,j)
  1359)             end do
  1360)          else
  1361)             sum = largest
  1362)          end if
  1363)          scol(nlev-id) = sum
  1364)       end do
  1365)       scol(nlev) = .95_r8*scol(nlev-1)
  1366) 
  1367)       end subroutine slant_col

  1369)       subroutine lymana( nlev, o2scol, rm, ro2 )
  1370) !-----------------------------------------------------------------------------!
  1371) !   PURPOSE:                                                                  !
  1372) !   Calculate the effective absorption cross section of O2 in the Lyman-Alpha !
  1373) !   bands and an effective O2 optical depth at all altitudes.  Parameterized  !
  1374) !   after:  Chabrillat, S., and G. Kockarts, Simple parameterization of the   !
  1375) !   absorption of the solar Lyman-Alpha line, Geophysical Research Letters,   !
  1376) !   Vol.24, No.21, pp 2659-2662, 1997.                                        !
  1377) !-----------------------------------------------------------------------------!
  1378) !   PARAMETERS:                                                               !
