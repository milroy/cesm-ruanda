   423) subroutine find_group_name(unit, group, status)
   424) 
   425) !---------------------------------------------------------------------------------------
   426) ! Purpose: 
   427) ! Search a file that contains namelist input for the specified namelist group name.
   428) ! Leave the file positioned so that the current record is the first record of the
   429) ! input for the specified group.
   430) ! 
   431) ! Method: 
   432) ! Read the file line by line.  Each line is searched for an '&' which may only
   433) ! be preceded by blanks, immediately followed by the group name which is case
   434) ! insensitive.  If found then backspace the file so the current record is the
   435) ! one containing the group name and return success.  Otherwise return -1.
   436) !
   437) ! Author:  B. Eaton, August 2007
   438) !---------------------------------------------------------------------------------------
   439) 
   440)    integer,          intent(in)  :: unit     ! fortran unit attached to file
   441)    character(len=*), intent(in)  :: group    ! namelist group name
   442)    integer,          intent(out) :: status   ! 0 for success, -1 if group name not found
   443) 
   444)    ! Local variables
   445) 
   446)    integer           :: len_grp
   447)    integer           :: ios    ! io status
   448)    character(len=80) :: inrec  ! first 80 characters of input record
   449)    character(len=80) :: inrec2 ! left adjusted input record
   450)    character(len=len(group)) :: lc_group
   451) 
   452)    !---------------------------------------------------------------------------
   453) 
   454)    len_grp = len_trim(group)
   455)    lc_group = to_lower(group)
   456) 
   457)    ios = 0
   458)    do while (ios <= 0)
   459) 
   460)       read(unit, '(a)', iostat=ios, end=102) inrec
   461) 
   462)       if (ios <= 0) then  ! ios < 0  indicates an end of record condition
   463) 
   464)          ! look for group name in this record
   465) 
   466)          ! remove leading blanks
   467)          inrec2 = to_lower(adjustl(inrec))
   468) 
   469)          ! check for leading '&'
   470)          if (inrec2(1:1) == '&') then
   471) 
   472)             ! check for case insensitive group name
   473)             if (trim(lc_group) == inrec2(2:len_grp+1)) then
   474) 
   475)                ! found group name.  backspace to leave file position at this record
   476)                backspace(unit)
   477)                status = 0
   478)                return
   479) 
   480)             end if
   481)          end if
   482)       end if
   483) 
   484)    end do
   485) 
   486)    102 continue  ! end of file processing
   487)    status = -1
   488) 
   489) end subroutine find_group_name

   495) function to_lower(str)
   496) 
   497) !----------------------------------------------------------------------- 
   498) ! Purpose: 
   499) ! Convert character string to lower case.
   500) ! 
   501) ! Method: 
   502) ! Use achar and iachar intrinsics to ensure use of ascii collating sequence.
   503) !
   504) ! Author:  B. Eaton, July 2001
   505) !     
   506) ! $Id$
   507) !----------------------------------------------------------------------- 
   508)    implicit none
   509) 
   510)    character(len=*), intent(in) :: str      ! String to convert to lower case
   511)    character(len=len(str))      :: to_lower
   512) 
   513) ! Local variables
   514) 
   515)    integer :: i                ! Index
   516)    integer :: aseq             ! ascii collating sequence
   517)    integer :: upper_to_lower   ! integer to convert case
   518)    character(len=1) :: ctmp    ! Character temporary
   519) !-----------------------------------------------------------------------
   520)    upper_to_lower = iachar("a") - iachar("A")
   521) 
   522)    do i = 1, len(str)
   523)       ctmp = str(i:i)
   524)       aseq = iachar(ctmp)
   525)       if ( aseq >= iachar("A") .and. aseq <= iachar("Z") ) &
   526)            ctmp = achar(aseq + upper_to_lower)	
   527)       to_lower(i:i) = ctmp
   528)    end do
   529) 
   530) end function to_lower
