   319)   subroutine check_files()
   320) 
   321)     use shr_sys_mod, only: shr_sys_system
   322)     implicit none
   323) 
   324) !-----------------------------------------------------------------------
   325) ! 	... local variables
   326) !-----------------------------------------------------------------------
   327)     character(len=128) :: ctmp
   328)     integer ::  istat
   329) 
   330) 
   331)     if (next_mod_time > curr_data_times(size(curr_data_times))) then
   332)        if ( .not. associated(next_data_times) ) then
   333)           ! open next file...
   334)           next_filename = incr_filename( curr_filename )
   335)           call open_spe_datafile( next_filename, next_fileid, next_data_times )
   336)        endif
   337)     endif
   338) 
   339)     if ( associated(next_data_times) ) then
   340)        if (curr_mod_time >= next_data_times(1)) then
   341) 
   342)           ! close current file ...
   343)           call pio_closefile( curr_fileid )
   344)           ! remove if requested
   345)           if(masterproc) then
   346)              if( remove_spe_file ) then
   347)                 write(iulog,*) 'check_files: removing file = ',trim(curr_filename) 
   348)                 ctmp = 'rm -f ' // trim(curr_filename) 
   349)                 write(iulog,*) 'check_files: fsystem issuing command - '
   350)                 write(iulog,*) trim(ctmp)
   351)                 call shr_sys_system( ctmp, istat )
   352)              end if
   353)           endif
   354) 
   355)           curr_filename = next_filename
   356)           curr_fileid = next_fileid
   357) 
   358)           deallocate( curr_data_times )
   359)           allocate( curr_data_times( size( next_data_times ) ) )
   360)           curr_data_times(:) = next_data_times(:)
   361) 
   362)           next_filename = ''
   363) 
   364)           deallocate( next_data_times )
   365)           nullify(  next_data_times )
   366) 
   367)        endif
   368)     endif

   374)   function incr_filename( filename )
   375) 
   376)     !-----------------------------------------------------------------------
   377)     ! 	... Increment or decrement a date string withing a filename
   378)     !           the filename date section is assumed to be of the form
   379)     !           yyyy-dd-mm
   380)     !-----------------------------------------------------------------------
   381) 
   382)     use string_utils,  only : incstr
   383) 
   384)     implicit none
   385) 
   386) 
   387)     character(len=*), intent(in) :: filename ! present dynamical dataset filename
   388)     character(len=shr_kind_cl) :: incr_filename      ! next filename in the sequence
   389) 
   390)     ! set new next_filename ...
   391) 
   392)     !-----------------------------------------------------------------------
   393)     !	... local variables
   394)     !-----------------------------------------------------------------------
   395)     integer :: pos, pos1, istat
   396)     character(len=shr_kind_cl) :: fn_new, line
   397)     character(len=6)   :: seconds
   398)     character(len=5)   :: num
   399) 
   400)     integer :: ios
   401) 
   402)     if ( len_trim(filenames_list) .eq. 0) then
   403) 
   404)        ! increment the number in the filename to the next number
   405)        pos = len_trim( filename )
   406)        fn_new = filename(:pos)
   407)        write(iulog,*) 'incr_flnm: old filename = ',trim(fn_new)
   408)        if( fn_new(pos-2:) == '.nc' ) then
   409)           pos = pos - 3
   410)        end if
   411)        istat = incstr( fn_new(:pos), 1 )
   412)        if( istat /= 0 ) then
   413)           write(iulog,*) 'incr_flnm: incstr returned ', istat
   414)           write(iulog,*) '           while trying to decrement ',trim( fn_new )
   415)           call endrun
   416)        end if
   417) 
   418)     else
   419) 
   420)        ! open filenames_list
   421)        write(iulog,*) 'incr_flnm: old filename = ',trim(filename)
   422)        write(iulog,*) 'incr_flnm: open filenames_list : ',filenames_list 
   423)        open( unit=9, file=filenames_list, iostat=ios, status="OLD")
   424)        if (ios /= 0) then
   425)           call endrun('not able to open filenames_list file: '//filenames_list)
   426)        endif
   427) 
   428)        ! read file names
   429)        read( unit=9, fmt='(A)', iostat=ios ) line 
   430)        if (ios /= 0) then
   431)           call endrun('not able to increment file name from filenames_list file: '//filenames_list)
   432)        endif
   433)        do while( trim(line) /= trim(filename) )
   434)           read( unit=9, fmt='(A)', iostat=ios ) line 
   435)           if (ios /= 0) then
   436)              call endrun('not able to increment file name from filenames_list file: '//filenames_list)
   437)           endif
   438)        enddo
   439) 
   440)        read( unit=9, fmt='(A)', iostat=ios ) line 
   441)        if (ios /= 0) then
   442)           call endrun('not able to increment file name from filenames_list file: '//filenames_list)
   443)        endif
   444)        fn_new = trim(line)
   445) 
   446)        close(unit=9)
   447) 
   448)     endif
   449) 
   450)     incr_filename = trim(fn_new)
   451)     write(iulog,*) 'incr_flnm: new filename = ',incr_filename
   452) 
   453)   end function incr_filename

   457)   subroutine find_times( itms, fids, datatm, datatp, time )
   458) 
   459)     implicit none
   460) 
   461) !------------------------------------------------------------------------------
   462) !	... dummy arguments
   463) !------------------------------------------------------------------------------
   464)     integer, intent(out) :: itms(2) ! record numbers that bracket time
   465)     type(file_desc_t), intent(out) :: fids(2) ! ids of files that contains these recs
   466)     real(r8), intent(in) :: time    ! time of interest
   467)     real(r8), intent(out):: datatm, datatp
   468) 
   469) !------------------------------------------------------------------------------
   470) !	... local variables
   471) !------------------------------------------------------------------------------
   472)     integer np1        ! current forward time index of dataset
   473)     integer n,i      ! 
   474)     integer :: curr_tsize, next_tsize, all_tsize
   475)     real(r8), allocatable, dimension(:):: all_data_times
   476)     logical :: found_time
   477) 
   478)     curr_tsize = size(curr_data_times)
   479)     next_tsize = 0
   480)     if ( associated(next_data_times)) next_tsize = size(next_data_times)
   481) 
   482)     all_tsize = curr_tsize + next_tsize
   483) 
   484)     allocate( all_data_times( all_tsize ) )
   485) 
   486)     all_data_times(:curr_tsize) = curr_data_times(:)
   487)     if (next_tsize > 0) all_data_times(curr_tsize+1:all_tsize) = next_data_times(:)
   488) 
   489)     ! find bracketing times 
   490)     found_time = .false.
   491)     do n = 1,all_tsize-1
   492)        np1 = n + 1
   493)        datatm = all_data_times(n)
   494)        datatp = all_data_times(np1)
   495)        if ( (time .ge. datatm) .and. (time .le. datatp) ) then
   496)           found_time = .true.
   497)           exit
   498)        endif
   499)     enddo
   500)   
   501)     if( found_time ) then
   502)        deallocate( all_data_times )
   503)   
   504)        itms(1) = n
   505)        itms(2) = np1
   506)        fids(:) = curr_fileid
   507)   
   508)        do i=1,2
   509)           if ( itms(i) > curr_tsize ) then 
   510)              itms(i) = itms(i) - curr_tsize 
   511)              fids(i) = next_fileid
   512)           endif
   513)        enddo
   514)     else
   515)        write(iulog,*)'FIND_TIMES: Failed to find dates bracketing desired time =', time
   516)        write(iulog,*)' datatm = ',datatm
   517)        write(iulog,*)' datatp = ',datatp
   518)        write(iulog,*)' all_data_times = ',all_data_times
   519)        call endrun
   520)     end if
   521) 
   522)   end subroutine find_times

   560)   subroutine interpolate_spedata()
   561) 
   562)     implicit none
   563) 
   564)     real(r4) fact1, fact2
   565)     real(r8) deltat 
   566) 
   567)     call t_startf('MET__interpolate_spedata')
   568) 
   569)     deltat = datatimep - datatimem
   570) 
   571)     fact1 = (datatimep - curr_mod_time)/deltat
   572)     fact2 = 1._r8-fact1
   573) 
   574)     ionpairs(:) = fact1*ionpairs_i(nm)%data(:) + fact2*ionpairs_i(np)%data(:)
   575) 
   576)     call t_stopf('MET__interpolate_spedata')
   577)

   658)   function get_time_float( year, month, day, sec )
   659) 
   660) 
   661) ! returns float representation of time -- number of days
   662) ! since 1 jan 0001 00:00:00.000
   663) 
   664)     implicit none
   665) 
   666)     integer, intent(in) :: year, month, day
   667)     integer, intent(in) :: sec
   668)     real(r8) :: get_time_float
   669) 
   670) ! ref date is 1 jan 0001
   671) 
   672)     integer  :: refyr, refmn, refdy
   673)     real(r8) :: refsc, fltdy
   674)     integer  :: doy(12)
   675) 
   676) !              jan feb mar apr may jun jul aug sep oct nov dec
   677) !              31  28  31  30  31  30  31  31  31  31  30  31
   678)     data doy /  1, 32, 60, 91,121,152,182,213,244,274,305,335 /
   679) 
   680)     refyr = 1
   681)     refmn = 1
   682)     refdy = 1
   683)     refsc = 0._r8

   685)     if ( timemgr_is_caltype(trim(shr_cal_gregorian))) then
   686)        fltdy = greg2jday(year, month, day) - greg2jday(refyr,refmn,refdy)
   687)     else ! assume no_leap (all years are 365 days)
   688)        fltdy = (year - refyr)*365._r8 + &
   689)                (doy(month)-doy(refmn)) + &
   690)                (day-refdy) 
   691)     endif

   697) !-----------------------------------------------------------------------
   698) ! 	... Return Julian day number given Gregorian date.
   699) !
   700) ! Algorithm from Hatcher,D.A., Simple Formulae for Julian Day Numbers
   701) ! and Calendar Dates, Q.Jl.R.astr.Soc. (1984) v25, pp 53-55.
   702) !-----------------------------------------------------------------------
   703)   function greg2jday( year, month, day )
   704) 
   705)     implicit none
   706) 
   707)     integer, intent(in) :: year, month, day
   708)     integer :: greg2jday
   709) 
   710)     !-----------------------------------------------------------------------
   711)     !	... Local variables
   712)     !-----------------------------------------------------------------------
   713)     integer :: ap, mp
   714)     integer :: y, d, n, g
   715) 
   716)     !-----------------------------------------------------------------------
   717)     !     	... Modify year and month numbers
   718)     !-----------------------------------------------------------------------
   719)     ap = year - (12 - month)/10
   720)     mp = MOD( month-3,12 )
   721)     if( mp < 0 ) then
   722)        mp = mp + 12
   723)     end if
   724) 
   725)     !-----------------------------------------------------------------------
   726)     !     	... Julian day
   727)     !-----------------------------------------------------------------------
   728)     y = INT( 365.25_r8*( ap + 4712 ) )
   729)     d = INT( 30.6_r8*mp + .5_r8 )
   730)     n = y + d + day  + 59
   731)     g = INT( .75_r8*INT( ap/100 + 49 ) ) - 38
   732)     greg2jday = n - g
   733)
