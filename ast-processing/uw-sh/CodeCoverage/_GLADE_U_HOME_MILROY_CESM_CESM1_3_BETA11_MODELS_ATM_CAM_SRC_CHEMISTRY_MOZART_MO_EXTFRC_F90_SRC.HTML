   291)   subroutine extfrc_set( lchnk, zint, frcing, ncol )
   292) 
   293)     !--------------------------------------------------------

   295)     !--------------------------------------------------------
   296)

   299)     !--------------------------------------------------------

   306) 
   307)     !--------------------------------------------------------
   308)     !	... local variables
   309)     !--------------------------------------------------------
   310)     integer  ::  i, m, n
   311)     character(len=16) :: xfcname
   312)     real(r8) :: frcing_col(1:ncol)
   313)     integer  :: k, isec
   314)     real(r8),parameter :: km_to_cm = 1.e5_r8
   315) 
   316)     if( extfrc_cnt < 1 .or. extcnt < 1 ) then
   317)        return
   318)     end if
   319) 
   320)     !--------------------------------------------------------
   321)     !	... set non-zero forcings
   322)     !--------------------------------------------------------
   323)     src_loop : do m = 1,extfrc_cnt
   324) 
   325)        n = forcings(m)%frc_ndx
   326) 
   327)        frcing(:ncol,:,n) = 0._r8
   328)        do isec = 1,forcings(m)%nsectors
   329)           frcing(:ncol,:,n) = frcing(:ncol,:,n) + forcings(m)%fields(isec)%data(:ncol,:,lchnk)
   330)        enddo
   331) 
   332)        xfcname = trim(forcings(m)%species)//'_XFRC'
   333)        call outfld( xfcname, frcing(:ncol,:,n), ncol, lchnk )
   334) 
   335)        frcing_col(:ncol) = 0._r8
   336)        do k = 1,pver
   337)           frcing_col(:ncol) = frcing_col(:ncol) + frcing(:ncol,k,n)*(zint(:ncol,k)-zint(:ncol,k+1))*km_to_cm
   338)        enddo
   339)        xfcname = trim(forcings(m)%species)//'_CLXF'
   340)        call outfld( xfcname, frcing_col(:ncol), ncol, lchnk )
   341) 
   342)     end do src_loop
