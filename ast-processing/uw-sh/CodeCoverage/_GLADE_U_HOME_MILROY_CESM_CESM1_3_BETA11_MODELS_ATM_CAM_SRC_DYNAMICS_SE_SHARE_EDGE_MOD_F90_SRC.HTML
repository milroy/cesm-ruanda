   214)     ! Notes about the buf_ptr/receive_ptr options:

   218)     ! are associated, their targets will be used as buffer space.

   222)     ! If an EdgeBuffer_t object is initialized from pre-existing storage

   224)     ! and must not be used if the underlying storage has been deallocated.
   225)     !
   226)     ! All these restrictions also applied to the old newbuf and newreceive
   227)     ! options.
   228) 
   229)     ! Workaround for NAG bug.
   230)     ! NAG 5.3.1 dies if you use pointer bounds remapping to set
   231)     ! a pointer that is also a component. So remap to temporary,
   232)     ! then use that to set component pointer.

   242) 
   243) !$OMP BARRIER
   244) 
   245) !   only master thread should allocate the buffer
   246) #if (! defined ELEMENT_OPENMP)
   247) !$OMP MASTER
   248) #endif
   249)     if (present(buf_ptr)) then
   250)        ! If buffer is passed in but not allocated, allocate it.
   251)        if (.not. associated(buf_ptr)) allocate(buf_ptr(nlyr*nbuf))
   252)        ! Verify dimensions
   253)        if (size(buf_ptr) < nlyr*nbuf) then
   254)           print *,'size(buf_ptr),nlyr,nbuf=',size(buf_ptr),nlyr,nbuf
   255)           call abortmp('Error: user provided edge buffer is too small')
   256)        end if

   258)        tmp_ptr(1:nlyr,1:nbuf) => buf_ptr
   259)        edge%buf => tmp_ptr
   260) #else
   261)        ! call F77 routine which will reshape array.
   262)        call remap_2D_ptr_buf(edge,nlyr,nbuf,buf_ptr)
   263) #endif
   264)     else
   265)        allocate(edge%buf    (nlyr,nbuf))
   266)     end if
   267)

   277)        tmp_ptr(1:nlyr,1:nbuf) => receive_ptr
   278)        edge%receive => tmp_ptr
   279) #else
   280)        ! call F77 routine which will reshape array.
   281)        call remap_2D_ptr_receive(edge,nlyr,nbuf,receive_ptr)
   282) #endif
   283)     else
   284)        allocate(edge%receive(nlyr,nbuf))

   293) !$OMP BARRIER
   294) #endif

   307)        !$OMP MASTER
   308) #endif
   309)        if (.not. associated(edgebuff_ptrs(0)%ptr, edgebuff_ptrs(1)%ptr)) then

   313)        !$OMP END MASTER
   314) #endif

   751)     use dimensions_mod, only : np, max_corner_elem
   752)     use control_mod, only : north, south, east, west, neast, nwest, seast, swest
   753)     type (EdgeBuffer_t),         intent(in)  :: edge
   754) 
   755)     integer,               intent(in)  :: vlyr

   757)     integer,               intent(in)  :: kptr

   863)         call haltmp("edgeVunpackVert should not be called with unstructured meshes")
   864)     end if

   869)     iw=desc%getmapP(west)
   870)

   875)        v(3,i ,np)%x = edge%buf(1,in+i)

   878)        ! South
   879)        v(2,i ,1)%x  = edge%buf(1,is+i)

   890)        v(1,i ,1)%x  = edge%buf(1,is+i)
