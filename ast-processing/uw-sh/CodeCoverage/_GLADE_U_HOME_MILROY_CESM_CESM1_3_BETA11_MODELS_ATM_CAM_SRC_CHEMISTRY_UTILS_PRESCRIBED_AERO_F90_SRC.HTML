     1) !-------------------------------------------------------------------
     2) ! Manages reading and interpolation of prescribed aerosol concentrations.
     3) ! This places the concentration fields in the physics buffer so that
     4) ! radiation package can access them.
     5) !
     6) ! This has been generalized so that the field names in the data files
     7) ! and the field names in the physics buffer can be arbitrary.
     8) !
     9) ! The prescribed_aero_specifier namelist variable specifies a list of 
    10) ! variable names of the concentration fields in the netCDF dataset (ncdf_fld_name)
    11) ! and the corresponding names used in the physics buffer:
    12) !
    13) ! prescribed_aero_specifier = 'pbuf_name1:ncdf_fld_name1','pbuf_name2:ncdf_fld_name2', ...
    14) !
    15) ! If there is no ":" then the specified name is used as both the 
    16) ! pbuf_name and ncdf_fld_name
    17) !
    18) ! Created by: Francis Vitt
    19) !-------------------------------------------------------------------

    98) !-------------------------------------------------------------------
    99) ! parses prescribed_aero_specifier namelist option
   100) !-------------------------------------------------------------------
   101)   subroutine prescribed_aero_init()
   102) 
   103)     use tracer_data, only : trcdata_init
   104)     use cam_history, only : addfld, phys_decomp
   105)     use ppgrid,      only : pver
   106)     use error_messages, only: handle_err
   107)     use ppgrid,         only: pcols, pver, begchunk, endchunk
   108)     use physics_buffer, only : physics_buffer_desc
   109) 
   110)     implicit none
   111) 
   112)     ! local vars
   113)     character(len=32)  :: spec_a
   114)     integer :: ndx, istat, i, i_c, j
   115)     
   116)     if ( has_prescribed_aero ) then
   117)        if ( masterproc ) then
   118)           write(iulog,*) 'aero is prescribed in :'//trim(filename)
   119)        endif
   120)     else
   121)        return
   122)     endif
   123) 
   124) 
   125)     allocate (file%in_pbuf(size(specifier)))
   126)     if (clim_modal_aero) then
   127)       file%in_pbuf(:) = .false.
   128)       do i = 1,N_AERO
   129)           do j=1,aero_cnt
   130)               if(specifier(i) .eq. pbuf_names(j)) then
   131)                   file%in_pbuf(i) = .true.
   132)                   exit
   133)               endif
   134)           enddo
   135)       enddo
   136)     else
   137)       file%in_pbuf(:) = .true.
   138)     endif
   139)     call trcdata_init( specifier, filename, filelist, datapath, fields, file, &
   140)                        rmv_file, cycle_yr, fixed_ymd, fixed_tod, datatype)
   141)         
   142)     number_flds = 0
   143)     if (associated(fields)) number_flds = size( fields )
   144) 
   145)     if( number_flds < 1 ) then
   146)        if ( masterproc ) then
   147)           write(iulog,*) 'There are no prescribed aerosols'
   148)           write(iulog,*) ' '
   149)        endif
   150)        return
   151)     end if
   152)     
   153)     ! Following loop add fields for output. For modal aerosols, first the cld borne aersols
   154)     ! are added and then their interstitial counterparts are added. The loop exits once all the cld borne
   155)     ! aerosols (and their interstitial counterparts) are added.  Note that the units(fields(i)%units), 
   156)     ! will be same for both interstitial and cloud borne species.
   157)     ! All other aerosol treatments(bulk) are left unchanged
   158)     i_c = 0
   159)     fldloop:do i = 1,number_flds
   160)        if(clim_modal_aero .and. index(trim(fields(i)%fldnam),'_c') > 1) then ! Only cloud borne species
   161)           call addfld(trim(fields(i)%fldnam)//'_D',trim(fields(i)%units), pver, 'A', 'prescribed aero', phys_decomp )
   162)           call spec_c_to_a(trim(fields(i)%fldnam),spec_a)
   163)           call addfld(trim(spec_a)//'_D',trim(fields(i)%units), pver, 'A', 'prescribed aero', phys_decomp )
   164)           i_c = i_c + 1
   165)           if(i_c >= aero_cnt_c) exit fldloop
   166)        else
   167)           call addfld(trim(fields(i)%fldnam)//'_D',trim(fields(i)%units), pver, 'A', 'prescribed aero', phys_decomp )
   168)        endif
   169)     enddo fldloop
   170) 
   171)   end subroutine prescribed_aero_init

   173) !-------------------------------------------------------------------
   174) ! reads namelist options
   175) !-------------------------------------------------------------------
   176) subroutine prescribed_aero_readnl(nlfile)
   177) 
   178)    use namelist_utils,  only: find_group_name
   179)    use units,           only: getunit, freeunit
   180)    use mpishorthand
   181)    use rad_constituents, only: rad_cnst_get_info ! Added to query if it is a modal aero sim or not
   182) 
   183)    character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
   184) 
   185)    ! Local variables
   186)    integer :: unitn, ierr, nmodes, aero_loop_end
   187)    logical :: skip_spec
   188)    character(len=*), parameter :: subname = 'prescribed_aero_readnl'
   189) 
   190)    character(len=32)  :: prescribed_aero_specifier(N_AERO)
   191)    character(len=256) :: prescribed_aero_file
   192)    character(len=256) :: prescribed_aero_filelist
   193)    character(len=256) :: prescribed_aero_datapath
   194)    character(len=32)  :: prescribed_aero_type
   195)    logical            :: prescribed_aero_rmfile
   196)    integer            :: prescribed_aero_cycle_yr
   197)    integer            :: prescribed_aero_fixed_ymd
   198)    integer            :: prescribed_aero_fixed_tod
   199)    integer :: i,k
   200) 
   201)    namelist /prescribed_aero_nl/ &
   202)       prescribed_aero_specifier, &
   203)       prescribed_aero_file,      &
   204)       prescribed_aero_filelist,  &
   205)       prescribed_aero_datapath,  &
   206)       prescribed_aero_type,      &
   207)       prescribed_aero_rmfile,    &
   208)       prescribed_aero_cycle_yr,  &
   209)       prescribed_aero_fixed_ymd, &
   210)       prescribed_aero_fixed_tod      
   211)    !-----------------------------------------------------------------------------
   212) 
   213)    ! Initialize namelist variables from local module variables.
   214)    prescribed_aero_specifier= specifier
   215)    prescribed_aero_file     = filename
   216)    prescribed_aero_filelist = filelist
   217)    prescribed_aero_datapath = datapath
   218)    prescribed_aero_type     = datatype
   219)    prescribed_aero_rmfile   = rmv_file
   220)    prescribed_aero_cycle_yr = cycle_yr
   221)    prescribed_aero_fixed_ymd= fixed_ymd
   222)    prescribed_aero_fixed_tod= fixed_tod
   223) 
   224)    ! Read namelist
   225)    if (masterproc) then
   226)       unitn = getunit()
   227)       open( unitn, file=trim(nlfile), status='old' )
   228)       call find_group_name(unitn, 'prescribed_aero_nl', status=ierr)
   229)       if (ierr == 0) then
   230)          read(unitn, prescribed_aero_nl, iostat=ierr)
   231)          if (ierr /= 0) then
   232)             call endrun(subname // ':: ERROR reading namelist')
   233)          end if
   234)       end if
   235)       close(unitn)
   236)       call freeunit(unitn)
   237)    end if
   238) 
   239) #ifdef SPMD
   240)    ! Broadcast namelist variables
   241)    call mpibcast(prescribed_aero_specifier,len(prescribed_aero_specifier(1))*N_AERO,     mpichar, 0, mpicom)
   242)    call mpibcast(prescribed_aero_file,     len(prescribed_aero_file),     mpichar, 0, mpicom)
   243)    call mpibcast(prescribed_aero_filelist, len(prescribed_aero_filelist), mpichar, 0, mpicom)
   244)    call mpibcast(prescribed_aero_datapath, len(prescribed_aero_datapath), mpichar, 0, mpicom)
   245)    call mpibcast(prescribed_aero_type,     len(prescribed_aero_type),     mpichar, 0, mpicom)
   246)    call mpibcast(prescribed_aero_rmfile,   1, mpilog,  0, mpicom)
   247)    call mpibcast(prescribed_aero_cycle_yr, 1, mpiint,  0, mpicom)
   248)    call mpibcast(prescribed_aero_fixed_ymd,1, mpiint,  0, mpicom)
   249)    call mpibcast(prescribed_aero_fixed_tod,1, mpiint,  0, mpicom)
   250) #endif
   251) 
   252)    ! Update module variables with user settings.
   253)    specifier  = prescribed_aero_specifier
   254)    filename   = prescribed_aero_file
   255)    filelist   = prescribed_aero_filelist
   256)    datapath   = prescribed_aero_datapath
   257)    datatype   = prescribed_aero_type
   258)    rmv_file   = prescribed_aero_rmfile
   259)    cycle_yr   = prescribed_aero_cycle_yr
   260)    fixed_ymd  = prescribed_aero_fixed_ymd
   261)    fixed_tod  = prescribed_aero_fixed_tod
   262) 
   263)    ! Turn on prescribed aerosols if user has specified an input dataset.
   264)    has_prescribed_aero = len_trim(filename) > 0 
   265) 
   266)    if ( .not. has_prescribed_aero) return
   267) 
   268)    ! Determine whether its a 'modal' aerosol simulation  or not
   269)    call rad_cnst_get_info(0, nmodes=nmodes)
   270)    clim_modal_aero = (nmodes > 0)
   271) 
   272)    ! For modal aerosols, interstitial species(*_a) are diagnosed from
   273)    ! their *_logm and *_logv counterparts (e.g. soa_a1 is diagnosed from
   274)    ! soa_a1_logm and soa_a1_logv). Therefore, only *_logm and *_logv and cloud
   275)    ! borne (*_c) species are specified in the build-namelist. In the following 
   276)    ! cnt_loop, we will count the cloud borne species and *_logm species(in lieu 
   277)    ! of *_a species). We will skip *_logv species. This will ensure that aero_cnt
   278)    ! variable is the sum of cloud borne and interstitial species (which we will 
   279)    ! manually add in pbuf_names array later). We are also counting cloud borne 
   280)    ! (*_c) species which will help adding the same number of interstitial species
   281)    ! in pbuf_names array
   282)   
   283)    ! determine which prescibed aerosols are specified
   284)    aero_cnt   = 0
   285)    aero_cnt_c = 0 ! cloud borne species count
   286)    cnt_loop: do i = 1,N_AERO
   287)       if ( len_trim(specifier(i))==0 ) exit cnt_loop
   288)       skip_spec = .FALSE.
   289)       if(clim_modal_aero) then
   290)          ! For modal aerosols, skip counting species ending with *_logv 
   291)          if(index(specifier(i),'_c') >= 1) aero_cnt_c = aero_cnt_c + 1
   292)          if(index(specifier(i),'_logv') >= 1) skip_spec = .TRUE.
   293)       endif
   294)       if(.NOT.skip_spec)aero_cnt = aero_cnt+1
   295)    end do cnt_loop
   296) 
   297)    has_prescribed_aero = aero_cnt>0
   298)    if ( .not. has_prescribed_aero) return
   299) 
   300)    allocate(pbuf_names(aero_cnt))
   301) 
   302)    !  For modal aerosols, the following loop will add the cloud borne 
   303)    ! species directly and interstitial species through the "add_interstitial_spec" 
   304)    ! call. Interstitial species are added at the end of the cloud borne species in
   305)    ! pbuf_names array.
   306)    ! Note that aero_cnt_c should be zero for all other aerosol trearments 
   307)    ! except the modal aerosols (e.g. bulk)
   308)    
   309)    if(.NOT. clim_modal_aero) aero_cnt_c = 0
   310)    aero_loop_end = aero_cnt - aero_cnt_c
   311)    
   312)    do i = 1,aero_loop_end
   313)       k = scan( specifier(i),':')
   314)       if (k>1) then
   315)          pbuf_names(i) = trim(specifier(i)(:k-1)) 
   316)          if(clim_modal_aero)call add_interstitial_spec(aero_loop_end,i)
   317)       else
   318)          pbuf_names(i) = trim(specifier(i))
   319)          if(clim_modal_aero)call add_interstitial_spec(aero_loop_end,i)
   320)       endif
   321)    enddo
   322) 
   323) end subroutine prescribed_aero_readnl

   342) !-------------------------------------------------------------------
   343) ! A generic subroutine which replaces 'c' in the cloud borne 
   344) ! species name with 'a' to make it interstital species
   345) !-------------------------------------------------------------------
   346) subroutine spec_c_to_a (spec_c_in,spec_a_out)
   347)   implicit none
   348) 
   349)   !Arguments
   350)   character(len=*), intent(in)  :: spec_c_in 
   351)   character(len=*), intent(out) :: spec_a_out
   352) 
   353)   !Local
   354)   character(len=32)   :: name
   355)   character(len=1000) :: errMsg
   356)   integer             :: k_c, k_cp1
   357) 
   358)   k_c = index(trim(adjustl(spec_c_in)),'_c')
   359)   if(k_c >= 1) then
   360)      k_cp1             = k_c + 1
   361)      name              = trim(adjustl(spec_c_in))
   362)      name(k_cp1:k_cp1) = 'a'
   363)      spec_a_out        = trim(adjustl(name))
   364)   else
   365)      write(errMsg,*) "Input string (",trim(spec_c_in)," is not a cld borne aerosol,", &
   366)           "cannot form interstitial species name"
   367)      call endrun(trim(errMsg))
   368)   endif  
   369) end subroutine spec_c_to_a

   372) ! advances the aerosol fields to the current time step

   374)   subroutine prescribed_aero_adv( state, pbuf2d )
   375) 
   376)     use tracer_data,  only : advance_trcdata
   377)     use physics_types,only : physics_state
   378)     use ppgrid,       only : begchunk, endchunk
   379)     use ppgrid,       only : pcols, pver
   380)     use string_utils, only : to_lower, GLC
   381)     use cam_history,  only : outfld
   382)     
   383)     use physics_buffer, only : physics_buffer_desc, pbuf_get_field, pbuf_get_chunk, &
   384)          pbuf_get_index
   385) 
   386)     implicit none
   387) 
   388)     type(physics_state), intent(in)    :: state(begchunk:endchunk)                 
   389)     
   390)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   391) 
   392)     type(physics_buffer_desc), pointer :: pbuf_chnk(:)
   393) 
   394)     character(len=32) :: spec_a
   395)     integer :: c,ncol, i, i_c, spec_a_ndx, errcode
   396)     real(r8),pointer :: outdata(:,:)
   397)     logical :: cld_borne_aero = .FALSE.
   398) 
   399)     if( .not. has_prescribed_aero ) return
   400) 
   401)     call advance_trcdata( fields, file, state, pbuf2d )
   402) 
   403)     ! Diagnose interstital species using random sampling
   404)     if ( clim_modal_aero ) then
   405)       call rand_sample_prescribed_aero(state, pbuf2d )
   406)     endif
   407)     
   408)     ! set the tracer fields with the correct units
   409)     i_c = 0
   410)     fldloop : do i = 1,number_flds
   411)        
   412) !$OMP PARALLEL DO PRIVATE (C, NCOL, OUTDATA,PBUF_CHNK)
   413)        do c = begchunk,endchunk
   414)           ncol = state(c)%ncol
   415)           pbuf_chnk => pbuf_get_chunk(pbuf2d, c)
   416)           if(clim_modal_aero .and. index(trim(fields(i)%fldnam),'_c') > 1) then ! Only cloud borne species
   417)              call pbuf_get_field(pbuf_chnk, fields(i)%pbuf_ndx, outdata)
   418)              call outfld( trim(fields(i)%fldnam)//'_D', outdata(:ncol,:), ncol, state(c)%lchnk )
   419)              
   420)              call spec_c_to_a(trim(fields(i)%fldnam),spec_a)
   421)              spec_a_ndx = pbuf_get_index(trim(fields(i)%fldnam),errcode)
   422)              call pbuf_get_field(pbuf_chnk, spec_a_ndx, outdata)
   423)              call outfld( trim(spec_a)//'_D', outdata(:ncol,:), ncol, state(c)%lchnk )
   424)              cld_borne_aero = .TRUE.
   425)           else
   426)              call pbuf_get_field(pbuf_chnk, fields(i)%pbuf_ndx, outdata)
   427)              call outfld( trim(fields(i)%fldnam)//'_D', outdata(:ncol,:), ncol, state(c)%lchnk )
   428)           endif
   429)        enddo
   430)        if(cld_borne_aero)then
   431)           i_c = i_c + 1
   432)           cld_borne_aero = .FALSE.
   433)           if(i_c >= aero_cnt_c) exit fldloop
   434)        endif
   435)     enddo fldloop
   436) 
   437)   end subroutine prescribed_aero_adv

   439) !-------------------------------------------------------------------
   440)   subroutine rand_sample_prescribed_aero(state, pbuf2d)
   441) 
   442)     !Purpose: Generates log normal distribution for the interstitial species.
   443)     !Note that only the interstitial aerosols are diagnosed here
   444)     !
   445)     !Written by Balwinder Singh (12/14/2012)
   446)     !Adapted from Jin-Ho Yoon
   447)     !
   448)     !Update log:
   449) 
   450)     use physics_types,  only : physics_state
   451)     use ppgrid,         only : begchunk, endchunk, pver
   452)     use physics_buffer, only : physics_buffer_desc, pbuf_get_field, pbuf_get_chunk, &
   453)          pbuf_get_index
   454)     use ppgrid,       only : pcols, pver
   455) 
   456)     !Arguments
   457)     type(physics_state), intent(in)    :: state(begchunk:endchunk)
   458)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   459)     
   460)     !Local
   461)     real(r8), parameter :: mean_max_val = 5.0_r8
   462)     real(r8), parameter :: std_max_val  = 3.0_r8
   463) 
   464)     integer  :: c, ncol, i, kc, klog, logv_ndx, logm_ndx, a_ndx, icol, kpver
   465)     real(r8) :: logm2, variance, std, mean_max, std_max, randn
   466)     type(physics_buffer_desc), pointer :: pbuf_chnk(:)
   467)     real(r8), pointer :: data(:,:)
   468)     real(r8) :: logm_data(pcols,pver),logv_data(pcols,pver)
   469) 
   470)     randn = randn_prescribed_aero()
   471)     do i = 1, aero_cnt
   472)        !Species with '_a' are updated using random sampling.
   473)        !Cloud borne species (ends with _c1,_c2, etc.) have to be skipped
   474)        kc   = index(pbuf_names(i),'_c')
   475)        ! if kc>1, species is cloud borne species
   476)        if( kc > 1) cycle
   477)        
   478)        !If species is ending with _a1, a2 etc., then find indicies of their 
   479)        !logv and lom conterparts
   480)        logv_ndx = logvm_get_index(pbuf_names(i),'v')
   481)        logm_ndx = logvm_get_index(pbuf_names(i),'m')
   482)        a_ndx    = pbuf_get_index(trim(adjustl(pbuf_names(i))))
   483) 
   484)        do c = begchunk, endchunk
   485)           ncol = state(c)%ncol
   486)           pbuf_chnk => pbuf_get_chunk(pbuf2d, c)
   487) 
   488)           !Get the species mixing ratio nad its logv and logm values
   489)           call pbuf_get_field(pbuf_chnk, a_ndx, data)
   490)           logv_data = fields(logv_ndx)%data(:,:,c)
   491)           logm_data = fields(logm_ndx)%data(:,:,c)
   492) 
   493)           do icol = 1, ncol
   494)              do kpver = 1, pver
   495)                 logm2    = logm_data(icol,kpver) * logm_data(icol,kpver)
   496) 
   497)                 !Compute variance
   498)                 variance = max( 0.0_r8, (logv_data(icol,kpver) - logm2) )
   499) 
   500)                 !Standard deviation
   501)                 std      = sqrt(variance)
   502) 
   503)                 !Bounds to keep mixing ratios from going unphysical
   504)                 mean_max = exp(logm_data(icol,kpver)) * mean_max_val
   505)                 std_max  = exp(logm_data(icol,kpver)  + std_max_val * std )
   506) 
   507)                 data(icol,kpver) = min(exp(logm_data(icol,kpver)+randn*std),mean_max,std_max)

   509)              enddo !pver
   510)           enddo !col
   511)        enddo !chunk
   512)     enddo !flds

   516)   function randn_prescribed_aero()
   517)     !Pupose: This function generates a new normally distributed random
   518)     ! number at end end of each day. This random number then stays the same
   519)     ! for the whole day
   520)     !
   521)     !Written by Balwinder Singh (12/14/2012)
   522)     !Adapted from Jin-Ho Yoon
   523)     !
   524)     !Update log:
   525)     
   526)     use time_manager,   only : is_end_curr_day, is_first_step, get_nstep
   527)     
   528)     integer, parameter :: rconst1_1 = 5000000
   529)     integer, parameter :: rconst1_2 = 50
   530)     integer, parameter :: rconst2_1 = 10000000
   531)     integer, parameter :: rconst2_2 = 10
   532)     
   533)     integer :: i, seed_size, nstep
   534)     integer, allocatable :: seed(:)
   535) 
   536)     real(r8) :: randn_prescribed_aero
   537)     real(r8) :: randu1, randu2
   538)     
   539)     !Use same random number for the entire day and generate a new normally 
   540)     !distributed random number at the start of the new day
   541)     if(is_first_step() .or. is_end_curr_day()) then
   542)        !Generate two uniformly distributed random numbers (between 0 and 1)
   543)        call random_seed(size=seed_size)
   544)        allocate(seed(seed_size))
   545) 
   546)        ! Using nstep as a seed to generate same sequence
   547)        nstep = get_nstep()
   548)        do i = 1, seed_size
   549)           seed(i) = rconst1_1*nstep + rconst1_2*(i-1)
   550)        end do
   551)        call random_seed(put=seed)
   552)        call random_number (randu1)
   553) 
   554)        do i = 1, seed_size
   555)           seed(i) = rconst2_1*nstep + rconst2_2*(i-1)
   556)        end do
   557)        call random_seed(put=seed)
   558)        call random_number (randu2)
   559)        deallocate(seed)
   560)       
   561)        !Normal distribution (Mean = 0, standard dev = 1)
   562)        randn_prescribed_aero = boxMuller(randu1,randu2)
   563)        randn_persists = randn_prescribed_aero
   564)     else
   565)        !Use the previously generated random number
   566)        randn_prescribed_aero = randn_persists
   567)     endif
   568)   end function randn_prescribed_aero

   599)   function boxMuller(ru1,ru2) result(rn)
   600)     use physconst,      only : pi 
   601)     implicit none
   602) 
   603)     !Args
   604)     real(r8), intent(in)  :: ru1, ru2
   605) 
   606)     !Loc
   607)     real(r8), parameter :: pi2 = 2._r8 * pi    
   608)     real(r8) :: ur, theta, rn
   609) 
   610)     !Based on Box Muller Method, generate normally distributed random numbers 
   611)     ur    = sqrt(-2._r8 * log(ru1))
   612)     theta = pi2 * ru2
   613)     
   614)     !Normal distribution (Mean = 0, standard dev = 1)
   615)     rn = ur * cos(theta)
   616)
