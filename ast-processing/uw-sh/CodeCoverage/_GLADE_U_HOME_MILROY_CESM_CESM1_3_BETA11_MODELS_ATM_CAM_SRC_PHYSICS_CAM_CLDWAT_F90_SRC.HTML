   157) subroutine inimc( tmeltx, rhonotx, gravitx, rh2ox)
   158) !----------------------------------------------------------------------- 
   159) ! 
   160) ! Purpose: 
   161) ! initialize constants for the prognostic condensate
   162) ! 
   163) ! Author: P. Rasch, April 1997
   164) ! 
   165) !-----------------------------------------------------------------------
   166)    use pmgrid,       only: plev, plevp
   167)    use dycore,       only: dycore_is, get_resolution
   168)    use ref_pres,     only: pref_mid
   169) 
   170)    integer k
   171)    real(r8), intent(in) :: tmeltx
   172)    real(r8), intent(in) :: rhonotx
   173)    real(r8), intent(in) :: gravitx
   174)    real(r8), intent(in) :: rh2ox
   175) 
   176) #ifdef UNICOSMP
   177)    real(r8) signgam              ! variable required by cray gamma function
   178)    external gamma
   179) #endif
   180) 
   181)    rhonot = rhonotx             ! air density at surface (gm/cm3)
   182)    gravit = gravitx
   183)    rh2o   = rh2ox
   184)    rhos = .1_r8                 ! assumed snow density (gm/cm3)
   185)    rhow = 1._r8                 ! water density
   186)    rhoi = 1._r8                 ! ice density
   187)    esi = 1.0_r8                 ! collection efficient for ice by snow
   188)    esw = 0.1_r8                 ! collection efficient for water by snow
   189)    t0 = tmeltx                  ! approximate freezing temp
   190)    cldmin = 0.02_r8             ! assumed minimum cloud amount
   191)    small = 1.e-22_r8            ! a small number compared to unity
   192)    c = 152.93_r8                ! constant for graupel like snow cm**(1-d)/s
   193)    d = 0.25_r8                  ! constant for graupel like snow
   194)    nos = 3.e-2_r8               ! particles snow / cm**4
   195)    pi = 4._r8*atan(1.0_r8)
   196)    prhonos = pi*rhos*nos
   197)    thrpd = 3._r8 + d
   198)    if (d==0.25_r8) then
   199)       gam3pd = 2.549256966718531_r8 ! only right for d = 0.25
   200)       gam4pd = 8.285085141835282_r8
   201)    else
   202) #ifdef UNICOSMP
   203)       call gamma(3._r8+d, signgam, gam3pd)
   204)       gam3pd = sign(exp(gam3pd),signgam)
   205)       call gamma(4._r8+d, signgam, gam4pd)
   206)       gam4pd = sign(exp(gam4pd),signgam)
   207)       write(iulog,*) ' d, gamma(3+d), gamma(4+d) =', gam3pd, gam4pd
   208) #else
   209)       call endrun(' can only use d ne 0.25 on a cray ')
   210) #endif
   211)    endif
   212)    mcon01 = pi*nos*c*gam3pd/4._r8
   213)    mcon02 = 1._r8/(c*gam4pd*sqrt(rhonot)/(6*prhonos**(d/4._r8)))
   214)    mcon03 = -(0.5_r8+d/4._r8)
   215)    mcon04 = 4._r8/(4._r8+d)
   216)    mcon05 = (3+d)/(4+d)
   217)    mcon06 = (3+d)/4._r8
   218)    mcon07 = mcon01*sqrt(rhonot)*mcon02**mcon05/prhonos**mcon06
   219)    mcon08 = -0.5_r8/(4._r8+d)
   220) 
   221)    if( masterproc ) write(iulog,*) 'cloud water initialization by inimc complete '
   222) 
   223) ! Initialize parameters used by findmcnew
   224)    capnw = 400._r8              ! warm continental cloud particles / cm3
   225)    capnc = 150._r8              ! cold and oceanic cloud particles / cm3
   226) !  capnsi = 40._r8              ! sea ice cloud particles density  / cm3
   227)    capnsi = 75._r8              ! sea ice cloud particles density  / cm3
   228) 
   229)    kconst = 1.18e6_r8           ! const for terminal velocity
   230) 
   231) !  effc = 1._r8                 ! autoconv collection efficiency following boucher 96
   232) !  effc = .55*0.05_r8           ! autoconv collection efficiency following baker 93
   233)    effc = 0.55_r8               ! autoconv collection efficiency following tripoli and cotton
   234) !  effc = 0._r8    ! turn off warm-cloud autoconv
   235)    alpha = 1.1_r8**4
   236)    capc = pi**(-.333_r8)*kconst*effc *(0.75_r8)**(1.333_r8)*alpha  ! constant for autoconversion
   237) 
   238) ! critical precip rate at which we assume the collector drops can change the
   239) ! drop size enough to enhance the auto-conversion process (mm/day)
   240)    critpr = 0.5_r8
   241)    convfw = 1.94_r8*2.13_r8*sqrt(rhow*1000._r8*9.81_r8*2.7e-4_r8)
   242) 
   243) ! liquid microphysics
   244) !  cracw = 6_r8                 ! beheng
   245)    cracw = .884_r8*sqrt(9.81_r8/(rhow*1000._r8*2.7e-4_r8)) ! tripoli and cotton
   246) 
   247) ! ice microphysics
   248)    ciautb = 5.e-4_r8
   249) 
   250)    if ( masterproc ) then
   251)       write(iulog,*)'tuning parameters cldwat: icritw',icritw,'icritc',icritc,'conke',conke,'r3lcrit',r3lcrit
   252)       write(iulog,*)'tuning parameters cldwat: capnw',capnw,'capnc',capnc,'capnsi',capnsi,'kconst',kconst
   253)       write(iulog,*)'tuning parameters cldwat: effc',effc,'alpha',alpha,'capc',capc
   254)       write(iulog,*)'tuning parameters cldwat: critpr',critpr,'convfw',convfw,'cracw',cracw,'ciautb',ciautb
   255)    endif
   256) 
   257) end subroutine inimc

   405)    real(r8) rcwn(pcols,2,pver), rliq(pcols,2,pver), rice(pcols,2,pver)
   406)    real(r8) cwnsave(pcols,2,pver), cmesave(pcols,2,pver)
   407)    real(r8) prodprecsave(pcols,2,pver)
   408)    logical error_found
   409) !
   410) !------------------------------------------------------------
   411) !
   412)    clrh2o = latvap/rh2o   ! Ratio of latvap to water vapor gas const
   413) #ifdef PERGRO
   414)    mincld = 1.e-4_r8
   415)    iter = 1   ! number of times to iterate the precipitation calculation
   416) #else
   417)    mincld = 1.e-4_r8
   418)    iter = 2
   419) #endif
   420) !   omsm = 0.99999
   421)    cpohl = cpair/latvap
   422)    hlocp = latvap/cpair
   423)    dto2=0.5_r8*deltat
   424) !
   425) ! Constant for computing rate of evaporation of precipitation:
   426) !
   427) !!$   conke = 1.e-5
   428) !!$   conke = 1.e-6
   429) !
   430) ! initialize a few single level fields
   431) !
   432)    do i = 1,ncol
   433)       precip(i) = 0.0_r8
   434)       precab(i) = 0.0_r8
   435)       snowab(i) = 0.0_r8
   436)       cldmax(i) = 0.0_r8
   437)    end do
   438) !
   439) ! initialize multi-level fields 
   440) !
   441)    do k = 1,pver
   442)       do i = 1,ncol
   443)          q(i,k) = qn(i,k) 
   444)          t(i,k) = tn(i,k)
   445) !         q(i,k)=qn(i,k)-qtend(i,k)*deltat
   446) !         t(i,k)=tn(i,k)-ttend(i,k)*deltat
   447)     end do
   448)    end do
   449)    cme     (:ncol,:) = 0._r8
   450)    evapprec(:ncol,:) = 0._r8
   451)    prodprec(:ncol,:) = 0._r8
   452)    evapsnow(:ncol,:) = 0._r8
   453)    prodsnow(:ncol,:) = 0._r8
   454)    evapheat(:ncol,:) = 0._r8
   455)    meltheat(:ncol,:) = 0._r8
   456)    prfzheat(:ncol,:) = 0._r8
   457)    ice2pr(:ncol,:)   = 0._r8
   458)    liq2pr(:ncol,:)   = 0._r8
   459)    liq2snow(:ncol,:) = 0._r8
   460)    fwaut(:ncol,:) = 0._r8
   461)    fsaut(:ncol,:) = 0._r8
   462)    fracw(:ncol,:) = 0._r8
   463)    fsacw(:ncol,:) = 0._r8
   464)    fsaci(:ncol,:) = 0._r8
   465)    rkflxprc(:ncol,:) = 0._r8
   466)    rkflxsnw(:ncol,:) = 0._r8
   467) 
   468)    pracwo(:ncol,:) = 0._r8
   469)    psacwo(:ncol,:) = 0._r8
   470)    psacio(:ncol,:) = 0._r8
   471) !
   472) ! find the wet bulb temp and saturation value
   473) ! for the provisional t and q without condensation
   474) !
   475)    do 800 k = top_lev,pver
   476) 
   477)       ! "True" means that ice will be taken into account.
   478)       call findsp_vc(qn(:ncol,k), tn(:ncol,k), p(:ncol,k), .true., &
   479)            tsp(:ncol,k), qsp(:ncol,k))
   480) 
   481)       call qsat(t(:ncol,k), p(:ncol,k), &
   482)            es(:ncol), qs(:ncol), gam=gamma(:ncol))
   483)       do i = 1,ncol
   484)          relhum(i) = q(i,k)/qs(i)
   485) !
   486)          cldm(i) = max(cldn(i,k),mincld)
   487) !
   488) ! the max cloud fraction above this level
   489) !
   490)          cldmax(i) = max(cldmax(i), cldm(i))
   491) 
   492) ! define the coefficients for C - E calculation
   493) 
   494)          calpha(i) = 1.0_r8/qs(i)
   495)          cbeta (i) = q(i,k)/qs(i)**2*gamma(i)*cpohl
   496)          cbetah(i) = 1.0_r8/qs(i)*gamma(i)*cpohl
   497)          cgamma(i) = calpha(i)+latvap*cbeta(i)/cpair
   498)          cgamah(i) = calpha(i)+latvap*cbetah(i)/cpair
   499)          rcgama(i) = cgamma(i)/cgamah(i)
   500) 
   501)          if(cldm(i) > mincld) then
   502)             icwc(i) = max(0._r8,cwat(i,k)/cldm(i))
   503)          else
   504)             icwc(i) = 0.0_r8
   505)          endif
   506) !PJR the above logic give zero icwc with nonzero cwat, dont like it!
   507) !PJR generates problems with csigma
   508) !PJR set the icwc to a very small number, so we can start from zero cloud cover and make some clouds
   509) !         icwc(i) = max(1.e-8_r8,cwat(i,k)/cldm(i))
   510) 
   511) !
   512) ! initial guess of evaporation, will be updated within iteration
   513) !
   514)          evapprec(i,k) = conke*(1._r8 - cldm(i))*sqrt(precab(i)) &
   515)                         *(1._r8 - min(relhum(i),1._r8))
   516) 
   517) !
   518) ! zero cmeres before iteration for each level
   519) !
   520)          cmeres(i)=0.0_r8
   521) 
   522)       end do
   523)       do i = 1,ncol
   524) !
   525) ! fractions of ice at this level
   526) !
   527) !!$         tc = t(i,k) - t0
   528) !!$         fice(i,k) = max(0._r8,min(-tc*0.05,1.0_r8))
   529) !
   530) ! calculate the cooling due to a phase change of the rainwater
   531) ! from above
   532) !
   533)          if (t(i,k) >= t0) then
   534)             meltheat(i,k) =  -latice * snowab(i) * gravit/pdel(i,k)
   535)             snowab(i) = 0._r8
   536)          else
   537)             meltheat(i,k) = 0._r8
   538)          endif
   539)       end do
   540) 
   541) !
   542) ! calculate cme and formation of precip. 
   543) !
   544) ! The cloud microphysics is highly nonlinear and coupled with cme
   545) ! Both rain processes and cme are calculated iteratively.
   546) ! 
   547)       do 100 l = 1,iter
   548) 
   549)         do i = 1,ncol
   550) 
   551) !
   552) ! calculation of cme has 4 scenarios
   553) ! ==================================
   554) !
   555)            if(relhum(i) > rhu00(i,k)) then
   556)     
   557)            ! 1. whole grid saturation
   558)            ! ========================
   559)               if(relhum(i) >= 0.999_r8 .or. cldm(i) >= 0.999_r8 ) then
   560)                  cme(i,k)=(calpha(i)*qtend(i,k)-cbetah(i)*ttend(i,k))/cgamah(i)
   561) 
   562)            ! 2. fractional saturation
   563)            ! ========================
   564)               else
   565)                  if (rhdfda(i,k) .eq. 0._r8 .and. icwc(i).eq.0._r8) then
   566)                     write (iulog,*) ' cldwat.F90:  empty rh cloud ', i, k, lchnk
   567)                     write (iulog,*) ' relhum, iter ', relhum(i), l, rhu00(i,k), cldm(i), cldn(i,k)
   568)                     call endrun ()
   569)                  endif
   570)                   csigma(i) = 1.0_r8/(rhdfda(i,k)+cgamma(i)*icwc(i))
   571)                   cmec1(i) = (1.0_r8-cldm(i))*csigma(i)*rhdfda(i,k)
   572)                   cmec2(i) = cldm(i)*calpha(i)/cgamah(i)+(1.0_r8-rcgama(i)*cldm(i))*   &
   573)                              csigma(i)*calpha(i)*icwc(i)
   574)                   cmec3(i) = cldm(i)*cbetah(i)/cgamah(i) +  &
   575)                            (cbeta(i)-rcgama(i)*cldm(i)*cbetah(i))*csigma(i)*icwc(i)
   576)                   cmec4(i) = csigma(i)*cgamma(i)*icwc(i)
   577) 
   578)                   ! Q=C-E=-C1*Al + C2*Aq - C3* At + C4*Er
   579)   
   580)                   cme(i,k) = -cmec1(i)*lctend(i,k) + cmec2(i)*qtend(i,k)  &
   581)                              -cmec3(i)*ttend(i,k) + cmec4(i)*evapprec(i,k)
   582)                endif
   583) 
   584)            ! 3. when rh < rhu00, evaporate existing cloud water
   585)            ! ================================================== 
   586)            else if(cwat(i,k) > 0.0_r8)then
   587)               ! liquid water should be evaporated but not to exceed 
   588)               ! saturation point. if qn > qsp, not to evaporate cwat
   589)               cme(i,k)=-min(max(0._r8,qsp(i,k)-qn(i,k)),cwat(i,k))/deltat 
   590) 
   591)            ! 4. no condensation nor evaporation
   592)            ! ==================================
   593)            else
   594)               cme(i,k)=0.0_r8
   595)            endif
   596) 
   597)   
   598)         end do    !end loop for cme update
   599) 
   600) ! Because of the finite time step, 
   601) ! place a bound here not to exceed wet bulb point
   602) ! and not to evaporate more than available water
   603) !
   604)          do i = 1, ncol
   605)             qtmp = qn(i,k) - cme(i,k)*deltat
   606) 
   607) ! possibilities to have qtmp > qsp
   608) !
   609) !   1. if qn > qs(tn), it condenses; 
   610) !      if after applying cme,  qtmp > qsp,  more condensation is applied. 
   611) !      
   612) !   2. if qn < qs, evaporation should not exceed qsp,
   613)     
   614)             if(qtmp > qsp(i,k)) then
   615)               cme(i,k) = cme(i,k) + (qtmp-qsp(i,k))/deltat
   616)             endif
   617) 
   618) !
   619) ! if net evaporation, it should not exceed available cwat
   620) !
   621)             if(cme(i,k) < -cwat(i,k)/deltat)  &
   622)                cme(i,k) = -cwat(i,k)/deltat
   623) !
   624) ! addition of residual condensation from previous step of iteration
   625) !
   626)             cme(i,k) = cme(i,k) + cmeres(i)
   627) 
   628)          end do
   629) 
   630)          !      limit cme for roundoff errors
   631)          do i = 1, ncol
   632)             cme(i,k) = cme(i,k)*omsm
   633)          end do
   634) 
   635)          do i = 1,ncol
   636) !
   637) ! as a safe limit, condensation should not reduce grid mean rh below rhu00
   638) ! 
   639)            if(cme(i,k) > 0.0_r8 .and. relhum(i) > rhu00(i,k) )  &
   640)               cme(i,k) = min(cme(i,k), (qn(i,k)-qs(i)*rhu00(i,k))/deltat)
   641) !
   642) ! initial guess for cwm (mean cloud water over time step) if 1st iteration
   643) !
   644)            if(l < 2) then
   645)              cwm(i) = max(cwat(i,k)+cme(i,k)*dto2,  0._r8)
   646)            endif
   647) 
   648)          enddo
   649) 
   650) ! provisional precipitation falling through model layer
   651)          do i = 1,ncol
   652) !!$            prprov(i) =  precab(i) + prodprec(i,k)*pdel(i,k)/gravit
   653) ! rain produced in this layer not too effective in collection process
   654)             wtthick = max(0._r8,min(0.5_r8,((zi(i,k)-zi(i,k+1))/1000._r8)**2))
   655)             prprov(i) =  precab(i) + wtthick*prodprec(i,k)*pdel(i,k)/gravit
   656)          end do
   657) 
   658) ! calculate conversion of condensate to precipitation by cloud microphysics 
   659)          call findmcnew (lchnk   ,ncol    , &
   660)                          k       ,prprov  ,snowab,  t       ,p        , &
   661)                          cwm     ,cldm    ,cldmax  ,fice(1,k),coef    , &
   662)                          fwaut(1,k),fsaut(1,k),fracw(1,k),fsacw(1,k),fsaci(1,k), &
   663)                          seaicef, snowh, pracwo(1,k), psacwo(1,k), psacio(1,k))
   664) 
   665) !
   666) ! calculate the precip rate
   667) !
   668)          error_found = .false.
   669)          do i = 1,ncol
   670)             if (cldm(i) > 0) then  
   671) !
   672) ! first predict the cloud water
   673) !
   674)                cdt = coef(i)*deltat
   675)                if (cdt > 0.01_r8) then
   676)                   pol = cme(i,k)/coef(i) ! production over loss
   677)                   cwn(i) = max(0._r8,(cwat(i,k)-pol)*exp(-cdt)+ pol)
   678)                else
   679)                   cwn(i) = max(0._r8,(cwat(i,k) + cme(i,k)*deltat)/(1+cdt))
   680)                endif
   681) !
   682) ! now back out the tendency of net rain production
   683) !
   684)                prodprec(i,k) =  max(0._r8,cme(i,k)-(cwn(i)-cwat(i,k))/deltat)
   685)             else
   686)                prodprec(i,k) = 0.0_r8
   687)                cwn(i) = 0._r8
   688)             endif
   689) 
   690)             ! provisional calculation of conversion terms
   691)             ice2pr(i,k) = prodprec(i,k)*(fsaut(i,k)+fsaci(i,k))
   692)             liq2pr(i,k) = prodprec(i,k)*(fwaut(i,k)+fsacw(i,k)+fracw(i,k))
   693) !old        liq2snow(i,k) = prodprec(i,k)*fsacw(i,k)
   694) 
   695) !           revision suggested by Jim McCaa
   696) !           it controls the amount of snow hitting the sfc 
   697) !           by forcing a lot of conversion of cloud liquid to snow phase
   698) !           it might be better done later by an explicit representation of 
   699) !           rain accreting ice (and freezing), or by an explicit freezing of raindrops
   700)             liq2snow(i,k) = max(prodprec(i,k)*fsacw(i,k), fsnow(i,k)*liq2pr(i,k))
   701) 
   702)             ! bounds
   703)             nice2pr = min(ice2pr(i,k),(cwat(i,k)+cme(i,k)*deltat)*fice(i,k)/deltat)
   704)             nliq2pr = min(liq2pr(i,k),(cwat(i,k)+cme(i,k)*deltat)*(1._r8-fice(i,k))/deltat)
   705) !            write(iulog,*) ' prodprec ', i, k, prodprec(i,k)
   706) !            write(iulog,*) ' nliq2pr, nice2pr ', nliq2pr, nice2pr
   707)             if (liq2pr(i,k).ne.0._r8) then
   708)                nliq2snow = liq2snow(i,k)*nliq2pr/liq2pr(i,k)   ! correction
   709)             else
   710)                nliq2snow = liq2snow(i,k)
   711)             endif
   712) 
   713) !           avoid roundoff problems generating negatives
   714)             nliq2snow = nliq2snow*omsm
   715)             nliq2pr = nliq2pr*omsm
   716)             nice2pr = nice2pr*omsm
   717)             
   718) !           final estimates of conversion to precip and snow
   719)             prodprec(i,k) = (nliq2pr + nice2pr)
   720)             prodsnow(i,k) = (nice2pr + nliq2snow)
   721) 
   722)             rcwn(i,l,k) =  cwat(i,k) + (cme(i,k)-   prodprec(i,k))*deltat
   723)             rliq(i,l,k) = (cwat(i,k) + cme(i,k)*deltat)*(1._r8-fice(i,k)) - nliq2pr * deltat
   724)             rice(i,l,k) = (cwat(i,k) + cme(i,k)*deltat)* fice(i,k)      -    nice2pr                     *deltat
   725) 
   726) !           Save for sanity check later...  
   727) !           Putting sanity checks inside loops 100 and 800 screws up the 
   728) !           IBM compiler for reasons as yet unknown.  TBH
   729)             cwnsave(i,l,k)      = cwn(i)
   730)             cmesave(i,l,k)      = cme(i,k)
   731)             prodprecsave(i,l,k) = prodprec(i,k)
   732) !           End of save for sanity check later...  
   733) 
   734) !           final version of condensate to precip terms
   735)             liq2pr(i,k) = nliq2pr
   736)             liq2snow(i,k) = nliq2snow
   737)             ice2pr(i,k) = nice2pr
   738) 
   739)             cwn(i) = rcwn(i,l,k)
   740) !
   741) ! update any remaining  provisional values
   742) !
   743)             cwm(i) = (cwn(i) + cwat(i,k))*0.5_r8
   744) !
   745) ! update in cloud water
   746) !
   747)             if(cldm(i) > mincld) then
   748)                icwc(i) = cwm(i)/cldm(i)
   749)             else
   750)                icwc(i) = 0.0_r8
   751)             endif
   752) !PJR the above logic give zero icwc with nonzero cwat, dont like it!
   753) !PJR generates problems with csigma
   754) !PJR set the icwc to a very small number, so we can start from zero cloud cover and make some clouds
   755) !         icwc(i) = max(1.e-8_r8,cwm(i)/cldm(i))
   756) 
   757)          end do              ! end of do i = 1,ncol
   758) 
   759) !
   760) ! calculate provisional value of cloud water for
   761) ! evaporation of precipitate (evapprec) calculation
   762) !
   763)       do i = 1,ncol
   764)          qtmp = qn(i,k) - cme(i,k)*deltat
   765)          ttmp = tn(i,k) + deltat/cpair * ( meltheat(i,k)       &
   766)               + (latvap + latice*fice(i,k)) * cme(i,k) )
   767)          esn = estblf(ttmp)
   768)          qsn = svp_to_qsat(esn, p(i,k))
   769)          qtl(i) = max((qsn - qtmp)/deltat,0._r8)
   770)          relhum1(i) = qtmp/qsn
   771)       end do
   772) !
   773)       do i = 1,ncol
   774) #ifdef PERGRO
   775)          evapprec(i,k) = conke*(1._r8 - max(cldm(i),mincld))* &
   776)                          sqrt(precab(i))*(1._r8 - min(relhum1(i),1._r8))
   777) #else
   778)          evapprec(i,k) = conke*(1._r8 - cldm(i))*sqrt(precab(i)) &
   779)                          *(1._r8 - min(relhum1(i),1._r8))
   780) #endif
   781) !
   782) ! limit the evaporation to the amount which is entering the box
   783) ! or saturates the box
   784) !
   785)          prtmp = precab(i)*gravit/pdel(i,k)
   786)          evapprec(i,k) = min(evapprec(i,k), prtmp, qtl(i))*omsm
   787) #ifdef PERGRO
   788) !           zeroing needed for pert growth
   789)          evapprec(i,k) = 0._r8
   790) #endif
   791) !
   792) ! Partition evaporation of precipitate between rain and snow using
   793) ! the fraction of snow falling into the box. Determine the heating
   794) ! due to evaporation. Note that evaporation is positive (loss of precip,
   795) ! gain of vapor) and that heating is negative.
   796)          if (evapprec(i,k) > 0._r8) then
   797)             evapsnow(i,k) = evapprec(i,k) * snowab(i) / precab(i)
   798)             evapheat(i,k) = -latvap * evapprec(i,k) - latice * evapsnow(i,k)
   799)          else 
   800)             evapsnow(i,k) = 0._r8
   801)             evapheat(i,k) = 0._r8
   802)          end if
   803) ! Account for the latent heat of fusion for liquid drops collected by falling snow
   804)          prfzheat(i,k) = latice * liq2snow(i,k)
   805)       end do
   806) 
   807) ! now remove the residual of any over-saturation. Normally,
   808) ! the oversaturated water vapor should have been removed by 
   809) ! cme formulation plus constraints by wet bulb tsp/qsp
   810) ! as computed above. However, because of non-linearity,
   811) ! addition of (cme-evapprec) to update t and q may still cause
   812) ! a very small amount of over saturation. It is called a
   813) ! residual of over-saturation because theoretically, cme
   814) ! should have taken care of all of large scale condensation.
   815) ! 
   816) 
   817)        do i = 1,ncol
   818)           qtmp = qn(i,k)-(cme(i,k)-evapprec(i,k))*deltat
   819)           ttmp = tn(i,k) + deltat/cpair * ( meltheat(i,k) + evapheat(i,k) + prfzheat(i,k)      &
   820)               + (latvap + latice*fice(i,k)) * cme(i,k) )
   821) 
   822)           call qsat(ttmp, p(i,k), esn, qsn, dqsdt=dqsdt)
   823) 
   824)           if( qtmp > qsn ) then
   825)              !
   826)              !now extra condensation to bring air to just saturation
   827)              !
   828)              ctmp = (qtmp-qsn)/(1._r8+hlocp*dqsdt)/deltat
   829)              cme(i,k) = cme(i,k)+ctmp
   830) !
   831) ! save residual on cmeres to addtion to cme on entering next iteration
   832) ! cme exit here contain the residual but overrided if back to iteration
   833) !
   834)              cmeres(i) = ctmp
   835)           else
   836)              cmeres(i) = 0.0_r8
   837)           endif
   838)        end do
   839)               
   840)  100 continue              ! end of do l = 1,iter
   841) 
   842) !
   843) ! precipitation
   844) !
   845)       do i = 1,ncol
   846)          precip(i) = precip(i) + pdel(i,k)/gravit * (prodprec(i,k) - evapprec(i,k))
   847)          precab(i) = precab(i) + pdel(i,k)/gravit * (prodprec(i,k) - evapprec(i,k))
   848)          if(precab(i).lt.0._r8) precab(i)=0._r8
   849) !         snowab(i) = snowab(i) + pdel(i,k)/gravit * (prodprec(i,k)*fice(i,k) - evapsnow(i,k))
   850)          snowab(i) = snowab(i) + pdel(i,k)/gravit * (prodsnow(i,k) - evapsnow(i,k))
   851) 
   852)          ! If temperature above freezing, all precip is rain flux.  if temperature below freezing, all precip is snow flux.
   853)          rkflxprc(i,k+1) = precab(i)   !! making this consistent with other precip fluxes.  prc = rain + snow
   854)          !!rkflxprc(i,k+1) = precab(i) - snowab(i)
   855)          rkflxsnw(i,k+1) = snowab(i)
   856) 
   857) !!$         if ((precab(i)) < 1.e-10) then      
   858) !!$            precab(i) = 0.
   859) !!$            snowab(i) = 0.
   860) !!$         endif
   861)       end do
   862)  800 continue                ! level loop (k=1,pver)
   863) 
   864) ! begin sanity checks
   865)    error_found = .false.
