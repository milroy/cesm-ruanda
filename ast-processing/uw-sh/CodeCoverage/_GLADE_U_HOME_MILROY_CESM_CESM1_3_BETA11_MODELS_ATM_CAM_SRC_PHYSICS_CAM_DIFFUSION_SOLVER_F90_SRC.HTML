     4)   !------------------------------------------------------------------------------------ !
     5)   ! Module to solve vertical diffusion equations using a tri-diagonal solver.           !

    15)   !    any              provides functionality of intrinsic any for type vdiff_selector !

    19)   ! Modularization      :  J. McCaa, September 2004                                     !

   131)   subroutine compute_vdiff( lchnk           ,                                                                   &
   132)                             pcols           , pver               , ncnst         , ncol         , pmid        , &
   133)                             pint            , rpdel              , t             , ztodt        , taux        , &
   134)                             tauy            , shflx              , cflx          , ntop         , nbot        , &
   135)                             kvh             , kvm                , kvq           , cgs          , cgh         , &
   136)                             zi              , ksrftms            , qmincg        , fieldlist    , fieldlistm  , &
   137)                             u               , v                  , q             , dse          ,               &
   138)                             tautmsx         , tautmsy            , dtk           , topflx       , errstring   , &
   139)                             tauresx         , tauresy            , itaures       , cpairv       , rairi       , &
   140)                             do_molec_diff  , compute_molec_diff, vd_lu_qdecomp, kvt )
   141) 
   142)     !-------------------------------------------------------------------------- !
   143)     ! Driver routine to compute vertical diffusion of momentum, moisture, trace !
   144)     ! constituents and dry static energy. The new temperature is computed from  !
   145)     ! the diffused dry static energy.                                           ! 
   146)     ! Turbulent diffusivities and boundary layer nonlocal transport terms are   !
   147)     ! obtained from the turbulence module.                                      !
   148)     !-------------------------------------------------------------------------- !
   149) 
   150) !    Used for CAM debugging.
   151) !    use phys_debug_util,    only : phys_debug_col
   152) !    use time_manager,       only : is_first_step, get_nstep
   153)     use vdiff_lu_solver,     only : lu_decomp, vd_lu_decomp, vd_lu_solve
   154)   
   155)   ! Modification : Ideally, we should diffuse 'liquid-ice static energy' (sl), not the dry static energy.
   156)   !                Also, vertical diffusion of cloud droplet number concentration and aerosol number
   157)   !                concentration should be done very carefully in the future version.
   158) 
   159)     ! --------------- !
   160)     ! Input Arguments !
   161)     ! --------------- !
   162) 
   163)     integer,  intent(in)    :: lchnk                   
   164)     integer,  intent(in)    :: pcols
   165)     integer,  intent(in)    :: pver
   166)     integer,  intent(in)    :: ncnst
   167)     integer,  intent(in)    :: ncol                      ! Number of atmospheric columns
   168)     integer,  intent(in)    :: ntop                      ! Top    interface level to which vertical diffusion is applied ( = 1 ).
   169)     integer,  intent(in)    :: nbot                      ! Bottom interface level to which vertical diffusion is applied ( = pver ).
   170)     integer,  intent(in)    :: itaures                   ! Indicator determining whether 'tauresx,tauresy'
   171)                                                          ! is updated (1) or non-updated (0) in this subroutine.
   172) 
   173)     real(r8), intent(in)    :: pmid(pcols,pver)          ! Mid-point pressures [ Pa ]
   174)     real(r8), intent(in)    :: pint(pcols,pver+1)        ! Interface pressures [ Pa ]
   175)     real(r8), intent(in)    :: rpdel(pcols,pver)         ! 1./pdel
   176)     real(r8), intent(in)    :: t(pcols,pver)             ! Temperature [ K ]
   177)     real(r8), intent(in)    :: ztodt                     ! 2 delta-t [ s ]
   178)     real(r8), intent(in)    :: taux(pcols)               ! Surface zonal      stress.
   179)                                                          ! Input u-momentum per unit time per unit area into the atmosphere [ N/m2 ]
   180)     real(r8), intent(in)    :: tauy(pcols)               ! Surface meridional stress.
   181)                                                          ! Input v-momentum per unit time per unit area into the atmosphere [ N/m2 ]
   182)     real(r8), intent(in)    :: shflx(pcols)              ! Surface sensible heat flux [ W/m2 ]
   183)     real(r8), intent(in)    :: cflx(pcols,ncnst)         ! Surface constituent flux [ kg/m2/s ]
   184)     real(r8), intent(in)    :: zi(pcols,pver+1)          ! Interface heights [ m ]
   185)     real(r8), intent(in)    :: ksrftms(pcols)            ! Surface drag coefficient for turbulent mountain stress. > 0. [ kg/s/m2 ]
   186)     real(r8), intent(in)    :: qmincg(ncnst)             ! Minimum constituent mixing ratios from cg fluxes
   187)     real(r8), intent(in)    :: cpairv(pcols,pver)        ! Specific heat at constant pressure
   188)     real(r8), intent(in)    :: rairi(pcols,pver+1)       ! Gas constant on interface levels
   189)     real(r8), intent(in)    :: kvh(pcols,pver+1)         ! Eddy diffusivity for heat [ m2/s ]
   190) 
   191)     logical,  intent(in)    :: do_molec_diff             ! Flag indicating multiple constituent diffusivities
   192) 
   193)     integer,  external, optional :: compute_molec_diff   ! Constituent-independent moleculuar diffusivity routine
   194)     integer,  external, optional :: vd_lu_qdecomp        ! Constituent-dependent moleculuar diffusivity routine
   195)     real(r8), intent(out), optional :: kvt(pcols,pver+1) ! Kinematic molecular conductivity
   196)     type(vdiff_selector), intent(in) :: fieldlist        ! Array of flags selecting which fields to diffuse
   197)     type(vdiff_selector), intent(in) :: fieldlistm       ! Array of flags selecting which fields for molecular diffusion
   198) 
   199)     ! ---------------------- !
   200)     ! Input-Output Arguments !
   201)     ! ---------------------- !
   202) 
   203)     real(r8), intent(inout) :: kvm(pcols,pver+1)         ! Eddy viscosity ( Eddy diffusivity for momentum ) [ m2/s ]
   204)     real(r8), intent(inout) :: kvq(pcols,pver+1)         ! Eddy diffusivity for constituents
   205)     real(r8), intent(inout) :: cgs(pcols,pver+1)         ! Counter-gradient star [ cg/flux ]
   206)     real(r8), intent(inout) :: cgh(pcols,pver+1)         ! Counter-gradient term for heat
   207) 
   208)     real(r8), intent(inout) :: u(pcols,pver)             ! U wind. This input is the 'raw' input wind to
   209)                                                          ! PBL scheme without iterative provisional update. [ m/s ]
   210)     real(r8), intent(inout) :: v(pcols,pver)             ! V wind. This input is the 'raw' input wind to PBL scheme
   211)                                                          ! without iterative provisional update. [ m/s ]
   212)     real(r8), intent(inout) :: q(pcols,pver,ncnst)       ! Moisture and trace constituents [ kg/kg, #/kg ? ]
   213)     real(r8), intent(inout) :: dse(pcols,pver)           ! Dry static energy [ J/kg ]
   214) 
   215)     real(r8), intent(inout) :: tauresx(pcols)            ! Input  : Reserved surface stress at previous time step
   216)     real(r8), intent(inout) :: tauresy(pcols)            ! Output : Reserved surface stress at current  time step
   217) 
   218)     ! ---------------- !
   219)     ! Output Arguments !
   220)     ! ---------------- !
   221) 
   222)     real(r8), intent(out)   :: dtk(pcols,pver)           ! T tendency from KE dissipation
   223)     real(r8), intent(out)   :: tautmsx(pcols)            ! Implicit zonal      turbulent mountain surface stress
   224)                                                          ! [ N/m2 = kg m/s /s/m2 ]
   225)     real(r8), intent(out)   :: tautmsy(pcols)            ! Implicit meridional turbulent mountain surface stress
   226)                                                          ! [ N/m2 = kg m/s /s/m2 ]
   227)     real(r8), intent(out)   :: topflx(pcols)             ! Molecular heat flux at the top interface
   228)     character(128), intent(out) :: errstring             ! Output status
   229) 
   230)     ! --------------- !
   231)     ! Local Variables ! 
   232)     ! --------------- !
   233) 
   234)     integer  :: i, k, m, icol                            ! Longitude, level, constituent indices
   235)     integer  :: status                                   ! Status indicator
   236)     integer  :: nbot_molec                               ! Bottom level where molecular diffusivity is applied
   237)     integer  :: ntop_molec                               ! Top level where molecular diffusivity is applied
   238)     logical  :: lqtst(pcols)                             ! Adjust vertical profiles
   239)     logical  :: need_decomp                              ! Whether to compute a new decomposition
   240)     logical  :: cnst_fixed_ubc(ncnst)                    ! Whether upper boundary condition is fixed
   241)     logical  :: cnst_fixed_ubflx(ncnst)                  ! Whether upper boundary flux is a fixed non-zero value
   242) 
   243)     ! LU decomposition information.
   244)     type(lu_decomp) :: decomp
   245) 
   246)     real(r8) :: tmp1(pcols)                              ! Temporary storage
   247)     real(r8) :: tmpi1(pcols,pver+1)                      ! Interface KE dissipation
   248)     real(r8) :: tint(pcols,pver+1)                       ! Interface temperature
   249)     real(r8) :: rhoi(pcols,pver+1)                       ! rho at interfaces
   250)     real(r8) :: tmpi2(pcols,pver+1)                      ! dt*(g*rho)**2/dp at interfaces
   251)     real(r8) :: rrho(pcols)                              ! 1./bottom level density 
   252) 
   253)     real(r8) :: zero(pcols)                              ! Zero array for surface heat exchange coefficients 
   254)     real(r8) :: tautotx(pcols)                           ! Total surface stress ( zonal )
   255)     real(r8) :: tautoty(pcols)                           ! Total surface stress ( meridional )
   256) 
   257)     real(r8) :: dinp_u(pcols,pver+1)                     ! Vertical difference at interfaces, input u
   258)     real(r8) :: dinp_v(pcols,pver+1)                     ! Vertical difference at interfaces, input v
   259)     real(r8) :: dout_u                                   ! Vertical difference at interfaces, output u
   260)     real(r8) :: dout_v                                   ! Vertical difference at interfaces, output v
   261)     real(r8) :: dse_top(pcols)                           ! dse on top boundary
   262)     real(r8) :: cc_top(pcols)                            ! Lower diagonal at top interface
   263)     real(r8) :: cd_top(pcols)                            ! 
   264)     real(r8) :: rghd(pcols,pver+1)                       ! (1/H_i - 1/H) *(g*rho)^(-1)
   265) 
   266)     real(r8) :: qtm(pcols,pver)                          ! Temporary copy of q
   267)     real(r8) :: kq_scal(pcols,pver+1)                    ! kq_fac*sqrt(T)*m_d/rho for molecular diffusivity
   268)     real(r8) :: mw_fac(ncnst)                            ! sqrt(1/M_q + 1/M_d) for this constituent
   269)     real(r8) :: cnst_mw(ncnst)                           ! Molecular weight [ kg/kmole ]
   270)     real(r8) :: ubc_mmr(pcols,ncnst)                     ! Upper boundary mixing ratios [ kg/kg ]
   271)     real(r8) :: ubc_flux(ncnst)                          ! Upper boundary flux [ kg/s/m^2 ]
   272)     real(r8) :: ubc_t(pcols)                             ! Upper boundary temperature [ K ]
   273) 
   274)     real(r8) :: ws(pcols)                                ! Lowest-level wind speed [ m/s ]
   275)     real(r8) :: tau(pcols)                               ! Turbulent surface stress ( not including mountain stress )
   276)     real(r8) :: ksrfturb(pcols)                          ! Surface drag coefficient of 'normal' stress. > 0.
   277)                                                          ! Virtual mass input per unit time per unit area [ kg/s/m2 ]
   278)     real(r8) :: ksrf(pcols)                              ! Surface drag coefficient of 'normal' stress +
   279)                                                          ! Surface drag coefficient of 'tms' stress.  > 0. [ kg/s/m2 ] 
   280)     real(r8) :: usum_in(pcols)                           ! Vertical integral of input u-momentum. Total zonal
   281)                                                          ! momentum per unit area in column  [ sum of u*dp/g = kg m/s m-2 ]
   282)     real(r8) :: vsum_in(pcols)                           ! Vertical integral of input v-momentum. Total meridional
   283)                                                          ! momentum per unit area in column [ sum of v*dp/g = kg m/s m-2 ]
   284)     real(r8) :: usum_mid(pcols)                          ! Vertical integral of u-momentum after adding explicit residual stress
   285)     real(r8) :: vsum_mid(pcols)                          ! Vertical integral of v-momentum after adding explicit residual stress
   286)     real(r8) :: usum_out(pcols)                          ! Vertical integral of u-momentum after doing implicit diffusion
   287)     real(r8) :: vsum_out(pcols)                          ! Vertical integral of v-momentum after doing implicit diffusion
   288)     real(r8) :: tauimpx(pcols)                           ! Actual net stress added at the current step other than mountain stress
   289)     real(r8) :: tauimpy(pcols)                           ! Actual net stress added at the current step other than mountain stress
   290)     real(r8) :: wsmin                                    ! Minimum sfc wind speed for estimating frictional
   291)                                                          ! transfer velocity ksrf. [ m/s ]
   292)     real(r8) :: ksrfmin                                  ! Minimum surface drag coefficient [ kg/s/m^2 ]
   293)     real(r8) :: timeres                                  ! Relaxation time scale of residual stress ( >= dt ) [ s ]
   294)     real(r8) :: ramda                                    ! dt/timeres [ no unit ]
   295)     real(r8) :: psum
   296)     real(r8) :: u_in, u_res, tauresx_in
   297)     real(r8) :: v_in, v_res, tauresy_in  
   298)     
   299)     real(r8) :: mw_fac_loc(pcols,pver+1,ncnst)           ! Local sqrt(1/M_q + 1/M_d) for this constituent
   300) 
   301)     !--------------------------------
   302)     ! Variables needed for WACCM-X
   303)     !--------------------------------
   304)     logical  :: kvt_returned
   305)     real(r8) :: ttemp(pcols,pver)			 ! temporary temperature array
   306)     real(r8) :: ttemp0(pcols,pver)			 ! temporary temperature array
   307) 
   308)     ! ------------------------------------------------ !
   309)     ! Parameters for implicit surface stress treatment !
   310)     ! ------------------------------------------------ !
   311) 
   312)     wsmin    = 1._r8                                     ! Minimum wind speed for ksrfturb computation        [ m/s ]
   313)     ksrfmin  = 1.e-4_r8                                  ! Minimum surface drag coefficient                   [ kg/s/m^2 ]
   314)     timeres  = 7200._r8                                  ! Relaxation time scale of residual stress ( >= dt ) [ s ]
   315) 
   316)     ! ----------------------- !
   317)     ! Main Computation Begins !
   318)     ! ----------------------- !
   319) 
   320)     errstring = ''
   321)     if( ( diffuse(fieldlist,'u') .or. diffuse(fieldlist,'v') ) .and. .not. diffuse(fieldlist,'s') ) then
   322)           errstring = 'diffusion_solver.compute_vdiff: must diffuse s if diffusing u or v'
   323)           return
   324)     end if
   325)     zero(:) = 0._r8
   326) 
   327)     ! Compute 'rho' and 'dt*(g*rho)^2/dp' at interfaces
   328) 
   329)     tint(:ncol,1) = t(:ncol,1)
   330)     rhoi(:ncol,1) = pint(:ncol,1) / (rairi(:ncol,1)*tint(:ncol,1))
   331)     do k = 2, pver
   332)        do i = 1, ncol
   333)           tint(i,k)  = 0.5_r8 * ( t(i,k) + t(i,k-1) )
   334)           rhoi(i,k)  = pint(i,k) / (rairi(i,k)*tint(i,k))
   335)           tmpi2(i,k) = ztodt * ( gravit*rhoi(i,k) )**2 / ( pmid(i,k) - pmid(i,k-1) )
   336)        end do
   337)     end do
   338)     tint(:ncol,pver+1) = t(:ncol,pver)
   339)     rhoi(:ncol,pver+1) = pint(:ncol,pver+1) / ( rair*tint(:ncol,pver+1) )
   340) 
   341)     rrho(:ncol) = rair  * t(:ncol,pver) / pmid(:ncol,pver)
   342)     tmp1(:ncol) = ztodt * gravit * rpdel(:ncol,pver)
   343) 
   344)     !--------------------------------------- !
   345)     ! Computation of Molecular Diffusivities !
   346)     !--------------------------------------- !
   347) 
   348)   ! Modification : Why 'kvq' is not changed by molecular diffusion ? 
   349) 
   350)     if( do_molec_diff ) then
   351) 
   352)         if( (.not.present(compute_molec_diff)) .or. (.not.present(vd_lu_qdecomp)) .or. (.not.present(kvt)) ) then
   353)               errstring = 'compute_vdiff: do_molec_diff true but compute_molec_diff or vd_lu_qdecomp or kvt missing'
   354)               return
   355)         endif
   356) 
   357)       ! The next subroutine 'compute_molec_diff' :
   358)       !     Modifies : kvh, kvm, tint, rhoi, and tmpi2
   359)       !     Returns  : kq_scal, ubc_t, ubc_mmr, dse_top, cc_top, cd_top, cnst_mw, 
   360)       !                cnst_fixed_ubc , mw_fac , ntop_molec 
   361) 
   362)         mw_fac_loc(:,:,:) = 0._r8
   363) 
   364)         !--------------------------------------------------------------------------------------------------------
   365)         ! In Extended WACCM, kvt is calculated rather than kvh. This is because molecular diffusion operates on 
   366)         ! temperature, while eddy diffusion operates on dse.  Also, pass in constituent dependent "constants"
   367)         !--------------------------------------------------------------------------------------------------------
   368) 
   369)         status = compute_molec_diff( lchnk         ,                                                                              &
   370)                                      pcols         , pver            , ncnst      , ncol      , t      , pmid   , pint   ,        &
   371)                                      zi            , ztodt           , kvm        , kvt       , tint   , rhoi   , tmpi2  ,        &
   372)                                      kq_scal       , ubc_t           , ubc_mmr    , ubc_flux  , dse_top, cc_top , cd_top ,cnst_mw,&
   373)                                      cnst_fixed_ubc, cnst_fixed_ubflx, mw_fac_loc , ntop_molec, nbot_molec,       kvt_returned )
   374) 
   375)     else
   376) 
   377)         kq_scal(:,:) = 0._r8
   378)         cd_top(:)    = 0._r8
   379)         cc_top(:)    = 0._r8
   380)         kvt_returned = .false.
   381) 
   382)     endif
   383) 
   384)     !---------------------------- !
   385)     ! Diffuse Horizontal Momentum !
   386)     !---------------------------- !
   387) 
   388)     if( diffuse(fieldlist,'u') .or. diffuse(fieldlist,'v') ) then
   389) 
   390)         ! Compute the vertical upward differences of the input u,v for KE dissipation
   391)         ! at each interface.
   392)         ! Velocity = 0 at surface, so difference at the bottom interface is -u,v(pver)
   393)         ! These 'dinp_u, dinp_v' are computed using the non-diffused input wind.
   394) 
   395)         do i = 1, ncol
   396)            dinp_u(i,1) = 0._r8
   397)            dinp_v(i,1) = 0._r8
   398)            dinp_u(i,pver+1) = -u(i,pver)
   399)            dinp_v(i,pver+1) = -v(i,pver)
   400)         end do
   401)         do k = 2, pver
   402)            do i = 1, ncol
   403)               dinp_u(i,k) = u(i,k) - u(i,k-1)
   404)               dinp_v(i,k) = v(i,k) - v(i,k-1)
   405)            end do
   406)         end do
   407) 
   408)        ! -------------------------------------------------------------- !
   409)        ! Do 'Implicit Surface Stress' treatment for numerical stability !
   410)        ! in the lowest model layer.                                     !
   411)        ! -------------------------------------------------------------- !
   412) 
   413)        if( do_iss ) then
   414) 
   415)          ! Compute surface drag coefficient for implicit diffusion 
   416)          ! including turbulent turbulent mountain stress. 
   417) 
   418)            do i = 1, ncol
   419)               ws(i)       = max( sqrt( u(i,pver)**2._r8 + v(i,pver)**2._r8 ), wsmin )
   420)               tau(i)      = sqrt( taux(i)**2._r8 + tauy(i)**2._r8 )
   421)               ksrfturb(i) = max( tau(i) / ws(i), ksrfmin )
   422)            end do              
   423)            ksrf(:ncol) = ksrfturb(:ncol) + ksrftms(:ncol)  ! Do all surface stress ( normal + tms ) implicitly
   424) 
   425)          ! Vertical integration of input momentum. 
   426)          ! This is total horizontal momentum per unit area [ kg*m/s/m2 ] in each column.
   427)          ! Note (u,v) are the raw input to the PBL scheme, not the
   428)          ! provisionally-marched ones within the iteration loop of the PBL scheme.  
   429) 
   430)            do i = 1, ncol
   431)               usum_in(i) = 0._r8
   432)               vsum_in(i) = 0._r8
   433)               do k = 1, pver
   434)                  usum_in(i) = usum_in(i) + (1._r8/gravit)*u(i,k)/rpdel(i,k)
   435)                  vsum_in(i) = vsum_in(i) + (1._r8/gravit)*v(i,k)/rpdel(i,k)
   436)               end do
   437)            end do              
   438) 
   439)          ! Add residual stress of previous time step explicitly into the lowest
   440)          ! model layer with a relaxation time scale of 'timeres'.
   441) 
   442)            ramda         = ztodt / timeres
   443)            u(:ncol,pver) = u(:ncol,pver) + tmp1(:ncol)*tauresx(:ncol)*ramda
   444)            v(:ncol,pver) = v(:ncol,pver) + tmp1(:ncol)*tauresy(:ncol)*ramda
   445) 
   446)          ! Vertical integration of momentum after adding explicit residual stress
   447)          ! into the lowest model layer.
   448) 
   449)            do i = 1, ncol
   450)               usum_mid(i) = 0._r8
   451)               vsum_mid(i) = 0._r8
   452)               do k = 1, pver
   453)                  usum_mid(i) = usum_mid(i) + (1._r8/gravit)*u(i,k)/rpdel(i,k)
   454)                  vsum_mid(i) = vsum_mid(i) + (1._r8/gravit)*v(i,k)/rpdel(i,k)
   455)               end do
   456)            end do              
   457) 
   458)          ! Debug 
   459)          ! icol = phys_debug_col(lchnk) 
   460)          ! if ( icol > 0 .and. get_nstep() .ge. 1 ) then
   461)          !      tauresx_in = tauresx(icol)
   462)          !      tauresy_in = tauresy(icol)
   463)          !      u_in  = u(icol,pver) - tmp1(icol) * tauresx(icol) * ramda
   464)          !      v_in  = v(icol,pver) - tmp1(icol) * tauresy(icol) * ramda
   465)          !      u_res = u(icol,pver)
   466)          !      v_res = v(icol,pver)
   467)          ! endif
   468)          ! Debug
   469) 
   470)        else
   471) 
   472)          ! In this case, do 'turbulent mountain stress' implicitly, 
   473)          ! but do 'normal turbulent stress' explicitly.
   474)          ! In this case, there is no 'redisual stress' as long as 'tms' is
   475)          ! treated in a fully implicit wway, which is true.
   476) 
   477)          ! 1. Do 'tms' implicitly
   478) 
   479)            ksrf(:ncol) = ksrftms(:ncol) 
   480) 
   481)          ! 2. Do 'normal stress' explicitly
   482) 
   483)            u(:ncol,pver) = u(:ncol,pver) + tmp1(:ncol)*taux(:ncol)
   484)            v(:ncol,pver) = v(:ncol,pver) + tmp1(:ncol)*tauy(:ncol)
   485) 
   486)        end if  ! End of 'do iss' ( implicit surface stress )
   487) 
   488)        ! --------------------------------------------------------------------------------------- !
   489)        ! Diffuse horizontal momentum implicitly using tri-diagnonal matrix.                      !
   490)        ! The 'u,v' are input-output: the output 'u,v' are implicitly diffused winds.             !
   491)        !    For implicit 'normal' stress : ksrf = ksrftms + ksrfturb,                            !
   492)        !                                   u(pver) : explicitly include 'redisual normal' stress !
   493)        !    For explicit 'normal' stress : ksrf = ksrftms                                        !
   494)        !                                   u(pver) : explicitly include 'normal' stress          !
   495)        ! Note that in all the two cases above, 'tms' is fully implicitly treated.                !
   496)        ! --------------------------------------------------------------------------------------- !
   497) 
   498)        call vd_lu_decomp( pcols , pver , ncol  ,                         &
   499)                           ksrf  , kvm  , tmpi2 , rpdel , ztodt , gravit, &
   500)                           zero  , ntop , nbot  , decomp)
   501) 
   502)        call vd_lu_solve(  pcols , pver  , ncol  ,                        &
   503)                           u     , decomp, ntop  , nbot , zero )
   504) 
   505)        call vd_lu_solve(  pcols , pver  , ncol  ,                        &
   506)                           v     , decomp, ntop  , nbot , zero )
   507) 
   508)        ! ---------------------------------------------------------------------- !
   509)        ! Calculate 'total' ( tautotx ) and 'tms' ( tautmsx ) stresses that      !
   510)        ! have been actually added into the atmosphere at the current time step. ! 
   511)        ! Also, update residual stress, if required.                             !
   512)        ! ---------------------------------------------------------------------- !
   513) 
   514)        do i = 1, ncol
   515) 
   516)           ! Compute the implicit 'tms' using the updated winds.
   517)           ! Below 'tautmsx(i),tautmsy(i)' are pure implicit mountain stresses
   518)           ! that has been actually added into the atmosphere both for explicit
   519)           ! and implicit approach. 
   520) 
   521)           tautmsx(i) = -ksrftms(i)*u(i,pver)
   522)           tautmsy(i) = -ksrftms(i)*v(i,pver)
   523) 
   524)           if( do_iss ) then
   525) 
   526)             ! Compute vertical integration of final horizontal momentum
   527) 
   528)               usum_out(i) = 0._r8
   529)               vsum_out(i) = 0._r8
   530)               do k = 1, pver
   531)                  usum_out(i) = usum_out(i) + (1._r8/gravit)*u(i,k)/rpdel(i,k)
   532)                  vsum_out(i) = vsum_out(i) + (1._r8/gravit)*v(i,k)/rpdel(i,k)
   533)               end do
   534) 
   535)             ! Compute net stress added into the atmosphere at the current time step.
   536)             ! Note that the difference between 'usum_in' and 'usum_out' are induced
   537)             ! by 'explicit residual stress + implicit total stress' for implicit case, while
   538)             ! by 'explicit normal   stress + implicit tms   stress' for explicit case. 
   539)             ! Here, 'tautotx(i)' is net stress added into the air at the current time step.
   540) 
   541)               tauimpx(i) = ( usum_out(i) - usum_in(i) ) / ztodt
   542)               tauimpy(i) = ( vsum_out(i) - vsum_in(i) ) / ztodt
   543) 
   544)               tautotx(i) = tauimpx(i) 
   545)               tautoty(i) = tauimpy(i) 
   546) 
   547)             ! Compute redisual stress and update if required.
   548)             ! Note that the total stress we should have added at the current step is
   549)             ! the sum of 'taux(i) - ksrftms(i)*u(i,pver) + tauresx(i)'.
   550) 
   551)               if( itaures .eq. 1 ) then
   552)                   tauresx(i) = taux(i) + tautmsx(i) + tauresx(i) - tauimpx(i)
   553)                   tauresy(i) = tauy(i) + tautmsy(i) + tauresy(i) - tauimpy(i)
   554)               endif
   555) 
   556)           else
   557) 
   558)               tautotx(i) = tautmsx(i) + taux(i)
   559)               tautoty(i) = tautmsy(i) + tauy(i)
   560)               tauresx(i) = 0._r8
   561)               tauresy(i) = 0._r8
   562) 
   563)           end if  ! End of 'do_iss' if
   564) 
   565)        end do ! End of 'do i = 1, ncol' loop
   566) 
   567)      ! Debug 
   568)      ! icol = phys_debug_col(lchnk) 
   569)      ! if ( icol > 0 .and. get_nstep() .ge. 1 ) then
   570)      !      write(iulog,*)
   571)      !      write(iulog,*)  'diffusion_solver debug'  
   572)      !      write(iulog,*)
   573)      !      write(iulog,*)  'u_in, u_res, u_out'
   574)      !      write(iulog,*)   u_in, u_res, u(icol,pver)
   575)      !      write(iulog,*)  'tauresx_in, tautmsx, tauimpx(actual), tauimpx(derived), tauresx_out, taux'
   576)      !      write(iulog,*)   tauresx_in, tautmsx(icol), tauimpx(icol), -ksrf(icol)*u(icol,pver), tauresx(icol), taux(icol)
   577)      !      write(iulog,*)
   578)      !      write(iulog,*)  'v_in, v_res, v_out'
   579)      !      write(iulog,*)   v_in, v_res, v(icol,pver)
   580)      !      write(iulog,*)  'tauresy_in, tautmsy, tauimpy(actual), tauimpy(derived), tauresy_out, tauy'
   581)      !      write(iulog,*)   tauresy_in, tautmsy(icol), tauimpy(icol), -ksrf(icol)*v(icol,pver), tauresy(icol), tauy(icol)
   582)      !      write(iulog,*)
   583)      !      write(iulog,*)  'itaures, ksrf, ksrfturb, ksrftms'
   584)      !      write(iulog,*)   itaures, ksrf(icol), ksrfturb(icol), ksrftms(icol)
   585)      !      write(iulog,*) 
   586)      ! endif
   587)      ! Debug
   588) 
   589)        ! ------------------------------------ !
   590)        ! Calculate kinetic energy dissipation !
   591)        ! ------------------------------------ !       
   592) 
   593)      ! Modification : In future, this should be set exactly same as 
   594)      !                the ones in the convection schemes 
   595) 
   596)        ! 1. Compute dissipation term at interfaces
   597)        !    Note that 'u,v' are already diffused wind, and 'tautotx,tautoty' are 
   598)        !    implicit stress that has been actually added. On the other hand,
   599)        !    'dinp_u, dinp_v' were computed using non-diffused input wind.
   600) 
   601)      ! Modification : I should check whether non-consistency between 'u' and 'dinp_u'
   602)      !                is correctly intended approach. I think so.
   603) 
   604)        k = pver + 1
   605)        do i = 1, ncol
   606)           tmpi1(i,1) = 0._r8
   607)           tmpi1(i,k) = 0.5_r8 * ztodt * gravit * &
   608)                        ( (-u(i,k-1) + dinp_u(i,k))*tautotx(i) + (-v(i,k-1) + dinp_v(i,k))*tautoty(i) )
   609)        end do
   610) 
   611)        do k = 2, pver
   612)           do i = 1, ncol
   613)              dout_u = u(i,k) - u(i,k-1)
   614)              dout_v = v(i,k) - v(i,k-1)
   615)              tmpi1(i,k) = 0.25_r8 * tmpi2(i,k) * kvm(i,k) * &
   616)                           ( dout_u**2 + dout_v**2 + dout_u*dinp_u(i,k) + dout_v*dinp_v(i,k) )
   617)           end do
   618)        end do
   619) 
   620)        ! 2. Compute dissipation term at midpoints, add to dry static energy
   621) 
   622)        do k = 1, pver
   623)           do i = 1, ncol
   624)              dtk(i,k) = ( tmpi1(i,k+1) + tmpi1(i,k) ) * rpdel(i,k)
   625)              dse(i,k) = dse(i,k) + dtk(i,k)
   626)           end do
   627)        end do
   628) 
   629)     end if ! End of diffuse horizontal momentum, diffuse(fieldlist,'u') routine
   630) 
   631)     !-------------------------- !
   632)     ! Diffuse Dry Static Energy !
   633)     !-------------------------- !
   634) 
   635)   ! Modification : In future, we should diffuse the fully conservative 
   636)   !                moist static energy,not the dry static energy.
   637) 
   638)     if( diffuse(fieldlist,'s') ) then
   639) 
   640)       ! Add counter-gradient to input static energy profiles
   641) 
   642)         do k = 1, pver
   643)            dse(:ncol,k) = dse(:ncol,k) + ztodt * rpdel(:ncol,k) * gravit  *                &
   644)                                        ( rhoi(:ncol,k+1) * kvh(:ncol,k+1) * cgh(:ncol,k+1) &
   645)                                        - rhoi(:ncol,k  ) * kvh(:ncol,k  ) * cgh(:ncol,k  ) )
   646)        end do
   647) 
   648)      ! Add the explicit surface fluxes to the lowest layer
   649) 
   650)        dse(:ncol,pver) = dse(:ncol,pver) + tmp1(:ncol) * shflx(:ncol)
   651) 
   652)      ! Diffuse dry static energy
   653)      !----------------------------------------------------------------------------------------------------
   654)      ! In Extended WACCM, kvt is calculated rather kvh. This is because molecular diffusion operates on 
   655)      ! temperature, while eddy diffusion operates on dse.  Also, pass in constituent dependent "constants"
   656)      !----------------------------------------------------------------------------------------------------
   657)        if ( kvt_returned ) then 
   658)          cc_top(:) = 0._r8      
   659)          cd_top(:) = 0._r8
   660)        endif
   661) 
   662)        call vd_lu_decomp( pcols , pver , ncol  ,                         &
   663)                           zero  , kvh  , tmpi2 , rpdel , ztodt , gravit, &
   664)                           cc_top, ntop , nbot  , decomp )
   665) 
   666)        call vd_lu_solve(  pcols , pver  , ncol  ,                         &
   667)                           dse   , decomp, ntop  , nbot  , cd_top )
   668) 
   669)      ! Calculate flux at top interface
   670)      
   671)      ! Modification : Why molecular diffusion does not work for dry static energy in all layers ?
   672) 
   673)        if( do_molec_diff ) then
   674)            topflx(:ncol) =  - kvh(:ncol,ntop_molec) * tmpi2(:ncol,ntop_molec) / (ztodt*gravit) * &
   675)                             ( dse(:ncol,ntop_molec) - dse_top(:ncol) )
   676)        end if
   677) 
   678)        !---------------------------------------------------
   679)        ! Solve for temperature using thermal conductivity 
   680)        !---------------------------------------------------
   681)        if ( kvt_returned ) then 
   682) 
   683)          call vd_lu_decomp( pcols , pver , ncol  ,                         &
   684)                             zero  , kvt  , tmpi2 , rpdel , ztodt , gravit, &
   685)                             zero  , ntop , nbot  , decomp, cpairv )
   686) 
   687)          do k = 1,pver
   688)            ttemp0(:ncol,k) = t(:ncol,k)
   689)            ttemp(:ncol,k) = ttemp0(:ncol,k)
   690)          enddo
   691) 
   692)          call vd_lu_solve(                                                &
   693)             pcols, pver, ncol     ,                                       &
   694)             ttemp      ,decomp    ,ntop     ,nbot,zero    )   ! upper boundary is zero flux for extended model
   695)     
   696)          !-------------------------------------
   697)          !  Update dry static energy
   698)          !-------------------------------------
   699)          do k = 1,pver
   700)            dse(:ncol,k) = dse(:ncol,k) + cpairv(:ncol,k)*(ttemp(:ncol,k) - ttemp0(:ncol,k))
   701)          enddo
   702) 
   703)        endif
   704) 
   705)     endif
   706) 
   707)     !---------------------------- !
   708)     ! Diffuse Water Vapor Tracers !
   709)     !---------------------------- !
   710) 
   711)   ! Modification : For aerosols, I need to use separate treatment 
   712)   !                for aerosol mass and aerosol number. 
   713) 
   714)     ! Loop through constituents
   715) 
   716)     need_decomp = .true.
   717) 
   718)     do m = 1, ncnst
   719) 
   720)        if( diffuse(fieldlist,'q',m) ) then
   721) 
   722)            ! Add the nonlocal transport terms to constituents in the PBL.
   723)            ! Check for neg q's in each constituent and put the original vertical
   724)            ! profile back if a neg value is found. A neg value implies that the
   725)            ! quasi-equilibrium conditions assumed for the countergradient term are
   726)            ! strongly violated.
   727) 
   728)            qtm(:ncol,:pver) = q(:ncol,:pver,m)
   729) 
   730)            do k = 1, pver
   731)               q(:ncol,k,m) = q(:ncol,k,m) + &
   732)                              ztodt * rpdel(:ncol,k) * gravit  * ( cflx(:ncol,m) * rrho(:ncol) ) * &
   733)                            ( rhoi(:ncol,k+1) * kvh(:ncol,k+1) * cgs(:ncol,k+1)                    &
   734)                            - rhoi(:ncol,k  ) * kvh(:ncol,k  ) * cgs(:ncol,k  ) )
   735)            end do
   736)            lqtst(:ncol) = all(q(:ncol,1:pver,m) >= qmincg(m), 2)
   737)            do k = 1, pver
   738)               q(:ncol,k,m) = merge( q(:ncol,k,m), qtm(:ncol,k), lqtst(:ncol) )
   739)            end do
   740) 
   741)            ! Add the explicit surface fluxes to the lowest layer
   742) 
   743)            q(:ncol,pver,m) = q(:ncol,pver,m) + tmp1(:ncol) * cflx(:ncol,m)
   744) 
   745)            ! Diffuse constituents.
   746) 
   747)            if( need_decomp ) then
   748) 
   749)                call vd_lu_decomp( pcols , pver , ncol  ,                         &
   750)                                   zero  , kvq  , tmpi2 , rpdel , ztodt , gravit, &
   751)                                   zero  , ntop , nbot  , decomp)
   752) 
   753)                if( do_molec_diff ) then
   754) 
   755)                  ! This is for solving molecular diffusion of minor species, thus, for WACCM-X, bypass O and O2 (major species)
   756)                  ! Major species diffusion is calculated separately.  -Hanli Liu
   757)                  !
   758)                    if ( diffuse(fieldlistm,'q',m) ) then                  ! decide if diffuse this constituent
   759) 
   760)                      status = vd_lu_qdecomp( pcols , pver   , ncol              , cnst_fixed_ubc(m), cnst_mw(m), ubc_mmr(:,m), &
   761)                                              kvq   , kq_scal, mw_fac_loc(:,:,m) ,tmpi2             , rpdel     ,               &
   762)                                              decomp, rhoi      ,               &
   763)                                              tint  , ztodt  , ntop_molec        , nbot_molec       , cd_top    ,               &
   764)                                              lchnk , pmid   , pint              , t                , m         )
   765) 
   766) 
   767)                      ! This to calculate the upper boundary flux of H.    -Hanli Liu
   768)                      if ((cnst_fixed_ubflx(m))) &
   769)                             q(:ncol,1,m) = q(:ncol,1,m) - ztodt * rpdel(:ncol,1) * gravit * ubc_flux(m)
   770) 
   771)                    endif
   772) 
   773)                else
   774)                    need_decomp =  .false.
   775)                endif
   776)            end if
   777) 
   778)            call vd_lu_solve(  pcols , pver , ncol  ,                         &
   779)                               q(1,1,m) , decomp    , ntop  , nbot  , cd_top )
   780)        end if
   781)     end do
   782) 
   783)     ! Probably unnecessary, but just in case there's an OpenMP problem,
   784)     ! deallocate the decomp object.
   785)     call decomp%finalize()
   786) 
   787)     return
   788)   end subroutine compute_vdiff
