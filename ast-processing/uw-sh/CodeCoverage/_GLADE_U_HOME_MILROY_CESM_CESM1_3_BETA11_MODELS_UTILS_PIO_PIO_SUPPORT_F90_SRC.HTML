   385)    subroutine pio_fc_gather_offset ( sendbuf, sendcnt, sendtype, &
   386)                                   recvbuf, recvcnt, recvtype, &
   387)                                   root, comm, flow_cntl )
   388) 
   389) !----------------------------------------------------------------------- 
   390) ! 
   391) !> Purpose: 
   392) !!   Gather collective with additional flow control, so as to 
   393) !!   be more robust when used with high process counts. 
   394) !!
   395) !! Method: 
   396) !!   If flow_cntl optional parameter 
   397) !!     < 0: use MPI_Gather
   398) !!     >= 0: use point-to-point with handshaking messages and 
   399) !!           preposting receive requests up to 
   400) !!           max(min(1,flow_cntl),max_gather_block_size) 
   401) !!           ahead if optional flow_cntl parameter is present.
   402) !!           Otherwise, fc_gather_flow_cntl is used in its place.
   403) !!     Default value is 64.
   404) !! 
   405) !! Author of original version:  P. Worley
   406) !! Ported from CAM: P. Worley, Jan 2010
   407) !< 
   408) !-----------------------------------------------------------------------
   409) 
   410) !-----------------------------------------------------------------------
   411)    implicit none
   412) 
   413) !---------------------------Parameters ---------------------------------
   414) !
   415)    integer, parameter :: max_gather_block_size = 64
   416) 
   417) !---------------------------Input arguments--------------------------
   418) !
   419)    integer(kind=pio_offset), intent(in)  :: sendbuf(:)       ! outgoing message buffer
   420)    integer, intent(in)  :: sendcnt          ! size of send buffer
   421)    integer, intent(in)  :: sendtype         ! MPI type of send buffer
   422)    integer, intent(in)  :: recvcnt          ! size of receive buffer
   423)    integer, intent(in)  :: recvtype         ! MPI type of receive buffer
   424)    integer, intent(in)  :: root             ! gather destination
   425)    integer, intent(in)  :: comm             ! MPI communicator
   426)    integer,optional, intent(in):: flow_cntl ! flow control variable
   427) 
   428) !---------------------------Output arguments--------------------------
   429) !
   430)    integer(kind=pio_offset), intent(out) :: recvbuf(*)       ! incoming message buffer
   431) !
   432) !---------------------------Local workspace---------------------------------
   433) !
   434)    character(len=*), parameter :: subName=modName//'::pio_fc_gather_int'
   435) 
   436)    logical :: fc_gather                     ! use explicit flow control?
   437)    integer :: hs                            ! handshake variable
   438)    integer :: gather_block_size             ! number of preposted receive requests
   439) 
   440)    integer :: nprocs                        ! size of communicator
   441)    integer :: mytask                        ! MPI task id with communicator
   442)    integer :: mtag                          ! MPI message tag
   443)    integer :: p, i                          ! loop indices
   444)    integer :: displs                        ! offset into receive buffer
   445)    integer :: count, preposts, head, tail   ! variables controlling recv-ahead logic
   446) 
   447)    integer :: rcvid(max_gather_block_size)  ! receive request ids
   448) 
   449)    integer :: ier                           ! return error status    
   450)    integer :: status(MPI_STATUS_SIZE)       ! MPI status 
   451) 
   452) !
   453) !-------------------------------------------------------------------------------------
   454) !
   455)    if ( present(flow_cntl) ) then
   456)       if (flow_cntl >= 0) then
   457)          gather_block_size = min(max(1,flow_cntl),max_gather_block_size)
   458)          fc_gather = .true.
   459)       else
   460)          fc_gather = .false.
   461)       endif
   462)    else
   463) #ifndef _NO_FLOW_CONTROL
   464)       gather_block_size = max(1,max_gather_block_size)
   465)       fc_gather = .true.
   466) #else
   467)       fc_gather = .false.
   468) #endif
   469)    endif
   470) 
   471)    if (fc_gather) then
   472)  
   473)       ! Determine task id and size of communicator
   474)       call mpi_comm_rank (comm, mytask, ier)
   475)       call mpi_comm_size (comm, nprocs, ier)
   476) 
   477)       ! Initialize tag and hs variable
   478) #ifdef _NO_PIO_SWAPM_TAG_OFFSET
   479)       mtag = 0
   480) #else
   481)       mtag = 2*nprocs
   482) #endif
   483)       hs = 1
   484) 
   485)       if (root .eq. mytask) then
   486) 
   487) ! prepost gather_block_size irecvs, and start receiving data
   488)          preposts = min(nprocs-1, gather_block_size)
   489)          head = 0
   490)          count = 0
   491)          do p=0, nprocs-1
   492)             if (p .ne. root) then
   493)                if (recvcnt > 0) then
   494)                   count = count + 1
   495)                   if (count > preposts) then
   496)                      tail = mod(head,preposts) + 1
   497)                      call mpi_wait (rcvid(tail), status, ier)
   498)                   end if
   499)                   head = mod(head,preposts) + 1
   500)                   displs = p*recvcnt
   501)                   call mpi_irecv ( recvbuf(displs+1), recvcnt, &
   502)                                    recvtype, p, mtag, comm, rcvid(head), &
   503)                                    ier )
   504)                   call mpi_send ( hs, 1, recvtype, p, mtag, comm, ier )
   505)                end if
   506)             end if
   507)          end do
   508) 
   509) ! copy local data
   510)          displs = mytask*recvcnt
   511)          do i=1,sendcnt
   512)             recvbuf(displs+i) = sendbuf(i)
   513)          enddo
   514) 
   515) ! wait for final data
   516)          do i=1,min(count,preposts)
   517)             call mpi_wait (rcvid(i), status, ier)
   518)          enddo
   519) 
   520)       else
   521) 
   522)          if (sendcnt > 0) then
   523)             call mpi_recv  ( hs, 1, sendtype, root, mtag, comm, &
   524)                              status, ier )
   525)             call mpi_rsend ( sendbuf, sendcnt, sendtype, root, mtag, &
   526)                              comm, ier )
   527)          end if
   528) 
   529)       endif
   530)       call CheckMPIReturn(subName,ier)
   531) 
   532)    else
   533)  
   534)       call mpi_gather (sendbuf, sendcnt, sendtype, &
   535)                        recvbuf, recvcnt, recvtype, &
   536)                        root, comm, ier)
   537)       call CheckMPIReturn(subName,ier)
   538) 
   539)    endif
   540) 
   541)    return
   542) 
   543)    end subroutine pio_fc_gather_offset
