    35) ! !INTERFACE:
    36)   subroutine SurfaceRunoff (lbc, ubc, lbp, ubp, num_hydrologyc, filter_hydrologyc, &
    37)                             num_urbanc, filter_urbanc, vol_liq, icefrac)
    38) !
    39) ! !DESCRIPTION:
    40) ! Calculate surface runoff
    41) !
    42) ! !USES:
    43)     use shr_kind_mod    , only : r8 => shr_kind_r8
    44)     use clmtype
    45)     use clm_varcon      , only : denice, denh2o, wimp, pondmx_urban, &
    46)                                  icol_roof, icol_sunwall, icol_shadewall, &
    47)                                  icol_road_imperv, icol_road_perv
    48)                              
    49)     use clm_varpar      , only : nlevsoi, maxpatch_pft
    50)     use clm_time_manager, only : get_step_size
    51) !
    52) ! !ARGUMENTS:
    53)     implicit none
    54)     integer , intent(in)  :: lbc, ubc                     ! column bounds
    55)     integer , intent(in)  :: lbp, ubp                     ! pft bounds   
    56)     integer , intent(in)  :: num_hydrologyc               ! number of column soil points in column filter
    57)     integer , intent(in)  :: filter_hydrologyc(ubc-lbc+1) ! column filter for soil points
    58)     integer , intent(in)  :: num_urbanc                   ! number of column urban points in column filter
    59)     integer , intent(in)  :: filter_urbanc(ubc-lbc+1)     ! column filter for urban points
    60)     real(r8), intent(out) :: vol_liq(lbc:ubc,1:nlevsoi)   ! partial volume of liquid water in layer
    61)     real(r8), intent(out) :: icefrac(lbc:ubc,1:nlevsoi)   ! fraction of ice in layer (-)
    62) !
    63) ! !CALLED FROM:
    64) ! subroutine Hydrology2 in module Hydrology2Mod
    65) !
    66) ! !REVISION HISTORY:
    67) ! 15 September 1999: Yongjiu Dai; Initial code
    68) ! 12 November 1999:  Z.-L. Yang and G.-Y. Niu
    69) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
    70) ! 2/26/02, Peter Thornton: Migrated to new data structures.
    71) ! 4/26/05, David Lawrence: Made surface runoff for dry soils a function
    72) !   of rooting fraction in top three soil layers.
    73) ! 04/25/07  Keith Oleson: Completely new routine for CLM3.5 hydrology
    74) !
    75) ! !LOCAL VARIABLES:
    76) !
    77) ! local pointers to original implicit in arguments
    78) !
    79) !rtm_flood
    80)     real(r8), pointer :: qflx_floodc(:)   ! column flux of flood water from RTM
    81) !rtm_flood
    82)     integer , pointer :: cgridcell(:)      ! gridcell index for each column
    83)     integer , pointer :: ctype(:)          ! column type index
    84)     real(r8), pointer :: qflx_top_soil(:)  !net water input into soil from top (mm/s)
    85)     real(r8), pointer :: watsat(:,:)       !volumetric soil water at saturation (porosity)
    86)     real(r8), pointer :: hkdepth(:)        !decay factor (m)
    87)     real(r8), pointer :: zwt(:)            !water table depth (m)
    88)     real(r8), pointer :: fcov(:)           !fractional impermeable area
    89)     real(r8), pointer :: fsat(:)           !fractional area with water table at surface
    90)     real(r8), pointer :: dz(:,:)           !layer depth (m)
    91)     real(r8), pointer :: h2osoi_ice(:,:)   !ice lens (kg/m2)
    92)     real(r8), pointer :: h2osoi_liq(:,:)   !liquid water (kg/m2)
    93)     real(r8), pointer :: wtfact(:)         !maximum saturated fraction for a gridcell
    94)     real(r8), pointer :: hksat(:,:)        ! hydraulic conductivity at saturation (mm H2O /s)
    95)     real(r8), pointer :: bsw(:,:)          ! Clapp and Hornberger "b"
    96)     real(r8), pointer :: sucsat(:,:)       ! minimum soil suction (mm)
    97)     integer , pointer :: snl(:)            ! minus number of snow layers
    98)     real(r8), pointer :: qflx_evap_grnd(:) ! ground surface evaporation rate (mm H2O/s) [+]
    99)     real(r8), pointer :: zi(:,:)           ! interface level below a "z" level (m)
   100) !
   101) ! local pointers to original implicit out arguments
   102) !
   103)     real(r8), pointer :: qflx_surf(:)      ! surface runoff (mm H2O /s)
   104)     real(r8), pointer :: eff_porosity(:,:) ! effective porosity = porosity - vol_ice
   105)     real(r8), pointer :: fracice(:,:)      !fractional impermeability (-)
   106) !
   107) !EOP
   108) !
   109) ! !OTHER LOCAL VARIABLES:
   110) !
   111)     integer  :: c,j,fc,g                   !indices
   112)     real(r8) :: dtime                      ! land model time step (sec)
   113)     real(r8) :: xs(lbc:ubc)                ! excess soil water above urban ponding limit
   114)     real(r8) :: vol_ice(lbc:ubc,1:nlevsoi) !partial volume of ice lens in layer
   115)     real(r8) :: fff(lbc:ubc)               !decay factor (m-1)
   116)     real(r8) :: s1                         !variable to calculate qinmax
   117)     real(r8) :: su                         !variable to calculate qinmax
   118)     real(r8) :: v                          !variable to calculate qinmax
   119)     real(r8) :: qinmax                     !maximum infiltration capacity (mm/s)
   120) 
   121) !-----------------------------------------------------------------------
   122) 
   123)     ! Assign local pointers to derived subtype components (column-level)
   124) 
   125) !rtm_flood
   126)     qflx_floodc      => cwf%qflx_floodc
   127) !rtm_flood
   128)     ctype         => col%itype
   129)     qflx_top_soil => cwf%qflx_top_soil
   130)     qflx_surf     => cwf%qflx_surf
   131)     watsat        => cps%watsat
   132)     hkdepth       => cps%hkdepth
   133)     dz            => cps%dz
   134)     h2osoi_ice    => cws%h2osoi_ice
   135)     h2osoi_liq    => cws%h2osoi_liq
   136)     fcov          => cws%fcov
   137)     fsat          => cws%fsat
   138)     eff_porosity  => cps%eff_porosity
   139)     wtfact        => cps%wtfact
   140)     zwt           => cws%zwt
   141)     fracice       => cps%fracice
   142)     hksat         => cps%hksat
   143)     bsw           => cps%bsw
   144)     sucsat        => cps%sucsat
   145)     snl            => cps%snl
   146)     qflx_evap_grnd => pwf_a%qflx_evap_grnd
   147)     zi            => cps%zi
   148) 
   149)     ! Get time step
   150) 
   151)     dtime = get_step_size()
   152) 
   153)     do j = 1,nlevsoi
   154)        do fc = 1, num_hydrologyc
   155)           c = filter_hydrologyc(fc)
   156) 
   157)           ! Porosity of soil, partial volume of ice and liquid, fraction of ice in each layer,
   158)           ! fractional impermeability
   159)    
   160)           vol_ice(c,j) = min(watsat(c,j), h2osoi_ice(c,j)/(dz(c,j)*denice))
   161)           eff_porosity(c,j) = max(0.01_r8,watsat(c,j)-vol_ice(c,j))
   162)           vol_liq(c,j) = min(eff_porosity(c,j), h2osoi_liq(c,j)/(dz(c,j)*denh2o))
   163) 
   164)           icefrac(c,j) = min(1._r8,h2osoi_ice(c,j)/(h2osoi_ice(c,j)+h2osoi_liq(c,j)))
   165) 
   166)           fracice(c,j) = max(0._r8,exp(-3._r8*(1._r8-icefrac(c,j)))- exp(-3._r8))/(1.0_r8-exp(-3._r8))
   167)        end do
   168)     end do
   169) 
   170)     ! Saturated fraction
   171) 
   172)     do fc = 1, num_hydrologyc
   173)        c = filter_hydrologyc(fc)
   174)        fff(c) = 0.5_r8
   175)        fsat(c) = wtfact(c) * exp(-0.5_r8*fff(c)*zwt(c))
   176)        fcov(c) = (1._r8 - fracice(c,1)) * fsat(c) + fracice(c,1)
   177)     end do
   178) 
   179)     do fc = 1, num_hydrologyc
   180)        c = filter_hydrologyc(fc)
   181) 
   182)        ! Maximum infiltration capacity
   183)        s1        = max(0.01_r8,vol_liq(c,1)/max(wimp,eff_porosity(c,1)))
   184)        su        = max(0._r8,(s1-fcov(c)) / (max(0.01_r8,1._r8-fcov(c))))
   185)        v         = -bsw(c,1)*sucsat(c,1)/(0.5_r8*dz(c,1)*1000._r8)
   186)        qinmax    = (1._r8+v*(su-1._r8))*hksat(c,1)
   187) 
   188)        ! Surface runoff
   189)        qflx_surf(c) =  fcov(c) * qflx_top_soil(c) + &
   190)                        (1._r8-fcov(c)) * max(0._r8, qflx_top_soil(c)-qinmax)
   191) 
   192)     end do
   193) 
   194)     ! Determine water in excess of ponding limit for urban roof and impervious road.
   195)     ! Excess goes to surface runoff. No surface runoff for sunwall and shadewall.
   196) 
   197)     do fc = 1, num_urbanc
   198)        c = filter_urbanc(fc)
   199)        if (ctype(c) == icol_roof .or. ctype(c) == icol_road_imperv) then
   200) 
   201)           ! If there are snow layers then all qflx_top_soil goes to surface runoff
   202)           if (snl(c) < 0) then
   203)              qflx_surf(c) = max(0._r8,qflx_top_soil(c))
   204)           else
   205)              xs(c) = max(0._r8, &
   206)                          h2osoi_liq(c,1)/dtime + qflx_top_soil(c) - qflx_evap_grnd(c) - &
   207)                          pondmx_urban/dtime)
   208)              if (xs(c) > 0.) then
   209)                 h2osoi_liq(c,1) = pondmx_urban
   210)              else
   211)                 h2osoi_liq(c,1) = max(0._r8,h2osoi_liq(c,1)+ &
   212)                                      (qflx_top_soil(c)-qflx_evap_grnd(c))*dtime)
   213)              end if
   214)              qflx_surf(c) = xs(c)
   215)           end if
   216)        else if (ctype(c) == icol_sunwall .or. ctype(c) == icol_shadewall) then
   217)          qflx_surf(c) = 0._r8
   218)        end if
   219) !rtm_flood:  send flood water flux to runoff for all urban columns
   220)        qflx_surf(c) = qflx_surf(c)  + qflx_floodc(c)
   221) !rtm_flood
   222)     end do
   223) 
   224) !rtm_flood: add qflx_flood to qflx_top_soil 
   225) !dir$ concurrent
   226) !cdir nodep
   227)     do fc = 1, num_hydrologyc
   228)        c = filter_hydrologyc(fc)
   229)        qflx_top_soil(c) = qflx_top_soil(c) + qflx_floodc(c)
   230)     end do
   231)   end subroutine SurfaceRunoff

   243) ! Calculate infiltration into surface soil layer (minus the evaporation)
   244) !
   245) ! !USES:

   247)     use clm_varcon  , only : icol_roof, icol_road_imperv, icol_sunwall, icol_shadewall, &

   249)     use clmtype

   271)     integer , pointer :: ctype(:)         ! column type index

   273)     real(r8), pointer :: qflx_top_soil(:) ! net water input into soil from top (mm/s)

   284) !
   285)     integer :: c, fc    !indices
   286) !-----------------------------------------------------------------------

   288)     ! Assign local pointers to derived type members (column-level)
   289) 
   290)     ctype          => col%itype
   291)     snl            => cps%snl
   292)     qflx_top_soil  => cwf%qflx_top_soil
   293)     qflx_surf      => cwf%qflx_surf
   294)     qflx_infl      => cwf%qflx_infl
   295)     qflx_evap_grnd => pwf_a%qflx_evap_grnd
   296) 
   297)     ! Infiltration into surface soil layer (minus the evaporation)
   298) 
   299)     do fc = 1, num_hydrologyc
   300)        c = filter_hydrologyc(fc)
   301)        if (snl(c) >= 0) then
   302)           qflx_infl(c) = qflx_top_soil(c) - qflx_surf(c) - qflx_evap_grnd(c)

   309) 
   310)     do fc = 1, num_urbanc
   311)        c = filter_urbanc(fc)
   312)        if (ctype(c) /= icol_road_perv) then
   313)           qflx_infl(c) = 0._r8
   314)        end if
   315)     end do
   316)

   322) ! !IROUTINE: SoilWater

   324) ! !INTERFACE:
   325)   subroutine SoilWater(lbc, ubc, num_hydrologyc, filter_hydrologyc, &
   326)                        num_urbanc, filter_urbanc, &
   327)                        vol_liq, dwat, hk, dhkdw)
   328) !
   329) ! !DESCRIPTION:
   330) ! Soil hydrology
   331) ! Soil moisture is predicted from a 10-layer model (as with soil
   332) ! temperature), in which the vertical soil moisture transport is governed
   333) ! by infiltration, runoff, gradient diffusion, gravity, and root
   334) ! extraction through canopy transpiration.  The net water applied to the
   335) ! surface layer is the snowmelt plus precipitation plus the throughfall
   336) ! of canopy dew minus surface runoff and evaporation.
   337) ! CLM3.5 uses a zero-flow bottom boundary condition.
   338) !
   339) ! The vertical water flow in an unsaturated porous media is described by
   340) ! Darcy's law, and the hydraulic conductivity and the soil negative
   341) ! potential vary with soil water content and soil texture based on the work
   342) ! of Clapp and Hornberger (1978) and Cosby et al. (1984). The equation is
   343) ! integrated over the layer thickness, in which the time rate of change in
   344) ! water mass must equal the net flow across the bounding interface, plus the
   345) ! rate of internal source or sink. The terms of water flow across the layer
   346) ! interfaces are linearly expanded by using first-order Taylor expansion.
   347) ! The equations result in a tridiagonal system equation.
   348) !
   349) ! Note: length units here are all millimeter
   350) ! (in temperature subroutine uses same soil layer
   351) ! structure required but lengths are m)
   352) !
   353) ! Richards equation:
   354) !
   355) ! d wat      d     d wat d psi
   356) ! ----- = - -- [ k(----- ----- - 1) ] + S
   357) !   dt      dz       dz  d wat
   358) !
   359) ! where: wat = volume of water per volume of soil (mm**3/mm**3)
   360) ! psi = soil matrix potential (mm)
   361) ! dt  = time step (s)
   362) ! z   = depth (mm)
   363) ! dz  = thickness (mm)
   364) ! qin = inflow at top (mm h2o /s)
   365) ! qout= outflow at bottom (mm h2o /s)
   366) ! s   = source/sink flux (mm h2o /s)
   367) ! k   = hydraulic conductivity (mm h2o /s)
   368) !
   369) !                       d qin                  d qin
   370) ! qin[n+1] = qin[n] +  --------  d wat(j-1) + --------- d wat(j)
   371) !                       d wat(j-1)             d wat(j)
   372) !                ==================|=================
   373) !                                  < qin
   374) !
   375) !                 d wat(j)/dt * dz = qin[n+1] - qout[n+1] + S(j)
   376) !
   377) !                                  > qout
   378) !                ==================|=================
   379) !                        d qout               d qout
   380) ! qout[n+1] = qout[n] + --------- d wat(j) + --------- d wat(j+1)
   381) !                        d wat(j)             d wat(j+1)
   382) !
   383) !
   384) ! Solution: linearize k and psi about d wat and use tridiagonal
   385) ! system of equations to solve for d wat,
   386) ! where for layer j
   387) !
   388) !
   389) ! r_j = a_j [d wat_j-1] + b_j [d wat_j] + c_j [d wat_j+1]
   390) !
   391) ! !USES:
   392)     use shr_kind_mod, only: r8 => shr_kind_r8
   393)     use clmtype
   394)     use clm_varcon    , only : wimp, icol_roof, icol_road_imperv
   395)     use clm_varpar    , only : nlevsoi, max_pft_per_col
   396)     use clm_varctl    , only : iulog
   397)     use shr_const_mod , only : SHR_CONST_TKFRZ, SHR_CONST_LATICE, SHR_CONST_G
   398)     use TridiagonalMod, only : Tridiagonal
   399)     use clm_time_manager  , only : get_step_size
   400) !
   401) ! !ARGUMENTS:
   402)     implicit none
   403)     integer , intent(in)  :: lbc, ubc                     ! column bounds
   404)     integer , intent(in)  :: num_hydrologyc               ! number of column soil points in column filter
   405)     integer , intent(in)  :: filter_hydrologyc(ubc-lbc+1) ! column filter for soil points
   406)     integer , intent(in)  :: num_urbanc                   ! number of column urban points in column filter
   407)     integer , intent(in)  :: filter_urbanc(ubc-lbc+1)     ! column filter for urban points
   408)     real(r8), intent(in)  :: vol_liq(lbc:ubc,1:nlevsoi)   ! soil water per unit volume [mm/mm]
   409)     real(r8), intent(out) :: dwat(lbc:ubc,1:nlevsoi)      ! change of soil water [m3/m3]
   410)     real(r8), intent(out) :: hk(lbc:ubc,1:nlevsoi)        ! hydraulic conductivity [mm h2o/s]
   411)     real(r8), intent(out) :: dhkdw(lbc:ubc,1:nlevsoi)     ! d(hk)/d(vol_liq)
   412) !

   430)     real(r8), pointer :: pwtgcell(:)          ! weight relative to gridcell for each pft

   436)     real(r8), pointer :: qflx_tran_veg_col(:) ! vegetation transpiration (mm H2O/s) (+ = to atm)
   437)     real(r8), pointer :: eff_porosity(:,:)    ! effective porosity = porosity - vol_ice
   438)     real(r8), pointer :: watsat(:,:)          ! volumetric soil water at saturation (porosity)
   439)     real(r8), pointer :: hksat(:,:)           ! hydraulic conductivity at saturation (mm H2O /s)
   440)     real(r8), pointer :: bsw(:,:)             ! Clapp and Hornberger "b"
   441)     real(r8), pointer :: sucsat(:,:)          ! minimum soil suction (mm)
   442)     real(r8), pointer :: t_soisno(:,:)        ! soil temperature (Kelvin)
   443)     real(r8), pointer :: rootr_pft(:,:)       ! effective fraction of roots in each soil layer
   444)     integer , pointer :: pfti(:)              ! beginning pft index for each column
   445)     real(r8), pointer :: fracice(:,:)         ! fractional impermeability (-)
   446)     real(r8), pointer :: h2osoi_vol(:,:)      ! volumetric soil water (0<=h2osoi_vol<=watsat) [m3/m3]
   447)     real(r8), pointer :: qcharge(:)           ! aquifer recharge rate (mm/s)
   448)     real(r8), pointer :: hkdepth(:)           ! decay factor (m)
   449)     real(r8), pointer :: zwt(:)               ! water table depth (m)
   450)     real(r8), pointer :: zi(:,:)              ! interface level below a "z" level (m)
   451) !
   452) ! local pointers to original implicit inout arguments
   453) !
   454)     real(r8), pointer :: h2osoi_liq(:,:)      ! liquid water (kg/m2)
   455) !
   456) ! local pointer s to original implicit out arguments
   457) !
   458)     real(r8), pointer :: rootr_col(:,:)       ! effective fraction of roots in each soil layer
   459)     real(r8), pointer :: smp_l(:,:)             ! soil matrix potential [mm]
   460)     real(r8), pointer :: hk_l(:,:)              ! hydraulic conductivity (mm/s)
   461) !
   462) !EOP
   463) !
   464) ! !OTHER LOCAL VARIABLES:
   465) !
   466)     integer  :: p,c,fc,j                  ! do loop indices
   467)     integer  :: jtop(lbc:ubc)             ! top level at each column
   468)     real(r8) :: dtime                     ! land model time step (sec)
   469)     real(r8) :: amx(lbc:ubc,1:nlevsoi+1)  ! "a" left off diagonal of tridiagonal matrix
   470)     real(r8) :: bmx(lbc:ubc,1:nlevsoi+1)  ! "b" diagonal column for tridiagonal matrix
   471)     real(r8) :: cmx(lbc:ubc,1:nlevsoi+1)  ! "c" right off diagonal tridiagonal matrix
   472)     real(r8) :: rmx(lbc:ubc,1:nlevsoi+1)  ! "r" forcing term of tridiagonal matrix
   473)     real(r8) :: zmm(lbc:ubc,1:nlevsoi+1)  ! layer depth [mm]
   474)     real(r8) :: dzmm(lbc:ubc,1:nlevsoi+1) ! layer thickness [mm]
   475)     real(r8) :: den                       ! used in calculating qin, qout
   476)     real(r8) :: dqidw0(lbc:ubc,1:nlevsoi+1) ! d(qin)/d(vol_liq(i-1))
   477)     real(r8) :: dqidw1(lbc:ubc,1:nlevsoi+1) ! d(qin)/d(vol_liq(i))
   478)     real(r8) :: dqodw1(lbc:ubc,1:nlevsoi+1) ! d(qout)/d(vol_liq(i))
   479)     real(r8) :: dqodw2(lbc:ubc,1:nlevsoi+1) ! d(qout)/d(vol_liq(i+1))
   480)     real(r8) :: dsmpdw(lbc:ubc,1:nlevsoi+1) ! d(smp)/d(vol_liq)
   481)     real(r8) :: num                         ! used in calculating qin, qout
   482)     real(r8) :: qin(lbc:ubc,1:nlevsoi+1)    ! flux of water into soil layer [mm h2o/s]
   483)     real(r8) :: qout(lbc:ubc,1:nlevsoi+1)   ! flux of water out of soil layer [mm h2o/s]
   484)     real(r8) :: s_node                    ! soil wetness
   485)     real(r8) :: s1                        ! "s" at interface of layer
   486)     real(r8) :: s2                        ! k*s**(2b+2)
   487)     real(r8) :: smp(lbc:ubc,1:nlevsoi)    ! soil matrix potential [mm]
   488)     real(r8) :: sdamp                     ! extrapolates soiwat dependence of evaporation
   489)     integer  :: pi                        ! pft index
   490)     real(r8) :: temp(lbc:ubc)             ! accumulator for rootr weighting
   491)     integer  :: jwt(lbc:ubc)              ! index of the soil layer right above the water table (-)
   492)     real(r8) :: smp1,dsmpdw1,wh,wh_zwt,ka
   493)     real(r8) :: dwat2(lbc:ubc,1:nlevsoi+1)
   494)     real(r8) :: dzq                         ! used in calculating qin, qout (difference in equilbirium matric potential)
   495)     real(r8) :: zimm(lbc:ubc,0:nlevsoi)     ! layer interface depth [mm]
   496)     real(r8) :: zq(lbc:ubc,1:nlevsoi+1)     ! equilibrium matric potential for each layer [mm]
   497)     real(r8) :: vol_eq(lbc:ubc,1:nlevsoi+1) ! equilibrium volumetric water content
   498)     real(r8) :: tempi                       ! temp variable for calculating vol_eq
   499)     real(r8) :: temp0                       ! temp variable for calculating vol_eq
   500)     real(r8) :: voleq1                      ! temp variable for calculating vol_eq
   501)     real(r8) :: zwtmm(lbc:ubc)              ! water table depth [mm]
   502) !-----------------------------------------------------------------------
   503) 
   504)     ! Assign local pointers to derived type members (column-level)
   505) 
   506)     qcharge           => cws%qcharge
   507)     hkdepth           => cps%hkdepth
   508)     zi                => cps%zi
   509)     zwt               => cws%zwt
   510)     ctype             => col%itype
   511)     npfts             => col%npfts
   512)     z                 => cps%z
   513)     dz                => cps%dz
   514)     smpmin            => cps%smpmin
   515)     watsat            => cps%watsat
   516)     hksat             => cps%hksat
   517)     bsw               => cps%bsw
   518)     sucsat            => cps%sucsat
   519)     eff_porosity      => cps%eff_porosity
   520)     rootr_col         => cps%rootr_column
   521)     t_soisno          => ces%t_soisno
   522)     h2osoi_liq        => cws%h2osoi_liq
   523)     h2osoi_vol        => cws%h2osoi_vol
   524)     qflx_infl         => cwf%qflx_infl
   525)     fracice           => cps%fracice
   526)     qflx_tran_veg_col => pwf_a%qflx_tran_veg
   527)     pfti              => col%pfti
   528)     smp_l             => cws%smp_l
   529)     hk_l              => cws%hk_l
   530) 
   531)     ! Assign local pointers to derived type members (pft-level)
   532) 
   533)     qflx_tran_veg_pft => pwf%qflx_tran_veg
   534)     rootr_pft         => pps%rootr
   535)     pwtcol            => pft%wtcol
   536)     pwtgcell          => pft%wtgcell
   537) 
   538)     ! Get time step
   539) 
   540)     dtime = get_step_size()
   541) 
   542)     ! Because the depths in this routine are in mm, use local
   543)     ! variable arrays instead of pointers
   544) 
   545)     do j = 1, nlevsoi
   546)        do fc = 1, num_hydrologyc
   547)           c = filter_hydrologyc(fc)
   548)           zmm(c,j) = z(c,j)*1.e3_r8
   549)           dzmm(c,j) = dz(c,j)*1.e3_r8
   550)           zimm(c,j) = zi(c,j)*1.e3_r8
   551)        end do
   552)     end do
   553) 
   554)     do fc = 1, num_hydrologyc 
   555)        c = filter_hydrologyc(fc)
   556)        zimm(c,0) = 0.0_r8
   557)        zwtmm(c)  = zwt(c)*1.e3_r8
   558)     end do
   559) 
   560)     ! First step is to calculate the column-level effective rooting
   561)     ! fraction in each soil layer. This is done outside the usual
   562)     ! PFT-to-column averaging routines because it is not a simple
   563)     ! weighted average of the PFT level rootr arrays. Instead, the
   564)     ! weighting depends on both the per-unit-area transpiration
   565)     ! of the PFT and the PFTs area relative to all PFTs.
   566) 
   567)     temp(:) = 0._r8
   568) 
   569)     do j = 1, nlevsoi
   570)        do fc = 1, num_hydrologyc
   571)           c = filter_hydrologyc(fc)
   572)           rootr_col(c,j) = 0._r8
   573)        end do
   574)     end do
   575) 
   576)     do pi = 1,max_pft_per_col
   577)        do j = 1,nlevsoi
   578)           do fc = 1, num_hydrologyc
   579)              c = filter_hydrologyc(fc)
   580)              if (pi <= npfts(c)) then
   581)                 p = pfti(c) + pi - 1
   582)                 if (pwtgcell(p)>0._r8) then
   583)                    rootr_col(c,j) = rootr_col(c,j) + rootr_pft(p,j) * qflx_tran_veg_pft(p) * pwtcol(p)
   584)                 end if
   585)              end if
   586)           end do
   587)        end do
   588)        do fc = 1, num_hydrologyc
   589)           c = filter_hydrologyc(fc)
   590)           if (pi <= npfts(c)) then
   591)              p = pfti(c) + pi - 1
   592)              if (pwtgcell(p)>0._r8) then
   593)                 temp(c) = temp(c) + qflx_tran_veg_pft(p) * pwtcol(p)
   594)              end if
   595)           end if
   596)        end do
   597)     end do
   598) 
   599)     do j = 1, nlevsoi
   600)        do fc = 1, num_hydrologyc
   601)           c = filter_hydrologyc(fc)
   602)           if (temp(c) /= 0._r8) then
   603)              rootr_col(c,j) = rootr_col(c,j)/temp(c)
   604)           end if
   605)        end do
   606)     end do
   607) 
   608)     !compute jwt index
   609)     ! The layer index of the first unsaturated layer, i.e., the layer right above
   610)     ! the water table
   611) 
   612)     do fc = 1, num_hydrologyc
   613)        c = filter_hydrologyc(fc)
   614)        jwt(c) = nlevsoi
   615)        do j = 2,nlevsoi
   616)           if(zwt(c) <= zi(c,j)) then
   617)              jwt(c) = j-1
   618)              exit
   619)           end if
   620)        enddo
   621)     end do
   622) 
   623)     ! calculate the equilibrium water content based on the water table depth
   624)             
   625)     do j=1,nlevsoi 
   626)        do fc=1, num_hydrologyc
   627)           c = filter_hydrologyc(fc)
   628)           if ((zwtmm(c) .lt. zimm(c,j-1))) then   !fully saturated when wtd is less than the layer top
   629)              vol_eq(c,j) = watsat(c,j)
   630)             
   631)           ! use the weighted average from the saturated part (depth > wtd) and the equilibrium solution for the
   632)           ! rest of the layer
   633) 
   634)           else if ((zwtmm(c) .lt. zimm(c,j)) .and. (zwtmm(c) .gt. zimm(c,j-1))) then
   635)              tempi = 1.0_r8
   636)              temp0 = (((sucsat(c,j)+zwtmm(c)-zimm(c,j-1))/sucsat(c,j)))**(1._r8-1._r8/bsw(c,j))
   637)              voleq1 = -sucsat(c,j)*watsat(c,j)/(1._r8-1._r8/bsw(c,j))/(zwtmm(c)-zimm(c,j-1))*(tempi-temp0)
   638)              vol_eq(c,j) = (voleq1*(zwtmm(c)-zimm(c,j-1)) + watsat(c,j)*(zimm(c,j)-zwtmm(c)))/(zimm(c,j)-zimm(c,j-1))
   639)              vol_eq(c,j) = min(watsat(c,j),vol_eq(c,j))
   640)              vol_eq(c,j) = max(vol_eq(c,j),0.0_r8)
   641)           else
   642)              tempi = (((sucsat(c,j)+zwtmm(c)-zimm(c,j))/sucsat(c,j)))**(1._r8-1._r8/bsw(c,j))
   643)              temp0 = (((sucsat(c,j)+zwtmm(c)-zimm(c,j-1))/sucsat(c,j)))**(1._r8-1._r8/bsw(c,j))
   644)              vol_eq(c,j) = -sucsat(c,j)*watsat(c,j)/(1._r8-1._r8/bsw(c,j))/(zimm(c,j)-zimm(c,j-1))*(tempi-temp0)
   645)              vol_eq(c,j) = max(vol_eq(c,j),0.0_r8)
   646)              vol_eq(c,j) = min(watsat(c,j),vol_eq(c,j))
   647)           endif
   648)           zq(c,j) = -sucsat(c,j)*(max(vol_eq(c,j)/watsat(c,j),0.01_r8))**(-bsw(c,j))
   649)           zq(c,j) = max(smpmin(c), zq(c,j))
   650)        end do
   651)     end do
   652) 
   653)     ! If water table is below soil column calculate zq for the 11th layer
   654)     j = nlevsoi
   655)     do fc=1, num_hydrologyc
   656)        c = filter_hydrologyc(fc)
   657)        if(jwt(c) == nlevsoi) then 
   658)           tempi = 1._r8
   659)           temp0 = (((sucsat(c,j)+zwtmm(c)-zimm(c,j))/sucsat(c,j)))**(1._r8-1._r8/bsw(c,j))
   660)           vol_eq(c,j+1) = -sucsat(c,j)*watsat(c,j)/(1._r8-1._r8/bsw(c,j))/(zwtmm(c)-zimm(c,j))*(tempi-temp0)
   661)           vol_eq(c,j+1) = max(vol_eq(c,j+1),0.0_r8)
   662)           vol_eq(c,j+1) = min(watsat(c,j),vol_eq(c,j+1))
   663)           zq(c,j+1) = -sucsat(c,j)*(max(vol_eq(c,j+1)/watsat(c,j),0.01_r8))**(-bsw(c,j))
   664)           zq(c,j+1) = max(smpmin(c), zq(c,j+1))
   665)        end if
   666)     end do
   667) 
   668)     ! Hydraulic conductivity and soil matric potential and their derivatives
   669) 
   670)     sdamp = 0._r8
   671)     do j = 1, nlevsoi
   672)        do fc = 1, num_hydrologyc
   673)           c = filter_hydrologyc(fc)
   674) 
   675)           s1 = 0.5_r8*(h2osoi_vol(c,j) + h2osoi_vol(c,min(nlevsoi, j+1))) / &
   676)                (0.5_r8*(watsat(c,j)+watsat(c,min(nlevsoi, j+1))))
   677)           s1 = min(1._r8, s1)
   678)           s2 = hksat(c,j)*s1**(2._r8*bsw(c,j)+2._r8)
   679) 
   680)           hk(c,j) = (1._r8-0.5_r8*(fracice(c,j)+fracice(c,min(nlevsoi, j+1))))*s1*s2
   681) 
   682)           dhkdw(c,j) = (1._r8-0.5_r8*(fracice(c,j)+fracice(c,min(nlevsoi, j+1))))* &
   683)                        (2._r8*bsw(c,j)+3._r8)*s2*0.5_r8/watsat(c,j)
   684) 
   685)           s_node = max(h2osoi_vol(c,j)/watsat(c,j), 0.01_r8)
   686)           s_node = min(1.0_r8, s_node)
   687) 
   688)           smp(c,j) = -sucsat(c,j)*s_node**(-bsw(c,j))
   689)           smp(c,j) = max(smpmin(c), smp(c,j))
   690) 
   691)           dsmpdw(c,j) = -bsw(c,j)*smp(c,j)/(s_node*watsat(c,j))
   692) 
   693)           smp_l(c,j) = smp(c,j)
   694)           hk_l(c,j) = hk(c,j)
   695) 
   696)        end do
   697)     end do
   698) 
   699)     ! aquifer (11th) layer
   700)     do fc = 1, num_hydrologyc
   701)        c = filter_hydrologyc(fc)
   702)        zmm(c,nlevsoi+1) = 0.5*(1.e3_r8*zwt(c) + zmm(c,nlevsoi))
   703)        if(jwt(c) < nlevsoi) then
   704)          dzmm(c,nlevsoi+1) = dzmm(c,nlevsoi)
   705)        else
   706)          dzmm(c,nlevsoi+1) = (1.e3_r8*zwt(c) - zmm(c,nlevsoi))
   707)        end if
   708)     end do
   709) 
   710)     ! Set up r, a, b, and c vectors for tridiagonal solution
   711) 
   712)     ! Node j=1 (top)
   713) 
   714)     j = 1
   715)     do fc = 1, num_hydrologyc
   716)        c = filter_hydrologyc(fc)
   717)        qin(c,j)    = qflx_infl(c)
   718)        den    = (zmm(c,j+1)-zmm(c,j))
   719)        dzq    = (zq(c,j+1)-zq(c,j))
   720)        num    = (smp(c,j+1)-smp(c,j)) - dzq
   721)        qout(c,j)   = -hk(c,j)*num/den
   722)        dqodw1(c,j) = -(-hk(c,j)*dsmpdw(c,j)   + num*dhkdw(c,j))/den
   723)        dqodw2(c,j) = -( hk(c,j)*dsmpdw(c,j+1) + num*dhkdw(c,j))/den
   724)        rmx(c,j) =  qin(c,j) - qout(c,j) - qflx_tran_veg_col(c) * rootr_col(c,j)
   725)        amx(c,j) =  0._r8
   726)        bmx(c,j) =  dzmm(c,j)*(sdamp+1._r8/dtime) + dqodw1(c,j)
   727)        cmx(c,j) =  dqodw2(c,j)
   728)     end do
   729) 
   730)     ! Nodes j=2 to j=nlevsoi-1
   731) 
   732)     do j = 2, nlevsoi - 1
   733)        do fc = 1, num_hydrologyc
   734)           c = filter_hydrologyc(fc)
   735)           den    = (zmm(c,j) - zmm(c,j-1))
   736)           dzq    = (zq(c,j)-zq(c,j-1))
   737)           num    = (smp(c,j)-smp(c,j-1)) - dzq
   738)           qin(c,j)    = -hk(c,j-1)*num/den
   739)           dqidw0(c,j) = -(-hk(c,j-1)*dsmpdw(c,j-1) + num*dhkdw(c,j-1))/den
   740)           dqidw1(c,j) = -( hk(c,j-1)*dsmpdw(c,j)   + num*dhkdw(c,j-1))/den
   741)           den    = (zmm(c,j+1)-zmm(c,j))
   742)           dzq    = (zq(c,j+1)-zq(c,j))
   743)           num    = (smp(c,j+1)-smp(c,j)) - dzq
   744)           qout(c,j)   = -hk(c,j)*num/den
   745)           dqodw1(c,j) = -(-hk(c,j)*dsmpdw(c,j)   + num*dhkdw(c,j))/den
   746)           dqodw2(c,j) = -( hk(c,j)*dsmpdw(c,j+1) + num*dhkdw(c,j))/den
   747)           rmx(c,j)    =  qin(c,j) - qout(c,j) - qflx_tran_veg_col(c)*rootr_col(c,j)
   748)           amx(c,j)    = -dqidw0(c,j)
   749)           bmx(c,j)    =  dzmm(c,j)/dtime - dqidw1(c,j) + dqodw1(c,j)
   750)           cmx(c,j)    =  dqodw2(c,j)
   751)        end do
   752)     end do
   753) 
   754)     ! Node j=nlevsoi (bottom)
   755) 
   756)     j = nlevsoi
   757)     do fc = 1, num_hydrologyc
   758)        c = filter_hydrologyc(fc)
   759)        if(j > jwt(c)) then !water table is in soil column
   760)          den    = (zmm(c,j) - zmm(c,j-1))
   761)          dzq    = (zq(c,j)-zq(c,j-1))
   762)          num    = (smp(c,j)-smp(c,j-1)) - dzq
   763)          qin(c,j)    = -hk(c,j-1)*num/den
   764)          dqidw0(c,j) = -(-hk(c,j-1)*dsmpdw(c,j-1) + num*dhkdw(c,j-1))/den
   765)          dqidw1(c,j) = -( hk(c,j-1)*dsmpdw(c,j)   + num*dhkdw(c,j-1))/den
   766)          qout(c,j)   =  0._r8
   767)          dqodw1(c,j) =  0._r8
   768)          rmx(c,j)    =  qin(c,j) - qout(c,j) - qflx_tran_veg_col(c)*rootr_col(c,j)
   769)          amx(c,j)    = -dqidw0(c,j)
   770)          bmx(c,j)    =  dzmm(c,j)/dtime - dqidw1(c,j) + dqodw1(c,j)
   771)          cmx(c,j)    =  0._r8
   772) 
   773)          !scs: next set up aquifer layer; hydrologically inactive
   774)          rmx(c,j+1) = 0._r8
   775)          amx(c,j+1) = 0._r8
   776)          bmx(c,j+1) = dzmm(c,j+1)/dtime
   777)          cmx(c,j+1) = 0._r8
   778)        else ! water table is below soil column
   779) 
   780)          !scs: compute aquifer soil moisture as average of layer 10 and saturation
   781)          s_node = max(0.5*(1.0_r8+h2osoi_vol(c,j)/watsat(c,j)), 0.01_r8)
   782)          s_node = min(1.0_r8, s_node)
   783) 
   784)          !scs: compute smp for aquifer layer
   785)          smp1 = -sucsat(c,j)*s_node**(-bsw(c,j))
   786)          smp1 = max(smpmin(c), smp1)
   787) 
   788)          !scs: compute dsmpdw for aquifer layer
   789)          dsmpdw1 = -bsw(c,j)*smp1/(s_node*watsat(c,j))
   790) 
   791)          !scs: first set up bottom layer of soil column
   792)          den    = (zmm(c,j) - zmm(c,j-1))
   793)          dzq    = (zq(c,j)-zq(c,j-1))
   794)          num    = (smp(c,j)-smp(c,j-1)) - dzq
   795)          qin(c,j)    = -hk(c,j-1)*num/den
   796)          dqidw0(c,j) = -(-hk(c,j-1)*dsmpdw(c,j-1) + num*dhkdw(c,j-1))/den
   797)          dqidw1(c,j) = -( hk(c,j-1)*dsmpdw(c,j)   + num*dhkdw(c,j-1))/den
   798)          den    = (zmm(c,j+1)-zmm(c,j))
   799)          dzq    = (zq(c,j+1)-zq(c,j))
   800)          num    = (smp1-smp(c,j)) - dzq
   801)          qout(c,j)   = -hk(c,j)*num/den
   802)          dqodw1(c,j) = -(-hk(c,j)*dsmpdw(c,j)   + num*dhkdw(c,j))/den
   803)          dqodw2(c,j) = -( hk(c,j)*dsmpdw1 + num*dhkdw(c,j))/den
   804) 
   805)          rmx(c,j) =  qin(c,j) - qout(c,j) - qflx_tran_veg_col(c)*rootr_col(c,j)
   806)          amx(c,j) = -dqidw0(c,j)
   807)          bmx(c,j) =  dzmm(c,j)/dtime - dqidw1(c,j) + dqodw1(c,j)
   808)          cmx(c,j) =  dqodw2(c,j)
   809) 
   810)          !scs: next set up aquifer layer; den/num unchanged, qin=qout
   811)          qin(c,j+1)    = qout(c,j)
   812)          dqidw0(c,j+1) = -(-hk(c,j)*dsmpdw(c,j) + num*dhkdw(c,j))/den
   813)          dqidw1(c,j+1) = -( hk(c,j)*dsmpdw1   + num*dhkdw(c,j))/den
   814)          qout(c,j+1)   =  0._r8  ! zero-flow bottom boundary condition
   815)          dqodw1(c,j+1) =  0._r8  ! zero-flow bottom boundary condition
   816)          rmx(c,j+1) =  qin(c,j+1) - qout(c,j+1)
   817)          amx(c,j+1) = -dqidw0(c,j+1)
   818)          bmx(c,j+1) =  dzmm(c,j+1)/dtime - dqidw1(c,j+1) + dqodw1(c,j+1)
   819)          cmx(c,j+1) =  0._r8
   820)        endif
   821)     end do
   822) 
   823)     ! Solve for dwat
   824) 
   825)     jtop(:) = 1
   826)     call Tridiagonal(lbc, ubc, 1, nlevsoi+1, jtop, num_hydrologyc, filter_hydrologyc, &
   827)                      amx, bmx, cmx, rmx, dwat2 )
   828)     !scs: set dwat
   829)     do fc = 1,num_hydrologyc
   830)        c = filter_hydrologyc(fc)
   831)        do j = 1, nlevsoi
   832)           dwat(c,j)=dwat2(c,j)
   833)        end do
   834)     end do
   835) 
   836)     ! Renew the mass of liquid water
   837)     !scs: also compute qcharge from dwat in aquifer layer
   838)     !scs: update in drainage for case jwt < nlevsoi
   839) 
   840)     do fc = 1,num_hydrologyc
   841)        c = filter_hydrologyc(fc)
   842)        do j = 1, nlevsoi
   843)           h2osoi_liq(c,j) = h2osoi_liq(c,j) + dwat2(c,j)*dzmm(c,j)
   844)        end do
   845) 
   846)        !scs: calculate qcharge for case jwt < nlevsoi
   847)        if(jwt(c) < nlevsoi) then
   848)           wh_zwt = 0._r8   !since wh_zwt = -sucsat - zq_zwt, where zq_zwt = -sucsat
   849) 
   850)           s_node = max(h2osoi_vol(c,jwt(c))/watsat(c,jwt(c)), 0.01_r8)
   851)           s_node = min(1.0_r8, s_node)
   852) 
   853)           !scs: use average moisture between water table and layer jwt
   854)           s1 = 0.5_r8*(1.0+s_node)
   855)           s1 = min(1._r8, s1)
   856) 
   857)           !scs: this is the expression for unsaturated hk
   858)           ka = hksat(c,jwt(c))*s1**(2._r8*bsw(c,jwt(c))+3._r8)
   859) 
   860)           ! Recharge rate qcharge to groundwater (positive to aquifer)
   861)           smp1 = -sucsat(c,jwt(c))*s_node**(-bsw(c,jwt(c)))
   862)           smp1 = max(smpmin(c), smp(c,jwt(c)))
   863)           wh      = smp1 - zq(c,jwt(c))
   864)           qcharge(c) = -ka * (wh_zwt-wh)  /((zwt(c)-z(c,jwt(c)))*1000._r8)
   865) 
   866)           ! To limit qcharge  (for the first several timesteps)
   867)           qcharge(c) = max(-10.0_r8/dtime,qcharge(c))
   868)           qcharge(c) = min( 10.0_r8/dtime,qcharge(c))
   869)        else
   870)           !scs: if water table is below soil column, compute qcharge from dwat2(11)
   871)           qcharge(c) = dwat2(c,nlevsoi+1)*dzmm(c,nlevsoi+1)/dtime
   872)        endif
   873)     end do
   874) 
   875)   end subroutine SoilWater
   876)

   883)   subroutine Drainage(lbc, ubc, num_hydrologyc, filter_hydrologyc, &
   884)                       num_urbanc, filter_urbanc, vol_liq, hk, &
   885)                       icefrac)
   886) !
   887) ! !DESCRIPTION:
   888) ! Calculate subsurface drainage
   889) !
   890) ! !USES:
   891)     use shr_kind_mod, only : r8 => shr_kind_r8
   892)     use clmtype
   893)     use clm_time_manager, only : get_step_size

   895)     use clm_varpar  , only : nlevsoi
   896) !

   900)     integer , intent(in) :: num_hydrologyc               ! number of column soil points in column filter
   901)     integer , intent(in) :: num_urbanc                   ! number of column urban points in column filter
   902)     integer , intent(in) :: filter_urbanc(ubc-lbc+1)     ! column filter for urban points
   903)     integer , intent(in) :: filter_hydrologyc(ubc-lbc+1) ! column filter for soil points
   904)     real(r8), intent(in) :: vol_liq(lbc:ubc,1:nlevsoi)   ! partial volume of liquid water in layer
   905)     real(r8), intent(in) :: hk(lbc:ubc,1:nlevsoi)        ! hydraulic conductivity (mm h2o/s)
   906)     real(r8), intent(in) :: icefrac(lbc:ubc,1:nlevsoi)   ! fraction of ice in layer
   907) !
   908) ! !CALLED FROM:
   909) !
   910) ! !REVISION HISTORY:
   911) ! 15 September 1999: Yongjiu Dai; Initial code
   912) ! 12 November 1999:  Z.-L. Yang and G.-Y. Niu
   913) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
   914) ! 4/26/05, Peter Thornton and David Lawrence: Turned off drainage from
   915) ! middle soil layers for both wet and dry fractions.
   916) ! 04/25/07  Keith Oleson: Completely new routine for CLM3.5 hydrology
   917) ! 27 February 2008: Keith Oleson; Saturation excess modification
   918) !
   919) ! !LOCAL VARIABLES:
   920) !
   921) ! local pointers to original implicit in arguments
   922) !
   923)     integer , pointer :: ctype(:)          !column type index
   924)     integer , pointer :: snl(:)            !number of snow layers
   925)     real(r8), pointer :: qflx_snwcp_liq(:) !excess rainfall due to snow capping (mm H2O /s) [+]
   926)     real(r8), pointer :: qflx_dew_grnd(:)  !ground surface dew formation (mm H2O /s) [+]
   927)     real(r8), pointer :: qflx_dew_snow(:)  !surface dew added to snow pack (mm H2O /s) [+]
   928)     real(r8), pointer :: qflx_sub_snow(:)  !sublimation rate from snow pack (mm H2O /s) [+]
   929)     real(r8), pointer :: dz(:,:)           !layer depth (m)
   930)     real(r8), pointer :: bsw(:,:)          !Clapp and Hornberger "b"
   931)     real(r8), pointer :: eff_porosity(:,:) !effective porosity = porosity - vol_ice
   932)     real(r8), pointer :: t_soisno(:,:)     !soil temperature (Kelvin)
   933)     real(r8), pointer :: hksat(:,:)        !hydraulic conductivity at saturation (mm H2O /s)
   934)     real(r8), pointer :: sucsat(:,:)       !minimum soil suction (mm)
   935)     real(r8), pointer :: z(:,:)            !layer depth (m)
   936)     real(r8), pointer :: zi(:,:)           !interface level below a "z" level (m)
   937)     real(r8), pointer :: watsat(:,:)       !volumetric soil water at saturation (porosity)
   938)     real(r8), pointer :: hkdepth(:)        !decay factor (m)
   939)     real(r8), pointer :: zwt(:)            !water table depth (m)
   940)     real(r8), pointer :: wa(:)             !water in the unconfined aquifer (mm)
   941)     real(r8), pointer :: wt(:)             !total water storage (unsaturated soil water + groundwater) (mm)
   942)     real(r8), pointer :: qcharge(:)        !aquifer recharge rate (mm/s)
   943) !
   944) ! local pointers to original implicit inout arguments
   945) !
   946)     real(r8), pointer :: h2osoi_ice(:,:)   !ice lens (kg/m2)
   947)     real(r8), pointer :: h2osoi_liq(:,:)   !liquid water (kg/m2)
   948) !
   949) ! local pointers to original implicit out arguments
   950) !
   951)     real(r8), pointer :: qflx_drain(:)     !sub-surface runoff (mm H2O /s)
   952)     real(r8), pointer :: qflx_irrig(:)     !irrigation flux (mm H2O /s)
   953)     real(r8), pointer :: qflx_qrgwl(:)     !qflx_surf at glaciers, wetlands, lakes (mm H2O /s)
   954)     real(r8), pointer :: eflx_impsoil(:)   !implicit evaporation for soil temperature equation
   955)     real(r8), pointer :: qflx_rsub_sat(:)  !soil saturation excess [mm h2o/s]
   956) !
   957) !EOP
   958) !
   959) ! !OTHER LOCAL VARIABLES:
   960) !
   961)     integer  :: c,j,fc,i                 !indices
   962)     real(r8) :: dtime                    !land model time step (sec)
   963)     real(r8) :: xs(lbc:ubc)              !water needed to bring soil moisture to watmin (mm)
   964)     real(r8) :: dzmm(lbc:ubc,1:nlevsoi)  !layer thickness (mm)
   965)     integer  :: jwt(lbc:ubc)             !index of the soil layer right above the water table (-)
   966)     real(r8) :: rsub_bot(lbc:ubc)        !subsurface runoff - bottom drainage (mm/s)
   967)     real(r8) :: rsub_top(lbc:ubc)        !subsurface runoff - topographic control (mm/s)
   968)     real(r8) :: fff(lbc:ubc)             !decay factor (m-1)
   969)     real(r8) :: xsi(lbc:ubc)             !excess soil water above saturation at layer i (mm)
   970)     real(r8) :: xsia(lbc:ubc)            !available pore space at layer i (mm)
   971)     real(r8) :: xs1(lbc:ubc)             !excess soil water above saturation at layer 1 (mm)
   972)     real(r8) :: smpfz(1:nlevsoi)         !matric potential of layer right above water table (mm)
   973)     real(r8) :: wtsub                    !summation of hk*dzmm for layers below water table (mm**2/s)
   974)     real(r8) :: rous                     !aquifer yield (-)
   975)     real(r8) :: wh                       !smpfz(jwt)-z(jwt) (mm)
   976)     real(r8) :: wh_zwt                   !water head at the water table depth (mm)
   977)     real(r8) :: ws                       !summation of pore space of layers below water table (mm)
   978)     real(r8) :: s_node                   !soil wetness (-)
   979)     real(r8) :: dzsum                    !summation of dzmm of layers below water table (mm)
   980)     real(r8) :: icefracsum               !summation of icefrac*dzmm of layers below water table (-)
   981)     real(r8) :: fracice_rsub(lbc:ubc)    !fractional impermeability of soil layers (-)
   982)     real(r8) :: ka                       !hydraulic conductivity of the aquifer (mm/s)
   983)     real(r8) :: dza                      !fff*(zwt-z(jwt)) (-)
   984)     real(r8) :: available_h2osoi_liq     !available soil liquid water in a layer
   985) !-----------------------------------------------------------------------
   986) 
   987)     ! Assign local pointers to derived subtypes components (column-level)
   988) 
   989)     ctype          => col%itype
   990) !   cgridcell      => col%gridcell
   991) 
   992)     snl           => cps%snl
   993)     dz            => cps%dz
   994)     bsw           => cps%bsw
   995)     t_soisno      => ces%t_soisno
   996)     hksat         => cps%hksat
   997)     sucsat        => cps%sucsat
   998)     z             => cps%z
   999)     zi            => cps%zi
  1000)     watsat        => cps%watsat
  1001)     hkdepth       => cps%hkdepth
  1002)     zwt           => cws%zwt
  1003)     wa            => cws%wa
  1004)     wt            => cws%wt
  1005)     qcharge       => cws%qcharge
  1006)     eff_porosity  => cps%eff_porosity
  1007)     qflx_snwcp_liq => pwf_a%qflx_snwcp_liq
  1008)     qflx_dew_grnd => pwf_a%qflx_dew_grnd
  1009)     qflx_dew_snow => pwf_a%qflx_dew_snow
  1010)     qflx_sub_snow => pwf_a%qflx_sub_snow
  1011)     qflx_drain    => cwf%qflx_drain
  1012)     qflx_irrig    => cwf%qflx_irrig
  1013)     qflx_qrgwl    => cwf%qflx_qrgwl
  1014)     qflx_rsub_sat => cwf%qflx_rsub_sat
  1015)     eflx_impsoil  => cef%eflx_impsoil
  1016)     h2osoi_liq    => cws%h2osoi_liq
  1017)     h2osoi_ice    => cws%h2osoi_ice
  1018) 
  1019)     ! Get time step
  1020) 
  1021)     dtime = get_step_size()
  1022) 
  1023)     ! Convert layer thicknesses from m to mm
  1024) 
  1025)     do j = 1,nlevsoi
  1026)        do fc = 1, num_hydrologyc
  1027)           c = filter_hydrologyc(fc)
  1028)           dzmm(c,j) = dz(c,j)*1.e3_r8
  1029)        end do
  1030)     end do
  1031) 
  1032)     ! Initial set
  1033) 
  1034)     do fc = 1, num_hydrologyc
  1035)        c = filter_hydrologyc(fc)
  1036)        qflx_drain(c)    = 0._r8 
  1037)        rsub_bot(c)      = 0._r8
  1038)        qflx_rsub_sat(c) = 0._r8
  1039)        rsub_top(c)      = 0._r8
  1040)        fracice_rsub(c)  = 0._r8
  1041)     end do
  1042) 
  1043)     ! The layer index of the first unsaturated layer, i.e., the layer right above
  1044)     ! the water table
  1045) 
  1046)     do fc = 1, num_hydrologyc
  1047)        c = filter_hydrologyc(fc)
  1048)        jwt(c) = nlevsoi
  1049)        do j = 2,nlevsoi
  1050)           if(zwt(c) <= zi(c,j)) then
  1051)              jwt(c) = j-1
  1052)              exit
  1053)           end if
  1054)        enddo
  1055)     end do
  1056) 
  1057)     ! Topographic runoff
  1058)     do fc = 1, num_hydrologyc
  1059)        c = filter_hydrologyc(fc)
  1060)        fff(c)         = 1._r8/ hkdepth(c)
  1061)        dzsum = 0._r8
  1062)        icefracsum = 0._r8
  1063)        do j = jwt(c), nlevsoi
  1064)           dzsum  = dzsum + dzmm(c,j)
  1065)           icefracsum = icefracsum + icefrac(c,j) * dzmm(c,j)
  1066)        end do
  1067)        fracice_rsub(c) = max(0._r8,exp(-3._r8*(1._r8-(icefracsum/dzsum)))- exp(-3._r8))/(1.0_r8-exp(-3._r8))
  1068)        rsub_top(c)    = (1._r8 - fracice_rsub(c)) * 5.5e-3_r8 * exp(-fff(c)*zwt(c))
  1069)     end do
  1070) 
  1071)     rous = 0.2_r8
  1072) 
  1073)     ! Water table calculation
  1074) 
  1075)     do fc = 1, num_hydrologyc
  1076)        c = filter_hydrologyc(fc)
  1077) 
  1078)        ! Water storage in aquifer + soil
  1079)        wt(c)  = wt(c) + (qcharge(c) - rsub_top(c)) * dtime
  1080) 
  1081)        if(jwt(c) == nlevsoi) then             ! water table is below the soil column
  1082)           wa(c)  = wa(c) + (qcharge(c) -rsub_top(c)) * dtime
  1083)           wt(c)  = wa(c)
  1084)           zwt(c)     = (zi(c,nlevsoi) + 25._r8) - wa(c)/1000._r8/rous
  1085)           h2osoi_liq(c,nlevsoi) = h2osoi_liq(c,nlevsoi) + max(0._r8,(wa(c)-5000._r8))
  1086)           wa(c)  = min(wa(c), 5000._r8)
  1087)        else                                ! water table within soil layers
  1088)           if (jwt(c) == nlevsoi-1) then       ! water table within bottom soil layer
  1089) 
  1090)              zwt(c) = zi(c,nlevsoi)- (wt(c)-rous*1000._r8*25._r8) /eff_porosity(c,nlevsoi)/1000._r8
  1091) 
  1092)           else                                   ! water table within soil layers 1-9
  1093)              ws = 0._r8   ! water used to fill soil air pores regardless of water content
  1094)              do j = jwt(c)+2,nlevsoi
  1095)                ws = ws + eff_porosity(c,j) * 1000._r8 * dz(c,j)
  1096)              enddo
  1097)              zwt(c) = zi(c,jwt(c)+1)-(wt(c)-rous*1000_r8*25._r8-ws) /eff_porosity(c,jwt(c)+1)/1000._r8
  1098)           endif
  1099) 
  1100)           wtsub = 0._r8
  1101)           do j = jwt(c)+1, nlevsoi
  1102)              wtsub = wtsub + hk(c,j)*dzmm(c,j)
  1103)           end do
  1104) 
  1105)           ! Remove subsurface runoff
  1106)           do j = jwt(c)+1, nlevsoi 
  1107)              h2osoi_liq(c,j) = h2osoi_liq(c,j) - rsub_top(c)*dtime*hk(c,j)*dzmm(c,j)/wtsub
  1108)           end do
  1109)        end if
  1110) 
  1111)        zwt(c) = max(0.05_r8,zwt(c))
  1112)        zwt(c) = min(80._r8,zwt(c))
  1113) 
  1114)     end do
  1115) 
  1116)     !  excessive water above saturation added to the above unsaturated layer like a bucket
  1117)     !  if column fully saturated, excess water goes to runoff
  1118) 
  1119)     do j = nlevsoi,2,-1
  1120)        do fc = 1, num_hydrologyc
  1121)           c = filter_hydrologyc(fc)
  1122)           xsi(c)            = max(h2osoi_liq(c,j)-eff_porosity(c,j)*dzmm(c,j),0._r8)
  1123)           h2osoi_liq(c,j)   = min(eff_porosity(c,j)*dzmm(c,j), h2osoi_liq(c,j))
  1124)           h2osoi_liq(c,j-1) = h2osoi_liq(c,j-1) + xsi(c)
  1125)        end do
  1126)     end do
  1127) 
  1128)     do fc = 1, num_hydrologyc
  1129)        c = filter_hydrologyc(fc)
  1130)        xs1(c)          = max(max(h2osoi_liq(c,1),0._r8)-max(0._r8,(pondmx+watsat(c,1)*dzmm(c,1)-h2osoi_ice(c,1))),0._r8)
  1131)        h2osoi_liq(c,1) = min(max(0._r8,pondmx+watsat(c,1)*dzmm(c,1)-h2osoi_ice(c,1)), h2osoi_liq(c,1))
  1132)        qflx_rsub_sat(c)     = xs1(c) / dtime
  1133)     end do
  1134) 
  1135)     ! Limit h2osoi_liq to be greater than or equal to watmin.
  1136)     ! Get water needed to bring h2osoi_liq equal watmin from lower layer.
  1137)     ! If insufficient water in soil layers, get from aquifer water
  1138) 
  1139)     do j = 1, nlevsoi-1
  1140)        do fc = 1, num_hydrologyc
  1141)           c = filter_hydrologyc(fc)
  1142)           if (h2osoi_liq(c,j) < watmin) then
  1143)              xs(c) = watmin - h2osoi_liq(c,j)
  1144)           else
  1145)              xs(c) = 0._r8
  1146)           end if
  1147)           h2osoi_liq(c,j  ) = h2osoi_liq(c,j  ) + xs(c)
  1148)           h2osoi_liq(c,j+1) = h2osoi_liq(c,j+1) - xs(c)
  1149)        end do
  1150)     end do
  1151) 
  1152) ! Get water for bottom layer from layers above if possible
  1153)     j = nlevsoi
  1154)     do fc = 1, num_hydrologyc
  1155)        c = filter_hydrologyc(fc)
  1156)        if (h2osoi_liq(c,j) < watmin) then
  1157)           xs(c) = watmin-h2osoi_liq(c,j)
  1158)           searchforwater: do i = nlevsoi-1, 1, -1
  1159)              available_h2osoi_liq = max(h2osoi_liq(c,i)-watmin-xs(c),0._r8)
  1160)              if (available_h2osoi_liq .ge. xs(c)) then
  1161)                h2osoi_liq(c,j) = h2osoi_liq(c,j) + xs(c)
  1162)                h2osoi_liq(c,i) = h2osoi_liq(c,i) - xs(c)
  1163)                xs(c) = 0._r8
  1164)                exit searchforwater
  1165)              else
  1166)                h2osoi_liq(c,j) = h2osoi_liq(c,j) + available_h2osoi_liq
  1167)                h2osoi_liq(c,i) = h2osoi_liq(c,i) - available_h2osoi_liq
  1168)                xs(c) = xs(c) - available_h2osoi_liq
  1169)              end if
  1170)           end do searchforwater
  1171)        else
  1172)           xs(c) = 0._r8
  1173)        end if
  1174) ! Needed in case there is no water to be found
  1175)        h2osoi_liq(c,j) = h2osoi_liq(c,j) + xs(c)
  1176)        wt(c) = wt(c) - xs(c)
  1177) ! Instead of removing water from aquifer where it eventually
  1178) ! shows up as excess drainage to the ocean, take it back out of 
  1179) ! drainage
  1180)        rsub_top(c) = rsub_top(c) - xs(c)/dtime
  1181)     end do
  1182) 
  1183)     do fc = 1, num_hydrologyc
  1184)        c = filter_hydrologyc(fc)
  1185) 
  1186)        ! Sub-surface runoff and drainage
  1187) 
  1188)        qflx_drain(c) = qflx_rsub_sat(c) + rsub_top(c)
  1189) 
  1190)        ! Set imbalance for snow capping
  1191) 
  1192)        qflx_qrgwl(c) = qflx_snwcp_liq(c)
  1193) 
  1194)        ! Implicit evaporation term is now zero
  1195) 
  1196)        eflx_impsoil(c) = 0._r8
  1197) 
  1198)        ! Renew the ice and liquid mass due to condensation
  1199) 
  1200)        if (snl(c)+1 >= 1) then
  1201)           h2osoi_liq(c,1) = h2osoi_liq(c,1) + qflx_dew_grnd(c) * dtime
  1202)           h2osoi_ice(c,1) = h2osoi_ice(c,1) + (qflx_dew_snow(c) * dtime)
  1203)           if (qflx_sub_snow(c)*dtime > h2osoi_ice(c,1)) then
  1204)              qflx_sub_snow(c) = h2osoi_ice(c,1)/dtime
  1205)              h2osoi_ice(c,1) = 0._r8
  1206)           else
  1207)              h2osoi_ice(c,1) = h2osoi_ice(c,1) - (qflx_sub_snow(c) * dtime)
  1208)           end if
  1209)        end if
  1210)     end do
  1211)
