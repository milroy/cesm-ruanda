   159)         tr_iage, restart_age,  &
   160)         tr_FY,   restart_FY,   &

   163)         tr_aero, restart_aero         !MH for soot

   168) 
   169)       days_per_year = 365    ! number of days in a year
   170)       year_init = 0          ! initial year
   171)       istep0 = 0             ! no. of steps taken in previous integrations,
   172)                              ! real (dumped) or imagined (to set calendar)
   173) #ifndef CCSMCOUPLED
   174)       dt = 3600.0_dbl_kind   ! time step, s 
   175) #endif
   176)       npt = 99999            ! total number of time steps (dt) 
   177)       diagfreq = 24          ! how often diag output is written
   178)       print_points = .false. ! if true, print point data
   179)       print_global = .true.  ! if true, print global diagnostic data
   180)       lprint_stats = .false. ! if true, prints decomposition statistics
   181)       diag_type = 'stdout'
   182)       diag_file = 'ice_diag.d'
   183)       histfreq(:) = 'd'      ! output frequency option for different streams  **** note this is only for testing - needs to get changed back'
   184)       histfreq_n(:) = 1      ! output frequency of histfreq
   185)       hist_avg = .true.      ! if true, write time-averages (not snapshots)
   186)       history_dir  = ' '     ! write to executable dir for default
   187)       history_file = 'iceh'  ! history file name prefix
   188)       history_format = 'nc'  ! file format ('bin'=binary or 'nc'=netcdf)
   189)       write_ic = .false.     ! write out initial condition
   190)       incond_dir = history_dir ! write to history dir for default
   191)       incond_file = 'iceh_ic'! file prefix
   192)       dumpfreq='y'           ! restart frequency option
   193)       dumpfreq_n = 1         ! restart frequency
   194)       restart = .false.      ! if true, read restart files for initialization
   195)       restart_dir  = ' '     ! write to executable dir for default
   196)       restart_file = 'iced'  ! restart file name prefix
   197)       restart_format = 'nc'  ! file format ('bin'=binary or 'nc'=netcdf)
   198)       lcdf64       = .false. ! 64 bit offset for netCDF
   199) #ifdef CCSMCOUPLED
   200)       pointer_file = 'rpointer.ice' // trim(inst_suffix)
   201) #else
   202)       pointer_file = 'ice.restart_file'
   203) #endif
   204)       ice_ic       = 'default'      ! latitude and sst-dependent
   205)       grid_format  = 'bin'          ! file format ('bin'=binary or 'nc'=netcdf)
   206)       grid_type    = 'rectangular'  ! define rectangular grid internally
   207)       grid_file    = 'unknown_grid_file'
   208)       kmt_file     = 'unknown_kmt_file'
   209)       gridcpl_file = 'unknown_gridcpl_file'
   210) 
   211)       kitd = 1           ! type of itd conversions (0 = delta, 1 = linear)
   212)       kcatbound = 1      ! category boundary formula (0 = old, 1 = new)
   213)       kdyn = 1           ! type of dynamics (1 = evp)
   214)       xndt_dyn = c1      ! dynamic time steps per thermodynamic time step
   215)       ndte = 120         ! subcycles per dynamics timestep:  ndte=dt_dyn/dte
   216)       evp_damping = .false.  ! if true, use damping procedure in evp dynamics

   218)       splitcomm_dyn = .false.    ! allow overlap of computation and halo update in subcycling
   219)       maskhalo_dyn = .true.     ! compute masked halo before subcycling for use in halo update
   220)       maskhalo_stress = .true.  ! compute masked halo before tripole stress update
   221)       maskhalo_remap = .true.   ! compute masked halo before large tracer update in remap
   222)       maskhalo_bound = .true.   ! compute masked halo before state bound
   223)       kstrength = 1          ! 1 = Rothrock 75 strength, 0 = Hibler 79
   224)       krdg_partic = 1        ! 1 = new participation, 0 = Thorndike et al 75
   225)       krdg_redist = 1        ! 1 = new redistribution, 0 = Hibler 80
   226)       advection  = 'remap'   ! incremental remapping transport scheme
   227)       shortwave = 'default'  ! or 'dEdd' (delta-Eddington)
   228)       albedo_type = 'default'! or 'constant'
   229)       heat_capacity = .true. ! nonzero heat capacity (F => 0-layer thermo)
   230)       calc_Tsfc = .true.     ! calculate surface temperature
   231)       Tfrzpt    = 'linear_S' ! ocean freezing temperature, 'constant'=-1.8C
   232)       update_ocn_f = .false. ! include fresh water and salt fluxes for frazil
   233)       R_ice     = 0.00_dbl_kind   ! tuning parameter for sea ice
   234)       R_pnd     = 0.00_dbl_kind   ! tuning parameter for ponded sea ice
   235)       R_snw     = 0.00_dbl_kind   ! tuning parameter for snow over sea ice
   236)       dT_mlt_in = 1.50_dbl_kind   ! melt transition (tuning)
   237)       rsnw_melt_in = 1500._dbl_kind ! max snow grain radius (tuning)
   238)       albicev   = 0.78_dbl_kind   ! visible ice albedo for h > ahmax
   239)       albicei   = 0.36_dbl_kind   ! near-ir ice albedo for h > ahmax
   240)       albsnowv  = 0.98_dbl_kind   ! cold snow albedo, visible
   241)       albsnowi  = 0.70_dbl_kind   ! cold snow albedo, near IR
   242)       atmbndy   = 'default'  ! or 'constant'
   243) 
   244)       fyear_init = 1900           ! first year of forcing cycle
   245)       ycycle = 1                  ! number of years in forcing cycle
   246)       atm_data_format = 'bin'     ! file format ('bin'=binary or 'nc'=netcdf)
   247)       atm_data_type   = 'default'
   248)       atm_data_dir    = ' '
   249)       calc_strair     = .true.    ! calculate wind stress
   250)       precip_units    = 'mks'     ! 'mm_per_month' or
   251)                                   ! 'mm_per_sec' = 'mks' = kg/m^2 s
   252)       oceanmixed_ice  = .false.   ! if true, use internal ocean mixed layer
   253)       ocn_data_format = 'bin'     ! file format ('bin'=binary or 'nc'=netcdf)
   254)       sss_data_type   = 'default'
   255)       sst_data_type   = 'default'
   256)       ocn_data_dir    = ' '
   257)       oceanmixed_file = 'unknown_oceanmixed_file' ! ocean forcing data
   258)       restore_sst     = .false.   ! restore sst if true
   259)       trestore        = 90        ! restoring timescale, days (0 instantaneous)
   260)       restore_ice     = .false.   ! restore ice state on grid edges if true
   261)       dbug      = .false.         ! true writes diagnostics for input forcing
   262) 
   263)       latpnt(1) =  90._dbl_kind   ! latitude of diagnostic point 1 (deg)
   264)       lonpnt(1) =   0._dbl_kind   ! longitude of point 1 (deg)
   265)       latpnt(2) = -65._dbl_kind   ! latitude of diagnostic point 2 (deg)
   266)       lonpnt(2) = -45._dbl_kind   ! longitude of point 2 (deg)
   267) 
   268) #ifndef CCSMCOUPLED
   269)       runid   = 'unknown'   ! run ID, only used in CCSM
   270)       runtype = 'initial'   ! run type: 'initial', 'continue'
   271) #endif
   272)

   279)       filename_iage = 'none'

   290)       restart_aero = .false. ! aerosol restart   MH

   296)       ! read from input file

   298) 
   299)       call get_fileunit(nu_nml)
   300) 
   301) #ifdef CCSMCOUPLED
   302)       nml_filename  = 'ice_in'//trim(inst_suffix)
   303) #endif
   304)       if (my_task == master_task) then
   305)          open (nu_nml, file=nml_filename, status='old',iostat=nml_error)
   306)          if (nml_error /= 0) then
   307)             nml_error = -1
   308)          else
   309)             nml_error =  1
   310)          endif
   311)          do while (nml_error > 0)
   312)             print*,'Reading setup_nml'
   313)             read(nu_nml, nml=setup_nml,iostat=nml_error)
   314)             print*,'Reading grid_nml'
   315)             read(nu_nml, nml=grid_nml,iostat=nml_error)

   317)             read(nu_nml, nml=ice_nml,iostat=nml_error)
   318)             print*,'Reading tracer_nml'
   319)             read(nu_nml, nml=tracer_nml,iostat=nml_error)

   321)          if (nml_error == 0) close(nu_nml)
   322)       endif
   323)       call broadcast_scalar(nml_error, master_task)

   325)          call abort_ice('ice: error reading namelist')

   329)

   346)          inquire(file='ice_modelio.nml'//trim(inst_suffix),exist=exists)

   348)             nu_diag = shr_file_getUnit()
   349)             call shr_file_setIO('ice_modelio.nml'//trim(inst_suffix),nu_diag)
   350)          end if
   351)       end if
   352)       if (trim(ice_ic) /= 'default' .and. trim(ice_ic) /= 'none') then

   354)       end if
   355) #else
   356)       if (trim(diag_type) == 'file') call get_fileunit(nu_diag)

   358) 
   359)

   361)          if (trim(diag_type) == 'file') then

   364)          endif
   365)          write(nu_diag,*) '--------------------------------'
   366)          write(nu_diag,*) '  CICE model diagnostic output  '
   367)          write(nu_diag,*) '--------------------------------'
   368)          write(nu_diag,*) ' '
   369)       endif
   370)

   379)             'WARNING: Need ice_ic = <filename>.'
   380)             write(nu_diag,*) &
   381)             'WARNING: Initializing using ice_ic conditions'
   382)             endif
   383)             restart = .false.
   384)          endif
   385)       endif
   386)       if (runtype == 'initial' .and. .not.(restart)) then
   387)          if (ice_ic /= 'none' .and. ice_ic /= 'default') then
   388)             if (my_task == master_task) then
   389)             write(nu_diag,*) &
   390)             'WARNING: runtype, restart, ice_ic are inconsistent:'
   391)             write(nu_diag,*) runtype, restart, ice_ic
   392)             write(nu_diag,*) &
   393)             'WARNING: Initializing with NO ICE: '
   394)             endif
   395)             ice_ic = 'none'
   396)          endif
   397)       endif
   398) 
   399) #ifndef ncdf

   401)       history_format  = 'bin'

   405)       ocn_data_format = 'bin'
   406) #endif
   407) 
   408)       if (days_per_year /= 365) shortwave = 'default' ! definite conflict
   409)

   412)

   416)          kitd = 0
   417)       endif
   418) 
   419)       if (trim(atm_data_type) == 'monthly' .and. calc_strair) &
   420)          calc_strair = .false.
   421) 
   422)       if (trim(atm_data_type) == 'hadgem' .and. & 
   423)              trim(precip_units) /= 'mks') then
   424)          if (my_task == master_task) &
   425)          write (nu_diag,*) &
   426)          'WARNING: HadGEM atmospheric data chosen with wrong precip_units'
   427)          write (nu_diag,*) &
   428)          'WARNING: Changing precip_units to mks (i.e. kg/m2 s).'
   429)          precip_units='mks'
   430)       endif
   431) 
   432)       call broadcast_scalar(days_per_year,      master_task)
   433)       call broadcast_scalar(year_init,          master_task)
   434)       call broadcast_scalar(istep0,             master_task)
   435)       call broadcast_scalar(dt,                 master_task)
   436)       call broadcast_scalar(npt,                master_task)
   437)       call broadcast_scalar(diagfreq,           master_task)
   438)       call broadcast_scalar(print_points,       master_task)
   439)       call broadcast_scalar(print_global,       master_task)
   440)       call broadcast_scalar(diag_type,          master_task)
   441)       call broadcast_scalar(diag_file,          master_task)
   442)       call broadcast_scalar(history_format,     master_task)
   443) !     call broadcast_scalar(histfreq,           master_task)
   444) !     call broadcast_scalar(histfreq_n,         master_task)
   445)       do ns=1,max_nstrm
   446)          call broadcast_scalar(histfreq(ns),         master_task)
   447)          call broadcast_scalar(histfreq_n(ns),       master_task)
   448)       enddo
   449) !     call broadcast_array(histfreq_n(:),       master_task)
   450)       call broadcast_scalar(hist_avg,           master_task)
   451)       call broadcast_scalar(history_dir,        master_task)
   452)       call broadcast_scalar(write_ic,           master_task)
   453)       call broadcast_scalar(incond_dir,         master_task)
   454)       call broadcast_scalar(incond_file,        master_task)
   455)       call broadcast_scalar(dumpfreq,           master_task)
   456)       call broadcast_scalar(dumpfreq_n,         master_task)
   457)       call broadcast_scalar(restart_file,       master_task)
   458)       call broadcast_scalar(restart,            master_task)
   459)       call broadcast_scalar(restart_dir,        master_task)
   460)       call broadcast_scalar(restart_format,     master_task)
   461)       call broadcast_scalar(lcdf64,             master_task)
   462) #ifndef CCSM_COUPLED
   463)       call broadcast_scalar(pointer_file,       master_task)
   464) #endif
   465)       call broadcast_scalar(grid_format,        master_task)
   466)       call broadcast_scalar(grid_type,          master_task)
   467)       call broadcast_scalar(grid_file,          master_task)
   468)       call broadcast_scalar(kmt_file,           master_task)
   469)       call broadcast_scalar(kitd,               master_task)
   470)       call broadcast_scalar(gridcpl_file,       master_task)
   471)       call broadcast_scalar(kcatbound,          master_task)

   791) ! !IROUTINE: init_state - initialize ice state variables
   792) !
   793) ! !INTERFACE:
   794) !
   795)       subroutine init_state
   796) !
   797) ! !DESCRIPTION:
   798) !
   799) ! Initialize state for the itd model
   800) !
   801) ! !REVISION HISTORY:
   802) !
   803) ! authors: C. M. Bitz, UW
   804) !          William H. Lipscomb, LANL
   805) !
   806) ! !USES:
   807) !
   808)       use ice_blocks
   809)       use ice_domain
   810)       use ice_flux, only: sst, Tf, Tair
   811)       use ice_grid
   812)       use ice_state
   813)       use ice_itd
   814)       use ice_exit
   815)       use ice_therm_vertical, only: heat_capacity
   816) !
   817) ! !INPUT/OUTPUT PARAMETERS:
   818) !
   819) !EOP
   820) !
   821)       integer (kind=int_kind) :: &
   822)          i, j        , & ! horizontal indices
   823)          it, n       , & ! tracer index
   824)          iblk            ! block index
   825) 
   826)       !-----------------------------------------------------------------
   827)       ! Check number of layers in ice and snow.
   828)       !-----------------------------------------------------------------
   829) 
   830)       if (my_task == master_task) then
   831)  
   832)          if (nilyr < 1) then
   833)             write (nu_diag,*) 'nilyr =', nilyr
   834)             write (nu_diag,*) 'Must have at least one ice layer'
   835)             call abort_ice ('ice_init: Not enough ice layers')
   836)          endif
   837) 
   838)          if (nslyr < 1) then
   839)             write (nu_diag,*) 'nslyr =', nslyr
   840)             write (nu_diag,*) 'Must have at least one snow layer'
   841)             call abort_ice('ice_init: Not enough snow layers')
   842)          endif
   843) 
   844)          if (.not.heat_capacity) then
   845) 
   846)             write (nu_diag,*) 'WARNING - Zero-layer thermodynamics'
   847) 
   848)             if (nilyr > 1) then
   849)                write (nu_diag,*) 'nilyr =', nilyr
   850)                write (nu_diag,*)        &
   851)                     'Must have nilyr = 1 if heat_capacity = F'
   852)                call abort_ice('ice_init: Too many ice layers')
   853)             endif
   854) 
   855)             if (nslyr > 1) then
   856)                write (nu_diag,*) 'nslyr =', nslyr
   857)                write (nu_diag,*)        &
   858)                     'Must have nslyr = 1 if heat_capacity = F'
   859)                call abort_ice('ice_init: Too many snow layers')
   860)             endif
   861) 
   862)          endif   ! heat_capacity = F
   863) 
   864)       endif      ! my_task
   865) 
   866)       !-----------------------------------------------------------------
   867)       ! Set tracer types
   868)       !-----------------------------------------------------------------
   869) 
   870)       trcr_depend(nt_Tsfc)  = 0   ! ice/snow surface temperature
   871)       if (tr_iage) trcr_depend(nt_iage)  = 1   ! volume-weighted ice age
   872)       if (tr_FY)   trcr_depend(nt_FY)    = 0   ! area-weighted FY conc
   873)       if (tr_lvl)  trcr_depend(nt_alvl)  = 0   ! level ice area
   874)       if (tr_lvl)  trcr_depend(nt_vlvl)  = 1   ! level ice volume
   875)       if (tr_pond) trcr_depend(nt_volpn) = 0   ! melt pond volume
   876)       if (tr_aero) then
   877)         do n=1,n_aero
   878)          trcr_depend(nt_aero+(n-1)*4  :nt_aero+(n-1)*4+1) = 2 ! snow volume-weighted MH
   879)          trcr_depend(nt_aero+(n-1)*4+2:nt_aero+(n-1)*4+3) = 1 ! volume-weighted MH
   880)         enddo
   881)       endif
   882) 
   883)       !$OMP PARALLEL DO PRIVATE(iblk,it)
   884)       do iblk = 1, nblocks
   885) 
   886)       !-----------------------------------------------------------------
   887)       ! Set state variables
   888)       !-----------------------------------------------------------------
   889) 
   890)          call set_state_var (nx_block,            ny_block,            &
   891)                              tmask(:,:,    iblk),                      &
   892)                              ULON (:,:,    iblk), ULAT (:,:,    iblk), &
   893)                              Tair (:,:,    iblk), sst  (:,:,    iblk), &
   894)                              Tf   (:,:,    iblk), trcr_depend,         &
   895)                              aicen(:,:,  :,iblk), trcrn(:,:,:,:,iblk), &
   896)                              vicen(:,:,  :,iblk), vsnon(:,:,  :,iblk), &
   897)                              eicen(:,:,  :,iblk), esnon(:,:,  :,iblk))
   898) 
   899)       !-----------------------------------------------------------------
   900)       ! compute aggregate ice state and open water area
   901)       !-----------------------------------------------------------------
   902) 
   903)          aice(:,:,iblk) = c0
   904)          vice(:,:,iblk) = c0
   905)          vsno(:,:,iblk) = c0
   906)          eice(:,:,iblk) = c0
   907)          esno(:,:,iblk) = c0
   908)          do it = 1, max_ntrcr
   909)             trcr(:,:,it,iblk) = c0
   910)          enddo
   911) 
   912)          call aggregate (nx_block, ny_block,  &
   913)                          aicen(:,:,:,iblk),   &
   914)                          trcrn(:,:,:,:,iblk), &
   915)                          vicen(:,:,:,iblk),   &
   916)                          vsnon(:,:,:,iblk),   &
   917)                          eicen(:,:,:,iblk),   &
   918)                          esnon(:,:,:,iblk),   &
   919)                          aice (:,:,  iblk),   &
   920)                          trcr (:,:,:,iblk),   &
   921)                          vice (:,:,  iblk),   &
   922)                          vsno (:,:,  iblk),   &
   923)                          eice (:,:,  iblk),   &
   924)                          esno (:,:,  iblk),   &
   925)                          aice0(:,:,  iblk),   &
   926)                          tmask(:,:,  iblk),   &
   927)                          ntrcr,               &
   928)                          trcr_depend)
   929) 
   930)          aice_init(:,:,iblk) = aice(:,:,iblk)
   931) 
   932)       enddo                     ! iblk
   933)       !$OMP END PARALLEL DO
   934) 
   935)       !-----------------------------------------------------------------
   936)       ! ghost cell updates
   937)       !-----------------------------------------------------------------
   938) 
   939)       call bound_state (aicen, trcrn, &
   940)                         vicen, vsnon, &
   941)                         eicen, esnon)
   942)
