   368) subroutine biharmonic_wk_scalar_minmax(elem,qtens,deriv,edgeq,hybrid,nets,nete,emin,emax)
   369) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   370) ! compute weak biharmonic operator
   371) !    input:  qtens = Q
   372) !    output: qtens = weak biharmonic of Q and Q element min/max
   373) !
   374) !    note: emin/emax must be initialized with Q element min/max.  
   375) !
   376) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   377) type (hybrid_t)      , intent(in) :: hybrid
   378) type (element_t)     , intent(inout), target :: elem(:)
   379) integer :: nets,nete
   380) real (kind=real_kind), dimension(np,np,nlev,qsize,nets:nete) :: qtens
   381) type (EdgeBuffer_t)  , intent(inout) :: edgeq
   382) type (derivative_t)  , intent(in) :: deriv
   383) real (kind=real_kind), intent(out), dimension(nlev,qsize,nets:nete) :: emin,emax
   384) 
   385) ! local
   386) integer :: k,kptr,i,j,ie,ic,q
   387) real (kind=real_kind), dimension(np,np) :: lap_p
   388) real (kind=real_kind) :: Qmin(np,np,nlev,qsize)
   389) real (kind=real_kind) :: Qmax(np,np,nlev,qsize)
   390) logical var_coef1
   391) 
   392)    !if tensor hyperviscosity with tensor V is used, then biharmonic operator is (\grad\cdot V\grad) (\grad \cdot \grad) 
   393)    !so tensor is only used on second call to laplace_sphere_wk
   394)    var_coef1 = .true.
   395)    if(hypervis_scaling > 0)    var_coef1 = .false.
   396) 
   397) 
   398) 
   399)    do ie=nets,nete
   400) #if (defined ELEMENT_OPENMP)
   401) !$omp parallel do private(k, q, lap_p)
   402) #endif
   403)       do q=1,qsize      
   404)       do k=1,nlev    !  Potential loop inversion (AAM)
   405)          Qmin(:,:,k,q)=emin(k,q,ie)  ! need to set all values in element for
   406)          Qmax(:,:,k,q)=emax(k,q,ie)  ! edgeVpack routine below
   407)          lap_p(:,:) = qtens(:,:,k,q,ie)
   408) ! Original use of qtens on left and right hand sides caused OpenMP errors (AAM)
   409)          qtens(:,:,k,q,ie)=laplace_sphere_wk(lap_p,deriv,elem(ie),var_coef=var_coef1)
   410)       enddo
   411)       enddo
   412)       call edgeVpack(edgeq, qtens(:,:,:,:,ie),qsize*nlev,0,elem(ie)%desc)
   413)       call edgeVpack(edgeq,Qmin,nlev*qsize,nlev*qsize,elem(ie)%desc)
   414)       call edgeVpack(edgeq,Qmax,nlev*qsize,2*nlev*qsize,elem(ie)%desc)
   415)    enddo
   416)    
   417)    call bndry_exchangeV(hybrid,edgeq)
   418)    
   419)    do ie=nets,nete
   420)       do q=1,qsize      
   421)       do k=1,nlev
   422)          Qmin(:,:,k,q)=emin(k,q,ie)  ! restore element data.  we could avoid
   423)          Qmax(:,:,k,q)=emax(k,q,ie)  ! this by adding a "ie" index to Qmin/max
   424)       enddo
   425)       enddo
   426) ! WARNING - edgeVunpackMin/Max take second argument as input/ouput
   427)       call edgeVunpack(edgeq, qtens(:,:,:,:,ie),qsize*nlev,0,elem(ie)%desc)
   428)       call edgeVunpackMin(edgeq, Qmin,qsize*nlev,qsize*nlev,elem(ie)%desc)
   429)       call edgeVunpackMax(edgeq, Qmax,qsize*nlev,2*qsize*nlev,elem(ie)%desc)
   430) 
   431)       ! apply inverse mass matrix, then apply laplace again
   432) #if (defined ELEMENT_OPENMP)
   433) !$omp parallel do private(k, q, lap_p)
   434) #endif
   435)       do q=1,qsize      
   436)       do k=1,nlev    !  Potential loop inversion (AAM)
   437)          lap_p(:,:)=elem(ie)%rspheremp(:,:)*qtens(:,:,k,q,ie)
   438)          qtens(:,:,k,q,ie)=laplace_sphere_wk(lap_p,deriv,elem(ie),var_coef=.true.)
   439)          ! note: only need to consider the corners, since the data we packed was
   440)          ! constant within each element
   441)          emin(k,q,ie)=min(qmin(1,1,k,q),qmin(1,np,k,q),qmin(np,1,k,q),qmin(np,np,k,q))
   442)          emin(k,q,ie)=max(emin(k,q,ie),0d0)
   443)          emax(k,q,ie)=max(qmax(1,1,k,q),qmax(1,np,k,q),qmax(np,1,k,q),qmax(np,np,k,q))
   444)       enddo
   445)       enddo
   446)    enddo
   447) #ifdef DEBUGOMP
   448) #if (! defined ELEMENT_OPENMP)
   449) !$OMP BARRIER
   450) #endif
   451) #endif
   452) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   453) end subroutine

   805) subroutine neighbor_minmax(elem,hybrid,edgeMinMax,nets,nete,min_neigh,max_neigh)
   806) !
   807) ! compute Q min&max over the element and all its neighbors
   808) !
   809) !
   810) integer :: nets,nete
   811) type (element_t)     , intent(in) :: elem(:)
   812) type (hybrid_t)      , intent(in) :: hybrid
   813) type (EdgeBuffer_t)  , intent(inout) :: edgeMinMax
   814) real (kind=real_kind) :: min_neigh(nlev,qsize,nets:nete)
   815) real (kind=real_kind) :: max_neigh(nlev,qsize,nets:nete)
   816) 
   817) ! local
   818) integer :: ie,k,q
   819) real (kind=real_kind) :: Qmin(np,np,nlev,qsize)
   820) real (kind=real_kind) :: Qmax(np,np,nlev,qsize)
   821) 
   822) 
   823)     ! compute Qmin, Qmax
   824)     do ie=nets,nete
   825) #if (defined ELEMENT_OPENMP)
   826) !$omp parallel do private(k, q)
   827) #endif
   828)        do q=1,qsize	
   829)           do k=1,nlev
   830)              Qmin(:,:,k,q)=min_neigh(k,q,ie)
   831)              Qmax(:,:,k,q)=max_neigh(k,q,ie)
   832)           enddo
   833)        end do
   834)        call edgeVpack(edgeMinMax,Qmin,nlev*qsize,0,elem(ie)%desc)
   835)        call edgeVpack(edgeMinMax,Qmax,nlev*qsize,nlev*qsize,elem(ie)%desc)
   836)     enddo
   837) 
   838)     call bndry_exchangeV(hybrid,edgeMinMax)
   839)        
   840)     do ie=nets,nete
   841) #if (defined ELEMENT_OPENMP)
   842) !$omp parallel do private(k, q)
   843) #endif
   844)        do q=1,qsize	
   845)           do k=1,nlev         
   846)              Qmin(:,:,k,q)=min_neigh(k,q,ie) ! restore element data.  we could avoid
   847)              Qmax(:,:,k,q)=max_neigh(k,q,ie) ! this by adding a "ie" index to Qmin/max
   848)           enddo
   849)        end do
   850) ! WARNING - edgeVunpackMin/Max take second argument as input/ouput
   851)        call edgeVunpackMin(edgeMinMax,Qmin,nlev*qsize,0,elem(ie)%desc)

   854) !$omp parallel do private(k, q)

   861)              min_neigh(k,q,ie)=max(min_neigh(k,q,ie),0d0)
   862)              max_neigh(k,q,ie)=max(qmax(1,1,k,q),qmax(1,np,k,q),qmax(np,1,k,q),qmax(np,np,k,q))
   863)           enddo
   864)        end do
   865)     end do
   866) #ifdef DEBUGOMP
   867) #if (! defined ELEMENT_OPENMP)
   868) !$OMP BARRIER
   869) #endif
   870) #endif

   872) end subroutine

   878) subroutine neighbor_minmax(elem,hybrid,edgeMinMax,nets,nete,nt,min_neigh,max_neigh,min_var,max_var,kmass)
   879) !
   880) ! compute Q min&max over the element and all its neighbors
   881) !
   882) !
   883) integer :: nets,nete,nt
   884) type (hybrid_t)      , intent(in) :: hybrid
   885) type (element_t)     , intent(inout) :: elem(:)
   886) type (EdgeBuffer_t)  , intent(in) :: edgeMinMax
   887) real (kind=real_kind) :: min_neigh(nlev,nets:nete)
   888) real (kind=real_kind) :: max_neigh(nlev,nets:nete)
   889) real (kind=real_kind),optional :: min_var(nlev,nets:nete)
   890) real (kind=real_kind),optional :: max_var(nlev,nets:nete)
   891) real (kind=real_kind) :: Qmin(np,np,nlev)
   892) real (kind=real_kind) :: Qmax(np,np,nlev)
   893) real (kind=real_kind) :: Qvar(np,np,nlev)
   894) type (EdgeBuffer_t)          :: edgebuf
   895) integer, optional :: kmass
   896) 
   897) ! local
   898) integer :: ie,k,q
   899) 
   900)   if(present(kmass))then
   901) !the check if kmass is a valid number is done in sweq_mod
   902)     do k=1,nlev
   903)       if(k.ne.kmass)then
   904)          do ie=nets,nete
   905)             elem(ie)%state%p(:,:,k,nt)=elem(ie)%state%p(:,:,k,nt)/&
   906)             elem(ie)%state%p(:,:,kmass,nt)
   907)          enddo
   908)       endif
   909)     enddo
   910)   endif
   911) 
   912) 
   913) 
   914)     ! create edge buffer for 3 fields
   915)     call initEdgeBuffer(edgebuf,3*nlev)
   916) 
   917) 
   918)     ! compute p min, max
   919)     do ie=nets,nete
   920) #if (defined ELEMENT_OPENMP)
   921) !$omp parallel do private(k)
   922) #endif
   923)        do k=1,nlev
   924)           Qmin(:,:,k)=minval(elem(ie)%state%p(:,:,k,nt))
   925)           Qmax(:,:,k)=maxval(elem(ie)%state%p(:,:,k,nt))
   926)           ! max - min - crude approximation to TV within the element:
   927)           Qvar(:,:,k)=Qmax(1,1,k)-Qmin(1,1,k)
   928)        enddo
   929)        call edgeVpack(edgebuf,Qmax,nlev,0,elem(ie)%desc)
   930)        call edgeVpack(edgebuf,Qmin,nlev,nlev,elem(ie)%desc)
   931)        call edgeVpack(edgebuf,Qvar,nlev,2*nlev,elem(ie)%desc)
   932)     enddo
   933)     
   934)     call bndry_exchangeV(hybrid,edgebuf)
   935)        
   936)     do ie=nets,nete
   937) #if (defined ELEMENT_OPENMP)
   938) !$omp parallel do private(k)
   939) #endif
   940)        do k=1,nlev
   941)           Qmin(:,:,k)=minval(elem(ie)%state%p(:,:,k,nt))
   942)           Qmax(:,:,k)=maxval(elem(ie)%state%p(:,:,k,nt))
   943)        enddo
   944) 
   945)        ! now unpack the min
   946)        if (present(min_var)) then
   947) #if (defined ELEMENT_OPENMP)
   948) !$omp parallel do private(k)
   949) #endif
   950)           do k=1,nlev
   951)              Qvar(:,:,k)=Qmax(1,1,k)-Qmin(1,1,k)
   952)           enddo
   953) ! WARNING - edgeVunpackMin/Max take second argument as input/ouput
   954)           call edgeVunpackMin(edgebuf,Qvar,nlev,2*nlev,elem(ie)%desc)
   955) #if (defined ELEMENT_OPENMP)
   956) !$omp parallel do private(k)
   957) #endif
   958)           do k=1,nlev
   959)              min_var(k,ie)=minval(Qvar(:,:,k))
   960)           enddo
   961)        endif
   962) 
   963)        ! now unpack the max
   964)        if (present(max_var)) then
   965) #if (defined ELEMENT_OPENMP)
   966) !$omp parallel do private(k)
   967) #endif
   968)           do k=1,nlev
   969)              Qvar(:,:,k)=Qmax(1,1,k)-Qmin(1,1,k)
   970)           enddo
   971) ! WARNING - edgeVunpackMin/Max take second argument as input/ouput
   972)           call edgeVunpackMax(edgebuf,Qvar,nlev,2*nlev,elem(ie)%desc)
   973) #if (defined ELEMENT_OPENMP)
   974) !$omp parallel do private(k)
   975) #endif
   976)           do k=1,nlev
   977)              max_var(k,ie)=maxval(Qvar(:,:,k))
   978)           enddo
   979)        endif
   980) 
   981) 
   982) ! WARNING - edgeVunpackMin/Max take second argument as input/ouput
   983)        call edgeVunpackMax(edgebuf,Qmax,nlev,0,elem(ie)%desc)
   984)        call edgeVunpackMin(edgebuf,Qmin,nlev,nlev,elem(ie)%desc)
   985) #if (defined ELEMENT_OPENMP)
