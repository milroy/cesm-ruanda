   338) !     Get the value of the {\tt ESMF\_Time} in units specified by the user
   339) !     via F90 optional arguments.
   340) !
   341) !     Time manager represents and manipulates time internally with integers
   342) !     to maintain precision. Hence, user-specified floating point values are
   343) !     converted internally from integers.
   344) !
   345) !     See {\tt ../include/ESMC\_BaseTime.h and ../include/ESMC\_Time.h} for
   346) !     complete description.
   347) !     
   348) !     The arguments are:
   349) !     \begin{description}
   350) !     \item[time]
   351) !          The object instance to query
   352) !     \item[{[YY]}]
   353) !          Integer year CCYR (>= 32-bit)
   354) !     \item[{[YRl]}]
   355) !          Integer year CCYR (large, >= 64-bit)
   356) !     \item[{[MM]}]
   357) !          Integer month 1-12
   358) !     \item[{[DD]}]
   359) !          Integer day of the month 1-31
   360) !     \item[{[D]}]
   361) !          Integer Julian days (>= 32-bit)
   362) !     \item[{[Dl]}]
   363) !          Integer Julian days (large, >= 64-bit)
   364) !     \item[{[H]}]
   365) !          Integer hours
   366) !     \item[{[M]}]
   367) !          Integer minutes
   368) !     \item[{[S]}]
   369) !          Integer seconds (>= 32-bit)
   370) !     \item[{[Sl]}]
   371) !          Integer seconds (large, >= 64-bit)
   372) !     \item[{[MS]}]
   373) !          Integer milliseconds
   374) !     \item[{[US]}]
   375) !          Integer microseconds
   376) !     \item[{[NS]}]
   377) !          Integer nanoseconds
   378) !     \item[{[d\_]}]
   379) !          Double precision days
   380) !     \item[{[h\_]}]
   381) !          Double precision hours
   382) !     \item[{[m\_]}]
   383) !          Double precision minutes
   384) !     \item[{[s\_]}]
   385) !          Double precision seconds
   386) !     \item[{[ms\_]}]
   387) !          Double precision milliseconds
   388) !     \item[{[us\_]}]
   389) !          Double precision microseconds
   390) !     \item[{[ns\_]}]
   391) !          Double precision nanoseconds
   392) !     \item[{[Sn]}]
   393) !          Integer fractional seconds - numerator
   394) !     \item[{[Sd]}]
   395) !          Integer fractional seconds - denominator
   396) !     \item[{[rc]}]
   397) !          Return code; equals {\tt ESMF\_SUCCESS} if there are no errors.
   398) !     \end{description}
   399) !
   400) ! !REQUIREMENTS:
   401) !     TMG2.1, TMG2.5.1, TMG2.5.6

   404)       integer :: ierr

   411)       ierr = ESMF_SUCCESS

   413)       IF ( PRESENT( YY ) ) THEN
   414)         YY = time%YR
   415)       ENDIF
   416)       IF ( PRESENT( MM ) ) THEN
   417)         CALL timegetmonth( time, MM )
   418)       ENDIF
   419)       IF ( PRESENT( DD ) ) THEN
   420)         CALL timegetdayofmonth( time, DD )
   421)       ENDIF
   422) 
   423)       if (present(d) .or. present(dl)) then
   424)         cnt = 0
   425)         do i = 0,time%yr-1
   426)            cnt = cnt + ndaysinyear(i,time%calendar%type)
   427)         enddo
   428)         do i = time%yr,-1
   429)            cnt = cnt - ndaysinyear(i,time%calendar%type)
   430)         enddo
   431)         call timegetmonth(time,month)
   432)         do i = 1,month-1
   433)            cnt = cnt + ndaysinmonth(time%yr,i,time%calendar%type)
   434)         enddo
   435)         call timegetdayofmonth( time, dayofmonth)
   436)         cnt = cnt + dayofmonth
   437)         if (present(d)) then
   438)           d = cnt
   439)         endif
   440)         if (present(dl)) then
   441)           dl = cnt
   442)         endif
   443)       endif
   444) !
   445) !$$$ push HMS down into ESMF_BaseTime
   446)       IF ( PRESENT( H ) ) THEN
   447)         H = mod( time%basetime%S, SECONDS_PER_DAY ) / SECONDS_PER_HOUR
   448)       ENDIF
   449)       IF ( PRESENT( M ) ) THEN
   450)         M = mod( time%basetime%S, SECONDS_PER_HOUR) / SECONDS_PER_MINUTE
   451)       ENDIF
   452)       IF ( PRESENT( S ) ) THEN
   453)         S = mod( time%basetime%S, SECONDS_PER_MINUTE )
   454)       ENDIF
   455) 
   456)       IF ( PRESENT( S ) .AND. PRESENT( DD ) ) THEN
   457)         IF ( ( .NOT. PRESENT( H ) ) .AND. ( .NOT. PRESENT( M ) ) ) THEN
   458)           S = mod( time%basetime%S, SECONDS_PER_DAY )
   459)         ENDIF
   460)       ENDIF
   461)       IF ( PRESENT( MS ) ) THEN
   462)         IF ( time%basetime%Sd /= 0 ) THEN
   463)           MS = NINT( ( time%basetime%Sn*1.0D0 / time%basetime%Sd*1.0D0 ) * 1000.0D0 )
   464)         ELSE
   465)           MS = 0
   466)         ENDIF
   467)       ENDIF
   468)       IF ( PRESENT( Sd ) .AND. PRESENT( Sn ) ) THEN
   469)         Sd = time%basetime%Sd
   470)         Sn = time%basetime%Sn
   471)       ENDIF
   472)       IF ( PRESENT( dayOfYear ) ) THEN
   473)         CALL ESMF_TimeGetDayOfYear( time, dayOfYear, rc=ierr )
   474)       ENDIF
   475)       IF ( PRESENT( timeString ) ) THEN
   476)         ! This duplication for YMD is an optimization that avoids calling 
   477)         ! timegetmonth() and timegetdayofmonth() when it is not needed.  
   478)         year = time%YR
   479)         CALL timegetmonth( time, month )
   480)         CALL timegetdayofmonth( time, dayofmonth )
   481) !$$$ push HMS down into ESMF_BaseTime
   482)         hour = mod( time%basetime%S, SECONDS_PER_DAY ) / SECONDS_PER_HOUR
   483)         minute = mod( time%basetime%S, SECONDS_PER_HOUR) / SECONDS_PER_MINUTE
   484)         second = mod( time%basetime%S, SECONDS_PER_MINUTE )
   485)         CALL ESMFold_TimeGetString( year, month, dayofmonth, &
   486)                                     hour, minute, second, timeString )
   487)       ENDIF
   488)       IF ( PRESENT( dayOfYear_intvl ) ) THEN
   489)         year = time%YR
   490)         CALL ESMF_TimeSet( begofyear, yy=year, mm=1, dd=1, s=0, &
   491)                            calendar=time%calendar, rc=ierr )
   492)         IF ( ierr == ESMF_FAILURE)THEN
   493)            rc = ierr
   494)            RETURN
   495)         END IF
   496)         dayOfYear_intvl = time - begofyear
   497)       ENDIF
   498)       IF ( PRESENT( dayOfYear_r8) ) THEN
   499)         year = time%YR
   500)         CALL ESMF_TimeSet( begofyear, yy=year, mm=1, dd=1, s=0, &
   501)                            calendar=time%calendar, rc=ierr )
   502)         IF ( ierr == ESMF_FAILURE)THEN
   503)            rc = ierr
   504)            RETURN
   505)         END IF
   506)         CALL ESMF_TimeIntervalSet( day_step, d=1, s=0, rc=ierr )
   507)         IF ( ierr == ESMF_FAILURE)THEN
   508)            rc = ierr
   509)            RETURN
   510)         END IF
   511)         difftobegofyear = time - begofyear + day_step
   512)         CALL ESMF_TimeIntervalGet( difftobegofyear, d_r8=dayOfYear_r8, rc=ierr )
   513)         IF ( ierr == ESMF_FAILURE)THEN
   514)            rc = ierr
   515)            RETURN
   516)         END IF
   517)       ENDIF
   518) 
   519)       IF ( PRESENT( rc ) ) THEN
   520)         rc = ierr
   521)       ENDIF
   522) 
   523)       end subroutine ESMF_TimeGet
