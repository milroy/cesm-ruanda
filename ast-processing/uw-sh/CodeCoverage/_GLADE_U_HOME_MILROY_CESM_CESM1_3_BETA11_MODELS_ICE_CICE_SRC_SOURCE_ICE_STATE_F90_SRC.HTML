     7) !
     8) ! Primary state variables in various configurations
     9) ! Note: other state variables are at the end of this...
    10) ! The primary state variable names are:
    11) !-------------------------------------------------------------------
    12) ! for each category   aggregated over     units
    13) !                       categories
    14) !-------------------------------------------------------------------
    15) ! aicen(i,j,n)         aice(i,j)           ---
    16) ! vicen(i,j,n)         vice(i,j)           m
    17) ! vsnon(i,j,n)         vsno(i,j)           m
    18) ! eicen(i,j,k)         eice(i,j)           J/m^2
    19) ! esnon(i,j,k)         esno(i,j)           J/m^2
    20) ! trcrn(i,j,it,n)      trcr(i,j,it)        
    21) !
    22) ! Area is dimensionless because aice is the fractional area
    23) ! (normalized so that the sum over all categories, including open
    24) ! water, is 1.0).  That is why vice/vsno have units of m instead of
    25) ! m^3, and eice/esno have units of J/m^2 instead of J.
    26) !
    27) ! Variable names follow these rules:
    28) !
    29) ! (1) For 3D variables (indices i,j,n), write 'ice' or 'sno' or
    30) !     'sfc' and put an 'n' at the end.
    31) ! (2) For 2D variables (indices i,j) aggregated over all categories,
    32) !     write 'ice' or 'sno' or 'sfc' without the 'n'.
    33) ! (3) For 2D variables (indices i,j) associated with an individual
    34) !     category, write 'i' or 's' instead of 'ice' or 'sno' and put an 'n'
    35) !     at the end: e.g. hin, hsn.  These are not declared here
    36) !     but in individual modules (e.g., ice_therm_vertical).
    37) !

   186) ! !IROUTINE: bound_state - bound calls for ice state variables
   187) !
   188) ! !INTERFACE:

   190)       subroutine bound_state (aicen, trcrn, &
   191)                               vicen, vsnon, &
   192)                               eicen, esnon)
   193) !
   194) ! !DESCRIPTION:
   195) !
   196) ! Get ghost cell values for ice state variables in each thickness category.
   197) ! NOTE: This subroutine cannot be called from inside a block loop!
   198) !
   199) ! !REVISION HISTORY:
   200) !
   201) ! author: William H. Lipscomb, LANL
   202) !
   203) ! !USES:
   204) !
   205)       use ice_boundary
   206)       use ice_domain
   207)       use ice_constants
   208) !
   209) ! !INPUT/OUTPUT PARAMETERS:
   210) !
   211)       real (kind=dbl_kind), &
   212)          dimension(nx_block,ny_block,ncat,max_blocks), intent(inout) :: &
   213)          aicen , & ! fractional ice area
   214)          vicen , & ! volume per unit area of ice          (m)
   215)          vsnon     ! volume per unit area of snow         (m)
   216) 
   217)       real (kind=dbl_kind), &
   218)          dimension(nx_block,ny_block,max_ntrcr,ncat,max_blocks), &
   219)          intent(inout) :: &
   220)          trcrn     ! ice tracers
   221) 
   222)       real (kind=dbl_kind), &
   223)          dimension(nx_block,ny_block,ntilyr,max_blocks),intent(inout) :: &
   224)          eicen     ! energy of melting for each ice layer  (J/m^2)
   225) 
   226)       real (kind=dbl_kind), &
   227)          dimension(nx_block,ny_block,ntslyr,max_blocks),intent(inout) :: &
   228)          esnon     ! energy of melting for each snow layer (J/m^2)
   229) !
   230) !EOP
   231) !
   232)       integer (kind=int_kind) :: i,j,n,iblk
   233)       integer (kind=int_kind), &
   234)          dimension(nx_block,ny_block,max_blocks) :: halomask
   235)       type (ice_halo) :: halo_info_aicemask
   236) !
   237) 
   238)    if (maskhalo_bound) then
   239) 
   240)       call t_barrierf('state_bound_h1_BARRIER',MPI_COMM_ICE)
   241)       call t_startf('state_bound_h1')
   242) 
   243)       call ice_HaloUpdate (aicen,            halo_info, &
   244)                            field_loc_center, field_type_scalar)
   245) 
   246)       call t_stopf('state_bound_h1')
   247) 
   248)       call t_barrierf('state_bound_hm_BARRIER',MPI_COMM_ICE)
   249)       call t_startf('state_bound_hm')
   250)       halomask = 0
   251)       do iblk=1,nblocks
   252)       do n=1,ncat
   253)       do j=1,ny_block
   254)       do i=1,nx_block
   255)          if (aicen(i,j,n,iblk) > c0) halomask(i,j,iblk) = 1
   256)       enddo
   257)       enddo
   258)       enddo
   259)       enddo
   260) !tcx
   261) !      halomask = 1
   262)       call t_stopf('state_bound_hm')
   263) 
   264)       call t_barrierf('state_bound_hc_BARRIER',MPI_COMM_ICE)
   265)       call t_startf('state_bound_hc')
   266)       call ice_HaloMask(halo_info_aicemask, halo_info, halomask)
   267)       call t_stopf('state_bound_hc')
   268) 
   269)       call t_barrierf('state_bound_h2_BARRIER',MPI_COMM_ICE)
   270)       call t_startf('state_bound_h2')
   271) 
   272)       call ice_HaloUpdate (trcrn(:,:,1:ntrcr,:,:), halo_info_aicemask, &
   273)                            field_loc_center, field_type_scalar)
   274) 
   275)       call ice_HaloUpdate (vicen,            halo_info_aicemask, &
   276)                            field_loc_center, field_type_scalar)
   277) 
   278)       call ice_HaloUpdate (vsnon,            halo_info_aicemask, &
   279)                            field_loc_center, field_type_scalar)
   280) 
   281)       call ice_HaloUpdate (eicen,            halo_info_aicemask, &
   282)                            field_loc_center, field_type_scalar)
   283) 
   284)       call ice_HaloUpdate (esnon,            halo_info_aicemask, &
   285)                            field_loc_center, field_type_scalar)
   286)       call t_stopf('state_bound_h2')
   287) 
   288)       call t_barrierf('state_bound_hd_BARRIER',MPI_COMM_ICE)
   289)       call t_startf('state_bound_hd')
   290)       call ice_HaloDestroy(halo_info_aicemask)
   291)       call t_stopf('state_bound_hd')
   292) 
   293)    else
   294) 
   295)       call t_barrierf('state_bound_h1_BARRIER',MPI_COMM_ICE)
   296)       call t_startf('state_bound_h1')
   297) 
   298)       call ice_HaloUpdate (aicen,            halo_info, &
   299)                            field_loc_center, field_type_scalar)
   300) 
   301)       call t_stopf('state_bound_h1')
   302) 
   303)       call t_barrierf('state_bound_h2_BARRIER',MPI_COMM_ICE)
   304)       call t_startf('state_bound_h2')
   305) 
   306)       call ice_HaloUpdate (trcrn(:,:,1:ntrcr,:,:), halo_info, &
   307)                            field_loc_center, field_type_scalar)
   308) 
   309)       call ice_HaloUpdate (vicen,            halo_info, &
   310)                            field_loc_center, field_type_scalar)
   311) 
   312)       call ice_HaloUpdate (vsnon,            halo_info, &
   313)                            field_loc_center, field_type_scalar)
   314) 
   315)       call ice_HaloUpdate (eicen,            halo_info, &
   316)                            field_loc_center, field_type_scalar)
   317) 
   318)       call ice_HaloUpdate (esnon,            halo_info, &
   319)                            field_loc_center, field_type_scalar)
   320)       call t_stopf('state_bound_h2')
   321) 
   322)    endif
   323) 
   324)       end subroutine bound_state
