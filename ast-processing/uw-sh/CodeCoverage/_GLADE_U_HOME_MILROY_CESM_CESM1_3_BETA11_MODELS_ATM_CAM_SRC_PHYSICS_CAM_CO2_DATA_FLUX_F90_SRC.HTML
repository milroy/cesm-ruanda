    70) subroutine read_data_flux (input_file, xin)
    71)  
    72) !-------------------------------------------------------------------------------              
    73) ! Do initial read of time-varying 2d(lat,lon NetCDF dataset, 
    74) ! reading two data bracketing current timestep
    75) !-------------------------------------------------------------------------------
    76)  
    77)   use time_manager, only : get_curr_date, get_curr_calday, &
    78)                            is_perpetual, get_perp_date, get_step_size, is_first_step
    79)   use ioFileMod,    only : getfil
    80) 
    81)   implicit none
    82)  
    83) !---------------------------Common blocks-------------------------------
    84) ! Dummy arguments
    85)   character(len=*),   intent(in)    :: input_file
    86)   TYPE(read_interp),  intent(inout) :: xin   
    87)  
    88)   integer lonid                 ! netcdf id for longitude variable
    89)   integer latid                 ! netcdf id for latitude variable
    90)   integer timeid                ! netcdf id for time variable
    91)   integer dateid                ! netcdf id for date variable
    92)   integer secid                 ! netcdf id for seconds variable
    93)   integer londimid              ! netcdf id for longitude variable
    94)   integer latdimid              ! netcdf id for latitude variable
    95)  
    96)   integer dtime                 ! timestep size [seconds]
    97)   integer cnt3(3)               ! array of counts for each dimension
    98)   integer strt3(3)              ! array of starting indices
    99)   integer n                     ! indices
   100)   integer j                     ! latitude index
   101)   integer istat                 ! error return
   102)   integer  :: yr, mon, day      ! components of a date
   103)   integer  :: ncdate            ! current date in integer format [yyyymmdd]
   104)   integer  :: ncsec             ! current time of day [seconds]
   105)   real(r8) calday               ! calendar day (includes yr if no cycling)
   106)   real(r8) caldayloc            ! calendar day (includes yr if no cycling)
   107)  
   108)   xin%nm_f = 1
   109)   xin%np_f = 2
   110)  
   111) ! Allocate space for data.
   112)  
   113)   allocate( xin%co2flx(pcols,begchunk:endchunk), stat=istat )
   114)   call alloc_err( istat, 'CO2FLUX_READ', 'co2flx', &
   115)        pcols*(endchunk-begchunk+1) )
   116)  
   117)   allocate( xin%co2bdy(pcols,begchunk:endchunk,2), stat=istat )
   118)   call alloc_err( istat, 'CO2FLUX_READ', 'co2bdy', &
   119)        pcols*(endchunk-begchunk+1)*2 )
   120)  
   121) ! SPMD: Master does all the work.
   122)  
   123)   if (masterproc) then
   124) !
   125) ! Use year information only if not cycling sst dataset
   126) !
   127)      if (is_first_step()) then
   128)         dtime = get_step_size()
   129)         dtime = -dtime
   130)         calday = get_curr_calday(offset=dtime)
   131)      else
   132)         calday = get_curr_calday()
   133)      endif
   134)      if ( is_perpetual() ) then
   135)         call get_perp_date(yr, mon, day, ncsec)
   136)      else
   137)         if (is_first_step()) then
   138)            call get_curr_date(yr, mon, day, ncsec,offset=dtime)
   139)         else
   140)            call get_curr_date(yr, mon, day, ncsec)
   141)         endif
   142)      end if
   143)  
   144)      ncdate = yr*10000 + mon*100 + day
   145)  
   146)      caldayloc = calday + yr*daysperyear
   147)  
   148) ! Open NetCDF File
   149)  
   150)      call getfil(input_file, xin%locfn)
   151)      call handle_ncerr( nf90_open(xin%locfn, 0, xin%ncid_f),&
   152)        'co2_data_flux.F90:154')
   153)      write(iulog,*)'CO2FLUX_READ: NCOPN returns id ',xin%ncid_f,' for file ',trim(xin%locfn)
   154)  
   155) ! Get and check dimension info
   156)  
   157)      call handle_ncerr( nf90_inq_dimid( xin%ncid_f, 'lon', londimid ),&
   158)        'co2_data_flux.F90:160')
   159)      call handle_ncerr( nf90_inq_dimid( xin%ncid_f, 'lat', latdimid ),&
   160)        'co2_data_flux.F90:162')
   161)      call handle_ncerr( nf90_inq_dimid( xin%ncid_f, 'time',  timeid ),&
   162)        'co2_data_flux.F90:164')
   163) 
   164)      call handle_ncerr( nf90_inquire_dimension( xin%ncid_f, londimid, len=lonsiz     ),&
   165)        'co2_data_flux.F90:167')
   166)      call handle_ncerr( nf90_inquire_dimension( xin%ncid_f, latdimid, len=latsiz     ),&
   167)        'co2_data_flux.F90:169')
   168)      call handle_ncerr( nf90_inquire_dimension( xin%ncid_f, timeid,   len=xin%timesz ),&
   169)        'co2_data_flux.F90:171')
   170) 
   171)      allocate(xin%date_f(xin%timesz), xin%sec_f(xin%timesz))
   172) ! Get data id
   173)         
   174)      call handle_ncerr( nf90_inq_varid( xin%ncid_f, 'date',     dateid     ),&
   175)        'co2_data_flux.F90:192')
   176)      call handle_ncerr( nf90_inq_varid( xin%ncid_f, 'datesec',  secid      ),&
   177)        'co2_data_flux.F90:194')
   178)      call handle_ncerr( nf90_inq_varid( xin%ncid_f, 'CO2_flux', xin%fluxid ),&
   179)        'co2_data_flux.F90:196')
   180)  
   181) ! Retrieve entire date and sec variables.
   182)  
   183)      call handle_ncerr( nf90_get_var ( xin%ncid_f, dateid, xin%date_f ),&
   184)        'co2_data_flux.F90:201')
   185)      call handle_ncerr( nf90_get_var ( xin%ncid_f, secid,  xin%sec_f  ),&
   186)        'co2_data_flux.F90:203')
   187) 
   188) ! initialize
   189)  
   190)      strt3(1) = 1
   191)      strt3(2) = 1
   192)      strt3(3) = 1
   193)      cnt3(1)  = lonsiz
   194)      cnt3(2)  = latsiz
   195)      cnt3(3)  = 1
   196) 
   197)   endif
   198) 
   199) #ifdef SPMD
   200)   call mpibcast( cnt3, 2,     mpiint, 0, mpicom )
   201) #endif
   202)   allocate(xin%xvar(cnt3(1),cnt3(2),2))
   203)   if (masterproc) then
   204) ! Normal interpolation between consecutive time slices.
   205) 
   206)      do n=1,xin%timesz-1
   207) 
   208)         xin%np1_f = n + 1
   209)  
   210)         call bnddyi(xin%date_f(n  ),       xin%sec_f(n  ),       xin%cdayfm)
   211)         call bnddyi(xin%date_f(xin%np1_f), xin%sec_f(xin%np1_f), xin%cdayfp)
   212)  
   213)         yr         = xin%date_f(n)/10000
   214)         xin%cdayfm = xin%cdayfm + yr*daysperyear
   215) 
   216)         yr         = xin%date_f(xin%np1_f)/10000
   217)         xin%cdayfp = xin%cdayfp + yr*daysperyear
   218)  
   219) !       read 2 time sample bracketing ncdate
   220) 
   221)         if ( caldayloc > xin%cdayfm .and. caldayloc <= xin%cdayfp ) then
   222)  
   223)            strt3(3) = n
   224)            call handle_ncerr( nf90_get_var ( xin%ncid_f, xin%fluxid, xin%xvar(:,:,xin%nm_f), strt3, cnt3), &
   225)              'co2_data_flux.F90:235')
   226)            strt3(3) = xin%np1_f                                      
   227)            call handle_ncerr( nf90_get_var ( xin%ncid_f, xin%fluxid, xin%xvar(:,:,xin%np_f), strt3, cnt3),&
   228)              'co2_data_flux.F90:238')
   229) 
   230)            goto 10
   231) 
   232)         end if
   233)  
   234)      end do
   235) 
   236)      write(iulog,*)'CO2FLUX_READ: Failed to find dates bracketing ncdate, ncsec=', ncdate, ncsec
   237)      call endrun
   238)  
   239) 10   continue
   240)      write(iulog,*)'CO2FLUX_READ: Read ', trim(xin%locfn), ' for dates ', xin%date_f(n), xin%sec_f(n), &
   241)           ' and ', xin%date_f(xin%np1_f), xin%sec_f(xin%np1_f)
   242)  
   243) #if (defined SPMD )
   244)      call mpibcast( xin%timesz, 1,     mpiint, 0, mpicom )
   245)      call mpibcast( xin%date_f, xin%timesz, mpiint, 0, mpicom )
   246)      call mpibcast( xin%sec_f,  xin%timesz, mpiint, 0, mpicom )
   247)      call mpibcast( xin%cdayfm, 1,     mpir8 , 0, mpicom )
   248)      call mpibcast( xin%cdayfp, 1,     mpir8,  0, mpicom )
   249)      call mpibcast( xin%np1_f,  1,     mpiint, 0, mpicom )
   250)   else
   251)      call mpibcast( xin%timesz, 1,     mpiint, 0, mpicom )
   252)      allocate(xin%date_f(xin%timesz), xin%sec_f(xin%timesz))
   253)      call mpibcast( xin%date_f, xin%timesz, mpiint, 0, mpicom )
   254)      call mpibcast( xin%sec_f,  xin%timesz, mpiint, 0, mpicom )
   255)      call mpibcast( xin%cdayfm, 1,     mpir8 , 0, mpicom )
   256)      call mpibcast( xin%cdayfp, 1,     mpir8,  0, mpicom )
   257)      call mpibcast( xin%np1_f,  1,     mpiint, 0, mpicom )
   258) #endif
   259)   end if
   260) 
   261)   call scatter_field_to_chunk ( 1,1,2,cnt3(1), xin%xvar, xin%co2bdy )
   262) 
   263)   return
   264) end subroutine read_data_flux

   268) subroutine interp_time_flux (xin, prev_timestep)
   269)  
   270) !-----------------------------------------------------------------------
   271) ! Time interpolate data to current time.
   272) ! Reading in new monthly data if necessary.
   273) !
   274) !-----------------------------------------------------------------------
   275)  
   276)   use time_manager, only : get_curr_date, get_curr_calday, &
   277)                           is_perpetual, get_perp_date, get_step_size, is_first_step
   278)   use interpolate_data,   only : get_timeinterp_factors
   279)  
   280)   logical, intent(in), optional     :: prev_timestep ! If using previous timestep, set to true
   281)   TYPE(read_interp),  intent(inout) :: xin
   282) 
   283) !---------------------------Local variables-----------------------------
   284)   integer dtime          ! timestep size [seconds]
   285)   integer cnt3(3)        ! array of counts for each dimension
   286)   integer strt3(3)       ! array of starting indices
   287)   integer i,j,lchnk      ! indices
   288)   integer ncol           ! number of columns in current chunk
   289)   integer ntmp           ! temporary
   290)   real(r8) fact1, fact2  ! time interpolation factors
   291)   integer :: yr, mon, day! components of a date
   292)   integer :: ncdate      ! current date in integer format [yyyymmdd]
   293)   integer :: ncsec       ! current time of day [seconds]
   294)   real(r8) :: calday     ! current calendar day
   295)   real(r8) caldayloc     ! calendar day (includes yr if no cycling)
   296)   real(r8) deltat        ! time (days) between interpolating data
   297)   logical :: previous
   298)   logical :: co2cyc=.false.
   299)  
   300) !-----------------------------------------------------------------------
   301)  
   302) ! SPMD: Master does all the work.  Sends needed info to slaves
   303)  
   304) ! Use year information only if a multiyear dataset
   305)  
   306)      if ( .not. present(prev_timestep) ) then
   307)         previous = .false.
   308)      else
   309)         previous = prev_timestep
   310)      end if
   311)  
   312)      if (previous .and. is_first_step()) then
   313)         dtime = get_step_size()
   314)         dtime = -dtime
   315)         calday = get_curr_calday(offset=dtime)
   316)      else
   317)         calday = get_curr_calday()
   318)      endif
   319)  
   320)      if ( is_perpetual() ) then
   321)         call get_perp_date(yr, mon, day, ncsec)
   322)      else
   323)         if (previous .and. is_first_step()) then
   324)            call get_curr_date(yr, mon, day, ncsec,offset=dtime)
   325)         else
   326)            call get_curr_date(yr, mon, day, ncsec)
   327)         endif
   328)      end if
   329)  
   330)      ncdate = yr*10000 + mon*100 + day
   331)  
   332)      caldayloc = calday + yr*daysperyear
   333) 
   334)      if (masterproc) then
   335) 
   336)         strt3(1) = 1
   337)         strt3(2) = 1
   338)         strt3(3) = 1
   339)         cnt3(1)  = lonsiz
   340)         cnt3(2)  = latsiz
   341)         cnt3(3)  = 1
   342) 
   343)      endif
   344) 
   345) #ifdef SPMD
   346)      call mpibcast(cnt3, 2, mpiint, 0, mpicom)
   347) #endif
   348)  
   349) ! If model time is past current forward data timeslice, read in the next
   350) ! timeslice for time interpolation. 
   351)  
   352)      if ( caldayloc > xin%cdayfp .and. .not. (xin%np1_f==1 .and. caldayloc > xin%cdayfm) ) then
   353)  
   354)         xin%np1_f = xin%np1_f + 1
   355)  
   356)         if ( xin%np1_f > xin%timesz ) then
   357)            call endrun ('CO2FLUX_INTERP: Attempt to read past end of dataset')
   358)         end if
   359)  
   360)         xin%cdayfm = xin%cdayfp
   361)  
   362)         call bnddyi( xin%date_f(xin%np1_f), xin%sec_f(xin%np1_f), xin%cdayfp )
   363) 
   364)         yr = xin%date_f(xin%np1_f)/10000
   365)         xin%cdayfp = xin%cdayfp + yr*daysperyear
   366) 
   367)         if ( .not. (xin%np1_f == 1 .or. caldayloc <= xin%cdayfp) ) then
   368)          
   369)            if (masterproc) then
   370)               write(iulog,*)'CO2FLUX_INTERP: Input data for date', xin%date_f(xin%np1_f), ' sec ', xin%sec_f(xin%np1_f), &
   371)                         ' does not exceed model date', ncdate, ' sec ', ncsec, ' Stopping.'
   372)            end if
   373)            call endrun ()
   374)         end if
   375) 
   376)         ntmp     = xin%nm_f
   377)         xin%nm_f = xin%np_f
   378)         xin%np_f = ntmp
   379)  
   380)         if (masterproc) then
   381)            strt3(3) = xin%np1_f
   382)            call handle_ncerr( nf90_get_var ( xin%ncid_f, xin%fluxid, xin%xvar(:,:,xin%np_f), strt3, cnt3 ),&
   383)              'co2_data_flux.F90:391')
   384)            write(iulog,*)'CO2FLUX_INTERP: Read ', trim(xin%locfn),' for date (yyyymmdd) ', xin%date_f(xin%np1_f), &
   385)                      ' sec ', xin%sec_f(xin%np1_f)
   386)         endif
   387)  
   388)         call scatter_field_to_chunk ( 1,1,2,cnt3(1), xin%xvar, xin%co2bdy )
   389)      end if
   390)  
   391) ! Determine time interpolation factors.
   392)  
   393)      call get_timeinterp_factors ( co2cyc, xin%np1_f, xin%cdayfm, xin%cdayfp, caldayloc, fact1, fact2, 'CO2FLUX_INTERP:' )
   394) 
   395)      do lchnk=begchunk,endchunk
   396)         ncol = get_ncols_p(lchnk)
   397)         do i=1,ncol
   398)            xin%co2flx(i,lchnk) = xin%co2bdy(i,lchnk,xin%nm_f)*fact1 + xin%co2bdy(i,lchnk,xin%np_f)*fact2
   399)         end do
   400)      end do
   401) 
   402)   return
   403) end subroutine interp_time_flux
