    21)   real (kind=real_kind),parameter, public   :: bignum = 1.0D20

    44) !                                 departure cell                                    !
    45) !         jall...number of intersections of an element                              !

    50)   use coordinate_systems_mod,  only :  cartesian2D_t, spherical_polar_t, &
    51)                                        cart2cubedspherexy, spherical_to_cart
    52)   use physical_constants, only : DD_PI
    53)   use control_mod, only : north, south, east, west, neast, nwest, seast, swest
    54) 
    55)   implicit none
    56)   type (fvm_struct), intent(inout)                                :: fvm
    57)   integer (kind=int_kind), intent(in)                            :: nreconstruction
    58)   ! arrays for collecting cell data
    59)   real (kind=real_kind),dimension(10*(nc+2*nhe)*(nc+2*nhe),nreconstruction), &

    61)   integer (kind=int_kind), dimension(10*(nc+2*nhe)*(nc+2*nhe),2), &
    62)                                                 intent(out) :: weights_eul_index_all
    63)   integer (kind=int_kind), dimension(10*(nc+2*nhe)*(nc+2*nhe),2), &

    65)   integer (kind=int_kind), intent(in)                       :: klev
    66)   integer (kind=int_kind), intent(out)                      :: jall
    67)   logical, intent(in)                                     :: debug
    68)   ! local workspace
    69)   ! max number of line segments is:
    70)   ! (number of longitudes)*(max average number of crossings per line segment = 3)*ncube*2
    71)   !
    72) 
    73)   integer (kind=int_kind)                     :: jx,jy
    74)   integer                                     :: jx_min, jx_max, jy_min, jy_max
    75)   integer                                     :: jx_min1, jx_max1, jy_min1, jy_max1
    76)   integer                                     :: jx_min2, jx_max2, jy_min2, jy_max2

    79)   integer (kind=int_kind)                     :: i, jtmp
    80)   
    81)   type (cartesian2D_t)                        :: dcart(nc+1,nc+1)       ! Cartesian coordinates

    83)   real (kind=real_kind), dimension(0:5)       :: xcell,ycell
    84)   real (kind=real_kind), dimension(0:5)       :: xcell2,ycell2

    86)   integer (kind=int_kind)                     :: fluxsign1, fluxsign2

    88)   
    89)   integer (kind=int_kind)                     :: inttmp

    92)   ! for Gaussian quadrature

    94)   ! weight-variables for individual cells
    95)   integer (kind=int_kind) :: jmax_segments_cell
    96)   real (kind=real_kind)   , dimension(nhe*50,nreconstruction)   :: weights_cell
    97)   integer (kind=int_kind),  dimension(nhe*50,2)                 :: weights_eul_index_cell
    98)   integer (kind=int_kind)                                       :: jcollect_cell
    99) 
   100) 
   101)   jx_min=fvm%jx_min; jx_max=fvm%jx_max; 
   102)   jy_min=fvm%jy_min; jy_max=fvm%jy_max;
   103)   if (fvm%cubeboundary > 0) then
   104)     jx_min1=fvm%jx_min1; jx_max1=fvm%jx_max1; 
   105)     jy_min1=fvm%jy_min1; jy_max1=fvm%jy_max1;
   106)     swap1=fvm%swap1
   107)     if (fvm%cubeboundary > 4) then
   108)       jx_min2=fvm%jx_min2; jx_max2=fvm%jx_max2;
   109)       jy_min2=fvm%jy_min2; jy_max2=fvm%jy_max2;
   110)       swap2=fvm%swap2
   111)     endif
   112)   endif
   113) 
   114)   jmax_segments_cell = nhe*50
   115)   call gauss_points(ngpc,gsweights,gspts)
   116)   tmp =0.0D0
   117)   jall=1
   118)   ! 
   119)   ! FYI: THESE is CACULATED twice in (xlfux and yflux)
   120)   ! calculate xy Cartesian on the cube of departure points on the corresponding face  
   121)   do jy=1,nc+1
   122)      do jx=1,nc+1               
   123)         call cart2cubedspherexy(spherical_to_cart(fvm%dsphere(jx,jy,klev)),&
   124)              fvm%faceno,dcart(jx,jy))              
   125)      end do
   126)   end do
   127)   
   128)   do jy=1, nc
   129)     do jx=0, nc
   130)       !call getdep_cellboundariesxyvec(xcell,ycell,jx,jy,dcart)    
   131)       ! xflux: 
   132)       xcell(1)=fvm%acartx(jx+1); ycell(1)=fvm%acarty(jy)
   133)       xcell(3)=fvm%acartx(jx+1); ycell(3)=fvm%acarty(jy+1)
   134)       xcell(2)=dcart(jx+1,jy+1)%x; ycell(2)=dcart(jx+1,jy+1)%y
   135)       xcell(0)=dcart(jx+1,jy)%x; ycell(0)=dcart(jx+1,jy)%y
   136)       
   137)      
   138)       
   139)       if ((debug) .and. (jx==2) .and. (jy==2)) then
   140)         write(*,*) 'eul', fvm%acartx(jx), fvm%acarty(jy) 
   141)         write(*,*) 'eul', fvm%acartx(jx), fvm%acarty(jy+1)
   142)         write(*,*) 'eul', fvm%acartx(jx+1), fvm%acarty(jy+1)
   143)         write(*,*) 'eul', fvm%acartx(jx+1), fvm%acarty(jy)
   144)         write(*,*)
   145)         write(*,*) 'dep', dcart(jx,jy)%x, dcart(jx,jy)%y
   146)         write(*,*) 'dep', dcart(jx,jy+1)%x, dcart(jx,jy+1)%y
   147)         write(*,*) 'dep', dcart(jx+1,jy+1)%x, dcart(jx+1,jy+1)%y
   148)         write(*,*) 'dep', dcart(jx+1,jy)%x, dcart(jx+1,jy)%y
   149)          debugon=.False.
   150)       else
   151)          debugon=.FALSE.
   152)       endif
   153)       
   154)       call makefluxarea(xcell,ycell,xcell2,ycell2,.FALSE.,twofluxareas,zeroflux,fluxsign1, fluxsign2)

   156)       if (.not. zeroflux) then
   157)         call compute_weights_cell(xcell,ycell,jx,jy,nreconstruction,&
   158)              fvm%acartx,fvm%acarty,jx_min, jx_max, jy_min, jy_max, &
   159)              tmp,ngpc,gsweights,gspts,&
   160)              weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell) 
   161)            
   162)         if (jcollect_cell>0) then
   163)           weights_all(jall:jall+jcollect_cell-1,:) = fluxsign1*weights_cell(1:jcollect_cell,:)
   164)           weights_eul_index_all(jall:jall+jcollect_cell-1,:) = &
   165)                                         weights_eul_index_cell(1:jcollect_cell,:)

   167)           weights_lgr_index_all(jall:jall+jcollect_cell-1,2) = jy
   168)           jall = jall+jcollect_cell          
   169)         endif
   170)         
   171)         ! only if the flow gives us hourglass flux area
   172)         if (twofluxareas) then
   173)           call compute_weights_cell(xcell2,ycell2,jx,jy,nreconstruction,&
   174)                fvm%acartx,fvm%acarty,jx_min, jx_max, jy_min, jy_max, &
   175)                tmp,ngpc,gsweights,gspts,&
   176)                weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell) 
   177)

   179)             weights_all(jall:jall+jcollect_cell-1,:) = fluxsign2*weights_cell(1:jcollect_cell,:)
   180)             weights_eul_index_all(jall:jall+jcollect_cell-1,:) = &
   181)                                           weights_eul_index_cell(1:jcollect_cell,:)
   182)             weights_lgr_index_all(jall:jall+jcollect_cell-1,1) = jx
   183)             weights_lgr_index_all(jall:jall+jcollect_cell-1,2) = jy
   184)             jall = jall+jcollect_cell          
   185)           endif
   186)         endif
   187)       end if  ! end zeroflux
   188)       
   189)     end do
   190)   end do
   191) 
   192)   !WEST SIDE
   193)   if (fvm%cubeboundary == west) then
   194)     jx=1
   195)     ! calculate xy Cartesian on the cube of departure points on the corresponding face  
   196)     do jy=1,nc+1
   197)       call cart2cubedspherexy(spherical_to_cart(fvm%dsphere(jx,jy,klev)),&
   198)            fvm%nbrsface(west),dcart(jx,jy))

   200)            fvm%nbrsface(west),dcart(jx+1,jy))                 
   201)     end do
   202)     do jy=1,nc
   203) !       call getdep_cellboundaries(xcell,ycell,jx,jy,fvm%nbrsface(west),fvm%dsphere) 
   204)       call getdep_cellboundariesxyvec(xcell,ycell,jx,jy,dcart)     
   205)           
   206)       if(swap1) then  !flip orientation
   207)         call compute_weights_cell(xcell,ycell,jy_min1,jx_min1,nreconstruction,&
   208)            fvm%acarty1,fvm%acartx1,jy_min1, jy_max1, jx_min1, jx_max1, &
   209)            tmp,ngpc,gsweights,gspts,&
   210)            weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell)
   211)         do i=1,jcollect_cell
   212)           inttmp=weights_eul_index_cell(i,1)
   213)           weights_eul_index_cell(i,1)=weights_eul_index_cell(i,2)
   214)           weights_eul_index_cell(i,2)=inttmp
   215)         end do
   216)       else  
   217)         call compute_weights_cell(xcell,ycell,jx,jy,nreconstruction,&
   218)            fvm%acartx1,fvm%acarty1,jx_min1, jx_max1, jy_min1, jy_max1, &
   219)            tmp,ngpc,gsweights,gspts,&
   220)            weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell)
   221)       end if
   222)       !I have to correct the number
   223)       if (fvm%faceno==5) then
   224)         do i=1,jcollect_cell
   225)           weights_eul_index_cell(i,1)=weights_eul_index_cell(i,1)+nhe-1
   226)         end do
   227)       end if
   228)       if (fvm%faceno==6) then
   229)           do i=1,jcollect_cell
   230)             weights_eul_index_cell(i,2)=jy_max1-jy_min1-weights_eul_index_cell(i,2)-nhe-nhe+1
   231)           end do
   232)         end if
   233)       if (jcollect_cell>0) then
   234)         weights_all(jall:jall+jcollect_cell-1,:) = weights_cell(1:jcollect_cell,:)
   235)         weights_eul_index_all(jall:jall+jcollect_cell-1,:) = &
   236)                                          weights_eul_index_cell(1:jcollect_cell,:)
   237)         weights_lgr_index_all(jall:jall+jcollect_cell-1,1) = jx
   238)         weights_lgr_index_all(jall:jall+jcollect_cell-1,2) = jy
   239)         jall = jall+jcollect_cell          
   240)       endif
   241)     end do
   242)   endif
   243)   !EAST SIDE
   244)   if (fvm%cubeboundary == east) then
   245)     jx=nc
   246)     ! calculate xy Cartesian on the cube of departure points on the corresponding face  
   247)     do jy=1,nc+1
   248)       call cart2cubedspherexy(spherical_to_cart(fvm%dsphere(jx,jy,klev)),&
   249)            fvm%nbrsface(east),dcart(jx,jy))  
   250)       call cart2cubedspherexy(spherical_to_cart(fvm%dsphere(jx+1,jy,klev)),&
   251)            fvm%nbrsface(east),dcart(jx+1,jy))                 
   252)     end do
   253)     do jy=1,nc
   254)       call getdep_cellboundariesxyvec(xcell,ycell,jx,jy,dcart)     
   255)       if(swap1) then !flip orientation
   256)         call compute_weights_cell(xcell,ycell,jy_min1,jx_min1,nreconstruction,&
   257)            fvm%acarty1,fvm%acartx1,jy_min1, jy_max1, jx_min1, jx_max1, &
   258)            tmp,ngpc,gsweights,gspts,&
   259)            weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell)
   260)         do i=1,jcollect_cell
   261)           inttmp=weights_eul_index_cell(i,1)
   262)           weights_eul_index_cell(i,1)=weights_eul_index_cell(i,2)
   263)           weights_eul_index_cell(i,2)=inttmp
   264)         end do
   265)       else
   266)         call compute_weights_cell(xcell,ycell,jx,jy,nreconstruction,&
   267)            fvm%acartx1,fvm%acarty1,jx_min1, jx_max1, jy_min1, jy_max1, &
   268)            tmp,ngpc,gsweights,gspts,&
   269)            weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell)
   270)       end if
   271)       !I have to correct the number
   272)       if (fvm%faceno==5) then
   273)         do i=1,jcollect_cell
   274)           weights_eul_index_cell(i,2)=jy_max1-jy_min1-weights_eul_index_cell(i,2)-nhe-nhe+1
   275)         end do
   276)       end if
   277)       if (fvm%faceno==6) then
   278)         do i=1,jcollect_cell
   279)           weights_eul_index_cell(i,1)=weights_eul_index_cell(i,1)-nhe+1
   280)         end do
   281)       end if
   282)       if (jcollect_cell>0) then
   283)         weights_all(jall:jall+jcollect_cell-1,:) = weights_cell(1:jcollect_cell,:)
   284)         weights_eul_index_all(jall:jall+jcollect_cell-1,:) = &
   285)                                          weights_eul_index_cell(1:jcollect_cell,:)
   286)         weights_lgr_index_all(jall:jall+jcollect_cell-1,1) = jx
   287)         weights_lgr_index_all(jall:jall+jcollect_cell-1,2) = jy
   288)         jall = jall+jcollect_cell          
   289)       endif
   290)     end do
   291)   endif  
   292)   !NORTH SIDE 
   293)   if (fvm%cubeboundary == north) then
   294)     jy=nc
   295)     ! calculate xy Cartesian on the cube of departure points on the corresponding face  
   296)     do jx=1,nc+1
   297)       call cart2cubedspherexy(spherical_to_cart(fvm%dsphere(jx,jy,klev)),&
   298)            fvm%nbrsface(north),dcart(jx,jy))  
   299)       call cart2cubedspherexy(spherical_to_cart(fvm%dsphere(jx,jy+1,klev)),&
   300)            fvm%nbrsface(north),dcart(jx,jy+1))                 
   301)     end do
   302)     do jx=1,nc
   303)       call getdep_cellboundariesxyvec(xcell,ycell,jx,jy,dcart)     
   304)       if(swap1) then !flip orientation
   305)         call compute_weights_cell(xcell,ycell,jy_min1,jx_min1,nreconstruction,&
   306)            fvm%acarty1,fvm%acartx1,jy_min1, jy_max1, jx_min1, jx_max1, &
   307)            tmp,ngpc,gsweights,gspts,&
   308)            weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell)
   309)         do i=1,jcollect_cell
   310)           inttmp=weights_eul_index_cell(i,1)
   311)           weights_eul_index_cell(i,1)=weights_eul_index_cell(i,2)
   312)           weights_eul_index_cell(i,2)=inttmp
   313)         end do
   314)       else
   315)         call compute_weights_cell(xcell,ycell,jx,jy,nreconstruction,&
   316)            fvm%acartx1,fvm%acarty1,jx_min1, jx_max1, jy_min1, jy_max1, &

   318)            weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell)

   321)       if (fvm%faceno==2) then
   322)         do i=1,jcollect_cell
   323)           weights_eul_index_cell(i,2)=weights_eul_index_cell(i,2)-nhe+1

   330)         end do
   331)       end if
   332)       if (fvm%faceno==4) then
   333)         do i=1,jcollect_cell
   334)           weights_eul_index_cell(i,1)=(jx_max1-jx_min1)-weights_eul_index_cell(i,1)-nhe-nhe+1
   335)         end do
   336)       end if
   337)       if (fvm%faceno==6) then
   338)         do i=1,jcollect_cell
   339)           weights_eul_index_cell(i,1)=(jx_max1-jx_min1)-weights_eul_index_cell(i,1)-nhe-nhe+1
   340)           weights_eul_index_cell(i,2)=weights_eul_index_cell(i,2)-nhe+1
   341)         end do
   342)       end if
   343)       if (jcollect_cell>0) then
   344)         weights_all(jall:jall+jcollect_cell-1,:) = weights_cell(1:jcollect_cell,:)
   345)         weights_eul_index_all(jall:jall+jcollect_cell-1,:) = &
   346)                                          weights_eul_index_cell(1:jcollect_cell,:)
   347)         weights_lgr_index_all(jall:jall+jcollect_cell-1,1) = jx
   348)         weights_lgr_index_all(jall:jall+jcollect_cell-1,2) = jy
   349)         jall = jall+jcollect_cell          
   350)       endif
   351)     end do
   352)   end if
   353) 
   354)   !SOUTH SIDE
   355)   if (fvm%cubeboundary == south) then
   356)     jy=1
   357)     ! calculate xy Cartesian on the cube of departure points on the corresponding face  
   358)     do jx=1,nc+1
   359)       call cart2cubedspherexy(spherical_to_cart(fvm%dsphere(jx,jy,klev)),&

   361)       call cart2cubedspherexy(spherical_to_cart(fvm%dsphere(jx,jy+1,klev)),&
   362)            fvm%nbrsface(south),dcart(jx,jy+1))                 
   363)     end do
   364)     do jx=1,nc
   365)       call getdep_cellboundariesxyvec(xcell,ycell,jx,jy,dcart)      
   366)       if(swap1) then !flip orientation
   367)         call compute_weights_cell(xcell,ycell,jy_min1,jx_min1,nreconstruction,&
   368)            fvm%acarty1,fvm%acartx1,jy_min1, jy_max1, jx_min1, jx_max1, &
   369)            tmp,ngpc,gsweights,gspts,&
   370)            weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell)
   371)         do i=1,jcollect_cell
   372)           inttmp=weights_eul_index_cell(i,1)
   373)           weights_eul_index_cell(i,1)=weights_eul_index_cell(i,2)
   374)           weights_eul_index_cell(i,2)=inttmp
   375)         end do
   376)       else
   377)         call compute_weights_cell(xcell,ycell,jx,jy,nreconstruction,&
   378)            fvm%acartx1,fvm%acarty1,jx_min1, jx_max1, jy_min1, jy_max1, &
   379)            tmp,ngpc,gsweights,gspts,&
   380)            weights_cell,weights_eul_index_cell,jcollect_cell,jmax_segments_cell)  
   381)       end if
   382)       !I have to correct the number   
   383)       if  (fvm%faceno==2) then
   384)         do i=1,jcollect_cell
   385)           weights_eul_index_cell(i,1)=(jx_max1-jx_min1)-(weights_eul_index_cell(i,1)+nhe)-nhe+1
   386)         end do

  1400) subroutine translate_then_rotate(x_in,y_in,x,y,imin,imax,xorigo,yorigo)
  1401)   !
  1402)   ! translates x_in and y_in to (x_in(3),y_in(3)) and rotates data -90 degrees about xorigo,yorigo
  1403)   !
  1404)   IMPLICIT NONE
  1405)   integer (kind=int_kind)                    , intent(in)  :: imin,imax
  1406)   real (kind=real_kind), dimension(imin:imax), intent(in)  :: x_in,y_in
  1407)   real (kind=real_kind)                                    :: xorigo, yorigo
  1408)   real (kind=real_kind), dimension(imin:imax), intent(out) :: x,y
  1409)   !
  1410)   !
  1411)   !
  1412)   real (kind=real_kind), dimension(imin:imax) :: xtmp,ytmp
  1413)   integer (kind=int_kind) :: i
  1414)   !
  1415)   !
  1416)   ! move origo to xorigo,yorigo
  1417)   !
  1418)   do i=imin,imax
  1419)      xtmp(i) = x_in(i)-xorigo
  1420)      ytmp(i) = y_in(i)-yorigo
  1421)   end do
  1422)   !
  1423)   ! rotate +90 degrees
  1424)   !
  1425)   x =  ytmp
  1426)   y = -xtmp
  1427) end subroutine translate_then_rotate
  1428) 
  1429) subroutine rotate_then_translate(x_in,y_in,x,y,imin,imax,xorigo,yorigo)
  1430)   !
  1431)   ! reverse of translate_then_rotate subroutine
  1432)   !
  1433)   IMPLICIT NONE
  1434)   integer (kind=int_kind)                    , intent(in)  :: imin,imax
  1435)   real (kind=real_kind), dimension(imin:imax), intent(in)  :: x_in,y_in
  1436)   real (kind=real_kind)                                    :: xorigo, yorigo
  1437)   real (kind=real_kind), dimension(imin:imax), intent(out) :: x,y
  1438)   !
  1439)   !
  1440)   !
  1441)   real (kind=real_kind), dimension(imin:imax) :: xtmp,ytmp
  1442)   integer (kind=int_kind) :: i
  1443)   !
  1444)   !
  1445)   ! rotate -90 degrees
  1446)   !
  1447)   xtmp =  -y_in
  1448)   ytmp =   x_in
  1449)   !
  1450)   ! move origo
  1451)   !
  1452)   do i=imin,imax
  1453)      x(i) = xtmp(i)+xorigo
  1454)      y(i) = ytmp(i)+yorigo
  1455)   end do
  1456) end subroutine rotate_then_translate
