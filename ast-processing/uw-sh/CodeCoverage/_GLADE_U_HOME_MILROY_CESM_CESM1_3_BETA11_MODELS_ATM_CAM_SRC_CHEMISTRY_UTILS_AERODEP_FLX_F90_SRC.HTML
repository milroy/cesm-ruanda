   111) ! parses the list of dep fluxes specified in aerodep_flx_specifier namelist
   112) ! variable and sets up index variables

   114)   subroutine aerodep_flx_init()
   115)     
   116)     use tracer_data, only : trcdata_init
   117)     use cam_history, only : addfld, phys_decomp
   118)     use physics_buffer, only : physics_buffer_desc
   119)     use modal_aero_deposition, only : modal_aero_deposition_init
   120)     
   121)     implicit none
   122) 
   123)     integer :: ndx, istat, i
   124) 
   125)     if ( has_aerodep_flx ) then
   126)        if ( masterproc ) then
   127)           write(iulog,*) 'aero dep fluxes are prescribed in :'//trim(filename)
   128)        endif
   129)     else
   130)        return
   131)     endif
   132) 
   133)     allocate(file%in_pbuf(size(specifier)))
   134)     file%in_pbuf(:) = .false.
   135)     call trcdata_init( specifier, filename, filelist, datapath, fields, file, &
   136)                        rmv_file, cycle_yr, fixed_ymd, fixed_tod, datatype)
   137) 
   138)     number_flds = 0
   139)     if (associated(fields)) number_flds = size( fields )
   140) 
   141)     if( number_flds < 1 ) then
   142)        has_aerodep_flx = .false.
   143)        if (masterproc) then
   144)           write(iulog,*) 'aerodep_flx_init: no aerosol deposition fluxes have been specified'
   145)        endif
   146)        return
   147)     end if
   148) 
   149)     index_bulk_map(:) = -1
   150)     index_modal_map(:) = -1
   151) 
   152)     do i = 1,number_flds
   153) 
   154)        ndx = get_ndx( fields(i)%fldnam, bulk_names )
   155)        if (ndx >0) then
   156)           index_bulk_map(ndx) = i
   157)        else
   158)           ndx = get_ndx( fields(i)%fldnam, modal_names )
   159)           if (ndx >0) then
   160)              index_modal_map(ndx) = i
   161)           endif
   162)        endif
   163)        if (ndx>0) then
   164)           call addfld(trim(fields(i)%fldnam)//'_D',fields(i)%units, 1, 'A', 'prescribed aero dep', phys_decomp )
   165)        else
   166)           call endrun('aerodep_flx_init: aerosol flux name not recognized: '//trim(fields(i)%fldnam))
   167)        endif
   168)     enddo
   169) 
   170)     modal_fluxes = any(index_modal_map(:)>0)
   171) 
   172)     if (modal_fluxes) then
   173) 
   174)        idx_bc1_dryis  = index_modal_map(1)
   175)        idx_bc1_drycw  = index_modal_map(2)
   176)        idx_pom1_dryis = index_modal_map(3)
   177)        idx_pom1_drycw = index_modal_map(4)
   178)        idx_soa1_dryis = index_modal_map(5)
   179)        idx_soa1_drycw = index_modal_map(6)
   180)        idx_soa2_dryis = index_modal_map(7)
   181)        idx_soa2_drycw = index_modal_map(8)
   182)        idx_dst1_dryis = index_modal_map(9)
   183)        idx_dst1_drycw = index_modal_map(10)
   184)        idx_dst3_dryis = index_modal_map(11)
   185)        idx_dst3_drycw = index_modal_map(12)
   186) 
   187)        idx_bc1_wetis  = index_modal_map(13)
   188)        idx_bc1_wetcw  = index_modal_map(14)
   189)        idx_pom1_wetis = index_modal_map(15)
   190)        idx_pom1_wetcw = index_modal_map(16)
   191)        idx_soa1_wetis = index_modal_map(17)
   192)        idx_soa1_wetcw = index_modal_map(18)
   193)        idx_dst1_wetis = index_modal_map(19)
   194)        idx_dst1_wetcw = index_modal_map(20)
   195)        idx_dst3_wetis = index_modal_map(21)
   196)        idx_dst3_wetcw = index_modal_map(22)
   197) 
   198)        call modal_aero_deposition_init( bc1_ndx=idx_bc1,   pom1_ndx=idx_pom1, soa1_ndx=idx_soa1, &
   199)                                         soa2_ndx=idx_soa2, dst1_ndx=idx_dst1, dst3_ndx=idx_dst3, &
   200)                                         ncl3_ndx=idx_ncl3, so43_ndx=idx_so43 )
   201)     else
   202) 
   203)        ibcphiwet = index_bulk_map(1)
   204)        ibcphodry = index_bulk_map(2)
   205)        ibcphidry = index_bulk_map(3)
   206)        iocphiwet = index_bulk_map(4)
   207)        iocphodry = index_bulk_map(5)
   208)        iocphidry = index_bulk_map(6)
   209)        idstdry1  = index_bulk_map(7)
   210)        idstdry2  = index_bulk_map(8)
   211)        idstdry3  = index_bulk_map(9)
   212)        idstdry4  = index_bulk_map(10)
   213)        idstwet1  = index_bulk_map(11)
   214)        idstwet2  = index_bulk_map(12)
   215)        idstwet3  = index_bulk_map(13)
   216)        idstwet4  = index_bulk_map(14)
   217) 
   218)     endif
   219) 
   220)   end subroutine aerodep_flx_init

   225) subroutine aerodep_flx_readnl(nlfile)
   226) 
   227)    use namelist_utils,  only: find_group_name
   228)    use units,           only: getunit, freeunit
   229)    use mpishorthand
   230) 
   231)    character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
   232) 
   233)    ! Local variables
   234)    integer :: unitn, ierr
   235)    character(len=*), parameter :: subname = 'aerodep_flx_readnl'
   236) 
   237)    character(len=32)  :: aerodep_flx_specifier(N_MODAL)
   238)    character(len=256) :: aerodep_flx_file
   239)    character(len=256) :: aerodep_flx_filelist
   240)    character(len=256) :: aerodep_flx_datapath
   241)    character(len=32)  :: aerodep_flx_type
   242)    logical            :: aerodep_flx_rmfile
   243)    integer            :: aerodep_flx_cycle_yr
   244)    integer            :: aerodep_flx_fixed_ymd
   245)    integer            :: aerodep_flx_fixed_tod
   246) 
   247)    namelist /aerodep_flx_nl/ &
   248)       aerodep_flx_specifier, &
   249)       aerodep_flx_file,      &
   250)       aerodep_flx_filelist,  &
   251)       aerodep_flx_datapath,  &
   252)       aerodep_flx_type,      &
   253)       aerodep_flx_rmfile,    &
   254)       aerodep_flx_cycle_yr,  &
   255)       aerodep_flx_fixed_ymd, &
   256)       aerodep_flx_fixed_tod      
   257)    !-----------------------------------------------------------------------------
   258) 
   259)    ! Initialize namelist variables from local module variables.
   260)    aerodep_flx_specifier= specifier
   261)    aerodep_flx_file     = filename
   262)    aerodep_flx_filelist = filelist
   263)    aerodep_flx_datapath = datapath
   264)    aerodep_flx_type     = datatype
   265)    aerodep_flx_rmfile   = rmv_file
   266)    aerodep_flx_cycle_yr = cycle_yr
   267)    aerodep_flx_fixed_ymd= fixed_ymd
   268)    aerodep_flx_fixed_tod= fixed_tod
   269) 
   270)    ! Read namelist
   271)    if (masterproc) then
   272)       unitn = getunit()
   273)       open( unitn, file=trim(nlfile), status='old' )
   274)       call find_group_name(unitn, 'aerodep_flx_nl', status=ierr)
   275)       if (ierr == 0) then
   276)          read(unitn, aerodep_flx_nl, iostat=ierr)
   277)          if (ierr /= 0) then
   278)             call endrun(subname // ':: ERROR reading namelist')
   279)          end if
   280)       end if
   281)       close(unitn)
   282)       call freeunit(unitn)
   283)    end if
   284) 
   285) #ifdef SPMD
   286)    ! Broadcast namelist variables
   287)    call mpibcast(aerodep_flx_specifier,len(aerodep_flx_specifier(1))*N_MODAL,     mpichar, 0, mpicom)
   288)    call mpibcast(aerodep_flx_file,     len(aerodep_flx_file),     mpichar, 0, mpicom)
   289)    call mpibcast(aerodep_flx_filelist, len(aerodep_flx_filelist), mpichar, 0, mpicom)
   290)    call mpibcast(aerodep_flx_datapath, len(aerodep_flx_datapath), mpichar, 0, mpicom)
   291)    call mpibcast(aerodep_flx_type,     len(aerodep_flx_type),     mpichar, 0, mpicom)
   292)    call mpibcast(aerodep_flx_rmfile,   1, mpilog,  0, mpicom)
   293)    call mpibcast(aerodep_flx_cycle_yr, 1, mpiint,  0, mpicom)
   294)    call mpibcast(aerodep_flx_fixed_ymd,1, mpiint,  0, mpicom)
   295)    call mpibcast(aerodep_flx_fixed_tod,1, mpiint,  0, mpicom)
   296) #endif
   297) 
   298)    ! Update module variables with user settings.
   299)    specifier  = aerodep_flx_specifier
   300)    filename   = aerodep_flx_file
   301)    filelist   = aerodep_flx_filelist
   302)    datapath   = aerodep_flx_datapath
   303)    datatype   = aerodep_flx_type
   304)    rmv_file   = aerodep_flx_rmfile
   305)    cycle_yr   = aerodep_flx_cycle_yr
   306)    fixed_ymd  = aerodep_flx_fixed_ymd
   307)    fixed_tod  = aerodep_flx_fixed_tod
   308) 
   309)    ! Turn on prescribed volcanics if user has specified an input dataset.
   310)    if (len_trim(filename) > 0 ) has_aerodep_flx = .true.
   311) 
   312) end subroutine aerodep_flx_readnl

   334) !-------------------------------------------------------------------
   335) ! advances the prescribed fluxes to the current time step
   336) !-------------------------------------------------------------------
   337)   subroutine aerodep_flx_adv( state, pbuf2d, cam_out )
   338) 
   339)     use tracer_data,      only : advance_trcdata
   340)     use physics_types,    only : physics_state
   341)     use camsrfexch,       only : cam_out_t
   342)     use physics_buffer, only : physics_buffer_desc
   343) 
   344)     implicit none
   345) 
   346)     type(physics_state), intent(in)    :: state(begchunk:endchunk)                 
   347)     type(cam_out_t),     intent(inout) :: cam_out(begchunk:endchunk)
   348)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   349) 
   350)     integer :: c, ncol
   351)     
   352)     if( .not. has_aerodep_flx ) return

   354)     call advance_trcdata( fields, file, state, pbuf2d  )
   355) 
   356) !$OMP PARALLEL DO PRIVATE (C, NCOL)
   357)     do c = begchunk, endchunk
   358)        ncol = state(c)%ncol
   359)        call aerodep_flx_set( cam_out(c), ncol, c )
   360)     enddo
