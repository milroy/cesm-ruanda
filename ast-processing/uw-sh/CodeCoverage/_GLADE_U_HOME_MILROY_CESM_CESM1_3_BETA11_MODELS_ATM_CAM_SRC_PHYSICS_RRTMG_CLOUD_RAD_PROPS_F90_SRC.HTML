   275) subroutine cloud_rad_props_get_sw(state, pbuf, &
   276)                                   tau, tau_w, tau_w_g, tau_w_f,&
   277)                                   diagnosticindex, oldliq, oldice)
   278) 
   279) ! return totaled (across all species) layer tau, omega, g, f 
   280) ! for all spectral interval for aerosols affecting the climate
   281) 
   282)    ! Arguments
   283)    type(physics_state), intent(in)   :: state
   284)    type(physics_buffer_desc),pointer :: pbuf(:)
   285)    integer, optional,   intent(in) :: diagnosticindex      ! index (if present) to radiation diagnostic information
   286) 
   287)    real(r8), intent(out) :: tau    (nswbands,pcols,pver) ! aerosol extinction optical depth
   288)    real(r8), intent(out) :: tau_w  (nswbands,pcols,pver) ! aerosol single scattering albedo * tau
   289)    real(r8), intent(out) :: tau_w_g(nswbands,pcols,pver) ! aerosol assymetry parameter * tau * w
   290)    real(r8), intent(out) :: tau_w_f(nswbands,pcols,pver) ! aerosol forward scattered fraction * tau * w
   291) 
   292)    logical, optional, intent(in) :: oldliq,oldice
   293) 
   294)    ! Local variables
   295) 
   296)    integer :: ncol
   297)    integer :: lchnk
   298)    integer :: k, i    ! lev and daycolumn indices
   299)    integer :: iswband ! sw band indices
   300) 
   301)    ! optical props for each aerosol
   302)    real(r8), pointer :: h_ext(:,:)
   303)    real(r8), pointer :: h_ssa(:,:)
   304)    real(r8), pointer :: h_asm(:,:)
   305)    real(r8), pointer :: n_ext(:)
   306)    real(r8), pointer :: n_ssa(:)
   307)    real(r8), pointer :: n_asm(:)
   308) 
   309)    ! rad properties for liquid clouds
   310)    real(r8) :: liq_tau    (nswbands,pcols,pver) ! aerosol extinction optical depth
   311)    real(r8) :: liq_tau_w  (nswbands,pcols,pver) ! aerosol single scattering albedo * tau
   312)    real(r8) :: liq_tau_w_g(nswbands,pcols,pver) ! aerosol assymetry parameter * tau * w
   313)    real(r8) :: liq_tau_w_f(nswbands,pcols,pver) ! aerosol forward scattered fraction * tau * w
   314) 
   315)    ! rad properties for ice clouds
   316)    real(r8) :: ice_tau    (nswbands,pcols,pver) ! aerosol extinction optical depth
   317)    real(r8) :: ice_tau_w  (nswbands,pcols,pver) ! aerosol single scattering albedo * tau
   318)    real(r8) :: ice_tau_w_g(nswbands,pcols,pver) ! aerosol assymetry parameter * tau * w
   319)    real(r8) :: ice_tau_w_f(nswbands,pcols,pver) ! aerosol forward scattered fraction * tau * w
   320) 
   321)    !-----------------------------------------------------------------------------
   322) 
   323)    ncol  = state%ncol
   324)    lchnk = state%lchnk
   325) 
   326)    ! initialize to conditions that would cause failure
   327)    tau     (:,:,:) = -100._r8
   328)    tau_w   (:,:,:) = -100._r8
   329)    tau_w_g (:,:,:) = -100._r8
   330)    tau_w_f (:,:,:) = -100._r8
   331) 
   332)    ! initialize layers to accumulate od's
   333)    tau    (:,1:ncol,:) = 0._r8
   334)    tau_w  (:,1:ncol,:) = 0._r8
   335)    tau_w_g(:,1:ncol,:) = 0._r8
   336)    tau_w_f(:,1:ncol,:) = 0._r8
   337) 
   338) 
   339)    call get_liquid_optics_sw(state, pbuf, liq_tau, liq_tau_w, liq_tau_w_g, liq_tau_w_f)
   340) 
   341)    call get_ice_optics_sw   (state, pbuf, ice_tau, ice_tau_w, ice_tau_w_g, ice_tau_w_f)
   342) 
   343)    tau    (:,1:ncol,:) =  liq_tau    (:,1:ncol,:) + ice_tau    (:,1:ncol,:)
   344)    tau_w  (:,1:ncol,:) =  liq_tau_w  (:,1:ncol,:) + ice_tau_w  (:,1:ncol,:)
   345)    tau_w_g(:,1:ncol,:) =  liq_tau_w_g(:,1:ncol,:) + ice_tau_w_g(:,1:ncol,:)
   346)    tau_w_f(:,1:ncol,:) =  liq_tau_w_f(:,1:ncol,:) + ice_tau_w_f(:,1:ncol,:)
   347) 
   348) end subroutine cloud_rad_props_get_sw

   351) subroutine cloud_rad_props_get_lw(state, pbuf, cld_abs_od, diagnosticindex, oldliq, oldice, oldcloud)
   352) 
   353) ! Purpose: Compute cloud longwave absorption optical depth
   354) !    cloud_rad_props_get_lw() is called by radlw() 
   355) 
   356)    ! Arguments
   357)    type(physics_state), intent(in)  :: state
   358)    type(physics_buffer_desc),pointer:: pbuf(:)
   359)    real(r8),            intent(out) :: cld_abs_od(nlwbands,pcols,pver) ! [fraction] absorption optical depth, per layer
   360)    integer, optional,   intent(in)  :: diagnosticindex
   361)    logical, optional,   intent(in)  :: oldliq  ! use old liquid optics
   362)    logical, optional,   intent(in)  :: oldice  ! use old ice optics
   363)    logical, optional,   intent(in)  :: oldcloud  ! use old optics for both (b4b)
   364) 
   365)    ! Local variables
   366) 
   367)    integer :: bnd_idx     ! LW band index
   368)    integer :: i           ! column index
   369)    integer :: k           ! lev index
   370)    integer :: ncol        ! number of columns
   371)    integer :: lchnk
   372) 
   373)    ! rad properties for liquid clouds
   374)    real(r8) :: liq_tau_abs_od(nlwbands,pcols,pver) ! liquid cloud absorption optical depth
   375) 
   376)    ! rad properties for ice clouds
   377)    real(r8) :: ice_tau_abs_od(nlwbands,pcols,pver) ! ice cloud absorption optical depth
   378) 
   379)    !-----------------------------------------------------------------------------
   380) 
   381)    ncol = state%ncol
   382)    lchnk = state%lchnk
   383) 
   384)    ! compute optical depths cld_absod 
   385)    cld_abs_od = 0._r8
   386) 
   387)    if(present(oldcloud))then
   388)       if(oldcloud) then
   389)          ! make diagnostic calls to these first to output ice and liq OD's
   390)          !call old_liq_get_rad_props_lw(state, pbuf, liq_tau_abs_od, oldliqwp=.false.)
   391)          !call old_ice_get_rad_props_lw(state, pbuf, ice_tau_abs_od, oldicewp=.false.)
   392)          ! This affects climate (cld_abs_od)
   393)          call oldcloud_lw(state,pbuf,cld_abs_od,oldwp=.false.)
   394)          return
   395)       endif
   396)    endif
   397) 
   398)    if(present(oldliq))then
   399)       if(oldliq) then
   400)          call old_liq_get_rad_props_lw(state, pbuf, liq_tau_abs_od, oldliqwp=.false.)
   401)       else
   402)          call liquid_cloud_get_rad_props_lw(state, pbuf, liq_tau_abs_od)
   403)       endif
   404)    else
   405)       call liquid_cloud_get_rad_props_lw(state, pbuf, liq_tau_abs_od)
   406)    endif
   407) 
   408)    if(present(oldice))then
   409)       if(oldice) then
   410)          call old_ice_get_rad_props_lw(state, pbuf, ice_tau_abs_od, oldicewp=.false.)
   411)       else
   412)          call ice_cloud_get_rad_props_lw(state, pbuf, ice_tau_abs_od)
