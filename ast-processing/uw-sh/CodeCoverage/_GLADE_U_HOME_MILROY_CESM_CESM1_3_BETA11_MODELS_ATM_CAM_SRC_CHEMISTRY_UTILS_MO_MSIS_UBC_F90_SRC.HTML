   114) !--------------------------------------------------------------------
   115) !	... get the upper boundary values for h, n, o, o2 and temp

   139)       real(r8), parameter :: mass_switch = 48._r8

   146)       integer  ::  offset

   149)       real(r8) ::  msis_ap(7)
   150)       real(r8) ::  msis_temp(2)
   151)       real(r8) ::  msis_conc(9)
   152)       real(r8) ::  rlons(pcols)
   153)       real(r8) ::  rlats(pcols)
   154)       real(r8) ::  dnom(pcols)

   178) 	    write(iulog,*) '===================================='
   179) 	    write(iulog,*) 'msis_timestep_init: diagnostics'
   180) 	    write(iulog,*) 'nstep,yr,mon,day,tod,date,ut,doy,offset,msis_frq = ',nstep, yr, mon, day, tod, date, ut, doy, offset, msis_frq

   215) 	       write(iulog,*) '===================================='

   230)                   msis_ubc(:ncol,4,c) = cnst_mw(ndx_o)*msis_ubc(:ncol,4,c)*dnom(:ncol)
   231)                end if
   232)                if( ndx(ndx_o2) > 0 ) then
   233)                   msis_ubc(:ncol,5,c) = cnst_mw(ndx_o2)*msis_ubc(:ncol,5,c)*dnom(:ncol)
   234)                end if
   235)             end if
   236)          end do chunk_loop
   237)       end if msis_retrieval
