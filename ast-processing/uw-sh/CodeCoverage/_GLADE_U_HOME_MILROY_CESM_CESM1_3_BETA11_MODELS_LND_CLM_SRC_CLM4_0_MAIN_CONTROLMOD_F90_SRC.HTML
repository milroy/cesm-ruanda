   117) ! !IROUTINE: control_init
   118) !
   119) ! !INTERFACE:
   120)   subroutine control_init( )
   121) !
   122) ! !DESCRIPTION:
   123) ! Initialize CLM run control information
   124) !
   125) ! !USES:
   126)     use clm_time_manager , only : set_timemgr_init, get_timemgr_defaults
   127)     use fileutils        , only : getavu, relavu
   128)     use shr_string_mod   , only : shr_string_getParentDir
   129)     use clm_varctl       , only : clmvarctl_init, clm_varctl_set, nsrBranch, nsrStartup, &
   130)                                   nsrContinue
   131)     use clm_cpl_indices  , only : glc_nec
   132) 
   133)     implicit none
   134) !
   135) !
   136) ! !REVISION HISTORY:
   137) ! Created by Mariana Vertenstein
   138) !
   139) !
   140) ! !LOCAL VARIABLES:
   141) !EOP
   142)     character(len=32)  :: starttype ! infodata start type
   143)     integer :: i,j,n                ! loop indices
   144)     integer :: ierr                 ! error code
   145)     integer :: unitn                ! unit for namelist file
   146)     integer :: dtime                ! Integer time-step
   147)     integer :: override_nsrest      ! If want to override the startup type sent from driver
   148)     character(len=32) :: subname = 'control_init'  ! subroutine name
   149) !------------------------------------------------------------------------
   150) 
   151)     ! ----------------------------------------------------------------------
   152)     ! Namelist Variables
   153)     ! ----------------------------------------------------------------------
   154) 
   155)     ! Time step
   156)     namelist / clm_inparm/ &
   157) 	 dtime	
   158) 
   159)     ! CLM namelist settings
   160) 
   161)     namelist /clm_inparm / &
   162)          fatmlndfrc, finidat, nrevsn
   163) 
   164)     ! Input datasets
   165) 
   166)     namelist /clm_inparm/  &
   167)          fsurdat, fatmtopo, flndtopo, &
   168)          fpftcon, flanduse_timeseries,  fsnowoptics, fsnowaging
   169) 
   170)     ! History, restart options
   171) 
   172)     namelist /clm_inparm/  &
   173)          hist_empty_htapes, hist_dov2xy, &
   174)          hist_avgflag_pertape, hist_type1d_pertape, &
   175)          hist_nhtfrq,  hist_ndens, hist_mfilt, &
   176)          hist_fincl1,  hist_fincl2, hist_fincl3, &
   177)          hist_fincl4,  hist_fincl5, hist_fincl6, &
   178)          hist_fexcl1,  hist_fexcl2, hist_fexcl3, &
   179)          hist_fexcl4,  hist_fexcl5, hist_fexcl6
   180) 
   181)     ! BGC info
   182) 
   183)     namelist /clm_inparm/  &
   184)          suplnitro
   185) 
   186)     namelist /clm_inparm / use_c13, use_c14
   187) 
   188)     namelist /clm_inparm / &
   189)          co2_type
   190) 
   191)     ! Glacier_mec info
   192)     namelist /clm_inparm / &    
   193)          maxpatch_glcmec, glc_smb, glc_dyntopo, glc_grid, fglcmask 
   194) 
   195)     ! Other options
   196) 
   197)     namelist /clm_inparm/  &
   198)          clump_pproc, wrtdia, &
   199)          create_crop_landunit, nsegspc, co2_ppmv, override_nsrest, &
   200)          albice
   201)     ! Urban options
   202) 
   203)     namelist /clm_inparm/  &
   204)          urban_hac, urban_traffic
   205)          
   206)     ! ----------------------------------------------------------------------
   207)     ! Default values
   208)     ! ----------------------------------------------------------------------
   209) 
   210)     if (masterproc) then
   211)        write(iulog,*) 'Attempting to initialize run control settings .....'
   212)     endif
   213) 
   214)     runtyp(:)               = 'missing'
   215)     runtyp(nsrStartup  + 1) = 'initial'
   216)     runtyp(nsrContinue + 1) = 'restart'
   217)     runtyp(nsrBranch   + 1) = 'branch '
   218) 
   219)     ! Set clumps per procoessor
   220) 
   221) #if (defined _OPENMP)
   222)     clump_pproc = omp_get_max_threads()
   223) #else
   224)     clump_pproc = 1
   225) #endif
   226) 
   227)     override_nsrest = nsrest
   228) 
   229)     if (masterproc) then
   230) 
   231)        ! ----------------------------------------------------------------------
   232)        ! Read namelist from standard input. 
   233)        ! ----------------------------------------------------------------------
   234) 
   235)        if ( len_trim(NLFilename) == 0  )then
   236)           call endrun( subname//' error: nlfilename not set' )
   237)        end if
   238)        unitn = getavu()
   239)        write(iulog,*) 'Read in clm_inparm namelist from: ', trim(NLFilename)
   240)        open( unitn, file=trim(NLFilename), status='old' )
   241)        ierr = 1
   242)        do while ( ierr /= 0 )
   243)           read(unitn, clm_inparm, iostat=ierr)
   244)           if (ierr < 0) then
   245)              call endrun( subname//' encountered end-of-file on clm_inparm read' )
   246)           endif
   247)        end do
   248)        call relavu( unitn )
   249) 
   250)        ! ----------------------------------------------------------------------
   251)        ! Consistency checks on input namelist.
   252)        ! ----------------------------------------------------------------------
   253) 
   254)        call set_timemgr_init( dtime_in=dtime )
   255) 
   256)        if (urban_traffic) then
   257)           write(iulog,*)'Urban traffic fluxes are not implemented currently'
   258)           call endrun()
   259)        end if
   260) 
   261)        ! History and restart files
   262) 
   263)        do i = 1, max_tapes
   264)           if (hist_nhtfrq(i) == 0) then
   265)              hist_mfilt(i) = 1
   266)           else if (hist_nhtfrq(i) < 0) then
   267)              hist_nhtfrq(i) = nint(-hist_nhtfrq(i)*SHR_CONST_CDAY/(24._r8*dtime))
   268)           endif
   269)        end do
   270) 
   271)        ! Override start-type (can only override to branch (3)  and only 
   272)        ! if the driver is a startup type
   273)        if ( override_nsrest /= nsrest )then
   274)            if ( override_nsrest /= nsrBranch .and. nsrest /= nsrStartup )then
   275)               call endrun( subname//' ERROR: can ONLY override clm start-type ' // &
   276)                            'to branch type and ONLY if driver is a startup type' )
   277)            end if
   278)            call clm_varctl_set( nsrest_in=override_nsrest )
   279)        end if
   280)        
   281)        ! Consistency of elevation classes on namelist to what's sent by the coupler
   282)        if (glc_nec /= maxpatch_glcmec ) then
   283)           write(iulog,*)'glc_nec, maxpatch_glcmec=',glc_nec, maxpatch_glcmec
   284)           write(iulog,*)'Number of glacier elevation classes from clm namelist and' // &
   285)                         ' sent by the coupler MUST be equal'
   286)           call endrun( subname //' ERROR: glc_nec and maxpatch_glcmec must be equal')  
   287)        end if
   288)        if (maxpatch_glcmec > 0) then
   289)           create_glacier_mec_landunit = .true.
   290)        else
   291)           create_glacier_mec_landunit = .false.
   292)        end if
   293)        
   294)     endif   ! end of if-masterproc if-block
   295) 
   296)     call clmvarctl_init( masterproc, dtime )
   297) 
   298)     ! ----------------------------------------------------------------------
   299)     ! Broadcast all control information if appropriate
   300)     ! ----------------------------------------------------------------------
   301) 
   302)     call control_spmd()
   303)     
   304)     if (masterproc) then
   305)        write(iulog,*) 'Successfully initialized run control settings'
   306)        write(iulog,*)
   307)     endif
   308) 
   309)   end subroutine control_init

   315) ! !IROUTINE: control_spmd
   316) !
   317) ! !INTERFACE:
   318)   subroutine control_spmd()
   319) !
   320) ! !DESCRIPTION:
   321) ! Distribute namelist data all processors. All program i/o is 
   322) ! funnelled through the master processor. Processor 0 either 
   323) ! reads restart/history data from the disk and distributes 
   324) ! it to all processors, or collects data from
   325) ! all processors and writes it to disk.
   326) !
   327) ! !USES:
   328) !
   329)     use spmdMod,    only : mpicom, MPI_CHARACTER, MPI_INTEGER, MPI_LOGICAL, MPI_REAL8
   330)     use clm_varctl, only : single_column, scmlat, scmlon, rpntfil
   331) !
   332) ! !ARGUMENTS:
   333)     implicit none
   334) !
   335) ! !REVISION HISTORY:
   336) ! Created by Mariana Vertenstein
   337) !
   338) !
   339) ! !LOCAL VARIABLES:
   340) !EOP
   341)     integer ier       !error code
   342) !-----------------------------------------------------------------------
   343) 
   344)     ! run control variables
   345) 
   346)     call mpi_bcast (caseid,         len(caseid),        MPI_CHARACTER, 0, mpicom, ier)
   347)     call mpi_bcast (ctitle,         len(ctitle),        MPI_CHARACTER, 0, mpicom, ier)
   348)     call mpi_bcast (model_version,  len(model_version), MPI_CHARACTER, 0, mpicom, ier)
   349)     call mpi_bcast (hostname,       len(hostname),      MPI_CHARACTER, 0, mpicom, ier)
   350)     call mpi_bcast (username,       len(username),      MPI_CHARACTER, 0, mpicom, ier)
   351)     call mpi_bcast (nsrest,                     1,      MPI_INTEGER  , 0, mpicom, ier)
   352) 
   353)     ! initial file variables
   354) 
   355)     call mpi_bcast (nrevsn  , len(nrevsn)  , MPI_CHARACTER, 0, mpicom, ier)
   356)     call mpi_bcast (finidat , len(finidat) , MPI_CHARACTER, 0, mpicom, ier)
   357)     call mpi_bcast (fsurdat , len(fsurdat) , MPI_CHARACTER, 0, mpicom, ier)
   358)     call mpi_bcast (fatmlndfrc,len(fatmlndfrc),MPI_CHARACTER, 0, mpicom, ier)
   359)     call mpi_bcast (fatmtopo, len(fatmtopo) ,MPI_CHARACTER, 0, mpicom, ier)
   360)     call mpi_bcast (flndtopo, len(flndtopo) ,MPI_CHARACTER, 0, mpicom, ier)
   361)     call mpi_bcast (fpftcon , len(fpftcon) , MPI_CHARACTER, 0, mpicom, ier)
   362)     call mpi_bcast (flanduse_timeseries , len(flanduse_timeseries) , MPI_CHARACTER, 0, mpicom, ier)
   363)     call mpi_bcast (fsnowoptics,  len(fsnowoptics),  MPI_CHARACTER, 0, mpicom, ier)
   364)     call mpi_bcast (fsnowaging,   len(fsnowaging),   MPI_CHARACTER, 0, mpicom, ier)
   365) 
   366)     ! Landunit generation
   367) 
   368)     call mpi_bcast(create_crop_landunit, 1, MPI_LOGICAL, 0, mpicom, ier)
   369)     call mpi_bcast(allocate_all_vegpfts, 1, MPI_LOGICAL, 0, mpicom, ier)
   370) 
   371)     ! BGC
   372) 
   373)     call mpi_bcast (co2_type, len(co2_type), MPI_CHARACTER, 0, mpicom, ier)
   374)     if (use_cn) then
   375)        call mpi_bcast (suplnitro, len(suplnitro), MPI_CHARACTER, 0, mpicom, ier)
   376)     end if
   377) 
   378)     ! isotopes
   379) 
   380)     call mpi_bcast (use_c13,          1, MPI_LOGICAL,     0, mpicom, ier)
   381)     call mpi_bcast (use_c14,          1, MPI_LOGICAL,     0, mpicom, ier)
   382) 
   383)     ! physics variables
   384) 
   385)     call mpi_bcast (urban_hac     , len(urban_hac), MPI_CHARACTER, 0, mpicom, ier)
   386)     call mpi_bcast (urban_traffic , 1, MPI_LOGICAL, 0, mpicom, ier)
   387)     call mpi_bcast (nsegspc     , 1, MPI_INTEGER, 0, mpicom, ier)
   388)     call mpi_bcast (wrtdia      , 1, MPI_LOGICAL, 0, mpicom, ier)
   389)     call mpi_bcast (single_column,1, MPI_LOGICAL, 0, mpicom, ier)
   390)     call mpi_bcast (scmlat,       1, MPI_REAL8,   0, mpicom, ier)
   391)     call mpi_bcast (scmlon,       1, MPI_REAL8,   0, mpicom, ier)
   392)     call mpi_bcast (co2_ppmv    , 1, MPI_REAL8,   0, mpicom, ier)
   393)     call mpi_bcast (albice      , 2, MPI_REAL8,   0, mpicom, ier)
   394) 
   395)     ! glacier_mec variables
   396) 
   397)     call mpi_bcast (create_glacier_mec_landunit, 1, MPI_LOGICAL  , 0, mpicom, ier)
   398)     call mpi_bcast (maxpatch_glcmec             ,1, MPI_INTEGER  , 0, mpicom, ier)
   399)     call mpi_bcast (glc_smb,                     1, MPI_LOGICAL  , 0, mpicom, ier)
   400)     call mpi_bcast (glc_dyntopo,                 1, MPI_LOGICAL  , 0, mpicom, ier)
   401)     call mpi_bcast (glc_grid,        len(glc_grid), MPI_CHARACTER, 0, mpicom, ier)
   402)     call mpi_bcast (fglcmask,        len(fglcmask), MPI_CHARACTER, 0, mpicom, ier)
   403) 
   404)     ! history file variables
   405) 
   406)     call mpi_bcast (hist_empty_htapes, 1, MPI_LOGICAL, 0, mpicom, ier)
   407)     call mpi_bcast (hist_dov2xy, size(hist_dov2xy), MPI_LOGICAL, 0, mpicom, ier)
   408)     call mpi_bcast (hist_nhtfrq, size(hist_nhtfrq), MPI_INTEGER, 0, mpicom, ier)
   409)     call mpi_bcast (hist_mfilt, size(hist_mfilt), MPI_INTEGER, 0, mpicom, ier)
   410)     call mpi_bcast (hist_ndens, size(hist_ndens), MPI_INTEGER, 0, mpicom, ier)
   411)     call mpi_bcast (hist_avgflag_pertape, size(hist_avgflag_pertape), MPI_CHARACTER, 0, mpicom, ier)
   412)     call mpi_bcast (hist_type1d_pertape, max_namlen*size(hist_type1d_pertape), MPI_CHARACTER, 0, mpicom, ier)
   413)     call mpi_bcast (hist_fexcl1, max_namlen*size(hist_fexcl1), MPI_CHARACTER, 0, mpicom, ier)
   414)     call mpi_bcast (hist_fexcl2, max_namlen*size(hist_fexcl2), MPI_CHARACTER, 0, mpicom, ier)
   415)     call mpi_bcast (hist_fexcl3, max_namlen*size(hist_fexcl3), MPI_CHARACTER, 0, mpicom, ier)
   416)     call mpi_bcast (hist_fexcl4, max_namlen*size(hist_fexcl4), MPI_CHARACTER, 0, mpicom, ier)
   417)     call mpi_bcast (hist_fexcl5, max_namlen*size(hist_fexcl5), MPI_CHARACTER, 0, mpicom, ier)
   418)     call mpi_bcast (hist_fexcl6, max_namlen*size(hist_fexcl6), MPI_CHARACTER, 0, mpicom, ier)
   419)     call mpi_bcast (hist_fincl1, (max_namlen+2)*size(hist_fincl1), MPI_CHARACTER, 0, mpicom, ier)
   420)     call mpi_bcast (hist_fincl2, (max_namlen+2)*size(hist_fincl2), MPI_CHARACTER, 0, mpicom, ier)
   421)     call mpi_bcast (hist_fincl3, (max_namlen+2)*size(hist_fincl3), MPI_CHARACTER, 0, mpicom, ier)
   422)     call mpi_bcast (hist_fincl4, (max_namlen+2)*size(hist_fincl4), MPI_CHARACTER, 0, mpicom, ier)
   423)     call mpi_bcast (hist_fincl5, (max_namlen+2)*size(hist_fincl5), MPI_CHARACTER, 0, mpicom, ier)
   424)     call mpi_bcast (hist_fincl6, (max_namlen+2)*size(hist_fincl6), MPI_CHARACTER, 0, mpicom, ier)
   425) 
   426)     ! restart file variables
   427) 
   428)     call mpi_bcast (rpntfil, len(rpntfil), MPI_CHARACTER, 0, mpicom, ier)
   429) 
   430)     ! clump decomposition variables
   431) 
   432)     call mpi_bcast (clump_pproc, 1, MPI_INTEGER, 0, mpicom, ier)
   433) 
   434)   end subroutine control_spmd

   441) ! !INTERFACE:
   442)   subroutine control_print ()
   443) !
   444) ! !DESCRIPTION:
   445) ! Write out the clm namelist run control variables
   446) !
   447) ! !USES:
   448) !
   449)     use clm_varctl,      only : source, rpntdir, rpntfil, nsrStartup, nsrBranch, &
   450)                                 nsrContinue
   451)     use CNAllocationMod, only : suplnitro, suplnNon
   452) !
   453) ! !ARGUMENTS:
   454)     implicit none
   455) !
   456) ! !REVISION HISTORY:
   457) ! Created by Mariana Vertenstein
   458) !
   459) !
   460) ! !LOCAL VARIABLES:
   461) !EOP
   462)     integer i  !loop index
   463)     character(len=32) :: subname = 'control_print'  ! subroutine name
   464) !------------------------------------------------------------------------
   465) 
   466)     write(iulog,*) 'define run:'
   467)     write(iulog,*) '   source                = ',trim(source)
   468)     write(iulog,*) '   model_version         = ',trim(model_version)
   469)     write(iulog,*) '   run type              = ',runtyp(nsrest+1)
   470)     write(iulog,*) '   case title            = ',trim(ctitle)
   471)     write(iulog,*) '   username              = ',trim(username)
   472)     write(iulog,*) '   hostname              = ',trim(hostname)
   473)     write(iulog,*) 'input data files:'
   474)     write(iulog,*) '   PFT physiology = ',trim(fpftcon)
   475)     if (fsurdat == ' ') then
   476)        write(iulog,*) '   fsurdat, surface dataset not set'
   477)     else
   478)        write(iulog,*) '   surface data   = ',trim(fsurdat)
   479)     end if
   480)     if (fatmlndfrc == ' ') then
   481)        write(iulog,*) '   fatmlndfrc not set, setting frac/mask to 1'
   482)     else
   483)        write(iulog,*) '   land frac data = ',trim(fatmlndfrc)
   484)     end if
   485)     if (flndtopo == ' ') then
