   457)           if (jr < 1 .or. jr > rtmlat .or. ir < 1 .or. ir > rtmlon) then
   458)              write(iulog,*) 'Rtmini ERROR ir jr bounds ',i,j,rdirc(n),ir,jr
   459)              call shr_sys_abort()
   460)           endif
   461)           nr = (jr-1)*rtmlon + ir
   462)           if (n == nr) then
   463)              write(iulog,*) 'Rtmini ERROR dwnstrm_index ',i,j,n,rdirc(n),ir,jr,nr
   464)              call shr_sys_abort()
   465)           endif
   466)           dwnstrm_index(n) = nr
   467)        endif
   468)     enddo
   469)     enddo
   470) 
   471)     !-------------------------------------------------------
   472)     ! Determine rtm ocn/land mask (global, all procs)
   473)     !-------------------------------------------------------
   474) 
   475)     !  0=none, 
   476)     !  1=land, 
   477)     !  2=ocean outflow, 
   478)     ! -1=reroute over ocean to ocean outflow points
   479) 
   480)     call t_startf('rtmi_decomp')
   481) 
   482)     allocate (gmask(rtmlon*rtmlat), stat=ier)
   483)     if (ier /= 0) then
   484)        write(iulog,*)'Rtmgridini: Allocation ERROR for gmask'
   485)        call shr_sys_abort()
   486)     end if
   487) 
   488)     ! Initialize gmask to 2 everywhere, then compute land cells
   489) 
   490)     gmask(:) = 2
   491) 
   492)     do n=1,rtmlon*rtmlat         ! override downstream setting from local info
   493)        nr = dwnstrm_index(n)
   494)        if (nr /= 0) then         ! n is always land if dwnstrm_index exists
   495)           if (rdirc(n) > 0) then
   496)              gmask(n) = 1
   497)           else if (rdirc(n) < 0) then 
   498)              gmask(n) = -1
   499)           end if
   500)        end if
   501)     enddo
   502) 
   503)     deallocate(rdirc)
   504) 
   505)     !-------------------------------------------------------
   506)     ! Compute river basins, actually compute ocean outlet gridcell
   507)     !-------------------------------------------------------
   508) 
   509)     ! iocn = final downstream cell, index is global 1d ocean gridcell
   510)     ! nocn = number of source gridcells for ocean gridcell
   511) 
   512)     allocate(iocn(rtmlon*rtmlat),nocn(rtmlon*rtmlat),stat=ier)
   513)     if (ier /= 0) then
   514)        write(iulog,*)'Rtmgridini: Allocation ERROR for ',&
   515)             'iocn,nocn'
   516)        call shr_sys_abort()
   517)     end if
   518) 
   519)     call t_startf('rtmi_dec_basins')
   520)     iocn = 0
   521)     nocn = 0
   522)     do nr=1,rtmlon*rtmlat
   523)        n = nr
   524)        if (abs(gmask(n)) == 1) then    ! land
   525)           g = 0
   526)           do while (abs(gmask(n)) == 1 .and. g < rtmlon*rtmlat)  ! follow downstream
   527)              n = dwnstrm_index(n)
   528)              g = g + 1
   529)           end do
   530)           if (gmask(n) == 2) then           ! found ocean outlet 
   531)              iocn(nr) = n                   ! set ocean outlet or nr to n
   532)              nocn(n) = nocn(n) + 1          ! one more land cell for n
   533)           elseif (abs(gmask(n)) == 1) then  ! no ocean outlet, warn user, ignore cell
   534)              write(iulog,*) 'rtmini ERROR no downstream ocean cell ', &
   535)                g,nr,gmask(nr),dwnstrm_index(nr), &
   536)                n,gmask(n),dwnstrm_index(n)
   537)              call shr_sys_abort()
   538)           else 
   539)              write(iulog,*) 'rtmini ERROR downstream cell is non-ocean,non-land', &
   540)                g,nr,gmask(nr),dwnstrm_index(nr), &
   541)                n,gmask(n),dwnstrm_index(n)
   542)              call shr_sys_abort()
   543)           endif
   544)        elseif (gmask(n) == 2) then  ! ocean, give to self
   545)           iocn(nr) = n
   546)           nocn(n) = nocn(n) + 1
   547)        endif
   548)     enddo
   549)     call t_stopf('rtmi_dec_basins')
   550) 
   551)     !-------------------------------------------------------
   552)     !--- Now allocate those basins to pes
   553)     !-------------------------------------------------------
   554) 
   555)     call t_startf('rtmi_dec_distr')
   556) 
   557)     !--- pocn is the pe that gets the basin associated with ocean outlet nr
   558)     !--- nop is a running count of the number of rtm cells/pe 
   559) 
   560)     nbas = 0
   561)     nrtm = 0
   562)     do nr=1,rtmlon*rtmlat
   563)        if (nocn(nr) > 0) then
   564)           nbas = nbas + 1
   565)           nrtm = nrtm + nocn(nr)
   566)        endif
   567)     enddo
   568) 
   569)     allocate(pocn(rtmlon*rtmlat),     &  !global rtm array
   570)              rglo2gdc(rtmlon*rtmlat), &  !global rtm array
   571)              nop(0:npes-1), &
   572)              nba(0:npes-1), &
   573)              nrs(0:npes-1), &
   574)              num_rtm(0:npes-1))
   575) 
   576)     nop = 0
   577)     nba = 0
   578)     nrs = 0
   579)     pocn = -99
   580)     rglo2gdc = 0
   581)     baspe = 0
   582)     maxrtm = int(float(nrtm)/float(npes)*0.445) + 1
   583)     nloops = 3
   584)     minbas = nrtm
   585)     do nl=1,nloops
   586)        maxbas = minbas - 1
   587)        minbas = maxval(nocn)/(2**nl)
   588)        if (nl == nloops) minbas = min(minbas,1)
   589)        do nr=1,rtmlon*rtmlat
   590)           if (nocn(nr) > 0 .and. nocn(nr) >= minbas .and. nocn(nr) <= maxbas) then
   591)              ! Decomp options
   592)              !   find min pe (implemented but scales poorly)
   593)              !   use increasing thresholds (implemented, ok load balance for l2r or calc)
   594)              !   distribute basins using above methods but work from max to min basin size
   595)              !
   596)              !--------------
   597)              ! find min pe
   598)              !             baspe = 0
   599)              !             do n = 1,npes-1
   600)              !                if (nop(n) < nop(baspe)) baspe = n
   601)              !             enddo
   602)              !--------------
   603)              ! find next pe below maxrtm threshhold and increment
   604)              do while (nop(baspe) > maxrtm)
   605)                 baspe = baspe + 1
   606)                 if (baspe > npes-1) then
   607)                    baspe = 0
   608)                    maxrtm = max(maxrtm*1.5, maxrtm+1.0)   ! 3 loop, .445 and 1.5 chosen carefully
   609)                 endif
   610)              enddo
   611)              !--------------
   612)              if (baspe > npes-1 .or. baspe < 0) then
   613)                 write(iulog,*) 'ERROR in decomp for rtm ',nr,npes,baspe
   614)                 call shr_sys_abort()
   615)              endif
   616)              nop(baspe) = nop(baspe) + nocn(nr)
   617)              nba(baspe) = nba(baspe) + 1
   618)              pocn(nr) = baspe
   619)           endif
   620)        enddo ! nr
   621)     enddo ! nl
   622) 
   623)     ! set pocn for land cells, was set for ocean above
   624)     do nr=1,rtmlon*rtmlat
   625)        if (iocn(nr) > 0) then
   626)           pocn(nr) = pocn(iocn(nr))
   627)           if (pocn(nr) < 0 .or. pocn(nr) > npes-1) then
   628)              write(iulog,*) 'Rtmini ERROR pocn lnd setting ',&
   629)                   nr,iocn(nr),iocn(iocn(nr)),pocn(iocn(nr)),pocn(nr),npes
   630)              call shr_sys_abort()
   631)           endif
   632)        endif
   633)     enddo
   634) 
   635)     if (masterproc) then
   636)        write(iulog,*) 'rtm cells and basins total  = ',nrtm,nbas
   637)        write(iulog,*) 'rtm cells per basin avg/max = ',nrtm/nbas,maxval(nocn)
   638)        write(iulog,*) 'rtm cells per pe    min/max = ',minval(nop),maxval(nop)
   639)        write(iulog,*) 'basins    per pe    min/max = ',minval(nba),maxval(nba)
   640)     end if
   641) 
   642)     !-------------------------------------------------------
   643)     !--- Count and distribute cells to rglo2gdc
   644)     !-------------------------------------------------------
   645) 
   646)     runoff%numr   = 0
   647)     runoff%numro  = 0
   648)     runoff%numrl  = 0
   649)     runoff%lnumr  = 0
   650)     lnumro = 0
   651)     lnumrl = 0
   652)     num_rtm = 0
   653) 
   654)     do n = 0,npes-1
   655)        if (iam == n) then
   656)           runoff%begr  = runoff%numr  + 1
   657)           begrl = runoff%numrl + 1
   658)           begro = runoff%numro + 1
   659)        endif
   660)        num_rtm(n)   = num_rtm(n)   + nop(n)
   661)        runoff%numr  = runoff%numr  + nop(n)
   662)        runoff%numro = runoff%numro + nba(n)
   663)        runoff%numrl = runoff%numrl + nop(n) - nba(n)
   664)        if (iam == n) then
   665)           runoff%lnumr = runoff%lnumr + nop(n)
   666)           runoff%endr  = runoff%begr  + runoff%lnumr  - 1
   667)           lnumro = lnumro + nba(n)
   668)           lnumrl = lnumrl + nop(n) - nba(n)
   669)           endro  = begro + lnumro - 1
   670)           endrl  = begrl + lnumrl - 1
   671)        endif
   672)     enddo
   673) 
   674)     deallocate(num_rtm)
   675) 
   676)     ! nrs is begr on each pe
   677)     nrs(0) = 1
   678)     do n = 1,npes-1
   679)        nrs(n) = nrs(n-1) + nop(n-1)
   680)     enddo
   681) 
   682)     ! reuse nba for nop-like counter here
   683)     ! pocn -99 is unused cell
   684)     nba = 0
   685)     do nr = 1,rtmlon*rtmlat
   686)        if (pocn(nr) >= 0) then
   687)           rglo2gdc(nr) = nrs(pocn(nr)) + nba(pocn(nr))
   688)           nba(pocn(nr)) = nba(pocn(nr)) + 1          
   689)        endif
   690)     enddo
   691)     do n = 0,npes-1
   692)        if (nba(n) /= nop(n)) then
   693)           write(iulog,*) 'Rtmini ERROR rtm cell count ',n,nba(n),nop(n)
   694)           call shr_sys_abort()
   695)        endif
   696)     enddo
   697) 
   698)     deallocate(nop,nba,nrs)
   699)     deallocate(iocn,nocn)
   700)     deallocate(pocn)
   701)     call t_stopf('rtmi_dec_distr')
   702) 
   703)     !--- set some local values
   704) 
   705)     nroflnd = runoff%numrl
   706)     nrofocn = runoff%numro
   707)     numr = nroflnd + nrofocn
   708)     begr = runoff%begr
   709)     endr = runoff%endr
   710)     call t_stopf('rtmi_decomp')
   711) 
   712)     !--- Write per-processor runoff bounds depending on dbug level
   713) 
   714)     call t_startf('rtmi_print')
   715) 
   716)     call shr_sys_flush(iulog)
   717)     if (masterproc) then
   718)        write(iulog,*) 'total runoff cells numr  = ',runoff%numr
   719)        write(iulog,*) 'total runoff cells numrl = ',runoff%numrl
   720)        write(iulog,*) 'total runoff cells numro = ',runoff%numro
   721)     endif
   722)     call shr_sys_flush(iulog)
   723)     call mpi_barrier(mpicom_rof,ier)
   724)     npmin = 0
   725)     npmax = npes-1
   726)     npint = 1
   727)     if (dbug == 0) then
   728)        npmax = 0
   729)     elseif (dbug == 1) then
   730)        npmax = min(npes-1,4)
   731)     elseif (dbug == 2) then
   732)        npint = npes/8
   733)     endif
   734)     do np = npmin,npmax,npint
   735)        pid = np
   736)        if (dbug == 1) then
   737)           if (np == 2) pid=npes/2-1
   738)           if (np == 3) pid=npes-2
   739)           if (np == 4) pid=npes-1
   740)        endif
   741)        pid = max(pid,0)
   742)        pid = min(pid,npes-1)
   743)        if (iam == pid) then
   744)           write(iulog,*) 'rtm decomp info',' proc = ',iam, &
   745)              ' begr = ',runoff%begr,&
   746)              ' endr = ',runoff%endr, &
   747)              ' numr = ',runoff%lnumr
   748)           write(iulog,*) '               ',' proc = ',iam, &
   749)              ' begrl= ',begrl,&
   750)              ' endrl= ',endrl, &
   751)              ' numrl= ',lnumrl
   752)           write(iulog,*) '               ',' proc = ',iam, &
   753)              ' begro= ',begro,&
   754)              ' endro= ',endro, &
   755)              ' numro= ',lnumro
   756)        endif
   757)        call shr_sys_flush(iulog)
   758)        call mpi_barrier(mpicom_rof,ier)
   759)     enddo
   760) 
   761)     call t_stopf('rtmi_print')
   762) 
   763)     !-------------------------------------------------------
   764)     ! Allocate local flux variables
   765)     !-------------------------------------------------------
   766) 
   767)     call t_startf('rtmi_vars')
   768) 
   769)     allocate (fluxout (begr:endr,nt_rtm), &
   770)               ddist   (begr:endr),        &
   771)               evel    (begr:endr,nt_rtm), &
   772)               sfluxin (begr:endr,nt_rtm), stat=ier)
   773) 
   774)     if (ier /= 0) then
   775)        write(iulog,*)'Rtmgridini: Allocation ERROR for ',&
   776)             'volr, fluxout, ddist'
   777)        call shr_sys_abort()
   778)     end if
   779)     fluxout(:,:) = 0._r8
   780)     ddist(:)     = 0._r8
   781)     sfluxin(:,:) = 0._r8
   782) 
   783)     !-------------------------------------------------------
   784)     ! Allocate runoff datatype 
   785)     !-------------------------------------------------------
   786) 
   787)     call RunoffInit(begr, endr, numr)
   788) 
   789)     allocate(rgdc2glo(numr), stat=ier)
   790)     if (ier /= 0) then
   791)        write(iulog,*)'Rtmini ERROR allocation of runoff%gcd2glo'
   792)        call shr_sys_abort()
   793)     end if
   794) 
   795)     ! Set map from local to global index space
   796)     numr = 0
   797)     do j = 1,rtmlat
   798)     do i = 1,rtmlon
   799)        n = (j-1)*rtmlon + i
   800)        nr = rglo2gdc(n)
   801)        if (nr /= 0) then
   802)           numr = numr + 1
   803)           rgdc2glo(nr) = n         
   804)           runoff%mask(nr) = gmask(n)
   805)        endif
   806)     end do
   807)     end do
   808) 
   809) 
   810)     !-------------------------------------------------------
   811)     ! Initialize runoff data type
   812)     !-------------------------------------------------------
   813) 
   814)     if (numr /= runoff%numr) then
   815)        write(iulog,*) 'Rtmini ERROR numr and runoff%numr are different ',numr,runoff%numr
   816)        call shr_sys_abort()
   817)     endif
   818)     deallocate(gmask)
   819) 
   820)     ! Set edge latitudes (assumes latitudes are constant for a given longitude)
   821)     rlats(:) = edges
   822)     rlatn(:) = edgen
   823)     do j = 2, rtmlat
   824)        if (rlatc(2) > rlatc(1)) then ! South to North grid
   825)           rlats(j)   = (rlatc(j-1) + rlatc(j)) / 2._r8
   826)           rlatn(j-1) = rlats(j)
   827)        else  ! North to South grid
   828)           rlatn(j)   = (rlatc(j-1) + rlatc(j)) / 2._r8
   829)           rlats(j-1) = rlatn(j)
   830)        end if
   831)     end do
   832) 
   833)     ! Set edge longitudes
   834)     rlonw(:) = edgew
   835)     rlone(:) = edgee
   836)     dx = (edgee - edgew) / rtmlon
   837)     do i = 2, rtmlon
   838)        rlonw(i)   = rlonw(i) + (i-1)*dx
   839)        rlone(i-1) = rlonw(i)
   840)     end do
   841) 
   842)     ! Determine runoff datatype variables
   843)     lrtmarea = 0.0_r8
   844)     do nr = begr,endr
   845)        runoff%gindex(nr) = rgdc2glo(nr)
   846)        n = rgdc2glo(nr)
   847)        i = mod(n-1,rtmlon) + 1
   848)        j = (n-1)/rtmlon + 1
   849)        if (n <= 0 .or. n > rtmlon*rtmlat) then
   850)           write(iulog,*) 'Rtmini ERROR gdc2glo, nr,ng= ',nr,n
   851)           call shr_sys_abort()
   852)        endif
   853)        runoff%lonc(nr) = runoff%rlon(i)
   854)        runoff%latc(nr) = runoff%rlat(j)
   855) 
   856)        if (runoff%mask(nr) == 1) then
   857)           do nt = 1,nt_rtm
   858)              runoff%runofflnd(nr,nt) = runoff%runoff(nr,nt)
   859)              runoff%dvolrdtlnd(nr,nt)= runoff%dvolrdt(nr,nt)
   860)              runoff%volrlnd(nr,nt)= runoff%volr(nr,nt)
   861)           enddo
   862)        elseif (runoff%mask(nr) == 2) then
   863)           do nt = 1,nt_rtm
   864)              runoff%runoffocn(nr,nt) = runoff%runoff(nr,nt)
   865)              runoff%dvolrdtocn(nr,nt)= runoff%dvolrdt(nr,nt)
   866)           enddo
   867)        endif
   868) 
   869)        dx = (rlone(i) - rlonw(i)) * deg2rad
   870)        dy = sin(rlatn(j)*deg2rad) - sin(rlats(j)*deg2rad)
   871)        runoff%area(nr) = 1.e6_r8 * dx*dy*re*re
   872)        lrtmarea = lrtmarea + runoff%area(nr)
   873)        if (dwnstrm_index(n) == 0) then
   874)           runoff%dsi(nr) = 0
   875)        else
   876)           if (rglo2gdc(dwnstrm_index(n)) == 0) then
   877)              write(iulog,*) 'Rtmini ERROR glo2gdc dwnstrm ',&
   878)                   nr,n,dwnstrm_index(n),rglo2gdc(dwnstrm_index(n))
   879)              call shr_sys_abort()
   880)           endif
   881)           runoff%dsi(nr) = rglo2gdc(dwnstrm_index(n))
   882)        endif
   883) 
   884)     enddo
   885)     deallocate(dwnstrm_index)
   886)     deallocate(rglo2gdc)
   887)     deallocate(rgdc2glo)
   888)     call shr_mpi_sum(lrtmarea,runoff%totarea,mpicom_rof,'rtm totarea',all=.true.)
   889)     if (masterproc) write(iulog,*) 'Rtmini earth area',4.0_r8*shr_const_pi*1.0e6_r8*re*re
   890)     if (masterproc) write(iulog,*) 'Rtmini rtm area  ',runoff%totarea
   891) 
   892)     !-------------------------------------------------------
   893)     ! Determine downstream distance 
   894)     !-------------------------------------------------------
   895) 
   896)     ! Instead of reading a distance file calculate the downstream distance
   897)     do nr=begr,endr
   898)        g = runoff%dsi(nr)
   899)        if (g == 0) then
   900)           ddist(nr) = 0._r8
   901)        elseif (g < begr .or. g > endr) then
   902)           write(iulog,*) 'Rtmini: ERROR in ddist calc ',nr,g,begr,endr
   903)           call shr_sys_abort()
   904)        else
   905)           dy  = deg2rad * abs(runoff%latc(nr)-runoff%latc(g)) * re*1000._r8
   906)           dx  = runoff%lonc(nr)-runoff%lonc(g)
   907)           dx1 = abs(dx)
   908)           dx2 = abs(dx+360._r8)
   909)           dx3 = abs(dx-360._r8)
   910)           dx  = min(dx1,dx2,dx3)
   911)           dx  = deg2rad * dx * re*1000._r8 * &
   912)                 0.5_r8*(cos(runoff%latc(nr)*deg2rad)+ &
   913)                         cos(runoff%latc(g)*deg2rad))
   914)           ddist(nr) = sqrt(dx*dx + dy*dy)
   915)        endif
   916)     enddo
   917) 
   918)     !-------------------------------------------------------
   919)     ! Initialize rtm flood - runoff%fthresh and evel
   920)     ! tcraig - recommend RtmFloodInit be called always here and that
   921)     !   the flood_active logic be inside rtmfloodinit, not here.
   922)     !   this will place the fthresh and evel initialization in
   923)     !   a consistent location for extensibility.
   924)     !-------------------------------------------------------
   925)     call RtmFloodInit (frivinp_rtm, begr, endr, nt_rtm, &
   926)          runoff%fthresh(begr:endr),                     &
   927)          evel(begr:endr, :),                            &
   928)          runoff%gindex(begr:endr),                      &
   929)          runoff%lnumr,                                  &
   930)          flood_active,                                  &
   931)          effvel_active)
   932) 
   933)     !-------------------------------------------------------
   934)     ! Compute timestep and subcycling number
   935)     !-------------------------------------------------------
   936) 
   937)     dtover = 0._r8
   938)     dtovermax = 0._r8
   939)     do nt=1,nt_rtm
   940)        do nr=begr,endr
   941)           if (ddist(nr) /= 0._r8) then
   942)              dtover = evel(nr,nt)/ddist(nr)
   943)           else
   944)              dtover = 0._r8
   945)           endif
   946)           dtovermax = max(dtovermax,dtover)
   947)        enddo
   948)     enddo
   949)     dtover = dtovermax
   950)     call mpi_allreduce(dtover,dtovermax,1,MPI_REAL8,MPI_MAX,mpicom_rof,ier)
   951)     if (dtovermax > 0._r8) then
   952)        delt_rtm_max = (1.0_r8/dtovermax)*cfl_scale
   953)     else
   954)        write(iulog,*) 'rtmini ERROR in delt_rtm_max ',delt_rtm_max,dtover
   955)        call shr_sys_abort()
   956)     endif
   957)     if (masterproc) write(iulog,*) 'rtm max timestep = ',delt_rtm_max,' (sec) for cfl_scale = ',cfl_scale
   958) 
   959)     call t_stopf('rtmi_vars')
   960) 
   961)     !-------------------------------------------------------
   962)     ! Read restart/initial info
   963)     !-------------------------------------------------------
   964) 
   965)     ! The call below opens and closes the file
   966)     if ((nsrest == nsrStartup .and. finidat_rtm /= ' ') .or. &

   994) ! !IROUTINE: Rtmrun

   996) ! !INTERFACE:
   997)   subroutine Rtmrun(totrunin, rstwr, nlend, rdate)
   998) !
   999) ! !DESCRIPTION:
  1000) ! River routing model (based on U. Texas code).
  1001) ! Input is totrunin
  1002) ! Input/output is fluxout, volr.
  1003) ! Outputs are dvolrdt\_r, dvolrdt\_lnd\_r, dvolrdt\_ocn\_r, flxocn\_r, flxlnd\_r.
  1004) !
  1005) ! !USES:
  1006)   use RunoffMod       , only : RunoffInit, runoff
  1007) !
  1008) ! !ARGUMENTS:
  1009)     implicit none
  1010)     real(r8),         intent(inout) :: totrunin( runoff%begr: ,: )  ! cell tracer lnd forcing on rtm grid (mm/s)
  1011)     logical ,         intent(in)    :: rstwr          ! true => write restart file this step
  1012)     logical ,         intent(in)    :: nlend          ! true => end of run on this step
  1013)     character(len=*), intent(in)    :: rdate          ! restart file time stamp for name
  1014) !
  1015) ! !CALLED FROM:
  1016) ! subroutine RtmMap in this module
  1017) !
  1018) ! !REVISION HISTORY:
  1019) ! Author: Sam Levis
  1020) !
  1021) !
  1022) ! !LOCAL VARIABLES:
  1023) !EOP
  1024)     integer  :: i, j, n, nr, ns, nt         ! indices
  1025)     real(r8) :: dvolrdt                     ! change in storage in discharge units (m3/s)
  1026)     real(r8) :: sumfin(nt_rtm),sumfex(nt_rtm)
  1027)     real(r8) :: sumrin(nt_rtm),sumdvt(nt_rtm)
  1028)     real(r8) :: budget1(nt_rtm)             ! budget check in m3/s
  1029)     real(r8) :: budget_global(nt_rtm)       ! global budget sum
  1030)     logical  :: budget_check                ! do global budget check
  1031)     real(r8),parameter :: budget_tolerance = 1.0e-6   ! budget tolerance, m3/day
  1032)     logical  :: abort                       ! abort flag
  1033)     real(r8) :: sum1,sum2
  1034)     integer  :: yr, mon, day, ymd, tod      ! time information
  1035)     integer  :: nsub                        ! subcyling for cfl
  1036)     real(r8) :: delt                        ! delt associated with subcycling
  1037)     real(r8) :: delt_rtm                    ! real value of rtm_tstep
  1038)     integer , save :: nsub_save             ! previous nsub
  1039)     real(r8), save :: delt_save             ! previous delt
  1040)     logical , save :: first_time = .true.   ! first time flag (for backwards compatibility)
  1041)     character(len=256) :: filer             ! restart file name
  1042)     integer,parameter  :: dbug = 1          ! local debug flag
  1043)     character(*),parameter :: subname = '(Rtmrun) '
  1044) !-----------------------------------------------------------------------
  1045) 
  1046)     call t_startf('RTMrun')
  1047)     call shr_sys_flush(iulog)
  1048) 
  1049)     call get_curr_date(yr, mon, day, tod)
  1050)     ymd = yr*10000 + mon*100 + day
  1051)     if (tod == 0 .and. masterproc) then
  1052)        write(iulog,*) ' '
  1053)        write(iulog,*) trim(subname),' model date is',ymd,tod
  1054)     endif
  1055) 
  1056)     delt_rtm = rtm_tstep*1.0_r8
  1057)     if (first_time) then
  1058)        if (masterproc) write(iulog,*) trim(subname),': rtm act timestep ~ ',delt_rtm
  1059)        first_time = .false.
  1060)     end if
  1061) 
  1062)     budget_check = .false.
  1063)     if (day == 1 .and. mon == 1) budget_check = .true.
  1064) 
  1065)     ! BUDGET per water tracer, in m3/s = totrunin + (volr_i - volr_f)/dt - flood - runoff ~ 0.0
  1066)     ! BUDGET, totrunin input, add initial volume, final volume removed later
  1067)     if (budget_check) then
  1068)        call t_startf('RTMbudget')
  1069)        budget1 = 0.0_r8
  1070)        do nt = 1,nt_rtm
  1071)        do nr = runoff%begr,runoff%endr
  1072)           budget1(nt) = budget1(nt) + totrunin(nr,nt)*runoff%area(nr)*0.001_r8 &
  1073)                                     + runoff%volr(nr,nt)/delt_rtm
  1074)        enddo
  1075)        enddo
  1076)        call t_stopf('RTMbudget')
  1077)     endif
  1078) 
  1079)     ! Remove water from rtm and send back to clm
  1080)     ! Just consider land points and only remove liquid water 
  1081)     ! runoff%flood needs to be a flux - in units of mm/s
  1082)     ! totrunin is a flux (mm/s)
  1083) 
  1084)     call t_startf('RTMflood')
  1085)     nt = 1 
  1086)     do nr = runoff%begr,runoff%endr
  1087)        ! initialize runoff%flood to zero
  1088)        runoff%flood(nr) = 0._r8
  1089)        if (flood_active .and. runoff%mask(nr) == 1) then
  1090)           if (runoff%volr(nr,nt) > runoff%fthresh(nr)) then 
  1091)              ! determine flux that is sent back to the land
  1092)              ! need to convert to mm/s to be consistent with totrunin units
  1093)              runoff%flood(nr) = &
  1094)                   1000._r8*(runoff%volr(nr,nt)-runoff%fthresh(nr)) / &
  1095)                   (delt_rtm*runoff%area(nr))
  1096)              ! runoff%flood will be sent back to land - so must subtract this 
  1097)              ! from the input runoff from land
  1098)              ! tcraig, comment - this seems like an odd approach, you
  1099)              !   might create negative forcing.  why not take it out of
  1100)              !   the volr directly?  it's also odd to compute this
  1101)              !   at the initial time of the time loop.  why not do
  1102)              !   it at the end or even during the run loop as the
  1103)              !   new volume is computed.  fluxout depends on volr, so
  1104)              !   how this is implemented does impact the solution.
  1105)              totrunin(nr,nt)= totrunin(nr,nt) - runoff%flood(nr)
  1106)           endif
  1107)        endif
  1108)     enddo
  1109)     call t_stopf('RTMflood')
  1110)     
  1111)     ! BUDGET, flood out, just liquid water term
  1112)     if (budget_check) then
  1113)        call t_startf('RTMbudget')
  1114)        nt = 1
  1115)        do nr = runoff%begr,runoff%endr
  1116)           budget1(nt) = budget1(nt) - runoff%flood(nr)*runoff%area(nr)*0.001_r8
  1117)        enddo
  1118)        call t_stopf('RTMbudget')
  1119)     endif
  1120) 
  1121)     ! Subcycling
  1122) 
  1123)     call t_startf('RTMsubcycling')
  1124) 
  1125)     nsub = int(delt_rtm/delt_rtm_max) + 1
  1126)     delt = delt_rtm/float(nsub)
  1127)     if (delt /= delt_save) then
  1128)        if (masterproc) then
  1129)           write(iulog,*) trim(subname),': rtm delt update from/to',delt_save,delt,nsub_save,nsub
  1130)        end if
  1131)     endif
  1132) 
  1133)     nsub_save = nsub
  1134)     delt_save = delt
  1135) 
  1136)     sumfin = 0._r8
  1137)     sumfex = 0._r8
  1138)     sumrin = 0._r8
  1139)     sumdvt = 0._r8
  1140)     runoff%runoff = 0._r8
  1141)     runoff%runofflnd = spval
  1142)     runoff%runoffocn = spval
  1143)     runoff%dvolrdt = 0._r8
  1144)     runoff%dvolrdtlnd = spval
  1145)     runoff%dvolrdtocn = spval
  1146) 
  1147)     do ns = 1,nsub
  1148) 
  1149)        sfluxin = 0._r8
  1150)        do n = runoff%begr,runoff%endr
  1151)           nr = runoff%dsi(n)
  1152)           if (abs(runoff%mask(n)) == 1) then
  1153)              if (nr < runoff%begr .or. nr > runoff%endr) then
  1154)                 write(iulog,*) trim(subname),':Rtm ERROR: non-local communication ',n,nr
  1155)                 call shr_sys_abort()
  1156)              endif
  1157)              do nt = 1,nt_rtm
  1158)                 sfluxin(nr,nt) = sfluxin(nr,nt) + fluxout(n,nt)
  1159)              enddo
  1160)           endif
  1161)        enddo
  1162) 
  1163)        if (dbug > 1) then
  1164)           do nt = 1,nt_rtm
  1165)              sum1 = 0._r8
  1166)              sum2 = 0._r8
  1167)              do n = runoff%begr,runoff%endr
  1168)                 sum1 = sum1 + sfluxin(n,nt)
  1169)                 sum2 = sum2 + fluxout(n,nt)
  1170)              enddo
  1171)              if (abs(sum1+sum2) > 0.0_r8) then
  1172)              if (abs(sum1-sum2)/(sum1+sum2) > 1.0e-12) then
  1173)                 write(iulog,*) trim(subname),':RTM Warning: fluxin = ',sum1,&
  1174)                      ' not equal to fluxout = ',sum2,' for tracer ',nt
  1175)              endif
  1176)              endif
  1177)           enddo
  1178)        endif
  1179) 
  1180)        do nt = 1,nt_rtm
  1181)        do n = runoff%begr,runoff%endr
  1182)           dvolrdt = sfluxin(n,nt) + 0.001_r8*totrunin(n,nt)*runoff%area(n) - fluxout(n,nt)
  1183) 
  1184)           if (dbug > 1) then
  1185)              sumfin(nt) = sumfin(nt) + sfluxin(n,nt)
  1186)              sumfex(nt) = sumfex(nt) + fluxout(n,nt)
  1187)              sumrin(nt) = sumrin(nt) + 0.001_r8*totrunin(n,nt)*runoff%area(n)
  1188)              sumdvt(nt) = sumdvt(nt) + dvolrdt
  1189)           endif
  1190) 
  1191)           if (abs(runoff%mask(n)) == 1) then         ! land points
  1192)              runoff%volr(n,nt) = runoff%volr(n,nt) + dvolrdt*delt
  1193)              fluxout(n,nt) = runoff%volr(n,nt) * evel(n,nt)/ddist(n)
  1194)              ! --- old cfl constraint.  now use subcycling.  for reference only
  1195)              ! fluxout(n)  = min(fluxout(n), volr(n)/delt)
  1196)              ! --- this would stop upstream flow if volr/fluxout < 0
  1197)              ! --- negative flow largely a function of negative forcing
  1198)              ! --- can still have negative runoff where there is negative
  1199)              !     forcing over a mask=2 (ocn) point since forcing is put onto
  1200)              !     the ocean instantly at these points
  1201)              ! --- also, want to allow negative flow so it doesn't build up
  1202)              ! fluxout(n) = max(fluxout(n),0._r8)
  1203)           else
  1204)              runoff%volr(n,nt) = 0._r8
  1205)              fluxout(n,nt) = 0._r8
  1206)           endif
  1207) 
  1208)           if (abs(runoff%mask(n)) == 1) then
  1209)              runoff%runoff(n,nt) = runoff%runoff(n,nt) + fluxout(n,nt)
  1210)           elseif (runoff%mask(n) == 2) then
  1211)              runoff%runoff(n,nt) = runoff%runoff(n,nt) + dvolrdt
  1212)           elseif (dvolrdt /= 0.0_r8) then
  1213)              ! this water has no where to go.....
  1214)              write(iulog,*) subname,' runoff dvolrdt ERROR ',nt,n,runoff%mask(n),dvolrdt
  1215)              call shr_sys_abort(subname//' runoff dvolrdt ERROR')
  1216)           endif
  1217)           ! Convert local dvolrdt (in m3/s) to output dvolrdt (in mm/s)
  1218)           runoff%dvolrdt(n,nt) = runoff%dvolrdt(n,nt) + 1000._r8*dvolrdt/runoff%area(n)
  1219) 
  1220)        enddo
  1221)        enddo
  1222) 
  1223)     enddo
  1224) 
  1225)     ! average fluxes over subcycling
  1226)     runoff%runoff  = runoff%runoff / float(nsub)
  1227)     runoff%dvolrdt = runoff%dvolrdt / float(nsub)
  1228)     runoff%fluxout = fluxout
  1229) 
  1230)     do n = runoff%begr,runoff%endr
  1231)        if (runoff%mask(n) == 1) then
  1232)           do nt = 1,nt_rtm
  1233)              runoff%volrlnd(n,nt)= runoff%volr(n,nt)
  1234)              runoff%runofflnd(n,nt) = runoff%runoff(n,nt)
  1235)              runoff%dvolrdtlnd(n,nt)= runoff%dvolrdt(n,nt)
  1236)           enddo
  1237)        elseif (runoff%mask(n) == 2) then
  1238)           do nt = 1,nt_rtm
  1239)              runoff%runoffocn(n,nt) = runoff%runoff(n,nt)
  1240)              runoff%dvolrdtocn(n,nt)= runoff%dvolrdt(n,nt)
  1241)           enddo
  1242)        endif
  1243)     enddo
  1244) 
  1245)     call t_stopf('RTMsubcycling')
  1246) 
  1247)     ! BUDGET, runoff out (only ocean points), subtract final volume out
  1248)     if (budget_check) then
  1249)        call t_startf('RTMbudget')
  1250)        do nt = 1,nt_rtm
  1251)        do nr = runoff%begr,runoff%endr
  1252)           if (runoff%mask(nr) == 2) then
  1253)              budget1(nt) = budget1(nt) - runoff%runoff(nr,nt)
  1254)           endif
  1255)           budget1(nt) = budget1(nt) - runoff%volr(nr,nt)/delt_rtm
  1256)        enddo
  1257)        enddo
  1258)        call t_stopf('RTMbudget')
  1259)     endif
  1260) 
  1261)     ! BUDGET, compute global sums and check
  1262)     if (budget_check) then
  1263)        call t_startf('RTMbudget')
  1264)        call shr_mpi_sum(budget1,budget_global,mpicom_rof,'rtm global budget',all=.false.)
  1265)        if (masterproc) then
  1266)           write(iulog,*) ' '
  1267)           abort = .false.
  1268)           do nt = 1,nt_rtm
  1269)              budget_global(nt) = budget_global(nt) * 1000._r8 / runoff%totarea * 1.0e6_r8   ! convert from m3/s to kg/m2s*1e6
  1270)              write(iulog,'(2a,i10,i6,i4,g20.12)') trim(subname),':BUDGET check (kg/m2s*1e6)= ',ymd,tod,nt,budget_global(nt)
  1271)              if (abs(budget_global(nt)) > budget_tolerance) abort = .true.
  1272)           enddo
  1273)           if (abort) then
  1274)              write(iulog,*) trim(subname),':BUDGET abort balance too large ',budget_tolerance
  1275)              call shr_sys_abort(trim(subname)//':rtm budget ERROR')
  1276)           endif
  1277)        endif
  1278)        call t_stopf('RTMbudget')
  1279)     endif
  1280) 
  1281)     ! Write out RTM history and restart file
  1282)     call t_startf('RTMhbuf')
  1283)     call RtmHistFldsSet()
  1284)     call RtmHistUpdateHbuf()
  1285)     call t_stopf('RTMhbuf')
  1286) 
  1287)     call t_startf('RTMhtapes')
  1288)     call RtmHistHtapesWrapup( rstwr, nlend )
  1289)     call t_stopf('RTMhtapes')
  1290) 
  1291)     if (rstwr) then
  1292)        call t_startf('RTMrest')
  1293)        filer = RtmRestFileName(rdate=rdate)
  1294)        call RtmRestFileWrite( filer, rdate=rdate )
  1295)        call t_stopf('RTMrest')
  1296)     end if
  1297) 
  1298)     ! Global water balance calculation and ERROR check
  1299)     if (dbug > 1) then
  1300)        do nt = 1,nt_rtm
  1301)        if (abs(sumdvt(nt)+sumrin(nt)) > 0.0_r8) then
  1302)        if (abs((sumdvt(nt)-sumrin(nt))/(sumdvt(nt)+sumrin(nt))) > 1.0e-6) then
  1303)           write(iulog,*) trim(subname),':RTM Warning: water balance nt,dvt,rin,fin,fex = ', &
  1304)              nt,sumdvt(nt),sumrin(nt),sumfin(nt),sumfex(nt)
  1305)           !call shr_sys_abort()
  1306)        endif
  1307)        endif
  1308)        enddo
  1309)     endif
  1310)     call shr_sys_flush(iulog)
  1311) 
  1312)     call t_stopf('RTMrun')
  1313)      
  1314)   end subroutine Rtmrun

  1401)        ! both have the same io descriptor

  1404)        ier = pio_inq_varid(ncid, name='SLOPE', vardesc=vardesc1)
