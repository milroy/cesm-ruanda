    38)   subroutine BeginWaterBalance(lbc, ubc, lbp, ubp, &
    39)              num_nolakec, filter_nolakec, num_lakec, filter_lakec, &
    40)              num_hydrologyc, filter_hydrologyc)
    41) !
    42) ! !DESCRIPTION:
    43) ! Initialize column-level water balance at beginning of time step
    44) !
    45) ! !USES:

    47)     use clmtype
    48)     use clm_varpar   , only : nlevgrnd, nlevsoi

    57)     integer, intent(in) :: num_nolakec                 ! number of column non-lake points in column filter
    58)     integer, intent(in) :: filter_nolakec(ubc-lbc+1)   ! column filter for non-lake points
    59)     integer, intent(in) :: num_lakec                   ! number of column non-lake points in column filter
    60)     integer, intent(in) :: filter_lakec(ubc-lbc+1)     ! column filter for non-lake points

    76)     real(r8), pointer :: h2osno(:)             ! snow water (mm H2O)
    77)     real(r8), pointer :: h2osoi_ice(:,:)       ! ice lens (kg/m2)
    78)     real(r8), pointer :: h2osoi_liq(:,:)       ! liquid water (kg/m2)
    79)     real(r8), pointer :: h2ocan_pft(:)         ! canopy water (mm H2O) (pft-level) 
    80)     real(r8), pointer :: wa(:)                 ! water in the unconfined aquifer (mm)
    81)     integer , pointer :: ctype(:)              ! column type

    83)     real(r8), pointer :: zi(:,:)               ! interface level below a "z" level (m)

    86) !

    88)     real(r8), pointer :: begwb(:)              ! water mass begining of the time step
    89) !
    90) ! !OTHER LOCAL VARIABLES:
    91) !
    92)     integer :: c, p, f, j, fc            ! indices
    93) !-----------------------------------------------------------------------
    94) 
    95)     ! Assign local pointers to derived type members (column-level)
    96) 
    97)     h2osno             => cws%h2osno
    98)     h2osoi_ice         => cws%h2osoi_ice
    99)     h2osoi_liq         => cws%h2osoi_liq
   100)     begwb              => cwbal%begwb
   101)     h2ocan_col         => pws_a%h2ocan
   102)     wa                 => cws%wa
   103)     ctype              => col%itype
   104)     zwt                => cws%zwt
   105)     zi                 => cps%zi
   106) 
   107)     ! Assign local pointers to derived type members (pft-level)
   108) 
   109)     h2ocan_pft         => pws%h2ocan
   110) 
   111)     ! Determine beginning water balance for time step
   112)     ! pft-level canopy water averaged to column
   113)     call p2c(num_nolakec, filter_nolakec, h2ocan_pft, h2ocan_col)
   114) 
   115)     do f = 1, num_hydrologyc
   116)        c = filter_hydrologyc(f)
   117)        if(zwt(c) <= zi(c,nlevsoi)) then
   118)           wa(c) = 5000._r8
   119)        end if
   120)     end do
   121) 
   122)     do f = 1, num_nolakec
   123)        c = filter_nolakec(f)
   124)        if (ctype(c) == icol_roof .or. ctype(c) == icol_sunwall &
   125)           .or. ctype(c) == icol_shadewall .or. ctype(c) == icol_road_imperv) then
   126)          begwb(c) = h2ocan_col(c) + h2osno(c)
   127)        else
   128)          begwb(c) = h2ocan_col(c) + h2osno(c) + wa(c)
   129)        end if
   130)     end do
   131)     do j = 1, nlevgrnd
   132)       do f = 1, num_nolakec
   133)          c = filter_nolakec(f)
   134)          begwb(c) = begwb(c) + h2osoi_ice(c,j) + h2osoi_liq(c,j)
   135)       end do
   136)     end do
   137) 
   138)     do f = 1, num_lakec
   139)        c = filter_lakec(f)
   140)        begwb(c) = h2osno(c)
   141)     end do
   142) 
   143)   end subroutine BeginWaterBalance

   152)   subroutine BalanceCheck(lbp, ubp, lbc, ubc, lbl, ubl, lbg, ubg)

   175)                               istdlak, istslak, istwet, istcrop, istsoil

   182)     integer :: lbl, ubl ! landunit-index bounds
   183)     integer :: lbg, ubg ! grid-index bounds
   184) !
   185) ! !CALLED FROM:
   186) ! subroutine clm_driver
   187) !
   188) ! !REVISION HISTORY:
   189) ! 15 September 1999: Yongjiu Dai; Initial code
   190) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
   191) ! 10 November 2000: Mariana Vertenstein
   192) ! Migrated to new data structures by Mariana Vertenstein and
   193) ! Peter Thornton
   194) !

   199)     logical , pointer :: do_capsnow(:)         ! true => do snow capping

   201)     real(r8), pointer :: qflx_snow_melt(:)  ! snow melt (net)
   202)     real(r8), pointer :: qflx_rain_grnd_col(:) ! rain on ground after interception (mm H2O/s) [+]
   203)     real(r8), pointer :: qflx_snow_grnd_col(:) ! snow on ground after interception (mm H2O/s) [+]

   207)     integer , pointer :: clandunit(:)       ! column's landunit index
   208)     integer , pointer :: ltype(:)           ! landunit type 
   209)     integer , pointer :: ctype(:)           ! column type 
   210)     real(r8), pointer :: pwtgcell(:)        ! pft's weight relative to corresponding gridcell
   211)     real(r8), pointer :: cwtgcell(:)        ! column's weight relative to corresponding gridcell
   212)     real(r8), pointer :: forc_rain(:)       ! rain rate [mm/s]
   213)     real(r8), pointer :: forc_snow(:)       ! snow rate [mm/s]
   214)     real(r8), pointer :: forc_lwrad(:)      ! downward infrared (longwave) radiation (W/m**2)
   215)     real(r8), pointer :: endwb(:)           ! water mass end of the time step
   216)     real(r8), pointer :: begwb(:)           ! water mass begining of the time step
   217)     real(r8), pointer :: fsa(:)             ! solar radiation absorbed (total) (W/m**2)
   218)     real(r8), pointer :: fsr(:)             ! solar radiation reflected (W/m**2)
   219)     real(r8), pointer :: eflx_lwrad_out(:)  ! emitted infrared (longwave) radiation (W/m**2)
   220)     real(r8), pointer :: eflx_lwrad_net(:)  ! net infrared (longwave) rad (W/m**2) [+ = to atm]
   221)     real(r8), pointer :: sabv(:)            ! solar radiation absorbed by vegetation (W/m**2)
   222)     real(r8), pointer :: sabg(:)            ! solar radiation absorbed by ground (W/m**2)
   223)     real(r8), pointer :: eflx_sh_tot(:)     ! total sensible heat flux (W/m**2) [+ to atm]
   224)     real(r8), pointer :: eflx_sh_totg(:)    ! total sensible heat flux at grid level (W/m**2) [+ to atm]
   225)     real(r8), pointer :: eflx_dynbal(:)     ! energy conversion flux due to dynamic land cover change(W/m**2) [+ to atm]
   226)     real(r8), pointer :: eflx_lh_tot(:)     ! total latent heat flux (W/m8*2)  [+ to atm]
   227)     real(r8), pointer :: eflx_soil_grnd(:)  ! soil heat flux (W/m**2) [+ = into soil]
   228)     real(r8), pointer :: qflx_evap_tot(:)   ! qflx_evap_soi + qflx_evap_can + qflx_tran_veg
   229)     real(r8), pointer :: qflx_irrig(:)      ! irrigation flux (mm H2O /s)
   230)     real(r8), pointer :: qflx_surf(:)       ! surface runoff (mm H2O /s)
   231)     real(r8), pointer :: qflx_qrgwl(:)      ! qflx_surf at glaciers, wetlands, lakes
   232)     real(r8), pointer :: qflx_drain(:)      ! sub-surface runoff (mm H2O /s)
   233)     real(r8), pointer :: qflx_runoff(:)     ! total runoff (mm H2O /s)
   234)     real(r8), pointer :: qflx_runoffg(:)    ! total runoff at gridcell level inc land cover change flux (mm H2O /s)
   235)     real(r8), pointer :: qflx_liq_dynbal(:) ! liq runoff due to dynamic land cover change (mm H2O /s)
   236)     real(r8), pointer :: qflx_snwcp_ice(:)  ! excess snowfall due to snow capping (mm H2O /s) [+]`
   237)     real(r8), pointer :: qflx_glcice(:)     ! flux of new glacier ice (mm H2O /s) [+ if ice grows]
   238)     real(r8), pointer :: qflx_glcice_frz(:) ! ice growth (mm H2O/s) [+]
   239)     real(r8), pointer :: qflx_snwcp_iceg(:) ! excess snowfall due to snow cap inc land cover change flux (mm H20/s)

   241)     real(r8), pointer :: forc_solad(:,:)    ! direct beam radiation (vis=forc_sols , nir=forc_soll )
   242)     real(r8), pointer :: forc_solai(:,:)    ! diffuse radiation     (vis=forc_solsd, nir=forc_solld)
   243)     real(r8), pointer :: eflx_traffic_pft(:)    ! traffic sensible heat flux (W/m**2)
   244)     real(r8), pointer :: eflx_wasteheat_pft(:)  ! sensible heat flux from urban heating/cooling sources of waste heat (W/m**2)
   245)     real(r8), pointer :: canyon_hwr(:)      ! ratio of building height to street width
   246)     real(r8), pointer :: eflx_heat_from_ac_pft(:) !sensible heat flux put back into canyon due to removal by AC (W/m**2)
   247)     real(r8), pointer :: h2osno(:)             ! snow water (mm H2O)
   248)     real(r8), pointer :: h2osno_old(:)         ! snow water (mm H2O) at previous time step
   249)     real(r8), pointer :: qflx_dew_snow(:)      ! surface dew added to snow pack (mm H2O /s) [+]
   250)     real(r8), pointer :: qflx_sub_snow(:)      ! sublimation rate from snow pack (mm H2O /s) [+]
   251)     real(r8), pointer :: qflx_top_soil(:)      ! net water input into soil from top (mm/s)
   252)     real(r8), pointer :: qflx_dew_grnd(:)      ! ground surface dew formation (mm H2O /s) [+]
   253)     real(r8), pointer :: qflx_evap_grnd(:)     ! ground surface evaporation rate (mm H2O/s) [+]
   254)     real(r8), pointer :: qflx_prec_grnd(:)     ! water onto ground including canopy runoff [kg/(m2 s)]
   255)     real(r8), pointer :: qflx_snwcp_liq(:)     ! excess liquid water due to snow capping (mm H2O /s) [+]`
   256)     real(r8), pointer :: qflx_sl_top_soil(:)   ! liquid water + ice from layer above soil to top soil layer or sent to qflx_qrgwl (mm H2O/s)
   257)     integer , pointer :: snl(:)                ! number of snow layers
   258) !
   259) ! local pointers to original implicit out arguments
   260) !
   261)     real(r8), pointer :: errh2o(:)          ! water conservation error (mm H2O)
   262)     real(r8), pointer :: errsol(:)          ! solar radiation conservation error (W/m**2)
   263)     real(r8), pointer :: errlon(:)          ! longwave radiation conservation error (W/m**2)
   264)     real(r8), pointer :: errseb(:)          ! surface energy conservation error (W/m**2)
   265)     real(r8), pointer :: netrad(:)          ! net radiation (positive downward) (W/m**2)
   266)     real(r8), pointer :: errsoi_col(:)      ! column-level soil/lake energy conservation error (W/m**2)
   267)     real(r8), pointer :: snow_sources(:)    ! snow sources (mm H2O /s)
   268)     real(r8), pointer :: snow_sinks(:)      ! snow sinks (mm H2O /s)
   269)     real(r8), pointer :: errh2osno(:)       ! error in h2osno (kg m-2)
   270) !
   271) !EOP
   272) !
   273) ! !OTHER LOCAL VARIABLES:
   274)     integer  :: p,c,l,g                     ! indices
   275)     real(r8) :: dtime                       ! land model time step (sec)
   276)     integer  :: nstep                       ! time step number
   277)     logical  :: found                       ! flag in search loop
   278)     integer  :: indexp,indexc,indexl,indexg ! index of first found in search loop
   279)     real(r8) :: forc_rain_col(lbc:ubc)      ! column level rain rate [mm/s]
   280)     real(r8) :: forc_snow_col(lbc:ubc)      ! column level snow rate [mm/s]
   281) 
   282) !-----------------------------------------------------------------------
   283) 
   284)     ! Assign local pointers to derived type scalar members (gridcell-level)
   285) 
   286)     do_capsnow          => cps%do_capsnow
   287)     qflx_floodc         => cwf%qflx_floodc
   288)     qflx_snow_melt      => cwf%qflx_snow_melt
   289)     qflx_rain_grnd_col  => pwf_a%qflx_rain_grnd
   290)     qflx_snow_grnd_col  => pwf_a%qflx_snow_grnd
   291)     clandunit           => col%landunit
   292)     forc_rain           => clm_a2l%forc_rain
   293)     forc_snow           => clm_a2l%forc_snow
   294)     forc_lwrad          => clm_a2l%forc_lwrad
   295)     forc_solad          => clm_a2l%forc_solad
   296)     forc_solai          => clm_a2l%forc_solai
   297) 
   298)     ! Assign local pointers to derived type scalar members (landunit-level)
   299) 
   300)     ltype             => lun%itype
   301)     canyon_hwr        => lun%canyon_hwr
   302) 
   303)     ! Assign local pointers to derived type scalar members (column-level)
   304) 
   305)     ctype             => col%itype
   306)     cgridcell         => col%gridcell
   307)     cwtgcell          => col%wtgcell
   308)     endwb             => cwbal%endwb
   309)     begwb             => cwbal%begwb
   310)     qflx_irrig        => cwf%qflx_irrig
   311)     qflx_surf         => cwf%qflx_surf
   312)     qflx_qrgwl        => cwf%qflx_qrgwl
   313)     qflx_drain        => cwf%qflx_drain
   314)     qflx_runoff       => cwf%qflx_runoff
   315)     qflx_snwcp_ice    => pwf_a%qflx_snwcp_ice
   316)     qflx_evap_tot     => pwf_a%qflx_evap_tot
   317)     qflx_glcice       => cwf%qflx_glcice
   318)     qflx_glcice_frz   => cwf%qflx_glcice_frz
   319)     errh2o            => cwbal%errh2o
   320)     errsoi_col        => cebal%errsoi
   321)     h2osno             => cws%h2osno
   322)     h2osno_old         => cws%h2osno_old
   323)     qflx_dew_snow      => pwf_a%qflx_dew_snow
   324)     qflx_sub_snow      => pwf_a%qflx_sub_snow
   325)     qflx_top_soil      => cwf%qflx_top_soil
   326)     qflx_evap_grnd     => pwf_a%qflx_evap_grnd
   327)     qflx_dew_grnd      => pwf_a%qflx_dew_grnd
   328)     qflx_prec_grnd     => pwf_a%qflx_prec_grnd
   329)     qflx_snwcp_liq     => pwf_a%qflx_snwcp_liq
   330)     qflx_sl_top_soil   => cwf%qflx_sl_top_soil
   331)     snow_sources       => cws%snow_sources
   332)     snow_sinks         => cws%snow_sinks
   333)     errh2osno          => cws%errh2osno
   334)     snl                => cps%snl
   335) 
   336)     ! Assign local pointers to derived type scalar members (pft-level)
   337) 
   338)     pgridcell         => pft%gridcell
   339)     plandunit         => pft%landunit
   340)     pwtgcell          => pft%wtgcell
   341)     fsa               => pef%fsa
   342)     fsr               => pef%fsr
   343)     eflx_lwrad_out    => pef%eflx_lwrad_out
   344)     eflx_lwrad_net    => pef%eflx_lwrad_net
   345)     sabv              => pef%sabv
   346)     sabg              => pef%sabg
   347)     eflx_sh_tot       => pef%eflx_sh_tot
   348)     eflx_lh_tot       => pef%eflx_lh_tot
   349)     eflx_soil_grnd    => pef%eflx_soil_grnd
   350)     errsol            => pebal%errsol
   351)     errseb            => pebal%errseb
   352)     errlon            => pebal%errlon
   353)     netrad            => pef%netrad
   354)     eflx_wasteheat_pft => pef%eflx_wasteheat_pft
   355)     eflx_heat_from_ac_pft => pef%eflx_heat_from_ac_pft
   356)     eflx_traffic_pft  => pef%eflx_traffic_pft
   357) 
   358)     ! Assign local pointers to derived type scalar members (gridcell-level)
   359) 
   360)     qflx_runoffg       => gwf%qflx_runoffg
   361)     qflx_liq_dynbal    => gwf%qflx_liq_dynbal
   362)     qflx_snwcp_iceg    => gwf%qflx_snwcp_iceg
   363)     qflx_ice_dynbal    => gwf%qflx_ice_dynbal
   364)     eflx_sh_totg       => gef%eflx_sh_totg
   365)     eflx_dynbal        => gef%eflx_dynbal
   366) 
   367)     ! Get step size and time step
   368) 
   369)     nstep = get_nstep()
   370)     dtime = get_step_size()
   371) 
   372)     ! Determine column level incoming snow and rain
   373)     ! Assume no incident precipitation on urban wall columns (as in Hydrology1Mod.F90).
   374) 
   375)     do c = lbc,ubc
   376)        g = cgridcell(c)
   377)        if (ctype(c) == icol_sunwall .or.  ctype(c) == icol_shadewall) then
   378)           forc_rain_col(c) = 0.
   379)           forc_snow_col(c) = 0.
   380)        else
   381)           forc_rain_col(c) = forc_rain(g)
   382)           forc_snow_col(c) = forc_snow(g)
   383)        end if
   384)     end do
   385) 
   386)     ! Water balance check
   387) 
   388)     do c = lbc, ubc
   389)        g = cgridcell(c)
   390)        l = clandunit(c)
   391) 
   392)        ! Note: Some glacier_mec cols may have zero weight
   393)        if (cwtgcell(c) > 0._r8 .or. ltype(l)==istice_mec)then
   394)           errh2o(c) = endwb(c) - begwb(c) &
   395)                - (forc_rain_col(c) + forc_snow_col(c) + qflx_irrig(c) + qflx_floodc(c) &
   396)                - qflx_evap_tot(c) - qflx_surf(c) &
   397)                - qflx_qrgwl(c) - qflx_drain(c) - qflx_snwcp_ice(c)) * dtime
   398) 
   399)           ! Suppose glc_dyntopo = T:   
   400)           ! (1) We have qflx_snwcp_ice = 0, and excess snow has been incorporated in qflx_glcice.  
   401)           !     This flux must be included here to complete the water balance.
   402)           ! (2) Meltwater from ice is allowed to run off and is included in qflx_qrgwl,
   403)           !     but the water content of the ice column has not changed (at least for now) because
   404)           !     an equivalent ice mass has been "borrowed" from the base of the column.  That
   405)           !     meltwater is included in qflx_glcice.
   406)           !
   407)           ! Note that qflx_glcice is only valid over ice_mec landunits; elsewhere it is spval
