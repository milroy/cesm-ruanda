   540)              if (.not.cg%converged(k)) then
   541) 
   542)                 ! ====================
   543)                 ! 2*np*np Flops
   544)                 ! ====================
   545) 
   546)                 do j=1,np
   547)                    do i=1,np
   548)                       gradp1 = gradp(i,j,1,k,ie)
   549)                       gradp2 = gradp(i,j,2,k,ie)
   550)                       gradp(i,j,1,k,ie) = metdet(i,j) * rmp(i,j) * &
   551)                              (Dinv(1,1,i,j)*gradp1 + Dinv(1,2,i,j)*gradp2)
   552)                       gradp(i,j,2,k,ie) = metdet(i,j) * rmp(i,j) * &
   553)                              (Dinv(2,1,i,j)*gradp1 + Dinv(2,2,i,j)*gradp2)
   554)                    end do
   555)                 end do
   556) 
   557)                 ! ================================================
   558)                 ! Compute  Pseudo Laplacian(p), store in div
   559)                 !JMD   2*np*np*(np + np) Flops 
   560)                 ! ================================================
   561) 
   562)                 div(:,:,k,ie) = divergence(gradp(:,:,:,k,ie),deriv)*rrearth
   563) 
   564)                 do j=1,np
   565)                    do i=1,np
   566)                       div(i,j,k,ie) = mp(i,j)*div(i,j,k,ie)
   567)                    end do
   568)                 end do
   569)              end if
   570)           end do
   571) 
   572)           kptr=0
   573)           call edgeVpack(edge1, div(1,1,1,ie), nlev, kptr, elem(ie)%desc)
   574) 
   575)        end do
   576) 
   577) 
   578)        call bndry_exchangeV(cg%hybrid,edge1)
   579) 
   580) 
   581)        ! ==========================================
   582)        ! compute Helmholtz operator, store in wrk3
   583)        !  4*np*np Flops
   584)        ! ==========================================
   585) 
   586)        do ie=nets,nete
   587)           ieptr=ie-nets+1
   588) 
   589)           rmp      => elem(ie)%rmp
   590)           rmetdet  => elem(ie)%rmetdet
   591)           metdet   => elem(ie)%metdet
   592) 
   593)           kptr=0
   594)           call edgeVunpack(edge1, div(1,1,1,ie), nlev, kptr, elem(ie)%desc)
   595) 
   596) #if (defined ELEMENT_OPENMP)
   597) !$omp parallel do private(k,i,j,iptr)
   598) #endif
   599)           do k=1,nlev
   600)              if (.not.cg%converged(k)) then
   601) 
   602)                 iptr=1
   603)                 do j=1,np
   604)                    do i=1,np
   605)                       cg%state(ieptr)%s(iptr,k) = metdet(i,j)*cg%state(ieptr)%z(iptr,k)+lambdasq(k)*rmp(i,j)*div(i,j,k,ie)
   606)                       iptr=iptr+1
   607)                    end do
   608)                 end do
   609) 
   610)              end if
   611)           end do
   612)        end do
   613) #ifdef DEBUGOMP
   614) #if (! defined ELEMENT_OPENMP)
   615) !$OMP BARRIER
   616) #endif
   617) #endif
   618)        call t_stopf('timer_helm')
   619) 
   620)     end do  ! CG solver while loop
   621) !    print *,'CG inter = ',cg%iter
   622)     ! ===============================
   623)     ! Converged! unpack wrk3 into x
   624)     ! ===============================
   625) 
   626)     do ie=nets,nete
   627)        ieptr=ie-nets+1
   628) #if (defined ELEMENT_OPENMP)
   629) !$omp parallel do private(k,i,j,iptr)
   630) #endif
   631)        do k=1,nlev
   632)           iptr=1
   633)           do j=1,np
   634)              do i=1,np
   635)                 !                x(i,j,k,ie)=cg%wrk3(iptr,k,ieptr)
   636)                 x(i,j,k,ie)=cg%state(ieptr)%x(iptr,k)
   637)                 iptr=iptr+1
   638)              end do
   639)           end do
   640)        end do
   641)     end do
   642)     call t_stopf('pcg_solver_nonstag')
   643) #endif
   644)   end function pcg_solver_nonstag

   864)   ! ================================================
   865)   ! solver_test:
   866)   !
   867)   !    L(x) = laplace_sphere_wk(x) = -< grad(PHI) dot grad(x) >
   868)   !    <   > = spheremp weighted inner-product
   869)   !    L is self-adjoint:  < L(x),y> = < x,L(y) >
   870)   !
   871)   ! solve for x:
   872)   !     <PHI,x> + a*L(x) =  < PHI, rhs >        
   873)   !     contant a ~ 10 dx^2 (typical scaling in semi-implicit solve)
   874)   !
   875)   ! 2D solve - but applied to every level  k=1,nlev
   876)   !
   877)   ! In matrix notation, following the convention in M.T. and A.L.'s 
   878)   ! "implicit.pdf" notes:
   879)   !     D QQ^t (M + a*L ) x = rhs
   880)   ! with:
   881)   !   M    = multiply by spheremp 
   882)   !   QQ^t = pack, exchange, unpack
   883)   !   D    = multiply by rspheremp  D = Q V^-1 Q^-L   
   884)   !          where V = the SEM diagonal mass matrix acting on vectors with no
   885)   !          duplicate degrees of freedom.  V does not appear in HOMME.
   886)   !   L is self adjoint w.r.t. M:    L(x) M y = x M L(y)
   887)   !
   888)   ! Note: if we solve laplace equation instead of Helmholz, we need to
   889)   ! ensure < rhs,1>=0
   890)   !
   891)   ! ================================================
   892)   subroutine solver_test(elem,edge1,red,hybrid,deriv,nets,nete)
   893)     use dimensions_mod, only : nlev, np,npsq
   894)     use element_mod, only : element_t
   895)     use reduction_mod, only : reductionbuffer_ordered_1d_t
   896)     use cg_mod, only : cg_t, congrad, cg_create
   897)     use edge_mod, only : edgebuffer_t, edgevpack, edgevunpack!, edgerotate
   898)     use derivative_mod, only : derivative_t, laplace_sphere_wk
   899)     use control_mod, only : maxits, while_iter, tol, precon_method
   900)     use physical_constants, only : rrearth, dd_pi, rearth, omega
   901)     use bndry_mod, only : bndry_exchangeV
   902)     use linear_algebra_mod, only : matvec
   903)     use parallel_mod, only : haltmp
   904)     use hybrid_mod, only : hybrid_t
   905)     use global_norms_mod, only : linf_snorm, l2_snorm
   906) 
   907)     integer, intent(in)  :: nets,nete
   908)     type(element_t), intent(in), target :: elem(:)
   909)     type (ReductionBuffer_ordered_1d_t)  :: red         ! CG reduction buffer   (shared memory)
   910)     type (derivative_t)               :: deriv          ! non staggered derivative struct     (private)
   911)     type (hybrid_t)             :: hybrid
   912)     type (EdgeBuffer_t)               :: edge1          ! Laplacian divergence edge buffer (shared memory)
   913) 
   914) 
   915)     ! ===========
   916)     ! Local
   917)     ! ===========
   918)     type (cg_t)                       :: cg             ! conjugate gradient    (private)
   919)     real (kind=real_kind) :: LHS(np,np,nlev,nets:nete)
   920)     real (kind=real_kind) :: RHS(np,np,nlev,nets:nete)
   921)     real (kind=real_kind) :: sol(np,np,nlev,nets:nete)   ! exact solution
   922)     real (kind=real_kind) :: solver_wts(npsq,nete-nets+1)
   923)     real (kind=real_kind) :: x(np,np)
   924)     real (kind=real_kind) :: alambda = 10*250e3**2      ! low res test, dx=250km grid spacing
   925) 
   926)     integer :: ie
   927)     integer :: i,j,k
   928)     integer :: kptr
   929)     integer :: iptr
   930)     integer :: ieptr
   931)     real (kind=real_kind) :: snlat,cslat,cslon,snlon,xc,yc,zc, res, res_sol
   932) 
   933)     if (hybrid%masterthread) print *,'creating manufactured solution'
   934)     do ie=nets,nete
   935)        iptr=1
   936)        do j=1,np
   937)           do i=1,np
   938)              solver_wts(iptr,ie-nets+1) = elem(ie)%spheremp(i,j)
   939)              iptr=iptr+1
   940)           end do
   941)        end do
   942)     end do
   943)     call cg_create(cg, npsq, nlev, nete-nets+1, hybrid, 0, solver_wts)
   944) 
   945)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   946)     ! make up an exact solution
   947)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   948)     do ie=nets,nete
   949)        ieptr=ie-nets+1
   950)        do k=1,nlev
   951)           iptr=1
   952)           do j=1,np
   953)              do i=1,np
   954)                 snlat = SIN(elem(ie)%spherep(i,j)%lat)
   955)                 cslat = COS(elem(ie)%spherep(i,j)%lat)
   956)                 snlon = SIN(elem(ie)%spherep(i,j)%lon)
   957)                 cslon = COS(elem(ie)%spherep(i,j)%lon)
   958)   
   959)                 xc = cslat*cslon
   960)                 yc = cslat*snlon
   961)                 zc = snlat
   962) 
   963)                 ! take a couple of low-freq spherical harmonics for the solution
   964)                 sol(i,j,k,ie) = 1*xc + 2*yc + 3*zc + 4*xc*yc + 5*xc*zc + 6*yc*zc + &
   965)                      7*(xc*yc*zc) 
   966) 
   967)              end do
   968)           end do
   969)        end do
   970)     end do
   971) 
   972)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   973)     ! compute the RHS from our exact solution
   974)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   975)     do ie=nets,nete
   976)        do k=1,nlev
   977)           RHS(:,:,k,ie)=elem(ie)%spheremp(:,:)*sol(:,:,k,ie) + &
   978)                alambda*laplace_sphere_wk(sol(:,:,k,ie),deriv,elem(ie),var_coef=.false.)
   979)           call edgeVpack(edge1, RHS(1,1,1,ie), nlev, 0, elem(ie)%desc)
   980)        end do
   981)     end do
   982)     call bndry_exchangeV(cg%hybrid,edge1)
   983)     do ie=nets,nete
   984)        ! unpack RHS
   985)        call edgeVunpack(edge1, RHS(1,1,1,ie), nlev, 0, elem(ie)%desc)
   986)        do k=1,nlev
   987)           RHS(:,:,k,ie)=RHS(:,:,k,ie)*elem(ie)%rspheremp(:,:)
   988)        enddo
   989)        !
   990)        !  Initialize CG solver:  set %r = residual from initial guess
   991)        !  if initial guess = 0, then we take %r=RHS
   992)        !
   993)        ieptr=ie-nets+1
   994)        do k=1,nlev
   995)           iptr=1
   996)           do j=1,np
   997)              do i=1,np
   998)                 cg%state(ieptr)%r(iptr,k) = rhs(i,j,k,ie)
   999)                 iptr=iptr+1
  1000)              enddo
  1001)           enddo
  1002)        enddo
  1003)     enddo
  1004)     
  1005)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1006)     ! Solver Loop
  1007)     ! in this version, we keep the residual C0 by DSSing the LHS
  1008)     ! and initializiont with a C0 RHS.  
  1009)     ! the update x, based on the last residual, will already be C0
  1010)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1011)     cg%debug_level=1  ! 2=output every iterations
  1012)     maxits = 250
  1013)     tol=1d-7
  1014)     if (hybrid%masterthread) print *,'running solver V1 (C0 RHS) tol=',tol
  1015)     do while (congrad(cg,red,maxits,tol))
  1016)        do ie=nets,nete
  1017)           ieptr=ie-nets+1
  1018)           do k=1,nlev
  1019)              ! apply preconditioner here:
  1020)              cg%state(ieptr)%z(:,k) = cg%state(ieptr)%r(:,k)
  1021)              
  1022)              !reshape(cg%state(ieptr)%z(:,k),(/np,np/))
  1023)              iptr=1
  1024)              do j=1,np
  1025)                 do i=1,np
  1026)                    x(i,j) = cg%state(ieptr)%z(iptr,k)
  1027)                    iptr=iptr+1
  1028)                 enddo
  1029)              enddo
  1030)              
  1031)              ! solve x + laplace(x)
  1032)              ! weak laplace operator already includes mass
  1033)              ! so only multiply x by mass;
  1034)              LHS(:,:,k,ie)=elem(ie)%spheremp(:,:)*x(:,:) + &
  1035)                   alambda*laplace_sphere_wk(x,deriv,elem(ie),var_coef=.false.)
  1036)              
  1037)           end do
  1038)           call edgeVpack(edge1, LHS(1,1,1,ie), nlev, 0, elem(ie)%desc)
  1039)        end do
  1040)        call bndry_exchangeV(cg%hybrid,edge1)
  1041)        do ie=nets,nete
  1042)           ! unpack LHS
  1043)           call edgeVunpack(edge1, LHS(1,1,1,ie), nlev, 0, elem(ie)%desc)
  1044)           do k=1,nlev
  1045)              LHS(:,:,k,ie)=LHS(:,:,k,ie)*elem(ie)%rspheremp(:,:)
  1046)           enddo
  1047) 
  1048)           ieptr=ie-nets+1
  1049)           do k=1,nlev
  1050)              iptr=1
