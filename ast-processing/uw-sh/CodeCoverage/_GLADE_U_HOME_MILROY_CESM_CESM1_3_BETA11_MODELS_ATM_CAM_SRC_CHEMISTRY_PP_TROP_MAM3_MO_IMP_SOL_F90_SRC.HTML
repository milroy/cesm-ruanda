   438)   subroutine imp_sol( base_sol, reaction_rates, het_rates, extfrc, delt, &
   439)        xhnm, ncol, lchnk, ltrop, o3s_loss )
   440)     !-----------------------------------------------------------------------
   441)     ! ... imp_sol advances the volumetric mixing ratio
   442)     ! forward one time step via the fully implicit euler scheme.
   443)     ! this source is meant for small l1 cache machines such as
   444)     ! the intel pentium and itanium cpus
   445)     !-----------------------------------------------------------------------
   446)     use chem_mods, only : rxntot, extcnt, nzcnt, permute, cls_rxt_cnt
   447)     use mo_tracname, only : solsym
   448)     use ppgrid, only : pver
   449)     use mo_lin_matrix, only : linmat
   450)     use mo_nln_matrix, only : nlnmat
   451)     use mo_lu_factor, only : lu_fac
   452)     use mo_lu_solve, only : lu_slv
   453)     use mo_prod_loss, only : imp_prod_loss
   454)     use mo_indprd, only : indprd
   455)     use time_manager, only : get_nstep
   456)     use cam_history, only : outfld
   457)     implicit none
   458)     !-----------------------------------------------------------------------
   459)     ! ... dummy args
   460)     !-----------------------------------------------------------------------
   461)     integer, intent(in) :: ncol ! columns in chunck
   462)     integer, intent(in) :: lchnk ! chunk id
   463)     real(r8), intent(in) :: delt ! time step (s)
   464)     real(r8), intent(in) :: reaction_rates(ncol,pver,max(1,rxntot)), & ! rxt rates (1/cm^3/s)
   465)          extfrc(ncol,pver,max(1,extcnt)), & ! external in-situ forcing (1/cm^3/s)
   466)          het_rates(ncol,pver,max(1,gas_pcnst)) ! washout rates (1/s)
   467)     real(r8), intent(inout) :: base_sol(ncol,pver,gas_pcnst) ! species mixing ratios (vmr)
   468)     real(r8), intent(in) :: xhnm(ncol,pver)
   469)     integer, intent(in) :: ltrop(ncol) ! chemistry troposphere boundary (index)
   470)     real(r8), optional, intent(out) :: o3s_loss(ncol,pver)
   471)     !-----------------------------------------------------------------------
   472)     ! ... local variables
   473)     !-----------------------------------------------------------------------
   474)     integer :: nr_iter, &
   475)          lev, &
   476)          i, &
   477)          j, &
   478)          k, l, &
   479)          m
   480)     integer :: fail_cnt, cut_cnt, stp_con_cnt
   481)     integer :: nstep
   482)     real(r8) :: interval_done, dt, dti, wrk
   483)     real(r8) :: max_delta(max(1,clscnt4))
   484)     real(r8) :: sys_jac(max(1,nzcnt))
   485)     real(r8) :: lin_jac(max(1,nzcnt))
   486)     real(r8), dimension(max(1,clscnt4)) :: &
   487)          solution, &
   488)          forcing, &
   489)          iter_invariant, &
   490)          prod, &
   491)          loss
   492)     real(r8) :: lrxt(max(1,rxntot))
   493)     real(r8) :: lsol(max(1,gas_pcnst))
   494)     real(r8) :: lhet(max(1,gas_pcnst))
   495)     real(r8), dimension(ncol,pver,max(1,clscnt4)) :: &
   496)          ind_prd
   497)     logical :: convergence
   498)     logical :: frc_mask, iter_conv
   499)     logical :: converged(max(1,clscnt4))
   500)     real(r8), dimension(ncol,pver,max(1,clscnt4)) :: prod_out, loss_out
   501)     real(r8), dimension(ncol,pver) :: prod_hydrogen_peroxides_out
   502)     if (present(o3s_loss)) then
   503)        o3s_loss(:,:) = 0._r8
   504)     endif
   505)     prod_out(:,:,:) = 0._r8
   506)     loss_out(:,:,:) = 0._r8
   507)     prod_hydrogen_peroxides_out(:,:) = 0._r8
   508)     solution(:) = 0._r8
   509)     !-----------------------------------------------------------------------
   510)     ! ... class independent forcing
   511)     !-----------------------------------------------------------------------
   512)     if( cls_rxt_cnt(1,4) > 0 .or. extcnt > 0 ) then
   513)        call indprd( 4, ind_prd, clscnt4, base_sol, extfrc, &
   514)             reaction_rates, ncol )
   515)     else
   516)        do m = 1,max(1,clscnt4)
   517)           ind_prd(:,:,m) = 0._r8
   518)        end do
   519)     end if
   520)     level_loop : do lev = 1,pver
   521)        column_loop : do i = 1,ncol
   522)           IF (lev <= ltrop(i)) CYCLE column_loop
   523)           !-----------------------------------------------------------------------
   524)           ! ... transfer from base to local work arrays
   525)           !-----------------------------------------------------------------------
   526)           do m = 1,rxntot
   527)              lrxt(m) = reaction_rates(i,lev,m)
   528)           end do
   529)           if( gas_pcnst > 0 ) then
   530)              do m = 1,gas_pcnst
   531)                 lhet(m) = het_rates(i,lev,m)
   532)              end do
   533)           end if
   534)           !-----------------------------------------------------------------------
   535)           ! ... time step loop
   536)           !-----------------------------------------------------------------------
   537)           dt = delt
   538)           cut_cnt = 0
   539)           fail_cnt = 0
   540)           stp_con_cnt = 0
   541)           interval_done = 0._r8
   542)           time_step_loop : do
   543)              dti = 1._r8 / dt
   544)              !-----------------------------------------------------------------------
   545)              ! ... transfer from base to local work arrays
   546)              !-----------------------------------------------------------------------
   547)              do m = 1,gas_pcnst
   548)                 lsol(m) = base_sol(i,lev,m)
   549)              end do
   550)              !-----------------------------------------------------------------------
   551)              ! ... transfer from base to class array
   552)              !-----------------------------------------------------------------------
   553)              do k = 1,clscnt4
   554)                 j = clsmap(k,4)
   555)                 m = permute(k,4)
   556)                 solution(m) = lsol(j)
   557)              end do
   558)              !-----------------------------------------------------------------------
   559)              ! ... set the iteration invariant part of the function f(y)
   560)              !-----------------------------------------------------------------------
   561)              if( cls_rxt_cnt(1,4) > 0 .or. extcnt > 0 ) then
   562)                 do m = 1,clscnt4
   563)                    iter_invariant(m) = dti * solution(m) + ind_prd(i,lev,m)
   564)                 end do
   565)              else
   566)                 do m = 1,clscnt4
   567)                    iter_invariant(m) = dti * solution(m)
   568)                 end do
   569)              end if
   570)              !-----------------------------------------------------------------------
   571)              ! ... the linear component
   572)              !-----------------------------------------------------------------------
   573)              !if( cls_rxt_cnt(2,4) > 0 ) then
   574)                 call linmat( lin_jac, lsol, lrxt, lhet )
   575)              !end if
   576)              !=======================================================================
   577)              ! the newton-raphson iteration for f(y) = 0
   578)              !=======================================================================
   579)              iter_loop : do nr_iter = 1,itermax
   580)                 !-----------------------------------------------------------------------
   581)                 ! ... the non-linear component
   582)                 !-----------------------------------------------------------------------
   583)                 if( factor(nr_iter) ) then
   584)                    call nlnmat( sys_jac, lsol, lrxt, lin_jac, dti )
   585)                    !-----------------------------------------------------------------------
   586)                    ! ... factor the "system" matrix
   587)                    !-----------------------------------------------------------------------
   588)                    call lu_fac( sys_jac )
   589)                 end if
   590)                 !-----------------------------------------------------------------------
   591)                 ! ... form f(y)
   592)                 !-----------------------------------------------------------------------
   593)                 call imp_prod_loss( prod, loss, lsol, lrxt, lhet )
   594)                 do m = 1,clscnt4
   595)                    forcing(m) = solution(m)*dti - (iter_invariant(m) + prod(m) - loss(m))
   596)                 end do
   597)                 !-----------------------------------------------------------------------
   598)                 ! ... solve for the mixing ratio at t(n+1)
   599)                 !-----------------------------------------------------------------------
   600)                 call lu_slv( sys_jac, forcing )
   601)                 do m = 1,clscnt4
   602)                    solution(m) = solution(m) + forcing(m)
   603)                 end do
   604)                 !-----------------------------------------------------------------------
   605)                 ! ... convergence measures
   606)                 !-----------------------------------------------------------------------
   607)                 if( nr_iter > 1 ) then
   608)                    do k = 1,clscnt4
   609)                       m = permute(k,4)
   610)                       if( abs(solution(m)) > 1.e-20_r8 ) then
   611)                          max_delta(k) = abs( forcing(m)/solution(m) )
   612)                       else
   613)                          max_delta(k) = 0._r8
   614)                       end if
   615)                    end do
   616)                 end if
   617)                 !-----------------------------------------------------------------------
   618)                 ! ... limit iterate
   619)                 !-----------------------------------------------------------------------
   620)                 where( solution(:) < 0._r8 )
   621)                    solution(:) = 0._r8
   622)                 endwhere
   623)                 !-----------------------------------------------------------------------
   624)                 ! ... transfer latest solution back to work array
   625)                 !-----------------------------------------------------------------------
   626)                 do k = 1,clscnt4
   627)                    j = clsmap(k,4)
   628)                    m = permute(k,4)
   629)                    lsol(j) = solution(m)
   630)                 end do
   631)                 !-----------------------------------------------------------------------
   632)                 ! ... check for convergence
   633)                 !-----------------------------------------------------------------------
   634)                 converged(:) = .true.
   635)                 if( nr_iter > 1 ) then
   636)                    do k = 1,clscnt4
   637)                       m = permute(k,4)
   638)                       frc_mask = abs( forcing(m) ) > small
   639)                       if( frc_mask ) then
   640)                          converged(k) = abs(forcing(m)) <= epsilon(k)*abs(solution(m))
   641)                       else
   642)                          converged(k) = .true.
   643)                       end if
   644)                    end do
   645)                    convergence = all( converged(:) )
   646)                    if( convergence ) then
   647)                       exit
   648)                    end if
   649)                 end if
   650)              end do iter_loop
   651)              !-----------------------------------------------------------------------
   652)              ! ... check for newton-raphson convergence
   653)              !-----------------------------------------------------------------------
   654)              if( .not. convergence ) then
   655)                 !-----------------------------------------------------------------------
   656)                 ! ... non-convergence
   657)                 !-----------------------------------------------------------------------
   658)                 fail_cnt = fail_cnt + 1
   659)                 nstep = get_nstep()
   660)                 write(iulog,'('' imp_sol: Time step '',1p,e21.13,'' failed to converge @ (lchnk,lev,col,nstep) = '',4i6)') &
   661)                      dt,lchnk,lev,i,nstep
   662)                 stp_con_cnt = 0
   663)                 if( cut_cnt < cut_limit ) then
   664)                    cut_cnt = cut_cnt + 1
   665)                    if( cut_cnt < cut_limit ) then
   666)                       dt = .5_r8 * dt
   667)                    else
   668)                       dt = .1_r8 * dt
   669)                    end if
   670)                    cycle time_step_loop
   671)                 else
   672)                    write(iulog,'('' imp_sol: Failed to converge @ (lchnk,lev,col,nstep,dt,time) = '',4i6,1p,2e21.13)') &
   673)                         lchnk,lev,i,nstep,dt,interval_done+dt
   674)                    do m = 1,clscnt4
   675)                       if( .not. converged(m) ) then
   676)                          write(iulog,'(1x,a8,1x,1pe10.3)') solsym(clsmap(m,4)), max_delta(m)
   677)                       end if
   678)                    end do
   679)                 end if
   680)              end if
   681)              !-----------------------------------------------------------------------
   682)              ! ... check for interval done
   683)              !-----------------------------------------------------------------------
   684)              interval_done = interval_done + dt
   685)              if( abs( delt - interval_done ) <= .0001_r8 ) then
   686)                 if( fail_cnt > 0 ) then
   687)                    write(iulog,*) 'imp_sol : @ (lchnk,lev,col) = ',lchnk,lev,i,' failed ',fail_cnt,' times'
   688)                 end if
   689)                 exit time_step_loop
   690)              else
   691)                 !-----------------------------------------------------------------------
   692)                 ! ... transfer latest solution back to base array
   693)                 !-----------------------------------------------------------------------
   694)                 if( convergence ) then
   695)                    stp_con_cnt = stp_con_cnt + 1
   696)                 end if
   697)                 do m = 1,gas_pcnst
   698)                    base_sol(i,lev,m) = lsol(m)
   699)                 end do
   700)                 if( stp_con_cnt >= 2 ) then
   701)                    dt = 2._r8*dt
   702)                    stp_con_cnt = 0
   703)                 end if
   704)                 dt = min( dt,delt-interval_done )
   705)                 ! write(iulog,'('' imp_sol: New time step '',1p,e21.13)') dt
   706)              end if
   707)           end do time_step_loop
   708)           !-----------------------------------------------------------------------
   709)           ! ... Transfer latest solution back to base array
   710)           !-----------------------------------------------------------------------
   711)           cls_loop: do k = 1,clscnt4
   712)              j = clsmap(k,4)
   713)              m = permute(k,4)
   714)              base_sol(i,lev,j) = solution(m)
   715)           end do cls_loop
   716)           !-----------------------------------------------------------------------
   717)           ! ... Prod/Loss history buffers...
   718)           !-----------------------------------------------------------------------
   719)           cls_loop2: do k = 1,clscnt4
   720)              j = clsmap(k,4)
   721)              m = permute(k,4)
   722)              has_o3_chem: if( ( full_ozone_chem .or. reduced_ozone_chem .or. middle_atm_chem ) .and. &
   723)                               (j == ox_ndx .or. j == o3a_ndx )) then
   724)                 if( o1d_ndx < 1 ) then
   725)                    loss_out(i,lev,k) = reaction_rates(i,lev,ox_l1_ndx)
   726)                 else
   727)                    if (j == ox_ndx) &
   728)                       loss_out(i,lev,k) = reaction_rates(i,lev,ox_l1_ndx) * base_sol(i,lev,o1d_ndx) &
   729)                                           / base_sol(i,lev,ox_ndx)
   730)                    if (j == o3a_ndx) &
   731)                       loss_out(i,lev,k) = reaction_rates(i,lev,ox_l1_ndx) * base_sol(i,lev,o1da_ndx)&
   732)                                           / base_sol(i,lev,o3a_ndx)
   733)                    if ( h2o_ndx > 0 ) &
   734)                       loss_out(i,lev,k) = loss_out(i,lev,k) * base_sol(i,lev,h2o_ndx)
   735)                 end if
   736)                 if ( full_ozone_chem ) then
   737)                    prod_out(i,lev,k) = reaction_rates(i,lev,ox_p1_ndx) * base_sol(i,lev,ho2_ndx) &
   738)                         + reaction_rates(i,lev,ox_p2_ndx) * base_sol(i,lev,ch3o2_ndx) &
   739)                         + reaction_rates(i,lev,ox_p3_ndx) * base_sol(i,lev,po2_ndx) &
   740)                         + reaction_rates(i,lev,ox_p4_ndx) * base_sol(i,lev,ch3co3_ndx) &
   741)                         + reaction_rates(i,lev,ox_p5_ndx) * base_sol(i,lev,c2h5o2_ndx) &
   742)                         + .92_r8* reaction_rates(i,lev,ox_p6_ndx) * base_sol(i,lev,isopo2_ndx) &
   743)                         + reaction_rates(i,lev,ox_p7_ndx) * base_sol(i,lev,macro2_ndx) &
   744)                         + reaction_rates(i,lev,ox_p8_ndx) * base_sol(i,lev,mco3_ndx) &
   745)                         + reaction_rates(i,lev,ox_p9_ndx) * base_sol(i,lev,c3h7o2_ndx) &
   746)                         + reaction_rates(i,lev,ox_p10_ndx)* base_sol(i,lev,ro2_ndx) &
   747)                         + reaction_rates(i,lev,ox_p11_ndx)* base_sol(i,lev,xo2_ndx) &
   748)                         + .9_r8*reaction_rates(i,lev,ox_p12_ndx)*base_sol(i,lev,tolo2_ndx) &
   749)                         + reaction_rates(i,lev,ox_p13_ndx)*base_sol(i,lev,terpo2_ndx)&
   750)                         + .9_r8*reaction_rates(i,lev,ox_p14_ndx)*base_sol(i,lev,alko2_ndx) &
   751)                         + reaction_rates(i,lev,ox_p15_ndx)*base_sol(i,lev,eneo2_ndx) &
   752)                         + reaction_rates(i,lev,ox_p16_ndx)*base_sol(i,lev,eo2_ndx) &
   753)                         + reaction_rates(i,lev,ox_p17_ndx)*base_sol(i,lev,meko2_ndx)
   754)                    loss_out(i,lev,k) = loss_out(i,lev,k) &
   755)                         + reaction_rates(i,lev,ox_l2_ndx) * base_sol(i,lev,oh_ndx) &
   756)                         + reaction_rates(i,lev,ox_l3_ndx) * base_sol(i,lev,ho2_ndx) &
   757)                         + reaction_rates(i,lev,ox_l6_ndx) * base_sol(i,lev,c2h4_ndx) &
   758)                         + reaction_rates(i,lev,ox_l4_ndx) * base_sol(i,lev,c3h6_ndx) &
   759)                         + .9_r8* reaction_rates(i,lev,ox_l5_ndx) * base_sol(i,lev,isop_ndx) &
   760)                         + .8_r8*( reaction_rates(i,lev,ox_l7_ndx) * base_sol(i,lev,mvk_ndx) &
   761)                         + reaction_rates(i,lev,ox_l8_ndx) * base_sol(i,lev,macro2_ndx)) &
   762)                         + .235_r8*reaction_rates(i,lev,ox_l9_ndx) * base_sol(i,lev,c10h16_ndx)
   763)                 else if ( reduced_ozone_chem ) then
   764)                    prod_out(i,lev,k) = reaction_rates(i,lev,ox_p1_ndx ) * base_sol(i,lev,ho2_ndx) &
   765)                         + reaction_rates(i,lev,ox_p2_ndx ) * base_sol(i,lev,ch3o2_ndx) &

   767)                         + reaction_rates(i,lev,ox_p11_ndx ) * base_sol(i,lev,xo2_ndx)

   770)                         + reaction_rates(i,lev,ox_l3_ndx) * base_sol(i,lev,ho2_ndx) &
   771)                         + .9_r8* reaction_rates(i,lev,ox_l5_ndx) * base_sol(i,lev,isop_ndx) &
   772)                         + reaction_rates(i,lev,ox_l6_ndx) * base_sol(i,lev,c2h4_ndx) &
   773)                         + reaction_rates(i,lev,ox_l7_ndx) * base_sol(i,lev,ole_ndx)

   789)                    endif
   790)                    loss_out(i,lev,k) = loss_out(i,lev,k) * base_sol(i,lev,ox_ndx)
   791)                    prod_out(i,lev,k) = prod_out(i,lev,k) * base_sol(i,lev,no_ndx)
   792)                 else if (j == o3a_ndx) then
   793)                    loss_out(i,lev,k) = loss_out(i,lev,k) &
   794)                         + ( reaction_rates(i,lev,usr4a_ndx) * base_sol(i,lev,xno2_ndx) * base_sol(i,lev,oh_ndx) &
