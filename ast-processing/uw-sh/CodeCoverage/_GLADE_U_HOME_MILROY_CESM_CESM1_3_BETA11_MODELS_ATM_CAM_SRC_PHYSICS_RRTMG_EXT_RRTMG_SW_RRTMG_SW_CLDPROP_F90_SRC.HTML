    18) ! ------- Modules -------
    19) 
    20)       use shr_kind_mod, only: r8 => shr_kind_r8
    21) 
    22) !      use parkind, only : jpim, jprb
    23)       use parrrsw, only : nbndsw, jpband, jpb1, jpb2
    24)       use rrsw_cld, only : extliq1, ssaliq1, asyliq1, &
    25)                            extice2, ssaice2, asyice2, &
    26)                            extice3, ssaice3, asyice3, fdlice3, &
    27)                            abari, bbari, cbari, dbari, ebari, fbari
    28)       use rrsw_wvn, only : wavenum1, wavenum2
    29)       use rrsw_vsn, only : hvrcld, hnamcld
    30) 
    31)       implicit none
    32) 
    33)       contains
    34) 
    35) ! ----------------------------------------------------------------------------
    36)       subroutine cldprop_sw(nlayers, inflag, iceflag, liqflag, cldfrac, &
    37)                             tauc, ssac, asmc, ciwp, clwp, rei, dge, rel, &
    38)                             taucldorig, taucloud, ssacloud, asmcloud)
    39) ! ----------------------------------------------------------------------------
    40) 
    41) ! Purpose: Compute the cloud optical properties for each cloudy layer.
    42) ! Note: Only inflag = 0 and inflag=2/liqflag=1/iceflag=2,3 are available;
    43) ! (Hu & Stamnes, Key, and Fu) are implemented.
    44) 
    45) ! ------- Input -------
    46) 
    47)       integer, intent(in) :: nlayers         ! total number of layers
    48)       integer, intent(in) :: inflag          ! see definitions
    49)       integer, intent(in) :: iceflag         ! see definitions
    50)       integer, intent(in) :: liqflag         ! see definitions
    51) 
    52)       real(kind=r8), intent(in) :: cldfrac(:)         ! cloud fraction
    53)                                                         !    Dimensions: (nlayers)
    54)       real(kind=r8), intent(in) :: ciwp(:)            ! cloud ice water path
    55)                                                         !    Dimensions: (nlayers)
    56)       real(kind=r8), intent(in) :: clwp(:)            ! cloud liquid water path
    57)                                                         !    Dimensions: (nlayers)
    58)       real(kind=r8), intent(in) :: rei(:)             ! cloud ice particle effective radius (microns)
    59)                                                         !    Dimensions: (nlayers)
    60)       real(kind=r8), intent(in) :: dge(:)             ! cloud ice particle generalized effective size (microns)
    61)                                                         !    Dimensions: (nlayers)
    62)       real(kind=r8), intent(in) :: rel(:)             ! cloud liquid particle effective radius (microns)
    63)                                                         !    Dimensions: (nlayers)
    64)       real(kind=r8), intent(in) :: tauc(:,:)          ! cloud optical depth
    65)                                                         !    Dimensions: (nbndsw,nlayers)
    66)       real(kind=r8), intent(in) :: ssac(:,:)          ! single scattering albedo
    67)                                                         !    Dimensions: (nbndsw,nlayers)
    68)       real(kind=r8), intent(in) :: asmc(:,:)          ! asymmetry parameter
    69)                                                         !    Dimensions: (nbndsw,nlayers)
    70) 
    71) ! ------- Output -------
    72) 
    73)       real(kind=r8), intent(out) :: taucloud(:,:)     ! cloud optical depth (delta scaled)
    74)                                                         !    Dimensions: (nlayers,jpband)
    75)       real(kind=r8), intent(out) :: taucldorig(:,:)   ! cloud optical depth (non-delta scaled)
    76)                                                         !    Dimensions: (nlayers,jpband)
    77)       real(kind=r8), intent(out) :: ssacloud(:,:)     ! single scattering albedo (delta scaled)
    78)                                                         !    Dimensions: (nlayers,jpband)
    79)       real(kind=r8), intent(out) :: asmcloud(:,:)     ! asymmetry parameter (delta scaled)
    80)                                                         !    Dimensions: (nlayers,jpband)
    81) 
    82) ! ------- Local -------
    83) 
    84) !      integer :: ncbands
    85)       integer :: ib, ib1, ib2, lay, istr, index, icx
    86) 
    87)       real(kind=r8), parameter :: eps = 1.e-06_r8     ! epsilon
    88)       real(kind=r8), parameter :: cldmin = 1.e-80_r8  ! minimum value for cloud quantities
    89)       real(kind=r8) :: cwp                            ! total cloud water path
    90)       real(kind=r8) :: radliq                         ! cloud liquid droplet radius (microns)
    91)       real(kind=r8) :: radice                         ! cloud ice effective radius (microns)
    92)       real(kind=r8) :: dgeice                         ! cloud ice generalized effective size (microns)
    93)       real(kind=r8) :: factor
    94)       real(kind=r8) :: fint
    95)       real(kind=r8) :: tauctot(nlayers)               ! band integrated cloud optical depth
    96) 
    97)       real(kind=r8) :: taucldorig_a, ssacloud_a, taucloud_a, ffp, ffp1, ffpssa
    98)       real(kind=r8) :: tauiceorig, scatice, ssaice, tauice, tauliqorig, scatliq, ssaliq, tauliq
    99) 
   100)       real(kind=r8) :: fdelta(jpb1:jpb2)
   101)       real(kind=r8) :: extcoice(jpb1:jpb2), gice(jpb1:jpb2)
   102)       real(kind=r8) :: ssacoice(jpb1:jpb2), forwice(jpb1:jpb2)
   103)       real(kind=r8) :: extcoliq(jpb1:jpb2), gliq(jpb1:jpb2)
   104)       real(kind=r8) :: ssacoliq(jpb1:jpb2), forwliq(jpb1:jpb2)
   105) 
   106) ! Initialize
   107) 
   108)       hvrcld = '$Revision: 1.4 $'
   109) 
   110) !      ncbands = 29
   111)       ib1 = jpb1
   112)       ib2 = jpb2
   113)       tauctot(:) = 0._r8
   114) 
   115)       do lay = 1, nlayers
   116)          do ib = ib1 , ib2
   117)             taucloud(lay,ib) = 0.0_r8
   118)             ssacloud(lay,ib) = 1.0_r8
   119)             asmcloud(lay,ib) = 0.0_r8
   120)             tauctot(lay) = tauctot(lay) + tauc(ib-15,lay)
   121)          enddo
   122)       enddo
   123) 
   124) ! Main layer loop
   125)       do lay = 1, nlayers
   126) 
   127)          cwp = ciwp(lay) + clwp(lay)
   128)          if (cldfrac(lay) .ge. cldmin .and. &
   129)             (cwp .ge. cldmin .or. tauctot(lay) .ge. cldmin)) then
   130) 
   131) ! (inflag=0): Cloud optical properties input directly
   132) ! Cloud optical properties already defined in tauc, ssac, asmc are unscaled;
   133) ! Apply delta-M scaling here
   134)             if (inflag .eq. 0) then
   135) 
   136)                do ib = ib1 , ib2
   137)                   taucldorig_a = tauc(ib-15,lay)
   138)                   ffp = asmc(ib-15,lay)**2
   139)                   ffp1 = 1.0_r8 - ffp
   140)                   ffpssa = 1.0_r8 - ffp * ssac(ib-15,lay)
   141)                   ssacloud_a = ffp1 * ssac(ib-15,lay) / ffpssa
   142)                   taucloud_a = ffpssa * taucldorig_a
   143) 
   144)                   taucldorig(lay,ib) = taucldorig_a
   145)                   ssacloud(lay,ib) = ssacloud_a
   146)                   taucloud(lay,ib) = taucloud_a
   147)                   asmcloud(lay,ib) = (asmc(ib-15,lay) - ffp) / (ffp1)
   148)                enddo
   149) 
   150) ! (inflag=2): Separate treatement of ice clouds and water clouds.
   151)             elseif (inflag .eq. 2) then       
   152)                radice = rei(lay)
   153) 
   154) ! Calculation of absorption coefficients due to ice clouds.
   155)                if (ciwp(lay) .eq. 0.0_r8) then
   156)                   do ib = ib1 , ib2
   157)                      extcoice(ib) = 0.0_r8
   158)                      ssacoice(ib) = 0.0_r8
   159)                      gice(ib)     = 0.0_r8
   160)                      forwice(ib)  = 0.0_r8
   161)                   enddo
   162) 
   163) ! (iceflag = 1): 
   164) ! Note: This option uses Ebert and Curry approach for all particle sizes similar to
   165) ! CAM3 implementation, though this is somewhat ineffective for large ice particles
   166)                elseif (iceflag .eq. 1) then
   167)                   do ib = ib1, ib2
   168)                      if (wavenum2(ib) .gt. 1.43e04_r8) then
   169)                         icx = 1
   170)                      elseif (wavenum2(ib) .gt. 7.7e03_r8) then
   171)                         icx = 2
   172)                      elseif (wavenum2(ib) .gt. 5.3e03_r8) then
   173)                         icx = 3
   174)                      elseif (wavenum2(ib) .gt. 4.0e03_r8) then
   175)                         icx = 4
   176)                      elseif (wavenum2(ib) .ge. 2.5e03_r8) then
   177)                         icx = 5
   178)                      endif
   179)                      extcoice(ib) = abari(icx) + bbari(icx)/radice
   180)                      ssacoice(ib) = 1._r8 - cbari(icx) - dbari(icx) * radice
   181)                      gice(ib) = ebari(icx) + fbari(icx) * radice
   182) 
   183) ! Check to ensure upper limit of gice is within physical limits for large particles
   184)                      if (gice(ib) .ge. 1.0_r8) gice(ib) = 1.0_r8 - eps
   185)                      forwice(ib) = gice(ib)*gice(ib)
   186) ! Check to ensure all calculated quantities are within physical limits.
   187)                      if (extcoice(ib) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   188)                      if (ssacoice(ib) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   189)                      if (ssacoice(ib) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   190)                      if (gice(ib) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   191)                      if (gice(ib) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   192)                   enddo
   193) 
   194) ! For iceflag=2 option, combine with iceflag=0 option to handle large particle sizes.
   195) ! Use iceflag=2 option for ice particle effective radii from 5.0 to 131.0 microns
   196) ! and use iceflag=0 option for ice particles greater than 131.0 microns.
   197) ! *** NOTE: Transition between two methods has not been smoothed.
   198) 
   199)                elseif (iceflag .eq. 2) then
   200)                   if (radice .lt. 5.0_r8) stop 'ICE RADIUS OUT OF BOUNDS'
   201)                   if (radice .ge. 5.0_r8 .and. radice .le. 131._r8) then
   202)                      factor = (radice - 2._r8)/3._r8
   203)                      index = int(factor)
   204)                      if (index .eq. 43) index = 42
   205)                      fint = factor - float(index)
   206)                      do ib = ib1, ib2
   207)                         extcoice(ib) = extice2(index,ib) + fint * &
   208)                                       (extice2(index+1,ib) -  extice2(index,ib))
   209)                         ssacoice(ib) = ssaice2(index,ib) + fint * &
   210)                                       (ssaice2(index+1,ib) -  ssaice2(index,ib))
   211)                         gice(ib) = asyice2(index,ib) + fint * &
   212)                                       (asyice2(index+1,ib) -  asyice2(index,ib))
   213)                         forwice(ib) = gice(ib)*gice(ib)
   214) ! Check to ensure all calculated quantities are within physical limits.
   215)                         if (extcoice(ib) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   216)                         if (ssacoice(ib) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   217)                         if (ssacoice(ib) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   218)                         if (gice(ib) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   219)                         if (gice(ib) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   220)                      enddo
   221)                   elseif (radice .gt. 131._r8) then
   222)                      do ib = ib1, ib2
   223)                         if (wavenum2(ib) .gt. 1.43e04_r8) then
   224)                            icx = 1
   225)                         elseif (wavenum2(ib) .gt. 7.7e03_r8) then
   226)                            icx = 2
   227)                         elseif (wavenum2(ib) .gt. 5.3e03_r8) then
   228)                            icx = 3
   229)                         elseif (wavenum2(ib) .gt. 4.0e03_r8) then
   230)                            icx = 4
   231)                         elseif (wavenum2(ib) .ge. 2.5e03_r8) then
   232)                            icx = 5
   233)                         endif
   234)                         extcoice(ib) = abari(icx) + bbari(icx) / radice
   235)                         ssacoice(ib) = 1._r8 - cbari(icx) - dbari(icx) * radice
   236)                         gice(ib) = ebari(icx) + fbari(icx) * radice
   237) ! Check to ensure upper limit of gice is within physical limits for large particles
   238)                         if (gice(ib) .ge. 1.0_r8) gice(ib) = 1.0_r8 - eps
   239)                         forwice(ib) = gice(ib)*gice(ib)
   240) ! Check to ensure all calculated quantities are within physical limits.
   241)                         if (extcoice(ib) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   242)                         if (ssacoice(ib) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   243)                         if (ssacoice(ib) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   244)                         if (gice(ib) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   245)                         if (gice(ib) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   246)                      enddo
   247)                   endif
   248) 
   249) ! For iceflag=3 option, combine with iceflag=0 option to handle large particle sizes
   250) ! Use iceflag=3 option for ice particle effective radii from 3.2 to 91.0 microns
   251) ! (generalized effective size, dge, from 5 to 140 microns), and use iceflag=0 option
   252) ! for ice particle effective radii greater than 91.0 microns (dge = 140 microns).
   253) ! *** NOTE: Fu parameterization requires particle size in generalized effective size.
   254) ! *** NOTE: Transition between two methods has not been smoothed. 
   255) 
   256)                elseif (iceflag .eq. 3) then
   257)                   dgeice = dge(lay)
   258)                   if (dgeice .lt. 5.0_r8) stop 'ICE GENERALIZED EFFECTIVE SIZE OUT OF BOUNDS'
   259)                   if (dgeice .ge. 5.0_r8 .and. dgeice .le. 140._r8) then
   260)                      factor = (dgeice - 2._r8)/3._r8
   261)                      index = int(factor)
   262)                      if (index .eq. 46) index = 45
   263)                      fint = factor - float(index)
   264) 
   265)                      do ib = ib1 , ib2
   266)                         extcoice(ib) = extice3(index,ib) + fint * &
   267)                                       (extice3(index+1,ib) - extice3(index,ib))
   268)                         ssacoice(ib) = ssaice3(index,ib) + fint * &
   269)                                       (ssaice3(index+1,ib) - ssaice3(index,ib))
   270)                         gice(ib) = asyice3(index,ib) + fint * &
   271)                                   (asyice3(index+1,ib) - asyice3(index,ib))
   272)                         fdelta(ib) = fdlice3(index,ib) + fint * &
   273)                                     (fdlice3(index+1,ib) - fdlice3(index,ib))
   274)                         if (fdelta(ib) .lt. 0.0_r8) stop 'FDELTA LESS THAN 0.0'
   275)                         if (fdelta(ib) .gt. 1.0_r8) stop 'FDELTA GT THAN 1.0'                     
   276)                         forwice(ib) = fdelta(ib) + 0.5_r8 / ssacoice(ib)
   277) ! See Fu 1996 p. 2067 
   278)                         if (forwice(ib) .gt. gice(ib)) forwice(ib) = gice(ib)
   279) ! Check to ensure all calculated quantities are within physical limits.
   280)                         if (extcoice(ib) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   281)                         if (ssacoice(ib) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   282)                         if (ssacoice(ib) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   283)                         if (gice(ib) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   284)                         if (gice(ib) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   285)                      enddo
   286)                   elseif (dgeice .gt. 140._r8) then
   287)                      do ib = ib1, ib2
   288)                         if (wavenum2(ib) .gt. 1.43e04_r8) then
   289)                            icx = 1
   290)                         elseif (wavenum2(ib) .gt. 7.7e03_r8) then
   291)                            icx = 2
   292)                         elseif (wavenum2(ib) .gt. 5.3e03_r8) then
   293)                            icx = 3
   294)                         elseif (wavenum2(ib) .gt. 4.0e03_r8) then
   295)                            icx = 4
   296)                         elseif (wavenum2(ib) .ge. 2.5e03_r8) then
   297)                            icx = 5
   298)                         endif
   299)                         extcoice(ib) = abari(icx) + bbari(icx)/radice
   300)                         ssacoice(ib) = 1._r8 - cbari(icx) - dbari(icx) * radice
   301)                         gice(ib) = ebari(icx) + fbari(icx) * radice
   302) ! Check to ensure upper limit of gice is within physical limits for large particles
   303)                         if (gice(ib).ge.1.0_r8) gice(ib) = 1.0_r8-eps
   304)                         forwice(ib) = gice(ib)*gice(ib)
   305) ! Check to ensure all calculated quantities are within physical limits.
   306)                         if (extcoice(ib) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   307)                         if (ssacoice(ib) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   308)                         if (ssacoice(ib) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   309)                         if (gice(ib) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   310)                         if (gice(ib) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   311)                      enddo
   312)                   endif
   313)                 endif
   314)                   
   315) ! Calculation of absorption coefficients due to water clouds.
   316)                 if (clwp(lay) .eq. 0.0_r8) then
   317)                    do ib = ib1 , ib2
   318)                       extcoliq(ib) = 0.0_r8
   319)                       ssacoliq(ib) = 0.0_r8
   320)                       gliq(ib) = 0.0_r8
   321)                       forwliq(ib) = 0.0_r8
   322)                    enddo
   323) 
   324)                 elseif (liqflag .eq. 1) then
   325)                    radliq = rel(lay)
   326)                    if (radliq .lt. 1.5_r8 .or. radliq .gt. 60._r8) stop &
   327)                       'LIQUID EFFECTIVE RADIUS OUT OF BOUNDS'
   328)                    index = int(radliq - 1.5_r8)
   329)                    if (index .eq. 0) index = 1
   330)                    if (index .eq. 58) index = 57
   331)                    fint = radliq - 1.5_r8 - float(index)
   332) 
   333)                    do ib = ib1 , ib2
   334)                       extcoliq(ib) = extliq1(index,ib) + fint * &
   335)                                     (extliq1(index+1,ib) - extliq1(index,ib))
   336)                       ssacoliq(ib) = ssaliq1(index,ib) + fint * &
   337)                                     (ssaliq1(index+1,ib) - ssaliq1(index,ib))
   338)                       if (fint .lt. 0._r8 .and. ssacoliq(ib) .gt. 1._r8) &
   339)                                      ssacoliq(ib) = ssaliq1(index,ib)
   340)                       gliq(ib) = asyliq1(index,ib) + fint * &
   341)                                 (asyliq1(index+1,ib) - asyliq1(index,ib))
   342)                       forwliq(ib) = gliq(ib)*gliq(ib)
   343) ! Check to ensure all calculated quantities are within physical limits.
   344)                       if (extcoliq(ib) .lt. 0.0_r8) stop 'LIQUID EXTINCTION LESS THAN 0.0'
   345)                       if (ssacoliq(ib) .gt. 1.0_r8) stop 'LIQUID SSA GRTR THAN 1.0'
   346)                       if (ssacoliq(ib) .lt. 0.0_r8) stop 'LIQUID SSA LESS THAN 0.0'
   347)                       if (gliq(ib) .gt. 1.0_r8) stop 'LIQUID ASYM GRTR THAN 1.0'
   348)                       if (gliq(ib) .lt. 0.0_r8) stop 'LIQUID ASYM LESS THAN 0.0'
   349)                    enddo
   350)                 endif
   351) 
   352)                 do ib = ib1 , ib2
   353)                    tauliqorig = clwp(lay) * extcoliq(ib)
   354)                    tauiceorig = ciwp(lay) * extcoice(ib)
   355)                    taucldorig(lay,ib) = tauliqorig + tauiceorig
   356) 
   357)                    ssaliq = ssacoliq(ib) * (1.0_r8 - forwliq(ib)) / &
   358)                            (1.0_r8 - forwliq(ib) * ssacoliq(ib))
   359)                    tauliq = (1.0_r8 - forwliq(ib) * ssacoliq(ib)) * tauliqorig
   360)                    ssaice = ssacoice(ib) * (1.0_r8 - forwice(ib)) / &
   361)                            (1.0_r8 - forwice(ib) * ssacoice(ib))
   362)                    tauice = (1.0_r8 - forwice(ib) * ssacoice(ib)) * tauiceorig
   363) 
   364)                    scatliq = ssaliq * tauliq
   365)                    scatice = ssaice * tauice
   366) 
   367)                    taucloud(lay,ib) = tauliq + tauice
   368) 
   369) ! Ensure non-zero taucmc and scatice
   370)                    if (taucloud(lay,ib).eq.0.0_r8) taucloud(lay,ib) = cldmin
   371)                    if (scatice.eq.0.0_r8) scatice = cldmin
   372) 
   373)                    ssacloud(lay,ib) = (scatliq + scatice) / taucloud(lay,ib)
   374) 
   375)                    if (iceflag .eq. 3) then
   376) ! In accordance with the 1996 Fu paper, equation A.3, 
   377) ! the moments for ice were calculated depending on whether using spheres
   378) ! or hexagonal ice crystals.
   379)                       istr = 1
   380)                       asmcloud(lay,ib) = (1.0_r8/(scatliq+scatice)) * &
   381)                          (scatliq*(gliq(ib)**istr - forwliq(ib)) / &
   382)                          (1.0_r8 - forwliq(ib)) + scatice * ((gice(ib)-forwice(ib)) / &
   383)                          (1.0_r8 - forwice(ib)))**istr)
   384)                    else 
   385) ! This code is the standard method for delta-m scaling. 
   386)                       istr = 1
   387)                       asmcloud(lay,ib) = (scatliq *  &
   388)                          (gliq(ib)**istr - forwliq(ib)) / &
   389)                          (1.0_r8 - forwliq(ib)) + scatice * (gice(ib)**istr - forwice(ib)) / &
