   553)           length      = nlyr * ntrac * pCycle%lengthP_ghost*nhc*npoints
   554)           tag             = pCycle%tag
   555)           iptr            = pCycle%ptrP_ghost
   556)           !print *,'ghost_exchangeV: MPI_Irecv: SRC:',source,'LENGTH:',length,'TAG: ',tag
   557)           call MPI_Irecv(buffer%receive(1,1,1,1,iptr),length,MPIreal_t, &
   558)                source,tag,hybrid%par%comm,Rrequest(icycle),ierr)
   559)           if(ierr .ne. MPI_SUCCESS) then

   561)              call MPI_Error_String(errorcode,errorstring,errorlen,ierr)
   562)              print *,'bndry_exchangeV: Error after call to MPI_Irecv: ',errorstring

   564)        end do    ! icycle
   565) 
   566) 
   567)        !==================================================
   568)        !  Wait for all the receives to complete
   569)        !==================================================
   570) 
   571)        call MPI_Waitall(nSendCycles,Srequest,status,ierr)
   572)        call MPI_Waitall(nRecvCycles,Rrequest,status,ierr)
   573) 
   574)        do icycle=1,nRecvCycles
   575)           pCycle         => pSchedule%RecvCycle(icycle)
   576)           length             = pCycle%lengthP_ghost
   577)           iptr            = pCycle%ptrP_ghost
   578)           do i=0,length-1
   579)              do i2=1,nhc
   580)                 do i1=1,npoints
   581)                    buffer%buf(i1,i2,1:nlyr,1:ntrac,iptr+i) = buffer%receive(i1,i2,1:nlyr,1:ntrac,iptr+i)
   582)                 enddo
   583)              enddo
   584)           enddo
   585)        end do   ! icycle
   586) 
   587) 
   588) #endif
   589)     endif  ! if (hybrid%ithr == 0)
   590) #if (! defined ELEMENT_OPENMP)
   591)     !$OMP BARRIER
   592) #endif

   594)   end subroutine ghost_exchangeV
