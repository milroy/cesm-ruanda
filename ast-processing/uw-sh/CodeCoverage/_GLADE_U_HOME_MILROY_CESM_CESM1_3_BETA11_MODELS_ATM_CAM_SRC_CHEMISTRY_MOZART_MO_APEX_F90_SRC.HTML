     1) module mo_apex
     2) 
     3) !-------------------------------------------------------------------------------
     4) ! Purpose:
     5) !
     6) !   Calculate apex coordinates and magnetic field magnitudes
     7) !   at global geographic grid for year of current model run. 
     8) !
     9) ! Method: 
    10) !
    11) !   The magnetic field parameters output by this module are time and height
    12) !     independent. They are chunked for waccm physics, i.e., allocated as 
    13) !     (pcols,begchunk:endchunk)
    14) !   Interface sub apexmag is called once per run from sub inti.
    15) !     Sub apexmag may be called for years 1900 through 2005.
    16) !   This module is dependent on routines in apex_subs.F (modified IGRF model).
    17) !   Apex_subs has several authors, but has been modified and maintained
    18) !     in recent years by Roy Barnes (bozo@ucar.edu).
    19) !   Subs apxmka and apxmall are called with the current lat x lon grid 
    20) !     resolution.
    21) !
    22) ! Author: Ben Foster, foster@ucar.edu (Nov, 2003)
    23) !-------------------------------------------------------------------------------
    24) 
    25)    use shr_kind_mod,    only: r8 => shr_kind_r8
    26)    use ppgrid,          only: pcols, begchunk, endchunk          ! physics grid
    27)    use phys_grid,       only: get_lat_p, get_lon_p, get_ncols_p
    28)    use abortutils,      only: endrun
    29)    use cam_control_mod, only: magfield_fix_year
    30)    use cam_logfile,     only: iulog
    31)    use spmd_utils,      only: masterproc
    32) 
    33)    implicit none
    34) 
    35)    private
    36)    public :: apexmag
    37)    public :: alatm, alonm, glonm, bnorth, beast, bdown, bmag
    38)    public :: d1vec, d2vec, colatp, elonp
    39) 
    40) !-------------------------------------------------------------------------------
    41) ! Magnetic field output arrays, chunked for physics:
    42) ! (these are allocated (pcols,begchunk:endchunk) by sub apex_alloc)
    43) !-------------------------------------------------------------------------------
    44)    real(r8), allocatable, dimension(:,:) :: & ! (pcols,begchunk:endchunk)
    45)      alatm,  & ! apex mag latitude at each geographic grid point (radians)
    46)      alonm,  & ! apex mag longitude at each geographic grid point (radians)
    47)      glonm,  & ! apex mag longitude at each geographic grid point (radians)
    48)      bnorth, & ! northward component of magnetic field
    49)      beast,  & ! eastward component of magnetic field
    50)      bdown,  & ! downward component of magnetic field
    51)      bmag      ! magnitude of magnetic field
    52)    real(r8), allocatable, dimension(:,:,:) :: & ! (3,pcols,begchunk:endchunk)
    53)      d1vec,    & ! base vectors more-or-less magnetic eastward direction
    54)      d2vec       ! base vectors more-or-less magnetic downward/equatorward direction
    55)    real(r8) :: &
    56)      colatp,   & ! geocentric colatitude of geomagnetic dipole north pole (deg)
    57)      elonp	 ! East longitude of geomagnetic dipole north pole (deg)
    58) 
    59)    logical :: initialized = .false.
    60)

    63) subroutine apexmag
    64) !-------------------------------------------------------------------------------
    65) ! Driver for apex code to calculate apex magnetic coordinates at 
    66) !   current geographic spatial resolution for given year. This calls
    67) !   routines in apex_subs.F.
    68) !
    69) ! This is called once per run from sub inti.
    70) !-------------------------------------------------------------------------------
    71) 
    72)    use physconst,      only: pi
    73)    use dyn_grid, only : get_dyn_grid_parm, get_horiz_grid_d
    74) !-------------------------------------------------------------------------------
    75) ! Local variables
    76) !-------------------------------------------------------------------------------
    77)    integer, parameter  :: nalt  = 2             ! number of altitudes
    78)    real(r8), parameter :: re    = 6.378165e8_r8 ! earth radius (cm)
    79)    real(r8), parameter :: h0    = 9.0e6_r8      ! base height (90 km)
    80)    real(r8), parameter :: hs    = 1.3e7_r8
    81)    real(r8), parameter :: eps   = 1.e-6_r8      ! epsilon
    82)    real(r8), parameter :: cm2km = 1.e-5_r8
    83) 
    84)    integer  :: i, j, ist, lchnk              ! indices
    85)    integer  :: ncol, ic, jc
    86) 
    87)    real(r8), allocatable :: dlat_mid(:)              ! midpoint latitudes to assist interp
    88)    real(r8), allocatable :: dlon_mid(:)              ! midpoint longitudes to assist interp
    89)    real(r8), allocatable :: dlon_shft(:)             ! shifted midpoint longitudes
    90) 
    91)    real(r8) :: rtd, dtr                      ! deg<->radians conversions
    92)    real(r8) :: rekm, h0km, alt, hr, ror03, alon, alat, & ! apxmall args
    93)                vmp, w, d, be3, sim, xlatqd, f, si, date1(1), collat, collon
    94)    real(r8) :: gpalt(nalt)                   ! altitudes passed to apxmka
    95)    real(r8), allocatable :: wk(:), clat(:), clon(:)
    96)    integer :: plon, plat, plonp1, platp1, lwk
    97) !-------------------------------------------------------------------------------
    98) ! Non-scalar arguments returned by APXMALL:
    99) !-------------------------------------------------------------------------------
   100)    real(r8) :: bhat(3)
   101)    real(r8) :: d3(3)
   102)    real(r8) :: e1(3), e2(3), e3(3)
   103)    real(r8) :: f1(2), f2(2)
   104) 
   105)    real(r8), allocatable :: glatm(:,:)
   106)    real(r8), allocatable :: bglobal(:,:,:)
   107)    real(r8), allocatable :: d1global(:,:,:)
   108)    real(r8), allocatable :: d2global(:,:,:)
   109)    real(r8), allocatable :: bmglobal(:,:)   
   110) 
   111)    real(r8) :: rdum
   112) 
   113)    if (initialized) return
   114) 
   115)    plat = get_dyn_grid_parm('plat')
   116)    plon = get_dyn_grid_parm('plon')
   117)    platp1 = plat+1
   118)    plonp1 = plon+1
   119)    lwk = platp1*plonp1*nalt*5+platp1 + plonp1 + nalt
   120) 
   121)    allocate(dlat_mid(platp1), dlon_mid(plonp1), dlon_shft(plonp1), wk(lwk))
   122)    allocate(glatm(plon,plat), bglobal(3,plon,plat), d1global(3,plon,plat), d2global(3,plon,plat))
   123)    allocate(bmglobal(plon,plat))
   124)    allocate(clat(plat), clon(plon))
   125) !-------------------------------------------------------------------------------
   126) ! Allocate output arrays
   127) !-------------------------------------------------------------------------------
   128)    call apex_alloc
   129) 
   130)    rtd = 180._r8/pi     ! radians to degrees
   131)    dtr = pi/180._r8     ! degrees to radians
   132) 
   133) !-------------------------------------------------------------------------------
   134) ! Extend staggered latitudes (midpoints) to assist apxmka build 
   135) ! interpolation tables (degrees).
   136) !-------------------------------------------------------------------------------
   137)    call get_horiz_grid_d(plat, clat_d_out=clat)
   138)    call get_horiz_grid_d(plon, clon_d_out=clon)
   139) 
   140)    dlat_mid(1) = clat(1)*rtd
   141)    do j = 2,plat
   142)       dlat_mid(j) = rtd*(clat(j-1) + clat(j))/2.0_r8
   143)    end do
   144)    dlat_mid(platp1) = rtd*clat(plat)
   145) 
   146) !-------------------------------------------------------------------------------
   147) ! Set longitude midpoints (assume non-reducing grid, so use londeg(:,1))
   148) !-------------------------------------------------------------------------------
   149)    dlon_mid(1) = rtd*(clon(1) - .5_r8*(clon(2) - clon(1)))
   150)    do i = 2,plon
   151)      dlon_mid(i) = .5_r8*rtd*(clon(i) + clon(i-1))
   152)    end do
   153)    dlon_mid(plon+1) = dlon_mid(plon) + rtd*(clon(plon) - clon(plon-1))
   154) 
   155) 
   156) #ifdef AURORA_DEBUG
   157)    write(iulog,*) '---------------------------------------'
   158)    write(iulog,*) 'apex: dlon_mid'
   159)    write(iulog,'(1p,5g15.7)') dlon_mid(:)
   160)    write(iulog,*) 'apex: dlat_mid'
   161)    write(iulog,'(1p,5g15.7)') dlat_mid(:)
   162)    write(iulog,*) '---------------------------------------'
   163) #endif
   164) 
   165)    date1(1) = magfield_fix_year
   166) 
   167) !-------------------------------------------------------------------------------
   168) ! Center min, max altitudes about 130 km
   169) !-------------------------------------------------------------------------------
   170)    gpalt(1) =  90._r8
   171)    gpalt(2) = 170._r8
   172) 
   173) !-------------------------------------------------------------------------------
   174) ! Set up interpolation.
   175) ! (Note apxmka expects longitudes in -180 -> +180)
   176) !-------------------------------------------------------------------------------
   177)    dlon_shft(:) = dlon_mid(:) - 180._r8
   178)    call apxmka( iulog, date1, dlat_mid, dlon_shft, gpalt, &
   179)                 platp1, plonp1, nalt, wk, lwk, ist )
   180)    if( ist /= 0 ) then
   181)      write(iulog,"(/,'>>> apexmag: Error from apxmka: ist=',i5)") ist
   182)      call endrun
   183)    end if
   184) 
   185)    rekm  = re*cm2km    ! earth radius (km)
   186)    h0km  = h0*cm2km    ! base height (km)
   187)    alt   = hs*cm2km    ! altitude for apxmall (km)
   188)    hr    = alt         ! reference altitude (km)
   189)    ror03 = ((rekm + alt)/(rekm + h0km))**3
   190) 
   191) !------------------------------------------------------------------------------
   192) ! Apex coords alon, alat are returned for each geographic grid point:
   193) ! first form global arrays
   194) !------------------------------------------------------------------------------
   195)    do j = 1,plat
   196)       collat = clat(j)*rtd                       ! latitude of current column (deg)
   197)       do i = 1,plon
   198)          collon = clon(i)*rtd ! longitude of current column (deg)
   199)          call apxmall(                                  &
   200)            collat, collon, alt, hr, wk, lwk,            & ! Inputs
   201)            bglobal(1,i,j), bhat, bmglobal(i,j), si,     & ! Mag Fld
   202)            alon, alat,                                  & ! Apex lon,lat output
   203)            vmp, w, d, be3, sim, d1global(1,i,j), d2global(1,i,j), d3, e1, e2, e3, & ! Mod Apex
   204)            xlatqd, f, f1, f2, ist )                       ! Qsi-Dpl
   205)          if( ist /= 0 ) then
   206)            write(iulog,"(/,'>>> apexmag: Error from apxmall: ist=',i4)") ist
   207)            call endrun
   208)          end if
   209)          glonm(i,j)   =  alon*dtr                         ! mag lons (radians)
   210)          glatm(i,j)   =  alat*dtr                         ! mag lats (radians)
   211)       end do
   212)    end do
   213) 
   214) #ifdef AURORA_DEBUG
   215)    write(iulog,*) '---------------------------------------'
   216)    write(iulog,*) 'apex: geo lons'
   217)    write(iulog,'(1p,5g15.7)') rtd*clon(:)
   218)    write(iulog,*) 'apex: geo lats'
   219)    write(iulog,'(1p,5g15.7)') rtd*clat(:)
   220)    write(iulog,*) ' '
   221)    do j = 1,plat
   222)       write(iulog,*) 'apexmag: mag lats @ lat ndx = ',j
   223)       write(iulog,'(1p,5g15.7)') glatm(:,j)/dtr
   224)    end do
   225)    write(iulog,*) ' '
   226)    do j = 1,plat
   227)       write(iulog,*) 'apexmag: mag lons @ lat ndx = ',j
   228)       write(iulog,'(1p,5g15.7)') glonm(:,j)/dtr
   229)    end do
   230)    write(iulog,*) ' '
   231)    write(iulog,*) '---------------------------------------'
   232) #endif
   233) 
   234) !------------------------------------------------------------------------------
   235) ! map globals to chunks
   236) !------------------------------------------------------------------------------
   237) chunk_loop : &
   238)    do lchnk = begchunk,endchunk  ! physics chunks (groups of arbitrary columns)
   239)      ncol = get_ncols_p( lchnk )
   240) column_loop : &
   241)      do i = 1,ncol             ! number of columns in each chunk
   242) !------------------------------------------------------------------------------
   243) ! set module output from apxmall output:
   244) !------------------------------------------------------------------------------
   245)        ic                = get_lon_p(lchnk,i)
   246)        jc                = get_lat_p(lchnk,i)
   247) !      write(iulog,*) 'apex: lchnk,i,ic,jc = ',lchnk,i,ic,jc
   248)        alatm (i,lchnk)   =  glatm(ic,jc)                ! mag lats (radians)
   249)        alonm (i,lchnk)   =  glonm(ic,jc)                ! mag lons (radians)
   250) !      bnorth(i,lchnk)   =  bglobal(2,ic,jc)*1.e-5_r8   ! northward nT -> gauss
   251) !      beast (i,lchnk)   =  bglobal(1,ic,jc)*1.e-5_r8   ! eastward  nT -> gauss
   252) !      bdown (i,lchnk)   = -bglobal(3,ic,jc)*1.e-5_r8   ! downward  nT -> gauss
   253) !      bmag  (i,lchnk)   =  bmglobal(ic,jc)*1.e-5_r8    ! magnitude nT -> gauss
   254)        bnorth(i,lchnk)   =  bglobal(2,ic,jc)            ! northward nT -> gauss
   255)        beast (i,lchnk)   =  bglobal(1,ic,jc)            ! eastward  nT -> gauss
   256)        bdown (i,lchnk)   = -bglobal(3,ic,jc)            ! downward  nT -> gauss
   257)        bmag  (i,lchnk)   =  bmglobal(ic,jc)             ! magnitude nT -> gauss
   258)        d1vec (:,i,lchnk) =  d1global(:,ic,jc)
   259)        d2vec (:,i,lchnk) =  d2global(:,ic,jc)
   260)      end do column_loop
   261) 
   262) #ifdef DEBUG
   263)      write(iulog,"('apexmag: lchnk=',i3,' alatm(:,lchnk)*rtd=',/,(1p,6g12.4))") &
   264)        lchnk,alatm(:,lchnk)*rtd
   265)      write(iulog,"('apexmag: lchnk=',i3,' alonm(:,lchnk)*rtd=',/,(1p,6g12.4))") &
   266)        lchnk,alonm(:,lchnk)*rtd
   267)      write(iulog,"('apexmag: lchnk=',i3,' bnorth(:,lchnk)=',/,(6e12.4))") &
   268)        lchnk,bnorth(:,lchnk)
   269)      write(iulog,"('apexmag: lchnk=',i3,' beast(:,lchnk)=' ,/,(6e12.4))") &
   270)        lchnk,beast(:,lchnk)
   271)      write(iulog,"('apexmag: lchnk=',i3,' bdown(:,lchnk)=' ,/,(6e12.4))") &
   272)        lchnk,bdown(:,lchnk)
   273)      write(iulog,"('apexmag: lchnk=',i3,' bmag(:,lchnk)='  ,/,(6e12.4))") &
   274)        lchnk,bmag(:,lchnk)
   275) #endif
   276)    end do chunk_loop
   277) 
   278) #ifdef DEBUG
   279)    write(iulog,*) '---------------------------------------'
   280)    write(iulog,*) 'apex: begchunk,endchunk = ', begchunk,endchunk
   281)    write(iulog,*) 'apex: mag lons at j = 81'
   282)    write(iulog,'(1p,5g15.7)') glonm(:,81)/dtr
   283)    write(iulog,*) 'apex: mag lons at j = 82'
   284)    write(iulog,'(1p,5g15.7)') glonm(:,82)/dtr
   285)    write(iulog,*) 'apex: mag lons at j = 83'
   286)    write(iulog,'(1p,5g15.7)') glonm(:,83)/dtr
   287)    write(iulog,*) '---------------------------------------'
   288) #endif
   289) 
   290)    call dypol( colatp, elonp, rdum )	! get geomagnetic dipole north pole 
   291) 
   292)    if (masterproc) write(iulog,*) 'apex: colatp,elonp ', colatp, elonp
   293) 
   294)   deallocate(dlat_mid, dlon_mid, dlon_shft, wk)
   295)   deallocate(glatm, bglobal, d1global, d2global)
   296)   deallocate(bmglobal)
   297)   deallocate(clat, clon)
   298)   if (masterproc) write(iulog,"(' APEXMAG: Calculated apex magnetic coordinates for year AD ',i4)") &
   299)                   magfield_fix_year
   300) 
   301)   initialized = .true.
   302) 
   303) end subroutine apexmag
