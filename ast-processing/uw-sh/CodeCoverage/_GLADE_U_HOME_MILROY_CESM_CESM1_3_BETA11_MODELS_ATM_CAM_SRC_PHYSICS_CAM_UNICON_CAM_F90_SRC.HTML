   123) subroutine unicon_cam_register
   124) 
   125) ! Register fields in the constituent array and the physics buffer.
   126) 
   127) 
   128)    ! Jun.02.2012. Sungsu for advecting organization-related horizontal heterogeneity
   129)    !              within PBL. 
   130)    !              For the time being, in order to save computation time, advection of aerosol perturbations 
   131)    !              are simply neglected.
   132) 
   133) #ifdef USE_UNICON
   134) 
   135)    call cnst_add('ORGawk', mwdry, cpair,    0._r8, awk_cnst_ind, & 
   136)       'Wake area within PBL associated with organization', readiv=.false., mixtype = 'dry')
   137)    call cnst_add('ORGthl', mwdry, cpair,    0._r8, thl_cnst_ind, & 
   138)       'Perturbation of  thl associated with organization', readiv=.false., mixtype = 'dry')
   139)    call cnst_add('ORGqto', mwdry, cpair,    0._r8,  qt_cnst_ind, & 
   140)       'Perturbation of  qt  associated with organization', readiv=.false., mixtype = 'dry')
   141)    call cnst_add('ORGuoo', mwdry, cpair,    0._r8,   u_cnst_ind, & 
   142)       'Perturbation of  u   associated with organization', readiv=.false., mixtype = 'dry')
   143)    call cnst_add('ORGvoo', mwdry, cpair,    0._r8,   v_cnst_ind, & 
   144)       'Perturbation of  v   associated with organization', readiv=.false., mixtype = 'dry')
   145) 
   146) 
   147)    call pbuf_add_field('cushavg',       'global', dtype_r8, (/pcols,dyn_time_lvls/),            cushavg_idx)
   148)    call pbuf_add_field('cuorg',         'global', dtype_r8, (/pcols,dyn_time_lvls/),            cuorg_idx)
   149)    call pbuf_add_field('awk_PBL',       'global', dtype_r8, (/pcols,dyn_time_lvls/),            awk_PBL_idx)
   150)    call pbuf_add_field('delta_thl_PBL', 'global', dtype_r8, (/pcols,dyn_time_lvls/),            delta_thl_PBL_idx)
   151)    call pbuf_add_field('delta_qt_PBL',  'global', dtype_r8, (/pcols,dyn_time_lvls/),            delta_qt_PBL_idx)
   152)    call pbuf_add_field('delta_u_PBL',   'global', dtype_r8, (/pcols,dyn_time_lvls/),            delta_u_PBL_idx)
   153)    call pbuf_add_field('delta_v_PBL',   'global', dtype_r8, (/pcols,dyn_time_lvls/),            delta_v_PBL_idx)
   154)    call pbuf_add_field('delta_tr_PBL',  'global', dtype_r8, (/pcols,pcnst,dyn_time_lvls/),      delta_tr_PBL_idx)
   155)    call pbuf_add_field('cu_cmfr',       'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),       cu_cmfr_idx)
   156)    call pbuf_add_field('cu_thlr',       'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),       cu_thlr_idx)
   157)    call pbuf_add_field('cu_qtr',        'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),       cu_qtr_idx)
   158)    call pbuf_add_field('cu_ur',         'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),       cu_ur_idx)
   159)    call pbuf_add_field('cu_vr',         'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),       cu_vr_idx)
   160)    call pbuf_add_field('cu_qlr',        'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),       cu_qlr_idx)
   161)    call pbuf_add_field('cu_qir',        'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),       cu_qir_idx)
   162)    call pbuf_add_field('cu_trr',        'global', dtype_r8, (/pcols,pver,pcnst,dyn_time_lvls/), cu_trr_idx)
   163) 
   164) #endif

   170) subroutine unicon_cam_init(pbuf2d)
   171) 
   172)    type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   173) 
   174)    ! local variables
   175)    character(len=*), parameter :: sub='unicon_cam_init: '
   176)    integer :: i, icnst, j, l, nmodes, nspec
   177) 
   178)    character(len=8)  :: units
   179)    character(len=30) :: varname

   184) 
   185) #ifdef USE_UNICON
   186)

   188)    xlv   = latvap
   189)    xlf   = latice
   190)    xls   = xlv + xlf
   191)    cp    = cpair
   192) 
   193)    call unicon_init(latvap, cpair, latice, zvir, rair, gravit)
   194) 
   195)    ! save some constituent indices
   196)    call cnst_get_ind('CLDLIQ', ixcldliq)
   197)    call cnst_get_ind('CLDICE', ixcldice)
   198)    call cnst_get_ind('NUMLIQ', ixnumliq)
   199)    call cnst_get_ind('NUMICE', ixnumice)
   200) 
   201)    ! save some physics buffer indices
   202)    ast_idx      = pbuf_get_index('AST')
   203)    tke_idx      = pbuf_get_index('tke')
   204)    bprod_idx    = pbuf_get_index('bprod')
   205)    kpblh_idx    = pbuf_get_index('kpblh')
   206)    pblh_idx    = pbuf_get_index('pblh')
   207)    went_idx    = pbuf_get_index('went')
   208)    cush_idx    = pbuf_get_index('cush')
   209)    shfrc_idx   = pbuf_get_index('shfrc')
   210)    icwmrsh_idx = pbuf_get_index('ICWMRSH')
   211)    rprdsh_idx  = pbuf_get_index('RPRDSH')
   212)    prec_sh_idx = pbuf_get_index('PREC_SH')
   213)    snow_sh_idx = pbuf_get_index('SNOW_SH')
   214)    nevapr_shcu_idx = pbuf_get_index('NEVAPR_SHCU')
   215)    am_evp_st_idx  = pbuf_get_index('am_evp_st')
   216)    evprain_st_idx = pbuf_get_index('evprain_st')
   217)    evpsnow_st_idx = pbuf_get_index('evpsnow_st')
   218) 
   219)    ! physics buffer fields that need initializers -- these are only
   220)    ! fields that have been added to pbuf by this module, or by the 
   221)    ! convection driver layer.
   222)    if (is_first_step()) then
   223) 
   224)       if (cush_idx > 0) then
   225)          call pbuf_set_field(pbuf2d, cush_idx,          1.e3_r8)
   226)       else
   227)          call endrun(sub//'cush not in pbuf')
   228)       end if
   229)       if (cushavg_idx > 0) then
   230)          call pbuf_set_field(pbuf2d, cushavg_idx,       1.e3_r8)
   231)       else
   232)          call endrun(sub//'cushavg not in pbuf')
   233)       end if
   234)       if (cuorg_idx > 0) then
   235)          call pbuf_set_field(pbuf2d, cuorg_idx,         0.0_r8)
   236)       else
   237)          call endrun(sub//'cuorg not in pbuf')
   238)       end if
   239)       if (awk_pbl_idx > 0) then
   240)          call pbuf_set_field(pbuf2d, awk_pbl_idx,       0.0_r8)
   241)       else
   242)          call endrun(sub//'awk_PBL not in pbuf')
   243)       end if
   244)       if (delta_thl_PBL_idx > 0) then
   245)          call pbuf_set_field(pbuf2d, delta_thl_PBL_idx, 0.0_r8)
   246)       else
   247)          call endrun(sub//'delta_thl_PBL not in pbuf')
   248)       end if
   249)       if (delta_qt_PBL_idx > 0) then
   250)          call pbuf_set_field(pbuf2d, delta_qt_PBL_idx,  0.0_r8)
   251)       else
   252)          call endrun(sub//'delta_qt_PBL not in pbuf')
   253)       end if
   254)       if (delta_u_PBL_idx > 0) then
   255)          call pbuf_set_field(pbuf2d, delta_u_PBL_idx,   0.0_r8)
   256)       else
   257)          call endrun(sub//'delta_u_PBL not in pbuf')
   258)       end if
   259)       if (delta_v_PBL_idx > 0) then
   260)          call pbuf_set_field(pbuf2d, delta_v_PBL_idx,   0.0_r8)
   261)       else
   262)          call endrun(sub//'delta_v_PBL not in pbuf')
   263)       end if
   264)       if (delta_tr_PBL_idx > 0) then
   265)          call pbuf_set_field(pbuf2d, delta_tr_PBL_idx,  0.0_r8)
   266)       else
   267)          call endrun(sub//'delta_tr_PBL not in pbuf')
   268)       end if
   269)       if (cu_cmfr_idx > 0) then
   270)          call pbuf_set_field(pbuf2d, cu_cmfr_idx,       0.0_r8)
   271)       else
   272)          call endrun(sub//'cu_cmfr not in pbuf')
   273)       end if
   274)       if (cu_thlr_idx > 0) then
   275)          call pbuf_set_field(pbuf2d, cu_thlr_idx,       0.0_r8)
   276)       else
   277)          call endrun(sub//'cu_thlr not in pbuf')
   278)       end if
   279)       if (cu_qtr_idx > 0) then
   280)          call pbuf_set_field(pbuf2d, cu_qtr_idx,        0.0_r8)
   281)       else
   282)          call endrun(sub//'cu_qtr not in pbuf')
   283)       end if
   284)       if (cu_ur_idx > 0) then
   285)          call pbuf_set_field(pbuf2d, cu_ur_idx,         0.0_r8)
   286)       else
   287)          call endrun(sub//'cu_ur not in pbuf')
   288)       end if
   289)       if (cu_vr_idx > 0) then
   290)          call pbuf_set_field(pbuf2d, cu_vr_idx,         0.0_r8)
   291)       else
   292)          call endrun(sub//'cu_vr not in pbuf')
   293)       end if
   294)       if (cu_qlr_idx > 0) then
   295)          call pbuf_set_field(pbuf2d, cu_qlr_idx,        0.0_r8)
   296)       else
   297)          call endrun(sub//'cu_qlr not in pbuf')
   298)       end if
   299)       if (cu_qir_idx > 0) then
   300)          call pbuf_set_field(pbuf2d, cu_qir_idx,        0.0_r8)
   301)       else
   302)          call endrun(sub//'cu_qir not in pbuf')
   303)       end if
   304)       if (cu_trr_idx > 0) then
   305)          call pbuf_set_field(pbuf2d, cu_trr_idx,        0.0_r8)
   306)       else
   307)          call endrun(sub//'cu_trr not in pbuf')
   308)       end if
   309) 
   310)    end if
   311) 
   312)    ! Set arrays to identify the modal aerosol constituents
   313)    cnst_is_mam_num = .false.
   314)    cnst_is_mam_mmr = .false.
   315)    call rad_cnst_get_info(0, nmodes=nmodes)
   316)    do i = 1, nmodes
   317)       call rad_cnst_get_mode_num_idx(i, icnst)
   318)       cnst_is_mam_num(icnst) = .true.
   319)       call rad_cnst_get_info(0, i, nspec=nspec)
   320)       do j = 1, nspec
   321)          call rad_cnst_get_mam_mmr_idx(i, j, icnst)
   322)          cnst_is_mam_mmr(icnst) = .true.
   323)       end do
   324)    end do
   325) 
   326)    ! ------------------------- !
   327)    ! Internal Output Variables !
   328)    ! ------------------------- !
   329) 
   330)    ! Sungsu for advection of convective organization
   331) 
   332)    call addfld('ORGawk', 'no', pver,  'A', 'Convective Organization - Wake Area', phys_decomp )
   333)    call addfld('ORGthl', 'K', pver,  'A', 'Convective Organization - Perturbation of thl in the non-wake area', phys_decomp )
   334)    call addfld('ORGqto', 'kg/kg', pver,  'A', 'Convective Organization - Perturbation of qt  in the non-wake area', phys_decomp )
   335)    call addfld('ORGuoo', 'm/s', pver,  'A', 'Convective Organization - Perturbation of u  in the non-wake area', phys_decomp )
   336)    call addfld('ORGvoo', 'm/s', pver,  'A', 'Convective Organization - Perturbation of v  in the non-wake area', phys_decomp )
   337) 
   338)    ! From the main unified convection scheme
   339) 
   340)    call addfld('cmf_SP',   'kg/m2/s', pverp, 'A', 'Convective net mass flux', phys_decomp)
   341)    call addfld('qtflx_SP', 'kg/m2/s', pverp, 'A', 'Convective net qt flux', phys_decomp)
   342)    call addfld('slflx_SP', 'J/m2/s' , pverp, 'A', 'Convective net sl flux', phys_decomp)
   343)    call addfld('uflx_SP',  'kg/m/s2', pverp, 'A', 'Convective net u flux', phys_decomp)
   344)    call addfld('vflx_SP',  'kg/m/s2', pverp, 'A', 'Convective net v flux', phys_decomp)
   345) 
   346)    call addfld('cmf_u_SP',   'kg/m2/s', pverp, 'A', 'Convective updraft mass flux', phys_decomp)
   347)    call addfld('qtflx_u_SP', 'kg/m2/s', pverp, 'A', 'Convective updraft qt flux', phys_decomp)
   348)    call addfld('slflx_u_SP', 'J/m2/s' , pverp, 'A', 'Convective updraft sl flux', phys_decomp)
   349)    call addfld('uflx_u_SP',  'kg/m/s2', pverp, 'A', 'Convective updraft u flux', phys_decomp)
   350)    call addfld('vflx_u_SP',  'kg/m/s2', pverp, 'A', 'Convective updraft v flux', phys_decomp)
   351)

   353)    call addfld('qtflx_d_SP', 'kg/m2/s', pverp, 'A', 'Convective downdraft qt flux', phys_decomp)

   359)    call addfld('slten_u_SP', 'J/kg/s',  pver, 'A', 'Convective tendency sl by updraft', phys_decomp)
   360)    call addfld('uten_u_SP',  'm/s/s',   pver, 'A', 'Convective tendency u  by updraft', phys_decomp)
   361)    call addfld('vten_u_SP',  'm/s/s' ,  pver, 'A', 'Convective tendency v  by updraft', phys_decomp)
   362)    call addfld('sten_u_SP',  'J/kg/s',  pver, 'A', 'Convective tendency s  by updraft', phys_decomp)
