    18) !   n means the field will not be written.  To output the same field at
    19) !   more than one frequency, for instance monthy and daily, set 
    20) !   f_<field> = 'md'.

   297) ! !IROUTINE: init_hist - initialize history files
   298) !
   299) ! !INTERFACE:
   300) !
   301)       subroutine init_hist (dt)
   302) !
   303) ! !DESCRIPTION:
   304) !
   305) ! Initialize history files
   306) !
   307) ! !REVISION HISTORY:
   308) !
   309) ! authors Tony Craig, NCAR
   310) !         Elizabeth C. Hunke, LANL
   311) !         C.M. Bitz, UW
   312) !         Bruce P. Briegleb, NCAR
   313) !         William H. Lipscomb, LANL
   314) !
   315) ! !USES:
   316) !
   317)       use ice_constants
   318)       use ice_calendar, only: yday, days_per_year, histfreq, &
   319)                               histfreq_n, nstreams
   320)       use ice_flux, only: mlt_onset, frz_onset, albcnt
   321)       use ice_restart, only: restart
   322)       use ice_state, only: tr_aero, tr_iage, tr_FY, tr_pond, tr_lvl
   323)       use ice_exit
   324) !
   325) ! !INPUT/OUTPUT PARAMETERS:
   326) !
   327)       real (kind=dbl_kind), intent(in) :: &
   328)          dt      ! time step
   329) !
   330) !EOP
   331) !
   332)       integer (kind=int_kind) :: n, k, ns, ns1, ns2, lenf
   333)       integer (kind=int_kind) :: hfreqn
   334)       integer (kind=int_kind), dimension(max_nstrm) :: &
   335)          ntmp
   336)       integer (kind=int_kind) :: nml_error ! namelist i/o error flag
   337) 
   338)       character (len=3) :: nchar
   339)       character (len=40) :: stmp
   340) 
   341)       !-----------------------------------------------------------------
   342)       ! read namelist
   343)       !-----------------------------------------------------------------
   344) 
   345)       call get_fileunit(nu_nml)
   346)       if (my_task == master_task) then
   347)          open (nu_nml, file=nml_filename, status='old',iostat=nml_error)
   348)          if (nml_error /= 0) then
   349)             nml_error = -1
   350)          else
   351)             nml_error =  1
   352)          endif
   353)          do while (nml_error > 0)
   354)             read(nu_nml, nml=icefields_nml,iostat=nml_error)
   355)          end do
   356)          if (nml_error == 0) close(nu_nml)
   357)       endif
   358)       call release_fileunit(nu_nml)
   359) 
   360)       call broadcast_scalar(nml_error, master_task)
   361)       if (nml_error /= 0) then
   362)          close (nu_nml)
   363)          call abort_ice('ice: error reading icefields_nml')
   364)       endif
   365) 
   366)       ! histfreq options ('1','h','d','m','y')
   367)       nstreams = 0
   368)       do ns = 1, max_nstrm
   369)          if (histfreq(ns) == '1' .or. histfreq(ns) == 'h' .or. &
   370)              histfreq(ns) == 'd' .or. histfreq(ns) == 'm' .or. &
   371)              histfreq(ns) == 'y') then
   372)                 nstreams = nstreams + 1
   373)          else if (histfreq(ns) /= 'x') then
   374)              call abort_ice('ice: histfreq contains illegal element')
   375)          endif
   376)       enddo
   377)       if (nstreams == 0) write (nu_diag,*) 'WARNING: No history output'
   378)       do ns1 = 1, nstreams
   379)          do ns2 = 1, nstreams
   380)             if (histfreq(ns1) == histfreq(ns2) .and. ns1/=ns2 &
   381)                .and. my_task == master_task) then
   382)                call abort_ice('ice: histfreq elements must be unique')
   383)             endif
   384)          enddo
   385)       enddo
   386) 
   387)       if (.not. tr_iage) f_iage   = 'xxxxx'
   388)       if (.not. tr_FY)   f_FY     = 'xxxxx'
   389)       if (.not. tr_pond) then
   390)           f_apond  = 'xxxxx'
   391)           f_apondn = 'xxxxx'
   392)       endif
   393)       if (.not. tr_lvl) then
   394)           f_ardg = 'xxxxx'
   395)           f_vrdg = 'xxxxx'
   396)           f_alvl = 'xxxxx'
   397)           f_vlvl = 'xxxxx'
   398)       endif
   399)       if (.not. tr_aero) then
   400)          f_faero_atm = 'xxxxx'
   401)          f_faero_ocn = 'xxxxx'
   402)          f_aero      = 'xxxxx' 
   403)          f_aeron     = 'xxxxx'
   404)       endif
   405) 
   406)       ! these must be output at the same frequency because of 
   407)       ! cos(zenith angle) averaging
   408)       if (f_albsno(1:1) /= 'x') f_albsno = f_albice
   409)       if (f_albpnd(1:1) /= 'x') f_albpnd = f_albice
   410)       if (f_coszen(1:1) /= 'x') f_coszen = f_albice
   411) 
   412)       ! to prevent array-out-of-bounds when aggregating
   413)       if (f_fmeltt_ai(1:1) /= 'x') f_fmelttn_ai = f_fmeltt_ai
   414) 
   415) #ifndef ncdf
   416)       f_bounds = .false.
   417) #endif
   418) 
   419)       call broadcast_scalar (f_tmask, master_task)
   420)       call broadcast_scalar (f_blkmask, master_task)
   421)       call broadcast_scalar (f_tarea, master_task)
   422)       call broadcast_scalar (f_uarea, master_task)
   423)       call broadcast_scalar (f_dxt, master_task)
   424)       call broadcast_scalar (f_dyt, master_task)
   425)       call broadcast_scalar (f_dxu, master_task)
   426)       call broadcast_scalar (f_dyu, master_task)
   427)       call broadcast_scalar (f_HTN, master_task)
   428)       call broadcast_scalar (f_HTE, master_task)
   429)       call broadcast_scalar (f_ANGLE, master_task)
   430)       call broadcast_scalar (f_ANGLET, master_task)
   431)       call broadcast_scalar (f_bounds, master_task)
   432) 
   433) !     call broadcast_scalar (f_example, master_task)
   434)       call broadcast_scalar (f_hi, master_task)
   435)       call broadcast_scalar (f_hs, master_task)
   436)       call broadcast_scalar (f_fs, master_task)
   437)       call broadcast_scalar (f_Tsfc, master_task)
   438)       call broadcast_scalar (f_aice, master_task)
   439)       call broadcast_scalar (f_uvel, master_task)
   440)       call broadcast_scalar (f_vvel, master_task)
   441)       call broadcast_scalar (f_transix, master_task)
   442)       call broadcast_scalar (f_transiy, master_task)
   443)       call broadcast_scalar (f_fswdn, master_task)
   444)       call broadcast_scalar (f_fswup, master_task)
   445)       call broadcast_scalar (f_flwdn, master_task)
   446)       call broadcast_scalar (f_snow, master_task)
   447)       call broadcast_scalar (f_snow_ai, master_task)
   448)       call broadcast_scalar (f_rain, master_task)
   449)       call broadcast_scalar (f_rain_ai, master_task)
   450)       call broadcast_scalar (f_faero_atm, master_task)
