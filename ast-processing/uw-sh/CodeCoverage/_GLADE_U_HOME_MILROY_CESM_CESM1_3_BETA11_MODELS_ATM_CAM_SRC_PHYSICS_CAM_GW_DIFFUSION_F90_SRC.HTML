    22) subroutine gw_ediff(ncol, pver, ngwv, kbot, ktop, tend_level, &
    23)      gwut, ubm, nm, rho, dt, gravit, pmid, rdpm, c, &
    24)      egwdffi, decomp)
    25) !
    26) ! Calculate effective diffusivity associated with GW forcing.
    27) !
    28) ! Author: F. Sassi, Jan 31, 2001
    29) !
    30)   use gw_utils, only: midpoint_interp
    31)   use vdiff_lu_solver, only: vd_lu_decomp
    32) 
    33) !-------------------------------Input Arguments----------------------------
    34) 
    35)   ! Column, level, and gravity wave spectrum dimensions.
    36)   integer, intent(in) :: ncol, pver, ngwv
    37)   ! Bottom and top levels to operate on.
    38)   integer, intent(in) :: kbot, ktop
    39)   ! Per-column bottom index where tendencies are applied.
    40)   integer, intent(in) :: tend_level(ncol)
    41)   ! GW zonal wind tendencies at midpoint.
    42)   real(r8), intent(in) :: gwut(ncol,pver,-ngwv:ngwv)
    43)   ! Projection of wind at midpoints.
    44)   real(r8), intent(in) :: ubm(ncol,pver)
    45)   ! Brunt-Vaisalla frequency.
    46)   real(r8), intent(in) :: nm(ncol,pver)
    47) 
    48)   ! Density at interfaces.
    49)   real(r8), intent(in) :: rho(ncol,pver+1)
    50)   ! Time step.
    51)   real(r8), intent(in) :: dt
    52)   ! Acceleration due to gravity.
    53)   real(r8), intent(in) :: gravit
    54)   ! Midpoint pressure.
    55)   real(r8), intent(in) :: pmid(ncol,pver)
    56)   ! 1./pdel (pdel is thickness between interfaces).
    57)   real(r8), intent(in) :: rdpm(ncol,pver)
    58)   ! Wave phase speeds for each column.
    59)   real(r8), intent(in) :: c(ncol,-ngwv:ngwv)
    60) 
    61) !-----------------------------Output Arguments-----------------------------
    62)   ! Effective gw diffusivity at interfaces.
    63)   real(r8), intent(out) :: egwdffi(ncol,0:pver)
    64)   ! LU decomposition.
    65)   type(lu_decomp), intent(out) :: decomp
    66) 
    67) !-----------------------------Local Workspace------------------------------
    68) 
    69)   ! 0._r8 (array to pass to vd_lu_decomp).
    70)   real(r8) :: zero(ncol)
    71)   ! Effective gw diffusivity at midpoints.
    72)   real(r8) :: egwdffm(ncol,pver)
    73)   ! dt * (gravit*rho)^2/dp
    74)   real(r8) :: tmpi2(ncol,pver+1)
    75)   ! Level and wave indices.
    76)   integer :: k, l
    77)   ! Inverse Prandtl number.
    78)   real(r8), parameter :: prndl=0.25_r8
    79)   ! Density scale height.
    80)   real(r8), parameter :: dscale=7000._r8
    81) 
    82) !--------------------------------------------------------------------------
    83) 
    84)   zero = 0.0_r8
    85) 
    86)   egwdffi = 0._r8
    87)   egwdffm = 0._r8
    88) 
    89)   ! Calculate effective diffusivity at midpoints.
    90)   do l = -ngwv, ngwv
    91)      do k = ktop+1, kbot
    92) 
    93)         egwdffm(:,k) = egwdffm(:,k) + &
    94)              prndl * 0.5_r8 * gwut(:,k,l) * (c(:,l)-ubm(:,k)) / nm(:,k)**2
    95) 
    96)      end do
    97)   end do
    98) 
    99) 
   100)   ! Interpolate effective diffusivity to interfaces.
   101)   ! Assume zero at top and bottom interfaces.
   102)   egwdffi(:,ktop+1:kbot-1) = midpoint_interp(egwdffm(:,ktop+1:kbot))
   103) 
   104)   ! Limit diffusivity to some reasonable value.
   105)   egwdffi = min(150._r8, egwdffi)
   106) 
   107)   ! Do not calculate diffusivities below level where tendencies are
   108)   ! actually allowed.
   109)   do k= ktop, kbot
   110)      where (k >= tend_level) egwdffi(:,k) = 0.0_r8
   111)   enddo
   112) 
   113)   ! Calculate dt * (gravit*rho)^2/dp at interior interfaces.
   114)   tmpi2 = 0.0_r8
   115)   do k = ktop+2 , kbot+1
   116)      tmpi2(:,k) = dt * (gravit * rho(:,k))**2 / (pmid(:,k) - pmid(:,k-1))
   117)   end do
   118) 
   119)   ! Decompose the diffusion matrix.
   120)   ! Note that [ktop,kbot] are model interfaces (beginning at zero), whereas
   121)   ! in vd_lu_decomp they are expected as midpoints.
   122)   call vd_lu_decomp(ncol, pver, ncol, &
   123)        zero, egwdffi, tmpi2, rdpm, dt, gravit, zero, ktop+1, kbot+1, decomp)
   124) 
   125) end subroutine gw_ediff

   129) subroutine gw_diff_tend(ncol, pver, kbot, ktop, q, dt, decomp, dq)
   130) 
   131) !
   132) ! Calculates tendencies from effective diffusion due to gravity wave
   133) ! breaking.
   134) !
   135) ! Method:
   136) ! A constituent flux on interfaces is given by:
   137) !
   138) !              rho * (w'q') = rho * Deff qz
   139) !
   140) ! where (all evaluated on interfaces):
   141) !
   142) !        rho   = density
   143) !        qz    = constituent vertical gradient
   144) !        Deff  = effective diffusivity
   145) !
   146) ! An effective diffusivity is calculated by adding up the diffusivities
   147) ! from all waves (see gw_ediff). The tendency is calculated by invoking LU
   148) ! decomposition and solving as for a regular diffusion equation.
   149) !
   150) ! Author: Sassi - Jan 2001
   151) !--------------------------------------------------------------------------
   152) 
   153)   use vdiff_lu_solver, only: vd_lu_solve
   154) 
   155) !---------------------------Input Arguments--------------------------------
   156) 
   157)   ! Column and level dimensions.
   158)   integer, intent(in) :: ncol, pver
   159)   ! Bottom and top levels to operate on.
   160)   integer, intent(in) :: kbot, ktop
   161) 
   162)   ! Constituent to diffuse.
   163)   real(r8), intent(in) :: q(ncol,pver)
   164)   ! Time step.
   165)   real(r8), intent(in) :: dt
   166) 
   167)   ! LU decomposition.
   168)   type(lu_decomp), intent(in) :: decomp
   169) 
   170) !--------------------------Output Arguments--------------------------------
   171) 
   172)   ! Constituent tendencies.
   173)   real(r8), intent(out) :: dq(ncol,pver)
   174) 
   175) !--------------------------Local Workspace---------------------------------
   176) 
   177)   ! Temporary storage for constituent.
   178)   real(r8) :: qnew(ncol,pver)
   179)   ! 0._r8 (array to pass to vd_lu_decomp).
   180)   real(r8) :: zero(ncol)
   181) 
   182) !--------------------------------------------------------------------------
   183)   zero = 0.0_r8
   184)   dq   = 0.0_r8
   185)   qnew = q
   186) 
   187)   ! Solve the diffusion matrix.
   188)   call vd_lu_solve(ncol, pver, ncol, qnew, decomp, ktop+1, kbot+1, zero)
   189) 
   190)   ! Evaluate tendency to be reported back.
   191)   dq = (qnew-q) / dt
   192) 
   193) end subroutine gw_diff_tend
   194) 
   195) end module gw_diffusion
