   151)   subroutine ubc_init()
   152) !-----------------------------------------------------------------------
   153) ! Initialization of time independent fields for the upper boundary condition
   154) ! Calls initialization routine for MSIS, TGCM and SNOE
   155) !-----------------------------------------------------------------------
   156)     use mo_tgcm_ubc, only: tgcm_ubc_inti
   157)     use mo_snoe,     only: snoe_inti
   158)     use mo_msis_ubc, only: msis_ubc_inti
   159)     use time_manager,only: get_step_size
   160)     use constituents,only: cnst_get_ind, cnst_mw !Needed for ubc_flux
   161)     use physics_buffer, only : physics_buffer_desc
   162) 
   163) !---------------------------Local workspace-----------------------------
   164)     integer :: steps_per_day
   165)     logical :: zonal_avg
   166) !-----------------------------------------------------------------------
   167) 
   168)     if (.not.waccm) return
   169) 
   170)     call cnst_get_ind('F', f_ndx, abort=.false.)
   171)     call cnst_get_ind('HF', hf_ndx, abort=.false.)
   172) 
   173)     steps_per_day = 86400/get_step_size()
   174)     zonal_avg     = .false.
   175) 
   176) !-----------------------------------------------------------------------
   177) !	... initialize the tgcm upper boundary module
   178) !-----------------------------------------------------------------------
   179)     call tgcm_ubc_inti( tgcm_ubc_file, tgcm_ubc_data_type, tgcm_ubc_cycle_yr, tgcm_ubc_fixed_ymd, tgcm_ubc_fixed_tod)
   180)     if (masterproc) write(iulog,*) 'ubc_init: after tgcm_ubc_inti'
   181) 
   182) !-----------------------------------------------------------------------
   183) !	... initialize the snoe module
   184) !-----------------------------------------------------------------------
   185)     call snoe_inti(snoe_ubc_file)
   186)     if (masterproc) write(iulog,*) 'ubc_init: after snoe_inti'
   187) 
   188) !-----------------------------------------------------------------------
   189) !	... initialize the msis module
   190) !-----------------------------------------------------------------------
   191)     call msis_ubc_inti( zonal_avg, steps_per_day )
   192)     if (masterproc) write(iulog,*) 'ubc_init: after msis_ubc_inti'
   193) 
   194)   end subroutine ubc_init

   227)   subroutine ubc_get_vals (lchnk, ncol, ntop_molec, pint, zi, msis_temp, ubc_mmr, ubc_flux)
   228) !-----------------------------------------------------------------------
   229) ! interface routine for vertical diffusion and pbl scheme
   230) !-----------------------------------------------------------------------
   231)     use mo_msis_ubc,  only: get_msis_ubc
   232)     use mo_snoe,      only: set_no_ubc, ndx_no
   233)     use mo_tgcm_ubc,  only: set_tgcm_ubc
   234)     use abortutils,   only: endrun
   235)     use physconst,    only: rairv, mbarv
   236)     use phys_control, only: waccmx_is
   237)     use physconst,    only: avogad
   238)     use constituents, only: cnst_get_ind, cnst_mw  ! Needed for ubc_flux
   239)   
   240) !------------------------------Arguments--------------------------------
   241)     integer,  intent(in)  :: lchnk                 ! chunk identifier
   242)     integer,  intent(in)  :: ncol                  ! number of atmospheric columns
   243)     integer,  intent(in)  :: ntop_molec            ! top of molecular diffusion region (=1)
   244)     real(r8), intent(in)  :: pint(pcols,pverp)     ! interface pressures
   245)     real(r8), intent(in)  :: zi(pcols,pverp)       ! interface geoptl height above sfc
   246) 
   247)     real(r8), intent(out) :: ubc_mmr(pcols,pcnst)  ! upper bndy mixing ratios (kg/kg)
   248)     real(r8), intent(out) :: msis_temp(pcols)      ! upper bndy temperature (K)
   249)     real(r8), intent(out) :: ubc_flux(pcnst)       ! upper bndy flux (kg/s/m^2)
   250) 
   251) !---------------------------Local storage-------------------------------
   252)     integer  :: ierr                               ! error flag for allocates
   253)     integer  :: indx_H                             ! cnst index for H
   254) 
   255)     real(r8), parameter :: m2km = 1.e-3_r8         ! meter to km
   256)     real(r8) :: rho_top(pcols)                     ! density at top interface
   257)     real(r8) :: z_top(pcols)                       ! height of top interface (km)
   258) 
   259) !-----------------------------------------------------------------------
   260) 
   261)     if (.not.waccm) return
   262) 
   263)     ubc_mmr(:,:) = 0._r8
   264)     ubc_flux(:) = 0._r8
   265) 
   266)     call get_msis_ubc( lchnk, ncol, msis_temp, ubc_mmr )
   267)     if( t_pert_ubc /= 0._r8 ) then
   268)        msis_temp(:ncol) = msis_temp(:ncol) + t_pert_ubc
   269)        if( any( msis_temp(:ncol) < 0._r8 ) ) then
   270)           write(iulog,*) 'ubc_get_vals: msis temp < 0 after applying offset = ',t_pert_ubc
   271)           call endrun
   272)        end if
   273)     end if
   274) 
   275)     rho_top(:ncol) = pint(:ncol,ntop_molec) / (rairv(:ncol,ntop_molec,lchnk)*msis_temp(:ncol))
   276)     z_top(:ncol)   = m2km * zi(:ncol,ntop_molec)
   277)     
   278)     if ( any( z_top(:ncol) < 180.0_r8 ) ) then  
   279)        call set_no_ubc  ( lchnk, ncol, z_top, ubc_mmr, rho_top )
   280)        if( ndx_no > 0 .and. no_xfac_ubc /= 1._r8 ) then
   281)           ubc_mmr(:ncol,ndx_no) = no_xfac_ubc * ubc_mmr(:ncol,ndx_no)
   282)        end if
   283)     else
   284)        ! waccmx
   285)        ubc_mmr(:ncol,ndx_no) = 0.0_r8
   286)     endif
   287)     
   288)     !--------------------------------------------------------------------------------------------
   289)     ! For WACCM-X, calculate upper boundary H flux
   290)     !--------------------------------------------------------------------------------------------
   291)     if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   292)     
   293)       call cnst_get_ind('H',  indx_H)
   294) 
   295)       ubc_flux(indx_H) = cnst_mw(indx_H)/ avogad * 0.9e12_r8      ! upward mass flux of H, kg/s/m^2
   296)    
   297)     endif     
   298) 
   299)     call set_tgcm_ubc( lchnk, ncol, ubc_mmr, mbarv(:,ntop_molec,lchnk))
   300) 
   301)     if (f_ndx .GT. 0) then
   302)       ubc_mmr(:ncol, f_ndx) = 1.0e-15_r8
   303)     endif
   304)     if (hf_ndx .GT. 0) then
   305)       ubc_mmr(:ncol, hf_ndx) = 1.0e-15_r8
   306)     endif
   307) 
   308)   end subroutine ubc_get_vals
