     1) Module DryDepVelocity                                              
     2) 
     3)   !-----------------------------------------------------------------------  
     4)   !  
     5)   ! Purpose:  
     6)   ! Deposition velocity (m/s) 
     7)   !  
     8)   ! Method:  
     9)   ! This code simulates dry deposition velocities using the Wesely scheme.   
    10)   ! Details of this method can be found in:  
    11)   ! 
    12)   ! M.L Wesely. Parameterization of surface resistances to gaseous dry deposition 
    13)   ! in regional-scale numericl models. 1989. Atmospheric Environment vol.23 No.6  
    14)   ! pp. 1293-1304. 
    15)   ! 
    16)   ! In Wesely (1998) "the magnitude of the dry deposition velocity can be found 
    17)   ! as: 
    18)   ! 
    19)   !  |vd|=(ra+rb+rc)^-1 
    20)   ! 
    21)   ! where ra is the aerodynamic resistance (common to all gases) between a  
    22)   ! specific height and the surface, rb is the quasilaminar sublayer resistance 
    23)   ! (whose only dependence on the porperties of the gas of interest is its 
    24)   ! molecular diffusivity in air), and rc is the bulk surface resistance". 
    25)   ! 
    26)   ! In this subroutine both ra and rb are calculated elsewhere in CLM.  Thus ra  
    27)   ! and rb were "globalized" in order to gain access to them for the calculation. 
    28)   ! "ram1" is the CLM variable used for ra.  ram1 was globalized in the following 
    29)   ! subroutines; BareGroundFluxes.F90, Biogeophysics_lake.F90, CanopyFluxes.F90, 
    30)   ! and clmtype.F90.  
    31)   ! 
    32)   ! "rb" is the CLM variable used for rb in the Wesely equation above.  rb was  
    33)   ! globalized in the following subroutines; clmtype.F90     
    34)   ! 
    35)   ! In Wesely (1989) rc is estimated for five seasonal categories and 11 landuse 
    36)   ! types.  For each season and landuse type, Wesely compiled data into a  
    37)   ! look-up-table for several parameters used to calculate rc. In this subroutine 
    38)   ! the same values are used as found in wesely's look-up-tables, the only  
    39)   ! difference is that this subroutine uses a CLM generated LAI to select values 
    40)   ! from the look-up-table instead of seasonality.  Inaddition, Wesely(1989)  
    41)   ! land use types are "mapped" into CLM plant function types (PFT). 
    42)   ! 
    43)   ! Subroutine written to operate at the patch level. 
    44)   ! 
    45)   ! Output: 
    46)   ! 
    47)   ! vd(n_species) !Dry deposition velocity [m s-1] for each molecule or species 
    48)   !  
    49)   ! Author: Beth Holland and  James Sulzman 
    50)   ! 
    51)   ! Modified: Francis Vitt -- 30 Mar 2007
    52)   !----------------------------------------------------------------------- 
    53) 
    54)   use shr_kind_mod,         only : r8 => shr_kind_r8 
    55)   use clmtype 
    56)   use abortutils,           only : endrun
    57)   use clm_time_manager,     only : get_nstep, get_curr_date, get_curr_time 
    58)   use clm_atmlnd,           only : clm_a2l
    59)   use spmdMod,              only : masterproc
    60)   use seq_drydep_mod,       only : n_drydep, drydep_list
    61)   use seq_drydep_mod,       only : drydep_method, DD_XLND
    62)   use seq_drydep_mod,       only : index_o3=>o3_ndx, index_o3a=>o3a_ndx, index_so2=>so2_ndx, index_h2=>h2_ndx, &
    63)                                    index_co=>co_ndx, index_ch4=>ch4_ndx, index_pan=>pan_ndx, &
    64)                                    index_xpan=>xpan_ndx
    65)   implicit none 
    66)   save 
    67) 
    68)   private
    69) 
    70)   public :: depvel_compute
    71) 
    72) CONTAINS 
    73) 
    74)   !----------------------------------------------------------------------- 
    75)   ! computes the dry deposition velocity of tracers
    76)   !----------------------------------------------------------------------- 
    77)   subroutine depvel_compute( lbp , ubp ) 
    78)     use shr_const_mod     , only :  tmelt => shr_const_tkfrz
    79)     use seq_drydep_mod    , only :  seq_drydep_setHCoeff, mapping, drat, foxd, &
    80)                                     rcls, h2_a, h2_b, h2_c, ri, rac, rclo, rlu, &
    81)                                     rgss, rgso
    82)     use clm_varcon        , only : istsoil, istice, istice_mec, istslak, istdlak, istwet, isturb
    83)     use clm_varctl        , only : iulog
    84)     use pftvarcon         , only : noveg, ndllf_evr_tmp_tree, ndllf_evr_brl_tree,   &
    85)                                    ndllf_dcd_brl_tree,        nbrdlf_evr_trp_tree,  &
    86)                                    nbrdlf_evr_tmp_tree,       nbrdlf_dcd_trp_tree,  &
    87)                                    nbrdlf_dcd_tmp_tree,       nbrdlf_dcd_brl_tree,  &
    88)                                    nbrdlf_evr_shrub,          nbrdlf_dcd_tmp_shrub, &
    89)                                    nbrdlf_dcd_brl_shrub,      nc3_arctic_grass,     &
    90)                                    nc3_nonarctic_grass,       nc4_grass, nc3crop,   &
    91)                                    nirrig,         npcropmin, npcropmax
    92) 
    93)     implicit none 
    94) 
    95)     !----Arguments-----------------------------------------------------
    96) 
    97)     integer, intent(in) :: lbp, ubp                    ! pft bounds
    98) 
    99)     ! ------------------------ local variables ------------------------ 
   100)     ! local pointers to implicit in arguments 
   101)     integer , pointer :: plandunit(:)     !pft's landunit index
   102)     integer , pointer :: ivt(:)           !landunit type
   103)     integer , pointer :: itypveg(:)       !vegetation type for current pft  
   104)     integer , pointer :: pgridcell(:)     !pft's gridcell index
   105)     real(r8), pointer :: pwtgcell(:)      !weight of pft relative to corresponding gridcell
   106)     real(r8), pointer :: elai(:)          !one-sided leaf area index with burying by snow 
   107)     real(r8), pointer :: forc_t(:)        !atmospheric temperature (Kelvin) 
   108)     real(r8), pointer :: forc_q(:)        !atmospheric specific humidity (kg/kg) 
   109)     real(r8), pointer :: forc_psrf(:)     !surface pressure (Pa) 
   110)     real(r8), pointer :: latdeg(:)        !latitude (degrees) 
   111)     real(r8), pointer :: londeg(:)        !longitude (degrees) 
   112)     real(r8), pointer :: forc_rain(:)     !rain rate [mm/s] 
   113)     real(r8), pointer :: forc_snow(:)     !snow rate [mm/s]   
   114)     real(r8), pointer :: forc_lwrad(:)    !direct beam radiation (visible only) 
   115)     real(r8), pointer :: forc_solad(:,:)  !direct beam radiation (visible only) 
   116)     real(r8), pointer :: forc_solai(:,:)  !direct beam radiation (visible only) 
   117)     real(r8), pointer :: ram1(:)          !aerodynamical resistance 
   118)     real(r8), pointer :: vds(:)           !aerodynamical resistance 
   119)     real(r8), pointer :: rssun(:)         !stomatal resistance 
   120)     real(r8), pointer :: rssha(:)         !shaded stomatal resistance (s/m) 
   121)     real(r8), pointer :: fsun(:)          !sunlit fraction of canopy 
   122)     real(r8), pointer :: rb1(:)           !leaf boundary layer resistance [s/m]
   123)     real(r8), pointer :: annlai(:,:)      !12 months of monthly lai from input data set 
   124)     real(r8), pointer :: mlaidiff(:)      !difference in lai between month one and month two 
   125)     real(r8), pointer :: velocity(:,:)
   126)     real(r8), pointer :: snowdp(:)        ! snow height (m)
   127) 
   128)     integer, pointer :: pcolumn(:)        ! column index associated with each pft
   129)     integer :: c
   130)     integer , pointer :: itypelun(:) 	   ! landunit type
   131) 
   132)     real(r8), pointer :: h2osoi_vol(:,:)    ! volumetric soil water (0<=h2osoi_vol<=watsat)
   133)     real(r8) :: soilw, var_soilw, fact_h2, dv_soil_h2
   134) 
   135)     ! new local variables  
   136)     integer :: pi,g, l
   137)     integer :: ispec 
   138)     integer :: length 
   139)     integer :: wesveg              !wesely vegegation index  
   140)     integer :: clmveg              !clm veg index from ivegtype 
   141)     integer :: i 
   142)     integer :: index_season        !seasonal index based on LAI.  This indexs wesely data tables 
   143)     integer :: nstep               !current step 
   144)     integer :: indexp 
   145) 
   146)     real(r8) :: pg          ! surface pressure 
   147)     real(r8) :: tc          ! temperature in celsius  
   148)     real(r8) :: rs          ! constant for calculating rsmx  
   149)     real(r8) :: es          ! saturation vapor pressur 
   150)     real(r8) :: ws          ! saturation mixing ratio 
   151)     real(r8) :: rmx         ! resistance by vegetation 
   152)     real(r8) :: qs          ! saturation specific humidity 
   153)     real(r8) :: dewm        ! multiplier for rs when dew occurs 
   154)     real(r8) :: crs         ! multiplier to calculate crs 
   155)     real(r8) :: rdc         ! part of lower canopy resistance 
   156)     real(r8) :: rain        ! rain fall 
   157)     real(r8) :: spec_hum    ! specific humidity 
   158)     real(r8) :: solar_flux  ! solar radiation(direct beam) W/m2 
   159)     real(r8) :: lat         ! latitude in degrees 
   160)     real(r8) :: lon         ! longitude in degrees 
   161)     real(r8) :: sfc_temp    ! surface temp 
   162)     real(r8) :: minlai      ! minimum of monthly lai 
   163)     real(r8) :: maxlai      ! maximum of monthly lai 
   164)     real(r8) :: rds
   165) 
   166)     logical  :: has_dew
   167)     logical  :: has_rain
   168)     real(r8), parameter :: rain_threshold      = 1.e-7_r8  ! of the order of 1cm/day expressed in m/s
   169) 
   170)     ! local arrays: dependent on species only 
   171)     ! 
   172) 
   173)     real(r8), dimension(n_drydep) :: rsmx !vegetative resistance (plant mesophyll) 
   174)     real(r8), dimension(n_drydep) :: rclx  !lower canopy resistance  
   175)     real(r8), dimension(n_drydep) :: rlux  !vegetative resistance (upper canopy) 
   176)     real(r8), dimension(n_drydep) :: rgsx  !gournd resistance 
   177)     real(r8), dimension(n_drydep) :: heff   
   178)     real(r8) :: rc    !combined surface resistance 
   179)     real(r8) :: cts   !correction to flu rcl and rgs for frost 
   180)     real(r8) :: rlux_o3
   181) 
   182)     ! constants 
   183)     real(r8), parameter :: slope = 0._r8      ! Used to calculate  rdc in (lower canopy resistance) 
   184)     integer, parameter :: wveg_unset = -1     ! Unset Wesley vegetation type
   185) 
   186)     character(len=32), parameter :: subname = "depvel_compute"
   187) 
   188)     !-------------------------------------------------------------------------------------
   189)     ! jfl : mods for PAN
   190)     !-------------------------------------------------------------------------------------
   191)     real(r8)                  :: dv_pan
   192)     real(r8) :: c0_pan(11) = (/ 0.000_r8, 0.006_r8, 0.002_r8, 0.009_r8, 0.015_r8, &
   193)                                 0.006_r8, 0.000_r8, 0.000_r8, 0.000_r8, 0.002_r8, 0.002_r8 /)
   194)     real(r8) :: k_pan (11) = (/ 0.000_r8, 0.010_r8, 0.005_r8, 0.004_r8, 0.003_r8, &
   195)                                 0.005_r8, 0.000_r8, 0.000_r8, 0.000_r8, 0.075_r8, 0.002_r8 /)
   196)     !----------------------------------------------------------------------- 
   197)     if ( n_drydep == 0 .or. drydep_method /= DD_XLND ) return
   198) 
   199)     ! local pointers to original implicit out arrays 
   200) 
   201)     ! Assign local pointers to derived subtypes components (column-level) 
   202)     forc_t     => clm_a2l%forc_t
   203)     forc_q     => clm_a2l%forc_q
   204)     forc_psrf  => clm_a2l%forc_pbot
   205)     forc_rain  => clm_a2l%forc_rain 
   206) 
   207)     latdeg     => grc%latdeg
   208)     londeg     => grc%londeg
   209)     ivt        => pft%itype
   210)     elai       => pps%elai 
   211)     ram1       => pps%ram1 
   212)     vds        => pps%vds
   213)     fsun       => pps%fsun 
   214)     rssun      => pps%rssun 
   215)     rssha      => pps%rssha 
   216)     rb1        => pps%rb1 
   217)     mlaidiff   => pps%mlaidiff 
   218)     annlai     => pps%annlai    
   219) 
   220)     forc_solai => clm_a2l%forc_solai 
   221)     forc_solad => clm_a2l%forc_solad
   222) 
   223)     pwtgcell   => pft%wtgcell  
   224)     pgridcell  => pft%gridcell
   225)     plandunit  => pft%landunit
   226) 
   227)     pcolumn    => pft%column
   228)     itypelun   => lun%itype
   229) 
   230)     h2osoi_vol => cws%h2osoi_vol
   231) 
   232)     velocity   => pdd%drydepvel ! cm/sec
   233) 
   234)     snowdp        => cps%snowdp
   235) 
   236)     ! Assign local pointers to original implicit out arrays 
   237)     !_________________________________________________________________ 
   238)     ! 
   239)     ! Begin loop through pfts 
   240)     pft_loop: do pi = lbp,ubp
   241)        l = plandunit(pi)
   242) 
   243)        ! Note: some glacier_mec pfts may have zero weight
   244)        gcell_wght: if (pwtgcell(pi)>0._r8 .or. itypelun(l)==istice_mec) then
   245) 
   246)           c = pcolumn(pi)
   247)           g = pgridcell(pi)
   248)           !solar_flux = forc_lwrad  !rename CLM variables to fit with Dry Dep variables 
   249) 
   250)           pg         = forc_psrf(g)  
   251)           spec_hum   = forc_q(g)
   252)           rain       = forc_rain(g) 
   253)           sfc_temp   = forc_t(g) 
   254)           lat        = latdeg(g) 
   255)           lon        = londeg(g) 
   256)           solar_flux = forc_solad(g,1) 
   257)           clmveg     = ivt(pi) 
   258)           soilw = h2osoi_vol(c,1)
   259) 
   260)           !  print *,'bb',pi,cps%npfts,lat,lon,clmveg 
   261)           !map CLM veg type into Wesely veg type  
   262)           wesveg = wveg_unset 
   263)           if (clmveg == noveg                               ) wesveg = 8 
   264)           if (clmveg == ndllf_evr_tmp_tree                  ) wesveg = 5 
   265)           if (clmveg == ndllf_evr_brl_tree                  ) wesveg = 5 
   266)           if (clmveg == ndllf_dcd_brl_tree                  ) wesveg = 5 
   267)           if (clmveg == nbrdlf_evr_trp_tree                 ) wesveg = 4 
   268)           if (clmveg == nbrdlf_evr_tmp_tree                 ) wesveg = 4 
   269)           if (clmveg == nbrdlf_dcd_trp_tree                 ) wesveg = 4 
   270)           if (clmveg == nbrdlf_dcd_tmp_tree                 ) wesveg = 4 
   271)           if (clmveg == nbrdlf_dcd_brl_tree                 ) wesveg = 4 
   272)           if (clmveg == nbrdlf_evr_shrub                    ) wesveg = 11 
   273)           if (clmveg == nbrdlf_dcd_tmp_shrub                ) wesveg = 11 
   274)           if (clmveg == nbrdlf_dcd_brl_shrub                ) wesveg = 11 
   275)           if (clmveg == nc3_arctic_grass                    ) wesveg = 3 
   276)           if (clmveg == nc3_nonarctic_grass                 ) wesveg = 3 
   277)           if (clmveg == nc4_grass                           ) wesveg = 3 
   278)           if (clmveg == nc3crop                             ) wesveg = 2 
   279)           if (clmveg == nirrig                              ) wesveg = 2 
   280)           if (clmveg >= npcropmin .and. clmveg <= npcropmax ) wesveg = 2 
   281)           if (wesveg == wveg_unset )then
   282)              write(iulog,*) 'clmveg = ', clmveg, 'itypelun = ', itypelun(l)
   283)              call endrun( subname//': Not able to determine Wesley vegetation type')
   284)           end if
   285) 
   286)           ! creat seasonality index used to index wesely data tables from LAI,  Bascially 
   287)           !if elai is between max lai from input data and half that max the index_season=1 
   288) 
   289) 
   290)           !mail1j and mlai2j are the two monthly lai values pulled from a CLM input data set 
   291)           !/fs/cgd/csm/inputdata/lnd/clm2/rawdata/mksrf_lai.nc.  lai for dates in the middle  
   292)           !of the month are interpolated using using these values and stored in the variable  
   293)           !elai (done elsewhere).  If the difference between mlai1j and mlai2j is greater 
   294)           !than zero it is assumed to be fall and less than zero it is assumed to be spring. 
   295) 
   296)           !wesely seasonal "index_season" 
   297)           ! 1 - midsummer with lush vegetation 
   298)           ! 2 - Autumn with unharvested cropland 
   299)           ! 3 - Late autumn after frost, no snow 
   300)           ! 4 - Winter, snow on ground and subfreezing 
   301)           ! 5 - Transitional spring with partially green short annuals 
   302) 
   303) 
   304)           !mlaidiff=jan-feb 
   305)           minlai=minval(annlai(:,pi)) 
   306)           maxlai=maxval(annlai(:,pi)) 
   307) 
   308)           index_season = -1
   309) 
   310)           if ( itypelun(l) /= istsoil )then
   311)              if ( itypelun(l) == istice .or. itypelun(l) == istice_mec ) then
   312)                 wesveg       = 8
   313)                 index_season = 4
   314)              elseif ( itypelun(l) == istdlak .or. itypelun(l) == istslak ) then
   315)                 wesveg       = 7
   316)                 index_season = 4
   317)              elseif ( itypelun(l) == istwet ) then
   318)                 wesveg       = 9
   319)                 index_season = 2
   320)              elseif ( itypelun(l) == isturb ) then
   321)                 wesveg       = 1
   322)                 index_season = 2
   323)              end if
   324)           else if ( snowdp(c) > 0 ) then
   325)              index_season = 4
   326)           else if(elai(pi).gt.0.5_r8*maxlai) then  
   327)              index_season = 1  
   328)           endif
   329) 
   330)           if (index_season<0) then 
   331)              if (elai(pi).lt.(minlai+0.05*(maxlai-minlai))) then  
   332)                 index_season = 3 
   333)              endif
   334)           endif
   335) 
   336)           if (index_season<0) then 
   337)              if (mlaidiff(pi).gt.0.0_r8) then 
   338)                 index_season = 2 
   339)              elseif (mlaidiff(pi).lt.0.0_r8) then 
   340)                 index_season = 5 
   341)              elseif (mlaidiff(pi).eq.0.0_r8) then 
   342)                 index_season = 3 
   343)              endif
   344)           endif
   345) 
   346)           if (index_season<0) then 
   347)              call endrun( subname//': not able to determine season')
   348)           endif
   349) 
   350)           ! saturation specific humidity 
   351)           ! 
   352)           es = 611_r8*exp(5414.77_r8*((1._r8/tmelt)-(1._r8/sfc_temp))) 
   353)           ws = .622_r8*es/(pg-es) 
   354)           qs = ws/(1._r8+ws) 
   355) 
   356)           has_dew = .false.
   357)           if( qs <= spec_hum ) then
   358)              has_dew = .true.
   359)           end if
   360)           if( sfc_temp < tmelt ) then
   361)              has_dew = .false.
   362)           end if
   363) 
   364)           has_rain = rain > rain_threshold
   365) 
   366)           if ( has_dew .or. has_rain ) then
   367)              dewm = 3._r8
   368)           else
   369)              dewm = 1._r8
   370)           end if
   371) 
   372)           ! 
   373)           ! constant in determining rs 
   374)           ! 
   375)           crs = 1.e36_r8
   376) 
   377)           tc = sfc_temp - tmelt 
   378)           if(sfc_temp.gt.tmelt.and.sfc_temp.lt.313.15_r8) then 
   379)              crs = (1._r8+(200._r8/(solar_flux+.1_r8))**2) * (400._r8/(tc*(40._r8-tc))) 
   380)           endif
   381)           ! 
   382)           ! rdc (lower canopy res) 
   383)           ! 
   384)           rdc=100._r8*(1._r8+1000._r8/(solar_flux+10._r8))/(1._r8+1000._r8*slope) 
   385) 
   386)           ! surface resistance : depends on both land type and species 
   387)           ! land types are computed seperately, then resistance is computed as average of values 
   388)           ! following wesely rc=(1/(rs+rm) + 1/rlu +1/(rdc+rcl) + 1/(rac+rgs))**-1 
   389)           ! 
   390)           ! compute rsmx = 1/(rs+rm) : multiply by 3 if surface is wet 
   391)           ! 
   392) 
   393)           !******************************************************* 
   394)           call seq_drydep_setHCoeff( sfc_temp, heff(:n_drydep) )
   395)           !********************************************************* 
   396) 
   397)           species_loop1: do ispec=1, n_drydep
   398)              if(mapping(ispec).le.0) cycle 
   399) 
   400)              if(ispec.eq.index_o3.or.ispec.eq.index_o3a.or.ispec.eq.index_so2) then 
   401)                 rmx=0._r8
   402)              else 
   403)                 rmx=1._r8/((heff(ispec)/3000._r8)+(100._r8*foxd(ispec))) 
   404)              endif
   405) 
   406)              ! correction for frost 
   407)              cts = 1000._r8*exp( -tc - 4._r8 )                 ! correction for frost
   408)              rgsx(ispec) = cts + 1._r8/((heff(ispec)/(1.e5_r8*rgss(index_season,wesveg))) + & 
   409)                                         (foxd(ispec)/rgso(index_season,wesveg))) 
   410) 
   411)              !-------------------------------------------------------------------------------------
   412)              ! special case for H2 and CO;; CH4 is set ot a fraction of dv(H2)
   413)              !-------------------------------------------------------------------------------------
   414)              if( ispec == index_h2 .or. ispec == index_co .or. ispec == index_ch4 ) then
   415) 
   416)                 if( ispec == index_co ) then
   417)                    fact_h2 = 1.0_r8
   418)                 elseif ( ispec == index_h2 ) then
   419)                    fact_h2 = 0.5_r8
   420)                 elseif ( ispec == index_ch4 ) then
   421)                    fact_h2 = 50.0_r8
   422)                 end if
   423)                 !-------------------------------------------------------------------------------------
   424)                 ! no deposition on snow, ice, desert, and water
   425)                 !-------------------------------------------------------------------------------------
   426)                 if( wesveg == 1 .or. wesveg == 7 .or. wesveg == 8 .or. index_season == 4 ) then
   427)                    rgsx(ispec) = 1.e36_r8
   428)                 else
   429)                    var_soilw = max( .1_r8,min( soilw,.3_r8 ) )
   430)                    if( wesveg == 3 ) then
   431)                       var_soilw = log( var_soilw )
   432)                    end if
   433)                    dv_soil_h2 = h2_c(wesveg) + var_soilw*(h2_b(wesveg) + var_soilw*h2_a(wesveg))
   434)                    if( dv_soil_h2 > 0._r8 ) then
   435)                       rgsx(ispec) = fact_h2/(dv_soil_h2*1.e-4_r8)
   436)                    end if
   437)                 end if
   438)              end if
   439) 
   440)              rs=(fsun(pi)*rssun(pi))+(rssha(pi)*(1._r8-fsun(pi)))
   441)              
   442)              !-------------------------------------------------------------------------------------
   443)              ! no deposition on snow, ice, desert, and water
   444)              !-------------------------------------------------------------------------------------
