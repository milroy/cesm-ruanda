   104) subroutine cam3_ozone_data_init(phys_state)
   105) !----------------------------------------------------------------------- 
   106) ! 
   107) ! Purpose: Do initial read of time-variant ozone boundary dataset, containing
   108) !          ozone mixing ratios as a function of latitude and pressure.  Read two
   109) !          consecutive months between which the current date lies.  Routine
   110) !          RADOZ2 then evaluates the two path length integrals (with and without
   111) !          pressure weighting) from zero to the interfaces between the input
   112) !          levels.  It also stores the contribution to the integral from each
   113) !          layer.
   114) ! 
   115) ! Method: Call appropriate netcdf wrapper routines and interpolate to model grid
   116) ! 
   117) ! Author: CCM Core Group
   118) ! Modified: P. Worley, August 2003, for chunking and performance optimization
   119) !           J. Edwards, Dec 2005, functionality now performed by zonalbndrydata
   120) !-----------------------------------------------------------------------
   121) 
   122)    use cam_history,      only: addfld, phys_decomp
   123) 
   124)    type(physics_state), intent(in) :: phys_state(begchunk:endchunk) 
   125)    !-----------------------------------------------------------------------
   126)     
   127)    call addfld ('O3VMR', 'm3/m3', pver, 'A', 'Ozone volume mixing ratio', phys_decomp, sampling_seq='rad_lwsw')
   128) 
   129) 
   130)    ! Initialize for one field (arg_4=1) and do not vertically interpolate (arg_6=3)
   131)    call boundarydata_init(bndtvo, phys_state, nc_name, 1, ozonedata, 3)
   132) 
   133)    if (masterproc) then
   134)       write(iulog,*)'cam3_ozone_data_init: Initializing CAM3 prescribed ozone'
   135)       write(iulog,*)'Time-variant boundary dataset (ozone) is: ', trim(bndtvo)
   136)       if (ozncyc) then
   137)          write(iulog,*)'OZONE dataset will be reused for each model year'
   138)       else
   139)          write(iulog,*)'OZONE dataset will not be cycled'
   140)       end if
   141)    end if
   142) 
   143) end subroutine cam3_ozone_data_init

   147) subroutine cam3_ozone_data_timestep_init(pbuf2d,  phys_state)
   148) !----------------------------------------------------------------------- 
   149) ! 
   150) ! Purpose: Interpolate ozone mixing ratios to current time, reading in new monthly
   151) !          data if necessary, and spatially interpolating it.
   152) ! 
   153) ! Method: Find next month of ozone data to interpolate.  Linearly interpolate 
   154) !         vertically and horizontally
   155) ! 
   156) !-----------------------------------------------------------------------
   157) 
   158)    
   159)    use physics_buffer, only : physics_buffer_desc, pbuf_get_field, pbuf_get_chunk
   160) 
   161)    
   162)    type(physics_state), intent(in) :: phys_state(begchunk:endchunk) 
   163)    type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   164)    real(r8),pointer :: tmpptr(:,:)
   165) 
   166)    integer lchnk
   167)     
   168)    call boundarydata_update(phys_state, ozonedata)
   169) 
   170)    do lchnk = begchunk, endchunk
   171)       call pbuf_get_field(pbuf_get_chunk(pbuf2d, lchnk), oz_idx, tmpptr)
   172)       call ozone_data_get_cnst(phys_state(lchnk), tmpptr)
   173)    enddo
   174) 
   175) end subroutine cam3_ozone_data_timestep_init

   179) subroutine ozone_data_get_cnst(state, q)
   180) 
   181)    use cam_history, only: outfld
   182)    use physconst,   only: mwo3
   183) 
   184)    type(physics_state),  intent(in) :: state
   185)    real(r8)                         :: q(:,:)     ! constituent mass mixing ratio
   186) 
   187)    ! local variables
   188)    integer :: lchnk            ! chunk identifier
   189)    integer :: i, k
   190)    real(r8) :: ozmixin(pcols,ozonedata%levsiz)
   191)    ! *** N.B. this hardwired mw of dry air needs to be changed to the share value
   192)    real(r8), parameter :: mwdry = 28.9644_r8  ! Effective molecular weight of dry air (g/mol)
   193)    real(r8), parameter :: mwr =  mwo3/mwdry   ! convert from the dataset values of vmr to mmr
   194)    !-------------------------------------------------------------------------------
   195) 
   196)    lchnk = state%lchnk
   197) 
   198)    ozmixin=0._r8
   199)    do k=1,ozonedata%levsiz
   200)       do i=1,state%ncol
   201)          ozmixin(i,k) = ozonedata%datainst(state%latmapback(i),k,lchnk,1)
   202)       end do
   203)    end do
   204)    call boundarydata_vert_interp(lchnk, state%ncol, ozonedata%levsiz, &
   205)                                  1, ozonedata%pin, state%pmid, ozmixin , q)
   206) 
   207)    call outfld('O3VMR', q, pcols, lchnk)
   208) 
   209)    do k=1,pver
   210)       do i=1,state%ncol
   211)          q(i,k) = mwr*q(i,k)
   212)       end do
   213)    end do
   214)     
   215) end subroutine ozone_data_get_cnst
