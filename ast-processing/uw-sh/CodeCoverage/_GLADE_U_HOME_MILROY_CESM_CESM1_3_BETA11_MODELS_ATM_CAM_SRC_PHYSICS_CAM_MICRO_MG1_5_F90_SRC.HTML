   256) subroutine micro_mg_init( &
   257)      kind, gravit, rair, rh2o, cpair,    &
   258)      tmelt_in, latvap, latice,           &
   259)      rhmini_in, microp_uniform_in, do_cldice_in, &
   260)      errstring)
   261)   
   262)   !----------------------------------------------------------------------- 
   263)   ! 
   264)   ! Purpose: 
   265)   ! initialize constants for MG microphysics
   266)   ! 
   267)   ! Author: Andrew Gettelman Dec 2005
   268)   ! 
   269)   !-----------------------------------------------------------------------
   270)   
   271)   integer,  intent(in)  :: kind         ! Kind used for reals
   272)   real(r8), intent(in)  :: gravit
   273)   real(r8), intent(in)  :: rair
   274)   real(r8), intent(in)  :: rh2o
   275)   real(r8), intent(in)  :: cpair
   276)   real(r8), intent(in)  :: tmelt_in     ! Freezing point of water (K)
   277)   real(r8), intent(in)  :: latvap
   278)   real(r8), intent(in)  :: latice
   279)   real(r8), intent(in)  :: rhmini_in    ! Minimum rh for ice cloud fraction > 0.
   280) 
   281)   logical,  intent(in)  :: microp_uniform_in    ! .true. = configure uniform for sub-columns 
   282)                                             ! .false. = use w/o sub-columns (standard)
   283)   logical,  intent(in)  :: do_cldice_in     ! .true. = do all processes (standard)
   284)                                             ! .false. = skip all processes affecting
   285)                                             !           cloud ice
   286) 
   287)   character(128), intent(out) :: errstring    ! Output status (non-blank for error return)
   288) 
   289)   !-----------------------------------------------------------------------
   290) 
   291)   errstring = ' '
   292) 
   293)   if( kind .ne. r8 ) then
   294)      errstring = 'micro_mg_init: KIND of reals does not match'
   295)      return
   296)   endif
   297) 
   298)   ! declarations for MG code (transforms variable names)
   299) 
   300)   g= gravit                 ! gravity
   301)   r= rair                   ! dry air gas constant: note units(phys_constants are in J/K/kmol)
   302)   rv= rh2o                  ! water vapor gas constant
   303)   cpp = cpair               ! specific heat of dry air
   304)   tmelt = tmelt_in
   305)   rhmini = rhmini_in
   306) 
   307)   ! latent heats
   308) 
   309)   xxlv = latvap         ! latent heat vaporization
   310)   xlf  = latice         ! latent heat freezing
   311)   xxls = xxlv + xlf     ! latent heat of sublimation
   312) 
   313)   ! flags
   314)   microp_uniform = microp_uniform_in
   315)   do_cldice  = do_cldice_in
   316) 
   317)   ! typical air density at 850 mb
   318) 
   319)   rhosu = 85000._r8/(rair * tmelt)
   320) 
   321)   ! Maximum temperature at which snow is allowed to exist
   322)   snowmelt = tmelt + 2._r8
   323)   ! Minimum temperature at which rain is allowed to exist
   324)   rainfrze = tmelt - 5._r8
   325) 
   326)   ! Ice nucleation temperature
   327)   icenuct  = tmelt - 5._r8
   328) 
   329)   ! Define constants to help speed up code (this limits calls to gamma function)
   330)   ! Unused names: cons6, cons15, cons21, cons26
   331)   cons1=gamma(1._r8+dsph)
   332)   cons4=gamma(1._r8+br)
   333)   cons5=gamma(4._r8+br)
   334)   cons7=gamma(1._r8+bs)     
   335)   cons8=gamma(4._r8+bs)     
   336)   cons11=gamma(3._r8+bs)
   337)   cons13=gamma(5._r8/2._r8+br/2._r8)
   338)   cons14=gamma(5._r8/2._r8+bs/2._r8)
   339)   cons16=gamma(1._r8+bi)
   340)   cons17=gamma(4._r8+bi)
   341)   cons22=(4._r8/3._r8*pi*rhow*(25.e-6_r8)**3)
   342)   cons23=dcs**3
   343)   cons24=dcs**2
   344)   cons25=dcs**bs
   345)   cons27=xxlv**2
   346)   cons28=xxls**2
   347)

  2917) ! get cloud droplet size distribution parameters
  2918) elemental subroutine size_dist_param_liq(qcic, ncic, cdnl, rho, nadjflag, pgam, lamc)
  2919) 
  2920)   real(r8), intent(in) :: qcic
  2921)   real(r8), intent(inout) :: ncic
  2922)   real(r8), intent(in) :: cdnl
  2923)   real(r8), intent(in) :: rho
  2924)   logical,  intent(in) :: nadjflag ! Whether to adjust number concentration to fall
  2925)                                    ! within certain bounds
  2926) 
  2927)   real(r8), intent(out) :: pgam
  2928)   real(r8), intent(out) :: lamc
  2929) 
  2930)   real(r8) :: dumgam1
  2931)   real(r8) :: dumgam2
  2932)   real(r8) :: lammin
  2933)   real(r8) :: lammax
  2934) 
  2935)   if (qcic > qsmall) then
  2936) 
  2937)      if (nadjflag) then
  2938)         ! add upper limit to in-cloud number concentration to prevent numerical error
  2939)         ncic=min(ncic,qcic*1.e20_r8)
  2940)         ! add lower limit to in-cloud number concentration
  2941)         ncic=max(ncic,cdnl/rho) ! sghan minimum in #/cm
  2942)      end if
  2943) 
  2944)      ! get pgam from fit to observations of martin et al. 1994
  2945) 
  2946)      pgam=0.0005714_r8*(ncic/1.e6_r8*rho)+0.2714_r8
  2947)      pgam=1._r8/(pgam**2)-1._r8
  2948)      pgam=max(pgam,2._r8)
  2949)      pgam=min(pgam,15._r8)
  2950) 
  2951)      ! calculate lamc
  2952)      dumgam1 = gamma(pgam+1._r8)
  2953)      dumgam2 = gamma(pgam+4._r8)
  2954) 
  2955)      lamc = (pi/6._r8*rhow*ncic*dumgam2/ &
  2956)           (qcic*dumgam1))**(1._r8/3._r8)
  2957) 
  2958)      ! lammin, 50 micron diameter max mean size
  2959)      ! omsm fudge factors are to guaranteed that lamcrad falls within the
  2960)      ! table used by RRTMG.
  2961)      lammin = (pgam+1._r8)/(omsm*50.e-6_r8)
  2962)      lammax = (pgam+1._r8)*omsm/2.e-6_r8
  2963) 
  2964)      if (lamc < lammin) then
  2965)         lamc = lammin
  2966) 
  2967)         if (nadjflag) then
  2968)            ncic = 6._r8 * lamc**3 * qcic * dumgam1/ &
  2969)                 (pi * rhow * dumgam2)
  2970)         end if
  2971)      else if (lamc > lammax) then
  2972)         lamc = lammax
  2973) 
  2974)         if (nadjflag) then
  2975)            ncic = 6._r8 * lamc**3 * qcic * dumgam1/ &
  2976)                 (pi * rhow * dumgam2)
  2977)         end if
  2978)      end if
  2979) 
  2980)   else
  2981)      ! pgam not calculated in this case, so set it to a value likely to cause an error
  2982)      ! if it is accidentally used
  2983)      ! (gamma function undefined for negative integers)
  2984)      pgam = -100._r8
  2985)      lamc = 0._r8
  2986)   end if
  2987) 
  2988) end subroutine size_dist_param_liq
