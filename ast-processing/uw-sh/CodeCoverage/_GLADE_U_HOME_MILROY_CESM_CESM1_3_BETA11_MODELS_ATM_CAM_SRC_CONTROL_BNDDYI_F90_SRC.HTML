     2) subroutine bnddyi (ncdate, ncsec, doy)
     3) !----------------------------------------------------------------------- 
     4) ! 
     5) ! Purpose: Convert date and seconds of day to floating point calendar day, for
     6) !          boundary dataset handling
     7) ! 
     8) ! Method: Use table of days per month to do conversion
     9) ! 
    10) ! Author: CCM Core Group
    11) ! 
    12) !-----------------------------------------------------------------------
    13)    use shr_kind_mod, only: r8 => shr_kind_r8
    14)    use abortutils,   only: endrun
    15)    use cam_logfile,  only: iulog
    16) !-----------------------------------------------------------------------
    17)    implicit none
    18) !--------------------------Arguments------------------------------------
    19) !
    20) ! Arguments
    21) !
    22)    integer, intent(in) :: ncdate      ! Current date as yymmdd or yyyymmdd
    23)    integer, intent(in) :: ncsec       ! Seconds of day for current date
    24) 
    25)    real(r8), intent(out) :: doy       ! Day of year
    26) !
    27) ! Local Variables
    28) !
    29)    integer mnth        ! Month number
    30)    integer mday        ! Day number of month
    31)    integer jdcon(12)   ! Starting day number for each month
    32)    save jdcon
    33)    data jdcon/0,31,59,90,120,151,181,212,243,273,304,334/
    34) !
    35) ! Decode month and day
    36) !
    37)    mnth = mod(ncdate,10000)/100
    38)    if (mnth < 1 .or. mnth > 12) then
    39)       write(iulog,*)'BNDDYI: Bad month index=', mnth
    40)       call endrun
    41)    end if
    42)    mday = mod(ncdate,100)
    43)    doy = jdcon(mnth) + mday + ncsec/86400._r8
    44) 
    45)    if (doy < 1._r8 .or. doy > 366._r8) then
    46)       write(iulog,*)'BNDDYI: bad day of year = ',doy
    47)       call endrun
    48)    end if
    49) !
    50)    return
    51) end subroutine bnddyi
