   429) 
   430)          endif
   431)       enddo  ! num_avail_hist_fields
   432)

   436)       ! ... the user should change these to something useful ...
   437)       !-----------------------------------------------------------------
   438) #ifdef CCSMCOUPLED
   439)       status = pio_put_att(File,pio_global,'title',runid)

   441)       title  = 'sea ice model output for CICE'
   442)       status = pio_put_att(File,pio_global,'title',title)
   443) #endif
   444)       title = 'Diagnostic and Prognostic Variables'
   445)       status = pio_put_att(File,pio_global,'contents',title)
   446) 
   447)       title  = 'sea ice model: Community Ice Code (CICE)'
   448)       status = pio_put_att(File,pio_global,'source',title)

   471)

   480)       !-----------------------------------------------------------------

   483)       status = pio_put_var(File,varid,ltime)
   484)

   494)          status = pio_put_var(File,varid,start=bnd_start(:),count=bnd_length(:),ival=time_bounds)
