    32) ! !INTERFACE:
    33) subroutine CNVegStructUpdate(num_soilp, filter_soilp)
    34) !
    35) ! !DESCRIPTION:
    36) ! On the radiation time step, use C state variables and epc to diagnose
    37) ! vegetation structure (LAI, SAI, height)
    38) !
    39) ! !USES:
    40)    use clmtype
    41)    use clm_atmlnd   , only: clm_a2l
    42)    use pftvarcon    , only: noveg, nc3crop, nirrig, nbrdlf_evr_shrub, nbrdlf_dcd_brl_shrub
    43)    use pftvarcon    , only: ncorn, npcropmin, ztopmx, laimx
    44)    use clm_varctl   , only: iulog, use_cndv
    45)    use shr_sys_mod  , only: shr_sys_flush
    46)    use shr_const_mod, only: SHR_CONST_PI
    47)    use clm_time_manager , only : get_rad_step_size
    48) !
    49) ! !ARGUMENTS:
    50)    implicit none
    51)    integer, intent(in) :: num_soilp                 ! number of column soil points in pft filter
    52)    integer, intent(in) :: filter_soilp(:)   ! pft filter for soil points
    53) !
    54) ! !CALLED FROM:
    55) ! subroutine CNEcosystemDyn
    56) !
    57) ! !REVISION HISTORY:
    58) ! 10/28/03: Created by Peter Thornton
    59) ! 2/29/08, David Lawrence: revised snow burial fraction for short vegetation
    60) !
    61) ! !LOCAL VARIABLES:
    62) ! local pointers to implicit in scalars
    63) !
    64)    real(r8), pointer :: allom2(:)     ! ecophys const
    65)    real(r8), pointer :: allom3(:)     ! ecophys const
    66)    real(r8), pointer :: nind(:)       ! number of individuals (#/m**2)
    67)    real(r8), pointer :: fpcgrid(:)    ! fractional area of pft (pft area/nat veg area)
    68)    integer , pointer :: ivt(:)        ! pft vegetation type
    69)    integer , pointer :: pcolumn(:)    ! column index associated with each pft
    70)    integer , pointer :: pgridcell(:)  ! pft's gridcell index
    71)    real(r8), pointer :: snowdp(:)     ! snow height (m)
    72)    real(r8), pointer :: leafc(:)      ! (gC/m2) leaf C
    73)    real(r8), pointer :: deadstemc(:)  ! (gC/m2) dead stem C
    74)    real(r8), pointer :: woody(:)      !binary flag for woody lifeform (1=woody, 0=not woody)
    75)    real(r8), pointer :: slatop(:)     !specific leaf area at top of canopy, projected area basis [m^2/gC]
    76)    real(r8), pointer :: dsladlai(:)   !dSLA/dLAI, projected area basis [m^2/gC]
    77)    real(r8), pointer :: z0mr(:)       !ratio of momentum roughness length to canopy top height (-)
    78)    real(r8), pointer :: displar(:)    !ratio of displacement height to canopy top height (-)
    79)    real(r8), pointer :: forc_hgt_u_pft(:) ! observational height of wind at pft-level [m]
    80)    real(r8), pointer :: dwood(:)      ! density of wood (gC/m^3)
    81) !
    82) ! local pointers to implicit in/out scalars
    83) !
    84)    integer , pointer :: frac_veg_nosno_alb(:) ! frac of vegetation not covered by snow [-]
    85)    real(r8), pointer :: tlai(:) !one-sided leaf area index, no burying by snow
    86)    real(r8), pointer :: tsai(:) !one-sided stem area index, no burying by snow
    87)    real(r8), pointer :: htop(:) !canopy top (m)
    88)    real(r8), pointer :: hbot(:) !canopy bottom (m)
    89)    real(r8), pointer :: elai(:)     ! one-sided leaf area index with burying by snow
    90)    real(r8), pointer :: esai(:)     ! one-sided stem area index with burying by snow
    91)    real(r8), pointer :: htmx(:)     ! max hgt attained by a crop during yr (m)
    92)    integer , pointer :: peaklai(:)  ! 1: max allowed lai; 0: not at max
    93)    integer , pointer :: harvdate(:) ! harvest date
    94) !
    95) ! local pointers to implicit out scalars
    96) !
    97) !
    98) ! !OTHER LOCAL VARIABLES:
    99)    integer :: p,c,g        !indices
   100)    integer :: fp           !lake filter indices
   101)    real(r8):: taper        ! ratio of height:radius_breast_height (tree allometry)
   102)    real(r8):: stocking     ! #stems / ha (stocking density)
   103)    real(r8):: ol           ! thickness of canopy layer covered by snow (m)
   104)    real(r8):: fb           ! fraction of canopy layer covered by snow
   105)    real(r8) :: tlai_old    ! for use in Zeng tsai formula
   106)    real(r8) :: tsai_old    ! for use in Zeng tsai formula
   107)    real(r8) :: tsai_min    ! PFT derived minimum tsai
   108)    real(r8) :: tsai_alpha  ! monthly decay rate of tsai
   109)    real(r8) dt             ! radiation time step (sec)
   110) 
   111)    real(r8), parameter :: dtsmonth = 2592000._r8 ! number of seconds in a 30 day month (60x60x24x30)
   112) !EOP
   113) !-----------------------------------------------------------------------
   114) ! tsai formula from Zeng et. al. 2002, Journal of Climate, p1835
   115) !
   116) ! tsai(p) = max( tsai_alpha(ivt(p))*tsai_old + max(tlai_old-tlai(p),0_r8), tsai_min(ivt(p)) )
   117) ! notes:
   118) ! * RHS tsai & tlai are from previous timestep
   119) ! * should create tsai_alpha(ivt(p)) & tsai_min(ivt(p)) in pftvarcon.F90 - slevis
   120) ! * all non-crop pfts use same values:
   121) !   crop    tsai_alpha,tsai_min = 0.0,0.1
   122) !   noncrop tsai_alpha,tsai_min = 0.5,1.0  (includes bare soil and urban)
   123) !-------------------------------------------------------------------------------
   124) 
   125)    ! assign local pointers to derived type arrays (in)
   126)     allom2                         => dgv_pftcon%allom2
   127)     allom3                         => dgv_pftcon%allom3
   128)     nind                           => pdgvs%nind
   129)     fpcgrid                        => pdgvs%fpcgrid
   130)     ivt                            => pft%itype
   131)     pcolumn                        => pft%column
   132)     pgridcell                      => pft%gridcell
   133)     leafc                          => pcs%leafc
   134)     deadstemc                      => pcs%deadstemc
   135)     snowdp                         => cps%snowdp
   136)     woody                          => pftcon%woody
   137)     slatop                         => pftcon%slatop
   138)     dsladlai                       => pftcon%dsladlai
   139)     z0mr                           => pftcon%z0mr
   140)     displar                        => pftcon%displar
   141)     dwood                          => pftcon%dwood
   142) 
   143)    ! assign local pointers to derived type arrays (out)
   144)     tlai                           => pps%tlai
   145)     tsai                           => pps%tsai
   146)     htop                           => pps%htop
   147)     hbot                           => pps%hbot
   148)     elai                           => pps%elai
   149)     esai                           => pps%esai
   150)     frac_veg_nosno_alb             => pps%frac_veg_nosno_alb
   151)     htmx                           => pps%htmx
   152)     peaklai                        => pps%peaklai
   153)     harvdate                       => pps%harvdate
   154)     forc_hgt_u_pft                 => pps%forc_hgt_u_pft
   155) 
   156)    dt = real( get_rad_step_size(), r8 )
   157) 
   158)    ! constant allometric parameters
   159)    taper = 200._r8
   160)    stocking = 1000._r8
   161) 
   162)    ! convert from stems/ha -> stems/m^2
   163)    stocking = stocking / 10000._r8
   164) 
   165)    ! pft loop
   166)    do fp = 1,num_soilp
   167)       p = filter_soilp(fp)
   168)       c = pcolumn(p)
   169)       g = pgridcell(p)
   170) 
   171)       if (ivt(p) /= noveg) then
   172) 
   173)           tlai_old = tlai(p) ! n-1 value
   174)           tsai_old = tsai(p) ! n-1 value
   175) 
   176)           ! update the leaf area index based on leafC and SLA
   177)           ! Eq 3 from Thornton and Zimmerman, 2007, J Clim, 20, 3902-3923. 
   178)           if (dsladlai(ivt(p)) > 0._r8) then
   179)              tlai(p) = (slatop(ivt(p))*(exp(leafc(p)*dsladlai(ivt(p))) - 1._r8))/dsladlai(ivt(p))
   180)           else
   181)              tlai(p) = slatop(ivt(p)) * leafc(p)
   182)           end if
   183)           tlai(p) = max(0._r8, tlai(p))
   184) 
   185)           ! update the stem area index and height based on LAI, stem mass, and veg type.
   186)           ! With the exception of htop for woody vegetation, this follows the DGVM logic.
   187) 
   188)           ! tsai formula from Zeng et. al. 2002, Journal of Climate, p1835 (see notes)
   189)           ! Assumes doalb time step .eq. CLM time step, SAI min and monthly decay factor
   190)           ! alpha are set by PFT, and alpha is scaled to CLM time step by multiplying by
   191)           ! dt and dividing by dtsmonth (seconds in average 30 day month)
   192)           ! tsai_min scaled by 0.5 to match MODIS satellite derived values
   193)           if (ivt(p) == nc3crop .or. ivt(p) == nirrig) then ! generic crops
   194) 
   195)              tsai_alpha = 1.0_r8-1.0_r8*dt/dtsmonth
   196)              tsai_min = 0.1_r8
   197)           else
   198)              tsai_alpha = 1.0_r8-0.5_r8*dt/dtsmonth
   199)              tsai_min = 1.0_r8
   200)           end if
   201)           tsai_min = tsai_min * 0.5_r8
   202)           tsai(p) = max(tsai_alpha*tsai_old+max(tlai_old-tlai(p),0._r8),tsai_min)
   203) 
   204)           if (woody(ivt(p)) == 1._r8) then
   205) 
   206)              ! trees and shrubs
   207) 
   208)              ! if shrubs have a squat taper 
   209)              if (ivt(p) >= nbrdlf_evr_shrub .and. ivt(p) <= nbrdlf_dcd_brl_shrub) then
   210)                 taper = 10._r8
   211)              ! otherwise have a tall taper
   212)              else
   213)                 taper = 200._r8
   214)              end if
   215) 
   216)              ! trees and shrubs for now have a very simple allometry, with hard-wired
   217)              ! stem taper (height:radius) and hard-wired stocking density (#individuals/area)
   218)              if (use_cndv) then
   219)                 if (fpcgrid(p) > 0._r8 .and. nind(p) > 0._r8) then
   220)                    stocking = nind(p)/fpcgrid(p) !#ind/m2 nat veg area -> #ind/m2 pft area
   221)                    ! lpj's htop w/ cn's stemdiam
   222)                    htop(p) = allom2(ivt(p)) * ( (24._r8 * deadstemc(p) / &
   223)                         (SHR_CONST_PI * stocking * dwood(ivt(p)) * taper))**(1._r8/3._r8) )**allom3(ivt(p)) 
   224)                 else
   225)                    htop(p) = 0._r8
   226)                 end if
   227)              else
   228)                 htop(p) = ((3._r8 * deadstemc(p) * taper * taper)/ &
   229)                      (SHR_CONST_PI * stocking * dwood(ivt(p))))**(1._r8/3._r8)
   230)              end if
   231) 
   232)              ! Peter Thornton, 5/3/2004
   233)              ! Adding test to keep htop from getting too close to forcing height for windspeed
   234)              ! Also added for grass, below, although it is not likely to ever be an issue.
   235)              htop(p) = min(htop(p),(forc_hgt_u_pft(p)/(displar(ivt(p))+z0mr(ivt(p))))-3._r8)
   236) 
   237)              ! Peter Thornton, 8/11/2004
   238)              ! Adding constraint to keep htop from going to 0.0.
   239)              ! This becomes an issue when fire mortality is pushing deadstemc
   240)              ! to 0.0.
   241)              htop(p) = max(htop(p), 0.01_r8)
   242) 
   243)              hbot(p) = max(0._r8, min(3._r8, htop(p)-1._r8))
   244) 
   245)           else if (ivt(p) >= npcropmin) then ! prognostic crops
   246) 
   247)              if (tlai(p) >= laimx(ivt(p))) peaklai(p) = 1 ! used in CNAllocation
   248) 
   249)              if (ivt(p) == ncorn) then
   250)                 tsai(p) = 0.1_r8 * tlai(p)
   251)              else
   252)                 tsai(p) = 0.2_r8 * tlai(p)
   253)              end if
   254) 
   255)              ! "stubble" after harvest
   256)              if (harvdate(p) < 999 .and. tlai(p) == 0._r8) then
   257)                 tsai(p) = 0.25_r8
   258)                 htmx(p) = 0._r8
   259)                 peaklai(p) = 0
   260)              end if
   261)              !if (harvdate(p) < 999 .and. tlai(p) > 0._r8) write(iulog,*) 'CNVegStructUpdate: tlai>0 after harvest!' ! remove after initial debugging?
   262) 
   263)              ! canopy top and bottom heights
   264)              htop(p) = ztopmx(ivt(p)) * (min(tlai(p)/(laimx(ivt(p))-1._r8),1._r8))**2
   265)              htmx(p) = max(htmx(p), htop(p))
   266)              htop(p) = max(0.05_r8, max(htmx(p),htop(p)))
   267)              hbot(p) = 0.02_r8
   268)           else ! generic crops and ...
   269)              ! grasses
   270) 
   271)              ! height for grasses depends only on LAI
   272)              htop(p) = max(0.25_r8, tlai(p) * 0.25_r8)
   273) 
   274)              htop(p) = min(htop(p),(forc_hgt_u_pft(p)/(displar(ivt(p))+z0mr(ivt(p))))-3._r8)
   275) 
   276)              ! Peter Thornton, 8/11/2004
   277)              ! Adding constraint to keep htop from going to 0.0.
   278)              htop(p) = max(htop(p), 0.01_r8)
   279) 
   280)              hbot(p) = max(0.0_r8, min(0.05_r8, htop(p)-0.20_r8))
   281)           end if
   282) 
   283)       else
   284)           tlai(p) = 0._r8
   285)           tsai(p) = 0._r8
   286)           htop(p) = 0._r8
   287)           hbot(p) = 0._r8
   288)       end if
   289)       
   290)       ! adjust lai and sai for burying by snow. 
   291) 
   292)       ! snow burial fraction for short vegetation (e.g. grasses) as in
   293)       ! Wang and Zeng, 2007.
   294)       if (ivt(p) > noveg .and. ivt(p) <= nbrdlf_dcd_brl_shrub ) then
   295)          ol = min( max(snowdp(c)-hbot(p), 0._r8), htop(p)-hbot(p))
   296)          fb = 1._r8 - ol / max(1.e-06_r8, htop(p)-hbot(p))
   297)       else
   298)          fb = 1._r8 - max(min(snowdp(c),0.2_r8),0._r8)/0.2_r8   ! 0.2m is assumed
   299)               !depth of snow required for complete burial of grasses
   300)       endif
   301) 
   302)       elai(p) = max(tlai(p)*fb, 0.0_r8)
   303)       esai(p) = max(tsai(p)*fb, 0.0_r8)
   304) 
   305)       ! Fraction of vegetation free of snow
   306)       if ((elai(p) + esai(p)) > 0._r8) then
   307)          frac_veg_nosno_alb(p) = 1
   308)       else
   309)          frac_veg_nosno_alb(p) = 0
   310)       end if
   311) 
   312)    end do
   313) 
   314) end subroutine CNVegStructUpdate
