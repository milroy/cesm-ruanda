   922) subroutine modal_aero_lw(list_idx, state, pbuf, tauxar)
   923) 
   924)    ! calculates aerosol lw radiative properties
   925) 
   926)    integer,             intent(in)  :: list_idx ! index of the climate or a diagnostic list
   927)    type(physics_state), intent(in), target :: state    ! state variables
   928)    
   929)    type(physics_buffer_desc), pointer :: pbuf(:)
   930) 
   931)    real(r8), intent(out) :: tauxar(pcols,pver,nlwbands) ! layer absorption optical depth
   932) 
   933)    ! Local variables
   934)    integer :: i, ifld, ilw, k, l, m, nc, ns
   935)    integer :: lchnk                    ! chunk id
   936)    integer :: ncol                     ! number of active columns in the chunk
   937)    integer :: nmodes
   938)    integer :: nspec
   939) 
   940)    real(r8), pointer :: dgnumwet(:,:)  ! wet number mode diameter (m)
   941)    real(r8), pointer :: qaerwat(:,:)   ! aerosol water (g/g)
   942) 
   943)    real(r8), pointer :: dgnumdry_m(:,:,:) ! number mode dry diameter for all modes
   944)    real(r8), pointer :: dgnumwet_m(:,:,:) ! number mode wet diameter for all modes
   945)    real(r8), pointer :: qaerwat_m(:,:,:)  ! aerosol water (g/g) for all modes
   946)    real(r8), pointer :: wetdens_m(:,:,:)  ! 
   947) 
   948)    real(r8) :: sigma_logr_aer          ! geometric standard deviation of number distribution
   949)    real(r8) :: alnsg_amode
   950)    real(r8) :: xrad(pcols)
   951)    real(r8) :: cheby(ncoef,pcols,pver)  ! chebychef polynomials
   952) 
   953)    real(r8) :: mass(pcols,pver) ! layer mass
   954) 
   955)    real(r8),    pointer :: specmmr(:,:)        ! species mass mixing ratio
   956)    real(r8)             :: specdens            ! species density (kg/m3)
   957)    complex(r8), pointer :: specrefindex(:)     ! species refractive index
   958) 
   959)    real(r8) :: vol(pcols)       ! volume concentration of aerosol specie (m3/kg)
   960)    real(r8) :: dryvol(pcols)    ! volume concentration of aerosol mode (m3/kg)
   961)    real(r8) :: wetvol(pcols)    ! volume concentration of wet mode (m3/kg)
   962)    real(r8) :: watervol(pcols)  ! volume concentration of water in each mode (m3/kg)
   963)    real(r8) :: refr(pcols)      ! real part of refractive index
   964)    real(r8) :: refi(pcols)      ! imaginary part of refractive index
   965)    complex(r8) :: crefin(pcols) ! complex refractive index
   966)    real(r8), pointer :: refrtablw(:,:) ! table of real refractive indices for aerosols
   967)    real(r8), pointer :: refitablw(:,:) ! table of imag refractive indices for aerosols
   968)    real(r8), pointer :: absplw(:,:,:,:) ! specific absorption
   969) 
   970)    integer  :: itab(pcols), jtab(pcols)
   971)    real(r8) :: ttab(pcols), utab(pcols)
   972)    real(r8) :: cabs(pcols,ncoef)
   973)    real(r8) :: pabs(pcols)      ! parameterized specific absorption (m2/kg)
   974)    real(r8) :: dopaer(pcols)    ! aerosol optical depth in layer
   975) 
   976)    integer, parameter :: nerrmax_dopaer=1000
   977)    integer  :: nerr_dopaer = 0
   978)    real(r8) :: volf             ! volume fraction of insoluble aerosol
   979) 
   980)    character(len=*), parameter :: subname = 'modal_aero_lw'
   981)    !----------------------------------------------------------------------------
   982) 
   983)    lchnk = state%lchnk
   984)    ncol  = state%ncol
   985) 
   986)    ! initialize output variables
   987)    tauxar(:ncol,:,:) = 0._r8
   988) 
   989)    ! dry mass in each cell
   990)    mass(:ncol,:) = state%pdeldry(:ncol,:)*rga
   991) 
   992)    ! loop over all aerosol modes
   993)    call rad_cnst_get_info(list_idx, nmodes=nmodes)
   994) 
   995)    if (list_idx == 0) then
   996)       ! water uptake and wet radius for the climate list has already been calculated
   997)       call pbuf_get_field(pbuf, dgnumwet_idx, dgnumwet_m)
   998)       call pbuf_get_field(pbuf, qaerwat_idx,  qaerwat_m)
   999)    else
  1000)       ! If doing a diagnostic calculation then need to calculate the wet radius
  1001)       ! and water uptake for the diagnostic modes
  1002)       call modal_aero_calcsize_diag(state, pbuf, list_idx, dgnumdry_m)  
  1003)       call modal_aero_wateruptake_dr(state, pbuf, list_idx, dgnumdry_m, dgnumwet_m, &
  1004)                                      qaerwat_m, wetdens_m)
  1005)    endif
  1006) 
  1007)    do m = 1, nmodes
  1008) 
  1009)       dgnumwet => dgnumwet_m(:,:,m)
  1010)       qaerwat  => qaerwat_m(:,:,m)
  1011) 
  1012)       ! get mode properties
  1013)       call rad_cnst_get_mode_props(list_idx, m, sigmag=sigma_logr_aer, refrtablw=refrtablw , &
  1014)          refitablw=refitablw, absplw=absplw)
  1015) 
  1016)       ! get mode info
  1017)       call rad_cnst_get_info(list_idx, m, nspec=nspec)
  1018) 
  1019)       ! calc size parameter for all columns
  1020)       ! this is the same calculation that's done in modal_size_parameters, but there
  1021)       ! some intermediate results are saved and the chebyshev polynomials are stored
  1022)       ! in a array with different index order.  Could be unified.
  1023)       do k = top_lev, pver
  1024)          do i = 1, ncol
  1025)             alnsg_amode = log( sigma_logr_aer )
  1026)             ! convert from number diameter to surface area
  1027)             xrad(i) = log(0.5_r8*dgnumwet(i,k)) + 2.0_r8*alnsg_amode*alnsg_amode
  1028)             ! normalize size parameter
  1029)             xrad(i) = max(xrad(i), xrmin)
  1030)             xrad(i) = min(xrad(i), xrmax)
  1031)             xrad(i) = (2*xrad(i)-xrmax-xrmin)/(xrmax-xrmin)
  1032)             ! chebyshev polynomials
  1033)             cheby(1,i,k) = 1.0_r8
  1034)             cheby(2,i,k) = xrad(i)
  1035)             do nc = 3, ncoef
  1036)                cheby(nc,i,k) = 2.0_r8*xrad(i)*cheby(nc-1,i,k)-cheby(nc-2,i,k)
  1037)             end do
  1038)          end do
  1039)       end do
  1040) 
  1041)       do ilw = 1, nlwbands
  1042) 
  1043)          do k = top_lev, pver
  1044) 
  1045)             ! form bulk refractive index. Use volume mixing for infrared
  1046)             crefin(:ncol) = (0._r8, 0._r8)
  1047)             dryvol(:ncol) = 0._r8
  1048) 
  1049)             ! aerosol species loop
  1050)             do l = 1, nspec
  1051)                call rad_cnst_get_aer_mmr(list_idx, m, l, 'a', state, pbuf, specmmr)
  1052)                call rad_cnst_get_aer_props(list_idx, m, l, density_aer=specdens, &
  1053)                                            refindex_aer_lw=specrefindex)
  1054) 
  1055)                do i = 1, ncol
  1056)                   vol(i)    = specmmr(i,k)/specdens
  1057)                   dryvol(i) = dryvol(i) + vol(i)
  1058)                   crefin(i) = crefin(i) + vol(i)*specrefindex(ilw)
  1059)                end do
  1060)             end do
  1061) 
  1062)             do i = 1, ncol
  1063)                watervol(i) = qaerwat(i,k)/rhoh2o
  1064)                wetvol(i)   = watervol(i) + dryvol(i)
  1065)                if (watervol(i) < 0.0_r8) then
  1066)                   if (abs(watervol(i)) .gt. 1.e-1_r8*wetvol(i)) then
  1067)                      write(iulog,*) 'watervol,wetvol,dryvol=',watervol(i),wetvol(i),dryvol(i),' in '//subname
  1068)                   end if
  1069)                   watervol(i) = 0._r8
  1070)                   wetvol(i)   = dryvol(i)
  1071)                end if
  1072) 
  1073)                crefin(i) = crefin(i) + watervol(i)*crefwlw(ilw)
  1074)                if (wetvol(i) > 1.e-40_r8) crefin(i) = crefin(i)/wetvol(i)
  1075)                refr(i) = real(crefin(i))
  1076)                refi(i) = aimag(crefin(i))
  1077)             end do
  1078) 
  1079)             ! interpolate coefficients linear in refractive index
  1080)             ! first call calcs itab,jtab,ttab,utab
  1081)             itab(:ncol) = 0
  1082)             call binterp(absplw(:,:,:,ilw), ncol, ncoef, prefr, prefi, &
  1083)                          refr, refi, refrtablw(:,ilw), refitablw(:,ilw), &
  1084)                          itab, jtab, ttab, utab, cabs)
  1085) 
  1086)             ! parameterized optical properties
  1087)             do i = 1, ncol
  1088)                pabs(i) = 0.5_r8*cabs(i,1)
  1089)                do nc = 2, ncoef
  1090)                   pabs(i) = pabs(i) + cheby(nc,i,k)*cabs(i,nc)
  1091)                end do
  1092)                pabs(i)   = pabs(i)*wetvol(i)*rhoh2o
  1093)                pabs(i)   = max(0._r8,pabs(i))
  1094)                dopaer(i) = pabs(i)*mass(i,k)
  1095)             end do
  1096) 
  1097)             do i = 1, ncol
  1098) 
  1099)                if ((dopaer(i) <= -1.e-10_r8) .or. (dopaer(i) >= 20._r8)) then
  1100) 
  1101)                   if (dopaer(i) <= -1.e-10_r8) then
  1102)                      write(iulog,*) "ERROR: Negative aerosol optical depth &
  1103)                           &in this layer."
  1104)                   else
  1105)                      write(iulog,*) "WARNING: Aerosol optical depth is &
  1106)                           &unreasonably high in this layer."
  1107)                   end if
  1108) 
  1109)                   write(iulog,*) 'dopaer(',i,',',k,',',m,',',lchnk,')=', dopaer(i)
  1110)                   write(iulog,*) 'k=',k,' pabs=', pabs(i)
  1111)                   write(iulog,*) 'wetvol=',wetvol(i),' dryvol=',dryvol(i),     &
  1112)                      ' watervol=',watervol(i)
  1113)                   write(iulog,*) 'cabs=', (cabs(i,l),l=1,ncoef)
  1114)                   write(iulog,*) 'crefin=', crefin(i)
  1115)                   write(iulog,*) 'nspec=', nspec
  1116)                   do l = 1,nspec
  1117)                      call rad_cnst_get_aer_mmr(list_idx, m, l, 'a', state, pbuf, specmmr)
  1118)                      call rad_cnst_get_aer_props(list_idx, m, l, density_aer=specdens, &
  1119)                                                  refindex_aer_lw=specrefindex)
  1120)                      volf = specmmr(i,k)/specdens
  1121)                      write(iulog,*) 'l=',l,'vol(l)=',volf
  1122)                      write(iulog,*) 'ilw=',ilw,' specrefindex(ilw)=',specrefindex(ilw)
  1123)                      write(iulog,*) 'specdens=',specdens
  1124)                   end do
  1125) 
  1126)                   nerr_dopaer = nerr_dopaer + 1
  1127)                   if (nerr_dopaer >= nerrmax_dopaer .or. dopaer(i) < -1.e-10_r8) then
  1128)                      write(iulog,*) '*** halting in '//subname//' after nerr_dopaer =', nerr_dopaer
  1129)                      call endrun()
  1130)                   end if
  1131) 
  1132)                end if
  1133)             end do
  1134) 
  1135)             do i = 1, ncol
  1136)                tauxar(i,k,ilw) = tauxar(i,k,ilw) + dopaer(i)
  1137)             end do
  1138) 
  1139)          end do ! k = top_lev, pver
  1140) 
  1141)       end do  ! nlwbands
  1142) 
  1143)    end do ! m = 1, nmodes
  1144) 
  1145)    if (list_idx > 0) then
  1146)       deallocate(dgnumdry_m)
  1147)       deallocate(dgnumwet_m)
  1148)       deallocate(qaerwat_m)
  1149)       deallocate(wetdens_m)
  1150)    end if
  1151) 
  1152) end subroutine modal_aero_lw

  1222) 
  1223)    integer,  intent(in)  :: ncol

  1225)    real(r8), intent(in)  :: dgnumwet(:,:)   ! aerosol wet number mode diameter (m)
