    16) ! DESCRIPTION
    17) !   This routine vertically redistributes condensed phase HNO3.
    18) !
    19) !   For each aerosol type the terminal velocity is calculated. This
    20) !   quantity is dependent on: 1) the mass density of the aerosol;
    21) !   2) the radius; 3) the dynamic viscosity; 4) shape; and the Cunningham
    22) !   correction factor for spherical particles. See Fuchs, The Mechanics
    23) !   of Aerosols Oxford, Pergmann Press, pp 27-31, 1964 and Kasten,
    24) !   Falling Speed of Aerosol Particles, J. of Appl. Met., 7, 944-947,
    25) !   1968 for details. For aerosol with a radius of 3 microns (e.g., NAT)
    26) !   and 10 microns (e.g., ICE) the terminal velocity (cm sec-1) is  0.2
    27) !   and 1.7 respectively.
    28) !
    29) !   The flux of condensed phase HNO3 is then derived using the
    30) !   following equation:
    31) !          Flux (molec cm-2 s-1) = V * C * exp (8*ln^2 sigma).
    32) !
    33) !          where: V is terminal velocity (cm sec-1)
    34) !                 C is condensed phase conc (molecules cm-3)
    35) !                 sigma is the width of the log normal distribution.
    36) !
    37) !   The approach of settling the entire aerosol size distribution is
    38) !   based on work of Considine et al., JGR, 1999.
    39) !
    40) !   The routine is a straighforward approach, starting at the top zone
    41) !   (highest altitude) and progressing towards the surface. The gross
    42) !   settling of condensed phase HNO3 is simply the quanity:
    43) !   Flux * dt/dz.
    44) !
    45) !
    46) ! ARGUMENTS
    47) !
    48) !      All of the following components are grid center quanitities.
    49) !
    50) !   INPUT:
    51) !      ad            Air Density (molecules cm-3)
    52) !      press         Pressure (Pascals)
    53) !      timestep      Gross chemistry timestep (in seconds)
    54) !      temp          Temperature (K)
    55) !      hno3_cond     Condensed phase HNO3 (mole fraction)
    56) !      radius_nat    Mean radius of NAT (cm)
    57) !      zstar         log pressure altitude coordinate (km)
    58) !
    59) !   OUTPUT:
    60) !      hno3_cond     vertically modified
    61) !
    62) !======================================================================

    87)       subroutine strat_aer_settling( ad, press, timestep, zstar, temp, &
    88)                                      hno3_cond, radius_nat, ncol, lchnk, aero_ndx )
    89) 
    90)       use ppgrid,      only : pcols, pver
    91)       use chem_mods,   only : adv_mass
    92)       use physconst,   only : gravit
    93)       use cam_history, only : outfld
    94) 
    95) !-----------------------------------------------------------------------
    96) !     ... dummy arguments
    97) !-----------------------------------------------------------------------
    98)       integer, intent(in)     ::  ncol                                    ! columns in chunk
    99)       integer, intent(in)     ::  lchnk                                   ! chunk id
   100)       integer, intent(in)     ::  aero_ndx                                ! aerosol index
   101)       real(r8), intent(in)    ::  timestep                                ! model time step (s)
   102)       real(r8), intent(in)    ::  ad(ncol,pver)                           ! Air density (molecules cm-3)
   103)       real(r8), intent(in)    ::  radius_nat(ncol,pver)                   ! Mean radius of NAT (cm)
   104)       real(r8), intent(in)    ::  zstar(ncol,pver)                        ! altitude (km)
   105)       real(r8), intent(in)    ::  press(pcols,pver)                       ! Pressure (Pa)
   106)       real(r8), intent(in)    ::  temp(pcols,pver)                        ! temperature (K)
   107)       real(r8), intent(inout) ::  hno3_cond(ncol,pver)                    ! Condensed Phase HNO3 (VMR)
   108) 
   109) !-----------------------------------------------------------------------
   110) !     ... local variables
   111) !-----------------------------------------------------------------------
   112)       real(r8), parameter :: avo_num       = 6.022e23_r8, &    ! molecules/mole
   113)                              MW_air        = 28.8_r8, &        ! grams/mole air
   114)                              nat_dens      = 1.6_r8, &         ! g/cm^3
   115)                              shape_fac_nat = 1.0_r8, &         ! TBD
   116)                              sigma_nat     = 1.6_r8, &         ! Width of distribution
   117)                              av_const      = 2.117265e4_r8, &  ! (8*8.31448*1000 / PI)
   118)                              km2cm         = 1.e5_r8, &        ! km to cm
   119)                              m2cm          = 1.e2_r8, &        ! m to cm
   120)                              c1            = 2._r8/9._r8
   121) 
   122)       integer  :: i, k, kp1
   123)       real(r8) :: gravity                   ! gravity cm/s^2
   124)       real(r8) :: Cc_nat                    ! Cunningham Correction Factor
   125)       real(r8) :: dt_dz                     ! dt / dz, sec cm-1
   126)       real(r8) :: flux_nat                  ! aerosol flux, molec cm-2 sec-1
   127)       real(r8) :: mean_vel                  ! mean velocity, cm sec-1
   128)       real(r8) :: mfp                       ! Mean Free Path
   129)       real(r8) :: vel_nat                   ! Terminal velocity (cm sec-1)
   130)       real(r8) :: rad_nat                   ! wrk radius nat (cm)
   131)       real(r8) :: visc                      ! Dynamic viscosity of air
   132)       real(r8) :: depos                     ! molecules/cm**3 deposited
   133)       real(r8) :: atm_dens, atm_densa       ! total atm density and inverse
   134)       real(r8) :: cond_hno3                 ! wrk variables
   135)       real(r8) :: const_nat                 ! wrk variables
   136)       real(r8) :: t                         ! working temperatue
   137)       real(r8) :: velnat(ncol,pver)         ! holding variable for output
   138)       logical  :: lon_mask(ncol)            ! longitude logic mask
   139) 
   140)       gravity       = gravit*m2cm       ! (cm/s^2)
   141) 
   142)       const_nat = exp( 8._r8*(log(sigma_nat))**2 )
   143)       do k = 1,pver
   144) 	 velnat(:,k) = 0._r8
   145)       end do
   146) 
   147) 
   148) !----------------------------------------------------------------------
   149) !     ... derive Aerosol Settling (explicit approach)
   150) !----------------------------------------------------------------------
   151) Level_loop : &
   152)       do k = 2,pver-1
   153) !----------------------------------------------------------------------
   154) !     ... operate between 2.0hPa and 300hPa and only where nat exist
   155) !----------------------------------------------------------------------
   156) 	 kp1 = k + 1
   157) 	 lon_mask(:) = press(:ncol,k) >= 2.e2_r8 .and. press(:ncol,k) <= 300.e2_r8 &
   158)                        .and. (radius_nat(:,k) > 0._r8 )
   159) 	 if( any( lon_mask(:) ) ) then
   160) Column_loop : &
   161)             do i = 1,ncol
   162) 	       if( lon_mask(i) ) then
   163)                   t = temp(i,k)
   164) !----------------------------------------------------------------------
   165) !     ... General Setup for NAT
   166) !         Calculate the settling of the NAT Aerosol
   167) !         NOTE: Index "k" is the box that is being calculated
   168) !               Index "k-1" is flux from above into the k box
   169) !               A positive NET {flux*dt/dz} adds to the "k" box
   170) !----------------------------------------------------------------------
   171) !     ... mean Molecular Velocity, cm sec-1
   172) !----------------------------------------------------------------------
   173)                   mean_vel  = sqrt( av_const*t/MW_air )*100._r8
   174) !----------------------------------------------------------------------
   175) !     ... dynamic Viscosity, g cm-1 sec-1
   176) !----------------------------------------------------------------------
   177) 	          visc = (t*1.458e-6_r8)**1.5_r8 /(t + 110.4_r8)*10000._r8
   178) !----------------------------------------------------------------------
   179) !     ... mean Free Path, cm
   180) !----------------------------------------------------------------------
   181) 		  atm_dens  = ad(i,k)
   182) 		  atm_densa = 1._r8/atm_dens
   183)                   mfp = visc* avo_num / (.499_r8*atm_dens*MW_air*mean_vel)
   184) !----------------------------------------------------------------------
   185) !     ... dt / dz, sec cm-1; NOTE: zstar is in km
   186) !----------------------------------------------------------------------
   187)                   dt_dz = timestep / ((zstar(i,k-1) - zstar(i,k))*km2cm)
   188) !----------------------------------------------------------------------
   189) !     ... calculate NAT Aerosol Settling
   190) !----------------------------------------------------------------------
   191) 		  rad_nat   = radius_nat(i,k)
   192)                   cond_hno3 = hno3_cond(i,k)*atm_dens
   193) !----------------------------------------------------------------------
   194) !     ... Cunningham Correction Factor, Unitless
   195) !----------------------------------------------------------------------
   196)                   Cc_nat = 1._r8 + (mfp/rad_nat)*(1.246_r8 + .42_r8*exp( -.87_r8*rad_nat/mfp ))
   197) !----------------------------------------------------------------------
   198) !     ... terminal Velocity of Aerosol, cm sec-1
   199) !----------------------------------------------------------------------
   200) 	          vel_nat = c1*rad_nat**2 * nat_dens*gravity*Cc_nat/visc*shape_fac_nat
   201) 		  velnat(i,k) = vel_nat
   202) !----------------------------------------------------------------------
   203) !     ... aerosol Flux, Cond-phase molecules cm-2 sec-1
   204) !----------------------------------------------------------------------
   205)          	  flux_nat = cond_hno3*vel_nat* const_nat
   206) !----------------------------------------------------------------------
   207) !     ... calculate NAT Aerosol Settling (i.e., HNO3 redistribution)
   208) !----------------------------------------------------------------------
   209) 		  depos = min( cond_hno3,flux_nat*dt_dz )
   210) !----------------------------------------------------------------------
   211) !     ... modify the HNO3_cond in level "k"
   212) !----------------------------------------------------------------------
   213) 	          hno3_cond(i,k) = (cond_hno3 - depos)*atm_densa
   214) !----------------------------------------------------------------------
   215) !     ... modify the HNO3_cond in level "k+1"
   216) !----------------------------------------------------------------------
   217) 		  hno3_cond(i,kp1) = hno3_cond(i,kp1) + depos/ad(i,kp1)
   218)                end if
   219)             end do Column_loop
   220)          end if
   221)       end do Level_loop
   222) 
   223) !----------------------------------------------------------------------
   224) !	... output nat velocity
   225) !----------------------------------------------------------------------
   226)       if( aero_ndx == 1 ) then
   227)          call outfld( 'VEL_NAT1', velnat, ncol, lchnk )
   228)       else if( aero_ndx == 2 ) then
   229)          call outfld( 'VEL_NAT2', velnat, ncol, lchnk )
   230)       end if
   231) 
   232)       end subroutine strat_aer_settling
   233) 
   234)       end module mo_aero_settling
