   127)

   129)             mpicom=mpicom_CPLID, iamroot=iamroot_CPLID)
   130)

   141)           call mct_aVect_init(o2x_ax(emi), rList=seq_flds_o2x_fields, lsize=lsize_a)
   142)           call mct_aVect_zero(o2x_ax(emi))
   143)        enddo

   146)           call mct_aVect_init(i2x_ax(eii), rList=seq_flds_i2x_fields, lsize=lsize_a)
   147)           call mct_aVect_zero(i2x_ax(eii))

   151)        samegrid_ao = .true.
   152)        if (trim(atm_gnam) /= trim(lnd_gnam)) samegrid_al = .false.
   153)        if (trim(atm_gnam) /= trim(ocn_gnam)) samegrid_ao = .false.
   154)        
   155)        if (ocn_c2_atm) then
   156)           if (iamroot_CPLID) then
   157)              write(logunit,*) ' '
   158)              write(logunit,F00) 'Initializing mapper_So2a'
   159)           end if
   160)           call seq_map_init_rcfile(mapper_So2a, ocn(1), atm(1), &
   161)                'seq_maps.rc','ocn2atm_smapname:','ocn2atm_smaptype:',samegrid_ao, &
   162)                'mapper_So2a initialization',esmf_map_flag)
   163)        end if
   164)        
   165)        ! needed for domain checking
   166)        if (ocn_present) then
   167)           if (iamroot_CPLID) then
   168)              write(logunit,*) ' '
   169)              write(logunit,F00) 'Initializing mapper_Fo2a'
   170)           end if
   171)           call seq_map_init_rcfile(mapper_Fo2a, ocn(1), atm(1), &
   172)                'seq_maps.rc','ocn2atm_fmapname:','ocn2atm_fmaptype:',samegrid_ao, &
   173)                'mapper_Fo2a initialization',esmf_map_flag)
   174)        endif
   175)        call shr_sys_flush(logunit)
   176)        
   177)        if (ice_c2_atm) then
   178)           if (iamroot_CPLID) then
   179)              write(logunit,*) ' '
   180)              write(logunit,F00) 'Initializing mapper_Si2a'
   181)           end if
   182)           call seq_map_init_rcfile(mapper_Si2a, ice(1), atm(1), &
   183)                'seq_maps.rc','ice2atm_smapname:','ice2atm_smaptype:',samegrid_ao, &
   184)                'mapper_Si2a initialization',esmf_map_flag)
   185)        end if
   186)        
   187)        ! needed for domain checking
   188)        if (ice_present) then
   189)           if (iamroot_CPLID) then
   190)              write(logunit,*) ' '
   191)              write(logunit,F00) 'Initializing mapper_Fi2a'
   192)           end if
   193)           call seq_map_init_rcfile(mapper_Fi2a, ice(1), atm(1), &
   194)                'seq_maps.rc','ice2atm_fmapname:','ice2atm_fmaptype:',samegrid_ao, &
   195)                'mapper_Fi2a initialization',esmf_map_flag)
   196)        endif
   197)        call shr_sys_flush(logunit)
   198)        
   199)        ! needed for domain checking
   200)        if (lnd_present) then
   201)           if (iamroot_CPLID) then
   202)              write(logunit,*) ' '
   203)              write(logunit,F00) 'Initializing mapper_Fl2a'
   204)           end if
   205)           call seq_map_init_rcfile(mapper_Fl2a, lnd(1), atm(1), &
   206)                'seq_maps.rc','lnd2atm_fmapname:','lnd2atm_fmaptype:',samegrid_al, &
   207)                'mapper_Fl2a initialization',esmf_map_flag)
   208)        endif
   209)        call shr_sys_flush(logunit)
   210) 
   211)        if (lnd_c2_atm) then
   212)           if (iamroot_CPLID) then
   213)              write(logunit,*) ' '
   214)              write(logunit,F00) 'Initializing mapper_Sl2a'
   215)           end if
   216)           call seq_map_init_rcfile(mapper_Sl2a, lnd(1), atm(1), &
   217)                'seq_maps.rc','lnd2atm_smapname:','lnd2atm_smaptype:',samegrid_al, &
   218)                'mapper_Sl2a initialization',esmf_map_flag)
   219)        end if
   220) 
   221)        
   222)     end if
   223)

   228)   subroutine prep_atm_mrg(infodata, fractions_ax, xao_ax, timer_mrg)
   229) 
   230)     !---------------------------------------------------------------
   231)     ! Description
   232)     ! Prepare run phase, including running the merge
   233)     !
   234)     ! Arguments
   235)     type(seq_infodata_type) , intent(in)    :: infodata
   236)     type(mct_aVect)         , intent(in)    :: fractions_ax(:)
   237)     type(mct_aVect)         , intent(in)    :: xao_ax(:) 
   238)     character(len=*)        , intent(in)    :: timer_mrg
   239)     !
   240)     ! Local Variables
   241)     integer                  :: eli, eoi, eii, exi, efi, eai, emi
   242)     type(mct_avect), pointer :: x2a_ax
   243)     character(*), parameter  :: subname = '(prep_atm_mrg)'
   244)     character(*), parameter  :: F00 = "('"//subname//" : ', 4A )"
   245)     !---------------------------------------------------------------
   246) 
   247)     call t_drvstartf (trim(timer_mrg),barrier=mpicom_CPLID)
   248)     do eai = 1,num_inst_atm
   249)        ! Use fortran mod to address ensembles in merge
   250)        eli = mod((eai-1),num_inst_lnd) + 1
   251)        eoi = mod((eai-1),num_inst_ocn) + 1
   252)        eii = mod((eai-1),num_inst_ice) + 1
   253)        exi = mod((eai-1),num_inst_xao) + 1
   254)        efi = mod((eai-1),num_inst_frc) + 1
   255)        emi = mod((eai-1),num_inst_max) + 1
   256) 
   257)        x2a_ax => component_get_x2c_cx(atm(eai)) ! This is actually modifying x2a_ax
   258)        call prep_atm_merge(l2x_ax(eli), o2x_ax(emi), xao_ax(exi), i2x_ax(eii), &
   259)             fractions_ax(efi), x2a_ax)
   260)     enddo
   261)     call t_drvstopf  (trim(timer_mrg))
   262) 
   263)   end subroutine prep_atm_mrg

   267)   subroutine prep_atm_merge( l2x_a, o2x_a, xao_a, i2x_a, fractions_a, x2a_a )
   268) 
   269)     !----------------------------------------------------------------------- 
   270)     !
   271)     ! Arguments
   272)     type(mct_aVect), intent(in)    :: l2x_a
   273)     type(mct_aVect), intent(in)    :: o2x_a
   274)     type(mct_aVect), intent(in)    :: xao_a
   275)     type(mct_aVect), intent(in)    :: i2x_a
   276)     type(mct_aVect), intent(in)    :: fractions_a
   277)     type(mct_aVect), intent(inout) :: x2a_a
   278)     !
   279)     ! Local workspace
   280)     real(r8) :: fracl, fraci, fraco
   281)     integer  :: n,ka,ki,kl,ko,kx,kof,kif,klf,i,i1,o1
   282)     integer  :: lsize       
   283)     integer  :: index_x2a_Sf_lfrac
   284)     integer  :: index_x2a_Sf_ifrac
   285)     integer  :: index_x2a_Sf_ofrac
   286)     character(CL),allocatable :: field_atm(:)   ! string converted to char
   287)     character(CL),allocatable :: field_lnd(:)   ! string converted to char
   288)     character(CL),allocatable :: field_ice(:)   ! string converted to char
   289)     character(CL),allocatable :: field_xao(:)   ! string converted to char
   290)     character(CL),allocatable :: field_ocn(:)   ! string converted to char
   291)     character(CL),allocatable :: itemc_atm(:)   ! string converted to char
   292)     character(CL),allocatable :: itemc_lnd(:)   ! string converted to char
   293)     character(CL),allocatable :: itemc_ice(:)   ! string converted to char
   294)     character(CL),allocatable :: itemc_xao(:)   ! string converted to char
   295)     character(CL),allocatable :: itemc_ocn(:)   ! string converted to char
   296)     logical :: iamroot  
   297)     character(CL),allocatable :: mrgstr(:)   ! temporary string
   298)     logical, save :: first_time = .true.
   299)     type(mct_aVect_sharedindices),save :: l2x_sharedindices
   300)     type(mct_aVect_sharedindices),save :: o2x_sharedindices
   301)     type(mct_aVect_sharedindices),save :: i2x_sharedindices
   302)     type(mct_aVect_sharedindices),save :: xao_sharedindices
   303)     logical, pointer, save :: lmerge(:),imerge(:),xmerge(:),omerge(:)
   304)     integer, pointer, save :: lindx(:), iindx(:), oindx(:),xindx(:)
   305)     integer, save          :: naflds, nlflds,niflds,noflds,nxflds
   306)     character(*), parameter   :: subname = '(prep_atm_merge) '
   307)     !-----------------------------------------------------------------------
   308)     !
   309)     call seq_comm_getdata(CPLID, iamroot=iamroot)
   310) 
   311)     if (first_time) then
   312)           
   313)        naflds = mct_aVect_nRattr(x2a_a)
   314)        nlflds = mct_aVect_nRattr(l2x_a)
   315)        niflds = mct_aVect_nRattr(i2x_a)
   316)        noflds = mct_aVect_nRattr(o2x_a)
   317)        nxflds = mct_aVect_nRattr(xao_a)
   318) 
   319)        allocate(lindx(naflds), lmerge(naflds))
   320)        allocate(iindx(naflds), imerge(naflds))
   321)        allocate(xindx(naflds), xmerge(naflds))
   322)        allocate(oindx(naflds), omerge(naflds))
   323)        allocate(field_atm(naflds), itemc_atm(naflds))
   324)        allocate(field_lnd(nlflds), itemc_lnd(nlflds))
   325)        allocate(field_ice(niflds), itemc_ice(niflds))
   326)        allocate(field_ocn(noflds), itemc_ocn(noflds))
   327)        allocate(field_xao(nxflds), itemc_xao(nxflds))
   328)        allocate(mrgstr(naflds))
   329) 
   330)        lindx(:) = 0
   331)        iindx(:) = 0
   332)        xindx(:) = 0
   333)        oindx(:) = 0
   334)        lmerge(:)  = .true.
   335)        imerge(:)  = .true.
   336)        xmerge(:)  = .true.
   337)        omerge(:)  = .true.
   338) 
   339)        do ka = 1,naflds
   340)           field_atm(ka) = mct_aVect_getRList2c(ka, x2a_a)
   341)           itemc_atm(ka) = trim(field_atm(ka)(scan(field_atm(ka),'_'):))
   342)        enddo
   343)        do kl = 1,nlflds
   344)           field_lnd(kl) = mct_aVect_getRList2c(kl, l2x_a)
   345)           itemc_lnd(kl) = trim(field_lnd(kl)(scan(field_lnd(kl),'_'):))
   346)        enddo
   347)        do ki = 1,niflds
   348)           field_ice(ki) = mct_aVect_getRList2c(ki, i2x_a)
   349)           itemc_ice(ki) = trim(field_ice(ki)(scan(field_ice(ki),'_'):))
   350)        enddo
   351)        do ko = 1,noflds
   352)           field_ocn(ko) = mct_aVect_getRList2c(ko, o2x_a)
   353)           itemc_ocn(ko) = trim(field_ocn(ko)(scan(field_ocn(ko),'_'):))
   354)        enddo
   355)        do kx = 1,nxflds
   356)           field_xao(kx) = mct_aVect_getRList2c(kx, xao_a)
   357)           itemc_xao(kx) = trim(field_xao(kx)(scan(field_xao(kx),'_'):))
   358)        enddo
   359) 
   360)        call mct_aVect_setSharedIndices(l2x_a, x2a_a, l2x_SharedIndices)
   361)        call mct_aVect_setSharedIndices(o2x_a, x2a_a, o2x_SharedIndices)
   362)        call mct_aVect_setSharedIndices(i2x_a, x2a_a, i2x_SharedIndices)
   363)        call mct_aVect_setSharedIndices(xao_a, x2a_a, xao_SharedIndices)
   364) 
   365)        ! Field naming rules
   366)        ! Only atm states that are Sx_... will be merged
   367)        ! Only fluxes that are F??x_... will be merged 
   368)        ! All fluxes will be multiplied by corresponding component fraction
   369) 
   370)        do ka = 1,naflds
   371)           !--- document merge ---
   372)           mrgstr(ka) = subname//'x2a%'//trim(field_atm(ka))//' ='
   373)           if (field_atm(ka)(1:2) == 'PF') then
   374)              cycle ! if flux has first character as P, pass straight through 
   375)           end if
   376)           if (field_atm(ka)(1:1) == 'S' .and. field_atm(ka)(2:2) /= 'x') then
   377)              cycle ! any state fields that are not Sx_ will just be copied
   378)           end if
   379) 
   380)           do kl = 1,nlflds
   381)              if (trim(itemc_atm(ka)) == trim(itemc_lnd(kl))) then
   382)                 if ((trim(field_atm(ka)) == trim(field_lnd(kl)))) then
   383)                    if (field_lnd(kl)(1:1) == 'F') lmerge(ka) = .false.
   384)                 end if
   385)                 ! --- make sure only one field matches ---
   386)                 if (lindx(ka) /= 0) then
   387)                    write(logunit,*) subname,' ERROR: found multiple kl field matches for ',trim(itemc_lnd(kl))
   388)                    call shr_sys_abort(subname//' ERROR multiple kl field matches')
   389)                 endif
   390)                 lindx(ka) = kl
   391)              end if
   392)           end do
   393)           do ki = 1,niflds
   394)              if (field_ice(ki)(1:1) == 'F' .and. field_ice(ki)(2:4) == 'ioi') then
   395)                 cycle ! ignore all fluxes that are ice/ocn fluxes
   396)              end if
   397)              if (trim(itemc_atm(ka)) == trim(itemc_ice(ki))) then
   398)                 if ((trim(field_atm(ka)) == trim(field_ice(ki)))) then
   399)                    if (field_ice(ki)(1:1) == 'F') imerge(ka) = .false.
   400)                 end if
   401)                 ! --- make sure only one field matches ---
   402)                 if (iindx(ka) /= 0) then
   403)                    write(logunit,*) subname,' ERROR: found multiple ki field matches for ',trim(itemc_ice(ki))
   404)                    call shr_sys_abort(subname//' ERROR multiple ki field matches')
   405)                 endif
   406)                 iindx(ka) = ki
   407)              end if
   408)           end do
   409)           do kx = 1,nxflds
   410)              if (trim(itemc_atm(ka)) == trim(itemc_xao(kx))) then
   411)                 if ((trim(field_atm(ka)) == trim(field_xao(kx)))) then
   412)                    if (field_xao(kx)(1:1) == 'F') xmerge(ka) = .false.
   413)                 end if
   414)                 ! --- make sure only one field matches ---
   415)                 if (xindx(ka) /= 0) then
   416)                    write(logunit,*) subname,' ERROR: found multiple kx field matches for ',trim(itemc_xao(kx))
   417)                    call shr_sys_abort(subname//' ERROR multiple kx field matches')
   418)                 endif
   419)                 xindx(ka) = kx
   420)              end if
   421)           end do
   422)           do ko = 1,noflds
   423)              if (trim(itemc_atm(ka)) == trim(itemc_ocn(ko))) then
   424)                 if ((trim(field_atm(ka)) == trim(field_ocn(ko)))) then
   425)                    if (field_ocn(ko)(1:1) == 'F') omerge(ka) = .false.
   426)                 end if
   427)                 ! --- make sure only one field matches ---
   428)                 if (oindx(ka) /= 0) then
   429)                    write(logunit,*) subname,' ERROR: found multiple ko field matches for ',trim(itemc_ocn(ko))
   430)                    call shr_sys_abort(subname//' ERROR multiple ko field matches')
   431)                 endif
   432)                 oindx(ka) = ko
   433)              end if
