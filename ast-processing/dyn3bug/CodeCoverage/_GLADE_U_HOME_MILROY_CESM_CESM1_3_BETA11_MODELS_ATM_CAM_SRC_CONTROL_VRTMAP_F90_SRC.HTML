     2) subroutine vrtmap (pkdim   ,pmap    ,sigln   ,dsigln  ,kdpmap  )
     3) !----------------------------------------------------------------------- 
     4) ! 
     5) ! Purpose: Map indices of an artificial evenly spaced (in log) vertical grid to
     6) !          the indices of the log of the model vertical grid.  The resultant
     7) !          array of mapped indices will be used by "kdpfnd" to find the vertical
     8) !          location of any departure point relative to the model grid.
     9) ! 
    10) ! Method: 
    11) ! 
    12) ! Author: Jerry Olson
    13) ! 
    14) !-----------------------------------------------------------------------
    15)    use shr_kind_mod, only: r8 => shr_kind_r8
    16)    use abortutils,   only: endrun
    17)    use cam_logfile,  only: iulog
    18) #if (!defined UNICOSMP)
    19)    use srchutil,     only: ismin
    20) #endif
    21) !-----------------------------------------------------------------------
    22)    implicit none
    23) !-----------------------------------------------------------------------
    24) !
    25) ! Arguments
    26) !
    27)    integer, intent(in) :: pkdim             ! dimension of "sigln" and "dsigln"
    28)    integer, intent(in) :: pmap              ! dimension of "kdpmap"
    29) 
    30)    real(r8), intent(in) :: sigln (pkdim)    ! model levels (log(eta))
    31)    real(r8), intent(in) :: dsigln(pkdim)    ! intervals between model levels (log)
    32) 
    33)    integer, intent(out) :: kdpmap(pmap)     ! array of mapped indices
    34) !
    35) !---------------------------Local variables-----------------------------
    36) !
    37)    integer imin              ! |
    38)    integer k                 ! |-- indices
    39)    integer kk                ! |
    40)    integer newmap            ! estimated value of "pmap"
    41) 
    42)    real(r8) del              ! artificial grid interval
    43)    real(r8) dp               ! artificial departure point
    44)    real(r8) eps              ! epsilon factor
    45) #if (defined UNICOSMP)
    46)    integer, external :: ismin
    47) #endif
    48) !
    49) !-----------------------------------------------------------------------
    50) !
    51)    eps = 1.e-05_r8
    52)    del = ( sigln(pkdim) - sigln(1) )/real(pmap,r8)
    53)    imin = ismin( pkdim-1,dsigln, 1 )
    54)    if (del + eps  >=  dsigln(imin)) then
    55)       newmap = ( sigln(pkdim) - sigln(1) )/dsigln(imin) + 1
    56)       write(iulog,9000) pmap,newmap
    57)       call endrun()
    58)    end if
    59) 
    60)    kdpmap(1) = 1
    61)    do kk = 2,pmap
    62)       dp = sigln(1) + real(kk-1,r8)*del
    63)       do k = 1,pkdim-1
    64)          if(dp > sigln(k) + eps) then
    65)             kdpmap(kk) = k
    66)          end if
    67)       end do
    68)    end do
    69) 
    70)    return
    71) 9000 format(' VRTMAP:  Not enough artificial grid intervals.'/ &
    72)             ' Currently, "pmap" is set to ',i20/ &
    73)             ' Reset parameter "pmap" to at least ',i20)
    74) end subroutine vrtmap
