     3)                   qbot    ,srfrpdel,shflx   ,lhflx   ,qflx    )
     4) !----------------------------------------------------------------------- 
     5) ! 
     6) ! Purpose: 
     7) ! Check if moisture flux into the ground is exceeding the total
     8) ! moisture content of the lowest model layer (creating negative moisture
     9) ! values).  If so, then subtract the excess from the moisture and
    10) ! latent heat fluxes and add it to the sensible heat flux.
    11) ! 
    12) ! Method: 
    13) ! <Describe the algorithm(s) used in the routine.> 
    14) ! <Also include any applicable external references.> 
    15) ! 
    16) ! Author: J. Olson
    17) ! 
    18) !-----------------------------------------------------------------------
    19)    use shr_kind_mod, only: r8 => shr_kind_r8
    20)    use ppgrid
    21)    use phys_grid,    only: get_lat_p, get_lon_p
    22)    use physconst,    only: gravit, latvap
    23)    use constituents, only: qmin, pcnst
    24)    use cam_logfile,  only: iulog
    25) 
    26)    implicit none
    27) 
    28) !
    29) ! Input arguments
    30) !
    31)    character*8, intent(in) :: subnam         ! name of calling routine
    32) !
    33)    integer, intent(in) :: lchnk              ! chunk index
    34)    integer, intent(in) :: ncol               ! number of atmospheric columns
    35) !
    36)    real(r8), intent(in) :: ztodt             ! two times model timestep (2 delta-t)
    37)    real(r8), intent(in) :: qbot(pcols,pcnst) ! moisture at lowest model level
    38)    real(r8), intent(in) :: srfrpdel(pcols)   ! 1./(pint(K+1)-pint(K))
    39) !
    40) ! Input/Output arguments
    41) !
    42)    real(r8), intent(inout) :: shflx(pcols)   ! Surface sensible heat flux (J/m2/s)
    43)    real(r8), intent(inout) :: lhflx(pcols)   ! Surface latent   heat flux (J/m2/s)
    44)    real(r8), intent(inout) :: qflx (pcols,pcnst)   ! surface water flux (kg/m^2/s)
    45) !
    46) !---------------------------Local workspace-----------------------------
    47) !
    48)    integer :: i,ii              ! longitude indices
    49)    integer :: iw                ! i index of worst violator
    50)    integer :: indxexc(pcols)    ! index array of points with excess flux
    51)    integer :: nptsexc           ! number of points with excess flux
    52) !
    53)    real(r8):: worst             ! biggest violator
    54)    real(r8):: excess(pcols)     ! Excess downward sfc latent heat flux
    55) !
    56) !-----------------------------------------------------------------------
    57) !
    58) ! Compute excess downward (negative) q flux compared to a theoretical
    59) ! maximum downward q flux.  The theoretical max is based upon the
    60) ! given moisture content of lowest level of the model atmosphere.
    61) !
    62)    nptsexc = 0
    63)    do i = 1,ncol
    64)       excess(i) = qflx(i,1) - (qmin(1) - qbot(i,1))/(ztodt*gravit*srfrpdel(i))
    65) !
    66) ! If there is an excess downward (negative) q flux, then subtract
    67) ! excess from "qflx" and "lhflx" and add to "shflx".
    68) !
    69)       if (excess(i) < 0._r8) then
    70)          nptsexc = nptsexc + 1
    71)          indxexc(nptsexc) = i
    72)          qflx (i,1) = qflx (i,1) - excess(i)
    73)          lhflx(i) = lhflx(i) - excess(i)*latvap
    74)          shflx(i) = shflx(i) + excess(i)*latvap
    75)       end if
    76)    end do
    77) !
    78) ! Write out worst value if excess
    79) !
    80)    if (nptsexc.gt.0) then
    81)       worst = 0._r8
    82)       do ii=1,nptsexc
    83)          i = indxexc(ii)
    84)          if (excess(i) < worst) then
    85)             worst = excess(i)
    86)             iw = i
    87)          end if
    88)       end do
    89)       write(iulog,9000) subnam,nptsexc,worst, lchnk, iw, get_lat_p(lchnk,iw),get_lon_p(lchnk,iw)
    90)    end if
    91) !
    92)    return
    93) 9000 format(' QNEG4 WARNING from ',a8 &
    94)             ,' Max possible LH flx exceeded at ',i4,' points. ' &
    95)             ,', Worst excess = ',1pe12.4 &
    96)             ,', lchnk = ',i3 &
    97)             ,', i = ',i4 &
    98)             ,', same as indices lat =', i4 &
    99)             ,', lon =', i4 &
   100)            )
   101) end subroutine qneg4
