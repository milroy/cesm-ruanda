    89) subroutine phys_register
    90)     !----------------------------------------------------------------------- 
    91)     ! 
    92)     ! Purpose: Register constituents and physics buffer fields.
    93)     ! 
    94)     ! Author:    CSM Contact: M. Vertenstein, Aug. 1997
    95)     !            B.A. Boville, Oct 2001
    96)     !            A. Gettelman, Nov 2010 - put micro/macro physics into separate routines
    97)     ! 
    98)     !-----------------------------------------------------------------------
    99)     use physics_buffer,     only: pbuf_init_time
   100)     use physics_buffer,     only: pbuf_add_field, dtype_r8, pbuf_register_subcol
   101)     use shr_kind_mod,       only: r8 => shr_kind_r8
   102)     use spmd_utils,         only: masterproc
   103)     use constituents,       only: pcnst, cnst_add, cnst_chk_dim, cnst_name
   104) 
   105)     use cam_control_mod,    only: moist_physics
   106)     use chemistry,          only: chem_register
   107)     use cloud_fraction,     only: cldfrc_register
   108)     use stratiform,         only: stratiform_register
   109)     use microp_driver,      only: microp_driver_register
   110)     use microp_aero,        only: microp_aero_register
   111)     use macrop_driver,      only: macrop_driver_register
   112)     use clubb_intr,         only: clubb_register_cam
   113)     use conv_water,         only: conv_water_register
   114)     use physconst,          only: mwdry, cpair, mwh2o, cpwv
   115)     use tracers,            only: tracers_register
   116)     use check_energy,       only: check_energy_register
   117)     use carma_intr,         only: carma_register
   118)     use cam3_aero_data,     only: cam3_aero_data_on, cam3_aero_data_register
   119)     use cam3_ozone_data,    only: cam3_ozone_data_on, cam3_ozone_data_register
   120)     use ghg_data,           only: ghg_data_register
   121)     use vertical_diffusion, only: vd_register
   122)     use convect_deep,       only: convect_deep_register
   123)     use convect_shallow,    only: convect_shallow_register
   124)     use radiation,          only: radiation_register
   125)     use co2_cycle,          only: co2_register
   126)     use flux_avg,           only: flux_avg_register
   127)     use iondrag,            only: iondrag_register
   128)     use ionosphere,         only: ionos_register
   129)     use string_utils,       only: to_lower
   130)     use prescribed_ozone,   only: prescribed_ozone_register
   131)     use prescribed_volcaero,only: prescribed_volcaero_register
   132)     use prescribed_strataero,only: prescribed_strataero_register
   133)     use prescribed_aero,    only: prescribed_aero_register
   134)     use prescribed_ghg,     only: prescribed_ghg_register
   135)     use sslt_rebin,         only: sslt_rebin_register
   136)     use aoa_tracers,        only: aoa_tracers_register
   137)     use aircraft_emit,      only: aircraft_emit_register
   138)     use cam_diagnostics,    only: diag_register
   139)     use cloud_diagnostics,  only: cloud_diagnostics_register
   140)     use rad_constituents,   only: rad_cnst_get_info ! Added to query if it is a modal aero sim or not
   141)     use subcol,             only: subcol_register
   142)     use subcol_utils,       only: is_subcol_on
   143) 
   144) 
   145)     !---------------------------Local variables-----------------------------
   146)     !
   147)     integer  :: m        ! loop index
   148)     integer  :: mm       ! constituent index 
   149)     integer  :: nmodes
   150)     !-----------------------------------------------------------------------
   151) 
   152)     ! Get physics options
   153)     call phys_getopts(shallow_scheme_out=shallow_scheme, &
   154)                       macrop_scheme_out=macrop_scheme,   &
   155)                       microp_scheme_out=microp_scheme,   &
   156)                       do_clubb_sgs_out=do_clubb_sgs,     &
   157)                       use_subcol_microp_out=use_subcol_microp, &
   158)                       state_debug_checks_out=state_debug_checks)
   159) 
   160)     ! Initialize dyn_time_lvls
   161)     call pbuf_init_time()
   162) 
   163)     ! Register the subcol scheme
   164)     call subcol_register()
   165) 
   166)     ! Register water vapor.
   167)     ! ***** N.B. ***** This must be the first call to cnst_add so that
   168)     !                  water vapor is constituent 1.
   169)     if (moist_physics) then
   170)        call cnst_add('Q', mwh2o, cpwv, 1.E-12_r8, mm, &
   171)             longname='Specific humidity', readiv=.true., is_convtran1=.true.)
   172)     else
   173)        call cnst_add('Q', mwh2o, cpwv, 0.0_r8, mm, &
   174)             longname='Specific humidity', readiv=.false., is_convtran1=.true.)
   175)     end if
   176) 
   177)     ! Fields for physics package diagnostics
   178)     call pbuf_add_field('TINI',      'physpkg', dtype_r8, (/pcols,pver/), tini_idx)
   179)     call pbuf_add_field('QINI',      'physpkg', dtype_r8, (/pcols,pver/), qini_idx)
   180)     call pbuf_add_field('CLDLIQINI', 'physpkg', dtype_r8, (/pcols,pver/), cldliqini_idx)
   181)     call pbuf_add_field('CLDICEINI', 'physpkg', dtype_r8, (/pcols,pver/), cldiceini_idx)
   182) 
   183)     ! check energy package
   184)     call check_energy_register
   185) 
   186)     ! If using an ideal/adiabatic physics option, the CAM physics parameterizations 
   187)     ! aren't called.
   188)     if (moist_physics) then
   189) 
   190)        ! register fluxes for saving across time
   191)        if (phys_do_flux_avg()) call flux_avg_register()
   192) 
   193)        call cldfrc_register()
   194) 
   195)        ! cloud water
   196)        if( microp_scheme == 'RK' ) then
   197)           call stratiform_register()
   198)        elseif( microp_scheme == 'MG' ) then
   199)           if (.not. do_clubb_sgs) call macrop_driver_register()
   200)           call microp_aero_register()
   201)           call microp_driver_register()
   202)        end if
   203)        
   204)        ! Register CLUBB_SGS here
   205)        if (do_clubb_sgs) call clubb_register_cam()
   206)        
   207) 
   208)        call pbuf_add_field('PREC_STR',  'physpkg',dtype_r8,(/pcols/),prec_str_idx)
   209)        call pbuf_add_field('SNOW_STR',  'physpkg',dtype_r8,(/pcols/),snow_str_idx)
   210)        call pbuf_add_field('PREC_PCW',  'physpkg',dtype_r8,(/pcols/),prec_pcw_idx)
   211)        call pbuf_add_field('SNOW_PCW',  'physpkg',dtype_r8,(/pcols/),snow_pcw_idx)
   212)        call pbuf_add_field('PREC_SED',  'physpkg',dtype_r8,(/pcols/),prec_sed_idx)
   213)        call pbuf_add_field('SNOW_SED',  'physpkg',dtype_r8,(/pcols/),snow_sed_idx)
   214)        if (is_subcol_on()) then
   215)          call pbuf_register_subcol('PREC_STR', 'phys_register', prec_str_idx)
   216)          call pbuf_register_subcol('SNOW_STR', 'phys_register', snow_str_idx)
   217)          call pbuf_register_subcol('PREC_PCW', 'phys_register', prec_pcw_idx)
   218)          call pbuf_register_subcol('SNOW_PCW', 'phys_register', snow_pcw_idx)
   219)          call pbuf_register_subcol('PREC_SED', 'phys_register', prec_sed_idx)
   220)          call pbuf_register_subcol('SNOW_SED', 'phys_register', snow_sed_idx)
   221)        end if
   222) 
   223)     ! Who should add FRACIS? 
   224)     ! -- It does not seem that aero_intr should add it since FRACIS is used in convection
   225)     !     even if there are no prognostic aerosols ... so do it here for now 
   226)        call pbuf_add_field('FRACIS','physpkg',dtype_r8,(/pcols,pver,pcnst/),m)
   227) 
   228)        call conv_water_register()
   229)        
   230)        ! Determine whether its a 'modal' aerosol simulation  or not
   231)        call rad_cnst_get_info(0, nmodes=nmodes)
   232)        clim_modal_aero = (nmodes > 0)
   233) 
   234)        if (clim_modal_aero) then
   235)           call modal_aero_calcsize_reg()
   236)           call modal_aero_wateruptake_reg()
   237)        endif
   238) 
   239)        ! register chemical constituents including aerosols ...
   240)        call chem_register()
   241) 
   242)        ! co2 constituents
   243)        call co2_register()
   244) 
   245)        ! register data model ozone with pbuf
   246)        if (cam3_ozone_data_on) then
   247)           call cam3_ozone_data_register()
   248)        end if
   249)        call prescribed_volcaero_register()
   250)        call prescribed_strataero_register()
   251)        call prescribed_ozone_register()
   252)        call prescribed_aero_register()
   253)        call prescribed_ghg_register()
   254)        call sslt_rebin_register
   255) 
   256)        ! CAM3 prescribed aerosols
   257)        if (cam3_aero_data_on) then
   258)           call cam3_aero_data_register()
   259)        end if
   260) 
   261)        ! register various data model gasses with pbuf
   262)        call ghg_data_register()
   263) 
   264)        ! carma microphysics
   265)        ! 
   266)        call carma_register()
   267) 
   268)        ! Register iondrag variables with pbuf
   269)        call iondrag_register()
   270) 
   271)        ! Register ionosphere variables with pbuf if mode set to ionosphere
   272)        if( waccmx_is('ionosphere') ) then
   273)           call ionos_register()
   274)        endif
   275) 
   276)        call aircraft_emit_register()
   277) 
   278)        ! deep convection
   279)        call convect_deep_register
   280) 
   281)        !  shallow convection
   282)        call convect_shallow_register
   283) 
   284)        ! radiation
   285)        call radiation_register
   286)        call cloud_diagnostics_register
   287) 
   288)        ! vertical diffusion
   289)        if (.not. do_clubb_sgs) call vd_register()
   290)     end if
   291) 
   292)     ! Register diagnostics PBUF
   293)     call diag_register()
   294) 
   295)     ! Register age of air tracers
   296)     call aoa_tracers_register()
   297) 
   298)     ! Register test tracers
   299)     ! ***** N.B. ***** This is the last call to register constituents because
   300)     !                  the test tracers fill the remaining available slots up
   301)     !                  to constituent number PCNST -- regardless of what PCNST is set to.
   302)     call tracers_register()
   303) 
   304)     ! All tracers registered, check that the dimensions are correct
   305)     call cnst_chk_dim()
   306) 
   307)     ! ***NOTE*** No registering constituents after the call to cnst_chk_dim.
   308) 
   309) end subroutine phys_register

   315) subroutine phys_inidat( cam_out, pbuf2d )
   316)     use abortutils, only : endrun
   317) 
   318)     use physics_buffer, only : pbuf_get_index, pbuf_get_field, physics_buffer_desc, pbuf_set_field, dyn_time_lvls
   319) 
   320) 
   321)     use cam_initfiles,       only: initial_file_get_id, topo_file_get_id
   322)     use pio,                 only: file_desc_t
   323)     use ncdio_atm,           only: infld
   324)     use dycore,              only: dycore_is
   325)     use polar_avg,           only: polar_average
   326)     use short_lived_species, only: initialize_short_lived_species
   327)     use comsrf,              only: landm, sgh, sgh30
   328)     use cam_control_mod,     only: aqua_planet
   329) 
   330)     type(cam_out_t),     intent(inout) :: cam_out(begchunk:endchunk)
   331)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   332)     integer :: lchnk, m, n, i, k, ncol
   333)     type(file_desc_t), pointer :: fh_ini, fh_topo
   334)     character(len=8) :: fieldname
   335)     real(r8), pointer :: cldptr(:,:,:,:), convptr_3d(:,:,:,:)
   336)     real(r8), pointer :: tptr(:,:), tptr3d(:,:,:), tptr3d_2(:,:,:)
   337)     real(r8), pointer :: qpert(:,:)
   338) 
   339)     character*11 :: subname='phys_inidat' ! subroutine name
   340)     integer :: tpert_idx, qpert_idx, pblh_idx
   341) 
   342)     logical :: found=.false., found2=.false.
   343)     integer :: ierr
   344)     character(len=4) :: dim1name
   345)     integer :: ixcldice, ixcldliq
   346)     nullify(tptr,tptr3d,tptr3d_2,cldptr,convptr_3d)
   347) 
   348)     fh_ini=>initial_file_get_id()
   349) 
   350)     !   dynamics variables are handled in dyn_init - here we read variables needed for physics 
   351)     !   but not dynamics
   352) 
   353)     if(dycore_is('UNSTRUCTURED')) then  
   354)        dim1name='ncol'
   355)     else
   356)        dim1name='lon'
   357)     end if
   358)     if(aqua_planet) then
   359)        sgh = 0._r8
   360)        sgh30 = 0._r8
   361)        landm = 0._r8
   362)     else
   363)        fh_topo=>topo_file_get_id()
   364)        call infld('SGH', fh_topo, dim1name, 'lat', 1, pcols, begchunk, endchunk, &
   365)             sgh, found, grid_map='PHYS')
   366)        if(.not. found) call endrun('ERROR: SGH not found on topo file')
   367) 
   368)        call infld('SGH30', fh_topo, dim1name, 'lat', 1, pcols, begchunk, endchunk, &
   369)             sgh30, found, grid_map='PHYS')
   370)        if(.not. found) then
   371)           if (masterproc) write(iulog,*) 'Warning: Error reading SGH30 from topo file.'
   372)           if (masterproc) write(iulog,*) 'The field SGH30 will be filled using data from SGH.'
   373)           sgh30 = sgh
   374)        end if
   375) 
   376)        call infld('LANDM_COSLAT', fh_topo, dim1name, 'lat', 1, pcols, begchunk, endchunk, &
   377)             landm, found, grid_map='PHYS')
   378) 
   379)        if(.not.found) call endrun(' ERROR: LANDM_COSLAT not found on topo dataset.')
   380)     end if
   381) 
   382)     allocate(tptr(1:pcols,begchunk:endchunk))
   383) 
   384)     call infld('PBLH', fh_ini, dim1name, 'lat', 1, pcols, begchunk, endchunk, &
   385)          tptr(:,:), found, grid_map='PHYS')
   386)     if(.not. found) then
   387)        tptr(:,:) = 0._r8
   388)        if (masterproc) write(iulog,*) 'PBLH initialized to 0.'
   389)     end if
   390)     pblh_idx = pbuf_get_index('pblh')
   391) 
   392)     call pbuf_set_field(pbuf2d, pblh_idx, tptr)
   393) 
   394)     call infld('TPERT', fh_ini, dim1name, 'lat', 1, pcols, begchunk, endchunk, &
   395)          tptr(:,:), found, grid_map='PHYS')
   396)     if(.not. found) then
   397)        tptr(:,:) = 0._r8
   398)        if (masterproc) write(iulog,*) 'TPERT initialized to 0.'
   399)     end if
   400)     tpert_idx = pbuf_get_index( 'tpert')
   401)     call pbuf_set_field(pbuf2d, tpert_idx, tptr)
   402) 
   403)     fieldname='QPERT'  
   404)     qpert_idx = pbuf_get_index( 'qpert',ierr)
   405)     if (qpert_idx > 0) then
   406)        call infld(fieldname, fh_ini, dim1name, 'lat', 1, pcols, begchunk, endchunk, &
   407)             tptr, found, grid_map='PHYS')
   408)        if(.not. found) then
   409)           tptr=0_r8
   410)           if (masterproc) write(iulog,*) trim(fieldname), ' initialized to 0.'
   411)        end if
   412) 
   413)        allocate(tptr3d_2(pcols,pcnst,begchunk:endchunk))
   414)        tptr3d_2 = 0_r8
   415)        tptr3d_2(:,1,:) = tptr(:,:)
   416) 
   417)        call pbuf_set_field(pbuf2d, qpert_idx, tptr3d_2)
   418)        deallocate(tptr3d_2)
   419)     end if
   420) 
   421)     fieldname='CUSH'
   422)     m = pbuf_get_index('cush')
   423)     call infld(fieldname, fh_ini, dim1name, 'lat', 1, pcols, begchunk, endchunk, &
   424)          tptr, found, grid_map='PHYS')
   425)     if(.not.found) then
   426)        if(masterproc) write(iulog,*) trim(fieldname), ' initialized to 1000.'
   427)        tptr=1000._r8
   428)     end if
   429)     do n=1,dyn_time_lvls
   430)        call pbuf_set_field(pbuf2d, m, tptr, start=(/1,n/), kount=(/pcols,1/))
   431)     end do
   432)     deallocate(tptr)
   433) 
   434)     do lchnk=begchunk,endchunk
   435)        cam_out(lchnk)%tbot(:) = posinf
   436)     end do
   437) 
   438)     !
   439)     ! 3-D fields
   440)     !
   441) 
   442)     allocate(tptr3d(pcols,pver,begchunk:endchunk))
   443) 
   444)     fieldname='CLOUD'
   445)     m = pbuf_get_index('CLD')
   446)     call infld(fieldname, fh_ini, dim1name, 'lev', 'lat', 1, pcols, 1, pver, begchunk, endchunk, &
   447)          tptr3d, found, grid_map='PHYS')
   448)     if(found) then
   449)        do n = 1, dyn_time_lvls
   450)           call pbuf_set_field(pbuf2d, m, tptr3d, (/1,1,n/),(/pcols,pver,1/))
   451)        end do
   452)     else
   453)        call pbuf_set_field(pbuf2d, m, 0._r8)
   454)        if (masterproc) write(iulog,*) trim(fieldname), ' initialized to 0.'
   455)     end if
   456) 
   457)     fieldname='QCWAT'
   458)     m = pbuf_get_index(fieldname,ierr)
   459)     if (m > 0) then
   460)        call infld(fieldname, fh_ini, dim1name, 'lev', 'lat', 1, pcols, 1, pver, begchunk, endchunk, &
   461)             tptr3d, found, grid_map='PHYS')
   462)        if(.not. found) then
   463)           call infld('Q',fh_ini,dim1name, 'lev', 'lat', 1, pcols, 1, pver, begchunk, endchunk, &
   464)                tptr3d, found, grid_map='PHYS')
   465)           if (found) then
   466)              if (masterproc) write(iulog,*) trim(fieldname), ' initialized with Q'
   467)              if(dycore_is('LR')) call polar_average(pver, tptr3d) 	
   468)           else
   469)              call endrun('  '//trim(subname)//' Error:  Q must be on Initial File')
   470)           end if
   471)        end if
   472)        do n = 1, dyn_time_lvls
   473)           call pbuf_set_field(pbuf2d, m, tptr3d, (/1,1,n/),(/pcols,pver,1/))
   474)        end do
   475)     end if
   476) 
   477)     fieldname = 'ICCWAT'
   478)     m = pbuf_get_index(fieldname, ierr)
   479)     if (m > 0) then
   480)        call infld(fieldname, fh_ini, dim1name, 'lev', 'lat', 1, pcols, 1, pver, begchunk, endchunk, &
   481)           tptr3d, found, grid_map='phys')
   482)        if(found) then
   483)           do n = 1, dyn_time_lvls

   863) subroutine phys_run1(phys_state, ztodt, phys_tend, pbuf2d,  cam_in, cam_out)
   864)     !----------------------------------------------------------------------- 
   865)     ! 
   866)     ! Purpose: 
   867)     ! First part of atmospheric physics package before updating of surface models
   868)     ! 
   869)     !-----------------------------------------------------------------------
   870)     use time_manager,   only: get_nstep
   871)     use cam_diagnostics,only: diag_allocate, diag_physvar_ic
   872)     use check_energy,   only: check_energy_gmean
   873) 
   874)     use physics_buffer,         only: physics_buffer_desc, pbuf_get_chunk, pbuf_allocate
   875) #if (defined BFB_CAM_SCAM_IOP )
   876)     use cam_history,    only: outfld
   877) #endif
   878)     use comsrf,         only: fsns, fsnt, flns, sgh30, flnt, landm, fsds
   879)     use abortutils,     only: endrun
   880) #if ( defined OFFLINE_DYN )
   881)      use metdata,       only: get_met_srf1
   882) #endif
   883)     !
   884)     ! Input arguments
   885)     !
   886)     real(r8), intent(in) :: ztodt            ! physics time step unless nstep=0
   887)     !
   888)     ! Input/Output arguments
   889)     !
   890)     type(physics_state), intent(inout), dimension(begchunk:endchunk) :: phys_state
   891)     type(physics_tend ), intent(inout), dimension(begchunk:endchunk) :: phys_tend
   892) 
   893)     type(physics_buffer_desc), pointer, dimension(:,:) :: pbuf2d
   894)     type(cam_in_t),                     dimension(begchunk:endchunk) :: cam_in
   895)     type(cam_out_t),                    dimension(begchunk:endchunk) :: cam_out
   896)     !-----------------------------------------------------------------------
   897)     !
   898)     !---------------------------Local workspace-----------------------------
   899)     !
   900)     integer :: c                                 ! indices
   901)     integer :: ncol                              ! number of columns
   902)     integer :: nstep                             ! current timestep number
   903) #if (! defined SPMD)
   904)     integer  :: mpicom = 0
   905) #endif
   906)     type(physics_buffer_desc), pointer :: phys_buffer_chunk(:)
   907) 
   908)     call t_startf ('physpkg_st1')
   909)     nstep = get_nstep()
   910) 
   911) #if ( defined OFFLINE_DYN )
   912)     !
   913)     ! if offline mode set SNOWH and TS for micro-phys
   914)     !
   915)     call get_met_srf1( cam_in )
   916) #endif
   917) 
   918)     ! The following initialization depends on the import state (cam_in)
   919)     ! being initialized.  This isn't true when cam_init is called, so need
   920)     ! to postpone this initialization to here.
   921)     if (nstep == 0 .and. phys_do_flux_avg()) call flux_avg_init(cam_in,  pbuf2d)
   922) 
   923)     ! Compute total energy of input state and previous output state
   924)     call t_startf ('chk_en_gmean')
   925)     call check_energy_gmean(phys_state, pbuf2d, ztodt, nstep)
   926)     call t_stopf ('chk_en_gmean')
   927) 
   928)     call t_stopf ('physpkg_st1')
   929) 
   930)     if ( adiabatic .or. ideal_phys )then
   931)        call t_startf ('bc_physics')
   932)        call phys_run1_adiabatic_or_ideal(ztodt, phys_state, phys_tend,  pbuf2d)
   933)        call t_stopf ('bc_physics')
   934)     else
   935)        call t_startf ('physpkg_st1')
   936) 
   937)        call pbuf_allocate(pbuf2d, 'physpkg')
   938)        call diag_allocate()
   939) 
   940)        !-----------------------------------------------------------------------
   941)        ! Advance time information
   942)        !-----------------------------------------------------------------------
   943) 
   944)        call phys_timestep_init( phys_state, cam_out, pbuf2d)
   945) 
   946)        call t_stopf ('physpkg_st1')
   947) 
   948) #ifdef TRACER_CHECK
   949)        call gmean_mass ('before tphysbc DRY', phys_state)
   950) #endif
   951) 
   952) 
   953)        !-----------------------------------------------------------------------
   954)        ! Tendency physics before flux coupler invocation
   955)        !-----------------------------------------------------------------------
   956)        !
   957) 
   958) #if (defined BFB_CAM_SCAM_IOP )
   959)        do c=begchunk, endchunk
   960)           call outfld('Tg',cam_in(c)%ts,pcols   ,c     )
   961)        end do
   962) #endif
   963) 
   964)        call t_barrierf('sync_bc_physics', mpicom)
   965)        call t_startf ('bc_physics')
   966)        call t_adj_detailf(+1)
   967) 
   968) !$OMP PARALLEL DO PRIVATE (C, phys_buffer_chunk)
   969)        do c=begchunk, endchunk
   970)           !
   971)           ! Output physics terms to IC file
   972)           !
   973)           phys_buffer_chunk => pbuf_get_chunk(pbuf2d, c)
   974) 
   975)           call t_startf ('diag_physvar_ic')
   976)           call diag_physvar_ic ( c,  phys_buffer_chunk, cam_out(c), cam_in(c) )
   977)           call t_stopf ('diag_physvar_ic')
   978) 
   979)           call tphysbc (ztodt, fsns(1,c), fsnt(1,c), flns(1,c), flnt(1,c), phys_state(c),        &
   980)                        phys_tend(c), phys_buffer_chunk,  fsds(1,c), landm(1,c),          &
   981)                        sgh30(1,c), cam_out(c), cam_in(c) )
   982) 
   983)        end do

  1222) subroutine tphysac (ztodt,   cam_in,  &

  1230)     ! Computes the following:
  1231)     !
  1232)     !   o Aerosol Emission at Surface
  1233)     !   o Source-Sink for Advected Tracers
  1234)     !   o Symmetric Turbulence Scheme - Vertical Diffusion
  1235)     !   o Rayleigh Friction

  1257)     use physconst,          only: rhoh2o, latvap,latice
  1258)     use aero_model,         only: aero_model_drydep
  1259)     use carma_intr,         only: carma_emission_tend, carma_timestep_tend
  1260)     use carma_flags_mod,    only: carma_do_aerosol, carma_do_emission
  1261)     use check_energy,       only: check_energy_chng
  1262)     use check_energy,       only: check_tracers_data, check_tracers_init, check_tracers_chng
  1263)     use time_manager,       only: get_nstep
  1264)     use abortutils,         only: endrun
  1265)     use dycore,             only: dycore_is
  1266)     use cam_control_mod,    only: aqua_planet

  1270)     use qbo,                only: qbo_relax
  1271)     use iondrag,            only: iondrag_calc, do_waccm_ions
  1272)     use clubb_intr,         only: clubb_surface
  1273)     use perf_mod
  1274)     use flux_avg,           only: flux_avg_run
  1275)     use unicon_cam,         only: unicon_cam_org_diags
  1276) 
  1277)     !
  1278)     ! Arguments
  1279)     !
  1280)     real(r8), intent(in) :: ztodt                  ! Two times model timestep (2 delta-t)
  1281)     real(r8), intent(in) :: fsds(pcols)            ! down solar flux
  1282)     real(r8), intent(in) :: sgh(pcols)             ! Std. deviation of orography for gwd
  1283)     real(r8), intent(in) :: sgh30(pcols)           ! Std. deviation of 30s orography for tms
  1284) 
  1285)     type(cam_in_t),      intent(inout) :: cam_in
  1286)     type(cam_out_t),     intent(inout) :: cam_out
  1287)     type(physics_state), intent(inout) :: state
  1288)     type(physics_tend ), intent(inout) :: tend
  1289)     type(physics_buffer_desc), pointer :: pbuf(:)
  1290) 
  1291) 
  1292)     type(check_tracers_data):: tracerint             ! tracer mass integrals and cummulative boundary fluxes
  1293) 
  1294)     !
  1295)     !---------------------------Local workspace-----------------------------
  1296)     !
  1297)     type(physics_ptend)     :: ptend               ! indivdual parameterization tendencies
  1298) 
  1299)     integer  :: nstep                              ! current timestep number
  1300)     real(r8) :: zero(pcols)                        ! array of zeros
  1301) 
  1302)     integer :: lchnk                                ! chunk identifier
  1303)     integer :: ncol                                 ! number of atmospheric columns
  1304)     integer i,k,m                 ! Longitude, level indices
  1305)     integer :: yr, mon, day, tod       ! components of a date
  1306)     integer :: ixcldice, ixcldliq      ! constituent indices for cloud liquid and ice water.
  1307) 
  1308)     logical :: labort                            ! abort flag
  1309) 
  1310)     real(r8) tvm(pcols,pver)           ! virtual temperature
  1311)     real(r8) prect(pcols)              ! total precipitation
  1312)     real(r8) surfric(pcols)            ! surface friction velocity
  1313)     real(r8) obklen(pcols)             ! Obukhov length
  1314)     real(r8) :: fh2o(pcols)            ! h2o flux to balance source from methane chemistry
  1315)     real(r8) :: tmp_q     (pcols,pver) ! tmp space
  1316)     real(r8) :: tmp_cldliq(pcols,pver) ! tmp space
  1317)     real(r8) :: tmp_cldice(pcols,pver) ! tmp space
  1318)     real(r8) :: tmp_t     (pcols,pver) ! tmp space

  1321)     integer itim_old, ifld
  1322) 
  1323)     real(r8), pointer, dimension(:,:) :: tini
  1324)     real(r8), pointer, dimension(:,:) :: cld
  1325)     real(r8), pointer, dimension(:,:) :: qini
  1326)     real(r8), pointer, dimension(:,:) :: cldliqini
  1327)     real(r8), pointer, dimension(:,:) :: cldiceini

  1329)     real(r8), pointer, dimension(:,:) :: ast     ! relative humidity cloud fraction

  1332)     !
  1333)     lchnk = state%lchnk
  1334)     ncol  = state%ncol
  1335) 
  1336)     nstep = get_nstep()
  1337)     
  1338)     ! Adjust the surface fluxes to reduce instabilities in near sfc layer
  1339)     if (phys_do_flux_avg()) then 
  1340)        call flux_avg_run(state, cam_in,  pbuf, nstep, ztodt)
  1341)     endif
  1342) 
  1343)     ! Validate the physics state.
  1344)     if (state_debug_checks) &

  1365)

  1368)     ! jrm Include latent heat of fusion for snow
  1369)     !
  1370)     do i=1,ncol
  1371)        tend%flx_net(i) = tend%flx_net(i) + cam_in%shf(i) + (cam_out%precc(i) &
  1372)             + cam_out%precl(i))*latvap*rhoh2o &
  1373)             + (cam_out%precsc(i) + cam_out%precsl(i))*latice*rhoh2o
  1374)     end do
  1375) 
  1376)     ! emissions of aerosols and gas-phase chemistry constituents at surface
  1377)     call chem_emissions( state, cam_in )
  1378) 
  1379)     if (carma_do_emission) then
  1380)        ! carma emissions
  1381)        call carma_emission_tend (state, ptend, cam_in, ztodt)
  1382)        call physics_update(state, ptend, ztodt, tend)
  1383)     end if
  1384) 
  1385)     ! get nstep and zero array for energy checker
  1386)     zero = 0._r8
  1387)     nstep = get_nstep()
  1388)     call check_tracers_init(state, tracerint)
  1389) 
  1390)     ! Check if latent heat flux exceeds the total moisture content of the
  1391)     ! lowest model layer, thereby creating negative moisture.
  1392) 
  1393)     call qneg4('TPHYSAC '       ,lchnk               ,ncol  ,ztodt ,               &
  1394)          state%q(1,pver,1),state%rpdel(1,pver) ,cam_in%shf ,         &
  1395)          cam_in%lhf , cam_in%cflx )
  1396) 
  1397)     call t_stopf('tphysac_init')
  1398)     !===================================================
  1399)     ! Source/sink terms for advected tracers.
  1400)     !===================================================
  1401)     call t_startf('adv_tracer_src_snk')

  1404)     call tracers_timestep_tend(state, ptend, cam_in%cflx, cam_in%landfrac, ztodt)      
  1405)     call physics_update(state, ptend, ztodt, tend)
  1406)     call check_tracers_chng(state, tracerint, "tracers_timestep_tend", nstep, ztodt,   &

  1408)

  1410)     call physics_update(state, ptend, ztodt, tend)

  1418)     ! elevated emission of aerosol species are treated in 'chem_timestep_tend' before
  1419)     ! Gas chemistry and MAM core aerosol conversion.

  1430)             cam_in%cflx)

  1437)     !===================================================
  1438)

  1440)     !   surface friction velocity still need to be computed.  In addition, 
  1441)     !   surface fluxes need to be updated here for constituents 
  1442)     if (do_clubb_sgs) then
  1443) 
  1444)        call clubb_surface ( state, ptend, ztodt, cam_in, surfric, obklen)
  1445)        
  1446)        ! Update surface flux constituents 
  1447)        call physics_update(state, ptend, ztodt, tend)
  1448) 
  1449)     else
  1450) 
  1451)        call t_startf('vertical_diffusion_tend')
  1452)        call vertical_diffusion_tend (ztodt ,state ,cam_in%wsx, cam_in%wsy,   &
  1453)             cam_in%shf     ,cam_in%cflx     ,surfric  ,obklen   ,ptend    ,ast    ,&
  1454)             cam_in%ocnfrac  , cam_in%landfrac ,        &
  1455)             sgh30    ,pbuf )
  1456) 
  1457)     !------------------------------------------
  1458)     ! Call major diffusion for extended model
  1459)     !------------------------------------------
  1460)     if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then
  1461)        call mspd_intr (ztodt    ,state    ,ptend)
  1462)     endif
  1463) 
  1464)        call physics_update(state, ptend, ztodt, tend)
  1465)        call t_stopf ('vertical_diffusion_tend')
  1466)     
  1467)     endif

  1471)     ! Rayleigh friction calculation
  1472)     !===================================================

  1475)     call physics_update(state, ptend, ztodt, tend)
  1476)     call t_stopf('rayleigh_friction')
  1477) 
  1478)     if (do_clubb_sgs) then
  1479)       call check_energy_chng(state, tend, "vdiff", nstep, ztodt, zero, zero, zero, zero)
  1480)     else
  1481)       call check_energy_chng(state, tend, "vdiff", nstep, ztodt, cam_in%cflx(:,1), zero, &
  1482)            zero, cam_in%shf)
  1483)     endif
  1484)     
  1485)     call check_tracers_chng(state, tracerint, "vdiff", nstep, ztodt, cam_in%cflx)
  1486) 
  1487)     !  aerosol dry deposition processes
  1488)     call t_startf('aero_drydep')
  1489)     call aero_model_drydep( state, pbuf, obklen, surfric, cam_in, ztodt, cam_out, ptend )
  1490)     call physics_update(state, ptend, ztodt, tend)
  1491)     call t_stopf('aero_drydep')

  1493)    ! CARMA microphysics
  1494)    !
  1495)    ! NOTE: This does both the timestep_tend for CARMA aerosols as well as doing the dry
  1496)    ! deposition for CARMA aerosols. It needs to follow vertical_diffusion_tend, so that
  1497)    ! obklen and surfric have been calculated. It needs to follow aero_model_drydep, so
  1498)    ! that cam_out%xxxdryxxx fields have already been set for CAM aerosols and cam_out
  1499)    ! can be added to for CARMA aerosols.
  1500)    if (carma_do_aerosol) then
  1501)      call t_startf('carma_timestep_tend')
  1502)      call carma_timestep_tend(state, cam_in, cam_out, ptend, ztodt, pbuf, obklen=obklen, ustar=surfric)
  1503)      call physics_update(state, ptend, ztodt, tend)
  1504)    
  1505)      call check_energy_chng(state, tend, "carma_tend", nstep, ztodt, zero, zero, zero, zero)
  1506)      call t_stopf('carma_timestep_tend')
  1507)    end if
  1508) 
  1509) 
  1510)     !---------------------------------------------------------------------------------
  1511)     !	... enforce charge neutrality
  1512)     !---------------------------------------------------------------------------------
  1513)     call charge_fix( ncol, state%q(:,:,:) )
  1514) 
  1515)     !===================================================
  1516)     ! Gravity wave drag
  1517)     !===================================================
  1518)     call t_startf('gw_tend')
  1519) 
  1520)     call gw_tend(state, sgh, pbuf, ztodt, ptend, cam_in)
  1521) 
  1522)     call physics_update(state, ptend, ztodt, tend)
  1523)     ! Check energy integrals
  1524)     call check_energy_chng(state, tend, "gwdrag", nstep, ztodt, zero, zero, zero, zero)
  1525)     call t_stopf('gw_tend')
  1526) 
  1527)     ! QBO relaxation
  1528)     call qbo_relax(state, pbuf, ptend)
  1529)     call physics_update(state, ptend, ztodt, tend)
  1530)     ! Check energy integrals
  1531)     call check_energy_chng(state, tend, "qborelax", nstep, ztodt, zero, zero, zero, zero)
  1532) 
  1533)     ! Ion drag calculation
  1534)     call t_startf ( 'iondrag' )
  1535) 
  1536)     if ( do_waccm_ions ) then
  1537)        call iondrag_calc( lchnk, ncol, state, ptend, pbuf,  ztodt )
  1538)     else
  1539)        call iondrag_calc( lchnk, ncol, state, ptend)
  1540)     endif
  1541)     !----------------------------------------------------------------------------
  1542)     ! Call ionosphere routines for extended model if mode is set to ionosphere
  1543)     !----------------------------------------------------------------------------
  1544)     if( waccmx_is('ionosphere') ) then
  1545)        call ionos_intr(state, ptend, pbuf, ztodt)
  1546)     endif
  1547) 
  1548)     call physics_update(state, ptend, ztodt, tend)
  1549)     ! Check energy integrals
  1550)     call check_energy_chng(state, tend, "iondrag", nstep, ztodt, zero, zero, zero, zero)
  1551)     call t_stopf  ( 'iondrag' )
  1552) 
  1553) 
  1554)     !-------------- Energy budget checks vvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvv
  1555) 
  1556)     call pbuf_set_field(pbuf, teout_idx, state%te_cur, (/1,itim_old/),(/pcols,1/))       
  1557) 
  1558)     if (shallow_scheme .eq. 'UNICON') then
  1559) 
  1560)        ! ------------------------------------------------------------------------
  1561)        ! Insert the organization-related heterogeneities computed inside the
  1562)        ! UNICON into the tracer arrays here before performing advection.
  1563)        ! This is necessary to prevent any modifications of organization-related
  1564)        ! heterogeneities by non convection-advection process, such as
  1565)        ! dry and wet deposition of aerosols, MAM, etc.
  1566)        ! Again, note that only UNICON and advection schemes are allowed to
  1567)        ! changes to organization at this stage, although we can include the
  1568)        ! effects of other physical processes in future.
  1569)        ! ------------------------------------------------------------------------
  1570) 
  1571)        call unicon_cam_org_diags(state, pbuf)
  1572) 
  1573)     end if
  1574) 
  1575)     !*** BAB's FV heating kludge *** apply the heating as temperature tendency.
  1576)     !*** BAB's FV heating kludge *** modify the temperature in the state structure
  1577)     tmp_t(:ncol,:pver) = state%t(:ncol,:pver)
  1578)     state%t(:ncol,:pver) = tini(:ncol,:pver) + ztodt*tend%dtdt(:ncol,:pver)
  1579) 
  1580)     ! store dse after tphysac in buffer
  1581)     do k = 1,pver
  1582)        dtcore(:ncol,k) = state%t(:ncol,k)
  1583)     end do
  1584) 
  1585)     !
  1586)     ! FV: convert dry-type mixing ratios to moist here because physics_dme_adjust
  1587)     !     assumes moist. This is done in p_d_coupling for other dynamics. Bundy, Feb 2004.
  1588) 
  1589) 
  1590)     if ( dycore_is('LR') .or. dycore_is('SE')) call set_dry_to_wet(state)    ! Physics had dry, dynamics wants moist
  1591) 
  1592) 
  1593)     ! Scale dry mass and energy (does nothing if dycore is EUL or SLD)
  1594)     call cnst_get_ind('CLDLIQ', ixcldliq)
  1595)     call cnst_get_ind('CLDICE', ixcldice)
  1596)     tmp_q     (:ncol,:pver) = state%q(:ncol,:pver,1)
  1597)     tmp_cldliq(:ncol,:pver) = state%q(:ncol,:pver,ixcldliq)
  1598)     tmp_cldice(:ncol,:pver) = state%q(:ncol,:pver,ixcldice)
  1599)     call physics_dme_adjust(state, tend, qini, ztodt)
  1600) !!!   REMOVE THIS CALL, SINCE ONLY Q IS BEING ADJUSTED. WON'T BALANCE ENERGY. TE IS SAVED BEFORE THIS
  1601) !!!   call check_energy_chng(state, tend, "drymass", nstep, ztodt, zero, zero, zero, zero)
  1602) 
  1603)     !-------------- Energy budget checks ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
  1604)     if (aqua_planet) then
  1605)        labort = .false.
  1606)        do i=1,ncol
  1607)           if (cam_in%ocnfrac(i) /= 1._r8) labort = .true.
  1608)        end do
  1609)        if (labort) then
  1610)           call endrun ('TPHYSAC error:  grid contains non-ocean point')
  1611)        endif
  1612)     endif
  1613) 
  1614) 
  1615)     call diag_phys_tend_writeout (state, pbuf,  tend, ztodt, tmp_q, tmp_cldliq, tmp_cldice, &
  1616)          tmp_t, qini, cldliqini, cldiceini)
  1617) 
  1618)     call clybry_fam_set( ncol, lchnk, map2chm, state%q, pbuf )
  1619) 
  1620) end subroutine tphysac

  1622) subroutine tphysbc (ztodt,               &
  1623)        fsns,    fsnt,    flns,    flnt,    state,   &
  1624)        tend,    pbuf,     fsds,    landm,            &
  1625)        sgh30, cam_out, cam_in )
  1626)     !----------------------------------------------------------------------- 
  1627)     ! 
  1628)     ! Purpose:

  1860)

  2069) 	  !    to account for it in the energy checker
  2070)           flx_cnd(:ncol) = -1._r8*rliq(:ncol)

  2078)        endif 
  2079) 
  2080)        call t_stopf('macrop_tend') 
  2081) 
  2082)        !===================================================
  2083)        ! Calculate cloud microphysics 
  2084)        !===================================================
  2085) 
  2086)        if (is_subcol_on()) then

  2103)

  2133)        call t_stopf('microp_tend')
  2134) 
  2135)     endif
  2136)

  2139)        prec_sed(:ncol) = prec_sed(:ncol) + prec_sed_carma(:ncol)
  2140)        snow_sed(:ncol) = snow_sed(:ncol) + snow_sed_carma(:ncol)

  2145)        ! -------------------------------------------------------------------------------

  2161)        call aero_model_wetdep( state, ztodt, dlf, cam_out, ptend, pbuf)

  2163) 
  2164) 
  2165)        if (carma_do_wetdep) then

  2167)           !
  2168)           ! NOTE: It needs to follow aero_model_wetdep, so that cam_out%xxxwetxxx
  2169)           ! fields have already been set for CAM aerosols and cam_out can be added

  2179)        call physics_update(state, ptend, ztodt, tend)
  2180)        call t_stopf ('convect_deep_tend2')
  2181) 
  2182)        ! check tracer integrals
  2183)        call check_tracers_chng(state, tracerint, "cmfmca", nstep, ztodt,  zero_tracers)
  2184) 
  2185)        call t_stopf('bc_aerosols')
  2186) 
  2187)    endif
  2188) 
  2189)     !===================================================
  2190)     ! Moist physical parameteriztions complete: 
  2191)     ! send dynamical variables, and derived variables to history file
  2192)     !===================================================
  2193) 
  2194)     call t_startf('bc_history_write')
  2195)     call diag_phys_writeout(state, cam_out%psl)
  2196)     call diag_conv(state, ztodt, pbuf)
  2197) 
  2198)     call t_stopf('bc_history_write')
  2199) 
  2200)     !===================================================
  2201)     ! Write cloud diagnostics on history file
  2202)     !===================================================
  2203) 
  2204)     call t_startf('bc_cld_diag_history_write')
  2205) 
  2206)     call cloud_diagnostics_calc(state, pbuf)
  2207) 
  2208)     call t_stopf('bc_cld_diag_history_write')
  2209) 
  2210)     !===================================================
  2211)     ! Radiation computations
  2212)     !===================================================
  2213)     call t_startf('radiation')
  2214) 
  2215) 
  2216)     call radiation_tend(state,ptend, pbuf, &
  2217)          cam_out, cam_in, &
  2218)          cam_in%landfrac, cam_in%icefrac, cam_in%snowhland, &
  2219)          fsns,    fsnt, flns,    flnt,  &
  2220)          fsds, net_flx)
  2221) 
  2222)     ! Set net flux used by spectral dycores
  2223)     do i=1,ncol
  2224)        tend%flx_net(i) = net_flx(i)
  2225)     end do
  2226)     call physics_update(state, ptend, ztodt, tend)
  2227)     call check_energy_chng(state, tend, "radheat", nstep, ztodt, zero, zero, zero, net_flx)
  2228) 
  2229)     call t_stopf('radiation')
  2230) 
  2231)     ! Diagnose the location of the tropopause and its location to the history file(s).
  2232)     call t_startf('tropopause')
  2233)     call tropopause_output(state)
  2234)     call t_stopf('tropopause')
  2235) 
  2236)     ! Save atmospheric fields to force surface models
  2237)     call t_startf('cam_export')
  2238)     call cam_export (state,cam_out,pbuf)
  2239)     call t_stopf('cam_export')
  2240) 
  2241)     ! Write export state to history file
  2242)     call t_startf('diag_export')
  2243)     call diag_export(cam_out)
  2244)     call t_stopf('diag_export')
  2245)
