     1) ! This module is used to diagnose the location of the tropopause. Multiple
     2) ! algorithms are provided, some of which may not be able to identify a
     3) ! tropopause in all situations. To handle these cases, an analytic
     4) ! definition and a climatology are provided that can be used to fill in
     5) ! when the original algorithm fails. The tropopause temperature and
     6) ! pressure are determined and can be output to the history file.
     7) !
     8) ! These routines are based upon code in the WACCM chemistry module
     9) ! including mo_tropoause.F90 and llnl_set_chem_trop.F90. The code
    10) ! for the Reichler et al. [2003] algorithm is from:
    11) !
    12) !   http://www.gfdl.noaa.gov/~tjr/TROPO/tropocode.htm
    13) !
    14) ! Author: Charles Bardeen
    15) ! Created: April, 2009

   209)   subroutine tropopause_read_file
   210)     !------------------------------------------------------------------
   211)     ! ... initialize upper boundary values
   212)     !------------------------------------------------------------------
   213)     use interpolate_data,  only : lininterp_init, lininterp, interp_type, lininterp_finish
   214)     use dyn_grid,     only : get_dyn_grid_parm
   215)     use phys_grid,    only : get_ncols_p, get_rlat_all_p, get_rlon_all_p	
   216)     use ioFileMod,    only : getfil
   217)     use time_manager, only : get_calday
   218)     use physconst,    only : pi
   219)     use cam_pio_utils, only: cam_pio_openfile
   220)     use pio,          only : file_desc_t, var_desc_t, pio_inq_dimid, pio_inq_dimlen, &
   221)          pio_inq_varid, pio_get_var, pio_closefile, pio_nowrite
   222) 
   223)     !------------------------------------------------------------------
   224)     ! ... local variables
   225)     !------------------------------------------------------------------
   226)     integer :: i, j, n
   227)     integer :: ierr
   228)     type(file_desc_t) :: pio_id
   229)     integer :: dimid
   230)     type(var_desc_t) :: vid
   231)     integer :: nlon, nlat, ntimes
   232)     integer :: start(3)
   233)     integer :: count(3)
   234)     integer, parameter :: dates(12) = (/ 116, 214, 316, 415,  516,  615, &
   235)          716, 816, 915, 1016, 1115, 1216 /)
   236)     integer :: plon, plat
   237)     type(interp_type) :: lon_wgts, lat_wgts
   238)     real(r8), allocatable :: tropp_p_in(:,:,:)
   239)     real(r8), allocatable :: lat(:)
   240)     real(r8), allocatable :: lon(:)
   241)     real(r8) :: to_lats(pcols), to_lons(pcols)
   242)     real(r8), parameter :: d2r=pi/180._r8, zero=0._r8, twopi=pi*2._r8
   243)     character(len=256) :: locfn
   244)     integer  :: c, ncols
   245) 
   246) 
   247)     plon = get_dyn_grid_parm('plon')
   248)     plat = get_dyn_grid_parm('plat')
   249) 
   250) 
   251)     !-----------------------------------------------------------------------
   252)     !       ... open netcdf file
   253)     !-----------------------------------------------------------------------
   254)     call getfil (tropopause_climo_file, locfn, 0)
   255)     call cam_pio_openfile(pio_id, trim(locfn), PIO_NOWRITE)
   256) 
   257)     !-----------------------------------------------------------------------
   258)     !       ... get time dimension
   259)     !-----------------------------------------------------------------------
   260)     ierr = pio_inq_dimid( pio_id, 'time', dimid )
   261)     ierr = pio_inq_dimlen( pio_id, dimid, ntimes )
   262)     if( ntimes /= 12 )then
   263)        write(iulog,*) 'tropopause_init: number of months = ',ntimes,'; expecting 12'
   264)        call endrun
   265)     end if
   266)     !-----------------------------------------------------------------------
   267)     !       ... get latitudes
   268)     !-----------------------------------------------------------------------
   269)     ierr = pio_inq_dimid( pio_id, 'lat', dimid )
   270)     ierr = pio_inq_dimlen( pio_id, dimid, nlat )
   271)     allocate( lat(nlat), stat=ierr )
   272)     if( ierr /= 0 ) then
   273)        write(iulog,*) 'tropopause_init: lat allocation error = ',ierr
   274)        call endrun
   275)     end if
   276)     ierr = pio_inq_varid( pio_id, 'lat', vid )
   277)     ierr = pio_get_var( pio_id, vid, lat )
   278)     lat(:nlat) = lat(:nlat) * d2r
   279)     !-----------------------------------------------------------------------
   280)     !       ... get longitudes
   281)     !-----------------------------------------------------------------------
   282)     ierr = pio_inq_dimid( pio_id, 'lon', dimid )
   283)     ierr = pio_inq_dimlen( pio_id, dimid, nlon )
   284)     allocate( lon(nlon), stat=ierr )
   285)     if( ierr /= 0 ) then
   286)        write(iulog,*) 'tropopause_init: lon allocation error = ',ierr
   287)        call endrun
   288)     end if
   289)     ierr = pio_inq_varid( pio_id, 'lon', vid )
   290)     ierr = pio_get_var( pio_id, vid, lon )
   291)     lon(:nlon) = lon(:nlon) * d2r
   292) 
   293)     !------------------------------------------------------------------
   294)     !  ... allocate arrays
   295)     !------------------------------------------------------------------
   296)     allocate( tropp_p_in(nlon,nlat,ntimes), stat=ierr )
   297)     if( ierr /= 0 ) then
   298)        write(iulog,*) 'tropopause_init: tropp_p_in allocation error = ',ierr
   299)        call endrun
   300)     end if
   301)     !------------------------------------------------------------------
   302)     !  ... read in the tropopause pressure
   303)     !------------------------------------------------------------------
   304)     ierr = pio_inq_varid( pio_id, 'trop_p', vid )
   305)     start = (/ 1, 1, 1 /)
   306)     count = (/ nlon, nlat, ntimes /)
   307)     ierr = pio_get_var( pio_id, vid, start, count, tropp_p_in )
   308) 
   309)     !------------------------------------------------------------------
   310)     !  ... close the netcdf file
   311)     !------------------------------------------------------------------
   312)     call pio_closefile( pio_id )
   313) 
   314)     !--------------------------------------------------------------------
   315)     !  ... regrid
   316)     !--------------------------------------------------------------------
   317) 
   318)     allocate( tropp_p_loc(pcols,begchunk:endchunk,ntimes), stat=ierr )
   319) 
   320)     if( ierr /= 0 ) then
   321)       write(iulog,*) 'tropopause_init: tropp_p_loc allocation error = ',ierr
   322)       call endrun
   323)     end if
   324) 
   325)     do c=begchunk,endchunk
   326)        ncols = get_ncols_p(c)
   327)        call get_rlat_all_p(c, pcols, to_lats)
   328)        call get_rlon_all_p(c, pcols, to_lons)
   329)        call lininterp_init(lon, nlon, to_lons, ncols, 2, lon_wgts, zero, twopi)
   330)        call lininterp_init(lat, nlat, to_lats, ncols, 1, lat_wgts)
   331)        do n=1,ntimes
   332)           call lininterp(tropp_p_in(:,:,n), nlon, nlat, tropp_p_loc(1:ncols,c,n), ncols, lon_wgts, lat_wgts)    
   333)        end do
   334)        call lininterp_finish(lon_wgts)
   335)        call lininterp_finish(lat_wgts)
   336)     end do
   337)     deallocate(lon)
   338)     deallocate(lat)
   339)     deallocate(tropp_p_in)
   340) 
   341)     !--------------------------------------------------------
   342)     ! ... initialize the monthly day of year times
   343)     !--------------------------------------------------------
   344) 
   345)     do n = 1,12
   346)        days(n) = get_calday( dates(n), 0 )
   347)     end do
   348)     if (masterproc) then
   349)        write(iulog,*) 'tropopause_init : days'
   350)        write(iulog,'(1p,5g15.8)') days(:)
   351)     endif
   352) 
   353)   end subroutine tropopause_read_file

   356)   ! This analytic expression closely matches the mean tropopause determined
   357)   ! by the NCEP reanalysis and has been used by the radiation code.
   358)   subroutine tropopause_analytic(pstate, tropLev, tropP, tropT, tropZ)
   359) 
   360)     implicit none
   361) 
   362)     type(physics_state), intent(in)     :: pstate
   363)     integer,            intent(inout)   :: tropLev(pcols)             ! tropopause level index   
   364)     real(r8), optional, intent(inout)   :: tropP(pcols)               ! tropopause pressure (Pa)  
   365)     real(r8), optional, intent(inout)   :: tropT(pcols)               ! tropopause temperature (K)
   366)     real(r8), optional, intent(inout)   :: tropZ(pcols)               ! tropopause height (m)
   367)     
   368)     ! Local Variables
   369)     integer       :: i
   370)     integer       :: k
   371)     integer       :: ncol                     ! number of columns in the chunk
   372)     integer       :: lchnk                    ! chunk identifier
   373)     real(r8)      :: tP                       ! tropopause pressure (Pa)
   374)  
   375)     ! Information about the chunk.  
   376)     lchnk = pstate%lchnk
   377)     ncol  = pstate%ncol
   378) 
   379)     ! Iterate over all of the columns.
   380)     do i = 1, ncol
   381)      
   382)       ! Skip column in which the tropopause has already been found.
   383)       if (tropLev(i) == NOTFOUND) then
   384)       
   385)         ! Calculate the pressure of the tropopause.
   386)         tP = (25000.0_r8 - 15000.0_r8 * (cos(pstate%lat(i)))**2)
   387)       
   388)         ! Find the level that contains the tropopause.
   389)         do k = pver, 2, -1
   390)           if (tP >= pstate%pint(i, k)) then
   391)             tropLev(i) = k
   392)             exit
   393)           end if
   394)         end do
   395)         
   396)         ! Return the optional outputs
   397)         if (present(tropP)) tropP(i) = tP
   398)         
   399)         if (present(tropT)) then
   400)           tropT(i) = tropopause_interpolateT(pstate, i, tropLev(i), tP)
   401)         end if
   402) 
   403)         if (present(tropZ)) then
   404)           tropZ(i) = tropopause_interpolateZ(pstate, i, tropLev(i), tP)
   405)         end if
   406)       end if
   407)     end do
   408)   end subroutine tropopause_analytic

   686)   ! This routine is an implementation of Reichler et al. [2003] done by
   687)   ! Reichler and downloaded from his web site. Minimal modifications were

   689)   ! CAM constants and types).
   690)   !
   691)   ! NOTE: I am not a big fan of the goto's and multiple returns in this
   692)   ! code, but for the moment I have left them to preserve as much of the
   693)   ! original and presumably well tested code as possible.
   694)   ! UPDATE: The most "obvious" substitutions have been made to replace
   695)   ! goto/return statements with cycle/exit. The structure is still
   696)   ! somewhat tangled.
   697)   ! UPDATE 2: "gamma" renamed to "gam" in order to avoid confusion
   698)   ! with the Fortran 2008 intrinsic. "level" argument removed because
   699)   ! a physics column is not contiguous, so using explicit dimensions
   700)   ! will cause the data to be needlessly copied.
   701)   !
   702)   !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   703)   !
   704)   ! determination of tropopause height from gridded temperature data
   705)   !
   706)   ! reference: Reichler, T., M. Dameris, and R. Sausen (2003)
   707)   !
   708)   !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   709)   subroutine twmo(t, p, plimu, pliml, gam, trp)
   710) 
   711)     real(r8), intent(in), dimension(:)      :: t, p
   712)     real(r8), intent(in)                    :: plimu, pliml, gam
   713)     real(r8), intent(out)                   :: trp
   714)     
   715)     real(r8), parameter                     :: deltaz = 2000.0_r8
   716) 
   717)     real(r8)                                :: pmk, pm, a, b, tm, dtdp, dtdz
   718)     real(r8)                                :: ag, bg, ptph
   719)     real(r8)                                :: pm0, pmk0, dtdz0
   720)     real(r8)                                :: p2km, asum, aquer
   721)     real(r8)                                :: pmk2, pm2, a2, b2, tm2, dtdp2, dtdz2
   722)     integer                                 :: level
   723)     integer                                 :: icount, jj
   724)     integer                                 :: j
   725)     
   726) 
   727)     trp=-99.0_r8                           ! negative means not valid
   728)     
   729)     ! initialize start level
   730)     ! dt/dz
   731)     level = size(t)
   732)     pmk= .5_r8 * (p(level-1)**cnst_kap+p(level)**cnst_kap)
   733)     pm = pmk**(1/cnst_kap)               
   734)     a = (t(level-1)-t(level))/(p(level-1)**cnst_kap-p(level)**cnst_kap)
   735)     b = t(level)-(a*p(level)**cnst_kap)
   736)     tm = a * pmk + b               
   737)     dtdp = a * cnst_kap * (pm**cnst_ka1)
   738)     dtdz = cnst_faktor*dtdp*pm/tm
   739) 
   740)     main_loop: do j=level-1,2,-1
   741)       pm0 = pm
   742)       pmk0 = pmk
   743)       dtdz0  = dtdz
   744)     
   745)       ! dt/dz
   746)       pmk= .5_r8 * (p(j-1)**cnst_kap+p(j)**cnst_kap)
   747)       pm = pmk**(1/cnst_kap)               
   748)       a = (t(j-1)-t(j))/(p(j-1)**cnst_kap-p(j)**cnst_kap)
   749)       b = t(j)-(a*p(j)**cnst_kap)
   750)       tm = a * pmk + b               
   751)       dtdp = a * cnst_kap * (pm**cnst_ka1)
   752)       dtdz = cnst_faktor*dtdp*pm/tm
   753)       ! dt/dz valid?
   754)       if (dtdz.le.gam) cycle main_loop    ! no, dt/dz < -2 K/km
   755)       if (pm.gt.plimu)   cycle main_loop    ! no, too low
   756)   
   757)       ! dtdz is valid, calculate tropopause pressure
   758)       if (dtdz0.lt.gam) then
   759)         ag = (dtdz-dtdz0) / (pmk-pmk0)     
   760)         bg = dtdz0 - (ag * pmk0)          
   761)         ptph = exp(log((gam-bg)/ag)/cnst_kap)
   762)       else
   763)         ptph = pm
   764)       endif
   765)   
   766)       if (ptph.lt.pliml) cycle main_loop
   767)       if (ptph.gt.plimu) cycle main_loop
   768)   
   769)       ! 2nd test: dtdz above 2 km must not exceed gam
   770)       p2km = ptph + deltaz*(pm/tm)*cnst_faktor     ! p at ptph + 2km
   771)       asum = 0.0_r8                                ! dtdz above
   772)       icount = 0                                   ! number of levels above
   773)   
   774)       ! test until apm < p2km
   775)       in_loop: do jj=j,2,-1
   776)     
   777)         pmk2 = .5_r8 * (p(jj-1)**cnst_kap+p(jj)**cnst_kap) ! p mean ^kappa
   778)         pm2 = pmk2**(1/cnst_kap)                           ! p mean
   779)         if(pm2.gt.ptph) cycle in_loop            ! doesn't happen
   780)         if(pm2.lt.p2km) exit in_loop             ! ptropo is valid
   781) 
   782)         a2 = (t(jj-1)-t(jj))                     ! a
   783)         a2 = a2/(p(jj-1)**cnst_kap-p(jj)**cnst_kap)
   784)         b2 = t(jj)-(a2*p(jj)**cnst_kap)          ! b
   785)         tm2 = a2 * pmk2 + b2                     ! T mean
   786)         dtdp2 = a2 * cnst_kap * (pm2**(cnst_kap-1))  ! dt/dp
   787)         dtdz2 = cnst_faktor*dtdp2*pm2/tm2
   788)         asum = asum+dtdz2
   789)         icount = icount+1
   790)         aquer = asum/float(icount)               ! dt/dz mean
   791)    
   792)         ! discard ptropo ?
   793)         if (aquer.le.gam) cycle main_loop      ! dt/dz above < gam
   794)     
   795)       enddo in_loop  ! test next level
   796)     
   797)       trp = ptph
   798)       exit main_loop
   799)     enddo main_loop
   800)     
   801)   end subroutine twmo

   870)   ! This requires that the lapse rate be less than 2 K/km for an altitude range
   871)   ! of 2 km. The search starts at the surface and stops the first time this
   872)   ! criteria is met.
   873)   !
   874)   ! NOTE: This code was modeled after the code in mo_tropopause; however, the
   875)   ! requirement that dt be greater than 0 was removed and the check to make
   876)   ! sure that the lapse rate is maintained for 2 km was added.
   877)   subroutine tropopause_wmo(pstate, tropLev, tropP, tropT, tropZ)
   878) 
   879)     implicit none
   880) 
   881)     type(physics_state), intent(in)    :: pstate 
   882)     integer,            intent(inout)  :: tropLev(pcols)            ! tropopause level index   
   883)     real(r8), optional, intent(inout)  :: tropP(pcols)              ! tropopause pressure (Pa)   
   884)     real(r8), optional, intent(inout)  :: tropT(pcols)              ! tropopause temperature (K)
   885)     real(r8), optional, intent(inout)  :: tropZ(pcols)              ! tropopause height (m)
   886)

   888)     real(r8), parameter    :: ztrop_low   = 5000._r8        ! lowest tropopause level allowed (m)
   889)     real(r8), parameter    :: ztrop_high  = 20000._r8       ! highest tropopause level allowed (m)
   890)     real(r8), parameter    :: max_dtdz    = 0.002_r8        ! max dt/dz for tropopause level (K/m)
   891)     real(r8), parameter    :: min_trop_dz = 2000._r8        ! min tropopause thickness (m)
   892)

   894)     integer                 :: k

   896)     integer                 :: ncol                         ! number of columns in the chunk
   897)     integer                 :: lchnk                        ! chunk identifier
   898)     real(r8)                :: tP                           ! tropopause pressure (Pa)
   899)     real(r8)                :: dt
   900)

   902)     lchnk = pstate%lchnk

   906)     do i = 1, ncol
   907)      
   908)       ! Skip column in which the tropopause has already been found.
   909)       if (tropLev(i) == NOTFOUND) then
   910) 
   911)         kloop: do k = pver-1, 2, -1
   912)          
   913)           ! Skip levels below the minimum and stop if nothing is found
   914)           ! before the maximum.
