   161)

   164) !  Restore values in cells surrounding the grid

   169)       this_block = get_block(blocks_ice(iblk),iblk)

   179)             do i = 1, ilo

   182)                vicen(i,j,n,iblk) = vicen(i,j,n,iblk) &
   183)                   + (vicen_rest(i,j,n,iblk)-vicen(i,j,n,iblk))*ctime
   184)                vsnon(i,j,n,iblk) = vsnon(i,j,n,iblk) &
   185)                   + (vsnon_rest(i,j,n,iblk)-vsnon(i,j,n,iblk))*ctime

   332)                   + (eicen_rest(i,j,n,iblk)-eicen(i,j,n,iblk))*ctime
