   296)    
   297)    ! pft loop
   298)    do fp = 1,num_soilp
   299)       p = filter_soilp(fp)
   300)       
   301)       ! initialize the pft-level C and N truncation terms
   302)       pc = 0._r8
   303)       if (use_c13) then
   304)          pc13 = 0._r8
   305)       end if
   306)       pn = 0._r8
   307)       
   308)       ! do tests on state variables for precision control
   309)       ! for linked C-N state variables, perform precision test on
   310)       ! the C component, but truncate C, C13, and N components
   311)       
   312)       ! leaf C and N
   313)       if (abs(leafc(p)) < ccrit) then
   314)           pc = pc + leafc(p)
   315)           leafc(p) = 0._r8
   316)           if (use_c13) then
   317)              pc13 = pc13 + c13_leafc(p)
   318)              c13_leafc(p) = 0._r8
   319)           endif
   320)           pn = pn + leafn(p)
   321)           leafn(p) = 0._r8
   322)       end if
   323) 
   324)       ! leaf storage C and N
   325)       if (abs(leafc_storage(p)) < ccrit) then
   326)           pc = pc + leafc_storage(p)
   327)           leafc_storage(p) = 0._r8
   328)           if (use_c13) then
   329)              pc13 = pc13 + c13_leafc_storage(p)
   330)              c13_leafc_storage(p) = 0._r8
   331)           endif
   332)           pn = pn + leafn_storage(p)
   333)           leafn_storage(p) = 0._r8
   334)       end if
   335)           
   336)       ! leaf transfer C and N
   337)       if (abs(leafc_xfer(p)) < ccrit) then
   338)           pc = pc + leafc_xfer(p)
   339)           leafc_xfer(p) = 0._r8
   340)           if (use_c13) then
   341)              pc13 = pc13 + c13_leafc_xfer(p)
   342)              c13_leafc_xfer(p) = 0._r8
   343)           endif
   344)           pn = pn + leafn_xfer(p)
   345)           leafn_xfer(p) = 0._r8
   346)       end if
   347)           
   348)       ! froot C and N
   349)       if (abs(frootc(p)) < ccrit) then
   350)           pc = pc + frootc(p)
   351)           frootc(p) = 0._r8
   352)           if (use_c13) then
   353)              pc13 = pc13 + c13_frootc(p)
   354)              c13_frootc(p) = 0._r8
   355)           endif
   356)           pn = pn + frootn(p)
   357)           frootn(p) = 0._r8
   358)       end if
   359) 
   360)       ! froot storage C and N
   361)       if (abs(frootc_storage(p)) < ccrit) then
   362)           pc = pc + frootc_storage(p)
   363)           frootc_storage(p) = 0._r8
   364)           if (use_c13) then
   365)              pc13 = pc13 + c13_frootc_storage(p)
   366)              c13_frootc_storage(p) = 0._r8
   367)           endif
   368)           pn = pn + frootn_storage(p)
   369)           frootn_storage(p) = 0._r8
   370)       end if
   371)           
   372)       ! froot transfer C and N
   373)       if (abs(frootc_xfer(p)) < ccrit) then
   374)           pc = pc + frootc_xfer(p)
   375)           frootc_xfer(p) = 0._r8
   376)           if (use_c13) then
   377)              pc13 = pc13 + c13_frootc_xfer(p)
   378)              c13_frootc_xfer(p) = 0._r8
   379)           endif
   380)           pn = pn + frootn_xfer(p)
   381)           frootn_xfer(p) = 0._r8
   382)       end if
   383)           
   384)       if ( crop_prog .and. ivt(p) >= nc3crop )then
   385)          ! grain C and N
   386)          if (abs(grainc(p)) < ccrit) then
   387)              pc = pc + grainc(p)
   388)              grainc(p) = 0._r8
   389)              pn = pn + grainn(p)
   390)              grainn(p) = 0._r8
   391)          end if
   392)    
   393)          ! grain storage C and N
