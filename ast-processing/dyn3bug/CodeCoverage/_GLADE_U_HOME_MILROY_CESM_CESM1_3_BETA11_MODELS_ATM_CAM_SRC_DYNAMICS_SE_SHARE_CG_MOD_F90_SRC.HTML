   164)     type (cg_t)                       :: cg      ! private cg structure
   165)     type (ReductionBuffer_ordered_1d_t)     :: red     ! shared data reduction data stucture

   172)

   174)     integer kptr

   183)     real (kind=real_kind) :: sigma

   193)        ! Assumes that r=rhs and
   194)        ! initial x1 = 0.0D0

   198) 
   199)        notdone=.true.

   203) 
   204)     else if (cg%iter==0) then

   206)        ! ==================================
   207)        ! assumes that z_1 = (M^-1) * r_1
   208)        ! assumes that v_1 = A * z_1
   209)        ! ==================================
   210) 
   211)        do k=1,cg%ninst
   212)           ! JPE This conditional should always be true - no need for it here
   213)           !          if (.not.cg%converged(k)) then
   214)           rhs_norm = 0.0D0
   215)           gamma    = 0.0D0
   216)           sigma    = 0.0D0
   217) 
   218)           do ie=1,cg%nblks
   219) 
   220)              global_shared_buf(ie,1:3) = 0.d0
   221)              do i=1,cg%len
   222)                 cg%state(ie)%p(i,k)     = cg%state(ie)%z(i,k) ! p_1 = z_1  r_1
   223)                 cg%state(ie)%v(i,k)     = cg%state(ie)%s(i,k) ! v_1 = s_1  LHS(r_1)
   224)                 global_shared_buf(ie,1) = global_shared_buf(ie,1) + &
   225)                      cg%wts(i,ie)*cg%state(ie)%z(i,k)*cg%state(ie)%r(i,k) ! <z_1,r_1>
   226)                 global_shared_buf(ie,2) = global_shared_buf(ie,2) + &
   227)                      cg%wts(i,ie)*cg%state(ie)%p(i,k)*cg%state(ie)%v(i,k) ! <p_1,v_1>
   228)                 global_shared_buf(ie,3) = global_shared_buf(ie,3) + &

   235)           red%buf(3*k-1,1) = global_shared_sum(2)
   236)           red%buf(3*k  ,1) = global_shared_sum(3)
   237) 
   238)           !          end if
   239)        end do
   240) 
   241)        kptr=0
   242)        do k=1,cg%ninst
   243)           ! JPE This conditional should always be true - no need for it here
   244)           !          if (.not.cg%converged(k)) then
   245)           gamma         = red%buf(3*kptr+1,1)     ! <z_1,r_1>
   246)           sigma         = red%buf(3*kptr+2,1)     ! <p_1,v_1>
   247)           cg%rhs_norm(k)= red%buf(3*kptr+3,1)     ! <b,b>
   248)           cg%gammam1(k) = gamma
   249)           cg%sigmam1(k) = sigma
   250) #if 1
   251)           eps(k)        = SQRT(cg%rhs_norm(k)/cg%rhs_norm(k))
   252) #else
   253)           eps(k) = ABS(gamma/cg%rhs_norm(k))

   272)           end if

   276)        !CG-JMD cg%wrk2 => cg%z         ! request z_2 = (M^-1) * r_2
   277)        !CG-JMD cg%wrk3 => cg%s         ! request s_2 = A*z_2

   314)        kptr=0

   323)              delta  = red%buf(3*kptr+2,1)

   326)              eps(k) = SQRT(l2_norm/cg%rhs_norm(k))
   327) #else
   328)              eps(k) = ABS(gamma/cg%rhs_norm(k))
   329) #endif             
   330)              beta = gamma/cg%gammam1(k)
   331)              sigma  = delta-beta*beta*cg%sigmam1(k)    ! eq(12) "method 1" of Lapack note 56
   332)              alpha  = gamma/sigma
   333)              cg%gammam1(k) = gamma
   334)              cg%sigmam1(k) = sigma
   335)              do ie=1,cg%nblks
   336)                 do i=1,cg%len
   337)                    cg%state(ie)%p(i,k)=cg%state(ie)%z(i,k) +  beta*cg%state(ie)%p(i,k)
   338)                    cg%state(ie)%v(i,k)=cg%state(ie)%s(i,k) +  beta*cg%state(ie)%v(i,k)

   340)                    cg%state(ie)%r(i,k)=cg%state(ie)%r(i,k) - alpha*cg%state(ie)%v(i,k)
   341)                 end do
   342)              end do
   343)              if (eps(k) < tol) cg%converged(k)=.true.
   344)              kptr=kptr+1
   345)           end if
   346)        end do
   347) 
   348)        if (cg%debug_level == 2) then
   349)           if (cg%hybrid%par%masterproc .and. cg%hybrid%ithr == 0) then
   350)              print *,"++++++++++++++++++++++++++++"
   351)              do k=1,SIZE(eps)
   352)                 print *,"iter = ",cg%iter,"eps(",k,")=",eps(k)," rhs_norm(",k,")=",cg%rhs_norm(k)
   353)              end do
   354)           end if
   355)        end if
   356) 
   357)        cg%iter=cg%iter+1
   358)        if (cg%iter==maxits .or. ALL(eps(:) < tol)) then
   359)           if (cg%debug_level == 1) then
   360)           if (cg%hybrid%par%masterproc .and. cg%hybrid%ithr == 0) then
   361)              print *
   362)              print *,"++++++++++++++++++++++++++++"
   363)              do k=1,SIZE(eps)
   364)                 print *,"iter = ",cg%iter,"eps(",k,")=",eps(k)," rhs_norm(",k,")=",cg%rhs_norm(k)
   365)              end do
   366)           end if
   367)           end if
   368) 
   369) 	  !=================================================================
   370) 	  ! hopefully this will provide a slightly more reliable branch test 
   371) 	  !=================================================================
   372)           cg%initalized=0
   373)           notdone=.false.
   374)        else
   375)           notdone=.true.
   376)        end if
   377)        return
