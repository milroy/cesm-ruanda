   128) SUBROUTINE normalize_basetime( basetime )
   129)   ! Factor so abs(Sn) < Sd and ensure that signs of S and Sn match.
   130)   ! Also, enforce consistency.
   131)   ! YR and MM fields are ignored.
   132)   IMPLICIT NONE
   133)   TYPE(ESMF_BaseTime), INTENT(INOUT) :: basetime
   134) 
   135)   !PRINT *,'DEBUG:  BEGIN normalize_basetime()'
   136)   ! Consistency check...
   137)   IF ( basetime%Sd < 0 ) THEN
   138)     CALL wrf_error_fatal( &
   139)       'normalize_basetime:  denominator of seconds cannot be negative' )
   140)   ENDIF
   141)   IF ( ( basetime%Sd == 0 ) .AND. ( basetime%Sn .NE. 0 ) ) THEN
   142)     CALL wrf_error_fatal( &
   143)       'normalize_basetime:  denominator of seconds cannot be zero when numerator is non-zero' )
   144)   ENDIF
   145)   ! factor so abs(Sn) < Sd
   146)   IF ( basetime%Sd > 0 ) THEN
   147)     IF ( ABS( basetime%Sn ) .GE. basetime%Sd ) THEN
   148)       !PRINT *,'DEBUG:  normalize_basetime() A1:  S,Sn,Sd = ',basetime%S,basetime%Sn,basetime%Sd
   149)       basetime%S = basetime%S + ( basetime%Sn / basetime%Sd )
   150)       basetime%Sn = mod( basetime%Sn, basetime%Sd )
   151)       !PRINT *,'DEBUG:  normalize_basetime() A2:  S,Sn,Sd = ',basetime%S,basetime%Sn,basetime%Sd
   152)     ENDIF
   153)     ! change sign of Sn if it does not match S
   154)     IF ( ( basetime%S > 0 ) .AND. ( basetime%Sn < 0 ) ) THEN
   155)       !PRINT *,'DEBUG:  normalize_basetime() B1:  S,Sn,Sd = ',basetime%S,basetime%Sn,basetime%Sd
   156)       basetime%S = basetime%S - 1_ESMF_KIND_I8
   157)       basetime%Sn = basetime%Sn + basetime%Sd
   158)       !PRINT *,'DEBUG:  normalize_basetime() B2:  S,Sn,Sd = ',basetime%S,basetime%Sn,basetime%Sd
   159)     ENDIF
   160)     IF ( ( basetime%S < 0 ) .AND. ( basetime%Sn > 0 ) ) THEN
   161)       !PRINT *,'DEBUG:  normalize_basetime() C1:  S,Sn,Sd = ',basetime%S,basetime%Sn,basetime%Sd
   162)       basetime%S = basetime%S + 1_ESMF_KIND_I8
   163)       basetime%Sn = basetime%Sn - basetime%Sd
   164)       !PRINT *,'DEBUG:  normalize_basetime() C2:  S,Sn,Sd = ',basetime%S,basetime%Sn,basetime%Sd
   165)     ENDIF
   166)   ENDIF
   167)   !PRINT *,'DEBUG:  END normalize_basetime()'
   168) END SUBROUTINE normalize_basetime

   193) !  PRINT *,'DEBUG:  ESMF_BaseTimeSum():  Sd1 = ',Sd1

   200)         ELSE IF ( ( Sd1 .NE. 0 ) .AND. ( Sd2 .EQ. 0 ) ) THEN
   201)           ESMF_BaseTimeSum%Sn = Sn1
   202)           ESMF_BaseTimeSum%Sd = Sd1
   203)         ELSE IF ( ( Sd1 .EQ. 0 ) .AND. ( Sd2 .NE. 0 ) ) THEN
   204)           ESMF_BaseTimeSum%Sn = Sn2
   205)           ESMF_BaseTimeSum%Sd = Sd2
   206)         ELSE IF ( ( Sd1 .NE. 0 ) .AND. ( Sd2 .NE. 0 ) ) THEN
   207)           CALL compute_lcd( Sd1 , Sd2 , lcd )
   208)           ESMF_BaseTimeSum%Sd = lcd

   211) !  PRINT *,'DEBUG:  ESMF_BaseTimeSum():  ESMF_BaseTimeSum%S = ',ESMF_BaseTimeSum%S
   212) !  PRINT *,'DEBUG:  ESMF_BaseTimeSum():  ESMF_BaseTimeSum%Sn = ',ESMF_BaseTimeSum%Sn
   213) !  PRINT *,'DEBUG:  ESMF_BaseTimeSum():  ESMF_BaseTimeSum%Sd = ',ESMF_BaseTimeSum%Sd
   214)         CALL normalize_basetime( ESMF_BaseTimeSum )

   365) SUBROUTINE compute_lcd( e1, e2, lcd )
   366)       IMPLICIT NONE
   367)       INTEGER(ESMF_KIND_I8), INTENT(IN) :: e1, e2
   368)       INTEGER(ESMF_KIND_I8), INTENT(OUT) :: lcd
   369)       INTEGER, PARAMETER ::  nprimes = 9
   370)       INTEGER(ESMF_KIND_I8), DIMENSION(nprimes), PARAMETER :: primes = (/2,3,5,7,11,13,17,19,23/)
   371)       INTEGER i
   372)       INTEGER(ESMF_KIND_I8) d1, d2, p
   373) 
   374)       d1 = e1 ; d2 = e2
   375)       IF ( d1 .EQ. 0 .AND. d2 .EQ. 0 ) THEN ; lcd = 1 ; RETURN ; ENDIF
   376)       IF ( d1 .EQ. 0 ) d1 = d2 
   377)       IF ( d2 .EQ. 0 ) d2 = d1 
   378)       IF ( d1 .EQ. d2 ) THEN ; lcd = d1 ; RETURN ; ENDIF
   379)       lcd = d1 * d2
   380)       DO i = 1, nprimes
   381)         p = primes(i)
   382)         DO WHILE (lcd/p .NE. 0 .AND. &
   383)           mod(lcd/p,d1) .EQ. 0 .AND. mod(lcd/p,d2) .EQ. 0) 
   384)           lcd = lcd / p 
   385)         END DO
   386)       ENDDO
   387) END SUBROUTINE compute_lcd

   391) SUBROUTINE simplify( ni, di, no, do ) 
   392)     IMPLICIT NONE
   393)     INTEGER(ESMF_KIND_I8), INTENT(IN)  :: ni, di
   394)     INTEGER(ESMF_KIND_I8), INTENT(OUT) :: no, do
   395)     INTEGER, PARAMETER ::  nprimes = 9
   396)     INTEGER(ESMF_KIND_I8), DIMENSION(nprimes), PARAMETER :: primes = (/2,3,5,7,11,13,17,19,23/)
   397)     INTEGER(ESMF_KIND_I8) :: pr, d, n
   398)     INTEGER :: np
   399)     LOGICAL keepgoing
   400)     IF ( ni .EQ. 0 ) THEN
   401)       do = 1
   402)       no = 0
   403)       RETURN
   404)     ENDIF
   405)     IF ( mod( di , ni ) .EQ. 0 ) THEN
   406)       do = di / ni
   407)       no = 1
   408)       RETURN
   409)     ENDIF
   410)     d = di
   411)     n = ni
   412)     DO np = 1, nprimes
   413)       pr = primes(np)
   414)       keepgoing = .TRUE.
   415)       DO WHILE ( keepgoing )
   416)         keepgoing = .FALSE.
   417)         IF ( d/pr .NE. 0 .AND. n/pr .NE. 0 .AND. MOD(d,pr) .EQ. 0 .AND. MOD(n,pr) .EQ. 0 ) THEN
   418)           d = d / pr
   419)           n = n / pr
   420)           keepgoing = .TRUE.
   421)         ENDIF
   422)       ENDDO
   423)     ENDDO
   424)     do = d
   425)     no = n
   426)     RETURN
   427) END SUBROUTINE simplify

   431) ! spaceship operator for seconds + Sn/Sd
   432) SUBROUTINE seccmp(S1, Sn1, Sd1, S2, Sn2, Sd2, retval )
   433)   IMPLICIT NONE
   434)   INTEGER, INTENT(OUT) :: retval
   435) !
   436) ! !ARGUMENTS:
   437)   INTEGER(ESMF_KIND_I8), INTENT(IN) :: S1, Sn1, Sd1
   438)   INTEGER(ESMF_KIND_I8), INTENT(IN) :: S2, Sn2, Sd2
   439) ! local
   440)   INTEGER(ESMF_KIND_I8) :: lcd, n1, n2
   441) 
   442)   n1 = Sn1
   443)   n2 = Sn2
   444)   if ( ( n1 .ne. 0 ) .or. ( n2 .ne. 0 ) ) then
   445)     CALL compute_lcd( Sd1, Sd2, lcd )
   446)     if ( Sd1 .ne. 0 ) n1 = n1 * ( lcd / Sd1 )
   447)     if ( Sd2 .ne. 0 ) n2 = n2 * ( lcd / Sd2 )
   448)   endif
   449) 
   450)   if ( S1 .GT. S2 ) retval = 1
   451)   if ( S1 .LT. S2 ) retval = -1
   452)   IF ( S1 .EQ. S2 ) THEN
   453)     IF (n1 .GT. n2) retval = 1
   454)     IF (n1 .LT. n2) retval = -1
   455)     IF (n1 .EQ. n2) retval = 0
   456)   ENDIF
   457) END SUBROUTINE seccmp
