     6) ! !DESCRIPTION:
     7) !
     8) ! Driver for ice mechanical redistribution (ridging)
     9) !
    10) ! See these references:
    11) !
    12) ! Flato, G. M., and W. D. Hibler III, 1995: Ridging and strength
    13) !  in modeling the thickness distribution of Arctic sea ice,
    14) !  J. Geophys. Res., 100, 18,611-18,626.
    15) !
    16) ! Hibler, W. D. III, 1980: Modeling a variable thickness sea ice
    17) !  cover, Mon. Wea. Rev., 108, 1943-1973, 1980.
    18) !
    19) ! Lipscomb, W. H., E. C. Hunke, W. Maslowski, and J. Jakacki, 2006: 
    20) !  Ridging, strength, and stability in sea ice models, submitted 
    21) !  to J. Geophys. Res. 
    22) ! 
    23) ! Rothrock, D. A., 1975: The energetics of the plastic deformation of
    24) !  pack ice by ridging, J. Geophys. Res., 80, 4514-4519.
    25) !
    26) ! Thorndike, A. S., D. A. Rothrock, G. A. Maykut, and R. Colony, 
    27) !  1975: The thickness distribution of sea ice, J. Geophys. Res., 
    28) !  80, 4501-4513. 
    29) !
    30) ! !REVISION HISTORY:
    31) !  SVN:$Id: ice_mechred.F90 53 2007-02-08 00:02:16Z dbailey $
    32) !
    33) ! authors: William H. Lipscomb, LANL
    34) !          Elizabeth C. Hunke, LANL
    35) !
    36) ! 2003: Vectorized by Clifford Chen (Fujitsu) and William Lipscomb
    37) ! 2004: Block structure added by William Lipscomb
    38) ! 2006: New options for participation and redistribution (WHL)
    39) ! 2006: Streamlined for efficiency by Elizabeth Hunke
    40) !       Converted to free source form (F90)

    60) !-----------------------------------------------------------------------
    61) ! Ridging parameters
    62) !-----------------------------------------------------------------------
    63) 
    64)       integer (kind=int_kind) :: & ! defined in namelist 
    65)          kstrength    , & ! 0 for simple Hibler (1979) formulation 
    66)                           ! 1 for Rothrock (1975) pressure formulation 
    67)          krdg_partic  , & ! 0 for Thorndike et al. (1975) formulation 
    68)                           ! 1 for exponential participation function 
    69)          krdg_redist      ! 0 for Hibler (1980) formulation 
    70)                           ! 1 for exponential redistribution function 
    71)  
    72)       real (kind=dbl_kind), parameter :: & 
    73)          Cf = 17._dbl_kind   , & ! ratio of ridging work to PE change in ridging 
    74)          Cs = p25            , & ! fraction of shear energy contrbtng to ridging 
    75)          Cp = p5*gravit*(rhow-rhoi)*rhoi/rhow, & ! proport const for PE 
    76)          fsnowrdg = p5       , & ! snow fraction that survives in ridging 
    77)          Gstar  = p15        , & ! max value of G(h) that participates 
    78)                                  ! (krdg_partic = 0) 
    79)          astar  = p05        , & ! e-folding scale for G(h) participation 
    80)                                  ! (krdg_partic = 1) 
    81)          maxraft= c1         , & ! max value of hrmin - hi = max thickness 
    82)                                  ! of ice that rafts (m) 
    83)          Hstar  = c25        , & ! determines mean thickness of ridged ice (m) 
    84)                                  ! (krdg_redist = 0) 
    85)                                  ! Flato & Hibler (1995) have Hstar = 100 
    86)          mu_rdg = c4         , & ! gives e-folding scale of ridged ice (m^.5) 
    87)                                  ! (krdg_redist = 1) 
    88)          Pstar = 2.75e4_dbl_kind, & ! constant in Hibler strength formula 
    89)                                  ! (kstrength = 0) 
    90)          Cstar = c20             ! constant in Hibler strength formula 
    91)                                  ! (kstrength = 0) 
    92) 
    93)       logical (kind=log_kind), parameter :: &
    94)          l_conservation_check = .true.  ! if true, check conservation
    95) !         l_conservation_check = .false.  ! if true, check conservation
    96)                                         ! (useful for debugging)
    97)

   104) !
   105) ! !ROUTINE: ridge_ice - driver for mechanical redistribution
   106) !
   107) ! !DESCRIPTION:
   108) !
   109) ! Compute changes in the ice thickness distribution due to divergence
   110) ! and shear.
   111) ! NOTE: This subroutine operates over a single block.
   112) !
   113) ! !REVISION HISTORY:
   114) !
   115) ! author: William H. Lipscomb, LANL
   116) !
   117) ! !INTERFACE:
   118) !
   119)       subroutine ridge_ice (nx_block,    ny_block,   &
   120)                             dt_dyn,      dt_thm,     &
   121)                             ntrcr,       icells,     &
   122)                             indxi,       indxj,      &
   123)                             rdg_conv,    rdg_shear,  &
   124)                             aicen,       trcrn,      &
   125)                             vicen,       vsnon,      &
   126)                             eicen,       esnon,      &
   127)                             aice0,                   &
   128)                             trcr_depend, l_stop,     &
   129)                             istop,       jstop,      &
   130)                             dardg1dt,    dardg2dt,   &
   131)                             dvirdgdt,    opening,    &
   132)                             fresh,       fhocn,      &
   133)                             fsoot)
   134) !
   135) ! !USES:
   136) !
   137) ! !INPUT/OUTPUT PARAMETERS:
   138) !
   139)       integer (kind=int_kind), intent(in) :: &
   140)          nx_block, ny_block, & ! block dimensions
   141)          icells            , & ! number of cells with ice present
   142)          ntrcr                 ! number of tracers in use
   143) 
   144)       integer (kind=int_kind), dimension (nx_block*ny_block), &
   145)          intent(in) :: &
   146)          indxi, indxj     ! compressed indices for cells with ice
   147) 
   148)       real (kind=dbl_kind), intent(in) :: &
   149)          dt_dyn  , & ! dynamic time step
   150)          dt_thm      ! thermodynamic time step for diagnostics
   151) 
   152)       real (kind=dbl_kind), dimension(nx_block,ny_block), intent(in) :: &
   153)          rdg_conv, & ! normalized energy dissipation due to convergence (1/s)
   154)          rdg_shear   ! normalized energy dissipation due to shear (1/s)
   155)  
   156)       real (kind=dbl_kind), dimension (nx_block,ny_block,ncat), &
   157)          intent(inout) :: &
   158)          aicen , & ! concentration of ice
   159)          vicen , & ! volume per unit area of ice          (m)
   160)          vsnon     ! volume per unit area of snow         (m)
   161)  
   162)       real (kind=dbl_kind), dimension (nx_block,ny_block,max_ntrcr,ncat), &
   163)          intent(inout) :: & 
   164)          trcrn     ! ice tracers 
   165)  
   166)       real (kind=dbl_kind), dimension (nx_block,ny_block,ntilyr), &
   167)          intent(inout) :: & 
   168)          eicen     ! energy of melting for each ice layer (J/m^2)
   169)  
   170)       real (kind=dbl_kind), dimension (nx_block,ny_block,ntslyr), &
   171)          intent(inout) :: & 
   172)          esnon     ! energy of melting for each snow layer (J/m^2)
   173) 
   174)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   175)          intent(inout) :: & 
   176)          aice0     ! concentration of open water
   177) 
   178)       integer (kind=int_kind), dimension(max_ntrcr), intent(in) :: &
   179)          trcr_depend
   180) 
   181)       logical (kind=log_kind), intent(out) :: &
   182)          l_stop   ! if true, abort on return
   183) 
   184)       integer (kind=int_kind), intent(out) :: &
   185)          istop, jstop ! indices of grid cell where model aborts
   186) 
   187)       ! optional history fields
   188)       real (kind=dbl_kind), dimension(nx_block,ny_block), &
   189)          intent(inout), optional :: &
   190)          dardg1dt  , & ! rate of fractional area loss by ridging ice (1/s)
   191)          dardg2dt  , & ! rate of fractional area gain by new ridges (1/s)
   192)          dvirdgdt  , & ! rate of ice volume ridged (m/s)
   193)          opening   , & ! rate of opening due to divergence/shear (1/s)
   194)          fresh     , & ! fresh water flux to ocean (kg/m^2/s)
   195)          fhocn         ! net heat flux to ocean (W/m^2)
   196) 
   197)       real (kind=dbl_kind), dimension(nx_block,ny_block,n_aeromx), &
   198)          intent(inout), optional :: &
   199)          fsoot      ! 
   200) !
   201) !EOP
   202) !
   203)       real (kind=dbl_kind), dimension (icells) :: &
   204)          asum       , & ! sum of ice and open water area
   205)          aksum      , & ! ratio of area removed to area ridged
   206)          msnow_mlt  , & ! mass of snow added to ocean (kg m-2)
   207)          esnow_mlt  , & ! energy needed to melt snow in ocean (J m-2)
   208)          closing_net, & ! net rate at which area is removed    (1/s)
   209)                         ! (ridging ice area - area of new ridges) / dt_dyn
   210)          divu_adv   , & ! divu as implied by transport scheme  (1/s)
   211)          opning     , & ! rate of opening due to divergence/shear
   212)                         ! opning is a local variable;
   213)                         ! opening is the history diagnostic variable
   214)          ardg1      , & ! fractional area loss by ridging ice
   215)          ardg2      , & ! fractional area gain by new ridges
   216)          virdg      , & ! ice volume ridged
   217)          aopen          ! area opening due to divergence/shear
   218) 
   219)       real (kind=dbl_kind), dimension (icells,n_aeromx) :: &
   220)          msoot       ! mass of soot added to ocean (kg m-2)
   221) 
   222)       real (kind=dbl_kind), dimension (icells,0:ncat) :: &
   223)          apartic          ! participation function; fraction of ridging
   224)                           ! and closing associated w/ category n
   225) 
   226)       real (kind=dbl_kind), dimension (icells,ncat) :: &
   227)          hrmin        , & ! minimum ridge thickness
   228)          hrmax        , & ! maximum ridge thickness (krdg_redist = 0)
   229)          hrexp        , & ! ridge e-folding thickness (krdg_redist = 1) 
   230)          krdg             ! mean ridge thickness/thickness of ridging ice
   231) 
   232)       real (kind=dbl_kind), dimension (icells) :: &
   233)          vice_init, vice_final, & ! ice volume summed over categories
   234)          vsno_init, vsno_final, & ! snow volume summed over categories
   235)          eice_init, eice_final, & ! ice energy summed over layers
   236)          esno_init, esno_final    ! snow energy summed over layers
   237) 
   238)       integer (kind=int_kind), parameter :: &
   239)          nitermax = 20    ! max number of ridging iterations
   240) 
   241)       integer (kind=int_kind) :: &
   242)          i,j          , & ! horizontal indices
   243)          n            , & ! thickness category index
   244)          niter        , & ! iteration counter
   245)          ij               ! horizontal index, combines i and j loops
   246) 
   247)       real (kind=dbl_kind) :: &
   248)          dti              ! 1 / dt_dyn or 1 / dt_thm
   249) 
   250)       logical (kind=log_kind) :: &
   251)          iterate_ridging, & ! if true, repeat the ridging
   252)          asum_error         ! flag for asum .ne. 1
   253) 
   254)       character (len=char_len) :: &
   255)          fieldid            ! field identifier
   256) 
   257)       !-----------------------------------------------------------------
   258)       ! Initialize
   259)       !-----------------------------------------------------------------
   260) 
   261)       l_stop = .false.
   262)       istop = 0
   263)       jstop = 0
   264) 
   265)       do ij = 1, icells
   266)          msnow_mlt(ij) = c0
   267)          esnow_mlt(ij) = c0
   268)          msoot    (ij,:) = c0
   269)          ardg1    (ij) = c0
   270)          ardg2    (ij) = c0
   271)          virdg    (ij) = c0
   272) !         aopen    (ij) = c0
   273)       enddo
   274) 
   275)       !-----------------------------------------------------------------
   276)       ! Compute area of ice plus open water before ridging.
   277)       !-----------------------------------------------------------------
   278)       call asum_ridging (nx_block, ny_block,        &
   279)                          icells,   indxi,    indxj, &
   280)                          aicen,    aice0,           &
   281)                          asum)
   282) 
   283)       !-----------------------------------------------------------------
   284)       ! Compute the area opening and closing.
   285)       !-----------------------------------------------------------------
   286)       call ridge_prep (nx_block, ny_block,      &
   287)                        icells,   indxi,  indxj, &
   288)                        dt_dyn,                  &
   289)                        rdg_conv,  rdg_shear,    &
   290)                        asum,      closing_net,  &
   291)                        divu_adv,  opning)
   292) 
   293)       !-----------------------------------------------------------------
   294)       ! Compute initial values of conserved quantities. 
   295)       !-----------------------------------------------------------------
   296) 
   297)       if (l_conservation_check) then
   298) 
   299)          call column_sum (nx_block,   ny_block,     &
   300)                           icells,   indxi,   indxj, &
   301)                           ncat,                     &
   302)                           vicen,      vice_init)
   303) 
   304)          call column_sum (nx_block,   ny_block,     &
   305)                           icells,   indxi,   indxj, &
   306)                           ncat,                     &
   307)                           vsnon,      vsno_init)
   308) 
   309)          call column_sum (nx_block,   ny_block,     &
   310)                           icells,   indxi,   indxj, &
   311)                           ntilyr,                   &
   312)                           eicen,      eice_init)
   313) 
   314)          call column_sum (nx_block,   ny_block,     &
   315)                           icells,   indxi,   indxj, &
   316)                           ntslyr,                   &
   317)                           esnon,      esno_init)
   318) 
   319)       endif            
   320) 
   321)       do niter = 1, nitermax
   322) 
   323)       !-----------------------------------------------------------------
   324)       ! Compute the thickness distribution of ridging ice
   325)       ! and various quantities associated with the new ridged ice.
   326)       !-----------------------------------------------------------------
   327) 
   328)          call ridge_itd (nx_block,  ny_block,        &
   329)                          icells,    indxi,    indxj, &
   330)                          aicen,     vicen,           &
   331)                          aice0,                      &
   332)                          aksum,     apartic,         &
   333)                          hrmin,     hrmax,           &
   334)                          hrexp,     krdg)
   335) 
   336)       !-----------------------------------------------------------------
   337)       ! Redistribute area, volume, and energy.
   338)       !-----------------------------------------------------------------
   339) 
   340)          call ridge_shift (nx_block,  ny_block,        &
   341)                            icells,    indxi,    indxj, &
   342)                            ntrcr,     dt_dyn,          &
   343)                            aicen,     trcrn,           &
   344)                            vicen,     vsnon,           &
   345)                            eicen,     esnon,           &
   346)                            aice0,     trcr_depend,     &
   347)                            aksum,     apartic,         &
   348)                            hrmin,     hrmax,           &
   349)                            hrexp,     krdg,            &
   350)                            closing_net, opning,        &
   351)                            ardg1,     ardg2,           &
   352)                            virdg,     aopen,           &
   353)                            msnow_mlt, esnow_mlt,       &
   354)                            msoot,                      &
   355)                            l_stop,                     &
   356)                            istop,     jstop)
   357) 
   358)          if (l_stop) return
   359) 
   360)       !-----------------------------------------------------------------
   361)       ! Make sure the new area = 1.  If not (because the closing
   362)       ! and opening rates were reduced above), prepare to ridge again
   363)       ! with new rates.
   364)       !-----------------------------------------------------------------
   365) 
   366)          call asum_ridging (nx_block,  ny_block,      &
   367)                             icells,    indxi,  indxj, &
   368)                             aicen,     aice0,         &
   369)                             asum)
   370) 
   371)          call ridge_check (nx_block,   ny_block,        &
   372)                            icells,     indxi,    indxj, &
   373)                            dt_dyn,                      &
   374)                            asum,       closing_net,     &
   375)                            divu_adv,   opning,          &
   376)                            iterate_ridging)
   377) 
   378)       !-----------------------------------------------------------------
   379)       ! If done, exit.  If not, prepare to ridge again.
   380)       !-----------------------------------------------------------------
   381) 
   382)          if (iterate_ridging) then
   383)             write(nu_diag,*) 'Repeat ridging, niter =', niter
   384)          else
   385)             exit
   386)          endif
   387) 
   388)          if (niter == nitermax) then
   389)             write(nu_diag,*) ' '
   390)             write(nu_diag,*) 'Exceeded max number of ridging iterations'
   391)             write(nu_diag,*) 'max =',nitermax
   392)             l_stop = .true.
   393)             return
   394)          endif
   395)             
   396)       enddo                     ! niter
   397) 
   398)       !-----------------------------------------------------------------
   399)       ! Compute final values of conserved quantities. 
   400)       ! Check for conservation (allowing for snow thrown into ocean).
   401)       !-----------------------------------------------------------------
   402) 
   403)       if (l_conservation_check) then
   404) 
   405)          call column_sum (nx_block,   ny_block,     &
   406)                           icells,   indxi,   indxj, &
   407)                           ncat,                     &
   408)                           vicen,      vice_final)
   409) 
   410)          call column_sum (nx_block,   ny_block,     &
   411)                           icells,   indxi,   indxj, &
   412)                           ncat,                     &
   413)                           vsnon,      vsno_final)
   414) 
   415)          call column_sum (nx_block,   ny_block,     &
   416)                           icells,   indxi,   indxj, &
   417)                           ntilyr,                   &
   418)                           eicen,      eice_final)
   419) 
   420)          call column_sum (nx_block,   ny_block,     &
   421)                           icells,   indxi,   indxj, &
   422)                           ntslyr,                   &
   423)                           esnon,      esno_final)
   424) 
   425)          do ij = 1, icells
   426)             vsno_final(ij) = vsno_final(ij) + msnow_mlt(ij)/rhos
   427)             esno_final(ij) = esno_final(ij) + esnow_mlt(ij)
   428)          enddo
   429) 
   430)          fieldid = 'vice, ridging'
   431)          call column_conservation_check (nx_block,  ny_block,       &
   432)                                          icells,    indxi,   indxj, &
   433)                                          fieldid,                   &
   434)                                          vice_init, vice_final,     &
   435)                                          puny,      l_stop,         &
   436)                                          istop,     jstop)
   437)          if (l_stop) return
   438) 
   439)          fieldid = 'vsno, ridging'
   440)          call column_conservation_check (nx_block,  ny_block,       &
   441)                                          icells,    indxi,   indxj, &
   442)                                          fieldid,                   &
   443)                                          vsno_init, vsno_final,     &
   444)                                          puny,      l_stop,         &
   445)                                          istop,     jstop)
   446)          if (l_stop) return
   447) 
   448)          fieldid = 'eice, ridging'
   449)          call column_conservation_check (nx_block,  ny_block,       &
   450)                                          icells,    indxi,   indxj, &
   451)                                          fieldid,                   &
   452)                                          eice_init, eice_final,     &
   453)                                          puny*Lfresh*rhoi,          &
   454)                                          l_stop,                    &
   455)                                          istop,     jstop)
   456)          if (l_stop) return
   457) 
   458)          fieldid = 'esno, ridging'
   459)          call column_conservation_check (nx_block,  ny_block,       &
   460)                                          icells,    indxi,   indxj, &
   461)                                          fieldid,                   &
   462)                                          esno_init, esno_final,     &
   463)                                          puny*Lfresh*rhos,          &
   464)                                          l_stop,                    &
   465)                                          istop,     jstop)
   466)          if (l_stop) return
   467)          
   468)       endif                     ! l_conservation_check            
   469) 
   470)       !-----------------------------------------------------------------
   471)       ! Compute ridging diagnostics.
   472)       !-----------------------------------------------------------------
   473) 
   474)       dti = c1/dt_dyn
   475) 
   476)       if (present(dardg1dt)) then
   477)          do ij = 1, icells
   478)             i = indxi(ij)
   479)             j = indxj(ij)
   480)             dardg1dt(i,j) = ardg1(ij)*dti
   481)          enddo
   482)       endif
   483)       if (present(dardg2dt)) then
   484)          do ij = 1, icells
   485)             i = indxi(ij)
   486)             j = indxj(ij)
   487)             dardg2dt(i,j) = ardg2(ij)*dti
   488)          enddo
   489)       endif
   490)       if (present(dvirdgdt)) then
   491)          do ij = 1, icells
   492)             i = indxi(ij)
   493)             j = indxj(ij)
   494)             dvirdgdt(i,j) = virdg(ij)*dti
   495)          enddo
   496)       endif
   497)       if (present(opening)) then
   498)          do ij = 1, icells
   499)             i = indxi(ij)
   500)             j = indxj(ij)
   501)             opening(i,j) = aopen(ij)*dti
   502)          enddo
   503)       endif
   504) 
   505)       !-----------------------------------------------------------------
   506)       ! Update fresh water and heat fluxes due to snow melt.
   507)       !-----------------------------------------------------------------
   508) 
   509)       dti = c1/dt_thm
   510) 
   511)       if (present(fsoot)) then
   512)          do ij = 1, icells
   513)             i = indxi(ij)
   514)             j = indxj(ij)
   515)             fsoot(i,j,:) = fsoot(i,j,:) + msoot(ij,:)*dti
   516)          enddo
   517)       endif
   518)       if (present(fresh)) then
   519)          do ij = 1, icells
   520)             i = indxi(ij)
   521)             j = indxj(ij)
   522)             fresh(i,j) = fresh(i,j) + msnow_mlt(ij)*dti
   523)          enddo
   524)       endif
   525)       if (present(fhocn)) then
   526)          do ij = 1, icells
   527)             i = indxi(ij)
   528)             j = indxj(ij)
   529)             fhocn(i,j) = fhocn(i,j) + esnow_mlt(ij)*dti
   530)          enddo
   531)       endif
   532) 
   533)       !-----------------------------------------------------------------
   534)       ! Check for fractional ice area > 1.
   535)       !-----------------------------------------------------------------
   536) 
   537)       do ij = 1, icells
   538)          i = indxi(ij)
   539)          j = indxj(ij)
   540)          if (abs(asum(ij) - c1) > puny) then
   541)             l_stop = .true.
   542)             istop = i
   543)             jstop = j
   544) 
   545)             write(nu_diag,*) ' '
   546)             write(nu_diag,*) 'Ridging error: total area > 1'
   547)             write(nu_diag,*) 'i, j, area:', i, j, asum(ij)
   548)             write(nu_diag,*) 'n, aicen:'
   549)             write(nu_diag,*)  0, aice0(i,j)
   550)             do n = 1, ncat
   551)                write(nu_diag,*) n, aicen(i,j,n)
   552)             enddo
   553)             return
   554)          endif

   641) ! !ROUTINE: ridge_prep - preparation for ridging
   642) !
   643) ! !DESCRIPTION: Initialize arrays, compute area of closing and opening
   644) !
   645) !
   646) ! !REVISION HISTORY:
   647) !
   648) ! author: William H. Lipscomb, LANL
   649) !
   650) ! !INTERFACE:
   651) !
   652)       subroutine ridge_prep (nx_block,   ny_block,        &
   653)                              icells,     indxi,    indxj, &
   654)                              dt,                          &
   655)                              rdg_conv,   rdg_shear,       &
   656)                              asum,       closing_net,     &
   657)                              divu_adv,   opning)
   658) !
   659) ! !USES:
   660) !
   661) ! !INPUT/OUTPUT PARAMETERS:
   662) !
   663)       integer (kind=int_kind), intent(in) :: &
   664)          nx_block, ny_block, & ! block dimensions
   665)          icells                ! number of cells with ice present
   666) 
   667)       integer (kind=int_kind), dimension (nx_block*ny_block), &
   668)          intent(in) :: &
   669)          indxi, indxj     ! compressed indices for cells with ice
   670) 
   671) 
   672)       real (kind=dbl_kind), intent(in) :: &
   673)          dt                  ! time step (s)
   674) 
   675)       real (kind=dbl_kind), dimension(nx_block,ny_block), intent(in) :: &
   676)          rdg_conv, & ! normalized energy dissipation due to convergence (1/s)
   677)          rdg_shear   ! normalized energy dissipation due to shear (1/s)
   678) 
   679)       real (kind=dbl_kind), dimension(icells), &
   680)          intent(inout):: &
   681)          asum      ! sum of ice and open water area
   682) 
   683)       real (kind=dbl_kind), dimension(icells), &
   684)          intent(out):: &
   685)          closing_net, & ! net rate at which area is removed    (1/s)
   686)          divu_adv   , & ! divu as implied by transport scheme  (1/s)
   687)          opning         ! rate of opening due to divergence/shear
   688) !
   689) !EOP
   690) !
   691)       real (kind=dbl_kind), parameter :: &
   692)          big = 1.0e+8_dbl_kind
   693) 
   694)       integer (kind=int_kind) :: &
   695)          i,j, &         ! horizontal indices
   696)          ij             ! horizontal index, combines i and j loops
   697) 
   698)       ! Set hin_max(ncat) to a big value to ensure that all ridged ice
   699)       ! is thinner than hin_max(ncat).
   700)       hin_max(ncat) = big
   701) 
   702)       do ij = 1, icells
   703)          i = indxi(ij)
   704)          j = indxj(ij)
   705) 
   706)       !-----------------------------------------------------------------
   707)       ! Compute the net rate of closing due to convergence
   708)       ! and shear, based on Flato and Hibler (1995).
   709)       !
   710)       ! For the elliptical yield curve:
   711)       !    rdg_conv  = -min (divu, 0)
   712)       !    rdg_shear = (1/2) * (Delta - abs(divu))
   713)       ! Note that the shear term also accounts for divergence.
   714)       !
   715)       ! The energy dissipation rate is equal to the net closing rate
   716)       ! times the ice strength.
   717)       !
   718)       ! NOTE: The NET closing rate is equal to the rate that open water
   719)       !  area is removed, plus the rate at which ice area is removed by
   720)       !  ridging, minus the rate at which area is added in new ridges.
   721)       !  The GROSS closing rate is equal to the first two terms (open
   722)       !  water closing and thin ice ridging) without the third term
   723)       !  (thick, newly ridged ice).
   724)       !-----------------------------------------------------------------
   725) 
   726)          closing_net(ij) = Cs*rdg_shear(i,j) + rdg_conv(i,j)
   727) 
   728)       !-----------------------------------------------------------------
   729)       ! Compute divu_adv, the divergence rate given by the transport/
   730)       ! advection scheme, which may not be equal to divu as computed
   731)       ! from the velocity field.
   732)       !
   733)       ! If divu_adv < 0, make sure the closing rate is large enough
   734)       ! to give asum = 1.0 after ridging.
   735)       !-----------------------------------------------------------------
   736) 
   737)          divu_adv(ij) = (c1-asum(ij)) / dt
   738) 
   739)          if (divu_adv(ij) < c0) &
   740)               closing_net(ij) = max(closing_net(ij), -divu_adv(ij))
   741) 
   742)       !-----------------------------------------------------------------
   743)       ! Compute the (non-negative) opening rate that will give
   744)       ! asum = 1.0 after ridging.
   745)       !-----------------------------------------------------------------
   746)          opning(ij) = closing_net(ij) + divu_adv(ij)
   747) 
   748)       enddo
   749) 
   750)       end subroutine ridge_prep

   755) ! !ROUTINE: ridge_itd - thickness distribution of ridging and ridged ice
   756) !
   757) ! !DESCRIPTION:
   758) !
   759) ! Compute the thickness distribution of the ice and open water
   760) ! participating in ridging and of the resulting ridges.
   761) !
   762) ! This version includes new options for ridging participation and
   763) !  redistribution.
   764) ! The new participation scheme (krdg_partic = 1) improves stability
   765) !  by increasing the time scale for large changes in ice strength.
   766) ! The new exponential redistribution function (krdg_redist = 1) improves 
   767) !  agreement between ITDs of modeled and observed ridges.   
   768) !
   769) ! !REVISION HISTORY:
   770) !
   771) ! author: William H. Lipscomb, LANL
   772) !
   773) ! 2006: Changed subroutine name to ridge_itd
   774) !       Added new options for ridging participation and redistribution.  
   775) !
   776) ! !INTERFACE:
   777) !
   778)       subroutine ridge_itd (nx_block,    ny_block,        &
   779)                             icells,      indxi,    indxj, &
   780)                             aicen,       vicen,           &
   781)                             aice0,                        &
   782)                             aksum,       apartic,         &
   783)                             hrmin,       hrmax,           &
   784)                             hrexp,       krdg)
   785) !
   786) ! !USES:
   787) !
   788) ! !INPUT/OUTPUT PARAMETERS:
   789) !
   790)       integer (kind=int_kind), intent(in) :: &
   791)          nx_block, ny_block, & ! block dimensions
   792)          icells                ! number of cells with ice present
   793) 
   794)       integer (kind=int_kind), dimension (nx_block*ny_block), &
   795)          intent(in) :: &
   796)          indxi, indxj     ! compressed indices for cells with ice
   797) 
   798) 
   799)       real (kind=dbl_kind), dimension (nx_block,ny_block,ncat), &
   800)          intent(in) :: &
   801)          aicen , & ! concentration of ice
   802)          vicen     ! volume per unit area of ice (m)
   803) 
   804)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(in) :: &
   805)          aice0     ! concentration of open water
   806) 
   807)       real (kind=dbl_kind), dimension (icells), intent(out):: &
   808)          aksum            ! ratio of area removed to area ridged
   809) 
   810)       real (kind=dbl_kind), dimension (icells,0:ncat), &
   811)          intent(out) :: &
   812)          apartic          ! participation function; fraction of ridging
   813)                           ! and closing associated w/ category n
   814) 
   815)       real (kind=dbl_kind), dimension (icells,ncat), &
   816)          intent(out) :: &
   817)          hrmin        , & ! minimum ridge thickness
   818)          hrmax        , & ! maximum ridge thickness (krdg_redist = 0)
   819)          hrexp        , & ! ridge e-folding thickness (krdg_redist = 1) 
   820)          krdg             ! mean ridge thickness/thickness of ridging ice
   821) !
   822) !EOP
   823) !
   824)       integer (kind=int_kind) :: &
   825)          i,j          , & ! horizontal indices
   826)          n            , & ! thickness category index
   827)          ij               ! horizontal index, combines i and j loops
   828) 
   829)       real (kind=dbl_kind), parameter :: &
   830)          Gstari   = c1/Gstar, &
   831)          astari   = c1/astar
   832) 
   833)       real (kind=dbl_kind), dimension(icells,-1:ncat) :: &
   834)          Gsum             ! Gsum(n) = sum of areas in categories 0 to n
   835) 
   836)       real (kind=dbl_kind), dimension(icells) :: &
   837)          work             ! temporary work array
   838) 
   839)       real (kind=dbl_kind) :: &
   840)          hi           , & ! ice thickness for each cat (m)
   841)          hieff        , & ! effective ice thickness (m) (krdg_redist = 2)
   842)          hrmean       , & ! mean ridge thickness (m)
   843)          xtmp             ! temporary variable
   844) 
   845)       !-----------------------------------------------------------------
   846)       ! Initialize
   847)       !-----------------------------------------------------------------
   848) 
   849)       do ij = 1, icells
   850)          Gsum   (ij,-1) = c0        ! by definition
   851)          Gsum   (ij,0)  = c1         ! to avoid divzero below
   852)          apartic(ij,0)  = c0
   853)       enddo
   854) 
   855)       do n = 1, ncat
   856)          do ij = 1, icells
   857)             Gsum   (ij,n) = c1    ! to avoid divzero below
   858)             apartic(ij,n) = c0
   859)             hrmin  (ij,n) = c0
   860)             hrmax  (ij,n) = c0
   861)             hrexp  (ij,n) = c0
   862)             krdg   (ij,n) = c1
   863)          enddo
   864)       enddo
   865) 
   866)       !-----------------------------------------------------------------
   867)       ! Compute the thickness distribution of ice participating in ridging.
   868)       !-----------------------------------------------------------------
   869) 
   870)       !-----------------------------------------------------------------
   871)       ! First compute the cumulative thickness distribution function Gsum,
   872)       !  where Gsum(n) is the fractional area in categories 0 to n.
   873)       ! Ignore categories with very small areas.
   874)       !-----------------------------------------------------------------
   875) 
   876) !DIR$ CONCURRENT !Cray
   877) !cdir nodep      !NEC
   878) !ocl novrec      !Fujitsu
   879)       do ij = 1, icells
   880)          i = indxi(ij)
   881)          j = indxj(ij)
   882)          if (aice0(i,j) > puny) then
   883)             Gsum(ij,0) = aice0(i,j)
   884)          else
   885)             Gsum(ij,0) = Gsum(ij,-1)
   886)          endif
   887)       enddo
   888) 
   889)       do n = 1, ncat
   890) !DIR$ CONCURRENT !Cray
   891) !cdir nodep      !NEC
   892) !ocl novrec      !Fujitsu
   893)          do ij = 1, icells
   894)             i = indxi(ij)
   895)             j = indxj(ij)
   896)             if (aicen(i,j,n) > puny) then
   897)                Gsum(ij,n) = Gsum(ij,n-1) + aicen(i,j,n)
   898)             else
   899)                Gsum(ij,n) = Gsum(ij,n-1)
   900)             endif
   901)          enddo
   902)       enddo
   903) 
   904)       ! normalize
   905) 
   906)       do ij = 1, icells
   907)          work(ij) = c1 / Gsum(ij,ncat)
   908)       enddo
   909)       do n = 0, ncat
   910) !DIR$ CONCURRENT !Cray
   911) !cdir nodep      !NEC
   912) !ocl novrec      !Fujitsu
   913)          do ij = 1, icells
   914)             Gsum(ij,n) = Gsum(ij,n) * work(ij)
   915)          enddo
   916)       enddo
   917) 
   918)       !-----------------------------------------------------------------
   919)       ! Compute the participation function apartic; this is analogous to
   920)       ! a(h) = b(h)g(h) as defined in Thorndike et al. (1975).
   921)       !
   922)       !                area lost from category n due to ridging/closing
   923)       !  apartic(n) = --------------------------------------------------
   924)       !                  total area lost due to ridging/closing
   925)       !
   926)       !-----------------------------------------------------------------
   927) 
   928)       if (krdg_partic == 0) then  ! Thornike et al. 1975 formulation
   929) 
   930)       !-----------------------------------------------------------------
   931)       ! Assume b(h) = (2/Gstar) * (1 - G(h)/Gstar).
   932)       ! The expressions for apartic are found by integrating b(h)g(h) between
   933)       ! the category boundaries.
   934)       !-----------------------------------------------------------------
   935) 
   936)          do n = 0, ncat
   937)             do ij = 1, icells
   938)                if (Gsum(ij,n) < Gstar) then
   939)                   apartic(ij,n) = Gstari*(Gsum(ij,n)-Gsum(ij,n-1)) * &
   940)                        (c2 - (Gsum(ij,n-1)+Gsum(ij,n))*Gstari)
   941)                elseif (Gsum(ij,n-1) < Gstar) then
   942)                   apartic(ij,n) = Gstari * (Gstar-Gsum(ij,n-1)) * &
   943)                        (c2 - (Gsum(ij,n-1)+Gstar)*Gstari)
   944)                endif
   945)             enddo               ! ij
   946)          enddo                  ! n
   947) 
   948)       elseif (krdg_partic==1) then   ! exponential dependence on G(h)
   949) 
   950)       !-----------------------------------------------------------------
   951)       ! b(h) = exp(-G(h)/astar)
   952)       ! apartic(n) = [exp(-G(n-1)/astar - exp(-G(n)/astar] / [1-exp(-1/astar)]. 
   953)       ! The expression for apartic is found by integrating b(h)g(h)
   954)       ! between the category boundaries.
   955)       !-----------------------------------------------------------------
   956) 
   957)          ! precompute exponential terms using Gsum as work array
   958) 
   959)          xtmp = c1 / (c1 - exp(-astari))
   960) 
   961)          do n = -1, ncat
   962) !DIR$ CONCURRENT !Cray
   963) !cdir nodep      !NEC
   964) !ocl novrec      !Fujitsu
   965)             do ij = 1, icells
   966)                Gsum(ij,n) = exp(-Gsum(ij,n)*astari) * xtmp
   967)             enddo               ! ij
   968)          enddo                  ! n
   969) 
   970)          do n = 0, ncat
   971)             do ij = 1, icells
   972)                apartic(ij,n) = Gsum(ij,n-1) - Gsum(ij,n)
   973)             enddo               ! ij
   974)          enddo                  ! n
   975) 
   976)       endif                     ! krdg_partic
   977) 
   978)       !-----------------------------------------------------------------
   979)       ! Compute variables related to ITD of ridged ice:
   980)       ! 
   981)       ! krdg = mean ridge thickness/ thickness of ridging ice
   982)       ! hrmin  = min ridge thickness
   983)       ! hrmax  = max ridge thickness (krdg_redist = 0)
   984)       ! hrexp  = ridge e-folding scale (krdg_redist = 1)
   985)       !----------------------------------------------------------------
   986) 
   987)       if (krdg_redist == 0) then  ! Hibler 1980 formulation
   988) 
   989)       !-----------------------------------------------------------------
   990)       ! Assume ridged ice is uniformly distributed between hrmin and hrmax. 
   991)       ! 
   992)       ! This parameterization is a modified version of Hibler (1980). 
   993)       ! In the original paper the min ridging thickness is hrmin = 2*hi, 
   994)       !  and the max thickness is hrmax = 2*sqrt(hi*Hstar). 
   995)       ! 
   996)       ! Here the min thickness is hrmin = min(2*hi, hi+maxraft), 
   997)       !  so thick ridging ice is not required to raft. 
   998)       !
   999)       !-----------------------------------------------------------------
  1000) 
  1001)          do n = 1, ncat
  1002)             do ij = 1, icells
  1003)                i = indxi(ij)
  1004)                j = indxj(ij)
  1005) 
  1006)                if (aicen(i,j,n) > puny) then 
  1007)                   hi = vicen(i,j,n) / aicen(i,j,n) 
  1008)                   hrmin(ij,n) = min(c2*hi, hi + maxraft) 
  1009)                   hrmax(ij,n) = c2*sqrt(Hstar*hi) 
  1010)                   hrmax(ij,n) = max(hrmax(ij,n), hrmin(ij,n)+puny) 
  1011)                   hrmean = p5 * (hrmin(ij,n) + hrmax(ij,n)) 
  1012)                   krdg(ij,n) = hrmean / hi 
  1013)                endif 
  1014) 
  1015)             enddo               ! ij
  1016)          enddo                  ! n
  1017) 
  1018)       else               ! krdg_redist = 1; exponential redistribution
  1019)  
  1020)       !----------------------------------------------------------------- 
  1021)       ! The ridge ITD is a negative exponential: 
  1022)       ! 
  1023)       !  g(h) ~ exp[-(h-hrmin)/hrexp], h >= hrmin 
  1024)       ! 
  1025)       ! where hrmin is the minimum thickness of ridging ice and 
  1026)       ! hrexp is the e-folding thickness.
  1027)       ! 
  1028)       ! Here, assume as above that hrmin = min(2*hi, hi+maxraft).
  1029)       ! That is, the minimum ridge thickness results from rafting,
  1030)       !  unless the ice is thicker than maxraft.
  1031)       !
  1032)       ! Also, assume that hrexp = mu_rdg*sqrt(hi).
  1033)       ! The parameter mu_rdg is tuned to give e-folding scales mostly
  1034)       !  in the range 2-4 m as observed by upward-looking sonar.
  1035)       !
  1036)       ! Values of mu_rdg in the right column give ice strengths
  1037)       !  roughly equal to values of Hstar in the left column
  1038)       !  (within ~10 kN/m for typical ITDs):
  1039)       !
  1040)       !   Hstar     mu_rdg
  1041)       !
  1042)       !     25        3.0
  1043)       !     50        4.0
  1044)       !     75        5.0
  1045)       !    100        6.0
  1046)       !----------------------------------------------------------------- 
  1047) 
  1048)          do n = 1, ncat
  1049)             do ij = 1, icells
  1050)                i = indxi(ij)
  1051)                j = indxj(ij)
  1052)                if (aicen(i,j,n) > puny) then
  1053)                   hi = vicen(i,j,n) / aicen(i,j,n)
  1054)                   hi = max(hi,puny)
  1055)                   hrmin(ij,n) = min(c2*hi, hi + maxraft)
  1056)                   hrexp(ij,n) = mu_rdg * sqrt(hi)
  1057)                   krdg(ij,n) = (hrmin(ij,n) + hrexp(ij,n)) / hi
  1058)                endif
  1059)             enddo
  1060)          enddo
  1061) 
  1062)       endif                     ! krdg_redist
  1063) 
  1064)       !----------------------------------------------------------------
  1065)       ! Compute aksum = net ice area removed / total area participating.
  1066)       ! For instance, if a unit area of ice with h = 1 participates in
  1067)       !  ridging to form a ridge with a = 1/3 and h = 3, then
  1068)       !  aksum = 1 - 1/3 = 2/3.
  1069)       !---------------------------------------------------------------- 
  1070) 
  1071)       do ij = 1, icells
  1072)          aksum(ij) = apartic(ij,0) ! area participating = area removed
  1073)       enddo
  1074) 
  1075)       do n = 1, ncat
  1076) !DIR$ CONCURRENT !Cray
  1077) !cdir nodep      !NEC
  1078) !ocl novrec      !Fujitsu
  1079)          do ij = 1, icells
  1080)             ! area participating > area removed
  1081)             aksum(ij) = aksum(ij)  &
  1082)                        + apartic(ij,n) * (c1 - c1/krdg(ij,n)) 
  1083)          enddo
  1084)       enddo
  1085) 
  1086)       end subroutine ridge_itd
