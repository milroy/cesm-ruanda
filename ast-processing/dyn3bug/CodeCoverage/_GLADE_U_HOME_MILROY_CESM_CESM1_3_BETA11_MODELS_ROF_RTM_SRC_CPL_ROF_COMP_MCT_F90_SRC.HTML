    67)   subroutine rof_init_mct( EClock, cdata_r, x2r_r, r2x_r, NLFilename)
    68) 
    69)     !---------------------------------------------------------------------------
    70)     ! DESCRIPTION:
    71)     ! Initialize runoff model and obtain relevant atmospheric model arrays
    72)     ! back from (i.e. albedos, surface temperature and snow cover over land).
    73)     !
    74)     ! !ARGUMENTS:
    75)     type(ESMF_Clock),           intent(inout) :: EClock     ! Input synchronization clock
    76)     type(seq_cdata),            intent(inout) :: cdata_r    ! Input runoff-model driver data
    77)     type(mct_aVect) ,           intent(inout) :: x2r_r      ! River import state
    78)     type(mct_aVect),            intent(inout) :: r2x_r      ! River export state
    79)     character(len=*), optional, intent(in)    :: NLFilename ! Namelist filename to read
    80)     !
    81)     ! !LOCAL VARIABLES:
    82)     integer :: ROFID                                 ! rof identifyer
    83)     integer :: mpicom_rof                            ! mpi communicator
    84)     type(mct_gsMap),         pointer :: gsMap_rof    ! runoff model MCT GS map
    85)     type(mct_gGrid),         pointer :: dom_r        ! runoff model domain
    86)     type(seq_infodata_type), pointer :: infodata     ! CESM driver level info data
    87)     integer :: lsize                                 ! size of attribute vector
    88)     integer :: g,i,j,n                               ! indices
    89)     logical :: exists                                ! true if file exists
    90)     integer :: nsrest                                ! restart type
    91)     integer :: ref_ymd                               ! reference date (YYYYMMDD)
    92)     integer :: ref_tod                               ! reference time of day (sec)
    93)     integer :: start_ymd                             ! start date (YYYYMMDD)
    94)     integer :: start_tod                             ! start time of day (sec)
    95)     integer :: stop_ymd                              ! stop date (YYYYMMDD)
    96)     integer :: stop_tod                              ! stop time of day (sec)
    97)     logical :: brnch_retain_casename                 ! flag if should retain the case name on a branch start type
    98)     integer :: lbnum                                 ! input to memory diagnostic
    99)     integer :: shrlogunit,shrloglev                  ! old values for log unit and log level
   100)     integer :: begr, endr
   101)     character(len=SHR_KIND_CL) :: caseid             ! case identifier name
   102)     character(len=SHR_KIND_CL) :: ctitle             ! case description title
   103)     character(len=SHR_KIND_CL) :: starttype          ! start-type (startup, continue, branch, hybrid)
   104)     character(len=SHR_KIND_CL) :: calendar           ! calendar type name
   105)     character(len=SHR_KIND_CL) :: hostname           ! hostname of machine running on
   106)     character(len=SHR_KIND_CL) :: version            ! Model version
   107)     character(len=SHR_KIND_CL) :: username           ! user running the model
   108)     character(len=32), parameter :: sub = 'rof_init_mct'
   109)     character(len=*),  parameter :: format = "('("//trim(sub)//") :',A)"
   110)     !---------------------------------------------------------------------------
   111) 
   112)     ! Obtain cdata_r (initalized in ccsm_comp_mod.F90 in the call to 
   113)     ! seq_cdata_init for cdata_rr)
   114)     call seq_cdata_setptrs(cdata_r, ID=ROFID, mpicom=mpicom_rof, &
   115)          gsMap=gsMap_rof, dom=dom_r, infodata=infodata)
   116) 
   117)     ! Determine attriute vector indices
   118)     call rtm_cpl_indices_set()
   119) 
   120)     ! Initialize rtm MPI communicator 
   121)     call RtmSpmdInit(mpicom_rof)
   122) 
   123) #if (defined _MEMTRACE)
   124)     if(masterproc) then
   125)        lbnum=1
   126)        call memmon_dump_fort('memmon.out','rof_init_mct:start::',lbnum)
   127)     endif
   128) #endif                      
   129) 
   130)     ! Initialize io log unit
   131)     inst_name   = seq_comm_name(ROFID)
   132)     inst_index  = seq_comm_inst(ROFID)
   133)     inst_suffix = seq_comm_suffix(ROFID)
   134) 
   135)     call shr_file_getLogUnit (shrlogunit)
   136)     if (masterproc) then
   137)        inquire(file='rof_modelio.nml'//trim(inst_suffix),exist=exists)
   138)        if (exists) then
   139)           iulog = shr_file_getUnit()
   140)           call shr_file_setIO('rof_modelio.nml'//trim(inst_suffix),iulog)
   141)        end if
   142)        write(iulog,format) "RTM land model initialization"
   143)     else
   144)        iulog = shrlogunit
   145)     end if
   146)     
   147)     call shr_file_getLogLevel(shrloglev)
   148)     call shr_file_setLogUnit (iulog)
   149)     
   150)     ! Initialize rtm
   151)     call seq_timemgr_EClockGetData(EClock,                               &
   152)                                    start_ymd=start_ymd,                  &
   153)                                    start_tod=start_tod, ref_ymd=ref_ymd, &
   154)                                    ref_tod=ref_tod, stop_ymd=stop_ymd,   &
   155)                                    stop_tod=stop_tod,                    &
   156)                                    calendar=calendar )
   157) 
   158)     call seq_infodata_GetData(infodata, case_name=caseid,                  &
   159)                               case_desc=ctitle, start_type=starttype,      &
   160)                               brnch_retain_casename=brnch_retain_casename, &
   161)                               model_version=version,                       &
   162)                               hostname=hostname, username=username)
   163) 
   164)     call timemgr_setup(calendar_in=calendar,                           &
   165)                        start_ymd_in=start_ymd, start_tod_in=start_tod, &
   166)                        ref_ymd_in=ref_ymd, ref_tod_in=ref_tod,         &
   167)                        stop_ymd_in=stop_ymd, stop_tod_in=stop_tod)  
   168) 
   169)     if (     trim(starttype) == trim(seq_infodata_start_type_start)) then
   170)        nsrest = nsrStartup
   171)     else if (trim(starttype) == trim(seq_infodata_start_type_cont) ) then
   172)        nsrest = nsrContinue
   173)     else if (trim(starttype) == trim(seq_infodata_start_type_brnch)) then
   174)        nsrest = nsrBranch
   175)     else
   176)        call shr_sys_abort( sub//' ERROR: unknown starttype' )
   177)     end if
   178) 
   179)     call RtmVarSet(caseid_in=caseid, ctitle_in=ctitle,             &
   180)                    brnch_retain_casename_in=brnch_retain_casename, &
   181)                    nsrest_in=nsrest, version_in=version,           &
   182)                    hostname_in=hostname, username_in=username)
   183) 
   184)     ! Read namelist, grid and surface data
   185)     call Rtmini()
   186) 
   187)     if (rtm_active) then
   188)        ! Initialize memory for input state
   189)        begr = runoff%begr
   190)        endr = runoff%endr
   191)        allocate (totrunin(begr:endr,nt_rtm))
   192)        
   193)        ! Initialize rof gsMap for ocean rof and land rof
   194)        call rof_SetgsMap_mct( mpicom_rof, ROFID, gsMap_rof)
   195)        
   196)        ! Initialize rof domain
   197)        lsize = mct_gsMap_lsize(gsMap_rof, mpicom_rof)
   198)        call rof_domain_mct( lsize, gsMap_rof, dom_r )
   199)        
   200)        ! Initialize lnd -> rtm attribute vector		
   201)        call mct_aVect_init(x2r_r, rList=seq_flds_x2r_fields, lsize=lsize)
   202)        call mct_aVect_zero(x2r_r)
   203)        
   204)        ! Initialize rtm -> ocn attribute vector		
   205)        call mct_aVect_init(r2x_r, rList=seq_flds_r2x_fields, lsize=lsize)
   206)        call mct_aVect_zero(r2x_r) 
   207)        
   208)        ! Create mct river runoff export state
   209)        call rof_export(r2x_r%rattr)
   210)     end if
   211) 
   212)     ! Fill in infodata
   213)     call seq_infodata_PutData( infodata, rof_present=rtm_active, &
   214)          rofice_present=.false., rof_prognostic=rtm_active, &
   215)          rof_nx = rtmlon, rof_ny = rtmlat)
   216)     call seq_infodata_PutData( infodata, flood_present=flood_active)
   217) 
   218)     ! Reset shr logging to original values
   219)     call shr_file_setLogUnit (shrlogunit)
   220)     call shr_file_setLogLevel(shrloglev)
   221) 
   222) #if (defined _MEMTRACE)
   223)     if(masterproc) then
   224)        write(iulog,*) TRIM(Sub) // ':end::'
   225)        lbnum=1
   226)        call memmon_dump_fort('memmon.out','rof_int_mct:end::',lbnum)
   227)        call memmon_reset_addr()
   228)     endif
   229) #endif
   230) 
   231)   end subroutine rof_init_mct

   235)   subroutine rof_run_mct( EClock, cdata_r, x2r_r, r2x_r)
   236) 
   237)     !-------------------------------------------------------
   238)     ! DESCRIPTION:
   239)     ! Run runoff model
   240) 
   241)     ! ARGUMENTS:
   242)     implicit none
   243)     type(ESMF_Clock) , intent(inout) :: EClock    ! Input synchronization clock from driver
   244)     type(seq_cdata)  , intent(inout) :: cdata_r   ! Input driver data for runoff model
   245)     type(mct_aVect)  , intent(inout) :: x2r_r     ! Import state from runoff model
   246)     type(mct_aVect)  , intent(inout) :: r2x_r     ! Export state from runoff model
   247) 
   248)     ! LOCAL VARIABLES:
   249)     integer :: ymd_sync, ymd              ! current date (YYYYMMDD)
   250)     integer :: yr_sync, yr                ! current year
   251)     integer :: mon_sync, mon              ! current month
   252)     integer :: day_sync, day              ! current day
   253)     integer :: tod_sync, tod              ! current time of day (sec)
   254)     logical :: rstwr                      ! .true. ==> write restart file before returning
   255)     logical :: nlend                      ! .true. ==> signaling last time-step
   256)     integer :: shrlogunit,shrloglev       ! old values for share log unit and log level
   257)     integer :: lsize                      ! local size
   258)     integer :: lbnum                      ! input to memory diagnostic
   259)     integer :: g,i                        ! indices
   260)     type(mct_gGrid),        pointer :: dom_r    ! runoff model domain
   261)     type(seq_infodata_type),pointer :: infodata ! CESM information from the driver
   262)     real(r8),               pointer :: data(:)  ! temporary
   263)     character(len=32)               :: rdate    ! date char string for restart file names
   264)     character(len=32), parameter    :: sub = "rof_run_mct"
   265)     !-------------------------------------------------------
   266) 
   267) #if (defined _MEMTRACE)
   268)     if(masterproc) then
   269)        lbnum=1
   270)        call memmon_dump_fort('memmon.out','rof_run_mct:start::',lbnum)
   271)     endif
   272) #endif
   273) 
   274)     if (.not.rtm_active) return
   275) 
   276)     ! Reset shr logging to my log file
   277)     call shr_file_getLogUnit (shrlogunit)
   278)     call shr_file_getLogLevel(shrloglev)
   279)     call shr_file_setLogUnit (iulog)
   280) 
   281)     ! Determine time of next atmospheric shortwave calculation
   282)     call seq_timemgr_EClockGetData(EClock, &
   283)          curr_ymd=ymd, curr_tod=tod_sync,  &
   284)          curr_yr=yr_sync, curr_mon=mon_sync, curr_day=day_sync)
   285) 
   286)     ! Map MCT to land data type (output is totrunin - module variable) 
   287)     call t_startf ('lc_rof_import')
   288)     call rof_import(x2r_r%rattr, totrunin=totrunin)
   289)     call t_stopf ('lc_rof_import')
   290) 
   291)     ! First advance rtm time step
   292)     write(rdate,'(i4.4,"-",i2.2,"-",i2.2,"-",i5.5)') yr_sync,mon_sync,day_sync,tod_sync
   293)     nlend = seq_timemgr_StopAlarmIsOn( EClock )
   294)     rstwr = seq_timemgr_RestartAlarmIsOn( EClock )
   295)     call advance_timestep()
   296) 
   297)     ! Run rtm (input is totrunin, output is runoff%runoff)
   298)     call Rtmrun(totrunin, rstwr, nlend, rdate)
   299) 
   300)     ! Map roff data to coupler datatype (input is runoff%runoff, output is r2x_r%rattr)
   301)     call t_startf ('lc_rof_export')
   302)     call rof_export(r2x_r%rattr)
   303)     call t_stopf ('lc_rof_export')
   304) 
   305)     ! Check that internal clock is in sync with master clock
   306)     call get_curr_date( yr, mon, day, tod )
   307)     ymd = yr*10000 + mon*100 + day
   308)     tod = tod
   309)     if ( .not. seq_timemgr_EClockDateInSync( EClock, ymd, tod ) )then
   310)        call seq_timemgr_EclockGetData( EClock, curr_ymd=ymd_sync, curr_tod=tod_sync )
   311)        write(iulog,*)' rtm ymd=',ymd     ,'  rtm tod= ',tod
   312)        write(iulog,*)'sync ymd=',ymd_sync,' sync tod= ',tod_sync
   313)        call shr_sys_abort( sub//":: RTM clock is not in sync with Master Sync clock" )
   314)     end if
   315)     
   316)     ! Reset shr logging to my original values
   317)     call shr_file_setLogUnit (shrlogunit)
   318)     call shr_file_setLogLevel(shrloglev)
   319)   
   320) #if (defined _MEMTRACE)
   321)     if(masterproc) then
   322)        lbnum=1
   323)        call memmon_dump_fort('memmon.out','rof_run_mct:end::',lbnum)
   324)        call memmon_reset_addr()
   325)     endif
   326) #endif
   327) 
   328)   end subroutine rof_run_mct

   364)   subroutine rof_SetgsMap_mct( mpicom_r, ROFID, gsMap_rof)
   365) 
   366)     !-----------------------------------------------------
   367)     ! DESCRIPTION:
   368)     ! Set the MCT GS map for the runoff model
   369)     !
   370)     ! ARGUMENTS:
   371)     implicit none
   372)     integer        , intent(in)    :: mpicom_r      ! MPI communicator for rof model
   373)     integer        , intent(in)    :: ROFID         ! Land model identifier
   374)     type(mct_gsMap), intent(inout) :: gsMap_rof     ! MCT gsmap for runoff -> land data
   375)     !
   376)     ! LOCAL VARIABLES
   377)     integer,allocatable :: gindex(:)         ! indexing for runoff grid cells
   378)     integer :: n, ni                         ! indices
   379)     integer :: lsize,gsize                   ! size of runoff data and number of grid cells
   380)     integer :: begr, endr                    ! beg, end runoff indices
   381)     integer :: ier                           ! error code
   382)     character(len=32), parameter :: sub = 'rof_SetgsMap_mct'
   383)     !-----------------------------------------------------
   384) 
   385)     begr  = runoff%begr
   386)     endr  = runoff%endr
   387)     lsize = runoff%lnumr
   388)     gsize = rtmlon*rtmlat
   389) 
   390)     ! Check 
   391)     ni = 0
   392)     do n = begr,endr
   393)        ni = ni + 1
   394)        if (ni > lsize) then
   395)           write(iulog,*) sub, ' : ERROR runoff count',n,ni,runoff%lnumr
   396)           call shr_sys_abort( sub//' ERROR: runoff > expected' )
   397)        endif
   398)     end do
   399)     if (ni /= lsize) then
   400)        write(iulog,*) sub, ' : ERROR runoff total count',ni,runoff%lnumr
   401)        call shr_sys_abort( sub//' ERROR: runoff not equal to expected' )
   402)     endif
   403) 
   404)     ! Determine gsmap_rof
   405)     allocate(gindex(lsize),stat=ier)
   406)     ni = 0
   407)     do n = begr,endr
   408)        ni = ni + 1
   409)        gindex(ni) = runoff%gindex(n)
   410)     end do
   411)     call mct_gsMap_init( gsMap_rof, gindex, mpicom_r, ROFID, lsize, gsize )
   412)     deallocate(gindex)
   413) 
   414)   end subroutine rof_SetgsMap_mct

   418)   subroutine rof_domain_mct( lsize, gsMap_r, dom_r )
   419) 
   420)     !-----------------------------------------------------
   421)     !
   422)     ! !DESCRIPTION:
   423)     ! Send the runoff model domain information to the coupler
   424)     !
   425)     ! !ARGUMENTS:
   426)     implicit none
   427)     integer        , intent(in)    :: lsize       ! Size of runoff domain information
   428)     type(mct_gsMap), intent(inout) :: gsMap_r     ! Output MCT GS map for runoff model
   429)     type(mct_ggrid), intent(out)   :: dom_r       ! Domain information from the runoff model
   430)     !
   431)     ! LOCAL VARIABLES
   432)     integer :: n, ni              ! index
   433)     integer , pointer :: idata(:) ! temporary
   434)     real(r8), pointer :: data(:)  ! temporary
   435)     real(r8) :: re = SHR_CONST_REARTH*0.001_r8 ! radius of earth (km)
   436)     character(len=32), parameter :: sub = 'rof_domain_mct'
   437)     !-----------------------------------------------------
   438) 
   439)     ! lat/lon in degrees,  area in radians^2, mask is 1 (land), 0 (non-land)
   440)     ! Note that in addition land carries around landfrac for the purposes of domain checking
   441)     call mct_gGrid_init( GGrid=dom_r, CoordChars=trim(seq_flds_dom_coord), &
   442)       OtherChars=trim(seq_flds_dom_other), lsize=lsize )
   443) 
   444)     ! Allocate memory
   445)     allocate(data(lsize))
   446) 
   447)     ! Determine global gridpoint number attribute, GlobGridNum, which is set automatically by MCT
   448)     call mct_gsMap_orderedPoints(gsMap_r, iam, idata)
   449)     call mct_gGrid_importIAttr(dom_r,'GlobGridNum',idata,lsize)
   450) 
   451)     ! Determine domain (numbering scheme is: West to East and South to North to South pole)
   452)     ! Initialize attribute vector with special value
   453)     data(:) = -9999.0_R8 
   454)     call mct_gGrid_importRAttr(dom_r,"lat"  ,data,lsize) 
   455)     call mct_gGrid_importRAttr(dom_r,"lon"  ,data,lsize) 
   456)     call mct_gGrid_importRAttr(dom_r,"area" ,data,lsize) 
   457)     call mct_gGrid_importRAttr(dom_r,"aream",data,lsize) 
   458)     data(:) = 0.0_R8     
   459)     call mct_gGrid_importRAttr(dom_r,"mask" ,data,lsize) 
   460) 
   461)     ! Determine bounds numbering consistency
   462)     ni = 0
   463)     do n = runoff%begr,runoff%endr
   464)        ni = ni + 1
