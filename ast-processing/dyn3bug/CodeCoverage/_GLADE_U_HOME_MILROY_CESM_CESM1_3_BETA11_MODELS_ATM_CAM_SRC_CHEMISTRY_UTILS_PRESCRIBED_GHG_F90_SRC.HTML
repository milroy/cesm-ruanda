   210)   subroutine prescribed_ghg_adv( state, pbuf2d)
   211) 
   212)     use tracer_data,  only : advance_trcdata
   213)     use physics_types,only : physics_state
   214)     use ppgrid,       only : begchunk, endchunk
   215)     use ppgrid,       only : pcols, pver
   216)     use string_utils, only : to_lower, GLC
   217)     use cam_history,  only : outfld
   218)     use physconst,    only : mwdry                ! molecular weight dry air ~ kg/kmole
   219)     use physconst,    only : boltz                ! J/K/molecule
   220)     
   221)     use physics_buffer, only : physics_buffer_desc, pbuf_get_field, pbuf_set_field, pbuf_get_chunk
   222) 
   223)     implicit none
   224) 
   225)     type(physics_state), intent(in)    :: state(begchunk:endchunk)                 
   226)     
   227)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   228) 
   229)     type(physics_buffer_desc), pointer :: pbuf_chnk(:)
   230)     integer :: ind,c,ncol,i
   231)     real(r8) :: to_mmr(pcols,pver)
   232)     real(r8) :: outdata(pcols,pver)
   233)     real(r8),pointer :: tmpptr(:,:)
   234) 
   235)     character(len=32) :: units_str
   236) 
   237)     if( .not. has_prescribed_ghg ) return
   238) 
   239)     call advance_trcdata( fields, file, state, pbuf2d )
   240)     
   241)     ! set the correct units and invoke history outfld
   242)     do i = 1,number_flds
   243)        ind = index_map(i)
   244) 
   245)        units_str = trim(to_lower(trim(fields(i)%units(:GLC(fields(i)%units)))))
   246) 
   247) !$OMP PARALLEL DO PRIVATE (C, NCOL, OUTDATA, TO_MMR, tmpptr, pbuf_chnk)
   248)        do c = begchunk,endchunk
   249)           ncol = state(c)%ncol
   250) 
   251)           select case ( units_str )
   252)           case ("molec/cm3","/cm3","molecules/cm3","cm^-3","cm**-3")
   253)              to_mmr(:ncol,:) = (molmass(ind)*1.e6_r8*boltz*state(c)%t(:ncol,:))/(mwdry*state(c)%pmiddry(:ncol,:))
   254)           case ('kg/kg','mmr')
   255)              to_mmr(:ncol,:) = 1._r8
   256)           case ('mol/mol','mole/mole','vmr','fraction')
   257)              to_mmr(:ncol,:) = molmass(ind)/mwdry
   258)           case default
   259)              print*, 'prescribed_ghg_adv: units = ',trim(fields(i)%units) ,' are not recognized'
   260)              call endrun('prescribed_ghg_adv: units are not recognized')
   261)           end select
   262) 
   263)           pbuf_chnk => pbuf_get_chunk(pbuf2d, c)
   264)           call pbuf_get_field(pbuf_chnk, fields(i)%pbuf_ndx, tmpptr )
   265) 
   266)           tmpptr(:ncol,:) = tmpptr(:ncol,:)*to_mmr(:ncol,:)
   267) 
   268)           outdata(:ncol,:) = tmpptr(:ncol,:) 
   269)           call outfld( fields(1)%fldnam, outdata(:ncol,:), ncol, state(c)%lchnk )
   270) 
   271)        enddo
   272)     enddo
   273) 
   274)   end subroutine prescribed_ghg_adv
