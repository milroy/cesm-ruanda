   101) ! ------- Description -------
   102) 
   103) ! This program is the driver for RRTMG_SW, the AER SW radiation model for 
   104) !  application to GCMs, that has been adapted from RRTM_SW for improved
   105) !  efficiency and to provide fractional cloudiness and cloud overlap
   106) !  capability using McICA.
   107) !
   108) ! Note: The call to RRTMG_SW_INI should be moved to the GCM initialization 
   109) !  area, since this has to be called only once. 
   110) !
   111) ! This routine
   112) !    b) calls INATM_SW to read in the atmospheric profile;
   113) !       all layering in RRTMG is ordered from surface to toa. 
   114) !    c) calls CLDPRMC_SW to set cloud optical depth for McICA based
   115) !       on input cloud properties
   116) !    d) calls SETCOEF_SW to calculate various quantities needed for 
   117) !       the radiative transfer algorithm
   118) !    e) calls SPCVMC to call the two-stream model that in turn 
   119) !       calls TAUMOL to calculate gaseous optical depths for each 
   120) !       of the 16 spectral bands and to perform the radiative transfer
   121) !       using McICA, the Monte-Carlo Independent Column Approximation,
   122) !       to represent sub-grid scale cloud variability
   123) !    f) passes the calculated fluxes and cooling rates back to GCM
   124) !
   125) ! Two modes of operation are possible:
   126) !     The mode is chosen by using either rrtmg_sw.nomcica.f90 (to not use
   127) !     McICA) or rrtmg_sw.f90 (to use McICA) to interface with a GCM.
   128) !
   129) !    1) Standard, single forward model calculation (imca = 0); this is 
   130) !       valid only for clear sky or fully overcast clouds
   131) !    2) Monte Carlo Independent Column Approximation (McICA, Pincus et al., 
   132) !       JC, 2003) method is applied to the forward model calculation (imca = 1)
   133) !       This method is valid for clear sky or partial cloud conditions.
   134) !
   135) ! This call to RRTMG_SW must be preceeded by a call to the module
   136) !     mcica_subcol_gen_sw.f90 to run the McICA sub-column cloud generator,
   137) !     which will provide the cloud physical or cloud optical properties
   138) !     on the RRTMG quadrature point (ngptsw) dimension.
   139) !
   140) ! Two methods of cloud property input are possible:
   141) !     Cloud properties can be input in one of two ways (controlled by input 
   142) !     flags inflag, iceflag and liqflag; see text file rrtmg_sw_instructions
   143) !     and subroutine rrtmg_sw_cldprop.f90 for further details):
   144) !
   145) !    1) Input cloud fraction, cloud optical depth, single scattering albedo 
   146) !       and asymmetry parameter directly (inflgsw = 0)
   147) !    2) Input cloud fraction and cloud physical properties: ice fracion,
   148) !       ice and liquid particle sizes (inflgsw = 1 or 2);  
   149) !       cloud optical properties are calculated by cldprop or cldprmc based
   150) !       on input settings of iceflgsw and liqflgsw
   151) !
   152) ! Two methods of aerosol property input are possible:
   153) !     Aerosol properties can be input in one of two ways (controlled by input 
   154) !     flag iaer, see text file rrtmg_sw_instructions for further details):
   155) !
   156) !    1) Input aerosol optical depth, single scattering albedo and asymmetry
   157) !       parameter directly by layer and spectral band (iaer=10)
   158) !    2) Input aerosol optical depth and 0.55 micron directly by layer and use
   159) !       one or more of six ECMWF aerosol types (iaer=6)
   160) !
   161) !
   162) ! ------- Modifications -------
   163) !
   164) ! This version of RRTMG_SW has been modified from RRTM_SW to use a reduced
   165) ! set of g-point intervals and a two-stream model for application to GCMs.

   167) !-- Original version (derived from RRTM_SW)
   168) !     2002: AER. Inc.

   173) !-- Total number of g-points reduced from 224 to 112.  Original

   175) !   and in file rrtmg_sw_init.f90.
   176) !     Apr 2004: M. J. Iacono, AER, Inc.
   177) !-- Modifications to include output for direct and diffuse 
   178) !   downward fluxes.  There are output as "true" fluxes without
   179) !   any delta scaling applied.  Code can be commented to exclude

   756)       real(kind=r8) function earth_sun(idn)
   757) !*************************************************************************
   758) !
   759) !  Purpose: Function to calculate the correction factor of Earth's orbit
   760) !  for current day of the year
   761) 
   762) !  idn        : Day of the year
   763) !  earth_sun  : square of the ratio of mean to actual Earth-Sun distance
   764)

   768) 
   769)       integer, intent(in) :: idn
   770) 
   771)       real(kind=r8) :: gamma
   772) 
   773)       gamma = 2._r8*pi*(idn-1)/365._r8
   774) 
   775) ! Use Iqbal's equation 1.2.1
   776) 
   777)       earth_sun = 1.000110_r8 + .034221_r8 * cos(gamma) + .001289_r8 * sin(gamma) + &
   778)                    .000719_r8 * cos(2._r8*gamma) + .000077_r8 * sin(2._r8*gamma)
   779) 
   780)       end function earth_sun
