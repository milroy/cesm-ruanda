   113) subroutine gw_drag_readnl(nlfile)
   114) 
   115)   use namelist_utils,  only: find_group_name
   116)   use units,           only: getunit, freeunit
   117)   use mpishorthand
   118) 
   119)   ! File containing namelist input.
   120)   character(len=*), intent(in) :: nlfile
   121) 
   122)   ! Local variables
   123)   integer :: unitn, ierr
   124)   character(len=*), parameter :: subname = 'gw_drag_readnl'
   125) 
   126)   ! More specific name for dc to prevent a name clash or confusion in the
   127)   ! namelist.
   128)   real(r8) :: gw_dc = unset_r8
   129) 
   130)   namelist /gw_drag_nl/ pgwv, gw_dc, tau_0_ubc, effgw_beres, effgw_cm, &
   131)        effgw_oro, fcrit2, frontgfc, gw_drag_file, taubgnd
   132)   !----------------------------------------------------------------------
   133) 
   134)   if (masterproc) then
   135)      unitn = getunit()
   136)      open( unitn, file=trim(nlfile), status='old' )
   137)      call find_group_name(unitn, 'gw_drag_nl', status=ierr)
   138)      if (ierr == 0) then
   139)         read(unitn, gw_drag_nl, iostat=ierr)
   140)         if (ierr /= 0) then
   141)            call endrun(subname // ':: ERROR reading namelist')
   142)         end if
   143)      end if
   144)      close(unitn)
   145)      call freeunit(unitn)
   146)   end if
   147) 
   148) #ifdef SPMD
   149)   ! Broadcast namelist variables
   150)   call mpibcast(pgwv,        1, mpiint, 0, mpicom)
   151)   call mpibcast(gw_dc,       1, mpir8,  0, mpicom)
   152)   call mpibcast(tau_0_ubc,   1, mpilog, 0, mpicom)
   153)   call mpibcast(effgw_beres, 1, mpir8,  0, mpicom)
   154)   call mpibcast(effgw_cm,    1, mpir8,  0, mpicom)
   155)   call mpibcast(effgw_oro,   1, mpir8,  0, mpicom)
   156)   call mpibcast(fcrit2,      1, mpir8,  0, mpicom)
   157)   call mpibcast(frontgfc,    1, mpir8,  0, mpicom)
   158)   call mpibcast(taubgnd,     1, mpir8,  0, mpicom)
   159)   call mpibcast(gw_drag_file, len(gw_drag_file), mpichar, 0, mpicom)
   160) #endif
   161) 
   162)   dc = gw_dc
   163) 
   164)   ! Check if pgwv was set.
   165)   if (pgwv < 0) then
   166)      call endrun('gw_drag_readnl: pgwv must be set via the namelist and &
   167)           &non-negative.')
   168)   end if
   169) 
   170)   ! Check if dc was set.
   171)   if (dc == unset_r8) then
   172)      call endrun('gw_drag_readnl: dc must be set via the namelist')
   173)   end if
   174) 
   175)   ! Check if fcrit2 was set.
   176)   if (fcrit2 == unset_r8) then
   177)      call endrun('gw_drag_readnl: fcrit2 must be set via the namelist')
   178)   end if
   179) 
   180) end subroutine gw_drag_readnl

   467)            call add_default('HDEPTH   ', 1, ' ')
   468)            call add_default('MAXQ0    ', 1, ' ')
   469)         end if
   470) 
   471)      end if
   472) 
   473)      call addfld ('EKGWSPEC' ,'M2/S   ',pver+1, 'A', &

   475)

   556) subroutine gw_tend(state, sgh, pbuf, dt, ptend, cam_in)
   557)   !-----------------------------------------------------------------------
   558)   ! Interface for multiple gravity wave drag parameterization.
   559)   !-----------------------------------------------------------------------
   560)   use physics_buffer, only: physics_buffer_desc, pbuf_get_field
   561)   use camsrfexch, only: cam_in_t
   562)   ! Location-dependent cpair
   563)   use physconst,  only: cpairv
   564)   use gw_common,  only: gw_prof, momentum_energy_conservation, &
   565)        gw_drag_prof
   566)   use gw_oro,     only: gw_oro_src
   567)   use gw_front,   only: gw_cm_src
   568)   use gw_convect, only: gw_beres_src
   569)   !------------------------------Arguments--------------------------------
   570)   type(physics_state), intent(in) :: state      ! physics state structure
   571)   ! Standard deviation of orography.
   572)   real(r8), intent(in) :: sgh(pcols)
   573)   type(physics_buffer_desc), pointer :: pbuf(:) ! Physics buffer
   574)   real(r8), intent(in) :: dt                    ! time step
   575)   ! Parameterization net tendencies.
   576)   type(physics_ptend), intent(out):: ptend
   577)   type(cam_in_t), intent(in) :: cam_in
   578) 
   579)   !---------------------------Local storage-------------------------------
   580)   integer :: lchnk                  ! chunk identifier
   581)   integer :: ncol                   ! number of atmospheric columns
   582) 
   583)   integer :: k                      ! loop index
   584) 
   585)   real(r8) :: ttgw(state%ncol,pver) ! temperature tendency
   586)   real(r8) :: utgw(state%ncol,pver) ! zonal wind tendency
   587)   real(r8) :: vtgw(state%ncol,pver) ! meridional wind tendency
   588) 
   589)   real(r8) :: ni(state%ncol,0:pver) ! interface Brunt-Vaisalla frequency
   590)   real(r8) :: nm(state%ncol,pver)   ! midpoint Brunt-Vaisalla frequency
   591)   real(r8) :: rhoi(state%ncol,0:pver)     ! interface density
   592)   real(r8) :: tau(state%ncol,-pgwv:pgwv,0:pver)  ! wave Reynolds stress
   593)   real(r8) :: tau0x(state%ncol)     ! c=0 sfc. stress (zonal)
   594)   real(r8) :: tau0y(state%ncol)     ! c=0 sfc. stress (meridional)
   595)   real(r8) :: ti(state%ncol,0:pver) ! interface temperature
   596)   real(r8) :: ubi(state%ncol,0:pver)! projection of wind at interfaces
   597)   real(r8) :: ubm(state%ncol,pver)  ! projection of wind at midpoints
   598)   real(r8) :: xv(state%ncol)        ! unit vector of source wind (x)
   599)   real(r8) :: yv(state%ncol)        ! unit vector of source wind (y)
   600) 
   601)   integer :: m                      ! dummy integers
   602)   real(r8) :: qtgw(state%ncol,pver,pcnst) ! constituents tendencies
   603) 
   604)   ! Reynolds stress for waves propagating in each cardinal direction.
   605)   real(r8) :: taucd(state%ncol,0:pver,4)
   606) 
   607)   ! gravity wave wind tendency for each wave
   608)   real(r8) :: gwut(state%ncol,pver,-pgwv:pgwv)
   609) 
   610)   ! Temperature tendencies from diffusion and kinetic energy.
   611)   real(r8) :: dttdf(state%ncol,pver)
   612)   real(r8) :: dttke(state%ncol,pver)
   613) 
   614)   ! spectrum phase speeds for each column
   615)   real(r8) :: c(state%ncol,-pgwv:pgwv)
   616) 
   617)   ! pbuf fields
   618)   ! Molecular diffusivity
   619)   real(r8), pointer :: kvt_in(:,:)
   620)   real(r8) :: kvtt(state%ncol,0:pver)
   621) 
   622)   ! Frontogenesis
   623)   real(r8), pointer :: frontgf(:,:)
   624)   real(r8), pointer :: frontga(:,:)
   625) 
   626)   ! Temperature change due to deep convection.
   627)   real(r8), pointer, dimension(:,:) :: ttend_dp
   628) 
   629)   ! Indices of gravity wave source and lowest level where wind tendencies
   630)   ! are allowed.
   631)   integer :: src_level(state%ncol)
   632)   integer :: tend_level(state%ncol)
   633) 
   634)   real(r8) :: hdepth(state%ncol)         ! heating depth array
   635)   real(r8) :: maxq0(state%ncol)          ! maximum heating rate array
   636) 
   637)   ! effective gw diffusivity at interfaces needed for output
   638)   real(r8) :: egwdffi(state%ncol,0:pver)
   639)   ! sum from the two types of spectral GW
   640)   real(r8) :: egwdffi_tot(state%ncol,0:pver)
   641) 
   642)   ! Which constituents are being affected by diffusion.
   643)   logical  :: lq(pcnst)
   644) 
   645)   ! Contiguous copies of state arrays.
   646)   real(r8) :: dse(state%ncol,pver)
   647)   real(r8) :: t(state%ncol,pver)
   648)   real(r8) :: u(state%ncol,pver)
   649)   real(r8) :: v(state%ncol,pver)
   650)   real(r8) :: q(state%ncol,pver,pcnst)
   651)   real(r8) :: pmid(state%ncol,pver)
   652)   real(r8) :: pint(state%ncol,pver+1)
   653)   real(r8) :: piln(state%ncol,pver+1)
   654)   real(r8) :: dpm(state%ncol,pver)
   655)   real(r8) :: rdpm(state%ncol,pver)
   656)   real(r8) :: zm(state%ncol,pver)
   657) 
   658)   !------------------------------------------------------------------------
   659) 
   660)   lchnk = state%lchnk
   661)   ncol  = state%ncol
   662) 
   663)   dse = state%s(:ncol,:)
   664)   t = state%t(:ncol,:)
   665)   u = state%u(:ncol,:)
   666)   v = state%v(:ncol,:)
   667)   q = state%q(:ncol,:,:)
   668)   pmid = state%pmid(:ncol,:)
   669)   pint = state%pint(:ncol,:)
   670)   piln = state%lnpint(:ncol,:)
   671)   dpm = state%pdel(:ncol,:)
   672)   rdpm = state%rpdel(:ncol,:)
   673)   zm = state%zm(:ncol,:)
   674) 
   675)   lq = .true.
   676)   call physics_ptend_init(ptend, state%psetcols, "Gravity wave drag", &
   677)        ls=.true., lu=.true., lv=.true., lq=lq)
   678) 
   679)   ! Profiles of background state variables
   680)   call gw_prof(ncol, cpair, &
   681)        t         , pmid         , pint      , rhoi      , ti        , &
   682)        nm        , ni)
   683) 
   684)   if (do_molec_diff) then
   685)      !--------------------------------------------------------
   686)      ! Initialize and calculate local molecular diffusivity
   687)      !--------------------------------------------------------
   688) 
   689)      call pbuf_get_field(pbuf, kvt_idx, kvt_in)  ! kvt_in(1:pcols,1:pver+1)
   690) 
   691)      ! Set kvtt from pbuf field; kvtt still needs a factor of 1/cpairv.
   692)      kvtt(:,0:pver) = kvt_in(:ncol,:)
   693) 
   694)      ! Use linear extrapolation of cpairv to top interface.
   695)      kvtt(:,ntop_molec-1) = kvtt(:,ntop_molec-1) / &
   696)           (1.5_r8*cpairv(:ncol,ntop_molec,lchnk) - &
   697)           0.5_r8*cpairv(:ncol,ntop_molec+1,lchnk))
   698) 
   699)      ! Interpolate cpairv to other interfaces.
   700)      do k = ntop_molec, nbot_molec-1
   701)         kvtt(:,k) = kvtt(:,k) / &
   702)              (cpairv(:ncol,k+1,lchnk)+cpairv(:ncol,k,lchnk)) * 2._r8
   703)      enddo
   704) 
   705)   end if
   706) 
   707)   !------------------------------------------------------------------------
   708)   ! Non-orographic background gravity wave spectra
   709)   !------------------------------------------------------------------------
   710) 
   711)   if (do_spectral_waves) then
   712) 
   713)      egwdffi_tot = 0._r8
   714) 
   715)      if (use_gw_convect) then
   716)         !------------------------------------------------------------------
   717)         ! Convective gravity waves (Beres scheme)
   718)         !------------------------------------------------------------------
   719) 
   720)         ! Set up heating
   721)         call pbuf_get_field(pbuf, ttend_dp_idx, ttend_dp)
   722) 
   723)         ! Determine wave sources for Beres04 scheme
   724)         call gw_beres_src(ncol, pgwv, state%lat(:ncol), u, v, ttend_dp, &
   725)              zm, src_level, tend_level, tau, ubm, ubi, xv, yv, c, &
   726)              hdepth, maxq0)
   727) 
   728)         ! Solve for the drag profile with Beres source spectrum.
   729)         call gw_drag_prof(ncol, pgwv, src_level, tend_level, .false., dt, &
   730)              state%lat(:ncol), t,    ti, pmid, pint, dpm,   rdpm, &
   731)              piln, rhoi,       nm,   ni, ubm,  ubi,  xv,    yv,   &
   732)              effgw_beres, c,   kvtt, q,  dse,  tau,  utgw,  vtgw, &
   733)              ttgw, qtgw,  taucd,     egwdffi,  gwut, dttdf, dttke)
   734) 
   735)         !  add the diffusion coefficients
   736)         do k = 0, pver
   737)            egwdffi_tot(:,k) = egwdffi_tot(:,k) + egwdffi(:,k)
   738)         end do
   739) 
   740)         ! Store constituents tendencies
   741)         do m=1, pcnst
   742)            do k = 1, pver
   743)               ptend%q(:ncol,k,m) = qtgw(:,k,m)
   744)            end do
   745)         end do
   746) 
   747)         ! add the momentum tendencies to the output tendency arrays
   748)         do k = 1, pver
   749)            ptend%u(:ncol,k) = utgw(:,k)
   750)            ptend%v(:ncol,k) = vtgw(:,k)
   751)            ptend%s(:ncol,k) = ttgw(:,k)
   752)         end do
   753) 
   754) 
   755)         ! C.-C. Chen, momentum & energy conservation
   756)         call momentum_energy_conservation(ncol, tend_level, dt, taucd, &
   757)              pint, dpm, u, v, ptend%u, ptend%v, ptend%s, utgw, vtgw, ttgw)
   758) 
   759)         call gw_spec_outflds(beres_pf, lchnk, ncol, pgwv, c, u, v, &
   760)              xv, yv, gwut, dttdf, dttke, tau(:,:,1:), utgw, vtgw, taucd)
   761) 
   762)         ! Note: This is probably redundant, because ZMDT is already being
   763)         ! output...
   764)         call outfld ('NETDT', ttend_dp, pcols, lchnk)
   765) 
   766)         ! Diagnostic outputs.
   767)         call outfld ('HDEPTH', hdepth, ncol, lchnk)
   768)         call outfld ('MAXQ0', maxq0, ncol, lchnk)
   769) 
   770)      end if
   771) 
   772)      if (use_gw_front) then
   773)         !------------------------------------------------------------------
   774)         ! Frontally generated gravity waves
   775)         !------------------------------------------------------------------
   776) 
   777)         ! Get frontogenesis physics buffer fields set by dynamics.
   778)         call pbuf_get_field(pbuf, frontgf_idx, frontgf)
   779)         call pbuf_get_field(pbuf, frontga_idx, frontga)
   780) 
   781)         ! Determine the wave source for C&M background spectrum
   782)         call gw_cm_src(ncol, pgwv, kbotbg, u, v, frontgf, &
   783)              src_level, tend_level, tau, ubm, ubi, xv, yv, c)
   784) 
   785)         ! Solve for the drag profile with C&M source spectrum.
   786)         call gw_drag_prof(ncol, pgwv, src_level, tend_level, .true., dt, &
   787)              state%lat(:ncol), t,    ti, pmid, pint, dpm,   rdpm, &
   788)              piln, rhoi,       nm,   ni, ubm,  ubi,  xv,    yv,   &
   789)              effgw_cm,    c,   kvtt, q,  dse,  tau,  utgw,  vtgw, &
   790)              ttgw, qtgw,  taucd,     egwdffi,  gwut, dttdf, dttke)
   791) 
   792)         !  add the diffusion coefficients
   793)         do k = 0, pver
   794)            egwdffi_tot(:,k) = egwdffi_tot(:,k) + egwdffi(:,k)
   795)         end do
   796) 
   797)         !Add the constituent tendencies
   798)         do m=1, pcnst
   799)            do k = 1, pver
   800)               ptend%q(:ncol,k,m) = ptend%q(:ncol,k,m) + qtgw(:,k,m)
   801)            end do
   802)         end do
   803) 
   804) 
   805)         ! add the momentum tendencies to the output tendency arrays
   806)         do k = 1, pver
   807)            ptend%u(:ncol,k) = ptend%u(:ncol,k) + utgw(:,k)
   808)            ptend%v(:ncol,k) = ptend%v(:ncol,k) + vtgw(:,k)
   809)            ptend%s(:ncol,k) = ptend%s(:ncol,k) + ttgw(:,k)
   810)         end do
   811) 
   812)         ! C.-C. Chen, momentum & energy conservation
   813)         call momentum_energy_conservation(ncol, tend_level, dt, taucd, &
   814)              pint, dpm, u, v, ptend%u, ptend%v, ptend%s, utgw, vtgw, ttgw)
   815) 
   816)         call gw_spec_outflds(cm_pf, lchnk, ncol, pgwv, c, u, v, &
   817)              xv, yv, gwut, dttdf, dttke, tau(:,:,1:), utgw, vtgw, taucd)
   818) 
   819)         call outfld ('FRONTGF', frontgf, pcols, lchnk)
   820)         call outfld ('FRONTGFA', frontga, pcols, lchnk)
   821) 
   822)      end if
   823) 
   824)      call outfld ('EKGWSPEC', egwdffi_tot , ncol, lchnk)
   825) 
   826)   end if
   827) 
   828)   if (use_gw_oro) then
   829)      !---------------------------------------------------------------------
   830)      ! Orographic stationary gravity waves
   831)      !---------------------------------------------------------------------
   832) 
   833)      ! Determine the orographic wave source
   834)      call gw_oro_src(ncol, &
   835)           u, v, t, sgh(:ncol), pmid, pint, dpm, zm, nm, &
   836)           src_level, tend_level, tau, ubm, ubi, xv, yv, c)
   837) 
   838)      ! Solve for the drag profile with orographic sources.
   839)      call gw_drag_prof(ncol, 0, src_level, tend_level, .false., dt, &
   840)           state%lat(:ncol), t,    ti, pmid, pint, dpm,   rdpm, &
   841)           piln, rhoi,       nm,   ni, ubm,  ubi,  xv,    yv,   &
   842)           effgw_oro,   c,   kvtt, q,  dse,  tau,  utgw,  vtgw, &
   843)           ttgw, qtgw,  taucd,     egwdffi,  gwut(:,:,0:0), dttdf, dttke)
   844) 
   845)      ! Add the orographic tendencies to the spectrum tendencies
   846)      ! Compute the temperature tendency from energy conservation
   847)      ! (includes spectrum).
   848)      do k = 1, pver
   849)         utgw(:,k) = utgw(:,k) * cam_in%landfrac(:ncol)
   850)         ptend%u(:ncol,k) = ptend%u(:ncol,k) + utgw(:,k)
   851)         vtgw(:,k) = vtgw(:,k) * cam_in%landfrac(:ncol)
   852)         ptend%v(:ncol,k) = ptend%v(:ncol,k) + vtgw(:,k)
   853)         ptend%s(:ncol,k) = ptend%s(:ncol,k) + ttgw(:,k) &
   854)              -(ptend%u(:ncol,k) * (u(:,k) + ptend%u(:ncol,k)*0.5_r8*dt) &
   855)              +ptend%v(:ncol,k) * (v(:,k) + ptend%v(:ncol,k)*0.5_r8*dt))
   856)         ttgw(:,k) = ttgw(:,k) &
   857)              -(ptend%u(:ncol,k) * (u(:,k) + ptend%u(:ncol,k)*0.5_r8*dt) &
   858)              +ptend%v(:ncol,k) * (v(:,k) + ptend%v(:ncol,k)*0.5_r8*dt))
   859)         ttgw(:,k) = ttgw(:,k) / cpairv(:ncol, k, lchnk)
   860)      end do
   861) 
   862)      do m = 1, pcnst
   863)         do k = 1, pver
   864)            ptend%q(:ncol,k,m) = ptend%q(:ncol,k,m) + qtgw(:,k,m)
   865)         end do
   866)      end do
   867) 
   868)      ! Write output fields to history file
   869)      call outfld('UTGWORO', utgw,  ncol, lchnk)
   870)      call outfld('VTGWORO', vtgw,  ncol, lchnk)
   871)      call outfld('TTGWORO', ttgw,  ncol, lchnk)
   872)      tau0x = tau(:,0,pver) * xv * effgw_oro
   873)      tau0y = tau(:,0,pver) * yv * effgw_oro
   874)      call outfld('TAUGWX', tau0x, ncol, lchnk)
   875)      call outfld('TAUGWY', tau0y, ncol, lchnk)
   876)      call outfld('SGH   ',   sgh,pcols, lchnk)
   877) 
   878)   end if
   879) 
   880)   ! Write total temperature tendency to history file
   881)   call outfld ('TTGW', ptend%s/cpairv(:,:,lchnk),  pcols, lchnk)
   882) 
   883) end subroutine gw_tend
