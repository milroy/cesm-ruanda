    43)   subroutine SnowWater(lbc, ubc, num_snowc, filter_snowc, &
    44)                        num_nosnowc, filter_nosnowc)
    45) !
    46) ! !DESCRIPTION:
    47) ! Evaluate the change of snow mass and the snow water onto soil.
    48) ! Water flow within snow is computed by an explicit and non-physical
    49) ! based scheme, which permits a part of liquid water over the holding
    50) ! capacity (a tentative value is used, i.e. equal to 0.033*porosity) to
    51) ! percolate into the underlying layer.  Except for cases where the
    52) ! porosity of one of the two neighboring layers is less than 0.05, zero
    53) ! flow is assumed. The water flow out of the bottom of the snow pack will
    54) ! participate as the input of the soil water and runoff.  This subroutine
    55) ! uses a filter for columns containing snow which must be constructed prior
    56) ! to being called.
    57) !
    58) ! !USES:
    59)     use clmtype
    60)     use clm_varcon  , only : denh2o, denice, wimp, ssi
    61)     use clm_time_manager, only : get_step_size
    62)     use clm_atmlnd        , only : clm_a2l
    63)     use SNICARMod         , only : scvng_fct_mlt_bcphi, scvng_fct_mlt_bcpho, &
    64)                                    scvng_fct_mlt_ocphi, scvng_fct_mlt_ocpho, &
    65)                                    scvng_fct_mlt_dst1,  scvng_fct_mlt_dst2,  &
    66)                                    scvng_fct_mlt_dst3,  scvng_fct_mlt_dst4
    67) !
    68) ! !ARGUMENTS:
    69)     implicit none
    70)     integer, intent(in) :: lbc, ubc                    ! column bounds
    71)     integer, intent(in) :: num_snowc                   ! number of snow points in column filter
    72)     integer, intent(in) :: filter_snowc(ubc-lbc+1)     ! column filter for snow points
    73)     integer, intent(in) :: num_nosnowc                 ! number of non-snow points in column filter
    74)     integer, intent(in) :: filter_nosnowc(ubc-lbc+1)   ! column filter for non-snow points
    75) !
    76) ! !CALLED FROM:
    77) !
    78) ! !REVISION HISTORY:
    79) ! 15 September 1999: Yongjiu Dai; Initial code
    80) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
    81) ! 15 November 2000: Mariana Vertenstein
    82) ! 2/26/02, Peter Thornton: Migrated to new data structures.
    83) ! 03/28/08, Mark Flanner: Added aerosol deposition and flushing with meltwater
    84) !
    85) ! !LOCAL VARIABLES:
    86) !
    87) ! local pointers to implicit in arguments
    88) !
    89)     integer , pointer :: snl(:)              !number of snow layers
    90)     logical , pointer :: do_capsnow(:)       !true => do snow capping
    91)     real(r8), pointer :: qflx_snow_melt(:)   !net snow melt
    92)     real(r8), pointer :: qflx_snomelt(:)     !snow melt (mm H2O /s)
    93)     real(r8), pointer :: qflx_rain_grnd(:)   !rain on ground after interception (mm H2O/s) [+]
    94)     real(r8), pointer :: qflx_sub_snow(:)    !sublimation rate from snow pack (mm H2O /s) [+]
    95)     real(r8), pointer :: qflx_evap_grnd(:)   !ground surface evaporation rate (mm H2O/s) [+]
    96)     real(r8), pointer :: qflx_dew_snow(:)    !surface dew added to snow pack (mm H2O /s) [+]
    97)     real(r8), pointer :: qflx_dew_grnd(:)    !ground surface dew formation (mm H2O /s) [+]
    98)     real(r8), pointer :: dz(:,:)             !layer depth (m)
    99) !
   100) ! local pointers to implicit out arguments
   101) !
   102)     real(r8), pointer :: qflx_top_soil(:)     !net water input into soil from top (mm/s)
   103) !
   104) ! local pointers to implicit inout arguments
   105) !
   106)     real(r8), pointer :: h2osoi_ice(:,:)     !ice lens (kg/m2)
   107)     real(r8), pointer :: h2osoi_liq(:,:)     !liquid water (kg/m2)
   108)     integer , pointer :: cgridcell(:)        ! columns's gridcell (col) 
   109)     real(r8), pointer :: mss_bcphi(:,:)      ! hydrophillic BC mass in snow (col,lyr) [kg]
   110)     real(r8), pointer :: mss_bcpho(:,:)      ! hydrophobic BC mass in snow (col,lyr) [kg]
   111)     real(r8), pointer :: mss_ocphi(:,:)      ! hydrophillic OC mass in snow (col,lyr) [kg]
   112)     real(r8), pointer :: mss_ocpho(:,:)      ! hydrophobic OC mass in snow (col,lyr) [kg]
   113)     real(r8), pointer :: mss_dst1(:,:)       ! mass of dust species 1 in snow (col,lyr) [kg]
   114)     real(r8), pointer :: mss_dst2(:,:)       ! mass of dust species 2 in snow (col,lyr) [kg]
   115)     real(r8), pointer :: mss_dst3(:,:)       ! mass of dust species 3 in snow (col,lyr) [kg]
   116)     real(r8), pointer :: mss_dst4(:,:)       ! mass of dust species 4 in snow (col,lyr) [kg]
   117)     real(r8), pointer :: flx_bc_dep_dry(:)   ! dry BC deposition (col) [kg m-2 s-1]
   118)     real(r8), pointer :: flx_bc_dep_wet(:)   ! wet BC deposition (col) [kg m-2 s-1]
   119)     real(r8), pointer :: flx_bc_dep(:)       ! total BC deposition (col) [kg m-2 s-1]
   120)     real(r8), pointer :: flx_bc_dep_pho(:)   ! hydrophobic BC deposition (col) [kg m-1 s-1]
   121)     real(r8), pointer :: flx_bc_dep_phi(:)   ! hydrophillic BC deposition (col) [kg m-1 s-1]
   122)     real(r8), pointer :: flx_oc_dep_dry(:)   ! dry OC deposition (col) [kg m-2 s-1]
   123)     real(r8), pointer :: flx_oc_dep_wet(:)   ! wet OC deposition (col) [kg m-2 s-1]
   124)     real(r8), pointer :: flx_oc_dep(:)       ! total OC deposition (col) [kg m-2 s-1]
   125)     real(r8), pointer :: flx_oc_dep_pho(:)   ! hydrophobic OC deposition (col) [kg m-1 s-1]
   126)     real(r8), pointer :: flx_oc_dep_phi(:)   ! hydrophillic OC deposition (col) [kg m-1 s-1]
   127)     real(r8), pointer :: flx_dst_dep_dry1(:) ! dry dust (species 1) deposition (col) [kg m-2 s-1]
   128)     real(r8), pointer :: flx_dst_dep_wet1(:) ! wet dust (species 1) deposition (col) [kg m-2 s-1]
   129)     real(r8), pointer :: flx_dst_dep_dry2(:) ! dry dust (species 2) deposition (col) [kg m-2 s-1]
   130)     real(r8), pointer :: flx_dst_dep_wet2(:) ! wet dust (species 2) deposition (col) [kg m-2 s-1]
   131)     real(r8), pointer :: flx_dst_dep_dry3(:) ! dry dust (species 3) deposition (col) [kg m-2 s-1]
   132)     real(r8), pointer :: flx_dst_dep_wet3(:) ! wet dust (species 3) deposition (col) [kg m-2 s-1]
   133)     real(r8), pointer :: flx_dst_dep_dry4(:) ! dry dust (species 4) deposition (col) [kg m-2 s-1]
   134)     real(r8), pointer :: flx_dst_dep_wet4(:) ! wet dust (species 4) deposition (col) [kg m-2 s-1]
   135)     real(r8), pointer :: flx_dst_dep(:)      ! total dust deposition (col) [kg m-2 s-1]
   136)     real(r8), pointer :: forc_aer(:,:)       ! aerosol deposition from atmosphere model (grd,aer) [kg m-1 s-1]
   137) !
   138) !
   139) ! !OTHER LOCAL VARIABLES:
   140) !EOP
   141) !
   142)     integer  :: c, j, fc                           !do loop/array indices
   143)     real(r8) :: dtime                              !land model time step (sec)
   144)     real(r8) :: qin(lbc:ubc)                       !water flow into the elmement (mm/s) 
   145)     real(r8) :: qout(lbc:ubc)                      !water flow out of the elmement (mm/s)
   146)     real(r8) :: wgdif                              !ice mass after minus sublimation
   147)     real(r8) :: vol_liq(lbc:ubc,-nlevsno+1:0)      !partial volume of liquid water in layer
   148)     real(r8) :: vol_ice(lbc:ubc,-nlevsno+1:0)      !partial volume of ice lens in layer
   149)     real(r8) :: eff_porosity(lbc:ubc,-nlevsno+1:0) !effective porosity = porosity - vol_ice
   150)     integer  :: g                                 ! gridcell loop index
   151)     real(r8) :: qin_bc_phi(lbc:ubc)               ! flux of hydrophilic BC into layer [kg]
   152)     real(r8) :: qout_bc_phi(lbc:ubc)              ! flux of hydrophilic BC out of layer [kg]
   153)     real(r8) :: qin_bc_pho(lbc:ubc)               ! flux of hydrophobic BC into layer [kg]
   154)     real(r8) :: qout_bc_pho(lbc:ubc)              ! flux of hydrophobic BC out of layer [kg]
   155)     real(r8) :: qin_oc_phi(lbc:ubc)               ! flux of hydrophilic OC into layer [kg]
   156)     real(r8) :: qout_oc_phi(lbc:ubc)              ! flux of hydrophilic OC out of layer [kg]
   157)     real(r8) :: qin_oc_pho(lbc:ubc)               ! flux of hydrophobic OC into layer [kg]
   158)     real(r8) :: qout_oc_pho(lbc:ubc)              ! flux of hydrophobic OC out of layer [kg]
   159)     real(r8) :: qin_dst1(lbc:ubc)                 ! flux of dust species 1 into layer [kg]
   160)     real(r8) :: qout_dst1(lbc:ubc)                ! flux of dust species 1 out of layer [kg]
   161)     real(r8) :: qin_dst2(lbc:ubc)                 ! flux of dust species 2 into layer [kg]
   162)     real(r8) :: qout_dst2(lbc:ubc)                ! flux of dust species 2 out of layer [kg]
   163)     real(r8) :: qin_dst3(lbc:ubc)                 ! flux of dust species 3 into layer [kg]
   164)     real(r8) :: qout_dst3(lbc:ubc)                ! flux of dust species 3 out of layer [kg]
   165)     real(r8) :: qin_dst4(lbc:ubc)                 ! flux of dust species 4 into layer [kg]
   166)     real(r8) :: qout_dst4(lbc:ubc)                ! flux of dust species 4 out of layer [kg]
   167)     real(r8) :: mss_liqice                        ! mass of liquid+ice in a layer
   168)  
   169) !-----------------------------------------------------------------------
   170) 
   171)     ! Assign local pointers to derived subtype components (column-level)
   172) 
   173)     snl              => cps%snl
   174)     do_capsnow       => cps%do_capsnow
   175)     qflx_snow_melt   => cwf%qflx_snow_melt
   176)     qflx_snomelt     => cwf%qflx_snomelt
   177)     qflx_rain_grnd   => pwf_a%qflx_rain_grnd
   178)     qflx_sub_snow    => pwf_a%qflx_sub_snow
   179)     qflx_evap_grnd   => pwf_a%qflx_evap_grnd
   180)     qflx_dew_snow    => pwf_a%qflx_dew_snow
   181)     qflx_dew_grnd    => pwf_a%qflx_dew_grnd
   182)     qflx_top_soil    => cwf%qflx_top_soil
   183)     dz               => cps%dz
   184)     h2osoi_ice       => cws%h2osoi_ice
   185)     h2osoi_liq       => cws%h2osoi_liq
   186)     cgridcell        => col%gridcell
   187)     mss_bcphi        => cps%mss_bcphi
   188)     mss_bcpho        => cps%mss_bcpho
   189)     mss_ocphi        => cps%mss_ocphi
   190)     mss_ocpho        => cps%mss_ocpho
   191)     mss_dst1         => cps%mss_dst1
   192)     mss_dst2         => cps%mss_dst2
   193)     mss_dst3         => cps%mss_dst3
   194)     mss_dst4         => cps%mss_dst4
   195)     flx_bc_dep       => cwf%flx_bc_dep
   196)     flx_bc_dep_wet   => cwf%flx_bc_dep_wet
   197)     flx_bc_dep_dry   => cwf%flx_bc_dep_dry
   198)     flx_bc_dep_phi   => cwf%flx_bc_dep_phi
   199)     flx_bc_dep_pho   => cwf%flx_bc_dep_pho
   200)     flx_oc_dep       => cwf%flx_oc_dep
   201)     flx_oc_dep_wet   => cwf%flx_oc_dep_wet
   202)     flx_oc_dep_dry   => cwf%flx_oc_dep_dry
   203)     flx_oc_dep_phi   => cwf%flx_oc_dep_phi
   204)     flx_oc_dep_pho   => cwf%flx_oc_dep_pho
   205)     flx_dst_dep      => cwf%flx_dst_dep
   206)     flx_dst_dep_wet1 => cwf%flx_dst_dep_wet1
   207)     flx_dst_dep_dry1 => cwf%flx_dst_dep_dry1
   208)     flx_dst_dep_wet2 => cwf%flx_dst_dep_wet2
   209)     flx_dst_dep_dry2 => cwf%flx_dst_dep_dry2
   210)     flx_dst_dep_wet3 => cwf%flx_dst_dep_wet3
   211)     flx_dst_dep_dry3 => cwf%flx_dst_dep_dry3
   212)     flx_dst_dep_wet4 => cwf%flx_dst_dep_wet4
   213)     flx_dst_dep_dry4 => cwf%flx_dst_dep_dry4
   214)     forc_aer         => clm_a2l%forc_aer
   215) 
   216)     ! Determine model time step
   217) 
   218)     dtime = get_step_size()
   219) 
   220)     ! Renew the mass of ice lens (h2osoi_ice) and liquid (h2osoi_liq) in the
   221)     ! surface snow layer resulting from sublimation (frost) / evaporation (condense)
   222) 
   223)     do fc = 1,num_snowc
   224)        c = filter_snowc(fc)
   225)        if (do_capsnow(c)) then
   226)           wgdif = h2osoi_ice(c,snl(c)+1) - qflx_sub_snow(c)*dtime
   227)           h2osoi_ice(c,snl(c)+1) = wgdif
   228)           if (wgdif < 0._r8) then
   229)              h2osoi_ice(c,snl(c)+1) = 0._r8
   230)              h2osoi_liq(c,snl(c)+1) = h2osoi_liq(c,snl(c)+1) + wgdif
   231)           end if
   232)           h2osoi_liq(c,snl(c)+1) = h2osoi_liq(c,snl(c)+1) - qflx_evap_grnd(c) * dtime
   233)        else
   234)           wgdif = h2osoi_ice(c,snl(c)+1) + (qflx_dew_snow(c) - qflx_sub_snow(c)) * dtime
   235)           h2osoi_ice(c,snl(c)+1) = wgdif
   236)           if (wgdif < 0._r8) then
   237)              h2osoi_ice(c,snl(c)+1) = 0._r8
   238)              h2osoi_liq(c,snl(c)+1) = h2osoi_liq(c,snl(c)+1) + wgdif
   239)           end if
   240)           h2osoi_liq(c,snl(c)+1) = h2osoi_liq(c,snl(c)+1) +  &
   241)                (qflx_rain_grnd(c) + qflx_dew_grnd(c) - qflx_evap_grnd(c)) * dtime
   242)        end if
   243)        h2osoi_liq(c,snl(c)+1) = max(0._r8, h2osoi_liq(c,snl(c)+1))
   244)     end do
   245) 
   246)     ! Porosity and partial volume
   247) 
   248)     do j = -nlevsno+1, 0
   249)        do fc = 1, num_snowc
   250)           c = filter_snowc(fc)
   251)           if (j >= snl(c)+1) then
   252)              vol_ice(c,j) = min(1._r8, h2osoi_ice(c,j)/(dz(c,j)*denice))
   253)              eff_porosity(c,j) = 1._r8 - vol_ice(c,j)
   254)              vol_liq(c,j) = min(eff_porosity(c,j),h2osoi_liq(c,j)/(dz(c,j)*denh2o))
   255)           end if
   256)        end do
   257)     end do
   258) 
   259)     ! Capillary forces within snow are usually two or more orders of magnitude
   260)     ! less than those of gravity. Only gravity terms are considered.
   261)     ! the genernal expression for water flow is "K * ss**3", however,
   262)     ! no effective parameterization for "K".  Thus, a very simple consideration
   263)     ! (not physically based) is introduced:
   264)     ! when the liquid water of layer exceeds the layer's holding
   265)     ! capacity, the excess meltwater adds to the underlying neighbor layer.
   266) 
   267)     ! Also compute aerosol fluxes through snowpack in this loop:
   268)     ! 1) compute aerosol mass in each layer
   269)     ! 2) add aerosol mass flux from above layer to mass of this layer
   270)     ! 3) qout_xxx is mass flux of aerosol species xxx out bottom of
   271)     !    layer in water flow, proportional to (current) concentration
   272)     !    of aerosol in layer multiplied by a scavenging ratio.
   273)     ! 4) update mass of aerosol in top layer, accordingly
   274)     ! 5) update mass concentration of aerosol accordingly
   275) 
   276)     qin(:) = 0._r8
   277)     qin_bc_phi(:) = 0._r8
   278)     qin_bc_pho(:) = 0._r8
   279)     qin_oc_phi(:) = 0._r8
   280)     qin_oc_pho(:) = 0._r8
   281)     qin_dst1(:)   = 0._r8
   282)     qin_dst2(:)   = 0._r8
   283)     qin_dst3(:)   = 0._r8
   284)     qin_dst4(:)   = 0._r8
   285) 
   286)     do j = -nlevsno+1, 0
   287)        do fc = 1, num_snowc
   288)           c = filter_snowc(fc)
   289)           if (j >= snl(c)+1) then
   290)              h2osoi_liq(c,j) = h2osoi_liq(c,j) + qin(c)
   291)              
   292)              mss_bcphi(c,j) = mss_bcphi(c,j) + qin_bc_phi(c)
   293)              mss_bcpho(c,j) = mss_bcpho(c,j) + qin_bc_pho(c)
   294)              mss_ocphi(c,j) = mss_ocphi(c,j) + qin_oc_phi(c)
   295)              mss_ocpho(c,j) = mss_ocpho(c,j) + qin_oc_pho(c)
   296)              mss_dst1(c,j)  = mss_dst1(c,j) + qin_dst1(c)
   297)              mss_dst2(c,j)  = mss_dst2(c,j) + qin_dst2(c)
   298)              mss_dst3(c,j)  = mss_dst3(c,j) + qin_dst3(c)
   299)              mss_dst4(c,j)  = mss_dst4(c,j) + qin_dst4(c)
   300) 
   301)              if (j <= -1) then
   302)                 ! No runoff over snow surface, just ponding on surface
   303)                 if (eff_porosity(c,j) < wimp .OR. eff_porosity(c,j+1) < wimp) then
   304)                    qout(c) = 0._r8
   305)                 else
   306)                    qout(c) = max(0._r8,(vol_liq(c,j)-ssi*eff_porosity(c,j))*dz(c,j))
   307)                    qout(c) = min(qout(c),(1._r8-vol_ice(c,j+1)-vol_liq(c,j+1))*dz(c,j+1))
   308)                 end if
   309)              else
   310)                 qout(c) = max(0._r8,(vol_liq(c,j) - ssi*eff_porosity(c,j))*dz(c,j))
   311)              end if
   312)              qout(c) = qout(c)*1000._r8
   313)              h2osoi_liq(c,j) = h2osoi_liq(c,j) - qout(c)
   314)              qin(c) = qout(c)
   315) 
   316)              ! mass of ice+water: in extremely rare circumstances, this can
   317)              ! be zero, even though there is a snow layer defined. In
   318)              ! this case, set the mass to a very small value to
   319)              ! prevent division by zero.
   320)              mss_liqice = h2osoi_liq(c,j)+h2osoi_ice(c,j)
   321)              if (mss_liqice < 1E-30_r8) then
   322)                 mss_liqice = 1E-30_r8
   323)              endif
   324)          
   325)              ! BCPHI:
   326)              ! 1. flux with meltwater:
   327)              qout_bc_phi(c) = qout(c)*scvng_fct_mlt_bcphi*(mss_bcphi(c,j)/mss_liqice)
   328)              if (qout_bc_phi(c) > mss_bcphi(c,j)) then
   329)                 qout_bc_phi(c) = mss_bcphi(c,j)
   330)              endif
   331)              mss_bcphi(c,j) = mss_bcphi(c,j) - qout_bc_phi(c)
   332)              qin_bc_phi(c) = qout_bc_phi(c)
   333) 
   334)              ! BCPHO:
   335)              ! 1. flux with meltwater:
   336)              qout_bc_pho(c) = qout(c)*scvng_fct_mlt_bcpho*(mss_bcpho(c,j)/mss_liqice)
   337)              if (qout_bc_pho(c) > mss_bcpho(c,j)) then
   338)                 qout_bc_pho(c) = mss_bcpho(c,j)
   339)              endif
   340)              mss_bcpho(c,j) = mss_bcpho(c,j) - qout_bc_pho(c)
   341)              qin_bc_pho(c) = qout_bc_pho(c)
   342) 
   343)              ! OCPHI:
   344)              ! 1. flux with meltwater:
   345)              qout_oc_phi(c) = qout(c)*scvng_fct_mlt_ocphi*(mss_ocphi(c,j)/mss_liqice)
   346)              if (qout_oc_phi(c) > mss_ocphi(c,j)) then
   347)                 qout_oc_phi(c) = mss_ocphi(c,j)
   348)              endif
   349)              mss_ocphi(c,j) = mss_ocphi(c,j) - qout_oc_phi(c)
   350)              qin_oc_phi(c) = qout_oc_phi(c)
   351) 
   352)              ! OCPHO:
   353)              ! 1. flux with meltwater:
   354)              qout_oc_pho(c) = qout(c)*scvng_fct_mlt_ocpho*(mss_ocpho(c,j)/mss_liqice)
   355)              if (qout_oc_pho(c) > mss_ocpho(c,j)) then
   356)                 qout_oc_pho(c) = mss_ocpho(c,j)
   357)              endif
   358)              mss_ocpho(c,j) = mss_ocpho(c,j) - qout_oc_pho(c)
   359)              qin_oc_pho(c) = qout_oc_pho(c)
   360) 
   361)              ! DUST 1:
   362)              ! 1. flux with meltwater:
   363)              qout_dst1(c) = qout(c)*scvng_fct_mlt_dst1*(mss_dst1(c,j)/mss_liqice)
   364)              if (qout_dst1(c) > mss_dst1(c,j)) then
   365)                 qout_dst1(c) = mss_dst1(c,j)
   366)              endif
   367)              mss_dst1(c,j) = mss_dst1(c,j) - qout_dst1(c)
   368)              qin_dst1(c) = qout_dst1(c)
   369) 
   370)              ! DUST 2:
   371)              ! 1. flux with meltwater:
   372)              qout_dst2(c) = qout(c)*scvng_fct_mlt_dst2*(mss_dst2(c,j)/mss_liqice)
   373)              if (qout_dst2(c) > mss_dst2(c,j)) then
   374)                 qout_dst2(c) = mss_dst2(c,j)
   375)              endif
   376)              mss_dst2(c,j) = mss_dst2(c,j) - qout_dst2(c)
   377)              qin_dst2(c) = qout_dst2(c)
   378) 
   379)              ! DUST 3:
   380)              ! 1. flux with meltwater:
   381)              qout_dst3(c) = qout(c)*scvng_fct_mlt_dst3*(mss_dst3(c,j)/mss_liqice)
   382)              if (qout_dst3(c) > mss_dst3(c,j)) then
   383)                 qout_dst3(c) = mss_dst3(c,j)
   384)              endif
   385)              mss_dst3(c,j) = mss_dst3(c,j) - qout_dst3(c)
   386)              qin_dst3(c) = qout_dst3(c)
   387) 
   388)              ! DUST 4:
   389)              ! 1. flux with meltwater:
   390)              qout_dst4(c) = qout(c)*scvng_fct_mlt_dst4*(mss_dst4(c,j)/mss_liqice)
   391)              if (qout_dst4(c) > mss_dst4(c,j)) then
   392)                 qout_dst4(c) = mss_dst4(c,j)
   393)              endif
   394)              mss_dst4(c,j) = mss_dst4(c,j) - qout_dst4(c)
   395)              qin_dst4(c) = qout_dst4(c)
   396)              
   397)           end if
   398)        end do
   399)     end do
   400) 
   401)     ! Adjust layer thickness for any water+ice content changes in excess of previous 
   402)     ! layer thickness. Strictly speaking, only necessary for top snow layer, but doing
   403)     ! it for all snow layers will catch problems with older initial files.
   404)     ! Layer interfaces (zi) and node depths (z) do not need adjustment here because they
   405)     ! are adjusted in CombineSnowLayers and are not used up to that point.
   406) 
   407)     do j = -nlevsno+1, 0
   408)        do fc = 1, num_snowc
   409)           c = filter_snowc(fc)
   410)           if (j >= snl(c)+1) then
   411)               dz(c,j) = max(dz(c,j),h2osoi_liq(c,j)/denh2o + h2osoi_ice(c,j)/denice)
   412)           end if
   413)        end do
   414)     end do
   415)  
   416)     do fc = 1, num_snowc
   417)        c = filter_snowc(fc)
   418)        ! Qout from snow bottom
   419)        qflx_top_soil(c) = qout(c) / dtime
   420)        qflx_snow_melt(c) = qflx_snow_melt(c) + (qout(c) / dtime)
   421)     end do
   422) 
   423)     do fc = 1, num_nosnowc
   424)        c = filter_nosnowc(fc)
   425)        qflx_top_soil(c) = qflx_rain_grnd(c) + qflx_snomelt(c)
   426)        qflx_snow_melt(c) = qflx_snomelt(c)
   427)     end do
   428)     
   429)     !  set aerosol deposition fluxes from forcing array
   430)     !  The forcing array is either set from an external file 
   431)     !  or from fluxes received from the atmosphere model
   432)     do c = lbc,ubc
   433)        g = cgridcell(c)
   434)        
   435)        flx_bc_dep_dry(c)   = forc_aer(g,1) + forc_aer(g,2)
   436)        flx_bc_dep_wet(c)   = forc_aer(g,3)
   437)        flx_bc_dep_phi(c)   = forc_aer(g,1) + forc_aer(g,3)
   438)        flx_bc_dep_pho(c)   = forc_aer(g,2)
   439)        flx_bc_dep(c)       = forc_aer(g,1) + forc_aer(g,2) + forc_aer(g,3)
   440)        
   441)        flx_oc_dep_dry(c)   = forc_aer(g,4) + forc_aer(g,5)
   442)        flx_oc_dep_wet(c)   = forc_aer(g,6)
   443)        flx_oc_dep_phi(c)   = forc_aer(g,4) + forc_aer(g,6)
   444)        flx_oc_dep_pho(c)   = forc_aer(g,5)
   445)        flx_oc_dep(c)       = forc_aer(g,4) + forc_aer(g,5) + forc_aer(g,6)
   446)        
   447)        flx_dst_dep_wet1(c) = forc_aer(g,7)
   448)        flx_dst_dep_dry1(c) = forc_aer(g,8)
   449)        flx_dst_dep_wet2(c) = forc_aer(g,9)
   450)        flx_dst_dep_dry2(c) = forc_aer(g,10)
   451)        flx_dst_dep_wet3(c) = forc_aer(g,11)
   452)        flx_dst_dep_dry3(c) = forc_aer(g,12)
   453)        flx_dst_dep_wet4(c) = forc_aer(g,13)
   454)        flx_dst_dep_dry4(c) = forc_aer(g,14)
   455)        flx_dst_dep(c)      = forc_aer(g,7) + forc_aer(g,8) + forc_aer(g,9) + &
   456)                              forc_aer(g,10) + forc_aer(g,11) + forc_aer(g,12) + &
   457)                              forc_aer(g,13) + forc_aer(g,14)
   458)     
   459)     end do
   460) 
   461)     ! aerosol deposition fluxes into top layer
   462)     ! This is done after the inter-layer fluxes so that some aerosol
   463)     ! is in the top layer after deposition, and is not immediately
   464)     ! washed out before radiative calculations are done
   465)     do fc = 1, num_snowc
   466)        c = filter_snowc(fc)
   467)        mss_bcphi(c,snl(c)+1) = mss_bcphi(c,snl(c)+1) + (flx_bc_dep_phi(c)*dtime)
   468)        mss_bcpho(c,snl(c)+1) = mss_bcpho(c,snl(c)+1) + (flx_bc_dep_pho(c)*dtime)
   469)        mss_ocphi(c,snl(c)+1) = mss_ocphi(c,snl(c)+1) + (flx_oc_dep_phi(c)*dtime)
   470)        mss_ocpho(c,snl(c)+1) = mss_ocpho(c,snl(c)+1) + (flx_oc_dep_pho(c)*dtime)
   471)        
   472)        mss_dst1(c,snl(c)+1) = mss_dst1(c,snl(c)+1) + (flx_dst_dep_dry1(c) + flx_dst_dep_wet1(c))*dtime
   473)        mss_dst2(c,snl(c)+1) = mss_dst2(c,snl(c)+1) + (flx_dst_dep_dry2(c) + flx_dst_dep_wet2(c))*dtime
   474)        mss_dst3(c,snl(c)+1) = mss_dst3(c,snl(c)+1) + (flx_dst_dep_dry3(c) + flx_dst_dep_wet3(c))*dtime
   475)        mss_dst4(c,snl(c)+1) = mss_dst4(c,snl(c)+1) + (flx_dst_dep_dry4(c) + flx_dst_dep_wet4(c))*dtime
   476)     end do
   477) 
   478)   end subroutine SnowWater

   483) ! !IROUTINE: SnowCompaction
   484) !
   485) ! !INTERFACE:
   486)   subroutine SnowCompaction(lbc, ubc, num_snowc, filter_snowc)
   487) !
   488) ! !DESCRIPTION:
   489) ! Determine the change in snow layer thickness due to compaction and
   490) ! settling.
   491) ! Three metamorphisms of changing snow characteristics are implemented,
   492) ! i.e., destructive, overburden, and melt. The treatments of the former
   493) ! two are from SNTHERM.89 and SNTHERM.99 (1991, 1999). The contribution
   494) ! due to melt metamorphism is simply taken as a ratio of snow ice
   495) ! fraction after the melting versus before the melting.
   496) !
   497) ! !USES:
   498)     use clmtype
   499)     use clm_time_manager, only : get_step_size
   500)     use clm_varcon  , only : denice, denh2o, tfrz, istice_mec
   501) !
   502) ! !ARGUMENTS:
   503)     implicit none
   504)     integer, intent(in) :: lbc, ubc                ! column bounds
   505)     integer, intent(in) :: num_snowc               ! number of column snow points in column filter
   506)     integer, intent(in) :: filter_snowc(ubc-lbc+1) ! column filter for snow points
   507) !
   508) ! !CALLED FROM:
   509) ! subroutine Hydrology2 in module Hydrology2Mod
   510) !
   511) ! !REVISION HISTORY:
   512) ! 15 September 1999: Yongjiu Dai; Initial code
   513) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
   514) ! 2/28/02, Peter Thornton: Migrated to new data structures
   515) ! 2/29/08, David Lawrence: Revised snow overburden to be include 0.5 weight of current layer
   516) !
   517) ! !LOCAL VARIABLES:
   518) !
   519) ! local pointers to implicit in scalars
   520) !
   521)     integer,  pointer :: snl(:)             !number of snow layers
   522) !
   523) ! local pointers to implicit in arguments
   524) !
   525)     integer,  pointer :: imelt(:,:)        !flag for melting (=1), freezing (=2), Not=0
   526)     real(r8), pointer :: frac_iceold(:,:)  !fraction of ice relative to the tot water
   527)     real(r8), pointer :: t_soisno(:,:)     !soil temperature (Kelvin)
   528)     real(r8), pointer :: h2osoi_ice(:,:)   !ice lens (kg/m2)
   529)     real(r8), pointer :: h2osoi_liq(:,:)   !liquid water (kg/m2)
   530) !
   531) ! local pointers to implicit inout arguments
   532) !
   533)     real(r8), pointer :: dz(:,:)           !layer depth (m)
   534) !
   535) !
   536) ! !OTHER LOCAL VARIABLES:
   537) !EOP
   538) !
   539)     integer :: j, l, c, fc                   ! indices
   540)     real(r8):: dtime                         ! land model time step (sec)
   541)     real(r8), parameter :: c2 = 23.e-3_r8    ! [m3/kg]
   542)     real(r8), parameter :: c3 = 2.777e-6_r8  ! [1/s]
   543)     real(r8), parameter :: c4 = 0.04_r8      ! [1/K]
   544)     real(r8), parameter :: c5 = 2.0_r8       !
   545)     real(r8), parameter :: dm = 100.0_r8     ! Upper Limit on Destructive Metamorphism Compaction [kg/m3]
   546)     real(r8), parameter :: eta0 = 9.e+5_r8   ! The Viscosity Coefficient Eta0 [kg-s/m2]
   547)     real(r8) :: burden(lbc:ubc) ! pressure of overlying snow [kg/m2]
   548)     real(r8) :: ddz1   ! Rate of settling of snowpack due to destructive metamorphism.
   549)     real(r8) :: ddz2   ! Rate of compaction of snowpack due to overburden.
   550)     real(r8) :: ddz3   ! Rate of compaction of snowpack due to melt [1/s]
   551)     real(r8) :: dexpf  ! expf=exp(-c4*(273.15-t_soisno)).
   552)     real(r8) :: fi     ! Fraction of ice relative to the total water content at current time step
   553)     real(r8) :: td     ! t_soisno - tfrz [K]
   554)     real(r8) :: pdzdtc ! Nodal rate of change in fractional-thickness due to compaction [fraction/s]
   555)     real(r8) :: void   ! void (1 - vol_ice - vol_liq)
   556)     real(r8) :: wx     ! water mass (ice+liquid) [kg/m2]
   557)     real(r8) :: bi     ! partial density of ice [kg/m3]
   558) 
   559)     integer, pointer :: clandunit(:)       !landunit index for each column
   560)     integer, pointer :: ltype(:)           !landunit type
   561) 
   562) !-----------------------------------------------------------------------
   563) 
   564)     ! Assign local pointers to derived subtypes (column-level)
   565) 
   566)     snl         => cps%snl
   567)     dz          => cps%dz
   568)     imelt       => cps%imelt
   569)     frac_iceold => cps%frac_iceold
   570)     t_soisno    => ces%t_soisno
   571)     h2osoi_ice  => cws%h2osoi_ice
   572)     h2osoi_liq  => cws%h2osoi_liq
   573)     clandunit   => col%landunit
   574)     ltype       => lun%itype
   575) 
   576)     ! Get time step
   577) 
   578)     dtime = get_step_size()
   579) 
   580)     ! Begin calculation - note that the following column loops are only invoked if snl(c) < 0
   581) 
   582)     burden(:) = 0._r8
   583) 
   584)     do j = -nlevsno+1, 0
   585)        do fc = 1, num_snowc
   586)           c = filter_snowc(fc)
   587)           if (j >= snl(c)+1) then
   588) 
   589)              wx = h2osoi_ice(c,j) + h2osoi_liq(c,j)
   590)              void = 1._r8 - (h2osoi_ice(c,j)/denice + h2osoi_liq(c,j)/denh2o) / dz(c,j)
   591) 
   592)              ! If void is negative, then increase dz such that void = 0.
   593)              ! This should be done for any landunit, but for now is done only for glacier_mec 1andunits.
   594)              l = clandunit(c)
   595)              if (ltype(l)==istice_mec .and. void < 0._r8) then
   596)                 dz(c,j) = h2osoi_ice(c,j)/denice + h2osoi_liq(c,j)/denh2o
   597)                 void = 0._r8
   598)              endif
   599) 
   600)              ! Allow compaction only for non-saturated node and higher ice lens node.
   601)              if (void > 0.001_r8 .and. h2osoi_ice(c,j) > .1_r8) then
   602)                 bi = h2osoi_ice(c,j) / dz(c,j)
   603)                 fi = h2osoi_ice(c,j) / wx
   604)                 td = tfrz-t_soisno(c,j)
   605)                 dexpf = exp(-c4*td)
   606) 
   607)                 ! Settling as a result of destructive metamorphism
   608) 
   609)                 ddz1 = -c3*dexpf
   610)                 if (bi > dm) ddz1 = ddz1*exp(-46.0e-3_r8*(bi-dm))
   611) 
   612)                 ! Liquid water term
   613) 
   614)                 if (h2osoi_liq(c,j) > 0.01_r8*dz(c,j)) ddz1=ddz1*c5
   615) 
   616)                 ! Compaction due to overburden
   617) 
   618)                 ddz2 = -(burden(c)+wx/2._r8)*exp(-0.08_r8*td - c2*bi)/eta0 
   619) 
   620)                 ! Compaction occurring during melt
   621) 
   622)                 if (imelt(c,j) == 1) then
   623)                    ddz3 = - 1._r8/dtime * max(0._r8,(frac_iceold(c,j) - fi)/frac_iceold(c,j))
   624)                 else
   625)                    ddz3 = 0._r8
   626)                 end if
   627) 
   628)                 ! Time rate of fractional change in dz (units of s-1)
   629) 
   630)                 pdzdtc = ddz1 + ddz2 + ddz3
   631) 
   632)                 ! The change in dz due to compaction
   633)                 ! Limit compaction to no less than fully saturated layer thickness
   634) 
   635)                 dz(c,j) = max(dz(c,j) * (1._r8+pdzdtc*dtime),h2osoi_ice(c,j)/denice &
   636)                                          + h2osoi_liq(c,j)/denh2o)
   637) 
   638)              end if
   639) 
   640)              ! Pressure of overlying snow
   641) 
   642)              burden(c) = burden(c) + wx
   643) 
   644)           end if
   645)        end do
   646)     end do
   647) 
   648)   end subroutine SnowCompaction

   658) ! !DESCRIPTION:

   660) ! If the snow element thickness or mass is less than a prescribed minimum,
   661) ! then it is combined with a neighboring element.  The subroutine
   662) ! clm\_combo.f90 then executes the combination of mass and energy.

   669) !

   672)     integer, intent(in)    :: lbc, ubc                    ! column bounds
   673)     integer, intent(inout) :: num_snowc                   ! number of column snow points in column filter
   674)     integer, intent(inout) :: filter_snowc(ubc-lbc+1)     ! column filter for snow points
   675) !
   676) ! !CALLED FROM:
   677) ! subroutine Hydrology2 in module Hydrology2Mod
   678) !
   679) ! !REVISION HISTORY:
   680) ! 15 September 1999: Yongjiu Dai; Initial code
   681) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
   682) ! 2/28/02, Peter Thornton: Migrated to new data structures.
   683) ! 03/28/08, Mark Flanner: Added aerosol masses and snow grain radius
   684) !
   685) ! !LOCAL VARIABLES:
   686) !
   687) ! local pointers to implicit in arguments
   688) !
   689)     integer, pointer :: clandunit(:)       !landunit index for each column
   690)     integer, pointer :: ltype(:)           !landunit type
   691) !
   692) ! local pointers to implicit inout arguments
   693) !
   694)     integer , pointer :: snl(:)            !number of snow layers
   695)     real(r8), pointer :: h2osno(:)         !snow water (mm H2O)
   696)     real(r8), pointer :: snowdp(:)         !snow height (m)

   698)     real(r8), pointer :: zi(:,:)           !interface level below a "z" level (m)
   699)     real(r8), pointer :: t_soisno(:,:)     !soil temperature (Kelvin)
   700)     real(r8), pointer :: h2osoi_ice(:,:)   !ice lens (kg/m2)
   701)     real(r8), pointer :: h2osoi_liq(:,:)   !liquid water (kg/m2)
   702) !
   703) ! local pointers to implicit out arguments
   704) !
   705)     real(r8), pointer :: z(:,:)          ! layer thickness (m)
   706)     real(r8), pointer :: mss_bcphi(:,:)  ! hydrophilic BC mass in snow (col,lyr) [kg]
   707)     real(r8), pointer :: mss_bcpho(:,:)  ! hydrophobic BC mass in snow (col,lyr) [kg]
   708)     real(r8), pointer :: mss_ocphi(:,:)  ! hydrophilic OC mass in snow (col,lyr) [kg]
   709)     real(r8), pointer :: mss_ocpho(:,:)  ! hydrophobic OC mass in snow (col,lyr) [kg]
   710)     real(r8), pointer :: mss_dst1(:,:)   ! dust species 1 mass in snow (col,lyr) [kg]
   711)     real(r8), pointer :: mss_dst2(:,:)   ! dust species 2 mass in snow (col,lyr) [kg]
   712)     real(r8), pointer :: mss_dst3(:,:)   ! dust species 3 mass in snow (col,lyr) [kg]
   713)     real(r8), pointer :: mss_dst4(:,:)   ! dust species 4 mass in snow (col,lyr) [kg]
   714)     real(r8), pointer :: snw_rds(:,:)    ! effective snow grain radius (col,lyr) [microns, m^-6]
   715)     real(r8), pointer :: qflx_sl_top_soil(:) ! liquid water + ice from layer above soil to top soil layer or sent to qflx_qrgwl (mm H2O/s)
   716) !
   717) !
   718) ! !OTHER LOCAL VARIABLES:
   719) !EOP
   720) !
   721)     integer :: c, fc                 ! column indices
   722)     integer :: i,k                   ! loop indices
   723)     integer :: j,l                   ! node indices
   724)     integer :: msn_old(lbc:ubc)      ! number of top snow layer
   725)     integer :: mssi(lbc:ubc)         ! node index
   726)     integer :: neibor                ! adjacent node selected for combination
   727)     real(r8):: zwice(lbc:ubc)        ! total ice mass in snow
   728)     real(r8):: zwliq (lbc:ubc)       ! total liquid water in snow
   729)     real(r8):: dzmin(5)              ! minimum of top snow layer
   730)     real(r8) :: dtime                !land model time step (sec)
   731) 
   732)     data dzmin /0.010_r8, 0.015_r8, 0.025_r8, 0.055_r8, 0.115_r8/
   733) !-----------------------------------------------------------------------
   734) 
   735)     ! Assign local pointers to derived subtypes (landunit-level)
   736) 
   737)     ltype      => lun%itype
   738) 
   739)     ! Assign local pointers to derived subtypes (column-level)
   740) 
   741)     clandunit  => col%landunit
   742)     snl        => cps%snl
   743)     snowdp     => cps%snowdp
   744)     h2osno     => cws%h2osno
   745)     dz         => cps%dz
   746)     zi         => cps%zi
   747)     z          => cps%z
   748)     t_soisno   => ces%t_soisno
   749)     h2osoi_ice => cws%h2osoi_ice
   750)     h2osoi_liq => cws%h2osoi_liq
   751)     mss_bcphi  => cps%mss_bcphi
   752)     mss_bcpho  => cps%mss_bcpho
   753)     mss_ocphi  => cps%mss_ocphi
   754)     mss_ocpho  => cps%mss_ocpho
   755)     mss_dst1   => cps%mss_dst1
   756)     mss_dst2   => cps%mss_dst2
   757)     mss_dst3   => cps%mss_dst3
   758)     mss_dst4   => cps%mss_dst4
   759)     snw_rds    => cps%snw_rds
   760)     qflx_sl_top_soil => cwf%qflx_sl_top_soil
   761) 
   762)     ! Determine model time step
   763) 
   764)     dtime = get_step_size()
   765) 
   766) 
   767)     ! Check the mass of ice lens of snow, when the total is less than a small value,
   768)     ! combine it with the underlying neighbor.
   769) 
   770)     do fc = 1, num_snowc
   771)        c = filter_snowc(fc)
   772)        msn_old(c) = snl(c)
   773)        qflx_sl_top_soil(c) = 0._r8
   774)     end do
   775) 
   776)     ! The following loop is NOT VECTORIZED
   777) 
   778)     do fc = 1, num_snowc
   779)        c = filter_snowc(fc)
   780)        l = clandunit(c)
   781)        do j = msn_old(c)+1,0
   782)           if (h2osoi_ice(c,j) <= .1_r8) then
   783)              if (ltype(l) == istsoil .or. ltype(l)==isturb .or. ltype(l) == istcrop) then
   784)                 h2osoi_liq(c,j+1) = h2osoi_liq(c,j+1) + h2osoi_liq(c,j)
   785)                 h2osoi_ice(c,j+1) = h2osoi_ice(c,j+1) + h2osoi_ice(c,j)
   786) 
   787)                 if (j == 0) then
   788)                   qflx_sl_top_soil(c) = (h2osoi_liq(c,j) + h2osoi_ice(c,j))/dtime
   789)                 end if
   790) 
   791)                 if (j /= 0) dz(c,j+1) = dz(c,j+1) + dz(c,j)
   792)                 
   793)                 ! NOTE: Temperature, and similarly snw_rds, of the
   794)                 ! underlying snow layer are NOT adjusted in this case. 
   795)                 ! Because the layer being eliminated has a small mass, 
   796)                 ! this should not make a large difference, but it 
   797)                 ! would be more thorough to do so.
   798)                 if (j /= 0) then
   799)                    mss_bcphi(c,j+1) = mss_bcphi(c,j+1) + mss_bcphi(c,j)
   800)                    mss_bcpho(c,j+1) = mss_bcpho(c,j+1) + mss_bcpho(c,j)
   801)                    mss_ocphi(c,j+1) = mss_ocphi(c,j+1) + mss_ocphi(c,j)
   802)                    mss_ocpho(c,j+1) = mss_ocpho(c,j+1) + mss_ocpho(c,j)
   803)                    mss_dst1(c,j+1)  = mss_dst1(c,j+1) + mss_dst1(c,j)
   804)                    mss_dst2(c,j+1)  = mss_dst2(c,j+1) + mss_dst2(c,j)
   805)                    mss_dst3(c,j+1)  = mss_dst3(c,j+1) + mss_dst3(c,j)
   806)                    mss_dst4(c,j+1)  = mss_dst4(c,j+1) + mss_dst4(c,j)
   807)                 end if
   808) 
   809)              else if (ltype(l) /= istsoil .and. ltype(l) /= isturb .and. ltype(l) /= istcrop .and. j /= 0) then
   810)                 h2osoi_liq(c,j+1) = h2osoi_liq(c,j+1) + h2osoi_liq(c,j)
   811)                 h2osoi_ice(c,j+1) = h2osoi_ice(c,j+1) + h2osoi_ice(c,j)
   812)                 dz(c,j+1) = dz(c,j+1) + dz(c,j)
   813)                 
   814)                 mss_bcphi(c,j+1) = mss_bcphi(c,j+1) + mss_bcphi(c,j)
   815)                 mss_bcpho(c,j+1) = mss_bcpho(c,j+1) + mss_bcpho(c,j)
   816)                 mss_ocphi(c,j+1) = mss_ocphi(c,j+1) + mss_ocphi(c,j)
   817)                 mss_ocpho(c,j+1) = mss_ocpho(c,j+1) + mss_ocpho(c,j)
   818)                 mss_dst1(c,j+1)  = mss_dst1(c,j+1) + mss_dst1(c,j)
   819)                 mss_dst2(c,j+1)  = mss_dst2(c,j+1) + mss_dst2(c,j)
   820)                 mss_dst3(c,j+1)  = mss_dst3(c,j+1) + mss_dst3(c,j)
   821)                 mss_dst4(c,j+1)  = mss_dst4(c,j+1) + mss_dst4(c,j)
   822) 
   823)              end if
   824) 
   825)              ! shift all elements above this down one.
   826)              if (j > snl(c)+1 .and. snl(c) < -1) then
   827)                 do i = j, snl(c)+2, -1
   828)                    ! If the layer closest to the surface is less than 0.1 mm and the ltype is not
   829)                    ! urban, soil or crop, the h2osoi_liq and h2osoi_ice associated with this layer is sent 
   830)                    ! to qflx_qrgwl later on in the code.  To keep track of this for the snow balance
   831)                    ! error check, we add this to qflx_sl_top_soil here
   832)                    if (ltype(l) /= istsoil .and. ltype(l) /= istcrop .and. ltype(l) /= isturb .and. i == 0) then
   833)                      qflx_sl_top_soil(c) = (h2osoi_liq(c,i) + h2osoi_ice(c,i))/dtime
   834)                    end if
   835) 
   836)                    t_soisno(c,i)   = t_soisno(c,i-1)
   837)                    h2osoi_liq(c,i) = h2osoi_liq(c,i-1)
   838)                    h2osoi_ice(c,i) = h2osoi_ice(c,i-1)
   839)                    
   840)                    mss_bcphi(c,i)   = mss_bcphi(c,i-1)
   841)                    mss_bcpho(c,i)   = mss_bcpho(c,i-1)
   842)                    mss_ocphi(c,i)   = mss_ocphi(c,i-1)
   843)                    mss_ocpho(c,i)   = mss_ocpho(c,i-1)
   844)                    mss_dst1(c,i)    = mss_dst1(c,i-1)
   845)                    mss_dst2(c,i)    = mss_dst2(c,i-1)
   846)                    mss_dst3(c,i)    = mss_dst3(c,i-1)
   847)                    mss_dst4(c,i)    = mss_dst4(c,i-1)
   848)                    snw_rds(c,i)     = snw_rds(c,i-1)
   849) 
   850)                    dz(c,i)         = dz(c,i-1)
   851)                 end do
   852)              end if
   853)              snl(c) = snl(c) + 1
   854)           end if
   855)        end do
   856)     end do
   857) 
   858)     do fc = 1, num_snowc
   859)        c = filter_snowc(fc)
   860)        h2osno(c) = 0._r8
   861)        snowdp(c) = 0._r8
   862)        zwice(c)  = 0._r8
   863)        zwliq(c)  = 0._r8
   864)     end do
   865) 
   866)     do j = -nlevsno+1,0
   867)        do fc = 1, num_snowc
   868)           c = filter_snowc(fc)
   869)           if (j >= snl(c)+1) then
   870)              h2osno(c) = h2osno(c) + h2osoi_ice(c,j) + h2osoi_liq(c,j)
   871)              snowdp(c) = snowdp(c) + dz(c,j)
   872)              zwice(c)  = zwice(c) + h2osoi_ice(c,j)
   873)              zwliq(c)  = zwliq(c) + h2osoi_liq(c,j)
   874)           end if
   875)        end do
   876)     end do
   877) 
   878)     ! Check the snow depth - all snow gone
   879)     ! The liquid water assumes ponding on soil surface.
   880) 
   881)     do fc = 1, num_snowc
   882)        c = filter_snowc(fc)
   883)        l = clandunit(c)
   884)        if (snowdp(c) < 0.01_r8 .and. snowdp(c) > 0._r8) then
   885)           snl(c) = 0
   886)           h2osno(c) = zwice(c)
   887) 
   888)           mss_bcphi(c,:) = 0._r8
   889)           mss_bcpho(c,:) = 0._r8
   890)           mss_ocphi(c,:) = 0._r8
   891)           mss_ocpho(c,:) = 0._r8
   892)           mss_dst1(c,:)  = 0._r8
   893)           mss_dst2(c,:)  = 0._r8
   894)           mss_dst3(c,:)  = 0._r8
   895)           mss_dst4(c,:)  = 0._r8
   896) 
   897)           if (h2osno(c) <= 0._r8) snowdp(c) = 0._r8
   898)           if (ltype(l) == istsoil .or. ltype(l) == isturb .or. ltype(l) == istcrop) then
   899)              h2osoi_liq(c,1) = h2osoi_liq(c,1) + zwliq(c)
   900)           end if
   901)        end if
   902)     end do
   903) 
   904)     ! Check the snow depth - snow layers combined
   905)     ! The following loop IS NOT VECTORIZED
   906) 
   907)     do fc = 1, num_snowc
   908)        c = filter_snowc(fc)
   909) 
   910)        ! Two or more layers
   911) 
   912)        if (snl(c) < -1) then
   913) 
   914)           msn_old(c) = snl(c)
   915)           mssi(c) = 1
   916) 
   917)           do i = msn_old(c)+1,0
   918)              if (dz(c,i) < dzmin(mssi(c))) then
   919) 
   920)                 if (i == snl(c)+1) then
   921)                    ! If top node is removed, combine with bottom neighbor.
   922)                    neibor = i + 1
   923)                 else if (i == 0) then
   924)                    ! If the bottom neighbor is not snow, combine with the top neighbor.
   925)                    neibor = i - 1
   926)                 else
   927)                    ! If none of the above special cases apply, combine with the thinnest neighbor
   928)                    neibor = i + 1
   929)                    if ((dz(c,i-1)+dz(c,i)) < (dz(c,i+1)+dz(c,i))) neibor = i-1
   930)                 end if
   931) 
   932)                 ! Node l and j are combined and stored as node j.
   933)                 if (neibor > i) then
   934)                    j = neibor
   935)                    l = i
   936)                 else
   937)                    j = i
   938)                    l = neibor
   939)                 end if
   940) 
   941)                 ! this should be included in 'Combo' for consistency,
   942)                 ! but functionally it is the same to do it here
   943)                 mss_bcphi(c,j)=mss_bcphi(c,j)+mss_bcphi(c,l)
   944)                 mss_bcpho(c,j)=mss_bcpho(c,j)+mss_bcpho(c,l)
   945)                 mss_ocphi(c,j)=mss_ocphi(c,j)+mss_ocphi(c,l)
   946)                 mss_ocpho(c,j)=mss_ocpho(c,j)+mss_ocpho(c,l)
   947)                 mss_dst1(c,j)=mss_dst1(c,j)+mss_dst1(c,l)
   948)                 mss_dst2(c,j)=mss_dst2(c,j)+mss_dst2(c,l)
   949)                 mss_dst3(c,j)=mss_dst3(c,j)+mss_dst3(c,l)
   950)                 mss_dst4(c,j)=mss_dst4(c,j)+mss_dst4(c,l)
   951)                 ! mass-weighted combination of effective grain size:
   952)                 snw_rds(c,j) = (snw_rds(c,j)*(h2osoi_liq(c,j)+h2osoi_ice(c,j)) + &
   953)                                snw_rds(c,l)*(h2osoi_liq(c,l)+h2osoi_ice(c,l))) / &
   954)                                (h2osoi_liq(c,j)+h2osoi_ice(c,j)+h2osoi_liq(c,l)+h2osoi_ice(c,l))
   955) 
   956)                 call Combo (dz(c,j), h2osoi_liq(c,j), h2osoi_ice(c,j), &
   957)                    t_soisno(c,j), dz(c,l), h2osoi_liq(c,l), h2osoi_ice(c,l), t_soisno(c,l) )
   958) 
   959)                 ! Now shift all elements above this down one.
   960)                 if (j-1 > snl(c)+1) then
   961)                    do k = j-1, snl(c)+2, -1
   962)                       t_soisno(c,k) = t_soisno(c,k-1)
   963)                       h2osoi_ice(c,k) = h2osoi_ice(c,k-1)
   964)                       h2osoi_liq(c,k) = h2osoi_liq(c,k-1)
   965) 
   966)                       mss_bcphi(c,k) = mss_bcphi(c,k-1)
   967)                       mss_bcpho(c,k) = mss_bcpho(c,k-1)
   968)                       mss_ocphi(c,k) = mss_ocphi(c,k-1)
   969)                       mss_ocpho(c,k) = mss_ocpho(c,k-1)
   970)                       mss_dst1(c,k)  = mss_dst1(c,k-1)
   971)                       mss_dst2(c,k)  = mss_dst2(c,k-1)
   972)                       mss_dst3(c,k)  = mss_dst3(c,k-1)
   973)                       mss_dst4(c,k)  = mss_dst4(c,k-1)
   974)                       snw_rds(c,k)   = snw_rds(c,k-1)
   975) 
   976)                       dz(c,k) = dz(c,k-1)
   977)                    end do
   978)                 end if
   979) 
   980)                 ! Decrease the number of snow layers
   981)                 snl(c) = snl(c) + 1
   982)                 if (snl(c) >= -1) EXIT
   983) 
   984)              else
   985) 
   986)                 ! The layer thickness is greater than the prescribed minimum value
   987)                 mssi(c) = mssi(c) + 1
   988) 
   989)              end if
   990)           end do
   991) 
   992)        end if
   993) 
   994)     end do
   995) 
   996)     ! Reset the node depth and the depth of layer interface
   997) 
   998)     do j = 0, -nlevsno+1, -1
   999)        do fc = 1, num_snowc
  1000)           c = filter_snowc(fc)
  1001)           if (j >= snl(c) + 1) then
  1002)              z(c,j) = zi(c,j) - 0.5_r8*dz(c,j)
  1003)              zi(c,j-1) = zi(c,j) - dz(c,j)
  1004)           end if
  1005)        end do
  1006)     end do
  1007) 
  1008)   end subroutine CombineSnowLayers
