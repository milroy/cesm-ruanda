    21)   subroutine preq_vertadv(T, v, eta_dot_dp_deta, rpdel, &
    22)        T_vadv, v_vadv)
    23)     use kinds,              only : real_kind
    24)     use dimensions_mod,     only : nlev, np, nlevp
    25)     implicit none
    26)     
    27)     real (kind=real_kind), intent(in) :: T(np,np,nlev)
    28)     real (kind=real_kind), intent(in) :: v(np,np,2,nlev)
    29)     real (kind=real_kind), intent(in) :: eta_dot_dp_deta(np,np,nlevp)
    30)     real (kind=real_kind), intent(in) :: rpdel(np,np,nlev)
    31) 
    32)     real (kind=real_kind), intent(out) :: T_vadv(np,np,nlev)
    33)     real (kind=real_kind), intent(out) :: v_vadv(np,np,2,nlev)
    34) 
    35)     ! ========================
    36)     ! Local Variables
    37)     ! ========================
    38) 
    39)     integer :: i,j,k
    40)     real (kind=real_kind) :: facp, facm
    41) 
    42) #if (defined ELEMENT_OPENMP)
    43) !$omp parallel do private(k,j,i,facp,facm)
    44) #endif
    45)     do j=1,np   !   Loop inversion (AAM)
    46) 
    47)     ! ===========================================================
    48)     ! Compute vertical advection of T and v from eq. (3.b.1)
    49)     !
    50)     ! k = 1 case:
    51)     ! ===========================================================
    52) 
    53)        k=1
    54)        do i=1,np 
    55)           facp            = (0.5_real_kind*rpdel(i,j,k))*eta_dot_dp_deta(i,j,k+1)
    56)           T_vadv(i,j,k)   = facp*(T(i,j,k+1)- T(i,j,k))
    57)           v_vadv(i,j,1,k) = facp*(v(i,j,1,k+1)- v(i,j,1,k))
    58)           v_vadv(i,j,2,k) = facp*(v(i,j,2,k+1)- v(i,j,2,k))
    59)        end do
    60) 
    61)     ! ===========================================================
    62)     ! vertical advection
    63)     !
    64)     ! 1 < k < nlev case:
    65)     ! ===========================================================
    66) 
    67)        do k=2,nlev-1
    68)           do i=1,np
    69)              facp            = (0.5_real_kind*rpdel(i,j,k))*eta_dot_dp_deta(i,j,k+1)
    70)              facm            = (0.5_real_kind*rpdel(i,j,k))*eta_dot_dp_deta(i,j,k)
    71)              T_vadv(i,j,k)   = facp*(T(i,j,k+1)- T(i,j,k)) + &
    72)                   facm*(T(i,j,k)- T(i,j,k-1))
    73)              v_vadv(i,j,1,k) = facp*(v(i,j,1,k+1)- v(i,j,1,k)) + &
    74)                   facm*(v(i,j,1,k)- v(i,j,1,k-1))
    75)              v_vadv(i,j,2,k) = facp*(v(i,j,2,k+1)- v(i,j,2,k)) + &
    76)                   facm*(v(i,j,2,k)- v(i,j,2,k-1))
    77)           end do
    78)        end do
    79) 
    80)     ! ===========================================================
    81)     ! vertical advection
    82)     !
    83)     ! k = nlev case:
    84)     ! ===========================================================
    85) 
    86)        k=nlev
    87)        do i=1,np
    88)           facm            = (0.5_real_kind*rpdel(i,j,k))*eta_dot_dp_deta(i,j,k)
    89)           T_vadv(i,j,k)   = facm*(T(i,j,k)- T(i,j,k-1))
    90)           v_vadv(i,j,1,k) = facm*(v(i,j,1,k)- v(i,j,1,k-1))
    91)           v_vadv(i,j,2,k) = facm*(v(i,j,2,k)- v(i,j,2,k-1))
    92)        end do
    93) 
    94)     end do
    95) 
    96)   end subroutine preq_vertadv

   102) ! preq_omegap:
   103) 
   104) ! Purpose: 
   105) ! Calculate (omega/p) needed for the Thermodynamics Equation
   106) ! 
   107) ! Method: 
   108) ! Simplified version in CAM2 for clarity
   109) ! 
   110) ! Author: Modified by Rich Loft for use in HOMME. 
   111) ! 
   112) !-----------------------------------------------------------------------
   113) 
   114)   subroutine preq_omegap(div     ,vgrad_ps,pdel    ,rpmid, &
   115)        hybm    ,hybd    ,omegap   )
   116)     use kinds, only : real_kind
   117)     use dimensions_mod, only : np, nlev
   118)     implicit none
   119) 
   120) 
   121)     !------------------------------Arguments---------------------------------------------------------------
   122)     real(kind=real_kind), intent(in) :: div(np,np,nlev)      ! divergence
   123)     real(kind=real_kind), intent(in) :: vgrad_ps(np,np,nlev) ! v.grad(ps)
   124)     real(kind=real_kind), intent(in) :: pdel(np,np,nlev)     ! layer thicknesses (pressure)
   125)     real(kind=real_kind), intent(in) :: rpmid(np,np,nlev)    ! 1./pmid
   126)     real(kind=real_kind), intent(in) :: hybm(nlev)           ! Hybrid B coefficient on mid levels
   127)     real(kind=real_kind), intent(in) :: hybd(nlev)           ! Hybrid dB coefficient on mid levels
   128)     real(kind=real_kind), intent(out):: omegap(np,np,nlev)   ! vertical pressure velocity
   129)     !------------------------------------------------------------------------------------------------------
   130) 
   131)     !---------------------------Local workspace-----------------------------
   132)     integer i,j,k                         ! longitude, level indices
   133)     real(kind=real_kind) term             ! one half of basic term in omega/p summation 
   134)     real(kind=real_kind) Ckk              ! diagonal term of energy conversion matrix
   135)     real(kind=real_kind) suml(np,np)      ! partial sum over l = (1, k-1)
   136)     !-----------------------------------------------------------------------
   137) 
   138)     ! =========================
   139)     ! Zero partial sum
   140)     ! =========================
   141) 
   142)     do j=1,np
   143)        do i=1,np
   144)           suml(i,j)=0.0_real_kind
   145)        end do
   146)     end do
   147) 
   148)     ! =============================
   149)     ! Compute omegap 
   150)     ! =============================
   151) 
   152)     do k=1,nlev
   153)        do j=1,np
   154)           do i=1,np
   155)              Ckk       = 0.5_real_kind
   156)              term      = Ckk*(div(i,j,k)*pdel(i,j,k) + vgrad_ps(i,j,k)*hybd(k))
   157)              suml(i,j) = suml(i,j) + term
   158)              omegap(i,j,k) = rpmid(i,j,k)*(hybm(k)*vgrad_ps(i,j,k) - suml(i,j))
   159)              suml(i,j) = suml(i,j) + term
   160)           end do
   161)        end do
   162)     end do
   163) 
   164)   end subroutine preq_omegap

   173)   subroutine preq_omega_ps(omega_p,hvcoord,p,vgrad_p,divdp)
   174)     use kinds, only : real_kind
   175)     use dimensions_mod, only : np, nlev
   176)     use hybvcoord_mod, only : hvcoord_t
   177)     implicit none
   178) 
   179) 
   180)     !------------------------------Arguments---------------------------------------------------------------
   181)     real(kind=real_kind), intent(in) :: divdp(np,np,nlev)      ! divergence
   182)     real(kind=real_kind), intent(in) :: vgrad_p(np,np,nlev) ! v.grad(p)
   183)     real(kind=real_kind), intent(in) :: p(np,np,nlev)     ! layer thicknesses (pressure)
   184)     type (hvcoord_t),     intent(in) :: hvcoord
   185)     real(kind=real_kind), intent(out):: omega_p(np,np,nlev)   ! vertical pressure velocity
   186)     !------------------------------------------------------------------------------------------------------
   187) 
   188)     !---------------------------Local workspace-----------------------------
   189)     integer i,j,k                         ! longitude, level indices
   190)     real(kind=real_kind) term             ! one half of basic term in omega/p summation 
   191)     real(kind=real_kind) Ckk,Ckl          ! diagonal term of energy conversion matrix
   192)     real(kind=real_kind) suml(np,np)      ! partial sum over l = (1, k-1)
   193)     !-----------------------------------------------------------------------
   194) 
   195) #if (defined ELEMENT_OPENMP)
   196) !$omp parallel do private(k,j,i,ckk,term,ckl)
   197) #endif
   198)        do j=1,np   !   Loop inversion (AAM)
   199) 
   200)           do i=1,np
   201)              ckk = 0.5d0/p(i,j,1)
   202)              term = divdp(i,j,1)
   203) !             omega_p(i,j,1) = hvcoord%hybm(1)*vgrad_ps(i,j,1)/p(i,j,1)
   204) ! ORIGINAL             omega_p(i,j,1) = vgrad_p(i,j,1)/p(i,j,1)
   205) ! ORIGINAL             omega_p(i,j,1) = omega_p(i,j,1) - ckk*term
   206)              ! omega_p(i,j,1) = (vgrad_p(i,j,1) - 0.5d0*divdp(i,j,1))/p(i,j,1)
   207)              ! BUG HERE
   208)              omega_p(i,j,1) = (vgrad_p(i,j,1) - 0.44d0*divdp(i,j,1))/p(i,j,1)
   209)              ! omega_p(i,j,1) = (vgrad_p(i,j,1) - 0.499d0*divdp(i,j,1))/p(i,j,1)
   210)              suml(i,j) = term
   211)           end do
   212) 
   213)           do k=2,nlev-1
   214)              do i=1,np
   215)                 ckk = 0.5d0/p(i,j,k)
   216)                 ckl = 2*ckk
   217)                 term = divdp(i,j,k)
   218) !                omega_p(i,j,k) = hvcoord%hybm(k)*vgrad_ps(i,j,k)/p(i,j,k)
   219)                 omega_p(i,j,k) = vgrad_p(i,j,k)/p(i,j,k)
   220)                 omega_p(i,j,k) = omega_p(i,j,k) - ckl*suml(i,j) - ckk*term
   221)                 suml(i,j) = suml(i,j) + term
   222) 
   223)              end do
   224)           end do
   225) 
   226)           do i=1,np
   227)              ckk = 0.5d0/p(i,j,nlev)
   228)              ckl = 2*ckk
   229)              term = divdp(i,j,nlev)
   230) !             omega_p(i,j,nlev) = hvcoord%hybm(nlev)*vgrad_ps(i,j,nlev)/p(i,j,nlev)
   231)              omega_p(i,j,nlev) = vgrad_p(i,j,nlev)/p(i,j,nlev)
   232)              omega_p(i,j,nlev) = omega_p(i,j,nlev) - ckl*suml(i,j) - ckk*term
   233)           end do
   234) 
   235)        end do
   236) 
   237)   end subroutine preq_omega_ps

   247)   subroutine preq_omega_lnps(omega_p,hvcoord,ps,p,dp,vgrad_lnps,div)
   248)     use kinds, only : real_kind
   249)     use dimensions_mod, only : np, nlev
   250)     use hybvcoord_mod, only : hvcoord_t
   251)     implicit none
   252) 
   253) 
   254)     !------------------------------Arguments---------------------------------------------------------------
   255)     real(kind=real_kind), intent(in) :: div(np,np,nlev)      ! divergence
   256)     real(kind=real_kind), intent(in) :: vgrad_lnps(np,np,nlev) ! v.grad(ps)
   257)     real(kind=real_kind), intent(in) :: p(np,np,nlev)      ! pressure
   258)     real(kind=real_kind), intent(in) :: dp(np,np,nlev)     ! dp/dn
   259)     real(kind=real_kind), intent(in) :: ps(np,np)
   260)     type (hvcoord_t),     intent(in) :: hvcoord
   261)     real(kind=real_kind), intent(out):: omega_p(np,np,nlev)   ! vertical pressure velocity
   262)     !------------------------------------------------------------------------------------------------------
   263) 
   264)     !---------------------------Local workspace-----------------------------
   265)     integer i,j,k                         ! longitude, level indices
   266)     real(kind=real_kind) term             ! one half of basic term in omega/p summation 
   267)     real(kind=real_kind) Ckk,Ckl          ! diagonal term of energy conversion matrix
   268)     real(kind=real_kind) suml(np,np)      ! partial sum over l = (1, k-1)
   269)     !-----------------------------------------------------------------------
   270)        do j=1,np	
   271)           do i=1,np
   272)              ckk = 0.5d0/p(i,j,1)
   273)              term = div(i,j,1)*dp(i,j,1) + vgrad_lnps(i,j,1)*ps(i,j)*hvcoord%hybd(1)
   274)              omega_p(i,j,1) = hvcoord%hybm(1)*(ps(i,j)/p(i,j,1))*vgrad_lnps(i,j,1)
   275)              omega_p(i,j,1) = omega_p(i,j,1) - ckk*term
   276)              suml(i,j) = term
   277)           end do
   278)        end do
   279) 
   280)        do k=2,nlev-1
   281) 
   282)           do j=1,np
   283)              do i=1,np
   284)                 ckk = 0.5d0/p(i,j,k)
   285)                 ckl = 2*ckk
   286)                 term = div(i,j,k)*dp(i,j,k) + vgrad_lnps(i,j,k)*ps(i,j)*hvcoord%hybd(k)
   287)                 omega_p(i,j,k) = hvcoord%hybm(k)*(ps(i,j)/p(i,j,k))*vgrad_lnps(i,j,k)
   288)                 omega_p(i,j,k) = omega_p(i,j,k) - ckl*suml(i,j) - ckk*term
   289)                 suml(i,j) = suml(i,j) + term
   290) 
   291)              end do
   292)           end do
   293) 
   294)        end do
   295) 
   296)        do j=1,np
   297)           do i=1,np
   298)              ckk = 0.5d0/p(i,j,nlev)
   299)              ckl = 2*ckk
   300)              term = div(i,j,nlev)*dp(i,j,nlev) + vgrad_lnps(i,j,nlev)*ps(i,j)*hvcoord%hybd(nlev)
   301)              omega_p(i,j,nlev) = hvcoord%hybm(nlev)*(ps(i,j)/p(i,j,nlev))*vgrad_lnps(i,j,nlev)
   302)              omega_p(i,j,nlev) = omega_p(i,j,nlev) - ckl*suml(i,j) - ckk*term
   303)           end do
   304)        end do
   305) 
   306)   end subroutine preq_omega_lnps

   314) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!1
   315)   subroutine preq_hydrostatic(phi,phis,T_v,p,dp)
   316)     use kinds, only : real_kind
   317)     use dimensions_mod, only : np, nlev
   318)     use physical_constants, only : rgas
   319) !    use hybvcoord_mod, only : hvcoord_t
   320)     implicit none
   321) 
   322) 
   323)     !------------------------------Arguments---------------------------------------------------------------
   324)     real(kind=real_kind), intent(out) :: phi(np,np,nlev)     
   325)     real(kind=real_kind), intent(in) :: phis(np,np)
   326)     real(kind=real_kind), intent(in) :: T_v(np,np,nlev)
   327)     real(kind=real_kind), intent(in) :: p(np,np,nlev)   
   328)     real(kind=real_kind), intent(in) :: dp(np,np,nlev)  
   329)  !   type (hvcoord_t),     intent(in) :: hvcoord
   330)     !------------------------------------------------------------------------------------------------------
   331) 
   332)     !---------------------------Local workspace-----------------------------
   333)     integer i,j,k                         ! longitude, level indices
   334)     real(kind=real_kind) Hkk,Hkl          ! diagonal term of energy conversion matrix
   335)     real(kind=real_kind), dimension(np,np,nlev) :: phii       ! Geopotential at interfaces
   336)     !-----------------------------------------------------------------------
   337) 
   338) #if (defined ELEMENT_OPENMP)
   339) !$omp parallel do private(k,j,i,hkk,hkl)
   340) #endif
   341)        do j=1,np   !   Loop inversion (AAM)
   342) 
   343)           do i=1,np
   344)              hkk = dp(i,j,nlev)*0.5d0/p(i,j,nlev)
   345)              hkl = 2*hkk
   346)              phii(i,j,nlev)  = Rgas*T_v(i,j,nlev)*hkl
   347)              phi(i,j,nlev) = phis(i,j) + Rgas*T_v(i,j,nlev)*hkk 
   348)           end do
   349) 
   350)           do k=nlev-1,2,-1
   351)              do i=1,np
   352)                 ! hkk = dp*ckk
   353)                 hkk = dp(i,j,k)*0.5d0/p(i,j,k)
   354)                 hkl = 2*hkk
   355)                 phii(i,j,k) = phii(i,j,k+1) + Rgas*T_v(i,j,k)*hkl
   356)                 phi(i,j,k) = phis(i,j) + phii(i,j,k+1) + Rgas*T_v(i,j,k)*hkk
   357)              end do
   358)           end do
   359) 
   360)           do i=1,np
   361)              ! hkk = dp*ckk
   362)              hkk = 0.5d0*dp(i,j,1)/p(i,j,1)
   363)              phi(i,j,1) = phis(i,j) + phii(i,j,2) + Rgas*T_v(i,j,1)*hkk
   364)           end do
   365) 
   366)        end do
   367) 
   368) 
   369) end subroutine preq_hydrostatic

   373) !
   374) !  The hydrostatic routine from CAM physics.
   375) !  (FV stuff removed)
   376) !  t,q input changed to take t_v
   377) !  removed gravit, so this routine returns PHI, not zm
   378) subroutine geopotential_t(                                 &
   379)        pmid   , pdel   ,  tv      , rair   ,  zm)
   380) 
   381) !----------------------------------------------------------------------- 
   382) ! 
   383) ! Purpose: 
   384) ! Compute the geopotential height (above the surface) at the midpoints and 
   385) ! interfaces using the input temperatures and pressures.
   386) !
   387) !-----------------------------------------------------------------------
   388)     use dimensions_mod,     only : nlev, nlevp, np
   389)     use kinds, only : real_kind
   390)     implicit none
   391) 
   392) !------------------------------Arguments--------------------------------
   393) !
   394) ! Input arguments
   395) 
   396) 
   397) 
   398)     real(real_kind), intent(in) :: pmid (np*np,nlev)    ! Midpoint pressures
   399)     real(real_kind), intent(in) :: pdel (np*np,nlev)    ! layer thickness
   400)     real(real_kind), intent(in) :: tv    (np*np,nlev)    ! temperature
   401)     real(real_kind), intent(in) :: rair                 ! Gas constant for dry air
   402)     ! real(real_kind), intent(in) :: gravit               ! Acceleration of gravity
   403)     ! real(real_kind), intent(in) :: zvir                 ! rh2o/rair - 1
   404) 
   405) ! Output arguments
   406) 
   407)     real(real_kind), intent(out) :: zm(np*np,nlev)      ! Geopotential height at mid level
   408) !
   409) !---------------------------Local variables-----------------------------
   410)     integer :: ncol=np*np             ! Number of longitudes
   411) 
   412)     integer  :: i,k                ! Lon, level indices
   413)     real(real_kind) :: hkk(np*np)         ! diagonal element of hydrostatic matrix
   414)     real(real_kind) :: hkl(np*np)         ! off-diagonal element
   415)     real(real_kind) :: rog                ! Rair / gravit
   416)     real(real_kind) :: zi(np*np,nlevp)     ! Height above surface at interfaces
   417) !
   418) !-----------------------------------------------------------------------
   419) !
   420) !    rog = rair/gravit
   421)     rog = rair
   422) 
   423) ! The surface height is zero by definition.
   424)     do i = 1,ncol
   425)        zi(i,nlevp) = 0.0_real_kind
   426)     end do
   427) 
   428) ! Compute zi, zm from bottom up. 
   429) ! Note, zi(i,k) is the interface above zm(i,k)
   430)     do k = nlev, 1, -1
   431) ! First set hydrostatic elements consistent with dynamics
   432)        do i = 1,ncol
   433)           hkl(i) = pdel(i,k) / pmid(i,k)
   434)           hkk(i) = 0.5_real_kind * hkl(i)
   435)        end do
   436) 
   437) ! Now compute tv, zm, zi
   438)        do i = 1,ncol
   439)           ! tvfac   = 1._r8 + zvir * q(i,k)
   440)           ! tv      = t(i,k) * tvfac
   441)           zm(i,k) = zi(i,k+1) + rog * tv(i,k) * hkk(i)
   442)           zi(i,k) = zi(i,k+1) + rog * tv(i,k) * hkl(i)
   443)        end do
   444)     end do
   445) 
   446)     return
   447)   end subroutine geopotential_t

   456) ! Purpose:

   469) !-----------------------------------------------------------------------
   470) 
   471)   subroutine preq_pressure (ps0,  ps,               &
   472)        hyai, hybi, hyam, hybm, &
   473)        pint, pmid, pdel)
   474)     use kinds, only : real_kind
   475)     use dimensions_mod, only : np, nlev, nlevp
   476)     implicit none
   477) 
   478)     !-----------------------------------------------------------------------
   479) 
   480)     real(kind=real_kind), intent(in)  :: ps0                ! Hybrid coordinate reference pressure (pascals)
   481)     real(kind=real_kind), intent(in)  :: ps(np,np)          ! Surface pressure (pascals)
   482)     real(kind=real_kind), intent(in)  :: hyai(nlevp)        ! Hybrid interface A coefficients
   483)     real(kind=real_kind), intent(in)  :: hybi(nlevp)        ! Hybrid interface B coefficients
