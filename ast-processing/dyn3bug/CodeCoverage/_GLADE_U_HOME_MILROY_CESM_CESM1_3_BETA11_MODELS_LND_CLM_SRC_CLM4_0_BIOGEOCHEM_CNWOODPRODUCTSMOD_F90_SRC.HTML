    36) subroutine CNWoodProducts(num_soilc, filter_soilc)

    39) ! Update all loss fluxes from wood product pools, and update product pool state variables
    40) ! for both loss and gain terms.  Gain terms are calculated in pftdyn_cnbal() for gains associated
    41) ! with changes in landcover, and in CNHarvest(), for gains associated with wood harvest.
    42) !
    43) ! !USES:
    44)    use clmtype
    45)    use clm_time_manager, only: get_step_size
    46)    use clm_varctl, only : use_c13
    47) !
    48) ! !ARGUMENTS:
    49)    implicit none
    50)    integer, intent(in) :: num_soilc       ! number of soil columns in filter
    51)    integer, intent(in) :: filter_soilc(:) ! filter for soil columns
    52) !
    53) ! !CALLED FROM:
    54) ! subroutine CNEcosystemDyn
    55) !
    56) ! !REVISION HISTORY:
    57) ! 5/21/09: Created by Peter Thornton
    58) !
    59) ! !LOCAL VARIABLES:

    61)    integer :: fc        ! lake filter indices
    62)    integer :: c         ! indices
    63)    real(r8):: dt        ! time step (seconds)
    64)    type(column_type),   pointer :: cptr         ! pointer to column derived subtype
    65)    real(r8) :: kprod10       ! decay constant for 10-year product pool
    66)    real(r8) :: kprod100      ! decay constant for 100-year product pool
    67) 
    68) !EOP
    69) !-----------------------------------------------------------------------
    70) 
    71)    cptr => col
    72) 	
    73)    ! calculate column-level losses from product pools
    74) 	! the following (1/s) rate constants result in ~90% loss of initial state over 10 and 100 years,
    75) 	! respectively, using a discrete-time fractional decay algorithm.
    76) 	kprod10 = 7.2e-9
    77) 	kprod100 = 7.2e-10
    78) 
    79)    do fc = 1,num_soilc
    80)       c = filter_soilc(fc)
    81) 
    82) 		! calculate fluxes (1/sec)
    83) 		ccf%prod10c_loss(c)    = ccs%prod10c(c)    * kprod10
    84) 		ccf%prod100c_loss(c)   = ccs%prod100c(c)   * kprod100
    85)                 if (use_c13) then
    86)                    cc13f%prod10c_loss(c)  = cc13s%prod10c(c)  * kprod10
    87)                    cc13f%prod100c_loss(c) = cc13s%prod100c(c) * kprod100
    88)                 end if
    89) 		cnf%prod10n_loss(c)    = cns%prod10n(c)    * kprod10
    90) 		cnf%prod100n_loss(c)   = cns%prod100n(c)   * kprod100
    91) 	end do
    92) 
    93)    ! set time steps
    94)    dt = real( get_step_size(), r8 )
    95) 
    96)    ! update wood product state variables
    97)    ! column loop
    98)    do fc = 1,num_soilc
    99)       c = filter_soilc(fc)
   100) 
   101)       ! column-level fluxes
   102) 
   103)       ! fluxes into wood product pools, from landcover change
   104)       ccs%prod10c(c)    = ccs%prod10c(c)    + ccf%dwt_prod10c_gain(c)*dt
   105)       ccs%prod100c(c)   = ccs%prod100c(c)   + ccf%dwt_prod100c_gain(c)*dt
   106)       if (use_c13) then
   107)          cc13s%prod10c(c)  = cc13s%prod10c(c)  + cc13f%dwt_prod10c_gain(c)*dt
   108)          cc13s%prod100c(c) = cc13s%prod100c(c) + cc13f%dwt_prod100c_gain(c)*dt
   109)       end if
   110)       cns%prod10n(c)    = cns%prod10n(c)    + cnf%dwt_prod10n_gain(c)*dt
   111)       cns%prod100n(c)   = cns%prod100n(c)   + cnf%dwt_prod100n_gain(c)*dt
   112) 
   113)       ! fluxes into wood product pools, from harvest
   114)       ccs%prod10c(c)    = ccs%prod10c(c)    + ccf%hrv_deadstemc_to_prod10c(c)*dt
   115)       ccs%prod100c(c)   = ccs%prod100c(c)   + ccf%hrv_deadstemc_to_prod100c(c)*dt
   116)       if (use_c13) then
   117)          cc13s%prod10c(c)  = cc13s%prod10c(c)  + cc13f%hrv_deadstemc_to_prod10c(c)*dt
   118)          cc13s%prod100c(c) = cc13s%prod100c(c) + cc13f%hrv_deadstemc_to_prod100c(c)*dt
   119)       end if
   120)       cns%prod10n(c)    = cns%prod10n(c)    + cnf%hrv_deadstemn_to_prod10n(c)*dt
   121)       cns%prod100n(c)   = cns%prod100n(c)   + cnf%hrv_deadstemn_to_prod100n(c)*dt
   122)      
   123)       ! fluxes out of wood product pools, from decomposition
   124)       ccs%prod10c(c)    = ccs%prod10c(c)    - ccf%prod10c_loss(c)*dt
   125)       ccs%prod100c(c)   = ccs%prod100c(c)   - ccf%prod100c_loss(c)*dt
   126)       if (use_c13) then
   127)          cc13s%prod10c(c)  = cc13s%prod10c(c)  - cc13f%prod10c_loss(c)*dt
   128)          cc13s%prod100c(c) = cc13s%prod100c(c) - cc13f%prod100c_loss(c)*dt
   129)       end if
   130)       cns%prod10n(c)    = cns%prod10n(c)    - cnf%prod10n_loss(c)*dt
   131)       cns%prod100n(c)   = cns%prod100n(c)   - cnf%prod100n_loss(c)*dt
   132)  
   133)    end do ! end of column loop
   134)
