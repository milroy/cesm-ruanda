    31)   ! ================================
    32)   ! global_integral:
    33)   !
    34)   ! eq 81 in Williamson, et. al. p 218
    35)   ! for spectral elements
    36)   !
    37)   ! ================================
    38)   ! --------------------------
    39)   function global_integral(elem, h,hybrid,npts,nets,nete) result(I_sphere)
    40)     use kinds,       only : real_kind
    41)     use hybrid_mod,  only : hybrid_t
    42)     use element_mod, only : element_t
    43)     use dimensions_mod, only : np, nelemd
    44)     use physical_constants, only : dd_pi
    45)     use parallel_mod, only: global_shared_buf, global_shared_sum
    46) 
    47)     type(element_t)      , intent(in) :: elem(:)
    48)     integer              , intent(in) :: npts,nets,nete
    49)     real (kind=real_kind), intent(in) :: h(npts,npts,nets:nete)
    50)     type (hybrid_t)      , intent(in) :: hybrid
    51) 
    52)     real (kind=real_kind) :: I_sphere
    53) 
    54)     real (kind=real_kind) :: I_priv
    55)     real (kind=real_kind) :: I_shared
    56)     common /gblintcom/I_shared
    57) 
    58)     ! Local variables
    59) 
    60)     integer :: ie,j,i
    61)     real(kind=real_kind) :: I_tmp(1)
    62) 
    63)     real (kind=real_kind) :: da
    64)     real (kind=real_kind) :: J_tmp(nets:nete)
    65) !
    66) ! This algorythm is independent of thread count and task count.
    67) ! This is a requirement of consistancy checking in cam.
    68) !
    69)     J_tmp = 0.0D0
    70) 
    71)        do ie=nets,nete
    72)           do j=1,np
    73)              do i=1,np
    74)                 da = elem(ie)%mp(i,j)*elem(ie)%metdet(i,j)
    75)                 J_tmp(ie) = J_tmp(ie) + da*h(i,j,ie)
    76)              end do
    77)           end do
    78)        end do       
    79)     do ie=nets,nete
    80)       global_shared_buf(ie,1) = J_tmp(ie)
    81)     enddo
    82)     call wrap_repro_sum(nvars=1, comm=hybrid%par%comm)
    83)     I_tmp = global_shared_sum(1)
    84)     I_sphere = I_tmp(1)/(4.0D0*DD_PI)
    85) 
    86)   end function global_integral

   110)     integer              , intent(in) :: nets,nete

   117)     ! Local variables
   118)     real (kind=real_kind), allocatable :: h(:,:,:)
   119)     ! Element statisics
   120)     real (kind=real_kind) :: min_area,max_area,avg_area
   121)     real (kind=real_kind) :: min_max_dx, max_max_dx, avg_max_dx
   122)     real (kind=real_kind) :: min_min_dx, max_min_dx, avg_min_dx

   124)     real (kind=real_kind) :: min_max_eig, max_max_eig, avg_max_eig

   141)     ! (Should be 1)
   142)     I_sphere = global_integral(elem, h(:,:,nets:nete),hybrid,np,nets,nete)
   143) 
   144)     min_area=1d99
   145)     max_area=0
   146)     avg_area=0_real_kind
   147) 
   148)     max_ratio = 0
   149) 
   150)     min_max_eig=1d99
   151)     max_max_eig=0
   152)     avg_max_eig=0_real_kind
   153) 
   154)     min_min_eig=1d99
   155)     max_min_eig=0
   156)     avg_min_eig=0_real_kind
   157) 
   158)     min_max_dx=1d99
   159)     max_max_dx=0
   160)     avg_max_dx=0_real_kind
   161) 
   162)     min_min_dx=1d99
   163)     max_min_dx=0
   164)     avg_min_dx=0_real_kind
   165) 
   166)     do ie=nets,nete
   167)        
   168)        elem(ie)%area = sum(elem(ie)%spheremp(:,:))
   169)        min_area=min(min_area,elem(ie)%area)
   170)        max_area=max(max_area,elem(ie)%area)
   171) 
   172)        min_max_eig = min(min_max_eig,elem(ie)%max_eig)
   173)        max_max_eig = max(max_max_eig,elem(ie)%max_eig)
   174) 
   175)        max_ratio   = max(max_ratio,elem(ie)%max_eig_ratio)

   177)        min_min_eig = min(min_min_eig,elem(ie)%min_eig)

   180)        min_max_dx = min(min_max_dx,elem(ie)%dx_long)
   181)        max_max_dx = max(max_max_dx,elem(ie)%dx_long)
   182) 
   183)        min_min_dx = min(min_min_dx,elem(ie)%dx_short)
   184)        max_min_dx = max(max_min_dx,elem(ie)%dx_short)
   185)        ! MNL: Want this output from all the uniform meshes
   186) 
   187) 
   188)        global_shared_buf(ie,1) = elem(ie)%area
   189)        global_shared_buf(ie,2) = elem(ie)%max_eig
   190)        global_shared_buf(ie,3) = elem(ie)%min_eig
   191)        global_shared_buf(ie,4) = elem(ie)%dx_long
   192)        global_shared_buf(ie,5) = elem(ie)%dx_short
   193) 
   194)        !      (Useful for setting level of variable hypervis)
   195)        !      (For now, only run on 1 proc if you want correct output!)
   196) !       if (elem(ie)%dx_short.eq.min_min_dx) then
   197) !           max_dx_at_min_dx = elem(ie)%dx_long
   198) !           max_dx_at_min_dx_id = elem(ie)%GlobalId
   199) !       end if
   200) 
   201)     enddo
   202)     ! MNL See comment above re: output from uniform meshes
   203) !    print*, "Min dx_short: ", min_min_dx
   204) !    print*, "Occurs at elem ", max_dx_at_min_dx_id, " where dx_long = ", max_dx_at_min_dx
   205) 
   206)     min_area=ParallelMin(min_area,hybrid)
   207)     max_area=ParallelMax(max_area,hybrid)
   208) 
   209)     min_max_eig=ParallelMin(min_max_eig,hybrid)
   210)     max_max_eig=ParallelMax(max_max_eig,hybrid)
   211) 
   212)     max_ratio=ParallelMax(max_ratio,hybrid)
   213) 
   214)     min_min_eig=ParallelMin(min_min_eig,hybrid)
   215)     max_min_eig=ParallelMax(max_min_eig,hybrid)
   216) 
   217)     min_max_dx=ParallelMin(min_max_dx,hybrid)
   218)     max_max_dx=ParallelMax(max_max_dx,hybrid)
   219) 
   220)     min_min_dx=ParallelMin(min_min_dx,hybrid)
   221)     max_min_dx=ParallelMax(max_min_dx,hybrid)
   222) 
   223)     call wrap_repro_sum(nvars=5, comm=hybrid%par%comm)
   224) 
   225)     avg_area = global_shared_sum(1)/dble(nelem)
   226)     avg_max_eig = global_shared_sum(2)/dble(nelem)
   227)     avg_min_eig = global_shared_sum(3)/dble(nelem)
   228)     avg_max_dx = global_shared_sum(4)/dble(nelem)
   229)     avg_min_dx = global_shared_sum(5)/dble(nelem)
   230) 
   231)     ! Physical units for area
   232)     min_area = min_area*rearth*rearth/1000000_real_kind
   233)     max_area = max_area*rearth*rearth/1000000_real_kind
   234)     avg_area = avg_area*rearth*rearth/1000000_real_kind
   235) 
   236) 
   237)     ! for an equation du/dt = i c u, leapfrog is stable for |c u dt| < 1
   238)     ! Consider a gravity wave at the equator, c=340m/s  
   239)     ! u = exp(i kmax x/ a ) with x = longitude,  and kmax =  pi a / dx, 
   240)     ! u = exp(i pi x / dx ),   so du/dt = c du/dx becomes du/dt = i c pi/dx u
   241)     ! stable for dt < dx/(c*pi)
   242)     ! CAM 26 level AMIP simulation: max gravity wave speed 341.75 m/s
   243)     if (hybrid%masterthread) then
   244)        write(iulog,* )""
   245)        write(iulog,* )"Running Global Integral Diagnostic..."
   246)        write(iulog,*)"Area of unit sphere is",I_sphere
   247)        write(iulog,*)"Should be 1.0 to round off..."
   248)        write(iulog,'(a,f9.3)') 'Element area:  max/min',(max_area/min_area)
   249)        if (.not.MeshUseMeshFile) then 
   250)            write(iulog,'(a,f6.3,f8.2)') "Average equatorial node spacing (deg, km) = ", &

   253)        write(iulog,'(a,2f9.3)') 'Min eigenvalue of Dinv (min, max): ', min_min_eig, max_min_eig

   256)        write(iulog,'(a,3f8.2)') 'dx for CFL (based on Dinv eigenvalue): ave,min,max = ', avg_min_dx, min_min_dx, max_min_dx
   257) !       write(iulog,'(a,3f8.2)') 'dx for hypervis (largest scale per elem): ave,min,max = ', avg_max_dx, min_max_dx, max_max_dx
   258)        write(iulog,'(a,3f8.2)') "dx based on sqrt element area: ave,min,max = ", &
   259)                 sqrt(avg_area)/(np-1),sqrt(min_area)/(np-1),sqrt(max_area)/(np-1)
   260)     end if
   261) 
   262)     deallocate(h)
   263)     
   264)     if(present(mindxout)) then
   265)         ! min_len now based on norm(Dinv)
   266)         mindxout=1000_real_kind*min_len
   267)         min_len = 0.002d0*rearth/(dble(np-1)*max_max_eig)
   268)     end if
   269) 
   270)   end subroutine test_global_integral

   284)   subroutine print_cfl(elem,hybrid,nets,nete,dtnu)
   285) !
   286) !   estimate various CFL limits
   287) !   also, for variable resolution viscosity coefficient, make sure
   288) !   worse viscosity CFL (given by dtnu) is not violated by reducing 
   289) !   viscosity coefficient in regions where CFL is violated
   290) !
   291)     use kinds,       only : real_kind
   292)     use hybrid_mod,  only : hybrid_t
   293)     use element_mod, only : element_t
   294)     use dimensions_mod, only : np,ne,nelem,nelemd
   295)     use quadrature_mod, only : gausslobatto, quadrature_t
   296) 
   297)     use reduction_mod, only : ParallelMin,ParallelMax
   298)     use physical_constants, only : rrearth, rearth,dd_pi
   299)     use control_mod, only : nu, nu_q, nu_div, hypervis_order, nu_top, hypervis_power, &
   300)                             fine_ne, rk_stage_user, max_hypervis_courant, hypervis_scaling
   301)     use parallel_mod, only : abortmp, global_shared_buf, global_shared_sum
   302)     use edge_mod, only : EdgeBuffer_t, initedgebuffer, FreeEdgeBuffer, edgeVpack, edgeVunpack
   303)     use bndry_mod, only : bndry_exchangeV
   304)     use time_mod, only : tstep
   305) 
   306)     type(element_t)      , intent(inout) :: elem(:)
   307)     integer              , intent(in) :: nets,nete
   308)     type (hybrid_t)      , intent(in) :: hybrid
   309)     real (kind=real_kind), intent(in) :: dtnu  
   310) 
   311)     ! Element statisics
   312)     real (kind=real_kind) :: min_max_dx, max_max_dx, avg_max_dx
   313)     real (kind=real_kind) :: min_min_dx, max_min_dx, avg_min_dx
   314)     real (kind=real_kind) :: max_unif_dx
   315)     real (kind=real_kind) :: min_max_eig, max_max_eig, avg_max_eig
   316)     real (kind=real_kind) :: min_min_eig, max_min_eig, avg_min_eig
   317)     real (kind=real_kind) :: min_hypervis, max_hypervis, avg_hypervis, stable_hv
   318)     real (kind=real_kind) :: max_eig_hypervis
   319)     real (kind=real_kind) :: x, y, noreast, nw, se, sw
   320)     real (kind=real_kind), dimension(np,np,nets:nete) :: zeta
   321)     real (kind=real_kind) :: lambda_max, lambda_vis, min_gw, lambda, nu_div_actual
   322)     integer :: ie,corner, i, j, rowind, colind
   323)     type (quadrature_t)    :: gp
   324) 
   325) 
   326)     ! Eigenvalues calculated by folks at UMich (Paul U & Jared W)
   327)     select case (np)
   328)         case (2)
   329)             lambda_max = 0.5d0
   330)         case (3)
   331)             lambda_max = 1.5d0
   332)             lambda_vis = 12.0d0
   333)         case (4)
   334)             lambda_max = 2.74d0
   335)             lambda_vis = 30.0d0
   336)         case (5)
   337)             lambda_max = 4.18d0
   338)             lambda_vis = 91.6742d0
   339)         case (6)
   340)             lambda_max = 5.86d0
   341)             lambda_vis = 190.1176d0
   342)         case (7)
   343)             lambda_max = 7.79d0
   344)             lambda_vis = 374.7788d0
   345)         case (8)
   346)             lambda_max = 10.0d0
   347)             lambda_vis = 652.3015d0
   348)         case DEFAULT
   349)             lambda_max = 0.0d0
   350)             lambda_vis = 0.0d0
   351)     end select
   352) 
   353)     if ((lambda_max.eq.0d0).and.(hybrid%masterthread)) then
   354)         print*, "lambda_max not calculated for NP = ",np
   355)         print*, "Estimate of gravity wave timestep will be incorrect"
   356)     end if
   357) 
   358)     do ie=nets,nete
   359)       elem(ie)%variable_hyperviscosity = 1.0
   360)     end do
   361) 
   362)     gp=gausslobatto(np)
   363)     min_gw = minval(gp%weights)
   364) 
   365)     min_max_eig=1d99
   366)     max_max_eig=0
   367)     avg_max_eig=0_real_kind
   368) 
   369)     min_min_eig=1d99
   370)     max_min_eig=0
   371)     avg_min_eig=0_real_kind
   372) 
   373)     min_max_dx=1d99
   374)     max_max_dx=0
   375)     avg_max_dx=0_real_kind
   376) 
   377)     min_min_dx=1d99
   378)     max_min_dx=0
   379)     avg_min_dx=0_real_kind
   380) 
   381)     do ie=nets,nete
   382)         min_max_eig = min(min_max_eig,elem(ie)%max_eig)
   383)         max_max_eig = max(max_max_eig,elem(ie)%max_eig)
   384) 
   385)         min_min_eig = min(min_min_eig,elem(ie)%min_eig)
   386)         max_min_eig = max(max_min_eig,elem(ie)%min_eig)
   387) 
   388)         min_max_dx = min(min_max_dx,elem(ie)%dx_long)
   389)         max_max_dx = max(max_max_dx,elem(ie)%dx_long)
   390) 
   391)         min_min_dx = min(min_min_dx,elem(ie)%dx_short)
   392)         max_min_dx = max(max_min_dx,elem(ie)%dx_short)
   393) 
   394)        global_shared_buf(ie,1) = elem(ie)%max_eig
   395)        global_shared_buf(ie,2) = elem(ie)%min_eig
   396)        global_shared_buf(ie,3) = elem(ie)%dx_long
   397)        global_shared_buf(ie,4) = elem(ie)%dx_short
   398)     enddo
   399) 
   400)     min_max_eig=ParallelMin(min_max_eig,hybrid)
   401)     max_max_eig=ParallelMax(max_max_eig,hybrid)
   402) 
   403)     min_min_eig=ParallelMin(min_min_eig,hybrid)
   404)     max_min_eig=ParallelMax(max_min_eig,hybrid)
   405) 
   406)     min_max_dx=ParallelMin(min_max_dx,hybrid)
   407)     max_max_dx=ParallelMax(max_max_dx,hybrid)
   408) 
   409)     min_min_dx=ParallelMin(min_min_dx,hybrid)
   410)     max_min_dx=ParallelMax(max_min_dx,hybrid)
   411) 
   412)     call wrap_repro_sum(nvars=4, comm=hybrid%par%comm)
   413) 
   414)     avg_max_eig = global_shared_sum(1)/dble(nelem)
   415)     avg_min_eig = global_shared_sum(2)/dble(nelem)
   416)     avg_max_dx = global_shared_sum(3)/dble(nelem)
   417)     avg_min_dx = global_shared_sum(4)/dble(nelem)
   418) 
   419) 
   420)     if (hypervis_power /= 0) then
   421) 
   422)         min_hypervis = 1d99
   423)         max_hypervis = 0
   424)         avg_hypervis = 0
   425) 
   426) !       viscosity in namelist specified for smallest element:
   427)         max_unif_dx = min_max_dx
   428)         if (fine_ne>0) then
   429)            ! viscosity in namelist specified for regions with a resolution
   430)            ! equivilant to a uniform grid with ne=fine_ne
   431) !!! og: is this in km? np=4? yes
   432)            max_unif_dx = (111.28*30)/dble(fine_ne)
   433)         endif
   434) 
   435) ! 
   436) ! note: if L = eigenvalue of metinv, then associated length scale (km) is
   437) ! dx = 1.0d0/( sqrt(L)*0.5d0*dble(np-1)*rrearth*1000.0d0)
   438) !
   439) !       for viscosity *tensor*, we take at each point: 
   440) !            nu1 = nu*(dx1/max_unif_dx)**3.2      dx1 associated with eigenvalue 1
   441) !            nu2 = nu*(dx2/max_unif_dx)**3.2      dx2 associated with eigenvalue 2
   442) !       with this approach:
   443) !          - with this formula, no need to adjust for CFL violations
   444) !          - if nu comes from a 3.2 scaling that is stable for coarse and fine resolutions,
   445) !            this formulat will be stable.  
   446) !          - gives the correct answer in long skinny rectangles:
   447) !            large viscosity in the long direction, small viscosity in the short direction 
   448) !            
   449) !
   450) 
   451)         max_eig_hypervis = 0
   452) 
   453)         do ie=nets,nete
   454)            ! variable viscosity based on map from ulatlon -> ucontra
   455) 
   456)            ! dx_long
   457)            elem(ie)%variable_hyperviscosity = sqrt((elem(ie)%dx_long/max_unif_dx) ** hypervis_power)
   458)            ! dx_short
   459)            !elem(ie)%variable_hyperviscosity = sqrt((elem(ie)%dx_short/1.25/min_min_dx) ** hypervis_power)
   460) 
   461)            ! geometric mean:
   462)            !elem(ie)%variable_hyperviscosity = sqrt(  (sqrt(elem(ie)%dx_short*elem(ie)%dx_long)/min_min_dx/1.9) ** hypervis_power)
   463) 
   464)            ! average
   465)            !elem(ie)%variable_hyperviscosity = sqrt( ( (elem(ie)%dx_short+elem(ie)%dx_long)/3.8/min_min_dx) ** hypervis_power)
   466) 
   467)            ! sqrt(area)
   468)            ! area = unit sphere.  other variables in km
   469)            !elem(ie)%variable_hyperviscosity = sqrt( ( sqrt(elem(ie)%area)*Rearth/2.4e3/max_unif_dx) ** hypervis_power)
   470)            
   471)            
   472)            elem(ie)%hv_courant = dtnu*(elem(ie)%variable_hyperviscosity(1,1)**2) * (lambda_vis**2) * ((rrearth*elem(ie)%max_eig)**4)
   473) 
   474)             ! Check to see if this is stable
   475)             if (elem(ie)%hv_courant.gt.max_hypervis_courant) then
   476)                 stable_hv = sqrt( max_hypervis_courant / &
   477)                      (  dtnu * (lambda_vis)**2 * (rrearth*elem(ie)%max_eig)**4 ) )
   478) 
   479) #if 0
   480)          ! Useful print statements for debugging the adjustments to hypervis 
   481)                  print*, "Adjusting hypervis on elem ", elem(ie)%GlobalId
   482)                  print*, "From ", nu*elem(ie)%variable_hyperviscosity(1,1)**2, " to ", nu*stable_hv
   483)                  print*, "Difference = ", nu*(/elem(ie)%variable_hyperviscosity(1,1)**2-stable_hv/)
   484)                  print*, "Factor of ", elem(ie)%variable_hyperviscosity(1,1)**2/stable_hv
   485)                  print*, " "
   486) #endif
   487) 
   488) !                make sure that: elem(ie)%hv_courant <=  max_hypervis_courant 
   489)                 elem(ie)%variable_hyperviscosity = stable_hv
   490)                 elem(ie)%hv_courant = dtnu*(stable_hv**2) * (lambda_vis)**2 * (rrearth*elem(ie)%max_eig)**4               
   491)             end if
   492)             max_eig_hypervis = max(max_eig_hypervis, elem(ie)%hv_courant/dtnu)
   493) 
   494)             min_hypervis = min(min_hypervis, elem(ie)%variable_hyperviscosity(1,1))
   495)             max_hypervis = max(max_hypervis, elem(ie)%variable_hyperviscosity(1,1))
   496)             global_shared_buf(ie,1) = elem(ie)%variable_hyperviscosity(1,1)
   497)         end do
   498) 
   499)         min_hypervis = ParallelMin(min_hypervis, hybrid)
   500)         max_hypervis = ParallelMax(max_hypervis, hybrid)
   501)         call wrap_repro_sum(nvars=1, comm=hybrid%par%comm)
   502)         avg_hypervis = global_shared_sum(1)/dble(nelem)
   503) 
   504)         max_eig_hypervis = ParallelMax(max_eig_hypervis, hybrid)
   505) 
   506)         ! apply DSS (aka assembly procedure) to variable_hyperviscosity (makes continuous)
   507)         call initEdgeBuffer(edgebuf,1)
   508)         do ie=nets,nete
   509)             zeta(:,:,ie) = elem(ie)%variable_hyperviscosity(:,:)*elem(ie)%spheremp(:,:)
   510)             call edgeVpack(edgebuf,zeta(1,1,ie),1,0,elem(ie)%desc)
   511)         end do
   512)         call bndry_exchangeV(hybrid,edgebuf)
   513)         do ie=nets,nete
   514)             call edgeVunpack(edgebuf,zeta(1,1,ie),1,0,elem(ie)%desc)
   515)             elem(ie)%variable_hyperviscosity(:,:) = zeta(:,:,ie)*elem(ie)%rspheremp(:,:)
   516)         end do
   517)         call FreeEdgeBuffer(edgebuf)
   518) 
   519)         ! replace hypervis w/ bilinear based on continuous corner values
   520)         do ie=nets,nete
   521)             noreast = elem(ie)%variable_hyperviscosity(np,np)
   522)             nw = elem(ie)%variable_hyperviscosity(1,np)
   523)             se = elem(ie)%variable_hyperviscosity(np,1)
   524)             sw = elem(ie)%variable_hyperviscosity(1,1)
   525)             do i=1,np
   526)                 x = gp%points(i)
   527)                 do j=1,np
   528)                     y = gp%points(j)
   529)                     elem(ie)%variable_hyperviscosity(i,j) = 0.25d0*( &
   530)                                             (1.0d0-x)*(1.0d0-y)*sw + &
   531)                                             (1.0d0-x)*(y+1.0d0)*nw + &
   532)                                             (x+1.0d0)*(1.0d0-y)*se + &
   533)                                             (x+1.0d0)*(y+1.0d0)*noreast)
   534)                 end do
   535)             end do
   536)         end do
   537)     else  if (hypervis_scaling/=0) then
   538)        ! tensorHV.  New eigenvalues are the eigenvalues of the tensor V
   539)        ! formulas here must match what is in cube_mod.F90
   540)        ! for tensorHV, we scale out the rearth dependency
   541)        lambda = max_max_eig**2
   542)        max_eig_hypervis = (lambda_vis**2) * (max_max_eig**4) * &
   543)             (lambda**(-hypervis_scaling/2) )
   544)     else
   545)        ! constant coefficient formula:
   546)        max_eig_hypervis = (lambda_vis**2) * (rrearth*max_max_eig)**4
   547)     endif
   548) 
   549) 
   550)    if (hypervis_scaling /= 0) then
   551) !this is a code for smoothing V for tensor HV
   552) 
   553) !if nu=0, we have extra 4 DSS here!
   554) !rowind, colind are from 1 to 2 cause they correspond to 2D tensor in lat/lon
   555) 
   556) !IF DSSED V NEEDED
   557) #if 1
   558)     call initEdgeBuffer(edgebuf,1)
   559)     do rowind=1,2
   560)       do colind=1,2
   561) 	do ie=nets,nete
   562) 	  zeta(:,:,ie) = elem(ie)%tensorVisc(rowind,colind,:,:)*elem(ie)%spheremp(:,:)
   563) 	  call edgeVpack(edgebuf,zeta(1,1,ie),1,0,elem(ie)%desc)
   564) 	end do
   565) 
   566) 	call bndry_exchangeV(hybrid,edgebuf)
   567) 	do ie=nets,nete
   568) 	  call edgeVunpack(edgebuf,zeta(1,1,ie),1,0,elem(ie)%desc)
   569)           elem(ie)%tensorVisc(rowind,colind,:,:) = zeta(:,:,ie)*elem(ie)%rspheremp(:,:)
   570) 	end do
   571)       enddo !rowind
   572)     enddo !colind
   573)     call FreeEdgeBuffer(edgebuf)
   574) #endif
   575) 
   576) !IF BILINEAR MAP OF V NEEDED
   577) #if 1
   578)     do rowind=1,2
   579)       do colind=1,2
   580)     ! replace hypervis w/ bilinear based on continuous corner values
   581) 	do ie=nets,nete
   582) 	  noreast = elem(ie)%tensorVisc(rowind,colind,np,np)
   583) 	  nw = elem(ie)%tensorVisc(rowind,colind,1,np)
   584) 	  se = elem(ie)%tensorVisc(rowind,colind,np,1)
   585) 	  sw = elem(ie)%tensorVisc(rowind,colind,1,1)
   586) 	  do i=1,np
   587) 	    x = gp%points(i)
   588) 	    do j=1,np
   589) 		y = gp%points(j)
   590) 		elem(ie)%tensorVisc(rowind,colind,i,j) = 0.25d0*( &
   591) 					(1.0d0-x)*(1.0d0-y)*sw + &
   592) 					(1.0d0-x)*(y+1.0d0)*nw + &
   593) 					(x+1.0d0)*(1.0d0-y)*se + &
   594) 					(x+1.0d0)*(y+1.0d0)*noreast)
   595) 	    end do
   596) 	  end do
   597) 	end do
   598)       enddo !rowind
   599)     enddo !colind
   600) #endif
   601)     endif
   602) 
   603) 
   604) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   605) 
   606)     deallocate(gp%points)
   607)     deallocate(gp%weights)
   608) 
   609)      if (hybrid%masterthread) then
   610)        write(iulog,'(a,f10.2)') 'CFL estimates in terms of S=time step stability region'
   611)        write(iulog,'(a,f10.2)') '(i.e. advection w/leapfrog: S=1, viscosity w/forward Euler: S=2)'
   612)        if (rk_stage_user>0) then
   613)           write(iulog,'(a,f10.2,a)') 'SSP preservation (120m/s) RKSSP euler step dt  < S *', &
   614)                min_gw/(120.0d0*max_max_eig*rrearth),'s'
   615)        endif
   616)        write(iulog,'(a,f10.2,a)') 'Stability: advective (120m/s)   dt_tracer < S *', 1/(120.0d0*max_max_eig*lambda_max*rrearth),'s'
   617)        write(iulog,'(a,f10.2,a)') 'Stability: gravity wave(342m/s)   dt_dyn  < S *', &
   618)             1/(342.0d0*max_max_eig*lambda_max*rrearth),'s'
   619)        if (nu>0) then
   620)           if (hypervis_order==1) then
   621)               write(iulog,'(a,f10.2,a)') 'Stability: viscosity dt < S *',1/(((rrearth*max_max_eig)**2)*lambda_vis),'s'
   622)           endif
   623)           if (hypervis_order==2) then
   624)              ! counrant number = dtnu*max_eig_hypervis  < S
   625)              !  dt < S  1/nu*max_eig
   626)              write(iulog,'(a,f10.2,a)') "Stability: nu_q   hyperviscosity dt < S *", 1/(nu_q*max_eig_hypervis),'s'
   627)              write(iulog,'(a,f10.2,a)') "Stability: nu_vor hyperviscosity dt < S *", 1/(nu*max_eig_hypervis),'s'
   628)              
   629)              ! bug in nu_div implimentation:
   630)              ! we apply nu_ration=(nu_div/nu) in laplace, so it is applied 2x
   631)              ! making the effective nu_div = nu * (nu_div/nu)**2 
   632)              ! should be fixed - but need all CAM defaults adjusted, 
   633)              ! so we have to coordiante with CAM
   634)              nu_div_actual = nu_div**2/nu
   635)              write(iulog,'(a,f10.2,a)') "Stability: nu_div hyperviscosity dt < S *", 1/(nu_div_actual*max_eig_hypervis),'s'
   636)           endif
   637)        endif
   638)        if(nu_top>0) then
   639)           write(iulog,'(a,f10.2,a)') 'TOP3 viscosity CFL: dt < S*', &
   640)                                   1.0d0/(4*nu_top*((rrearth*max_max_eig)**2)*lambda_vis),'s'
   641)        end if
   642)       if (hypervis_power /= 0) then 
   643)         write(iulog,'(a,3e11.4)')'Hyperviscosity (dynamics): ave,min,max = ', &
   644)                                   nu*(/avg_hypervis**2,min_hypervis**2,max_hypervis**2/)
   645) !         print*, 'fine_ne = ', fine_ne
   646) !         print*, 'Using max_unif_dx = ', max_unif_dx
   647)       end if
   648) 
   649)     end if
