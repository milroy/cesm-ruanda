     1) module phys_debug
     2) 
     3) ! This module contains subroutines that are intended to print diagnostic
     4) ! information about a single column to the log file.  The module uses the
     5) ! phys_debug_util module to locate the column using local indices of the
     6) ! physics decomposition.  The reason to encapsulate the debug print
     7) ! statements into a subroutine is to minimize the impact of the debugging
     8) ! code on the development code.  This makes it easier to maintain debug
     9) ! code on a branch which can be updated to the head of the trunk when
    10) ! convenient.  Minimizing the footprint of debug code helps to minimize the
    11) ! chances of conflicts that can occur while updating between the debug code
    12) ! and changes that have occurred on the trunk.
    13) 
    14) ! This module also contains some generic code to write a collection of variables
    15) ! from the physics state object to the history file.
    16)

    61) subroutine phys_debug_state_init(tags)
    62) 
    63)    ! Write selected state variables from locations corresponding to the 'tags'.
    64)    ! This method does the addfld/add_default calls.
    65) 
    66)    ! ** Note ** This init needs to be called after the rad_constituent module has finished
    67)    !            its init.
    68) 
    69) 
    70)    character(len=8), dimension(:), intent(in) :: tags
    71) 
    72)    ! local variables
    73)    integer :: i, icnst, j, nmodes, nspec, ntags
    74)    character(len=5) :: unit
    75)    !-----------------------------------------------------------------------------------------------
    76) 
    77)    ! Initialize some constituent indices and names
    78)    call cnst_get_ind('CLDLIQ', ixcldliq, abort=.false.)
    79)    call cnst_get_ind('CLDICE', ixcldice, abort=.false.)
    80)    call cnst_get_ind('NUMLIQ', ixnumliq, abort=.false.)
    81)    call cnst_get_ind('NUMICE', ixnumice, abort=.false.)
    82) 
    83)    ! Set arrays to identify the modal aerosol constituents
    84)    cnst_is_mam_num = .false.
    85)    cnst_is_mam_mmr = .false.
    86)    call rad_cnst_get_info(0, nmodes=nmodes)
    87)    do i = 1, nmodes
    88)       call rad_cnst_get_mode_num_idx(i, icnst)
    89)       cnst_is_mam_num(icnst) = .true.
    90)       call rad_cnst_get_info(0, i, nspec=nspec)
    91)       do j = 1, nspec
    92)          call rad_cnst_get_mam_mmr_idx(i, j, icnst)
    93)          cnst_is_mam_mmr(icnst) = .true.
    94)       end do
    95)    end do
    96) 
    97)    ! For each supplied tag, issue addfld/add_default calls for the state variables
    98)    ! with the tag name appended.
    99)    ntags = size(tags)
   100)    do i = 1, ntags
   101) 
   102)       call addfld('T_'//trim(tags(i)), 'K', pver, 'A', 'T at tag '//trim(tags(i)), phys_decomp)
   103)       call add_default('T_'//trim(tags(i)), 1, ' ')
   104) 
   105)       call addfld('U_'//trim(tags(i)), 'm/s', pver, 'A', 'U at tag '//trim(tags(i)), phys_decomp)
   106)       call add_default('U_'//trim(tags(i)), 1, ' ')
   107) 
   108)       call addfld('V_'//trim(tags(i)), 'm/s', pver, 'A', 'V at tag '//trim(tags(i)), phys_decomp)
   109)       call add_default('V_'//trim(tags(i)), 1, ' ')
   110) 
   111)       call addfld('QV_'//trim(tags(i)), 'kg/kg', pver, 'A', 'QV at tag '//trim(tags(i)), phys_decomp)
   112)       call add_default('QV_'//trim(tags(i)), 1, ' ')
   113) 
   114)       if (ixcldliq > 0) then
   115)          call addfld('QL_'//trim(tags(i)), 'kg/kg', pver, 'A', 'QL at tag '//trim(tags(i)), phys_decomp)
   116)          call add_default('QL_'//trim(tags(i)), 1, ' ')
   117)       end if
   118) 
   119)       if (ixcldice > 0) then
   120)          call addfld('QI_'//trim(tags(i)), 'kg/kg', pver, 'A', 'QI at tag '//trim(tags(i)), phys_decomp)
   121)          call add_default('QI_'//trim(tags(i)), 1, ' ')
   122)       end if
   123) 
   124)       if (ixnumliq > 0) then
   125)          call addfld('NL_'//trim(tags(i)), '1/kg', pver, 'A', 'NL at tag '//trim(tags(i)), phys_decomp)
   126)          call add_default('NL_'//trim(tags(i)), 1, ' ')
   127)       end if
   128) 
   129)       if (ixnumice > 0) then
   130)          call addfld('NI_'//trim(tags(i)), '1/kg', pver, 'A', 'NI at tag '//trim(tags(i)), phys_decomp)
   131)          call add_default('NI_'//trim(tags(i)), 1, ' ')
   132)       end if
   133) 
   134)       do j = 1, pcnst
   135) 
   136)          if (cnst_is_mam_num(j) .or. cnst_is_mam_mmr(j)) then
   137) 
   138)             unit = '1/kg'
   139)             if (cnst_is_mam_mmr(j)) unit = 'kg/kg'
   140) 
   141)             call addfld(trim(cnst_name(j))//'_'//trim(tags(i)), trim(unit), pver, &
   142)                'A', trim(cnst_name(j))//' at tag '//trim(tags(i)), phys_decomp)
   143)             call add_default(trim(cnst_name(j))//'_'//trim(tags(i)), 1, ' ')
   144) 
   145)          end if
   146)       end do
   147) 
   148)    end do
   149) 
   150) end subroutine phys_debug_state_init

   154) subroutine phys_debug_state_out(tag, state)
   155) 
   156)    ! Dump selected state variables at location corresponding to given tag.
   157) 
   158)    character(len=*),    intent(in) :: tag
   159)    type(physics_state), intent(in) :: state 
   160) 
   161)    ! Local variables
   162)    integer :: lchnk, j
   163)    !-----------------------------------------------------------------------------------------------
   164) 
   165)    lchnk = state%lchnk
   166) 
   167)    call outfld('T_'//trim(tag),  state%t,        pcols, lchnk)
   168)    call outfld('U_'//trim(tag),  state%u,        pcols, lchnk)
   169)    call outfld('V_'//trim(tag),  state%v,        pcols, lchnk)
   170)    call outfld('QV_'//trim(tag), state%q(:,:,1), pcols, lchnk)
   171)    if (ixcldliq > 0) call outfld('QL_'//trim(tag), state%q(:,:,ixcldliq), pcols, lchnk)
   172)    if (ixcldice > 0) call outfld('QI_'//trim(tag), state%q(:,:,ixcldice), pcols, lchnk)
   173)    if (ixnumliq > 0) call outfld('NL_'//trim(tag), state%q(:,:,ixnumliq), pcols, lchnk)
   174)    if (ixnumice > 0) call outfld('NI_'//trim(tag), state%q(:,:,ixnumice), pcols, lchnk)
   175) 
   176)    do j = 1, pcnst
   177)       if (cnst_is_mam_num(j) .or. cnst_is_mam_mmr(j)) then
   178)          call outfld(trim(cnst_name(j))//'_'//trim(tag), state%q(:,:,j), pcols, lchnk)
   179)       end if
   180)    end do
   181) 
   182) end subroutine phys_debug_state_out

   209) subroutine phys_debug_shallow1(state, ptend, nstep, prec_cmf, rliq2, ztodt, kmx)
   210) 
   211)    use constituents, only: cnst_get_ind
   212) 
   213)    type(physics_state), intent(in) :: state             ! Physics state variables
   214)    type(physics_ptend), intent(in) :: ptend             ! Physics process tendencies
   215)    integer,             intent(in) :: nstep
   216)    real(r8),            intent(in) :: prec_cmf(pcols)   ! total precipitation from Hack convection
   217)    real(r8),            intent(in) :: rliq2(pcols)      ! vertical integral of liquid from shallow scheme
   218)    real(r8),            intent(in) :: ztodt             ! physics time step
   219)    integer,            intent(out) :: kmx
   220) 
   221)    integer  :: icol, k, kmn, ixcldliq
   222)    real(r8) :: qtmx, qtmn
   223)    !-----------------------------------------------------------------------------
   224) 
   225)    icol = phys_debug_col(state%lchnk) 
   226)    if (icol > 0) then
   227)       qtmx = 0._r8
   228)       kmx = 0
   229)       qtmn = 0._r8
   230)       kmn = 0
   231)       do k = 1,pver
   232)          !        write (iulog,*) 'aaa ', k, ptend%q(icol,k,1), ptend%q(icol,k,2)
   233)          if (ptend%q(icol,k,1).ge.qtmx) then
   234)             kmx = k
   235)             qtmx = ptend%q(icol,k,1)
   236)          endif
   237)          if (ptend%q(icol,k,1).le.qtmn) then
   238)             kmn = k
   239)             qtmn = ptend%q(icol,k,1)
   240)          endif
   241)       end do
   242)       k = kmx
   243) 66    format ('tphysbc, aft shallow:', 4i5, 6f9.4) 
   244)       call cnst_get_ind('CLDLIQ', ixcldliq, abort=.false.)
   245)       write (iulog,66)  nstep, icol, &
   246)            kmx, kmn, &
   247)            prec_cmf(icol)*8.64e7_r8, rliq2(icol)*8.64e7_r8,  &
   248)            qtmx*8.64e7_r8, qtmn*8.64e7_r8, &
   249)            (state%q(icol,k,1)+ptend%q(icol,k,1)*ztodt)*1.e3_r8, &
   250)            (state%q(icol,k,ixcldliq)+ptend%q(icol,k,ixcldliq)*ztodt)*1.e3_r8
