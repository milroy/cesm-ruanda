   103)   subroutine set_interp_parameter(parm_name, value)
   104)     character*(*), intent(in) :: parm_name
   105)     character(len=80) :: msg
   106)     integer :: value,power
   107)     real (kind=real_kind) :: value_target
   108) 
   109)     if(parm_name .eq. 'itype') then
   110)        itype=value
   111)     else if(parm_name .eq. 'nlon') then
   112)        nlon=value
   113)     else if(parm_name .eq. 'nlat') then
   114)        nlat=value
   115)     else if(parm_name.eq. 'gridtype') then
   116)        gridtype=value
   117)     else if(parm_name.eq. 'auto') then
   118)        auto_grid=1
   119)        ! compute recommended nlat,nlon which has slightly higher
   120)        ! resolution than the specifed number of points around equator given in "value"
   121)        ! computed recommended lat-lon grid.
   122)        ! nlon > peq   peq = points around equator cubed sphere grid
   123)        ! take nlon power of 2, and at most 1 power of 3
   124)        if (value.eq.0) then
   125)            ! If reading in unstructured mesh, ne = 0
   126)            ! This makes it hard to guess how many interpolation points to use
   127)            ! So We'll set the default as 720 x 360
   128)            ! BUT if you're running with an unstructured mesh, set interp_nlon and interp_nlat
   129)            nlon = 1536
   130)            nlat = 768
   131)        else
   132)            value_target=value*1.25
   133)            power = nint(.5 +  log( value_target)/log(2d0) )
   134)            power = max(power,7) ! min grid: 64x128
   135)            if ( 3*2**(power-2) > value_target) then
   136)                nlon=3*2**(power-2)   ! use 1 power of 3
   137)            else
   138)                nlon=2**power
   139)            endif
   140)        endif
   141)        nlat=nlon/2
   142)        if (gridtype==1) nlat=nlat+1
   143)     else
   144)        write(msg,*) 'Did not recognize parameter named ',parm_name,' in interpolate_mod:set_interp_parameter'
   145)        call abortmp(msg)

   228)   subroutine interpolate_tracers(r, tracers, f) 
   229)     use kinds,          only : longdouble_kind
   230)     use dimensions_mod, only : np, qsize
   231)     use quadrature_mod, only : quadrature_t, gausslobatto
   232) 
   233) 
   234)     implicit none
   235) 
   236)     type (cartesian2D_t), intent(in)  :: r
   237)     real (kind=real_kind),intent(in)  :: tracers(np*np,qsize)
   238)     real (kind=real_kind),intent(out) :: f(qsize)
   239) 
   240)     type (quadrature_t        )       :: gll        
   241)     real (kind=real_kind      )       :: dp    (np)
   242)     real (kind=real_kind      )       :: x     (np)
   243)     real (kind=real_kind      )       :: y     (np)
   244)     real (kind=real_kind      )       :: c     (np,np)
   245)     real (kind=real_kind      )       :: xy    (np*np)
   246) 
   247)     integer                           :: i,j
   248)     logical                           :: first_time=.true.
   249)     
   250)     save c
   251)     save gll
   252)    
   253)     if (first_time) then
   254)       first_time = .false.
   255)       gll=gausslobatto(np)
   256)       dp = 1
   257)       do i=1,np
   258)         do j=1,np
   259)           if (i /= j) then
   260)             dp(i) = dp(i) * (gll%points(i) - gll%points(j))
   261)           end if
   262)         end do
   263)       end do 
   264)       do i=1,np
   265)         do j=1,np
   266)           c(i,j) = 1/(dp(i)*dp(j))
   267)         end do
   268)       end do 
   269)     end if
   270) 
   271)     x = 1
   272)     y = 1
   273)     do i=1,np
   274)       do j=1,np
   275)         if (i /= j) then
   276)           x(i) = x(i) * (r%x - gll%points(j))
   277)           y(i) = y(i) * (r%y - gll%points(j))
   278)         end if
   279)       end do
   280)     end do 
   281)     do j=1,np  
   282)       do i=1,np
   283)         xy(i + (j-1)*np) = x(i)*y(j)*c(i,j)
   284)       end do
   285)     end do 
   286)     f = MATMUL(xy,tracers)
   287)   end subroutine interpolate_tracers

   324)   function interpolate_2d(cart, f, interp, npts, fillvalue) result(fxy)
   325)     integer, intent(in)               :: npts
   326)     type (cartesian2D_t), intent(in)  :: cart
   327)     real (kind=real_kind), intent(in) :: f(npts,npts)
   328)     type (interpolate_t)              :: interp
   329)     real (kind=real_kind)             :: fxy     ! value of f interpolated to (x,y)
   330)     real (kind=real_kind), intent(in), optional :: fillvalue
   331)     ! local variables
   332) 
   333)     real (kind=real_kind)             :: tmp_1,tmp_2
   334)     real (kind=real_kind)             :: fk0,fk1
   335)     real (kind=real_kind)             :: pk
   336) 
   337)     integer                           :: l,j,k
   338) 
   339)     if(present(fillvalue)) then
   340)        if (any(f==fillvalue)) then
   341)           fxy = fillvalue
   342)           return
   343)        endif
   344)     endif
   345) 
   346) 
   347)     do l=1,npts,2
   348) 
   349)        ! Compute Pk(cart%x) for Legendre order 0
   350) 
   351)        pk = 1.0D0
   352) 
   353)        fk0=0.0D0
   354)        fk1=0.0D0
   355)        do j=1,npts
   356)           fk0 = fk0 + interp%Imat(j,1)*f(j,l  )
   357)           fk1 = fk1 + interp%Imat(j,1)*f(j,l+1)
   358)        end do
   359)        interp%vtemp(l  ) = pk*fk0
   360)        interp%vtemp(l+1) = pk*fk1
   361) 
   362)        ! Compute Pk(cart%x) for Legendre order 1
   363) 
   364)        tmp_2 = pk
   365)        pk    = cart%x
   366) 
   367)        fk0=0.0D0
   368)        fk1=0.0D0
   369)        do j=1,npts
   370)           fk0 = fk0 + interp%Imat(j,2)*f(j,l  )
   371)           fk1 = fk1 + interp%Imat(j,2)*f(j,l+1)
   372)        end do
   373)        interp%vtemp(l  ) = interp%vtemp(l  ) + pk*fk0
   374)        interp%vtemp(l+1) = interp%vtemp(l+1) + pk*fk1
   375) 
   376)        ! Compute Pk(cart%x) for Legendre order 2 to npts-1
   377) 
   378)        do k = 2,npts-1
   379) 
   380)           tmp_1  = tmp_2
   381)           tmp_2  = pk
   382)           pk = ( (2*k-1)*cart%x*tmp_2 - (k-1)*tmp_1 )*interp%rk(k)
   383) 
   384)           fk0=0.0D0
   385)           fk1=0.0D0
   386)           do j=1,npts
   387)              fk0 = fk0 + interp%Imat(j,k+1)*f(j,l  )
   388)              fk1 = fk1 + interp%Imat(j,k+1)*f(j,l+1)
   389)           end do
   390)           interp%vtemp(l  ) = interp%vtemp(l  ) + pk*fk0
   391)           interp%vtemp(l+1) = interp%vtemp(l+1) + pk*fk1
   392) 
   393)        end do
   394) 
   395)     end do
   396) 
   397)     ! Compute Pk(cart%y) for Legendre order 0
   398) 
   399)     pk = 1.0
   400) 
   401)     fk0 = 0.0D0
   402)     do j=1,npts
   403)        fk0 = fk0 + interp%Imat(j,1)*interp%vtemp(j)
   404)     end do
   405)     fxy = pk*fk0
   406) 
   407)     ! Compute Pk(cart%y) for Legendre order 1
   408) 
   409)     tmp_2 = pk
   410)     pk    = cart%y
   411) 
   412)     fk0=0.0D0
   413)     do j=1,npts
   414)        fk0 = fk0 + interp%Imat(j,2)*interp%vtemp(j)
   415)     end do
   416)     fxy = fxy + pk*fk0
   417) 
   418)     ! Compute Pk(cart%y) for Legendre order 2, npts-1
   419) 
   420)     do k = 2,npts-1
   421)        tmp_1  = tmp_2
   422)        tmp_2  = pk
   423)        pk = ( (2*k-1)*cart%y*tmp_2 - (k-1)*tmp_1 )*interp%rk(k)
   424) 
   425)        fk0 = 0.0D0
   426)        do j=1,npts
   427)           fk0 = fk0 + interp%Imat(j,k+1)*interp%vtemp(j)
   428)        end do
   429) 
   430)        fxy = fxy + pk*fk0
   431) 
   432)     end do
   433) 
   434)   end function interpolate_2d

   440)   function interpol_bilinear(cart, f, interp, npts, fillvalue) result(fxy)
   441) 
   442)     integer, intent(in)               :: npts
   443)     type (cartesian2D_t), intent(in)  :: cart
   444)     real (kind=real_kind), intent(in) :: f(npts,npts)
   445)     type (interpolate_t)              :: interp
   446)     real (kind=real_kind)             :: fxy     ! value of f interpolated to (x,y)
   447)     real (kind=real_kind), intent(in), optional :: fillvalue
   448)     ! local variables
   449) 
   450)     real (kind=real_kind)             :: xoy(npts)
   451)     real (kind=real_kind)             :: p,q,xp,yp ,y4(4)
   452) 
   453)     integer                           :: l,j,k, ii, jj, na,nb,nm
   454) 
   455)     xp = cart%x
   456)     yp = cart%y
   457) 
   458)     xoy(:) = interp%glp(:)
   459) 
   460)     ! Search index along "x"  (bisection method)
   461) 
   462)     na = 1
   463)     nb = npts
   464)     do
   465)        if  ((nb-na) <=  1)  exit
   466)        nm = (nb + na)/2
   467)        if (xp  >  xoy(nm)) then
   468)           na = nm
   469)        else
   470)           nb = nm
   471)        endif
   472)     enddo
   473)     ii = na
   474) 
   475)     ! Search index along "y"
   476) 
   477)     na = 1
   478)     nb = npts
   479)     do
   480)        if  ((nb-na) <=  1)  exit
   481)        nm = (nb + na)/2
   482)        if (yp  >  xoy(nm)) then
   483)           na = nm
   484)        else
   485)           nb = nm
   486)        endif
   487)     enddo
   488)     jj = na
   489) 
   490)     ! GLL cell containing (xp,yp)
   491) 
   492)     y4(1) = f(ii,jj)
   493)     y4(2) = f(ii+1,jj)
   494)     y4(3) = f(ii+1,jj+1)
   495)     y4(4) = f(ii,jj+1)
   496) 
   497)     if(present(fillvalue)) then
   498)        if (any(y4==fillvalue)) then
   499)           fxy = fillvalue
   500)           return
   501)        endif
   502)     endif
   503)        p = (xp - xoy(ii))/(xoy(ii+1) - xoy(ii))
   504)        q = (yp - xoy(jj))/(xoy(jj+1) - xoy(jj))
   505) 
   506)        fxy = (1.0D0 - p)*(1.0D0 - q)* y4(1) + p*(1.0D0 - q) * y4(2)   &
   507)             + p*q* y4(3) + (1.0D0 - p)*q * y4(4)
   508) 
   509)   end function interpol_bilinear

   697)   function parametric_coordinates(sphere, elem) result (ref)
   698)     implicit none
   699)     type (spherical_polar_t), intent(in) :: sphere
   700)     type (element_t), intent(in) :: elem
   701)     type (cartesian2D_t) :: ref
   702) 
   703)     ! local
   704)     integer               :: face_no, i, MAX_NR_ITER=10
   705)     real(kind=real_kind)  :: D(2,2),Dinv(2,2),detD,a,b,resa,resb,dela,delb,costh
   706)     real(kind=real_kind)  :: tol_sq=1e-26
   707)     type (spherical_polar_t) :: sphere1, sphere_tmp
   708) 
   709)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   710)     ! newton iteration on: ref=ref - df^-1 (ref2sphere(ref) - sphere)
   711)     !
   712)     ! Generic version written in terms of HOMME's 'ref2sphere' and 'Dmap' operaters,
   713)     ! with no assumption as to the type of map (gnomonic, equi-angular, parametric)
   714)     !
   715)     ! f = ref2sphere(xvec) - sphere
   716)     ! df = d(ref2sphere) 
   717)     !
   718)     ! D = diag(cos(theta),1) * d(ref2sphere)       d(ref2sphere) = diag(1/cos(theta),1)*D
   719)     ! 
   720)     ! df = diag(1/cos(theta),1)*D 
   721)     ! df^-1 =  D^-1 *  diag(cos(theta),1)
   722)     ! 
   723)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   724)     costh=cos(sphere%lat)
   725)     a=0
   726)     b=0
   727)     i=0
   728)     do
   729)        sphere1 = ref2sphere(a,b,elem)
   730)        resa = sphere1%lon - sphere%lon
   731)        if (resa>dd_pi) resa=resa-2*dd_pi
   732)        if (resa<-dd_pi) resa=resa+2*dd_pi
   733) 
   734)        resb = sphere1%lat - sphere%lat 
   735) 
   736)        call Dmap(D,elem,a,b)
   737)        detD = D(1,1)*D(2,2) - D(1,2)*D(2,1)      
   738)        Dinv(1,1) =  D(2,2)/detD
   739)        Dinv(1,2) = -D(1,2)/detD
   740)        Dinv(2,1) = -D(2,1)/detD
   741)        Dinv(2,2) =  D(1,1)/detD
   742)        
   743)        dela =  Dinv(1,1)*costh*resa + Dinv(1,2)*resb 
   744)        delb =  Dinv(2,1)*costh*resa + Dinv(2,2)*resb 
   745)        a = a - dela
   746)        b = b - delb
   747)        i=i+1
   748)        if ( (costh*resa)**2 + resb**2 < tol_sq .or. MAX_NR_ITER < i) exit
   749)     end do
   750)     ref%x=a
   751)     ref%y=b
   752)   end function parametric_coordinates

  1069)   subroutine setup_latlon_interp(elem,interpdata,par)
  1070)     !
  1071)     ! initialize interpolation data structures to interpolate to a lat-lon grid
  1072)     !
  1073)     !
  1074) 
  1075)     implicit none
  1076)     type (element_t)     , intent(in), target :: elem(:)
  1077)     type (parallel_t)      , intent(in)       :: par
  1078)     type (interpdata_t)  , intent(out)        :: interpdata(:)
  1079) 
  1080)     ! local
  1081)     integer i,j,ii,count_total,n_interp,count_max
  1082)     integer ngrid, number, elem_num, plat
  1083)     integer countx, missing_pts,ierr
  1084)     integer :: npts_mult_claims,max_claims
  1085) 
  1086)     real (kind=real_kind)    ::  dp,latdeg(nlat+1),clat(nlat+1),w(nlat+1),w_staggered(nlat)
  1087)     real (kind=real_kind)    ::  clat_staggered(nlat),latdeg_st(nlat),err,err2
  1088) 
  1089)     type (spherical_polar_t) :: sphere
  1090)     type (cartesian2D_t)     :: cart
  1091)     type (cartesian3D_t)     :: sphere_xyz,sphere2_xyz
  1092) 
  1093)     type (quadrature_t)       :: gp
  1094) 
  1095) 
  1096)     logical,save                                  :: first_time=.TRUE.
  1097) 
  1098)     ! Array to make sure each interp point is on exactly one process
  1099)     type (cartesian2D_t),allocatable    :: cart_vec(:,:)
  1100)     integer :: k
  1101)     integer, allocatable :: global_elem_gid(:,:),local_elem_gid(:,:), local_elem_num(:,:)
  1102) 
  1103)     ! these arrays often are too large for stack, so lets make sure
  1104)     ! they go on the heap:
  1105)     allocate(local_elem_num(nlat,nlon))
  1106)     allocate(local_elem_gid(nlat,nlon))
  1107)     allocate(global_elem_gid(nlat,nlon))
  1108)     allocate(cart_vec(nlat,nlon))
  1109) 
  1110)     if (par%masterproc) then
  1111)        write(iulog,'(a,i4,a,i4,a)') 'Initializing ',nlat,' x ',nlon,' lat-lon interpolation grid: '
  1112)     endif
  1113) 
  1114)     do ii=1,nelemd
  1115)        interpdata(ii)%n_interp=0  ! reset counter
  1116)     enddo
  1117) 
  1118)     if(first_time)then
  1119)        NULLIFY(lat)
  1120)        NULLIFY(gweight)
  1121)        NULLIFY(lon)
  1122)        first_time=.FALSE.
  1123)     end if
  1124) 
  1125)     if (associated(lat))then
  1126)        if(size(lat)>0)deallocate(lat)
  1127)     endif
  1128)     if (associated(gweight))then
  1129)        if(size(gweight)>0)deallocate(gweight)
  1130)     endif
  1131) 
  1132)     if (associated(lon))then
  1133)        if(size(lon)>0)deallocate(lon)
  1134)     endif
  1135) 
  1136)     allocate(lat(nlat))
  1137)     allocate(gweight(nlat))
  1138)     allocate(lon(nlon))
  1139)     call interp_init()
  1140)     gweight=0
  1141)     do i=1,nlon
  1142)        lon(i)=2*dd_pi*(i-1)/nlon
  1143)     enddo
  1144)     if (gridtype==1) then
  1145)        do j=1,nlat
  1146)           lat(j) = -dd_pi/2 + dd_pi*(j-1)/(nlat-1)
  1147)        end do
  1148)        plat=nlat
  1149)     endif
  1150)     if (gridtype==2) then
  1151)        gp=gauss(nlat)
  1152)        do j=1,nlat
  1153)           lat(j) = asin(gp%points(j))
  1154)           gweight(j) = gp%weights(j)
  1155)        end do
  1156)     endif
  1157)     if (gridtype==3) then
  1158)        do j=1,nlat
  1159)           lat(j) = -dd_pi/2 + dd_pi*(j-.5d0)/nlat
  1160)        end do
  1161)        plat=nlat+1
  1162)     endif
  1163) 
  1164)     if (gridtype==1 .or. gridtype==3) then
  1165)        ! gridtype=1    plat=nlat    gweight(1:nlat)=w(1:plat)
  1166)        ! gridtype=3    plat=nlat+1  gweight(1:nlat)=w_staggered(1:plat-1)
  1167) 
  1168)        ! L-R dynamics uses a regular latitude distribution (not gausian).
  1169)        ! The algorithm below is a bastardized version of LSM: map.F.
  1170)        dp = 180d0/(plat-1)
  1171)        do j = 1, plat
  1172)           latdeg(j) = -90d0 + (j-1)*dp
  1173)           clat(j) = latdeg(j)*dd_pi/180d0
  1174)        end do
  1175) 
  1176)        ! Calculate latitudes for the staggered grid
  1177) 
  1178)        do j = 1, plat-1
  1179)           clat_staggered(j) = (clat(j) + clat(j+1)) / 2
  1180)           latdeg_st     (j) = clat_staggered(j)*180d0/dd_pi
  1181)        end do
  1182) 
  1183)        ! Weights are defined as cos(phi)*(delta-phi)
  1184)        ! For a sanity check, the sum of w across all lats should be 2, or 1 across
  1185)        ! half of the latitudes.
  1186) 
  1187)        do j = 2, plat-1
  1188)           w(j) = sin(clat_staggered(j)) - sin(clat_staggered(j-1))
  1189)        end do
  1190)        w(1) = sin(clat_staggered(1)) + 1
  1191)        w(plat) = w(1)
  1192) 
  1193)        ! with nlat=2048, this error was 4e-16
  1194)        if (abs(sum(w(1:plat)) - 2) > 1e-8) then
  1195)           write(iulog,*) 'interpolate_mod: w weights do not sum to 2. sum=',sum(w(1:plat))
  1196)           call abortmp('interpolate_mod: weights do not sum to 2.')
  1197)        end if
  1198) 
  1199)        dp = dd_pi / (plat-1)
  1200)        do j = 1, plat-1
  1201)           w_staggered(j) = sin(clat(j+1)) - sin(clat(j))
  1202)        end do
  1203) 
  1204) 
  1205)        if (abs(sum(w_staggered(1:plat-1)) - 2) > 1e-8) then
  1206)           write(iulog,*) 'interpolate_mod: staggered weights do not sum to 2. sum=',sum(w_staggered(1:plat-1))
  1207)           call abortmp('interpolate_mod: weights do not sum to 2.')
  1208)        end if
  1209) 
  1210)        if (gridtype==1) then
  1211)           gweight(1:nlat)=w(1:plat)
  1212)        endif
  1213)        if (gridtype==3) then
  1214)           gweight(1:nlat)=w_staggered(1:plat-1)
  1215)        endif
  1216)     endif
  1217) 
  1218) 
  1219)     ! go through once, counting the number of points on each element
  1220)     sphere%r=1
  1221)     local_elem_num  = -1
  1222)     local_elem_gid  = -1
  1223)     global_elem_gid = -1
  1224)     err=0
  1225)     do j=1,nlat
  1226)        do i=1,nlon
  1227)           sphere%lat=lat(j)
  1228)           sphere%lon=lon(i)
  1229) 
  1230)           !debug=.false.
  1231)           !if (j==18 .and. i==95) debug=.true.
  1232)           number = -1
  1233)           if ( (cubed_sphere_map /= 0) .or. MeshUseMeshFile) then
  1234)              call cube_facepoint_unstructured(sphere, cart, number, elem)
  1235)              if (number /= -1) then
  1236)                 ! If points are outside element but within tolerance, move to boundary
  1237)                 if (cart%x + 1.0d0.le.0.0d0) cart%x = -1.0d0
  1238)                 if (cart%x - 1.0d0.ge.0.0d0) cart%x = 1.0d0
  1239)                 if (cart%y + 1.0d0.le.0.0d0) cart%y = -1.0d0
  1240)                 if (cart%y - 1.0d0.ge.0.0d0) cart%y = 1.0d0
  1241) 
  1242)                 local_elem_num(j,i) = number
  1243)                 local_elem_gid(j,i) = elem(number)%vertex%number
  1244)                 cart_vec(j,i)    = cart  ! local element coordiante of interpolation point
  1245)              endif
  1246)           else
  1247)              call cube_facepoint_ne(sphere, ne, cart, number)
  1248)              ! the sphere point belongs to the element number on face = face_no.
  1249)              ! do I own this element?
  1250)              if (number /= -1) then
  1251)                 do ii=1,nelemd
  1252)                    if (number == elem(ii)%vertex%number) then
  1253)                       local_elem_gid(j,i) = number
  1254)                       local_elem_num(j,i) = ii
  1255)                       cart_vec(j,i)        = cart   ! local element coordinate found above
  1256)                       exit
  1257)                    endif
  1258)                 enddo
  1259)              endif
  1260)           endif
  1261)           ii=local_elem_num(j,i)
  1262)           if (ii /= -1) then
  1263)              ! compute error: map 'cart' back to sphere and compare with original
  1264)              ! interpolation point:
  1265)              sphere2_xyz = spherical_to_cart( ref2sphere(cart%x,cart%y,elem(ii)))
  1266)              sphere_xyz = spherical_to_cart(sphere)
  1267)              err=max(err,distance(sphere2_xyz,sphere_xyz))
  1268)           endif
  1269)        enddo
  1270)        if (par%masterproc) then
  1271)           if ((MOD(j,64).eq.1).or.(j.eq.nlat)) then
  1272)              print *,'finished latitude ',j,' of ',nlat
  1273)           endif
  1274)        endif
  1275)     enddo
  1276)     err2=err
  1277) #ifdef _MPI
  1278)     call MPI_Allreduce(err,err2,1,MPIreal_t,MPI_MAX,par%comm,ierr)
  1279) #endif
  1280)     if (par%masterproc) then
  1281)        write(iulog,'(a,e12.4)') 'Max interpolation point search error: ',err2
  1282)     endif
  1283) 
  1284)     ! if multile elements claim a interpolation point, take the one with largest gid:
  1285)     global_elem_gid = local_elem_gid
  1286) #ifdef _MPI
  1287)     call MPI_Allreduce(local_elem_gid, global_elem_gid, nlat*nlon, MPIinteger_t, MPI_MAX, par%comm,ierr)
  1288) #endif
  1289) 
  1290)     missing_pts=0
  1291)     do j=1,nlat
  1292)        do i=1,nlon
  1293)           if (global_elem_gid(j,i) == -1 ) then
  1294)              missing_pts = missing_pts + 1
  1295)              if (par%masterproc) &
  1296)                   print *,'Error: point not claimed by any element j,i,lat(j),lon(i)=',j,i,lat(j),lon(i)
  1297)           else if (local_elem_gid(j,i) == global_elem_gid(j,i)  ) then
  1298)              ii = local_elem_num(j,i)
  1299)              interpdata(ii)%n_interp = interpdata(ii)%n_interp + 1
  1300)           endif
  1301)        end do
  1302)     end do
  1303) 
  1304)     countx=maxval(interpdata(1:nelemd)%n_interp)
  1305)     count_max = countx
  1306) #ifdef _MPI
  1307)     call MPI_Allreduce(countx,count_max,1,MPIinteger_t,MPI_MAX,par%comm,ierr)
  1308) #endif
  1309) 
  1310)     if (par%masterproc) then
  1311)        write(iulog,'(a,i6)') 'Maximum number of interpolation points claimed by an element: ',count_max
  1312)     endif
  1313) 
  1314)     ! allocate storage
  1315)     do ii=1,nelemd
  1316)        ngrid = interpdata(ii)%n_interp
  1317)        if(interpdata(ii)%first_entry)then
  1318)           NULLIFY(interpdata(ii)%interp_xy)
  1319)           NULLIFY(interpdata(ii)%ilat)
  1320)           NULLIFY(interpdata(ii)%ilon)
  1321) 
  1322)           interpdata(ii)%first_entry=.FALSE.
  1323)        endif
  1324)        if(associated(interpdata(ii)%interp_xy))then
  1325)           if(size(interpdata(ii)%interp_xy)>0)deallocate(interpdata(ii)%interp_xy)
  1326)        endif
  1327)        if(associated(interpdata(ii)%ilat))then
  1328)           if(size(interpdata(ii)%ilat)>0)deallocate(interpdata(ii)%ilat)
  1329)        endif
  1330) 
  1331)        if (associated(interpdata(ii)%ilon))then
  1332)           if(size(interpdata(ii)%ilon)>0)deallocate(interpdata(ii)%ilon)
  1333)        endif
  1334)        allocate(interpdata(ii)%interp_xy( ngrid ) )
  1335)        allocate(interpdata(ii)%ilat( ngrid ) )
  1336)        allocate(interpdata(ii)%ilon( ngrid ) )
  1337)        interpdata(ii)%n_interp=0  ! reset counter
  1338)     enddo
  1339)     do j=1,nlat
  1340)        do i=1,nlon
  1341)           if (local_elem_gid(j,i) == global_elem_gid(j,i) .and. &
  1342)                local_elem_gid(j,i) /= -1 ) then
  1343)              ii = local_elem_num(j,i)
  1344)              ngrid = interpdata(ii)%n_interp + 1
  1345)              interpdata(ii)%n_interp = ngrid
  1346)              interpdata(ii)%interp_xy( ngrid )   = cart_vec(j,i)
  1347)              interpdata(ii)%ilon( ngrid ) = i
  1348)              interpdata(ii)%ilat( ngrid ) = j
  1349)           endif
  1350)        enddo
  1351)     enddo
  1352) 
  1353)     ! now lets compute the number of points that were claimed by
  1354)     ! more than one element:
  1355)     do j=1,nlat
  1356)        do i=1,nlon
  1357)           if (local_elem_gid(j,i) == -1) then
  1358)              local_elem_gid(j,i)=0
  1359)           else
  1360)              local_elem_gid(j,i)=1
  1361)           endif
  1362)        enddo
  1363)     enddo
  1364)     global_elem_gid = local_elem_gid
  1365) #ifdef _MPI
  1366)     call MPI_Allreduce(local_elem_gid, global_elem_gid, nlat*nlon, MPIinteger_t, MPI_SUM, par%comm,ierr)

  1370)        do j=1,nlat
  1371)           do i=1,nlon
  1372)              if (global_elem_gid(j,i)>1) countx=countx+1

  1374)        enddo
  1375)        npts_mult_claims=countx
  1376)        max_claims=maxval(global_elem_gid)
  1377)     endif
  1378) 
  1379)     if (par%masterproc) then
  1380)        print *,'Number of interpolation points claimed by more than one element: ',npts_mult_claims
  1381)        print *,'max number of elements which claimed the same interpolation point:',max_claims
  1382)     endif
  1383)     
  1384)     deallocate(global_elem_gid)
  1385)     deallocate(local_elem_num)
  1386)     deallocate(local_elem_gid)

  1394)        end if
