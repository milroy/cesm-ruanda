    77) subroutine zmconv_readnl(nlfile)
    78) 
    79)    use namelist_utils,  only: find_group_name
    80)    use units,           only: getunit, freeunit
    81)    use mpishorthand
    82) 
    83)    character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
    84) 
    85)    ! Local variables
    86)    integer :: unitn, ierr
    87)    character(len=*), parameter :: subname = 'zmconv_readnl'
    88) 
    89)    namelist /zmconv_nl/ zmconv_c0_lnd, zmconv_c0_ocn, zmconv_ke
    90)    !-----------------------------------------------------------------------------
    91) 
    92)    if (masterproc) then
    93)       unitn = getunit()
    94)       open( unitn, file=trim(nlfile), status='old' )
    95)       call find_group_name(unitn, 'zmconv_nl', status=ierr)
    96)       if (ierr == 0) then
    97)          read(unitn, zmconv_nl, iostat=ierr)
    98)          if (ierr /= 0) then
    99)             call endrun(subname // ':: ERROR reading namelist')
   100)          end if
   101)       end if
   102)       close(unitn)
   103)       call freeunit(unitn)
   104) 
   105)       ! set local variables
   106)       c0_lnd = zmconv_c0_lnd
   107)       c0_ocn = zmconv_c0_ocn
   108)       ke = zmconv_ke
   109) 
   110)    end if
   111) 
   112) #ifdef SPMD
   113)    ! Broadcast namelist variables
   114)    call mpibcast(c0_lnd,            1, mpir8,  0, mpicom)
   115)    call mpibcast(c0_ocn,            1, mpir8,  0, mpicom)
   116)    call mpibcast(ke,                1, mpir8,  0, mpicom)
   117) #endif
   118) 
   119) end subroutine zmconv_readnl

   122) subroutine zm_convi(limcnv_in, no_deep_pbl_in)
   123) 
   124)    use dycore,       only: dycore_is, get_resolution
   125) 
   126)    integer, intent(in)           :: limcnv_in       ! top interface level limit for convection
   127)    logical, intent(in), optional :: no_deep_pbl_in  ! no_deep_pbl = .true. eliminates ZM convection entirely within PBL 
   128) 
   129)    ! local variables
   130)    character(len=32)   :: hgrid           ! horizontal grid specifier
   131) 
   132)    ! Initialization of ZM constants
   133)    limcnv = limcnv_in
   134)    tfreez = tmelt
   135)    eps1   = epsilo
   136)    rl     = latvap
   137)    cpres  = cpair
   138)    rgrav  = 1.0_r8/gravit
   139)    rgas   = rair
   140)    grav   = gravit
   141)    cp     = cpres
   142) 
   143)    if ( present(no_deep_pbl_in) )  then
   144)       no_deep_pbl = no_deep_pbl_in
   145)    else
   146)       no_deep_pbl = .false.
   147)    endif
   148) 
   149)    ! tau=4800. were used in canadian climate center. however, in echam3 t42, 
   150)    ! convection is too weak, thus adjusted to 2400.
   151) 
   152)    hgrid = get_resolution()
   153)    tau = 3600._r8
   154) 
   155)    if ( masterproc ) then
   156)       write(iulog,*) 'tuning parameters zm_convi: tau',tau
   157)       write(iulog,*) 'tuning parameters zm_convi: c0_lnd',c0_lnd, ', c0_ocn', c0_ocn 
   158)       write(iulog,*) 'tuning parameters zm_convi: ke',ke
   159)       write(iulog,*) 'tuning parameters zm_convi: no_deep_pbl',no_deep_pbl
   160)    endif
   161) 
   162)    if (masterproc) write(iulog,*)'**** ZM: DILUTE Buoyancy Calculation ****'
   163) 
   164) end subroutine zm_convi

   445)    mcon(:,:) = 0._r8
   446)    rliq(:ncol)   = 0._r8
   447) !
   448) ! initialize convective tendencies
   449) !
   450)    prec(:ncol) = 0._r8
   451)    do k = 1,pver
   452)       do i = 1,ncol
   453)          dqdt(i,k)  = 0._r8
   454)          dsdt(i,k)  = 0._r8
   455)          dudt(i,k)  = 0._r8
   456)          dvdt(i,k)  = 0._r8
   457)          pflx(i,k)  = 0._r8
   458)          pflxg(i,k) = 0._r8
   459)          cme(i,k)   = 0._r8
   460)          rprd(i,k)  = 0._r8
   461)          zdu(i,k)   = 0._r8
   462)          ql(i,k)    = 0._r8
   463)          qlg(i,k)   = 0._r8
   464)          dlf(i,k)   = 0._r8
   465)          dlg(i,k)   = 0._r8
   466)       end do
   467)    end do
   468)    do i = 1,ncol
   469)       pflx(i,pverp) = 0
   470)       pflxg(i,pverp) = 0
   471)    end do
   472) !
   473)    do i = 1,ncol
   474)       pblt(i) = pver
   475)       dsubcld(i) = 0._r8
   476) 
   477)       jctop(i) = pver
   478)       jcbot(i) = 1
   479) 
   480)    end do
   481) !
   482) ! calculate local pressure (mbs) and height (m) for both interface
   483) ! and mid-layer locations.
   484) !
   485)    do i = 1,ncol
   486)       zs(i) = geos(i)*rgrav
   487)       pf(i,pver+1) = paph(i,pver+1)*0.01_r8
   488)       zf(i,pver+1) = zi(i,pver+1) + zs(i)
   489)    end do
   490)    do k = 1,pver
   491)       do i = 1,ncol
   492)          p(i,k) = pap(i,k)*0.01_r8
   493)          pf(i,k) = paph(i,k)*0.01_r8
   494)          z(i,k) = zm(i,k) + zs(i)
   495)          zf(i,k) = zi(i,k) + zs(i)
   496)       end do
   497)    end do
   498) !
   499)    do k = pver - 1,msg + 1,-1
   500)       do i = 1,ncol
   501)          if (abs(z(i,k)-zs(i)-pblh(i)) < (zf(i,k)-zf(i,k+1))*0.5_r8) pblt(i) = k
   502)       end do
   503)    end do
   504) !
   505) ! store incoming specific humidity field for subsequent calculation
   506) ! of precipitation (through change in storage).
   507) ! define dry static energy (normalized by cp).
   508) !
   509)    do k = 1,pver
   510)       do i = 1,ncol
   511)          q(i,k) = qh(i,k)
   512)          s(i,k) = t(i,k) + (grav/cpres)*z(i,k)
   513)          tp(i,k)=0.0_r8
   514)          shat(i,k) = s(i,k)
   515)          qhat(i,k) = q(i,k)
   516)       end do
   517)    end do
   518) 
   519)    do i = 1,ncol
   520)       capeg(i) = 0._r8
   521)       lclg(i) = 1
   522)       lelg(i) = pver
   523)       maxg(i) = 1
   524)       tlg(i) = 400._r8
   525)       dsubcld(i) = 0._r8
   526)    end do
   527) 
   528)    if( cam_physpkg_is('cam3')) then
   529) 
   530)       !  For cam3 physics package, call non-dilute
   531) 
   532)       call buoyan(lchnk   ,ncol    , &
   533)                   q       ,t       ,p       ,z       ,pf       , &
   534)                   tp      ,qstp    ,tl      ,rl      ,cape     , &
   535)                   pblt    ,lcl     ,lel     ,lon     ,maxi     , &
   536)                   rgas    ,grav    ,cpres   ,msg     , &
   537)                   tpert   )
   538)    else
   539) 
   540)       !  Evaluate Tparcel, qs(Tparcel), buoyancy and CAPE, 
   541)       !     lcl, lel, parcel launch level at index maxi()=hmax
   542) 
   543)       call buoyan_dilute(lchnk   ,ncol    , &
   544)                   q       ,t       ,p       ,z       ,pf       , &
   545)                   tp      ,qstp    ,tl      ,rl      ,cape     , &
   546)                   pblt    ,lcl     ,lel     ,lon     ,maxi     , &
   547)                   rgas    ,grav    ,cpres   ,msg     , &
   548)                   tpert   )
   549)    end if
   550) 
   551) !
   552) ! determine whether grid points will undergo some deep convection
   553) ! (ideep=1) or not (ideep=0), based on values of cape,lcl,lel
   554) ! (require cape.gt. 0 and lel<lcl as minimum conditions).
   555) !
   556)    lengath = 0
   557)    do i=1,ncol
   558)       if (cape(i) > capelmt) then
   559)          lengath = lengath + 1
   560)          index(lengath) = i
   561)       end if
   562)    end do
   563) 
   564)    if (lengath.eq.0) return
   565)    do ii=1,lengath
   566)       i=index(ii)
   567)       ideep(ii)=i
   568)    end do
   569) !
   570) ! obtain gathered arrays necessary for ensuing calculations.
   571) !
   572)    do k = 1,pver
   573)       do i = 1,lengath
   574)          dp(i,k) = 0.01_r8*dpp(ideep(i),k)
   575)          qg(i,k) = q(ideep(i),k)
   576)          tg(i,k) = t(ideep(i),k)
   577)          pg(i,k) = p(ideep(i),k)
   578)          zg(i,k) = z(ideep(i),k)
   579)          sg(i,k) = s(ideep(i),k)
   580)          tpg(i,k) = tp(ideep(i),k)
   581)          zfg(i,k) = zf(ideep(i),k)
   582)          qstpg(i,k) = qstp(ideep(i),k)
   583)          ug(i,k) = 0._r8
   584)          vg(i,k) = 0._r8
   585)       end do
   586)    end do
   587) !
   588)    do i = 1,lengath
   589)       zfg(i,pver+1) = zf(ideep(i),pver+1)
   590)    end do
   591)    do i = 1,lengath
   592)       capeg(i) = cape(ideep(i))
   593)       lclg(i) = lcl(ideep(i))
   594)       lelg(i) = lel(ideep(i))
   595)       maxg(i) = maxi(ideep(i))
   596)       tlg(i) = tl(ideep(i))
   597)       landfracg(i) = landfrac(ideep(i))
   598)    end do
   599) !
   600) ! calculate sub-cloud layer pressure "thickness" for use in
   601) ! closure and tendency routines.
   602) !
   603)    do k = msg + 1,pver
   604)       do i = 1,lengath
   605)          if (k >= maxg(i)) then
   606)             dsubcld(i) = dsubcld(i) + dp(i,k)
   607)          end if
   608)       end do
   609)    end do
   610) !
   611) ! define array of factors (alpha) which defines interfacial
   612) ! values, as well as interfacial values for (q,s) used in
   613) ! subsequent routines.
   614) !
   615)    do k = msg + 2,pver
   616)       do i = 1,lengath
   617) !            alpha(i,k) = 0.5
   618)          sdifr = 0._r8
   619)          qdifr = 0._r8
   620)          if (sg(i,k) > 0._r8 .or. sg(i,k-1) > 0._r8) &
   621)             sdifr = abs((sg(i,k)-sg(i,k-1))/max(sg(i,k-1),sg(i,k)))
   622)          if (qg(i,k) > 0._r8 .or. qg(i,k-1) > 0._r8) &
   623)             qdifr = abs((qg(i,k)-qg(i,k-1))/max(qg(i,k-1),qg(i,k)))
   624)          if (sdifr > 1.E-6_r8) then
   625)             shat(i,k) = log(sg(i,k-1)/sg(i,k))*sg(i,k-1)*sg(i,k)/(sg(i,k-1)-sg(i,k))
   626)          else
   627)             shat(i,k) = 0.5_r8* (sg(i,k)+sg(i,k-1))
   628)          end if
   629)          if (qdifr > 1.E-6_r8) then
   630)             qhat(i,k) = log(qg(i,k-1)/qg(i,k))*qg(i,k-1)*qg(i,k)/(qg(i,k-1)-qg(i,k))
   631)          else
   632)             qhat(i,k) = 0.5_r8* (qg(i,k)+qg(i,k-1))
   633)          end if
   634)       end do
   635)    end do
   636) !
   637) ! obtain cloud properties.
   638) !
   639) 
   640)    call cldprp(lchnk   , &
   641)                qg      ,tg      ,ug      ,vg      ,pg      , &
   642)                zg      ,sg      ,mu      ,eu      ,du      , &
   643)                md      ,ed      ,sd      ,qd      ,mc      , &
   644)                qu      ,su      ,zfg     ,qs      ,hmn     , &
   645)                hsat    ,shat    ,qlg     , &
   646)                cmeg    ,maxg    ,lelg    ,jt      ,jlcl    , &
   647)                maxg    ,j0      ,jd      ,rl      ,lengath , &
   648)                rgas    ,grav    ,cpres   ,msg     , &
   649)                pflxg   ,evpg    ,cug     ,rprdg   ,limcnv  ,landfracg)
   650) !
   651) ! convert detrainment from units of "1/m" to "1/mb".
   652) !
   653)    do k = msg + 1,pver
   654)       do i = 1,lengath
   655)          du   (i,k) = du   (i,k)* (zfg(i,k)-zfg(i,k+1))/dp(i,k)
   656)          eu   (i,k) = eu   (i,k)* (zfg(i,k)-zfg(i,k+1))/dp(i,k)
   657)          ed   (i,k) = ed   (i,k)* (zfg(i,k)-zfg(i,k+1))/dp(i,k)
   658)          cug  (i,k) = cug  (i,k)* (zfg(i,k)-zfg(i,k+1))/dp(i,k)
   659)          cmeg (i,k) = cmeg (i,k)* (zfg(i,k)-zfg(i,k+1))/dp(i,k)
   660)          rprdg(i,k) = rprdg(i,k)* (zfg(i,k)-zfg(i,k+1))/dp(i,k)
   661)          evpg (i,k) = evpg (i,k)* (zfg(i,k)-zfg(i,k+1))/dp(i,k)
   662)       end do
   663)    end do
   664) 
   665)    call closure(lchnk   , &
   666)                 qg      ,tg      ,pg      ,zg      ,sg      , &
   667)                 tpg     ,qs      ,qu      ,su      ,mc      , &
   668)                 du      ,mu      ,md      ,qd      ,sd      , &
   669)                 qhat    ,shat    ,dp      ,qstpg   ,zfg     , &
   670)                 qlg     ,dsubcld ,mb      ,capeg   ,tlg     , &
   671)                 lclg    ,lelg    ,jt      ,maxg    ,1       , &
   672)                 lengath ,rgas    ,grav    ,cpres   ,rl      , &
   673)                 msg     ,capelmt    )
   674) !
   675) ! limit cloud base mass flux to theoretical upper bound.
   676) !
   677)    do i=1,lengath
   678)       mumax(i) = 0
   679)    end do
   680)    do k=msg + 2,pver
   681)       do i=1,lengath
   682)         mumax(i) = max(mumax(i), mu(i,k)/dp(i,k))
   683)       end do
   684)    end do
   685) 
   686)    do i=1,lengath
   687)       if (mumax(i) > 0._r8) then
   688)          mb(i) = min(mb(i),0.5_r8/(delt*mumax(i)))
   689)       else
   690)          mb(i) = 0._r8
   691)       endif
   692)    end do
   693)    ! If no_deep_pbl = .true., don't allow convection entirely 
   694)    ! within PBL (suggestion of Bjorn Stevens, 8-2000)
   695) 
   696)    if (no_deep_pbl) then
   697)       do i=1,lengath
   698)          if (zm(ideep(i),jt(i)) < pblh(ideep(i))) mb(i) = 0
   699)       end do
   700)    end if
   701) 
   702) 
   703)    do k=msg+1,pver
   704)       do i=1,lengath
   705)          mu   (i,k)  = mu   (i,k)*mb(i)
   706)          md   (i,k)  = md   (i,k)*mb(i)
   707)          mc   (i,k)  = mc   (i,k)*mb(i)
   708)          du   (i,k)  = du   (i,k)*mb(i)
   709)          eu   (i,k)  = eu   (i,k)*mb(i)
   710)          ed   (i,k)  = ed   (i,k)*mb(i)
   711)          cmeg (i,k)  = cmeg (i,k)*mb(i)
   712)          rprdg(i,k)  = rprdg(i,k)*mb(i)
   713)          cug  (i,k)  = cug  (i,k)*mb(i)
   714)          evpg (i,k)  = evpg (i,k)*mb(i)
   715)          pflxg(i,k+1)= pflxg(i,k+1)*mb(i)*100._r8/grav
   716)       end do
   717)    end do
   718) !
   719) ! compute temperature and moisture changes due to convection.
   720) !
   721)    call q1q2_pjr(lchnk   , &
   722)                  dqdt    ,dsdt    ,qg      ,qs      ,qu      , &
   723)                  su      ,du      ,qhat    ,shat    ,dp      , &
   724)                  mu      ,md      ,sd      ,qd      ,qlg     , &
   725)                  dsubcld ,jt      ,maxg    ,1       ,lengath , &
   726)                  cpres   ,rl      ,msg     ,          &
   727)                  dlg     ,evpg    ,cug     )
   728) !
   729) ! gather back temperature and mixing ratio.
   730) !
   731)    do k = msg + 1,pver
   732) !DIR$ CONCURRENT
   733)       do i = 1,lengath
   734) !
   735) ! q is updated to compute net precip.
   736) !
   737)          q(ideep(i),k) = qh(ideep(i),k) + 2._r8*delt*dqdt(i,k)
   738)          qtnd(ideep(i),k) = dqdt (i,k)
   739)          cme (ideep(i),k) = cmeg (i,k)
   740)          rprd(ideep(i),k) = rprdg(i,k)
   741)          zdu (ideep(i),k) = du   (i,k)
   742)          mcon(ideep(i),k) = mc   (i,k)
   743)          heat(ideep(i),k) = dsdt (i,k)*cpres
   744)          dlf (ideep(i),k) = dlg  (i,k)
   745)          pflx(ideep(i),k) = pflxg(i,k)
   746)          ql  (ideep(i),k) = qlg  (i,k)
   747)       end do
   748)    end do
   749) !
   750) !DIR$ CONCURRENT
   751)    do i = 1,lengath
   752)       jctop(ideep(i)) = jt(i)
   753) !++bee
   754)       jcbot(ideep(i)) = maxg(i)
   755) !--bee
   756)       pflx(ideep(i),pverp) = pflxg(i,pverp)
   757)    end do
   758) 
   759) ! Compute precip by integrating change in water vapor minus detrained cloud water
   760)    do k = pver,msg + 1,-1
   761)       do i = 1,ncol
   762)          prec(i) = prec(i) - dpp(i,k)* (q(i,k)-qh(i,k)) - dpp(i,k)*dlf(i,k)*2*delt
   763)       end do
   764)    end do
   765) 
   766) ! obtain final precipitation rate in m/s.
   767)    do i = 1,ncol
   768)       prec(i) = rgrav*max(prec(i),0._r8)/ (2._r8*delt)/1000._r8
   769)    end do
   770) 
   771) ! Compute reserved liquid (not yet in cldliq) for energy integrals.
   772) ! Treat rliq as flux out bottom, to be added back later.
   773)    do k = 1, pver
   774)       do i = 1, ncol
   775)          rliq(i) = rliq(i) + dlf(i,k)*dpp(i,k)/gravit
   776)       end do
   777)    end do
   778)    rliq(:ncol) = rliq(:ncol) /1000._r8
   779) 
   780)    return
   781) end subroutine zm_convr

   783) !===============================================================================
   784) subroutine zm_conv_evap(ncol,lchnk, &
   785)      t,pmid,pdel,q, &
   786)      tend_s, tend_s_snwprd, tend_s_snwevmlt, tend_q, &
   787)      prdprec, cldfrc, deltat,  &
   788)      prec, snow, ntprprd, ntsnprd, flxprec, flxsnow )
   789) 
   790) !-----------------------------------------------------------------------
   791) ! Compute tendencies due to evaporation of rain from ZM scheme
   792) !--
   793) ! Compute the total precipitation and snow fluxes at the surface.
   794) ! Add in the latent heat of fusion for snow formation and melt, since it not dealt with
   795) ! in the Zhang-MacFarlane parameterization.
   796) ! Evaporate some of the precip directly into the environment using a Sundqvist type algorithm
   797) !-----------------------------------------------------------------------
   798) 
   799)     use wv_saturation,  only: qsat
   800)     use phys_grid, only: get_rlat_all_p
   801) 
   802) !------------------------------Arguments--------------------------------
   803)     integer,intent(in) :: ncol, lchnk             ! number of columns and chunk index
   804)     real(r8),intent(in), dimension(pcols,pver) :: t          ! temperature (K)
   805)     real(r8),intent(in), dimension(pcols,pver) :: pmid       ! midpoint pressure (Pa) 
   806)     real(r8),intent(in), dimension(pcols,pver) :: pdel       ! layer thickness (Pa)
   807)     real(r8),intent(in), dimension(pcols,pver) :: q          ! water vapor (kg/kg)
   808)     real(r8),intent(inout), dimension(pcols,pver) :: tend_s     ! heating rate (J/kg/s)
   809)     real(r8),intent(inout), dimension(pcols,pver) :: tend_q     ! water vapor tendency (kg/kg/s)
   810)     real(r8),intent(out  ), dimension(pcols,pver) :: tend_s_snwprd ! Heating rate of snow production
   811)     real(r8),intent(out  ), dimension(pcols,pver) :: tend_s_snwevmlt ! Heating rate of evap/melting of snow
   812)     
   813) 
   814) 
   815)     real(r8), intent(in   ) :: prdprec(pcols,pver)! precipitation production (kg/ks/s)
   816)     real(r8), intent(in   ) :: cldfrc(pcols,pver) ! cloud fraction
   817)     real(r8), intent(in   ) :: deltat             ! time step
   818) 
   819)     real(r8), intent(inout) :: prec(pcols)        ! Convective-scale preciptn rate
   820)     real(r8), intent(out)   :: snow(pcols)        ! Convective-scale snowfall rate
   821) !
   822) !---------------------------Local storage-------------------------------
   823) 
   824)     real(r8) :: es    (pcols,pver)    ! Saturation vapor pressure
   825)     real(r8) :: fice   (pcols,pver)    ! ice fraction in precip production
   826)     real(r8) :: fsnow_conv(pcols,pver) ! snow fraction in precip production
   827)     real(r8) :: qs   (pcols,pver)    ! saturation specific humidity
   828)     real(r8),intent(out) :: flxprec(pcols,pverp)   ! Convective-scale flux of precip at interfaces (kg/m2/s)
   829)     real(r8),intent(out) :: flxsnow(pcols,pverp)   ! Convective-scale flux of snow   at interfaces (kg/m2/s)
   830)     real(r8),intent(out) :: ntprprd(pcols,pver)    ! net precip production in layer
   831)     real(r8),intent(out) :: ntsnprd(pcols,pver)    ! net snow production in layer
   832)     real(r8) :: work1                  ! temp variable (pjr)
   833)     real(r8) :: work2                  ! temp variable (pjr)
   834) 
   835)     real(r8) :: evpvint(pcols)         ! vertical integral of evaporation
   836)     real(r8) :: evpprec(pcols)         ! evaporation of precipitation (kg/kg/s)
   837)     real(r8) :: evpsnow(pcols)         ! evaporation of snowfall (kg/kg/s)
   838)     real(r8) :: snowmlt(pcols)         ! snow melt tendency in layer
   839)     real(r8) :: flxsntm(pcols)         ! flux of snow into layer, after melting
   840) 
   841)     real(r8) :: evplimit               ! temp variable for evaporation limits
   842)     real(r8) :: rlat(pcols)
   843) 
   844)     integer :: i,k                     ! longitude,level indices
   845) 
   846) 
   847) !-----------------------------------------------------------------------
   848) 
   849) ! convert input precip to kg/m2/s
   850)     prec(:ncol) = prec(:ncol)*1000._r8
   851) 
   852) ! determine saturation vapor pressure
   853)     call qsat(t(1:ncol, 1:pver), pmid(1:ncol, 1:pver), &
   854)          es(1:ncol, 1:pver), qs(1:ncol, 1:pver))
   855) 
   856) ! determine ice fraction in rain production (use cloud water parameterization fraction at present)
   857)     call cldfrc_fice(ncol, t, fice, fsnow_conv)
   858) 
   859) ! zero the flux integrals on the top boundary
   860)     flxprec(:ncol,1) = 0._r8
   861)     flxsnow(:ncol,1) = 0._r8
   862)     evpvint(:ncol)   = 0._r8
   863) 
   864)     do k = 1, pver
   865)        do i = 1, ncol
   866) 
   867) ! Melt snow falling into layer, if necessary. 
   868)           if (t(i,k) > tmelt) then
   869)              flxsntm(i) = 0._r8
   870)              snowmlt(i) = flxsnow(i,k) * gravit/ pdel(i,k)
   871)           else
   872)              flxsntm(i) = flxsnow(i,k)
   873)              snowmlt(i) = 0._r8
   874)           end if
   875) 
   876) ! relative humidity depression must be > 0 for evaporation
   877)           evplimit = max(1._r8 - q(i,k)/qs(i,k), 0._r8)
   878) 
   879) ! total evaporation depends on flux in the top of the layer
   880) ! flux prec is the net production above layer minus evaporation into environmet
   881)           evpprec(i) = ke * (1._r8 - cldfrc(i,k)) * evplimit * sqrt(flxprec(i,k))
   882) !**********************************************************
   883) !!          evpprec(i) = 0.    ! turn off evaporation for now
   884) !**********************************************************
   885) 
   886) ! Don't let evaporation supersaturate layer (approx). Layer may already be saturated.
   887) ! Currently does not include heating/cooling change to qs
   888)           evplimit   = max(0._r8, (qs(i,k)-q(i,k)) / deltat)
   889) 
   890) ! Don't evaporate more than is falling into the layer - do not evaporate rain formed
   891) ! in this layer but if precip production is negative, remove from the available precip
   892) ! Negative precip production occurs because of evaporation in downdrafts.
   893) !!$          evplimit   = flxprec(i,k) * gravit / pdel(i,k) + min(prdprec(i,k), 0.)
   894)           evplimit   = min(evplimit, flxprec(i,k) * gravit / pdel(i,k))
   895) 
   896) ! Total evaporation cannot exceed input precipitation
   897)           evplimit   = min(evplimit, (prec(i) - evpvint(i)) * gravit / pdel(i,k))
   898) 
   899)           evpprec(i) = min(evplimit, evpprec(i))
   900) 
   901) ! evaporation of snow depends on snow fraction of total precipitation in the top after melting
   902)           if (flxprec(i,k) > 0._r8) then
   903) !            evpsnow(i) = evpprec(i) * flxsntm(i) / flxprec(i,k)
   904) !            prevent roundoff problems
   905)              work1 = min(max(0._r8,flxsntm(i)/flxprec(i,k)),1._r8)
   906)              evpsnow(i) = evpprec(i) * work1
   907)           else
   908)              evpsnow(i) = 0._r8
   909)           end if
   910) 
   911) ! vertically integrated evaporation
   912)           evpvint(i) = evpvint(i) + evpprec(i) * pdel(i,k)/gravit
   913) 
   914) ! net precip production is production - evaporation
   915)           ntprprd(i,k) = prdprec(i,k) - evpprec(i)
   916) ! net snow production is precip production * ice fraction - evaporation - melting
   917) !pjrworks ntsnprd(i,k) = prdprec(i,k)*fice(i,k) - evpsnow(i) - snowmlt(i)
   918) !pjrwrks2 ntsnprd(i,k) = prdprec(i,k)*fsnow_conv(i,k) - evpsnow(i) - snowmlt(i)
   919) ! the small amount added to flxprec in the work1 expression has been increased from 
   920) ! 1e-36 to 8.64e-11 (1e-5 mm/day).  This causes the temperature based partitioning
   921) ! scheme to be used for small flxprec amounts.  This is to address error growth problems.
   922) #ifdef PERGRO
   923)           work1 = min(max(0._r8,flxsnow(i,k)/(flxprec(i,k)+8.64e-11_r8)),1._r8)
   924) #else
   925)           if (flxprec(i,k).gt.0._r8) then
   926)              work1 = min(max(0._r8,flxsnow(i,k)/flxprec(i,k)),1._r8)
   927)           else
   928)              work1 = 0._r8
   929)           endif
   930) #endif
   931)           work2 = max(fsnow_conv(i,k), work1)
   932)           if (snowmlt(i).gt.0._r8) work2 = 0._r8
   933) !         work2 = fsnow_conv(i,k)
   934)           ntsnprd(i,k) = prdprec(i,k)*work2 - evpsnow(i) - snowmlt(i)
   935)           tend_s_snwprd  (i,k) = prdprec(i,k)*work2*latice
   936)           tend_s_snwevmlt(i,k) = - ( evpsnow(i) + snowmlt(i) )*latice
   937) 
   938) ! precipitation fluxes
   939)           flxprec(i,k+1) = flxprec(i,k) + ntprprd(i,k) * pdel(i,k)/gravit
   940)           flxsnow(i,k+1) = flxsnow(i,k) + ntsnprd(i,k) * pdel(i,k)/gravit
   941) 
   942) ! protect against rounding error
   943)           flxprec(i,k+1) = max(flxprec(i,k+1), 0._r8)
   944)           flxsnow(i,k+1) = max(flxsnow(i,k+1), 0._r8)
   945) ! more protection (pjr)
   946) !         flxsnow(i,k+1) = min(flxsnow(i,k+1), flxprec(i,k+1))
   947) 
   948) ! heating (cooling) and moistening due to evaporation 
   949) ! - latent heat of vaporization for precip production has already been accounted for
   950) ! - snow is contained in prec
   951)           tend_s(i,k)   =-evpprec(i)*latvap + ntsnprd(i,k)*latice
   952)           tend_q(i,k) = evpprec(i)
   953)        end do
   954)     end do
   955) 
   956) ! set output precipitation rates (m/s)
   957)     prec(:ncol) = flxprec(:ncol,pver+1) / 1000._r8
   958)     snow(:ncol) = flxsnow(:ncol,pver+1) / 1000._r8
   959) 
   960) !**********************************************************
   961) !!$    tend_s(:ncol,:)   = 0.      ! turn heating off
   962) !**********************************************************
   963) 
   964)   end subroutine zm_conv_evap

   968) subroutine convtran(lchnk   , &
   969)                     doconvtran,q       ,ncnst   ,mu      ,md      , &
   970)                     du      ,eu      ,ed      ,dp      ,dsubcld , &
   971)                     jt      ,mx      ,ideep   ,il1g    ,il2g    , &
   972)                     nstep   ,fracis  ,dqdt    ,dpdry   )
   973) !----------------------------------------------------------------------- 
   974) ! 
   975) ! Purpose: 
   976) ! Convective transport of trace species
   977) !
   978) ! Mixing ratios may be with respect to either dry or moist air
   979) ! 
   980) ! Method: 
   981) ! <Describe the algorithm(s) used in the routine.> 
   982) ! <Also include any applicable external references.> 
   983) ! 
   984) ! Author: P. Rasch
   985) ! 
   986) !-----------------------------------------------------------------------
   987)    use shr_kind_mod, only: r8 => shr_kind_r8
   988)    use constituents,    only: cnst_get_type_byind
   989)    use ppgrid
   990)    use abortutils, only: endrun
   991) 
   992)    implicit none
   993) !-----------------------------------------------------------------------
   994) !
   995) ! Input arguments
   996) !
   997)    integer, intent(in) :: lchnk                 ! chunk identifier
   998)    integer, intent(in) :: ncnst                 ! number of tracers to transport
   999)    logical, intent(in) :: doconvtran(ncnst)     ! flag for doing convective transport
  1000)    real(r8), intent(in) :: q(pcols,pver,ncnst)  ! Tracer array including moisture
  1001)    real(r8), intent(in) :: mu(pcols,pver)       ! Mass flux up
  1002)    real(r8), intent(in) :: md(pcols,pver)       ! Mass flux down
  1003)    real(r8), intent(in) :: du(pcols,pver)       ! Mass detraining from updraft
  1004)    real(r8), intent(in) :: eu(pcols,pver)       ! Mass entraining from updraft
  1005)    real(r8), intent(in) :: ed(pcols,pver)       ! Mass entraining from downdraft
  1006)    real(r8), intent(in) :: dp(pcols,pver)       ! Delta pressure between interfaces
  1007)    real(r8), intent(in) :: dsubcld(pcols)       ! Delta pressure from cloud base to sfc
  1008)    real(r8), intent(in) :: fracis(pcols,pver,ncnst) ! fraction of tracer that is insoluble
  1009) 
  1010)    integer, intent(in) :: jt(pcols)         ! Index of cloud top for each column
  1011)    integer, intent(in) :: mx(pcols)         ! Index of cloud top for each column
  1012)    integer, intent(in) :: ideep(pcols)      ! Gathering array
  1013)    integer, intent(in) :: il1g              ! Gathered min lon indices over which to operate
  1014)    integer, intent(in) :: il2g              ! Gathered max lon indices over which to operate
  1015)    integer, intent(in) :: nstep             ! Time step index
  1016) 
  1017)    real(r8), intent(in) :: dpdry(pcols,pver)       ! Delta pressure between interfaces
  1018) 
  1019) 
  1020) ! input/output
  1021) 
  1022)    real(r8), intent(out) :: dqdt(pcols,pver,ncnst)  ! Tracer tendency array
  1023) 
  1024) !--------------------------Local Variables------------------------------
  1025) 
  1026)    integer i                 ! Work index
  1027)    integer k                 ! Work index
  1028)    integer kbm               ! Highest altitude index of cloud base
  1029)    integer kk                ! Work index
  1030)    integer kkp1              ! Work index
  1031)    integer km1               ! Work index
  1032)    integer kp1               ! Work index
  1033)    integer ktm               ! Highest altitude index of cloud top
  1034)    integer m                 ! Work index
  1035) 
  1036)    real(r8) cabv                 ! Mix ratio of constituent above
  1037)    real(r8) cbel                 ! Mix ratio of constituent below
  1038)    real(r8) cdifr                ! Normalized diff between cabv and cbel
  1039)    real(r8) chat(pcols,pver)     ! Mix ratio in env at interfaces
  1040)    real(r8) cond(pcols,pver)     ! Mix ratio in downdraft at interfaces
  1041)    real(r8) const(pcols,pver)    ! Gathered tracer array
  1042)    real(r8) fisg(pcols,pver)     ! gathered insoluble fraction of tracer
  1043)    real(r8) conu(pcols,pver)     ! Mix ratio in updraft at interfaces
  1044)    real(r8) dcondt(pcols,pver)   ! Gathered tend array
  1045)    real(r8) small                ! A small number
  1046)    real(r8) mbsth                ! Threshold for mass fluxes
  1047)    real(r8) mupdudp              ! A work variable
  1048)    real(r8) minc                 ! A work variable
  1049)    real(r8) maxc                 ! A work variable
  1050)    real(r8) fluxin               ! A work variable
  1051)    real(r8) fluxout              ! A work variable
  1052)    real(r8) netflux              ! A work variable
  1053) 
  1054)    real(r8) dutmp(pcols,pver)       ! Mass detraining from updraft
  1055)    real(r8) eutmp(pcols,pver)       ! Mass entraining from updraft
  1056)    real(r8) edtmp(pcols,pver)       ! Mass entraining from downdraft
  1057)    real(r8) dptmp(pcols,pver)    ! Delta pressure between interfaces
  1058) !-----------------------------------------------------------------------
  1059) !
  1060)    small = 1.e-36_r8
  1061) ! mbsth is the threshold below which we treat the mass fluxes as zero (in mb/s)
  1062)    mbsth = 1.e-15_r8
  1063) 
  1064) ! Find the highest level top and bottom levels of convection
  1065)    ktm = pver
  1066)    kbm = pver
  1067)    do i = il1g, il2g
  1068)       ktm = min(ktm,jt(i))
  1069)       kbm = min(kbm,mx(i))
  1070)    end do
  1071) 
  1072) ! Loop ever each constituent
  1073)    do m = 2, ncnst
  1074)       if (doconvtran(m)) then
  1075) 
  1076)          if (cnst_get_type_byind(m).eq.'dry') then
  1077)             do k = 1,pver
  1078)                do i =il1g,il2g
  1079)                   dptmp(i,k) = dpdry(i,k)
  1080)                   dutmp(i,k) = du(i,k)*dp(i,k)/dpdry(i,k)
  1081)                   eutmp(i,k) = eu(i,k)*dp(i,k)/dpdry(i,k)
  1082)                   edtmp(i,k) = ed(i,k)*dp(i,k)/dpdry(i,k)
  1083)                end do
  1084)             end do
  1085)          else
  1086)             do k = 1,pver
  1087)                do i =il1g,il2g
  1088)                   dptmp(i,k) = dp(i,k)
  1089)                   dutmp(i,k) = du(i,k)
  1090)                   eutmp(i,k) = eu(i,k)
  1091)                   edtmp(i,k) = ed(i,k)
  1092)                end do
  1093)             end do
  1094)          endif
  1095) !        dptmp = dp
  1096) 
  1097) ! Gather up the constituent and set tend to zero
  1098)          do k = 1,pver
  1099)             do i =il1g,il2g
  1100)                const(i,k) = q(ideep(i),k,m)
  1101)                fisg(i,k) = fracis(ideep(i),k,m)
  1102)             end do
  1103)          end do
  1104) 
  1105) ! From now on work only with gathered data
  1106) 
  1107) ! Interpolate environment tracer values to interfaces
  1108)          do k = 1,pver
  1109)             km1 = max(1,k-1)
  1110)             do i = il1g, il2g
  1111)                minc = min(const(i,km1),const(i,k))
  1112)                maxc = max(const(i,km1),const(i,k))
  1113)                if (minc < 0) then
  1114)                   cdifr = 0._r8
  1115)                else
  1116)                   cdifr = abs(const(i,k)-const(i,km1))/max(maxc,small)
  1117)                endif
  1118) 
  1119) ! If the two layers differ significantly use a geometric averaging
  1120) ! procedure
  1121)                if (cdifr > 1.E-6_r8) then
  1122)                   cabv = max(const(i,km1),maxc*1.e-12_r8)
  1123)                   cbel = max(const(i,k),maxc*1.e-12_r8)
  1124)                   chat(i,k) = log(cabv/cbel)/(cabv-cbel)*cabv*cbel
  1125) 
  1126)                else             ! Small diff, so just arithmetic mean
  1127)                   chat(i,k) = 0.5_r8* (const(i,k)+const(i,km1))
  1128)                end if
  1129) 
  1130) ! Provisional up and down draft values
  1131)                conu(i,k) = chat(i,k)
  1132)                cond(i,k) = chat(i,k)
  1133) 
  1134) !              provisional tends
  1135)                dcondt(i,k) = 0._r8
  1136) 
  1137)             end do
  1138)          end do
  1139) 
  1140) ! Do levels adjacent to top and bottom
  1141)          k = 2
  1142)          km1 = 1
  1143)          kk = pver
  1144)          do i = il1g,il2g
  1145)             mupdudp = mu(i,kk) + dutmp(i,kk)*dptmp(i,kk)
  1146)             if (mupdudp > mbsth) then
  1147)                conu(i,kk) = (+eutmp(i,kk)*fisg(i,kk)*const(i,kk)*dptmp(i,kk))/mupdudp
  1148)             endif
  1149)             if (md(i,k) < -mbsth) then
  1150)                cond(i,k) =  (-edtmp(i,km1)*fisg(i,km1)*const(i,km1)*dptmp(i,km1))/md(i,k)
  1151)             endif
  1152)          end do
  1153) 
  1154) ! Updraft from bottom to top
  1155)          do kk = pver-1,1,-1
  1156)             kkp1 = min(pver,kk+1)
  1157)             do i = il1g,il2g
  1158)                mupdudp = mu(i,kk) + dutmp(i,kk)*dptmp(i,kk)
  1159)                if (mupdudp > mbsth) then
  1160)                   conu(i,kk) = (  mu(i,kkp1)*conu(i,kkp1)+eutmp(i,kk)*fisg(i,kk)* &
  1161)                                   const(i,kk)*dptmp(i,kk) )/mupdudp
  1162)                endif
  1163)             end do
  1164)          end do
  1165) 
  1166) ! Downdraft from top to bottom
  1167)          do k = 3,pver
  1168)             km1 = max(1,k-1)
  1169)             do i = il1g,il2g
  1170)                if (md(i,k) < -mbsth) then
  1171)                   cond(i,k) =  (  md(i,km1)*cond(i,km1)-edtmp(i,km1)*fisg(i,km1)*const(i,km1) &
  1172)                                   *dptmp(i,km1) )/md(i,k)
  1173)                endif
  1174)             end do
  1175)          end do
  1176) 
  1177) 
  1178)          do k = ktm,pver
  1179)             km1 = max(1,k-1)
  1180)             kp1 = min(pver,k+1)
  1181)             do i = il1g,il2g
  1182) 
  1183) ! version 1 hard to check for roundoff errors
  1184) !               dcondt(i,k) =
  1185) !     $                  +(+mu(i,kp1)* (conu(i,kp1)-chat(i,kp1))
  1186) !     $                    -mu(i,k)*   (conu(i,k)-chat(i,k))
  1187) !     $                    +md(i,kp1)* (cond(i,kp1)-chat(i,kp1))
  1188) !     $                    -md(i,k)*   (cond(i,k)-chat(i,k))
  1189) !     $                   )/dp(i,k)
  1190) 
  1191) ! version 2 hard to limit fluxes
  1192) !               fluxin =  mu(i,kp1)*conu(i,kp1) + mu(i,k)*chat(i,k)
  1193) !     $                 -(md(i,k)  *cond(i,k)   + md(i,kp1)*chat(i,kp1))
  1194) !               fluxout = mu(i,k)*conu(i,k)     + mu(i,kp1)*chat(i,kp1)
  1195) !     $                 -(md(i,kp1)*cond(i,kp1) + md(i,k)*chat(i,k))
  1196) 
  1197) ! version 3 limit fluxes outside convection to mass in appropriate layer
  1198) ! these limiters are probably only safe for positive definite quantitities
  1199) ! it assumes that mu and md already satify a courant number limit of 1
  1200)                fluxin =  mu(i,kp1)*conu(i,kp1)+ mu(i,k)*min(chat(i,k),const(i,km1)) &
  1201)                          -(md(i,k)  *cond(i,k) + md(i,kp1)*min(chat(i,kp1),const(i,kp1)))
  1202)                fluxout = mu(i,k)*conu(i,k) + mu(i,kp1)*min(chat(i,kp1),const(i,k)) &
  1203)                          -(md(i,kp1)*cond(i,kp1) + md(i,k)*min(chat(i,k),const(i,k)))
  1204) 
  1205)                netflux = fluxin - fluxout
  1206)                if (abs(netflux) < max(fluxin,fluxout)*1.e-12_r8) then
  1207)                   netflux = 0._r8
  1208)                endif
  1209)                dcondt(i,k) = netflux/dptmp(i,k)
  1210)             end do
  1211)          end do
  1212) ! %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  1213) !
  1214) !DIR$ NOINTERCHANGE
  1215)          do k = kbm,pver
  1216)             km1 = max(1,k-1)
  1217)             do i = il1g,il2g
  1218)                if (k == mx(i)) then
  1219) 
  1220) ! version 1
  1221) !                  dcondt(i,k) = (1./dsubcld(i))*
  1222) !     $              (-mu(i,k)*(conu(i,k)-chat(i,k))
  1223) !     $               -md(i,k)*(cond(i,k)-chat(i,k))
  1224) !     $              )
  1225) 
  1226) ! version 2
  1227) !                  fluxin =  mu(i,k)*chat(i,k) - md(i,k)*cond(i,k)
  1228) !                  fluxout = mu(i,k)*conu(i,k) - md(i,k)*chat(i,k)
  1229) ! version 3
  1230)                   fluxin =  mu(i,k)*min(chat(i,k),const(i,km1)) - md(i,k)*cond(i,k)
  1231)                   fluxout = mu(i,k)*conu(i,k) - md(i,k)*min(chat(i,k),const(i,k))
  1232) 
  1233)                   netflux = fluxin - fluxout
  1234)                   if (abs(netflux) < max(fluxin,fluxout)*1.e-12_r8) then
  1235)                      netflux = 0._r8
  1236)                   endif
  1237) !                  dcondt(i,k) = netflux/dsubcld(i)
  1238)                   dcondt(i,k) = netflux/dptmp(i,k)
  1239)                else if (k > mx(i)) then
  1240) !                  dcondt(i,k) = dcondt(i,k-1)
  1241)                   dcondt(i,k) = 0._r8
  1242)                end if
  1243)             end do
  1244)          end do
  1245) 
  1246) ! Initialize to zero everywhere, then scatter tendency back to full array
  1247)          dqdt(:,:,m) = 0._r8
  1248)          do k = 1,pver
  1249)             kp1 = min(pver,k+1)
  1250) !DIR$ CONCURRENT
  1251)             do i = il1g,il2g
  1252)                dqdt(ideep(i),k,m) = dcondt(i,k)
  1253)             end do
  1254)          end do
  1255) 
  1256)       end if      ! for doconvtran
  1257) 
  1258)    end do
  1259) 
  1260)    return
  1261) end subroutine convtran

  1337)    real(r8) cdifr                ! Normalized diff between cabv and cbel
  1338)    real(r8) chat(pcols,pver)     ! Mix ratio in env at interfaces
  1339)    real(r8) cond(pcols,pver)     ! Mix ratio in downdraft at interfaces
  1340)    real(r8) const(pcols,pver)    ! Gathered wind array
  1341)    real(r8) conu(pcols,pver)     ! Mix ratio in updraft at interfaces
  1342)    real(r8) dcondt(pcols,pver)   ! Gathered tend array
  1343)    real(r8) small                ! A small number
  1344)    real(r8) mbsth                ! Threshold for mass fluxes
  1345)    real(r8) mupdudp              ! A work variable
  1346)    real(r8) minc                 ! A work variable
  1347)    real(r8) maxc                 ! A work variable
  1348)    real(r8) fluxin               ! A work variable
  1349)    real(r8) fluxout              ! A work variable
  1350)    real(r8) netflux              ! A work variable
  1351) 
  1352)    real(r8) momcu                ! constant for updraft pressure gradient term
  1353)    real(r8) momcd                ! constant for downdraft pressure gradient term
  1354)    real(r8) sum                  ! sum
  1355)    real(r8) sum2                  ! sum2
  1356)  
  1357)    real(r8) mududp(pcols,pver) ! working variable
  1358)    real(r8) mddudp(pcols,pver)     ! working variable
  1359) 
  1360)    real(r8) pgu(pcols,pver)      ! Pressure gradient term for updraft
  1361)    real(r8) pgd(pcols,pver)      ! Pressure gradient term for downdraft
  1362) 
  1363)    real(r8),intent(out) ::  pguall(pcols,pver,ncnst)      ! Apparent force from  updraft PG
  1364)    real(r8),intent(out) ::  pgdall(pcols,pver,ncnst)      ! Apparent force from  downdraft PG
  1365) 
  1366)    real(r8),intent(out) ::  icwu(pcols,pver,ncnst)      ! In-cloud winds in updraft
  1367)    real(r8),intent(out) ::  icwd(pcols,pver,ncnst)      ! In-cloud winds in downdraft
  1368) 
  1369)    real(r8),intent(out) ::  seten(pcols,pver) ! Dry static energy tendency
  1370)    real(r8)                 gseten(pcols,pver) ! Gathered dry static energy tendency
  1371) 
  1372)    real(r8)  mflux(pcols,pverp,ncnst)   ! Gathered momentum flux
  1373) 
  1374)    real(r8)  wind0(pcols,pver,ncnst)       !  gathered  wind before time step
  1375)    real(r8)  windf(pcols,pver,ncnst)       !  gathered  wind after time step
  1376)    real(r8) fkeb, fket, ketend_cons, ketend, utop, ubot, vtop, vbot, gset2
  1377)    
  1378) 
  1379) !-----------------------------------------------------------------------
  1380) !
  1381) 
  1382) ! Initialize outgoing fields
  1383)    pguall(:,:,:)     = 0.0_r8
  1384)    pgdall(:,:,:)     = 0.0_r8
  1385) ! Initialize in-cloud winds to environmental wind
  1386)    icwu(:ncol,:,:)       = q(:ncol,:,:)
  1387)    icwd(:ncol,:,:)       = q(:ncol,:,:)
  1388) 
  1389) ! Initialize momentum flux and  final winds
  1390)    mflux(:,:,:)       = 0.0_r8
  1391)    wind0(:,:,:)         = 0.0_r8
  1392)    windf(:,:,:)         = 0.0_r8
  1393) 
  1394) ! Initialize dry static energy
  1395) 
  1396)    seten(:,:)         = 0.0_r8
  1397)    gseten(:,:)         = 0.0_r8
  1398) 
  1399) ! Define constants for parameterization
  1400) 
  1401)    momcu = 0.4_r8
  1402)    momcd = 0.4_r8
  1403) 
  1404)    small = 1.e-36_r8
  1405) ! mbsth is the threshold below which we treat the mass fluxes as zero (in mb/s)
  1406)    mbsth = 1.e-15_r8
  1407) 
  1408) ! Find the highest level top and bottom levels of convection
  1409)    ktm = pver
  1410)    kbm = pver
  1411)    do i = il1g, il2g
  1412)       ktm = min(ktm,jt(i))
  1413)       kbm = min(kbm,mx(i))
  1414)    end do
  1415) 
  1416) ! Loop ever each wind component
  1417)    do m = 1, ncnst                    !start at m = 1 to transport momentum
  1418)       if (domomtran(m)) then
  1419) 
  1420) ! Gather up the winds and set tend to zero
  1421)          do k = 1,pver
  1422)             do i =il1g,il2g
  1423)                const(i,k) = q(ideep(i),k,m)
  1424)                 wind0(i,k,m) = const(i,k)
  1425)             end do
  1426)          end do
  1427) 
  1428) 
  1429) ! From now on work only with gathered data
  1430) 
  1431) ! Interpolate winds to interfaces
  1432) 
  1433)          do k = 1,pver
  1434)             km1 = max(1,k-1)
  1435)             do i = il1g, il2g
  1436) 
  1437)                ! use arithmetic mean
  1438)                chat(i,k) = 0.5_r8* (const(i,k)+const(i,km1))
  1439) 
  1440) ! Provisional up and down draft values
  1441)                conu(i,k) = chat(i,k)
  1442)                cond(i,k) = chat(i,k)
  1443) 
  1444) !              provisional tends
  1445)                dcondt(i,k) = 0._r8
  1446) 
  1447)             end do
  1448)          end do
  1449) 
  1450) 
  1451) !
  1452) ! Pressure Perturbation Term
  1453) ! 
  1454) 
  1455)       !Top boundary:  assume mu is zero 
  1456) 
  1457)          k=1
  1458)          pgu(:il2g,k) = 0.0_r8
  1459)          pgd(:il2g,k) = 0.0_r8
  1460) 
  1461)          do k=2,pver-1
  1462)             km1 = max(1,k-1)
  1463)             kp1 = min(pver,k+1)
  1464)             do i = il1g,il2g
  1465)             
  1466)                !interior points
  1467) 
  1468)                mududp(i,k) =  ( mu(i,k) * (const(i,k)- const(i,km1))/dp(i,km1) &
  1469)                            +  mu(i,kp1) * (const(i,kp1) - const(i,k))/dp(i,k))
  1470) 
  1471)                pgu(i,k) = - momcu * 0.5_r8 * mududp(i,k)
  1472)                            
  1473) 
  1474)                mddudp(i,k) =  ( md(i,k) * (const(i,k)- const(i,km1))/dp(i,km1) &
  1475)                            +  md(i,kp1) * (const(i,kp1) - const(i,k))/dp(i,k))
  1476) 
  1477)                pgd(i,k) = - momcd * 0.5_r8 * mddudp(i,k)
  1478) 
  1479) 
  1480)             end do
  1481)          end do
  1482) 
  1483)        ! bottom boundary 
  1484)        k = pver
  1485)        km1 = max(1,k-1)
  1486)        do i=il1g,il2g
  1487) 
  1488)           mududp(i,k) =   mu(i,k) * (const(i,k)- const(i,km1))/dp(i,km1)
  1489)           pgu(i,k) = - momcu *  mududp(i,k)
  1490)           
  1491)           mddudp(i,k) =   md(i,k) * (const(i,k)- const(i,km1))/dp(i,km1) 
  1492) 
  1493)           pgd(i,k) = - momcd * mddudp(i,k)
  1494)           
  1495)        end do
  1496)        
  1497) 
  1498) !
  1499) ! In-cloud velocity calculations
  1500) !
  1501) 
  1502) ! Do levels adjacent to top and bottom
  1503)          k = 2
  1504)          km1 = 1
  1505)          kk = pver
  1506)          kkm1 = max(1,kk-1)
  1507)          do i = il1g,il2g
  1508)             mupdudp = mu(i,kk) + du(i,kk)*dp(i,kk)
  1509)             if (mupdudp > mbsth) then
  1510)                  
  1511)                conu(i,kk) = (+eu(i,kk)*const(i,kk)*dp(i,kk)+pgu(i,kk)*dp(i,kk))/mupdudp
  1512)             endif
  1513)             if (md(i,k) < -mbsth) then
  1514)                cond(i,k) =  (-ed(i,km1)*const(i,km1)*dp(i,km1))-pgd(i,km1)*dp(i,km1)/md(i,k)
  1515)             endif
  1516) 
  1517)                         
  1518)          end do
  1519) 
  1520) 
  1521) 
  1522) ! Updraft from bottom to top
  1523)          do kk = pver-1,1,-1
  1524)             kkm1 = max(1,kk-1)
  1525)             kkp1 = min(pver,kk+1)
  1526)             do i = il1g,il2g
  1527)                mupdudp = mu(i,kk) + du(i,kk)*dp(i,kk)
  1528)                if (mupdudp > mbsth) then
  1529)             
  1530)                   conu(i,kk) = (  mu(i,kkp1)*conu(i,kkp1)+eu(i,kk)* &
  1531)                                   const(i,kk)*dp(i,kk)+pgu(i,kk)*dp(i,kk))/mupdudp
  1532)                endif
  1533)             end do
  1534) 
  1535)          end do
  1536) 
  1537) 
  1538) ! Downdraft from top to bottom
  1539)          do k = 3,pver
  1540)             km1 = max(1,k-1)
  1541)             do i = il1g,il2g
  1542)                if (md(i,k) < -mbsth) then
  1543)                             
  1544)                   cond(i,k) =  (  md(i,km1)*cond(i,km1)-ed(i,km1)*const(i,km1) &
  1545)                                   *dp(i,km1)-pgd(i,km1)*dp(i,km1) )/md(i,k)
  1546) 
  1547)                endif
  1548)             end do
  1549)          end do
  1550) 
  1551) 
  1552)          sum = 0._r8
  1553)          sum2 = 0._r8
  1554) 
  1555) 
  1556)          do k = ktm,pver
  1557)             km1 = max(1,k-1)
  1558)             kp1 = min(pver,k+1)
  1559)             do i = il1g,il2g
  1560)                ii = ideep(i)
  1561) 	
  1562) ! version 1 hard to check for roundoff errors
  1563)                dcondt(i,k) =  &
  1564)                            +(mu(i,kp1)* (conu(i,kp1)-chat(i,kp1)) &
  1565)                            -mu(i,k)*   (conu(i,k)-chat(i,k))      &
  1566)                            +md(i,kp1)* (cond(i,kp1)-chat(i,kp1)) &
  1567)                            -md(i,k)*   (cond(i,k)-chat(i,k)) &
  1568)                           )/dp(i,k)
  1569) 
  1570)             end do
  1571)          end do
  1572) 
  1573)   ! dcont for bottom layer
  1574)           !
  1575)           !DIR$ NOINTERCHANGE
  1576)           do k = kbm,pver
  1577)              km1 = max(1,k-1)
  1578)              do i = il1g,il2g
  1579)                 if (k == mx(i)) then
  1580) 
  1581)                    ! version 1
  1582)                    dcondt(i,k) = (1._r8/dp(i,k))*   &  
  1583)                         (-mu(i,k)*(conu(i,k)-chat(i,k)) &
  1584)                         -md(i,k)*(cond(i,k)-chat(i,k)) &
  1585)                         )
  1586)                 end if
  1587)              end do
  1588)           end do
  1589) 
  1590) ! Initialize to zero everywhere, then scatter tendency back to full array
  1591)          dqdt(:,:,m) = 0._r8
  1592) 
  1593)          do k = 1,pver
  1594)             do i = il1g,il2g
  1595)                ii = ideep(i)
  1596)                dqdt(ii,k,m) = dcondt(i,k)
  1597)     ! Output apparent force on the mean flow from pressure gradient
  1598)                pguall(ii,k,m) = -pgu(i,k)
  1599)                pgdall(ii,k,m) = -pgd(i,k)
  1600)                icwu(ii,k,m)   =  conu(i,k)
  1601)                icwd(ii,k,m)   =  cond(i,k)
  1602)             end do
  1603)          end do
  1604) 
  1605)           ! Calculate momentum flux in units of mb*m/s2 
  1606) 
  1607)           do k = ktm,pver
  1608)              do i = il1g,il2g
  1609)                 ii = ideep(i)
  1610)                 mflux(i,k,m) = &
  1611)                      -mu(i,k)*   (conu(i,k)-chat(i,k))      &
  1612)                      -md(i,k)*   (cond(i,k)-chat(i,k))
  1613)              end do
  1614)           end do
  1615) 
  1616) 
  1617)           ! Calculate winds at the end of the time step 
  1618) 
  1619)           do k = ktm,pver
  1620)              do i = il1g,il2g
  1621)                 ii = ideep(i)
  1622)                 km1 = max(1,k-1)
  1623)                 kp1 = k+1
  1624)                 windf(i,k,m) = const(i,k)    -   (mflux(i,kp1,m) - mflux(i,k,m)) * dt /dp(i,k)
  1625) 
  1626)              end do
  1627)           end do
  1628) 
  1629)        end if      ! for domomtran
  1630)    end do
  1631) 
  1632)  ! Need to add an energy fix to account for the dissipation of kinetic energy
  1633)     ! Formulation follows from Boville and Bretherton (2003)
  1634)     ! formulation by PJR
  1635) 
  1636)     do k = ktm,pver
  1637)        km1 = max(1,k-1)
  1638)        kp1 = min(pver,k+1)
  1639)        do i = il1g,il2g
  1640) 
  1641)           ii = ideep(i)
  1642) 
  1643)           ! calculate the KE fluxes at top and bot of layer 
  1644)           ! based on a discrete approximation to b&b eq(35) F_KE = u*F_u + v*F_v at interface
  1645)           utop = (wind0(i,k,1)+wind0(i,km1,1))/2._r8
  1646)           vtop = (wind0(i,k,2)+wind0(i,km1,2))/2._r8
  1647)           ubot = (wind0(i,kp1,1)+wind0(i,k,1))/2._r8
  1648)           vbot = (wind0(i,kp1,2)+wind0(i,k,2))/2._r8
  1649)           fket = utop*mflux(i,k,1)   + vtop*mflux(i,k,2)    ! top of layer
  1650)           fkeb = ubot*mflux(i,k+1,1) + vbot*mflux(i,k+1,2)  ! bot of layer
  1651) 
  1652)           ! divergence of these fluxes should give a conservative redistribution of KE
  1653)           ketend_cons = (fket-fkeb)/dp(i,k)
  1654) 
  1655)           ! tendency in kinetic energy resulting from the momentum transport
  1656)           ketend = ((windf(i,k,1)**2 + windf(i,k,2)**2) - (wind0(i,k,1)**2 + wind0(i,k,2)**2))*0.5_r8/dt
  1657) 
  1658)           ! the difference should be the dissipation
  1659)           gset2 = ketend_cons - ketend
  1660)           gseten(i,k) = gset2
  1661) 
  1662)        end do
  1663) 
  1664)     end do
  1665) 
  1666)     ! Scatter dry static energy to full array
  1667)     do k = 1,pver
  1668)        do i = il1g,il2g
  1669)           ii = ideep(i)
  1670)           seten(ii,k) = gseten(i,k)
  1671) 
  1672)        end do
  1673)     end do
  1674) 
  1675)    return
  1676) end subroutine momtran

  1733)    real(r8) tpv(pcols,pver)      !

  1736)    real(r8) a1(pcols)
  1737)    real(r8) a2(pcols)

  1928)             call qsat_hPa(tp(i,k), p(i,k), estp(i), qstp(i,k))
  1929) !-jjh          tpv(i,k) =tp(i,k)*(1.+1.608*qstp(i,k))/

  1932)             buoy(i,k) = tpv(i,k) - tv(i,k) + tiedke_add
  1933)          end if
  1934)       end do
  1935)    end do
  1936) 
  1937) !
  1938)    do k = msg + 2,pver
  1939)       do i = 1,ncol
  1940)          if (k < lcl(i) .and. plge600(i)) then
  1941)             if (buoy(i,k+1) > 0._r8 .and. buoy(i,k) <= 0._r8) then

  1943)                lelten(i,knt(i)) = k
  1944)             end if
  1945)          end if
  1946)       end do
  1947)    end do
  1948) !
  1949) ! calculate convective available potential energy (cape).
  1950) !
  1951)    do n = 1,5
  1952)       do k = msg + 1,pver
  1953)          do i = 1,ncol
  1954)             if (plge600(i) .and. k <= mx(i) .and. k > lelten(i,n)) then
  1955)                capeten(i,n) = capeten(i,n) + rd*buoy(i,k)*log(pf(i,k+1)/pf(i,k))
  1956)             end if
  1957)          end do
  1958)       end do
  1959)    end do
  1960) !
  1961) ! find maximum cape from all possible tentative capes from
  1962) ! one sounding,
  1963) ! and use it as the final cape, april 26, 1995
  1964) !
  1965)    do n = 1,5
  1966)       do i = 1,ncol
  1967)          if (capeten(i,n) > cape(i)) then
  1968)             cape(i) = capeten(i,n)
  1969)             lel(i) = lelten(i,n)
  1970)          end if
  1971)       end do
  1972)    end do
  1973) !
  1974) ! put lower bound on cape for diagnostic purposes.
  1975) !
  1976)    do i = 1,ncol
  1977)       cape(i) = max(cape(i), 0._r8)
  1978)    end do
  1979) !
  1980)    return

  3436) !            write(iulog,*)' s',smix(i,k+1),slcl,smix(i,lcl(i))
  3437) !            write(iulog,*)'qt',qtmix(i,k+1),qtlcl,qtmix(i,lcl(i))
  3438) !            write(iulog,*)'qs',qsmix(i,k+1),qslcl,qsmix(i,lcl(i))
