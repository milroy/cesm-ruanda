   101)   subroutine write_interpolated_scalar(File, varid, fld, numlev, data_type, decomp_type) 
   102)     use pio, only : file_desc_t, io_desc_t, var_desc_t, pio_write_darray, iosystem_desc_t, &
   103)          pio_initdecomp, pio_freedecomp, pio_setdebuglevel
   104)     use cam_instance, only: atm_id
   105)     use spmd_dyn,       only: local_dp_map, block_buf_nrecs, chunk_buf_nrecs
   106)     use ppgrid, only : begchunk, endchunk, pcols, pver
   107)     use phys_grid, only : get_gcol_all_p, get_ncols_p, chunk_to_block_send_pters, chunk_to_block_recv_pters, &
   108)          transpose_chunk_to_block
   109)     use dyn_grid,       only: get_gcol_block_d
   110)     use dimensions_mod, only: npsq
   111)     use element_mod, only : element_t
   112)     use dof_mod, only : PutUniquePoints
   113)     use interpolate_mod, only : get_interp_parameter
   114)     use shr_pio_mod, only : shr_pio_getiosys
   115)     use edge_mod, only : edgebuffer_t, edgevpack, edgevunpack, initedgebuffer, freeedgebuffer
   116)     use bndry_mod, only : bndry_exchangeV
   117)     use parallel_mod,   only: par
   118)     use abortutils, only : endrun
   119)     
   120)     implicit none
   121)     type(file_desc_t), intent(inout) :: File
   122)     type(var_desc_t), intent(inout) :: varid
   123)     real(r8), intent(in) :: fld(:,:,:)
   124)     integer, intent(in) :: numlev, data_type, decomp_type
   125) 
   126)     type(io_desc_t) :: iodesc
   127) 
   128)     integer :: lchnk, i, j, m, icol, ncols, pgcols(pcols), ierr
   129)     integer :: idmb1(1), idmb2(1), idmb3(1)
   130)     integer :: bpter(npsq,0:pver)    ! offsets into block buffer for packing data
   131)     integer :: cpter(pcols,0:pver)   ! offsets into chunk buffer for unpacking data
   132) 
   133)     real(r8), pointer :: dest(:,:,:,:) 
   134)     real(r8), pointer :: bbuffer(:), cbuffer(:), fldout(:,:)
   135)     real(r8) :: fld_dyn(npsq,numlev,nelemd)
   136)     integer :: st, en, ie, ioff, ncnt_out, k
   137)     integer, pointer :: idof(:)
   138)     integer :: nlon, nlat, ncol
   139)     logical :: usefillvalues=.false.
   140)     type(iosystem_desc_t), pointer :: pio_subsystem
   141)     type (EdgeBuffer_t) :: edgebuf              ! edge buffer
   142) 
   143) 
   144) 
   145)     nlon=get_interp_parameter('nlon')
   146)     nlat=get_interp_parameter('nlat')
   147)     pio_subsystem => shr_pio_getiosys(atm_id)
   148) 
   149)     if(decomp_type==phys_decomp) then
   150)        fld_dyn = -999_R8
   151)        if(local_dp_map) then
   152)           !$omp parallel do private (lchnk, ncols, pgcols, icol, idmb1, idmb2, idmb3, ie, ioff, m)
   153)           do lchnk=begchunk,endchunk
   154)              ncols=get_ncols_p(lchnk)
   155)              call get_gcol_all_p(lchnk,pcols,pgcols)
   156)              do icol=1,ncols
   157)                 call get_gcol_block_d(pgcols(icol),1,idmb1,idmb2,idmb3)
   158)                 ie = idmb3(1)
   159)                 ioff=idmb2(1)
   160)                 do k=1,numlev
   161)                    fld_dyn(ioff,k,ie)      = fld(icol, k, lchnk-begchunk+1)
   162)                 end do
   163)              end do
   164)              
   165)           end do
   166)        else
   167) 
   168)           allocate( bbuffer(block_buf_nrecs*numlev) )
   169)           allocate( cbuffer(chunk_buf_nrecs*numlev) )
   170) 
   171)           !$omp parallel do private (lchnk, ncols, cpter, i, icol)
   172)           do lchnk = begchunk,endchunk
   173)              ncols = get_ncols_p(lchnk)
   174)              
   175)              call chunk_to_block_send_pters(lchnk,pcols,pver+1,1,cpter)
   176)              
   177)              do i=1,ncols
   178)                 cbuffer(cpter(i,1):cpter(i,1)) = 0.0_r8
   179)              end do
   180) 
   181)              do icol=1,ncols
   182)                 
   183)                 cbuffer   (cpter(icol,:))     = fld(icol,:,lchnk-begchunk+1)
   184)              end do
   185) 
   186)           end do
   187) 
   188)           call transpose_chunk_to_block(1, cbuffer, bbuffer)
   189)           if(iam < par%nprocs) then
   190) !$omp parallel do private (ie, bpter, icol)
   191)              do ie=1,nelemd
   192)           
   193)                 call chunk_to_block_recv_pters(elem(ie)%GlobalID,npsq,pver+1,1,bpter)
   194)                 ncols = elem(ie)%idxp%NumUniquePts
   195)                 do icol=1,ncols
   196)                    fld_dyn   (icol,:,ie)   = bbuffer(bpter(icol,:))
   197)                 end do
   198) 
   199)              end do
   200)           end if
   201)           deallocate( bbuffer )
   202)           deallocate( cbuffer )
   203) 
   204)        end if
   205)        allocate(dest(np,np,numlev,nelemd))
   206)        call initEdgeBuffer(edgebuf, numlev)
   207) 
   208)        do ie=1,nelemd
   209)           ncols = elem(ie)%idxp%NumUniquePts
   210)           call putUniquePoints(elem(ie)%idxP, numlev, fld_dyn(1:ncols,:,ie), dest(:,:,:,ie))
   211)           call edgeVpack(edgebuf, dest(:,:,:,ie), numlev, 0, elem(ie)%desc)
   212)        enddo
   213)        if(iam < par%nprocs) then
   214)           call bndry_exchangeV(par, edgebuf)
   215)        end if
   216)        do ie=1,nelemd
   217)           call edgeVunpack(edgebuf, dest(:,:,:,ie), numlev, 0, elem(ie)%desc)
   218)        end do
   219)        call freeEdgeBuffer(edgebuf)
   220)        usefillvalues = any(dest == fillvalue)
   221)     else
   222)         usefillvalues=any(fld==fillvalue)
   223)        allocate(dest(np,np,numlev,1))
   224)     end if
   225) 
   226)     ncnt_out = sum(cam_interpolate(1:nelemd)%n_interp)
   227)     allocate(fldout(ncnt_out,numlev))
   228)     allocate(idof(ncnt_out*numlev))
   229)     fldout = -999_r8
   230)     idof = 0
   231)     st = 1
   232)     
   233)     
   234) 
   235)     do ie=1,nelemd
   236)        ncol = cam_interpolate(ie)%n_interp
   237)        do k=0,numlev-1
   238)           do i=1,ncol
   239)              idof(st+i-1+k*ncnt_out)=cam_interpolate(ie)%ilon(i)+nlon*(cam_interpolate(ie)%ilat(i)-1)+nlon*nlat*k
   240)           enddo
   241)        enddo
   242)        
   243)        ! Now that we have the field on the dyn grid we need to interpolate
   244)        en = st+cam_interpolate(ie)%n_interp-1
   245)        if(decomp_type==phys_decomp) then
   246)           if(usefillvalues) then
   247)              call interpolate_scalar(cam_interpolate(ie),dest(:,:,:,ie), np, numlev, fldout(st:en,:), fillvalue) 
   248)           else
   249)              call interpolate_scalar(cam_interpolate(ie),dest(:,:,:,ie), np, numlev, fldout(st:en,:)) 
   250)           end if
   251)        else
   252)           do j=1,np
   253)              do i=1,np
   254)                 dest(i,j,:,1) = fld(i+(j-1)*np,:,ie)
   255)              end do
   256)           end do
   257)           if(usefillvalues) then
   258)              call interpolate_scalar(cam_interpolate(ie),dest(:,:,:,1), &
   259)                   np, numlev, fldout(st:en,:), fillvalue) 
   260)           else
   261)              call interpolate_scalar(cam_interpolate(ie),dest(:,:,:,1), &
   262)                   np, numlev, fldout(st:en,:)) 
   263)           end if
   264)        end if
   265) 
   266) 
   267)        st = en+1
   268)     end do
   269) 
   270) 
   271)     if(numlev==1) then
   272)        call pio_initdecomp(pio_subsystem, data_type, (/nlon,nlat/), idof, iodesc)
   273)     else
   274)        call pio_initdecomp(pio_subsystem, data_type, (/nlon,nlat,numlev/), idof, iodesc)
   275)     end if
   276)     call pio_write_darray(File, varid, iodesc, fldout, ierr)
   277) 
   278)     deallocate(dest)
   279) 
   280)     deallocate(fldout)
   281)     deallocate(idof)
   282)     call pio_freedecomp(file,iodesc)
   283) 
   284)   end subroutine write_interpolated_scalar
   285)

   289)   subroutine write_interpolated_vector(File, varidu, varidv, fldu, fldv, numlev, data_type, decomp_type)
   290)     use pio, only : file_desc_t, io_desc_t, var_desc_t, pio_write_darray, iosystem_desc_t, &
   291)          pio_initdecomp, pio_freedecomp, pio_setdebuglevel
   292)     use spmd_dyn,       only: local_dp_map, block_buf_nrecs, chunk_buf_nrecs
   293)     use ppgrid, only : begchunk, endchunk, pcols, pver
   294)     use phys_grid, only : get_gcol_all_p, get_ncols_p, chunk_to_block_send_pters, chunk_to_block_recv_pters, &
   295)          transpose_chunk_to_block
   296)     use dyn_grid,       only: get_gcol_block_d
   297)     use dimensions_mod, only: npsq
   298)     use element_mod, only : element_t
   299)     use dof_mod, only : PutUniquePoints
   300)     use interpolate_mod, only : get_interp_parameter
   301)     use shr_pio_mod, only : shr_pio_getiosys
   302)     use edge_mod, only : edgebuffer_t, edgevpack, edgevunpack, initedgebuffer, freeedgebuffer
   303)     use bndry_mod, only : bndry_exchangeV
   304)     use parallel_mod,   only: par
   305)     implicit none
   306)     type(file_desc_t), intent(inout) :: File
   307)     type(var_desc_t), intent(inout) :: varidu, varidv
   308)     real(r8), intent(in) :: fldu(:,:,:), fldv(:,:,:)
   309)     integer, intent(in) :: numlev, data_type, decomp_type
   310) 
   311)     type(io_desc_t) :: iodesc
   312) 
   313)     integer :: lchnk, i, j, m, icol, ncols, pgcols(pcols), ierr
   314)     integer :: idmb1(1), idmb2(1), idmb3(1)
   315)     integer :: bpter(npsq,0:pver)    ! offsets into block buffer for packing data
   316)     integer :: cpter(pcols,0:pver)   ! offsets into chunk buffer for unpacking data
   317) 
   318)     real(r8), allocatable :: dest(:,:,:,:,:)
   319)     real(r8), pointer :: bbuffer(:), cbuffer(:), fldout(:,:,:)
   320)     real(r8) :: fld_dyn(npsq,2,numlev,nelemd)
   321)     integer :: st, en, ie, ioff, ncnt_out, k
   322)     integer, pointer :: idof(:)
   323)     integer :: nlon, nlat, ncol
   324)     logical :: usefillvalues=.false.
   325) 
   326)     type(iosystem_desc_t), pointer :: pio_subsystem
   327)     type (EdgeBuffer_t) :: edgebuf              ! edge buffer
   328) 
   329) 
   330) 
   331)     nlon=get_interp_parameter('nlon')
   332)     nlat=get_interp_parameter('nlat')
   333)     pio_subsystem => shr_pio_getiosys('ATM')
   334)     fld_dyn = -999_R8
   335)     if(decomp_type==phys_decomp) then
   336)        allocate(dest(np,np,2,numlev,nelemd))
   337)        if(local_dp_map) then
   338)           !$omp parallel do private (lchnk, ncols, pgcols, icol, idmb1, idmb2, idmb3, ie, ioff, m)
   339)           do lchnk=begchunk,endchunk
   340)              ncols=get_ncols_p(lchnk)
   341)              call get_gcol_all_p(lchnk,pcols,pgcols)
   342)              
   343)              do icol=1,ncols
   344)                 call get_gcol_block_d(pgcols(icol),1,idmb1,idmb2,idmb3)
   345)                 ie = idmb3(1)
   346)                 ioff=idmb2(1)
   347)                 do k=1,numlev
   348)                    fld_dyn(ioff,1,k,ie)      = fldu(icol, k, lchnk-begchunk+1)
   349)                    fld_dyn(ioff,2,k,ie)      = fldv(icol, k, lchnk-begchunk+1)
   350)                 end do
   351)              end do
   352)              
   353)           end do
   354)        else
   355) 
   356)           allocate( bbuffer(2*block_buf_nrecs*numlev) )
   357)           allocate( cbuffer(2*chunk_buf_nrecs*numlev) )
   358) 
   359)           !$omp parallel do private (lchnk, ncols, cpter, i, icol)
   360)           do lchnk = begchunk,endchunk
   361)              ncols = get_ncols_p(lchnk)
   362)              
   363)              call chunk_to_block_send_pters(lchnk,pcols,pver+1,2,cpter)
   364)              
   365)              do i=1,ncols
   366)                 cbuffer(cpter(i,1):cpter(i,1)) = 0.0_r8
   367)              end do
   368) 
   369)              do icol=1,ncols
   370)                 do k=1,numlev
   371)                    cbuffer   (cpter(icol,k))     = fldu(icol,k,lchnk-begchunk+1)
   372)                    cbuffer   (cpter(icol,k)+1)   = fldv(icol,k,lchnk-begchunk+1)
   373)                 end do
   374)              end do
   375) 
   376)           end do
   377) 
   378)           call transpose_chunk_to_block(2, cbuffer, bbuffer)
   379)           if(iam < par%nprocs) then
   380)              !$omp parallel do private (ie, bpter, icol)
   381)              do ie=1,nelemd
   382)                 
   383)                 call chunk_to_block_recv_pters(elem(ie)%GlobalID,npsq,pver+1,2,bpter)
   384)                 ncols = elem(ie)%idxp%NumUniquePts
   385)                 do icol=1,ncols
   386)                    do k=1,numlev
   387)                       fld_dyn   (icol,1,k,ie)   = bbuffer(bpter(icol,k))
   388)                       fld_dyn   (icol,2,k,ie)   = bbuffer(bpter(icol,k)+1)
   389)                    enddo
   390)                 end do
   391)                 
   392)              end do
   393)           end if
   394)           deallocate( bbuffer )
   395)           deallocate( cbuffer )
   396) 
   397)        end if
   398)        call initEdgeBuffer(edgebuf, 2*numlev)
   399) 
   400)        do ie=1,nelemd
   401)           ncols = elem(ie)%idxp%NumUniquePts
   402)           call putUniquePoints(elem(ie)%idxP, 2, numlev, fld_dyn(1:ncols,:,:,ie), dest(:,:,:,:,ie))
   403)           
   404)           call edgeVpack(edgebuf, dest(:,:,:,:,ie), 2*numlev, 0, elem(ie)%desc)
   405)        enddo
   406)        if(iam < par%nprocs) then
   407)           call bndry_exchangeV(par, edgebuf)
   408)        end if
   409) 
   410)        do ie=1,nelemd
   411)           call edgeVunpack(edgebuf, dest(:,:,:,:,ie), 2*numlev, 0, elem(ie)%desc)
   412)        enddo
   413)        call freeEdgeBuffer(edgebuf)
   414)        usefillvalues = any(dest==fillvalue)
   415)     else
   416)        usefillvalues = (any(fldu==fillvalue) .or. any(fldv==fillvalue))
   417)        allocate(dest(np,np,2,numlev,1))
   418)     endif
   419)     ncnt_out = sum(cam_interpolate(1:nelemd)%n_interp)
   420)     allocate(fldout(ncnt_out,numlev,2))
   421)     allocate(idof(ncnt_out*numlev))
   422)     
   423)     fldout = -999_r8
   424)     idof = 0
   425)     st = 1
   426)     do ie=1,nelemd
   427)        ncol = cam_interpolate(ie)%n_interp
   428)        do k=0,numlev-1
   429)           do i=1,ncol
   430)              idof(st+i-1+k*ncnt_out)=cam_interpolate(ie)%ilon(i)+nlon*(cam_interpolate(ie)%ilat(i)-1)+nlon*nlat*k
   431)           enddo
   432)        enddo
   433)        
   434) 
   435)     ! Now that we have the field on the dyn grid we need to interpolate
   436)        en = st+cam_interpolate(ie)%n_interp-1
   437)        if(decomp_type==phys_decomp) then
   438)           if(usefillvalues) then
   439)              call interpolate_vector(cam_interpolate(ie),elem(ie), &
   440)                   dest(:,:,:,:,ie), np, numlev, fldout(st:en,:,:), 0, fillvalue) 
   441)           else
   442)              call interpolate_vector(cam_interpolate(ie),elem(ie),&
   443)                   dest(:,:,:,:,ie), np, numlev, fldout(st:en,:,:), 0) 
   444)           endif
   445)        else
   446)           do k=1,numlev
   447)              do j=1,np
   448)                 do i=1,np
   449)                    dest(i,j,1,k,1) = fldu(i+(j-1)*np,k,ie)
   450)                    dest(i,j,1,k,1) = fldv(i+(j-1)*np,k,ie)
   451)                 end do
   452)              end do
   453)           end do
   454)           if(usefillvalues) then
   455)              call interpolate_vector(cam_interpolate(ie),elem(ie),&
   456)                   dest(:,:,:,:,1), np, numlev, fldout(st:en,:,:), 0, fillvalue) 
   457)           else
   458)              call interpolate_vector(cam_interpolate(ie),elem(ie),&
   459)                   dest(:,:,:,:,1), np, numlev, fldout(st:en,:,:), 0) 
   460)           end if
   461)        end if
   462) 
   463)        st = en+1
   464)     end do
   465) 
   466)     if(numlev==1) then
   467)        call pio_initdecomp(pio_subsystem, data_type, (/nlon,nlat/), idof, iodesc)
   468)     else
   469)        call pio_initdecomp(pio_subsystem, data_type, (/nlon,nlat,numlev/), idof, iodesc)
   470)     end if
   471) 
   472)     call pio_write_darray(File, varidu, iodesc, fldout(:,:,1), ierr)
   473) 
   474)     call pio_write_darray(File, varidv, iodesc, fldout(:,:,2), ierr)
   475) 
   476) 
   477)     deallocate(fldout)
