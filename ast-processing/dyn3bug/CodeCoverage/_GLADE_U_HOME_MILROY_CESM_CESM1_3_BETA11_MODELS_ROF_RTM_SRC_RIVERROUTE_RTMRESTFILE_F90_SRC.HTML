   178)   subroutine RtmRestGetfile( file, path )
   179) 
   180)     !---------------------------------------------------
   181)     ! DESCRIPTION:
   182)     ! Determine and obtain netcdf restart file
   183) 
   184)     ! ARGUMENTS:
   185)     implicit none
   186)     character(len=*), intent(out) :: file  ! name of netcdf restart file
   187)     character(len=*), intent(out) :: path  ! full pathname of netcdf restart file
   188) 
   189)     ! LOCAL VARIABLES:
   190)     integer :: status                      ! return status
   191)     integer :: length                      ! temporary          
   192)     character(len=256) :: ftest,ctest      ! temporaries
   193)     !---------------------------------------------------
   194) 
   195)     ! Continue run:
   196)     ! Restart file pathname is read restart pointer file 
   197)     if (nsrest==nsrContinue) then
   198)        call restFile_read_pfile( path )
   199)        call getfil( path, file, 0 )
   200)     end if
   201)        
   202)     ! Branch run: 
   203)     ! Restart file pathname is obtained from namelist "nrevsn_rtm"
   204)     if (nsrest==nsrBranch) then
   205)        length = len_trim(nrevsn_rtm)
   206)        if (nrevsn_rtm(length-2:length) == '.nc') then
   207)           path = trim(nrevsn_rtm) 
   208)        else
   209)           path = trim(nrevsn_rtm) // '.nc'
   210)        end if
   211)        call getfil( path, file, 0 )
   212)        
   213)        ! Check case name consistency (case name must be different 
   214)        ! for branch run, unless brnch_retain_casename is set)
   215)        ctest = 'xx.'//trim(caseid)//'.rtm'
   216)        ftest = 'xx.'//trim(file)
   217)        status = index(trim(ftest),trim(ctest))
   218)        if (status /= 0 .and. .not.(brnch_retain_casename)) then
   219)           write(iulog,*) 'Must change case name on branch run if ',&
   220)                'brnch_retain_casename namelist is not set'
   221)           write(iulog,*) 'previous case filename= ',trim(file),&
   222)                ' current case = ',trim(caseid), ' ctest = ',trim(ctest), &
   223)                ' ftest = ',trim(ftest)
   224)           call shr_sys_abort()
   225)        end if
   226)     end if
   227) 
   228)     ! Initial run 
   229)     if (nsrest==nsrStartup) then
   230)        call getfil( finidat_rtm, file, 0 )
   231)     end if
   232)     
   233)   end subroutine RtmRestGetfile

   238) 
   239)     ! !DESCRIPTION:
   240)     ! Setup restart file and perform necessary consistency checks
   241) 
   242)     ! !ARGUMENTS:
   243)     implicit none

   245) 
   246)     ! !LOCAL VARIABLES:
   247)     integer :: i                  ! indices
   248)     integer :: nio                ! restart unit
   249)     integer :: status             ! substring check status
   250)     character(len=256) :: locfn   ! Restart pointer file name
   251)     !--------------------------------------------------------
   252) 
   253)     ! Obtain the restart file from the restart pointer file. 
   254)     ! For restart runs, the restart pointer file contains the full pathname 
   255)     ! of the restart file. For branch runs, the namelist variable 
   256)     ! [nrevsn_rtm] contains the full pathname of the restart file. 
   257)     ! New history files are always created for branch runs.
   258)        
   259)     if (masterproc) then
   260)        write(iulog,*) 'Reading restart pointer file....'
   261)     endif
   262) 
   263)     nio = getavu()
   264)     locfn = './'// trim(rpntfil)//trim(inst_suffix)
   265)     call opnfil (locfn, nio, 'f')
   266)     read (nio,'(a256)') pnamer
   267)     call relavu (nio)
   268) 
   269)     if (masterproc) then
   270)        write(iulog,*) 'Reading restart data.....'
   271)        write(iulog,'(72a1)') ("-",i=1,60)

   367)   subroutine RtmRestart(ncid, flag)
   368) 
   369)     !-----------------------------------------------------------------------
   370)     ! DESCRIPTION:
   371)     ! Read/write RTM restart data.
   372)     !
   373)     ! ARGUMENTS:
   374)     implicit none
   375)     type(file_desc_t), intent(inout)  :: ncid ! netcdf id
   376)     character(len=*) , intent(in) :: flag   ! 'read' or 'write'
   377)     ! LOCAL VARIABLES:
   378)     logical :: readvar          ! determine if variable is on initial file
   379)     integer :: nt,nv,n          ! indices
   380)     real(r8) , pointer :: dfld(:) ! temporary array
   381)     character(len=32)  :: vname,uname
   382)     character(len=128) :: lname
   383)     !-----------------------------------------------------------------------
   384) 
   385)     do nv = 1,4
   386)     do nt = 1,nt_rtm
   387) 
   388)        if (nv == 1) then
   389)           vname = 'RTM_VOLR_'//trim(rtm_tracers(nt))
   390)           lname = 'water volume in cell (volr)'
   391)           uname = 'm3'
   392)           dfld  => runoff%volr(:,nt)
   393)        elseif (nv == 2) then
   394)           vname = 'RTM_FLUXOUT_'//trim(rtm_tracers(nt))
   395)           lname = 'water fluxout in cell (fluxout)'
   396)           uname = 'm3/s'
   397)           dfld  => runoff%fluxout(:,nt)
   398)        elseif (nv == 3) then
   399)           vname = 'RTM_RUNOFF_'//trim(rtm_tracers(nt))
   400)           lname = 'runoff (runoff)'
   401)           uname = 'm3/s'
   402)           dfld  => runoff%runoff(:,nt)
   403)        elseif (nv == 4) then
   404)           vname = 'RTM_DVOLRDT_'//trim(rtm_tracers(nt))
   405)           lname = 'water volume change in cell (dvolrdt)'
   406)           uname = 'mm/s'
   407)           dfld  => runoff%dvolrdt(:,nt)
   408)        else
   409)           write(iulog,*) 'Rtm ERROR: illegal nv value a ',nv
   410)           call shr_sys_abort()
   411)        endif
   412) 
   413)        if (flag == 'define') then
   414)           call ncd_defvar(ncid=ncid, varname=trim(vname), &
   415)                xtype=ncd_double,  dim1name='rtmlon', dim2name='rtmlat', &
   416)                long_name=trim(lname), units=trim(uname))
   417)        else if (flag == 'read' .or. flag == 'write') then
   418)           call ncd_io(varname=trim(vname), data=dfld, dim1name='allrof', &
   419)                ncid=ncid, flag=flag, readvar=readvar)
   420)           if (flag=='read' .and. .not. readvar) then
   421)              if (nsrest == nsrContinue) then
   422)                 call shr_sys_abort()
   423)              else
   424)                 dfld = 0._r8
   425)              end if
   426)           end if
   427)        end if
   428) 
   429)     enddo
   430)     enddo
   431) 
   432)     if (flag == 'read') then
   433)        do n = runoff%begr,runoff%endr
   434)           do nt = 1,nt_rtm
   435)              if (abs(runoff%volr(n,nt))    > 1.e30) runoff%volr(n,nt) = 0.
   436)              if (abs(runoff%runoff(n,nt))  > 1.e30) runoff%runoff(n,nt) = 0.
   437)              if (abs(runoff%dvolrdt(n,nt)) > 1.e30) runoff%dvolrdt(n,nt) = 0.
   438)              if (abs(runoff%fluxout(n,nt)) > 1.e30) runoff%fluxout(n,nt) = 0.
   439)           end do
   440)           if (runoff%mask(n) == 1) then
   441)              do nt = 1,nt_rtm
   442)                 runoff%runofflnd(n,nt) = runoff%runoff(n,nt)
   443)                 runoff%dvolrdtlnd(n,nt)= runoff%dvolrdt(n,nt)
   444)                 runoff%volrlnd(n,nt)   = runoff%volr(n,nt)
   445)              end do
   446)           elseif (runoff%mask(n) == 2) then
   447)              do nt = 1,nt_rtm
   448)                 runoff%runoffocn(n,nt) = runoff%runoff(n,nt)
   449)                 runoff%dvolrdtocn(n,nt)= runoff%dvolrdt(n,nt)
   450)              enddo
   451)           endif
   452)        enddo
   453)     endif
   454) 
   455)   end subroutine RtmRestart
