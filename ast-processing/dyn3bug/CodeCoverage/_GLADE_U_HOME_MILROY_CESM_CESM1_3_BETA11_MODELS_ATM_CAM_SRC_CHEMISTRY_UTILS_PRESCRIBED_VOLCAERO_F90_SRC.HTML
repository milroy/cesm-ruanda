   203)   subroutine prescribed_volcaero_adv( state, pbuf2d)
   204) 
   205)     use tracer_data,  only : advance_trcdata
   206)     use physics_types,only : physics_state
   207)     use ppgrid,       only : begchunk, endchunk
   208)     use ppgrid,       only : pcols, pver
   209)     use string_utils, only : to_lower, GLC
   210)     use cam_history,  only : outfld
   211)     use physconst,    only : mwdry                ! molecular weight dry air ~ kg/kmole
   212)     use physconst,    only : boltz, gravit        ! J/K/molecule
   213)     use tropopause,   only : tropopause_find, TROP_ALG_TWMO, TROP_ALG_CLIMATE
   214)     
   215)     use physics_buffer, only : physics_buffer_desc, pbuf_get_field, pbuf_get_chunk
   216)     
   217)     implicit none
   218) 
   219)     type(physics_state), intent(in)    :: state(begchunk:endchunk)                 
   220)     
   221)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   222) 
   223)     type(physics_buffer_desc), pointer :: pbuf_chnk(:)
   224) 
   225)     integer :: c,ncol,i,k
   226)     real(r8) :: to_mmr(pcols,pver)
   227)     real(r8), parameter :: molmass = 47.9981995_r8
   228)     real(r8) :: ptrop
   229)     real(r8) :: concvolc ! micrograms of wetted aerosol per cubic centimeter
   230)     real(r8) :: volcmass(pcols,pver)
   231)     real(r8) :: columnmass(pcols)
   232)     real(r8) :: mmrvolc
   233)     integer  :: tropLev(pcols)
   234) 
   235)     real(r8) :: outdata(pcols,pver)
   236)     real(r8), pointer :: data(:,:)
   237)     real(r8), pointer :: radius(:,:)
   238) 
   239)     !WACCM-derived relation between mass concentration and wet aerosol radius in meters
   240)     real(r8),parameter :: radius_conversion = 1.9e-4_r8
   241) 
   242)     if( .not. has_prescribed_volcaero ) return
   243) 
   244)     call advance_trcdata( fields, file, state, pbuf2d )
   245) 
   246)     ! copy prescribed tracer fields into state svariable with the correct units
   247)     do c = begchunk,endchunk
   248)        pbuf_chnk => pbuf_get_chunk(pbuf2d, c)
   249)        call pbuf_get_field(pbuf_chnk, radius_ndx, radius)
   250)        radius(:,:) = 0._r8
   251)        ncol = state(c)%ncol
   252)        select case ( to_lower(trim(fields(1)%units(:GLC(fields(1)%units)))) )
   253)        case ("molec/cm3","/cm3","molecules/cm3","cm^-3","cm**-3")
   254)           to_mmr(:ncol,:) = (molmass*1.e6_r8*boltz*state(c)%t(:ncol,:))/(mwdry*state(c)%pmiddry(:ncol,:))
   255)        case ('kg/kg','mmr','kg kg-1')
   256)           to_mmr(:ncol,:) = 1._r8
   257)        case ('mol/mol','mole/mole','vmr','fraction')
   258)           to_mmr(:ncol,:) = molmass/mwdry
   259)        case default
   260)           write(iulog,*) 'prescribed_volcaero_adv: units = ',trim(fields(1)%units) ,' are not recognized'
   261)           call endrun('prescribed_volcaero_adv: units are not recognized')
   262)        end select
   263) 
   264)        call pbuf_get_field(pbuf_chnk, fields(1)%pbuf_ndx, data)
   265)        data(:ncol,:) = to_mmr(:ncol,:) * data(:ncol,:) ! mmr
   266) 
   267)        call tropopause_find(state(c), tropLev, primary=TROP_ALG_TWMO, backup=TROP_ALG_CLIMATE)
   268)        do i = 1,ncol
   269)           do k = 1,pver
   270)              ! set to zero below tropopause
   271)              if ( k >= tropLev(i) ) then
   272)                 data(i,k) = 0._r8
   273)              endif
   274)              mmrvolc = data(i,k)
   275)              if (mmrvolc > 0._r8) then
   276)                 concvolc = (mmrvolc * state(c)%pdel(i,k))/(gravit * state(c)%zm(i,k))
   277)                 radius(i,k) = radius_conversion*(concvolc**(1._r8/3._r8))
   278)              endif
   279)           enddo
   280)        enddo
   281) 
   282)        volcmass(:ncol,:) = data(:ncol,:)*state(c)%pdel(:ncol,:)/gravit
   283)        columnmass(:ncol) = sum(volcmass(:ncol,:), 2)
   284) 
   285)        call outfld( volcaero_name,        data(:,:),     pcols, state(c)%lchnk)
   286)        call outfld( volcrad_name,         radius(:,:),   pcols, state(c)%lchnk)
   287)        call outfld( volcmass_name,        volcmass(:,:), pcols, state(c)%lchnk)
   288)        call outfld( volcmass_column_name, columnmass(:), pcols, state(c)%lchnk)
   289) 
   290)     enddo
   291) 
   292)   end subroutine prescribed_volcaero_adv
