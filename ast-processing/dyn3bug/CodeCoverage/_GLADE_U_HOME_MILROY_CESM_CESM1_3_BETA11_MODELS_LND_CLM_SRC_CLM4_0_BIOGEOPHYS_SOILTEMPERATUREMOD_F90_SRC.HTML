    39) ! !DESCRIPTION:
    40) ! Snow and soil temperatures including phase change
    41) ! o The volumetric heat capacity is calculated as a linear combination
    42) !   in terms of the volumetric fraction of the constituent phases.
    43) ! o The thermal conductivity of soil is computed from
    44) !   the algorithm of Johansen (as reported by Farouki 1981), and the
    45) !   conductivity of snow is from the formulation used in
    46) !   SNTHERM (Jordan 1991).
    47) ! o Boundary conditions:
    48) !   F = Rnet - Hg - LEg (top),  F= 0 (base of the soil column).
    49) ! o Soil / snow temperature is predicted from heat conduction
    50) !   in 10 soil layers and up to 5 snow layers.
    51) !   The thermal conductivities at the interfaces between two
    52) !   neighboring layers (j, j+1) are derived from an assumption that
    53) !   the flux across the interface is equal to that from the node j
    54) !   to the interface and the flux from the interface to the node j+1.
    55) !   The equation is solved using the Crank-Nicholson method and
    56) !   results in a tridiagonal system equation.
    57) !
    58) ! !USES:

    68)     use TridiagonalMod, only : Tridiagonal
    69) 
    70) !
    71) ! !ARGUMENTS:
    72)     implicit none
    73)     integer , intent(in)  :: lbc, ubc                    ! column bounds
    74)     integer , intent(in)  :: num_nolakec                 ! number of column non-lake points in column filter
    75)     integer , intent(in)  :: filter_nolakec(ubc-lbc+1)   ! column filter for non-lake points
    76)     integer , intent(in)  :: lbl, ubl                    ! landunit-index bounds
    77)     integer , intent(in)  :: num_urbanl                  ! number of urban landunits in clump
    78)     integer , intent(in)  :: filter_urbanl(ubl-lbl+1)    ! urban landunit filter
    79)     real(r8), intent(out) :: xmf(lbc:ubc)                ! total latent heat of phase change of ground water
    80)     real(r8), intent(out) :: fact(lbc:ubc, -nlevsno+1:nlevgrnd) ! used in computing tridiagonal matrix
    81) !

    83) ! subroutine Biogeophysics2 in module Biogeophysics2Mod
    84) !
    85) ! !REVISION HISTORY:
    86) ! 15 September 1999: Yongjiu Dai; Initial code
    87) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
    88) ! 12/19/01, Peter Thornton

    92) ! in calculation of net ground heat flux.

    94) ! 03/28/08, Mark Flanner: Changes to allow solar radiative absorption in all snow layers and top soil layer
    95) ! !LOCAL VARIABLES:
    96) !
    97) ! local pointers to original implicit in arguments
    98) !
    99)     integer , pointer :: pgridcell(:)       ! pft's gridcell index
   100)     integer , pointer :: plandunit(:)       ! pft's landunit index
   101)     integer , pointer :: clandunit(:)       ! column's landunit
   102)     integer , pointer :: ltype(:)           ! landunit type
   103)     integer , pointer :: ctype(:)           ! column type
   104)     integer , pointer :: npfts(:)           ! column's number of pfts 
   105)     integer , pointer :: pfti(:)            ! column's beginning pft index 
   106)     real(r8), pointer :: pwtcol(:)          ! weight of pft relative to column
   107)     real(r8), pointer :: pwtgcell(:)        ! weight of pft relative to corresponding gridcell
   108)     real(r8), pointer :: forc_lwrad(:)      ! downward infrared (longwave) radiation (W/m**2)
   109)     integer , pointer :: snl(:)             ! number of snow layers
   110)     real(r8), pointer :: htvp(:)            ! latent heat of vapor of water (or sublimation) [j/kg]
   111)     real(r8), pointer :: emg(:)             ! ground emissivity
   112)     real(r8), pointer :: cgrnd(:)           ! deriv. of soil energy flux wrt to soil temp [w/m2/k]
   113)     real(r8), pointer :: dlrad(:)           ! downward longwave radiation blow the canopy [W/m2]
   114)     real(r8), pointer :: sabg(:)            ! solar radiation absorbed by ground (W/m**2)
   115)     integer , pointer :: frac_veg_nosno(:)  ! fraction of vegetation not covered by snow (0 OR 1 now) [-] (new)
   116)     real(r8), pointer :: eflx_sh_grnd(:)    ! sensible heat flux from ground (W/m**2) [+ to atm]
   117)     real(r8), pointer :: qflx_evap_soi(:)   ! soil evaporation (mm H2O/s) (+ = to atm)
   118)     real(r8), pointer :: qflx_tran_veg(:)   ! vegetation transpiration (mm H2O/s) (+ = to atm)
   119)     real(r8), pointer :: zi(:,:)            ! interface level below a "z" level (m)
   120)     real(r8), pointer :: dz(:,:)            ! layer depth (m)
   121)     real(r8), pointer :: z(:,:)             ! layer thickness (m)
   122)     real(r8), pointer :: t_soisno(:,:)      ! soil temperature (Kelvin)
   123)     real(r8), pointer :: eflx_lwrad_net(:)  ! net infrared (longwave) rad (W/m**2) [+ = to atm]
   124)     real(r8), pointer :: tssbef(:,:)        ! temperature at previous time step [K]
   125)     real(r8), pointer :: t_building(:)      ! internal building temperature (K)
   126)     real(r8), pointer :: t_building_max(:)  ! maximum internal building temperature (K)
   127)     real(r8), pointer :: t_building_min(:)  ! minimum internal building temperature (K)
   128)     real(r8), pointer :: hc_soi(:)          ! soil heat content (MJ/m2)
   129)     real(r8), pointer :: hc_soisno(:)       ! soil plus snow plus lake heat content (MJ/m2)
   130)     real(r8), pointer :: eflx_fgr12(:)      ! heat flux between soil layer 1 and 2 (W/m2)
   131)     real(r8), pointer :: eflx_traffic(:)    ! traffic sensible heat flux (W/m**2)
   132)     real(r8), pointer :: eflx_wasteheat(:)  ! sensible heat flux from urban heating/cooling sources of waste heat (W/m**2)
   133)     real(r8), pointer :: eflx_wasteheat_pft(:)  ! sensible heat flux from urban heating/cooling sources of waste heat (W/m**2)
   134)     real(r8), pointer :: eflx_heat_from_ac(:) !sensible heat flux put back into canyon due to removal by AC (W/m**2)
   135)     real(r8), pointer :: eflx_heat_from_ac_pft(:) !sensible heat flux put back into canyon due to removal by AC (W/m**2)
   136)     real(r8), pointer :: eflx_traffic_pft(:)    ! traffic sensible heat flux (W/m**2)
   137)     real(r8), pointer :: eflx_anthro(:)         ! total anthropogenic heat flux (W/m**2)
   138)     real(r8), pointer :: canyon_hwr(:)      ! urban canyon height to width ratio
   139)     real(r8), pointer :: wtlunit_roof(:)    ! weight of roof with respect to landunit
   140)     real(r8), pointer :: eflx_bot(:)        ! heat flux from beneath column (W/m**2) [+ = upward]
   141) ! 
   142) ! local pointers to  original implicit inout arguments
   143) !
   144)     real(r8), pointer :: t_grnd(:)          ! ground surface temperature [K]
   145) !
   146) ! local pointers to original implicit out arguments
   147) !
   148)     real(r8), pointer :: eflx_gnet(:)          ! net ground heat flux into the surface (W/m**2)
   149)     real(r8), pointer :: dgnetdT(:)            ! temperature derivative of ground net heat flux
   150)     real(r8), pointer :: eflx_building_heat(:) ! heat flux from urban building interior to walls, roof (W/m**2)
   151) 
   152) ! variables needed for SNICAR
   153)     real(r8), pointer :: sabg_lyr(:,:)      ! absorbed solar radiation (pft,lyr) [W/m2]
   154)     real(r8), pointer :: h2osno(:)          ! total snow water (col) [kg/m2]
   155)     real(r8), pointer :: h2osoi_liq(:,:)    ! liquid water (col,lyr) [kg/m2]
   156)     real(r8), pointer :: h2osoi_ice(:,:)    ! ice content (col,lyr) [kg/m2]
   157) 
   158) ! Urban building HAC fluxes
   159)     real(r8), pointer :: eflx_urban_ac(:)      ! urban air conditioning flux (W/m**2)
   160)     real(r8), pointer :: eflx_urban_heat(:)    ! urban heating flux (W/m**2)
   161) !
   162) !
   163) ! !OTHER LOCAL VARIABLES:
   164) !EOP
   165) !
   166)     integer  :: j,c,p,l,g,pi                       !  indices
   167)     integer  :: fc                                 ! lake filtered column indices
   168)     integer  :: fl                                 ! urban filtered landunit indices
   169)     integer  :: jtop(lbc:ubc)                      ! top level at each column
   170)     real(r8) :: dtime                              ! land model time step (sec)
   171)     real(r8) :: at (lbc:ubc,-nlevsno+1:nlevgrnd)   ! "a" vector for tridiagonal matrix
   172)     real(r8) :: bt (lbc:ubc,-nlevsno+1:nlevgrnd)   ! "b" vector for tridiagonal matrix
   173)     real(r8) :: ct (lbc:ubc,-nlevsno+1:nlevgrnd)   ! "c" vector for tridiagonal matrix
   174)     real(r8) :: rt (lbc:ubc,-nlevsno+1:nlevgrnd)   ! "r" vector for tridiagonal solution
   175)     real(r8) :: cv (lbc:ubc,-nlevsno+1:nlevgrnd)   ! heat capacity [J/(m2 K)]
   176)     real(r8) :: tk (lbc:ubc,-nlevsno+1:nlevgrnd)   ! thermal conductivity [W/(m K)]
   177)     real(r8) :: fn (lbc:ubc,-nlevsno+1:nlevgrnd)   ! heat diffusion through the layer interface [W/m2]
   178)     real(r8) :: fn1(lbc:ubc,-nlevsno+1:nlevgrnd)   ! heat diffusion through the layer interface [W/m2]
   179)     real(r8) :: brr(lbc:ubc,-nlevsno+1:nlevgrnd)   ! temporary
   180)     real(r8) :: dzm                                ! used in computing tridiagonal matrix
   181)     real(r8) :: dzp                                ! used in computing tridiagonal matrix
   182)     real(r8) :: hs(lbc:ubc)                        ! net energy flux into the surface (w/m2)
   183)     real(r8) :: dhsdT(lbc:ubc)                     ! d(hs)/dT
   184)     real(r8) :: lwrad_emit(lbc:ubc)                ! emitted longwave radiation
   185)     real(r8) :: dlwrad_emit(lbc:ubc)               ! time derivative of emitted longwave radiation
   186)     integer  :: lyr_top                            ! index of top layer of snowpack (-4 to 0) [idx]
   187)     real(r8) :: sabg_lyr_col(lbc:ubc,-nlevsno+1:1) ! absorbed solar radiation (col,lyr) [W/m2]
   188)     real(r8) :: eflx_gnet_top                      ! net energy flux into surface layer, pft-level [W/m2]
   189)     real(r8) :: hs_top(lbc:ubc)                    ! net energy flux into surface layer (col) [W/m2]
   190)     logical  :: cool_on(lbl:ubl)                   ! is urban air conditioning on?
   191)     logical  :: heat_on(lbl:ubl)                   ! is urban heating on?
   192) !-----------------------------------------------------------------------
   193) 
   194)     ! Assign local pointers to derived subtypes components (gridcell-level)
   195) 
   196)     forc_lwrad     => clm_a2l%forc_lwrad
   197) 
   198)     ! Assign local pointers to derived subtypes components (landunit-level)
   199) 
   200)     ltype          => lun%itype
   201)     t_building     => lps%t_building
   202)     t_building_max => lps%t_building_max
   203)     t_building_min => lps%t_building_min
   204)     eflx_traffic   => lef%eflx_traffic
   205)     canyon_hwr     => lun%canyon_hwr
   206)     eflx_wasteheat => lef%eflx_wasteheat
   207)     eflx_heat_from_ac => lef%eflx_heat_from_ac
   208)     wtlunit_roof   => lun%wtlunit_roof
   209) 
   210)     ! Assign local pointers to derived subtypes components (column-level)
   211) 
   212)     ctype          => col%itype
   213)     clandunit      => col%landunit
   214)     npfts          => col%npfts
   215)     pfti           => col%pfti
   216)     snl            => cps%snl
   217)     htvp           => cps%htvp
   218)     emg            => cps%emg
   219)     t_grnd         => ces%t_grnd
   220)     hc_soi         => ces%hc_soi
   221)     hc_soisno      => ces%hc_soisno
   222)     eflx_fgr12     => cef%eflx_fgr12
   223)     zi             => cps%zi
   224)     dz             => cps%dz
   225)     z              => cps%z
   226)     t_soisno       => ces%t_soisno
   227)     eflx_building_heat => cef%eflx_building_heat
   228)     tssbef             => ces%tssbef
   229)     eflx_urban_ac      => cef%eflx_urban_ac
   230)     eflx_urban_heat    => cef%eflx_urban_heat
   231)     eflx_bot           => cef%eflx_bot
   232) 
   233)     ! Assign local pointers to derived subtypes components (pft-level)
   234) 
   235)     pgridcell      => pft%gridcell
   236)     plandunit      => pft%landunit
   237)     pwtcol         => pft%wtcol
   238)     pwtgcell       => pft%wtgcell  
   239)     frac_veg_nosno => pps%frac_veg_nosno
   240)     cgrnd          => pef%cgrnd
   241)     dlrad          => pef%dlrad
   242)     sabg           => pef%sabg
   243)     eflx_sh_grnd   => pef%eflx_sh_grnd
   244)     qflx_evap_soi  => pwf%qflx_evap_soi
   245)     qflx_tran_veg  => pwf%qflx_tran_veg
   246)     eflx_gnet      => pef%eflx_gnet
   247)     dgnetdT        => pef%dgnetdT
   248)     eflx_lwrad_net => pef%eflx_lwrad_net
   249)     eflx_wasteheat_pft => pef%eflx_wasteheat_pft
   250)     eflx_heat_from_ac_pft => pef%eflx_heat_from_ac_pft
   251)     eflx_traffic_pft => pef%eflx_traffic_pft
   252)     eflx_anthro => pef%eflx_anthro
   253) 
   254)     sabg_lyr       => pef%sabg_lyr
   255)     h2osno         => cws%h2osno
   256)     h2osoi_liq     => cws%h2osoi_liq
   257)     h2osoi_ice     => cws%h2osoi_ice
   258) 
   259)     ! Get step size
   260) 
   261)     dtime = get_step_size()
   262) 
   263)     ! Compute ground surface and soil temperatures
   264) 
   265)     ! Thermal conductivity and Heat capacity
   266) 
   267)     call SoilThermProp(lbc, ubc, num_nolakec, filter_nolakec, tk, cv)
   268) 
   269)     ! Net ground heat flux into the surface and its temperature derivative
   270)     ! Added a pfts loop here to get the average of hs and dhsdT over 
   271)     ! all PFTs on the column. Precalculate the terms that do not depend on PFT.
   272) 
   273)     do fc = 1,num_nolakec
   274)        c = filter_nolakec(fc)
   275)        lwrad_emit(c)  =    emg(c) * sb * t_grnd(c)**4
   276)        dlwrad_emit(c) = 4._r8*emg(c) * sb * t_grnd(c)**3
   277)     end do
   278) 
   279)     hs(lbc:ubc) = 0._r8
   280)     dhsdT(lbc:ubc) = 0._r8
   281)     do pi = 1,max_pft_per_col
   282)        do fc = 1,num_nolakec
   283)           c = filter_nolakec(fc)
   284)           if ( pi <= npfts(c) ) then
   285)              p = pfti(c) + pi - 1
   286)              l = plandunit(p)
   287)              g = pgridcell(p)
   288) 
   289)              ! Note: Some glacier_mec pfts may have zero weight
   290)              if (pwtgcell(p)>0._r8 .or. ltype(l)==istice_mec) then
   291)                 if (ltype(l) /= isturb) then
   292)                    eflx_gnet(p) = sabg(p) + dlrad(p) &
   293)                                   + (1-frac_veg_nosno(p))*emg(c)*forc_lwrad(g) - lwrad_emit(c) &
   294)                                   - (eflx_sh_grnd(p)+qflx_evap_soi(p)*htvp(c))
   295)                 else
   296)                    ! For urban columns we use the net longwave radiation (eflx_lwrad_net) because of 
   297)                    ! interactions between urban columns.
   298)                    
   299)                    ! All wasteheat and traffic flux goes into canyon floor
   300)                    if (ctype(c) == icol_road_perv .or. ctype(c) == icol_road_imperv) then
   301)                       eflx_wasteheat_pft(p) = eflx_wasteheat(l)/(1._r8-wtlunit_roof(l))
   302)                       eflx_heat_from_ac_pft(p) = eflx_heat_from_ac(l)/(1._r8-wtlunit_roof(l))
   303)                       eflx_traffic_pft(p) = eflx_traffic(l)/(1._r8-wtlunit_roof(l))
   304)                    else
   305)                       eflx_wasteheat_pft(p) = 0._r8
   306)                       eflx_heat_from_ac_pft(p) = 0._r8
   307)                       eflx_traffic_pft(p) = 0._r8
   308)                    end if
   309)                    ! Include transpiration term because needed for previous road
   310)                    ! and include wasteheat and traffic flux
   311)                    eflx_gnet(p) = sabg(p) + dlrad(p)  &
   312)                                   - eflx_lwrad_net(p) &
   313)                                   - (eflx_sh_grnd(p) + qflx_evap_soi(p)*htvp(c) + qflx_tran_veg(p)*hvap) &
   314)                                   + eflx_wasteheat_pft(p) + eflx_heat_from_ac_pft(p) + eflx_traffic_pft(p)
   315)                    eflx_anthro(p) = eflx_wasteheat_pft(p) + eflx_traffic_pft(p)
   316)                 end if
   317)                 dgnetdT(p) = - cgrnd(p) - dlwrad_emit(c)
   318)                 hs(c) = hs(c) + eflx_gnet(p) * pwtcol(p)
   319)                 dhsdT(c) = dhsdT(c) + dgnetdT(p) * pwtcol(p)
   320)              end if
   321) 
   322)           end if
   323)        end do
   324)     end do
   325) 
   326)     !       Additional calculations with SNICAR: 
   327)     !       Set up tridiagonal matrix in a new manner. There is now 
   328)     !       absorbed solar radiation in each snow layer, instead of 
   329)     !       only the surface. Following the current implementation, 
   330)     !       absorbed solar flux should be: S + ((delS/delT)*dT), 
   331)     !       where S is absorbed radiation, and T is temperature. Now, 
   332)     !       assume delS/delT is zero, then it is OK to just add S 
   333)     !       to each layer
   334) 
   335)     ! Initialize:
   336)     sabg_lyr_col(lbc:ubc,-nlevsno+1:1) = 0._r8
   337)     hs_top(lbc:ubc) = 0._r8
   338) 
   339)     do pi = 1,max_pft_per_col
   340)        do fc = 1,num_nolakec
   341)           c = filter_nolakec(fc)
   342)           lyr_top = snl(c) + 1
   343)           if ( pi <= npfts(c) ) then
   344)              p = pfti(c) + pi - 1
   345)              l = plandunit(p)
   346)              if (pwtgcell(p)>0._r8 .or. ltype(l)==istice_mec) then
   347)                 g = pgridcell(p)
   348)                 if (ltype(l) /= isturb )then
   349) 
   350)                    eflx_gnet_top = sabg_lyr(p,lyr_top) + dlrad(p) + (1-frac_veg_nosno(p))*emg(c)*forc_lwrad(g) &
   351)                         - lwrad_emit(c) - (eflx_sh_grnd(p)+qflx_evap_soi(p)*htvp(c))
   352) 
   353)                    hs_top(c) = hs_top(c) + eflx_gnet_top*pwtcol(p)
   354)              
   355)                    do j = lyr_top,1,1
   356)                       sabg_lyr_col(c,j) = sabg_lyr_col(c,j) + sabg_lyr(p,j) * pwtcol(p)
   357)                    enddo
   358)                 else
   359) 
   360)                    hs_top(c) = hs_top(c) + eflx_gnet(p)*pwtcol(p)
   361) 
   362)                    sabg_lyr_col(c,lyr_top) = sabg_lyr_col(c,lyr_top) + sabg(p) * pwtcol(p)
   363)              
   364)                 endif
   365)              endif
   366) 
   367)           endif
   368)        enddo
   369)     enddo
   370) 
   371)     ! Restrict internal building temperature to between min and max
   372)     ! and determine if heating or air conditioning is on
   373)     do fl = 1,num_urbanl
   374)        l = filter_urbanl(fl)
   375)        if (ltype(l) == isturb) then
   376)           cool_on(l) = .false. 
   377)           heat_on(l) = .false. 
   378)           if (t_building(l) > t_building_max(l)) then
   379)             t_building(l) = t_building_max(l)
   380)             cool_on(l) = .true.
   381)             heat_on(l) = .false.
   382)           else if (t_building(l) < t_building_min(l)) then
   383)             t_building(l) = t_building_min(l)
   384)             cool_on(l) = .false.
   385)             heat_on(l) = .true.
   386)           end if
   387)        end if
   388)     end do
   389) 
   390)     ! Determine heat diffusion through the layer interface and factor used in computing
   391)     ! tridiagonal matrix and set up vector r and vectors a, b, c that define tridiagonal
   392)     ! matrix and solve system
   393) 
   394)     do j = -nlevsno+1,nlevgrnd
   395)        do fc = 1,num_nolakec
   396)           c = filter_nolakec(fc)
   397)           l = clandunit(c)
   398)           if (j >= snl(c)+1) then
   399)              if (j == snl(c)+1) then
   400)                 if (ctype(c)==icol_sunwall .or. ctype(c)==icol_shadewall .or. ctype(c)==icol_roof) then
   401)                   fact(c,j) = dtime/cv(c,j)
   402)                 else
   403)                   fact(c,j) = dtime/cv(c,j) * dz(c,j) / (0.5_r8*(z(c,j)-zi(c,j-1)+capr*(z(c,j+1)-zi(c,j-1))))
   404)                 end if
   405)                 fn(c,j) = tk(c,j)*(t_soisno(c,j+1)-t_soisno(c,j))/(z(c,j+1)-z(c,j))
   406)              else if (j <= nlevgrnd-1) then
   407)                 fact(c,j) = dtime/cv(c,j)
   408)                 fn(c,j) = tk(c,j)*(t_soisno(c,j+1)-t_soisno(c,j))/(z(c,j+1)-z(c,j))
   409)                 dzm     = (z(c,j)-z(c,j-1))
   410)              else if (j == nlevgrnd) then
   411)                 fact(c,j) = dtime/cv(c,j)
   412) 
   413)                 ! For urban sunwall, shadewall, and roof columns, there is a non-zero heat flux across
   414)                 ! the bottom "soil" layer and the equations are derived assuming a prescribed internal
   415)                 ! building temperature. (See Oleson urban notes of 6/18/03).
   416)                 if (ctype(c)==icol_sunwall .or. ctype(c)==icol_shadewall .or. ctype(c)==icol_roof) then
   417)                    fn(c,j) = tk(c,j) * (t_building(l) - cnfac*t_soisno(c,j))/(zi(c,j) - z(c,j))
   418)                 else
   419)                    fn(c,j) = eflx_bot(c)
   420)                 end if
   421)              end if
   422)           end if
   423)        enddo
   424)     end do
   425) 
   426)     do j = -nlevsno+1,nlevgrnd
   427)        do fc = 1,num_nolakec
   428)           c = filter_nolakec(fc)
   429)           l = clandunit(c)
   430)           if (j >= snl(c)+1) then
   431)              if (j == snl(c)+1) then
   432)                 dzp     = z(c,j+1)-z(c,j)
   433)                 at(c,j) = 0._r8
   434)                 bt(c,j) = 1+(1._r8-cnfac)*fact(c,j)*tk(c,j)/dzp-fact(c,j)*dhsdT(c)
   435)                 ct(c,j) =  -(1._r8-cnfac)*fact(c,j)*tk(c,j)/dzp
   436)                 ! changed hs to hs_top
   437)                 rt(c,j) = t_soisno(c,j) +  fact(c,j)*( hs_top(c) - dhsdT(c)*t_soisno(c,j) + cnfac*fn(c,j) )
   438)              else if (j <= nlevgrnd-1) then
   439)                 dzm     = (z(c,j)-z(c,j-1))
   440)                 dzp     = (z(c,j+1)-z(c,j))
   441)                 at(c,j) =   - (1._r8-cnfac)*fact(c,j)* tk(c,j-1)/dzm
   442)                 bt(c,j) = 1._r8+ (1._r8-cnfac)*fact(c,j)*(tk(c,j)/dzp + tk(c,j-1)/dzm)
   443)                 ct(c,j) =   - (1._r8-cnfac)*fact(c,j)* tk(c,j)/dzp
   444) 
   445)                 ! if this is a snow layer or the top soil layer,
   446)                 ! add absorbed solar flux to factor 'rt'
   447)                 if (j <= 1) then
   448)                    rt(c,j) = t_soisno(c,j) + cnfac*fact(c,j)*( fn(c,j) - fn(c,j-1) ) + (fact(c,j)*sabg_lyr_col(c,j))
   449)                 else
   450)                    rt(c,j) = t_soisno(c,j) + cnfac*fact(c,j)*( fn(c,j) - fn(c,j-1) )
   451)                 endif
   452) 
   453)              else if (j == nlevgrnd) then
   454) 
   455)                 ! For urban sunwall, shadewall, and roof columns, there is a non-zero heat flux across
   456)                 ! the bottom "soil" layer and the equations are derived assuming a prescribed internal
   457)                 ! building temperature. (See Oleson urban notes of 6/18/03).
   458)                 if (ctype(c)==icol_sunwall .or. ctype(c)==icol_shadewall .or. ctype(c)==icol_roof) then
   459)                    dzm     = ( z(c,j)-z(c,j-1))
   460)                    dzp     = (zi(c,j)-z(c,j))
   461)                    at(c,j) =   - (1._r8-cnfac)*fact(c,j)*(tk(c,j-1)/dzm)
   462)                    bt(c,j) = 1._r8+ (1._r8-cnfac)*fact(c,j)*(tk(c,j-1)/dzm + tk(c,j)/dzp)
   463)                    ct(c,j) = 0._r8
   464)                    rt(c,j) = t_soisno(c,j) + fact(c,j)*( fn(c,j) - cnfac*fn(c,j-1) )
   465)                 else
   466)                    dzm     = (z(c,j)-z(c,j-1))
   467)                    at(c,j) =   - (1._r8-cnfac)*fact(c,j)*tk(c,j-1)/dzm
   468)                    bt(c,j) = 1._r8+ (1._r8-cnfac)*fact(c,j)*tk(c,j-1)/dzm
   469)                    ct(c,j) = 0._r8
   470)                    rt(c,j) = t_soisno(c,j) - cnfac*fact(c,j)*fn(c,j-1) + fact(c,j)*fn(c,j)
   471)                 end if
   472)              end if
   473) 
   474)           end if
   475)        enddo
   476)     end do
   477) 
   478)     do fc = 1,num_nolakec
   479)        c = filter_nolakec(fc)
   480)        jtop(c) = snl(c) + 1
   481)     end do
   482)     call Tridiagonal(lbc, ubc, -nlevsno+1, nlevgrnd, jtop, num_nolakec, filter_nolakec, &
   483)                      at, bt, ct, rt, t_soisno(lbc:ubc,-nlevsno+1:nlevgrnd))
   484) 
   485)     ! Melting or Freezing
   486) 
   487)     do j = -nlevsno+1,nlevgrnd
   488)        do fc = 1,num_nolakec
   489)           c = filter_nolakec(fc)
   490)           l = clandunit(c)
   491)           if (j >= snl(c)+1) then
   492)              if (j <= nlevgrnd-1) then
   493)                 fn1(c,j) = tk(c,j)*(t_soisno(c,j+1)-t_soisno(c,j))/(z(c,j+1)-z(c,j))
   494)              else if (j == nlevgrnd) then
   495) 
   496)                 ! For urban sunwall, shadewall, and roof columns, there is a non-zero heat flux across
   497)                 ! the bottom "soil" layer and the equations are derived assuming a prescribed internal
   498)                 ! building temperature. (See Oleson urban notes of 6/18/03).
   499)                 ! Note new formulation for fn, this will be used below in brr computation
   500)                 if (ctype(c)==icol_sunwall .or. ctype(c)==icol_shadewall .or. ctype(c)==icol_roof) then
   501)                    fn1(c,j) = tk(c,j) * (t_building(l) - t_soisno(c,j))/(zi(c,j) - z(c,j))
   502)                    fn(c,j)  = tk(c,j) * (t_building(l) - tssbef(c,j))/(zi(c,j) - z(c,j))
   503)                 else
   504)                    fn1(c,j) = 0._r8
   505)                 end if
   506)              end if
   507)           end if
   508)        end do
   509)     end do
   510) 
   511)     do fc = 1,num_nolakec
   512)        c = filter_nolakec(fc)
   513)        l = clandunit(c)
   514)        if (ltype(l) == isturb) then
   515)          eflx_building_heat(c) = cnfac*fn(c,nlevurb) + (1-cnfac)*fn1(c,nlevurb)
   516)          if (cool_on(l)) then
   517)            eflx_urban_ac(c) = abs(eflx_building_heat(c))
   518)            eflx_urban_heat(c) = 0._r8
   519)          else if (heat_on(l)) then
   520)            eflx_urban_ac(c) = 0._r8
   521)            eflx_urban_heat(c) = abs(eflx_building_heat(c))
   522)          else
   523)            eflx_urban_ac(c) = 0._r8
   524)            eflx_urban_heat(c) = 0._r8
   525)          end if
   526)        end if
   527)     end do
   528) 
   529)     do j = -nlevsno+1,nlevgrnd
   530)        do fc = 1,num_nolakec
   531)           c = filter_nolakec(fc)
   532)           l = clandunit(c)
   533)           if (j >= snl(c)+1) then
   534)              if (j == snl(c)+1) then
   535)                 brr(c,j) = cnfac*fn(c,j) + (1._r8-cnfac)*fn1(c,j)
   536)              else
   537)                 brr(c,j) = cnfac*(fn(c,j)-fn(c,j-1)) + (1._r8-cnfac)*(fn1(c,j)-fn1(c,j-1))
   538)              end if
   539)           end if
   540)        end do
   541)     end do
   542) 
   543)     call PhaseChange (lbc, ubc, num_nolakec, filter_nolakec, fact, brr, hs, dhsdT, xmf, hs_top, sabg_lyr_col)
   544) 
   545)     do fc = 1,num_nolakec
   546)        c = filter_nolakec(fc)
   547)        t_grnd(c) = t_soisno(c,snl(c)+1)
   548)     end do
   549) 
   550) 
   551) ! Initialize soil heat content
   552)     do fc = 1,num_nolakec
   553)        c = filter_nolakec(fc)
   554)        l = clandunit(c)
   555)        if (ltype(l) /= isturb) then
   556)          hc_soisno(c) = 0._r8
   557)          hc_soi(c)    = 0._r8
   558)        end if
   559)        eflx_fgr12(c)= 0._r8
   560)     end do
   561) 
   562) ! Calculate soil heat content and soil plus snow heat content
   563)     do j = -nlevsno+1,nlevgrnd
   564)        do fc = 1,num_nolakec
   565)           c = filter_nolakec(fc)
   566)           l = clandunit(c)
   567)           eflx_fgr12(c) = -cnfac*fn(c,1) - (1._r8-cnfac)*fn1(c,1)
   568)           if (ltype(l) /= isturb) then
   569)             if (j >= snl(c)+1) then
   570)                hc_soisno(c) = hc_soisno(c) + cv(c,j)*t_soisno(c,j) / 1.e6_r8
   571)             endif
   572)             if (j >= 1) then
   573)                hc_soi(c) = hc_soi(c) + cv(c,j)*t_soisno(c,j) / 1.e6_r8
   574)             end if
   575)           end if
   576)        end do
   577)     end do
   578) 
   579)   end subroutine SoilTemperature

   584) ! !IROUTINE: SoilThermProp

   586) ! !INTERFACE:
   587)   subroutine SoilThermProp (lbc, ubc,  num_nolakec, filter_nolakec, tk, cv)
   588) !
   589) ! !DESCRIPTION:
   590) ! Calculation of thermal conductivities and heat capacities of
   591) ! snow/soil layers
   592) ! (1) The volumetric heat capacity is calculated as a linear combination
   593) !     in terms of the volumetric fraction of the constituent phases.
   594) !
   595) ! (2) The thermal conductivity of soil is computed from the algorithm of
   596) !     Johansen (as reported by Farouki 1981), and of snow is from the
   597) !     formulation used in SNTHERM (Jordan 1991).
   598) ! The thermal conductivities at the interfaces between two neighboring
   599) ! layers (j, j+1) are derived from an assumption that the flux across
   600) ! the interface is equal to that from the node j to the interface and the
   601) ! flux from the interface to the node j+1.
   602) !
   603) ! !USES:
   604)     use shr_kind_mod, only : r8 => shr_kind_r8
   605)     use clmtype
   606)     use clm_varcon  , only : denh2o, denice, tfrz, tkwat, tkice, tkair, &
   607)                              cpice,  cpliq,  istice, istice_mec, istwet, &
   608)                              icol_roof, icol_sunwall, icol_shadewall, &
   609)                              icol_road_perv, icol_road_imperv
   610)     use clm_varpar  , only : nlevsno, nlevgrnd, nlevurb, nlevsoi
   611) !
   612) ! !ARGUMENTS:
   613)     implicit none
   614)     integer , intent(in)  :: lbc, ubc                       ! column bounds
   615)     integer , intent(in)  :: num_nolakec                    ! number of column non-lake points in column filter
   616)     integer , intent(in)  :: filter_nolakec(ubc-lbc+1)      ! column filter for non-lake points
   617)     real(r8), intent(out) :: cv(lbc:ubc,-nlevsno+1:nlevgrnd)! heat capacity [J/(m2 K)]
   618)     real(r8), intent(out) :: tk(lbc:ubc,-nlevsno+1:nlevgrnd)! thermal conductivity [W/(m K)]
   619) !
   620) ! !CALLED FROM:
   621) ! subroutine SoilTemperature in this module
   622) !
   623) ! !REVISION HISTORY:
   624) ! 15 September 1999: Yongjiu Dai; Initial code
   625) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
   626) ! 2/13/02, Peter Thornton: migrated to new data structures
   627) ! 7/01/03, Mariana Vertenstein: migrated to vector code
   628) !
   629) ! !LOCAL VARIABLES:
   630) !
   631) ! local pointers to original implicit in scalars
   632) !
   633)     integer , pointer :: ctype(:)         ! column type
   634)     integer , pointer :: clandunit(:)     ! column's landunit
   635)     integer , pointer :: ltype(:)       ! landunit type
   636)     integer , pointer :: snl(:)           ! number of snow layers
   637)     real(r8), pointer :: h2osno(:)        ! snow water (mm H2O)
   638) !
   639) ! local pointers to original implicit in arrays
   640) !
   641)     real(r8), pointer :: watsat(:,:)      ! volumetric soil water at saturation (porosity)
   642)     real(r8), pointer :: tksatu(:,:)      ! thermal conductivity, saturated soil [W/m-K]
   643)     real(r8), pointer :: tkmg(:,:)        ! thermal conductivity, soil minerals  [W/m-K]
   644)     real(r8), pointer :: tkdry(:,:)       ! thermal conductivity, dry soil (W/m/Kelvin)
   645)     real(r8), pointer :: csol(:,:)        ! heat capacity, soil solids (J/m**3/Kelvin)
   646)     real(r8), pointer :: dz(:,:)          ! layer depth (m)
   647)     real(r8), pointer :: zi(:,:)          ! interface level below a "z" level (m)
   648)     real(r8), pointer :: z(:,:)           ! layer thickness (m)
   649)     real(r8), pointer :: t_soisno(:,:)    ! soil temperature (Kelvin)
   650)     real(r8), pointer :: h2osoi_liq(:,:)  ! liquid water (kg/m2)
   651)     real(r8), pointer :: h2osoi_ice(:,:)  ! ice lens (kg/m2)
   652)     real(r8), pointer :: tk_wall(:,:)     ! thermal conductivity of urban wall
   653)     real(r8), pointer :: tk_roof(:,:)     ! thermal conductivity of urban roof
   654)     real(r8), pointer :: tk_improad(:,:)  ! thermal conductivity of urban impervious road
   655)     real(r8), pointer :: cv_wall(:,:)     ! thermal conductivity of urban wall
   656)     real(r8), pointer :: cv_roof(:,:)     ! thermal conductivity of urban roof
   657)     real(r8), pointer :: cv_improad(:,:)  ! thermal conductivity of urban impervious road
   658)     integer,  pointer :: nlev_improad(:)  ! number of impervious road layers
   659) 
   660) !
   661) !
   662) ! !OTHER LOCAL VARIABLES:
   663) !EOP
   664) !
   665)     integer  :: l,c,j                     ! indices
   666)     integer  :: fc                        ! lake filtered column indices
   667)     real(r8) :: bw                        ! partial density of water (ice + liquid)
   668)     real(r8) :: dksat                     ! thermal conductivity for saturated soil (j/(k s m))
   669)     real(r8) :: dke                       ! kersten number
   670)     real(r8) :: fl                        ! fraction of liquid or unfrozen water to total water
   671)     real(r8) :: satw                      ! relative total water content of soil.
   672)     real(r8) :: thk(lbc:ubc,-nlevsno+1:nlevgrnd) ! thermal conductivity of layer
   673)     real(r8) :: thk_bedrock = 3.0_r8      ! thermal conductivity of 'typical' saturated granitic rock 
   674)                                           ! (Clauser and Huenges, 1995)(W/m/K)
   675) !-----------------------------------------------------------------------
   676) 
   677)     ! Assign local pointers to derived subtypes components (landunit-level)
   678) 
   679)     ltype    => lun%itype
   680) 
   681)     ! Assign local pointers to derived subtypes components (column-level)
   682) 
   683)     ctype      => col%itype
   684)     clandunit  => col%landunit
   685)     snl        => cps%snl
   686)     h2osno     => cws%h2osno
   687)     watsat     => cps%watsat
   688)     tksatu     => cps%tksatu
   689)     tkmg       => cps%tkmg
   690)     tkdry      => cps%tkdry
   691)     csol       => cps%csol
   692)     dz         => cps%dz
   693)     zi         => cps%zi
   694)     z          => cps%z
   695)     t_soisno   => ces%t_soisno
   696)     h2osoi_liq => cws%h2osoi_liq
   697)     h2osoi_ice => cws%h2osoi_ice
   698)     tk_wall    => lps%tk_wall
   699)     tk_roof    => lps%tk_roof
   700)     tk_improad => lps%tk_improad
   701)     cv_wall    => lps%cv_wall
   702)     cv_roof    => lps%cv_roof
   703)     cv_improad => lps%cv_improad
   704)     nlev_improad => lps%nlev_improad
   705) 
   706)     ! Thermal conductivity of soil from Farouki (1981)
   707)     ! Urban values are from Masson et al. 2002, Evaluation of the Town Energy Balance (TEB)
   708)     ! scheme with direct measurements from dry districts in two cities, J. Appl. Meteorol.,
   709)     ! 41, 1011-1026.
   710) 
   711)     do j = -nlevsno+1,nlevgrnd
   712)        do fc = 1, num_nolakec
   713)           c = filter_nolakec(fc)
   714) 
   715)           ! Only examine levels from 1->nlevgrnd
   716)           if (j >= 1) then    
   717)              l = clandunit(c)
   718)              if (ctype(c) == icol_sunwall .OR. ctype(c) == icol_shadewall) then
   719)                 thk(c,j) = tk_wall(l,j)
   720)              else if (ctype(c) == icol_roof) then
   721)                 thk(c,j) = tk_roof(l,j)
   722)              else if (ctype(c) == icol_road_imperv .and. j >= 1 .and. j <= nlev_improad(l)) then
   723)                 thk(c,j) = tk_improad(l,j)
   724)              elseif (ltype(l) /= istwet .AND. ltype(l) /= istice  &
   725)                                         .AND. ltype(l) /= istice_mec) then
   726)                 satw = (h2osoi_liq(c,j)/denh2o + h2osoi_ice(c,j)/denice)/(dz(c,j)*watsat(c,j))
   727)                 satw = min(1._r8, satw)
   728)                 if (satw > .1e-6_r8) then
   729)                    fl = h2osoi_liq(c,j)/(h2osoi_ice(c,j)+h2osoi_liq(c,j))
   730)                    if (t_soisno(c,j) >= tfrz) then       ! Unfrozen soil
   731)                       dke = max(0._r8, log10(satw) + 1.0_r8)
   732)                       dksat = tksatu(c,j)
   733)                    else                               ! Frozen soil
   734)                       dke = satw
   735)                       dksat = tkmg(c,j)*0.249_r8**(fl*watsat(c,j))*2.29_r8**watsat(c,j)
   736)                    endif
   737)                    thk(c,j) = dke*dksat + (1._r8-dke)*tkdry(c,j)
   738)                 else
   739)                    thk(c,j) = tkdry(c,j)
   740)                 endif
   741)                 if (j > nlevsoi) thk(c,j) = thk_bedrock
   742)              else if (ltype(l) == istice .OR. ltype(l) == istice_mec) then
   743)                 thk(c,j) = tkwat
   744)                 if (t_soisno(c,j) < tfrz) thk(c,j) = tkice
   745)              else if (ltype(l) == istwet) then                         
   746)                 if (j > nlevsoi) then 
   747)                    thk(c,j) = thk_bedrock
   748)                 else
   749)                    thk(c,j) = tkwat
   750)                    if (t_soisno(c,j) < tfrz) thk(c,j) = tkice
   751)                 endif
   752)              endif
   753)           endif
   754) 
   755)           ! Thermal conductivity of snow, which from Jordan (1991) pp. 18
   756)           ! Only examine levels from snl(c)+1 -> 0 where snl(c) < 1
   757)           if (snl(c)+1 < 1 .AND. (j >= snl(c)+1) .AND. (j <= 0)) then  
   758)              bw = (h2osoi_ice(c,j)+h2osoi_liq(c,j))/dz(c,j)
   759)              thk(c,j) = tkair + (7.75e-5_r8 *bw + 1.105e-6_r8*bw*bw)*(tkice-tkair)
   760)           end if
   761) 
   762)        end do
   763)     end do
   764) 
   765)     ! Thermal conductivity at the layer interface
   766) 
   767)     do j = -nlevsno+1,nlevgrnd
   768)        do fc = 1,num_nolakec
   769)           c = filter_nolakec(fc)
   770)           if (j >= snl(c)+1 .AND. j <= nlevgrnd-1) then
   771)              tk(c,j) = thk(c,j)*thk(c,j+1)*(z(c,j+1)-z(c,j)) &
   772)                        /(thk(c,j)*(z(c,j+1)-zi(c,j))+thk(c,j+1)*(zi(c,j)-z(c,j)))
   773)           else if (j == nlevgrnd) then
   774) 
   775)              ! For urban sunwall, shadewall, and roof columns, there is a non-zero heat flux across
   776)              ! the bottom "soil" layer and the equations are derived assuming a prescribed internal
   777)              ! building temperature. (See Oleson urban notes of 6/18/03).
   778)              if (ctype(c)==icol_sunwall .OR. ctype(c)==icol_shadewall .OR. ctype(c)==icol_roof) then
   779)                 tk(c,j) = thk(c,j)
   780)              else
   781)                 tk(c,j) = 0._r8
   782)              end if
   783)           end if
   784)        end do
   785)     end do
   786) 
   787)     ! Soil heat capacity, from de Vires (1963)
   788)     ! Urban values are from Masson et al. 2002, Evaluation of the Town Energy Balance (TEB)
   789)     ! scheme with direct measurements from dry districts in two cities, J. Appl. Meteorol.,
   790)     ! 41, 1011-1026.
   791) 
   792)     do j = 1, nlevgrnd
   793)        do fc = 1,num_nolakec
   794)           c = filter_nolakec(fc)
   795)           l = clandunit(c)
   796)           if (ctype(c)==icol_sunwall .OR. ctype(c)==icol_shadewall) then
   797)              cv(c,j) = cv_wall(l,j) * dz(c,j)
   798)           else if (ctype(c) == icol_roof) then
   799)              cv(c,j) = cv_roof(l,j) * dz(c,j)
   800)           else if (ctype(c) == icol_road_imperv .and. j >= 1 .and. j <= nlev_improad(l)) then
   801)              cv(c,j) = cv_improad(l,j) * dz(c,j)
   802)           elseif (ltype(l) /= istwet .AND. ltype(l) /= istice  &
   803)                                      .AND. ltype(l) /= istice_mec) then
   804)              cv(c,j) = csol(c,j)*(1-watsat(c,j))*dz(c,j) + (h2osoi_ice(c,j)*cpice + h2osoi_liq(c,j)*cpliq)
   805)           else if (ltype(l) == istwet) then 
   806)              cv(c,j) = (h2osoi_ice(c,j)*cpice + h2osoi_liq(c,j)*cpliq)
   807)              if (j > nlevsoi) cv(c,j) = csol(c,j)*dz(c,j)
   808)           else if (ltype(l) == istice .OR. ltype(l) == istice_mec) then 
   809)              cv(c,j) = (h2osoi_ice(c,j)*cpice + h2osoi_liq(c,j)*cpliq)
   810)           endif
   811)           if (j == 1) then
   812)              if (snl(c)+1 == 1 .AND. h2osno(c) > 0._r8) then
   813)                 cv(c,j) = cv(c,j) + cpice*h2osno(c)
   814)              end if
   815)           end if
   816)        enddo
   817)     end do
   818) 
   819)     ! Snow heat capacity
   820) 
   821)     do j = -nlevsno+1,0
   822)        do fc = 1,num_nolakec
   823)           c = filter_nolakec(fc)
   824)           if (snl(c)+1 < 1 .and. j >= snl(c)+1) then
   825)              cv(c,j) = cpliq*h2osoi_liq(c,j) + cpice*h2osoi_ice(c,j)
   826)           end if
   827)        end do
   828)     end do
   829) 
   830)   end subroutine SoilThermProp

   918)     real(r8), pointer :: sucsat(:,:)      !minimum soil suction (mm)
   919)     real(r8), pointer :: watsat(:,:)      !volumetric soil water at saturation (porosity)
   920)     real(r8), pointer :: bsw(:,:)         !Clapp and Hornberger "b"
   921)     real(r8), pointer :: dz(:,:)          !layer thickness (m)
   922) !
   923) ! local pointers to original implicit inout arrays
   924) !
   925)     real(r8), pointer :: t_soisno(:,:)    !soil temperature (Kelvin)
   926) !
   927) ! local pointers to original implicit out arrays
   928) !
   929)     integer, pointer :: imelt(:,:)        !flag for melting (=1), freezing (=2), Not=0 (new)
   930) !
   931) !
   932) ! !OTHER LOCAL VARIABLES:
   933) !EOP
   934) !
   935)     integer  :: j,c,g,l                            !do loop index
   936)     integer  :: fc                                 !lake filtered column indices
   937)     real(r8) :: dtime                              !land model time step (sec)
   938)     real(r8) :: heatr                              !energy residual or loss after melting or freezing
   939)     real(r8) :: temp1                              !temporary variables [kg/m2]
   940)     real(r8) :: hm(lbc:ubc,-nlevsno+1:nlevgrnd)    !energy residual [W/m2]
   941)     real(r8) :: xm(lbc:ubc,-nlevsno+1:nlevgrnd)    !melting or freezing within a time step [kg/m2]
   942)     real(r8) :: wmass0(lbc:ubc,-nlevsno+1:nlevgrnd)!initial mass of ice and liquid (kg/m2)
   943)     real(r8) :: wice0 (lbc:ubc,-nlevsno+1:nlevgrnd)!initial mass of ice (kg/m2)
   944)     real(r8) :: wliq0 (lbc:ubc,-nlevsno+1:nlevgrnd)!initial mass of liquid (kg/m2)
   945)     real(r8) :: supercool(lbc:ubc,nlevgrnd)        !supercooled water in soil (kg/m2) 
   946)     real(r8) :: propor                             !proportionality constant (-)
   947)     real(r8) :: tinc                               !t(n+1)-t(n) (K)
   948)     real(r8) :: smp                                !frozen water potential (mm)

   950) 
   951)     ! Assign local pointers to derived subtypes components (column-level)
   952) 
   953)     snl          => cps%snl
   954)     h2osno       => cws%h2osno
   955)     snowdp       => cps%snowdp
   956)     qflx_snow_melt => cwf%qflx_snow_melt
   957)     qflx_snomelt => cwf%qflx_snomelt
   958)     eflx_snomelt => cef%eflx_snomelt
   959)     eflx_snomelt_u => cef%eflx_snomelt_u
   960)     eflx_snomelt_r => cef%eflx_snomelt_r
   961)     h2osoi_liq   => cws%h2osoi_liq
   962)     h2osoi_ice   => cws%h2osoi_ice
   963)     imelt        => cps%imelt
   964)     t_soisno     => ces%t_soisno
   965)     tssbef       => ces%tssbef
   966)     bsw          => cps%bsw
   967)     sucsat       => cps%sucsat
   968)     watsat       => cps%watsat
   969)     dz           => cps%dz
   970)     ctype        => col%itype
   971)     clandunit    => col%landunit
   972)     ltype        => lun%itype
   973)     qflx_snofrz_lyr => cwf%qflx_snofrz_lyr
   974)     qflx_snofrz_col => cwf%qflx_snofrz_col
   975)     qflx_glcice  => cwf%qflx_glcice
   976)     qflx_glcice_melt  => cwf%qflx_glcice_melt
   977) 
   978)     ! Get step size
   979) 
   980)     dtime = get_step_size()
   981) 
   982)     ! Initialization
   983) 
   984)     do fc = 1,num_nolakec
   985)        c = filter_nolakec(fc)
   986)        l = clandunit(c)
   987) 
   988)        qflx_snomelt(c) = 0._r8
   989)        qflx_snow_melt(c) = 0._r8
   990)        xmf(c) = 0._r8
   991)        qflx_snofrz_lyr(c,-nlevsno+1:0) = 0._r8
   992)        qflx_snofrz_col(c) = 0._r8
   993)        if (ltype(l)==istice_mec) then
   994)           ! only need to initialize qflx_glcice_melt over ice_mec landunits, because
   995)           ! those are the only places where it is computed
   996)           qflx_glcice_melt(c) = 0._r8
   997)        end if
   998)     end do
   999) 
  1000)     do j = -nlevsno+1,nlevgrnd       ! all layers
  1001)        do fc = 1,num_nolakec
  1002)           c = filter_nolakec(fc)
  1003)           if (j >= snl(c)+1) then
  1004) 
  1005)              ! Initialization
  1006)              imelt(c,j) = 0
  1007)              hm(c,j) = 0._r8
  1008)              xm(c,j) = 0._r8
  1009)              wice0(c,j) = h2osoi_ice(c,j)
  1010)              wliq0(c,j) = h2osoi_liq(c,j)
  1011)              wmass0(c,j) = h2osoi_ice(c,j) + h2osoi_liq(c,j)
  1012)           endif   ! end of snow layer if-block
  1013)        end do   ! end of column-loop
  1014)     enddo   ! end of level-loop
  1015) 
  1016)     do j = -nlevsno+1,0             ! snow layers 
  1017)        do fc = 1,num_nolakec
  1018)           c = filter_nolakec(fc)
  1019)           if (j >= snl(c)+1) then
  1020) 
  1021)              ! Melting identification
  1022)              ! If ice exists above melt point, melt some to liquid.
  1023)              if (h2osoi_ice(c,j) > 0._r8 .AND. t_soisno(c,j) > tfrz) then
  1024)                 imelt(c,j) = 1
  1025)                 t_soisno(c,j) = tfrz
  1026)              endif
  1027) 
  1028)              ! Freezing identification
  1029)              ! If liquid exists below melt point, freeze some to ice.
  1030)              if (h2osoi_liq(c,j) > 0._r8 .AND. t_soisno(c,j) < tfrz) then
  1031)                 imelt(c,j) = 2
  1032)                 t_soisno(c,j) = tfrz
  1033)              endif
  1034)           endif   ! end of snow layer if-block
  1035)        end do   ! end of column-loop
  1036)     enddo   ! end of level-loop
  1037) 
  1038)     do j = 1,nlevgrnd             ! soil layers 
  1039)        do fc = 1,num_nolakec
  1040)           c = filter_nolakec(fc)
  1041)           l = clandunit(c)
  1042)           if (h2osoi_ice(c,j) > 0. .AND. t_soisno(c,j) > tfrz) then
  1043)              imelt(c,j) = 1
  1044)              t_soisno(c,j) = tfrz
  1045)           endif
  1046) 
  1047)           ! from Zhao (1997) and Koren (1999)
  1048)           supercool(c,j) = 0.0_r8

  1051)                 smp = hfus*(tfrz-t_soisno(c,j))/(grav*t_soisno(c,j)) * 1000._r8  !(mm)

  1054)              endif
  1055)           endif

  1058)              imelt(c,j) = 2
  1059)              t_soisno(c,j) = tfrz
  1060)           endif
  1061) 
  1062)           ! If snow exists, but its thickness is less than the critical value (0.01 m)
  1063)           if (snl(c)+1 == 1 .AND. h2osno(c) > 0._r8 .AND. j == 1) then
  1064)              if (t_soisno(c,j) > tfrz) then
  1065)                 imelt(c,j) = 1
  1066)                 t_soisno(c,j) = tfrz
  1067)              endif
  1068)           endif
  1069)        end do
  1070)     enddo
  1071) 
  1072)     do j = -nlevsno+1,nlevgrnd       ! all layers
  1073)        do fc = 1,num_nolakec
  1074)           c = filter_nolakec(fc)
  1075) 
  1076)           if (j >= snl(c)+1) then
  1077) 
  1078)              ! Calculate the energy surplus and loss for melting and freezing
  1079)              if (imelt(c,j) > 0) then
  1080)                 tinc = t_soisno(c,j)-tssbef(c,j)
  1081)                 
  1082)                 ! added unique cases for this calculation,
  1083)                 ! to account for absorbed solar radiation in each layer
  1084)                 if (j == snl(c)+1) then
  1085)                    ! top layer
  1086)                    hm(c,j) = hs_top(c) + dhsdT(c)*tinc + brr(c,j) - tinc/fact(c,j)
  1087)                 elseif (j <= 1) then
  1088)                    ! snow layer or top soil layer (where sabg_lyr_col is defined)
  1089)                    hm(c,j) = brr(c,j) - tinc/fact(c,j) + sabg_lyr_col(c,j)
  1090)                 else
  1091)                    ! soil layer
  1092)                    hm(c,j) = brr(c,j) - tinc/fact(c,j)
  1093)                 endif
  1094) 
  1095)              endif
  1096) 
  1097)              ! These two errors were checked carefully (Y. Dai).  They result from the
  1098)              ! computed error of "Tridiagonal-Matrix" in subroutine "thermal".
  1099)              if (imelt(c,j) == 1 .AND. hm(c,j) < 0._r8) then
  1100)                 hm(c,j) = 0._r8
  1101)                 imelt(c,j) = 0
  1102)              endif
  1103)              if (imelt(c,j) == 2 .AND. hm(c,j) > 0._r8) then
  1104)                 hm(c,j) = 0._r8
  1105)                 imelt(c,j) = 0
  1106)              endif
  1107) 
  1108)              ! The rate of melting and freezing
  1109) 
  1110)              if (imelt(c,j) > 0 .and. abs(hm(c,j)) > 0._r8) then
  1111)                 xm(c,j) = hm(c,j)*dtime/hfus                           ! kg/m2
  1112) 
  1113)                 ! If snow exists, but its thickness is less than the critical value
  1114)                 ! (1 cm). Note: more work is needed to determine how to tune the
  1115)                 ! snow depth for this case
  1116)                 if (j == 1) then
  1117)                    if (snl(c)+1 == 1 .AND. h2osno(c) > 0._r8 .AND. xm(c,j) > 0._r8) then
  1118)                       temp1 = h2osno(c)                           ! kg/m2
  1119)                       h2osno(c) = max(0._r8,temp1-xm(c,j))
  1120)                       propor = h2osno(c)/temp1
  1121)                       snowdp(c) = propor * snowdp(c)
  1122)                       heatr = hm(c,j) - hfus*(temp1-h2osno(c))/dtime   ! W/m2
  1123)                       if (heatr > 0._r8) then
  1124)                          xm(c,j) = heatr*dtime/hfus                    ! kg/m2
  1125)                          hm(c,j) = heatr                               ! W/m2
  1126)                       else
  1127)                          xm(c,j) = 0._r8
  1128)                          hm(c,j) = 0._r8
  1129)                       endif
  1130)                       qflx_snomelt(c) = max(0._r8,(temp1-h2osno(c)))/dtime   ! kg/(m2 s)
  1131)                       xmf(c) = hfus*qflx_snomelt(c)
  1132)                       qflx_snow_melt(c) = qflx_snomelt(c)
  1133)                    endif
  1134)                 endif
  1135) 
  1136)                 heatr = 0._r8
  1137)                 if (xm(c,j) > 0._r8) then
  1138)                    h2osoi_ice(c,j) = max(0._r8, wice0(c,j)-xm(c,j))
  1139)                    heatr = hm(c,j) - hfus*(wice0(c,j)-h2osoi_ice(c,j))/dtime
  1140)                 else if (xm(c,j) < 0._r8) then
  1141)                    if (j <= 0) then
  1142)                       h2osoi_ice(c,j) = min(wmass0(c,j), wice0(c,j)-xm(c,j))  ! snow
  1143)                    else
  1144)                       if (wmass0(c,j) < supercool(c,j)) then
  1145)                          h2osoi_ice(c,j) = 0._r8
  1146)                       else
  1147)                          h2osoi_ice(c,j) = min(wmass0(c,j) - supercool(c,j),wice0(c,j)-xm(c,j))
  1148)                       endif
  1149)                    endif
  1150)                    heatr = hm(c,j) - hfus*(wice0(c,j)-h2osoi_ice(c,j))/dtime
  1151)                 endif
  1152) 
  1153)                 h2osoi_liq(c,j) = max(0._r8,wmass0(c,j)-h2osoi_ice(c,j))
  1154) 
  1155)                 if (abs(heatr) > 0._r8) then
  1156)                    if (j > snl(c)+1) then
  1157)                       t_soisno(c,j) = t_soisno(c,j) + fact(c,j)*heatr
  1158)                    else
  1159)                       t_soisno(c,j) = t_soisno(c,j) + fact(c,j)*heatr/(1._r8-fact(c,j)*dhsdT(c))
  1160)                    endif
  1161)                    if (j <= 0) then    ! snow
  1162)                       if (h2osoi_liq(c,j)*h2osoi_ice(c,j)>0._r8) t_soisno(c,j) = tfrz
  1163)                    end if
  1164)                 endif
  1165) 
  1166)                 xmf(c) = xmf(c) + hfus * (wice0(c,j)-h2osoi_ice(c,j))/dtime
  1167) 
  1168)                 if (imelt(c,j) == 1 .AND. j < 1) then
  1169)                    qflx_snomelt(c) = qflx_snomelt(c) + max(0._r8,(wice0(c,j)-h2osoi_ice(c,j)))/dtime
  1170)                 endif
  1171) 
  1172)                 ! layer freezing mass flux (positive):
  1173)                 if (imelt(c,j) == 2 .AND. j < 1) then
  1174)                    qflx_snofrz_lyr(c,j) = max(0._r8,(h2osoi_ice(c,j)-wice0(c,j)))/dtime
  1175)                 endif
  1176) 
  1177)              endif
  1178)           endif   ! end of snow layer if-block
  1179) 
  1180)        ! For glacier_mec columns, compute negative ice flux from melted ice.
  1181)        ! Note that qflx_glcice can also include a positive component from excess snow,
  1182)        !  as computed in Hydrology2Mod.F90.
  1183) 
  1184)           l = clandunit(c)
  1185)           if (ltype(l)==istice_mec) then
  1186) 
  1187)              if (j>=1 .and. h2osoi_liq(c,j) > 0._r8) then   ! ice layer with meltwater
  1188)                 ! melting corresponds to a negative ice flux
  1189)                 qflx_glcice_melt(c) = qflx_glcice_melt(c) + h2osoi_liq(c,j)/dtime
  1190)                 qflx_glcice(c) = qflx_glcice(c) - h2osoi_liq(c,j)/dtime
  1191) 
  1192)                 ! convert layer back to pure ice by "borrowing" ice from below the column
  1193)                 h2osoi_ice(c,j) = h2osoi_ice(c,j) + h2osoi_liq(c,j)
  1194)                 h2osoi_liq(c,j) = 0._r8
  1195)
