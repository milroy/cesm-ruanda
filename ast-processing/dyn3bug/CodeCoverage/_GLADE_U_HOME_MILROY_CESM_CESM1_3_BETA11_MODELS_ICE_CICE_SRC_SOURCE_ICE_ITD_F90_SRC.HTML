    53)                          !   0 = delta function
    54)                          !   1 = linear remap
    55)          kcatbound   , & !   0 = old category boundary formula
    56)                          !   1 = new formula giving round numbers
    57)                          !   2 = WMO standard

    59)          ilyrn (ncat), & ! array position of bottom ice layer in each cat

    76) ! hi_min is a more general purpose parameter, but is specifically

    78) ! hin_max(0) = 0.1 m for the delta function itd

    93) ! !IROUTINE: init_itd - initalize area fraction and thickness boundaries for ITD
    94) !
    95) ! !INTERFACE:
    96) !
    97)       subroutine init_itd
    98) !
    99) ! !DESCRIPTION:
   100) !
   101) ! Initialize area fraction and thickness boundaries for the itd model
   102) !
   103) ! !REVISION HISTORY:
   104) !
   105) ! authors: William H. Lipscomb and Elizabeth C. Hunke, LANL
   106) !          C. M. Bitz, UW
   107) !
   108) ! !USES:
   109) !
   110) ! !INPUT/OUTPUT PARAMETERS:
   111) !
   112) !EOP
   113) !
   114)       integer (kind=int_kind) :: &
   115)            n    ! thickness category index
   116) 
   117)       real (kind=dbl_kind) :: &
   118)            cc1, cc2, cc3, & ! parameters for kcatbound = 0
   119)            x1           , &
   120)            rn           , & ! real(n)
   121)            rncat        , & ! real(ncat)
   122)            d1           , & ! parameters for kcatbound = 1 (m)
   123)            d2
   124) 
   125)       real (kind=dbl_kind), dimension(5) :: wmo5 ! data for wmo itd
   126)       real (kind=dbl_kind), dimension(6) :: wmo6 ! data for wmo itd
   127)       real (kind=dbl_kind), dimension(7) :: wmo7 ! data for wmo itd
   128) 
   129)       character(len=8) :: c_hinmax1,c_hinmax2
   130)       character(len=2) :: c_nc
   131) 
   132)       rncat = real(ncat, kind=dbl_kind)
   133)       d1 = 3.0_dbl_kind / rncat
   134)       d2 = 0.5_dbl_kind / rncat
   135) 
   136)       !-----------------------------------------------------------------
   137)       ! Choose category boundaries based on one of three options.
   138)       !
   139)       ! The first formula (kcatbound = 0) was used in Lipscomb (2001) 
   140)       !  and in CICE versions 3.0 and 3.1.
   141)       !
   142)       ! The second formula is more user-friendly in the sense that it
   143)       !  is easy to obtain round numbers for category boundaries:
   144)       !
   145)       !    H(n) = n * [d1 + d2*(n-1)] 
   146)       ! 
   147)       ! Default values are d1 = 300/ncat, d2 = 50/ncat.
   148)       ! For ncat = 5, boundaries in cm are 60, 140, 240, 360, which are 
   149)       !  close to the standard values given by the first formula.
   150)       ! For ncat = 10, boundaries in cm are 30, 70, 120, 180, 250, 330,
   151)       !  420, 520, 630.    
   152)       !
   153)       ! The third option provides support for World Meteorological
   154)       !  Organization classification based on thickness.  The full
   155)       !  WMO thickness distribution is used if ncat = 7;  if ncat=5 
   156)       !  or ncat = 6, some of the thinner categories are combined.
   157)       ! For ncat = 5,  boundaries are         30, 70, 120, 200, >200 cm.
   158)       ! For ncat = 6,  boundaries are     15, 30, 70, 120, 200, >200 cm.
   159)       ! For ncat = 7,  boundaries are 10, 15, 30, 70, 120, 200, >200 cm.
   160)       !-----------------------------------------------------------------
   161) 
   162)       if (kcatbound == 0) then   ! original scheme
   163) 
   164)          if (kitd == 1) then
   165)             ! linear remapping itd category limits
   166)             cc1 = c3/rncat
   167)             cc2 = c15*cc1
   168)             cc3 = c3
   169) 
   170)             hin_max(0) = c0     ! minimum ice thickness, m
   171)          else
   172)             ! delta function itd category limits
   173)             cc1 = max(1.1_dbl_kind/rncat,c1*hi_min)
   174)             cc2 = c25*cc1
   175)             cc3 = 2.25_dbl_kind
   176) 
   177)             ! hin_max(0) should not be zero
   178)             ! use some caution in making it less than 0.10
   179)             hin_max(0) = hi_min ! minimum ice thickness, m
   180)          endif                  ! kitd
   181) 
   182)          do n = 1, ncat
   183)             x1 = real(n-1,kind=dbl_kind) / rncat
   184)             hin_max(n) = hin_max(n-1) &
   185)                        + cc1 + cc2*(c1 + tanh(cc3*(x1-c1)))
   186)          enddo
   187) 
   188)       elseif (kcatbound == 1) then  ! new scheme
   189) 
   190)          hin_max(0) = c0
   191)          do n = 1, ncat
   192)             rn = real(n, kind=dbl_kind)
   193)             hin_max(n) = rn * (d1 + (rn-c1)*d2)
   194)          enddo
   195) 
   196)       elseif (kcatbound == 2) then  ! WMO standard
   197) 
   198)         if (ncat == 5) then
   199)          ! thinnest 3 categories combined
   200)          data wmo5 / 0.30_dbl_kind, 0.70_dbl_kind, &
   201)                     1.20_dbl_kind, 2.00_dbl_kind,  &
   202)                     999._dbl_kind  /
   203)          hin_max(0) = c0
   204)          do n = 1, ncat
   205)             hin_max(n) = wmo5(n)
   206)          enddo
   207)        elseif (ncat == 6) then
   208)          ! thinnest 2 categories combined
   209)          data wmo6 / 0.15_dbl_kind, &
   210)                     0.30_dbl_kind, 0.70_dbl_kind,  &
   211)                     1.20_dbl_kind, 2.00_dbl_kind,  &
   212)                     999._dbl_kind /
   213)          hin_max(0) = c0
   214)          do n = 1, ncat
   215)             hin_max(n) = wmo6(n)
   216)          enddo
   217)        elseif (ncat == 7) then
   218)          ! all thickness categories 
   219)          data wmo7 / 0.10_dbl_kind, 0.15_dbl_kind, &
   220)                     0.30_dbl_kind, 0.70_dbl_kind,  &
   221)                     1.20_dbl_kind, 2.00_dbl_kind,  &
   222)                     999._dbl_kind  /
   223)          hin_max(0) = c0
   224)          do n = 1, ncat
   225)             hin_max(n) = wmo7(n)
   226)          enddo
   227)        else
   228)          write (nu_diag,*) 'kcatbound=3 (WMO) must have ncat=5, 6 or 7'
   229)          stop
   230)        endif
   231) 
   232)       endif ! kcatbound
   233) 
   234)       if (my_task == master_task) then
   235)          write (nu_diag,*) ' '
   236)          write (nu_diag,*) 'hin_max(n-1) < Cat n < hin_max(n)'
   237)          do n = 1, ncat
   238)             write (nu_diag,*) hin_max(n-1),' < Cat ',n, ' < ',hin_max(n)
   239)             ! Write integer n to character string
   240)             write (c_nc, '(i2)') n    
   241) 
   242)             ! Write hin_max to character string
   243)             write (c_hinmax1, '(f5.3)') hin_max(n-1)
   244)             write (c_hinmax2, '(f5.3)') hin_max(n)
   245) 
   246)             ! Save character string to write to history file
   247)             c_hi_range(n)=c_hinmax1//'m < hi Cat '//c_nc//' < '// &
   248)                           c_hinmax2//'m'
   249)          enddo
   250)          write (nu_diag,*) ' '
   251)       endif
   252) 
   253)       !-----------------------------------------------------------------
   254)       ! vectors identifying first and last layer in each category
   255)       !-----------------------------------------------------------------
   256)       ilyr1(1) = 1                       ! if nilyr  = 4
   257)       ilyrn(1) = nilyr                   !   ilyr1 = { 1,5,9 }
   258)       do n = 2, ncat                     !   ilyrn = { 4,8,12} etc
   259)          ilyr1(n) = ilyrn(n-1) + 1
   260)          ilyrn(n) = ilyrn(n-1) + nilyr
   261)       enddo
   262) 
   263)       slyr1(1) = 1
   264)       slyrn(1) = nslyr
   265)       do n = 2, ncat
   266)          slyr1(n) = slyrn(n-1) + 1
   267)          slyrn(n) = slyrn(n-1) + nslyr
   268)       enddo
   269) 
   270)       end subroutine init_itd
   271)

   275) ! !IROUTINE: aggregate - aggregate ice state variables
   276) !
   277) ! !INTERFACE:
   278) !
   279)       subroutine aggregate (nx_block, ny_block, &
   280)                             aicen,    trcrn,    &
   281)                             vicen,    vsnon,    &
   282)                             eicen,    esnon,    &
   283)                             aice,     trcr,     &
   284)                             vice,     vsno,     &
   285)                             eice,     esno,     &
   286)                             aice0,    tmask,    &
   287)                             ntrcr,    trcr_depend)
   288) !
   289) ! !DESCRIPTION:
   290) !
   291) ! Aggregate ice state variables over thickness categories.
   292) !
   293) ! !REVISION HISTORY:
   294) !
   295) ! authors: C. M. Bitz, UW
   296) !          W. H. Lipscomb, LANL
   297) !
   298) ! !USES:
   299) !
   300) ! !INPUT/OUTPUT PARAMETERS:
   301) !
   302)       integer (kind=int_kind), intent(in) :: &
   303)          nx_block, ny_block, & ! block dimensions
   304)          ntrcr                 ! number of tracers in use
   305) 
   306)       real (kind=dbl_kind), dimension (nx_block,ny_block,ncat), &
   307)          intent(in) :: &
   308)          aicen , & ! concentration of ice
   309)          vicen , & ! volume per unit area of ice          (m)
   310)          vsnon     ! volume per unit area of snow         (m)
   311) 
   312)       real (kind=dbl_kind), dimension (nx_block,ny_block,max_ntrcr,ncat), &
   313)          intent(in) :: &
   314)          trcrn     ! ice tracers
   315) 
   316)       real (kind=dbl_kind), dimension (nx_block,ny_block,ntilyr), &
   317)          intent(in) :: &
   318)          eicen     ! energy of melting for each ice layer  (J/m^2)
   319) 
   320)       real (kind=dbl_kind), dimension (nx_block,ny_block,ntslyr), &
   321)          intent(in) :: &
   322)          esnon     ! energy of melting for each snow layer (J/m^2)
   323) 
   324)       logical (kind=log_kind), dimension (nx_block,ny_block), &
   325)          intent(in) :: &
   326)          tmask     ! land/boundary mask, thickness (T-cell)
   327) 
   328)       integer (kind=int_kind), dimension (max_ntrcr), intent(in) :: &
   329)          trcr_depend ! = 0 for aicen tracers, 1 for vicen, 2 for vsnon
   330) 
   331)       real (kind=dbl_kind), dimension (nx_block,ny_block),  &
   332)          intent(out) :: &
   333)          aice  , & ! concentration of ice
   334)          vice  , & ! volume per unit area of ice          (m)
   335)          vsno  , & ! volume per unit area of snow         (m)
   336)          eice  , & ! energy of melt. of ice           (J/m^2)
   337)          esno  , & ! energy of melt. of snow layer    (J/m^2)
   338)          aice0     ! concentration of open water
   339) 
   340)       real (kind=dbl_kind), dimension (nx_block,ny_block,max_ntrcr),  &
   341)          intent(out) :: &
   342)          trcr      ! ice tracers
   343) !
   344) !EOP
   345) !
   346)       integer (kind=int_kind) :: &
   347)         icells                ! number of ocean/ice cells
   348) 
   349)       integer (kind=int_kind), dimension (nx_block*ny_block) :: &
   350)         indxi, &              ! compressed indices in i/j directions
   351)         indxj
   352) 
   353)       integer (kind=int_kind) :: &
   354)         i, j, k, n, it, &
   355)         ij                    ! combined i/j horizontal index
   356) 
   357)       real (kind=dbl_kind), dimension (:,:), allocatable :: &
   358)         atrcr      ! sum of aicen*trcrn or vicen*trcrn or vsnon*trcrn
   359) 
   360)       !-----------------------------------------------------------------
   361)       ! Initialize
   362)       !-----------------------------------------------------------------
   363) 
   364)       icells = 0
   365)       do j = 1, ny_block
   366)       do i = 1, nx_block
   367)          if (tmask(i,j)) then
   368)             icells = icells + 1
   369)             indxi(icells) = i
   370)             indxj(icells) = j
   371)          endif                  ! tmask
   372) 
   373)          aice0(i,j) = c1
   374)          aice (i,j) = c0
   375)          vice (i,j) = c0
   376)          vsno (i,j) = c0
   377)          eice (i,j) = c0
   378)          esno (i,j) = c0
   379)       enddo
   380)       enddo
   381) 
   382) 
   383)       allocate (atrcr(icells,ntrcr))
   384) 
   385)       !-----------------------------------------------------------------
   386)       ! Aggregate
   387)       !-----------------------------------------------------------------
   388) 
   389)       atrcr(:,:) = c0
   390) 
   391)       do n = 1, ncat
   392) 
   393) !DIR$ CONCURRENT !Cray
   394) !cdir nodep      !NEC
   395) !ocl novrec      !Fujitsu
   396)          do ij = 1, icells
   397)             i = indxi(ij)
   398)             j = indxj(ij)
   399)             aice(i,j) = aice(i,j) + aicen(i,j,n)
   400)             vice(i,j) = vice(i,j) + vicen(i,j,n)
   401)             vsno(i,j) = vsno(i,j) + vsnon(i,j,n)
   402)          enddo                  ! ij
   403) 
   404)          do it = 1, ntrcr
   405)             if (trcr_depend(it) == 0) then  ! ice area tracer
   406) !DIR$ CONCURRENT !Cray
   407) !cdir nodep      !NEC
   408) !ocl novrec      !Fujitsu
   409)                do ij = 1, icells
   410)                   i = indxi(ij)
   411)                   j = indxj(ij)
   412)                   atrcr(ij,it) = atrcr(ij,it)  &
   413)                                 + trcrn(i,j,it,n)*aicen(i,j,n)
   414)                enddo            ! ij
   415) 
   416)             elseif (trcr_depend(it) == 1) then  ! ice volume tracer
   417) !DIR$ CONCURRENT !Cray
   418) !cdir nodep      !NEC
   419) !ocl novrec      !Fujitsu
   420)                do ij = 1, icells
   421)                   i = indxi(ij)
   422)                   j = indxj(ij)
   423)                   atrcr(ij,it) = atrcr(ij,it)  &
   424)                                 + trcrn(i,j,it,n)*vicen(i,j,n)
   425)                enddo            ! ij
   426) 
   427)             elseif (trcr_depend(it) ==2) then ! snow volume tracer
   428) 
   429) !DIR$ CONCURRENT !Cray
   430) !cdir nodep      !NEC
   431) !ocl novrec      !Fujitsu
   432)                do ij = 1, icells
   433)                   i = indxi(ij)
   434)                   j = indxj(ij)
   435)                   atrcr(ij,it) = atrcr(ij,it)  &
   436)                                 + trcrn(i,j,it,n)*vsnon(i,j,n)
   437)                enddo            ! ij
   438) 
   439)             endif               ! trcr_depend
   440)          enddo                  ! ntrcr
   441) 
   442)          do k = 1, nilyr
   443) !DIR$ CONCURRENT !Cray
   444) !cdir nodep      !NEC
   445) !ocl novrec      !Fujitsu
   446)             do ij = 1, icells
   447)                i = indxi(ij)
   448)                j = indxj(ij)
   449)                eice(i,j) = eice(i,j) + eicen(i,j,ilyr1(n)+k-1)
   450)             enddo
   451)          enddo                  ! nilyr
   452) 
   453)          do k = 1, nslyr
   454) !DIR$ CONCURRENT !Cray
   455) !cdir nodep      !NEC
   456) !ocl novrec      !Fujitsu
   457)             do ij = 1, icells
   458)                i = indxi(ij)
   459)                j = indxj(ij)
   460)                esno(i,j) = esno(i,j) + esnon(i,j,slyr1(n)+k-1)
   461)             enddo
   462)          enddo                  ! nslyr
   463) 
   464)       enddo                     ! ncat
   465) 
   466)       ! Open water fraction
   467) 
   468)       do ij = 1, icells
   469)          i = indxi(ij)
   470)          j = indxj(ij)
   471)          aice0(i,j) = max (c1 - aice(i,j), c0)
   472)       enddo                     ! ij
   473) 
   474)       ! Tracers
   475) 
   476)       call compute_tracers (nx_block,     ny_block,   &
   477)                             icells,   indxi,   indxj, &
   478)                             ntrcr,    trcr_depend,    &
   479)                             atrcr(:,:), aice(:,:),    &
   480)                             vice (:,:),   vsno(:,:),  &
   481)                             trcr(:,:,:))
   482) 
   483)       deallocate (atrcr)
   484) 
   485)       end subroutine aggregate

   494)       subroutine aggregate_area (nx_block, ny_block,        &
   495)                                  aicen,    aice,     aice0)
   496) !
   497) ! !DESCRIPTION:
   498) !
   499) ! Aggregate ice area (but not other state variables) over thickness 
   500) ! categories.
   501) !
   502) ! !REVISION HISTORY:
   503) !
   504) ! authors: William H. Lipscomb, LANL
   505) !          modified Jan 2004 by Clifford Chen, Fujitsu
   506) !
   507) ! !USES:
   508) !
   509) ! !INPUT/OUTPUT PARAMETERS:
   510) !
   511)       integer (kind=int_kind), intent(in) :: &
   512)          nx_block, ny_block  ! block dimensions
   513) 
   514)       real (kind=dbl_kind), dimension (:,:,:), intent(in) :: &
   515)          aicen     ! concentration of ice
   516) 
   517)       real (kind=dbl_kind), dimension (:,:), intent(inout) :: &
   518)          aice, &   ! concentration of ice
   519)          aice0     ! concentration of open water
   520) !
   521) !EOP
   522) !
   523)       integer (kind=int_kind) :: i, j, n
   524) 
   525)       !-----------------------------------------------------------------
   526)       ! Aggregate
   527)       !-----------------------------------------------------------------
   528) 
   529)       aice(:,:) = c0
   530) 
   531)       do n = 1, ncat
   532)          do j = 1, ny_block
   533)          do i = 1, nx_block
   534)             aice(i,j) = aice(i,j) + aicen(i,j,n)
   535)          enddo                  ! i
   536)          enddo                  ! j
   537)       enddo                     ! n
   538) 
   539)       do j = 1, ny_block
   540)       do i = 1, nx_block
   541) 
   542)          ! open water fraction
   543)          aice0(i,j) = max (c1 - aice(i,j), c0)
   544) 
   545)       enddo                     ! i
   546)       enddo                     ! j
   547) 
   548)       end subroutine aggregate_area

   579)          nx_block, ny_block, & ! block dimensions
   580)          icells            , & ! number of grid cells with ice

   591)          intent(inout) :: &
   592)          aicen , & ! concentration of ice
   593)          vicen , & ! volume per unit area of ice           (m)
   594)          vsnon     ! volume per unit area of snow          (m)
   595) 
   596)       real (kind=dbl_kind), dimension (nx_block,ny_block,max_ntrcr,ncat), &
   597)          intent(inout) :: &
   598)          trcrn     ! ice tracers

   601)          intent(inout) :: &

   603)

   611)       integer (kind=int_kind), intent(out) :: &
   612)          istop, jstop    ! indices of grid cell where model aborts

   616)       integer (kind=int_kind) :: &
   617)          i,j          , & ! horizontal indices
   618)          n            , & ! category index
   619)          ij                ! combined horizontal index
   620) 
   621)       logical (kind=log_kind) :: &

   623)

   625)          donor              ! donor category index
   626) 
   627)       real (kind=dbl_kind), dimension (icells,ncat) :: &
   628)          daice          , & ! ice area transferred
   629)          dvice          , & ! ice volume transferred

   631) 
   632)       !-----------------------------------------------------------------
   633)       ! Initialize
   634)       !-----------------------------------------------------------------
   635) 
   636)       l_stop = .false.
   637)       istop = 0
   638)       jstop = 0
   639) 
   640)       do n = 1, ncat
   641)          do ij = 1, icells       ! aice(i,j) > puny
   642)             i = indxi(ij)
   643)             j = indxj(ij)
   644) 
   645)             donor(ij,n) = 0
   646)             daice(ij,n) = c0
   647)             dvice(ij,n) = c0
   648) 
   649)       !-----------------------------------------------------------------
   650)       ! Compute ice thickness.
   651)       !-----------------------------------------------------------------
   652)             if (aicen(i,j,n) > puny) then
   653)                hicen(ij,n) = vicen(i,j,n) / aicen(i,j,n)
   654)             else
   655)                hicen(ij,n) = c0
   656)             endif
   657)          enddo                  ! ij
   658)       enddo                     ! n
   659) 
   660)       !-----------------------------------------------------------------
   661)       ! make sure thickness of cat 1 is at least hin_max(0)
   662)       !-----------------------------------------------------------------
   663)       do ij = 1, icells       ! aice(i,j) > puny
   664)          i = indxi(ij)
   665)          j = indxj(ij)
   666) 
   667)          if (aicen(i,j,1) > puny) then
   668)             if (hicen(ij,1) <= hin_max(0) .and. hin_max(0) > c0 ) then
   669)                aicen(i,j,1) = vicen(i,j,1) / hin_max(0)
   670)                hicen(ij,1) = hin_max(0)
   671)             endif
   672)          endif
   673)       enddo                     ! ij
   674) 
   675)       !-----------------------------------------------------------------
   676)       ! If a category thickness is not in bounds, shift the
   677)       ! entire area, volume, and energy to the neighboring category
   678)       !-----------------------------------------------------------------
   679) 
   680)       !-----------------------------------------------------------------
   681)       ! Move thin categories up
   682)       !-----------------------------------------------------------------
   683) 
   684)       do n = 1, ncat-1          ! loop over category boundaries
   685) 
   686)       !-----------------------------------------------------------------
   687)       ! identify thicknesses that are too big
   688)       !-----------------------------------------------------------------
   689)          shiftflag = .false.
   690)          do ij = 1, icells       ! aice(i,j) > puny
   691)             i = indxi(ij)
   692)             j = indxj(ij)
   693) 
   694)             if (aicen(i,j,n) > puny .and. &
   695)                 hicen(ij,n) > hin_max(n)) then
   696)                shiftflag = .true.
   697)                donor(ij,n) = n
   698)                daice(ij,n) = aicen(i,j,n)
   699)                dvice(ij,n) = vicen(i,j,n)
   700)             endif
   701)          enddo                  ! ij
   702) 
   703)          if (shiftflag) then
   704) 
   705)       !-----------------------------------------------------------------
   706)       ! shift ice between categories
   707)       !-----------------------------------------------------------------
   708)             call shift_ice (nx_block, ny_block,    &
   709)                             indxi,    indxj,       &
   710)                             icells,                &
   711)                             ntrcr,    trcr_depend, &
   712)                             aicen,    trcrn,       &
   713)                             vicen,    vsnon,       &
   714)                             eicen,    esnon,       &
   715)                             hicen,    donor,       &
   716)                             daice,    dvice,       &
   717)                             l_stop,                &
   718)                             istop,    jstop)
   719) 
   720)             if (l_stop) return
   721) 
   722)       !-----------------------------------------------------------------
   723)       ! reset shift parameters
   724)       !-----------------------------------------------------------------
   725) 
   726)          do ij = 1, icells       ! aice(i,j) > puny
   727)             donor(ij,n) = 0
   728)             daice(ij,n) = c0
   729)             dvice(ij,n) = c0
   730)          enddo
   731) 
   732)          endif                  ! shiftflag
   733) 
   734)       enddo                     ! n
   735) 
   736)       !-----------------------------------------------------------------
   737)       ! Move thick categories down
   738)       !-----------------------------------------------------------------
   739) 
   740)       do n = ncat-1, 1, -1      ! loop over category boundaries
   741) 
   742)       !-----------------------------------------------------------------
   743)       ! identify thicknesses that are too small
   744)       !-----------------------------------------------------------------
   745)          shiftflag = .false.
   746)          do ij = 1, icells       ! aice(i,j) > puny
   747)             i = indxi(ij)
   748)             j = indxj(ij)
   749) 
   750)             if (aicen(i,j,n+1) > puny .and. &
   751)                 hicen(ij,n+1) <= hin_max(n)) then
   752)                shiftflag = .true.
   753)                donor(ij,n) = n+1
   754)                daice(ij,n) = aicen(i,j,n+1)
   755)                dvice(ij,n) = vicen(i,j,n+1)
   756)             endif
   757)          enddo                  ! ij
   758) 
   759)          if (shiftflag) then
   760) 
   761)       !-----------------------------------------------------------------
   762)       ! shift ice between categories
   763)       !-----------------------------------------------------------------
   764)             call shift_ice (nx_block, ny_block,    &
   765)                             indxi,    indxj,       &
   766)                             icells,                &
   767)                             ntrcr,    trcr_depend, &
   768)                             aicen,    trcrn,       &
   769)                             vicen,    vsnon,       &
   770)                             eicen,    esnon,       &
   771)                             hicen,    donor,       &
   772)                             daice,    dvice,       &
   773)                             l_stop,                &
   774)                             istop,    jstop)
   775) 
   776)             if (l_stop) return
   777) 
   778)       !-----------------------------------------------------------------
   779)       ! reset shift parameters
   780)       !-----------------------------------------------------------------
   781) 
   782)          do ij = 1, icells       ! aice(i,j) > puny
   783)             donor(ij,n) = 0
   784)             daice(ij,n) = c0
   785)             dvice(ij,n) = c0
   786)          enddo
   787) 
   788)          endif                  ! shiftflag
   789) 
   790)       enddo                     ! n
   791) 
   792) 
   793)       end subroutine rebin

   798) ! !IROUTINE: reduce_area - reduce area when ice melts for special case ncat=1
   799) !
   800) ! !INTERFACE:
   801) !
   802)       subroutine reduce_area (nx_block, ny_block, &
   803)                               ilo, ihi, jlo, jhi, &
   804)                               tmask,              &
   805)                               aicen,     vicen,   &
   806)                               aicen_init,vicen_init)
   807) !
   808) ! !DESCRIPTION:
   809) !
   810) ! Reduce area when ice melts for special case of ncat=1
   811) !
   812) ! Use CSM 1.0-like method of reducing ice area
   813) ! when melting occurs: assume only half the ice volume
   814) ! change goes to thickness decrease, the other half
   815) ! to reduction in ice fraction
   816) !
   817) ! !REVISION HISTORY:
   818) !
   819) ! authors: C. M. Bitz, UW
   820) ! modified by: Elizabeth C. Hunke, LANL
   821) !
   822) ! !USES:
   823) !
   824) ! !INPUT/OUTPUT PARAMETERS:
   825) !
   826)       integer (kind=int_kind), intent(in) :: &
   827)          nx_block, ny_block, & ! block dimensions
   828)          ilo,ihi,jlo,jhi       ! beginning and end of physical domain
   829) 
   830)       logical (kind=log_kind), dimension (nx_block,ny_block), &
   831)          intent(in) :: &
   832)          tmask     ! land/boundary mask, thickness (T-cell)
   833) 
   834)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   835)          intent(inout) :: &
   836)          aicen , & ! concentration of ice
   837)          vicen     ! volume per unit area of ice          (m)
   838) 
   839)       real (kind=dbl_kind), dimension(nx_block,ny_block), intent(in) :: &
   840)          aicen_init, & ! old ice area for category 1 (m)
   841)          vicen_init    ! old ice volume for category 1 (m)
   842) !
   843) !EOP
   844) !
   845)       integer (kind=int_kind) :: &
   846)          i, j        ! horizontal indices
   847) 
   848)       real (kind=dbl_kind) :: &
   849)          hi0     , & ! initial hi
   850)          hi1     , & ! current hi
   851)          dhi         ! hi1 - hi0
   852) 
   853)       do j = jlo, jhi
   854)       do i = ilo, ihi
   855)          if (tmask(i,j)) then
   856) 
   857)             hi0 = c0
   858)             if (aicen_init(i,j) > c0) &
   859)                 hi0 = vicen_init(i,j) / aicen_init(i,j)
   860) 
   861)             hi1 = c0
   862)             if (aicen(i,j) > c0) &
   863)                 hi1 = vicen(i,j) / aicen(i,j)
   864) 
   865)             ! make sure thickness of cat 1 is at least hin_max(0)
   866)             if (hi1 <= hin_max(0) .and. hin_max(0) > c0 ) then
   867)                aicen(i,j) = vicen(i,j) / hin_max(0)
   868)                hi1 = hin_max(0)
   869)             endif
   870) 
   871)             if (aicen(i,j) > c0) then
   872)                dhi = hi1 - hi0
   873)                if (dhi < c0) then
   874)                   hi1  = vicen(i,j) / aicen(i,j)
   875)                   aicen(i,j) = c2 * vicen(i,j) / (hi1 + hi0)
   876)                endif
   877)             endif
   878) 
   879)          endif                  ! tmask
   880)       enddo                     ! i
   881)       enddo                     ! j
   882) 
   883)       end subroutine reduce_area

   893)                             indxi,    indxj,       &

   896)                             aicen,    trcrn,       &

   903) !
   904) ! !DESCRIPTION:

   908) !
   909) ! !REVISION HISTORY:
   910) !

   918)       integer (kind=int_kind), intent(in) :: &

   921)          ntrcr                 ! number of tracers in use

   925)          indxi             , & ! compressed indices in i/j directions

   927) 
   928)       integer (kind=int_kind), dimension (max_ntrcr), intent(in) :: &
   929)          trcr_depend ! = 0 for aicen tracers, 1 for vicen, 2 for vsnon
   930) 
   931)       real (kind=dbl_kind), dimension (nx_block,ny_block,ncat), &

   935)          vsnon     ! volume per unit area of snow         (m)

   946)          intent(inout) :: &
   947)          esnon     ! energy of melting for each snow layer (J/m^2)
   948) 
   949)       ! NOTE: Third index of donor, daice, dvice should be ncat-1,
   950)       !       except that compilers would have trouble when ncat = 1 
   951)       integer (kind=int_kind), dimension(icells,ncat), &
   952)          intent(in) :: &
   953)          donor             ! donor category index
   954) 
   955)       real (kind=dbl_kind), dimension(icells,ncat), &
   956)            intent(inout) :: &
   957)          daice         , & ! ice area transferred across boundary
   958)          dvice         , & ! ice volume transferred across boundary
   959)          hicen             ! ice thickness for each cat        (m)
   960) 
   961)       logical (kind=log_kind), intent(out) :: &
   962)          l_stop    ! if true, abort on return
   963) 
   964)       integer (kind=int_kind), intent(out) :: &
   965)          istop, jstop    ! indices of grid cell where model aborts
   966) !
   967) !EOP
   968) !
   969)       integer (kind=int_kind) :: &
   970)          i, j, m       , & ! horizontal indices
   971)          n             , & ! thickness category index
   972)          nr            , & ! receiver category
   973)          nd            , & ! donor category
   974)          k             , & ! ice layer index
   975)          it            , & ! tracer index
   976)          ilo,ihi,jlo,jhi   ! beginning and end of physical domain

   978)       real (kind=dbl_kind), dimension(icells,max_ntrcr,ncat) :: &
   979)          atrcrn            ! aicen*trcrn
   980) 
   981)       real (kind=dbl_kind) :: &
   982)          dvsnow        , & ! snow volume transferred
   983)          desnow        , & ! snow energy transferred

   985)          datrcr            ! aicen*train transferred
   986)

   990)         indxij

   992)       integer (kind=int_kind) :: &
   993)         ishift      , & ! number of cells with ice to transfer
   994)         ij              ! combined i/j horizontal index
   995) 
   996)       logical (kind=log_kind) :: &
   997)         daice_negative     , & ! true if daice < -puny
   998)         dvice_negative     , & ! true if dvice < -puny
   999)         daice_greater_aicen, & ! true if daice > aicen
  1000)         dvice_greater_vicen    ! true if dvice > vicen
  1001) 
  1002)       real (kind=dbl_kind), dimension (nx_block,ny_block) :: &
  1003)          worka, &
  1004)          workb
  1005) 
  1006)       !-----------------------------------------------------------------
  1007)       ! Initialize
  1008)       !-----------------------------------------------------------------
  1009) 
  1010)       l_stop = .false.
  1011)       istop = 0
  1012)       jstop = 0
  1013) 
  1014)       worka(:,:) = c0
  1015)       workb(:,:) = c0
  1016) 
  1017)       !-----------------------------------------------------------------
  1018)       ! Define variables equal to aicen*trcrn, vicen*trcrn, vsnon*trcrn
  1019)       !-----------------------------------------------------------------
  1020) 
  1021)       do n = 1, ncat
  1022)          do it = 1, ntrcr
  1023)             if (trcr_depend(it) == 0) then ! ice area tracer
  1024)                do ij = 1, icells
  1025)                   i = indxi(ij)
  1026)                   j = indxj(ij)
  1027)                   atrcrn(ij,it,n) = aicen(i,j,n)*trcrn(i,j,it,n)

  1030)                do ij = 1, icells
  1031)                   i = indxi(ij)
  1032)                   j = indxj(ij)
  1033)                   atrcrn(ij,it,n) = vicen(i,j,n)*trcrn(i,j,it,n)

  1037)                   i = indxi(ij)
  1038)                   j = indxj(ij)
  1039)                   atrcrn(ij,it,n) = vsnon(i,j,n)*trcrn(i,j,it,n)
  1040)                enddo
  1041)             endif
  1042)          enddo
  1043)       enddo

  1047)       !-----------------------------------------------------------------
  1048) 
  1049)       do n = 1, ncat-1
  1050) 
  1051)          daice_negative = .false.
  1052)          dvice_negative = .false.
  1053)          daice_greater_aicen = .false.
  1054)          dvice_greater_vicen = .false.
  1055) 
  1056) 
  1057)          do ij = 1, icells
  1058)             i = indxi(ij)
  1059)             j = indxj(ij)
  1060)

  1065)                   if (daice(ij,n) > -puny*aicen(i,j,nd)) then
  1066)                      daice(ij,n) = c0 ! shift no ice
  1067)                      dvice(ij,n) = c0
  1068)                   else
  1069)                      daice_negative = .true.
  1070)                   endif
  1071)                endif
  1072)          
  1073)                if (dvice(ij,n) < c0) then
  1074)                   if (dvice(ij,n) > -puny*vicen(i,j,nd)) then   
  1075)                      daice(ij,n) = c0 ! shift no ice
  1076)                      dvice(ij,n) = c0
  1077)                   else

  1079)                   endif
  1080)                endif

  1085)                      dvice(ij,n) = vicen(i,j,nd)
  1086)                   else
  1087)                      daice_greater_aicen = .true.
  1088)                   endif
  1089)                endif    
  1090) 
  1091)                if (dvice(ij,n) > vicen(i,j,nd)*(c1-puny)) then
  1092)                   if (dvice(ij,n) < vicen(i,j,nd)*(c1+puny)) then
  1093)                      daice(ij,n) = aicen(i,j,nd)
  1094)                      dvice(ij,n) = vicen(i,j,nd)
  1095)                   else
  1096)                      dvice_greater_vicen = .true.
  1097)                   endif
  1098)                endif
  1099) 
  1100)             endif               ! donor > 0
  1101)          enddo                  ! ij
  1102) 
  1103)       !-----------------------------------------------------------------
  1104)       ! error messages
  1105)       !-----------------------------------------------------------------
  1106) 
  1107)          if (daice_negative) then
  1108)          do ij = 1, icells
  1109)             i = indxi(ij)
  1110)             j = indxj(ij)
  1111) 
  1112)                if (donor(ij,n) > 0 .and.  &
  1113)                    daice(ij,n) <= -puny*aicen(i,j,nd)) then
  1114)                   write(nu_diag,*) ' '
  1115)                   write(nu_diag,*) 'shift_ice: negative daice'
  1116)                   write(nu_diag,*) 'i, j:', i, j
  1117)                   write(nu_diag,*) 'boundary, donor cat:', n, nd
  1118)                   write(nu_diag,*) 'daice =', daice(ij,n)
  1119)                   write(nu_diag,*) 'dvice =', dvice(ij,n)
  1120)                   l_stop = .true.
  1121)                   istop = i
  1122)                   jstop = j
  1123)                endif
  1124)             enddo
  1125)          endif
  1126)          if (l_stop) return
  1127) 
  1128)          if (dvice_negative) then
  1129)          do ij = 1, icells
  1130)             i = indxi(ij)
  1131)             j = indxj(ij)
  1132) 
  1133)                if (donor(ij,n) > 0 .and.  &
  1134)                    dvice(ij,n) <= -puny*vicen(i,j,nd)) then
  1135)                   write(nu_diag,*) ' '
  1136)                   write(nu_diag,*) 'shift_ice: negative dvice'
  1137)                   write(nu_diag,*) 'i, j:', i, j
  1138)                   write(nu_diag,*) 'boundary, donor cat:', n, nd
  1139)                   write(nu_diag,*) 'daice =', daice(ij,n)
  1140)                   write(nu_diag,*) 'dvice =', dvice(ij,n)

  1806)                                dt,           ntrcr,          &

  1808)                                aicen(:,:,:), trcrn(:,:,:,:), &
  1809)                                vicen(:,:,:), vsnon(:,:,:),   &
  1810)                                eicen(:,:,:), esnon(:,:,:),   &

  1813)                                tr_aero,                      &
  1814)                                l_stop,                       &

  1816)          if (l_stop) return
  1817)       endif   ! l_limit_aice
  1818) 
  1819)     !-------------------------------------------------------------------
  1820)     ! Update ice-ocean fluxes for strict conservation
  1821)     !-------------------------------------------------------------------
  1822) 
  1823)       if (present(fresh)) &
  1824)            fresh     (:,:) = fresh(:,:)      + dfresh(:,:) 
  1825)       if (present(fsalt)) &
  1826)            fsalt     (:,:) = fsalt(:,:)      + dfsalt(:,:)
  1827)       if (present(fhocn)) &

  1829)       if (present(fsoot)) &
  1830)            fsoot   (:,:,:) = fsoot(:,:,:)    + dfsoot(:,:,:)
  1831) 
  1832)       !----------------------------------------------------------------
  1833)       ! If using zero-layer model (no heat capacity), check that the 
  1834)       ! energy of snow and ice is correct. 
  1835)       !----------------------------------------------------------------
  1836) 
  1837)       if (.not. heat_capacity) then
  1838) 
  1839)          call zerolayer_check(nx_block,    ny_block,   &
  1840)                               icells,  indxi,   indxj, &
  1841)                               aicen,                   &
  1842)                               vicen,       vsnon,      &
  1843)                               eicen,       esnon,      &
  1844)                               l_stop,                  &
  1845)                               istop,       jstop)

  1847)       endif

  1860)                                   dt,       ntrcr,    &
  1861)                                   aice,     aice0,    &

  1889)          ntrcr                 ! number of tracers in use

  2188)          if (tr_aero) then

  2203)
