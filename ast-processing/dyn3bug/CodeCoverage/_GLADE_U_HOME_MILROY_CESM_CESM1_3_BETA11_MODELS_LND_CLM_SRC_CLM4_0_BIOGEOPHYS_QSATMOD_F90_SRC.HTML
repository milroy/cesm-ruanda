    51)     real(r8), intent(out) :: esdT     ! d(es)/d(T)

    73)     real(r8), parameter :: a0 =  6.11213476_r8
    74)     real(r8), parameter :: a1 =  0.444007856_r8
    75)     real(r8), parameter :: a2 =  0.143064234e-01_r8
    76)     real(r8), parameter :: a3 =  0.264461437e-03_r8
    77)     real(r8), parameter :: a4 =  0.305903558e-05_r8
    78)     real(r8), parameter :: a5 =  0.196237241e-07_r8
    79)     real(r8), parameter :: a6 =  0.892344772e-10_r8
    80)     real(r8), parameter :: a7 = -0.373208410e-12_r8

    87)     real(r8), parameter :: b2 =  0.794683137e-03_r8
    88)     real(r8), parameter :: b3 =  0.121211669e-04_r8

    92)     real(r8), parameter :: b7 = -0.114596802e-13_r8

    95) ! For ice (temperature range -75C-0C)
    96) !
    97)     real(r8), parameter :: c0 =  6.11123516_r8

    99)     real(r8), parameter :: c2 =  0.188369801e-01_r8
   100)     real(r8), parameter :: c3 =  0.420547422e-03_r8
   101)     real(r8), parameter :: c4 =  0.614396778e-05_r8
   102)     real(r8), parameter :: c5 =  0.602780717e-07_r8
   103)     real(r8), parameter :: c6 =  0.387940929e-09_r8
   104)     real(r8), parameter :: c7 =  0.149436277e-11_r8
