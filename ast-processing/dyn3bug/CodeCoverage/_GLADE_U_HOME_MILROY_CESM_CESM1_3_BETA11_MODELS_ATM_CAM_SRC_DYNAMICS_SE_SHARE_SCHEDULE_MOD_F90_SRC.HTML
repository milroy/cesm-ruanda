    64)   subroutine genEdgeSched(elem, PartNumber,LSchedule,MetaVertex)
    65)     use element_mod, only : element_t
    66)     use metagraph_mod, only : metavertex_t

    93)     integer			  :: iSched
    94)     logical, parameter            :: VerbosePrint=.FALSE.

    99) 
   100)     nSched=SIZE(schedule)
   101)     ! ================================================
   102)     ! allocate some arrays for the call to MPI_gatherv
   103)     ! ================================================
   104) 
   105)     MinNelemd = nelem
   106)     MaxNelemd = 0
   107)     ! =====================================================
   108)     ! It looks like this is only used in this routine...
   109)     ! so no need to put it in the schedule data-structure
   110)     ! =====================================================
   111)     allocate(Global2Local(nelem))
   112)     if(Debug) write(iulog,*)'genEdgeSched: point #1'
   113)     iSched = PartNumber
   114) 
   115)     nelemd0 = MetaVertex%nmembers
   116)     if(VerbosePrint) then
   117)        if(iam .eq. 1)  write(iulog,*)'genEdgeSched: Part # ',i,' has ',nelemd0, ' elements '
   118)     endif
   119)     MaxNelemd = AMAX0(MaxNelemd,nelemd0)
   120)     MinNelemd = AMIN0(MinNelemd,nelemd0)
   121)     if(Debug) write(iulog,*)'genEdgeSched: point #2'
   122) 
   123)     if(Debug) write(iulog,*)'genEdgeSched: point #3'
   124)     LSchedule%ncycles = MetaVertex%nedges
   125)     LSchedule%nelemd  = nelemd0
   126)     if(Debug) write(iulog,*)'genEdgeSched: point #4'
   127) 
   128)     !  Note the minus one is for the internal node
   129)     nedges = MetaVertex%nedges
   130)     if(2*(nedges/2) .eq. nedges) then
   131)        nedges = nedges/2
   132)     else
   133)        nedges = (nedges-1)/2
   134)     endif
   135)     LSchedule%nSendCycles = nedges
   136)     LSchedule%nRecvCycles = nedges
   137)     if(Debug) write(iulog,*)'genEdgeSched: point #5'
   138) 
   139)     ! Temporary array to calculate the Buffer Slot
   140)     allocate(tmpP(2,nedges+1))
   141)     allocate(tmpP_ghost(2,nedges+1))
   142) 
   143)     tmpP(1,:) = -1
   144)     tmpP(2,:) = 0
   145)     tmpP_ghost(1,:) = -1
   146)     tmpP_ghost(2,:) = 0
   147) 
   148)     !  Allocate all the cycle structures
   149)     allocate(LSchedule%SendCycle(nedges))
   150)     allocate(LSchedule%RecvCycle(nedges))
   151)     allocate(LSchedule%MoveCycle(1))
   152)     if(Debug) write(iulog,*)'genEdgeSched: point #6'
   153) 
   154)     !==================================================================
   155)     !  Allocate and initalized the index translation arrays
   156)     Global2Local = -1
   157)     allocate(LSchedule%Local2Global(nelemd0))
   158)     if(Debug) write(iulog,*)'genEdgeSched: point #7'
   159) 
   160)     do il=1,nelemd0
   161)        ig     = MetaVertex%members(il)%number
   162)        Global2Local(ig)=il
   163)        LSchedule%Local2Global(il)=ig
   164) #ifndef _PREDICT
   165)        elem(il)%desc%putmapP=-1
   166)        elem(il)%desc%getmapP=-1
   167)        elem(il)%desc%putmapP_ghost=-1
   168)        elem(il)%desc%getmapP_ghost=-1
   169)        elem(il)%desc%reverse = .FALSE.
   170) #endif
   171)     enddo
   172)     !==================================================================
   173)     if(Debug) write(iulog,*)'genEdgeSched: point #8'
   174) 
   175) 
   176) 
   177)     total_length = 0
   178)     ncycle = LSchedule%ncycles
   179)     is=1
   180)     ir=1
   181)     do j=1,ncycle
   182)        lengthP     =  MetaVertex%edges(j)%wgtP
   183)        lengthP_ghost     =  MetaVertex%edges(j)%wgtP_ghost
   184) 
   185)        if((MetaVertex%edges(j)%HeadVertex == PartNumber) .AND. &
   186)             (MetaVertex%edges(j)%TailVertex == PartNumber)) then
   187)           inbr                            = PartNumber
   188)           if(Debug) write(iulog,*)'genEdgeSched: point #9', iam
   189)           LSchedule%MoveCycle%ptrP         = FindBufferSlot(inbr,lengthP,tmpP)
   190)           LSchedule%MoveCycle%ptrP_ghost   = FindBufferSlot(inbr,lengthP_ghost,tmpP_ghost)
   191)           call SetCycle(elem, LSchedule,LSchedule%MoveCycle(1),MetaVertex%edges(j))
   192)           if(Debug) write(iulog,*)'genEdgeSched: point #10',iam
   193)        else if (MetaVertex%edges(j)%TailVertex == PartNumber) then
   194)           inbr                            = MetaVertex%edges(j)%HeadVertex
   195)           if(Debug) write(iulog,*)'genEdgeSched: point #11', iam
   196)           LSchedule%SendCycle(is)%ptrP     = FindBufferSlot(inbr,lengthP,tmpP)
   197)           LSchedule%SendCycle(is)%ptrP_ghost= FindBufferSlot(inbr,lengthP_ghost,tmpP_ghost)
   198)           call SetCycle(elem, LSchedule,LSchedule%SendCycle(is),MetaVertex%edges(j))
   199)           if(Debug) write(iulog,*)'genEdgeSched: point #12',iam
   200)           is = is+1
   201)        else if (MetaVertex%edges(j)%HeadVertex == PartNumber) then
   202)           inbr                            = MetaVertex%edges(j)%TailVertex
   203)           if(Debug) write(iulog,*)'genEdgeSched: point #13',iam
   204)           LSchedule%RecvCycle(ir)%ptrP     = FindBufferSlot(inbr,lengthP,tmpP)
   205)           LSchedule%RecvCycle(ir)%ptrP_ghost= FindBufferSlot(inbr,lengthP_ghost,tmpP_ghost)
   206)           call SetCycle(elem, LSchedule,LSchedule%RecvCycle(ir),MetaVertex%edges(j))
   207)           if(Debug) write(iulog,*)'genEdgeSched: point #14',iam
   208)           ir = ir+1
   209)        endif
   210)     enddo
   211) 
   212)     deallocate(tmpP)
   213)     deallocate(tmpP_ghost)
   214) 
   215) 
   216) 
   217)     do ie=1,nelemd0
   218)        ! compute number of neighbers for each element
   219)        elem(ie)%desc%actual_neigh_edges=0
   220)        do i=1,max_neigh_edges
   221)           if (elem(ie)%desc%globalID(i)>0) then
   222)              elem(ie)%desc%actual_neigh_edges=elem(ie)%desc%actual_neigh_edges+1
   223)           endif
   224)        enddo
   225) 
   226)        ! normally, we loop over max_neigh_edges, checking if there is an edge
   227)        ! let's create a mapping so that we can loop over actual_neigh_edges
   228)        ! sort in REVERSE global id order (so the ones with globalID=0 are last)
   229)        do l1 = 1,max_neigh_edges-1
   230)           do l2=l1+1,max_neigh_edges
   231)              l1id=elem(ie)%desc%loc2buf(l1)
   232)              l2id=elem(ie)%desc%loc2buf(l2)
   233)              if (elem(ie)%desc%globalID(l2id) > elem(ie)%desc%globalID(l1id)) then
   234)                 ! swap index:
   235)                 l1id=elem(ie)%desc%loc2buf(l2)
   236)                 elem(ie)%desc%loc2buf(l2)=elem(ie)%desc%loc2buf(l1)
   237)                 elem(ie)%desc%loc2buf(l1)=l1id
   238)              endif
   239)           enddo
   240)        enddo
   241) 
   242)        
   243) 
   244) 
   245)        elem(ie)%vertex     = MetaVertex%members(ie)
   246)        ig                  = MetaVertex%members(ie)%number
   247)        elem(ie)%GlobalId   = ig
   248)        elem(ie)%LocalId    = ie  
   249) #if 0
   250)        call LLInsertEdge(eroot,ig,jmd)
   251)        !DBG write(iulog,*)'After call to LLInsertEdge in schedule: ie,ig ',ie,ig,jmd
   252) #endif
   253)     enddo
   254) 
   255)     deallocate(Global2Local)
   256)     !S-JMD call CheckSchedule()
   257) #ifdef _MPI
   258)     !================================================================
   259)     !     Allocate a couple of structures for bndry_exchange
   260)     !        done here to remove it from the critical path
   261)     !================================================================
   262) 
   263)     nComPoints=0
   264) 
   265)     nSend = nedges
   266)     nRecv = nedges
   267)     allocate(Rrequest(nRecv))
   268)     allocate(Srequest(nSend))
   269)     allocate(status(MPI_STATUS_SIZE,nRecv))
   270) 
   271)     !===============================================================
   272)     !   Number of communication points ... to be used later to
   273)     !    setup the size of the communication buffer for MPI_Ibsend
   274)     !===============================================================
   275)     do icycle=1,nSend
   276)        nComPoints = nComPoints + LSchedule%SendCycle(icycle)%lengthP
   277)     enddo
   278)     nPackPoints = nComPoints + LSchedule%MoveCycle(1)%lengthP
   279) 
   280) 
   281)     !   nbuf = 4*(nv+1)*nelemd*8*4*nlev
   282)     !   write(iulog,*)'before call to allocate(combuffer) ',nbuf
   283)     !   allocate(combuffer(nbuf))
   284)     !   write(iulog,*)'IAM: ',iSched,'Before call to MPI_Buffer_Attach '
   285)     !   call MPI_Buffer_Attach(combuffer,nbuf,ierr)
   286)     !   write(iulog,*)'IAM: ',iSched,'After call to MPI_Buffer_Attach '
   287) 
   288) #endif
   289) #ifdef DEBUGPART
   290)     call haltmp("genEdgeSched: Just testing the partitioning algorithms")
   291) #endif
   292)   end subroutine genEdgeSched

   792)   subroutine fooCalc(Schedule,bytes_per_point,offnode,onnode,count,Time)
   793)     use kinds, only : real_kind
   794)     use perfmodel_mod, only : commtime_t, perf_t ! _EXTERNAL
   795)     use flops_mod, only : min_number,max_number, avg_number, & ! _EXTERNAL
   796)          min_volume,max_volume, avg_volume, &
   797)          min_message,max_message, avg_message, &
   798)          LocalComVolume,TotalComVolume, tnsend
   799)     use parallel_mod, only : iam, ncompoints
   800)     use dimensions_mod, only : nmpi_per_node, npart
   801)     ! results:: TnSend,{min,max}_{volume,number},Time_{total,calc,latency,bandwidth}
   802)     ! inputs:: results + Schedule + bytes_per_point 
   803) 
   804)     type (Schedule_t),intent(in) :: Schedule(:)
   805)     integer(kind=int_kind),intent(in) :: bytes_per_point
   806)     type (commtime_t),intent(in)    :: offnode,onnode
   807)     type (GraphStats_t), intent(in)   :: count(:)
   808)     type (commtime_t),intent(inout)   :: Time(:)
   809) 
   810)     real(kind=real_kind)   :: bandwidth_tmp,latency_tmp
   811)     integer(kind=int_kind) :: nbr,node1,node2,nsend,ip,icycle,nrecv
   812)     integer(kind=int_kind) :: length
   813)     real(kind=real_kind) :: lb_volume
   814)     integer(kind=int_kind) :: imax_volume,imin_volume
   815)     type (perf_t)          :: tcv
   816)     integer                :: nSched
   817)     integer                :: nelemd0
   818)     logical, parameter     :: Debug = .FALSE.
   819) 
   820)     nSched=SIZE(Schedule)
   821)     min_message = 1000000
   822)     min_volume  = 1000000
   823)     max_message = 0
   824)     max_volume  = 0
   825)     TotalComVolume = 0
   826)     TnSend = 0
   827) 
   828)     tcv%onnode  = 0.0
   829)     tcv%offnode = 0.0
   830)     do ip=1,nSched
   831)        nSend = Schedule(ip)%nSendCycles
   832)        nRecv = Schedule(ip)%nRecvCycles
   833)        nelemd0 = Schedule(ip)%nelemd
   834)        nComPoints = 0
   835)        Time(ip)%latency = 0
   836)        Time(ip)%bandwidth = 0
   837)        do icycle=1,nSend
   838)           length = Schedule(ip)%SendCycle(icycle)%lengthV
   839)           nComPoints = nComPoints + length
   840)           min_message = MIN(min_message, bytes_per_point*length)
   841)           max_message = MAX(max_message, bytes_per_point*length)
   842)        enddo
   843)        do icycle=1,nRecv
   844)           length = Schedule(ip)%RecvCycle(icycle)%lengthV
   845)           nbr = Schedule(ip)%RecvCycle(icycle)%source
   846)           node1 = ((ip-1)/nmpi_per_node) + 1
   847)           node2 = ((nbr-1)/nmpi_per_node) + 1
   848)           if(node1 .eq. node2 )  then
   849)              ! Message is set to an on-node processor
   850)              tcv%onnode = tcv%onnode + length*bytes_per_point
   851)              bandwidth_tmp = onnode%bandwidth
   852)              latency_tmp = onnode%latency
   853)           else
   854)              ! Message is set to an off-node processor
   855)              tcv%offnode = tcv%offnode + length*bytes_per_point
   856)              bandwidth_tmp = real(count(node1)%offnode,kind=real_kind)*offnode%bandwidth
   857)              latency_tmp = offnode%latency
   858)              if(Debug) write(iulog,*)'fooCalc: node # ',node1,' offnode count is ',real(count(node1)%offnode,kind=real_kind)
   859)              if(Debug) write(iulog,*)'fooCalc: node # ',node1,' offnode ={total,effective} ',offnode%bandwidth,bandwidth_tmp
   860) 
   861)           endif
   862)           Time(ip)%latency = Time(ip)%latency + latency_tmp
   863)           if(Debug) then 
   864)              write(iulog,*)'fooCalc: node # ',node1,' Contribution in time ',length*bytes_per_point,  &
   865)                   length*bytes_per_point*bandwidth_tmp
   866)           endif
   867)           Time(ip)%bandwidth = Time(ip)%bandwidth + length*bytes_per_point*bandwidth_tmp
   868)        enddo
   869)        !JMD  Note: I am removing the size of the communication buffer out of the computation
   870)        LocalComVolume = bytes_per_point*nComPoints
   871)        min_volume = MIN(min_volume,LocalComVolume)
   872)        max_volume = MAX(max_volume,LocalComVolume)
   873)        TnSend     = TnSend + nSend
   874) 
   875)        TotalComVolume = TotalComVolume + LocalComVolume
   876)     enddo
   877) 
   878)     !------------------------------------------------------
   879)     ! This Outputs information about each boundary exchange 
   880)     !------------------------------------------------------
   881)     avg_volume  = dble(TotalComVolume)/dble(npart)
   882)     lb_volume  = (max_volume-avg_volume)/max_volume
   883)     avg_message = dble(TotalComVolume)/dble(TnSend)
   884)     imin_volume = NINT(min_volume)
   885)     imax_volume = NINT(max_volume)
   886)     if(iam .eq. 1) then 
   887) 
   888)        write(iulog,*)'MessageStats: Total Message volume is (Mbytes): ',1.0e-6*TotalComVolume
   889) #if 1
   890)        write(iulog,*) 'MessageStats: imin_volume',imin_volume
   891)        write(iulog,*) 'MessageStats: imax_volume',imax_volume
   892)        write(iulog,*) 'MessageStats: avg_volume',avg_volume
   893)        write(iulog,*) 'MessageStats: lb_volume',lb_volume
   894) #else
   895)        !JMD For some reason on AIX this line breaks when the numbers are high
   896)        write(iulog,110) 'MessageStats: Single process volume (bytes) {MIN,MAX,AVG,LB} =: ', &
   897)             imin_volume, imax_volume, avg_volume,lb_volume
   898) #endif
   899)        write(iulog,*) 'MessageStats: Single message (bytes) {MIN,MAX,AVG} =: ',  &
   900)             min_message,max_message,avg_message
