   151)     call init_carbon_balance_type(begc, endc, ccbal)

   156)     call init_nitrogen_balance_type(begc, endc, cnbal)

   160)     call init_pft_pstate_type(begp, endp, pps)
   161)     call init_pft_pstate_type(begc, endc, pps_a)
   162)

   164)     call init_pft_epv_type(begp, endp, pepv)
   165) 
   166)     ! pft DGVM state variables at pft level 
   167) 
   168)     if (use_cndv) then

   170)     end if

   172)

   174) 
   175)     call init_pft_estate_type(begp, endp, pes)
   176) 
   177)     ! pft water state variables at the pft level and averaged to the column
   178)

   185)     call init_pft_cstate_type(begc, endc, pcs_a)

   188)        ! Adding isotope code

   190)        call init_pft_cstate_type(begc, endc, pc13s_a)

   193)        end if
   194)     endif
   195) 
   196)     ! pft nitrogen state variables at the pft level and averaged to the column
   197) 
   198)     call init_pft_nstate_type(begp, endp, pns)
   199)     call init_pft_nstate_type(begc, endc, pns_a)
   200)

   215) 
   216)     call init_pft_cflux_type(begp, endp, pcf)
   217)     call init_pft_cflux_type(begc, endc, pcf_a)
   218)     if (use_c13) then

   220)        ! Adding isotope code

   224)

   228)     call init_pft_nflux_type(begc, endc, pnf_a)
   229)

   237) 
   238)     ! pft dust flux variables at pft level 
   239) 
   240)     call init_pft_dflux_type(begp, endp, pdf)
   241) 
   242)     ! pft dry dep velocity variables at pft level

   245) 
   246)     ! column physical state variables at column level

   249) 
   250)     ! column energy state variables at column level 
   251) 
   252) 
   253)     call init_column_estate_type(begc, endc, ces)
   254) 
   255)     ! column water state variables at column level 
   256) 
   257)     call init_column_wstate_type(begc, endc, cws)
   258) 
   259)     ! column carbon state variables at column level
   260) 
   261)     call init_column_cstate_type(begc, endc, ccs)
   262)     if (use_c13) then
   263)        ! 4/14/05: PET
   264)        ! Adding isotope code
   265)        call init_column_cstate_type(begc, endc, cc13s)
   266)     endif
   267) 
   268)     ! column nitrogen state variables at column level
   269) 
   270)     call init_column_nstate_type(begc, endc, cns)
   271) 
   272)     ! column energy flux variables at column level 
   273) 
   274)     call init_column_eflux_type(begc, endc, cef)
   275) 
   276)     ! column water flux variables at column level 
   277) 
   278)     call init_column_wflux_type(begc, endc, cwf)
   279) 
   280)     ! column carbon flux variables at column level
   281) 
   282)     call init_column_cflux_type(begc, endc, ccf)
   283)     if (use_c13) then
   284)        ! 4/14/05: PET
   285)        ! Adding isotope code
   286)        call init_column_cflux_type(begc, endc, cc13f)
   287)     endif
   288) 
   289)     ! column nitrogen flux variables at column level
   290) 
   291)     call init_column_nflux_type(begc, endc, cnf)
   292) 
   293)     ! land unit physical state variables
   294) 
   295)     call init_landunit_pstate_type(begl, endl, lps)
   296) 
   297)     ! land unit energy flux variables 
   298) 
   299)     call init_landunit_eflux_type(begl, endl, lef)
   300) 
   301)     ! gridcell DGVM variables
   302) 
   303)     if (use_cndv) then
   304)        call init_gridcell_dgvstate_type(begg, endg, gdgvs)
   305)     end if
   306) 
   307)     ! gridcell physical state variables
   308) 
   309) 
   310)     ! gridcell: water flux variables
   311) 
   312)     call init_gridcell_wflux_type(begg, endg, gwf)
   313) 
   314)     ! gridcell: energy flux variables
   315) 
   316)     call init_gridcell_eflux_type(begg, endg, gef)
   317) 
   318)     ! gridcell: water state variables
   319) 
   320)     call init_gridcell_wstate_type(begg, endg, gws)
   321)

   323)
