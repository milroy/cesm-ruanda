   166) ! !IROUTINE: init_forcing_atmo - initialize atmospheric forcing
   167) !
   168) ! !INTERFACE:
   169) !
   170)       subroutine init_forcing_atmo
   171) !
   172) ! !DESCRIPTION:
   173) !
   174) ! Determine the current and final year of the forcing cycle based on
   175) ! namelist input; initialize the forcing data filenames.
   176) !
   177) ! !REVISION HISTORY:
   178) !
   179) ! authors: same as module
   180) !
   181) ! !USES:
   182) !
   183) ! !INPUT/OUTPUT PARAMETERS:
   184) !
   185) !EOP
   186) !
   187)       fyear       = fyear_init + mod(nyr-1,ycycle) ! current year
   188)       fyear_final = fyear_init + ycycle - 1 ! last year in forcing cycle
   189) 
   190)       if (trim(atm_data_type) /= 'default' .and. &
   191)                           my_task == master_task) then
   192)          write (nu_diag,*) ' Initial forcing data year = ',fyear_init
   193)          write (nu_diag,*) ' Final   forcing data year = ',fyear_final
   194)       endif
   195) 
   196)     !-------------------------------------------------------------------
   197)     ! Get filenames for input forcing data     
   198)     !-------------------------------------------------------------------
   199) 
   200)       ! default forcing values from init_flux_atm
   201)       if (trim(atm_data_type) == 'ncar') then
   202)          call NCAR_files(fyear)
   203)       elseif (trim(atm_data_type) == 'ecmwf') then
   204)          call ecmwf_files(fyear)    
   205)       elseif (trim(atm_data_type) == 'LYq') then
   206)          call LY_files(fyear)
   207)       elseif (trim(atm_data_type) == 'hadgem') then
   208)          call hadgem_files(fyear)
   209)       elseif (trim(atm_data_type) == 'monthly') then
   210)          call monthly_files(fyear)
   211)       endif
   212) 
   213)       end subroutine init_forcing_atmo

   218) ! !IROUTINE: init_forcing_ocn - initialize sss and sst
   219) !
   220) ! !INTERFACE:
   221) !
   222)       subroutine init_forcing_ocn(dt)
   223) !
   224) ! !DESCRIPTION:
   225) !
   226) ! Set sea surface salinity and freezing temperature to annual mean value 
   227) !  using a 12-month climatology.
   228) ! Read sst data for current month, and adjust sst based on freezing
   229) ! temperature.  No interpolation in time.
   230) 
   231) ! Note: SST is subsequently prognosed if CICE is run with a mixed layer
   232) ! ocean (oceanmixed\_ice = T), and can be restored to data 
   233) ! (restore\_sst = T). SSS is not prognosed by CICE. 
   234) !
   235) ! !REVISION HISTORY:
   236) !
   237) ! authors: same as module
   238) !
   239) ! !USES:
   240) !
   241)       use ice_domain, only: nblocks
   242)       use ice_flux, only: sss, sst, Tf, Tfrzpt
   243)       use ice_work, only:  work1
   244)       use ice_read_write
   245)       use ice_therm_vertical, only: ustar_scale
   246) !
   247) ! !INPUT/OUTPUT PARAMETERS:
   248) !
   249)       real (kind=dbl_kind), intent(in) :: &
   250)            dt                   ! time step
   251) !
   252) !EOP
   253) !
   254)       integer (kind=int_kind) :: &
   255)            i, j, iblk       , & ! horizontal indices
   256)            k                , & ! month index
   257)            fid              , & ! file id for netCDF file 
   258)           nbits
   259) 
   260)       logical (kind=log_kind) :: diag
   261) 
   262)       character (char_len) :: & 
   263)             fieldname    ! field name in netcdf file
   264) 
   265)       nbits = 64                ! double precision data
   266) 
   267)     !-------------------------------------------------------------------
   268)     ! Sea surface salinity (SSS)
   269)     ! initialize to annual climatology created from monthly data
   270)     !-------------------------------------------------------------------
   271) 
   272)       if (trim(sss_data_type) == 'clim') then
   273) 
   274) !         sss_file = trim(ocn_data_dir)//'sss_Lev.mm'
   275)             sss_file = trim(ocn_data_dir)//'sss.mm.100x116.da' ! gx3 only
   276) !!!         sss_file = trim(ocn_data_dir)//'sss_12.r'
   277) 
   278)          if (my_task == master_task) then
   279)             write (nu_diag,*) ' '
   280)             write (nu_diag,*) 'SSS climatology computed from:'
   281)             write (nu_diag,*) trim(sss_file)
   282)          endif
   283) 
   284)          if (my_task == master_task) &
   285)               call ice_open (nu_forcing, sss_file, nbits)
   286) 
   287)          sss(:,:,:) = c0
   288) 
   289)          do k = 1,12            ! loop over 12 months
   290)             call ice_read (nu_forcing, k, work1, 'rda8', dbug, &
   291)                            field_loc_center, field_type_scalar)
   292)             do iblk = 1, nblocks
   293)                do j = 1, ny_block
   294)                do i = 1, nx_block
   295)                   sss(i,j,iblk) = sss(i,j,iblk) + work1(i,j,iblk)
   296)                enddo
   297)                enddo
   298)             enddo
   299)          enddo                  ! k
   300) 
   301)          do iblk = 1, nblocks
   302)             do j = 1, ny_block
   303)             do i = 1, nx_block
   304)                sss(i,j,iblk) = sss(i,j,iblk) / c12   ! annual average
   305)                sss(i,j,iblk) = max(sss(i,j,iblk),c0)
   306)                if (trim(Tfrzpt) == 'constant') then
   307)                   Tf (i,j,iblk) = -1.8_dbl_kind ! deg C
   308)                else ! default:  Tfrzpt = 'linear_S'
   309)                   Tf (i,j,iblk) = -depressT * sss(i,j,iblk) ! deg C
   310)                endif
   311)             enddo
   312)             enddo
   313)          enddo
   314) 
   315)          ! close file
   316)          if (my_task == master_task) close(nu_forcing)
   317) 
   318)       endif                     ! sss_data_type
   319) 
   320)     !-------------------------------------------------------------------
   321)     ! Sea surface temperature (SST)
   322)     ! initialize to data for current month
   323)     !-------------------------------------------------------------------
   324) 
   325)       if (restore_sst) then
   326)          if (trestore == 0) then
   327)             trest = dt          ! use data instantaneously
   328)          else
   329)             trest = real(trestore,kind=dbl_kind) * secday ! seconds
   330)          endif
   331)       endif
   332) 
   333)       if (trim(sst_data_type) == 'clim') then
   334) 
   335)          if (nx_global == 320) then ! gx1
   336)             sst_file = trim(ocn_data_dir)//'sst_clim_hurrell.dat'
   337)          else                   ! gx3
   338) !            sst_file = trim(ocn_data_dir)//'sst_Lev.mm'
   339)             sst_file = trim(ocn_data_dir)//'sst.mm.100x116.da'
   340) !!!            sst_file = trim(ocn_data_dir)//'sst_12.r'
   341)          endif
   342) 
   343)          if (my_task == master_task) then
   344)             write (nu_diag,*) ' '
   345)             write (nu_diag,*) 'Initial SST file:', trim(sst_file)
   346)          endif
   347) 
   348)          if (my_task == master_task) &
   349)               call ice_open (nu_forcing, sst_file, nbits)
   350) 
   351)          call ice_read (nu_forcing, month, sst, 'rda8', dbug, &
   352)                         field_loc_center, field_type_scalar)
   353) 
   354)          if (my_task == master_task) close(nu_forcing)
   355) 
   356)          ! Make sure sst is not less than freezing temperature Tf
   357)          do iblk = 1, nblocks
   358)             do j = 1, ny_block
   359)             do i = 1, nx_block
   360)                sst(i,j,iblk) = max(sst(i,j,iblk),Tf(i,j,iblk))
   361)             enddo
   362)             enddo
   363)          enddo
   364) 
   365)       endif                     ! init_sst_data
   366) 
   367) 
   368)       if (trim(sst_data_type) == 'hadgem_sst' .or.  &
   369)           trim(sst_data_type) == 'hadgem_sst_uvocn') then
   370) 
   371)        	 diag = .true.   ! write diagnostic information 
   372) 
   373)          sst_file = trim (ocn_data_dir)//'MONTHLY/sst.1997.nc'
   374) 
   375)        	 if (my_task == master_task) then
   376) 
   377)              write (nu_diag,*) ' '
   378)              write (nu_diag,*) 'Initial SST file:', trim(sst_file)
   379) 
   380)              call ice_open_nc(sst_file,fid)
   381) 
   382)          endif
   383)  
   384)          fieldname='sst'
   385)          call ice_read_nc(fid,month,fieldname,sst,diag)
   386) 
   387)          if (my_task == master_task) call ice_close_nc(fid)  
   388) 
   389)          ! Make sure sst is not less than freezing temperature Tf
   390)          do iblk = 1, nblocks
   391)             do j = 1, ny_block
   392)             do i = 1, nx_block
   393)                sst(i,j,iblk) = max(sst(i,j,iblk),Tf(i,j,iblk))
   394)             enddo
   395)             enddo
   396)          enddo
   397) 
   398)       endif                        ! sst_data_type
   399) 
   400) 
   401)       if (trim(sst_data_type) == 'ncar' .or.  &
   402)           trim(sss_data_type) == 'ncar') then
   403)          call ocn_data_ncar_init
   404)       endif
   405) 
   406)       ! set ustar_scale for case of zero currents
   407)       ! default value of c1 (for nonzero currents) set in init_thermo_vertical
   408) 
   409)       if (trim(sst_data_type) /= 'ncar' .or.  &
   410)           trim(sss_data_type) /= 'ncar' .or.  & 
   411)           trim(sst_data_type) /= 'hadgem_sst_uvocn') then
   412)          ustar_scale = c10            ! for zero currents
   413)       endif
   414) 
   415)       end subroutine init_forcing_ocn

   420) ! !IROUTINE: get_forcing_atmo - Get atmospheric forcing data and interpolate
   421) !
   422) ! !INTERFACE:
   423) !
   424)       subroutine get_forcing_atmo
   425) !
   426) ! !DESCRIPTION:
   427) !
   428) ! Get atmospheric forcing data and interpolate as necessary
   429) !
   430) ! !REVISION HISTORY:
   431) !
   432) ! authors: same as module
   433) !
   434) ! !USES:
   435) !
   436)       use ice_boundary, only: ice_HaloUpdate
   437)       use ice_domain
   438)       use ice_blocks
   439)       use ice_flux
   440)       use ice_state
   441)       use ice_grid, only: ANGLET, hm
   442) !
   443) !EOP
   444) !
   445)       integer (kind=int_kind) :: &
   446)          iblk, &              ! block index
   447)          ilo,ihi,jlo,jhi      ! beginning and end of physical domain
   448) 
   449)       type (block) :: &
   450)          this_block           ! block information for current block
   451)       
   452)       fyear = fyear_init + mod(nyr-1,ycycle)  ! current year
   453)       if (trim(atm_data_type) /= 'default' .and. istep <= 1 &
   454)                    .and. my_task == master_task) then
   455)          write (nu_diag,*) ' Current forcing data year = ',fyear
   456)       endif
   457) 
   458)       ftime = time         ! forcing time
   459)       time_forc = ftime    ! for restarting
   460) 
   461)     !-------------------------------------------------------------------
   462)     ! Read and interpolate atmospheric data
   463)     !-------------------------------------------------------------------
   464) 
   465)       if (trim(atm_data_type) == 'ncar') then
   466)          call ncar_data
   467)       elseif (trim(atm_data_type) == 'ecmwf') then
   468)          call ecmwf_data
   469)       elseif (trim(atm_data_type) == 'LYq') then
   470)          call LY_data
   471)       elseif (trim(atm_data_type) == 'hadgem') then
   472)          call hadgem_data
   473)       elseif (trim(atm_data_type) == 'monthly') then
   474)          call monthly_data
   475)       else    ! default values set in init_flux
   476)          return
   477)       endif
   478) 
   479)     !-------------------------------------------------------------------
   480)     ! Convert forcing data to fields needed by ice model
   481)     !-------------------------------------------------------------------
   482) 
   483)       do iblk = 1, nblocks
   484) 
   485)          this_block = get_block(blocks_ice(iblk),iblk)         
   486)          ilo = this_block%ilo
   487)          ihi = this_block%ihi
   488)          jlo = this_block%jlo
   489)          jhi = this_block%jhi
   490) 
   491)          call prepare_forcing (nx_block, ny_block, &
   492)                                ilo, ihi, jlo, jhi, &
   493)                                hm    (:,:,iblk),   &
   494)                                Tair  (:,:,iblk),   &
   495)                                fsw   (:,:,iblk),   &   
   496)                                cldf  (:,:,iblk),   &
   497)                                flw   (:,:,iblk),   &
   498)                                frain (:,:,iblk),   &
   499)                                fsnow (:,:,iblk),   &
   500)                                Qa    (:,:,iblk),   &
   501)                                rhoa  (:,:,iblk),   &
   502)                                uatm  (:,:,iblk),   &
   503)                                vatm  (:,:,iblk),   &
   504)                                strax (:,:,iblk),   &
   505)                                stray (:,:,iblk),   &
   506)                                zlvl  (:,:,iblk),   &
   507)                                wind  (:,:,iblk),   &
   508)                                swvdr (:,:,iblk),   &
   509)                                swvdf (:,:,iblk),   &
   510)                                swidr (:,:,iblk),   &
   511)                                swidf (:,:,iblk),   &
   512)                                potT  (:,:,iblk),   &
   513)                                ANGLET(:,:,iblk),   &
   514)                                trcr  (:,:,nt_Tsfc,iblk), &
   515)                                sst   (:,:,iblk),   &
   516)                                aice  (:,:,iblk) )
   517) 
   518)       enddo                     ! iblk
   519) 
   520)       call ice_timer_start(timer_bound)
   521)       call ice_HaloUpdate (swvdr,             halo_info, &
   522)                            field_loc_center,  field_type_scalar)
   523)       call ice_HaloUpdate (swvdf,             halo_info, &
   524)                            field_loc_center,  field_type_scalar)
   525)       call ice_HaloUpdate (swidr,             halo_info, &
   526)                            field_loc_center,  field_type_scalar)
   527)       call ice_HaloUpdate (swidf,             halo_info, &
   528)                            field_loc_center,  field_type_scalar)
   529)       call ice_timer_stop(timer_bound)
   530) 
   531)       end subroutine get_forcing_atmo

   540)       subroutine get_forcing_ocn (dt)
   541) !
   542) ! !DESCRIPTION:
   543) !
   544) ! Read and interpolate annual climatologies of SSS and SST.
   545) ! Restore model SST to data if desired.
   546) ! Interpolate ocean fields to U grid if necessary.
   547) !
   548) ! !REVISION HISTORY:
   549) !
   550) ! author: Elizabeth C. Hunke, LANL
   551) !
   552) ! !USES:

   554)       use ice_domain, only: nblocks

   556) !

   560)          dt      ! time step
   561) !
   562) !EOP
   563) !

   565)           trim(sss_data_type) == 'clim') then
   566)          call ocn_data_clim(dt)
   567)       elseif (trim(sst_data_type) == 'ncar' .or.  &
   568)               trim(sss_data_type) == 'ncar') then
   569)          call ocn_data_ncar(dt)      
   570)       elseif (trim(sst_data_type) == 'hadgem_sst' .or.  &
   571)               trim(sst_data_type) == 'hadgem_sst_uvocn') then
   572)          call ocn_data_hadgem(dt)

  1017) !
  1018)       subroutine interp_coeff_monthly (recslot)
  1019) !
  1020) ! !DESCRIPTION:
  1021) !
  1022) ! Compute coefficients for interpolating monthly data to current time step.
  1023) !
  1024) ! !REVISION HISTORY:
  1025) !
  1026) ! authors: same as module
  1027) !
  1028) ! !USES:
  1029) !
  1030) ! !INPUT/OUTPUT PARAMETERS:
  1031) !
  1032)       integer (kind=int_kind), intent(in) :: &
  1033)           recslot         ! slot (1 or 2) for current record
  1034) !
  1035) !EOP
  1036) !
  1037)       real (kind=dbl_kind) :: &
  1038)           tt           , & ! seconds elapsed in current year
  1039)           t1, t2           ! seconds elapsed at month midpoint
  1040) 
  1041)       real (kind=dbl_kind) :: &
  1042)           daymid(0:13)     ! month mid-points
  1043) 
  1044)       daymid(1:13) = 14._dbl_kind   ! time frame ends 0 sec into day 15
  1045)       daymid(0)    = 14._dbl_kind - daymo(12)  ! Dec 15, 0 sec
  1046) 
  1047)       ! make time cyclic
  1048)       tt = mod(ftime/secday,dayyr)
  1049) 
  1050)       ! Find neighboring times
  1051) 
  1052)       if (recslot==2) then      ! first half of month
  1053)         t2 = daycal(month) + daymid(month)   ! midpoint, current month
  1054)         if (month == 1) then
  1055)           t1 = daymid(0)                 ! Dec 15 (0 sec)
  1056)         else
  1057)           t1 = daycal(month-1) + daymid(month-1) ! midpoint, previous month
  1058)         endif
  1059)       else                      ! second half of month
  1060)         t1 = daycal(month) + daymid(month)    ! midpoint, current month
  1061)         t2 = daycal(month+1) + daymid(month+1)! day 15 of next month (0 sec)
  1062)       endif
  1063) 
  1064)       ! Compute coefficients
  1065)       c1intp = (t2 - tt) / (t2 - t1)
  1066)       c2intp =  c1 - c1intp
  1067) 
  1068)       end subroutine interp_coeff_monthly

  2291)                                    ilo, ihi, jlo, jhi, &
  2292)                                    TLON, TLAT, hm, Qa, cldf, fsw)

  2296) !---!-------------------------------------------------------------------

  2305)          cldf           , & ! cloud fraction
  2306)          hm                 ! land mask
  2307) 
  2308)       real (kind=dbl_kind), dimension(nx_block,ny_block),  &
  2309)          intent(inout) :: &

  2311)

  2314)          solar_time, &
  2315)          declin    , &
  2316)          cosZ      , &
  2317)          e, d      , &
  2318)          sw0       , &
  2319)          deg2rad

  2322)          i, j

  2326)         deg2rad = pi/c180

  2332)         cosZ = sin(TLAT(i,j))*sin(declin) &

  2341)         Fsw(i,j) = sw0*(c1-p6*cldf(i,j)**3)

  2351) 
  2352)       use ice_work, only: worka
  2353) 
  2354)       integer (kind=int_kind), intent(in) :: &
  2355)          nx_block, ny_block ! block dimensions
  2356)

  2358)          Tair               ! air temperature

  2362)          Qa                 ! specific humidity
  2363) 
  2364)       worka = Tair - Tffresh
  2365)       worka = c2 + (0.7859_dbl_kind + 0.03477_dbl_kind*worka) &
  2366)                      /(c1 + 0.00412_dbl_kind*worka) & ! 2+ converts ea mb -> Pa
  2367)                 + 0.00422_dbl_kind*worka              ! for ice

  2369)       worka = (c10**worka)      ! saturated

  2372)       worka = 0.622_dbl_kind*worka/(1.e5_dbl_kind-0.378_dbl_kind*worka)
