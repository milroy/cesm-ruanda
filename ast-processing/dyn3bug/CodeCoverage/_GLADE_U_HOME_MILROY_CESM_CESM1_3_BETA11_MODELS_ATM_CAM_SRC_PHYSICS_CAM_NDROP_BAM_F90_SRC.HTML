    72) subroutine ndrop_bam_init
    73) 
    74)   use phys_control, only: phys_getopts
    75) 
    76)    !----------------------------------------------------------------------- 
    77)    ! 
    78)    ! Initialize constants for droplet activation by bulk aerosols
    79)    ! 
    80)    !-----------------------------------------------------------------------
    81) 
    82)    integer  :: l, m, iaer
    83)    real(r8) :: surften       ! surface tension of water w/respect to air (N/m)
    84)    real(r8) :: arg
    85)    logical  :: history_amwg
    86)    !-------------------------------------------------------------------------------
    87) 
    88)    call phys_getopts(history_amwg_out=history_amwg)
    89) 
    90)    ! Access the physical properties of the bulk aerosols that are affecting the climate
    91)    ! by using routines from the rad_constituents module.
    92) 
    93)    call rad_cnst_get_info(0, naero=naer_all)
    94)    allocate( &
    95)       aername(naer_all),        &
    96)       dryrad_aer(naer_all),     &
    97)       density_aer(naer_all),    &
    98)       hygro_aer(naer_all),      &
    99)       dispersion_aer(naer_all), &
   100)       num_to_mass_aer(naer_all) )
   101) 
   102)    do iaer = 1, naer_all
   103)       call rad_cnst_get_aer_props(0, iaer, &
   104)          aername         = aername(iaer), &
   105)          dryrad_aer      = dryrad_aer(iaer), &
   106)          density_aer     = density_aer(iaer), &
   107)          hygro_aer       = hygro_aer(iaer), &
   108)          dispersion_aer  = dispersion_aer(iaer), &
   109)          num_to_mass_aer = num_to_mass_aer(iaer) )
   110) 
   111)       ! Look for sulfate aerosol in this list (Bulk aerosol only)
   112)       if (trim(aername(iaer)) == 'SULFATE') idxsul   = iaer
   113) 
   114)       ! aerosol number concentration
   115)       call addfld(trim(aername(iaer))//'_m3', 'm-3', pver, 'A', 'aerosol number concentration', phys_decomp)
   116) 
   117)    end do
   118) 
   119)    if (masterproc) then
   120)       write(iulog,*) 'ndrop_bam_init: iaer, name, dryrad, density, hygro, dispersion, num_to_mass'
   121)       do iaer = 1, naer_all
   122)          write(iulog,*) iaer, aername(iaer), dryrad_aer(iaer), density_aer(iaer), hygro_aer(iaer), &
   123)             dispersion_aer(iaer), num_to_mass_aer(iaer)
   124)       end do
   125)       if (idxsul < 1) then
   126)          write(iulog,*) 'ndrop_bam_init: SULFATE aerosol properties NOT FOUND'
   127)       else
   128)          write(iulog,*) 'ndrop_bam_init: SULFATE aerosol properties FOUND at index ',idxsul
   129)       end if
   130)    end if
   131) 
   132)    call addfld ('CCN1    ','#/cm3   ',pver, 'A','CCN concentration at S=0.02%',phys_decomp)
   133)    call addfld ('CCN2    ','#/cm3   ',pver, 'A','CCN concentration at S=0.05%',phys_decomp)
   134)    call addfld ('CCN3    ','#/cm3   ',pver, 'A','CCN concentration at S=0.1%',phys_decomp)
   135)    call addfld ('CCN4    ','#/cm3   ',pver, 'A','CCN concentration at S=0.2%',phys_decomp)
   136)    call addfld ('CCN5    ','#/cm3   ',pver, 'A','CCN concentration at S=0.5%',phys_decomp)
   137)    call addfld ('CCN6    ','#/cm3   ',pver, 'A','CCN concentration at S=1.0%',phys_decomp)
   138) 
   139)    if (history_amwg) then
   140)       call add_default('CCN3', 1, ' ')
   141)    end if
   142) 
   143)    ! set parameters for droplet activation, following abdul-razzak and ghan 2000, JGR
   144) 
   145)    third   = 1._r8/3._r8
   146)    sixth   = 1._r8/6._r8
   147)    sq2     = sqrt(2._r8)
   148)    pi      = 4._r8*atan(1.0_r8)
   149)    sqpi    = sqrt(pi)
   150)    surften = 0.076_r8
   151)    aten    = 2._r8*mwh2o*surften/(r_universal*tmelt*rhoh2o)
   152)    alogaten= log(aten)
   153)    alog2   = log(2._r8)
   154)    alog3   = log(3._r8)
   155)    super(:)= 0.01_r8*supersat(:)
   156) 
   157)    allocate(                  &
   158)       alogsig(naer_all),      &
   159)       exp45logsig(naer_all),  &
   160)       argfactor(naer_all),    &
   161)       f1(naer_all),           &
   162)       f2(naer_all),           &
   163)       amcubefactor(naer_all), &
   164)       smcritfactor(naer_all), &
   165)       amcube(naer_all),       &
   166)       smcrit(naer_all),       &
   167)       lnsm(naer_all),         &
   168)       ccnfact(psat,naer_all)  )
   169) 
   170) 
   171)    do m = 1, naer_all
   172) 
   173)       ! Skip aerosols that don't have a dispersion defined.
   174)       if (dispersion_aer(m) == 0._r8) cycle
   175)       
   176)       alogsig(m)     = log(dispersion_aer(m))
   177)       exp45logsig(m) = exp(4.5_r8*alogsig(m)*alogsig(m))
   178)       argfactor(m)   = 2._r8/(3._r8*sqrt(2._r8)*alogsig(m))
   179)       f1(m)          = 0.5_r8*exp(2.5_r8*alogsig(m)*alogsig(m))
   180)       f2(m)          = 1._r8 + 0.25_r8*alogsig(m)
   181)       amcubefactor(m)= 3._r8/(4._r8*pi*exp45logsig(m)*density_aer(m))
   182)       smcritfactor(m)= 2._r8*aten*sqrt(aten/(27._r8*max(1.e-10_r8,hygro_aer(m))))
   183)       amcube(m)      = amcubefactor(m)/num_to_mass_aer(m)
   184) 
   185)       if (hygro_aer(m) .gt. 1.e-10_r8) then
   186)          smcrit(m) = smcritfactor(m)/sqrt(amcube(m))
   187)       else
   188)          smcrit(m) = 100._r8
   189)       endif
   190)       lnsm(m) = log(smcrit(m))
   191) 
   192)       do l = 1, psat
   193)          arg = argfactor(m)*log(smcrit(m)/super(l))
   194)          if (arg < 2) then
   195)             if (arg < -2) then
   196)                ccnfact(l,m) = 1.e-6_r8
   197)             else
   198)                ccnfact(l,m) = 1.e-6_r8*0.5_r8*erfc(arg)
   199)             endif
   200)          else
   201)             ccnfact(l,m) = 0._r8
   202)          endif
   203)       enddo
   204) 
   205)    end do
   206) 
   207) end subroutine ndrop_bam_init

   211) subroutine ndrop_bam_run( &
   212)    wbar, tair, rhoair, na, pmode, &
   213)    nmode, ma, nact)
   214) 
   215)    ! calculates number fraction of aerosols activated as CCN
   216)    ! assumes an internal mixture within each of up to pmode multiple aerosol modes
   217)    ! a gaussian spectrum of updrafts can be treated.
   218) 
   219)    !      mks units
   220) 
   221)    !      Abdul-Razzak and Ghan, A parameterization of aerosol activation.
   222)    !      2. Multiple aerosol types. J. Geophys. Res., 105, 6837-6844.
   223) 
   224)    ! input
   225)    integer,  intent(in) :: pmode         ! dimension of modes
   226)    integer,  intent(in) :: nmode         ! number of aerosol modes
   227)    real(r8), intent(in) :: wbar          ! grid cell mean vertical velocity (m/s)
   228)    real(r8), intent(in) :: tair          ! air temperature (K)
   229)    real(r8), intent(in) :: rhoair        ! air density (kg/m3)
   230)    real(r8), intent(in) :: na(pmode)     ! aerosol number concentration (1/m3)
   231)    real(r8), intent(in) :: ma(pmode)     ! aerosol mass concentration (kg/m3)
   232) 
   233)    ! output
   234)    real(r8), intent(out) :: nact         ! number fraction of aerosols activated
   235) 
   236)    ! local variables
   237)    integer :: maxmodes
   238) 
   239)    real(r8), allocatable :: volc(:) ! total aerosol volume  concentration (m3/m3)
   240)    real(r8), allocatable :: eta(:)
   241)    real(r8), allocatable :: smc(:)
   242)    real(r8), allocatable :: etafactor2(:)
   243)    real(r8), allocatable :: amcubeloc(:)
   244)    real(r8), allocatable :: lnsmloc(:)
   245) 
   246)    real(r8) :: pres     ! pressure (Pa)
   247)    real(r8) :: diff0
   248)    real(r8) :: conduct0 ! thermal conductivity (Joule/m/sec/deg)
   249)    real(r8) :: qs       ! water vapor saturation mixing ratio
   250)    real(r8) :: dqsdt    ! change in qs with temperature
   251)    real(r8) :: gloc     ! thermodynamic function (m2/s)
   252)    real(r8) :: zeta
   253)    real(r8) :: lnsmax   ! ln(smax)
   254)    real(r8) :: alpha
   255)    real(r8) :: gammaloc
   256)    real(r8) :: beta
   257)    real(r8) :: sqrtg
   258)    real(r8) :: wnuc
   259)    real(r8) :: alw
   260)    real(r8) :: sqrtalw
   261)    real(r8) :: smax
   262)    real(r8) :: x
   263)    real(r8) :: etafactor1
   264)    real(r8) :: etafactor2max
   265)    real(r8) :: es
   266)    integer  :: m
   267)    !-------------------------------------------------------------------------------
   268) 
   269)    maxmodes = naer_all
   270)    allocate( &
   271)       volc(maxmodes),       &
   272)       eta(maxmodes),        &
   273)       smc(maxmodes),        &
   274)       etafactor2(maxmodes), &
   275)       amcubeloc(maxmodes),  &
   276)       lnsmloc(maxmodes)     )
   277) 
   278)    if (maxmodes < pmode) then
   279)       write(iulog,*)'ndrop_bam_run: maxmodes,pmode=',maxmodes,pmode
   280)       call endrun('ndrop_bam_run')
   281)    endif
   282) 
   283)    nact = 0._r8
   284) 
   285)    if (nmode .eq. 1  .and.  na(1) .lt. 1.e-20_r8) return
   286) 
   287)    if (wbar .le. 0._r8) return
   288) 
   289)    pres     = rair*rhoair*tair
   290)    diff0    = 0.211e-4_r8*(pref/pres)*(tair/tmelt)**1.94_r8
   291)    conduct0 = (5.69_r8 + 0.017_r8*(tair-tmelt))*4.186e2_r8*1.e-5_r8 ! convert to J/m/s/deg
   292)    call qsat(tair, pres, es, qs)
   293)    dqsdt    = latvap/(rh2o*tair*tair)*qs
   294)    alpha    = gravit*(latvap/(cpair*rh2o*tair*tair) - 1._r8/(rair*tair))
   295)    gammaloc = (1 + latvap/cpair*dqsdt)/(rhoair*qs)
   296)    ! growth coefficent Abdul-Razzak & Ghan 1998 eqn 16
   297)    ! should depend on mean radius of mode to account for gas kinetic effects
   298)    gloc     = 1._r8/(rhoh2o/(diff0*rhoair*qs)                               &
   299)               + latvap*rhoh2o/(conduct0*tair)*(latvap/(rh2o*tair) - 1._r8))
   300)    sqrtg    = sqrt(gloc)
   301)    beta     = 4._r8*pi*rhoh2o*gloc*gammaloc
   302)    etafactor2max = 1.e10_r8/(alpha*wbar)**1.5_r8 ! this should make eta big if na is very small.
   303) 
   304)    do m = 1, nmode
   305)       ! skip aerosols with no dispersion, since they aren't meant to be CCN
   306)       if (dispersion_aer(m) == 0._r8) then
   307)          smc(m)=100._r8
   308)          cycle
   309)       end if
   310)       ! internal mixture of aerosols
   311)       volc(m) = ma(m)/(density_aer(m)) ! only if variable size dist
   312)       if (volc(m) > 1.e-39_r8 .and. na(m) > 1.e-39_r8) then
   313)          etafactor2(m) = 1._r8/(na(m)*beta*sqrtg)  !fixed or variable size dist
   314)          ! number mode radius (m)
   315)          amcubeloc(m) = (3._r8*volc(m)/(4._r8*pi*exp45logsig(m)*na(m)))  ! only if variable size dist
   316)          smc(m) = smcrit(m) ! only for prescribed size dist
   317) 
   318)          if (hygro_aer(m) > 1.e-10_r8) then   ! loop only if variable size dist
   319)             smc(m) = 2._r8*aten*sqrt(aten/(27._r8*hygro_aer(m)*amcubeloc(m))) 
   320)          else
   321)             smc(m) = 100._r8
   322)          endif
   323)       else
   324)          smc(m) = 1._r8
   325)          etafactor2(m) = etafactor2max ! this should make eta big if na is very small.
   326)       end if
   327)       lnsmloc(m) = log(smc(m)) ! only if variable size dist
   328)    end do
   329) 
   330)    ! single  updraft
   331)    wnuc    = wbar
   332)    alw     = alpha*wnuc
   333)    sqrtalw = sqrt(alw)
   334)    zeta    = 2._r8*sqrtalw*aten/(3._r8*sqrtg)
   335)    etafactor1 = 2._r8*alw*sqrtalw
   336) 
   337)    do m = 1, nmode
   338)       ! skip aerosols with no dispersion, since they aren't meant to be CCN
   339)       if (dispersion_aer(m) /= 0._r8) eta(m) = etafactor1*etafactor2(m)
   340)    end do
   341) 
   342)    call maxsat(zeta, eta, nmode, smc, smax)
   343)    lnsmax = log(smax)
   344) 
   345)    nact = 0._r8
   346)    do m = 1, nmode
   347)       ! skip aerosols with no dispersion, since they aren't meant to be CCN
   348)       if (dispersion_aer(m) == 0._r8) cycle
   349)       x = 2*(lnsmloc(m) - lnsmax)/(3*sq2*alogsig(m))
   350)       nact = nact + 0.5_r8*(1._r8 - erf(x))*na(m)
   351)    end do
   352)    nact = nact/rhoair ! convert from #/m3 to #/kg
   353) 
   354)    deallocate( &
   355)       volc,       &
   356)       eta,        &
   357)       smc,        &
   358)       etafactor2, &
   359)       amcubeloc,  &
   360)       lnsmloc     )
   361) 
   362) end subroutine ndrop_bam_run
   363)

   366) subroutine ndrop_bam_ccn(lchnk, ncol, maerosol, naer2)
   367) 
   368)    !-------------------------------------------------------------------------------
   369)    !
   370)    ! Write diagnostic bulk aerosol ccn concentration
   371)    !
   372)    !-------------------------------------------------------------------------------
   373) 
   374)    ! Input arguments
   375)    integer,  intent(in) :: lchnk           ! chunk identifier
   376)    integer,  intent(in) :: ncol            ! number of columns
   377)    real(r8), intent(in) :: naer2(:,:,:)    ! aerosol number concentration (1/m3)
   378)    real(r8), intent(in) :: maerosol(:,:,:) ! aerosol mass conc (kg/m3)
   379) 
   380)    ! Local variables
   381)    integer :: i, k, l, m
   382)    real(r8) :: arg
   383) 
   384)    character*8, parameter :: ccn_name(psat)=(/'CCN1','CCN2','CCN3','CCN4','CCN5','CCN6'/)
   385)    real(r8) :: amcubesulfate(pcols)  ! cube of dry mode radius (m) of sulfate
   386)    real(r8) :: smcritsulfate(pcols)  ! critical supersatuation for activation of sulfate
   387)    real(r8) :: ccnfactsulfate
   388)    real(r8) :: ccn(pcols,pver,psat)  ! number conc of aerosols activated at supersat
   389)    !-------------------------------------------------------------------------------
   390) 
   391)    ccn(:ncol,:,:) = 0._r8
   392) 
   393)    do k = top_lev, pver
   394) 
   395)       do m = 1, naer_all
   396) 
   397)          if (m == idxsul) then
   398)             ! Lohmann treatment for sulfate has variable size distribution
   399)             do i = 1, ncol
   400)                if (naer2(i,k,m) > 0._r8) then 
   401)                   amcubesulfate(i) = amcubefactor(m)*maerosol(i,k,m)/(naer2(i,k,m))
   402)                   smcritsulfate(i) = smcritfactor(m)/sqrt(amcubesulfate(i))
   403)                else
   404)                   smcritsulfate(i) = 1._r8
   405)                end if
   406)             end do
   407)          end if
   408) 
   409)          do l = 1, psat
   410) 
   411)             if (m == idxsul) then
   412)                ! This code is modifying ccnfact for sulfate only.
   413)                do i = 1, ncol
   414)                   arg = argfactor(m)*log(smcritsulfate(i)/super(l))
   415)                   if (arg < 2) then
   416)                      if (arg < -2) then
   417)                         ccnfactsulfate = 1.0e-6_r8
   418)                      else
   419)                         ccnfactsulfate = 0.5e-6_r8*erfc(arg)
   420)                      end if
   421)                   else
   422)                      ccnfactsulfate = 0.0_r8
   423)                   end if
   424)                   ccn(i,k,l) = ccn(i,k,l) + naer2(i,k,m)*ccnfactsulfate
   425)                end do
   426)             else
   427)                ! Non-sulfate species use ccnfact computed by the init routine
   428)                ccn(:ncol,k,l) = ccn(:ncol,k,l) + naer2(:ncol,k,m)*ccnfact(l,m)
   429)             end if
   430) 
   431)          end do   ! supersaturation
   432)       end do      ! bulk aerosol
   433)    end do         ! level
   434) 
   435)    do l = 1, psat
   436)       call outfld(ccn_name(l), ccn(1,1,l), pcols, lchnk)
   437)    end do
   438) 
   439)    do l = 1, naer_all
   440)       call outfld(trim(aername(l))//'_m3', naer2(:,:,l), pcols, lchnk)
   441)    end do
   442) 
   443) end subroutine ndrop_bam_ccn

   447) subroutine maxsat(zeta, eta, nmode, smc, smax)
   448) 
   449)    ! calculates maximum supersaturation for multiple
   450)    ! competing aerosol modes.
   451) 
   452)    ! Abdul-Razzak and Ghan, A parameterization of aerosol activation.
   453)    ! 2. Multiple aerosol types. J. Geophys. Res., 105, 6837-6844.
   454) 
   455)    real(r8), intent(in) :: zeta
   456)    integer,  intent(in) :: nmode ! number of modes
   457)    real(r8), intent(in) :: smc(:) ! critical supersaturation for number mode radius
   458)    real(r8), intent(in) :: eta(:)
   459) 
   460)    real(r8), intent(out) :: smax ! maximum supersaturation
   461) 
   462)    integer :: m  ! mode index
   463)    real(r8) :: sum, g1, g2
   464) 
   465)    do m=1,nmode
   466)       if(zeta.gt.1.e5_r8*eta(m).or.smc(m)*smc(m).gt.1.e5_r8*eta(m))then
   467)          ! weak forcing. essentially none activated
   468)          smax=1.e-20_r8
   469)       else
   470)          ! significant activation of this mode. calc activation all modes.
   471)          go to 1
   472)       endif
   473)    enddo
   474) 
   475)    return
   476) 
   477) 1  continue
   478) 
   479)    sum=0
   480)    do m=1,nmode
   481)       if(eta(m).gt.1.e-20_r8)then
   482)          g1=sqrt(zeta/eta(m))
   483)          g1=g1*g1*g1
   484)          g2=smc(m)/sqrt(eta(m)+3*zeta)
   485)          g2=sqrt(g2)
   486)          g2=g2*g2*g2
