    60) elemental subroutine calc_ustar( t,    pmid, taux, tauy, &
    61)                                  rrho, ustar)
    62) 
    63)   !-----------------------------------------------------------------------!
    64)   ! Purpose: Calculate ustar and bottom level density (necessary for      !
    65)   !  Obukhov length calculation).                                         !
    66)   !-----------------------------------------------------------------------!
    67) 
    68)   real(r8), intent(in) :: t         ! surface temperature
    69)   real(r8), intent(in) :: pmid      ! midpoint pressure (bottom level)
    70)   real(r8), intent(in) :: taux      ! surface u stress [N/m2]
    71)   real(r8), intent(in) :: tauy      ! surface v stress [N/m2]
    72) 
    73)   real(r8), intent(out) :: rrho     ! 1./bottom level density
    74)   real(r8), intent(out) :: ustar    ! surface friction velocity [m/s]
    75) 
    76)   rrho = rair * t / pmid
    77)   ustar = max( sqrt( sqrt(taux**2 + tauy**2)*rrho ), ustar_min )
    78)   
    79) end subroutine calc_ustar

    81) elemental subroutine calc_obklen( ths,  thvs, qflx, shflx, rrho, ustar, &
    82)                                   khfs, kqfs, kbfs, obklen)
    83) 
    84)   !-----------------------------------------------------------------------!
    85)   ! Purpose: Calculate Obukhov length and kinematic fluxes.               !
    86)   !-----------------------------------------------------------------------!
    87) 
    88)   real(r8), intent(in)  :: ths           ! potential temperature at surface [K]
    89)   real(r8), intent(in)  :: thvs          ! virtual potential temperature at surface
    90)   real(r8), intent(in)  :: qflx          ! water vapor flux (kg/m2/s)
    91)   real(r8), intent(in)  :: shflx         ! surface heat flux (W/m2)
    92) 
    93)   real(r8), intent(in)  :: rrho          ! 1./bottom level density [ m3/kg ]
    94)   real(r8), intent(in)  :: ustar         ! Surface friction velocity [ m/s ]
    95)   
    96)   real(r8), intent(out) :: khfs          ! sfc kinematic heat flux [mK/s]
    97)   real(r8), intent(out) :: kqfs          ! sfc kinematic water vapor flux [m/s]
    98)   real(r8), intent(out) :: kbfs          ! sfc kinematic buoyancy flux [m^2/s^3]
    99)   real(r8), intent(out) :: obklen        ! Obukhov length
   100)   
   101)   ! Need kinematic fluxes for Obukhov:
   102)   khfs = shflx*rrho/cpair
   103)   kqfs = qflx*rrho
   104)   kbfs = khfs + zvir*ths*kqfs
   105)   
   106)   ! Compute Obukhov length:
   107)   obklen = -thvs * ustar**3 / (g*vk*(kbfs + sign(1.e-10_r8,kbfs)))
   108) 
   109) end subroutine calc_obklen
