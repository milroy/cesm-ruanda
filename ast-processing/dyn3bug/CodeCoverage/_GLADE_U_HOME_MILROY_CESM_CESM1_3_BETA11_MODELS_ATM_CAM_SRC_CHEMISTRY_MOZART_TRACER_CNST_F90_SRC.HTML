   219)   subroutine tracer_cnst_adv( pbuf2d, state )
   220) 
   221)     use physics_buffer, only : physics_buffer_desc
   222)     use tracer_data,  only : advance_trcdata
   223)     use physics_types,only : physics_state
   224)     use ppgrid,       only : begchunk, endchunk
   225)     use ppgrid,       only : pcols, pver
   226)     use string_utils, only : to_lower, GLC
   227)     use chem_mods,    only : fix_mass
   228)     use mo_chem_utls, only : get_inv_ndx
   229)     use cam_history,  only : outfld
   230)     use physconst,    only: mwdry   ! molecular weight dry air ~ kg/kmole
   231)     use physconst,    only: boltz
   232) 
   233)     implicit none
   234) 
   235)     type(physics_state), intent(in):: state(begchunk:endchunk)                 
   236)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   237) 
   238)     integer :: i,ind,c,ncol
   239)     real(r8) :: to_vmr(pcols,pver)
   240) 
   241)     if( num_tracer_cnst < 1 ) return
   242) 
   243)     call advance_trcdata( fields, file, state, pbuf2d )
   244) 
   245)     ! copy prescribed tracer fields into state variable with the correct units
   246) 
   247)     do i = 1,num_tracer_cnst
   248)        ind = get_inv_ndx( tracer_cnst_flds(i) )
   249)        do c = begchunk,endchunk
   250)           ncol = state(c)%ncol
   251) 
   252)           select case ( to_lower(trim(fields(i)%units(:GLC(fields(i)%units)))) )
   253)           case ("molec/cm3","/cm3","molecules/cm3","cm^-3","cm**-3")
   254)              to_vmr(:ncol,:) = (1.e6_r8*boltz*state(c)%t(:ncol,:))/(state(c)%pmiddry(:ncol,:))
   255)           case ('kg/kg','mmr')
   256)              to_vmr(:ncol,:) = mwdry/fix_mass(ind)
   257)           case ('mol/mol','mole/mole','vmr')
   258)              to_vmr(:ncol,:) = 1._r8
   259)           case default
   260)              write(iulog,*) 'tracer_cnst_adv: units = ',trim(fields(i)%units) ,' are not recognized'
   261)              call endrun('tracer_cnst_adv: units are not recognized')
   262)           end select
   263) 
   264)           fields(i)%data(:ncol,:,c) = to_vmr(:ncol,:) * fields(i)%data(:ncol,:,c)      ! vmr
   265)           call outfld( trim(tracer_cnst_flds(i)), fields(i)%data(:ncol,:,c), ncol, state(c)%lchnk )
   266) 
   267)        enddo
   268)     enddo
   269) 
   270)   end subroutine tracer_cnst_adv

   295)   subroutine get_cnst_data_ptr(name, state, q, pbuf)
   296) 
   297)     use tracer_data,   only : get_fld_data, get_fld_ndx
   298)     use physconst,     only : mwdry            ! molecular weight dry air ~ kg/kmole
   299)     use chem_mods,     only : fix_mass
   300)     use mo_chem_utls,  only : get_inv_ndx
   301)     use physics_types, only : physics_state
   302)     use ppgrid,        only : pcols, pver
   303)     use physics_buffer, only : physics_buffer_desc
   304) 
   305)     implicit none
   306) 
   307)     character(len=*),    intent(in) :: name
   308)     type(physics_state), intent(in) :: state
   309)     real(r8), pointer, dimension(:,:) :: q     ! constituent mass mixing ratio
   310)     type(physics_buffer_desc), pointer :: pbuf(:)
   311) 
   312)     integer :: lchnk
   313)     integer :: ncol
   314)     integer :: inv_id, idx
   315) 
   316)     lchnk = state%lchnk
   317)     ncol  = state%ncol
   318) 
   319)     ! make sure the requested constituent can be provided
   320)     inv_id = get_inv_ndx(name) 
   321)     if (.not. inv_id > 0) then
   322)        if (masterproc) then
   323)           write(iulog,*) 'get_cnst_data_ptr: '//name//' is not a prescribed tracer constituent'
   324)        endif
   325)        return
   326)     endif
   327) 
   328) 
   329)     call get_fld_ndx( fields, name, idx  )
   330)     call get_fld_data( fields, name, data_q(:,:,idx,lchnk), ncol, lchnk, pbuf  )
   331) 
   332)     data_q(:ncol,:,idx,lchnk) = data_q(:ncol,:,idx,lchnk)*fix_mass(inv_id)/mwdry   ! vmr->mmr
   333)     q => data_q(:,:,idx,lchnk)
   334) 
   335)   end subroutine get_cnst_data_ptr
