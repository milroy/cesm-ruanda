    68) ! !IROUTINE: shr_tInterp_getFactors -- calculate time interpolation factors
    69) !
    70) ! !DESCRIPTION:
    71) !     Returns two interpolation factors
    72) !     Legal algorithms are (algo):
    73) !       lower   - sets factors to data 1 (lower-bound), f1=1, f2=0
    74) !       upper   - sets factors to data 2 (upper-bound), f1=0, f2=1
    75) !       nearest - sets factors to nearest data in time
    76) !       linear  - sets factors to linear interpolation between lb and ub
    77) !     \newline
    78) !     call shr\_tInterp\_getFactors(D1,s1,D2,s2,D,s,f1,f2,'linear',rc)
    79) !     \newline
    80) !     time of 2 >= time of 1  for all algos
    81) !     time of 2 >= time of model data >= time of 1  for linear
    82) !
    83) ! !REVISION HISTORY:
    84) !     2005-Apr-10 - T. Craig - first version
    85) !
    86) ! !INTERFACE: ------------------------------------------------------------------
    87) 
    88) subroutine shr_tInterp_getFactors(D1,S1,D2,S2,Din,Sin,f1,f2,calendar,algo,rc)
    89) 
    90)    implicit none
    91) 
    92) ! !USES:
    93) 
    94) ! !INPUT/OUTPUT PARAMETERS:
    95) 
    96)    integer(SHR_KIND_IN),intent(in)           :: D1,S1   ! LB date & sec (20010115,3600)
    97)    integer(SHR_KIND_IN),intent(in)           :: D2,S2   ! UB date & sec
    98)    integer(SHR_KIND_IN),intent(in)           :: Din,Sin ! desired/model date & sec
    99)    real(SHR_KIND_R8)   ,intent(out)          :: f1      ! wgt for 1
   100)    real(SHR_KIND_R8)   ,intent(out)          :: f2      ! wgt for 2
   101)    character(*)        ,intent(in)           :: calendar!calendar type 
   102)    character(*)        ,intent(in) ,optional :: algo    ! algorithm
   103)    integer(SHR_KIND_IN),intent(out),optional :: rc      ! return code
   104) 
   105) !EOP
   106) 
   107)    !----- local  ------
   108)    type(ESMF_Time)        :: itime1      ! time for 1 (lower-bound)
   109)    type(ESMF_Time)        :: itime2      ! time for 2 (upper-bound)
   110)    type(ESMF_Time)        :: itimeIn     ! time for model date
   111)    type(ESMF_TimeInterval):: timeint     ! timeinterval
   112)    integer(SHR_KIND_I8)   :: snum, sden  ! delta times in seconds
   113)    integer(SHR_KIND_I8)   :: sint1,sint2 ! delta times in seconds
   114)    character(SHR_KIND_CS) :: lalgo       ! local algo variable
   115)    integer(SHR_KIND_IN)   :: lrc         ! local rc
   116) 
   117)    !----- formats -----
   118)    character(*),parameter :: subName = "(shr_tInterp_getFactors) "
   119)    character(*),parameter :: F00   = "('(shr_tInterp_getFactors) ',8a)" 
   120)    character(*),parameter :: F01   = "('(shr_tInterp_getFactors) ',a,2f17.8)" 
   121)    character(*),parameter :: F02   = "('(shr_tInterp_getFactors) ',a,3i9)" 
   122)    character(*),parameter :: F03   = "('(shr_tInterp_getFactors) ',2a,3(i9.8,i6))" 
   123) 
   124) !-------------------------------------------------------------------------------
   125) ! Computes time interpolation factors
   126) !-------------------------------------------------------------------------------
   127) 
   128)    lrc = 0
   129) 
   130)    if (present(algo)) then
   131)      lalgo = algo
   132)    else
   133)      lalgo = 'linear'
   134)    endif
   135) 
   136)    !--- compute elapsed time ---
   137) 
   138)    call shr_cal_timeSet(itimein,Din,Sin,calendar)
   139)    call shr_cal_timeSet(itime1 ,D1 ,S1 ,calendar)
   140)    call shr_cal_timeSet(itime2 ,D2 ,S2 ,calendar)
   141) 
   142)    !DML
   143) !  write(s_logunit,*) subName,' DTIME ',Din,D1,D2
   144) !  write(s_logunit,*) subName,' STIME ',sin,s1,s2
   145) !  write(s_logunit,*) subName,' ETIME ',etime1,etime2,etimein
   146) !  write(s_logunit,*) subName,' ITIME ',itime1,itime2,itimein
   147) 
   148)    ! --- always check that 1 <= 2, although we could relax this requirement ---
   149)    if (itime2 < itime1) then
   150)      if (s_loglev > 0) write(s_logunit,F01) ' ERROR: itime2 < itime1 D=',D1,S1,D2,S2
   151)      lrc = 1
   152)      call shr_tInterp_abort(subName//' itime2 < itime1 ')
   153)    endif
   154) 
   155)    f1 = -1.0
   156)    ! --- set interpolation factors ---
   157)    if (trim(lalgo) == 'lower') then
   158)      if (itime1 < itime2) then
   159)        f1 = c1
   160)      else
   161)        f1 = c0
   162)      endif
   163)    elseif (trim(lalgo) == 'upper') then
   164)      if (itime1 < itime2) then
   165)        f1 = c0
   166)      else
   167)        f1 = c1
   168)      endif
   169)    elseif (trim(lalgo) == 'nearest') then
   170)      timeint = itime1-itimein
   171)      call ESMF_TimeIntervalGet(timeint,StartTimeIn=itimein,s_i8=sint1)
   172)      timeint = itime2-itimein
   173)      call ESMF_TimeIntervalGet(timeint,StartTimeIn=itimein,s_i8=sint2)
   174)      if (abs(sint1) <= abs(sint2)) then
   175)        f1 = c1
   176)      else
   177)        f1 = c0
   178)      endif
   179)    elseif (trim(lalgo) == 'linear') then
   180)      !--- check that itimein is between itime1 and itime2 ---
   181)      if (itime2 < itimein .or. itime1 > itimein) then
   182)        write(s_logunit,F02) ' ERROR illegal linear times: ',D1,S1,Din,Sin,D2,S2
   183)        lrc = 1
   184)        call shr_tInterp_abort(subName//' illegal itimes ')
   185)      endif
   186)      if (itime2 == itime1) then
   187)        f1 = 0.5_SHR_KIND_R8
   188)      else
   189)        timeint = itime2 - itimein
   190)        call ESMF_TimeIntervalGet(timeint,StartTimeIn=itimein,s_i8=snum)
   191)        timeint = itime2 - itime1
   192)        call ESMF_TimeIntervalGet(timeint,StartTimeIn=itime1,s_i8=sden)
   193)        f1 = real(snum,SHR_KIND_R8)/real(sden,SHR_KIND_R8)
   194)      endif
   195)    else
   196)      if (s_loglev > 0) write(s_logunit,F00) 'ERROR: illegal lalgo option: ',trim(lalgo)
   197)      lrc = 1
   198)      call shr_tInterp_abort(subName//' illegal algo option '//trim(lalgo))
   199)    endif
   200) 
   201)    f2 = c1 - f1
   202) 
   203)    !--- check that f1 and f2 are OK, each between 0 and 1 and they sum to 1 ---
   204)    if (f1 < c0-eps .or. f1 > c1+eps .or. &
   205)        f2 < c0-eps .or. f2 > c1+eps .or. &
   206)        abs(f1+f2-c1) > eps) then
   207)      if (s_loglev > 0) write(s_logunit,F01) 'ERROR: illegal tInterp values ',f1,f2
   208)      lrc = 1
   209)      call shr_tInterp_abort(subName//' illegal tInterp values ')
   210)    endif
   211) 
   212)    if (debug > 0 .and. s_loglev > 0) then
   213)      write(s_logunit,F03) 'DEBUG: algo,D1,S1,Din,Sin,D2,S2=',trim(lAlgo),D1,S1,Din,Sin,D2,S2
   214)      write(s_logunit,F01) 'DEBUG: algo,f1,f2= '//trim(lAlgo),f1,f2
   215)    endif
   216) 
   217)    if (present(rc)) rc = lrc
   218) 
   219) end subroutine shr_tInterp_getFactors

   224) ! !IROUTINE: shr_tInterp_getAvgCosz -- returns avg cos(z) between LB and UB
   225) !
   226) ! !DESCRIPTION:
   227) !
   228) !   Returns a time-average of cos(z) over the time interval [LB,UB].  
   229) !   The time-avg is calculated via a right-sum Riemann sum where the partitian 
   230) !   width is the model dt, and the left-most partitian starts at the LB.
   231) !
   232) ! NOTE: For cosine of solar zenith angle forcing the time-stamps MUST be for
   233) !       the beginning of the interval.
   234) !
   235) ! !REVISION HISTORY:
   236) !     2010-Apr - B. Kauffman - change to t-avg cosz computation, uses model dt
   237) !     2009-Oct - T. Craig - migrated from dshr code
   238) !     2008-Jun - E. Kluzek - first version
   239) !
   240) ! !INTERFACE: ------------------------------------------------------------------
   241) 
   242) subroutine shr_tInterp_getAvgCosz(tavCosz,lonr,latr,ymd1,tod1,ymd2,tod2,eccen,mvelpp,lambm0,obliqr,dt,&
   243) 	                          calendar)
   244) 
   245)    implicit none
   246) 
   247) ! !USES:
   248) 
   249) ! !INPUT/OUTPUT PARAMETERS:
   250) 
   251)    real(SHR_KIND_R8)   ,intent(out)  :: tavCosz(:) ! t-avg of cosz over [LB,UB]
   252)    real(SHR_KIND_R8)   ,intent(in)   :: latr(:)    ! latitude
   253)    real(SHR_KIND_R8)   ,intent(in)   :: lonr(:)    ! longitude
   254)    integer(SHR_KIND_IN),intent(in)   :: ymd1,tod1  ! date of lb
   255)    integer(SHR_KIND_IN),intent(in)   :: ymd2,tod2  ! date of ub
   256)    real(SHR_KIND_R8)   ,intent(in)   :: eccen      ! orb param
   257)    real(SHR_KIND_R8)   ,intent(in)   :: mvelpp     ! orb param
   258)    real(SHR_KIND_R8)   ,intent(in)   :: lambm0     ! orb param
   259)    real(SHR_KIND_R8)   ,intent(in)   :: obliqr     ! orb param
   260)    integer(SHR_KIND_IN),intent(in)   :: dt         ! model time step in secs
   261)    character(*)        ,intent(in)   :: calendar   ! calendar type
   262) 
   263) !EOP
   264) 
   265)    !----- local  ------
   266)    integer(SHR_KIND_IN) :: lsize              ! size of local data
   267)    type(ESMF_Time)      :: reday1, reday2     ! LB, UB time
   268)    type(ESMF_TimeInterval) :: timeint         ! time interval
   269)    integer(SHR_KIND_I8) :: n                  ! number of time-samples in t-avg
   270)    type(ESMF_Time)      :: reday,reday0       ! time sample's elapsed seconds
   271)    integer(SHR_KIND_IN) :: ymd,tod,ymd0,tod0  ! used to compute time of time-sample
   272)    integer(SHR_KIND_IN) :: ldt                ! local dt as needed
   273)    integer(SHR_KIND_I8) :: ldt8               ! local dt as needed in i8
   274)    integer(SHR_KIND_I8) :: dtsec              ! delta time from timeint 
   275)    real(SHR_KIND_R8),pointer :: cosz(:)       ! cos(zenith angle)
   276) 
   277)    !----- formats -----
   278)    character(*),parameter :: subName = "(shr_tInterp_getAvgCosz) "
   279)    character(*),parameter :: F00   = "('(shr_tInterp_getAvgCosz) ',8a)" 
   280) 
   281) !-------------------------------------------------------------------------------
   282) ! Computes time avg cosz over interval [LB,UB]
   283) !-------------------------------------------------------------------------------
   284) 
   285)    lsize = size(lonr)
   286)    allocate(cosz(lsize))
   287)    if (lsize < 1 .or. size(latr) /= lsize .or. size(tavCosz) /= lsize) then
   288)       call shr_sys_abort(subname//' ERROR: lon lat tavCosz sizes disagree')
   289)    endif
   290) 
   291)    ldt = dt
   292) 
   293)    !--- get LB & UB dates ---
   294)    call shr_cal_timeSet(reday1,ymd1,tod1,calendar)
   295)    call shr_cal_timeSet(reday2,ymd2,tod2,calendar)
   296)    if (reday1 > reday2) call shr_sys_abort(subname//'ERROR: lower-bound > upper-bound')
   297) 
   298)    timeint = reday2-reday1
   299)    call ESMF_TimeIntervalGet(timeint,s_i8=dtsec)
   300)    ldt8 = ldt
   301)    if (mod(dtsec,ldt8) /= 0) then
   302)       ldt8 = (dtsec)/((dtsec)/ldt8+1)
   303)       ldt = ldt8
   304)    endif
   305) 
   306)    !--- compute time average ---
   307)    tavCosz = 0.0_SHR_KIND_R8 ! initialize partial sum
   308)    n       = 0               ! dt weighted average in t-avg
   309)    reday   = reday1          ! mid [LB,UB] interval t-step starts at LB 
   310)    ymd     = ymd1
   311)    tod     = tod1
   312)    do while( reday < reday2) ! mid-interval t-steps thru interval [LB,UB]
   313) 
   314)       !--- advance to next time in [LB,UB] ---
   315)       ymd0 = ymd
   316)       tod0 = tod
   317)       reday0 = reday
   318)       call shr_cal_advDateInt(ldt,'seconds',ymd0,tod0,ymd,tod,calendar)
   319)       call shr_cal_timeSet(reday,ymd,tod,calendar)
   320) 
   321)       if (reday > reday2) then
   322)          ymd = ymd2
   323)          tod = tod2
   324)          timeint = reday2-reday0
   325)          call ESMF_TimeIntervalGet(timeint,s_i8=dtsec)
   326)          ldt = dtsec
   327)       endif
   328) 
   329)       !--- get next cosz value for t-avg ---
   330)       call shr_tInterp_getCosz(cosz,lonr,latr,ymd,tod,eccen,mvelpp,lambm0,obliqr,calendar)
   331)       n = n + ldt
   332)       tavCosz = tavCosz + cosz*real(ldt,SHR_KIND_R8)  ! add to partial sum
   333) 
   334)    end do
   335)    tavCosz = tavCosz/real(n,SHR_KIND_R8) ! form t-avg
   336) 
   337)    deallocate( cosz )
   338) 
   339) end subroutine shr_tInterp_getAvgCosz

   344) ! !IROUTINE: shr_tInterp_getCosz -- calculate cosine(solar-zenith angle).
   345) !
   346) ! !DESCRIPTION:
   347) !
   348) !    Calculate the cos(solar-zenith angle).
   349) !
   350) ! !REVISION HISTORY:
   351) !     2010-Apr - B. Kauffman - returns cosz
   352) !     2009-Oct - T. Craig - added
   353) !
   354) ! !INTERFACE: ------------------------------------------------------------------
   355) 
   356) subroutine shr_tInterp_getCosz(cosz,lonr,latr,ymd,tod,eccen,mvelpp,lambm0,obliqr,calendar)
   357) 
   358)    implicit none
   359) 
   360) ! !USES:
   361) 
   362) ! !INPUT/OUTPUT PARAMETERS:
   363) 
   364)    real(SHR_KIND_R8)   ,intent(out)   :: cosz(:)    ! cos(zenith angle)
   365)    real(SHR_KIND_R8)   ,intent(in)    :: latr(:)    ! latitude
   366)    real(SHR_KIND_R8)   ,intent(in)    :: lonr(:)    ! longitude
   367)    integer(SHR_KIND_IN),intent(in)    :: ymd,tod    ! date of interest
   368)    real(SHR_KIND_R8)   ,intent(in)    :: eccen      ! orb param
   369)    real(SHR_KIND_R8)   ,intent(in)    :: mvelpp     ! orb param
   370)    real(SHR_KIND_R8)   ,intent(in)    :: lambm0     ! orb param
   371)    real(SHR_KIND_R8)   ,intent(in)    :: obliqr     ! orb param
   372)    character(*),        intent(in)    :: calendar   ! calendar type
   373) 
   374) !EOP
   375) 
   376)    !----- local  ------
   377)    integer(SHR_KIND_IN) :: n
   378)    integer(SHR_KIND_IN) :: lsize
   379)    real(SHR_KIND_R8)    :: calday      ! julian days
   380)    real(SHR_KIND_R8)    :: declin,eccf ! orb params
   381) 
   382)    real(SHR_KIND_R8),parameter :: solZenMin = 0.001_SHR_KIND_R8 ! min solar zenith angle
   383) 
   384)    !----- formats -----
   385)    character(*),parameter :: subName = "(shr_tInterp_getCosz) "
   386)    character(*),parameter :: F00   = "('(shr_tInterp_getCosz) ',8a)" 
   387) 
   388) !-------------------------------------------------------------------------------
   389) ! Returns cos(zenith angle)
   390) !-------------------------------------------------------------------------------
   391) 
   392)    lsize = size(lonr)
   393)    if (lsize < 1 .or. size(latr) /= lsize .or. size(cosz) /= lsize) then
   394)       call shr_sys_abort(subname//' ERROR: lon lat cosz sizes disagree')
   395)    endif
   396) 
   397)    call shr_cal_date2julian(ymd,tod,calday,calendar)
   398)    call shr_orb_decl( calday, eccen, mvelpp, lambm0, obliqr, declin, eccf)
   399)    do n = 1,lsize
   400)       cosz(n) = max(solZenMin, shr_orb_cosz( calday, latr(n), lonr(n), declin ))
   401)    end do
   402) 
   403) end subroutine shr_tInterp_getCosz
