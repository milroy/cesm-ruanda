    62) int GPTLget_memusage (int *size, int *rss, int *share, int *text, int *datastack)
    63) {
    64) #if defined (BGP) || defined(BGQ)
    65) 
    66)   long long alloc;
    67)   struct mallinfo m;
    68) #if defined (BGP)
    69)   Personality pers;
    70) #endif
    71) #if defined (BGQ)
    72)   uint64_t shared_mem_count;
    73) #endif
    74)   long long total;
    75)   int node_config;
    76)  
    77)  /* memory available */
    78) #if defined(BGP)
    79)   Kernel_GetPersonality(&pers, sizeof(pers));
    80)   total = BGP_Personality_DDRSizeMB(&pers);
    81) 
    82)   node_config  = BGP_Personality_processConfig(&pers);
    83)   if (node_config == _BGP_PERS_PROCESSCONFIG_VNM) total /= 4;
    84)   else if (node_config == _BGP_PERS_PROCESSCONFIG_2x2) total /= 2;
    85)   total *= 1024*1024;
    86) 
    87)   *size = total;
    88) #endif
    89) 
    90) #if defined(BGQ)
    91)   Kernel_GetMemorySize(KERNEL_MEMSIZE_SHARED, &shared_mem_count);
    92) 
    93)   shared_mem_count *= 1024*1024;
    94)   *size = shared_mem_count;
    95) 
    96) #endif
    97)   /* total memory used  - heap only (not static memory)*/
    98) 
    99)   m = mallinfo();
   100)   alloc = m.hblkhd + m.uordblks;
   101) 
   102)   *rss = alloc;
   103)   *share     = -1;
   104)   *text     = -1;
   105)   *datastack = -1;
   106) 
   107) 
   108) #elif (defined HAVE_SLASHPROC)
   109)   FILE *fd;                       /* file descriptor for fopen */
   110)   int pid;                        /* process id */
   111)   static char *head = "/proc/";   /* part of path */
   112)   static char *tail = "/statm";   /* part of path */
   113)   char file[19];                  /* full path to file in /proc */
   114)   int dum;                        /* placeholder for unused return arguments */
   115)   int ret;                        /* function return value */
   116) 
   117)   /*
   118)   ** The file we want to open is /proc/<pid>/statm
   119)   */
   120) 
   121)   pid = (int) getpid ();
   122)   if (pid > 999999) {
   123)     fprintf (stderr, "get_memusage: pid %d is too large\n", pid);
   124)     return -1;
   125)   }
   126) 
   127)   sprintf (file, "%s%d%s", head, pid, tail);
   128)   if ((fd = fopen (file, "r")) < 0) {
   129)     fprintf (stderr, "get_memusage: bad attempt to open %s\n", file);
   130)     return -1;
   131)   }
   132) 
   133)   /*
   134)   ** Read the desired data from the /proc filesystem directly into the output
   135)   ** arguments, close the file and return.
   136)   */
   137) 
   138)   ret = fscanf (fd, "%d %d %d %d %d %d %d", 
   139) 		size, rss, share, text, datastack, &dum, &dum);
   140)   ret = fclose (fd);
   141)   return 0;
   142) 
   143) #elif (defined __APPLE__)
   144) 
   145)   FILE *fd;
   146)   char cmd[60];  
   147)   int pid = (int) getpid ();
   148)   
   149)   sprintf (cmd, "ps -o vsz -o rss -o tsiz -p %d | grep -v RSS", pid);
   150)   fd = popen (cmd, "r");
   151) 
   152)   if (fd) {
   153)     fscanf (fd, "%d %d %d", size, rss, text);
   154)     *share     = -1;
   155)     *datastack = -1;
   156)     (void) pclose (fd);
   157)   }
   158) 
   159)   return 0;
   160) 
   161) #else
   162) 
   163)   struct rusage usage;         /* structure filled in by getrusage */
   164) 
   165)   if (getrusage (RUSAGE_SELF, &usage) < 0)
   166)     return -1;
   167)   
   168)   *size      = -1;
   169)   *rss       = usage.ru_maxrss;
   170)   *share     = -1;
   171)   *text      = -1;
   172)   *datastack = -1;
   173) #ifdef IRIX64
   174)   *datastack = usage.ru_idrss + usage.ru_isrss;
   175) #endif
   176)   return 0;
   177) 
   178) #endif
