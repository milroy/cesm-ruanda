   145)   subroutine get_tidal_coeffs( dcoef )
   146) 
   147)     !----------------------------------------------------------------------- 
   148)     ! Purpose: calculate fourier coefficients
   149)     !-----------------------------------------------------------------------
   150) 
   151)     use time_manager,  only: get_curr_date               
   152)     use physconst, only: pi, cday
   153) 
   154)     real(r8), intent(out) :: dcoef(4) 
   155) 
   156)  !  variables to calculate tidal coeffs
   157)     real(r8), parameter :: pi_x_2 = 2._r8*pi
   158)     real(r8), parameter :: pi_x_4 = 4._r8*pi
   159)     integer  :: year, month
   160)     integer  :: day              ! day of month
   161)     integer  :: tod              ! time of day (seconds past 0Z) 
   162)     real(r8) :: gmtfrac 
   163) 
   164)  !  calculate multipliers for Fourier transform in time (tidal analysis)
   165)     call get_curr_date(year, month, day, tod)
   166)     gmtfrac = tod / cday
   167) 
   168)     dcoef(1) = 2._r8*sin(pi_x_2*gmtfrac)
   169)     dcoef(2) = 2._r8*cos(pi_x_2*gmtfrac)
   170)     dcoef(3) = 2._r8*sin(pi_x_4*gmtfrac)
   171)     dcoef(4) = 2._r8*cos(pi_x_4*gmtfrac)
   172) 
   173)   end subroutine get_tidal_coeffs
