   239)    subroutine subcol_set_nsubcol(lchnk, ngrdcol, nsubcol)
   240)    !-----------------------------------------------------------------------
   241)    ! Set a chunk of the nsubcol module variable
   242)    ! Also, recompute indcol for lchnk
   243)    !-----------------------------------------------------------------------
   244) 
   245)       integer,    intent(in)    :: lchnk
   246)       integer,    intent(in)    :: ngrdcol
   247)       integer,    intent(in)    :: nsubcol(:)
   248) 
   249)       integer                   :: i, j, indx
   250) 
   251)       if (any(nsubcol(:) > psubcols)) then
   252)          call endrun('subcol_set_nsubcol: psubcols not set large enough to hold the number of subcolumns requested')
   253)       end if
   254)       if (any(nsubcol(:) < 0)) then
   255)          call endrun('subcol_set_nsubcol: nsubcols must be non-negative')
   256)       end if
   257)       if (ngrdcol < pcols) then
   258)          if (any(nsubcol(ngrdcol+1:) > 0)) then
   259)             call endrun('subcol_set_nsubcol: Cannot set subcolumns for columns past ngrdcol')
   260)          end if
   261)       end if
   262)       nsubcol2d(:, lchnk) = nsubcol(:)
   263)       ! Recalculate indcol for the chunk
   264)       indx = 1
   265)       do i = 1, pcols
   266)          do j = 1, nsubcol2d(i, lchnk)
   267)             indcol2d(indx, lchnk) = i
   268)             indx = indx + 1
   269)          end do
   270)       end do
   271)       ! Fill with zeros
   272)       if (indx <= pcols * psubcols) then
   273)          indcol2d(indx:pcols*psubcols, lchnk) = 0
   274)       end if
   275)    end subroutine subcol_set_nsubcol

   352)    subroutine subcol_field_copy_{DIMS}d{TYPE} (field, lchnk, field_sc)
   353)    !-----------------------------------------------------------------------
   354)    ! Copy a pbuf field dimensioned pcols to one with pcols*psubcols and fill appropriately
   355)    !-----------------------------------------------------------------------
   356) 
   357)       {VTYPE},          intent(in)      :: field{DIMSTR}
   358)       integer,          intent(in)      :: lchnk
   359)       {VTYPE},          intent(out)     :: field_sc{DIMSTR}   ! intent out
   360) 
   361)       ! Local Variables
   362)       integer                           :: ncol
   363)       integer                           :: indcol(pcols*psubcols)
   364)       integer :: i, indx, j
   365) 
   366)       if (size(field,dim=1) .ne. pcols) then
   367)          call endrun('subcol_field_copy error: only fields with first dimension pcols may use this routine')
   368)       end if
   369) 
   370)       field_sc = fillval_{TYPE}
   371)       ncol = subcol_get_ncol(lchnk)
   372)       call subcol_get_indcol(lchnk, indcol)
   373) 
   374) #if ({DIMS} == 1)
   375)       field_sc(:ncol) = field(indcol(:ncol))
   376) #endif
   377) 
   378) #if ({DIMS} == 2)
   379)       field_sc(:ncol,:) = field(indcol(:ncol),:)
   380) #endif
   381) 
   382) #if ({DIMS} == 3)
   383)       field_sc(:ncol,:,:) = field(indcol(:ncol),:,:)
   384) #endif
   385) 
   386) #if ({DIMS} == 4)
   387)       field_sc(:ncol,:,:,:) = field(indcol(:ncol),:,:,:)
   388) #endif
   389) 
   390) #if ({DIMS} == 5)
   391)       field_sc(:ncol,:,:,:,:) = field(indcol(:ncol),:,:,:,:)
   392) #endif
   393) 
   394)    end subroutine subcol_field_copy_{DIMS}d{TYPE}
