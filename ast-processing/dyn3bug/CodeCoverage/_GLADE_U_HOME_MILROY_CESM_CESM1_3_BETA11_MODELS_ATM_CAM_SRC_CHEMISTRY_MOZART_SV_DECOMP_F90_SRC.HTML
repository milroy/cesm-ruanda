   307) ! purpose: solves a*x = b
   308) !
   309) ! method:     
   310) ! solves a*x = b for a vector x, where a is specified by the arrays
   311) ! u,w,v as returned by svdcmp. m and n
   312) ! are the logical dimensions of a, and will be equal for square matrices.
   313) ! mp and np are the physical dimensions of a. b(1:m) is the input right-hand

   315) ! destroyed, so the routine may be called sequentially with different b

   322)       subroutine svbksb( u, w, v, m, n, mp, np, b, x )
   323) !------------------------------------------------------------------------- 
   324) !	... dummy arguments
   325) !------------------------------------------------------------------------- 
   326)       integer, intent(in)   :: m

   328)       integer, intent(in)   :: mp
   329)       integer, intent(in)   :: np
   330)       real(r8), intent(in)  :: u(mp,np)
   331)       real(r8), intent(in)  :: w(np)
   332)       real(r8), intent(in)  :: v(np,np)
   333)       real(r8), intent(in)  :: b(mp)
   334)       real(r8), intent(out) :: x(np)

   336) !------------------------------------------------------------------------- 
   337) !	... local variables
   338) !------------------------------------------------------------------------- 
   339)       integer  :: i, j, jj
   340)       real(r8) :: s
   341)       real(r8) :: tmp(nmax)
   342) 
   343)       do j = 1,n
   344)         s = 0._r8
   345)         if( w(j) /= 0._r8 ) then
   346)           do i = 1,m
   347)             s = s + u(i,j)*b(i)
   348)           end do
   349)           s = s/w(j)
   350)         endif

   356)         do jj = 1,n
   357)           s = s + v(j,jj)*tmp(jj)
   358)         end do
