     7) ! !MODULE: shr_stream_mod -- Data type and methods to manage input data streams.
     8) !
     9) ! !DESCRIPTION:
    10) !    A "data stream" is a sequence of input files where each file contains the 
    11) !    same set of data fields and all the data fields are on the same grid.
    12) !    The sequence of input data files provides an uninterupted time series of 
    13) !    data.
    14) !
    15) !    A stream data type stores information about one data stream, including the
    16) !    range of data date years to use and how data dates align with model dates.  
    17) !
    18) !    Given a model date, this module can return data dates that are upper and 
    19) !    lower time bounds around the given model date and the names of the files
    20) !    containing those dates.
    21) !
    22) ! !REMARKS:
    23) !

   156) ! !IROUTINE: shr_stream_init -- initialize stream datatype, read description text file
   157) !
   158) ! !DESCRIPTION:
   159) !
   160) ! !REMARKS:
   161) !    should input be via standard Fortran namelist?

   167) ! !INTERFACE: ------------------------------------------------------------------  
   168) 
   169) subroutine shr_stream_init(strm,infoFile,yearFirst,yearLast,yearAlign,taxMode,rc)
   170) 
   171) ! !INPUT/OUTPUT PARAMETERS:
   172) 
   173)    type(shr_stream_streamType)    ,intent(out) :: strm      ! data stream
   174)    character(*)                   ,intent(in)  :: infoFile  ! file with stream info, must read
   175)    integer  (SHR_KIND_IN)         ,intent(in)  :: yearFirst ! first year to use 
   176)    integer  (SHR_KIND_IN)         ,intent(in)  :: yearLast  ! last  year to use 
   177)    integer  (SHR_KIND_IN)         ,intent(in)  :: yearAlign ! align yearFirst with this model year
   178)    character(*)          ,optional,intent(in)  :: taxMode   ! time axis cycling option
   179)    integer  (SHR_KIND_IN),optional,intent(out) :: rc        ! return code
   180)  
   181) !EOP
   182) 
   183)    !----- local -----
   184)    integer  (SHR_KIND_IN) :: n             ! generic index
   185)    character(SHR_KIND_CL) :: str           ! string to parse from input data file
   186)    integer  (SHR_KIND_IN) :: int           ! integer to parse from input data file
   187)    character(SHR_KIND_CL) :: subStr        ! sub-string of interest
   188)    integer  (SHR_KIND_IN) :: nUnit         ! file i/o unit number
   189)    character(SHR_KIND_CS) :: startTag      ! input file start tag
   190)    character(SHR_KIND_CS) ::   endTag      ! input file   end tag
   191)    character(SHR_KIND_CS) :: fldNameFile   ! field name in data file field list
   192)    character(SHR_KIND_CS) :: fldNameModel  ! field name in model     field list
   193)    character(SHR_KIND_CXX):: fldListFile   ! list of data file fields, colon delim list
   194)    character(SHR_KIND_CXX):: fldListModel  ! list of model     fields, colon delim list
   195)    character(SHR_KIND_CL) :: calendar      ! stream calendar
   196)    integer  (SHR_KIND_IN) :: rCode, rCode2 ! return code
   197) 
   198)    !----- formats -----
   199)    character(*),parameter :: subName = '(shr_stream_init) ' 
   200)    character(*),parameter :: F00   = "('(shr_stream_init) ',8a)" 
   201) 
   202) !-------------------------------------------------------------------------------
   203) ! notes: 
   204) ! * should this use standard namelist input?
   205) ! * needs more robust error checking
   206) ! o yearFirst,yearLast,yearAlign are provided by calling routine
   207) ! o parse infoFile for remaining, except for...
   208) ! o fileNT,fileDates, & fileSecs, which are initially set to -1, but but are replaced with
   209) !   valid values as each file is opened for the first time
   210) !-------------------------------------------------------------------------------
   211) 
   212)    rCode = 0
   213)    write(s_logunit,F00) 'Reading file ',trim(infoFile)
   214) 
   215)    call shr_stream_default(strm)
   216) 
   217)    strm%yearFirst        = yearFirst
   218)    strm%yearLast         = yearLast
   219)    strm%yearAlign        = yearAlign
   220)    if (present(taxMode)) then
   221)       strm%taxMode       = trim(taxMode)
   222)    endif
   223) 
   224)    !-----------------------------------------------------------------------------  
   225)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  reading data source'
   226)    !-----------------------------------------------------------------------------  
   227) 
   228)    nUnit = shr_file_getUnit() ! get unused unit number
   229) 
   230)    !--- find start tag ---
   231)    startTag =  "<dataSource>"
   232)      endTag = "</dataSource>"
   233)    open(nUnit,file=infoFile,STATUS='OLD',FORM='FORMATTED',ACTION='READ',iostat=rCode)
   234)    if (rCode /= 0) goto 999
   235)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode)
   236)    if (rCode /= 0) goto 999
   237) 
   238)    !--- read data ---
   239)    read(nUnit,'(a)',END=999) str
   240)    call shr_string_leftAlign(str)
   241)    strm%dataSource = str
   242)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  * format = ', trim(strm%dataSource)
   243) 
   244)    close(nUnit)
   245)    call shr_file_freeUnit(nUnit)
   246) 
   247)    !-----------------------------------------------------------------------------  
   248)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  reading field data variable names'
   249)    !-----------------------------------------------------------------------------  
   250) 
   251)    !--- find start tag ---
   252)    open(nUnit,file=infoFile,STATUS='OLD',FORM='FORMATTED',ACTION='READ')
   253)    startTag =  "<fieldInfo>"
   254)      endTag = "</fieldInfo>"
   255)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode)
   256)    if (rCode /= 0) goto 999
   257)    startTag =  "<variableNames>"
   258)      endTag = "</variableNames>"
   259)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode)
   260)    if (rCode /= 0) goto 999
   261) 
   262)    !--- read data ---
   263)    n=0
   264)    do while (.true.)
   265)       read(nUnit,'(a)',END=999) str
   266)       call shr_string_leftAlign(str)
   267)       n=n+1
   268)       if (str(1:len_trim(endTag)) == trim(endTag)) exit
   269)       fldNameFile  = ""
   270)       fldNameModel = ""
   271)       read(str,*,iostat=rCode) fldNameFile,fldNameModel
   272)       if (len_trim(fldNameFile)==0 .or. len_trim(fldNameModel)==0 ) then
   273)          rCode = 1
   274)          write(s_logunit,F00) "ERROR: reading field names"
   275)          write(s_logunit,F00) '* fldNameFile  = ',trim(fldNameFile)
   276)          write(s_logunit,F00) '* fldNameModel = ',trim(fldNameModel)
   277)          call shr_stream_abort(subName//"ERROR: reading field names")
   278)       end if
   279)       if (n==1) then
   280)          strm%fldListFile  = trim(fldNameFile )
   281)          strm%fldListModel = trim(fldNameModel)
   282)       else  
   283)          strm%fldListFile  = trim(strm%fldListFile ) // ":" // trim(fldNameFile )
   284)          strm%fldListModel = trim(strm%fldListModel) // ":" // trim(fldNameModel)
   285)       end if
   286)    end do
   287)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  * file  field list = ',trim(strm%fldListFile )
   288)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  * model field list = ',trim(strm%fldListModel)
   289)    if (n==0) then
   290)       rCode = 1
   291)       write(s_logunit,F00) "ERROR: no input field names"
   292)       call shr_stream_abort(subName//"ERROR: no input field names")
   293)    end if
   294) 
   295)    close(nUnit) 
   296) 
   297)    !-----------------------------------------------------------------------------  
   298)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  reading time-interpolation alogrithm '
   299)    !-----------------------------------------------------------------------------  
   300) 
   301)    !--- find start tag ---
   302)    open(nUnit,file=infoFile,STATUS='OLD',FORM='FORMATTED',ACTION='READ')
   303)    startTag =  "<fieldInfo>"
   304)      endTag = "</fieldInfo>"
   305)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode2)
   306)    if (rCode2 /= 0)then
   307)       rCode = rCode2
   308)       goto 999
   309)    end if
   310) 
   311)    close(nUnit) 
   312) 
   313)    !-----------------------------------------------------------------------------  
   314)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  reading offset'
   315)    !-----------------------------------------------------------------------------  
   316) 
   317)    !--- find start tag ---
   318)    open(nUnit,file=infoFile,STATUS='OLD',FORM='FORMATTED',ACTION='READ')
   319)    startTag =  "<fieldInfo>"
   320)      endTag = "</fieldInfo>"
   321)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode2)
   322)    if (rCode2 /= 0)then
   323)       rCode = rCode2
   324)       goto 999
   325)    end if
   326)    startTag =  "<offset>"
   327)      endTag = "</offset>"
   328)    call shr_stream_readUpToTag(nUnit,startTag,optionalTag=.true.,rc=rCode2)
   329)    if (rCode2 == 0) then
   330)       !--- read data ---
   331)       read(nUnit,*,END=999) int
   332)       strm%offset = int
   333)    else
   334)       strm%offset = 0
   335)    end if
   336)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  * offset ',strm%offset
   337) 
   338)    close(nUnit) 
   339) 
   340)    !-----------------------------------------------------------------------------  
   341)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  reading data file path'
   342)    !-----------------------------------------------------------------------------  
   343) 
   344)    !--- find start tag ---
   345)    open(nUnit,file=infoFile,STATUS='OLD',FORM='FORMATTED',ACTION='READ')
   346)    startTag =  "<fieldInfo>"
   347)      endTag = "</fieldInfo>"
   348)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode2)
   349)    if (rCode2 /= 0)then
   350)       rCode = rCode2
   351)       goto 999
   352)    end if
   353)    startTag =  "<filePath>"
   354)      endTag = "</filePath>"
   355)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode2)
   356)    if (rCode2 /= 0)then
   357)       rCode = rCode2
   358)       goto 999
   359)    end if
   360) 
   361)    !--- read data ---
   362)    read(nUnit,'(a)',END=999) str
   363)    call shr_string_leftAlign(str)
   364)    n = len_trim(str)
   365)    if (n>0 .and. str(n:n) /= '/') str(n+1:n+2) = "/ " ! must have trailing slash
   366)    if (n==0) str = "./ "                              ! null path => ./
   367)    strm%FilePath = str
   368)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  * data file path = ', trim(strm%FilePath)
   369) 
   370)    close(nUnit) 
   371) 
   372)    !-----------------------------------------------------------------------------  
   373)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  reading field data file names'
   374)    !-----------------------------------------------------------------------------  
   375) 
   376)    !--- find start tag ---
   377)    open(nUnit,file=infoFile,STATUS='OLD',FORM='FORMATTED',ACTION='READ')
   378)    startTag =  "<fieldInfo>"
   379)      endTag = "</fieldInfo>"
   380)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode2)
   381)    if (rCode2 /= 0)then
   382)       rCode = rCode2
   383)       goto 999
   384)    end if
   385)    startTag =  "<fileNames>"
   386)      endTag = "</fileNames>"
   387)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode2)
   388)    if (rCode2 /= 0)then
   389)       rCode = rCode2
   390)       goto 999
   391)    end if
   392) 
   393)    !--- read data ---
   394)    n=0
   395)    do while (.true.)
   396)       read(nUnit,'(a)',END=999) str
   397)       call shr_string_leftAlign(str)
   398)       if (str(1:len_trim(endTag)) == trim(endTag)) exit
   399)       n=n+1
   400)       if (n > nFileMax) then
   401)          rCode = 1
   402)          write(s_logunit,F00) "ERROR: exceeded max number of files"
   403)          call shr_stream_abort(subName//"ERROR: exceeded max number of files")
   404)          if ( present(rc) ) rc = rCode
   405)          close(nUnit) 
   406)          call shr_file_freeUnit(nUnit)
   407)          return
   408)       end if
   409)       strm%file(n)%name = str
   410)       if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  * ',trim(strm%file(n)%name)
   411)    end do
   412)    strm%nFiles = n
   413)    if (n==0) then
   414)       rCode = 1
   415)       write(s_logunit,F00) "ERROR: no input file names"
   416)       call shr_stream_abort(subName//"ERROR: no input file names")
   417)    end if
   418) 
   419)    close(nUnit) 
   420) 
   421)    !-----------------------------------------------------------------------------  
   422)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  reading domain data variable names'
   423)    !-----------------------------------------------------------------------------  
   424) 
   425)    !--- find start tag ---
   426)    open(nUnit,file=infoFile,STATUS='OLD',FORM='FORMATTED',ACTION='READ')
   427)    startTag =  "<domainInfo>"
   428)      endTag = "</domainInfo>"
   429)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode2)
   430)    if (rCode2 /= 0)then
   431)       rCode = rCode2
   432)       goto 999
   433)    end if
   434)    startTag =  "<variableNames>"
   435)      endTag = "</variableNames>"
   436)    call shr_stream_readUpToTag(nUnit,startTag,rc=rCode2)
   437)    if (rCode2 /= 0)then
   438)       rCode = rCode2
   439)       goto 999
   440)    end if
   441) 
   442)    !--- read data ---
   443)    n=0
   444)    do while (.true.)
   445)       read(nUnit,'(a)',END=999) str
   446)       call shr_string_leftAlign(str)
   447)       n=n+1
   448)       if (str(1:len_trim(endTag)) == trim(endTag)) exit
   449)       fldNameFile  = ""
   450)       fldNameModel = ""
   451)       read(str,*,iostat=rCode2) fldNameFile,fldNameModel
   452)       if (len_trim(fldNameFile)==0 .or. len_trim(fldNameModel)==0 ) then
   453)          rCode = 1
   454)          write(s_logunit,F00) "ERROR: reading field names"
   455)          write(s_logunit,F00) '* fldNameFile  = ',trim(fldNameFile)
   456)          write(s_logunit,F00) '* fldNameModel = ',trim(fldNameModel)
   457)          call shr_stream_abort(subName//"ERROR: reading field names")
   458)       end if
   459)       if (n==1) then
   460)          fldListFile  = trim(fldNameFile )
   461)          fldListModel = trim(fldNameModel)
   462)       else  
   463)          fldListFile  = trim(fldListFile ) // ":" // trim(fldNameFile )
   464)          fldListModel = trim(fldListModel) // ":" // trim(fldNameModel)
   465)       end if
   466)    end do
   467)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  * file  field list = ',trim(fldListFile )
   468)    if (debug>0 .and. s_loglev>0) write(s_logunit,F00) '  * model field list = ',trim(fldListModel)
   469) 
   470)    if (n==0) then

  1565) subroutine shr_stream_verifyTCoord(strm,k,rc)
  1566) 
  1567) ! !INPUT/OUTPUT PARAMETERS:
  1568) 
  1569)    type(shr_stream_streamType),intent(in) :: strm  ! data stream
  1570)    integer(SHR_KIND_IN)                   :: k     ! index of file to check
  1571)    integer(SHR_KIND_IN)                   :: rc    ! return code

  1573) !EOP

  1575)    !----- local -----
  1576)    integer(SHR_KIND_IN)   :: n           ! generic loop index

  1578)    integer(SHR_KIND_IN)   :: date1,secs1 ! date and seconds for a    time coord

  1582)    !----- formats -----

  1585)    character(*),parameter :: F01   = "('(shr_stream_verifyTCoord) ',a,2i7)" 
  1586)    character(*),parameter :: F02   = "('(shr_stream_verifyTCoord) ',a,2i9.8)" 
  1587) 
  1588) !-------------------------------------------------------------------------------
  1589) ! Notes:
  1590) !   o checks that dates are increasing (must not decrease)
  1591) !   o does not check for valid dates (eg. day=0 & month = 13 are "OK")
  1592) !   o checks that secs are strictly increasing within any one day
  1593) !   o checks that 0 <= secs <= spd (seconds per day)
  1594) !   o checks all dates from one file plus last date of previous file and 
  1595) !     first date of next file
  1596) !-------------------------------------------------------------------------------
  1597) 
  1598)    rc = 0
  1599)    if (debug>1 .and. s_loglev > 0) write(s_logunit,F01) "checking t-coordinate data   for file k =",k
  1600)

  1602)       rc = 1
  1603)       write(s_logunit,F01) "Don't have data for file ",k
  1604)       call shr_stream_abort(subName//"ERROR: can't check -- file not read.")
  1605)       return
  1606)    end if
  1607) 
  1608)    do n=1,strm%file(k)%nt+1
  1609)       checkIt = .false.
  1610)

  1615)             if ( strm%file(k-1)%haveData ) then

  1617)                date1 = strm%file(k-1)%date(nt)
  1618)                secs1 = strm%file(k-1)%secs(nt)
  1619)                date2 = strm%file(k  )%date(n)

  1623)             end if
  1624)          end if
  1625)       else if (n==strm%file(k)%nt+1) then
  1626)          !--- compare with next file? ---
  1627)          if (k<strm%nFiles) then
  1628)             if ( strm%file(k+1)%haveData ) then
  1629)                nt    = strm%file(k  )%nt
  1630)                date1 = strm%file(k  )%date(nt)
  1631)                secs1 = strm%file(k  )%secs(nt)
  1632)                date2 = strm%file(k+1)%date(1)
  1633)                secs2 = strm%file(k+1)%secs(1)
  1634)                checkIt = .true.
  1635)                if (debug>1 .and. s_loglev > 0) write(s_logunit,F01) "comparing with next     file for file k =",k
  1636)             end if
  1637)          end if
  1638)       else
  1639)          !--- compare within this file ---
  1640)          date1 = strm%file(k)%date(n-1)
  1641)          secs1 = strm%file(k)%secs(n-1)
  1642)          date2 = strm%file(k)%date(n  )
  1643)          secs2 = strm%file(k)%secs(n  )
  1644)          checkIt = .true.
  1645)       end if
  1646) 
  1647)       !--- compare two consecutive dates ---
  1648)       if (checkIt) then
  1649)          if ( date1 > date2 ) then
  1650)             rc = 1
  1651)             write(s_logunit,F01) "ERROR: calendar dates must be increasing"
  1652)             write(s_logunit,F02) "date(n), date(n+1) = ",date1,date2
  1653)             call shr_stream_abort(subName//"ERROR: calendar dates must be increasing")
  1654)             return
  1655)          else if ( date1 == date2 ) then
  1656)             if ( secs1 >= secs2 ) then
  1657)                rc = 1
  1658)                write(s_logunit,F01) "ERROR: elapsed seconds on a date must be strickly increasing"
  1659)                write(s_logunit,F02) "secs(n), secs(n+1) = ",secs1,secs2
  1660)                call shr_stream_abort(subName//"ERROR: elapsed seconds must be increasing")
  1661)                return
  1662)             end if
  1663)          end if
  1664)          if ( secs1 < 0 .or. spd < secs1 ) then
  1665)             rc = 1
  1666)             write(s_logunit,F01) "ERROR: elapsed seconds out of valid range [0,spd]"
  1667)             write(s_logunit,F02) "secs(n) = ",secs1
  1668)             call shr_stream_abort(subName//"ERROR: elapsed seconds out of range")
  1669)             return
  1670)          end if
  1671)       end if
  1672)    end do
  1673) 
  1674)    if (debug>0 .and. s_loglev > 0) write(s_logunit,F01) "data is OK (non-decreasing)  for file k =",k
  1675) 
  1676) end subroutine shr_stream_verifyTCoord
