    97) 
    98)     !
    99)     ! check horizontal grid ...

   101)     call pio_seterrorhandling( indata%piofile, PIO_BCAST_ERROR)
   102)     lonname = 'ncol'

   109)     call pio_seterrorhandling( indata%piofile, PIO_INTERNAL_ERROR)
   110)

   112)     ierr = pio_inq_dimlen( indata%piofile, id, nlons )

   114)     call get_horiz_grid_dim_d(hdim1_d,hdim2_d)

   116)     if (hdim1_d /= nlons) then

   118)     endif
   119) 
   120)     !
   121)     ! get time/date info ...
   122)     !

   125)

   127) 
   128)     ierr = pio_inq_varid( indata%piofile, 'date',  id  )
   129)     ierr = pio_get_var( indata%piofile, id, indata%dates )
   130) 
   131)     ierr = pio_inq_varid( indata%piofile, 'datesec',  id  )

   133)
