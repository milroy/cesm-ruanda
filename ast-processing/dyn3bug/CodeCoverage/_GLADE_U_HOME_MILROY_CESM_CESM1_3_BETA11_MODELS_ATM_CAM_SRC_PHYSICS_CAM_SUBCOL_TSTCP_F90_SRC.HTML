    97)    subroutine subcol_gen_tstcp(state, tend, state_sc, tend_sc, pbuf)
    98) 
    99)       use subcol_utils,   only: subcol_set_subcols, subcol_get_nsubcol, subcol_set_weight, subcol_set_filter
   100)       use physics_buffer, only: physics_buffer_desc, pbuf_get_field, col_type_subcol
   101)       use phys_grid,      only: get_gcol_p
   102)       use time_manager,   only: is_first_step, is_first_restart_step

   105)       !-----------------------------------
   106)       ! sub-column generator
   107)       !-----------------------------------
   108)       type(physics_state), intent(inout) :: state
   109)       type(physics_tend),  intent(inout) :: tend
   110)       type(physics_state), intent(inout) :: state_sc        ! sub-column state
   111)       type(physics_tend),  intent(inout) :: tend_sc         ! sub-column tend
   112)       type(physics_buffer_desc), pointer :: pbuf(:)
   113) 
   114) 
   115)       !
   116)       ! Local variables
   117)       !
   118)       integer            :: i, k, ngrdcol, indx, indx1, indx2
   119)       integer            :: nsubcol(pcols)
   120)       real(r8)           :: weight(state_sc%psetcols)
   121)       integer            :: filter(state_sc%psetcols)
   122)       integer, pointer   :: test_field(:,:)
   123)       character(len=128) :: errmsg
   124) 
   125)       ngrdcol    = state%ngrdcol
   126) 
   127)       !----------------------
   128)       ! Set the number of subcolumns on the 0th time step -- current implementation does not allow
   129)       ! number of subcolumns to vary within a run.  Cannot be done in init as ngrdcol is not known
   130)       ! at init
   131)       !----------------------
   132)       ! Test differing number of subcolumns by setting columns > 45 degrees to
   133)       ! have 1 subcolumn, columns < -45 to 2 subcolumns and others to 3 subcols
   134)       if (is_first_step()) then
   135)          nsubcol = 0
   136)          do i = 1, ngrdcol
   137)             if (state%lat(i) > 0.7854_r8) then
   138)                nsubcol(i) = 1
   139)             else if (state%lat(i) < -0.7854_r8) then
   140)                nsubcol(i) = 2
   141)             else
   142)                nsubcol(i) = psubcols
   143)             end if
   144)          end do
   145)       else
   146)          call subcol_get_nsubcol(state%lchnk, nsubcol)
   147)          ! Since this is a test generator, check for nsubcol correctness.
   148) 10       format(a,i3,a,i5)
   149)          do i = 1, pcols
   150)             if (i > ngrdcol) then
   151)                if (nsubcol(i) /= 0) then
   152)                   write(errmsg, 10) 'subcol_gen_tstcp: Bad value for nsubcol(',&
   153)                        i,') = ',nsubcol(i),', /= 0'
   154)                   call endrun(errmsg)
   155)                end if
   156)             else if (state%lat(i) > 0.7854_r8) then
   157)                if (nsubcol(i) /= 1) then
   158)                   write(errmsg, 10) 'subcol_gen_tstcp: Bad value for nsubcol(',&
   159)                        i,') = ',nsubcol(i),', /= 1'
   160)                   call endrun(errmsg)
   161)                end if
   162)             else if (state%lat(i) < -0.7854_r8) then
   163)                if (nsubcol(i) /= 2) then
   164)                   write(errmsg, 10) 'subcol_gen_tstcp: Bad value for nsubcol(',&
   165)                        i,') = ',nsubcol(i),', /= 2'
   166)                   call endrun(errmsg)
   167)                end if
   168)             else
   169)                if (nsubcol(i) /= psubcols) then
   170)                   write(errmsg, 10) 'subcol_gen_tstcp: Bad value for nsubcol(',&
   171)                        i,') = ',nsubcol(i),', /=',psubcols
   172)                   call endrun(errmsg)
   173)                end if
   174)             end if
   175)          end do
   176)       end if
   177) 
   178)       call subcol_set_subcols(state, tend, nsubcol, state_sc, tend_sc)
   179) 
   180)       ! For perturb case, adjust Temperature up and down one degree
   181)       if (subcol_tstcp_perturb) then
   182)         indx=1
   183)         do i=1,ngrdcol
   184)           if (nsubcol(i) >= 2) then
   185)             state_sc%t(indx,:) = state_sc%t(indx,:)+1
   186)             state_sc%t(indx+1,:) = state_sc%t(indx+1,:)-1
   187)           end if
   188)           indx=indx+nsubcol(i)
   189)         end do
   190)       end if
   191)           
   192)       ! Set weight to 1 for first column, 0 for all others -- will be BFB with noUniAv case
   193)       if(subcol_tstcp_filter .and. subcol_tstcp_weight) then
   194)         weight=1._r8
   195)         ! Initialize to 1 - will match doAv_noUni, init to 0 - will match noUniAv
   196)         filter=1
   197)         indx=1
   198)         do i=1,ngrdcol
   199)            weight(indx) = 1.0_r8
   200)            filter(indx) = 1
   201)            indx = indx+nsubcol(i)
   202)         end do
   203)         call subcol_set_weight(state%lchnk, weight)
   204)         call subcol_set_filter(state%lchnk, filter)
   205)       ! Set weight to 1 for first column, 0 for all others -- will be BFB with noUniAv case
   206)       else if(subcol_tstcp_weight) then
   207)         weight=0._r8
   208)         indx=1
   209)         do i=1,ngrdcol
   210)            weight(indx) = 1.0_r8
   211)            indx = indx+nsubcol(i)
   212)         end do
   213)         call subcol_set_weight(state%lchnk, weight)
   214) 
   215)       ! Set filter to 1 for first column, 0 for all others -- will be BFB with noUniAv case
   216)       else if(subcol_tstcp_filter) then
   217)         filter=0
   218)         indx=1
   219)         do i=1,ngrdcol
   220)            filter(indx) = 1
   221)            indx = indx+nsubcol(i)
   222)         end do
   223)         call subcol_set_filter(state%lchnk, filter)
   224)       end if
   225) 
   226) 
   227)       if (is_first_restart_step()) then
   228)          ! Test values for the test pbuf
   229)          call pbuf_get_field(pbuf, tstcpy_scol_idx, test_field,               &
   230)               col_type=col_type_subcol, copy_if_needed=.false.)
   231)          indx = 1
   232)          do i=1,ngrdcol
   233)             do indx1 = 1, nsubcol(i)
   234)                do k = 1, pver
   235)                   indx2 = (get_gcol_p(state%lchnk, i) * 10000)
   236)                   indx2 = k + (100 * (indx1 + indx2))
   237)                   if(test_field(indx, k) /= indx2) then
   238)                      write(iulog, *) 'TSTCPY_SCOL check(',indx,',',k,         &
   239)                           '): expected',indx2,', found',test_field(indx, k)
   240)                      call endrun("Restart check for TSTCPY_SCOL failed")
   241)                   end if
   242)                end do
   243)                indx = indx + 1
   244)             end do
   245)          end do
   246)          ! Unused subcolumn space is not initialized so no check
   247)       else if (is_first_step()) then
   248)          ! Set values for the test pbuf
   249)          call pbuf_get_field(pbuf, tstcpy_scol_idx, test_field,               &
   250)               col_type=col_type_subcol, copy_if_needed=.false.)
   251)          test_field = -1
   252)          indx = 1
   253)          do i=1,ngrdcol
   254)             do indx1 = 1, nsubcol(i)
   255)                do k = 1, pver
   256)                   indx2 = (get_gcol_p(state%lchnk, i) * 10000)
   257)                   indx2 = k + (100 * (indx1 + indx2))
   258)                   test_field(indx, k) = indx2
   259)                end do
   260)                indx = indx + 1
   261)             end do
   262)          end do
   263)       end if
   264) 
   265) end subroutine subcol_gen_tstcp
