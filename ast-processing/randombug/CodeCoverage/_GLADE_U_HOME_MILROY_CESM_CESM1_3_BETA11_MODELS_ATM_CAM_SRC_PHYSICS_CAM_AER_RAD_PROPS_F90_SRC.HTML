    45) subroutine aer_rad_props_init()
    46)    use phys_control, only: phys_getopts
    47) 
    48) 
    49)    integer                    :: i
    50)    integer                    :: numaerosols  ! number of aerosols
    51)    character(len=64), pointer :: aernames(:)  ! aerosol names
    52)    logical                    :: history_amwg         ! output the variables used by the AMWG diag package
    53)    logical                    :: history_aero_optics  ! Output aerosol optics diagnostics
    54)    logical                    :: prog_modal_aero      ! Prognostic modal aerosols present
    55) 
    56)    !----------------------------------------------------------------------------
    57) 
    58)    call phys_getopts( history_aero_optics_out    = history_aero_optics, &
    59)                       history_amwg_out           = history_amwg,    &
    60)                       prog_modal_aero_out        = prog_modal_aero )
    61) 
    62)    ! Limit modal aerosols with top_lev here.
    63)    if (prog_modal_aero) top_lev = clim_modal_aero_top_lev
    64) 
    65)    call addfld ('AEROD_v ', '1', 1, 'A', &
    66)       'Total Aerosol Optical Depth in visible band', phys_decomp, flag_xyfill=.true.)
    67) 
    68)    ! Contributions to AEROD_v from individual aerosols (climate species).
    69) 
    70)    ! number of bulk aerosols in climate list
    71)    call rad_cnst_get_info(0, naero=numaerosols)
    72) 
    73)    ! get names of bulk aerosols
    74)    allocate(aernames(numaerosols))
    75)    call rad_cnst_get_info(0, aernames=aernames)
    76) 
    77)    ! diagnostic output for bulk aerosols
    78)    ! create outfld names for visible OD
    79)    allocate(odv_names(numaerosols))
    80)    do i = 1, numaerosols
    81)       odv_names(i) = 'ODV_'//trim(aernames(i))
    82)       call addfld (odv_names(i), '1', 1, 'A', &
    83)          trim(aernames(i))//' optical depth in visible band', phys_decomp, flag_xyfill=.true.)
    84)    end do
    85) 
    86)    ! Determine default fields
    87)    if (history_amwg ) then 
    88)       call add_default ('AEROD_v', 1, ' ')
    89)    endif   
    90)    
    91)    if ( history_aero_optics ) then 
    92)       call add_default ('AEROD_v', 1, ' ')
    93)       do i = 1, numaerosols
    94)          odv_names(i) = 'ODV_'//trim(aernames(i))
    95)          call add_default (odv_names(i), 1, ' ')
    96)       end do
    97)     endif
    98) 
    99) 
   100)    deallocate(aernames)
   101) 
   102) end subroutine aer_rad_props_init

   106) subroutine aer_rad_props_sw(list_idx, state, pbuf,  nnite, idxnite, &
   107)                             tau, tau_w, tau_w_g, tau_w_f)
   108) 
   109)    ! Return bulk layer tau, omega, g, f for all spectral intervals.
   110) 
   111)    use physics_buffer, only : physics_buffer_desc
   112)    ! Arguments
   113)    integer,             intent(in) :: list_idx      ! index of the climate or a diagnostic list
   114)    type(physics_state), intent(in), target :: state
   115)    
   116)    type(physics_buffer_desc), pointer :: pbuf(:)
   117)    integer,             intent(in) :: nnite                ! number of night columns
   118)    integer,             intent(in) :: idxnite(:)           ! local column indices of night columns
   119) 
   120)    real(r8), intent(out) :: tau    (pcols,0:pver,nswbands) ! aerosol extinction optical depth
   121)    real(r8), intent(out) :: tau_w  (pcols,0:pver,nswbands) ! aerosol single scattering albedo * tau
   122)    real(r8), intent(out) :: tau_w_g(pcols,0:pver,nswbands) ! aerosol assymetry parameter * tau * w
   123)    real(r8), intent(out) :: tau_w_f(pcols,0:pver,nswbands) ! aerosol forward scattered fraction * tau * w
   124) 
   125)    ! Local variables
   126) 
   127)    integer :: ncol
   128)    integer :: lchnk
   129)    integer :: k, i    ! lev and daycolumn indices
   130)    integer :: iswband ! sw band indices
   131) 
   132)    ! optical props for each aerosol
   133)    ! hygroscopic
   134)    real(r8), pointer :: h_ext(:,:)
   135)    real(r8), pointer :: h_ssa(:,:)
   136)    real(r8), pointer :: h_asm(:,:)
   137)    ! non-hygroscopic
   138)    real(r8), pointer :: n_ext(:)
   139)    real(r8), pointer :: n_ssa(:)
   140)    real(r8), pointer :: n_asm(:)
   141)    real(r8), pointer :: n_scat(:)
   142)    real(r8), pointer :: n_ascat(:)
   143)    ! radius-dependent
   144)    real(r8), pointer :: r_ext(:,:)    ! radius-dependent mass-specific extinction
   145)    real(r8), pointer :: r_scat(:,:)
   146)    real(r8), pointer :: r_ascat(:,:)
   147)    real(r8), pointer :: r_mu(:)       ! log(radius) domain variable for r_ext, r_scat, r_ascat
   148) 
   149)    ! radiative properties for each aerosol
   150)    real(r8) :: ta (pcols,pver,nswbands)
   151)    real(r8) :: tw (pcols,pver,nswbands)
   152)    real(r8) :: twf(pcols,pver,nswbands)
   153)    real(r8) :: twg(pcols,pver,nswbands)
   154) 
   155)    ! aerosol masses
   156)    real(r8), pointer :: aermmr(:,:)    ! mass mixing ratio of aerosols
   157)    real(r8) :: mmr_to_mass(pcols,pver) ! conversion factor for mmr to mass
   158)    real(r8) :: aermass(pcols,pver)     ! mass of aerosols
   159) 
   160)    ! for table lookup into rh grid
   161)    real(r8) :: es(pcols,pver)     ! saturation vapor pressure
   162)    real(r8) :: qs(pcols,pver)     ! saturation specific humidity
   163)    real(r8) :: rh(pcols,pver)
   164)    real(r8) :: rhtrunc(pcols,pver)
   165)    real(r8) :: wrh(pcols,pver)
   166)    integer  :: krh(pcols,pver)
   167)  
   168)    integer  :: numaerosols     ! number of bulk aerosols in climate/diagnostic list
   169)    integer  :: nmodes          ! number of aerosol modes in climate/diagnostic list
   170)    integer  :: iaerosol        ! index into bulk aerosol list
   171) 
   172)    character(len=ot_length) :: opticstype       ! hygro or nonhygro
   173)    !-----------------------------------------------------------------------------
   174) 
   175)    ncol  = state%ncol
   176)    lchnk = state%lchnk
   177) 
   178)    ! compute mixing ratio to mass conversion
   179)    do k = 1, pver
   180)       mmr_to_mass(:ncol,k) = rga * state%pdeldry(:ncol,k)
   181)    enddo
   182) 
   183)    ! initialize to conditions that would cause failure
   184)    tau     (:,:,:) = -100._r8
   185)    tau_w   (:,:,:) = -100._r8
   186)    tau_w_g (:,:,:) = -100._r8
   187)    tau_w_f (:,:,:) = -100._r8
   188) 
   189)    ! top layer (ilev = 0) has no aerosol (ie tau = 0)
   190)    ! also initialize rest of layers to accumulate od's
   191)    tau    (1:ncol,:,:) = 0._r8
   192)    tau_w  (1:ncol,:,:) = 0._r8
   193)    tau_w_g(1:ncol,:,:) = 0._r8
   194)    tau_w_f(1:ncol,:,:) = 0._r8
   195) 
   196)    ! calculate relative humidity for table lookup into rh grid
   197)    call qsat(state%t(1:ncol,1:pver), state%pmid(1:ncol,1:pver), &
   198)         es(1:ncol,1:pver), qs(1:ncol,1:pver))
   199)    rh(1:ncol,1:pver) = state%q(1:ncol,1:pver,1) / qs(1:ncol,1:pver)
   200) 
   201)    rhtrunc(1:ncol,1:pver) = min(rh(1:ncol,1:pver),1._r8)
   202)    krh(1:ncol,1:pver) = min(floor( rhtrunc(1:ncol,1:pver) * nrh ) + 1, nrh - 1) ! index into rh mesh
   203)    wrh(1:ncol,1:pver) = rhtrunc(1:ncol,1:pver) * nrh - krh(1:ncol,1:pver)       ! (-) weighting on left side values
   204) 
   205)    ! get number of bulk aerosols and number of modes in current list
   206)    call rad_cnst_get_info(list_idx, naero=numaerosols, nmodes=nmodes)
   207) 
   208)    ! Contributions from modal aerosols.
   209)    if (nmodes > 0) then
   210)       call modal_aero_sw(list_idx, state, pbuf, nnite, idxnite, &
   211)                          tau, tau_w, tau_w_g, tau_w_f)
   212)    else
   213)       tau    (1:ncol,:,:) = 0._r8
   214)       tau_w  (1:ncol,:,:) = 0._r8
   215)       tau_w_g(1:ncol,:,:) = 0._r8
   216)       tau_w_f(1:ncol,:,:) = 0._r8
   217)    end if
   218) 
   219)    ! Contributions from bulk aerosols.
   220)    do iaerosol = 1, numaerosols
   221) 
   222)       ! get bulk aerosol mass mixing ratio
   223)       call rad_cnst_get_aer_mmr(list_idx, iaerosol, state, pbuf, aermmr)
   224)       aermass(1:ncol,1:top_lev-1) = 0._r8
   225)       aermass(1:ncol,top_lev:pver) = aermmr(1:ncol,top_lev:pver) * mmr_to_mass(1:ncol,top_lev:pver)
   226) 
   227)       ! get optics type
   228)       call rad_cnst_get_aer_props(list_idx, iaerosol, opticstype=opticstype)
   229) 
   230)       select case (trim(opticstype))
   231)       case('hygro','hygroscopic','hygroscopi')
   232)          ! get optical properties for hygroscopic aerosols
   233)          call rad_cnst_get_aer_props(list_idx, iaerosol, sw_hygro_ext=h_ext, sw_hygro_ssa=h_ssa, sw_hygro_asm=h_asm)
   234)          call get_hygro_rad_props(ncol, krh, wrh, aermass, h_ext, h_ssa, h_asm, ta, tw, twg, twf)
   235)          tau    (1:ncol,1:pver,:) = tau    (1:ncol,1:pver,:) + ta (1:ncol,:,:)
   236)          tau_w  (1:ncol,1:pver,:) = tau_w  (1:ncol,1:pver,:) + tw (1:ncol,:,:)
   237)          tau_w_g(1:ncol,1:pver,:) = tau_w_g(1:ncol,1:pver,:) + twg(1:ncol,:,:)
   238)          tau_w_f(1:ncol,1:pver,:) = tau_w_f(1:ncol,1:pver,:) + twf(1:ncol,:,:)
   239) 
   240)       case('nonhygro','insoluble ')
   241)          ! get optical properties for non-hygroscopic aerosols
   242)          call rad_cnst_get_aer_props(list_idx, iaerosol, sw_nonhygro_ext=n_ext, sw_nonhygro_ssa=n_ssa, sw_nonhygro_asm=n_asm)
   243) 
   244)          call get_nonhygro_rad_props(ncol, aermass, n_ext, n_ssa, n_asm, ta, tw, twg, twf)
   245)          tau    (1:ncol,1:pver,:) = tau    (1:ncol,1:pver,:) + ta (1:ncol,:,:)
   246)          tau_w  (1:ncol,1:pver,:) = tau_w  (1:ncol,1:pver,:) + tw (1:ncol,:,:)
   247)          tau_w_g(1:ncol,1:pver,:) = tau_w_g(1:ncol,1:pver,:) + twg(1:ncol,:,:)
   248)          tau_w_f(1:ncol,1:pver,:) = tau_w_f(1:ncol,1:pver,:) + twf(1:ncol,:,:)
   249) 
   250)       case('volcanic')
   251)          ! get optical properties for volcanic aerosols
   252)          call rad_cnst_get_aer_props(list_idx, iaerosol, sw_nonhygro_ext=n_ext, sw_nonhygro_scat=n_scat, sw_nonhygro_ascat=n_ascat)
   253) 
   254)          call get_volcanic_rad_props(ncol, aermass, n_ext, n_scat, n_ascat, ta, tw, twg, twf)
   255)          tau    (1:ncol,1:pver,:) = tau    (1:ncol,1:pver,:) + ta (1:ncol,:,:)
   256)          tau_w  (1:ncol,1:pver,:) = tau_w  (1:ncol,1:pver,:) + tw (1:ncol,:,:)
   257)          tau_w_g(1:ncol,1:pver,:) = tau_w_g(1:ncol,1:pver,:) + twg(1:ncol,:,:)
   258)          tau_w_f(1:ncol,1:pver,:) = tau_w_f(1:ncol,1:pver,:) + twf(1:ncol,:,:)
   259) 
   260)       case('volcanic_radius')
   261)          ! get optical properties for volcanic aerosols
   262)          call rad_cnst_get_aer_props(list_idx, iaerosol, r_sw_ext=r_ext, r_sw_scat=r_scat, r_sw_ascat=r_ascat, mu=r_mu)
   263)          call get_volcanic_radius_rad_props(lchnk, ncol, aermass, pbuf, r_ext, r_scat, r_ascat, r_mu, ta, tw, twg, twf)
   264)          tau    (1:ncol,1:pver,:) = tau    (1:ncol,1:pver,:) + ta (1:ncol,:,:)
   265)          tau_w  (1:ncol,1:pver,:) = tau_w  (1:ncol,1:pver,:) + tw (1:ncol,:,:)
   266)          tau_w_g(1:ncol,1:pver,:) = tau_w_g(1:ncol,1:pver,:) + twg(1:ncol,:,:)
   267)          tau_w_f(1:ncol,1:pver,:) = tau_w_f(1:ncol,1:pver,:) + twf(1:ncol,:,:)
   268) 
   269)       case('zero')
   270)          ! no effect of "zero" aerosols, so update nothing
   271)       case default
   272)          call endrun('aer_rad_props_sw: unsupported opticstype :'//trim(opticstype)//':')
   273)       end select
   274) 
   275)       ! diagnostic output of individual aerosol optical properties
   276)       ! currently implemented for climate list only
   277)       call aer_vis_diag_out(lchnk, ncol, nnite, idxnite, iaerosol, ta(:,:,idx_sw_diag), list_idx)
   278) 
   279)    enddo
   280) 
   281)    ! diagnostic output of total aerosol optical properties
   282)    ! currently implemented for climate list only
   283)    call aer_vis_diag_out(lchnk, ncol, nnite, idxnite, 0, tau(:,:,idx_sw_diag), list_idx)
   284) 
   285) end subroutine aer_rad_props_sw

   289) subroutine aer_rad_props_lw(list_idx, state, pbuf,  odap_aer)
   290) 
   291)    use radconstants,  only: ot_length
   292) 
   293)    use physics_buffer, only : pbuf_get_field, pbuf_get_index, physics_buffer_desc
   294)    ! Purpose: Compute aerosol transmissions needed in absorptivity/
   295)    !    emissivity calculations
   296) 
   297)    ! lw extinction is the same representation for all 
   298)    ! species.  If this changes, this routine will need to do something
   299)    ! similar to the sw with routines like get_hygro_lw_abs
   300) 
   301)    ! Arguments
   302)    integer,             intent(in)  :: list_idx                      ! index of the climate or a diagnostic list
   303)    type(physics_state), intent(in), target :: state
   304)    
   305)    type(physics_buffer_desc), pointer :: pbuf(:)
   306)    real(r8),            intent(out) :: odap_aer(pcols,pver,nlwbands) ! [fraction] absorption optical depth, per layer
   307) 
   308)    ! Local variables
   309) 
   310)    integer :: bnd_idx     ! LW band index
   311)    integer :: i           ! column index
   312)    integer :: k           ! lev index
   313)    integer :: ncol        ! number of columns
   314)    integer :: lchnk       ! chunk index
   315)    integer :: numaerosols ! number of bulk aerosols in climate/diagnostic list
   316)    integer :: nmodes      ! number of aerosol modes in climate/diagnostic list
   317)    integer :: iaerosol    ! index into bulk aerosol list
   318)    character(len=ot_length) :: opticstype       ! hygro or nonhygro
   319) 
   320)    ! optical props for each aerosol
   321)    real(r8), pointer :: lw_abs(:)
   322)    real(r8), pointer :: lw_hygro_abs(:,:)
   323)    real(r8), pointer :: geometric_radius(:,:)
   324)  
   325)    ! volcanic lookup table
   326)    real(r8), pointer :: r_lw_abs(:,:)  ! radius dependent mass-specific absorption coefficient
   327)    real(r8), pointer :: r_mu(:)        ! log(geometric_mean_radius) domain samples of r_lw_abs(:,:)
   328)    integer  :: idx                     ! index to pbuf for geometric radius
   329)    real(r8) :: mu(pcols,pver)          ! log(geometric_radius)
   330)    real(r8) :: r_mu_min, r_mu_max, wmu, mutrunc
   331)    integer  :: nmu, kmu
   332) 
   333)    ! for table lookup into rh grid
   334)    real(r8) :: es(pcols,pver)     ! saturation vapor pressure
   335)    real(r8) :: qs(pcols,pver)     ! saturation specific humidity
   336)    real(r8) :: rh(pcols,pver)
   337)    real(r8) :: rhtrunc(pcols,pver)
   338)    real(r8) :: wrh(pcols,pver)
   339)    integer  :: krh(pcols,pver)
   340) 
   341)    ! aerosol (vertical) mass path and extinction
   342)    ! aerosol masses
   343)    real(r8), pointer :: aermmr(:,:)    ! mass mixing ratio of aerosols
   344)    real(r8) :: mmr_to_mass(pcols,pver) ! conversion factor for mmr to mass
   345)    real(r8) :: aermass(pcols,pver)     ! mass of aerosols
   346)    !-----------------------------------------------------------------------------
   347) 
   348)    ncol = state%ncol
   349)    lchnk = state%lchnk
   350) 
   351)    ! get number of bulk aerosols and number of modes in current list
   352)    call rad_cnst_get_info(list_idx, naero=numaerosols, nmodes=nmodes)
   353) 
   354)    ! Contributions from modal aerosols.
   355)    if (nmodes > 0) then
   356)       call modal_aero_lw(list_idx, state, pbuf, odap_aer)
   357)    else
   358)       odap_aer = 0._r8
   359)    end if
   360) 
   361)    ! Contributions from bulk aerosols.
   362)    if (numaerosols > 0) then
   363) 
   364)       ! compute mixing ratio to mass conversion
   365)       do k = 1, pver
   366)          mmr_to_mass(:ncol,k) = rga * state%pdeldry(:ncol,k)
   367)       end do
   368) 
   369)       ! calculate relative humidity for table lookup into rh grid
   370)       call qsat(state%t(1:ncol,1:pver), state%pmid(1:ncol,1:pver), &
   371)            es(1:ncol,1:pver), qs(1:ncol,1:pver))
   372)       rh(1:ncol,1:pver) = state%q(1:ncol,1:pver,1) / qs(1:ncol,1:pver)
   373) 
   374)       rhtrunc(1:ncol,1:pver) = min(rh(1:ncol,1:pver),1._r8)
   375)       krh(1:ncol,1:pver) = min(floor( rhtrunc(1:ncol,1:pver) * nrh ) + 1, nrh - 1) ! index into rh mesh
   376)       wrh(1:ncol,1:pver) = rhtrunc(1:ncol,1:pver) * nrh - krh(1:ncol,1:pver)       ! (-) weighting on left side values
   377) 
   378)    end if
   379) 
   380)    ! Loop over bulk aerosols in list.
   381)    do iaerosol = 1, numaerosols
   382) 
   383)       ! get aerosol mass mixing ratio
   384)       call rad_cnst_get_aer_mmr(list_idx, iaerosol, state, pbuf,  aermmr)
   385)       aermass(1:ncol,1:top_lev-1) = 0._r8
   386)       aermass(1:ncol,top_lev:pver) = aermmr(1:ncol,top_lev:pver) * mmr_to_mass(1:ncol,top_lev:pver)
   387) 
   388)       ! get optics type
   389)       call rad_cnst_get_aer_props(list_idx, iaerosol, opticstype=opticstype)
   390)       select case (trim(opticstype))
   391)       case('hygroscopic')
   392)           ! get optical properties for hygroscopic aerosols
   393)          call rad_cnst_get_aer_props(list_idx, iaerosol, lw_hygro_ext=lw_hygro_abs)
   394)          do bnd_idx = 1, nlwbands
   395)             do k = 1, pver
   396)                do i = 1, ncol
   397)                   odap_aer(i, k, bnd_idx) = odap_aer(i, k, bnd_idx) + &
   398)                        aermass(i, k) * &
   399)                        ((1 + wrh(i,k)) * lw_hygro_abs(krh(i,k)+1,bnd_idx) &
   400)                        - wrh(i,k)  * lw_hygro_abs(krh(i,k),  bnd_idx))
   401)                end do
   402)             end do
   403)          end do
   404)       case('insoluble','nonhygro','hygro','volcanic')
   405)           ! get optical properties for hygroscopic aerosols
   406)          call rad_cnst_get_aer_props(list_idx, iaerosol, lw_ext=lw_abs)
   407)          do bnd_idx = 1, nlwbands
   408)             do k = 1, pver          
   409)                do i = 1, ncol
   410)                   odap_aer(i,k,bnd_idx) = odap_aer(i,k,bnd_idx) + lw_abs(bnd_idx)*aermass(i,k)
   411)                end do
   412)             end do
   413)          end do
   414)          
   415)       case('volcanic_radius')
   416)           ! get optical properties for hygroscopic aerosols
   417)          call rad_cnst_get_aer_props(list_idx, iaerosol, r_lw_abs=r_lw_abs, mu=r_mu)
   418)          ! get microphysical properties for volcanic aerosols
   419)          idx = pbuf_get_index('VOLC_RAD_GEOM')
   420)          call pbuf_get_field(pbuf, idx, geometric_radius )
   421)          
   422)          ! interpolate in radius
   423)          ! caution: clip the table with no warning when outside bounds
   424)          nmu = size(r_mu)
   425)          r_mu_max = r_mu(nmu)
   426)          r_mu_min = r_mu(1)
   427)          do i = 1, ncol
   428)             do k = 1, pver
   429)                if(geometric_radius(i,k) > 0._r8) then
   430)                   mu(i,k) = log(geometric_radius(i,k))
   431)                else
   432)                   mu(i,k) = 0._r8
   433)                endif
   434)                mutrunc = max(min(mu(i,k),r_mu_max),r_mu_min)
   435)                kmu = max(min(1 + (mutrunc-r_mu_min)/(r_mu_max-r_mu_min)*(nmu-1),nmu-1._r8),1._r8)
   436)                wmu = max(min( (mutrunc -r_mu(kmu)) / (r_mu(kmu+1) - r_mu(kmu)) ,1._r8),0._r8)
   437)                do bnd_idx = 1, nlwbands
   438)                   odap_aer(i,k,bnd_idx) = odap_aer(i,k,bnd_idx) + &
   439)                      aermass(i,k) * &
   440)                      ((1._r8 - wmu) * r_lw_abs(bnd_idx, kmu  ) + &
   441)                      (wmu) * r_lw_abs(bnd_idx, kmu+1))
   442)                end do
   443)             end do
   444)          end do
   445) 
   446)        case('zero')
   447)           ! zero aerosols types have no optical effect, so do nothing.
   448)        case default
   449)           call endrun('aer_rad_props_lw: unsupported opticstype: '//trim(opticstype))
