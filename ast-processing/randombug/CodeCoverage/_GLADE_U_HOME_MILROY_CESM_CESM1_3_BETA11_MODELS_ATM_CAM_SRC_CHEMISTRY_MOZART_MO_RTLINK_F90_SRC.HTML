   199)           dssal4 = dtsal(k,wn,4)*wrk
   200)           dasal4 = dtsal(k,wn,4)*(1._r8 - wrk)
   201)           !-----------------------------------------------------------------------

   203)           !-----------------------------------------------------------------------
   204)           dtsct = dtsct + dssal1 + dssal2 + dssal3 + dssal4
   205)           dtabs = dtabs + dasal1 + dasal2 + dasal3 + dasal4
