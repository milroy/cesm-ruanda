   476)           albsfc_lcl(nir_bnd_bgn:nir_bnd_end) = albsfc(c_idx,2)
   477) 
   478) 
   479)           ! Error check for snow grain size:
   480)           do i=snl_top,snl_btm,1
   481)              if ((snw_rds_lcl(i) < snw_rds_min_tbl) .or. (snw_rds_lcl(i) > snw_rds_max_tbl)) then
   482)                 write (iulog,*) "SNICAR ERROR: snow grain radius of ", snw_rds_lcl(i), " out of bounds."
   483)                 write (iulog,*) "NSTEP= ", nstep
   484)                 write (iulog,*) "flg_snw_ice= ", flg_snw_ice
   485)                 write (iulog,*) "column: ", c_idx, " level: ", i, " snl(c)= ", snl_lcl
   486)                 write (iulog,*) "lat= ", lat_coord, " lon= ", lon_coord
   487)                 write (iulog,*) "h2osno(c)= ", h2osno_lcl
   488)                 call endrun()
   489)              endif
   490)           enddo
   491) 
   492)           ! Incident flux weighting parameters
   493)           !  - sum of all VIS bands must equal 1
   494)           !  - sum of all NIR bands must equal 1
   495)           !
   496)           ! Spectral bands (5-band case)
   497)           !  Band 1: 0.3-0.7um (VIS)
   498)           !  Band 2: 0.7-1.0um (NIR)
   499)           !  Band 3: 1.0-1.2um (NIR)
   500)           !  Band 4: 1.2-1.5um (NIR)
   501)           !  Band 5: 1.5-5.0um (NIR)
   502)           !
   503)           ! The following weights are appropriate for surface-incident flux in a mid-latitude winter atmosphere
   504)           !
   505)           ! 3-band weights
   506)           if (numrad_snw==3) then
   507)              ! Direct:
   508)              if (flg_slr_in == 1) then
   509)                 flx_wgt(1) = 1._r8
   510)                 flx_wgt(2) = 0.66628670195247_r8
   511)                 flx_wgt(3) = 0.33371329804753_r8
   512)              ! Diffuse:
   513)              elseif (flg_slr_in == 2) then
   514)                 flx_wgt(1) = 1._r8
   515)                 flx_wgt(2) = 0.77887652162877_r8
   516)                 flx_wgt(3) = 0.22112347837123_r8
   517)              endif
   518) 
   519)           ! 5-band weights
   520)           elseif(numrad_snw==5) then
   521)              ! Direct:
   522)              if (flg_slr_in == 1) then
   523)                 flx_wgt(1) = 1._r8
   524)                 flx_wgt(2) = 0.49352158521175_r8
   525)                 flx_wgt(3) = 0.18099494230665_r8
   526)                 flx_wgt(4) = 0.12094898498813_r8
   527)                 flx_wgt(5) = 0.20453448749347_r8
   528)              ! Diffuse:
   529)              elseif (flg_slr_in == 2) then
   530)                 flx_wgt(1) = 1._r8
   531)                 flx_wgt(2) = 0.58581507618433_r8
   532)                 flx_wgt(3) = 0.20156903770812_r8
   533)                 flx_wgt(4) = 0.10917889346386_r8
   534)                 flx_wgt(5) = 0.10343699264369_r8
   535)              endif
   536)           endif
   537) 
   538)           ! Loop over snow spectral bands
   539)           do bnd_idx = 1,numrad_snw
   540) 
   541)              mu_not    = coszen(c_idx)  ! must set here, because of error handling
   542)              flg_dover = 1              ! default is to redo
   543)              err_idx   = 0              ! number of times through loop
   544) 
   545)              do while (flg_dover > 0)
   546) 
   547)                 ! DEFAULT APPROXIMATIONS:
   548)                 !  VIS:       Delta-Eddington
   549)                 !  NIR (all): Delta-Hemispheric Mean
   550)                 !  WARNING:   DO NOT USE DELTA-EDDINGTON FOR NIR DIFFUSE - this sometimes results in negative albedo
   551)                 !  
   552)                 ! ERROR CONDITIONS:
   553)                 !  Conditions which cause "trip", resulting in redo of RT approximation:
   554)                 !   1. negative absorbed flux
   555)                 !   2. total absorbed flux greater than incident flux
   556)                 !   3. negative albedo
   557)                 !   NOTE: These errors have only been encountered in spectral bands 4 and 5
   558)                 !
   559)                 ! ERROR HANDLING
   560)                 !  1st error (flg_dover=2): switch approximation (Edd->HM or HM->Edd)
   561)                 !  2nd error (flg_dover=3): change zenith angle by 0.02 (this happens about 1 in 10^6 cases)
   562)                 !  3rd error (flg_dover=4): switch approximation with new zenith
   563)                 !  Subsequent errors: repeatedly change zenith and approximations...
   564)               
   565)                 if (bnd_idx == 1) then
   566)                    if (flg_dover == 2) then
   567)                       APRX_TYP = 3
   568)                    elseif (flg_dover == 3) then
   569)                       APRX_TYP = 1
   570)                       if (coszen(c_idx) > 0.5_r8) then
   571)                          mu_not = mu_not - 0.02_r8
   572)                       else
   573)                          mu_not = mu_not + 0.02_r8
   574)                       endif
   575)                    elseif (flg_dover == 4) then
   576)                       APRX_TYP = 3
   577)                    else
   578)                       APRX_TYP = 1
   579)                    endif
   580)                    
   581)                 else
   582)                    if (flg_dover == 2) then
   583)                       APRX_TYP = 1
   584)                    elseif (flg_dover == 3) then
   585)                       APRX_TYP = 3
   586)                       if (coszen(c_idx) > 0.5_r8) then
   587)                          mu_not = mu_not - 0.02_r8
   588)                       else
   589)                          mu_not = mu_not + 0.02_r8
   590)                       endif
   591)                    elseif (flg_dover == 4) then
   592)                       APRX_TYP = 1
   593)                    else
   594)                       APRX_TYP = 3
   595)                    endif
   596) 
   597)                 endif

   657)                 ext_cff_mss_aer_lcl(4)   = ext_cff_mss_oc2(bnd_idx)

   660)                 ss_alb_aer_lcl(5)        = ss_alb_dst1(bnd_idx)      
   661)                 asm_prm_aer_lcl(5)       = asm_prm_dst1(bnd_idx)

   664)                 ! aerosol species 6 optical properties

   680)                 ! 1. snow and aerosol layer column mass (L_snw, L_aer [kg/m^2])
   681)                 ! 2. optical Depths (tau_snw, tau_aer)
   682)                 ! 3. weighted Mie properties (tau, omega, g)
   683) 
   684)                 ! Weighted Mie parameters of each layer
   685)                 do i=snl_top,snl_btm,1
   686)                    L_snw(i)   = h2osno_ice_lcl(i)+h2osno_liq_lcl(i)

   689)                    do j=1,sno_nbr_aer
   690)                       L_aer(i,j)   = L_snw(i)*mss_cnc_aer_lcl(i,j)
   691)                       tau_aer(i,j) = L_aer(i,j)*ext_cff_mss_aer_lcl(j)
   692)                    enddo
   693) 
   694)                    tau_sum   = 0._r8
   695)                    omega_sum = 0._r8
   696)                    g_sum     = 0._r8
   697)   
   698)                    do j=1,sno_nbr_aer
   699)                       tau_sum    = tau_sum + tau_aer(i,j) 
   700)                       omega_sum  = omega_sum + (tau_aer(i,j)*ss_alb_aer_lcl(j))
   701)                       g_sum      = g_sum + (tau_aer(i,j)*ss_alb_aer_lcl(j)*asm_prm_aer_lcl(j))
   702)                    enddo
   703) 
   704)                    tau(i)    = tau_sum + tau_snw(i)
   705)                    omega(i)  = (1/tau(i))*(omega_sum+(ss_alb_snw_lcl(i)*tau_snw(i)))
   706)                    g(i)      = (1/(tau(i)*omega(i)))*(g_sum+ (asm_prm_snw_lcl(i)*ss_alb_snw_lcl(i)*tau_snw(i)))
   707)                 enddo
   708) 
   709)                 ! DELTA transformations, if requested
   710)                 if (DELTA == 1) then
   711)                    do i=snl_top,snl_btm,1
   712)                       g_star(i)     = g(i)/(1+g(i))
   713)                       omega_star(i) = ((1-(g(i)**2))*omega(i)) / (1-(omega(i)*(g(i)**2)))
   714)                       tau_star(i)   = (1-(omega(i)*(g(i)**2)))*tau(i)
   715)                    enddo
   716)                 else
   717)                    do i=snl_top,snl_btm,1
   718)                       g_star(i)     = g(i)
   719)                       omega_star(i) = omega(i)
   720)                       tau_star(i)   = tau(i)
   721)                    enddo
   722)                 endif
   723) 
   724)                 ! Total column optical depth:
   725)                 ! tau_clm(i) = total optical depth above the bottom of layer i
   726)                 tau_clm(snl_top) = 0._r8
   727)                 do i=snl_top+1,snl_btm,1
   728)                    tau_clm(i) = tau_clm(i-1)+tau_star(i-1)
   729)                 enddo
   730) 
   731)                 ! Direct radiation at bottom of snowpack:
   732)                 F_direct_btm = albsfc_lcl(bnd_idx)*mu_not*exp(-(tau_clm(snl_btm)+tau_star(snl_btm))/mu_not)*pi*flx_slrd_lcl(bnd_idx)
   733) 
   734)                 ! Intermediates
   735)                 ! Gamma values are approximation-specific.
   736) 
   737)                 ! Eddington
   738)                 if (APRX_TYP==1) then
   739)                    do i=snl_top,snl_btm,1
   740)                       gamma1(i) = (7-(omega_star(i)*(4+(3*g_star(i)))))/4
   741)                       gamma2(i) = -(1-(omega_star(i)*(4-(3*g_star(i)))))/4
   742)                       gamma3(i) = (2-(3*g_star(i)*mu_not))/4
   743)                       gamma4(i) = 1-gamma3(i)
   744)                       mu_one    = 0.5
   745)                    enddo
   746)                    
   747)                 ! Quadrature
   748)                 elseif (APRX_TYP==2) then
   749)                    do i=snl_top,snl_btm,1
   750)                       gamma1(i) = (3**0.5)*(2-(omega_star(i)*(1+g_star(i))))/2
   751)                       gamma2(i) = omega_star(i)*(3**0.5)*(1-g_star(i))/2
   752)                       gamma3(i) = (1-((3**0.5)*g_star(i)*mu_not))/2
   753)                       gamma4(i) = 1-gamma3(i)
   754)                       mu_one    = 1/(3**0.5)
   755)                    enddo
   756) 
   757)                 ! Hemispheric Mean
   758)                 elseif (APRX_TYP==3) then
   759)                    do i=snl_top,snl_btm,1
   760)                       gamma1(i) = 2 - (omega_star(i)*(1+g_star(i)))
   761)                       gamma2(i) = omega_star(i)*(1-g_star(i))
   762)                       gamma3(i) = (1-((3**0.5)*g_star(i)*mu_not))/2
   763)                       gamma4(i) = 1-gamma3(i)
   764)                       mu_one    = 0.5
   765)                    enddo
   766)                 endif
   767) 
   768)                 ! Intermediates for tri-diagonal solution
   769)                 do i=snl_top,snl_btm,1
   770)                    lambda(i) = sqrt(abs((gamma1(i)**2) - (gamma2(i)**2)))
   771)                    GAMMA(i)  = gamma2(i)/(gamma1(i)+lambda(i))
   772) 
   773)                    e1(i)     = 1+(GAMMA(i)*exp(-lambda(i)*tau_star(i)))
   774)                    e2(i)     = 1-(GAMMA(i)*exp(-lambda(i)*tau_star(i)))
   775)                    e3(i)     = GAMMA(i) + exp(-lambda(i)*tau_star(i))
   776)                    e4(i)     = GAMMA(i) - exp(-lambda(i)*tau_star(i))
   777)                 enddo !enddo over snow layers
   778) 
   779) 
   780)                 ! Intermediates for tri-diagonal solution
   781)                 do i=snl_top,snl_btm,1
   782)                    if (flg_slr_in == 1) then
   783) 
   784)                       C_pls_btm(i) = (omega_star(i)*pi*flx_slrd_lcl(bnd_idx)* &
   785)                                      exp(-(tau_clm(i)+tau_star(i))/mu_not)*   &
   786)                                      (((gamma1(i)-(1/mu_not))*gamma3(i))+     &
   787)                                      (gamma4(i)*gamma2(i))))/((lambda(i)**2)-(1/(mu_not**2)))
   788) 
   789)                       C_mns_btm(i) = (omega_star(i)*pi*flx_slrd_lcl(bnd_idx)* &
   790)                                      exp(-(tau_clm(i)+tau_star(i))/mu_not)*   &
   791)                                      (((gamma1(i)+(1/mu_not))*gamma4(i))+     &
   792)                                      (gamma2(i)*gamma3(i))))/((lambda(i)**2)-(1/(mu_not**2)))
   793) 
   794)                       C_pls_top(i) = (omega_star(i)*pi*flx_slrd_lcl(bnd_idx)* &
   795)                                      exp(-tau_clm(i)/mu_not)*(((gamma1(i)-(1/mu_not))* &
   796)                                      gamma3(i))+(gamma4(i)*gamma2(i))))/((lambda(i)**2)-(1/(mu_not**2)))
   797) 
   798)                       C_mns_top(i) = (omega_star(i)*pi*flx_slrd_lcl(bnd_idx)* &
   799)                                      exp(-tau_clm(i)/mu_not)*(((gamma1(i)+(1/mu_not))* &
   800)                                      gamma4(i))+(gamma2(i)*gamma3(i))))/((lambda(i)**2)-(1/(mu_not**2)))
   801) 
   802)                    else
   803)                       C_pls_btm(i) = 0._r8
   804)                       C_mns_btm(i) = 0._r8
   805)                       C_pls_top(i) = 0._r8
   806)                       C_mns_top(i) = 0._r8
   807)                    endif
   808)                 enddo
   809) 
   810)                 ! Coefficients for tridiaganol matrix solution
   811)                 do i=2*snl_lcl+1,0,1
   812) 
   813)                    !Boundary values for i=1 and i=2*snl_lcl, specifics for i=odd and i=even    
   814)                    if (i==(2*snl_lcl+1)) then
   815)                       A(i) = 0
   816)                       B(i) = e1(snl_top)
   817)                       D(i) = -e2(snl_top)
   818)                       E(i) = flx_slri_lcl(bnd_idx)-C_mns_top(snl_top)
   819) 
   820)                    elseif(i==0) then
   821)                       A(i) = e1(snl_btm)-(albsfc_lcl(bnd_idx)*e3(snl_btm))
   822)                       B(i) = e2(snl_btm)-(albsfc_lcl(bnd_idx)*e4(snl_btm))
   823)                       D(i) = 0
   824)                       E(i) = F_direct_btm-C_pls_btm(snl_btm)+(albsfc_lcl(bnd_idx)*C_mns_btm(snl_btm))
   825) 
   826)                    elseif(mod(i,2)==-1) then   ! If odd and i>=3 (n=1 for i=3)
   827)                       n=floor(i/2.0)
   828)                       A(i) = (e2(n)*e3(n))-(e4(n)*e1(n))
   829)                       B(i) = (e1(n)*e1(n+1))-(e3(n)*e3(n+1))
   830)                       D(i) = (e3(n)*e4(n+1))-(e1(n)*e2(n+1))
   831)                       E(i) = (e3(n)*(C_pls_top(n+1)-C_pls_btm(n)))+(e1(n)*(C_mns_btm(n)-C_mns_top(n+1)))
   832) 
   833)                    elseif(mod(i,2)==0) then    ! If even and i<=2*snl_lcl
   834)                       n=(i/2)
   835)                       A(i) = (e2(n+1)*e1(n))-(e3(n)*e4(n+1))
   836)                       B(i) = (e2(n)*e2(n+1))-(e4(n)*e4(n+1))
   837)                       D(i) = (e1(n+1)*e4(n+1))-(e2(n+1)*e3(n+1))
   838)                       E(i) = (e2(n+1)*(C_pls_top(n+1)-C_pls_btm(n)))+(e4(n+1)*(C_mns_top(n+1)-C_mns_btm(n))) 
   839)                    endif
   840)                 enddo
   841) 
   842)                 AS(0) = A(0)/B(0)
   843)                 DS(0) = E(0)/B(0)
   844) 
   845)                 do i=-1,(2*snl_lcl+1),-1
   846)                    X(i)  = 1/(B(i)-(D(i)*AS(i+1)))
   847)                    AS(i) = A(i)*X(i)
   848)                    DS(i) = (E(i)-(D(i)*DS(i+1)))*X(i)
   849)                 enddo
   850) 
   851)                 Y(2*snl_lcl+1) = DS(2*snl_lcl+1)
   852)                 do i=(2*snl_lcl+2),0,1
   853)                    Y(i) = DS(i)-(AS(i)*Y(i-1))
   854)                 enddo
   855) 
   856)                 ! Downward direct-beam and net flux (F_net) at the base of each layer:
   857)                 do i=snl_top,snl_btm,1
   858)                    F_direct(i) = mu_not*pi*flx_slrd_lcl(bnd_idx)*exp(-(tau_clm(i)+tau_star(i))/mu_not)
   859)                    F_net(i)    = (Y(2*i-1)*(e1(i)-e3(i))) + (Y(2*i)*(e2(i)-e4(i))) + &
   860)                                  C_pls_btm(i) - C_mns_btm(i) - F_direct(i)
   861)                 enddo
   862) 
   863)                 ! Upward flux at snowpack top:

   865)                             GAMMA(snl_top))) + (Y(2*snl_lcl+2)*(exp(-lambda(snl_top)* &
   866)                             tau_star(snl_top))-GAMMA(snl_top))) + C_pls_top(snl_top)
   867) 
   868)                 ! Net flux at bottom = absorbed radiation by underlying surface:
   869)                 F_btm_net = -F_net(snl_btm)
   870) 
   871) 
   872)                 ! Bulk column albedo and surface net flux
   873)                 albedo    = F_sfc_pls/((mu_not*pi*flx_slrd_lcl(bnd_idx))+flx_slri_lcl(bnd_idx))
   874)                 F_sfc_net = F_sfc_pls - ((mu_not*pi*flx_slrd_lcl(bnd_idx))+flx_slri_lcl(bnd_idx))
   875) 
   876)                 trip = 0
   877)                 ! Absorbed flux in each layer
   878)                 do i=snl_top,snl_btm,1
   879)                    if(i==snl_top) then
   880)                       F_abs(i) = F_net(i)-F_sfc_net
   881)                    else
   882)                       F_abs(i) = F_net(i)-F_net(i-1)
   883)                    endif
   884)                    flx_abs_lcl(i,bnd_idx) = F_abs(i)
   885)                    
   886) 
   887)                    ! ERROR check: negative absorption
   888)                    if (flx_abs_lcl(i,bnd_idx) < -0.00001) then
   889)                       trip = 1
   890)                    endif
   891)                 enddo
   892)                 
   893)                 flx_abs_lcl(1,bnd_idx) = F_btm_net
   894)              
   895)                 if (flg_nosnl == 1) then
   896)                    ! If there are no snow layers (but still snow), all absorbed energy must be in top soil layer
   897)                    !flx_abs_lcl(:,bnd_idx) = 0._r8
   898)                    !flx_abs_lcl(1,bnd_idx) = F_abs(0) + F_btm_net
   899) 
   900)                    ! changed on 20070408:
   901)                    ! OK to put absorbed energy in the fictitous snow layer because routine SurfaceRadiation
   902)                    ! handles the case of no snow layers. Then, if a snow layer is addded between now and
   903)                    ! SurfaceRadiation (called in Hydrology1), absorbed energy will be properly distributed.
   904)                    flx_abs_lcl(0,bnd_idx) = F_abs(0)
   905)                    flx_abs_lcl(1,bnd_idx) = F_btm_net
   906)                 endif
   907)                 
   908)                 !Underflow check (we've already tripped the error condition above)
   909)                 do i=snl_top,1,1
   910)                    if (flx_abs_lcl(i,bnd_idx) < 0._r8) then
   911)                       flx_abs_lcl(i,bnd_idx) = 0._r8
   912)                    endif
   913)                 enddo
   914) 
   915)                 F_abs_sum = 0._r8
   916)                 do i=snl_top,snl_btm,1
   917)                    F_abs_sum = F_abs_sum + F_abs(i)
   918)                 enddo
   919) 
   920) 
   921)                 !ERROR check: absorption greater than incident flux
   922)                 ! (should make condition more generic than "1._r8")
   923)                 if (F_abs_sum > 1._r8) then
   924)                    trip = 1
   925)                 endif
   926) 
   927)                 !ERROR check:
   928)                 if ((albedo < 0._r8).and.(trip==0)) then
   929)                    trip = 1
   930)                 endif
   931)                 
   932)                 ! Set conditions for redoing RT calculation 
   933)                 if ((trip == 1).and.(flg_dover == 1)) then
   934)                    flg_dover = 2
   935)                 elseif ((trip == 1).and.(flg_dover == 2)) then
   936)                    flg_dover = 3
   937)                 elseif ((trip == 1).and.(flg_dover == 3)) then
   938)                    flg_dover = 4
   939)                 elseif((trip == 1).and.(flg_dover == 4).and.(err_idx < 20)) then
   940)                    flg_dover = 3
   941)                    err_idx = err_idx + 1
   942)                 elseif((trip == 1).and.(flg_dover == 4).and.(err_idx >= 20)) then
   943)                    flg_dover = 0
   944)                    write(iulog,*) "SNICAR ERROR: FOUND A WORMHOLE. STUCK IN INFINITE LOOP! Called from: ", flg_snw_ice
   945)                    write(iulog,*) "SNICAR STATS: snw_rds(0)= ", snw_rds(c_idx,0)
   946)                    write(iulog,*) "SNICAR STATS: L_snw(0)= ", L_snw(0)
   947)                    write(iulog,*) "SNICAR STATS: h2osno= ", h2osno_lcl, " snl= ", snl_lcl
   948)                    write(iulog,*) "SNICAR STATS: soot1(0)= ", mss_cnc_aer_lcl(0,1)
   949)                    write(iulog,*) "SNICAR STATS: soot2(0)= ", mss_cnc_aer_lcl(0,2)
   950)                    write(iulog,*) "SNICAR STATS: dust1(0)= ", mss_cnc_aer_lcl(0,3)
   951)                    write(iulog,*) "SNICAR STATS: dust2(0)= ", mss_cnc_aer_lcl(0,4)
   952)                    write(iulog,*) "SNICAR STATS: dust3(0)= ", mss_cnc_aer_lcl(0,5)
   953)                    write(iulog,*) "SNICAR STATS: dust4(0)= ", mss_cnc_aer_lcl(0,6)

   957)                    flg_dover = 0
   958)                 endif
   959) 
   960)              enddo !enddo while (flg_dover > 0)
   961)              
   962)              ! Energy conservation check:
   963)              ! Incident direct+diffuse radiation equals (absorbed+bulk_transmitted+bulk_reflected)
   964)              energy_sum = (mu_not*pi*flx_slrd_lcl(bnd_idx)) + flx_slri_lcl(bnd_idx) - (F_abs_sum + F_btm_net + F_sfc_pls)
   965)              if (abs(energy_sum) > 0.00001_r8) then
   966)                 write (iulog,"(a,e12.6,a,i6,a,i6)") "SNICAR ERROR: Energy conservation error of : ", energy_sum, &
   967)                              " at timestep: ", nstep, " at column: ", c_idx
   968)                 call endrun()
   969)              endif
   970) 
   971)              albout_lcl(bnd_idx) = albedo
   972) 
   973) 
   974)              ! Check that albedo is less than 1
   975)              if (albout_lcl(bnd_idx) > 1.0) then
   976) 
   977)                 write (iulog,*) "SNICAR ERROR: Albedo > 1.0 at c: ", c_idx, " NSTEP= ",nstep
   978)                 write (iulog,*) "SNICAR STATS: bnd_idx= ",bnd_idx
   979)                 write (iulog,*) "SNICAR STATS: albout_lcl(bnd)= ",albout_lcl(bnd_idx), " albsfc_lcl(bnd_idx)= ",albsfc_lcl(bnd_idx)
   980)                 write (iulog,*) "SNICAR STATS: landtype= ", sfctype
   981)                 write (iulog,*) "SNICAR STATS: h2osno= ", h2osno_lcl, " snl= ", snl_lcl
   982)                 write (iulog,*) "SNICAR STATS: coszen= ", coszen(c_idx), " flg_slr= ", flg_slr_in
   983) 
   984)                 write (iulog,*) "SNICAR STATS: soot(-4)= ", mss_cnc_aer_lcl(-4,1)
   985)                 write (iulog,*) "SNICAR STATS: soot(-3)= ", mss_cnc_aer_lcl(-3,1)
   986)                 write (iulog,*) "SNICAR STATS: soot(-2)= ", mss_cnc_aer_lcl(-2,1)
   987)                 write (iulog,*) "SNICAR STATS: soot(-1)= ", mss_cnc_aer_lcl(-1,1)

  1000)                 write (iulog,*) "SNICAR STATS: snw_rds(0)= ", snw_rds(c_idx,0)
  1001)                 
  1002)                 call endrun()
  1003)              endif
  1004)              
  1005)           enddo   ! loop over wvl bands
  1006) 
  1007) 
  1008)           ! Weight output NIR albedo appropriately
  1009)           albout(c_idx,1) = albout_lcl(1)
  1010)           flx_sum         = 0._r8
  1011)           do bnd_idx= nir_bnd_bgn,nir_bnd_end
  1012)              flx_sum = flx_sum + flx_wgt(bnd_idx)*albout_lcl(bnd_idx)
  1013)           end do
  1014)           albout(c_idx,2) = flx_sum / sum(flx_wgt(nir_bnd_bgn:nir_bnd_end))
  1015) 
  1016)           ! Weight output NIR absorbed layer fluxes (flx_abs) appropriately
  1017)           flx_abs(c_idx,:,1) = flx_abs_lcl(:,1)

  1019)              flx_sum = 0._r8
  1020)              do bnd_idx= nir_bnd_bgn,nir_bnd_end
  1021)                 flx_sum = flx_sum + flx_wgt(bnd_idx)*flx_abs_lcl(i,bnd_idx)

  1023)              flx_abs(c_idx,i,2) = flx_sum / sum(flx_wgt(nir_bnd_bgn:nir_bnd_end))          
  1024)           end do
  1025) 
  1026)        ! If snow < minimum_snow, but > 0, and there is sun, set albedo to underlying surface albedo
  1027)        elseif ( (coszen(c_idx) > 0._r8) .and. (h2osno_lcl < min_snw) .and. (h2osno_lcl > 0._r8) ) then
  1028)           albout(c_idx,1) = albsfc(c_idx,1)
  1029)           albout(c_idx,2) = albsfc(c_idx,2)
  1030)

  1032)        else
  1033)           albout(c_idx,1) = 0._r8
  1034)           albout(c_idx,2) = 0._r8

  1049)   subroutine SnowAge_grain(lbc, ubc, num_snowc, filter_snowc, num_nosnowc, filter_nosnowc)
  1050)     !
  1051)     ! !DESCRIPTION:
  1052)     ! Updates the snow effective grain size (radius). 
  1053)     ! Contributions to grain size evolution are from:
  1054)     !   1. vapor redistribution (dry snow) 
  1055)     !   2. liquid water redistribution (wet snow)
  1056)     !   3. re-freezing of liquid water
  1057)     ! 
  1058)     ! Vapor redistribution: Method is to retrieve 3 best-bit parameters that
  1059)     ! depend on snow temperature, temperature gradient, and density,
  1060)     ! that are derived from the microphysical model described in: 
  1061)     ! Flanner and Zender (2006), Linking snowpack microphysics and albedo
  1062)     ! evolution, J. Geophys. Res., 111, D12208, doi:10.1029/2005JD006834. 
  1063)     ! The parametric equation has the form: 
  1064)     ! dr/dt = drdt_0*(tau/(dr_fresh+tau))^(1/kappa), where:
  1065)     !   r is the effective radius,
  1066)     !   tau and kappa are best-fit parameters,
  1067)     !   drdt_0 is the initial rate of change of effective radius, and
  1068)     !   dr_fresh is the difference between the current and fresh snow states 
  1069)     !  (r_current - r_fresh).
  1070)     !
  1071)     ! Liquid water redistribution: Apply the grain growth function from:
  1072)     !   Brun, E. (1989), Investigation of wet-snow metamorphism in respect of 
  1073)     !   liquid-water content, Annals of Glaciology, 13, 22-26.
  1074)     !   There are two parameters that describe the grain growth rate as 
  1075)     !   a function of snow liquid water content (LWC). The "LWC=0" parameter
  1076)     !   is zeroed here because we are accounting for dry snowing with a 
  1077)     !   different representation
  1078)     !
  1079)     ! Re-freezing of liquid water: Assume that re-frozen liquid water clumps
  1080)     !   into an arbitrarily large effective grain size (snw_rds_refrz). 
  1081)     !   The phenomenon is observed (Grenfell), but so far unquantified, as far as 
  1082)     !   I am aware.
  1083)     !
  1084)     !
  1085)     ! !USES:
  1086)     use clmtype
  1087)     use clm_time_manager , only : get_step_size, get_nstep
  1088)     use clm_varpar       , only : nlevsno
  1089)     use clm_varcon       , only : spval
  1090)     use abortutils       , only : endrun
  1091)     use shr_const_mod    , only : SHR_CONST_RHOICE, SHR_CONST_PI
  1092)     !
  1093)     ! !ARGUMENTS:
  1094)     implicit none
  1095)     integer, intent(in) :: lbc, ubc                  ! column bounds
  1096)     integer, intent(in) :: num_snowc                 ! number of column snow points in column filter
  1097)     integer, intent(in) :: filter_snowc(ubc-lbc+1)   ! column filter for snow points
  1098)     integer, intent(in) :: num_nosnowc               ! number of column non-snow points in column filter
  1099)     integer, intent(in) :: filter_nosnowc(ubc-lbc+1) ! column filter for non-snow points
  1100)     !
  1101)     !
  1102)     ! !CALLED FROM: clm_driver1
  1103)     !
  1104) 
  1105)     ! !LOCAL VARIABLES:
  1106)     !
  1107)     ! local pointers to implicit arguments
  1108)     !
  1109) 
  1110)     real(r8), pointer :: t_soisno(:,:)         ! soil and snow temperature (col,lyr) [K]
  1111)     integer,  pointer :: snl(:)                ! negative number of snow layers (col) [nbr]
  1112)     real(r8), pointer :: t_grnd(:)             ! ground temperature (col) [K]
  1113)     real(r8), pointer :: dz(:,:)               ! layer thickness (col,lyr) [m]
  1114)     real(r8), pointer :: h2osno(:)             ! snow water (col) [mm H2O]
  1115)     real(r8), pointer :: snw_rds(:,:)          ! effective grain radius (col,lyr) [microns, m-6]
  1116)     real(r8), pointer :: snw_rds_top(:)        ! effective grain radius, top layer (col) [microns, m-6]
  1117)     real(r8), pointer :: sno_liq_top(:)        ! liquid water fraction (mass) in top snow layer (col) [frc]
  1118)     real(r8), pointer :: h2osoi_liq(:,:)       ! liquid water content (col,lyr) [kg m-2]
  1119)     real(r8), pointer :: h2osoi_ice(:,:)       ! ice content (col,lyr) [kg m-2]
  1120)     real(r8), pointer :: snot_top(:)           ! snow temperature in top layer (col) [K]
  1121)     real(r8), pointer :: dTdz_top(:)           ! temperature gradient in top layer (col) [K m-1]
  1122)     real(r8), pointer :: qflx_snow_grnd_col(:) ! snow on ground after interception (col) [kg m-2 s-1]
  1123)     real(r8), pointer :: qflx_snwcp_ice(:)     ! excess precipitation due to snow capping [kg m-2 s-1]
  1124)     real(r8), pointer :: qflx_snofrz_lyr(:,:)  ! snow freezing rate (col,lyr) [kg m-2 s-1]
  1125)     logical , pointer :: do_capsnow(:)         ! true => do snow capping
  1126)  
  1127)     !
  1128)     ! !OTHER LOCAL VARIABLES:
  1129)     !
  1130)     integer :: snl_top                      ! top snow layer index [idx]
  1131)     integer :: snl_btm                      ! bottom snow layer index [idx]
  1132)     integer :: i                            ! layer index [idx]
  1133)     integer :: c_idx                        ! column index [idx]
  1134)     integer :: fc                           ! snow column filter index [idx]
  1135)     integer :: T_idx                        ! snow aging lookup table temperature index [idx]
  1136)     integer :: Tgrd_idx                     ! snow aging lookup table temperature gradient index [idx]
  1137)     integer :: rhos_idx                     ! snow aging lookup table snow density index [idx]
  1138)     real(r8) :: t_snotop                    ! temperature at upper layer boundary [K]
  1139)     real(r8) :: t_snobtm                    ! temperature at lower layer boundary [K]
  1140)     real(r8) :: dTdz(lbc:ubc,-nlevsno:0)    ! snow temperature gradient (col,lyr) [K m-1]
  1141)     real(r8) :: bst_tau                     ! snow aging parameter retrieved from lookup table [hour]
  1142)     real(r8) :: bst_kappa                   ! snow aging parameter retrieved from lookup table [unitless]
  1143)     real(r8) :: bst_drdt0                   ! snow aging parameter retrieved from lookup table [um hr-1]
  1144)     real(r8) :: dr                          ! incremental change in snow effective radius [um]
  1145)     real(r8) :: dr_wet                      ! incremental change in snow effective radius from wet growth [um]
  1146)     real(r8) :: dr_fresh                    ! difference between fresh snow r_e and current r_e [um]
  1147)     real(r8) :: newsnow                     ! fresh snowfall [kg m-2]
  1148)     real(r8) :: refrzsnow                   ! re-frozen snow [kg m-2]
  1149)     real(r8) :: frc_newsnow                 ! fraction of layer mass that is new snow [frc]
  1150)     real(r8) :: frc_oldsnow                 ! fraction of layer mass that is old snow [frc]
  1151)     real(r8) :: frc_refrz                   ! fraction of layer mass that is re-frozen snow [frc]
  1152)     real(r8) :: frc_liq                     ! fraction of layer mass that is liquid water[frc]    
  1153)     real(r8) :: dtime                       ! land model time step [sec]
  1154)     real(r8) :: rhos                        ! snow density [kg m-3]
  1155)     real(r8) :: h2osno_lyr                  ! liquid + solid H2O in snow layer [kg m-2]
  1156) 
  1157) 
  1158)     ! Assign local pointers to derived subtypes components (column-level)
  1159)     t_soisno           => ces%t_soisno
  1160)     snl                => cps%snl
  1161)     t_grnd             => ces%t_grnd
  1162)     dz                 => cps%dz
  1163)     h2osno             => cws%h2osno
  1164)     snw_rds            => cps%snw_rds
  1165)     h2osoi_liq         => cws%h2osoi_liq
  1166)     h2osoi_ice         => cws%h2osoi_ice
  1167)     snot_top           => cps%snot_top
  1168)     dTdz_top           => cps%dTdz_top
  1169)     snw_rds_top        => cps%snw_rds_top
  1170)     sno_liq_top        => cps%sno_liq_top
  1171)     qflx_snow_grnd_col => pwf_a%qflx_snow_grnd
  1172)     qflx_snwcp_ice     => pwf_a%qflx_snwcp_ice
  1173)     qflx_snofrz_lyr    => cwf%qflx_snofrz_lyr
  1174)     do_capsnow         => cps%do_capsnow
  1175)   
  1176) 
  1177)     ! set timestep and step interval
  1178)     dtime = get_step_size()
  1179) 
  1180)     ! loop over columns that have at least one snow layer
  1181)     do fc = 1, num_snowc
  1182)        c_idx = filter_snowc(fc)
  1183) 
  1184)        snl_btm = 0
  1185)        snl_top = snl(c_idx) + 1
  1186) 
  1187)        ! loop over snow layers
  1188)        do i=snl_top,snl_btm,1
  1189)           !
  1190)           !**********  1. DRY SNOW AGING  ***********
  1191)           !
  1192)           h2osno_lyr = h2osoi_liq(c_idx,i) + h2osoi_ice(c_idx,i)
  1193) 
  1194)           ! temperature gradient
  1195)           if (i == snl_top) then 
  1196)              ! top layer
  1197)              t_snotop = t_grnd(c_idx)
  1198)              t_snobtm = (t_soisno(c_idx,i+1)*dz(c_idx,i) + t_soisno(c_idx,i)*dz(c_idx,i+1)) / (dz(c_idx,i)+dz(c_idx,i+1))
  1199)           else
  1200)              t_snotop = (t_soisno(c_idx,i-1)*dz(c_idx,i) + t_soisno(c_idx,i)*dz(c_idx,i-1)) / (dz(c_idx,i)+dz(c_idx,i-1))
  1201)              t_snobtm = (t_soisno(c_idx,i+1)*dz(c_idx,i) + t_soisno(c_idx,i)*dz(c_idx,i+1)) / (dz(c_idx,i)+dz(c_idx,i+1))
  1202)           endif
  1203)           
  1204)           dTdz(c_idx,i) = abs((t_snotop - t_snobtm) / dz(c_idx,i))
  1205)           
  1206)           ! snow density
  1207)           rhos = (h2osoi_liq(c_idx,i)+h2osoi_ice(c_idx,i)) / dz(c_idx,i)
  1208) 
  1209)           ! best-fit table indecies
  1210)           T_idx    = nint((t_soisno(c_idx,i)-223) / 5) + 1
  1211)           Tgrd_idx = nint(dTdz(c_idx,i) / 10) + 1
  1212)           rhos_idx = nint((rhos-50) / 50) + 1
  1213) 
  1214)           ! boundary check:
  1215)           if (T_idx < idx_T_min) then 
  1216)              T_idx = idx_T_min
  1217)           endif
  1218)           if (T_idx > idx_T_max) then 
  1219)              T_idx = idx_T_max
  1220)           endif
  1221)           if (Tgrd_idx < idx_Tgrd_min) then 
  1222)              Tgrd_idx = idx_Tgrd_min
  1223)           endif
  1224)           if (Tgrd_idx > idx_Tgrd_max) then 
  1225)              Tgrd_idx = idx_Tgrd_max
  1226)           endif
  1227)           if (rhos_idx < idx_rhos_min) then 
  1228)              rhos_idx = idx_rhos_min
  1229)           endif
  1230)           if (rhos_idx > idx_rhos_max) then 
  1231)              rhos_idx = idx_rhos_max
  1232)           endif
  1233)              
  1234)           ! best-fit parameters
  1235)           bst_tau   = snowage_tau(rhos_idx,Tgrd_idx,T_idx)
  1236)           bst_kappa = snowage_kappa(rhos_idx,Tgrd_idx,T_idx)     
  1237)           bst_drdt0 = snowage_drdt0(rhos_idx,Tgrd_idx,T_idx)
  1238) 
  1239) 
  1240)           ! change in snow effective radius, using best-fit parameters
  1241)           dr_fresh = snw_rds(c_idx,i)-snw_rds_min
  1242)           dr = (bst_drdt0*(bst_tau/(dr_fresh+bst_tau))**(1/bst_kappa)) * (dtime/3600)
  1243)           
  1244) 
  1245)           !
  1246)           !**********  2. WET SNOW AGING  ***********
  1247)           !
  1248)           ! We are assuming wet and dry evolution occur simultaneously, and 
  1249)           ! the contributions from both can be summed. 
  1250)           ! This is justified by setting the linear offset constant C1_liq_Brun89 to zero [Brun, 1989]
  1251)           
  1252)           ! liquid water faction
  1253)           frc_liq = min(0.1_r8, (h2osoi_liq(c_idx,i) / (h2osoi_liq(c_idx,i)+h2osoi_ice(c_idx,i))))
  1254) 
  1255)           !dr_wet = 1E6_r8*(dtime*(C1_liq_Brun89 + C2_liq_Brun89*(frc_liq**(3))) / (4*SHR_CONST_PI*(snw_rds(c_idx,i)/1E6)**(2)))
  1256)           !simplified, units of microns:
  1257)           dr_wet = 1E18_r8*(dtime*(C2_liq_Brun89*(frc_liq**(3))) / (4*SHR_CONST_PI*snw_rds(c_idx,i)**(2)))
  1258) 
  1259)           dr = dr + dr_wet
  1260)     
  1261)           !
  1262)           !**********  3. SNOWAGE SCALING (TURNED OFF BY DEFAULT)  *************
  1263)           !
  1264)           ! Multiply rate of change of effective radius by some constant, xdrdt
  1265)           if (flg_snoage_scl) then
  1266)              dr = dr*xdrdt
  1267)           endif
  1268) 
  1269)           
  1270)           !
  1271)           !**********  4. INCREMENT EFFECTIVE RADIUS, ACCOUNTING FOR:  ***********
  1272)           !               DRY AGING
  1273)           !               WET AGING
  1274)           !               FRESH SNOW
  1275)           !               RE-FREEZING
  1276)           !
  1277)           ! new snowfall [kg/m2]
  1278)           if (do_capsnow(c_idx)) then
  1279)              newsnow = max(0._r8, (qflx_snwcp_ice(c_idx)*dtime))
  1280)           else
  1281)              newsnow = max(0._r8, (qflx_snow_grnd_col(c_idx)*dtime))
  1282)           endif
  1283) 
  1284)           ! snow that has re-frozen [kg/m2]
  1285)           refrzsnow = max(0._r8, (qflx_snofrz_lyr(c_idx,i)*dtime))
  1286)           
  1287)           ! fraction of layer mass that is re-frozen
  1288)           frc_refrz = refrzsnow / h2osno_lyr
  1289)                   
  1290)           ! fraction of layer mass that is new snow
  1291)           if (i == snl_top) then
  1292)              frc_newsnow = newsnow / h2osno_lyr
  1293)           else
  1294)              frc_newsnow = 0._r8
  1295)           endif
  1296) 
  1297)           if ((frc_refrz + frc_newsnow) > 1._r8) then
  1298)              frc_refrz = frc_refrz / (frc_refrz + frc_newsnow)
  1299)              frc_newsnow = 1._r8 - frc_refrz
  1300)              frc_oldsnow = 0._r8
  1301)           else
  1302)              frc_oldsnow = 1._r8 - frc_refrz - frc_newsnow
  1303)           endif
  1304) 
  1305)           ! mass-weighted mean of fresh snow, old snow, and re-frozen snow effective radius
  1306)           snw_rds(c_idx,i) = (snw_rds(c_idx,i)+dr)*frc_oldsnow + snw_rds_min*frc_newsnow + snw_rds_refrz*frc_refrz
  1307) 
  1308)              
  1309)           !
  1310)           !**********  5. CHECK BOUNDARIES   ***********
  1311)           !
  1312)           ! boundary check
  1313)           if (snw_rds(c_idx,i) < snw_rds_min) then
  1314)              snw_rds(c_idx,i) = snw_rds_min
  1315)           endif
  1316) 
  1317)           if (snw_rds(c_idx,i) > snw_rds_max) then
  1318)              snw_rds(c_idx,i) = snw_rds_max
  1319)           end if
  1320) 
  1321)           ! set top layer variables for history files
  1322)           if (i == snl_top) then
  1323)              snot_top(c_idx)    = t_soisno(c_idx,i)
  1324)              dTdz_top(c_idx)    = dTdz(c_idx,i)
  1325)              snw_rds_top(c_idx) = snw_rds(c_idx,i)
  1326)              sno_liq_top(c_idx) = h2osoi_liq(c_idx,i) / (h2osoi_liq(c_idx,i)+h2osoi_ice(c_idx,i))
  1327)           endif
  1328) 
  1329)        enddo
  1330)     enddo
  1331) 
  1332)     ! Special case: snow on ground, but not enough to have defined a snow layer:
  1333)     !   set snw_rds to fresh snow grain size:
  1334)     do fc = 1, num_nosnowc
  1335)        c_idx = filter_nosnowc(fc)
  1336)        if (h2osno(c_idx) > 0._r8) then
  1337)           snw_rds(c_idx,0) = snw_rds_min
  1338)        endif
  1339)     enddo
  1340)         
  1341)   end subroutine SnowAge_grain
