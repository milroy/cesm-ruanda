    62)     ! Scan list from right end until element <= reference is found
    63)     DO
    64)       j = j - 1

    66)     END DO

    72)       itemp = order(i); order(i) = order(j); order(j) = itemp

    75)       EXIT

    77)       EXIT
    78)     END IF
    79)   END DO
