    39) subroutine gw_oro_src(ncol, &
    40)      u, v, t, sgh, pmid, pint, dpm, zm, nm, &
    41)      src_level, tend_level, tau, ubm, ubi, xv, yv, c)
    42)   use gw_common, only: pgwv, fcrit2, rair
    43)   use gw_utils, only: get_unit_vector, dot_2d, midpoint_interp
    44)   !-----------------------------------------------------------------------
    45)   ! Orographic source for multiple gravity wave drag parameterization.
    46)   !
    47)   ! The stress is returned for a single wave with c=0, over orography.
    48)   ! For points where the orographic variance is small (including ocean),
    49)   ! the returned stress is zero.
    50)   !------------------------------Arguments--------------------------------
    51)   ! Column dimension.
    52)   integer, intent(in) :: ncol
    53) 
    54)   ! Midpoint zonal/meridional winds.
    55)   real(r8), intent(in) :: u(ncol,pver), v(ncol,pver)
    56)   ! Midpoint temperatures.
    57)   real(r8), intent(in) :: t(ncol,pver)
    58)   ! Standard deviation of orography.
    59)   real(r8), intent(in) :: sgh(ncol)
    60)   ! Midpoint and interface pressures.
    61)   real(r8), intent(in) :: pmid(ncol,pver), pint(ncol,0:pver)
    62)   ! Layer thickness (pressure delta).
    63)   real(r8), intent(in) :: dpm(ncol,pver)
    64)   ! Midpoint altitudes.
    65)   real(r8), intent(in) :: zm(ncol,pver)
    66)   ! Midpoint Brunt-Vaisalla frequencies.
    67)   real(r8), intent(in) :: nm(ncol,pver)
    68) 
    69)   ! Indices of top gravity wave source level and lowest level where wind
    70)   ! tendencies are allowed.
    71)   integer, intent(out) :: src_level(ncol)
    72)   integer, intent(out) :: tend_level(ncol)
    73) 
    74)   ! Wave Reynolds stress.
    75)   real(r8), intent(out) :: tau(ncol,-pgwv:pgwv,0:pver)
    76)   ! Projection of wind at midpoints and interfaces.
    77)   real(r8), intent(out) :: ubm(ncol,pver), ubi(ncol,0:pver)
    78)   ! Unit vectors of source wind (zonal and meridional components).
    79)   real(r8), intent(out) :: xv(ncol), yv(ncol)
    80)   ! Phase speeds.
    81)   real(r8), intent(out) :: c(ncol,-pgwv:pgwv)
    82) 
    83)   !---------------------------Local Storage-------------------------------
    84)   ! Column and level indices.
    85)   integer :: i, k
    86) 
    87)   ! Surface streamline displacement height (2*sgh).
    88)   real(r8) :: hdsp(ncol)
    89)   ! Max orographic standard deviation to use.
    90)   real(r8) :: sghmax
    91)   ! c=0 stress from orography.
    92)   real(r8) :: tauoro(ncol)
    93)   ! Averages over source region.
    94)   real(r8) :: nsrc(ncol) ! B-V frequency.
    95)   real(r8) :: rsrc(ncol) ! Density.
    96)   real(r8) :: usrc(ncol) ! Zonal wind.
    97)   real(r8) :: vsrc(ncol) ! Meridional wind.
    98) 
    99)   ! Difference in interface pressure across source region.
   100)   real(r8) :: dpsrc(ncol)
   101) 
   102)   ! Limiters (min/max values)
   103)   ! min surface displacement height for orographic waves
   104)   real(r8), parameter :: orohmin = 10._r8
   105)   ! min wind speed for orographic waves
   106)   real(r8), parameter :: orovmin = 2._r8
   107) 
   108) !--------------------------------------------------------------------------
   109) ! Average the basic state variables for the wave source over the depth of
   110) ! the orographic standard deviation. Here we assume that the appropiate
   111) ! values of wind, stability, etc. for determining the wave source are
   112) ! averages over the depth of the atmosphere penterated by the typical
   113) ! mountain.
   114) ! Reduces to the bottom midpoint values when sgh=0, such as over ocean.
   115) !--------------------------------------------------------------------------
   116) 
   117)   hdsp = 2.0_r8 * sgh
   118) 
   119)   k = pver
   120)   src_level = k-1
   121)   rsrc = pmid(:,k)/(rair*t(:,k)) * dpm(:,k)
   122)   usrc = u(:,k) * dpm(:,k)
   123)   vsrc = v(:,k) * dpm(:,k)
   124)   nsrc = nm(:,k)* dpm(:,k)
   125) 
   126)   do k = pver-1, pver/2, -1
   127)      do i = 1, ncol
   128)         if (hdsp(i) > sqrt(zm(i,k)*zm(i,k+1))) then
   129)            src_level(i) = k-1
   130)            rsrc(i) = rsrc(i) + pmid(i,k) / (rair*t(i,k))* dpm(i,k)
   131)            usrc(i) = usrc(i) + u(i,k) * dpm(i,k)
   132)            vsrc(i) = vsrc(i) + v(i,k) * dpm(i,k)
   133)            nsrc(i) = nsrc(i) + nm(i,k)* dpm(i,k)
   134)         end if
   135)      end do
   136)   end do
   137) 
   138)   do i = 1, ncol
   139)      dpsrc(i) = pint(i,pver) - pint(i,src_level(i))
   140)   end do
   141) 
   142)   rsrc = rsrc / dpsrc
   143)   usrc = usrc / dpsrc
   144)   vsrc = vsrc / dpsrc
   145)   nsrc = nsrc / dpsrc
   146) 
   147)   ! Get the unit vector components and magnitude at the surface.
   148)   call get_unit_vector(usrc, vsrc, xv, yv, ubi(:,pver))
   149) 
   150)   ! Project the local wind at midpoints onto the source wind.
   151)   do k = 1, pver
   152)      ubm(:,k) = dot_2d(u(:,k), v(:,k), xv, yv)
   153)   end do
   154) 
   155)   ! Compute the interface wind projection by averaging the midpoint winds.
   156)   ! Use the top level wind at the top interface.
   157)   ubi(:,0) = ubm(:,1)
   158) 
   159)   ubi(:,1:pver-1) = midpoint_interp(ubm)
   160) 
   161)   ! Determine the orographic c=0 source term following McFarlane (1987).
   162)   ! Set the source top interface index to pver, if the orographic term is
   163)   ! zero.
   164)   do i = 1, ncol
   165)      if ((ubi(i,pver) .gt. orovmin) .and. (hdsp(i) .gt. orohmin)) then
   166)         sghmax = fcrit2 * (ubi(i,pver) / nsrc(i))**2
   167)         tauoro(i) = oroko2 * min(hdsp(i)**2, sghmax) * rsrc(i) * nsrc(i) &
   168)              * ubi(i,pver)
   169)      else
   170)         tauoro(i) = 0._r8
   171)         src_level(i) = pver
   172)      end if
   173)   end do
   174) 
   175)   ! Set the phase speeds and wave numbers in the direction of the source
   176)   ! wind. Set the source stress magnitude (positive only, note that the
   177)   ! sign of the stress is the same as (c-u).
   178)   do k = pver, minval(src_level), -1
   179)      where (src_level <= k) tau(:,0,k) = tauoro
   180)   end do
   181) 
   182)   ! Allow wind tendencies all the way to the model bottom.
   183)   tend_level = pver
   184) 
   185)   ! No spectrum; phase speed is just 0.
   186)   c = 0._r8
   187) 
   188) end subroutine gw_oro_src
   189) 
   190) end module gw_oro
