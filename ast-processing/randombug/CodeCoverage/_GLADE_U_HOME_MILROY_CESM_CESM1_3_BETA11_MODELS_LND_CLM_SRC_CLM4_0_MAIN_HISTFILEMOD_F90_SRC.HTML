   419) ! !INTERFACE:
   420)   subroutine hist_htapes_build ()
   421) !
   422) ! !DESCRIPTION:
   423) ! Initialize history file for initial or continuation run.  For example,
   424) ! on an initial run, this routine initializes ``ntapes'' history files.
   425) ! On a restart run, this routine only initializes history files declared
   426) ! beyond what existed on the previous run.  Files which already existed on
   427) ! the previous run have already been initialized (i.e. named and opened)
   428) ! in routine restart\_history.  Loop over tapes and fields per tape setting
   429) ! appropriate variables and calling appropriate routines
   430) !
   431) ! !USES:
   432)     use clm_time_manager, only: get_prev_time
   433)     use clm_varcon      , only: secspday
   434) !
   435) ! !ARGUMENTS:
   436)     implicit none
   437) !
   438) ! !REVISION HISTORY:
   439) ! Created by Mariana Vertenstein
   440) !
   441) !
   442) ! !LOCAL VARIABLES:
   443) !EOP
   444)     integer :: i                   ! index
   445)     integer :: ier                 ! error code
   446)     integer :: t, f                ! tape, field indices
   447)     integer :: day, sec            ! day and seconds from base date
   448)     character(len=*),parameter :: subname = 'hist_htapes_build'
   449) !-----------------------------------------------------------------------
   450) 
   451)     if (masterproc) then
   452)        write(iulog,*)  trim(subname),' Initializing clm2 history files'
   453)        write(iulog,'(72a1)') ("-",i=1,60)
   454)        call shr_sys_flush(iulog)
   455)     endif
   456) 
   457)     ! Define field list information for all history files.
   458)     ! Update ntapes to reflect number of active history files
   459)     ! Note - branch runs can have additional auxiliary history files
   460)     ! declared).
   461) 
   462)     call htapes_fieldlist()
   463) 
   464)     ! Determine if gridcell (xy) averaging is done for all fields on tape
   465) 
   466)     do t=1,ntapes
   467)        tape(t)%dov2xy = hist_dov2xy(t)
   468)        write(iulog,*)trim(subname),' hist tape = ',t,&
   469)             ' written with dov2xy= ',tape(t)%dov2xy
   470)     end do
   471) 
   472)     ! Set number of time samples in each history file and
   473)     ! Note - the following entries will be overwritten by history restart
   474)     ! Note - with netcdf, only 1 (ncd_double) and 2 (ncd_float) are allowed
   475) 
   476)     do t=1,ntapes
   477)        tape(t)%ntimes = 0
   478)        tape(t)%dov2xy = hist_dov2xy(t)
   479)        tape(t)%nhtfrq = hist_nhtfrq(t)
   480)        tape(t)%mfilt = hist_mfilt(t)
   481)        if (hist_ndens(t) == 1) then
   482)           tape(t)%ncprec = ncd_double
   483)        else
   484)           tape(t)%ncprec = ncd_float
   485)        endif
   486)     end do
   487) 
   488)     ! Set time of beginning of current averaging interval
   489)     ! First etermine elapased time since reference date
   490) 
   491)     call get_prev_time(day, sec)
   492)     do t=1,ntapes
   493)        tape(t)%begtime = day + sec/secspday
   494)     end do

   623) ! !IROUTINE: htapes_fieldlist
   624) !
   625) ! !INTERFACE:
   626)   subroutine htapes_fieldlist()
   627) !
   628) ! !DESCRIPTION:
   629) ! Define the contents of each history file based on namelist
   630) ! input for initial or branch run, and restart data if a restart run.
   631) ! Use arrays fincl and fexcl to modify default history tape contents.
   632) ! Then sort the result alphanumerically.
   633) !
   634) ! !USES:
   635) !
   636) ! !ARGUMENTS:
   637)     implicit none
   638) !
   639) ! !REVISION HISTORY:
   640) ! Created by Mariana Vertenstein
   641) !
   642) !
   643) ! !LOCAL VARIABLES:
   644) !EOP
   645)     integer :: t, f                ! tape, field indices
   646)     integer :: ff                  ! index into include, exclude and fprec list
   647)     character(len=max_namlen) :: name       ! field name portion of fincl (i.e. no avgflag separator)
   648)     character(len=max_namlen) :: mastername ! name from masterlist field
   649)     character(len=1)  :: avgflag    ! averaging flag
   650)     character(len=1)  :: prec_acc   ! history buffer precision flag
   651)     character(len=1)  :: prec_wrt   ! history buffer write precision flag
   652)     type (history_entry) :: tmp     ! temporary used for swapping
   653)     character(len=*),parameter :: subname = 'htapes_fieldlist'
   654) !-----------------------------------------------------------------------
   655) 
   656)     ! Override averaging flag for all fields on a particular tape
   657)     ! if namelist input so specifies
   658) 
   659)     do t=1,max_tapes
   660)        if (hist_avgflag_pertape(t) /= ' ') then
   661)           call masterlist_change_timeavg (t)
   662)        end if
   663)     end do
   664) 
   665)     fincl(:,1) = hist_fincl1(:)
   666)     fincl(:,2) = hist_fincl2(:)
   667)     fincl(:,3) = hist_fincl3(:)
   668)     fincl(:,4) = hist_fincl4(:)
   669)     fincl(:,5) = hist_fincl5(:)
   670)     fincl(:,6) = hist_fincl6(:)
   671) 
   672)     fexcl(:,1) = hist_fexcl1(:)
   673)     fexcl(:,2) = hist_fexcl2(:)
   674)     fexcl(:,3) = hist_fexcl3(:)
   675)     fexcl(:,4) = hist_fexcl4(:)
   676)     fexcl(:,5) = hist_fexcl5(:)
   677)     fexcl(:,6) = hist_fexcl6(:)
   678) 
   679) 
   680)     ! First ensure contents of fincl and fexcl are valid names
   681) 
   682)     do t = 1,max_tapes
   683)        f = 1
   684)        do while (f < max_flds .and. fincl(f,t) /= ' ')
   685)           name = getname (fincl(f,t))
   686)           do ff = 1,nfmaster
   687)              mastername = masterlist(ff)%field%name
   688)              if (name == mastername) exit
   689)           end do
   690)           if (name /= mastername) then
   691)              write(iulog,*) trim(subname),' ERROR: ', trim(name), ' in fincl(', f, ') ',&
   692)                   'for history tape ',t,' not found'
   693)              call endrun()
   694)           end if
   695)           f = f + 1
   696)        end do
   697) 
   698)        f = 1
   699)        do while (f < max_flds .and. fexcl(f,t) /= ' ')
   700)           do ff = 1,nfmaster
   701)              mastername = masterlist(ff)%field%name
   702)              if (fexcl(f,t) == mastername) exit
   703)           end do
   704)           if (fexcl(f,t) /= mastername) then
   705)              write(iulog,*) trim(subname),' ERROR: ', fexcl(f,t), ' in fexcl(', f, ') ', &
   706)                   'for history tape ',t,' not found'
   707)              call endrun()
   708)           end if
   709)           f = f + 1
   710)        end do
   711)     end do
   712) 
   713)     tape(:)%nflds = 0
   714)     do t = 1,max_tapes
   715) 
   716)        ! Loop through the masterlist set of field names and determine if any of those
   717)        ! are in the FINCL or FEXCL arrays
   718)        ! The call to list_index determines the index in the FINCL or FEXCL arrays
   719)        ! that the masterlist field corresponds to
   720)        ! Add the field to the tape if specified via namelist (FINCL[1-max_tapes]),
   721)        ! or if it is on by default and was not excluded via namelist (FEXCL[1-max_tapes]).
   722) 
   723)        do f = 1,nfmaster
   724)           mastername = masterlist(f)%field%name
   725)           call list_index (fincl(1,t), mastername, ff)
   726) 
   727)           if (ff > 0) then
   728) 
   729)              ! if field is in include list, ff > 0 and htape_addfld
   730)              ! will not be called for field
   731) 
   732)              avgflag = getflag (fincl(ff,t))
   733)              call htape_addfld (t, f, avgflag)
   734) 
   735)           else if (.not. hist_empty_htapes) then
   736) 
   737)              ! find index of field in exclude list
   738) 
   739)              call list_index (fexcl(1,t), mastername, ff)
   740) 
   741)              ! if field is in exclude list, ff > 0 and htape_addfld
   742)              ! will not be called for field
   743)              ! if field is not in exclude list, ff =0 and htape_addfld
   744)              ! will be called for field (note that htape_addfld will be
   745)              ! called below only if field is not in exclude list OR in
   746)              ! include list
   747) 
   748)              if (ff == 0 .and. masterlist(f)%actflag(t)) then
   749)                 call htape_addfld (t, f, ' ')
   750)              end if
   751) 
   752)           end if
   753)        end do
   754) 
   755)        ! Specification of tape contents now complete.
   756)        ! Sort each list of active entries
   757) 
   758)        do f = tape(t)%nflds-1,1,-1
   759)           do ff = 1,f
   760)              if (tape(t)%hlist(ff)%field%name > tape(t)%hlist(ff+1)%field%name) then
   761) 
   762)                 tmp = tape(t)%hlist(ff)
   763)                 tape(t)%hlist(ff  ) = tape(t)%hlist(ff+1)
   764)                 tape(t)%hlist(ff+1) = tmp
   765) 
   766)              else if (tape(t)%hlist(ff)%field%name == tape(t)%hlist(ff+1)%field%name) then
   767) 
   768)                 write(iulog,*) trim(subname),' ERROR: Duplicate field ', &
   769)                    tape(t)%hlist(ff)%field%name, &
   770)                    't,ff,name=',t,ff,tape(t)%hlist(ff+1)%field%name
   771)                 call endrun()
   772) 
   773)              end if
   774)           end do
   775)        end do
   776) 
   777)        if (masterproc) then
   778)           if (tape(t)%nflds > 0) then
   779)              write(iulog,*) trim(subname),' : Included fields tape ',t,'=',tape(t)%nflds
   780)           end if
   781)           do f = 1,tape(t)%nflds
   782)              write(iulog,*) f,' ',tape(t)%hlist(f)%field%name, &
   783)                   tape(t)%hlist(f)%field%num2d,' ',tape(t)%hlist(f)%avgflag
   784)           end do
   785)           call shr_sys_flush(iulog)
   786)        end if
   787)     end do
   788) 
   789)     ! Determine total number of active history tapes
   790) 
   791)     ntapes = 0
   792)     do t = max_tapes,1,-1
   793)        if (tape(t)%nflds > 0) then
   794)           ntapes = t
   795)           exit
   796)        end if
   797)     end do
   798) 
   799)     ! Ensure there are no "holes" in tape specification, i.e. empty tapes.
   800)     ! Enabling holes should not be difficult if necessary.
   801) 
   802)     do t = 1,ntapes
   803)        if (tape(t)%nflds  ==  0) then
   804)           write(iulog,*) trim(subname),' ERROR: Tape ',t,' is empty'
   805)           call endrun()
   806)        end if
   807)     end do
   808) 
   809)     ! Check that the number of history files declared does not exceed
   810)     ! the maximum allowed.
   811) 
   812)     if (ntapes > max_tapes) then
   813)        write(iulog,*) trim(subname),' ERROR: Too many history files declared, max_tapes=',max_tapes
   814)        call endrun()
   815)     end if
   816) 
   817)     ! Change 1d output per tape output flag if requested - only for history
   818)     ! tapes where 2d xy averaging is not enabled
   819) 
   820)     do t = 1,ntapes
   821)        if (hist_type1d_pertape(t) /= ' ' .and. (.not. hist_dov2xy(t))) then
   822)           select case (trim(hist_type1d_pertape(t)))
   823)           case ('PFTS','COLS', 'LAND', 'GRID')
   824)              if ( masterproc ) &
   825)              write(iulog,*)'history tape ',t,' will have 1d output type of ',hist_type1d_pertape(t)
   826)           case default
   827)              write(iulog,*) trim(subname),' ERROR: unknown namelist type1d per tape=',hist_type1d_pertape(t)
   828)              call endrun()
   829)           end select
   830)        end if
   831)     end do
   832) 
   833)     if (masterproc) then
   834)        write(iulog,*) 'There will be a total of ',ntapes,' history tapes'
   835)        do t=1,ntapes
   836)           write(iulog,*)
   837)           if (hist_nhtfrq(t) == 0) then
   838)              write(iulog,*)'History tape ',t,' write frequency is MONTHLY'
   839)           else
   840)              write(iulog,*)'History tape ',t,' write frequency = ',hist_nhtfrq(t)
   841)           endif
   842)           if (hist_dov2xy(t)) then
   843)              write(iulog,*)'All fields on history tape ',t,' are grid averaged'
   844)           else
   845)              write(iulog,*)'All fields on history tape ',t,' are not grid averaged'
   846)           end if
   847)           write(iulog,*)'Number of time samples on history tape ',t,' is ',hist_mfilt(t)
   848)           write(iulog,*)'Output precision on history tape ',t,'=',hist_ndens(t)
   849)           write(iulog,*)
   850)        end do
   851)        call shr_sys_flush(iulog)
   852)     end if
   853) 
   854)     ! Set flag indicating h-tape contents are now defined (needed by masterlist_addfld)
   855) 
   856)     htapes_defined = .true.
   857) 
   858)   end subroutine htapes_fieldlist

   863) ! !IROUTINE: htape_addfld

   865) ! !INTERFACE:
   866)   subroutine htape_addfld (t, f, avgflag)
   867) !
   868) ! !DESCRIPTION:
   869) ! Add a field to the active list for a history tape. Copy the data from
   870) ! the master field list to the active list for the tape.
   871) !
   872) ! !USES:
   873) !
   874) ! !ARGUMENTS:
   875)     implicit none
   876)     integer, intent(in) :: t                 ! history tape index
   877)     integer, intent(in) :: f                 ! field index from master field list
   878)     character(len=1), intent(in) :: avgflag  ! time averaging flag
   879) !
   880) ! !REVISION HISTORY:
   881) ! Created by Mariana Vertenstein
   882) !
   883) !
   884) ! !LOCAL VARIABLES:
   885) !EOP
   886)     integer :: n                    ! field index on defined tape
   887)     character(len=8) :: type1d      ! clm pointer 1d type
   888)     character(len=8) :: type1d_out  ! history buffer 1d type
   889)     integer :: begp, endp           ! per-proc beginning and ending pft indices
   890)     integer :: begc, endc           ! per-proc beginning and ending column indices
   891)     integer :: begl, endl           ! per-proc beginning and ending landunit indices
   892)     integer :: begg, endg           ! per-proc gridcell ending gridcell indices
   893)     integer :: numa                 ! total number of atm cells across all processors
   894)     integer :: numg                 ! total number of gridcells across all processors
   895)     integer :: numl                 ! total number of landunits across all processors
   896)     integer :: numc                 ! total number of columns across all processors
   897)     integer :: nump                 ! total number of pfts across all processors
   898)     integer :: num2d                ! size of second dimension (e.g. .number of vertical levels)
   899)     integer :: beg1d_out,end1d_out  ! history output per-proc 1d beginning and ending indices
   900)     integer :: num1d_out            ! history output 1d size
   901)     character(len=*),parameter :: subname = 'htape_addfld'
   902) !-----------------------------------------------------------------------
   903) 
   904)     ! Ensure that it is not to late to add a field to the history tape
   905) 
   906)     if (htapes_defined) then
   907)        write(iulog,*) trim(subname),' ERROR: attempt to add field ', &
   908)             masterlist(f)%field%name, ' after history files are set'
   909)        call endrun()
   910)     end if
   911) 
   912)     tape(t)%nflds = tape(t)%nflds + 1
   913)     n = tape(t)%nflds
   914) 
   915)     ! Copy field information
   916) 
   917)     tape(t)%hlist(n)%field = masterlist(f)%field
   918) 
   919)     ! Determine bounds
   920) 
   921)     call get_proc_bounds(begg, endg, begl, endl, begc, endc, begp, endp)
   922)     call get_proc_global(numg, numl, numc, nump)
   923) 
   924)     ! Modify type1d_out if necessary
   925) 
   926)     if (hist_dov2xy(t)) then
   927) 
   928)        ! If xy output averaging is requested, set output 1d type to grlnd
   929)        ! ***NOTE- the following logic is what permits non lat/lon grids to
   930)        ! be written to clm history file
   931) 
   932)        type1d = tape(t)%hlist(n)%field%type1d
   933) 
   934)        if (type1d == nameg .or. &
   935)            type1d == namel .or. &
   936)            type1d == namec .or. &
   937)            type1d == namep) then
   938)           tape(t)%hlist(n)%field%type1d_out = grlnd
   939)        end if
   940)        if (type1d == grlnd) then
   941)           tape(t)%hlist(n)%field%type1d_out = grlnd
   942)        end if
   943) 
   944)     else if (hist_type1d_pertape(t) /= ' ') then
   945) 
   946)        ! Set output 1d type  based on namelist setting of  hist_type1d_pertape
   947)        ! Only applies to tapes when xy output is not required
   948) 
   949)        type1d = tape(t)%hlist(n)%field%type1d
   950) 
   951)        select case (trim(hist_type1d_pertape(t)))
   952)        case('GRID')
   953)           tape(t)%hlist(n)%field%type1d_out = nameg
   954)        case('LAND')
   955)           tape(t)%hlist(n)%field%type1d_out = namel
   956)        case('COLS')
   957)           tape(t)%hlist(n)%field%type1d_out = namec
   958)        case ('PFTS')
   959)           tape(t)%hlist(n)%field%type1d_out = namep
   960)        case default
   961)           write(iulog,*) trim(subname),' ERROR: unknown input hist_type1d_pertape= ', hist_type1d_pertape(t)
   962)           call endrun()
   963)        end select
   964) 
   965)     endif
   966) 
   967)     ! Determine output 1d dimensions
   968) 
   969)     type1d_out = tape(t)%hlist(n)%field%type1d_out
   970)     if (type1d_out == grlnd) then
   971)        beg1d_out = begg
   972)        end1d_out = endg
   973)        num1d_out = numg
   974)     else if (type1d_out == nameg) then
   975)        beg1d_out = begg
   976)        end1d_out = endg
   977)        num1d_out = numg
   978)     else if (type1d_out == namel) then
   979)        beg1d_out = begl
   980)        end1d_out = endl
   981)        num1d_out = numl
   982)     else if (type1d_out == namec) then
   983)        beg1d_out = begc
   984)        end1d_out = endc
   985)        num1d_out = numc
   986)     else if (type1d_out == namep) then
   987)        beg1d_out = begp
   988)        end1d_out = endp
   989)        num1d_out = nump
   990)     else
   991)        write(iulog,*) trim(subname),' ERROR: incorrect value of type1d_out= ',type1d_out
   992)        call endrun()
   993)     end if
   994) 
   995)     tape(t)%hlist(n)%field%beg1d_out = beg1d_out
   996)     tape(t)%hlist(n)%field%end1d_out = end1d_out
   997)     tape(t)%hlist(n)%field%num1d_out = num1d_out
   998)     
   999)     ! Alloccate and initialize history buffer and related info
  1000) 
  1001)     num2d = tape(t)%hlist(n)%field%num2d
  1002)     allocate (tape(t)%hlist(n)%hbuf(beg1d_out:end1d_out,num2d))
  1003)     allocate (tape(t)%hlist(n)%nacs(beg1d_out:end1d_out,num2d))
  1004)     tape(t)%hlist(n)%hbuf(:,:) = 0._r8
  1005)     tape(t)%hlist(n)%nacs(:,:) = 0
  1006) 
  1007)     ! Set time averaging flag based on masterlist setting or
  1008)     ! override the default averaging flag with namelist setting
  1009) 
  1010)     select case (avgflag)
  1011)     case (' ')
  1012)        tape(t)%hlist(n)%avgflag = masterlist(f)%avgflag(t)
  1013)     case ('A','I','X','M')
  1014)        tape(t)%hlist(n)%avgflag = avgflag
  1015)     case default
  1016)        write(iulog,*) trim(subname),' ERROR: unknown avgflag=', avgflag
  1017)        call endrun()
  1018)     end select
  1019) 
  1020)   end subroutine htape_addfld
