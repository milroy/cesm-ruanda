    89)    real(r8), pointer :: m_frootc_storage_to_litter(:)

    99)    real(r8), pointer :: m_livecrootc_xfer_to_litter(:)
   100)    real(r8), pointer :: m_livestemc_storage_to_litter(:)

   104) ! local pointers to implicit in/out arrays
   105)    real(r8), pointer :: cwdc(:)               ! (gC/m2) coarse woody debris C
   106)    real(r8), pointer :: litr1c(:)             ! (gC/m2) litter labile C
   107)    real(r8), pointer :: litr2c(:)             ! (gC/m2) litter cellulose C
   108)    real(r8), pointer :: litr3c(:)             ! (gC/m2) litter lignin C
   109)    real(r8), pointer :: deadcrootc(:)         ! (gC/m2) dead coarse root C
   110)    real(r8), pointer :: deadcrootc_storage(:) ! (gC/m2) dead coarse root C storage
   111)    real(r8), pointer :: deadcrootc_xfer(:)    !(gC/m2) dead coarse root C transfer
   112)    real(r8), pointer :: deadstemc(:)          ! (gC/m2) dead stem C
   113)    real(r8), pointer :: deadstemc_storage(:)  ! (gC/m2) dead stem C storage
   114)    real(r8), pointer :: deadstemc_xfer(:)     ! (gC/m2) dead stem C transfer
   115)    real(r8), pointer :: frootc(:)             ! (gC/m2) fine root C
   116)    real(r8), pointer :: frootc_storage(:)     ! (gC/m2) fine root C storage
   117)    real(r8), pointer :: frootc_xfer(:)        ! (gC/m2) fine root C transfer
   118)    real(r8), pointer :: gresp_storage(:)      ! (gC/m2) growth respiration storage
   119)    real(r8), pointer :: gresp_xfer(:)         ! (gC/m2) growth respiration transfer
   120)    real(r8), pointer :: leafc(:)              ! (gC/m2) leaf C
   121)    real(r8), pointer :: leafc_storage(:)      ! (gC/m2) leaf C storage
   122)    real(r8), pointer :: leafc_xfer(:)         ! (gC/m2) leaf C transfer
   123)    real(r8), pointer :: livecrootc(:)         ! (gC/m2) live coarse root C
   124)    real(r8), pointer :: livecrootc_storage(:) ! (gC/m2) live coarse root C storage
   125)    real(r8), pointer :: livecrootc_xfer(:)    !(gC/m2) live coarse root C transfer
   126)    real(r8), pointer :: livestemc(:)          ! (gC/m2) live stem C
   127)    real(r8), pointer :: livestemc_storage(:)  ! (gC/m2) live stem C storage
   128)    real(r8), pointer :: livestemc_xfer(:)     ! (gC/m2) live stem C transfer
   129) !
   130) !
   131) ! local pointers to implicit out arrays
   132) !
   133) !
   134) ! !OTHER LOCAL VARIABLES:
   135)    integer :: c,p      ! indices
   136)    integer :: fp,fc    ! lake filter indices
   137)    real(r8):: dt       ! radiation time step (seconds)
   138) !
   139) !EOP
   140) !-----------------------------------------------------------------------
   141)     ! assign local pointers at the column level
   142)     m_deadcrootc_storage_to_litr1c => ccf%m_deadcrootc_storage_to_litr1c
   143)     m_deadcrootc_to_cwdc           => ccf%m_deadcrootc_to_cwdc
   144)     m_deadcrootc_xfer_to_litr1c    => ccf%m_deadcrootc_xfer_to_litr1c
   145)     m_deadstemc_storage_to_litr1c  => ccf%m_deadstemc_storage_to_litr1c
   146)     m_deadstemc_to_cwdc            => ccf%m_deadstemc_to_cwdc
   147)     m_deadstemc_xfer_to_litr1c     => ccf%m_deadstemc_xfer_to_litr1c
   148)     m_frootc_storage_to_litr1c     => ccf%m_frootc_storage_to_litr1c
   149)     m_frootc_to_litr1c             => ccf%m_frootc_to_litr1c
   150)     m_frootc_to_litr2c             => ccf%m_frootc_to_litr2c
   151)     m_frootc_to_litr3c             => ccf%m_frootc_to_litr3c
   152)     m_frootc_xfer_to_litr1c        => ccf%m_frootc_xfer_to_litr1c
   153)     m_gresp_storage_to_litr1c      => ccf%m_gresp_storage_to_litr1c
   154)     m_gresp_xfer_to_litr1c         => ccf%m_gresp_xfer_to_litr1c
   155)     m_leafc_storage_to_litr1c      => ccf%m_leafc_storage_to_litr1c
   156)     m_leafc_to_litr1c              => ccf%m_leafc_to_litr1c
   157)     m_leafc_to_litr2c              => ccf%m_leafc_to_litr2c
   158)     m_leafc_to_litr3c              => ccf%m_leafc_to_litr3c
   159)     m_leafc_xfer_to_litr1c         => ccf%m_leafc_xfer_to_litr1c
   160)     m_livecrootc_storage_to_litr1c => ccf%m_livecrootc_storage_to_litr1c
   161)     m_livecrootc_to_cwdc           => ccf%m_livecrootc_to_cwdc
   162)     m_livecrootc_xfer_to_litr1c    => ccf%m_livecrootc_xfer_to_litr1c
   163)     m_livestemc_storage_to_litr1c  => ccf%m_livestemc_storage_to_litr1c
   164)     m_livestemc_to_cwdc            => ccf%m_livestemc_to_cwdc
   165)     m_livestemc_xfer_to_litr1c     => ccf%m_livestemc_xfer_to_litr1c
   166)     cwdc                           => ccs%cwdc
   167)     litr1c                         => ccs%litr1c
   168)     litr2c                         => ccs%litr2c
   169)     litr3c                         => ccs%litr3c
   170) 
   171)     ! assign local pointers at the pft level
   172)     m_deadcrootc_storage_to_litter => pcf%m_deadcrootc_storage_to_litter
   173)     m_deadcrootc_to_litter         => pcf%m_deadcrootc_to_litter
   174)     m_deadcrootc_xfer_to_litter    => pcf%m_deadcrootc_xfer_to_litter
   175)     m_deadstemc_storage_to_litter  => pcf%m_deadstemc_storage_to_litter
   176)     m_deadstemc_to_litter          => pcf%m_deadstemc_to_litter
   177)     m_deadstemc_xfer_to_litter     => pcf%m_deadstemc_xfer_to_litter
   178)     m_frootc_storage_to_litter     => pcf%m_frootc_storage_to_litter
   179)     m_frootc_to_litter             => pcf%m_frootc_to_litter
   180)     m_frootc_xfer_to_litter        => pcf%m_frootc_xfer_to_litter
   181)     m_gresp_storage_to_litter      => pcf%m_gresp_storage_to_litter
   182)     m_gresp_xfer_to_litter         => pcf%m_gresp_xfer_to_litter
   183)     m_leafc_storage_to_litter      => pcf%m_leafc_storage_to_litter
   184)     m_leafc_to_litter              => pcf%m_leafc_to_litter
   185)     m_leafc_xfer_to_litter         => pcf%m_leafc_xfer_to_litter
   186)     m_livecrootc_storage_to_litter => pcf%m_livecrootc_storage_to_litter
   187)     m_livecrootc_to_litter         => pcf%m_livecrootc_to_litter
   188)     m_livecrootc_xfer_to_litter    => pcf%m_livecrootc_xfer_to_litter
   189)     m_livestemc_storage_to_litter  => pcf%m_livestemc_storage_to_litter
   190)     m_livestemc_to_litter          => pcf%m_livestemc_to_litter
   191)     m_livestemc_xfer_to_litter     => pcf%m_livestemc_xfer_to_litter
   192)     deadcrootc                     => pcs%deadcrootc
   193)     deadcrootc_storage             => pcs%deadcrootc_storage
   194)     deadcrootc_xfer                => pcs%deadcrootc_xfer
   195)     deadstemc                      => pcs%deadstemc
   196)     deadstemc_storage              => pcs%deadstemc_storage
   197)     deadstemc_xfer                 => pcs%deadstemc_xfer
   198)     frootc                         => pcs%frootc
   199)     frootc_storage                 => pcs%frootc_storage
   200)     frootc_xfer                    => pcs%frootc_xfer
   201)     gresp_storage                  => pcs%gresp_storage
   202)     gresp_xfer                     => pcs%gresp_xfer
   203)     leafc                          => pcs%leafc
   204)     leafc_storage                  => pcs%leafc_storage
   205)     leafc_xfer                     => pcs%leafc_xfer
   206)     livecrootc                     => pcs%livecrootc
   207)     livecrootc_storage             => pcs%livecrootc_storage
   208)     livecrootc_xfer                => pcs%livecrootc_xfer
   209)     livestemc                      => pcs%livestemc
   210)     livestemc_storage              => pcs%livestemc_storage
   211)     livestemc_xfer                 => pcs%livestemc_xfer
   212) 
   213)     ! set time steps
   214)     dt = real( get_step_size(), r8 )
   215) 
   216)     ! column loop
   217)     do fc = 1,num_soilc
   218)        c = filter_soilc(fc)
   219) 
   220)        ! column level carbon fluxes from gap-phase mortality
   221) 
   222)        ! leaf to litter
   223)        litr1c(c) = litr1c(c) + m_leafc_to_litr1c(c) * dt
   224)        litr2c(c) = litr2c(c) + m_leafc_to_litr2c(c) * dt
   225)        litr3c(c) = litr3c(c) + m_leafc_to_litr3c(c) * dt
   226) 
   227)        ! fine root to litter
   228)        litr1c(c) = litr1c(c) + m_frootc_to_litr1c(c) * dt
   229)        litr2c(c) = litr2c(c) + m_frootc_to_litr2c(c) * dt
   230)        litr3c(c) = litr3c(c) + m_frootc_to_litr3c(c) * dt
   231) 
   232)        ! wood to CWD
   233)        cwdc(c) = cwdc(c) + m_livestemc_to_cwdc(c)  * dt
   234)        cwdc(c) = cwdc(c) + m_deadstemc_to_cwdc(c)  * dt
   235)        cwdc(c) = cwdc(c) + m_livecrootc_to_cwdc(c) * dt
   236)        cwdc(c) = cwdc(c) + m_deadcrootc_to_cwdc(c) * dt
   237) 
   238)        ! storage pools to litter
   239)        litr1c(c) = litr1c(c) + m_leafc_storage_to_litr1c(c)      * dt
   240)        litr1c(c) = litr1c(c) + m_frootc_storage_to_litr1c(c)     * dt
   241)        litr1c(c) = litr1c(c) + m_livestemc_storage_to_litr1c(c)  * dt
   242)        litr1c(c) = litr1c(c) + m_deadstemc_storage_to_litr1c(c)  * dt
   243)        litr1c(c) = litr1c(c) + m_livecrootc_storage_to_litr1c(c) * dt
   244)        litr1c(c) = litr1c(c) + m_deadcrootc_storage_to_litr1c(c) * dt
   245)        litr1c(c) = litr1c(c) + m_gresp_storage_to_litr1c(c)      * dt
   246) 
   247)        ! transfer pools to litter
   248)        litr1c(c) = litr1c(c) + m_leafc_xfer_to_litr1c(c)      * dt
   249)        litr1c(c) = litr1c(c) + m_frootc_xfer_to_litr1c(c)     * dt
   250)        litr1c(c) = litr1c(c) + m_livestemc_xfer_to_litr1c(c)  * dt
   251)        litr1c(c) = litr1c(c) + m_deadstemc_xfer_to_litr1c(c)  * dt
   252)        litr1c(c) = litr1c(c) + m_livecrootc_xfer_to_litr1c(c) * dt
   253)        litr1c(c) = litr1c(c) + m_deadcrootc_xfer_to_litr1c(c) * dt
   254)        litr1c(c) = litr1c(c) + m_gresp_xfer_to_litr1c(c)      * dt
   255) 
   256)     end do ! end of columns loop
   257) 
   258)     ! pft loop
   259)     do fp = 1,num_soilp
   260)        p = filter_soilp(fp)
   261) 
   262)        ! pft-level carbon fluxes from gap-phase mortality
   263)        ! displayed pools
   264)        leafc(p)               = leafc(p)              - m_leafc_to_litter(p)              * dt
   265)        frootc(p)              = frootc(p)             - m_frootc_to_litter(p)             * dt
   266)        livestemc(p)           = livestemc(p)          - m_livestemc_to_litter(p)          * dt
   267)        deadstemc(p)           = deadstemc(p)          - m_deadstemc_to_litter(p)          * dt
   268)        livecrootc(p)          = livecrootc(p)         - m_livecrootc_to_litter(p)         * dt
   269)        deadcrootc(p)          = deadcrootc(p)         - m_deadcrootc_to_litter(p)         * dt
   270) 
   271)        ! storage pools
   272)        leafc_storage(p)       = leafc_storage(p)      - m_leafc_storage_to_litter(p)      * dt
   273)        frootc_storage(p)      = frootc_storage(p)     - m_frootc_storage_to_litter(p)     * dt
   274)        livestemc_storage(p)   = livestemc_storage(p)  - m_livestemc_storage_to_litter(p)  * dt
   275)        deadstemc_storage(p)   = deadstemc_storage(p)  - m_deadstemc_storage_to_litter(p)  * dt
   276)        livecrootc_storage(p)  = livecrootc_storage(p) - m_livecrootc_storage_to_litter(p) * dt
   277)        deadcrootc_storage(p)  = deadcrootc_storage(p) - m_deadcrootc_storage_to_litter(p) * dt
   278)        gresp_storage(p)       = gresp_storage(p)      - m_gresp_storage_to_litter(p)      * dt
   279) 
   280)        ! transfer pools
   281)        leafc_xfer(p)          = leafc_xfer(p)         - m_leafc_xfer_to_litter(p)         * dt
   282)        frootc_xfer(p)         = frootc_xfer(p)        - m_frootc_xfer_to_litter(p)        * dt
   283)        livestemc_xfer(p)      = livestemc_xfer(p)     - m_livestemc_xfer_to_litter(p)     * dt
   284)        deadstemc_xfer(p)      = deadstemc_xfer(p)     - m_deadstemc_xfer_to_litter(p)     * dt
   285)        livecrootc_xfer(p)     = livecrootc_xfer(p)    - m_livecrootc_xfer_to_litter(p)    * dt
   286)        deadcrootc_xfer(p)     = deadcrootc_xfer(p)    - m_deadcrootc_xfer_to_litter(p)    * dt
   287)        gresp_xfer(p)          = gresp_xfer(p)         - m_gresp_xfer_to_litter(p)         * dt
   288) 
   289)     end do ! end of pft loop
   290) 
   291) end subroutine CStateUpdate2
