     6) ! !MODULE: DUSTMod
     7) !
     8) ! !DESCRIPTION: 
     9) ! Routines in this module calculate Dust mobilization and dry deposition for dust.
    10) ! Simulates dust mobilization due to wind from the surface into the 
    11) ! lowest atmospheric layer. On output flx_mss_vrt_dst(ndst) is the surface dust 
    12) ! emission (kg/m**2/s) [ + = to atm].
    13) ! Calculates the turbulent component of dust dry deposition, (the turbulent deposition 
    14) ! velocity through the lowest atmospheric layer). CAM will calculate the settling 
    15) ! velocity through the whole atmospheric column. The two calculations will determine 
    16) ! the dust dry deposition flux to the surface.
    17) !                              
    18) ! !USES:
    19)   use shr_kind_mod, only: r8 => shr_kind_r8 
    20)   use clmtype
    21)   use clm_varpar  , only : dst_src_nbr, ndst, sz_nbr
    22)   use clm_varcon  , only : grav, istsoil
    23)   use clm_varcon  , only : istcrop, istice_mec
    24)   use clm_varctl  , only : iulog
    25)   use abortutils  , only : endrun
    26)   use subgridAveMod, only: p2l_1d
    27)   use clm_varcon, only: spval

    31)   save
    32) !

    34) !

    36)   public DustEmission   ! Dust mobilization 
    37)   public DustDryDep     ! Turbulent dry deposition for dust
    38) !

    48)   real(r8) ovr_src_snk_mss(dst_src_nbr,ndst)  
    49)   real(r8) tmp1          !Factor in saltation computation (named as in Charlie's code)
    50)   real(r8) dmt_vwr(ndst) ![m] Mass-weighted mean diameter resolved
    51)   real(r8) stk_crc(ndst) ![frc] Correction to Stokes settling velocity
    52)   real(r8) dns_aer       ![kg m-3] Aerosol density

    60) ! !IROUTINE: DustEmission
    61) !
    62) ! !INTERFACE:
    63)   subroutine DustEmission (lbp, ubp, lbc,ubc,lbl,ubl,num_nolakep, filter_nolakep)
    64) !
    65) ! !DESCRIPTION: 
    66) ! Dust mobilization. This code simulates dust mobilization due to wind
    67) ! from the surface into the lowest atmospheric layer
    68) ! On output flx_mss_vrt_dst(ndst) is the surface dust emission 
    69) ! (kg/m**2/s) [ + = to atm]
    70) ! Source: C. Zender's dust model
    71) !
    72) ! !USES
    73)    use clm_atmlnd   , only : clm_a2l
    74)    use shr_const_mod, only : SHR_CONST_RHOFW
    75) !
    76) ! !ARGUMENTS:
    77)     implicit none
    78)     integer, intent(in) :: lbp, ubp,lbc,ubc,ubl,lbl                    ! pft bounds
    79)     integer, intent(in) :: num_nolakep                 ! number of column non-lake points in pft filter
    80)     integer, intent(in) :: filter_nolakep(num_nolakep) ! pft filter for non-lake points
    81) !
    82) ! !LOCAL VARIABLES
    83) !
    84) ! local pointers to implicit in arguments
    85) !
    86)     integer , pointer :: pcolumn(:)         ! pft's column index
    87)     integer , pointer :: plandunit(:)       ! pft's landunit index
    88)     integer , pointer :: pgridcell(:)       ! pft's gridcell index
    89)     integer , pointer :: ityplun(:)         ! landunit type
    90)     real(r8), pointer :: tlai(:)            ! one-sided leaf area index, no burying by snow
    91)     real(r8), pointer :: tsai(:)            ! one-sided stem area index, no burying by snow
    92)     real(r8), pointer :: frac_sno(:)        ! fraction of ground covered by snow (0 to 1)
    93)     real(r8), pointer :: gwc_thr(:)         ! threshold gravimetric soil moisture based on clay content
    94)     real(r8), pointer :: forc_rho(:)        ! density (kg/m**3)
    95)     real(r8), pointer :: fv(:)              ! friction velocity (m/s) (for dust model)
    96)     real(r8), pointer :: u10(:)             ! 10-m wind (m/s) (created for dust model)
    97)     real(r8), pointer :: mbl_bsn_fct(:)     ! basin factor
    98)     real(r8), pointer :: mss_frc_cly_vld(:) ! [frc] Mass fraction clay limited to 0.20
    99)     real(r8), pointer :: h2osoi_vol(:,:)    ! volumetric soil water (0<=h2osoi_vol<=watsat)
   100)     real(r8), pointer :: h2osoi_liq(:,:)    ! liquid soil water (kg/m2)
   101)     real(r8), pointer :: h2osoi_ice(:,:)    ! frozen soil water (kg/m2)
   102)     real(r8), pointer :: watsat(:,:)        ! saturated volumetric soil water
   103) 
   104) ! local pointers to implicit out arguments
   105) !
   106)     real(r8), pointer :: flx_mss_vrt_dst(:,:)   ! surface dust emission (kg/m**2/s) 
   107)     real(r8), pointer :: flx_mss_vrt_dst_tot(:) ! total dust flux into atmosphere 
   108) 
   109) ! !REVISION HISTORY
   110) ! Created by Sam Levis
   111) ! Migrated to new data structures by Peter Thornton and Mariana Vertenstein
   112) ! !Created by Peter Thornton and Mariana Vertenstein
   113) !
   114) !
   115) ! !OTHER LOCAL VARIABLES:
   116) !EOP
   117) !
   118)     integer  :: fp,p,c,l,g,m,n      ! indices
   119)     real(r8) :: liqfrac             ! fraction of total water that is liquid
   120)     real(r8) :: wnd_frc_rat         ! [frc] Wind friction threshold over wind friction
   121)     real(r8) :: wnd_frc_slt_dlt     ! [m s-1] Friction velocity increase from saltatn
   122)     real(r8) :: wnd_rfr_dlt         ! [m s-1] Reference windspeed excess over threshld
   123)     real(r8) :: dst_slt_flx_rat_ttl
   124)     real(r8) :: flx_mss_hrz_slt_ttl
   125)     real(r8) :: flx_mss_vrt_dst_ttl(lbp:ubp)
   126)     real(r8) :: frc_thr_wet_fct
   127)     real(r8) :: frc_thr_rgh_fct
   128)     real(r8) :: wnd_frc_thr_slt
   129)     real(r8) :: wnd_rfr_thr_slt
   130)     real(r8) :: wnd_frc_slt
   131)     real(r8) :: lnd_frc_mbl(lbp:ubp)
   132)     real(r8) :: bd
   133)     real(r8) :: gwc_sfc
   134)     real(r8) :: ttlai(lbp:ubp)
   135)     real(r8) :: tlai_lu(lbl:ubl)
   136) !    
   137) ! constants
   138) !
   139)     real(r8), parameter :: cst_slt = 2.61_r8           ! [frc] Saltation constant
   140)     real(r8), parameter :: flx_mss_fdg_fct = 5.0e-4_r8 ! [frc] Empir. mass flx tuning eflx_lh_vegt
   141)     real(r8), parameter :: vai_mbl_thr = 0.3_r8        ! [m2 m-2] VAI threshold quenching dust mobilization
   142)     real(r8), pointer :: wtlunit(:)         ! weight of pft relative to landunit
   143)     real(r8) :: sumwt(lbl:ubl)              ! sum of weights
   144)     logical  :: found                       ! temporary for error check
   145)     integer  :: index
   146) 
   147) !------------------------------------------------------------------------
   148) 
   149)     ! Assign local pointers to derived type scalar members (gridcell-level)
   150) 
   151)     forc_rho        => clm_a2l%forc_rho
   152) 
   153)     ! Assign local pointers to derived type scalar members (landunit-level)
   154) 
   155)     ityplun         => lun%itype
   156) 
   157)     ! Assign local pointers to derived type scalar members (column-level)
   158) 
   159)     frac_sno        => cps%frac_sno
   160)     gwc_thr         => cps%gwc_thr
   161)     mbl_bsn_fct     => cps%mbl_bsn_fct
   162)     mss_frc_cly_vld => cps%mss_frc_cly_vld
   163)     h2osoi_vol      => cws%h2osoi_vol
   164)     h2osoi_liq      => cws%h2osoi_liq
   165)     h2osoi_ice      => cws%h2osoi_ice
   166)     watsat          => cps%watsat
   167) 
   168)     ! Assign local pointers to derived type scalar members (pft-level)
   169) 
   170)     pgridcell       => pft%gridcell
   171)     plandunit       => pft%landunit
   172)     pcolumn         => pft%column
   173)     tlai            => pps%tlai
   174)     tsai            => pps%tsai
   175)     fv              => pps%fv
   176)     u10             => pps%u10
   177)     flx_mss_vrt_dst => pdf%flx_mss_vrt_dst
   178)     flx_mss_vrt_dst_tot => pdf%flx_mss_vrt_dst_tot
   179)    !local pointers from subgridAveMod/p2l_1d
   180)     wtlunit         => pft%wtlunit
   181) 
   182)     ttlai(:) = 0._r8
   183) ! make lai average at landunit level
   184)     do fp = 1,num_nolakep
   185)        p = filter_nolakep(fp)
   186)        ttlai(p) = tlai(p)+tsai(p)
   187)     enddo
   188) 
   189)     tlai_lu(:) = spval
   190)     sumwt(:) = 0._r8
   191)     do p = lbp,ubp
   192)        if (ttlai(p) /= spval .and. wtlunit(p) /= 0._r8) then
   193)           c = pcolumn(p)
   194)           l = plandunit(p)
   195)           if (sumwt(l) == 0._r8) tlai_lu(l) = 0._r8
   196)           tlai_lu(l) = tlai_lu(l) + ttlai(p) * wtlunit(p)
   197)           sumwt(l) = sumwt(l) + wtlunit(p)
   198)        end if
   199)     end do
   200)     found = .false.
   201)     do l = lbl,ubl
   202)        if (sumwt(l) > 1.0_r8 + 1.e-6_r8) then
   203)           found = .true.
   204)           index = l
   205)           exit
   206)        else if (sumwt(l) /= 0._r8) then
   207)           tlai_lu(l) = tlai_lu(l)/sumwt(l)
   208)        end if
   209)     end do
   210)     if (found) then
   211)        write(iulog,*) 'p2l_1d error: sumwt is greater than 1.0 at l= ',index
   212)        call endrun()
   213)     end if
   214) 
   215) ! Loop through pfts
   216) 
   217) ! initialize variables which get passed to the atmosphere
   218)     flx_mss_vrt_dst(lbp:ubp,:)=0._r8
   219) 
   220)     do fp = 1,num_nolakep
   221)        p = filter_nolakep(fp)
   222)        c = pcolumn(p)
   223)        l = plandunit(p)
   224)        
   225)        ! the following code from subr. lnd_frc_mbl_get was adapted for lsm use
   226)        ! purpose: return fraction of each gridcell suitable for dust mobilization
   227)        
   228)        ! the "bare ground" fraction of the current sub-gridscale cell decreases
   229)        ! linearly from 1 to 0 as VAI(=tlai+tsai) increases from 0 to vai_mbl_thr
   230)        ! if ice sheet, wetland, or lake, no dust allowed
   231)        
   232)        if (ityplun(l) == istsoil .or. ityplun(l) == istcrop) then
   233)           if (tlai_lu(l) < vai_mbl_thr) then
   234)              lnd_frc_mbl(p) = 1.0_r8 - (tlai_lu(l))/vai_mbl_thr
   235)           else
   236)              lnd_frc_mbl(p) = 0.0_r8
   237)           endif
   238)           lnd_frc_mbl(p) = lnd_frc_mbl(p) * (1.0_r8 - frac_sno(c))
   239)        else          
   240)           lnd_frc_mbl(p) = 0.0_r8   
   241)        end if
   242)     end do
   243)     
   244)     do fp = 1,num_nolakep
   245)        p = filter_nolakep(fp)
   246)        if (lnd_frc_mbl(p)>1.0_r8 .or. lnd_frc_mbl(p)<0.0_r8) then
   247)           write(iulog,*)'Error dstmbl: pft= ',p,' lnd_frc_mbl(p)= ',lnd_frc_mbl(p)
   248)           call endrun 
   249)        end if
   250)     end do
   251)     
   252)     ! reset history output variables before next if-statement to avoid output = inf
   253)     
   254)     do fp = 1,num_nolakep
   255)        p = filter_nolakep(fp)
   256)        flx_mss_vrt_dst_tot(p) = 0.0_r8
   257)     end do
   258)     do n = 1, ndst
   259)        do fp = 1,num_nolakep
   260)           p = filter_nolakep(fp)
   261)           flx_mss_vrt_dst(p,n) = 0.0_r8
   262)        end do
   263)     end do
   264)     
   265)     do fp = 1,num_nolakep
   266)        p = filter_nolakep(fp)
   267)        c = pcolumn(p)
   268)        l = plandunit(p)
   269)        g = pgridcell(p)
   270)        
   271)        ! only perform the following calculations if lnd_frc_mbl is non-zero 
   272)        
   273)        if (lnd_frc_mbl(p) > 0.0_r8) then
   274) 
   275)           ! the following comes from subr. frc_thr_rgh_fct_get
   276)           ! purpose: compute factor by which surface roughness increases threshold
   277)           !          friction velocity (currently a constant)
   278)           
   279)           frc_thr_rgh_fct = 1.0_r8
   280)           
   281)           ! the following comes from subr. frc_thr_wet_fct_get
   282)           ! purpose: compute factor by which soil moisture increases threshold friction velocity
   283)           ! adjust threshold velocity for inhibition by moisture
   284)           ! modified 4/5/2002 (slevis) to use gravimetric instead of volumetric
   285)           ! water content
   286)           
   287)           bd = (1._r8-watsat(c,1))*2.7e3_r8      ![kg m-3] Bulk density of dry surface soil
   288)           gwc_sfc = h2osoi_vol(c,1)*SHR_CONST_RHOFW/bd    ![kg kg-1] Gravimetric H2O cont
   289)           if (gwc_sfc > gwc_thr(c)) then
   290)              frc_thr_wet_fct = sqrt(1.0_r8 + 1.21_r8 * (100.0_r8*(gwc_sfc - gwc_thr(c)))**0.68_r8)
   291)           else
   292)              frc_thr_wet_fct = 1.0_r8
   293)           end if
   294)           
   295)           ! slevis: adding liqfrac here, because related to effects from soil water
   296) 
   297)           liqfrac = max( 0.0_r8, min( 1.0_r8, h2osoi_liq(c,1) / (h2osoi_ice(c,1)+h2osoi_liq(c,1)+1.0e-6_r8) ) )
   298)           
   299)           ! the following lines come from subr. dst_mbl
   300)           ! purpose: adjust threshold friction velocity to acct for moisture and
   301)           !          roughness. The ratio tmp1 / sqrt(forc_rho) comes from
   302)           !          subr. wnd_frc_thr_slt_get which computes dry threshold
   303)           !          friction velocity for saltation
   304)           
   305)           wnd_frc_thr_slt = tmp1 / sqrt(forc_rho(g)) * frc_thr_wet_fct * frc_thr_rgh_fct
   306)           
   307)           ! reset these variables which will be updated in the following if-block
   308)           
   309)           wnd_frc_slt = fv(p)
   310)           flx_mss_hrz_slt_ttl = 0.0_r8
   311)           flx_mss_vrt_dst_ttl(p) = 0.0_r8
   312)           
   313)           ! the following line comes from subr. dst_mbl
   314)           ! purpose: threshold saltation wind speed
   315)           
   316)           wnd_rfr_thr_slt = u10(p) * wnd_frc_thr_slt / fv(p)
   317)           
   318)           ! the following if-block comes from subr. wnd_frc_slt_get 
   319)           ! purpose: compute the saltating friction velocity
   320)           ! theory: saltation roughens the boundary layer, AKA "Owen's effect"
   321)           
   322)           if (u10(p) >= wnd_rfr_thr_slt) then
   323)              wnd_rfr_dlt = u10(p) - wnd_rfr_thr_slt
   324)              wnd_frc_slt_dlt = 0.003_r8 * wnd_rfr_dlt * wnd_rfr_dlt
   325)              wnd_frc_slt = fv(p) + wnd_frc_slt_dlt
   326)           end if
   327)           
   328)           ! the following comes from subr. flx_mss_hrz_slt_ttl_Whi79_get
   329)           ! purpose: compute vertically integrated streamwise mass flux of particles
   330)           
   331)           if (wnd_frc_slt > wnd_frc_thr_slt) then
   332)              wnd_frc_rat = wnd_frc_thr_slt / wnd_frc_slt
   333)              flx_mss_hrz_slt_ttl = cst_slt * forc_rho(g) * (wnd_frc_slt**3.0_r8) * &
   334)                   (1.0_r8 - wnd_frc_rat) * (1.0_r8 + wnd_frc_rat) * (1.0_r8 + wnd_frc_rat) / grav
   335)              
   336)              ! the following loop originates from subr. dst_mbl
   337)              ! purpose: apply land sfc and veg limitations and global tuning factor
   338)              ! slevis: multiply flx_mss_hrz_slt_ttl by liqfrac to incude the effect 
   339)              ! of frozen soil
   340)              
   341)              flx_mss_hrz_slt_ttl = flx_mss_hrz_slt_ttl * lnd_frc_mbl(p) * mbl_bsn_fct(c) * &
   342)                   flx_mss_fdg_fct * liqfrac
   343)           end if
   344)           
   345)           ! the following comes from subr. flx_mss_vrt_dst_ttl_MaB95_get
   346)           ! purpose: diagnose total vertical mass flux of dust from vertically
   347)           !          integrated streamwise mass flux
   348)           
   349)           dst_slt_flx_rat_ttl = 100.0_r8 * exp( log(10.0_r8) * (13.4_r8 * mss_frc_cly_vld(c) - 6.0_r8) )
   350)           flx_mss_vrt_dst_ttl(p) = flx_mss_hrz_slt_ttl * dst_slt_flx_rat_ttl
   351)           
   352)        end if   ! lnd_frc_mbl > 0.0
   353) 
   354)     end do
   355) 
   356)     ! the following comes from subr. flx_mss_vrt_dst_prt in C. Zender's code
   357)     ! purpose: partition total vertical mass flux of dust into transport bins
   358)           
   359)     do n = 1, ndst
   360)        do m = 1, dst_src_nbr
   361)           do fp = 1,num_nolakep
   362)              p = filter_nolakep(fp)
   363)              if (lnd_frc_mbl(p) > 0.0_r8) then
   364)                 flx_mss_vrt_dst(p,n) = flx_mss_vrt_dst(p,n) +  ovr_src_snk_mss(m,n) * flx_mss_vrt_dst_ttl(p)
   365)              end if
   366)           end do
   367)        end do
   368)     end do
   369) 
   370)     do n = 1, ndst
   371)        do fp = 1,num_nolakep
   372)           p = filter_nolakep(fp)
   373)           if (lnd_frc_mbl(p) > 0.0_r8) then
   374)              flx_mss_vrt_dst_tot(p) = flx_mss_vrt_dst_tot(p) + flx_mss_vrt_dst(p,n)
   375)           end if
   376)        end do
   377)     end do
   378) 
   379)   end subroutine DustEmission

   384) ! !IROUTINE: subroutine DustDryDep
   385) !
   386) ! !INTERFACE:
   387) !
   388)   subroutine DustDryDep (lbp, ubp)
   389) !
   390) ! !DESCRIPTION: 
   391) !
   392) ! Determine Turbulent dry deposition for dust. Calculate the turbulent 
   393) ! component of dust dry deposition, (the turbulent deposition velocity 
   394) ! through the lowest atmospheric layer. CAM will calculate the settling 
   395) ! velocity through the whole atmospheric column. The two calculations 
   396) ! will determine the dust dry deposition flux to the surface.
   397) ! Note: Same process should occur over oceans. For the coupled CESM,
   398) ! we may find it more efficient to let CAM calculate the turbulent dep
   399) ! velocity over all surfaces. This would require passing the
   400) ! aerodynamic resistance, ram(1), and the friction velocity, fv, from
   401) ! the land to the atmosphere component. In that case, dustini need not
   402) ! calculate particle diamter (dmt_vwr) and particle density (dns_aer).
   403) ! Source: C. Zender's dry deposition code
   404) !
   405) ! !USES
   406) !
   407)     use shr_const_mod, only : SHR_CONST_PI, SHR_CONST_RDAIR, SHR_CONST_BOLTZ
   408)     use clm_atmlnd   , only : clm_a2l
   409) !
   410) ! !ARGUMENTS:
   411) !
   412)     implicit none
   413)     integer, intent(in) :: lbp, ubp     ! pft bounds
   414) !
   415) ! !LOCAL VARIABLES
   416) !
   417) ! local pointers to implicit in arguments
   418) !
   419)     integer , pointer :: plandunit(:)   ! pft's landunit index
   420)     integer , pointer :: pgridcell(:)   ! pft's gridcell index
   421)     integer , pointer :: ityplun(:)     ! landunit type
   422)     real(r8), pointer :: pwtgcell(:)    ! weight of pft relative to corresponding gridcell
   423)     real(r8), pointer :: forc_t(:)      ! atm temperature (K)
   424)     real(r8), pointer :: forc_pbot(:)   ! atm pressure (Pa)
   425)     real(r8), pointer :: forc_rho(:)    ! atm density (kg/m**3)
   426)     real(r8), pointer :: fv(:)          ! friction velocity (m/s)
   427)     real(r8), pointer :: ram1(:)        ! aerodynamical resistance (s/m)
   428)     real(r8), pointer :: vlc_trb(:,:)   ! Turbulent deposn velocity (m/s)
   429)     real(r8), pointer :: vlc_trb_1(:)   ! Turbulent deposition velocity 1
   430)     real(r8), pointer :: vlc_trb_2(:)   ! Turbulent deposition velocity 2
   431)     real(r8), pointer :: vlc_trb_3(:)   ! Turbulent deposition velocity 3
   432)     real(r8), pointer :: vlc_trb_4(:)   ! Turbulent deposition velocity 4
   433) !
   434) ! !REVISION HISTORY
   435) ! Created by Sam Levis
   436) !
   437) !
   438) ! !LOCAL VARIABLES
   439) !EOP
   440) !
   441)     integer  :: p,l,g,m,n             ! indices
   442)     real(r8) :: vsc_dyn_atm(lbp:ubp)  ! [kg m-1 s-1] Dynamic viscosity of air
   443)     real(r8) :: vsc_knm_atm(lbp:ubp)  ! [m2 s-1] Kinematic viscosity of atmosphere
   444)     real(r8) :: shm_nbr_xpn           ! [frc] Sfc-dep exponent for aerosol-diffusion dependence on Schmidt number
   445)     real(r8) :: shm_nbr               ! [frc] Schmidt number
   446)     real(r8) :: stk_nbr               ! [frc] Stokes number
   447)     real(r8) :: mfp_atm               ! [m] Mean free path of air
   448)     real(r8) :: dff_aer               ! [m2 s-1] Brownian diffusivity of particle

   455) ! constants
   456)

   484)        l = plandunit(p)

   486)        if (pwtgcell(p)>0._r8 .or. ityplun(l)==istice_mec) then
   487)           g = pgridcell(p)

   496)           vsc_dyn_atm(p) = 1.72e-5_r8 * ((forc_t(g)/273.0_r8)**1.5_r8) * 393.0_r8 / &
   497)                (forc_t(g)+120.0_r8)      ![kg m-1 s-1] RoY94 p. 102

   499)                (forc_pbot(g)*sqrt(8.0_r8/(SHR_CONST_PI*SHR_CONST_RDAIR*forc_t(g))))
   500)           vsc_knm_atm(p) = vsc_dyn_atm(p) / forc_rho(g) ![m2 s-1] Kinematic viscosity of air
   501)           
   502)           do m = 1, ndst
   503)              slp_crc(p,m) = 1.0_r8 + 2.0_r8 * mfp_atm * &
   504)                   (1.257_r8+0.4_r8*exp(-1.1_r8*dmt_vwr(m)/(2.0_r8*mfp_atm))) / &
   505)                   dmt_vwr(m)   ![frc] Slip correction factor SeP97 p. 464
   506)              vlc_grv(p,m) = (1.0_r8/18.0_r8) * dmt_vwr(m) * dmt_vwr(m) * dns_aer * &
   507)                   grav * slp_crc(p,m) / vsc_dyn_atm(p)   ![m s-1] Stokes' settling velocity SeP97 p. 466
   508)              vlc_grv(p,m) = vlc_grv(p,m) * stk_crc(m)    ![m s-1] Correction to Stokes settling velocity
   509)           end do
   510)        end if
   511)     end do
   512) 
   513)     do m = 1, ndst
   514)        do p = lbp,ubp
   515)           l = plandunit(p)
   516)           if (pwtgcell(p)>0._r8 .or. ityplun(l)==istice_mec) then
   517)              g = pgridcell(p)
   518)              
   519)              stk_nbr = vlc_grv(p,m) * fv(p) * fv(p) / (grav * vsc_knm_atm(p))  ![frc] SeP97 p.965
   520)              dff_aer = SHR_CONST_BOLTZ * forc_t(g) * slp_crc(p,m) / &          ![m2 s-1]
   521)                   (3.0_r8*SHR_CONST_PI * vsc_dyn_atm(p) * dmt_vwr(m))          !SeP97 p.474
   522)              shm_nbr = vsc_knm_atm(p) / dff_aer                                ![frc] SeP97 p.972
   523)              shm_nbr_xpn = shm_nbr_xpn_lnd                                     ![frc]
   524)              
   525)              ! fxm: Turning this on dramatically reduces
   526)              ! deposition velocity in low wind regimes
   527)              ! Schmidt number exponent is -2/3 over solid surfaces and
   528)              ! -1/2 over liquid surfaces SlS80 p. 1014
   529)              ! if (oro(i)==0.0) shm_nbr_xpn=shm_nbr_xpn_ocn else shm_nbr_xpn=shm_nbr_xpn_lnd
   530)              ! [frc] Surface-dependent exponent for aerosol-diffusion dependence on Schmidt # 
   531)              
   532)              tmp = shm_nbr**shm_nbr_xpn + 10.0_r8**(-3.0_r8/stk_nbr)
   533)              rss_lmn(p,m) = 1.0_r8 / (tmp * fv(p)) ![s m-1] SeP97 p.972,965
   534)           end if
   535)        end do
   536)     end do
   537) 
   538)     ! Lowest layer: Turbulent deposition (CAM will calc. gravitational dep)
   539) 
   540)     do m = 1, ndst
   541)        do p = lbp,ubp
   542)           l = plandunit(p)
   543)           if (pwtgcell(p)>0._r8 .or. ityplun(l)==istice_mec) then
   544)              rss_trb = ram1(p) + rss_lmn(p,m) + ram1(p) * rss_lmn(p,m) * vlc_grv(p,m) ![s m-1]
   545)              vlc_trb(p,m) = 1.0_r8 / rss_trb                                          ![m s-1]
   546)           end if
   547)        end do
   548)     end do
   549) 
   550)     do p = lbp,ubp
   551)        l = plandunit(p)
   552)        if (pwtgcell(p)>0._r8 .or. ityplun(l)==istice_mec) then
   553)           vlc_trb_1(p) = vlc_trb(p,1)
   554)           vlc_trb_2(p) = vlc_trb(p,2)
   555)           vlc_trb_3(p) = vlc_trb(p,3)
   556)           vlc_trb_4(p) = vlc_trb(p,4)
   557)        end if
   558)     end do 
   559) 
   560)   end subroutine DustDryDep

   681)                                    erf(lndminjovrdmdni/sqrt2lngsdi))

   774)

   777)           sz_ctr(m) = 0.5_r8 * (sz_min(m)+sz_max(m))

   779)        end do
   780) 
   781)        lngsdsqrttwopi_rcp = 1.0_r8 / (ln_gsd*sqrt(2.0_r8*SHR_CONST_PI))

   783)        vlm_rsl(n) = 0.0_r8 ! [m3 m-3] Volume concentration resolved
   784) 
   785)        do m = 1, sz_nbr
   786) 
   787)           ! Evaluate lognormal distribution for these sizes (call lgn_evl)
   788)           tmp = log(sz_ctr(m)/dmt_nma) / ln_gsd
   789)           lgn_dst = lngsdsqrttwopi_rcp * exp(-0.5_r8*tmp*tmp) / sz_ctr(m)
   790) 
   791)           ! Integrate moments of size distribution
   792)           dmt_vwr(n) = dmt_vwr(n) + sz_ctr(m) *                    &
   793)                SHR_CONST_PI / 6.0_r8 * (sz_ctr(m)**3.0_r8) * & ![m3] Volume
   794)                lgn_dst * sz_dlt(m)                ![# m-3] Number concentrn
   795)           vlm_rsl(n) = vlm_rsl(n) +                                &
   796)                SHR_CONST_PI / 6.0_r8 * (sz_ctr(m)**3.0_r8) * & ![m3] Volume
   797)                lgn_dst * sz_dlt(m)                ![# m-3] Number concentrn
   798) 
   799)        end do
   800) 
   801)        dmt_vwr(n) = dmt_vwr(n) / vlm_rsl(n) ![m] Mass weighted diameter resolved
   802) 
   803)     end do
   804) 
   805)     ! calculate correction to Stokes' settling velocity (subroutine stk_crc_get)
   806) 
   807)     eps_max = 1.0e-4_r8
   808)     dns_mdp = 100000._r8 / (295.0_r8*SHR_CONST_RDAIR) ![kg m-3] const prs_mdp & tpt_vrt
   809) 
   810)     ! Size-independent thermokinetic properties
   811) 
   812)     vsc_dyn_atm = 1.72e-5_r8 * ((295.0_r8/273.0_r8)**1.5_r8) * 393.0_r8 / &
   813)          (295.0_r8+120.0_r8)      ![kg m-1 s-1] RoY94 p.102 tpt_mdp=295.0
   814)     mfp_atm = 2.0_r8 * vsc_dyn_atm / &  !SeP97 p. 455 constant prs_mdp, tpt_mdp
   815)          (100000._r8*sqrt(8.0_r8/(SHR_CONST_PI*SHR_CONST_RDAIR*295.0_r8)))
   816)     vsc_knm_atm = vsc_dyn_atm / dns_mdp ![m2 s-1] Kinematic viscosity of air
   817) 
   818)     do m = 1, ndst
   819)        slp_crc(m) = 1.0_r8 + 2.0_r8 * mfp_atm *                      &
