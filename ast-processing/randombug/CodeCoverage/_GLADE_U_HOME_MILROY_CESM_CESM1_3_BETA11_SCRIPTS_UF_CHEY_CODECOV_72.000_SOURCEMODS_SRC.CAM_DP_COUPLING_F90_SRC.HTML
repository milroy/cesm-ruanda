    59)     integer (kind=int_kind)  :: lchnk, icol, ilyr      ! indices over chunks, columns, layers

    69)     real (kind=real_kind), allocatable :: frontga(:,:,:) !   function (frontgf) and angle (frontga)

   126)        uv_tmp(:,:,:,:) = 0._r8

   129)           frontgf(:,:,:) = 0._r8
   130)           frontga(:,:,:) = 0._r8
   131)        end if
   132)        phis_tmp(:,:) = 0._r8
   133)        Q_tmp(:,:,:,:) = 0._r8
   134)     endif !! iam .lt. par%nprocs
   135) 
   136)     call t_startf('dpcopy')
   137)     if (local_dp_map) then
   138)

   140)        do lchnk=begchunk,endchunk
   141)           ncols=get_ncols_p(lchnk)
   142)           call get_gcol_all_p(lchnk,pcols,pgcols)
   143) 
   144)           pbuf_chnk => pbuf_get_chunk(pbuf2d, lchnk)
   145) 
   146)           if (use_gw_front) then
   147)              call pbuf_get_field(pbuf_chnk, frontgf_idx, pbuf_frontgf)
   148)              call pbuf_get_field(pbuf_chnk, frontga_idx, pbuf_frontga)
   149)           end if
   150) 
   151)           do icol=1,ncols
   152)              call get_gcol_block_d(pgcols(icol),1,idmb1,idmb2,idmb3)
   153)              ie = idmb3(1)
   154)              ioff=idmb2(1)
   155) 
   156)              phys_state(lchnk)%ps(icol)=ps_tmp(ioff,ie)
   157)              phys_state(lchnk)%phis(icol)=phis_tmp(ioff,ie)
   158)              do ilyr=1,pver
   159)                 phys_state(lchnk)%t(icol,ilyr)=T_tmp(ioff,ilyr,ie)	   
   160)                 phys_state(lchnk)%u(icol,ilyr)=uv_tmp(ioff,1,ilyr,ie)
   161)                 phys_state(lchnk)%v(icol,ilyr)=uv_tmp(ioff,2,ilyr,ie)
   162)                 ! phys_state(lchnk)%omega(icol,ilyr)=omega_tmp(ioff,ilyr,ie)
   163)                 ! BUG HERE
   164)                 phys_state(lchnk)%omega(icol,ilyr)=omega_tmp(ioff,1,ie)
   165) 
   166)                 if (use_gw_front) then
   167)                    pbuf_frontgf(icol,ilyr) = frontgf(ioff,ilyr,ie)
   168)                    pbuf_frontga(icol,ilyr) = frontga(ioff,ilyr,ie)
   169)                 endif
   170)              end do
   171) 
   172)              do m=1,pcnst
   173)                 do ilyr=1,pver
   174)                    phys_state(lchnk)%q(icol,ilyr,m)=Q_tmp(ioff,ilyr,m,ie)
   175)                 end do
   176)              end do
   177)           end do
   178)        end do
   179) 
   180)     else  ! .not. local_dp_map
   181) 
   182)        tsize = 4 + pcnst
   183)        if (use_gw_front) tsize = tsize + 2
   184) 
   185)        allocate(bbuffer(tsize*block_buf_nrecs))
   186)        allocate(cbuffer(tsize*chunk_buf_nrecs))
   187)        if(iam .lt. par%nprocs) then
   188) !$omp parallel do private (ie, bpter, icol, ilyr, m)
   189)           do ie=1,nelemd
   190) 
   191)              call block_to_chunk_send_pters(elem(ie)%GlobalID,npsq,pver+1,tsize,bpter)
   192) 
   193)              do icol=1,elem(ie)%idxP%NumUniquePts
   194)                 
   195)                 bbuffer(bpter(icol,0)+2:bpter(icol,0)+tsize-1) = 0.0_r8
   196)                 
   197)                 bbuffer(bpter(icol,0))   = ps_tmp(icol,ie)
   198)                 bbuffer(bpter(icol,0)+1) = phis_tmp(icol,ie)
   199) 
   200)                 do ilyr=1,pver
   201)                    bbuffer(bpter(icol,ilyr))   = T_tmp(icol,ilyr,ie)
   202)                    bbuffer(bpter(icol,ilyr)+1) = uv_tmp(icol,1,ilyr,ie)
   203)                    bbuffer(bpter(icol,ilyr)+2) = uv_tmp(icol,2,ilyr,ie)
   204)                    bbuffer(bpter(icol,ilyr)+3) = omega_tmp(icol,ilyr,ie)
   205) 
   206)                    if (use_gw_front) then
   207)                       bbuffer(bpter(icol,ilyr)+4) = frontgf(icol,ilyr,ie)
   208)                       bbuffer(bpter(icol,ilyr)+5) = frontga(icol,ilyr,ie)
   209)                    end if
   210) 
   211)                    do m=1,pcnst
   212)                       bbuffer(bpter(icol,ilyr)+tsize-pcnst-1+m) = Q_tmp(icol,ilyr,m,ie)
   213)                    end do
   214)                 end do
   215) 
   216)              end do
   217) 
   218)           end do
   219)        else
   220)           bbuffer(:) = 0._r8
   221)        end if
   222) 
   223)        call t_barrierf ('sync_blk_to_chk', mpicom)
   224)        call t_startf ('block_to_chunk')
   225)        call transpose_block_to_chunk(tsize, bbuffer, cbuffer)
   226)        call t_stopf  ('block_to_chunk')
   227) 
   228) !$omp parallel do private (lchnk, ncols, cpter, icol, ilyr, m, pbuf_chnk, pbuf_frontgf, pbuf_frontga)
   229)        do lchnk = begchunk,endchunk
   230)           ncols = phys_state(lchnk)%ncol
   231) 
   232)           pbuf_chnk => pbuf_get_chunk(pbuf2d, lchnk)
   233) 
   234)           if (use_gw_front) then
   235)              call pbuf_get_field(pbuf_chnk, frontgf_idx, pbuf_frontgf)
   236)              call pbuf_get_field(pbuf_chnk, frontga_idx, pbuf_frontga)
   237)           end if
   238) 
   239)           call block_to_chunk_recv_pters(lchnk,pcols,pver+1,tsize,cpter)
   240) 
   241)           do icol=1,ncols
   242) 
   243)              phys_state(lchnk)%ps   (icol)     = cbuffer(cpter(icol,0))
   244)              phys_state(lchnk)%phis (icol)     = cbuffer(cpter(icol,0)+1)
   245) 
   246)              do ilyr=1,pver
   247) 
   248)                 phys_state(lchnk)%t     (icol,ilyr)   = cbuffer(cpter(icol,ilyr))
   249)                 phys_state(lchnk)%u     (icol,ilyr)   = cbuffer(cpter(icol,ilyr)+1)
   250)                 phys_state(lchnk)%v     (icol,ilyr)   = cbuffer(cpter(icol,ilyr)+2)
   251)                 phys_state(lchnk)%omega (icol,ilyr)   = cbuffer(cpter(icol,ilyr)+3)
   252) 
   253)                 if (use_gw_front) then
   254)                    pbuf_frontgf(icol,ilyr) = cbuffer(cpter(icol,ilyr)+4)
   255)                    pbuf_frontga(icol,ilyr) = cbuffer(cpter(icol,ilyr)+5)
   256)                 endif
   257) 
   258)                 do m=1,pcnst
   259)                    phys_state(lchnk)%q  (icol,ilyr,m) = cbuffer(cpter(icol,ilyr)+tsize-pcnst-1+m)
   260)                 end do
   261) 
   262)              end do
   263) 
   264)           end do
   265) 
   266)        end do
   267) 
   268)        deallocate( bbuffer )
   269)        deallocate( cbuffer )
   270) 
   271)     end if
   272)     call t_stopf('dpcopy')
   273) 
   274)     call t_startf('derived_phys')
   275)     call derived_phys(phys_state,phys_tend,pbuf2d)
   276)     call t_stopf('derived_phys')
   277) 
   278) !$omp parallel do private (lchnk, ncols, ilyr, icol)
   279)     do lchnk=begchunk,endchunk
   280)        ncols=get_ncols_p(lchnk)
   281)        do ilyr=1,pver
   282)           do icol=1,ncols
   283)              phys_state(lchnk)%omega(icol,ilyr)=phys_state(lchnk)%omega(icol,ilyr)*phys_state(lchnk)%pmid(icol,ilyr)
   284)           end do
   285)        end do
   286)     end do
   287) 
   288) 
   289)    if (write_inithist() ) then
   290)       do lchnk=begchunk,endchunk
   291)          call outfld('T&IC',phys_state(lchnk)%t,pcols,lchnk)
   292)          call outfld('U&IC',phys_state(lchnk)%u,pcols,lchnk)
   293)          call outfld('V&IC',phys_state(lchnk)%v,pcols,lchnk)
   294)          call outfld('PS&IC',phys_state(lchnk)%ps,pcols,lchnk)
   295)          do m=1,pcnst
   296)             call outfld(trim(cnst_name(m))//'&IC',phys_state(lchnk)%q(1,1,m), pcols,lchnk)
   297)          end do
   298)       end do
   299)    endif
   300)    
   301)        
   302)   end subroutine d_p_coupling

   304)   subroutine p_d_coupling(phys_state, phys_tend,  dyn_in)
   305)     use shr_vmath_mod, only: shr_vmath_log
   306)     use cam_control_mod, only : adiabatic
   307)     implicit none
   308) 
   309) ! !INPUT PARAMETERS:
   310)     type(physics_state), intent(inout), dimension(begchunk:endchunk) :: phys_state
   311)     type(physics_tend),  intent(inout), dimension(begchunk:endchunk) :: phys_tend
   312)     
   313) 
   314) ! !OUTPUT PARAMETERS:
   315)     type(dyn_import_t),  intent(inout)   :: dyn_in
   316) 
   317)     ! LOCAL VARIABLES
   318)     integer :: ic , ncols                            ! index
   319)     type(element_t), pointer :: elem(:)               ! pointer to dyn_in element array
   320)     integer (kind=int_kind)  :: ie, iep               ! indices over elements
   321)     integer (kind=int_kind)  :: lchnk, icol, ilyr      ! indices over chunks, columns, layers
   322) 
   323)     real (kind=real_kind)    :: T_tmp(npsq,pver,nelemd)       ! temporary array to hold T
   324)     real (kind=real_kind)    :: uv_tmp(npsq,2,pver,nelemd)    ! temporary array to hold uv
   325) !   real (kind=real_kind)    :: omega_tmp(npsq,pver,nelemd)   ! temporary array to hold omega
   326)     real (kind=real_kind)    :: q_tmp(npsq,pver,pcnst,nelemd) ! temporary array to hold q
   327)     integer (kind=int_kind)  :: ioff, m, i, j, k
   328)     integer(kind=int_kind)   :: pgcols(pcols), idmb1(1), idmb2(1), idmb3(1)
   329) 
   330)     integer :: tsize                 ! amount of data per grid point passed to physics
   331)     integer :: cpter(pcols,0:pver)   ! offsets into chunk buffer for packing data
   332)     integer :: bpter(npsq,0:pver)    ! offsets into block buffer for unpacking data
   333) 
   334)     real (kind=real_kind), allocatable, dimension(:) :: bbuffer, cbuffer ! transpose buffers
   335) 
   336)     if (iam .lt. par%nprocs) then
   337)        elem => dyn_in%elem
   338)     else
   339)        nullify(elem)
   340)     end if
   341) 
   342)     T_tmp=0.0_r8
   343)     uv_tmp=0.0_r8
   344)     q_tmp=0.0_r8
   345) 
   346)     if(adiabatic) return
   347) 
   348)     call t_startf('pd_copy')
   349)     if(local_dp_map) then
   350) 
   351) !$omp parallel do private (lchnk, ncols, pgcols, icol, idmb1, idmb2, idmb3, ie, ioff, ilyr, m)
   352)        do lchnk=begchunk,endchunk
   353)           ncols=get_ncols_p(lchnk)
   354)           call get_gcol_all_p(lchnk,pcols,pgcols)
   355) 
   356)           do icol=1,ncols
   357)              call get_gcol_block_d(pgcols(icol),1,idmb1,idmb2,idmb3)
   358)              ie = idmb3(1)
   359)              ioff=idmb2(1)
   360) 
   361)              do ilyr=1,pver
   362)                 T_tmp(ioff,ilyr,ie)      = phys_tend(lchnk)%dtdt(icol,ilyr)
   363)                 uv_tmp(ioff,1,ilyr,ie)   = phys_tend(lchnk)%dudt(icol,ilyr)
   364)                 uv_tmp(ioff,2,ilyr,ie)   = phys_tend(lchnk)%dvdt(icol,ilyr)
   365) 
   366)                 do m=1,pcnst
   367)                    q_tmp(ioff,ilyr,m,ie) = phys_state(lchnk)%q(icol,ilyr,m)
   368)                 end do
   369)              end do
   370) 
   371)        	  end do	
   372) 
   373)        end do
   374) 
   375)     else
   376) 
   377)        tsize = 3 + pcnst
   378) 
   379)        allocate( bbuffer(tsize*block_buf_nrecs) )
   380)        allocate( cbuffer(tsize*chunk_buf_nrecs) )
   381) 
   382) !$omp parallel do private (lchnk, ncols, cpter, i, icol, ilyr, m)
   383)        do lchnk = begchunk,endchunk
   384)           ncols = get_ncols_p(lchnk)
   385) 
   386)           call chunk_to_block_send_pters(lchnk,pcols,pver+1,tsize,cpter)
   387) 
   388)           do i=1,ncols
   389)              cbuffer(cpter(i,0):cpter(i,0)+2+pcnst) = 0.0_r8
   390)           end do
   391) 
   392)           do icol=1,ncols
   393) 
   394)              do ilyr=1,pver
   395)                 cbuffer   (cpter(icol,ilyr))     = phys_tend(lchnk)%dtdt(icol,ilyr)
   396)                 cbuffer   (cpter(icol,ilyr)+1)   = phys_tend(lchnk)%dudt(icol,ilyr)
   397)                 cbuffer   (cpter(icol,ilyr)+2)   = phys_tend(lchnk)%dvdt(icol,ilyr)
   398) 
   399)                 do m=1,pcnst
   400)                    cbuffer(cpter(icol,ilyr)+2+m) = phys_state(lchnk)%q(icol,ilyr,m)
   401)                 end do
   402)              end do
   403) 
   404)           end do
   405) 
   406)        end do
   407) 
   408)        call t_barrierf('sync_chk_to_blk', mpicom)
   409)        call t_startf ('chunk_to_block')
   410)        call transpose_chunk_to_block(tsize, cbuffer, bbuffer)
   411)        call t_stopf  ('chunk_to_block')
   412)        if(iam < par%nprocs) then
   413) !$omp parallel do private (ie, bpter, icol, ilyr, m)
   414)           do ie=1,nelemd
   415) 
   416)              call chunk_to_block_recv_pters(elem(ie)%GlobalID,npsq,pver+1,tsize,bpter)
   417) 
   418)              do icol=1,elem(ie)%idxP%NumUniquePts
   419) 
   420)                 do ilyr=1,pver
   421) 
   422)                    T_tmp   (icol,ilyr,ie)   = bbuffer(bpter(icol,ilyr))
   423)                    uv_tmp  (icol,1,ilyr,ie) = bbuffer(bpter(icol,ilyr)+1)
   424)                    uv_tmp  (icol,2,ilyr,ie) = bbuffer(bpter(icol,ilyr)+2)
   425) 
   426)                    do m=1,pcnst
   427)                       q_tmp(icol,ilyr,m,ie) = bbuffer(bpter(icol,ilyr)+2+m)
   428)                    end do
   429) 
   430)                 end do
   431)                 
   432)              end do
   433)              
   434)           end do
   435)        endif
   436)        deallocate( bbuffer )
   437)        deallocate( cbuffer )
   438)        
   439)     end if
   440)     call t_stopf('pd_copy')
   441)     if(iam < par%nprocs) then
   442)        call t_startf('putUniquePoints')
   443)        do ie=1,nelemd
   444)           ncols = elem(ie)%idxP%NumUniquePts
   445)           call putUniquePoints(elem(ie)%idxP, nlev, T_tmp(1:ncols,:,ie), elem(ie)%derived%fT(:,:,:,1))
   446)           call putUniquePoints(elem(ie)%idxV, 2, nlev, uv_tmp(1:ncols,:,:,ie), &
   447)                elem(ie)%derived%fM(:,:,:,:,1))
