   176)   subroutine spmdinit( mpicom_atm )
   177)     !----------------------------------------------------------------------- 
   178)     ! 
   179)     ! Purpose: MPI initialization routine:  
   180)     ! 
   181)     ! Method: get number of cpus, processes, tids, etc
   182)     !         dynamics and physics decompositions are set up later
   183)     ! 
   184)     ! Author: CCM Core Group
   185)     ! 
   186)     !-----------------------------------------------------------------------
   187) 
   188)     implicit none
   189)     integer, intent(in) :: mpicom_atm
   190) 
   191) #if ( defined SPMD )
   192)     !
   193)     ! Local workspace
   194)     !
   195)     integer i,j,c             ! indices
   196)     integer npthreads         ! thread status
   197)     integer ier               ! return error status    
   198)     integer length            ! length of name
   199)     integer max_len           ! maximum name length
   200)     integer, allocatable :: lengths(:)! max lengths of names for use in gatherv
   201)     integer, allocatable :: displs(:) ! offsets for use in gatherv
   202)     logical done
   203)     character, allocatable                             :: proc_name(:)  ! processor name, this task
   204)     character, allocatable                             :: proc_names(:) ! processor names, all tasks
   205)     character(len=mpi_max_processor_name)              :: tmp_name      ! temporary storage
   206)     character(len=mpi_max_processor_name), allocatable :: smp_names(:)  ! SMP name
   207)     logical mpi_running       ! returned value indicates if MPI_INIT has been called
   208) 
   209)     !---------------------------------------------------------------------------
   210)     !
   211)     ! Determine CAM MPI communicator group
   212)     !
   213)     mpicom  = mpicom_atm
   214)     !
   215)     ! Set mpishorthand variables.  Need to set as variables rather than parameters since
   216)     ! some MPI implementations set values for MPI tags at run time
   217)     !
   218)     mpiint  = mpi_integer
   219)     mpii8   = mpi_integer8
   220)     mpichar = mpi_character
   221)     mpilog  = mpi_logical
   222)     mpir4   = mpi_real4
   223)     mpir8   = mpi_real8
   224)     mpic16  = mpi_complex16
   225)     mpipk   = mpi_packed
   226)     mpimax  = mpi_max
   227)     !
   228)     ! Get my id  
   229)     !
   230)     call mpi_comm_rank (mpicom, iam, ier) 
   231)     masterprocid = DEFAULT_MASTERPROC
   232)     if (iam == DEFAULT_MASTERPROC) then 
   233)        masterproc = .true.
   234)     else
   235)        masterproc = .false.
   236)     end if
   237)     !
   238)     ! Get number of processors
   239)     !
   240)     max_len = mpi_max_processor_name
   241)     call mpi_comm_size (mpicom, npes, ier)
   242)     allocate ( displs(npes) )
   243)     allocate ( lengths(npes) )
   244)     allocate ( proc_name(max_len) )
   245)     allocate ( proc_names(max_len*npes) )
   246)  
   247)     !
   248)     ! Get processor names and send to root. 
   249)     !
   250)     call mpi_get_processor_name (tmp_name, length, ier)
   251)     proc_name(:) = ' '
   252)     do i = 1, length
   253)        proc_name(i) = tmp_name(i:i)
   254)     end do
   255) 
   256)     proc_names(:) = ' '
   257)     lengths(:) = max_len
   258)     do i=1,npes
   259)        displs(i) = (i-1)*max_len
   260)     enddo
   261)     call fc_gathervc (proc_name,  max_len, mpichar, &
   262)                       proc_names, lengths, displs, mpichar, &
   263)                       0, mpicom, flow_cntl=-1)
   264)     if (masterproc) then
   265)        write(iulog,*) npes, 'pes participating in computation'
   266)        write(iulog,*) '-----------------------------------'
   267)        write(iulog,*) 'TASK#  NAME'
   268)        do i=0,min(npes-1,256)  ! dont print too many of these
   269)           do c=1,max_len
   270)              tmp_name(c:c) = proc_names(i*max_len+c)
   271)           enddo
   272)           write(iulog,'(i3,2x,a)') i,trim(tmp_name)
   273)        end do
   274)        if(npes-1>256) then
   275)           write(iulog,*) '... list truncated at 256'
   276)        end if
   277)     end if
   278)     !
   279)     ! Identify SMP nodes and process/SMP mapping.
   280)     ! (Assume that processor names are SMP node names on SMP clusters.)
   281)     !
   282)     allocate ( proc_smp_map(0:npes-1) )
   283)     if (masterproc) then
   284)        allocate ( smp_names(0:npes-1) )
   285)        smp_names(:) = ' '
   286)        proc_smp_map(:) = -1
   287)        !
   288)        nsmps = 1
   289)        do c=1,max_len
   290)           tmp_name(c:c) = proc_names(c)
   291)        enddo
   292)        smp_names(0) = trim(tmp_name)
   293)        proc_smp_map(0) = 0
   294)        !
   295)        do i=1,npes-1
   296)           do c=1,max_len
   297)              tmp_name(c:c) = proc_names(i*max_len+c)
   298)           enddo
   299) 
   300)           j = 0
   301)           done = .false.
   302)           do while ((.not. done) .and. (j < nsmps))
   303)              if (smp_names(j) .eq. trim(tmp_name)) then
   304)                 proc_smp_map(i) = j
   305)                 done = .true.
   306)              endif
   307)              j = j + 1
   308)           enddo
   309) 
   310)           if (.not. done) then
   311)              smp_names(nsmps) = trim(tmp_name)
   312)              proc_smp_map(i) = nsmps
   313)              nsmps = nsmps + 1
   314)           endif
   315) 
   316)        enddo
   317)        deallocate(smp_names)
   318)     endif
   319)     call mpibcast(nsmps, 1, mpiint, 0, mpicom)
   320)     call mpibcast(proc_smp_map, npes, mpiint, 0, mpicom)
   321)     !
   322)     deallocate(displs)
   323)     deallocate(lengths)
   324)     deallocate(proc_name)
   325)     deallocate(proc_names)
   326) 
   327) #else
   328)     !	
   329)     ! spmd is not defined
   330)     !
   331)     mpicom = mpicom_atm
   332)     iam = 0
   333)     masterprocid = 0
   334)     masterproc = .true.
   335)     npes = 1
   336)     nsmps = 1
   337)     allocate ( proc_smp_map(0:0) )
   338)     proc_smp_map(:) = -1
   339) 
   340) #endif   
   341) 
   342)   end subroutine spmdinit

   348)    subroutine swapm (steps, nprocs, swapids,               &
   349)                      sndbuf, sbuf_siz, sndlths, sdispls,   &
   350)                      rcvbuf, rbuf_siz, rcvlths, rdispls,   &
   351)                      comm, comm_protocol, comm_maxreq      )
   352) 
   353) !----------------------------------------------------------------------- 
   354) ! 
   355) ! Purpose: 
   356) !   Reduced version of original swapm (for swap of multiple messages 
   357) !   using MPI point-to-point routines), more efficiently implementing a 
   358) !   subset of the swap protocols.
   359) ! 
   360) ! Method: 
   361) ! comm_protocol:
   362) !  = 3 or 5: use nonblocking send
   363) !  = 2 or 4: use blocking send
   364) !  = 4 or 5: use handshaking protocol
   365) ! comm_maxreq:
   366) !  =-1,0: do not limit number of outstanding send/receive requests
   367) !     >0: do not allow more than min(comm_maxreq, steps) outstanding
   368) !         nonblocking send requests or nonblocking receive requests
   369) !
   370) ! Author of original version:  P. Worley
   371) ! Ported to CAM: P. Worley, December 2003
   372) ! Simplified version: P. Worley, October, 2008
   373) ! 
   374) !-----------------------------------------------------------------------
   375) 
   376) !-----------------------------------------------------------------------
   377)    implicit none
   378) !---------------------------Input arguments--------------------------
   379) !
   380)    integer, intent(in)   :: steps              ! number of swaps to initiate
   381)    integer, intent(in)   :: nprocs             ! size of communicator
   382)    integer, intent(in)   :: sbuf_siz           ! size of send buffer
   383)    integer, intent(in)   :: rbuf_siz           ! size of receive buffer
   384)    integer, intent(in)   :: swapids(steps)     ! MPI process id of swap partners
   385) 
   386)    integer, intent(in)   :: sndlths(0:nprocs-1)! length of outgoing message
   387)    integer, intent(in)   :: sdispls(0:nprocs-1)! offset from beginning of send
   388)                                                !  buffer where outgoing messages
   389)                                                !  should be sent from
   390)    integer, intent(in)   :: rcvlths(0:nprocs-1)! length of incoming messages
   391)    integer, intent(in)   :: rdispls(0:nprocs-1)! offset from beginning of receive 
   392)                                                !  buffer where incoming messages
   393)                                                !  should be placed
   394)    real(r8), intent(in)  :: sndbuf(sbuf_siz)   ! outgoing message buffer
   395)    real(r8), intent(out) :: rcvbuf(rbuf_siz)   ! incoming message buffer
   396) 
   397)    integer, intent(in)   :: comm               ! MPI communicator
   398)    integer, intent(in)   :: comm_protocol      ! swap_comm protocol
   399)    integer, intent(in)   :: comm_maxreq        ! maximum number of outstanding 
   400)                                                !  nonblocking requests
   401) 
   402) !
   403) !---------------------------Local workspace-----------------------------
   404) !
   405)    integer :: p                                ! process index
   406)    integer :: istep                            ! loop index
   407)    integer :: offset_s                         ! index of message beginning in 
   408)                                                !  send buffer
   409)    integer :: offset_r                         ! index of message beginning in 
   410)                                                !  receive buffer
   411)    integer :: sndids(steps)                    ! send request ids
   412)    integer :: rcvids(steps)                    ! receive request ids
   413)    integer :: hs_rcvids(steps)                 ! handshake receive request ids
   414) 
   415)    integer :: maxreq, maxreqh                  ! maximum number of outstanding 
   416)                                                !  nonblocking requests (and half)
   417)    integer :: hs_s, hs_r(steps)                ! handshake variables (send/receive)
   418)    integer :: rstep                            ! "receive" step index
   419) 
   420)    logical :: handshake, sendd                 ! protocol option flags
   421) 
   422)    integer :: ier                              ! return error status    
   423)    integer :: status(MPI_STATUS_SIZE)          ! MPI status 
   424) !
   425) !-------------------------------------------------------------------------------------
   426) !
   427)    if (steps .eq. 0) return
   428) 
   429)    ! identify communication protocol
   430)    if ((comm_protocol < 2) .or. (comm_protocol > 5)) then
   431)       sendd = .true.
   432)       handshake = .true.
   433)    else
   434)       if ((comm_protocol .eq. 4) .or. (comm_protocol .eq. 5)) then
   435)          handshake = .true.
   436)       else
   437)          handshake = .false.
   438)       endif
   439) 
   440)       if ((comm_protocol .eq. 2) .or. (comm_protocol .eq. 4)) then
   441)          sendd = .true.
   442)       else
   443)          sendd = .false.
   444)       endif
   445)    endif
   446) 
   447)    ! identify maximum number of outstanding nonblocking requests to permit
   448)    if (steps .eq. 1) then
   449)       maxreq  = 1
   450)       maxreqh = 1
   451)    else
   452)       if (comm_maxreq >= -1) then
   453)          maxreq = comm_maxreq
   454)       else
   455)          maxreq = steps
   456)       endif
   457) 
   458)       if ((maxreq .le. steps) .and. (maxreq > 0)) then
   459)          if (maxreq > 1) then
   460)             maxreqh = maxreq/2
   461)          else
   462)             maxreq  = 2
   463)             maxreqh = 1
   464)          endif
   465)       else
   466)          maxreq  = steps
   467)          maxreqh = steps
   468)       endif
   469)    endif
   470) 
   471) ! Four protocol options:
   472) !  (1) handshaking + blocking sends
   473)    if ((handshake) .and. (sendd)) then
   474) 
   475)       ! Initialize handshake variable
   476)       hs_s = 1
   477) 
   478)       ! Post initial handshake receive requests
   479)       do istep=1,maxreq
   480)          p = swapids(istep)
   481)          if (sndlths(p) > 0) then
   482)             call mpi_irecv( hs_r(istep), 1, mpiint, p, iam, comm, &
   483)                             hs_rcvids(istep), ier )
   484)          endif
   485)       enddo
   486) 
   487)       ! Post initial receive requests
   488)       do istep=1,maxreq
   489)          p = swapids(istep)
   490)          if (rcvlths(p) > 0) then
   491)             offset_r = rdispls(p)+1
   492)             call mpi_irecv( rcvbuf(offset_r), rcvlths(p), mpir8, p, p, &
   493)                             comm, rcvids(istep), ier )
   494)             call mpi_send ( hs_s, 1, mpiint, p, p, comm, ier )
   495)          endif
   496)       enddo
   497)       rstep = maxreq
   498) 
   499)       ! Send (and start receiving) data 
   500)       do istep=1,steps
   501)          p = swapids(istep)
   502) 
   503)          ! Submit new rsend request
   504)          if (sndlths(p) > 0) then
   505)             offset_s = sdispls(p)+1
   506)             call mpi_wait  ( hs_rcvids(istep), MPI_STATUS_IGNORE, ier )
   507)             call mpi_rsend ( sndbuf(offset_s), sndlths(p), mpir8, p, iam, &
   508)                              comm, ier )
   509)          endif
   510) 
   511)          if (istep > maxreqh) then
   512) 
   513)             ! Wait for oldest irecv request to complete
   514)             p = swapids(istep-maxreqh)
   515)             if (rcvlths(p) > 0) then
   516)                call mpi_wait( rcvids(istep-maxreqh), status, ier )
   517)             endif
   518) 
   519)             if (rstep < steps) then
   520)                rstep = rstep + 1
   521)                p = swapids(rstep)
   522) 
   523)                ! Submit a new handshake irecv request
   524)                if (sndlths(p) > 0) then
   525)                   call mpi_irecv( hs_r(rstep), 1, mpiint, p, iam, comm, &
   526)                                   hs_rcvids(rstep), ier )
   527)                endif
   528) 
   529)                ! Submit a new irecv request
   530)                if (rcvlths(p) > 0) then
   531)                   offset_r = rdispls(p)+1
   532)                   call mpi_irecv( rcvbuf(offset_r), rcvlths(p), mpir8, p, p, &
   533)                                   comm, rcvids(rstep), ier )
   534)                   call mpi_send ( hs_s, 1, mpiint, p, p, comm, ier )
   535)                endif
   536)             endif
   537) 
   538)          endif
   539) !
   540)       enddo
   541) 
   542)       ! wait for rest of receive requests to complete
   543)       do istep=steps-maxreqh+1,steps
   544)          p = swapids(istep)
   545)          if (rcvlths(p) > 0) then
   546)             call mpi_wait( rcvids(istep), status, ier )
   547)          endif
   548)       enddo
   549) 
   550) !  (2) handshaking + nonblocking sends
   551)    elseif ((handshake) .and. (.not. sendd)) then
   552) 
   553)       ! Initialize handshake variable
   554)       hs_s = 1
   555) 
   556)       ! Post initial handshake receive requests
   557)       do istep=1,maxreq
   558)          p = swapids(istep)
   559)          if (sndlths(p) > 0) then
   560)             call mpi_irecv( hs_r(istep), 1, mpiint, p, iam, comm, &
   561)                             hs_rcvids(istep), ier )
   562)          endif
   563)       enddo
   564) 
   565)       ! Post initial receive requests
   566)       do istep=1,maxreq
   567)          p = swapids(istep)
   568)          if (rcvlths(p) > 0) then
   569)             offset_r = rdispls(p)+1
   570)             call mpi_irecv( rcvbuf(offset_r), rcvlths(p), mpir8, p, p, &
   571)                             comm, rcvids(istep), ier )
   572)             call mpi_send ( hs_s, 1, mpiint, p, p, comm, ier )
   573)          endif
   574)       enddo
   575)       rstep = maxreq
   576) 
   577)       ! Send (and start receiving) data 
   578)       do istep=1,steps
   579)          p = swapids(istep)
   580) 
   581)          ! Submit new irsend request
   582)          if (sndlths(p) > 0) then
   583)             offset_s = sdispls(p)+1
   584)             call mpi_wait  ( hs_rcvids(istep), MPI_STATUS_IGNORE, ier )
   585)             call mpi_irsend( sndbuf(offset_s), sndlths(p), mpir8, p, iam, &
   586)                              comm, sndids(istep), ier )
   587)          endif
   588) 
   589)          if (istep > maxreqh) then
   590) 
   591)             ! Wait for oldest irecv request to complete
   592)             p = swapids(istep-maxreqh)
   593)             if (rcvlths(p) > 0) then
   594)                call mpi_wait( rcvids(istep-maxreqh), status, ier )
   595)             endif
   596) 
   597)             if (rstep < steps) then
   598)                rstep = rstep + 1
   599)                p = swapids(rstep)
   600) 
   601)                ! Submit a new handshake irecv request
   602)                if (sndlths(p) > 0) then
   603)                   call mpi_irecv( hs_r(rstep), 1, mpiint, p, iam, comm, &
   604)                                   hs_rcvids(rstep), ier )
   605)                endif
   606) 
   607)                ! Submit a new irecv request
   608)                if (rcvlths(p) > 0) then
   609)                   offset_r = rdispls(p)+1
   610)                   call mpi_irecv( rcvbuf(offset_r), rcvlths(p), mpir8, p, p, &
   611)                                   comm, rcvids(rstep), ier )
   612)                   call mpi_send ( hs_s, 1, mpiint, p, p, comm, ier )
   613)                endif
   614)             endif
   615) 
   616)             ! Wait for outstanding i(r)send request to complete
   617)             p = swapids(istep-maxreqh)
   618)             if (sndlths(p) > 0) then
   619)                call mpi_wait( sndids(istep-maxreqh), status, ier )
   620)             endif
   621) 
   622)          endif
   623) 
   624)       enddo
   625) 
   626)       ! wait for rest of send and receive requests to complete
   627)       do istep=steps-maxreqh+1,steps
   628)          p = swapids(istep)
   629)          if (rcvlths(p) > 0) then
   630)             call mpi_wait( rcvids(istep), status, ier )
   631)          endif
   632)          if (sndlths(p) > 0) then
   633)             call mpi_wait( sndids(istep), status, ier )
   634)          endif
   635)       enddo
   636) 
   637) !  (3) no handshaking + blocking sends
   638)    elseif ((.not. handshake) .and. (sendd)) then
   639) 
   640)       ! Post receive requests
   641)       do istep=1,maxreq
   642)          p = swapids(istep)
   643)          if (rcvlths(p) > 0) then
   644)             offset_r = rdispls(p)+1
   645)             call mpi_irecv( rcvbuf(offset_r), rcvlths(p), mpir8, p, p, &
   646)                             comm, rcvids(istep), ier )
   647)          endif
   648)       enddo
   649)       rstep = maxreq
   650) 
   651)       ! Send (and start receiving) data 
   652)       do istep=1,steps
   653)          p = swapids(istep)
   654) 
   655)          ! Submit new send request
   656)          if (sndlths(p) > 0) then
   657)             offset_s = sdispls(p)+1
   658)             call mpi_send( sndbuf(offset_s), sndlths(p), mpir8, p, iam, &
   659)                            comm, ier )
   660)          endif
   661) 
   662)          if (istep > maxreqh) then
   663) 
   664)             ! Wait for oldest irecv request to complete
   665)             p = swapids(istep-maxreqh)
   666)             if (rcvlths(p) > 0) then
   667)                call mpi_wait( rcvids(istep-maxreqh), status, ier )
   668)             endif
   669) 
   670)             ! Submit a new irecv request
   671)             if (rstep < steps) then
   672)                rstep = rstep + 1
   673)                p = swapids(rstep)
   674)                if (rcvlths(p) > 0) then
   675)                   offset_r = rdispls(p)+1
   676)                   call mpi_irecv( rcvbuf(offset_r), rcvlths(p), mpir8, p, p, &
   677)                                   comm, rcvids(rstep), ier )
   678)                endif
   679)             endif
   680) 
   681)          endif
   682) 
   683)       enddo
   684) 
   685)       ! wait for rest of send and receive requests to complete
   686)       do istep=steps-maxreqh+1,steps
   687)          p = swapids(istep)
   688)          if (rcvlths(p) > 0) then
   689)             call mpi_wait( rcvids(istep), status, ier )
   690)          endif
   691)       enddo
   692) 
   693) !  (4) no handshaking + nonblocking sends
   694)    elseif ((.not. handshake) .and. (.not. sendd)) then
   695) 
   696)       ! Post receive requests
   697)       do istep=1,maxreq
   698)          p = swapids(istep)
   699)          if (rcvlths(p) > 0) then
   700)             offset_r = rdispls(p)+1
   701)             call mpi_irecv( rcvbuf(offset_r), rcvlths(p), mpir8, p, p, &
   702)                             comm, rcvids(istep), ier )
   703)          endif
   704)       enddo
   705)       rstep = maxreq
   706) 
   707)       ! Send (and start receiving) data 
   708)       do istep=1,steps
   709)          p = swapids(istep)
   710) 
   711)          ! Submit new isend request
   712)          if (sndlths(p) > 0) then
   713)             offset_s = sdispls(p)+1
   714)             call mpi_isend( sndbuf(offset_s), sndlths(p), mpir8, p, iam, &
   715)                             comm, sndids(istep), ier )
   716)          endif
   717) 
   718)          if (istep > maxreqh) then
   719) 
   720)             ! Wait for oldest irecv request to complete
   721)             p = swapids(istep-maxreqh)
   722)             if (rcvlths(p) > 0) then
   723)                call mpi_wait( rcvids(istep-maxreqh), status, ier )
   724)             endif
   725) 
   726)             ! Submit a new irecv request
   727)             if (rstep < steps) then
   728)                rstep = rstep + 1
   729)                p = swapids(rstep)
   730)                if (rcvlths(p) > 0) then
   731)                   offset_r = rdispls(p)+1
   732)                   call mpi_irecv( rcvbuf(offset_r), rcvlths(p), mpir8, p, p, &
   733)                                   comm, rcvids(rstep), ier )
   734)                endif
   735)             endif
   736) 
   737)             ! Wait for outstanding i(r)send request to complete
   738)             p = swapids(istep-maxreqh)
   739)             if (sndlths(p) > 0) then
   740)                call mpi_wait( sndids(istep-maxreqh), status, ier )
   741)             endif
   742) 
   743)          endif
   744) 
   745)       enddo
   746) 
   747)       ! wait for rest of send and receive requests to complete
   748)       do istep=steps-maxreqh+1,steps
   749)          p = swapids(istep)
   750)          if (rcvlths(p) > 0) then
   751)             call mpi_wait( rcvids(istep), status, ier )
   752)          endif
   753)          if (sndlths(p) > 0) then
   754)             call mpi_wait( sndids(istep), status, ier )
   755)          endif
   756)       enddo
   757) 
   758)    endif
   759) 
   760)    return

   768) ! Purpose: gather collective with additional flow control, so as to

   772) !          >= 0: use point-to-point with handshaking messages and 
   773) !                preposting receive requests up to 
   774) !                 min(max(1,flow_cntl),max_gather_block_size)

   776) !                Otherwise, fc_gather_flow_cntl is used in its place.
   777) !          Default value is 64.
   778) ! 
   779) ! Entry points:
   780) !      fc_gatherv       functionally equivalent to mpi_gatherv
   781) !      fc_gathervr4     functionally equivalent to mpi_gatherv for real*4 data
   782) !      fc_gathervint    functionally equivalent to mpi_gatherv for integer data

   785) ! Author: P. Worley

   787) 
   788) !
   789) !========================================================================
   790) !
   791)    subroutine fc_gatherv (sendbuf, sendcnt, sendtype, &
   792)                          recvbuf, recvcnts, displs, recvtype, &
   793)                          root, comm, flow_cntl )
   794) !
   795) ! Collects different messages from each process on masterproc
   796) !
   797)    use shr_kind_mod, only: r8 => shr_kind_r8
   798)    use mpishorthand
   799)    use abortutils, only: endrun
   800)    use cam_logfile,  only: iulog
   801) #if defined( WRAP_MPI_TIMING )
   802)    use perf_mod
   803) #endif
   804) 
   805)    implicit none
   806) 
   807)    real (r8), intent(in)  :: sendbuf(*)
   808)    real (r8), intent(out) :: recvbuf(*)
   809)    integer, intent(in) :: displs(*)
   810)    integer, intent(in) :: sendcnt
   811)    integer, intent(in) :: sendtype
   812)    integer, intent(in) :: recvcnts(*)
   813)    integer, intent(in) :: recvtype
   814)    integer, intent(in) :: root
   815)    integer, intent(in) :: comm
   816)    integer, optional, intent(in) :: flow_cntl
   817) 
   818)    real (r8) :: signal
   819)    logical fc_gather         ! use explicit flow control?
   820)    integer gather_block_size ! number of preposted receive requests
   821) 
   822)    integer :: mytid, mysize, mtag, p, q, i, count
   823)    integer :: preposts, head, tail
   824)    integer :: rcvid(max_gather_block_size)
   825)    integer :: status(MPI_STATUS_SIZE)
   826)    integer ier               ! MPI error code
   827) 
   828)    if ( present(flow_cntl) ) then
   829)       if (flow_cntl >= 0) then
   830)          gather_block_size = min(max(1,flow_cntl),max_gather_block_size)
   831)          fc_gather = .true.
   832)       else
   833)          fc_gather = .false.
   834)       endif
   835)    else
   836)       if (fc_gather_flow_cntl >= 0) then
   837)          gather_block_size = min(max(1,fc_gather_flow_cntl),max_gather_block_size)
   838)          fc_gather = .true.
   839)       else
   840)          fc_gather = .false.
   841)       endif
   842)    endif
   843) 
   844)    if (fc_gather) then
   845)  
   846) #if defined( WRAP_MPI_TIMING )
   847)       call t_startf ('fc_gatherv_r8')
   848) #endif
   849)       call mpi_comm_rank (comm, mytid, ier)
   850)       call mpi_comm_size (comm, mysize, ier)
   851)       mtag = 0
   852)       if (root .eq. mytid) then
   853) 
   854) ! prepost gather_block_size irecvs, and start receiving data
   855)          preposts = min(mysize-1, gather_block_size)
   856)          head = 0
   857)          count = 0
   858)          do p=0, mysize-1
   859)             if (p .ne. root) then
   860)                q = p+1
   861)                if (recvcnts(q) > 0) then
   862)                   count = count + 1
   863)                   if (count > preposts) then
   864)                      tail = mod(head,preposts) + 1
   865)                      call mpi_wait (rcvid(tail), status, ier)
   866)                   end if
   867)                   head = mod(head,preposts) + 1
   868)                   call mpi_irecv ( recvbuf(displs(q)+1), recvcnts(q), &
   869)                                    recvtype, p, mtag, comm, rcvid(head), &
   870)                                    ier )
   871)                   call mpi_send ( signal, 1, mpir8, p, mtag, comm, ier )
   872)                end if
   873)             end if
   874)          end do
   875) 
   876) ! copy local data
   877)          q = mytid+1
   878)          do i=1,sendcnt
   879)             recvbuf(displs(q)+i) = sendbuf(i)
   880)          enddo
   881) 
   882) ! wait for final data
   883)          do i=1,min(count,preposts)
   884)             call mpi_wait (rcvid(i), status, ier)
   885)          enddo
   886) 
   887)       else
   888) 
   889)          if (sendcnt > 0) then
   890)             call mpi_recv ( signal, 1, mpir8, root, mtag, comm, &
   891)                             status, ier )
   892)             call mpi_rsend ( sendbuf, sendcnt, sendtype, root, mtag, &
   893)                              comm, ier )
   894)          end if
   895) 
   896)       endif
   897)       if (ier /= mpi_success) then
   898)          write(iulog,*)'fc_gatherv_r8 failed ier=',ier
   899)          call endrun
   900)       end if
   901) #if defined( WRAP_MPI_TIMING )
   902)       call t_stopf ('fc_gatherv_r8')
   903) #endif
   904) 
   905)    else
   906)  
   907) #if defined( WRAP_MPI_TIMING )
   908)       call t_startf ('mpi_gatherv')
   909) #endif
   910)       call mpi_gatherv (sendbuf, sendcnt, sendtype, &
   911)                         recvbuf, recvcnts, displs, recvtype, &
   912)                         root, comm, ier)
   913)       if (ier /= mpi_success) then
   914)          write(iulog,*)'mpi_gatherv failed ier=',ier
   915)          call endrun
   916)       end if
   917) #if defined( WRAP_MPI_TIMING )
   918)       call t_stopf ('mpi_gatherv')
   919) #endif
   920) 
   921)    endif
   922) 
   923)    return
