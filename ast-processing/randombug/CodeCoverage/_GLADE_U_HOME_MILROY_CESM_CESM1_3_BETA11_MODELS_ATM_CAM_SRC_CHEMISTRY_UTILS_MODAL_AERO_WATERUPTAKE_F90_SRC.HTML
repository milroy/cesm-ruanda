   113) subroutine modal_aero_wateruptake_dr(state, pbuf, list_idx_in, dgnumdry_m, dgnumwet_m, &
   114)                                      qaerwat_m, wetdens_m)
   115) !-----------------------------------------------------------------------
   116) !
   117) ! CAM specific driver for modal aerosol water uptake code.
   118) !
   119) ! *** N.B. *** The calculation has been enabled for diagnostic mode lists
   120) !              via optional arguments.  If the list_idx arg is present then
   121) !              all the optional args must be present.
   122) !
   123) !-----------------------------------------------------------------------
   124) 
   125)    ! Arguments
   126)    type(physics_state), target, intent(in)    :: state          ! Physics state variables
   127)    type(physics_buffer_desc),   pointer       :: pbuf(:)        ! physics buffer
   128) 
   129)    integer,  optional,          intent(in)    :: list_idx_in
   130)    real(r8), optional, target,  intent(in)    :: dgnumdry_m(:,:,:)
   131)    real(r8), optional,          pointer       :: dgnumwet_m(:,:,:)
   132)    real(r8), optional,          pointer       :: qaerwat_m(:,:,:)
   133)    real(r8), optional,          pointer       :: wetdens_m(:,:,:)
   134) 
   135)    ! local variables
   136) 
   137)    integer  :: lchnk              ! chunk index
   138)    integer  :: ncol               ! number of columns
   139)    integer  :: list_idx           ! radiative constituents list index
   140)    integer  :: stat
   141) 
   142)    integer :: i, k, l, m
   143)    integer :: itim_old
   144)    integer :: nmodes
   145)    integer :: nspec
   146) 
   147)    real(r8), pointer :: h2ommr(:,:) ! specific humidity
   148)    real(r8), pointer :: t(:,:)      ! temperatures (K)
   149)    real(r8), pointer :: pmid(:,:)   ! layer pressure (Pa)
   150)    real(r8), pointer :: raer(:,:)   ! aerosol species MRs (kg/kg and #/kg)
   151) 
   152)    real(r8), pointer :: cldn(:,:)      ! layer cloud fraction (0-1)
   153)    real(r8), pointer :: dgncur_a(:,:,:)
   154)    real(r8), pointer :: dgncur_awet(:,:,:)
   155)    real(r8), pointer :: wetdens(:,:,:)
   156)    real(r8), pointer :: qaerwat(:,:,:)
   157) 
   158)    real(r8), allocatable :: maer(:,:,:)      ! aerosol wet mass MR (including water) (kg/kg-air)
   159)    real(r8), allocatable :: hygro(:,:,:)     ! volume-weighted mean hygroscopicity (--)
   160)    real(r8), allocatable :: naer(:,:,:)      ! aerosol number MR (bounded!) (#/kg-air)
   161)    real(r8), allocatable :: dryvol(:,:,:)    ! single-particle-mean dry volume (m3)
   162)    real(r8), allocatable :: drymass(:,:,:)   ! single-particle-mean dry mass  (kg)
   163)    real(r8), allocatable :: dryrad(:,:,:)    ! dry volume mean radius of aerosol (m)
   164) 
   165)    real(r8), allocatable :: wetrad(:,:,:)    ! wet radius of aerosol (m)
   166)    real(r8), allocatable :: wetvol(:,:,:)    ! single-particle-mean wet volume (m3)
   167)    real(r8), allocatable :: wtrvol(:,:,:)    ! single-particle-mean water volume in wet aerosol (m3)
   168) 
   169)    real(r8), allocatable :: rhcrystal(:)
   170)    real(r8), allocatable :: rhdeliques(:)
   171)    real(r8), allocatable :: specdens_1(:)
   172) 
   173)    real(r8) :: dryvolmr(pcols,pver)          ! volume MR for aerosol mode (m3/kg)
   174)    real(r8) :: specdens
   175)    real(r8) :: spechygro, spechygro_1
   176)    real(r8) :: duma, dumb
   177)    real(r8) :: sigmag
   178)    real(r8) :: alnsg
   179)    real(r8) :: v2ncur_a
   180)    real(r8) :: drydens               ! dry particle density  (kg/m^3)
   181)    real(r8) :: rh(pcols,pver)        ! relative humidity (0-1)
   182) 
   183)    real(r8) :: es(pcols)             ! saturation vapor pressure
   184)    real(r8) :: qs(pcols)             ! saturation specific humidity
   185) 
   186)    character(len=3) :: trnum       ! used to hold mode number (as characters)
   187)    !-----------------------------------------------------------------------
   188) 
   189)    lchnk = state%lchnk
   190)    ncol = state%ncol
   191) 
   192)    list_idx = 0
   193)    if (present(list_idx_in)) then
   194)       list_idx = list_idx_in
   195) 
   196)       ! check that all optional args are present
   197)       if (.not. present(dgnumdry_m)   .or. .not. present(dgnumwet_m) .or. &
   198)           .not. present(qaerwat_m) .or. .not. present(wetdens_m)) then
   199)          call endrun('modal_aero_wateruptake_dr called for'// &
   200)                      'diagnostic list but required args not present')
   201)       end if
   202)    end if
   203) 
   204)    ! loop over all aerosol modes
   205)    call rad_cnst_get_info(list_idx, nmodes=nmodes)
   206) 
   207)    allocate( &
   208)       maer(pcols,pver,nmodes),     &
   209)       hygro(pcols,pver,nmodes),    &
   210)       naer(pcols,pver,nmodes),     &
   211)       dryvol(pcols,pver,nmodes),   &
   212)       drymass(pcols,pver,nmodes),  &
   213)       dryrad(pcols,pver,nmodes),   &
   214)       wetrad(pcols,pver,nmodes),   &
   215)       wetvol(pcols,pver,nmodes),   &
   216)       wtrvol(pcols,pver,nmodes),   &
   217)       rhcrystal(nmodes),           &
   218)       rhdeliques(nmodes),          &
   219)       specdens_1(nmodes)           )
   220) 
   221)    maer(:,:,:)     = 0._r8
   222)    hygro(:,:,:)    = 0._r8
   223) 
   224) 
   225)    if (list_idx == 0) then
   226)       call pbuf_get_field(pbuf, dgnum_idx, dgncur_a)
   227)    else
   228)       dgncur_a => dgnumdry_m
   229)    end if
   230) 
   231)    do m = 1, nmodes
   232) 
   233)       dryvolmr(:,:) = 0._r8
   234) 
   235)       ! get mode properties
   236)       call rad_cnst_get_mode_props(list_idx, m, sigmag=sigmag,  &
   237)          rhcrystal=rhcrystal(m), rhdeliques=rhdeliques(m))
   238) 
   239)       ! get mode info
   240)       call rad_cnst_get_info(list_idx, m, nspec=nspec)
   241) 
   242)       do l = 1, nspec
   243) 
   244)          ! get species interstitial mixing ratio ('a')
   245)          call rad_cnst_get_aer_mmr(list_idx, m, l, 'a', state, pbuf, raer)
   246)          call rad_cnst_get_aer_props(list_idx, m, l, density_aer=specdens, hygro_aer=spechygro)
   247) 
   248)          if (l == 1) then
   249)             ! save off these values to be used as defaults
   250)             specdens_1(m)  = specdens
   251)             spechygro_1    = spechygro
   252)          end if
   253) 
   254)          do k = top_lev, pver
   255)             do i = 1, ncol
   256)                duma          = raer(i,k)
   257)                maer(i,k,m)   = maer(i,k,m) + duma
   258)                dumb          = duma/specdens
   259)                dryvolmr(i,k) = dryvolmr(i,k) + dumb
   260)                hygro(i,k,m)  = hygro(i,k,m) + dumb*spechygro
   261)             end do
   262)          end do
   263)       end do
   264) 
   265)       alnsg = log(sigmag)
   266) 
   267)       do k = top_lev, pver
   268)          do i = 1, ncol
   269) 
   270)             if (dryvolmr(i,k) > 1.0e-30_r8) then
   271)                hygro(i,k,m) = hygro(i,k,m)/dryvolmr(i,k)
   272)             else
   273)                hygro(i,k,m) = spechygro_1
   274)             end if
   275) 
   276)             ! dry aerosol properties
   277) 
   278)             v2ncur_a = 1._r8 / ( (pi/6._r8)*(dgncur_a(i,k,m)**3._r8)*exp(4.5_r8*alnsg**2._r8) )
   279)             ! naer = aerosol number (#/kg)
   280)             naer(i,k,m) = dryvolmr(i,k)*v2ncur_a
   281) 
   282)             ! compute mean (1 particle) dry volume and mass for each mode
   283)             ! old coding is replaced because the new (1/v2ncur_a) is equal to
   284)             ! the mean particle volume
   285)             ! also moletomass forces maer >= 1.0e-30, so (maer/dryvolmr)
   286)             ! should never cause problems (but check for maer < 1.0e-31 anyway)
   287)             if (maer(i,k,m) .gt. 1.0e-31_r8) then
   288)                drydens = maer(i,k,m)/dryvolmr(i,k)
   289)             else
   290)                drydens = 1.0_r8
   291)             end if
   292)             dryvol(i,k,m)   = 1.0_r8/v2ncur_a
   293)             drymass(i,k,m)  = drydens*dryvol(i,k,m)
   294)             dryrad(i,k,m)   = (dryvol(i,k,m)/pi43)**third
   295) 
   296)          end do
   297)       end do
   298) 
   299)    end do    ! modes
   300) 
   301)    ! relative humidity calc
   302) 
   303)    h2ommr => state%q(:,:,1)
   304)    t      => state%t
   305)    pmid   => state%pmid
   306) 
   307)    itim_old    =  pbuf_old_tim_idx()
   308)    call pbuf_get_field(pbuf, cld_idx, cldn, start=(/1,1,itim_old/), kount=(/pcols,pver,1/) )
   309) 
   310)    do k = top_lev, pver
   311)       call qsat_water(t(:ncol,k), pmid(:ncol,k), es(:ncol), qs(:ncol))
   312)       do i = 1, ncol
   313)          if (qs(i) > h2ommr(i,k)) then
   314)             rh(i,k) = h2ommr(i,k)/qs(i)
   315)          else
   316)             rh(i,k) = 0.98_r8
   317)          endif
   318)          rh(i,k) = max(rh(i,k), 0.0_r8)
   319)          rh(i,k) = min(rh(i,k), 0.98_r8)
   320)          if (cldn(i,k) .lt. 1.0_r8) then
   321)             rh(i,k) = (rh(i,k) - cldn(i,k)) / (1.0_r8 - cldn(i,k))  ! clear portion
   322)          end if
   323)          rh(i,k) = max(rh(i,k), 0.0_r8)
   324)       end do
   325)    end do
   326) 
   327)    call modal_aero_wateruptake_sub( &
   328)       ncol, nmodes, rhcrystal, rhdeliques, dryrad, &
   329)       hygro, rh, dryvol, wetrad, wetvol,           &
   330)       wtrvol)
   331) 
   332)    if (list_idx == 0) then
   333)       call pbuf_get_field(pbuf, dgnumwet_idx,   dgncur_awet)
   334)       call pbuf_get_field(pbuf, wetdens_ap_idx, wetdens)
   335)       call pbuf_get_field(pbuf, qaerwat_idx,    qaerwat)
   336)    else
   337)       allocate(dgncur_awet(pcols,pver,nmodes), wetdens(pcols,pver,nmodes), &
   338)                qaerwat(pcols,pver,nmodes), stat=stat)
   339)       if (stat > 0) then
   340)          call endrun('modal_aero_wateruptake_dr: allocation FAILURE')
   341)       end if
   342) 
   343)    end if
   344) 
   345)    do m = 1, nmodes
   346) 
   347)       do k = top_lev, pver
   348)          do i = 1, ncol
   349) 
   350)             dgncur_awet(i,k,m) = dgncur_a(i,k,m) * (wetrad(i,k,m)/dryrad(i,k,m))
   351)             qaerwat(i,k,m)     = rhoh2o*naer(i,k,m)*wtrvol(i,k,m)
   352) 
   353)             ! compute aerosol wet density (kg/m3)
   354)             if (wetvol(i,k,m) > 1.0e-30_r8) then
   355)                wetdens(i,k,m) = (drymass(i,k,m) + rhoh2o*wtrvol(i,k,m))/wetvol(i,k,m)
   356)             else
   357)                wetdens(i,k,m) = specdens_1(m)
   358)             end if
   359)          end do
   360)       end do
   361) 
   362)    end do    ! modes
   363) 
   364)    if (list_idx == 0) then
   365) 
   366)       do m = 1, nmodes
   367)          ! output to history
   368)          write( trnum, '(i3.3)' ) m
   369)          call outfld( 'wat_a'//trnum(3:3),  qaerwat(:,:,m),     pcols, lchnk)
   370)          call outfld( 'dgnd_a'//trnum(2:3), dgncur_a(:,:,m),    pcols, lchnk)
   371)          call outfld( 'dgnw_a'//trnum(2:3), dgncur_awet(:,:,m), pcols, lchnk)
   372)       end do
   373) 
   374)    else
   375) 
   376)       ! for diagnostic calcs just return results
   377)       dgnumwet_m => dgncur_awet
   378)       qaerwat_m  => qaerwat
   379)       wetdens_m  => wetdens
   380) 
   381)    end if
   382) 
   383)    deallocate( &
   384)       maer,   hygro,  naer,   dryvol,    drymass,    dryrad, &
   385)       wetrad, wetvol, wtrvol, rhcrystal, rhdeliques, specdens_1)
   386) 
   387) end subroutine modal_aero_wateruptake_dr

   391) subroutine modal_aero_wateruptake_sub( &
   392)    ncol, nmodes, rhcrystal, rhdeliques, dryrad, &
   393)    hygro, rh, dryvol, wetrad, wetvol,           &
   394)    wtrvol)
   395) 
   396) !-----------------------------------------------------------------------
   397) !
   398) ! Purpose: Compute aerosol wet radius
   399) !
   400) ! Method:  Kohler theory
   401) !
   402) ! Author:  S. Ghan
   403) !
   404) !-----------------------------------------------------------------------
   405) 
   406)    ! Arguments
   407)    integer, intent(in)  :: ncol                    ! number of columns
   408)    integer, intent(in)  :: nmodes
   409) 
   410)    real(r8), intent(in) :: rhcrystal(:)
   411)    real(r8), intent(in) :: rhdeliques(:)
   412)    real(r8), intent(in) :: dryrad(:,:,:)         ! dry volume mean radius of aerosol (m)
   413)    real(r8), intent(in) :: hygro(:,:,:)          ! volume-weighted mean hygroscopicity (--)
   414)    real(r8), intent(in) :: rh(:,:)               ! relative humidity (0-1)
   415)    real(r8), intent(in) :: dryvol(:,:,:)
   416) 
   417)    real(r8), intent(out) :: wetrad(:,:,:)        ! wet radius of aerosol (m)
   418)    real(r8), intent(out) :: wetvol(:,:,:)        ! single-particle-mean wet volume (m3)
   419)    real(r8), intent(out) :: wtrvol(:,:,:)        ! single-particle-mean water volume in wet aerosol (m3)
   420) 
   421)    ! local variables
   422) 
   423)    integer :: i, k, m
   424) 
   425)    real(r8) :: hystfac                ! working variable for hysteresis
   426)    !-----------------------------------------------------------------------
   427) 
   428) 
   429)    ! loop over all aerosol modes
   430)    do m = 1, nmodes
   431) 
   432)       hystfac = 1.0_r8 / max(1.0e-5_r8, (rhdeliques(m) - rhcrystal(m)))
   433) 
   434)       do k = top_lev, pver
   435)          do i = 1, ncol
   436) 
   437)             ! compute wet radius for each mode
   438)             call modal_aero_kohler(dryrad(i:i,k,m), hygro(i:i,k,m), rh(i:i,k), wetrad(i:i,k,m), 1)
   439) 
   440)             wetrad(i,k,m) = max(wetrad(i,k,m), dryrad(i,k,m))
   441)             wetvol(i,k,m) = pi43*wetrad(i,k,m)**3
   442)             wetvol(i,k,m) = max(wetvol(i,k,m), dryvol(i,k,m))
   443)             wtrvol(i,k,m) = wetvol(i,k,m) - dryvol(i,k,m)
   444)             wtrvol(i,k,m) = max(wtrvol(i,k,m), 0.0_r8)
   445) 
   446)             ! apply simple treatment of deliquesence/crystallization hysteresis
   447)             ! for rhcrystal < rh < rhdeliques, aerosol water is a fraction of
   448)             ! the "upper curve" value, and the fraction is a linear function of rh
   449)             if (rh(i,k) < rhcrystal(m)) then
   450)                wetrad(i,k,m) = dryrad(i,k,m)
   451)                wetvol(i,k,m) = dryvol(i,k,m)
   452)                wtrvol(i,k,m) = 0.0_r8
   453)             else if (rh(i,k) < rhdeliques(m)) then
   454)                wtrvol(i,k,m) = wtrvol(i,k,m)*hystfac*(rh(i,k) - rhcrystal(m))
   455)                wtrvol(i,k,m) = max(wtrvol(i,k,m), 0.0_r8)
   456)                wetvol(i,k,m) = dryvol(i,k,m) + wtrvol(i,k,m)
   457)                wetrad(i,k,m) = (wetvol(i,k,m)/pi43)**third
   458)             end if
   459) 
   460)          end do  ! columns
   461)       end do     ! levels
   462) 
   463)    end do ! modes
   464) 
   465) end subroutine modal_aero_wateruptake_sub

   468)       subroutine modal_aero_kohler(   &
   469)           rdry_in, hygro, s, rwet_out, im )
   470) 
   471) ! calculates equlibrium radius r of haze droplets as function of
   472) ! dry particle mass and relative humidity s using kohler solution
   473) ! given in pruppacher and klett (eqn 6-35)
   474) 
   475) ! for multiple aerosol types, assumes an internal mixture of aerosols
   476) 
   477)       implicit none
   478) 
   479) ! arguments
   480)       integer :: im         ! number of grid points to be processed
   481)       real(r8) :: rdry_in(:)    ! aerosol dry radius (m)
   482)       real(r8) :: hygro(:)      ! aerosol volume-mean hygroscopicity (--)
