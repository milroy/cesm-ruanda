    24) 	FUNCTION EpotVal(gLAT,gMLT)
    25) !
    26) !-----------------------------------------------------------------------
    27) ! Return the value of the electric potential in kV at
    28) ! corrected geomagnetic coordinates gLAT (degrees) and gMLT (hours).
    29) !
    30) ! Must first call ReadCoef and SetModel to set up the model coeficients for
    31) ! the desired values of Bt, IMF clock angle, Dipole tilt angle, and SW Vel.
    32) !-----------------------------------------------------------------------
    33) !
    34)         use shr_kind_mod, only: r8 => shr_kind_r8
    35)         implicit none 
    36) !
    37) !-----------------------------Return Value------------------------------
    38) !
    39)         real(r8) EpotVal
    40) !
    41) !-------------------------------Commons---------------------------------
    42) !
    43) 	INTEGER ML,MM
    44) 	REAL(r8) Coef(0:1,0:8,0:3),pi
    45) 	COMMON/SetCoef/Coef,pi,ML,MM
    46) !
    47) !------------------------------Arguments--------------------------------
    48) !
    49) 	REAL(r8) gLAT,gMLT
    50) !
    51) !---------------------------Local variables-----------------------------
    52) !
    53)         integer limit,l,m
    54) 
    55) 	Real(r8) Theta,Phi,Z,ct,Phim
    56)         real(r8) r
    57) 	REAL(r8) Plm(0:20,0:20)
    58) !
    59) !-----------------------------------------------------------------------
    60) !
    61) 	r=90._r8-gLAT
    62) 	IF(r .LT. 45._r8)THEN
    63) 	  Theta=r*pi/45._r8
    64)           Phi=gMLT*pi/12._r8
    65) 	  Z=Coef(0,0,0)
    66) 	  ct=COS(Theta)
    67) 	  CALL Legendre(ct,ML,MM,Plm)
    68) 	  DO l=1,ML
    69) 	    Z=Z + Coef(0,l,0)*Plm(l,0)
    70) 	    IF(l.LT.MM)THEN
    71) 	      limit=l
    72) 	    ELSE
    73) 	      limit=MM
    74) 	    ENDIF
    75) 	    DO m=1,limit
    76) 	      phim=phi*m
    77)               Z=Z + Coef(0,l,m)*Plm(l,m)*COS(phim) + &
    78)      		   Coef(1,l,m)*Plm(l,m)*SIN(phim)
    79) 	    ENDDO
    80) 	  ENDDO
    81) 	ELSE
    82) 	  Z=0._r8
    83) 	ENDIF
    84) 	EpotVal=Z
    85) 	RETURN
    86) 	END FUNCTION EpotVal

    90) 	SUBROUTINE ReadCoef (wei96_file)
    91) !
    92) !-----------------------------------------------------------------------
    93) !
    94) ! Read in the data file with the model coefficients
    95) !
    96) !*********************** Copyright 1996, Dan Weimer/MRC ***********************
    97) !
    98) ! NCAR addition (Jan 97):  initialize constants used in GECMP
    99) !-----------------------------------------------------------------------
   100) !
   101)       use shr_kind_mod,  only: r8 => shr_kind_r8
   102)       use ioFileMod,     only : getfil
   103)       use units,         only : getunit, freeunit
   104)       use abortutils,    only : endrun
   105)       use cam_logfile,   only : iulog
   106)       implicit none 
   107) !
   108) !-------------------------------Commons---------------------------------
   109) !
   110)       real(r8) alamn, alamx, alamr, stpd, stp2, cstp, sstp
   111)       COMMON /CECMP/ ALAMN,ALAMX,ALAMR,STPD,STP2,CSTP,SSTP
   112) !            ALAMN = Absolute min latitude (deg) of model
   113) !            ALAMX = Absolute max latitude (deg) for normal gradient calc.
   114) !            STPD  = Angular dist (deg) of step @ 300km above earth (r=6371km)
   115) !            STP2  = Denominator in gradient calc
   116) 
   117) !
   118) !------------------------------Arguments--------------------------------
   119) !
   120)       character(len=*), intent(in) :: wei96_file
   121) !
   122) !-----------------------------Parameters------------------------------
   123) !
   124)       real(r8) d2r, r2d
   125)       PARAMETER ( D2R =  0.0174532925199432957692369076847_r8 , &
   126)                   R2D = 57.2957795130823208767981548147_r8)
   127) !
   128) !---------------------------Local variables-----------------------------
   129) !
   130)       INTEGER udat,unit,ios
   131)       integer ll, mm, k, m, klimit, kk, nn, ii, i, n, ilimit, mlimit, l
   132) 
   133)       REAL(r8) C(0:3)
   134)       real(r8) stpr, step
   135) 
   136)       CHARACTER*15 skip
   137) 
   138)       INTEGER MaxL,MaxM,MaxN
   139)       REAL(r8) Cn( 0:3 , 0:1 , 0:4 , 0:1 , 0:8 , 0:3 )
   140)       COMMON /AllCoefs/Cn,MaxL,MaxM,MaxN
   141) 
   142)       character(len=256) :: locfn
   143)       logical, parameter :: debug = .false.
   144) !
   145) !-----------------------------------------------------------------------
   146) !
   147)       STEP = 10._r8
   148)       STPR = STEP/6671._r8
   149)       STPD = STPR*R2D
   150)       STP2 = 2._r8*STEP
   151)       CSTP = COS (STPR)
   152)       SSTP = SQRT (1._r8 - CSTP*CSTP)
   153)       ALAMN = 45._r8
   154)       ALAMX = 90._r8 - STPD
   155)       ALAMR = ALAMN*D2R
   156) !          End NCAR addition
   157) ! 
   158) !  get coeff_file  
   159)       unit= getunit()
   160)       if (debug) write(iulog,*) 'Weimer: getting file ',trim(wei96_file),' unit ',unit
   161)       call getfil( wei96_file, locfn, 0 )
   162) !      
   163)       if (debug) write(iulog,*) 'Weimer: opening file ',trim(locfn),' unit ',unit	
   164)       OPEN(unit=unit,file=trim(locfn), &
   165)            status = 'old',iostat = ios)
   166)       if(ios.gt.0) then
   167) 	write(iulog,*) 'Weimer: error in opening wei96.cofcnts',' unit ',unit
   168)         call endrun
   169)       endif
   170)       
   171)   900 FORMAT(A15)
   172)  1000 FORMAT(3I8)
   173)  2000 FORMAT(3I2)
   174)  3000 FORMAT(2I2,4E15.6)
   175) 
   176) !     READ(udat,900) skip
   177)       if (debug) write(iulog,*) 'Weimer: reading file ',trim(locfn),' unit ',unit	
   178)       READ(unit,1000,iostat = ios) MaxL,MaxM,MaxN
   179)       if(ios.gt.0) then
   180) 	write(iulog,*) 'ReadCoef: error in reading wei96.cofcnts file',' unit ',unit	
   181)         call endrun
   182)       endif
   183)       DO l=0,MaxL
   184)         IF(l.LT.MaxM)THEN
   185)           mlimit=l
   186)         ELSE
   187)           mlimit=MaxM
   188)         ENDIF
   189)         DO m=0,mlimit
   190)           IF(m.LT.1)THEN
   191)             klimit=0
   192)           ELSE
   193)             klimit=1
   194)           ENDIF
   195)           DO k=0,klimit
   196)             READ(unit,2000,iostat = ios) ll,mm,kk
   197)             if(ios.gt.0) then
   198)       	      write(iulog,*) 'ReadCoef: error in reading wei96.cofcnts file',' unit ',unit	
   199)               call endrun
   200)             endif
   201)             IF(ll.NE.l .OR. mm.NE.m .OR. kk.NE.k)THEN
   202)               WRITE(IULOG,*)'Data File Format Error'
   203)               CALL ENDRUN
   204)             ENDIF
   205)             DO n=0,MaxN
   206)               IF(n.LT.1)THEN
   207)         	ilimit=0
   208)               ELSE
   209)         	ilimit=1
   210)               ENDIF
   211)               DO i=0,ilimit
   212)         	READ(unit,3000,iostat = ios) nn,ii,C
   213)                 if(ios.gt.0) then
   214)       	          write(iulog,*) 'ReadCoef: error in reading', &
   215)                        ' wei96.cofcnts file',' unit ',unit	
   216)                   call endrun
   217)                 endif
   218)         	IF(nn.NE.n .OR. ii.NE.i)THEN
   219)         	  WRITE(IULOG,*)'Data File Format Error'
   220)         	  CALL ENDRUN
   221)         	ENDIF
   222)         	Cn(0,i,n,k,l,m)=C(0)
   223)         	Cn(1,i,n,k,l,m)=C(1)
   224)         	Cn(2,i,n,k,l,m)=C(2)
   225)         	Cn(3,i,n,k,l,m)=C(3)
   226)               ENDDO
   227)             ENDDO
   228)           ENDDO
   229)         ENDDO
   230)       ENDDO
   231) !      
   232)       close(unit)
   233)       call freeunit(unit)
   234) !    
   235)       RETURN
   236)       END SUBROUTINE ReadCoef

   240) 	FUNCTION FSVal(omega,MaxN,FSC)
   241) !
   242) !-----------------------------------------------------------------------
   243) ! Evaluate a  Sine/Cosine Fourier series for N terms up to MaxN
   244) ! at angle omega, given the coefficients in FSC
   245) !
   246) !*********************** Copyright 1996, Dan Weimer/MRC ***********************
   247) !-----------------------------------------------------------------------
   248) !
   249)         use shr_kind_mod, only: r8 => shr_kind_r8
   250)         implicit none 
   251) !
   252) !-----------------------------Return Value------------------------------
   253) !
   254)         real(r8) FSVal
   255) !
   256) !------------------------------Arguments--------------------------------
   257) !
   258) 	INTEGER MaxN
   259) 	REAL(r8) omega,FSC(0:1,0:*)
   260) !
   261) !---------------------------Local variables-----------------------------
   262) !
   263) 	INTEGER n
   264) 	REAL(r8) Y,theta
   265) !
   266) !-----------------------------------------------------------------------
   267) !
   268) 	Y=0._r8
   269) 	DO n=0,MaxN
   270) 	  theta=omega*n
   271) 	  Y=Y + FSC(0,n)*COS(theta) + FSC(1,n)*SIN(theta)
   272) 	ENDDO
   273) 	FSVal=Y
   274) 	RETURN
   275) 	END FUNCTION FSVal

   279) 	SUBROUTINE SetModel(angle,Bt,Tilt,SWVel)
   280) !
   281) !-----------------------------------------------------------------------
   282) ! Calculate the complete set of spherical harmonic coefficients,
   283) ! given an arbitrary IMF angle (degrees from northward toward +Y),
   284) ! magnitude Bt (nT), dipole tilt angle (degrees),
   285) ! and solar wind velocity (km/sec).
   286) ! Returns the Coef in the common block SetCoef.
   287) !
   288) !*********************** Copyright 1996, Dan Weimer/MRC ***********************
   289) !-----------------------------------------------------------------------
   290) !
   291)         use shr_kind_mod, only: r8 => shr_kind_r8
   292)         implicit none 
   293) !
   294) !-------------------------------Commons---------------------------------
   295) !
   296) 	INTEGER MaxL,MaxM,MaxN
   297) 	REAL(r8) Cn( 0:3 , 0:1 , 0:4 , 0:1 , 0:8 , 0:3 )
   298) 	COMMON /AllCoefs/Cn,MaxL,MaxM,MaxN
   299) 
   300) 	INTEGER ML,MM
   301) 	REAL(r8) Coef(0:1,0:8,0:3),pi
   302) 	COMMON/SetCoef/Coef,pi,ML,MM
   303) !
   304) !------------------------------Arguments--------------------------------
   305) !
   306) 	REAL(r8) angle,Bt,Tilt,SWVel
   307) !
   308) !---------------------------Local variables-----------------------------
   309) !
   310)         integer n, k, ilimit, i, klimit, l, m, mlimit
   311) 	REAL(r8) FSC(0:1,0:4), fsval, omega, sintilt
   312) !
   313) !-----------------------------------------------------------------------
   314) !
   315) 	pi=2._r8*ASIN(1._r8)
   316) 	ML=MaxL
   317) 	MM=MaxM
   318) 	SinTilt=SIN(Tilt*pi/180._r8)
   319) !	SinTilt=SIND(Tilt)
   320) 
   321) 	omega=angle*pi/180._r8
   322) 
   323)         fsc(1,0) = 0._r8
   324) 	DO l=0,MaxL
   325) 	  IF(l.LT.MaxM)THEN
   326) 	    mlimit=l
   327) 	  ELSE
   328) 	    mlimit=MaxM
   329) 	  ENDIF
   330) 	  DO m=0,mlimit
   331) 	    IF(m.LT.1)THEN
   332) 	      klimit=0
   333) 	    ELSE
   334) 	      klimit=1
   335) 	    ENDIF
   336) 	    DO k=0,klimit
   337) ! Retrieve the regression coefficients and evaluate the function
   338) ! as a function of Bt,Tilt,and SWVel to get each Fourier coefficient.
   339) 	      DO n=0,MaxN
   340) 	        IF(n.LT.1)THEN
   341) 	          ilimit=0
   342) 	        ELSE
   343) 	          ilimit=1
   344) 	        ENDIF
   345) 		DO i=0,ilimit
   346) 		  FSC(i,n)=Cn(0,i,n,k,l,m) + Bt*Cn(1,i,n,k,l,m) + &
   347)      		   SinTilt*Cn(2,i,n,k,l,m) + SWVel*Cn(3,i,n,k,l,m)
   348) 		ENDDO
   349) 	      ENDDO
   350) ! Next evaluate the Fourier series as a function of angle.
   351)       	      Coef(k,l,m)=FSVal(omega,MaxN,FSC)
   352) 	    ENDDO
   353) 	  ENDDO
   354) 	ENDDO
   355) 	RETURN
   356) 	END SUBROUTINE SetModel

   360) 	SUBROUTINE LEGENDRE(x,lmax,mmax,Plm)
   361) !
   362) !-----------------------------------------------------------------------
   363) ! compute Associate Legendre Function P_l^m(x)
   364) ! for all l up to lmax and all m up to mmax.
   365) ! returns results in array Plm
   366) ! if X is out of range ( abs(x)>1 ) then value is returned as if x=1.
   367) !
   368) !*********************** Copyright 1996, Dan Weimer/MRC ***********************
   369) !-----------------------------------------------------------------------
   370) !
   371)         use shr_kind_mod, only: r8 => shr_kind_r8
   372)         use cam_logfile,  only : iulog
   373) 
   374)         implicit none 
   375) !
   376) !------------------------------Arguments--------------------------------
   377) !
   378)         integer lmax, mmax
   379) 	real(r8) x, Plm(0:20,0:20)
   380) !
   381) !---------------------------Local variables-----------------------------
   382) !
   383)         integer m, lm2, l
   384)         real(r8) xx, fact
   385) !
   386) !-----------------------------------------------------------------------
   387) !
   388) 	  DO l=0,20
   389) 	    DO m=0,20
   390) 		Plm(l,m)=0._r8
   391) 	    ENDDO
   392) 	  ENDDO
   393) 	xx=MIN(x,1._r8)
   394) 	xx=MAX(xx,-1._r8)
   395) 	IF(lmax .LT. 0 .OR. mmax .LT. 0 .OR. mmax .GT. lmax )THEN
   396) 	  write(iulog,*)'Bad arguments to Legendre'
   397) 	  RETURN
   398) 	ENDIF
   399) ! First calculate all Pl0 for l=0 to l
   400) 	Plm(0,0)=1._r8
   401) 	IF(lmax.GT.0)Plm(1,0)=xx
   402) 	IF (lmax .GT. 1 )THEN
   403) 	  DO L=2,lmax
   404) 	    Plm(L,0)=( (2._r8*L-1)*xx*Plm(L-1,0) - (L-1)*Plm(L-2,0) )/L
   405) 	  ENDDO
   406) 	ENDIF
   407) 	IF (mmax .EQ. 0 )RETURN
   408) 	fact=SQRT( (1._r8-xx)*(1._r8+xx) )
   409) 	DO M=1,mmax
   410) 	  DO L=m,lmax
   411) 	    lm2=MAX(L-2,0)
   412) 	    Plm(L,M)=Plm(lm2,M) - ( 2*L-1)*fact*Plm(L-1,M-1)
   413) 	  ENDDO
   414) 	ENDDO
   415) 	RETURN
   416) 	END SUBROUTINE LEGENDRE

   431) 	FUNCTION GET_TILT(YEAR,MONTH,DAY,HOUR)
   432) !
   433) !-----------------------------------------------------------------------
   434) !CC NCAR MODIFIED (3/96) CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC
   435) !  The following line initially was:                                           C
   436) !       SUBROUTINE TRANS(YEAR,MONTH,DAY,HOUR,IDBUG)                            C
   437) !  It has been changed to return the dipole tilt from this function call.      C
   438) !CC NCAR MODIFIED (3/96) CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC
   439) !         
   440) !      THIS SUBROUTINE DERIVES THE ROTATION MATRICES AM(I,J,K) FOR 11
   441) !      TRANSFORMATIONS, IDENTIFIED BY K.
   442) !          K=1 TRANSFORMS GSE to GEO
   443) !          K=2     "      GEO to MAG
   444) !          K=3     "      GSE to MAG
   445) !          K=4     "      GSE to GSM
   446) !          K=5     "      GEO to GSM
   447) !          K=6     "      GSM to MAG
   448) !          K=7     "      GSE to GEI
   449) !          K=8     "      GEI to GEO
   450) !          K=9     "      GSM to SM 
   451) !	   K=10    "      GEO to SM 
   452) !	   K=11    "      MAG to SM 
   453) !
   454) !      IF IDBUG IS NOT 0, THEN OUTPUTS DIAGNOSTIC INFORMATION TO
   455) !      FILE UNIT=IDBUG
   456) !
   457) !      The formal names of the coordinate systems are:
   458) !	GSE - Geocentric Solar Ecliptic
   459) !	GEO - Geographic
   460) !	MAG - Geomagnetic
   461) !	GSM - Geocentric Solar Magnetospheric
   462) !	SM  - Solar Magnetic
   463) !	
   464) !      THE ARRAY CX(I) ENCODES VARIOUS ANGLES, STORED IN DEGREES
   465) !      ST(I) AND CT(I) ARE SINES & COSINES.       
   466) !
   467) !      Program author:  D. R. Weimer
   468) !
   469) !      Some of this code has been copied from subroutines which had been
   470) !      obtained from D. Stern, NASA/GSFC.  Other formulas are from "Space 
   471) !      Physics Coordinate Transformations: A User Guide" by M. Hapgood (1991).
   472) !
   473) !      The formulas for the calculation of Greenwich mean sidereal time (GMST)
   474) !      and the sun's location are from "Almanac for Computers 1990",
   475) !      U.S. Naval Observatory.
   476) !
   477) !-----------------------------------------------------------------------
   478) !
   479)         use shr_kind_mod, only: r8 => shr_kind_r8
   480)         implicit none 
   481) !
   482) !-----------------------------Return Value------------------------------
   483) !
   484)         real(r8) get_tilt
   485) !
   486) !-------------------------------Commons---------------------------------
   487) !
   488)         real(r8) cx, st, ct, am
   489) 	COMMON/TRANSDAT/CX(9),ST(6),CT(6),AM(3,3,11)
   490) 
   491)         real(r8) :: th0=11.19_r8
   492)         real(r8) :: ph0=-70.76_r8
   493) !
   494) !------------------------------Arguments--------------------------------
   495) !
   496) 	INTEGER YEAR, MONTH, DAY
   497) 	REAL(r8) HOUR
   498) !
   499) !-----------------------------Parameters------------------------------
   500) !
   501) 	INTEGER GSEGEO,GEOGSE,GEOMAG,MAGGEO
   502) 	INTEGER GSEMAG,MAGGSE,GSEGSM,GSMGSE
   503) 	INTEGER GEOGSM,GSMGEO,GSMMAG,MAGGSM
   504) 	INTEGER GSEGEI,GEIGSE,GEIGEO,GEOGEI
   505) 	INTEGER GSMSM,SMGSM,GEOSM,SMGEO,MAGSM,SMMAG
   506) 
   507) 	PARAMETER (GSEGEO= 1,GEOGSE=-1,GEOMAG= 2,MAGGEO=-2)
   508) 	PARAMETER (GSEMAG= 3,MAGGSE=-3,GSEGSM= 4,GSMGSE=-4)
   509) 	PARAMETER (GEOGSM= 5,GSMGEO=-5,GSMMAG= 6,MAGGSM=-6)
   510) 	PARAMETER (GSEGEI= 7,GEIGSE=-7,GEIGEO= 8,GEOGEI=-8)
   511) 	PARAMETER (GSMSM = 9,SMGSM =-9,GEOSM =10,SMGEO=-10)
   512) 	PARAMETER (MAGSM =11,SMMAG =-11)
   513) !
   514) !---------------------------Local variables-----------------------------
   515) !
   516)         integer IDBUG
   517)         integer j, k, jd, iyr, i, mjd
   518) 
   519)         REAL(r8) UT, T0, GMSTD, GMSTH, ECLIP, MA, LAMD, SUNLON, pi
   520)         real(r8) b32, b33, b3
   521) !
   522) !-------------------------External Functions----------------------------
   523) !
   524)         integer julday
   525)         external julday
   526) !
   527) !-----------------------------------------------------------------------
   528) !
   529) 	pi=2._r8*ASIN(1._r8)
   530) !CC NCAR MODIFICATION (3/96) CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC
   531) ! IDBUG=0 to prevent printing data to the screen or writing data to a file.    C
   532)         IDBUG = 0
   533) !CC NCAR MODIFICATION (3/96) CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC
   534) 
   535) 	IF(YEAR.LT.1900)THEN
   536) 	  IYR=1900+YEAR
   537) 	ELSE
   538) 	  IYR=YEAR
   539) 	ENDIF
   540) 	UT=HOUR
   541) 	JD=JULDAY(MONTH,DAY,IYR)
   542) 	MJD=JD-2400001
   543) 	T0=(real(MJD,r8)-51544.5_r8)/36525.0_r8
   544) 	GMSTD=100.4606184_r8 + 36000.770_r8*T0 + 3.87933E-4_r8*T0*T0 + &
   545)              15.0410686_r8*UT
   546) 	CALL ADJUST(GMSTD)
   547) 	GMSTH=GMSTD*24._r8/360._r8
   548) 	ECLIP=23.439_r8 - 0.013_r8*T0
   549)         MA=357.528_r8 + 35999.050_r8*T0 + 0.041066678_r8*UT
   550)         CALL ADJUST(MA)
   551)         LAMD=280.460_r8 + 36000.772_r8*T0 + 0.041068642_r8*UT
   552)         CALL ADJUST(LAMD)
   553)         SUNLON=LAMD + (1.915_r8-0.0048_r8*T0)*SIN(MA*pi/180._r8) + 0.020_r8* &
   554)            SIN(2._r8*MA*pi/180._r8)
   555)         CALL ADJUST(SUNLON)
   556) 	IF(IDBUG.NE.0)THEN
   557) 	  WRITE(IDBUG,*) YEAR,MONTH,DAY,HOUR
   558) 	  WRITE(IDBUG,*) 'MJD=',MJD
   559) 	  WRITE(IDBUG,*) 'T0=',T0
   560) 	  WRITE(IDBUG,*) 'GMSTH=',GMSTH
   561) 	  WRITE(IDBUG,*) 'ECLIPTIC OBLIQUITY=',ECLIP
   562) 	  WRITE(IDBUG,*) 'MEAN ANOMALY=',MA
   563) 	  WRITE(IDBUG,*) 'MEAN LONGITUDE=',LAMD
   564) 	  WRITE(IDBUG,*) 'TRUE LONGITUDE=',SUNLON
   565) 	ENDIF
   566) 
   567) 	CX(1)= GMSTD
   568) 	CX(2) = ECLIP
   569) 	CX(3) = SUNLON
   570) 	CX(4) = TH0
   571) 	CX(5) = PH0
   572) ! Derived later:
   573) !       CX(6) = Dipole tilt angle  
   574) !       CX(7) = Angle between sun and magnetic pole
   575) !       CX(8) = Subsolar point latitude
   576) !       CX(9) = Subsolar point longitude
   577) 
   578) 	DO I=1,5
   579) 	  ST(I) = SIN(CX(I)*pi/180._r8)
   580) 	  CT(I) = COS(CX(I)*pi/180._r8)
   581) 	ENDDO
   582) !         
   583)       AM(1,1,GSEGEI) = CT(3)
   584)       AM(1,2,GSEGEI) = -ST(3)
   585)       AM(1,3,GSEGEI) = 0._r8         
   586)       AM(2,1,GSEGEI) = ST(3)*CT(2)
   587)       AM(2,2,GSEGEI) = CT(3)*CT(2)
   588)       AM(2,3,GSEGEI) = -ST(2)
   589)       AM(3,1,GSEGEI) = ST(3)*ST(2)
   590)       AM(3,2,GSEGEI) = CT(3)*ST(2)
   591)       AM(3,3,GSEGEI) = CT(2)      
   592) !         
   593)       AM(1,1,GEIGEO) = CT(1)      
   594)       AM(1,2,GEIGEO) = ST(1)      
   595)       AM(1,3,GEIGEO) = 0._r8         
   596)       AM(2,1,GEIGEO) = -ST(1)     
   597)       AM(2,2,GEIGEO) = CT(1)      
   598)       AM(2,3,GEIGEO) = 0._r8         
   599)       AM(3,1,GEIGEO) = 0._r8         
   600)       AM(3,2,GEIGEO) = 0._r8         
   601)       AM(3,3,GEIGEO) = 1._r8         
   602) !         
   603)       DO I=1,3   
   604)       DO J=1,3   
   605)         AM(I,J,GSEGEO) = AM(I,1,GEIGEO)*AM(1,J,GSEGEI) + &
   606)          AM(I,2,GEIGEO)*AM(2,J,GSEGEI) +  AM(I,3,GEIGEO)*AM(3,J,GSEGEI)
   607)       ENDDO
   608)       ENDDO
   609) !         
   610)       AM(1,1,GEOMAG) = CT(4)*CT(5) 
   611)       AM(1,2,GEOMAG) = CT(4)*ST(5) 
   612)       AM(1,3,GEOMAG) =-ST(4)       
   613)       AM(2,1,GEOMAG) =-ST(5)       
   614)       AM(2,2,GEOMAG) = CT(5)       
   615)       AM(2,3,GEOMAG) = 0._r8
   616)       AM(3,1,GEOMAG) = ST(4)*CT(5) 
   617)       AM(3,2,GEOMAG) = ST(4)*ST(5) 
   618)       AM(3,3,GEOMAG) = CT(4)       
   619) !         
   620)       DO I=1,3   
   621)       DO J=1,3   
   622)        AM(I,J,GSEMAG) = AM(I,1,GEOMAG)*AM(1,J,GSEGEO) + &
   623)         AM(I,2,GEOMAG)*AM(2,J,GSEGEO) +  AM(I,3,GEOMAG)*AM(3,J,GSEGEO)
   624)       ENDDO
   625)       ENDDO
   626) !         
   627)       B32 = AM(3,2,GSEMAG)         
   628)       B33 = AM(3,3,GSEMAG)         
   629)       B3  = SQRT(B32*B32+B33*B33)       
   630)       IF (B33.LE.0._r8) B3 = -B3  
   631) !         
   632)       AM(2,2,GSEGSM) = B33/B3      
   633)       AM(3,3,GSEGSM) = AM(2,2,GSEGSM)   
   634)       AM(3,2,GSEGSM) = B32/B3      
   635)       AM(2,3,GSEGSM) =-AM(3,2,GSEGSM)   
   636)       AM(1,1,GSEGSM) = 1._r8
   637)       AM(1,2,GSEGSM) = 0._r8
   638)       AM(1,3,GSEGSM) = 0._r8
   639)       AM(2,1,GSEGSM) = 0._r8
   640)       AM(3,1,GSEGSM) = 0._r8
   641) !         
   642)       DO I=1,3   
   643)       DO J=1,3   
   644)         AM(I,J,GEOGSM) = AM(I,1,GSEGSM)*AM(J,1,GSEGEO) + &
   645)          AM(I,2,GSEGSM)*AM(J,2,GSEGEO) + AM(I,3,GSEGSM)*AM(J,3,GSEGEO)
   646)       ENDDO
   647)       ENDDO
   648) !         
   649)       DO I=1,3   
   650)       DO J=1,3   
   651)         AM(I,J,GSMMAG) = AM(I,1,GEOMAG)*AM(J,1,GEOGSM) + &
   652)         AM(I,2,GEOMAG)*AM(J,2,GEOGSM) + AM(I,3,GEOMAG)*AM(J,3,GEOGSM)
   653)       ENDDO
   654)       ENDDO 
   655) !
   656) 	ST(6) = AM(3,1,GSEMAG)        
   657) 	CT(6) = SQRT(1._r8-ST(6)*ST(6))      
   658) 	CX(6) = ASIN(ST(6)*pi/180._r8)  
   659) 
   660)         AM(1,1,GSMSM) = CT(6)
   661)         AM(1,2,GSMSM) = 0._r8
   662)         AM(1,3,GSMSM) = -ST(6)
   663)         AM(2,1,GSMSM) = 0._r8
   664)         AM(2,2,GSMSM) = 1._r8
   665)         AM(2,3,GSMSM) = 0._r8
   666)         AM(3,1,GSMSM) = ST(6)
   667)         AM(3,2,GSMSM) = 0._r8
   668)         AM(3,3,GSMSM) = CT(6)  
   669) !         
   670)       DO I=1,3   
   671)       DO J=1,3   
   672)         AM(I,J,GEOSM) = AM(I,1,GSMSM)*AM(1,J,GEOGSM) + &
   673)          AM(I,2,GSMSM)*AM(2,J,GEOGSM) +  AM(I,3,GSMSM)*AM(3,J,GEOGSM)
   674)       ENDDO
   675)       ENDDO
   676) !         
   677)       DO I=1,3   
   678)       DO J=1,3   
   679)         AM(I,J,MAGSM) = AM(I,1,GSMSM)*AM(J,1,GSMMAG) + &
   680)         AM(I,2,GSMSM)*AM(J,2,GSMMAG) + AM(I,3,GSMSM)*AM(J,3,GSMMAG)
   681)       ENDDO
   682)       ENDDO
   683)       
   684) !
   685)       CX(7)=ATAN2( AM(2,1,11) , AM(1,1,11) )
   686)       
   687)       CX(7)=CX(7)*180._r8/pi
   688)       CX(8)=ASIN( AM(3,1,1)*pi/180._r8 )
   689)       CX(9)=ATAN2( AM(2,1,1) , AM(1,1,1) )
   690)       CX(9)=CX(9)*180._r8/pi
   691) 
   692)       IF(IDBUG.NE.0)THEN
   693) 	  WRITE(IDBUG,*) 'Dipole tilt angle=',CX(6)
   694) 	  WRITE(IDBUG,*) 'Angle between sun and magnetic pole=',CX(7)
   695) 	  WRITE(IDBUG,*) 'Subsolar point latitude=',CX(8)
   696) 	  WRITE(IDBUG,*) 'Subsolar point longitude=',CX(9)
   697) 
   698)         DO K=1,11
   699)          WRITE(IDBUG,1001) K
   700)          DO I=1,3
   701)            WRITE(IDBUG,1002) (AM(I,J,K),J=1,3)
   702)          ENDDO
   703)         ENDDO
   704)  1001   FORMAT(' ROTATION MATRIX ',I2)
   705)  1002   FORMAT(3F9.5)
   706)       ENDIF
   707) 
   708) !CC NCAR MODIFICATION (3/96) CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC
   709) !  The next line was added to return the dipole tilt from this function call.  C
   710)       GET_TILT = CX(6)
   711) !CC NCAR MODIFICATION (3/96) CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC

   713)       RETURN

   718)       SUBROUTINE ROTATE (X,Y,Z,I)       
   719) !
   720) !-----------------------------------------------------------------------
   721) !     THIS SUBROUTINE APPLIES TO THE VECTOR (X,Y,Z) THE ITH ROTATION  
   722) !     MATRIX AM(N,M,I) GENERATED BY SUBROUTINE TRANS
   723) !     IF I IS NEGATIVE, THEN THE INVERSE ROTATION IS APPLIED

   753)       SUBROUTINE ROTATEV (A,B,I)       
   754) !         
   755) !-----------------------------------------------------------------------
   756) !     THIS SUBROUTINE APPLIES TO THE VECTOR A(3) THE ITH ROTATION  
   757) !     MATRIX AM(N,M,I) GENERATED BY SUBROUTINE TRANS
   758) !     AND OUTPUTS THE CONVERTED VECTOR B(3), WITH NO CHANGE TO A.
   759) !     IF I IS NEGATIVE, THEN THE INVERSE ROTATION IS APPLIED
   760) !-----------------------------------------------------------------------
   761) !
   762)       use shr_kind_mod, only: r8 => shr_kind_r8
   763)       use cam_logfile,  only : iulog
   764)       use abortutils,   only : endrun
   765) 
   766)       implicit none 
   767) !
   768) !-------------------------------Commons---------------------------------
   769) !
   770)       real(r8) cx, st, ct, am
   771)       COMMON/TRANSDAT/CX(9),ST(6),CT(6),AM(3,3,11)
   772) !
   773) !------------------------------Arguments--------------------------------
   774) !
   775)       integer i
   776)       REAL(r8) A(3),B(3)
   777) !
   778) !---------------------------Local variables-----------------------------
   779) !
   780)       integer id, j
   781)       real(r8) xa, ya, za
   782) !
   783) !-----------------------------------------------------------------------
   784) !
   785)       IF(I.EQ.0 .OR. IABS(I).GT.11)THEN
   786)       WRITE(IULOG,*)'ROTATEV CALLED WITH UNDEFINED TRANSFORMATION'
   787)       CALL ENDRUN
   788)       ENDIF
   789) 
   790)       XA = A(1)
   791)       YA = A(2)
   792)       ZA = A(3)
   793)       IF(I.GT.0)THEN
   794) 	ID=I
   795)         DO J=1,3
   796)           B(J) = XA*AM(J,1,ID) + YA*AM(J,2,ID) + ZA*AM(J,3,ID)
   797)         ENDDO
   798)       ELSE
   799) 	ID=-I
   800)         DO J=1,3
   801)           B(J) = XA*AM(1,J,ID) + YA*AM(2,J,ID) + ZA*AM(3,J,ID)
   802)         ENDDO
   803)       ENDIF
   804)       RETURN        
   805)       END SUBROUTINE ROTATEV

   809) 	SUBROUTINE FROMCART(R,LAT,LONG,POS)
   810) !
   811) !-----------------------------------------------------------------------
   812) ! CONVERT CARTESIAN COORDINATES POS(3)
   813) ! TO SPHERICAL COORDINATES R, LATITUDE, AND LONGITUDE (DEGREES)
   814) !-----------------------------------------------------------------------
   815) !
   816)         use shr_kind_mod, only: r8 => shr_kind_r8
   817)         implicit none 
   818) !
   819) !------------------------------Arguments--------------------------------
   820) !
   821) 	REAL(r8) R, LAT, LONG, POS(3)

   823) !---------------------------Local variables-----------------------------
   824) !
   825)         real(r8) pi
   826) !
   827) !-----------------------------------------------------------------------
   828) !
   829) 	pi=2._r8*ASIN(1._r8)
   830) 	R=SQRT(POS(1)*POS(1) + POS(2)*POS(2) + POS(3)*POS(3))
   831) 	IF(R.EQ.0._r8)THEN
   832) 	  LAT=0._r8
   833) 	  LONG=0._r8
   834) 	ELSE
   835) 	  LAT=ASIN(POS(3)*pi/180._r8/R)
   836) 	  LONG=ATAN2(POS(2),POS(1))
   837) 	  LONG=LONG*180._r8/pi
   838) 	ENDIF
   839) 	RETURN
   840) 	END SUBROUTINE FROMCART

   845) !
   846) !-----------------------------------------------------------------------
   847) ! CONVERT SPHERICAL COORDINATES R, LATITUDE, AND LONGITUDE (DEGREES)

  1064)       SUBROUTINE GECMP (AMLA,RMLT,ET,EP)
  1065) !
  1066) !-----------------------------------------------------------------------
  1067) !          Get Electric field components for the Weimer electrostatic
  1068) !          potential model.  Before use, first load coefficients (CALL
  1069) !          READCOEF) and initialize model conditions (CALL SETMODEL).
  1070) !
  1071) !          INPUTS:
  1072) !            AMLA = Absolute value of magnetic latitude (deg)
  1073) !            RMLT = Magnetic local time (hours).
  1074) !          RETURNS:
  1075) !            ET = Etheta (magnetic equatorward*) E field component (V/m)
  1076) !            EP = Ephi   (magnetic eastward)     E field component (V/m)
  1077) !
  1078) !          * ET direction is along the magnetic meridian away from the
  1079) !            current hemisphere; i.e., when ET > 0, the direction is
  1080) !              southward when RMLA > 0
  1081) !              northward when RMLA < 0
  1082) !
  1083) !          NCAR addition (Jan 97).  R.Barnes
  1084) !-----------------------------------------------------------------------
  1085) !
  1086)       use shr_kind_mod, only: r8 => shr_kind_r8
  1087)       implicit none 
  1088) !
  1089) !-------------------------------Commons---------------------------------
  1090) !
  1091) !          CECMP contains constants initialized in READCOEF
  1092)       real(r8) alamn, alamx, alamr, stpd, stp2, cstp, sstp
  1093)       COMMON /CECMP/ ALAMN,ALAMX,ALAMR,STPD,STP2,CSTP,SSTP
  1094) !
