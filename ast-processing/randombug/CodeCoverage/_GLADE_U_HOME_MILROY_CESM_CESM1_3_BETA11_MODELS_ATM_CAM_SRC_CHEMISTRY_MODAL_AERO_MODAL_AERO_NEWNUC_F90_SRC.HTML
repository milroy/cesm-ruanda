   791)             if (so4vol_in >= 1.0e4_r8) then

   793)                rh_bb = max( 1.0e-4_r8, min( 1.0_r8, rh_in ) )
   794)                so4vol_bb = max( 1.0e4_r8, min( 1.0e11_r8, so4vol_in ) )
   795)                call binary_nuc_vehk2002(   &
   796)                   temp_bb, rh_bb, so4vol_bb,   &
   797)                   ratenuclt, rateloge,   &
   798)                   cnum_h2so4, cnum_tot, radius_cluster )
   799)             end if
   800)             cnum_nh3 = 0.0_r8
   801)             newnuc_method_flagaa2 = 2
   802) 
   803)         end if
   804) 
   805) 
   806) ! do boundary layer nuc
   807)         if ((newnuc_method_flagaa == 11) .or.   &
   808)             (newnuc_method_flagaa == 12)) then
   809)            if ( zm_in <= max(pblh_in,100.0_r8) ) then
   810)               so4vol_bb = so4vol_in
   811)               call pbl_nuc_wang2008( so4vol_bb,   &
   812)                  newnuc_method_flagaa, newnuc_method_flagaa2,   &
   813)                  ratenuclt, rateloge,   &
   814)                  cnum_tot, cnum_h2so4, cnum_nh3, radius_cluster )
   815)            end if
   816)         end if
   817) 
   818) 
   819) ! if nucleation rate is less than 1e-6 #/m3/s ~= 0.1 #/cm3/day,
   820) ! exit with new particle formation = 0
   821)         if (rateloge  .le. -13.82_r8) return
   822) !       if (ratenuclt .le. 1.0e-6) return
   823)         ratenuclt = exp( rateloge )
   824)         ratenuclt_bb = ratenuclt*1.0e6_r8
   825) 
   826) 
   827) ! wet/dry volume ratio - use simple kohler approx for ammsulf/ammbisulf
   828)         tmpa = max( 0.10_r8, min( 0.95_r8, rh_in ) )
   829)         wetvol_dryvol = 1.0_r8 - 0.56_r8/log(tmpa)
   830) 
   831) 
   832) ! determine size bin into which the new particles go
   833) ! (probably it will always be bin #1, but ...)
   834)         voldry_clus = ( max(cnum_h2so4,1.0_r8)*mw_so4a + cnum_nh3*mw_nh4a ) /   &
   835)                       (1.0e3_r8*dens_sulfacid*avogad)
   836) ! correction when host code sulfate is really ammonium bisulfate/sulfate
   837)         voldry_clus = voldry_clus * (mw_so4a_host/mw_so4a)
   838)         dpdry_clus = (voldry_clus*6.0_r8/pi)**onethird
   839) 
   840)         isize_nuc = 1
   841)         dpdry_part = dplom_sect(1)
   842)         if (dpdry_clus <= dplom_sect(1)) then
   843)            igrow = 1   ! need to clusters to larger size
   844)         else if (dpdry_clus >= dphim_sect(nsize)) then
   845)            igrow = 0
   846)            isize_nuc = nsize
   847)            dpdry_part = dphim_sect(nsize)
   848)         else
   849)            igrow = 0
   850)            do i = 1, nsize
   851)               if (dpdry_clus < dphim_sect(i)) then
   852)                  isize_nuc = i
   853)                  dpdry_part = dpdry_clus
   854)                  dpdry_part = min( dpdry_part, dphim_sect(i) )
   855)                  dpdry_part = max( dpdry_part, dplom_sect(i) )
   856)                  exit
   857)               end if
   858)            end do
   859)         end if
   860)         voldry_part = (pi/6.0_r8)*(dpdry_part**3)
   861) 
   862) 
   863) !
   864) ! determine composition and density of the "grown particles"
   865) ! the grown particles are assumed to be liquid
   866) !    (since critical clusters contain water)
   867) !    so any (nh4/so4) molar ratio between 0 and 2 is allowed
   868) ! assume that the grown particles will have 
   869) !    (nh4/so4 molar ratio) = min( 2, (nh3/h2so4 gas molar ratio) )
   870) !
   871)         if (igrow .le. 0) then
   872) ! no "growing" so pure sulfuric acid
   873)            tmp_n1 = 0.0_r8
   874)            tmp_n2 = 0.0_r8
   875)            tmp_n3 = 1.0_r8
   876)         else if (qnh3_cur .ge. qh2so4_cur) then
   877) ! combination of ammonium sulfate and ammonium bisulfate
   878) ! tmp_n1 & tmp_n2 = mole fractions of the ammsulf & ammbisulf
   879)            tmp_n1 = (qnh3_cur/qh2so4_cur) - 1.0_r8
   880)            tmp_n1 = max( 0.0_r8, min( 1.0_r8, tmp_n1 ) )
   881)            tmp_n2 = 1.0_r8 - tmp_n1
   882)            tmp_n3 = 0.0_r8
   883)         else
   884) ! combination of ammonium bisulfate and sulfuric acid
   885) ! tmp_n2 & tmp_n3 = mole fractions of the ammbisulf & sulfacid
   886)            tmp_n1 = 0.0_r8
   887)            tmp_n2 = (qnh3_cur/qh2so4_cur)
   888)            tmp_n2 = max( 0.0_r8, min( 1.0_r8, tmp_n2 ) )
   889)            tmp_n3 = 1.0_r8 - tmp_n2
   890) 	end if
   891) 
   892)         tmp_m1 = tmp_n1*mw_ammsulf
   893)         tmp_m2 = tmp_n2*mw_ammbisulf
   894)         tmp_m3 = tmp_n3*mw_sulfacid
   895)         dens_part = (tmp_m1 + tmp_m2 + tmp_m3)/   &
   896)            ((tmp_m1/dens_ammsulf) + (tmp_m2/dens_ammbisulf)   &
   897)                                   + (tmp_m3/dens_sulfacid))
   898)         dens_nh4so4a = dens_part
   899)         mass_part  = voldry_part*dens_part 
   900) ! (mol aerosol nh4)/(mol aerosol so4)
   901)         molenh4a_per_moleso4a = 2.0_r8*tmp_n1 + tmp_n2  
   902) ! (kg dry aerosol)/(mol aerosol so4)
   903)         kgaero_per_moleso4a = 1.0e-3_r8*(tmp_m1 + tmp_m2 + tmp_m3)  
   904) ! correction when host code sulfate is really ammonium bisulfate/sulfate
   905)         kgaero_per_moleso4a = kgaero_per_moleso4a * (mw_so4a_host/mw_so4a)
   906) 
   907) ! fraction of wet volume due to so4a
   908)         tmpb = 1.0_r8 + molenh4a_per_moleso4a*17.0_r8/98.0_r8
   909)         wet_volfrac_so4a = 1.0_r8 / ( wetvol_dryvol * tmpb )
   910) 
   911) 
   912) !
   913) ! calc kerminen & kulmala (2002) correction
   914) !
   915)         if (igrow <=  0) then
   916)             factor_kk = 1.0_r8
   917) 
   918)         else
   919) ! "gr" parameter (nm/h) = condensation growth rate of new particles
   920) ! use kk2002 eqn 21 for h2so4 uptake, and correct for nh3 & h2o uptake
   921)             tmp_spd = 14.7_r8*sqrt(temp_in)   ! h2so4 molecular speed (m/s)
   922)             gr_kk = 3.0e-9_r8*tmp_spd*mw_sulfacid*so4vol_in/   &
   923)                     (dens_part*wet_volfrac_so4a)
   924) 
   925) ! "gamma" parameter (nm2/m2/h)
   926) ! use kk2002 eqn 22
   927) !
   928) ! dfin_kk = wet diam (nm) of grown particle having dry dia = dpdry_part (m)
   929)             dfin_kk = 1.0e9_r8 * dpdry_part * (wetvol_dryvol**onethird)
   930) ! dnuc_kk = wet diam (nm) of cluster
   931)             dnuc_kk = 2.0_r8*radius_cluster
   932)             dnuc_kk = max( dnuc_kk, 1.0_r8 )
   933) ! neglect (dmean/150)**0.048 factor, 
   934) ! which should be very close to 1.0 because of small exponent
   935)             gamma_kk = 0.23_r8 * (dnuc_kk)**0.2_r8   &
   936)                      * (dfin_kk/3.0_r8)**0.075_r8   &
   937)                      * (dens_part*1.0e-3_r8)**(-0.33_r8)   &
   938)                      * (temp_in/293.0_r8)**(-0.75_r8)
   939) 
   940) ! "cs_prime parameter" (1/m2) 
   941) ! instead kk2002 eqn 3, use
   942) !     cs_prime ~= tmpa / (4*pi*tmpb * h2so4_accom_coef)
   943) ! where
   944) !     tmpa = -d(ln(h2so4))/dt by conden to particles   (1/h units)
   945) !     tmpb = h2so4 vapor diffusivity (m2/h units)
   946) ! this approx is generally within a few percent of the cs_prime
   947) !     calculated directly from eqn 2, 
   948) !     which is acceptable, given overall uncertainties
   949) ! tmpa = -d(ln(h2so4))/dt by conden to particles   (1/h units)
   950)             tmpa = h2so4_uptkrate * 3600.0_r8
   951)             tmpa1 = tmpa
   952)             tmpa = max( tmpa, 0.0_r8 )
   953) ! tmpb = h2so4 gas diffusivity (m2/s, then m2/h)
   954)             tmpb = 6.7037e-6_r8 * (temp_in**0.75_r8) / cair
   955)             tmpb1 = tmpb         ! m2/s
   956)             tmpb = tmpb*3600.0_r8   ! m2/h
   957)             cs_prime_kk = tmpa/(4.0_r8*pi*tmpb*accom_coef_h2so4)
   958)             cs_kk = cs_prime_kk*4.0_r8*pi*tmpb1
   959) 
   960) ! "nu" parameter (nm) -- kk2002 eqn 11
   961)             nu_kk = gamma_kk*cs_prime_kk/gr_kk
   962) ! nucleation rate adjustment factor (--) -- kk2002 eqn 13
   963)             factor_kk = exp( (nu_kk/dfin_kk) - (nu_kk/dnuc_kk) )
   964) 
   965)         end if
   966)         ratenuclt_kk = ratenuclt_bb*factor_kk
   967) 
   968) 
   969) ! max production of aerosol dry mass (kg-aero/m3-air)
   970)         tmpa = max( 0.0_r8, (ratenuclt_kk*dtnuc*mass_part) )
   971) ! max production of aerosol so4 (mol-so4a/mol-air)
   972)         tmpe = tmpa/(kgaero_per_moleso4a*cair)
   973) ! max production of aerosol so4 (mol/mol-air)
   974) ! based on ratenuclt_kk and mass_part
   975)         qmolso4a_del_max = tmpe
   976) 
   977) ! check if max production exceeds available h2so4 vapor
   978)         freducea = 1.0_r8
   979)         if (qmolso4a_del_max .gt. qh2so4_cur) then
   980)            freducea = qh2so4_cur/qmolso4a_del_max
   981)         end if
   982) 
   983) ! check if max production exceeds available nh3 vapor
   984)         freduceb = 1.0_r8
   985)         if (molenh4a_per_moleso4a .ge. 1.0e-10_r8) then
   986) ! max production of aerosol nh4 (ppm) based on ratenuclt_kk and mass_part
   987)            qmolnh4a_del_max = qmolso4a_del_max*molenh4a_per_moleso4a
   988)            if (qmolnh4a_del_max .gt. qnh3_cur) then
   989)               freduceb = qnh3_cur/qmolnh4a_del_max
   990)            end if
   991)         end if
   992)         freduce = min( freducea, freduceb )
   993) 
   994) ! if adjusted nucleation rate is less than 1e-12 #/m3/s ~= 0.1 #/cm3/day,
   995) ! exit with new particle formation = 0
   996)         if (freduce*ratenuclt_kk .le. 1.0e-12_r8) return
   997) 
   998) 
   999) ! note:  suppose that at this point, freduce < 1.0 (no gas-available 
  1000) !    constraints) and molenh4a_per_moleso4a < 2.0
  1001) ! if the gas-available constraints is do to h2so4 availability,
  1002) !    then it would be possible to condense "additional" nh3 and have
  1003) !    (nh3/h2so4 gas molar ratio) < (nh4/so4 aerosol molar ratio) <= 2 
  1004) ! one could do some additional calculations of 
  1005) !    dens_part & molenh4a_per_moleso4a to realize this
  1006) ! however, the particle "growing" is a crude approximate way to get
  1007) !    the new particles to the host code's minimum particle size,
  1008) ! are such refinements worth the effort?
  1009) 
  1010) 
  1011) ! changes to h2so4 & nh3 gas (in mol/mol-air), limited by amounts available
  1012)         tmpa = 0.9999_r8
  1013)         qh2so4_del = min( tmpa*qh2so4_cur, freduce*qmolso4a_del_max )
  1014)         qnh3_del   = min( tmpa*qnh3_cur, qh2so4_del*molenh4a_per_moleso4a )
  1015)         qh2so4_del = -qh2so4_del
  1016)         qnh3_del   = -qnh3_del
  1017) 
  1018) ! changes to so4 & nh4 aerosol (in mol/mol-air)
  1019)         qso4a_del = -qh2so4_del
  1020)         qnh4a_del =   -qnh3_del
  1021) ! change to aerosol number (in #/mol-air)
  1022)         qnuma_del = 1.0e-3_r8*(qso4a_del*mw_so4a + qnh4a_del*mw_nh4a)/mass_part
  1023) 
  1024) ! do the following (tmpa, tmpb, tmpc) calculations as a check
  1025) ! max production of aerosol number (#/mol-air)
  1026)         tmpa = max( 0.0_r8, (ratenuclt_kk*dtnuc/cair) )
  1027) ! adjusted production of aerosol number (#/mol-air)
  1028)         tmpb = tmpa*freduce
  1029) ! relative difference from qnuma_del
  1030)         tmpc = (tmpb - qnuma_del)/max(tmpb, qnuma_del, 1.0e-35_r8)
  1031) 
  1032) 
  1033) !
  1034) ! diagnostic output to fort.41
  1035) ! (this should be commented-out or deleted in the wrf-chem version)
  1036) !
  1037)         if (ldiagaa <= 0) return
  1038) 
  1039)         icase = icase + 1
  1040)         if (abs(tmpc) .gt. abs(reldiffmax)) then
  1041)            reldiffmax = tmpc
  1042)            icase_reldiffmax = icase
  1043)         end if
  1044) !       do lun = 41, 51, 10
  1045)         do lun = 6, 6
  1046) !          write(lun,'(/)')
  1047)            write(lun,'(a,2i9,1p,e10.2)')   &
  1048)                'vehkam bin-nuc icase, icase_rdmax =',   &
  1049)                icase, icase_reldiffmax, reldiffmax
  1050)            if (freduceb .lt. freducea) then
  1051)               if (abs(freducea-freduceb) .gt.   &
  1052)                    3.0e-7_r8*max(freduceb,freducea)) write(lun,'(a,1p,2e15.7)')   &
  1053)                  'freducea, b =', freducea, freduceb
  1054)            end if
  1055)         end do
  1056) 
  1057) ! output factors so that output matches that of ternucl03
  1058) !       fogas  = 1.0e6                     ! convert mol/mol-air to ppm
  1059) !       foso4a = 1.0e9*mw_so4a/mw_air      ! convert mol-so4a/mol-air to ug/kg-air
  1060) !       fonh4a = 1.0e9*mw_nh4a/mw_air      ! convert mol-nh4a/mol-air to ug/kg-air
  1061) !       fonuma = 1.0e3/mw_air              ! convert #/mol-air to #/kg-air
  1062)         fogas  = 1.0_r8
  1063)         foso4a = 1.0_r8
  1064)         fonh4a = 1.0_r8
  1065)         fonuma = 1.0_r8
  1066) 
  1067) !       do lun = 41, 51, 10
  1068)         do lun = 6, 6
  1069) 
  1070)         write(lun,'(a,2i5)') 'newnuc_method_flagaa/aa2',   &
  1071)            newnuc_method_flagaa, newnuc_method_flagaa2
  1072) 
  1073)         write(lun,9210)
  1074)         write(lun,9201) temp_in, rh_in,   &
  1075)            ratenuclt, 2.0_r8*radius_cluster*1.0e-7_r8, dpdry_part*1.0e2_r8,   &
  1076)            voldry_part*1.0e6_r8, float(igrow)
  1077)         write(lun,9215)
  1078)         write(lun,9201)   &
  1079)            qh2so4_avg*fogas, 0.0_r8,  &
  1080)            qh2so4_cur*fogas, qnh3_cur*fogas,  &
  1081)            qh2so4_del*fogas, qnh3_del*fogas,  &
  1082)            qso4a_del*foso4a, qnh4a_del*fonh4a
  1083) 
  1084)         write(lun,9220)
  1085)         write(lun,9201)   &
  1086)            dtnuc, dens_nh4so4a*1.0e-3_r8,   &
  1087)            (qnh3_cur/qh2so4_cur), molenh4a_per_moleso4a,   &
  1088)            qnuma_del*fonuma, tmpb*fonuma, tmpc, freduce
  1089) 
  1090)         end do
  1091) 
  1092) !       lun = 51
  1093)         lun = 6
  1094)         write(lun,9230)
  1095)         write(lun,9201)   &
  1096)            press_in, cair*1.0e-6_r8, so4vol_in,   &
  1097)            wet_volfrac_so4a, wetvol_dryvol, dens_part*1.0e-3_r8
  1098) 
  1099)         if (igrow > 0) then
  1100)         write(lun,9240)
  1101)         write(lun,9201)   &
  1102)            tmp_spd, gr_kk, dnuc_kk, dfin_kk,   &
  1103)            gamma_kk, tmpa1, tmpb1, cs_kk
  1104) 
  1105)         write(lun,9250)
  1106)         write(lun,9201)   &
  1107)            cs_prime_kk, nu_kk, factor_kk, ratenuclt,   &
  1108)            ratenuclt_kk*1.0e-6_r8
  1109)         end if
  1110) 
  1111) 9201    format ( 1p, 40e10.2  )
  1112) 9210    format (   &
  1113)         '      temp        rh',   &
  1114)         '   ratenuc  dia_clus ddry_part',   &
  1115)         ' vdry_part     igrow' )
  1116) 9215    format (   &
  1117)         '  h2so4avg  h2so4pre',   &
  1118)         '  h2so4cur   nh3_cur',   &
  1119)         '  h2so4del   nh3_del',   &
  1120)         '  so4a_del  nh4a_del' )
  1121) 9220    format (    &
  1122)         '     dtnuc    dens_a   nh/so g   nh/so a',   &
  1123)         '  numa_del  numa_dl2   reldiff   freduce' )
  1124) 9230    format (   &
  1125)         '  press_in      cair so4_volin',   &
  1126)         ' wet_volfr wetv_dryv dens_part' )
  1127) 9240    format (   &
  1128)         '   tmp_spd     gr_kk   dnuc_kk   dfin_kk',   &
  1129)         '  gamma_kk     tmpa1     tmpb1     cs_kk' )
  1130) 9250    format (   &
  1131)         ' cs_pri_kk     nu_kk factor_kk ratenuclt',   &
  1132)         ' ratenu_kk' )
  1133) 
  1134)

  1142)         subroutine pbl_nuc_wang2008( so4vol,   &
  1143)             newnuc_method_flagaa, newnuc_method_flagaa2,   &
  1144)             ratenucl, rateloge,   &
  1145)             cnum_tot, cnum_h2so4, cnum_nh3, radius_cluster )
  1146) !
  1147) ! calculates boundary nucleation nucleation rate
  1148) ! using the first or second-order parameterization in  
  1149) !     wang, m., and j.e. penner, 2008,
  1150) !        aerosol indirect forcing in a global model with particle nucleation,
  1151) !        atmos. chem. phys. discuss., 8, 13943-13998
  1152) !
  1153)         implicit none
  1154) 
  1155) ! subr arguments (in)
  1156)         real(r8), intent(in) :: so4vol            ! concentration of h2so4 (molecules cm-3)
  1157)         integer, intent(in)  :: newnuc_method_flagaa  
  1158)                                 ! [11,12] value selects [first,second]-order parameterization
  1159) 
  1160) ! subr arguments (inout)
  1161)         integer, intent(inout)  :: newnuc_method_flagaa2
  1162)         real(r8), intent(inout) :: ratenucl         ! binary nucleation rate, j (# cm-3 s-1)
  1163)         real(r8), intent(inout) :: rateloge         ! log( ratenucl )
  1164) 
  1165)         real(r8), intent(inout) :: cnum_tot         ! total number of molecules
  1166)                                                     ! in the critical nucleus
  1167)         real(r8), intent(inout) :: cnum_h2so4       ! number of h2so4 molecules
  1168)         real(r8), intent(inout) :: cnum_nh3         ! number of nh3 molecules
  1169)         real(r8), intent(inout) :: radius_cluster   ! the radius of cluster (nm)
  1170) 
  1171) 
  1172) ! local variables
  1173)         real(r8) :: tmp_diam, tmp_mass, tmp_volu
  1174)         real(r8) :: tmp_rateloge, tmp_ratenucl
  1175) 
  1176) ! executable
  1177) 
  1178) 
  1179) ! nucleation rate
  1180)         if (newnuc_method_flagaa == 11) then
  1181)            tmp_ratenucl = 1.0e-6_r8 * so4vol
  1182)         else if (newnuc_method_flagaa == 12) then
  1183)            tmp_ratenucl = 1.0e-12_r8 * (so4vol**2)
  1184)         else
  1185)            return
  1186)         end if
  1187)         tmp_rateloge = log( tmp_ratenucl )
  1188) 
  1189) ! exit if pbl nuc rate is lower than (incoming) ternary/binary rate
  1190)         if (tmp_rateloge <= rateloge) return
  1191) 
  1192)         rateloge = tmp_rateloge
  1193)         ratenucl = tmp_ratenucl
  1194)         newnuc_method_flagaa2 = newnuc_method_flagaa
  1195) 
  1196) ! following wang 2002, assume fresh nuclei are 1 nm diameter
  1197) !    subsequent code will "grow" them to aitken mode size
  1198)         radius_cluster = 0.5_r8
  1199) 
  1200) ! assume fresh nuclei are pure h2so4
  1201) !    since aitken size >> initial size, the initial composition 
  1202) !    has very little impact on the results
  1203)         tmp_diam = radius_cluster * 2.0e-7_r8   ! diameter in cm
  1204)         tmp_volu = (tmp_diam**3) * (pi/6.0_r8)  ! volume in cm^3
  1205)         tmp_mass = tmp_volu * 1.8_r8            ! mass in g
  1206)         cnum_h2so4 = (tmp_mass / 98.0_r8) * 6.023e23_r8   ! no. of h2so4 molec assuming pure h2so4
  1207)         cnum_tot = cnum_h2so4
  1208)         cnum_nh3 = 0.0_r8
  1209) 
  1210) 
  1211)         return
  1212)         end subroutine pbl_nuc_wang2008
