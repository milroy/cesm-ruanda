    31)     use dyn_comp,      only: dyn_import_t

    35)     use dimensions_mod,     only: nelemd, nlev, np
    36)     use dof_mod, only           : putUniquePoints
    37)     use edge_mod, only : edgevpack, edgevunpack, InitEdgeBuffer, FreeEdgeBuffer, EdgeBuffer_t
    38)     use ncdio_atm, only : infld
    39)     use shr_vmath_mod, only: shr_vmath_log
    40)     use hycoef,           only: ps0
    41)     use abortutils,     only: endrun
    42)     use pio, only : file_desc_t, io_desc_t, pio_double, pio_get_local_array_size, pio_freedecomp
    43)     use dyn_grid, only : get_horiz_grid_dim_d
    44)     use chemistry   , only: chem_implements_cnst, chem_init_cnst
    45)     use carma_intr,   only: carma_implements_cnst, carma_init_cnst
    46)     use tracers     , only: tracers_implements_cnst, tracers_init_cnst
    47)     use aoa_tracers , only: aoa_tracers_implements_cnst, aoa_tracers_init_cnst
    48)     use stratiform,   only: stratiform_implements_cnst, stratiform_init_cnst
    49)     use microp_driver, only: microp_driver_implements_cnst, microp_driver_init_cnst
    50)     use phys_control,  only: phys_getopts
    51)     use co2_cycle   , only: co2_implements_cnst, co2_init_cnst
    52)     use nctopo_util_mod, only: nctopo_util_inidat
    53)     implicit none
    54)     type(file_desc_t),intent(inout) :: ncid_ini, ncid_topo
    55)     type (dyn_import_t), target, intent(inout) :: dyn_in   ! dynamics import
    56) 
    57)     type(element_t), pointer :: elem(:)
    58)     real(r8), allocatable :: tmp(:,:)
    59)     integer, allocatable :: gcols(:)
    60)     integer :: tlncols, ig, ie, start, j, t, k
    61)     character(len=40) :: fieldname
    62)     logical :: found
    63)     integer :: kptr, m_cnst
    64)     type(EdgeBuffer_t) :: edge
    65)     type(io_desc_t) :: iodesc
    66)     integer :: lsize
    67) 
    68)     integer,parameter :: pcnst = PCNST
    69)     integer, pointer :: gcid(:)
    70) 
    71)     integer :: rndm_seed_sz
    72)     integer, allocatable :: rndm_seed(:)
    73)     real(r8) :: pertval
    74)     integer :: i
    75)     real(r8), parameter :: D0_0 = 0.0_r8
    76)     real(r8), parameter :: D0_5 = 0.5_r8
    77)     real(r8), parameter :: D1_0 = 1.0_r8
    78)     real(r8), parameter :: D2_0 = 2.0_r8
    79)     character*16 :: subname='READ_INIDAT'
    80) 
    81)     if(iam < par%nprocs) then
    82)        elem=> dyn_in%elem
    83)     else
    84)        nullify(elem)
    85)     end if
    86) 
    87)     call get_dyn_decomp(elem, nlev, pio_double, iodesc)
    88) 
    89)     lsize = pio_get_local_array_size(iodesc)	
    90) 
    91)     tlncols = lsize/nlev
    92) 
    93)     allocate(tmp(tlncols,nlev))    
    94) 
    95)     if (iam < par%nprocs) then
    96)       if(elem(1)%idxP%NumUniquePts <=0 .or. elem(1)%idxP%NumUniquePts > np*np) then
    97)          write(iulog,*)  elem(1)%idxP%NumUniquePts
    98)          call endrun('inidat')
    99)       end if
   100)     end if
   101) 
   102)     fieldname = 'U'
   103)     call infld(fieldname, ncid_ini, iodesc, tlncols,'lev',  tmp, found)
   104)     if(.not. found) then
   105)        call endrun('Could not find U field on input datafile')
   106)     end if
   107)     
   108)     start=1
   109)     do ie=1,nelemd
   110)        elem(ie)%state%v=0.0_r8
   111)        call putUniquePoints(elem(ie)%idxP, nlev, tmp(start:,:), &
   112)             elem(ie)%state%v(:,:,1,:,1))
   113)        start=start+elem(ie)%idxP%numUniquePts
   114)     end do
   115) 
   116)     fieldname = 'V'
   117)     call infld(fieldname, ncid_ini, iodesc, tlncols,'lev',  tmp, found)
   118)     if(.not. found) then
   119)        call endrun('Could not find V field on input datafile')
   120)     end if
   121)     start=1
   122)     do ie=1,nelemd
   123)        call putUniquePoints(elem(ie)%idxP, nlev, tmp(start:,:), &
   124)             elem(ie)%state%v(:,:,2,:,1))
   125)        start=start+elem(ie)%idxP%numUniquePts
   126)     end do
   127) 
   128)     fieldname = 'T'
   129)     call infld(fieldname, ncid_ini, iodesc, tlncols ,'lev',  tmp, found)
   130)     if(.not. found) then
   131)        call endrun('Could not find T field on input datafile')
   132)     end if
   133)     start=1
   134)     do ie=1,nelemd
   135)        elem(ie)%state%T=0.0_r8
   136)        call putUniquePoints(elem(ie)%idxP, nlev, tmp(start:,:), &
   137)             elem(ie)%state%T(:,:,:,1))
   138)        start=start+elem(ie)%idxP%numUniquePts
   139)     end do
   140) 
   141)     if (pertlim .ne. D0_0) then
   142)       if(masterproc) then
   143)         write(iulog,*) trim(subname), ': Adding random perturbation bounded', &
   144)                        'by +/- ', pertlim, ' to initial temperature field'
   145)       end if
   146) 
   147)       call random_seed(size=rndm_seed_sz)
   148)       allocate(rndm_seed(rndm_seed_sz))
   149) 
   150)       do ie=1,nelemd
   151)         ! seed random number generator based on element ID
   152)         ! (possibly include a flag to allow clock-based random seeding)
   153)         rndm_seed = elem(ie)%GlobalId
   154)         call random_seed(put=rndm_seed)
   155)         do i=1,np
   156)           do j=1,np
   157)             do k=1,nlev
   158)               call random_number(pertval)
   159)               pertval = D2_0*pertlim*(D0_5 - pertval)
   160)               elem(ie)%state%T(i,j,k,1) = elem(ie)%state%T(i,j,k,1)*(D1_0 + pertval)
   161)             end do
   162)           end do
   163)         end do
   164)       end do
   165) 
   166)       deallocate(rndm_seed)
   167)     end if
   168) 
   169)     gcid => get_ldof(elem, 1)
   170) 
   171)     ! qmin = 1e-12,0,0
   172) 
   173)     do m_cnst=1,pcnst
   174)        found = .false.
   175)        if(cnst_read_iv(m_cnst)) then
   176)           call infld(cnst_name(m_cnst), ncid_ini, iodesc, tlncols, 'lev', tmp, found)
   177)        end if
   178)        if(.not. found) then
   179) 
   180)           if(par%masterproc  ) write(iulog,*) 'Field ',cnst_name(m_cnst),' not found on initial dataset'
   181) 
   182)           if (microp_driver_implements_cnst(cnst_name(m_cnst))) then
   183)              call microp_driver_init_cnst(cnst_name(m_cnst),tmp , gcid)
   184)               if(par%masterproc) write(iulog,*) '          ', cnst_name(m_cnst), ' initialized by "microp_driver_init_cnst"'
   185)           else if (stratiform_implements_cnst(cnst_name(m_cnst))) then
   186)              call stratiform_init_cnst(cnst_name(m_cnst), tmp, gcid)
   187)               if(par%masterproc) write(iulog,*) '          ', cnst_name(m_cnst), &
   188)                    ' initialized by "stratiform_init_cnst"'
   189)           else if (chem_implements_cnst(cnst_name(m_cnst))) then
   190)              call chem_init_cnst(cnst_name(m_cnst), tmp, gcid)
   191)               if(par%masterproc) write(iulog,*) '          ', cnst_name(m_cnst), &
   192)                    ' initialized by "chem_init_cnst"'
   193)           else if (tracers_implements_cnst(cnst_name(m_cnst))) then
   194)              call tracers_init_cnst(cnst_name(m_cnst), tmp, gcid)
   195)               if(par%masterproc) write(iulog,*) '          ', cnst_name(m_cnst), &
   196)                    ' initialized by "tracers_init_cnst"'
   197)           else if (aoa_tracers_implements_cnst(cnst_name(m_cnst))) then
   198)              call aoa_tracers_init_cnst(cnst_name(m_cnst), tmp, gcid)
   199)               if(par%masterproc) write(iulog,*) '          ', cnst_name(m_cnst), &
   200)                    ' initialized by "aoa_tracers_init_cnst"'
   201)           else if (carma_implements_cnst(cnst_name(m_cnst))) then
   202)              call carma_init_cnst(cnst_name(m_cnst), tmp, gcid)
   203)               if(par%masterproc) write(iulog,*) '          ', cnst_name(m_cnst), &
   204)                    ' initialized by "carma_init_cnst"'
   205)           else if (co2_implements_cnst(cnst_name(m_cnst))) then
   206)              call co2_init_cnst(cnst_name(m_cnst), tmp, gcid)
   207)               if(par%masterproc) write(iulog,*) '          ', cnst_name(m_cnst), &
   208)                    ' initialized by "co2_init_cnst"'
   209)           else
   210)               if(par%masterproc) write(iulog,*) '          ', cnst_name(m_cnst), ' set to 0.'
   211)           end if
   212)        end if
   213)        do k=1,nlev
   214)           do ig=1,tlncols
   215)              tmp(ig,k)=max(qmin(m_cnst),tmp(ig,k))
   216)           end do
   217)        end do
   218)        
   219)        start=1
   220)        do ie=1,nelemd
   221)           elem(ie)%state%Q(:,:,:,m_cnst)=0.0_r8
   222)           call putUniquePoints(elem(ie)%idxP, nlev, tmp(start:,:), &
   223)                elem(ie)%state%Q(:,:,:,m_cnst))
   224)           start=start+elem(ie)%idxP%numUniquePts
   225)        end do
   226)     end do
   227)     deallocate(gcid)
   228) 
   229)     call pio_freedecomp(ncid_ini, iodesc)
   230) 
   231)     call get_dyn_decomp(elem, 1, pio_double, iodesc)
   232) 
   233)     fieldname = 'PS'
   234)     call infld(fieldname, ncid_ini, iodesc, tmp(:,1), found)
   235)     if(.not. found) then
   236)        call endrun('Could not find PS field on input datafile')
   237)     end if
   238)     start=1
   239) 
   240)     if(minval(tmp(:,1)) < 10000) then
   241)        call endrun('Problem reading ps field')
   242)     end if
   243) 
   244)     do ie=1,nelemd
   245)        elem(ie)%state%ps_v=0.0_r8
   246)        call putUniquePoints(elem(ie)%idxP, tmp(start:start+elem(ie)%idxP%numUniquePts-1,1), &
   247)             elem(ie)%state%ps_v(:,:,1))
   248)        start=start+elem(ie)%idxP%numUniquePts
   249)     end do
   250) 
   251)     if ( (ideal_phys .or. aqua_planet)) then
   252)        tmp(:,:) = 0._r8
   253)     else    
   254)        fieldname = 'PHIS'
   255)        call infld(fieldname, ncid_topo, iodesc, tmp(:,1), found)
   256)        if(.not. found) then
   257)           call endrun('Could not find PHIS field on input datafile')
   258)        end if
   259)     end if
   260)     start=1
   261)     do ie=1,nelemd
   262)        elem(ie)%state%phis=0.0_r8
   263)        call putUniquePoints(elem(ie)%idxP, tmp(start:,1), &
   264)             elem(ie)%state%phis(:,:))
   265)        start=start+elem(ie)%idxP%numUniquePts
   266)     end do
   267)     
   268)     ! once we've read all the fields we do a boundary exchange to 
   269)     ! update the redundent columns in the dynamics
   270)     if(iam < par%nprocs) then
   271)        call initEdgeBuffer(edge, (3+pcnst)*nlev+2)
   272)     end if
   273)     do ie=1,nelemd
   274)        kptr=0
   275)        call edgeVpack(edge, elem(ie)%state%ps_v(:,:,1),1,kptr,elem(ie)%desc)
   276)        kptr=kptr+1
   277)        call edgeVpack(edge, elem(ie)%state%phis,1,kptr,elem(ie)%desc)
   278)        kptr=kptr+1
   279)        call edgeVpack(edge, elem(ie)%state%v(:,:,:,:,1),2*nlev,kptr,elem(ie)%desc)
   280)        kptr=kptr+2*nlev
   281)        call edgeVpack(edge, elem(ie)%state%T(:,:,:,1),nlev,kptr,elem(ie)%desc)
   282)        kptr=kptr+nlev
   283)        call edgeVpack(edge, elem(ie)%state%Q(:,:,:,:),nlev*pcnst,kptr,elem(ie)%desc)
   284)     end do
   285)     if(iam < par%nprocs) then
   286)        call bndry_exchangeV(par,edge)
   287)     end if
   288)     do ie=1,nelemd
   289)        kptr=0
   290)        call edgeVunpack(edge, elem(ie)%state%ps_v(:,:,1),1,kptr,elem(ie)%desc)
   291)        kptr=kptr+1
   292)        call edgeVunpack(edge, elem(ie)%state%phis,1,kptr,elem(ie)%desc)
   293)        kptr=kptr+1
   294)        call edgeVunpack(edge, elem(ie)%state%v(:,:,:,:,1),2*nlev,kptr,elem(ie)%desc)
   295)        kptr=kptr+2*nlev
   296)        call edgeVunpack(edge, elem(ie)%state%T(:,:,:,1),nlev,kptr,elem(ie)%desc)
   297)        kptr=kptr+nlev
   298)        call edgeVunpack(edge, elem(ie)%state%Q(:,:,:,:),nlev*pcnst,kptr,elem(ie)%desc)
   299)     end do
   300) 
   301) !$omp parallel do private(ie, t, m_cnst)
   302)     do ie=1,nelemd
   303)        do t=2,3
   304)           elem(ie)%state%ps_v(:,:,t)=elem(ie)%state%ps_v(:,:,1)
   305)           elem(ie)%state%v(:,:,:,:,t)=elem(ie)%state%v(:,:,:,:,1)
   306)           elem(ie)%state%T(:,:,:,t)=elem(ie)%state%T(:,:,:,1)
   307)        end do
   308)        call shr_vmath_log(elem(ie)%state%ps_v,elem(ie)%state%lnps,size(elem(ie)%state%lnps))
   309)     end do
   310) 
   311)     if(iam < par%nprocs) then
   312)        call FreeEdgeBuffer(edge)
   313)     end if
   314) 
   315)     !
   316)     ! This subroutine is used to create nc_topo files, if requested
   317)     ! 
   318) 
   319)     call nctopo_util_inidat(ncid_topo,iodesc,elem)
   320) 
   321)     call pio_freedecomp(ncid_topo, iodesc)
   322)     deallocate(tmp)
   323) 
   324)   end subroutine read_inidat

   328)   subroutine get_dyn_decomp(elem, nlev, datatype, iodesc)
   329)     use pio, only : io_desc_t, pio_initdecomp
   330)     use cam_pio_utils, only : pio_subsystem
   331)     use dyn_grid, only : get_horiz_grid_dim_d
   332) 
   333)     type(element_t), pointer :: elem(:)
   334)     integer, intent(in) :: nlev, datatype
   335)     type(io_desc_t), intent(out) :: iodesc
   336)     integer, pointer :: ldof(:)
   337)     integer :: dimlens(2), dimcnt
   338) 
   339)     dimcnt=1
   340)     call get_horiz_grid_dim_d(dimlens(1)) 
   341)     if(nlev>1) then
   342)        dimlens(2) = nlev
   343)        dimcnt=dimcnt+1
   344)     end if
   345) 
   346)     ldof => get_ldof(elem, nlev)
   347)     call pio_initdecomp(pio_subsystem, datatype, dimlens(1:dimcnt), ldof, iodesc)
   348) 
   349)     deallocate(ldof)
   350) 
   351)   end subroutine get_dyn_decomp

   354)   function get_ldof(elem, nlev) result(ldof)
   355)     use dimensions_mod,     only: nelemd
   356)     use dyn_grid, only : get_horiz_grid_dim_d
   357)     use abortutils,     only: endrun
   358) 
   359)     type(element_t), pointer :: elem(:)
   360)     integer, intent(in) :: nlev
   361)     integer, pointer :: ldof(:)
   362) 
   363)     integer :: lcnt, ie, j, k, ig, numpts, offset, hdim
   364) 
   365) 
   366)     call get_horiz_grid_dim_d(hdim)
   367) 
   368)     lcnt = 0
   369)     do ie=1,nelemd
   370)        lcnt = lcnt+nlev*elem(ie)%idxP%NumUniquePts
   371)     end do
   372)     allocate(ldof(lcnt))
   373)     ig=1
   374)     ldof(:) = 0
   375)     do k=1,nlev
   376)        do ie=1,nelemd
   377)           numpts = elem(ie)%idxP%NumUniquePts
   378)           offset = elem(ie)%idxP%UniquePtOffset
   379)           do j=1,numpts
   380)              ldof(ig)=offset+(j-1)+(k-1)*hdim
   381)              ig=ig+1
   382)           end do
   383)        end do
   384)     end do
   385) 
   386) 
   387)   end function get_ldof
