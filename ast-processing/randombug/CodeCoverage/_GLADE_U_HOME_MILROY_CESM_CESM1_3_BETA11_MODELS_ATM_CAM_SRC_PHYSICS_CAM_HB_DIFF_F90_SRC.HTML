     1) module hb_diff
     2)   !---------------------------------------------------------------------------------
     3)   ! Module to compute mixing coefficients associated with turbulence in the 
     4)   ! planetary boundary layer and elsewhere.  PBL coefficients are based on Holtslag 
     5)   ! and Boville, 1991.
     6)   !
     7)   ! Public interfaces:
     8)   !    init_hb_diff     initializes time independent coefficients
     9)   !    compute_hb_diff  computes eddy diffusivities and counter-gradient fluxes
    10)   !
    11)   ! Private methods:
    12)   !       trbintd         initializes time dependent variables
    13)   !       pblintd         initializes time dependent variables that depend pbl depth
    14)   !       austausch_atm   computes free atmosphere exchange coefficients
    15)   !       austausch_pbl   computes pbl exchange coefficients
    16)   !
    17)   !---------------------------Code history--------------------------------
    18)   ! Standardized:      J. Rosinski, June 1992
    19)   ! Reviewed:          P. Rasch, B. Boville, August 1992
    20)   ! Reviewed:          P. Rasch, April 1996
    21)   ! Reviewed:          B. Boville, April 1996
    22)   ! rewritten:         B. Boville, May 2000
    23)   ! rewritten:         B. Stevens, August 2000
    24)   ! modularized:       J. McCaa, September 2004
    25)   !---------------------------------------------------------------------------------
    26)   use shr_kind_mod, only: r8 => shr_kind_r8
    27)   use spmd_utils,   only: masterproc          ! output from hb_init should be eliminated
    28)   use ppgrid,       only: pver, pverp, pcols  ! these should be passed in
    29)   use cam_logfile,  only: iulog
    30) 
    31)   implicit none
    32)   private
    33)   save
    34) 
    35)   ! Public interfaces
    36)   public init_hb_diff
    37)   public compute_hb_diff
    38)   public pblintd
    39)   !
    40)   ! PBL limits
    41)   !
    42)   real(r8), parameter :: pblmaxp   = 4.e4_r8        ! pbl max depth in pressure units
    43)   real(r8), parameter :: zkmin     = 0.01_r8        ! Minimum kneutral*f(ri)
    44)   !
    45)   ! PBL Parameters
    46)   !
    47)   real(r8), parameter :: onet  = 1._r8/3._r8 ! 1/3 power in wind gradient expression
    48)   real(r8), parameter :: betam = 15.0_r8  ! Constant in wind gradient expression
    49)   real(r8), parameter :: betas =  5.0_r8  ! Constant in surface layer gradient expression
    50)   real(r8), parameter :: betah = 15.0_r8  ! Constant in temperature gradient expression 
    51)   real(r8), parameter :: fakn  =  7.2_r8  ! Constant in turbulent prandtl number
    52)   real(r8), parameter :: fak   =  8.5_r8  ! Constant in surface temperature excess         
    53)   real(r8), parameter :: ricr  =  0.3_r8  ! Critical richardson number
    54)   real(r8), parameter :: sffrac=  0.1_r8  ! Surface layer fraction of boundary layer
    55)   real(r8), parameter :: binm  = betam*sffrac       ! betam * sffrac
    56)   real(r8), parameter :: binh  = betah*sffrac       ! betah * sffrac
    57) 
    58)   ! Pbl constants set using values from other parts of code
    59) 
    60)   real(r8) :: cpair      ! Specific heat of dry air
    61)   real(r8) :: g          ! Gravitational acceleration
    62)   real(r8) :: ml2(pverp) ! Mixing lengths squared
    63)   real(r8) :: vk         ! Von Karman's constant
    64)   real(r8) :: ccon       ! fak * sffrac * vk
    65) 
    66)   integer :: npbl       ! Maximum number of levels in pbl from surface
    67)   integer :: ntop_turb  ! Top level to which turbulent vertical diffusion is applied.
    68)   integer :: nbot_turb  ! Bottom level to which turbulent vertical diff is applied.
    69) 
    70) !===============================================================================
    71) CONTAINS
    72) !===============================================================================
    73) 
    74) subroutine init_hb_diff(gravx, cpairx, ntop_eddy, nbot_eddy, pref_mid, &
    75)                         vkx, eddy_scheme)
    76) 
    77)    !----------------------------------------------------------------------- 
    78)    ! 
    79)    ! Initialize time independent variables of turbulence/pbl package.
    80)    ! 
    81)    !-----------------------------------------------------------------------
    82) 
    83)    !------------------------------Arguments--------------------------------
    84)    real(r8), intent(in) :: gravx     ! acceleration of gravity
    85)    real(r8), intent(in) :: cpairx    ! specific heat of dry air
    86)    real(r8), intent(in) :: pref_mid(pver)! reference pressures at midpoints
    87)    real(r8), intent(in) :: vkx       ! Von Karman's constant
    88)    integer, intent(in)  :: ntop_eddy ! Top level to which eddy vert diff is applied.
    89)    integer, intent(in)  :: nbot_eddy ! Bottom level to which eddy vert diff is applied.
    90)    character(len=16),  intent(in) :: eddy_scheme
    91) 
    92)    !---------------------------Local workspace-----------------------------
    93)    integer :: k                     ! vertical loop index
    94)    !-----------------------------------------------------------------------
    95) 
    96)    ! Basic constants
    97)    cpair = cpairx
    98)    g     = gravx
    99)    vk    = vkx
   100)    ccon  = fak*sffrac*vk
   101)    ntop_turb = ntop_eddy
   102)    nbot_turb = nbot_eddy
   103) 
   104)    ! Set the square of the mixing lengths.
   105)    ml2(ntop_turb) = 0._r8
   106)    do k = ntop_turb+1, nbot_turb
   107)       ml2(k) = 30.0_r8**2                 ! HB scheme: length scale = 30m  
   108)       if  ( eddy_scheme .eq. 'HBR' ) then      
   109)          ml2(k) = 1.0_r8**2               ! HBR scheme: length scale = 1m  
   110)       end if
   111)    end do
   112)    ml2(nbot_turb+1) = 0._r8
   113) 
   114)    ! Limit pbl height to regions below 400 mb
   115)    ! npbl = max number of levels (from bottom) in pbl
   116) 
   117)    npbl = 0
   118)    do k=nbot_turb,ntop_turb,-1
   119)       if (pref_mid(k) >= pblmaxp) then
   120)          npbl = npbl + 1
   121)       end if
   122)    end do
   123)    npbl = max(npbl,1)
   124) 
   125)    if (masterproc) then
   126)       write(iulog,*)'INIT_HB_DIFF: PBL height will be limited to bottom ',npbl, &
   127)          ' model levels. Top is ',pref_mid(pverp-npbl),' pascals'
   128)    end if
   129) 
   130) end subroutine init_hb_diff
   131)

   134)   subroutine compute_hb_diff(lchnk, ncol,            &
   135)        th      ,t       ,q       ,z       ,zi      , &
   136)        pmid    ,u       ,v       ,taux    ,tauy    , &
   137)        shflx   ,qflx    ,obklen  ,ustar   ,pblh    , &
   138)        kvm     ,kvh     ,kvq     ,cgh     ,cgs     , &
   139)        tpert   ,qpert   ,cldn    ,ocnfrac ,tke     , &
   140)        ri      , &
   141)        eddy_scheme)
   142)     !----------------------------------------------------------------------- 
   143)     ! 
   144)     ! Purpose: 
   145)     !  Interface routines for calcualtion and diatnostics of turbulence related
   146)     !  coefficients
   147)     !
   148)     ! Author: B. Stevens (rewrite August 2000)
   149)     ! 
   150)     !-----------------------------------------------------------------------
   151) 
   152)     use pbl_utils, only: virtem, calc_ustar, calc_obklen
   153) 
   154)     !------------------------------Arguments--------------------------------
   155)     !
   156)     ! Input arguments
   157)     !
   158)     integer, intent(in) :: lchnk                      ! chunk index (for debug only)
   159)     integer, intent(in) :: ncol                       ! number of atmospheric columns
   160) 
   161)     real(r8), intent(in)  :: th(pcols,pver)           ! potential temperature [K]
   162)     real(r8), intent(in)  :: t(pcols,pver)            ! temperature (used for density)
   163)     real(r8), intent(in)  :: q(pcols,pver)            ! specific humidity [kg/kg]
   164)     real(r8), intent(in)  :: z(pcols,pver)            ! height above surface [m]
   165)     real(r8), intent(in)  :: zi(pcols,pverp)          ! height above surface [m]
   166)     real(r8), intent(in)  :: u(pcols,pver)            ! zonal velocity
   167)     real(r8), intent(in)  :: v(pcols,pver)            ! meridional velocity
   168)     real(r8), intent(in)  :: taux(pcols)              ! zonal stress [N/m2]
   169)     real(r8), intent(in)  :: tauy(pcols)              ! meridional stress [N/m2]
   170)     real(r8), intent(in)  :: shflx(pcols)             ! sensible heat flux
   171)     real(r8), intent(in)  :: qflx(pcols)              ! water vapor flux
   172)     real(r8), intent(in)  :: pmid(pcols,pver)         ! midpoint pressures
   173)     real(r8), intent(in)  :: cldn(pcols,pver)         ! new cloud fraction
   174)     real(r8), intent(in)  :: ocnfrac(pcols)           ! Land fraction
   175)     character(len=16), intent(in) :: eddy_scheme
   176) 
   177)     !
   178)     ! Output arguments
   179)     !
   180)     real(r8), intent(out) :: kvm(pcols,pverp)         ! eddy diffusivity for momentum [m2/s]
   181)     real(r8), intent(out) :: kvh(pcols,pverp)         ! eddy diffusivity for heat [m2/s]
   182)     real(r8), intent(out) :: kvq(pcols,pverp)         ! eddy diffusivity for constituents [m2/s]
   183)     real(r8), intent(out) :: cgh(pcols,pverp)         ! counter-gradient term for heat [J/kg/m]
   184)     real(r8), intent(out) :: cgs(pcols,pverp)         ! counter-gradient star (cg/flux)
   185)     real(r8), intent(out) :: tpert(pcols)             ! convective temperature excess
   186)     real(r8), intent(out) :: qpert(pcols)             ! convective humidity excess
   187)     real(r8), intent(out) :: ustar(pcols)             ! surface friction velocity [m/s]
   188)     real(r8), intent(out) :: obklen(pcols)            ! Obukhov length
   189)     real(r8), intent(out) :: pblh(pcols)              ! boundary-layer height [m]
   190)     real(r8), intent(out) :: tke(pcols,pverp)         ! turbulent kinetic energy (estimated)
   191)     real(r8), intent(out) :: ri(pcols,pver)           ! richardson number: n2/s2
   192)     !
   193)     !---------------------------Local workspace-----------------------------
   194)     !
   195)     real(r8) :: thv(pcols,pver)         ! virtual temperature
   196)     real(r8) :: rrho(pcols)             ! 1./bottom level density
   197)     real(r8) :: wstar(pcols)            ! convective velocity scale [m/s]
   198)     real(r8) :: kqfs(pcols)             ! kinematic surf constituent flux (kg/m2/s)
   199)     real(r8) :: khfs(pcols)             ! kinimatic surface heat flux 
   200)     real(r8) :: kbfs(pcols)             ! surface buoyancy flux 
   201)     real(r8) :: kvf(pcols,pverp)        ! free atmospheric eddy diffsvty [m2/s]
   202)     real(r8) :: s2(pcols,pver)          ! shear squared
   203)     real(r8) :: n2(pcols,pver)          ! brunt vaisaila frequency
   204)     real(r8) :: bge(pcols)              ! buoyancy gradient enhancment
   205)     integer  :: ktopbl(pcols)           ! index of first midpoint inside pbl
   206)     !
   207)     ! Initialize time dependent variables that do not depend on pbl height
   208)     !
   209) 
   210)     ! virtual temperature
   211)     thv(:ncol,ntop_turb:) = virtem(th(:ncol,ntop_turb:),q(:ncol,ntop_turb:))
   212) 
   213)     ! Compute ustar, Obukhov length, and kinematic surface fluxes.
   214)     call calc_ustar(t(:ncol,pver),pmid(:ncol,pver),taux(:ncol),tauy(:ncol), &
   215)          rrho(:ncol),ustar(:ncol))
   216)     call calc_obklen(th(:ncol,pver), thv(:ncol,pver), qflx(:ncol),  &
   217)                      shflx(:ncol),   rrho(:ncol),     ustar(:ncol), &
   218)                      khfs(:ncol),    kqfs(:ncol),     kbfs(:ncol),  &
   219)                      obklen(:ncol))
   220)     ! Calculate s2, n2, and Richardson number.
   221)     call trbintd(ncol    ,                            &
   222)          thv     ,z       ,u       ,v       , &
   223)          s2      ,n2      ,ri      )
   224)     !
   225)     ! Initialize time dependent variables that do depend on pbl height
   226)     !
   227)     call  pblintd(ncol    ,                            &
   228)          thv     ,z       ,u       ,v       , &
   229)          ustar   ,obklen  ,kbfs    ,pblh    ,wstar   , &
   230)          zi      ,cldn    ,ocnfrac ,bge     )
   231)     !
   232)     ! Get free atmosphere exchange coefficients
   233)     !
   234)     call austausch_atm(ncol    ,ri      ,s2      ,kvf     )
   235)     ! 
   236)     ! Get pbl exchange coefficients
   237)     !
   238)     call austausch_pbl(lchnk, ncol,                    &
   239)          z       ,kvf     ,kqfs    ,khfs    ,kbfs    , &
   240)          obklen  ,ustar   ,wstar   ,pblh    ,kvm     , &
   241)          kvh     ,cgh     ,cgs     ,tpert   ,qpert   , &
   242)          ktopbl  ,tke     ,bge     ,eddy_scheme)
   243)     !
   244) 
   245)     kvq(:ncol,:) = kvh(:ncol,:)
   246) 
   247)     return
   248)   end subroutine compute_hb_diff
   249)   !
   250)   !===============================================================================
   251)   subroutine trbintd(ncol    ,                            &
   252)        thv     ,z       ,u       ,v       , &
   253)        s2      ,n2      ,ri      )
   254) 
   255)     !----------------------------------------------------------------------- 
   256)     ! 
   257)     ! Purpose: 
   258)     !  Time dependent initialization
   259)     ! 
   260)     ! Method: 
   261)     !  Diagnosis of variables that do not depend on mixing assumptions or
   262)     !  PBL depth
   263)     !
   264)     ! Author: B. Stevens (extracted from pbldiff, August, 2000)
   265)     ! 
   266)     !-----------------------------------------------------------------------
   267)     !------------------------------Arguments--------------------------------
   268)     !
   269)     ! Input arguments
   270)     !
   271)     integer, intent(in) :: ncol                      ! number of atmospheric columns
   272) 
   273)     real(r8), intent(in)  :: thv(pcols,pver)         ! virtual temperature
   274)     real(r8), intent(in)  :: z(pcols,pver)           ! height above surface [m]
   275)     real(r8), intent(in)  :: u(pcols,pver)           ! windspeed x-direction [m/s]
   276)     real(r8), intent(in)  :: v(pcols,pver)           ! windspeed y-direction [m/s]
   277) 
   278)     !
   279)     ! Output arguments
   280)     !
   281)     real(r8), intent(out) :: s2(pcols,pver)          ! shear squared
   282)     real(r8), intent(out) :: n2(pcols,pver)          ! brunt vaisaila frequency
   283)     real(r8), intent(out) :: ri(pcols,pver)          ! richardson number: n2/s2
   284)     !
   285)     !---------------------------Local workspace-----------------------------
   286)     !
   287)     integer  :: i                        ! longitude index
   288)     integer  :: k                        ! level index
   289) 
   290)     real(r8) :: vvk                      ! velocity magnitude squared
   291)     real(r8) :: dvdz2                    ! velocity shear squared
   292)     real(r8) :: dz                       ! delta z between midpoints
   293)     !
   294)     ! Compute shear squared (s2), brunt vaisaila frequency (n2) and related richardson
   295)     ! number (ri). Use virtual temperature to compute n2.
   296)     !
   297) 
   298)     do k=ntop_turb,nbot_turb-1
   299)        do i=1,ncol
   300)           dvdz2   = (u(i,k)-u(i,k+1))**2 + (v(i,k)-v(i,k+1))**2
   301)           dvdz2   = max(dvdz2,1.e-36_r8)
   302)           dz      = z(i,k) - z(i,k+1)
   303)           s2(i,k) = dvdz2/(dz**2)
   304)           n2(i,k) = g*2.0_r8*( thv(i,k) - thv(i,k+1))/((thv(i,k) + thv(i,k+1))*dz)
   305)           ri(i,k) = n2(i,k)/s2(i,k)
   306)        end do
   307)     end do
   308) 
   309)     return
   310)   end subroutine trbintd

   312)   !===============================================================================
   313)   subroutine pblintd(ncol    ,                            &
   314)        thv     ,z       ,u       ,v       , &
   315)        ustar   ,obklen  ,kbfs    ,pblh    ,wstar   , &
   316)        zi      ,cldn    ,ocnfrac ,bge     )
   317)     !----------------------------------------------------------------------- 
   318)     ! 
   319)     ! Purpose: 
   320)     ! Diagnose standard PBL variables
   321)     ! 
   322)     ! Method: 
   323)     ! Diagnosis of PBL depth and related variables.  In this case only wstar.
   324)     ! The PBL depth follows:
   325)     !    Holtslag, A.A.M., and B.A. Boville, 1993:
   326)     !    Local versus Nonlocal Boundary-Layer Diffusion in a Global Climate
   327)     !    Model. J. Clim., vol. 6., p. 1825--1842.
   328)     !
   329)     ! Updated by Holtslag and Hack to exclude the surface layer from the
   330)     ! definition of the boundary layer Richardson number. Ri is now defined
   331)     ! across the outer layer of the pbl (between the top of the surface
   332)     ! layer and the pbl top) instead of the full pbl (between the surface and
   333)     ! the pbl top). For simiplicity, the surface layer is assumed to be the
   334)     ! region below the first model level (otherwise the boundary layer depth
   335)     ! determination would require iteration).
   336)     !
   337)     ! Modified for boundary layer height diagnosis: Bert Holtslag, june 1994
   338)     ! >>>>>>>>>  (Use ricr = 0.3 in this formulation)
   339)     ! 
   340)     ! Author: B. Stevens (extracted from pbldiff, August 2000)
   341)     ! 
   342)     !-----------------------------------------------------------------------
   343)     !------------------------------Arguments--------------------------------
   344)     !
   345)     ! Input arguments
   346)     !
   347)     integer, intent(in) :: ncol                      ! number of atmospheric columns
   348) 
   349)     real(r8), intent(in)  :: thv(pcols,pver)         ! virtual temperature
   350)     real(r8), intent(in)  :: z(pcols,pver)           ! height above surface [m]
   351)     real(r8), intent(in)  :: u(pcols,pver)           ! windspeed x-direction [m/s]
   352)     real(r8), intent(in)  :: v(pcols,pver)           ! windspeed y-direction [m/s]
   353)     real(r8), intent(in)  :: ustar(pcols)            ! surface friction velocity [m/s]
   354)     real(r8), intent(in)  :: obklen(pcols)           ! Obukhov length
   355)     real(r8), intent(in)  :: kbfs(pcols)             ! sfc kinematic buoyancy flux [m^2/s^3]
   356)     real(r8), intent(in)  :: zi(pcols,pverp)         ! height above surface [m]
   357)     real(r8), intent(in)  :: cldn(pcols,pver)        ! new cloud fraction
   358)     real(r8), intent(in)  :: ocnfrac(pcols)          ! Land fraction
   359) 
   360)     !
   361)     ! Output arguments
   362)     !
   363)     real(r8), intent(out) :: wstar(pcols)            ! convective sclae velocity [m/s]
   364)     real(r8), intent(out) :: pblh(pcols)             ! boundary-layer height [m]
   365)     real(r8), intent(out) :: bge(pcols)              ! buoyancy gradient enhancment
   366)     !
   367)     !---------------------------Local parameters----------------------------
   368)     !
   369)     real(r8), parameter   :: tiny = 1.e-36_r8           ! lower bound for wind magnitude
   370)     real(r8), parameter   :: fac  = 100._r8             ! ustar parameter in height diagnosis 
   371)     !
   372)     !---------------------------Local workspace-----------------------------
   373)     !
   374)     integer  :: i                       ! longitude index
   375)     integer  :: k                       ! level index
   376) 
   377)     real(r8) :: phiminv(pcols)          ! inverse phi function for momentum
   378)     real(r8) :: phihinv(pcols)          ! inverse phi function for heat
   379)     real(r8) :: rino(pcols,pver)        ! bulk Richardson no. from level to ref lev
   380)     real(r8) :: tlv(pcols)              ! ref. level pot tmp + tmp excess
   381)     real(r8) :: vvk                     ! velocity magnitude squared
   382) 
   383)     logical  :: unstbl(pcols)           ! pts w/unstbl pbl (positive virtual ht flx)
   384)     logical  :: check(pcols)            ! True=>chk if Richardson no.>critcal
   385)     logical  :: ocncldcheck(pcols)      ! True=>if ocean surface and cloud in lowest layer
   386)     !
   387)     ! Compute Obukhov length virtual temperature flux and various arrays for use later:
   388)     !
   389)     do i=1,ncol
   390)        check(i)     = .true.
   391)        rino(i,pver) = 0.0_r8
   392)        pblh(i)      = z(i,pver)
   393)     end do
   394)     !
   395)     !
   396)     ! PBL height calculation:  Scan upward until the Richardson number between
   397)     ! the first level and the current level exceeds the "critical" value.
   398)     !
   399)     do k=pver-1,pver-npbl+1,-1
   400)        do i=1,ncol
   401)           if (check(i)) then
   402)              vvk = (u(i,k) - u(i,pver))**2 + (v(i,k) - v(i,pver))**2 + fac*ustar(i)**2
   403)              vvk = max(vvk,tiny)
   404)              rino(i,k) = g*(thv(i,k) - thv(i,pver))*(z(i,k)-z(i,pver))/(thv(i,pver)*vvk)
   405)              if (rino(i,k) >= ricr) then
   406)                 pblh(i) = z(i,k+1) + (ricr - rino(i,k+1))/(rino(i,k) - rino(i,k+1)) * &
   407)                      (z(i,k) - z(i,k+1))
   408)                 check(i) = .false.
   409)              end if
   410)           end if
   411)        end do
   412)     end do
   413)     !
   414)     ! Estimate an effective surface temperature to account for surface fluctuations
   415)     !
   416)     do i=1,ncol
   417)        if (check(i)) pblh(i) = z(i,pverp-npbl)
   418)        unstbl(i) = (kbfs(i) > 0._r8)
   419)        check(i)  = (kbfs(i) > 0._r8)
   420)        if (check(i)) then
   421)           phiminv(i)   = (1._r8 - binm*pblh(i)/obklen(i))**onet
   422)           rino(i,pver) = 0.0_r8
   423)           tlv(i)       = thv(i,pver) + kbfs(i)*fak/( ustar(i)*phiminv(i) )
   424)        end if
   425)     end do
   426)     !
   427)     ! Improve pblh estimate for unstable conditions using the convective temperature excess:
   428)     !
   429)     do i = 1,ncol
   430)        bge(i) = 1.e-8_r8
   431)     end do
   432)     do k=pver-1,pver-npbl+1,-1
   433)        do i=1,ncol
   434)           if (check(i)) then
   435)              vvk = (u(i,k) - u(i,pver))**2 + (v(i,k) - v(i,pver))**2 + fac*ustar(i)**2
   436)              vvk = max(vvk,tiny)
   437)              rino(i,k) = g*(thv(i,k) - tlv(i))*(z(i,k)-z(i,pver))/(thv(i,pver)*vvk)
   438)              if (rino(i,k) >= ricr) then
   439)                 pblh(i) = z(i,k+1) + (ricr - rino(i,k+1))/(rino(i,k) - rino(i,k+1))* &
   440)                      (z(i,k) - z(i,k+1))
   441)                 bge(i) = 2._r8*g/(thv(i,k)+thv(i,k+1))*(thv(i,k)-thv(i,k+1))/(z(i,k)-z(i,k+1))*pblh(i)
   442)                 if (bge(i).lt.0._r8) then
   443)                    bge(i) = 1.e-8_r8
   444)                 endif
   445)                 check(i) = .false.
   446)              end if
   447)           end if
   448)        end do
   449)     end do
   450)     !
   451)     ! PBL height must be greater than some minimum mechanical mixing depth
   452)     ! Several investigators have proposed minimum mechanical mixing depth
   453)     ! relationships as a function of the local friction velocity, u*.  We
   454)     ! make use of a linear relationship of the form h = c u* where c=700.
   455)     ! The scaling arguments that give rise to this relationship most often
   456)     ! represent the coefficient c as some constant over the local coriolis
   457)     ! parameter.  Here we make use of the experimental results of Koracin
   458)     ! and Berkowicz (1988) [BLM, Vol 43] for wich they recommend 0.07/f
   459)     ! where f was evaluated at 39.5 N and 52 N.  Thus we use a typical mid
   460)     ! latitude value for f so that c = 0.07/f = 700.  Also, do not allow 
   461)     ! PBL to exceed some maximum (npbl) number of allowable points
   462)     !
   463)     do i=1,ncol
   464)        if (check(i)) pblh(i) = z(i,pverp-npbl)
   465)        pblh(i) = max(pblh(i),700.0_r8*ustar(i))
   466)        wstar(i) = (max(0._r8,kbfs(i))*g*pblh(i)/thv(i,pver))**onet
   467)     end do
   468)     !
   469)     ! Final requirement on PBL heightis that it must be greater than the depth
   470)     ! of the lowest model level over ocean if there is any cloud diagnosed in 
   471)     ! the lowest model level.  This is to deal with the inadequacies of the 
   472)     ! current "dry" formulation of the boundary layer, where this test is 
   473)     ! used to identify circumstances where there is marine stratus in the 
   474)     ! lowest level, and to provide a weak ventilation of the layer to avoid
   475)     ! a pathology in the cloud scheme (locking in low-level stratiform cloud)
   476)     ! If over an ocean surface, and any cloud is diagnosed in the 
   477)     ! lowest level, set pblh to 50 meters higher than top interface of lowest level
   478)     !
   479)     !  jrm This is being applied everywhere (not just ocean)!
   480)     do i=1,ncol
   481)        ocncldcheck(i) = .false.
   482)        if (cldn(i,pver).ge.0.0_r8) ocncldcheck(i) = .true.
   483)        if (ocncldcheck(i)) pblh(i) = max(pblh(i),zi(i,pver) + 50._r8)
   484)     end do
   485)     !
   486)     return
   487)   end subroutine pblintd

   490)   subroutine austausch_atm(ncol    ,ri      ,s2      ,kvf     )
   491)     !----------------------------------------------------------------------- 
   492)     ! 
   493)     ! Purpose: 
   494)     !  Computes exchange coefficients for free turbulent flows. 
   495)     ! 
   496)     ! Method: 
   497)     !
   498)     ! The free atmosphere diffusivities are based on standard mixing length
   499)     ! forms for the neutral diffusivity multiplied by functns of Richardson
   500)     ! number. K = l^2 * |dV/dz| * f(Ri). The same functions are used for
   501)     ! momentum, potential temperature, and constitutents. 
   502)     !
   503)     ! The stable Richardson num function (Ri>0) is taken from Holtslag and
   504)     ! Beljaars (1989), ECMWF proceedings. f = 1 / (1 + 10*Ri*(1 + 8*Ri))
   505)     ! The unstable Richardson number function (Ri<0) is taken from  CCM1.
   506)     ! f = sqrt(1 - 18*Ri)
   507)     ! 
   508)     ! Author: B. Stevens (rewrite, August 2000)
   509)     ! 
   510)     !-----------------------------------------------------------------------
   511)     !------------------------------Arguments--------------------------------
   512)     !
   513)     ! Input arguments
   514)     !
   515)     integer, intent(in) :: ncol                     ! number of atmospheric columns
   516) 
   517)     real(r8), intent(in)  ::  s2(pcols,pver)        ! shear squared
   518)     real(r8), intent(in)  ::  ri(pcols,pver)        ! richardson no
   519)     !
   520)     ! Output arguments
   521)     !
   522)     real(r8), intent(out) :: kvf(pcols,pverp)       ! coefficient for heat and tracers
   523)     !
   524)     !---------------------------Local workspace-----------------------------
   525)     !
   526)     real(r8) :: fofri                  ! f(ri)
   527)     real(r8) :: kvn                    ! neutral Kv
   528) 
   529)     integer  :: i                      ! longitude index
   530)     integer  :: k                      ! vertical index
   531)     !
   532)     !-----------------------------------------------------------------------
   533)     !
   534)     ! The surface diffusivity is always zero
   535)     !
   536)     kvf(:ncol,pverp) = 0.0_r8
   537)     !
   538)     ! Set the vertical diffusion coefficient above the top diffusion level
   539)     ! Note that nbot_turb != pver is not supported
   540)     !
   541)     kvf(:ncol,1:ntop_turb) = 0.0_r8
   542)     !
   543)     ! Compute the free atmosphere vertical diffusion coefficients: kvh = kvq = kvm. 
   544)     !
   545)     do k = ntop_turb, nbot_turb-1
   546)        do i=1,ncol
   547)           if (ri(i,k) < 0.0_r8) then
   548)              fofri = sqrt(max(1._r8 - 18._r8*ri(i,k),0._r8))
   549)           else 
   550)              fofri = 1.0_r8/(1.0_r8 + 10.0_r8*ri(i,k)*(1.0_r8 + 8.0_r8*ri(i,k)))    
   551)           end if
   552)           kvn = ml2(k)*sqrt(s2(i,k))
   553)           kvf(i,k+1) = max(zkmin,kvn*fofri)
   554)        end do
   555)     end do
   556) 
   557)     return
   558)   end subroutine austausch_atm

   561)   subroutine austausch_pbl(lchnk ,ncol    ,          &
   562)        z       ,kvf     ,kqfs    ,khfs    ,kbfs    , &
   563)        obklen  ,ustar   ,wstar   ,pblh    ,kvm     , &
   564)        kvh     ,cgh     ,cgs     ,tpert   ,qpert   , &
   565)        ktopbl  ,tke     ,bge     ,eddy_scheme)
   566)     !----------------------------------------------------------------------- 
   567)     ! 
   568)     ! Purpose: 
   569)     ! Atmospheric Boundary Layer Computation
   570)     ! 
   571)     ! Method: 
   572)     ! Nonlocal scheme that determines eddy diffusivities based on a
   573)     ! specified boundary layer height and a turbulent velocity scale;
   574)     ! also, countergradient effects for heat and moisture, and constituents
   575)     ! are included, along with temperature and humidity perturbations which
   576)     ! measure the strength of convective thermals in the lower part of the
   577)     ! atmospheric boundary layer.
   578)     !
   579)     ! For more information, see Holtslag, A.A.M., and B.A. Boville, 1993:
   580)     ! Local versus Nonlocal Boundary-Layer Diffusion in a Global Climate
   581)     ! Model. J. Clim., vol. 6., p. 1825--1842.
   582)     !
   583)     ! Updated by Holtslag and Hack to exclude the surface layer from the
   584)     ! definition of the boundary layer Richardson number. Ri is now defined
   585)     ! across the outer layer of the pbl (between the top of the surface
   586)     ! layer and the pbl top) instead of the full pbl (between the surface and
   587)     ! the pbl top). For simiplicity, the surface layer is assumed to be the
   588)     ! region below the first model level (otherwise the boundary layer depth
   589)     ! determination would require iteration).
   590)     !
   591)     ! Author: B. Boville, B. Stevens (rewrite August 2000)
   592)     ! 
   593)     !-----------------------------------------------------------------------
   594) !++ debug code to be removed after validation of PBL codes
   595)      use phys_debug, only: phys_debug_hbdiff1
   596) !++ debug code to be removed after validation of PBL codes
   597)     !------------------------------Arguments--------------------------------
   598)     !
   599)     ! Input arguments
   600)     !
   601)     integer, intent(in) :: lchnk                    ! local chunk index (for debug only)
   602)     integer, intent(in) :: ncol                     ! number of atmospheric columns
   603) 
   604)     real(r8), intent(in) :: z(pcols,pver)           ! height above surface [m]
   605)     real(r8), intent(in) :: kvf(pcols,pverp)        ! free atmospheric eddy diffsvty [m2/s]
   606)     real(r8), intent(in) :: kqfs(pcols)             ! kinematic surf cnstituent flux (kg/m2/s)
   607)     real(r8), intent(in) :: khfs(pcols)             ! kinimatic surface heat flux 
   608)     real(r8), intent(in) :: kbfs(pcols)             ! surface buoyancy flux 
   609)     real(r8), intent(in) :: pblh(pcols)             ! boundary-layer height [m]
   610)     real(r8), intent(in) :: obklen(pcols)           ! Obukhov length
   611)     real(r8), intent(in) :: ustar(pcols)            ! surface friction velocity [m/s]
   612)     real(r8), intent(in) :: wstar(pcols)            ! convective velocity scale [m/s]
   613)     real(r8), intent(in) :: bge(pcols)              ! buoyancy gradient enhancment
   614)     character(len=16), intent(in) :: eddy_scheme
   615) 
   616)     !
   617)     ! Output arguments
   618)     !
   619)     real(r8), intent(out) :: kvm(pcols,pverp)       ! eddy diffusivity for momentum [m2/s]
   620)     real(r8), intent(out) :: kvh(pcols,pverp)       ! eddy diffusivity for heat [m2/s]
   621)     real(r8), intent(out) :: cgh(pcols,pverp)       ! counter-gradient term for heat [J/kg/m]
   622)     real(r8), intent(out) :: cgs(pcols,pverp)       ! counter-gradient star (cg/flux)
   623)     real(r8), intent(out) :: tpert(pcols)           ! convective temperature excess
   624)     real(r8), intent(out) :: qpert(pcols)           ! convective humidity excess
   625) 
   626)     integer,  intent(out) :: ktopbl(pcols)          ! index of first midpoint inside pbl
   627)     real(r8), intent(out) :: tke(pcols,pverp)       ! turbulent kinetic energy (estimated)
   628)     !
   629)     !---------------------------Local workspace-----------------------------
   630)     !
   631)     integer  :: i                       ! longitude index
   632)     integer  :: k                       ! level index
   633) 
   634)     real(r8) :: phiminv(pcols)          ! inverse phi function for momentum
   635)     real(r8) :: phihinv(pcols)          ! inverse phi function for heat
   636)     real(r8) :: wm(pcols)               ! turbulent velocity scale for momentum
   637)     real(r8) :: zp(pcols)               ! current level height + one level up 
   638)     real(r8) :: fak1(pcols)             ! k*ustar*pblh     
   639)     real(r8) :: fak2(pcols)             ! k*wm*pblh
   640)     real(r8) :: fak3(pcols)             ! fakn*wstar/wm
   641)     real(r8) :: pblk(pcols)             ! level eddy diffusivity for momentum
   642)     real(r8) :: pr(pcols)               ! Prandtl number for eddy diffusivities
   643)     real(r8) :: zl(pcols)               ! zmzp / Obukhov length
   644)     real(r8) :: zh(pcols)               ! zmzp / pblh
   645)     real(r8) :: zzh(pcols)              ! (1-(zmzp/pblh))**2
   646)     real(r8) :: zmzp                    ! level height halfway between zm and zp
   647)     real(r8) :: term                    ! intermediate calculation
   648)     real(r8) :: kve                     ! diffusivity at entrainment layer in unstable cases 
   649) 
   650)     logical  :: unstbl(pcols)           ! pts w/unstbl pbl (positive virtual ht flx)
   651)     logical  :: pblpt(pcols)            ! pts within pbl
   652)     !
   653)     ! Initialize height independent arrays
   654)     !
   655) 
   656)     !drb initialize variables for runtime error checking
   657)     kvm = 0._r8	
   658)     kvh = 0._r8
   659)     kve = 0._r8
   660)     cgh = 0._r8
   661)     cgs = 0._r8
   662)     tpert = 0._r8
   663)     qpert = 0._r8
   664)     ktopbl = 0._r8
   665)     tke = 0._r8
   666) 
   667)     do i=1,ncol
   668)        unstbl(i) = (kbfs(i) > 0._r8)
   669)        pblk(i) = 0.0_r8
   670)        fak1(i) = ustar(i)*pblh(i)*vk
   671)        if (unstbl(i)) then
   672)           phiminv(i) = (1._r8 - binm*pblh(i)/obklen(i))**onet
   673)           phihinv(i) = sqrt(1._r8 - binh*pblh(i)/obklen(i))
   674)           wm(i)      = ustar(i)*phiminv(i)
   675)           fak2(i)    = wm(i)*pblh(i)*vk
   676)           fak3(i)    = fakn*wstar(i)/wm(i)
   677)           tpert(i)   = max(khfs(i)*fak/wm(i),0._r8)
   678)           qpert(i)   = max(kqfs(i)*fak/wm(i),0._r8)
   679)        else
   680)           tpert(i)   = max(khfs(i)*fak/ustar(i),0._r8)
   681)           qpert(i)   = max(kqfs(i)*fak/ustar(i),0._r8)
   682)        end if
   683)     end do
   684)     !
   685)     ! Initialize output arrays with free atmosphere values
   686)     !
   687)     do k=1,pverp
   688)        do i=1,ncol
   689)           kvm(i,k) = kvf(i,k)
   690)           kvh(i,k) = kvf(i,k)
   691)           cgh(i,k) = 0.0_r8
   692)           cgs(i,k) = 0.0_r8
   693)        end do
   694)     end do
   695)     !
   696)     ! Main level loop to compute the diffusivities and counter-gradient terms. These terms are 
   697)     ! only calculated at points determined to be in the interior of the pbl (pblpt(i)==.true.),
   698)     ! and then calculations are directed toward regime: stable vs unstable, surface vs outer 
   699)     ! layer.
   700)     !
   701)     do k=pver,pver-npbl+2,-1
   702)        do i=1,ncol
   703)           pblpt(i) = (z(i,k) < pblh(i))
   704)           if (pblpt(i)) then
   705)              ktopbl(i) = k
   706)              zp(i)  = z(i,k-1)
   707)              if (zkmin == 0.0_r8 .and. zp(i) > pblh(i)) zp(i) = pblh(i)
   708)              zmzp    = 0.5_r8*(z(i,k) + zp(i)) ! we think this is an approximation to the interface height (where KVs are calculated)
   709)              zh(i)   = zmzp/pblh(i)
   710)              zl(i)   = zmzp/obklen(i)
   711)              zzh(i)  = zh(i)*max(0._r8,(1._r8 - zh(i)))**2
   712)              if (unstbl(i)) then
   713)                 if (zh(i) < sffrac) then
   714)                    term     = (1._r8 - betam*zl(i))**onet
   715)                    pblk(i)  = fak1(i)*zzh(i)*term
   716)                    pr(i)    = term/sqrt(1._r8 - betah*zl(i))
   717)                 else
   718)                    pblk(i)  = fak2(i)*zzh(i)
   719)                    pr(i)    = phiminv(i)/phihinv(i) + ccon*fak3(i)/fak
   720)                    cgs(i,k) = fak3(i)/(pblh(i)*wm(i))
   721)                    cgh(i,k) = khfs(i)*cgs(i,k)*cpair
   722)                 end if
   723)              else
   724)                 if (zl(i) <= 1._r8) then
   725)                    pblk(i) = fak1(i)*zzh(i)/(1._r8 + betas*zl(i))
   726)                 else
   727)                    pblk(i) = fak1(i)*zzh(i)/(betas + zl(i))
   728)                 end if
   729)                 pr(i)    = 1._r8
   730)              end if
   731)              kvm(i,k) = max(pblk(i),kvf(i,k))
   732)              kvh(i,k) = max(pblk(i)/pr(i),kvf(i,k))
   733)           end if
   734)        end do
   735)     end do
   736) 
   737) !++ debug code to be removed after validation of PBL codes
   738)     call phys_debug_hbdiff1(lchnk, pblh, zl, zh)
   739) !++ debug code to be removed after validation of PBL codes
   740) 
   741)     !
   742)     ! Check whether last allowed midpoint is within pbl
   743)     !
   744) 
   745)     if  ( eddy_scheme .eq. 'HBR' ) then  
   746)        ! apply new diffusivity at entrainment zone 
   747)        do i = 1,ncol
   748)           if (bge(i) > 1.e-7_r8) then
   749)              k = ktopbl(i)
   750)              kve = 0.2_r8*(wstar(i)**3+5._r8*ustar(i)**3)/bge(i)
   751)              kvm(i,k) = kve
   752)              kvh(i,k) = kve
   753)           end if
   754)        end do
   755)     end if
   756) 
   757)     ! Crude estimate of tke (tke=0 above boundary layer)
   758)     do k = max(pverp-npbl,2),pverp
   759)        do i = 1, ncol
   760)           if (z(i,k-1) < pblh(i)) then
   761)              tke(i,k) = ( kvm(i,k) / pblh(i) ) ** 2
   762)           endif
   763)        end do
   764)     end do
   765)     return
   766)   end subroutine austausch_pbl
