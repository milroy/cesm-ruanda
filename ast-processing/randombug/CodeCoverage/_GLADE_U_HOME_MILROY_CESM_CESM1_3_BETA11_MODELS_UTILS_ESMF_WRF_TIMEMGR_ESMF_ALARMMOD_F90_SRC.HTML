   157) ! !IROUTINE: ESMF_AlarmSet - Initializes an alarm
   158) 
   159) ! !INTERFACE:
   160)       subroutine ESMF_AlarmSet(alarm, name, RingTime, RingInterval, &
   161)                                StopTime, Enabled, rc)
   162) 
   163) ! !ARGUMENTS:
   164)       type(ESMF_Alarm), intent(inout) :: alarm
   165)       character(len=*), intent(in), optional :: name
   166)       type(ESMF_Time), intent(in), optional :: RingTime
   167)       type(ESMF_TimeInterval), intent(in), optional :: RingInterval
   168)       type(ESMF_Time), intent(in), optional :: StopTime
   169)       logical, intent(in), optional :: Enabled
   170)       integer, intent(out), optional :: rc
   171) 
   172) ! !DESCRIPTION:
   173) !     Initializes an {\tt ESMF\_Alarm}
   174) !
   175) !     The arguments are:
   176) !     \begin{description}
   177) !     \item[alarm]
   178) !          The object instance to initialize
   179) !     \item[{[RingTime]}]
   180) !          Optional ring time for one-shot or first repeating alarm
   181) !     \item[{[RingInterval]}]
   182) !          Optional ring interval for repeating alarms
   183) !     \item[{[StopTime]}]
   184) !          Optional stop time for repeating alarms
   185) !     \item[Enabled]
   186) !          Alarm enabled/disabled
   187) !     \item[{[rc]}]
   188) !          Return code; equals {\tt ESMF\_SUCCESS} if there are no errors.
   189) !     \end{description}
   190) !
   191) ! !REQUIREMENTS:
   192) !     TMG4.1, TMG4.7
   193) !EOP
   194)       IF ( ASSOCIATED( alarm%alarmint ) ) THEN
   195)         alarm%alarmint%RingTimeSet = .FALSE.
   196)         alarm%alarmint%RingIntervalSet = .FALSE.
   197)         alarm%alarmint%StopTimeSet = .FALSE.
   198)         IF ( PRESENT( name ) ) THEN
   199)           alarm%alarmint%name = name
   200)         END IF
   201)         IF ( PRESENT( RingInterval ) ) THEN
   202)           alarm%alarmint%RingInterval = RingInterval
   203)           alarm%alarmint%RingIntervalSet = .TRUE.
   204)         ENDIF
   205)         IF ( PRESENT( RingTime ) ) THEN
   206)           alarm%alarmint%RingTime = RingTime
   207)           alarm%alarmint%RingTimeSet = .TRUE.
   208)         ENDIF
   209)         IF ( PRESENT( StopTime ) ) THEN
   210)           alarm%alarmint%StopTime = StopTime
   211)           alarm%alarmint%StopTimeSet = .TRUE.
   212)         ENDIF
   213)         alarm%alarmint%Enabled = .TRUE.
   214)         IF ( PRESENT( Enabled ) ) THEN
   215)           alarm%alarmint%Enabled = Enabled
   216)         ENDIF
   217)         IF ( PRESENT( rc ) ) THEN
   218)           rc = ESMF_SUCCESS
   219)         ENDIF
   220)         alarm%alarmint%Ringing = .FALSE.
   221)         alarm%alarmint%Enabled = .TRUE.
   222)       ELSE
   223)         IF ( PRESENT( rc ) ) rc = ESMF_FAILURE
   224)       ENDIF
   225) 
   226)       end subroutine ESMF_AlarmSet

   321) ! !ARGUMENTS:

   326) ! !DESCRIPTION:
   327) !     Get an {\tt ESMF\_Alarm}'s time to ring
   328) !
   329) !     The arguments are:
   330) !     \begin{description}
   331) !     \item[alarm]
   332) !          The object instance to get the ring time
   333) !     \item[RingTime]
   334) !          The {\tt ESMF\_Alarm}'s ring time
   335) !     \item[{[rc]}]
   336) !          Return code; equals {\tt ESMF\_SUCCESS} if there are no errors.

   339) ! !REQUIREMENTS:
   340) !     TMG4.7, TMG4.8
   341) !EOP
   342)       type(ESMF_Time) :: PrevRingTime
   343)       type(ESMF_TimeInterval) :: RingInterval
   344)       integer :: ierr
   345) 
   346)       IF ( ASSOCIATED( alarm%alarmint ) ) THEN
   347)         IF ( alarm%alarmint%RingIntervalSet )THEN
   348)            PrevRingTime = alarm%alarmint%PrevRingTime

   355)            IF ( PRESENT( rc ) ) rc = ESMF_SUCCESS
   356)         ELSE IF ( alarm%alarmint%RingTimeSet )THEN
   357)            RingTime = alarm%alarmint%RingTime
   358)            IF ( PRESENT( rc ) ) rc = ESMF_SUCCESS
   359)         ELSE
   360)            IF ( PRESENT( rc ) ) rc = ESMF_FAILURE
   361)         END IF
