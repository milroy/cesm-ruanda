    97) 
    98)    ! Local variables
    99)    integer :: unitn, ierr, i
   100)    character(len=*), parameter :: subname = 'chem_surfvals_readnl'
   101)

   110)                                ghg_yearStart_model, ghg_yearStart_data

   112)    namelist /chem_surfvals_nl/ flbc_type, flbc_cycle_yr, flbc_fixed_ymd, flbc_fixed_tod, flbc_list, flbc_file

   176) subroutine chem_surfvals_init()
   177) 
   178) !----------------------------------------------------------------------- 
   179) ! 
   180) ! Purpose: 
   181) ! Initialize the ramp options that are controlled by namelist input.
   182) ! Set surface values at initial time.
   183) ! N.B. This routine must be called after the time manager has been initialized
   184) !      since chem_surfvals_set calls time manager methods.
   185) ! 
   186) ! Author: B. Eaton - merged code from parse_namelist and rampnl_ghg.
   187) ! 
   188) !-----------------------------------------------------------------------
   189) 
   190)    use infnan,  only : posinf, assignment(=)
   191)    use mo_flbc, only : flbc_inti
   192) 
   193)    !---------------------------Local variables-----------------------------
   194)    integer :: yr, mon, day, ncsec
   195)    !-----------------------------------------------------------------------
   196) 
   197)    if (scenario_ghg == 'FIXED') then
   198)       doRamp_ghg = .false.
   199)       ramp_just_co2 = .false.
   200)       if (masterproc) &
   201)          write(iulog,*)'chem_surfvals_init: ghg surface values are fixed as follows'
   202) 
   203)    else if (scenario_ghg == 'RAMPED') then
   204)       doRamp_ghg = .true.
   205)       ramp_just_co2 = .false.
   206)       call ghg_ramp_read
   207) 
   208)       fixYear_ghg = rampYear_ghg     ! set private member to namelist var
   209)       if (masterproc) then
   210)          if ( fixYear_ghg > 0 ) then
   211)             write(iulog,*) '  FIXED values from year ',fixYear_ghg
   212)          else
   213)             write(iulog,*) '  RAMPED values initialized to'
   214)          end if
   215)       end if
   216)       call chem_surfvals_set()
   217) 
   218)    else if (scenario_ghg == 'RAMP_CO2_ONLY') then
   219)       if(ramp_co2_start_ymd == 0) then
   220)          ! by default start the ramp at the initial run time
   221)          call get_start_date(yr, mon, day, ncsec)
   222)          ramp_co2_start_ymd = yr*10000 + mon*100 + day
   223)       end if
   224)       co2_start = ramp_co2_start_ymd
   225) 
   226)       if(ramp_co2_annual_rate <= -100.0_r8) then
   227)          write(iulog,*) 'RAMP_CO2:  invalid ramp_co2_annual_rate= ',ramp_co2_annual_rate
   228)          call endrun ('chem_surfvals_init: RAMP_CO2_ANNUAL_RATE must be greater than -100.0')
   229)       end if
   230) 
   231)       doRamp_ghg = .true.
   232)       ramp_just_co2 = .true.
   233)       co2_base = co2vmr        ! save initial setting 
   234)       if (masterproc) &
   235)            write(iulog,*) '  RAMPED values initialized to'
   236) 
   237)       co2_daily_factor = (ramp_co2_annual_rate*0.01_r8+1.0_r8)**(1.0_r8/365.0_r8)
   238) 
   239)       if(ramp_co2_cap > 0.0_r8) then  
   240)          co2_limit = ramp_co2_cap * co2_base
   241)       else                                  ! if no cap/floor specified, provide default
   242)          if(ramp_co2_annual_rate < 0.0_r8) then
   243)             co2_limit = 0.0_r8
   244)          else
   245)             co2_limit = posinf
   246)          end if
   247)       end if
   248)       if((ramp_co2_annual_rate<0.0_r8 .and. co2_limit>co2_base) .or. &
   249)          (ramp_co2_annual_rate>0.0_r8 .and. co2_limit<co2_base)) then
   250)          write(iulog,*) 'RAMP_CO2: ramp_co2_cap is unreachable'
   251)          write(iulog,*) 'RAMP_CO2: ramp_co2_annual_rate= ',ramp_co2_annual_rate,' ramp_co2_cap= ',ramp_co2_cap
   252)          call endrun('chem_surfvals_init:  ramp_co2_annual_rate and ramp_co2_cap incompatible')
   253)       end if
   254) 
   255)       call chem_surfvals_set()
   256)    else if (scenario_ghg == 'CHEM_LBC_FILE') then
   257)       ! set by lower boundary conditions file
   258)       call flbc_inti( flbc_file, flbc_list, flbc_timing, co2vmr, ch4vmr, n2ovmr, f11vmr, f12vmr )
   259)       call chem_surfvals_set()
   260)    else
   261)       call endrun ('chem_surfvals_init: input namelist SCENARIO_GHG must be set to either FIXED, RAMPED, RAMP_CO2_ONLY, &
   262)                     or CHEM_LBC_FILE')
   263)    endif
   264) 
   265)    if (masterproc) then
   266)       write(iulog,*) '  co2 volume mixing ratio = ',co2vmr
   267)       write(iulog,*) '  ch4 volume mixing ratio = ',ch4vmr
   268)       write(iulog,*) '  n2o volume mixing ratio = ',n2ovmr
   269)       write(iulog,*) '  f11 volume mixing ratio = ',f11vmr
   270)       write(iulog,*) '  f12 volume mixing ratio = ',f12vmr
   271)    end if
   272) 
   273) end subroutine chem_surfvals_init

   277) subroutine ghg_ramp_read()
   278) 
   279) !----------------------------------------------------------------------- 
   280) ! 
   281) ! Purpose: 
   282) ! Read ramped greenhouse gas surface data.  
   283) ! 
   284) ! Author: T. Henderson
   285) ! 
   286) !-----------------------------------------------------------------------
   287) 
   288)    use ioFileMod, only: getfil
   289) #if ( defined SPMD )
   290)    use mpishorthand, only: mpicom, mpiint, mpir8
   291) #endif
   292)    character(len=*), parameter :: subname = 'ghg_ramp_read'
   293) 
   294) !---------------------------Local variables-----------------------------
   295)    integer :: ncid
   296)    integer :: co2_id
   297)    integer :: ch4_id
   298)    integer :: n2o_id
   299)    integer :: f11_id
   300)    integer :: f12_id
   301)    integer :: adj_id
   302)    integer :: date_id
   303)    integer :: time_id
   304)    integer :: ierror
   305)    character(len=256) :: locfn          ! netcdf local filename to open
   306) 
   307)    if (masterproc) then
   308)      call getfil (bndtvghg, locfn, 0)
   309)      call handle_ncerr( nf90_open (trim(locfn), NF90_NOWRITE, ncid),subname,__LINE__)
   310) 
   311)      write(iulog,*)'GHG_RAMP_READ:  reading ramped greenhouse gas surface data from file ',trim(locfn)
   312) 
   313)      call handle_ncerr( nf90_inq_varid( ncid, 'date', date_id ),subname,__LINE__)
   314)      call handle_ncerr( nf90_inq_varid( ncid, 'CO2', co2_id ),subname,__LINE__)
   315)      call handle_ncerr( nf90_inq_varid( ncid, 'CH4', ch4_id ),subname,__LINE__)
   316)      call handle_ncerr( nf90_inq_varid( ncid, 'N2O', n2o_id ),subname,__LINE__)
   317)      call handle_ncerr( nf90_inq_varid( ncid, 'f11', f11_id ),subname,__LINE__)
   318)      call handle_ncerr( nf90_inq_varid( ncid, 'f12', f12_id ),subname,__LINE__)
   319)      call handle_ncerr( nf90_inq_varid( ncid, 'adj', adj_id ),subname,__LINE__)
   320)      call handle_ncerr( nf90_inq_dimid( ncid, 'time', time_id ),subname,__LINE__)
   321)      call handle_ncerr( nf90_inquire_dimension( ncid, time_id, len=ntim ),subname,__LINE__)
   322) 
   323)    endif
   324) #if (defined SPMD )
   325)    call mpibcast (ntim, 1, mpiint, 0, mpicom)
   326) #endif
   327)    ! these arrays are never deallocated
   328)    allocate ( yrdata(ntim), co2(ntim), ch4(ntim), n2o(ntim),    &
   329)                  f11(ntim), f12(ntim), adj(ntim), stat=ierror )
   330)    if (ierror /= 0) then
   331)      write(iulog,*)'GHG_RAMP_READ:  ERROR, allocate() failed!'
   332)      call endrun
   333)    endif
   334)    if (masterproc) then
   335)      call handle_ncerr( nf90_get_var (ncid, date_id, yrdata ),subname,__LINE__)
   336)      yrdata = yrdata / 10000
   337)      call handle_ncerr( nf90_get_var (ncid, co2_id, co2 ),subname,__LINE__)
   338)      call handle_ncerr( nf90_get_var (ncid, ch4_id, ch4 ),subname,__LINE__)
   339)      call handle_ncerr( nf90_get_var (ncid, n2o_id, n2o ),subname,__LINE__)
   340)      call handle_ncerr( nf90_get_var (ncid, f11_id, f11 ),subname,__LINE__)
   341)      call handle_ncerr( nf90_get_var (ncid, f12_id, f12 ),subname,__LINE__)
   342)      call handle_ncerr( nf90_get_var (ncid, adj_id, adj ),subname,__LINE__)
   343)      call handle_ncerr( nf90_close (ncid),subname,__LINE__)
   344)      write(iulog,*)'GHG_RAMP_READ:  successfully read ramped greenhouse gas surface data from years ',&
   345) 	yrdata(1),' through ',yrdata(ntim)
   346)    endif
   347) #if (defined SPMD )
   348)    call mpibcast (co2, ntim, mpir8, 0, mpicom)
   349)    call mpibcast (ch4, ntim, mpir8, 0, mpicom)
   350)    call mpibcast (n2o, ntim, mpir8, 0, mpicom)
   351)    call mpibcast (f11, ntim, mpir8, 0, mpicom)
   352)    call mpibcast (f12, ntim, mpir8, 0, mpicom)
   353)    call mpibcast (adj, ntim, mpir8, 0, mpicom)
   354)    call mpibcast (yrdata, ntim, mpiint, 0, mpicom)
   355) #endif
   356) 
   357)    return
   358) 
   359) end subroutine ghg_ramp_read

   363) function chem_surfvals_get(name)
   364)   use physconst,    only: mwdry, mwco2
   365) 
   366)   character(len=*), intent(in) :: name
   367) 
   368)   real(r8) :: rmwco2 
   369)   real(r8) :: chem_surfvals_get
   370) 
   371)   rmwco2 = mwco2/mwdry    ! ratio of molecular weights of co2 to dry air
   372)   select case (name)
   373)   case ('CO2VMR')
   374)      chem_surfvals_get = co2vmr
   375)   case ('CO2MMR')
   376)      chem_surfvals_get = rmwco2 * co2vmr
   377)   case ('N2OVMR')
   378)      chem_surfvals_get = n2ovmr
   379)   case ('CH4VMR')
   380)      chem_surfvals_get = ch4vmr
   381)   case ('F11VMR')
   382)      chem_surfvals_get = f11vmr
   383)   case ('F12VMR')
   384)      chem_surfvals_get = f12vmr
   385)   case ('O2MMR')
   386)      chem_surfvals_get = o2mmr
   387)   case default
   388)      call endrun('chem_surfvals_get does not know name')
   389)   end select
   390) 
   391) end function chem_surfvals_get

   396) function chem_surfvals_co2_rad(vmr_in)
   397)  
   398)    ! Return the value of CO2 (as mmr) that is radiatively active.
   399) 
   400)    ! This method is used by ghg_data to set the prescribed value of CO2 in
   401)    ! the physics buffer.  If the user has set the co2vmr_rad namelist
   402)    ! variable then that value will override either the value set by the
   403)    ! co2vmr namelist variable, or the values time interpolated from a
   404)    ! dataset.
   405)    
   406)    ! This method is also used by cam_history to write the radiatively active
   407)    ! CO2 to the history file.  The optional argument allows returning the
   408)    ! value as vmr.
   409) 
   410)    use physconst,    only: mwdry, mwco2
   411) 
   412)    ! Arguments
   413)    logical, intent(in), optional :: vmr_in  ! return CO2 as vmr
   414) 
   415)    ! Return value
   416)    real(r8) :: chem_surfvals_co2_rad
   417) 
   418)    ! Local variables
   419)    real(r8) :: convert_vmr      ! convert vmr to desired output
   420)    !-----------------------------------------------------------------------
   421) 
   422)    ! by default convert vmr to mmr
   423)    convert_vmr = mwco2/mwdry    ! ratio of molecular weights of co2 to dry air
   424)    if (present(vmr_in)) then
   425)       ! if request return vmr
   426)       if (vmr_in) convert_vmr = 1.0_r8
   427)    end if
   428) 
   429)    if (co2vmr_rad > 0._r8) then
   430)       chem_surfvals_co2_rad = convert_vmr * co2vmr_rad
   431)    else                           
   432)       chem_surfvals_co2_rad = convert_vmr * co2vmr     
   433)    end if
   434) 
   435) end function chem_surfvals_co2_rad

   439) subroutine chem_surfvals_set()
   440) 
   441)    use ppgrid,         only: begchunk, endchunk
   442)    use mo_flbc,        only: flbc_gmean_vmr, flbc_chk
   443) 
   444) !---------------------------Local variables-----------------------------
   445) 
   446)    integer  :: yr, mon, day, ncsec ! components of a date
   447)    integer  :: ncdate              ! current date in integer format [yyyymmdd]
   448)    
   449)    if ( doRamp_ghg ) then
   450)       if(ramp_just_co2) then
   451)          call chem_surfvals_set_co2()
   452)       else
   453)          call chem_surfvals_set_all()
   454)       end if
   455)    elseif (scenario_ghg == 'CHEM_LBC_FILE') then

   478) subroutine chem_surfvals_set_all()
   479) !----------------------------------------------------------------------- 
   480) ! 
   481) ! Purpose: 
   482) ! Computes greenhouse gas volume mixing ratios via interpolation of
   483) ! yearly input data.
   484) ! 
   485) ! Author: B. Eaton - updated ramp_ghg for use in chem_surfvals module
   486) ! 
   487) !-----------------------------------------------------------------------
   488)    use interpolate_data, only: get_timeinterp_factors
   489) 
   490) !---------------------------Local variables-----------------------------
   491) 
   492)    integer yrmodel           ! model year
   493)    integer nyrm              ! year index
   494)    integer nyrp              ! year index
   495)    integer :: yr, mon, day   ! components of a date
   496)    integer :: ncdate         ! current date in integer format [yyyymmdd]
   497)    integer :: ncsec          ! current time of day [seconds]

   499)    real(r8) :: calday            ! current calendar day
   500)    real(r8) doymodel             ! model day of year

   502)    real(r8) doydatap             ! day or year for input data yrdata(nyrp)
   503)    real(r8) deltat               ! delta time
   504)    real(r8) fact1, fact2         ! time interpolation factors
   505)    real(r8) cfcscl               ! cfc scale factor for f11

   507)    integer yearRan_model         ! model ran year

   509) ! ---------------------------------------------------------------------
   510) !
   511)    calday = get_curr_calday()
   512)    call get_curr_date(yr, mon, day, ncsec)
   513)    ncdate = yr*10000 + mon*100 + day
   514) !
   515) ! determine ghg_use_calendar      
   516) !
   517)    if ( ghg_yearStart_model > 0 .and. ghg_yearStart_data > 0 ) then
   518)       ghg_use_calendar = .false.
   519)    else
   520)       ghg_use_calendar = .true.
   521)    end if
   522) !
   523) ! determine index into input data
   524) !
   525)    if ( fixYear_ghg > 0) then
   526)       yrmodel  = fixYear_ghg
   527)       nyrm = fixYear_ghg - yrdata(1) + 1
   528)    else
   529)       if ( ghg_use_calendar) then
   530)          yrmodel  = yr          
   531)          nyrm = yr - yrdata(1) + 1
   532)       else 
   533)          yearRan_model = yr - ghg_yearStart_model
   534)          if ( yearRan_model < 0 ) then
   535)             call endrun('chem_surfvals_set_all: incorrect ghg_yearStart_model')
   536)          endif
   537)          yrmodel  = yearRan_model + ghg_yearStart_data
   538)  
   539)          nyrm = ghg_yearStart_data + yearRan_model - yrdata(1) + 1
   540)       end if
   541)    end if
   542) 
   543)    nyrp       = nyrm + 1
   544) !
   545) ! if current date is before yrdata(1), quit
   546) !
   547)    if (nyrm < 1) then
   548)       write(iulog,*)'chem_surfvals_set_all: data time index is out of bounds'
   549)       write(iulog,*)'nyrm = ',nyrm,' nyrp= ',nyrp, ' ncdate= ', ncdate
   550)       call endrun
   551)    endif
   552) !
   553) ! if current date later than yrdata(ntim), call endrun.
   554) ! if want to use ntim values - uncomment the following lines

   556) !
   557)    if (nyrp > ntim) then
   558)       call endrun ('chem_surfvals_set_all: error - current date is past the end of valid data')
   559) !         write(iulog,*)'chem_surfvals_set_all: using ghg data for ',yrdata(ntim)
   560) !         co2vmr = co2(ntim)*1.e-06
   561) !         ch4vmr = ch4(ntim)*1.e-09
   562) !         n2ovmr = n2o(ntim)*1.e-09

   565) !         co2mmr = rmwco2 * co2vmr
   566) !         return

   570) ! of interpolation factors to within 32-bit roundoff
   571) ! assume that day of year is 1 for all input data
   572) !
   573)    doymodel = yrmodel*365._r8    + calday
   574)    doydatam = yrdata(nyrm)*365._r8 + 1._r8

   586)    co2vmr = (co2(nyrm)*fact1 + co2(nyrp)*fact2)*1.e-06_r8

   592)    f12vmr = (f12(nyrm)*fact1 + f12(nyrp)*fact2)*1.e-12_r8
