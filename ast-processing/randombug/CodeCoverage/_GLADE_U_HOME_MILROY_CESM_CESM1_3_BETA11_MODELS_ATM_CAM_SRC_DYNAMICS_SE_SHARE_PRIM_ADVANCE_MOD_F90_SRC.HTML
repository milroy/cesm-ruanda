  1746)                  lap_p=laplace_sphere_wk(elem(ie)%state%T(:,:,k,nt),deriv,elem(ie),var_coef=.false.)
  1747)                  lap_v=vlaplace_sphere_wk(elem(ie)%state%v(:,:,:,k,nt),deriv,elem(ie),var_coef=.false.)
  1748)               endif
  1749)               nu_scale_top = 1
  1750)               if (k==1) nu_scale_top=4
  1751)               if (k==2) nu_scale_top=2
  1752)               
  1753)               do j=1,np
  1754)                  do i=1,np
  1755)                     if (nu_p==0) then
  1756)                        ! normalize so as to conserve IE  
  1757)                        ! scale by 1/rho (normalized to be O(1))
  1758)                        ! dp/dn = O(ps0)*O(delta_eta) = O(ps0)/O(nlev)
  1759)                        dpdn = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
  1760)                             ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*elem(ie)%state%ps_v(i,j,nt)  
  1761)                        dpdn0 = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
  1762)                             ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*hvcoord%ps0
  1763)                        nu_scale = dpdn0/dpdn
  1764)                     endif
  1765) 
  1766)                     ! biharmonic terms need a negative sign:
  1767)                     if (nu_top>0 .and. k<=3) then
  1768)                        utens_tmp=(-nu*vtens(i,j,1,k,ie) + nu_scale_top*nu_top*lap_v(i,j,1))
  1769)                        vtens_tmp=(-nu*vtens(i,j,2,k,ie) + nu_scale_top*nu_top*lap_v(i,j,2))
  1770)                        ptens_tmp=nu_scale*(-nu_s*ptens(i,j,k,ie) + nu_scale_top*nu_top*lap_p(i,j) )
  1771)                     else
  1772)                        utens_tmp=-nu*vtens(i,j,1,k,ie)
  1773)                        vtens_tmp=-nu*vtens(i,j,2,k,ie)
  1774)                        ptens_tmp=-nu_scale*nu_s*ptens(i,j,k,ie)
  1775)                     endif
  1776)                     
  1777)                     ptens(i,j,k,ie) = ptens_tmp  
  1778)                     vtens(i,j,1,k,ie)=utens_tmp
  1779)                     vtens(i,j,2,k,ie)=vtens_tmp
  1780)                  enddo
  1781)               enddo
  1782)            enddo
  1783)            
  1784)            pstens(:,:,ie)  =  -nu_p*pstens(:,:,ie)
  1785)            kptr=0
  1786)            call edgeVpack(edge3, ptens(:,:,:,ie),nlev,kptr,elem(ie)%desc)
  1787)            kptr=nlev
  1788)            call edgeVpack(edge3,vtens(:,:,:,:,ie),2*nlev,kptr,elem(ie)%desc)
  1789)            kptr=3*nlev
  1790)            call edgeVpack(edge3,pstens(:,:,ie),1,kptr,elem(ie)%desc)
  1791)         enddo
  1792)         
  1793)         
  1794)         call bndry_exchangeV(hybrid,edge3)
  1795)         
  1796)         do ie=nets,nete
  1797)            
  1798)            kptr=0
  1799)            call edgeVunpack(edge3, ptens(:,:,:,ie), nlev, kptr, elem(ie)%desc)
  1800)            kptr=nlev
  1801)            call edgeVunpack(edge3, vtens(:,:,:,:,ie), 2*nlev, kptr, elem(ie)%desc)
  1802) 
  1803)            
  1804)            ! apply inverse mass matrix, accumulate tendencies
  1805) #if (defined ELEMENT_OPENMP)
  1806) !$omp parallel do private(k,i,j,v1,v2,heating)
  1807) #endif
  1808)            do k=1,nlev
  1809)               vtens(:,:,1,k,ie)=dt*vtens(:,:,1,k,ie)*elem(ie)%rspheremp(:,:)
  1810)               vtens(:,:,2,k,ie)=dt*vtens(:,:,2,k,ie)*elem(ie)%rspheremp(:,:)
  1811)               ptens(:,:,k,ie)=dt*ptens(:,:,k,ie)*elem(ie)%rspheremp(:,:)
  1812)            enddo
  1813) 
  1814)            ! apply hypervis to u -> u+utens:  
  1815)            ! E0 = dpdn * .5*u dot u + dpdn * T  + dpdn*PHIS
  1816)            ! E1 = dpdn * .5*(u+utens) dot (u+utens) + dpdn * (T-X) + dpdn*PHIS
  1817)            ! E1-E0:   dpdn (u dot utens) + dpdn .5 utens dot utens   - dpdn X
  1818)            !      X = (u dot utens) + .5 utens dot utens
  1819)            !  alt:  (u+utens) dot utens
  1820)            do k=1,nlev
  1821)               do j=1,np
  1822)                  do i=1,np
  1823)                     ! update v first (gives better results than updating v after heating)
  1824)                     elem(ie)%state%v(i,j,:,k,nt)=elem(ie)%state%v(i,j,:,k,nt) + &
  1825)                          vtens(i,j,:,k,ie)
  1826) 
  1827)                     v1=elem(ie)%state%v(i,j,1,k,nt)
  1828)                     v2=elem(ie)%state%v(i,j,2,k,nt)
  1829)                     heating = (vtens(i,j,1,k,ie)*v1  + vtens(i,j,2,k,ie)*v2 ) 
  1830)                     elem(ie)%state%T(i,j,k,nt)=elem(ie)%state%T(i,j,k,nt) &
  1831)                          +ptens(i,j,k,ie)-heating/cp
  1832) 
  1833)                  enddo
  1834)               enddo
  1835)            enddo              
  1836) 
  1837)            if (nu_p>0) then
  1838)               kptr=3*nlev
  1839)               call edgeVunpack(edge3, pstens(:,:,ie), 1, kptr, elem(ie)%desc)
  1840)               pstens(:,:,ie)=dt*pstens(:,:,ie)*elem(ie)%rspheremp(:,:)
  1841)               elem(ie)%state%ps_v(:,:,nt)=elem(ie)%state%ps_v(:,:,nt) + pstens(:,:,ie)
  1842)            endif
  1843)            
  1844)         enddo
  1845) #ifdef DEBUGOMP
  1846) #if (! defined ELEMENT_OPENMP)
  1847) !$OMP BARRIER
  1848) #endif
  1849) #endif
  1850)      enddo
  1851)   endif
  1852)   
  1853)   call t_stopf('advance_hypervis')
  1854)   
  1855)   end subroutine advance_hypervis
