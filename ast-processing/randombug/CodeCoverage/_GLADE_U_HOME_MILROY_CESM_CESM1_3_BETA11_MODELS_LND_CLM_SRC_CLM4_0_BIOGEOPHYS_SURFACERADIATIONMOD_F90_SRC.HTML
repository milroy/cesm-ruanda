    43) ! !DESCRIPTION: 
    44) ! Solar fluxes absorbed by vegetation and ground surface
    45) ! Note possible problem when land is on different grid than atmosphere.
    46) ! Land may have sun above the horizon (coszen > 0) but atmosphere may
    47) ! have sun below the horizon (forc_solad = 0 and forc_solai = 0). This is okay
    48) ! because all fluxes (absorbed, reflected, transmitted) are multiplied
    49) ! by the incoming flux and all will equal zero.
    50) ! Atmosphere may have sun above horizon (forc_solad > 0 and forc_solai > 0) but
    51) ! land may have sun below horizon. This is okay because fabd, fabi,
    52) ! ftdd, ftid, and ftii all equal zero so that sabv=sabg=fsa=0. Also,
    53) ! albd and albi equal one so that fsr=forc_solad+forc_solai. In other words, all
    54) ! the radiation is reflected. NDVI should equal zero in this case.
    55) ! However, the way the code is currently implemented this is only true
    56) ! if (forc_solad+forc_solai)|vis = (forc_solad+forc_solai)|nir.
    57) ! Output variables are parsun,parsha,sabv,sabg,fsa,fsr,ndvi
    58) !

    83) ! Author: Gordon Bonan

    86) ! had all radiation being absorbed in the sunlit canopy, and now the sunlit and shaded
    87) ! canopies are each given the appropriate fluxes.  There was also an inconsistency in
    88) ! the original code, where parsun was not being scaled by leaf area, and so represented
    89) ! the entire canopy flux.  This goes into Stomata (in CanopyFluxes) where it is assumed
    90) ! to be a flux per unit leaf area. In addition, the fpsn flux coming out of Stomata was

    92) ! for the entire canopy to begin with.  Corrected this inconsistency in this version, so that
    93) ! the parsun and parsha fluxes going into canopy fluxes are per unit lai in the sunlit and
    94) ! shaded canopies.
    95) ! 6/9/03, Peter Thornton: Moved coszen from gps to c%cps to avoid problem

    98) ! 6/10/03, Peter Thornton: Added constraint on negative tot_aid, instead of

   353)      sfc_frc_bc_sno   => pef%sfc_frc_bc_sno
   354)      albgrd_bc        => cps%albgrd_bc
   355)      albgri_bc        => cps%albgri_bc
   356)      sfc_frc_oc       => pef%sfc_frc_oc
   357)      sfc_frc_oc_sno   => pef%sfc_frc_oc_sno
   358)      albgrd_oc        => cps%albgrd_oc
   359)      albgri_oc        => cps%albgri_oc
   360)      sfc_frc_dst      => pef%sfc_frc_dst
   361)      sfc_frc_dst_sno  => pef%sfc_frc_dst_sno
   362)      albgrd_dst       => cps%albgrd_dst
   363)      albgri_dst       => cps%albgri_dst
   364)      albsnd_hst       => cps%albsnd_hst
   365)      albsni_hst       => cps%albsni_hst
   366)      fsr_sno_vd       => pef%fsr_sno_vd
   367)      fsr_sno_nd       => pef%fsr_sno_nd
   368)      fsr_sno_vi       => pef%fsr_sno_vi
   369)      fsr_sno_ni       => pef%fsr_sno_ni
   370)      fsds_sno_vd      => pef%fsds_sno_vd
   371)      fsds_sno_nd      => pef%fsds_sno_nd
   372)      fsds_sno_vi      => pef%fsds_sno_vi
   373)      fsds_sno_ni      => pef%fsds_sno_ni
   374) 
   375)      ! Determine seconds off current time step
   376)      
   377)      dtime = get_step_size()
   378)      call get_curr_date (year, month, day, secs)
   379) 
   380)      ! Determine fluxes
   381) 
   382)      do fp = 1,num_nourbanp
   383)         p = filter_nourbanp(fp)
   384)         ! was redundant b/c filter already included wt>0; 
   385)         ! not redundant anymore with chg in filter definition
   386)         l = plandunit(p)
   387)         !Note: Some glacier_mec pfts may have zero weight
   388)         if (pwtgcell(p)>0._r8 .or. ityplun(l)==istice_mec) then
   389)            sabg(p)       = 0._r8
   390)            sabv(p)       = 0._r8
   391)            fsa(p)        = 0._r8
   392)            l = plandunit(p)
   393)            if (ityplun(l)==istsoil .or. ityplun(l)==istcrop) then
   394)              fsa_r(p)      = 0._r8
   395)            end if
   396)            sabg_lyr(p,:) = 0._r8
   397)            sabg_pur(p)   = 0._r8
   398)            sabg_bc(p)    = 0._r8
   399)            sabg_oc(p)    = 0._r8
   400)            sabg_dst(p)   = 0._r8
   401)         end if
   402)      end do 
   403) 
   404)      ! Loop over pfts to calculate fsun, etc
   405)      do fp = 1,num_nourbanp
   406)         p = filter_nourbanp(fp)
   407)         l = plandunit(p)
   408)         if (pwtgcell(p)>0._r8 .or. ityplun(l)==istice_mec) then
   409)            c = pcolumn(p)
   410)            g = pgridcell(p)
   411)         
   412)            vai(p) = elai(p) + esai(p)
   413)            if (coszen(c) > 0._r8 .and. elai(p) > 0._r8 .and. gdir(p) > 0._r8) then
   414)               cosz = max(0.001_r8, coszen(c))
   415)               ext = gdir(p)/cosz
   416)               t1 = min(ext*elai(p), 40.0_r8)
   417)               t2 = exp(-t1)
   418)               fsun(p) = (1._r8-t2)/t1
   419)               
   420)               ! new control on low lai, to avoid numerical problems in
   421)               ! calculation of slasun, slasha
   422)               ! PET: 2/29/04
   423)               
   424)               if (elai(p) > 0.01_r8) then
   425)                  laisun(p) = elai(p)*fsun(p)
   426)                  laisha(p) = elai(p)*(1._r8-fsun(p))
   427)                  
   428)                  ! calculate the average specific leaf area for sunlit and shaded
   429)                  ! canopies, when effective LAI > 0
   430)                  slasun(p) = (t2*dsladlai(ivt(p))*ext*elai(p) + &
   431)                               t2*dsladlai(ivt(p)) + &
   432)                               t2*slatop(ivt(p))*ext - &
   433)                               dsladlai(ivt(p)) - &
   434)                               slatop(ivt(p))*ext) / &
   435)                               (ext*(t2-1._r8))
   436)                  slasha(p) = ((slatop(ivt(p)) + &
   437)                              (dsladlai(ivt(p)) * elai(p)/2.0_r8)) * elai(p) - &
   438)                              laisun(p)*slasun(p)) / laisha(p)
   439)               else
   440)                  ! special case for low elai
   441)                  fsun(p) = 1._r8
   442)                  laisun(p) = elai(p)
   443)                  laisha(p) = 0._r8
   444)                  slasun(p) = slatop(ivt(p))
   445)                  slasha(p) = 0._r8
   446)               end if
   447)            else
   448)               fsun(p)   = 0._r8
   449)               laisun(p) = 0._r8
   450)               laisha(p) = elai(p)
   451)               slasun(p) = 0._r8
   452)               slasha(p) = 0._r8
   453)            end if
   454)         end if
   455)      end do
   456)         
   457)      ! Loop over nband wavebands
   458)      do ib = 1, nband
   459)         do fp = 1,num_nourbanp
   460)            p = filter_nourbanp(fp)
   461)            l = plandunit(p)
   462)            if (pwtgcell(p)>0._r8 .or. ityplun(l)==istice_mec) then
   463)               c = pcolumn(p)
   464)               g = pgridcell(p)
   465)               
   466)               ! Absorbed by canopy
   467)               
   468)               cad(p,ib) = forc_solad(g,ib)*fabd(p,ib)
   469)               cai(p,ib) = forc_solai(g,ib)*fabi(p,ib)
   470)               sabv(p) = sabv(p) + cad(p,ib) + cai(p,ib)
   471)               fsa(p)  = fsa(p)  + cad(p,ib) + cai(p,ib)
   472)               l = plandunit(p)
   473)               if (ityplun(l)==istsoil .or. ityplun(l)==istcrop) then
   474)                 fsa_r(p)  = fsa_r(p)  + cad(p,ib) + cai(p,ib)
   475)               end if
   476)               
   477)               ! Transmitted = solar fluxes incident on ground
   478)               
   479)               trd(p,ib) = forc_solad(g,ib)*ftdd(p,ib)
   480)               tri(p,ib) = forc_solad(g,ib)*ftid(p,ib) + forc_solai(g,ib)*ftii(p,ib)
   481)      
   482)               ! Solar radiation absorbed by ground surface
   483)               
   484)               absrad  = trd(p,ib)*(1._r8-albgrd(c,ib)) + tri(p,ib)*(1._r8-albgri(c,ib))
   485)               sabg(p) = sabg(p) + absrad
   486)               fsa(p)  = fsa(p)  + absrad
   487)               if (ityplun(l)==istsoil .or. ityplun(l)==istcrop) then
   488)                 fsa_r(p)  = fsa_r(p)  + absrad
   489)               end if
   490) 
   491)               if (use_snicar_frc) then
   492)                  ! Solar radiation absorbed by ground surface without BC
   493)                  absrad_bc = trd(p,ib)*(1._r8-albgrd_bc(c,ib)) + tri(p,ib)*(1._r8-albgri_bc(c,ib))
   494)                  sabg_bc(p) = sabg_bc(p) + absrad_bc
   495)                  
   496)                  ! Solar radiation absorbed by ground surface without OC
   497)                  absrad_oc = trd(p,ib)*(1._r8-albgrd_oc(c,ib)) + tri(p,ib)*(1._r8-albgri_oc(c,ib))
   498)                  sabg_oc(p) = sabg_oc(p) + absrad_oc
   499)                  
   500)                  ! Solar radiation absorbed by ground surface without dust
   501)                  absrad_dst = trd(p,ib)*(1._r8-albgrd_dst(c,ib)) + tri(p,ib)*(1._r8-albgri_dst(c,ib))
   502)                  sabg_dst(p) = sabg_dst(p) + absrad_dst
   503)                  
   504)                  ! Solar radiation absorbed by ground surface without any aerosols
   505)                  absrad_pur = trd(p,ib)*(1._r8-albgrd_pur(c,ib)) + tri(p,ib)*(1._r8-albgri_pur(c,ib))
   506)                  sabg_pur(p) = sabg_pur(p) + absrad_pur
   507)               end if
   508) 
   509) 
   510)               ! New sunlit.shaded canopy algorithm
   511)               
   512)               if (coszen(c) > 0._r8 .and. elai(p) > 0._r8 .and. gdir(p) > 0._r8 ) then
   513)                  
   514)                  ! 1. calculate flux of direct beam radiation absorbed in the 
   515)                  ! sunlit canopy as direct (sun_add), and the flux of direct
   516)                  ! beam radiation absorbed in the total canopy as indirect
   517)                  
   518)                  sun_add(p,ib) = forc_solad(g,ib) * (1._r8-ftdd(p,ib)) * (1._r8-omega(p,ib))
   519)                  tot_aid(p,ib) = (forc_solad(g,ib) * fabd(p,ib)) - sun_add(p,ib)
   520)                  
   521)                  ! the following constraint set to catch round-off level errors
   522)                  ! that can cause negative tot_aid
   523)                  
   524)                  tot_aid(p,ib) = max(tot_aid(p,ib), 0._r8)
   525)                  
   526)                  ! 2. calculate the effective extinction coefficients for indirect
   527)                  ! transmission originating from direct and indirect streams,
   528)                  ! using ftid and ftii
   529)                  
   530)                  !eff_kid(p,ib) = -(log(ftid(p,ib)))/vai(p)
   531)                  !eff_kii(p,ib) = -(log(ftii(p,ib)))/vai(p)
   532)                  
   533)                  ! 3. calculate the fraction of indirect radiation being absorbed 
   534)                  ! in the sunlit and shaded canopy fraction. Some of this indirect originates in
   535)                  ! the direct beam and some originates in the indirect beam.
   536) 
   537)                  !sun_faid(p,ib) = 1.-exp(-eff_kid(p,ib) * vaisun(p))
   538)                  !sun_faii(p,ib) = 1.-exp(-eff_kii(p,ib) * vaisun(p))
   539)                  sun_faid(p,ib) = fsun(p)
   540)                  sun_faii(p,ib) = fsun(p)
   541)                  sha_faid(p,ib) = 1._r8-sun_faid(p,ib)
   542)                  sha_faii(p,ib) = 1._r8-sun_faii(p,ib)
   543) 
   544)                  ! 4. calculate the total indirect flux absorbed by the sunlit
   545)                  ! and shaded canopy based on these fractions and the fabd and
   546)                  ! fabi from surface albedo calculations
   547) 
   548)                  sun_aid(p,ib) = tot_aid(p,ib) * sun_faid(p,ib)
   549)                  sun_aii(p,ib) = forc_solai(g,ib)*fabi(p,ib)*sun_faii(p,ib)
   550)                  sha_aid(p,ib) = tot_aid(p,ib) * sha_faid(p,ib)
   551)                  sha_aii(p,ib) = forc_solai(g,ib)*fabi(p,ib)*sha_faii(p,ib)
   552)                  
   553)                  ! 5. calculate the total flux absorbed in the sunlit and shaded
   554)                  ! canopy as the sum of these terms
   555)                  
   556)                  sun_atot(p,ib) = sun_add(p,ib) + sun_aid(p,ib) + sun_aii(p,ib)
   557)                  sha_atot(p,ib) = sha_aid(p,ib) + sha_aii(p,ib)
   558)                  
   559)                  ! 6. calculate the total flux absorbed by leaves in the sunlit
   560)                  ! and shaded canopies
   561)                  
   562)                  laifra = elai(p)/vai(p)
   563)                  sun_alf(p,ib) = sun_atot(p,ib) * laifra
   564)                  sha_alf(p,ib) = sha_atot(p,ib) * laifra
   565)                  
   566)                  ! 7. calculate the fluxes per unit lai in the sunlit and shaded
   567)                  ! canopies
   568)                  
   569)                  if (laisun(p) > 0._r8) then
   570)                     sun_aperlai(p,ib) = sun_alf(p,ib)/laisun(p)
   571)                  else
   572)                     sun_aperlai(p,ib) = 0._r8
   573)                  endif
   574)                  if (laisha(p) > 0._r8) then
   575)                     sha_aperlai(p,ib) = sha_alf(p,ib)/laisha(p)
   576)                  else
   577)                     sha_aperlai(p,ib) = 0._r8
   578)                  endif
   579)              
   580)               else   ! coszen = 0 or elai = 0
   581)                  
   582)                  sun_add(p,ib)     = 0._r8
   583)                  tot_aid(p,ib)     = 0._r8
   584)                  eff_kid(p,ib)     = 0._r8
   585)                  eff_kii(p,ib)     = 0._r8
   586)                  sun_faid(p,ib)    = 0._r8
   587)                  sun_faii(p,ib)    = 0._r8
   588)                  sha_faid(p,ib)    = 0._r8
   589)                  sha_faii(p,ib)    = 0._r8
   590)                  sun_aid(p,ib)     = 0._r8
   591)                  sun_aii(p,ib)     = 0._r8
   592)                  sha_aid(p,ib)     = 0._r8
   593)                  sha_aii(p,ib)     = 0._r8
   594)                  sun_atot(p,ib)    = 0._r8
   595)                  sha_atot(p,ib)    = 0._r8
   596)                  sun_alf(p,ib)     = 0._r8
   597)                  sha_alf(p,ib)     = 0._r8
   598)                  sun_aperlai(p,ib) = 0._r8
   599)                  sha_aperlai(p,ib) = 0._r8
   600)                  
   601)               end if
   602)            end if
   603)         end do ! end of pft loop
   604)      end do ! end nbands loop   
   605) 
   606)      
   607)      !   compute absorbed flux in each snow layer and top soil layer,
   608)      !   based on flux factors computed in the radiative transfer portion of SNICAR.
   609)      do fp = 1,num_nourbanp
   610)         p = filter_nourbanp(fp)
   611)            l = plandunit(p)
   612)            if (pwtgcell(p)>0._r8 .or. ityplun(l)==istice_mec) then
   613)            c = pcolumn(p)
   614)            sabg_snl_sum = 0._r8
   615) 
   616)            ! CASE1: No snow layers: all energy is absorbed in top soil layer
   617)            if (snl(c) == 0) then
   618)               sabg_lyr(p,:) = 0._r8
   619)               sabg_lyr(p,1) = sabg(p)
   620)               sabg_snl_sum  = sabg_lyr(p,1)
   621)    
   622)            ! CASE 2: Snow layers present: absorbed radiation is scaled according to 
   623)            ! flux factors computed by SNICAR
   624)            else
   625)               do i = -nlevsno+1,1,1
   626)                  sabg_lyr(p,i) = flx_absdv(c,i)*trd(p,1) + flx_absdn(c,i)*trd(p,2) + &
   627)                                  flx_absiv(c,i)*tri(p,1) + flx_absin(c,i)*tri(p,2)
   628)                  ! summed radiation in active snow layers:
   629)                  if (i >= snl(c)+1) then
   630)                     sabg_snl_sum = sabg_snl_sum + sabg_lyr(p,i)
   631)                  endif
   632)               enddo
   633)    
   634)               ! Error handling: The situation below can occur when solar radiation is 
   635)               ! NOT computed every timestep.
   636)               ! When the number of snow layers has changed in between computations of the 
   637)               ! absorbed solar energy in each layer, we must redistribute the absorbed energy
   638)               ! to avoid physically unrealistic conditions. The assumptions made below are 
   639)               ! somewhat arbitrary, but this situation does not arise very frequently. 
   640)               ! This error handling is implemented to accomodate any value of the
   641)               ! radiation frequency.
   642)               if (abs(sabg_snl_sum-sabg(p)) > 0.00001_r8) then
   643)                  if (snl(c) == 0) then
   644)                     sabg_lyr(p,-4:0) = 0._r8
   645)                     sabg_lyr(p,1) = sabg(p)
   646)                  elseif (snl(c) == -1) then
   647)                     sabg_lyr(p,-4:-1) = 0._r8
   648)                     sabg_lyr(p,0) = sabg(p)*0.6_r8
   649)                     sabg_lyr(p,1) = sabg(p)*0.4_r8
   650)                  else
   651)                     sabg_lyr(p,:) = 0._r8
   652)                     sabg_lyr(p,snl(c)+1) = sabg(p)*0.75_r8
   653)                     sabg_lyr(p,snl(c)+2) = sabg(p)*0.25_r8
   654)                  endif
   655)               endif
   656) 
   657)               ! If shallow snow depth, all solar radiation absorbed in top or top two snow layers
   658)               ! to prevent unrealistic timestep soil warming 
   659)               if (snowdp(c) < 0.10_r8) then
   660)                  if (snl(c) == 0) then
   661)                     sabg_lyr(p,-4:0) = 0._r8
   662)                     sabg_lyr(p,1) = sabg(p)
   663)                  elseif (snl(c) == -1) then
   664)                     sabg_lyr(p,-4:-1) = 0._r8
   665)                     sabg_lyr(p,0) = sabg(p)
   666)                     sabg_lyr(p,1) = 0._r8
   667)                  else
   668)                     sabg_lyr(p,:) = 0._r8
   669)                     sabg_lyr(p,snl(c)+1) = sabg(p)*0.75_r8
   670)                     sabg_lyr(p,snl(c)+2) = sabg(p)*0.25_r8
   671)                  endif
   672)               endif
   673) 
   674)            endif
   675) 
   676)            ! This situation should not happen:
   677)            if (abs(sum(sabg_lyr(p,:))-sabg(p)) > 0.00001_r8) then
   678)               write(iulog,*) "SNICAR ERROR: Absorbed ground radiation not equal to summed snow layer radiation. pft = ",   &
   679)                              p," Col= ", c, " Diff= ",sum(sabg_lyr(p,:))-sabg(p), " sabg(p)= ", sabg(p), " sabg_sum(p)= ", &
   680)                              sum(sabg_lyr(p,:)), " snl(c)= ", snl(c)
   681)               write(iulog,*) "flx_absdv1= ", trd(p,1)*(1.-albgrd(c,1)), "flx_absdv2= ", sum(flx_absdv(c,:))*trd(p,1)
   682)               write(iulog,*) "flx_absiv1= ", tri(p,1)*(1.-albgri(c,1))," flx_absiv2= ", sum(flx_absiv(c,:))*tri(p,1)
   683)               write(iulog,*) "flx_absdn1= ", trd(p,2)*(1.-albgrd(c,2))," flx_absdn2= ", sum(flx_absdn(c,:))*trd(p,2)
   684)               write(iulog,*) "flx_absin1= ", tri(p,2)*(1.-albgri(c,2))," flx_absin2= ", sum(flx_absin(c,:))*tri(p,2)
   685)    
   686)               write(iulog,*) "albgrd_nir= ", albgrd(c,2)
   687)               write(iulog,*) "coszen= ", coszen(c)
   688)               call endrun()
   689)            endif
   690) 
   691)  
   692)            if (use_snicar_frc) then
   693) 
   694)               ! BC aerosol forcing (pft-level):
   695)               sfc_frc_bc(p) = sabg(p) - sabg_bc(p)
   696)               
   697)               ! OC aerosol forcing (pft-level):
   698)               if (DO_SNO_OC) then
   699)                  sfc_frc_oc(p) = sabg(p) - sabg_oc(p)
   700)               else
   701)                  sfc_frc_oc(p) = 0._r8
   702)               endif
   703)               
   704)               ! dust aerosol forcing (pft-level):
   705)               sfc_frc_dst(p) = sabg(p) - sabg_dst(p)
   706)               
   707)               ! all-aerosol forcing (pft-level):
   708)               sfc_frc_aer(p) = sabg(p) - sabg_pur(p)        
   709)               
   710)               ! forcings averaged only over snow:
   711)               if (frac_sno(c) > 0._r8) then
   712)                  sfc_frc_bc_sno(p)  = sfc_frc_bc(p)/frac_sno(c)
   713)                  sfc_frc_oc_sno(p)  = sfc_frc_oc(p)/frac_sno(c)
   714)                  sfc_frc_dst_sno(p) = sfc_frc_dst(p)/frac_sno(c)
   715)                  sfc_frc_aer_sno(p) = sfc_frc_aer(p)/frac_sno(c)
   716)               else
   717)                  sfc_frc_bc_sno(p)  = spval
   718)                  sfc_frc_oc_sno(p)  = spval
   719)                  sfc_frc_dst_sno(p) = spval
   720)                  sfc_frc_aer_sno(p) = spval
   721)               endif
   722) 
   723)            end if
   724)         endif
   725)      enddo
   726) 
   727) 
   728)      do fp = 1,num_nourbanp
   729)         p = filter_nourbanp(fp)
   730)         l = plandunit(p)
   731)         if (pwtgcell(p)>0._r8 .or. ityplun(l)==istice_mec) then
   732)            g = pgridcell(p)
   733)         
   734)            ! Final step of new sunlit/shaded canopy algorithm
   735)            ! 8. calculate the total and per-unit-lai fluxes for PAR in the
   736)            ! sunlit and shaded canopy leaf fractions
   737)            
   738)            parsun(p) = sun_aperlai(p,1)
   739)            parsha(p) = sha_aperlai(p,1)
   740)            
   741)            ! The following code is duplicated from SurfaceRadiation
   742)            ! NDVI and reflected solar radiation
   743)            
   744)            rvis = albd(p,1)*forc_solad(g,1) + albi(p,1)*forc_solai(g,1)
   745)            rnir = albd(p,2)*forc_solad(g,2) + albi(p,2)*forc_solai(g,2)
   746)            fsr(p) = rvis + rnir
   747)            
   748)            fsds_vis_d(p) = forc_solad(g,1)
   749)            fsds_nir_d(p) = forc_solad(g,2)
   750)            fsds_vis_i(p) = forc_solai(g,1)
   751)            fsds_nir_i(p) = forc_solai(g,2)
   752)            fsr_vis_d(p)  = albd(p,1)*forc_solad(g,1)
   753)            fsr_nir_d(p)  = albd(p,2)*forc_solad(g,2)
   754)            fsr_vis_i(p)  = albi(p,1)*forc_solai(g,1)
   755)            fsr_nir_i(p)  = albi(p,2)*forc_solai(g,2)
   756)            
   757)            local_secp1 = secs + nint((londeg(g)/degpsec)/dtime)*dtime
   758)            local_secp1 = mod(local_secp1,isecspday)
   759)            if (local_secp1 == isecspday/2) then
