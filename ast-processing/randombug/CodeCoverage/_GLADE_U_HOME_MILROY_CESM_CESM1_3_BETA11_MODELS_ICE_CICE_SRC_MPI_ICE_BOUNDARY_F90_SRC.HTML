   157) ! !IROUTINE: ice_HaloCreate
   158) ! !INTERFACE:

   160)  function ice_HaloCreate(dist, nsBoundaryType, ewBoundaryType, &
   161)                          nxGlobal)  result(halo)
   162) 
   163) ! !DESCRIPTION:
   164) !  This routine creates a halo type with info necessary for
   165) !  performing a halo (ghost cell) update. This info is computed
   166) !  based on the input block distribution.
   167) !
   168) ! !REVISION HISTORY:
   169) !  same as module
   170) 
   171) ! !INPUT PARAMETERS:
   172) 
   173)    type (distrb), intent(in) :: &
   174)       dist             ! distribution of blocks across procs
   175) 
   176)    character (*), intent(in) :: &
   177)       nsBoundaryType,   &! type of boundary to use in logical ns dir
   178)       ewBoundaryType     ! type of boundary to use in logical ew dir
   179) 
   180)    integer (int_kind), intent(in) :: &
   181)       nxGlobal           ! global grid extent for tripole grids
   182) 
   183) ! !OUTPUT PARAMETERS:
   184) 
   185)    type (ice_halo) :: &
   186)       halo               ! a new halo type with info for halo updates
   187) 
   188) !EOP
   189) !BOC
   190) !-----------------------------------------------------------------------
   191) !
   192) !  local variables
   193) !
   194) !-----------------------------------------------------------------------
   195) 
   196)    integer (int_kind) ::             &
   197)       i,j,k,l,n,m,&
   198)       istat,                       &! allocate status flag
   199)       numProcs,                    &! num of processors involved
   200)       communicator,                &! communicator for message passing
   201)       iblock,                      &! block counter
   202)       eastBlock, westBlock,        &! block id  east,  west neighbors
   203)       northBlock, southBlock,      &! block id north, south neighbors
   204)       neBlock, nwBlock,            &! block id northeast, northwest nbrs
   205)       seBlock, swBlock,            &! block id southeast, southwest nbrs
   206)       srcProc, dstProc,            &! source, dest processor locations
   207)       srcLocalID, dstLocalID,      &! local block index of src,dst blocks
   208)       maxTmp,                      &! temp for global maxval      
   209)       blockSizeX,                  &! size of default physical domain in X 
   210)       blockSizeY,                  &! size of default physical domain in Y 
   211)       maxSizeSend, maxSizeRecv,    &! max buffer sizes
   212)       numMsgSend, numMsgRecv,      &! number of messages for this halo
   213)       eastMsgSize, westMsgSize,    &! nominal sizes for e-w msgs
   214)       northMsgSize, southMsgSize,  &! nominal sizes for n-s msgs
   215)       tripoleMsgSize,              &! size for tripole messages
   216)       tripoleMsgSizeOut,           &! size for tripole messages
   217)       tripoleRows,                 &! number of rows in tripole buffer
   218)       cornerMsgSize, msgSize        ! nominal size for corner msg
   219) 
   220)    integer (int_kind), dimension(:), allocatable :: &
   221)       sendCount, recvCount          ! count number of words to each proc
   222) 
   223)    logical (log_kind) :: &
   224)       resize,               &! flag for resizing buffers
   225)       tripoleFlag,          &! flag for allocating tripole buffers
   226)       tripoleBlock,         &! flag for identifying north tripole blocks
   227)       tripoleTFlag           ! flag for processing tripole buffer as T-fold
   228) 
   229) !-----------------------------------------------------------------------
   230) !
   231) !  Initialize some useful variables and return if this task not
   232) !  in the current distribution.
   233) !
   234) !-----------------------------------------------------------------------
   235) 
   236)    call ice_distributionGet(dist,          &
   237)                             nprocs = numProcs,       &
   238)                             communicator = communicator)
   239) 
   240)    if (my_task >= numProcs) return
   241) 
   242)    halo%communicator = communicator
   243) 
   244)    blockSizeX = nx_block - 2*nghost
   245)    blockSizeY = ny_block - 2*nghost
   246)    eastMsgSize  = nghost*blockSizeY
   247)    westMsgSize  = nghost*blockSizeY
   248)    southMsgSize = nghost*blockSizeX
   249)    northMsgSize = nghost*blockSizeX
   250)    cornerMsgSize = nghost*nghost
   251)    tripoleRows = nghost+1
   252) 
   253)    if (nsBoundaryType == 'tripole' .or. nsBoundaryType == 'tripoleT') then
   254)       tripoleFlag = .true.
   255)       tripoleTFlag = (nsBoundaryType == 'tripoleT')
   256)       if (tripoleTflag) tripoleRows = tripoleRows+1
   257) 
   258)       !*** allocate tripole message buffers if not already done
   259) 
   260)       if (.not. allocated(bufTripoleR8)) then
   261)          allocate (bufTripoleI4(nxGlobal, tripoleRows), &
   262)                    bufTripoleR4(nxGlobal, tripoleRows), &
   263)                    bufTripoleR8(nxGlobal, tripoleRows), &
   264)                    stat=istat)
   265) 
   266)          if (istat > 0) then
   267)             call abort_ice( &
   268)                'ice_HaloCreate: error allocating tripole buffers')
   269)             return
   270)          endif
   271)       endif
   272) 
   273)    else
   274)       tripoleFlag = .false.
   275)       tripoleTFlag = .false.
   276)    endif
   277)    halo%tripoleTFlag = tripoleTFlag
   278)    halo%tripoleRows = tripoleRows
   279)    tripoleMsgSize = tripoleRows*blockSizeX
   280)    tripoleMsgSizeOut = tripoleRows*nx_block
   281) 
   282) !-----------------------------------------------------------------------
   283) !
   284) !  Count the number of messages to send/recv from each processor
   285) !  and number of words in each message.  These quantities are
   286) !  necessary for allocating future arrays.
   287) !
   288) !-----------------------------------------------------------------------
   289) 
   290)    allocate (sendCount(numProcs), recvCount(numProcs), stat=istat)
   291) 
   292)    if (istat > 0) then
   293)       call abort_ice( &
   294)                         'ice_HaloCreate: error allocating count arrays')
   295)       return
   296)    endif
   297) 
   298)    sendCount  = 0
   299)    recvCount  = 0
   300) 
   301)    msgCountLoop: do iblock=1,nblocks_tot
   302) 
   303)       call ice_distributionGetBlockLoc(dist, iblock, srcProc, &
   304)                                        srcLocalID)
   305) 
   306)       !*** find north neighbor block and add to message count
   307)       !***  also set tripole block flag for later special cases
   308) 
   309)       northBlock = ice_blocksGetNbrID(iblock, ice_blocksNorth,        &
   310)                                       ewBoundaryType, nsBoundaryType)
   311)       if (northBlock > 0) then
   312)          tripoleBlock = .false.
   313)          msgSize = northMsgSize
   314)          call ice_distributionGetBlockLoc(dist, northBlock, dstProc, &
   315)                                           dstLocalID)
   316)       else if (northBlock < 0) then ! tripole north row, count block
   317)          tripoleBlock = .true.
   318)          msgSize = tripoleMsgSize
   319)          call ice_distributionGetBlockLoc(dist, abs(northBlock), &
   320)                                  dstProc, dstLocalID)
   321)       else
   322)          tripoleBlock = .false.
   323)          msgSize = northMsgSize
   324)          dstProc = 0
   325)          dstLocalID = 0
   326)       endif
   327) 
   328)       call ice_HaloIncrementMsgCount(sendCount, recvCount,           &
   329)                                      srcProc, dstProc, msgSize)
   330) 
   331)       !*** if a tripole boundary block, also create a local
   332)       !*** message into and out of tripole buffer 
   333) 
   334)       if (tripoleBlock) then
   335)          !*** copy out of tripole buffer - includes halo
   336)          call ice_HaloIncrementMsgCount(sendCount, recvCount,        &
   337)                                         srcProc, srcProc,            &
   338)                                         tripoleMsgSizeOut)
   339) 
   340)          !*** copy in only required if dstProc not same as srcProc
   341)          if (dstProc /= srcProc) then
   342)             call ice_HaloIncrementMsgCount(sendCount, recvCount,  &
   343)                                            srcProc, srcProc,      & 
   344)                                            msgSize)
   345)          endif
   346)       endif
   347) 
   348)       !*** find south neighbor block and add to message count
   349) 
   350)       southBlock = ice_blocksGetNbrID(iblock, ice_blocksSouth,        &
   351)                                       ewBoundaryType, nsBoundaryType)
   352) 
   353)       if (southBlock > 0) then
   354)          call ice_distributionGetBlockLoc(dist, southBlock, dstProc, &
   355)                                           dstLocalID)
   356)       else
   357)          dstProc = 0
   358)          dstLocalID = 0
   359)       endif
   360) 
   361)       call ice_HaloIncrementMsgCount(sendCount, recvCount,           &
   362)                                      srcProc, dstProc, southMsgSize)
   363) 
   364)       !*** find east neighbor block and add to message count
   365) 
   366)       eastBlock = ice_blocksGetNbrID(iblock, ice_blocksEast,         &
   367)                                      ewBoundaryType, nsBoundaryType)
   368) 
   369)       if (eastBlock > 0) then
   370)          call ice_distributionGetBlockLoc(dist, eastBlock, dstProc, &
   371)                                           dstLocalID)
   372)       else
   373)          dstProc = 0
   374)          dstLocalID = 0
   375)       endif
   376) 
   377)       call ice_HaloIncrementMsgCount(sendCount, recvCount,          &
   378)                                      srcProc, dstProc, eastMsgSize)
   379) 
   380)       !*** if a tripole boundary block, non-local east neighbor
   381)       !*** needs a chunk of the north boundary, so add a message
   382)       !*** for that
   383) 
   384)       if (tripoleBlock .and. dstProc /= srcProc) then
   385)          call ice_HaloIncrementMsgCount(sendCount, recvCount,          &
   386)                                      srcProc, dstProc, tripoleMsgSize)
   387)       endif
   388) 
   389)       !*** find west neighbor block and add to message count
   390) 
   391)       westBlock = ice_blocksGetNbrID(iblock, ice_blocksWest,         &
   392)                                      ewBoundaryType, nsBoundaryType)
   393) 
   394)       if (westBlock > 0) then
   395)          call ice_distributionGetBlockLoc(dist, westBlock, dstProc, &
   396)                                           dstLocalID)
   397)       else
   398)          dstProc = 0
   399)          dstLocalID = 0
   400)       endif
   401) 
   402)       call ice_HaloIncrementMsgCount(sendCount, recvCount,          &
   403)                                      srcProc, dstProc, westMsgSize)
   404) 
   405)       !*** if a tripole boundary block, non-local west neighbor
   406)       !*** needs a chunk of the north boundary, so add a message
   407)       !*** for that
   408) 
   409)       if (tripoleBlock .and. dstProc /= srcProc) then
   410)          call ice_HaloIncrementMsgCount(sendCount, recvCount,          &
   411)                                      srcProc, dstProc, tripoleMsgSize)
   412)       endif
   413) 
   414)       !*** find northeast neighbor block and add to message count
   415) 
   416)       neBlock = ice_blocksGetNbrID(iblock, ice_blocksNorthEast,    &
   417)                                    ewBoundaryType, nsBoundaryType)
   418) 
   419)       if (neBlock > 0) then
   420)          msgSize = cornerMsgSize  ! normal corner message 
   421) 
   422)          call ice_distributionGetBlockLoc(dist, neBlock, dstProc, &
   423)                                           dstLocalID)
   424) 
   425)       else if (neBlock < 0) then ! tripole north row
   426)          msgSize = tripoleMsgSize  ! tripole needs whole top row of block
   427) 
   428)          call ice_distributionGetBlockLoc(dist, abs(neBlock), dstProc, &
   429)                                           dstLocalID)
   430)       else
   431)          dstProc = 0
   432)          dstLocalID = 0
   433)       endif
   434) 
   435)       call ice_HaloIncrementMsgCount(sendCount, recvCount,      &
   436)                                      srcProc, dstProc, msgSize)
   437) 
   438)       !*** find northwest neighbor block and add to message count
   439) 
   440)       nwBlock = ice_blocksGetNbrID(iblock, ice_blocksNorthWest,    &
   441)                                    ewBoundaryType, nsBoundaryType)
   442) 
   443)       if (nwBlock > 0) then
   444)          msgSize = cornerMsgSize ! normal NE corner update
   445) 
   446)          call ice_distributionGetBlockLoc(dist, nwBlock, dstProc, &
   447)                                           dstLocalID)
   448) 
   449)       else if (nwBlock < 0) then ! tripole north row, count block
   450)          msgSize = tripoleMsgSize ! tripole NE corner update - entire row needed
   451) 
   452)          call ice_distributionGetBlockLoc(dist, abs(nwBlock), dstProc, &
   453)                                           dstLocalID)
   454) 
   455)       else
   456)          dstProc = 0
   457)          dstLocalID = 0
   458)       endif
   459) 
   460)       call ice_HaloIncrementMsgCount(sendCount, recvCount,      &
   461)                                      srcProc, dstProc, msgSize)
   462) 
   463)       !*** find southeast neighbor block and add to message count
   464) 
   465)       seBlock = ice_blocksGetNbrID(iblock, ice_blocksSouthEast,    &
   466)                                    ewBoundaryType, nsBoundaryType)
   467) 
   468)       if (seBlock > 0) then
   469)          call ice_distributionGetBlockLoc(dist, seBlock, dstProc, &
   470)                                           dstLocalID)
   471)       else
   472)          dstProc = 0
   473)          dstLocalID = 0
   474)       endif
   475) 
   476)       call ice_HaloIncrementMsgCount(sendCount, recvCount,            &
   477)                                      srcProc, dstProc, cornerMsgSize)
   478) 
   479)       !*** find southwest neighbor block and add to message count
   480) 
   481)       swBlock = ice_blocksGetNbrID(iblock, ice_blocksSouthWest,    &
   482)                                    ewBoundaryType, nsBoundaryType)
   483) 
   484)       if (swBlock > 0) then
   485)          call ice_distributionGetBlockLoc(dist, swBlock, dstProc, &
   486)                                           dstLocalID)
   487)       else
   488)          dstProc = 0
   489)          dstLocalID = 0
   490)       endif
   491) 
   492)       call ice_HaloIncrementMsgCount(sendCount, recvCount,            &
   493)                                      srcProc, dstProc, cornerMsgSize)
   494) 
   495)       !*** for tripole grids with padded domain, padding will
   496)       !*** prevent tripole buffer from getting all the info
   497)       !*** it needs - must extend footprint at top boundary
   498) 
   499)       if (tripoleBlock                  .and. & !tripole
   500)           mod(nxGlobal,blockSizeX) /= 0) then   !padding
   501) 
   502)          !*** find east2 neighbor block and add to message count
   503) 
   504)          eastBlock = ice_blocksGetNbrID(iBlock, ice_blocksEast2,     &
   505)                                      ewBoundaryType, nsBoundaryType)
   506) 
   507)          if (eastBlock > 0) then
   508)             call ice_distributionGetBlockLoc(dist, eastBlock, dstProc, &
   509)                                              dstLocalID)
   510)          else
   511)             dstProc = 0
   512)             dstLocalID = 0
   513)          endif
   514) 
   515)          if (dstProc /= srcProc) then
   516)             call ice_HaloIncrementMsgCount(sendCount, recvCount,       &
   517)                                      srcProc, dstProc, tripoleMsgSize)
   518)          endif
   519) 
   520)          !*** find EastNorthEast neighbor block and add to message count
   521) 
   522)          neBlock = ice_blocksGetNbrID(iBlock, ice_blocksEastNorthEast, &
   523)                                      ewBoundaryType, nsBoundaryType)
   524) 
   525)          if (neBlock < 0) then ! tripole north row
   526)             msgSize = tripoleMsgSize  ! tripole needs whole top row of block
   527) 
   528)             call ice_distributionGetBlockLoc(dist, abs(neBlock), dstProc, &
   529)                                              dstLocalID)
   530)          else
   531)             dstProc = 0
   532)             dstLocalID = 0
   533)          endif
   534) 
   535)          if (dstProc /= srcProc) then
   536)             call ice_HaloIncrementMsgCount(sendCount, recvCount,   &
   537)                                         srcProc, dstProc, msgSize)
   538)          endif
   539) 
   540)          !*** find west2 neighbor block and add to message count
   541) 
   542)          westBlock = ice_blocksGetNbrID(iBlock, ice_blocksWest2,     &
   543)                                      ewBoundaryType, nsBoundaryType)
   544) 
   545)          if (westBlock > 0) then
   546)             call ice_distributionGetBlockLoc(dist, westBlock, dstProc, &
   547)                                              dstLocalID)
   548)          else
   549)             dstProc = 0
   550)             dstLocalID = 0
   551)          endif
   552) 
   553)          if (dstProc /= srcProc) then
   554)             call ice_HaloIncrementMsgCount(sendCount, recvCount,       &
   555)                                      srcProc, dstProc, tripoleMsgSize)
   556)          endif
   557) 
   558)          !*** find WestNorthWest neighbor block and add to message count
   559) 
   560)          nwBlock = ice_blocksGetNbrID(iBlock, ice_blocksWestNorthWest, &
   561)                                      ewBoundaryType, nsBoundaryType)
   562) 
   563)          if (nwBlock < 0) then ! tripole north row
   564)             msgSize = tripoleMsgSize  ! tripole needs whole top row of block
   565) 
   566)             call ice_distributionGetBlockLoc(dist, abs(nwBlock), dstProc, &
   567)                                              dstLocalID)
   568)          else
   569)             dstProc = 0
   570)             dstLocalID = 0
   571)          endif
   572) 
   573)          if (dstProc /= srcProc) then
   574)             call ice_HaloIncrementMsgCount(sendCount, recvCount,   &
   575)                                         srcProc, dstProc, msgSize)
   576)          endif
   577) 
   578)       endif
   579) 
   580)    end do msgCountLoop
   581) 
   582) !-----------------------------------------------------------------------
   583) !
   584) !  if messages are received from the same processor, the message is 
   585) !  actually a local copy - count them and reset to zero
   586) !
   587) !-----------------------------------------------------------------------
   588) 
   589)    halo%numLocalCopies = recvCount(my_task+1)
   590) 
   591)    sendCount(my_task+1) = 0
   592)    recvCount(my_task+1) = 0
   593) 
   594) !-----------------------------------------------------------------------
   595) !
   596) !  now count the number of actual messages to be sent and received
   597) !
   598) !-----------------------------------------------------------------------
   599)

  5207)             !*** on T-fold, do not need to replace
  5208)             !*** top row of physical domain, so jSrc should be
  5209)             !*** out of range and skipped
  5210)             !*** otherwise do the copy

  5216)                                   bufTripole(iSrc,jSrc,k,l)

  5220) 
  5221)          endif
  5222)       end do

  5224)    endif

  5416) 
  5417)       i=0
  5418)       do n=1,halo%sizeSend(nmsg)

  5420)          jSrc     = halo%sendAddr(2,n,nmsg)

  5424)          do k=1,nz

  5707)             !*** out of range and skipped

  5718) 
  5719)          endif
  5720)       end do
  5721) 
  5722)    endif
  5723) 
  5724) !-----------------------------------------------------------------------

  5728) !-----------------------------------------------------------------------
  5729)

  5743)       call abort_ice( &
  5744)          'ice_HaloUpdate4DI4: error deallocating 4d buffers')
