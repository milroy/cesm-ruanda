   350) ! !IROUTINE: init_coupler_flux - initialize fluxes exchanged with coupler
   351) !
   352) ! !INTERFACE:
   353) !
   354)       subroutine init_coupler_flux
   355) !
   356) ! !DESCRIPTION:
   357) !
   358) ! Initialize all fluxes exchanged with flux coupler
   359) ! and some data-derived fields
   360) !
   361) ! !REVISION HISTORY:
   362) !
   363) ! author Elizabeth C. Hunke, LANL
   364) !
   365) ! !USES:
   366) !
   367) ! !INPUT/OUTPUT PARAMETERS:
   368) !
   369) !EOP
   370) !
   371)       integer (kind=int_kind) :: i, j, n, iblk
   372) 
   373)       logical (kind=log_kind), parameter ::     &
   374) !         l_winter = .true.   ! winter/summer default switch
   375)          l_winter = .false.   ! winter/summer default switch
   376) 
   377)       real (kind=dbl_kind) :: fcondtopn_d(6), fsurfn_d(6)
   378) 
   379)       data fcondtopn_d / -50.0_dbl_kind,-17.0_dbl_kind,-12.0_dbl_kind, &
   380)                           -9.0_dbl_kind, -7.0_dbl_kind, -3.0_dbl_kind /
   381)       data fsurfn_d    /  0.20_dbl_kind, 0.15_dbl_kind, 0.10_dbl_kind, &
   382)                           0.05_dbl_kind, 0.01_dbl_kind, 0.01_dbl_kind /
   383) 
   384)       !-----------------------------------------------------------------
   385)       ! fluxes received from atmosphere
   386)       !-----------------------------------------------------------------
   387)       zlvl  (:,:,:) = c10             ! atm level height (m)
   388)       rhoa  (:,:,:) = 1.3_dbl_kind    ! air density (kg/m^3)
   389)       uatm  (:,:,:) = c5              ! wind velocity    (m/s)
   390)       vatm  (:,:,:) = c5
   391)       strax (:,:,:) = 0.05_dbl_kind
   392)       stray (:,:,:) = 0.05_dbl_kind
   393)       if (l_winter) then
   394)          !typical winter values
   395)          potT  (:,:,:) = 253.0_dbl_kind  ! air potential temp (K)
   396)          Tair  (:,:,:) = 253.0_dbl_kind  ! air temperature  (K)
   397)          Qa    (:,:,:) = 0.0006_dbl_kind ! specific humidity (kg/kg)
   398)          swvdr (:,:,:) = c0              ! shortwave radiation (W/m^2)
   399)          swvdf (:,:,:) = c0              ! shortwave radiation (W/m^2)
   400)          swidr (:,:,:) = c0              ! shortwave radiation (W/m^2)
   401)          swidf (:,:,:) = c0              ! shortwave radiation (W/m^2)
   402)          flw   (:,:,:) = c180            ! incoming longwave rad (W/m^2)
   403)          frain (:,:,:) = c0              ! rainfall rate (kg/m2/s)
   404)          fsnow (:,:,:) = 4.0e-6_dbl_kind ! snowfall rate (kg/m2/s)
   405)          do n = 1, ncat                  ! conductive heat flux (W/m^2)
   406)             fcondtopn_f(:,:,n,:) = fcondtopn_d(n)
   407)          enddo
   408)          fsurfn_f = fcondtopn_f          ! surface heat flux (W/m^2)
   409)          flatn_f(:,:,:,:) = c0           ! latent heat flux (kg/m2/s)
   410)       else
   411)          !typical summer values
   412)          potT  (:,:,:) = 273.0_dbl_kind  ! air potential temp (K)
   413)          Tair  (:,:,:) = 273.0_dbl_kind  ! air temperature  (K)
   414)          Qa    (:,:,:) = 0.0035_dbl_kind ! specific humidity (kg/kg)
   415)          swvdr (:,:,:) = 50._dbl_kind    ! shortwave radiation (W/m^2)
   416)          swvdf (:,:,:) = 50._dbl_kind    ! shortwave radiation (W/m^2)
   417)          swidr (:,:,:) = 50._dbl_kind    ! shortwave radiation (W/m^2)
   418)          swidf (:,:,:) = 50._dbl_kind    ! shortwave radiation (W/m^2)
   419)          flw   (:,:,:) = 280.0_dbl_kind  ! incoming longwave rad (W/m^2)
   420)          frain (:,:,:) = c0              ! rainfall rate (kg/m2/s)
   421)          fsnow (:,:,:) = c0              ! snowfall rate (kg/m2/s)
   422)          do n = 1, ncat                  ! surface heat flux (W/m^2)
   423)             fsurfn_f(:,:,n,:) = fsurfn_d(n)
   424)          enddo
   425)          fcondtopn_f(:,:,:,:) = 0.0_dbl_kind ! conductive heat flux (W/m^2)
   426)          flatn_f(:,:,:,:) = -2.0_dbl_kind    ! latent heat flux (W/m^2)
   427)       endif !     l_winter
   428) 
   429)       faero (:,:,:,:) = c0            ! aerosol deposition rate (kg/m2/s) MH
   430) 
   431)       !-----------------------------------------------------------------
   432)       ! fluxes received from ocean
   433)       !-----------------------------------------------------------------
   434) 
   435)       ss_tltx(:,:,:)= c0              ! sea surface tilt (m/m)
   436)       ss_tlty(:,:,:)= c0
   437)       uocn  (:,:,:) = c0              ! surface ocean currents (m/s)
   438)       vocn  (:,:,:) = c0
   439)       frzmlt(:,:,:) = c0              ! freezing/melting potential (W/m^2)
   440)       sss   (:,:,:) = 34.0_dbl_kind   ! sea surface salinity (o/oo)
   441)       if (trim(Tfrzpt) == 'constant') then
   442)          Tf    (:,:,:) = -1.8_dbl_kind   ! freezing temp (C)
   443)       else ! default:  Tfrzpt = 'linear_S'
