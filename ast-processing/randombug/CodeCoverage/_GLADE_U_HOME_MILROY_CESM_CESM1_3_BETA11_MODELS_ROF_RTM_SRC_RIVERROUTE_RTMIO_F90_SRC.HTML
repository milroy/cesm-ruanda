   403)     !-----------------------------------------------------------------------

   433)     if (ier == PIO_NOERR) ier = pio_inq_dimlen(ncid, dimid, nj)

   449)        isgrid2d = .false.
   450)     else
   451)        isgrid2d = .true.

   720)     endif
   721)

   724)     else
   725)        lxtype = xtype
   726)     end if
   727)     if (masterproc .and. debug > 1) then
   728)        write(iulog,*) 'Error in defining variable = ', trim(varname)

   772)           if ( n > 1 ) str = trim(str)//" "//flag_meanings(n)
   773)        end do
   774)        status = PIO_put_att(ncid,varid,'flag_meanings', trim(str) )
   775)     end if
   776)     if (present(comment)) then
   777)        call ncd_putatt(ncid, varid, 'comment', trim(comment))
   778)     end if
   779)     if (present(units)) then
   780)        call ncd_putatt(ncid, varid, 'units', trim(units))
   781)     end if
   782)     if (present(cell_method)) then
   783)        str = 'time: ' // trim(cell_method)
   784)        call ncd_putatt(ncid, varid, 'cell_methods', trim(str))
   785)     end if
   786)     if (present(fill_value)) then
   787)        call ncd_putatt(ncid, varid, '_FillValue', fill_value, lxtype)
   788)     end if
   789)     if (present(missing_value)) then
   790)        call ncd_putatt(ncid, varid, 'missing_value', missing_value, lxtype)
   791)     end if
   792)     if (present(ifill_value)) then
   793)        call ncd_putatt(ncid, varid, '_FillValue', ifill_value, lxtype)
   794)     end if
   795)     if (present(imissing_value)) then

   798)     if (present(nvalid_range)) then

   800)     end if
   801)     if ( xtype == ncd_log )then

   803)        status = PIO_put_att(ncid,varid,'flag_meanings',  "FALSE TRUE" )
