    45) subroutine vd_lu_decomp(pcols, pver, ncol,                          &
    46)                         ksrf,  kv,   tmpi,   rpdel,  ztodt,  gravit,&
    47)                         cc_top, ntop,  nbot, decomp, cpairv)
    48) 
    49)   !---------------------------------------------------------------------- !
    50)   ! Determine superdiagonal (ca(k)) and subdiagonal (cc(k)) coeffs of the !
    51)   ! tridiagonal diffusion matrix.                                         !
    52)   ! The diagonal elements (1+ca(k)+cc(k)) are not required by the solver. !
    53)   ! Also determine ze factor and denominator for ze and zf (see solver).  !
    54)   !---------------------------------------------------------------------- !
    55) 
    56)   ! ---------------------- !
    57)   ! Input-Output Arguments !
    58)   ! ---------------------- !
    59) 
    60)   ! Allocated column and level dimensions.
    61)   integer,  intent(in)  :: pcols, pver
    62)   ! Columns actually used during computation.
    63)   integer,  intent(in)  :: ncol
    64)   ! Top and bottom levels to operate on.
    65)   integer,  intent(in)  :: ntop, nbot
    66) 
    67)   ! Surface "drag" coefficient. [ kg/s/m2 ]
    68)   real(r8), intent(in)  :: ksrf(pcols)
    69)   ! Vertical diffusion coefficients. [ m2/s ]
    70)   real(r8), intent(in)  :: kv(pcols,pver+1)
    71)   ! dt*(g/R)**2/dp*pi(k+1)/(.5*(tm(k+1)+tm(k))**2
    72)   real(r8), intent(in)  :: tmpi(pcols,pver+1)
    73)   ! 1./pdel (pdel is thickness between interfaces). [ Pa^-1 ]
    74)   real(r8), intent(in)  :: rpdel(pcols,pver)
    75)   ! 2 delta-t [ s ]
    76)   real(r8), intent(in)  :: ztodt
    77)   ! Acceleration due to gravity. [ m/s2 ]
    78)   real(r8), intent(in)  :: gravit
    79)   ! Lower diagonal on top interface (for fixed ubc only).
    80)   real(r8), intent(in)  :: cc_top(pcols)
    81) 
    82)   ! Output decomposition.
    83)   type(lu_decomp), intent(out) :: decomp
    84) 
    85)   ! "Variable" specific heat at constant pressure.
    86)   real(r8), intent(in), optional  :: cpairv(pcols,pver)
    87) 
    88)   ! --------------- !
    89)   ! Local Variables !
    90)   ! --------------- !
    91) 
    92)   ! Column and level indices.
    93)   integer :: i, k
    94) 
    95)   ! ----------------------- !
    96)   ! Main Computation Begins !
    97)   ! ----------------------- !
    98) 
    99)   decomp = lu_decomp(ncol, pver)
   100) 
   101)   ! Determine superdiagonal (ca(k)) and subdiagonal (cc(k)) coeffs of the
   102)   ! tridiagonal diffusion matrix. The diagonal elements  (cb=1+ca+cc) are
   103)   ! a combination of ca and cc; they are not required by the solver.
   104) 
   105)   !
   106)   ! If switch present and set true, then input kv is kvt for use in
   107)   ! diagonal calculations
   108)   !
   109) 
   110)   if ( present(cpairv) ) then
   111)      do k = nbot-1, ntop, -1
   112)         do i = 1, ncol
   113)            decomp%ca(i,k  ) = kv(i,k+1)*tmpi(i,k+1)*rpdel(i,k  ) &
   114)                 / cpairv(i,k)
   115)            decomp%cc(i,k+1) = kv(i,k+1)*tmpi(i,k+1)*rpdel(i,k+1) &
   116)                 / cpairv(i,k+1)
   117)         end do
   118)      end do
   119)   else
   120)      do k = nbot - 1, ntop, -1
   121)         do i = 1, ncol
   122)            decomp%ca(i,k  ) = kv(i,k+1) * tmpi(i,k+1) * rpdel(i,k  )
   123)            decomp%cc(i,k+1) = kv(i,k+1) * tmpi(i,k+1) * rpdel(i,k+1)
   124)         end do
   125)      end do
   126)   endif
   127) 
   128)   ! The bottom element of the upper diagonal (ca) is zero (not used).
   129)   ! The subdiagonal (cc) is not needed in the solver.
   130)   decomp%ca(:,nbot) = 0._r8
   131) 
   132)   ! Calculate e(k). This term is required in the solution of the
   133)   ! tridiagonal matrix as defined by the implicit diffusion equation.
   134) 
   135)   do i = 1, ncol
   136)      decomp%dnom(i,nbot) = 1._r8/ &
   137)           (1._r8 + decomp%cc(i,nbot) + ksrf(i)*ztodt*gravit*rpdel(i,nbot))
   138)      decomp%ze(i,nbot)   = decomp%cc(i,nbot) * decomp%dnom(i,nbot)
   139)   end do
   140) 
   141)   do k = nbot - 1, ntop + 1, -1
   142)      do i = 1, ncol
   143)         decomp%dnom(i,k) = 1._r8/ &
   144)              (1._r8 + decomp%ca(i,k) + decomp%cc(i,k) - &
   145)              decomp%ca(i,k)*decomp%ze(i,k+1))
   146)         decomp%ze(i,k)   = decomp%cc(i,k) * decomp%dnom(i,k)
   147)      end do
   148)   end do
   149) 
   150)   do i = 1, ncol
   151)      decomp%dnom(i,ntop) = 1._r8/ &
   152)           (1._r8 + decomp%ca(i,ntop) + cc_top(i) - &
   153)           decomp%ca(i,ntop)*decomp%ze(i,ntop+1))
   154)   end do
   155) 
   156) end subroutine vd_lu_decomp

   160) subroutine vd_lu_solve(pcols, pver,   ncol, &
   161)                        q,     decomp, ntop, nbot, cd_top)
   162)   !-----------------------------------------------------------------------!
   163)   ! Solve the implicit vertical diffusion equation with zero flux         !
   164)   ! boundary conditions. Actual surface fluxes are explicit (rather than  !
   165)   ! implicit) and applied separately. Procedure for solution of the       !
   166)   ! implicit equation follows Richtmyer and Morton (1967,pp 198-200).     !
   167)   !                                                                       !
   168)   ! The equation solved is                                                !
   169)   !                                                                       !
   170)   !     -ca(k)*q(k+1) + cb(k)*q(k) - cc(k)*q(k-1) = d(k),                 !
   171)   !                                                                       !
   172)   ! where d(k) is the input profile and q(k) is the output profile        !
   173)   !                                                                       !
   174)   ! The solution has the form                                             !
   175)   !                                                                       !
   176)   !     q(k) = ze(k)*q(k-1) + zf(k)                                       !
   177)   !                                                                       !
   178)   !     ze(k) = cc(k) * dnom(k)                                           !
   179)   !                                                                       !
   180)   !     zf(k) = [d(k) + ca(k)*zf(k+1)] * dnom(k)                          !
   181)   !                                                                       !
   182)   !     dnom(k) = 1/[cb(k) - ca(k)*ze(k+1)]                               !
   183)   !             = 1/[1 + ca(k) + cc(k) - ca(k)*ze(k+1)]                   !
   184)   !                                                                       !
   185)   ! Note that the same routine is used for temperature, momentum and      !
   186)   ! tracers, and that input variables are replaced.                       !
   187)   ! ----------------------------------------------------------------------!
   188) 
   189)   ! ---------------------- !
   190)   ! Input-Output Arguments !
   191)   ! ---------------------- !
   192) 
   193)   ! Allocated column and level dimensions.
   194)   integer,  intent(in)  :: pcols, pver
   195)   ! Columns actually used during computation.
   196)   integer,  intent(in)  :: ncol
   197)   ! Top and bottom levels to operate on.
   198)   integer,  intent(in)  :: ntop, nbot
   199) 
   200)   ! LU decomposition information.
   201)   type(lu_decomp), intent(in) :: decomp
   202) 
   203)   ! cc_top * ubc value
   204)   real(r8), intent(in)    :: cd_top(pcols)
   205) 
   206)   ! Constituent field.
   207)   real(r8), intent(inout) :: q(pcols,pver)
   208) 
   209)   ! --------------- !
   210)   ! Local Variables !
   211)   ! --------------- !
   212) 
   213)   ! Term in tri-diag solution.
   214)   real(r8) :: zf(ncol,pver)
   215)   ! Column and level indices.
   216)   integer :: i, k
   217) 
   218)   ! ----------------------- !
   219)   ! Main Computation Begins !
   220)   ! ----------------------- !
   221) 
   222)   ! Calculate zf(k). Terms zf(k) and ze(k) are required in solution of
   223)   ! tridiagonal matrix defined by implicit diffusion equation.
   224)   ! Note that only levels ntop through nbot need be solved for.
   225) 
   226)   do i = 1, ncol
   227)      zf(i,nbot) = q(i,nbot)*decomp%dnom(i,nbot)
   228)   end do
   229) 
   230)   do k = nbot - 1, ntop + 1, -1
   231)      do i = 1, ncol
   232)         zf(i,k) = (q(i,k) + decomp%ca(i,k)*zf(i,k+1)) * decomp%dnom(i,k)
   233)      end do
   234)   end do
   235) 
   236)   ! Include boundary condition on top element
   237) 
   238)   k = ntop
   239)   do i = 1, ncol
   240)      zf(i,k) = (q(i,k) + cd_top(i) + decomp%ca(i,k)*zf(i,k+1)) * &
   241)           decomp%dnom(i,k)
   242)   end do
   243) 
   244)   ! Perform back substitution
   245) 
   246)   do i = 1, ncol
   247)      q(i,ntop) = zf(i,ntop)
   248)   end do
   249) 
   250)   do k = ntop + 1, nbot, +1
   251)      do i = 1, ncol
   252)         q(i,k) = zf(i,k) + decomp%ze(i,k)*q(i,k-1)
   253)      end do
   254)   end do
   255) 
   256) end subroutine vd_lu_solve
