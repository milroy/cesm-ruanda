   184)   subroutine setFilters( nc )
   185) !
   186) ! !DESCRIPTION:
   187) ! Set CLM filters.
   188) !
   189) ! !USES:
   190)     use clmtype
   191)     use decompMod , only : get_clump_bounds
   192)     use pftvarcon , only : npcropmin
   193)     use clm_varcon, only : istsoil, isturb, icol_road_perv, istice_mec
   194)     use clm_varcon, only : istcrop
   195) !
   196) ! !ARGUMENTS:
   197)     implicit none
   198)     integer, intent(IN) :: nc          ! clump index
   199) !
   200) ! !REVISION HISTORY:
   201) ! Created by Mariana Vertenstein
   202) ! 2004.04.27 DGVM naturally-vegetated filter added by Forrest Hoffman
   203) ! 2005.09.12 Urban related filters added by Mariana Vertenstein
   204) !
   205) !EOP
   206) !
   207) ! LOCAL VARAIBLES:
   208)     integer , pointer :: ctype(:) ! column type
   209)     integer :: c,l,p       ! column, landunit, pft indices
   210)     integer :: fl          ! lake filter index
   211)     integer :: fnl,fnlu    ! non-lake filter index
   212)     integer :: fs          ! soil filter index
   213)     integer :: f, fn       ! general indices
   214)     integer :: begp, endp  ! per-clump beginning and ending pft indices
   215)     integer :: begc, endc  ! per-clump beginning and ending column indices
   216)     integer :: begl, endl  ! per-clump beginning and ending landunit indices
   217)     integer :: begg, endg  ! per-clump beginning and ending gridcell indices

   222)     ! Determine clump boundaries
   223) 
   224)     call get_clump_bounds(nc, begg, endg, begl, endl, begc, endc, begp, endp)
   225) 
   226)     ! Create lake and non-lake filters at column-level 
   227) 
   228)     fl = 0
   229)     fnl = 0
   230)     do c = begc,endc
   231)        l = col%landunit(c)
   232)        if (lun%lakpoi(l)) then
   233)           fl = fl + 1
   234)           filter(nc)%lakec(fl) = c
   235)        else
   236)           fnl = fnl + 1
   237)           filter(nc)%nolakec(fnl) = c
   238)        end if
   239)     end do
   240)     filter(nc)%num_lakec = fl
   241)     filter(nc)%num_nolakec = fnl
   242) 
   243)     ! Create lake and non-lake filters at pft-level 
   244)     ! Filter will only be active if weight of pft wrt gcell is nonzero
   245) 
   246)     fl = 0
   247)     fnl = 0
   248)     fnlu = 0
   249)     do p = begp,endp
   250)        l = pft%landunit(p)
   251)        if (pft%wtgcell(p) > 0._r8    &
   252)                    .or.                       &
   253)            lun%itype(l)==istice_mec) then  ! some glacier_mec columns have zero weight
   254) 
   255)           l = pft%landunit(p)
   256)           if (lun%lakpoi(l) ) then
   257)              fl = fl + 1
   258)              filter(nc)%lakep(fl) = p
   259)           else
   260)              fnl = fnl + 1
   261)              filter(nc)%nolakep(fnl) = p
   262)              if (lun%itype(l) /= isturb) then
   263)                 fnlu = fnlu + 1
   264)                 filter(nc)%nolakeurbanp(fnlu) = p
   265)              end if
   266)           end if
   267)        end if
   268)     end do
   269)     filter(nc)%num_lakep = fl
   270)     filter(nc)%num_nolakep = fnl
   271)     filter(nc)%num_nolakeurbanp = fnlu
   272) 
   273)     ! Create soil filter at column-level
   274) 
   275)     fs = 0
   276)     do c = begc,endc
   277)        l = col%landunit(c)
   278)        if (lun%itype(l) == istsoil .or. lun%itype(l) == istcrop) then
   279)           fs = fs + 1
   280)           filter(nc)%soilc(fs) = c
   281)        end if
   282)     end do
   283)     filter(nc)%num_soilc = fs
   284) 
   285)     ! Create soil filter at pft-level
   286)     ! Filter will only be active if weight of pft wrt gcell is nonzero
   287) 
   288)     fs = 0
   289)     do p = begp,endp
   290)        if (pft%wtgcell(p) > 0._r8) then
   291)           l = pft%landunit(p)
   292)           if (lun%itype(l) == istsoil .or. lun%itype(l) == istcrop) then
   293)              fs = fs + 1
   294)              filter(nc)%soilp(fs) = p
   295)           end if
   296)        end if
   297)     end do
   298)     filter(nc)%num_soilp = fs
   299) 
   300)     ! Create column-level hydrology filter (soil and Urban pervious road cols) 
   301) 
   302)     f = 0
   303)     do c = begc,endc
   304)        l = col%landunit(c)
   305)        if (lun%itype(l) == istsoil .or. ctype(c) == icol_road_perv .or. &
   306)            lun%itype(l) == istcrop) then
   307)           f = f + 1
   308)           filter(nc)%hydrologyc(f) = c
   309)        end if
   310)     end do
   311)     filter(nc)%num_hydrologyc = f
   312) 
   313)     ! Create prognostic crop and soil w/o prog. crop filters at pft-level
   314)     ! according to where the crop model should be used
   315) 
   316)     fl  = 0
   317)     fnl = 0
   318)     do p = begp,endp
   319)        if (pft%wtgcell(p) > 0._r8) then
   320)           if (pft%itype(p) >= npcropmin) then !skips 2 generic crop types
   321)              fl = fl + 1
   322)              filter(nc)%pcropp(fl) = p
   323)           else
   324)              l = pft%landunit(p)
   325)              if (lun%itype(l) == istsoil .or. lun%itype(l) == istcrop) then
   326)                 fnl = fnl + 1
   327)                 filter(nc)%soilnopcropp(fnl) = p
   328)              end if
   329)           end if
   330)        end if
   331)     end do
   332)     filter(nc)%num_pcropp   = fl
   333)     filter(nc)%soilnopcropp = fnl   ! This wasn't being set before...
   334) 
   335)     ! Create landunit-level urban and non-urban filters
   336) 
   337)     f = 0
   338)     fn = 0
   339)     do l = begl,endl
   340)        if (lun%itype(l) == isturb) then
   341)           f = f + 1
   342)           filter(nc)%urbanl(f) = l
   343)        else
   344)           fn = fn + 1
   345)           filter(nc)%nourbanl(fn) = l
   346)        end if
   347)     end do
   348)     filter(nc)%num_urbanl = f
   349)     filter(nc)%num_nourbanl = fn
   350) 
   351)     ! Create column-level urban and non-urban filters
   352) 
   353)     f = 0
   354)     fn = 0
   355)     do c = begc,endc
   356)        l = col%landunit(c)
   357)        if (lun%itype(l) == isturb) then
   358)           f = f + 1
   359)           filter(nc)%urbanc(f) = c
   360)        else
   361)           fn = fn + 1
   362)           filter(nc)%nourbanc(fn) = c
   363)        end if
   364)     end do
   365)     filter(nc)%num_urbanc = f
   366)     filter(nc)%num_nourbanc = fn
   367) 
   368)     ! Create pft-level urban and non-urban filters
   369) 
   370)     f = 0
   371)     fn = 0
   372)     do p = begp,endp
   373)        l = pft%landunit(p)
   374)        if (lun%itype(l) == isturb .and. pft%wtgcell(p) > 0._r8) then
   375)           f = f + 1
   376)           filter(nc)%urbanp(f) = p
   377)        else
   378)           fn = fn + 1
   379)           filter(nc)%nourbanp(fn) = p 
   380)        end if
   381)     end do
   382)     filter(nc)%num_urbanp = f
   383)     filter(nc)%num_nourbanp = fn
   384) 
   385)     ! Note: snow filters are reconstructed each time step in Hydrology2
   386)     ! Note: CNDV "pft present" filter is reconstructed each time CNDV is run
   387) 
   388)   end subroutine setFilters
