    79)   ! This reads the megan_emis_nl namelist group in drv_flds_in and parses the
    80)   ! namelist information for the driver, CLM, and CAM.
    81)   !
    82)   ! Namelist variables:
    83)   !   megan_specifier, megan_mapped_emisfctrs, megan_factors_file
    84)   !
    85)   ! megan_specifier is a series of strings where each string contains one 
    86)   !  CAM chemistry constituent name (left of = sign) and one or more MEGAN 
    87)   !  compounds (seperated by + sign if more than one).  The specification of 
    88)   !  the MEGAN compounds to the right of the = signs tells the MEGAN VOC 
    89)   !  model within CLM how to construct the VOC fluxes using the factors in 
    90)   !  megan_factors_file and land surface state.
    91)   !
    92)   ! megan_factors_file read by CLM contains valid MEGAN compound names,
    93)   !  MEGAN class groupings and scalar emission factors
    94)   ! 
    95)   ! megan_mapped_emisfctrs switch is used to tell the MEGAN model to use 
    96)   !  mapped emission factors read in from the CLM surface data input file
    97)   !  rather than the scalar factors from megan_factors_file
    98)   !
    99)   ! Example:
   100)   ! &megan_emis_nl
   101)   !  megan_specifier = 'ISOP = isoprene',
   102)   !     'C10H16 = myrcene + sabinene + limonene + carene_3 + ocimene_t_b + pinene_b + ...',
   103)   !     'CH3OH = methanol',
   104)   !     'C2H5OH = ethanol',
   105)   !     'CH2O = formaldehyde',
   106)   !     'CH3CHO = acetaldehyde',
   107)   ! ...
   108)   !  megan_factors_file = '$datapath/megan_emis_factors.nc'
   109)   ! /
