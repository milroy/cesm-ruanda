   117) subroutine wv_sat_readnl(nlfile)
   118)   !------------------------------------------------------------------!
   119)   ! Purpose:                                                         !
   120)   !   Get runtime options for wv_saturation.                         !
   121)   !------------------------------------------------------------------!
   122) 
   123)   use wv_sat_methods, only: wv_sat_get_scheme_idx, &
   124)                             wv_sat_valid_idx, &
   125)                             wv_sat_set_default
   126) 
   127)   use spmd_utils,      only: masterproc
   128)   use namelist_utils,  only: find_group_name
   129)   use units,           only: getunit, freeunit
   130)   use mpishorthand
   131)   use abortutils,      only: endrun
   132) 
   133)   character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
   134)    
   135)   ! Local variables
   136)   integer :: unitn, ierr
   137) 
   138)   character(len=32) :: wv_sat_scheme = "GoffGratch"
   139) 
   140)   character(len=*), parameter :: subname = 'wv_sat_readnl'
   141) 
   142)   namelist /wv_sat_nl/ wv_sat_scheme
   143)   !-----------------------------------------------------------------------------
   144) 
   145)   if (masterproc) then
   146)      unitn = getunit()
   147)      open( unitn, file=trim(nlfile), status='old' )
   148)      call find_group_name(unitn, 'wv_sat_nl', status=ierr)
   149)      if (ierr == 0) then
   150)         read(unitn, wv_sat_nl, iostat=ierr)
   151)         if (ierr /= 0) then
   152)            call endrun(subname // ':: ERROR reading namelist')
   153)            return
   154)         end if
   155)      end if
   156)      close(unitn)
   157)      call freeunit(unitn)
   158) 
   159)   end if
   160) 
   161) #ifdef SPMD
   162)   call mpibcast(wv_sat_scheme, len(wv_sat_scheme) , mpichar, 0, mpicom)
   163) #endif
   164) 
   165)   if (.not. wv_sat_set_default(wv_sat_scheme)) then
   166)      call endrun('wv_sat_readnl :: Invalid wv_sat_scheme.')
   167)      return
   168)   end if
   169) 
   170) end subroutine wv_sat_readnl

   172) subroutine wv_sat_init
   173)   !------------------------------------------------------------------!
   174)   ! Purpose:                                                         !
   175)   !   Initialize module (e.g. setting parameters, initializing the   !
   176)   !   SVP lookup table).                                             !
   177)   !------------------------------------------------------------------!
   178) 
   179)   use wv_sat_methods, only: wv_sat_methods_init, &
   180)                             wv_sat_get_scheme_idx, &
   181)                             wv_sat_valid_idx
   182)   use spmd_utils,     only: masterproc
   183)   use cam_logfile,    only: iulog
   184)   use abortutils,     only: endrun
   185)   use shr_assert_mod, only: shr_assert_in_domain
   186)   use error_messages, only: handle_errmsg
   187) 
   188)   integer :: status
   189) 
   190)   ! For wv_sat_methods error reporting.
   191)   character(len=256) :: errstring
   192) 
   193)   ! For generating internal SVP table.
   194)   real(r8) :: t         ! Temperature
   195)   integer  :: i         ! Increment counter
   196) 
   197)   ! Precalculated because so frequently used.
   198)   omeps  = 1.0_r8 - epsilo
   199) 
   200)   ! Transition range method is only valid for transition temperatures at:
   201)   ! -40 deg C < T < 0 deg C
   202)   call shr_assert_in_domain(ttrice, ge=0._r8, le=40._r8, varname="ttrice",&
   203)        msg="wv_sat_init: Invalid transition temperature range.")
   204) 
   205) ! This parameter uses a hardcoded 287.04_r8?
   206)   c3 = 287.04_r8*(7.5_r8*log(10._r8))/cpair
   207) 
   208) ! Init "methods" module containing actual SVP formulae.
   209) 
   210)   call wv_sat_methods_init(r8, tmelt, h2otrip, tboil, ttrice, &
   211)        epsilo, errstring)
   212) 
   213)   call handle_errmsg(errstring, subname="wv_sat_methods_init")
   214) 
   215)   ! Add two to make the table slightly too big, just in case.
   216)   plenest = ceiling(tmax-tmin) + 2
   217) 
   218)   ! Allocate SVP table.
   219)   allocate(estbl(plenest), stat=status)
   220)   if (status /= 0) then 
   221)      call endrun('wv_sat_init :: ERROR allocating saturation vapor pressure table')
   222)      return
   223)   end if
   224) 
   225)   do i = 1, plenest
   226)     estbl(i) = svp_trans(tmin + real(i-1,r8))
   227)   end do
   228) 
   229)   if (masterproc) then
   230)      write(iulog,*)' *** SATURATION VAPOR PRESSURE TABLE COMPLETED ***'
   231)   end if
   232) 
   233) end subroutine wv_sat_init

   304) ! Does linear interpolation from nearest values found
   305) ! in the table (estbl).
   306) elemental function estblf(t) result(es)
   307) 
   308)   real(r8), intent(in) :: t ! Temperature 
   309)   real(r8) :: es            ! SVP (Pa)
   310) 
   311)   integer  :: i         ! Index for t in the table
   312)   real(r8) :: t_tmp     ! intermediate temperature for es look-up
   313) 
   314)   real(r8) :: weight ! Weight for interpolation
   315) 
   316)   t_tmp = max(min(t,tmax)-tmin, 0._r8)   ! Number of table entries above tmin
   317)   i = int(t_tmp) + 1                     ! Corresponding index.
   318)   weight = t_tmp - aint(t_tmp, r8)       ! Fractional part of t_tmp (for interpolation).
   319)   es = (1._r8 - weight)*estbl(i) + weight*estbl(i+1)
   320) 
   321) end function estblf

   323) ! Get enthalpy based only on temperature

   341) elemental subroutine no_ip_hltalt(t, hltalt)
   342)   !------------------------------------------------------------------!
   343)   ! Purpose:                                                         !
   344)   !   Calculate latent heat of vaporization of pure liquid water at  !
   345)   !   a given temperature.                                           !
   346)   !------------------------------------------------------------------!
   347) 
   348)   ! Inputs
   349)   real(r8), intent(in) :: t        ! Temperature
   350)   ! Outputs
   351)   real(r8), intent(out) :: hltalt  ! Appropriately modified hlat
   352) 
   353)   hltalt = latvap
   354) 
   355)   ! Account for change of latvap with t above freezing where
   356)   ! constant slope is given by -2369 j/(kg c) = cpv - cw
   357)   if (t >= tmelt) then
   358)      hltalt = hltalt - 2369.0_r8*(t-tmelt)
   359)   end if
   360) 
   361) end subroutine no_ip_hltalt
   362) 
   363) elemental subroutine calc_hltalt(t, hltalt, tterm)
   364)   !------------------------------------------------------------------!
   365)   ! Purpose:                                                         !
   366)   !   Calculate latent heat of vaporization of water at a given      !
   367)   !   temperature, taking into account the ice phase if temperature  !
   368)   !   is below freezing.                                             !
   369)   !   Optional argument also calculates a term used to calculate     !
   370)   !   d(es)/dT within the water-ice transition range.                !
   371)   !------------------------------------------------------------------!
   372) 
   373)   ! Inputs
   374)   real(r8), intent(in) :: t        ! Temperature
   375)   ! Outputs
   376)   real(r8), intent(out) :: hltalt  ! Appropriately modified hlat
   377)   ! Term to account for d(es)/dT in transition region.
   378)   real(r8), intent(out), optional :: tterm
   379) 
   380)   ! Local variables
   381)   real(r8) :: tc      ! Temperature in degrees C
   382)   real(r8) :: weight  ! Weight for es transition from water to ice
   383)   ! Loop iterator
   384)   integer :: i
   385) 
   386)   if (present(tterm)) tterm = 0.0_r8
   387) 
   388)   call no_ip_hltalt(t,hltalt)
   389)   if (t < tmelt) then
   390)      ! Weighting of hlat accounts for transition from water to ice.
   391)      tc = t - tmelt
   392) 
   393)      if (tc >= -ttrice) then
   394)         weight = -tc/ttrice
   395) 
   396)         ! polynomial expression approximates difference between es
   397)         ! over water and es over ice from 0 to -ttrice (C) (max of
   398)         ! ttrice is 40): required for accurate estimate of es
   399)         ! derivative in transition range from ice to water
   400)         if (present(tterm)) then
   401)            do i = size(pcf), 1, -1
   402)               tterm = pcf(i) + tc*tterm
   403)            end do
   404)            tterm = tterm/ttrice
   405)         end if
   406) 
   407)      else
   408)         weight = 1.0_r8
   409)      end if
   410) 
   411)      hltalt = hltalt + weight*latice
   412) 
   413)   end if
   414) 
   415) end subroutine calc_hltalt

   421) ! Temperature derivative outputs, for qsat_*
   422) elemental subroutine deriv_outputs(t, p, es, qs, hltalt, tterm, &
   423)      gam, dqsdt)
   424) 
   425)   ! Inputs
   426)   real(r8), intent(in) :: t      ! Temperature
   427)   real(r8), intent(in) :: p      ! Pressure
   428)   real(r8), intent(in) :: es     ! Saturation vapor pressure
   429)   real(r8), intent(in) :: qs     ! Saturation specific humidity
   430)   real(r8), intent(in) :: hltalt ! Modified latent heat
   431)   real(r8), intent(in) :: tterm  ! Extra term for d(es)/dT in
   432)                                  ! transition region.
   433) 
   434)   ! Outputs
   435)   real(r8), intent(out), optional :: gam      ! (hltalt/cpair)*(d(qs)/dt)
   436)   real(r8), intent(out), optional :: dqsdt    ! (d(qs)/dt)
   437) 
   438)   ! Local variables
   439)   real(r8) :: desdt        ! d(es)/dt
   440)   real(r8) :: dqsdt_loc    ! local copy of dqsdt
   441) 
   442)   if (qs == 1.0_r8) then
   443)      dqsdt_loc = 0._r8
   444)   else
   445)      desdt = hltalt*es/(rh2o*t*t) + tterm
   446)      dqsdt_loc = qs*p*desdt/(es*(p-omeps*es))
   447)   end if
   448) 
   449)   if (present(dqsdt)) dqsdt = dqsdt_loc
   450)   if (present(gam))   gam   = dqsdt_loc * (hltalt/cpair)
   451) 
   452) end subroutine deriv_outputs

   458) elemental subroutine qsat(t, p, es, qs, gam, dqsdt, enthalpy)
   459)   !------------------------------------------------------------------!
   460)   ! Purpose:                                                         !
   461)   !   Look up and return saturation vapor pressure from precomputed  !
   462)   !   table, then calculate and return saturation specific humidity. !
   463)   !   Optionally return various temperature derivatives or enthalpy  !
   464)   !   at saturation.                                                 !
   465)   !------------------------------------------------------------------!
   466) 
   467)   ! Inputs
   468)   real(r8), intent(in) :: t    ! Temperature
   469)   real(r8), intent(in) :: p    ! Pressure
   470)   ! Outputs
   471)   real(r8), intent(out) :: es  ! Saturation vapor pressure
   472)   real(r8), intent(out) :: qs  ! Saturation specific humidity
   473) 
   474)   real(r8), intent(out), optional :: gam    ! (l/cpair)*(d(qs)/dt)
   475)   real(r8), intent(out), optional :: dqsdt  ! (d(qs)/dt)
   476)   real(r8), intent(out), optional :: enthalpy ! cpair*t + hltalt*q
   477) 
   478)   ! Local variables
   479)   real(r8) :: hltalt       ! Modified latent heat for T derivatives
   480)   real(r8) :: tterm        ! Account for d(es)/dT in transition region
   481) 
   482)   es = estblf(t)
   483) 
   484)   qs = svp_to_qsat(es, p)
   485) 
   486)   ! Ensures returned es is consistent with limiters on qs.
   487)   es = min(es, p)
   488) 
   489)   ! Calculate optional arguments.
   490)   if (present(gam) .or. present(dqsdt) .or. present(enthalpy)) then
   491) 
   492)      ! "generalized" analytic expression for t derivative of es
   493)      ! accurate to within 1 percent for 173.16 < t < 373.16
   494)      call calc_hltalt(t, hltalt, tterm)
   495) 
   496)      if (present(enthalpy)) enthalpy = tq_enthalpy(t, qs, hltalt)
   497) 
   498)      call deriv_outputs(t, p, es, qs, hltalt, tterm, &
   499)           gam=gam, dqsdt=dqsdt)
   500) 
   501)   end if
   502) 
   503) end subroutine qsat

   505) elemental subroutine qsat_water(t, p, es, qs, gam, dqsdt, enthalpy)
   506)   !------------------------------------------------------------------!
   507)   ! Purpose:                                                         !
   508)   !   Calculate SVP over water at a given temperature, and then      !
   509)   !   calculate and return saturation specific humidity.             !
   510)   !   Optionally return various temperature derivatives or enthalpy  !
   511)   !   at saturation.                                                 !
   512)   !------------------------------------------------------------------!
   513) 
   514)   use wv_sat_methods, only: wv_sat_qsat_water
   515) 
   516)   ! Inputs
   517)   real(r8), intent(in) :: t    ! Temperature
   518)   real(r8), intent(in) :: p    ! Pressure
   519)   ! Outputs
   520)   real(r8), intent(out) :: es  ! Saturation vapor pressure
   521)   real(r8), intent(out) :: qs  ! Saturation specific humidity
   522) 
   523)   real(r8), intent(out), optional :: gam    ! (l/cpair)*(d(qs)/dt)
   524)   real(r8), intent(out), optional :: dqsdt  ! (d(qs)/dt)
   525)   real(r8), intent(out), optional :: enthalpy ! cpair*t + hltalt*q
   526) 
   527)   ! Local variables
   528)   real(r8) :: hltalt       ! Modified latent heat for T derivatives
   529) 
   530)   call wv_sat_qsat_water(t, p, es, qs)
   531) 
   532)   if (present(gam) .or. present(dqsdt) .or. present(enthalpy)) then
   533) 
   534)      ! "generalized" analytic expression for t derivative of es
   535)      ! accurate to within 1 percent for 173.16 < t < 373.16
   536)      call no_ip_hltalt(t, hltalt)
   537) 
   538)      if (present(enthalpy)) enthalpy = tq_enthalpy(t, qs, hltalt)
   539) 
   540)      ! For pure water/ice transition term is 0.
   541)      call deriv_outputs(t, p, es, qs, hltalt, 0._r8, &
   542)           gam=gam, dqsdt=dqsdt)
   543) 
   544)   end if
   545) 
   546) end subroutine qsat_water

   548) elemental subroutine qsat_ice(t, p, es, qs, gam, dqsdt, enthalpy)
   549)   !------------------------------------------------------------------!
   550)   ! Purpose:                                                         !
   551)   !   Calculate SVP over ice at a given temperature, and then        !
   552)   !   calculate and return saturation specific humidity.             !
   553)   !   Optionally return various temperature derivatives or enthalpy  !
   554)   !   at saturation.                                                 !
   555)   !------------------------------------------------------------------!
   556) 
   557)   use wv_sat_methods, only: wv_sat_qsat_ice
   558) 
   559)   ! Inputs
   560)   real(r8), intent(in) :: t    ! Temperature
   561)   real(r8), intent(in) :: p    ! Pressure
   562)   ! Outputs
   563)   real(r8), intent(out) :: es  ! Saturation vapor pressure
   564)   real(r8), intent(out) :: qs  ! Saturation specific humidity
   565) 
   566)   real(r8), intent(out), optional :: gam    ! (l/cpair)*(d(qs)/dt)
   567)   real(r8), intent(out), optional :: dqsdt  ! (d(qs)/dt)
   568)   real(r8), intent(out), optional :: enthalpy ! cpair*t + hltalt*q
   569) 
   570)   ! Local variables
   571)   real(r8) :: hltalt       ! Modified latent heat for T derivatives
   572) 
   573)   call wv_sat_qsat_ice(t, p, es, qs)
   574) 
   575)   if (present(gam) .or. present(dqsdt) .or. present(enthalpy)) then
   576)

   644) elemental subroutine findsp (q, t, p, use_ice, tsp, qsp, status)
   645) !----------------------------------------------------------------------- 
   646) ! 
   647) ! Purpose: 
   648) !     find the wet bulb temperature for a given t and q
   649) !     in a longitude height section
   650) !     wet bulb temp is the temperature and spec humidity that is 
   651) !     just saturated and has the same enthalpy
   652) !     if q > qs(t) then tsp > t and qsp = qs(tsp) < q
   653) !     if q < qs(t) then tsp < t and qsp = qs(tsp) > q
   654) !
   655) ! Method: 
   656) ! a Newton method is used
   657) ! first guess uses an algorithm provided by John Petch from the UKMO
   658) ! we exclude points where the physical situation is unrealistic
   659) ! e.g. where the temperature is outside the range of validity for the
   660) !      saturation vapor pressure, or where the water vapor pressure
   661) !      exceeds the ambient pressure, or the saturation specific humidity is 
   662) !      unrealistic
   663) ! 
   664) ! Author: P. Rasch
   665) ! 
   666) !-----------------------------------------------------------------------
   667) !
   668) !     input arguments
   669) !
   670) 
   671)   real(r8), intent(in) :: q        ! water vapor (kg/kg)
   672)   real(r8), intent(in) :: t        ! temperature (K)
   673)   real(r8), intent(in) :: p        ! pressure    (Pa)
   674)   logical,  intent(in) :: use_ice  ! flag to include ice phase in calculations
   675) !
   676) ! output arguments
   677) !
   678)   real(r8), intent(out) :: tsp      ! saturation temp (K)
   679)   real(r8), intent(out) :: qsp      ! saturation mixing ratio (kg/kg)
   680)   integer,  intent(out) :: status   ! flag representing state of output
   681)                                     ! 0 => Successful convergence
   682)                                     ! 1 => No calculation done: pressure or specific
   683)                                     !      humidity not within usable range
   684)                                     ! 2 => Run failed to converge
   685)                                     ! 4 => Temperature fell below minimum
   686)                                     ! 8 => Enthalpy not conserved
   687) !
   688) ! local variables
   689) !
   690)   integer, parameter :: iter = 8    ! max number of times to iterate the calculation
   691)   integer :: l                      ! iterator
   692) 
   693)   real(r8) es                   ! sat. vapor pressure
   694)   real(r8) gam                  ! change in sat spec. hum. wrt temperature (times hltalt/cpair)
   695)   real(r8) dgdt                 ! work variable
   696)   real(r8) g                    ! work variable
   697)   real(r8) hltalt               ! lat. heat. of vap.
   698)   real(r8) qs                   ! spec. hum. of water vapor
   699) 
   700) ! work variables
   701)   real(r8) t1, q1, dt, dq
   702)   real(r8) qvd
   703)   real(r8) r1b, c1, c2
   704)   real(r8), parameter :: dttol = 1.e-4_r8 ! the relative temp error tolerance required to quit the iteration
   705)   real(r8), parameter :: dqtol = 1.e-4_r8 ! the relative moisture error tolerance required to quit the iteration
   706)   real(r8) enin, enout
   707) 
   708)   ! Saturation specific humidity at this temperature
   709)   if (use_ice) then
   710)      call qsat(t, p, es, qs)
   711)   else
   712)      call qsat_water(t, p, es, qs)
   713)   end if
   714) 
   715)   ! make sure a meaningful calculation is possible
   716)   if (p <= 5._r8*es .or. qs <= 0._r8 .or. qs >= 0.5_r8 &
   717)        .or. t < tmin .or. t > tmax) then
   718)      status = 1
   719)      ! Keep initial parameters when conditions aren't suitable
   720)      tsp = t
   721)      qsp = q
   722)      enin = 1._r8
   723)      enout = 1._r8
   724) 
   725)      return
   726)   end if
   727) 
   728)   ! Prepare to iterate
   729)   status = 2
   730) 
   731)   ! Get initial enthalpy
   732)   if (use_ice) then
   733)      call calc_hltalt(t,hltalt)
   734)   else
   735)      call no_ip_hltalt(t,hltalt)
   736)   end if
   737)   enin = tq_enthalpy(t, q, hltalt)
   738) 
   739)   ! make a guess at the wet bulb temp using a UKMO algorithm (from J. Petch)
   740)   c1 = hltalt*c3
   741)   c2 = (t + 36._r8)**2
   742)   r1b = c2/(c2 + c1*qs)
   743)   qvd = r1b * (q - qs)
   744)   tsp = t + ((hltalt/cpair)*qvd)
   745) 
   746)   ! Generate qsp, gam, and enout from tsp.
   747)   if (use_ice) then
   748)      call qsat(tsp, p, es, qsp, gam=gam, enthalpy=enout)
   749)   else
   750)      call qsat_water(tsp, p, es, qsp, gam=gam, enthalpy=enout)
   751)   end if
   752) 
   753)   ! iterate on first guess
   754)   do l = 1, iter
   755) 
   756)      g = enin - enout
   757)      dgdt = -cpair * (1 + gam)
   758) 
   759)      ! New tsp
   760)      t1 = tsp - g/dgdt
   761)      dt = abs(t1 - tsp)/t1
   762)      tsp = t1
   763) 
   764)      ! bail out if past end of temperature range
   765)      if ( tsp < tmin ) then
   766)         tsp = tmin
   767)         ! Get latent heat and set qsp to a value
   768)         ! that preserves enthalpy.
   769)         if (use_ice) then
   770)            call calc_hltalt(tsp,hltalt)
   771)         else
   772)            call no_ip_hltalt(tsp,hltalt)
   773)         end if
   774)         qsp = (enin - cpair*tsp)/hltalt
   775)         enout = tq_enthalpy(tsp, qsp, hltalt)
   776)         status = 4
   777)         exit
   778)      end if
   779) 
   780)      ! Re-generate qsp, gam, and enout from new tsp.
   781)      if (use_ice) then
   782)         call qsat(tsp, p, es, q1, gam=gam, enthalpy=enout)
   783)      else
   784)         call qsat_water(tsp, p, es, q1, gam=gam, enthalpy=enout)
   785)      end if
   786)      dq = abs(q1 - qsp)/max(q1,1.e-12_r8)
   787)      qsp = q1
   788) 
   789)      ! if converged at this point, exclude it from more iterations
   790)      if (dt < dttol .and. dq < dqtol) then
   791)         status = 0
   792)         exit
   793)      endif
   794)   end do
   795) 
   796)   ! Test for enthalpy conservation
   797)   if (abs((enin-enout)/(enin+enout)) > 1.e-4_r8) status = 8
   798) 
   799) end subroutine findsp
