   248)   subroutine modal_aero_initialize(pbuf2d)
   249) 
   250)        use constituents,          only: pcnst
   251)        use physconst,             only: rhoh2o, mwh2o
   252)        use modal_aero_calcsize,   only: modal_aero_calcsize_init
   253)        use modal_aero_coag,       only: modal_aero_coag_init
   254)        use modal_aero_deposition, only: modal_aero_deposition_init
   255)        use modal_aero_gasaerexch, only: modal_aero_gasaerexch_init
   256)        use modal_aero_newnuc,     only: modal_aero_newnuc_init
   257)        use modal_aero_rename,     only: modal_aero_rename_init
   258)        use rad_constituents,      only: rad_cnst_get_info, rad_cnst_get_aer_props, &
   259)                                         rad_cnst_get_mode_props
   260)        use aerodep_flx,           only: aerodep_flx_prescribed
   261)        use physics_buffer,        only: physics_buffer_desc, pbuf_get_chunk
   262) 
   263)        type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   264) 
   265)        !--------------------------------------------------------------
   266)        ! ... local variables
   267)        !--------------------------------------------------------------
   268)        integer :: l, m, i, lchnk
   269)        integer :: m_idx, s_idx
   270) 
   271)        character(len=3) :: trnum       ! used to hold mode number (as characters)
   272)        integer :: iaerosol, ibulk
   273)        integer  :: numaerosols     ! number of bulk aerosols in climate list
   274)        character(len=20) :: bulkname
   275)        real(r8) :: pi
   276)        complex(r8), pointer  :: refindex_aer_sw(:), &
   277)             refindex_aer_lw(:)
   278)        real(r8), pointer :: qqcw(:,:)
   279)        real(r8), parameter :: huge_r8 = huge(1._r8)
   280)        character(len=*), parameter :: routine='modal_aero_initialize'
   281)        !-----------------------------------------------------------------------
   282) 
   283)        pi = 4._r8*atan(1._r8)    
   284) 
   285)        ! safety check on modal_aero, and modal_aero_3mode, modal_aero_7mode
   286) #if ( defined MODAL_AERO_3MODE ) && ( defined MODAL_AERO_7MODE )
   287)        call endrun( 'Error - when modal_aero defined, just 1 of modal_aero_3/7mode must be defined'
   288) #elif ( ! ( defined MODAL_AERO_3MODE ) ) && ( ! ( defined MODAL_AERO_7MODE ) )
   289)        call endrun( 'Error - when modal_aero defined, at least 1 of modal_aero_3/7mode must be defined'
   290) #endif
   291) 
   292)        ! Mode specific properties.
   293)        do m = 1, ntot_amode
   294)           call rad_cnst_get_mode_props(0, m, &
   295)              sigmag=sigmag_amode(m), dgnum=dgnum_amode(m), dgnumlo=dgnumlo_amode(m), &
   296)              dgnumhi=dgnumhi_amode(m), rhcrystal=rhcrystal_amode(m), rhdeliques=rhdeliques_amode(m))
   297) 
   298)           !   compute frequently used parameters: ln(sigmag),
   299)           !   volume-to-number and volume-to-surface conversions, ...
   300)           alnsg_amode(m) = log( sigmag_amode(m) )
   301) 
   302)           voltonumb_amode(m) = 1._r8 / ( (pi/6._r8)*                            &
   303)              (dgnum_amode(m)**3._r8)*exp(4.5_r8*alnsg_amode(m)**2._r8) )
   304)           voltonumblo_amode(m) = 1._r8 / ( (pi/6._r8)*                          &
   305)              (dgnumlo_amode(m)**3._r8)*exp(4.5_r8*alnsg_amode(m)**2._r8) )
   306)           voltonumbhi_amode(m) = 1._r8 / ( (pi/6._r8)*                          &
   307)              (dgnumhi_amode(m)**3._r8)*exp(4.5_r8*alnsg_amode(m)**2._r8) )
   308) 
   309)           alnv2n_amode(m)   = log( voltonumb_amode(m) )
   310)           alnv2nlo_amode(m) = log( voltonumblo_amode(m) )
   311)           alnv2nhi_amode(m) = log( voltonumbhi_amode(m) )
   312)        end do
   313) 
   314) 
   315)        ! Properties of mode specie types.
   316) 
   317)        !     values from Koepke, Hess, Schult and Shettle, Global Aerosol Data Set 
   318)        !     Report #243, Max-Planck Institute for Meteorology, 1997a
   319)        !     See also Hess, Koepke and Schult, Optical Properties of Aerosols and Clouds (OPAC)
   320)        !     BAMS, 1998.
   321) 
   322)        !      specrefndxsw(:ntot_aspectype)     = (/ (1.53,  0.01),   (1.53,  0.01),  (1.53,  0.01), &
   323)        !                                           (1.55,  0.01),   (1.55,  0.01),  (1.90, 0.60), &
   324)        !                                           (1.50, 1.0e-8), (1.50, 0.005) /)
   325)        !      specrefndxlw(:ntot_aspectype)   = (/ (2.0, 0.5),   (2.0, 0.5), (2.0, 0.5), &
   326)        !                                           (1.7, 0.5),   (1.7, 0.5), (2.22, 0.73), &
   327)        !                                           (1.50, 0.02), (2.6, 0.6) /)
   328)        !     get refractive indices from phys_prop files
   329) 
   330)        ! The following use of the rad_constituent interfaces makes the assumption that the
   331)        ! prognostic modes are used in the mode climate (index 0) list.
   332)        do l = 1, ntot_aspectype
   333) 
   334)           ! specname_amode is the species type.  This info call will return the mode and species
   335)           ! indices of the first occurance of the species type.
   336)           call rad_cnst_get_info(0, specname_amode(l), mode_idx=m_idx, spec_idx=s_idx)
   337) 
   338)           if (m_idx > 0 .and. s_idx > 0) then
   339) 
   340)              call rad_cnst_get_aer_props(0, m_idx, s_idx, &
   341)                 refindex_aer_sw=refindex_aer_sw, &
   342)                 refindex_aer_lw=refindex_aer_lw, &
   343)                 density_aer=specdens_amode(l), &
   344)                 hygro_aer=spechygro(l))
   345) 
   346)              specrefndxsw(:nswbands,l) = refindex_aer_sw(:nswbands)
   347)              specrefndxlw(:nlwbands,l) = refindex_aer_lw(:nlwbands)
   348) 
   349)           else
   350)              if (masterproc) then
   351)                 write(iulog,*) routine//': INFO: props not found for species type: ',trim(specname_amode(l))
   352)              end if
   353)              specdens_amode(l)         = huge_r8
   354)              spechygro(l)              = huge_r8
   355)              specrefndxsw(:nswbands,l) = (huge_r8, huge_r8)
   356)              specrefndxlw(:nlwbands,l) = (huge_r8, huge_r8)
   357)           endif
   358) 
   359)        end do
   360) 
   361) 
   362)        if (masterproc) write(iulog,9210)
   363)        do l = 1, ntot_aspectype
   364)           !            spechygro(l) = specnu(l)*specphi(l)*specsolfrac(l)*mwh2o*specdens_amode(l) / &
   365)           !	               (rhoh2o*specmw_amode(l))
   366)           if (masterproc) then
   367)              write(iulog,9211) l
   368)              write(iulog,9212) 'name            ', specname_amode(l)
   369)              write(iulog,9213) 'density, MW     ',                  &
   370)                   specdens_amode(l), specmw_amode(l)
   371)              write(iulog,9213) 'hygro', spechygro(l)
   372)              do i=1,nswbands
   373)                 write(iulog,9213) 'ref index sw    ', (specrefndxsw(i,l))
   374)              end do
   375)              do i=1,nlwbands
   376)                 write(iulog,9213) 'ref index ir    ', (specrefndxlw(i,l))
   377)              end do
   378)           end if
   379)        end do
   380) 
   381) 9210   format( // '*** init_aer_modes aerosol species-types' )
   382) 9211   format( 'spectype =', i4)

   783)      subroutine initaermodes_set_cnstnamecw
   784)        !
   785)        !   sets the cnst_name_cw
   786)        !
   787)        use constituents, only: pcnst, cnst_name
   788)        implicit none
   789) 
   790)        !   subr arguments (none)
   791) 
   792)        !   local variables
   793)        integer j, l, la, lc, ll, m
   794) 
   795)        !   set cnst_name_cw
   796)        cnst_name_cw = ' '
   797)        do m = 1, ntot_amode
   798)           do ll = 0, nspec_amode(m)
   799)              if (ll == 0) then
   800)                 la = numptr_amode(m)
   801)                 lc = numptrcw_amode(m)
   802)              else
   803)                 la = lmassptr_amode(ll,m)
   804)                 lc = lmassptrcw_amode(ll,m)
   805)              end if
   806)              if ((la < 1) .or. (la > pcnst) .or.   &
   807)                   (lc < 1) .or. (lc > pcnst)) then
   808)                 write(*,'(/2a/a,5(1x,i10))')   &
   809)                      '*** initaermodes_set_cnstnamecw error',   &
   810)                      ' -- bad la or lc',   &
   811)                      '    m, ll, la, lc, pcnst =', m, ll, la, lc, pcnst
   812)                 call endrun( '*** initaermodes_set_cnstnamecw error' )
   813)              end if
   814)              do j = 2, len( cnst_name(la) ) - 1
   815)                 if (cnst_name(la)(j:j+1) == '_a') then
   816)                    cnst_name_cw(lc) = cnst_name(la)
   817)                    cnst_name_cw(lc)(j:j+1) = '_c'
   818)                    exit
   819)                 else if (cnst_name(la)(j:j+1) == '_A') then
   820)                    cnst_name_cw(lc) = cnst_name(la)
   821)                    cnst_name_cw(lc)(j:j+1) = '_C'
   822)                    exit
   823)                 end if
   824)              end do
   825)              if (cnst_name_cw(lc) == ' ') then
   826)                 write(*,'(/2a/a,3(1x,i10),2x,a)')   &
