    40)   subroutine UrbanInitAero( )
    41) !
    42) ! !DESCRIPTION: 
    43) ! Calculate urban land unit aerodynamic constants using Macdonald (1998) as used in
    44) ! Grimmond and Oke (1999)
    45) !
    46) ! !USES:
    47)     use clmtype 
    48)     use clm_varcon, only : isturb, vkc
    49)     use decompMod , only : get_proc_bounds
    50) !
    51) ! !ARGUMENTS:
    52)     implicit none
    53) !
    54) ! local pointers to original implicit in arguments (urban clump)
    55) !
    56)     real(r8), pointer :: ht_roof(:)    ! height of urban roof (m)
    57)     real(r8), pointer :: canyon_hwr(:) ! ratio of building height to street width (-)
    58)     integer , pointer :: ltype(:)      ! landunit type
    59) !
    60) ! local pointers to original implicit out arguments
    61) !
    62)     real(r8), pointer :: z_0_town(:)   ! urban landunit momentum roughness length (m)
    63)     real(r8), pointer :: z_d_town(:)   ! urban landunit displacement height (m)
    64) !
    65) ! !CALLED FROM:
    66) ! subroutine initialize
    67) !
    68) ! !REVISION HISTORY:
    69) ! Created by Keith Oleson January 2005
    70) !
    71) !
    72) ! !LOCAL VARIABLES:
    73) !EOP
    74)     real(r8), parameter :: alpha = 4.43_r8 ! coefficient used to calculate z_d_town
    75)     real(r8), parameter :: beta = 1.0_r8   ! coefficient used to calculate z_d_town
    76)     real(r8), parameter :: C_d = 1.2_r8    ! drag coefficient as used in Grimmond and Oke (1999)
    77)     real(r8) :: plan_ai                    ! plan area index - ratio building area to plan area (-)
    78)     real(r8) :: frontal_ai                 ! frontal area index of buildings (-)
    79)     real(r8) :: build_lw_ratio             ! building short/long side ratio (-)
    80)     integer  :: l,g                        ! indices
    81)     integer  :: begp, endp                 ! clump beginning and ending pft indices
    82)     integer  :: begc, endc                 ! clump beginning and ending column indices
    83)     integer  :: begl, endl                 ! clump beginning and ending landunit indices
    84)     integer  :: begg, endg                 ! clump beginning and ending gridcell indices
    85) !-----------------------------------------------------------------------
    86) 
    87)     ! Assign local pointers to derived type members (landunit level)
    88) 
    89)     ltype      => lun%itype
    90)     z_0_town   => lun%z_0_town
    91)     z_d_town   => lun%z_d_town
    92)     ht_roof    => lun%ht_roof
    93)     canyon_hwr => lun%canyon_hwr
    94) 
    95)     call get_proc_bounds(begg, endg, begl, endl, begc, endc, begp, endp)
    96) 
    97)     do l = begl, endl 
    98)       if (ltype(l) == isturb) then 
    99) 
   100)          ! Calculate plan area index 
   101)          plan_ai = canyon_hwr(l)/(canyon_hwr(l) + 1._r8)
   102) 
   103)          ! Building shape shortside/longside ratio (e.g. 1 = square )
   104)          ! This assumes the building occupies the entire canyon length
   105)          build_lw_ratio = plan_ai
   106) 
   107)          ! Calculate frontal area index
   108)          frontal_ai = (1._r8 - plan_ai) * canyon_hwr(l)
   109) 
   110)          ! Adjust frontal area index for different building configuration
   111)          frontal_ai = frontal_ai * sqrt(1/build_lw_ratio) * sqrt(plan_ai)
   112)          
   113)          ! Calculate displacement height
   114)          
   115)          if (use_vancouver) then
   116)             z_d_town(l) = 3.5_r8
   117)          else if (use_mexicocity) then
   118)             z_d_town(l) = 10.9_r8
   119)          else
   120)             z_d_town(l) = (1._r8 + alpha**(-plan_ai) * (plan_ai - 1._r8)) * ht_roof(l)
   121)          end if
   122) 
   123)          ! Calculate the roughness length
   124)          
   125)          if (use_vancouver) then
   126)             z_0_town(l) = 0.35_r8
   127)          else if (use_mexicocity) then
   128)             z_0_town(l) = 2.2_r8
   129)          else
   130)             z_0_town(l) = ht_roof(l) * (1._r8 - z_d_town(l) / ht_roof(l)) * &
   131)                           exp(-1.0_r8 * (0.5_r8 * beta * C_d / vkc**2 * &
   132)                           (1 - z_d_town(l) / ht_roof(l)) * frontal_ai)**(-0.5_r8))
   133)          end if
   134)       end if
   135)    end do
   136) 
   137)  end subroutine UrbanInitAero

   257)              if (ctype(c) == icol_roof       ) emg(c) = urbinp%em_roof(g)
   258)              if (ctype(c) == icol_sunwall    ) emg(c) = urbinp%em_wall(g)

   261)              if (ctype(c) == icol_road_perv  ) emg(c) = urbinp%em_perroad(g)
   262)           end do
   263) 
   264)           ! Inferred from Sailor and Lu 2004
   265)           if (urban_traffic) then
   266)              eflx_traffic_factor(l) = 3.6_r8 * (canyon_hwr(l)-0.5_r8) + 1.0_r8
   267)           else
   268)              eflx_traffic_factor(l) = 0.0_r8
   269)           end if
   270) 
   271)           if (use_vancouver .or. use_mexicocity) then 
   272)              ! Freely evolving
   273)              t_building_max(l) = 380.00_r8
   274)              t_building_min(l) = 200.00_r8
   275)           else

   277)                 ! Overwrite values read in from urbinp by freely evolving values
   278)                 t_building_max(l) = 380.00_r8
   279)                 t_building_min(l) = 200.00_r8
   280)              end if
   281)           end if
   282)        else
   283)           eflx_traffic_factor(l) = spval
   284)           t_building_max(l) = spval
   285)           t_building_min(l) = spval
   286)        end if
   287)     end do
