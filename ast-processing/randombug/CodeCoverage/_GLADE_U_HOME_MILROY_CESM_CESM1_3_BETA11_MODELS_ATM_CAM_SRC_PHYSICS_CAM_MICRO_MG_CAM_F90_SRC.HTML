   149) subroutine micro_mg_cam_readnl(nlfile)
   150) 
   151)   use namelist_utils,  only: find_group_name
   152)   use units,           only: getunit, freeunit
   153)   use mpishorthand
   154) 
   155)   character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
   156) 
   157)   ! Namelist variables
   158)   logical :: micro_mg_do_cldice   = .true. ! do_cldice = .true., MG microphysics is prognosing cldice
   159)   logical :: micro_mg_do_cldliq   = .true. ! do_cldliq = .true., MG microphysics is prognosing cldliq
   160) 
   161)   ! Local variables
   162)   integer :: unitn, ierr
   163)   character(len=*), parameter :: subname = 'micro_mg_cam_readnl'
   164) 
   165)   namelist /micro_mg_nl/ micro_mg_version, micro_mg_sub_version, &
   166)        micro_mg_do_cldice, micro_mg_do_cldliq, microp_uniform
   167) 
   168)   !-----------------------------------------------------------------------------
   169) 
   170)   if (masterproc) then
   171)      unitn = getunit()
   172)      open( unitn, file=trim(nlfile), status='old' )
   173)      call find_group_name(unitn, 'micro_mg_nl', status=ierr)
   174)      if (ierr == 0) then
   175)         read(unitn, micro_mg_nl, iostat=ierr)
   176)         if (ierr /= 0) then
   177)            call endrun(subname // ':: ERROR reading namelist')
   178)         end if
   179)      end if
   180)      close(unitn)
   181)      call freeunit(unitn)
   182) 
   183)      ! set local variables
   184)      do_cldice  = micro_mg_do_cldice
   185)      do_cldliq  = micro_mg_do_cldliq
   186) 
   187)      ! Verify that version numbers are valid.
   188)      select case (micro_mg_version)
   189)      case (1)
   190)         select case (micro_mg_sub_version)
   191)         case(0)
   192)            ! MG version 1.0
   193)         case(5)
   194)            ! MG version 1.5 - MG2 development
   195)         case default
   196)            call bad_version_endrun()
   197)         end select
   198)      case default
   199)         call bad_version_endrun()
   200)      end select
   201) 
   202)   end if
   203) 
   204) #ifdef SPMD
   205)   ! Broadcast namelist variables
   206)   call mpibcast(micro_mg_version,     1, mpiint, 0, mpicom)
   207)   call mpibcast(micro_mg_sub_version, 1, mpiint, 0, mpicom)
   208)   call mpibcast(do_cldice,            1, mpilog, 0, mpicom)
   209)   call mpibcast(do_cldliq,            1, mpilog, 0, mpicom)
   210)   call mpibcast(microp_uniform,       1, mpilog, 0, mpicom)
   211) #endif
   212) 
   213) contains
   214) 
   215)   subroutine bad_version_endrun
   216)     ! Endrun wrapper with a more useful error message.
   217)     character(len=128) :: errstring
   218)     write(errstring,*) "Invalid version number specified for MG microphysics: ", &
   219)          micro_mg_version,".",micro_mg_sub_version
   220)     call endrun(errstring)
   221)   end subroutine bad_version_endrun
   222) 
   223) end subroutine micro_mg_cam_readnl
