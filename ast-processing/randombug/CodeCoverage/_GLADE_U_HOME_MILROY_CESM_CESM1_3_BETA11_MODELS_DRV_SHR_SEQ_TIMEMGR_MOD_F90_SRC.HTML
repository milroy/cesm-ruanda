  1080)       endif

  1104)

  1111)       call ESMF_ClockAdvance(SyncClock%ECP(seq_timemgr_nclock_ice)%EClock, rc=rc )

  1127)          if (seq_timemgr_alarmIsOn(SyncClock%ECP(n)%EClock,seq_timemgr_alarm_stop) .or. &

  1455) ! !IROUTINE: seq_timemgr_alarmSetOn -- turn alarm on
  1456) !

  1460) !      
  1461) ! !INTERFACE: ------------------------------------------------------------------
  1462) 
  1463) subroutine seq_timemgr_AlarmSetOn( EClock, alarmname)
  1464) 
  1465)     implicit none
  1466) 
  1467) ! !INPUT/OUTPUT PARAMETERS:
  1468) 
  1469)     type(ESMF_Clock), intent(INOUT) :: EClock      ! clock/alarm
  1470)     character(len=*), intent(IN), optional :: alarmname  ! alarmname
  1471) 
  1472) !EOP
  1473) 
  1474)     !----- local -----
  1475)     character(len=*), parameter :: subname = '(seq_timemgr_alarmSetOn) '
  1476)     character(len=*), parameter :: xalarm = 'unset'
  1477)     integer :: n
  1478)     integer :: rc
  1479)     logical :: found
  1480)     logical :: set
  1481)     character(len=64) :: name
  1482)     type(ESMF_Alarm),pointer :: EAlarm
  1483)     type(ESMF_Alarm),pointer :: EAlarm_list(:)
  1484)     integer(SHR_KIND_IN) :: AlarmCount ! Number of valid alarms
  1485) 
  1486) !-------------------------------------------------------------------------------
  1487) ! Notes: The Alarm_list is returned and only a subset of the alarms may
  1488) !   be initialized.  In the esmf_wrf_timemgr, numalarms is not used internally,
  1489) !   and the alarm pointer is valid if it's associated.  If it's not associated
  1490) !   the AlarmGet calls will generally return an error code.  What we really
  1491) !   want is to ignore the unset alarms.  So below, we have to kind of kludge
  1492) !   this up.  We set name=xalarm, a special value, before the AlarmGet call so 
  1493) !   if Alarm_list(n) is not associated, the name will remain the value of 
  1494) !   xalarm.  Then we check whether it's a valid alarm by first checking
  1495) !   the name vs xalarm.  If name is not xalarm, then it must be a valid alarm
  1496) !   and we either set found to true if we are setting all alarms or we compare
  1497) !   the name returned to the alarm name we're looking for and only set found
  1498) !   to true if the names match.
  1499) !-------------------------------------------------------------------------------
  1500) 
  1501)     set = .false.
  1502) 
  1503)     call seq_timemgr_EClockGetData(EClock, AlarmCount=AlarmCount)
  1504) #ifdef USE_ESMF_LIB
  1505)     allocate(EAlarm_list(AlarmCount))
  1506)     call ESMF_ClockGetAlarmList(EClock, alarmListFlag=ESMF_ALARMLIST_ALL, &
  1507)        alarmList=EAlarm_list, alarmCount=AlarmCount, rc=rc)
  1508) #else
  1509)     call ESMF_ClockGetAlarmList(EClock, EAlarm_list, rc=rc)
  1510) #endif
  1511)     call seq_timemgr_ESMFCodeCheck( rc, msg=subname//"Error from ESMF_ClockGetAlarmList" )
  1512)     do n = 1,AlarmCount
  1513)        found = .false.
  1514)        if (present(alarmname)) then
  1515)           call ESMF_AlarmGet(EAlarm_list(n), name=name)
  1516)           if (trim(name) == trim(alarmname)) found = .true.
  1517)        else
  1518)           found = .true.
  1519)        endif
  1520)        if (found) then
  1521)           set = .true.
  1522)           call ESMF_AlarmRingerOn( EAlarm_list(n), rc=rc )
  1523)           call seq_timemgr_ESMFCodeCheck( rc, msg=subname//"Error from ESMF_AlarmRingerOn" )
  1524)        endif
  1525)     enddo
  1526) 
  1527)     if (present(alarmname) .and. .not. set) then
  1528)        write(logunit,*) subname,' ERROR in alarmname ',trim(alarmname)
  1529)        call shr_sys_abort()
  1530)     endif
  1531) #ifdef USE_ESMF_LIB
  1532)     deallocate(EAlarm_list)
  1533) #endif
  1534) 
  1535) end subroutine seq_timemgr_AlarmSetOn

  1785)        endif
