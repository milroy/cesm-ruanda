    57)   subroutine restFile_write( file, nlend, noptr, rdate )
    58) !
    59) ! !DESCRIPTION:
    60) ! Read/write CLM restart file.
    61) !
    62) ! !USES:
    63)     use clm_time_manager , only : timemgr_restart_io, get_nstep
    64)     use subgridRestMod   , only : SubgridRest
    65)     use BiogeophysRestMod, only : BiogeophysRest
    66)     use CNRestMod        , only : CNRest
    67)     use CropRestMod      , only : CropRest
    68)     use accumulMod       , only : accumulRest
    69)     use histFileMod      , only : hist_restart_ncd
    70) !
    71) ! !ARGUMENTS:
    72)     implicit none
    73)     character(len=*) , intent(in) :: file            ! output netcdf restart file
    74)     logical,           intent(in) :: nlend	     ! if at the end of the simulation
    75)     character(len=*) , intent(in) :: rdate           ! restart file time stamp for name
    76)     logical,           intent(in), optional :: noptr ! if should NOT write to the restart pointer file
    77) !
    78) ! !CALLED FROM:
    79) ! subroutine clm_driver2
    80) !
    81) ! !REVISION HISTORY:
    82) ! Author: Mariana Vertenstein
    83) !
    84) !
    85) ! !LOCAL VARIABLES:
    86) !EOP
    87)     type(file_desc_t) :: ncid ! netcdf id
    88)     integer :: i       ! index
    89)     logical :: ptrfile ! write out the restart pointer file
    90) !-----------------------------------------------------------------------
    91) 
    92)     if ( present(noptr) )then
    93)        ptrfile = .not. noptr
    94)     else
    95)        ptrfile = .true.
    96)     end if
    97) 
    98)     ! --------------------------------------------
    99)     ! Open restart file
   100)     ! --------------------------------------------
   101) 
   102)     call restFile_open( flag='write', file=file, ncid=ncid )
   103) 
   104)     ! --------------------------------------------
   105)     ! Define dimensions and variables
   106)     ! --------------------------------------------
   107) 
   108)     call restFile_dimset ( ncid )
   109) 
   110)     ! Define restart file variables
   111) 
   112)     call timemgr_restart_io( ncid, flag='define' )
   113) 
   114)     call SubgridRest( ncid, flag='define' )
   115) 
   116)     call BiogeophysRest( ncid, flag='define' )
   117) 
   118)     if (use_cn) then
   119)        call CNRest( ncid, flag='define' )
   120)        if ( crop_prog ) call CropRest( ncid, flag='define' )
   121)     end if
   122) 
   123)     call accumulRest( ncid, flag='define' )
   124) 
   125)     call hist_restart_ncd ( ncid, flag='define', rdate=rdate )
   126) 
   127)     call restFile_enddef( ncid )
   128) 
   129)     ! --------------------------------------------
   130)     ! Write restart file variables
   131)     ! --------------------------------------------
   132)     
   133)     call timemgr_restart_io( ncid, flag='write' )
   134) 
   135)     call SubgridRest( ncid, flag='write' )
   136) 
   137)     call BiogeophysRest( ncid, flag='write' )
   138) 
   139)     if (use_cn) then
   140)        call CNRest( ncid, flag='write' )
   141)        if ( crop_prog ) call CropRest( ncid, flag='write' )
   142)     end if
   143) 
   144)     call accumulRest( ncid, flag='write' )
   145)     
   146)     call hist_restart_ncd (ncid, flag='write' )
   147) 
   148)     ! --------------------------------------------
   149)     ! Close restart file and write restart pointer file
   150)     ! --------------------------------------------
   151)     
   152)     call restFile_close( ncid )
   153)     call restFile_closeRestart( file, nlend )
   154)     
   155)     ! Write restart pointer file
   156)     
   157)     if ( ptrfile ) call restFile_write_pfile( file )
   158)     
   159)     ! Write out diagnostic info
   160) 
   161)     if (masterproc) then
   162)        write(iulog,*) 'Successfully wrote out restart data at nstep = ',get_nstep()
   163)        write(iulog,'(72a1)') ("-",i=1,60)
   164)     end if
   165)     
   166)   end subroutine restFile_write

   242)   subroutine restFile_getfile( file, path )
   243) !
   244) ! !DESCRIPTION:
   245) ! Determine and obtain netcdf restart file
   246) !
   247) ! !USES:
   248)     use clm_varctl, only : caseid, finidat, nrevsn, nsrest, brnch_retain_casename, &
   249)                            nsrContinue, nsrBranch, nsrStartup
   250)     use fileutils , only : getfil
   251) !
   252) ! !ARGUMENTS:
   253)     implicit none
   254)     character(len=*), intent(out) :: file  ! name of netcdf restart file
   255)     character(len=*), intent(out) :: path  ! full pathname of netcdf restart file
   256) !
   257) ! !CALLED FROM:
   258) ! subroutine initialize2
   259) !
   260) ! !REVISION HISTORY:
   261) ! Author: Mariana Vertenstein
   262) !
   263) !
   264) ! !LOCAL VARIABLES:
   265) !EOP
   266)     integer :: status                      ! return status
   267)     integer :: length                      ! temporary          
   268)     character(len=256) :: ftest,ctest      ! temporaries
   269) !-----------------------------------------------------------------------
   270) 
   271)     ! Continue run:
   272)     ! Restart file pathname is read restart pointer file 
   273)     
   274)     if (nsrest==nsrContinue) then
   275)        call restFile_read_pfile( path )
   276)        call getfil( path, file, 0 )
   277)     end if
   278)        
   279)     ! Branch run: 
   280)     ! Restart file pathname is obtained from namelist "nrevsn"
   281)     ! Check case name consistency (case name must be different for branch run, 
   282)     ! unless namelist specification states otherwise)
   283)     
   284)     if (nsrest==nsrBranch) then
   285)        length = len_trim(nrevsn)
   286)        if (nrevsn(length-2:length) == '.nc') then
   287)           path = trim(nrevsn) 
   288)        else
   289)           path = trim(nrevsn) // '.nc'
   290)        end if
   291)        call getfil( path, file, 0 )
   292)        
   293)        ! tcraig, adding xx. and .clm2 makes this more robust
   294)        ctest = 'xx.'//trim(caseid)//'.clm2'
   295)        ftest = 'xx.'//trim(file)
   296)        status = index(trim(ftest),trim(ctest))
   297)        if (status /= 0 .and. .not.(brnch_retain_casename)) then
   298)           write(iulog,*) 'Must change case name on branch run if ',&
   299)                'brnch_retain_casename namelist is not set'
   300)           write(iulog,*) 'previous case filename= ',trim(file),&
   301)                ' current case = ',trim(caseid), ' ctest = ',trim(ctest), &
   302)                ' ftest = ',trim(ftest)
   303)           call endrun()
   304)        end if
   305)     end if
   306) 
   307)     ! Initial run: 
   308)     ! Restart file pathname is obtained from namelist "finidat"
   309)     
   310)     if (nsrest==nsrStartup) then
   311)        call getfil( finidat, file, 0 )
   312)     end if
   313)     
   314)   end subroutine restFile_getfile

   322)   subroutine restFile_read_pfile( pnamer )
   323) !
   324) ! !DESCRIPTION:
   325) ! Setup restart file and perform necessary consistency checks
   326) !
   327) ! !USES:
   328)     use fileutils , only : opnfil, getavu, relavu
   329)     use clm_varctl, only : rpntfil, rpntdir, inst_suffix
   330) !
   331) ! !ARGUMENTS:
   332)     implicit none
   333)     character(len=*), intent(out) :: pnamer ! full path of restart file
   334) !
   335) ! !CALLED FROM:
   336) ! subroutine restart in this module
   337) !
   338) ! !REVISION HISTORY:
   339) ! Created by Mariana Vertenstein
   340) !
   341) !
   342) ! !LOCAL VARIABLES:
   343) !EOP
   344)     integer :: i                  ! indices
   345)     integer :: nio                ! restart unit
   346)     integer :: status             ! substring check status
   347)     character(len=256) :: locfn   ! Restart pointer file name
   348) !-----------------------------------------------------------------------
   349) 
   350)     ! Obtain the restart file from the restart pointer file. 
   351)     ! For restart runs, the restart pointer file contains the full pathname 
   352)     ! of the restart file. For branch runs, the namelist variable 
   353)     ! [nrevsn] contains the full pathname of the restart file. 
   354)     ! New history files are always created for branch runs.
   355)        
   356)     if (masterproc) then
   357)        write(iulog,*) 'Reading restart pointer file....'
   358)     endif
   359) 
   360)     nio = getavu()
   361)     locfn = trim(rpntdir) //'/'// trim(rpntfil)//trim(inst_suffix)
   362)     call opnfil (locfn, nio, 'f')
   363)     read (nio,'(a256)') pnamer
   364)     call relavu (nio)
   365) 
   366)     if (masterproc) then
   367)        write(iulog,*) 'Reading restart data.....'
   368)        write(iulog,'(72a1)') ("-",i=1,60)
   369)     end if
   370) 
   371)   end subroutine restFile_read_pfile

   540)   subroutine restFile_dimset( ncid )
   541) !
   542) ! !DESCRIPTION:
   543) ! Read/Write initial data from/to netCDF instantaneous initial data file
   544) !
   545) ! !USES:
   546)     use shr_kind_mod, only : r8 => shr_kind_r8
   547)     use clm_time_manager, only : get_nstep, get_curr_date
   548)     use spmdMod     , only : mpicom, MPI_LOGICAL
   549)     use clm_varctl  , only : caseid, ctitle, version, username, hostname, fsurdat, &
   550)                              conventions, source
   551)     use clm_varpar  , only : numrad, nlevlak, nlevsno, nlevgrnd
   552)     use decompMod   , only : get_proc_bounds, get_proc_global
   553) !
   554) ! !ARGUMENTS:
   555)     implicit none
   556)     type(file_desc_t), intent(inout) :: ncid
   557) !
   558) ! !REVISION HISTORY:
   559) !
   560) !
   561) ! !LOCAL VARIABLES:
   562) !EOP
   563)     integer :: yr                  ! current year (0 -> ...)
   564)     integer :: mon                 ! current month (1 -> 12)
   565)     integer :: day                 ! current day (1 -> 31)
   566)     integer :: mcsec               ! seconds of current date
   567)     integer :: mcdate              ! current date
   568)     integer :: dimid               ! netCDF dimension id
   569)     integer :: numg                ! total number of gridcells across all processors
   570)     integer :: numl                ! total number of landunits across all processors
   571)     integer :: numc                ! total number of columns across all processors
   572)     integer :: nump                ! total number of pfts across all processors
   573)     integer :: ier                 ! error status
   574)     integer :: strlen_dimid        ! string dimension id
   575)     character(len=  8) :: curdate  ! current date
   576)     character(len=  8) :: curtime  ! current time
   577)     character(len=256) :: str
   578)     character(len= 32) :: subname='restFile_dimset' ! subroutine name
   579) !------------------------------------------------------------------------
   580) 
   581)     call get_proc_global(numg, numl, numc, nump)
   582) 
   583)     ! Define dimensions
   584)     
   585)     call ncd_defdim(ncid, 'gridcell', numg           , dimid)
   586)     call ncd_defdim(ncid, 'landunit', numl           , dimid)
   587)     call ncd_defdim(ncid, 'column'  , numc           , dimid)
   588)     call ncd_defdim(ncid, 'pft'     , nump           , dimid)
   589)     
   590)     call ncd_defdim(ncid, 'levgrnd' , nlevgrnd       , dimid)
   591)     call ncd_defdim(ncid, 'levlak'  , nlevlak        , dimid)
   592)     call ncd_defdim(ncid, 'levsno'  , nlevsno        , dimid)
   593)     call ncd_defdim(ncid, 'levsno1'  , nlevsno+1     , dimid)
   594)     call ncd_defdim(ncid, 'levtot'  , nlevsno+nlevgrnd, dimid)
   595)     call ncd_defdim(ncid, 'numrad'  , numrad         , dimid)
   596)     call ncd_defdim(ncid, 'string_length', 64        , dimid)
   597)        
   598)     ! Define global attributes
   599)     
   600)     call ncd_putatt(ncid, NCD_GLOBAL, 'Conventions', trim(conventions))
   601)     call getdatetime(curdate, curtime)
   602)     str = 'created on ' // curdate // ' ' // curtime
   603)     call ncd_putatt(ncid, NCD_GLOBAL, 'history' , trim(str))
   604)     call ncd_putatt(ncid, NCD_GLOBAL, 'username', trim(username))
   605)     call ncd_putatt(ncid, NCD_GLOBAL, 'host'    , trim(hostname))
   606)     call ncd_putatt(ncid, NCD_GLOBAL, 'version' , trim(version))
   607)     call ncd_putatt(ncid, NCD_GLOBAL, 'source'  , trim(source))
   608)     str = '$Id: restFileMod.F90 60694 2014-05-27 22:16:25Z erik $'
