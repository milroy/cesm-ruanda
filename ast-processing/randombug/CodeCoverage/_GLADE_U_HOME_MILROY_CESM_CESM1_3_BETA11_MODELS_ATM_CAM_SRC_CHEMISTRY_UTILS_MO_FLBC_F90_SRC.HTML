    66)   subroutine flbc_inti( flbc_file, flbc_list, flbc_timing_in, co2vmr, ch4vmr, n2ovmr, f11vmr, f12vmr )
    67)     !-----------------------------------------------------------------------
    68)     ! 	... initialize the fixed lower bndy cond
    69)     !-----------------------------------------------------------------------
    70) 
    71)     use mo_constants,  only : d2r, pi, rearth
    72)     use string_utils,  only : to_upper
    73)     use constituents,  only : cnst_get_ind
    74)     use cam_pio_utils, only : cam_pio_openfile
    75)     use pio,           only : pio_get_var,pio_inq_varid,pio_inq_dimid, pio_inq_dimlen
    76)     use pio,           only : file_desc_t, pio_closefile, pio_nowrite
    77) 
    78)     implicit none
    79) 
    80)     !-----------------------------------------------------------------------
    81)     ! 	... dummy arguments
    82)     !-----------------------------------------------------------------------
    83)     character(len=*), intent(in) :: flbc_file
    84)     character(len=*), intent(in) :: flbc_list(:)
    85)     type(time_ramp),  intent(in) :: flbc_timing_in
    86)     real(r8),         intent(in) :: co2vmr, ch4vmr, n2ovmr, f11vmr, f12vmr
    87) 
    88)     !-----------------------------------------------------------------------
    89)     ! 	... local variables
    90)     !-----------------------------------------------------------------------
    91)     integer :: astat
    92)     integer :: j, l, m, n                     ! Indices
    93)     integer :: t1, t2
    94)     type(file_desc_t) :: ncid
    95)     integer :: dimid
    96)     integer :: varid
    97)     integer :: yr, mon, day, wrk_date, wrk_sec
    98)     real(r8)    :: seq
    99)     real(r8)    :: wrk_time
   100)     character(len=16)  :: species
   101)     character(len=16)  :: spc_name
   102)     character(len=8)   :: time_type
   103)     integer :: ierr
   104) 
   105)     if ( len_trim( flbc_file ) == 0 ) return
   106) 
   107)     call get_curr_date( yr, mon, day, ncsec )
   108)     ncdate = yr*10000 + mon*100 + day
   109) 
   110)     !-----------------------------------------------------------------------
   111)     ! 	... check timing
   112)     !-----------------------------------------------------------------------
   113)     flbc_timing = flbc_timing_in
   114)     time_type = to_upper(flbc_timing%type)
   115)     flbc_timing%type = time_type
   116)     if( time_type /= 'SERIAL' .and. time_type /= 'CYCLICAL' &
   117)          .and. time_type /= 'FIXED' ) then
   118)        write(iulog,*) 'flbc_inti: time type ',trim(time_type),' is not SERIAL,CYCLICAL, or FIXED'
   119)        call endrun('flbc_inti: invalid time type ')
   120)     end if
   121) 
   122)     if ( (flbc_timing%cycle_yr>0) .and. (time_type/='CYCLICAL') ) then
   123)        call endrun('flbc_inti: cannot specify  flbc_cycle_yr if flbc_type is not CYCLICAL')
   124)     endif
   125)     if ( ((flbc_timing%fixed_ymd>0).or.(flbc_timing%fixed_tod>0)).and.(time_type/='FIXED') ) then
   126)        call endrun('flbc_inti: cannot specify  flbc_fixed_ymd or flbc_fixed_tod if flbc_type is not FIXED')
   127)     endif
   128) 
   129)     wrk_sec  = ncsec
   130)     if( time_type == 'SERIAL' ) then
   131)        wrk_date = ncdate 
   132)     else if( time_type == 'CYCLICAL' ) then
   133) 
   134)     	! If this is a leap-day, we have to avoid asking for a non-leap-year
   135)     	! on a cyclical dataset. When this happens, just use Feb 28 instead
   136)     	if (( mon .eq. 2 ) .and. ( day.eq.29 )) then
   137) 	   ncdate = yr*10000 + mon*100 + (day-1)
   138)            write(iulog,*)'WARNING: flbc_inti using Feb 28 instead of Feb 29 for cyclical dataset'
   139)         endif 	
   140)        wrk_date = flbc_timing%cycle_yr*10000 + mod(ncdate,10000)
   141)     else
   142)        wrk_date = flbc_timing%fixed_ymd
   143)        wrk_sec  = flbc_timing%fixed_tod
   144)     end if
   145)     wrk_time = flt_date( wrk_date, wrk_sec )
   146)     if (masterproc) write(iulog,*) 'flbc_inti: wrk_date,wrk_sec,wrk_time = ',wrk_date,wrk_sec,wrk_time
   147) 
   148)     !-----------------------------------------------------------------------
   149)     ! 	... species with fixed lbc ?
   150)     !-----------------------------------------------------------------------
   151)     has_flbc(:) = .false.
   152)     flbc_cnt = 0
   153)     
   154)     do m = 1,max_nflbc
   155) 
   156)        if ( len_trim(flbc_list(m))==0 ) exit
   157) 
   158)        flbc_cnt = flbc_cnt + 1
   159) 
   160)        call cnst_get_ind (flbc_list(m), n, abort=.false.)
   161) 
   162)        if (n > 0) then
   163)           has_flbc(n) = .true.
   164)           flbcs(flbc_cnt)%spc_ndx = n
   165)        else ! must be one of the GHGs which is not prognosted
   166)           if( .not. any( ghg_names(:) == flbc_list(m) ) ) then
   167)              call endrun('flbc_inti: flbc_list member '// trim(flbc_list(m)) //' is not allowed')
   168)           endif
   169)           flbcs(flbc_cnt)%spc_ndx = -1
   170)        endif
   171) 
   172)        flbcs(flbc_cnt)%species = trim( flbc_list(m) )
   173) 
   174)        where( ghg_names(:) == flbc_list(m) )
   175)           ghg_indices = m
   176)        endwhere
   177) 
   178)        if( trim(flbcs(flbc_cnt)%species) == 'CFC11' ) then
   179)           flbcs(flbc_cnt)%species = 'CFCL3'
   180)        elseif( trim(flbcs(flbc_cnt)%species) == 'CFC12' ) then
   181)           flbcs(flbc_cnt)%species = 'CF2CL2'
   182)        endif
   183) 
   184)     enddo
   185) 
   186)     ! check that user has not set vmr namelist values... 
   187)     if ( ghg_indices(co2_ndx) > 0 .and. co2vmr>1.e-6_r8) then
   188)        call endrun('flbc_inti: cannot specify both co2vmr and CO2 in flbc_file')
   189)     endif
   190)     if ( ghg_indices(ch4_ndx) > 0 .and. ch4vmr > 0._r8) then
   191)        call endrun('flbc_inti: cannot specify both ch4vmr and CH4 in flbc_file')
   192)     endif
   193)     if ( ghg_indices(n2o_ndx) > 0 .and. n2ovmr > 0._r8) then
   194)        call endrun('flbc_inti: cannot specify both n2ovmr and N2O in flbc_file')
   195)     endif
   196)     if ( ghg_indices(f11_ndx) > 0 .and. f11vmr > 0._r8) then
   197)        call endrun('flbc_inti: cannot specify both f11vmr and CFC11 in flbc_file')
   198)     endif
   199)     if ( ghg_indices(f12_ndx) > 0 .and. f12vmr > 0._r8) then
   200)        call endrun('flbc_inti: cannot specify both f12vmr and CFC12 in flbc_file')
   201)     endif
   202)     
   203)     if( flbc_cnt == 0 ) then
   204)        return
   205)     end if
   206) 
   207)     if(masterproc) then
   208)        write(iulog,*) ' '
   209)        if( flbc_cnt > 0 ) then
   210)           write(iulog,*) 'flbc_inti: Species with specified lower boundary values'
   211)           do n = 1,flbc_cnt
   212)              write(iulog,*) trim(flbcs(n)%species)
   213)           enddo
   214)        else
   215)           write(iulog,*) 'There are no species with specified lower boundary values'
   216)        end if
   217)        write(iulog,*) ' '
   218) 
   219)        !-----------------------------------------------------------------------
   220)        ! 	... diagnostics
   221)        !-----------------------------------------------------------------------
   222)        write(iulog,*) ' '
   223)        write(iulog,*) 'flbc_inti: diagnostics'
   224)        write(iulog,*) ' '
   225)        write(iulog,*) 'lower bndy timing specs'
   226)        write(iulog,*) 'type = ',flbc_timing%type
   227)        if( time_type == 'CYCLICAL' ) then
   228)           write(iulog,*) 'cycle year = ',flbc_timing%cycle_yr
   229)        else
   230)           write(iulog,*) 'fixed date = ',flbc_timing%fixed_ymd
   231)           write(iulog,*) 'fixed time = ',flbc_timing%fixed_tod
   232)        end if
   233)        write(iulog,*) ' '
   234)        write(iulog,*) 'there are ',flbc_cnt,' species with specified lower bndy values'
   235)        write(iulog,*) ' '
   236)     end if
   237)     !-----------------------------------------------------------------------
   238)     ! 	... get timing information, allocate arrays, and read in dates
   239)     !-----------------------------------------------------------------------
   240)     call getfil ( flbc_file, filename, 0)
   241)     call cam_pio_openfile (ncid, trim(filename), PIO_NOWRITE)
   242)     ierr = pio_inq_dimid( ncid, 'time', dimid )
   243)     ierr = pio_inq_dimlen( ncid, dimid, ntimes )
   244) 
   245)     allocate( dates(ntimes),stat=astat )
   246)     if( astat/= 0 ) then
   247)        write(iulog,*) 'flbc_inti: failed to allocate dates array; error = ',astat
   248)        call endrun
   249)     end if
   250)     allocate( times(ntimes),stat=astat )
   251)     if( astat/= 0 ) then
   252)        write(iulog,*) 'flbc_inti: failed to allocate times array; error = ',astat
   253)        call endrun
   254)     end if
   255) 
   256)     ierr = pio_inq_varid( ncid, 'date', varid )
   257)     ierr = pio_get_var( ncid, varid, dates )
   258) 
   259)     do n = 1,ntimes
   260)        times(n) = flt_date( dates(n), 0 )
   261)     end do
   262)     if( time_type /= 'CYCLICAL' ) then
   263)        if( wrk_time < times(1) .or. wrk_time > times(ntimes) ) then
   264)           write(iulog,*) 'flbc_inti: time out of bounds for dataset = ',trim(filename)
   265)           call endrun
   266)        end if
   267)        do n = 2,ntimes
   268)           if( wrk_time <= times(n) ) then
   269)              exit
   270)           end if
   271)        end do
   272)        tim_ndx(1) = n - 1
   273)     else
   274)        yr = flbc_timing%cycle_yr
   275)        do n = 1,ntimes
   276)           if( yr == dates(n)/10000 ) then
   277)              exit
   278)           end if
   279)        end do
   280)        if( n >= ntimes ) then
   281)           write(iulog,*) 'flbc_inti: time out of bounds for dataset = ',trim(filename)
   282)           call endrun
   283)        end if
   284)        tim_ndx(1) = n
   285)     end if
   286)     select case( time_type )
   287)     case( 'FIXED' )
   288)        tim_ndx(2) = n
   289)     case( 'CYCLICAL' )
   290)        do n = tim_ndx(1),ntimes
   291)           if( yr /= dates(n)/10000 ) then
   292)              exit
   293)           end if
   294)        end do
   295)        tim_ndx(2) = n - 1
   296)        if( (tim_ndx(2) - tim_ndx(1)) < 2 ) then
   297)           write(iulog,*) 'flbc_inti: cyclical lb conds require at least two time points'
   298)           call endrun
   299)        end if
   300)     case( 'SERIAL' )
   301)        tim_ndx(2) = min( ntimes,tim_ndx(1) + time_span )
   302)     end select
   303)     t1 = tim_ndx(1)
   304)     t2 = tim_ndx(2)
   305) 
   306)     if( masterproc .and. debug ) then
   307)        write(iulog,*) ' '
   308)        write(iulog,*) 'flbc time cnt = ',ntimes
   309)        write(iulog,*) 'flbc times'
   310)        write(iulog,'(10i10)') dates(:)
   311)        write(iulog,'(1p,5g15.7)') times(:)
   312)        write(iulog,*) 'flbc time indicies = ',tim_ndx(:)
   313)        write(iulog,'(10i10)') dates(tim_ndx(1):tim_ndx(2))
   314)        write(iulog,*) ' '
   315)     endif
   316) 
   317)     do m = 1,flbc_cnt
   318)        !-----------------------------------------------------------------------
   319)        ! 	... allocate array
   320)        !-----------------------------------------------------------------------
   321)        allocate( flbcs(m)%vmr(pcols,begchunk:endchunk,t1:t2),stat=astat )
   322)        if( astat/= 0 ) then
   323)           write(iulog,*) 'flbc_inti: failed to allocate lbc vmr; error = ',astat
   324)           call endrun
   325)        end if
   326)        flbcs(m)%has_mean = file_has_gmean(ncid,flbcs(m)%species)
   327)        if ( flbcs(m)%has_mean) then
   328)           allocate( flbcs(m)%vmr_mean(t1:t2),stat=astat )
   329)           if( astat/= 0 ) then
   330)              write(iulog,*) 'flbc_inti: failed to allocate lbc vmr_mean; error = ',astat
   331)              call endrun
   332)           end if
   333)        endif
   334)        !-----------------------------------------------------------------------
   335)        ! 	... readin the flbc vmr
   336)        !-----------------------------------------------------------------------
   337)        call flbc_get( ncid, flbcs(m), .true., read_gmean=flbcs(m)%has_mean )
   338)     end do
   339) 
   340)     !-----------------------------------------------------------------------
   341)     ! 	... close the file
   342)     !-----------------------------------------------------------------------
   343)     call pio_closefile( ncid )
   344) 
   345)   end subroutine flbc_inti

   347)   subroutine flbc_chk( )
   348)     use cam_pio_utils, only : cam_pio_openfile
   349)     use pio,           only : file_desc_t, pio_closefile, pio_nowrite
   350)     !-----------------------------------------------------------------------
   351)     !       ... check serial case for time span
   352)     !-----------------------------------------------------------------------
   353) 
   354)     implicit none
   355) 
   356)     !-----------------------------------------------------------------------
   357)     !       ... dummy arguments
   358)     !-----------------------------------------------------------------------
   359) 
   360)     !-----------------------------------------------------------------------
   361)     !       ... local variables
   362)     !-----------------------------------------------------------------------
   363)     integer                     :: m
   364)     integer                     :: t1, t2, tcnt
   365)     integer                     :: astat
   366)     type(file_desc_t)           :: ncid
   367)     real(r8)                        :: wrk_time
   368)     integer ::  yr, mon, day
   369) 
   370)     call get_curr_date( yr, mon, day, ncsec )
   371)     ncdate = yr*10000 + mon*100 + day
   372) 
   373)     if( flbc_cnt > 0 .and. flbc_timing%type == 'SERIAL' ) then
   374)        wrk_time = flt_date( ncdate, ncsec )
   375)        if( wrk_time > times(tim_ndx(2)) ) then
   376)           tcnt = tim_ndx(2) - tim_ndx(1)
   377)           tim_ndx(1) = tim_ndx(2)
   378)           tim_ndx(2) = min( ntimes,tim_ndx(1) + time_span )
   379)           t1 = tim_ndx(1)
   380)           t2 = tim_ndx(2)
   381) !!$          if( tcnt /= (t2 - t1) ) then
   382)           !-----------------------------------------------------------------------
   383)           ! 	... allocate array
   384)           !-----------------------------------------------------------------------
   385)           do m = 1,flbc_cnt
   386)              if( associated( flbcs(m)%vmr ) ) then
   387)                 deallocate( flbcs(m)%vmr,stat=astat )
   388)                 if( astat/= 0 ) then
   389)                    write(iulog,*) 'flbc_chk: failed to deallocate flbc vmr; error = ',astat
   390)                    call endrun
   391)                 end if
   392)              end if
   393)              allocate( flbcs(m)%vmr(pcols,begchunk:endchunk,t1:t2),stat=astat )
   394)              if( astat/= 0 ) then
   395)                 write(iulog,*) 'flbc_chk: failed to allocate flbc vmr; error = ',astat
   396)                 call endrun
   397)              end if
   398)                 
   399)              if (flbcs(m)%has_mean) then
   400)                 if( associated( flbcs(m)%vmr_mean ) ) then
   401)                    deallocate( flbcs(m)%vmr_mean,stat=astat )
   402)                    if( astat/= 0 ) then
   403)                       write(iulog,*) 'flbc_chk: failed to deallocate flbc vmr; error = ',astat
   404)                       call endrun
   405)                    end if
   406)                 end if
   407)                 allocate( flbcs(m)%vmr_mean(t1:t2),stat=astat )
   408)                 if( astat/= 0 ) then
   409)                    write(iulog,*) 'flbc_chk: failed to allocate flbc vmr; error = ',astat
   410)                    call endrun
   411)                 end if
   412) 
   413)              endif
   414)           end do
   415) !!$          end if
   416) 
   417)           call cam_pio_openfile (ncid, trim(filename), PIO_NOWRITE)
   418)           !-----------------------------------------------------------------------
   419)           ! 	... readin the lb concentrations
   420)           !-----------------------------------------------------------------------
   421)           do m = 1,flbc_cnt
   422)              call flbc_get( ncid, flbcs(m), .true., read_gmean=flbcs(m)%has_mean )
   423)           end do
   424) 
   425)           !-----------------------------------------------------------------------
   426)           ! 	... close the file
   427)           !-----------------------------------------------------------------------
   428)           call pio_closefile( ncid )
   429) 
   430)        end if
   431)     end if
   432) 
   433)   end subroutine flbc_chk

   435)   ! checks for global mean in input file
   436)   function file_has_gmean(ncid,species)
   437)     use pio, only : file_desc_t, pio_inq_varid, pio_noerr, pio_seterrorhandling, &
   438)          pio_bcast_error, pio_internal_error
   439)     implicit none
   440) 
   441)     type(file_desc_t),      intent(inout) :: ncid
   442)     character(*), intent(in) :: species
   443)     logical :: file_has_gmean
   444) 
   445)     integer :: varid, ierr
   446) 
   447)     ! Allow pio to return the potential error and handle it locally
   448)     call pio_seterrorhandling(ncid, PIO_BCAST_ERROR)
   449)     ierr = pio_inq_varid( ncid, trim(species)//'_LBC_mean', varid)
   450)     call pio_seterrorhandling(ncid, PIO_INTERNAL_ERROR)
   451) 
   452) 
   453)     file_has_gmean = (ierr==PIO_NOERR)
   454) 
   455)   endfunction file_has_gmean

   457)   subroutine flbc_get( ncid, lbcs, initial, read_gmean )
   458)     !-----------------------------------------------------------------------
   459)     !       ... read lower bndy values
   460)     !-----------------------------------------------------------------------
   461)     use mo_constants,  only : d2r, pi
   462)     use phys_grid,     only: get_ncols_p, get_rlat_all_p, get_rlon_all_p
   463)     use pio,           only: file_desc_t, pio_get_var, pio_inq_varndims, &
   464)          pio_max_name, pio_inq_varid, pio_inq_dimlen, pio_inq_dimid
   465)     use interpolate_data, only : interp_type, lininterp_init, lininterp_finish, lininterp
   466) 
   467)     implicit none
   468) 
   469)     !-----------------------------------------------------------------------
   470)     !       ... dummy arguments
   471)     !-----------------------------------------------------------------------
   472)     type(file_desc_t), intent(in)           :: ncid
   473)     logical, intent(in)           :: initial
   474)     type(flbc), intent(inout) :: lbcs
   475) 
   476)     logical, intent(in), optional :: read_gmean
   477) 
   478)     !-----------------------------------------------------------------------
   479)     !       ... local variables
   480)     !-----------------------------------------------------------------------
   481)     integer                     :: j, m               ! Indices
   482)     integer                     :: t1, t2, tcnt
   483)     integer                     :: ierr
   484)     integer                     :: vid, nlat, nlon
   485)     integer                     :: dimid_lat, dimid_lon
   486)     integer                     :: plon, plat
   487)     real(r8), allocatable           :: lat(:)
   488)     real(r8), allocatable           :: lon(:)
   489)     real(r8), allocatable           :: wrk(:,:,:), wrk_zonal(:,:)
   490)     real(r8), allocatable           :: wrk2d(:,:)
   491)     character(len=pio_max_name)  :: varname
   492)     real(r8), allocatable       :: locl_vmr(:,:,:)
   493)     integer :: ndims, t, c, ncols
   494)     type(interp_type) :: lon_wgts, lat_wgts
   495)     real(r8) :: to_lats(pcols), to_lons(pcols)
   496)     real(r8), parameter :: twopi=2._r8*pi, zero=0._r8
   497) 
   498)     t1 = tim_ndx(1)
   499)     t2 = tim_ndx(2)
   500)     tcnt = t2 - t1 + 1
   501)     allocate( locl_vmr(pcols,begchunk:endchunk,tcnt), stat=ierr )

   663)   subroutine get_dels( dels, last, next )
   664) 
   665)     implicit none
   666) 
   667)     real(r8), intent(out) :: dels
   668)     integer,  intent(out) :: last
   669)     integer,  intent(out) :: next
   670) 
   671)     !--------------------------------------------------------

   677) 
   678)     !--------------------------------------------------------
   679)     !	... setup the time interpolation
   680)     !--------------------------------------------------------

   683)     case( 'SERIAL' )

   685)     case( 'CYCLICAL' )

   699)           if( wrk_time <= times(n) ) then

   701)              next = n
   702)              exit
   703)           end if
   704)        end do

   706)           write(iulog,*) 'flbc_set: interp time is out of bounds'
   707)           call endrun
   708)        end if
   709)        dels = (wrk_time - times(last))/(times(next) - times(last))
