   233)     type(mct_gGrid), pointer :: dom_o

   236)     integer                  :: ier
   237)     integer                  :: mytask

   239)     character(len=128)       :: ConfigFileName  ! config file to read
   240)     character(len=128)       :: MapLabel        ! map name
   241)     character(len=128)       :: MapTypeLabel    ! map type
   242)     character(len=256)       :: fileName
   243)     character(len=1)         :: maptype
   244)     character(len=3)         :: Smaptype
   245)     type(mct_aVect)          :: avdom_oe
   246)     type(mct_list)           :: sort_keys
   247)     character(*),parameter :: subName =   '(seq_flux_initexch_mct) '
   248)     !-----------------------------------------------------------------------
   249) 
   250)     gsmap_a => component_get_gsmap_cx(atm) ! gsmap_ax
   251)     gsmap_o => component_get_gsmap_cx(ocn) ! gsmap_ox

   253)     dom_o   => component_get_dom_cx(ocn)   ! dom_ox
   254)

   261)           MapLabel = "atm2ocn_fmapname:"
   262)           MapTypeLabel = "atm2ocn_fmaptype:"
   263)        elseif (n == 2) then
   264)           MapLabel = "ocn2atm_fmapname:"
   265)           MapTypeLabel = "ocn2atm_fmaptype:"
   266)        else
   267)           call shr_sys_abort(trim(subname)//' do error1')
   268)        endif
   269) 
   270)        call shr_mct_queryConfigFile(mpicom_cplid, ConfigFilename, &
   271)           trim(MapLabel),fileName,trim(MapTypeLabel),maptype)
   272) 
   273)        !--- hardwire decomposition to gsmap_o
   274)        if (n == 1) then
   275)           Smaptype = "src"
   276)           call shr_mct_sMatReaddnc(sMata2o, gsmap_a, gsmap_o, Smaptype, &

   278)        elseif (n == 2) then
   279)           Smaptype = "dst"
   280)           call shr_mct_sMatReaddnc(sMato2a, gsmap_o, gsmap_a, Smaptype, &
   281)              filename=fileName, mytask=mytask, mpicom=mpicom_cplid)
   282)        else
   283)           call shr_sys_abort(trim(subname)//' do error2')
   284)        endif
   285) 
   286)     enddo
   287) 
   288)     !--- the two mapping files must have their local indices in identical order
   289)     !--- sort the global indices as a starting point
   290) 
   291)     call mct_list_init(sort_keys,'grow:gcol')

   293)     call mct_list_clean(sort_keys)
   294)     call mct_list_init(sort_keys,'gcol:grow')
   295)     call mct_sMat_SortPermute(sMato2a,sort_keys)
   296)     call mct_list_clean(sort_keys)
   297) 
   298)     !--- now check that they are sorted properly
   299) 
   300)     nloc_a2o= mct_sMat_lsize(sMata2o)
   301)     nloc_o2a= mct_sMat_lsize(sMato2a)
   302) 
   303)     if (nloc_a2o /= nloc_o2a) then
   304)        write(logunit,*) trim(subname),' ERROR: sMat sizes',nloc_a2o,nloc_o2a
   305)        call shr_sys_abort(trim(subname)//' ERROR in sMat sizes')
   306)     endif
   307)     ko = mct_sMat_indexIA(sMata2o,'grow')    ! local row (dst) index
   308)     ka = mct_sMat_indexIA(sMato2a,'gcol')    ! local column (src) index
   309)     do n = 1,nloc_a2o
   310)        io = sMata2o%data%iAttr(ko,n)
   311)        ia = sMato2a%data%iAttr(ka,n)
   312)        if (io /= ia) then
   313)           write(logunit,*) trim(subname),' ERROR: sMat indices1 ',io,ia
   314)           call shr_sys_abort(trim(subname)//' ERROR in sMat indices1')
   315)        endif
   316)     enddo
   317)     ko = mct_sMat_indexIA(sMata2o,'gcol')    ! local column (src) index
   318)     ka = mct_sMat_indexIA(sMato2a,'grow')    ! local row (dst) index
   319)     do n = 1,nloc_a2o
   320)        io = sMata2o%data%iAttr(ko,n)
   321)        ia = sMato2a%data%iAttr(ka,n)
   322)        if (io /= ia) then
   323)           write(logunit,*) trim(subname),' ERROR: sMat indices2 ',io,ia
   324)           call shr_sys_abort(trim(subname)//' ERROR in sMat indices2')
   325)        endif
   326)     enddo
   327) 
   328)     !--- instantiate/create/compute various datatypes
   329) 
   330)     call mct_sMat_2XgsMap(sMata2o , gsmap_ae, 0, mpicom_cplid, cplid)
   331)     call mct_sMat_2YgsMap(sMata2o , gsmap_oe, 0, mpicom_cplid, cplid)
   332) 
   333)     call mct_rearr_init(gsmap_a   , gsmap_ae, mpicom_cplid, Re_a2e)
   334)     call mct_rearr_init(gsmap_ae  , gsmap_a,  mpicom_cplid, Re_e2a)
   335)     call mct_rearr_init(gsmap_o   , gsmap_oe, mpicom_cplid, Re_o2e)
   336)     call mct_rearr_init(gsmap_oe  , gsmap_o,  mpicom_cplid, Re_e2o)
   337) 
   338)     call mct_sMat_g2lMat(sMata2o  , gsmap_ae, 'column',mpicom_cplid)
   339)     call mct_sMat_g2lMat(sMata2o  , gsmap_oe, 'row',   mpicom_cplid)
   340)     call mct_sMat_g2lMat(sMato2a  , gsmap_ae, 'row',   mpicom_cplid)
   341)     call mct_sMat_g2lMat(sMato2a  , gsmap_oe, 'column',mpicom_cplid)
   342) 
   343)     nloc_a  = mct_gsmap_lsize(gsmap_a  , mpicom_cplid)
   344)     nloc_o  = mct_gsmap_lsize(gsmap_o  , mpicom_cplid)
   345)     nloc_ae = mct_gsmap_lsize(gsmap_ae , mpicom_cplid)
   346)     nloc_oe = mct_gsmap_lsize(gsmap_oe , mpicom_cplid)
   347) 
   348)     call mct_gsmap_clean(gsmap_ae)
   349)     call mct_gsmap_clean(gsmap_oe)
   350) 
   351)     ! Input fields atm
   352)     allocate( emask(nloc_a2o),stat=ier)
   353)     if(ier/=0) call mct_die(subName,'allocate emask',ier)
   354)     allocate( zbot(nloc_a2o),stat=ier)
   355)     if(ier/=0) call mct_die(subName,'allocate zbot',ier)
   356)     allocate( ubot(nloc_a2o),stat=ier)
   357)     if(ier/=0) call mct_die(subName,'allocate ubot',ier)
   358)     allocate( vbot(nloc_a2o))
   359)     if(ier/=0) call mct_die(subName,'allocate vbot',ier)
   360)     allocate(thbot(nloc_a2o),stat=ier)
   361)     if(ier/=0) call mct_die(subName,'allocate thbot',ier)
   362)     allocate(shum(nloc_a2o),stat=ier)
   363)     if(ier/=0) call mct_die(subName,'allocate shum',ier)
   364)     allocate(dens(nloc_a2o),stat=ier)
   365)     if(ier/=0) call mct_die(subName,'allocate dens',ier)
   366)     allocate(tbot(nloc_a2o),stat=ier)
   367)     if(ier/=0) call mct_die(subName,'allocate tbot',ier)
   368)     allocate(ustar(nloc_a2o),stat=ier)
   369)     if(ier/=0) call mct_die(subName,'allocate ustar',ier)
   370)     allocate(re(nloc_a2o), stat=ier)
   371)     if(ier/=0) call mct_die(subName,'allocate re',ier)
   372)     allocate(ssq(nloc_a2o),stat=ier)
   373)     if(ier/=0) call mct_die(subName,'allocate ssq',ier)
   374)     allocate( uocn(nloc_a2o),stat=ier)
   375)     if(ier/=0) call mct_die(subName,'allocate uocn',ier)
   376)     allocate( vocn(nloc_a2o),stat=ier)
   377)     if(ier/=0) call mct_die(subName,'allocate vocn',ier)
   378)     allocate( tocn(nloc_a2o),stat=ier)
   379)     if(ier/=0) call mct_die(subName,'allocate tocn',ier)
   380) 
   381)     ! Output fields 
   382)     allocate(sen (nloc_a2o),stat=ier)
   383)     if(ier/=0) call mct_die(subName,'allocate sen',ier)
   384)     allocate(lat (nloc_a2o),stat=ier)
   385)     if(ier/=0) call mct_die(subName,'allocate lat',ier)
   386)     allocate(evap(nloc_a2o),stat=ier)
   387)     if(ier/=0) call mct_die(subName,'allocate evap',ier)
   388)     allocate(lwup(nloc_a2o),stat=ier)
   389)     if(ier/=0) call mct_die(subName,'allocate lwup',ier)
   390)     allocate(taux(nloc_a2o),stat=ier)
   391)     if(ier/=0) call mct_die(subName,'allocate taux',ier)
   392)     allocate(tauy(nloc_a2o),stat=ier)
   393)     if(ier/=0) call mct_die(subName,'allocate tauy',ier)
   394)     allocate(tref(nloc_a2o),stat=ier)
   395)     if(ier/=0) call mct_die(subName,'allocate tref',ier)
   396)     allocate(qref(nloc_a2o),stat=ier)
   397)     if(ier/=0) call mct_die(subName,'allocate qref',ier)
   398)     allocate(duu10n(nloc_a2o),stat=ier)
   399)     if(ier/=0) call mct_die(subName,'allocate duu10n',ier)
   400) 
   401)     ! set emask
   402) 
   403)     call mct_avect_init(avdom_oe,dom_o%data,lsize=nloc_oe)
   404)     call mct_rearr_rearrange(dom_o%data, avdom_oe, Re_o2e, VECTOR=mct_usevector, ALLTOALL=mct_usealltoall)
   405)     ko = mct_sMat_indexIA(sMata2o,'lrow')    ! local dst index
   406)     kmsk = mct_aVect_indexRA(avdom_oe,"mask",dieWith=subName)

   414)     enddo
