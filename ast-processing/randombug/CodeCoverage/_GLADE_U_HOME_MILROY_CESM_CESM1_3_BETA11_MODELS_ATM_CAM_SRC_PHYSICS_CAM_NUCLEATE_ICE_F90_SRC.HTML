    23) subroutine nucleati(  &
    24)    wbar, tair, relhum, cldn, qc,              &
    25)    nfice, rhoair, so4_num, dst_num, soot_num, &
    26)    nuci, onihf, oniimm, onidep, onimey) 
    27) 
    28)    !---------------------------------------------------------------
    29)    ! Purpose:
    30)    !  The parameterization of ice nucleation.
    31)    !
    32)    ! Method: The current method is based on Liu & Penner (2005)
    33)    !  It related the ice nucleation with the aerosol number, temperature and the
    34)    !  updraft velocity. It includes homogeneous freezing of sulfate, immersion
    35)    !  freezing of soot, and Meyers et al. (1992) deposition nucleation
    36)    !
    37)    ! Authors: Xiaohong Liu, 01/2005, modifications by A. Gettelman 2009-2010
    38)    !----------------------------------------------------------------
    39) 
    40)    ! Input Arguments
    41)    real(r8), intent(in) :: wbar        ! grid cell mean vertical velocity (m/s)
    42)    real(r8), intent(in) :: tair        ! temperature (K)
    43)    real(r8), intent(in) :: relhum      ! relative humidity with respective to liquid
    44)    real(r8), intent(in) :: cldn        ! new value of cloud fraction    (fraction)
    45)    real(r8), intent(in) :: qc          ! liquid water mixing ratio (kg/kg)
    46)    real(r8), intent(in) :: nfice       ! ice mass fraction
    47)    real(r8), intent(in) :: rhoair      ! air density (kg/m3)
    48)    real(r8), intent(in) :: so4_num     ! so4 aerosol number (#/cm^3)
    49)    real(r8), intent(in) :: dst_num     ! total dust aerosol number (#/cm^3)
    50)    real(r8), intent(in) :: soot_num    ! soot (hydrophilic) aerosol number (#/cm^3)
    51) 
    52)    ! Output Arguments
    53)    real(r8), intent(out) :: nuci       ! ice number nucleated (#/kg)
    54)    real(r8), intent(out) :: onihf      ! nucleated number from homogeneous freezing of so4
    55)    real(r8), intent(out) :: oniimm     ! nucleated number from immersion freezing
    56)    real(r8), intent(out) :: onidep     ! nucleated number from deposition nucleation
    57)    real(r8), intent(out) :: onimey     ! nucleated number from deposition nucleation  (meyers: mixed phase)
    58) 
    59)    ! Local workspace
    60)    real(r8) :: nihf                      ! nucleated number from homogeneous freezing of so4
    61)    real(r8) :: niimm                     ! nucleated number from immersion freezing
    62)    real(r8) :: nidep                     ! nucleated number from deposition nucleation
    63)    real(r8) :: nimey                     ! nucleated number from deposition nucleation (meyers)
    64)    real(r8) :: n1, ni                    ! nucleated number
    65)    real(r8) :: tc, A, B, C, regm, RHw    ! work variable
    66)    real(r8) :: esl, esi, deles           ! work variable
    67)    real(r8) :: subgrid
    68)    !-------------------------------------------------------------------------------
    69) 
    70)    ni = 0._r8
    71)    tc = tair - 273.15_r8
    72) 
    73)    ! initialize
    74)    niimm = 0._r8
    75)    nidep = 0._r8
    76)    nihf  = 0._r8
    77) 
    78)    if(so4_num.ge.1.0e-10_r8 .and. (soot_num+dst_num).ge.1.0e-10_r8 .and. cldn.gt.0._r8) then
    79) 
    80)       !-----------------------------
    81)       ! RHw parameterization for heterogeneous immersion nucleation
    82)       A = 0.0073_r8
    83)       B = 1.477_r8
    84)       C = 131.74_r8
    85)       RHw=(A*tc*tc+B*tc+C)*0.01_r8   ! RHi ~ 120-130%
    86) 
    87)       subgrid = 1.2_r8
    88) 
    89)       if((tc.le.-35.0_r8) .and. ((relhum*svp_water(tair)/svp_ice(tair)*subgrid).ge.1.2_r8)) then ! use higher RHi threshold
    90) 
    91)          A = -1.4938_r8 * log(soot_num+dst_num) + 12.884_r8
    92)          B = -10.41_r8  * log(soot_num+dst_num) - 67.69_r8
    93)          regm = A * log(wbar) + B
    94) 
    95)          if(tc.gt.regm) then    ! heterogeneous nucleation only
    96)             if(tc.lt.-40._r8 .and. wbar.gt.1._r8) then ! exclude T<-40 & W>1m/s from hetero. nucleation
    97)                call hf(tc,wbar,relhum,subgrid,so4_num,nihf)
    98)                niimm=0._r8
    99)                nidep=0._r8
   100)                n1=nihf
   101)             else
   102)                call hetero(tc,wbar,soot_num+dst_num,niimm,nidep)
   103)                nihf=0._r8
   104)                n1=niimm+nidep
   105)             endif
   106)          elseif (tc.lt.regm-5._r8) then ! homogeneous nucleation only
   107)             call hf(tc,wbar,relhum,subgrid,so4_num,nihf)
   108)             niimm=0._r8
   109)             nidep=0._r8
   110)             n1=nihf
   111)          else        ! transition between homogeneous and heterogeneous: interpolate in-between
   112)             if(tc.lt.-40._r8 .and. wbar.gt.1._r8) then ! exclude T<-40 & W>1m/s from hetero. nucleation
   113)                call hf(tc,wbar,relhum,subgrid,so4_num,nihf)
   114)                niimm=0._r8
   115)                nidep=0._r8
   116)                n1=nihf
   117)             else
   118) 
   119)                call hf(regm-5._r8,wbar,relhum,subgrid,so4_num,nihf)
   120)                call hetero(regm,wbar,soot_num+dst_num,niimm,nidep)
   121) 
   122)                if(nihf.le.(niimm+nidep)) then
   123)                   n1=nihf
   124)                else
   125)                   n1=(niimm+nidep)*((niimm+nidep)/nihf)**((tc-regm)/5._r8)
   126)                endif
   127)             endif
   128)          endif
   129) 
   130)          ni=n1
   131) 
   132)       endif
   133)    endif
   134) 
   135)    ! deposition/condensation nucleation in mixed clouds (-40<T<0C) (Meyers, 1992)
   136)    if(tc.lt.0._r8 .and. tc.gt.-37._r8 .and. qc.gt.1.e-12_r8) then
   137)       esl = svp_water(tair)     ! over water in mixed clouds
   138)       esi = svp_ice(tair)     ! over ice
   139)       deles = (esl - esi)
   140)       nimey=1.e-3_r8*exp(12.96_r8*deles/esi - 0.639_r8) 
   141)    else
   142)       nimey=0._r8
   143)    endif
   144) 
   145)    nuci=ni+nimey
   146)    if(nuci.gt.9999._r8.or.nuci.lt.0._r8) then
   147)       write(iulog, *) 'Warning: incorrect ice nucleation number (nuci reset =0)'
   148)       write(iulog, *) ni, tair, relhum, wbar, nihf, niimm, nidep,deles,esi,dst_num,so4_num
   149)       nuci=0._r8
   150)    endif
   151) 
   152)    nuci   = nuci*1.e+6_r8/rhoair    ! change unit from #/cm3 to #/kg
   153)    onimey = nimey*1.e+6_r8/rhoair
   154)    onidep = nidep*1.e+6_r8/rhoair
   155)    oniimm = niimm*1.e+6_r8/rhoair
   156)    onihf  = nihf*1.e+6_r8/rhoair
   157) 
   158) end subroutine nucleati

   162) subroutine hetero(T,ww,Ns,Nis,Nid)
   163) 
   164)     real(r8), intent(in)  :: T, ww, Ns
   165)     real(r8), intent(out) :: Nis, Nid
   166) 
   167)     real(r8) A11,A12,A21,A22,B11,B12,B21,B22
   168)     real(r8) A,B,C
   169) 
   170) !---------------------------------------------------------------------
   171) ! parameters
   172) 
   173)       A11 = 0.0263_r8
   174)       A12 = -0.0185_r8
   175)       A21 = 2.758_r8
   176)       A22 = 1.3221_r8
   177)       B11 = -0.008_r8
   178)       B12 = -0.0468_r8
   179)       B21 = -0.2667_r8
   180)       B22 = -1.4588_r8
   181) 
   182) !     ice from immersion nucleation (cm^-3)
   183) 
   184)       B = (A11+B11*log(Ns)) * log(ww) + (A12+B12*log(Ns))
   185)       C =  A21+B21*log(Ns)
   186) 
   187)       Nis = exp(A22) * Ns**B22 * exp(B*T) * ww**C
   188)       Nis = min(Nis,Ns)
   189) 
   190)       Nid = 0.0_r8    ! don't include deposition nucleation for cirrus clouds when T<-37C
   191) 
   192) end subroutine hetero

   196) subroutine hf(T,ww,RH,subgrid,Na,Ni)
   197) 
   198)       real(r8), intent(in)  :: T, ww, RH, subgrid, Na
   199)       real(r8), intent(out) :: Ni
   200) 
   201)       real(r8)    A1_fast,A21_fast,A22_fast,B1_fast,B21_fast,B22_fast
   202)       real(r8)    A2_fast,B2_fast
   203)       real(r8)    C1_fast,C2_fast,k1_fast,k2_fast
   204)       real(r8)    A1_slow,A2_slow,B1_slow,B2_slow,B3_slow
   205)       real(r8)    C1_slow,C2_slow,k1_slow,k2_slow
   206)       real(r8)    regm
   207)       real(r8)    A,B,C
   208)       real(r8)    RHw
   209) 
   210) !---------------------------------------------------------------------
   211) ! parameters
   212) 
   213)       A1_fast  =0.0231_r8
   214)       A21_fast =-1.6387_r8  !(T>-64 deg)
   215)       A22_fast =-6.045_r8   !(T<=-64 deg)
   216)       B1_fast  =-0.008_r8
   217)       B21_fast =-0.042_r8   !(T>-64 deg)
   218)       B22_fast =-0.112_r8   !(T<=-64 deg)
   219)       C1_fast  =0.0739_r8
   220)       C2_fast  =1.2372_r8
   221) 
   222)       A1_slow  =-0.3949_r8
   223)       A2_slow  =1.282_r8
   224)       B1_slow  =-0.0156_r8
   225)       B2_slow  =0.0111_r8
   226)       B3_slow  =0.0217_r8
   227)       C1_slow  =0.120_r8
   228)       C2_slow  =2.312_r8
   229) 
   230)       Ni = 0.0_r8
   231) 
   232) !----------------------------
   233) !RHw parameters
   234)       A = 6.0e-4_r8*log(ww)+6.6e-3_r8
   235)       B = 6.0e-2_r8*log(ww)+1.052_r8
   236)       C = 1.68_r8  *log(ww)+129.35_r8
   237)       RHw=(A*T*T+B*T+C)*0.01_r8
   238) 
   239)       if((T.le.-37.0_r8) .and. ((RH*subgrid).ge.RHw)) then
   240) 
   241)         regm = 6.07_r8*log(ww)-55.0_r8
   242) 
   243)         if(T.ge.regm) then    ! fast-growth regime
   244) 
   245)           if(T.gt.-64.0_r8) then
   246)             A2_fast=A21_fast
   247)             B2_fast=B21_fast
   248)           else
   249)             A2_fast=A22_fast
   250)             B2_fast=B22_fast
   251)           endif
   252) 
   253)           k1_fast = exp(A2_fast + B2_fast*T + C2_fast*log(ww))
   254)           k2_fast = A1_fast+B1_fast*T+C1_fast*log(ww)
   255) 
   256)           Ni = k1_fast*Na**(k2_fast)
   257)           Ni = min(Ni,Na)
   258) 
   259)         else       ! slow-growth regime
   260) 
   261)           k1_slow = exp(A2_slow + (B2_slow+B3_slow*log(ww))*T + C2_slow*log(ww))
   262)           k2_slow = A1_slow+B1_slow*T+C1_slow*log(ww)
   263) 
   264)           Ni = k1_slow*Na**(k2_slow)
   265)           Ni = min(Ni,Na)
   266) 
   267)         endif
   268) 
   269)       end if
   270) 
   271) end subroutine hf
