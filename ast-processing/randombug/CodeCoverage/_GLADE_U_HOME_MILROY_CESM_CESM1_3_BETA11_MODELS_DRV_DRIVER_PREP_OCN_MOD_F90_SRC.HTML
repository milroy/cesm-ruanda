   338)   subroutine prep_ocn_accum(timer)
   339)     !---------------------------------------------------------------
   340)     ! Description
   341)     ! Accumulate ocn inputs
   342)     ! Form partial sum of tavg ocn inputs (virtual "send" to ocn) 
   343)     ! NOTE: this is done AFTER the call to the merge in prep_ocn_mrg
   344)     !
   345)     ! Arguments
   346)     character(len=*)        , intent(in) :: timer
   347)     !
   348)     ! Local Variables
   349)     integer :: eoi
   350)     type(mct_avect) , pointer   :: x2o_ox
   351)     character(*)    , parameter :: subname = '(prep_ocn_accum)'
   352)     !---------------------------------------------------------------
   353) 
   354)     call t_drvstartf (trim(timer), barrier=mpicom_CPLID)
   355)     do eoi = 1,num_inst_ocn
   356)        x2o_ox => component_get_x2c_cx(ocn(eoi))
   357) 
   358)        if (x2oacc_ox_cnt == 0) then
   359)           call mct_avect_copy(x2o_ox, x2oacc_ox(eoi))
   360)        else
   361)           call mct_avect_accum(x2o_ox, x2oacc_ox(eoi))
   362)        endif
   363)     enddo
   364)     x2oacc_ox_cnt = x2oacc_ox_cnt + 1
   365)     call t_drvstopf  (trim(timer))
   366) 
   367)   end subroutine prep_ocn_accum

   372)     !---------------------------------------------------------------

   376)     ! Arguments

   393)        x2o_ox   => component_get_x2c_cx(ocn(eoi))

   395)     enddo

   403)   subroutine prep_ocn_mrg(infodata, fractions_ox, xao_ox, timer_mrg)
   404) 
   405)     !---------------------------------------------------------------
   406)     ! Description
   407)     ! Merge all ocn inputs
   408)     !
   409)     ! Arguments
   410)     type(seq_infodata_type) , intent(in)    :: infodata
   411)     type(mct_aVect)         , intent(in)    :: fractions_ox(:)
   412)     type(mct_aVect)         , intent(in)    :: xao_ox(:) ! Atm-ocn fluxes, ocn grid, cpl pes 
   413)     character(len=*)        , intent(in)    :: timer_mrg
   414)     !
   415)     ! Local Variables
   416)     integer                  :: eii, ewi, egi, eoi, eai, eri, exi, efi, emi
   417)     real(r8)                 :: flux_epbalfact ! adjusted precip factor
   418)     type(mct_avect), pointer :: x2o_ox
   419)     integer                  :: cnt
   420)     character(*), parameter  :: subname = '(prep_ocn_mrg)'
   421)     !---------------------------------------------------------------
   422) 
   423)     call seq_infodata_GetData(infodata, &
   424)          flux_epbalfact=flux_epbalfact)
   425) 
   426)     call t_drvstartf (trim(timer_mrg), barrier=mpicom_CPLID)
   427) 
   428)     ! Use emi here for instance averaging capability, num_inst_max = num_inst_ocn normally
   429)     ! if NOT x2o_average, just fill each instance of component_get_x2c_cx(ocn(eoi))
   430)     ! if     x2o_average, then computer merge into x2o_ox_inst and accumulate that to 
   431)     !                     component_get_x2c_cx(ocn(1)) and then average it at the end
   432) 
   433)     if (x2o_average) then
   434)        x2o_ox   => component_get_x2c_cx(ocn(1)) 
   435)        call mct_aVect_zero(x2o_ox)
   436)     endif
   437) 
   438)     cnt = 0
   439)     do emi = 1,num_inst_max
   440)        ! Use fortran mod to address ensembles in merge
   441)        eoi = mod((emi-1),num_inst_ocn) + 1
   442)        eai = mod((emi-1),num_inst_atm) + 1
   443)        eii = mod((emi-1),num_inst_ice) + 1
   444)        eri = mod((emi-1),num_inst_rof) + 1
   445)        ewi = mod((emi-1),num_inst_wav) + 1
   446)        egi = mod((emi-1),num_inst_glc) + 1
   447)        exi = mod((emi-1),num_inst_xao) + 1
   448)        efi = mod((emi-1),num_inst_frc) + 1
