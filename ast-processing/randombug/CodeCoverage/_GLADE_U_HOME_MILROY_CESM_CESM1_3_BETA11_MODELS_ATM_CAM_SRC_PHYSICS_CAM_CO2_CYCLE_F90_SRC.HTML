   109)    end if

   188) subroutine co2_init
   189) 
   190) !----------------------------------------------------------------------- 
   191) ! 
   192) ! Purpose: initialize co2,
   193) !          declare history variables,
   194) !          read co2 flux form ocn,  as data_flux_ocn
   195) !          read co2 flux form fule, as data_flux_fuel
   196) !
   197) !-----------------------------------------------------------------------
   198) 
   199)     use cam_history, only: addfld, add_default, phys_decomp
   200)  
   201)     integer :: m, mm
   202)       
   203)     if (.not. co2_flag) return
   204)  
   205)     ! Add constituents and fluxes to history file
   206)     do m = 1, ncnst
   207)        call cnst_get_ind(c_names(m), mm)
   208) 
   209)        call addfld(trim(cnst_name(mm))//'_BOT', 'kg/kg',     1, 'A', trim(cnst_longname(mm))//', Bottom Layer', phys_decomp)
   210)        call addfld(cnst_name(mm),               'kg/kg',  pver, 'A', cnst_longname(mm), phys_decomp)
   211)        call addfld(sflxnam(mm),                 'kg/m2/s',   1, 'A', trim(cnst_name(mm))//' surface flux', phys_decomp)
   212) 
   213)        call add_default(cnst_name(mm), 1, ' ')
   214)        call add_default(sflxnam(mm),   1, ' ')
   215) 
   216)        ! The addfld call for the 'TM*' fields are made by default in the 
   217)        ! constituent_burden module.
   218)        call add_default('TM'//trim(cnst_name(mm)), 1, ' ')
   219)     end do
   220)  
   221)     ! Read flux data
   222)     if (co2_readFlux_ocn) then
   223)        call read_data_flux ( co2flux_ocn_file,  data_flux_ocn )
   224)     end if
   225)  
   226)     if (co2_readFlux_fuel) then
   227)        call read_data_flux ( co2flux_fuel_file, data_flux_fuel )
   228)     end if
   229)  
   230)   end subroutine co2_init
