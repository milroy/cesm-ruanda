   182) subroutine initialize_clock( start_date, ref_date, curr_date, stop_date )
   183) !
   184) ! Purpose: Initialize the clock based on the start_date, ref_date, and curr_date
   185) ! as well as the settings from the namelist specifying the time to stop
   186) !
   187) ! Input variables
   188)    type(ESMF_Time), intent(inout) :: start_date  ! start date for run
   189)    type(ESMF_Time), intent(inout) :: ref_date    ! reference date for time coordinate
   190)    type(ESMF_Time), intent(inout) :: curr_date   ! current date (equal to start_date)
   191)    type(ESMF_Time), intent(inout) :: stop_date   ! stop date for run
   192) 
   193) ! Local variables
   194)    character(len=*), parameter :: sub = 'initialize_clock'
   195)    type(ESMF_TimeInterval) :: step_size       ! timestep size
   196)    type(ESMF_Time) :: current     ! current date (from clock)
   197)    integer :: yr, mon, day, tod   ! Year, month, day, and second as integers
   198)    integer :: rc                  ! return code
   199) 
   200)    if ( mod(86400,dtime) /= 0 ) then
   201) !!!!      call endrun (sub//': timestep must divide evenly into 1 day')
   202)    end if
   203) 
   204)    call ESMF_TimeIntervalSet( step_size, s=dtime, rc=rc )
   205)    call chkrc(rc, sub//': error return from ESMF_TimeIntervalSet: setting step_size')
   206) 
   207)    if ( stop_date <= start_date ) then
   208)       write(iulog,*)sub, ': stop date must be specified later than start date: '
   209)       call ESMF_TimeGet( start_date, yy=yr, mm=mon, dd=day, s=tod )
   210)       write(iulog,*) ' Start date (yr, mon, day, tod): ', yr, mon, day, tod
   211)       call ESMF_TimeGet( stop_date, yy=yr, mm=mon, dd=day, s=tod )
   212)       write(iulog,*) ' Stop date  (yr, mon, day, tod): ', yr, mon, day, tod
   213)       call endrun
   214)    end if
   215)    if ( curr_date >= stop_date ) then
   216)       write(iulog,*)sub, ': stop date must be specified later than current date: '
   217)       call ESMF_TimeGet( curr_date, yy=yr, mm=mon, dd=day, s=tod )
   218)       write(iulog,*) ' Current date (yr, mon, day, tod): ', yr, mon, day, tod
   219)       call ESMF_TimeGet( stop_date, yy=yr, mm=mon, dd=day, s=tod )
   220)       write(iulog,*) ' Stop date    (yr, mon, day, tod): ', yr, mon, day, tod
   221)       call endrun
   222)    end if
   223) 
   224) ! Initialize the clock
   225) 
   226)    tm_clock = ESMF_ClockCreate(name="CAM Time-manager clock", timeStep=step_size, startTime=start_date, &
   227)                                stopTime=stop_date, refTime=ref_date, rc=rc)
   228)    call chkrc(rc, sub//': error return from ESMF_ClockSetup')
   229) 
   230) ! Advance clock to the current time (in case of a restart)
   231) 
   232)    call ESMF_ClockGet(tm_clock, currTime=current, rc=rc )
   233)    call chkrc(rc, sub//': error return from ESMF_ClockGet')
   234)    do while( curr_date > current )
   235)       call ESMF_ClockAdvance( tm_clock, rc=rc )
   236)       call chkrc(rc, sub//': error return from ESMF_ClockAdvance')
   237)       call ESMF_ClockGet(tm_clock, currTime=current )
   238)       call chkrc(rc, sub//': error return from ESMF_ClockGet')
   239)    end do
   240) end subroutine initialize_clock

   245) !
   246) ! Set the time by an integer as YYYYMMDD and integer seconds in the day
   247) !
   248)    integer, intent(in) :: ymd            ! Year, month, day YYYYMMDD
   249)    integer, intent(in) :: tod            ! Time of day in seconds
   250)    character(len=*), intent(in) :: desc  ! Description of time to set

   273) subroutine set_time_float_from_date( time, year, month, day, sec )
   274) !
   275) ! Set the time as a float given year, month, day, sec
   276) !
   277)   implicit none
   278) 
   279)   real(r8),intent(out):: time
   280)   integer, intent(in) :: year
   281)   integer, intent(in) :: month
   282)   integer, intent(in) :: day
   283)   integer, intent(in) :: sec
   284) 
   285)   integer :: rc                    ! return code
   286)   character(len=*), parameter :: sub = 'set_time_float_from_date'
   287)   type(ESMF_Time) :: date, ref_date
   288)   type(ESMF_TimeInterval) :: diff
   289)   integer :: useday
   290) 
   291)   call ESMF_TimeSet( date, yy=year, mm=month, dd=day, s=sec, calendar=tm_cal, rc=rc)
   292)   !
   293)   ! If the subroutine returned error, check if it is Feb 29 of a non-leap year
   294)   ! (legitimately used by the time-interpolation routines in tracer_data.F90)
   295)   ! in which case, substitute Feb 28 for the day
   296)   !
   297)   if ( rc .ne. ESMF_SUCCESS ) then
   298)      if ( ( month .eq. 2 ) .and. ( day .eq. 29 ) ) then ! assume the failure is because it is leap day
   299)         useday = 28
   300)         call ESMF_TimeSet( date, yy=year, mm=month, dd=useday, s=sec, calendar=tm_cal, rc=rc)
   301)      else  ! legitimate error, let the model quit
   302)         call chkrc(rc, sub//': error return from ESMF_TimeSet for set_time_float_from_date')        
   303)      endif 
   304)   endif  
   305) 
   306)   call ESMF_ClockGet(tm_clock, refTime=ref_date, rc=rc )
   307)   call chkrc(rc, sub//': error return from ESMF_ClockGet for set_time_float_from_date')
   308) 
   309)   diff = date - ref_date
   310) 
   311)   call ESMF_TimeIntervalGet( diff, d_r8=time, rc=rc)
   312)   call chkrc(rc, sub//': error return from ESMF_TimeIntervalGet for set_time_float_from_date')
   313) 
   314) endsubroutine set_time_float_from_date

   412) subroutine timemgr_restart( stop_ymd, stop_tod )
   413) 
   414) ! Restart the ESMF time manager.
   415) !
   416) ! NOTE - Assumptions:
   417) !      1) Restart data have been read on the master process before this routine is called.
   418) !         (timemgr_read_restart called from control/restart.F90::cam_read_restart)
   419) !      2) Stopping time has been set and input to this routine.
   420) ! Arguments
   421)    integer,          intent(IN) :: stop_ymd       ! Stop date (YYYYMMDD)
   422)    integer,          intent(IN) :: stop_tod       ! Stop time of day (sec)
   423) 
   424) ! Local variables
   425)    character(len=*), parameter :: sub = 'timemgr_restart'
   426)    integer :: rc                  ! return code
   427)    type(ESMF_Time) :: start_date  ! start date for run
   428)    type(ESMF_Time) :: stop_date   ! stop date for run
   429)    type(ESMF_Time) :: ref_date    ! reference date
   430)    type(ESMF_Time) :: curr_date   ! date of data in restart file
   431) !-----------------------------------------------------------------------------------------
   432) 
   433) #if ( defined SPMD ) 
   434)    call mpibcast(rst_calendar,  len(rst_calendar), mpichar, 0, mpicom)
   435)    call mpibcast(rst_step_sec,  1, mpiint, 0, mpicom)
   436)    call mpibcast(rst_start_ymd, 1, mpiint, 0, mpicom)
   437)    call mpibcast(rst_start_tod, 1, mpiint, 0, mpicom)
   438)    call mpibcast(rst_stop_ymd,  1, mpiint, 0, mpicom)
   439)    call mpibcast(rst_stop_tod,  1, mpiint, 0, mpicom)
   440)    call mpibcast(rst_ref_ymd,   1, mpiint, 0, mpicom)
   441)    call mpibcast(rst_ref_tod,   1, mpiint, 0, mpicom)
   442)    call mpibcast(rst_curr_ymd,  1, mpiint, 0, mpicom)
   443)    call mpibcast(rst_curr_tod,  1, mpiint, 0, mpicom)
   444)    call mpibcast(rst_perp_ymd,  1, mpiint, 0, mpicom)
   445)    call mpibcast(rst_perp_cal,  1, mpilog, 0, mpicom)
   446) #endif
   447) 
   448)   calendar = trim(rst_calendar)
   449) 
   450) ! Initialize calendar type.
   451) 
   452)    call init_calendar( )
   453) 
   454) ! Initialize the timestep.
   455) 
   456)    dtime = rst_step_sec
   457) 
   458) ! Initialize start date.
   459) 
   460)    start_date = TimeSetymd( rst_start_ymd, rst_start_tod, "start_date" )
   461) 
   462) ! Initialize stop date.
   463) 
   464)    stop_date = TimeSetymd( stop_ymd, stop_tod, "stop_date" )
   465) 
   466) ! Initialize current date.
   467) 
   468)    curr_date = TimeSetymd( rst_curr_ymd, rst_curr_tod, "curr_date" )
   469) 
   470) ! Initialize ref date.
   471) 
   472)    ref_date = TimeSetymd( rst_ref_ymd, rst_ref_tod, "ref_date" )
   473) 
   474) ! Initialize clock and the stop date
   475) 
   476) 
   477)    call initialize_clock( start_date, ref_date, curr_date, stop_date )
   478) 
   479) ! Advance the timestep.  Data from the restart file corresponds to the
   480) ! last timestep of the previous run.
   481) 
   482)    call advance_timestep()
   483)

   988) !!!!  call ESMF_TimePrint( date, "string" )
   989)    end if
