    98) subroutine CNNFixation(num_soilc, filter_soilc)
    99) !
   100) ! !DESCRIPTION:
   101) ! On the radiation time step, update the nitrogen fixation rate
   102) ! as a function of annual total NPP. This rate gets updated once per year.
   103) ! All N fixation goes to the soil mineral N pool.
   104) !
   105) ! !USES:
   106)    use clmtype
   107)    use clm_varctl      , only: iulog
   108)    use clm_time_manager, only: get_days_per_year
   109)    use shr_sys_mod     , only: shr_sys_flush
   110)    use clm_varcon      , only: secspday
   111) !
   112) ! !ARGUMENTS:
   113)    implicit none
   114)    integer, intent(in) :: num_soilc       ! number of soil columns in filter
   115)    integer, intent(in) :: filter_soilc(:) ! filter for soil columns
   116) !
   117) ! !CALLED FROM:
   118) ! subroutine CNEcosystemDyn, in module CNEcosystemDynMod.F90
   119) !
   120) ! !REVISION HISTORY:
   121) ! 6/1/04: Created by Peter Thornton
   122) ! 2/14/05, PET: After looking at a number of point simulations,
   123) !               it looks like a constant Nfix might be more efficient and 
   124) !               maybe more realistic - setting to constant 0.4 gN/m2/yr.
   125) !
   126) ! !LOCAL VARIABLES:
   127) ! local pointers to implicit in scalars
   128) !
   129)    real(r8), pointer :: cannsum_npp(:) ! nitrogen deposition rate (gN/m2/s)
   130) !
   131) ! local pointers to implicit out scalars
   132) !
   133)    real(r8), pointer :: nfix_to_sminn(:)
   134) !
   135) ! !OTHER LOCAL VARIABLES:
   136)    integer  :: c,fc                  ! indices
   137)    real(r8) :: t                     ! temporary
   138)    real(r8) :: dayspyr               ! days per year
   139) 
   140) !EOP
   141) !-----------------------------------------------------------------------
   142)    ! Assign local pointers to derived type arrays (in)
   143)    cannsum_npp   => cps%cannsum_npp
   144) 
   145)    ! Assign local pointers to derived type arrays (out)
   146)    nfix_to_sminn => cnf%nfix_to_sminn
   147) 
   148)    dayspyr = get_days_per_year()
   149) 
   150)    ! Loop through columns
   151)    do fc = 1,num_soilc
   152)       c = filter_soilc(fc)
   153) 
   154)       ! the value 0.001666 is set to give 100 TgN/yr when global
   155)       ! NPP = 60 PgC/yr.  (Cleveland et al., 1999)
   156)       ! Convert from gN/m2/yr -> gN/m2/s
   157)       !t = cannsum_npp(c) * 0.001666_r8 / (secspday * dayspyr)
   158)       t = (1.8_r8 * (1._r8 - exp(-0.003_r8 * cannsum_npp(c))))/(secspday * dayspyr)
   159)       nfix_to_sminn(c) = max(0._r8,t)
   160)       ! PET 2/14/05: commenting out the dependence on NPP, and
   161)       ! forcing Nfix to global constant = 0.4 gN/m2/yr
   162)       !nfix_to_sminn(c) = 0.4 / (secspday*dayspyr)
   163) 
   164)    end do

   171) ! !IROUTINE: CNNLeaching
   172) !
   173) ! !INTERFACE:
   174) subroutine CNNLeaching(lbc, ubc, num_soilc, filter_soilc)
   175) !
   176) ! !DESCRIPTION:
   177) ! On the radiation time step, update the nitrogen leaching rate
   178) ! as a function of soluble mineral N and total soil water outflow.
   179) !
   180) ! !USES:
   181)    use clmtype
   182)    use clm_varpar      , only : nlevsoi
   183)    use clm_time_manager    , only : get_step_size
   184) !
   185) ! !ARGUMENTS:
   186)    implicit none
   187)    integer, intent(in) :: lbc, ubc        ! column bounds
   188)    integer, intent(in) :: num_soilc       ! number of soil columns in filter
   189)    integer, intent(in) :: filter_soilc(:) ! filter for soil columns
   190) !
   191) ! !CALLED FROM:
   192) ! subroutine CNEcosystemDyn
   193) !
   194) ! !REVISION HISTORY:
   195) ! 6/9/04: Created by Peter Thornton
   196) !
   197) ! !LOCAL VARIABLES:
   198) ! local pointers to implicit in scalars
   199) !
   200)    real(r8), pointer :: h2osoi_liq(:,:)  ! liquid water (kg/m2) (new) (-nlevsno+1:nlevgrnd)
   201)    real(r8), pointer :: qflx_drain(:)    ! sub-surface runoff (mm H2O /s)
   202)    real(r8), pointer :: sminn(:)         ! (gN/m2) soil mineral N
   203) !
   204) ! local pointers to implicit out scalars
   205) !
   206)    real(r8), pointer :: sminn_leached(:) ! rate of mineral N leaching (gN/m2/s)
   207) !
   208) ! !OTHER LOCAL VARIABLES:
   209)    integer  :: j,c,fc             ! indices
   210)    real(r8) :: dt                 ! radiation time step (seconds)
   211)    real(r8) :: tot_water(lbc:ubc) ! total column liquid water (kg water/m2)
   212)    real(r8) :: sf                 ! soluble fraction of mineral N (unitless)
   213)    real(r8) :: disn_conc          ! dissolved mineral N concentration
   214)                                   ! (gN/kg water)
   215) 
   216) !EOP
   217) !-----------------------------------------------------------------------
   218)    ! Assign local pointers to derived type arrays (in)
   219)    h2osoi_liq    => cws%h2osoi_liq
   220)    qflx_drain    => cwf%qflx_drain
   221)    sminn         => cns%sminn
   222) 
   223)    ! Assign local pointers to derived type arrays (out)
   224)    sminn_leached => cnf%sminn_leached
   225) 
   226)    ! set time steps
   227)    dt = real( get_step_size(), r8 )
   228) 
   229)    ! Assume that 10% of the soil mineral N is in a soluble form
   230)    sf = 0.1_r8
   231) 
   232)    ! calculate the total soil water
   233)    tot_water(lbc:ubc) = 0._r8
   234)    do j = 1,nlevsoi
   235)       do fc = 1,num_soilc
   236)          c = filter_soilc(fc)
   237)          tot_water(c) = tot_water(c) + h2osoi_liq(c,j)
   238)       end do
   239)    end do
   240) 
   241)    ! Loop through columns
   242)    do fc = 1,num_soilc
   243)       c = filter_soilc(fc)
   244) 
   245)       ! calculate the dissolved mineral N concentration (gN/kg water)
   246)       ! assumes that 10% of mineral nitrogen is soluble
   247)       disn_conc = 0._r8
   248)       if (tot_water(c) > 0._r8) then
   249)          disn_conc = (sf * sminn(c))/tot_water(c)
   250)       end if
   251) 
   252)       ! calculate the N leaching flux as a function of the dissolved
   253)       ! concentration and the sub-surface drainage flux
   254)       sminn_leached(c) = disn_conc * qflx_drain(c)
   255) 
   256)       ! limit the flux based on current sminn state
   257)       ! only let at most the assumed soluble fraction
   258)       ! of sminn be leached on any given timestep
   259)       sminn_leached(c) = min(sminn_leached(c), (sf * sminn(c))/dt)
   260)       
   261)       ! limit the flux to a positive value
   262)       sminn_leached(c) = max(sminn_leached(c), 0._r8)
   263) 
   264)    end do
   265) 
   266) end subroutine CNNLeaching
   267)
