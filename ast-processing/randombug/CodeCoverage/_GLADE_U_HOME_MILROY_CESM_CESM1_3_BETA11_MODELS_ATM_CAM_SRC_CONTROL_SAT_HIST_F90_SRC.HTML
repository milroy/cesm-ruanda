    95)   subroutine sat_hist_readnl(nlfile, hfilename_spec, mfilt, fincl, nhtfrq, avgflag_pertape)
    96)     
    97)     use namelist_utils,      only: find_group_name
    98)     use units,               only: getunit, freeunit
    99)     use cam_history_support, only: pflds
   100)     use cam_instance,        only: inst_suffix
   101) 
   102)     implicit none
   103) 
   104)     character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
   105)     character(len=*), intent(inout) :: hfilename_spec(:)
   106)     character(len=*), intent(inout) :: fincl(:,:)
   107)     character(len=1), intent(inout) :: avgflag_pertape(:)
   108)     integer,          intent(inout) :: mfilt(:), nhtfrq(:)
   109)     
   110)     ! Local variables
   111)     integer :: unitn, ierr
   112)     character(len=*), parameter :: subname = 'sat_hist_readnl'
   113)     integer :: f, fcnt
   114) 
   115)     character(len=fieldname_lenp2) :: sathist_fincl(pflds)
   116)     character(len=max_string_len)  :: sathist_hfilename_spec
   117)     integer :: sathist_mfilt, sat_tape_num
   118) 
   119)     namelist /satellite_options_nl/ sathist_track_infile, sathist_hfilename_spec, sathist_fincl, &
   120)          sathist_mfilt, sathist_nclosest, sathist_ntimestep
   121) 
   122)     ! set defaults
   123) 
   124)     sathist_track_infile = ' '
   125)     sathist_hfilename_spec = '%c.cam' // trim(inst_suffix) // '.hs.%y-%m-%d-%s.nc'
   126)     sathist_fincl(:) = ' '
   127)     sathist_mfilt = 100000
   128)     sathist_nclosest = 1
   129)     sathist_ntimestep = 1
   130) 
   131)     !read namelist options
   132) 
   133)     if (masterproc) then
   134)        unitn = getunit()
   135)        open( unitn, file=trim(nlfile), status='old' )
   136)        call find_group_name(unitn, 'satellite_options_nl', status=ierr)
   137)        if (ierr == 0) then
   138)           read(unitn, satellite_options_nl, iostat=ierr)
   139)           if (ierr /= 0) then
   140)              call endrun(subname // ':: ERROR reading namelist')
   141)           end if
   142)        end if
   143)        close(unitn)
   144)        call freeunit(unitn)
   145)     end if
   146) 
   147) #ifdef SPMD
   148)     ! broadcast the options to all MPI tasks
   149)     call mpibcast(sathist_track_infile,   len(sathist_track_infile),   mpichar, 0, mpicom)
   150)     call mpibcast(sathist_hfilename_spec, len(sathist_hfilename_spec), mpichar, 0, mpicom)
   151)     call mpibcast(sathist_fincl,          pflds*len(sathist_fincl(1)), mpichar, 0, mpicom)
   152)     call mpibcast(sathist_mfilt,          1,                           mpiint,  0, mpicom)
   153)     call mpibcast(sathist_nclosest,       1,                           mpiint,  0, mpicom)
   154)     call mpibcast(sathist_ntimestep,      1,                           mpiint,  0, mpicom)
   155) #endif
   156) 
   157)     has_sat_hist = len_trim(sathist_track_infile) > 0
   158) 
   159)     if (.not.has_sat_hist) return
   160) 
   161)      sat_tape_num=ptapes-1
   162)      hfilename_spec(sat_tape_num) = sathist_hfilename_spec
   163)      mfilt(sat_tape_num) = sathist_mfilt
   164)      fcnt=0
   165)      do f=1, pflds
   166)         fincl(f,sat_tape_num) = sathist_fincl(f)
   167)         if(len_trim(sathist_fincl(f)) > 0) then
   168)            fcnt=fcnt+1
   169)         end if
   170)      enddo
   171)      
   172)      nhtfrq(sat_tape_num) = 1
   173)      avgflag_pertape(sat_tape_num) = 'I'
   174) 
   175)      if(masterproc) then
   176)         write(iulog,*) 'sathist_track_infile: ',trim(sathist_track_infile)
   177)         write(iulog,*) 'sathist_hfilename_spec: ',trim(sathist_hfilename_spec)
   178)         write(iulog,*) 'sathist_fincl: ',(trim(sathist_fincl(f))//' ', f=1,fcnt)
   179)         write(iulog,*) 'max columns per file sathist_mfilt: ',sathist_mfilt
   180)         write(iulog,*) 'sathist_nclosest: ',sathist_nclosest
   181)         write(iulog,*) 'sathist_ntimestep: ',sathist_ntimestep
   182)      end if
   183) 
   184)    end subroutine sat_hist_readnl

   189)   subroutine sat_hist_init
   190)     use cam_pio_utils, only: cam_pio_openfile
   191)     use ioFileMod,     only: getfil
   192)     use spmd_utils,    only: npes
   193)     use time_manager,  only: get_step_size
   194)     use string_utils,  only: to_lower, GLC
   195) 
   196)     implicit none
   197) 
   198)     character(len=max_string_len)  :: locfn       ! Local filename
   199)     integer :: ierr, dimid, i
   200) 
   201)     character(len=128) :: date_format
   202) 
   203)     if (.not.has_sat_hist) return
   204) 
   205)     call getfil (sathist_track_infile, locfn)
   206)     call cam_pio_openfile(infile, locfn, PIO_NOWRITE)
   207) 
   208)     ierr = pio_inq_dimid(infile,'profs',dimid)
   209)     ierr = pio_inq_dimlen(infile, dimid, n_profiles)
   210) 
   211)     ierr = pio_inq_varid( infile, 'time', time_vid )
   212)     ierr = pio_inq_varid( infile, 'date', date_vid )
   213) 
   214)     ierr = pio_get_att( infile, date_vid, 'long_name', date_format)
   215)     date_format = to_lower(trim( date_format(:GLC(date_format))))
   216) 
   217)     if ( index( date_format, 'yyyymmdd') > 0 ) then
   218)        doy_format = .false.
   219)     else if  ( index( date_format, 'yyyyddd') > 0 ) then
   220)        doy_format = .true.
   221)     else
   222)        call endrun('sat_hist_init: date_format not recognized : '//trim(date_format))
   223)     endif
   224) 
   225)     ierr = pio_inq_varid( infile, 'lat', lat_vid )
   226)     ierr = pio_inq_varid( infile, 'lon', lon_vid )
   227) 
   228)     call pio_seterrorhandling(infile, PIO_BCAST_ERROR)
   229)     ierr = pio_inq_varid( infile, 'instr_num', instr_vid )
   230)     if(ierr/=PIO_NOERR) instr_vid=-1
   231) 
   232)     ierr = pio_inq_varid( infile, 'orbit_num', orbit_vid )
   233)     if(ierr/=PIO_NOERR) orbit_vid=-1
   234) 
   235)     ierr = pio_inq_varid( infile, 'prof_num',  prof_vid )
   236)     if(ierr/=PIO_NOERR) prof_vid=-1
   237) 
   238)     ierr = pio_inq_varid( infile, 'instr_sza', zenith_vid )
   239)     if(ierr/=PIO_NOERR) zenith_vid=-1
   240) 
   241)     ierr = pio_inq_varid( infile, 'julian', in_julian_vid )
   242)     if(ierr/=PIO_NOERR) in_julian_vid=-1
   243) 
   244)     ierr = pio_inq_varid( infile, 'local_time', in_localtime_vid )
   245)     if(ierr/=PIO_NOERR) in_localtime_vid=-1
   246) 
   247)     ierr = pio_inq_varid( infile, 'doy', in_doy_vid )
   248)     if(ierr/=PIO_NOERR) in_doy_vid=-1
   249) 
   250)     ierr = pio_inq_varid( infile, 'occ_type', in_occ_type_vid )
   251)     if(ierr/=PIO_NOERR) in_occ_type_vid=-1
   252) 
   253)     call pio_seterrorhandling(infile, PIO_INTERNAL_ERROR)
   254) 
   255)     call read_datetime( first_datetime, 1 )
   256)     call read_datetime( last_datetime, n_profiles )
   257)     last_start_index = -1
   258)     t_buffer_size = min(1000,n_profiles)
   259)     allocate( date_buffer(t_buffer_size), time_buffer(t_buffer_size) )
   260)     if (masterproc) write(iulog,*) "sathist_init:", n_profiles, first_datetime, last_datetime
   261)     if ( last_datetime<first_datetime ) then
   262)        call endrun('sat_hist_init: satellite track file has invalid date time info')
   263)     endif
   264) 
   265)     time_ndx = 1
   266)     half_step = get_step_size()*0.5_r8
   267) 
   268)   end subroutine sat_hist_init

   272)   subroutine read_datetime( datetime, index )
   273) 
   274)     real(r8), intent( out ) :: datetime

   294)   subroutine read_buffered_datetime( datetime, index )
   295) 
   296)     real(r8), intent( out ) :: datetime
   297)     integer,  intent( in )  :: index
   298) 
   299)     integer :: ii
   300) 
   301)     integer :: ierr
   302)     integer :: cnt
   303)     integer :: start
   304)     integer :: date, time
   305)     
   306)     ! If the request is outside of the buffer then reload the buffer.
   307)     if ((last_start_index == -1) .or. (index < last_start_index) &
   308)          .or. (index >= (last_start_index + t_buffer_size))) then
   309) 
   310)        start = (index - 1) / t_buffer_size * t_buffer_size + 1
   311)        if ( start+t_buffer_size-1 <= n_profiles ) then
   312)           cnt = t_buffer_size 
   313)        else
   314)           cnt = n_profiles-start+1
   315)        endif
   316)        ierr = pio_get_var( infile, time_vid, (/ start /), (/ cnt /), time_buffer(1:cnt) )
   317)        ierr = pio_get_var( infile, date_vid, (/ start /), (/ cnt /), date_buffer(1:cnt) )
   318) 
   319)        last_start_index = start
   320)     endif
   321) 
   322)     ii = mod( index - 1, t_buffer_size ) + 1
   323)     time = time_buffer(ii)
   324)     date = date_buffer(ii)
   325)     datetime = convert_date_time( date,time )
   326) 
   327)   end subroutine read_buffered_datetime

   331)   function convert_date_time( date,time )
   332)     use time_manager, only: set_time_float_from_date
   333) 
   334)     integer, intent(in) :: date,time
   335)     real(r8) :: convert_date_time
   336) 
   337)     real(r8) :: datetime
   338)     integer :: yr, doy, mon, dom
   339) 
   340)     if ( doy_format ) then
   341)        yr = date/1000
   342)        doy = date - yr*1000
   343)        call set_time_float_from_date( datetime, yr, 1, doy, time )
   344)     else 
   345)        yr = date/10000
   346)        mon = (date - yr*10000)/100
   347)        dom = date - yr*10000 - mon*100
   348)        call set_time_float_from_date( datetime, yr, mon, dom, time )
   349)     endif
   350)     convert_date_time = datetime
   351) 
   352)   end function convert_date_time

   407)   subroutine sat_hist_write( tape , nflds, nfils)
   408) 
   409)     use ppgrid,   only : pcols, begchunk, endchunk
   410)     use dyn_grid, only : get_dyn_grid_parm
   411)     use cam_pio_utils, only: phys_decomp, dyn_decomp
   412)     use cam_history_support, only : active_entry
   413)     use pio, only : pio_file_is_open
   414)     implicit none
   415)     type(active_entry) :: tape
   416)     integer, intent(in) :: nflds
   417)     integer, intent(inout) :: nfils
   418) 
   419)     integer :: t, f, i, ncols, nocols    
   420)     integer :: ierr
   421) 
   422)     integer, allocatable :: col_ndxs(:)
   423)     integer, allocatable :: chk_ndxs(:)
   424)     integer, allocatable :: fdyn_ndxs(:)
   425)     integer, allocatable :: ldyn_ndxs(:)
   426)     integer, allocatable :: phs_owners(:)
   427)     integer, allocatable :: dyn_owners(:)
   428)     real(r8),allocatable :: mlats(:)
   429)     real(r8),allocatable :: mlons(:)
   430)     real(r8),allocatable :: phs_dists(:)
   431) 
   432)     integer :: coldim
   433) 
   434)     integer :: io_type
   435)     logical :: has_dyn_flds
   436) 
   437)     if (.not.has_sat_hist) return
   438) 
   439)     call read_next_position( ncols )
   440) 
   441)     if ( ncols < 1 ) return
   442) 
   443)     call t_startf ('sat_hist_write')
   444) 
   445)     ! The n closest columns to the observation will be output,
   446)     ! so increase the size of the columns used for output/
   447)     nocols = ncols * sathist_nclosest
   448) 
   449)     allocate( col_ndxs(nocols) )
   450)     allocate( chk_ndxs(nocols) )
   451)     allocate( fdyn_ndxs(nocols) )
   452)     allocate( ldyn_ndxs(nocols) )
   453)     allocate( phs_owners(nocols) )
   454)     allocate( dyn_owners(nocols) )
   455)     allocate( mlats(nocols) )
   456)     allocate( mlons(nocols) )
   457)     allocate( phs_dists(nocols) )
   458) 
   459)     has_dyn_flds = .false.
   460)     dyn_flds_loop: do f=1,nflds
   461)        if ( tape%hlist(f)%field%decomp_type == dyn_decomp ) then
   462)           has_dyn_flds = .true.
   463)           exit dyn_flds_loop
   464)        endif
   465)     enddo dyn_flds_loop
   466) 
   467)     call get_indices( obs_lats, obs_lons, ncols, nocols, has_dyn_flds, col_ndxs, chk_ndxs, &
   468)          fdyn_ndxs, ldyn_ndxs, phs_owners, dyn_owners, mlats, mlons, phs_dists )
   469) 
   470)     if ( .not. pio_file_is_open(tape%File) ) then
   471)        call endrun('sat file not open')
   472)     endif
   473) 
   474)     ierr = pio_inq_dimid(tape%File,'ncol',coldim )
   475)     
   476)     ierr = pio_inq_varid(tape%File, 'lat', out_latid )
   477)     ierr = pio_inq_varid(tape%File, 'lon', out_lonid )
   478)     ierr = pio_inq_varid(tape%File, 'distance', out_dstid )
   479) 
   480)     call write_record_coord( tape, mlats(:), mlons(:), phs_dists(:), ncols, nfils )
   481) 
   482)     do f=1,nflds
   483) 
   484)        select case (tape%hlist(f)%field%decomp_type)
   485)        case (phys_decomp)
   486)           call dump_columns(tape%File, tape%hlist(f), nocols, nfils, col_ndxs(:), chk_ndxs(:), phs_owners(:) )
   487)        case (dyn_decomp)
   488)           call dump_columns(tape%File, tape%hlist(f), nocols, nfils, fdyn_ndxs(:), ldyn_ndxs(:), dyn_owners(:) )
   489)        end select
   490) 
   491)     enddo
   492) 
   493)     deallocate( col_ndxs, chk_ndxs, fdyn_ndxs, ldyn_ndxs, phs_owners, dyn_owners )
   494)     deallocate( mlons, mlats, phs_dists )
   495)     deallocate( obs_lons, obs_lats )
   496) 
   497)     nfils = nfils + nocols
   498) 
   499)     call t_stopf ('sat_hist_write')
   500)

   504)   subroutine dump_columns( File, hitem, ncols, nfils, fdims, ldims, owners  )
   505)     use cam_history_support,  only: field_info, hentry, hist_coords
   506)     use pionfwrite_mod, only: write_nf
   507)     use cam_pio_utils, only : fillvalue
   508)     use pio,            only: pio_initdecomp, pio_freedecomp, pio_setframe, pio_offset, pio_iam_iotask, pio_setdebuglevel
   509) 
   510)     type(File_desc_t),intent(inout)  :: File
   511)     type(hentry),     intent(in), target     :: hitem
   512)     integer,          intent(in)     :: ncols
   513)     integer,          intent(in)     :: nfils
   514)     integer,          intent(in)     :: fdims(:)
   515)     integer,          intent(in)     :: ldims(:)
   516)     integer,          intent(in)     :: owners(:)
   517) 
   518)     type(field_info), pointer :: field
   519)     type(var_desc_t) :: vardesc
   520)     type(iosystem_desc_t), pointer :: sat_iosystem
   521)     type(io_desc_t) :: iodesc
   522)     integer :: t, ierr, ndims
   523)     integer, allocatable :: dimlens(:)
   524) 
   525)     real(r8), allocatable :: buf(:)
   526)     integer,  allocatable :: dof(:)
   527)     integer :: i,k, cnt
   528) 
   529)     call t_startf ('sat_hist::dump_columns')
   530) 
   531)     sat_iosystem => File%iosystem
   532)     field => hitem%field
   533)     vardesc = hitem%varid(1)
   534) 
   535) 
   536)     ndims=1
   537)     if(associated(field%mdims)) then
   538)        ndims = size(field%mdims)+1
   539)     else if(field%numlev>1) then
   540)        ndims=2
   541)     end if
   542)     allocate(dimlens(ndims))
   543)     dimlens(ndims)=ncols
   544)     if(ndims>2) then
   545)        do i=1,ndims-1
   546)           dimlens(i)=hist_coords(field%mdims(i))%dimsize

   549)        dimlens(1) = field%numlev

   551)    
   552)

   554)     allocate( dof( product(dimlens) ) )

   558)     dof = 0

   565)              dof(cnt) = cnt
   566)           endif
   567)        enddo
   568)     enddo
   569) 
   570)     call pio_setframe(vardesc, int(-1,kind=PIO_OFFSET))
   571) 
   572)     call pio_initdecomp(sat_iosystem, pio_double, dimlens, dof, iodesc )
   573)     if(pio_iam_iotask(sat_iosystem)) &
   574)          iodesc%start(ndims)=iodesc%start(ndims)+nfils-1
   575) 
   576)     call pio_write_darray(File, vardesc, iodesc, buf, ierr, fillval=fillvalue)
   577) 
   578)     call pio_freedecomp(sat_iosystem, iodesc)
   579) 
   580)     deallocate( buf )
   581)     deallocate( dof )
   582)     deallocate( dimlens )
   583) 
   584)     call t_stopf ('sat_hist::dump_columns')
   585) 
   586)   end subroutine dump_columns

   590)   subroutine read_next_position( ncols )
   591)     use time_manager, only: get_curr_date, get_prev_date
   592)     use time_manager, only: set_time_float_from_date
   593)    
   594)     implicit none
   595) 
   596)     integer,  intent(out) :: ncols
   597) 
   598)     integer :: ierr
   599)     integer :: yr, mon, day, tod
   600)     real(r8) :: begdatetime, enddatetime
   601)     integer :: beg_ndx, end_ndx, i
   602) 
   603)     real(r8) :: datetime
   604) 
   605)     call get_curr_date(yr, mon, day, tod)
   606)     call set_time_float_from_date(begdatetime, yr, mon, day, tod-half_step*sathist_ntimestep)
   607)     call set_time_float_from_date(enddatetime, yr, mon, day, tod+half_step*sathist_ntimestep)
   608) 
   609)     ncols = 0
   610) 
   611)     if ( first_datetime > enddatetime ) then
   612)        if (masterproc) write(iulog,'(a,2f16.6)') &
   613)             'sat_hist->read_next_position: all of the satellite date times are after the time window', first_datetime, enddatetime
   614)        return
   615)     endif
   616)     if ( last_datetime < begdatetime ) then
   617)        if (masterproc) write(iulog,'(a,2f16.6)') &
   618)             'sat_hist->read_next_position: all of the satellite date times are before the time window', begdatetime, last_datetime
   619)        return
   620)     endif
   621) 
   622)     call t_startf ('sat_hist::read_next_position')
   623) 
   624)     beg_ndx = -99
   625)     end_ndx = -99
   626) 
   627)     bnds_loop: do i = time_ndx,n_profiles
   628) 
   629)        call read_buffered_datetime( datetime, i )
   630) 
   631)        if ( datetime>begdatetime .and. beg_ndx<0 ) beg_ndx = i
   632)        if ( datetime>enddatetime ) exit bnds_loop
   633)        end_ndx = i
   634) 
   635)     enddo bnds_loop
   636) 
   637)     if (beg_ndx == -99 .and. end_ndx== -99) then
   638)        if (masterproc) write(iulog,'(a)')  'sat_hist->read_next_position: must be beyond last position -- returning.'
   639)        return
   640)     endif
   641) 
   642)     ! Advance the search forward, but because of ntimesteps, it is possible
   643)     ! for observations used here to be used again. However, we should not go
   644)     ! back before the previous beginning time.
   645)     if (beg_ndx>0) time_ndx = beg_ndx
   646) 
   647)     ncols = end_ndx-beg_ndx+1
   648) 
   649)     if (ncols > 0) then
   650)        allocate( obs_lats(ncols), obs_lons(ncols) )
   651)        in_start_col = beg_ndx
   652) 
   653)        ierr = pio_get_var( infile, lat_vid, (/beg_ndx/), (/ncols/), obs_lats )
   654)        ierr = pio_get_var( infile, lon_vid, (/beg_ndx/), (/ncols/), obs_lons )
   655) 
   656)     endif
   657) 
   658)     call t_stopf ('sat_hist::read_next_position')
   659)   end subroutine read_next_position

   663)   subroutine write_record_coord( tape, mod_lats, mod_lons, mod_dists, ncols, nfils )
   664) 
   665)     use time_manager,  only: get_nstep, get_curr_date, get_curr_time
   666)     use cam_history_support, only : active_entry
   667)     implicit none
   668)     type(active_entry), intent(inout) :: tape
   669) 
   670)     integer,  intent(in) :: ncols
   671)     real(r8), intent(in) :: mod_lats(ncols * sathist_nclosest)
   672)     real(r8), intent(in) :: mod_lons(ncols * sathist_nclosest)
   673)     real(r8), intent(in) :: mod_dists(ncols * sathist_nclosest)
   674)     integer,  intent(in) ::  nfils
   675) 
   676)     integer :: t, ierr, i
   677)     integer :: yr, mon, day      ! year, month, and day components of a date
   678)     integer :: nstep             ! current timestep number
   679)     integer :: ncdate            ! current date in integer format [yyyymmdd]
   680)     integer :: ncsec             ! current time of day [seconds]
   681)     integer :: ndcur             ! day component of current time
   682)     integer :: nscur             ! seconds component of current time
   683)     real(r8) :: time             ! current time
   684)     integer, allocatable  :: itmp(:)
   685)     real(r8), allocatable :: rtmp(:)
   686)     real(r8), allocatable :: out_lats(:)
   687)     real(r8), allocatable :: out_lons(:)
   688) 
   689)     call t_startf ('sat_hist::write_record_coord')
   690) 
   691)     nstep = get_nstep()
   692)     call get_curr_date(yr, mon, day, ncsec)
   693)     ncdate = yr*10000 + mon*100 + day
   694)     call get_curr_time(ndcur, nscur)
   695) 
   696) 
   697)     time = ndcur + nscur/86400._r8
   698) 
   699)     allocate( itmp(ncols * sathist_nclosest) )
   700)     allocate( rtmp(ncols * sathist_nclosest) )
   701)     
   702)     itmp(:) = ncdate
   703)     ierr = pio_put_var(tape%File, tape%dateid,(/nfils/), (/ncols * sathist_nclosest/),itmp)
   704)     itmp(:) = ncsec
   705)     ierr = pio_put_var(tape%File, tape%datesecid,(/nfils/),(/ncols * sathist_nclosest/),itmp)
   706)     rtmp(:) = time
   707)     ierr = pio_put_var(tape%File, tape%timeid, (/nfils/),(/ncols * sathist_nclosest/),rtmp)
   708)     
   709)     deallocate(itmp)
   710)     deallocate(rtmp)
   711) 
   712)     ! output model column coordinates
   713)     ierr = pio_put_var(tape%File, out_latid, (/nfils/),(/ncols * sathist_nclosest/), mod_lats)
   714)     ierr = pio_put_var(tape%File, out_lonid, (/nfils/),(/ncols * sathist_nclosest/), mod_lons)
   715)     ierr = pio_put_var(tape%File, out_dstid, (/nfils/),(/ncols * sathist_nclosest/), mod_dists / 1000._r8)
   716)     
   717)     ! output instrument location
   718)     allocate( out_lats(ncols * sathist_nclosest) )
   719)     allocate( out_lons(ncols * sathist_nclosest) )
   720)     
   721)     do i = 1, ncols
   722)       out_lats(((i-1)*sathist_nclosest)+1 : (i*sathist_nclosest)) = obs_lats(i)
   723)       out_lons(((i-1)*sathist_nclosest)+1 : (i*sathist_nclosest)) = obs_lons(i)
   724)     enddo
   725) 
   726)     ierr = pio_put_var(tape%File, out_instr_lat_vid, (/nfils/),(/ncols * sathist_nclosest/), out_lats)
   727)     ierr = pio_put_var(tape%File, out_instr_lon_vid, (/nfils/),(/ncols * sathist_nclosest/), out_lons)
   728) 
   729)     deallocate(out_lats)
   730)     deallocate(out_lons)
   731)     
   732)     
   733)     ierr = copy_data( infile, date_vid, tape%File, out_obs_date_vid, in_start_col, nfils, ncols )
   734)     ierr = copy_data( infile, time_vid, tape%File, out_obs_time_vid, in_start_col, nfils, ncols )
   735)     
   736)     ! output observation identifiers
   737)     if (instr_vid>0) then
   738)        ierr = copy_data( infile, instr_vid, tape%File, out_instrid, in_start_col, nfils, ncols )
   739)     endif
   740)     if (orbit_vid>0) then
   741)        ierr = copy_data( infile, orbit_vid, tape%File, out_orbid, in_start_col, nfils, ncols )
   742)     endif
   743)     if (prof_vid>0) then
   744)        ierr = copy_data( infile, prof_vid, tape%File, out_profid, in_start_col, nfils, ncols )
   745)     endif
   746)     if (zenith_vid>0) then
   747)        ierr = copy_data( infile, zenith_vid, tape%File, out_zenithid, in_start_col, nfils, ncols )
   748)     endif
   749)     if (in_julian_vid>0) then
   750)        ierr = copy_data( infile, in_julian_vid, tape%File, out_julian_vid, in_start_col, nfils, ncols )
   751)     endif
   752)     if (in_occ_type_vid>0) then
   753)        ierr = copy_data( infile, in_occ_type_vid, tape%File, out_occ_type_vid, in_start_col, nfils, ncols )
   754)     endif
   755)     if (in_localtime_vid>0) then
   756)        ierr = copy_data( infile, in_localtime_vid, tape%File, out_localtime_vid, in_start_col, nfils, ncols )
   757)     endif
   758)     if (in_doy_vid>0) then
   759)        ierr = copy_data( infile, in_doy_vid, tape%File, out_doy_vid, in_start_col, nfils, ncols )
   760)     endif
   761) 
   762)     call t_stopf ('sat_hist::write_record_coord')
   763)   end subroutine write_record_coord

   768)   subroutine get_indices( lats, lons, ncols, nocols, has_dyn_flds, col_ndxs, chk_ndxs, &
   769)        fdyn_ndxs, ldyn_ndxs, phs_owners, dyn_owners, mlats, mlons, phs_dists )
   770) 
   771)     use dyn_grid, only : dyn_grid_get_colndx
   772)     use phys_grid, only: get_rlat_p, get_rlon_p
   773) 
   774)     integer,  intent(in)  :: ncols
   775)     real(r8), intent(in)  :: lats(ncols)
   776)     real(r8), intent(in)  :: lons(ncols)
   777)     integer,  intent(in)  :: nocols
   778)     logical,  intent(in)  :: has_dyn_flds
   779)     integer,  intent(out) :: col_ndxs(nocols)
   780)     integer,  intent(out) :: chk_ndxs(nocols)
   781)     integer,  intent(out) :: fdyn_ndxs(nocols)
   782)     integer,  intent(out) :: ldyn_ndxs(nocols)
   783)     integer,  intent(out) :: phs_owners(nocols)
   784)     integer,  intent(out) :: dyn_owners(nocols)
   785)     real(r8), intent(out) :: mlats(nocols)
   786)     real(r8), intent(out) :: mlons(nocols)
   787)     real(r8), intent(out) :: phs_dists(nocols)
   788) 
   789)     integer :: i, j, ndx
   790)     real(r8) :: lat, lon
   791)     
   792)     integer,  allocatable :: ichks(:),icols(:),idyn1s(:),idyn2s(:), iphs_owners(:), idyn_owners(:)
   793)     real(r8), allocatable :: rlats(:), rlons(:), plats(:), plons(:), iphs_dists(:)
   794) 
   795)     integer :: gcols(sathist_nclosest)
   796) 
   797)     call t_startf ('sat_hist::get_indices')
   798) 
   799)     allocate(ichks(sathist_nclosest),icols(sathist_nclosest),idyn1s(sathist_nclosest), &
   800)          idyn2s(sathist_nclosest),iphs_owners(sathist_nclosest),idyn_owners(sathist_nclosest))
   801)     allocate(rlats(sathist_nclosest), rlons(sathist_nclosest), plats(sathist_nclosest), &
   802)          plons(sathist_nclosest), iphs_dists(sathist_nclosest) )
   803) 
   804)     col_ndxs = -1
   805)     chk_ndxs = -1
   806)     fdyn_ndxs = -1
   807)     ldyn_ndxs = -1
   808)     phs_owners = -1
   809)     dyn_owners = -1
   810)     phs_dists = -1
   811) 
   812)     ndx = 0
   813)     do i = 1,ncols
   814) 
   815)        lat = lats(i)
   816)        lon = lons(i)
   817) 
   818)        if ( lon >= 360._r8) then
   819)          lon = lon-360._r8
   820)        endif
   821)        if ( lon < 0._r8) then
   822)          lon = lon+360._r8
   823)        endif
   824)        if (lat<-90._r8 .or. lat>90._r8) then
   825)           write(iulog,*) 'sat_hist::get_indices lat = ',lat
   826)           call endrun('sat_hist::get_indices : lat must be between -90 and 90 degrees (-90<=lat<=90)')
   827)        endif
   828)        if (lon<0._r8 .or. lon>=360._r8) then
   829)           write(iulog,*) 'sat_hist::get_indices lon = ',lon
   830)           call endrun('sat_hist::get_indices : lon must be between 0 and 360 degrees (0<=lon<360)')
   831)        endif
   832)        
   833)        call find_cols( lat, lon, sathist_nclosest, iphs_owners, ichks, icols, &
   834)                        gcols, iphs_dists, plats, plons )
   835) 
   836)        if (has_dyn_flds) then
   837)           call dyn_grid_get_colndx( gcols, sathist_nclosest, idyn_owners, idyn1s, idyn2s )
   838)        endif
   839) 
   840)        do j = 1, sathist_nclosest
   841)           
   842)           if (debug .and. iam==iphs_owners(j) ) then
   843)              if ( abs(plats(j)-rlats(j))>1.e-3_r8 ) then
   844)                 write(*,'(a,3f20.12)') ' lat, plat, rlat = ', lat, plats(j), rlats(j)
   845)                 write(*,'(a,3f20.12)') ' lon, plon, rlon = ', lon, plons(j), rlons(j)
   846)                 call endrun('sat_hist::get_indices: dyn lat is different than phys lat ')
   847)              endif
   848)              if ( abs(plons(j)-rlons(j))>1.e-3_r8 ) then
   849)                 write(*,'(a,3f20.12)') ' lat, plat, rlat = ', lat, plats(j), rlats(j)
   850)                 write(*,'(a,3f20.12)') ' lon, plon, rlon = ', lon, plons(j), rlons(j)
   851)                 call endrun('sat_hist::get_indices: dyn lon is different than phys lon ')
   852)              endif
   853)           endif
   854)           
   855)           ndx = ndx+1
   856)           
   857)           chk_ndxs(ndx)   = ichks(j)
   858)           col_ndxs(ndx)   = icols(j)
   859)           fdyn_ndxs(ndx)  = idyn1s(j)
   860)           ldyn_ndxs(ndx)  = idyn2s(j)
   861)           mlats(ndx)      = plats(j)
   862)           mlons(ndx)      = plons(j)
   863)           phs_owners(ndx) = iphs_owners(j)
   864)           dyn_owners(ndx) = idyn_owners(j)
   865)           phs_dists(ndx)  = iphs_dists(j)
   866)        enddo
   867)     enddo
   868) 
   869)     deallocate(ichks, icols, idyn1s, idyn2s, iphs_owners, idyn_owners)
   870)     deallocate(rlats, rlons, plats, plons, iphs_dists )
   871) 
   872)     call t_stopf ('sat_hist::get_indices')
   873)   end subroutine get_indices
