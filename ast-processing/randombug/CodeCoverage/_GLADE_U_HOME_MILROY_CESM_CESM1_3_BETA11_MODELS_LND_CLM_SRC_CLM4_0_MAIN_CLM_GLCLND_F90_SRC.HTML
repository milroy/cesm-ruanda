     8) ! !DESCRIPTION:
     9) ! Handle arrays used for exchanging data between glc and land model.
    10) ! Based on clm_atmlnd (but without mapping routines because glc data
    11) !  is send and received on the lnd decomposition, at least for now).
    12) !
    13) ! The fields sent from the lnd component to the glc component via
    14) !  the coupler are labeled 's2x', or sno to coupler.
    15) ! The fields received by the lnd component from the glc component
    16) !  via the coupler are labeled 'x2s', or coupler to sno.
    17) ! 'Sno' is a misnomer in that the exchanged data are related to
    18) !  the ice beneath the snow, not the snow itself.  But by CESM convention,
    19) ! 'ice' refers to sea ice, not land ice.
    20) !

   153) ! !IROUTINE: update_clm_s2x
   154) !
   155) ! !INTERFACE:
   156)   subroutine update_clm_s2x(init)
   157) !
   158) ! !DESCRIPTION:
   159) ! Assign values to clm_s2x based on the appropriate derived types
   160) !
   161) ! !USES:
   162)   use clmtype     
   163)   use domainMod   , only : ldomain
   164)   use clm_varcon  , only : istice_mec
   165)   use clm_atmlnd  , only : clm_l2a, clm_a2l
   166)   use clm_varcon  , only : spval, tfrz
   167) !
   168) ! !ARGUMENTS:
   169)   implicit none
   170) 
   171)   logical, intent(in)   :: init    ! if true=>only set a subset of fields
   172) !
   173) ! !REVISION HISTORY:
   174) ! Written by William Lipscomb, Feb. 2009 
   175) !
   176) 
   177)     integer :: begg, endg              ! per-proc beginning and ending gridcell indices
   178)     integer :: begc, endc              ! per-proc beginning and ending column indices
   179)     integer :: c, l, g, n              ! indices
   180)     integer , pointer :: ityplun(:)    ! landunit type
   181)     integer , pointer :: clandunit(:)  ! column's landunit index
   182)     integer , pointer :: cgridcell(:)  ! column's gridcell index
   183) 
   184)     ! Assign local pointers to derived type members
   185) 
   186)     clandunit => col%landunit
   187)     cgridcell => col%gridcell
   188)     ityplun   => lun%itype
   189) 
   190)     ! Get processor bounds
   191)     call get_proc_bounds(begg, endg, begc=begc, endc=endc)
   192) 
   193)     ! Initialize qice because otherwise it will remain unset if init=true and
   194)     ! glc_smb=true; note that the value here is the value qice will keep if these
   195)     ! conditions hold. Also, need to set a reasonable default for elevation class 0,
   196)     ! which is used by the CLM4.5 code, but not by CLM4.0.
   197) 
   198)     clm_s2x%qice(:,0:) = 0._r8
   199) 
   200)     ! and initialize the other variables just to be safe; also need to set a reasonable
   201)     ! default for elevation class 0 (for tsrf, note that we initialize other elevation
   202)     ! classes to 0, to remain consistent with what was done before, but set elevation
   203)     ! class 0 to a reasonable default, because it will remain at its default value).
   204) 
   205)     clm_s2x%tsrf(:,0)  = tfrz
   206)     clm_s2x%tsrf(:,1:) = 0._r8
   207)     clm_s2x%topo(:,:)  = 0._r8
   208) 
   209)     ! Fill the clm_s2x vector on the clm grid
   210) 
   211)     if (glc_smb) then   ! send surface mass balance info
   212)        do c = begc, endc
   213)           l = clandunit(c)
   214)           g = cgridcell(c)
   215) 
   216)           ! Following assumes all elevation classes are populated
   217)           if (ityplun(l) == istice_mec) then
   218)              n = c - lun%coli(l) + 1    ! elevation class index
   219) 
   220)              ! t_soisno and glc_topo are valid even in initialization, so tsrf and topo
   221)              ! are set here regardless of the value of init. But qflx_glcice is not valid
   222)              ! until the run loop; thus, in initialization, we will use the default value
   223)              ! for qice, as set above.
   224)              clm_s2x%tsrf(g,n) = ces%t_soisno(c,1)
   225)              clm_s2x%topo(g,n) = cps%glc_topo(c)
   226)              if (.not. init) then
   227)                 clm_s2x%qice(g,n) = cwf%qflx_glcice(c)
   228) 
   229)                 ! Check for bad values of qice
   230)                 if ( abs(clm_s2x%qice(g,n)) > 1.0_r8 .and. clm_s2x%qice(g,n) /= spval) then
   231)                    write(iulog,*) 'WARNING: qice out of bounds: g, n, qice =', g, n, clm_s2x%qice(g,n)
   232)                 endif
   233)              end if
   234) 
   235)            endif    ! istice_mec
   236)        enddo        ! c
   237)     else  ! Pass PDD info (same info in each elevation class)
   238)        ! Require maxpatch_glcmec = 1 for this case 
   239)        if (maxpatch_glcmec .ne. 1) then
   240)           call endrun('update_clm_s2x error: maxpatch_glcmec must be 1 if glc_smb is false') 
   241)        end if
   242)        n = 1
   243)        do g = begg, endg
   244)           clm_s2x%tsrf(g,n) = clm_l2a%t_ref2m(g)
   245)           clm_s2x%qice(g,n) = clm_a2l%forc_snow(g)   ! Assume rain runs off
   246)           clm_s2x%topo(g,n) = ldomain%topo(g)
   247)           ! Check for bad values of qice
   248)           if (clm_s2x%qice(g,n) > -1.0_r8 .and. clm_s2x%qice(g,n) < 1.0_r8) then
   249)              continue
   250)           else
   251)              write(iulog,*) 'WARNING: qice out of bounds: g, n, qice =', g, n, clm_s2x%qice(g,n)
   252)              write(iulog,*) 'forc_rain =', clm_a2l%forc_rain(g)
   253)              write(iulog,*) 'forc_snow =', clm_a2l%forc_snow(g)
   254)           endif
   255)        enddo
   256)     endif   ! glc_smb
   257) 
   258) end subroutine update_clm_s2x
