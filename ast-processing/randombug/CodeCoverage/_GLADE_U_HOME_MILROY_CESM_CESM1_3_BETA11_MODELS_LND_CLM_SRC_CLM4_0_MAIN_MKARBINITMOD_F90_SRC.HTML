    40) !

    51)     use clm_varcon   , only : bdsno, istice, istwet, istsoil, isturb, &
    52)                               denice, denh2o, spval, sb, icol_road_perv, &
    53)                               icol_road_imperv, icol_roof, icol_sunwall, &
    54)                               icol_shadewall

    60)     use SNICARMod    , only : snw_rds_min
    61) 
    62) !
    63) ! !ARGUMENTS:
    64)     implicit none
    65) !
    66) ! !CALLED FROM:
    67) ! subroutine initialize in module initializeMod
    68) !
    69) ! !REVISION HISTORY:
    70) ! Created by Mariana Vertenstein
    71) ! 3/07/08 Keith Oleson: initialize h2osoi_vol for all soil layers to 0.3
    72) ! 3/18/08 David Lawrence, initialize deep layers
    73) ! 03/28/08 Mark Flanner, initialize snow aerosols and grain size
    74) !
    75) ! !LOCAL VARIABLES:
    76) !
    77) ! local pointers to implicit in arguments
    78) !
    79)     integer , pointer :: pcolumn(:)        ! column index associated with each pft
    80)     integer , pointer :: ctype(:)          ! column type
    81)     integer , pointer :: clandunit(:)      ! landunit index associated with each column
    82)     integer , pointer :: ltype(:)          ! landunit type
    83)     logical , pointer :: lakpoi(:)         ! true => landunit is a lake point
    84)     integer , pointer :: plandunit(:)      ! landunit index associated with each pft
    85)     logical , pointer :: urbpoi(:)         ! true => landunit is an urban point
    86)     logical , pointer :: ifspecial(:)      ! true => landunit is not vegetated
    87)     real(r8), pointer :: dz(:,:)           ! layer thickness depth (m)
    88)     real(r8), pointer :: watsat(:,:)       ! volumetric soil water at saturation (porosity) (nlevgrnd)
    89)     real(r8), pointer :: h2osoi_ice(:,:)   ! ice lens (kg/m2)
    90)     real(r8), pointer :: h2osoi_liq(:,:)   ! liquid water (kg/m2)
    91)     real(r8), pointer :: bsw2(:,:)         ! Clapp and Hornberger "b" for CN code
    92)     real(r8), pointer :: psisat(:,:)       ! soil water potential at saturation for CN code (MPa)
    93)     real(r8), pointer :: vwcsat(:,:)       ! volumetric water content at saturation for CN code (m3/m3)
    94)     real(r8), pointer :: zi(:,:)           ! interface level below a "z" level (m)
    95)     real(r8), pointer :: wa(:)             ! water in the unconfined aquifer (mm)
    96)     real(r8), pointer :: wt(:)             ! total water storage (unsaturated soil water + groundwater) (mm)
    97)     real(r8), pointer :: zwt(:)            ! water table depth (m)
    98)     real(r8), pointer :: qflx_snow_melt(:)  ! snow melt (net)
    99) !
   100) ! local pointers to implicit out arguments
   101) !
   102)     integer , pointer :: snl(:)             ! number of snow layers
   103)     real(r8), pointer :: t_soisno(:,:)      ! soil temperature (Kelvin)  (-nlevsno+1:nlevgrnd)
   104)     real(r8), pointer :: t_lake(:,:)        ! lake temperature (Kelvin)  (1:nlevlak)
   105)     real(r8), pointer :: t_grnd(:)          ! ground temperature (Kelvin)
   106)     real(r8), pointer :: t_veg(:)           ! vegetation temperature (Kelvin)
   107)     real(r8), pointer :: t_ref2m(:)         ! 2 m height surface air temperature (Kelvin)
   108)     real(r8), pointer :: t_ref2m_u(:)       ! Urban 2 m height surface air temperature (Kelvin)
   109)     real(r8), pointer :: t_ref2m_r(:)       ! Rural 2 m height surface air temperature (Kelvin)
   110)     real(r8), pointer :: h2osoi_vol(:,:)    ! volumetric soil water (0<=h2osoi_vol<=watsat) [m3/m3]
   111)     real(r8), pointer :: h2ocan_col(:)      ! canopy water (mm H2O) (column-level)
   112)     real(r8), pointer :: h2ocan_pft(:)      ! canopy water (mm H2O) (pft-level)
   113)     real(r8), pointer :: h2osno(:)          ! snow water (mm H2O)
   114)     real(r8), pointer :: snowdp(:)          ! snow height (m)
   115)     real(r8), pointer :: qflx_irrig(:)      ! irrigation flux (mm H2O/s)
   116)     real(r8), pointer :: eflx_lwrad_out(:)  ! emitted infrared (longwave) radiation (W/m**2)
   117)     real(r8), pointer :: soilpsi(:,:)       ! soil water potential in each soil layer (MPa)
   118)     real(r8), pointer :: snw_rds(:,:)       ! effective snow grain radius (col,lyr) [microns, m^-6]
   119)     real(r8), pointer :: snw_rds_top(:)     ! snow grain size, top (col) [microns]
   120)     real(r8), pointer :: sno_liq_top(:)     ! liquid water fraction (mass) in top snow layer (col) [frc]
   121)     real(r8), pointer :: mss_bcpho(:,:)     ! mass of hydrophobic BC in snow (col,lyr) [kg]
   122)     real(r8), pointer :: mss_bcphi(:,:)     ! mass of hydrophillic BC in snow (col,lyr) [kg]
   123)     real(r8), pointer :: mss_bctot(:,:)     ! total mass of BC (pho+phi) (col,lyr) [kg]
   124)     real(r8), pointer :: mss_bc_col(:)      ! total mass of BC in snow column (col) [kg]
   125)     real(r8), pointer :: mss_bc_top(:)      ! total mass of BC in top snow layer (col) [kg]
   126)     real(r8), pointer :: mss_cnc_bcphi(:,:) ! mass concentration of BC species 1 (col,lyr) [kg/kg]
   127)     real(r8), pointer :: mss_cnc_bcpho(:,:) ! mass concentration of BC species 2 (col,lyr) [kg/kg]
   128)     real(r8), pointer :: mss_ocpho(:,:)     ! mass of hydrophobic OC in snow (col,lyr) [kg]
   129)     real(r8), pointer :: mss_ocphi(:,:)     ! mass of hydrophillic OC in snow (col,lyr) [kg]
   130)     real(r8), pointer :: mss_octot(:,:)     ! total mass of OC (pho+phi) (col,lyr) [kg]
   131)     real(r8), pointer :: mss_oc_col(:)      ! total mass of OC in snow column (col) [kg]
   132)     real(r8), pointer :: mss_oc_top(:)      ! total mass of OC in top snow layer (col) [kg]
   133)     real(r8), pointer :: mss_cnc_ocphi(:,:) ! mass concentration of OC species 1 (col,lyr) [kg/kg]
   134)     real(r8), pointer :: mss_cnc_ocpho(:,:) ! mass concentration of OC species 2 (col,lyr) [kg/kg]
   135)     real(r8), pointer :: mss_dst1(:,:)      ! mass of dust species 1 in snow (col,lyr) [kg]
   136)     real(r8), pointer :: mss_dst2(:,:)      ! mass of dust species 2 in snow (col,lyr) [kg]
   137)     real(r8), pointer :: mss_dst3(:,:)      ! mass of dust species 3 in snow (col,lyr) [kg]
   138)     real(r8), pointer :: mss_dst4(:,:)      ! mass of dust species 4 in snow (col,lyr) [kg]
   139)     real(r8), pointer :: mss_dsttot(:,:)    ! total mass of dust in snow (col,lyr) [kg]
   140)     real(r8), pointer :: mss_dst_col(:)     ! total mass of dust in snow column (col) [kg]
   141)     real(r8), pointer :: mss_dst_top(:)     ! total mass of dust in top snow layer (col) [kg]
   142)     real(r8), pointer :: mss_cnc_dst1(:,:)  ! mass concentration of dust species 1 (col,lyr) [kg/kg]
   143)     real(r8), pointer :: mss_cnc_dst2(:,:)  ! mass concentration of dust species 2 (col,lyr) [kg/kg]
   144)     real(r8), pointer :: mss_cnc_dst3(:,:)  ! mass concentration of dust species 3 (col,lyr) [kg/kg]
   145)     real(r8), pointer :: mss_cnc_dst4(:,:)  ! mass concentration of dust species 4 (col,lyr) [kg/kg]
   146)     real(r8), pointer :: irrig_rate(:)         ! current irrigation rate [mm/s]
   147)     integer,  pointer :: n_irrig_steps_left(:) ! number of time steps for which we still need to irrigate today (if 0, ignore irrig_rate)
   148) 
   149) !
   150) !
   151) ! !OTHER LOCAL VARIABLES:
   152) !EOP
   153)     integer :: j,l,c,p      ! indices
   154)     integer :: nlevs        ! number of levels
   155)     integer :: begp, endp   ! per-proc beginning and ending pft indices
   156)     integer :: begc, endc   ! per-proc beginning and ending column indices
   157)     integer :: begl, endl   ! per-proc beginning and ending landunit indices
   158)     integer :: begg, endg   ! per-proc gridcell ending gridcell indices
   159)     real(r8):: vwc,psi      ! for calculating soilpsi
   160) !-----------------------------------------------------------------------
   161) 
   162)     if ( masterproc )then
   163)         write(iulog,*) 'Setting initial data to non-spun up values'
   164)     end if
   165) 
   166)     ! Assign local pointers to derived subtypes components (landunit-level)
   167) 
   168)     ltype      => lun%itype
   169)     lakpoi     => lun%lakpoi
   170)     ifspecial  => lun%ifspecial
   171)     urbpoi     => lun%urbpoi
   172) 
   173)     ! Assign local pointers to derived subtypes components (column-level)
   174) 
   175)     ctype            => col%itype
   176)     clandunit        => col%landunit
   177)     snl              => cps%snl
   178)     dz               => cps%dz
   179)     watsat           => cps%watsat
   180)     bsw2             => cps%bsw2
   181)     vwcsat           => cps%vwcsat
   182)     psisat           => cps%psisat
   183)     soilpsi          => cps%soilpsi
   184)     h2osoi_ice       => cws%h2osoi_ice
   185)     h2osoi_liq       => cws%h2osoi_liq
   186)     h2osoi_vol       => cws%h2osoi_vol
   187)     h2ocan_col       => pws_a%h2ocan
   188)     qflx_irrig       => cwf%qflx_irrig
   189)     qflx_snow_melt   => cwf%qflx_snow_melt
   190)     snowdp           => cps%snowdp
   191)     h2osno           => cws%h2osno
   192)     t_soisno         => ces%t_soisno
   193)     t_lake           => ces%t_lake
   194)     t_grnd           => ces%t_grnd
   195)     zi               => cps%zi
   196)     wa               => cws%wa
   197)     wt               => cws%wt
   198)     zwt              => cws%zwt
   199)     snw_rds          => cps%snw_rds
   200)     snw_rds_top      => cps%snw_rds_top
   201)     sno_liq_top      => cps%sno_liq_top
   202)     mss_bcpho        => cps%mss_bcpho
   203)     mss_bcphi        => cps%mss_bcphi
   204)     mss_bctot        => cps%mss_bctot
   205)     mss_bc_col       => cps%mss_bc_col
   206)     mss_bc_top       => cps%mss_bc_top
   207)     mss_cnc_bcphi    => cps%mss_cnc_bcphi
   208)     mss_cnc_bcpho    => cps%mss_cnc_bcpho
   209)     mss_ocpho        => cps%mss_ocpho
   210)     mss_ocphi        => cps%mss_ocphi
   211)     mss_octot        => cps%mss_octot
   212)     mss_oc_col       => cps%mss_oc_col
   213)     mss_oc_top       => cps%mss_oc_top
   214)     mss_cnc_ocphi    => cps%mss_cnc_ocphi
   215)     mss_cnc_ocpho    => cps%mss_cnc_ocpho
   216)     mss_dst1         => cps%mss_dst1
   217)     mss_dst2         => cps%mss_dst2
   218)     mss_dst3         => cps%mss_dst3
   219)     mss_dst4         => cps%mss_dst4
   220)     mss_dsttot       => cps%mss_dsttot
   221)     mss_dst_col      => cps%mss_dst_col
   222)     mss_dst_top      => cps%mss_dst_top
   223)     mss_cnc_dst1     => cps%mss_cnc_dst1
   224)     mss_cnc_dst2     => cps%mss_cnc_dst2
   225)     mss_cnc_dst3     => cps%mss_cnc_dst3
   226)     mss_cnc_dst4     => cps%mss_cnc_dst4
   227)     n_irrig_steps_left => cps%n_irrig_steps_left
   228)     irrig_rate       => cps%irrig_rate
   229) 
   230)     ! Assign local pointers to derived subtypes components (pft-level)
   231) 
   232)     pcolumn        => pft%column
   233)     h2ocan_pft     => pws%h2ocan
   234)     t_veg          => pes%t_veg
   235)     t_ref2m        => pes%t_ref2m
   236)     t_ref2m_u      => pes%t_ref2m_u
   237)     t_ref2m_r      => pes%t_ref2m_r
   238)     plandunit      => pft%landunit
   239)     eflx_lwrad_out => pef%eflx_lwrad_out  
   240) 
   241)     ! Determine subgrid bounds on this processor
   242) 
   243)     call get_proc_bounds(begg, endg, begl, endl, begc, endc, begp, endp)
   244) 
   245)     ! NOTE: h2ocan, h2osno, and snowdp has valid values everywhere
   246)     ! canopy water (pft level)
   247) 
   248)     do p = begp, endp
   249)        h2ocan_pft(p) = 0._r8
   250)        
   251)        ! added for canopy water mass balance under dynamic pft weights
   252)        !pps%tlai(p) = 0._r8
   253)        !pps%tsai(p) = 0._r8
   254)        !pps%elai(p) = 0._r8
   255)        !pps%esai(p) = 0._r8
   256)        !pps%htop(p) = 0._r8
   257)        !pps%hbot(p) = 0._r8
   258)        !pps%frac_veg_nosno_alb(p) = 0._r8
   259)     end do
   260) 
   261)     do c = begc,endc
   262) 
   263)        ! canopy water (column level)
   264) 
   265)        h2ocan_col(c) = 0._r8
   266)        qflx_snow_melt(c) = 0._r8
   267) 
   268)        ! snow water
   269) 
   270)        l = clandunit(c)
   271) 
   272)        ! Note: Glacier_mec columns are initialized with half the maximum snow cover.
   273)        ! This gives more realistic values of qflx_glcice sooner in the simulation
   274)        !  for columns with net ablation, at the cost of delaying ice formation
   275)        !  in columns with net accumulation.
   276)        if (ltype(l)==istice) then
   277)           h2osno(c) = h2osno_max
   278)        elseif (ltype(l)==istice_mec) then
   279)           h2osno(c) = 0.5_r8 * h2osno_max   ! 50 cm if h2osno_max = 1 m
   280)        else
   281)           h2osno(c) = 0._r8
   282)        endif
   283) 
   284)        ! snow depth
   285) 
   286)        snowdp(c)  = h2osno(c) / bdsno
   287) 
   288)        ! Initialize Irrigation to zero
   289)        if (ltype(l)==istsoil) then
   290)           n_irrig_steps_left(c) = 0
   291)           irrig_rate(c)         = 0.0_r8
   292)        end if
   293) 
   294)     end do
   295) 
   296)     ! Set snow layer number, depth and thickiness
   297) 
   298)     call snowdp2lev(begc, endc)
   299) 
   300)     ! Set snow/soil temperature, note:
   301)     ! t_soisno only has valid values over non-lake
   302)     ! t_lake   only has valid values over lake
   303)     ! t_grnd has valid values over all land
   304)     ! t_veg  has valid values over all land
   305) 
   306)     ! NOTE: THESE MEMORY COPIES ARE INEFFICIENT -- SINCE nlev LOOP IS NESTED FIRST!!!!
   307)     do c = begc,endc
   308) 
   309)        t_soisno(c,-nlevsno+1:nlevgrnd) = spval
   310)        t_lake(c,1:nlevlak) = spval
   311) 
   312)        l = clandunit(c)
   313)        if (.not. lakpoi(l)) then  !not lake
   314)           t_soisno(c,-nlevsno+1:0) = spval
   315)           if (snl(c) < 0) then    !snow layer temperatures
   316)              do j = snl(c)+1, 0
   317)                 t_soisno(c,j) = 250._r8
   318)              enddo
   319)           endif
   320)           if (ltype(l)==istice .or. ltype(l)==istice_mec) then
   321)              do j = 1, nlevgrnd
   322)                 t_soisno(c,j) = 250._r8
   323) 
   324)              end do
   325)           else if (ltype(l) == istwet) then
   326)              do j = 1, nlevgrnd
   327)                 t_soisno(c,j) = 277._r8
   328)              end do
   329)           else if (ltype(l) == isturb) then
   330)              if (use_vancouver) then
   331)                 if (ctype(c) == icol_road_perv .or. ctype(c) == icol_road_imperv) then 
   332)                    ! Set road top layer to initial air temperature and interpolate other
   333)                    ! layers down to 20C in bottom layer
   334)                    do j = 1, nlevurb
   335)                       t_soisno(c,j) = 297.56 - (j-1) * ((297.56-293.16)/(nlevurb-1)) 
   336)                    end do
   337)                    ! Set wall and roof layers to initial air temperature
   338)                 else if (ctype(c) == icol_sunwall .or. ctype(c) == icol_shadewall .or. ctype(c) == icol_roof) then
   339)                    do j = 1, nlevurb
   340)                       t_soisno(c,j) = 297.56
   341)                    end do
   342)                 else
   343)                    do j = 1, nlevurb
   344)                       t_soisno(c,j) = 283._r8
   345)                    end do
   346)                 end if
   347)              else if (use_mexicocity) then
   348)                 if (ctype(c) == icol_road_perv .or. ctype(c) == icol_road_imperv) then 
   349)                    ! Set road top layer to initial air temperature and interpolate other
   350)                    ! layers down to 22C in bottom layer
   351)                    do j = 1, nlevurb
   352)                       t_soisno(c,j) = 289.46 - (j-1) * ((289.46-295.16)/(nlevurb-1)) 
   353)                    end do
   354)                 else if (ctype(c) == icol_sunwall .or. ctype(c) == icol_shadewall .or. ctype(c) == icol_roof) then
   355)                    ! Set wall and roof layers to initial air temperature
   356)                    do j = 1, nlevurb
   357)                       t_soisno(c,j) = 289.46
   358)                    end do
   359)                 else
   360)                    do j = 1, nlevurb
   361)                       t_soisno(c,j) = 283._r8
   362)                    end do
   363)                 end if
   364)              else
   365)                 if (ctype(c) == icol_road_perv .or. ctype(c) == icol_road_imperv) then 
   366)                    do j = 1, nlevurb
   367)                       t_soisno(c,j) = 274._r8
   368)                    end do
   369)                 else if (ctype(c) == icol_sunwall .or. ctype(c) == icol_shadewall .or. ctype(c) == icol_roof) then
   370)                    ! Set sunwall, shadewall, roof to fairly high temperature to avoid initialization
   371)                    ! shock from large heating/air conditioning flux
   372)                    do j = 1, nlevurb
   373)                       t_soisno(c,j) = 292._r8
   374)                    end do
   375)                 end if
   376)              end if
   377)           else
   378)              do j = 1, nlevgrnd
   379)                 t_soisno(c,j) = 274._r8
   380)              end do
   381)           endif
   382)           t_grnd(c) = t_soisno(c,snl(c)+1)
   383)        else                     !lake
   384)           t_lake(c,1:nlevlak) = 277._r8
   385)           t_grnd(c) = t_lake(c,1)
   386)        endif
   387) 
   388)     end do
   389) 
   390)     do p = begp, endp
   391)        c = pcolumn(p)
   392)        l = plandunit(p)
   393) 
   394)        ! Initialize Irrigation to zero
   395)        if (ltype(l)==istsoil) then
   396)           qflx_irrig(c)      = 0.0_r8
   397)        end if
   398) 
   399)        if (use_vancouver) then
   400)           t_veg(p) = 297.56
   401)           t_ref2m(p) = 297.56
   402)           if (urbpoi(l)) then
   403)              t_ref2m_u(p) = 297.56
   404)           else
   405)              t_ref2m_u(p) = spval
   406)           end if
   407)           if (ifspecial(l)) then
   408)              t_ref2m_r(p) = spval
   409)           else
   410)              t_ref2m_r(p) = 297.56
   411)           end if
   412)        else if (use_mexicocity) then
   413)           t_veg(p) = 289.46
   414)           t_ref2m(p) = 289.46
   415)           if (urbpoi(l)) then
   416)              t_ref2m_u(p) = 289.46
   417)           else
   418)              t_ref2m_u(p) = spval
   419)           end if
   420)           if (ifspecial(l)) then
   421)              t_ref2m_r(p) = spval
   422)           else
   423)              t_ref2m_r(p) = 289.46
   424)           end if
   425)        else
   426)           t_veg(p) = 283._r8
   427)           t_ref2m(p) = 283._r8
   428)           if (urbpoi(l)) then
   429)              t_ref2m_u(p) = 283._r8
   430)           else
   431)              t_ref2m_u(p) = spval
   432)           end if
   433)           if (ifspecial(l)) then
   434)              t_ref2m_r(p) = spval
   435)           else
   436)              t_ref2m_r(p) = 283._r8
