    35)   logical, public                :: carma_do_incloud  = .false.   ! If .true. then do incloud particle calculations
    36)   logical, public                :: carma_do_grow     = .false.   ! If .true. then do growth
    37)   logical, public                :: carma_do_optics   = .false.   ! If .true. then do optical properties file
    38)   logical, public                :: carma_do_pheat    = .false.   ! If .true. then do particle heating

    41)   logical, public                :: carma_do_thermo   = .false.   ! If .true. then do solve thermodynamics equation

    51)   real(r8), public               :: carma_dt_threshold  = 0.0_r8  ! When non-zero, the largest change in temperature (K) allowed per substep.
