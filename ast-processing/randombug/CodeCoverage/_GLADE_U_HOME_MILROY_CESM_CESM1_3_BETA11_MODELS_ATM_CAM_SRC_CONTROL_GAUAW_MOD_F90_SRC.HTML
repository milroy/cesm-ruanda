    49)    subroutine gauaw(a, w, k)
    50) !-----------------------------------------------------------------------
    51) !
    52) ! Calculate sine of latitudes a(k) and weights w(k) for the gaussian
    53) ! quadrature. The algorithm is described in Davis and Rabinowitz,
    54) ! Journal of Research of the NBS, V 56, Jan 1956.
    55) ! The zeros of the bessel function j0, which are obtained from bsslzr,
    56) ! are used as a first guess for the abscissa.
    57) !
    58) ! Modified 1/23/97 by Jim Rosinski to use real*16 arithmetic in order to
    59) ! achieve (nearly) identical weights and latitudes on all machines.
    60) !
    61) !---------------------------Code history--------------------------------
    62) !
    63) ! Original version:  CCM1
    64) ! Standardized:      L. Bath, Jun 1992
    65) !                    L. Buja, Feb 1996
    66) ! Reviewed:          D. Williamson, J. Hack, Aug 1992
    67) !                    D. Williamson, J. Hack, Feb 1996
    68) !
    69) !-----------------------------------------------------------------------
    70) !
    71) ! $Id$
    72) ! $Author$
    73) !
    74) !-----------------------------------------------------------------------
    75)    implicit none
    76) !------------------------------Arguments--------------------------------
    77) !
    78)    integer , intent(in)  :: k      ! number of latitudes pole to pole
    79)    real(r8), intent(out) :: a(k)   ! sine of latitudes
    80)    real(r8), intent(out) :: w(k)   ! gaussian weights
    81) !
    82) !---------------------------Local workspace-----------------------------
    83) !
    84)    real(r16) sinlat(k)    ! sine of latitudes
    85)    real(r16) wgt(k)       ! gaussian weights
    86)  
    87)    real(r16) eps          ! convergence criterion
    88)    real(r16) c            ! constant combination
    89)    real(r16) fk           ! real k
    90)    real(r16) xz           ! abscissa estimate
    91)    real(r16) pkm1         ! |
    92)    real(r16) pkm2         ! |-polynomials
    93)    real(r16) pkmrk        ! |
    94)    real(r16) pk           ! |
    95)    real(r16) sp           ! current iteration latitude increment
    96)    real(r16) avsp         ! |sp|
    97)    real(r16) fn           ! real n
    98) #ifdef NO_R16
    99)    parameter (eps = 1.e-15_r16)
   100) #else
   101)    parameter (eps = 1.e-27_r16)
   102) #endif
   103)  
   104)    integer kk           ! k/2 (number of latitudes in hemisphere)
   105)    integer is           ! latitude index
   106)    integer iter         ! iteration counter
   107)    integer n,l          ! indices
   108) !
   109) !-----------------------------------------------------------------------
   110) !
   111)    pi  = 4._r16*atan(one)
   112) !
   113) ! The value eps, used for convergence tests in the iterations,
   114) ! can be changed.  Newton iteration is used to find the abscissas.
   115) !
   116)    c = (1._r16-(2._r16/pi)**2)*0.25_r16
   117)    fk = k
   118)    kk = k/2
   119)    call bsslzr(sinlat,kk)
   120)    do is=1,kk
   121)      xz = cos(sinlat(is)/(((fk+0.5_r16)**2+c)**0.5_r16))
   122) !
   123) ! This is the first approximation to xz
   124) !
   125)      iter = 0
   126)  10  continue
   127)      pkm2 = 1._r16
   128)      pkm1 = xz
   129)      iter = iter + 1
   130)      if (iter.gt.10) then
   131) !
   132) ! Error exit
   133) !
   134)        call endrun ('GAUAW: no convergence in 10 iterations')
   135)      end if
   136) !
   137) ! Computation of the legendre polynomial
   138) !
   139)      do n=2,k
   140)        fn = n
   141)        pk = ((2._r16*fn-1._r16)*xz*pkm1-(fn-1._r16)*pkm2)/fn
   142)        pkm2 = pkm1
   143)        pkm1 = pk
   144)      enddo
   145)      pkm1 = pkm2
   146)      pkmrk = (fk*(pkm1-xz*pk))/(1._r16-xz**2)
   147)      sp = pk/pkmrk
   148)      xz = xz - sp
   149)      avsp = abs(sp)
   150)      if (avsp.gt.eps) go to 10
   151)      sinlat(is) = xz
   152)      wgt(is) = (2._r16*(1._r16-xz**2))/(fk*pkm1)**2
   153)    end do
   154) !
   155)    if (k.ne.kk*2) then
   156) !
   157) ! For odd k computation of weight at the equator
   158) !
   159)      sinlat(kk+1) = 0._r16
   160)      pk = 2._r16/fk**2
   161)      do n=2,k,2
   162)        fn = n
   163)        pk = pk*fn**2/(fn-1._r16)**2
   164)      end do
   165)      wgt(kk+1) = pk
   166)    end if
   167) !
   168) ! Complete the sets of abscissas and weights, using the symmetry.
   169) ! Also note truncation from real(r16) to real*8
   170) !
   171)    do n=1,kk
   172)      l = k + 1 - n
   173)      a(n) = sinlat(n)
   174)      a(l) = -sinlat(n)
   175)  
   176)      w(n) = wgt(n)
   177)      w(l) = wgt(n)
   178)    end do
   179)  
   180)    return
   181)    end subroutine gauaw
