    54) ! !INTERFACE:
    55) subroutine CNAllocationInit ( lbc, ubc, lbp, ubp )
    56) !
    57) ! !DESCRIPTION:
    58) !
    59) ! !USES:
    60)    use clm_varcon      , only: secspday
    61)    use clm_time_manager, only: get_step_size
    62)    use surfrdMod       , only: crop_prog
    63)    use clm_varctl      , only: iulog, use_c13
    64)    use shr_infnan_mod  , only: nan => shr_infnan_nan, assignment(=)
    65) ! !ARGUMENTS:
    66)    implicit none
    67)    integer, intent(in) :: lbc, ubc        ! column-index bounds
    68)    integer, intent(in) :: lbp, ubp        ! pft-index bounds
    69) !
    70) ! !CALLED FROM:
    71) !
    72) ! !REVISION HISTORY:
    73) ! 4/6/11: Created by Erik Kluzek
    74) !
    75) ! !LOCAL VARIABLES:
    76)     character(len=32) :: subname = 'CNAllocationInit'
    77) !EOP
    78) !-----------------------------------------------------------------------
    79)    if ( crop_prog )then
    80)       allocate(arepr(lbp:ubp))
    81)       allocate(aroot(lbp:ubp))
    82)       arepr(:) = nan
    83)       aroot(:) = nan
    84)    end if
    85)    allocate(col_plant_ndemand(lbc:ubc))
    86)    col_plant_ndemand(:) = nan
    87) 
    88)    ! set time steps
    89)    dt = real( get_step_size(), r8 )
    90) 
    91)    ! set some space-and-time constant parameters 
    92)    bdnr         = 0.5_r8 * (dt/secspday)
    93)    dayscrecover = 30.0_r8
    94) 
    95)    ! Change namelist settings into private logical variables
    96)    select case(suplnitro)
    97)       case(suplnNon)
    98)          Carbon_only = .false.
    99)          crop_supln  = .false.
   100)       case(suplnCrp)
   101)          Carbon_only = .false.
   102)          crop_supln  = .true.
   103)          if ( .not. crop_prog )then
   104)             call endrun( trim(subname)//'ERROR: '//trim(suplnCrp)// &
   105)                          ' can NOT be on when crop is NOT' )
   106)          end if
   107)       case(suplnAll)
   108)          Carbon_only = .true.
   109)          crop_supln  = .false.
   110)       case default
   111)          write(iulog,*) 'Supplemental Nitrogen flag (suplnitro) can only be: ', &
   112)                         suplnNon, ",", suplnCrp, ', or ', suplnAll
   113)          call endrun( trim(subname)//'ERROR: supplemental Nitrogen flag is not correct' )
   114)    end select
   115) 
   116) end subroutine CNAllocationInit

   404)    npool_to_livestemn_storage  => pnf%npool_to_livestemn_storage
   405)    npool_to_deadstemn          => pnf%npool_to_deadstemn
   406)    npool_to_deadstemn_storage  => pnf%npool_to_deadstemn_storage
   407)    npool_to_livecrootn         => pnf%npool_to_livecrootn
   408)    npool_to_livecrootn_storage => pnf%npool_to_livecrootn_storage
   409)    npool_to_deadcrootn         => pnf%npool_to_deadcrootn
   410)    npool_to_deadcrootn_storage => pnf%npool_to_deadcrootn_storage
   411)    fpi                         => cps%fpi
   412)    fpg                         => cps%fpg
   413)    potential_immob             => cnf%potential_immob
   414)    actual_immob                => cnf%actual_immob
   415)    sminn_to_plant              => cnf%sminn_to_plant
   416)    sminn_to_denit_excess       => cnf%sminn_to_denit_excess
   417)    supplement_to_sminn         => cnf%supplement_to_sminn
   418)    aleafi                      => pps%aleafi
   419)    astemi                      => pps%astemi
   420)    aleaf                       => pps%aleaf
   421)    astem                       => pps%astem
   422) 
   423)    ! loop over pfts to assess the total plant N demand
   424)    do fp=1,num_soilp
   425)       p = filter_soilp(fp)
   426) 
   427)       ! get the time step total gross photosynthesis
   428)       ! this is coming from the canopy fluxes code, and is the
   429)       ! gpp that is used to control stomatal conductance.
   430)       ! For the nitrogen downregulation code, this is assumed
   431)       ! to be the potential gpp, and the actual gpp will be
   432)       ! reduced due to N limitation. 
   433)       
   434)       ! Convert psn from umol/m2/s -> gC/m2/s
   435) 
   436)       ! The input psn (psnsun and psnsha) are expressed per unit LAI
   437)       ! in the sunlit and shaded canopy, respectively. These need to be
   438)       ! scaled by laisun and laisha to get the total gpp for allocation
   439) 
   440)       psnsun_to_cpool(p) = psnsun(p) * laisun(p) * 12.011e-6_r8
   441)       psnshade_to_cpool(p) = psnsha(p) * laisha(p) * 12.011e-6_r8
   442)       if (use_c13) then
   443)          c13_psnsun_to_cpool(p) = c13_psnsun(p) * laisun(p) * 12.011e-6_r8
   444)          c13_psnshade_to_cpool(p) = c13_psnsha(p) * laisha(p) * 12.011e-6_r8
   445)       endif
   446)       
   447)       gpp(p) = psnsun_to_cpool(p) + psnshade_to_cpool(p)
   448) 
   449)       ! get the time step total maintenance respiration
   450)       ! These fluxes should already be in gC/m2/s
   451) 
   452)       mr = leaf_mr(p) + froot_mr(p)
   453)       if (woody(ivt(p)) == 1.0_r8) then
   454)          mr = mr + livestem_mr(p) + livecroot_mr(p)
   455)       else if (ivt(p) >= npcropmin)then
   456)          if (croplive(p)) mr = mr + livestem_mr(p)
   457)       end if
   458) 
   459)       ! carbon flux available for allocation
   460)       availc(p) = gpp(p) - mr
   461)       
   462)       ! new code added for isotope calculations, 7/1/05, PET
   463)       ! If mr > gpp, then some mr comes from gpp, the rest comes from
   464)       ! cpool (xsmr)
   465)       if (mr > 0._r8 .and. availc(p) < 0._r8) then
   466)          curmr = gpp(p)
   467)          curmr_ratio = curmr / mr
   468)       else
   469)          curmr_ratio = 1._r8
   470)       end if
   471)       leaf_curmr(p) = leaf_mr(p) * curmr_ratio
   472)       leaf_xsmr(p) = leaf_mr(p) - leaf_curmr(p)
   473)       froot_curmr(p) = froot_mr(p) * curmr_ratio
   474)       froot_xsmr(p) = froot_mr(p) - froot_curmr(p)
   475)       livestem_curmr(p) = livestem_mr(p) * curmr_ratio
   476)       livestem_xsmr(p) = livestem_mr(p) - livestem_curmr(p)
   477)       livecroot_curmr(p) = livecroot_mr(p) * curmr_ratio
   478)       livecroot_xsmr(p) = livecroot_mr(p) - livecroot_curmr(p)
   479)       
   480)       ! no allocation when available c is negative
   481)       availc(p) = max(availc(p),0.0_r8)
   482) 
   483)       ! test for an xsmrpool deficit
   484)       if (xsmrpool(p) < 0.0_r8) then
   485)          ! Running a deficit in the xsmrpool, so the first priority is to let
   486)          ! some availc from this timestep accumulate in xsmrpool.
   487)          ! Determine rate of recovery for xsmrpool deficit
   488) 
   489)          xsmrpool_recover(p) = -xsmrpool(p)/(dayscrecover*secspday)
   490)          if (xsmrpool_recover(p) < availc(p)) then
   491)              ! available carbon reduced by amount for xsmrpool recovery
   492)              availc(p) = availc(p) - xsmrpool_recover(p)
   493)          else
   494)              ! all of the available carbon goes to xsmrpool recovery
   495)              xsmrpool_recover(p) = availc(p)
   496)              availc(p) = 0.0_r8
   497)          end if
   498)          cpool_to_xsmrpool(p) = xsmrpool_recover(p)
   499)       end if
   500) 
   501)       f1 = froot_leaf(ivt(p))
   502)       f2 = croot_stem(ivt(p))
   503)      
   504)       ! modified wood allocation to be 2.2 at npp=800 gC/m2/yr, 0.2 at npp=0,
   505)       ! constrained so that it does not go lower than 0.2 (under negative annsum_npp)
   506)       ! This variable allocation is only for trees. Shrubs have a constant
   507)       ! allocation as specified in the pft-physiology file.  The value is also used
   508)       ! as a trigger here: -1.0 means to use the dynamic allocation (trees).
   509)       if (stem_leaf(ivt(p)) == -1._r8) then
   510)          f3 = (2.7/(1.0+exp(-0.004*(annsum_npp(p) - 300.0)))) - 0.4
   511)       else
   512)          f3 = stem_leaf(ivt(p))
   513)       end if
   514)       
   515)       f4 = flivewd(ivt(p))
   516)       g1 = grperc(ivt(p))
   517)       g2 = grpnow(ivt(p))
   518)       cnl = leafcn(ivt(p))
   519)       cnfr = frootcn(ivt(p))
   520)       cnlw = livewdcn(ivt(p))
   521)       cndw = deadwdcn(ivt(p))
   522) 
   523)       ! calculate f1 to f5 for prog crops following AgroIBIS subr phenocrop
   524) 
   525)       f5 = 0._r8 ! continued intializations from above
   526) 
   527)       if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   528) 
   529)          if (croplive(p)) then
   530)             ! same phases appear in subroutine CropPhenology
   531) 
   532)             ! Phase 1 completed:
   533)             ! ==================
   534)             ! if hui is less than the number of gdd needed for filling of grain
   535)             ! leaf emergence also has to have taken place for lai changes to occur
   536)             ! and carbon assimilation
   537)             ! Next phase: leaf emergence to start of leaf decline
   538) 
   539)             if (leafout(p) >= huileaf(p) .and. hui(p) < huigrain(p)) then
   540) 
   541)                ! allocation rules for crops based on maturity and linear decrease
   542)                ! of amount allocated to roots over course of the growing season
   543)    
   544)                if (peaklai(p) == 1) then ! lai at maximum allowed
   545)                   arepr(p) = 0._r8
   546)                   aleaf(p) = 1.e-5_r8
   547)                   astem(p) = 0._r8
   548)                   aroot(p) = 1._r8 - arepr(p) - aleaf(p) - astem(p)
   549)                else
   550)                   arepr(p) = 0._r8
   551)                   aroot(p) = max(0._r8, min(1._r8, arooti(ivt(p)) -   &
   552)                                  (arooti(ivt(p)) - arootf(ivt(p))) *  &
   553)                                  min(1._r8, hui(p)/gddmaturity(p))))
   554)                   fleaf = fleafi(ivt(p)) * (exp(-bfact(ivt(p))) -         &
   555)                                 exp(-bfact(ivt(p))*hui(p)/huigrain(p))) / &
   556)                                 (exp(-bfact(ivt(p)))-1) ! fraction alloc to leaf (from J Norman alloc curve)
   557)                   aleaf(p) = max(1.e-5_r8, (1._r8 - aroot(p)) * fleaf)
   558)                   astem(p) = 1._r8 - arepr(p) - aleaf(p) - aroot(p)
   559)                end if
   560)    
   561)                ! AgroIBIS included here an immediate adjustment to aleaf & astem if the 
   562)                ! predicted lai from the above allocation coefficients exceeded laimx.
   563)                ! We have decided to live with lais slightly higher than laimx by
   564)                ! enforcing the cap in the following tstep through the peaklai logic above.
   565) 
   566)                astemi(p) = astem(p) ! save for use by equations after shift
   567)                aleafi(p) = aleaf(p) ! to reproductive phenology stage begins
   568) 
   569)                ! Phase 2 completed:
   570)                ! ==================
   571)                ! shift allocation either when enough gdd are accumulated or maximum number
   572)                ! of days has elapsed since planting
   573) 
   574)             else if (hui(p) >= huigrain(p)) then
   575)    
   576)                aroot(p) = max(0._r8, min(1._r8, arooti(ivt(p)) - &
   577)                          (arooti(ivt(p)) - arootf(ivt(p))) * min(1._r8, hui(p)/gddmaturity(p))))
   578)                if (astemi(p) > astemf(ivt(p))) then
   579)                   astem(p) = max(0._r8, max(astemf(ivt(p)), astem(p) * &
   580)                                  (1._r8 - min((hui(p)-                 &
   581)                                  huigrain(p))/((gddmaturity(p)*declfact(ivt(p)))- &
   582)                                  huigrain(p)),1._r8)**allconss(ivt(p)) )))
   583)                end if
   584)                if (aleafi(p) > aleaff(ivt(p))) then
   585)                   aleaf(p) = max(1.e-5_r8, max(aleaff(ivt(p)), aleaf(p) * &
   586)                                  (1._r8 - min((hui(p)-                    &
   587)                                  huigrain(p))/((gddmaturity(p)*declfact(ivt(p)))- &
   588)                                  huigrain(p)),1._r8)**allconsl(ivt(p)) )))
   589)                end if
   590)                arepr(p) = 1._r8 - aroot(p) - astem(p) - aleaf(p)
   591)                astem(p) = astem(p)+arepr(p)
   592)                arepr(p) = 0._r8
   593) 
   594)             else                   ! pre emergence
   595)                aleaf(p) = 1.e-5_r8 ! allocation coefficients should be irrelevant
   596)                astem(p) = 0._r8    ! because crops have no live carbon pools;
   597)                aroot(p) = 0._r8    ! this applies to this "else" and to the "else"
   598)                arepr(p) = 0._r8    ! a few lines down
   599)             end if
   600) 
   601)             f1 = aroot(p) / aleaf(p)
   602)             f3 = astem(p) / aleaf(p)
   603)             f5 = arepr(p) / aleaf(p)
   604)             g1 = 0.25_r8
   605) 
   606)          else   ! .not croplive
   607)             f1 = 0._r8
   608)             f3 = 0._r8
   609)             f5 = 0._r8
   610)             g1 = 0.25_r8
   611)          end if
   612)       end if
   613) 
   614)       ! based on available C, use constant allometric relationships to
   615)       ! determine N requirements
   616)       if (woody(ivt(p)) == 1.0_r8) then
   617)          c_allometry(p) = (1._r8+g1)*(1._r8+f1+f3*(1._r8+f2))
   618)          n_allometry(p) = 1._r8/cnl + f1/cnfr + (f3*f4*(1._r8+f2))/cnlw + &
   619)                        (f3*(1._r8-f4)*(1._r8+f2))/cndw
   620)       else if (ivt(p) >= npcropmin) then ! skip generic crops
   621)          c_allometry(p) = (1._r8+g1)*(1._r8+f1+f3*(1._r8+f2))
   622)          n_allometry(p) = 1._r8/cnl + f1/cnfr + (f3*f4*(1._r8+f2))/cnlw + &
   623)                        (f3*(1._r8-f4)*(1._r8+f2))/cndw
   624)       else
   625)          c_allometry(p) = 1._r8+g1+f1+f1*g1
   626)          n_allometry(p) = 1._r8/cnl + f1/cnfr
   627)       end if
   628)       plant_ndemand(p) = availc(p)*(n_allometry(p)/c_allometry(p))
   629) 
   630)       ! retranslocated N deployment depends on seasonal cycle of potential GPP
   631)       ! (requires one year run to accumulate demand)
   632) 
   633)       tempsum_potential_gpp(p) = tempsum_potential_gpp(p) + gpp(p)
   634) 
   635)       ! Adding the following line to carry max retransn info to CN Annual Update
   636)       tempmax_retransn(p) = max(tempmax_retransn(p),retransn(p))
   637) 
   638)       if (annsum_potential_gpp(p) > 0.0_r8) then
   639)          avail_retransn(p) = (annmax_retransn(p)/2.0)*(gpp(p)/annsum_potential_gpp(p))/dt
   640)       else
   641)          avail_retransn(p) = 0.0_r8
   642)       end if
   643) 
   644)       ! make sure available retrans N doesn't exceed storage
   645)       avail_retransn(p) = min(avail_retransn(p), retransn(p)/dt)
   646) 
   647)       ! modify plant N demand according to the availability of
   648)       ! retranslocated N
   649)       ! take from retransn pool at most the flux required to meet
   650)       ! plant ndemand
   651) 
   652)       if (plant_ndemand(p) > avail_retransn(p)) then
   653)          retransn_to_npool(p) = avail_retransn(p)
   654)       else
   655)          retransn_to_npool(p) = plant_ndemand(p)
   656)       end if
   657)       plant_ndemand(p) = plant_ndemand(p) - retransn_to_npool(p)
   658) 
   659)    end do ! end pft loop
   660) 
   661)    ! now use the p2c routine to get the column-averaged plant_ndemand
   662)    call p2c(num_soilc,filter_soilc,plant_ndemand,col_plant_ndemand)
   663) 
   664)    ! column loop to resolve plant/heterotroph competition for mineral N
   665)    do fc=1,num_soilc
   666)       c = filter_soilc(fc)
   667)       l = clandunit(c)
   668) 
   669)       sum_ndemand = col_plant_ndemand(c) + potential_immob(c)
   670) 
   671)       if (sum_ndemand*dt < sminn(c)) then
   672)          ! N availability is not limiting immobilization of plant
   673)          ! uptake, and both can proceed at their potential rates
   674) 
   675)          nlimit = 0
   676)          fpi(c) = 1.0_r8
   677)          actual_immob(c) = potential_immob(c)
   678)          sminn_to_plant(c) = col_plant_ndemand(c)
   679) 
   680)          ! under conditions of excess N, some proportion is assumed to
   681)          ! be lost to denitrification, in addition to the constant
   682)          ! proportion lost in the decomposition pathways
   683) 
   684)          sminn_to_denit_excess(c) = bdnr*((sminn(c)/dt) - sum_ndemand)
   685)       else if ( ((.not. Carbon_only) .and.  (.not. crop_supln)) .or. &
   686)                 (crop_supln .and. ( (itypelun(l) /= istcrop)  .or. &
   687)                 ((itypelun(l) == istcrop) .and. (ivt(pfti(c)) < npcropmin) )) ) )then
   688) 
   689)          ! N availability can not satisfy the sum of immobilization and
   690)          ! plant growth demands, so these two demands compete for available
   691)          ! soil mineral N resource.
   692) 
   693)          nlimit = 1
   694)          if (sum_ndemand > 0.0_r8) then
   695)             actual_immob(c) = (sminn(c)/dt)*(potential_immob(c) / sum_ndemand)
   696)          else
   697)             actual_immob(c) = 0.0_r8
   698)          end if
   699) 
   700)          if (potential_immob(c) > 0.0_r8) then
   701)             fpi(c) = actual_immob(c) / potential_immob(c)
   702)          else
   703)             fpi(c) = 0.0_r8
   704)          end if
   705) 
   706)          sminn_to_plant(c) = (sminn(c)/dt) - actual_immob(c)
   707)       else if ( Carbon_only .or. &
   708)                (crop_supln .and. (itypelun(l) == istcrop) .and. &
   709)                 (ivt(pfti(c)) >= npcropmin)) )then
   710)          ! this code block controls the addition of N to sminn pool
   711)          ! to eliminate any N limitation, when Carbon_Only is set.  This lets the
   712)          ! model behave essentially as a carbon-only model, but with the
   713)          ! benefit of keeping track of the N additions needed to
   714)          ! eliminate N limitations, so there is still a diagnostic quantity
   715)          ! that describes the degree of N limitation at steady-state.
   716) 
   717)          nlimit = 1
   718)          fpi(c) = 1.0_r8
   719)          actual_immob(c) = potential_immob(c)
   720)          sminn_to_plant(c) = col_plant_ndemand(c)
   721)          supplement_to_sminn(c) = sum_ndemand - (sminn(c)/dt)
   722)       else
   723)          call endrun( 'This else should NOT be able to happen' )
   724)       end if
   725) 
   726)       ! calculate the fraction of potential growth that can be
   727)       ! acheived with the N available to plants
   728) 
   729)       if (col_plant_ndemand(c) > 0.0_r8) then
   730)          fpg(c) = sminn_to_plant(c) / col_plant_ndemand(c)
   731)       else
   732)          fpg(c) = 1.0_r8
   733)       end if
   734) 
   735)    end do ! end of column loop
   736) 
   737)    ! start new pft loop to distribute the available N between the
   738)    ! competing pfts on the basis of relative demand, and allocate C and N to
   739)    ! new growth and storage
   740) 
   741)    do fp=1,num_soilp
   742)       p = filter_soilp(fp)
   743)       c = pcolumn(p)
   744) 
   745)       ! set some local allocation variables
   746)       f1 = froot_leaf(ivt(p))
   747)       f2 = croot_stem(ivt(p))
   748) 
   749)       ! modified wood allocation to be 2.2 at npp=800 gC/m2/yr, 0.2 at npp=0,
   750)       ! constrained so that it does not go lower than 0.2 (under negative annsum_npp)
   751)       ! There was an error in this formula in previous version, where the coefficient
   752)       ! was 0.004 instead of 0.0025.
   753)       ! This variable allocation is only for trees. Shrubs have a constant
   754)       ! allocation as specified in the pft-physiology file.  The value is also used
   755)       ! as a trigger here: -1.0 means to use the dynamic allocation (trees).
   756)       if (stem_leaf(ivt(p)) == -1._r8) then
   757)         f3 = (2.7/(1.0+exp(-0.004*(annsum_npp(p) - 300.0)))) - 0.4
   758)       else
   759)         f3 = stem_leaf(ivt(p))
   760)       end if
   761)       
   762)       f4 = flivewd(ivt(p))
   763)       g1 = grperc(ivt(p))
   764)       g2 = grpnow(ivt(p))
   765)       cnl = leafcn(ivt(p))
   766)       cnfr = frootcn(ivt(p))
   767)       cnlw = livewdcn(ivt(p))
   768)       cndw = deadwdcn(ivt(p))
   769)       fcur = fcur2(ivt(p))
   770) 
   771)       if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   772)          if (croplive(p)) then
   773)             f1 = aroot(p) / aleaf(p)
   774)             f3 = astem(p) / aleaf(p)
   775)             f5 = arepr(p) / aleaf(p)
   776)             g1 = 0.25_r8
   777)          else
   778)             f1 = 0._r8
   779)             f3 = 0._r8
   780)             f5 = 0._r8
   781)             g1 = 0.25_r8
   782)          end if
   783)       end if
   784) 
   785)       ! increase fcur linearly with ndays_active, until fcur reaches 1.0 at
   786)       ! ndays_active = days/year.  This prevents the continued storage of C and N.
   787)       ! turning off this correction (PET, 12/11/03), instead using bgtr in
   788)       ! phenology algorithm.
   789)       !fcur = fcur + (1._r8 - fcur)*lgsf(p)
   790) 
   791)       sminn_to_npool(p) = plant_ndemand(p) * fpg(c)
   792)       plant_nalloc(p) = sminn_to_npool(p) + retransn_to_npool(p)
   793) 
   794)       ! calculate the associated carbon allocation, and the excess
   795)       ! carbon flux that must be accounted for through downregulation
   796) 
   797)       plant_calloc(p) = plant_nalloc(p) * (c_allometry(p)/n_allometry(p))
   798)       excess_cflux(p) = availc(p) - plant_calloc(p)
   799) 
   800)       ! reduce gpp fluxes due to N limitation
   801)       if (gpp(p) > 0.0_r8) then
   802)          downreg(p) = excess_cflux(p)/gpp(p)
   803)          psnsun_to_cpool(p) = psnsun_to_cpool(p)*(1._r8 - downreg(p))
   804)          psnshade_to_cpool(p) = psnshade_to_cpool(p)*(1._r8 - downreg(p))
   805)          if (use_c13) then
   806)             c13_psnsun_to_cpool(p) = c13_psnsun_to_cpool(p)*(1._r8 - downreg(p))
   807)             c13_psnshade_to_cpool(p) = c13_psnshade_to_cpool(p)*(1._r8 - downreg(p))
   808)          endif
   809)       end if
   810) 
   811)       ! calculate the amount of new leaf C dictated by these allocation
   812)       ! decisions, and calculate the daily fluxes of C and N to current
   813)       ! growth and storage pools
   814) 
   815)       ! fcur is the proportion of this day's growth that is displayed now,
   816)       ! the remainder going into storage for display next year through the
   817)       ! transfer pools
   818) 
   819)       nlc = plant_calloc(p) / c_allometry(p)
   820)       cpool_to_leafc(p)          = nlc * fcur
   821)       cpool_to_leafc_storage(p)  = nlc * (1._r8 - fcur)
   822)       cpool_to_frootc(p)         = nlc * f1 * fcur
   823)       cpool_to_frootc_storage(p) = nlc * f1 * (1._r8 - fcur)
   824)       if (woody(ivt(p)) == 1._r8) then
   825)          cpool_to_livestemc(p)          = nlc * f3 * f4 * fcur
