     1)       module mo_snoe
     2) !----------------------------------------------------------------------
     3) ! An empirical model of nitric oxide (NO) in the lower thermosphere
     4) ! (100 - 150 km altitude), based on measurements from the Student
     5) ! Nitric Oxide Explorer (SNOE). Model uses empirical orthogonal functions
     6) ! (EOFs) derived from the SNOE dataset to describe spatial variability
     7) ! in NO. Model NO is the sum of a mean distribution and EOFs multiplied
     8) ! by coefficients based on geophysical parameters. These geophysical
     9) ! parameters are day of year, Kp magnetic index and F10.7 solar uv index.
    10) !
    11) ! Model is utilized by calling subroutine snoe_3d(), which returns
    12) ! a 3-D distribution of NO on geographic coordinates (lon/lat are
    13) ! supplied by user). Altitude is fixed to SNOE grid (every 3.33 km).
    14) !
    15) ! 
    16) !  DRM 4/03
    17) !----------------------------------------------------------------------
    18) 
    19)       use shr_kind_mod, only : r8 => shr_kind_r8
    20)       use ppgrid,       only : pcols
    21)       use physconst,    only : pi
    22)       use mo_constants, only : r2d, d2r
    23)       use abortutils,   only : endrun
    24)       use cam_logfile,  only : iulog
    25) 
    26)       implicit none
    27) 
    28)       private
    29) 
    30)       public :: snoe_inti, snoe_timestep_init, set_no_ubc
    31)       public :: ndx_no
    32) 
    33)       save
    34) 
    35)       integer,  parameter    :: nmodes  = 3
    36)       real(r8), parameter    :: delz    = 104.6_r8             ! delta z (km)
    37)       real(r8), parameter    :: re      = 6471._r8             ! radial distance from ED at 100 km altitude (km)
    38)       real(r8), parameter    :: delx    = -399.1_r8            ! delta x (km)
    39)       real(r8), parameter    :: dely    = -286.1_r8            ! delta y (km)
    40)       real(r8), parameter    :: twopi   = 2._r8*pi, pid2=0.5_r8*pi
    41)       real(r8), parameter    :: theta_n = d2r*(90._r8 - 78.98_r8)    ! co-latitude of CD North Pole (radians)
    42)       real(r8), parameter    :: phi_n   = d2r*289.1_r8               ! longitude of CD North Pole (radians)
    43) 
    44)       integer :: nmlat

   178) 
   179)       ierr = pio_inq_varid( ncid, 'lat', var_id )
   180)       ierr = pio_get_var( ncid, var_id, mlat2d )
   181)       mlat2d(:) = d2r*mlat2d(:)
   182) 
   183)       ierr = pio_inq_varid( ncid, 'z', var_id )
   184)       ierr = pio_get_var( ncid, var_id, lev )
   185) 
   186) !----------------------------------------------------------------------
   187) !	... close the netcdf file
   188) !----------------------------------------------------------------------
   189)       call pio_closefile( ncid)
   190)

   193)       subroutine geo2mag
   194) !----------------------------------------------------------------------
   195) !	... converts geographic latitude and longitude pair
   196) !           to eccentric dipole (ED) geomagnetic coordinates
   197) !----------------------------------------------------------------------
   198) 
   199)       use ppgrid,    only : begchunk, endchunk
   200)       use phys_grid, only : get_ncols_p, get_rlon_all_p, get_rlat_all_p
   201) 
   202)       implicit none
   203) 
   204) !----------------------------------------------------------------------
   205) !	... local variables
   206) !----------------------------------------------------------------------
   207)       integer  :: c, i, j
   208)       integer  :: ncol
   209)       real(r8) :: cdlat
   210)       real(r8) :: wrk
   211)       real(r8) :: smaglat, cmaglat, cmaglon, smaglon
   212)       real(r8) :: edlon, sedlon, cedlon, edlat, cedlat
   213)       real(r8) :: singlat, cosglat
   214)       real(r8) :: rlons(pcols)
   215)       real(r8) :: rlats(pcols)
   216) 
   217) !----------------------------------------------------------------------
   218) !	... start with centered dipole (CD) coordinates
   219) !----------------------------------------------------------------------
   220) chunk_loop : &
   221)       do c = begchunk,endchunk
   222)          ncol = get_ncols_p( c )
   223)          call get_rlon_all_p( c, ncol, rlons )
   224)          call get_rlat_all_p( c, ncol, rlats )
   225) column_loop : &
   226)          do i = 1,ncol
   227)             singlat = sin( rlats(i) )
   228)             cosglat = cos( rlats(i) )
   229)             wrk     = rlons(i) - phi_n
   230)             cdlat = acos( singlat * cthetan + cosglat * sthetan * cos(wrk) )
   231)             smaglat = sin( cdlat )
   232)             cmaglat = cos( cdlat )
   233)             cmaglon = -(singlat - cthetan * cmaglat) / (sthetan * smaglat)
   234)             smaglon = cosglat * sin(wrk) / smaglat
   235) !----------------------------------------------------------------------
   236) !	... convert CD coords to ED coords- equation (39 a,b) of Fraser-Smith
   237) !----------------------------------------------------------------------
   238)             sedlon = re * smaglat * smaglon - dely
   239)             cedlon = re * smaglat * cmaglon - delx
   240)             edlon  = atan2( sedlon, cedlon )	   	! ED magnetic long. (degrees)
   241)             cedlat = (re * cmaglat - delz) * cos(edlon)
   242)             edlat  = atan2( cedlon, cedlat )		! ED magnetic latitude (degrees)
   243) !----------------------------------------------------------------------
   244) !	... convert co-latitudes into latitudes 
   245) !----------------------------------------------------------------------
   246)             if( edlat < 0._r8 ) then
   247)                edlat = edlat + pi
   248)             end if
   249)             mlat(i,c) = pid2 - edlat
   250)          end do column_loop
   251)       end do chunk_loop
   252) 
   253)       end subroutine geo2mag

   314)       subroutine set_no_ubc( lchunk, ncol, zint, mmr, rho )
   315) !----------------------------------------------------------------------
   316) !	... set no mixing ratio at the model top interface level
   317) !----------------------------------------------------------------------
   318) 
   319)       use constituents, only : pcnst, cnst_mw
   320) 
   321)       implicit none
   322) 
   323) !----------------------------------------------------------------------
   324) !	... dummy arguments
   325) !----------------------------------------------------------------------
   326)       integer,  intent(in)    :: lchunk
   327)       integer,  intent(in)    :: ncol
   328)       real(r8), intent(in)    :: zint(pcols)          ! geopot height (km)
   329)       real(r8), intent(in)    :: rho(pcols)           ! total atm density (kg/m^3)
   330)       real(r8), intent(inout) :: mmr(pcols,pcnst)     ! concentration at top interface (kg/kg)
   331) 
   332) !----------------------------------------------------------------------
   333) !	... local variables
   334) !----------------------------------------------------------------------
   335)       real(r8), parameter :: amu_fac  = 1.65979e-27_r8 ! kg/amu
   336)       real(r8), parameter :: cm3_2_m3  = 1.e6_r8
   337)       integer    :: astat, i, k, ks, ks1
   338)       real(r8)   :: zinterp, zfac, mfac
   339) 
   340) !----------------------------------------------------------------------
   341) !	... map to model levels
   342) !----------------------------------------------------------------------
   343)       if( ndx_no > 0 ) then
   344) column_loop : &
   345)          do i = 1,ncol
   346) 	    zinterp = zint(i)
   347) 	    if( zinterp >= lev(1) ) then
   348) 	       mmr(i,ndx_no) = snoe_no(i,1,lchunk)
   349) 	    else if( zint(i) < lev(nlev) ) then
   350) 	       mmr(i,ndx_no) = 0._r8
   351) 	    else
   352) 	       do ks = 2,nlev
   353) 	          if( zinterp >= lev(ks) ) then
   354) 		     ks1 = ks - 1
   355) 		     zfac = (zinterp - lev(ks))/(lev(ks1) - lev(ks))
   356) 	             mmr(i,ndx_no) = snoe_no(i,ks,lchunk) &
   357)                                      + zfac*(snoe_no(i,ks1,lchunk) - snoe_no(i,ks,lchunk))
   358) 		     exit
   359) 	          end if
   360) 	       end do
   361) 	    end if
   362)          end do column_loop
   363)          mfac = amu_fac * cm3_2_m3 * cnst_mw(ndx_no)
   364)          mmr(:ncol,ndx_no) = mmr(:ncol,ndx_no) * mfac /rho(:ncol)
   365)       end if
   366) 
   367)       end subroutine set_no_ubc

   369)       subroutine snoe_zm( doy, kp, f107, zm )
   370) !----------------------------------------------------------------------
   371) !	... calculates zonal mean nitric oxide distribution on a given day 
   372) !           and solar conditions (represented by the f10.7 and kp indices)
   373) !----------------------------------------------------------------------
   374) 
   375)       implicit none
   376) 
   377) !----------------------------------------------------------------------
   378) !	... dummy arguments
   379) !----------------------------------------------------------------------
   380)       real(r8), intent(in)  :: doy
   381)       real(r8), intent(in)  :: kp 
   382)       real(r8), intent(in)  :: f107
   383)       real(r8), intent(out) :: zm(:,:)
   384) 
   385) !----------------------------------------------------------------------
   386) !	... local variables
   387) !----------------------------------------------------------------------
   388)       integer  :: k
   389)       real(r8) :: theta0         ! day number in degrees
   390)       real(r8) :: dec            ! solar declination angle
   391)       real(r8) :: m1, m2, m3     ! coefficients for first 3 eofs
   392) 
   393) #ifdef SNOE_DIAGS
   394)       write(iulog,*) 'snoe_zm: doy, kp, f107 = ',doy,kp,f107
   395) #endif
   396) 
   397) !----------------------------------------------------------------------
   398) !	... calculate coefficients (m1 to m3) for eofs based on
   399) !           geophysical parametes eof1 - kp 
   400) !----------------------------------------------------------------------
   401) 
   402)       m1 =  kp * 0.785760_r8 - 1.94262_r8
   403) 
   404) !----------------------------------------------------------------------
   405) !	... eof2 - declination
   406) !----------------------------------------------------------------------
   407)       theta0 = twopi * (doy - 1._r8) / 365._r8
   408) 
   409)       dec = .006918_r8 &
   410)           - .399912_r8 * cos(theta0)   + .070257_r8 * sin(theta0) &
   411)           - .006758_r8 * cos(2._r8*theta0) + .000907_r8 * sin(2._r8*theta0) & 
   412)           - .002697_r8 * cos(3._r8*theta0) + .001480_r8 * sin(3._r8*theta0)
   413) 
   414)       dec = dec * r2d
   415) 
   416) #ifdef SNOE_DIAGS
   417)       write(iulog,*) 'snoe_zm: dec = ',dec
   418) #endif
   419)       m2 = -.319782_r8 + dec*(.0973109_r8 + dec*(.000489814_r8 - dec*.000103608_r8))
   420)       
   421) !----------------------------------------------------------------------
   422) !	... eof3 - f107 
   423) !----------------------------------------------------------------------
   424)       m3 =  log10(f107) * 6.44069_r8 - 13.9832_r8
   425) 
   426) #ifdef SNOE_DIAGS
   427)       write(iulog,*) 'snoe_zm: m1, m2, m3 = ',m1,m2,m3
   428) #endif
   429) 
   430) !----------------------------------------------------------------------
   431) !	... zonal mean distrib. is sum of mean and eofs
   432) !----------------------------------------------------------------------
   433)       do k = 1,nlev
   434)          zm(:,k) = no_mean(:,k) - m1 * eofs(:,k,1) + m2 * eofs(:,k,2) - m3 * eofs(:,k,3) 
   435)       end do
   436) 
   437)       end subroutine snoe_zm

   467)       subroutine interpol( nin, xin, yin, nout, xout, yout )
   468) !-----------------------------------------------------------------------
   469) !	... linear interpolation
   470) !           does not extrapolate, but repeats edge values
   471) !-----------------------------------------------------------------------
   472) 
   473)       implicit none
   474) 
   475) !-----------------------------------------------------------------------
   476) !	... dummy arguments
   477) !-----------------------------------------------------------------------
   478)       integer, intent(in) :: nin, nout
   479)       real(r8), intent(in)    :: xin(nin)
   480)       real(r8), intent(in)    :: yin(nin)
   481)       real(r8), intent(in)    :: xout(nout)
   482)       real(r8), intent(out)   :: yout(nout)
   483) 
   484) !-----------------------------------------------------------------------
   485) !	... local variables
   486) !-----------------------------------------------------------------------
   487)       integer :: i, j
   488) 
   489)       do j = 1,nout
   490)         if( xout(j) < xin(1) ) then
   491)           yout(j) = yin(1)
   492)         else
   493)           if( xout(j) > xin(nin) ) then
   494)             yout(j) = yin(nin)
   495)           else
   496)             do i = 1, nin-1
   497)               if ((xout(j) >= xin(i)) .and. (xout(j) < xin(i+1)) ) then
   498)                 yout(j) =  yin(i) &
   499) 		           + (yin(i+1) - yin(i)) * (xout(j) - xin(i)) / (xin(i+1) - xin(i))
   500)               end if
   501)             end do
   502)           end if
   503)         end if
   504)       end do
   505) 
   506)       end subroutine interpol
