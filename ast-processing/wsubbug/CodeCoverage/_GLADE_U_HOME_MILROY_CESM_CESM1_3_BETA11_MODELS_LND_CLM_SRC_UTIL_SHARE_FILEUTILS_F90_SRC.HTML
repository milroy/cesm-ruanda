    81) ! !DESCRIPTION:

    89)      implicit none

    98) ! !LOCAL VARIABLES:
    99) !EOP

   103) !------------------------------------------------------------------------
   104) 
   105)      ! get local file name from full name
   106) 
   107)      locfn = get_filename( fulpath )
   108)      if (len_trim(locfn) == 0) then
   109) 	if (masterproc) write(iulog,*)'(GETFIL): local filename has zero length'
   110)         call endrun
   111)      else
   112)         if (masterproc) write(iulog,*)'(GETFIL): attempting to find local file ',  &
   113)              trim(locfn)
   114)      endif
   115) 
   116)      ! first check if file is in current working directory.
   117) 
   118)      inquire (file=locfn,exist=lexist)
   119)      if (lexist) then
   120)         if (masterproc) write(iulog,*) '(GETFIL): using ',trim(locfn), &
   121)              ' in current working directory'
   122)         RETURN
   123)      endif
   124) 
   125)      ! second check for full pathname on disk

   127)

   130)         if (masterproc) write(iulog,*) '(GETFIL): using ',trim(fulpath)
   131)         RETURN
   132)      else

   134)         if (iflag==0) then
   135)            call endrun ('GETFIL: FAILED to get '//trim(fulpath))
