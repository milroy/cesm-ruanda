    46) ! !IROUTINE: atmo_boundary_layer - compute coefficients for atm-ice fluxes, 
    47) !                                  stress and Tref/Qref
    48) !
    49) ! !INTERFACE:
    50) !
    51)       subroutine atmo_boundary_layer (nx_block, ny_block, &
    52)                                       sfctype,  icells,   &
    53)                                       indxi,    indxj,    & 
    54)                                       Tsf,      potT,     &
    55)                                       uatm,     vatm,     &  
    56)                                       uvel,     vvel,     &  
    57)                                       wind,     zlvl,     &  
    58)                                       Qa,       rhoa,     &
    59)                                       strx,     stry,     &   
    60)                                       Uref,               &
    61)                                       Tref,     Qref,     &
    62)                                       delt,     delq,     &
    63)                                       lhcoef,   shcoef)
    64) 
    65) ! !DESCRIPTION:
    66) !
    67) ! Compute coefficients for atm/ice fluxes, stress, and reference
    68) ! temperature and humidity. NOTE: \\
    69) ! (1) all fluxes are positive downward,  \\
    70) ! (2) here, tstar = (WT)/U*, and qstar = (WQ)/U*,  \\
    71) ! (3a) wind speeds should all be above a minimum speed (eg. 1.0 m/s). \\
    72) ! (3b) (wind-ice) speeds should all be above a minimum speed (eg. 1.0 m/s). \\
    73) !
    74) ! ASSUME:
    75) !  The saturation humidity of air at T(K): qsat(T)  (kg/m**3)
    76) !
    77) ! Code originally based on CSM1
    78) !
    79) ! !REVISION HISTORY: same as module
    80) !
    81) ! !USES:
    82) !
    83) ! !INPUT/OUTPUT PARAMETERS:
    84) !
    85)       integer (kind=int_kind), intent(in) :: &
    86)          nx_block, ny_block, & ! block dimensions
    87)          icells                ! number of cells that require atmo fluxes
    88) 
    89)       integer (kind=int_kind), dimension(nx_block*ny_block), &
    90)          intent(in) :: &
    91)          indxi, indxj    ! compressed i and j indices
    92) 
    93)       character (len=3), intent(in) :: &
    94)          sfctype      ! ice or ocean
    95) 
    96)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(in) :: &
    97)          Tsf      , & ! surface temperature of ice or ocean
    98)          potT     , & ! air potential temperature  (K)
    99)          uatm     , & ! x-direction wind speed (m/s)
   100)          vatm     , & ! y-direction wind speed (m/s)
   101)          uvel     , & ! x-direction ice speed (m/s)
   102)          vvel     , & ! y-direction ice speed (m/s)
   103)          wind     , & ! wind speed (m/s)
   104)          zlvl     , & ! atm level height (m)
   105)          Qa       , & ! specific humidity (kg/kg)
   106)          rhoa         ! air density (kg/m^3)
   107) 
   108)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   109)          intent(inout) :: &
   110)          strx     , & ! x surface stress (N)
   111)          stry         ! y surface stress (N)
   112) 
   113)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(out) :: &
   114)          Uref     , & ! reference height wind speed (m/s)
   115)          Tref     , & ! reference height temperature  (K)
   116)          Qref     , & ! reference height specific humidity (kg/kg)
   117)          delt     , & ! potential T difference   (K)
   118)          delq     , & ! humidity difference      (kg/kg)
   119)          shcoef   , & ! transfer coefficient for sensible heat
   120)          lhcoef       ! transfer coefficient for latent heat
   121) !
   122) !EOP
   123) !
   124)        integer (kind=int_kind) :: &
   125)          k     , & ! iteration index
   126)          i, j  , & ! horizontal indices
   127)          ij        ! combined ij index
   128) 
   129)       real (kind=dbl_kind) :: &
   130)          TsfK  , & ! surface temperature in Kelvin (K)
   131)          xqq   , & ! temporary variable
   132)          psimh , & ! stability function at zlvl   (momentum)
   133)          tau   , & ! stress at zlvl
   134)          fac   , & ! interpolation factor
   135)          al2   , & ! ln(z10   /zTrf)
   136)          psix2 , & ! stability function at zTrf   (heat and water)
   137)          psimhs, & ! stable profile
   138)          ssq   , & ! sat surface humidity     (kg/kg)
   139)          qqq   , & ! for qsat, dqsfcdt
   140)          TTT   , & ! for qsat, dqsfcdt
   141)          qsat  , & ! the saturation humidity of air (kg/m^3)
   142)          Lheat     ! Lvap or Lsub, depending on surface type
   143) 
   144)       real (kind=dbl_kind), dimension (icells) :: &
   145)          ustar , & ! ustar (m/s)
   146)          tstar , & ! tstar
   147)          qstar , & ! qstar
   148)          rdn   , & ! sqrt of neutral exchange coefficient (momentum)
   149)          rhn   , & ! sqrt of neutral exchange coefficient (heat)
   150)          ren   , & ! sqrt of neutral exchange coefficient (water)
   151)          rd    , & ! sqrt of exchange coefficient (momentum)
   152)          re    , & ! sqrt of exchange coefficient (water)
   153)          rh    , & ! sqrt of exchange coefficient (heat)
   154)          vmag  , & ! surface wind magnitude   (m/s)
   155)          alz   , & ! ln(zlvl  /z10)
   156)          thva  , & ! virtual temperature      (K)
   157)          cp    , & ! specific heat of moist air
   158)          hol   , & ! H (at zlvl  ) over L
   159)          stable, & ! stability factor
   160)          psixh     ! stability function at zlvl   (heat and water)
   161) 
   162)       real (kind=dbl_kind), parameter :: &
   163)          cpvir = cp_wv/cp_air-c1, & ! defined as cp_wv/cp_air - 1.
   164)          zTrf  = c2             , & ! reference height for air temp (m)
   165)          umin  = c1                 ! minimum wind speed (m/s)
   166) 
   167)       ! local functions
   168)       real (kind=dbl_kind) :: &
   169)          xd    , & ! dummy argument
   170)          psimhu, & ! unstable part of psimh
   171)          psixhu    ! unstable part of psimx
   172) 
   173)       !------------------------------------------------------------
   174)       ! Define functions
   175)       !------------------------------------------------------------
   176) 
   177)       psimhu(xd)  = log((c1+xd*(c2+xd))*(c1+xd*xd)/c8) &
   178)                   - c2*atan(xd) + pih
   179) !ech                  - c2*atan(xd) + 1.571_dbl_kind
   180) 
   181)       psixhu(xd)  =  c2 * log((c1 + xd*xd)/c2)
   182) 
   183)       al2 = log(zref/zTrf)
   184) 
   185)       !------------------------------------------------------------
   186)       ! Initialize
   187)       !------------------------------------------------------------
   188) 
   189)       do j = 1, ny_block
   190)       do i = 1, nx_block
   191)          Uref(i,j) = c0
   192)          Tref(i,j) = c0
   193)          Qref(i,j) = c0
   194)          delt(i,j) = c0
   195)          delq(i,j) = c0
   196)          shcoef(i,j) = c0
   197)          lhcoef(i,j) = c0
   198)       enddo
   199)       enddo
   200) 
   201)       !------------------------------------------------------------
   202)       ! Compute turbulent flux coefficients, wind stress, and
   203)       ! reference temperature and humidity.
   204)       !------------------------------------------------------------
   205) 
   206)       !------------------------------------------------------------
   207)       ! define variables that depend on surface type
   208)       !------------------------------------------------------------
   209) 
   210)       if (sfctype(1:3)=='ice') then
   211) 
   212)          qqq  = qqqice          ! for qsat
   213)          TTT  = TTTice          ! for qsat
   214)          Lheat = Lsub           ! ice to vapor
   215)          do ij = 1, icells
   216)             i = indxi(ij)
   217)             j = indxj(ij)
   218)             vmag(ij) = max(umin, wind(i,j))
   219) !---------- (3b) option by Andrew Roberts
   220) !            vmag(ij)   = max(umin, sqrt( (uatm(i,j)-uvel(i,j))**2 + (vatm(i,j)-vvel(i,j))**2) )
   221) !---------- (3b) option end
   222)             rdn(ij)  = vonkar/log(zref/iceruf) ! neutral coefficient
   223)          enddo   ! ij
   224) 
   225)       elseif (sfctype(1:3)=='ocn') then
   226) 
   227)          qqq  = qqqocn
   228)          TTT  = TTTocn
   229)          Lheat = Lvap           ! liquid to vapor
   230)          do ij = 1, icells
   231)             i = indxi(ij)
   232)             j = indxj(ij)
   233)             vmag(ij) = max(umin, wind(i,j))
   234) !---------- (3b) option by Andrew Roberts
   235) !            vmag(ij)   = max(umin, sqrt( (uatm(i,j)-uvel(i,j))**2 + (vatm(i,j)-vvel(i,j))**2) )
   236) !---------- (3b) option end
   237)             rdn(ij)  = sqrt(0.0027_dbl_kind/vmag(ij) &
   238)                     + .000142_dbl_kind + .0000764_dbl_kind*vmag(ij))
   239)          enddo   ! ij
   240) 
   241)       endif   ! sfctype
   242) 
   243)       do ij = 1, icells
   244)          i = indxi(ij)
   245)          j = indxj(ij)
   246) 
   247)       !------------------------------------------------------------
   248)       ! define some more needed variables
   249)       !------------------------------------------------------------
   250) 
   251)          TsfK       = Tsf(i,j) + Tffresh     ! surface temp (K)
   252)          qsat       = qqq * exp(-TTT/TsfK)   ! saturation humidity (kg/m^3)
   253)          ssq        = qsat / rhoa(i,j)       ! sat surf hum (kg/kg)
   254) 
   255)          thva(ij)   = potT(i,j) * (c1 + zvir * Qa(i,j)) ! virtual pot temp (K)
   256)          delt(i,j)  = potT(i,j) - TsfK       ! pot temp diff (K)
   257)          delq(i,j)  = Qa(i,j) - ssq          ! spec hum dif (kg/kg)
   258)          alz(ij)    = log(zlvl(i,j)/zref)
   259)          cp(ij)     = cp_air*(c1 + cpvir*ssq)
   260) 
   261)       !------------------------------------------------------------
   262)       ! first estimate of Z/L and ustar, tstar and qstar
   263)       !------------------------------------------------------------
   264) 
   265)          ! neutral coefficients, z/L = 0.0
   266)          rhn(ij) = rdn(ij)
   267)          ren(ij) = rdn(ij)
   268) 
   269)          ! ustar,tstar,qstar
   270)          ustar(ij) = rdn(ij) * vmag(ij)
   271)          tstar(ij) = rhn(ij) * delt(i,j)
   272)          qstar(ij) = ren(ij) * delq(i,j)
   273) 
   274)       enddo                     ! ij
   275) 
   276)       !------------------------------------------------------------
   277)       ! iterate to converge on Z/L, ustar, tstar and qstar
   278)       !------------------------------------------------------------
   279) 
   280)       do k=1,5
   281) 
   282)          do ij = 1, icells
   283)             i = indxi(ij)
   284)             j = indxj(ij)
   285) 
   286)         ! compute stability & evaluate all stability functions
   287)             hol(ij) = vonkar * gravit * zlvl(i,j) &
   288)                    * (tstar(ij)/thva(ij) &
   289)                     + qstar(ij)/(c1/zvir+Qa(i,j))) &
   290)                    / ustar(ij)**2
   291)             hol(ij)    = sign( min(abs(hol(ij)),c10), hol(ij) )
   292)             stable(ij) = p5 + sign(p5 , hol(ij))
   293)             xqq    = max(sqrt(abs(c1 - c16*hol(ij))) , c1)
   294)             xqq    = sqrt(xqq)
   295) 
   296)             ! Jordan et al 1999
   297)             psimhs = -(0.7_dbl_kind*hol(ij) &
   298)                    + 0.75_dbl_kind*(hol(ij)-14.3_dbl_kind) &
   299)                    * exp(-0.35_dbl_kind*hol(ij)) + 10.7_dbl_kind)
   300)             psimh  = psimhs*stable(ij) &
   301)                     + (c1 - stable(ij))*psimhu(xqq)
   302)             psixh(ij)  = psimhs*stable(ij) &
   303)                     + (c1 - stable(ij))*psixhu(xqq)
   304) 
   305)         ! shift all coeffs to measurement height and stability
   306)             rd(ij) = rdn(ij) / (c1+rdn(ij)/vonkar*(alz(ij)-psimh))
   307)             rh(ij) = rhn(ij) / (c1+rhn(ij)/vonkar*(alz(ij)-psixh(ij)))
   308)             re(ij) = ren(ij) / (c1+ren(ij)/vonkar*(alz(ij)-psixh(ij)))
   309) 
   310)         ! update ustar, tstar, qstar using updated, shifted coeffs
   311)             ustar(ij) = rd(ij) * vmag(ij)
   312)             tstar(ij) = rh(ij) * delt(i,j)
   313)             qstar(ij) = re(ij) * delq(i,j)
   314) 
   315)          enddo                  ! ij
   316)       enddo                     ! end iteration
   317) 
   318)       if (calc_strair) then
   319) 
   320)       ! initialize
   321)       do j = 1, ny_block
   322)       do i = 1, nx_block
   323)          strx(i,j) = c0
   324)          stry(i,j) = c0
   325)       enddo
   326)       enddo
   327) 
   328) !DIR$ CONCURRENT !Cray
   329) !cdir nodep      !NEC
   330) !ocl novrec      !Fujitsu
   331)       do ij = 1, icells
   332)          i = indxi(ij)
   333)          j = indxj(ij)
   334) 
   335)       !------------------------------------------------------------
   336)       ! momentum flux
   337)       !------------------------------------------------------------
   338)       ! tau = rhoa(i,j) * ustar * ustar
   339)       ! strx = tau * uatm(i,j) / vmag
   340)       ! stry = tau * vatm(i,j) / vmag
   341)       !------------------------------------------------------------
   342) 
   343)          tau = rhoa(i,j) * ustar(ij) * rd(ij) ! not the stress at zlvl(i,j)
   344)          strx(i,j) = tau * (uatm(i,j)-uvel(i,j))
   345)          stry(i,j) = tau * (vatm(i,j)-vvel(i,j))
   346) 
   347)       enddo                     ! ij
   348) 
   349)       endif                     ! calc_strair
   350) 
   351) !DIR$ CONCURRENT !Cray
   352) !cdir nodep      !NEC
   353) !ocl novrec      !Fujitsu
   354)       do ij = 1, icells
   355)          i = indxi(ij)
   356)          j = indxj(ij)
   357) 
   358)       !------------------------------------------------------------
   359)       ! coefficients for turbulent flux calculation
   360)       !------------------------------------------------------------
   361)       ! add windless coefficient for sensible heat flux
   362)       ! as in Jordan et al (JGR, 1999)
   363)       !------------------------------------------------------------
   364) 
   365)          shcoef(i,j) = rhoa(i,j) * ustar(ij) * cp(ij) * rh(ij) + c1
   366)          lhcoef(i,j) = rhoa(i,j) * ustar(ij) * Lheat  * re(ij)
   367) 
   368)       !------------------------------------------------------------
   369)       ! Compute diagnostics: 2m ref T & Q and 10m wind speed.
   370)       !------------------------------------------------------------
   371)          hol(ij)  = hol(ij)*zTrf/zlvl(i,j)
   372)          xqq      = max( c1, sqrt(abs(c1-c16*hol(ij))) )
   373)          xqq      = sqrt(xqq)
   374)          psix2    = -c5*hol(ij)*stable(ij) + (c1-stable(ij))*psixhu(xqq)
   375)          fac      = (rh(ij)/vonkar) &
   376)                   * (alz(ij) + al2 - psixh(ij) + psix2)
   377)          Tref(i,j)= potT(i,j) - delt(i,j)*fac
   378)          Tref(i,j)= Tref(i,j) - p01*zTrf ! pot temp to temp correction
   379)          fac      = (re(ij)/vonkar) &
   380)                   * (alz(ij) + al2 - psixh(ij) + psix2)
   381)          Qref(i,j)= Qa(i,j) - delq(i,j)*fac
   382) 
   383)          Uref(i,j)= vmag(ij) * rd(ij) / rdn(ij)
   384)       enddo                     ! ij
   385) 
   386)       end subroutine atmo_boundary_layer

   391) ! !IROUTINE: atmo_boundary_const - compute coeeficients for atm-ice fluxes

   396)       subroutine atmo_boundary_const (nx_block, ny_block, &
   397)                                       sfctype,  icells,   &
   398)                                       indxi,    indxj,    & 
   399)                                       uatm,     vatm,     &  
   400)                                       wind,     rhoa,     &
   401)                                       strx,     stry,     &   
   402)                                       lhcoef,   shcoef)
   403) 
   404) ! !DESCRIPTION:
   405) !
   406) ! Compute coefficients for atm/ice fluxes, stress
   407) ! NOTE: \\
   408) ! (1) all fluxes are positive downward,  \\
   409) ! (2) reference temperature and humidity are NOT computed
   410) !
   411) ! !REVISION HISTORY: same as module
   412) !
   413) ! !USES:
   414) !
   415) ! !INPUT/OUTPUT PARAMETERS:
   416) !
   417)       integer (kind=int_kind), intent(in) :: &
   418)          nx_block, ny_block, & ! block dimensions
   419)          icells                ! number of cells that require atmo fluxes
   420) 
   421)       integer (kind=int_kind), dimension(nx_block*ny_block), &
   422)          intent(in) :: &
   423)          indxi, indxj    ! compressed i and j indices
   424) 
   425)       character (len=3), intent(in) :: &
   426)          sfctype      ! ice or ocean
   427) 
   428)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(in) :: &
   429)          uatm     , & ! x-direction wind speed (m/s)
   430)          vatm     , & ! y-direction wind speed (m/s)
   431)          wind     , & ! wind speed (m/s)
   432)          rhoa         ! air density (kg/m^3)
   433) 
   434)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(inout):: &
   435)          strx     , & ! x surface stress (N)
   436)          stry         ! y surface stress (N)
   437) 
   438)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(out):: &
   439)          shcoef   , & ! transfer coefficient for sensible heat
   440)          lhcoef       ! transfer coefficient for latent heat
   441) !
   442) !EOP
   443) !
   444)        integer (kind=int_kind) :: &
   445)          i, j, & ! horizontal indices
   446)          ij      ! combined ij index
   447) 
   448)       real (kind=dbl_kind) :: &
   449)          tau, &  ! stress at zlvl
   450)          Lheat   ! Lvap or Lsub, depending on surface type
   451) 
   452)       !------------------------------------------------------------
   453)       ! Initialize
   454)       !------------------------------------------------------------
   455) 
   456)       do j = 1, ny_block
   457)       do i = 1, nx_block
   458)          shcoef(i,j) = c0
   459)          lhcoef(i,j) = c0
   460)       enddo
   461)       enddo
   462) 
   463)       if (calc_strair) then
   464) 
   465)       do j = 1, ny_block
   466)       do i = 1, nx_block
   467)          strx(i,j) = c0
   468)          stry(i,j) = c0
   469)       enddo
   470)       enddo
   471) 
   472) !DIR$ CONCURRENT !Cray
   473) !cdir nodep      !NEC
   474) !ocl novrec      !Fujitsu
   475)       do ij = 1, icells
   476)          i = indxi(ij)
   477)          j = indxj(ij)
   478) 
   479)       !------------------------------------------------------------
   480)       ! momentum flux
   481)       !------------------------------------------------------------
   482) 
   483)          tau = rhoa(i,j) * 0.0012_dbl_kind * wind(i,j)
   484) !AOMIP         tau = rhoa(i,j) * (1.10_dbl_kind + c4*p01*wind(i,j)) &
   485) !AOMIP                         * wind(i,j) * p001
   486)          strx(i,j) = tau * uatm(i,j)
   487)          stry(i,j) = tau * vatm(i,j)
   488) 
   489)       enddo                     ! ij
   490) 
   491)       endif                     ! calc_strair
   492) 
   493)       !------------------------------------------------------------
   494)       ! define variables that depend on surface type
   495)       !------------------------------------------------------------
   496) 
   497)       if (sfctype(1:3)=='ice') then
   498)          Lheat = Lsub           ! ice to vapor
   499)       elseif (sfctype(1:3)=='ocn') then
   500)          Lheat = Lvap           ! liquid to vapor
   501)       endif   ! sfctype
   502) 
   503) !DIR$ CONCURRENT !Cray
   504) !cdir nodep      !NEC
   505) !ocl novrec      !Fujitsu
   506)       do ij = 1, icells
   507)          i = indxi(ij)
   508)          j = indxj(ij)
   509) 
   510)       !------------------------------------------------------------
   511)       ! coefficients for turbulent flux calculation
   512)       !------------------------------------------------------------
   513) 
   514)          shcoef(i,j) = (1.20e-3_dbl_kind)*cp_air*rhoa(i,j)*wind(i,j)
   515)          lhcoef(i,j) = (1.50e-3_dbl_kind)*Lheat *rhoa(i,j)*wind(i,j)
   516) 
   517)       enddo                     ! ij
   518) 
   519)       end subroutine atmo_boundary_const
