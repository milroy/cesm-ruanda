    21) real(r8) :: frontgfc = huge(1._r8)

    23) ! Level at which to check the frontogenesis function to determine when
    24) ! waves will be launched.
    25) integer :: kfront = 0

    33) ! Initialize module parameters.
    34) subroutine gw_front_init(taubgnd, frontgfc_in, kfront_in, errstring)
    35)   use gw_common, only: dc
    36) 
    37)   ! The following are used to set the module data.
    38)   real(r8), intent(in) :: taubgnd
    39)   real(r8), intent(in) :: frontgfc_in
    40)   integer, intent(in) :: kfront_in
    41)   ! Report any errors from this routine.
    42)   character(len=*), intent(out) :: errstring
    43) 
    44)   ! Parameters to calculate fav (average value of gaussian over bin).
    45) 
    46)   ! Integration interval to get bin average.
    47)   real(r8), parameter :: dca  = 0.1_r8
    48)   ! Width of gaussian in phase speed.
    49)   real(r8), parameter :: c0   = 30._r8
    50) 
    51)   ! Integration bounds and indices.
    52)   integer :: l, n
    53)   real(r8) :: cmn, cmx
    54) 
    55)   integer :: ierr
    56) 
    57)   errstring = ""
    58) 
    59)   frontgfc = frontgfc_in
    60)   kfront = kfront_in
    61) 
    62)   ! Allocate and calculate fav.
    63)   allocate(fav(-pgwv:pgwv), stat=ierr, errmsg=errstring)
    64)   if (ierr /= 0) return
    65) 
    66)   if (pgwv > 0) then
    67)      do l = -pgwv,pgwv
    68)         ! Lower bound of bin.
    69)         cmn = cref(l) - 0.5_r8*dc
    70)         cmx = cref(l) + 0.5_r8*dc
    71)         ! Loop over integration intervals in bin.
    72)         fav(l) = 0.5_r8 * dca * (exp(-(cmn/c0)**2) + exp(-(cmx/c0)**2))
    73)         do n = 1,nint(dc/dca)-1
    74)            fav(l) = fav(l) + dca * exp(-((cmn+n*dca)/c0)**2)
    75)         end do
    76)      end do
    77)      fav = fav/dc
    78)   else
    79)      fav(0) = 1._r8
    80)   end if
    81) 
    82)   ! Multiply by source strength.
    83)   fav = taubgnd * fav
    84)   ! Prohibit wavenumber 0.
    85)   fav(0) = 0._r8
    86) 
    87) end subroutine gw_front_init

    90) subroutine gw_cm_src(ncol, ngwv, kbot, u, v, frontgf, &
    91)      src_level, tend_level, tau, ubm, ubi, xv, yv, c)
    92)   use gw_utils, only: get_unit_vector, dot_2d, midpoint_interp
    93)   !-----------------------------------------------------------------------
    94)   ! Driver for multiple gravity wave drag parameterization.
    95)   !
    96)   ! The parameterization is assumed to operate only where water vapor
    97)   ! concentrations are negligible in determining the density.
    98)   !-----------------------------------------------------------------------
    99) 
   100)   !------------------------------Arguments--------------------------------
   101)   ! Column and gravity wave spectrum dimensions.
   102)   integer, intent(in) :: ncol, ngwv
   103) 
   104)   ! Index of source interface.
   105)   integer, intent(in) :: kbot
   106) 
   107)   ! Midpoint zonal/meridional winds.
   108)   real(r8), intent(in) :: u(ncol,pver), v(ncol,pver)
   109)   ! Frontogenesis function.
   110)   real(r8), intent(in) :: frontgf(:,:)
   111) 
   112)   ! Indices of top gravity wave source level and lowest level where wind
   113)   ! tendencies are allowed.
   114)   integer, intent(out) :: src_level(ncol)
   115)   integer, intent(out) :: tend_level(ncol)
   116) 
   117)   ! Wave Reynolds stress.
   118)   real(r8), intent(out) :: tau(ncol,-pgwv:pgwv,0:pver)
   119)   ! Projection of wind at midpoints and interfaces.
   120)   real(r8), intent(out) :: ubm(ncol,pver), ubi(ncol,0:pver)
   121)   ! Unit vectors of source wind (zonal and meridional components).
   122)   real(r8), intent(out) :: xv(ncol), yv(ncol)
   123)   ! Phase speeds.
   124)   real(r8), intent(out) :: c(ncol,-pgwv:pgwv)
   125) 
   126)   !---------------------------Local Storage-------------------------------
   127)   ! Column and wavenumber indices.
   128)   integer :: k, l
   129) 
   130)   ! Whether or not to launch waves in this column.
   131)   logical :: launch_wave(ncol)
   132) 
   133)   ! Zonal/meridional wind averaged over source region.
   134)   real(r8) :: usrc(ncol), vsrc(ncol)
   135) 
   136)   !------------------------------------------------------------------------
   137)   ! Determine the source layer wind and unit vectors, then project winds.
   138)   !------------------------------------------------------------------------
   139) 
   140)   ! Just use the source level interface values for the source wind speed
   141)   ! and direction (unit vector).
   142)   src_level = kbot
   143)   tend_level = kbot
   144)   usrc = 0.5_r8*(u(:,kbot+1)+u(:,kbot))
   145)   vsrc = 0.5_r8*(v(:,kbot+1)+v(:,kbot))
   146) 
   147)   ! Get the unit vector components and magnitude at the surface.
   148)   call get_unit_vector(usrc, vsrc, xv, yv, ubi(:,kbot))
   149) 
   150)   ! Project the local wind at midpoints onto the source wind.
   151)   do k = 1, kbot
   152)      ubm(:,k) = dot_2d(u(:,k), v(:,k), xv, yv)
   153)   end do
   154) 
   155)   ! Compute the interface wind projection by averaging the midpoint winds.
   156)   ! Use the top level wind at the top interface.
   157)   ubi(:,0) = ubm(:,1)
   158) 
   159)   ubi(:,1:kbot-1) = midpoint_interp(ubm(:,1:kbot))
   160) 
   161)   !-----------------------------------------------------------------------
   162)   ! Gravity wave sources
   163)   !-----------------------------------------------------------------------
   164) 
   165)   tau = 0._r8
   166) 
   167)   ! GW generation depends on frontogenesis at specified level (may be below
   168)   ! actual source level).
   169)   launch_wave = (frontgf(:ncol,kfront) > frontgfc)
   170) 
   171)   do l = 0, ngwv
   172)      where (launch_wave)
   173)         tau(:,l,kbot) = fav(l)
   174)         tau(:,-l,kbot) = fav(l)
   175)      end where
   176)   end do
   177) 
   178)   ! Set phase speeds as reference speeds plus the wind speed at the source
   179)   ! level.
   180)   c = spread(cref, 1, ncol) + spread(abs(ubi(:,kbot)),2,2*ngwv+1)
   181) 
   182) end subroutine gw_cm_src
