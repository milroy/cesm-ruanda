    64)     call trcdata_init( specifier, filename, filelist, datapath, fields, file, &
    65)                        rmv_file, cycle_yr, fixed_ymd, fixed_tod, data_type)
    66) 
    67)     num_tracer_srcs = 0
    68)     if (associated(fields)) num_tracer_srcs = size( fields )
    69) 
    70)     if( num_tracer_srcs < 1 ) then
    71)        
    72)        if (masterproc) then
    73)           write(iulog,*) 'There are no offline tracer sources'
    74)           write(iulog,*) ' '
    75)        end if
    76)        return
    77)     end if
    78) 
    79)     allocate( tracer_src_flds(num_tracer_srcs))
    80) 
    81)     do i = 1, num_tracer_srcs
    82) 
    83)        ndx = get_extfrc_ndx( fields(i)%fldnam )
    84) 
    85)        if (ndx < 1) then
    86)           write(iulog,*) fields(i)%fldnam//' is not configured to have an external source'
    87)           call endrun('tracer_srcs_init')
    88)        endif
    89) 
    90)        tracer_src_flds(i) = fields(i)%fldnam
    91)  
    92)        call addfld(trim(fields(i)%fldnam)//'_trsrc','/cm3/s ', pver, 'I', 'tracer source rate', phys_decomp )
    93) 
    94)     enddo
