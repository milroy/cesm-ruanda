     8) ! !MODULE: m_AttrVect - Multi-field Storage
     9) !
    10) ! !DESCRIPTION:
    11) !
    12) ! An {\em attribute vector} is a scheme for storing bundles of integer 
    13) ! and real data vectors, indexed by the names of the fields stored in 
    14) ! {\tt List} format (see the mpeu module {\tt m\_List} for more 
    15) ! information about the {\tt List} datatype).  The ordering of the 
    16) ! fieldnames in the integer and real attribute {\tt List} components 
    17) ! ({\tt AttrVect\%iList} and {\tt AttrVect\%rList}, respectively) 
    18) ! corresponds to the storage order of the attributes in their respective 
    19) ! data buffers (the components {\tt AttrVect\%iAttr(:,:)} and 
    20) ! {\tt AttrVect\%rAttr(:,:)}, respectively).   The organization of 
    21) ! the fieldnames in {\tt List} format, along with the direct mapping
    22) ! between {\tt List} items and locations in the data buffer, allows 
    23) ! the user to have {\em random access} to the field data.  This 
    24) ! approach also allows the user to set the number and the names of fields 
    25) ! stored in an {\tt AttrVect} at run-time.  
    26) !
    27) ! The {\tt AttrVect} stores field data in a {\em pointwise} fashion 
    28) ! (that is, the data are grouped so that all the integer or real data 
    29) ! associated with an individual point are adjacent to each other in memory. 
    30) ! This amounts to the having the integer and real field data arrays in 
    31) ! the {\tt AttrVect} (the components {\tt AttrVect\%iAttr(:,:)} and 
    32) ! {\tt AttrVect\%rAttr(:,:)}, respectively) having the attribute index
    33) ! as the major (or fastest-varying) index.  A prime example of this is 
    34) ! observational data input to a data assimilation system.  In the Model 
    35) ! Coupling Toolkit, this datatype is the fundamental type for storing 
    36) ! field data exchanged by component models, and forms a basis for other
    37) ! MCT datatypes that encapsulate time accumulation/averaging buffers (the 
    38) ! {\tt Accumulator} datatype defined in the module {\tt m\_Accumulator}), 
    39) ! coordinate grid information (the {\tt GeneralGrid} datatype defined in 
    40) ! the module {\tt m\_GeneralGrid}), and sparse interpolation matrices
    41) ! (the {\tt SparseMatrix} datatype defined in the module 
    42) ! {\tt m\_SparseMatrix}).
    43) !
    44) ! The attribute vector is implemented in Fortran 90 using the 
    45) ! {\tt AttrVect} derived type.  This module contains the definition 
    46) ! of the {\tt AttrVect}, and the numerous methods that service it.  There
    47) ! are a number of initialization (creation) schemes, and a routine for 
    48) ! zeroing out the elements of an {\tt AttrVect}.  There is a method 
    49) ! to {\em clean} up allocated memory used by an {\tt AttrVect} 
    50) ! (destruction).  There are numerous query methods that return:  the 
    51) ! number of datapoints (or {\em length}; the numbers of integer and 
    52) ! real attributes; the data buffer index of a given real or integer 
    53) ! attribute; and return the lists of real and integer attributes.  There 
    54) ! also exist methods for exporting a given attribute as a one-dimensional
    55) ! array and importing a given attribute from a one-dimensional array.  
    56) ! There is a method for copying attributes from one {\tt AttrVect} to 
    57) ! another.  There is also a method for cross-indexing the attributes in 
    58) ! two {\tt AttrVect} variables.  In addition, there are methods that
    59) ! return those cross-indexed attributes along with some auxiliary data
    60) ! in a {\tt AVSharedIndicesOneType} or {\tt AVSharedIndices} structure.
    61) ! Finally, there are methods for sorting and permuting {\tt AttrVect}
    62) ! entries using a MergeSort scheme keyed by the attributes of the {\tt
    63) ! AttrVect}.

   258) ! !IROUTINE: init_ - Initialize an AttrVect Given Attribute Lists and Length
   259) !
   260) ! !DESCRIPTION:
   261) ! This routine creates an {\tt AttrVect} (the output argument {\tt aV}) 
   262) ! using the optional input {\tt CHARACTER} arguments {\tt iList}, and 
   263) ! {\tt rList} to define its integer and real attributes, respectively.
   264) ! The optional input {\tt INTEGER} argument {\tt lsize} defines the 
   265) ! number of points for which we are storing attributes, or the 
   266) ! {\em length} of {\tt aV}.  The expected form for the arguments 
   267) ! {\tt iList} and {\tt rList} are colon-delimited strings where each
   268) ! substring defines an attribute.  Suppose we wish to store {\tt N} 
   269) ! observations that have the real attributes {\tt 'latitude'}, 
   270) ! {\tt 'longitude'}, {\tt pressure}, {\tt 'u-wind'}, and 
   271) ! {\tt 'v-wind'}.  Suppose we also wish to store the integer 
   272) ! attributes {\tt 'hour'}, {\tt 'day'}, {\tt 'month'}, {\tt 'year'}, 
   273) ! and {\tt 'data source'}.  This can be accomplished by invoking 
   274) ! {\tt init\_()} as follows:
   275) ! \begin{verbatim} 
   276) ! call init_(aV, 'hour:day:month:year:data source', &
   277) !            'latitude:longitude:pressure:u-wind:v-wind', N)
   278) ! \end{verbatim} 
   279) ! The resulting {\tt AttrVect} {\tt aV} will have five integer 
   280) ! attributes, five real attributes, and length {\tt N}.
   281) !
   282) ! !INTERFACE:
   283) 
   284)  subroutine init_(aV, iList, rList, lsize)
   285) !
   286) ! !USES:
   287) !
   288)       use m_List, only : List
   289)       use m_List, only : init,nitem
   290)       use m_List, only : List_nullify => nullify
   291)       use m_mall
   292)       use m_die
   293) 
   294)       implicit none
   295) 
   296) ! !INPUT PARAMETERS: 
   297) !
   298)       character(len=*), optional, intent(in)  :: iList
   299)       character(len=*), optional, intent(in)  :: rList
   300)       integer,          optional, intent(in)  :: lsize
   301) 
   302) ! !OUTPUT PARAMETERS: 
   303) !
   304)       type(AttrVect),             intent(out) :: aV
   305) 
   306) ! !REVISION HISTORY:
   307) ! 09Apr98 - Jing Guo <guo@thunder> - initial prototype/prolog/code
   308) ! 09Oct01 - J.W. Larson <larson@mcs.anl.gov> - added feature to
   309) !           nullify all pointers before usage.  This was done to
   310) !           accomodate behavior of the f90 ASSOCIATED intrinsic 
   311) !           function on the AIX platform.
   312) ! 07Dec01 - E.T. Ong <eong@mcs.anl.gov> - added support for 
   313) !           intialization with blank character strings for iList
   314) !           and rList
   315) !EOP ___________________________________________________________________
   316) !
   317)   character(len=*),parameter :: myname_=myname//'::init_'
   318)   integer :: nIA,nRA,n,ier
   319) 
   320)        ! Initially, nullify all pointers in the AttrVect aV:
   321) 
   322)   nullify(aV%iAttr)
   323)   nullify(aV%rAttr)
   324)   call List_nullify(aV%iList)
   325)   call List_nullify(aV%rList)
   326) 
   327)   if(present(rList)) then
   328)      if(len_trim(rList) > 0) then
   329) 	call init(aV%rList,rList)	! init.List()
   330)      endif
   331)   endif
   332) 
   333)   if(present(iList)) then
   334)      if(len_trim(iList) > 0) then
   335) 	call init(aV%iList,iList)	! init.List()
   336)      endif
   337)   endif
   338) 
   339)   nIA=nitem(aV%iList)		! nitem.List()
   340)   nRA=nitem(aV%rList)		! nitem.List()
   341) 
   342)   n=0
   343)   if(present(lsize)) n=lsize
   344) 
   345)   allocate( aV%iAttr(nIA,n),aV%rAttr(nRA,n),	stat=ier)
   346)   if(ier /= 0) call die(myname_,'allocate()',ier)
   347) 
   348) #ifdef MALL_ON
   349) 	call mall_ci(size(transfer(aV%iAttr,(/1/)),myname_)
   350) 	call mall_ci(size(transfer(aV%rAttr,(/1/)),myname_)
   351) #endif
   352) 
   353)  end subroutine init_

   369) !
   370) ! !USES:
   371) !
   372)       use m_String, only : String,char
   373)       use m_String, only : String_clean => clean    
   374)       use m_List,   only : get
   375)       use m_List,   only : List_nullify => nullify
   376)       use m_die
   377)       use m_stdio

   379)       implicit none
   380) 
   381) ! !INPUT PARAMETERS: 
   382) !
   383)       type(AttrVect),intent(in)  :: bV
   384)       integer,       intent(in)  :: lsize
   385) 
   386) ! !OUTPUT PARAMETERS: 
   387) !
   388)       type(AttrVect),intent(out) :: aV
   389) 
   390) ! !REVISION HISTORY:
   391) ! 22Apr98 - Jing Guo <guo@thunder> - initial prototype/prolog/code
   392) ! 17May01 - R. Jacob <jacob@mcs.anl.gov> - add a check to see if
   393) !           input argument has been defined.  SGI will dump
   394) !           core if its not.
   395) ! 10Oct01 - J. Larson <larson@mcs.anl.gov> - Nullify all pointers
   396) !           in ouput AttrVect aV before initializing aV.
   397) ! 19Sep08 - J. Wolfe <jwolfe@ucar.edu> - plug memory leak from not deallocating
   398) !           strings.
   399) !EOP ___________________________________________________________________
   400) 
   401)   character(len=*),parameter :: myname_=myname//'::initv_'
   402)   type(String) :: iLStr,rLStr
   403) 
   404) 	! Step One:  Nullify all pointers in aV.  We will set
   405)         ! only the pointers we really need for aV based on those
   406)         ! currently ASSOCIATED in bV.
   407) 
   408)   call List_nullify(aV%iList)
   409)   call List_nullify(aV%rList)
   410)   nullify(aV%iAttr)
   411)   nullify(aV%rAttr)
   412) 
   413) 	! Convert the two Lists to two Strings
   414) 
   415)   if(.not.associated(bv%iList%bf) .and. & 
   416)        .not.associated(bv%rList%bf)) then
   417)      write(stderr,'(2a)')myname_, &
   418)       'MCTERROR:  Trying to initialize a new AttrVect off an undefined AttrVect'
   419)       call die(myname_,'undefined input argument',0)
   420)   endif
   421) 
   422)   if(associated(bv%iList%bf)) then
   423)      call get(iLStr,bv%iList)
   424)   endif
   425) 
   426)   if(associated(bv%rList%bf)) then
   427)      call get(rLStr,bv%rList)
   428)   endif
   429) 
   430)        ! Initialize the AttrVect aV depending on which parts of
   431)        ! the input bV are valid:
   432) 
   433)   if(associated(bv%iList%bf) .and. associated(bv%rList%bf)) then
   434)      call init_(aV,iList=char(iLStr),rList=char(rLStr),lsize=lsize)
   435)   endif
   436)   if(.not.associated(bv%iList%bf) .and. associated(bv%rList%bf)) then
   437)      call init_(aV,rList=char(rLStr),lsize=lsize)
   438)   endif
   439)   if(associated(bv%iList%bf) .and. .not.associated(bv%rList%bf)) then
   440)      call init_(aV,iList=char(iLStr),lsize=lsize)
   441)   endif
   442) 
   443)   if(associated(bv%iList%bf)) then  
   444)      call String_clean(iLStr)       
   445)   endif                             
   446)   if(associated(bv%rList%bf)) then  
   447)      call String_clean(rLStr)       
   448)   endif                             
   449) 
   450)  end subroutine initv_

   687) ! !IROUTINE: lsize_ - Length of an AttrVect
   688) !
   689) ! !DESCRIPTION:
   690) ! This function returns the number of elements, or {\em length} of the 
   691) ! input {\tt AttrVect} argument {\tt aV}.  This function examines the 
   692) ! length of the second dimension of the arrays {\tt aV\%iAttr(:,:)} 
   693) ! and {\tt aV\%rAttr(:,:)}.  If neither {\tt aV\%iAttr(:,:)} nor
   694) ! {\tt aV\%rAttr(:,:)} are associated, then ${\tt lsize\_(aV)} = 0$.
   695) ! If {\tt aV\%iAttr(:,:)} is associated, but {\tt aV\%rAttr(:,:)} is 
   696) ! not, then ${\tt lsize\_(aV)} = {\tt size(aV\%iAttr,2)}$. If 
   697) ! {\tt aV\%iAttr(:,:)} is not associated, but {\tt aV\%rAttr(:,:)} is, 
   698) ! then ${\tt lsize\_(aV)} = {\tt size(aV\%rAttr,2)}$. If both 
   699) ! {\tt aV\%iAttr(:,:)} and {\tt aV\%rAttr(:,:)} are associated, the
   700) ! function {\tt lsize\_()} will do one of two things:  If 
   701) ! ${\tt size(aV\%iAttr,2)} = {\tt size(aV\%rAttr,2)}$, this equal value 
   702) ! will be returned.  If ${\tt size(aV\%iAttr,2)} \neq 
   703) ! {\tt size(aV\%rAttr,2)}$, termination with an error message will occur.
   704) !
   705) ! !INTERFACE:
   706) 
   707)  integer function lsize_(aV)
   708) 
   709) ! !USES:
   710) 
   711)      use m_List,  only : List
   712)      use m_List,  only : List_allocated => allocated
   713) 
   714)      use m_stdio, only : stderr
   715)      use m_die
   716)  
   717)      implicit none
   718) 
   719) ! !INPUT PARAMETERS: 
   720) !
   721)       type(AttrVect), intent(in) :: aV
   722) 
   723) ! !REVISION HISTORY:
   724) ! 09Apr98 - Jing Guo <guo@thunder> - initial prototype/prolog/code
   725) ! 10Oct01 - J. Larson <larson@mcs.anl.gov> - made code more robust
   726) !           to handle cases where the length of either aV%iAttr or
   727) !           aV%rAttr is zero, but the other is positive.
   728) !EOP ___________________________________________________________________
   729) 
   730)   character(len=*),parameter :: myname_=myname//'::lsize_'
   731)   integer :: iLength, rLength
   732) 
   733) 	! One should try to avoid using this function on an undefined
   734) 	! or disassocated pointer.  However, it is understandable 
   735) 	! that an undefined or disassocated pointer has a size 0, if
   736) 	! the associated() test sucesses.
   737) 
   738)   lsize_=0
   739) 
   740)   if(List_allocated(aV%iList) .and. associated(aV%iAttr)) then
   741)      iLength = size(aV%iAttr,2)
   742)   else
   743)      iLength = 0
   744)   endif
   745) 
   746)   if(List_allocated(aV%rList) .and. associated(aV%rAttr)) then
   747)      rLength = size(aV%rAttr,2)
   748)   else
   749)      rLength = 0
   750)   endif
   751) 
   752)   if(iLength /= rLength) then
   753) 
   754)      if((rLength > 0) .and. (iLength > 0)) then
   755) 	call die(myname_,'attribute array length mismatch', &
   756) 	     iLength-rLength)
   757)      endif
   758) 
   759)      if((rLength > 0) .and. (iLength == 0)) then
   760) 	lsize_ = rLength
   761)      endif
   762) 
   763)      if((iLength > 0) .and. (rLength == 0)) then
   764) 	lsize_ = iLength
   765)      endif
   766) 
   767)   endif
   768) 
   769)   if(iLength == rLength) lsize_ = iLength
   770) 
   771)  end function lsize_
