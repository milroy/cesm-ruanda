   134) subroutine microp_aero_init
   135) 
   136)    !----------------------------------------------------------------------- 
   137)    ! 
   138)    ! Purpose: 
   139)    ! Initialize constants for aerosols needed by microphysics
   140)    ! 
   141)    ! Author: Andrew Gettelman May 2010
   142)    ! 
   143)    !-----------------------------------------------------------------------
   144) 
   145)    ! local variables
   146)    integer  :: iaer
   147)    integer  :: m, n, nmodes, nspec
   148) 
   149)    character(len=32) :: str32
   150)    character(len=*), parameter :: routine = 'microp_aero_init'
   151)    logical :: history_amwg
   152)    !-----------------------------------------------------------------------
   153) 
   154)    ! Query the PBL eddy scheme
   155)    call phys_getopts(eddy_scheme_out          = eddy_scheme, &
   156)                      history_amwg_out = history_amwg)
   157) 
   158)    ! Access the physical properties of the aerosols that are affecting the climate
   159)    ! by using routines from the rad_constituents module.
   160) 
   161)    ! get indices into state and pbuf structures
   162)    call cnst_get_ind('CLDLIQ', cldliq_idx)
   163)    call cnst_get_ind('CLDICE', cldice_idx)
   164)    call cnst_get_ind('NUMLIQ', numliq_idx)
   165)    call cnst_get_ind('NUMICE', numice_idx)
   166) 
   167)    select case(trim(eddy_scheme))
   168)    case ('diag_TKE')
   169)       tke_idx      = pbuf_get_index('tke')
   170)    case ('CLUBB_SGS')
   171)       wp2_idx = pbuf_get_index('WP2')
   172)    case default
   173)       kvh_idx      = pbuf_get_index('kvh')
   174)    end select
   175) 
   176)    ! prog_modal_aero determines whether prognostic modal aerosols are present in the run.
   177)    call phys_getopts(prog_modal_aero_out=prog_modal_aero)
   178) 
   179)    ! clim_modal_aero determines whether modal aerosols are used in the climate calculation.
   180)    ! The modal aerosols can be either prognostic or prescribed.
   181)    call rad_cnst_get_info(0, nmodes=nmodes)
   182)    clim_modal_aero = (nmodes > 0)
   183) 
   184)    ast_idx      = pbuf_get_index('AST')
   185) 
   186)    if (clim_modal_aero) then
   187) 
   188)       cldo_idx     = pbuf_get_index('CLDO')
   189)       dgnum_idx    = pbuf_get_index('DGNUM' )
   190)       dgnumwet_idx = pbuf_get_index('DGNUMWET')
   191) 
   192)       call ndrop_init()
   193) 
   194)       ! Init indices for specific modes/species
   195) 
   196)       ! mode index for specified mode types
   197)       do m = 1, nmodes
   198)          call rad_cnst_get_info(0, m, mode_type=str32)
   199)          select case (trim(str32))
   200)          case ('accum')
   201)             mode_accum_idx = m
   202)          case ('aitken')
   203)             mode_aitken_idx = m
   204)          case ('coarse')
   205)             mode_coarse_idx = m
   206)          case ('coarse_dust')
   207)             mode_coarse_dst_idx = m
   208)          case ('coarse_seasalt')
   209)             mode_coarse_slt_idx = m
   210)          end select
   211)       end do
   212) 
   213)       ! check if coarse dust is in separate mode
   214)       separate_dust = mode_coarse_dst_idx > 0
   215) 
   216)       ! for 3-mode 
   217)       if ( mode_coarse_dst_idx<0 ) mode_coarse_dst_idx = mode_coarse_idx
   218)       if ( mode_coarse_slt_idx<0 ) mode_coarse_slt_idx = mode_coarse_idx
   219) 
   220)       ! Check that required mode types were found
   221)       if (mode_accum_idx == -1 .or. mode_aitken_idx == -1 .or. &
   222)           mode_coarse_dst_idx == -1.or. mode_coarse_slt_idx == -1) then
   223)          write(iulog,*) routine//': ERROR required mode type not found - mode idx:', &
   224)             mode_accum_idx, mode_aitken_idx, mode_coarse_dst_idx, mode_coarse_slt_idx
   225)          call endrun(routine//': ERROR required mode type not found')
   226)       end if
   227) 
   228)       ! species indices for specified types
   229)       ! find indices for the dust and seasalt species in the coarse mode
   230)       call rad_cnst_get_info(0, mode_coarse_dst_idx, nspec=nspec)
   231)       do n = 1, nspec
   232)          call rad_cnst_get_info(0, mode_coarse_dst_idx, n, spec_type=str32)
   233)          select case (trim(str32))
   234)          case ('dust')
   235)             coarse_dust_idx = n
   236)          end select
   237)       end do
   238)       call rad_cnst_get_info(0, mode_coarse_slt_idx, nspec=nspec)
   239)       do n = 1, nspec
   240)          call rad_cnst_get_info(0, mode_coarse_slt_idx, n, spec_type=str32)
   241)          select case (trim(str32))
   242)          case ('seasalt')
   243)             coarse_nacl_idx = n
   244)          end select
   245)       end do
   246) 
   247)       ! Check that required mode specie types were found
   248)       if ( coarse_dust_idx == -1 .or. coarse_nacl_idx == -1) then
   249)          write(iulog,*) routine//': ERROR required mode-species type not found - indicies:', &
   250)             coarse_dust_idx, coarse_nacl_idx
   251)          call endrun(routine//': ERROR required mode-species type not found')
   252)       end if
   253) 
   254)       ! get specific mode properties
   255)       call rad_cnst_get_mode_props(0, mode_aitken_idx, sigmag=sigmag_aitken)
   256) 
   257)    else
   258) 
   259)       ! Props needed for BAM number concentration calcs.
   260) 
   261)       call rad_cnst_get_info(0, naero=naer_all)
   262)       allocate( &
   263)          aername(naer_all),        &
   264)          num_to_mass_aer(naer_all) )
   265) 
   266)       do iaer = 1, naer_all
   267)          call rad_cnst_get_aer_props(0, iaer, &
   268)             aername         = aername(iaer), &
   269)             num_to_mass_aer = num_to_mass_aer(iaer) )
   270) 
   271)          ! Look for sulfate, dust, and soot in this list (Bulk aerosol only)
   272)          if (trim(aername(iaer)) == 'SULFATE') idxsul = iaer
   273)          if (trim(aername(iaer)) == 'DUST1') idxdst1 = iaer
   274)          if (trim(aername(iaer)) == 'DUST2') idxdst2 = iaer
   275)          if (trim(aername(iaer)) == 'DUST3') idxdst3 = iaer
   276)          if (trim(aername(iaer)) == 'DUST4') idxdst4 = iaer
   277)          if (trim(aername(iaer)) == 'BCPHIL') idxbcphi = iaer
   278)       end do
   279) 
   280)       call ndrop_bam_init()
   281) 
   282)    end if
   283) 
   284)    call addfld('LCLOUD', ' ', pver, 'A', 'Liquid cloud fraction used in stratus activation', phys_decomp)
   285) 
   286)    call addfld('WSUB     ', 'm/s     ', pver, 'A', 'Diagnostic sub-grid vertical velocity'                   ,phys_decomp)
   287)    call addfld('WSUBI    ', 'm/s     ', pver, 'A', 'Diagnostic sub-grid vertical velocity for ice'           ,phys_decomp)
   288)    call addfld('NIHF',  '1/m3', pver, 'A', 'Activated Ice Number Concentation due to homogenous freezing',  phys_decomp)
   289)    call addfld('NIDEP', '1/m3', pver, 'A', 'Activated Ice Number Concentation due to deposition nucleation',phys_decomp)
   290)    call addfld('NIIMM', '1/m3', pver, 'A', 'Activated Ice Number Concentation due to immersion freezing',   phys_decomp)
   291)    call addfld('NIMEY', '1/m3', pver, 'A', 'Activated Ice Number Concentation due to meyers deposition',    phys_decomp)
   292) 
   293)    if (history_amwg) then
   294)       call add_default ('WSUB     ', 1, ' ')
   295)    end if
   296) 
   297) end subroutine microp_aero_init

   301) subroutine microp_aero_readnl(nlfile)
   302) 
   303)    use namelist_utils,  only: find_group_name
   304)    use units,           only: getunit, freeunit
   305)    use mpishorthand
   306) 
   307)    character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
   308) 
   309)    ! Namelist variables
   310)    real(r8) :: microp_aero_bulk_scale = 2._r8  ! prescribed aerosol bulk sulfur scale factor
   311)  
   312)    ! Local variables
   313)    integer :: unitn, ierr
   314)    character(len=*), parameter :: subname = 'microp_aero_readnl'
   315) 
   316)    namelist /microp_aero_nl/ microp_aero_bulk_scale
   317)    !-----------------------------------------------------------------------------
   318) 
   319)    if (masterproc) then
   320)       unitn = getunit()
   321)       open( unitn, file=trim(nlfile), status='old' )
   322)       call find_group_name(unitn, 'microp_aero_nl', status=ierr)
   323)       if (ierr == 0) then
   324)          read(unitn, microp_aero_nl, iostat=ierr)
   325)          if (ierr /= 0) then
   326)             call endrun(subname // ':: ERROR reading namelist')
   327)          end if
   328)       end if
   329)       close(unitn)
   330)       call freeunit(unitn)
   331)    end if
   332) 
   333) #ifdef SPMD
   334)    ! Broadcast namelist variable
   335)    call mpibcast(microp_aero_bulk_scale, 1, mpir8, 0, mpicom)
   336) #endif
   337) 
   338)    ! set local variables
   339)    bulk_scale = microp_aero_bulk_scale
   340) 
   341) end subroutine microp_aero_readnl

   345) subroutine microp_aero_run ( &
   346)    state, ptend, deltatin, pbuf)
   347) 
   348)    ! input arguments
   349)    type(physics_state), target, intent(in)    :: state
   350)    type(physics_ptend),         intent(out)   :: ptend
   351)    real(r8),                    intent(in)    :: deltatin     ! time step (s)
   352)    type(physics_buffer_desc),   pointer       :: pbuf(:)
   353) 
   354) 
   355) 
   356) 
   357)    ! local workspace
   358)    ! all units mks unless otherwise stated
   359) 
   360)    integer :: i, k, m
   361)    integer :: itim_old
   362)    integer :: lchnk
   363)    integer :: ncol
   364)    integer :: nmodes
   365)    integer :: nucboast
   366) 
   367)    real(r8), pointer :: ast(:,:)        
   368) 
   369)    real(r8)          :: icecldf(pcols,pver)    ! ice cloud fraction   
   370)    real(r8)          :: liqcldf(pcols,pver)    ! liquid cloud fraction
   371) 
   372)    real(r8), pointer :: naai(:,:)       ! number of activated aerosol for ice nucleation 
   373)    real(r8), pointer :: naai_hom(:,:)   ! number of activated aerosol for ice nucleation (homogeneous freezing only)
   374)    real(r8), pointer :: npccn(:,:)      ! number of CCN (liquid activated)
   375)    real(r8), pointer :: rndst(:,:,:)    ! radius of 4 dust bins for contact freezing
   376)    real(r8), pointer :: nacon(:,:,:)    ! number in 4 dust bins for contact freezing
   377) 
   378)    real(r8), pointer :: t(:,:)          ! input temperature (K)
   379)    real(r8), pointer :: qn(:,:)         ! input water vapor mixing ratio (kg/kg)
   380)    ! note: all input cloud variables are grid-averaged
   381)    real(r8), pointer :: qc(:,:)         ! cloud water mixing ratio (kg/kg)
   382)    real(r8), pointer :: qi(:,:)         ! cloud ice mixing ratio (kg/kg)
   383)    real(r8), pointer :: nc(:,:)         ! cloud water number conc (1/kg)
   384)    real(r8), pointer :: ni(:,:)         ! cloud ice number conc (1/kg)
   385)    real(r8), pointer :: pmid(:,:)       ! pressure at layer midpoints (pa)
   386)    real(r8), pointer :: pdel(:,:)       ! pressure difference across level (pa)
   387)    real(r8), pointer :: pint(:,:)       ! air pressure layer interfaces (pa)
   388)    real(r8), pointer :: rpdel(:,:)      ! inverse pressure difference across level (pa)
   389)    real(r8), pointer :: zm(:,:)         ! geopotential height of model levels (m)
   390)    real(r8), pointer :: omega(:,:)      ! vertical velocity (Pa/s)
   391)    real(r8), pointer :: num_accum(:,:)  ! number m.r. of accumulation mode
   392)    real(r8), pointer :: num_aitken(:,:) ! number m.r. of aitken mode
   393)    real(r8), pointer :: num_coarse(:,:) ! number m.r. of coarse mode
   394)    real(r8), pointer :: coarse_dust(:,:) ! mass m.r. of coarse dust
   395)    real(r8), pointer :: coarse_nacl(:,:) ! mass m.r. of coarse nacl
   396) 
   397)    real(r8), pointer :: kvh(:,:)        ! vertical eddy diff coef (m2 s-1)
   398)    real(r8), pointer :: tke(:,:)        ! TKE from the UW PBL scheme (m2 s-2)
   399)    real(r8), pointer :: wp2(:,:)        ! CLUBB vertical velocity variance
   400) 
   401)    real(r8), pointer :: cldn(:,:)       ! cloud fraction
   402)    real(r8), pointer :: cldo(:,:)       ! old cloud fraction
   403) 
   404)    real(r8), pointer :: dgnum(:,:,:)    ! aerosol mode dry diameter
   405)    real(r8), pointer :: dgnumwet(:,:,:) ! aerosol mode diameter
   406) 
   407)    real(r8), pointer :: aer_mmr(:,:)    ! aerosol mass mixing ratio
   408) 
   409)    real(r8) :: rho(pcols,pver)     ! air density (kg m-3)
   410)    real(r8) :: relhum(pcols,pver)  ! relative humidity
   411)    real(r8) :: icldm(pcols,pver)   ! ice cloud fraction
   412)    real(r8) :: lcldm(pcols,pver)   ! liq cloud fraction
   413)    real(r8) :: nfice(pcols,pver)   ! fice variable
   414)    real(r8) :: dumfice             ! dummy var in fice calc
   415)    real(r8) :: lcldn(pcols,pver)   ! fractional coverage of new liquid cloud
   416)    real(r8) :: lcldo(pcols,pver)   ! fractional coverage of old liquid cloud
   417)    real(r8) :: qcld                ! total cloud water
   418)    real(r8) :: nctend_mixnuc(pcols,pver)
   419)    real(r8) :: dum, dum2           ! temporary dummy variable
   420)    real(r8) :: dmc, ssmc           ! variables for modal scheme.
   421) 
   422)    real(r8) :: so4_num                               ! so4 aerosol number (#/cm^3)
   423)    real(r8) :: soot_num                              ! soot (hydrophilic) aerosol number (#/cm^3)
   424)    real(r8) :: dst1_num,dst2_num,dst3_num,dst4_num   ! dust aerosol number (#/cm^3)
   425)    real(r8) :: dst_num                               ! total dust aerosol number (#/cm^3)
   426) 
   427)    real(r8) :: qs(pcols)            ! liquid-ice weighted sat mixing rat (kg/kg)
   428)    real(r8) :: es(pcols)            ! liquid-ice weighted sat vapor press (pa)
   429)    real(r8) :: gammas(pcols)        ! parameter for cond/evap of cloud water
   430) 
   431)    ! bulk aerosol variables
   432)    real(r8), allocatable :: naer2(:,:,:)    ! bulk aerosol number concentration (1/m3)
   433)    real(r8), allocatable :: maerosol(:,:,:) ! bulk aerosol mass conc (kg/m3)
   434) 
   435)    real(r8) :: wsub(pcols,pver)    ! diagnosed sub-grid vertical velocity st. dev. (m/s)
   436)    real(r8) :: wsubi(pcols,pver)   ! diagnosed sub-grid vertical velocity ice (m/s)
   437) 
   438)    ! history output for ice nucleation
   439)    real(r8) :: nihf(pcols,pver)  !output number conc of ice nuclei due to heterogenous freezing (1/m3)
   440)    real(r8) :: niimm(pcols,pver) !output number conc of ice nuclei due to immersion freezing (hetero nuc) (1/m3)
   441)    real(r8) :: nidep(pcols,pver) !output number conc of ice nuclei due to deoposion nucleation (hetero nuc) (1/m3)
   442)    real(r8) :: nimey(pcols,pver) !output number conc of ice nuclei due to meyers deposition (1/m3)
   443) 
   444)    real(r8) :: wght
   445) 
   446)    !-------------------------------------------------------------------------------
   447) 
   448)    lchnk = state%lchnk
   449)    ncol  = state%ncol
   450)    t     => state%t
   451)    qn    => state%q(:,:,1)
   452)    qc    => state%q(:,:,cldliq_idx)
   453)    qi    => state%q(:,:,cldice_idx)
   454)    nc    => state%q(:,:,numliq_idx)
   455)    ni    => state%q(:,:,numice_idx)
   456)    pmid  => state%pmid
   457)    pdel  => state%pdel
   458)    pint  => state%pint
   459)    rpdel => state%rpdel
   460)    zm    => state%zm
   461)    omega => state%omega
   462) 
   463)    itim_old = pbuf_old_tim_idx()
   464)    call pbuf_get_field(pbuf, ast_idx,      ast, start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
