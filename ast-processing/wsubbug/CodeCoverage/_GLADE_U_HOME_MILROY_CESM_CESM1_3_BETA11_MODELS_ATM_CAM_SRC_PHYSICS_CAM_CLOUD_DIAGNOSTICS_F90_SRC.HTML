    49)   subroutine cloud_diagnostics_register
    50) 
    51)     use phys_control,  only: phys_getopts
    52)     use physics_buffer,only: pbuf_add_field, dtype_r8, dtype_i4
    53) 
    54)     character(len=16) :: rad_pkg, microp_pgk
    55) 
    56)     call phys_getopts(radiation_scheme_out=rad_pkg,microp_scheme_out=microp_pgk)
    57)     camrt_rad = rad_pkg .eq. 'camrt'
    58)     rk_clouds = microp_pgk == 'RK'
    59)     mg_clouds = microp_pgk == 'MG'
    60) 
    61)     if (rk_clouds) then
    62)        call pbuf_add_field('CLDEMIS','physpkg', dtype_r8,(/pcols,pver/), cldemis_idx)
    63)        call pbuf_add_field('CLDTAU', 'physpkg', dtype_r8,(/pcols,pver/), cldtau_idx)
    64) 
    65)        call pbuf_add_field('CICEWP', 'physpkg', dtype_r8,(/pcols,pver/), cicewp_idx)
    66)        call pbuf_add_field('CLIQWP', 'physpkg', dtype_r8,(/pcols,pver/), cliqwp_idx)
    67) 
    68)        call pbuf_add_field('PMXRGN', 'physpkg', dtype_r8,(/pcols,pverp/), pmxrgn_idx)
    69)        call pbuf_add_field('NMXRGN', 'physpkg', dtype_i4,(/pcols /),      nmxrgn_idx)
    70)     else if (mg_clouds) then
    71)        ! In cloud ice water path for radiation
    72)        call pbuf_add_field('ICIWP',      'global', dtype_r8,(/pcols,pver/), iciwp_idx)
    73)        ! In cloud liquid water path for radiation
    74)        call pbuf_add_field('ICLWP',      'global', dtype_r8,(/pcols,pver/), iclwp_idx)
    75)     endif
    76)   end subroutine cloud_diagnostics_register
    77) 
    78) !===============================================================================
    79)   subroutine cloud_diagnostics_init()
    80) !-----------------------------------------------------------------------
    81)     use physics_buffer,only: pbuf_get_index
    82)     use phys_control,  only: phys_getopts
    83)     use constituents,  only: cnst_get_ind
    84)     use cloud_cover_diags, only: cloud_cover_diags_init
    85) 
    86)     implicit none
    87) 
    88) !-----------------------------------------------------------------------
    89) 
    90)     character(len=16) :: wpunits, sampling_seq
    91)     logical           :: history_amwg                  ! output the variables used by the AMWG diag package
    92) 
    93) 
    94)     !-----------------------------------------------------------------------
    95) 
    96)     cld_idx    = pbuf_get_index('CLD')
    97) 
    98)     if (mg_clouds) then
    99) 
   100)        call addfld ('ICWMR    ', 'kg/kg   ', pver, 'A', 'Prognostic in-cloud water mixing ratio'                  ,phys_decomp)
   101)        call addfld ('ICIMR    ', 'kg/kg   ', pver, 'A', 'Prognostic in-cloud ice mixing ratio'                    ,phys_decomp)
   102)        call addfld ('IWC      ', 'kg/m3   ', pver, 'A', 'Grid box average ice water content'                      ,phys_decomp)
   103)        call addfld ('LWC      ', 'kg/m3   ', pver, 'A', 'Grid box average liquid water content'                   ,phys_decomp)
   104) 
   105)        ! determine the add_default fields
   106)        call phys_getopts(history_amwg_out           = history_amwg) 
   107) 
   108)        if (history_amwg) then
   109)           call add_default ('ICWMR', 1, ' ')
   110)           call add_default ('ICIMR', 1, ' ')
   111)           call add_default ('IWC      ', 1, ' ')
   112)        end if
   113) 
   114)        dei_idx    = pbuf_get_index('DEI')
   115)        mu_idx     = pbuf_get_index('MU')
   116)        lambda_idx = pbuf_get_index('LAMBDAC')
   117) 
   118)     elseif (rk_clouds) then
   119) 
   120)        rei_idx    = pbuf_get_index('REI')
   121)        rel_idx    = pbuf_get_index('REL')
   122) 
   123)     endif
   124) 
   125)     call cnst_get_ind('CLDICE', ixcldice)
   126)     call cnst_get_ind('CLDLIQ', ixcldliq)
   127) 
   128)     do_cld_diag = rk_clouds .or. mg_clouds
   129) 
   130)     if (.not.do_cld_diag) return
   131)     
   132)     if (rk_clouds) then 
   133)        wpunits = 'gram/m2'
   134)        sampling_seq='rad_lwsw'
   135)     else if (mg_clouds) then 
   136)        wpunits = 'kg/m2'
   137)        sampling_seq=''
   138)     endif
   139) 
   140)     call addfld ('ICLDIWP', wpunits, pver, 'A','In-cloud ice water path'               ,phys_decomp, sampling_seq=sampling_seq)
   141)     call addfld ('ICLDTWP ',wpunits, pver, 'A','In-cloud cloud total water path (liquid and ice)',phys_decomp, &
   142)          sampling_seq=sampling_seq)
   143) 
   144)     call addfld ('GCLDLWP ',wpunits,pver, 'A','Grid-box cloud water path'             ,phys_decomp, &
   145)          sampling_seq=sampling_seq)
   146)     call addfld ('TGCLDCWP',wpunits,1,    'A','Total grid-box cloud water path (liquid and ice)',phys_decomp, &
   147)          sampling_seq=sampling_seq)
   148)     call addfld ('TGCLDLWP',wpunits,1,    'A','Total grid-box cloud liquid water path',phys_decomp, &
   149)          sampling_seq=sampling_seq)
   150)     call addfld ('TGCLDIWP',wpunits,1,    'A','Total grid-box cloud ice water path'   ,phys_decomp, &
   151)          sampling_seq=sampling_seq)
   152)     
   153)     if(mg_clouds) then
   154)        call addfld ('lambda_cloud','1/meter',pver,'I','lambda in cloud', phys_decomp)
   155)        call addfld ('mu_cloud','1',pver,'I','mu in cloud', phys_decomp)
   156)        call addfld ('dei_cloud','micrometers',pver,'I','ice radiative effective diameter in cloud', phys_decomp)
   157)     endif
   158) 
   159)     if(rk_clouds) then
   160)        call addfld ('rel_cloud','1/meter',pver,'I','effective radius of liq in cloud', phys_decomp, sampling_seq=sampling_seq)
   161)        call addfld ('rei_cloud','1',pver,'I','effective radius of ice in cloud', phys_decomp, sampling_seq=sampling_seq)
   162)     endif
   163) 
   164)     call addfld ('SETLWP  ','gram/m2 ',pver, 'A','Prescribed liquid water path'          ,phys_decomp, sampling_seq=sampling_seq)
   165)     call addfld ('LWSH    ','m       ',1,    'A','Liquid water scale height'             ,phys_decomp, sampling_seq=sampling_seq)
   166) 
   167)     call addfld ('EFFCLD  ','fraction',pver, 'A','Effective cloud fraction'              ,phys_decomp, sampling_seq=sampling_seq)
   168) 
   169)     if (camrt_rad) then
   170)        call addfld ('EMIS', '1', pver, 'A','cloud emissivity'                      ,phys_decomp, sampling_seq=sampling_seq)
   171)     else
   172)        call addfld ('EMISCLD', '1', pver, 'A','cloud emissivity'                      ,phys_decomp, sampling_seq=sampling_seq)
   173)     endif
   174) 
   175)     call cloud_cover_diags_init(sampling_seq)
   176) 
   177)     ! ----------------------------
   178)     ! determine default variables
   179)     ! ----------------------------
   180)     call phys_getopts( history_amwg_out = history_amwg)
   181) 
   182)     if (history_amwg) then
   183)        call add_default ('TGCLDLWP', 1, ' ')
   184)        call add_default ('TGCLDIWP', 1, ' ')
   185)        call add_default ('TGCLDCWP', 1, ' ')
   186)        if (camrt_rad) then
   187)            call add_default ('EMIS', 1, ' ')
   188)        else
   189)            call add_default ('EMISCLD', 1, ' ')
   190)        endif
   191)     endif
   192) 
   193)     return
   194)   end subroutine cloud_diagnostics_init
   195) 
   196) subroutine cloud_diagnostics_calc(state,  pbuf)
   197) !===============================================================================
   198) !
   199) ! Compute (liquid+ice) water path and cloud water/ice diagnostics
   200) ! *** soon this code will compute liquid and ice paths from input liquid and ice mixing ratios
   201) ! 
   202) ! **** mixes interface and physics code temporarily
   203) !-----------------------------------------------------------------------
   204)     use physics_types, only: physics_state    
   205)     use physics_buffer,only: physics_buffer_desc, pbuf_get_field, pbuf_old_tim_idx
   206)     use pkg_cldoptics, only: cldovrlap, cldclw,  cldems
   207)     use conv_water,    only: conv_water_in_rad, conv_water_4rad
   208)     use radiation,     only: radiation_do
   209)     use cloud_cover_diags, only: cloud_cover_diags_out
   210) 
   211)     use ref_pres,       only: top_lev=>trop_cloud_top_lev
   212) 
   213)     implicit none
   214) 
   215) ! Arguments
   216)     type(physics_state), intent(in)    :: state        ! state variables
   217)     type(physics_buffer_desc), pointer :: pbuf(:)
   218) 
   219) ! Local variables
   220) 
   221)     real(r8), pointer :: cld(:,:)       ! cloud fraction
   222)     real(r8), pointer :: iciwp(:,:)   ! in-cloud cloud ice water path
   223)     real(r8), pointer :: iclwp(:,:)   ! in-cloud cloud liquid water path
   224)     real(r8), pointer :: dei(:,:)       ! effective radiative diameter of ice
   225)     real(r8), pointer :: mu(:,:)        ! gamma distribution for liq clouds
   226)     real(r8), pointer :: lambda(:,:)    ! gamma distribution for liq clouds
   227)     real(r8), pointer :: rei(:,:)       ! effective radiative radius of ice
   228)     real(r8), pointer :: rel(:,:)       ! effective radiative radius of liq
   229) 
   230)     real(r8), pointer :: cldemis(:,:)   ! cloud emissivity
   231)     real(r8), pointer :: cldtau(:,:)    ! cloud optical depth
   232)     real(r8), pointer :: cicewp(:,:)    ! in-cloud cloud ice water path
   233)     real(r8), pointer :: cliqwp(:,:)    ! in-cloud cloud liquid water path
   234) 
   235)     integer,  pointer :: nmxrgn(:)      ! Number of maximally overlapped regions
   236)     real(r8), pointer :: pmxrgn(:,:)    ! Maximum values of pressure for each
   237) 
   238)     integer :: itim_old
   239) 
   240)     real(r8) :: cwp   (pcols,pver)      ! in-cloud cloud (total) water path
   241)     real(r8) :: gicewp(pcols,pver)      ! grid-box cloud ice water path
   242)     real(r8) :: gliqwp(pcols,pver)      ! grid-box cloud liquid water path
   243)     real(r8) :: gwp   (pcols,pver)      ! grid-box cloud (total) water path
   244)     real(r8) :: tgicewp(pcols)          ! Vertically integrated ice water path
   245)     real(r8) :: tgliqwp(pcols)          ! Vertically integrated liquid water path
   246)     real(r8) :: tgwp   (pcols)          ! Vertically integrated (total) cloud water path
   247) 
   248)     real(r8) :: ficemr (pcols,pver)     ! Ice fraction from ice and liquid mixing ratios
   249) 
   250)     real(r8) :: icimr(pcols,pver)       ! In cloud ice mixing ratio
   251)     real(r8) :: icwmr(pcols,pver)       ! In cloud water mixing ratio
   252)     real(r8) :: iwc(pcols,pver)         ! Grid box average ice water content
   253)     real(r8) :: lwc(pcols,pver)         ! Grid box average liquid water content
   254) 
   255) ! old data
   256)     real(r8) :: tpw    (pcols)          ! total precipitable water
   257)     real(r8) :: clwpold(pcols,pver)     ! Presribed cloud liq. h2o path
   258)     real(r8) :: hl     (pcols)          ! Liquid water scale height
   259) 
   260)     integer :: i,k                      ! loop indexes
   261)     integer :: ncol, lchnk
   262)     real(r8) :: rgrav
   263) 
   264)     real(r8) :: allcld_ice (pcols,pver) ! Convective cloud ice
   265)     real(r8) :: allcld_liq (pcols,pver) ! Convective cloud liquid
   266) 
   267)     real(r8) :: effcld(pcols,pver)      ! effective cloud=cld*emis
   268) 
   269)     logical :: dosw,dolw
   270)   
   271) !-----------------------------------------------------------------------
   272)     if (.not.do_cld_diag) return
   273) 
   274) 
   275)     if(rk_clouds) then
   276)        dosw     = radiation_do('sw')      ! do shortwave heating calc this timestep?
   277)        dolw     = radiation_do('lw')      ! do longwave heating calc this timestep?
   278)     else
   279)        dosw     = .true.
   280)        dolw     = .true.
   281)     endif
   282) 
   283)     if (.not.(dosw .or. dolw)) return
   284) 
   285)     ncol  = state%ncol
   286)     lchnk = state%lchnk
   287) 
   288)     itim_old = pbuf_old_tim_idx()
   289)     call pbuf_get_field(pbuf, cld_idx, cld, start=(/1,1,itim_old/), kount=(/pcols,pver,1/) )
   290) 
   291)     if(mg_clouds)then
   292) 
   293)        call pbuf_get_field(pbuf, iclwp_idx, iclwp )
   294)        call pbuf_get_field(pbuf, iciwp_idx, iciwp )
   295)        call pbuf_get_field(pbuf, dei_idx, dei )
   296)        call pbuf_get_field(pbuf, mu_idx, mu )
   297)        call pbuf_get_field(pbuf, lambda_idx, lambda )
   298) 
   299)        call outfld('dei_cloud',dei(:,:),pcols,lchnk)
   300)        call outfld('mu_cloud',mu(:,:),pcols,lchnk)
   301)        call outfld('lambda_cloud',lambda(:,:),pcols,lchnk)
   302) 
   303)     elseif(rk_clouds) then
   304) 
   305)        call pbuf_get_field(pbuf, rei_idx, rei )
   306)        call pbuf_get_field(pbuf, rel_idx, rel )
   307) 
   308)        call outfld('rel_cloud', rel, pcols, lchnk)
   309)        call outfld('rei_cloud', rei, pcols, lchnk)
   310) 
   311)        if (cldemis_idx>0) then
   312)           call pbuf_get_field(pbuf, cldemis_idx, cldemis )
   313)        else
   314)           allocate(cldemis(pcols,pver))
   315)        endif
   316)        if (cldtau_idx>0) then
   317)           call pbuf_get_field(pbuf, cldtau_idx, cldtau )
   318)        else
   319)           allocate(cldtau(pcols,pver))
   320)        endif
   321) 
   322)     endif
   323) 
   324)     if (cicewp_idx>0) then
   325)        call pbuf_get_field(pbuf, cicewp_idx, cicewp )
   326)     else
   327)        allocate(cicewp(pcols,pver))
   328)     endif
   329)     if (cliqwp_idx>0) then
   330)        call pbuf_get_field(pbuf, cliqwp_idx, cliqwp )
   331)     else
   332)        allocate(cliqwp(pcols,pver))
   333)     endif
   334) 
   335)     if (nmxrgn_idx>0) then
   336)        call pbuf_get_field(pbuf, nmxrgn_idx, nmxrgn )
   337)     else
   338)        allocate(nmxrgn(pcols))
   339)     endif
   340) 
   341)     if (pmxrgn_idx>0) then
   342)        call pbuf_get_field(pbuf, pmxrgn_idx, pmxrgn )
   343)     else
   344)        allocate(pmxrgn(pcols,pverp))
   345)     endif
   346) 
   347) ! Compute liquid and ice water paths
   348)     if(mg_clouds) then
   349) 
   350)        ! ----------------------------------------------------------- !
   351)        ! Adjust in-cloud water values to take account of convective  !
   352)        ! in-cloud water. It is used to calculate the values of       !
   353)        ! iclwp and iciwp to pass to the radiation.                   !
   354)        ! ----------------------------------------------------------- !
   355)        if( conv_water_in_rad /= 0 ) then
   356)           allcld_ice(:ncol,:) = 0._r8 ! Grid-avg all cloud liquid
   357)           allcld_liq(:ncol,:) = 0._r8 ! Grid-avg all cloud ice
   358)     
   359)           call conv_water_4rad(state, pbuf, allcld_liq, allcld_ice)
   360)        else
   361)           allcld_liq(:ncol,top_lev:pver) = state%q(:ncol,top_lev:pver,ixcldliq)  ! Grid-ave all cloud liquid
   362)           allcld_ice(:ncol,top_lev:pver) = state%q(:ncol,top_lev:pver,ixcldice)  !           "        ice
   363)        end if
   364) 
   365)        ! ------------------------------------------------------------ !
   366)        ! Compute in cloud ice and liquid mixing ratios                !
   367)        ! Note that 'iclwp, iciwp' are used for radiation computation. !
   368)        ! ------------------------------------------------------------ !
   369) 
   370) 
   371)        iciwp = 0._r8
   372)        iclwp = 0._r8
   373)        icimr = 0._r8
   374)        icwmr = 0._r8
   375)        iwc = 0._r8
   376)        lwc = 0._r8
   377) 
   378)        do k = top_lev, pver
   379)           do i = 1, ncol
   380)              ! Limits for in-cloud mixing ratios consistent with MG microphysics
   381)              ! in-cloud mixing ratio maximum limit of 0.005 kg/kg
   382)              icimr(i,k)     = min( allcld_ice(i,k) / max(0.0001_r8,cld(i,k)),0.005_r8 )
   383)              icwmr(i,k)     = min( allcld_liq(i,k) / max(0.0001_r8,cld(i,k)),0.005_r8 )
   384)              iwc(i,k)       = allcld_ice(i,k) * state%pmid(i,k) / (287.15_r8*state%t(i,k))
   385)              lwc(i,k)       = allcld_liq(i,k) * state%pmid(i,k) / (287.15_r8*state%t(i,k))
   386)              ! Calculate total cloud water paths in each layer
   387)              iciwp(i,k)     = icimr(i,k) * state%pdel(i,k) / gravit
   388)              iclwp(i,k)     = icwmr(i,k) * state%pdel(i,k) / gravit
   389)           end do
   390)        end do
   391) 
   392)        do k=1,pver
   393)           do i = 1,ncol
   394)              gicewp(i,k) = iciwp(i,k)*cld(i,k)
   395)              gliqwp(i,k) = iclwp(i,k)*cld(i,k)
   396)              cicewp(i,k) = iciwp(i,k)
   397)              cliqwp(i,k) = iclwp(i,k)
   398)           end do
   399)        end do
   400) 
   401)     elseif(rk_clouds) then
   402) 
   403)        if (conv_water_in_rad /= 0) then
   404)           call conv_water_4rad(state, pbuf, allcld_liq, allcld_ice)
   405)        else
   406)           allcld_liq = state%q(:,:,ixcldliq)
   407)           allcld_ice = state%q(:,:,ixcldice)
   408)        end if
   409)     
   410)        do k=1,pver
   411)           do i = 1,ncol
   412)              gicewp(i,k) = allcld_ice(i,k)*state%pdel(i,k)/gravit*1000.0_r8  ! Grid box ice water path.
   413)              gliqwp(i,k) = allcld_liq(i,k)*state%pdel(i,k)/gravit*1000.0_r8  ! Grid box liquid water path.
   414)              cicewp(i,k) = gicewp(i,k) / max(0.01_r8,cld(i,k))               ! In-cloud ice water path.
   415)              cliqwp(i,k) = gliqwp(i,k) / max(0.01_r8,cld(i,k))               ! In-cloud liquid water path.
   416)              ficemr(i,k) = allcld_ice(i,k) / max(1.e-10_r8,(allcld_ice(i,k) + allcld_liq(i,k)))
   417)           end do
   418)        end do
   419)     endif
   420) 
   421) ! Determine parameters for maximum/random overlap
   422)     call cldovrlap(lchnk, ncol, state%pint, cld, nmxrgn, pmxrgn)
   423) 
   424) ! Cloud cover diagnostics (done in radiation_tend for camrt)
   425)     if (.not.camrt_rad) then
   426)        call cloud_cover_diags_out(lchnk, ncol, cld, state%pmid, nmxrgn, pmxrgn )
   427)     endif
   428)     
   429)     tgicewp(:ncol) = 0._r8
   430)     tgliqwp(:ncol) = 0._r8
   431) 
   432)     do k=1,pver
   433)        tgicewp(:ncol)  = tgicewp(:ncol) + gicewp(:ncol,k)
   434)        tgliqwp(:ncol)  = tgliqwp(:ncol) + gliqwp(:ncol,k)
   435)     end do
   436) 
   437)     tgwp(:ncol) = tgicewp(:ncol) + tgliqwp(:ncol)
   438)     gwp(:ncol,:pver) = gicewp(:ncol,:pver) + gliqwp(:ncol,:pver)
   439)     cwp(:ncol,:pver) = cicewp(:ncol,:pver) + cliqwp(:ncol,:pver)
   440) 
   441)     if(rk_clouds) then
   442) 
   443)        ! Cloud emissivity.
   444)        call cldems(lchnk, ncol, cwp, ficemr, rei, cldemis, cldtau)
   445)        
   446)        ! Effective cloud cover
   447)        do k=1,pver
   448)           do i=1,ncol
   449)              effcld(i,k) = cld(i,k)*cldemis(i,k)
   450)           end do
   451)        end do
   452)        
   453)        call outfld('EFFCLD'  ,effcld , pcols,lchnk)
   454)        if (camrt_rad) then
   455)           call outfld('EMIS' ,cldemis, pcols,lchnk)
   456)        else
   457)           call outfld('EMISCLD' ,cldemis, pcols,lchnk)
   458)        endif
   459) 
   460)     else if (mg_clouds) then
   461) 
   462)        ! --------------------------------------------- !
   463)        ! General outfield calls for microphysics       !
   464)        ! --------------------------------------------- !
   465) 
   466)        call outfld( 'IWC'      , iwc,         pcols, lchnk )
   467)        call outfld( 'LWC'      , lwc,         pcols, lchnk )
   468)        call outfld( 'ICIMR'    , icimr,       pcols, lchnk )
