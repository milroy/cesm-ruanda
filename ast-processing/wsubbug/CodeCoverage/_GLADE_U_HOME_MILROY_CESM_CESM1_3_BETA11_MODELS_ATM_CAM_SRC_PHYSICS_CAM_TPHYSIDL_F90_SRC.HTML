     3) !----------------------------------------------------------------------- 
     4) ! 
     5) ! Purpose: 
     6) !  algorithm 1: Held/Suarez IDEALIZED physics
     7) !  algorithm 2: Held/Suarez IDEALIZED physics (Williamson modified stratosphere
     8) !  algorithm 3: Held/Suarez IDEALIZED physics (Lin/Williamson modified strato/meso-sphere
     9) !  algorithm 4: Boer/Denis  IDEALIZED physics

    14)    use shr_kind_mod,       only: r8 => shr_kind_r8

    22)    use cam_history,        only: outfld
    23)    use cam_logfile,        only: iulog

    32)    real(r8), intent(in) :: ztodt                   ! Two times model timestep (2 delta-t)

    34)    type(physics_state),  intent(inout) :: state

    36) ! Output arguments

    46)    integer :: nstep                                ! timestep number

    48)    real(r8) clat(pcols)                        ! latitudes(radians) for columns
    49)    real(r8) pmid(pcols,pver)                   ! mid-point pressure
    50)    integer  i,k                                ! Longitude, level indices
    51) 
    52)    real(r8) tmp                                ! temporary
    53)    real(r8) kf                                 ! 1./efolding_time for wind dissipation
    54)    real(r8) ka                                 ! 1./efolding_time for temperature diss.
    55)    real(r8) kaa                                ! 1./efolding_time for temperature diss.
    56)    real(r8) ks                                 ! 1./efolding_time for temperature diss.
    57)    real(r8) kv                                 ! 1./efolding_time (normalized) for wind
    58)    real(r8) kt                                 ! 1./efolding_time for temperature diss.
    59)    real(r8) trefa                              ! "radiative equilibrium" T
    60)    real(r8) trefc                              ! used in calc of "radiative equilibrium" T
    61)    real(r8) cossq(pcols)                       ! coslat**2
    62)    real(r8) cossqsq(pcols)                     ! coslat**4
    63)    real(r8) sinsq(pcols)                       ! sinlat**2
    64)    real(r8) onemsig                            ! 1. - sigma_reference
    65)    real(r8) efoldf                             ! efolding time for wind dissipation
    66)    real(r8) efolda                             ! efolding time for T dissipation
    67)    real(r8) efoldaa                            ! efolding time for T dissipation
    68)    real(r8) efolds                             ! efolding time for T dissipation
    69)    real(r8) efold_strat                        ! efolding time for T dissipation in Strat
    70)    real(r8) efold_meso                         ! efolding time for T dissipation in Meso
    71)    real(r8) efoldv                             ! efolding time for wind dissipation
    72)    real(r8) p_infint                           ! effective top of model
    73)    real(r8) constw                             ! constant
    74)    real(r8) lapsew                             ! lapse rate
    75)    real(r8) p0strat                            ! threshold pressure
    76)    real(r8) phi0                               ! threshold latitude
    77)    real(r8) dphi0                              ! del-latitude
    78)    real(r8) a0                                 ! coefficient
    79)    real(r8) aeq                                ! 100 mb
    80)    real(r8) apole                              ! 2   mb
    81)    real(r8) pi                                 ! 3.14159...
    82)    real(r8) coslat(pcols)                      ! cosine(latitude)
    83)    real(r8) acoslat                            ! abs(acos(coslat))
    84)    real(r8) constc                             ! constant
    85)    real(r8) lapsec                             ! lapse rate
    86)    real(r8) lapse                              ! lapse rate
    87)    real(r8) h0                                 ! scale height (7 km)
    88)    real(r8) sigmab                             ! threshold sigma level
    89)    real(r8) pressmb                            ! model pressure in mb
    90)    real(r8) t00                                ! minimum reference temperature
    91)    integer  idlflag                            ! Flag to choose which idealized physics
    92)    real(r8) :: zero(pcols)
    93) !
    94) !-----------------------------------------------------------------------
    95) !
    96)    idlflag = 1
    97) 
    98)    lchnk = state%lchnk
    99)    ncol  = state%ncol
   100) 
   101)    nstep   = get_nstep()
   102)    zero    = 0._r8
   103) 
   104)    call get_rlat_all_p(lchnk, ncol, clat)
   105)    do i=1,ncol
   106)       coslat (i) = cos(clat(i))
   107)       sinsq  (i) = sin(clat(i))*sin(clat(i))
   108)       cossq  (i) = coslat(i)*coslat(i)
   109)       cossqsq(i) = cossq (i)*cossq (i)
   110)    end do
   111)    do k = 1, pver
   112)       do i = 1, ncol
   113)          pmid(i,k) = state%pmid(i,k)
   114)       end do
   115)    end do
   116) 
   117)    ! initialize individual parameterization tendencies
   118)    call physics_ptend_init(ptend, state%psetcols, 'tphysidl', ls=.true., lu=.true., lv=.true.)
   119) 
   120)    if (idlflag == 1) then
   121) !
   122) !-----------------------------------------------------------------------
   123) !
   124) ! Held/Suarez IDEALIZED physics algorithm:
   125) !
   126) !   Held, I. M., and M. J. Suarez, 1994: A proposal for the
   127) !   intercomparison of the dynamical cores of atmospheric general
   128) !   circulation models.
   129) !   Bulletin of the Amer. Meteor. Soc., vol. 75, pp. 1825-1830.
   130) !
   131) !-----------------------------------------------------------------------
   132) !
   133) ! Compute idealized radiative heating rates (as dry static energy)
   134) !
   135)       efoldf =  1._r8
   136)       efolda = 40._r8
   137)       efolds =  4._r8
   138)       sigmab =  0.7_r8
   139)       t00    = 200._r8
   140) !
   141)       onemsig = 1._r8 - sigmab
   142) !
   143)       ka = 1._r8/(86400._r8*efolda)
   144)       ks = 1._r8/(86400._r8*efolds)
   145) !
   146)       do k=1,pver
   147)          if (pref_mid_norm(k) > sigmab) then
   148)             do i=1,ncol
   149)                kt = ka + (ks - ka)*cossqsq(i)*(pref_mid_norm(k) - sigmab)/onemsig
   150) #ifdef MODHS
   151)                tmp   = kt/(1._r8+ ztodt*kt)
   152) #else
   153) 	       tmp = kt
   154) #endif
   155)                trefc   = 315._r8 - 60._r8*sinsq(i)
   156)                trefa = (trefc - 10._r8*cossq(i)*log((pmid(i,k)/psurf_ref)))*(pmid(i,k)/psurf_ref)**cappa
   157)                trefa    = max(t00,trefa)
   158)                ptend%s(i,k) = (trefa - state%t(i,k))*tmp*cpair
   159)             end do
   160)          else
   161) #ifdef MODHS
   162)             tmp   = ka/(1._r8+ ztodt*ka)
   163) #else
   164) 	    tmp = ka
   165) #endif
   166)             do i=1,ncol
   167)                trefc   = 315._r8 - 60._r8*sinsq(i)
   168)                trefa = (trefc - 10._r8*cossq(i)*log((pmid(i,k)/psurf_ref)))*(pmid(i,k)/psurf_ref)**cappa
   169)                trefa    = max(t00,trefa)
   170)                ptend%s(i,k) = (trefa - state%t(i,k))*tmp*cpair
   171)             end do
   172)          endif
   173)       end do
   174) !
   175) ! Add diffusion near the surface for the wind fields
   176) !
   177)       do k=1,pver
   178)          do i=1,pcols
   179)             ptend%u(i,k) = 0._r8
   180)             ptend%v(i,k) = 0._r8
   181)          end do
   182)       end do
   183) 
   184)       kf = 1._r8/(86400._r8*efoldf)
   185) !
   186)       do k=1,pver
   187)          if (pref_mid_norm(k) > sigmab) then
   188)             kv  = kf*(pref_mid_norm(k) - sigmab)/onemsig
   189) #ifdef MODHS
   190)             tmp = -kv/(1._r8+ ztodt*kv)
   191) #else
   192)             tmp = -kv
   193) #endif
   194)             do i=1,ncol
   195)                ptend%u(i,k) = tmp*state%u(i,k)
   196)                ptend%v(i,k) = tmp*state%v(i,k)
   197)             end do
   198)          endif
   199)       end do
   200) 
   201)    elseif (idlflag == 2) then
   202) !
   203) !-----------------------------------------------------------------------
   204) !
   205) ! Modified Held/Suarez IDEALIZED physics algorithm
   206) ! (modified with Williamson stratosphere):
   207) !
   208) !   Williamson, D. L., J. G. Olson and B. A. Boville, 1998: A comparison
   209) !   of semi--Lagrangian and Eulerian tropical climate simulations.
   210) !   Mon. Wea. Rev., vol 126, pp. 1001-1012.
   211) !
   212) !-----------------------------------------------------------------------
   213) !
   214) ! Compute idealized radiative heating rates (as dry static energy)
   215) !
   216)       efoldf  =  1._r8
   217)       efolda  = 40._r8
   218)       efoldaa = 40._r8
   219)       efolds  =  4._r8
   220)       sigmab  =  0.7_r8
   221)       t00     = 200._r8
   222) !
   223)       onemsig = 1._r8 - sigmab
   224) !
   225)       ka  = 1._r8/(86400._r8*efolda)
   226)       kaa = 1._r8/(86400._r8*efoldaa)
   227)       ks  = 1._r8/(86400._r8*efolds)
   228) !
   229)       pi     = 4._r8*atan(1._r8)
   230)       phi0   = 60._r8*pi/180._r8
   231)       dphi0  = 15._r8*pi/180._r8
   232)       a0     = 2.65_r8/dphi0
   233)       aeq    = 10000._r8
   234)       apole  = 200._r8
   235)       lapsew = -3.345e-03_r8
   236)       constw = rair*lapsew/gravit
   237)       lapsec =  2.00e-03_r8
   238)       constc = rair*lapsec/gravit
   239)       do k=1,pver
   240)          if (pref_mid_norm(k) > sigmab) then
   241)             do i=1,ncol
   242)                kt = ka + (ks - ka)*cossqsq(i)*(pref_mid_norm(k) - sigmab)/onemsig
   243)                acoslat = abs(acos(coslat(i)))
   244)                p0strat = aeq - (aeq - apole)*0.5_r8*(1._r8 + tanh(a0*(acoslat - phi0)))
   245)                tmp     = kt/(1._r8+ ztodt*kt)
   246)                trefc   = 315._r8 - 60._r8*sinsq(i)
   247)                trefa   = (trefc - 10._r8*cossq(i)*log((pmid(i,k)/psurf_ref)))*(pmid(i,k)/psurf_ref)** &
   248)                                                                                      cappa
   249)                trefa   = max(t00,trefa)
   250)                if (pmid(i,k) < 10000._r8) then
   251)                   trefa = t00*((pmid(i,k)/10000._r8))**constc
   252)                   tmp   = kaa/(1._r8+ ztodt*kaa)
   253)                endif
   254)                if (pmid(i,k) < p0strat) then
   255)                   trefa = trefa + t00*( ((pmid(i,k)/p0strat))**constw - 1._r8 )
   256)                   tmp   = kaa/(1._r8+ ztodt*kaa)
   257)                endif
   258)                ptend%s(i,k) = (trefa - state%t(i,k))*tmp*cpair
   259)             end do
   260)          else
   261)             do i=1,ncol
   262)                acoslat = abs(acos(coslat(i)))
   263)                p0strat = aeq - (aeq - apole)*0.5_r8*(1._r8 + tanh(a0*(acoslat - phi0)))
   264)                tmp     = ka/(1._r8+ ztodt*ka)
   265)                trefc   = 315._r8 - 60._r8*sinsq(i)
   266)                trefa   = (trefc - 10._r8*cossq(i)*log((pmid(i,k)/psurf_ref)))*(pmid(i,k)/psurf_ref)** &
   267)                                                                                      cappa
   268)                trefa   = max(t00,trefa)
   269)                if (pmid(i,k) < 10000._r8) then
   270)                   trefa = t00*((pmid(i,k)/10000._r8))**constc
   271)                   tmp   = kaa/(1._r8+ ztodt*kaa)
   272)                endif
   273)                if (pmid(i,k) < p0strat) then
   274)                   trefa = trefa + t00*( ((pmid(i,k)/p0strat))**constw - 1._r8 )
   275)                   tmp   = kaa/(1._r8+ ztodt*kaa)
   276)                endif
   277)                ptend%s(i,k) = (trefa - state%t(i,k))*tmp*cpair
   278)             end do
   279)          endif
   280)       end do
   281) !
   282) ! Add diffusion near the surface for the wind fields
   283) !
   284)       do k=1,pver
   285)          do i=1,pcols
   286)             ptend%u(i,k) = 0._r8
   287)             ptend%v(i,k) = 0._r8
   288)          end do
   289)       end do
   290) 
   291)       kf = 1._r8/(86400._r8*efoldf)
   292) !
   293)       do k=1,pver
   294)          if (pref_mid_norm(k) > sigmab) then
   295)             kv  = kf*(pref_mid_norm(k) - sigmab)/onemsig
   296)             tmp = -kv/(1._r8+ ztodt*kv)
   297)             do i=1,ncol
   298)                ptend%u(i,k) = tmp*state%u(i,k)
   299)                ptend%v(i,k) = tmp*state%v(i,k)
   300)             end do
   301)          endif
   302)       end do
   303) 
   304)    elseif (idlflag == 3) then
   305) !
   306) !-----------------------------------------------------------------------
   307) !
   308) ! Held/Suarez IDEALIZED physics algorithm:
   309) ! (modified with Lin/Williamson stratosphere/mesosphere):
   310) !
   311) !   Held, I. M., and M. J. Suarez, 1994: A proposal for the
   312) !   intercomparison of the dynamical cores of atmospheric general
   313) !   circulation models.
   314) !   Bulletin of the Amer. Meteor. Soc., vol. 75, pp. 1825-1830.
   315) !
   316) !-----------------------------------------------------------------------
   317) !
   318) ! Compute idealized radiative heating rates (as dry static energy)
   319) !
   320)       efoldf      =  1._r8
   321)       efolda      = 40._r8
   322)       efolds      =  4._r8
   323)       efold_strat = 40._r8
   324)       efold_meso  = 10._r8
   325)       efoldv      = 0.5_r8
   326)       sigmab      = 0.7_r8
   327)       lapse       = 0.00225_r8
   328)       h0          = 7000._r8
   329)       t00         = 200._r8
   330)       p_infint    = 0.01_r8
   331) !
   332)       onemsig = 1._r8 - sigmab
   333) !
   334)       ka = 1._r8/(86400._r8*efolda)
   335)       ks = 1._r8/(86400._r8*efolds)
   336) !
   337)       do k=1,pver
   338)          if (pref_mid_norm(k) > sigmab) then
   339)             do i=1,ncol
   340)                kt    = ka + (ks - ka)*cossqsq(i)*(pref_mid_norm(k) - sigmab)/onemsig
   341)                tmp   = kt/(1._r8+ ztodt*kt)
   342)                trefc = 315._r8 - 60._r8*sinsq(i)
   343)                trefa = (trefc - 10._r8*cossq(i)*log((pmid(i,k)/psurf_ref)))*(pmid(i,k)/psurf_ref)**cappa
   344)                trefa = max(t00,trefa)
   345)                ptend%s(i,k) = (trefa - state%t(i,k))*tmp*cpair
   346)             end do
   347)          else
   348)             do i=1,ncol
   349)                tmp     = ka/(1._r8+ ztodt*ka)
   350)                pressmb = pmid(i,k)*0.01_r8
   351)                trefc   = 315._r8 - 60._r8*sinsq(i)
   352)                trefa   = (trefc - 10._r8*cossq(i)*log((pmid(i,k)/psurf_ref)))*(pmid(i,k)/psurf_ref)** &
   353)                                                                                      cappa
   354)                trefa   = max(t00,trefa)
   355)                if (pressmb <= 100._r8 .and. pressmb > 1._r8) then
   356)                   trefa = t00 + lapse*h0*coslat(i)*log(100._r8/pressmb)
   357)                   tmp   = (efold_strat-efold_meso)*log(pressmb)/log(100._r8)
   358)                   tmp   = efold_meso + tmp
   359)                   tmp   = 1._r8/(86400._r8*tmp)
   360)                   tmp   = tmp/(1._r8+ ztodt*tmp)
   361)                endif
   362)                if (pressmb <= 1._r8 .and. pressmb > 0.01_r8) then
   363)                   trefa = t00 + lapse*h0*coslat(i)*log(100._r8*pressmb)
   364)                   tmp   = 1._r8/(86400._r8*efold_meso)
   365)                   tmp   = tmp/(1._r8+ ztodt*tmp)
   366)                endif
   367)                if (pressmb <= 0.01_r8) then
   368)                   tmp   = 1._r8/(86400._r8*efold_meso)
   369)                   tmp   = tmp/(1._r8+ ztodt*tmp)
   370)                endif
   371)                ptend%s(i,k) = (trefa - state%t(i,k))*tmp*cpair
   372)             end do
   373)          endif
   374)       end do
   375) !
   376) ! Add diffusion near the surface for the wind fields
   377) !
   378)       do k=1,pver
   379)          do i=1,pcols
   380)             ptend%u(i,k) = 0._r8
   381)             ptend%v(i,k) = 0._r8
   382)          end do
   383)       end do
   384) 
   385)       kf = 1._r8/(86400._r8*efoldf)
   386) !
   387)       do k=1,pver
   388)          if (pref_mid_norm(k) > sigmab) then
   389)             kv  = kf*(pref_mid_norm(k) - sigmab)/onemsig
   390)             tmp = -kv/(1._r8+ ztodt*kv)
   391)             do i=1,ncol
   392)                ptend%u(i,k) = tmp*state%u(i,k)
   393)                ptend%v(i,k) = tmp*state%v(i,k)
   394)             end do
   395)          else
   396)             do i=1,ncol
   397)                pressmb  = pmid(i,k)*0.01_r8
   398)                if (pressmb <= 100._r8) then
   399)                   kv       = 1._r8/(86400._r8*efoldv)
   400)                   tmp      = 1._r8 + tanh(1.5_r8*log10(p_infint/pressmb))
   401)                   kv       = kv*tmp
   402)                   tmp      = -kv/(1._r8+ ztodt*kv)
   403)                   ptend%u(i,k) = tmp*state%u(i,k)
   404)                   ptend%v(i,k) = tmp*state%v(i,k)
   405)                endif
   406)             end do
   407)          endif
   408)       end do
   409) 
   410)    else
   411)       write(iulog,*) 'TPHYSIDL: flag for choosing desired type of idealized ', &
   412)                  'physics ("idlflag") is set incorrectly.'
   413)       write(iulog,*) 'The valid options are 1, 2, or 3.'
   414)       write(iulog,*) 'idlflag is currently set to: ',idlflag
   415)       call endrun('tphysidl: invalid option')
   416)    endif
   417) 
   418)    ! update the state and total physics tendency
   419)    call physics_update(state, ptend, ztodt, tend)
   420) 
   421)    ! Can't turn on conservation error messages unless the appropriate heat
   422)    ! surface flux is computed and supplied as an argument to
   423)    ! check_energy_chng to account for how the ideal physics forcings are
   424)    ! changing the total energy.
   425)    call check_energy_chng(state, tend, "tphysidl", nstep, ztodt, zero, zero, zero, zero)
   426) 
   427)    call outfld('QRS', tend%dtdt, pcols, lchnk)
   428) 
   429) end subroutine tphysidl
