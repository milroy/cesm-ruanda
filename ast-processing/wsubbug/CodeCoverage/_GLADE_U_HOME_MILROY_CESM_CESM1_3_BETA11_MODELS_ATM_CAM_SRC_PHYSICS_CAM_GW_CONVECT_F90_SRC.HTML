    61) subroutine gw_beres_src(ncol, ngwv, lat, u, v, netdt, &
    62)      zm, src_level, tend_level, tau, ubm, ubi, xv, yv, c, &
    63)      hdepth, maxq0)
    64) !-----------------------------------------------------------------------
    65) ! Driver for multiple gravity wave drag parameterization.
    66) !
    67) ! The parameterization is assumed to operate only where water vapor
    68) ! concentrations are negligible in determining the density.
    69) !
    70) ! Beres, J.H., M.J. Alexander, and J.R. Holton, 2004: "A method of
    71) ! specifying the gravity wave spectrum above convection based on latent
    72) ! heating properties and background wind". J. Atmos. Sci., Vol 61, No. 3,
    73) ! pp. 324-337.
    74) !
    75) !-----------------------------------------------------------------------
    76)   use gw_common, only: dc, cref
    77)   use gw_utils, only: get_unit_vector, dot_2d, midpoint_interp
    78) !------------------------------Arguments--------------------------------
    79)   ! Column and gravity wave spectrum dimensions.
    80)   integer, intent(in) :: ncol, ngwv
    81) 
    82)   ! Column latitudes [rad].
    83)   real(r8), intent(in) :: lat(ncol)
    84) 
    85)   ! Midpoint zonal/meridional winds.
    86)   real(r8), intent(in) :: u(ncol,pver), v(ncol,pver)
    87)   ! Heating rate due to convection.
    88)   real(r8), intent(in) :: netdt(:,:)
    89)   ! Midpoint altitudes.
    90)   real(r8), intent(in) :: zm(ncol,pver)
    91) 
    92)   ! Indices of top gravity wave source level and lowest level where wind
    93)   ! tendencies are allowed.
    94)   integer, intent(out) :: src_level(ncol)
    95)   integer, intent(out) :: tend_level(ncol)
    96) 
    97)   ! Wave Reynolds stress.
    98)   real(r8), intent(out) :: tau(ncol,-pgwv:pgwv,0:pver)
    99)   ! Projection of wind at midpoints and interfaces.
   100)   real(r8), intent(out) :: ubm(ncol,pver), ubi(ncol,0:pver)
   101)   ! Unit vectors of source wind (zonal and meridional components).
   102)   real(r8), intent(out) :: xv(ncol), yv(ncol)
   103)   ! Phase speeds.
   104)   real(r8), intent(out) :: c(ncol,-pgwv:pgwv)
   105) 
   106)   ! Heating depth and maximum heating in each column.
   107)   real(r8), intent(out) :: hdepth(ncol), maxq0(ncol)
   108) 
   109) !---------------------------Local Storage-------------------------------
   110)   ! Column and level indices.
   111)   integer :: i, k
   112) 
   113)   ! Zonal/meridional wind at 700mb.
   114)   real(r8) :: u700(ncol), v700(ncol)
   115)   ! 3.14...
   116)   real(r8), parameter :: pi = 4._r8*atan(1._r8)
   117) 
   118)   ! Maximum heating rate.
   119)   real(r8) :: q0(ncol)
   120) 
   121)   ! Bottom/top heating range index.
   122)   integer  :: mini(ncol), maxi(ncol)
   123)   ! Min/max wavenumber for critical level filtering.
   124)   integer  :: Umini,Umaxi
   125)   ! Mean wind in heating region.
   126)   real(r8) :: uh(ncol)
   127)   ! Min/max projected wind value in each column.
   128)   real(r8) :: Umin(ncol), Umax(ncol)
   129)   ! Source level tau for a column.
   130)   real(r8) :: tau0(-PGWV:PGWV)
   131)   ! Speed of convective cells relative to storm.
   132)   integer :: CS(ncol)
   133)   ! Index to shift spectra relative to ground.
   134)   integer :: shift
   135) 
   136)   ! Heating rate conversion factor.
   137)   real(r8), parameter :: CF = 20._r8
   138)   ! Averaging length.
   139)   real(r8), parameter :: AL = 1.0e5_r8
   140) 
   141)   !----------------------------------------------------------------------
   142)   ! Initialize tau array
   143)   !----------------------------------------------------------------------
   144) 
   145)   tau = 0.0_r8
   146)   hdepth = 0.0_r8
   147)   q0 = 0.0_r8
   148)   tau0 = 0.0_r8
   149) 
   150)   !------------------------------------------------------------------------
   151)   ! Determine 700 mb layer wind and unit vectors, then project winds.
   152)   !------------------------------------------------------------------------
   153) 
   154)   ! Just use the 700 mb interface values for the source wind speed and
   155)   ! direction (unit vector).
   156) 
   157)   u700 = u(:,k700)
   158)   v700 = v(:,k700)
   159) 
   160)   ! Get the unit vector components and magnitude at the surface.
   161)   call get_unit_vector(u700, v700, xv, yv, ubi(:,k700))
   162) 
   163)   ! Project the local wind at midpoints onto the source wind.
   164)   do k = 1, pver
   165)      ubm(:,k) = dot_2d(u(:,k), v(:,k), xv, yv)
   166)   end do
   167) 
   168)   ! Compute the interface wind projection by averaging the midpoint winds.
   169)   ! Use the top level wind at the top interface.
   170)   ubi(:,0) = ubm(:,1)
   171) 
   172)   ubi(:,1:pver-1) = midpoint_interp(ubm)
   173) 
   174)   !-----------------------------------------------------------------------
   175)   ! Calculate heating depth.
   176)   !
   177)   ! Heating depth is defined as the first height range from the bottom in
   178)   ! which heating rate is continuously positive.
   179)   !-----------------------------------------------------------------------
   180) 
   181)   ! First find the indices for the top and bottom of the heating range.
   182)   mini = 0
   183)   maxi = 0
   184)   do k = pver, 1, -1
   185)      do i = 1, ncol
   186)         if (mini(i) == 0) then
   187)            ! Detect if we are outside the maximum range (where z = 20 km).
   188)            if (zm(i,k) >= 20000._r8) then
   189)               mini(i) = k
   190)               maxi(i) = k
   191)            else
   192)               ! First spot where heating rate is positive.
   193)               if (netdt(i,k) > 0.0_r8) mini(i) = k
   194)            end if
   195)         else if (maxi(i) == 0) then
   196)            ! Detect if we are outside the maximum range (z = 20 km).
   197)            if (zm(i,k) >= 20000._r8) then
   198)               maxi(i) = k
   199)            else
   200)               ! First spot where heating rate is no longer positive.
   201)               if (.not. (netdt(i,k) > 0.0_r8)) maxi(i) = k
   202)            end if
   203)         end if
   204)      end do
   205)      ! When all done, exit
   206)      if (all(maxi /= 0)) exit
   207)   end do
   208) 
   209)   ! Heating depth in km.
   210)   hdepth = [ ( (zm(i,maxi(i))-zm(i,mini(i)))/1000._r8, i = 1, ncol ) ]
   211)   ! Confine depth to table range.
   212)   hdepth = min(hdepth, real(maxh, r8))
   213) 
   214)   ! Maximum heating rate.
   215)   do k = minval(maxi), maxval(mini)
   216)      where (k >= maxi .and. k <= mini)
   217)         q0 = max(q0, netdt(:ncol,k))
   218)      end where
   219)   end do
   220) 
   221)   !output max heating rate in K/day
   222)   maxq0 = q0*24._r8*3600._r8
   223) 
   224)   ! Multipy by conversion factor
   225)   q0 = q0 * CF
   226) 
   227)   ! Taking ubm at 700 mb to be the storm speed, find the cell speed where
   228)   ! the storm speed is > 10 m/s.
   229)   CS = int(sign(max(abs(ubm(:,k700))-10._r8, 0._r8), ubm(:,k700)))
   230) 
   231)   uh = 0._r8
   232)   do k = minval(maxi), maxval(mini)
   233)      where (k >= maxi .and. k <= mini)
   234)         uh = uh + ubm(:,k)/(mini-maxi+1)
   235)      end where
   236)   end do
   237) 
   238)   uh = uh - real(CS, r8)
   239) 
   240)   ! Limit uh to table range.
   241)   uh = min(uh, real(maxuh, r8))
   242)   uh = max(uh, -real(maxuh, r8))
   243) 
   244)   ! Speeds for critical level filtering.
   245)   Umin =  pgwv*dc
   246)   Umax = -pgwv*dc
   247)   do k = minval(maxi), maxval(mini)
   248)      where (k >= maxi .and. k <= mini)
   249)         Umin = min(Umin, ubm(:,k))
   250)         Umax = max(Umax, ubm(:,k))
   251)      end where
   252)   end do
   253) 
   254)   !-----------------------------------------------------------------------
   255)   ! Gravity wave sources
   256)   !-----------------------------------------------------------------------
   257)   ! Start loop over all columns.
   258)   !-----------------------------------------------------------------------
   259)   do i=1,ncol
   260) 
   261)      !---------------------------------------------------------------------
   262)      ! Look up spectrum only if depth >= 2.5 km, else set tau0 = 0.
   263)      !---------------------------------------------------------------------
   264) 
   265)      if ((hdepth(i) >= 2.5_r8) .and. (abs(lat(i)) < (pi/2._r8))) then
   266) 
   267)         !------------------------------------------------------------------
   268)         ! Look up the spectrum using depth and uh.
   269)         !------------------------------------------------------------------
   270) 
   271)         tau0 = mfcc(NINT(hdepth(i)),NINT(uh(i)),:)
   272) 
   273)         ! Shift spectrum so that it is relative to the ground.
   274)         shift = -nint(real(CS(i), r8)/dc)
   275)         tau0 = cshift(tau0,shift)
   276) 
   277)         ! Adjust magnitude.
   278)         tau0 = tau0*q0(i)*q0(i)/AL
   279) 
   280)         ! Adjust for critical level filtering.
   281)         Umini = max(nint(Umin(i)/dc),-PGWV)
   282)         Umaxi = min(nint(Umax(i)/dc),PGWV)
   283) 
   284)         if (Umaxi > Umini) then
   285)            tau0(Umini:Umaxi) = 0.0_r8
   286)         end if
   287) 
   288)         tau(i,-ngwv:ngwv,maxi(i)) = tau0(-ngwv:ngwv)
   289) 
   290)      end if ! depth >= 2.5
   291) 
   292)   enddo
   293) 
   294)   !-----------------------------------------------------------------------
   295)   ! End loop over all columns.
   296)   !-----------------------------------------------------------------------
   297) 
   298)   ! Output the source level.
   299)   src_level = maxi
   300)   tend_level = maxi
   301) 
   302)   ! Set phase speeds; just use reference speeds.
   303)   c = spread(cref, 1, ncol)
   304) 
   305) end subroutine gw_beres_src
   306) 
   307) end module gw_convect
