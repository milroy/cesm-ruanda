     4) ! !MODULE: shr_cal_mod -- calendar module, relates elapsed days to calendar date.
     5) !
     6) ! !DESCRIPTION:
     7) !   These calendar routines do conversions between...
     8) !   \begin{itemize}
     9) !   \item the integer number of elapsed days 
    10) !   \item the integers year, month, day (three inter-related integers)
    11) !   \item the integer coded calendar date (yyyymmdd)
    12) !   \end{itemize}
    13) !   Possible uses include: a calling routine can increment the elapsed days 
    14) !   integer and use this module to determine what the corresponding calendar 
    15) !   date is;  this module can be used to determine how many days apart two
    16) !   arbitrary calendar dates are.

    18) ! !REVISION HISTORY:
    19) !     2001-dec-28 - B. Kauffman - created initial version, taken from cpl5

    21) !       associated with elapsed days must still be done internally
    22) !
    23) ! !REMARKS:
    24) !   Following are some internal assumptions.  These assumptions are somewhat
    25) !   arbitrary -- they were chosen because they result in the simplest code given
    26) !   the requirements of this module.  These assumptions can be relaxed as 
    27) !   necessary: 
    28) !   o the valid range of years is [-999,9999]
    29) !   o elapsed days = 0 <=> January 1st, year 0000 for noleap
    30) !

    98) ! !IROUTINE: shr_cal_calendarName - check calendar name and translate
    99) !
   100) ! !DESCRIPTION:
   101) !    Check the validity of the calendar name and translate to standard naming

   103) ! !REVISION HISTORY:
   104) !     2010-oct-7 - T Craig - initial version.
   105) !
   106) ! !INTERFACE:  -----------------------------------------------------------------
   107) 
   108) function shr_cal_calendarName(calendar,trap)
   109) 
   110)    implicit none
   111) 
   112) ! !INPUT/OUTPUT PARAMETERS:
   113) 
   114)    character(*)         ,intent(in) :: calendar  ! calendar type
   115)    logical     ,optional,intent(in) :: trap
   116)    character(len=shr_cal_calMaxLen) :: shr_cal_calendarName
   117) 
   118) !EOP
   119) 
   120)    character(len=shr_cal_calMaxLen) :: lcal
   121)    logical :: ltrap
   122)    character(len=shr_cal_calMaxLen) :: lowercalendar
   123)    character(*),parameter :: subName = "(shr_cal_calendarName)"
   124) 
   125)    ltrap = .true.
   126)    if (present(trap)) then
   127)       ltrap = trap
   128)    endif
   129) 
   130)    lcal = ' '
   131)    lowercalendar = trim(shr_string_toLower(trim(calendar)))
   132)    lcal = trim(calendar)
   133) 
   134)    selectcase(trim(lowercalendar))
   135) 
   136)    case ('noleap')
   137)       lcal = trim(shr_cal_noleap)
   138)    case ('no_leap')
   139)       lcal = trim(shr_cal_noleap)
   140)    case ('365_day')
   141)       lcal = trim(shr_cal_noleap)
   142)    case ('365day')
   143)       lcal = trim(shr_cal_noleap)
   144)    case ('gregorian')
   145)       lcal = trim(shr_cal_gregorian)
   146)    case ('standard')
   147)       lcal = trim(shr_cal_gregorian)
   148)    case ('proleptic_gregorian')
   149)       lcal = trim(shr_cal_gregorian)
   150)    case default
   151)       if (ltrap) then
   152)          write(s_logunit,*) trim(subname),' : ERROR calendar not supported : ',trim(calendar)
   153)          call shr_sys_abort(trim(subname)//' : calendar not supported : '//trim(calendar))
   154)       endif
   155)    end select
   156) 
   157)    shr_cal_calendarName = trim(lcal)
   158) 
   159)    end function shr_cal_calendarName

   174) subroutine shr_cal_timeSet(etime,ymd,sec,calendar)
   175) 
   176)    implicit none
   177) 
   178) ! !INPUT/OUTPUT PARAMETERS:
   179) 
   180)    type(ESMF_Time),intent(out) :: etime
   181)    integer(SHR_KIND_IN),intent(in ) :: ymd,sec   ! ymd, sec
   182)    character(*)        ,intent(in)  :: calendar  ! calendar type
   183) 
   184) !EOP
   185) 
   186)    integer(SHR_KIND_IN) :: year,month,day
   187)    type(ESMF_CALKIND_FLAG) :: calkind
   188)    character(len=shr_cal_calMaxLen) :: lcalendar

   191) 
   192) !-------------------------------------------------------------------------------
   193) !
   194) !-------------------------------------------------------------------------------
   195) 
   196) 
   197)    lcalendar = shr_cal_calendarName(calendar)
   198) 
   199)    calkind = ESMF_CALKIND_NOLEAP
   200)    if (trim(lcalendar) == trim(shr_cal_gregorian)) then
   201)       calkind = ESMF_CALKIND_GREGORIAN
   202)    endif
   203) 
   204)    call shr_cal_date2ymd(ymd,year,month,day)
   205)    call ESMF_TimeSet(etime,yy=year,mm=month,dd=day,s=sec,calkindflag=calkind,rc=rc)
   206)    if(rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   207)

   220) ! !INTERFACE:  -----------------------------------------------------------------

   222) integer function shr_cal_numDaysInMonth(year,month,calendar)
   223) 
   224)    implicit none
   225) 
   226) ! !INPUT/OUTPUT PARAMETERS:
   227) 
   228)    integer(SHR_KIND_IN),intent(in ) :: year,month  ! calendar year,month
   229)    character(*)        ,intent(in)  :: calendar  ! calendar type
   230) 
   231) !EOP
   232) 
   233)    type(ESMF_time) :: time1,time2
   234)    type(ESMF_timeInterval) :: timeint
   235)    type(ESMF_CALKIND_FLAG) :: calkind
   236)    integer(SHR_KIND_IN) :: eday
   237)    character(len=shr_cal_calMaxLen) :: lcalendar
   238)    integer :: rc
   239)    character(*),parameter :: subName = "(shr_cal_numDaysInMonth)"
   240) 
   241) !-------------------------------------------------------------------------------
   242) !
   243) !-------------------------------------------------------------------------------
   244) 
   245)    lcalendar = shr_cal_calendarName(calendar)
   246) 
   247)    calkind = ESMF_CALKIND_NOLEAP
   248)    if (trim(lcalendar) == trim(shr_cal_gregorian)) then
   249)       calkind = ESMF_CALKIND_GREGORIAN
   250)    endif
   251) 
   252)    call ESMF_TimeSet(time1,yy=year,mm=month,dd=1,s=0,calkindflag=calkind,rc=rc)
   253)    if(rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   254)    if (month < 12) then
   255)       call ESMF_TimeSet(time2,yy=year,mm=month+1,dd=1,s=0,calkindflag=calkind,rc=rc)
   256)       if(rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   257)    else
   258)       call ESMF_TimeSet(time2,yy=year+1,mm=1,dd=1,s=0,calkindflag=calkind,rc=rc)
   259)       if(rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   260)    endif
   261)    timeint = time2-time1
   262)    call ESMF_TimeIntervalGet(timeint,StartTimeIn=time1,d=eday)
   263)    shr_cal_numDaysInMonth = eday
   264) 
   265) end function shr_cal_numDaysInMonth
