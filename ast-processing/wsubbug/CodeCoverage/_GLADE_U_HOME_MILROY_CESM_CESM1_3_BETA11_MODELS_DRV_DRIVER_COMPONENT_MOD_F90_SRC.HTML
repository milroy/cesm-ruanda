    86)   subroutine component_init_pre(comp, compid, cplcompid, cplallcompid, &
    87)        infodata, ntype)
    88) 
    89)     !---------------------------------------------------------------
    90)     ! Initialize driver rearrangers and AVs on driver
    91)     ! Initialize cdata_*x data
    92)     ! Zero out x2*_** in case it never gets used then it'll produce zeros in diags
    93)     ! For ensembles, create only a single dom_*x for the coupler based on the
    94)     !   first ensemble member.  otherwise, just extend the dom_** and dom_*x to
    95)     !   other ensemble members.
    96)     !
    97)     ! Arguments
    98)     type(component_type)     , intent(inout)         :: comp(:)
    99)     integer                  , intent(in)            :: compid(:)
   100)     integer                  , intent(in)            :: cplcompid(:)
   101)     integer                  , intent(in)            :: cplallcompid
   102)     type (seq_infodata_type) , intent(inout), target :: infodata
   103)     character(len=3)         , intent(in)            :: ntype
   104)     !
   105)     ! Local Variables
   106)     integer  :: eci       ! index
   107)     !---------------------------------------------------------------
   108) 
   109)     ! initialize module variables (this is repetitive here- but does not require a different routine)
   110) 
   111)     call seq_infodata_getdata(infodata, drv_threading=drv_threading)
   112)     call seq_comm_getinfo(GLOID, mpicom=mpicom_GLOID, iamroot=iamroot_GLOID, nthreads=nthreads_GLOID)
   113)     call seq_comm_getinfo(CPLID, mpicom=mpicom_CPLID, iamroot=iamroot_CPLID, nthreads=nthreads_CPLID)
   114)     iamin_CPLID = seq_comm_iamin(CPLID)
   115)    
   116)     ! Initialize component type variables
   117)     do eci = 1,size(comp)
   118) 
   119)        comp(eci)%compid       = compid(eci)
   120)        comp(eci)%cplcompid    = cplcompid(eci)
   121)        comp(eci)%cplallcompid = cplallcompid
   122) 
   123)        call seq_comm_getinfo(comp(eci)%cplallcompid, mpicom=comp(eci)%mpicom_cplallcompid)
   124)        call seq_comm_getinfo(comp(eci)%cplcompid   , mpicom=comp(eci)%mpicom_cplcompid)
   125)        call seq_comm_getinfo(comp(eci)%compid      , mpicom=comp(eci)%mpicom_compid)
   126)        call seq_comm_getinfo(comp(eci)%compid      , iamroot=comp(eci)%iamroot_compid)
   127)        call seq_comm_getinfo(comp(eci)%compid      , nthreads=comp(eci)%nthreads_compid)
   128) 
   129)        comp(eci)%iamin_compid       =  seq_comm_iamin (comp(eci)%compid)
   130)        comp(eci)%iamin_cplcompid    =  seq_comm_iamin (comp(eci)%cplcompid)
   131)        comp(eci)%iamin_cplallcompid =  seq_comm_iamin (comp(eci)%cplallcompid)
   132)        comp(eci)%suffix             =  seq_comm_suffix(comp(eci)%compid)
   133)        comp(eci)%name               =  seq_comm_name  (comp(eci)%compid)
   134)        comp(eci)%ntype              =  ntype(1:3)
   135)        comp(eci)%oneletterid        =  ntype(1:1)
   136) 
   137)        if (eci == 1) then
   138)           allocate(comp(1)%dom_cx)
   139)           allocate(comp(1)%gsmap_cx)
   140)        else
   141)           comp(eci)%dom_cx   => comp(1)%dom_cx 
   142)           comp(eci)%gsmap_cx => comp(1)%gsmap_cx 
   143)        end if
   144) 
   145)        ! Set cdata_cc - unique for each instance
   146)        allocate(comp(eci)%dom_cc)
   147)        allocate(comp(eci)%gsmap_cc)
   148)        allocate(comp(eci)%cdata_cc)
   149)        comp(eci)%cdata_cc%name     = 'cdata_'//ntype(1:1)//ntype(1:1)
   150)        comp(eci)%cdata_cc%ID       =  comp(eci)%compid
   151)        comp(eci)%cdata_cc%mpicom   =  comp(eci)%mpicom_compid
   152)        comp(eci)%cdata_cc%dom      => comp(eci)%dom_cc 
   153)        comp(eci)%cdata_cc%gsmap    => comp(eci)%gsmap_cc
   154)        comp(eci)%cdata_cc%infodata => infodata
   155) 
   156)        ! Determine initial value of comp_present in infodata - to do - add this to component 
   157) 
   158)        if (comp(1)%oneletterid == 'a') call seq_infodata_getData(infodata, atm_present=comp(eci)%present)
   159)        if (comp(1)%oneletterid == 'l') call seq_infodata_getData(infodata, lnd_present=comp(eci)%present)
   160)        if (comp(1)%oneletterid == 'i') call seq_infodata_getData(infodata, ice_present=comp(eci)%present)
   161)        if (comp(1)%oneletterid == 'o') call seq_infodata_getData(infodata, ocn_present=comp(eci)%present)
   162)        if (comp(1)%oneletterid == 'r') call seq_infodata_getData(infodata, rof_present=comp(eci)%present)
   163)        if (comp(1)%oneletterid == 'g') call seq_infodata_getData(infodata, glc_present=comp(eci)%present)
   164)        if (comp(1)%oneletterid == 'w') call seq_infodata_getData(infodata, wav_present=comp(eci)%present)
   165) 
   166)     end do
   167) 
   168)   end subroutine component_init_pre

   173)   subroutine component_init_cc(Eclock, comp, comp_init, infodata, NLFilename, &
   174)        seq_flds_x2c_fluxes, seq_flds_c2x_fluxes)
   175) 
   176)     !---------------------------------------------------------------
   177)     !
   178)     ! Arguments
   179)     type(ESMF_Clock)         , intent(inout) :: EClock
   180)     type(component_type)     , intent(inout) :: comp(:)
   181)     interface 
   182)        subroutine comp_init( Eclock, cdata, x2c, c2x, nlfilename)
   183)          use ESMF         , only: ESMF_Clock
   184)          use seq_cdata_mod, only: seq_cdata
   185)          use mct_mod      , only: mct_avect
   186)          implicit none
   187)          type(ESMF_Clock), intent(inout) :: EClock
   188)          type(seq_cdata) , intent(inout) :: cdata
   189)          type(mct_aVect) , intent(inout) :: x2c
   190)          type(mct_aVect) , intent(inout) :: c2x   
   191)          character(len=*), optional, intent(IN) :: NLFilename ! Namelist filename
   192)        end subroutine comp_init
   193)     end interface 
   194)     type (seq_infodata_type) , intent(inout)        :: infodata
   195)     character(len=*)         , intent(in)           :: NLFilename 
   196)     character(len=*)         , intent(in), optional :: seq_flds_x2c_fluxes
   197)     character(len=*)         , intent(in), optional :: seq_flds_c2x_fluxes
   198)     !
   199)     ! Local Variables
   200)     integer :: k1, k2
   201)     integer :: eci
   202)     !---------------------------------------------------------------
   203) 
   204)     ! **** Initialize component - this initializes  x2c_cc and c2x_cc ***
   205)     ! the following will call the appropriate comp_init_mct routine
   206) 
   207)     if (comp(1)%iamin_cplallcompid) then
   208)        call seq_infodata_exchange(infodata, comp(1)%cplallcompid, &
   209)             'cpl2'//comp(1)%ntype(1:3)//'_init')
   210)     end if
   211) 
   212)     ! The following initializes the component instance cdata_cc (gsmap and dom), 
   213)     ! x2c_cc and c2x_cc
   214) 
   215)     do eci = 1,size(comp)
   216)        if (iamroot_CPLID .and. comp(eci)%present) then
   217)           write(logunit,F00) 'Initialize component '//trim(comp(eci)%ntype)
   218)           call shr_sys_flush(logunit)
   219)        endif
   220) 
   221)        if (.not. associated(comp(eci)%x2c_cc)) allocate(comp(eci)%x2c_cc)
   222)        if (.not. associated(comp(eci)%c2x_cc)) allocate(comp(eci)%c2x_cc)
   223) 
   224)        if (comp(eci)%iamin_compid .and. comp(eci)%present) then
   225)           if (drv_threading) call seq_comm_setnthreads(comp(eci)%nthreads_compid)
   226)           call shr_sys_flush(logunit)
   227)           
   228)           if (present(seq_flds_x2c_fluxes)) then
   229)              call mct_avect_vecmult(comp(eci)%x2c_cc, comp(eci)%drv2mdl, seq_flds_x2c_fluxes, mask_spval=.true.)
   230)           end if
   231) 
   232)           call comp_init( EClock, comp(eci)%cdata_cc, comp(eci)%x2c_cc, comp(eci)%c2x_cc, &
   233)                NLFilename=NLFilename )
   234)           
   235)           if (present(seq_flds_c2x_fluxes)) then
   236)              call mct_avect_vecmult(comp(eci)%c2x_cc, comp(eci)%mdl2drv, seq_flds_c2x_fluxes, mask_spval=.true.)
   237)           end if
   238)           
   239)           if (drv_threading) call seq_comm_setnthreads(nthreads_GLOID)
   240)        end if
   241)     end do
   242) 
   243)     if (comp(1)%iamin_cplcompid) then
   244)        call seq_infodata_exchange(infodata, comp(1)%cplcompid, &
   245)             comp(1)%ntype(1:3)//'2cpl_init')
   246)     endif
   247) 
   248)     ! Determine final value of comp_present in infodata (after component initialization)
   249) 
   250)     do eci = 1,size(comp) 
   251)        if (comp(1)%oneletterid == 'a') call seq_infodata_getData(infodata, atm_present=comp(eci)%present)
   252)        if (comp(1)%oneletterid == 'l') call seq_infodata_getData(infodata, lnd_present=comp(eci)%present)
   253)        if (comp(1)%oneletterid == 'i') call seq_infodata_getData(infodata, ice_present=comp(eci)%present)
   254)        if (comp(1)%oneletterid == 'o') call seq_infodata_getData(infodata, ocn_present=comp(eci)%present)
   255)        if (comp(1)%oneletterid == 'r') call seq_infodata_getData(infodata, rof_present=comp(eci)%present)
   256)        if (comp(1)%oneletterid == 'g') call seq_infodata_getData(infodata, glc_present=comp(eci)%present)
   257)        if (comp(1)%oneletterid == 'w') call seq_infodata_getData(infodata, wav_present=comp(eci)%present)
   258)     end do
   259) 
   260) 
   261)     ! Initialize aream, set it to area for now until maps are read
   262)     !   in some cases, maps are not read at all !!
   263)     ! Entire domain must have reasonable values before calling xxx2xxx init
   264) 
   265)     do eci = 1,size(comp)
   266)        if (comp(eci)%iamin_compid .and. comp(eci)%present) then
   267)           if (drv_threading) call seq_comm_setnthreads(comp(eci)%nthreads_compid)
   268)           k1 = mct_aVect_indexRa(comp(eci)%cdata_cc%dom%data, "area"  ,perrWith='aa area ')
   269)           k2 = mct_aVect_indexRa(comp(eci)%cdata_cc%dom%data, "aream" ,perrWith='aa aream')
   270) 
   271)           comp(eci)%cdata_cc%dom%data%rAttr(k2,:) = comp(eci)%cdata_cc%dom%data%rAttr(k1,:)
   272) 
   273)           if (drv_threading) call seq_comm_setnthreads(nthreads_GLOID)
   274)        endif
   275)     end do
   276) 
   277)   end subroutine component_init_cc
   278) #endif

   282) #ifdef ESMF_INTERFACE
   283)   subroutine component_init_cc(Eclock, drvcomp, comp, gridcomp_register,  &
   284)        infodata, NlFilename, seq_flds_x2c_fields, seq_flds_c2x_fields, &
   285)        seq_flds_x2c_fluxes, seq_flds_c2x_fluxes) 
   286) 
   287)     !---------------------------------------------------------------
   288)     ! Uses
   289)     use esmf2mct_mod,         only: esmf2mct_init, esmf2mct_copy
   290)     use mct2esmf_mod,         only: mct2esmf_init
   291)     use seq_flds_mod,         only: seq_flds_dom_coord, seq_flds_dom_other, seq_flds_dom_fields
   292)     !
   293)     ! Arguments
   294)     type(ESMF_Clock)     , intent(inout) :: EClock
   295)     type(ESMF_CplComp)   , intent(inout) :: drvComp
   296)     type(component_type) , intent(inout) :: comp(:)
   297)     interface 
   298)        subroutine gridcomp_register(gridcomp, rc)
   299)          use ESMF
   300)          implicit none
   301)          type(ESMF_GridComp)  :: gridcomp
   302)          integer, intent(out) :: rc
   303)        end subroutine gridcomp_register
   304)     end interface 
   305)     type (seq_infodata_type) , intent(inout)        :: infodata
   306)     character(len=*)         , intent(in), optional :: NLFilename 
   307)     character(len=*)         , intent(in), optional :: seq_flds_x2c_fields
   308)     character(len=*)         , intent(in), optional :: seq_flds_c2x_fields
   309)     character(len=*)         , intent(in), optional :: seq_flds_x2c_fluxes
   310)     character(len=*)         , intent(in), optional :: seq_flds_c2x_fluxes
   311)     !
   312)     ! Local Variables
   313)     integer             :: k1, k2
   314)     integer             :: nx, ny
   315)     integer             :: eci
   316)     integer             :: rc, urc
   317)     integer             :: init_phase
   318)     integer             :: lsize
   319)     type(ESMF_DistGrid) :: distgrid_cc
   320)     type(ESMF_DistGrid) :: distgrid_cx
   321)     type(ESMF_Array)    :: dom_cc_array
   322)     type(ESMF_Array)    :: dom_cx_array
   323)     type(ESMF_Array)    :: x2c_cc_array
   324)     type(ESMF_Array)    :: x2c_cx_array
   325)     type(ESMF_Array)    :: c2x_cc_array
   326)     type(ESMF_Array)    :: c2x_cx_array
   327)     integer , pointer   :: petlist(:)
   328)     real(R8), pointer   :: fptr(:,:)            ! pointer into    array data
   329)     character(len=1)    :: cid
   330)     character(len=8196) :: mct_names_x2c, mct_names_c2x
   331)     !---------------------------------------------------------------
   332) 
   333)     if (present(seq_flds_x2c_fluxes) .and. present(seq_flds_c2x_fluxes)) then
   334)        init_phase = 2
   335)     else
   336)        init_phase = 1
   337)     end if
   338) 
   339)     do eci = 1,size(comp)
   340) 
   341)        if (init_phase == 1) then
   342) 
   343)           ! Create gridcomp for this instance
   344) 
   345)           call seq_comm_petlist(comp(eci)%compid, petlist) 
   346) 
   347)           comp(eci)%gridcomp_cc = ESMF_GridCompCreate(name=trim(comp(eci)%name), petList=petlist, rc=rc)
   348)           if (rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   349) 
   350)           call ESMF_GridCompSetServices(comp(eci)%gridcomp_cc, userRoutine=gridcomp_register, rc=rc)
   351)           if (rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   352) 
   353)           ! create import and export states state_x2c_cc and state_c2x_cc
   354) 
   355)           comp(eci)%x2c_cc_state = ESMF_StateCreate(name=trim(comp(eci)%ntype)//" x2c_cc", &
   356)                stateintent=ESMF_STATEINTENT_IMPORT, rc=rc)
   357)           if (rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   358) 
   359)           comp(eci)%c2x_cc_state = ESMF_StateCreate(name=trim(comp(eci)%ntype)//" c2x_cc", &
   360)                stateintent=ESMF_STATEINTENT_EXPORT, rc=rc)
   361)           if (rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   362) 
   363)           ! link attributes
   364) 
   365)           call ESMF_AttributeLink(drvcomp, comp(eci)%gridcomp_cc, rc=rc)
   366)           if (rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   367) 
   368)        end if
   369) 
   370)        !--------------------------------------------------
   371)        ! COMPONENT-CPL PES
   372)        !--------------------------------------------------
   373)        ! Transfer infodata from coupler pes -> component pes
   374) 
   375)        if (comp(1)%iamin_cplallcompid) then
   376)           call seq_infodata_exchange(infodata, comp(1)%cplallcompid, &
   377)                'cpl2'//comp(1)%ntype(1:3)//'_init')
   378)        end if
   379) 
   380)        !--------------------------------------------------
   381)        ! COMPONENT PES
   382)        !--------------------------------------------------
   383) 
   384)        ! The following initializes the component instance values of x2c_cc and c2x_cc
   385)        
   386)        if (iamroot_CPLID .and. comp(eci)%present) then
   387)           write(logunit,F00) 'Initialize component '//trim(comp(eci)%ntype)
   388)           call shr_sys_flush(logunit)
   389)        end if
   390) 
   391)        if (comp(eci)%iamin_compid) then
   392) 
   393)           if (comp(eci)%present) then
   394) 
   395)              if (drv_threading) call seq_comm_setnthreads(comp(eci)%nthreads_compid)
   396)              call shr_sys_flush(logunit)
   397) 
   398)              if (init_phase == 1) then 
   399)                 call ESMF_AttributeSet(comp(eci)%c2x_cc_state, name="ID", &
   400)                      value=comp(eci)%compid, rc=rc)
   401)                 if (rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   402)              end if
   403) 
   404)              ! Set init_phase attribute value
   405)              call ESMF_AttributeSet(comp(eci)%c2x_cc_state, name=trim(comp(eci)%ntype)//"_phase", &
   406)                   value=init_phase, rc=rc)
   407)              if (rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   408) 
   409)              ! Convert infodata information into appropriate export state attributes
   410)              call esmfshr_infodata_infodata2state(infodata, comp(eci)%c2x_cc_state, &
   411)                   id=comp(eci)%compid, rc=rc)
   412)              if (rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)
   413) 
   414)              ! Rescale the attribute vector before sending for phase 2 of atm initialization
   415)              ! Note that x2c_cc attribute vector and x2c_cc_array share the same memory 
   416)              ! (set in init_phase=1)
   417) 
   418)              if (init_phase == 2) then ! phase 2 (only for atm for now)
   419)                 call ESMF_StateGet(comp(eci)%x2c_cc_state, itemName="x2d", array=x2c_cc_array, rc=rc)
   420)                 if (rc /= ESMF_SUCCESS) call ESMF_Finalize(rc=rc, endflag=ESMF_END_ABORT)

   654)   subroutine component_init_cx(comp, infodata) 
   655) 
   656)     !---------------------------------------------------------------
   657)     ! Uses
   658)     use cplcomp_exchange_mod, only: seq_mctext_gsmapinit, seq_mctext_avInit
   659)     use cplcomp_exchange_mod, only: seq_mctext_avExtend, seq_mctext_gGridInit
   660)     use cplcomp_exchange_mod, only: seq_map_init_exchange, seq_map_map_exchange

   662)     use mct_mod,              only: mct_ggrid_clean
   663)     !
   664)     ! Arguments
   665)     type(component_type)     , intent(inout) :: comp(:)
   666)     type (seq_infodata_type) , intent(inout) :: infodata
   667)     !
   668)     ! Local Variables
   669)     integer         :: eci
   670)     integer         :: rc        ! return code
   671)     type(mct_gGrid) :: dom_tmp   ! temporary
   672)     !---------------------------------------------------------------
   673) 
   674)     ! Initialize driver rearrangers and AVs on driver
   675)     ! Initialize cdata_*x data
   676)     ! Zero out x2*_** in case it never gets used then it'll produce zeros in diags
   677)     ! For ensembles, create only a single dom_*x for the coupler based on the
   678)     !   first ensemble member.  otherwise, just extend the dom_** and dom_*x to
   679)     !   other ensemble members.
   680) 
   681)     do eci = 1,size(comp)
   682)        if (comp(eci)%present) then
   683) 
   684)           if (iamroot_CPLID) then
   685)              write(logunit,*) ' '
   686)              call shr_sys_flush(logunit)
   687)           end if
   688) 
   689)           if (comp(eci)%iamin_cplcompid) then
   690) 
   691)              ! Create gsmap_cx (note that comp(eci)%gsmap_cx all point to comp(1)%gsmap_cx
   692)              ! This will only be valid on the coupler pes
   693)              if (eci == 1) then
   694)                 if (iamroot_CPLID) then
   695)                    write(logunit,F0I) 'creating gsmap_cx for '//comp(eci)%ntype(1:3)
   696)                    call shr_sys_flush(logunit)
   697)                 end if
   698)                 call seq_mctext_gsmapInit(comp(1))
   699)              endif
   700) 
   701)              ! Create mapper_Cc2x and mapper_Cx2c
   702)              allocate(comp(eci)%mapper_Cc2x, comp(eci)%mapper_Cx2c)  
   703)              if (iamroot_CPLID) then
   704)                 write(logunit,F0I) 'Initializing mapper_C'//comp(eci)%ntype(1:1)//'2x',eci
   705)                 call shr_sys_flush(logunit)
   706)              end if
   707)              call seq_map_init_exchange(comp(eci), flow='c2x', mapper=comp(eci)%mapper_Cc2x)    
   708)              if (iamroot_CPLID) then
   709)                 write(logunit,F0I) 'Initializing mapper_Cx2'//comp(eci)%ntype(1:1),eci
   710)                 call shr_sys_flush(logunit)
   711)              end if
   712)              call seq_map_init_exchange(comp(eci), flow='x2c', mapper=comp(eci)%mapper_Cx2c)  
   713) 
   714)              ! Create x2c_cx and c2x_cx
   715)              allocate(comp(eci)%x2c_cx, comp(eci)%c2x_cx)
   716)              call seq_mctext_avinit(comp(eci), flow='x2c')
   717)              call seq_mctext_avinit(comp(eci), flow='c2x')
   718) 
   719)              ! Create dom_cx (note that  comp(eci)%dom_cx all point to  comp(1)%dom_cx
   720)              ! Then verify other ensembles have same domain by comparing to dom_cx
   721)              if (eci == 1) then  ! create dom_cx
   722)                 if (iamroot_CPLID) then
   723)                    write(logunit,F0I) 'creating dom_cx'
   724)                    call shr_sys_flush(logunit)
   725)                 end if
   726)                 call seq_mctext_gGridInit(comp(1))
   727)                 call seq_map_map_exchange(comp(1), flow='c2x', dom_flag=.true., msgtag=comp(1)%cplcompid*100+1*10+1)
   728)              else if (eci > 1) then  
   729)                 if (iamroot_CPLID) then
   730)                    write(logunit,F0I) 'comparing comp domain ensemble number ',eci
   731)                    call shr_sys_flush(logunit)
   732)                 end if
   733)                 call seq_mctext_avExtend(comp(eci)%dom_cx%data, cplid, comp(eci)%cplcompid)
   734)                 call seq_mctext_gGridInit(comp(eci), dom_tmp)
   735)                 call seq_map_map_exchange(comp(eci), flow='c2x', dom_flag=.true., dom_tmp=dom_tmp)
   736)                 if (iamin_CPLID) then
   737)                    call seq_domain_compare(comp(eci)%dom_cx, dom_tmp, mpicom_CPLID)
   738)                 end if
   739)                 call mct_ggrid_clean(dom_tmp,rc)
   740)              endif
   741) 
   742)              call mct_avect_zero(comp(eci)%x2c_cc)
   743)              call mct_avect_zero(comp(eci)%x2c_cx)
   744) 
   745)           end if ! if comp(eci)%iamin_cplcompid
   746)        end if  ! if comp(eci)%present
   747)     end do  ! end of eci loop
   748) 
   749)   end subroutine component_init_cx

   754)

   766)     ! Arguments

   768)     logical                  , intent(in)    :: rof_c2_ocn
   769)     logical                  , intent(in)    :: samegrid_ao
