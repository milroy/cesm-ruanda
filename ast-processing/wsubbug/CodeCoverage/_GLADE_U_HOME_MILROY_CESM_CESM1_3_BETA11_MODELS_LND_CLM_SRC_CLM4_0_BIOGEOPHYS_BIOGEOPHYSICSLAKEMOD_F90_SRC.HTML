    34) !
    35) ! !DESCRIPTION:
    36) ! Calculates lake temperatures and surface fluxes.
    37) ! Lake temperatures are determined from a one-dimensional thermal
    38) ! stratification model based on eddy diffusion concepts to
    39) ! represent vertical mixing of heat.
    40) !
    41) ! d ts    d            d ts     1 ds
    42) ! ---- = -- [(km + ke) ----] + -- --
    43) !  dt    dz             dz     cw dz
    44) !
    45) ! where: ts = temperature (kelvin)
    46) !         t = time (s)
    47) !         z = depth (m)
    48) !        km = molecular diffusion coefficient (m**2/s)
    49) !        ke = eddy diffusion coefficient (m**2/s)
    50) !        cw = heat capacity (j/m**3/kelvin)
    51) !         s = heat source term (w/m**2)
    52) !
    53) ! There are two types of lakes:
    54) !   Deep lakes are 50 m.
    55) !   Shallow lakes are 10 m deep.
    56) !
    57) !   For unfrozen deep lakes:    ke > 0 and    convective mixing
    58) !   For unfrozen shallow lakes: ke = 0 and no convective mixing
    59) !
    60) ! Use the Crank-Nicholson method to set up tridiagonal system of equations to
    61) ! solve for ts at time n+1, where the temperature equation for layer i is
    62) ! r_i = a_i [ts_i-1] n+1 + b_i [ts_i] n+1 + c_i [ts_i+1] n+1
    63) !
    64) ! The solution conserves energy as:
    65) !
    66) ! cw*([ts(      1)] n+1 - [ts(      1)] n)*dz(      1)/dt + ... +
    67) ! cw*([ts(nlevlak)] n+1 - [ts(nlevlak)] n)*dz(nlevlak)/dt = fin
    68) !
    69) ! where:
    70) ! [ts] n   = old temperature (kelvin)
    71) ! [ts] n+1 = new temperature (kelvin)
    72) ! fin      = heat flux into lake (w/m**2)
    73) !          = beta*sabg + forc_lwrad - eflx_lwrad_out - eflx_sh_tot - eflx_lh_tot
    74) !            - hm + phi(1) + ... + phi(nlevlak)
    75) !
    76) ! WARNING: This subroutine assumes lake columns have one and only one pft.
    77) !
    78) ! !USES:
    79)     use shr_kind_mod, only: r8 => shr_kind_r8
    80)     use clmtype
    81)     use clm_atmlnd         , only : clm_a2l
    82)     use clm_time_manager       , only : get_step_size

    85)                                     sb, vkc, grav, denh2o, tfrz, spval
    86)     use QSatMod            , only : QSat
    87)     use FrictionVelocityMod, only : FrictionVelocity, MoninObukIni
    88)     use TridiagonalMod     , only : Tridiagonal
    89) !
    90) ! !ARGUMENTS:
    91)     implicit none
    92)     integer, intent(in) :: lbc, ubc                ! column-index bounds
    93)     integer, intent(in) :: lbp, ubp                ! pft-index bounds
    94)     integer, intent(in) :: num_lakec               ! number of column non-lake points in column filter
    95)     integer, intent(in) :: filter_lakec(ubc-lbc+1) ! column filter for non-lake points

    97)     integer, intent(in) :: filter_lakep(ubp-lbp+1) ! pft filter for non-lake points
    98) !

   150)     real(r8), pointer :: eflx_lh_grnd(:)    ! ground evaporation heat flux (W/m**2) [+ to atm]
   151)     real(r8), pointer :: t_veg(:)           ! vegetation temperature (Kelvin)
   152)     real(r8), pointer :: t_ref2m(:)         ! 2 m height surface air temperature (Kelvin)
   153)     real(r8), pointer :: q_ref2m(:)         ! 2 m height surface specific humidity (kg/kg)
   154)     real(r8), pointer :: rh_ref2m(:)        ! 2 m height surface relative humidity (%)
   155)     real(r8), pointer :: taux(:)            ! wind (shear) stress: e-w (kg/m/s**2)
   156)     real(r8), pointer :: tauy(:)            ! wind (shear) stress: n-s (kg/m/s**2)
   157)     real(r8), pointer :: qmelt(:)           ! snow melt [mm/s]
   158)     real(r8), pointer :: ram1(:)            ! aerodynamical resistance (s/m)
   159)     real(r8), pointer :: errsoi(:)          ! soil/lake energy conservation error (W/m**2)
   160)     real(r8), pointer :: t_lake(:,:)        ! lake temperature (Kelvin)
   161) !
   162) !
   163) ! !OTHER LOCAL VARIABLES:
   164) !EOP
   165) !
   166)     integer , parameter  :: idlak = 1     ! index of lake, 1 = deep lake, 2 = shallow lake
   167)     integer , parameter  :: niters = 3    ! maximum number of iterations for surface temperature
   168)     real(r8), parameter :: beta1 = 1._r8  ! coefficient of connective velocity (in computing W_*) [-]
   169)     real(r8), parameter :: emg = 0.97_r8     ! ground emissivity (0.97 for snow)
   170)     real(r8), parameter :: zii = 1000._r8 ! convective boundary height [m]
   171)     real(r8), parameter :: p0 = 1._r8     ! neutral value of turbulent prandtl number
   172)     integer  :: i,j,fc,fp,g,c,p         ! do loop or array index
   173)     integer  :: fncopy                  ! number of values in pft filter copy
   174)     integer  :: fnold                   ! previous number of pft filter values
   175)     integer  :: fpcopy(num_lakep)       ! pft filter copy for iteration loop
   176)     integer  :: num_unfrzc              ! number of values in unfrozen column filter
   177)     integer  :: filter_unfrzc(ubc-lbc+1)! unfrozen column filter
   178)     integer  :: iter                    ! iteration index
   179)     integer  :: nmozsgn(lbp:ubp)        ! number of times moz changes sign
   180)     integer  :: jtop(lbc:ubc)           ! number of levels for each column (all 1)
   181)     real(r8) :: dtime                   ! land model time step (sec)
   182)     real(r8) :: ax                      !
   183)     real(r8) :: bx                      !
   184)     real(r8) :: degdT                   ! d(eg)/dT
   185)     real(r8) :: dqh(lbp:ubp)            ! diff of humidity between ref. height and surface
   186)     real(r8) :: dth(lbp:ubp)            ! diff of virtual temp. between ref. height and surface
   187)     real(r8) :: dthv                    ! diff of vir. poten. temp. between ref. height and surface
   188)     real(r8) :: dzsur(lbc:ubc)          !
   189)     real(r8) :: eg                      ! water vapor pressure at temperature T [pa]
   190)     real(r8) :: hm                      ! energy residual [W/m2]
   191)     real(r8) :: htvp(lbc:ubc)           ! latent heat of vapor of water (or sublimation) [j/kg]
   192)     real(r8) :: obu(lbp:ubp)            ! monin-obukhov length (m)
   193)     real(r8) :: obuold(lbp:ubp)         ! monin-obukhov length of previous iteration
   194)     real(r8) :: qsatg(lbc:ubc)          ! saturated humidity [kg/kg]
   195)     real(r8) :: qsatgdT(lbc:ubc)        ! d(qsatg)/dT
   196)     real(r8) :: qstar                   ! moisture scaling parameter
   197)     real(r8) :: ram(lbp:ubp)            ! aerodynamical resistance [s/m]
   198)     real(r8) :: rah(lbp:ubp)            ! thermal resistance [s/m]
   199)     real(r8) :: raw(lbp:ubp)            ! moisture resistance [s/m]
   200)     real(r8) :: stftg3(lbp:ubp)         ! derivative of fluxes w.r.t ground temperature
   201)     real(r8) :: temp1(lbp:ubp)          ! relation for potential temperature profile
   202)     real(r8) :: temp12m(lbp:ubp)        ! relation for potential temperature profile applied at 2-m
   203)     real(r8) :: temp2(lbp:ubp)          ! relation for specific humidity profile
   204)     real(r8) :: temp22m(lbp:ubp)        ! relation for specific humidity profile applied at 2-m
   205)     real(r8) :: tgbef(lbc:ubc)          ! initial ground temperature
   206)     real(r8) :: thm(lbp:ubp)            ! intermediate variable (forc_t+0.0098*forc_hgt_t_pft)
   207)     real(r8) :: thv(lbc:ubc)            ! virtual potential temperature (kelvin)
   208)     real(r8) :: thvstar                 ! virtual potential temperature scaling parameter
   209)     real(r8) :: tksur                   ! thermal conductivity of snow/soil (w/m/kelvin)
   210)     real(r8) :: tstar                   ! temperature scaling parameter
   211)     real(r8) :: um(lbp:ubp)             ! wind speed including the stablity effect [m/s]
   212)     real(r8) :: ur(lbp:ubp)             ! wind speed at reference height [m/s]
   213)     real(r8) :: ustar(lbp:ubp)          ! friction velocity [m/s]
   214)     real(r8) :: wc                      ! convective velocity [m/s]
   215)     real(r8) :: zeta                    ! dimensionless height used in Monin-Obukhov theory
   216)     real(r8) :: zldis(lbp:ubp)          ! reference height "minus" zero displacement height [m]
   217)     real(r8) :: displa(lbp:ubp)         ! displacement (always zero) [m]
   218)     real(r8) :: z0mg(lbp:ubp)           ! roughness length over ground, momentum [m]
   219)     real(r8) :: z0hg(lbp:ubp)           ! roughness length over ground, sensible heat [m]
   220)     real(r8) :: z0qg(lbp:ubp)           ! roughness length over ground, latent heat [m]
   221)     real(r8) :: beta(2)                 ! fraction solar rad absorbed at surface: depends on lake type
   222)     real(r8) :: za(2)                   ! base of surface absorption layer (m): depends on lake type
   223)     real(r8) :: eta(2)                  ! light extinction coefficient (/m): depends on lake type
   224)     real(r8) :: a(lbc:ubc,nlevlak)      ! "a" vector for tridiagonal matrix
   225)     real(r8) :: b(lbc:ubc,nlevlak)      ! "b" vector for tridiagonal matrix
   226)     real(r8) :: c1(lbc:ubc,nlevlak)     ! "c" vector for tridiagonal matrix
   227)     real(r8) :: r(lbc:ubc,nlevlak)      ! "r" vector for tridiagonal solution
   228)     real(r8) :: rhow(lbc:ubc,nlevlak)   ! density of water (kg/m**3)
   229)     real(r8) :: phi(lbc:ubc,nlevlak)    ! solar radiation absorbed by layer (w/m**2)
   230)     real(r8) :: kme(lbc:ubc,nlevlak)    ! molecular + eddy diffusion coefficient (m**2/s)
   231)     real(r8) :: cwat                    ! specific heat capacity of water (j/m**3/kelvin)
   232)     real(r8) :: ws(lbc:ubc)             ! surface friction velocity (m/s)
   233)     real(r8) :: ks(lbc:ubc)             ! coefficient
   234)     real(r8) :: in                      ! relative flux of solar radiation into layer
   235)     real(r8) :: out                     ! relative flux of solar radiation out of layer
   236)     real(r8) :: ri                      ! richardson number
   237)     real(r8) :: fin(lbc:ubc)            ! heat flux into lake - flux out of lake (w/m**2)
   238)     real(r8) :: ocvts(lbc:ubc)          ! (cwat*(t_lake[n  ])*dz
   239)     real(r8) :: ncvts(lbc:ubc)          ! (cwat*(t_lake[n+1])*dz
   240)     real(r8) :: m1                      ! intermediate variable for calculating r, a, b, c
   241)     real(r8) :: m2                      ! intermediate variable for calculating r, a, b, c
   242)     real(r8) :: m3                      ! intermediate variable for calculating r, a, b, c
   243)     real(r8) :: ke                      ! eddy diffusion coefficient (m**2/s)
   244)     real(r8) :: km                      ! molecular diffusion coefficient (m**2/s)
   245)     real(r8) :: zin                     ! depth at top of layer (m)
   246)     real(r8) :: zout                    ! depth at bottom of layer (m)
   247)     real(r8) :: drhodz                  ! d [rhow] /dz (kg/m**4)
   248)     real(r8) :: n2                      ! brunt-vaisala frequency (/s**2)
   249)     real(r8) :: num                     ! used in calculating ri
   250)     real(r8) :: den                     ! used in calculating ri
   251)     real(r8) :: tav(lbc:ubc)            ! used in aver temp for convectively mixed layers
   252)     real(r8) :: nav(lbc:ubc)            ! used in aver temp for convectively mixed layers
   253)     real(r8) :: phidum                  ! temporary value of phi
   254)     real(r8) :: u2m                     ! 2 m wind speed (m/s)
   255)     real(r8) :: fm(lbp:ubp)             ! needed for BGC only to diagnose 10m wind speed
   256)     real(r8) :: e_ref2m                 ! 2 m height surface saturated vapor pressure [Pa]
   257)     real(r8) :: de2mdT                  ! derivative of 2 m height surface saturated vapor pressure on t_ref2m
   258)     real(r8) :: qsat_ref2m              ! 2 m height surface saturated specific humidity [kg/kg]
   259)     real(r8) :: dqsat2mdT               ! derivative of 2 m height surface saturated specific humidity on t_ref2m
   260) !
   261) ! Constants for lake temperature model
   262) !
   263)     data beta/0.4_r8, 0.4_r8/  ! (deep lake, shallow lake)
   264)     data za  /0.6_r8, 0.5_r8/
   265)     data eta /0.1_r8, 0.5_r8/
   266) !-----------------------------------------------------------------------
   267) 
   268)     ! Assign local pointers to derived type members (gridcell-level)
   269) 
   270)     forc_t         => clm_a2l%forc_t
   271)     forc_pbot      => clm_a2l%forc_pbot
   272)     forc_th        => clm_a2l%forc_th
   273)     forc_q         => clm_a2l%forc_q
   274)     forc_u         => clm_a2l%forc_u
   275)     forc_v         => clm_a2l%forc_v
   276)     forc_rho       => clm_a2l%forc_rho
   277)     forc_lwrad     => clm_a2l%forc_lwrad
   278)     forc_snow      => clm_a2l%forc_snow
   279)     forc_rain      => clm_a2l%forc_rain
   280)     lat            => grc%lat
   281) 
   282)     ! Assign local pointers to derived type members (column-level)
   283) 
   284)     cgridcell      => col%gridcell
   285)     dz             => cps%dz
   286)     z              => cps%z
   287)     t_lake         => ces%t_lake
   288)     h2osno         => cws%h2osno
   289)     snowdp         => cps%snowdp
   290)     t_grnd         => ces%t_grnd
   291)     hc_soisno      => ces%hc_soisno
   292)     errsoi         => cebal%errsoi
   293)     qmelt          => cwf%qmelt
   294) 
   295)     ! Assign local pointers to derived type members (pft-level)
   296) 
   297)     pcolumn        => pft%column
   298)     pgridcell      => pft%gridcell
   299)     sabg           => pef%sabg
   300)     t_ref2m        => pes%t_ref2m
   301)     q_ref2m        => pes%q_ref2m
   302)     rh_ref2m       => pes%rh_ref2m
   303)     t_veg          => pes%t_veg
   304)     eflx_lwrad_out => pef%eflx_lwrad_out
   305)     eflx_lwrad_net => pef%eflx_lwrad_net
   306)     eflx_soil_grnd => pef%eflx_soil_grnd
   307)     eflx_lh_tot    => pef%eflx_lh_tot
   308)     eflx_lh_grnd   => pef%eflx_lh_grnd
   309)     eflx_sh_grnd   => pef%eflx_sh_grnd
   310)     eflx_sh_tot    => pef%eflx_sh_tot
   311)     ram1           => pps%ram1
   312)     taux           => pmf%taux
   313)     tauy           => pmf%tauy
   314)     qflx_prec_grnd => pwf%qflx_prec_grnd
   315)     qflx_evap_soi  => pwf%qflx_evap_soi
   316)     qflx_evap_tot  => pwf%qflx_evap_tot
   317)     forc_hgt_u_pft => pps%forc_hgt_u_pft
   318)     forc_hgt_t_pft => pps%forc_hgt_t_pft
   319)     forc_hgt_q_pft => pps%forc_hgt_q_pft
   320)     qflx_snwcp_ice => pwf%qflx_snwcp_ice    
   321)     qflx_snwcp_liq => pwf%qflx_snwcp_liq    
   322) 
   323)     ! Determine step size
   324) 
   325)     dtime = get_step_size()
   326) 
   327)     ! Begin calculations
   328) 
   329)     do fc = 1, num_lakec
   330)        c = filter_lakec(fc)
   331)        g = cgridcell(c)
   332) 
   333)        ! Initialize quantities computed below
   334) 
   335)        ocvts(c) = 0._r8
   336)        ncvts(c) = 0._r8
   337)        hc_soisno(c) = 0._r8
   338) 
   339)        ! Surface temperature and fluxes
   340) 
   341)        dzsur(c) = dz(c,1) + snowdp(c)
   342) 
   343)        ! Saturated vapor pressure, specific humidity and their derivatives
   344)        ! at lake surface
   345) 
   346)        call QSat(t_grnd(c), forc_pbot(g), eg, degdT, qsatg(c), qsatgdT(c))
   347) 
   348)        ! Potential, virtual potential temperature, and wind speed at the
   349)        ! reference height
   350) 
   351)        !zii = 1000.    ! m  (pbl height)
   352)        thv(c) = forc_th(g)*(1._r8+0.61_r8*forc_q(g))     ! virtual potential T
   353)     end do
   354) 
   355)     do fp = 1, num_lakep
   356)        p = filter_lakep(fp)
   357)        c = pcolumn(p)
   358)        g = pgridcell(p)
   359) 
   360)        nmozsgn(p) = 0
   361)        obuold(p) = 0._r8
   362)        displa(p) = 0._r8
   363)        thm(p) = forc_t(g) + 0.0098_r8*forc_hgt_t_pft(p)   ! intermediate variable
   364) 
   365)        ! Roughness lengths
   366) 
   367)        if (t_grnd(c) >= tfrz) then   ! for unfrozen lake
   368)           z0mg(p) = 0.01_r8
   369)        else                          ! for frozen lake
   370)           z0mg(p) = 0.04_r8
   371)        end if
   372)        z0hg(p) = z0mg(p)
   373)        z0qg(p) = z0mg(p)
   374) 
   375)        ! Latent heat
   376) 
   377)        if (forc_t(g) > tfrz) then
   378)           htvp(c) = hvap
   379)        else
   380)           htvp(c) = hsub
   381)        end if
   382) 
   383)        ! Initialize stability variables
   384)
