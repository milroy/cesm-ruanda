    23)   !... Exception to the Atkinson et al. reference  (@0K unless noted)

   101)   !-------------------------------------------------------------------------
   102)   subroutine cph( cph_tot, vmr, rxt, cp, mbar, kbot, ncol, lchnk )
   103) 
   104)     !-----------------------------------------------------------------------
   105)     !      	... forms the chemical potential heating rates
   106)     !-----------------------------------------------------------------------
   107) 
   108)     use chem_mods,     only : gas_pcnst, rxntot
   109)     use ppgrid,        only : pver
   110)     use cam_history,   only : outfld
   111)     use mo_rxt_rates_conv, only : set_rates
   112) 
   113)     implicit none
   114) 
   115)     !-----------------------------------------------------------------------
   116)     !     	... dummy arguments
   117)     !-----------------------------------------------------------------------
   118)     integer, intent(in)   ::  ncol                                ! columns in chunck
   119)     integer, intent(in)   ::  lchnk                               ! chunk index
   120)     integer, intent(in)   ::  kbot                                ! bottom vert index
   121)     real(r8), intent(in)  ::  rxt(ncol,pver,rxntot)               ! rxt rates (1/cm^3/s)
   122)     real(r8), intent(in)  ::  cp(ncol,pver)                       ! specific heat capacity (J/K/kg)
   123)     real(r8), intent(in)  ::  mbar(ncol,pver)                     ! atm mean mass (g/mole)
   124)     real(r8), intent(in)  ::  vmr(ncol,pver,gas_pcnst)            ! concentrations (mol/mol)
   125)     real(r8), intent(out) ::  cph_tot(ncol,pver)                  ! total heating (K/s)
   126) 
   127)     !-----------------------------------------------------------------------
   128)     !     	... local variables
   129)     !-----------------------------------------------------------------------
   130)     integer  ::  i, k
   131)     real(r8) ::  tmp(ncol,pver)
   132)     real(r8) ::  cph_rate(ncol,pver,ncph)
   133)     real(r8) ::  rxt_rates(ncol,pver,rxntot)
   134) 
   135)     if (.not.has_cph) return
   136) 
   137)     ! get the reaction rates from rate constants ...
   138)     rxt_rates(:ncol,:,:) = rxt(:ncol,:,:)
   139)     call set_rates( rxt_rates, vmr, ncol )
   140) 
   141)     ! compute corresponding chem heating rates ...
   142)     cph_rate(:,:,:) = 0._r8
   143)     tmp(:ncol,:) =  1._r8 / (1.e-6_r8*cp(:ncol,:)*mbar(:ncol,:))
   144)     do i = 1,ncph
   145)        cph_rate(:ncol,:,i) = tmp(:ncol,:) * rxt_rates(:ncol,:,cph_rid(i)) * exotherm(i)
   146)     enddo
   147) 
   148)     ! compute total heating rate ...
   149)     cph_tot(:,:) = 0._r8
   150)     do k = 1,kbot
   151)        do i = 1,ncol
   152)           cph_tot(i,k) = sum( cph_rate(i,k,:) )
   153)        end do
   154)     end do
   155) 
   156)     ! output diagnostics
   157)     do i = 1,ncph
   158)        if ( exotherm(i)>0._r8) then
   159)           call outfld( fldnames(i), cph_rate(:,:,i), ncol, lchnk )
   160)        endif
   161)     enddo
   162) 
   163)     call outfld( 'QCP', cph_tot(:,:), ncol, lchnk )
   164) 
   165)   end subroutine cph
   166)
