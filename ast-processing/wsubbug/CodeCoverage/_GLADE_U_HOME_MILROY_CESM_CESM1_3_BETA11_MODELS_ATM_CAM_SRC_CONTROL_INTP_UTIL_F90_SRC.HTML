    49)    !----------------------------------------------------------------------- 
    50)    ! Purpose: 
    51)    ! "find periodic lower bound"
    52)    ! Search the input array for the lower bound of the interval that
    53)    ! contains the input value.  The returned index satifies:
    54)    ! x(index) .le. xval .lt. x(index+1)
    55)    ! Assume the array represents values in one cycle of a periodic coordinate.
    56)    ! So, if xval .lt. x(1), or xval .ge. x(nx), then the index returned is nx.
    57)    !
    58)    ! Author: B. Eaton
    59)    !----------------------------------------------------------------------- 
    60) 
    61)    use shr_kind_mod, only: r8 => shr_kind_r8
    62) 
    63)    implicit none
    64) 
    65)    integer, intent(in) ::   nx         ! size of x
    66)    real(r8), intent(in) ::  x(nx)      ! strictly increasing array
    67)    real(r8), intent(in) ::  xval       ! value to be searched for in x
    68)    
    69)    integer, intent(out) ::  index
    70) 
    71)    ! Local variables:
    72)    integer i
    73)    !-----------------------------------------------------------------------
    74) 
    75)    if ( xval .lt. x(1) .or. xval .ge. x(nx) ) then
    76)       index = nx
    77)       return
    78)    end if
    79) 
    80)    do i = 2, nx
    81)       if ( xval .lt. x(i) ) then
    82)          index = i-1
    83)          return
    84)       end if
    85)    end do
