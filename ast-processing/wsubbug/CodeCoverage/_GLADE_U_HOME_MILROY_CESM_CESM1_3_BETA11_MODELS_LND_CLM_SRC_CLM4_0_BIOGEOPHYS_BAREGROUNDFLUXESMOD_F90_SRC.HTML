    36)   subroutine BareGroundFluxes(lbp, ubp, num_nolakep, filter_nolakep)
    37) !
    38) ! !DESCRIPTION:
    39) ! Compute sensible and latent fluxes and their derivatives with respect
    40) ! to ground temperature using ground temperatures from previous time step.
    41) !
    42) ! !USES:
    43)     use clmtype
    44)     use clm_atmlnd         , only : clm_a2l
    45)     use clm_varpar         , only : nlevgrnd
    46)     use clm_varcon         , only : cpair, vkc, grav, denice, denh2o, istsoil
    47)     use clm_varcon         , only : istcrop
    48)     use clm_varctl         , only : use_c13
    49)     use shr_const_mod      , only : SHR_CONST_RGAS
    50)     use FrictionVelocityMod, only : FrictionVelocity, MoninObukIni
    51)     use QSatMod            , only : QSat
    52) 
    53) !
    54) ! !ARGUMENTS:
    55)     implicit none
    56)     integer, intent(in) :: lbp, ubp                     ! pft bounds
    57)     integer, intent(in) :: num_nolakep                  ! number of pft non-lake points in pft filter
    58)     integer, intent(in) :: filter_nolakep(ubp-lbp+1)    ! pft filter for non-lake points
    59) !
    60) ! !CALLED FROM:
    61) ! subroutine Biogeophysics1 in module Biogeophysics1Mod
    62) !
    63) ! !REVISION HISTORY:
    64) ! 15 September 1999: Yongjiu Dai; Initial code
    65) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
    66) ! 12/19/01, Peter Thornton
    67) ! This routine originally had a long list of parameters, and also a reference to
    68) ! the entire clm derived type.  For consistency, only the derived type reference
    69) ! is passed (now pointing to the current column and pft), and the other original
    70) ! parameters are initialized locally. Using t_grnd instead of tg (tg eliminated
    71) ! as redundant).
    72) ! 1/23/02, PET: Added pft reference as parameter. All outputs will be written
    73) ! to the pft data structures, and averaged to the column level outside of
    74) ! this routine.
    75) !
    76) ! !LOCAL VARIABLES:
    77) !
    78) ! local pointers to implicit in arguments
    79) !
    80)     integer , pointer :: pcolumn(:)        ! pft's column index
    81)     integer , pointer :: pgridcell(:)      ! pft's gridcell index
    82)     integer , pointer :: plandunit(:)      ! pft's landunit index
    83)     integer , pointer :: ltype(:)          ! landunit type
    84)     integer , pointer :: frac_veg_nosno(:) ! fraction of vegetation not covered by snow (0 OR 1) [-]
    85)     real(r8), pointer :: t_grnd(:)         ! ground surface temperature [K]
    86)     real(r8), pointer :: thm(:)            ! intermediate variable (forc_t+0.0098*forc_hgt_t_pft)
    87)     real(r8), pointer :: qg(:)             ! specific humidity at ground surface [kg/kg]
    88)     real(r8), pointer :: thv(:)            ! virtual potential temperature (kelvin)
    89)     real(r8), pointer :: dqgdT(:)          ! temperature derivative of "qg"
    90)     real(r8), pointer :: htvp(:)           ! latent heat of evaporation (/sublimation) [J/kg]
    91)     real(r8), pointer :: beta(:)           ! coefficient of conective velocity [-]
    92)     real(r8), pointer :: zii(:)            ! convective boundary height [m]
    93)     real(r8), pointer :: forc_u(:)         ! atmospheric wind speed in east direction (m/s)
    94)     real(r8), pointer :: forc_v(:)         ! atmospheric wind speed in north direction (m/s)
    95)     real(r8), pointer :: forc_t(:)         ! atmospheric temperature (Kelvin)
    96)     real(r8), pointer :: forc_th(:)        ! atmospheric potential temperature (Kelvin)
    97)     real(r8), pointer :: forc_q(:)         ! atmospheric specific humidity (kg/kg)
    98)     real(r8), pointer :: forc_rho(:)       ! density (kg/m**3)
    99)     real(r8), pointer :: forc_pbot(:)      ! atmospheric pressure (Pa)
   100)     real(r8), pointer :: forc_hgt_u_pft(:) ! observational height of wind at pft level [m]
   101)     real(r8), pointer :: psnsun(:)         ! sunlit leaf photosynthesis (umol CO2 /m**2/ s)
   102)     real(r8), pointer :: psnsha(:)         ! shaded leaf photosynthesis (umol CO2 /m**2/ s)
   103)     real(r8), pointer :: z0mg_col(:)       ! roughness length, momentum [m]
   104)     real(r8), pointer :: h2osoi_ice(:,:)   ! ice lens (kg/m2)
   105)     real(r8), pointer :: h2osoi_liq(:,:)   ! liquid water (kg/m2)
   106)     real(r8), pointer :: dz(:,:)           ! layer depth (m)
   107)     real(r8), pointer :: watsat(:,:)       ! volumetric soil water at saturation (porosity)
   108)     real(r8), pointer :: frac_sno(:)       ! fraction of ground covered by snow (0 to 1)
   109)     real(r8), pointer :: soilbeta(:)       ! soil wetness relative to field capacity
   110) !
   111) ! local pointers to implicit inout arguments
   112) !
   113)     real(r8), pointer :: z0hg_col(:)       ! roughness length, sensible heat [m]
   114)     real(r8), pointer :: z0qg_col(:)       ! roughness length, latent heat [m]
   115) !
   116) ! local pointers to implicit out arguments
   117) !
   118)     real(r8), pointer :: dlrad(:)         ! downward longwave radiation below the canopy [W/m2]
   119)     real(r8), pointer :: ulrad(:)         ! upward longwave radiation above the canopy [W/m2]
   120)     real(r8), pointer :: cgrnds(:)        ! deriv, of soil sensible heat flux wrt soil temp [w/m2/k]
   121)     real(r8), pointer :: cgrndl(:)        ! deriv of soil latent heat flux wrt soil temp [w/m**2/k]
   122)     real(r8), pointer :: cgrnd(:)         ! deriv. of soil energy flux wrt to soil temp [w/m2/k]
   123)     real(r8), pointer :: taux(:)          ! wind (shear) stress: e-w (kg/m/s**2)
   124)     real(r8), pointer :: tauy(:)          ! wind (shear) stress: n-s (kg/m/s**2)
   125)     real(r8), pointer :: eflx_sh_grnd(:)  ! sensible heat flux from ground (W/m**2) [+ to atm]
   126)     real(r8), pointer :: eflx_sh_tot(:)   ! total sensible heat flux (W/m**2) [+ to atm]
   127)     real(r8), pointer :: qflx_evap_soi(:) ! soil evaporation (mm H2O/s) (+ = to atm)
   128)     real(r8), pointer :: qflx_evap_tot(:) ! qflx_evap_soi + qflx_evap_can + qflx_tran_veg
   129)     real(r8), pointer :: t_ref2m(:)       ! 2 m height surface air temperature (Kelvin)
   130)     real(r8), pointer :: q_ref2m(:)       ! 2 m height surface specific humidity (kg/kg)
   131)     real(r8), pointer :: t_ref2m_r(:)     ! Rural 2 m height surface air temperature (Kelvin)
   132)     real(r8), pointer :: rh_ref2m_r(:)    ! Rural 2 m height surface relative humidity (%)
   133)     real(r8), pointer :: rh_ref2m(:)      ! 2 m height surface relative humidity (%)
   134)     real(r8), pointer :: t_veg(:)         ! vegetation temperature (Kelvin)
   135)     real(r8), pointer :: btran(:)         ! transpiration wetness factor (0 to 1)
   136)     real(r8), pointer :: rssun(:)         ! sunlit stomatal resistance (s/m)
   137)     real(r8), pointer :: rssha(:)         ! shaded stomatal resistance (s/m)
   138)     real(r8), pointer :: ram1(:)          ! aerodynamical resistance (s/m)
   139)     real(r8), pointer :: fpsn(:)          ! photosynthesis (umol CO2 /m**2 /s)
   140)     real(r8), pointer :: rootr(:,:)       ! effective fraction of roots in each soil layer
   141)     real(r8), pointer :: rresis(:,:)      ! root resistance by layer (0-1)  (nlevgrnd)	
   142) !
   143) !
   144) ! !OTHER LOCAL VARIABLES:
   145) !EOP
   146) !
   147)     integer, parameter  :: niters = 3  ! maximum number of iterations for surface temperature
   148)     integer  :: p,c,g,f,j,l            ! indices
   149)     integer  :: filterp(ubp-lbp+1)     ! pft filter for vegetated pfts
   150)     integer  :: fn                     ! number of values in local pft filter
   151)     integer  :: fp                     ! lake filter pft index
   152)     integer  :: iter                   ! iteration index
   153)     real(r8) :: zldis(lbp:ubp)         ! reference height "minus" zero displacement height [m]
   154)     real(r8) :: displa(lbp:ubp)        ! displacement height [m]
   155)     real(r8) :: zeta                   ! dimensionless height used in Monin-Obukhov theory
   156)     real(r8) :: wc                     ! convective velocity [m/s]
   157)     real(r8) :: dth(lbp:ubp)           ! diff of virtual temp. between ref. height and surface
   158)     real(r8) :: dthv                   ! diff of vir. poten. temp. between ref. height and surface
   159)     real(r8) :: dqh(lbp:ubp)           ! diff of humidity between ref. height and surface
   160)     real(r8) :: obu(lbp:ubp)           ! Monin-Obukhov length (m)
   161)     real(r8) :: ur(lbp:ubp)            ! wind speed at reference height [m/s]
   162)     real(r8) :: um(lbp:ubp)            ! wind speed including the stablity effect [m/s]
   163)     real(r8) :: temp1(lbp:ubp)         ! relation for potential temperature profile
   164)     real(r8) :: temp12m(lbp:ubp)       ! relation for potential temperature profile applied at 2-m
   165)     real(r8) :: temp2(lbp:ubp)         ! relation for specific humidity profile
   166)     real(r8) :: temp22m(lbp:ubp)       ! relation for specific humidity profile applied at 2-m
   167)     real(r8) :: ustar(lbp:ubp)         ! friction velocity [m/s]
   168)     real(r8) :: tstar                  ! temperature scaling parameter
   169)     real(r8) :: qstar                  ! moisture scaling parameter
   170)     real(r8) :: thvstar                ! virtual potential temperature scaling parameter
   171)     real(r8) :: cf                     ! heat transfer coefficient from leaves [-]
   172)     real(r8) :: ram                    ! aerodynamical resistance [s/m]
   173)     real(r8) :: rah                    ! thermal resistance [s/m]
   174)     real(r8) :: raw                    ! moisture resistance [s/m]
   175)     real(r8) :: raih                   ! temporary variable [kg/m2/s]
   176)     real(r8) :: raiw                   ! temporary variable [kg/m2/s]
   177)     real(r8) :: fm(lbp:ubp)            ! needed for BGC only to diagnose 10m wind speed
   178)     real(r8) :: z0mg_pft(lbp:ubp)
   179)     real(r8) :: z0hg_pft(lbp:ubp)
   180)     real(r8) :: z0qg_pft(lbp:ubp)
   181)     real(r8) :: e_ref2m                ! 2 m height surface saturated vapor pressure [Pa]
   182)     real(r8) :: de2mdT                 ! derivative of 2 m height surface saturated vapor pressure on t_ref2m
   183)     real(r8) :: qsat_ref2m             ! 2 m height surface saturated specific humidity [kg/kg]
   184)     real(r8) :: dqsat2mdT              ! derivative of 2 m height surface saturated specific humidity on t_ref2m 
   185)     real(r8) :: www                    ! surface soil wetness [-]
   186) !------------------------------------------------------------------------------
   187) 
   188)     ! Assign local pointers to derived type members (gridcell-level)
   189) 
   190)     forc_u     => clm_a2l%forc_u
   191)     forc_v     => clm_a2l%forc_v
   192) 
   193)     ! Assign local pointers to derived type members (landunit-level)
   194) 
   195)     ltype      => lun%itype
   196) 
   197)     ! Assign local pointers to derived type members (column-level)
   198) 
   199)     forc_th    => ces%forc_th
   200)     forc_t     => ces%forc_t
   201)     forc_pbot  => cps%forc_pbot
   202)     forc_rho   => cps%forc_rho
   203)     forc_q     => cws%forc_q
   204)     pcolumn    => pft%column
   205)     pgridcell  => pft%gridcell
   206)     frac_veg_nosno => pps%frac_veg_nosno
   207)     dlrad  => pef%dlrad
   208)     ulrad  => pef%ulrad
   209)     t_grnd => ces%t_grnd
   210)     qg     => cws%qg
   211)     z0mg_col => cps%z0mg
   212)     z0hg_col => cps%z0hg
   213)     z0qg_col => cps%z0qg
   214)     thv    => ces%thv
   215)     beta   => cps%beta
   216)     zii    => cps%zii
   217)     ram1   => pps%ram1
   218)     cgrnds => pef%cgrnds
   219)     cgrndl => pef%cgrndl
   220)     cgrnd  => pef%cgrnd
   221)     dqgdT  => cws%dqgdT
   222)     htvp   => cps%htvp
   223)     watsat         => cps%watsat
   224)     h2osoi_ice     => cws%h2osoi_ice
   225)     dz             => cps%dz
   226)     h2osoi_liq     => cws%h2osoi_liq
   227)     frac_sno       => cps%frac_sno
   228)     soilbeta       => cws%soilbeta
   229) 
   230)     ! Assign local pointers to derived type members (pft-level)
   231) 
   232)     taux => pmf%taux
   233)     tauy => pmf%tauy
   234)     eflx_sh_grnd => pef%eflx_sh_grnd
   235)     eflx_sh_tot => pef%eflx_sh_tot
   236)     qflx_evap_soi => pwf%qflx_evap_soi
   237)     qflx_evap_tot => pwf%qflx_evap_tot
   238)     t_ref2m => pes%t_ref2m
   239)     q_ref2m => pes%q_ref2m
   240)     t_ref2m_r => pes%t_ref2m_r
   241)     rh_ref2m_r => pes%rh_ref2m_r
   242)     plandunit => pft%landunit
   243)     rh_ref2m => pes%rh_ref2m
   244)     t_veg => pes%t_veg
   245)     thm => pes%thm
   246)     btran => pps%btran
   247)     rssun => pps%rssun
   248)     rssha => pps%rssha
   249)     rootr => pps%rootr
   250)     rresis => pps%rresis
   251)     psnsun => pcf%psnsun
   252)     psnsha => pcf%psnsha
   253)     fpsn => pcf%fpsn
   254)     forc_hgt_u_pft => pps%forc_hgt_u_pft
   255) 
   256)     ! Filter pfts where frac_veg_nosno is zero
   257) 
   258)     fn = 0
   259)     do fp = 1,num_nolakep
   260)        p = filter_nolakep(fp)
   261)        if (frac_veg_nosno(p) == 0) then
   262)           fn = fn + 1
   263)           filterp(fn) = p
   264)        end if
   265)     end do
   266) 
   267)     ! Compute sensible and latent fluxes and their derivatives with respect
   268)     ! to ground temperature using ground temperatures from previous time step
   269) 
   270)     do f = 1, fn
   271)        p = filterp(f)
   272)        c = pcolumn(p)
   273)        g = pgridcell(p)
   274) 
   275)        ! Initialization variables
   276) 
   277)        displa(p) = 0._r8
   278)        dlrad(p)  = 0._r8
   279)        ulrad(p)  = 0._r8
   280) 
   281)        ur(p) = max(1.0_r8,sqrt(forc_u(g)*forc_u(g)+forc_v(g)*forc_v(g)))
   282)        dth(p) = thm(p)-t_grnd(c)
   283)        dqh(p) = forc_q(c) - qg(c)
   284)        dthv = dth(p)*(1._r8+0.61_r8*forc_q(c))+0.61_r8*forc_th(c)*dqh(p)
   285)        zldis(p) = forc_hgt_u_pft(p)
   286) 
   287)        ! Copy column roughness to local pft-level arrays
   288) 
   289)        z0mg_pft(p) = z0mg_col(c)
   290)        z0hg_pft(p) = z0hg_col(c)
   291)        z0qg_pft(p) = z0qg_col(c)
   292) 
   293)        ! Initialize Monin-Obukhov length and wind speed
   294) 
   295)        call MoninObukIni(ur(p), thv(c), dthv, zldis(p), z0mg_pft(p), um(p), obu(p))
   296) 
   297)     end do
   298) 
   299)     ! Perform stability iteration
   300)     ! Determine friction velocity, and potential temperature and humidity
   301)     ! profiles of the surface boundary layer
   302) 
   303)     do iter = 1, niters
   304) 
   305)        call FrictionVelocity(lbp, ubp, fn, filterp, &
   306)                              displa, z0mg_pft, z0hg_pft, z0qg_pft, &
   307)                              obu, iter, ur, um, ustar, &
   308)                              temp1, temp2, temp12m, temp22m, fm)
   309) 
   310)        do f = 1, fn
   311)           p = filterp(f)
   312)           c = pcolumn(p)
   313)           g = pgridcell(p)
   314) 
   315)           tstar = temp1(p)*dth(p)
   316)           qstar = temp2(p)*dqh(p)
   317)           z0hg_pft(p) = z0mg_pft(p)/exp(0.13_r8 * (ustar(p)*z0mg_pft(p)/1.5e-5_r8)**0.45_r8)
   318)           z0qg_pft(p) = z0hg_pft(p)
   319)           thvstar = tstar*(1._r8+0.61_r8*forc_q(c)) + 0.61_r8*forc_th(c)*qstar
   320)           zeta = zldis(p)*vkc*grav*thvstar/(ustar(p)**2*thv(c))
   321) 
   322)           if (zeta >= 0._r8) then                   !stable
   323)              zeta = min(2._r8,max(zeta,0.01_r8))
   324)              um(p) = max(ur(p),0.1_r8)
   325)           else                                      !unstable
   326)              zeta = max(-100._r8,min(zeta,-0.01_r8))
   327)              wc = beta(c)*(-grav*ustar(p)*thvstar*zii(c)/thv(c))**0.333_r8
   328)              um(p) = sqrt(ur(p)*ur(p) + wc*wc)
   329)           end if
   330)           obu(p) = zldis(p)/zeta
   331)        end do
   332) 
   333)     end do ! end stability iteration
   334) 
   335)      do j = 1, nlevgrnd
   336)        do f = 1, fn
   337)           p = filterp(f)
   338)           rootr(p,j) = 0._r8
   339)           rresis(p,j) = 0._r8
   340)         end do
   341)      end do
   342) 
   343)     do f = 1, fn
   344)        p = filterp(f)
   345)        c = pcolumn(p)
   346)        g = pgridcell(p)
   347)        l = plandunit(p)
   348) 
   349)        ! Determine aerodynamic resistances
   350) 
   351)        ram     = 1._r8/(ustar(p)*ustar(p)/um(p))
   352)        rah     = 1._r8/(temp1(p)*ustar(p))
   353)        raw     = 1._r8/(temp2(p)*ustar(p))
   354)        raih    = forc_rho(c)*cpair/rah
   355) 
   356)        ! Soil evaporation resistance
   357)        www     = (h2osoi_liq(c,1)/denh2o+h2osoi_ice(c,1)/denice)/dz(c,1)/watsat(c,1)
   358)        www     = min(max(www,0.0_r8),1._r8)
   359) 
   360)        !changed by K.Sakaguchi. Soilbeta is used for evaporation
   361)        if (dqh(p) .gt. 0._r8) then   !dew  (beta is not applied, just like rsoil used to be)
   362)           raiw    = forc_rho(c)/(raw)
   363)        else
   364)        ! Lee and Pielke 1992 beta is applied
   365)           raiw    = soilbeta(c)*forc_rho(c)/(raw)
   366)        end if
   367) 
   368)        ram1(p) = ram  !pass value to global variable
   369) 
   370)        ! Output to pft-level data structures
   371)        ! Derivative of fluxes with respect to ground temperature
   372) 
   373)        cgrnds(p) = raih
   374)        cgrndl(p) = raiw*dqgdT(c)
   375)        cgrnd(p)  = cgrnds(p) + htvp(c)*cgrndl(p)
   376) 
   377)        ! Surface fluxes of momentum, sensible and latent heat
   378)        ! using ground temperatures from previous time step
   379) 
   380)        taux(p)          = -forc_rho(c)*forc_u(g)/ram
   381)        tauy(p)          = -forc_rho(c)*forc_v(g)/ram
   382)        eflx_sh_grnd(p)  = -raih*dth(p)
   383)        eflx_sh_tot(p)   = eflx_sh_grnd(p)
   384)        qflx_evap_soi(p) = -raiw*dqh(p)
   385)        qflx_evap_tot(p) = qflx_evap_soi(p)
   386) 
   387)        ! 2 m height air temperature
   388) 
   389)        t_ref2m(p) = thm(p) + temp1(p)*dth(p)*(1._r8/temp12m(p) - 1._r8/temp1(p))
   390) 
   391)        ! 2 m height specific humidity
   392) 
   393)        q_ref2m(p) = forc_q(c) + temp2(p)*dqh(p)*(1._r8/temp22m(p) - 1._r8/temp2(p))
   394) 
   395)        ! 2 m height relative humidity
   396)                                                                                 
   397)        call QSat(t_ref2m(p), forc_pbot(c), e_ref2m, de2mdT, qsat_ref2m, dqsat2mdT)
   398) 
   399)        rh_ref2m(p) = min(100._r8, q_ref2m(p) / qsat_ref2m * 100._r8)
   400) 
   401)        if (ltype(l) == istsoil .or. ltype(l) == istcrop) then
   402)          rh_ref2m_r(p) = rh_ref2m(p)
   403)          t_ref2m_r(p) = t_ref2m(p)
   404)        end if
   405) 
   406)        ! Variables needed by history tape
   407) 
   408)        t_veg(p) = forc_t(c)
   409)        btran(p) = 0._r8
   410)        cf = forc_pbot(c)/(SHR_CONST_RGAS*0.001_r8*thm(p))*1.e06_r8
   411)        rssun(p) = 1._r8/1.e15_r8 * cf
   412)        rssha(p) = 1._r8/1.e15_r8 * cf
   413) 
   414)        ! Add the following to avoid NaN
   415) 
   416)        psnsun(p) = 0._r8
   417)        psnsha(p) = 0._r8
   418)        fpsn(p) = 0._r8
   419)        pps%lncsun(p) = 0._r8
   420)        pps%lncsha(p) = 0._r8
   421)        pps%vcmxsun(p) = 0._r8
   422)        pps%vcmxsha(p) = 0._r8
   423)        ! adding code for isotopes, 8/17/05, PET
   424)        pps%cisun(p) = 0._r8
   425)        pps%cisha(p) = 0._r8
