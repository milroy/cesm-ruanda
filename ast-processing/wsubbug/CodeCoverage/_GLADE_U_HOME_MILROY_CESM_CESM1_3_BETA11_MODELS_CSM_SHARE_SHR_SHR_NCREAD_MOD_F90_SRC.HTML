   346) 
   347)    implicit none
   348) 
   349) ! !INPUT/OUTPUT PARAMETERS:
   350) 
   351)    character(*)        ,intent(in)           :: fileName ! nc file name
   352)    character(*)        ,intent(in)           :: varName  ! name of variable
   353)    integer(SHR_KIND_IN),intent(out),optional :: n1       ! size of dim1 in var
   354)    integer(SHR_KIND_IN),intent(out),optional :: n2       ! size of dim2 in var
   355)    integer(SHR_KIND_IN),intent(out),optional :: n3       ! size of dim3 in var
   356)    integer(SHR_KIND_IN),intent(out),optional :: n4       ! size of dim4 in var

   358)    integer(SHR_KIND_IN),intent(out),optional :: n6       ! size of dim6 in var

   360) 
   361) !EOP
   362) 
   363)    !----- local -----
   364)    integer(SHR_KIND_IN),parameter :: maxn = 6     ! max number of dims available
   365)    integer(SHR_KIND_IN) :: n                      ! counter
   366)    integer(SHR_KIND_IN) :: fid                    ! file id
   367)    integer(SHR_KIND_IN) :: vid                    ! variable id
   368)    integer(SHR_KIND_IN) :: ndims                  ! number of dims
   369)    integer(SHR_KIND_IN),allocatable :: dids(:)    ! dimids
   370)    integer(SHR_KIND_IN),allocatable :: ns(:)      ! size of dims
   371)    integer(SHR_KIND_IN) :: rCode                  ! error code
   372) 
   373)    !----- formats -----
   374)    character(*),parameter :: subName = "(shr_ncread_varDimSizes)"
   375)    character(*),parameter :: F00     = "('(shr_ncread_varDimSizes) ',4a)"
   376)    character(*),parameter :: F01     = "('(shr_ncread_varDimSizes) ',a,i6)"
   377) 
   378) !-------------------------------------------------------------------------------
   379) ! 
   380) !-------------------------------------------------------------------------------
   381) 
   382)    call shr_ncread_open(fileName,fid,rCode)
   383) 
   384)    rCode = nf90_inq_varid(fid,trim(varName),vid)
   385)    call shr_ncread_handleErr(rCode,subName//' ERROR inq varid vid')
   386)    rCode = nf90_inquire_variable(fid,vid,ndims=ndims)
   387)    call shr_ncread_handleErr(rCode,subName//' ERROR inquire variable ndims')
   388)    allocate(dids(ndims))
   389)    allocate(ns(maxn))
   390)    rCode = nf90_inquire_variable(fid,vid,dimids=dids)
   391)    call shr_ncread_handleErr(rCode,subName//' ERROR inquire variable dimids')
   392) 
   393)    !--- get dim sizes for all dims or to maxn, default result is 1 ---
   394)    ns = 1
   395)    do n=1,min(ndims,maxn)
   396)      rcode = nf90_inquire_dimension(fid,dids(n),len=ns(n))
   397)      call shr_ncread_handleErr(rCode, subName//" ERROR inquire dimension")
   398)    enddo
   399) 
   400)    call shr_ncread_close(fid,rCode)
   401) 
   402)    !--- copy to output optional arguments ---
   403)    if (present(n1)) then
   404)      n1 = ns(1)
   405)    endif
   406)    if (present(n2)) then
   407)      n2 = ns(2)
   408)    endif
   409)    if (present(n3)) then
   410)      n3 = ns(3)
   411)    endif
   412)    if (present(n4)) then
   413)      n4 = ns(4)
   414)    endif
   415)    if (present(n5)) then
   416)      n5 = ns(5)
   417)    endif
   418)    if (present(n6)) then
   419)      n6 = ns(6)
   420)    endif
   421) 
   422)    deallocate(dids)
   423)    deallocate(ns)
   424) 
   425)    if (present(rc)) rc = rCode
   426)

   488) ! !IROUTINE: shr_ncread_domain -- read in domain info from a file
   489) !
   490) ! !DESCRIPTION:
   491) !     Read in domain information from a file.  The subroutine is designed
   492) !     specially for certain criteria in CCSM.  longitude, latitude, mask,
   493) !     a area arrays will be read in from the cdf file for each variable
   494) !     name input.  Across the subroutine interface, all arrays are 2d,
   495) !     and each is real*8 except mask which is an integer array.  Within
   496) !     the netcdf file, other scenarios are possible.
   497) ! note:
   498) ! o always returns 2d lat/lon arrays even if data is 1d in netCDF file
   499) ! o works if lat & lon are dimensions or variables
   500) ! o assumes area and mask are variables without a time dimension
   501) ! o mask is an integer array, all others are real*8
   502) ! o mask is read as real*8 array then copied via nint
   503) ! o assumes arrays are already allocated by the caller
   504) !
   505) ! \newline
   506) ! General Usage:
   507) !    call shr_ncread_domain('myfile','xc',lon,'yc',lat,'mask',imask,'area',area)
   508) ! \newline
   509) ! !REVISION HISTORY:
   510) !     2005-Apr-21 - T. Craig - first version
   511) !
   512) ! !INTERFACE: ------------------------------------------------------------------
   513) 
   514) subroutine shr_ncread_domain(fn, lonName,  lon,  latName,  lat, &
   515)                              &  maskName, mask, areaName, area, &
   516)                              &  fracName, frac, rc)
   517) 
   518)    implicit none
   519) 
   520) ! !INPUT/OUTPUT PARAMETERS:
   521) 
   522)    character(*)        ,intent(in)          :: fn        ! nc file name
   523)    character(*)        ,intent(in)          :: lonName   ! name of longitude var
   524)    real(SHR_KIND_R8)   ,intent(out)         :: lon(:,:)  ! longitudes
   525)    character(*)        ,intent(in)          :: latName   ! name of latitude var
   526)    real(SHR_KIND_R8)   ,intent(out)         :: lat(:,:)  ! latitudes

   536)    !----- local -----
   537)    real(SHR_KIND_R8),allocatable :: A4d(:,:,:,:)  ! local 4d array

   540)    integer(SHR_KIND_IN)          :: nflds         ! number of flds to read
   541)    integer(SHR_KIND_IN)          :: n,i,j         ! counters
   542)    integer(SHR_KIND_IN)          :: ndim,nd1,nd2  ! dims and size of 2 dims for cdf field
   543)    integer(SHR_KIND_IN)          ::      pd1,pd2  ! size of 2 dims for P2d

   591)    readarea = .true.

   600)      elseif (n > 2) then
   601)         if (present(maskName) .and. readmask) then

   603)            !--- since mask in an integer, allocate P2d and copy back later ---

   605)            readmask = .false.
   606)         else if (present(areaName) .and. readarea) then
   607)            varName = trim(areaName)
   608)            allocate(P2d(size(area,1),size(area,2)))
   609)            readarea = .false.
   610)         else if (present(fracName) .and. readfrac) then
   611)            varName = trim(fracName)
   612)            allocate(P2d(size(frac,1),size(frac,2)))
   613)            readfrac = .false.
   614)         endif
   615)      end if
   616) 
   617)      if (.not.shr_ncread_varExists(fn,varName)) &
   618)        call shr_ncread_abort(subName//' ERROR var does not exist '//trim(varName))
   619) 
   620)      !--- get size of input array ---
   621)      pd1 = size(P2d,1)
   622)      pd2 = size(P2d,2)
   623) 
   624)      !--- get var dims and check ---
   625)      call shr_ncread_varDimNum(fn,varName,ndim)
   626)      if (n > 2 .and. ndim /= 2) then
   627)        write(s_logunit,F02) 'ERROR '//trim(varName)//' ndim = ',ndim
   628)        call shr_ncread_abort(subName//' ERROR ndim must be 2 for '//trim(varName))

   630)        write(s_logunit,F02) 'ERROR '//trim(varName)//' ndim = ',ndim
   631)        call shr_ncread_abort(subName//' ERROR ndim must be 1 or 2 for '//trim(varName))
   632)      endif
   633)      nd1 = 1
   634)      nd2 = 1
   635)      if (ndim > 0)  call shr_ncread_varDimSize(fn,varName,1,nd1)
   636)      if (ndim > 1)  call shr_ncread_varDimSize(fn,varName,2,nd2)
   637) 
   638)      !--- error check dimensions, special case for 1d lat  ---
   639)      if (n == 2 .and. ndim == 1) then
   640)        if ( nd1 /= pd2) then
   641)          write(s_logunit,F03) ' nd1 pd2 error ',nd1,pd2
   642)          call shr_ncread_abort(subName//' ERROR nd1 pd2 error')
   643)        endif
   644)      elseif (ndim > 0 .and. nd1 /= pd1) then
   645)        write(s_logunit,F03) ' nd1 pd1 error ',nd1,pd1
   646)        call shr_ncread_abort(subName//' ERROR nd1 pd1 error')
   647)      endif
   648)      if (ndim > 1 .and. nd2 /= pd2) then
   649)        write(s_logunit,F03) ' nd2 pd2 error ',nd2,pd2
   650)        call shr_ncread_abort(subName//' ERROR nd2 pd2 error')
   651)      endif
   652) 
   653)      !--- allocate A4d and read ---
   654)      allocate(A4d(nd1,nd2,1,1))
   655)      A4d = 0.0_SHR_KIND_R8
   656)      call shr_ncread_field4dG(fn,varName,rfld=A4d)
   657) 
   658)      !--- copy into P2d as appropriate ---
   659)      do j = 1,pd2
   660)      do i = 1,pd1
   661)        if (n == 2 .and. ndim == 1) then
   662)          P2d(i,j) = A4d(j,1,1,1)
   663)        elseif (ndim == 1) then
   664)          P2d(i,j) = A4d(i,1,1,1)
   665)        else
   666)          P2d(i,j) = A4d(i,j,1,1)
   667)        endif
   668)      enddo
   669)      enddo
   670) 
   671)      !--- copy into mask R8 to IN ---
   672)      if (n == 1) then
   673)        lon(:,:) = P2d(:,:)
   674)      elseif (n == 2) then
   675)        lat(:,:) = P2d(:,:)
   676)      elseif (n == 3) then
   677)        mask(:,:) = nint(P2d(:,:))
   678)      elseif (n == 4) then
   679)        area(:,:) = P2d(:,:)
   680)      elseif (n == 5) then
   681)        frac(:,:) = P2d(:,:)
   682)      endif
   683) 
   684)      !--- clean up ---
   685)      deallocate(A4d)
   686)      deallocate(P2d,stat=rCode)
   687) !     nullify(P2d)
   688)      
   689)    enddo
   690) 
   691)    if (debug > 1 .and. s_loglev > 0) then
   692)      write(s_logunit,F04) 'min/max lon  ',minval(lon),maxval(lon)
   693)      write(s_logunit,F04) 'min/max lat  ',minval(lat),maxval(lat)
   694)      write(s_logunit,F04) 'min/max mask ',minval(mask),maxval(mask)
   695)      write(s_logunit,F04) 'min/max area ',minval(area),maxval(area)
   696)      if (nflds >= 5 .and. s_loglev > 0) write(s_logunit,F04) 'min/max frac ',minval(frac),maxval(frac)
   697)    endif
   698) 
   699)    if (present(rc)) rc = rCode
   700) 
   701) end subroutine shr_ncread_domain
