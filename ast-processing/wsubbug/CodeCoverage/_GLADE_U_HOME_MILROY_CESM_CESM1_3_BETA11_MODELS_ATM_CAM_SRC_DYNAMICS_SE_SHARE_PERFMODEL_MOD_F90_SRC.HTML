    31) 	subroutine SetPtoPNetworkParams(offnode,onnode,network,found)
    32) 
    33) 	type (commtime_t),intent(inout)     :: offnode
    34) 	type (commtime_t),intent(inout)     :: onnode
    35)         character(len=80)                   :: network
    36)         logical(kind=log_kind),intent(inout)  :: found
    37)         real (kind=real_kind) :: contention
    38) 
    39) 	! Macine specific characteristics
    40) 	! bluesky Power4 with dual rail 8-way LPAR 
    41)         ! bandwidth: microseconds per byte
    42)         ! latency:   microseconds
    43) 
    44) 	found=.FALSE.
    45)         if(network(1:7) == "bluesky") then 
    46)            offnode%latency   = 17.9 
    47)            onnode%latency    = .9   
    48)            offnode%bandwidth = 0.002777
    49)            !offnode%bandwidth = 3.*0.002777
    50)            onnode%bandwidth  = 0.001203
    51)            found=.TRUE.
    52)         elseif (network(1:11) == "blackforest") then 
    53)            offnode%latency   = 30.0 
    54)            onnode%latency    = 2.   
    55)            offnode%bandwidth = 0.0125
    56)            onnode%bandwidth  = 0.00416
    57)            found=.TRUE.
    58)         elseif (network(1:8) == "protoBGL") then 
    59) 	   ! This is the from the description of the prototype BGL from Oct 2003 meeting
    60)            offnode%latency   = 6. 
    61)            onnode%latency    = .1   
    62)            !contention = 3.0
    63)            !contention = 1.0
    64)            contention =  9.0
    65)            offnode%bandwidth = contention/420.! Note the value of 420 is 3 times what I get on point-to-point
    66) 					      ! messaging... I use the three because they only get 3 receive 
    67) 					      ! streams going at once.  The other three is due to the fact 
    68) 					      ! that the corner processors only have 3 network connections
    69) 					      ! to simulate contention 8 neighbors/6 links * 2.25 = 3
    70)            onnode%bandwidth  =1.0/(3.*420.) 
    71)            found=.TRUE.
    72)         elseif (network(1:8) == "BGL-fast") then 
    73) 	   ! BGL with network as described in the Paper "???"
    74)            offnode%latency   = 1.5 
    75)            onnode%latency    = .1   
    76)            !contention = 2.25
    77)            contention = 4.5
    78)            offnode%bandwidth = contention/1050.    !JMD I added 2.25 because I
    79)                                             !JMD wanted simulate contention
    80)                                             !JMD 8 neighbors/6 links * 2.25 = 3
    81)            onnode%bandwidth  =1.0/(3.*1050.) 
    82)            found=.TRUE.
    83)         elseif (network(1:8) == "BGL-slow") then 
    84) 	   ! BGL with network running at 1/2 of described in the Paper "???"
    85)            offnode%latency   = 4.5 
    86)            onnode%latency    = .1   
    87)            !contention = 2.25
    88)            contention = 4.5
    89)            offnode%bandwidth = contention/525.    !JMD I added 2.25 because I
    90)                                             !JMD wanted simulate contention
    91)                                             !JMD 8 neighbors/6 links * 2.25 = 3
    92)            onnode%bandwidth  =1.0/(3.*525.) 
    93)            found=.TRUE.
    94)         else 
    95) 	   print *,"SetPtoPNetworkParams:  Undefined network name"
    96) 	   print *,"SetPtoPNetworkParams:  Please select from the following or create your own:"
    97)            print *,"SetPtoPNetworkParams:  bluesky     == Power4 with dual rail colony and 8-way LPARs"
    98)            print *,"SetPtoPNetworkParams:  blackforest == Power3 with TBMX network"
    99)            print *,"SetPtoPNetworkParams:  protoBGL    == Prototype BGL machine (November 2003 configuration)"
   100)            print *,"SetPtoPNetworkParams:  BGL-fast    == BGL network as described in Paper ??? "
   101)            print *,"SetPtoPNetworkParams:  BGL-slow    == 1/2 perf. of BGL network as described in Paper ???? "
   102)         endif
