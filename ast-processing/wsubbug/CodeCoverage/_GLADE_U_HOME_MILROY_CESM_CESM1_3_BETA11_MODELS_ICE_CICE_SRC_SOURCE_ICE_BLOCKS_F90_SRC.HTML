   208)  subroutine create_blocks(nx_global, ny_global, ew_boundary_type, &
   209)                                                 ns_boundary_type)
   210) 
   211) ! !DESCRIPTION:
   212) !  This subroutine decomposes the global domain into blocks and
   213) !  fills the data structures with all the necessary block information.
   214) !
   215) ! !REVISION HISTORY: 
   216) !  same as module
   217) !
   218) ! !USES:
   219) 
   220)    use ice_fileunits, only: nu_diag
   221)    use ice_communicate, only: my_task, master_task
   222) 
   223) ! !INPUT PARAMETERS:
   224) 
   225)    integer (int_kind), intent(in) :: &
   226)       nx_global, ny_global           ! global domain size in x,y
   227) 
   228)    character (*), intent(in) :: &
   229)       ew_boundary_type,  &! type of boundary in logical east-west dir
   230)       ns_boundary_type    ! type of boundary in logical north-south dir
   231) 
   232) !EOP
   233) !BOC
   234) !----------------------------------------------------------------------
   235) !
   236) !  local variables
   237) !
   238) !----------------------------------------------------------------------
   239) 
   240)    integer (int_kind) :: &
   241)       i, ip1, j, jp1, n    ,&! loop indices
   242)       iblock, jblock       ,&! block loop indices
   243)       is, ie, js, je         ! temp start, end indices
   244) 
   245)    logical (log_kind) :: dbug
   246) 
   247) !----------------------------------------------------------------------
   248) !
   249) !  compute number of blocks and cartesian decomposition
   250) !  if the requested block size does not divide the global domain
   251) !  size evenly, add additional block space to accomodate padding
   252) !
   253) !----------------------------------------------------------------------
   254) 
   255)    nblocks_x   = (nx_global-1)/block_size_x + 1
   256)    nblocks_y   = (ny_global-1)/block_size_y + 1
   257)    nblocks_tot = nblocks_x*nblocks_y
   258) 
   259) !----------------------------------------------------------------------
   260) !
   261) !  allocate block arrays
   262) !
   263) !----------------------------------------------------------------------
   264) 
   265)    allocate(all_blocks(nblocks_tot))
   266)    allocate(i_global(nx_block,nblocks_tot), &
   267)             j_global(ny_block,nblocks_tot))
   268)    allocate(all_blocks_ij(nblocks_x,nblocks_y))
   269) 
   270) !----------------------------------------------------------------------
   271) !
   272) !  fill block data structures for all blocks in domain
   273) !
   274) !----------------------------------------------------------------------
   275) 
   276)    n = 0
   277)    do jblock=1,nblocks_y
   278)       js = (jblock-1)*block_size_y + 1
   279)       if (js > ny_global) call abort_ice(&
   280)             'ice: create_blocks: Bad block decomp: ny_block too large?')
   281)       je = js + block_size_y - 1
   282)       if (je > ny_global) je = ny_global ! pad array
   283) 
   284)       do iblock=1,nblocks_x
   285)          n = n + 1  ! global block id
   286) 
   287)          is = (iblock-1)*block_size_x + 1
   288)          if (is > nx_global) call abort_ice(&
   289)             'ice: create_blocks: Bad block decomp: nx_block too large?')
   290)          ie = is + block_size_x - 1
   291)          if (ie > nx_global) ie = nx_global
   292) 
   293)          all_blocks(n)%block_id = n
   294)          all_blocks(n)%iblock   = iblock
   295)          all_blocks(n)%jblock   = jblock
   296)          all_blocks(n)%ilo      = nghost + 1
   297)          all_blocks(n)%jlo      = nghost + 1
   298)          all_blocks(n)%ihi      = nx_block - nghost ! default value
   299)          all_blocks(n)%jhi      = ny_block - nghost ! default value
   300) 
   301)          if (jblock == nblocks_y .and. &
   302)              (ns_boundary_type == 'tripole' .or. &
   303)              ns_boundary_type == 'tripoleT')) then
   304)              all_blocks(n)%tripole = .true.
   305)          else
   306)              all_blocks(n)%tripole = .false.
   307)          endif
   308)          all_blocks(n)%tripoleTFlag = (ns_boundary_type == 'tripoleT')
   309) 
   310)          all_blocks_ij(iblock,jblock) = n
   311) 
   312)          do j=1,ny_block
   313)             j_global(j,n) = js - nghost + j - 1
   314) 
   315)             !*** southern ghost cells
   316) 
   317)             if (j_global(j,n) < 1) then
   318)                select case (ns_boundary_type)
   319)                case ('cyclic')
   320)                   j_global(j,n) = j_global(j,n) + ny_global
   321)                case ('open')
   322)                   j_global(j,n) = nghost - j + 1
   323)                case ('closed')
   324)                   j_global(j,n) = 0
   325)                case ('tripole')
   326)                   j_global(j,n) = nghost - j + 1 ! open
   327)                case ('tripoleT')
   328)                   j_global(j,n) = -j_global(j,n) + 1 ! open
   329)                case default
   330)                   call abort_ice(&
   331)                            'ice: create_blocks: unknown n-s bndy type')
   332)                end select
   333)             endif
   334) 
   335)             !*** padding required
   336) 
   337)             if (j_global(j,n) > ny_global + nghost) then
   338)                j_global(j,n) = 0   ! padding
   339) 
   340)             !*** northern ghost cells
   341) 
   342)             else if (j_global(j,n) > ny_global) then
   343)                select case (ns_boundary_type)
   344)                case ('cyclic')
   345)                   j_global(j,n) = j_global(j,n) - ny_global
   346)                case ('open')
   347)                   j_global(j,n) = 2*ny_global - j_global(j,n) + 1
   348)                case ('closed')
   349)                   j_global(j,n) = 0
   350)                case ('tripole')
   351)                   j_global(j,n) = -j_global(j,n)
   352)                case ('tripoleT')
   353)                   j_global(j,n) = -j_global(j,n)
   354)                case default
   355)                   call abort_ice(&
   356)                            'ice: create_blocks: unknown n-s bndy type')
   357)                end select
   358) 
   359)             !*** set last physical point if padded domain
   360) 
   361)             else if (j_global(j,n) == ny_global .and. &
   362)                      j > all_blocks(n)%jlo      .and. &
   363)                      j < all_blocks(n)%jhi) then
   364)                all_blocks(n)%jhi = j   ! last physical point in padded domain
   365)             endif
   366)          end do
   367) 
   368)          all_blocks(n)%j_glob => j_global(:,n)
   369) 
   370)          do i=1,nx_block
   371)             i_global(i,n) = is - nghost + i - 1
   372) 
   373)             !*** western ghost cells
   374) 
   375)             if (i_global(i,n) < 1) then
   376)                select case (ew_boundary_type)
   377)                case ('cyclic')
   378)                   i_global(i,n) = i_global(i,n) + nx_global
   379)                case ('open')
   380)                   i_global(i,n) = nghost - i + 1
   381)                case ('closed')
   382)                   i_global(i,n) = 0
   383)                case default
   384)                   call abort_ice(&
   385)                            'ice: create_blocks: unknown e-w bndy type')
   386)                end select
   387)             endif
   388) 
   389)             !*** padded domain - fill padded region with zero
   390) 
   391)             if (i_global(i,n) > nx_global + nghost) then
   392)                i_global(i,n) = 0
   393) 
   394)             !*** eastern ghost cells
   395) 
   396)             else if (i_global(i,n) > nx_global) then
   397)                select case (ew_boundary_type)
   398)                case ('cyclic')
   399)                   i_global(i,n) = i_global(i,n) - nx_global
   400)                case ('open')
   401)                   i_global(i,n) = 2*nx_global - i_global(i,n) + 1
   402)                case ('closed')
   403)                   i_global(i,n) = 0
   404)                case default
   405)                   call abort_ice(&
   406)                            'ice: create_blocks: unknown e-w bndy type')
   407)                end select
   408) 
   409)             !*** last physical point in padded domain
   410) 
   411)             else if (i_global(i,n) == nx_global .and. &
   412)                      i > all_blocks(n)%ilo      .and. &
   413)                      i < all_blocks(n)%ihi) then
   414)                all_blocks(n)%ihi = i
   415)             endif
   416)          end do
   417) 
   418)          all_blocks(n)%i_glob => i_global(:,n)
   419) 
   420)       end do
   421)    end do
   422) 
   423) !   dbug = .true.
   424)    dbug = .false.
   425)    if (dbug) then
   426)       if (my_task == master_task) then
   427)          write(nu_diag,*) 'block i,j locations'
   428)          do n = 1, nblocks_tot
   429)             write(nu_diag,*) 'block id, iblock, jblock:', &
   430)             all_blocks(n)%block_id, &
   431)             all_blocks(n)%iblock,   & 
   432)             all_blocks(n)%jblock
   433)          enddo
   434)       endif
   435)    endif
   436) 
   437) !EOC
   438) !----------------------------------------------------------------------
   439) 
   440) end subroutine create_blocks

   444) ! !IROUTINE: ice_blocksGetNbrID
   445) ! !INTERFACE:
   446) 
   447)  function ice_blocksGetNbrID(blockID, direction, iBoundary, jBoundary) &
   448)                              result (nbrID)
   449) 
   450) ! !DESCRIPTION:
   451) !  This function returns the block id of a neighboring block in a
   452) !  requested direction.  Directions:
   453) !      ice\_blocksNorth             (i  ,j+1)
   454) !      ice\_blocksSouth             (i  ,j-1)
   455) !      ice\_blocksEast              (i+1,j  )
   456) !      ice\_blocksWest              (i-1,j  )
   457) !      ice\_blocksNorthEast         (i+1,j+1)
   458) !      ice\_blocksNorthWest         (i-1,j+1)
   459) !      ice\_blocksSouthEast         (i  ,j-1)
   460) !      ice\_blocksSouthWest         (i-1,j-1)
   461) !      ice\_blocksNorth2            (i  ,j+2)
   462) !      ice\_blocksSouth2            (i  ,j-2)
   463) !      ice\_blocksEast2             (i+2,j  )
   464) !      ice\_blocksWest2             (i-2,j  )
   465) !      ice\_blocksNorthEast2        (i+2,j+2)
   466) !      ice\_blocksNorthWest2        (i-2,j+2)
   467) !      ice\_blocksSouthEast2        (i+2,j-2)
   468) !      ice\_blocksSouthWest2        (i-2,j-2)
   469) !      ice\_blocksEastNorthEast     (i+2,j+1)
   470) !      ice\_blocksEastSouthEast     (i+2,j-1)
   471) !      ice\_blocksWestNorthWest     (i-2,j+1)
   472) !      ice\_blocksWestSouthWest     (i-2,j-1)
   473) !      ice\_blocksNorthNorthEast    (i+1,j-2)
   474) !      ice\_blocksSouthSouthEast    (i+1,j-2)
   475) !      ice\_blocksNorthNorthWest    (i-1,j+2)
   476) !      ice\_blocksSouthSouthWest    (i-1,j-2)
   477) !
   478) 
   479) ! !INPUT PARAMETERS:
   480) 
   481)    integer (int_kind), intent(in)  :: &
   482)       blockID,       &! id of block for which neighbor id requested
   483)       direction       ! direction for which to look for neighbor -
   484)                       !   must be one of the predefined module
   485)                       !   variables for block direction
   486) 
   487)    character (*), intent(in) :: &
   488)       iBoundary,     &! determines what to do at edges of domain
   489)       jBoundary       !  options are - open, closed, cyclic, tripole, tripoleT
   490) 
   491) ! !OUTPUT PARAMETERS:
   492) 
   493)    integer (int_kind) :: &
   494)       nbrID           ! block ID of neighbor in requested dir
   495) 
   496) !EOP
   497) !BOC
   498) !----------------------------------------------------------------------
   499) !
   500) !  local variables
   501) !
   502) !----------------------------------------------------------------------
   503)     
   504)    integer (int_kind) :: &
   505)       iBlock, jBlock,  &! i,j block location of current block
   506)       inbr,   jnbr      ! i,j block location of neighboring block
   507) 
   508) !----------------------------------------------------------------------
   509) !
   510) !  retrieve info for current block
   511) !
   512) !----------------------------------------------------------------------
   513) 
   514)    call get_block_parameter(blockID, iblock=iBlock, jblock=jBlock)
   515) 
   516) !----------------------------------------------------------------------
   517) !
   518) !  compute i,j block location of neighbor
   519) !
   520) !----------------------------------------------------------------------
   521) 
   522)    select case(direction)
   523) 
   524)    case (ice_blocksNorth)
   525) 
   526)       inbr = iBlock
   527)       jnbr = jBlock + 1
   528)       if (jnbr > nblocks_y) then
   529)          select case(jBoundary)
   530)          case ('open')
   531)             jnbr = 0 ! do not write into the neighbor's ghost cells
   532)          case ('closed')
   533)             jnbr = 0
   534)          case ('cyclic')
   535)             jnbr = 1
   536)          case ('tripole':'tripoleT')
   537)             !*** return negative j value to flag tripole
   538)             !*** i index of main northern neighbor across the
   539)             !*** tripole cut - may also need i+1,i-1 to get
   540)             !*** other points if there has been padding or
   541)             !*** if the block size does not divide the domain
   542)             !*** evenly
   543)             inbr =  nblocks_x - iBlock + 1 
   544)             jnbr = -jBlock
   545)          case default
   546)             call abort_ice( &
   547)                'ice_blocksGetNbrID: unknown north boundary')
   548)          end select
   549)       endif
   550) 
   551)    case (ice_blocksSouth)
   552) 
   553)       inbr = iBlock
   554)       jnbr = jBlock - 1
   555)       if (jnbr < 1) then
   556)          select case(jBoundary)
   557)          case ('open')
   558)             jnbr = 0 ! do not write into the neighbor's ghost cells
   559)          case ('closed')
   560)             jnbr = 0
   561)          case ('cyclic')
   562)             jnbr = nblocks_y
   563)          case ('tripole')
   564)             jnbr = 0 ! do not write into the neighbor's ghost cells
   565)          case ('tripoleT')
   566)             jnbr = 0 ! do not write into the neighbor's ghost cells
   567)          case default
   568)             call abort_ice( &
   569)                'ice_blocksGetNbrID: unknown south boundary')
   570)          end select
   571)       endif
   572) 
   573)    case (ice_blocksEast )
   574) 
   575)       inbr = iBlock + 1
   576)       jnbr = jBlock
   577)       if (inbr > nblocks_x) then
   578)          select case(iBoundary)
   579)          case ('open')
