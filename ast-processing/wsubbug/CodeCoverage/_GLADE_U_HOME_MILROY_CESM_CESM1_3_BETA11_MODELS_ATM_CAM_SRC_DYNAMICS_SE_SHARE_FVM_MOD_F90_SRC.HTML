   533) ! do the remapping
   534) subroutine cslam_remap(tracer0,tracer1,weights,recons,centroid, &
   535)                  weights_eul_index_all, weights_lgr_index_all, jall)
   536)   real (kind=real_kind), intent(in)           :: tracer0(1-nhc:nc+nhc,1-nhc:nc+nhc)
   537)   real (kind=real_kind), intent(inout)        :: tracer1(1:nc,1:nc)
   538)   real (kind=real_kind), intent(in)           :: weights(10*(nc+2*nhe)*(nc+2*nhe),6)
   539)   real (kind=real_kind), intent(in)           :: recons(5,1-nhe:nc+nhe,1-nhe:nc+nhe)
   540)   real (kind=real_kind), intent(in)           :: centroid(5,1-nhe:nc+nhe,1-nhe:nc+nhe)
   541)   integer (kind=int_kind), intent(in)        :: weights_eul_index_all(10*(nc+2*nhe)*(nc+2*nhe),2)
   542)   integer (kind=int_kind), intent(in)        :: weights_lgr_index_all(10*(nc+2*nhe)*(nc+2*nhe),2)
   543)   integer (kind=int_kind), intent(in)        :: jall  
   544)   
   545)   integer                                     :: h, jx, jy, jdx, jdy
   546)        
   547)     do h=1,jall
   548)       jx  = weights_lgr_index_all(h,1)
   549)       jy  = weights_lgr_index_all(h,2)
   550)       jdx = weights_eul_index_all(h,1)
   551)       jdy = weights_eul_index_all(h,2)
   552) 
   553)       tracer1(jx,jy) = tracer1(jx,jy)+weights(h,1)*(&
   554)          ! all constant terms 
   555)          tracer0(jdx,jdy) - recons(1,jdx,jdy)*centroid(1,jdx,jdy) &
   556)          - recons(2,jdx,jdy)*centroid(2,jdx,jdy) &
   557)          + recons(3,jdx,jdy)*(2.0D0*centroid(1,jdx,jdy)**2 -centroid(3,jdx,jdy)) &
   558)          + recons(4,jdx,jdy)*(2.0D0*centroid(2,jdx,jdy)**2 -centroid(4,jdx,jdy)) &
   559)          + recons(5,jdx,jdy)*(2.0D0*centroid(1,jdx,jdy)*centroid(2,jdx,jdy)-centroid(5,jdx,jdy))) + &
   560)          ! linear terms
   561)          weights(h,2)*&
   562)          (recons(1,jdx,jdy) - recons(3,jdx,jdy)*2.0D0*centroid(1,jdx,jdy) &
   563)           - recons(5,jdx,jdy)*centroid(2,jdx,jdy)) + &
   564)          weights(h,3)*&
   565)          (recons(2,jdx,jdy) - recons(4,jdx,jdy)*2.0D0*centroid(2,jdx,jdy) &
   566)          - recons(5,jdx,jdy)*centroid(1,jdx,jdy)) + &
   567)          ! quadratic terms

   569)          weights(h,5)*recons(4,jdx,jdy)+&
   570)          weights(h,6)*recons(5,jdx,jdy)

   897) !SUBROUTINE FVM_DEP_FROM_GLL----------------------------------------------CE-for FVM!
   898) ! AUTHOR: CHRISTOPH ERATH, MARK TAYLOR 14. December 2011                            !
   899) ! DESCRIPTION: calculates the deparute grid for fvm coming from the gll points      !
   900) !                                                                                   !
   901) ! CALLS: 
   902) ! INPUT: 
   903) !        
   904) ! OUTPUT: 
   905) !-----------------------------------------------------------------------------------!
   906) subroutine fvm_dep_from_gll(elem, deriv, asphere,dsphere,dt,tl,klev)
   907)   use physical_constants, only : DD_PI, rearth
   908)   use coordinate_systems_mod, only : spherical_polar_t, cartesian3D_t, change_coordinates
   909)   use time_mod, only : timelevel_t
   910)   use element_mod, only : element_t
   911)   use derivative_mod, only : derivative_t, interpolate_gll2fvm_corners
   912) 
   913)   implicit none
   914)   type (element_t), intent(in)          :: elem
   915)   type (derivative_t)  , intent(in) :: deriv
   916)   type (spherical_polar_t),intent(in)   :: asphere(nc+1,nc+1)
   917)   type (spherical_polar_t),intent(out)  :: dsphere(-1:nc+3,-1:nc+3,nlev)
   918)   type (timelevel_t),intent(in)        :: tl
   919)   real (kind=real_kind),intent(in)      :: dt
   920)   integer,intent(in)                   :: klev
   921)   
   922)   real (kind=real_kind)                 :: uxyz_gll(np,np,3),uxyz(nc+1,nc+1,3)
   923)   real (kind=real_kind)                 :: un,ue,ur,clon,clat,slon,slat
   924)   type(cartesian3D_t)                   :: acart
   925)   integer :: i,j,k
   926)   
   927)   real (kind=real_kind)  :: vstar(np,np,2)
   928)   
   929)    ! convert velocity from lat/lon to cartesian 3D
   930)    vstar = elem%derived%vstar(:,:,:,klev)
   931)     
   932)    do i=1,np
   933)      do j=1,np
   934)         clon = cos(elem%spherep(i,j)%lon)
   935)         slon = sin(elem%spherep(i,j)%lon)
   936)         clat = cos(elem%spherep(i,j)%lat)
   937)         slat = sin(elem%spherep(i,j)%lat)
   938) 
   939)         ur = 0
   940)         ue = vstar(i,j,1) 
   941)         un = vstar(i,j,2)
   942) 
   943)         uxyz_gll(i,j,1)= clon*clat*ur - clon*slat*un - slon*ue
   944)         uxyz_gll(i,j,2)= slon*clon*ur - slon*slat*un + clon*ue
   945)         uxyz_gll(i,j,3)= slat          *ur + clat          *un
   946)     
   947)      enddo
   948)    enddo
   949)    ! interpolate velocity to fvm nodes
   950)    do i=1,3
   951)       uxyz(:,:,i)=interpolate_gll2fvm_corners(uxyz_gll(:,:,i),deriv)
   952)    end do 
   953)    ! compute departure point 
   954)    ! crude, 1st order accurate approximation.  to be improved
   955)    do i=1,nc+1
   956)       do j=1,nc+1
   957)          ! note: asphere()%r=1, so we need to convert velocity to radians/sec:
   958)          acart = change_coordinates(asphere(i,j))  
   959)          acart%x = acart%x - dt*uxyz(i,j,1)/rearth
   960)          acart%y = acart%y - dt*uxyz(i,j,2)/rearth
   961)          acart%z = acart%z - dt*uxyz(i,j,3)/rearth
   962)          dsphere(i,j,klev) = change_coordinates(acart)
   963)          dsphere(i,j,klev)%r = asphere(i,j)%r
   964)       enddo
   965)    enddo
   966)   
   967) end subroutine fvm_dep_from_gll

   980)   use coordinate_systems_mod, only: cartesian2D_t, cart2cubedspherexy, spherical_to_cart
   981)   use fvm_control_volume_mod, only:  fvm_struct                                         
   982)   implicit none

   989)   integer                              :: i,j
   990)   logical                              :: crossline
   991)             
   992)   ! calculate xy Cartesian on the cube of departure points on the corresponding face  
   993)   ! the calculation of dcart is also done in fvm_line_integrals_mod.F90, one could 
   994)   ! save the points here...
   995)   ! BUT: this subroutine should only be used for test reasons
   996)   do j=1,nc+1
   997)      do i=1,nc+1               
   998)         call cart2cubedspherexy(spherical_to_cart(fvm%dsphere(i,j,klev)),&
   999)              fvm%faceno,dcart(i,j))

  1018)       call check_lines_cross(dcart(i,j),dcart(i,j+1),dcart(i+1,j),dcart(i+1,j+1),crossline)
  1019)      end do
  1020)   end do

  1023)     cflx=abs((atan(dcart(i,nc+1)%x)-atan(fvm%acartx(i)))/fvm%dalpha)
  1024)     cfly=abs((atan(dcart(nc+1,j)%y)-atan(fvm%acarty(j)))/fvm%dbeta)

  1033)     write(*,*) "max CFL in this element: maxcflx", fvm%maxcfl(1,klev), "maxcfly", fvm%maxcfl(2,klev) 
  1034)     STOP "Exit program!"
  1035)   endif

  1039)     write(*,*) "Choose a smaller time step! max CFL in this element: maxcflx", fvm%maxcfl(1,klev), "maxcfly", fvm%maxcfl(2,klev)

  1041)   endif
