   223)     !---------------------------------------------------------------

   225)     ! Create input land state directly from atm, runoff and glc outputs 
   226)     !
   227)     ! Arguments

   236)     character(CL),allocatable :: mrgstr(:)   ! temporary string
   237)     character(CL) :: field   ! string converted to char
   238)     type(mct_aVect_sharedindices),save :: a2x_sharedindices
   239)     type(mct_aVect_sharedindices),save :: r2x_sharedindices
   240)     type(mct_aVect_sharedindices),save :: g2x_sharedindices
   241)     character(*), parameter   :: subname = '(prep_lnd_merge) '
   242) 
   243)     !----------------------------------------------------------------------- 
   244) 
   245)     call seq_comm_getdata(CPLID, iamroot=iamroot)
   246) 
   247)     if (first_time) then
   248)        nflds = mct_aVect_nRattr(x2l_l)
   249) 
   250)        allocate(mrgstr(nflds))
   251)        do i = 1,nflds
   252)           field = mct_aVect_getRList2c(i, x2l_l)
   253)           mrgstr(i) = subname//'x2l%'//trim(field)//' ='
   254)        enddo
   255) 
   256)        call mct_aVect_setSharedIndices(a2x_l, x2l_l, a2x_SharedIndices)
   257)        call mct_aVect_setSharedIndices(r2x_l, x2l_l, r2x_SharedIndices)
   258)        call mct_aVect_setSharedIndices(g2x_l, x2l_l, g2x_SharedIndices)
   259) 
   260)        !--- document copy operations ---
   261)        do i=1,a2x_SharedIndices%shared_real%num_indices
   262)           i1=a2x_SharedIndices%shared_real%aVindices1(i)
   263)           o1=a2x_SharedIndices%shared_real%aVindices2(i)
   264)           field = mct_aVect_getRList2c(i1, a2x_l)
   265)           mrgstr(o1) = trim(mrgstr(o1))//' = a2x%'//trim(field)
   266)        enddo
   267)        do i=1,r2x_SharedIndices%shared_real%num_indices
   268)           i1=r2x_SharedIndices%shared_real%aVindices1(i)
   269)           o1=r2x_SharedIndices%shared_real%aVindices2(i)
   270)           field = mct_aVect_getRList2c(i1, r2x_l)
   271)           mrgstr(o1) = trim(mrgstr(o1))//' = r2x%'//trim(field)
   272)        enddo
   273)        do i=1,g2x_SharedIndices%shared_real%num_indices
   274)           i1=g2x_SharedIndices%shared_real%aVindices1(i)
   275)           o1=g2x_SharedIndices%shared_real%aVindices2(i)
   276)           field = mct_aVect_getRList2c(i1, g2x_l)
   277)           mrgstr(o1) = trim(mrgstr(o1))//' = g2x%'//trim(field)
   278)        enddo
   279)     endif
   280) 
   281)     call mct_aVect_copy(aVin=a2x_l, aVout=x2l_l, vector=mct_usevector, sharedIndices=a2x_SharedIndices)
   282)     call mct_aVect_copy(aVin=r2x_l, aVout=x2l_l, vector=mct_usevector, sharedIndices=r2x_SharedIndices)
   283)     call mct_aVect_copy(aVin=g2x_l, aVout=x2l_l, vector=mct_usevector, sharedIndices=g2x_SharedIndices)
   284) 
   285)     if (first_time) then
   286)        if (iamroot) then
   287)           write(logunit,'(A)') subname//' Summary:'
   288)           do i = 1,nflds
   289)              write(logunit,'(A)') trim(mrgstr(i))
   290)           enddo
   291)        endif
   292)        deallocate(mrgstr)
   293)     endif
   294) 
   295)     first_time = .false.
   296)
