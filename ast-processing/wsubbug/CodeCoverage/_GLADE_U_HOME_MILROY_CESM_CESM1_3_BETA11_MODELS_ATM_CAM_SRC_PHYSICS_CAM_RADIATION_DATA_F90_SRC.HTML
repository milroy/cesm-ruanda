   128)   subroutine rad_data_readnl(nlfile)
   129) 
   130)     ! Read rad_data_nl namelist group.  Parse input.
   131) 
   132)     use namelist_utils,  only: find_group_name
   133)     use units,           only: getunit, freeunit
   134)     use mpishorthand
   135) 
   136)     character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
   137)     logical :: rad_data_fdh = .false.
   138) 
   139)     ! Local variables
   140)     integer :: unitn, ierr, i
   141)     character(len=*), parameter :: subname = 'rad_data_readnl'
   142) 
   143)     namelist /rad_data_nl/ rad_data_output, rad_data_histfile_num, rad_data_avgflag, rad_data_fdh
   144) 
   145)     !-----------------------------------------------------------------------------
   146) 
   147)     if (masterproc) then
   148)        unitn = getunit()
   149)        open( unitn, file=trim(nlfile), status='old' )
   150)        call find_group_name(unitn, 'rad_data_nl', status=ierr)
   151)        if (ierr == 0) then
   152)           read(unitn, rad_data_nl, iostat=ierr)
   153)           if (ierr /= 0) then
   154)              call endrun(subname // ':: ERROR reading namelist')
   155)           end if
   156)        end if
   157)        close(unitn)
   158)        call freeunit(unitn)
   159)     end if
   160) 
   161) #ifdef SPMD
   162)     ! Broadcast namelist variables
   163)     call mpibcast (rad_data_output,       1,   mpilog ,  0, mpicom)
   164)     call mpibcast (rad_data_fdh,          1,   mpilog ,  0, mpicom)
   165)     call mpibcast (rad_data_histfile_num, 1,   mpiint ,  0, mpicom)
   166)     call mpibcast (rad_data_avgflag,      1,   mpichar , 0, mpicom)
   167) #endif
   168)     do_fdh = rad_data_fdh
   169) 
   170)   end subroutine rad_data_readnl
