    88)   subroutine macrop_driver_readnl(nlfile)
    89) 
    90)     use namelist_utils,  only: find_group_name
    91)     use units,           only: getunit, freeunit
    92)     use mpishorthand
    93) 
    94)     character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
    95) 
    96)    ! Namelist variables
    97)    logical  :: macro_park_do_cldice  = .true.   !   do_cldice = .true., park macrophysics is prognosing cldice
    98)    logical  :: macro_park_do_cldliq  = .true.   !   do_cldliq = .true., park macrophysics is prognosing cldliq
    99)    logical  :: macro_park_do_detrain = .true.   !   do_detrain = .true., park macrophysics is detraining ice into stratiform
   100) 
   101)    ! Local variables
   102)    integer :: unitn, ierr
   103)    character(len=*), parameter :: subname = 'macrop_driver_readnl'
   104) 
   105)    namelist /macro_park_nl/ macro_park_do_cldice, macro_park_do_cldliq, macro_park_do_detrain
   106) 
   107)    !-----------------------------------------------------------------------------
   108) 
   109)    if (masterproc) then
   110)       unitn = getunit()
   111)       open( unitn, file=trim(nlfile), status='old' )
   112)       call find_group_name(unitn, 'macro_park_nl', status=ierr)
   113)       if (ierr == 0) then
   114)          read(unitn, macro_park_nl, iostat=ierr)
   115)          if (ierr /= 0) then
   116)             call endrun(subname // ':: ERROR reading namelist')
   117)          end if
   118)       end if
   119)       close(unitn)
   120)       call freeunit(unitn)
   121) 
   122)       ! set local variables
   123) 
   124)       do_cldice  = macro_park_do_cldice
   125)       do_cldliq  = macro_park_do_cldliq
   126)       do_detrain = macro_park_do_detrain
   127) 
   128)    end if
   129) 
   130) 
   131) 
   132) #ifdef SPMD
   133)    ! Broadcast namelist variables
   134)    call mpibcast(do_cldice,         1, mpilog, 0, mpicom)
   135)    call mpibcast(do_cldliq,         1, mpilog, 0, mpicom)
   136)    call mpibcast(do_detrain,        1, mpilog, 0, mpicom)
   137) #endif
   138) 
   139) end subroutine macrop_driver_readnl

   635)            (4._r8*3.14_r8*50.e-6_r8**3*500._r8)     ! Shallow Convection
