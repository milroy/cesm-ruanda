   125)      call shr_mpi_abort(string,rcode)

   880) SUBROUTINE shr_mpi_gathScatvInitr1(comm, rootid, locArr, glob1DArr, globSize, &
   881)                                    displs, string )
   882) 
   883)    IMPLICIT none
   884) 
   885)    !----- arguments -----
   886)    integer(SHR_KIND_IN), intent(in)   :: comm          ! mpi communicator
   887)    integer(SHR_KIND_IN), intent(in)   :: rootid        ! MPI task to gather/scatter on
   888)    real(SHR_KIND_R8),    intent(in)   :: locArr(:)     ! Local array of distributed data
   889)    real(SHR_KIND_R8),    pointer      :: glob1DArr(:)  ! Global 1D array of gathered data
   890)    integer(SHR_KIND_IN), pointer      :: globSize(:)   ! Size of each distributed piece
   891)    integer(SHR_KIND_IN), pointer      :: displs(:)     ! Displacements for receive
   892)    character(*),optional,intent(in)   :: string        ! message
   893) 
   894)    !----- local -----
   895)    integer(SHR_KIND_IN)               :: npes          ! Number of MPI tasks
   896)    integer(SHR_KIND_IN)               :: locSize       ! Size of local distributed data
   897)    integer(SHR_KIND_IN), pointer      :: sendSize(:)   ! Size to send for initial gather
   898)    integer(SHR_KIND_IN)               :: i             ! Index
   899)    integer(SHR_KIND_IN)               :: rank          ! Rank of this MPI task
   900)    integer(SHR_KIND_IN)               :: nSize         ! Maximum size to send
   901)    integer(SHR_KIND_IN)               :: ierr          ! Error code
   902)    integer(SHR_KIND_IN)               :: nSiz1D        ! Size of 1D global array
   903)    integer(SHR_KIND_IN)               :: maxSize       ! Maximum size
   904) 
   905)    !----- formats -----
   906)    character(*),parameter             :: subName = '(shr_mpi_gathScatvInitr1) '
   907) 
   908) !-------------------------------------------------------------------------------
   909) ! PURPOSE: Setup arrays for a gatherv/scatterv operation
   910) !-------------------------------------------------------------------------------
   911) 
   912)    locSize = size(locarr)
   913)    call shr_mpi_commsize( comm, npes )
   914)    call shr_mpi_commrank( comm, rank )
   915)    allocate( globSize(npes) )
   916)    !
   917)    ! --- Gather the send global sizes from each MPI task -----------------------
   918)    !
   919)    allocate( sendSize(npes) )
   920)    sendSize(:) = 1
   921)    globSize(:) = 1
   922)    call MPI_GATHER( locSize, 1, MPI_INTEGER, globSize, sendSize, &
   923)                     MPI_INTEGER, rootid, comm, ierr )
   924)    if (present(string)) then
   925)      call shr_mpi_chkerr(ierr,subName//trim(string))
   926)    else
   927)      call shr_mpi_chkerr(ierr,subName)
   928)    endif
   929)    deallocate( sendSize )
   930)    !
   931)    ! --- Prepare the displacement and allocate arrays -------------------------
   932)    !
   933)    allocate( displs(npes) )
   934)    displs(1) = 0
   935)    if ( rootid /= rank )then
   936)       maxSize = 1
   937)       globSize = 1
   938)    else
   939)       maxSize = maxval(globSize)
   940)    end if
   941)    nsiz1D  = min(maxSize,globSize(1))
   942)    do i = 2, npes
   943)       nSize = min(maxSize,globSize(i-1))
   944)       displs(i) = displs(i-1) + nSize
   945)       nsiz1D = nsiz1D + min(maxSize,globSize(i))
   946)    end do
   947)    allocate( glob1DArr(nsiz1D) )
   948)    !----- Do some error checking for the root task arrays computed ----
   949)    if ( rootid == rank )then
   950)       if ( nsiz1D /= sum(globSize) ) &
   951)          call shr_mpi_abort( subName//" : Error, size of global array not right" )
   952)       if ( any(displs < 0) .or. any(displs >= nsiz1D) ) &
   953)          call shr_mpi_abort( subName//" : Error, displacement array not right" )
   954)       if ( (displs(npes)+globSize(npes)) /= nsiz1D ) &
   955)          call shr_mpi_abort( subName//" : Error, displacement array values too big" )
   956)    end if
   957) 
   958) END SUBROUTINE shr_mpi_gathScatvInitr1
