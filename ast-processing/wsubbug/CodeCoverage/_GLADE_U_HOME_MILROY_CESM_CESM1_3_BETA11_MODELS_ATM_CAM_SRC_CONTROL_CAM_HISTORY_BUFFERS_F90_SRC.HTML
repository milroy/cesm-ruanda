   501)   subroutine check_accum (field, idim, ieu, jeu, fillvalue)
   502)     !
   503)     integer, intent(in)  :: idim
   504)     real(r8), intent(in) :: field(idim,*)   ! real*8 array
   505)     integer, intent(in)  :: ieu,jeu         ! loop ranges
   506) 
   507)     logical :: bad
   508)     integer :: i,k
   509)     real(r8), intent(in) :: fillvalue
   510)     !
   511)     ! For multilevel fields ensure that all levels have fillvalue applied consistently
   512)     !
   513)     bad = .false.
   514)     do k=2,jeu
   515)        do i=1,ieu
   516)           if (field(i,1) == fillvalue .and. field(i,k) /= fillvalue .or. &
   517)                field(i,1) /= fillvalue .and. field(i,k) == fillvalue) then
   518)              bad = .true.
   519)           end if
   520)        end do
   521)     end do
   522) 
   523)     if (bad) then
   524)        call endrun ('CHECK_ACCUM: inconsistent level values')
   525)     end if
   526) 
   527)     return
   528)   end subroutine check_accum
