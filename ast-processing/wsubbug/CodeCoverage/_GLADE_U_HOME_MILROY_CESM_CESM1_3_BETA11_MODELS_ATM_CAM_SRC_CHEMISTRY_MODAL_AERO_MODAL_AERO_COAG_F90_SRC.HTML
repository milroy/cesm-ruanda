   996)       subroutine getcoags_wrapper_f(              &
   997)           airtemp, airprs,                        &
   998)           dgatk, dgacc,                           &
   999)           sgatk, sgacc,                           &
  1000)           xxlsgat, xxlsgac,                       &
  1001)           pdensat, pdensac,                       &

  1003)           betaii0, betaii2, betajj0, betajj2      )
  1004)         use physconst, only: p0 => pstd, &
  1005)                              tmelt, &
  1006)                              boltz
  1007) !
  1008) ! interface to subr. getcoags
  1009) !
  1010) ! interface code adapted from subr. aeroproc of cmaq v4.6,
  1011) !     with some of the parameter values from module aero_info_ae4
  1012) !
  1013)       implicit none
  1014) 
  1015) ! *** arguments
  1016) 
  1017)       real(r8), intent(in) :: airtemp  ! air temperature [ k ]
  1018)       real(r8), intent(in) :: airprs   ! air pressure in [ pa ]
  1019) 
  1020)       real(r8), intent(in) :: dgatk    ! aitken mode geometric mean diameter [m]
  1021)       real(r8), intent(in) :: dgacc    ! accumulation mode geometric mean diam [m]
  1022) 
  1023)       real(r8), intent(in) :: sgatk    ! aitken mode geometric standard deviation
  1024)       real(r8), intent(in) :: sgacc    ! accumulation mode geometric standard deviation

  1027)       real(r8), intent(in) :: xxlsgac  !  deviations

  1033)                                betaii0, betaii2,  betajj0,  betajj2

  1036) ! *** local parameters
  1037)       real(r8) :: t0  ! standard surface temperature (15 deg C) [ k ]
  1038)       real(r8), parameter :: two3 = 2.0_r8/3.0_r8
  1039) 
  1040) ! *** local variables
  1041)       real(r8) amu            ! atmospheric dynamic viscosity [ kg/m s ]
  1042)       real(r8) sqrt_temp      ! square root of ambient temperature
  1043)       real(r8) lamda          ! mean free path [ m ]
  1044) 
  1045) ! *** intramodal coagulation rates [ m**3/s ] ( 0th & 2nd moments )
  1046)       real(r8)    batat( 2 )  ! aitken mode
  1047)       real(r8)    bacac( 2 )  ! accumulation mode
  1048) ! *** intermodal coagulation rates [ m**3/s ] ( 0th & 2nd moments )
  1049)       real(r8)    batac( 2 )  ! aitken to accumulation
  1050)       real(r8)    bacat( 2 )  ! accumulation from aitken

  1053) ! *** 3rd moment intermodal transfer rate by coagulation
  1054)       real(r8)    c30atac     ! aitken to accumulation
  1055) 
  1056) ! *** near continnuum regime (independent of mode)
  1057)       real(r8)    knc         ! knc = two3 * boltz *  airtemp / amu
  1058) ! *** free molecular regime (depends upon modal density)
  1059)       real(r8)    kfmat       ! kfmat = sqrt(3.0*boltz*airtemp/pdensat)
  1060)       real(r8)    kfmac       ! kfmac = sqrt(3.0*boltz*airtemp/pdensac)
  1061)       real(r8)    kfmatac     ! kfmatac = sqrt( 6.0 * boltz * airtemp /
  1062)                               !                ( pdensat + pdensac ) )
  1063) 
  1064)       real(r8)    dumacc2, dumatk2, dumatk3
  1065) 
  1066)       t0 = tmelt + 15._r8
  1067) 
  1068)       sqrt_temp = sqrt( airtemp)
  1069) 
  1070) ! *** calculate mean free path [ m ]:
  1071) !     6.6328e-8 is the sea level value given in table i.2.8
  1072) !     on page 10 of u.s. standard atmosphere 1962
  1073)       lamda = 6.6328e-8_r8 * p0 * airtemp  / ( t0 * airprs )
  1074) 
  1075) ! *** calculate dynamic viscosity [ kg m**-1 s**-1 ]:
  1076) !     u.s. standard atmosphere 1962 page 14 expression
  1077) !     for dynamic viscosity is:
  1078) !     dynamic viscosity =  beta * t * sqrt(t) / ( t + s)
  1079) !     where beta = 1.458e-6 [ kg sec^-1 k**-0.5 ], s = 110.4 [ k ].
  1080)       amu = 1.458e-6_r8 * airtemp * sqrt_temp / ( airtemp + 110.4_r8 )
  1081) 
  1082) ! *** coagulation
  1083) !     calculate coagulation coefficients using a method dictated by
  1084) !     the value of fastcoag_flag.  if true, the computationally-
  1085) !     efficient getcoags routine is used.  if false, the more intensive
  1086) !     gauss-hermite numerical quadrature method is used.  see section
  1087) !     2.1 of bhave et al. (2004) for further discussion.
  1088) 
  1089) ! *** calculate term used in equation a6 of binkowski & shankar (1995)
  1090)       knc      = two3 * boltz *  airtemp / amu
  1091) ! *** calculate terms used in equation a5 of binkowski & shankar (1995)
  1092)       kfmat    = sqrt( 3.0_r8 * boltz * airtemp / pdensat )
  1093)       kfmac    = sqrt( 3.0_r8 * boltz * airtemp / pdensac )
  1094)       kfmatac  = sqrt( 6.0_r8 * boltz * airtemp / ( pdensat + pdensac ) )
  1095) 
  1096) ! *** transfer of number to accumulation mode from aitken mode is zero
  1097)       bacat(1) = 0.0_r8
  1098) 
  1099) ! *** calculate intermodal and intramodal coagulation coefficients
  1100) !     for zeroth and second moments, and intermodal coagulation
  1101) !     coefficient for third moment
  1102)         call getcoags( lamda, kfmatac, kfmat, kfmac, knc,   &
  1103)                        dgatk,   dgacc,   sgatk,   sgacc,     &
  1104)                        xxlsgat,  xxlsgac,     &
  1105)                        batat(2), batat(1), bacac(2), bacac(1),   &
  1106)                        batac(2), bacat(2), batac(1), c3ij )
  1107) 
  1108) ! convert from the "cmaq" coag rate parameters 
  1109) ! to the "mirage2" parameters
  1110)         dumacc2 = ( (dgacc**2) * exp( 2.0_r8*xxlsgac*xxlsgac ) )
  1111)         dumatk2 = ( (dgatk**2) * exp( 2.0_r8*xxlsgat*xxlsgat ) )
  1112)         dumatk3 = ( (dgatk**3) * exp( 4.5_r8*xxlsgat*xxlsgat ) )
  1113) 
  1114)         betaii0  = max( 0.0_r8, batat(1) )
  1115)         betajj0  = max( 0.0_r8, bacac(1) )
  1116)         betaij0  = max( 0.0_r8, batac(1) )
  1117)         betaij3  = max( 0.0_r8, c3ij / dumatk3 )
  1118) 
  1119)         betajj2  = max( 0.0_r8, bacac(2) / dumacc2 )
  1120)         betaii2  = max( 0.0_r8, batat(2) / dumatk2 )
  1121)         betaij2i = max( 0.0_r8, batac(2) / dumatk2 )
  1122)         betaij2j = max( 0.0_r8, bacat(2) / dumatk2 )
  1123) 
  1124) 
  1125)       return
