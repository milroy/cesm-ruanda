    31)       subroutine reftra_sw(nlayers, lrtchk, pgg, prmuz, ptau, pw, &
    32)                            pref, prefd, ptra, ptrad)
    33) ! --------------------------------------------------------------------
    34)   
    35) ! Purpose: computes the reflectivity and transmissivity of a clear or 
    36) !   cloudy layer using a choice of various approximations.
    37) !
    38) ! Interface:  *rrtmg_sw_reftra* is called by *rrtmg_sw_spcvrt*
    39) !
    40) ! Description:
    41) ! explicit arguments :
    42) ! --------------------
    43) ! inputs
    44) ! ------ 
    45) !      lrtchk  = .t. for all layers in clear profile
    46) !      lrtchk  = .t. for cloudy layers in cloud profile 
    47) !              = .f. for clear layers in cloud profile
    48) !      pgg     = assymetry factor
    49) !      prmuz   = cosine solar zenith angle
    50) !      ptau    = optical thickness
    51) !      pw      = single scattering albedo
    52) !
    53) ! outputs
    54) ! -------
    55) !      pref    : collimated beam reflectivity
    56) !      prefd   : diffuse beam reflectivity 
    57) !      ptra    : collimated beam transmissivity
    58) !      ptrad   : diffuse beam transmissivity
    59) !
    60) !
    61) ! Method:
    62) ! -------
    63) !      standard delta-eddington, p.i.f.m., or d.o.m. layer calculations.
    64) !      kmodts  = 1 eddington (joseph et al., 1976)
    65) !              = 2 pifm (zdunkowski et al., 1980)
    66) !              = 3 discrete ordinates (liou, 1973)
    67) !
    68) !
    69) ! Modifications:
    70) ! --------------
    71) ! Original: J-JMorcrette, ECMWF, Feb 2003
    72) ! Revised for F90 reformatting: MJIacono, AER, Jul 2006
    73) ! Revised to add exponential lookup table: MJIacono, AER, Aug 2007
    74) !
    75) ! ------------------------------------------------------------------
    76) 
    77) ! ------- Declarations ------
    78) 
    79) ! ------- Input -------
    80) 
    81)       integer, intent(in) :: nlayers
    82) 
    83)       logical, intent(in) :: lrtchk(:)                           ! Logical flag for reflectivity and
    84)                                                                  ! and transmissivity calculation; 
    85)                                                                  !   Dimensions: (nlayers)
    86) 
    87)       real(kind=r8), intent(in) :: pgg(:)                      ! asymmetry parameter
    88)                                                                  !   Dimensions: (nlayers)
    89)       real(kind=r8), intent(in) :: ptau(:)                     ! optical depth
    90)                                                                  !   Dimensions: (nlayers)
    91)       real(kind=r8), intent(in) :: pw(:)                       ! single scattering albedo 
    92)                                                                  !   Dimensions: (nlayers)
    93)       real(kind=r8), intent(in) :: prmuz                       ! cosine of solar zenith angle
    94) 
    95) ! ------- Output -------
    96) 
    97)       real(kind=r8), intent(inout) :: pref(:)                    ! direct beam reflectivity
    98)                                                                  !   Dimensions: (nlayers+1)
    99)       real(kind=r8), intent(inout) :: prefd(:)                   ! diffuse beam reflectivity
   100)                                                                  !   Dimensions: (nlayers+1)
   101)       real(kind=r8), intent(inout) :: ptra(:)                    ! direct beam transmissivity
   102)                                                                  !   Dimensions: (nlayers+1)
   103)       real(kind=r8), intent(inout) :: ptrad(:)                   ! diffuse beam transmissivity
   104)                                                                  !   Dimensions: (nlayers+1)
   105) 
   106) ! ------- Local -------
   107) 
   108)       integer :: jk, jl, kmodts
   109)       integer :: itind
   110) 
   111)       real(kind=r8) :: tblind
   112)       real(kind=r8) :: za, za1, za2
   113)       real(kind=r8) :: zbeta, zdend, zdenr, zdent
   114)       real(kind=r8) :: ze1, ze2, zem1, zem2, zemm, zep1, zep2
   115)       real(kind=r8) :: zg, zg3, zgamma1, zgamma2, zgamma3, zgamma4, zgt
   116)       real(kind=r8) :: zr1, zr2, zr3, zr4, zr5
   117)       real(kind=r8) :: zrk, zrk2, zrkg, zrm1, zrp, zrp1, zrpp
   118)       real(kind=r8) :: zsr3, zt1, zt2, zt3, zt4, zt5, zto1
   119)       real(kind=r8) :: zw, zwcrit, zwo
   120) 
   121)       real(kind=r8), parameter :: eps = 1.e-08_r8
   122) 
   123) !     ------------------------------------------------------------------
   124) 
   125) ! Initialize
   126) 
   127)       hvrrft = '$Revision: 1.2 $'
   128) 
   129)       zsr3=sqrt(3._r8)
   130)       zwcrit=0.9999995_r8
   131)       kmodts=2
   132) 
   133)       do jk=1, nlayers
   134)          if (.not.lrtchk(jk)) then
   135)             pref(jk) =0._r8
   136)             ptra(jk) =1._r8
   137)             prefd(jk)=0._r8
   138)             ptrad(jk)=1._r8
   139)          else
   140)             zto1=ptau(jk)
   141)             zw  =pw(jk)
   142)             zg  =pgg(jk)  
   143) 
   144) ! General two-stream expressions
   145) 
   146)             zg3= 3._r8 * zg
   147)             if (kmodts == 1) then
   148)                zgamma1= (7._r8 - zw * (4._r8 + zg3)) * 0.25_r8
   149)                zgamma2=-(1._r8 - zw * (4._r8 - zg3)) * 0.25_r8
   150)                zgamma3= (2._r8 - zg3 * prmuz ) * 0.25_r8
   151)             else if (kmodts == 2) then  
   152)                zgamma1= (8._r8 - zw * (5._r8 + zg3)) * 0.25_r8
   153)                zgamma2=  3._r8 *(zw * (1._r8 - zg )) * 0.25_r8
   154)                zgamma3= (2._r8 - zg3 * prmuz ) * 0.25_r8
   155)             else if (kmodts == 3) then  
   156)                zgamma1= zsr3 * (2._r8 - zw * (1._r8 + zg)) * 0.5_r8
   157)                zgamma2= zsr3 * zw * (1._r8 - zg ) * 0.5_r8
   158)                zgamma3= (1._r8 - zsr3 * zg * prmuz ) * 0.5_r8
   159)             end if
   160)             zgamma4= 1._r8 - zgamma3
   161)     
   162) ! Recompute original s.s.a. to test for conservative solution
   163) 
   164)             zwo= zw / (1._r8 - (1._r8 - zw) * (zg / (1._r8 - zg))**2)
   165)     
   166)             if (zwo >= zwcrit) then
   167) ! Conservative scattering
   168) 
   169)                za  = zgamma1 * prmuz 
   170)                za1 = za - zgamma3
   171)                zgt = zgamma1 * zto1
   172)         
   173) ! Homogeneous reflectance and transmittance,
   174) ! collimated beam
   175) 
   176)                ze1 = min ( zto1 / prmuz , 500._r8)
   177) !               ze2 = exp( -ze1 )
   178) 
   179) ! Use exponential lookup table for transmittance, or expansion of 
   180) ! exponential for low tau
   181)                if (ze1 .le. od_lo) then 
   182)                   ze2 = 1._r8 - ze1 + 0.5_r8 * ze1 * ze1
   183)                else
   184)                   tblind = ze1 / (bpade + ze1)
   185)                   itind = tblint * tblind + 0.5_r8
   186)                   ze2 = exp_tbl(itind)
   187)                endif
   188) !
   189) 
   190)                pref(jk) = (zgt - za1 * (1._r8 - ze2)) / (1._r8 + zgt)
   191)                ptra(jk) = 1._r8 - pref(jk)
   192) 
   193) ! isotropic incidence
   194) 
   195)                prefd(jk) = zgt / (1._r8 + zgt)
   196)                ptrad(jk) = 1._r8 - prefd(jk)        
   197) 
   198) ! This is applied for consistency between total (delta-scaled) and direct (unscaled) 
   199) ! calculations at very low optical depths (tau < 1.e-4) when the exponential lookup
   200) ! table returns a transmittance of 1.0.
   201)                if (ze2 .eq. 1.0_r8) then 
   202)                   pref(jk) = 0.0_r8
   203)                   ptra(jk) = 1.0_r8
   204)                   prefd(jk) = 0.0_r8
   205)                   ptrad(jk) = 1.0_r8
   206)                endif
   207) 
   208)             else
   209) ! Non-conservative scattering
   210) 
   211)                za1 = zgamma1 * zgamma4 + zgamma2 * zgamma3
   212)                za2 = zgamma1 * zgamma3 + zgamma2 * zgamma4
   213)                zrk = sqrt ( zgamma1**2 - zgamma2**2)
   214)                zrp = zrk * prmuz               
   215)                zrp1 = 1._r8 + zrp
   216)                zrm1 = 1._r8 - zrp
   217)                zrk2 = 2._r8 * zrk
   218)                zrpp = 1._r8 - zrp*zrp
   219)                zrkg = zrk + zgamma1
   220)                zr1  = zrm1 * (za2 + zrk * zgamma3)
   221)                zr2  = zrp1 * (za2 - zrk * zgamma3)
   222)                zr3  = zrk2 * (zgamma3 - za2 * prmuz )
   223)                zr4  = zrpp * zrkg
   224)                zr5  = zrpp * (zrk - zgamma1)
   225)                zt1  = zrp1 * (za1 + zrk * zgamma4)
   226)                zt2  = zrm1 * (za1 - zrk * zgamma4)
   227)                zt3  = zrk2 * (zgamma4 + za1 * prmuz )
   228)                zt4  = zr4
   229)                zt5  = zr5
   230)                zbeta = (zgamma1 - zrk) / zrkg !- zr5 / zr4
   231)         
   232) ! Homogeneous reflectance and transmittance
   233) 
   234)                ze1 = min ( zrk * zto1, 500._r8)
   235)                ze2 = min ( zto1 / prmuz , 500._r8)
   236) !
   237) ! Original
   238) !              zep1 = exp( ze1 )
   239) !              zem1 = exp(-ze1 )
   240) !              zep2 = exp( ze2 )
   241) !              zem2 = exp(-ze2 )
   242) !
   243) ! Revised original, to reduce exponentials
   244) !              zep1 = exp( ze1 )
   245) !              zem1 = 1._r8 / zep1
   246) !              zep2 = exp( ze2 )
   247) !              zem2 = 1._r8 / zep2
   248) !
   249) ! Use exponential lookup table for transmittance, or expansion of 
   250) ! exponential for low tau
   251)                if (ze1 .le. od_lo) then 
   252)                   zem1 = 1._r8 - ze1 + 0.5_r8 * ze1 * ze1
   253)                   zep1 = 1._r8 / zem1
   254)                else
   255)                   tblind = ze1 / (bpade + ze1)
   256)                   itind = tblint * tblind + 0.5_r8
   257)                   zem1 = exp_tbl(itind)
   258)                   zep1 = 1._r8 / zem1
   259)                endif
   260) 
   261)                if (ze2 .le. od_lo) then 
   262)                   zem2 = 1._r8 - ze2 + 0.5_r8 * ze2 * ze2
   263)                   zep2 = 1._r8 / zem2
   264)                else
   265)                   tblind = ze2 / (bpade + ze2)
   266)                   itind = tblint * tblind + 0.5_r8
   267)                   zem2 = exp_tbl(itind)
   268)                   zep2 = 1._r8 / zem2
   269)                endif
   270) 
   271) ! collimated beam
   272) 
   273)                zdenr = zr4*zep1 + zr5*zem1
   274)                zdent = zt4*zep1 + zt5*zem1
   275)                if (zdenr .ge. -eps .and. zdenr .le. eps) then
   276)                   pref(jk) = eps
   277)                   ptra(jk) = zem2
   278)                else
   279)                   pref(jk) = zw * (zr1*zep1 - zr2*zem1 - zr3*zem2) / zdenr
   280)                   ptra(jk) = zem2 - zem2 * zw * (zt1*zep1 - zt2*zem1 - zt3*zep2) / zdent
   281)                endif 
   282) 
   283) ! diffuse beam
   284) 
   285)                zemm = zem1*zem1
   286)                zdend = 1._r8 / ( (1._r8 - zbeta*zemm ) * zrkg)
   287)                prefd(jk) =  zgamma2 * (1._r8 - zemm) * zdend
   288)                ptrad(jk) =  zrk2*zem1*zdend
   289) 
   290)             endif
   291) 
   292)          endif         
   293) 
   294)       enddo    
   295) 
   296)       end subroutine reftra_sw
   297) 
   298)       end module rrtmg_sw_reftra
