     4)       use shr_kind_mod,      only : r8 => shr_kind_r8
     5)       use cam_logfile,       only : iulog
     6) 
     7)       implicit none
     8) 
     9)       save
    10) 
    11)       real(r8), parameter :: secpday       = 86400._r8
    12)       real(r8), parameter :: daypsec       = 1._r8/secpday
    13)       real(r8), parameter :: aur_therm     = 807._r8
    14)       real(r8), parameter :: jkcal         = 4184._r8
    15)       real(r8), parameter :: aur_heat_eff  = .05_r8
    16)       real(r8), parameter :: aur_hconst    = 1.e3_r8*jkcal*aur_therm*aur_heat_eff
    17) 
    18)       real(r8) :: max_zen_angle
    19) 
    20)       private
    21)       public :: waccm_hrates, init_hrates, has_hrates
    22) 
    23)       integer :: id_co2, id_o2, id_o3, id_o2_1d, id_o2_1s, id_h2o, id_o, id_h
    24)       logical :: has_hrates
    25)

    28)       subroutine init_hrates( )
    29)         use mo_chem_utls, only : get_spc_ndx
    30)         use cam_history,  only : addfld, phys_decomp
    31)         use ppgrid,       only : pver
    32)         use ref_pres,     only : ptop_ref, psurf_ref
    33) 
    34) 
    35)         implicit none
    36) 
    37)         integer :: ids(8)
    38)         character(len=128) :: attr  ! netcdf variable attribute
    39) 
    40)         id_co2   = get_spc_ndx( 'CO2' )
    41)         id_o2    = get_spc_ndx( 'O2' )
    42)         id_o3    = get_spc_ndx( 'O3' )
    43)         id_o2_1d = get_spc_ndx( 'O2_1D' )
    44)         id_o2_1s = get_spc_ndx( 'O2_1S' )
    45)         id_h2o   = get_spc_ndx( 'H2O' )
    46)         id_o     = get_spc_ndx( 'O' )
    47)         id_h     = get_spc_ndx( 'H' )
    48) 
    49)         ids = (/ id_co2, id_o2, id_o3, id_o2_1d, id_o2_1s, id_h2o, id_o, id_h /)
    50) 
    51)         has_hrates = all( ids(:) > 0 ) .and. ptop_ref < 0.0004_r8 * psurf_ref
    52) 
    53)         if (.not. has_hrates) return
    54) 
    55)         call addfld( 'CPAIR', 'J/K/kg', pver, 'I', 'specific heat cap air', phys_decomp )
    56)         call addfld( 'QRS_AUR', 'K/s', pver, 'I', 'total auroral heating rate', phys_decomp )
    57)         call addfld( 'QRS_CO2NIR', 'K/s', pver, 'I', 'co2 nir heating rate', phys_decomp )
    58)         call addfld( 'QTHERMAL', 'K/s', pver, 'I', 'non-euv photolysis heating rate', phys_decomp )
    59)         call addfld( 'QRS_MLT', 'K/s', pver, 'I', 'Total heating rate (unmerged with tropospheric RT heating)', phys_decomp )
    60) 
    61)         attr = 'O2 + hv -> O1D + O3P solar heating rate < 200nm'
    62)         call addfld( 'QRS_SO2A', 'K/s ', pver, 'I', trim(attr), phys_decomp )
    63)         attr = 'O2 + hv -> O3P + O3P solar heating rate < 200nm'
    64)         call addfld( 'QRS_SO2B', 'K/s ', pver, 'I', trim(attr), phys_decomp )
    65)         attr = 'O3 + hv -> O1D + O2_1S solar heating rate < 200nm'
    66)         call addfld( 'QRS_SO3A', 'K/s ', pver, 'I', trim(attr), phys_decomp )
    67)         attr = 'O3 + hv -> O3P + O2 solar heating rate < 200nm'
    68)         call addfld( 'QRS_SO3B', 'K/s ', pver, 'I', trim(attr), phys_decomp )
    69)         attr = 'O2 + hv -> 2*O3P solar heating rate > 200nm'
    70)         call addfld( 'QRS_LO2B', 'K/s ', pver, 'I', trim(attr), phys_decomp )
    71)         attr = 'O3 + hv -> O1D + O2_1S solar heating rate > 200nm'
    72)         call addfld( 'QRS_LO3A', 'K/s ', pver, 'I', trim(attr), phys_decomp )
    73)         attr = 'O3 + hv -> O3P + O2 solar heating rate > 200nm'
    74)         call addfld( 'QRS_LO3B', 'K/s ', pver, 'I', trim(attr), phys_decomp )
    75)         attr = 'Total O3 solar heating > 200nm'
    76)         call addfld( 'QRS_LO3',  'K/s ', pver, 'I', trim(attr), phys_decomp )
    77)         attr = 'total euv heating rate'
    78)         call addfld( 'QRS_EUV', 'K/s', pver, 'I', trim(attr), phys_decomp )
    79)         attr = 'total jo2 euv photolysis rate'
    80)         call addfld( 'JO2_EUV', '/s', pver, 'I', trim(attr), phys_decomp )
    81) 
    82)       end subroutine init_hrates
    83) 
    84)       subroutine waccm_hrates(ncol, state, asdir, bot_mlt_lev, qrs_tot, pbuf )
    85) !-----------------------------------------------------------------------
    86) !     ... computes the short wavelength heating rates
    87) !-----------------------------------------------------------------------
    88) 
    89)       use chem_mods,         only : nabscol, nfs, gas_pcnst, rxntot, indexm
    90)       use ppgrid,            only : pcols, pver
    91)       use physconst,         only : rga, mbarv, cpairv
    92)       use constituents,      only : pcnst
    93)       use mo_gas_phase_chemdr,only: map2chm
    94)       use mo_photo,          only : set_ub_col, setcol
    95)       use mo_jlong,          only : jlong
    96)       use mo_jshort,         only : jshort
    97)       use mo_jeuv,           only : heuv
    98)       use mo_cph,            only : cph
    99)       use mo_heatnirco2,     only : heatnirco2
   100)       use mo_airglow,        only : airglow
   101)       use mo_aurora,         only : aurora
   102)       use mo_setrxt,         only : setrxt_hrates
   103)       use mo_adjrxt,         only : adjrxt
   104)       use mo_usrrxt,         only : usrrxt_hrates
   105)       use mo_setinv,         only : setinv
   106)       use mo_mass_xforms,    only : mmr2vmr
   107)       use physics_types,     only : physics_state
   108)       use phys_grid,         only : get_rlat_all_p, get_rlon_all_p, &
   109)                                     get_lat_all_p, get_lon_all_p
   110)       use mo_mean_mass,      only : set_mean_mass
   111)       use set_cp,            only : calc_cp
   112)       use cam_history,       only : outfld
   113)       use shr_orb_mod,       only : shr_orb_decl
   114)       use time_manager,      only : get_curr_calday
   115)       use cam_control_mod,   only : lambm0, eccen, mvelpp, obliqr
   116)       use mo_constants,      only : r2d
   117)       use short_lived_species,only: get_short_lived_species
   118)       use physics_buffer,    only : physics_buffer_desc
   119)       use phys_control,      only : waccmx_is
   120) 
   121) !-----------------------------------------------------------------------
   122) !        ... dummy arguments
   123) !-----------------------------------------------------------------------
   124)       integer,             intent(in)  ::  ncol                  ! number columns in chunk
   125)       type(physics_state), intent(in)  ::  state                 ! physics state structure
   126)       real(r8),            intent(in)  ::  asdir(pcols)          ! shortwave, direct albedo
   127)       integer,             intent(in)  ::  bot_mlt_lev           ! lowest model level where MLT heating is needed
   128)       real(r8),            intent(out) ::  qrs_tot(pcols,pver)   ! total heating (K/s)
   129)       type(physics_buffer_desc), pointer :: pbuf(:)
   130) 
   131) !-----------------------------------------------------------------------
   132) !     	... local variables
   133) !-----------------------------------------------------------------------
   134)       integer             :: lchnk                 ! chunk index
   135)       real(r8), parameter :: m2km  = 1.e-3_r8
   136)       real(r8), parameter :: Pa2mb = 1.e-2_r8
   137) 
   138)       integer      ::  i, k, m, n
   139)       integer      ::  kbot_hrates
   140)       real(r8)     ::  esfact
   141)       real(r8)     ::  sza                                           ! solar zenith angle (degrees)
   142)       integer      ::  latndx(pcols)                                 ! chunk lat indicies
   143)       integer      ::  lonndx(pcols)                                 ! chunk lon indicies
   144)       real(r8)     ::  invariants(ncol,pver,nfs)
   145)       real(r8)     ::  col_dens(ncol,pver,max(1,nabscol))            ! column densities (molecules/cm^2)
   146)       real(r8)     ::  col_delta(ncol,0:pver,max(1,nabscol))         ! layer column densities (molecules/cm^2)
   147)       real(r8)     ::  vmr(ncol,pver,gas_pcnst)                      ! xported species (vmr)
   148)       real(r8)     ::  reaction_rates(ncol,pver,rxntot)              ! reaction rates
   149)       real(r8)     ::  mmr(pcols,pver,gas_pcnst)                     ! chem working concentrations (kg/kg)
   150)       real(r8)     ::  h2ovmr(ncol,pver)                             ! water vapor concentration (mol/mol)
   151)       real(r8)     ::  mbar(ncol,pver)                               ! mean wet atmospheric mass (kg/mole)
   152)       real(r8)     ::  zmid(ncol,pver)                               ! midpoint geopotential (km)
   153)       real(r8)     ::  cpair(ncol,pver)                              ! specific heat capacity (J/K/kg)
   154)       real(r8)     ::  cphrate(ncol,pver)                            ! chemical pot heat rate (K/s)
   155)       real(r8)     ::  aghrate(ncol,pver)                            ! airglow heat rate (K/s)
   156)       real(r8)     ::  qrs_col(pver,4)                               ! column thermal heating < 200nm
   157)       real(r8)     ::  qrl_col(pver,4)                               ! column thermal heating > 200nm
   158)       real(r8)     ::  qrs(ncol,pver,4)                              ! chunk thermal heating < 200nm
   159)       real(r8)     ::  qrl(ncol,pver,4)                              ! chunk thermal heating > 200nm
   160)       real(r8)     ::  euv_hrate_col(pver)                           ! column euv thermal heating rate
   161)       real(r8)     ::  co2_hrate_col(pver)                           ! column co2 nir heating rate
   162)       real(r8)     ::  euv_hrate(ncol,pver)                          ! chunk euv thermal heating rate
   163)       real(r8)     ::  aur_hrate(ncol,pver)                          ! chunk auroral heating rate
   164)       real(r8)     ::  co2_hrate(ncol,pver)                          ! chunk co2 nir heating rate
   165)       real(r8)     ::  o2mmr(ncol,pver)                              ! chunk o2 concentration (kg/kg)
   166)       real(r8)     ::  ommr(ncol,pver)                               ! chunk o concentration (kg/kg)
   167)       real(r8)     ::  fac1(pver)                                    ! work array
   168)       real(r8)     ::  colo3(pver)                                   ! vertical o3 column density
   169)       real(r8)     ::  zarg(pver)                                    ! vertical height array
   170)       real(r8)     ::  parg(pver)                                    ! vertical pressure array (hPa)
   171)       real(r8)     ::  tline(pver)                                   ! vertical temperature array
   172)       real(r8)     ::  eff_alb(pver)                                 ! albedo
   173)       real(r8)     ::  mw(pver)                                      ! atms molecular weight
   174)       real(r8)     ::  n2_line(pver)                                 ! n2 density (mol/mol)
   175)       real(r8)     ::  o_line(pver)                                  ! o density (mol/mol)
   176)       real(r8)     ::  o2_line(pver)                                 ! o2 density (mol/mol)
   177)       real(r8)     ::  o3_line(pver)                                 ! o3 density (mol/mol)
   178)       real(r8)     ::  co2_line(pver)                                ! co2 density (mol/mol)
   179)       real(r8)     ::  scco2(pver)                                   ! co2 slant column concentration (molec/cm^2)
   180)       real(r8)     ::  scco2i(pver)                                  ! co2 slant column concentration (molec/cm^2)
   181)       real(r8)     ::  occ(pver)                                     ! o density (molecules/cm^3)
   182)       real(r8)     ::  o2cc(pver)                                    ! o2 density (molecules/cm^3)
   183)       real(r8)     ::  co2cc(pver)                                   ! co2 density (molecules/cm^3)
   184)       real(r8)     ::  n2cc(pver)                                    ! n2 density (molecules/cm^3)
   185)       real(r8)     ::  o3cc(pver)                                    ! o3 density (molecules/cm^3)
   186)       real(r8)     ::  cparg(pver)                                   ! specific heat capacity
   187)       real(r8)     ::  zen_angle(ncol)                               ! solar zenith angles (radians)
   188)       real(r8)     ::  zsurf(ncol)                                   ! surface height (m)
   189)       real(r8)     ::  rlats(ncol)                                   ! chunk latitudes (radians)
   190)       real(r8)     ::  rlons(ncol)                                   ! chunk longitudes (radians)
   191)       real(r8)     ::  calday                                        ! day of year
   192)       real(r8)     ::  delta                                         ! solar declination (radians)
   193)       logical      ::  do_diag
   194) 
   195)       qrs_tot(:ncol,:) = 0._r8
   196)       if (.not. has_hrates) return
   197)       
   198) !-------------------------------------------------------------------------      
   199) !        ... set maximum zenith angle - higher value for higher top model
   200) !-------------------------------------------------------------------------      
   201)       if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   202)          max_zen_angle = 116._r8
   203)       else
   204)          max_zen_angle = 97.01_r8 ! degrees
   205)       endif
   206) 
   207) !-----------------------------------------------------------------------      
   208) !        ... get chunk latitudes and longitudes
   209) !-----------------------------------------------------------------------      
   210)       lchnk = state%lchnk
   211) 
   212)       call get_lat_all_p( lchnk, ncol, latndx )
   213)       call get_lon_all_p( lchnk, ncol, lonndx )
   214)       call get_rlat_all_p( lchnk, ncol, rlats )
   215)       call get_rlon_all_p( lchnk, ncol, rlons )
   216) 
   217) !-----------------------------------------------------------------------      
   218) !        ... set lower limit for heating rates which is now dictated by radheat module
   219) !-----------------------------------------------------------------------      
   220)       kbot_hrates = bot_mlt_lev
   221)       kbot_hrates = min( kbot_hrates,pver )
   222) !     write(iulog,*) 'hrates: kbot_hrates = ',kbot_hrates
   223) 
   224) !-----------------------------------------------------------------------      
   225) !        ... calculate cosine of zenith angle then cast back to angle
   226) !-----------------------------------------------------------------------      
   227)       calday = get_curr_calday()
   228)       call zenith( calday, rlats, rlons, zen_angle, ncol )
   229)       zen_angle(:) = acos( zen_angle(:) )
   230) 
   231) !-----------------------------------------------------------------------      
   232) !        ... map incoming concentrations to working array
   233) !-----------------------------------------------------------------------      
   234)       do m = 1,pcnst
   235)          n = map2chm(m)
   236)          if( n > 0 ) then
   237)             do k = 1,pver
   238)                mmr(:ncol,k,n) = state%q(:ncol,k,m)
   239)             end do
   240)          end if
   241)       end do
   242)       call get_short_lived_species( mmr, lchnk, ncol, pbuf )
   243) 
   244) !-----------------------------------------------------------------------      
   245) !        ... set atmosphere mean mass
   246) !-----------------------------------------------------------------------      
   247)       if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   248)         do k = 1,pver
   249)           mbar(:ncol,k) = mbarv(:ncol,k,lchnk)
   250)         enddo
   251)       else      
   252)         call set_mean_mass( ncol, mmr, mbar )
   253)       endif
   254) !
   255) !-----------------------------------------------------------------------      
   256) !        ... xform from mmr to vmr
   257) !-----------------------------------------------------------------------      
   258)       call mmr2vmr( mmr, vmr, mbar, ncol )
   259) !-----------------------------------------------------------------------      
   260) !        ... xform water vapor from mmr to vmr
   261) !-----------------------------------------------------------------------      
   262)       do k = 1,pver
   263)          h2ovmr(:ncol,k) = vmr(:ncol,k,id_h2o)
   264)       end do
   265) !-----------------------------------------------------------------------      
   266) !        ... xform geopotential height from m to km 
   267) !            and pressure from Pa to mb
   268) !-----------------------------------------------------------------------      
   269)       zsurf(:ncol) = rga * state%phis(:ncol)
   270)       do k = 1,pver
   271)          zmid(:ncol,k) = m2km * (state%zm(:ncol,k) + zsurf(:ncol))
   272)       end do
   273) 
   274) !-----------------------------------------------------------------------      
   275) !        ... set the "invariants"
   276) !-----------------------------------------------------------------------      
   277)       call setinv( invariants, state%t, h2ovmr, vmr, state%pmid, ncol, lchnk, pbuf )
   278) 
   279) !-----------------------------------------------------------------------      
   280) !        ... set the column densities at the upper boundary
   281) !-----------------------------------------------------------------------      
   282)       call set_ub_col( col_delta, vmr, invariants, state%pint(:,1), state%pdel, ncol, lchnk )
   283) 
   284) !-----------------------------------------------------------------------      
   285) !       ...  set rates for "tabular" and user specified reactions
   286) !-----------------------------------------------------------------------      
   287)       do m = 1,rxntot
   288)          do k = 1,pver
   289)             reaction_rates(:,k,m) = 0._r8
   290)          end do
   291)       end do
   292)       call setrxt_hrates( reaction_rates, state%t, invariants(1,1,indexm), ncol, kbot_hrates )
   293)       call usrrxt_hrates( reaction_rates, state%t, state%t, state%t, invariants, &
   294)                           h2ovmr, state%pmid, invariants(:,:,indexm), ncol, kbot_hrates )
   295)       call adjrxt( reaction_rates, invariants, invariants(1,1,indexm), ncol )
   296)       
   297) !-----------------------------------------------------------------------      
   298) !     	... set cp array
   299) !-----------------------------------------------------------------------      
   300)       if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   301)         do k = 1, pver
   302)            cpair(:ncol,k) = cpairv(:ncol,k,lchnk)
   303)         enddo
   304)       else      
   305)         call calc_cp( ncol, vmr, cpair )
   306)       endif
   307) 
   308)       call outfld( 'CPAIR', cpair, ncol, lchnk )
   309) #ifdef HRATES_DEBUG
   310)       write(iulog,*) ' '
   311)       write(iulog,*) '---------------------------------------------'
   312)       write(iulog,*) 'waccm_hrates: cp at lchnk = ',lchnk
   313)       write(iulog,'(1p,5g15.7)') cpair(1,:)
   314)       write(iulog,*) '---------------------------------------------'
   315)       write(iulog,*) ' '
   316) #endif
   317) 
   318) !-----------------------------------------------------------------------      
   319) !     	... set the earth-sun distance factor
   320) !-----------------------------------------------------------------------      
   321)       call shr_orb_decl( calday, eccen, mvelpp, lambm0, obliqr  , &
   322)                          delta, esfact )
   323) !-----------------------------------------------------------------------      
   324) !     	... set the column densities
   325) !-----------------------------------------------------------------------      
   326)       call setcol( col_delta, col_dens, vmr, state%pdel,  ncol )
   327) !-----------------------------------------------------------------------
   328) !        ... compute the thermal heating rates
   329) !-----------------------------------------------------------------------      
   330)       do m = 1,4
   331)          do k = 1,pver
   332)             qrs(:,k,m) = 0._r8
   333)             qrl(:,k,m) = 0._r8
   334)          end do
   335)       end do
   336)       do k = 1,pver
   337)          euv_hrate(:,k) = 0._r8
   338)          co2_hrate(:,k) = 0._r8
   339)       end do
   340) column_loop : &
   341)       do i = 1,ncol
   342)          sza = zen_angle(i)*r2d
   343)          if( sza < max_zen_angle ) then
   344)             zarg(:)     = zmid(i,:)
   345)             parg(:)     = Pa2mb*state%pmid(i,:)
   346)             colo3(:)    = col_dens(i,:,1)
   347)             tline(:)    = state%t(i,:)
   348)             eff_alb(:)  = asdir(i)
   349)             o_line(:)   = vmr(i,:,id_o)
   350)             o2_line(:)  = vmr(i,:,id_o2)
   351)             co2_line(:) = vmr(i,:,id_co2)
   352)             n2_line(:)  = 1._r8 - (o_line(:) + o2_line(:) + vmr(i,:,id_h))
   353)             o3_line(:)  = vmr(i,:,id_o3)
   354)             occ(:)      = o_line(:) * invariants(i,:,indexm)
   355)             o2cc(:)     = o2_line(:) * invariants(i,:,indexm)
   356)             co2cc(:)    = co2_line(:) * invariants(i,:,indexm)
   357)             n2cc(:)     = n2_line(:) * invariants(i,:,indexm)
   358)             o3cc(:)     = o3_line(:) * invariants(i,:,indexm)
   359)             mw(:)       = mbar(i,:)
   360)             cparg(:)    = cpair(i,:)
   361)             do_diag     = .false.
   362)             call jshort( pver, sza, o2_line, o3_line, o2cc, &
   363)                          o3cc, tline, zarg, mw, qrs_col, &
   364)                          cparg, lchnk, i, co2cc, scco2, do_diag )
   365)             call jlong( pver, sza, eff_alb, parg, tline, &
   366)                         mw, o2_line, o3_line, colo3, qrl_col, &
   367)                         cparg, kbot_hrates )
   368)             do m = 1,4
   369)                qrs(i,pver:1:-1,m) = qrs_col(:,m) * esfact
   370)             end do
   371)             do m = 2,4
   372)                qrl(i,:,m) = qrl_col(:,m) * esfact
   373)             end do
   374)             call heuv( pver, sza, occ, o2cc, n2cc, &
   375)                        o_line, o2_line, n2_line, cparg, mw, &
   376)                        zarg, euv_hrate_col, kbot_hrates )
   377)             euv_hrate(i,:) = euv_hrate_col(:) * esfact
   378)             scco2i(1:pver) = scco2(pver:1:-1)
   379)             call heatnirco2( co2_line, scco2i, state%pmid(i,:kbot_hrates), co2_hrate_col, kbot_hrates, &
   380)                              zarg, sza )
   381) #ifdef HRATES_DEBUG
   382)             write(iulog,*) '==================================='
   383)             write(iulog,*) 'hrates: diagnostics for heatco2nir'
   384)             write(iulog,*) 'hrates: co2_line'
   385)             write(iulog,'(1p,5g15.7)') co2_line(:)
   386)             write(iulog,*) 'hrates: scco2'
   387)             write(iulog,'(1p,5g15.7)') scco2i(:)
   388)             write(iulog,*) 'hrates: co2_hrate'
   389)             write(iulog,'(1p,5g15.7)') co2_hrate_col(:)
   390)             write(iulog,*) '==================================='
   391) #endif
   392)             co2_hrate(i,:kbot_hrates) = co2_hrate_col(:kbot_hrates) * esfact * daypsec
   393)          end if
   394)       end do column_loop
   395) 
   396) 
   397)       call outfld( 'QRS_SO2A', qrs(:,:,1), ncol, lchnk )
   398)       call outfld( 'QRS_SO2B', qrs(:,:,2), ncol, lchnk )
   399)       call outfld( 'QRS_SO3A', qrs(:,:,3), ncol, lchnk )
