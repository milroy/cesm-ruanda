   149)  subroutine create_communicator(new_comm, num_procs)
   150) 
   151) ! !DESCRIPTION:
   152) !  This routine creates a separate communicator for a subset of
   153) !  processors under default ice communicator.
   154) !
   155) !  this routine should be called from init_domain1 when the
   156) !  domain configuration (e.g. nprocs_btrop) has been determined
   157) !
   158) ! !REVISION HISTORY:
   159) !  same as module
   160) 
   161) ! !INCLUDES:
   162) 
   163)    include 'mpif.h'
   164) 
   165) ! !INPUT PARAMETERS:

   167)    integer (int_kind), intent(in) :: &

   169) 
   170) ! !OUTPUT PARAMETERS:
   171) 
   172)    integer (int_kind), intent(out) :: &
   173)       new_comm          ! new communicator for this distribution
   174) 
   175) !EOP
   176) !BOC
   177) !-----------------------------------------------------------------------
   178) !
   179) !  local variables
   180) !
   181) !-----------------------------------------------------------------------
   182) 
   183)    integer (int_kind) :: &
   184)      MPI_GROUP_ICE,         &! group of processors assigned to ice
   185)      MPI_GROUP_NEW           ! group of processors assigned to new dist
   186) 
   187)    integer (int_kind) :: &
   188)      ierr                    ! error flag for MPI comms
   189) 
   190)    integer (int_kind), dimension(3) :: &
   191)      range                   ! range of tasks assigned to new dist
   192)                              !  (assumed 0,num_procs-1)
   193) 
   194) !-----------------------------------------------------------------------
   195) !
   196) !  determine group of processes assigned to distribution
   197) !
   198) !-----------------------------------------------------------------------
   199) 
   200)    call MPI_COMM_GROUP (MPI_COMM_ICE, MPI_GROUP_ICE, ierr)
   201) 
   202)    range(1) = 0
   203)    range(2) = num_procs-1
   204)    range(3) = 1
   205) 
   206) !-----------------------------------------------------------------------
   207) !
   208) !  create subroup and communicator for new distribution
   209) !  note: MPI_COMM_CREATE must be called by all procs in MPI_COMM_ICE
   210) !
   211) !-----------------------------------------------------------------------
   212) 
   213) #ifdef SPMD
   214)    call MPI_GROUP_RANGE_INCL(MPI_GROUP_ICE, 1, range, &
   215)                              MPI_GROUP_NEW, ierr)
   216) 
   217)    call MPI_COMM_CREATE (MPI_COMM_ICE, MPI_GROUP_NEW,  &
   218)                          new_comm, ierr)
   219) #else
   220)    call MPI_COMM_DUP(MPI_COMM_ICE, new_comm, ierr)
   221) #endif
   222) 
   223) !-----------------------------------------------------------------------
   224) !EOC
   225) 
   226)  end subroutine create_communicator
