    11)       subroutine ps2str( nw, zen, rsfc, tauu, omu, &
    12)                          gu, dsdh, nid, radfld )
    13) !-----------------------------------------------------------------------------
    14) !   purpose:
    15) !   solve two-stream equations for multiple layers.  the subroutine is based
    16) !   on equations from:  toon et al., j.geophys.res., v94 (d13), nov 20, 1989.
    17) !   it contains 9 two-stream methods to choose from.  a pseudo-spherical
    18) !   correction has also been added.
    19) !-----------------------------------------------------------------------------
    20) !   parameters:
    21) !   nlevel  - integer, number of specified altitude levels in the working (i)
    22) !             grid
    23) !   zen     - real(r8), solar zenith angle (degrees)                          (i)
    24) !   rsfc    - real(r8), surface albedo at current wavelength                  (i)
    25) !   tauu    - real(r8), unscaled optical depth of each layer                  (i)
    26) !   omu     - real(r8), unscaled single scattering albedo of each layer       (i)
    27) !   gu      - real(r8), unscaled asymmetry parameter of each layer            (i)
    28) !   dsdh    - real(r8), slant path of direct beam through each layer crossed  (i)
    29) !             when travelling from the top of the atmosphere to layer i;
    30) !             dsdh(i,j), i = 0..nz-1, j = 1..nz-1
    31) !   nid     - integer, number of layers crossed by the direct beam when   (i)
    32) !             travelling from the top of the atmosphere to layer i;
    33) !             nid(i), i = 0..nz-1
    34) !   delta   - logical, switch to use delta-scaling                        (i)
    35) !             .true. -> apply delta-scaling
    36) !             .false.-> do not apply delta-scaling
    37) !   fdr     - real(r8), contribution of the direct component to the total     (o)
    38) !             actinic flux at each altitude level
    39) !   fup     - real(r8), contribution of the diffuse upwelling component to    (o)
    40) !             the total actinic flux at each altitude level
    41) !   fdn     - real(r8), contribution of the diffuse downwelling component to  (o)
    42) !             the total actinic flux at each altitude level
    43) !   edr     - real(r8), contribution of the direct component to the total     (o)
    44) !             spectral irradiance at each altitude level
    45) !   eup     - real(r8), contribution of the diffuse upwelling component to    (o)
    46) !             the total spectral irradiance at each altitude level
    47) !   edn     - real(r8), contribution of the diffuse downwelling component to  (o)
    48) !             the total spectral irradiance at each altitude level
    49) !-----------------------------------------------------------------------------
    50) 
    51)       use mo_params,    only : smallest, largest
    52)       use mo_constants, only : d2r
    53)       use ppgrid,       only : pver, pverp
    54)       use mo_trislv,    only : tridec, trislv
    55) 
    56)       implicit none
    57) 
    58) !-----------------------------------------------------------------------------
    59) !	... dummy arguments
    60) !-----------------------------------------------------------------------------
    61)       integer, intent(in)     :: nw
    62)       integer, intent(in)     :: nid(0:pver)
    63)       real(r8), intent(in)    :: zen
    64)       real(r8), intent(in)    :: rsfc(nw)
    65)       real(r8), intent(in)    :: tauu(pver,nw)
    66)       real(r8), intent(in)    :: omu(pver,nw)
    67)       real(r8), intent(in)    :: gu(pver,nw)
    68)       real(r8), intent(in)    :: dsdh(0:pver,pver)
    69)       real(r8), intent(out)   :: radfld(pverp,nw)
    70) 
    71) !-----------------------------------------------------------------------------
    72) !	... local variables
    73) !-----------------------------------------------------------------------------
    74) !-----------------------------------------------------------------------------
    75) ! 	... mu = cosine of solar zenith angle
    76) !           rsfc = surface albedo
    77) !           tauu =  unscaled optical depth of each layer
    78) !           omu  =  unscaled single scattering albedo
    79) !           gu   =  unscaled asymmetry factor
    80) !           klev = max dimension of number of layers in atmosphere
    81) !           nlayer = number of layers in the atmosphere
    82) !           nlevel = nlayer + 1 = number of levels
    83) !-----------------------------------------------------------------------------
    84)       integer, parameter  :: mrows = 2*pver
    85)       integer, parameter  :: pverm = pver - 1
    86)       real(r8), parameter :: eps = 1.e-3_r8
    87)       real(r8), parameter :: pifs = 1._r8
    88)       real(r8), parameter :: fdn0 = 0._r8
    89) 
    90)       integer :: row
    91)       integer :: lev
    92)       integer :: i, ip1, wn
    93)       integer :: j, jl, ju
    94) 
    95)       real(r8) :: precis, wrk
    96)       real(r8) :: tempg
    97)       real(r8) :: mu, suma
    98)       real(r8) :: g, om
    99)       real(r8) :: gam1, gam2, gam3, gam4
   100)       real(r8), dimension(pver)    :: f, gi, omi
   101)       real(r8), dimension(0:pverp) :: tauc, mu2
   102)       real(r8), dimension(pver)    :: lam, taun, bgam
   103)       real(r8), dimension(pver)    :: cdn
   104)       real(r8), dimension(0:pverp,nw) :: tausla
   105)       real(r8), dimension(pver,nw)  :: cup, cuptn, cdntn
   106)       real(r8), dimension(pver,nw)  :: e1, e2, e3, e4
   107)       real(r8), dimension(mrows)    :: a, b, d, e
   108)       real(r8), dimension(nw,mrows) :: sub, main, super, y
   109) !-----------------------------------------------------------------------------
   110) ! 	... for calculations of associated legendre polynomials for gama1,2,3,4
   111) !           in delta-function, modified quadrature, hemispheric constant,
   112) !           hybrid modified eddington-delta function metods, p633,table1.
   113) !           w.e.meador and w.r.weaver, gas,1980,v37,p.630
   114) !           w.j.wiscombe and g.w. grams, gas,1976,v33,p2440, 
   115) !           uncomment the following two lines and the appropriate statements
   116) !           further down.
   117) !-----------------------------------------------------------------------------
   118)       real(r8) :: expon, expon0, expon1, divisr, temp, up, dn
   119)       real(r8) :: ssfc
   120) 
   121) !-----------------------------------------------------------------------------
   122) ! 	... initial conditions:  pi*solar flux = 1;  diffuse incidence = 0
   123) !-----------------------------------------------------------------------------
   124)       precis = epsilon( precis )
   125) 
   126)       mu = cos( zen*d2r )
   127) wave_loop : &
   128)       do wn = 1,nw
   129) !-----------------------------------------------------------------------------
   130) !	... compute coefficients for each layer:
   131) !           gam1 - gam4 = 2-stream coefficients, different for different approximations
   132) !           expon0 = calculation of e when tau is zero
   133) !           expon1 = calculation of e when tau is taun
   134) !           cup and cdn = calculation when tau is zero
   135) !           cuptn and cdntn = calc. when tau is taun
   136) !           divisr = prevents division by zero
   137) !-----------------------------------------------------------------------------
   138)          tauc(0:pverp)      = 0._r8
   139)          tausla(0:pverp,wn) = 0._r8
   140)          mu2(0:pverp)       = sqrt( smallest )
   141) 
   142) !-----------------------------------------------------------------------------
   143) ! 	... delta-scaling. have to be done for delta-eddington approximation, 
   144) !           delta discrete ordinate, practical improved flux method, delta function,
   145) !           and hybrid modified eddington-delta function methods approximations
   146) !-----------------------------------------------------------------------------
   147)          f(1:pver)    = gu(:,wn)*gu(:,wn)
   148)          gi(1:pver)   = (gu(:,wn) - f(1:pver))/(1._r8 - f(1:pver))
   149)          omi(1:pver)  = (1._r8 - f(1:pver))*omu(1:pver,wn)/(1._r8 - omu(1:pver,wn)*f(1:pver))       
   150)          taun(1:pver) = (1._r8 - omu(1:pver,wn)*f(1:pver))*tauu(1:pver,wn)
   151) 
   152) !-----------------------------------------------------------------------------
   153) ! 	... calculate slant optical depth at the top of the atmosphere when zen>90.
   154) !           in this case, higher altitude of the top layer is recommended which can 
   155) !           be easily changed in gridz.f.
   156) !-----------------------------------------------------------------------------
   157)          if( zen > 90._r8 ) then
   158)             if( nid(0) < 0 ) then
   159)                tausla(0,wn) = largest
   160)             else
   161) 	       ju = nid(0)
   162)                tausla(0,wn) = 2._r8*dot_product( taun(1:ju),dsdh(0,1:ju) )
   163)             end if
   164)          end if
   165) level_loop : &  
   166)          do i = 1,pver
   167)             g = gi(i)
   168)             om = omi(i)
   169)             tauc(i) = tauc(i-1) + taun(i)
   170) !-----------------------------------------------------------------------------
   171) ! 	... stay away from 1 by precision.  for g, also stay away from -1
   172) !-----------------------------------------------------------------------------
   173)             tempg = min( abs(g),1._r8 - precis )
   174)             g     = sign( tempg,g )
   175)             om    = min( om,1._r8 - precis )
   176) !-----------------------------------------------------------------------------
   177) ! 	... calculate slant optical depth
   178) !-----------------------------------------------------------------------------
   179)             if( nid(i) < 0 ) then
   180)                tausla(i,wn) = largest
   181)             else
   182)                ju = min( nid(i),i )
   183)                suma = dot_product( taun(1:ju),dsdh(i,1:ju) )
   184)                jl = min( nid(i),i ) + 1
   185)                tausla(i,wn) = suma + 2._r8*dot_product( taun(jl:nid(i)),dsdh(i,jl:nid(i)) )
   186)                if( tausla(i,wn) == tausla(i-1,wn) ) then
   187)                  mu2(i) = sqrt( largest )
   188)                else
   189)                  mu2(i) = (tauc(i) - tauc(i-1))/(tausla(i,wn) - tausla(i-1,wn))
   190)                  mu2(i) = sign( max( abs(mu2(i)),sqrt(smallest) ),mu2(i) )
   191)                end if
   192)             end if
   193) !-----------------------------------------------------------------------------
   194) !	... the following gamma equations are from pg 16,289, table 1
   195) !           eddington approximation(joseph et al., 1976, jas, 33, 2452):
   196) !-----------------------------------------------------------------------------
   197)             gam1 =  .25_r8*(7._r8 - om*(4._r8 + 3._r8*g))
   198)             gam2 = -.25_r8*(1._r8 - om*(4._r8 - 3._r8*g))
   199)             gam3 = .25_r8*(2._r8 - 3._r8*g*mu)
   200)             gam4 = 1._r8 - gam3
   201) !-----------------------------------------------------------------------------
   202) ! 	... lambda = pg 16,290 equation 21
   203) !           big gamma = pg 16,290 equation 22
   204) !-----------------------------------------------------------------------------
   205)             lam(i) = sqrt( gam1*gam1 - gam2*gam2 )
   206)             bgam(i) = (gam1 - lam(i))/gam2
   207) 	    wrk = lam(i)*taun(i)
   208) 	    if( wrk < 500._r8 ) then
   209)                expon = exp( -wrk )
   210) 	    else
   211)                expon = 0._r8
   212) 	    end if
   213) !-----------------------------------------------------------------------------
   214) ! 	... e1 - e4 = pg 16,292 equation 44
   215) !-----------------------------------------------------------------------------
   216)             e1(i,wn) = 1._r8 + bgam(i)*expon
   217)             e2(i,wn) = 1._r8 - bgam(i)*expon
   218)             e3(i,wn) = bgam(i) + expon
   219)             e4(i,wn) = bgam(i) - expon
   220) !-----------------------------------------------------------------------------
   221) ! 	... the following sets up for the c equations 23, and 24
   222) !           found on page 16,290
   223) !           prevent division by zero (if lambda=1/mu, shift 1/mu^2 by eps = 1.e-3
   224) !           which is approx equiv to shifting mu by 0.5*eps* (mu)**3
   225) !-----------------------------------------------------------------------------
   226) 	    if( tausla(i-1,wn) < 500._r8 ) then
   227)                expon0 = exp( -tausla(i-1,wn) )
   228) 	    else
   229)                expon0 = 0._r8
   230) 	    end if
   231) 	    if( tausla(i,wn) < 500._r8 ) then
   232)                expon1 = exp( -tausla(i,wn) )
   233) 	    else
   234)                expon1 = 0._r8
   235) 	    end if
   236)             divisr = lam(i)*lam(i) - 1._r8/(mu2(i)*mu2(i))
   237)             temp = max( eps,abs(divisr) )
   238)             divisr = 1._r8/sign( temp,divisr )
   239)             up = om*pifs*((gam1 - 1._r8/mu2(i))*gam3 + gam4*gam2)*divisr
   240)             dn = om*pifs*((gam1 + 1._r8/mu2(i))*gam4 + gam2*gam3)*divisr
   241) !-----------------------------------------------------------------------------
   242) ! 	... cup and cdn are when tau is equal to zero
   243) !           cuptn and cdntn are when tau is equal to taun
   244) !-----------------------------------------------------------------------------
   245)             cup(i,wn) = up*expon0
   246)             cdn(i)    = dn*expon0
   247)             cuptn(i,wn) = up*expon1
   248)             cdntn(i,wn) = dn*expon1
   249)          end do level_loop
   250) 
   251) !-----------------------------------------------------------------------------
   252) !	... set up matrix
   253) !           ssfc = pg 16,292 equation 37  where pi fs is one (unity).
   254) !-----------------------------------------------------------------------------
   255) 	if( tausla(pver,wn) < 500._r8 ) then
   256)            ssfc = rsfc(wn)*mu*exp( -tausla(pver,wn) )*pifs
   257) 	else
   258)            ssfc = 0._r8
   259) 	end if
   260) 
   261) !-----------------------------------------------------------------------------
   262) ! 	... the following are from pg 16,292  equations 39 - 43.
   263) !           set up first row of matrix:
   264) !-----------------------------------------------------------------------------
   265)         a(1) = 0._r8
   266)         b(1) = e1(1,wn)
   267)         d(1) = -e2(1,wn)
   268)         e(1) = fdn0 - cdn(1)
   269) 
   270) !-----------------------------------------------------------------------------
   271) ! 	... set up odd rows 3 thru (mrows - 1):
   272) !-----------------------------------------------------------------------------
   273)         a(3:mrows-1:2) = e2(1:pverm,wn)*e3(1:pverm,wn) - e4(1:pverm,wn)*e1(1:pverm,wn)
   274)         b(3:mrows-1:2) = e1(1:pverm,wn)*e1(2:pver,wn) - e3(1:pverm,wn)*e3(2:pver,wn)
   275)         d(3:mrows-1:2) = e3(1:pverm,wn)*e4(2:pver,wn) - e1(1:pverm,wn)*e2(2:pver,wn)
   276)         e(3:mrows-1:2) = e3(1:pverm,wn)*(cup(2:pver,wn) - cuptn(1:pverm,wn)) + e1(1:pverm,wn)*(cdntn(1:pverm,wn) - cdn(2:pver))
   277) 
   278) !-----------------------------------------------------------------------------
   279) ! 	... set up even rows 2 thru (mrows - 2): 
   280) !-----------------------------------------------------------------------------
   281)         a(2:mrows-2:2) = e2(2:pver,wn)*e1(1:pverm,wn) - e3(1:pverm,wn)*e4(2:pver,wn)
   282)         b(2:mrows-2:2) = e2(1:pverm,wn)*e2(2:pver,wn) - e4(1:pverm,wn)*e4(2:pver,wn)
   283)         d(2:mrows-2:2) = e1(2:pver,wn)*e4(2:pver,wn) - e2(2:pver,wn)*e3(2:pver,wn)
   284)         e(2:mrows-2:2) = (cup(2:pver,wn) - cuptn(1:pverm,wn))*e2(2:pver,wn) - (cdn(2:pver) - cdntn(1:pverm,wn))*e4(2:pver,wn)
   285) 
   286) !-----------------------------------------------------------------------------
   287) ! 	... set up last row of matrix at mrows:
   288) !-----------------------------------------------------------------------------
   289)         a(mrows) = e1(pver,wn) - rsfc(wn)*e3(pver,wn)
   290)         b(mrows) = e2(pver,wn) - rsfc(wn)*e4(pver,wn)
   291)         d(mrows) = 0._r8
   292)         e(mrows) = ssfc - cuptn(pver,wn) + rsfc(wn)*cdntn(pver,wn)
   293) 
   294) 	sub(wn,1:mrows)   = a(1:mrows)
   295) 	main(wn,1:mrows)  = b(1:mrows)
   296) 	super(wn,1:mrows) = d(1:mrows)
   297) 	y(wn,1:mrows)     = e(1:mrows)
   298)       end do wave_loop
   299) 
   300) !-----------------------------------------------------------------------------
   301) ! 	... solve the system
   302) !-----------------------------------------------------------------------------
   303)       call tridec( nw, mrows, sub, main, super )
   304)       call trislv( nw, mrows, sub, main, super, y )
   305) 
   306) !-----------------------------------------------------------------------------
   307) !	... unfold solution of matrix, compute output fluxes
   308) !-----------------------------------------------------------------------------
   309)       do wn = 1,nw
   310) !-----------------------------------------------------------------------------
   311) ! 	... the following equations are from pg 16,291  equations 31 & 32
   312) !-----------------------------------------------------------------------------
   313) 	 e(:mrows) = y(wn,:mrows)
   314) 	 if( tausla(0,wn) < 500._r8 ) then
   315)             radfld(1,wn) = 2._r8*(fdn0 +  e(1)*e3(1,wn) - e(2)*e4(1,wn) + cup(1,wn)) + exp( -tausla(0,wn) )
   316) 	 else
   317)             radfld(1,wn) = 2._r8*(fdn0 +  e(1)*e3(1,wn) - e(2)*e4(1,wn) + cup(1,wn))
   318) 	 end if
   319) 	 where( tausla(1:pver,wn) < 500._r8 )
   320) 	    cdn(1:pver) = exp( -tausla(1:pver,wn) )
   321) 	 elsewhere
   322) 	    cdn(1:pver) = 0._r8
   323) 	 endwhere
   324)          radfld(2:pverp,wn) = 2._r8*(e(1:mrows-1:2)*(e3(1:pver,wn) + e1(1:pver,wn)) &
   325)                             + e(2:mrows:2)*(e4(1:pver,wn) + e2(1:pver,wn))       &
   326)                             + cdntn(1:pver,wn) + cuptn(1:pver,wn)) + cdn(1:pver)
   327)       end do
   328) 
   329)       end subroutine ps2str
