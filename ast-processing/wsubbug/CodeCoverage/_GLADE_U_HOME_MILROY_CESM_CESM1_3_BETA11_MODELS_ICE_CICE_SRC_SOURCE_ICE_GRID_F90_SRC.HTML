   147) ! !IROUTINE: init_grid1 - - distribute blocks across processors
   148) !
   149) ! !INTERFACE:
   150) !
   151)       subroutine init_grid1 
   152) !
   153) ! !DESCRIPTION:
   154) !
   155) ! Distribute blocks across processors.  The distribution is optimized
   156) ! based on latitude and topography, contained in the ULAT and KMT arrays. 
   157) !
   158) ! !REVISION HISTORY:
   159) !
   160) ! authors: William Lipscomb and Phil Jones, LANL
   161) !
   162) ! !USES:
   163) !
   164)       use ice_broadcast
   165)       use ice_work, only: work_g1, work_g2
   166) !
   167) ! !INPUT/OUTPUT PARAMETERS:
   168) !
   169) !EOP
   170) !
   171) 
   172)       integer (kind=int_kind) :: &
   173)          i, j, iblk, &
   174)          fid_grid, &     ! file id for netCDF grid file
   175)          fid_kmt         ! file id for netCDF kmt file
   176) 
   177)       real (dbl_kind), allocatable, dimension(:,:) :: bStats
   178) 
   179)       integer(kind=int_kind), allocatable :: WorkPerBlock(:),blockType(:)
   180)       real(kind=dbl_kind), allocatable :: ProbPerBlock(:)
   181) 
   182)       character (char_len) :: &
   183)          fieldname       ! field name in netCDF file
   184) 
   185)       !-----------------------------------------------------------------
   186)       ! Get global ULAT and KMT arrays used for block decomposition.
   187)       !-----------------------------------------------------------------
   188) 
   189)       allocate(work_g1(nx_global,ny_global))
   190)       allocate(work_g2(nx_global,ny_global))
   191) 
   192)       if (trim(grid_type) == 'displaced_pole' .or. &
   193)           trim(grid_type) == 'tripole'      ) then
   194) 
   195)          if (trim(grid_format) == 'nc') then
   196) 
   197)             call ice_open_nc(grid_file,fid_grid)
   198)             call ice_open_nc(kmt_file,fid_kmt)
   199) 
   200)             fieldname='ulat'
   201)             call ice_read_global_nc(fid_grid,1,fieldname,work_g1,.true.)
   202)             fieldname='kmt'
   203)             call ice_read_global_nc(fid_kmt,1,fieldname,work_g2,.true.)
   204) 
   205)             if (my_task == master_task) then
   206)                call ice_close_nc(fid_grid)
   207)                call ice_close_nc(fid_kmt)
   208)             endif
   209) 
   210)          else
   211) 
   212)             call ice_open(nu_grid,grid_file,64) ! ULAT
   213)             call ice_open(nu_kmt, kmt_file, 32) ! KMT
   214) 
   215)             call ice_read_global(nu_grid,1,work_g1,'rda8',.true.)  ! ULAT
   216)             call ice_read_global(nu_kmt, 1,work_g2,'ida4',.true.)  ! KMT
   217) 
   218)             if (my_task == master_task) then
   219)                close (nu_grid)
   220)                close (nu_kmt)
   221)             endif
   222) 
   223)          endif
   224) 
   225)       elseif (trim(grid_type) == 'panarctic') then
   226) 
   227)          call ice_open(nu_grid,grid_file,64) ! ULAT, KMT
   228) 
   229)          call ice_read_global(nu_grid,1,work_g2,'ida8',.true.)  ! KMT
   230)          call ice_read_global(nu_grid,2,work_g1,'rda8',.true.)  ! ULAT
   231) 
   232) 
   233) 
   234)          if (my_task == master_task) close (nu_grid)
   235) 
   236)       else   ! rectangular grid
   237) 
   238)          work_g1(:,:) = 75._dbl_kind/rad_to_deg  ! arbitrary polar latitude
   239)          work_g2(:,:) = c1
   240) 
   241)       endif
   242) 
   243)       call broadcast_array(work_g1, master_task)   ! ULAT
   244)       call broadcast_array(work_g2, master_task)   ! KMT
   245) 
   246)       allocate(WorkPerBlock(nblocks_tot),ProbPerBlock(nblocks_tot),blockType(nblocks_tot))
   247)       allocate(bStats(numCoeff,nblocks_tot))
   248)       call CalcWorkPerBlock(distribution_wght, work_g2,work_g1, &
   249)                 WorkPerBlock,ProbPerBlock,blockType,bStats)
   250) 
   251) !      call abort_ice('init_grid1: after call to CalcWorkPerBlock')
   252) !      stop 'init_grid1: after call to CalcWorkPerBlock'
   253) 
   254)       call init_domain_distribution(work_g2, work_g1,  &
   255)                 WorkPerBlock,ProbPerBlock,blockType,bStats,maxDil)  ! KMT, ULAT
   256) !DBG      print *,'init_grid1: after call to init_domain_distribution'
   257) 
   258)       deallocate(bStats)
   259)       deallocate(WorkPerBlock,ProbPerBlock,blockType)
   260) 
   261)       deallocate(work_g1)
   262)       deallocate(work_g2)
   263) 
   264)       !-----------------------------------------------------------------
   265)       ! write additional domain information
   266)       !-----------------------------------------------------------------
   267) 
   268)       if (my_task == master_task) then
   269)         write(nu_diag,'(a26,i6)') '  Block size:  nx_block = ',nx_block
   270)         write(nu_diag,'(a26,i6)') '               ny_block = ',ny_block
   271)       endif
   272) 
   273)       end subroutine init_grid1

   278) ! !IROUTINE: init_grid2 - horizontal grid initialization
   279) !
   280) ! !INTERFACE:
   281) !
   282)       subroutine init_grid2
   283) !
   284) ! !DESCRIPTION:
   285) !
   286) ! Horizontal grid initialization:
   287) !
   288) !     U{LAT,LONG} = true {latitude,longitude} of U points
   289) !     HT{N,E} = cell widths on {N,E} sides of T cell
   290) !     ANGLE = angle between local x direction and true east
   291) !     hm = land mask (c1 for ocean points, c0 for land points)
   292) !     D{X,Y}{T,U} = {x,y} spacing centered at {T,U} points
   293) !     T-grid and ghost cell values
   294) !     Various grid quantities needed for dynamics and transport
   295) !
   296) ! !REVISION HISTORY:
   297) !
   298) ! author: Elizabeth C. Hunke, LANL
   299) !
   300) ! !USES:
   301) !
   302)       use ice_work, only: work_g1
   303)       use ice_exit
   304)       use ice_blocks, only: get_block, block
   305) !
   306) ! !INPUT/OUTPUT PARAMETERS:
   307) !
   308) !EOP
   309) !
   310)       integer (kind=int_kind) :: &
   311)          i, j, iblk, &
   312)          ilo,ihi,jlo,jhi      ! beginning and end of physical domain
   313) 
   314)       real (kind=dbl_kind) :: &
   315)          angle_0, angle_w, angle_s, angle_sw
   316) 
   317)       logical (kind=log_kind), dimension(nx_block,ny_block,max_blocks):: &
   318)          out_of_range
   319) 
   320)       type (block) :: &
   321)          this_block           ! block information for current block
   322)       
   323)       !-----------------------------------------------------------------
   324)       ! lat, lon, cell widths, angle, land mask
   325)       !-----------------------------------------------------------------
   326) 
   327)       if (trim(grid_type) == 'displaced_pole' .or. &
   328)           trim(grid_type) == 'tripole'      ) then
   329)          if (trim(grid_format) == 'nc') then
   330)             call popgrid_nc     ! read POP grid lengths from nc file
   331)          else
   332)             call popgrid        ! read POP grid lengths directly
   333)          endif 
   334)       elseif (trim(grid_type) == 'panarctic') then
   335)          call panarctic_grid    ! pan-Arctic grid
   336) #ifdef CCSMCOUPLED
   337)       elseif (trim(grid_type) == 'latlon') then
   338)          call latlongrid        ! lat lon grid for sequential CCSM (CAM mode)
   339)          return
   340) #endif
   341)       else
   342)          call rectgrid          ! regular rectangular grid
   343)       endif
   344) 
   345)       !-----------------------------------------------------------------
   346)       ! T-grid cell and U-grid cell quantities
   347)       !-----------------------------------------------------------------
   348) 
   349)       !$OMP PARALLEL DO PRIVATE(iblk,this_block,ilo,ihi,jlo,jhi,i,j)
   350)       do iblk = 1, nblocks
   351)          this_block = get_block(blocks_ice(iblk),iblk)         
   352)          ilo = this_block%ilo
   353)          ihi = this_block%ihi
   354)          jlo = this_block%jlo
   355)          jhi = this_block%jhi
   356) 
   357)          do j = jlo, jhi
   358)          do i = ilo, ihi
   359)             tarea(i,j,iblk) = dxt(i,j,iblk)*dyt(i,j,iblk)
   360)             uarea(i,j,iblk) = dxu(i,j,iblk)*dyu(i,j,iblk)
   361)             if (tarea(i,j,iblk) > c0) then
   362)                tarear(i,j,iblk) = c1/tarea(i,j,iblk)
   363)             else
   364)                tarear(i,j,iblk) = c0 ! possible on boundaries
   365)             endif
   366)             if (uarea(i,j,iblk) > c0) then
   367)                uarear(i,j,iblk) = c1/uarea(i,j,iblk)
   368)             else
   369)                uarear(i,j,iblk) = c0 ! possible on boundaries
   370)             endif
   371)             tinyarea(i,j,iblk) = puny*tarea(i,j,iblk)
   372) 
   373)             dxhy(i,j,iblk) = p5*(HTE(i,j,iblk) - HTE(i-1,j,iblk))
   374)             dyhx(i,j,iblk) = p5*(HTN(i,j,iblk) - HTN(i,j-1,iblk))
   375)          enddo
   376)          enddo
   377) 
   378)          do j = jlo, jhi+1
   379)          do i = ilo, ihi+1
   380)             cyp(i,j,iblk) = (c1p5*HTE(i,j,iblk) - p5*HTE(i-1,j,iblk))
   381)             cxp(i,j,iblk) = (c1p5*HTN(i,j,iblk) - p5*HTN(i,j-1,iblk))
   382)             ! match order of operations in cyp, cxp for tripole grids
   383)             cym(i,j,iblk) = -(c1p5*HTE(i-1,j,iblk) - p5*HTE(i,j,iblk))
   384)             cxm(i,j,iblk) = -(c1p5*HTN(i,j-1,iblk) - p5*HTN(i,j,iblk))
   385)          enddo
   386)          enddo
   387) 
   388)       enddo                     ! iblk
   389)       !$OMP END PARALLEL DO
   390) 
   391)       !-----------------------------------------------------------------
   392)       ! Ghost cell updates
   393)       ! On the tripole grid, one must be careful with updates of
   394)       !  quantities that involve a difference of cell lengths.
   395)       ! For example, dyhx and dxhy are cell-centered vector components.
   396)       ! Also note that on the tripole grid, cxp and cxm would swap places,
   397)       !  as would cyp and cym.  These quantities are computed only
   398)       !  in north and east ghost cells (above), not south and west.
   399)       !-----------------------------------------------------------------
   400) 
   401)       call ice_timer_start(timer_bound)
   402)       call ice_HaloUpdate (tarea,              halo_info, &
   403)                            field_loc_center,   field_type_scalar, &
   404)                            fillValue=c1)
   405)       call ice_HaloUpdate (uarea,              halo_info, &
   406)                            field_loc_NEcorner, field_type_scalar, &
   407)                            fillValue=c1)
   408)       call ice_HaloUpdate (tarear,             halo_info, &
   409)                            field_loc_center,   field_type_scalar, &
   410)                            fillValue=c1)
   411)       call ice_HaloUpdate (uarear,             halo_info, &
   412)                            field_loc_NEcorner, field_type_scalar, &
   413)                            fillValue=c1)
   414)       call ice_HaloUpdate (tinyarea,           halo_info, &
   415)                            field_loc_center,   field_type_scalar, &
   416)                            fillValue=c1)
   417)       call ice_HaloUpdate (dxhy,               halo_info, &
   418)                            field_loc_center,   field_type_vector, &
   419)                            fillValue=c1)
   420)       call ice_HaloUpdate (dyhx,               halo_info, &
   421)                            field_loc_center,   field_type_vector, &
   422)                            fillValue=c1)
   423)       call ice_timer_stop(timer_bound)
   424) 
   425)       !-----------------------------------------------------------------
   426)       ! Calculate ANGLET to be compatible with POP ocean model
   427)       ! First, ensure that -pi <= ANGLE <= pi
   428)       !-----------------------------------------------------------------
   429) 
   430)       out_of_range = .false.
   431)       where (ANGLE < -pi .or. ANGLE > pi) out_of_range = .true.
   432)       if (count(out_of_range) > 0) then
   433)          call abort_ice ('ice: init_grid: ANGLE out of expected range')
   434)       endif
   435) 
   436)       !-----------------------------------------------------------------
   437)       ! Compute ANGLE on T-grid
   438)       !-----------------------------------------------------------------
   439)       ANGLET = c0
   440) 
   441)       !$OMP PARALLEL DO PRIVATE(iblk,this_block,ilo,ihi,jlo,jhi,i,j,&
   442)       !$OMP                     angle_0,angle_w,angle_s,angle_sw)
   443)       do iblk = 1, nblocks
   444)          this_block = get_block(blocks_ice(iblk),iblk)         
   445)          ilo = this_block%ilo
   446)          ihi = this_block%ihi
   447)          jlo = this_block%jlo
   448)          jhi = this_block%jhi
   449) 
   450)          do j = jlo, jhi
   451)          do i = ilo, ihi
   452)             angle_0  = ANGLE(i  ,j  ,iblk) !   w----0
   453)             angle_w  = ANGLE(i-1,j  ,iblk) !   |    |
   454)             angle_s  = ANGLE(i,  j-1,iblk) !   |    |
   455)             angle_sw = ANGLE(i-1,j-1,iblk) !   sw---s
   456) 
   457)             if ( angle_0 < c0 ) then
   458)                if ( abs(angle_w - angle_0) > pi) &
   459)                         angle_w = angle_w  - pi2
   460)                if ( abs(angle_s - angle_0) > pi) &
   461)                         angle_s = angle_s  - pi2
   462)                if ( abs(angle_sw - angle_0) > pi) &
   463)                         angle_sw = angle_sw - pi2
   464)             endif
   465) 
   466)             ANGLET(i,j,iblk) = angle_0 * p25 + angle_w * p25 &
   467)                              + angle_s * p25 + angle_sw* p25
   468)          enddo
   469)          enddo
   470)       enddo
   471)       !$OMP END PARALLEL DO
   472)       
   473)       call ice_timer_start(timer_bound)
   474)       call ice_HaloUpdate (ANGLET,           halo_info, &
   475)                            field_loc_center, field_type_angle, &
   476)                            fillValue=c1)
   477)       call ice_timer_stop(timer_bound)
   478) 
   479)       call makemask          ! velocity mask, hemisphere masks
   480) 
   481)       call Tlatlon           ! get lat, lon on the T grid
   482) 
   483)       !----------------------------------------------------------------
   484)       ! Corner coordinates for CF compliant history files
   485)       !----------------------------------------------------------------
   486) 
   487)       call gridbox_corners
   488) 
   489)       !-----------------------------------------------------------------
   490)       ! Compute global index (used for unpacking messages from coupler)
   491)       !-----------------------------------------------------------------
   492) 
   493)       if (my_task==master_task) then
   494)          allocate(work_g1(nx_global,ny_global))
   495)          do j=1,ny_global
   496)          do i=1,nx_global
   497)             work_g1(i,j) = real((j-1)*nx_global + i,kind=dbl_kind)
   498)          enddo
   499)          enddo
   500)       else
   501)          allocate(work_g1(1,1)) ! to save memory
   502)       endif
   503) 
   504)       call scatter_global(rndex_global, work_g1,  &
   505)                           master_task,  distrb_info, &
   506)                           field_loc_center, field_type_scalar)
   507) 
   508)       deallocate(work_g1)
   509) 
   510)       end subroutine init_grid2

   515) ! !IROUTINE: popgrid - read and set POP displaced pole (or tripole)
   516) !                      grid and land mask
   517) !
   518) ! !INTERFACE:
   519) !
   520)       subroutine popgrid
   521) !
   522) ! !DESCRIPTION:
   523) !
   524) ! POP displaced pole grid and land mask. 
   525) ! Grid record number, field and units are: \\
   526) ! (1) ULAT  (radians)    \\
   527) ! (2) ULON  (radians)    \\
   528) ! (3) HTN   (cm)         \\
   529) ! (4) HTE   (cm)         \\
   530) ! (5) HUS   (cm)         \\
   531) ! (6) HUW   (cm)         \\
   532) ! (7) ANGLE (radians)   
   533) !
   534) ! Land mask record number and field is (1) KMT.
   535) !
   536) ! !REVISION HISTORY:

   937) ! !REVISION HISTORY:

   948) !
   949) ! !INPUT/OUTPUT PARAMETERS:
   950) !
   951) !EOP
   952) !
   953)       integer (kind=int_kind) :: &
   954)          i, j, iblk    
   955)       
   956)       integer (kind=int_kind) :: &
   957)          ni, nj, ncid, dimid, varid, ier
   958)

   960)          subname='latlongrid' ! subroutine name
   961)

   963)          this_block           ! block information for current block
   964) 
   965)       integer (kind=int_kind) :: &

   969)            closelat, &        ! Single-column latitude value

   971)            closelatidx, &     ! Single-column latitude index to retrieve
   972)            closelonidx        ! Single-column longitude index to retrieve
   973) 
   974)       integer (kind=int_kind) :: &
   975)            start(2), &        ! Start index to read in
   976)            count(2)           ! Number of points to read in
   977) 
   978)       integer (kind=int_kind) :: &
   979)            start3(3), &        ! Start index to read in
   980)            count3(3)           ! Number of points to read in
   981) 
   982)       integer (kind=int_kind) :: &
   983)         status                ! status flag
   984) 
   985)       real (kind=dbl_kind), allocatable :: &
   986)            lats(:),lons(:),pos_lons(:), glob_grid(:,:)  ! temporaries 
   987) 
   988)       real (kind=dbl_kind) :: &
   989)          pos_scmlon,&         ! temporary
   990)          scamdata             ! temporary
   991) 
   992)       !-----------------------------------------------------------------
   993)       ! - kmt file is actually clm fractional land file
   994)       ! - Determine consistency of dimensions
   995)       ! - Read in lon/lat centers in degrees from kmt file
   996)       ! - Read in ocean from "kmt" file (1 for ocean, 0 for land)
   997)       !-----------------------------------------------------------------
   998) 
   999)       ! Determine dimension of domain file and check for consistency
  1000) 
  1001)       if (my_task == master_task) then
  1002)          call ice_open_nc(kmt_file, ncid)
  1003) 
  1004)          status = nf90_inq_dimid (ncid, 'ni', dimid)
  1005)          status = nf90_inquire_dimension(ncid, dimid, len=ni)
  1006)          status = nf90_inq_dimid (ncid, 'nj', dimid)
  1007)          status = nf90_inquire_dimension(ncid, dimid, len=nj)
  1008)       end if
  1009) 
  1010)       ! Determine start/count to read in for either single column or global lat-lon grid
  1011)       ! If single_column, then assume that only master_task is used since there is only one task
  1012) 
  1013)       if (single_column) then
  1014)          ! Check for consistency
  1015)          if (my_task == master_task) then
  1016)             if ((nx_global /= 1).or. (ny_global /= 1)) then
  1017)                write(nu_diag,*) 'Because you have selected the column model flag'
  1018)                write(nu_diag,*) 'Please set nx_global=ny_global=1 in file'
  1019)                write(nu_diag,*) 'ice_domain_size.F and recompile'
  1020)                call abort_ice ('latlongrid: check nx_global, ny_global')
  1021)             endif
  1022)          end if
  1023) 
  1024)          ! Read in domain file for single column
  1025)          allocate(lats(nj))
  1026)          allocate(lons(ni))
  1027)          allocate(pos_lons(ni))
  1028)          allocate(glob_grid(ni,nj))
  1029) 
  1030)          start3=(/1,1,1/)
  1031)          count3=(/ni,nj,1/)
  1032)          call check_ret(nf90_inq_varid(ncid, 'xc' , varid), subname)
  1033)          call check_ret(nf90_get_var(ncid, varid, glob_grid, start3, count3), subname)
  1034)          do i = 1,ni
  1035)             lons(i) = glob_grid(i,1)
  1036)          end do
  1037) 
  1038)          call check_ret(nf90_inq_varid(ncid, 'yc' , varid), subname)
  1039)          call check_ret(nf90_get_var(ncid, varid, glob_grid, start3, count3), subname)
  1040)          do j = 1,nj
  1041)             lats(j) = glob_grid(1,j) 
  1042)          end do
  1043)          
  1044)          ! convert lons array and scmlon to 0,360 and find index of value closest to 0
  1045)          ! and obtain single-column longitude/latitude indices to retrieve
  1046)          
  1047)          pos_lons(:)= mod(lons(:) + 360._dbl_kind,360._dbl_kind)
  1048)          pos_scmlon = mod(scmlon  + 360._dbl_kind,360._dbl_kind)
  1049)          start(1) = (MINLOC(abs(pos_lons-pos_scmlon),dim=1))
  1050)          start(2) = (MINLOC(abs(lats    -scmlat    ),dim=1))
  1051) 
  1052)          deallocate(lats)
  1053)          deallocate(lons)
  1054)          deallocate(pos_lons)
  1055)          deallocate(glob_grid)
  1056) 
  1057)          call check_ret(nf90_inq_varid(ncid, 'xc' , varid), subname)
  1058)          call check_ret(nf90_get_var(ncid, varid, scamdata, start), subname)
  1059)          TLON = scamdata
  1060)          call check_ret(nf90_inq_varid(ncid, 'yc' , varid), subname)
  1061)          call check_ret(nf90_get_var(ncid, varid, scamdata, start), subname)
  1062)          TLAT = scamdata
  1063)          call check_ret(nf90_inq_varid(ncid, 'area' , varid), subname)
  1064)          call check_ret(nf90_get_var(ncid, varid, scamdata, start), subname)
  1065)          tarea = scamdata
  1066)          call check_ret(nf90_inq_varid(ncid, 'mask' , varid), subname)
  1067)          call check_ret(nf90_get_var(ncid, varid, scamdata, start), subname)
  1068)          hm = scamdata
  1069)          call check_ret(nf90_inq_varid(ncid, 'frac' , varid), subname)
  1070)          call check_ret(nf90_get_var(ncid, varid, scamdata, start), subname)
  1071)          ocn_gridcell_frac = scamdata
  1072)       else
  1073)          ! Check for consistency
  1074)          if (my_task == master_task) then
  1075)             if (nx_global /= ni .and. ny_global /= nj) then

  1091)       end if

  1096)          ilo = this_block%ilo
  1097)          ihi = this_block%ihi
  1098)          jlo = this_block%jlo
  1099)          jhi = this_block%jhi

  1121)       ! because we only use thermodynamic sea ice.  However, ULAT is used
  1122)       ! in the default initialization of CICE so we calculate it here as 
  1123)       ! a "dummy" so that CICE will initialize with ice.  If a no ice
  1124)       ! initialization is OK (or desired) this can be commented out and
  1125)       ! ULAT will remain 0 as specified above.  ULAT is located at the
  1126)       ! NE corner of the grid cell, TLAT at the center, so here ULAT is
  1127)       ! hacked by adding half the latitudinal spacing (in radians) to
  1128)       ! TLAT.
  1129)       !-----------------------------------------------------------------

  1134)          ilo = this_block%ilo
  1135)          ihi = this_block%ihi
  1136)          jlo = this_block%jlo
  1137)          jhi = this_block%jhi

  1140)          do i = ilo, ihi
  1141) 
  1142)             if (ny_global == 1) then
  1143)                uarea(i,j,iblk)  = tarea(i,j,  iblk)
  1144)             else
  1145)                uarea(i,j,iblk)  = p25*  &
  1146)                                  (tarea(i,j,  iblk) + tarea(i+1,j,  iblk) &
  1147)                                 + tarea(i,j+1,iblk) + tarea(i+1,j+1,iblk))
  1148)             endif

  1150)             uarear(i,j,iblk)   = c1/uarea(i,j,iblk)
  1151)             tinyarea(i,j,iblk) = puny*tarea(i,j,iblk)
  1152) 
  1153)             if (single_column) then
  1154)                ULAT  (i,j,iblk) = TLAT(i,j,iblk)+(pi/nj)  
  1155)             else
  1156)                if (ny_global == 1) then
  1157)                   ULAT  (i,j,iblk) = TLAT(i,j,iblk)
  1158)                else
  1159)                   ULAT  (i,j,iblk) = TLAT(i,j,iblk)+(pi/ny_global)  
  1160)                endif
  1161)             endif
  1162)             ULON  (i,j,iblk) = c0
  1163)             ANGLE (i,j,iblk) = c0                             
  1164) 
  1165)             ANGLET(i,j,iblk) = c0

  1167)             HTE   (i,j,iblk) = 1.e36_dbl_kind
  1168)             dxt   (i,j,iblk) = 1.e36_dbl_kind
  1169)             dyt   (i,j,iblk) = 1.e36_dbl_kind
  1170)             dxu   (i,j,iblk) = 1.e36_dbl_kind
  1171)             dyu   (i,j,iblk) = 1.e36_dbl_kind
  1172)             dxhy  (i,j,iblk) = 1.e36_dbl_kind
  1173)             dyhx  (i,j,iblk) = 1.e36_dbl_kind
  1174)             cyp   (i,j,iblk) = 1.e36_dbl_kind
  1175)             cxp   (i,j,iblk) = 1.e36_dbl_kind
  1176)             cym   (i,j,iblk) = 1.e36_dbl_kind
  1177)             cxm   (i,j,iblk) = 1.e36_dbl_kind
  1178)          enddo
  1179)          enddo
  1180)       enddo

  1182)

  1993) ! !IROUTINE: sinlat - calculates sin of latitudes
  1994) !
  1995) ! !INTERFACE:
  1996) !
  1997)       subroutine sinlat(a, k)
  1998) !
  1999) ! !DESCRIPTION:
  2000) !
  2001) ! Calculates the sin of latitudes on the grid based on ny_global and
  2002) ! using code from CAM (/control/gauaw_mod.F90)  In CAM, gauaw_mod.F90 is
  2003) ! only used to calculate the sin of latitudes (and thence latitude) if
  2004) ! the dynamical core is set to eul.  If using one of the other dynamical
  2005) ! cores and coupling to stand alone CAM the latitudes calculated in this
  2006) ! way may not match the grid from CAM.
  2007) !
  2008) ! !REVISION HISTORY:
  2009) !
  2010) ! author: Jacob Sewall
  2011) !
  2012) ! !USES:
  2013)       use ice_exit
  2014) 
  2015) !
  2016) ! !INPUT/OUTPUT PARAMETERS:
  2017) !
  2018) 
  2019)       real (kind=dbl_kind), dimension(ny_global), intent(out) :: & 
  2020)            a            ! sin of latitudes
  2021) 
  2022)       integer (kind=int_kind), intent(in) :: &
  2023)            k            ! number of latitudes (ny_global)
  2024) !
  2025) !EOP
  2026) !
  2027)       real (kind=dbl_kind), dimension(k) :: &
  2028)            sinlats      ! sine of latitudes
  2029) 
  2030)       real (kind=dbl_kind) :: &
  2031)            eps      , & ! convergence criterion
  2032)            c        , & ! constant combination
  2033)            fk       , & ! real k
  2034)            xz       , & ! abscissa estimate
  2035)            pkm1     , & ! |
  2036)            pkm2     , & ! |-polynomials
  2037)            pkmrk    , & ! |
  2038)            pk       , & ! |
  2039)            sp       , & ! current iteration latitude increment
  2040)            avsp     , & ! |sp|
  2041)            fn       , & ! real n
  2042)            avsp_prev, &
  2043)            testdiff
  2044) 
  2045)       real (kind=dbl_kind), parameter :: &
  2046)            eps27 = 1.e-27_dbl_kind
  2047) 
  2048)       integer (kind=int_kind) :: &
  2049)            n, l     , &
  2050)            iter     , & ! iteration counter
  2051)            is       , & ! latitude index
  2052)            kk           ! k/2 (number of latitudes in a hemisphere)
  2053) 
  2054) !
  2055) !----------------------------------------------------------------------
  2056) !
  2057) 
  2058)       c = (c1-(c2/pi)**2)*p25
  2059)       fk = k
  2060)       kk = k/2
  2061)       call bsslzr(sinlats,kk)
  2062)       do is=1,kk
  2063)         xz = cos(sinlats(is)/sqrt((fk+p5)**2+c))
  2064) !
  2065) ! This is the first approximation to xz
  2066) !
  2067)         iter = 0
  2068)         avsp = c100     !initialize avsp to a very large number
  2069) 10      continue
  2070)         avsp_prev = avsp
  2071)         pkm2 = c1
  2072)         pkm1 = xz
  2073)         iter = iter + 1
  2074)         if (iter > 100) then  ! Error exit
  2075)            call abort_ice('SINLAT: no convergence in 100 iterations')
  2076)         end if
  2077) !
  2078) ! Computation of the legendre polynomial
  2079) !
  2080)         do n=2,k
  2081)           fn = n
  2082)           pk = ((c2*fn-1._dbl_kind)*xz*pkm1-(fn-c1)*pkm2)/fn
  2083)           pkm2 = pkm1
  2084)           pkm1 = pk
  2085)         enddo
  2086)         pkm1 = pkm2
  2087)         pkmrk = (fk*(pkm1-xz*pk))/(c1-xz**2)
  2088)         sp = pk/pkmrk
  2089)         xz = xz - sp
  2090)         avsp = abs(sp)
  2091)         testdiff = avsp_prev - avsp
  2092)         if (testdiff > eps27) go to 10
  2093)         sinlats(is) = xz
  2094)       end do
  2095) !
  2096) ! Complete the sets of abscissas and weights, using the symmetry.
  2097) ! Also note truncation from real(r8) to real*8
  2098) !
  2099)       do n=1,kk
  2100)         l = k + 1 - n
  2101)         a(n) = sinlats(n)
  2102)         a(l) = -sinlats(n)
  2103)       end do
  2104)  
  2105)       end subroutine sinlat
