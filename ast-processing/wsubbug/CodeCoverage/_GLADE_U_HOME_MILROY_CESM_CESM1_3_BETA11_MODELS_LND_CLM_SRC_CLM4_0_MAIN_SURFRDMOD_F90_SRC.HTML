     8) ! !DESCRIPTION:
     9) ! Contains methods for reading in surface data file and determining
    10) ! two-dimensional subgrid weights as well as writing out new surface
    11) ! dataset. When reading in the surface dataset, determines array
    12) ! which sets the PFT for each of the [maxpatch] patches and
    13) ! array which sets the relative abundance of the PFT.
    14) ! Also fills in the PFTs for vegetated portion of each grid cell.
    15) ! Fractional areas for these points pertain to "vegetated"
    16) ! area not to total grid area. Need to adjust them for fraction of grid
    17) ! that is vegetated. Also fills in urban, lake, wetland, and glacier patches.

    26)                            maxpatch_glcmec

    34)                            ncd_io, check_var, ncd_inqfdims, check_dim

    70) ! !INTERFACE:
    71)   subroutine surfrd_get_globmask(filename, mask, ni, nj)
    72) !
    73) ! !DESCRIPTION:
    74) ! Read the surface dataset grid related information:
    75) ! This is the first routine called by clm_initialize 
    76) ! NO DOMAIN DECOMPOSITION  HAS BEEN SET YET
    77) !
    78) ! !USES:
    79)     use fileutils , only : getfil
    80) !
    81) ! !ARGUMENTS:
    82)     implicit none
    83)     character(len=*), intent(in)    :: filename  ! grid filename
    84)     integer         , pointer       :: mask(:)   ! grid mask 
    85)     integer         , intent(out)   :: ni, nj    ! global grid sizes
    86) !
    87) ! !CALLED FROM:
    88) ! subroutine surfrd in this module
    89) !
    90) ! !REVISION HISTORY:
    91) ! Created by Mariana Vertenstein
    92) !
    93) !
    94) ! !LOCAL VARIABLES:
    95) !EOP
    96)     logical :: isgrid2d
    97)     integer :: dimid,varid         ! netCDF id's
    98)     integer :: ns                  ! size of grid on file
    99)     integer :: n,i,j               ! index 
   100)     integer :: ier                 ! error status
   101)     type(file_desc_t)  :: ncid     ! netcdf id
   102)     type(var_desc_t)   :: vardesc  ! variable descriptor
   103)     character(len=256) :: varname  ! variable name
   104)     character(len=256) :: locfn    ! local file name
   105)     logical :: readvar             ! read variable in or not
   106)     integer , allocatable :: idata2d(:,:)
   107)     character(len=32) :: subname = 'surfrd_get_globmask' ! subroutine name
   108) !-----------------------------------------------------------------------
   109) 
   110)     if (filename == ' ') then
   111)        mask(:) = 1
   112)        RETURN
   113)     end if
   114) 
   115)     if (masterproc) then
   116)        if (filename == ' ') then
   117)           write(iulog,*) trim(subname),' ERROR: filename must be specified '
   118)           call endrun()
   119)        endif
   120)     end if
   121) 
   122)     call getfil( filename, locfn, 0 )
   123)     call ncd_pio_openfile (ncid, trim(locfn), 0)
   124) 
   125)     ! Determine dimensions and if grid file is 2d or 1d
   126) 
   127)     call ncd_inqfdims(ncid, isgrid2d, ni, nj, ns)
   128)     if (masterproc) then
   129)        write(iulog,*)'lat/lon grid flag (isgrid2d) is ',isgrid2d
   130)     end if
   131) 
   132)     allocate(mask(ns))
   133)     mask(:) = 1
   134) 
   135)     if (isgrid2d) then
   136)        allocate(idata2d(ni,nj))
   137)        idata2d(:,:) = 1	
   138)        call ncd_io(ncid=ncid, varname='LANDMASK', data=idata2d, flag='read', readvar=readvar)
   139)        if (.not. readvar) then
   140)           call ncd_io(ncid=ncid, varname='mask', data=idata2d, flag='read', readvar=readvar)
   141)        end if
   142)        if (readvar) then
   143)           do j = 1,nj
   144)           do i = 1,ni
   145)              n = (j-1)*ni + i	
   146)              mask(n) = idata2d(i,j)
   147)           enddo
   148)           enddo
   149)        end if
   150)        deallocate(idata2d)
   151)     else
   152)        call ncd_io(ncid=ncid, varname='LANDMASK', data=mask, flag='read', readvar=readvar)
   153)        if (.not. readvar) then
   154)           call ncd_io(ncid=ncid, varname='mask', data=mask, flag='read', readvar=readvar)
   155)        end if
   156)     end if
   157)     if (.not. readvar) call endrun( trim(subname)//' ERROR: landmask not on fatmlndfrc file' )
   158) 
   159)     call ncd_pio_closefile(ncid)
   160) 
   161)   end subroutine surfrd_get_globmask

   168) ! !INTERFACE:
   169)   subroutine surfrd_get_grid(ldomain, filename, glcfilename)
   170) !
   171) ! !DESCRIPTION:
   172) ! THIS IS CALLED AFTER THE DOMAIN DECOMPOSITION HAS BEEN CREATED
   173) ! Read the surface dataset grid related information:
   174) ! o real latitude  of grid cell (degrees)
   175) ! o real longitude of grid cell (degrees)
   176) !
   177) ! !USES:
   178)     use clm_varcon, only : spval, re
   179)     use domainMod , only : domain_type, domain_init, domain_clean, lon1d, lat1d
   180)     use decompMod , only : get_proc_bounds
   181)     use fileutils , only : getfil
   182) !
   183) ! !ARGUMENTS:
   184)     implicit none
   185)     type(domain_type),intent(inout) :: ldomain   ! domain to init
   186)     character(len=*) ,intent(in)    :: filename  ! grid filename
   187)     character(len=*) ,optional, intent(in) :: glcfilename ! glc mask filename
   188) !
   189) ! !CALLED FROM:
   190) ! subroutine surfrd in this module
   191) !
   192) ! !REVISION HISTORY:
   193) ! Created by Mariana Vertenstein
   194) !
   195) !
   196) ! !LOCAL VARIABLES:
   197) !EOP
   198)     type(file_desc_t) :: ncid               ! netcdf id
   199)     type(file_desc_t) :: ncidg              ! netCDF id for glcmask
   200)     type(var_desc_t)  :: vardesc            ! variable descriptor
   201)     integer :: beg                          ! local beg index
   202)     integer :: end                          ! local end index
   203)     integer :: ni,nj,ns                     ! size of grid on file
   204)     integer :: dimid,varid                  ! netCDF id's
   205)     integer :: start(1), count(1)           ! 1d lat/lon array sections
   206)     integer :: ier,ret                      ! error status
   207)     logical :: readvar                      ! true => variable is on input file 
   208)     logical :: isgrid2d                     ! true => file is 2d lat/lon
   209)     logical :: istype_domain                ! true => input file is of type domain
   210)     real(r8), allocatable :: rdata2d(:,:)   ! temporary
   211)     character(len=16) :: vname              ! temporary
   212)     character(len=256):: locfn              ! local file name
   213)     integer :: n                            ! indices
   214)     real(r8):: eps = 1.0e-12_r8             ! lat/lon error tolerance
   215)     character(len=32) :: subname = 'surfrd_get_grid'     ! subroutine name
   216) !-----------------------------------------------------------------------
   217) 
   218)     if (masterproc) then
   219)        if (filename == ' ') then
   220)           write(iulog,*) trim(subname),' ERROR: filename must be specified '
   221)           call endrun()
   222)        endif
   223)     end if
   224) 
   225)     call getfil( filename, locfn, 0 )
   226)     call ncd_pio_openfile (ncid, trim(locfn), 0)
   227) 
   228)     ! Determine dimensions
   229) 
   230)     call ncd_inqfdims(ncid, isgrid2d, ni, nj, ns)
   231) 
   232)     ! Determine isgrid2d flag for domain
   233) 
   234)     call get_proc_bounds(beg, end)
   235)     call domain_init(ldomain, isgrid2d=isgrid2d, ni=ni, nj=nj, nbeg=beg, nend=end)
   236)     ! Determine type of file - old style grid file or new style domain file
   237) 
   238)     call check_var(ncid=ncid, varname='LONGXY', vardesc=vardesc, readvar=readvar) 
   239)     if (readvar) istype_domain = .false.
   240) 
   241)     call check_var(ncid=ncid, varname='xc', vardesc=vardesc, readvar=readvar) 
   242)     if (readvar) istype_domain = .true.
   243) 
   244)     ! Read in area, lon, lat
   245) 
   246)     if (istype_domain) then
   247)        call ncd_io(ncid=ncid, varname= 'area', flag='read', data=ldomain%area, &
   248)             dim1name=grlnd, readvar=readvar)
   249)        ! convert from radians**2 to km**2
   250)        ldomain%area = ldomain%area * (re**2)
   251)        if (.not. readvar) call endrun( trim(subname)//' ERROR: area NOT on file' )
   252)        
   253)        call ncd_io(ncid=ncid, varname= 'xc', flag='read', data=ldomain%lonc, &
   254)             dim1name=grlnd, readvar=readvar)
   255)        if (.not. readvar) call endrun( trim(subname)//' ERROR: xc NOT on file' )
   256)        
   257)        call ncd_io(ncid=ncid, varname= 'yc', flag='read', data=ldomain%latc, &
   258)             dim1name=grlnd, readvar=readvar)
   259)        if (.not. readvar) call endrun( trim(subname)//' ERROR: yc NOT on file' )
   260)     else
   261)        call ncd_io(ncid=ncid, varname= 'AREA', flag='read', data=ldomain%area, &
   262)             dim1name=grlnd, readvar=readvar)
   263)        if (.not. readvar) call endrun( trim(subname)//' ERROR: AREA NOT on file' )
   264)        
   265)        call ncd_io(ncid=ncid, varname= 'LONGXY', flag='read', data=ldomain%lonc, &
   266)             dim1name=grlnd, readvar=readvar)
   267)        if (.not. readvar) call endrun( trim(subname)//' ERROR: LONGXY NOT on file' )
   268)        
   269)        call ncd_io(ncid=ncid, varname= 'LATIXY', flag='read', data=ldomain%latc, &
   270)             dim1name=grlnd, readvar=readvar)
   271)        if (.not. readvar) call endrun( trim(subname)//' ERROR: LATIXY NOT on file' )
   272)     end if
   273) 
   274)     if (isgrid2d) then
   275)        allocate(rdata2d(ni,nj), lon1d(ni), lat1d(nj))
   276)        if (istype_domain) then
   277)           vname = 'xc'
   278)        else
   279)           vname = 'LONGXY'
   280)        end if
   281)        call ncd_io(ncid=ncid, varname=trim(vname), data=rdata2d, flag='read', readvar=readvar)
   282)        lon1d(:) = rdata2d(:,1)
   283)        if (istype_domain) then
   284)           vname = 'yc'
   285)        else
   286)           vname = 'LATIXY'
   287)        end if
   288)        call ncd_io(ncid=ncid, varname=trim(vname), data=rdata2d, flag='read', readvar=readvar)
   289)        lat1d(:) = rdata2d(1,:)
   290)        deallocate(rdata2d)
   291)     end if
   292) 
   293)     ! Check lat limited to -90,90
   294) 
   295)     if (minval(ldomain%latc) < -90.0_r8 .or. &
   296)         maxval(ldomain%latc) >  90.0_r8) then
   297)        write(iulog,*) trim(subname),' WARNING: lat/lon min/max is ', &
   298)             minval(ldomain%latc),maxval(ldomain%latc)
   299)        ! call endrun( trim(subname)//' ERROR: lat is outside [-90,90]' )
   300)        ! write(iulog,*) trim(subname),' Limiting lat/lon to [-90/90] from ', &
   301)        !     minval(domain%latc),maxval(domain%latc)
   302)        ! where (ldomain%latc < -90.0_r8) ldomain%latc = -90.0_r8
   303)        ! where (ldomain%latc >  90.0_r8) ldomain%latc =  90.0_r8
   304)     endif
   305) 
   306)     call ncd_io(ncid=ncid, varname='LANDMASK', flag='read', data=ldomain%mask, &
   307)          dim1name=grlnd, readvar=readvar)
   308)     if (.not. readvar) then
   309)        call ncd_io(ncid=ncid, varname='mask', flag='read', data=ldomain%mask, &
   310)             dim1name=grlnd, readvar=readvar)
   311)        if (.not. readvar) then
   312)           call endrun( trim(subname)//' ERROR: LANDMASK NOT on fracdata file' )
   313)        end if
   314)     end if
   315) 
   316)     call ncd_io(ncid=ncid, varname='LANDFRAC', flag='read', data=ldomain%frac, &
   317)          dim1name=grlnd, readvar=readvar)
   318)     if (.not. readvar) then
   319)        call ncd_io(ncid=ncid, varname='frac', flag='read', data=ldomain%frac, &
   320)             dim1name=grlnd, readvar=readvar)
   321)        if (.not. readvar) then
   322)           call endrun( trim(subname)//' ERROR: LANDFRAC NOT on fracdata file' )
   323)        end if
   324)     end if
   325) 
   326)     call ncd_pio_closefile(ncid)
   327) 
   328)     if (present(glcfilename)) then
   329)        if (masterproc) then
   330)           if (glcfilename == ' ') then
   331)              write(iulog,*) trim(subname),' ERROR: glc filename must be specified '
   332)              call endrun()
   333)           endif
   334)        end if
   335)        call getfil( glcfilename, locfn, 0 )
   336)        call ncd_pio_openfile (ncidg, trim(locfn), 0)
   337) 
   338)        ldomain%glcmask(:) = 0
   339)        call ncd_io(ncid=ncidg, varname='GLCMASK', flag='read', data=ldomain%glcmask, &
   340)             dim1name=grlnd, readvar=readvar)
   341)        if (.not. readvar) call endrun( trim(subname)//' ERROR: GLCMASK NOT in file' )
   342) 
   343)        ! Make sure the glc mask is a subset of the land mask
   344)        do n = beg,end
   345)           if (ldomain%glcmask(n)==1 .and. ldomain%mask(n)==0) then
   346)              write(iulog,*)trim(subname),&
   347)                   'initialize1: landmask/glcmask mismatch'
   348)              write(iulog,*)trim(subname),&
   349)                   'glc requires input where landmask = 0, gridcell index', n
   350)              call endrun()
   351)           endif
   352)        enddo
   353)        call ncd_pio_closefile(ncidg)
   354)     endif   ! present(glcfilename)
   355) 
   356)   end subroutine surfrd_get_grid

   363) ! !INTERFACE:
   364)   subroutine surfrd_get_topo(domain,filename)
   365) !
   366) ! !DESCRIPTION:
   367) ! Read the topo dataset grid related information:
   368) ! Assume domain has already been initialized and read
   369) !
   370) ! !USES:
   371)     use domainMod , only : domain_type
   372)     use fileutils , only : getfil
   373) !
   374) ! !ARGUMENTS:
   375)     implicit none
   376)     type(domain_type),intent(inout) :: domain   ! domain to init
   377)     character(len=*) ,intent(in)    :: filename ! grid filename
   378) !
   379) ! !CALLED FROM:
   380) ! subroutine initialize
   381) !
   382) ! !REVISION HISTORY:
   383) ! Created by T Craig
   384) !
   385) !
   386) ! !LOCAL VARIABLES:
   387) !EOP
   388)     type(file_desc_t) :: ncid      ! netcdf file id
   389)     integer :: n                   ! indices
   390)     integer :: ni,nj,ns            ! size of grid on file
   391)     integer :: dimid,varid         ! netCDF id's
   392)     integer :: ier                 ! error status
   393)     real(r8):: eps = 1.0e-12_r8             ! lat/lon error tolerance
   394)     integer :: beg,end                      ! local beg,end indices
   395)     logical             :: isgrid2d         ! true => file is 2d lat/lon
   396)     real(r8),pointer    :: lonc(:),latc(:)  ! local lat/lon
   397)     character(len=256)  :: locfn            ! local file name
   398)     logical :: readvar                      ! is variable on file
   399)     character(len=32) :: subname = 'surfrd_get_topo'     ! subroutine name
   400) !-----------------------------------------------------------------------
   401) 
   402)     if (masterproc) then
   403)        if (filename == ' ') then
   404)           write(iulog,*) trim(subname),' ERROR: filename must be specified '
   405)           call endrun()
   406)        else
   407)           write(iulog,*) 'Attempting to read lnd topo from flndtopo ',trim(filename)
   408)        endif
   409)     end if
   410) 
   411)     call getfil( filename, locfn, 0 )
   412)     call ncd_pio_openfile (ncid, trim(locfn), 0)
   413)     call ncd_inqfdims(ncid, isgrid2d, ni, nj, ns)
   414) 
   415)     if (domain%ns /= ns) then
   416)        write(iulog,*) trim(subname),' ERROR: topo file mismatch ns',&
   417)             domain%ns,ns
   418)        call endrun()
   419)     endif
   420)     
   421)     beg = domain%nbeg
   422)     end = domain%nend
   423) 
   424)     allocate(latc(beg:end),lonc(beg:end))
   425) 
   426)     call ncd_io(ncid=ncid, varname='LONGXY', flag='read', data=lonc, &
   427)          dim1name=grlnd, readvar=readvar)
   428)     if (.not. readvar) call endrun( trim(subname)//' ERROR: LONGXY  NOT on topodata file' )
   429) 
   430)     call ncd_io(ncid=ncid, varname='LATIXY', flag='read', data=latc, &
   431)          dim1name=grlnd, readvar=readvar)
   432)     if (.not. readvar) call endrun( trim(subname)//' ERROR: LONGXY  NOT on topodata file' )
   433) 
   434)     do n = beg,end
   435)        if (abs(latc(n)-domain%latc(n)) > eps .or. &
   436)            abs(lonc(n)-domain%lonc(n)) > eps) then
   437)           write(iulog,*) trim(subname),' ERROR: topo file mismatch lat,lon',latc(n),&
   438)                domain%latc(n),lonc(n),domain%lonc(n),eps
   439)           call endrun()
   440)        endif
   441)     enddo
   442) 
   443)     call ncd_io(ncid=ncid, varname='TOPO', flag='read', data=domain%topo, &
   444)          dim1name=grlnd, readvar=readvar)
   445)     if (.not. readvar) call endrun( trim(subname)//' ERROR: LONGXY  NOT on topodata file' )
   446) 
   447)     deallocate(latc,lonc)
   448) 
   449)     call ncd_pio_closefile(ncid)
   450) 
   451)   end subroutine surfrd_get_topo

   456) ! !IROUTINE: surfrd_get_data

   458) ! !INTERFACE:
   459)   subroutine surfrd_get_data (ldomain, lfsurdat)
   460) !
   461) ! !DESCRIPTION:
   462) ! Read the surface dataset and create subgrid weights.
   463) ! The model's surface dataset recognizes 6 basic land cover types within a grid
   464) ! cell: lake, wetland, urban, glacier, glacier_mec and vegetated. The vegetated
   465) ! portion of the grid cell is comprised of up to [maxpatch_pft] PFTs. These
   466) ! subgrid patches are read in explicitly for each grid cell. This is in
   467) ! contrast to LSMv1, where the PFTs were built implicitly from biome types.
   468) !    o real latitude  of grid cell (degrees)
   469) !    o real longitude of grid cell (degrees)
   470) !    o integer surface type: 0 = ocean or 1 = land
   471) !    o integer soil color (1 to 20) for use with soil albedos
   472) !    o real soil texture, %sand, for thermal and hydraulic properties
   473) !    o real soil texture, %clay, for thermal and hydraulic properties
   474) !    o real % of cell covered by lake    for use as subgrid patch
   475) !    o real % of cell covered by wetland for use as subgrid patch
   476) !    o real % of cell that is urban      for use as subgrid patch
   477) !    o real % of cell that is glacier    for use as subgrid patch
   478) !    o real % of cell that is glacier_mec for use as subgrid patch
   479) !    o integer PFTs
   480) !    o real % abundance PFTs (as a percent of vegetated area)
   481) !
   482) ! !USES:
   483)     use clm_varctl  , only : allocate_all_vegpfts, create_crop_landunit
   484)     use pftvarcon   , only : noveg
   485)     use fileutils   , only : getfil
   486)     use domainMod   , only : domain_type, domain_init, domain_clean
   487) !
   488) ! !ARGUMENTS:
   489)     implicit none
   490)     type(domain_type),intent(in) :: ldomain     ! land domain associated with wtxy
   491)     character(len=*), intent(in) :: lfsurdat    ! surface dataset filename
   492) !
   493) ! !CALLED FROM:
   494) ! subroutine initialize in module initializeMod
   495) !
   496) ! !REVISION HISTORY:
   497) ! Created by Mariana Vertenstein, Sam Levis and Gordon Bonan
   498) !
   499) !
   500) ! !LOCAL VARIABLES:
   501) !EOP
   502)     type(var_desc_t)  :: vardesc              ! pio variable descriptor
   503)     type(domain_type) :: surfdata_domain      ! local domain associated with surface dataset
   504)     character(len=256):: locfn                ! local file name
   505)     integer           :: n                    ! loop indices
   506)     integer           :: ni,nj,ns             ! domain sizes
   507)     character(len=16) :: lon_var, lat_var     ! names of lat/lon on dataset
   508)     logical           :: readvar              ! true => variable is on dataset
   509)     real(r8)          :: rmaxlon,rmaxlat      ! local min/max vars
   510)     type(file_desc_t) :: ncid                 ! netcdf id
   511)     integer           :: begg,endg            ! beg,end gridcell indices
   512)     logical           :: istype_domain        ! true => input file is of type domain
   513)     logical           :: isgrid2d             ! true => intut grid is 2d 
   514)     character(len=32) :: subname = 'surfrd_get_data'    ! subroutine name
   515) !-----------------------------------------------------------------------
   516) 
   517)     if (masterproc) then
   518)        write(iulog,*) 'Attempting to read surface boundary data .....'
   519)        if (lfsurdat == ' ') then
   520)           write(iulog,*)'lfsurdat must be specified'; call endrun()
   521)        endif
   522)     endif
   523) 
   524)     call get_proc_bounds(begg,endg)
   525)     allocate(pctspec(begg:endg))
   526) 
   527)     vegxy(:,:) = noveg
   528)     wtxy(:,:)  = 0._r8
   529)     pctspec(:) = 0._r8
   530)     if (allocated(topoxy)) topoxy(:,:) = 0._r8
   531) 
   532)     ! Read surface data
   533) 
   534)     call getfil( lfsurdat, locfn, 0 )
   535)     call ncd_pio_openfile (ncid, trim(locfn), 0)
   536) 
   537)     ! Read in pft mask - this variable is only on the surface dataset - but not
   538)     ! on the domain dataset
   539) 
   540)     call ncd_io(ncid=ncid, varname= 'PFTDATA_MASK', flag='read', data=ldomain%pftm, &
   541)          dim1name=grlnd, readvar=readvar)
   542)     if (.not. readvar) call endrun( trim(subname)//' ERROR: pftm NOT on surface dataset' )
   543) 
   544)     ! Check if fsurdat grid is "close" to fatmlndfrc grid, exit if lats/lon > 0.001
   545) 
   546)     call check_var(ncid=ncid, varname='xc', vardesc=vardesc, readvar=readvar) 
   547)     if (readvar) then
   548)        istype_domain = .true.
   549)     else
   550)        call check_var(ncid=ncid, varname='LONGXY', vardesc=vardesc, readvar=readvar) 
   551)        if (readvar) then
   552)           istype_domain = .false.
   553)        else
   554)           call endrun( trim(subname)//' ERROR: unknown domain type')
   555)        end if
   556)     end if
   557)     if (istype_domain) then
   558)        lon_var  = 'xc'
   559)        lat_var  = 'yc'
   560)     else
   561)        lon_var  = 'LONGXY'
   562)        lat_var  = 'LATIXY'
   563)     end if
   564)     if ( masterproc )then
   565)        write(iulog,*) trim(subname),' lon_var = ',trim(lon_var),' lat_var =',trim(lat_var)
   566)     end if
   567) 
   568)     call ncd_inqfdims(ncid, isgrid2d, ni, nj, ns)
   569)     call domain_init(surfdata_domain, isgrid2d, ni, nj, begg, endg, clmlevel=grlnd)
   570) 
   571)     call ncd_io(ncid=ncid, varname=lon_var, flag='read', data=surfdata_domain%lonc, &
   572)          dim1name=grlnd, readvar=readvar)
   573)     if (.not. readvar) call endrun( trim(subname)//' ERROR: lon var NOT on surface dataset' )
   574) 
   575)     call ncd_io(ncid=ncid, varname=lat_var, flag='read', data=surfdata_domain%latc, &
   576)          dim1name=grlnd, readvar=readvar)
   577)     if (.not. readvar) call endrun( trim(subname)//' ERROR: lat var NOT on surface dataset' )
   578) 
   579)     rmaxlon = 0.0_r8
   580)     rmaxlat = 0.0_r8
   581)     do n = begg,endg
   582)        if (ldomain%lonc(n)-surfdata_domain%lonc(n) > 300.) then
   583)           rmaxlon = max(rmaxlon,abs(ldomain%lonc(n)-surfdata_domain%lonc(n)-360._r8))
   584)        elseif (ldomain%lonc(n)-surfdata_domain%lonc(n) < -300.) then
   585)           rmaxlon = max(rmaxlon,abs(ldomain%lonc(n)-surfdata_domain%lonc(n)+360._r8))
   586)        else
   587)           rmaxlon = max(rmaxlon,abs(ldomain%lonc(n)-surfdata_domain%lonc(n)))
   588)        endif
   589)        rmaxlat = max(rmaxlat,abs(ldomain%latc(n)-surfdata_domain%latc(n)))
   590)     enddo
   591)     if (rmaxlon > 0.001_r8 .or. rmaxlat > 0.001_r8) then
   592)        write(iulog,*) trim(subname)//': surfdata/fatmgrid lon/lat mismatch error',&
   593)             rmaxlon,rmaxlat
   594)        call endrun(trim(subname))
   595)     end if
   596)     call domain_clean(surfdata_domain)
   597) 
   598)     ! Obtain special landunit info
   599) 
   600)     call surfrd_wtxy_special(ncid, ldomain%ns)
   601) 
   602)     ! Obtain vegetated landunit info
   603) 
   604)     if (use_cndv) then
   605)        if (create_crop_landunit) then ! CNDV means allocate_all_vegpfts = .true.
   606)           call surfrd_wtxy_veg_all(ncid, ldomain%ns, ldomain%pftm)
   607)        end if
   608)        call surfrd_wtxy_veg_dgvm()
   609)     else
   610)        if (allocate_all_vegpfts) then
   611)           call surfrd_wtxy_veg_all(ncid, ldomain%ns, ldomain%pftm)
   612)        else
   613)           call endrun (trim(subname) // 'only allocate_all_vegpfts is supported')
   614)        end if
   615)     end if
   616) 
   617)     call ncd_pio_closefile(ncid)
   618) 
   619)     if ( masterproc )then
   620)        write(iulog,*) 'Successfully read surface boundary data'
   621)        write(iulog,*)
   622)     end if
   623) 
   624)   end subroutine surfrd_get_data

   629) ! !IROUTINE: surfrd_wtxy_special 
   630) !
   631) ! !INTERFACE:
   632)   subroutine surfrd_wtxy_special(ncid, ns)
   633) !
   634) ! !DESCRIPTION:
   635) ! Determine weight with respect to gridcell of all special "pfts" as well
   636) ! as soil color and percent sand and clay
   637) !
   638) ! !USES:
   639)     use pftvarcon     , only : noveg
   640)     use UrbanInputMod , only : urbinp
   641)     use clm_varpar    , only : maxpatch_glcmec
   642) !
   643) ! !ARGUMENTS:
   644)     implicit none
   645)     type(file_desc_t), intent(inout) :: ncid   ! netcdf id
   646)     integer          , intent(in)    :: ns     ! domain size
   647) !
   648) ! !CALLED FROM:
   649) ! subroutine surfrd in this module
   650) !
   651) ! !REVISION HISTORY:
   652) ! Created by Mariana Vertenstein, Sam Levis and Gordon Bonan
   653) !
   654) !
   655) ! !LOCAL VARIABLES:
   656) !EOP
   657)     integer  :: n,nl,nurb,g                ! indices
   658)     integer  :: begg,endg                  ! gcell beg/end
   659)     integer  :: dimid,varid                ! netCDF id's
   660)     real(r8) :: nlevsoidata(nlevsoi)
   661)     logical  :: found                      ! temporary for error check
   662)     integer  :: nindx                      ! temporary for error check
   663)     integer  :: ier                        ! error status
   664)     integer  :: nlev                       ! level
   665)     integer  :: npatch
   666)     logical  :: readvar
   667)     real(r8),pointer :: pctgla(:)      ! percent of grid cell is glacier
   668)     real(r8),pointer :: pctlak(:)      ! percent of grid cell is lake
   669)     real(r8),pointer :: pctwet(:)      ! percent of grid cell is wetland
   670)     real(r8),pointer :: pcturb(:)      ! percent of grid cell is urbanized
   671)     real(r8),pointer :: pctglc_mec(:,:)   ! percent of grid cell is glacier_mec (in each elev class)
   672)     real(r8),pointer :: pctglc_mec_tot(:) ! percent of grid cell is glacier (sum over classes)
   673)     real(r8),pointer :: topoglc_mec(:,:)  ! surface elevation in each elev class
   674)     character(len=32) :: subname = 'surfrd_wtxy_special'  ! subroutine name
   675)     real(r8) closelat,closelon
   676) !!-----------------------------------------------------------------------
   677) 
   678)     call get_proc_bounds(begg,endg)
   679) 
   680)     allocate(pctgla(begg:endg),pctlak(begg:endg))
   681)     allocate(pctwet(begg:endg),pcturb(begg:endg))
   682)     if (create_glacier_mec_landunit) then
   683)        allocate(pctglc_mec(begg:endg,maxpatch_glcmec))
   684)        allocate(pctglc_mec_tot(begg:endg))
   685)        allocate(topoglc_mec(begg:endg,maxpatch_glcmec))
   686)        allocate(glc_topomax(0:maxpatch_glcmec))
   687)     endif
   688) 
   689)     call check_dim(ncid, 'nlevsoi', nlevsoi)
   690) 
   691)        ! Obtain non-grid surface properties of surface dataset other than percent pft
   692) 
   693)     call ncd_io(ncid=ncid, varname='PCT_WETLAND', flag='read', data=pctwet, &
   694)          dim1name=grlnd, readvar=readvar)
   695)     if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_WETLAND  NOT on surfdata file' )
   696) 
   697)     call ncd_io(ncid=ncid, varname='PCT_LAKE'   , flag='read', data=pctlak, &
   698)          dim1name=grlnd, readvar=readvar)
   699)     if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_LAKE NOT on surfdata file' )
   700) 
   701)     call ncd_io(ncid=ncid, varname='PCT_GLACIER', flag='read', data=pctgla, &
   702)          dim1name=grlnd, readvar=readvar)
   703)     if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_GLACIER NOT on surfdata file' )
   704) 
   705)     call ncd_io(ncid=ncid, varname='PCT_URBAN'  , flag='read', data=pcturb, &
   706)          dim1name=grlnd, readvar=readvar)
   707)     if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_URBAN NOT on surfdata file' )
   708) 
   709)     if (create_glacier_mec_landunit) then          ! call ncd_io_gs_int2d
   710) 
   711)        call check_dim(ncid, 'nglcec',   maxpatch_glcmec   )
   712)        call check_dim(ncid, 'nglcecp1', maxpatch_glcmec+1 )
   713) 
   714)        call ncd_io(ncid=ncid, varname='GLC_MEC', flag='read', data=glc_topomax, &
   715)             readvar=readvar)
   716)        if (.not. readvar) call endrun( trim(subname)//'ERROR: GLC_MEC was NOT on the input surfdata file' )
   717) 
   718)        call ncd_io(ncid=ncid, varname='PCT_GLC_MEC', flag='read', data=pctglc_mec, &
   719)             dim1name=grlnd, readvar=readvar)
   720)        if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_GLC_MEC NOT on surfdata file' )
   721) 
   722)        call ncd_io(ncid=ncid, varname='TOPO_GLC_MEC',  flag='read', data=topoglc_mec, &
   723)             dim1name=grlnd, readvar=readvar)
   724)        if (.not. readvar) call endrun( trim(subname)//' ERROR: TOPO_GLC_MEC NOT on surfdata file' )
   725) 
   726)        pctglc_mec_tot(:) = 0._r8
   727)        do n = 1, maxpatch_glcmec
   728)           do nl = begg,endg
   729)              pctglc_mec_tot(nl) = pctglc_mec_tot(nl) + pctglc_mec(nl,n)
   730)           enddo
   731)        enddo
   732) 
   733)        ! Make sure sum of pctglc_mec = pctgla (approximately), then correct pctglc_mec so
   734)        ! that its sum more exactly equals pctgla, then zero out pctgla
   735)        !
   736)        ! WJS (9-28-12): The reason for the correction piece of this is: in the surface
   737)        ! dataset, pctgla underwent various minor corrections that make it the trusted
   738)        ! value (as opposed to sum(pctglc_mec). sum(pctglc_mec) can differ from pctgla by
   739)        ! single precision roundoff. This difference can cause problems later (e.g., in the
   740)        ! consistency check in pftdynMod), so we do this correction here. It might be
   741)        ! better to do this correction in mksurfdata_map, but because of time constraints,
   742)        ! which make me unable to recreate surface datasets, I have to do it here instead
   743)        ! (and there are arguments for putting it here anyway).
   744) 
   745)        do nl = begg,endg
   746)           ! We need to use a fairly high threshold for equality (2.0e-5) because pctgla
   747)           ! and pctglc_mec are computed using single precision inputs. Note that this
   748)           ! threshold agrees with the threshold in the error checks in mkglcmecMod:
   749)           ! mkglcmec in mksurfdata_map. 
   750)           if (abs(pctgla(nl) - pctglc_mec_tot(nl)) > 2.0e-5) then
   751)              write(iulog,*) ' '
   752)              write(iulog,*) 'surfrd error: pctgla not equal to sum of pctglc_mec for nl=', nl
   753)              write(iulog,*) 'pctgla =', pctgla(nl)
   754)              write(iulog,*) 'pctglc_mec_tot =', pctglc_mec_tot(nl)
   755)              call endrun()
   756)           endif
   757) 
   758)           ! Correct the remaining minor differences in pctglc_mec so sum more exactly
   759)           ! equals pctglc (see notes above for justification)
   760)           if (pctglc_mec_tot(nl) > 0.0_r8) then
   761)              pctglc_mec(nl,:) = pctglc_mec(nl,:) * pctgla(nl)/pctglc_mec_tot(nl)
   762)           end if
   763) 
   764)           ! Now recompute pctglc_mec_tot, and confirm that we are now much closer to pctgla
   765)           pctglc_mec_tot(nl) = 0._r8
   766)           do n = 1, maxpatch_glcmec
   767)              pctglc_mec_tot(nl) = pctglc_mec_tot(nl) + pctglc_mec(nl,n)
   768)           end do
   769) 
   770)           if (abs(pctgla(nl) - pctglc_mec_tot(nl)) > 1.0e-13) then
   771)              write(iulog,*) ' '
   772)              write(iulog,*) 'surfrd error: after correction, pctgla not equal to sum of pctglc_mec for nl=', nl
   773)              write(iulog,*) 'pctgla =', pctgla(nl)
   774)              write(iulog,*) 'pctglc_mec_tot =', pctglc_mec_tot(nl)
   775)              call endrun()
   776)           endif
   777) 
   778)           ! Finally, zero out pctgla
   779)           pctgla(nl) = 0._r8
   780)        enddo
   781) 
   782)        ! If pctglc_mec_tot is very close to 100%, round to 100%
   783) 
   784)        do nl = begg,endg
   785)           ! The inequality here ( <= 2.0e-5 ) is designed to be the complement of the
   786)           ! above check that makes sure pctglc_mec_tot is close to pctgla: so if pctglc=
   787)           ! 100 (exactly), then exactly one of these conditionals will be triggered.
   788)           ! Update 9-28-12: Now that there is a rescaling of pctglc_mec to bring it more
   789)           ! in line with pctgla, we could probably decrease this tolerance again (the
   790)           ! point about exactly one of these conditionals being triggered no longer holds)
   791)           ! - or perhaps even get rid of this whole block of code. But I'm keeping this as
   792)           ! is for now because that's how I tested it, and I don't think it will hurt
   793)           ! anything to use this larger tolerance.
   794)           if (abs(pctglc_mec_tot(nl) - 100._r8) <= 2.0e-5) then
   795)              pctglc_mec(nl,:) = pctglc_mec(nl,:) * 100._r8 / pctglc_mec_tot(nl)
   796)              pctglc_mec_tot(nl) = 100._r8
   797)           endif
   798)        enddo
   799) 
   800)        pctspec = pctwet + pctlak + pcturb + pctglc_mec_tot
   801) 
   802)        if ( masterproc ) write(iulog,*) '   elevation limits = ', glc_topomax
   803) 
   804)     else
   805) 
   806)        pctspec = pctwet + pctlak + pcturb + pctgla
   807)  
   808)     endif
   809) 
   810)     ! Error check: glacier, lake, wetland, urban sum must be less than 100
   811) 
   812)     found = .false.
   813)     do nl = begg,endg
   814)        if (pctspec(nl) > 100._r8+1.e-04_r8) then
   815)           found = .true.
   816)           nindx = nl
   817)           exit
   818)        end if
   819)        if (found) exit
   820)     end do
   821)     if ( found ) then
   822)        write(iulog,*)'surfrd error: PFT cover>100 for nl=',nindx
   823)        call endrun()
   824)     end if
   825) 
   826)     ! Determine veg and wtxy for special landunits
   827) 
   828)     do nl = begg,endg
   829) 
   830)        vegxy(nl,npatch_lake)   = noveg
   831)        wtxy(nl,npatch_lake)    = pctlak(nl)/100._r8
   832) 
   833)        vegxy(nl,npatch_wet)    = noveg
   834)        wtxy(nl,npatch_wet)     = pctwet(nl)/100._r8
   835) 
   836)        vegxy(nl,npatch_glacier)= noveg
   837)        wtxy(nl,npatch_glacier) = pctgla(nl)/100._r8
   838) 
   839)        ! Initialize urban weights
   840) 
   841)        do nurb = npatch_urban, npatch_lake-1 
   842)           vegxy(nl,nurb) = noveg
   843)           wtxy(nl,nurb)  = pcturb(nl) / 100._r8
   844)        end do
   845)        if ( pcturb(nl) > 0.0_r8 )then
   846)           wtxy(nl,npatch_urban)   = wtxy(nl,npatch_urban)*urbinp%wtlunit_roof(nl)
   847)           wtxy(nl,npatch_urban+1) = wtxy(nl,npatch_urban+1)*(1 - urbinp%wtlunit_roof(nl))/3
   848)           wtxy(nl,npatch_urban+2) = wtxy(nl,npatch_urban+2)*(1 - urbinp%wtlunit_roof(nl))/3
   849)           wtxy(nl,npatch_urban+3) = wtxy(nl,npatch_urban+3)*(1 - urbinp%wtlunit_roof(nl))/3 * (1.-urbinp%wtroad_perv(nl))
   850)           wtxy(nl,npatch_urban+4) = wtxy(nl,npatch_urban+4)*(1 - urbinp%wtlunit_roof(nl))/3 * urbinp%wtroad_perv(nl)
   851)        end if
   852) 
   853)     end do
   854) 
   855)     ! Check to make sure we have valid urban data for each urban patch
   856) 
   857)     found = .false.
   858)     do nl = begg,endg
   859)        if ( pcturb(nl) > 0.0_r8 )then
   860)          if (urbinp%canyon_hwr(nl)            .le. 0._r8 .or. &
   861)              urbinp%em_improad(nl)            .le. 0._r8 .or. &
   862)              urbinp%em_perroad(nl)            .le. 0._r8 .or. &
   863)              urbinp%em_roof(nl)               .le. 0._r8 .or. &
   864)              urbinp%em_wall(nl)               .le. 0._r8 .or. &
   865)              urbinp%ht_roof(nl)               .le. 0._r8 .or. &
   866)              urbinp%thick_roof(nl)            .le. 0._r8 .or. &
   867)              urbinp%thick_wall(nl)            .le. 0._r8 .or. &
   868)              urbinp%t_building_max(nl)        .le. 0._r8 .or. &
   869)              urbinp%t_building_min(nl)        .le. 0._r8 .or. &
   870)              urbinp%wind_hgt_canyon(nl)       .le. 0._r8 .or. &
   871)              urbinp%wtlunit_roof(nl)          .le. 0._r8 .or. &
   872)              urbinp%wtroad_perv(nl)           .le. 0._r8 .or. &
   873)              any(urbinp%alb_improad_dir(nl,:) .le. 0._r8) .or. &
   874)              any(urbinp%alb_improad_dif(nl,:) .le. 0._r8) .or. &
   875)              any(urbinp%alb_perroad_dir(nl,:) .le. 0._r8) .or. &
   876)              any(urbinp%alb_perroad_dif(nl,:) .le. 0._r8) .or. &
   877)              any(urbinp%alb_roof_dir(nl,:)    .le. 0._r8) .or. &
   878)              any(urbinp%alb_roof_dif(nl,:)    .le. 0._r8) .or. &
   879)              any(urbinp%alb_wall_dir(nl,:)    .le. 0._r8) .or. &
   880)              any(urbinp%alb_wall_dif(nl,:)    .le. 0._r8) .or. &
   881)              any(urbinp%tk_roof(nl,:)         .le. 0._r8) .or. &
   882)              any(urbinp%tk_wall(nl,:)         .le. 0._r8) .or. &
   883)              any(urbinp%cv_roof(nl,:)         .le. 0._r8) .or. &
   884)              any(urbinp%cv_wall(nl,:)         .le. 0._r8)) then
   885)             found = .true.
   886)             nindx = nl
   887)             exit
   888)          else
   889)             if (urbinp%nlev_improad(nl) .gt. 0) then
   890)                nlev = urbinp%nlev_improad(nl)
   891)                if (any(urbinp%tk_improad(nl,1:nlev) .le. 0._r8) .or. &
   892)                    any(urbinp%cv_improad(nl,1:nlev) .le. 0._r8)) then
   893)                   found = .true.
   894)                   nindx = nl
   895)                   exit
   896)                end if
   897)             end if
   898)          end if
   899)          if (found) exit
   900)        end if
   901)     end do
   902)     if ( found ) then
   903)        write(iulog,*)'surfrd error: no valid urban data for nl=',nindx
   904)        write(iulog,*)'canyon_hwr:      ',urbinp%canyon_hwr(nindx)
   905)        write(iulog,*)'em_improad:      ',urbinp%em_improad(nindx)
   906)        write(iulog,*)'em_perroad:      ',urbinp%em_perroad(nindx)
   907)        write(iulog,*)'em_roof:         ',urbinp%em_roof(nindx)
   908)        write(iulog,*)'em_wall:         ',urbinp%em_wall(nindx)
