   220) subroutine rad_cnst_readnl(nlfile)
   221) 
   222)    ! Read rad_cnst_nl namelist group.  Parse input.
   223) 
   224)    use namelist_utils,  only: find_group_name
   225)    use units,           only: getunit, freeunit
   226)    use mpishorthand
   227) 
   228)    character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
   229) 
   230)    ! Local variables
   231)    integer :: unitn, ierr, i
   232)    character(len=2) :: suffix
   233)    character(len=1), pointer   :: ctype(:)
   234)    character(len=*), parameter :: subname = 'rad_cnst_readnl'
   235) 
   236)    namelist /rad_cnst_nl/ mode_defs,     &
   237)                           rad_climate,   &
   238)                           rad_diag_1,    &
   239)                           rad_diag_2,    &
   240)                           rad_diag_3,    &
   241)                           rad_diag_4,    &
   242)                           rad_diag_5,    &
   243)                           rad_diag_6,    &
   244)                           rad_diag_7,    &
   245)                           rad_diag_8,    &
   246)                           rad_diag_9,    &
   247)                           rad_diag_10,   &
   248)                           iceopticsfile, &
   249)                           liqopticsfile, &
   250)                           icecldoptics,  &
   251)                           liqcldoptics,  &
   252)                           oldcldoptics
   253) 
   254)    !-----------------------------------------------------------------------------
   255) 
   256)    if (masterproc) then
   257)       unitn = getunit()
   258)       open( unitn, file=trim(nlfile), status='old' )
   259)       call find_group_name(unitn, 'rad_cnst_nl', status=ierr)
   260)       if (ierr == 0) then
   261)          read(unitn, rad_cnst_nl, iostat=ierr)
   262)          if (ierr /= 0) then
   263)             call endrun(subname // ':: ERROR reading namelist')
   264)          end if
   265)       end if
   266)       close(unitn)
   267)       call freeunit(unitn)
   268)    end if
   269) 
   270) #ifdef SPMD
   271)    ! Broadcast namelist variables
   272)    call mpibcast (mode_defs,     len(mode_defs(1))*n_mode_str,     mpichar, 0, mpicom)
   273)    call mpibcast (rad_climate,   len(rad_climate(1))*n_rad_cnst,   mpichar, 0, mpicom)
   274)    call mpibcast (rad_diag_1,    len(rad_diag_1(1))*n_rad_cnst,    mpichar, 0, mpicom)
   275)    call mpibcast (rad_diag_2,    len(rad_diag_2(1))*n_rad_cnst,    mpichar, 0, mpicom)
   276)    call mpibcast (rad_diag_3,    len(rad_diag_3(1))*n_rad_cnst,    mpichar, 0, mpicom)
   277)    call mpibcast (rad_diag_4,    len(rad_diag_4(1))*n_rad_cnst,    mpichar, 0, mpicom)
   278)    call mpibcast (rad_diag_5,    len(rad_diag_5(1))*n_rad_cnst,    mpichar, 0, mpicom)
   279)    call mpibcast (rad_diag_6,    len(rad_diag_6(1))*n_rad_cnst,    mpichar, 0, mpicom)
   280)    call mpibcast (rad_diag_7,    len(rad_diag_7(1))*n_rad_cnst,    mpichar, 0, mpicom)
   281)    call mpibcast (rad_diag_8,    len(rad_diag_8(1))*n_rad_cnst,    mpichar, 0, mpicom)
   282)    call mpibcast (rad_diag_9,    len(rad_diag_9(1))*n_rad_cnst,    mpichar, 0, mpicom)
   283)    call mpibcast (rad_diag_10,   len(rad_diag_10(1))*n_rad_cnst,   mpichar, 0, mpicom)
   284)    call mpibcast (iceopticsfile, len(iceopticsfile),               mpichar, 0, mpicom)
   285)    call mpibcast (liqopticsfile, len(liqopticsfile),               mpichar, 0, mpicom)
   286)    call mpibcast (liqcldoptics,  len(liqcldoptics),                mpichar, 0, mpicom)
   287)    call mpibcast (icecldoptics,  len(icecldoptics),                mpichar, 0, mpicom)
   288)    call mpibcast (oldcldoptics,  1,                                mpilog , 0, mpicom)
   289) #endif
   290) 
   291)    ! Parse the namelist input strings
   292) 
   293)    ! Mode definition stings
   294)    call parse_mode_defs(mode_defs, modes)
   295)    
   296)    ! Lists of externally mixed entities for climate and diagnostic calculations
   297)    do i = 0,N_DIAG
   298)       select case (i)
   299)       case(0)
   300)          call parse_rad_specifier(rad_climate, namelist(i))
   301)       case (1)
   302)          call parse_rad_specifier(rad_diag_1, namelist(i))
   303)       case (2)
   304)          call parse_rad_specifier(rad_diag_2, namelist(i))
   305)       case (3)
   306)          call parse_rad_specifier(rad_diag_3, namelist(i))
   307)       case (4)
   308)          call parse_rad_specifier(rad_diag_4, namelist(i))
   309)       case (5)
   310)          call parse_rad_specifier(rad_diag_5, namelist(i))
   311)       case (6)
   312)          call parse_rad_specifier(rad_diag_6, namelist(i))
   313)       case (7)
   314)          call parse_rad_specifier(rad_diag_7, namelist(i))
   315)       case (8)
   316)          call parse_rad_specifier(rad_diag_8, namelist(i))
   317)       case (9)
   318)          call parse_rad_specifier(rad_diag_9, namelist(i))
   319)       case (10)
   320)          call parse_rad_specifier(rad_diag_10, namelist(i))
   321)       end select
   322)    enddo
   323) 
   324)    ! were there any constituents specified for the nth diagnostic call?
   325)    ! if so, radiation will make a call with those consituents
   326)    active_calls(:) = (namelist(:)%ncnst > 0)
   327)    	
   328)    ! Initialize the gas and aerosol lists with the information from the
   329)    ! namelist.  This is done here so that this information is available via
   330)    ! the query functions at the time when the register methods are called.
   331) 
   332)    ! Set the list_id fields which distinquish the climate and diagnostic lists
   333)    do i = 0, N_DIAG
   334)       if (active_calls(i)) then
   335)          if (i > 0) then
   336)             write(suffix, fmt = '(i2.2)') i
   337)          else
   338)             suffix='  '
   339)          end if
   340)          aerosollist(i)%list_id = suffix
   341)          gaslist(i)%list_id     = suffix
   342)          ma_list(i)%list_id     = suffix
   343)       end if
   344)    end do
   345) 
   346)    ! Create a list of the unique set of filenames containing property data
   347) 
   348)    ! Start with the bulk aerosol species in the climate/diagnostic lists.
   349)    ! The physprop_accum_unique_files routine has the side effect of returning the number
   350)    ! of bulk aerosols in each list (they're identified by type='A').
   351)    do i = 0, N_DIAG
   352)       if (active_calls(i)) then
   353)          call physprop_accum_unique_files(namelist(i)%radname, namelist(i)%type)
   354)       endif
   355)    enddo
   356) 
   357)    ! Add physprop files for the species from the mode definitions.
   358)    do i = 1, modes%nmodes
   359)       allocate(ctype(modes%comps(i)%nspec))
   360)       ctype = 'A'
   361)       call physprop_accum_unique_files(modes%comps(i)%props, ctype)
   362)       deallocate(ctype)
   363)    end do
   364) 
   365)    ! Initialize the gas, bulk aerosol, and modal aerosol lists.  This step splits the
   366)    ! input climate/diagnostic lists into the corresponding gas, bulk and modal aerosol
   367)    ! lists.
   368)    if (masterproc) write(iulog,*) nl//subname//': Radiation constituent lists:'
   369)    do i = 0, N_DIAG
   370)       if (active_calls(i)) then
   371)          call list_init1(namelist(i), gaslist(i), aerosollist(i), ma_list(i))
   372) 
   373)          if (masterproc .and. verbose) then
   374)             call print_lists(gaslist(i), aerosollist(i), ma_list(i))
   375)          end if
   376) 
   377)       end if
   378)    end do
   379) 
   380)    if (masterproc .and. verbose) call print_modes(modes)
   381) 
   382) end subroutine rad_cnst_readnl

   386) subroutine rad_cnst_init()
   387) 
   388)    ! The initialization of the gas and aerosol lists is finished by
   389)    ! 1) read the physprop files
   390)    ! 2) find the index of each constituent in the constituent or physics buffer arrays
   391)    ! 3) find the index of the aerosol constituents used to access its properties from the
   392)    !    physprop module.
   393) 
   394)    integer :: i
   395)    integer :: num_aerosols
   396)    logical, parameter :: stricttest = .true.
   397)    character(len=*), parameter :: subname = 'rad_cnst_init'
   398)    !-----------------------------------------------------------------------------
   399) 
   400)    ! memory to point to if zero value requested
   401)    allocate(zero_cols(pcols,pver))
   402)    zero_cols = 0._r8
   403) 
   404)    ! Allocate storage for the physical properties of each aerosol; read properties from
   405)    ! the data files.
   406)    call physprop_init()
   407) 
   408)    ! Start checking that specified radiative constituents are present in the constituent
   409)    ! or physics buffer arrays.
   410)    if (masterproc) write(iulog,*) nl//subname//': checking for radiative constituents'
   411) 
   412)    ! Finish initializing the mode definitions.
   413)    call init_mode_comps(modes)
   414) 
   415)    ! Finish initializing the gas, bulk aerosol, and mode lists.
   416)    do i = 0, N_DIAG
   417)       if (active_calls(i)) then
   418)          call list_init2(gaslist(i), aerosollist(i), ma_list(i))
   419)       end if
   420)    end do
   421) 
   422)    ! Check that all gases supported by the radiative transfer code have been specified.
   423)    if (stricttest) then
   424)       do i = 1, nradgas
   425)          if (gaslist(0)%gas(i)%source .eq. 'Z' ) then
   426)             call endrun(subname//': list of radiative gasses must include all radiation gasses for the climate specication')
   427)          endif
   428)       enddo
   429)    endif
   430) 
   431)    ! Initialize history output of climate diagnostic quantities
   432)    call rad_gas_diag_init(gaslist(0))
   433)    call rad_aer_diag_init(aerosollist(0))
   434) 
   435) 
   436) end subroutine rad_cnst_init

   440) subroutine rad_cnst_get_gas(list_idx, gasname, state, pbuf, mmr)
   441) 
   442)    ! Return pointer to mass mixing ratio for the gas from the specified
   443)    ! climate or diagnostic list.
   444) 
   445)    ! Arguments
   446)    integer,                     intent(in) :: list_idx    ! index of the climate or a diagnostic list
   447)    character(len=*),            intent(in) :: gasname
   448)    type(physics_state), target, intent(in) :: state
   449)    type(physics_buffer_desc),   pointer    :: pbuf(:)
   450)    real(r8),                    pointer    :: mmr(:,:)
   451) 
   452)    ! Local variables
   453)    integer :: lchnk
   454)    integer :: igas
   455)    integer :: idx
   456)    character(len=1) :: source
   457)    type(gaslist_t), pointer :: list
   458)    character(len=*), parameter :: subname = 'rad_cnst_get_gas'
   459)    !-----------------------------------------------------------------------------
   460) 
   461)    if (list_idx >= 0 .and. list_idx <= N_DIAG) then
   462)       list => gaslist(list_idx)
   463)    else
   464)       write(iulog,*) subname//': list_idx =', list_idx
   465)       call endrun(subname//': list_idx out of bounds')
   466)    endif
   467)       
   468)    lchnk = state%lchnk
   469) 
   470)    ! Get index of gas in internal arrays.  rad_gas_index will abort if the 
   471)    ! specified gasname is not recognized by the radiative transfer code.
   472)    igas = rad_gas_index(trim(gasname))

   776) subroutine rad_cnst_out(list_idx, state, pbuf)
   777) 
   778)    ! Output the mass per layer, and total column burdens for gas and aerosol
   779)    ! constituents in either the climate or diagnostic lists
   780) 
   781)    ! Arguments
   782)    integer,                     intent(in) :: list_idx
   783)    type(physics_state), target, intent(in) :: state
   784)    type(physics_buffer_desc), pointer      :: pbuf(:)
   785) 
   786) 
   787)    ! Local variables
   788)    integer :: i, naer, ngas, lchnk, ncol
   789)    integer :: idx
   790)    character(len=1)  :: source
   791)    character(len=32) :: name, cbname
   792)    real(r8)          :: mass(pcols,pver)
   793)    real(r8)          :: cb(pcols)
   794)    real(r8), pointer :: mmr(:,:)
   795)    type(aerlist_t), pointer :: aerlist
   796)    type(gaslist_t), pointer :: g_list
   797)    character(len=*), parameter :: subname = 'rad_cnst_out'
   798)    !-----------------------------------------------------------------------------
   799) 
   800)    lchnk = state%lchnk
   801)    ncol  = state%ncol
   802) 
   803)    ! Associate pointer with requested aerosol list
   804)    if (list_idx >= 0 .and. list_idx <= N_DIAG) then
   805)       aerlist => aerosollist(list_idx)
   806)    else
   807)       write(iulog,*) subname//': list_idx = ', list_idx
   808)       call endrun(subname//': list_idx out of range')
   809)    endif
   810) 
   811)    naer = aerlist%numaerosols
   812)    do i = 1, naer
   813) 
   814)       source = aerlist%aer(i)%source
   815)       idx    = aerlist%aer(i)%idx
   816)       name   = aerlist%aer(i)%mass_name
   817)       ! construct name for column burden field by replacing the 'm_' prefix by 'cb_'
   818)       cbname = 'cb_' // name(3:len_trim(name))
   819) 
   820)       select case( source )
   821)       case ('A')
   822)          mmr => state%q(:,:,idx)
   823)       case ('N')
   824)          call pbuf_get_field(pbuf, idx, mmr)
   825)       end select
   826) 
   827)       mass(:ncol,:) = mmr(:ncol,:) * state%pdeldry(:ncol,:) * rga
   828)       call outfld(trim(name), mass, pcols, lchnk)
   829) 
   830)       cb(:ncol) = sum(mass(:ncol,:),2)
   831)       call outfld(trim(cbname), cb, pcols, lchnk)
   832) 
   833)    end do
   834) 
   835)    ! Associate pointer with requested gas list
   836)    g_list => gaslist(list_idx)
   837) 
   838)    ngas = g_list%ngas
   839)    do i = 1, ngas
   840) 
   841)       source = g_list%gas(i)%source
   842)       idx    = g_list%gas(i)%idx
   843)       name   = g_list%gas(i)%mass_name
   844)       cbname = 'cb_' // name(3:len_trim(name))
   845)       select case( source )
   846)       case ('A')
   847)          mmr => state%q(:,:,idx)
   848)       case ('N')
   849)          call pbuf_get_field(pbuf, idx, mmr)
   850)       end select
   851) 
   852)       mass(:ncol,:) = mmr(:ncol,:) * state%pdeldry(:ncol,:) * rga
   853)       call outfld(trim(name), mass, pcols, lchnk)
   854) 
   855)       cb(:ncol) = sum(mass(:ncol,:),2)
   856)       call outfld(trim(cbname), cb, pcols, lchnk)
   857) 
   858)    end do
   859) 
   860) end subroutine rad_cnst_out

   958) subroutine list_init1(namelist, gaslist, aerlist, ma_list)
   959) 
   960)    ! Initialize the gas and bulk and modal aerosol lists with the 
   961)    ! entities specified in the climate or diagnostic lists.
   962) 
   963)    ! This first phase initialization just sets the information that
   964)    ! is available at the time the namelist is read.
   965) 
   966)    type(rad_cnst_namelist_t), intent(in) :: namelist ! parsed namelist input for climate or diagnostic lists
   967) 
   968)    type(gaslist_t),        intent(inout) :: gaslist
   969)    type(aerlist_t),        intent(inout) :: aerlist
   970)    type(modelist_t),       intent(inout) :: ma_list
   971) 
   972) 
   973)    ! Local variables
   974)    integer :: ii, idx, m, naero, nmodes
   975)    integer :: igas, ifileindex, ba_idx, ma_idx
   976)    integer :: istat
   977)    character(len=*), parameter :: routine = 'list_init1'
   978)    !-----------------------------------------------------------------------------
   979) 
   980)    ! nradgas is set by the radiative transfer code
   981)    gaslist%ngas = nradgas
   982) 
   983)    ! Determine the number of bulk aerosols and aerosol modes in the list
   984)    naero = 0
   985)    nmodes = 0
   986)    do ii = 1, namelist%ncnst
   987)       if (trim(namelist%type(ii)) == 'A') naero  = naero + 1
   988)       if (trim(namelist%type(ii)) == 'M') nmodes = nmodes + 1
   989)    end do
   990)    aerlist%numaerosols = naero
   991)    ma_list%nmodes      = nmodes

   995)       aerlist%aer(aerlist%numaerosols),      &
   996)       gaslist%gas(gaslist%ngas),             &
   997)       ma_list%idx(ma_list%nmodes),           &
   998)       ma_list%physprop_files(ma_list%nmodes), &
   999)       ma_list%idx_props(ma_list%nmodes),     &
  1000)       stat=istat)
  1001)    if (istat /= 0) call endrun(routine//': allocate ERROR; aero and gas list components')
  1002) 
  1003)    if (masterproc .and. verbose) then
  1004)       if (len_trim(gaslist%list_id) == 0) then
  1005)          write(iulog,*) nl//' '//routine//': namelist input for climate list'
  1006)       else
  1007)          write(iulog,*) nl//' '//routine//': namelist input for diagnostic list:'//gaslist%list_id
  1008)       end if
  1009)    end if
  1010)

  1012)    ba_idx = 0
  1013)    ma_idx = 0
  1014)    do ii = 1, namelist%ncnst
  1015) 
  1016)       if (masterproc .and. verbose) &

  1022)           namelist%source(ii) /= 'N' .and. namelist%source(ii) /= 'Z' ) then
  1023)          call endrun(routine//": source must either be A, M, N or Z:"//&
  1024)                      " illegal specifier in namelist input: "//namelist%source(ii))

  1026)

  1028)       if (namelist%type(ii) == 'A') then 
  1029) 
  1030)          ! Add to bulk aerosol list

  1233) subroutine parse_mode_defs(nl_in, modes)
  1234) 
  1235)    ! Parse the mode definition specifiers.  The specifiers are of the form:
  1236)    ! 
  1237)    ! 'mode_name:mode_type:=',
  1238)    !  'source_num_a:camname_num_a:source_num_c:camname_num_c:num_mr:+',
  1239)    !  'source_mmr_a:camname_mmr_a:source_mmr_c:camname_mmr_c:spec_type:prop_file[:+]'[,]
  1240)    !  ['source_mmr_a:camname_mmr_a:source_mmr_c:camname_mmr_c:spec_type:prop_file][:+][']
  1241)    !
  1242)    ! where the ':' separated fields are:
  1243)    ! mode_name -- name of the mode.
  1244)    ! mode_type -- type of mode.  Valid values are from the MAM code.
  1245)    ! =         -- this line terminator identifies the initial string in a
  1246)    !              mode definition
  1247)    ! +         -- this line terminator indicates that the mode definition is
  1248)    !              continued in the next string
  1249)    ! source_num_a  -- Source of interstitial number mixing ratio,  'A', 'N', or 'Z'
  1250)    ! camname_num_a -- the name of the interstitial number component.  This name must be
  1251)    !                  registered in the constituent arrays when source=A or in the
  1252)    !                  physics buffer when source=N
  1253)    ! source_num_c  -- Source of cloud borne number mixing ratio,  'A', 'N', or 'Z'
  1254)    ! camname_num_c -- the name of the cloud borne number component.  This name must be
  1255)    !                  registered in the constituent arrays when source=A or in the
  1256)    !                  physics buffer when source=N
  1257)    ! source_mmr_a  -- Source of interstitial specie mass mixing ratio,  'A', 'N' or 'Z'
  1258)    ! camname_mmr_a -- the name of the interstitial specie.  This name must be
  1259)    !                  registered in the constituent arrays when source=A or in the
  1260)    !                  physics buffer when source=N
  1261)    ! source_mmr_c  -- Source of cloud borne specie mass mixing ratio,  'A', 'N' or 'Z'
  1262)    ! camname_mmr_c -- the name of the cloud borne specie.  This name must be
  1263)    !                  registered in the constituent arrays when source=A or in the
  1264)    !                  physics buffer when source=N
  1265)    ! spec_type -- species type.  Valid values far from the MAM code, except that
  1266)    !              the value 'num_mr' designates a number mixing ratio and has no
  1267)    !              associated field for the prop_file.  There can only be one entry
  1268)    !              with the num_mr type in a mode definition.
  1269)    ! prop_file -- For aerosol species this is a filename, which is
  1270)    !              identified by a ".nc" suffix.  The file contains optical and 
  1271)    !              other physical properties of the aerosol.
  1272)    !
  1273)    ! A mode definition must contain only 1 string for the number mixing ratio components
  1274)    ! and at least 1 string for the species.
  1275) 
  1276) 
  1277)    character(len=*), intent(inout) :: nl_in(:)    ! namelist input (blanks are removed on output)
  1278)    type(modes_t),    intent(inout) :: modes       ! structure containing parsed input
  1279) 
  1280)    ! Local variables
  1281)    integer :: i, m
  1282)    integer :: istat
  1283)    integer :: nmodes, nstr, istr
  1284)    integer :: mbeg, mcur
  1285)    integer :: nspec, ispec
  1286)    integer :: strlen, ibeg, iend, ipos
  1287)    logical :: num_mr_found
  1288)    character(len=*), parameter :: routine = 'parse_mode_defs'
  1289)    character(len=len(nl_in(1))) :: tmpstr
  1290)    character(len=1)  :: tmp_src_a
  1291)    character(len=32) :: tmp_name_a
  1292)    character(len=1)  :: tmp_src_c
  1293)    character(len=32) :: tmp_name_c
  1294)    character(len=32) :: tmp_type
  1295)    !-------------------------------------------------------------------------
  1296)   
  1297)    ! Determine number of modes defined by counting number of strings that are
  1298)    ! terminated by ':='
  1299)    ! (algorithm stops counting at first blank element).
  1300)    nmodes = 0
  1301)    nstr = 0
  1302)    do m = 1, n_mode_str
  1303) 
  1304)       if (len_trim(nl_in(m)) == 0) exit
  1305)       nstr = nstr + 1
  1306)       
  1307)       ! There are no fields in the input strings in which a blank character is allowed.
  1308)       ! To simplify the parsing go through the input strings and remove blanks.
  1309)       tmpstr = adjustl(nl_in(m))
  1310)       nl_in(m) = tmpstr
  1311)       do
  1312)          strlen = len_trim(nl_in(m))
  1313)          ipos = index(nl_in(m), ' ')
  1314)          if (ipos == 0 .or. ipos > strlen) exit
  1315)          tmpstr = nl_in(m)(:ipos-1) // nl_in(m)(ipos+1:strlen)
  1316)          nl_in(m) = tmpstr
  1317)       end do
  1318)       ! count strings with ':=' terminator
  1319)       if (nl_in(m)(strlen-1:strlen) == ':=') nmodes = nmodes + 1
  1320) 
  1321)    end do
  1322)    modes%nmodes = nmodes
  1323) 
  1324)    ! return if no modes defined
  1325)    if (nmodes == 0) return
  1326) 
  1327)    ! allocate components that depend on nmodes
  1328)    allocate( &
  1329)       modes%names(nmodes),  &
  1330)       modes%types(nmodes),  &
  1331)       modes%comps(nmodes),  &
  1332)       stat=istat )
  1333)    if (istat > 0) then
  1334)       write(iulog,*) routine//': ERROR: cannot allocate storage for modes.  nmodes=', nmodes
  1335)       call endrun(routine//': ERROR allocating storage for modes')
  1336)    end if
  1337)    
  1338) 
  1339)    mcur = 1              ! index of current string being processed
  1340) 
  1341)    ! loop over modes
  1342)    do m = 1, nmodes
  1343) 
  1344)       mbeg = mcur  ! remember the first string of a mode
  1345) 
  1346)       ! check that first string in mode definition is ':=' terminated
  1347)       iend = len_trim(nl_in(mcur))
  1348)       if (nl_in(mcur)(iend-1:iend) /= ':=') call parse_error('= not found', nl_in(mcur))
  1349) 
  1350)       ! count species in mode definition.  definition will contain 1 string with
  1351)       ! with a ':+' terminator for each specie
  1352)       nspec = 0
  1353)       mcur = mcur + 1
  1354)       do
  1355)          iend = len_trim(nl_in(mcur))
  1356)          if (nl_in(mcur)(iend-1:iend) /= ':+') exit
  1357)          nspec = nspec + 1
  1358)          mcur = mcur + 1
  1359)       end do
  1360)       
  1361)       ! a mode must have at least one specie
  1362)       if (nspec == 0) call parse_error('mode must have at least one specie', nl_in(mbeg))
  1363) 
  1364)       ! allocate components that depend on number of species
  1365)       allocate( &
  1366)          modes%comps(m)%source_mmr_a(nspec),  &
  1367)          modes%comps(m)%camname_mmr_a(nspec), &
  1368)          modes%comps(m)%source_mmr_c(nspec),  &
  1369)          modes%comps(m)%camname_mmr_c(nspec), &
  1370)          modes%comps(m)%type(nspec),          &
  1371)          modes%comps(m)%props(nspec),         &
  1372)          stat=istat)
  1373) 
  1374)       if (istat > 0) then
  1375)          write(iulog,*) routine//': ERROR: cannot allocate storage for species.  nspec=', nspec
  1376)          call endrun(routine//': ERROR allocating storage for species')
  1377)       end if
  1378) 
  1379)       ! initialize components
  1380)       modes%comps(m)%nspec         = nspec
  1381)       modes%comps(m)%source_num_a  = ' '
  1382)       modes%comps(m)%camname_num_a = ' '
  1383)       modes%comps(m)%source_num_c  = ' '
  1384)       modes%comps(m)%camname_num_c = ' '
  1385)       do ispec = 1, nspec
  1386)          modes%comps(m)%source_mmr_a(ispec)  = ' '
  1387)          modes%comps(m)%camname_mmr_a(ispec) = ' '
  1388)          modes%comps(m)%source_mmr_c(ispec)  = ' '
  1389)          modes%comps(m)%camname_mmr_c(ispec) = ' '
  1390)          modes%comps(m)%type(ispec)          = ' '
  1391)          modes%comps(m)%props(ispec)         = ' '
  1392)       end do
  1393) 
  1394)       ! return to first string in mode definition
  1395)       mcur = mbeg
  1396)       tmpstr = nl_in(mcur)
  1397)       
  1398)       ! mode name
  1399)       ipos = index(tmpstr, ':')
  1400)       if (ipos < 2) call parse_error('mode name not found', tmpstr)
  1401)       modes%names(m) = tmpstr(:ipos-1)
  1402)       tmpstr         = tmpstr(ipos+1:)
  1403) 
  1404)       ! mode type
  1405)       ipos = index(tmpstr, ':')
  1406)       if (ipos == 0) call parse_error('mode type not found', tmpstr)
  1407)       ! check for valid mode type
  1408)       call check_mode_type(tmpstr, 1, ipos-1)
  1409)       modes%types(m) = tmpstr(:ipos-1)
  1410)       tmpstr         = tmpstr(ipos+1:)
  1411) 
  1412)       ! mode type must be followed by '='
  1413)       if (tmpstr(1:1) /= '=') call parse_error('= not found', tmpstr)
  1414) 
  1415)       ! move to next string
  1416)       mcur = mcur + 1
  1417)       tmpstr = nl_in(mcur)
  1418) 
  1419)       ! process mode component strings
  1420)       num_mr_found = .false.   ! keep track of whether number mixing ratio component is found
  1421)       ispec = 0                ! keep track of the number of species found
  1422)       do
  1423) 
  1424)          ! source of interstitial component
  1425)          ipos = index(tmpstr, ':')
  1426)          if (ipos < 2) call parse_error('expect to find source field first', tmpstr)
  1427)          ! check for valid source
  1428)          if (tmpstr(:ipos-1) /= 'A' .and. tmpstr(:ipos-1) /= 'N' .and. tmpstr(:ipos-1) /= 'Z') &
  1429)             call parse_error('source must be A, N or Z', tmpstr)
  1430)          tmp_src_a = tmpstr(:ipos-1)
  1431)          tmpstr    = tmpstr(ipos+1:)
  1432) 
  1433)          ! name of interstitial component
  1434)          ipos = index(tmpstr, ':')
  1435)          if (ipos == 0) call parse_error('next separator not found', tmpstr)
  1436)          tmp_name_a = tmpstr(:ipos-1)
  1437)          tmpstr     = tmpstr(ipos+1:)
  1438) 
  1439)          ! source of cloud borne component
  1440)          ipos = index(tmpstr, ':')
  1441)          if (ipos < 2) call parse_error('expect to find a source field', tmpstr)
  1442)          ! check for valid source
  1443)          if (tmpstr(:ipos-1) /= 'A' .and. tmpstr(:ipos-1) /= 'N' .and. tmpstr(:ipos-1) /= 'Z') &
  1444)             call parse_error('source must be A, N or Z', tmpstr)
  1445)          tmp_src_c = tmpstr(:ipos-1)
  1446)          tmpstr    = tmpstr(ipos+1:)
  1447) 
  1448)          ! name of cloud borne component
  1449)          ipos = index(tmpstr, ':')
  1450)          if (ipos == 0) call parse_error('next separator not found', tmpstr)
  1451)          tmp_name_c = tmpstr(:ipos-1)
  1452)          tmpstr     = tmpstr(ipos+1:)
  1453) 
  1454)          ! component type
  1455)          ipos = scan(tmpstr, ': ')
  1456)          if (ipos == 0) call parse_error('next separator not found', tmpstr)
  1457) 
  1458)          if (tmpstr(:ipos-1) == 'num_mr') then
  1459) 
  1460)             ! there can only be one number mixing ratio component
  1461)             if (num_mr_found) call parse_error('more than 1 number component', nl_in(mcur))
  1462) 
  1463)             num_mr_found = .true.
  1464)             modes%comps(m)%source_num_a  = tmp_src_a
  1465)             modes%comps(m)%camname_num_a = tmp_name_a
  1466)             modes%comps(m)%source_num_c  = tmp_src_c
  1467)             modes%comps(m)%camname_num_c = tmp_name_c
  1468)             tmpstr                       = tmpstr(ipos+1:)
  1469) 
  1470)          else
  1471) 
  1472)             ! check for valid specie type
  1473)             call check_specie_type(tmpstr, 1, ipos-1)
  1474)             tmp_type = tmpstr(:ipos-1)
  1475)             tmpstr   = tmpstr(ipos+1:)
  1476) 
  1477)             ! get the properties file
  1478)             ipos = scan(tmpstr, ': ')
  1479)             if (ipos == 0) call parse_error('next separator not found', tmpstr)
  1480)             ! check for valid filename -- must have .nc extension
  1481)             if (tmpstr(ipos-3:ipos-1) /= '.nc') &
  1482)                call parse_error('filename not valid', tmpstr)
  1483) 
  1484)             ispec = ispec + 1
  1485)             modes%comps(m)%source_mmr_a(ispec)  = tmp_src_a
  1486)             modes%comps(m)%camname_mmr_a(ispec) = tmp_name_a
  1487)             modes%comps(m)%source_mmr_c(ispec)  = tmp_src_c
  1488)             modes%comps(m)%camname_mmr_c(ispec) = tmp_name_c
  1489)             modes%comps(m)%type(ispec)          = tmp_type
  1490)             modes%comps(m)%props(ispec)         = tmpstr(:ipos-1)
  1491)             tmpstr                              = tmpstr(ipos+1:)
  1492)          end if
  1493) 
  1494)          ! check if there are more components.  either the current character is
  1495)          ! a ' ' which means this string is the final mode component, or the character
  1496)          ! is a '+' which means there are more components
  1497)          if (tmpstr(1:1) == ' ') exit
  1498) 
  1499)          if (tmpstr(1:1) /= '+') &
  1500)                call parse_error('+ field not found', tmpstr)
  1501) 
  1502)          ! continue to next component...
  1503)          mcur = mcur + 1
  1504)          tmpstr = nl_in(mcur)
  1505)       end do

  1575) subroutine parse_rad_specifier(specifier, namelist_data)
  1576) 
  1577) !-----------------------------------------------------------------------------
  1578) ! Private method for parsing the radiation namelist specifiers.  The specifiers
  1579) ! are of the form 'source_camname:radname' where:
  1580) ! source  -- either 'N' for pbuf (non-advected) or 'A' for state (advected)
  1581) ! camname -- the name of a constituent that must be found in the constituent
  1582) !            component of the state when source=A or in the physics buffer
  1583) !            when source=N
  1584) ! radname -- For gases this is a name that identifies the constituent to the
  1585) !            radiative transfer codes.  These names are contained in the
  1586) !            radconstants module.  For aerosols this is a filename, which is
  1587) !            identified by a ".nc" suffix.  The file contains optical and 
  1588) !            other physical properties of the aerosol.
  1589) !
  1590) ! This code also identifies whether the constituent is a gas or an aerosol
  1591) ! and adds that info to a structure that stores the parsed data.
  1592) !-----------------------------------------------------------------------------
  1593) 
  1594)     character(len=*), dimension(:), intent(in) :: specifier
  1595)     type(rad_cnst_namelist_t),   intent(inout) :: namelist_data
  1596) 
  1597)     ! Local variables
  1598)     integer            :: number, i, j
  1599)     integer            :: ipos, strlen
  1600)     integer            :: astat
  1601)     character(len=cs1) :: tmpstr
  1602)     character(len=1)   :: source(n_rad_cnst)
  1603)     character(len=64)  :: camname(n_rad_cnst)
  1604)     character(len=cs1) :: radname(n_rad_cnst)
  1605)     character(len=1)   :: type(n_rad_cnst)
  1606)     !-------------------------------------------------------------------------
  1607)   
  1608)     number = 0
  1609) 
  1610)     parse_loop: do i = 1, n_rad_cnst
  1611)       if ( len_trim(specifier(i)) == 0 ) then 
  1612)          exit parse_loop
  1613)       endif
  1614) 
  1615)       ! There are no fields in the input strings in which a blank character is allowed.
  1616)       ! To simplify the parsing go through the input strings and remove blanks.
  1617)       tmpstr = adjustl(specifier(i))
  1618)       do
  1619)          strlen = len_trim(tmpstr)
  1620)          ipos = index(tmpstr, ' ')
  1621)          if (ipos == 0 .or. ipos > strlen) exit
  1622)          tmpstr = tmpstr(:ipos-1) // tmpstr(ipos+1:strlen)
  1623)       end do
  1624) 
  1625)       ! Locate the ':' separating source from camname.
  1626)       j = index(tmpstr, ':')
  1627)       source(i) = tmpstr(:j-1)
  1628)       tmpstr = tmpstr(j+1:)
  1629) 
  1630)       ! locate the ':' separating camname from radname
  1631)       j = scan(tmpstr, ':')
  1632)  
  1633)       camname(i) = tmpstr(:j-1)
  1634)       radname(i) = tmpstr(j+1:)
  1635) 
  1636)       ! determine the type of constituent
  1637)       if (source(i) == 'M') then 
  1638)          type(i) = 'M'
  1639)       else if(index(radname(i),".nc") .gt. 0) then
  1640)          type(i) = 'A'
  1641)       else
  1642)          type(i) = 'G'
  1643)       end if
  1644) 
  1645)       number = number+1    
  1646)     end do parse_loop
  1647) 
  1648)     namelist_data%ncnst = number
  1649) 
  1650)     if (number == 0) return
  1651) 
  1652)     allocate(namelist_data%source (number), stat=astat)
  1653)     if( astat/= 0 ) call endrun('parse_rad_specifier: not able to allocate namelist_data%source')
  1654)     allocate(namelist_data%camname(number), stat=astat)
  1655)     if( astat/= 0 ) call endrun('parse_rad_specifier: not able to allocate namelist_data%camname')
  1656)     allocate(namelist_data%radname(number), stat=astat)
  1657)     if( astat/= 0 ) call endrun('parse_rad_specifier: not able to allocate namelist_data%radname')
  1658)     allocate(namelist_data%type(number), stat=astat)
  1659)     if( astat/= 0 ) call endrun('parse_rad_specifier: not able to allocate namelist_data%type')
  1660) 
  1661)     namelist_data%source(:namelist_data%ncnst)  = source (:namelist_data%ncnst)
  1662)     namelist_data%camname(:namelist_data%ncnst) = camname(:namelist_data%ncnst)
  1663)     namelist_data%radname(:namelist_data%ncnst) = radname(:namelist_data%ncnst)
  1664)     namelist_data%type(:namelist_data%ncnst)    = type(:namelist_data%ncnst)
  1665) 
  1666) end subroutine parse_rad_specifier
