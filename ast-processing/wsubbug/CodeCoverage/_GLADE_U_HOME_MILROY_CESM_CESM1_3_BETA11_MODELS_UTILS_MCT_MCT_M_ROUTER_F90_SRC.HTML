    11) ! The Router data type contains all the information needed

    36) #ifdef SEQUENCE

    47) !% On return, pe_list is the processor ranks of the other

    59)       sequence

    72)       integer,dimension(:),pointer   :: permarr    ! possible permutation array
    73)       integer,dimension(:,:),pointer :: seg_starts ! starting index
    74)       integer,dimension(:,:),pointer :: seg_lengths! total length
    75)       type(rptr),dimension(:),pointer :: rp1       ! buffer to hold real data
    76)       type(iptr),dimension(:),pointer :: ip1       ! buffer to hold integer data
    77)       integer,dimension(:),pointer   :: ireqs,rreqs  ! buffer for MPI_Requests
    78)       integer,dimension(:,:),pointer :: istatus,rstatus  ! buffer for MPI_Status
    79)     end type Router
    80)
