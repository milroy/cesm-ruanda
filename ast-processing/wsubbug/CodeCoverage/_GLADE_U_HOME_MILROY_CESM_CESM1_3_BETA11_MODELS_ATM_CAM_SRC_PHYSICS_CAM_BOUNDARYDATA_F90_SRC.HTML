    54)   subroutine boundarydata_init(bndyfilename,phys_state,fieldnames,fieldcnt,bndydata,vertextrap)
    55)     implicit none    
    56)     character(len=*),intent(in) :: bndyfilename
    57)     type(physics_state), intent(in):: phys_state(begchunk:endchunk)
    58)     integer,intent(in) :: fieldcnt
    59)     character(len=*), intent(in) :: fieldnames(fieldcnt)
    60)     type(boundarydata_type),intent(out) :: bndydata 
    61)     integer,intent(in), optional :: vertextrap ! if 0 set values outside output grid to 0
    62)                                                ! if 1 set to boundary value
    63)                                                ! if 2 set to cyclic boundaries 
    64)                                                ! if 3 leave on input data grid and extrapolate later
    65)     real(r8), pointer :: datain(:,:,:,:,:)
    66)     integer :: lchnk
    67) 
    68)     bndydata%fieldcnt=fieldcnt
    69)     if(present(vertextrap)) then
    70)        bndydata%vertextrap=vertextrap
    71)     else
    72)        bndydata%vertextrap=0
    73)     end if
    74)     nullify(bndydata%fields)
    75) 
    76)     call boundarydata_read(phys_state,bndyfilename,fieldcnt,fieldnames,bndydata,datain)
    77) 
    78)     if(bndydata%iszonal) then
    79)        call boundarydata_interpolate(phys_state,datain,bndydata)
    80) 
    81)        allocate(bndydata%datainst(size(bndydata%fields,1),size(bndydata%fields,2), &
    82)             begchunk:endchunk,bndydata%fieldcnt))
    83)     
    84)        deallocate(datain)    
    85)     end if
    86)   end subroutine boundarydata_init

    88)   subroutine boundarydata_update(phys_state, bndydata, update_out)
    89)     use interpolate_data,only : get_timeinterp_factors
    90)     type(physics_state), intent(in) :: phys_state(begchunk:endchunk)
    91)     type(boundarydata_type), intent(inout) :: bndydata
    92)     logical, intent(out), optional :: update_out
    93)     real(r8) :: cdate
    94)     integer :: nm, np, lchnk, j, k, fld, cols, cole, ncol, ndims
    95)     real(r8) :: fact1, fact2
    96)     real(r8), allocatable :: datain(:,:,:,:,:)
    97)     logical :: update
    98)     integer :: latspan
    99)     integer :: kmax
   100)     integer :: count(4), start(4), ierr
   101) 
   102)     
   103)     call get_data_bounding_date_indices(bndydata%cdates,bndydata%nm,bndydata%np,cdate,update)
   104)     if(present(update_out)) update_out=update
   105)     nm= bndydata%nm
   106)     np= bndydata%np
   107) 
   108)     call get_timeinterp_factors(.true., np, bndydata%cdates(nm), bndydata%cdates(np), &
   109)          cdate, fact1, fact2, _FILE)
   110) 
   111)     if(size(bndydata%fields,5).eq.2) then 
   112)        nm=1
   113)        np=2
   114)        if(update) then ! we need to read in the next month and interpolate
   115)           if(bndydata%isncol) then
   116)              bndydata%fields(:,:,:,:,nm)=bndydata%fields(:,:,:,:,np)
   117)              do lchnk=begchunk,endchunk
   118)                 ncol=phys_state(lchnk)%ncol
   119)                 cols=1
   120)                 cole=cols+bndydata%count(cols,lchnk)-1
   121)                 do while(cole<=ncol)
   122)                    
   123)                    if(bndydata%levsiz==1) then               
   124)                       ndims=2
   125)                       start=(/bndydata%start(cols,lchnk),bndydata%np,-1,-1/)
   126)                       count=(/bndydata%count(cols,lchnk),1,-1,-1/)
   127)                    else
   128)                       ndims=3
   129)                       start=(/bndydata%start(cols,lchnk),bndydata%levsiz,bndydata%np,-1/)
   130)                       count=(/bndydata%count(cols,lchnk),1,1,-1/)
   131)                    end if
   132)                    do fld=1,bndydata%fieldcnt
   133)                       call handle_ncerr( nf90_get_var(bndydata%ncid, bndydata%dataid(fld) , &
   134)                            bndydata%fields(cols:cole,:,lchnk,fld,np),  &
   135)                            start(1:ndims), count(1:ndims)),&
   136)                            _FILE,__LINE__)
   137) 
   138)                    end do
   139)                    if(cols==ncol) exit
   140)                    cols=cols+bndydata%count(cols,lchnk)
   141)                    cole=cols+bndydata%count(cols,lchnk)-1
   142)                 end do
   143)              end do
   144) 
   145)           else
   146)              allocate(datain(ptrlon,bndydata%levsiz,bndydata%latsiz,1,bndydata%fieldcnt))
   147)              if(masterproc) then
   148)                 count(1)=ptrlon
   149)                 count(2)=bndydata%levsiz
   150)                 count(3)=bndydata%latsiz
   151)                 count(4)=1
   152)                 start(1)=1
   153)                 start(2)=1
   154)                 start(3)=1
   155)                 start(4)=bndydata%np
   156)                 write(iulog,*) 'boundarydata reading data for month: ',bndydata%np
   157)                 do fld=1,bndydata%fieldcnt
   158)                    call handle_ncerr( nf90_get_var(bndydata%ncid, bndydata%dataid(fld), &
   159)                         datain(:,:,:,:,fld), start, count),_FILE,__LINE__)
   160)                 end do
   161)              end if
   162) #ifdef SPMD
   163)              call mpibcast (datain, bndydata%levsiz*bndydata%latsiz*1*bndydata%fieldcnt, mpir8, 0, mpicom, ierr)
   164) #endif          
   165)              bndydata%fields(:,:,:,:,nm) = bndydata%fields(:,:,:,:,np) 
   166)              call boundarydata_interpolate(phys_state,datain,bndydata)
   167)              deallocate(datain)
   168)           end if
   169)        end if
   170)     end if
   171)     kmax = size(bndydata%fields,2)
   172)    
   173)     do fld=1,bndydata%fieldcnt
   174)        do lchnk=begchunk,endchunk
   175)           if(bndydata%isncol) then
   176)              latspan = phys_state(lchnk)%ncol
   177)           else
   178)              latspan=phys_state(lchnk)%ulatcnt
   179)           end if
   180)           do k=1,kmax
   181)              do j=1,latspan
   182)                 bndydata%datainst(j,k,lchnk,fld)=bndydata%fields(j,k,lchnk,fld,nm)*fact1 + &
   183)                      bndydata%fields(j,k,lchnk,fld,np)*fact2
   184)              end do
   185)           end do
   186)        end do
   187)     end do
   188)   end subroutine boundarydata_update

   191)   subroutine boundarydata_read(phys_state,bndyfilename,fieldcnt,fieldnames,bndydata,datain)
   192)     !----------------------------------------------------------------------- 
   193)     ! 
   194)     ! Purpose: 
   195)     ! Do initial read of time-variant boundary dataset, containing
   196)     ! 12 monthly fields as a function of latitude and pressure.  Determine the two
   197)     ! consecutive months between which the current date lies.
   198)     ! 
   199)     ! Method: 
   200)     ! 
   201)     ! Author: NCAR CMS
   202)     !-----------------------------------------------------------------------
   203)     use ioFileMod, only : getfil
   204) 
   205) 
   206)     implicit none    
   207)     type(physics_state), intent(in) :: phys_state(begchunk:endchunk)
   208)     character(len=*),intent(in) :: bndyfilename
   209)     integer,intent(in) :: fieldcnt
   210)     character(len=*), intent(in) :: fieldnames(fieldcnt)
   211)     type(boundarydata_type), intent(inout) :: bndydata
   212)     real(r8), pointer :: datain(:,:,:,:,:)  ! 
   213)     !
   214)     ! Local variables
   215)     !
   216)     integer :: londimid
   217)     integer :: latdimid
   218)     integer :: ncoldimid
   219)     integer :: levdimid
   220)     integer :: ilevdimid
   221)     integer :: timdimid
   222)     integer :: ndims
   223)     integer :: dimlen
   224)     integer :: ilevsiz
   225)     integer :: ncolsiz
   226)     character(len=nf90_max_name) :: dimname
   227) 
   228) 
   229) !    integer ::  ncid              ! netcdf id for file
   230)     integer ::  dateid            ! netcdf id for date variable
   231)     integer ::  secid             ! netcdf id for seconds variable
   232)     integer ::  lonid             ! netcdf id for longitude variable
   233)     integer ::  ncolid             ! netcdf id for longitude variable
   234)     integer ::  latid             ! netcdf id for latitude variable
   235)     integer ::  levid             ! netcdf id for level variable
   236)     integer ::  timid             ! netcdf id for time variable
   237)     integer :: hybid
   238) 
   239)     integer ::  dataid  ! netcdf id for data fields
   240) 
   241)     integer ::  lonsiz            ! size of longitude dimension on tracer dataset
   242)     integer ::  levsiz            ! size of level dimension on tracer dataset
   243)     integer ::  latsiz            ! size of latitude dimension on tracer dataset
   244) 
   245)     integer ::  j,n,k,nt,id          ! indices
   246)     integer ::  ki,ko,ji,jo       ! indices
   247)     integer :: date_tr(ptrtim), sec_tr(ptrtim)
   248) 
   249)     integer :: dimids(4), start(4), count(4)
   250)     integer, pointer :: columnmap(:)
   251)     real(r8) :: calday        ! current calendar day
   252)     real(r8), pointer :: pin(:)
   253)     real(r8), allocatable :: tmp_ps(:,:), tmp_fld(:,:,:)
   254)     integer :: mincid,maxcid
   255)     real(r8), allocatable, target:: lati(:)
   256)     integer :: cols, cole
   257)     integer ::  ierr, dimcnt
   258)     integer ::  i, ncol, lchnk
   259)     character(len=256) :: locfn    ! netcdf local filename to open 
   260) 
   261)     !
   262)     !-----------------------------------------------------------------------
   263)     !
   264)     ! SPMD: Master reads dataset and does broadcast.  All subsequent interpolation is
   265)     !       done in every process.  This is not required, one could remove this conditional
   266)     !       and read the dataset independently on each task.
   267)     !
   268)     if(masterproc) then
   269)        write(iulog,*)'boundarydata_read: Reading from: ', trim(bndyfilename)
   270) #ifndef USE_MASTERPROC
   271)     end if
   272) #endif
   273)     call getfil(bndyfilename, locfn)
   274)     call handle_ncerr( nf90_open(locfn, 0, bndydata%ncid),&
   275)          _FILE,__LINE__)
   276) 
   277)     !       write(iulog,*)'boundarydata_read: NCOPN returns id ',bndydata%ncid,' for file ',trim(locfn)
   278)     !
   279)     !------------------------------------------------------------------------
   280)     ! Read tracer data
   281)     !------------------------------------------------------------------------
   282)     !
   283)     ! Get dimension info
   284)     !
   285)     nullify(columnmap)
   286)     nullify(pin)
   287)     call handle_ncerr( nf90_inquire(bndydata%ncid, bndydata%ndims, unlimiteddimid=timdimid), &
   288)          _FILE,__LINE__)
   289)     ncolsiz=-1
   290)     levsiz=-1
   291)     lonsiz=-1
   292)     latsiz=-1
   293)     do i=1,bndydata%ndims
   294)        call handle_ncerr( nf90_inquire_dimension(bndydata%ncid, i, dimname, dimlen),&
   295)             _FILE,__LINE__)
   296)        if (dimname(1:3).eq.'lat') then
   297)           latdimid=i
   298)           latsiz=dimlen
   299)        else if (dimname(1:3) .eq.'lon') then
   300)           londimid=i
   301)           lonsiz=dimlen
   302)        else if (dimname(1:4) .eq. 'ncol') then
   303)           ncoldimid=i
   304)           ncolsiz=dimlen
   305)           bndydata%isncol=.true. 
   306)        else if (dimname(1:3) .eq. 'lev') then
   307)           levdimid=i
   308)           levsiz=dimlen
   309)        else if (dimname(1:4) .eq. 'ilev') then
   310)           ilevdimid=i
   311)           ilevsiz=dimlen
   312)        else if (dimname(1:4) .eq. 'time') then
   313)           if(timdimid/=i) then
   314)              timdimid=i
   315)           end if
   316)           bndydata%timsiz=dimlen
   317)        else
   318)           write(iulog,*) 'Warning: do not know how to handle dimension ',&
   319)                trim(dimname), ' in boundarydata.F90:313'
   320)        end if
   321)     end do
   322) 
   323)     if (latsiz>0 .and. lonsiz<=1) then
   324)        bndydata%iszonal=.true.
   325)        allocate(bndydata%lat(latsiz))
   326)     end if
   327)     if(bndydata%isncol) then
   328) !       allocate (columnmap(ncolsiz))
   329) !       call handle_ncerr( nf90_inq_varid(bndydata%ncid, 'ncol'    , ncolid),&
   330) !            _FILE,__LINE__)
   331) !       call handle_ncerr( nf90_get_var(bndydata%ncid,ncolid,columnmap), &
   332) !            _FILE,__LINE__)
   333)        if(levsiz>0) then
   334)           allocate(bndydata%fields(pcols,levsiz,begchunk:endchunk,fieldcnt,2))
   335) 
   336)           ierr = nf90_inq_varid(bndydata%ncid, 'PS', bndydata%psid)
   337)           if(ierr.eq.NF90_NOERR) then
   338)              allocate(bndydata%ps(pcols,begchunk:endchunk,2))
   339)              allocate(bndydata%hybi(levsiz+1))
   340)              call handle_ncerr(nf90_inq_varid(bndydata%ncid,'hybi',hybid),&
   341)                   _FILE,__LINE__)
   342)              call handle_ncerr( nf90_get_var(bndydata%ncid, hybid, bndydata%hybi ),&
   343)                   _FILE,__LINE__)
   344)           else 
   345)              call endrun('Did not recognize a vertical coordinate variable')
   346)           end if
   347)        else
   348)           levsiz=1
   349)           allocate(bndydata%fields(pcols,1,begchunk:endchunk,fieldcnt,2))
   350)        end if
   351)     else
   352)        allocate(datain(lonsiz,levsiz,latsiz,2,fieldcnt))
   353)        !
   354)        ! Check dimension info
   355)        !
   356)        if (lonsiz/=ptrlon) then
   357)           call endrun ('BOUNDARYDATA_READ: longitude dependence not implemented')
   358)        endif
   359) 
   360)        if (bndydata%timsiz /= ptrtim) then
   361)           write(iulog,*)'BOUNDARYDATA_READ: timsiz=',bndydata%timsiz,' must = ptrtim=',ptrtim
   362)           call endrun
   363)        end if
   364)        if( bndydata%vertextrap.lt.3) then
   365)           allocate(pin(levsiz))
   366)        else
   367)           allocate(bndydata%pin(levsiz))
   368)           pin => bndydata%pin
   369)        end if
   370)        allocate(bndydata%lat(latsiz))
   371) 
   372) 
   373)        allocate(datain(ptrlon,levsiz,latsiz,2,fieldcnt))
   374) 
   375)        call handle_ncerr( nf90_inq_varid(bndydata%ncid, 'lat'    , latid),&
   376)             _FILE,__LINE__)
   377)     end if
   378)     !
   379)     ! Determine necessary dimension and variable id's
   380)     !
   381)     allocate(bndydata%cdates(bndydata%timsiz))
   382) 
   383)     call handle_ncerr(nf90_inq_varid(bndydata%ncid, 'date'   , dateid), &
   384)          _FILE,__LINE__)
   385)     call handle_ncerr( nf90_get_var(bndydata%ncid, dateid, date_tr),&
   386)          _FILE,__LINE__)
   387)     ierr = nf90_inq_varid(bndydata%ncid, 'datesec', secid)
   388)     if(ierr==NF90_NOERR) then
   389)        call handle_ncerr( nf90_get_var(bndydata%ncid, secid , sec_tr),&
   390)             _FILE,__LINE__)
   391)     else
   392)        sec_tr=0
   393)     end if
   394) 
   395)     if (mod(date_tr(1),10000)/100 /= 1) then
   396)        call endrun ('(boundarydata_read): error when cycling data: 1st month must be 1')
   397)     end if
   398)     if (mod(date_tr(ptrtim),10000)/100 /= 12) then
   399)        call endrun ('(boundarydata_read): error when cycling data: last month must be 12')
   400)     end if
   401)     !
   402)     !    return the calander dates of the file data
   403)     !
   404)     do n=1,ptrtim
   405)        call bnddyi(date_tr(n), sec_tr(n), bndydata%cdates(n))
   406)     end do
   407) !    else
   408) !       call handle_ncerr( nf90_inq_varid(bndydata%ncid, 'time', dateid),&
   409) !            _FILE,__LINE__)
   410) !
   411) !       call handle_ncerr( nf90_get_var(bndydata%ncid, dateid, bndydata%cdates),&
   412) !            _FILE,__LINE__)
   413) !
   414) !    end if
   415) #ifdef USE_MASTERPROC
   416)  else
   417)     allocate(bndydata%cdates(ptrtim))
   418)  end if
   419) #ifdef SPMD
   420)  call mpibcast (bndydata%cdates, ptrtim, mpir8, 0, mpicom, ierr)
   421) #endif
   422) #endif
   423)  bndydata%nm=12
   424)  bndydata%np=1
   425)  call get_data_bounding_date_indices(bndydata%cdates,bndydata%nm,bndydata%np)
   426) #ifdef USE_MASTERPROC
   427)  if(masterproc) then
   428) #endif
   429)     !
   430)     ! Obtain entire date and sec variables. Assume that will always
   431)     ! cycle over 12 month data.
   432)     !
   433)     !
   434)     ! Obtain input data latitude and level arrays.
   435)     !
   436)     if(bndydata%iszonal) then
   437)        call handle_ncerr( nf90_get_var(bndydata%ncid, latid, bndydata%lat),&
   438)             _FILE,__LINE__)
   439)        ierr = nf90_inq_varid(bndydata%ncid, 'lev'    , levid)
   440)        call handle_ncerr( nf90_get_var(bndydata%ncid, levid, pin ),&
   441)             _FILE,__LINE__)
   442)     end if
   443) 
   444)     allocate(bndydata%dataid(fieldcnt))
   445)     if(masterproc) then
   446)        write(iulog,*) 'boundarydata reading data for months: ',bndydata%nm,bndydata%np
   447)     end if
   448)     do i=1,fieldcnt
   449)        call handle_ncerr( nf90_inq_varid(bndydata%ncid, fieldnames(i)   , bndydata%dataid(i)),&
   450)             _FILE,__LINE__)
   451)     end do
   452)     if(bndydata%isncol) then
   453)        allocate(bndydata%start(pcols,begchunk:endchunk), &
   454)             bndydata%count(pcols,begchunk:endchunk))
   455) 
   456) !
   457) !  For i/o efficiency we read in a block of data which includes the data needed on this 
   458) !  processor but which may in fact include data not needed here.  physics cids are just the
   459) !  offset into the file.
   460) ! 
   461) 
   462)        bndydata%start=-1
   463)        bndydata%count=1
   464)        mincid=2147483647
   465)        maxcid=-1
   466)        do lchnk=begchunk,endchunk
   467)           ncol=phys_state(lchnk)%ncol
   468)           i=minval(phys_state(lchnk)%cid(1:ncol)) 
   469)           if(i < mincid) mincid = i
   470)           i=maxval(phys_state(lchnk)%cid(1:ncol))
   471)           if(i > maxcid) maxcid = i
   472)        end do
   473) 
   474)        allocate(tmp_ps(mincid:maxcid,2))
   475)        start=(/mincid,bndydata%nm,1,-1/)
   476)        if(bndydata%np>bndydata%nm) then
   477)           count=(/maxcid-mincid+1,2,-1,-1/)
   478)        else
   479)           count=(/maxcid-mincid+1,1,-1,-1/)
   480)        end if
   481)        if(associated(bndydata%ps) ) then
   482)           call handle_ncerr( nf90_get_var(bndydata%ncid, bndydata%psid , &
   483)                tmp_ps(:,1:count(2)), start(1:2), &
   484)                count(1:2)),&
   485)                _FILE,__LINE__)
   486)           if(bndydata%np<bndydata%nm) then
   487)              start(2)=bndydata%np
   488)              call handle_ncerr( nf90_get_var(bndydata%ncid, bndydata%psid , &
   489)                   tmp_ps(:,2:2), start(1:2), &
   490)                   count(1:2)),&
   491)                   _FILE,__LINE__)
   492)           end if
   493) 
   494)           do lchnk=begchunk,endchunk
   495)              do n=1,phys_state(lchnk)%ncol
   496)                 bndydata%ps(n ,lchnk,:) = tmp_ps(phys_state(lchnk)%cid(n),:)
   497)              end do
   498)           end do
   499)           deallocate(tmp_ps)
   500) 
   501)        end if
   502)

   797)   subroutine boundarydata_vert_interp(lchnk, ncol, levsiz, fldcnt, pin, pmid, datain, dataout)
   798)     !----------------------------------------------------------------------- 
   799)     ! 
   800)     ! Purpose: Interpolate ozone from current time-interpolated values to model levels
   801)     ! 
   802)     ! Method: Use pressure values to determine interpolation levels
   803)     ! 
   804)     ! Author: Bruce Briegleb
   805)     ! 
   806)     !--------------------------------------------------------------------------
   807)     implicit none
   808)     ! Arguments
   809)     !
   810)     integer, intent(in) :: lchnk               ! chunk identifier
   811)     integer, intent(in) :: ncol                ! number of atmospheric columns
   812)     integer, intent(in) :: levsiz
   813)     integer, intent(in) :: fldcnt
   814)     real(r8), intent(in) :: pin(levsiz)
   815)     real(r8), intent(in) :: pmid(pcols,pver)   ! level pressures (mks)
   816)     real(r8), intent(in) :: datain(pcols,levsiz,fldcnt)
   817)     real(r8), intent(out) :: dataout(pcols,pver,fldcnt)    ! ozone mass mixing ratio
   818)     !
   819)     ! local storage
   820)     !
   821) 
   822)     integer ::  i                   ! longitude index
   823)     integer ::  k, kk, kkstart      ! level indices
   824)     integer ::  kupper(pcols)       ! Level indices for interpolation
   825)     integer ::  kount               ! Counter
   826)     integer ::  fld
   827)     real(r8) dpu                ! upper level pressure difference
   828)     real(r8) dpl                ! lower level pressure difference
   829)     !--------------------------------------------------------------------------
   830)     !
   831)     ! Initialize index array
   832)     !
   833)     do i=1,ncol
   834)        kupper(i) = 1
   835)     end do
   836) 
   837)     do k=1,pver
   838)        !
   839)        ! Top level we need to start looking is the top level for the previous k
   840)        ! for all longitude points
   841)        !
   842)        kkstart = levsiz
   843)        do i=1,ncol
   844)           kkstart = min0(kkstart,kupper(i))
   845)        end do
   846)        kount = 0
   847)        !
   848)        ! Store level indices for interpolation
   849)        !
   850)        do kk=kkstart,levsiz-1
   851)           do i=1,ncol
   852)              if (pin(kk).lt.pmid(i,k) .and. pmid(i,k).le.pin(kk+1)) then
   853)                 kupper(i) = kk
   854)                 kount = kount + 1
   855)              end if
   856)           end do
   857)           !
   858)           ! If all indices for this level have been found, do the interpolation and
   859)           ! go to the next level
   860)           !
   861)           if (kount.eq.ncol) then
   862)              do fld=1,fldcnt
   863)                 do i=1,ncol
   864)                    dpu = pmid(i,k) - pin(kupper(i))
   865)                    dpl = pin(kupper(i)+1) - pmid(i,k)
   866)                    dataout(i,k,fld) = (datain(i,kupper(i),fld )*dpl + &
   867)                         datain(i,kupper(i)+1,fld)*dpu)/(dpl + dpu)
   868) 
   869)                 end do
   870)              end do
   871)              goto 35
   872)           end if
   873)        end do
   874)        !
   875)        ! If we've fallen through the kk=1,levsiz-1 loop, we cannot interpolate and
   876)        ! must extrapolate from the bottom or top data level for at least some
   877)        ! of the longitude points.
   878)        !
   879)        do fld=1,fldcnt
   880)           do i=1,ncol
   881)              if (pmid(i,k) .lt. pin(1)) then
   882)                 dataout(i,k,fld) = datain(i,1,fld)*pmid(i,k)/pin(1)
   883)              else if (pmid(i,k) .gt. pin(levsiz)) then
   884)                 dataout(i,k,fld) = datain(i,levsiz,fld)
   885)              else
   886)                 dpu = pmid(i,k) - pin(kupper(i))
   887)                 dpl = pin(kupper(i)+1) - pmid(i,k)
   888)                 dataout(i,k,fld) = (datain(i,kupper(i),fld )*dpl + &
   889)                      datain(i,kupper(i)+1,fld)*dpu)/(dpl + dpu)
   890)              end if
   891)           end do
   892)        end do
   893)        if (kount.gt.ncol) then
   894)           call endrun ('ozone_data_vert_interp: Bad ozone data: non-monotonicity suspected')
   895)        end if
   896) 35     continue
   897)     end do
   898)   end subroutine boundarydata_vert_interp
