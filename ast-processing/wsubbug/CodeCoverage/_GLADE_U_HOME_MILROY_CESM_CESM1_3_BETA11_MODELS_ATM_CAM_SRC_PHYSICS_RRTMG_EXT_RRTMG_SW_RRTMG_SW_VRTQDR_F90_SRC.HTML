    36) ! Purpose: This routine performs the vertical quadrature integration

    52)       integer, intent (in) :: klev                   ! number of model layers
    53)       integer, intent (in) :: kw                     ! g-point index

    55)       real(kind=r8), intent(in) :: pref(:)                    ! direct beam reflectivity
    56)                                                                  !   Dimensions: (nlayers+1)
    57)       real(kind=r8), intent(in) :: prefd(:)                   ! diffuse beam reflectivity
    58)                                                                  !   Dimensions: (nlayers+1)
    59)       real(kind=r8), intent(in) :: ptra(:)                    ! direct beam transmissivity
    60)                                                                  !   Dimensions: (nlayers+1)
    61)       real(kind=r8), intent(in) :: ptrad(:)                   ! diffuse beam transmissivity
    62)                                                                  !   Dimensions: (nlayers+1)
    63) 
    64)       real(kind=r8), intent(in) :: pdbt(:)
    65)                                                                  !   Dimensions: (nlayers+1)
    66)       real(kind=r8), intent(in) :: ptdbt(:)
    67)                                                                  !   Dimensions: (nlayers+1)
    68)

    70)                                                                  !   Dimensions: (nlayers+1)
    71)       real(kind=r8), intent(inout) :: prup(:)
    72)                                                                  !   Dimensions: (nlayers+1)
    73)       real(kind=r8), intent(inout) :: prupd(:)
    74)                                                                  !   Dimensions: (nlayers+1)
    75) 
    76) ! Output
    77)       real(kind=r8), intent(out) :: pfd(:,:)                   ! downwelling flux (W/m2)
    78)                                                                  !   Dimensions: (nlayers+1,ngptsw)
    79)                                                                  ! unadjusted for earth/sun distance or zenith angle
    80)       real(kind=r8), intent(out) :: pfu(:,:)                   ! upwelling flux (W/m2)

    82)                                                                  ! unadjusted for earth/sun distance or zenith angle
    83)

    91) ! Definitions
    92) !
    93) ! pref(jk)   direct reflectance
    94) ! prefd(jk)  diffuse reflectance
    95) ! ptra(jk)   direct transmittance
    96) ! ptrad(jk)  diffuse transmittance
    97) !
    98) ! pdbt(jk)   layer mean direct beam transmittance
    99) ! ptdbt(jk)  total direct beam transmittance at levels
   100) !
   101) !-----------------------------------------------------------------------------
   102)                    
   103) ! Link lowest layer with surface
   104)              
   105)       zreflect = 1._r8 / (1._r8 - prefd(klev+1) * prefd(klev))
   106)       prup(klev) = pref(klev) + (ptrad(klev) * &
   107)                  ((ptra(klev) - pdbt(klev)) * prefd(klev+1) + &
   108)                    pdbt(klev) * pref(klev+1))) * zreflect
   109)       prupd(klev) = prefd(klev) + ptrad(klev) * ptrad(klev) * &
   110)                     prefd(klev+1) * zreflect
   111) 
   112) ! Pass from bottom to top 
   113) 
   114)       do jk = 1,klev-1
   115)          ikp = klev+1-jk                       
   116)          ikx = ikp-1
   117)          zreflect = 1._r8 / (1._r8 -prupd(ikp) * prefd(ikx))
   118)          prup(ikx) = pref(ikx) + (ptrad(ikx) * &
   119)                    ((ptra(ikx) - pdbt(ikx)) * prupd(ikp) + &
   120)                      pdbt(ikx) * prup(ikp))) * zreflect
   121)          prupd(ikx) = prefd(ikx) + ptrad(ikx) * ptrad(ikx) * &
   122)                       prupd(ikp) * zreflect
   123)       enddo
   124)     
   125) ! Upper boundary conditions
   126) 
   127)       ztdn(1) = 1._r8
   128)       prdnd(1) = 0._r8
   129)       ztdn(2) = ptra(1)
   130)       prdnd(2) = prefd(1)
   131) 
   132) ! Pass from top to bottom
   133) 
   134)       do jk = 2,klev
   135)          ikp = jk+1
   136)          zreflect = 1._r8 / (1._r8 - prefd(jk) * prdnd(jk))
   137)          ztdn(ikp) = ptdbt(jk) * ptra(jk) + &
   138)                     (ptrad(jk) * ((ztdn(jk) - ptdbt(jk)) + &
   139)                      ptdbt(jk) * pref(jk) * prdnd(jk))) * zreflect
   140)          prdnd(ikp) = prefd(jk) + ptrad(jk) * ptrad(jk) * &
   141)                       prdnd(jk) * zreflect
   142)       enddo
   143)     
   144) ! Up and down-welling fluxes at levels
   145) 
   146)       do jk = 1,klev+1
   147)          zreflect = 1._r8 / (1._r8 - prdnd(jk) * prupd(jk))
   148)          pfu(jk,kw) = (ptdbt(jk) * prup(jk) + &
   149)                       (ztdn(jk) - ptdbt(jk)) * prupd(jk)) * zreflect
   150)          pfd(jk,kw) = ptdbt(jk) + (ztdn(jk) - ptdbt(jk)+ &
   151)                       ptdbt(jk) * prup(jk) * prdnd(jk)) * zreflect
   152)       enddo
