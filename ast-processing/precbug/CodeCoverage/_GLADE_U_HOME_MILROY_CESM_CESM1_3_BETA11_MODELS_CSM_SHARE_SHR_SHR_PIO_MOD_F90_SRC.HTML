    62)   subroutine shr_pio_init1(ncomps, nlfilename, Global_Comm)
    63)     integer, intent(in) :: ncomps

    65)     integer, intent(inout) :: Global_Comm

    67) 
    68)     integer :: i, pio_root, pio_stride, pio_numiotasks, pio_iotype

    71)     integer :: pelist(3,1)

    73)     call shr_pio_read_default_namelist(nlfilename, Global_Comm, pio_stride, pio_root, pio_numiotasks, &
    74)          pio_iotype, pio_async_interface)
    75) 
    76)     io_comm = MPI_COMM_NULL
    77)     allocate(pio_comp_settings(ncomps))
    78)     do i=1,ncomps
    79)        pio_comp_settings(i)%pio_root = pio_root
    80)        pio_comp_settings(i)%pio_stride = pio_stride
    81)        pio_comp_settings(i)%pio_numiotasks = pio_numiotasks
    82)        pio_comp_settings(i)%pio_iotype = pio_iotype
    83)     end do

    89)        pelist(1,1) = pio_root
    90)        pelist(2,1) = pio_root + (pio_numiotasks-1)*pio_stride
    91)        pelist(3,1) = pio_stride
    92) 
    93)        call mpi_comm_group(GLOBAL_COMM, mpigrp_world, ierr)

   102)        Global_COMM=mpicom
   103) 
   104)        print *,__FILE__,__LINE__,subname, ' complete'

   353)   subroutine shr_pio_read_default_namelist(nlfilename, Comm, pio_stride, pio_root, pio_numiotasks, &
   354)        pio_iotype, pio_async_interface)
   355)     
   356)     character(len=*), intent(in) :: nlfilename
   357)     integer, intent(in) :: Comm
   358)     logical, intent(out) :: pio_async_interface
   359)     integer, intent(out) :: pio_stride, pio_root, pio_numiotasks, pio_iotype
   360) 
   361)     character(len=shr_kind_cs) :: pio_typename
   362)     character(*),parameter :: subName =   '(shr_pio_read_default_namelist) '
   363) 
   364)     integer :: iam, ierr, npes, unitn
   365)     logical :: iamroot
   366) 
   367)     namelist /pio_default_inparm/ pio_stride, pio_root, pio_numiotasks, &

   374)     call mpi_comm_size(Comm, npes, ierr)

   376) 
   377)     if(iam==0) then
   378)        iamroot=.true.
   379)     else
   380)        iamroot=.false.
   381)     end if
   382) 
   383)     !--------------------------------------------------------------------------
   384)     ! read io nml parameters 
   385)     !--------------------------------------------------------------------------
   386)     pio_stride   = -99 ! set based on pio_numiotasks value when initialized < 0
   387)     pio_numiotasks = -99 ! set based on pio_stride   value when initialized < 0
   388)     pio_root     = -99
   389)     pio_typename = 'nothing'
   390)     pio_blocksize= -99  ! io blocking size set internally in pio when < 0
   391)     pio_buffer_size_limit = -99 ! io task memory buffer maximum set internally in pio when < 0
   392)     pio_debug_level = 0 ! no debug info by default
   393)     pio_async_interface = .false.   ! pio tasks are a subset of component tasks
   394) 
   395)     if(iamroot) then
   396)        unitn=shr_file_getunit()
   397)        open( unitn, file=trim(nlfilename), status='old' , iostat=ierr)
   398)        if(ierr/=0) then
   399)           write(shr_log_unit,*) 'File ',trim(nlfilename),' not found, setting default values.'
   400)        else
   401)           ierr = 1
   402)           do while( ierr /= 0 )
   403)              read(unitn,nml=pio_default_inparm,iostat=ierr)

   405)                 call shr_sys_abort( subname//':: namelist read returns an'// &

   407)              end if
   408)           end do
   409)           close(unitn)
   410)           call shr_file_freeUnit( unitn )
   411) 
   412)           call shr_pio_getiotypefromname(pio_typename, pio_iotype, pio_iotype_netcdf)
   413)        end if
   414)     end if
   415) 
   416) 
   417) 
   418)     call shr_pio_namelist_set(npes, Comm, pio_stride, pio_root, pio_numiotasks, pio_iotype, iamroot)
   419)     call shr_mpi_bcast(pio_debug_level, Comm)
   420)     call shr_mpi_bcast(pio_blocksize, Comm)
   421)     call shr_mpi_bcast(pio_buffer_size_limit, Comm)
   422)     call shr_mpi_bcast(pio_async_interface, Comm)
   423)

   427)   subroutine shr_pio_read_component_namelist(nlfilename, Comm, pio_stride, pio_root, pio_numiotasks, pio_iotype)
   428)     character(len=*), intent(in) :: nlfilename

   430) 
   431)     integer, intent(inout) :: pio_stride, pio_root, pio_numiotasks, pio_iotype

   433)     integer :: unitn

   447)     call mpi_comm_size(Comm, npes, ierr)

   451)        iamroot=.true.

   466)     pio_numiotasks = -99 ! set based on pio_stride   value when initialized < 0

   479)              if (ierr < 0) then
   480)                 call shr_sys_abort( subname//':: namelist read returns an'// &
   481)                      ' end of file or end of record condition' )
   482)              end if
   483)           end do
   484)           close(unitn)
   485)           call shr_file_freeUnit( unitn )
   486) 
   487)           call shr_pio_getiotypefromname(pio_typename, pio_iotype, pio_default_iotype)
   488) 
   489)           if(pio_stride== -99) pio_stride = pio_default_stride
   490)           if(pio_root == -99) pio_root = pio_default_root
   491)           if(pio_numiotasks == -99) then
