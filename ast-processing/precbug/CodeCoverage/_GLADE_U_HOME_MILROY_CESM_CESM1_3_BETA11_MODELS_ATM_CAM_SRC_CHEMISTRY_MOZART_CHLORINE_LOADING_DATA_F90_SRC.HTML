   162)   subroutine chlorine_loading_advance( )
   163)     use cam_pio_utils, only : cam_pio_openfile
   164)     use pio, only : file_desc_t, pio_closefile, pio_inq_varid, pio_get_var
   165)     use physconst,    only : cday
   166)     use ioFileMod, only : getfil
   167) 
   168)     integer  :: year, month, day, sec
   169)     integer  :: index, i
   170)     integer  :: offset(1), count(1)
   171)     logical  :: do_adv, read_data
   172)     real(r8) :: time, delt
   173)     type(file_desc_t) :: file_id
   174)     integer :: data_vid, ierr
   175)     character(len=256) :: filen
   176) 
   177) 
   178)     if (.not.has_linoz_data) return
   179)     if ( fixed .and. initialized ) return
   180) 
   181)     index = -1
   182)     call get_model_time( time, year=year, month=month, day=day, seconds=sec )
   183) 
   184)     read_data = time > data_times(last_index) .or. .not.initialized
   185) 
   186)     if ( read_data ) then
   187) 
   188)        find_ndx: do i = last_index, ntimes
   189)           if ( data_times(i) - time > 1.e-6_r8 ) then
   190)              index = i-1
   191)              exit find_ndx
   192)           endif
   193)        enddo find_ndx
   194) 
   195)        last_index = index+1
   196) 
   197)        if ( index < 1 ) then
   198)           write(iulog,102) year,month,day,sec
   199)           call endrun('chlorine_loading_advance: failed to read data from '//trim(chlorine_loading_file))
   200)        endif
   201) 
   202)        ! get the surrounding time slices
   203)        offset = (/ index /)
   204)        count =  (/ nt /)
   205) 
   206)        call getfil( chlorine_loading_file, filen, 0 )
   207)        call cam_pio_openfile( file_id, filen, 0 )
   208)        if ( masterproc ) write(iulog,*)'chlorine_loading_advance: data file = ',trim(filen)
   209)        ierr = pio_inq_varid( file_id, 'chlorine_loading', data_vid )
   210)        ierr = pio_get_var( file_id, data_vid, offset, count, iloading )
   211)        call pio_closefile(file_id)
   212)     else
   213)        index = last_index - 1
   214)     endif
   215) 
   216)     delt = ( time - data_times(index) ) / ( data_times(index+1) - data_times(index) )
   217) 
   218)     ! this assures that FIXED data are b4b on restarts
   219)     if ( fixed ) then
   220)        delt = dble(int(delt*cday+.5_r8))/dble(cday)
   221)     endif
   222) 
   223)     chlorine_loading = iloading(1) + delt*( iloading(2) - iloading(1) )
   224)     
   225)     if ( masterproc ) then
   226)        write(iulog,101) year, month, day, sec, chlorine_loading
   227)     endif
   228)

   230) 102 FORMAT('chlorine_loading_advance: not able to find data for : ',i4.4,'-',i2.2,'-',i2.2,'-',i5.5)
