     1) module accumulMod

     3)   !-----------------------------------------------------------------------
     4)   ! !DESCRIPTION:
     5)   ! This module contains generic subroutines that can be used to
     6)   ! define, accumulate and extract  user-specified fields over
     7)   ! user-defined intervals. Each interval  and accumulation type is
     8)   ! unique to each field processed.
     9)   ! Subroutine [init_accumulator] defines the values of the accumulated
    10)   ! field data structure. Subroutine [update_accum_field] does
    11)   ! the actual accumulation for a given field.
    12)   ! Four types of accumulations are possible:
    13)   ! - Average over time interval. Time average fields are only
    14)   !   valid at the end of the averaging interval.
    15)   ! - Running mean over time interval. Running means are valid once the
    16)   !   length of the simulation exceeds the
    17)   ! - Running accumulation over time interval. Accumulated fields are
    18)   !   continuously accumulated. The trigger value "-99999." resets
    19)   !   the accumulation to zero.
    20)   !
    21)   ! !USES:
    22)   use shr_kind_mod, only: r8 => shr_kind_r8
    23)   use shr_sys_mod , only: shr_sys_abort
    24)   use clm_varctl  , only: iulog
    25)   !
    26)   ! !PUBLIC TYPES:
    27)   implicit none
    28)   save
    29)   !
    30)   ! !PUBLIC MEMBER FUNCTIONS:
    31)   public :: accumulRest          ! Write/read restart of accumulation fields
    32)   public :: init_accum_field     ! Initialize an accumulator field
    33)   public :: print_accum_fields   ! Print info about accumulator fields
    34)   public :: extract_accum_field  ! Extracts the current value of an accumulator field
    35)   public :: update_accum_field   ! Update the current value of an accumulator field
    36) 
    37)   interface extract_accum_field

    39)      module procedure extract_accum_field_ml ! Extract current val of multi-level accumulator field
    40)   end interface
    41)   interface update_accum_field               ! Updates the current value of an accumulator field
    42)      module procedure update_accum_field_sl  ! Update single-level accumulator field
    43)      module procedure update_accum_field_ml  ! Update multi-level accumulator field
    44)   end interface

    47)   ! !PRIVATE TYPES:

    78)   subroutine init_accum_field (name, units, desc, &
    79)        accum_type, accum_period, numlev, subgrid_type, init_value, type2d)
    80)     !
    81)     ! !DESCRIPTION:
    82)     ! Initialize accumulation fields. This subroutine sets:
    83)     ! o name  of accumulated field
    84)     ! o units of accumulated field
    85)     ! o accumulation type of accumulated field
    86)     ! o description of accumulated fields: accdes
    87)     ! o accumulation period for accumulated field (in iterations)
    88)     ! o initial value of accumulated field
    89)     !
    90)     ! !USES:
    91)     use shr_const_mod, only: SHR_CONST_CDAY
    92)     use clm_time_manager, only : get_step_size
    93)     use decompMod, only : get_proc_bounds, get_proc_global
    94)     !
    95)     ! !ARGUMENTS:
    96)     implicit none
    97)     character(len=*), intent(in)           :: name         !field name
    98)     character(len=*), intent(in)           :: units        !field units
    99)     character(len=*), intent(in)           :: desc         !field description
   100)     character(len=*), intent(in)           :: accum_type   !field type: tavg, runm, runa, ins
   101)     integer , intent(in)                   :: accum_period !field accumulation period
   102)     character(len=*), intent(in)           :: subgrid_type !["gridcell","landunit","column" or "pft"]
   103)     integer , intent(in)                   :: numlev       !number of vertical levels
   104)     real(r8), intent(in)                   :: init_value   !field initial or reset value
   105)     character(len=*), intent(in), optional :: type2d       !level type (optional) - needed if numlev > 1
   106)     !
   107)     ! !LOCAL VARIABLES:
   108)     integer :: nf           ! field index
   109)     integer :: beg1d,end1d  ! beggining and end subgrid indices
   110)     integer :: num1d        ! total number subgrid indices
   111)     integer :: begp, endp   ! per-proc beginning and ending pft indices
   112)     integer :: begc, endc   ! per-proc beginning and ending column indices
   113)     integer :: begl, endl   ! per-proc beginning and ending landunit indices
   114)     integer :: begg, endg   ! per-proc gridcell ending gridcell indices
   115)     integer :: begCohort, endCohort   ! per-proc beg end cohort indices
   116)     integer :: numg         ! total number of gridcells across all processors
   117)     integer :: numl         ! total number of landunits across all processors
   118)     integer :: numc         ! total number of columns across all processors
   119)     integer :: nump         ! total number of pfts across all processors
   120)     integer :: numCohort    ! total number of cohorts across all processors
   121)     !------------------------------------------------------------------------
   122) 
   123)     ! Determine necessary indices
   124) 
   125)     call get_proc_bounds(begg, endg, begl, endl, begc, endc, begp, endp, &
   126)          begCohort, endCohort )
   127)     call get_proc_global(numg, numl, numc, nump, numCohort)
   128) 
   129)     ! update field index
   130)     ! Consistency check that number of accumulated does not exceed maximum.
   131) 
   132)     naccflds = naccflds + 1
   133)     if (naccflds > max_accum) then
   134)        write(iulog,*) 'ACCUMULINIT error: user-defined accumulation fields ', &
   135)             'equal to ',naccflds,' exceeds max_accum'
   136)        call shr_sys_abort()
   137)     end if
   138)     nf = naccflds
   139) 
   140)     ! Note accumulation period must be converted from days
   141)     ! to number of iterations
   142) 
   143)     accum(nf)%name    = trim(name)
   144)     accum(nf)%units   = trim(units)
   145)     accum(nf)%desc    = trim(desc)
   146)     accum(nf)%acctype = trim(accum_type)
   147)     accum(nf)%initval = init_value
   148)     accum(nf)%period  = accum_period
   149)     if (accum(nf)%period < 0) then
   150)        accum(nf)%period = -accum(nf)%period * nint(SHR_CONST_CDAY) / get_step_size()
   151)     end if
   152) 
   153)     select case (trim(subgrid_type))
   154)     case ('gridcell')
   155)        beg1d = begg
   156)        end1d = endg
   157)        num1d = numg
   158)     case ('landunit')
   159)        beg1d = begl
   160)        end1d = endl
   161)        num1d = numl
   162)     case ('column')
   163)        beg1d = begc
   164)        end1d = endc
   165)        num1d = numc
   166)     case ('pft')
   167)        beg1d = begp
   168)        end1d = endp
   169)        num1d = nump
   170)     case default
   171)        write(iulog,*)'ACCUMULINIT: unknown subgrid type ',subgrid_type
   172)        call shr_sys_abort ()
   173)     end select
   174) 
   175)     accum(nf)%type1d = trim(subgrid_type)
   176)     accum(nf)%beg1d = beg1d
   177)     accum(nf)%end1d = end1d
   178)     accum(nf)%num1d = num1d
   179)     accum(nf)%numlev = numlev
   180) 
   181)     if (present(type2d)) then
   182)        accum(nf)%type2d = type2d
   183)     else
   184)        accum(nf)%type2d = ' '
   185)     end if
   186)     
   187)     ! Allocate and initialize accumulation field
   188) 
   189)     allocate(accum(nf)%val(beg1d:end1d,numlev))
   190)     accum(nf)%val(beg1d:end1d,numlev) = init_value
   191) 
   192)   end subroutine init_accum_field

   240)   subroutine extract_accum_field_sl (name, field, nstep)
   241)     !
   242)     ! !DESCRIPTION:
   243)     ! Extract single-level accumulated field.
   244)     ! This routine extracts the field values from the multi-level
   245)     ! accumulation field. It extracts the current value except if
   246)     ! the field type is a time average. In this case, an absurd value
   247)     ! is assigned to  indicate the time average is not yet valid.
   248)     !
   249)     ! !USES:
   250)     use clm_varcon, only : spval, ispval
   251)     !
   252)     ! !ARGUMENTS:
   253)     implicit none
   254)     character(len=*), intent(in) :: name     !field name
   255)     real(r8), pointer, dimension(:) :: field !field values for current time step
   256)     integer , intent(in) :: nstep            !timestep index
   257)     !
   258)     ! !LOCAL VARIABLES:
   259)     integer :: i,k,nf        !indices
   260)     integer :: beg,end         !subgrid beginning,ending indices
   261)     !------------------------------------------------------------------------
   262) 
   263)     ! find field index. return if "name" is not on list
   264) 
   265)     nf = 0
   266)     do i = 1, naccflds
   267)        if (name == accum(i)%name) nf = i
   268)     end do
   269)     if (nf == 0) then
   270)        write(iulog,*) 'EXTRACT_ACCUM_FIELD_SL error: field name ',name,' not found'
   271)        call shr_sys_abort
   272)     endif
   273) 
   274)     ! error check
   275) 
   276)     beg = accum(nf)%beg1d
   277)     end = accum(nf)%end1d
   278)     if (size(field,dim=1) /= end-beg+1) then
   279)        write(iulog,*)'ERROR in extract_accum_field for field ',accum(nf)%name
   280)        write(iulog,*)'size of first dimension of field is ',&
   281)             size(field,dim=1),' and should be ',end-beg+1
   282)        call shr_sys_abort
   283)     endif
   284) 
   285)     ! extract field
   286) 
   287)     if (accum(nf)%acctype == 'timeavg' .and. &
   288)          mod(nstep,accum(nf)%period) /= 0) then
   289)        do k = beg,end
   290)           field(k) = spval  !assign absurd value when avg not ready
   291)        end do
   292)     else
   293)        do k = beg,end
   294)           field(k) = accum(nf)%val(k,1)
   295)        end do
   296)     end if
   297) 
   298)   end subroutine extract_accum_field_sl
   299)

   372)   !------------------------------------------------------------------------
   373)   subroutine update_accum_field_sl (name, field, nstep)
   374)     !
   375)     ! !DESCRIPTION:
   376)     ! Accumulate single level field over specified time interval.
   377)     ! The appropriate field is accumulated in the array [accval].
   378)     !
   379)     ! !ARGUMENTS:
   380)     implicit none
   381)     character(len=*), intent(in) :: name     !field name
   382)     real(r8), pointer, dimension(:) :: field !field values for current time step
   383)     integer , intent(in) :: nstep            !time step index
   384)     !
   385)     ! !LOCAL VARIABLES:
   386)     integer :: i,k,nf              !indices
   387)     integer :: accper              !temporary accumulation period
   388)     integer :: beg,end             !subgrid beginning,ending indices
   389)     !------------------------------------------------------------------------
   390) 
   391)     ! find field index. return if "name" is not on list
   392) 
   393)     nf = 0
   394)     do i = 1, naccflds
   395)        if (name == accum(i)%name) nf = i
   396)     end do
   397)     if (nf == 0) then
   398)        write(iulog,*) 'UPDATE_ACCUM_FIELD_SL error: field name ',name,' not found'
   399)        call shr_sys_abort
   400)     endif
   401) 
   402)     ! error check
   403) 
   404)     beg = accum(nf)%beg1d
   405)     end = accum(nf)%end1d
   406)     if (size(field,dim=1) /= end-beg+1) then
   407)        write(iulog,*)'ERROR in UPDATE_ACCUM_FIELD_SL for field ',accum(nf)%name
   408)        write(iulog,*)'size of first dimension of field is ',size(field,dim=1),&
   409)             ' and should be ',end-beg+1
   410)        call shr_sys_abort
   411)     endif
   412) 
   413)     ! accumulate field
   414) 
   415)     if (accum(nf)%acctype /= 'timeavg' .AND. &
   416)         accum(nf)%acctype /= 'runmean' .AND. &
   417)         accum(nf)%acctype /= 'runaccum') then
   418)        write(iulog,*) 'UPDATE_ACCUM_FIELD_SL error: incorrect accumulation type'
   419)        write(iulog,*) ' was specified for field ',name
   420)        write(iulog,*)' accumulation type specified is ',accum(nf)%acctype
   421)        write(iulog,*)' only [timeavg, runmean, runaccum] are currently acceptable'
   422)        call shr_sys_abort()
   423)     end if
   424) 
   425) 
   426)     ! reset accumulated field value if necessary and  update
   427)     ! accumulation field
   428)     ! running mean never reset
   429) 
   430)     if (accum(nf)%acctype == 'timeavg') then
   431) 
   432)        !time average field reset every accumulation period
   433)        !normalize at end of accumulation period
   434) 
   435)        if ((mod(nstep,accum(nf)%period) == 1 .or. accum(nf)%period == 1) .and. (nstep /= 0))then
   436)           accum(nf)%val(beg:end,1) = 0._r8
   437)        end if
   438)        accum(nf)%val(beg:end,1) =  accum(nf)%val(beg:end,1) + field(beg:end)
   439)        if (mod(nstep,accum(nf)%period) == 0) then
   440)           accum(nf)%val(beg:end,1) = accum(nf)%val(beg:end,1) / accum(nf)%period
   441)        endif
   442) 
   443)     else if (accum(nf)%acctype == 'runmean') then
   444) 
   445)        !running mean - reset accumulation period until greater than nstep
   446) 
   447)        accper = min (nstep,accum(nf)%period)
   448)        accum(nf)%val(beg:end,1) = ((accper-1)*accum(nf)%val(beg:end,1) + field(beg:end)) / accper
   449) 
   450)     else if (accum(nf)%acctype == 'runaccum') then
   451) 
   452)        !running accumulation field reset at trigger -99999
   453) 
   454)        do k = beg,end
   455)           if (nint(field(k)) == -99999) then
   456)              accum(nf)%val(k,1) = 0._r8
   457)           end if
   458)        end do
   459)        accum(nf)%val(beg:end,1) = min(max(accum(nf)%val(beg:end,1) + field(beg:end), 0._r8), 99999._r8)
   460) 
   461)     end if
   462) 
   463)   end subroutine update_accum_field_sl

   476)     !
   477)     ! !LOCAL VARIABLES:
   478)     integer :: i,j,k,nf            !indices
   479)     integer :: accper              !temporary accumulation period
   480)     integer :: beg,end             !subgrid beginning,ending indices
   481)     integer :: numlev              !number of vertical levels
   482)     !------------------------------------------------------------------------

   485)

   511) 
   512)     ! accumulate field

   514)     if (accum(nf)%acctype /= 'timeavg' .AND. &
   515)         accum(nf)%acctype /= 'runmean' .AND. &
   516)         accum(nf)%acctype /= 'runaccum') then
   517)        write(iulog,*) 'UPDATE_ACCUM_FIELD_ML error: incorrect accumulation type'
   518)        write(iulog,*) ' was specified for field ',name

   520)        write(iulog,*)' only [timeavg, runmean, runaccum] are currently acceptable'
   521)        call shr_sys_abort()
   522)     end if
   523) 
   524)     ! accumulate field
   525) 
   526)     ! reset accumulated field value if necessary and  update
   527)     ! accumulation field
   528)     ! running mean never reset
   529) 
   530)     if (accum(nf)%acctype == 'timeavg') then
   531) 
   532)        !time average field reset every accumulation period
   533)        !normalize at end of accumulation period
   534) 
   535)        if ((mod(nstep,accum(nf)%period) == 1 .or. accum(nf)%period == 1) .and. (nstep /= 0))then
   536)           accum(nf)%val(beg:end,1:numlev) = 0._r8
   537)        endif
   538)        accum(nf)%val(beg:end,1:numlev) =  accum(nf)%val(beg:end,1:numlev) + field(beg:end,1:numlev)
   539)        if (mod(nstep,accum(nf)%period) == 0) then
   540)           accum(nf)%val(beg:end,1:numlev) = accum(nf)%val(beg:end,1:numlev) / accum(nf)%period
   541)        endif
   542) 
   543)     else if (accum(nf)%acctype == 'runmean') then
   544) 
   545)        !running mean - reset accumulation period until greater than nstep
   546) 
   547)        accper = min (nstep,accum(nf)%period)
   548)        accum(nf)%val(beg:end,1:numlev) = &
   549)             ((accper-1)*accum(nf)%val(beg:end,1:numlev) + field(beg:end,1:numlev)) / accper
   550) 
   551)     else if (accum(nf)%acctype == 'runaccum') then
   552) 
   553)        !running accumulation field reset at trigger -99999
   554)

   558)                 accum(nf)%val(k,j) = 0._r8
   559)              end if
   560)           end do
   561)        end do
   562)        accum(nf)%val(beg:end,1:numlev) = &
   563)             min(max(accum(nf)%val(beg:end,1:numlev) + field(beg:end,1:numlev), 0._r8), 99999._r8)
   564) 
   565)     end if
