    36)   subroutine ice_import( x2i )
    37) 
    38)     !-----------------------------------------------------
    39)     ! Arguments
    40)     real(r8), intent(inout) :: x2i(:,:)
    41)     !
    42)     ! Local variables
    43)     integer     :: i, j, iblk, n
    44)     integer     :: ilo, ihi, jlo, jhi !beginning and end of physical domain
    45)     type(block) :: this_block         ! block information for current block
    46)     integer,parameter                :: nflds=15,nfldv=6
    47)     real (kind=dbl_kind),allocatable :: aflds(:,:,:,:)
    48)     real (kind=dbl_kind)             :: workx, worky
    49)     logical (kind=log_kind)          :: first_call = .true.
    50)     !-----------------------------------------------------
    51) 
    52)     ! Note that the precipitation fluxes received  from the coupler
    53)     ! are in units of kg/s/m^2 which is what CICE requires.
    54)     ! Note also that the read in below includes only values needed
    55)     ! by the thermodynamic component of CICE.  Variables uocn, vocn,
    56)     ! ss_tltx, and ss_tlty are excluded. Also, because the SOM and
    57)     ! DOM don't  compute SSS.   SSS is not read in and is left at
    58)     ! the initilized value (see ice_flux.F init_coupler_flux) of
    59)     ! 34 ppt
    60) 
    61)     ! Use aflds to gather the halo updates of multiple fields
    62)     ! Need to separate the scalar from the vector halo updates
    63) 
    64)     allocate(aflds(nx_block,ny_block,nflds,nblocks))
    65)     aflds = c0
    66) 
    67)     n=0
    68)     do iblk = 1, nblocks
    69)        this_block = get_block(blocks_ice(iblk),iblk)         
    70)        ilo = this_block%ilo
    71)        ihi = this_block%ihi
    72)        jlo = this_block%jlo
    73)        jhi = this_block%jhi
    74) 
    75)        do j = jlo, jhi
    76)           do i = ilo, ihi
    77) 
    78)              n = n+1
    79)              aflds(i,j, 1,iblk)   = x2i(index_x2i_So_t,n)
    80)              aflds(i,j, 2,iblk)   = x2i(index_x2i_So_s,n)
    81)              aflds(i,j, 3,iblk)   = x2i(index_x2i_Sa_z,n)
    82)              aflds(i,j, 4,iblk)   = x2i(index_x2i_Sa_ptem,n)
    83)              aflds(i,j, 5,iblk)   = x2i(index_x2i_Sa_tbot,n)
    84)              aflds(i,j, 6,iblk)   = x2i(index_x2i_Sa_shum,n)
    85)              aflds(i,j, 7,iblk)   = x2i(index_x2i_Sa_dens,n)
    86)              aflds(i,j, 8,iblk)   = x2i(index_x2i_Fioo_q,n)
    87)              aflds(i,j, 9,iblk)   = x2i(index_x2i_Faxa_swvdr,n)
    88)              aflds(i,j,10,iblk)   = x2i(index_x2i_Faxa_swndr,n)
    89)              aflds(i,j,11,iblk)   = x2i(index_x2i_Faxa_swvdf,n)
    90)              aflds(i,j,12,iblk)   = x2i(index_x2i_Faxa_swndf,n)
    91)              aflds(i,j,13,iblk)   = x2i(index_x2i_Faxa_lwdn,n)
    92)              aflds(i,j,14,iblk)   = x2i(index_x2i_Faxa_rain,n)
    93)              aflds(i,j,15,iblk)   = x2i(index_x2i_Faxa_snow,n)
    94) 
    95)           enddo    !i
    96)        enddo    !j
    97) 
    98)     enddo        !iblk
    99) 
   100)     if (.not.prescribed_ice) then
   101)        call t_startf ('cice_imp_halo')
   102)        call ice_HaloUpdate(aflds, halo_info, field_loc_center, &
   103)             field_type_scalar)
   104)        call t_stopf ('cice_imp_halo')
   105)     endif
   106) 
   107)     !$OMP PARALLEL DO PRIVATE(iblk,i,j)
   108)     do iblk = 1, nblocks
   109)        do j = 1,ny_block
   110)           do i = 1,nx_block
   111)              sst  (i,j,iblk)   = aflds(i,j, 1,iblk)
   112)              sss  (i,j,iblk)   = aflds(i,j, 2,iblk)
   113)              zlvl (i,j,iblk)   = aflds(i,j, 3,iblk)
   114)              potT (i,j,iblk)   = aflds(i,j, 4,iblk)
   115)              Tair (i,j,iblk)   = aflds(i,j, 5,iblk)
   116)              Qa   (i,j,iblk)   = aflds(i,j, 6,iblk)
   117)              rhoa (i,j,iblk)   = aflds(i,j, 7,iblk)
   118)              frzmlt (i,j,iblk) = aflds(i,j, 8,iblk)
   119)              swvdr(i,j,iblk)   = aflds(i,j, 9,iblk)
   120)              swidr(i,j,iblk)   = aflds(i,j,10,iblk)
   121)              swvdf(i,j,iblk)   = aflds(i,j,11,iblk)
   122)              swidf(i,j,iblk)   = aflds(i,j,12,iblk)
   123)              flw  (i,j,iblk)   = aflds(i,j,13,iblk)
   124)              frain(i,j,iblk)   = aflds(i,j,14,iblk)
   125)              fsnow(i,j,iblk)   = aflds(i,j,15,iblk)
   126)           enddo    !i
   127)        enddo    !j
   128)     enddo        !iblk
   129)     !$OMP END PARALLEL DO
   130) 
   131)     deallocate(aflds)
   132)     allocate(aflds(nx_block,ny_block,nfldv,nblocks))
   133)     aflds = c0
   134) 
   135)     n=0
   136)     do iblk = 1, nblocks
   137)        this_block = get_block(blocks_ice(iblk),iblk)
   138)        ilo = this_block%ilo
   139)        ihi = this_block%ihi
   140)        jlo = this_block%jlo
   141)        jhi = this_block%jhi
   142) 
   143)        do j = jlo, jhi
   144)           do i = ilo, ihi
   145)              n = n+1
   146)              aflds(i,j, 1,iblk)   = x2i(index_x2i_So_u,n)
   147)              aflds(i,j, 2,iblk)   = x2i(index_x2i_So_v,n)
   148)              aflds(i,j, 3,iblk)   = x2i(index_x2i_Sa_u,n)
   149)              aflds(i,j, 4,iblk)   = x2i(index_x2i_Sa_v,n)
   150)              aflds(i,j, 5,iblk)   = x2i(index_x2i_So_dhdx,n)
   151)              aflds(i,j, 6,iblk)   = x2i(index_x2i_So_dhdy,n)
   152)           enddo
   153)        enddo
   154)     enddo
   155) 
   156)     if (.not.prescribed_ice) then
   157)        call t_startf ('cice_imp_halo')
   158)        call ice_HaloUpdate(aflds, halo_info, field_loc_center, &
   159)             field_type_vector)
   160)        call t_stopf ('cice_imp_halo')
   161)     endif
   162) 
   163)     !$OMP PARALLEL DO PRIVATE(iblk,i,j)
   164)     do iblk = 1, nblocks
   165)        do j = 1,ny_block
   166)           do i = 1,nx_block
   167)              uocn (i,j,iblk)   = aflds(i,j, 1,iblk)
   168)              vocn (i,j,iblk)   = aflds(i,j, 2,iblk)
   169)              uatm (i,j,iblk)   = aflds(i,j, 3,iblk)
   170)              vatm (i,j,iblk)   = aflds(i,j, 4,iblk)
   171)              ss_tltx(i,j,iblk) = aflds(i,j, 5,iblk)
   172)              ss_tlty(i,j,iblk) = aflds(i,j, 6,iblk)
   173)           enddo    !i
   174)        enddo    !j
   175)     enddo        !iblk
   176)     !$OMP END PARALLEL DO
   177) 
   178)     deallocate(aflds)
   179) 
   180)     !-------------------------------------------------------
   181)     ! Set aerosols from coupler 
   182)     !-------------------------------------------------------
   183) 
   184)     n=0
   185)     do iblk = 1, nblocks
   186)        this_block = get_block(blocks_ice(iblk),iblk)         
   187)        ilo = this_block%ilo
   188)        ihi = this_block%ihi
   189)        jlo = this_block%jlo
   190)        jhi = this_block%jhi
   191) 
   192)        do j = jlo, jhi
   193)           do i = ilo, ihi
   194) 
   195)              n = n+1
   196)              faero(i,j,1,iblk) = x2i(index_x2i_Faxa_bcphodry,n)
   197) 
   198)              faero(i,j,2,iblk) = x2i(index_x2i_Faxa_bcphidry,n) &
   199)                   + x2i(index_x2i_Faxa_bcphiwet,n)
   200)              ! Combine all of the dust into one category
   201)              faero(i,j,3,iblk) = x2i(index_x2i_Faxa_dstwet1,n) &
   202)                   + x2i(index_x2i_Faxa_dstdry1,n) &
   203)                   + x2i(index_x2i_Faxa_dstwet2,n) &
   204)                   + x2i(index_x2i_Faxa_dstdry2,n) &
   205)                   + x2i(index_x2i_Faxa_dstwet3,n) &
   206)                   + x2i(index_x2i_Faxa_dstdry3,n) &
   207)                   + x2i(index_x2i_Faxa_dstwet4,n) &
   208)                   + x2i(index_x2i_Faxa_dstdry4,n)
   209) 
   210)           enddo    !i
   211)        enddo    !j
   212) 
   213)     enddo        !iblk
   214) 
   215) 
   216)     !-----------------------------------------------------------------
   217)     ! rotate zonal/meridional vectors to local coordinates
   218)     ! compute data derived quantities
   219)     !-----------------------------------------------------------------
   220) 
   221)     ! Vector fields come in on T grid, but are oriented geographically
   222)     ! need to rotate to pop-grid FIRST using ANGLET
   223)     ! then interpolate to the U-cell centers  (otherwise we
   224)     ! interpolate across the pole)
   225)     ! use ANGLET which is on the T grid !
   226) 
   227)     call t_startf ('cice_imp_ocn')
   228)     !$OMP PARALLEL DO PRIVATE(iblk,i,j,workx,worky)
   229)     do iblk = 1, nblocks
   230) 
   231)        do j = 1,ny_block
   232)           do i = 1,nx_block
   233) 
   234)              ! ocean
   235)              workx      = uocn  (i,j,iblk) ! currents, m/s 
   236)              worky      = vocn  (i,j,iblk)
   237)              uocn(i,j,iblk) = workx*cos(ANGLET(i,j,iblk)) & ! convert to POP grid 
   238)                   + worky*sin(ANGLET(i,j,iblk))
   239)              vocn(i,j,iblk) = worky*cos(ANGLET(i,j,iblk)) &
   240)                   - workx*sin(ANGLET(i,j,iblk))
   241) 
   242)              workx      = ss_tltx  (i,j,iblk)           ! sea sfc tilt, m/m
   243)              worky      = ss_tlty  (i,j,iblk)
   244)              ss_tltx(i,j,iblk) = workx*cos(ANGLET(i,j,iblk)) & ! convert to POP grid 
   245)                   + worky*sin(ANGLET(i,j,iblk))
   246)              ss_tlty(i,j,iblk) = worky*cos(ANGLET(i,j,iblk)) &
   247)                   - workx*sin(ANGLET(i,j,iblk))
   248) 
   249)              sst(i,j,iblk) = sst(i,j,iblk) - Tffresh       ! sea sfc temp (C)
   250)              Tf (i,j,iblk) = -1.8_dbl_kind                 ! hardwired for NCOM
   251)              !         Tf (i,j,iblk) = -depressT*sss(i,j,iblk)       ! freezing temp (C)
   252)              !         Tf (i,j,iblk) = -depressT*max(sss(i,j,iblk),ice_ref_salinity)
   253) 
   254)           enddo
   255)        enddo
   256)     enddo
   257)     !$OMP END PARALLEL DO
   258)     call t_stopf ('cice_imp_ocn')
   259) 
   260)     ! Interpolate ocean dynamics variables from T-cell centers to 
   261)     ! U-cell centers.
   262) 
   263)     if (.not.prescribed_ice) then
   264)        call t_startf ('cice_imp_t2u')
   265)        call t2ugrid_vector(uocn)
   266)        call t2ugrid_vector(vocn)
   267)        call t2ugrid_vector(ss_tltx)
   268)        call t2ugrid_vector(ss_tlty)
   269)        call t_stopf ('cice_imp_t2u')
   270)     end if
   271) 
   272)     ! Atmosphere variables are needed in T cell centers in
   273)     ! subroutine stability and are interpolated to the U grid
   274)     ! later as necessary.
   275) 
   276)     call t_startf ('cice_imp_atm')
   277)     !$OMP PARALLEL DO PRIVATE(iblk,i,j,workx,worky)
   278)     do iblk = 1, nblocks
   279)        do j = 1, ny_block
   280)           do i = 1, nx_block
   281) 
   282)              ! atmosphere
   283)              workx      = uatm(i,j,iblk) ! wind velocity, m/s
   284)              worky      = vatm(i,j,iblk) 
   285)              uatm (i,j,iblk) = workx*cos(ANGLET(i,j,iblk)) & ! convert to POP grid
   286)                              + worky*sin(ANGLET(i,j,iblk))   ! note uatm, vatm, wind
   287)              vatm (i,j,iblk) = worky*cos(ANGLET(i,j,iblk)) & ! are on the T-grid here
   288)                   - workx*sin(ANGLET(i,j,iblk))
   289) 
   290)              wind (i,j,iblk) = sqrt(uatm(i,j,iblk)**2 + vatm(i,j,iblk)**2)
   291)              fsw  (i,j,iblk) = swvdr(i,j,iblk) + swvdf(i,j,iblk) &
   292)                              + swidr(i,j,iblk) + swidf(i,j,iblk)
   293)           enddo
   294)        enddo
   295)     enddo
   296)     !$OMP END PARALLEL DO
   297)     call t_stopf ('cice_imp_atm')
   298) 
   299)   end subroutine ice_import

   303)   subroutine ice_export( i2x )   
   304) 
   305)     !-----------------------------------------------------
   306)     !
   307)     ! Arguments
   308)     real(r8), intent(inout) :: i2x(:,:)
   309)     !
   310)     ! Local Variables
   311)     integer :: i, j, iblk, n, ij 
   312)     integer :: ilo, ihi, jlo, jhi !beginning and end of physical domain
   313)     integer (kind=int_kind)                                :: icells ! number of ocean/ice cells
   314)     integer (kind=int_kind), dimension (nx_block*ny_block) :: indxi  ! compressed indices in i
   315)     integer (kind=int_kind), dimension (nx_block*ny_block) :: indxj  ! compressed indices in i
   316) 
   317)     real (kind=dbl_kind), dimension(nx_block,ny_block,max_blocks) :: &
   318)          Tsrf  &      ! surface temperature
   319)          ,  tauxa &      ! atmo/ice stress
   320)          ,  tauya &
   321)          ,  tauxo &      ! ice/ocean stress
   322)          ,  tauyo &
   323)          ,  ailohi       ! fractional ice area
   324) 
   325)     real (kind=dbl_kind) :: &
   326)          workx, worky           ! tmps for converting grid
   327) 
   328)     type(block)        :: this_block                           ! block information for current block
   329)     logical :: flag
   330)     !-----------------------------------------------------
   331) 
   332)     flag=.false.
   333) 
   334)     !calculate ice thickness from aice and vice. Also
   335)     !create Tsrf from the first tracer (trcr) in ice_state.F
   336) 
   337)     !$OMP PARALLEL DO PRIVATE(iblk,i,j,workx,worky)
   338)     do iblk = 1, nblocks
   339)        do j = 1, ny_block
   340)           do i = 1, nx_block
   341) 
   342)              ! ice fraction
   343)              ailohi(i,j,iblk) = min(aice(i,j,iblk), c1)
   344) 
   345)              ! surface temperature
   346)              Tsrf(i,j,iblk)  = Tffresh + trcr(i,j,1,iblk)             !Kelvin (original ???)
   347) 
   348)              ! wind stress  (on POP T-grid:  convert to lat-lon)
   349)              workx = strairxT(i,j,iblk)                             ! N/m^2
   350)              worky = strairyT(i,j,iblk)                             ! N/m^2
   351)              tauxa(i,j,iblk) = workx*cos(ANGLET(i,j,iblk)) &
   352)                              - worky*sin(ANGLET(i,j,iblk))
   353)              tauya(i,j,iblk) = worky*cos(ANGLET(i,j,iblk)) &
   354)                              + workx*sin(ANGLET(i,j,iblk))
   355) 
   356)              ! ice/ocean stress (on POP T-grid:  convert to lat-lon)
   357)              workx = -strocnxT(i,j,iblk)                            ! N/m^2
   358)              worky = -strocnyT(i,j,iblk)                            ! N/m^2
   359)              tauxo(i,j,iblk) = workx*cos(ANGLET(i,j,iblk)) &
   360)                              - worky*sin(ANGLET(i,j,iblk))
   361)              tauyo(i,j,iblk) = worky*cos(ANGLET(i,j,iblk)) &
   362)                              + workx*sin(ANGLET(i,j,iblk))
   363) 
   364)           enddo
   365)        enddo
   366)     enddo
   367)     !$OMP END PARALLEL DO
   368) 
   369)     do iblk = 1, nblocks
   370)        do j = 1, ny_block
   371)           do i = 1, nx_block
   372)              if (tmask(i,j,iblk) .and. ailohi(i,j,iblk) < c0 ) then
   373)                 flag = .true.
   374)              endif
   375)           end do
   376)        end do
   377)     end do
   378)     if (flag) then
   379)        do iblk = 1, nblocks
   380)           do j = 1, ny_block
   381)              do i = 1, nx_block
   382)                 if (tmask(i,j,iblk) .and. ailohi(i,j,iblk) < c0 ) then
   383)                    write(nu_diag,*) &
   384)                         ' (ice) send: ERROR ailohi < 0.0 ',i,j,ailohi(i,j,iblk)
   385)                    call shr_sys_flush(nu_diag)
   386)                 endif
   387)              end do
   388)           end do
   389)        end do
   390)     endif
   391) 
   392)     ! Fill export state i2x_i
   393) 
   394)     i2x(:,:) = spval_dbl
   395) 
   396)     n=0
   397)     do iblk = 1, nblocks
   398)        this_block = get_block(blocks_ice(iblk),iblk)         
   399)        ilo = this_block%ilo
   400)        ihi = this_block%ihi
   401)        jlo = this_block%jlo
   402)        jhi = this_block%jhi
   403) 
   404)        do j = jlo, jhi
   405)           do i = ilo, ihi
   406) 
   407)              n = n+1
   408) 
   409)              !-------states-------------------- 
   410)              i2x(index_i2x_Si_ifrac ,n)    = ailohi(i,j,iblk)   
   411) 
   412)              if ( tmask(i,j,iblk) .and. ailohi(i,j,iblk) > c0 ) then
   413)                 !-------states-------------------- 
   414)                 i2x(index_i2x_Si_t     ,n)    = Tsrf(i,j,iblk)
   415)                 i2x(index_i2x_Si_avsdr ,n)    = alvdr(i,j,iblk)
   416)                 i2x(index_i2x_Si_anidr ,n)    = alidr(i,j,iblk)
   417)                 i2x(index_i2x_Si_avsdf ,n)    = alvdf(i,j,iblk)
   418)                 i2x(index_i2x_Si_anidf ,n)    = alidf(i,j,iblk)
   419)                 i2x(index_i2x_Si_u10  ,n)     = Uref(i,j,iblk)
   420)                 i2x(index_i2x_Si_tref  ,n)    = Tref(i,j,iblk)
   421)                 i2x(index_i2x_Si_qref  ,n)    = Qref(i,j,iblk)
   422)                 i2x(index_i2x_Si_snowh ,n)    = vsno(i,j,iblk) &
   423)                      / ailohi(i,j,iblk)
   424)
