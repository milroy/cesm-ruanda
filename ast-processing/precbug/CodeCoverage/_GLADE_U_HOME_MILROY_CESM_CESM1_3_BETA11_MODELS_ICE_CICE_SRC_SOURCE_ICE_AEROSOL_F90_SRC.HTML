    78) ! !ROUTINE: update_aerosol
    79) !
    80) ! !DESCRIPTION:
    81) !
    82) !  Increase aerosol in ice or snow surface due to deposition
    83) !
    84) ! !REVISION HISTORY: same as module
    85) !
    86) ! !INTERFACE:
    87) !
    88)       subroutine update_aerosol (nx_block, ny_block,  &
    89)                                 dt,       icells,     &
    90)                                 indxi,    indxj,      &
    91)                                 meltt,    melts,      &
    92)                                 meltb,    congel,     &
    93)                                 snoice,               &
    94)                                 fsnow,                &
    95)                                 trcrn,                &
    96)                                 aice_old,             &
    97)                                 vice_old, vsno_old,   &
    98)                                 vicen, vsnon, aicen,  &
    99)                                 faero, fsoot)
   100) !
   101) ! !USES:
   102) !
   103)       use ice_domain_size, only: max_ntrcr, nilyr, nslyr, n_aero, n_aeromx
   104)       use ice_state, only: nt_aero 
   105) !
   106) ! !INPUT/OUTPUT PARAMETERS:
   107) !
   108)       integer (kind=int_kind), intent(in) :: &
   109)          nx_block, ny_block, & ! block dimensions
   110)          icells                ! number of cells with ice present
   111) 
   112)       integer (kind=int_kind), dimension (nx_block*ny_block), &
   113)          intent(in) :: &
   114)          indxi, indxj     ! compressed indices for cells with ice
   115) 
   116)       real (kind=dbl_kind), intent(in) :: &
   117)          dt                    ! time step
   118) 
   119)       real (kind=dbl_kind), dimension(nx_block,ny_block), &
   120)          intent(in) :: &
   121)          meltt,    &
   122)          melts,    &
   123)          meltb,    &
   124)          congel,   &
   125)          snoice,   &
   126)          fsnow,    &
   127)          vicen,    &
   128)          vsnon,    &
   129)          aicen,    &
   130)          aice_old, &
   131)          vice_old, &
   132)          vsno_old 
   133) 
   134)       real (kind=dbl_kind), dimension(nx_block,ny_block,n_aeromx), &
   135)          intent(in) :: &
   136)          faero
   137) 
   138)       real (kind=dbl_kind), dimension(nx_block,ny_block,n_aeromx), &
   139)          intent(inout) :: &
   140)          fsoot
   141) 
   142)       real (kind=dbl_kind), dimension(nx_block,ny_block,max_ntrcr), &
   143)          intent(inout) :: &
   144)          trcrn
   145) 
   146) !
   147) !  local variables
   148) !
   149)       integer (kind=int_kind) :: i, j, ij, k
   150)       integer (kind=int_kind) :: n  ! print_points
   151) !
   152)       real (kind=dbl_kind), dimension(icells) :: &
   153)          dzssl, &
   154)          dzint, &
   155)          dzssli, &
   156)          dzinti
   157) 
   158)       real (kind=dbl_kind), dimension(icells) :: &
   159)         dhs_evap, dhi_evap,  &
   160)         dhs_melts, dhs_snoice, dhi_meltt, dhi_snoice, &
   161)         dhi_congel, dhi_meltb 
   162)       real (kind=dbl_kind), dimension(icells,n_aeromx) :: &
   163)         aerotot, aerotot0   ! for diagnostics 
   164) 
   165)       real (kind=dbl_kind) :: &
   166)          dzssl_new, &
   167)          dzint_new, &
   168)          dzssli_new, &
   169)          dzinti_new, &
   170)          dznew
   171) 
   172)       real (kind=dbl_kind), dimension(nx_block,ny_block,n_aeromx,2) :: &
   173)          aerosno, aeroice, &
   174)          aerosno0, aeroice0  ! for diagnostic prints
   175) 
   176)       real (kind=dbl_kind), dimension(nx_block,ny_block,n_aeromx) :: &
   177)          fsoot_old
   178) 
   179)       real (kind=dbl_kind) :: &
   180)          hs_old, hi_old, hslyr_old, hilyr_old, dhs, dhi, hs, hi, &
   181)          hslyr, hilyr, sloss1, sloss2 
   182)       real (kind=dbl_kind), dimension(n_aeromx) :: &
   183)          kscav, kscavsi
   184) 
   185) !MH These need to be the same as in the DE code. Put in a common place?
   186)       real (kind=dbl_kind) :: &
   187)         hi_ssl, hs_ssl
   188) 
   189)       data hs_ssl  / .040_dbl_kind /
   190)       data hi_ssl  / .050_dbl_kind /
   191)       data kscav   / .03_dbl_kind, .20_dbl_kind,&
   192)            .02_dbl_kind,.02_dbl_kind,.01_dbl_kind,.01_dbl_kind /
   193)       data kscavsi / .03_dbl_kind, .20_dbl_kind,&
   194)            .02_dbl_kind,.02_dbl_kind,.01_dbl_kind,.01_dbl_kind /
   195) 
   196)       aerosno(:,:,:,:) = c0
   197)       aeroice(:,:,:,:) = c0
   198)       aerosno0(:,:,:,:) = c0
   199)       aeroice0(:,:,:,:) = c0
   200)       fsoot_old(:,:,:) = fsoot(:,:,:)
   201) 
   202)       do ij = 1, icells
   203)          i = indxi(ij)
   204)          j = indxj(ij)
   205) 
   206)          hs_old=vsno_old(i,j)/aice_old(i,j)
   207)          hi_old=vice_old(i,j)/aice_old(i,j)
   208)          hslyr_old=hs_old/real(nslyr,kind=dbl_kind)
   209)          hilyr_old=hi_old/real(nilyr,kind=dbl_kind)
   210) 
   211)          dzssl(ij)=min(hslyr_old/c2,hs_ssl)
   212)          dzint(ij)=hs_old-dzssl(ij)
   213)          dzssli(ij)=min(hilyr_old/c2,hi_ssl)
   214)          dzinti(ij)=hi_old-dzssli(ij)
   215) 
   216)          if (aicen(i,j) > c0) then
   217)             hs = vsnon(i,j)/aicen(i,j)
   218)             hi = vicen(i,j)/aicen(i,j)
   219)             dhs_melts(ij)=-melts(i,j)/aicen(i,j)
   220)             dhi_snoice(ij)=snoice(i,j)/aicen(i,j)
   221)             dhs_snoice(ij)=dhi_snoice(ij)*rhoi/rhos
   222)             dhi_meltt(ij)=-meltt(i,j)/aicen(i,j)
   223)             dhi_meltb(ij)=-meltb(i,j)/aicen(i,j)
   224)             dhi_congel(ij)=congel(i,j)/aicen(i,j)
   225)          else
   226)             hs = vsnon(i,j)/aice_old(i,j)
   227)             hi = vicen(i,j)/aice_old(i,j)
   228)             dhs_melts(ij)=-melts(i,j)/aice_old(i,j)
   229)             dhi_snoice(ij)=snoice(i,j)/aice_old(i,j)
   230)             dhs_snoice(ij)=dhi_snoice(ij)*rhoi/rhos
   231)             dhi_meltt(ij)=-meltt(i,j)/aice_old(i,j)
   232)             dhi_meltb(ij)=-meltb(i,j)/aice_old(i,j)
   233)             dhi_congel(ij)=congel(i,j)/aice_old(i,j)
   234)          endif
   235) 
   236)          dhs_evap(ij)=hs-(hs_old+dhs_melts(ij)-dhs_snoice(ij)+&
   237)              fsnow(i,j)/rhos*dt)
   238)          dhi_evap(ij)=hi-(hi_old+dhi_meltt(ij)+dhi_meltb(ij)+ &
   239)              dhi_congel(ij)+dhi_snoice(ij))
   240)       enddo
   241) 
   242)       do ij = 1, icells
   243)        i = indxi(ij)
   244)        j = indxj(ij)
   245)        do k=1,n_aero
   246)          aerosno(i,j,k,:)=&
   247)           trcrn(i,j,nt_aero+(k-1)*4  :nt_aero+(k-1)*4+1)*vsno_old(i,j)    ! aerosol in snow
   248)          aeroice(i,j,k,:)=&
   249)           trcrn(i,j,nt_aero+(k-1)*4+2:nt_aero+(k-1)*4+3)*vice_old(i,j)    ! aerosol in ice
   250)          aerosno0(i,j,k,:)=aerosno(i,j,k,:)
   251)          aeroice0(i,j,k,:)=aeroice(i,j,k,:)
   252)          aerotot0(ij,k)=aerosno(i,j,k,2)+aerosno(i,j,k,1) &
   253)            +aeroice(i,j,k,2)+aeroice(i,j,k,1)
   254)        enddo
   255)       enddo
   256) 
   257) ! apply evaporation
   258)       do ij = 1, icells
   259)          i = indxi(ij)
   260)          j = indxj(ij)
   261)          dzint(ij)=dzint(ij) + min(dzssl(ij)+dhs_evap(ij),c0)
   262)          dzssl(ij)=max(dzssl(ij)+dhs_evap(ij),c0)
   263)          dzinti(ij)=dzinti(ij) + min(dzssli(ij)+dhi_evap(ij),c0)
   264)          dzssli(ij)=max(dzssli(ij)+dhi_evap(ij),c0)
   265)       enddo
   266) 
   267) !     basal ice growth
   268)       do ij = 1, icells
   269)          i = indxi(ij)
   270)          j = indxj(ij)
   271)          dzinti(ij)=dzinti(ij)+dhi_congel(ij)
   272)       enddo
   273) 
   274) !     surface snow melt
   275)       do ij = 1, icells
   276)         i = indxi(ij)
   277)         j = indxj(ij)
   278)         if (-dhs_melts(ij) > puny) then
   279)          do k=1,n_aero
   280)           sloss1=c0
   281)           sloss2=c0
   282)           if (dzssl(ij) > puny)  &
   283)            sloss1=kscav(k)*aerosno(i,j,k,1)  &
   284)                  *min(-dhs_melts(ij),dzssl(ij))/dzssl(ij)
   285)           aerosno(i,j,k,1)=aerosno(i,j,k,1)-sloss1
   286)           if (dzint(ij) > puny)  &
   287)            sloss2=kscav(k)*aerosno(i,j,k,2) &
   288)                  *max(-dhs_melts(ij)-dzssl(ij),c0)/dzint(ij)
   289)           aerosno(i,j,k,2)=aerosno(i,j,k,2)-sloss2
   290)           fsoot(i,j,k)=fsoot(i,j,k)+(sloss1+sloss2)/dt
   291)          enddo  ! n_aero
   292) 
   293) !      update snow thickness
   294)          dzint(ij)=dzint(ij)+min(dzssl(ij)+dhs_melts(ij),c0)
   295)          dzssl(ij)=max(dzssl(ij)+dhs_melts(ij),c0)
   296) 
   297)          if ( dzssl(ij) <= puny ) then ! ssl melts away
   298)           aerosno(i,j,:,2)=aerosno(i,j,:,1)+aerosno(i,j,:,2)
   299)           aerosno(i,j,:,1)=c0
   300)           dzssl(ij)=max(dzssl(ij),c0)
   301)          endif
   302)          if (dzint(ij) <= puny ) then   ! all snow melts away
   303)           aeroice(i,j,:,1)=&
   304)              aeroice(i,j,:,1)+aerosno(i,j,:,1)+aerosno(i,j,:,2)
   305)           aerosno(i,j,:,:)=c0
   306)           dzint(ij)=max(dzint(ij),c0)
   307)          endif
   308)         endif
   309)        enddo
   310) 
   311) !     surface ice melt
   312)       do ij = 1, icells
   313)         i = indxi(ij)
   314)         j = indxj(ij)
   315)         if (-dhi_meltt(ij) > puny) then
   316)          do k=1,n_aero
   317)           sloss1=c0
   318)           sloss2=c0
   319)           if (dzssli(ij) > puny)  &
   320)            sloss1=kscav(k)*aeroice(i,j,k,1)  &
   321)                  *min(-dhi_meltt(ij),dzssli(ij))/dzssli(ij)
   322)           aeroice(i,j,k,1)=aeroice(i,j,k,1)-sloss1
   323)           if (dzinti(ij) > puny)  &
   324)            sloss2=kscav(k)*aeroice(i,j,k,2)  &
   325)                  *max(-dhi_meltt(ij)-dzssli(ij),c0)/dzinti(ij)
   326)           aeroice(i,j,k,2)=aeroice(i,j,k,2)-sloss2
   327)           fsoot(i,j,k)=fsoot(i,j,k)+(sloss1+sloss2)/dt
   328)          enddo
   329) 
   330)          dzinti(ij)=dzinti(ij)+min(dzssli(ij)+dhi_meltt(ij),c0)
   331)          dzssli(ij)=max(dzssli(ij)+dhi_meltt(ij),c0)
   332)          if (dzssli(ij) <= puny) then   ! ssl ice melts away
   333)           do k=1,n_aero
   334)            aeroice(i,j,k,2)=aeroice(i,j,k,1)+aeroice(i,j,k,2)
   335)            aeroice(i,j,k,1)=c0
   336)           enddo
   337)           dzssli(ij)=max(dzssli(ij),c0)
   338)          endif
   339)          if (dzinti(ij) <= puny) then   ! all ice melts away
   340)           do k=1,n_aero
   341)            fsoot(i,j,k)=fsoot(i,j,k)  &
   342)                        +(aeroice(i,j,k,1)+aeroice(i,j,k,2))/dt
   343)            aeroice(i,j,k,:)=c0
   344)           enddo
   345)           dzinti(ij)=max(dzinti(ij),c0)
   346)          endif
   347)         endif
   348)        enddo
   349) 
   350) !     basal ice melt.  Assume all soot lost in basal melt
   351)        do ij = 1, icells
   352)         i = indxi(ij)
   353)         j = indxj(ij)
   354)         if (-dhi_meltb(ij) > puny) then
   355)          do k=1,n_aero
   356)           sloss1=c0
   357)           sloss2=c0
   358)           if (dzssli(ij) > puny)  &
   359)            sloss1=max(-dhi_meltb(ij)-dzinti(ij),c0)  &
   360)                  *aeroice(i,j,k,1)/dzssli(ij)
   361)           aeroice(i,j,k,1)=aeroice(i,j,k,1)-sloss1
   362)           if (dzinti(ij) > puny)  &
   363)            sloss2=min(-dhi_meltb(ij),dzinti(ij))  &
   364)                  *aeroice(i,j,k,2)/dzinti(ij)
   365)           aeroice(i,j,k,2)=aeroice(i,j,k,2)-sloss2
   366)           fsoot(i,j,k)=fsoot(i,j,k)+(sloss1+sloss2)/dt
   367)          enddo
   368) 
   369)          dzssli(ij) = dzssli(ij)+min(dzinti(ij)+dhi_meltb(ij), c0)
   370)          dzinti(ij) = max(dzinti(ij)+dhi_meltb(ij), c0)           
   371)         endif
   372)        enddo
   373) 
   374) !     snowfall
   375)        do ij = 1, icells
   376)          i = indxi(ij)
   377)          j = indxj(ij)
   378)          if (fsnow(i,j) > c0) &
   379)            dzssl(ij)=dzssl(ij)+fsnow(i,j)/rhos*dt
   380)       enddo 
   381) 
   382) !     snoice formation
   383)       do ij = 1, icells
   384)         i = indxi(ij)
   385)         j = indxj(ij)
   386)         if (dhs_snoice(ij) > puny) then
   387)          do k=1,n_aero
   388)           sloss1=c0
   389)           sloss2=c0
   390)           if (dzint(ij) > puny)  &
   391)            sloss2 = min(dhs_snoice(ij),dzint(ij))  &
   392)                   *aerosno(i,j,k,2)/dzint(ij)
   393)           aerosno(i,j,k,2) = aerosno(i,j,k,2) - sloss2
   394)           if (dzssl(ij) > puny)  &
   395)            sloss1 = max(dhs_snoice(ij)-dzint(ij),c0)  &
   396)                   *aerosno(i,j,k,1)/dzssl(ij)
   397)           aerosno(i,j,k,1) = aerosno(i,j,k,1) - sloss1
   398)           aeroice(i,j,k,1) = aeroice(i,j,k,1) &
   399)                            + (c1-kscavsi(k))*(sloss2+sloss1)
   400)           fsoot(i,j,k)=fsoot(i,j,k)+kscavsi(k)*(sloss2+sloss1)/dt
   401)          enddo
   402)          dzssl(ij)=dzssl(ij)-max(dhs_snoice(ij)-dzint(ij),c0)
   403)          dzint(ij)=max(dzint(ij)-dhs_snoice(ij),c0)
   404)          dzssli(ij)=dzssli(ij)+dhi_snoice(ij)
   405)         endif
   406)       enddo
   407) 
   408) !     aerosol deposition
   409)       do ij = 1, icells
   410)          i = indxi(ij)
   411)          j = indxj(ij)
   412)          if (aicen(i,j) > c0) then
   413)             hs = vsnon(i,j) / aicen(i,j)
   414)          else
   415)             hs = c0
   416)          endif
   417)          if (hs > hsmin) then    ! should this really be hsmin or 0? 
   418)                                  ! should use same hsmin value as in radiation
   419)           do k=1,n_aero
   420)            aerosno(i,j,k,1)=aerosno(i,j,k,1) &
   421)                            + faero(i,j,k)*dt*aicen(i,j)
   422)           enddo
   423)          else
   424)           do k=1,n_aero
   425)            aeroice(i,j,k,1)=aeroice(i,j,k,1) &
   426)                            + faero(i,j,k)*dt*aicen(i,j)
   427)           enddo
   428)          endif
   429)       enddo
   430) 
   431) !     redistribute aerosol within vertical layers
   432)       do ij = 1, icells
   433)          i = indxi(ij)
   434)          j = indxj(ij)
   435)          if (aicen(i,j) > c0) then
   436)             hs = vsnon(i,j) / aicen(i,j)     ! new snow thickness
   437)             hi = vicen(i,j) / aicen(i,j)     ! new ice thickness
   438)          else
   439)             hs = c0
   440)             hi = c0
   441)          endif
   442)          if (dzssl(ij) <= puny) then   ! nothing in SSL
   443)           do k=1,n_aero
   444)            aerosno(i,j,k,2)=aerosno(i,j,k,2)+aerosno(i,j,k,1)
   445)            aerosno(i,j,k,1)=c0
   446)           enddo
   447)          endif
   448)          if (dzint(ij) <= puny) then   ! nothing in Snow Int
   449)           do k=1,n_aero
   450)            aeroice(i,j,k,1)=aeroice(i,j,k,1)+aerosno(i,j,k,2)
   451)            aerosno(i,j,k,2)=c0
   452)           enddo
   453)          endif
   454)          if (dzssli(ij) <= puny) then   ! nothing in Ice SSL
   455)           do k=1,n_aero
   456)            aeroice(i,j,k,2)=aeroice(i,j,k,2)+aeroice(i,j,k,1)
   457)            aeroice(i,j,k,1)=c0
   458)           enddo
   459)          endif
   460) 
   461)          if (dzinti(ij) <= puny) then   ! nothing in Ice INT
   462)           do k=1,n_aero
   463)            fsoot(i,j,k)=fsoot(i,j,k)+&
   464)                (aeroice(i,j,k,1)+aeroice(i,j,k,2))/dt
   465)            aeroice(i,j,k,:)=c0
   466)           enddo
   467)          endif
   468) 
   469)          hslyr=hs/real(nslyr,kind=dbl_kind)
   470)          hilyr=hi/real(nilyr,kind=dbl_kind)
   471)          dzssl_new=min(hslyr/c2,hs_ssl)         ! ssl for snow
   472)          dzint_new=hs-dzssl_new
   473)          dzssli_new=min(hilyr/c2,hi_ssl)        ! ssl for ice
   474)          dzinti_new=hi-dzssli_new
   475)  
   476)          if (hs > hsmin) then
   477)           do k=1,n_aero
   478)            dznew=min(dzssl_new-dzssl(ij),c0)
   479)            sloss1=c0
   480)            if (dzssl(ij) > puny) &
   481)             sloss1=dznew*aerosno(i,j,k,1)/dzssl(ij) ! not neccesarily a loss term
   482)            dznew=max(dzssl_new-dzssl(ij),c0)
   483)            if (dzint(ij) > puny) &
