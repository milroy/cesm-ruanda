  2272) ** construct_tree: Build the parent->children tree starting with knowledge of
  2273) **                 parent list for each child.
  2274) **
  2275) ** Input arguments:
  2276) **   timerst: Linked list of timers
  2277) **   method:  method to be used to define the links
  2278) **
  2279) ** Return value: 0 (success) or GPTLerror (failure)
  2280) */
  2281) 
  2282) int construct_tree (Timer *timerst, Method method)
  2283) {
  2284)   Timer *ptr;       /* loop through linked list */
  2285)   Timer *pptr = 0;  /* parent (init to NULL to avoid compiler warning) */
  2286)   int nparent;      /* number of parents */
  2287)   int maxcount;     /* max calls by a single parent */
  2288)   int n;            /* loop over nparent */
  2289) 
  2290)   /*
  2291)   ** Walk the linked list to build the parent-child tree, using whichever
  2292)   ** mechanism is in place. newchild() will prevent loops.
  2293)   */
  2294) 
  2295)   for (ptr = timerst; ptr; ptr = ptr->next) {
  2296)     switch (method) {
  2297)     case GPTLfirst_parent:
  2298)       if (ptr->nparent > 0) {
  2299) 	pptr = ptr->parent[0];
  2300) 	if (newchild (pptr, ptr) != 0);
  2301)       }
  2302)       break;
  2303)     case GPTLlast_parent:
  2304)       if (ptr->nparent > 0) {
  2305) 	nparent = ptr->nparent;
  2306) 	pptr = ptr->parent[nparent-1];
  2307) 	if (newchild (pptr, ptr) != 0);
  2308)       }
  2309)       break;
  2310)     case GPTLmost_frequent:
  2311)       maxcount = 0;
  2312)       for (n = 0; n < ptr->nparent; ++n) {
  2313) 	if (ptr->parent_count[n] > maxcount) {
  2314) 	  pptr = ptr->parent[n];
  2315) 	  maxcount = ptr->parent_count[n];
  2316) 	}
  2317)       }
  2318)       if (maxcount > 0) {   /* not an orphan */
  2319) 	if (newchild (pptr, ptr) != 0);
  2320)       }
  2321)       break;
  2322)     case GPTLfull_tree:
  2323)       /* 
  2324)       ** Careful: this one can create *lots* of output!
  2325)       */
  2326)       for (n = 0; n < ptr->nparent; ++n) {
  2327) 	pptr = ptr->parent[n];
  2328) 	if (newchild (pptr, ptr) != 0);
  2329)       }
  2330)       break;
  2331)     default:
  2332)       return GPTLerror ("construct_tree: method %d is not known\n", method);
  2333)     }
  2334)   }
  2335)   return 0;
  2336) }

  2360) ** newchild: Add an entry to the children list of parent. Use function
  2361) **   is_descendant() to prevent infinite loops. 
  2362) **
  2363) ** Input arguments:
  2364) **   parent: parent node
  2365) **   child:  child to be added
  2366) **
  2367) ** Return value: 0 (success) or GPTLerror (failure)
  2368) */
  2369) 
  2370) static int newchild (Timer *parent, Timer *child)
  2371) {
  2372)   int nchildren;     /* number of children (temporary) */
  2373)   Timer **chptr;     /* array of pointers to children */
  2374)   int n;             /* loop over nchildren */
  2375) 
  2376)   static const char *thisfunc = "newchild";
  2377) 
  2378)   if (parent == child)
  2379)     return GPTLerror ("%s: child %s can't be a parent of itself\n", thisfunc, child->name);
  2380) 
  2381)   /*
  2382)   ** To allow construct_tree to be called multiple times, check that proposed child
  2383)   ** is not a known child
  2384)   */
  2385) 
  2386)   for (n = 0; n < parent->nchildren; ++n) {
  2387)      if (parent->children[n] == child){
  2388)         n = parent->nchildren + 1;
  2389)      }
  2390)   }
  2391)   if (n > parent->nchildren){
  2392)     return 0;
  2393)   }
  2394) 
  2395)   /*
  2396)   ** To guarantee no loops, ensure that proposed parent isn't already a descendant of 
  2397)   ** proposed child
  2398)   */
  2399) 
  2400)   if (is_descendant (child, parent)) {
  2401)     show_descendant (0, child, parent);
  2402)     return GPTLerror ("%s: loop detected: NOT adding %s to descendant list of %s. "
  2403) 		      "Proposed parent is in child's descendant path.\n",
  2404) 		      thisfunc, child->name, parent->name);
  2405)   }
  2406) 
  2407)   /* Safe to add the child to the parent's list of children */
  2408) 
  2409)   ++parent->nchildren;
  2410)   nchildren = parent->nchildren;
  2411)   chptr = (Timer **) realloc (parent->children, nchildren * sizeof (Timer *));
  2412)   if ( ! chptr)
  2413)     return GPTLerror ("%s: realloc error\n", thisfunc);
  2414)   parent->children = chptr;
  2415)   parent->children[nchildren-1] = child;
  2416) 
  2417)   return 0;
  2418) }

  2477) static int is_descendant (const Timer *node1, const Timer *node2)
