     4) ! !MODULE: seq_frac_mct -- handles surface fractions.
     5) !
     6) !  Fraction Notes: tcraig, august 2008
     7) !  Assumes is running on CPLID pes
     8) !
     9) !  the fractions fields are now afrac, ifrac, ofrac, lfrac, and lfrin.
    10) !    afrac = fraction of atm on a grid
    11) !    lfrac = fraction of lnd on a grid
    12) !    ifrac = fraction of ice on a grid
    13) !    ofrac = fraction of ocn on a grid
    14) !    lfrin = land fraction defined by the land model
    15) !    ifrad = fraction of ocn on a grid at last radiation time
    16) !    ofrad = fraction of ice on a grid at last radiation time
    17) !      afrac, lfrac, ifrac, and ofrac are the self-consistent values in the 
    18) !      system.  lfrin is the fraction on the land grid and is allowed to
    19) !      vary from the self-consistent value as descibed below.  ifrad
    20) !      and ofrad are needed for the swnet calculation.
    21) !  the fractions fields are defined for each grid in the fraction bundles as 
    22) !    needed as follows.
    23) !    character(*),parameter :: fraclist_a = 'afrac:ifrac:ofrac:lfrac:lfrin'
    24) !    character(*),parameter :: fraclist_o = 'afrac:ifrac:ofrac:ifrad:ofrad'
    25) !    character(*),parameter :: fraclist_i = 'afrac:ifrac:ofrac'
    26) !    character(*),parameter :: fraclist_l = 'afrac:lfrac:lfrin'
    27) !    character(*),parameter :: fraclist_g = 'gfrac'
    28) !    character(*),parameter :: fraclist_r = 'lfrac:rfrac'
    29) !
    30) !  we assume ocean and ice are on the same grids, same masks
    31) !  we assume ocn2atm and ice2atm are masked maps
    32) !  we assume lnd2atm is a global map
    33) !  we assume that the ice fraction evolves in time but that
    34) !    the land model fraction does not.  the ocean fraction then
    35) !    is just the complement of the ice fraction over the region
    36) !    of the ocean/ice mask.
    37) !  we assume that component domains are filled with the total 
    38) !    potential mask/fraction on that grid, but that the fractions
    39) !    sent at run time are always the relative fraction covered.
    40) !    for example, if an ice cell can be up to 50% covered in 
    41) !    ice and 50% land, then the ice domain should have a fraction
    42) !    value of 0.5 at that grid cell.  at run time though, the ice
    43) !    fraction will be between 0.0 and 1.0 meaning that grid cells
    44) !    is covered with between 0.0 and 0.5 by ice.  the "relative" fractions 
    45) !    sent at run-time are corrected by the model to be total fractions
    46) !    such that
    47) !  in general, on every grid,
    48) !              fractions_*(afrac) = 1.0
    49) !              fractions_*(ifrac) + fractions_*(ofrac) + fractions_*(lfrac) = 1.0
    50) !  where fractions_* are a bundle of fractions on a particular grid and 
    51) !    *frac (ie afrac) is the fraction of a particular component in the bundle.
    52) !
    53) !  fraclist_g and fraclist_r don't yet interact with atm, lnd, ice, ocn.
    54) !
    55) !  the fractions are computed fundamentally as follows (although the 
    56) !    detailed implementation might be slightly different)
    57) !  initialization (frac_init):
    58) !    afrac is set on all grids
    59) !      fractions_a(afrac) = 1.0
    60) !      fractions_o(afrac) = mapa2o(fractions_a(afrac))
    61) !      fractions_i(afrac) = mapa2i(fractions_a(afrac))
    62) !      fractions_l(afrac) = mapa2l(fractions_a(afrac))
    63) !    initially assume ifrac on all grids is zero
    64) !      fractions_*(ifrac) = 0.0
    65) !    fractions/masks provided by surface components
    66) !      fractions_o(ofrac) = dom_o(frac)  ! ocean "mask"
    67) !      fractions_l(lfrin) = dom_l(frac)  ! land model fraction
    68) !    then mapped to the atm model
    69) !      fractions_a(ofrac) = mapo2a(fractions_o(ofrac))
    70) !      fractions_a(lfrin) = mapl2a(fractions_l(lfrin))
    71) !    and a few things are then derived
    72) !      fractions_a(lfrac) = 1.0 - fractions_a(ofrac)
    73) !        this is truncated to zero for very small values (< 0.001)
    74) !        to attempt to preserve non-land gridcells.
    75) !      fractions_l(lfrac) = mapa2l(fractions_a(lfrac))
    76) !      fractions_r(lfrac) = mapl2r(fractions_l(lfrac))
    77) !
    78) !  run-time (frac_set):
    79) !    update fractions on ice grid
    80) !      fractions_i(ifrac) = i2x_i(Si_ifrac)  ! ice frac from ice model
    81) !      fractions_i(ofrac) = 1.0 - fractions_i(ifrac)
    82) !        note: the relative fractions are corrected to total fractions
    83) !      fractions_o(ifrac) = mapi2o(fractions_i(ifrac))
    84) !      fractions_o(ofrac) = mapi2o(fractions_i(ofrac))
    85) !      fractions_a(ifrac) = mapi2a(fractions_i(ifrac))
    86) !      fractions_a(ofrac) = mapi2a(fractions_i(ofrac))
    87) ! 
    88) !  fractions used in merging are as follows
    89) !    mrg_x2a uses fractions_a(lfrac,ofrac,ifrac)
    90) !    mrg_x2o needs to use fractions_o(ofrac,ifrac) normalized to one
    91) !      normalization happens in mrg routine
    92) !
    93) !  fraction corrections in mapping are as follows
    94) !    mapo2a uses *fractions_o(ofrac) and /fractions_a(ofrac)
    95) !    mapi2a uses *fractions_i(ifrac) and /fractions_a(ifrac)
    96) !    mapl2a uses *fractions_l(lfrin) and /fractions_a(lfrin)
    97) !    mapa2* should use *fractions_a(afrac) and /fractions_*(afrac) but this
    98) !      has been defered since the ratio always close to 1.0
    99) !
   100) !  budgets use the standard afrac, ofrac, ifrac, and lfrac to compute
   101) !
   102) !  fraction and domain checks
   103) !    initialization:
   104) !      dom_i = mapo2i(dom_o)  ! lat, lon, mask, area
   105) !      where fractions_a(lfrac) > 0.0, fractions_a(lfrin) is also > 0.0
   106) !         this ensures the land will provide data everywhere the atm needs it
   107) !         and allows the land frac to be subtlely different from the
   108) !         land fraction specified in the atm.
   109) !      dom_a = mapl2a(dom_l)  ! if atm/lnd same grids
   110) !      dom_a = mapo2a(dom_o)  ! if atm/ocn same grids
   111) !      dom_a = mapi2a(dom_i)  ! if atm/ocn same grids
   112) !      0.0-eps < fractions_*(*) < 1.0+eps
   113) !    run time:
   114) !      fractions_a(lfrac) + fractions_a(ofrac) + fractions_a(ifrac) ~ 1.0
   115) !      0.0-eps < fractions_*(*) < 1.0+eps
   116) !    
   117) !!
   118) !
   119) ! !REVISION HISTORY:
   120) !    2007-may-07 - M. Vertenstein - initial port to cpl7.
   121) !

   178)   !  real(r8),parameter :: eps_fracsum = 1.0e-12   ! allowed error in sum of fracs

   180)   !  real(r8),parameter :: eps_fraclim = 1.0e-03   ! truncation limit in fractions_a(lfrac)

   206) ! !IROUTINE: seq_frac_init 
   207) !
   208) ! !DESCRIPTION:
   209) !    Initialize fraction attribute vectors and necessary ocn/ice domain
   210) !    fraction input if appropriate
   211) !
   212) ! !REVISION HISTORY:
   213) !    2007-may-07 - M. Vertenstein - initial cpl7 version.
   214) !
   215) ! !INTERFACE: ------------------------------------------------------------------
   216) 
   217) subroutine seq_frac_init( infodata,         &
   218)      atm, ice, lnd, ocn, glc, rof, wav,     &
   219)      fractions_a, fractions_i, fractions_l, &
   220)      fractions_o, fractions_g, fractions_r, fractions_w)
   221) 
   222) ! !INPUT/OUTPUT PARAMETERS:
   223)    type(seq_infodata_type) , intent(in)    :: infodata
   224)    type(component_type)    , intent(in)    :: atm
   225)    type(component_type)    , intent(in)    :: ice
   226)    type(component_type)    , intent(in)    :: lnd
   227)    type(component_type)    , intent(in)    :: ocn
   228)    type(component_type)    , intent(in)    :: glc
   229)    type(component_type)    , intent(in)    :: rof
   230)    type(component_type)    , intent(in)    :: wav
   231)    type(mct_aVect)         , intent(inout) :: fractions_a   ! Fractions on atm grid/decomp
   232)    type(mct_aVect)         , intent(inout) :: fractions_i   ! Fractions on ice grid/decomp
   233)    type(mct_aVect)         , intent(inout) :: fractions_l   ! Fractions on lnd grid/decomp
   234)    type(mct_aVect)         , intent(inout) :: fractions_o   ! Fractions on ocn grid/decomp
   235)    type(mct_aVect)         , intent(inout) :: fractions_g   ! Fractions on glc grid/decomp
   236)    type(mct_aVect)         , intent(inout) :: fractions_r   ! Fractions on rof grid/decomp
   237)    type(mct_aVect)         , intent(inout) :: fractions_w   ! Fractions on wav grid/decomp
   238) !EOP
   239) 
   240)    !----- local -----
   241)    type(mct_ggrid), pointer    :: dom_a
   242)    type(mct_ggrid), pointer    :: dom_i
   243)    type(mct_ggrid), pointer    :: dom_l
   244)    type(mct_ggrid), pointer    :: dom_o
   245)    type(mct_ggrid), pointer    :: dom_g
   246)    type(mct_ggrid), pointer    :: dom_r
   247)    type(mct_ggrid), pointer    :: dom_w
   248) 
   249)    logical :: atm_present   ! .true. => atm is present
   250)    logical :: ice_present   ! .true. => ice is present
   251)    logical :: ocn_present   ! .true. => ocean is present
   252)    logical :: lnd_present   ! .true. => land is present
   253)    logical :: glc_present   ! .true. => glc is present
   254)    logical :: rof_present   ! .true. => rof is present
   255)    logical :: wav_present   ! .true. => wav is present
   256)    logical :: dead_comps    ! .true. => dead models present
   257) 
   258)    integer :: j,n            ! indices
   259)    integer :: ka, ki, kl, ko ! indices
   260)    integer :: kf, kk, kr     ! indices 
   261)    integer :: lsize          ! local size of ice av
   262)    integer :: debug_old      ! old debug value
   263) 
   264)    character(*),parameter :: fraclist_a = 'afrac:ifrac:ofrac:lfrac:lfrin'
   265)    character(*),parameter :: fraclist_o = 'afrac:ifrac:ofrac:ifrad:ofrad'
   266)    character(*),parameter :: fraclist_i = 'afrac:ifrac:ofrac'
   267)    character(*),parameter :: fraclist_l = 'afrac:lfrac:lfrin'
   268)    character(*),parameter :: fraclist_g = 'gfrac'
   269)    character(*),parameter :: fraclist_r = 'lfrac:rfrac'
   270)    character(*),parameter :: fraclist_w = 'wfrac'
   271) 
   272)    !----- formats -----
   273)    character(*),parameter :: subName = '(seq_frac_init) '
   274) 
   275) !-------------------------------------------------------------------------------
   276) !
   277) !-------------------------------------------------------------------------------
   278) 
   279)    call seq_infodata_getData(infodata, &
   280)         atm_present=atm_present,       &
   281)         lnd_present=lnd_present,       &
   282)         rof_present=rof_present,       &
   283)         ice_present=ice_present,       &
   284)         ocn_present=ocn_present,       &
   285)         glc_present=glc_present,       &
   286)         wav_present=wav_present,       &
   287)         dead_comps=dead_comps)
   288) 
   289)    dom_a => component_get_dom_cx(atm)   
   290)    dom_l => component_get_dom_cx(lnd)   
   291)    dom_i => component_get_dom_cx(ice)   
   292)    dom_o => component_get_dom_cx(ocn)   
   293)    dom_r => component_get_dom_cx(rof)   
   294)    dom_g => component_get_dom_cx(glc)   
   295)    dom_w => component_get_dom_cx(wav)   
   296) 
   297)    debug_old = seq_frac_debug
   298)    seq_frac_debug = 2
   299) 
   300)    ! Initialize fractions on atm grid/decomp (initialize ice fraction to zero)
   301) 
   302)    if (atm_present) then
   303)       lSize = mct_aVect_lSize(dom_a%data)
   304)       call mct_aVect_init(fractions_a,rList=fraclist_a,lsize=lsize)
   305)       call mct_aVect_zero(fractions_a)
   306) 
   307)       ka = mct_aVect_indexRa(fractions_a,"afrac",perrWith=subName)
   308)       fractions_a%rAttr(ka,:) = 1.0_r8
   309)    endif
   310) 
   311)    ! Initialize fractions on glc grid decomp, just an initial "guess", updated later
   312) 
   313)    if (glc_present) then
   314)       lSize = mct_aVect_lSize(dom_g%data)
   315)       call mct_aVect_init(fractions_g,rList=fraclist_g,lsize=lsize)
   316)       call mct_aVect_zero(fractions_g)
   317)       fractions_g%rAttr(:,:) = 1.0_r8
   318)    end if
   319)       
   320)    ! Initialize fractions on land grid decomp, just an initial "guess", updated later
   321) 
   322)    if (lnd_present) then
   323)       lSize = mct_aVect_lSize(dom_l%data)
   324)       call mct_aVect_init(fractions_l,rList=fraclist_l,lsize=lsize)
   325)       call mct_aVect_zero(fractions_l)
   326)       
   327)       kk = mct_aVect_indexRA(fractions_l,"lfrin",perrWith=subName)
   328)       kf = mct_aVect_indexRA(dom_l%data ,"frac" ,perrWith=subName)
   329)       fractions_l%rAttr(kk,:) = dom_l%data%rAttr(kf,:)
   330) 
   331)       if (atm_present) then
   332)          mapper_l2a => prep_atm_get_mapper_Fl2a()
   333)          mapper_a2l => prep_lnd_get_mapper_Fa2l()
   334)          call seq_map_map(mapper_l2a, fractions_l, fractions_a, fldlist='lfrin', norm=.false.)
   335)          call seq_map_map(mapper_a2l, fractions_a, fractions_l, fldlist='afrac', norm=.false.)
   336)       endif
   337) 
   338)    end if
   339)       
   340)    ! Initialize fractions on ice grid/decomp (initialize ice fraction to zero)
   341) 
   342)    if (rof_present) then
   343)       lSize = mct_aVect_lSize(dom_r%data)
   344)       call mct_aVect_init(fractions_r,rList=fraclist_r,lsize=lsize)
   345)       call mct_aVect_zero(fractions_r)
   346) 
   347)       kr = mct_aVect_indexRa(fractions_r,"rfrac",perrWith=subName)	
   348)       kf = mct_aVect_indexRA(dom_r%data ,"frac" ,perrWith=subName)
   349)       fractions_r%rAttr(kr,:) = dom_r%data%rAttr(kf,:)
   350)    end if
   351) 
   352)    ! Initialize fractions on wav grid decomp, just an initial "guess", updated later
   353) 
   354)    if (wav_present) then
   355)       lSize = mct_aVect_lSize(dom_w%data)
   356)       call mct_aVect_init(fractions_w,rList=fraclist_w,lsize=lsize)
   357)       call mct_aVect_zero(fractions_w)
   358)       fractions_w%rAttr(:,:) = 1.0_r8
   359)    end if
   360) 
   361)    ! Initialize fractions on ice grid/decomp (initialize ice fraction to zero)
   362) 
   363)    if (ice_present) then
   364)       lSize = mct_aVect_lSize(dom_i%data)
   365)       call mct_aVect_init(fractions_i,rList=fraclist_i,lsize=lsize)
   366)       call mct_aVect_zero(fractions_i)
   367)       
   368)       ko = mct_aVect_indexRa(fractions_i,"ofrac",perrWith=subName)	
   369)       kf = mct_aVect_indexRA(dom_i%data ,"frac" ,perrWith=subName)
   370)       fractions_i%rAttr(ko,:) = dom_i%data%rAttr(kf,:)
   371) 
   372)       if (atm_present) then
   373)          mapper_i2a => prep_atm_get_mapper_Fi2a()
   374)          call seq_map_map(mapper_i2a,fractions_i,fractions_a,fldlist='ofrac',norm=.false.)
   375)       endif
   376)    end if
   377) 
   378)    ! Initialize fractions on ocean grid/decomp (initialize ice fraction to zero)
   379)    ! These are initialize the same as for ice
   380) 
   381)    if (ocn_present) then
   382)       lSize = mct_aVect_lSize(dom_o%data)
   383)       call mct_aVect_init(fractions_o,rList=fraclist_o,lsize=lsize)
   384)       call mct_aVect_zero(fractions_o)
   385)       
   386)       if (ice_present) then
   387)          mapper_i2o => prep_ocn_get_mapper_SFi2o()
   388)          call seq_map_map(mapper_i2o,fractions_i,fractions_o,fldlist='ofrac',norm=.false.)
   389)       else
   390)          ko = mct_aVect_indexRa(fractions_o,"ofrac",perrWith=subName)	
   391)          kf = mct_aVect_indexRA(dom_o%data ,"frac" ,perrWith=subName)
   392)          fractions_o%rAttr(ko,:) = dom_o%data%rAttr(kf,:)
   393)          mapper_o2a => prep_atm_get_mapper_Fo2a()
   394)          call seq_map_map(mapper_o2a, fractions_o, fractions_a, fldlist='ofrac',norm=.false.)
   395)       endif
   396) 
   397)       if (atm_present) then
   398)          mapper_a2o => prep_ocn_get_mapper_Fa2o()
   399)          call seq_map_map(mapper_a2o, fractions_a, fractions_o, fldlist='afrac',norm=.false.)
   400)       endif
   401)       if (ice_present) then
   402)           ! --- this should be an atm2ice call above, but atm2ice doesn't work
   403)          mapper_o2i => prep_ice_get_mapper_SFo2i()
   404)          call seq_map_map(mapper_o2i,fractions_o,fractions_i,fldlist='afrac',norm=.false.)
   405)       endif
   406)    end if
   407) 
   408)    ! --- Set ofrac and lfrac on atm grid.  These should actually be mapo2a of
   409)    !     ofrac and lfrac but we can't map lfrac from o2a due to masked mapping
   410)    !     weights.  So we have to settle for a residual calculation that is
   411)    !     truncated to zero to try to preserve "all ocean" cells.
   412) 
   413)    if (atm_present) then
   414)       ka = mct_aVect_indexRa(fractions_a,"afrac",perrWith=subName)
   415)       ki = mct_aVect_indexRa(fractions_a,"ifrac",perrWith=subName)	
   416)       kl = mct_aVect_indexRa(fractions_a,"lfrac",perrWith=subName)
   417)       ko = mct_aVect_indexRa(fractions_a,"ofrac",perrWith=subName)	
   418)       kk = mct_aVect_indexRa(fractions_a,"lfrin",perrWith=subName)	
   419)       lSize = mct_aVect_lSize(fractions_a)
   420) 
   421)       if (ice_present .or. ocn_present) then
   422)          do n = 1,lsize
   423)             fractions_a%rAttr(kl,n) = 1.0_r8 - fractions_a%rAttr(ko,n)
   424)             if (abs(fractions_a%rAttr(kl,n)) < eps_fraclim) then
   425)                fractions_a%rAttr(kl,n) = 0.0_r8
