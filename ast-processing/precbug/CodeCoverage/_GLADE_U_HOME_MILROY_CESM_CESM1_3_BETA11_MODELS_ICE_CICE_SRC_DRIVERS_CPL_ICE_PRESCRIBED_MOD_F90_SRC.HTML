     4) ! !MODULE: ice_prescribed_mod - Prescribed Ice Model

     6) ! !DESCRIPTION:
     7) !
     8) ! The prescribed ice model reads in ice concentration data from a netCDF
     9) ! file.  Ice thickness, temperature, the ice temperature profile are
    10) ! prescribed.  Air/ice fluxes are computed to get surface temperature,
    11) ! Ice/ocean fluxes are set to zero, and ice dynamics are not calculated.
    12) ! Regridding and data cycling capabilities are included.

   114) ! !DESCRIPTION:
   115) !    Prescribed ice initialization - needed to

   129) !EOP
   130)    !----- Local ------
   131)    integer(kind=int_kind) :: nml_error ! namelist i/o error flag

   133)    character(len=8)       :: fillalgo

   141)         stream_year_last  ,  &

   143)         stream_fldFileName,  &
   144)         stream_domTvarName,  &
   145)         stream_domXvarName,  &
   146)         stream_domYvarName,  &
   147)         stream_domAreaName,  &
   148)         stream_domMaskName,  &
   149)         stream_domFileName,  &
   150)         stream_mapread,      &
   151)         prescribed_ice_fill
   152) 
   153)    ! default values for namelist
   154)    prescribed_ice         = .false.          ! if true, prescribe ice
   155)    stream_year_first      = 1                ! first year in  pice stream to use
   156)    stream_year_last       = 1                ! last  year in  pice stream to use
   157)    model_year_align       = 1                ! align stream_year_first with this model year
   158)    stream_fldVarName      = 'ice_cov'
   159)    stream_fldFileName(:)  = ' '
   160)    stream_domTvarName     = 'time'
   161)    stream_domXvarName     = 'lon'
   162)    stream_domYvarName     = 'lat'
   163)    stream_domAreaName     = 'area'
   164)    stream_domMaskName     = 'mask'
   165)    stream_domFileName     = ' '
   166)    stream_mapread         = 'NOT_SET'
   167)    prescribed_ice_fill    = .false.          ! true if pice data fill required
   168)

   170)    call get_fileunit(nu_nml)
   171)    if (my_task == master_task) then
   172)       open (nu_nml, file=nml_filename, status='old',iostat=nml_error)
   173)       if (nml_error /= 0) then
   174)          nml_error = -1
   175)       else
   176)          nml_error =  1
   177)       endif

   179)          read(nu_nml, nml=ice_prescribed_nml,iostat=nml_error)
   180)       end do
   181)       if (nml_error == 0) close(nu_nml)
   182)    endif
   183)    call release_fileunit(nu_nml)
   184)    call broadcast_scalar(nml_error,master_task)
   185)    if (nml_error /= 0) then
   186)       call abort_ice ('ice: Namelist read error in ice_prescribed_mod')
   187)    endif
   188) 
   189)    call broadcast_scalar(prescribed_ice,master_task)
   190) 
   191)    ! *** If not prescribed ice then return ***
   192)    if (.not. prescribed_ice) RETURN
   193) 
   194)    call broadcast_scalar(model_year_align,master_task)
   195)    call broadcast_scalar(stream_year_first,master_task)
   196)    call broadcast_scalar(stream_year_last,master_task)
   197)    call broadcast_scalar(stream_fldVarName,master_task)
   198)    call broadcast_scalar(stream_domTvarName,master_task)
   199)    call broadcast_scalar(stream_domXvarName,master_task)
   200)    call broadcast_scalar(stream_domYvarName,master_task)
   201)    call broadcast_scalar(stream_domAreaName,master_task)
   202)    call broadcast_scalar(stream_domMaskName,master_task)
   203)    call broadcast_scalar(stream_domFileName,master_task)
   204)    call broadcast_scalar(stream_mapread,master_task)
   205)    call broadcast_scalar(prescribed_ice_fill,master_task)
   206)    call mpi_bcast(stream_fldFileName, len(stream_fldFileName(1))*NFilesMaximum, &
   207)         MPI_CHARACTER, 0, MPI_COMM_ICE, ierr)
   208) 
   209)    nFile = 0
   210)    do n=1,nFilesMaximum
   211)       if (stream_fldFileName(n) /= ' ') nFile = nFile + 1
   212)    end do
   213) 
   214)    ! Read shr_strdata_nml namelist
   215)    if (prescribed_ice_fill) then
   216)       fillalgo='nn'
   217)    else
   218)       fillalgo='none'
   219)    endif
   220) 
   221)    if (my_task == master_task) then
   222)       write(nu_diag,*) ' '
   223)       write(nu_diag,*) 'This is the prescribed ice coverage option.'
   224)       write(nu_diag,*) '  stream_year_first  = ',stream_year_first  
   225)       write(nu_diag,*) '  stream_year_last   = ',stream_year_last   
   226)       write(nu_diag,*) '  model_year_align   = ',model_year_align   
   227)       write(nu_diag,*) '  stream_fldVarName  = ',trim(stream_fldVarName)
   228)       do n = 1,nFile
   229)          write(nu_diag,*) '  stream_fldFileName = ',trim(stream_fldFileName(n)),n
   230)       end do
   231)       write(nu_diag,*) '  stream_domTvarName = ',trim(stream_domTvarName)
   232)       write(nu_diag,*) '  stream_domXvarName = ',trim(stream_domXvarName)
   233)       write(nu_diag,*) '  stream_domYvarName = ',trim(stream_domYvarName)
   234)       write(nu_diag,*) '  stream_domFileName = ',trim(stream_domFileName)
   235)       write(nu_diag,*) '  stream_mapread     = ',trim(stream_mapread)
   236)       write(nu_diag,*) '  stream_fillalgo    = ',trim(fillalgo)
   237)       write(nu_diag,*) ' '
   238)    endif
   239) 
   240)    call shr_strdata_create(sdat,name="prescribed_ice", &
   241)         mpicom=MPI_COMM_ICE, compid=compid, &
   242)         gsmap=gsmap, ggrid=dom,          &
   243)         nxg=nx_global,nyg=ny_global,     &
   244)         yearFirst=stream_year_first,     &
   245)         yearLast=stream_year_last,       &
   246)         yearAlign=model_year_align,      &
   247)         offset=0,                        &
   248)         domFilePath='',                  &
   249)         domFileName=trim(stream_domFileName), &
   250)         domTvarName=stream_domTvarName,  &
   251)         domXvarName=stream_domXvarName,  &
   252)         domYvarName=stream_domYvarName,  &
   253)         domAreaName=stream_domAreaName,  &
   254)         domMaskName=stream_domMaskName,  &
   255)         filePath='',                     &
   256)         filename=stream_fldFileName(1:nFile), &
   257)         fldListFile=stream_fldVarName,   &
   258)         fldListModel=stream_fldVarName,  &
   259)         pio_subsystem=shr_pio_getiosys(inst_name), &
   260)         pio_iotype=shr_pio_getiotype(inst_name),   &
   261)         fillalgo=trim(fillalgo),       &
   262)         calendar=trim(calendar_type),  &
   263)         mapread=trim(stream_mapread))
   264) 
   265)    if (my_task == master_task) then
   266)       call shr_strdata_print(sdat,'SPRESICE data')
   267)    endif
   268) 
   269)    !-----------------------------------------------------------------
   270)    ! For one ice category, set hin_max(1) to something big
   271)    !-----------------------------------------------------------------

   280) ! !IROUTINE: ice_prescribed_run -- Update ice coverage
   281) !
   282) ! !DESCRIPTION:
   283) !
   284) !  Finds two time slices bounding current model time, remaps if necessary
   285) !
   286) ! !REVISION HISTORY:
   287) !     2005-May-19 - J. Schramm - first version
   288) !     2009-Oct-15 - M. Vertenstein - update to new data model changes
   289) !
   290) ! !INTERFACE: -----------------------------------------------------------
   291) 
   292) subroutine ice_prescribed_run(mDateIn, secIn)
   293) 
   294) ! !USES:
   295) 
   296)    implicit none
   297) 
   298) ! !INPUT/OUTPUT PARAMETERS:
   299) 
   300)    integer(kind=int_kind), intent(in) :: mDateIn  ! Current model date (yyyymmdd)
   301)    integer(kind=int_kind), intent(in) :: secIn    ! Elapsed seconds on model date
   302) 
   303) !EOP
   304) 
   305)    integer(kind=int_kind) :: i,j,n,iblk       ! loop indices and counter
   306)    integer(kind=int_kind) :: ilo,ihi,jlo,jhi  ! beginning and end of physical domain
   307)    type (block)           :: this_block
   308)    real(kind=dbl_kind)    :: aice_max         ! maximun ice concentration
   309)    logical, save          :: first_time = .true.
   310)    character(*),parameter :: subName = "('ice_prescribed_run')"
   311)    character(*),parameter :: F00 = "('(ice_prescribed_run) ',a,2g20.13)"
   312)  
   313)    !------------------------------------------------------------------------
   314)    ! Interpolate to new ice coverage
   315)    !------------------------------------------------------------------------
   316) 
   317)    call shr_strdata_advance(sdat,mDateIn,SecIn,MPI_COMM_ICE,'cice_pice')
   318)    
   319)    ice_cov(:,:,:) = c0  ! This initializes ghost cells as well 
   320) 
   321)    n=0
   322)    do iblk = 1, nblocks
   323)       this_block = get_block(blocks_ice(iblk),iblk)         
   324)       ilo = this_block%ilo
   325)       ihi = this_block%ihi
   326)       jlo = this_block%jlo
   327)       jhi = this_block%jhi
   328)       
   329)       do j = jlo, jhi
   330)       do i = ilo, ihi
   331)          n = n+1
   332)          ice_cov(i,j,iblk) = sdat%avs(1)%rAttr(1,n)
   333)       end do
   334)       end do
   335)    end do
   336) 
   337)    !--------------------------------------------------------------------
   338)    ! Check to see that ice concentration is in fraction, not percent
   339)    !--------------------------------------------------------------------
   340)    if (first_time) then
   341)       aice_max = maxval(ice_cov)
   342) 
   343)       if (aice_max > c10) then
   344)          write(nu_diag,F00) "ERROR: Ice conc data must be in fraction, aice_max= ",&
   345)               aice_max
   346)          call abort_ice(subName)
   347)       end if
   348)       first_time = .false.
   349)    end if
   350) 
   351)    !-----------------------------------------------------------------
   352)    ! Set prescribed ice state and fluxes
   353)    !-----------------------------------------------------------------
   354) 
   355)    call ice_prescribed_phys()
   356) 
   357) end subroutine ice_prescribed_run

   362) ! !IROUTINE: ice_prescribed_phys -- set prescribed ice state and fluxes
   363) !
   364) ! !DESCRIPTION:
   365) !
   366) ! Set prescribed ice state using input ice concentration;
   367) ! set surface ice temperature to atmospheric value; use
   368) ! linear temperature gradient in ice to ocean temperature.
   369) !
   370) ! !REVISION HISTORY:
   371) !     2005-May-23 - J. Schramm - Updated with data models
   372) !     2004-July   - J. Schramm - Modified to allow variable snow cover
   373) !     2001-May    - B. P. Briegleb - Original version
   374) !
   375) ! !INTERFACE: ------------------------------------------------------------------
   376)  
   377) subroutine ice_prescribed_phys
   378) 
   379) ! !USES:
   380)  
   381)    use ice_flux
   382) !  use ice_grid, only : bound
   383)    use ice_state
   384)    use ice_itd, only  : aggregate
   385)    use ice_dyn_evp
   386) 
   387)    implicit none
   388)  
   389) ! !INPUT/OUTPUT PARAMETERS:
   390)  
   391) !EOP
   392) 
   393)    !----- Local ------
   394)    integer(kind=int_kind) :: layer    ! level index
   395)    integer(kind=int_kind) :: nc       ! ice category index
   396)    integer(kind=int_kind) :: i,j,k    ! longitude, latitude and level indices
   397)    integer(kind=int_kind) :: iblk
   398) 
   399)    real(kind=dbl_kind) :: slope     ! diff in underlying ocean tmp and ice surface tmp
   400)    real(kind=dbl_kind) :: Ti        ! ice level temperature
   401)    real(kind=dbl_kind) :: Tmlt      ! ice level melt temperature
   402)    real(kind=dbl_kind) :: qin_save(nilyr) 
   403)    real(kind=dbl_kind) :: qsn_save(nslyr)
   404)    real(kind=dbl_kind) :: hi        ! ice prescribed (hemispheric) ice thickness
   405)    real(kind=dbl_kind) :: hs        ! snow thickness
   406)    real(kind=dbl_kind) :: zn        ! normalized ice thickness
   407)    real(kind=dbl_kind) :: salin(nilyr)  ! salinity (ppt) 
   408) 
   409)    real(kind=dbl_kind), parameter :: nsal    = 0.407_dbl_kind
   410)    real(kind=dbl_kind), parameter :: msal    = 0.573_dbl_kind
   411)    real(kind=dbl_kind), parameter :: saltmax = 3.2_dbl_kind   ! max salinity at ice base (ppm)
   412) 
   413)    !-----------------------------------------------------------------
   414)    ! Initialize ice state
   415)    !-----------------------------------------------------------------
   416) 
   417)    ! TODO  - can we now get rid of the following???
   418) 
   419)    !  aicen(:,:,:,:) = c0
   420)    !  vicen(:,:,:,:) = c0
   421)    !  eicen(:,:,:,:) = c0
   422)    
   423)    !  do nc=1,ncat
   424)    !     trcrn(:,:,nt_Tsfc,nc,:) = Tf(:,:,:)
   425)    !  enddo
   426)    
   427)    !-----------------------------------------------------------------
   428)    ! Set ice cover over land to zero, not sure if this should be
   429)    ! be done earier, before time/spatial interp??????
   430)    !-----------------------------------------------------------------
   431)    do iblk = 1,nblocks
   432)    do j = 1,ny_block
   433)    do i = 1,nx_block
   434)       if (tmask(i,j,iblk)) then
   435)          if (ice_cov(i,j,iblk) .lt. eps04) ice_cov(i,j,iblk) = c0
   436)          if (ice_cov(i,j,iblk) .gt. c1)    ice_cov(i,j,iblk) = c1
   437)       else
   438)          ice_cov(i,j,iblk) = c0
   439)       end if
   440)    enddo
   441)    enddo
   442)    enddo
   443) 
   444)    do iblk = 1,nblocks
   445)    do j = 1,ny_block
   446)    do i = 1,nx_block
   447) 
   448)       if (tmask(i,j,iblk)) then   ! Over ocean points
   449) 
   450)          !--------------------------------------------------------------
   451)          ! Place ice where ice concentration > .0001
   452)          !--------------------------------------------------------------
   453)          if (ice_cov(i,j,iblk) >= eps04) then
   454) 
   455)             hi = 0.0_dbl_kind
   456)             !----------------------------------------------------------
   457)             ! Set ice thickness in each hemisphere
   458)             !----------------------------------------------------------
   459)             if(TLAT(i,j,iblk)*rad_to_deg > 40.0_dbl_kind) then
   460)               hi  = 2.0_dbl_kind
   461)             else if(TLAT(i,j,iblk)*rad_to_deg < -40.0_dbl_kind) then
   462)               hi  = 1.0_dbl_kind
   463)             end if
   464) 
   465)             !----------------------------------------------------------
   466)             ! All ice in appropriate thickness category
   467)             !----------------------------------------------------------
   468)             do nc = 1,ncat
   469) 
   470)               if(hin_max(nc-1) < hi .and. hi < hin_max(nc)) then
   471) 
   472)                   if (aicen(i,j,nc,iblk) > c0) then
   473)                      hs = vsnon(i,j,nc,iblk) / aicen(i,j,nc,iblk)
   474)                   else
   475)                      hs = c0
   476)                   endif
   477) 
   478)                   qin_save(:) = c0
   479)                   qsn_save(:) = c0
   480) 
   481)                   if (vicen(i,j,nc,iblk) > c0) then
   482)                      do k=1,nilyr
   483)                         qin_save(k) = eicen(i,j,ilyr1(nc)+k-1,iblk)         &
   484)                                     * real(nilyr,kind=dbl_kind)             &
   485)                                     / Vicen(i,j,nc,iblk)
   486)                      enddo
   487)                   endif
   488) 
   489)                   if (vsnon(i,j,nc,iblk) > c0) then
   490)                      do k=1,nslyr
   491)                         qsn_save(k) = esnon(i,j,slyr1(nc)+k-1,iblk)         &
   492)                                     * real(nslyr,kind=dbl_kind)             &
   493)                                     / vsnon(i,j,nc,iblk)
   494)                      enddo
   495)                   endif
   496) 
   497)                   aicen(i,j,nc,iblk) = ice_cov(i,j,iblk)
   498)                   vicen(i,j,nc,iblk) = hi*aicen(i,j,nc,iblk) 
   499)                   vsnon(i,j,nc,iblk) = hs*aicen(i,j,nc,iblk) 
   500) 
   501)                   ! remember enthalpy profile to compute energy
   502)                   do k=1,nilyr
   503)                      eicen(i,j,ilyr1(nc)+k-1,iblk)                          &
   504)                         = qin_save(k) * vicen(i,j,nc,iblk)                  &
   505)                         / real(nilyr,kind=dbl_kind)
   506)                   enddo
   507) 
   508)                   do k=1,nslyr
   509)                      esnon(i,j,slyr1(nc)+k-1,iblk)                          &
   510)                         = qsn_save(k) * vsnon(i,j,nc,iblk)                  &
   511)                         / real(nslyr,kind=dbl_kind)
   512)                   enddo
   513) 
   514)                   !---------------------------------------------------------
