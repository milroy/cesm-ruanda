   394)  function ErfcProbability(lat) result(prob)
   395) 
   396)       real (kind=dbl_kind), intent(in), &
   397)          dimension (nx_global,ny_global) :: lat

   399)       real (kind=real_kind),  &

   401) 
   402) 
   403)       real (kind=dbl_kind) :: ltmp,thetai,sigmai,arg
   404)       integer :: i,j
   405) 
   406)       
   407)       do j=1,ny_global
   408)       do i=1,nx_global
   409)          ltmp = lat(i,j)         
   410)          if(ltmp > 0.0d0) then 
   411)                    !---------------------
   412)            ! northern latitude
   413)            !---------------------
   414) !JMD           thetai = (70.0_dbl_kind/180.0_dbl_kind)*pi
   415) !JMD  For startup lots of random ice at low lattitudes
   416)            thetai = (55.0_dbl_kind/180.0_dbl_kind)*pi
   417)            sigmai = (5.0_dbl_kind/180.0_dbl_kind)*pi
   418)            arg = (thetai-ltmp)/sigmai

   420)            !---------------------
   421)            ! southern hemisphere
   422)            !---------------------
   423) !JMD           thetai = (60.0_dbl_kind/180.0_dbl_kind)*pi
   424) !JMD  For startup lots of random ice at low lattitudes

   426)            sigmai = (5.0_dbl_kind/180.0_dbl_kind)*pi
   427)            arg = (thetai+ltmp)/sigmai
   428) 	 endif
