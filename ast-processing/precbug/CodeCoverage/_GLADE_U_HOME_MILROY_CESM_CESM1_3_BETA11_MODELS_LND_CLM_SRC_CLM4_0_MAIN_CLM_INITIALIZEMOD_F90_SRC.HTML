    51) !
    52) ! !INTERFACE:
    53)   subroutine initialize1( )
    54) !
    55) ! !DESCRIPTION:
    56) ! Land model initialization.
    57) ! o Initializes run control variables via the [clm_inparm] namelist.
    58) ! o Reads surface data on model grid.
    59) ! o Defines the multiple plant types and fraction areas for each surface type.
    60) ! o Builds the appropriate subgrid <-> grid mapping indices and weights.
    61) ! o Set up parallel processing.
    62) ! o Initializes time constant variables.
    63) ! o Reads restart data for a restart or branch run.
    64) ! o Reads initial data and initializes the time variant variables for an initial run.
    65) ! o Initializes history file output.
    66) ! o Initializes river routing model.
    67) ! o Initializes accumulation variables.
    68) !
    69) ! !USES:
    70)     use clmtypeInitMod  , only : initClmtype
    71)     use clm_varpar      , only : maxpatch, clm_varpar_init
    72)     use clm_varctl      , only : fsurdat, fatmlndfrc, flndtopo, fglcmask, noland 
    73)     use pftvarcon       , only : pftconrd
    74)     use decompInitMod   , only : decompInit_lnd, decompInit_glcp
    75)     use decompMod       , only : get_proc_bounds
    76)     use domainMod       , only : domain_check, ldomain, domain_init
    77)     use surfrdMod       , only : surfrd_get_globmask, surfrd_get_grid, surfrd_get_topo, &
    78)                                  surfrd_get_data 
    79)     use controlMod      , only : control_init, control_print, nlfilename
    80)     use UrbanInputMod   , only : UrbanInput
    81)     use ncdio_pio       , only : ncd_pio_init
    82)     use clm_atmlnd      , only : init_atm2lnd_type, init_lnd2atm_type, clm_a2l, clm_l2a
    83)     use clm_glclnd      , only : init_glc2lnd_type, init_lnd2glc_type, clm_x2s, clm_s2x
    84)     use initGridCellsMod, only : initGridCells
    85) !
    86) ! !ARGUMENTS:
    87) !
    88) ! !REVISION HISTORY:
    89) ! Created by Gordon Bonan, Sam Levis and Mariana Vertenstein
    90) !
    91) !
    92) ! !LOCAL VARIABLES:
    93) !EOP
    94)     integer  :: ier                   ! error status
    95)     integer  :: i,j,n,k               ! loop indices
    96)     integer  :: nl                    ! gdc and glo lnd indices
    97)     integer  :: ns, ni, nj            ! global grid sizes
    98)     logical  :: isgrid2d              ! true => global grid is regular lat/lon
    99)     integer  :: begp, endp            ! clump beg and ending pft indices
   100)     integer  :: begc, endc            ! clump beg and ending column indices
   101)     integer  :: begl, endl            ! clump beg and ending landunit indices
   102)     integer  :: begg, endg            ! clump beg and ending gridcell indices
   103)     integer ,pointer  :: amask(:)     ! global land mask
   104)     character(len=32) :: subname = 'initialize1' ! subroutine name
   105) !-----------------------------------------------------------------------
   106) 
   107)     call t_startf('clm_init1')
   108) 
   109)     ! ------------------------------------------------------------------------
   110)     ! Initialize run control variables, timestep
   111)     ! ------------------------------------------------------------------------
   112) 
   113)     call header()
   114) 
   115)     if (masterproc) then
   116)        write(iulog,*) 'Attempting to initialize the land model .....'
   117)        write(iulog,*)
   118)        call shr_sys_flush(iulog)
   119)     endif
   120) 
   121)     call control_init()
   122)     call clm_varpar_init()
   123)     call ncd_pio_init()
   124) 
   125)     if (masterproc) call control_print()
   126) 
   127)     ! ------------------------------------------------------------------------
   128)     ! Read in global land grid and land mask (amask)- needed to set decomposition
   129)     ! ------------------------------------------------------------------------
   130) 
   131)     ! global memory for amask is allocate in surfrd_get_glomask - must be
   132)     ! deallocated below
   133)     if (masterproc) then
   134)        write(iulog,*) 'Attempting to read global land mask from ',trim(fatmlndfrc)
   135)        call shr_sys_flush(iulog)
   136)     endif
   137)     call surfrd_get_globmask(filename=fatmlndfrc, mask=amask, ni=ni, nj=nj)
   138) 
   139)     ! Exit early if no valid land points
   140)     if ( all(amask == 0) )then
   141)        if (masterproc) write(iulog,*) trim(subname)//': no valid land points do NOT run clm'
   142)        noland = .true.
   143)        return
   144)     end if
   145) 
   146)     ! Determine clm decomposition
   147) 
   148)     call decompInit_lnd(ni, nj, amask)
   149)     deallocate(amask)
   150) 
   151)     ! Get grid and land fraction (set ldomain)
   152) 
   153)     if (masterproc) then
   154)        write(iulog,*) 'Attempting to read ldomain from ',trim(fatmlndfrc)
   155)        call shr_sys_flush(iulog)
   156)     endif
   157)     if (create_glacier_mec_landunit) then
   158)        call surfrd_get_grid(ldomain, fatmlndfrc, fglcmask)
   159)     else
   160)        call surfrd_get_grid(ldomain, fatmlndfrc)
   161)     endif
   162)     if (masterproc) then
   163)        call domain_check(ldomain)
   164)     endif
   165)     ldomain%mask = 1  !!! TODO - is this needed?
   166) 
   167)     ! Get topo if appropriate (set ldomain%topo)
   168) 
   169)     if (flndtopo /= " ") then
   170)        if (masterproc) then
   171)           write(iulog,*) 'Attempting to read atm topo from ',trim(flndtopo)
   172)           call shr_sys_flush(iulog)
   173)        endif
   174)        call surfrd_get_topo(ldomain, flndtopo)  
   175)     endif
   176) 
   177)     ! Initialize urban model input (initialize urbinp data structure)
   178) 
   179)     call UrbanInput(mode='initialize')
   180) 
   181)     ! Allocate surface grid dynamic memory (for wtxy and vegxy arrays)
   182)     ! Allocate additional dynamic memory for glacier_mec topo and thickness
   183) 
   184)     call get_proc_bounds(begg, endg)
   185)     allocate (vegxy(begg:endg,maxpatch), wtxy(begg:endg,maxpatch), stat=ier)   
   186)     if (create_glacier_mec_landunit) then
   187)        allocate (topoxy(begg:endg,maxpatch), stat=ier)
   188)     else
   189)        allocate (topoxy(1,1), stat=ier)
   190)     endif
   191)     if (ier /= 0) then
   192)        write(iulog,*)'initialize allocation error'; call endrun()
   193)     endif
   194) 
   195)     ! Read list of PFTs and their corresponding parameter values
   196)     ! Independent of model resolution, Needs to stay before surfrd_get_data
   197) 
   198)     call pftconrd()
   199) 
   200)     ! Read surface dataset and set up vegetation type [vegxy] and 
   201)     ! weight [wtxy] arrays for [maxpatch] subgrid patches.
   202)     
   203)     call surfrd_get_data(ldomain, fsurdat)
   204) 
   205)     ! Determine decomposition of subgrid scale landunits, columns, pfts
   206) 
   207)     if (create_glacier_mec_landunit) then
   208)        call decompInit_glcp (ns, ni, nj, ldomain%glcmask)
   209)     else
   210)        call decompInit_glcp (ns, ni, nj)
   211)     endif
   212) 
   213)     ! Allocate memory and initialize values of clmtype data structures
   214) 
   215)     call initClmtype()
   216) 
   217)     ! Initialize atm->lnd, lnd->atm, glc->lnd and lnd->glc data structures
   218) 
   219)     call init_atm2lnd_type(begg, endg, clm_a2l)
   220)     call init_lnd2atm_type(begg, endg, clm_l2a)
   221)     if (create_glacier_mec_landunit) then
   222)        call init_glc2lnd_type(begg, endg, clm_x2s)
   223)        call init_lnd2glc_type(begg, endg, clm_s2x)
   224)     endif
   225) 
   226)     ! Build hierarchy and topological info for derived types
   227) 
   228)     call initGridCells()
   229) 
   230)     ! Deallocate surface grid dynamic memory (for wtxy and vegxy arrays)
   231) 
   232)     deallocate (vegxy, wtxy, topoxy)
   233) 
   234)     call t_stopf('clm_init1')
   235) 
   236)   end subroutine initialize1
   237)

   239) !BOP

   241) ! !IROUTINE: initialize2
   242) !
   243) ! !INTERFACE:
   244)   subroutine initialize2( )
   245) !
   246) ! !DESCRIPTION:
   247) ! Land model initialization.
   248) ! o Initializes run control variables via the [clm_inparm] namelist.
   249) ! o Reads surface data on model grid.
   250) ! o Defines the multiple plant types and fraction areas for each surface type.
   251) ! o Builds the appropriate subgrid <-> grid mapping indices and weights.
   252) ! o Set up parallel processing.
   253) ! o Initializes time constant variables.
   254) ! o Reads restart data for a restart or branch run.
   255) ! o Reads initial data and initializes the time variant variables for an initial run.
   256) ! o Initializes history file output.
   257) ! o Initializes river routing model.
   258) ! o Initializes accumulation variables.
   259) !
   260) ! !USES:
   261)     use clm_atmlnd      , only : clm_map2gcell_minimal
   262)     use clm_glclnd      , only : update_clm_s2x
   263)     use clm_varctl      , only : finidat, flanduse_timeseries
   264)     use decompMod       , only : get_proc_clumps, get_proc_bounds
   265)     use filterMod       , only : allocFilters, setFilters
   266)     use histFldsMod     , only : hist_initFlds
   267)     use histFileMod     , only : hist_htapes_build, htapes_fieldlist
   268)     use restFileMod     , only : restFile_getfile, &
   269)                                  restFile_open, restFile_close, restFile_read 
   270)     use accFldsMod      , only : initAccFlds, initAccClmtype
   271)     use mkarbinitMod    , only : mkarbinit
   272)     use pftdynMod       , only : pftdyn_init, pftdyn_interp
   273)     use ndepStreamMod    , only : ndep_init, ndep_interp
   274)     use CNEcosystemDynMod, only : CNEcosystemDynInit
   275)     use pftdynMod             , only : pftwt_init
   276)     use CNDVEcosystemDyniniMod, only : CNDVEcosystemDynini
   277)     use STATICEcosysDynMod , only : EcosystemDynini, readAnnualVegetation
   278)     use STATICEcosysDynMod , only : interpMonthlyVeg
   279)     use DustMod         , only : Dustini
   280)     use clm_time_manager, only : get_curr_date, get_nstep, advance_timestep, &
   281)                                  timemgr_init, timemgr_restart_io, timemgr_restart
   282)     use clm_time_manager, only : get_step_size, get_curr_calday
   283)     use fileutils       , only : getfil
   284)     use UrbanMod        , only : UrbanClumpInit
   285)     use UrbanInitMod    , only : UrbanInitTimeConst, UrbanInitTimeVar, UrbanInitAero 
   286)     use UrbanInputMod   , only : UrbanInput
   287)     use seq_drydep_mod  , only : n_drydep, drydep_method, DD_XLND
   288)     use shr_orb_mod        , only : shr_orb_decl
   289)     use initSurfAlbMod     , only : initSurfAlb, do_initsurfalb 
   290)     use clm_varorb         , only : eccen, mvelpp, lambm0, obliqr
   291)     use VOCEmissionMod  , only : VOCEmission_init
   292) 
   293) 
   294) ! !Arguments    
   295)     implicit none
   296) !
   297) ! !REVISION HISTORY:
   298) ! Created by Gordon Bonan, Sam Levis and Mariana Vertenstein
   299) !
   300) !
   301) ! !LOCAL VARIABLES:
   302) !EOP
   303)     integer  :: nl,na,nag             ! indices
   304)     integer  :: i,j,k                 ! indices
   305)     integer  :: yr                    ! current year (0, ...)
   306)     integer  :: mon                   ! current month (1 -> 12)
   307)     integer  :: day                   ! current day (1 -> 31)
   308)     integer  :: ncsec                 ! current time of day [seconds]
   309)     integer  :: nc                    ! clump index
   310)     integer  :: nclumps               ! number of clumps on this processor
   311)     integer  :: begp, endp            ! clump beg and ending pft indices
   312)     integer  :: begc, endc            ! clump beg and ending column indices
   313)     integer  :: begl, endl            ! clump beg and ending landunit indices
   314)     integer  :: begg, endg            ! clump beg and ending gridcell indices
   315)     character(len=256) :: fnamer      ! name of netcdf restart file 
   316)     character(len=256) :: pnamer      ! full pathname of netcdf restart file
   317)     type(file_desc_t)  :: ncid        ! netcdf id
   318)     real(r8) :: dtime                 ! time step increment (sec)
   319)     integer  :: nstep                 ! model time step
   320)     real(r8) :: calday                ! calendar day for nstep
   321)     real(r8) :: caldaym1              ! calendar day for nstep-1
   322)     real(r8) :: declin                ! solar declination angle in radians for nstep
   323)     real(r8) :: declinm1              ! solar declination angle in radians for nstep-1
   324)     real(r8) :: eccf                  ! earth orbit eccentricity factor
   325)     character(len=32) :: subname = 'initialize2' ! subroutine name
   326) !----------------------------------------------------------------------
   327) 
   328)     call t_startf('clm_init2')
   329) 
   330)     ! ------------------------------------------------------------------------
   331)     ! Initialize time constant variables 
   332)     ! ------------------------------------------------------------------------
   333) 
   334)     call get_proc_bounds(begg, endg, begl, endl, begc, endc, begp, endp )
   335) 
   336)     ! Initialize Ecosystem Dynamics 
   337) 
   338)     call t_startf('init_ecosys')
   339)     if (use_cndv) then
   340)        call CNDVEcosystemDynini()
   341)     else if (.not. use_cn) then
   342)        call EcosystemDynini()
   343)     end if
   344) 
   345)     if (use_cn .or. use_cndv) then
   346)        ! --------------------------------------------------------------
   347)        ! Initialize CLMSP ecosystem dynamics when drydeposition is used
   348)        ! so that estimates of monthly differences in LAI can be computed
   349)        ! --------------------------------------------------------------
   350)        if ( n_drydep > 0 .and. drydep_method == DD_XLND )then
   351)           call EcosystemDynini()
   352)        end if
   353)     end if
   354)     call t_stopf('init_ecosys')
   355) 
   356)     ! Initialize dust emissions model 
   357) 
   358)     call t_startf('init_dust')
   359)     call Dustini()
   360)     call t_stopf('init_dust')
   361)     
   362)     ! Initialize MEGAN emissions model 
   363) 
   364)     call VOCEmission_init( )
   365) 
   366)     ! ------------------------------------------------------------------------
   367)     ! Initialize time constant urban variables
   368)     ! ------------------------------------------------------------------------
   369) 
   370)     call t_startf('init_io1')
   371)     call UrbanInitTimeConst()
   372)     call iniTimeConst()
   373) 
   374)     ! ------------------------------------------------------------------------
   375)     ! Obtain restart file if appropriate
   376)     ! ------------------------------------------------------------------------
   377) 
   378)     if (do_restread()) then
   379)        call restFile_getfile( file=fnamer, path=pnamer )
   380)     end if
   381) 
   382)     ! ------------------------------------------------------------------------
   383)     ! Initialize master history list. 
   384)     ! ------------------------------------------------------------------------
   385)     call t_startf('hist_initFlds')
   386) 
   387)     call hist_initFlds()
   388)     ! On restart process the history namelist. Later the namelist from the restart file
   389)     ! will be used. But, this allows some basic checking to make sure you didn't
   390)     ! try to change the history namelist on restart.
   391)     if (nsrest == nsrContinue ) call htapes_fieldlist()
   392) 
   393)     call t_stopf('hist_initFlds')
   394)     ! ------------------------------------------------------------------------
   395)     ! Initialize time manager
   396)     ! ------------------------------------------------------------------------
   397) 
   398)     if (nsrest == nsrStartup) then  
   399)        call timemgr_init()
   400)     else
   401)        call restFile_open( flag='read', file=fnamer, ncid=ncid )
   402)        call timemgr_restart_io( ncid=ncid, flag='read' )
   403)        call restFile_close( ncid=ncid )
   404)        call timemgr_restart()
   405)     end if
   406)     call t_stopf('init_io1')
   407) 
   408)     ! ------------------------------------------------------------------------
   409)     ! Initialize CN Ecosystem Dynamics (must be after time-manager initialization)
   410)     ! ------------------------------------------------------------------------
   411)     if (use_cn .or. use_cndv) then
   412)        call CNEcosystemDynInit( begc, endc, begp, endp )
   413)     end if
   414) 
   415)     ! ------------------------------------------------------------------------
   416)     ! Initialize accumulated fields
   417)     ! ------------------------------------------------------------------------
   418) 
   419)     ! Initialize accumulator fields to be time accumulated for various purposes.
   420)     ! The time manager needs to be initialized before this called is made, since
   421)     ! the step size is needed.
   422) 
   423)     call t_startf('init_accflds')
   424)     call initAccFlds()
   425)     call t_stopf('init_accflds')
   426) 
   427)     ! ------------------------------------------------------------------------
   428)     ! Set arbitrary initial conditions for time varying fields 
   429)     ! used in coupled carbon-nitrogen code
   430)     ! ------------------------------------------------------------------------
   431)     
   432)     if (use_cn) then
   433)        call t_startf('init_cninitim')
   434)        if (nsrest == nsrStartup) then
   435)           call CNiniTimeVar()
   436)        end if
   437)        call t_stopf('init_cninitim')
   438)     end if
   439) 
   440)     ! ------------------------------------------------------------------------
   441)     ! Initialization of dynamic pft weights
   442)     ! ------------------------------------------------------------------------
   443) 
   444)     ! Determine correct pft weights (interpolate pftdyn dataset if initial run)
   445)     ! Otherwise these are read in for a restart run
   446) 
   447)     if (use_cndv) then
   448)        call pftwt_init()
   449)     else
   450)        if (flanduse_timeseries /= ' ') then
   451)           call t_startf('init_pftdyn')
   452)           call pftdyn_init()
   453)           call pftdyn_interp( )
   454)           call t_stopf('init_pftdyn')
   455)        end if
   456)     end if
   457) 
   458)     ! ------------------------------------------------------------------------
   459)     ! Read restart/initial info
   460)     ! ------------------------------------------------------------------------
   461) 
   462)     ! No weight related information can be contained in the routines,  
   463)     ! "mkarbinit, inicfile and restFile". 
   464) 
   465)     call t_startf('init_io2')
   466)     if (do_restread()) then
   467)        if (masterproc) write(iulog,*)'reading restart file ',fnamer
   468)        call restFile_read( fnamer )
   469)     else if (nsrest == nsrStartup .and. finidat == ' ') then
   470)        call mkarbinit()
   471)        call UrbanInitTimeVar( )
   472)     end if
   473)     call t_stopf('init_io2')
   474) 
   475)     ! ------------------------------------------------------------------------
   476)     ! Initialize nitrogen deposition
   477)     ! ------------------------------------------------------------------------
   478) 
   479)     if (use_cn) then
   480)        call t_startf('init_ndep')
   481)        call ndep_init()
   482)        call ndep_interp()
   483)        call t_stopf('init_ndep')
   484)     end if
   485)     
   486)     ! ------------------------------------------------------------------------
   487)     ! Initialization of model parameterizations that are needed after
   488)     ! restart file is read in
   489)     ! ------------------------------------------------------------------------
   490) 
   491)     ! ------------------------------------------------------------------------
   492)     ! Initialize history and accumator buffers
   493)     ! ------------------------------------------------------------------------
   494) 
   495)     call t_startf('init_hist1')
   496) 
   497)     ! Initialize active history fields. This is only done if not a restart run. 
   498)     ! If a restart run, then this information has already been obtained from the 
   499)     ! restart data read above. Note that routine hist_htapes_build needs time manager 
   500)     ! information, so this call must be made after the restart information has been read.
   501) 
   502)     if (nsrest == nsrStartup .or. nsrest == nsrBranch) call hist_htapes_build()
   503) 
   504)     ! Initialize clmtype variables that are obtained from accumulated fields.
   505)     ! This routine is called in an initial run at nstep=0
   506)     ! This routine is also always called for a restart run and must 
   507)     ! therefore be called after the restart file is read in
   508) 
   509)     call initAccClmtype()
   510) 
   511)     call t_stopf('init_hist1')
   512) 
   513)     ! --------------------------------------------------------------
   514)     ! Note - everything below this point needs updated weights
   515)     ! --------------------------------------------------------------
   516) 
   517)     ! Initialize filters
   518)     
   519)     call t_startf('init_filters')
   520) 
   521)     call allocFilters()
   522)     nclumps = get_proc_clumps()
   523) !$OMP PARALLEL DO PRIVATE (nc)
   524)     do nc = 1, nclumps
   525)        call setFilters(nc)
   526)     end do
   527) !$OMP END PARALLEL DO
   528) 
   529)     call t_stopf('init_filters')
   530) 
   531)     ! Calculate urban "town" roughness length and displacement 
   532)     ! height for urban landunits
   533) 
   534)     call UrbanInitAero()
   535) 
   536)     ! Initialize urban radiation model - this uses urbinp data structure
   537) 
   538)     call UrbanClumpInit()
   539) 
   540)     ! Finalize urban model initialization
   541)     
   542)     call UrbanInput(mode='finalize')
   543) 
   544)     !
   545)     ! Even if CN is on, and dry-deposition is active, read CLMSP annual vegetation to get estimates of monthly LAI
   546)     !
   547)     if ( n_drydep > 0 .and. drydep_method == DD_XLND )then
   548)        call readAnnualVegetation()
   549)     end if
   550) 
   551)     ! End initialization
   552) 
   553)     call t_startf('init_wlog')
   554)     if (masterproc) then
   555)        write(iulog,*) 'Successfully initialized the land model'
   556)        if (nsrest == nsrStartup) then
   557)           write(iulog,*) 'begin initial run at: '
   558)        else
   559)           write(iulog,*) 'begin continuation run at:'
   560)        end if
   561)        call get_curr_date(yr, mon, day, ncsec)
   562)        write(iulog,*) '   nstep= ',get_nstep(), ' year= ',yr,' month= ',mon,&
   563)             ' day= ',day,' seconds= ',ncsec
   564)        write(iulog,*)
   565)        write(iulog,'(72a1)') ("*",i=1,60)
   566)        write(iulog,*)
