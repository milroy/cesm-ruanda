    87)        rmp      => elem(ie)%rmp

    89)        metinv   => elem(ie)%metinv

    97) 
    98)           grad_tmp(:,:,:) = gradient(elem(ie)%state%T(:,:,k,np1),deriv)*rrearth

   100)           do j=1,np
   101)              do i=1,np
   102) !                 grad_T_np1(i,j,1,k,ie) = mp(i,j)*(metinv(1,1,i,j)*grad_tmp(i,j,1)+metinv(1,2,i,j)*grad_tmp(i,j,2))
   103) !                 grad_T_np1(i,j,2,k,ie) = mp(i,j)*(metinv(2,1,i,j)*grad_tmp(i,j,1)+metinv(2,2,i,j)*grad_tmp(i,j,2))

   105)                  grad_T_np1(i,j,1,k,ie) = mp(i,j)*(elem(ie)%Dinv(1,1,i,j)*grad_tmp(i,j,1)+elem(ie)%Dinv(2,1,i,j)*grad_tmp(i,j,2))

   108)                 v2     = elem(ie)%state%v(i,j,2,k,np1)

   110)                 gp(i,j,1) = metdet(i,j)*(elem(ie)%Dinv(1,1,i,j)*v1 + elem(ie)%Dinv(1,2,i,j)*v2)
   111)                 gp(i,j,2) = metdet(i,j)*(elem(ie)%Dinv(2,1,i,j)*v1 + elem(ie)%Dinv(2,2,i,j)*v2)
   112)                 vco(i,j,1) = elem(ie)%D(1,1,i,j)*v1 + elem(ie)%D(2,1,i,j)*v2
   113)                 vco(i,j,2) = elem(ie)%D(1,2,i,j)*v1 + elem(ie)%D(2,2,i,j)*v2
   114)              end do
   115)           end do
   116) 
   117)           div_tmp(:,:)  = divergence(gp,deriv)*rrearth
   118)           zeta_tmp(:,:) = vorticity(vco,deriv)*rrearth
   119) 
   120)           do j=1,np
   121)              do i=1,np
   122)                 div_np1(i,j,k,ie)  = mp(i,j)*rmetdetv(i,j)*div_tmp(i,j)
   123)                 zeta_np1(i,j,k,ie) = mp(i,j)*rmetdetv(i,j)*zeta_tmp(i,j)
   124)              end do
   125)           end do
   126) 
   127)        end do
   128) 
   129)        kptr=0
   130)        call edgeVpack(edge4,grad_T_np1(1,1,1,1,ie),2*nlev,kptr,elem(ie)%desc)
   131) 
   132)        kptr=2*nlev
   133)        call edgeVpack(edge4,zeta_np1(1,1,1,ie),nlev,kptr,elem(ie)%desc)
   134) 
   135)        kptr=3*nlev
   136)        call edgeVpack(edge4,div_np1(1,1,1,ie),nlev,kptr,elem(ie)%desc)
   137) 
   138) !       kptr=0
   139) !       call edgerotate(edge4,2*nlev,kptr,elem(ie)%desc)
   140) 
   141)     end do
   142)

   144) 
   145)     do ie=nets,nete
   146) 
   147)        mp       => elem(ie)%mp
   148)        rmp      => elem(ie)%rmp
   149)        metdet   => elem(ie)%metdet
   150)        metinv   => elem(ie)%metinv
   151)        rmetdetv(:,:)=1.0_real_kind/elem(ie)%metdet(:,:)
   152)        D        => elem(ie)%D
   153)        Dinv     => elem(ie)%Dinv
   154) 
   155)        kptr=0
   156)        call edgeVunpack(edge4, grad_T_np1(1,1,1,1,ie), 2*nlev, kptr, elem(ie)%desc)
   157) 
   158)        kptr=2*nlev
   159)        call edgeVunpack(edge4, zeta_np1(1,1,1,ie), nlev, kptr, elem(ie)%desc)
   160) 
   161)        kptr=3*nlev
   162)        call edgeVunpack(edge4, div_np1(1,1,1,ie), nlev, kptr, elem(ie)%desc)
   163) #ifdef DEBUGOMP
   164) #if (! defined ELEMENT_OPENMP)
   165) !$OMP BARRIER
   166) #endif
   167) #endif
   168)        ! ======================================
   169)        ! compute Laplacian of T(n+1)
   170)        ! ======================================
   171) 
   172) #if (defined ELEMENT_OPENMP)
   173) !$omp parallel do private(k,i,j,grad_tmp,div_tmp,grad_div,grad_zeta,v1,v2)
   174) #endif
   175)        do k=1,nlev
   176)           do j=1,np
   177)              do i=1,np
   178) 
   179)                 ! ======================================================
   180)                 ! complete global assembly of gradT(n+1), zeta(n+1), div(n+1)
   181)                 ! ======================================================
   182) 
   183) !                v1 = rmp(i,j)*grad_T_np1(i,j,1,k,ie)
   184) !                v2 = rmp(i,j)*grad_T_np1(i,j,2,k,ie)
   185) 
   186)                 zeta_np1(i,j,k,ie) = rmp(i,j)*zeta_np1(i,j,k,ie)
   187)                 div_np1(i,j,k,ie)  = rmp(i,j)*div_np1(i,j,k,ie)
   188)                 elem(ie)%derived%zeta(i,j,k) = zeta_np1(i,j,k,ie)
   189) 
   190)                 ! ==========================================
   191)                 ! Compute contravariant gradient(T(n+1))
   192)                 ! ==========================================
   193) 
   194) !                grad_T_np1(i,j,1,k,ie) = metdet(i,j)*(metinv(1,1,i,j)*v1 + metinv(1,2,i,j)*v2)
   195) !                grad_T_np1(i,j,2,k,ie) = metdet(i,j)*(metinv(2,1,i,j)*v1 + metinv(2,2,i,j)*v2)
   196)                 grad_tmp(i,j,:) = grad_T_np1(i,j,:,k,ie)
   197)                 grad_T_np1(i,j,1,k,ie) = metdet(i,j)*rmp(i,j)*&
   198)        (elem(ie)%Dinv(1,1,i,j)*grad_tmp(i,j,1)+elem(ie)%Dinv(1,2,i,j)*grad_tmp(i,j,2))
   199)                 grad_T_np1(i,j,2,k,ie) = metdet(i,j)*rmp(i,j)*&
   200)        (elem(ie)%Dinv(2,1,i,j)*grad_tmp(i,j,1)+elem(ie)%Dinv(2,2,i,j)*grad_tmp(i,j,2))
   201)                 !grad_T_np1(i,j,1,k,ie) = metdet(i,j)*rmp(i,j)*grad_T_np1(i,j,1,k,ie)
   202)                 !grad_T_np1(i,j,2,k,ie) = metdet(i,j)*rmp(i,j)*grad_T_np1(i,j,2,k,ie)
   203) 
   204)              end do
   205)           end do
   206) 
   207)           div_tmp(:,:) = divergence(grad_T_np1(:,:,:,k,ie),deriv)*rrearth
   208) 
   209)           do j=1,np
   210)              do i=1,np
   211)                 lap_T_np1(i,j,k,ie)   = mp(i,j)*rmetdetv(i,j)*div_tmp(i,j)
   212)              end do
   213)           end do
   214) 
   215)           ! ==========================================
   216)           ! compute covariant Laplacian of v(n+1)  
   217)           ! ==========================================
   218) 
   219)           grad_div(:,:,:)  = gradient(div_np1(:,:,k,ie),deriv)*rrearth
   220)           grad_zeta(:,:,:) = gradient(zeta_np1(:,:,k,ie),deriv)*rrearth
   221) 
   222)           do j=1,np
   223)              do i=1,np
   224) 
   225)                 v2 = Dinv(1,1,i,j)*grad_zeta(i,j,1) + Dinv(2,1,i,j)*grad_zeta(i,j,2)
   226)                 v1 = Dinv(1,2,i,j)*grad_zeta(i,j,1) + Dinv(2,2,i,j)*grad_zeta(i,j,2)
   227) 
   228)                 v2 = -v2
   229)                 grad_zeta(i,j,1) = v1
   230)                 grad_zeta(i,j,2) = v2
   231)                 v1 = grad_div(i,j,1)
   232)                 v2 = grad_div(i,j,2)
   233) 
   234)                 grad_div(i,j,1) = Dinv(1,1,i,j)*v1 + Dinv(2,1,i,j)*v2
   235)                 grad_div(i,j,2) = Dinv(1,2,i,j)*v1 + Dinv(2,2,i,j)*v2
   236)                 lap_v_np1(i,j,1,k,ie) = mp(i,j)*(grad_div(i,j,1) - grad_zeta(i,j,1))
   237)                 lap_v_np1(i,j,2,k,ie) = mp(i,j)*(grad_div(i,j,2) - grad_zeta(i,j,2))
   238) 
   239)              end do
   240)           end do
   241) 
   242)        end do
   243) 
   244)        kptr=0
   245)        call edgeVpack(edge3, lap_v_np1(1,1,1,1,ie),2*nlev,kptr,elem(ie)%desc)
   246) 
   247)        kptr=2*nlev
   248)        call edgeVpack(edge3, lap_T_np1(1,1,1,ie),nlev,kptr,elem(ie)%desc)
   249) 
   250)     end do
   251) 
   252)     call bndry_exchangeV(hybrid,edge3)
   253) 
   254)     do ie=nets,nete
   255) 
   256)        rmp   => elem(ie)%rmp
   257)        metdet   => elem(ie)%metdet
   258)        rmetdetv(:,:)=1.0_real_kind/elem(ie)%metdet(:,:)
   259) 
   260)        kptr=0
   261)        call edgeVunpack(edge3, lap_v_np1(1,1,1,1,ie), 2*nlev, kptr, elem(ie)%desc)
   262) 
   263)        kptr=2*nlev
   264)        call edgeVunpack(edge3, lap_T_np1(1,1,1,ie), nlev, kptr, elem(ie)%desc)
   265) 
   266) #if (defined ELEMENT_OPENMP)
   267) !$omp parallel do private(k,i,j)
   268) #endif
   269)        do k=1,nlev   
   270) 
   271)           do j=1,np
   272)              do i=1,np
   273)                 lap_T_np1(i,j,k,ie)   = rmp(i,j)*lap_T_np1(i,j,k,ie)

   275)                 lap_v_np1(i,j,2,k,ie) = rmp(i,j)*lap_v_np1(i,j,2,k,ie)
   276) 
   277)                 elem(ie)%state%T(i,j,k,np1)   = elem(ie)%state%T(i,j,k,np1)   + nu*dt2*lap_T_np1(i,j,k,ie)
   278)                 elem(ie)%state%v(i,j,1,k,np1) = elem(ie)%state%v(i,j,1,k,np1) + nu*dt2*lap_v_np1(i,j,1,k,ie)
   279)                 elem(ie)%state%v(i,j,2,k,np1) = elem(ie)%state%v(i,j,2,k,np1) + nu*dt2*lap_v_np1(i,j,2,k,ie)
   280)              end do
   281)           end do
   282) 
   283)        end do
   284) 
   285)     end do
   286) #ifdef DEBUGOMP
   287) #if (! defined ELEMENT_OPENMP)
   288) !$OMP BARRIER
   289) #endif
   290) #endif
   291)     call t_stopf('prim_diffusion')
   292)

   297)   subroutine scalar_diffusion(elem, nets,nete,np1,deriv,dt2,hybrid)
   298)     use hybrid_mod, only : hybrid_t
   299)     use physical_constants, only : rrearth
   300)     use element_mod, only : element_t
   301)     use bndry_mod, only : bndry_exchangev

   304)     implicit none
   305)     type(element_t), intent(inout), target :: elem(:)

   359)                    v2 = mp(i,j)*grad_tmp(i,j,2)

   380) 
   381)        mp       => elem(ie)%mp
   382)        rmp      => elem(ie)%rmp
   383)        metdet   => elem(ie)%metdet
   384)        metinv   => elem(ie)%metinv
   385)        rmetdetv(:,:)=1.0_real_kind/elem(ie)%metdet(:,:)
   386)        D        => elem(ie)%D
   387)        Dinv     => elem(ie)%Dinv
   388)

   390) #ifdef DEBUGOMP
   391) #if (! defined ELEMENT_OPENMP)
   392) !$OMP BARRIER
   393) #endif
   394) #endif
   395) 
   396)        do q=1,qsize
   397) 
   398) #if (defined ELEMENT_OPENMP)
   399) !$omp parallel do private(k,i,j,div_tmp)
   400) #endif
   401)           do k=1,nlev
   402)              do j=1,np
   403)                 do i=1,np
   404) 
   405)                    !                v1 = rmp(i,j)*grad_Q_np1(i,j,1,k,ie)
   406)                    !                v2 = rmp(i,j)*grad_Q_np1(i,j,2,k,ie)
   407) 
   408)                    !                grad_Q_np1(i,j,1,k,ie) = metdet(i,j)*(metinv(1,1,i,j)*v1 + metinv(1,2,i,j)*v2)
   409)                    !                grad_Q_np1(i,j,2,k,ie) = metdet(i,j)*(metinv(2,1,i,j)*v1 + metinv(2,2,i,j)*v2)

   415)              end do
   416) 
   417)              div_tmp(:,:) = divergence(grad_Q_np1(:,:,:,k,q,ie),deriv)*rrearth
   418) 
   419)              do j=1,np
   420)                 do i=1,np
   421)                    lap_Q_np1(i,j,k,q,ie)   = mp(i,j)*rmetdetv(i,j)*div_tmp(i,j)
   422)                 end do
   423)              end do
   424) 
   425)           end do
   426) 
   427)        end do
   428) 
   429)        call edgeVpack(edgeS1, lap_Q_np1(1,1,1,1,ie),nlev*qsize, 0,elem(ie)%desc)
