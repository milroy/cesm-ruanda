   157) 
   158)        call set_landunit_urban( &
   159) !           ltype=isturb, wtxy=wtxy, vegxy=vegxy,   &

   165)        call set_landunit_wet_ice_lake(              &
   166)             ltype=istdlak, &
   167)             nw=nwtxy, gi=gdc, li=li, ci=ci, pi=pi, setdata=my_gcell)
   168) 
   169)        call set_landunit_wet_ice_lake(              &
   170)             ltype=istwet, &
   171)             nw=nwtxy, gi=gdc, li=li, ci=ci, pi=pi, setdata=my_gcell)
   172) 
   173)        call set_landunit_wet_ice_lake(              &
   174)             ltype=istice, &
   175)             nw=nwtxy, gi=gdc, li=li, ci=ci, pi=pi, setdata=my_gcell)

   177)        if (create_glacier_mec_landunit) then
   178)           call set_landunit_wet_ice_lake(              &
   179)                ltype=istice_mec, &
   180)                nw=nwtxy, gi=gdc, li=li, ci=ci, pi=pi, setdata=my_gcell, &

   186)        gptr%gris_mask(gdc) = 0._r8
   187)        gptr%gris_area(gdc) = 0._r8
   188)        gptr%aais_mask(gdc) = 0._r8
   189)        gptr%aais_area(gdc) = 0._r8

   191)        ! Greenland mask
   192)        if ( (ldomain%latc(gdc) >  58. .and. ldomain%latc(gdc) <= 67.  .and.   &

   195)             (ldomain%latc(gdc) >  67. .and. ldomain%latc(gdc) <= 70. .and.    &
   196)              ldomain%lonc(gdc) > 300. .and. ldomain%lonc(gdc) < 345.)         &
   197)                                       .or.                                 &
   198)             (ldomain%latc(gdc) >  70. .and. ldomain%latc(gdc) <= 75. .and.    &
   199)              ldomain%lonc(gdc) > 295. .and. ldomain%lonc(gdc) < 350.)         &
   200)                                       .or.                                 &
   201)             (ldomain%latc(gdc) >  75. .and. ldomain%latc(gdc) <= 79. .and.    &
   202)              ldomain%lonc(gdc) > 285. .and. ldomain%lonc(gdc) < 350.)         &
   203)                                       .or.                                 &
   204)             (ldomain%latc(gdc) >  79. .and. ldomain%latc(gdc) <  85. .and.    &
   205)              ldomain%lonc(gdc) > 290. .and. ldomain%lonc(gdc) < 355.) ) then

   207)             gptr%gris_mask(gdc) = 1.0_r8
   208) 
   209)       elseif (ldomain%latc(gdc) < -60.) then
   210) 
   211)             gptr%aais_mask(gdc) = 1.0_r8
   212) 
   213)        endif  ! Greenland or Antarctic grid cell
   214) 
   215)        ! Set clm3 lats/lons
   216) 
   217)        if (my_gcell) then
   218)           gptr%gindex(gdc) = glo
   219)           gptr%latdeg(gdc) = ldomain%latc(gdc)

   226)     enddo

   238) ! !IROUTINE: clm_ptrs_compdown
   239) !
   240) ! !INTERFACE:
   241)   subroutine clm_ptrs_compdown()
   242) !
   243) ! !DESCRIPTION:
   244) ! Assumes the part of the subgrid pointing up has been set.  Fills 
   245) ! in the data pointing down.  Up is p_c, p_l, p_g, c_l, c_g, and l_g.
   246) !
   247) ! This algorithm assumes all indices are monotonically increasing.
   248) !
   249) ! Algorithm works as follows.  The p, c, and l loops march through
   250) ! the full arrays (nump, numc, and numl) checking the "up" indexes.
   251) ! As soon as the "up" index of the current (p,c,l) cell changes relative 
   252) ! to the previous (p,c,l) cell, the *i array will be set to point down 
   253) ! to that cell.  The *f array follows the same logic, so it's always the
   254) ! last "up" index from the previous cell when an "up" index changes.
   255) !
   256) ! For example, a case where p_c(1:4) = 1 and p_c(5:12) = 2.  This 
   257) ! subroutine will set c_pi(1) = 1, c_pf(1) = 4, c_pi(2) = 5, c_pf(2) = 12.
   258) !
   259) ! !USES
   260)     use clmtype
   261)     use decompMod , only : get_proc_bounds
   262) 
   263) ! !ARGUMENTS
   264)     implicit none
   265) !
   266) ! !CALLED FROM:
   267) ! subroutines initGridCellsMod
   268) !
   269) ! !REVISION HISTORY:
   270) ! 2005.11.15  T Craig Creation
   271) !
   272) !
   273) ! !LOCAL VARIABLES:
   274)     integer :: begg,endg,begl,endl,begc,endc,begp,endp ! beg/end glcp
   275)     integer :: g,l,c,p               ! loop counters
   276)     integer :: curg,curl,curc,curp   ! tracks g,l,c,p indexes in arrays
   277)     type(gridcell_type), pointer  :: gptr ! pointer to gridcell derived subtype
   278)     type(landunit_type), pointer  :: lptr ! pointer to landunit derived subtype
   279)     type(column_type)  , pointer  :: cptr ! pointer to column derived subtype
   280)     type(pft_type)     , pointer  :: pptr ! pointer to pft derived subtype
   281) !EOP
   282) !------------------------------------------------------------------------------
   283) 
   284)     gptr => grc
   285)     lptr => lun
   286)     cptr => col
   287)     pptr => pft
   288) 
   289)     call get_proc_bounds(begg,endg,begl,endl,begc,endc,begp,endp)
   290) 
   291)     !--- Set the current c,l,g (curc, curl, curg) to zero for initialization,
   292)     !---   these indices track the current "up" index.
   293)     !--- Take advantage of locality of g/l/c/p cells
   294)     !--- Loop p through full local begp:endp length
   295)     !--- Separately check the p_c, p_l, and p_g indexes for a change in
   296)     !---   the "up" index.
   297)     !--- If there is a change, verify that the current c,l,g is within the 
   298)     !---   valid range, and set c_pi, l_pi, or g_pi to that current c,l,g
   299)     !--- Constantly update the c_pf, l_pf, and g_pf array.  When the
   300)     !---   g, l, c index changes, the *_pf array will be set correctly
   301)     !--- Do the same for cols setting c_li, c_gi, c_lf, c_gf and
   302)     !---   lunits setting l_gi, l_gf.
   303) 
   304)     curc = 0
   305)     curl = 0
   306)     curg = 0
   307)     do p = begp,endp
   308)        if (pptr%column(p) /= curc) then
   309)           curc = pptr%column(p)
   310)           if (curc < begc .or. curc > endc) then
   311)              write(iulog,*) 'clm_ptrs_compdown ERROR: pcolumn ',p,curc,begc,endc
   312)              call endrun()
   313)           endif
   314)           cptr%pfti(curc) = p
   315)        endif
   316)        cptr%pftf(curc) = p
   317)        cptr%npfts(curc) = cptr%pftf(curc) - cptr%pfti(curc) + 1
   318)        if (pptr%landunit(p) /= curl) then
   319)           curl = pptr%landunit(p)
   320)           if (curl < begl .or. curl > endl) then
   321)              write(iulog,*) 'clm_ptrs_compdown ERROR: plandunit ',p,curl,begl,endl
   322)              call endrun()
   323)           endif
   324)           lptr%pfti(curl) = p
   325)        endif
   326)        lptr%pftf(curl) = p
   327)        lptr%npfts(curl) = lptr%pftf(curl) - lptr%pfti(curl) + 1
   328)        if (pptr%gridcell(p) /= curg) then
   329)           curg = pptr%gridcell(p)
   330)           if (curg < begg .or. curg > endg) then
   331)              write(iulog,*) 'clm_ptrs_compdown ERROR: pgridcell ',p,curg,begg,endg
   332)              call endrun()
   333)           endif
   334)           gptr%pfti(curg) = p
   335)        endif
   336)        gptr%pftf(curg) = p
   337)        gptr%npfts(curg) = gptr%pftf(curg) - gptr%pfti(curg) + 1
   338)     enddo
   339) 
   340)     curg = 0
   341)     curl = 0
   342)     do c = begc,endc
   343)        if (cptr%landunit(c) /= curl) then
   344)           curl = cptr%landunit(c)
   345)           if (curl < begl .or. curl > endl) then
   346)              write(iulog,*) 'clm_ptrs_compdown ERROR: clandunit ',c,curl,begl,endl
   347)              call endrun()
   348)           endif
   349)           lptr%coli(curl) = c
   350)        endif
   351)        lptr%colf(curl) = c
   352)        lptr%ncolumns(curl) = lptr%colf(curl) - lptr%coli(curl) + 1
   353)        if (cptr%gridcell(c) /= curg) then
   354)           curg = cptr%gridcell(c)
   355)           if (curg < begg .or. curg > endg) then
   356)              write(iulog,*) 'clm_ptrs_compdown ERROR: cgridcell ',c,curg,begg,endg
   357)              call endrun()
   358)           endif
   359)           gptr%coli(curg) = c
   360)        endif
   361)        gptr%colf(curg) = c
   362)        gptr%ncolumns(curg) = gptr%colf(curg) - gptr%coli(curg) + 1
   363)     enddo
   364) 
   365)     curg = 0
   366)     do l = begl,endl
   367)        if (lptr%gridcell(l) /= curg) then
   368)           curg = lptr%gridcell(l)
   369)           if (curg < begg .or. curg > endg) then
   370)              write(iulog,*) 'clm_ptrs_compdown ERROR: lgridcell ',l,curg,begg,endg
   371)              call endrun()
   372)           endif
   373)           gptr%luni(curg) = l
   374)        endif
   375)        gptr%lunf(curg) = l
   376)        gptr%nlandunits(curg) = gptr%lunf(curg) - gptr%luni(curg) + 1
   377)     enddo
   378) 
   379)     end subroutine clm_ptrs_compdown

   398) ! !CALLED FROM:

   406)     type(landunit_type), pointer  :: lptr ! pointer to landunit derived subtype
   407)     type(column_type)  , pointer  :: cptr ! pointer to column derived subtype
   408)     type(pft_type)     , pointer  :: pptr ! pointer to pft derived subtype
   409)     integer :: begg,endg,begl,endl,begc,endc,begp,endp   ! beg/end indices

   417)     cptr => col

   421)     if (masterproc) write(iulog,*) '---clm_ptrs_check:'
   422)     call get_proc_bounds(begg,endg,begl,endl,begc,endc,begp,endp)
   423) 
   424)     !--- check index ranges ---
   425)     error = .false.
   426)     if (minval(gptr%luni) < begl .or. maxval(gptr%luni) > endl) error=.true.
   427)     if (minval(gptr%lunf) < begl .or. maxval(gptr%lunf) > endl) error=.true.
   428)     if (minval(gptr%coli) < begc .or. maxval(gptr%coli) > endc) error=.true.
   429)     if (minval(gptr%colf) < begc .or. maxval(gptr%colf) > endc) error=.true.
   430)     if (minval(gptr%pfti) < begp .or. maxval(gptr%pfti) > endp) error=.true.
   431)     if (minval(gptr%pftf) < begp .or. maxval(gptr%pftf) > endp) error=.true.
   432)     if (error) then
   433)        write(iulog,*) '   clm_ptrs_check: g index ranges - ERROR'
   434)        write(iulog,*)'minval,beg,maxval,end'
   435)        write(iulog,*) minval(gptr%luni),begl,maxval(gptr%luni),endl
   436)        write(iulog,*) minval(gptr%lunf),begl,maxval(gptr%lunf),endl
   437)        write(iulog,*) minval(gptr%coli),begc,maxval(gptr%coli),endc
   438)        write(iulog,*) minval(gptr%colf),begc,maxval(gptr%colf),endc
   439)        write(iulog,*) minval(gptr%pfti),begp,maxval(gptr%pfti),endp
   440)        write(iulog,*) minval(gptr%pftf),begp,maxval(gptr%pftf),endp
   441)        call endrun()
   442)     endif
   443)     if (masterproc) write(iulog,*) '   clm_ptrs_check: g index ranges - OK'
   444) 
   445)     error = .false.
   446)     if (minval(lptr%gridcell) < begg .or. maxval(lptr%gridcell) > endg) error=.true.
   447)     if (minval(lptr%coli) < begc .or. maxval(lptr%coli) > endc) error=.true.
   448)     if (minval(lptr%colf) < begc .or. maxval(lptr%colf) > endc) error=.true.
   449)     if (minval(lptr%pfti) < begp .or. maxval(lptr%pfti) > endp) error=.true.
   450)     if (minval(lptr%pftf) < begp .or. maxval(lptr%pftf) > endp) error=.true.
   451)     if (error) then
   452)        write(iulog,*) '   clm_ptrs_check: l index ranges - ERROR'
   453)        call endrun()
   454)     endif
   455)     if (masterproc) write(iulog,*) '   clm_ptrs_check: l index ranges - OK'
   456) 
   457)     error = .false.
   458)     if (minval(cptr%gridcell) < begg .or. maxval(cptr%gridcell) > endg) error=.true.
   459)     if (minval(cptr%landunit) < begl .or. maxval(cptr%landunit) > endl) error=.true.
   460)     if (minval(cptr%pfti) < begp .or. maxval(cptr%pfti) > endp) error=.true.
   461)     if (minval(cptr%pftf) < begp .or. maxval(cptr%pftf) > endp) error=.true.
   462)     if (error) then
   463)        write(iulog,*) '   clm_ptrs_check: c index ranges - ERROR'
   464)        call endrun()
   465)     endif
   466)     if (masterproc) write(iulog,*) '   clm_ptrs_check: c index ranges - OK'
   467) 
   468)     error = .false.
   469)     if (minval(pptr%gridcell) < begg .or. maxval(pptr%gridcell) > endg) error=.true.
   470)     if (minval(pptr%landunit) < begl .or. maxval(pptr%landunit) > endl) error=.true.
   471)     if (minval(pptr%column) < begc .or. maxval(pptr%column) > endc) error=.true.
   472)     if (error) then
   473)        write(iulog,*) '   clm_ptrs_check: p index ranges - ERROR'
   474)        call endrun()
   475)     endif
   476)     if (masterproc) write(iulog,*) '   clm_ptrs_check: p index ranges - OK'
   477) 
   478)     !--- check that indices in arrays are monotonically increasing ---
   479)     error = .false.
   480)     do g=begg+1,endg
   481)       if (gptr%luni(g) < gptr%luni(g-1)) error = .true.
   482)       if (gptr%lunf(g) < gptr%lunf(g-1)) error = .true.

   484)       if (gptr%colf(g) < gptr%colf(g-1)) error = .true.
   485)       if (gptr%pfti(g) < gptr%pfti(g-1)) error = .true.
   486)       if (gptr%pftf(g) < gptr%pftf(g-1)) error = .true.
   487)       if (error) then
   488)          write(iulog,*) '   clm_ptrs_check: g mono increasing - ERROR'
   489)          call endrun()
   490)       endif
   491)     enddo
   492)     if (masterproc) write(iulog,*) '   clm_ptrs_check: g mono increasing - OK'

   494)     error = .false.
   495)     do l=begl+1,endl
   496)       if (lptr%gridcell(l) < lptr%gridcell(l-1)) error = .true.
   497)       if (lptr%coli(l) < lptr%coli(l-1)) error = .true.
   498)       if (lptr%colf(l) < lptr%colf(l-1)) error = .true.
   499)       if (lptr%pfti(l) < lptr%pfti(l-1)) error = .true.
   500)       if (lptr%pftf(l) < lptr%pftf(l-1)) error = .true.
   501)       if (error) then

   503)          call endrun()
   504)       endif
   505)     enddo
   506)     if (masterproc) write(iulog,*) '   clm_ptrs_check: l mono increasing - OK'

   508)     error = .false.
   509)     do c=begc+1,endc
   510)       if (cptr%gridcell(c) < cptr%gridcell(c-1)) error = .true.
   511)       if (cptr%landunit(c) < cptr%landunit(c-1)) error = .true.
   512)       if (cptr%pfti(c)     < cptr%pfti(c-1)) error = .true.
   513)       if (cptr%pftf(c)     < cptr%pftf(c-1)) error = .true.
   514)       if (error) then
   515)          write(iulog,*) '   clm_ptrs_check: c mono increasing - ERROR'
   516)          call endrun()
   517)       endif
   518)     enddo
   519)     if (masterproc) write(iulog,*) '   clm_ptrs_check: c mono increasing - OK'
   520) 
   521)     error = .false.
   522)     do p=begp+1,endp
   523)       if (pptr%gridcell(p) < pptr%gridcell(p-1)) error = .true.
   524)       if (pptr%landunit(p) < pptr%landunit(p-1)) error = .true.
   525)       if (pptr%column  (p) < pptr%column  (p-1)) error = .true.

   527)          write(iulog,*) '   clm_ptrs_check: p mono increasing - ERROR'

   529)       endif
   530)     enddo
   531)     if (masterproc) write(iulog,*) '   clm_ptrs_check: p mono increasing - OK'

   534)     error = .false.
   535)     do g = begg, endg
   536)        do l = gptr%luni(g),gptr%lunf(g)
   537)           if (lptr%gridcell(l) /= g) error = .true.
   538)           do c = lptr%coli(l),lptr%colf(l)
   539)              if (cptr%gridcell(c) /= g) error = .true.
   540)              if (cptr%landunit(c) /= l) error = .true.
   541)              do p = cptr%pfti(c),cptr%pftf(c)
   542)                 if (pptr%gridcell(p) /= g) error = .true.
   543)                 if (pptr%landunit(p) /= l) error = .true.
   544)                 if (pptr%column(p)   /= c) error = .true.
   545)                 if (error) then
   546)                    write(iulog,*) '   clm_ptrs_check: tree consistent - ERROR'
   547)                    call endrun()
   548)                 endif
   549)              enddo
   550)           enddo
   551)        enddo
   552)     enddo
   553)     if (masterproc) write(iulog,*) '   clm_ptrs_check: tree consistent - OK'
   554)     if (masterproc) write(iulog,*) ' '
   555)

   717) ! !INTERFACE:
   718) !  subroutine set_landunit_wet_ice_lake (ltype, wtxy, vegxy, &
   719)   subroutine set_landunit_wet_ice_lake (ltype, &
   720)                            nw, gi, li, ci, pi, setdata, glcmask)
   721) !
   722) ! !DESCRIPTION: 
   723) ! Initialize wet_ice_lake landunits that are non-urban (lake, wetland, glacier, glacier_mec)
   724) !
   725) ! !USES
   726)     use clmtype 
   727)     use subgridMod, only : subgrid_get_gcellinfo
   728)     use clm_varcon, only : istice, istwet, istdlak, istice_mec
   729)     use clm_varpar, only : npatch_lake, npatch_glacier, npatch_wet
   730)     use clm_varpar, only : npatch_glacier_mec
   731) 
   732) !
   733) ! !ARGUMENTS:
   734)     implicit none
   735)     integer , intent(in)    :: ltype             ! landunit type
   736) !    real(r8), intent(in)    :: wtxy(:,:)         ! subgrid patch weights
   737) !    integer , intent(in)    :: vegxy(:,:)        ! PFT types 
   738)     integer , intent(in)    :: nw                ! cell index
   739)     integer , intent(in)    :: gi                ! gridcell index
   740)     integer , intent(inout) :: li                ! landunit index
   741)     integer , intent(inout) :: ci                ! column index
   742)     integer , intent(inout) :: pi                ! pft index
   743)     logical , intent(in)    :: setdata           ! set info or just compute
   744)     integer , intent(in), optional :: glcmask    ! = 1 where glc requires sfc mass balance
   745)                                                  ! = 0 otherwise
   746) !
   747) ! !REVISION HISTORY:
   748) ! Created by Sam Levis
   749) ! 2005.11.25 Updated by T Craig
   750) !
   751) !
   752) ! !LOCAL VARIABLES:
   753) !EOP
   754)     integer  :: m                                ! m index in wtxy(nw,m)
   755)     integer  :: c                                ! column loop index
   756)     integer  :: ctype                            ! column type
   757)     integer  :: ier                              ! error status 
   758)     integer  :: npfts                            ! number of pfts in landunit
   759)     integer  :: ncols                            ! number of columns in landu
   760)     real(r8) :: wtlunit2gcell                    ! landunit weight in gridcell
   761)     real(r8) :: wtcol2lunit                      ! col weight in landunit
   762)     type(landunit_type), pointer :: lptr         ! pointer to landunit
   763)     type(column_type)  , pointer :: cptr         ! pointer to column
   764)     type(pft_type)     , pointer :: pptr         ! pointer to pft
   765) 
   766) !------------------------------------------------------------------------
   767) 
   768)     ! Set decomposition properties
   769) 
   770)     if (ltype == istwet) then
   771) !       call subgrid_get_gcellinfo(nw, wtxy, nwetland=npfts, wtwetland=wtlunit2gcell)
   772)        call subgrid_get_gcellinfo(nw, nwetland=npfts, wtwetland=wtlunit2gcell)
   773)        m = npatch_wet
   774)     else if (ltype == istdlak) then
   775) !       call subgrid_get_gcellinfo(nw, wtxy, nlake=npfts, wtlake=wtlunit2gcell)
   776)        call subgrid_get_gcellinfo(nw, nlake=npfts, wtlake=wtlunit2gcell)
   777)        m = npatch_lake
   778)     else if (ltype == istice) then 
   779) !       call subgrid_get_gcellinfo(nw, wtxy, nglacier=npfts, wtglacier=wtlunit2gcell)
   780)        call subgrid_get_gcellinfo(nw, nglacier=npfts, wtglacier=wtlunit2gcell)
   781)        m = npatch_glacier
   782)     else if (ltype == istice_mec) then
   783) !       call subgrid_get_gcellinfo(nw, wtxy, nglacier_mec=npfts, wtglacier_mec=wtlunit2gcell)
   784)        call subgrid_get_gcellinfo(nw, nglacier_mec=npfts, wtglacier_mec=wtlunit2gcell, &
   785)                                   glcmask = glcmask)
   786)        ! NOTE: multiple columns per landunit, so m is not set here
   787) 
   788)     else
   789)        write(iulog,*)' set_landunit_wet_ice_lake: ltype of ',ltype,' not valid'
   790)        write(iulog,*)' only istwet, istdlak, istice and istice_mec ltypes are valid'
   791)        call endrun()
   792)     end if
   793) 
   794)     if (npfts > 0) then
   795) 
   796)        ! Set pointers into derived types for this module
   797) 
   798)        lptr => lun
   799)        cptr => col
   800)        pptr => pft
   801) 
   802)        if (npfts /=1 .and. ltype /= istice_mec) then
   803)           write(iulog,*)' set_landunit_wet_ice_lake: compete landunit must'// &
   804)                     ' have one column and one pft '

   806)           call endrun()
   807)        end if
   808)

   810) 
   811)           ! Assume that columns are of type 1 and that each column has its own pft
   812) 
   813)           ctype = 1
   814)           li = li + 1
   815)

   817) 
   818)              ! Determine landunit properties
   819)

   821)              lptr%ifspecial(li) = .true.
   822)              lptr%glcmecpoi(li) = .true.
   823)              lptr%lakpoi   (li) = .false.

   826)              lptr%wtgcell  (li) = wtlunit2gcell
   827) 
   828)              ! Determine column and properties
   829)              ! (Each column has its own pft)
   830)              ! 
   831)              ! For grid cells with glcmask = 1, make sure all the elevations classes
   832)              !  are populated, even if some have zero fractional area.  This ensures that the 
   833)              !  ice sheet component, glc, will receive a surface mass balance in each elevation 
   834)              !  class wherever the SMB is needed.
   835)              ! Columns with zero weight are referred to as "virtual" columns.

   842)                    pi = pi + 1

   844)                       wtcol2lunit = wtxy(nw,m)/wtlunit2gcell

   850)                    cptr%gridcell (ci) = gi
   851)                    cptr%wtgcell  (ci) = wtcol2lunit * wtlunit2gcell
   852)                    cptr%landunit (ci) = li
   853)                    cptr%wtlunit  (ci) = wtcol2lunit
   854) 
   855)                    ! Set sfc elevation too
   856) 
   857)                    cps%glc_topo(ci) = topoxy(nw,m)
   858) 
   859)                    ! Set pft properties

   862)                    pptr%itype    (pi) = vegxy(nw,m)
   863)                    pptr%gridcell (pi) = gi
   864)                    pptr%wtgcell  (pi) = wtcol2lunit * wtlunit2gcell
   865)                    pptr%landunit (pi) = li
   866)                    pptr%wtlunit  (pi) = wtcol2lunit
   867)                    pptr%column   (pi) = ci
   868)                    pptr%wtcol    (pi) = 1.0_r8
   869) 
   870)                 endif   ! wtxy > 0 or glcmask = 1
   871)              enddo      ! loop over columns
   872)           endif         ! setdata
   873) 
   874)        else
   875) 
   876)           ncols = 1
   877) 
   878)           ! Currently assume that each landunit only has only one column 
   879)           ! (of type 1) and that each column has its own pft
   880)        
   881)           wtcol2lunit = 1.0_r8/ncols
   882)           ctype = 1
   883) 
   884)           li = li + 1
   885)           ci = ci + 1
   886)           pi = pi + 1
   887)

   897)              else
   898)                 lptr%lakpoi(li) = .false.
   899)              end if
   900)        
   901)              lptr%gridcell (li) = gi
   902)              lptr%wtgcell(li) = wtlunit2gcell
   903) 
   904)              ! Determine column and properties
   905)              ! For the wet, ice or lake landunits it is assumed that each 
   906)              ! column has its own pft
   907)        
   908)              cptr%itype(ci)    = ctype
   909)        
   910)              cptr%gridcell (ci) = gi
   911)              cptr%wtgcell(ci) = wtcol2lunit * wtlunit2gcell
   912)              cptr%landunit (ci) = li
   913)              cptr%wtlunit(ci) = wtcol2lunit
   914) 
   915)              ! Set pft properties
   916) 
   917)              pptr%mxy(pi)      = m
   918)              pptr%itype(pi)    = vegxy(nw,m)
   919)      
   920)              pptr%gridcell (pi) = gi
   921)              pptr%wtgcell(pi) = wtcol2lunit * wtlunit2gcell
   922)              pptr%landunit (pi) = li
   923)              pptr%wtlunit(pi) = wtcol2lunit
   924)              pptr%column (pi) = ci
   925)              pptr%wtcol(pi) = 1.0_r8
   926)           endif ! setdata
   927)        end if   ! ltype = istice_mec
   928)     endif       ! npfts > 0       
   929) 
   930)   end subroutine set_landunit_wet_ice_lake
