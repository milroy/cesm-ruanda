   120)       subroutine get_xsqy( xs_long_file, lng_indexer )
   121) !=============================================================================!
   122) !   PURPOSE:                                                                  !
   123) !   Reads a NetCDF file that contains:                                        !
   124) !     cross section * QY temperature dependence, >200nm                       !
   125) !                                                                             !
   126) !=============================================================================!
   127) !   PARAMETERS:                                                               !
   128) !     Input:                                                                  !
   129) !      filepath.... NetCDF filepath that contains the "cross sections"        !
   130) !=============================================================================!
   131) !   EDIT HISTORY:                                                             !
   132) !   Created by Doug Kinnison, 3/14/2002                                       !
   133) !=============================================================================!
   134) 
   135)       use ioFileMod,      only : getfil
   136)       use error_messages, only : alloc_err
   137)       use chem_mods,      only : phtcnt, pht_alias_lst, rxt_tag_lst
   138)       use netcdf, only: &
   139)            nf90_open, &
   140)            nf90_nowrite, &
   141)            nf90_inq_dimid, &
   142)            nf90_inquire_dimension, &
   143)            nf90_inq_varid, &
   144)            nf90_noerr, &
   145)            nf90_get_var, &
   146)            nf90_close
   147) 
   148) !------------------------------------------------------------------------------
   149) !    ... Dummy arguments
   150) !------------------------------------------------------------------------------
   151)       integer, intent(inout) :: lng_indexer(:)
   152)       character(len=*) :: xs_long_file
   153) 
   154) !------------------------------------------------------------------------------
   155) !       ... Local variables
   156) !------------------------------------------------------------------------------
   157)       integer :: varid, dimid, ndx
   158)       integer :: ncid
   159)       integer :: iret
   160)       integer :: i, k, m, n
   161)       integer :: wrk_ndx(phtcnt)
   162)       character(len=256) :: locfn
   163) 
   164)       Masterproc_only : if( masterproc ) then
   165)          !------------------------------------------------------------------------------
   166)          !       ... open NetCDF File
   167)          !------------------------------------------------------------------------------
   168)          call getfil(xs_long_file, locfn, 0)
   169)          iret = nf90_open(trim(locfn), NF90_NOWRITE, ncid)
   170) 
   171)          !------------------------------------------------------------------------------
   172)          !       ... get dimensions
   173)          !------------------------------------------------------------------------------
   174)          iret = nf90_inq_dimid( ncid, 'numtemp', dimid )
   175)          iret = nf90_inquire_dimension( ncid, dimid,len= nt )
   176)          iret = nf90_inq_dimid( ncid, 'numwl', dimid )
   177)          iret = nf90_inquire_dimension( ncid, dimid,len= nw )
   178)          iret = nf90_inq_dimid( ncid, 'numprs', dimid )
   179)          iret = nf90_inquire_dimension( ncid, dimid,len= np_xs )
   180)          !------------------------------------------------------------------------------
   181)          !       ... check for cross section in dataset
   182)          !------------------------------------------------------------------------------
   183)          do m = 1,phtcnt
   184)             if( pht_alias_lst(m,2) == ' ' ) then
   185)                iret = nf90_inq_varid( ncid, rxt_tag_lst(m), varid )
   186)                if( iret == nf90_noerr ) then 
   187)                   lng_indexer(m) = varid
   188)                end if
   189)             else if( pht_alias_lst(m,2) == 'userdefined' ) then
   190)                lng_indexer(m) = -1
   191)             else
   192)                iret = nf90_inq_varid( ncid, pht_alias_lst(m,2), varid )
   193)                if( iret == nf90_noerr ) then 
   194)                   lng_indexer(m) = varid
   195)                else
   196)                   write(iulog,*) 'get_xsqy : ',rxt_tag_lst(m)(:len_trim(rxt_tag_lst(m))),' alias ', &
   197)                        pht_alias_lst(m,2)(:len_trim(pht_alias_lst(m,2))),' not in dataset'            
   198)                   call endrun
   199)                end if
   200)             end if
   201)          end do
   202)          numj = 0
   203)          do m = 1,phtcnt
   204)             if( lng_indexer(m) > 0 ) then
   205)                if( any( lng_indexer(:m-1) == lng_indexer(m) ) ) then
   206)                   cycle
   207)                end if
   208)                numj = numj + 1
   209)             end if
   210)          end do
   211) 
   212)          !------------------------------------------------------------------------------
   213)          !       ... allocate arrays
   214)          !------------------------------------------------------------------------------
   215) 
   216)          allocate( xsqy(numj,nw,nt,np_xs),stat=iret )
   217)          if( iret /= 0 ) then 
   218)             call alloc_err( iret, 'get_xsqy', 'xsqy', numj*nt*np_xs*nw )
   219)          end if
   220)          allocate( prs(np_xs),dprs(np_xs-1),stat=iret )
   221)          if( iret /= 0 ) then 
   222)             call alloc_err( iret, 'get_xsqy', 'prs,dprs', np_xs )
   223)          end if
   224)          allocate( xs_o2b(nw,nt,np_xs),xs_o3a(nw,nt,np_xs),xs_o3b(nw,nt,np_xs),stat=iret )
   225)          if( iret /= 0 ) then 
   226)             call alloc_err( iret, 'get_xsqy', 'xs_o2b ... xs_o3b', np_xs )
   227)          end if
   228)          !------------------------------------------------------------------------------
   229)          !       ... read cross sections
   230)          !------------------------------------------------------------------------------
   231)          ndx = 0
   232)          do m = 1,phtcnt
   233)             if( lng_indexer(m) > 0 ) then
   234)                if( any( lng_indexer(:m-1) == lng_indexer(m) ) ) then
   235)                   cycle
   236)                end if
   237)                ndx = ndx + 1
   238)                iret = nf90_get_var( ncid, lng_indexer(m), xsqy(ndx,:,:,:) )
   239)             end if
   240)          end do
   241)          if( ndx /= numj ) then
   242)             write(iulog,*) 'get_xsqy : ndx count /= cross section count'
   243)             call endrun
   244)          end if
   245)          iret = nf90_inq_varid( ncid, 'jo2_b', varid )
   246)          iret = nf90_get_var( ncid, varid, xs_o2b )
   247)          iret = nf90_inq_varid( ncid, 'jo3_a', varid )
   248)          iret = nf90_get_var( ncid, varid, xs_o3a )
   249)          iret = nf90_inq_varid( ncid, 'jo3_b', varid )
   250)          iret = nf90_get_var( ncid, varid, xs_o3b )
   251)          !------------------------------------------------------------------------------
   252)          !       ... setup final lng_indexer
   253)          !------------------------------------------------------------------------------
   254)          ndx = 0
   255)          wrk_ndx(:) = lng_indexer(:)
   256)          do m = 1,phtcnt
   257)             if( wrk_ndx(m) > 0 ) then
   258)                ndx = ndx + 1
   259)                i = wrk_ndx(m)
   260)                where( wrk_ndx(:) == i )
   261)                   lng_indexer(:) = ndx
   262)                   wrk_ndx(:)     = -100000
   263)                end where
   264)             end if
   265)          end do
   266) 
   267)          iret = nf90_inq_varid( ncid, 'pressure', varid )
   268)          iret = nf90_get_var( ncid, varid, prs )
   269)          iret = nf90_close( ncid )
   270)       end if Masterproc_only
   271) 
   272) #ifdef SPMD
   273) !        call mpibarrier( mpicom )
   274)       call mpibcast( numj,  1, mpiint, 0, mpicom )
   275)       call mpibcast( nt,    1, mpiint, 0, mpicom )
   276)       call mpibcast( nw,    1, mpiint, 0, mpicom )
   277)       call mpibcast( np_xs, 1, mpiint, 0, mpicom )
   278)       call mpibcast( lng_indexer, phtcnt, mpiint, 0, mpicom )
   279) #endif
   280)       if( .not. masterproc ) then
   281)          !------------------------------------------------------------------------------
   282)          !       ... allocate arrays
   283)          !------------------------------------------------------------------------------
   284)          allocate( xsqy(numj,nw,nt,np_xs),stat=iret )
   285)          if( iret /= nf90_noerr) then 
   286)             write(iulog,*) 'get_xsqy : failed to allocate xsqy ; error = ',iret
   287)             call endrun
   288)          end if
   289)          allocate( prs(np_xs),dprs(np_xs-1),stat=iret )
   290)          if( iret /= nf90_noerr) then 
   291)             write(iulog,*) 'get_xsqy : failed to allocate prs,dprs ; error = ',iret
   292)             call endrun
   293)          end if
   294)          allocate( xs_o2b(nw,nt,np_xs),xs_o3a(nw,nt,np_xs),xs_o3b(nw,nt,np_xs),stat=iret )
   295)          if( iret /= 0 ) then 
   296)             call alloc_err( iret, 'get_xsqy', 'xs_o2b ... xs_o3b', np_xs )
   297)          end if
   298)       end if
   299) #ifdef SPMD
   300) !        call mpibarrier( mpicom )
   301)       call mpibcast( prs, np_xs, mpir8, 0, mpicom )
   302)       call mpibcast( xsqy, numj*nt*np_xs*nw, mpir4, 0, mpicom )
   303)       call mpibcast( xs_o2b, nw*nt*np_xs, mpir8, 0, mpicom )
   304)       call mpibcast( xs_o3a, nw*nt*np_xs, mpir8, 0, mpicom )
   305)       call mpibcast( xs_o3b, nw*nt*np_xs, mpir8, 0, mpicom )
   306) #endif
   307)       dprs(:np_xs-1) = 1._r8/(prs(1:np_xs-1) - prs(2:np_xs))
   308) 
   309)       end subroutine get_xsqy

   311)       subroutine get_rsf(rsf_file)
   312) !=============================================================================!
   313) !   PURPOSE:                                                                  !
   314) !   Reads a NetCDF file that contains:
   315) !     Radiative Souce function                                                !
   316) !=============================================================================!
   317) !   PARAMETERS:                                                               !
   318) !     Input:                                                                  !
   319) !      filepath.... NetCDF file that contains the RSF                         !
   320) !                                                                             !
   321) !     Output:                                                                 !
   322) !      rsf ........ Radiative Source Function (quanta cm-2 sec-1              !
   323) !                                                                             !
   324) !   EDIT HISTORY:                                                             !
   325) !   Created by Doug Kinnison, 3/14/2002                                       !
   326) !=============================================================================!
   327) 
   328)       use ioFileMod,      only : getfil
   329)       use error_messages, only : alloc_err
   330)       use netcdf, only: &
   331)            nf90_open, &
   332)            nf90_nowrite, &
   333)            nf90_inq_dimid, &
   334)            nf90_inquire_dimension, &
   335)            nf90_inq_varid, &
   336)            nf90_noerr, &
   337)            nf90_get_var, &
   338)            nf90_close
   339) 
   340) !------------------------------------------------------------------------------
   341) !    ... dummy arguments
   342) !------------------------------------------------------------------------------
   343)       character(len=*) :: rsf_file
   344) 
   345) !------------------------------------------------------------------------------
   346) !       ... local variables
   347) !------------------------------------------------------------------------------
   348)       integer :: varid, dimid
   349)       integer :: ncid
   350)       integer :: i, j, k, l, w
   351)       integer :: iret
   352)       integer :: count(5)
   353)       integer :: start(5)
   354)       real(r8) :: wrk
   355)       character(len=256) :: locfn
   356) 
   357)       Masterproc_only : if( masterproc ) then
   358)          !------------------------------------------------------------------------------
   359)          !       ... open NetCDF File
   360)          !------------------------------------------------------------------------------
   361)          call getfil(rsf_file, locfn, 0)
   362)          iret = nf90_open(trim(locfn), NF90_NOWRITE, ncid)
   363) 
   364)          !------------------------------------------------------------------------------
   365)          !       ... get dimensions
   366)          !------------------------------------------------------------------------------
   367)          iret = nf90_inq_dimid( ncid, 'numz', dimid )
   368)          iret = nf90_inquire_dimension( ncid, dimid,len= nump )
   369)          iret = nf90_inq_dimid( ncid, 'numsza', dimid )
   370)          iret = nf90_inquire_dimension( ncid, dimid,len= numsza )
   371)          iret = nf90_inq_dimid( ncid, 'numalb', dimid )
   372)          iret = nf90_inquire_dimension( ncid, dimid,len= numalb )
   373)          iret = nf90_inq_dimid( ncid, 'numcolo3fact', dimid )
   374)          iret = nf90_inquire_dimension( ncid, dimid,len= numcolo3 )
   375) 
   376)       end if Masterproc_only
   377) #ifdef SPMD
   378) !        call mpibarrier( mpicom )
   379)       call mpibcast( nump,     1, mpiint, 0, mpicom )
   380)       call mpibcast( numsza,   1, mpiint, 0, mpicom )
   381)       call mpibcast( numalb,   1, mpiint, 0, mpicom )
   382)       call mpibcast( numcolo3, 1, mpiint, 0, mpicom )
   383) #endif
   384) !------------------------------------------------------------------------------
   385) !       ... allocate arrays
   386) !------------------------------------------------------------------------------
   387)       allocate( wc(nw),stat=iret )
   388)       if( iret /= 0 ) then 
   389)          call alloc_err( iret, 'get_rsf', 'wc', nw )
   390)       end if
   391)       allocate( wlintv(nw),we(nw+1),etfphot(nw),stat=iret )
   392)       if( iret /= 0 ) then 
   393)          call alloc_err( iret, 'get_rsf', 'wlintv,etfphot', nw )
   394)       end if
   395)       allocate( bde_o2_b(nw),bde_o3_a(nw),bde_o3_b(nw),stat=iret )
   396)       if( iret /= 0 ) then 
   397)          call alloc_err( iret, 'get_rsf', 'bde', nw )
   398)       end if
   399)       allocate( p(nump),del_p(nump-1),stat=iret )
   400)       if( iret /= 0 ) then 
   401)          call alloc_err( iret, 'get_rsf', 'p,delp', nump )
   402)       end if
   403)       allocate( sza(numsza),del_sza(numsza-1),stat=iret )
   404)       if( iret /= 0 ) then 
   405)          call alloc_err( iret, 'get_rsf', 'sza,del_sza', numsza )
   406)       end if
   407)       allocate( alb(numalb),del_alb(numalb-1),stat=iret )
   408)       if( iret /= 0 ) then 
   409)          call alloc_err( iret, 'get_rsf', 'alb,del_alb', numalb )
   410)       end if
   411)       allocate( o3rat(numcolo3),del_o3rat(numcolo3-1),stat=iret )
   412)       if( iret /= 0 ) then 
   413)          call alloc_err( iret, 'get_rsf', 'o3rat,del_o3rat', numcolo3 )
   414)       end if
   415)       allocate( colo3(nump),stat=iret )
   416)       if( iret /= 0 ) then 
   417)          call alloc_err( iret, 'get_rsf', 'colo3', nump )
   418)       end if
   419)       allocate( rsf_tab(nw,nump,numsza,numcolo3,numalb),stat=iret )
   420)       if( iret /= 0 ) then 
   421)          write(iulog,*) 'get_rsf : dimensions = ',nw,nump,numsza,numcolo3,numalb
   422)          call alloc_err( iret, 'get_rsf', 'rsf_tab', numalb*numcolo3*numsza*nump )
   423)       end if
   424) 
   425)       Masterproc_only2 : if( masterproc ) then
   426)          !------------------------------------------------------------------------------
   427)          !       ... read variables
   428)          !------------------------------------------------------------------------------
   429)          iret = nf90_inq_varid( ncid, 'wc', varid )
   430)          iret = nf90_get_var( ncid, varid, wc )
   431)          iret = nf90_inq_varid( ncid, 'wlintv', varid )
   432)          iret = nf90_get_var( ncid, varid, wlintv )
   433)          iret = nf90_inq_varid( ncid, 'pm', varid )
   434)          iret = nf90_get_var( ncid, varid, p )
   435)          iret = nf90_inq_varid( ncid, 'sza', varid )
   436)          iret = nf90_get_var( ncid, varid, sza )
   437)          iret = nf90_inq_varid( ncid, 'alb', varid )
   438)          iret = nf90_get_var( ncid, varid, alb )
   439)          iret = nf90_inq_varid( ncid, 'colo3fact', varid )
   440)          iret = nf90_get_var( ncid, varid, o3rat )
   441)          iret = nf90_inq_varid( ncid, 'colo3', varid )
   442)          iret = nf90_get_var( ncid, varid, colo3 )
   443) 
   444)          iret = nf90_inq_varid( ncid, 'RSF', varid )
   445)          
   446)          if (masterproc) then
   447)             write(iulog,*) ' '
   448)             write(iulog,*) '----------------------------------------------'
   449)             write(iulog,*) 'get_rsf: numalb, numcolo3, numsza, nump = ', &
   450)                  numalb, numcolo3, numsza, nump
   451)             write(iulog,*) 'get_rsf: size of rsf_tab = ',size(rsf_tab,dim=1),size(rsf_tab,dim=2), &
   452)                  size(rsf_tab,dim=3),size(rsf_tab,dim=4),size(rsf_tab,dim=5)
   453)             write(iulog,*) '----------------------------------------------'
   454)             write(iulog,*) ' '
   455)          endif
   456) 
   457)          iret = nf90_get_var( ncid, varid, rsf_tab )
   458)          iret = nf90_close( ncid )
   459) 
   460)          do w = 1,nw
   461)             wrk = wlintv(w)
   462)             rsf_tab(w,:,:,:,:) = wrk*rsf_tab(w,:,:,:,:)
   463)          enddo
   464) 
   465)       end if Masterproc_only2
   466) #ifdef SPMD
   467)       call mpibcast( wc,      nw,       mpir8, 0, mpicom )
   468)       call mpibcast( wlintv,  nw,       mpir8, 0, mpicom )
   469)       call mpibcast( p,       nump,     mpir8, 0, mpicom )
   470)       call mpibcast( sza,     numsza,   mpir8, 0, mpicom )
   471)       call mpibcast( alb,     numalb,   mpir8, 0, mpicom )
   472)       call mpibcast( o3rat,   numcolo3, mpir8, 0, mpicom )
   473)       call mpibcast( colo3,   nump,     mpir8, 0, mpicom )
   474)       do w = 1,nw
   475)          call mpibcast( rsf_tab(w,:,:,:,:), numalb*numcolo3*numsza*nump, mpir4, 0, mpicom )
   476)       enddo
   477) #endif
   478) #ifdef USE_BDE
   479)       if (masterproc) write(iulog,*) 'Jlong using bdes'
   480)       bde_o2_b(:) = max( 0._r8, hc*(wc_o2_b - wc(:))/(wc_o2_b*wc(:)) )
   481)       bde_o3_a(:) = max( 0._r8, hc*(wc_o3_a - wc(:))/(wc_o3_a*wc(:)) )
   482)       bde_o3_b(:) = max( 0._r8, hc*(wc_o3_b - wc(:))/(wc_o3_b*wc(:)) )
   483) #else
   484)       if (masterproc) write(iulog,*) 'Jlong not using bdes'
   485)       bde_o2_b(:) = hc/wc(:)
   486)       bde_o3_a(:) = hc/wc(:)
   487)       bde_o3_b(:) = hc/wc(:)
   488) #endif
   489) 
   490)       del_p(:nump-1)         = 1._r8/abs(p(1:nump-1)- p(2:nump))
   491)       del_sza(:numsza-1)     = 1._r8/(sza(2:numsza) - sza(1:numsza-1))
   492)       del_alb(:numalb-1)     = 1._r8/(alb(2:numalb) - alb(1:numalb-1))
   493)       del_o3rat(:numcolo3-1) = 1._r8/(o3rat(2:numcolo3) - o3rat(1:numcolo3-1))
   494) 
   495)       end subroutine get_rsf

   515)       subroutine jlong_hrates( nlev, sza_in, alb_in, p_in, t_in, &
   516)                                mw, o2_vmr, o3_vmr, colo3_in, qrl_col, &
   517)                                cparg, kbot )
   518) !==============================================================================
   519) !   Purpose:                                                                   
   520) !     To calculate the thermal heating rates longward of 200nm.        
   521) !==============================================================================
   522) !   Approach:
   523) !     1) Reads the Cross Section*QY NetCDF file
   524) !     2) Given a temperature profile, derives the appropriate XS*QY
   525) !
   526) !     3) Reads the Radiative Source function (RSF) NetCDF file
   527) !        Units = quanta cm-2 sec-1
   528) !
   529) !     4) Indices are supplied to select a RSF that is consistent with
   530) !        the reference atmosphere in TUV (for direct comparision of J's).
   531) !        This approach will be replaced in the global model. Here colo3, zenith
   532) !        angle, and altitude will be inputed and the correct entry in the table
   533) !        will be derived.
   534) !==============================================================================
   535) 
   536) 	use physconst,       only : avogad
   537)         use error_messages, only : alloc_err
   538) 
   539) 	implicit none
   540) 
   541) !------------------------------------------------------------------------------
   542) !    	... dummy arguments
   543) !------------------------------------------------------------------------------
   544)       integer, intent (in)     :: nlev               ! number vertical levels
   545)       integer, intent (in)     :: kbot               ! heating levels
   546)       real(r8), intent(in)     :: o2_vmr(nlev)       ! o2 conc (mol/mol)
   547)       real(r8), intent(in)     :: o3_vmr(nlev)       ! o3 conc (mol/mol)
   548)       real(r8), intent(in)     :: sza_in             ! solar zenith angle (degrees)
   549)       real(r8), intent(in)     :: alb_in(nlev)       ! albedo
   550)       real(r8), intent(in)     :: p_in(nlev)         ! midpoint pressure (hPa)
   551)       real(r8), intent(in)     :: t_in(nlev)         ! Temperature profile (K)
   552)       real(r8), intent(in)     :: colo3_in(nlev)     ! o3 column density (molecules/cm^3)
   553)       real(r8), intent(in)     :: mw(nlev)           ! atms molecular weight
   554)       real(r8), intent(in)     :: cparg(nlev)        ! specific heat capacity
   555)       real(r8), intent(inout)  :: qrl_col(:,:)	     ! heating rates
   556) 
   557) !----------------------------------------------------------------------
   558) !  	... local variables
   559) !----------------------------------------------------------------------
   560)       integer  ::  astat
   561)       integer  ::  k, km, m
   562)       integer  ::  t_index					! Temperature index
   563)       integer  ::  pndx
   564)       real(r8) ::  ptarget
   565)       real(r8) ::  delp
   566)       real(r8) ::  hfactor
   567)       real(r8), allocatable                :: rsf(:,:)	        ! Radiative source function
   568)       real(r8), allocatable                :: xswk(:,:)	        ! working xsection array
   569)       real(r8), allocatable                :: wrk(:)	        ! work vector
   570) 
   571) !----------------------------------------------------------------------
   572) !        ... allocate variables
   573) !----------------------------------------------------------------------
   574)       allocate( rsf(nw,kbot),stat=astat )
   575)       if( astat /= 0 ) then
   576)          call alloc_err( astat, 'jlong_hrates', 'rsf', nw*nlev )
   577)       end if
   578)       allocate( xswk(nw,3),wrk(nw),stat=astat )
   579)       if( astat /= 0 ) then
   580)          call alloc_err( astat, 'jlong_hrates', 'xswk,wrk', 3*nw )
   581)       end if
   582) 
   583) !----------------------------------------------------------------------
   584) !        ... interpolate table rsf to model variables
   585) !----------------------------------------------------------------------
   586)       call interpolate_rsf( alb_in, sza_in, p_in, colo3_in, kbot, rsf )
   587) 
   588) !------------------------------------------------------------------------------
   589) !     ... calculate thermal heating rates for wavelengths >200nm
   590) !------------------------------------------------------------------------------
   591) !------------------------------------------------------------------------------
   592) !     LLNL LUT approach to finding temperature index...
   593) !     Calculate the temperature index into the cross section
   594) !     data which lists coss sections for temperatures from
   595) !     150 to 350 degrees K.  Make sure the index is a value
   596) !     between 1 and 201.
   597) !------------------------------------------------------------------------------
   598)       qrl_col(kbot+1:nlev,:) = 0._r8
   599) level_loop_1 : &
   600)       do k = 1,kbot
   601) !----------------------------------------------------------------------
   602) !   	... get index into xsqy
   603) !----------------------------------------------------------------------
   604)         t_index = t_in(k) - 148.5_r8
   605)         t_index = min( 201,max( t_index,1) )
   606) !----------------------------------------------------------------------
   607) !   	... find pressure level
   608) !----------------------------------------------------------------------
   609) 	ptarget = p_in(k)
   610) 	if( ptarget >= prs(1) ) then
   611) 	   xswk(:,1) = xs_o2b(:,t_index,1)
   612) 	   xswk(:,2) = xs_o3a(:,t_index,1)
   613) 	   xswk(:,3) = xs_o3b(:,t_index,1)
   614) 	else if( ptarget <= prs(np_xs) ) then
   615) 	   xswk(:,1) = xs_o2b(:,t_index,np_xs)
   616) 	   xswk(:,2) = xs_o3a(:,t_index,np_xs)
   617) 	   xswk(:,3) = xs_o3b(:,t_index,np_xs)
   618) 	else
   619) 	   do km = 2,np_xs
   620) 	      if( ptarget >= prs(km) ) then
   621) 		 pndx = km - 1
   622) 		 delp = (prs(pndx) - ptarget)*dprs(pndx)
   623) 		 exit
   624) 	      end if
   625) 	   end do
   626) 	   xswk(:,1) = xs_o2b(:,t_index,pndx) &
   627)                        + delp*(xs_o2b(:,t_index,pndx+1) - xs_o2b(:,t_index,pndx))
   628) 	   xswk(:,2) = xs_o3a(:,t_index,pndx) &
   629)                        + delp*(xs_o3a(:,t_index,pndx+1) - xs_o3a(:,t_index,pndx))
   630) 	   xswk(:,3) = xs_o3b(:,t_index,pndx) &
   631)                        + delp*(xs_o3b(:,t_index,pndx+1) - xs_o3b(:,t_index,pndx))
   632) 	end if
   633) 	hfactor      = avogad/(cparg(k)*mw(k))
   634) 	wrk(:)       = xswk(:,1)*bde_o2_b(:)
   635) 	qrl_col(k,2) = dot_product( wrk(:),rsf(:,k) ) * o2_vmr(k) * hfactor
   636) 	wrk(:)       = xswk(:,2)*bde_o3_a(:)
   637) 	qrl_col(k,3) = dot_product( wrk(:),rsf(:,k) ) * o3_vmr(k) * hfactor
   638) 	wrk(:)       = xswk(:,3)*bde_o3_b(:)
   639) 	qrl_col(k,4) = dot_product( wrk(:),rsf(:,k) ) * o3_vmr(k) * hfactor
   640)       end do level_loop_1
   641) 
   642)       deallocate( rsf, xswk, wrk )
   643) 
   644)       end subroutine jlong_hrates
   645) 
   646)        subroutine jlong_photo( nlev, sza_in, alb_in, p_in, t_in, &
   647)                               colo3_in, j_long )
   648) !==============================================================================
   649) !   Purpose:                                                                   
   650) !     To calculate the total J for selective species longward of 200nm.        
   651) !==============================================================================
   652) !   Approach:
   653) !     1) Reads the Cross Section*QY NetCDF file
   654) !     2) Given a temperature profile, derives the appropriate XS*QY
   655) !
   656) !     3) Reads the Radiative Source function (RSF) NetCDF file
   657) !        Units = quanta cm-2 sec-1
   658) !
   659) !     4) Indices are supplied to select a RSF that is consistent with
   660) !        the reference atmosphere in TUV (for direct comparision of J's).
   661) !        This approach will be replaced in the global model. Here colo3, zenith
   662) !        angle, and altitude will be inputed and the correct entry in the table
   663) !        will be derived.
   664) !==============================================================================
   665) 
   666)  use spmd_utils,   only : masterproc
   667)         use error_messages, only : alloc_err
   668) 
   669) 	implicit none
   670) 
   671) !------------------------------------------------------------------------------
   672) !    	... dummy arguments
   673) !------------------------------------------------------------------------------
   674)       integer, intent (in)     :: nlev               ! number vertical levels
   675)       real(r8), intent(in)     :: sza_in             ! solar zenith angle (degrees)
   676)       real(r8), intent(in)     :: alb_in(nlev)       ! albedo
   677)       real(r8), intent(in)     :: p_in(nlev)         ! midpoint pressure (hPa)
   678)       real(r8), intent(in)     :: t_in(nlev)         ! Temperature profile (K)
   679)       real(r8), intent(in)     :: colo3_in(nlev)     ! o3 column density (molecules/cm^3)
   680)       real(r8), intent(out)    :: j_long(:,:)	     ! photo rates (1/s)
   681) 
   682) !----------------------------------------------------------------------
   683) !  	... local variables
   684) !----------------------------------------------------------------------
   685)       integer  ::  astat
   686)       integer  ::  k, km, m
   687)       integer  ::  wn
   688)       integer  ::  t_index					! Temperature index
   689)       integer  ::  pndx
   690)       real(r8) ::  ptarget
   691)       real(r8) ::  delp
   692)       real(r8) ::  hfactor
   693)       real(r8), allocatable :: rsf(:,:)	        ! Radiative source function
   694)       real(r8), allocatable :: xswk(:,:)	! working xsection array
   695) 
   696) !----------------------------------------------------------------------
   697) !        ... allocate variables
   698) !----------------------------------------------------------------------
   699)       allocate( rsf(nw,nlev),stat=astat )
   700)       if( astat /= 0 ) then
   701)          call alloc_err( astat, 'jlong_photo', 'rsf', nw*nlev )
   702)       end if
   703)       allocate( xswk(numj,nw),stat=astat )
   704)       if( astat /= 0 ) then
   705)          call alloc_err( astat, 'jlong_photo', 'xswk', numj*nw )
   706)       end if
   707) 
   708) !----------------------------------------------------------------------
   709) !        ... interpolate table rsf to model variables
   710) !----------------------------------------------------------------------
   711)       call interpolate_rsf( alb_in, sza_in, p_in, colo3_in, nlev, rsf )
   712) 
   713) !------------------------------------------------------------------------------
   714) !     ... calculate total Jlong for wavelengths >200nm
   715) !------------------------------------------------------------------------------
   716) !------------------------------------------------------------------------------
   717) !     LLNL LUT approach to finding temperature index...
   718) !     Calculate the temperature index into the cross section
   719) !     data which lists coss sections for temperatures from
   720) !     150 to 350 degrees K.  Make sure the index is a value
   721) !     between 1 and 201.
   722) !------------------------------------------------------------------------------
   723) level_loop_1 : &
   724)       do k = 1,nlev
   725) !----------------------------------------------------------------------
   726) !   	... get index into xsqy
   727) !----------------------------------------------------------------------
   728)         t_index = t_in(k) - 148.5_r8
   729)         t_index = min( 201,max( t_index,1) )
   730) !----------------------------------------------------------------------
   731) !   	... find pressure level
   732) !----------------------------------------------------------------------
   733) 	ptarget = p_in(k)
   734) 	if( ptarget >= prs(1) ) then
   735) 	   do wn = 1,nw
   736) 	      xswk(:,wn) = xsqy(:,wn,t_index,1)
   737) 	   end do
   738) 	else if( ptarget <= prs(np_xs) ) then
   739) 	   do wn = 1,nw
   740) 	      xswk(:,wn) = xsqy(:,wn,t_index,np_xs)
   741) 	   end do
   742) 	else
   743) 	   do km = 2,np_xs
   744) 	      if( ptarget >= prs(km) ) then
   745) 		 pndx = km - 1
   746) 		 delp = (prs(pndx) - ptarget)*dprs(pndx)
   747) 		 exit
   748) 	      end if
   749) 	   end do
   750) 	   do wn = 1,nw
   751) 	      xswk(:,wn) = xsqy(:,wn,t_index,pndx) &
   752)                            + delp*(xsqy(:,wn,t_index,pndx+1) - xsqy(:,wn,t_index,pndx))
   753) 	   end do
   754) 	end if
   755) #ifdef USE_ESSL
   756)         call dgemm( 'N', 'N', numj, 1, nw, &
   757) 		    1._r8, xswk, numj, rsf(1,k), nw, &
   758) 		    0._r8, j_long(1,k), numj )
   759) #else
   760)         j_long(:,k) = matmul( xswk(:,:),rsf(:,k) )
   761) #endif
   762)       end do level_loop_1
   763) 
   764)       deallocate( rsf, xswk )
   765) 
   766)       end subroutine jlong_photo
   767) 
   768) !----------------------------------------------------------------------
   769) !----------------------------------------------------------------------
   770) !        ... interpolate table rsf to model variables
   771) !----------------------------------------------------------------------
   772) !----------------------------------------------------------------------
   773)       subroutine interpolate_rsf( alb_in, sza_in, p_in, colo3_in, kbot, rsf )
   774) 
   775)         use error_messages, only : alloc_err
   776) 
   777)       implicit none
   778) 
   779) !------------------------------------------------------------------------------
   780) !    	... dummy arguments
   781) !------------------------------------------------------------------------------
   782)       real(r8), intent(in)  :: alb_in(:)       ! albedo
   783)       real(r8), intent(in)  :: sza_in          ! solar zenith angle (degrees)
   784)       integer,  intent(in)  :: kbot            ! heating levels
   785)       real(r8), intent(in)  :: p_in(:)         ! midpoint pressure (hPa)
   786)       real(r8), intent(in)  :: colo3_in(:)     ! o3 column density (molecules/cm^3)
   787)       real(r8), intent(out) :: rsf(:,:)
   788) 
   789) !----------------------------------------------------------------------
   790) !  	... local variables
   791) !----------------------------------------------------------------------
   792)       integer  ::  astat
   793)       integer  ::  is, iv, ial
   794)       integer  ::  isp1, ivp1, ialp1
   795)       real(r8), dimension(3)               :: dels
   796)       real(r8), dimension(0:1,0:1,0:1)     :: wghtl, wghtu
   797)       real(r8) ::  psum_u
   798)       real(r8), allocatable                :: psum_l(:)
   799)       real(r8) ::  v3ratl, v3ratu
   800)       integer  ::  pind, albind
   801)       real(r8) ::  wrk0, wrk1, wght1
   802)       integer  ::  iz, k, m
   803)       integer  ::  izl
   804)       integer  ::  ratindl, ratindu
   805)       integer  ::  wn
   806) 
   807) !----------------------------------------------------------------------
   808) !        ... allocate variables
   809) !----------------------------------------------------------------------
   810)       allocate( psum_l(nw),stat=astat )
   811)       if( astat /= 0 ) then
   812)          call alloc_err( astat, 'jlong_hrates', 'psum_l', nw )
   813)       end if
   814) 
   815) !----------------------------------------------------------------------
   816) !        ... find the zenith angle index ( same for all levels )
   817) !----------------------------------------------------------------------
   818)       do is = 1,numsza
   819)          if( sza(is) > sza_in ) then
   820)             exit
   821)          end if
   822)       end do
   823)       is   = max( min( is,numsza ) - 1,1 )
   824)       isp1 = is + 1
   825)       dels(1)  = max( 0._r8,min( 1._r8,(sza_in - sza(is)) * del_sza(is) ) )
   826)       wrk0     = 1._r8 - dels(1)
   827) 
   828)       izl = 2
   829) Level_loop : &
   830)       do k = kbot,1,-1
   831) !----------------------------------------------------------------------
   832) !        ... find albedo indicies
   833) !----------------------------------------------------------------------
   834)          do ial = 1,numalb
   835) 	    if( alb(ial) > alb_in(k) ) then
   836) 	       exit
   837) 	    end if
   838) 	 end do
   839) 	 albind = max( min( ial,numalb ) - 1,1 )
   840) !----------------------------------------------------------------------
   841) !        ... find pressure level indicies
   842) !----------------------------------------------------------------------
   843)          if( p_in(k) > p(1) ) then
   844)             pind  = 2
   845)             wght1 = 1._r8
   846)          else if( p_in(k) <= p(nump) ) then
   847)             pind  = nump
   848)             wght1 = 0._r8
   849)          else
   850)             do iz = izl,nump
   851)                if( p(iz) < p_in(k) ) then
   852) 	          izl = iz
   853) 	          exit
   854)                end if
   855)             end do
   856)             pind  = max( min( iz,nump ),2 )
   857)             wght1 = max( 0._r8,min( 1._r8,(p_in(k) - p(pind)) * del_p(pind-1) ) )
   858)          end if
   859) !----------------------------------------------------------------------
   860) !        ... find "o3 ratios"
   861) !----------------------------------------------------------------------
   862)          v3ratu  = colo3_in(k) / colo3(pind-1)
   863)          do iv = 1,numcolo3
   864)             if( o3rat(iv) > v3ratu ) then
   865)                exit
   866)             end if
   867)          end do
   868)          ratindu = max( min( iv,numcolo3 ) - 1,1 )
   869) 
   870)          if( colo3(pind) /= 0._r8 ) then
   871)             v3ratl = colo3_in(k) / colo3(pind)
   872)             do iv = 1,numcolo3
   873)                if( o3rat(iv) > v3ratl ) then
   874)                   exit
   875)                end if
   876)             end do
   877)             ratindl = max( min( iv,numcolo3 ) - 1,1 )
   878) 	 else
   879)             ratindl = ratindu
   880)             v3ratl  = o3rat(ratindu)
   881) 	 end if
   882) 
   883) !----------------------------------------------------------------------
   884) !        ... compute the weigths
   885) !----------------------------------------------------------------------
   886) 	 ial   = albind
   887) 	 ialp1 = ial + 1
   888) 	 iv    = ratindl
   889) 
   890)          dels(2)  = max( 0._r8,min( 1._r8,(v3ratl - o3rat(iv)) * del_o3rat(iv) ) )
   891)          dels(3)  = max( 0._r8,min( 1._r8,(alb_in(k) - alb(ial)) * del_alb(ial) ) )
   892) 
   893) 	 wrk1         = (1._r8 - dels(2))*(1._r8 - dels(3))
   894) 	 wghtl(0,0,0) = wrk0*wrk1
   895) 	 wghtl(1,0,0) = dels(1)*wrk1
   896) 	 wrk1         = (1._r8 - dels(2))*dels(3)
   897) 	 wghtl(0,0,1) = wrk0*wrk1
   898) 	 wghtl(1,0,1) = dels(1)*wrk1
   899) 	 wrk1         = dels(2)*(1._r8 - dels(3))
   900) 	 wghtl(0,1,0) = wrk0*wrk1
   901) 	 wghtl(1,1,0) = dels(1)*wrk1
   902) 	 wrk1         = dels(2)*dels(3)
   903) 	 wghtl(0,1,1) = wrk0*wrk1
   904) 	 wghtl(1,1,1) = dels(1)*wrk1
   905) 
   906) 	 iv  = ratindu
   907)          dels(2)  = max( 0._r8,min( 1._r8,(v3ratu - o3rat(iv)) * del_o3rat(iv) ) )
   908) 
   909) 	 wrk1         = (1._r8 - dels(2))*(1._r8 - dels(3))
   910) 	 wghtu(0,0,0) = wrk0*wrk1
   911) 	 wghtu(1,0,0) = dels(1)*wrk1
   912) 	 wrk1         = (1._r8 - dels(2))*dels(3)
   913) 	 wghtu(0,0,1) = wrk0*wrk1
   914) 	 wghtu(1,0,1) = dels(1)*wrk1
   915) 	 wrk1         = dels(2)*(1._r8 - dels(3))
   916) 	 wghtu(0,1,0) = wrk0*wrk1
   917) 	 wghtu(1,1,0) = dels(1)*wrk1
   918) 	 wrk1         = dels(2)*dels(3)
   919) 	 wghtu(0,1,1) = wrk0*wrk1
   920) 	 wghtu(1,1,1) = dels(1)*wrk1
   921) 
   922) 	 iz   = pind
   923) 	 iv   = ratindl
   924) 	 ivp1 = iv + 1
   925)          do wn = 1,nw
   926)             psum_l(wn) = wghtl(0,0,0) * rsf_tab(wn,iz,is,iv,ial) &
   927)                          + wghtl(0,0,1) * rsf_tab(wn,iz,is,iv,ialp1) &
   928)                          + wghtl(0,1,0) * rsf_tab(wn,iz,is,ivp1,ial) &
