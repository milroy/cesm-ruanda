   228)    icol_diag = -1

   230) !!$   if (nstep < 3) then

   241)    call cnst_get_ind( 'MSA',   l_msag, .false. )

   244)    l_nh4g = l_nh4g - loffset

   249)          '*** modal_aero_gasaerexch_sub -- cannot find H2SO4 species',   &

   252)    end if
   253)    do_nh4g = .false.
   254)    do_msag = .false.
   255)    if ((l_nh4g > 0) .and. (l_nh4g <= pcnstxx)) do_nh4g = .true.
   256)    if ((l_msag > 0) .and. (l_msag <= pcnstxx)) do_msag = .true.
   257)    do_soag = .false.
   258)    if ((method_soa == 1) .or. (method_soa == 2)) then
   259)       if ((l_soag > 0) .and. (l_soag <= pcnstxx)) do_soag = .true.
   260)    else if (method_soa /= 0) then
   261)       write(*,'(/a,1x,i10)') '*** modal_aero_gasaerexch_sub - bad method_soa =', method_soa
   262)       call endrun( 'modal_aero_gasaerexch_sub error' )
   263)    end if
   264)

   266)    dotend(:) = .false.
   267)    dotendqqcw(:) = .false.
   268)    ido_so4a(:) = 0
   269)    ido_nh4a(:) = 0
   270)    ido_soaa(:) = 0
   271) 
   272)    dotend(l_so4g) = .true.
   273)    if ( do_nh4g ) dotend(l_nh4g) = .true.
   274)    if ( do_msag ) dotend(l_msag) = .true.
   275)    if ( do_soag ) dotend(l_soag) = .true.
   276)    ntot_soamode = 0
   277)    do n = 1, ntot_amode
   278)       l = lptr_so4_a_amode(n)-loffset
   279)       if ((l > 0) .and. (l <= pcnstxx)) then
   280)          dotend(l) = .true.
   281)          ido_so4a(n) = 1
   282)          if ( do_nh4g ) then
   283)             l = lptr_nh4_a_amode(n)-loffset
   284)             if ((l > 0) .and. (l <= pcnstxx)) then
   285)                dotend(l) = .true.
   286)                ido_nh4a(n) = 1
   287)             end if
   288)          end if
   289)       end if
   290)       if ( do_soag ) then
   291)          l = lptr_soa_a_amode(n)-loffset
   292)          if ((l > 0) .and. (l <= pcnstxx)) then
   293)             dotend(l) = .true.
   294)             ido_soaa(n) = 1
   295)             ntot_soamode = n
   296)          end if
   297)       end if
   298)    end do
   299)    if ( do_soag ) ntot_soamode = max( ntot_soamode, modefrm_pcage )
   300) 
   301)    if (modefrm_pcage > 0) then
   302)       ido_so4a(modefrm_pcage) = 2
   303)       if (ido_nh4a(modetoo_pcage) == 1) ido_nh4a(modefrm_pcage) = 2
   304)       if (ido_soaa(modetoo_pcage) == 1) ido_soaa(modefrm_pcage) = 2
   305)       do iq = 1, nspecfrm_pcage
   306)          lsfrm = lspecfrm_pcage(iq)-loffset
   307)          lstoo = lspectoo_pcage(iq)-loffset
   308)          if ((lsfrm > 0) .and. (lsfrm <= pcnst)) then
   309)             dotend(lsfrm) = .true.
   310)             if ((lstoo > 0) .and. (lstoo <= pcnst)) then
   311)                dotend(lstoo) = .true.
   312)             end if
   313)          end if
   314)       end do
   315) 
   316) 
   317)       fac_m2v_so4 = specmw_so4_amode / specdens_so4_amode
   318)       fac_m2v_nh4 = specmw_nh4_amode / specdens_nh4_amode
   319)       fac_m2v_soa = specmw_soa_amode / specdens_soa_amode
   320)       fac_m2v_pcarbon(:) = 0.0_r8
   321)       n = modeptr_pcarbon
   322)       do l = 1, nspec_amode(n)
   323)          l2 = lspectype_amode(l,n)
   324) !   fac_m2v converts (kmol-AP/kmol-air) to (m3-AP/kmol-air)
   325) !        [m3-AP/kmol-AP]    = [kg-AP/kmol-AP]  / [kg-AP/m3-AP]
   326)          fac_m2v_pcarbon(l) = specmw_amode(l2) / specdens_amode(l2)
   327)       end do
   328)       fac_volsfc_pcarbon = exp( 2.5_r8*(alnsg_amode(n)**2) )
   329)       xferfrac_max = 1.0_r8 - 10.0_r8*epsilon(1.0_r8)   ! 1-eps
   330)    end if
   331) 
   332) 
   333) ! zero out tendencies and other

   335)    dqqcwdt(:,:,:) = 0.0_r8

   342)                            dgncur_awet,            uptkrate    )
   343) 
   344)

   348)

   351)      do i=1,ncol
   352)

   354) !   fgain_nh4(n) = fraction of total  nh3  uptake going to mode n
   355)         sum_uprt_so4 = 0.0_r8
   356)         sum_uprt_nh4 = 0.0_r8
   357)         sum_uprt_soa = 0.0_r8
   358)         do n = 1, ntot_amode
   359)             uptkratebb(n) = uptkrate(n,i,k)
   360)             if (ido_so4a(n) > 0) then
   361)                 fgain_so4(n) = uptkratebb(n)
   362)                 sum_uprt_so4 = sum_uprt_so4 + fgain_so4(n)
   363)                 if (ido_so4a(n) == 1) then
   364)                     qold_so4(n) = q(i,k,lptr_so4_a_amode(n)-loffset)
   365)                 else
   366)                     qold_so4(n) = 0.0_r8
   367)                 end if
   368)             else
   369)                 fgain_so4(n) = 0.0_r8
   370)                 qold_so4(n) = 0.0_r8
   371)             end if
   372) 
   373)             if (ido_nh4a(n) > 0) then
   374) !   2.08 factor is for gas diffusivity (nh3/h2so4)
   375) !   differences in fuch-sutugin and accom coef ignored
   376)                 fgain_nh4(n) = uptkratebb(n)*2.08_r8
   377)                 sum_uprt_nh4 = sum_uprt_nh4 + fgain_nh4(n)
   378)                 if (ido_nh4a(n) == 1) then
   379)                     qold_nh4(n) = q(i,k,lptr_nh4_a_amode(n)-loffset)
   380)                 else
   381)                     qold_nh4(n) = 0.0_r8
   382)                 end if
   383)             else
   384)                 fgain_nh4(n) = 0.0_r8
   385)                 qold_nh4(n) = 0.0_r8
   386)             end if
   387) 
   388)             if (ido_soaa(n) > 0) then
   389) !   0.81 factor is for gas diffusivity (soa/h2so4)
   390) !   (differences in fuch-sutugin and accom coef ignored)
   391)                 fgain_soa(n) = uptkratebb(n)*0.81_r8
   392)                 sum_uprt_soa = sum_uprt_soa + fgain_soa(n)
   393)                 if (ido_soaa(n) == 1) then
   394)                     qold_soa(n) = q(i,k,lptr_soa_a_amode(n)-loffset)
   395)                     l = lptr_pom_a_amode(n)-loffset
   396)                     if (l > 0) then
   397)                        qold_poa(n) = q(i,k,l)
   398)                     else
   399)                        qold_poa(n) = 0.0_r8
   400)                     end if
   401)                 else
   402)                     qold_soa(n) = 0.0_r8
   403)                     qold_poa(n) = 0.0_r8
   404)                 end if
   405)             else
   406)                 fgain_soa(n) = 0.0_r8
   407)                 qold_soa(n) = 0.0_r8
   408)                 qold_poa(n) = 0.0_r8
   409)             end if
   410)             uptkrate_soa(n) = fgain_soa(n)
   411)         end do
   412) 
   413)         if (sum_uprt_so4 > 0.0_r8) then
   414)             do n = 1, ntot_amode
   415)                 fgain_so4(n) = fgain_so4(n) / sum_uprt_so4
   416)             end do
   417)         end if
   418) !       at this point (sum_uprt_so4 <= 0.0) only when all the fgain_so4 are zero
   419)         if (sum_uprt_nh4 > 0.0_r8) then
   420)             do n = 1, ntot_amode
   421)                 fgain_nh4(n) = fgain_nh4(n) / sum_uprt_nh4
   422)             end do
   423)         end if
   424)         if (sum_uprt_soa > 0.0_r8) then
   425)             do n = 1, ntot_amode
   426)                 fgain_soa(n) = fgain_soa(n) / sum_uprt_soa
   427)             end do
   428)         end if
   429) 
   430) !   uptake amount (fraction of gas uptaken) over deltat
   431)         avg_uprt_so4 = (1.0_r8 - exp(-deltatxx*sum_uprt_so4))/deltatxx
   432)         avg_uprt_nh4 = (1.0_r8 - exp(-deltatxx*sum_uprt_nh4))/deltatxx
   433)         avg_uprt_soa = (1.0_r8 - exp(-deltatxx*sum_uprt_soa))/deltatxx
   434)  
   435) !   sum_dqdt_so4 = so4_a tendency from h2so4 gas uptake (mol/mol/s)
   436) !   sum_dqdt_msa = msa_a tendency from msa   gas uptake (mol/mol/s)

   840) subroutine gas_aer_uptkrates( ncol,       loffset,                &
   841)                               q,          t,          pmid,       &
   842)                               dgncur_awet,            uptkrate    )
   843) 
   844) !
   845) !                         /
   846) !   computes   uptkrate = | dx  dN/dx  gas_conden_rate(Dp(x))
   847) !                         /
   848) !   using Gauss-Hermite quadrature of order nghq=2
   849) !
   850) !       Dp = particle diameter (cm)
   851) !       x = ln(Dp)
   852) !       dN/dx = log-normal particle number density distribution
   853) !       gas_conden_rate(Dp) = 2 * pi * gasdiffus * Dp * F(Kn,ac)
   854) !           F(Kn,ac) = Fuchs-Sutugin correction factor
   855) !           Kn = Knudsen number
   856) !           ac = accomodation coefficient
   857) !
   858) 
   859) !use modal_aero_data, only:  ntot_amode, ntot_amode, nspec_amode,   &
   860) !                            lspectype_amode, lmassptr_amode,   &
   861) !                            sigmag_amode,   &
   862) !                            specdens_amode, specmw_amode
   863) use modal_aero_data, only:  ntot_amode, ntot_amode,   &
   864)                             numptr_amode,   &
   865)                             sigmag_amode
   866) 
   867) use ppgrid
   868) use constituents, only: pcnst, cnst_name
   869) use physconst, only: mwdry, rair
   870) 
   871) implicit none
   872) 
   873) 
   874)    integer,  intent(in) :: ncol                 ! number of atmospheric column
   875)    integer,  intent(in) :: loffset
   876)    real(r8), intent(in) :: q(ncol,pver,pcnstxx) ! Tracer array (mol,#/mol-air)
   877)    real(r8), intent(in) :: t(pcols,pver)        ! Temperature in Kelvin
   878)    real(r8), intent(in) :: pmid(pcols,pver)     ! Air pressure in Pa
   879)    real(r8), intent(in) :: dgncur_awet(pcols,pver,ntot_amode)
   880) 
   881)    real(r8), intent(out) :: uptkrate(ntot_amode,pcols,pver)  
   882)                             ! gas-to-aerosol mass transfer rates (1/s)
   883) 
   884) 
   885) ! local 
   886)    integer, parameter :: nghq = 2
   887)    integer :: i, iq, k, l1, l2, la, n
   888) 
   889)    ! Can use sqrt here once Lahey is gone.
   890)    real(r8), parameter :: tworootpi = 3.5449077_r8
   891)    real(r8), parameter :: root2 = 1.4142135_r8
   892)    real(r8), parameter :: beta = 2.0_r8
   893) 
   894)    real(r8) :: aircon
   895)    real(r8) :: const
   896)    real(r8) :: dp, dum_m2v
   897)    real(r8) :: dryvol_a(pcols,pver)
   898)    real(r8) :: gasdiffus, gasspeed
   899)    real(r8) :: freepathx2, fuchs_sutugin
   900)    real(r8) :: knudsen
   901)    real(r8) :: lndp, lndpgn, lnsg
   902)    real(r8) :: num_a
   903)    real(r8) :: rhoair
   904)    real(r8) :: sumghq
   905)    real(r8), save :: xghq(nghq), wghq(nghq) ! quadrature abscissae and weights
   906) 
   907)    data xghq / 0.70710678_r8, -0.70710678_r8 /
   908)    data wghq / 0.88622693_r8,  0.88622693_r8 /
   909) 
   910) 
   911) ! outermost loop over all modes
   912)    do n = 1, ntot_amode
   913) 
   914) ! 22-aug-2007 rc easter - get number from q array rather
   915) !    than computing a "bounded" number conc.
   916) !! compute dry volume = sum_over_components{ component_mass / density }
   917) !!    (m3-AP/mol-air)
   918) !! compute it for all i,k to improve accessing q array
   919) !      dryvol_a(1:ncol,:) = 0.0_r8
   920) !      do l1 = 1, nspec_amode(n)
   921) !         l2 = lspectype_amode(l1,n)
   922) !! dum_m2v converts (kmol-AP/kmol-air) to (m3-AP/kmol-air)
   923) !! [m3-AP/kmol-AP]= [kg-AP/kmol-AP]  / [kg-AP/m3-AP]
   924) !         dum_m2v = specmw_amode(l2) / specdens_amode(l2)
   925) !         la = lmassptr_amode(l1,n)
   926) !         dryvol_a(1:ncol,:) = dryvol_a(1:ncol,:)    &
   927) !                            + max(0.0_r8,q(1:ncol,:,la))*dum_m2v
   928) !      end do
   929) 
   930) ! loops k and i
   931)       do k=top_lev,pver
   932)       do i=1,ncol
   933) 
   934)          rhoair = pmid(i,k)/(rair*t(i,k))   ! (kg-air/m3)
   935) !        aircon = 1.0e3*rhoair/mwdry        ! (mol-air/m3)
   936) 
   937) !!   "bounded" number conc. (#/m3)
   938) !        num_a = dryvol_a(i,k)*v2ncur_a(i,k,n)*aircon
   939) 
   940) !   number conc. (#/m3) -- note q(i,k,numptr) is (#/kmol-air) 
   941) !   so need aircon in (kmol-air/m3)
   942)          aircon = rhoair/mwdry              ! (kmol-air/m3)
   943)          num_a = q(i,k,numptr_amode(n)-loffset)*aircon
   944) 
   945) !   gasdiffus = h2so4 gas diffusivity from mosaic code (m^2/s)
   946) !               (pmid must be Pa)
   947)          gasdiffus = 0.557e-4_r8 * (t(i,k)**1.75_r8) / pmid(i,k)
   948) !   gasspeed = h2so4 gas mean molecular speed from mosaic code (m/s)
   949)          gasspeed  = 1.470e1_r8 * sqrt(t(i,k))
   950) !   freepathx2 = 2 * (h2so4 mean free path)  (m)
   951)          freepathx2 = 6.0_r8*gasdiffus/gasspeed
   952) 
   953)          lnsg   = log( sigmag_amode(n) )
   954)          lndpgn = log( dgncur_awet(i,k,n) )   ! (m)
   955)          const  = tworootpi * num_a * exp(beta*lndpgn + 0.5_r8*(beta*lnsg)**2)
   956)          
   957) !   sum over gauss-hermite quadrature points
   958)          sumghq = 0.0_r8
   959)          do iq = 1, nghq
   960)             lndp = lndpgn + beta*lnsg**2 + root2*lnsg*xghq(iq)
   961)             dp = exp(lndp)
   962) 
   963) !   knudsen number
   964)             knudsen = freepathx2/dp
   965) !   following assumes accomodation coefficient = ac = 0.65
   966) !   (Adams & Seinfeld, 2002, JGR, and references therein)
   967) !           fuchs_sutugin = (0.75*ac*(1. + knudsen)) /
   968) !                           (knudsen*(1.0 + knudsen + 0.283*ac) + 0.75*ac)
   969)             fuchs_sutugin = (0.4875_r8*(1._r8 + knudsen)) /   &
   970)                             (knudsen*(1.184_r8 + knudsen) + 0.4875_r8)
   971) 
   972)             sumghq = sumghq + wghq(iq)*dp*fuchs_sutugin/(dp**beta)
   973)          end do
   974)          uptkrate(n,i,k) = const * gasdiffus * sumghq    
   975) 
   976)       end do   ! "do i = 1, ncol"
   977)       end do   ! "do k = 1, pver"
   978) 
   979)    end do   ! "do n = 1, ntot_soamode"
   980) 
   981) 
   982)    return
   983)    end subroutine gas_aer_uptkrates

   988)       subroutine modal_aero_soaexch( dtfull, temp, pres, &
   989)           niter, niter_max, ntot_soamode, &
   990)           g_soa_in, a_soa_in, a_poa_in, xferrate, &
   991)           g_soa_tend, a_soa_tend )
   992) !         g_soa_tend, a_soa_tend, g0_soa, idiagss )
   993) 
   994)         use mo_constants, only: rgas ! Gas constant (J/K/mol)
   995) !-----------------------------------------------------------------------
   996) !
   997) ! Purpose:
   998) !
   999) ! calculates condensation/evaporation of "soa gas" 
  1000) ! to/from multiple aerosol modes in 1 grid cell
  1001) !
  1002) ! key assumptions
  1003) ! (1) ambient equilibrium vapor pressure of soa gas
  1004) !     is given by p0_soa_298 and delh_vap_soa
  1005) ! (2) equilibrium vapor pressure of soa gas at aerosol
  1006) !     particle surface is given by raoults law in the form
  1007) !     g_star = g0_soa*[a_soa/(a_soa + a_opoa)]
  1008) ! (3) (oxidized poa)/(total poa) is equal to frac_opoa (constant)
  1009) ! 
  1010) !
  1011) ! Author: R. Easter and R. Zaveri
  1012) !
  1013) !-----------------------------------------------------------------------
  1014)       implicit none
  1015) 
  1016)       real(r8), intent(in)  :: dtfull     ! full integration time step (s)
  1017)       real(r8), intent(in)  :: temp       ! air temperature (K)
  1018)       real(r8), intent(in)  :: pres       ! air pressure (Pa)
  1019)       integer,  intent(out) :: niter      ! number of iterations performed
  1020)       integer,  intent(in)  :: niter_max  ! max allowed number of iterations
  1021)       integer,  intent(in)  :: ntot_soamode             ! number of modes having soa
  1022)       real(r8), intent(in)  :: g_soa_in                 ! initial soa gas mixrat (mol/mol)
  1023)       real(r8), intent(in)  :: a_soa_in(ntot_soamode)   ! initial soa aerosol mixrat (mol/mol)
  1024)       real(r8), intent(in)  :: a_poa_in(ntot_soamode)   ! initial poa aerosol mixrat (mol/mol)
  1025)       real(r8), intent(in)  :: xferrate(ntot_soamode)   ! gas-aerosol mass transfer rate (1/s)
  1026)       real(r8), intent(out) :: g_soa_tend               ! soa gas mixrat tendency (mol/mol/s)
  1027)       real(r8), intent(out) :: a_soa_tend(ntot_soamode) ! soa aerosol mixrat tendency (mol/mol/s)
  1028) !     real(r8), intent(out) :: g0_soa   ! ambient soa gas equilib mixrat (mol/mol)
  1029) !     integer,  intent(in)  :: idiagss
  1030) 
  1031)       integer :: luna=6
  1032)       integer :: m
  1033) 
  1034)       real(r8), parameter :: alpha = 0.05_r8  ! parameter used in calc of time step
  1035)       real(r8), parameter :: g_min1 = 1.0e-20_r8
  1036)       real(r8), parameter :: opoa_frac = 0.1_r8  ! fraction of poa that is opoa
  1037)       real(r8), parameter :: delh_vap_soa = 156.0e3_r8
  1038)       ! delh_vap_soa = heat of vaporization for gas soa (J/mol)
  1039)       real(r8), parameter :: p0_soa_298 = 1.0e-10_r8
  1040) 
  1041)       real(r8) :: a_opoa(ntot_soamode)    ! oxidized-poa aerosol mixrat (mol/mol)
  1042)       real(r8) :: a_soa(ntot_soamode)     ! soa aerosol mixrat (mol/mol)
  1043)       real(r8) :: a_soa_tmp               ! temporary soa aerosol mixrat (mol/mol)
  1044)       real(r8) :: beta(ntot_soamode)      ! dtcur*xferrate
  1045)       real(r8) :: dtcur    ! current time step (s)
  1046)       real(r8) :: dtmax    ! = (dtfull-tcur)
  1047)       real(r8) :: g_soa    ! soa gas mixrat (mol/mol)
  1048)       real(r8) :: g0_soa   ! ambient soa gas equilib mixrat (mol/mol)
  1049)       real(r8) :: g_star(ntot_soamode)    ! soa gas mixrat that is in equilib
  1050)                                           ! with each aerosol mode (mol/mol)
  1051)       real(r8) :: phi(ntot_soamode)       ! "relative driving force"
  1052)       real(r8) :: p0_soa   ! soa gas equilib vapor presssure (atm)
  1053)       real(r8) :: sat(ntot_soamode)
  1054)       real(r8) :: tcur     ! current integration time (from 0 s)
  1055)       real(r8) :: tmpa, tmpb
  1056)       real(r8) :: tot_soa  ! g_soa + sum( a_soa(:) )
  1057) 
  1058) 
  1059) ! force things to be non-negative and calc tot_soa
  1060) ! calc a_opoa (always slightly >0)
  1061)       g_soa = max( g_soa_in, 0.0_r8 )
  1062)       tot_soa = g_soa
  1063)       do m = 1, ntot_soamode
  1064)          a_soa(m) = max( a_soa_in(m), 0.0_r8 )
  1065)          tot_soa = tot_soa + a_soa(m)
  1066)          a_opoa(m) = opoa_frac*a_poa_in(m)
  1067)          a_opoa(m) = max( a_opoa(m), 1.0e-20_r8 )  ! force to small non-zero value
  1068)       end do
  1069) 
  1070) ! calc ambient equilibrium soa gas
  1071)       p0_soa = p0_soa_298 * &
  1072)                exp( -(delh_vap_soa/rgas)*((1.0_r8/temp)-(1.0_r8/298.0_r8)) )
  1073)       g0_soa = 1.01325e5_r8*p0_soa/pres
  1074) ! molecular weight adjustment
  1075) ! the soa parameterization assumes that real gsoa and asoa have mw=150
  1076) ! currently in cam3,
  1077) !    mw=12 is used (this has to do with the mozart preprocessor)
  1078) !    soag emission files (molec/cm2/s units) are set to give the desired 
  1079) !       mass emissions (kg/m2/s) and mass mixing ratios (kg/kg)
  1080) !       when mw=12 is applied
  1081) !    as a result, the molar mixing ratios for both gsoa and asoa
  1082) !       are artificially scaled up by (150/12)
  1083) !       and g0_soa must be similarly scaled up
  1084)       g0_soa = g0_soa*(150.0_r8/12.0_r8)
  1085) !     g0_soa = 0.0   ! force irreversible uptake
  1086) 
  1087)       niter = 0
  1088)       tcur = 0.0_r8
  1089)       dtcur = 0.0_r8
  1090)       phi(:) = 0.0_r8
  1091)       g_star(:) = 0.0_r8
  1092) 
  1093) !     if (idiagss > 0) then
  1094) !        write(luna,'(a,1p,10e11.3)') 'p0, g0_soa', p0_soa, g0_soa
  1095) !        write(luna,'(3a)') &
  1096) !           'niter, tcur,   dtcur,    phi(:),                       ', &
  1097) !           'g_star(:),                    ', &
  1098) !           'a_soa(:),                     g_soa'      
  1099) !        write(luna,'(3a)') &
  1100) !           '                         sat(:),                       ', &
  1101) !           'sat(:)*a_soa(:)               ', &
  1102) !           'a_opoa(:)'
  1103) !        write(luna,'(i3,1p,20e10.2)') niter, tcur, dtcur, &
  1104) !           phi(:), g_star(:), a_soa(:), g_soa
  1105) !     end if
  1106) 
  1107) 
  1108) ! integration loop -- does multiple substeps to reach dtfull
  1109) timeloop: do while (tcur < dtfull-1.0e-3_r8 )
  1110) 
  1111)       niter = niter + 1
  1112)       if (niter > niter_max) exit
  1113) 
  1114)       tmpa = 0.0_r8
  1115)       do m = 1, ntot_soamode
  1116)          sat(m) = g0_soa/(a_soa(m) + a_opoa(m))
  1117)          g_star(m) = sat(m)*a_soa(m)
  1118)          phi(m) = (g_soa - g_star(m))/max(g_soa,g_star(m),g_min1)
  1119)          tmpa = tmpa + xferrate(m)*abs(phi(m))
  1120)       end do
  1121) 
  1122)       dtmax = dtfull-tcur
  1123)       if (dtmax*tmpa <= alpha) then
  1124) ! here alpha/tmpa >= dtmax, so this is final substep
  1125)          dtcur = dtmax
  1126)          tcur = dtfull
  1127)       else
  1128)          dtcur = alpha/tmpa
  1129)          tcur = tcur + dtcur
  1130)       end if
  1131) 
  1132) ! step 1 - for modes where soa is condensing, estimate "new" a_soa(m)
  1133) !    using an explicit calculation with "old" g_soa
  1134) !    and g_star(m) calculated using "old" a_soa(m)
  1135) ! do this to get better estimate of "new" a_soa(m) and sat(m)
  1136)       do m = 1, ntot_soamode
  1137)          beta(m) = dtcur*xferrate(m)
  1138)          tmpa = g_soa - g_star(m)
  1139)          if (tmpa > 0.0_r8) then
  1140)             a_soa_tmp = a_soa(m) + beta(m)*tmpa
  1141)             sat(m) = g0_soa/(a_soa_tmp + a_opoa(m))
  1142)             g_star(m) = sat(m)*a_soa_tmp   ! this just needed for diagnostics
  1143)          end if
  1144)       end do
  1145) 
  1146) ! step 2 - implicit in g_soa and semi-implicit in a_soa,
  1147) !    with g_star(m) calculated semi-implicitly
  1148)       tmpa = 0.0_r8
  1149)       tmpb = 0.0_r8
  1150)       do m = 1, ntot_soamode
  1151)          tmpa = tmpa + a_soa(m)/(1.0_r8 + beta(m)*sat(m))
  1152)          tmpb = tmpb + beta(m)/(1.0_r8 + beta(m)*sat(m))
  1153)       end do
  1154)       g_soa = (tot_soa - tmpa)/(1.0_r8 + tmpb)
  1155)       g_soa = max( 0.0_r8, g_soa )
  1156)       do m = 1, ntot_soamode
  1157)          a_soa(m) = (a_soa(m) + beta(m)*g_soa)/   &
  1158)                     (1.0_r8 + beta(m)*sat(m))
  1159)       end do
  1160) 
  1161) !     if (idiagss > 0) then
  1162) !        write(luna,'(i3,1p,20e10.2)') niter, tcur, dtcur, &
  1163) !           phi(:), g_star(:), a_soa(:), g_soa
  1164) !        write(luna,'(23x,1p,20e10.2)') &
  1165) !           sat(:), sat(:)*a_soa(:), a_opoa(:)
  1166) !     end if
  1167) 
  1168) !     if (niter > 9992000) then
  1169) !        write(luna,'(a)') '*** to many iterations'
  1170) !        exit
  1171) !     end if
  1172) 
  1173)       end do timeloop
  1174) 
  1175) 
  1176)       g_soa_tend = (g_soa - g_soa_in)/dtfull
  1177)       do m = 1, ntot_soamode
  1178)          a_soa_tend(m) = (a_soa(m) - a_soa_in(m))/dtfull
  1179)       end do
  1180) 
  1181) 
  1182)       return
  1183)       end subroutine modal_aero_soaexch
