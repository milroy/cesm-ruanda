    90)   subroutine aero_model_readnl(nlfile)
    91) 
    92)     use namelist_utils,  only: find_group_name
    93)     use units,           only: getunit, freeunit
    94)     use mpishorthand
    95) 
    96)     character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
    97) 
    98)     ! Local variables
    99)     integer :: unitn, ierr
   100)     character(len=*), parameter :: subname = 'aero_model_readnl'
   101) 
   102)     ! Namelist variables
   103)     character(len=16) :: aer_wetdep_list(pcnst) = ' '
   104)     character(len=16) :: aer_drydep_list(pcnst) = ' '
   105) 
   106)     namelist /aerosol_nl/ aer_wetdep_list, aer_drydep_list, sol_facti_cloud_borne, &
   107)        sol_factb_interstitial, sol_factic_interstitial
   108) 
   109)     !-----------------------------------------------------------------------------
   110) 
   111)     ! Read namelist
   112)     if (masterproc) then
   113)        unitn = getunit()
   114)        open( unitn, file=trim(nlfile), status='old' )
   115)        call find_group_name(unitn, 'aerosol_nl', status=ierr)
   116)        if (ierr == 0) then
   117)           read(unitn, aerosol_nl, iostat=ierr)
   118)           if (ierr /= 0) then
   119)              call endrun(subname // ':: ERROR reading namelist')
   120)           end if
   121)        end if
   122)        close(unitn)
   123)        call freeunit(unitn)
   124)     end if
   125) 
   126) #ifdef SPMD
   127)     ! Broadcast namelist variables
   128)     call mpibcast(aer_wetdep_list,   len(aer_wetdep_list(1))*pcnst, mpichar, 0, mpicom)
   129)     call mpibcast(aer_drydep_list,   len(aer_drydep_list(1))*pcnst, mpichar, 0, mpicom)
   130)     call mpibcast(sol_facti_cloud_borne, 1,                         mpir8,   0, mpicom)
   131)     call mpibcast(sol_factb_interstitial, 1,                        mpir8,   0, mpicom)
   132)     call mpibcast(sol_factic_interstitial, 1,                       mpir8,   0, mpicom)
   133) #endif
   134) 
   135)     wetdep_list = aer_wetdep_list
   136)     drydep_list = aer_drydep_list
   137) 
   138)   end subroutine aero_model_readnl

   210)        endif

   214)          allocate(wetdep_indices(nwetdep))
   215)     if (ndrydep>0) &
   216)          allocate(drydep_indices(ndrydep))

   221)           drydep_indices(m) = id
   222)        else
   223)           call endrun(subrname//': invalid drydep species: '//trim(drydep_list(m)) )
   224)        endif
   225) 
   226)        if (masterproc) then
   227)           write(iulog,*) subrname//': '//drydep_list(m)//' will have drydep applied'
   228)        endif

   231)        call cnst_get_ind ( wetdep_list(m), id, abort=.false. )
   232)        if (id>0) then
   233)           wetdep_indices(m) = id
   234)        else
   235)           call endrun(subrname//': invalid wetdep species: '//trim(wetdep_list(m)) )

   238)        if (masterproc) then
   239)           write(iulog,*) subrname//': '//wetdep_list(m)//' will have wet removal'

   255)           call add_default (dummy, 1, ' ')
   256)        endif
   257) 
   258)     endif

   267)              call add_default (dummy, 1, ' ')
   268)           endif
   269)        enddo
   270) 
   271)        dummy = 'DSTSFMBL'
   272)        call addfld (dummy,'kg/m2/s',1, 'A','Mobilization flux at surface',phys_decomp)

   274)           call add_default (dummy, 1, ' ')
   275)        endif
   276) 
   277)        dummy = 'LND_MBL'
   278)        call addfld (dummy,'frac ',1, 'A','Soil erodibility factor',phys_decomp)
   279)        if (history_aerosol) then
   280)           call add_default (dummy, 1, ' ')
   281)        endif
   282) 
   283)     endif
   284) 
   285)     if (seasalt_active) then
   286)        
   287)        dummy = 'SSTSFMBL'
   288)        call addfld (dummy,'kg/m2/s',1, 'A','Mobilization flux at surface',phys_decomp)

   290)           call add_default (dummy, 1, ' ')
   291)        endif
   292)

   296)           if (history_aerosol) then
   297)              call add_default (dummy, 1, ' ')
   298)           endif
   299)        enddo
   300) 
   301)     endif
   302) 
   303)

   306)     do m=1,ndrydep

   313)     do m=1,nwetdep
   314)        id = wetdep_indices(m)
   315)        wetdep_lq(id) = .true.
   316)     enddo
   317) 
   318)     wetdens_ap_idx = pbuf_get_index('WETDENS_AP')
   319)     qaerwat_idx    = pbuf_get_index('QAERWAT')
   320)     pblh_idx       = pbuf_get_index('pblh')
   321) 
   322)     rate1_cw2pr_st_idx  = pbuf_get_index('RATE1_CW2PR_ST') 
   323)     call pbuf_set_field(pbuf2d, rate1_cw2pr_st_idx, 0.0_r8)
   324) 
   325)     do m = 1,ndrydep
   326)        
   327)        ! units 
   328)        if (drydep_list(m)(1:3) == 'num') then
   329)           unit_basename = ' 1'
   330)        else
   331)           unit_basename = 'kg'  
   332)        endif
   333) 
   334)        call addfld (trim(drydep_list(m))//'DDF',unit_basename//'/m2/s ',   1, 'A', &
   335)             trim(drydep_list(m))//' dry deposition flux at bottom (grav + turb)',phys_decomp)
   336)        call addfld (trim(drydep_list(m))//'TBF',unit_basename//'/m2/s',   1, 'A', &
   337)             trim(drydep_list(m))//' turbulent dry deposition flux',phys_decomp)
   338)        call addfld (trim(drydep_list(m))//'GVF',unit_basename//'/m2/s ',   1, 'A', &
   339)             trim(drydep_list(m))//' gravitational dry deposition flux',phys_decomp)
   340)        call addfld (trim(drydep_list(m))//'DTQ',unit_basename//'/kg/s ',pver, 'A', &
   341)             trim(drydep_list(m))//' dry deposition',phys_decomp)
   342)        call addfld (trim(drydep_list(m))//'DDV','m/s     ',pver, 'A', &
   343)             trim(drydep_list(m))//' deposition velocity',phys_decomp)
   344) 
   345)        if ( history_aerosol ) then 
   346)           call add_default (trim(drydep_list(m))//'DDF', 1, ' ')
   347)           call add_default (trim(drydep_list(m))//'TBF', 1, ' ')
   348)           call add_default (trim(drydep_list(m))//'GVF', 1, ' ')
   349)        endif
   350) 
   351)     enddo
   352) 
   353)     do m = 1,nwetdep
   354)        
   355)        ! units 
   356)        if (wetdep_list(m)(1:3) == 'num') then
   357)           unit_basename = ' 1'
   358)        else
   359)           unit_basename = 'kg'  
   360)        endif
   361) 
   362)        call addfld (trim(wetdep_list(m))//'SFWET',unit_basename//'/m2/s ', &
   363)             1,  'A','Wet deposition flux at surface',phys_decomp)

   368)        call addfld (trim(wetdep_list(m))//'SFSBC',unit_basename//'/m2/s ', &

   370)        call addfld (trim(wetdep_list(m))//'SFSBS',unit_basename//'/m2/s ', &

   373)        call addfld (trim(wetdep_list(m))//'SIC',unit_basename//'/kg/s ',pver, 'A', &

   379)        call addfld (trim(wetdep_list(m))//'SBS',unit_basename//'/kg/s ',pver, 'A', &
   380)             trim(wetdep_list(m))//' bs wet deposition',phys_decomp)

   383)           call add_default (trim(wetdep_list(m))//'SFWET', 1, ' ')
   384)           call add_default (trim(wetdep_list(m))//'SFSIC', 1, ' ')
   385)           call add_default (trim(wetdep_list(m))//'SFSIS', 1, ' ')
   386)           call add_default (trim(wetdep_list(m))//'SFSBC', 1, ' ')
   387)           call add_default (trim(wetdep_list(m))//'SFSBS', 1, ' ')
   388)        endif
   389)

   391)

   393)

   395)           unit_basename = ' 1'  ! Units 'kg' or '1'

   397)           unit_basename = 'kg'  ! Units 'kg' or '1' 
   398)        end if
   399) 
   400)        call addfld( 'GS_'//trim(solsym(m)), unit_basename//'/m2/s ',1,  'A', &
   401)                     trim(solsym(m))//' gas chemistry/wet removal (for gas species)', phys_decomp)
   402)        call addfld( 'AQ_'//trim(solsym(m)), unit_basename//'/m2/s ',1,  'A', &
   403)                     trim(solsym(m))//' aqueous chemistry (for gas species)', phys_decomp)
   404)        if ( history_aerosol ) then 
   405)           call add_default( 'GS_'//trim(solsym(m)), 1, ' ')
   406)           call add_default( 'AQ_'//trim(solsym(m)), 1, ' ')
   407)        endif
   408)     enddo
   409)     do n = 1,pcnst
   410)        if( .not. (cnst_name_cw(n) == ' ') ) then
   411) 
   412)           if (cnst_name_cw(n)(1:3) == 'num') then

   417)

   419)                trim(cnst_name_cw(n))//' in cloud water',phys_decomp)
   420)           call addfld (trim(cnst_name_cw(n))//'SFWET', unit_basename//'/m2/s ',1,  'A', &

   422)           call addfld (trim(cnst_name_cw(n))//'SFSIC', unit_basename//'/m2/s ',1,  'A', &
   423)                trim(cnst_name_cw(n))//' wet deposition flux (incloud, convective) at surface',phys_decomp)
   424)           call addfld (trim(cnst_name_cw(n))//'SFSIS', unit_basename//'/m2/s ',1,  'A', &

   435)                trim(cnst_name_cw(n))//' gravitational dry deposition flux',phys_decomp)     
   436) 
   437)           if ( history_aerosol ) then 
   438)              call add_default( cnst_name_cw(n), 1, ' ' )
   439)              call add_default (trim(cnst_name_cw(n))//'GVF', 1, ' ')
   440)              call add_default (trim(cnst_name_cw(n))//'SFWET', 1, ' ') 
   441)              call add_default (trim(cnst_name_cw(n))//'TBF', 1, ' ')
   442)              call add_default (trim(cnst_name_cw(n))//'DDF', 1, ' ')
   443)              call add_default (trim(cnst_name_cw(n))//'SFSBS', 1, ' ')      
   444)              call add_default (trim(cnst_name_cw(n))//'SFSIC', 1, ' ')
   445)              call add_default (trim(cnst_name_cw(n))//'SFSBC', 1, ' ')
   446)              call add_default (trim(cnst_name_cw(n))//'SFSIS', 1, ' ')

  1912)           endif
  1913)        enddo  ! n_land_type
  1914)        vlc_trb(i) = wrk2
  1915)        vlc_dry(i,k) = wrk3
  1916)     enddo !ncol
  1917) 
  1918)     return
