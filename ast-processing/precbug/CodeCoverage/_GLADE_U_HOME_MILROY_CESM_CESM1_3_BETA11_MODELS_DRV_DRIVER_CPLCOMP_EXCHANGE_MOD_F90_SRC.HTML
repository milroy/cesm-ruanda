   158)        !       mapid = -1

   160)        !       call shr_mpi_min(mapid,mapidmin,mpicom_join,subname//' min')

   169)        mapper%strategy = "rearrange"

   179)        call mct_rearr_init(gsmap_s_join, gsmap_d_join, mpicom_join, mapper%rearr)

   182) 
   183)        call mct_gsMap_clean(gsmap_s_join)
   184)        call mct_gsMap_clean(gsmap_d_join)
   185) 
   186)     endif
   187) 
   188)     if (seq_comm_iamroot(CPLID)) then
   189)        write(logunit,'(2A,I6,4A)') subname,' mapper counter, strategy, mapfile = ', &
   190)           mapper%counter,' ',trim(mapper%strategy),' ',trim(mapper%mapfile)

   288)     ! on other pes.  It addresses non-overlap of pes.
   289) 
   290)     !-----------------------------------------------------
   291)     ! 
   292)     ! Arguments
   293)     !

   296)     ! Local Variables

   305)     type(mct_gsMap), pointer :: gsmap_old

   308)     character(len=*),parameter :: subname = "(seq_mctext_gsmapInit) "
   309)     !-----------------------------------------------------
   310) 
   311)     call seq_comm_getinfo(CPLID, mpicom=mpicom_CPLID)
   312) 
   313)     id_new  = cplid
   314)     id_old  = comp%compid 
   315)     id_join = comp%cplcompid
   316)

   318)     mpicom_old  = comp%mpicom_compid
   319)     mpicom_join = comp%mpicom_cplcompid
   320) 
   321)     gsmap_new => component_get_gsmap_cx(comp)
   322)     gsmap_old => component_get_gsmap_cc(comp)
   323) 
   324)     call seq_comm_getinfo(ID_old ,mpicom=mpicom_old)
   325)     call seq_comm_getinfo(ID_new ,mpicom=mpicom_new)
   326)     call seq_comm_getinfo(ID_join,mpicom=mpicom_join)
   327)

   330)     ! ---   Create a new gsmap on pes associated with ID_new using info from the old one
   331) 
   332)     call seq_mctext_gsmapExtend(gsmap_old     , mpicom_old  , gsmap_old_join, mpicom_join, ID_join)
   333)     call seq_mctext_gsmapCreate(gsmap_old_join, mpicom_join , gsmap_new     , mpicom_new , ID_new  )
   334) 
   335)     call mct_gsMap_clean(gsmap_old_join)
   336)

   458)   subroutine seq_mctext_gsmapExtend(gsmapi, mpicomi, gsmapo, mpicomo, compido)
   459) 
   460)     !----------------------------------------------------------------
   461)     ! Extend/Convert a gsmap from one mpicom to another mpicom that contains
   462)     ! at least all the pes that gsmap uses, but with different ranks
   463)     !----------------------------------------------------------------
   464) 
   465)     implicit none
   466)     type(mct_gsMap), intent(IN) :: gsmapi
   467)     integer        , intent(IN) :: mpicomi
   468)     type(mct_gsMap), intent(OUT):: gsmapo
   469)     integer        , intent(IN) :: mpicomo
   470)     integer        , intent(IN) :: compido
   471) 
   472)     character(len=*),parameter :: subname = "(seq_mctext_gsmapExtend) "
   473)     integer :: n
   474)     integer :: ngseg
   475)     integer :: gsize
   476)     integer :: msizei,msizeo
   477)     integer :: mrank,mranko,mrankog   ! sets pe rank of root mpicomi pe in mpicomo
   478)     integer :: mpigrpi,mpigrpo
   479)     integer :: ierr
   480)     integer, pointer :: pei(:),peo(:)
   481)     integer, pointer :: start(:),length(:),peloc(:)
   482) 
   483)     mranko = -1
   484) 
   485)     ! --- create the new gsmap on the mpicomi root only
   486) 
   487)     if (mpicomi /= MPI_COMM_NULL) then
   488)        call mpi_comm_rank(mpicomi,mrank,ierr)
   489)        call shr_mpi_chkerr(ierr,subname//' gsm_cop mpi_comm_rank i')
   490)        if (mrank == 0) then
   491)           call mpi_comm_group(mpicomi,mpigrpi,ierr)
   492)           call shr_mpi_chkerr(ierr,subname//' gsm_cop mpi_comm_group i')
   493)           call mpi_comm_group(mpicomo,mpigrpo,ierr)
   494)           call shr_mpi_chkerr(ierr,subname//' gsm_cop mpi_comm_group o')
   495)           call mpi_comm_size(mpicomi,msizei,ierr)
   496)           call shr_mpi_chkerr(ierr,subname//' gsm_cop mpi_comm_size i')
   497)           call mpi_comm_size(mpicomo,msizeo,ierr)
   498)           call shr_mpi_chkerr(ierr,subname//' gsm_cop mpi_comm_size o')
   499) 
   500)           ! --- setup the translation of pe numbers from the old gsmap(mpicom)
   501)           ! --- to the new one, pei -> peo
   502) 
   503)           allocate(pei(0:msizei-1),peo(0:msizei-1))
   504)           do n = 0,msizei-1
   505)              pei(n) = n
   506)           enddo
   507) 
   508)           peo = -1
   509)           call mpi_group_translate_ranks(mpigrpi,msizei,pei,mpigrpo,peo,ierr)

   593)     ! creates a new gsmap on a subset of pes, requires setting a new decomp
   594)     !---------------------------------------------------------------------
   595) 
   596)     implicit none
   597)     type(mct_gsMap), intent(IN) :: gsmapi
   598)     integer        , intent(IN) :: mpicomi
   599)     type(mct_gsMap), intent(OUT):: gsmapo
   600)     integer        , intent(IN) :: mpicomo
   601)     integer        , intent(IN) :: compido
   602) 
   603)     character(len=*),parameter :: subname = "(seq_mctext_gsmapCreate) "
   604)     integer :: n,m,k
   605)     integer :: ktot            ! number of active cells in gsmap
   606)     integer :: apesi, apeso    ! number of active pes in gsmap
   607)     integer ::        lsizeo   ! local size for lindex
   608)     integer :: ngsegi,ngsego   ! ngseg of mpicomi, mpicomo
   609)     integer :: gsizei,gsizeo   ! gsize of mpicomi, mpicomo
   610)     integer :: msizei,msizeo   ! size of mpicomi, mpicomo
   611)     integer :: mranki,mranko   ! rank in mpicomi, mpicomo
   612)     integer :: ierr
   613)     integer :: decomp_type
   614)     integer, pointer :: start(:),length(:),peloc(:),perm(:),gindex(:),lindex(:)
   615)     real(r8):: rpeloc

   617) 
   618)     ! --- create a new gsmap on new pes based on the old gsmap
   619)     ! --- gsmapi must be known on all mpicomo pes, compute the same 
   620)     ! --- thing on all pes in parallel
   621) 
   622)     if (mpicomo /= MPI_COMM_NULL) then
   623)        call mpi_comm_rank(mpicomi,mranki,ierr)
   624)        call shr_mpi_chkerr(ierr,subname//' mpi_comm_rank i')
   625)        call mpi_comm_size(mpicomi,msizei,ierr)
   626)        call shr_mpi_chkerr(ierr,subname//' mpi_comm_size i')
   627)        call mpi_comm_rank(mpicomo,mranko,ierr)
   628)        call shr_mpi_chkerr(ierr,subname//' mpi_comm_rank o')
   629)        call mpi_comm_size(mpicomo,msizeo,ierr)
   630)        call shr_mpi_chkerr(ierr,subname//' mpi_comm_size o')
   631) 
   632)        ngsegi = gsmapi%ngseg
   633)        gsizei = gsmapi%gsize
   634)        gsizeo = gsizei
   635)        call mct_gsMap_activepes(gsmapi,apesi)
   636) 
   637)        decomp_type = 0
   638) 
   639)        if (seq_mctext_decomp == 0) then
   640)           if (msizeo == apesi) then      ! preserve segments and decomp
   641)              ! For testing - set decomp_type to 1 - to have gsmapi and gsmapo identical
   642)              if (gsmap_bfbflag) then
   643)                 decomp_type = 1     ! better in cpl to have all decomps "same-ish"
   644)              else
   645)                 decomp_type = 2
   646)              end if
   647)           elseif (ngsegi >= msizeo) then ! preserve segments, new decomp
   648)              decomp_type = 2
   649)           else                           ! new segments
   650)              decomp_type = 3
   651)           endif
   652)        else
   653)           decomp_type = seq_mctext_decomp
   654)        endif
   655) 
   656)        !tcx       decomp_type = 3 ! over ride setting above for testing
   657)        !       if (mranko == 0) write(logunit,'(2A,4I)') trim(subname),' decomp_type =',decomp_type,ngsegi,msizeo,apesi
   658) 
   659)        select case (decomp_type)
   660) 
   661)        case(1)   ! --- preserve segments and decomp ---------------------
   662) 
   663)           ! -- copy the gsmap and translate the pes
   664)           call mct_gsMap_copy(gsmapi,gsmapo)
   665)           ngsego = ngsegi
   666)           do n = 1,ngsego
   667)              gsmapo%pe_loc(n) = mod(gsmapo%pe_loc(n),msizeo)    ! translate pes 1:1 from old to new
   668)           enddo
   669) 
   670)        case(2)   ! --- preserve segments, new decomp --------------------
   671) 
   672)           ! --- preserve segments, sort the start and length, assign a new pe list
   673)           ngsego = ngsegi
   674)           allocate(start(ngsego),length(ngsego),peloc(ngsego),perm(ngsego))
   675)           do n = 1,ngsego
   676)              start(n)  = gsmapi%start(n)
   677)              length(n) = gsmapi%length(n)
   678)           enddo
   679)           ! --- sort gsmap to minimize permute cost in mct
   680)           call mct_indexset(perm)
   681)           call mct_indexsort(ngsego,perm,start)
   682)           call mct_permute(start,perm,ngsego)
   683)           call mct_permute(length,perm,ngsego)
   684)           ! --- give each pe "equal" number of segments, use reals to avoid integer overflow
   685)           do n = 1,ngsego
   686)              rpeloc = (((msizeo*c1)*((n-1)*c1))/(ngsego*c1))      ! give each pe "equal" number of segments, use reals to avoid integer overflow
   687)              peloc(n) = int(rpeloc)
   688)           enddo
   689)           call mct_gsmap_init(gsmapo,ngsego,start,length,peloc,0,mpicomo,compido,gsizeo)
   690)           deallocate(start,length,peloc,perm)
   691) 
   692)        case(3)   ! --- new segments, new decomp -------------------------
   693) 
   694)           ! --- new segments, compute gindex, then parse the gridcells out evenly
   695) 
   696)           k = 0
   697)           do n = 1,ngsegi
   698)              do m = 1,gsmapi%length(n)
   699)                 k = k + 1
   700)                 if (k > gsizei) then
   701)                    write(logunit,*) trim(subname),' ERROR in gindex ',k,gsizei
   702)                    call shr_sys_abort()
   703)                 endif
   704)              enddo
   705)           enddo
   706)           ktot = k
   707) 
   708)           allocate(gindex(ktot),perm(ktot))  
   709) 
   710)           k = 0
   711)           do n = 1,ngsegi
   712)              do m = 1,gsmapi%length(n)
   713)                 k = k + 1
   714)                 gindex(k) = gsmapi%start(n) + m - 1
   715)              enddo
   716)           enddo
   717)           call mct_indexset(perm)
   718)           call mct_indexsort(ktot,perm,gindex)
   719)           call mct_permute(gindex,perm,ktot)
   720) 
   721)           k = 0
   722)           do m = 0,msizeo-1
   723)              lsizeo = ktot/msizeo
   724)              if (m < (ktot - lsizeo*msizeo)) lsizeo = lsizeo + 1
   725)              if (mranko == m) then
   726)                 allocate(lindex(lsizeo))
   727)                 if (k+lsizeo > ktot) then
   728)                    write(logunit,*) trim(subname),' ERROR: decomp out of bounds ',mranko,k,lsizeo,ktot
   729)                    call shr_sys_abort()
   730)                 endif
   731)                 lindex(1:lsizeo) = gindex(k+1:k+lsizeo)
   732)                 !                write(logunit,*) trim(subname),' decomp is ',mranko,lsizeo,k+1,k+lsizeo
   733)              endif
   734)              k = k + lsizeo
   735)           enddo
   736)           if (k /= ktot) then
   737)              write(logunit,*) trim(subname),' ERROR: decomp incomplete ',k,ktot
   738)              call shr_sys_abort()
   739)           endif
   740) 
   741)           call mct_gsmap_init(gsmapo,lindex,mpicomo,compido,size(lindex),gsizeo)
   742)           deallocate(gindex,perm,lindex)
   743) 
   744)        case default   ! --- unknown ---
   745)           write(logunit,*) trim(subname),' ERROR decomp_type unknown ',decomp_type
   746)           call shr_sys_abort(trim(subname)//' ERROR decomp_type unknown')
   747) 
   748)        end select
   749) 
   750)        if (mranko == 0) then
   751)           write(logunit,102) trim(subname),' created new gsmap decomp_type =',decomp_type
   752)           write(logunit,102) trim(subname),'   ngseg/gsize        = ', &
   753)                mct_gsmap_ngseg(gsmapo),mct_gsmap_gsize(gsmapo)
   754)           call mct_gsmap_activepes(gsmapo,apeso)
   755)           write(logunit,102) trim(subname),'   mpisize/active_pes = ', &
   756)                msizeo,apeso
   757)           write(logunit,102) trim(subname),'   avg seg per pe/ape = ', &
   758)                mct_gsmap_ngseg(gsmapo)/msizeo,mct_gsmap_ngseg(gsmapo)/apeso
   759)           write(logunit,102) trim(subname),'   nlseg/maxnlsegs    = ', &
   760)                mct_gsmap_nlseg(gsmapo,0),mct_gsmap_maxnlseg(gsmapo)
   761) 102       format(2A,2I8)
   762)        endif
   763) 
   764)        !       if (.not. mct_gsmap_increasing(gsmapo) ) then
   765)        !          write(logunit,*) trim(subname),' ERROR: gsmapo not increasing'
   766)        !          call shr_sys_abort()
   767)        !       endif
   768) 
   769)     endif
