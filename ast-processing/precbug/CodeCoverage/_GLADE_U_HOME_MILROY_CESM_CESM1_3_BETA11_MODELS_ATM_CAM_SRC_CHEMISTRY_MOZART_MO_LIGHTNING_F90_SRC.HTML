    90)     vdist(:,3) = (/  2.0_r8, 2.0_r8, 2.0_r8, 1.5_r8, 1.5_r8, 1.5_r8, 3.0_r8, 5.8_r8, &       ! trop cont

   123)   subroutine lightning_no_prod( state, pbuf2d,  cam_in )

   125)     !	... set no production from lightning
   126)     !----------------------------------------------------------------------
   127)     use physics_types,    only : physics_state
   128)     
   129)     use physics_buffer,   only : pbuf_get_index, physics_buffer_desc, pbuf_get_field, pbuf_get_chunk
   130)     use physconst,        only : rga
   131)     use phys_grid,        only : get_rlat_all_p, get_lat_all_p, get_lon_all_p, get_wght_all_p
   132)     use cam_history,      only : outfld
   133)     use camsrfexch,       only : cam_in_t
   134)     use shr_reprosum_mod, only : shr_reprosum_calc
   135)     use mo_constants,  only : rearth, d2r
   136)     implicit none
   137) 
   138)     !----------------------------------------------------------------------
   139)     !	... dummy args
   140)     !----------------------------------------------------------------------
   141)     type(physics_state), intent(in) :: state(begchunk:endchunk) ! physics state
   142)     
   143)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   144)     type(cam_in_t), intent(in) :: cam_in(begchunk:endchunk) ! physics state
   145) 
   146)     !----------------------------------------------------------------------
   147)     !	... local variables
   148)     !----------------------------------------------------------------------
   149)     real(r8), parameter    :: land   = 1._r8
   150)     real(r8), parameter    :: secpyr = 365._r8 * 8.64e4_r8
   151) 
   152)     integer :: i, c
   153)     integer :: cldtind             ! level index for cloud top
   154)     integer :: cldbind             ! level index for cloud base > 273k
   155)     integer :: surf_type
   156)     integer :: file                ! file index
   157)     integer :: k, kk, zlow_ind, zhigh_ind, itype
   158)     real(r8) :: glob_flashfreq     ! global flash frequency [s-1]
   159)     real(r8) :: glob_noprod        ! global rate of no production [as tgn/yr]
   160)     real(r8) :: frac_sum           ! work variable
   161)     real(r8) :: zlow
   162)     real(r8) :: zhigh
   163)     real(r8) :: zlow_scal
   164)     real(r8) :: zhigh_scal
   165)     real(r8) :: fraction
   166)     real(r8) :: dchgz
   167)     real(r8) :: dchgzone(pcols,begchunk:endchunk)           ! depth of discharge zone [km]
   168)     real(r8) :: cldhgt(pcols,begchunk:endchunk)             ! cloud top height [km]
   169)     real(r8) :: cgic(pcols,begchunk:endchunk)               ! cloud-ground/intracloud discharge ratio
   170)     real(r8) :: flash_energy(pcols,begchunk:endchunk)       ! energy of flashes per second
   171)     real(r8) :: prod_no_col(pcols,begchunk:endchunk)        ! global no production rate for diagnostics
   172)     real(r8) :: wrk, wrk1, wrk2(1)
   173)     integer  :: ncol                                    ! columns per chunk
   174)     integer  :: lchnk                                   ! columns per chunk
   175)     real(r8),pointer :: cldtop(:)                       ! cloud top level index
   176)     real(r8),pointer :: cldbot(:)                       ! cloud bottom level index
   177)     real(r8) :: zmid(pcols,pver)                        ! geopot height above surface at midpoints (m)
   178)     real(r8) :: zint(pcols,pver+1,begchunk:endchunk)    ! geopot height above surface at interfaces (m)
   179)     real(r8) :: zsurf(pcols)                            ! geopot height above surface at interfaces (m)
   180)     real(r8) :: rlats(pcols,begchunk:endchunk)          ! column latitudes in chunks
   181)     real(r8) :: wght(pcols)
   182) 
   183)     !----------------------------------------------------------------------
   184)     ! 	... parameters to determine cg/ic ratio [price and rind, 1993]
   185)     !----------------------------------------------------------------------
   186)     real(r8), parameter  :: ca = .021_r8
   187)     real(r8), parameter  :: cb = -.648_r8
   188)     real(r8), parameter  :: cc = 7.49_r8
   189)     real(r8), parameter  :: cd = -36.54_r8
   190)     real(r8), parameter  :: ce = 64.09_r8
   191)     real(r8), parameter  :: t0 = 273._r8
   192)     real(r8), parameter  :: m2km  = 1.e-3_r8
   193)     real(r8), parameter  :: km2cm = 1.e5_r8
   194)     real(r8), parameter  :: lat25 = 25._r8*d2r      ! 25 degrees latitude in radians
   195)     integer  :: cldtop_ndx, cldbot_ndx
   196)     integer  :: istat
   197)     real(r8) :: flash_freq_land, flash_freq_ocn
   198) 
   199)     if (.not.has_no_lightning_prod) return
   200) 
   201)     !----------------------------------------------------------------------
   202)     !	... initialization
   203)     !----------------------------------------------------------------------
   204) 
   205)     flash_freq(:,:)       = 0._r8
   206)     prod_no(:,:,:)        = 0._r8
   207)     prod_no_col(:,:)      = 0._r8
   208)     cldhgt(:,:)           = 0._r8
   209)     dchgzone(:,:)         = 0._r8
   210)     cgic(:,:)             = 0._r8
   211)     flash_energy(:,:)     = 0._r8
   212)     glob_prod_no_col(:,:) = 0._r8
   213) 
   214)     cldtop_ndx = pbuf_get_index('CLDTOP')
   215)     cldbot_ndx = pbuf_get_index('CLDBOT')
   216) 
   217)     !--------------------------------------------------------------------------------
   218)     !	... estimate flash frequency and resulting no emissions
   219)     !           [price, penner, prather, 1997 (jgr)]
   220)     !    lightning only occurs in convective clouds with a discharge zone, i.e.
   221)     !    an altitude range where liquid water, ice crystals, and graupel coexist.
   222)     !    we test this by examining the temperature at the cloud base.
   223)     !    it is assumed that only one thunderstorm occurs per grid box, and its
   224)     !    flash frequency is determined by the maximum cloud top height (not the
   225)     !    depth of the discharge zone). this is somewhat speculative but yields
   226)     !    reasonable results.
   227)     !
   228)     !       the cg/ic ratio is determined by an empirical formula from price and
   229)     !    rind [1993]. the average energy of a cg flash is estimated as 6.7e9 j,
   230)     !    and the average energy of a ic flash is assumed to be 1/10 of that value.
   231)     !       the no production rate is assumed proportional to the discharge energy
   232)     !    with 1e17 n atoms per j. the total number of n atoms is then distributed
   233)     !    over the complete column of grid boxes.
   234)     !--------------------------------------------------------------------------------
   235)     Chunk_loop : do c = begchunk,endchunk
   236)        ncol  = state(c)%ncol
   237)        lchnk = state(c)%lchnk
   238)        call pbuf_get_field(pbuf_get_chunk(pbuf2d,lchnk), cldtop_ndx, cldtop )
   239)        call pbuf_get_field(pbuf_get_chunk(pbuf2d,lchnk), cldbot_ndx, cldbot )
   240)        zsurf(:ncol) = state(c)%phis(:ncol)*rga
   241)        call get_rlat_all_p( c, ncol, rlats(1,c) )
   242)        call get_wght_all_p(c, ncol, wght)
   243) 
   244)        do k = 1,pver
   245)           zmid(:ncol,k)   = state(c)%zm(:ncol,k) + zsurf(:ncol)
   246)           zint(:ncol,k,c) = state(c)%zi(:ncol,k) + zsurf(:ncol)
   247)        end do
   248)        zint(:ncol,pver+1,c) = state(c)%zi(:ncol,pver+1) + zsurf(:ncol)
   249) 
   250)        col_loop : do i = 1,ncol
   251)           !--------------------------------------------------------------------------------
   252)           ! 	... find cloud top and bottom level above 273k
   253)           !--------------------------------------------------------------------------------
   254)           cldtind = nint( cldtop(i) )
   255)           cldbind = nint( cldbot(i) )
   256)           do
   257)              if( cldbind <= cldtind .or. state(c)%t(i,cldbind) < t0 ) then
   258)                 exit
   259)              end if
   260)              cldbind = cldbind - 1
   261)           end do
   262)           cloud_layer : if( cldtind < pver .and. cldtind > 0 .and. cldtind < cldbind ) then
   263)              !--------------------------------------------------------------------------------
   264)              !       ... compute cloud top height and depth of charging zone
   265)              !--------------------------------------------------------------------------------
   266)              cldhgt(i,c)   = m2km*max( 0._r8,zint(i,cldtind,c) )
   267)              dchgz = cldhgt(i,c) - m2km*zmid(i,cldbind)
   268)              dchgzone(i,c) = dchgz
   269)              !--------------------------------------------------------------------------------
   270)              !       ... compute flash frequency for given cloud top height
   271)              !           (flashes storm^-1 min^-1)
   272)              !--------------------------------------------------------------------------------
   273)              flash_freq_land = 3.44e-5_r8 * cldhgt(i,c)**4.9_r8 
   274)              flash_freq_ocn  = 6.40e-4_r8 * cldhgt(i,c)**1.7_r8
   275)              flash_freq(i,c) = cam_in(c)%landfrac(i)*flash_freq_land + &
   276)                                cam_in(c)%ocnfrac(i) *flash_freq_ocn
   277) 
   278)              !--------------------------------------------------------------------------------
   279)              !       ... compute cg/ic ratio
   280)              !           cgic = proportion of cg flashes (=pg from ppp paper)
   281)              !--------------------------------------------------------------------------------
   282)              cgic(i,c) = 1._r8/((((ca*dchgz + cb)*dchgz + cc) *dchgz + cd)*dchgz + ce)
   283)              if( dchgz < 5.5_r8 ) then
   284)                 cgic(i,c) = 0._r8
   285)              else if( dchgz > 14._r8 ) then
   286)                 cgic(i,c) = .02_r8
   287)              end if
   288)              !--------------------------------------------------------------------------------
   289)              !       ... compute flash energy (cg*6.7e9 + ic*6.7e8)
   290)              !           and convert to total energy per second
   291)              !           set ic = cg
   292)              !--------------------------------------------------------------------------------
   293)              flash_energy(i,c) = 6.7e9_r8 * flash_freq(i,c)/60._r8
   294)              !--------------------------------------------------------------------------------
   295)              !       ... LKE Aug 23, 2005: scale production to account for different grid
   296)              !           box sizes. This requires a reduction in the overall fudge factor 
   297)              !           (e.g., from 1.2 to 0.5) 
   298)              !--------------------------------------------------------------------------------
   299)              flash_energy(i,c) =  flash_energy(i,c) * wght(i) * geo_factor
   300)              !--------------------------------------------------------------------------------
   301)              ! 	... compute number of n atoms produced per second
   302)              !           and convert to n atoms per second per cm2 and apply fudge factor
   303)              !--------------------------------------------------------------------------------
   304)              prod_no_col(i,c) = 1.e17_r8*flash_energy(i,c)/(1.e4_r8*rearth*rearth*wght(i)) * factor
   305) 
   306)              !--------------------------------------------------------------------------------
   307)              ! 	... compute global no production rate in tgn/yr:
   308)              !           tgn per second: * 14.00674 * 1.65979e-24 * 1.e-12
   309)              !             nb: 1.65979e-24 = 1/avo
   310)              !           tgn per year: * secpyr
   311)              !--------------------------------------------------------------------------------
   312)              glob_prod_no_col(i,c) = 1.e17_r8*flash_energy(i,c) &
   313)                   * 14.00674_r8 * 1.65979e-24_r8 * 1.e-12_r8 * secpyr * factor
   314) 
   315)           end if cloud_layer
   316)        end do Col_loop
   317)     end do Chunk_loop
   318)     !--------------------------------------------------------------------------------
   319)     ! 	... Accumulate global total, convert to flashes per second
   320)     ! 	... Accumulate global NO production rate
   321)     !--------------------------------------------------------------------------------
   322)     kk = pcols*(endchunk-begchunk+1)
   323)     call shr_reprosum_calc( flash_freq, wrk2,kk,kk,1, commid=mpicom)
   324)     glob_flashfreq=wrk2(1)/60._r8
   325)     call shr_reprosum_calc( glob_prod_no_col, wrk2,kk,kk,1, commid=mpicom)
   326)     glob_noprod = wrk2(1)
   327)     if( masterproc ) then
   328)        write(iulog,*) ' '
   329)        write(iulog,'(''Global flash freq (/s), lightning NOx (TgN/y) = '',2f10.4)') &
   330)             glob_flashfreq, glob_noprod
   331)     end if
   332) 
   333)     if( glob_noprod > 0._r8 ) then
   334)        !--------------------------------------------------------------------------------
   335)        !	... Distribute production up to cloud top [Pickering et al., 1998 (JGR)]
   336)        !--------------------------------------------------------------------------------
   337)        do c = begchunk,endchunk
   338)           ncol  = state(c)%ncol
   339)           lchnk = state(c)%lchnk
   340)           call pbuf_get_field(pbuf_get_chunk(pbuf2d,lchnk), cldtop_ndx, cldtop )
   341)           do i = 1,ncol
   342)              cldtind = nint( cldtop(i) )
   343)              if( prod_no_col(i,c) > 0._r8 ) then
   344)                 if( cldhgt(i,c) > 0._r8 ) then
   345)                    if( abs( rlats(i,c) ) > lat25 ) then
   346)                       itype = 1                                                    ! midlatitude continental
   347)                    else if( nint( cam_in(c)%landfrac(i) ) == land ) then
   348)                       itype = 3                                                    ! tropical continental
   349)                    else
   350)                       itype = 2                                                    ! topical marine
   351)                    end if
   352)                    frac_sum = 0._r8
   353)                    do k = cldtind,pver
   354)                       zlow       = zint(i,k+1,c) * m2km                            ! lower interface height (km)
   355)                       zlow_scal  = zlow * 16._r8/cldhgt(i,c)                       ! scale to 16 km convection height
   356)                       zlow_ind   = max( 1,INT(zlow_scal)+1 )                       ! lowest vdist index to include in layer
   357)                       zhigh      = zint(i,k,c) * m2km                              ! upper interface height (km)
   358)                       zhigh_scal = zhigh * 16._r8/cldhgt(i,c)                      ! height (km) scaled to 16km convection height
   359)                       zhigh_ind  = max( 1,MIN( 16,INT(zhigh_scal)+1 ) )            ! highest vdist index to include in layer
   360)                       do kk = zlow_ind,zhigh_ind
   361)                          wrk  = kk
   362)                          wrk1 = kk - 1
   363)                          fraction = min( zhigh_scal,wrk ) &                         ! fraction of vdist in this model layer
   364)                               - max( zlow_scal,wrk1 )
   365)                          fraction = max( 0._r8, min( 1._r8,fraction ) )
   366)                          frac_sum = frac_sum + fraction*vdist(kk,itype)
   367)                          prod_no(i,k,c) = prod_no(i,k,c) &                         ! sum the fraction of column NOx in layer k
   368)                               + fraction*vdist(kk,itype)*.01_r8
   369)                       end do
   370)                       prod_no(i,k,c) = prod_no_col(i,c) * prod_no(i,k,c) &         ! multiply fraction by column amount
   371)                            / (km2cm*(zhigh - zlow))                    ! and convert to atom N cm^-3 s^-1
   372)                    end do
   373)                 end if
   374)              end if
   375)           end do
   376)        end do
   377)     end if
   378) 
   379)     !--------------------------------------------------------------------------------
   380)     !       ... output lightning no production to history file
   381)     !--------------------------------------------------------------------------------
   382)     do c = begchunk,endchunk
   383)        lchnk = state(c)%lchnk
   384)        call outfld( 'LNO_PROD',     prod_no(:,:,c),        pcols, lchnk )
   385)        call outfld( 'LNO_COL_PROD', glob_prod_no_col(:,c), pcols, lchnk )
   386)        call outfld( 'FLASHFRQ',     flash_freq(:,c),       pcols, lchnk )
   387)        call outfld( 'FLASHENGY',    flash_energy(:,c),     pcols, lchnk )
   388)        call outfld( 'CLDHGT',       cldhgt(:,c),           pcols, lchnk )
   389)        call outfld( 'DCHGZONE',     dchgzone(:,c),         pcols, lchnk )
   390)        call outfld( 'CGIC',         cgic(:,c),             pcols, lchnk )
   391)     enddo
   392) 
   393)   end subroutine lightning_no_prod
