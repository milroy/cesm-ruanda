   377)        ! zero aerosol input arrays
   378)        mss_cnc_aer_in_frc_pur(c,:,:) = 0._r8
   379)        mss_cnc_aer_in_frc_bc(c,:,:)  = 0._r8
   380)        mss_cnc_aer_in_frc_oc(c,:,:)  = 0._r8
   381)        mss_cnc_aer_in_frc_dst(c,:,:) = 0._r8
   382)        mss_cnc_aer_in_fdb(c,:,:)     = 0._r8
   383)     end do
   384) 
   385)     ! Set aerosol input arrays
   386)     ! feedback input arrays have been zeroed
   387)     ! set soot and dust aerosol concentrations:
   388)     if (DO_SNO_AER) then
   389)        mss_cnc_aer_in_fdb(lbc:ubc,:,1) = mss_cnc_bcphi(lbc:ubc,:)
   390)        mss_cnc_aer_in_fdb(lbc:ubc,:,2) = mss_cnc_bcpho(lbc:ubc,:)
   391)        
   392)        ! DO_SNO_OC is set in SNICAR_varpar. Default case is to ignore OC concentrations because:
   393)        !  1) Knowledge of their optical properties is primitive
   394)        !  2) When 'water-soluble' OPAC optical properties are applied to OC in snow, 
   395)        !     it has a negligible darkening effect.
   396)        if (DO_SNO_OC) then
   397)           mss_cnc_aer_in_fdb(lbc:ubc,:,3) = mss_cnc_ocphi(lbc:ubc,:)
   398)           mss_cnc_aer_in_fdb(lbc:ubc,:,4) = mss_cnc_ocpho(lbc:ubc,:)
   399)        endif
   400)        
   401)        mss_cnc_aer_in_fdb(lbc:ubc,:,5) = mss_cnc_dst1(lbc:ubc,:)
   402)        mss_cnc_aer_in_fdb(lbc:ubc,:,6) = mss_cnc_dst2(lbc:ubc,:)
   403)        mss_cnc_aer_in_fdb(lbc:ubc,:,7) = mss_cnc_dst3(lbc:ubc,:)
   404)        mss_cnc_aer_in_fdb(lbc:ubc,:,8) = mss_cnc_dst4(lbc:ubc,:)
   405)     endif
   406) 
   407) 
   408) ! If radiative forcing is being calculated, first estimate clean-snow albedo
   409) 
   410)     if (use_snicar_frc) then
   411) 
   412)        ! 1. BC input array:
   413)        !  set dust and (optionally) OC concentrations, so BC_FRC=[(BC+OC+dust)-(OC+dust)]
   414)        mss_cnc_aer_in_frc_bc(lbc:ubc,:,5) = mss_cnc_dst1(lbc:ubc,:)
   415)        mss_cnc_aer_in_frc_bc(lbc:ubc,:,6) = mss_cnc_dst2(lbc:ubc,:)
   416)        mss_cnc_aer_in_frc_bc(lbc:ubc,:,7) = mss_cnc_dst3(lbc:ubc,:)
   417)        mss_cnc_aer_in_frc_bc(lbc:ubc,:,8) = mss_cnc_dst4(lbc:ubc,:)
   418)        if (DO_SNO_OC) then
   419)           mss_cnc_aer_in_frc_bc(lbc:ubc,:,3) = mss_cnc_ocphi(lbc:ubc,:)
   420)           mss_cnc_aer_in_frc_bc(lbc:ubc,:,4) = mss_cnc_ocpho(lbc:ubc,:)
   421)        endif
   422)        
   423)        ! BC FORCING CALCULATIONS
   424)        flg_slr = 1; ! direct-beam
   425)        call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   426)             coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   427)             mss_cnc_aer_in_frc_bc, albsfc, albsnd_bc, foo_snw)
   428)        
   429)        flg_slr = 2; ! diffuse
   430)        call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   431)             coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   432)             mss_cnc_aer_in_frc_bc, albsfc, albsni_bc, foo_snw)
   433)        
   434)        
   435)        ! 2. OC input array:
   436)        !  set BC and dust concentrations, so OC_FRC=[(BC+OC+dust)-(BC+dust)]
   437)        if (DO_SNO_OC) then
   438)           mss_cnc_aer_in_frc_oc(lbc:ubc,:,1) = mss_cnc_bcphi(lbc:ubc,:)
   439)           mss_cnc_aer_in_frc_oc(lbc:ubc,:,2) = mss_cnc_bcpho(lbc:ubc,:)
   440)           mss_cnc_aer_in_frc_oc(lbc:ubc,:,5) = mss_cnc_dst1(lbc:ubc,:)
   441)           mss_cnc_aer_in_frc_oc(lbc:ubc,:,6) = mss_cnc_dst2(lbc:ubc,:)
   442)           mss_cnc_aer_in_frc_oc(lbc:ubc,:,7) = mss_cnc_dst3(lbc:ubc,:)
   443)           mss_cnc_aer_in_frc_oc(lbc:ubc,:,8) = mss_cnc_dst4(lbc:ubc,:)
   444)           
   445)           ! OC FORCING CALCULATIONS
   446)           flg_slr = 1; ! direct-beam
   447)           call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   448)                coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   449)                mss_cnc_aer_in_frc_oc, albsfc, albsnd_oc, foo_snw)
   450)           
   451)           flg_slr = 2; ! diffuse
   452)           call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   453)                coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   454)                mss_cnc_aer_in_frc_oc, albsfc, albsni_oc, foo_snw)
   455)        endif
   456)     
   457)        ! 3. DUST input array:
   458)        ! set BC and OC concentrations, so DST_FRC=[(BC+OC+dust)-(BC+OC)]
   459)        mss_cnc_aer_in_frc_dst(lbc:ubc,:,1) = mss_cnc_bcphi(lbc:ubc,:)
   460)        mss_cnc_aer_in_frc_dst(lbc:ubc,:,2) = mss_cnc_bcpho(lbc:ubc,:)
   461)        if (DO_SNO_OC) then
   462)           mss_cnc_aer_in_frc_dst(lbc:ubc,:,3) = mss_cnc_ocphi(lbc:ubc,:)
   463)           mss_cnc_aer_in_frc_dst(lbc:ubc,:,4) = mss_cnc_ocpho(lbc:ubc,:)
   464)        endif
   465)        
   466)        ! DUST FORCING CALCULATIONS
   467)        flg_slr = 1; ! direct-beam
   468)        call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   469)             coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   470)             mss_cnc_aer_in_frc_dst, albsfc, albsnd_dst, foo_snw)
   471)        
   472)        flg_slr = 2; ! diffuse
   473)        call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   474)             coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   475)             mss_cnc_aer_in_frc_dst, albsfc, albsni_dst, foo_snw)
   476)        
   477)        
   478)        ! 4. ALL AEROSOL FORCING CALCULATION
   479)        ! (pure snow albedo)
   480)        flg_slr = 1; ! direct-beam
   481)        call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   482)             coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   483)             mss_cnc_aer_in_frc_pur, albsfc, albsnd_pur, foo_snw)
   484)        
   485)        flg_slr = 2; ! diffuse
   486)        call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   487)             coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   488)             mss_cnc_aer_in_frc_pur, albsfc, albsni_pur, foo_snw)
   489) 
   490)     end if
   491) 
   492)     ! CLIMATE FEEDBACK CALCULATIONS, ALL AEROSOLS:
   493)     flg_slr = 1; ! direct-beam
   494)     call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   495)                    coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   496)                    mss_cnc_aer_in_fdb, albsfc, albsnd, flx_absd_snw)
   497) 
   498)     flg_slr = 2; ! diffuse
   499)     call SNICAR_RT(flg_snw_ice, lbc, ubc, num_nourbanc, filter_nourbanc,    &
   500)                    coszen_col, flg_slr, h2osno_liq, h2osno_ice, snw_rds_in, &
   501)                    mss_cnc_aer_in_fdb, albsfc, albsni, flx_absi_snw)
   502) 
   503)     ! ground albedos and snow-fraction weighting of snow absorption factors
   504)     do ib = 1, nband
   505)        do fc = 1,num_nourbanc
   506)           c = filter_nourbanc(fc)
   507)           if (coszen(c) > 0._r8) then
   508)              ! ground albedo was originally computed in SoilAlbedo, but is now computed here
   509)              ! because the order of SoilAlbedo and SNICAR_RT was switched for SNICAR.
   510)              albgrd(c,ib) = albsod(c,ib)*(1._r8-frac_sno(c)) + albsnd(c,ib)*frac_sno(c)
   511)              albgri(c,ib) = albsoi(c,ib)*(1._r8-frac_sno(c)) + albsni(c,ib)*frac_sno(c)
   512) 
   513)              ! albedos for radiative forcing calculations:
   514)              if (use_snicar_frc) then
   515) 
   516)                 ! BC forcing albedo
   517)                 albgrd_bc(c,ib) = albsod(c,ib)*(1.-frac_sno(c)) + albsnd_bc(c,ib)*frac_sno(c)
   518)                 albgri_bc(c,ib) = albsoi(c,ib)*(1.-frac_sno(c)) + albsni_bc(c,ib)*frac_sno(c)
   519)                 
   520)                 if (DO_SNO_OC) then
   521)                    ! OC forcing albedo
   522)                    albgrd_oc(c,ib) = albsod(c,ib)*(1.-frac_sno(c)) + albsnd_oc(c,ib)*frac_sno(c)
   523)                    albgri_oc(c,ib) = albsoi(c,ib)*(1.-frac_sno(c)) + albsni_oc(c,ib)*frac_sno(c)
   524)                 endif
   525)                 
   526)                 ! dust forcing albedo
   527)                 albgrd_dst(c,ib) = albsod(c,ib)*(1.-frac_sno(c)) + albsnd_dst(c,ib)*frac_sno(c)
   528)                 albgri_dst(c,ib) = albsoi(c,ib)*(1.-frac_sno(c)) + albsni_dst(c,ib)*frac_sno(c)
   529)                 
   530)                 ! pure snow albedo for all-aerosol radiative forcing
   531)                 albgrd_pur(c,ib) = albsod(c,ib)*(1.-frac_sno(c)) + albsnd_pur(c,ib)*frac_sno(c)
   532)                 albgri_pur(c,ib) = albsoi(c,ib)*(1.-frac_sno(c)) + albsni_pur(c,ib)*frac_sno(c)
   533) 
   534)              end if
   535) 
   536)              ! also in this loop (but optionally in a different loop for vectorized code)
   537)              !  weight snow layer radiative absorption factors based on snow fraction and soil albedo
   538)              !  (NEEDED FOR ENERGY CONSERVATION)
   539)              do i = -nlevsno+1,1,1
   540)                 if (ib == 1) then
   541)                    flx_absdv(c,i) = flx_absd_snw(c,i,ib)*frac_sno(c) + &
   542)                         ((1.-frac_sno(c))*(1-albsod(c,ib))*(flx_absd_snw(c,i,ib)/(1.-albsnd(c,ib))))
   543)                    flx_absiv(c,i) = flx_absi_snw(c,i,ib)*frac_sno(c) + &
   544)                         ((1.-frac_sno(c))*(1-albsoi(c,ib))*(flx_absi_snw(c,i,ib)/(1.-albsni(c,ib))))
   545)                 elseif (ib == 2) then
   546)                    flx_absdn(c,i) = flx_absd_snw(c,i,ib)*frac_sno(c) + &
   547)                         ((1.-frac_sno(c))*(1-albsod(c,ib))*(flx_absd_snw(c,i,ib)/(1.-albsnd(c,ib))))
   548)                    flx_absin(c,i) = flx_absi_snw(c,i,ib)*frac_sno(c) + &
   549)                         ((1.-frac_sno(c))*(1-albsoi(c,ib))*(flx_absi_snw(c,i,ib)/(1.-albsni(c,ib))))
   550)                 endif
   551)              enddo
   552)           endif
   553)        enddo
   554)     enddo
   555) 
   556)     ! for diagnostics, set snow albedo to spval over non-snow points 
   557)     ! so that it is not averaged in history buffer
   558)     ! (OPTIONAL)
   559)     do ib = 1, nband
   560)        do fc = 1,num_nourbanc
   561)           c = filter_nourbanc(fc)
   562)           if ((coszen(c) > 0._r8) .and. (h2osno(c) > 0._r8)) then
   563)              albsnd_hst(c,ib) = albsnd(c,ib)
   564)              albsni_hst(c,ib) = albsni(c,ib)
   565)           else
   566)              albsnd_hst(c,ib) = 0._r8
   567)              albsni_hst(c,ib) = 0._r8
   568)           endif
   569)        enddo
   570)     enddo
   571) 
   572)     ! Create solar-vegetated filter for the following calculations
   573) 
   574)     num_vegsol = 0
   575)     num_novegsol = 0
   576)     do fp = 1,num_nourbanp
   577)        p = filter_nourbanp(fp)
   578)           if (coszen_pft(p) > 0._r8) then
   579)              if ((itypelun(plandunit(p)) == istsoil .or.  &
   580)                   itypelun(plandunit(p)) == istcrop     ) &
   581)                  .and. (elai(p) + esai(p)) > 0._r8        &
   582)                  .and. pwtgcell(p) > 0._r8) then
   583)                 num_vegsol = num_vegsol + 1
   584)                 filter_vegsol(num_vegsol) = p
   585)              else
   586)                 num_novegsol = num_novegsol + 1
   587)                 filter_novegsol(num_novegsol) = p
   588)              end if
   589)           end if
   590)     end do
   591) 
   592)     ! Weight reflectance/transmittance by lai and sai
   593)     ! Only perform on vegetated pfts where coszen > 0
   594) 
   595)     do fp = 1,num_vegsol
   596)        p = filter_vegsol(fp)
   597)        vai(p) = elai(p) + esai(p)
   598)        wl(p) = elai(p) / max( vai(p), mpe )
   599)        ws(p) = esai(p) / max( vai(p), mpe )
   600)     end do
   601) 
   602)     do ib = 1, numrad
   603)        do fp = 1,num_vegsol
   604)           p = filter_vegsol(fp)
   605)           rho(p,ib) = max( rhol(ivt(p),ib)*wl(p) + rhos(ivt(p),ib)*ws(p), mpe )
   606)           tau(p,ib) = max( taul(ivt(p),ib)*wl(p) + taus(ivt(p),ib)*ws(p), mpe )
   607)        end do
   608)     end do
   609) 
   610)     ! Calculate surface albedos and fluxes 
   611)     ! Only perform on vegetated pfts where coszen > 0
   612) 
   613)     call TwoStream (lbc, ubc, lbp, ubp, filter_vegsol, num_vegsol, &
   614)                     coszen_pft, vai, rho, tau)
   615)        
   616)     ! Determine values for non-vegetated pfts where coszen > 0
   617) 
   618)     do ib = 1,numrad
   619)        do fp = 1,num_novegsol
   620)           p = filter_novegsol(fp)
   621)           c = pcolumn(p)
   622)           fabd(p,ib) = 0._r8
   623)           fabi(p,ib) = 0._r8
   624)           ftdd(p,ib) = 1._r8
   625)           ftid(p,ib) = 0._r8
   626)           ftii(p,ib) = 1._r8
   627)           albd(p,ib) = albgrd(c,ib)
   628)           albi(p,ib) = albgri(c,ib)
   629)           gdir(p) = 0._r8
   630)        end do
   631)     end do
   632) 
   633)   end subroutine SurfaceAlbedo

   729)              if (ltype(l) == istsoil .or. ltype(l) == istcrop)  then ! soil
   730)                 inc    = max(0.11_r8-0.40_r8*h2osoi_vol(c,1), 0._r8)
   731)                 soilcol = isoicol(c)

   733)                 !albsod = min(albsat(soilcol,ib)+inc, albdry(soilcol,ib))

   744)                 ! changed from local variable to clm_type:
