    22)       subroutine r01( nw, wl, wc, tlev, airlev, jlabel, xs )
    23) !-----------------------------------------------------------------------------
    24) !   purpose:
    25) !   provide the product of (cross section) x (quantum yield) for the two
    26) !   o3 photolysis reactions:
    27) !              (a) o3 + hv -> o2 + o(1d)
    28) !              (b) o3 + hv -> o2 + o(3p)
    29) !   cross section:  combined data from wmo 85 ozone assessment (use 273k
    30) !                   value from 175.439-847.5 nm) and data from molina and
    31) !                   molina (use in hartley and huggins bans (240.5-350 nm)
    32) !   quantum yield:  choice between
    33) !                    (1) data from michelsen et al, 1994
    34) !                    (2) jpl 87 recommendation
    35) !                    (3) jpl 90/92 recommendation (no "tail")
    36) !                    (4) data from shetter et al., 1996
    37) !                    (5) jpl 97 recommendation
    38) !-----------------------------------------------------------------------------
    39) !   parameters:
    40) !   nw     - integer, number of specified intervals + 1 in working        (i)
    41) !            wavelength grid
    42) !   wl     - real(r8), vector of lower limits of wavelength intervals in      (i)
    43) !            working wavelength grid
    44) !   wc     - real(r8), vector of center points of wavelength intervals in     (i)
    45) !            working wavelength grid
    46) !   nz     - integer, number of altitude levels in working altitude grid  (i)
    47) !   tlev   - real(r8), temperature (k) at each specified altitude level       (i)
    48) !   airlev - real(r8), air density (molec/cc) at each altitude level          (i)
    49) !   j      - integer, counter for number of weighting functions defined  (io)
    50) !   sq     - real(r8), cross section x quantum yield (cm^2) for each          (o)
    51) !            photolysis reaction defined, at each defined wavelength and
    52) !            at each defined altitude level
    53) !   jlabel - character*40, string identifier for each photolysis reaction (o)
    54) !            defined
    55) !-----------------------------------------------------------------------------
    56) !   edit history:
    57) !   05/98  original, adapted from former jspec1 subroutine

    82)       real(r8), intent(in)         :: wc(kw)
    83)       real(r8), intent(in)         :: tlev(pverp)
    84)       real(r8), intent(in)         :: airlev(pverp)
    85)       real(r8), intent(inout)      :: xs(:,:)
    86)       character(len=*), intent(in) :: jlabel

    89) !	... local variables
    90) !-----------------------------------------------------------------------------
    91)       real(r8), parameter ::     c0  = 12._r8/19._r8
    92)       real(r8), parameter ::     a1  = 0.887_r8
    93)       real(r8), parameter ::     a2  = 2.35_r8
    94)       real(r8), parameter ::     a3  = 57.0_r8
    95)       real(r8), parameter ::     wc1 = 302._r8
    96)       real(r8), parameter ::     wc2 = 311.1_r8
    97)       real(r8), parameter ::     wc3 = 313.9_r8
    98)       real(r8), parameter ::     v2  = 820.0_r8
    99)       real(r8), parameter ::     v3  = 1190.0_r8
   100)       real(r8), parameter ::     w1  = 7.9_r8
   101)       real(r8), parameter ::     w2  = 2.2_r8
   102)       real(r8), parameter ::     w3  = 7.4_r8
   103)       real(r8), parameter ::     xk  = 0.695_r8
   104) 
   105)       integer  :: i, wn
   106)       integer  :: myld
   107)       real(r8) :: qy1d, qy3p, so3, wrk, tinv

   118) level_loop : &
   119)       do i = 1,pverp
   120)          tinv = 1._r8/tlev(i)
   121)          tl   = tlev(i) < 263._r8
   122)          if( tl ) then
   123)             wrk = (tlev(i) - 226._r8)/(263._r8 - 226._r8)
   124)          else
   125)             wrk = (tlev(i) - 263._r8)/(298._r8 - 263._r8)
   126)          end if
   127) wave_loop : &
   128)          do wn = 1,nw
   129)             if( wl(wn) > 240.5_r8  .and. wl(wn+1) < 350._r8 ) then
   130)                if( tl ) then
   131)                   so3 = s226(wn) + (s263(wn) - s226(wn)) * wrk
   132)                else
   133)                   so3 = s263(wn) + (s298(wn) - s263(wn)) * wrk
   134)                end if
   135)             else
   136)                so3 = xso3(wn)
   137)             end if
   138) !-----------------------------------------------------------------------------
   139) ! 	... from jpl97
   140) !-----------------------------------------------------------------------------
   141)              if( wc(wn) < 271._r8 ) then
   142)                 qy1d = .87_r8
   143)              else if( wc(wn) >= 271._r8 .and. wc(wn) < 290._r8 ) then
   144)                 qy1d = .87_r8 + (wc(wn) - 271._r8)*c0
   145)              else if( wc(wn) >= 290._r8 .and. wc(wn) < 305._r8 ) then
   146)                 qy1d = .95_r8
   147)              else if( wc(wn) >= 305._r8 .and. wc(wn) <= 325._r8 ) then
   148)                 qy1d = r01g1(wn) * exp ( -r01g2(wn)*tinv )
   149)              else
   150)                 qy1d = 0._r8
   151)              end if
   152) !-------------------------------------------------------------------------------
   153) !	... from jpl2000
   154) !-------------------------------------------------------------------------------
   155)              if( wc(wn) < 300._r8 ) then
   156)                 qy1d = 0.95_r8
   157)              else if( wc(wn) >= 300._r8 .and. wc(wn) < 331._r8 ) then
   158)                 qy1d = a1*exp( -((wc(wn) - wc1 )/w1)**4 ) &
   159)                      + a2*(tlev(i)/300._r8)**4*exp( -v2/xk*tinv ) &
   160)                     * exp( -((wc(wn) - wc2)/w2)**2 ) &
   161)                      + a3*exp( -v3/xk*tinv ) * exp( -((wc(wn) - wc3)/w3)**2 ) &
   162)                      + 0.06_r8
   163)              else if( wc(wn) >= 331._r8 .and. wc(wn) <= 345._r8 ) then
   164)                 qy1d = 0.06_r8
   165)              else
   166)                 qy1d = 0._r8
   167)              end if
   168) 
   169)              if( myld == kmats ) then
   170)                 qy1d = fo3qy( wc(wn), tlev(i) )
   171)              end if
   172) 
   173)              if( (trim( jlabel ) == 'jo1d') .or. (trim( jlabel ) == 'j2oh') ) then
   174)                 xs(wn,i) = qy1d*so3
   175)              else
   176)                 qy3p     = 1._r8 - qy1d
   177)                 xs(wn,i) = qy3p*so3
   178)              end if
   179)          end do wave_loop
   180)       end do level_loop
   181)

   184)       function fo3qy( w, t )
   185) !-----------------------------------------------------------------------------
   186) !   PURPOSE:
   187) ! function to calculate the quantum yield O3 + hv -> O(1D) + O2,
   188) ! according to:                                                             
   189) ! Matsumi, Y., F. J. Comes, G. Hancock, A. Hofzumanhays, A. J. Hynes,
   190) ! M. Kawasaki, and A. R. Ravishankara, QUantum yields for production of O(1D)
   191) ! in the ultraviolet photolysis of ozone:  Recommendation based on evaluation
   192) ! of laboratory data, J. Geophys. Res., 107, 10.1029/2001JD000510, 2002.
   193) !-----------------------------------------------------------------------------
   194) 
   195)       implicit none
   196) 
   197) !-----------------------------------------------------------------------------
   198) !	... dummy arguments
   199) !-----------------------------------------------------------------------------
   200)       real(r8), intent(in)  :: w
   201)       real(r8), intent(in)  :: t
   202) 
   203) !-----------------------------------------------------------------------------
   204) !	... local variables
   205) !-----------------------------------------------------------------------------
   206)       real(r8) :: kt
   207)       real(r8) :: q1
   208)       real(r8) :: q2 
   209)       real(r8) :: a(3)  = (/ 0.8036_r8, 8.9061_r8, 0.1192_r8 /)
   210)       real(r8) :: x(3)  = (/ 304.225_r8, 314.957_r8, 310.737_r8 /)
   211)       real(r8) :: om(3) = (/ 5.576_r8, 6.601_r8, 2.187_r8 /)
   212) 
   213) !-----------------------------------------------------------------------------
   214) !	... function declarations
   215) !-----------------------------------------------------------------------------
   216)       real(r8) :: fo3qy
   217)       
   218)       fo3qy = 0._r8
   219)       kt = 0.695_r8 * t
   220)       q1 = 1._r8
   221)       q2 = exp( -825.518_r8/kt )
   222)       
   223)       if( w <= 305._r8 ) then
   224)          fo3qy = .90_r8
   225)       else if( w > 305._r8 .and. w <= 328._r8 ) then
   226)          fo3qy = 0.0765_r8 &
   227)                 + a(1)*             (q1/(q1+q2))*exp( -((x(1)-w)/om(1))**4 ) &
   228)                 + a(2)*(t/300._r8)**2 *(q2/(q1+q2))*exp( -((x(2)-w)/om(2))**2 ) &
   229)                 + a(3)*(t/300._r8)**1.5_r8            *exp( -((x(3)-w)/om(3))**2 )
   230)       else if( w > 328._r8 .and. w <= 340._r8 ) then
   231)          fo3qy = 0.08_r8
   232)       else if( w > 340._r8 ) then
   233)          fo3qy = 0._r8
   234)       end if
   235) 
   236)       end function fo3qy

   238)       subroutine r04( nw, wl, wc, tlev, airlev, jlabel, xs )
   239) !-----------------------------------------------------------------------------
   240) !   purpose:
   241) !   provide product of (cross section) x (quantum yiels) for n2o5 photolysis
   242) !   reactions:
   243) !        (a) n2o5 + hv -> no3 + no + o(3p)
   244) !        (b) n2o5 + hv -> no3 + no2
   245) !   cross section from jpl97: use tabulated values up to 280 nm, use expon.
   246) !                             expression for >285nm, linearly interpolate
   247) !                             between s(280) and s(285,t) in between
   248) !   quantum yield: analysis of data in jpl94 (->dataj1/yld/n2o5.qy)
   249) !-----------------------------------------------------------------------------
   250) !   parameters:
   251) !   nw     - integer, number of specified intervals + 1 in working        (i)
   252) !            wavelength grid
   253) !   wl     - real(r8), vector of lower limits of wavelength intervals in      (i)
   254) !            working wavelength grid
   255) !   wc     - real(r8), vector of center points of wavelength intervals in     (i)
   256) !            working wavelength grid
   257) !   nz     - integer, number of altitude levels in working altitude grid  (i)
   258) !   tlev   - real(r8), temperature (k) at each specified altitude level       (i)
   259) !   airlev - real(r8), air density (molec/cc) at each altitude level          (i)
   260) !   j      - integer, counter for number of weighting functions defined  (io)
   261) !   sq     - real(r8), cross section x quantum yield (cm^2) for each          (o)
   262) !            photolysis reaction defined, at each defined wavelength and
   263) !            at each defined altitude level
   264) !   jlabel - character*40, string identifier for each photolysis reaction (o)
   265) !            defined
   266) !-----------------------------------------------------------------------------
   267) !   edit history:
   268) !   05/98  original, adapted from former jspec1 subroutine
   269) !-----------------------------------------------------------------------------
   270) 
   271)       use mo_params
   272)       use mo_waveall
   273)       use ppgrid,     only : pverp
   274) 
   275)       implicit none
   276) 
   277) !-----------------------------------------------------------------------------
   278) !	... dummy arguments
   279) !-----------------------------------------------------------------------------
   280)       integer, intent(in)          :: nw
   281)       real(r8), intent(in)         :: wl(kw)
   282)       real(r8), intent(in)         :: wc(kw)
   283)       real(r8), intent(in)         :: tlev(pverp)
   284)       real(r8), intent(in)         :: airlev(pverp)
   285)       real(r8), intent(inout)      :: xs(:,:)
   286)       character(len=*), intent(in) :: jlabel
   287) 
   288) !-----------------------------------------------------------------------------
   289) !	... local variables
   290) !-----------------------------------------------------------------------------
   291)       real(r8), parameter :: xs280 = 1.16999993e-19_r8
   292) 
   293)       integer  :: k, wn
   294)       real(r8) :: qy
   295)       real(r8) :: xsect, xst285
   296)       real(r8) :: t
   297) 
   298) !-----------------------------------------------------------------------------
   299) !	... n2o5 photodissociation
   300) !-----------------------------------------------------------------------------
   301) ! 	... cross section from jpl97, table up to 280 nm
   302) !           quantum yield : see dataj1/yld/n2o5.qy for explanation
   303) !           correct for t-dependence of cross section
   304) !-----------------------------------------------------------------------------
   305) level_loop : &
   306)       do k = 1,pverp
   307) !-----------------------------------------------------------------------------
   308) ! 	... temperature dependence only valid for 225 - 300 k.
   309) !-----------------------------------------------------------------------------
   310)          t = 1._r8/max( 225._r8,min( tlev(k),300._r8 ) )
   311) wave_loop : &
   312)          do wn = 1,nw
   313)             qy = max( 0._r8,min( 1._r8, 3.832441_r8 - 0.012809638_r8 * wc(wn) ) )
   314) !-----------------------------------------------------------------------------
   315) ! 	... evaluate exponential
   316) !-----------------------------------------------------------------------------
   317)             if( wl(wn) >= 285._r8 .and. wl(wn+1) <= 380._r8 ) then
   318) !!$               xs(wn,k) = qy        * 1.e-20_r8*exp( 2.735_r8 + (4728.5_r8 - 17.127_r8*wc(wn)) * t )
   319) ! fvitt - made to match the old jn2o5
   320)                xs(wn,k) = (1._r8 - qy) * 1.e-20_r8*exp( 2.735_r8 + (4728.5_r8 - 17.127_r8*wc(wn)) * t )
   321) !-----------------------------------------------------------------------------
   322) ! 	... between 280 and 285, interpolate between temperature evaluated exponential
   323) !           at 285 nm and the tabulated value at 280 nm.
   324) !-----------------------------------------------------------------------------
   325)             else if( wl(wn) >= 280._r8 .and. wl(wn+1) <= 286._r8 ) then
   326)                xst285 = 1.e-20_r8* exp( 2.735_r8 + (4728.5_r8 - 17.127_r8*286._r8)*t )
   327)                xsect  = xs280 + (wc(wn) - 280._r8)*(xst285 - xs280)/(286._r8 - 280._r8)
   328) !!$               xs(wn,k) = qy * xsect 
   329)                xs(wn,k) = (1._r8-qy) * xsect 
   330) !-----------------------------------------------------------------------------
   331) ! 	... use tabulated values
   332) !-----------------------------------------------------------------------------
   333)             else if (wl(wn) <= 280._r8 ) then
   334) !!$               xs(wn,k) = qy * r04g(wn)
   335)                xs(wn,k) = (1._r8-qy) * r04g(wn)
   336) !-----------------------------------------------------------------------------
   337) ! 	... beyond 380 nm, set to zero
   338) !-----------------------------------------------------------------------------
   339)             else
   340)                xs(wn,k) = 0._r8
   341)             end if
   342)          end do wave_loop
   343)       end do level_loop
   344) 
   345)       end subroutine r04

   397)       end do

   401)       subroutine r44( nw, wl, wc, tlev, airlev, jlabel, xs )
   402) !-----------------------------------------------------------------------------
   403) !   purpose:
   404) !   provide product (cross section) x (quantum yield) for n2o photolysis:
   405) !               n2o + hv -> n2 + o(1d)
   406) !   cross section: from jpl 97 recommendation
   407) !   quantum yield: assumed to be unity, based on greenblatt and ravishankara
   408) !-----------------------------------------------------------------------------
   409) !   parameters:
   410) !   nw     - integer, number of specified intervals + 1 in working        (i)
   411) !            wavelength grid
   412) !   wl     - real(r8), vector of lower limits of wavelength intervals in      (i)
   413) !            working wavelength grid
   414) !   wc     - real(r8), vector of center points of wavelength intervals in     (i)
   415) !            working wavelength grid
   416) !   nz     - integer, number of altitude levels in working altitude grid  (i)
   417) !   tlev   - real(r8), temperature (k) at each specified altitude level       (i)
   418) !   airlev - real(r8), air density (molec/cc) at each altitude level          (i)
   419) !   j      - integer, counter for number of weighting functions defined  (io)
   420) !   sq     - real(r8), cross section x quantum yield (cm^2) for each          (o)
   421) !            photolysis reaction defined, at each defined wavelength and
   422) !            at each defined altitude level
   423) !   jlabel - character*40, string identifier for each photolysis reaction (o)
   424) !            defined
   425) !-----------------------------------------------------------------------------
   426) !   edit history:
   427) !   05/98  original, adapted from former jspec1 subroutine
   428) !-----------------------------------------------------------------------------
   429) 
   430)       use mo_params
   431)       use ppgrid,    only : pverp
   432) 
   433)       implicit none
   434) 
   435) !-----------------------------------------------------------------------------
   436) !	... dummy arguments
   437) !-----------------------------------------------------------------------------
   438)       integer, intent(in)          :: nw
   439)       real(r8), intent(in)         :: wl(kw), wc(kw)
   440)       real(r8), intent(in)         :: tlev(pverp)
   441)       real(r8), intent(in)         :: airlev(pverp)
   442)       real(r8), intent(inout)      :: xs(:,:)
   443)       character(len=*), intent(in) :: jlabel
   444) 
   445) !-----------------------------------------------------------------------------
   446) !	... local variables
   447) !-----------------------------------------------------------------------------
   448)       integer  :: k
   449)       real(r8) :: t
   450) 
   451) !-----------------------------------------------------------------------------
   452) !	... n2o photodissociation
   453) !-----------------------------------------------------------------------------
   454) !	... quantum yield of n(4s) and no(2pi) is less than 1% (greenblatt and
   455) !           ravishankara), so quantum yield of o(1d) is assumed to be unity
   456) !-----------------------------------------------------------------------------
   457)       do k = 1,pverp
   458)          t = max( 194._r8,min( tlev(k),320._r8 ) ) - 300._r8
   459)          where( wc(:nw) >= 173._r8 .and. wc(:nw) <= 240._r8 )
   460)             xs(:nw,k) = exp( a(:nw) + t*exp( b(:nw) ) )
   461)          elsewhere
   462)             xs(:nw,k) = 0._r8
   463)          endwhere
   464)       end do
   465) 
   466)       end subroutine r44
