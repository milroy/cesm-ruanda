   114)   subroutine ice_init_mct( EClock, cdata_i, x2i_i, i2x_i, NLFilename )
   115) !
   116) ! !DESCRIPTION:
   117) ! Initialize thermodynamic ice model and obtain relevant atmospheric model
   118) ! arrays back from driver 
   119) !
   120) ! !USES:
   121) 
   122)     use CICE_InitMod
   123)     use ice_restart, only: runid, runtype, restart_dir, restart_format
   124)     use ice_history, only: history_dir, history_file
   125) !
   126) ! !ARGUMENTS:
   127)     type(ESMF_Clock)         , intent(inout) :: EClock
   128)     type(seq_cdata)          , intent(inout) :: cdata_i
   129)     type(mct_aVect)          , intent(inout) :: x2i_i, i2x_i
   130)     character(len=*), optional  , intent(in) :: NLFilename ! Namelist filename
   131) !
   132) ! !LOCAL VARIABLES:
   133) !
   134)     type(mct_gsMap)             , pointer :: gsMap_ice
   135)     type(mct_gGrid)             , pointer :: dom_i
   136)     type(seq_infodata_type)     , pointer :: infodata   ! Input init object
   137)     integer                               :: lsize,lsize_loc
   138)     integer                               :: xoff,yoff
   139)     integer                               :: nxg,nyg
   140)     integer                               :: k
   141)  
   142)     type(mct_gsMap) :: gsmap_extend  ! local gsmaps
   143) 
   144)     character(len=256) :: drvarchdir         ! driver archive directory
   145)     character(len=32)  :: starttype          ! infodata start type
   146)     integer            :: start_ymd          ! Start date (YYYYMMDD)
   147)     integer            :: start_tod          ! start time of day (s)
   148)     integer            :: curr_ymd           ! Current date (YYYYMMDD)
   149)     integer            :: curr_tod           ! Current time of day (s)
   150)     integer            :: ref_ymd            ! Reference date (YYYYMMDD)
   151)     integer            :: ref_tod            ! reference time of day (s)
   152)     integer            :: iyear              ! yyyy
   153)     integer            :: nyrp               ! yyyy
   154)     integer            :: dtime              ! time step
   155)     integer            :: shrlogunit,shrloglev ! old values
   156)     integer            :: iam,ierr
   157)     integer            :: lbnum
   158)     integer            :: daycal(13)  !number of cumulative days per month
   159)     integer            :: nleaps      ! number of leap days before current year
   160)     integer            :: mpicom_loc  ! temporary mpicom
   161)     logical (kind=log_kind) :: atm_aero
   162)     real(r8) :: mrss, mrss0,msize,msize0
   163)     character(len=*), parameter  :: SubName = "ice_init_mct"
   164) ! !REVISION HISTORY:
   165) ! Author: Mariana Vertenstein
   166) !EOP
   167) !-----------------------------------------------------------------------
   168) 
   169)     !--------------------------------------------------------------------------
   170)     ! Determine attribute vector indices
   171)     !--------------------------------------------------------------------------
   172) 
   173)     call ice_cpl_indices_set()
   174) 
   175)     !---------------------------------------------------------------------------
   176)     ! Set cdata pointers
   177)     !---------------------------------------------------------------------------
   178) 
   179)     call seq_cdata_setptrs(cdata_i, ID=ICEID, mpicom=mpicom_loc, &
   180)          gsMap=gsMap_ice, dom=dom_i, infodata=infodata)
   181) 
   182)     ! Determine time of next atmospheric shortwave calculation
   183)     call seq_infodata_GetData(infodata, nextsw_cday=nextsw_cday )
   184) 
   185)     ! Determine if aerosols are coming from the coupler
   186)     call seq_infodata_GetData(infodata, atm_aero=atm_aero )
   187) 
   188)     ! Determine orbital parameters
   189)     call seq_infodata_GetData(infodata, orb_eccen=eccen, orb_mvelpp=mvelpp, &
   190)          orb_lambm0=lambm0, orb_obliqr=obliqr)
   191) 
   192)     !   call shr_init_memusage()
   193) 
   194)     !---------------------------------------------------------------------------
   195)     ! use infodata to determine type of run
   196)     !---------------------------------------------------------------------------
   197) 
   198)     ! Preset single column values
   199) 
   200)     single_column = .false.
   201)     scmlat = -999.
   202)     scmlon = -999.
   203) 
   204)     call seq_infodata_GetData( infodata, case_name=runid   ,  &  
   205)        single_column=single_column ,scmlat=scmlat,scmlon=scmlon)
   206)     call seq_infodata_GetData( infodata, start_type=starttype)
   207) 
   208)     if (     trim(starttype) == trim(seq_infodata_start_type_start)) then
   209)        runtype = "initial"
   210)     else if (trim(starttype) == trim(seq_infodata_start_type_cont) ) then
   211)        runtype = "continue"
   212)     else if (trim(starttype) == trim(seq_infodata_start_type_brnch)) then
   213)        runtype = "branch"
   214)     else
   215)        write(nu_diag,*) 'ice_comp_mct ERROR: unknown starttype'
   216)        call shr_sys_abort()
   217)     end if
   218) 
   219)     ! Set nextsw_cday to -1 for continue and branch runs.
   220) 
   221)     if (trim(runtype) /= 'initial') nextsw_cday = -1
   222) 
   223)     !=============================================================
   224)     ! Set ice dtime to ice coupling frequency
   225)     !=============================================================
   226) 
   227)     call seq_timemgr_EClockGetData(EClock, dtime=dtime, calendar=calendar_type)
   228)     dt = real(dtime)
   229) 
   230)     !=============================================================
   231)     ! Initialize cice because grid information is needed for
   232)     ! creation of GSMap_ice.  cice_init also sets time manager info
   233)     !=============================================================
   234) 
   235)     inst_name   = seq_comm_name(ICEID)
   236)     inst_index  = seq_comm_inst(ICEID)
   237)     inst_suffix = seq_comm_suffix(ICEID)
   238) 
   239)     call t_startf ('cice_init')
   240)     call cice_init( mpicom_loc )
   241)     call t_stopf ('cice_init')
   242)     call init_numIceCells
   243) 
   244)     !---------------------------------------------------------------------------
   245)     ! Reset shr logging to my log file
   246)     !---------------------------------------------------------------------------
   247) 
   248)     call shr_file_getLogUnit (shrlogunit)
   249)     call shr_file_getLogLevel(shrloglev)
   250)     call shr_file_setLogUnit (nu_diag)
   251)    
   252)     !---------------------------------------------------------------------------
   253)     ! use EClock to reset calendar information on initial start
   254)     !---------------------------------------------------------------------------
   255) 
   256)     ! - the following logic duplicates the logic for the concurrent system - 
   257)     ! cice_init is called then init_cpl is called where the start date is received
   258)     ! from the flux coupler
   259)     ! - in the following calculation for the variable time, iyear is used
   260)     ! rather than (iyear-1) (as in init_cpl) since the sequential system permits you 
   261)     ! to start with year "0" not year "1"
   262)     ! - on restart run 
   263)     !   - istep0, time and time_forc are read from restart file
   264)     !   - istep1 is set to istep0
   265)     !   - idate is determined from time via the call to calendar (see below)
   266)     ! - on initial run 
   267)     !   - iyear, month and mday obtained from sync clock
   268)     !   - time determined from iyear, month and mday
   269)     !   - istep0 and istep1 are set to 0 
   270) 
   271)     call seq_timemgr_EClockGetData(EClock,               &
   272)          start_ymd=start_ymd, start_tod=start_tod,       &
   273)          curr_ymd=curr_ymd,   curr_tod=curr_tod,         &
   274)          ref_ymd=ref_ymd,     ref_tod=ref_tod)
   275) 
   276)     if (runtype == 'initial') then
   277)        if (ref_ymd /= start_ymd .or. ref_tod /= start_tod) then
   278)           if (my_task == master_task) then
   279)              write(nu_diag,*) 'ice_comp_mct: ref_ymd ',ref_ymd, &
   280)                   ' must equal start_ymd ',start_ymd
   281)              write(nu_diag,*) 'ice_comp_mct: ref_ymd ',ref_tod, &
   282)                   ' must equal start_ymd ',start_tod
   283)           end if
   284)        end if
   285) 
   286)        if (my_task == master_task) then
   287)           write(nu_diag,*) '(ice_init_mct) idate from sync clock = ', &
   288)                start_ymd
   289)           write(nu_diag,*) '(ice_init_mct)   tod from sync clock = ', &
   290)                start_tod
   291)           write(nu_diag,*) &
   292)                '(ice_init_mct) resetting idate to match sync clock'
   293)        end if
   294) 
   295)        idate = curr_ymd
   296)        iyear = (idate/10000)                     ! integer year of basedate
   297)        if (calendar_type .eq. "GREGORIAN") then 	
   298)           nyr = iyear+1
   299)           new_year = .false.
   300)        end if
   301)        month = (idate-iyear*10000)/100           ! integer month of basedate
   302)        mday  =  idate-iyear*10000-month*100-1    ! day of month of basedate
   303)                                                  ! (starts at 0)
   304) 
   305)        call get_daycal(year=iyear,days_per_year_in=days_per_year, &
   306)                        daycal_out=daycal)
   307) 
   308)        nleaps = leap_year_count(iyear)   ! this sets nleaps in ice_calendar
   309)        time  = (((iyear)*days_per_year  + nleaps + daycal(month)+mday)*secday) &
   310)              + start_tod    
   311) 
   312)        call shr_sys_flush(nu_diag)
   313)     end if
   314) 
   315)     call calendar(time)     ! update calendar info
   316)  
   317)     !---------------------------------------------------------------------------
   318)     ! Initialize MCT attribute vectors and indices
   319)     !---------------------------------------------------------------------------
   320) 
   321)     call t_startf ('cice_mct_init')
   322) 
   323)     ! Initialize ice gsMap
   324) 
   325)     if (trim(gridcpl_file) == 'unknown_gridcpl_file') then
   326)        call ice_SetGSMap_mct( MPI_COMM_ICE, ICEID, GSMap_ice ) 
   327)        lsize = mct_gsMap_lsize(gsMap_ice, MPI_COMM_ICE)
   328)        call ice_domain_mct( lsize, gsMap_ice, dom_i )
   329)        other_cplgrid = .false.
   330)        nxg = nx_global
   331)        nyg = ny_global
   332)     else
   333)        call ice_SetGSMap_mct( MPI_COMM_ICE, ICEID, GSMap_iloc ) 
   334)        lsize_loc = mct_gsMap_lsize(gsMap_iloc, MPI_COMM_ICE)
   335)        call ice_domain_mct( lsize_loc, gsMap_iloc, dom_iloc )
   336)  
   337)        call ice_setcoupling_mct(MPI_COMM_ICE, ICEID, gsmap_ice, dom_i)
   338)        lsize = mct_gsMap_lsize(gsMap_ice, MPI_COMM_ICE)
   339)  
   340)        call ice_coffset_mct(xoff,yoff,gsmap_iloc,dom_iloc,gsmap_ice,dom_i,MPI_COMM_ICE)
   341)  
   342)        call ice_SetGSMap_mct( MPI_COMM_ICE, ICEID, gsmap_extend, xoff, yoff, nxcpl, nycpl)
   343)        if (lsize_loc /= mct_gsmap_lsize(gsmap_extend,MPI_COMM_ICE)) then
   344)           write(nu_diag,*) subname,' :: gsmap_extend extended ',lsize_loc, &
   345)              mct_gsmap_lsize(gsmap_extend,MPI_COMM_ICE)
   346)           call shr_sys_abort(subname//' :: error in gsmap_extend extended')
   347)        endif
   348) 
   349)        call mct_rearr_init(gsmap_ice, gsmap_extend, MPI_COMM_ICE, rearr_ice2iloc)
   350)        call mct_rearr_init(gsmap_extend, gsmap_ice, MPI_COMM_ICE, rearr_iloc2ice)
   351)        call mct_aVect_init(x2i_iloc, rList=seq_flds_x2i_fields, lsize=lsize_loc)
   352)        call mct_aVect_zero(x2i_iloc)
   353)        call mct_aVect_init(i2x_iloc, rList=seq_flds_i2x_fields, lsize=lsize_loc)
   354)        call mct_aVect_zero(i2x_iloc)
   355)        call mct_gsmap_clean(gsmap_extend)
   356)  
   357)        other_cplgrid = .true.
   358)        nxg = nxcpl
   359)        nyg = nycpl
   360)     endif
   361) 
   362)     ! Inialize mct attribute vectors
   363) 
   364)     call mct_aVect_init(x2i_i, rList=seq_flds_x2i_fields, lsize=lsize)
   365)     call mct_aVect_zero(x2i_i)
   366) 
   367)     call mct_aVect_init(i2x_i, rList=seq_flds_i2x_fields, lsize=lsize) 
   368)     call mct_aVect_zero(i2x_i)
   369) 
   370)     !-----------------------------------------------------------------
   371)     ! Prescribed ice initialization
   372)     !-----------------------------------------------------------------
   373) 
   374)     if (other_cplgrid) then
   375)        call ice_prescribed_init(ICEID, gsmap_iloc, dom_iloc)
   376)     else
   377)        call ice_prescribed_init(ICEID, gsmap_ice, dom_i)
   378)     endif
   379) 
   380)     !-----------------------------------------------------------------
   381)     ! Get ready for coupling
   382)     !-----------------------------------------------------------------
   383) 
   384)     call coupling_prep
   385) 
   386)     !---------------------------------------------------------------------------
   387)     ! Fill in export state for driver
   388)     !---------------------------------------------------------------------------
   389) 
   390)     if (other_cplgrid) then
   391)        call ice_export (i2x_iloc%rattr)  !Send initial state to driver
   392)        call ice_setdef_mct ( i2x_i )
   393)        call mct_rearr_rearrange(i2x_iloc, i2x_i, rearr_iloc2ice)
   394)     else
   395)        call ice_export (i2x_i%rattr)  !Send initial state to driver
   396)     endif
   397)     call seq_infodata_PutData( infodata, ice_prognostic=.true., &
   398)       iceberg_prognostic=.false., ice_nx = nxg, ice_ny = nyg )
   399)     call t_stopf ('cice_mct_init')
   400) 
   401)     ! Error check
   402)     if (tr_aero .and. .not. atm_aero) then
   403)        write(nu_diag,*) 'ice_import ERROR: atm_aero must be set for tr_aero' 
   404)        call shr_sys_abort()
   405)     end if
   406) 
   407)     !---------------------------------------------------------------------------
   408)     ! Reset shr logging to original values
   409)     !---------------------------------------------------------------------------
   410) 
   411)     call shr_file_setLogUnit (shrlogunit)
   412)     call shr_file_setLogLevel(shrloglev)
   413) 
   414)     !   call ice_timer_stop(timer_total) ! time entire run
   415)     !   call shr_get_memusage(msize,mrss)
   416)     !   call shr_mpi_max(mrss, mrss0, MPI_COMM_ICE,'ice_init_mct mrss0')
   417)     !   call shr_mpi_max(msize,msize0,MPI_COMM_ICE,'ice_init_mct msize0')
   418)     !   if(my_task == 0) then
   419)     !   write(shrlogunit,105) 'ice_init_mct: memory_write: model date = ',start_ymd,start_tod, &
   420)     !           ' memory = ',msize0,' MB (highwater)    ',mrss0,' MB (usage)'
   421)     !   endif
   422)  
   423)   105  format( A, 2i8, A, f10.2, A, f10.2, A)
   424) 
   425)   end subroutine ice_init_mct

   858)   subroutine ice_SetGSMap_mct( mpicom, ID, gsMap_ice, xoff, yoff, nxgin, nygin )
   859) 
   860)     !-------------------------------------------------------------------
   861)     !
   862)     ! Arguments
   863)     !
   864)     implicit none

   869)     integer,optional, intent(in)   :: yoff   ! y offset
   870)     integer,optional, intent(in)   :: nxgin ! global size
   871)     integer,optional, intent(in)   :: nygin ! global size

   873)     ! Local variables

   875)     integer,allocatable :: gindex(:)

   878)     integer     :: i, j, iblk, n, gi

   884)     !-------------------------------------------------------------------
   885) 
   886)     ! Build the CICE grid numbering for MCT
   887)     ! NOTE:  Numbering scheme is: West to East and South to North
   888)     ! starting at south pole.  Should be the same as what's used

   890)

   906)        nyg = nygin
   907)     endif
   908)     gsize = nxg*nyg
   909) 
   910)     ! number the local grid
   911) 
   912)     n=0
   913)     do iblk = 1, nblocks
   914)        this_block = get_block(blocks_ice(iblk),iblk)         
   915)        ilo = this_block%ilo
   916)        ihi = this_block%ihi
   917)        jlo = this_block%jlo
   918)        jhi = this_block%jhi
   919)        
   920)        do j = jlo, jhi
   921)           do i = ilo, ihi
   922)              n = n+1
   923)           enddo !i
   924)        enddo    !j
   925)     enddo        !iblk
   926)     lsize = n

   929)     n=0

   932)        ilo = this_block%ilo
   933)        ihi = this_block%ihi

   935)        jhi = this_block%jhi
   936)        
   937)        do j = jlo, jhi

   939)              n = n+1
   940)              lon = this_block%i_glob(i) + lxoff - 1
   941)              lat = this_block%j_glob(j) + lyoff - 1
   942)              gi = (lat-1)*nxg + lon
   943)              gindex(n) = gi
   944)           enddo !i
   945)        enddo    !j

   947)     
   948)     call mct_gsMap_init( gsMap_ice, gindex, mpicom, ID, lsize, gsize )

   954)   subroutine ice_domain_mct( lsize, gsMap_i, dom_i )
   955) 
   956)     !-------------------------------------------------------------------
   957)     !
   958)     ! Arguments
   959)     !
   960)     integer        , intent(in)    :: lsize
   961)     type(mct_gsMap), intent(in)    :: gsMap_i

   964)     ! Local Variables
   965)     !
   966)     integer :: i, j, iblk, n, gi           ! indices

   969)     real(dbl_kind), pointer :: data(:)     ! temporary
   970)     integer       , pointer :: idata(:)    ! temporary
   971)     type(block)             :: this_block  ! block information for current block
   972)     !-------------------------------------------------------------------
   973)     !
   974)     ! Initialize mct domain type
   975)     ! lat/lon in degrees,  area in radians^2, mask is 1 (ocean), 0 (non-ocean)
   976)     !
   977)     call mct_gGrid_init( GGrid=dom_i, CoordChars=trim(seq_flds_dom_coord), &
   978)        OtherChars=trim(seq_flds_dom_other), lsize=lsize )
   979)     call mct_aVect_zero(dom_i%data)
   980)     !  
   981)     allocate(data(lsize))

   983)     ! Determine global gridpoint number attribute, GlobGridNum, which is set automatically by MCT
   984)     !
   985)     call mct_gsMap_orderedPoints(gsMap_i, my_task, idata)
   986)     call mct_gGrid_importIAttr(dom_i,'GlobGridNum',idata,lsize)
   987)     !
   988)     ! Determine domain (numbering scheme is: West to East and South to North to South pole)
   989)     ! Initialize attribute vector with special value

   991)     data(:) = -9999.0_R8

   993)     call mct_gGrid_importRAttr(dom_i,"lon"  ,data,lsize) 
   994)     call mct_gGrid_importRAttr(dom_i,"area" ,data,lsize) 
   995)     call mct_gGrid_importRAttr(dom_i,"aream",data,lsize) 
   996)     data(:) = 0.0_R8     
   997)     call mct_gGrid_importRAttr(dom_i,"mask",data,lsize) 
   998)     call mct_gGrid_importRAttr(dom_i,"frac",data,lsize) 
   999)     !
  1000)     ! Fill in correct values for domain components
  1001)     !
  1002)     allocate(work_dom(lsize)) 
  1003)     work_dom(:) = 0.0_dbl_kind
  1004) 
  1005)     n=0
  1006)     do iblk = 1, nblocks
  1007)        this_block = get_block(blocks_ice(iblk),iblk)         
  1008)        ilo = this_block%ilo
  1009)        ihi = this_block%ihi
  1010)        jlo = this_block%jlo
  1011)        jhi = this_block%jhi
  1012)        
  1013)        do j = jlo, jhi
