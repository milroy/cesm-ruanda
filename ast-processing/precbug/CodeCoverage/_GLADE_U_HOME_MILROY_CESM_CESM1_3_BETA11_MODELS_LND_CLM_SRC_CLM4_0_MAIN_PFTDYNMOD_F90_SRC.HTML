    66)   subroutine pftdyn_init()
    67) !
    68) ! !DESCRIPTION:
    69) ! Initialize dynamic landuse dataset (position it to the right time samples
    70) ! that bound the initial model date)
    71) !
    72) ! !USES:
    73)     use clm_time_manager, only : get_curr_date
    74)     use clm_varctl  , only : flanduse_timeseries
    75)     use clm_varpar  , only : numpft, maxpatch_pft
    76)     use fileutils   , only : getfil
    77) !
    78) ! !ARGUMENTS:
    79)     implicit none
    80) !
    81) !
    82) ! !LOCAL VARIABLES:
    83) !EOP
    84)     integer  :: i,j,m,n,g                       ! indices
    85)     real(r8) :: sumpct                          ! sum for error check
    86)     integer  :: varid                           ! netcdf ids
    87)     integer  :: year                            ! year (0, ...) for nstep+1
    88)     integer  :: mon                             ! month (1, ..., 12) for nstep+1
    89)     integer  :: day                             ! day of month (1, ..., 31) for nstep+1
    90)     integer  :: sec                             ! seconds into current date for nstep+1
    91)     integer  :: ier, ret                        ! error status
    92)     logical  :: found                           ! true => input dataset bounding dates found
    93)     logical  :: readvar	                        ! true => variable is on input dataset
    94)     integer  :: begg,endg                       ! beg/end indices for land gridcells
    95)     integer  :: begl,endl                       ! beg/end indices for land landunits
    96)     integer  :: begc,endc                       ! beg/end indices for land columns
    97)     integer  :: begp,endp                       ! beg/end indices for land pfts
    98)     real(r8), pointer :: pctgla(:)          ! percent of gcell is glacier
    99)     real(r8), pointer :: pctlak(:)          ! percent of gcell is lake
   100)     real(r8), pointer :: pctwet(:)          ! percent of gcell is wetland
   101)     real(r8), pointer :: pcturb(:)          ! percent of gcell is urbanized
   102)     type(gridcell_type), pointer :: gptr        ! pointer to gridcell derived subtype
   103)     character(len=256) :: locfn                 ! local file name
   104)     character(len= 32) :: subname='pftdyn_init' ! subroutine name
   105)  !-----------------------------------------------------------------------
   106) 
   107)     call get_proc_bounds(begg,endg,begl,endl,begc,endc,begp,endp)
   108) 
   109)     ! Error check
   110) 
   111)     if ( maxpatch_pft /= numpft+1 )then
   112)        call endrun( subname//' maxpatch_pft does NOT equal numpft+1 -- this is invalid for dynamic PFT case' )
   113)     end if
   114) 
   115)     allocate(pctgla(begg:endg),pctlak(begg:endg))
   116)     allocate(pctwet(begg:endg),pcturb(begg:endg))
   117) 
   118)     ! Set pointers into derived type
   119) 
   120)     gptr => grc
   121) 
   122)     ! pctspec must be saved between time samples
   123)     ! position to first time sample - assume that first time sample must match starting date
   124)     ! check consistency -  special landunits, grid, frac and mask
   125)     ! only do this once
   126) 
   127)     ! read data PCT_PFT corresponding to correct year
   128) 
   129)     allocate(wtpft1(begg:endg,0:numpft), wtpft2(begg:endg,0:numpft), stat=ier)
   130)     if (ier /= 0) then
   131)        call endrun( subname//' allocation error for wtpft1, wtpft2' )
   132)     end if
   133)     
   134)     allocate(harvest(begg:endg),stat=ier)
   135)     if (ier /= 0) then
   136)        call endrun( subname//' allocation error for harvest')
   137)     end if
   138) 
   139)     allocate(wtcol_old(begp:endp),stat=ier)
   140)     if (ier /= 0) then
   141)        call endrun( subname//' allocation error for wtcol_old' )
   142)     end if
   143) 
   144)     if (masterproc) then
   145)        write(iulog,*) 'Attempting to read pft dynamic landuse data .....'
   146)     end if
   147) 
   148)     ! Obtain file
   149)     call getfil (flanduse_timeseries, locfn, 0)
   150)     call ncd_pio_openfile (ncid, locfn, 0)
   151) 
   152)     ! Obtain pft years from dynamic landuse file
   153)     
   154)     call ncd_inqdid(ncid, 'time', varid)
   155)     call ncd_inqdlen(ncid, varid, ntimes)
   156) 
   157)     ! Consistency check
   158)     
   159)     call check_dim(ncid, 'lsmpft', numpft+1)
   160) 
   161)     allocate (yearspft(ntimes), stat=ier)
   162)     if (ier /= 0) then
   163)        write(iulog,*) subname//' allocation error for yearspft'; call endrun()
   164)     end if
   165) 
   166)     call ncd_io(ncid=ncid, varname='YEAR', flag='read', data=yearspft)
   167) 
   168)     call ncd_io(ncid=ncid, varname='PCT_WETLAND', flag='read', data=pctwet, &
   169)          dim1name=grlnd, readvar=readvar)
   170)     if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_WETLAND NOT on landuse_timeseries file' )
   171) 
   172)     call ncd_io(ncid=ncid, varname= 'PCT_LAKE', flag='read', data=pctlak, &
   173)          dim1name=grlnd, readvar=readvar)
   174)     if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_LAKE NOT on landuse_timeseries file' )
   175) 
   176)     call ncd_io(ncid=ncid, varname= 'PCT_GLACIER', flag='read', data=pctgla, &
   177)          dim1name=grlnd, readvar=readvar)
   178)     if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_GLACIER NOT on landuse_timeseries file' )
   179) 
   180)     call ncd_io(ncid=ncid, varname= 'PCT_URBAN'  , flag='read', data=pcturb, &
   181)          dim1name=grlnd, readvar=readvar)
   182)     if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_URBAN NOT on landuse_timeseries file' )
   183) 
   184)     ! Consistency check
   185)     do g = begg,endg
   186)     !   this was causing a fail, even though values are the same to within 1e-15
   187)     !   if (pctlak(g)+pctwet(g)+pcturb(g)+pctgla(g) /= pctspec(g)) then 
   188)        if (abs((pctlak(g)+pctwet(g)+pcturb(g)+pctgla(g))-pctspec(g)) > 1e-13_r8) then 
   189)           write(iulog,*) subname//'mismatch between input pctspec = ',&
   190)                      pctlak(g)+pctwet(g)+pcturb(g)+pctgla(g),&
   191)                     ' and that obtained from surface dataset ', pctspec(g),' at g= ',g
   192)            call endrun()
   193)        end if
   194)     end do
   195) 
   196)     ! Determine if current date spans the years
   197)     ! If current year is less than first dynamic PFT timeseries year,
   198)     ! then use the first year from dynamic pft file for both nt1 and nt2,
   199)     ! forcing constant weights until the model year enters the dynamic
   200)     ! pft dataset timeseries range.
   201)     ! If current year is equal to or greater than the last dynamic pft
   202)     ! timeseries year, then use the last year for both nt1 and nt2, 
   203)     ! forcing constant weights for the remainder of the simulation.
   204)     ! This mechanism permits the introduction of a dynamic pft period in the middle
   205)     ! of a simulation, with constant weights before and after the dynamic period.
   206)     ! PET: harvest - since harvest is specified as a rate for each year, this
   207)     ! approach will not work. Instead, need to seta flag that indicates harvest is
   208)     ! zero for the period before the beginning and after the end of the dynpft timeseries.
   209) 
   210)     call get_curr_date(year, mon, day, sec)
   211) 
   212)     if (year < yearspft(1)) then
   213)        nt1 = 1
   214)        nt2 = 1
   215)        do_harvest = .false.
   216)     else if (year >= yearspft(ntimes)) then
   217)        nt1 = ntimes
   218)        nt2 = ntimes
   219)        do_harvest = .false.
   220)     else
   221)        found = .false.
   222)        do n = 1,ntimes-1 
   223)           if (year == yearspft(n)) then
   224)              nt1 = n
   225)              nt2 = nt1 + 1
   226)              found = .true.
   227)              do_harvest = .true.
   228)           end if   
   229)        end do
   230)        if (.not. found) then
   231)           write(iulog,*) subname//' error: model year not found in landuse_timeseries file'
   232)           write(iulog,*)'model year = ',year
   233)           call endrun()
   234)        end if
   235)     end if
   236) 
   237)     ! Get pctpft time samples bracketing the current time
   238) 
   239)     if (masterproc) then
   240)        write(iulog,*) 'Get PFTDYN data for year: ', yearspft(nt1)
   241)     end if
   242)     call pftdyn_getdata(nt1, wtpft1, begg,endg,0,numpft)
   243)     if (masterproc) then
   244)        write(iulog,*) 'Get PFTDYN data for year: ', yearspft(nt2)
   245)     end if
   246)     call pftdyn_getdata(nt2, wtpft2, begg,endg,0,numpft)
   247)     
   248)     if (use_cn) then
   249)        ! Get harvest rate at the nt1 time
   250)        call pftdyn_getharvest(nt1,begg,endg)
   251)     end if
   252) 
   253)     ! convert weights from percent to proportion
   254)     do m = 0,numpft
   255)        do g = begg,endg
   256)           wtpft1(g,m) = wtpft1(g,m)/100._r8
   257)           wtpft2(g,m) = wtpft2(g,m)/100._r8
   258)        end do
   259)     end do
   260)        
   261)     deallocate(pctgla,pctlak,pctwet,pcturb)
   262) 
   263)   end subroutine pftdyn_init

   268) ! !ROUTINE: pftdyn_interp
   269) !
   270) ! !INTERFACE:
   271)   subroutine pftdyn_interp()
   272) !
   273) ! !DESCRIPTION:
   274) ! Time interpolate dynamic landuse data to get pft weights for model time
   275) ! Note that harvest data are stored as rates (not weights) and so time interpolation is 
   276) ! not necessary - the harvest rate is held constant through the year.  This is consistent with
   277) ! the treatment of changing PFT weights, where interpolation of the annual endpoint weights leads to 
   278) ! a constant rate of change in PFT weight through the year, with abrupt changes in the rate at
   279) ! annual boundaries. This routine is still used to get the next harvest time slice, when needed.
   280) ! This routine is also used to turn off the harvest switch when the model year runs past the end of
   281) ! the dynpft time series.
   282) !
   283) ! !USES:
   284)     use clm_time_manager, only : get_curr_date, get_curr_calday, &
   285)                                  get_days_per_year
   286)     use clm_varcon      , only : istsoil
   287)     use clm_varcon      , only : istcrop
   288)     use clm_varpar      , only : numpft
   289)     implicit none
   290) !
   291) !
   292) ! !LOCAL VARIABLES:
   293) !EOP
   294) !
   295) ! !ARGUMENTS:
   296)     integer  :: begg,endg        ! beg/end indices for land gridcells
   297)     integer  :: begl,endl        ! beg/end indices for land landunits
   298)     integer  :: begc,endc        ! beg/end indices for land columns
   299)     integer  :: begp,endp        ! beg/end indices for land pfts
   300)     integer  :: i,j,m,p,l,g,c    ! indices
   301)     integer  :: year             ! year (0, ...) for nstep+1
   302)     integer  :: mon              ! month (1, ..., 12) for nstep+1
   303)     integer  :: day              ! day of month (1, ..., 31) for nstep+1
   304)     integer  :: sec              ! seconds into current date for nstep+1
   305)     real(r8) :: cday             ! current calendar day (1.0 = 0Z on Jan 1)
   306)     real(r8) :: days_per_year    ! days per year
   307)     integer  :: ier              ! error status
   308)     integer  :: lbc,ubc
   309)     real(r8) :: wt1              ! time interpolation weights
   310)     real(r8), pointer :: wtpfttot1(:)            ! summation of pft weights for renormalization
   311)     real(r8), pointer :: wtpfttot2(:)            ! summation of pft weights for renormalization
   312)     real(r8), parameter :: wtpfttol = 1.e-10     ! tolerance for pft weight renormalization
   313)     type(gridcell_type), pointer :: gptr         ! pointer to gridcell derived subtype
   314)     type(landunit_type), pointer :: lptr         ! pointer to landunit derived subtype
   315)     type(pft_type)     , pointer :: pptr         ! pointer to pft derived subtype
   316)     character(len=32) :: subname='pftdyn_interp' ! subroutine name
   317) !-----------------------------------------------------------------------
   318) 
   319)     call get_proc_bounds(begg,endg,begl,endl,begc,endc,begp,endp)
   320) 
   321)     ! Set pointers into derived type
   322) 
   323)     gptr => grc
   324)     lptr => lun
   325)     pptr => pft
   326) 
   327)     allocate(wtpfttot1(begc:endc),wtpfttot2(begc:endc))
   328)     wtpfttot1(:) = 0._r8
   329)     wtpfttot2(:) = 0._r8
   330) 
   331)     ! Interpolat pctpft to current time step - output in pctpft_intp
   332)     ! Map interpolated pctpft to subgrid weights
   333)     ! assumes that maxpatch_pft = numpft + 1, that each landunit has only 1 column, 
   334)     ! SCAM and CNDV have not been defined, and create_croplandunit = .false.
   335) 
   336)     ! If necessary, obtain new time sample
   337) 
   338)     ! Get current date
   339) 
   340)     call get_curr_date(year, mon, day, sec)
   341) 
   342)     ! Obtain new time sample if necessary.
   343)     ! The first condition is the regular crossing of a year boundary
   344)     ! when within the dynpft timeseries range. The second condition is
   345)     ! the case of the first entry into the dynpft timeseries range from
   346)     ! an earlier period of constant weights.
   347) 
   348)     if (year > yearspft(nt1) .or. (nt1 == 1 .and. nt2 == 1 .and. year == yearspft(1))) then
   349) 
   350)        if (year >= yearspft(ntimes)) then
   351)           nt1 = ntimes
   352)           nt2 = ntimes
   353)        else
   354)           nt1        = nt2
   355)           nt2        = nt2 + 1
   356)           do_harvest = .true.
   357)        end if
   358)        
   359)        if (year > yearspft(ntimes)) then
   360)           do_harvest = .false.
   361)        endif
   362)        
   363)        if (nt2 > ntimes .and. masterproc) then
   364)           write(iulog,*)subname,' error - current year is past input data boundary'
   365)        end if
   366)        
   367)        do m = 0,numpft
   368)           do g = begg,endg
   369)              wtpft1(g,m) = wtpft2(g,m)
   370)           end do
   371)        end do
   372) 
   373)        if (masterproc) then
   374)           write(iulog,*) 'Get PFTDYN data for year: ', yearspft(nt2)
   375)        end if
   376)        call pftdyn_getdata(nt2, wtpft2, begg,endg,0,numpft)
   377)        if (use_cn) then
   378)           call pftdyn_getharvest(nt1,begg,endg)
   379)        end if
   380) 
   381)        do m = 0,numpft
   382)           do g = begg,endg
   383)              wtpft2(g,m) = wtpft2(g,m)/100._r8
   384)           end do
   385)        end do
   386)     
   387)     end if  ! end of need new data if-block 
   388) 
   389)     ! Interpolate pft weight to current time
   390) 
   391)     cday          = get_curr_calday() 
   392)     days_per_year = get_days_per_year()
   393) 
   394)     wt1 = ((days_per_year + 1._r8) - cday)/days_per_year
   395) 
   396)     do p = begp,endp
   397)        c = pptr%column(p)
   398)        g = pptr%gridcell(p)
   399)        l = pptr%landunit(p)
   400)        if (lptr%itype(l) == istsoil .or. lptr%itype(l) == istcrop) then
   401)           m = pptr%itype(p)
   402)           wtcol_old(p)      = pptr%wtcol(p)
   403) !         --- recoded for roundoff performance, tcraig 3/07 from k.lindsay
   404) !         pptr%wtgcell(p)   = wtpft1(g,m)*wt1 + wtpft2(g,m)*wt2
   405)           wtpfttot1(c) = wtpfttot1(c)+pptr%wtgcell(p)    
   406)           pptr%wtgcell(p)   = wtpft2(g,m) + wt1*(wtpft1(g,m)-wtpft2(g,m))
   407)           pptr%wtlunit(p)   = pptr%wtgcell(p) / lptr%wtgcell(l)
   408)           pptr%wtcol(p)     = pptr%wtgcell(p) / lptr%wtgcell(l)
   409)           wtpfttot2(c) = wtpfttot2(c)+pptr%wtgcell(p)
   410)        end if
   411) 
   412)     end do
   413) 
   414) !   Renormalize pft weights so that sum of pft weights relative to grid cell 
   415) !   remain constant even as land cover changes.  Doing this eliminates 
   416) !   soil balance error warnings.  (DML, 4/8/2009)
   417)     do p = begp,endp
   418)        c = pptr%column(p)
   419)        g = pptr%gridcell(p)
   420)        l = pptr%landunit(p)
   421)        if (lptr%itype(l) == istsoil .or. lptr%itype(l) == istcrop) then
   422)           if (wtpfttot2(c) /= 0 .and. &
   423)               abs(wtpfttot1(c)-wtpfttot2(c)) > wtpfttol) then
   424)              pptr%wtgcell(p)   = (wtpfttot1(c)/wtpfttot2(c))*pptr%wtgcell(p)
   425)              pptr%wtlunit(p)   = pptr%wtgcell(p) / lptr%wtgcell(l)
   426)              pptr%wtcol(p)     = pptr%wtgcell(p) / lptr%wtgcell(l)
   427)           end if
   428)        end if
   429) 
   430)     end do
   431)    
   432)     deallocate(wtpfttot1,wtpfttot2) 
   433)     
   434)   end subroutine pftdyn_interp

   441) ! !INTERFACE:
   442)   subroutine pftdyn_getdata(ntime, pctpft, begg, endg, pft0, maxpft)
   443) !
   444) ! !DESCRIPTION:
   445) ! Obtain dynamic landuse data (pctpft) and make sure that
   446) ! percentage of PFTs sum to 100% cover for vegetated landunit
   447) !
   448) ! !USES:
   449)     use clm_varpar  , only : numpft
   450) !
   451) ! !ARGUMENTS:
   452)     implicit none
   453)     integer , intent(in)  :: ntime
   454)     integer , intent(in)  :: begg,endg,pft0,maxpft
   455)     real(r8), intent(out) :: pctpft(begg:endg,pft0:maxpft)
   456) !
   457) !
   458) ! !LOCAL VARIABLES:
   459) !EOP
   460)     integer  :: i,j,m,n
   461)     integer  :: err, ierr, ret
   462)     real(r8) :: sumpct,sumerr                     ! temporary
   463)     real(r8), pointer :: arrayl(:,:)              ! temporary array
   464)     logical  :: readvar
   465)    
   466)     character(len=32) :: subname='pftdyn_getdata' ! subroutine name
   467) !-----------------------------------------------------------------------
   468)     
   469)     allocate(arrayl(begg:endg,pft0:maxpft))	
   470)     call ncd_io(ncid=ncid, varname= 'PCT_PFT', flag='read', data=arrayl, &
   471)          dim1name=grlnd, nt=ntime, readvar=readvar)
   472)     pctpft(begg:endg,pft0:maxpft) = arrayl(begg:endg,pft0:maxpft)
   473)     deallocate(arrayl)		
   474)     if (.not. readvar) call endrun( trim(subname)//' ERROR: PCT_PFT NOT on pftdyn file' )
   475) 
   476)     err = 0
   477)     do n = begg,endg
   478)        if (pctspec(n) < 100._r8) then
   479)           sumpct = 0._r8
   480)           do m = 0, numpft
   481)              sumpct = sumpct + pctpft(n,m) * 100._r8/(100._r8-pctspec(n))
   482)           end do
   483)           if (abs(sumpct - 100._r8) > 0.1_r8) then
   484)              err = 1; ierr = n; sumerr = sumpct
   485)           end if
   486)           if (sumpct < -0.000001_r8) then
   487)              err = 2; ierr = n; sumerr = sumpct
   488)           end if
   489)        end if
   490)     end do
   491)     if (err == 1) then
   492)        write(iulog,*) subname,' error: sum(pct) over numpft+1 is not = 100.',sumerr,ierr,pctspec(ierr),pctpft(ierr,:)
   493)        call endrun()
   494)     else if (err == 2) then

   601) !
   602) ! !INTERFACE:
   603)   subroutine pftdyn_wbal( begg, endg, begc, endc, begp, endp )
   604) !
   605) ! !DESCRIPTION:
   606) ! modify pft-level state and flux variables to maintain water balance with
   607) ! dynamic pft-weights.
   608) ! Canopy water balance does not need to consider harvest fluxes, since pft weights are
   609) ! not affected by harvest.
   610) !
   611) ! !USES:
   612)     use clm_varcon  , only : istsoil
   613)     use clm_varcon  , only : istcrop
   614)     use clm_time_manager, only : get_step_size
   615) !
   616) ! !ARGUMENTS:
   617)     implicit none
   618)     integer, intent(IN)  :: begg     ! beg indices for land gridcells
   619)     integer, intent(IN)  :: endg     ! end indices for land gridcells
   620)     integer, intent(IN)  :: begc     ! beg indices for land columns
   621)     integer, intent(IN)  :: endc     ! end indices for land columns
   622)     integer, intent(IN)  :: begp     ! beg indices for land plant function types
   623)     integer, intent(IN)  :: endp     ! end indices for land plant function types
   624) !
   625) !
   626) ! !LOCAL VARIABLES:
   627) !EOP
   628)     integer  :: pi,p,c,l,g    ! indices
   629)     integer  :: ier           ! error code
   630)     real(r8) :: dtime         ! land model time step (sec)
   631)     real(r8) :: dwt           ! change in pft weight (relative to column)
   632)     real(r8) :: init_h2ocan   ! initial canopy water mass
   633)     real(r8) :: new_h2ocan    ! canopy water mass after weight shift
   634)     real(r8), allocatable :: loss_h2ocan(:) ! canopy water mass loss due to weight shift
   635)     type(landunit_type), pointer :: lptr         ! pointer to landunit derived subtype
   636)     type(column_type),   pointer :: cptr         ! pointer to column derived subtype
   637)     type(pft_type)   ,   pointer :: pptr         ! pointer to pft derived subtype
   638)     character(len=32) :: subname='pftdyn_wbal' ! subroutine name
   639) !-----------------------------------------------------------------------
   640) 
   641)     ! Set pointers into derived type
   642) 
   643)     lptr => lun
   644)     cptr => col
   645)     pptr => pft
   646) 
   647)     ! Allocate loss_h2ocan
   648)     allocate(loss_h2ocan(begp:endp), stat=ier)
   649)     if (ier /= 0) then
   650)           write(iulog,*)subname,' allocation error for loss_h2ocan'; call endrun()
   651)     end if
   652) 
   653)     ! Get time step
   654) 
   655)     dtime = get_step_size()
   656) 
   657)     ! set column-level canopy water mass balance correction flux
   658)     ! term to 0 at the beginning of every weight-shifting timestep
   659) 
   660)     do c = begc,endc
   661)        cwf%h2ocan_loss(c) = 0._r8 ! is this OR pftdyn_wbal_init redundant?
   662)     end do
   663) 
   664)     do p = begp,endp
   665)        l = pptr%landunit(p)
   666)        loss_h2ocan(p) = 0._r8
   667) 
   668)        if (lptr%itype(l) == istsoil .or. lptr%itype(l) == istcrop) then
   669) 
   670)           ! calculate the change in weight for the timestep
   671)           dwt = pptr%wtcol(p)-wtcol_old(p)
   672)   
   673)           if (dwt > 0._r8) then
   674)           
   675)              ! if the pft gained weight, then the 
   676)              ! initial canopy water state is redistributed over the
   677)              ! new (larger) area, conserving mass.
   678) 
   679)              pws%h2ocan(p) = pws%h2ocan(p) * (wtcol_old(p)/pptr%wtcol(p))
   680)           
   681)           else if (dwt < 0._r8) then
   682)           
   683)              ! if the pft lost weight on the timestep, then the canopy water
   684)              ! mass associated with the lost weight is directed to a 
   685)              ! column-level flux term that gets added to the precip flux
   686)              ! for every pft calculation in Hydrology1()
   687)              
   688)              init_h2ocan = pws%h2ocan(p) * wtcol_old(p)
   689)              loss_h2ocan(p) = pws%h2ocan(p) * (-dwt)
   690)              new_h2ocan = init_h2ocan - loss_h2ocan(p)
   691)              if (abs(new_h2ocan) < 1e-8_r8) then
   692)                 new_h2ocan = 0._r8
   693)                 loss_h2ocan(p) = init_h2ocan
   694)              end if
   695)              if (pptr%wtcol(p) /= 0._r8) then  
   696)                 pws%h2ocan(p) = new_h2ocan/pptr%wtcol(p)
   697)              else
   698)                 pws%h2ocan(p) = 0._r8
   699)                 loss_h2ocan(p) = init_h2ocan
   700)              end if 
   701)        
   702) 
   703)           end if
   704) 
   705)        end if
   706)     end do
   707) 
   708)     do pi = 1,max_pft_per_col
   709)        do c = begc,endc
   710)           if (pi <= cptr%npfts(c)) then
   711)              p = cptr%pfti(c) + pi - 1
   712)              cwf%h2ocan_loss(c) = cwf%h2ocan_loss(c) + loss_h2ocan(p)/dtime
   713)           end if
   714)        end do
   715)     end do
   716) 
   717)     ! Deallocate loss_h2ocan
   718)     deallocate(loss_h2ocan)
   719)     
   720)   end subroutine pftdyn_wbal
