    73) ! !ROUTINE: step_therm1 - step pre-coupler thermodynamics
    74) !
    75) ! !DESCRIPTION:
    76) !
    77) ! Wrapper to driver for updating ice and snow internal temperatures and
    78) ! computing thermodynamic growth rates and coupler fluxes. Needed for 
    79) ! introducing OpenMP threading more simply into code.
    80) !
    81) ! !REVISION HISTORY:
    82) !
    83) ! authors: Mariana Vertenstein, NCAR
    84) !
    85) ! !INTERFACE:
    86) 
    87)       subroutine step_therm1(dt)
    88) !
    89) ! !USES:
    90) !
    91) ! !INPUT/OUTPUT PARAMETERS:
    92) !
    93)       real (kind=dbl_kind), intent(in) :: &
    94)          dt      ! time step
    95) !
    96) !EOP
    97) !
    98)       integer (kind=int_kind) :: &
    99)          iblk    ! block index
   100) 
   101)       call ice_timer_start(timer_column)  ! column physics
   102)       call ice_timer_start(timer_thermo)  ! thermodynamics
   103) 
   104)       call init_history_therm    ! initialize thermo history variables
   105)       call init_flux_ocn         ! initialize ocean fluxes sent to coupler
   106) 
   107)       if (oceanmixed_ice) &
   108)            call ocean_mixed_layer (dt)   ! ocean surface fluxes and sst
   109) 
   110) !      call ice_timer_start(timer_tmp)  ! temporary timer
   111) 
   112)       !$OMP PARALLEL DO PRIVATE(iblk)
   113)       do iblk = 1, nblocks
   114)          call step_therm1_iblk(dt, iblk)
   115)       end do
   116)       !$OMP END PARALLEL DO
   117) 
   118) !      call ice_timer_stop(timer_tmp)  ! temporary timer
   119)       call ice_timer_stop(timer_thermo) ! thermodynamics
   120)       call ice_timer_stop(timer_column) ! column physics
   121) 
   122)       end subroutine step_therm1

   127) ! !ROUTINE: step_therm1_iblk - step pre-coupler thermodynamics
   128) !
   129) ! !DESCRIPTION:
   130) !
   131) ! Driver for updating ice and snow internal temperatures and

   133) !
   134) ! !REVISION HISTORY:
   135) !

   137) !
   138) ! !INTERFACE:
   139) 
   140)       subroutine step_therm1_iblk (dt, iblk)
   141) !
   142) ! !USES:
   143) !
   144) ! !INPUT/OUTPUT PARAMETERS:
   145) !
   146)       real (kind=dbl_kind), intent(in) :: &
   147)          dt      ! time step
   148) 
   149)       integer (kind=int_kind), intent(in) :: &
   150)          iblk ! block index
   151) 
   152) !
   153) !EOP
   154) !
   155)       integer (kind=int_kind) :: &
   156)          i, j, ij    , & ! horizontal indices
   157)          ilo,ihi,jlo,jhi, & ! beginning and end of physical domain
   158)          n           , & ! thickness category index
   159)          il1, il2    , & ! ice layer indices for eice
   160)          sl1, sl2        ! snow layer indices for esno
   161) 
   162)       integer (kind=int_kind) :: &  
   163)          icells          ! number of cells with aicen > puny
   164) 
   165)       integer (kind=int_kind), dimension(nx_block*ny_block) :: &
   166)          indxi, indxj    ! indirect indices for cells with aicen > puny

   168) ! 2D coupler variables (computed for each category, then aggregated)
   169) 
   170)       real (kind=dbl_kind), dimension (nx_block,ny_block) :: &
   171)          fsensn      , & ! surface downward sensible heat     (W/m^2)
   172)          fswabsn     , & ! shortwave absorbed by ice          (W/m^2)
   173)          flwoutn     , & ! upward LW at surface               (W/m^2)
   174)          evapn       , & ! flux of vapor, atmos to ice   (kg m-2 s-1)
   175)          freshn      , & ! flux of water, ice to ocean     (kg/m^2/s)
   176)          fsaltn      , & ! flux of salt, ice to ocean      (kg/m^2/s)
   177)          fhocnn      , & ! fbot corrected for leftover energy (W/m^2)
   178)          strairxn    , & ! air/ice zonal  strss,              (N/m^2)
   179)          strairyn    , & ! air/ice merdnl strss,              (N/m^2)
   180)          Urefn       , & ! wind speed reference level         (m/s)
   181)          Trefn       , & ! air tmp reference level            (K)
   182)          Qrefn           ! air sp hum reference level         (kg/kg)
   183) 
   184)       ! other local variables
   185)       real (kind=dbl_kind), dimension (nx_block,ny_block) :: &
   186)          Tbot        , & ! ice bottom surface temperature (deg C)
   187)          fbot        , & ! ice-ocean heat flux at bottom surface (W/m^2)
   188)          shcoef      , & ! transfer coefficient for sensible heat
   189)          lhcoef          ! transfer coefficient for latent heat
   190) 
   191)       real (kind=dbl_kind), dimension (nx_block,ny_block) :: &
   192)          melttn      , & ! top melt in category n (m)
   193)          meltbn      , & ! bottom melt in category n (m)
   194)          meltsn      , & ! snow melt in category n (m)
   195)          congeln     , & ! congelation ice formation in category n (m)
   196)          snoicen         ! snow-ice formation in category n (m)
   197) 
   198)       ! Local variables to keep track of mass budget for aerosols MH
   199)       real (kind=dbl_kind), dimension (nx_block,ny_block) :: &
   200)          vsno_old   
   201) 
   202)       type (block) :: &
   203)          this_block      ! block information for current block
   204) 
   205)       logical (kind=log_kind) :: &
   206)          l_stop          ! if true, abort the model
   207) 
   208)       integer (kind=int_kind) :: &
   209)          istop, jstop    ! indices of grid cell where model aborts 
   210) 
   211)       ! local (single block)
   212)       real (kind=dbl_kind), dimension (nx_block,ny_block) :: &
   213)          worka, workb
   214) 
   215)       l_stop = .false.
   216) 
   217)       worka(:,:) = c0
   218)       workb(:,:) = c0
   219) 
   220)       this_block = get_block(blocks_ice(iblk),iblk)         
   221)       ilo = this_block%ilo
   222)       ihi = this_block%ihi
   223)       jlo = this_block%jlo
   224)       jhi = this_block%jhi
   225) 
   226)       !-----------------------------------------------------------------
   227)       ! Save the ice area passed to the coupler (so that history fields
   228)       !  can be made consistent with coupler fields).
   229)       ! Save the initial ice area and volume in each category.
   230)       !-----------------------------------------------------------------
   231) 
   232)       do j = 1, ny_block
   233)       do i = 1, nx_block
   234)          aice_init (i,j,  iblk) = aice (i,j,  iblk)
   235)       enddo
   236)       enddo
   237) 
   238)       do n = 1, ncat
   239)       do j = 1, ny_block
   240)       do i = 1, nx_block
   241)          aicen_init(i,j,n,iblk) = aicen(i,j,n,iblk)
   242)          vicen_init(i,j,n,iblk) = vicen(i,j,n,iblk)
   243)       enddo
   244)       enddo
   245)       enddo
   246) 
   247)       if (mod(time-dt_thm,dt_dyn) == c0) then
   248)          strairxT_accum(:,:,iblk) = c0
   249)          strairyT_accum(:,:,iblk) = c0
   250)       endif
   251) 
   252)       !-----------------------------------------------------------------
   253)       ! Adjust frzmlt to account for ice-ocean heat fluxes since last

   255)       ! Compute lateral and bottom heat fluxes.
   256)       !-----------------------------------------------------------------
   257) 
   258)       call frzmlt_bottom_lateral                                      &
   259)                              (nx_block,           ny_block,           &
   260)                               ilo, ihi,           jlo, jhi,           &
   261)                               dt,                                     &
   262)                               aice  (:,:,  iblk), frzmlt(:,:,  iblk), &
   263)                               eicen (:,:,:,iblk), esnon (:,:,:,iblk), &
   264)                               sst   (:,:,  iblk), Tf    (:,:,  iblk), &
   265)                               strocnxT(:,:,iblk), strocnyT(:,:,iblk), &

   267)                               rside (:,:,  iblk) )
   268) 
   269) 
   270)       do n = 1, ncat
   271) 
   272)       !-----------------------------------------------------------------
   273)       ! Identify cells with nonzero ice area
   274)       !-----------------------------------------------------------------
   275)            
   276)          icells = 0
   277)          do j = jlo, jhi
   278)          do i = ilo, ihi
   279)             if (aicen(i,j,n,iblk) > puny) then
   280)                icells = icells + 1
   281)                indxi(icells) = i
   282)                indxj(icells) = j
   283)             endif
   284)          enddo               ! i
   285)          enddo               ! j
   286) 
   287)          if (calc_Tsfc .or. calc_strair .and. icells > 0) then
   288) 
   289)       !-----------------------------------------------------------------
   290)       ! Atmosphere boundary layer calculation; compute coefficients
   291)       ! for sensible and latent heat fluxes.
   292)       !
   293)       ! NOTE: The wind stress is computed here for later use if 
   294)       !       calc_strair = .true.   Otherwise, the wind stress
   295)       !       components are set to the data values.
   296)       !-----------------------------------------------------------------
   297) 
   298)             if (trim(atmbndy) == 'constant') then
   299)                call atmo_boundary_const(nx_block,      ny_block,        &
   300)                                         'ice',          icells,         &
   301)                                         indxi,          indxj,          &
   302)                                         uatm(:,:,iblk), vatm(:,:,iblk), &
   303)                                         wind(:,:,iblk), rhoa(:,:,iblk), &
   304)                                         strairxn,       strairyn,       &
   305)                                         lhcoef,         shcoef)
   306)             else ! default
   307)                call atmo_boundary_layer(nx_block,       ny_block,       &
   308)                                         'ice',          icells,         &
   309)                                         indxi,          indxj,          &
   310)                                         trcrn(:,:,nt_Tsfc,n,iblk),      &
   311)                                         potT(:,:,iblk),                 &
   312)                                         uatm(:,:,iblk), vatm(:,:,iblk), &
   313)                                         uvel(:,:,iblk), vvel(:,:,iblk), &
   314)                                         wind(:,:,iblk), zlvl(:,:,iblk), &
   315)                                         Qa  (:,:,iblk), rhoa(:,:,iblk), &
   316)                                         strairxn,       strairyn,       &
   317)                                         Urefn,                          &
   318)                                         Trefn,          Qrefn,          &
   319)                                         worka,          workb,          &
   320)                                         lhcoef,         shcoef)
   321)             endif ! atmbndy
   322) 
   323)          else
   324) 
   325)             ! Initialize for safety
   326)             Urefn (:,:)  = c0
   327)             Trefn (:,:)  = c0
   328)             Qrefn (:,:)  = c0
   329)             lhcoef(:,:)  = c0
   330)             shcoef(:,:)  = c0
   331) 
   332)          endif   ! calc_Tsfc or calc_strair
   333) 
   334)          if (.not.(calc_strair)) then
   335)             strairxn(:,:) = strax(:,:,iblk)
   336)             strairyn(:,:) = stray(:,:,iblk)
   337)          endif
   338) 
   339)       !-----------------------------------------------------------------
   340)       ! Update ice age
   341)       ! This is further adjusted for freezing in the thermodynamics.
   342)       ! Melting does not alter the ice age.
   343)       !-----------------------------------------------------------------
   344) 
   345)          if (tr_iage) then
   346)             call increment_age (nx_block, ny_block,      &
   347)                                 dt, icells,              &
   348)                                 indxi, indxj,            &
   349)                                 trcrn(:,:,nt_iage,n,iblk))
   350)          endif
   351)          if (tr_FY) then
   352)             call update_FYarea (nx_block, ny_block,      &
   353)                                 dt, icells,              &
   354)                                 indxi, indxj,            &
   355)                                 lmask_n(:,:,iblk),       &
   356)                                 lmask_s(:,:,iblk),       &
   357)                                 trcrn(:,:,nt_FY,n,iblk))
   358)          endif
   359) 
   360)       !-----------------------------------------------------------------
   361)       ! Vertical thermodynamics: Heat conduction, growth and melting.
   362)       !----------------------------------------------------------------- 
   363) 
   364)          il1 = ilyr1(n)
   365)          il2 = ilyrn(n)
   366)          sl1 = slyr1(n)
   367)          sl2 = slyrn(n)
   368) 
   369)          vsno_old   = vsnon(:,:,n,iblk) !MH
   370) 
   371)          call thermo_vertical                                       &
   372)                          (nx_block,            ny_block,            &
   373)                           dt,                  icells,              &
   374)                           indxi,               indxj,               &
   375)                           aicen(:,:,n,iblk),                        &
   376)                           trcrn(:,:,:,n,iblk),                      &
   377)                           vicen(:,:,n,iblk),   vsnon(:,:,n,iblk),   &
   378)                           eicen  (:,:,il1:il2,iblk),                &
   379)                           esnon  (:,:,sl1:sl2,iblk),                &
   380)                           flw    (:,:,iblk),   potT (:,:,iblk),     &
   381)                           Qa     (:,:,iblk),   rhoa (:,:,iblk),     &
   382)                           fsnow  (:,:,iblk),                        &
   383)                           fbot,                Tbot,                &
   384)                           lhcoef,              shcoef,              &
   385)                           fswsfcn(:,:,n,iblk), fswintn(:,:,n,iblk), &
   386)                           fswthrun(:,:,n,iblk),                     &
   387)                           Sswabsn(:,:,sl1:sl2,iblk),                &
   388)                           Iswabsn(:,:,il1:il2,iblk),                &
   389)                           fsurfn(:,:,n,iblk),  fcondtopn(:,:,n,iblk),&
   390)                           fsensn,              flatn(:,:,n,iblk),   &
   391)                           fswabsn,             flwoutn,             &
   392)                           evapn,               freshn,              &
   393)                           fsaltn,              fhocnn,              &
   394)                           melttn,              meltsn,              &
   395)                           meltbn,                                   &
   396)                           congeln,             snoicen,             &
   397)                           mlt_onset(:,:,iblk), frz_onset(:,:,iblk), &
   398)                           yday,                l_stop,              &
   399)                           istop,               jstop)
   400) 
   401)          if (l_stop) then
   402)             write (nu_diag,*) 'istep1, my_task, iblk =', &
   403)                                istep1, my_task, iblk
   404)             write (nu_diag,*) 'Global block:', this_block%block_id
   405)             if (istop > 0 .and. jstop > 0) &
   406)                  write(nu_diag,*) 'Global i and j:', &
   407)                                   this_block%i_glob(istop), &
   408)                                   this_block%j_glob(jstop) 
   409)                  write(nu_diag,*) 'Lat, Lon:', &
   410)                                  TLAT(istop,jstop,iblk)*rad_to_deg, &
   411)                                  TLON(istop,jstop,iblk)*rad_to_deg
   412)             call abort_ice ('ice: Vertical thermo error')
   413)          endif
   414) 
   415)       !-----------------------------------------------------------------
   416)       ! Aerosol update  MH
   417)       !-----------------------------------------------------------------
   418)          if (tr_aero) then
   419) 
   420)             if (icells > 0) then
   421) 
   422)                call update_aerosol (nx_block, ny_block,                  &
   423)                                     dt, icells,                          &
   424)                                     indxi, indxj,                        &
   425)                                     melttn, meltsn,                      &
   426)                                     meltbn, congeln, snoicen,            &
   427)                                     fsnow(:,:,iblk),                     &
   428)                                     trcrn(:,:,:,n,iblk),                 &
   429)                                     aicen_init(:,:,n,iblk),              &
   430)                                     vicen_init(:,:,n,iblk),              &
   431)                                     vsno_old,                            &
   432)                                     vicen(:,:,n,iblk),                   &
   433)                                     vsnon(:,:,n,iblk),                   &
   434)                                     aicen(:,:,n,iblk),faero(:,:,:,iblk), &
   435)                                     fsoot(:,:,:,iblk))
   436) 
   437)             endif
   438) 
   439)          endif
   440) 
   441)       !-----------------------------------------------------------------
   442)       ! Melt ponds
   443)       !-----------------------------------------------------------------
   444) 
   445)          if (tr_pond) then
   446) 
   447)             call compute_ponds(nx_block, ny_block,                      &
   448)                                ilo, ihi, jlo, jhi,                      &
   449)                                melttn, meltsn, frain(:,:,iblk),         &
   450)                                aicen (:,:,n,iblk), vicen (:,:,n,iblk),  &
   451)                                vsnon (:,:,n,iblk), trcrn (:,:,:,n,iblk),&

   455)

   460)          call merge_fluxes (nx_block,           ny_block,             &
   461)                             icells,                                   &
   462)                             indxi,              indxj,                &
   463)                             aicen_init(:,:,n,iblk),                   &
   464)                             flw(:,:,iblk),      coszen(:,:,iblk),     &
   465)                             strairxn,           strairyn,             &
   466)                             fsurfn(:,:,n,iblk), fcondtopn(:,:,n,iblk),&
   467)                             fsensn,             flatn(:,:,n,iblk),    &
   468)                             fswabsn,            flwoutn,              &
   469) #if (defined AEROFRC) || (defined PONDFRC) || (defined CCSM3FRC)

   471) #endif
   472) #ifdef AEROFRC
   473)                             dfswabsn_noaero(:,:,n,iblk),               &
   474)                             dfswsfcn_noaero(:,:,n,iblk),               &
   475)                             dfswintn_noaero(:,:,n,iblk),               &
   476)                             dfswthrun_noaero(:,:,n,iblk),              &
   477) #endif
   478) #ifdef CCSM3FRC
   479)                             dfswabsn_ccsm3(:,:,n,iblk),               &
   480)                             dfswsfcn_ccsm3(:,:,n,iblk),               &
   481)                             dfswintn_ccsm3(:,:,n,iblk),               &
   482)                             dfswthrun_ccsm3(:,:,n,iblk),              &
   483) #endif
   484) #ifdef PONDFRC
   485)                             dfswabsn_nopond(:,:,n,iblk),               &
   486)                             dfswsfcn_nopond(:,:,n,iblk),               &
   487)                             dfswintn_nopond(:,:,n,iblk),               &
   488)                             dfswthrun_nopond(:,:,n,iblk),              &
   489) #endif
   490)                             evapn,                                    &
   491)                             Urefn,                                    &
   492)                             Trefn,              Qrefn,                &
   493)                             freshn,             fsaltn,               &
   494)                             fhocnn,             fswthrun(:,:,n,iblk), &
   495)                             strairxT(:,:,iblk), strairyT  (:,:,iblk), &
   496)                             fsurf   (:,:,iblk), fcondtop  (:,:,iblk), &
   497)                             fsens   (:,:,iblk), flat      (:,:,iblk), &
   498)                             fswabs  (:,:,iblk), flwout    (:,:,iblk), &
   499) #if (defined AEROFRC) || (defined PONDFRC) || (defined CCSM3FRC)
   500)                             fswsfc  (:,:,iblk), fswint    (:,:,iblk), &
   501) #endif
   502) #ifdef AEROFRC
   503)                             dfswabs_noaero(:,:,iblk),dfswsfc_noaero(:,:,iblk), &
   504)                             dfswint_noaero(:,:,iblk),dfswthru_noaero(:,:,iblk),&
   505) #endif
   506) #ifdef CCSM3FRC
   507)                             dfswabs_ccsm3(:,:,iblk),dfswsfc_ccsm3(:,:,iblk), &
   508)                             dfswint_ccsm3(:,:,iblk),dfswthru_ccsm3(:,:,iblk),&
   509) #endif
   510) #ifdef PONDFRC
   511)                             dfswabs_nopond(:,:,iblk),dfswsfc_nopond(:,:,iblk), &
   512)                             dfswint_nopond(:,:,iblk),dfswthru_nopond(:,:,iblk),&
   513) #endif
   514)                             evap    (:,:,iblk),                       &
   515)                             Uref    (:,:,iblk),                       &
   516)                             Tref    (:,:,iblk), Qref      (:,:,iblk), &
   517)                             fresh   (:,:,iblk), fsalt   (:,:,iblk),   &
   518)                             fhocn   (:,:,iblk), fswthru (:,:,iblk),   &
   519)                             melttn, meltsn, meltbn, congeln, snoicen, &
   520)                             meltt   (:,:,iblk),  melts   (:,:,iblk),  &
   521)                             meltb   (:,:,iblk),                       &
   522)                             congel  (:,:,iblk),  snoice  (:,:,iblk))
   523) 
   524)       enddo                  ! ncat
   525) 
   526) 
   527) ! Accumulate stresses when super-cycling the dynamics. Otherwise just
   528) ! use the stresses as computed.
   529)
