   280) !----------------------------------------------------------------------------
   281)       subroutine taugb1
   282) !----------------------------------------------------------------------------
   283) 
   284) ! ------- Modifications -------
   285) !  Written by Eli J. Mlawer, Atmospheric & Environmental Research.
   286) !  Revised by Michael J. Iacono, Atmospheric & Environmental Research.
   287) !
   288) !     band 1:  10-350 cm-1 (low key - h2o; low minor - n2)
   289) !                          (high key - h2o; high minor - n2)
   290) !
   291) !     note: previous versions of rrtm band 1: 
   292) !           10-250 cm-1 (low - h2o; high - h2o)
   293) !----------------------------------------------------------------------------
   294) 
   295) ! ------- Modules -------
   296) 
   297)       use parrrtm, only : ng1
   298)       use rrlw_kg01, only : fracrefa, fracrefb, absa, absb, &
   299)                             ka_mn2, kb_mn2, selfref, forref
   300) 
   301) ! ------- Declarations -------
   302) 
   303) ! Local 
   304)       integer :: lay, ind0, ind1, inds, indf, indm, ig
   305)       real(kind=r8) :: pp, corradj, scalen2, tauself, taufor, taun2
   306) 
   307) 
   308) ! Minor gas mapping levels:
   309) !     lower - n2, p = 142.5490 mbar, t = 215.70 k
   310) !     upper - n2, p = 142.5490 mbar, t = 215.70 k
   311) 
   312) ! Compute the optical depth by interpolating in ln(pressure) and 
   313) ! temperature.  Below laytrop, the water vapor self-continuum and
   314) ! foreign continuum is interpolated (in temperature) separately.
   315) 
   316) ! Lower atmosphere loop
   317)       do lay = 1, laytrop
   318) 
   319)          ind0 = ((jp(lay)-1)*5+(jt(lay)-1))*nspa(1) + 1
   320)          ind1 = (jp(lay)*5+(jt1(lay)-1))*nspa(1) + 1
   321)          inds = indself(lay)
   322)          indf = indfor(lay)
   323)          indm = indminor(lay)
   324)          pp = pavel(lay)
   325)          corradj =  1.
   326)          if (pp .lt. 250._r8) then
   327)             corradj = 1._r8 - 0.15_r8 * (250._r8-pp) / 154.4_r8
   328)          endif
   329) 
   330)          scalen2 = colbrd(lay) * scaleminorn2(lay)
   331)          do ig = 1, ng1
   332)             tauself = selffac(lay) * (selfref(inds,ig) + selffrac(lay) * &
   333)                  (selfref(inds+1,ig) - selfref(inds,ig)))
   334)             taufor =  forfac(lay) * (forref(indf,ig) + forfrac(lay) * &
   335)                  (forref(indf+1,ig) -  forref(indf,ig))) 
   336)             taun2 = scalen2*(ka_mn2(indm,ig) + & 
   337)                  minorfrac(lay) * (ka_mn2(indm+1,ig) - ka_mn2(indm,ig)))
   338)             taug(lay,ig) = corradj * (colh2o(lay) * &
   339)                 (fac00(lay) * absa(ind0,ig) + &
   340)                  fac10(lay) * absa(ind0+1,ig) + &
   341)                  fac01(lay) * absa(ind1,ig) + &
   342)                  fac11(lay) * absa(ind1+1,ig)) & 
   343)                  + tauself + taufor + taun2)
   344)              fracs(lay,ig) = fracrefa(ig)
   345)          enddo
   346)       enddo
   347) 
   348) ! Upper atmosphere loop
   349)       do lay = laytrop+1, nlayers
   350) 
   351)          ind0 = ((jp(lay)-13)*5+(jt(lay)-1))*nspb(1) + 1
   352)          ind1 = ((jp(lay)-12)*5+(jt1(lay)-1))*nspb(1) + 1
   353)          indf = indfor(lay)
   354)          indm = indminor(lay)
   355)          pp = pavel(lay)
   356)          corradj =  1._r8 - 0.15_r8 * (pp / 95.6_r8)
   357) 
   358)          scalen2 = colbrd(lay) * scaleminorn2(lay)
   359)          do ig = 1, ng1
   360)             taufor = forfac(lay) * (forref(indf,ig) + &
   361)                  forfrac(lay) * (forref(indf+1,ig) - forref(indf,ig))) 
   362)             taun2 = scalen2*(kb_mn2(indm,ig) + & 
   363)                  minorfrac(lay) * (kb_mn2(indm+1,ig) - kb_mn2(indm,ig)))
   364)             taug(lay,ig) = corradj * (colh2o(lay) * &
   365)                 (fac00(lay) * absb(ind0,ig) + &
   366)                  fac10(lay) * absb(ind0+1,ig) + &
   367)                  fac01(lay) * absb(ind1,ig) + &
   368)                  fac11(lay) * absb(ind1+1,ig)) &  
   369)                  + taufor + taun2)
   370)             fracs(lay,ig) = fracrefb(ig)
   371)          enddo
   372)       enddo
   373) 
   374)       end subroutine taugb1

   377)       subroutine taugb2
   378) !----------------------------------------------------------------------------
   379) !
   380) !     band 2:  350-500 cm-1 (low key - h2o; high key - h2o)
   381) !
   382) !     note: previous version of rrtm band 2: 
   383) !           250 - 500 cm-1 (low - h2o; high - h2o)
   384) !----------------------------------------------------------------------------
   385) 
   386) ! ------- Modules -------
   387) 
   388)       use parrrtm, only : ng2, ngs1
   389)       use rrlw_kg02, only : fracrefa, fracrefb, absa, absb, &
   390)                             selfref, forref
   391) 
   392) ! ------- Declarations -------
   393) 
   394) ! Local 
   395)       integer :: lay, ind0, ind1, inds, indf, ig
   396)       real(kind=r8) :: pp, corradj, tauself, taufor
   397) 
   398) 
   399) ! Compute the optical depth by interpolating in ln(pressure) and 
   400) ! temperature.  Below laytrop, the water vapor self-continuum and
   401) ! foreign continuum is interpolated (in temperature) separately.
   402) 
   403) ! Lower atmosphere loop
   404)       do lay = 1, laytrop
   405) 
   406)          ind0 = ((jp(lay)-1)*5+(jt(lay)-1))*nspa(2) + 1
   407)          ind1 = (jp(lay)*5+(jt1(lay)-1))*nspa(2) + 1
   408)          inds = indself(lay)
   409)          indf = indfor(lay)
   410)          pp = pavel(lay)
   411)          corradj = 1._r8 - .05_r8 * (pp - 100._r8) / 900._r8
   412)          do ig = 1, ng2
   413)             tauself = selffac(lay) * (selfref(inds,ig) + selffrac(lay) * &
   414)                  (selfref(inds+1,ig) - selfref(inds,ig)))
   415)             taufor =  forfac(lay) * (forref(indf,ig) + forfrac(lay) * &
   416)                  (forref(indf+1,ig) - forref(indf,ig))) 
   417)             taug(lay,ngs1+ig) = corradj * (colh2o(lay) * &
   418)                 (fac00(lay) * absa(ind0,ig) + &
   419)                  fac10(lay) * absa(ind0+1,ig) + &
   420)                  fac01(lay) * absa(ind1,ig) + &
   421)                  fac11(lay) * absa(ind1+1,ig)) &
   422)                  + tauself + taufor)
   423)             fracs(lay,ngs1+ig) = fracrefa(ig)
   424)          enddo
   425)       enddo
   426) 
   427) ! Upper atmosphere loop
   428)       do lay = laytrop+1, nlayers
   429) 
   430)          ind0 = ((jp(lay)-13)*5+(jt(lay)-1))*nspb(2) + 1
   431)          ind1 = ((jp(lay)-12)*5+(jt1(lay)-1))*nspb(2) + 1
   432)          indf = indfor(lay)
   433)          do ig = 1, ng2
   434)             taufor =  forfac(lay) * (forref(indf,ig) + &
   435)                  forfrac(lay) * (forref(indf+1,ig) - forref(indf,ig)))
