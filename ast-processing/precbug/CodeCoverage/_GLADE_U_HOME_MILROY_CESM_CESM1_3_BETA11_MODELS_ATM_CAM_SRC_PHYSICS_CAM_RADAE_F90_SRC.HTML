   227) ! Purpose: 
   228) ! Compute absorptivities for h2o, co2, o3, ch4, n2o, cfc11 and cfc12
   229) ! 
   230) ! Method: 
   231) ! h2o  ....  Uses nonisothermal emissivity method for water vapor from
   232) !            Ramanathan, V. and  P.Downey, 1986: A Nonisothermal
   233) !            Emissivity and Absorptivity Formulation for Water Vapor
   234) !            Journal of Geophysical Research, vol. 91., D8, pp 8649-8666
   235) !
   236) !            Implementation updated by Collins, Hackney, and Edwards (2001)
   237) !               using line-by-line calculations based upon Hitran 1996 and
   238) !               CKD 2.1 for absorptivity and emissivity
   239) !
   240) !            Implementation updated by Collins, Lee-Taylor, and Edwards (2003)
   241) !               using line-by-line calculations based upon Hitran 2000 and
   242) !               CKD 2.4 for absorptivity and emissivity
   243) !
   244) ! co2  ....  Uses absorptance parameterization of the 15 micro-meter
   245) !            (500 - 800 cm-1) band system of Carbon Dioxide, from
   246) !            Kiehl, J.T. and B.P.Briegleb, 1991: A New Parameterization
   247) !            of the Absorptance Due to the 15 micro-meter Band System
   248) !            of Carbon Dioxide Jouranl of Geophysical Research,
   249) !            vol. 96., D5, pp 9013-9019.
   250) !            Parameterizations for the 9.4 and 10.4 mircon bands of CO2
   251) !            are also included.
   252) !
   253) ! o3   ....  Uses absorptance parameterization of the 9.6 micro-meter
   254) !            band system of ozone, from Ramanathan, V. and R.Dickinson,
   255) !            1979: The Role of stratospheric ozone in the zonal and
   256) !            seasonal radiative energy balance of the earth-troposphere
   257) !            system. Journal of the Atmospheric Sciences, Vol. 36,
   258) !            pp 1084-1104
   259) !
   260) ! ch4  ....  Uses a broad band model for the 7.7 micron band of methane.
   261) !
   262) ! n20  ....  Uses a broad band model for the 7.8, 8.6 and 17.0 micron
   263) !            bands of nitrous oxide
   264) !
   265) ! cfc11 ...  Uses a quasi-linear model for the 9.2, 10.7, 11.8 and 12.5
   266) !            micron bands of CFC11
   267) !
   268) ! cfc12 ...  Uses a quasi-linear model for the 8.6, 9.1, 10.8 and 11.2
   269) !            micron bands of CFC12
   270) !
   271) !
   272) ! Computes individual absorptivities for non-adjacent layers, accounting
   273) ! for band overlap, and sums to obtain the total; then, computes the
   274) ! nearest layer contribution.
   275) ! 
   276) ! Author: W. Collins (H2O absorptivity) and J. Kiehl
   277) !------------------------------Arguments--------------------------------
   278) !
   279) ! Input arguments
   280) !
   281)    integer, intent(in) :: lchnk                       ! chunk identifier

   283) 
   284)    real(r8), intent(in) :: pbr(pcols,pver)            ! Prssr at mid-levels (dynes/cm2)

   287)    real(r8), intent(in) :: co2eml(pcols,pver)         ! Co2 emissivity function
   288)    real(r8), intent(in) :: tplnka(pcols,pverp)        ! Planck fnctn level temperature

   292)    real(r8), intent(in) :: h2otr(pcols,pverp)         ! H2o trnsmssn fnct for o3 overlap

   297)    real(r8), intent(in) :: tlayr(pcols,pverp)         ! K-1 level temperatures
   298)    real(r8), intent(in) :: plol(pcols,pverp)          ! Ozone prs wghted path length
   299)    real(r8), intent(in) :: plos(pcols,pverp)          ! Ozone path length
   300)    real(r8), intent(in) :: pmln(pcols,pver)           ! Ln(pmidm1)

   325)    real(r8), intent(in) :: uco212(pcols,pverp)        ! CO2 9.4 micron band path length
   326)    real(r8), intent(in) :: uco213(pcols,pverp)        ! CO2 9.4 micron band path length

   337) ! Output arguments

  1580)             tmp2     = log(1._r8 + f2co2(i))
  1581)             tmp3     = log(1._r8 + f3co2(i))

  2491)          posqt  = pi/(2._r8*sqti)
  2492)          rbeta7 =  1._r8/( 5.3288_r8*posqt)
  2493)          rbeta8 = 1._r8/ (10.6576_r8*posqt)
  2494)          rbeta9 = rbeta7
  2495)          rbeta13= rbeta9
  2496)          f2co2  = (u7/sqrt(4._r8 + u7*(1._r8 + rbeta7))) + &
  2497)                   (u8/sqrt(4._r8 + u8*(1._r8 + rbeta8))) + &
  2498)                   (u9/sqrt(4._r8 + u9*(1._r8 + rbeta9)))
  2499)          f3co2  = u13/sqrt(4._r8 + u13*(1._r8 + rbeta13))
  2500)          tmp1   = log(1._r8 + f1sqwp)
  2501)          tmp2   = log(1._r8 +  f2co2)
  2502)          tmp3   = log(1._r8 +  f3co2)
  2503)          absbnd = (tmp1 + 2._r8*t1co2*tmp2 + 2._r8*tmp3)*sqti
  2504)          tco2(i)=1.0_r8/(1.0_r8+10.0_r8*(u7/sqrt(4._r8 + u7*(1._r8 + rbeta7))))
  2505)          co2ems(i,k1)  = troco2(i,k1)*absbnd*co2plk(i)
  2506)          ex     = exp(960._r8/tint(i,k1))
  2507)          exm1sq = (ex - 1._r8)**2
  2508)          co2em(i,k1) = 1.2e11_r8*ex/(tint(i,k1)*tint4(i,k1)*exm1sq)
  2509)       end do
  2510) !
  2511) ! O3 emissivity
  2512) !
  2513)       do i=1,ncol
  2514)          h2otr(i,k1) = exp(-12._r8*s2c(i,k1))
  2515)           h2otr(i,k1)=h2otr(i,k1)*aer_trn_ttl(i,k1,1,idx_LW_1000_1200)
  2516)          te          = (co2t(i,k1)/293._r8)**.7_r8
  2517)          u1          = 18.29_r8*plos(i,k1)/te
  2518)          u2          = .5649_r8*plos(i,k1)/te
  2519)          phat        = plos(i,k1)/plol(i,k1)
  2520)          tlocal      = tplnke(i)
  2521)          tcrfac      = sqrt(tlocal*r250)*te
  2522)          beta        = (1._r8/.3205_r8)*((1._r8/phat) + (dpfo3*tcrfac))
  2523)          realnu      = (1._r8/beta)*te
  2524)          o3bndi      = 74._r8*te*(tplnke(i)/375._r8)*log(1._r8 + fo3(u1,realnu) + fo3(u2,realnu))
  2525)          o3ems(i,k1) = dbvtt(i)*h2otr(i,k1)*o3bndi
  2526)          to3(i)=1.0_r8/(1._r8 + 0.1_r8*fo3(u1,realnu) + 0.1_r8*fo3(u2,realnu))
  2527)       end do
  2528) !
  2529) !   Calculate trace gas emissivities
  2530) !
  2531)       call trcems(ncol    ,                                     &
  2532)                   k1      ,co2t    ,pnm     ,ucfc11  ,ucfc12  , &
  2533)                   un2o0   ,un2o1   ,bn2o0   ,bn2o1   ,uch4    , &
  2534)                   bch4    ,uco211  ,uco212  ,uco213  ,uco221  , &
  2535)                   uco222  ,uco223  ,uptype  ,w       ,s2c     , &
  2536)                   u       ,emplnk  ,th2o    ,tco2    ,to3     , &
  2537)                   emstrc  , &
  2538)                   aer_trn_ttl)
  2539) !
  2540) ! Total emissivity:
  2541) !
  2542)       do i=1,ncol
  2543)          emstot(i,k1) = h2oems(i,k1) + co2ems(i,k1) + o3ems(i,k1)  &
  2544)                         + emstrc(i,k1)
  2545)       end do
  2546)    end do ! End of interface loop
  2547) 
  2548) end subroutine radems

  2552) subroutine radtpl(ncol    ,                                     &
  2553)                   tnm     ,lwupcgs ,qnm     ,pnm     ,plco2   ,plh2o   , &
  2554)                   tplnka  ,s2c     ,tcg     ,w       ,tplnke  , &
  2555)                   tint    ,tint4   ,tlayr   ,tlayr4  ,pmln    , &
  2556)                   piln    ,plh2ob  ,wb      ,co2mmr)
  2557) !--------------------------------------------------------------------
  2558) !
  2559) ! Purpose:
  2560) ! Compute temperatures and path lengths for longwave radiation
  2561) !
  2562) ! Method:
  2563) ! <Describe the algorithm(s) used in the routine.>
  2564) ! <Also include any applicable external references.>
  2565) !
  2566) ! Author: CCM1
  2567) !------------------------------Arguments-----------------------------
  2568) !
  2569) ! Input arguments
  2570) !
  2571)    integer, intent(in) :: ncol                  ! number of atmospheric columns
  2572) 
  2573)    real(r8), intent(in) :: tnm(pcols,pver)      ! Model level temperatures
  2574)    real(r8), intent(in) :: lwupcgs(pcols)       ! Surface longwave up flux
  2575)    real(r8), intent(in) :: qnm(pcols,pver)      ! Model level specific humidity
  2576)    real(r8), intent(in) :: pnm(pcols,pverp)     ! Pressure at model interfaces (dynes/cm2)
  2577)    real(r8), intent(in) :: pmln(pcols,pver)     ! Ln(pmidm1)
  2578)    real(r8), intent(in) :: piln(pcols,pverp)    ! Ln(pintm1)
  2579)    real(r8), intent(in) :: co2mmr(pcols)        ! co2 column mean mass mixing ratio
  2580) !
  2581) ! Output arguments
  2582) !
  2583)    real(r8), intent(out) :: plco2(pcols,pverp)   ! Pressure weighted co2 path
  2584)    real(r8), intent(out) :: plh2o(pcols,pverp)   ! Pressure weighted h2o path
  2585)    real(r8), intent(out) :: tplnka(pcols,pverp)  ! Level temperature from interface temperatures
  2586)    real(r8), intent(out) :: s2c(pcols,pverp)     ! H2o continuum path length
  2587)    real(r8), intent(out) :: tcg(pcols,pverp)     ! H2o-mass-wgted temp. (Curtis-Godson approx.)
  2588)    real(r8), intent(out) :: w(pcols,pverp)       ! H2o path length
  2589)    real(r8), intent(out) :: tplnke(pcols)        ! Equal to tplnka
  2590)    real(r8), intent(out) :: tint(pcols,pverp)    ! Layer interface temperature
  2591)    real(r8), intent(out) :: tint4(pcols,pverp)   ! Tint to the 4th power
  2592)    real(r8), intent(out) :: tlayr(pcols,pverp)   ! K-1 level temperature
  2593)    real(r8), intent(out) :: tlayr4(pcols,pverp)  ! Tlayr to the 4th power
  2594)    real(r8), intent(out) :: plh2ob(nbands,pcols,pverp)! Pressure weighted h2o path with 
  2595)                                                       !    Hulst-Curtis-Godson temp. factor 
  2596)                                                       !    for H2O bands 
  2597)    real(r8), intent(out) :: wb(nbands,pcols,pverp)    ! H2o path length with 
  2598)                                                       !    Hulst-Curtis-Godson temp. factor 
  2599)                                                       !    for H2O bands 
  2600) 
  2601) !
  2602) !---------------------------Local variables--------------------------
  2603) !
  2604)    integer i                 ! Longitude index
  2605)    integer k                 ! Level index
  2606)    integer kp1               ! Level index + 1
  2607) 
  2608)    real(r8) repsil               ! Inver ratio mol weight h2o to dry air
  2609)    real(r8) dy                   ! Thickness of layer for tmp interp
  2610)    real(r8) dpnm                 ! Pressure thickness of layer
  2611)    real(r8) dpnmsq               ! Prs squared difference across layer
  2612)    real(r8) dw                   ! Increment in H2O path length
  2613)    real(r8) dplh2o               ! Increment in plh2o
  2614)    real(r8) cpwpl                ! Const in co2 mix ratio to path length conversn
  2615) 
  2616) !--------------------------------------------------------------------
  2617) !
  2618)    repsil = 1._r8/epsilo
  2619) !
  2620) ! Compute co2 and h2o paths
  2621) !
  2622)    cpwpl = 0.5_r8/(gravit_cgs*p0)
  2623)    do i=1,ncol
  2624)       plh2o(i,ntoplw)  = rgsslp*qnm(i,ntoplw)*pnm(i,ntoplw)*pnm(i,ntoplw)
  2625)       plco2(i,ntoplw)  = co2mmr(i)*cpwpl*pnm(i,ntoplw)*pnm(i,ntoplw)
  2626)    end do
  2627)    do k=ntoplw,pver
  2628)       do i=1,ncol
  2629)          plh2o(i,k+1)  = plh2o(i,k) + rgsslp* &
  2630)                          (pnm(i,k+1)**2 - pnm(i,k)**2)*qnm(i,k)
  2631)          plco2(i,k+1)  = co2mmr(i)*cpwpl*pnm(i,k+1)**2
  2632)       end do
  2633)    end do
  2634) !
  2635) ! Set the top and bottom intermediate level temperatures,
  2636) ! top level planck temperature and top layer temp**4.
  2637) !
  2638) ! Tint is lower interface temperature
  2639) ! (not available for bottom layer, so use ground temperature)
  2640) !
  2641)    do i=1,ncol
  2642)       tint4(i,pverp)   = lwupcgs(i)/stebol_cgs
  2643)       tint(i,pverp)    = sqrt(sqrt(tint4(i,pverp)))
  2644)       tplnka(i,ntoplw) = tnm(i,ntoplw)
  2645)       tint(i,ntoplw)   = tplnka(i,ntoplw)
  2646)       tlayr4(i,ntoplw) = tplnka(i,ntoplw)**4
  2647)       tint4(i,ntoplw)  = tlayr4(i,ntoplw)
  2648)    end do
  2649) !
  2650) ! Intermediate level temperatures are computed using temperature
  2651) ! at the full level below less dy*delta t,between the full level
  2652) !
  2653)    do k=ntoplw+1,pver
  2654)       do i=1,ncol
  2655)          dy = (piln(i,k) - pmln(i,k))/(pmln(i,k-1) - pmln(i,k))
  2656)          tint(i,k)  = tnm(i,k) - dy*(tnm(i,k)-tnm(i,k-1))
  2657)          tint4(i,k) = tint(i,k)**4
  2658)       end do
  2659)    end do
  2660) !
  2661) ! Now set the layer temp=full level temperatures and establish a
  2662) ! planck temperature for absorption (tplnka) which is the average
  2663) ! the intermediate level temperatures.  Note that tplnka is not
  2664) ! equal to the full level temperatures.
  2665) !
  2666)    do k=ntoplw+1,pverp
  2667)       do i=1,ncol
  2668)          tlayr(i,k)  = tnm(i,k-1)
  2669)          tlayr4(i,k) = tlayr(i,k)**4
  2670)          tplnka(i,k) = .5_r8*(tint(i,k) + tint(i,k-1))
  2671)       end do
  2672)    end do
  2673) !
  2674) ! Calculate tplank for emissivity calculation.
  2675) ! Assume isothermal tplnke i.e. all levels=ttop.
  2676) !
  2677)    do i=1,ncol
  2678)       tplnke(i)       = tplnka(i,ntoplw)
  2679)       tlayr(i,ntoplw) = tint(i,ntoplw)
  2680)    end do
  2681) !
  2682) ! Now compute h2o path fields:
  2683) !
  2684)    do i=1,ncol
  2685) !
  2686) ! Changed effective path temperature to std. Curtis-Godson form
  2687) !
  2688)       tcg(i,ntoplw) = rga*qnm(i,ntoplw)*pnm(i,ntoplw)*tnm(i,ntoplw)
  2689)       w(i,ntoplw)   = sslp * (plh2o(i,ntoplw)*2._r8) / pnm(i,ntoplw)
  2690) !
  2691) ! Hulst-Curtis-Godson scaling for H2O path
  2692) !
  2693)       wb(1,i,ntoplw) = w(i,ntoplw) * phi(tnm(i,ntoplw),1)
  2694)       wb(2,i,ntoplw) = w(i,ntoplw) * phi(tnm(i,ntoplw),2)
  2695) !
  2696) ! Hulst-Curtis-Godson scaling for effective pressure along H2O path
  2697) !
  2698)       plh2ob(1,i,ntoplw) = plh2o(i,ntoplw) * psi(tnm(i,ntoplw),1)
  2699)       plh2ob(2,i,ntoplw) = plh2o(i,ntoplw) * psi(tnm(i,ntoplw),2)
  2700) 
  2701)       s2c(i,ntoplw) = plh2o(i,ntoplw)*fh2oself(tnm(i,ntoplw))*qnm(i,ntoplw)*repsil
  2702)    end do
  2703) 
  2704)    do k=ntoplw,pver
  2705)       do i=1,ncol
  2706)          dpnm       = pnm(i,k+1) - pnm(i,k)
  2707)          dpnmsq     = pnm(i,k+1)**2 - pnm(i,k)**2
  2708)          dw         = rga*qnm(i,k)*dpnm
  2709)          kp1        = k+1
  2710)          w(i,kp1)   = w(i,k) + dw
  2711) !
  2712) ! Hulst-Curtis-Godson scaling for H2O path
  2713) !
  2714)          wb(1,i,kp1) = wb(1,i,k) + dw * phi(tnm(i,k),1)
  2715)          wb(2,i,kp1) = wb(2,i,k) + dw * phi(tnm(i,k),2)
  2716) !
  2717) ! Hulst-Curtis-Godson scaling for effective pressure along H2O path
  2718) !
  2719)          dplh2o = plh2o(i,kp1) - plh2o(i,k)
  2720) 
  2721)          plh2ob(1,i,kp1) = plh2ob(1,i,k) + dplh2o * psi(tnm(i,k),1)
  2722)          plh2ob(2,i,kp1) = plh2ob(2,i,k) + dplh2o * psi(tnm(i,k),2)
  2723) !
  2724) ! Changed effective path temperature to std. Curtis-Godson form
  2725) !
  2726)          tcg(i,kp1) = tcg(i,k) + dw*tnm(i,k)
  2727)          s2c(i,kp1) = s2c(i,k) + rgsslp*dpnmsq*qnm(i,k)* &
  2728)                       fh2oself(tnm(i,k))*qnm(i,k)*repsil
  2729)       end do
  2730)    end do
  2731) 
  2732) end subroutine radtpl

  2987) subroutine radoz2(ncol, o3, pint, plol, plos)
  2988) !----------------------------------------------------------------------- 
  2989) ! 
  2990) ! Purpose: 
  2991) ! Computes the path length integrals to the model interfaces given the
  2992) ! ozone volume mixing ratio
  2993) ! 
  2994) ! Method: 
  2995) ! <Describe the algorithm(s) used in the routine.> 
  2996) ! <Also include any applicable external references.> 
  2997) ! 
  2998) ! Author: CCM1, CMS Contact J. Kiehl
  2999) ! 
  3000) !------------------------------Input arguments--------------------------
  3001) !
  3002)    integer, intent(in) :: ncol                 ! number of atmospheric columns
  3003) 
  3004)    real(r8), intent(in) :: o3(pcols,pver)      ! ozone mass mixing ratio
  3005)    real(r8), intent(in) :: pint(pcols,pverp)   ! Model interface pressures
  3006) !
  3007) !----------------------------Output arguments---------------------------
  3008) !
  3009)    real(r8), intent(out) :: plol(pcols,pverp)   ! Ozone prs weighted path length (cm)
  3010)    real(r8), intent(out) :: plos(pcols,pverp)   ! Ozone path length (cm)
  3011) !
  3012) !---------------------------Local workspace-----------------------------
  3013) !
  3014)    integer i                ! longitude index
  3015)    integer k                ! level index
  3016) 
  3017)    real(r8) :: v0    ! Volume of a gas at stp (m**3/kmol)
  3018)    real(r8) :: p0    ! Standard pressure (pascals)
  3019)    real(r8) :: cplos ! constant for ozone path length integral
  3020)    real(r8) :: cplol ! constant for ozone path length integral
  3021) 
  3022) !
  3023) !-----------------------------------------------------------------------
  3024) !*******************************************************************
  3025) ! These hardwired constants need to be replaced with common values.
  3026) ! They are here for testing infrastructure changes that should not
  3027) ! change answers.
  3028) ! Constants for ozone path integrals (multiplication by 100 for unit
  3029) ! conversion to cgs from mks):
  3030) !
  3031)    v0    = 22.4136_r8         ! Volume of a gas at stp (m**3/kmol)
  3032)    p0    = 0.1_r8*sslp        ! Standard pressure (pascals)
  3033)    cplos = v0/(mwo3*gravit)       *100.0_r8
  3034)    cplol = v0/(mwo3*gravit*p0)*0.5_r8*100.0_r8
  3035) !*******************************************************************
  3036) !
  3037) ! Evaluate the ozone path length integrals to interfaces;
  3038) ! factors of .1 and .01 to convert pressures from cgs to mks:
  3039) !
  3040)    do i=1,ncol
  3041)       plos(i,ntoplw) = 0.1_r8 *cplos*o3(i,ntoplw)*pint(i,ntoplw)
  3042)       plol(i,ntoplw) = 0.01_r8*cplol*o3(i,ntoplw)*pint(i,ntoplw)*pint(i,ntoplw)
  3043)    end do
  3044)    do k=ntoplw+1,pverp
  3045)       do i=1,ncol
  3046)          plos(i,k) = plos(i,k-1) + 0.1_r8*cplos*o3(i,k-1)*(pint(i,k) - pint(i,k-1))
  3047)          plol(i,k) = plol(i,k-1) + 0.01_r8*cplol*o3(i,k-1)* &
  3048)                     (pint(i,k)*pint(i,k) - pint(i,k-1)*pint(i,k-1))
  3049)       end do
  3050)    end do
  3051) 
  3052) end subroutine radoz2

  3056) subroutine trcpth(ncol                                        , &
  3057)                   tnm     ,pnm     ,cfc11   ,cfc12   ,n2o     , &
  3058)                   ch4     ,qnm     ,ucfc11  ,ucfc12  ,un2o0   , &
  3059)                   un2o1   ,uch4    ,uco211  ,uco212  ,uco213  , &
  3060)                   uco221  ,uco222  ,uco223  ,bn2o0   ,bn2o1   , &
  3061)                   bch4    ,uptype  ,co2mmr)
  3062) !----------------------------------------------------------------------- 
  3063) ! 
  3064) ! Purpose: 
  3065) ! Calculate path lengths and pressure factors for CH4, N2O, CFC11
  3066) ! and CFC12.
  3067) ! 
  3068) ! Method: 
  3069) ! See CCM3 description for details
  3070) ! 
  3071) ! Author: J. Kiehl
  3072) ! 
  3073) !-----------------------------------------------------------------------
  3074) !
  3075) ! Input arguments
  3076) !
  3077)    integer, intent(in) :: ncol                  ! number of atmospheric columns
  3078) 
  3079)    real(r8), intent(in) :: tnm(pcols,pver)      ! Model level temperatures
  3080)    real(r8), intent(in) :: pnm(pcols,pverp)     ! Pres. at model interfaces (dynes/cm2)
  3081)    real(r8), intent(in) :: qnm(pcols,pver)      ! h2o specific humidity
  3082)    real(r8), intent(in) :: cfc11(pcols,pver)    ! CFC11 mass mixing ratio
  3083) !
  3084)    real(r8), intent(in) :: cfc12(pcols,pver)    ! CFC12 mass mixing ratio
  3085)    real(r8), intent(in) :: n2o(pcols,pver)      ! N2O mass mixing ratio
  3086)    real(r8), intent(in) :: ch4(pcols,pver)      ! CH4 mass mixing ratio
  3087)    real(r8), intent(in) :: co2mmr(pcols)        ! co2 column mean mass mixing ratio
  3088) 
  3089) !
  3090) ! Output arguments
  3091) !
  3092)    real(r8), intent(out) :: ucfc11(pcols,pverp)  ! CFC11 path length
  3093)    real(r8), intent(out) :: ucfc12(pcols,pverp)  ! CFC12 path length
  3094)    real(r8), intent(out) :: un2o0(pcols,pverp)   ! N2O path length
  3095)    real(r8), intent(out) :: un2o1(pcols,pverp)   ! N2O path length (hot band)
  3096)    real(r8), intent(out) :: uch4(pcols,pverp)    ! CH4 path length
  3097) !
  3098)    real(r8), intent(out) :: uco211(pcols,pverp)  ! CO2 9.4 micron band path length
  3099)    real(r8), intent(out) :: uco212(pcols,pverp)  ! CO2 9.4 micron band path length
  3100)    real(r8), intent(out) :: uco213(pcols,pverp)  ! CO2 9.4 micron band path length
  3101)    real(r8), intent(out) :: uco221(pcols,pverp)  ! CO2 10.4 micron band path length
  3102)    real(r8), intent(out) :: uco222(pcols,pverp)  ! CO2 10.4 micron band path length
  3103) !
  3104)    real(r8), intent(out) :: uco223(pcols,pverp)  ! CO2 10.4 micron band path length
  3105)    real(r8), intent(out) :: bn2o0(pcols,pverp)   ! pressure factor for n2o
  3106)    real(r8), intent(out) :: bn2o1(pcols,pverp)   ! pressure factor for n2o
  3107)    real(r8), intent(out) :: bch4(pcols,pverp)    ! pressure factor for ch4
  3108)    real(r8), intent(out) :: uptype(pcols,pverp)  ! p-type continuum path length
  3109) 
  3110) !
  3111) !---------------------------Local variables-----------------------------
  3112) !
  3113)    integer   i               ! Longitude index
  3114)    integer   k               ! Level index
  3115) !
  3116)    real(r8) co2fac(pcols,1)      ! co2 factor
  3117)    real(r8) alpha1(pcols)        ! stimulated emission term
  3118)    real(r8) alpha2(pcols)        ! stimulated emission term
  3119)    real(r8) rt(pcols)            ! reciprocal of local temperature
  3120)    real(r8) rsqrt(pcols)         ! reciprocal of sqrt of temp
  3121) !
  3122)    real(r8) pbar(pcols)          ! mean pressure
  3123)    real(r8) dpnm(pcols)          ! difference in pressure
  3124)    real(r8) diff                 ! diffusivity factor
  3125) !
  3126) !--------------------------Data Statements------------------------------
  3127) !
  3128)    data diff /1.66_r8/
  3129) !
  3130) !-----------------------------------------------------------------------
  3131) !
  3132) !  Calculate path lengths for the trace gases at model top
  3133) !
  3134) 
  3135)    do i = 1,ncol
  3136)       ucfc11(i,ntoplw) = 1.8_r8 * cfc11(i,ntoplw) * pnm(i,ntoplw) * rga
  3137)       ucfc12(i,ntoplw) = 1.8_r8 * cfc12(i,ntoplw) * pnm(i,ntoplw) * rga
  3138)       un2o0(i,ntoplw) = diff * 1.02346e5_r8 * n2o(i,ntoplw) * pnm(i,ntoplw) * rga / sqrt(tnm(i,ntoplw))
  3139)       un2o1(i,ntoplw) = diff * 2.01909_r8 * un2o0(i,ntoplw) * exp(-847.36_r8/tnm(i,ntoplw))
  3140)       uch4(i,ntoplw)  = diff * 8.60957e4_r8 * ch4(i,ntoplw) * pnm(i,ntoplw) * rga / sqrt(tnm(i,ntoplw))
  3141)       co2fac(i,1)     = diff * co2mmr(i) * pnm(i,ntoplw) * rga
  3142)       alpha1(i) = (1.0_r8 - exp(-1540.0_r8/tnm(i,ntoplw)))**3.0_r8/sqrt(tnm(i,ntoplw))
  3143)       alpha2(i) = (1.0_r8 - exp(-1360.0_r8/tnm(i,ntoplw)))**3.0_r8/sqrt(tnm(i,ntoplw))
  3144)       uco211(i,ntoplw) = 3.42217e3_r8 * co2fac(i,1) * alpha1(i) * exp(-1849.7_r8/tnm(i,ntoplw))
  3145)       uco212(i,ntoplw) = 6.02454e3_r8 * co2fac(i,1) * alpha1(i) * exp(-2782.1_r8/tnm(i,ntoplw))
  3146)       uco213(i,ntoplw) = 5.53143e3_r8 * co2fac(i,1) * alpha1(i) * exp(-3723.2_r8/tnm(i,ntoplw))
  3147)       uco221(i,ntoplw) = 3.88984e3_r8 * co2fac(i,1) * alpha2(i) * exp(-1997.6_r8/tnm(i,ntoplw))
  3148)       uco222(i,ntoplw) = 3.67108e3_r8 * co2fac(i,1) * alpha2(i) * exp(-3843.8_r8/tnm(i,ntoplw))
  3149)       uco223(i,ntoplw) = 6.50642e3_r8 * co2fac(i,1) * alpha2(i) * exp(-2989.7_r8/tnm(i,ntoplw))
  3150)       bn2o0(i,ntoplw) = diff * 19.399_r8 * pnm(i,ntoplw)**2.0_r8 * n2o(i,ntoplw) * &
  3151)                    1.02346e5_r8 * rga / (sslp*tnm(i,ntoplw))
  3152)       bn2o1(i,ntoplw) = bn2o0(i,ntoplw) * exp(-847.36_r8/tnm(i,ntoplw)) * 2.06646e5_r8
  3153)       bch4(i,ntoplw) = diff * 2.94449_r8 * ch4(i,ntoplw) * pnm(i,ntoplw)**2.0_r8 * rga * &
  3154)                   8.60957e4_r8 / (sslp*tnm(i,ntoplw))
  3155)       uptype(i,ntoplw) = diff * qnm(i,ntoplw) * pnm(i,ntoplw)**2.0_r8 *  &
  3156)                     exp(1800.0_r8*(1.0_r8/tnm(i,ntoplw) - 1.0_r8/296.0_r8)) * rga / sslp
  3157)    end do
  3158) !
  3159) ! Calculate trace gas path lengths through model atmosphere
  3160) !
  3161)    do k = ntoplw,pver
  3162)       do i = 1,ncol
  3163)          rt(i) = 1._r8/tnm(i,k)
  3164)          rsqrt(i) = sqrt(rt(i))
  3165)          pbar(i) = 0.5_r8 * (pnm(i,k+1) + pnm(i,k)) / sslp
  3166)          dpnm(i) = (pnm(i,k+1) - pnm(i,k)) * rga
  3167)          alpha1(i) = diff * rsqrt(i) * (1.0_r8 - exp(-1540.0_r8/tnm(i,k)))**3.0_r8
  3168)          alpha2(i) = diff * rsqrt(i) * (1.0_r8 - exp(-1360.0_r8/tnm(i,k)))**3.0_r8
  3169)          ucfc11(i,k+1) = ucfc11(i,k) +  1.8_r8 * cfc11(i,k) * dpnm(i)
  3170)          ucfc12(i,k+1) = ucfc12(i,k) +  1.8_r8 * cfc12(i,k) * dpnm(i)
  3171)          un2o0(i,k+1) = un2o0(i,k) + diff * 1.02346e5_r8 * n2o(i,k) * rsqrt(i) * dpnm(i)
  3172)          un2o1(i,k+1) = un2o1(i,k) + diff * 2.06646e5_r8 * n2o(i,k) * &
  3173)                         rsqrt(i) * exp(-847.36_r8/tnm(i,k)) * dpnm(i)
  3174)          uch4(i,k+1) = uch4(i,k) + diff * 8.60957e4_r8 * ch4(i,k) * rsqrt(i) * dpnm(i)
  3175)          uco211(i,k+1) = uco211(i,k) + 1.15_r8*3.42217e3_r8 * alpha1(i) * &
  3176)                          co2mmr(i) * exp(-1849.7_r8/tnm(i,k)) * dpnm(i)
  3177)          uco212(i,k+1) = uco212(i,k) + 1.15_r8*6.02454e3_r8 * alpha1(i) * &
  3178)                          co2mmr(i) * exp(-2782.1_r8/tnm(i,k)) * dpnm(i)
  3179)          uco213(i,k+1) = uco213(i,k) + 1.15_r8*5.53143e3_r8 * alpha1(i) * &
  3180)                          co2mmr(i) * exp(-3723.2_r8/tnm(i,k)) * dpnm(i)
  3181)          uco221(i,k+1) = uco221(i,k) + 1.15_r8*3.88984e3_r8 * alpha2(i) * &
  3182)                          co2mmr(i) * exp(-1997.6_r8/tnm(i,k)) * dpnm(i)
  3183)          uco222(i,k+1) = uco222(i,k) + 1.15_r8*3.67108e3_r8 * alpha2(i) * &

  3220) !    260K and 296K by:
  3221) !
  3222) !         TFAC =  (T(IPATH) - 296.0)/(260.0 - 296.0) 
  3223) !         CSFFT = CSFF296*(CSFF260/CSFF296)**TFAC
  3224) !
  3225) ! For 800-1200 cm^-1, (CSFF260/CSFF296) ranges from ~2.1 to ~1.9
  3226) !     with increasing wavenumber.  The ratio <CSFF260>/<CSFF296>,
  3227) !     where <> indicates average over wavenumber, is ~2.07
  3228) ! 
  3229) ! fh2oself is (<CSFF260>/<CSFF296>)**TFAC
  3230) !
  3231)    real(r8),intent(in) :: temp     ! path temperature
  3232)    real(r8) fh2oself               ! mean ratio of self-continuum at temp and 296K
  3233) 
  3234)    fh2oself = 2.0727484_r8**((296.0_r8 - temp) / 36.0_r8)
  3235) end function fh2oself
