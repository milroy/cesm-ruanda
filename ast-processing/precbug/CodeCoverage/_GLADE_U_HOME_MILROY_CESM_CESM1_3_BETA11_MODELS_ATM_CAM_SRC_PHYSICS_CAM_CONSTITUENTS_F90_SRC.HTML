    89)   subroutine cnst_add (name, mwc, cpc, qminc, &
    90)                        ind, longname, readiv, mixtype, molectype, cam_outfld, &
    91)                        fixed_ubc, fixed_ubflx, is_convtran1)
    92) !----------------------------------------------------------------------- 
    93) ! 
    94) ! Purpose: Register a constituent to be advected by the large scale winds and transported by
    95) !          subgrid scale processes.
    96) !
    97) !---------------------------------------------------------------------------------
    98) !
    99)     character(len=*), intent(in) :: &
   100)        name      ! constituent name used as variable name in history file output (8 char max)
   101)     real(r8),intent(in)    :: mwc    ! constituent molecular weight (kg/kmol)
   102)     real(r8),intent(in)    :: cpc    ! constituent specific heat at constant pressure (J/kg/K)
   103)     real(r8),intent(in)    :: qminc  ! minimum value of mass mixing ratio (kg/kg)
   104)                                      ! normally 0., except water 1.E-12, for radiation.
   105)     integer, intent(out)   :: ind    ! global constituent index (in q array)
   106) 
   107)     character(len=*), intent(in), optional :: &
   108)        longname    ! value for long_name attribute in netcdf output (128 char max, defaults to name)
   109)     logical,          intent(in), optional :: &
   110)        readiv      ! true => read initial values from initial file (default: true)
   111)     character(len=*), intent(in), optional :: &
   112)        mixtype     ! mixing ratio type (dry, wet)
   113)     character(len=*), intent(in), optional :: &
   114)        molectype     ! molecular diffusion type (minor, major)
   115)     logical,          intent(in), optional :: &
   116)        cam_outfld  ! true => default CAM output of constituent in kg/kg
   117)     logical,          intent(in), optional :: &
   118)        fixed_ubc   ! true => const has a fixed upper bndy condition
   119)     logical,          intent(in), optional :: &
   120)        fixed_ubflx ! true => const has a non-zero fixed upper bndy flux value
   121)     logical,          intent(in), optional :: &
   122)        is_convtran1 ! true => convect in convtran1 rather than convtran2
   123) 
   124) !-----------------------------------------------------------------------
   125) 
   126) ! set tracer index and check validity, advected tracer
   127)     padv = padv+1
   128)     ind  = padv
   129)     if (padv > pcnst) then
   130)        write(iulog,*) 'CNST_ADD: advected tracer index greater than pcnst = ', pcnst
   131)        call endrun
   132)     end if
   133) 
   134) ! set tracer name and constants
   135)     cnst_name(ind) = name
   136)     if ( present(longname) )then
   137)        cnst_longname(ind) = longname
   138)     else
   139)        cnst_longname(ind) = name
   140)     end if
   141) 
   142) ! set whether to read initial values from initial file
   143)     if ( present(readiv) ) then
   144)        read_init_vals(ind) = readiv
   145)     else
   146)        read_init_vals(ind) = readtrace
   147)     end if
   148) 
   149) ! set constituent mixing ratio type
   150)     if ( present(mixtype) )then
   151)        cnst_type(ind) = mixtype
   152)     else
   153)        cnst_type(ind) = 'wet'
   154)     end if
   155) 
   156) ! set constituent molecular diffusion type
   157)     if ( present(molectype) )then
   158)        cnst_molec(ind) = molectype
   159)     else
   160)        cnst_molec(ind) = 'minor'
   161)     end if
   162) 
   163) ! set outfld type 
   164) ! (false: the module declaring the constituent is responsible for outfld calls)
   165)     if ( present(cam_outfld) ) then
   166)        cam_outfld_(ind) = cam_outfld
   167)     else
   168)        cam_outfld_(ind) = .true.
   169)     end if
   170) 
   171) ! set upper boundary condition type
   172)     if ( present(fixed_ubc) ) then
   173)        cnst_fixed_ubc(ind) = fixed_ubc
   174)     else
   175)        cnst_fixed_ubc(ind) = .false.
   176)     end if
   177) 
   178)  ! set upper boundary flux type
   179)     if ( present(fixed_ubflx) ) then
   180)        cnst_fixed_ubflx(ind) = fixed_ubflx
   181)     else
   182)        cnst_fixed_ubflx(ind) = .false.
   183)     end if
   184) 
   185) ! set upper convective transport flag
   186)     if ( present(is_convtran1) ) then
   187)        cnst_is_convtran1(ind) = is_convtran1
   188)     else
   189)        cnst_is_convtran1(ind) = .false.
   190)     end if
   191) 
   192)     cnst_cp  (ind) = cpc
   193)     cnst_mw  (ind) = mwc
   194)     qmin     (ind) = qminc
   195)     qmincg   (ind) = qminc
   196)     if (ind == 1) qmincg = 0._r8  ! This crap is replicate what was there before ****
   197) 
   198)     cnst_rgas(ind) = r_universal * mwc
   199)     cnst_cv  (ind) = cpc - cnst_rgas(ind)
   200) 
   201)     return
   202)   end subroutine cnst_add

   218)   subroutine cnst_get_ind (name, ind, abort)
   219) !----------------------------------------------------------------------- 
   220) ! 
   221) ! Purpose: Get the index of a constituent 
   222) ! 
   223) ! Author:  B.A. Boville
   224) ! 
   225) !-----------------------------Arguments---------------------------------
   226) !
   227)     character(len=*),  intent(in)  :: name  ! constituent name
   228)     integer,           intent(out) :: ind   ! global constituent index (in q array)
   229)     logical, optional, intent(in)  :: abort ! optional flag controlling abort
   230) 
   231) !---------------------------Local workspace-----------------------------
   232)     integer :: m                                   ! tracer index
   233)     logical :: abort_on_error
   234) !-----------------------------------------------------------------------
   235) 
   236) ! Find tracer name in list
   237)     do m = 1, pcnst
   238)        if (name == cnst_name(m)) then
   239)           ind  = m
   240)           return
   241)        end if
   242)     end do
   243) 
   244) ! Unrecognized name
   245)     abort_on_error = .true.
   246)     if ( present(abort) ) abort_on_error = abort
   247) 
   248)     if ( abort_on_error ) then
   249)        write(iulog,*) 'CNST_GET_IND, name:', name,  ' not found in list:', cnst_name(:)
   250)        call endrun('CNST_GET_IND: name not found')
   251)     end if
   252) 
   253) ! error return
   254)     ind = -1
   255) 
   256)   end subroutine cnst_get_ind

   376) !----------------------------------------------------------------------- 
   377) !

   380) !
