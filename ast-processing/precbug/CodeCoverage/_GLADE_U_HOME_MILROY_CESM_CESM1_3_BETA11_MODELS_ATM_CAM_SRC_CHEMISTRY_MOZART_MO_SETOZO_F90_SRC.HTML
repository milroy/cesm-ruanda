     8)       subroutine setozo( z, nw, wl, tlay, dto3, &
     9)                          to3, o3, airlev, o3top )
    10) !-----------------------------------------------------------------------------
    11) !   purpose:
    12) !   set up an altitude profile of ozone, and corresponding absorption
    13) !   optical depths.  subroutine includes a shape-conserving scaling method
    14) !   that allows scaling of the entire profile to a given overhead ozone
    15) !   column amount.
    16) !-----------------------------------------------------------------------------
    17) !   parameters:
    18) !   dobnew - real, overhead ozone column amount (du) to which profile     (i)
    19) !            should be scaled.  if dobnew < 0, no scaling is done
    20) !   nz     - integer, number of specified altitude levels in the working  (i)
    21) !            grid
    22) !   z      - real, specified altitude working grid (km)
    23) !   nw     - integer, number of specified intervals + 1 in working        (i)
    24) !            wavelength grid
    25) !   wl     - real, vector of lower limits of wavelength intervals in      (i)
    26) !            working wavelength grid
    27) !   xso3   - real, molecular absoprtion cross section (cm^2) of o3 at     (i)
    28) !            each specified wavelength (wmo value at 273)
    29) !   s226   - real, molecular absoprtion cross section (cm^2) of o3 at     (i)
    30) !            each specified wavelength (value from molina and molina at 226k)
    31) !   s263   - real, molecular absoprtion cross section (cm^2) of o3 at     (i)
    32) !            each specified wavelength (value from molina and molina at 263k)
    33) !   s298   - real, molecular absoprtion cross section (cm^2) of o3 at     (i)
    34) !            each specified wavelength (value from molina and molina at 298k)
    35) !   tlay   - real, temperature (k) at each specified altitude layer       (i)
    36) !   dto3   - real, optical depth due to ozone absorption at each          (o)
    37) !            specified altitude at each specified wavelength
    38) !   to3    - real, totol ozone column                                     (o)
    39) !-----------------------------------------------------------------------------
    40) 
    41)       use mo_params, only : kw
    42)       use mo_waveo3, only : xso3, s226, s263, s298
    43)       use ppgrid,    only : pver, pverp
    44) 
    45)       implicit none
    46) 
    47) !-----------------------------------------------------------------------------
    48) !	... dummy arguments
    49) !-----------------------------------------------------------------------------
    50)       integer, intent(in) :: nw
    51)       real(r8), intent(in)    :: o3top
    52)       real(r8), intent(in)    :: wl(kw)
    53)       real(r8), intent(in)    :: z(pverp)
    54)       real(r8), intent(in)    :: tlay(pver) 
    55)       real(r8), intent(in)    :: airlev(pverp) 
    56)       real(r8), intent(out)   :: dto3(pver,nw)     
    57)       real(r8), intent(out)   :: to3(pverp)
    58)       real(r8), intent(inout) :: o3(pverp)
    59) 
    60) !-----------------------------------------------------------------------------
    61) !	... local variables
    62) !-----------------------------------------------------------------------------
    63)       real(r8), parameter :: wave_lims(2) = (/ 240.5_r8, 350._r8 /)
    64)       real(r8), parameter :: t_lims(3)    = (/ 226._r8, 263._r8, 298._r8 /)
    65)       real(r8), parameter :: tfac1 = 1._r8/(t_lims(2) - t_lims(1))
    66)       real(r8), parameter :: tfac2 = 1._r8/(t_lims(3) - t_lims(2))
    67) 
    68)       integer :: k, wn
    69)       real(r8)    :: so3
    70)       real(r8)    :: cz(pverp)
    71) 
    72) !-----------------------------------------------------------------------------
    73) ! 	... compute column increments
    74) !-----------------------------------------------------------------------------
    75)       o3(1:pverp) = o3(1:pverp)*airlev(1:pverp) 
    76)       cz(1:pver)  = .5_r8*(o3(2:pverp) + o3(1:pver)) * 1.e5_r8 * (z(2:pverp) - z(1:pver))
    77)       to3(pverp)  = o3top
    78)       do k = pver,1,-1
    79)          to3(k) = to3(k+1) + cz(k)
    80)       end do
    81) 
    82) !-----------------------------------------------------------------------------
    83) ! 	... include exponential tail integral from infinity to 50 km,
    84) !           fold tail integral into top layer
    85) !           specify scale height near top of data.
    86) !-----------------------------------------------------------------------------
    87) 
    88)       cz(pver) = cz(pver) + o3top
    89) !-----------------------------------------------------------------------------
    90) ! 	... calculate ozone optical depth for each layer, with temperature 
    91) !           correction.  output, dto3(kz,kw)
    92) !-----------------------------------------------------------------------------
    93)       do wn = 1,nw
    94)          if( wl(wn) > wave_lims(1)  .and. wl(wn+1) < wave_lims(2) ) then
    95)             do k = 1,pver
    96)                if( tlay(k) < t_lims(2) ) then
    97)                   so3 = s226(wn) + (s263(wn) - s226(wn)) * tfac1 * (tlay(k) - t_lims(1))
    98)                else
    99)                   so3 = s263(wn) + (s298(wn) - s263(wn)) * tfac2 * (tlay(k) - t_lims(2))
   100)                end if
   101)                dto3(k,wn) = cz(k)*so3
   102)             end do
   103)          else
   104)             dto3(1:pver,wn) = cz(1:pver) * xso3(wn)
   105)          end if
   106)       end do
   107) 
   108)       end subroutine setozo
