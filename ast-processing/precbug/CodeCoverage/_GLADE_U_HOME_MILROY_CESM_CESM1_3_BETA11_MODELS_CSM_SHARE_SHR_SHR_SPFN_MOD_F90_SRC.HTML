   313) ! 6 December 2006 -- B. Eaton
   314) ! The following comments are from the original version of CALERF.
   315) ! The only changes in implementing this module are that the function
   316) ! names previously used for the single precision versions have been
   317) ! adopted for the new generic interfaces.  To support these interfaces
   318) ! there is now both a single precision version (calerf_r4) and a
   319) ! double precision version (calerf_r8) of CALERF below.  These versions
   320) ! are hardcoded to use IEEE arithmetic.
   321) !
   322) !------------------------------------------------------------------
   323) !
   324) ! This packet evaluates  erf(x),  erfc(x),  and  exp(x*x)*erfc(x)
   325) !   for a real argument  x.  It contains three FUNCTION type
   326) !   subprograms: ERF, ERFC, and ERFCX (or ERF_R8, ERFC_R8, and ERFCX_R8),
   327) !   and one SUBROUTINE type subprogram, CALERF.  The calling
   328) !   statements for the primary entries are:
   329) !
   330) !                   Y=ERF(X)     (or   Y=ERF_R8(X)),
   331) !
   332) !                   Y=ERFC(X)    (or   Y=ERFC_R8(X)),
   333) !   and
   334) !                   Y=ERFCX(X)   (or   Y=ERFCX_R8(X)).
   335) !
   336) !   The routine  CALERF  is intended for internal packet use only,
   337) !   all computations within the packet being concentrated in this
   338) !   routine.  The function subprograms invoke  CALERF  with the
   339) !   statement
   340) !
   341) !          CALL CALERF(ARG,RESULT,JINT)
   342) !
   343) !   where the parameter usage is as follows
   344) !
   345) !      Function                     Parameters for CALERF
   346) !       call              ARG                  Result          JINT
   347) !
   348) !     ERF(ARG)      ANY REAL ARGUMENT         ERF(ARG)          0
   349) !     ERFC(ARG)     ABS(ARG) .LT. XBIG        ERFC(ARG)         1
   350) !     ERFCX(ARG)    XNEG .LT. ARG .LT. XMAX   ERFCX(ARG)        2
   351) !
   352) !   The main computation evaluates near-minimax approximations
   353) !   from "Rational Chebyshev approximations for the error function"
   354) !   by W. J. Cody, Math. Comp., 1969, PP. 631-638.  This
   355) !   transportable program uses rational functions that theoretically
   356) !   approximate  erf(x)  and  erfc(x)  to at least 18 significant
   357) !   decimal digits.  The accuracy achieved depends on the arithmetic
   358) !   system, the compiler, the intrinsic functions, and proper
   359) !   selection of the machine-dependent constants.
   360) !
   361) !*******************************************************************
   362) !*******************************************************************
   363) !
   364) ! Explanation of machine-dependent constants
   365) !
   366) !   XMIN   = the smallest positive floating-point number.
   367) !   XINF   = the largest positive finite floating-point number.
   368) !   XNEG   = the largest negative argument acceptable to ERFCX;
   369) !            the negative of the solution to the equation
   370) !            2*exp(x*x) = XINF.
   371) !   XSMALL = argument below which erf(x) may be represented by
   372) !            2*x/sqrt(pi)  and above which  x*x  will not underflow.
   373) !            A conservative value is the largest machine number X
   374) !            such that   1.0 + X = 1.0   to machine precision.
   375) !   XBIG   = largest argument acceptable to ERFC;  solution to
   376) !            the equation:  W(x) * (1-0.5/x**2) = XMIN,  where
   377) !            W(x) = exp(-x*x)/[x*sqrt(pi)].
   378) !   XHUGE  = argument above which  1.0 - 1/(2*x*x) = 1.0  to
   379) !            machine precision.  A conservative value is
   380) !            1/[2*sqrt(XSMALL)]
   381) !   XMAX   = largest acceptable argument to ERFCX; the minimum
   382) !            of XINF and 1/[sqrt(pi)*XMIN].
   383) !
   384) !   Approximate values for some important machines are:
   385) !
   386) !                          XMIN       XINF        XNEG     XSMALL
   387) !
   388) !  CDC 7600      (S.P.)  3.13E-294   1.26E+322   -27.220  7.11E-15
   389) !  CRAY-1        (S.P.)  4.58E-2467  5.45E+2465  -75.345  7.11E-15
   390) !  IEEE (IBM/XT,
   391) !    SUN, etc.)  (S.P.)  1.18E-38    3.40E+38     -9.382  5.96E-8
   392) !  IEEE (IBM/XT,
   393) !    SUN, etc.)  (D.P.)  2.23D-308   1.79D+308   -26.628  1.11D-16
   394) !  IBM 195       (D.P.)  5.40D-79    7.23E+75    -13.190  1.39D-17
   395) !  UNIVAC 1108   (D.P.)  2.78D-309   8.98D+307   -26.615  1.73D-18
   396) !  VAX D-Format  (D.P.)  2.94D-39    1.70D+38     -9.345  1.39D-17
   397) !  VAX G-Format  (D.P.)  5.56D-309   8.98D+307   -26.615  1.11D-16
   398) !
   399) !
   400) !                          XBIG       XHUGE       XMAX
   401) !
   402) !  CDC 7600      (S.P.)  25.922      8.39E+6     1.80X+293
   403) !  CRAY-1        (S.P.)  75.326      8.39E+6     5.45E+2465
   404) !  IEEE (IBM/XT,
   405) !    SUN, etc.)  (S.P.)   9.194      2.90E+3     4.79E+37
   406) !  IEEE (IBM/XT,
   407) !    SUN, etc.)  (D.P.)  26.543      6.71D+7     2.53D+307
   408) !  IBM 195       (D.P.)  13.306      1.90D+8     7.23E+75
   409) !  UNIVAC 1108   (D.P.)  26.582      5.37D+8     8.98D+307
   410) !  VAX D-Format  (D.P.)   9.269      1.90D+8     1.70D+38
   411) !  VAX G-Format  (D.P.)  26.569      6.71D+7     8.98D+307
   412) !
   413) !*******************************************************************
   414) !*******************************************************************
   415) !
   416) ! Error returns
   417) !
   418) !  The program returns  ERFC = 0      for  ARG .GE. XBIG;
   419) !
   420) !                       ERFCX = XINF  for  ARG .LT. XNEG;
   421) !      and
   422) !                       ERFCX = 0     for  ARG .GE. XMAX.
   423) !
   424) !
   425) ! Intrinsic functions required are:
   426) !
   427) !     ABS, AINT, EXP
   428) !
   429) !

   431) !          Mathematics and Computer Science Division
   432) !          Argonne National Laboratory
   433) !          Argonne, IL 60439

   439) SUBROUTINE CALERF_r8(ARG, RESULT, JINT)
   440) 
   441)    !------------------------------------------------------------------
   442)    !  This version uses 8-byte reals
   443)    !------------------------------------------------------------------
   444)    integer, parameter :: rk = r8
   445) 
   446)    ! arguments
   447)    real(rk), intent(in)  :: arg
   448)    integer,  intent(in)  :: jint
   449)    real(rk), intent(out) :: result
   450) 
   451)    ! local variables
   452)    INTEGER :: I
   453) 
   454)    real(rk) :: X, Y, YSQ, XNUM, XDEN, DEL
   455) 
   456)    !------------------------------------------------------------------
   457)    !  Mathematical constants
   458)    !------------------------------------------------------------------
   459)    real(rk), parameter :: ZERO   = 0.0E0_rk
   460)    real(rk), parameter :: FOUR   = 4.0E0_rk
   461)    real(rk), parameter :: ONE    = 1.0E0_rk
   462)    real(rk), parameter :: HALF   = 0.5E0_rk
   463)    real(rk), parameter :: TWO    = 2.0E0_rk
   464)    ! 1/sqrt(pi)
   465)    real(rk), parameter :: SQRPI  = 5.6418958354775628695E-1_rk
   466)    real(rk), parameter :: THRESH = 0.46875E0_rk
   467)    real(rk), parameter :: SIXTEN = 16.0E0_rk
   468) 
   469)    !------------------------------------------------------------------
   470)    !  Machine-dependent constants: IEEE double precision values
   471)    !------------------------------------------------------------------
   472)    real(rk), parameter :: XNEG   = -26.628E0_r8
   473)    real(rk), parameter :: XBIG   =  26.543E0_r8
   474)    real(rk), parameter :: XHUGE  =   6.71E7_r8
   475) 
   476)    !------------------------------------------------------------------
   477)    !  Coefficients for approximation to  erf  in first interval
   478)    !------------------------------------------------------------------
   479)    real(rk), parameter :: A(5) = (/ 3.16112374387056560E00_rk, 1.13864154151050156E02_rk, &
   480)                                     3.77485237685302021E02_rk, 3.20937758913846947E03_rk, &
   481)                                     1.85777706184603153E-1_rk /)
   482)    real(rk), parameter :: B(4) = (/ 2.36012909523441209E01_rk, 2.44024637934444173E02_rk, &
   483)                                     1.28261652607737228E03_rk, 2.84423683343917062E03_rk /)
   484) 
   485)    !------------------------------------------------------------------
   486)    !  Coefficients for approximation to  erfc  in second interval
   487)    !------------------------------------------------------------------
   488)    real(rk), parameter :: C(9) = (/ 5.64188496988670089E-1_rk, 8.88314979438837594E00_rk, &
   489)                                     6.61191906371416295E01_rk, 2.98635138197400131E02_rk, &
   490)                                     8.81952221241769090E02_rk, 1.71204761263407058E03_rk, &
   491)                                     2.05107837782607147E03_rk, 1.23033935479799725E03_rk, &
   492)                                     2.15311535474403846E-8_rk /)
   493)    real(rk), parameter :: D(8) = (/ 1.57449261107098347E01_rk, 1.17693950891312499E02_rk, &
   494)                                     5.37181101862009858E02_rk, 1.62138957456669019E03_rk, &
   495)                                     3.29079923573345963E03_rk, 4.36261909014324716E03_rk, &
   496)                                     3.43936767414372164E03_rk, 1.23033935480374942E03_rk /)
   497) 
   498)    !------------------------------------------------------------------
   499)    !  Coefficients for approximation to  erfc  in third interval
   500)    !------------------------------------------------------------------
   501)    real(rk), parameter :: P(6) = (/ 3.05326634961232344E-1_rk, 3.60344899949804439E-1_rk, &
   502)                                     1.25781726111229246E-1_rk, 1.60837851487422766E-2_rk, &
   503)                                     6.58749161529837803E-4_rk, 1.63153871373020978E-2_rk /)
   504)    real(rk), parameter :: Q(5) = (/ 2.56852019228982242E00_rk, 1.87295284992346047E00_rk, &
   505)                                     5.27905102951428412E-1_rk, 6.05183413124413191E-2_rk, &
   506)                                     2.33520497626869185E-3_rk /)
   507) 
   508)    !------------------------------------------------------------------
   509)    X = ARG
   510)    Y = ABS(X)
   511)    IF (Y .LE. THRESH) THEN
   512)       !------------------------------------------------------------------
   513)       !  Evaluate  erf  for  |X| <= 0.46875
   514)       !------------------------------------------------------------------
   515)       YSQ = ZERO
   516)       IF (Y .GT. XSMALLR8) YSQ = Y * Y
   517)       XNUM = A(5)*YSQ
   518)       XDEN = YSQ
   519)       DO I = 1, 3
   520)          XNUM = (XNUM + A(I)) * YSQ
   521)          XDEN = (XDEN + B(I)) * YSQ
   522)       end do
   523)       RESULT = X * (XNUM + A(4)) / (XDEN + B(4))
   524)       IF (JINT .NE. 0) RESULT = ONE - RESULT
   525)       IF (JINT .EQ. 2) RESULT = EXP(YSQ) * RESULT
   526)       GO TO 80
   527)    ELSE IF (Y .LE. FOUR) THEN
   528)       !------------------------------------------------------------------
   529)       !  Evaluate  erfc  for 0.46875 <= |X| <= 4.0
   530)       !------------------------------------------------------------------
   531)       XNUM = C(9)*Y
   532)       XDEN = Y
   533)       DO I = 1, 7
   534)          XNUM = (XNUM + C(I)) * Y
   535)          XDEN = (XDEN + D(I)) * Y
   536)       end do
   537)       RESULT = (XNUM + C(8)) / (XDEN + D(8))
   538)       IF (JINT .NE. 2) THEN
   539)          YSQ = AINT(Y*SIXTEN)/SIXTEN
   540)          DEL = (Y-YSQ)*(Y+YSQ)
   541)          RESULT = EXP(-YSQ*YSQ) * EXP(-DEL) * RESULT
   542)       END IF
   543)    ELSE
   544)       !------------------------------------------------------------------
   545)       !  Evaluate  erfc  for |X| > 4.0
   546)       !------------------------------------------------------------------
   547)       RESULT = ZERO
   548)       IF (Y .GE. XBIG) THEN
   549)          IF ((JINT .NE. 2) .OR. (Y .GE. XMAXR8)) GO TO 30
   550)          IF (Y .GE. XHUGE) THEN
   551)             RESULT = SQRPI / Y
   552)             GO TO 30
   553)          END IF
   554)       END IF
   555)       YSQ = ONE / (Y * Y)
   556)       XNUM = P(6)*YSQ
   557)       XDEN = YSQ
   558)       DO I = 1, 4
   559)          XNUM = (XNUM + P(I)) * YSQ
   560)          XDEN = (XDEN + Q(I)) * YSQ
   561)       end do
   562)       RESULT = YSQ *(XNUM + P(5)) / (XDEN + Q(5))
   563)       RESULT = (SQRPI -  RESULT) / Y
   564)       IF (JINT .NE. 2) THEN
   565)          YSQ = AINT(Y*SIXTEN)/SIXTEN
   566)          DEL = (Y-YSQ)*(Y+YSQ)
   567)          RESULT = EXP(-YSQ*YSQ) * EXP(-DEL) * RESULT
   568)       END IF
   569)    END IF
   570) 30 continue
   571)    !------------------------------------------------------------------
   572)    !  Fix up for negative argument, erf, etc.
   573)    !------------------------------------------------------------------
   574)    IF (JINT .EQ. 0) THEN
   575)       RESULT = (HALF - RESULT) + HALF
   576)       IF (X .LT. ZERO) RESULT = -RESULT
   577)    ELSE IF (JINT .EQ. 1) THEN
   578)       IF (X .LT. ZERO) RESULT = TWO - RESULT
   579)    ELSE
   580)       IF (X .LT. ZERO) THEN
   581)          IF (X .LT. XNEG) THEN
   582)             RESULT = XINFR8
   583)          ELSE
   584)             YSQ = AINT(X*SIXTEN)/SIXTEN
   585)             DEL = (X-YSQ)*(X+YSQ)
   586)             Y = EXP(YSQ*YSQ) * EXP(DEL)
   587)             RESULT = (Y+Y) - RESULT
   588)          END IF
   589)       END IF
   590)    END IF
   591) 80 continue
   592) end SUBROUTINE CALERF_r8

   723)          DEL = (Y-YSQ)*(Y+YSQ)
   724)          RESULT = EXP(-YSQ*YSQ) * EXP(-DEL) * RESULT

   733)       IF (X .LT. ZERO) RESULT = -RESULT

   755) pure function shr_spfn_gamma_nonintrinsic_r8(X) result(gamma)
   756) 
   757) !cccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccc
   758) !
   759) ! 7 Feb 2013 -- S. Santos
   760) ! The following comments are from the original version. Changes have
   761) ! been made to update syntax and allow inclusion into this module.
   762) !
   763) !----------------------------------------------------------------------
   764) !
   765) ! THIS ROUTINE CALCULATES THE GAMMA FUNCTION FOR A REAL ARGUMENT X.
   766) !   COMPUTATION IS BASED ON AN ALGORITHM OUTLINED IN REFERENCE 1.
   767) !   THE PROGRAM USES RATIONAL FUNCTIONS THAT APPROXIMATE THE GAMMA
   768) !   FUNCTION TO AT LEAST 20 SIGNIFICANT DECIMAL DIGITS.  COEFFICIENTS
   769) !   FOR THE APPROXIMATION OVER THE INTERVAL (1,2) ARE UNPUBLISHED.
   770) !   THOSE FOR THE APPROXIMATION FOR X .GE. 12 ARE FROM REFERENCE 2.
   771) !   THE ACCURACY ACHIEVED DEPENDS ON THE ARITHMETIC SYSTEM, THE
   772) !   COMPILER, THE INTRINSIC FUNCTIONS, AND PROPER SELECTION OF THE
   773) !   MACHINE-DEPENDENT CONSTANTS.
   774) !
   775) !
   776) !*******************************************************************
   777) !*******************************************************************
   778) !
   779) ! EXPLANATION OF MACHINE-DEPENDENT CONSTANTS
   780) !
   781) ! BETA   - RADIX FOR THE FLOATING-POINT REPRESENTATION
   782) ! MAXEXP - THE SMALLEST POSITIVE POWER OF BETA THAT OVERFLOWS
   783) ! XBIG   - THE LARGEST ARGUMENT FOR WHICH GAMMA(X) IS REPRESENTABLE
   784) !          IN THE MACHINE, I.E., THE SOLUTION TO THE EQUATION
   785) !                  GAMMA(XBIG) = BETA**MAXEXP
   786) ! XINF   - THE LARGEST MACHINE REPRESENTABLE FLOATING-POINT NUMBER;
   787) !          APPROXIMATELY BETA**MAXEXP
   788) ! EPS    - THE SMALLEST POSITIVE FLOATING-POINT NUMBER SUCH THAT
   789) !          1.0+EPS .GT. 1.0
   790) ! XMININ - THE SMALLEST POSITIVE FLOATING-POINT NUMBER SUCH THAT
   791) !          1/XMININ IS MACHINE REPRESENTABLE
   792) !
   793) !     APPROXIMATE VALUES FOR SOME IMPORTANT MACHINES ARE:
   794) !
   795) !                            BETA       MAXEXP        XBIG
   796) !
   797) ! CRAY-1         (S.P.)        2         8191        966.961
   798) ! CYBER 180/855
   799) !   UNDER NOS    (S.P.)        2         1070        177.803
   800) ! IEEE (IBM/XT,
   801) !   SUN, ETC.)   (S.P.)        2          128        35.040
   802) ! IEEE (IBM/XT,
   803) !   SUN, ETC.)   (D.P.)        2         1024        171.624
   804) ! IBM 3033       (D.P.)       16           63        57.574
   805) ! VAX D-FORMAT   (D.P.)        2          127        34.844
   806) ! VAX G-FORMAT   (D.P.)        2         1023        171.489
   807) !
   808) !                            XINF         EPS        XMININ
   809) !
   810) ! CRAY-1         (S.P.)   5.45E+2465   7.11E-15    1.84E-2466
   811) ! CYBER 180/855
   812) !   UNDER NOS    (S.P.)   1.26E+322    3.55E-15    3.14E-294
   813) ! IEEE (IBM/XT,
   814) !   SUN, ETC.)   (S.P.)   3.40E+38     1.19E-7     1.18E-38
   815) ! IEEE (IBM/XT,
   816) !   SUN, ETC.)   (D.P.)   1.79D+308    2.22D-16    2.23D-308
   817) ! IBM 3033       (D.P.)   7.23D+75     2.22D-16    1.39D-76
   818) ! VAX D-FORMAT   (D.P.)   1.70D+38     1.39D-17    5.88D-39
   819) ! VAX G-FORMAT   (D.P.)   8.98D+307    1.11D-16    1.12D-308
   820) !

   822) !*******************************************************************
   823) !
   824) ! ERROR RETURNS
   825) !
   826) !  THE PROGRAM RETURNS THE VALUE XINF FOR SINGULARITIES OR
   827) !     WHEN OVERFLOW WOULD OCCUR.  THE COMPUTATION IS BELIEVED
   828) !     TO BE FREE OF UNDERFLOW AND OVERFLOW.
   829) !
   830) !
   831) !  INTRINSIC FUNCTIONS REQUIRED ARE:
   832) !
   833) !     INT, DBLE, EXP, LOG, REAL, SIN
   834) !
   835) !
   836) ! REFERENCES:  AN OVERVIEW OF SOFTWARE DEVELOPMENT FOR SPECIAL
   837) !              FUNCTIONS   W. J. CODY, LECTURE NOTES IN MATHEMATICS,
   838) !              506, NUMERICAL ANALYSIS DUNDEE, 1975, G. A. WATSON
   839) !              (ED.), SPRINGER VERLAG, BERLIN, 1976.
   840) !
   841) !              COMPUTER APPROXIMATIONS, HART, ET. AL., WILEY AND
   842) !              SONS, NEW YORK, 1968.
   843) !
   844) !  LATEST MODIFICATION: OCTOBER 12, 1989
   845) !
   846) !  AUTHORS: W. J. CODY AND L. STOLTZ
   847) !           APPLIED MATHEMATICS DIVISION
   848) !           ARGONNE NATIONAL LABORATORY
   849) !           ARGONNE, IL 60439
   850) !
   851) !----------------------------------------------------------------------
   852) 
   853)   real(r8), intent(in) :: x
   854)   real(r8) :: gamma
   855)   real(r8) :: fact, res, sum, xden, xnum, y, y1, ysq, z
   856) 
   857)   integer :: i, n
   858)   logical :: negative_odd
   859) 
   860)   ! log(2*pi)/2
   861)   real(r8), parameter :: logsqrt2pi = 0.9189385332046727417803297E0_r8
   862) 
   863) !----------------------------------------------------------------------
   864) !  NUMERATOR AND DENOMINATOR COEFFICIENTS FOR RATIONAL MINIMAX
   865) !     APPROXIMATION OVER (1,2).
   866) !----------------------------------------------------------------------
   867)   real(r8), parameter :: P(8) = &
   868)        (/-1.71618513886549492533811E+0_r8, 2.47656508055759199108314E+1_r8, &
   869)          -3.79804256470945635097577E+2_r8, 6.29331155312818442661052E+2_r8, &
   870)           8.66966202790413211295064E+2_r8,-3.14512729688483675254357E+4_r8, &
   871)          -3.61444134186911729807069E+4_r8, 6.64561438202405440627855E+4_r8 /)
   872)   real(r8), parameter :: Q(8) = &
   873)        (/-3.08402300119738975254353E+1_r8, 3.15350626979604161529144E+2_r8, &
   874)          -1.01515636749021914166146E+3_r8,-3.10777167157231109440444E+3_r8, &
   875)           2.25381184209801510330112E+4_r8, 4.75584627752788110767815E+3_r8, &
   876)          -1.34659959864969306392456E+5_r8,-1.15132259675553483497211E+5_r8 /)
   877) !----------------------------------------------------------------------
   878) !  COEFFICIENTS FOR MINIMAX APPROXIMATION OVER (12, INF).
   879) !----------------------------------------------------------------------
   880)   real(r8), parameter :: C(7) = &
   881)        (/-1.910444077728E-03_r8,          8.4171387781295E-04_r8, &
   882)          -5.952379913043012E-04_r8,       7.93650793500350248E-04_r8, &
   883)          -2.777777777777681622553E-03_r8, 8.333333333333333331554247E-02_r8, &
   884)           5.7083835261E-03_r8 /)
   885) 
   886)   negative_odd = .false.
   887)   fact = 1._r8
   888)   n = 0
   889)   y = x
   890)   if (y <= 0._r8) then
   891) !----------------------------------------------------------------------
   892) !  ARGUMENT IS NEGATIVE
   893) !----------------------------------------------------------------------
   894)      y = -x
   895)      y1 = aint(y)
   896)      res = y - y1
   897)      if (res /= 0._r8) then
   898)         negative_odd = (y1 /= aint(y1*0.5_r8)*2._r8)
   899)         fact = -pi/sin(pi*res)
   900)         y = y + 1._r8
   901)      else
   902)         gamma = xinfr8
   903)         return
   904)      end if
   905)   end if
   906) !----------------------------------------------------------------------
   907) !  ARGUMENT IS POSITIVE
   908) !----------------------------------------------------------------------
   909)   if (y < epsr8) then
   910) !----------------------------------------------------------------------
   911) !  ARGUMENT .LT. EPS
   912) !----------------------------------------------------------------------
   913)      if (y >= xminr8) then
   914)         res = 1._r8/y
   915)      else
   916)         gamma = xinfr8
   917)         return
   918)      end if
   919)   elseif (y < 12._r8) then
   920)      y1 = y
   921)      if (y < 1._r8) then
   922) !----------------------------------------------------------------------
   923) !  0.0 .LT. ARGUMENT .LT. 1.0
   924) !----------------------------------------------------------------------
   925)         z = y
   926)         y = y + 1._r8
   927)      else
   928) !----------------------------------------------------------------------
   929) !  1.0 .LT. ARGUMENT .LT. 12.0, REDUCE ARGUMENT IF NECESSARY
   930) !----------------------------------------------------------------------
   931)         n = int(y) - 1
   932)         y = y - real(n, r8)
   933)         z = y - 1._r8
   934)      end if
   935) !----------------------------------------------------------------------
   936) !  EVALUATE APPROXIMATION FOR 1.0 .LT. ARGUMENT .LT. 2.0
   937) !----------------------------------------------------------------------
   938)      xnum = 0._r8
   939)      xden = 1._r8
   940)      do i=1,8
   941)         xnum = (xnum+P(i))*z
   942)         xden = xden*z + Q(i)
   943)      end do
   944)      res = xnum/xden + 1._r8
   945)      if (y1 < y) then
   946) !----------------------------------------------------------------------
   947) !  ADJUST RESULT FOR CASE  0.0 .LT. ARGUMENT .LT. 1.0
   948) !----------------------------------------------------------------------
   949)         res = res/y1
   950)      elseif (y1 > y) then
   951) !----------------------------------------------------------------------
   952) !  ADJUST RESULT FOR CASE  2.0 .LT. ARGUMENT .LT. 12.0
   953) !----------------------------------------------------------------------
   954)         do i = 1,n
   955)            res = res*y
   956)            y = y + 1._r8
   957)         end do
   958)      end if
   959)   else
   960) !----------------------------------------------------------------------
   961) !  EVALUATE FOR ARGUMENT .GE. 12.0,
   962) !----------------------------------------------------------------------
   963)      if (y <= xbig_gamma) then
   964)         ysq = y*y
   965)         sum = C(7)
   966)         do i=1,6
   967)            sum = sum/ysq + C(i)
   968)         end do
   969)         sum = sum/y - y + logsqrt2pi
   970)         sum = sum + (y-0.5_r8)*log(y)
   971)         res = exp(sum)
   972)      else
   973)         gamma = xinfr8
   974)         return
   975)      end if
   976)   end if
   977) !----------------------------------------------------------------------
   978) !  FINAL ADJUSTMENTS AND RETURN
   979) !----------------------------------------------------------------------
   980)   if (negative_odd)  res = -res
   981)   if (fact /= 1._r8) res = fact/res
   982)   gamma = res
   983) ! ---------- LAST LINE OF GAMMA ----------
   984) end function shr_spfn_gamma_nonintrinsic_r8

   986) !! Incomplete Gamma function
   987) !!
   988) !! @author  Tianyi Fan
   989) !! @version August-2010
   990) real(r8) elemental function shr_spfn_igamma(a, x)
   991)   ! Upper incomplete gamma function.
   992)   ! Modified for inclusion in this module and made
   993)   ! pure elemental, September 2012
   994) 
   995)   real(r8), intent(in) ::      a
   996)   real(r8), intent(in) ::      x
   997) 
   998)   ! local variable
   999)   real(r8) :: xam, gin, s, r, t0
  1000)   integer  :: k
  1001) 
  1002) 
  1003)   if (x == 0.0_r8) then
  1004)      shr_spfn_igamma = shr_spfn_gamma(a)
  1005)      return
  1006)   end if
  1007) 
  1008)   xam = -x + a * log(x)
  1009)   
  1010)   if ((xam > 700.0_r8) .or. (a > xbig_gamma)) then
  1011)      ! Out of bounds
  1012)      ! Return "huge" value.
  1013)      shr_spfn_igamma = xinfr8
  1014)      return
  1015) 
  1016)   else if (x <= (1.0_r8 + a)) then
  1017)      s = 1.0_r8 / a
  1018)      r = s
  1019) 
  1020)      do  k = 1,60
  1021)         r = r * x / (a+k)
  1022)         s = s + r
  1023) 
  1024)         if (abs(r/s) < 1.0e-15_r8) exit
  1025)      end do
