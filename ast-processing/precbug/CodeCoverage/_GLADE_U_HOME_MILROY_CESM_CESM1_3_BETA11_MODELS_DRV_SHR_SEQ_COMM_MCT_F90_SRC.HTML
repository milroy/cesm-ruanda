   210)     integer :: gmin(num_inst_glc), gmax(num_inst_glc), gstr(num_inst_glc)
   211)     integer :: wmin(num_inst_wav), wmax(num_inst_wav), wstr(num_inst_wav)

   213)     integer :: cmin,cmax,cstr

   215)     integer, pointer :: comps(:) ! array with component ids
   216)     integer, pointer :: comms(:) ! array with mpicoms
   217)     integer :: nu, i
   218)     logical, save :: first_pass = .true.   ! 
   219) 
   220)     integer :: &
   221)          atm_ntasks, atm_rootpe, atm_pestride, atm_nthreads, &
   222)          lnd_ntasks, lnd_rootpe, lnd_pestride, lnd_nthreads, &
   223)          ice_ntasks, ice_rootpe, ice_pestride, ice_nthreads, &
   224)          glc_ntasks, glc_rootpe, glc_pestride, glc_nthreads, &
   225)          wav_ntasks, wav_rootpe, wav_pestride, wav_nthreads, &
   226)          rof_ntasks, rof_rootpe, rof_pestride, rof_nthreads, &

   230)          atm_ntasks, atm_rootpe, atm_pestride, atm_nthreads, atm_layout, &

   232)          ice_ntasks, ice_rootpe, ice_pestride, ice_nthreads, ice_layout, &
   233)          glc_ntasks, glc_rootpe, glc_pestride, glc_nthreads, glc_layout, &
   234)          wav_ntasks, wav_rootpe, wav_pestride, wav_nthreads, wav_layout, &
   235)          rof_ntasks, rof_rootpe, rof_pestride, rof_nthreads, rof_layout, &
   236)          ocn_ntasks, ocn_rootpe, ocn_pestride, ocn_nthreads, ocn_layout, &
   237)          cpl_ntasks, cpl_rootpe, cpl_pestride, cpl_nthreads 
   238)     !----------------------------------------------------------
   239) 
   240)     ! make sure this is first pass and set comms unset
   241)     if (.not. first_pass) then
   242)        write(logunit,*) trim(subname),' ERROR seq_comm_init already called '
   243)        call shr_sys_abort()
   244)     endif
   245)     first_pass = .false.
   246)     Global_Comm = Comm_in
   247) 
   248)     !! Initialize seq_comms elements
   249) 
   250)     do n = 1,ncomps
   251)        seq_comms(n)%name = 'unknown'
   252)        seq_comms(n)%suffix = ' '
   253)        seq_comms(n)%inst = 0
   254)        seq_comms(n)%set = .false.
   255)        seq_comms(n)%mpicom = MPI_COMM_NULL    ! do some initialization here 
   256)        seq_comms(n)%iam = -1
   257)        seq_comms(n)%iamroot = .false.
   258)        seq_comms(n)%npes = -1
   259)        seq_comms(n)%nthreads = -1
   260)        seq_comms(n)%gloiam = -1
   261)        seq_comms(n)%gloroot = -1
   262)        seq_comms(n)%pethreads = -1
   263)        seq_comms(n)%cplpe = -1
   264)        seq_comms(n)%cmppe = -1
   265)     enddo
   266) 
   267) 
   268)     ! Initialize MPI
   269)     ! Note that if no MPI, will call MCTs fake version
   270) 
   271)     call mpi_comm_rank(GLOBAL_COMM, mype  , ierr)

   274)     call shr_mpi_chkerr(ierr,subname//' mpi_comm_size comm_world')
   275) 
   276)     ! Initialize gloiam on all IDs
   277) 
   278)     global_mype = mype
   279)

   281)        seq_comms(n)%gloiam = mype
   282)     enddo
   283) 
   284)     ! Set ntasks, rootpe, pestride, nthreads for all components
   285) 
   286)     if (mype == 0) then
   287) 
   288)        !! Set up default atmosphere process parameters
   289) 
   290)        atm_ntasks = numpes
   291)        atm_rootpe = 0
   292)        atm_pestride = 1
   293)        atm_nthreads = 1
   294)        atm_layout = trim(layout_concurrent)
   295) 
   296)        lnd_ntasks = numpes
   297)        lnd_rootpe = 0
   298)        lnd_pestride = 1
   299)        lnd_nthreads = 1
   300)        lnd_layout = trim(layout_concurrent)
   301) 
   302)        ocn_ntasks = numpes
   303)        ocn_rootpe = 0
   304)        ocn_pestride = 1
   305)        ocn_nthreads = 1
   306)        ocn_layout = trim(layout_concurrent)
   307) 
   308)        ice_ntasks = numpes
   309)        ice_rootpe = 0
   310)        ice_pestride = 1
   311)        ice_nthreads = 1
   312)        ice_layout = trim(layout_concurrent)
   313) 
   314)        glc_ntasks = numpes
   315)        glc_rootpe = 0
   316)        glc_pestride = 1
   317)        glc_nthreads = 1
   318)        glc_layout = trim(layout_concurrent)
   319) 
   320)        rof_ntasks = numpes
   321)        rof_rootpe = 0
   322)        rof_pestride = 1
   323)        rof_nthreads = 1
   324)        rof_layout = trim(layout_concurrent)
   325) 
   326)        wav_ntasks = numpes
   327)        wav_rootpe = 0
   328)        wav_pestride = 1
   329)        wav_nthreads = 1
   330)        wav_layout = trim(layout_concurrent)
   331) 
   332)        cpl_ntasks = numpes
   333)        cpl_rootpe = 0
   334)        cpl_pestride = 1
   335)        cpl_nthreads = 1
   336) 
   337)        ! Read namelist if it exists
   338) 
   339)        nu = shr_file_getUnit()
   340)        open(nu, file=trim(nmlfile), status='old', iostat=ierr)
   341) 
   342)        if (ierr == 0) then
   343)           ierr = 1
   344)           do while( ierr > 0 )
   345)              read(nu, nml=ccsm_pes, iostat=ierr)
   346)           end do
   347)           close(nu)
   348)        end if
   349)        call shr_file_freeUnit(nu)
   350) 
   351)     end if
   352) 
   353)     !--- compute some other num_inst values
   354)        
   355)     num_inst_xao = max(num_inst_atm,num_inst_ocn)
   356)     num_inst_frc = num_inst_ice
   357)

   365)     num_inst_min = min(num_inst_min, num_inst_ocn)
   366)     num_inst_min = min(num_inst_min, num_inst_ice)
   367)     num_inst_min = min(num_inst_min, num_inst_glc)
   368)     num_inst_min = min(num_inst_min, num_inst_wav)
   369)     num_inst_min = min(num_inst_min, num_inst_rof)
   370)     num_inst_max = num_inst_atm
   371)     num_inst_max = max(num_inst_max, num_inst_lnd)
   372)     num_inst_max = max(num_inst_max, num_inst_ocn)
   373)     num_inst_max = max(num_inst_max, num_inst_ice)
   374)     num_inst_max = max(num_inst_max, num_inst_glc)
   375)     num_inst_max = max(num_inst_max, num_inst_wav)
   376)     num_inst_max = max(num_inst_max, num_inst_rof)
   377)

   379)     if (num_inst_atm /= num_inst_min .and. num_inst_atm /= num_inst_max) error_state = .true.
   380)     if (num_inst_lnd /= num_inst_min .and. num_inst_lnd /= num_inst_max) error_state = .true.
   381)     if (num_inst_ocn /= num_inst_min .and. num_inst_ocn /= num_inst_max) error_state = .true.
   382)     if (num_inst_ice /= num_inst_min .and. num_inst_ice /= num_inst_max) error_state = .true.
   383)     if (num_inst_glc /= num_inst_min .and. num_inst_glc /= num_inst_max) error_state = .true.
   384)     if (num_inst_wav /= num_inst_min .and. num_inst_wav /= num_inst_max) error_state = .true.
   385)     if (num_inst_rof /= num_inst_min .and. num_inst_rof /= num_inst_max) error_state = .true.
   386)

   388)        write(logunit,*) trim(subname),' ERROR: num_inst inconsistent'
   389)        call shr_sys_abort(trim(subname)//' ERROR: num_inst inconsistent')
   390)     endif
   391) 
   392)     ! Initialize IDs
   393) 
   394)     count = 0
   395) 
   396)     count = count + 1
   397)     GLOID = count
   398)     count = count + 1
   399)     CPLID = count
   400) 
   401)     count = count + 1
   402)     ALLATMID = count
   403)     count = count + 1
   404)     ALLLNDID = count
   405)     count = count + 1
   406)     ALLOCNID = count
   407)     count = count + 1
   408)     ALLICEID = count
   409)     count = count + 1
   410)     ALLGLCID = count
   411)     count = count + 1
   412)     ALLROFID = count
   413)     count = count + 1
   414)     ALLWAVID = count
   415) 
   416)     count = count + 1
   417)     CPLALLATMID = count
   418)     count = count + 1
   419)     CPLALLLNDID = count
   420)     count = count + 1
   421)     CPLALLOCNID = count
   422)     count = count + 1
   423)     CPLALLICEID = count
   424)     count = count + 1
   425)     CPLALLGLCID = count
   426)     count = count + 1
   427)     CPLALLROFID = count
   428)     count = count + 1
   429)     CPLALLWAVID = count
   430) 
   431)     do n = 1, num_inst_atm
   432)        count = count + 1
   433)        ATMID(n) = count
   434)        count = count + 1
   435)        CPLATMID(n) = count
   436)     end do
   437)          
   438)     do n = 1, num_inst_lnd
   439)        count = count + 1
   440)        LNDID(n) = count
   441)        count = count + 1
   442)        CPLLNDID(n) = count
   443)     end do
   444)        
   445)     do n = 1, num_inst_ocn
   446)        count = count + 1
   447)        OCNID(n) = count
   448)        count = count + 1
   449)        CPLOCNID(n) = count
   450)     end do
   451)        
   452)     do n = 1, num_inst_ice
   453)        count = count + 1
   454)        ICEID(n) = count
   455)        count = count + 1
   456)        CPLICEID(n) = count
   457)     end do
   458) 
   459)     do n = 1, num_inst_glc
   460)        count = count + 1
   461)        GLCID(n) = count
   462)        count = count + 1
   463)        CPLGLCID(n) = count
   464)     end do
   465) 
   466)     do n = 1, num_inst_rof
   467)        count = count + 1
   468)        ROFID(n) = count
   469)        count = count + 1
   470)        CPLROFID(n) = count
   471)     end do
   472) 
   473)     do n = 1, num_inst_wav
   474)        count = count + 1
   475)        WAVID(n) = count
   476)        count = count + 1
   477)        CPLWAVID(n) = count
   478)     end do
   479) 
   480)     if (count /= ncomps) then
   481)        write(logunit,*) trim(subname),' ERROR in ID count ',count,ncomps
   482)        call shr_sys_abort(trim(subname)//' ERROR in ID count')
   483)     endif
   484) 
   485)     if (mype == 0) then
   486)        !--- validation of inputs ---
   487)        ! rootpes >= 0
   488) 
   489)        error_state = .false.
   490) 
   491)        if (atm_rootpe < 0) error_state = .true.
   492)        if (lnd_rootpe < 0) error_state = .true.
   493)        if (ice_rootpe < 0) error_state = .true.
   494)        if (ocn_rootpe < 0) error_state = .true.
   495)        if (glc_rootpe < 0) error_state = .true.
   496)        if (wav_rootpe < 0) error_state = .true.
   497)        if (rof_rootpe < 0) error_state = .true.
   498)        if (cpl_rootpe < 0) error_state = .true.
   499)        
   500)        if (error_state) then
   501)           write(logunit,*) trim(subname),' ERROR: rootpes must be >= 0'
   502)           call shr_sys_abort(trim(subname)//' ERROR: rootpes >= 0')
   503)        endif
