    25) !SUBROUTINE RECONSTRUCTION------------------------------------------------CE-for FVM!
    26) ! AUTHOR: CHRISTOPH ERATH, 17.October 2011                                          !
    27) ! DESCRIPTION: controls the cubic (3rd order) reconstructions:                      !
    28) !                                                                                   !
    29) ! CALLS: fillhalo_cubic, reconstruction_cubic                                       !
    30) ! INPUT: fcube    ...  tracer values incl. the halo zone                            !
    31) !        fvm    ...  structure incl. tracer values aso                            !                                   ! 
    32) ! OUTPUT:recons   ...  has the reconstruction coefficients (5) for the 3rd order    !
    33) !                      reconstruction: dx, dy, dx^2, dy^2, dxdy                     !
    34) !-----------------------------------------------------------------------------------!
    35) subroutine reconstruction(fcube,fvm,recons)
    36)   use fvm_control_volume_mod, only: fvm_struct
    37) 
    38)   implicit none
    39)   real (kind=real_kind), dimension(1-nhc:nc+nhc, 1-nhc:nc+nhc), intent(in) :: fcube
    40)   type (fvm_struct), intent(in)                                     :: fvm
    41)   real (kind=real_kind), dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe), &
    42)                                                     intent(out)       :: recons
    43)   
    44)   real (kind=real_kind),dimension(-1:nc+2,2,2)                        :: fnewval
    45)   real (kind=real_kind), dimension(-1:nc+2,2,2)                       :: fhalo
    46)   real (kind=real_kind), dimension(0:nhe+1,2,2)                       :: fhaloex
    47)           
    48)   ! if element lies on a cube edge, recalculation the values in the halo zone
    49)   if (fvm%cubeboundary > 0) then    
    50)     call fillhalo_cubic(fcube,fvm,fnewval,fhalo,fhaloex)               
    51)   end if
    52) 
    53)   call reconstruction_cubic(fcube,fvm,fnewval,fhalo,&
    54)                             fhaloex, recons)
    55) end subroutine reconstruction

   393) !FUNCTION CUBIC_EQUISPACE_INTERP------------------------------------------CE-for FVM!
   394) ! AUTHOR: CHRISTOPH ERATH, 17.October 2011                                          !
   395) ! DESCRIPTION: apply cubic interpolation of Lagrange type on the specified array of !
   396) !        values, where all points are equally spaced.                               !
   397) !        alpha/beta coordinates                                                     !
   398) !                                                                                   !
   399) ! CALLS: none                                                                       !
   400) ! INPUT: dx ... spacing of points, alpha/beta                                       !
   401) !        x  ... X coordinate where interpolation is to be applied, alpha/beta       !
   402) !        y  ... array of 4 values = f(x + k * dx) where k = 0,1,2,3                 !
   403) ! OUTPUT/RETURN:                                                                    !
   404) !        cubic_equispace_interp ... interpolation value                             !                 
   405) !-----------------------------------------------------------------------------------!
   406) function cubic_equispace_interp(dx, x, y)
   407) 
   408)   implicit none
   409)   real (kind=real_kind)                             :: cubic_equispace_interp
   410)   real (kind=real_kind),intent(in)                  :: dx, x
   411)   real (kind=real_kind), dimension(1:4),intent(in)  :: y
   412) 
   413)   cubic_equispace_interp =                                                   &
   414)     (-y(1) / (6 * dx**3)) * (x - dx) * (x - 2 * dx) * (x - 3 * dx) +         &
   415)     ( y(2) / (2 * dx**3)) * (x) * (x - 2 * dx) * (x - 3 * dx) +              &
   416)     (-y(3) / (2 * dx**3)) * (x) * (x - dx) * (x - 3 * dx) +                  &
   417)     ( y(4) / (6 * dx**3)) * (x) * (x - dx) * (x - 2 * dx)
   418) end function cubic_equispace_interp
   419) !END FUNCTION CUBIC_EQUISPACE_INTERP--------------------------------------CE-for FVM!

   778) !SUBROUTINE RECONSTRUCT_CUBIC_ONFACE--------------------------------------CE-for FVM!
   779) ! AUTHOR: CHRISTOPH ERATH, 14.November 2011                                         !
   780) ! DESCRIPTION: construct a piecewise cubic reconstruction in the alpha/beta         !
   781) !              coordinate system for the interior element and halo zone on the same !
   782) !              face as the element                                                  !
   783) !                                                                                   !
   784) ! INPUT: fvm    ... fvm structure                                               !
   785) !        fcube    ... tracer values in the patch                                    !                                    
   786) ! INPUT/OUTPUT:                                                                     !
   787) !          recons ... reconstruction coefficients df/dx, df/dy, d^2f/dx^2, d^2/dy^2 !                 
   788) !-----------------------------------------------------------------------------------!
   789) subroutine reconstruct_cubic_onface(fvm, fcube, recons)
   790)   implicit none
   791) 
   792)   type (fvm_struct), intent(in)                               :: fvm
   793)   real (kind=real_kind),   &
   794)       dimension(1-nhc:nc+nhc, 1-nhc:nc+nhc)                     :: fcube
   795)   real (kind=real_kind), &
   796)       dimension(1:5,1-nhe:nc+nhe,1-nhe:nc+nhe), intent(inout)   :: recons
   797)   
   798)   integer  :: i, j
   799)         
   800)   do j = fvm%jy_min, fvm%jy_max-1
   801)     do i = fvm%jx_min, fvm%jx_max-1
   802)       ! df/dx
   803)       recons(1,i,j) = (- fcube(i+2,j) + 8 * fcube(i+1,j)      &
   804)                        - 8 * fcube(i-1,j) + fcube(i-2,j))/(12 * fvm%dalpha)     
   805)       ! df/dy
   806)       recons(2,i,j) = (- fcube(i,j+2) + 8 * fcube(i,j+1)      &
   807)                        - 8 * fcube(i,j-1) + fcube(i,j-2)) /( 12 * fvm%dbeta)
   808)       ! Stretching
   809)       recons(1,i,j) = recons(1,i,j) / (1 + fvm%spherecentroid(1,i,j)**2)
   810)       
   811)       recons(2,i,j) = recons(2,i,j) / (1 + fvm%spherecentroid(2,i,j)**2)
   812) 
   813)       ! d^2f/dx^2
   814)       recons(3,i,j) = (- fcube(i+2,j) + 16 * fcube(i+1,j)     &
   815)                        - 30 * fcube(i,j) + 16 * fcube(i-1,j)  &
