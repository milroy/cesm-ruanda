     6) ! !DESCRIPTION:
     7) !
     8) ! Thermo calculations after call to coupler, related to ITD:
     9) ! ice thickness redistribution, lateral growth and melting.
    10) !
    11) ! NOTE: The thermodynamic calculation is split in two for load balancing.
    12) !       First ice_therm_vertical computes vertical growth rates and coupler
    13) !       fluxes.  Then ice_therm_itd does thermodynamic calculations not
    14) !       needed for coupling.
    15) !

    37)       use ice_fileunits

    40) !

    42)       save
    43) 
    44)       real (kind=dbl_kind), parameter, private :: &
    45)          hfrazilmin = 0.05_dbl_kind ! min thickness of new frazil ice (m)

    54) ! !IROUTINE: linear_itd - ITD scheme that shifts ice among categories
    55) !
    56) ! !INTERFACE:
    57) !
    58)       subroutine linear_itd (nx_block,    ny_block,    & 
    59)                              icells, indxi, indxj,     & 
    60)                              ntrcr,       trcr_depend, & 
    61)                              aicen_init,  vicen_init,  & 
    62)                              aicen,       trcrn,       & 
    63)                              vicen,       vsnon,       & 
    64)                              eicen,       esnon,       & 
    65)                              aice,        aice0,       & 
    66)                              l_stop,                   & 
    67)                              istop,       jstop)
    68) 
    69) ! See Lipscomb, W. H.  Remapping the thickness distribution in sea
    70) !     ice models. 2001, J. Geophys. Res., Vol 106, 13989--14000.
    71) !
    72) ! Using the thermodynamic "velocities", interpolate to find the
    73) ! velocities in thickness space at the category boundaries, and
    74) ! compute the new locations of the boundaries.  Then for each
    75) ! category, compute the thickness distribution function,  g(h),
    76) ! between hL and hR, the left and right boundaries of the category.
    77) ! Assume g(h) is a linear polynomial that satisfies two conditions:
    78) !
    79) ! (1) The ice area implied by g(h) equals aicen(n).
    80) ! (2) The ice volume implied by g(h) equals aicen(n)*hicen(n).
    81) !
    82) ! Given g(h), at each boundary compute the ice area and volume lying
    83) ! between the original and new boundary locations.  Transfer area
    84) ! and volume across each boundary in the appropriate direction, thus
    85) ! restoring the original boundaries.
    86) !
    87) ! !REVISION HISTORY:
    88) !
    89) ! authors: William H. Lipscomb, LANL
    90) !          Elizabeth C. Hunke, LANL
    91) !
    92) ! !USES:
    93) !
    94)       use ice_itd, only: hin_max, hi_min, aggregate_area, shift_ice, & 
    95)                          column_sum, column_conservation_check 
    96) !
    97) ! !INPUT/OUTPUT PARAMETERS:
    98) !
    99)       integer (kind=int_kind), intent(in) :: &
   100)          nx_block, ny_block, & ! block dimensions
   101)          icells            , & ! number of grid cells with ice
   102)          ntrcr                 ! number of tracers in use
   103) 
   104)        integer (kind=int_kind), dimension (nx_block*ny_block), &
   105)          intent(in) :: &
   106)          indxi, indxj      ! compressed i/j indices
   107) 
   108)       integer (kind=int_kind), dimension (max_ntrcr), intent(in) :: &
   109)          trcr_depend ! = 0 for aicen tracers, 1 for vicen, 2 for vsnon
   110) 
   111)       real (kind=dbl_kind), dimension(nx_block,ny_block,ncat), &
   112)          intent(in) :: &
   113)          aicen_init, & ! initial ice concentration (before vertical thermo)
   114)          vicen_init    ! initial ice volume               (m)
   115) 
   116)       real (kind=dbl_kind), dimension (nx_block,ny_block,ncat), &
   117)          intent(inout) :: &
   118)          aicen  , & ! ice concentration
   119)          vicen  , & ! volume per unit area of ice      (m)
   120)          vsnon      ! volume per unit area of snow     (m)
   121) 
   122)       real (kind=dbl_kind), dimension (nx_block,ny_block,max_ntrcr,ncat), &
   123)          intent(inout) :: &
   124)          trcrn     ! ice tracers
   125) 
   126)       real (kind=dbl_kind), dimension (nx_block,ny_block,ntilyr), &
   127)          intent(inout) :: &
   128)          eicen     ! energy of melting for each ice layer (J/m^2)
   129) 
   130)       real (kind=dbl_kind), dimension (nx_block,ny_block,ntslyr), &
   131)          intent(inout) :: &
   132)          esnon     ! energy of melting for each snow layer (J/m^2)
   133) 
   134)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   135)          intent(inout) :: &
   136)          aice  , & ! concentration of ice
   137)          aice0     ! concentration of open water
   138) 
   139)       logical (kind=log_kind), intent(out) :: &
   140)          l_stop    ! if true, abort on return
   141) 
   142)       integer (kind=int_kind), intent(out) :: &
   143)          istop, jstop    ! indices of grid cell where model aborts 
   144) !
   145) !EOP
   146) !
   147)       integer (kind=int_kind) :: &
   148)          i, j         , & ! horizontal indices
   149)          n, nd        , & ! category indices
   150)          k                ! ice layer index
   151) 
   152)       real (kind=dbl_kind) :: &
   153)          slope        , & ! rate of change of dhice with hice
   154)          dh0          , & ! change in ice thickness at h = 0
   155)          da0          , & ! area melting from category 1
   156)          damax        , & ! max allowed reduction in category 1 area
   157)          etamin, etamax,& ! left and right limits of integration
   158)          x1           , & ! etamax - etamin
   159)          x2           , & ! (etamax^2 - etamin^2) / 2
   160)          x3           , & ! (etamax^3 - etamin^3) / 3
   161)          wk1, wk2         ! temporary variables
   162) 
   163)       real (kind=dbl_kind), dimension(icells,0:ncat) :: &
   164)          hbnew            ! new boundary locations
   165) 
   166)       real (kind=dbl_kind), dimension(icells) :: &
   167)          work             ! temporary work array (for hbnew)
   168) 
   169)       integer (kind=int_kind), dimension(icells) :: &
   170)          indxii, indxjj,& ! compressed i/j indices
   171)          indxij           ! compressed 1D i/j indices
   172) 
   173)       real (kind=dbl_kind), dimension(:,:), allocatable :: &
   174)          g0           , & ! constant coefficient in g(h)
   175)          g1           , & ! linear coefficient in g(h)
   176)          hL           , & ! left end of range over which g(h) > 0
   177)          hR               ! right end of range over which g(h) > 0
   178) 
   179)       real (kind=dbl_kind), dimension(icells,ncat) :: &
   180)          hicen        , & ! ice thickness for each cat     (m)
   181)          hicen_init   , & ! initial ice thickness for each cat (pre-thermo)
   182)          dhicen       , & ! thickness change for remapping (m)
   183)          daice        , & ! ice area transferred across boundary
   184)          dvice            ! ice volume transferred across boundary
   185) 
   186)       real (kind=dbl_kind), dimension(icells) :: &
   187)          vice_init, vice_final, & ! ice volume summed over categories
   188)          vsno_init, vsno_final, & ! snow volume summed over categories
   189)          eice_init, eice_final, & ! ice energy summed over categories
   190)          esno_init, esno_final    ! snow energy summed over categories
   191) 
   192)       ! NOTE: Third index of donor, daice, dvice should be ncat-1,
   193)       !       except that compilers would have trouble when ncat = 1 
   194)       integer (kind=int_kind), dimension(icells,ncat) :: &
   195)          donor            ! donor category index
   196) 
   197)       logical (kind=log_kind), dimension(icells) :: &
   198)          remap_flag       ! remap ITD if remap_flag(ij) is true
   199) 
   200)       character (len=char_len) :: &
   201)          fieldid           ! field identifier
   202) 
   203)       logical (kind=log_kind), parameter :: &
   204)          l_conservation_check = .true.   ! if true, check conservation
   205) !         l_conservation_check = .false.   ! if true, check conservation
   206)                                          ! (useful for debugging)
   207) 
   208)        integer (kind=int_kind) :: &
   209)          iflag         , & ! number of grid cells with remap_flag = .true.
   210)          ij, m             ! combined horizontal indices
   211) 
   212)       !-----------------------------------------------------------------
   213)       ! Initialize
   214)       !-----------------------------------------------------------------
   215) 
   216)       l_stop = .false.
   217)       istop = 0
   218)       jstop = 0
   219) 
   220)       hin_max(ncat) = 999.9_dbl_kind ! arbitrary big number
   221) 
   222)       !-----------------------------------------------------------------
   223)       ! Compute volume and energy sums that linear remapping should
   224)       !  conserve.
   225)       !-----------------------------------------------------------------
   226) 
   227)       if (l_conservation_check) then
   228)          call column_sum (nx_block, ny_block,       &
   229)                           icells,   indxi,   indxj, &
   230)                           ncat,                     &
   231)                           vicen,    vice_init)
   232)          call column_sum (nx_block, ny_block,       &
   233)                           icells,   indxi,   indxj, &
   234)                           ncat,                     &
   235)                           vsnon,    vsno_init)
   236)          call column_sum (nx_block, ny_block,       &
   237)                           icells,   indxi,   indxj, &
   238)                           ntilyr,                   &
   239)                           eicen,    eice_init)
   240)          call column_sum (nx_block, ny_block,       &
   241)                           icells,   indxi,   indxj, &
   242)                           ntslyr,                   &
   243)                           esnon,    esno_init)
   244)       endif
   245) 
   246)       !-----------------------------------------------------------------
   247)       ! Initialize remapping flag.
   248)       ! Remapping is done wherever remap_flag = .true.
   249)       ! In rare cases the category boundaries may shift too far for the
   250)       !  remapping algorithm to work, and remap_flag is set to .false.
   251)       ! In these cases the simpler 'rebin' subroutine will shift ice
   252)       !  between categories if needed.
   253)       !-----------------------------------------------------------------
   254) 
   255)       do ij = 1, icells
   256)          remap_flag(ij) = .true.
   257)       enddo
   258) 
   259)       !-----------------------------------------------------------------
   260)       ! Compute thickness change in each category.
   261)       !-----------------------------------------------------------------
   262) 
   263)       do n = 1, ncat
   264)          do ij = 1, icells       ! aice(i,j) > puny
   265)             i = indxi(ij)
   266)             j = indxj(ij)
   267) 
   268)             if (aicen_init(i,j,n) > puny) then
   269)                hicen_init (ij,n) = vicen_init(i,j,n) /  &
   270)                                     aicen_init(i,j,n) 
   271)             else
   272)                hicen_init(ij,n) = c0
   273)             endif               ! aicen_init > puny
   274) 
   275)             if (aicen(i,j,n) > puny) then
   276)                hicen (ij,n) = vicen(i,j,n) / aicen(i,j,n) 
   277)                dhicen(ij,n) = hicen(ij,n) - hicen_init(ij,n)
   278)             else
   279)                hicen (ij,n) = c0
   280)                dhicen(ij,n) = c0
   281)             endif               ! aicen > puny
   282) 
   283)          enddo                  ! ij
   284)       enddo                     ! n
   285) 
   286)       !-----------------------------------------------------------------
   287)       ! Compute new category boundaries, hbnew, based on changes in
   288)       ! ice thickness from vertical thermodynamics.
   289)       !-----------------------------------------------------------------
   290) 
   291)       do ij = 1, icells       ! aice(i,j) > puny
   292)          hbnew(ij,0) = hin_max(0)
   293)       enddo
   294) 
   295)       do n = 1, ncat-1
   296) 
   297) !DIR$ CONCURRENT !Cray
   298) !cdir nodep      !NEC
   299) !ocl novrec      !Fujitsu
   300)          do ij = 1, icells       ! aice(i,j) > puny
   301)             i = indxi(ij)
   302)             j = indxj(ij)
   303) 
   304)             if (hicen_init(ij,n)   > puny .and. &
   305)                 hicen_init(ij,n+1) > puny) then
   306)                  ! interpolate between adjacent category growth rates
   307)                slope = (dhicen(ij,n+1) - dhicen(ij,n)) / &
   308)                        (hicen_init(ij,n+1) - hicen_init(ij,n))
   309)                hbnew(ij,n) = hin_max(n) + dhicen(ij,n) &
   310)                             + slope * (hin_max(n) - hicen_init(ij,n))
   311)             elseif (hicen_init(ij,n) > puny) then ! hicen_init(n+1)=0
   312)                hbnew(ij,n) = hin_max(n) + dhicen(ij,n)
   313)             elseif (hicen_init(ij,n+1) > puny) then ! hicen_init(n)=0
   314)                hbnew(ij,n) = hin_max(n) + dhicen(ij,n+1)
   315)             else
   316)                hbnew(ij,n) = hin_max(n)
   317)             endif
   318) 
   319)       !-----------------------------------------------------------------
   320)       ! Check that each boundary lies between adjacent values of hicen.
   321)       ! If not, set remap_flag = .false.
   322)       ! Write diagnosis outputs if remap_flag was changed to false
   323)       !-----------------------------------------------------------------
   324) 
   325)             if (aicen(i,j,n) > puny .and. &
   326)                 hicen(ij,n) >= hbnew(ij,n)) then
   327)                remap_flag(ij) = .false.
   328) 
   329)                   write(nu_diag,*) my_task,':',i,j, &
   330)                        'ITD: hicen(n) > hbnew(n)'
   331)                   write(nu_diag,*) 'cat ',n
   332)                   write(nu_diag,*) my_task,':',i,j, &
   333)                        'hicen(n) =', hicen(ij,n)
   334)                   write(nu_diag,*) my_task,':',i,j, &
   335)                        'hbnew(n) =', hbnew(ij,n)
   336) 
   337)             elseif (aicen(i,j,n+1) > puny .and. &
   338)                     hicen(ij,n+1) <= hbnew(ij,n)) then
   339)                remap_flag(ij) = .false.
   340) 
   341)                   write(nu_diag,*) my_task,':',i,j, &
   342)                        'ITD: hicen(n+1) < hbnew(n)'
   343)                   write(nu_diag,*) 'cat ',n
   344)                   write(nu_diag,*) my_task,':',i,j, &
   345)                        'hicen(n+1) =', hicen(ij,n+1)
   346)                   write(nu_diag,*) my_task,':',i,j, &
   347)                        'hbnew(n) =', hbnew(ij,n)
   348)             endif
   349) 
   350)       !-----------------------------------------------------------------
   351)       ! Check that hbnew(n) lies between hin_max(n-1) and hin_max(n+1).
   352)       ! If not, set remap_flag = .false.
   353)       ! (In principle we could allow this, but it would make the code
   354)       ! more complicated.)
   355)       ! Write diagnosis outputs if remap_flag was changed to false
   356)       !-----------------------------------------------------------------
   357) 
   358)             if (hbnew(ij,n) > hin_max(n+1)) then
   359)                remap_flag(ij) = .false.
   360) 
   361)                   write(nu_diag,*) my_task,':',i,j, &
   362)                        'ITD hbnew(n) > hin_max(n+1)'
   363)                   write(nu_diag,*) 'cat ',n
   364)                   write(nu_diag,*) my_task,':',i,j, &
   365)                        'hbnew(n) =', hbnew(ij,n)
   366)                   write(nu_diag,*) my_task,':',i,j, &
   367)                        'hin_max(n+1) =', hin_max(n+1)
   368)             endif
   369) 
   370)             if (hbnew(ij,n) < hin_max(n-1)) then
   371)                remap_flag(ij) = .false.
   372) 
   373)                   write(nu_diag,*) my_task,':',i,j, &
   374)                        'ITD: hbnew(n) < hin_max(n-1)'
   375)                   write(nu_diag,*) 'cat ',n
   376)                   write(nu_diag,*) my_task,':',i,j, &
   377)                        'hbnew(n) =', hbnew(ij,n)
   378)                   write(nu_diag,*) my_task,':',i,j, &
   379)                        'hin_max(n-1) =', hin_max(n-1)
   380)             endif
   381) 
   382)          enddo                  ! ij
   383) 
   384)       enddo                     ! boundaries, 1 to ncat-1
   385) 
   386)       !-----------------------------------------------------------------
   387)       ! Compute hbnew(ncat)
   388)       !-----------------------------------------------------------------
   389) 
   390) !DIR$ CONCURRENT !Cray
   391) !cdir nodep      !NEC
   392) !ocl novrec      !Fujitsu
   393)       do ij = 1, icells         ! aice(i,j) > puny
   394)          i = indxi(ij)
   395)          j = indxj(ij)
   396)          if (aicen(i,j,ncat) > puny) then
   397)             hbnew(ij,ncat) = c3*hicen(ij,ncat) - c2*hbnew(ij,ncat-1)
   398)          else
   399)             hbnew(ij,ncat) = hin_max(ncat)
   400)          endif
   401)          hbnew(ij,ncat) = max(hbnew(ij,ncat),hin_max(ncat-1))
   402)       enddo
   403) 
   404)       !-----------------------------------------------------------------
   405)       ! Identify cells where the ITD is to be remapped
   406)       !-----------------------------------------------------------------
   407) 
   408)       iflag = 0
   409)       do ij = 1, icells
   410)          i = indxi(ij)
   411)          j = indxj(ij)
   412)          if (remap_flag(ij)) then
   413)             iflag = iflag + 1
   414)             indxii(iflag) = i
   415)             indxjj(iflag) = j
   416)             indxij(iflag) = ij
   417)          endif
   418)       enddo
   419) 
   420)       allocate(g0(iflag,ncat))
   421)       allocate(g1(iflag,ncat))
   422)       allocate(hL(iflag,ncat))
   423)       allocate(hR(iflag,ncat))
   424) 
   425)       !-----------------------------------------------------------------
   426)       ! Compute g(h) for category 1 at start of time step
   427)       ! (hicen = hicen_init)
   428)       !-----------------------------------------------------------------
   429) 
   430)       do ij = 1, icells       ! aice(i,j) > puny
   431)          work(ij) = hin_max(1)
   432)       enddo
   433) 
   434)       call fit_line(nx_block,       ny_block,          &
   435)                     iflag,          icells,            &
   436)                     indxii,         indxjj,    indxij, &
   437)                     aicen(:,:,1),   hicen_init(:,1),   &
   438)                     hbnew(:,0),   work     (:),        &
   439)                     g0   (:,1),   g1        (:,1),     &
   440)                     hL   (:,1),   hR        (:,1))
   441) 
   442)       !-----------------------------------------------------------------
   443)       ! Find area lost due to melting of thin (category 1) ice
   444)       !-----------------------------------------------------------------
   445) 
   446) !DIR$ CONCURRENT !Cray
   447) !cdir nodep      !NEC
   448) !ocl novrec      !Fujitsu
   449)       do ij = 1, iflag    ! remap_flag = .true.
   450)          i = indxii(ij)
   451)          j = indxjj(ij)
   452)          m = indxij(ij)
   453) 
   454)          if (aicen(i,j,1) > puny) then
   455) 
   456)             dh0 = dhicen(m,1)
   457) 
   458)             if (dh0 < c0) then   ! remove area from category 1
   459)                dh0 = min(-dh0,hin_max(1))   ! dh0 --> |dh0|
   460) 
   461)       !-----------------------------------------------------------------
   462)       ! Integrate g(1) from 0 to dh0 to estimate area melted
   463)       !-----------------------------------------------------------------
   464) 
   465)                ! right integration limit (left limit = 0)
   466)                etamax = min(dh0,hR(ij,1)) - hL(ij,1)
   467) 
   468)                if (etamax > c0) then
   469)                   x1 = etamax
   470)                   x2 = p5 * etamax*etamax
   471)                   da0 = g1(ij,1)*x2 + g0(ij,1)*x1 ! ice area removed
   472) 
   473)                ! constrain new thickness <= hicen_init
   474)                   damax = aicen(i,j,1) &
   475)                         * (c1-hicen(m,1)/hicen_init(m,1)) ! damax > 0
   476)                   da0 = min (da0, damax)
   477) 
   478)                ! remove area, conserving volume
   479)                   hicen(m,1) = hicen(m,1) &
   480)                                * aicen(i,j,1) / (aicen(i,j,1)-da0)
   481)                   aicen(i,j,1) = aicen(i,j,1) - da0
   482)                endif            ! etamax > 0
   483) 
   484)             else                ! dh0 >= 0
   485)                hbnew(m,0) = min(dh0,hin_max(1))  ! shift hbnew(0) to right
   486)             endif
   487) 
   488)          endif                  ! aicen(i,j,1) > puny
   489)       enddo                     ! ij
   490) 
   491)       !-----------------------------------------------------------------
   492)       ! Compute g(h) for each ice thickness category.
   493)       !-----------------------------------------------------------------
   494) 
   495)       do n = 1, ncat
   496)          call fit_line(nx_block,       ny_block,          &
   497)                        iflag,          icells,            &
   498)                        indxii,         indxjj,    indxij, &
   499)                        aicen(:,:,n),   hicen(:,n),        &
   500)                        hbnew(:,n-1), hbnew(:,n),          &
   501)                        g0   (:,n),   g1   (:,n),          &
   502)                        hL   (:,n),   hR   (:,n))
   503) 
   504)       enddo
   505) 
   506)       !-----------------------------------------------------------------
   507)       ! Compute area and volume to be shifted across each boundary.
   508)       !-----------------------------------------------------------------
   509) 
   510)       do n = 1, ncat
   511)          do ij = 1, icells
   512)             donor(ij,n) = 0
   513)             daice(ij,n) = c0
   514)             dvice(ij,n) = c0
   515)          enddo
   516)       enddo
   517) 
   518)       do n = 1, ncat-1
   519) 
   520) !DIR$ CONCURRENT !Cray
   521) !cdir nodep      !NEC
   522) !ocl novrec      !Fujitsu
   523)          do ij = 1, iflag   ! remap_flag = .true.
   524)             i = indxii(ij)
   525)             j = indxjj(ij)
   526)             m = indxij(ij)
   527) 
   528)             if (hbnew(m,n) > hin_max(n)) then ! transfer from n to n+1
   529) 
   530)                ! left and right integration limits in eta space
   531)                etamin = max(hin_max(n), hL(ij,n)) - hL(ij,n)
   532)                etamax = min(hbnew(m,n), hR(ij,n)) - hL(ij,n)
   533)                donor(m,n) = n
   534) 
   535)             else             ! hbnew(n) <= hin_max(n); transfer from n+1 to n
   536) 
   537)                ! left and right integration limits in eta space
   538)                etamin = c0
   539)                etamax = min(hin_max(n), hR(ij,n+1)) - hL(ij,n+1)
   540)                donor(m,n) = n+1
   541) 
   542)             endif            ! hbnew(n) > hin_max(n)
   543) 
   544)             if (etamax > etamin) then
   545)                x1  = etamax - etamin
   546)                wk1 = etamin*etamin
   547)                wk2 = etamax*etamax
   548)                x2  = p5 * (wk2 - wk1)
   549)                wk1 = wk1*etamin
   550)                wk2 = wk2*etamax
   551)                x3  = p333 * (wk2 - wk1)
   552)                nd  = donor(m,n)
   553)                daice(m,n) = g1(ij,nd)*x2 + g0(ij,nd)*x1
   554)                dvice(m,n) = g1(ij,nd)*x3 + g0(ij,nd)*x2 &
   555)                             + daice(m,n)*hL(ij,nd)
   556)             endif               ! etamax > etamin
   557) 
   558)             ! If daice or dvice is very small, shift no ice.
   559) 
   560)             nd = donor(m,n)
   561) 
   562)             if (daice(m,n) < aicen(i,j,nd)*puny) then
   563)                daice(m,n) = c0
   564)                dvice(m,n) = c0
   565)                donor(m,n) = 0
   566)             endif 
   567) 
   568)             if (dvice(m,n) < vicen(i,j,nd)*puny) then
   569)                daice(m,n) = c0
   570)                dvice(m,n) = c0
   571)                donor(m,n) = 0
   572)             endif
   573) 
   574)             ! If daice is close to aicen or dvice is close to vicen,
   575)             ! shift entire category
   576) 
   577)             if (daice(m,n) > aicen(i,j,nd)*(c1-puny)) then
   578)                daice(m,n) = aicen(i,j,nd)
   579)                dvice(m,n) = vicen(i,j,nd)
   580)             endif
   581) 
   582)             if (dvice(m,n) > vicen(i,j,nd)*(c1-puny)) then
   583)                daice(m,n) = aicen(i,j,nd)
   584)                dvice(m,n) = vicen(i,j,nd)
   585)             endif
   586) 
   587)          enddo                  ! ij
   588)       enddo                     ! boundaries, 1 to ncat-1
   589) 
   590)       deallocate(g0)
   591)       deallocate(g1)
   592)       deallocate(hL)
   593)       deallocate(hR)
   594) 
   595)       !-----------------------------------------------------------------
   596)       ! Shift ice between categories as necessary
   597)       !-----------------------------------------------------------------
   598) 
   599)       call shift_ice (nx_block, ny_block,    &
   600)                       indxi,    indxj,       &
   601)                       icells,                &
   602)                       ntrcr,    trcr_depend, &
   603)                       aicen,    trcrn,       &
   604)                       vicen,    vsnon,       &
   605)                       eicen,    esnon,       &
   606)                       hicen,    donor,       &
   607)                       daice,    dvice,       &
   608)                       l_stop,                &
   609)                       istop,    jstop)
   610) 
   611)       if (l_stop) return
   612) 
   613)       !-----------------------------------------------------------------
   614)       ! Make sure hice(i,j,1) >= minimum ice thickness hi_min.
   615)       !-----------------------------------------------------------------
   616) 
   617) !DIR$ CONCURRENT !Cray
   618) !cdir nodep      !NEC
   619) !ocl novrec      !Fujitsu
   620)       do ij = 1, iflag          ! remap_flag = .true.
   621)          i = indxii(ij)
   622)          j = indxjj(ij)
   623)          m = indxij(ij)
   624)          if (hi_min > c0 .and. &
   625)               aicen(i,j,1) > puny .and. hicen(m,1) < hi_min) then
   626)             aicen(i,j,1) = aicen(i,j,1) * hicen(m,1)/hi_min
   627)             hicen(m,1) = hi_min
   628)          endif
   629)       enddo                     ! ij
   630) 
   631)       !-----------------------------------------------------------------
   632)       ! Update fractional ice area in each grid cell.
   633)       !-----------------------------------------------------------------
   634)       call aggregate_area (nx_block, ny_block, &
   635)                            aicen,              &
   636)                            aice,     aice0)
   637) 
   638)       if (l_stop) return
   639) 
   640)       !-----------------------------------------------------------------
   641)       ! Check volume and energy conservation.
   642)       !-----------------------------------------------------------------
   643) 
   644)       if (l_conservation_check) then
   645) 
   646)          call column_sum (nx_block, ny_block,       &
   647)                           icells,   indxi,   indxj, &
   648)                           ncat,                     &
   649)                           vicen,    vice_final)
   650)          fieldid = 'vice, ITD remap'
   651)          call column_conservation_check (nx_block,  ny_block,      &
   652)                                          icells,   indxi,   indxj, &
   653)                                          fieldid,                  &
   654)                                          vice_init, vice_final,    &
   655)                                          puny,      l_stop,        &
   656)                                          istop,     jstop)
   657)          if (l_stop) return
   658) 
   659)          call column_sum (nx_block, ny_block,       &
   660)                           icells,   indxi,   indxj, &
   661)                           ncat,                     &
   662)                           vsnon,    vsno_final)
   663)          fieldid = 'vsno, ITD remap'
   664)          call column_conservation_check (nx_block,  ny_block,      &
   665)                                          icells,   indxi,   indxj, &
   666)                                          fieldid,                  &
   667)                                          vsno_init, vsno_final,    &
   668)                                          puny,      l_stop,        &
   669)                                          istop,     jstop)
   670)          if (l_stop) return
   671) 
   672)          call column_sum (nx_block, ny_block,       &
   673)                           icells,   indxi,   indxj, &
   674)                           ntilyr,                   &
   675)                           eicen,    eice_final)
   676)          fieldid = 'eice, ITD remap'
   677)          call column_conservation_check (nx_block,   ny_block,     &
   678)                                          icells,   indxi,   indxj, &
   679)                                          fieldid,                  &
   680)                                          eice_init,  eice_final,   &
   681)                                          puny*Lfresh*rhoi,         &
   682)                                          l_stop,                   &
   683)                                          istop,     jstop)
   684)          if (l_stop) return
   685) 
   686)          call column_sum (nx_block, ny_block,       &
   687)                           icells,   indxi,   indxj, &
   688)                           ntslyr,                   &
   689)                           esnon,    esno_final)
   690)          fieldid = 'esno, ITD remap'
   691)          call column_conservation_check (nx_block,   ny_block,     &
   692)                                          icells,   indxi,   indxj, &
   693)                                          fieldid,                  &
   694)                                          esno_init,  esno_final,   &
   695)                                          puny*Lfresh*rhos,         &
   696)                                          l_stop,                   &
   697)                                          istop,     jstop)
   698)          if (l_stop) return
   699) 
   700)       endif                     ! conservation check
   701) 
   702)       end subroutine linear_itd

   707) ! !IROUTINE: fit_line - fit g(h) with a line using area, volume constraints
   708) !
   709) ! !INTERFACE:
   710) !
   711) 
   712)       subroutine fit_line (nx_block, ny_block,        &
   713)                            iflag,   icells,           &
   714)                            indxii,   indxjj,  indxij, &
   715)                            aicen,    hice,            &
   716)                            hbL,      hbR,             &
   717)                            g0,       g1,              &
   718)                            hL,       hR)
   719) !
   720) ! !DESCRIPTION:
   721) !
   722) ! Fit g(h) with a line, satisfying area and volume constraints.
   723) ! To reduce roundoff errors caused by large values of g0 and g1,
   724) ! we actually compute g(eta), where eta = h - hL, and hL is the
   725) ! left boundary.
   726) !
   727) ! !REVISION HISTORY:
   728) !
   729) ! authors: William H. Lipscomb, LANL
   730) !          Elizabeth C. Hunke, LANL
   731) !
   732) ! !USES:
   733) !
   734) ! !INPUT/OUTPUT PARAMETERS:
   735) !
   736)       integer (kind=int_kind), intent(in) :: &
   737)          nx_block, ny_block, & ! block dimensions
   738)          icells            , & ! number of grid cells with ice
   739)          iflag                 ! number of grid cells with remap_flag = .true.
   740) 
   741)        integer (kind=int_kind), dimension (icells), &
   742)          intent(in) :: &
   743)          indxii, indxjj, & ! compressed i/j indices (from iflag)
   744)          indxij            ! compressed i/j indices (from icells)
   745) 
   746)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(in) :: &
   747)          aicen           ! concentration of ice
   748) 
   749)       real (kind=dbl_kind), dimension (icells), intent(in) :: &
   750)          hbL, hbR    , & ! left and right category boundaries
   751)          hice            ! ice thickness
   752) 
   753)       real (kind=dbl_kind), dimension (iflag), intent(out):: &
   754)          g0, g1      , & ! coefficients in linear equation for g(eta)
   755)          hL          , & ! min value of range over which g(h) > 0
   756)          hR              ! max value of range over which g(h) > 0
   757) !
   758) !EOP
   759) !
   760)       integer (kind=int_kind) :: &
   761)          i,j         , & ! horizontal indices
   762)          ij, m           ! combined horizontal indices
   763) 
   764)       real  (kind=dbl_kind) :: &
   765)          h13         , & ! hbL + 1/3 * (hbR - hbL)
   766)          h23         , & ! hbL + 2/3 * (hbR - hbL)
   767)          dhr         , & ! 1 / (hR - hL)
   768)          wk1, wk2        ! temporary variables
   769) 
   770)       !-----------------------------------------------------------------
   771)       ! Compute g0, g1, hL, and hR for each category to be remapped.
   772)       !-----------------------------------------------------------------
   773) 
   774)        do ij = 1, iflag
   775)          i = indxii(ij)
   776)          j = indxjj(ij)
   777)          m = indxij(ij)
   778) 
   779)          if (aicen(i,j) > puny .and. hbR(m) - hbL(m) > puny) then
   780) 
   781)          ! Initialize hL and hR
   782) 
   783)             hL(ij) = hbL(m)
   784)             hR(ij) = hbR(m)
   785) 
   786)          ! Change hL or hR if hicen(n) falls outside central third of range
   787) 
   788)             h13 = p333 * (c2*hL(ij) + hR(ij))
   789)             h23 = p333 * (hL(ij) + c2*hR(ij))
   790)             if (hice(m) < h13) then
   791)                hR(ij) = c3*hice(m) - c2*hL(ij)
   792)             elseif (hice(m) > h23) then
   793)                hL(ij) = c3*hice(m) - c2*hR(ij)
   794)             endif
   795) 
   796)          ! Compute coefficients of g(eta) = g0 + g1*eta
   797) 
   798)             dhr = c1 / (hR(ij) - hL(ij))
   799)             wk1 = c6 * aicen(i,j) * dhr
   800)             wk2 = (hice(m) - hL(ij)) * dhr
   801)             g0(ij) = wk1 * (p666 - wk2)
   802)             g1(ij) = c2*dhr * wk1 * (wk2 - p5)
   803) 
   804)          else
   805) 
   806)             g0(ij) = c0
   807)             g1(ij) = c0
   808)             hL(ij) = c0
   809)             hR(ij) = c0
   810) 
   811)          endif                  ! aicen > puny
   812) 
   813)       enddo                     ! ij
   814) 
   815)       end subroutine fit_line

   820) ! !ROUTINE: add_new_ice - add frazil ice to ice thickness distribution
   821) !
   822) ! !DESCRIPTION:
   823) !
   824) ! Given the volume of new ice grown in open water, compute its area
   825) ! and thickness and add it to the appropriate category or categories.
   826) !
   827) ! NOTE: Usually all the new ice is added to category 1.  An exception is
   828) !       made if there is no open water or if the new ice is too thick
   829) !       for category 1, in which case ice is distributed evenly over the
   830) !       entire cell.  Subroutine rebin should be called in case the ice
   831) !       thickness lies outside category bounds after new ice formation.
   832) !
   833) ! !REVISION HISTORY:
   834) !
   835) ! authors William H. Lipscomb, LANL
   836) !         Elizabeth C. Hunke, LANL
   837) !
   838) ! !INTERFACE:
   839) !
   840)       subroutine add_new_ice (nx_block,  ny_block, &
   841)                               ntrcr,     icells,   &
   842)                               indxi,     indxj,    &
   843)                               tmask,     dt,       &
   844)                               aicen,     trcrn,    &
   845)                               vicen,     eicen,    &
   846)                               aice0,     aice,     &
   847)                               frzmlt,    frazil,   &
   848)                               frz_onset, yday,     &
   849)                               fresh,     fsalt,    &
   850)                               Tf,        l_stop,   &
   851)                               istop,     jstop)
   852) !
   853) ! !USES:
   854) !
   855)       use ice_itd, only: hin_max, ilyr1, column_sum, &
   856)                          column_conservation_check
   857)       use ice_state, only: nt_Tsfc, nt_iage, nt_FY, nt_aero, &
   858)                            tr_iage, tr_FY, tr_aero, &
   859)                            nt_alvl, nt_vlvl, tr_lvl
   860)       use ice_flux, only: update_ocn_f
   861) 
   862) ! !INPUT/OUTPUT PARAMETERS:
   863) !
   864)       integer (kind=int_kind), intent(in) :: &
   865)          nx_block, ny_block, & ! block dimensions
   866)          ntrcr             , & ! number of tracers in use
   867)          icells                ! number of ice/ocean grid cells
   868) 
   869)       integer (kind=int_kind), dimension (nx_block*ny_block), &
   870)          intent(in) :: &
   871)          indxi,  indxj          ! compressed i/j indices
   872) 
   873)       logical (kind=log_kind), dimension (nx_block,ny_block), &
   874)           intent(in) :: &
   875)          tmask     ! land/boundary mask, thickness (T-cell)
   876) 
   877)       real (kind=dbl_kind), intent(in) :: &
   878)          dt        ! time step (s)
   879) 
   880)       real (kind=dbl_kind), dimension (nx_block,ny_block), intent(in) :: &
   881)          aice  , & ! total concentration of ice
   882)          frzmlt, & ! freezing/melting potential (W/m^2)
   883)          Tf        ! freezing temperature (C)
   884) 
   885)       real (kind=dbl_kind), dimension (nx_block,ny_block,ncat), &
   886)          intent(inout) :: &
   887)          aicen , & ! concentration of ice
   888)          vicen     ! volume per unit area of ice          (m)
   889) 
   890)       real (kind=dbl_kind), dimension (nx_block,ny_block,max_ntrcr,ncat), &
   891)          intent(inout) :: &
   892)          trcrn     ! ice tracers
   893)                    ! 1: surface temperature
   894) 
   895)       real (kind=dbl_kind), dimension (nx_block,ny_block,ntilyr), &
   896)          intent(inout) :: &
   897)          eicen     ! energy of melting for each ice layer (J/m^2)
   898) 
   899)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   900)          intent(inout) :: &
   901)          aice0     , & ! concentration of open water
   902)          frazil    , & ! frazil ice growth        (m/step-->cm/day)
   903)          fresh     , & ! fresh water flux to ocean (kg/m^2/s)
   904)          fsalt         ! salt flux to ocean (kg/m^2/s)
   905) 
   906)       real (kind=dbl_kind), dimension (nx_block,ny_block), &
   907)          intent(inout), optional :: &
   908)          frz_onset ! day of year that freezing begins (congel or frazil)
   909) 
   910)       real (kind=dbl_kind), intent(in), optional :: &
   911)          yday      ! day of year
   912) 
   913)       logical (kind=log_kind), intent(out) :: &
   914)          l_stop    ! if true, abort on return
   915) 
   916)       integer (kind=int_kind), intent(out) :: &
   917)          istop, jstop    ! indices of grid cell where model aborts
   918) !
   919) !EOP
   920) !
   921)       integer (kind=int_kind) :: &
   922)          i, j         , & ! horizontal indices
   923)          n            , & ! ice category index
   924)          k            , & ! ice layer index
   925)          it               ! aerosol tracer index
   926) 
   927)       real (kind=dbl_kind), dimension (icells) :: &
   928)          ai0new       , & ! area of new ice added to cat 1
   929)          vi0new       , & ! volume of new ice added to cat 1
   930)          hsurp        , & ! thickness of new ice added to each cat
   931)          vlyr             ! ice layer volume
   932) 
   933)       real (kind=dbl_kind), dimension (icells) :: &
   934)          vice_init, vice_final  ! ice volume summed over categories
   935) 
   936)       real (kind=dbl_kind) :: &
   937)          fnew         , & ! heat flx to open water for new ice (W/m^2)
   938)          hi0new       , & ! thickness of new ice
   939)          hi0max       , & ! max allowed thickness of new ice
   940)          qi0(nilyr)   , & ! frazil ice enthalpy
   941)          qi0av        , & ! mean value of qi0 for new ice (J kg-1)
   942)          vsurp        , & ! volume of new ice added to each cat
   943)          area1        , & ! starting fractional area of existing ice
   944)          vice1        , & ! starting volume of existing ice
   945)          rnilyr       , & ! real(nilyr)
   946)          dfresh       , & ! change in fresh
   947)          dfsalt       , & ! change in fsalt
   948)          vtmp
   949) 
   950)       integer (kind=int_kind) :: &
   951)          jcells, kcells     , & ! grid cell counters
   952)          ij, m                  ! combined i/j horizontal indices
   953) 
   954)       integer (kind=int_kind), dimension (icells) :: &
   955)          indxij2,  indxij3  , & ! compressed i/j indices
   956)          indxi2, indxj2     , &
   957)          indxi3, indxj3
   958) 
   959)       character (len=char_len) :: &
   960)          fieldid           ! field identifier
   961) 
   962)       l_stop = .false.
   963)       istop = 0
   964)       jstop = 0
   965) 
   966)       jcells = 0
   967)       kcells = 0
   968) 
   969)       if (ncat > 1) then
   970)          hi0max = hin_max(1)*0.9_dbl_kind  ! not too close to boundary
   971)       else
   972)          hi0max = bignum                   ! big number
   973)       endif
   974) 
   975)       ! initial ice volume in each grid cell
   976)       call column_sum (nx_block, ny_block,       &
   977)                        icells,   indxi,   indxj, &
   978)                        ncat,                     &
   979)                        vicen,    vice_init)
   980) 
   981)       !-----------------------------------------------------------------
   982)       ! Compute average enthalpy of new ice.
   983)       !
   984)       ! POP assumes new ice is fresh.  Otherwise, it would be better
   985)       ! to do something like this:
   986)       !  qi0(i,j,k) = -rhoi * (cp_ice*(Tmlt(k)-Tf(i,j))
   987)       !             + Lfresh*(1.-Tmlt(k)/Tf(i,j)) - cp_ocn*Tmlt(k))
   988)       !-----------------------------------------------------------------
   989) 
   990)       rnilyr = real(nilyr,kind=dbl_kind)
   991)       qi0av = c0
   992)       do k = 1, nilyr
   993)          qi0(k) = -rhoi*Lfresh  ! note sign convention, qi < 0
   994)          qi0av  = qi0av + qi0(k)
   995)       enddo
   996)       qi0av = qi0av/rnilyr
   997) 
   998)       !-----------------------------------------------------------------
   999)       ! Compute the volume, area, and thickness of new ice.
  1000)       !-----------------------------------------------------------------
  1001) 
  1002) !DIR$ CONCURRENT !Cray
  1003) !cdir nodep      !NEC
  1004) !ocl novrec      !Fujitsu
  1005)       do ij = 1, icells
  1006)          i = indxi(ij)
  1007)          j = indxj(ij)
  1008) 
  1009)          fnew = max (frzmlt(i,j), c0)   ! fnew > 0 iff frzmlt > 0
  1010)          vi0new(ij) = -fnew*dt / qi0av ! note sign convention, qi < 0
  1011) 
  1012)          ! increment ice volume
  1013)          vice_init(ij) = vice_init(ij) + vi0new(ij)
  1014) 
  1015)          ! history diagnostics
  1016)          frazil(i,j) = vi0new(ij)
  1017) 
  1018)          if (present(frz_onset) .and. present(yday)) then
  1019)             if (frazil(i,j) > puny .and. frz_onset(i,j) < puny) &
  1020)                  frz_onset(i,j) = yday
  1021)          endif
  1022) 
  1023)       !-----------------------------------------------------------------
  1024)       ! Update fresh water and salt fluxes.
  1025)       !
  1026)       ! NOTE: POP assumes fresh water and salt flux due to frzmlt > 0
  1027)       !       is NOT included in fluxes fresh and fsalt.
  1028)       !-----------------------------------------------------------------
  1029) 
  1030)          if (update_ocn_f) then
  1031)             dfresh = -rhoi*vi0new(ij)/dt
  1032)             dfsalt = ice_ref_salinity*p001*dfresh
  1033)             fresh(i,j)      = fresh(i,j)      + dfresh
  1034)             fsalt(i,j)      = fsalt(i,j)      + dfsalt
  1035)          endif
  1036) 
  1037)       !-----------------------------------------------------------------
  1038)       ! Decide how to distribute the new ice.
  1039)       !-----------------------------------------------------------------
  1040) 
  1041)          hsurp(ij)  = c0
  1042)          ai0new(ij) = c0
  1043) 
  1044)          if (vi0new(ij) > c0) then
  1045) 
  1046)             ! new ice area and thickness
  1047)             ! hin_max(0) < new ice thickness < hin_max(1)
  1048)             if (aice0(i,j) > puny) then
  1049)                hi0new = max(vi0new(ij)/aice0(i,j), hfrazilmin)
  1050)                if (hi0new > hi0max .and. aice0(i,j)+puny < c1) then
  1051)                   ! distribute excess volume over all categories (below)
  1052)                   hi0new = hi0max
  1053)                   ai0new(ij) = aice0(i,j)
  1054)                   vsurp       = vi0new(ij) - ai0new(ij)*hi0new
  1055)                   hsurp(ij)  = vsurp / aice(i,j)
  1056)                   vi0new(ij) = ai0new(ij)*hi0new
