   440) ! !IROUTINE:  ESMF_BaseInit - initialize a Base object
   441) !
   442) ! !INTERFACE:
   443)       subroutine ESMF_BaseInit(base, rc)
   444) !
   445) ! !ARGUMENTS:
   446)       type(ESMF_Base) :: base                 
   447)       integer, intent(out), optional :: rc     
   448) 
   449) !
   450) ! !DESCRIPTION:
   451) !     Set initial state on a Base object.
   452) !
   453) !     \begin{description}
   454) !     \item [base]
   455) !           In the Fortran interface, this must in fact be a {\tt Base}
   456) !           derived type object.  It is expected that all specialized 
   457) !           derived types will include a {\tt Base} object as the first
   458) !           entry.
   459) !     \item [{[rc]}]
   460) !           Return code; equals {\tt ESMF\_SUCCESS} if there are no errors.
   461) !
   462) !     \end{description}
   463) !
   464) !EOP
   465) 
   466)       logical :: rcpresent                          ! Return code present   
   467) 
   468) !     !Initialize return code
   469)       rcpresent = .FALSE.
   470)       if(present(rc)) then
   471)         rcpresent = .TRUE.
   472)         rc = ESMF_FAILURE
   473)       endif
   474) 
   475)       global_count = global_count + 1
   476)       base%ID = global_count
   477)       base%ref_count = 1
   478)       base%base_status = ESMF_STATE_READY
   479)       base%name = "undefined"
   480) 
   481)       if (rcpresent) rc = ESMF_SUCCESS
   482) 
   483)       end subroutine ESMF_BaseInit

   487) ! !IROUTINE:  ESMF_SetName - set the name of this object
   488) !
   489) ! !INTERFACE:
   490)       subroutine ESMF_SetName(anytype, name, namespace, rc)
   491) !
   492) ! !ARGUMENTS:
   493)       type(ESMF_Base) :: anytype                 
   494)       character (len = *), intent(in), optional :: name   
   495)       character (len = *), intent(in), optional :: namespace
   496)       integer, intent(out), optional :: rc     
   497) 
   498) !
   499) ! !DESCRIPTION:
   500) !     Associate a name with any object in the system.
   501) !
   502) !     \begin{description}
   503) !     \item [anytype]
   504) !           In the Fortran interface, this must in fact be a {\tt Base}
   505) !           derived type object.  It is expected that all specialized 
   506) !           derived types will include a {\tt Base} object as the first
   507) !           entry.
   508) !     \item [[name]]
   509) !           Object name.  An error will be returned if a duplicate name 
   510) !           is specified.  If a name is not given a unique name will be
   511) !           generated and can be queried by the {\tt ESMF_GetName} routine.
   512) !     \item [[namespace]]
   513) !           Object namespace (e.g. "Application", "Component", "Grid", etc).
   514) !           If given, the name will be checked that it is unique within
   515) !           this namespace.  If not given, the generated name will be 
   516) !           unique within this namespace.  If namespace is not specified,
   517) !           a default "global" namespace will be assumed and the same rules
   518) !           for names will be followed.
   519) !     \item [[rc]]
   520) !           Return code; equals {\tt ESMF\_SUCCESS} if there are no errors.
   521) !
   522) !     \end{description}
   523) !
   524) ! 
   525) 
   526) !
   527) !EOP
   528) ! !REQUIREMENTS:  FLD1.5, FLD1.5.3
   529)       logical :: rcpresent                          ! Return code present   
   530)       character (len = ESMF_MAXSTR) :: ournamespace ! Namespace if not given
   531)       character (len = ESMF_MAXSTR) :: defaultname  ! Name if not given
   532)       integer, save :: seqnum = 0       ! HACK - generate uniq names
   533)                                         ! but not coordinated across procs
   534) 
   535) !     !Initialize return code
   536)       rcpresent = .FALSE.
   537)       if(present(rc)) then
   538)         rcpresent = .TRUE.
   539)         rc = ESMF_FAILURE
   540)       endif
   541) 
   542) !     ! TODO: this code should generate a unique name if a name
   543) !     !   is not given.  If a namespace is given, the name has to
   544) !     !   be unique within that namespace.  Example namespaces could
   545) !     !   be: Applications, Components, Fields/Bundles, Grids.
   546) !      
   547) !     ! Construct a default namespace if one is not given
   548)       if((.not. present(namespace)) .or. (namespace .eq. "")) then
   549)           ournamespace = "global"
   550)       else
   551)           ournamespace = namespace
   552)       endif
   553) !     ! Construct a default name if one is not given
   554)       if((.not. present(name)) .or. (name .eq. "")) then
   555) 
   556)           write(defaultname, 20) trim(ournamespace), seqnum
   557) 20        format(A,I3.3)
   558)           seqnum = seqnum + 1
   559)           anytype%name = defaultname
   560)       else
   561)           anytype%name = name
   562)       endif
   563) 
   564)       if (rcpresent) rc = ESMF_SUCCESS
   565) 
   566)       end subroutine ESMF_SetName
