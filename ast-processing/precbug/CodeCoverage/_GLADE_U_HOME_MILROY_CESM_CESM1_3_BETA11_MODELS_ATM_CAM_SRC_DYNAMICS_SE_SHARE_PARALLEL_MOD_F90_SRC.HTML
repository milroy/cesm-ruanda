   241)            (my_name(1:2) .eq. 'bs') .or.  &    ! BlueSky

   263) 	   endif

   266) 	      tarray(:) = 0
   267) 	      tarray(FrameNumber+1) = 1

   272) 	      allocate(FrameIndex(numFrames))

   279) 		    ii=ii+1
   280) 	         endif
   281) 	      enddo
   282) 	   endif
   283) 
   284)            FrameCount(:)=FrameCount(:)/nmpi_per_node

   286)            ! We are not running on more than one frame 
   287) 	   ! ==========================================

   289) 
   290) 	endif
   291) 
   292)          write(iulog,*)'initmp: mpi task ',par%rank,': ',nmpi_per_node,' task(s) on node ',my_name(1:namelen),  &

   294) #endif
   295) 	 if(PartitionForFrames) then

   298) 	 endif
   299) 
   300)             
   301) 
   302)          deallocate(the_names)
   303) 	 
   304) #else

   306)          par%rank          =  0
   307)          par%nprocs        =  1
   308)          par%comm          = -1

   377) ! split:
   378) !
   379) ! splits the message passing world into components
   380) ! and returns a new parallel structure for the
   381) ! component resident at this process, i.e. lcl_component
   382) ! =========================================================

   384)        function split(par,leader,lcl_component) result(newpar)
   385) 
   386)          type (parallel_t)  :: par
   387)          type (parallel_t)  :: newpar
   388)          integer            :: lcl_component
   389)          integer            :: leader(0:ncomponents-1)
   390) 
   391) #ifdef _MPI
   392)          integer ierr
   393)          integer info
   394)          integer            :: key
   395) #endif
   396) 
   397)          lcl_component=ncomponents-1
   398)          do while(leader(lcl_component) > par%rank)
   399)             lcl_component=lcl_component-1
   400)          end do
   401) 
   402) #ifdef _MPI
   403)          key=par%rank   ! simplest key for most cases
   404) 
   405)          call MPI_comm_split(par%comm, lcl_component, key, newpar%comm,ierr);
   406) 
   407)          call MPI_comm_rank(newpar%comm,newpar%rank,info)
   408)          call MPI_comm_size(newpar%comm,newpar%nprocs,info)
   409)          newpar%root=0
   410) #else
   411)          newpar%comm=-1
   412)          newpar%root=0
   413)          newpar%rank=0
   414)          newpar%nprocs=1
   415) #endif
   416) 
   417)        end function split

   420) ! connect:
   421) !
   422) ! connects this MPI component to all others by constructing
   423) ! intercommunicator array and storing it in the local parallel
   424) ! structure lcl_par. Connect assumes you have called split
   425) ! to create the lcl_par structure.
   426) !
   427) ! =========================================================
   428) 
   429)        subroutine connect(gbl_par, lcl_par, lcl_component, leader)
   430) 
   431)          type (parallel_t) :: gbl_par
   432)          type (parallel_t) :: lcl_par
   433)          integer           :: lcl_component
   434)          integer           :: leader(0:ncomponents-1) ! leader rank in bridge group
   435) 
   436) #ifdef _MPI
   437)          integer tag
   438)          integer i
   439)          integer ierr
   440) 
   441)          do i=0,ncomponents-1
   442) 
   443)             if (i > lcl_component) then
   444)                tag=ncomponents*lcl_component + i
   445)             else
   446)                tag=ncomponents*i+lcl_component
   447)             end if
   448)             if (i .ne. lcl_component) then
   449) #ifdef _DEBUG
   450)                write(iulog,10)lcl_component,
   451)      &                    gbl_par%rank,
   452)      &                    leader(lcl_component),
   453)      &                    leader(i),
   454)                           tag
   455)  10            format("component=",i4, 
   456)      &                " gbl rank =",i4,   
   457)      &                " lcl leader=",i4,  
   458)      &                " rem leader=",i4, 
   459)      &                " tag=",i4)
   460) #endif
   461)                call MPI_Intercomm_create(lcl_par%comm, &
   462) 	                                 lcl_par%root, &
   463)                                          gbl_par%comm, &
   464)                                          leader(i),    &
   465)                                          tag,          &
   466)                                          lcl_par%intercomm(i),  &
   467)                                          ierr)  
   468)             end if
   469)          end do     
   470) #endif
