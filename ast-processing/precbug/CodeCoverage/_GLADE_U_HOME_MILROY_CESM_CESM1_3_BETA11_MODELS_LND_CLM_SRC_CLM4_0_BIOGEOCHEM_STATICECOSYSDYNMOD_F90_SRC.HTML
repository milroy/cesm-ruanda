   103) ! !INTERFACE:
   104)   subroutine EcosystemDyn(lbp, ubp, num_nolakep, filter_nolakep, doalb)
   105) !
   106) ! !DESCRIPTION:
   107) ! Ecosystem dynamics: phenology, vegetation
   108) ! Calculates leaf areas (tlai, elai),  stem areas (tsai, esai) and
   109) ! height (htop).
   110) !
   111) ! !USES:
   112)     use clmtype
   113)     use pftvarcon, only : noveg, nc3crop, nbrdlf_dcd_brl_shrub
   114) !
   115) ! !ARGUMENTS:
   116)     implicit none
   117)     integer, intent(in) :: lbp, ubp                    ! pft bounds
   118)     integer, intent(in) :: num_nolakep                 ! number of column non-lake points in pft filter
   119)     integer, intent(in) :: filter_nolakep(ubp-lbp+1)   ! pft filter for non-lake points
   120)     logical, intent(in) :: doalb                       ! true = surface albedo calculation time step
   121) !
   122) ! !CALLED FROM:
   123) !
   124) ! !REVISION HISTORY:
   125) ! Author: Gordon Bonan
   126) ! 2/1/02, Peter Thornton: Migrated to new data structure.
   127) ! 2/29/08, David Lawrence: revised snow burial fraction for short vegetation   
   128) !
   129) ! !LOCAL VARIABLES:
   130) !
   131) ! local pointers to implicit in arguments
   132) !
   133)     integer , pointer :: pcolumn(:)  ! column index associated with each pft
   134)     real(r8), pointer :: snowdp(:)   ! snow height (m)
   135)     integer , pointer :: ivt(:)      ! pft vegetation type
   136) !
   137) ! local pointers to implicit out arguments
   138) !
   139)     real(r8), pointer :: tlai(:)     ! one-sided leaf area index, no burying by snow
   140)     real(r8), pointer :: tsai(:)     ! one-sided stem area index, no burying by snow
   141)     real(r8), pointer :: htop(:)     ! canopy top (m)
   142)     real(r8), pointer :: hbot(:)     ! canopy bottom (m)
   143)     real(r8), pointer :: elai(:)     ! one-sided leaf area index with burying by snow
   144)     real(r8), pointer :: esai(:)     ! one-sided stem area index with burying by snow
   145)     integer , pointer :: frac_veg_nosno_alb(:) ! frac of vegetation not covered by snow [-]
   146) !
   147) !
   148) ! !OTHER LOCAL VARIABLES:
   149) !EOP
   150) !
   151)     integer  :: fp,p,c   ! indices
   152)     real(r8) :: ol       ! thickness of canopy layer covered by snow (m)
   153)     real(r8) :: fb       ! fraction of canopy layer covered by snow
   154) !-----------------------------------------------------------------------
   155) 
   156)     if (doalb) then
   157) 
   158)        ! Assign local pointers to derived type scalar members (column-level)
   159) 
   160)        snowdp  => cps%snowdp
   161) 
   162)        ! Assign local pointers to derived type scalar members (pftlevel)
   163) 
   164)        pcolumn => pft%column
   165)        tlai    => pps%tlai
   166)        tsai    => pps%tsai
   167)        elai    => pps%elai
   168)        esai    => pps%esai
   169)        htop    => pps%htop
   170)        hbot    => pps%hbot
   171)        frac_veg_nosno_alb => pps%frac_veg_nosno_alb
   172)        ivt     => pft%itype
   173) 
   174)        do fp = 1, num_nolakep
   175)           p = filter_nolakep(fp)
   176)           c = pcolumn(p)
   177) 
   178)           ! need to update elai and esai only every albedo time step so do not
   179)           ! have any inconsistency in lai and sai between SurfaceAlbedo calls (i.e.,
   180)           ! if albedos are not done every time step).
   181)           ! leaf phenology
   182)           ! Set leaf and stem areas based on day of year
   183)           ! Interpolate leaf area index, stem area index, and vegetation heights
   184)           ! between two monthly
   185)           ! The weights below (timwt(1) and timwt(2)) were obtained by a call to
   186)           ! routine InterpMonthlyVeg in subroutine NCARlsm.
   187)           !                 Field   Monthly Values
   188)           !                -------------------------
   189)           ! leaf area index LAI  <- mlai1 and mlai2
   190)           ! leaf area index SAI  <- msai1 and msai2
   191)           ! top height      HTOP <- mhvt1 and mhvt2
   192)           ! bottom height   HBOT <- mhvb1 and mhvb2
   193) 
   194)           tlai(p) = timwt(1)*mlai2t(p,1) + timwt(2)*mlai2t(p,2)
   195)           tsai(p) = timwt(1)*msai2t(p,1) + timwt(2)*msai2t(p,2)
   196)           htop(p) = timwt(1)*mhvt2t(p,1) + timwt(2)*mhvt2t(p,2)
   197)           hbot(p) = timwt(1)*mhvb2t(p,1) + timwt(2)*mhvb2t(p,2)
   198) 
   199)           ! adjust lai and sai for burying by snow. if exposed lai and sai
   200)           ! are less than 0.05, set equal to zero to prevent numerical
   201)           ! problems associated with very small lai and sai.
   202) 
   203)           ! snow burial fraction for short vegetation (e.g. grasses) as in
   204)           ! Wang and Zeng, 2007. 
   205) 
   206)           if (ivt(p) > noveg .and. ivt(p) <= nbrdlf_dcd_brl_shrub ) then
   207)              ol = min( max(snowdp(c)-hbot(p), 0._r8), htop(p)-hbot(p))
   208)              fb = 1._r8 - ol / max(1.e-06_r8, htop(p)-hbot(p))
   209)           else
   210)              fb = 1._r8 - max(min(snowdp(c),0.2_r8),0._r8)/0.2_r8   ! 0.2m is assumed
   211)                   !depth of snow required for complete burial of grasses
   212)           endif
   213) 
   214)           elai(p) = max(tlai(p)*fb, 0.0_r8)
   215)           esai(p) = max(tsai(p)*fb, 0.0_r8)
   216)           if (elai(p) < 0.05_r8) elai(p) = 0._r8
   217)           if (esai(p) < 0.05_r8) esai(p) = 0._r8
   218) 
   219)           ! Fraction of vegetation free of snow
   220) 
   221)           if ((elai(p) + esai(p)) >= 0.05_r8) then
   222)              frac_veg_nosno_alb(p) = 1
   223)           else
   224)              frac_veg_nosno_alb(p) = 0
   225)           end if
   226) 
   227)        end do ! end of pft loop
   228) 
   229)     end if  !end of if-doalb block
   230) 
   231)   end subroutine EcosystemDyn

   238) ! !INTERFACE:
   239)   subroutine interpMonthlyVeg ()
   240) !
   241) ! !DESCRIPTION:
   242) ! Determine if 2 new months of data are to be read.
   243) !
   244) ! !USES:
   245)     use clm_varctl      , only : fsurdat
   246)     use clm_time_manager, only : get_curr_date, get_step_size, &
   247)                                  get_perp_date, is_perpetual, get_nstep
   248) !
   249) ! !ARGUMENTS:
   250)     implicit none
   251) !
   252) ! !REVISION HISTORY:
   253) ! Created by Mariana Vertenstein
   254) !
   255) !
   256) ! !LOCAL VARIABLES:
   257) !EOP
   258)     integer :: kyr         ! year (0, ...) for nstep+1
   259)     integer :: kmo         ! month (1, ..., 12)
   260)     integer :: kda         ! day of month (1, ..., 31)
   261)     integer :: ksec        ! seconds into current date for nstep+1
   262)     real(r8):: dtime       ! land model time step (sec)
   263)     real(r8):: t           ! a fraction: kda/ndaypm
   264)     integer :: it(2)       ! month 1 and month 2 (step 1)
   265)     integer :: months(2)   ! months to be interpolated (1 to 12)
   266)     integer, dimension(12) :: ndaypm= &
   267)          (/31,28,31,30,31,30,31,31,30,31,30,31/) !days per month
   268) !-----------------------------------------------------------------------
   269) 
   270)     dtime = get_step_size()
   271) 
   272)     if ( is_perpetual() ) then
   273)        call get_perp_date(kyr, kmo, kda, ksec, offset=int(dtime))
   274)     else
   275)        call get_curr_date(kyr, kmo, kda, ksec, offset=int(dtime))
   276)     end if
   277) 
   278)     t = (kda-0.5_r8) / ndaypm(kmo)
   279)     it(1) = t + 0.5_r8
   280)     it(2) = it(1) + 1
   281)     months(1) = kmo + it(1) - 1
   282)     months(2) = kmo + it(2) - 1
   283)     if (months(1) <  1) months(1) = 12
   284)     if (months(2) > 12) months(2) = 1
   285)     timwt(1) = (it(1)+0.5_r8) - t
   286)     timwt(2) = 1._r8-timwt(1)
   287) 
   288)     if (InterpMonths1 /= months(1)) then
   289)        if (masterproc) then
   290)           write(iulog,*) 'Attempting to read monthly vegetation data .....'
   291)           write(iulog,*) 'nstep = ',get_nstep(),' month = ',kmo,' day = ',kda
   292)        end if
   293)        call t_startf('readMonthlyVeg')
   294)        call readMonthlyVegetation (fsurdat, months)
   295)        InterpMonths1 = months(1)
   296)        call t_stopf('readMonthlyVeg')
   297)     end if
   298) 
   299)   end subroutine interpMonthlyVeg

   305)   subroutine readAnnualVegetation ( )
   306) 
   307)     use clmtype
   308)     use clm_varpar  , only : numpft
   309)     use pftvarcon   , only : noveg
   310)     use decompMod   , only : get_proc_bounds
   311)     use domainMod   , only : ldomain
   312)     use fileutils   , only : getfil
   313)     use clm_varctl  , only : fsurdat
   314)     use shr_scam_mod, only : shr_scam_getCloseLatLon
   315) 
   316)     implicit none
   317) 
   318)     ! local vars
   319) 
   320)     type(file_desc_t) :: ncid             ! netcdf id
   321)     real(r8), pointer :: annlai(:,:)      ! 12 months of monthly lai from input data set 
   322)     real(r8), pointer :: mlai(:,:)        ! lai read from input files
   323)     integer :: ier                        ! error code
   324)     character(len=256) :: locfn           ! local file name
   325)     integer :: g,k,l,m,n,p,ivt            ! indices
   326)     integer :: ni,nj,ns                   ! indices
   327)     integer :: dimid,varid                ! input netCDF id's
   328)     integer :: ntim                       ! number of input data time samples
   329)     integer :: nlon_i                     ! number of input data longitudes
   330)     integer :: nlat_i                     ! number of input data latitudes
   331)     integer :: npft_i                     ! number of input data pft types
   332)     integer :: begp,endp                  ! beg and end local p index
   333)     integer :: begg,endg                  ! beg and end local g index
   334)     integer :: closelatidx,closelonidx    ! single column vars
   335)     real(r8):: closelat,closelon          ! single column vars
   336)     logical :: isgrid2d                   ! true => file is 2d
   337)     character(len=32) :: subname = 'readAnnualVegetation'
   338) 
   339)     annlai    => pps%annlai
   340) 
   341)     ! Determine necessary indices
   342) 
   343)     call get_proc_bounds(begg=begg,endg=endg,begp=begp,endp=endp)
   344) 
   345)     allocate(mlai(begg:endg,0:numpft), stat=ier)
   346)     if (ier /= 0) then
   347)        write(iulog,*)subname, 'allocation error '; call endrun()
   348)     end if
   349) 
   350)     if (masterproc) then
   351)        write (iulog,*) 'Attempting to read annual vegetation data .....'
   352)     end if
   353) 
   354)     call getfil(fsurdat, locfn, 0)
   355)     call ncd_pio_openfile (ncid, trim(locfn), 0)
   356)     call ncd_inqfdims (ncid, isgrid2d, ni, nj, ns)
   357) 
   358)     if (ldomain%ns /= ns .or. ldomain%ni /= ni .or. ldomain%nj /= nj) then
   359)        write(iulog,*)trim(subname), 'ldomain and input file do not match dims '
   360)        write(iulog,*)trim(subname), 'ldomain%ni,ni,= ',ldomain%ni,ni
   361)        write(iulog,*)trim(subname), 'ldomain%nj,nj,= ',ldomain%nj,nj
   362)        write(iulog,*)trim(subname), 'ldomain%ns,ns,= ',ldomain%ns,ns
   363)        call endrun()
   364)     end if
   365)     call check_dim(ncid, 'lsmpft', numpft+1)
   366) 
   367)     if (single_column) then
   368)        call shr_scam_getCloseLatLon(locfn, scmlat, scmlon, &
   369)             closelat, closelon, closelatidx, closelonidx)
   370)     endif
   371) 
   372)     do k=1,12   !! loop over months and read vegetated data
   373) 
   374)        call ncd_io(ncid=ncid, varname='MONTHLY_LAI', flag='read', data=mlai, &
   375)             dim1name=grlnd, nt=k)
   376) 
   377)        !! store data directly in clmtype structure
   378)        !! only vegetated pfts have nonzero values
   379)        !! Assign lai/sai/hgtt/hgtb to the top [maxpatch_pft] pfts
   380)        !! as determined in subroutine surfrd
   381) 
   382)        do p = begp,endp
   383)           g = pft%gridcell(p)
   384)           ivt = pft%itype(p)
   385)           if (ivt /= noveg) then     !! vegetated pft
   386)              do l = 0, numpft
   387)                 if (l == ivt) then
   388)                    annlai(k,p) = mlai(g,l)
   389)                 end if
   390)              end do
   391)           else                       !! non-vegetated pft
   392)              annlai(k,p) = 0._r8
   393)           end if
   394)        end do   ! end of loop over pfts  
   395) 
   396)     enddo ! months loop
   397) 
   398)     call ncd_pio_closefile(ncid)
   399) 
   400)     deallocate(mlai)
   401) 
   402)   endsubroutine readAnnualVegetation

   409) ! !INTERFACE:
   410)   subroutine readMonthlyVegetation (fveg, months)
   411) !
   412) ! !DESCRIPTION:
   413) ! Read monthly vegetation data for two consec. months.
   414) !
   415) ! !USES:
   416)     use clmtype
   417)     use decompMod   , only : get_proc_bounds
   418)     use clm_varpar  , only : numpft
   419)     use pftvarcon   , only : noveg
   420)     use fileutils   , only : getfil
   421)     use spmdMod     , only : masterproc, mpicom, MPI_REAL8, MPI_INTEGER
   422)     use shr_scam_mod, only : shr_scam_getCloseLatLon
   423)     use clm_time_manager, only : get_nstep
   424)     use netcdf
   425) !
   426) ! !ARGUMENTS:
   427)     implicit none
   428) 
   429)     character(len=*), intent(in) :: fveg  ! file with monthly vegetation data
   430)     integer, intent(in) :: months(2)      ! months to be interpolated (1 to 12)
   431) !
   432) ! !REVISION HISTORY:
   433) ! Created by Sam Levis
   434) !
   435) !
   436) ! !LOCAL VARIABLES:
   437) !EOP
   438)     character(len=256) :: locfn           ! local file name
   439)     type(file_desc_t)  :: ncid            ! netcdf id
   440)     integer :: g,n,k,l,m,p,ivt,ni,nj,ns   ! indices
   441)     integer :: dimid,varid                ! input netCDF id's
   442)     integer :: ntim                       ! number of input data time samples
   443)     integer :: nlon_i                     ! number of input data longitudes
   444)     integer :: nlat_i                     ! number of input data latitudes
   445)     integer :: npft_i                     ! number of input data pft types
   446)     integer :: begp,endp                  ! beg and end local p index
   447)     integer :: begg,endg                  ! beg and end local g index
   448)     integer :: ier                        ! error code
   449)     integer :: closelatidx,closelonidx
   450)     real(r8):: closelat,closelon
   451)     logical :: readvar
   452)     real(r8), pointer :: mlai(:,:)        ! lai read from input files
   453)     real(r8), pointer :: msai(:,:)        ! sai read from input files
   454)     real(r8), pointer :: mhgtt(:,:)       ! top vegetation height
   455)     real(r8), pointer :: mhgtb(:,:)       ! bottom vegetation height
   456)     real(r8), pointer :: mlaidiff(:)      ! difference between lai month one and month two
   457)     character(len=32) :: subname = 'readMonthlyVegetation'
   458) !-----------------------------------------------------------------------
   459) 
   460)     ! Determine necessary indices
   461) 
   462)     call get_proc_bounds(begg=begg,endg=endg,begp=begp,endp=endp)
   463) 
   464)     allocate(mlai(begg:endg,0:numpft), &
   465)              msai(begg:endg,0:numpft), &
   466)              mhgtt(begg:endg,0:numpft), &
   467)              mhgtb(begg:endg,0:numpft), &
   468)              stat=ier)
   469)     if (ier /= 0) then
   470)        write(iulog,*)subname, 'allocation big error '; call endrun()
   471)     end if
   472) 
   473)     ! ----------------------------------------------------------------------
   474)     ! Open monthly vegetation file
   475)     ! Read data and convert from gridcell to pft data
   476)     ! ----------------------------------------------------------------------
   477)    
   478)     call getfil(fveg, locfn, 0)
   479)     call ncd_pio_openfile (ncid, trim(locfn), 0)
   480)     
   481)     if (single_column) then
   482)        call shr_scam_getCloseLatLon (ncid, scmlat, scmlon, closelat, closelon,&
   483)             closelatidx, closelonidx)
   484)     endif
   485)     
   486)     do k=1,2   !loop over months and read vegetated data
   487) 
   488)        call ncd_io(ncid=ncid, varname='MONTHLY_LAI', flag='read', data=mlai, dim1name=grlnd, &
   489)             nt=months(k), readvar=readvar)
   490)        if (.not. readvar) call endrun( trim(subname)//' ERROR: MONTHLY_LAI NOT on fveg file' )
   491) 
   492)        call ncd_io(ncid=ncid, varname='MONTHLY_SAI', flag='read', data=msai, dim1name=grlnd, &
   493)             nt=months(k), readvar=readvar)
   494)        if (.not. readvar) call endrun( trim(subname)//' ERROR: MONTHLY_SAI NOT on fveg file' )
   495) 
   496)        call ncd_io(ncid=ncid, varname='MONTHLY_HEIGHT_TOP', flag='read', data=mhgtt, dim1name=grlnd, &
   497)             nt=months(k), readvar=readvar)
   498)        if (.not. readvar) call endrun( trim(subname)//' ERROR: MONTHLY_HEIGHT_TOP NOT on fveg file' )
   499) 
   500)        call ncd_io(ncid=ncid, varname='MONTHLY_HEIGHT_BOT', flag='read', data=mhgtb, dim1name=grlnd, &
   501)             nt=months(k), readvar=readvar)
   502)        if (.not. readvar) call endrun( trim(subname)//' ERROR: MONTHLY_HEIGHT_TOP NOT on fveg file' )
   503)
