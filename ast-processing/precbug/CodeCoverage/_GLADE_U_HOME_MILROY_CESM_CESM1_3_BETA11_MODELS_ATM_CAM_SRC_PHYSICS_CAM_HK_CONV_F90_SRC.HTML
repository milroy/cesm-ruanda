   185) subroutine cmfmca(lchnk   ,ncol    , &
   186)                   nstep   ,ztodt     ,pmid    ,pdel    , &
   187)                   rpdel   ,zm      ,tpert   ,qpert   ,phis    , &
   188)                   pblh    ,t       ,q       ,cmfdt   ,dq      , &
   189)                   cmfmc   ,cmfdqr  ,cmfsl   ,cmflq   ,precc   , &
   190)                   qc      ,cnt     ,cnb     ,icwmr   ,rliq    , & 
   191)                   pmiddry ,pdeldry ,rpdeldry)
   192) !----------------------------------------------------------------------- 
   193) ! 
   194) ! Purpose: 
   195) ! Moist convective mass flux procedure:
   196) ! 
   197) ! Method: 
   198) ! If stratification is unstable to nonentraining parcel ascent,
   199) ! complete an adjustment making successive use of a simple cloud model
   200) ! consisting of three layers (sometimes referred to as a triplet)
   201) !
   202) ! Code generalized to allow specification of parcel ("updraft")
   203) ! properties, as well as convective transport of an arbitrary
   204) ! number of passive constituents (see q array).  The code
   205) ! is written so the water vapor field is passed independently
   206) ! in the calling list from the block of other transported
   207) ! constituents, even though as currently designed, it is the
   208) ! first component in the constituents field.
   209) ! 
   210) ! Author: J. Hack
   211) !
   212) ! BAB: changed code to report tendencies in cmfdt and dq, instead of
   213) ! updating profiles. Cmfdq contains water only, made it a local variable
   214) ! made dq (all constituents) the argument.
   215) ! 
   216) !-----------------------------------------------------------------------
   217) 
   218) !#######################################################################
   219) !#                                                                     #
   220) !# Debugging blocks are marked this way for easy identification        #
   221) !#                                                                     #
   222) !#######################################################################
   223)    use constituents,  only: pcnst
   224)    use constituents,    only: cnst_get_type_byind
   225)    use ppgrid,    only: pcols, pver, pverp
   226)    use phys_grid, only: get_lat_all_p, get_lon_all_p
   227)    use wv_saturation, only: qsat
   228) 
   229)    real(r8) ssfac               ! supersaturation bound (detrained air)
   230)    parameter (ssfac = 1.001_r8)
   231) 
   232) !------------------------------Arguments--------------------------------
   233) !
   234) ! Input arguments
   235) !
   236)    integer, intent(in) :: lchnk                ! chunk identifier
   237)    integer, intent(in) :: ncol                 ! number of atmospheric columns
   238)    integer, intent(in) :: nstep                ! current time step index
   239) 
   240)    real(r8), intent(in) :: ztodt               ! 2 delta-t (seconds)
   241)    real(r8), intent(in) :: pmid(pcols,pver)    ! pressure
   242)    real(r8), intent(in) :: pdel(pcols,pver)    ! delta-p
   243)    real(r8), intent(in) :: pmiddry(pcols,pver)    ! pressure
   244)    real(r8), intent(in) :: pdeldry(pcols,pver)    ! delta-p
   245)    real(r8), intent(in) :: rpdel(pcols,pver)   ! 1./pdel
   246)    real(r8), intent(in) :: rpdeldry(pcols,pver)   ! 1./pdel
   247)    real(r8), intent(in) :: zm(pcols,pver)      ! height abv sfc at midpoints
   248)    real(r8), intent(in) :: tpert(pcols)        ! PBL perturbation theta
   249)    real(r8), intent(in) :: qpert(pcols,pcnst)  ! PBL perturbation specific humidity
   250)    real(r8), intent(in) :: phis(pcols)         ! surface geopotential
   251)    real(r8), intent(in) :: pblh(pcols)         ! PBL height (provided by PBL routine)
   252)    real(r8), intent(in) :: t(pcols,pver)       ! temperature (t bar)
   253)    real(r8), intent(in) :: q(pcols,pver,pcnst) ! specific humidity (sh bar)
   254) !
   255) ! Output arguments
   256) !
   257)    real(r8), intent(out) :: cmfdt(pcols,pver)   ! dt/dt due to moist convection
   258)    real(r8), intent(out) :: cmfmc(pcols,pverp)  ! moist convection cloud mass flux
   259)    real(r8), intent(out) :: cmfdqr(pcols,pver)  ! dq/dt due to convective rainout
   260)    real(r8), intent(out) :: cmfsl(pcols,pver )  ! convective lw static energy flux
   261)    real(r8), intent(out) :: cmflq(pcols,pver )  ! convective total water flux
   262)    real(r8), intent(out) :: precc(pcols)        ! convective precipitation rate
   263) ! JJH mod to explicitly export cloud water
   264)    real(r8), intent(out) :: qc(pcols,pver)      ! dq/dt due to export of cloud water
   265)    real(r8), intent(out) :: cnt(pcols)          ! top level of convective activity
   266)    real(r8), intent(out) :: cnb(pcols)          ! bottom level of convective activity
   267)    real(r8), intent(out) :: dq(pcols,pver,pcnst) ! constituent tendencies
   268)    real(r8), intent(out) :: icwmr(pcols,pver)
   269)    real(r8), intent(out) :: rliq(pcols) 
   270) !
   271) !---------------------------Local workspace-----------------------------
   272) !
   273)    real(r8) pm(pcols,pver)    ! pressure
   274)    real(r8) pd(pcols,pver)    ! delta-p
   275)    real(r8) rpd(pcols,pver)   ! 1./pdel
   276) 
   277)    real(r8) cmfdq(pcols,pver)   ! dq/dt due to moist convection
   278)    real(r8) gam(pcols,pver)     ! 1/cp (d(qsat)/dT)
   279)    real(r8) sb(pcols,pver)      ! dry static energy (s bar)
   280)    real(r8) hb(pcols,pver)      ! moist static energy (h bar)
   281)    real(r8) shbs(pcols,pver)    ! sat. specific humidity (sh bar star)
   282)    real(r8) hbs(pcols,pver)     ! sat. moist static energy (h bar star)
   283)    real(r8) shbh(pcols,pverp)   ! specific humidity on interfaces
   284)    real(r8) sbh(pcols,pverp)    ! s bar on interfaces
   285)    real(r8) hbh(pcols,pverp)    ! h bar on interfaces
   286)    real(r8) cmrh(pcols,pverp)   ! interface constituent mixing ratio
   287)    real(r8) prec(pcols)         ! instantaneous total precipitation
   288)    real(r8) dzcld(pcols)        ! depth of convective layer (m)
   289)    real(r8) beta(pcols)         ! overshoot parameter (fraction)
   290)    real(r8) betamx(pcols)       ! local maximum on overshoot
   291)    real(r8) eta(pcols)          ! convective mass flux (kg/m^2 s)
   292)    real(r8) etagdt(pcols)       ! eta*grav*dt
   293)    real(r8) cldwtr(pcols)       ! cloud water (mass)
   294)    real(r8) rnwtr(pcols)        ! rain water  (mass)
   295) !  JJH extension to facilitate export of cloud liquid water
   296)    real(r8) totcond(pcols)	! total condensate; mix of precip and cloud water (mass)
   297)    real(r8) sc  (pcols)         ! dry static energy   ("in-cloud")
   298)    real(r8) shc (pcols)         ! specific humidity   ("in-cloud")
   299)    real(r8) hc  (pcols)         ! moist static energy ("in-cloud")
   300)    real(r8) cmrc(pcols)         ! constituent mix rat ("in-cloud")
   301)    real(r8) dq1(pcols)          ! shb  convective change (lower lvl)
   302)    real(r8) dq2(pcols)          ! shb  convective change (mid level)
   303)    real(r8) dq3(pcols)          ! shb  convective change (upper lvl)
   304)    real(r8) ds1(pcols)          ! sb   convective change (lower lvl)
   305)    real(r8) ds2(pcols)          ! sb   convective change (mid level)
   306)    real(r8) ds3(pcols)          ! sb   convective change (upper lvl)
   307)    real(r8) dcmr1(pcols)        ! q convective change (lower lvl)
   308)    real(r8) dcmr2(pcols)        ! q convective change (mid level)
   309)    real(r8) dcmr3(pcols)        ! q convective change (upper lvl)
   310)    real(r8) estemp(pcols,pver)  ! saturation vapor pressure (scratch)
   311)    real(r8) vtemp1(2*pcols)     ! intermediate scratch vector
   312)    real(r8) vtemp2(2*pcols)     ! intermediate scratch vector
   313)    real(r8) vtemp3(2*pcols)     ! intermediate scratch vector
   314)    real(r8) vtemp4(2*pcols)     ! intermediate scratch vector
   315)    real(r8) vtemp5(2*pcols)     ! intermediate scratch vector
   316)    integer indx1(pcols)     ! longitude indices for condition true
   317)    logical etagt0           ! true if eta > 0.0
   318)    real(r8) sh1                 ! dummy arg in qhalf statement func.
   319)    real(r8) sh2                 ! dummy arg in qhalf statement func.
   320)    real(r8) shbs1               ! dummy arg in qhalf statement func.
   321)    real(r8) shbs2               ! dummy arg in qhalf statement func.
   322)    real(r8) cats                ! modified characteristic adj. time
   323)    real(r8) rtdt                ! 1./ztodt
   324)    real(r8) qprime              ! modified specific humidity pert.
   325)    real(r8) tprime              ! modified thermal perturbation
   326)    real(r8) pblhgt              ! bounded pbl height (max[pblh,1m])
   327)    real(r8) fac1                ! intermediate scratch variable
   328)    real(r8) shprme              ! intermediate specific humidity pert.
   329)    real(r8) qsattp              ! sat mix rat for thermally pert PBL parcels
   330)    real(r8) dz                  ! local layer depth
   331)    real(r8) temp1               ! intermediate scratch variable
   332)    real(r8) b1                  ! bouyancy measure in detrainment lvl
   333)    real(r8) b2                  ! bouyancy measure in condensation lvl
   334)    real(r8) temp2               ! intermediate scratch variable
   335)    real(r8) temp3               ! intermediate scratch variable
   336)    real(r8) g                   ! bounded vertical gradient of hb
   337)    real(r8) tmass               ! total mass available for convective exch
   338)    real(r8) denom               ! intermediate scratch variable
   339)    real(r8) qtest1              ! used in negative q test (middle lvl)
   340)    real(r8) qtest2              ! used in negative q test (lower lvl)
   341)    real(r8) fslkp               ! flux lw static energy (bot interface)
   342)    real(r8) fslkm               ! flux lw static energy (top interface)
   343)    real(r8) fqlkp               ! flux total water (bottom interface)
   344)    real(r8) fqlkm               ! flux total water (top interface)
   345)    real(r8) botflx              ! bottom constituent mixing ratio flux
   346)    real(r8) topflx              ! top constituent mixing ratio flux
   347)    real(r8) efac1               ! ratio q to convectively induced chg (btm lvl)
   348)    real(r8) efac2               ! ratio q to convectively induced chg (mid lvl)
   349)    real(r8) efac3               ! ratio q to convectively induced chg (top lvl)
   350)    real(r8) tb(pcols,pver)      ! working storage for temp (t bar)
   351)    real(r8) shb(pcols,pver)     ! working storage for spec hum (sh bar)
   352)    real(r8) adjfac              ! adjustment factor (relaxation related)
   353)    real(r8) rktp
   354)    real(r8) rk
   355) #if ( defined DIAGNS )
   356) !
   357) !  Following 7 real variables are used in diagnostics calculations
   358) !
   359)    real(r8) rh                  ! relative humidity
   360)    real(r8) es                  ! sat vapor pressure
   361)    real(r8) hsum1               ! moist static energy integral
   362)    real(r8) qsum1               ! total water integral
   363)    real(r8) hsum2               ! final moist static energy integral
   364)    real(r8) qsum2               ! final total water integral
   365)    real(r8) fac                 ! intermediate scratch variable
   366) #endif
   367)    integer i,k              ! longitude, level indices
   368)    integer ii               ! index on "gathered" vectors
   369)    integer len1             ! vector length of "gathered" vectors
   370)    integer m                ! constituent index
   371)    integer ktp              ! tmp indx used to track top of convective layer
   372) #if ( defined DIAGNS )
   373)    integer n                ! vertical index     (diagnostics)
   374)    integer kp               ! vertical index     (diagnostics)
   375)    integer kpp              ! index offset, kp+1 (diagnostics)
   376)    integer kpm1             ! index offset, kp-1 (diagnostics)
   377)    integer lat(pcols)       ! latitude indices
   378)    integer lon(pcols)       ! longitude indices
   379) #endif
   380) !
   381) !---------------------------Statement functions-------------------------
   382) !
   383)    real(r8) qhalf
   384)    qhalf(sh1,sh2,shbs1,shbs2) = min(max(sh1,sh2),(shbs2*sh1 + shbs1*sh2)/(shbs1+shbs2))
   385) !
   386) !-----------------------------------------------------------------------
   387) 
   388) !** BAB initialize output tendencies here
   389) !       copy q to dq; use dq below for passive tracer transport
   390)    cmfdt(:ncol,:)  = 0._r8
   391)    cmfdq(:ncol,:)  = 0._r8
   392)    dq(:ncol,:,2:)  = q(:ncol,:,2:)
   393)    cmfmc(:ncol,:)  = 0._r8
   394)    cmfdqr(:ncol,:) = 0._r8
   395)    cmfsl(:ncol,:)  = 0._r8
   396)    cmflq(:ncol,:)  = 0._r8
   397)    qc(:ncol,:)     = 0._r8
   398)    rliq(:ncol)     = 0._r8
   399) !
   400) #if ( defined DIAGNS )
   401) ! Determine chunk latitudes and longitudes
   402)    call get_lat_all_p(lchnk, ncol, lat)
   403)    call get_lon_all_p(lchnk, ncol, lon)
   404) #endif
   405) !
   406) ! Ensure that characteristic adjustment time scale (cmftau) assumed
   407) ! in estimate of eta isn't smaller than model time scale (ztodt)
   408) ! The time over which the convection is assumed to act (the adjustment
   409) ! time scale) can be applied with each application of the three-level
   410) ! cloud model, or applied to the column tendencies after a "hard"
   411) ! adjustment (i.e., on a 2-delta t time scale) is evaluated
   412) !
   413)    if (rlxclm) then
   414)       cats   = ztodt             ! relaxation applied to column
   415)       adjfac = ztodt/(max(ztodt,cmftau))
   416)    else
   417)       cats   = max(ztodt,cmftau) ! relaxation applied to triplet
   418)       adjfac = 1.0_r8
   419)    endif
   420)    rtdt = 1.0_r8/ztodt
   421) !
   422) ! Move temperature and moisture into working storage
   423) !
   424)    do k=limcnv,pver
   425)       do i=1,ncol
   426)          tb (i,k) = t(i,k)
   427)          shb(i,k) = q(i,k,1)
   428)       end do
   429)    end do
   430)    do k=1,pver
   431)       do i=1,ncol
   432)          icwmr(i,k) = 0._r8
   433)       end do
   434)    end do
   435) !
   436) ! Compute sb,hb,shbs,hbs
   437) !
   438)    call qsat(tb(:ncol,limcnv:pver), pmid(:ncol,limcnv:pver), &
   439)         estemp(:ncol,limcnv:pver), shbs(:ncol,limcnv:pver), &
   440)         gam=gam(:ncol,limcnv:pver))
   441) !
   442)    do k=limcnv,pver
   443)       do i=1,ncol
   444)          sb (i,k) = cp*tb(i,k) + zm(i,k)*grav + phis(i)
   445)          hb (i,k) = sb(i,k) + hlat*shb(i,k)
   446)          hbs(i,k) = sb(i,k) + hlat*shbs(i,k)
   447)       end do
   448)    end do
   449) !
   450) ! Compute sbh, shbh
   451) !
   452)    do k=limcnv+1,pver
   453)       do i=1,ncol
   454)          sbh (i,k) = 0.5_r8*(sb(i,k-1) + sb(i,k))
   455)          shbh(i,k) = qhalf(shb(i,k-1),shb(i,k),shbs(i,k-1),shbs(i,k))
   456)          hbh (i,k) = sbh(i,k) + hlat*shbh(i,k)
   457)       end do
   458)    end do
   459) !
   460) ! Specify properties at top of model (not used, but filling anyway)
   461) !
   462)    do i=1,ncol
   463)       sbh (i,limcnv) = sb(i,limcnv)
   464)       shbh(i,limcnv) = shb(i,limcnv)
   465)       hbh (i,limcnv) = hb(i,limcnv)
   466)    end do
   467) !
   468) ! Zero vertically independent control, tendency & diagnostic arrays
   469) !
   470)    do i=1,ncol
   471)       prec(i)  = 0.0_r8
   472)       dzcld(i) = 0.0_r8
   473)       cnb(i)   = 0.0_r8
   474)       cnt(i)   = real(pver+1,r8)
   475)    end do
   476) #if ( defined DIAGNS )
   477) !#######################################################################
   478) !#                                                                     #
   479) !#    output initial thermodynamic profile if debug diagnostics        #
   480) !#                                                                     #
   481)    do i=1,ncol
   482)      if ((lat(i).eq.jloc) .and. (lon(i).eq.iloc) &
   483)          .and. (nstep.ge.nsloc)) then
   484) !#                                                                     #
   485) !#       approximate vertical integral of moist static energy          #
   486) !#       and total preciptable water                                   #
   487) !#                                                                     #
   488)       hsum1 = 0.0_r8
   489)       qsum1 = 0.0_r8
   490)       do k=limcnv,pver
   491)          hsum1 = hsum1 + pdel(i,k)*rgrav*hb(i,k)
   492)          qsum1 = qsum1 + pdel(i,k)*rgrav*shb(i,k)
   493)       end do
   494) !#                                                                     #
   495)       write(iulog,8010)
   496)       fac = grav*864._r8
   497)       do k=limcnv,pver
   498)          rh = shb(i,k)/shbs(i,k)
   499)          write(iulog,8020) shbh(i,k),sbh(i,k),hbh(i,k),fac*cmfmc(i,k),cmfsl(i,k), cmflq(i,k)
   500)          write(iulog,8040) tb(i,k),shb(i,k),rh,sb(i,k),hb(i,k),hbs(i,k),ztodt*cmfdt(i,k), &
   501)                        ztodt*cmfdq(i,k),ztodt*cmfdqr(i,k)
   502)       end do
   503)       write(iulog, 8000) prec(i)
   504)      end if
   505)    enddo
   506) #endif
   507) !#                                                                     #
   508) !#                                                                     #
   509) !#######################################################################
   510) !
   511) ! Begin moist convective mass flux adjustment procedure.
   512) ! Formalism ensures that negative cloud liquid water can never occur
   513) !
   514)    do 70 k=pver-1,limcnv+1,-1
   515)       do 10 i=1,ncol
   516)          etagdt(i) = 0.0_r8
   517)          eta   (i) = 0.0_r8
   518)          beta  (i) = 0.0_r8
   519)          ds1   (i) = 0.0_r8
   520)          ds2   (i) = 0.0_r8
   521)          ds3   (i) = 0.0_r8
   522)          dq1   (i) = 0.0_r8
   523)          dq2   (i) = 0.0_r8
   524)          dq3   (i) = 0.0_r8
   525) !
   526) ! Specification of "cloud base" conditions
   527) !
   528)          qprime    = 0.0_r8
   529)          tprime    = 0.0_r8
   530) !
   531) ! Assign tprime within the PBL to be proportional to the quantity
   532) ! tpert (which will be bounded by tpmax), passed to this routine by
   533) ! the PBL routine.  Don't allow perturbation to produce a dry
   534) ! adiabatically unstable parcel.  Assign qprime within the PBL to be
   535) ! an appropriately modified value of the quantity qpert (which will be
   536) ! bounded by shpmax) passed to this routine by the PBL routine.  The
   537) ! quantity qprime should be less than the local saturation value
   538) ! (qsattp=qsat[t+tprime,p]).  In both cases, tpert and qpert are
   539) ! linearly reduced toward zero as the PBL top is approached.
   540) !
   541)          pblhgt = max(pblh(i),1.0_r8)
   542)          if ( (zm(i,k+1) <= pblhgt) .and. dzcld(i) == 0.0_r8 ) then
   543)             fac1   = max(0.0_r8,1.0_r8-zm(i,k+1)/pblhgt)
   544)             tprime = min(tpert(i),tpmax)*fac1
   545)             qsattp = shbs(i,k+1) + cp*rhlat*gam(i,k+1)*tprime
   546)             shprme = min(min(qpert(i,1),shpmax)*fac1,max(qsattp-shb(i,k+1),0.0_r8))
   547)             qprime = max(qprime,shprme)
   548)          else
   549)             tprime = 0.0_r8
   550)             qprime = 0.0_r8
   551)          end if
   552) !
   553) ! Specify "updraft" (in-cloud) thermodynamic properties
   554) !
   555)          sc (i)    = sb (i,k+1) + cp*tprime
   556)          shc(i)    = shb(i,k+1) + qprime
   557)          hc (i)    = sc (i    ) + hlat*shc(i)
   558)          vtemp4(i) = hc(i) - hbs(i,k)
   559)          dz        = pdel(i,k)*rgas*tb(i,k)*rgrav/pmid(i,k)
   560)          if (vtemp4(i) > 0.0_r8) then
   561)             dzcld(i) = dzcld(i) + dz
   562)          else
   563)             dzcld(i) = 0.0_r8
   564)          end if
   565) 10       continue
   566) #if ( defined DIAGNS )
   567) !#######################################################################
   568) !#                                                                     #
   569) !#    output thermodynamic perturbation information                    #
   570) !#                                                                     #
   571)          do i=1,ncol
   572)            if ((lat(i)==jloc).and.(lon(i)==iloc).and.(nstep>=nsloc)) then
   573)             write(iulog,8090) k+1,sc(iloc),shc(iloc),hc(iloc)
   574)            end if
   575)          enddo
   576) !#                                                                     #
   577) !#######################################################################
   578) #endif
   579) !
   580) ! Check on moist convective instability
   581) ! Build index vector of points where instability exists
   582) !
   583)          len1 = 0
   584)          do i=1,ncol
   585)             if (vtemp4(i) > 0.0_r8) then
   586)                len1 = len1 + 1
   587)                indx1(len1) = i
   588)             end if
   589)          end do
   590)          if (len1 <= 0) go to 70
   591) !
   592) ! Current level just below top level => no overshoot
   593) !
   594)          if (k <= limcnv+1) then
   595)             do ii=1,len1
   596)                i = indx1(ii)
   597)                temp1     = vtemp4(i)/(1.0_r8 + gam(i,k))
   598)                cldwtr(i) = max(0.0_r8,(sb(i,k) - sc(i) + temp1))
   599)                beta(i)   = 0.0_r8
   600)                vtemp3(i) = (1.0_r8 + gam(i,k))*(sc(i) - sbh(i,k))
   601)             end do
   602)          else
   603) !
   604) ! First guess at overshoot parameter using crude buoyancy closure
   605) ! 10% overshoot assumed as a minimum and 1-c0*dz maximum to start
   606) ! If pre-existing supersaturation in detrainment layer, beta=0
   607) ! cldwtr is temporarily equal to hlat*l (l=> liquid water)
   608) !
   609) !cdir nodep
   610) !DIR$ CONCURRENT
   611)             do ii=1,len1
   612)                i = indx1(ii)
   613)                temp1     = vtemp4(i)/(1.0_r8 + gam(i,k))
   614)                cldwtr(i) = max(0.0_r8,(sb(i,k)-sc(i)+temp1))
   615)                betamx(i) = 1.0_r8 - c0*max(0.0_r8,(dzcld(i)-dzmin))
   616)                b1        = (hc(i) - hbs(i,k-1))*pdel(i,k-1)
   617)                b2        = (hc(i) - hbs(i,k  ))*pdel(i,k  )
   618)                beta(i)   = max(betamn,min(betamx(i), 1.0_r8 + b1/b2))
   619)                if (hbs(i,k-1) <= hb(i,k-1)) beta(i) = 0.0_r8
   620) !
   621) ! Bound maximum beta to ensure physically realistic solutions
   622) !
   623) ! First check constrains beta so that eta remains positive
   624) ! (assuming that eta is already positive for beta equal zero)
   625) !
   626)                vtemp1(i) = -(hbh(i,k+1) - hc(i))*pdel(i,k)*rpdel(i,k+1)+ &
   627)                            (1.0_r8 + gam(i,k))*(sc(i) - sbh(i,k+1) + cldwtr(i))
   628)                vtemp2(i) = (1.0_r8 + gam(i,k))*(sc(i) - sbh(i,k))
   629)                vtemp3(i) = vtemp2(i)
   630)                if ((beta(i)*vtemp2(i) - vtemp1(i)) > 0._r8) then
   631)                   betamx(i) = 0.99_r8*(vtemp1(i)/vtemp2(i))
   632)                   beta(i)   = max(0.0_r8,min(betamx(i),beta(i)))
   633)                end if
   634)             end do
   635) !
   636) ! Second check involves supersaturation of "detrainment layer"
   637) ! small amount of supersaturation acceptable (by ssfac factor)
   638) !
   639) !cdir nodep
   640) !DIR$ CONCURRENT
   641)             do ii=1,len1
   642)                i = indx1(ii)
   643)                if (hb(i,k-1) < hbs(i,k-1)) then
   644)                   vtemp1(i) = vtemp1(i)*rpdel(i,k)
   645)                   temp2 = gam(i,k-1)*(sbh(i,k) - sc(i) + cldwtr(i)) -  &
   646)                           hbh(i,k) + hc(i) - sc(i) + sbh(i,k)
   647)                   temp3 = vtemp3(i)*rpdel(i,k)
   648)                   vtemp2(i) = (ztodt/cats)*(hc(i) - hbs(i,k))*temp2/ &
   649)                               (pdel(i,k-1)*(hbs(i,k-1) - hb(i,k-1))) + temp3
   650)                   if ((beta(i)*vtemp2(i) - vtemp1(i)) > 0._r8) then
   651)                      betamx(i) = ssfac*(vtemp1(i)/vtemp2(i))
   652)                      beta(i)   = max(0.0_r8,min(betamx(i),beta(i)))
   653)                   end if
   654)                else
   655)                   beta(i) = 0.0_r8
   656)                end if
   657)             end do
   658) !
   659) ! Third check to avoid introducing 2 delta x thermodynamic
   660) ! noise in the vertical ... constrain adjusted h (or theta e)
   661) ! so that the adjustment doesn't contribute to "kinks" in h
   662) !
   663) !cdir nodep
   664) !DIR$ CONCURRENT
   665)             do ii=1,len1
   666)                i = indx1(ii)
   667)                g = min(0.0_r8,hb(i,k) - hb(i,k-1))
   668)                temp1 = (hb(i,k) - hb(i,k-1) - g)*(cats/ztodt)/(hc(i) - hbs(i,k))
   669)                vtemp1(i) = temp1*vtemp1(i) + (hc(i) - hbh(i,k+1))*rpdel(i,k)
   670)                vtemp2(i) = temp1*vtemp3(i)*rpdel(i,k) + (hc(i) - hbh(i,k) - cldwtr(i))* &
   671)                            (rpdel(i,k) + rpdel(i,k+1))
   672)                if ((beta(i)*vtemp2(i) - vtemp1(i)) > 0._r8) then
   673)                   if (vtemp2(i) /= 0.0_r8) then
   674)                      betamx(i) = vtemp1(i)/vtemp2(i)
   675)                   else
   676)                      betamx(i) = 0.0_r8
   677)                   end if
   678)                   beta(i) = max(0.0_r8,min(betamx(i),beta(i)))
   679)                end if
   680)             end do
   681)          end if
   682) !
   683) ! Calculate mass flux required for stabilization.
   684) !
   685) ! Ensure that the convective mass flux, eta, is positive by
   686) ! setting negative values of eta to zero..
   687) ! Ensure that estimated mass flux cannot move more than the
   688) ! minimum of total mass contained in either layer k or layer k+1.
   689) ! Also test for other pathological cases that result in non-
   690) ! physical states and adjust eta accordingly.
   691) !
   692) !cdir nodep
   693) !DIR$ CONCURRENT
   694)          do ii=1,len1
   695)             i = indx1(ii)
   696)             beta(i) = max(0.0_r8,beta(i))
   697)             temp1 = hc(i) - hbs(i,k)
   698)             temp2 = ((1.0_r8 + gam(i,k))*(sc(i) - sbh(i,k+1) + cldwtr(i)) - &
   699)                       beta(i)*vtemp3(i))*rpdel(i,k) - (hbh(i,k+1) - hc(i))*rpdel(i,k+1)
   700)             eta(i) = temp1/(temp2*grav*cats)
   701)             tmass = min(pdel(i,k),pdel(i,k+1))*rgrav
   702)             if (eta(i) > tmass*rtdt .or. eta(i) <= 0.0_r8) eta(i) = 0.0_r8
   703) !
   704) ! Check on negative q in top layer (bound beta)
   705) !
   706)             if (shc(i)-shbh(i,k) < 0.0_r8 .and. beta(i)*eta(i) /= 0.0_r8) then
   707)                denom = eta(i)*grav*ztodt*(shc(i) - shbh(i,k))*rpdel(i,k-1)
   708)                beta(i) = max(0.0_r8,min(-0.999_r8*shb(i,k-1)/denom,beta(i)))
   709)             end if
   710) !
   711) ! Check on negative q in middle layer (zero eta)
   712) !
   713)             qtest1 = shb(i,k) + eta(i)*grav*ztodt*((shc(i) - shbh(i,k+1)) - &
   714)                      (1.0_r8 - beta(i))*cldwtr(i)*rhlat - beta(i)*(shc(i) - shbh(i,k)))* &
   715) 	             rpdel(i,k)
   716)             if (qtest1 <= 0.0_r8) eta(i) = 0.0_r8
   717) !
   718) ! Check on negative q in lower layer (bound eta)
   719) !
   720)             fac1 = -(shbh(i,k+1) - shc(i))*rpdel(i,k+1)
   721)             qtest2 = shb(i,k+1) - eta(i)*grav*ztodt*fac1
   722)             if (qtest2 < 0.0_r8) then
   723)                eta(i) = 0.99_r8*shb(i,k+1)/(grav*ztodt*fac1)
   724)             end if
   725)             etagdt(i) = eta(i)*grav*ztodt
   726)          end do
   727) !
   728) #if ( defined DIAGNS )
   729) !#######################################################################
   730) !#                                                                     #
   731)          do i=1,ncol
   732)            if ((lat(i)==jloc).and.(lon(i)==iloc).and.(nstep >= nsloc)) then
   733)             write(iulog,8080) beta(iloc), eta(iloc)
   734)            end if
   735)          enddo
   736) !#                                                                     #
   737) !#######################################################################
   738) #endif
   739) !
   740) ! Calculate cloud water, rain water, and thermodynamic changes
   741) !
   742) !cdir nodep
   743) !DIR$ CONCURRENT
   744)          do 30 ii=1,len1
   745)             i = indx1(ii)
   746)             icwmr(i,k) = cldwtr(i)*rhlat
   747)             cldwtr(i) = etagdt(i)*cldwtr(i)*rhlat*rgrav
   748) ! JJH changes to facilitate export of cloud liquid water --------------------------------
   749)             totcond(i) = (1.0_r8 - beta(i))*cldwtr(i)
   750)             rnwtr(i) = min(totcond(i),c0*(dzcld(i)-dzmin)*cldwtr(i))
   751)             ds1(i) = etagdt(i)*(sbh(i,k+1) - sc(i))*rpdel(i,k+1)
   752)             dq1(i) = etagdt(i)*(shbh(i,k+1) - shc(i))*rpdel(i,k+1)
   753)             ds2(i) = (etagdt(i)*(sc(i) - sbh(i,k+1)) +  &
   754)                      hlat*grav*cldwtr(i) - beta(i)*etagdt(i)*(sc(i) - sbh(i,k)))*rpdel(i,k)
   755) ! JJH change for export of cloud liquid water; must use total condensate 
   756) ! since rainwater no longer represents total condensate
   757)             dq2(i) = (etagdt(i)*(shc(i) - shbh(i,k+1)) - grav*totcond(i) - beta(i)* &
   758)                      etagdt(i)*(shc(i) - shbh(i,k)))*rpdel(i,k)
   759)             ds3(i) = beta(i)*(etagdt(i)*(sc(i) - sbh(i,k)) - hlat*grav*cldwtr(i))* &
   760)                      rpdel(i,k-1)
   761)             dq3(i) = beta(i)*etagdt(i)*(shc(i) - shbh(i,k))*rpdel(i,k-1)
   762) !
   763) ! Isolate convective fluxes for later diagnostics
   764) !
   765)             fslkp = eta(i)*(sc(i) - sbh(i,k+1))
   766)             fslkm = beta(i)*(eta(i)*(sc(i) - sbh(i,k)) - hlat*cldwtr(i)*rtdt)
   767)             fqlkp = eta(i)*(shc(i) - shbh(i,k+1))
   768)             fqlkm = beta(i)*eta(i)*(shc(i) - shbh(i,k))
   769) !
   770) ! Update thermodynamic profile (update sb, hb, & hbs later)
   771) !
   772)             tb (i,k+1) = tb(i,k+1)  + ds1(i)*rcp
   773)             tb (i,k  ) = tb(i,k  )  + ds2(i)*rcp
   774)             tb (i,k-1) = tb(i,k-1)  + ds3(i)*rcp
   775)             shb(i,k+1) = shb(i,k+1) + dq1(i)
   776)             shb(i,k  ) = shb(i,k  ) + dq2(i)
   777)             shb(i,k-1) = shb(i,k-1) + dq3(i)
   778) !
   779) ! ** Update diagnostic information for final budget **
   780) ! Tracking precipitation, temperature & specific humidity tendencies,
   781) ! rainout term, convective mass flux, convective liquid
   782) ! water static energy flux, and convective total water flux
   783) ! The variable afac makes the necessary adjustment to the
   784) ! diagnostic fluxes to account for adjustment time scale based on
   785) ! how relaxation time scale is to be applied (column vs. triplet)
   786) !
   787)             prec(i)    = prec(i) + (rnwtr(i)/rhoh2o)*adjfac
   788) !
   789) ! The following variables have units of "units"/second
   790) !
   791)             cmfdt (i,k+1) = cmfdt (i,k+1) + ds1(i)*rtdt*adjfac
   792)             cmfdt (i,k  ) = cmfdt (i,k  ) + ds2(i)*rtdt*adjfac
   793)             cmfdt (i,k-1) = cmfdt (i,k-1) + ds3(i)*rtdt*adjfac
   794)             cmfdq (i,k+1) = cmfdq (i,k+1) + dq1(i)*rtdt*adjfac
   795)             cmfdq (i,k  ) = cmfdq (i,k  ) + dq2(i)*rtdt*adjfac
   796)             cmfdq (i,k-1) = cmfdq (i,k-1) + dq3(i)*rtdt*adjfac
   797) ! JJH changes to export cloud liquid water --------------------------------
   798)             qc    (i,k  ) = (grav*(totcond(i)-rnwtr(i))*rpdel(i,k))*rtdt*adjfac
   799)             cmfdqr(i,k  ) = cmfdqr(i,k  ) + (grav*rnwtr(i)*rpdel(i,k))*rtdt*adjfac
   800)             cmfmc (i,k+1) = cmfmc (i,k+1) + eta(i)*adjfac
   801)             cmfmc (i,k  ) = cmfmc (i,k  ) + beta(i)*eta(i)*adjfac
   802) !
   803) ! The following variables have units of w/m**2
   804) !
   805)             cmfsl (i,k+1) = cmfsl (i,k+1) + fslkp*adjfac
   806)             cmfsl (i,k  ) = cmfsl (i,k  ) + fslkm*adjfac
   807)             cmflq (i,k+1) = cmflq (i,k+1) + hlat*fqlkp*adjfac
   808)             cmflq (i,k  ) = cmflq (i,k  ) + hlat*fqlkm*adjfac
   809) 30          continue
   810) !
   811) ! Next, convectively modify passive constituents
   812) ! For now, when applying relaxation time scale to thermal fields after
   813) ! entire column has undergone convective overturning, constituents will
   814) ! be mixed using a "relaxed" value of the mass flux determined above
   815) ! Although this will be inconsistant with the treatment of the thermal
   816) ! fields, it's computationally much cheaper, no more-or-less justifiable,
   817) ! and consistent with how the history tape mass fluxes would be used in
   818) ! an off-line mode (i.e., using an off-line transport model)
   819) !
   820)             do 50 m=2,pcnst    ! note: indexing assumes water is first field
   821)                if (cnst_get_type_byind(m).eq.'dry') then
   822)                   pd(:ncol,:) = pdeldry(:ncol,:)
   823)                   rpd(:ncol,:) = rpdeldry(:ncol,:)
   824)                   pm(:ncol,:) = pmiddry(:ncol,:)
   825)                else
   826)                   pd(:ncol,:) = pdel(:ncol,:)
   827)                   rpd(:ncol,:) = rpdel(:ncol,:)
   828)                   pm(:ncol,:) = pmid(:ncol,:)
   829)                endif
   830) !cdir nodep
   831) !DIR$ CONCURRENT
   832)                do 40 ii=1,len1
   833)                   i = indx1(ii)
   834) !
   835) ! If any of the reported values of the constituent is negative in
   836) ! the three adjacent levels, nothing will be done to the profile
   837) !
   838)                   if ((dq(i,k+1,m) < 0.0_r8) .or. (dq(i,k,m) < 0.0_r8) .or. (dq(i,k-1,m) < 0.0_r8)) go to 40
   839) !
   840) ! Specify constituent interface values (linear interpolation)
   841) !
   842)                   cmrh(i,k  ) = 0.5_r8*(dq(i,k-1,m) + dq(i,k  ,m))
   843)                   cmrh(i,k+1) = 0.5_r8*(dq(i,k  ,m) + dq(i,k+1,m))
   844) !
   845) ! Specify perturbation properties of constituents in PBL
   846) !
   847)                   pblhgt = max(pblh(i),1.0_r8)
   848)                   if ( (zm(i,k+1) <= pblhgt) .and. dzcld(i) == 0.0_r8 ) then
   849)                      fac1 = max(0.0_r8,1.0_r8-zm(i,k+1)/pblhgt)
   850)                      cmrc(i) = dq(i,k+1,m) + qpert(i,m)*fac1
   851)                   else
   852)                      cmrc(i) = dq(i,k+1,m)
   853)                   end if
   854) !
   855) ! Determine fluxes, flux divergence => changes due to convection
   856) ! Logic must be included to avoid producing negative values. A bit
   857) ! messy since there are no a priori assumptions about profiles.
   858) ! Tendency is modified (reduced) when pending disaster detected.
   859) !
   860)                   botflx   = etagdt(i)*(cmrc(i) - cmrh(i,k+1))*adjfac
   861)                   topflx   = beta(i)*etagdt(i)*(cmrc(i)-cmrh(i,k))*adjfac
   862)                   dcmr1(i) = -botflx*rpd(i,k+1)
   863)                   efac1    = 1.0_r8
   864)                   efac2    = 1.0_r8
   865)                   efac3    = 1.0_r8
   866) !
   867)                   if (dq(i,k+1,m)+dcmr1(i) < 0.0_r8) then
