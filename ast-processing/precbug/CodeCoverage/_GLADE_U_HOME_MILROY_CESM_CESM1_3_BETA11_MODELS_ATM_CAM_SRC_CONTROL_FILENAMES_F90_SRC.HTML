    59) character(len=nlen) function interpret_filename_spec( filename_spec, number, prev, case, &
    60)    yr_spec, mon_spec, day_spec, sec_spec )
    61) 
    62) ! Create a filename from a filename specifier. The 
    63) ! filename specifyer includes codes for setting things such as the
    64) ! year, month, day, seconds in day, caseid, and tape number. This
    65) ! routine is private to filenames.F90
    66) !
    67) ! Interpret filename specifyer string with: 
    68) !
    69) !      %c for case, 
    70) !      %t for optional number argument sent into function
    71) !      %y for year
    72) !      %m for month
    73) !      %d for day
    74) !      %s for second
    75) !      %% for the "%" character
    76) !
    77) ! If the filename specifyer has spaces " ", they will be trimmed out
    78) ! of the resulting filename.
    79) 
    80)    ! arguments
    81)    character(len=*), intent(in)           :: filename_spec   ! Filename specifier to use
    82)    integer         , intent(in), optional :: number          ! Number to use for %t field
    83)    logical         , intent(in), optional :: prev            ! If should label with previous time-step
    84)    character(len=*), intent(in), optional :: case            ! Optional casename
    85)    integer         , intent(in), optional :: yr_spec         ! Simulation year
    86)    integer         , intent(in), optional :: mon_spec        ! Simulation month
    87)    integer         , intent(in), optional :: day_spec        ! Simulation day
    88)    integer         , intent(in), optional :: sec_spec        ! Seconds into current simulation day
    89) 
    90)    ! Local variables
    91)    integer :: year  ! Simulation year
    92)    integer :: month ! Simulation month
    93)    integer :: day   ! Simulation day
    94)    integer :: ncsec ! Seconds into current simulation day
    95)    character(len=nlen) :: string    ! Temporary character string 
    96)    character(len=nlen) :: format    ! Format character string 
    97)    integer :: i, n  ! Loop variables
    98)    logical :: previous              ! If should label with previous time-step
    99)    logical :: done
   100)    !-----------------------------------------------------------------------------
   101) 
   102)    if ( len_trim(filename_spec) == 0 )then
   103)       call endrun ('INTERPRET_FILENAME_SPEC: filename specifier is empty')
   104)    end if
   105)    if ( index(trim(filename_spec)," ") /= 0 )then
   106)       call endrun ('INTERPRET_FILENAME_SPEC: filename specifier can not contain a space:'//trim(filename_spec))
   107)    end if
   108)    !
   109)    ! Determine year, month, day and sec to put in filename
   110)    !
   111)    if (present(yr_spec) .and. present(mon_spec) .and. present(day_spec) .and. present(sec_spec)) then
   112)       year  = yr_spec
   113)       month = mon_spec
   114)       day   = day_spec
   115)       ncsec = sec_spec
   116)    else
   117)       if ( .not. present(prev) ) then
   118)          previous = .false.
   119)       else
   120)          previous = prev
   121)       end if
   122)       if ( previous ) then
   123)          call get_prev_date(year, month, day, ncsec)
   124)       else
   125)          call get_curr_date(year, month, day, ncsec)
   126)       end if
   127)    end if
   128)    !
   129)    ! Go through each character in the filename specifyer and interpret if special string
   130)    !
   131)    i = 1
   132)    interpret_filename_spec = ''
   133)    do while ( i <= len_trim(filename_spec) )
   134)       !
   135)       ! If following is an expansion string
   136)       !
   137)       if ( filename_spec(i:i) == "%" )then
   138)          i = i + 1
   139)          select case( filename_spec(i:i) )
   140)          case( 'c' )   ! caseid
   141)             if ( present(case) )then
   142)                string = trim(case)
   143)             else
   144)                string = trim(caseid)
   145)             end if
   146)          case( 't' )   ! number
   147)             if ( .not. present(number) )then
   148)                write(iulog,*) 'INTERPRET_FILENAME_SPEC: number needed in filename_spec' &
   149)                   , ', but not provided to subroutine'
   150)                write(iulog,*) 'filename_spec = ', filename_spec
   151)                call endrun
   152)             end if
   153)             if (      number > 999 ) then
   154)                format = '(i4.4)'
   155)                if ( number > 9999 ) then
   156)                   write(iulog,*) 'INTERPRET_FILENAME_SPEC: number is too large: ', number
   157)                   call endrun
   158)                end if
   159)             else if ( number > 99  ) then
   160)                format = '(i3.3)'
   161)             else if ( number > 9   ) then
   162)                format = '(i2.2)'
   163)             else
   164)                format = '(i1.1)'
   165)             end if
   166)             write(string,format) number
   167)          case( 'y' )   ! year
   168)             if ( year > 99999   ) then
   169)                format = '(i6.6)'
   170)             else if ( year > 9999    ) then
   171)                format = '(i5.5)'
   172)             else
   173)                format = '(i4.4)'
   174)             end if
   175)             write(string,format) year
   176)          case( 'm' )   ! month
   177)             write(string,'(i2.2)') month
   178)          case( 'd' )   ! day
   179)             write(string,'(i2.2)') day
   180)          case( 's' )   ! second
   181)             write(string,'(i5.5)') ncsec
   182)          case( '%' )   ! percent character
   183)             string = "%"
   184)          case default
   185)             call endrun ('INTERPRET_FILENAME_SPEC: Invalid expansion character: '//filename_spec(i:i))
   186)          end select
   187)          !
   188)          ! Otherwise take normal text up to the next "%" character
   189)          !
   190)       else
   191)          n = index( filename_spec(i:), "%" )
   192)          if ( n == 0 ) n = len_trim( filename_spec(i:) ) + 1
   193)          if ( n == 0 ) exit 
   194)          string = filename_spec(i:n+i-2)
   195)          i = n + i - 2
   196)       end if
   197)       if ( len_trim(interpret_filename_spec) == 0 )then
   198)          interpret_filename_spec = trim(string)
   199)       else
   200)          if ( (len_trim(interpret_filename_spec)+len_trim(string)) >= nlen )then
   201)             call endrun ('INTERPRET_FILENAME_SPEC: Resultant filename too long')
   202)          end if
   203)          interpret_filename_spec = trim(interpret_filename_spec) // trim(string)
   204)       end if
   205)       i = i + 1
   206) 
   207)    end do
   208)    if ( len_trim(interpret_filename_spec) == 0 )then
   209)       call endrun ('INTERPRET_FILENAME_SPEC: Resulting filename is empty')
   210)    end if
   211) 
   212) end function interpret_filename_spec
