   129) #define GEN_SIZE(x) size(x)
   130) #define GEN_ALL(x) all(x)
   131) #else

   133) ! Scalar extensions:
   134) !   GEN_SIZE always returns 1 for a scalar.
   135) !   GEN_ALL (logical reduction) is a no-op for a scalar.
   136) !   GEN_[MAX,MIN]LOC should return a 1D, size 0 (empty), integer array.

   167)   logical :: lt_passed
   168)   logical :: gt_passed
   169)   logical :: le_passed
   170)   logical :: ge_passed
   171)   logical :: eq_passed
   172)   logical :: ne_passed
   173) 
   174)   {VTYPE} :: abs_tol_loc

   176)   ! Handling of abs_tol makes a couple of fairly safe assumptions.
   177)   !  1. It is not the most negative integer.
   178)   !  2. It is finite (not a floating point infinity or NaN).
   179)   if (present(abs_tol)) then
   180)      abs_tol_loc = abs(abs_tol)
   181)   else
   182)      abs_tol_loc = 0_i4
   183)   end if

   187)   gt_passed = .true.
   188)   le_passed = .true.
   189)   ge_passed = .true.
   190)   eq_passed = .true.
   191)   ne_passed = .true.

   193)   ! Do one pass just to find out if we can return with no problem.
   194) 
   195) #ifdef TYPEFP
   196)   ! Only floating-point values can actually be Inf/NaN.
   197)   if (present(is_nan)) &
   198)      is_nan_passed = GEN_ALL(shr_infnan_isnan(var) .eqv. is_nan)
   199) #else

   202) #endif

   206) 
   207)   if (present(gt)) &
   208)      gt_passed = GEN_ALL(var > gt)
   209) 
   210)   if (present(le)) &
   211)      le_passed = GEN_ALL(var <= le)
   212) 
   213)   if (present(ge)) &
   214)      ge_passed = GEN_ALL(var >= ge)
   215) 
   216)   if (present(eq)) then
   217)      eq_passed = GEN_ALL(within_tolerance(eq, var, abs_tol_loc))
   218)   end if

   220)   if (present(ne)) then

   222)   end if

   225)        lt_passed .and. &
   226)        gt_passed .and. &
   227)        le_passed .and. &
   228)        ge_passed .and. &
   229)        eq_passed .and. &
   230)        ne_passed) &

   234)   ! can try to print something useful.

   242)      else

   244)      end if
   245) #else
   246)      loc_vec = spread(1,1,{DIMS})
   247)      call print_bad_loc(var, loc_vec, varname)
   248)      if (is_nan) then
   249)         write(shr_log_Unit,*) &
   250)              "Asserted NaN, but the variable is not floating-point!"
   251)      end if
   252) #endif
   253)   end if
   254) 
   255)   if (.not. lt_passed) then
   256)      loc_vec = find_first_loc(var >= lt)
   257)      call print_bad_loc(var, loc_vec, varname)
   258)      write(shr_log_Unit,*) "Expected value to be less than ",lt
   259)   end if
   260) 
   261)   if (.not. gt_passed) then
   262)      loc_vec = find_first_loc(var <= gt)
   263)      call print_bad_loc(var, loc_vec, varname)
   264)      write(shr_log_Unit,*) "Expected value to be greater than ",gt
   265)   end if
   266) 
   267)   if (.not. le_passed) then
   268)      loc_vec = find_first_loc(var > le)
   269)      call print_bad_loc(var, loc_vec, varname)
   270)      write(shr_log_Unit,*) "Expected value to be less than or &
   271)           &equal to ",le
   272)   end if
   273) 
   274)   if (.not. ge_passed) then
   275)      loc_vec = find_first_loc(var < ge)
   276)      call print_bad_loc(var, loc_vec, varname)
   277)      write(shr_log_Unit,*) "Expected value to be greater than or &
   278)           &equal to ",ge
   279)   end if
   280) 
   281)   if (.not. eq_passed) then
   282)      loc_vec = find_first_loc(.not. within_tolerance(eq, var, abs_tol_loc))
   283)      call print_bad_loc(var, loc_vec, varname)
   284)      write(shr_log_Unit,*) "Expected value to be equal to ",eq
   285)      if (abs_tol_loc > 0) &
   286)           write(shr_log_Unit,*) "Asserted with tolerance ", abs_tol_loc
   287)   end if
   288) 
   289)   if (.not. ne_passed) then
   290)      loc_vec = find_first_loc(within_tolerance(ne, var, abs_tol_loc))
   291)      call print_bad_loc(var, loc_vec, varname)
   292)      write(shr_log_Unit,*) "Expected value to never be equal to ",ne
   293)      if (abs_tol_loc > 0) &
   294)           write(shr_log_Unit,*) "Asserted with tolerance ", abs_tol_loc
   295)   end if
   296) 
   297)   call shr_sys_abort(msg)

   378) elemental function within_tolerance_{TYPE}(expected, actual, tolerance) &
   379)      result(is_in_tol)
   380)   ! Precondition: tolerance must be >= 0.
   381)   {VTYPE}, intent(in) :: expected
   382)   {VTYPE}, intent(in) :: actual
   383)   {VTYPE}, intent(in) :: tolerance
   384)   logical :: is_in_tol
   385) 
   386)   ! The following conditionals are to ensure that we don't overflow.
   387) 
   388)   ! This takes care of two identical infinities.
   389)   if (actual == expected) then
   390)      is_in_tol = .true.
   391)   else if (actual > expected) then
   392)      if (expected >= 0) then
   393)         is_in_tol = (actual - expected) <= tolerance
   394)      else
   395)         is_in_tol = actual <= (expected + tolerance)
   396)      end if
   397)   else
   398)      if (expected < 0) then
   399)         is_in_tol = (expected - actual) <= tolerance
   400)      else
   401)         is_in_tol = actual >= (expected - tolerance)
   402)      end if
   403)   end if
