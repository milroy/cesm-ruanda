    93) subroutine old_liquid_optics_sw(state, pbuf, liq_tau, liq_tau_w, liq_tau_w_g, liq_tau_w_f, oldliqwp)
    94) 
    95)    use physconst, only: gravit
    96) 
    97)    type(physics_state), intent(in) :: state
    98)    type(physics_buffer_desc),pointer :: pbuf(:)
    99) 
   100)    real(r8),intent(out) :: liq_tau    (nswbands,pcols,pver) ! extinction optical depth
   101)    real(r8),intent(out) :: liq_tau_w  (nswbands,pcols,pver) ! single scattering albedo * tau
   102)    real(r8),intent(out) :: liq_tau_w_g(nswbands,pcols,pver) ! assymetry parameter * tau * w
   103)    real(r8),intent(out) :: liq_tau_w_f(nswbands,pcols,pver) ! forward scattered fraction * tau * w
   104)    logical, intent(in) :: oldliqwp
   105) 
   106)    real(r8), pointer, dimension(:,:) :: rel
   107)    real(r8), pointer, dimension(:,:) :: cldn
   108)    real(r8), pointer, dimension(:,:) :: tmpptr
   109)    real(r8), dimension(pcols,pver) :: cliqwp
   110)    real(r8), dimension(nswbands) :: wavmin
   111)    real(r8), dimension(nswbands) :: wavmax
   112) 
   113)    ! Minimum cloud amount (as a fraction of the grid-box area) to 
   114)    ! distinguish from clear sky
   115)    real(r8), parameter :: cldmin = 1.0e-80_r8
   116) 
   117)    ! Decimal precision of cloud amount (0 -> preserve full resolution;
   118)    ! 10^-n -> preserve n digits of cloud amount)
   119)    real(r8), parameter :: cldeps = 0.0_r8
   120) 
   121)    ! A. Slingo's data for cloud particle radiative properties (from 'A GCM
   122)    ! Parameterization for the Shortwave Properties of Water Clouds' JAS
   123)    ! vol. 46 may 1989 pp 1419-1427)
   124)    real(r8) :: abarl(4) = &  ! A coefficient for extinction optical depth
   125)       (/ 2.817e-02_r8, 2.682e-02_r8,2.264e-02_r8,1.281e-02_r8/)
   126)    real(r8) :: bbarl(4) = &  ! B coefficient for extinction optical depth
   127)       (/ 1.305_r8    , 1.346_r8    ,1.454_r8    ,1.641_r8    /)
   128)    real(r8) :: cbarl(4) = &  ! C coefficient for single scat albedo
   129)       (/-5.62e-08_r8 ,-6.94e-06_r8 ,4.64e-04_r8 ,0.201_r8    /)
   130)    real(r8) :: dbarl(4) = &  ! D coefficient for single  scat albedo
   131)       (/ 1.63e-07_r8 , 2.35e-05_r8 ,1.24e-03_r8 ,7.56e-03_r8 /)
   132)    real(r8) :: ebarl(4) = &  ! E coefficient for asymmetry parameter
   133)       (/ 0.829_r8    , 0.794_r8    ,0.754_r8    ,0.826_r8    /)
   134)    real(r8) :: fbarl(4) = &  ! F coefficient for asymmetry parameter
   135)       (/ 2.482e-03_r8, 4.226e-03_r8,6.560e-03_r8,4.353e-03_r8/)
   136) 
   137)    real(r8) :: abarli        ! A coefficient for current spectral band
   138)    real(r8) :: bbarli        ! B coefficient for current spectral band
   139)    real(r8) :: cbarli        ! C coefficient for current spectral band
   140)    real(r8) :: dbarli        ! D coefficient for current spectral band
   141)    real(r8) :: ebarli        ! E coefficient for current spectral band
   142)    real(r8) :: fbarli        ! F coefficient for current spectral band
   143) 
   144)    ! Caution... A. Slingo recommends no less than 4.0 micro-meters nor
   145)    ! greater than 20 micro-meters
   146) 
   147)    integer :: ns, i, k, indxsl, Nday
   148)    integer :: lchnk, itim_old
   149)    real(r8) :: tmp1l, tmp2l, tmp3l, g
   150)    real(r8) :: kext(pcols,pver)
   151)    real(r8), pointer, dimension(:,:) :: iclwpth
   152) 
   153)    Nday = state%ncol
   154)    lchnk = state%lchnk
   155) 
   156)    itim_old = pbuf_old_tim_idx()
   157)    call pbuf_get_field(pbuf, cld_idx,cldn, start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
   158)    call pbuf_get_field(pbuf, rel_idx,rel)
   159) 
   160)    if (oldliqwp) then
   161)      do k=1,pver
   162)         do i = 1,Nday
   163)            cliqwp(i,k) = state%q(i,k,ixcldliq)*state%pdel(i,k)/(gravit*max(0.01_r8,cldn(i,k)))
   164)         end do
   165)      end do
   166)    else
   167)      if (iclwp_idx<0) then 
   168)         call endrun('old_liquid_optics_sw: oldliqwp must be set to true since ICLWP was not found in pbuf')
   169)      endif
   170)      ! The following is the eventual target specification for in cloud liquid water path.
   171)      call pbuf_get_field(pbuf, iclwp_idx, tmpptr)
   172)      cliqwp = tmpptr
   173)    endif
   174)   
   175)    call get_sw_spectral_boundaries(wavmin,wavmax,'microns')
   176) 
   177)    do ns = 1, nswbands
   178)       ! Set index for cloud particle properties based on the wavelength,
   179)       ! according to A. Slingo (1989) equations 1-3:
   180)       ! Use index 1 (0.25 to 0.69 micrometers) for visible
   181)       ! Use index 2 (0.69 - 1.19 micrometers) for near-infrared
   182)       ! Use index 3 (1.19 to 2.38 micrometers) for near-infrared
   183)       ! Use index 4 (2.38 to 4.00 micrometers) for near-infrared
   184)       if(wavmax(ns) <= 0.7_r8) then
   185)          indxsl = 1
   186)       else if(wavmax(ns) <= 1.25_r8) then
   187)          indxsl = 2
   188)       else if(wavmax(ns) <= 2.38_r8) then
   189)          indxsl = 3
   190)       else if(wavmin(ns) > 2.38_r8) then
   191)          indxsl = 4
   192)       end if
   193) 
   194)       ! Set cloud extinction optical depth, single scatter albedo,
   195)       ! asymmetry parameter, and forward scattered fraction:
   196)       abarli = abarl(indxsl)
   197)       bbarli = bbarl(indxsl)
   198)       cbarli = cbarl(indxsl)
   199)       dbarli = dbarl(indxsl)
   200)       ebarli = ebarl(indxsl)
   201)       fbarli = fbarl(indxsl)
   202) 
   203)       do k=1,pver
   204)          do i=1,Nday
   205) 
   206)             ! note that optical properties for liquid valid only
   207)             ! in range of 4.2 > rel > 16 micron (Slingo 89)
   208)             if (cldn(i,k) >= cldmin .and. cldn(i,k) >= cldeps) then
   209)                tmp1l = abarli + bbarli/min(max(4.2_r8,rel(i,k)),16._r8)
   210)                liq_tau(ns,i,k) = 1000._r8*cliqwp(i,k)*tmp1l
   211)             else
   212)                liq_tau(ns,i,k) = 0.0_r8
   213)             endif
   214) 
   215)             tmp2l = 1._r8 - cbarli - dbarli*min(max(4.2_r8,rel(i,k)),16._r8)
   216)             tmp3l = fbarli*min(max(4.2_r8,rel(i,k)),16._r8)
   217)             ! Do not let single scatter albedo be 1.  Delta-eddington solution
   218)             ! for non-conservative case has different analytic form from solution
   219)             ! for conservative case, and raddedmx is written for non-conservative case.
   220)             liq_tau_w(ns,i,k) = liq_tau(ns,i,k) * min(tmp2l,.999999_r8)
   221)             g = ebarli + tmp3l
   222)             liq_tau_w_g(ns,i,k) = liq_tau_w(ns,i,k) * g
   223)             liq_tau_w_f(ns,i,k) = liq_tau_w(ns,i,k) * g * g
   224) 
   225)          end do ! End do i=1,Nday
   226)       end do    ! End do k=1,pver
   227)    end do ! nswbands
   228) 
   229)    !call outfld('CL_OD_SW_OLD',liq_tau(idx_sw_diag,:,:), pcols, lchnk)
   230)    !call outfld('REL_OLD',rel(:,:), pcols, lchnk)
   231)    !call outfld('CLWPTH_OLD',cliqwp(:,:), pcols, lchnk)
   232)    !call outfld('KEXT_OLD',kext(:,:), pcols, lchnk)
   233) 
   234) 
   235) end subroutine old_liquid_optics_sw

   238) subroutine old_ice_optics_sw   (state, pbuf, ice_tau, ice_tau_w, ice_tau_w_g, ice_tau_w_f, oldicewp)
   239) 
   240)    use physconst, only: gravit
   241) 
   242)    type(physics_state), intent(in) :: state
   243)    type(physics_buffer_desc),pointer :: pbuf(:)
   244) 
   245)    real(r8),intent(out) :: ice_tau    (nswbands,pcols,pver) ! extinction optical depth
   246)    real(r8),intent(out) :: ice_tau_w  (nswbands,pcols,pver) ! single scattering albedo * tau
   247)    real(r8),intent(out) :: ice_tau_w_g(nswbands,pcols,pver) ! assymetry parameter * tau * w
   248)    real(r8),intent(out) :: ice_tau_w_f(nswbands,pcols,pver) ! forward scattered fraction * tau * w
   249)    logical, intent(in) :: oldicewp
   250) 
   251)    real(r8), pointer, dimension(:,:) :: rei
   252)    real(r8), pointer, dimension(:,:) :: cldn
   253)    real(r8), pointer, dimension(:,:) :: tmpptr
   254)    real(r8), dimension(pcols,pver) :: cicewp
   255)    real(r8), dimension(nswbands) :: wavmin
   256)    real(r8), dimension(nswbands) :: wavmax
   257)    !
   258)    ! ice water coefficients (Ebert and Curry,1992, JGR, 97, 3831-3836)
   259)    real(r8) :: abari(4) = &     ! a coefficient for extinction optical depth
   260)       (/ 3.448e-03_r8, 3.448e-03_r8,3.448e-03_r8,3.448e-03_r8/)
   261)    real(r8) :: bbari(4) = &     ! b coefficient for extinction optical depth
   262)       (/ 2.431_r8    , 2.431_r8    ,2.431_r8    ,2.431_r8    /)
   263)    real(r8) :: cbari(4) = &     ! c coefficient for single scat albedo
   264)       (/ 1.00e-05_r8 , 1.10e-04_r8 ,1.861e-02_r8,.46658_r8   /)
   265)    real(r8) :: dbari(4) = &     ! d coefficient for single scat albedo
   266)       (/ 0.0_r8      , 1.405e-05_r8,8.328e-04_r8,2.05e-05_r8 /)
   267)    real(r8) :: ebari(4) = &     ! e coefficient for asymmetry parameter
   268)       (/ 0.7661_r8   , 0.7730_r8   ,0.794_r8    ,0.9595_r8   /)
   269)    real(r8) :: fbari(4) = &     ! f coefficient for asymmetry parameter
   270)       (/ 5.851e-04_r8, 5.665e-04_r8,7.267e-04_r8,1.076e-04_r8/)
   271) 
   272)    real(r8) :: abarii           ! A coefficient for current spectral band
   273)    real(r8) :: bbarii           ! B coefficient for current spectral band
   274)    real(r8) :: cbarii           ! C coefficient for current spectral band
   275)    real(r8) :: dbarii           ! D coefficient for current spectral band
   276)    real(r8) :: ebarii           ! E coefficient for current spectral band
   277)    real(r8) :: fbarii           ! F coefficient for current spectral band
   278) 
   279)    ! Minimum cloud amount (as a fraction of the grid-box area) to 
   280)    ! distinguish from clear sky
   281)    real(r8), parameter :: cldmin = 1.0e-80_r8
   282) 
   283)    ! Decimal precision of cloud amount (0 -> preserve full resolution;
   284)    ! 10^-n -> preserve n digits of cloud amount)
   285)    real(r8), parameter :: cldeps = 0.0_r8
   286) 
   287)    integer :: ns, i, k, indxsl, lchnk, Nday
   288)    integer :: itim_old
   289)    real(r8) :: tmp1i, tmp2i, tmp3i, g
   290) 
   291)    Nday = state%ncol
   292)    lchnk = state%lchnk
   293) 
   294)    itim_old = pbuf_old_tim_idx()
   295)    call pbuf_get_field(pbuf, cld_idx,cldn, start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
   296)    call pbuf_get_field(pbuf, rei_idx,rei)
   297) 
   298)    if(oldicewp) then
   299)      do k=1,pver
   300)         do i = 1,Nday
   301)            cicewp(i,k) = 1000.0_r8*state%q(i,k,ixcldice)*state%pdel(i,k) /(gravit* max(0.01_r8,cldn(i,k)))
   302)         end do
   303)      end do
   304)    else
   305)      if (iciwp_idx<=0) then
   306)         call endrun('old_ice_optics_sw: oldicewp must be set to true since ICIWP was not found in pbuf')
   307)      endif
   308)      call pbuf_get_field(pbuf, iciwp_idx, tmpptr)
   309)      cicewp(1:pcols,1:pver) =  1000.0_r8*tmpptr
   310)    endif
   311)    
   312)    call get_sw_spectral_boundaries(wavmin,wavmax,'microns')
   313) 
   314)    do ns = 1, nswbands
   315) 
   316)       if(wavmax(ns) <= 0.7_r8) then
   317)          indxsl = 1
   318)       else if(wavmax(ns) <= 1.25_r8) then
   319)          indxsl = 2
   320)       else if(wavmax(ns) <= 2.38_r8) then
   321)          indxsl = 3
   322)       else if(wavmin(ns) > 2.38_r8) then
   323)          indxsl = 4
   324)       end if
   325) 
   326)       abarii = abari(indxsl)
   327)       bbarii = bbari(indxsl)
   328)       cbarii = cbari(indxsl)
   329)       dbarii = dbari(indxsl)
   330)       ebarii = ebari(indxsl)
   331)       fbarii = fbari(indxsl)
   332) 
   333)       do k=1,pver
   334)          do i=1,Nday
   335) 
   336)             ! note that optical properties for ice valid only
   337)             ! in range of 13 > rei > 130 micron (Ebert and Curry 92)
   338)             if (cldn(i,k) >= cldmin .and. cldn(i,k) >= cldeps) then
   339)                tmp1i = abarii + bbarii/max(13._r8,min(scalefactor*rei(i,k),130._r8))
   340)                ice_tau(ns,i,k) = cicewp(i,k)*tmp1i
   341)             else
   342)                ice_tau(ns,i,k) = 0.0_r8
   343)             endif
   344) 
   345)             tmp2i = 1._r8 - cbarii - dbarii*min(max(13._r8,scalefactor*rei(i,k)),130._r8)
   346)             tmp3i = fbarii*min(max(13._r8,scalefactor*rei(i,k)),130._r8)
   347)             ! Do not let single scatter albedo be 1.  Delta-eddington solution
   348)             ! for non-conservative case has different analytic form from solution
   349)             ! for conservative case, and raddedmx is written for non-conservative case.
   350)             ice_tau_w(ns,i,k) = ice_tau(ns,i,k) * min(tmp2i,.999999_r8)
   351)             g = ebarii + tmp3i
   352)             ice_tau_w_g(ns,i,k) = ice_tau_w(ns,i,k) * g
   353)             ice_tau_w_f(ns,i,k) = ice_tau_w(ns,i,k) * g * g
   354) 
   355)          end do ! End do i=1,Nday
   356)       end do    ! End do k=1,pver
   357)    end do ! nswbands
   358) 
   359) end subroutine old_ice_optics_sw

   362) subroutine oldcloud_lw(state,pbuf,cld_abs_od,oldwp)
   363)    use physconst, only: gravit
   364)    type(physics_state), intent(in) :: state
   365)    type(physics_buffer_desc),pointer :: pbuf(:)
   366)    real(r8),            intent(out) :: cld_abs_od(nlwbands,pcols,pver) ! [fraction] absorption optical depth, per layer
   367)    logical,intent(in)              :: oldwp ! use old definition of waterpath
   368) 
   369) 
   370)    real(r8) :: gicewp(pcols,pver)
   371)    real(r8) :: gliqwp(pcols,pver)
   372)    real(r8) :: cicewp(pcols,pver)
   373)    real(r8) :: cliqwp(pcols,pver)
   374)    real(r8) :: ficemr(pcols,pver)
   375)    real(r8) :: cwp(pcols,pver)
   376)    real(r8) :: cldtau(pcols,pver)
   377) 
   378)    real(r8), pointer, dimension(:,:) :: cldn
   379)    real(r8), pointer, dimension(:,:) :: rei 
   380)    integer :: ncol, itim_old, lwband, i, k, lchnk
   381)    real(r8), pointer, dimension(:,:) :: iclwpth, iciwpth
   382) 
   383)     real(r8) :: kabs, kabsi
   384)     real(r8) kabsl                  ! longwave liquid absorption coeff (m**2/g)
   385)     parameter (kabsl = 0.090361_r8)
   386) 
   387) 
   388)  
   389)    ncol = state%ncol
   390)    lchnk = state%lchnk
   391) 
   392)    itim_old  =  pbuf_old_tim_idx()
   393)    call pbuf_get_field(pbuf, rei_idx,   rei)
   394)    call pbuf_get_field(pbuf, cld_idx,   cldn,   start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
   395) 
   396)    if (oldwp) then
   397)      do k=1,pver
   398)          do i = 1,ncol
   399)             gicewp(i,k) = state%q(i,k,ixcldice)*state%pdel(i,k)/gravit*1000.0_r8  ! Grid box ice water path.
   400)             gliqwp(i,k) = state%q(i,k,ixcldliq)*state%pdel(i,k)/gravit*1000.0_r8  ! Grid box liquid water path.
   401)             cicewp(i,k) = gicewp(i,k) / max(0.01_r8,cldn(i,k))                 ! In-cloud ice water path.
   402)             cliqwp(i,k) = gliqwp(i,k) / max(0.01_r8,cldn(i,k))                 ! In-cloud liquid water path.
   403)             ficemr(i,k) = state%q(i,k,ixcldice) /                 &
   404)                  max(1.e-10_r8,(state%q(i,k,ixcldice)+state%q(i,k,ixcldliq)))
   405)          end do
   406)      end do
   407)      cwp(:ncol,:pver) = cicewp(:ncol,:pver) + cliqwp(:ncol,:pver)
   408)    else
   409)       if (iclwp_idx<=0 .or. iciwp_idx<=0) then 
   410)          call endrun('oldcloud_lw: oldwp must be set to true since ICIWP and/or ICLWP were not found in pbuf')
   411)       endif
   412)       call pbuf_get_field(pbuf, iclwp_idx, iclwpth)
   413)       call pbuf_get_field(pbuf, iciwp_idx, iciwpth)
   414)       do k=1,pver
   415)          do i = 1,ncol
   416)             cwp(i,k) = 1000.0_r8 *iclwpth(i,k) + 1000.0_r8 *iciwpth(i, k)
   417)             ficemr(i,k) = 1000.0_r8 * iciwpth(i,k)/(max(1.e-18_r8,cwp(i,k)))
   418)          end do
   419)       end do
   420)    endif
   421) 
   422)    do k=1,pver
   423)        do i=1,ncol
   424)
