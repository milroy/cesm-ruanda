    94) subroutine wetdep_inputs_set( state, pbuf, inputs )
    95)   use phys_control,   only: cam_physpkg_is
    96)   use physics_types,  only: physics_state
    97)   use physics_buffer, only: physics_buffer_desc, pbuf_get_field, pbuf_old_tim_idx
    98) 
    99)   ! args
   100) 
   101)   type(physics_state),  intent(in )  :: state           !! physics state
   102)   type(physics_buffer_desc), pointer :: pbuf(:)         !! physics buffer
   103)   type(wetdep_inputs_t), intent(out) :: inputs          !! collection of wetdepa inputs
   104) 
   105)   ! local vars
   106) 
   107)   real(r8), pointer :: icwmrdp(:,:)    ! in cloud water mixing ratio, deep convection
   108)   real(r8), pointer :: rprddp(:,:)     ! rain production, deep convection
   109)   real(r8), pointer :: icwmrsh(:,:)    ! in cloud water mixing ratio, deep convection
   110)   real(r8), pointer :: rprdsh(:,:)     ! rain production, deep convection
   111)   real(r8), pointer :: sh_frac(:,:)    ! Shallow convective cloud fraction
   112)   real(r8), pointer :: dp_frac(:,:)    ! Deep convective cloud fraction
   113)   real(r8), pointer :: evapcsh(:,:)    ! Evaporation rate of shallow convective precipitation >=0.
   114)   real(r8), pointer :: evapcdp(:,:)    ! Evaporation rate of deep    convective precipitation >=0.
   115) 
   116)   real(r8) :: rainmr(pcols,pver)       ! mixing ratio of rain within cloud volume
   117)   real(r8) :: cldst(pcols,pver)        ! Stratiform cloud fraction
   118) 
   119)   integer :: itim, ncol
   120) 
   121)   ncol = state%ncol
   122)   itim = pbuf_old_tim_idx()
   123) 
   124)   call pbuf_get_field(pbuf, cld_idx,         inputs%cldt, start=(/1,1,itim/), kount=(/pcols,pver,1/) )
   125)   call pbuf_get_field(pbuf, qme_idx,         inputs%qme     )
   126)   call pbuf_get_field(pbuf, prain_idx,       inputs%prain   )
   127)   call pbuf_get_field(pbuf, nevapr_idx,      inputs%evapr   )
   128)   call pbuf_get_field(pbuf, icwmrdp_idx,     icwmrdp )
   129)   call pbuf_get_field(pbuf, icwmrsh_idx,     icwmrsh )
   130)   call pbuf_get_field(pbuf, rprddp_idx,      rprddp  )
   131)   call pbuf_get_field(pbuf, rprdsh_idx,      rprdsh  )
   132)   call pbuf_get_field(pbuf, sh_frac_idx,     sh_frac )
   133)   call pbuf_get_field(pbuf, dp_frac_idx,     dp_frac )
   134)   call pbuf_get_field(pbuf, nevapr_shcu_idx, evapcsh )
   135)   call pbuf_get_field(pbuf, nevapr_dpcu_idx, evapcdp )
   136) 
   137)   inputs%cldcu(:ncol,:)  = dp_frac(:ncol,:) + sh_frac(:ncol,:)
   138)   cldst(:ncol,:)          = inputs%cldt(:ncol,:) - inputs%cldcu(:ncol,:)       ! Stratiform cloud fraction
   139)   inputs%evapc(:ncol,:)  = evapcsh(:ncol,:) + evapcdp(:ncol,:)
   140)   inputs%cmfdqr(:ncol,:) = rprddp(:ncol,:)  + rprdsh(:ncol,:)
   141) 
   142)   ! sum deep and shallow convection contributions
   143)   if (cam_physpkg_is('cam5')) then
   144)      ! Dec.29.2009. Sungsu
   145)      inputs%conicw(:ncol,:) = (icwmrdp(:ncol,:)*dp_frac(:ncol,:) + icwmrsh(:ncol,:)*sh_frac(:ncol,:))/ &
   146)                               max(0.01_r8, sh_frac(:ncol,:) + dp_frac(:ncol,:))
   147)   else
   148)      inputs%conicw(:ncol,:) = icwmrdp(:ncol,:) + icwmrsh(:ncol,:)
   149)   end if
   150) 
   151)   inputs%totcond(:ncol,:) = state%q(:ncol,:,ixcldliq) + state%q(:ncol,:,ixcldice)
   152) 
   153)   call clddiag( state%t,     state%pmid,   state%pdel,   inputs%cmfdqr, inputs%evapc, &
   154)                inputs%cldt,  inputs%cldcu,       cldst,  inputs%qme,    inputs%evapr, &
   155)                inputs%prain, inputs%cldv, inputs%cldvcu, inputs%cldvst,       rainmr, &
   156)                 state%ncol )
   157) 
   158) end subroutine wetdep_inputs_set

   160) subroutine clddiag(t, pmid, pdel, cmfdqr, evapc, &
   161)                    cldt, cldcu, cldst, cme, evapr, &
   162)                    prain, cldv, cldvcu, cldvst, rain, &
   163)                    ncol)
   164) 
   165)    ! ------------------------------------------------------------------------------------ 
   166)    ! Estimate the cloudy volume which is occupied by rain or cloud water as
   167)    ! the max between the local cloud amount or the
   168)    ! sum above of (cloud*positive precip production)      sum total precip from above
   169)    !              ----------------------------------   x ------------------------
   170)    ! sum above of     (positive precip           )        sum positive precip from above
   171)    ! Author: P. Rasch
   172)    !         Sungsu Park. Mar.2010 
   173)    ! ------------------------------------------------------------------------------------
   174) 
   175)    ! Input arguments:
   176)    real(r8), intent(in) :: t(pcols,pver)        ! temperature (K)
   177)    real(r8), intent(in) :: pmid(pcols,pver)     ! pressure at layer midpoints
   178)    real(r8), intent(in) :: pdel(pcols,pver)     ! pressure difference across layers
   179)    real(r8), intent(in) :: cmfdqr(pcols,pver)   ! dq/dt due to convective rainout 
   180)    real(r8), intent(in) :: evapc(pcols,pver)    ! Evaporation rate of convective precipitation ( >= 0 ) 
   181)    real(r8), intent(in) :: cldt(pcols,pver)    ! total cloud fraction
   182)    real(r8), intent(in) :: cldcu(pcols,pver)    ! Cumulus cloud fraction
   183)    real(r8), intent(in) :: cldst(pcols,pver)    ! Stratus cloud fraction
   184)    real(r8), intent(in) :: cme(pcols,pver)      ! rate of cond-evap within the cloud
   185)    real(r8), intent(in) :: evapr(pcols,pver)    ! rate of evaporation of falling precipitation (kg/kg/s)
   186)    real(r8), intent(in) :: prain(pcols,pver)    ! rate of conversion of condensate to precipitation (kg/kg/s)
   187)    integer, intent(in) :: ncol
   188) 
   189)    ! Output arguments:
   190)    real(r8), intent(out) :: cldv(pcols,pver)     ! fraction occupied by rain or cloud water 
   191)    real(r8), intent(out) :: cldvcu(pcols,pver)   ! Convective precipitation volume
   192)    real(r8), intent(out) :: cldvst(pcols,pver)   ! Stratiform precipitation volume
   193)    real(r8), intent(out) :: rain(pcols,pver)     ! mixing ratio of rain (kg/kg)
   194) 
   195)    ! Local variables:
   196)    integer  i, k
   197)    real(r8) convfw         ! used in fallspeed calculation; taken from findmcnew
   198)    real(r8) sumppr(pcols)        ! precipitation rate (kg/m2-s)
   199)    real(r8) sumpppr(pcols)       ! sum of positive precips from above
   200)    real(r8) cldv1(pcols)         ! precip weighted cloud fraction from above
   201)    real(r8) lprec                ! local production rate of precip (kg/m2/s)
   202)    real(r8) lprecp               ! local production rate of precip (kg/m2/s) if positive
   203)    real(r8) rho                  ! air density
   204)    real(r8) vfall
   205)    real(r8) sumppr_cu(pcols)     ! Convective precipitation rate (kg/m2-s)
   206)    real(r8) sumpppr_cu(pcols)    ! Sum of positive convective precips from above
   207)    real(r8) cldv1_cu(pcols)      ! Convective precip weighted convective cloud fraction from above
   208)    real(r8) lprec_cu             ! Local production rate of convective precip (kg/m2/s)
   209)    real(r8) lprecp_cu            ! Local production rate of convective precip (kg/m2/s) if positive
   210)    real(r8) sumppr_st(pcols)     ! Stratiform precipitation rate (kg/m2-s)
   211)    real(r8) sumpppr_st(pcols)    ! Sum of positive stratiform precips from above
   212)    real(r8) cldv1_st(pcols)      ! Stratiform precip weighted stratiform cloud fraction from above
   213)    real(r8) lprec_st             ! Local production rate of stratiform precip (kg/m2/s)
   214)    real(r8) lprecp_st            ! Local production rate of stratiform precip (kg/m2/s) if positive
   215)    ! -----------------------------------------------------------------------
   216) 
   217)    convfw = 1.94_r8*2.13_r8*sqrt(rhoh2o*gravit*2.7e-4_r8)
   218)    do i=1,ncol
   219)       sumppr(i) = 0._r8
   220)       cldv1(i) = 0._r8
   221)       sumpppr(i) = 1.e-36_r8
   222)       sumppr_cu(i)  = 0._r8
   223)       cldv1_cu(i)   = 0._r8
   224)       sumpppr_cu(i) = 1.e-36_r8
   225)       sumppr_st(i)  = 0._r8
   226)       cldv1_st(i)   = 0._r8
   227)       sumpppr_st(i) = 1.e-36_r8
   228)    end do
   229) 
   230)    do k = 1,pver
   231)       do i = 1,ncol
   232)          cldv(i,k) = &
   233)             max(min(1._r8, &
   234)             cldv1(i)/sumpppr(i) &
   235)             )*sumppr(i)/sumpppr(i), &
   236)             cldt(i,k) &
   237)             )
   238)          lprec = pdel(i,k)/gravit &
   239)             *(prain(i,k)+cmfdqr(i,k)-evapr(i,k))
   240)          lprecp = max(lprec,1.e-30_r8)
   241)          cldv1(i) = cldv1(i)  + cldt(i,k)*lprecp
   242)          sumppr(i) = sumppr(i) + lprec
   243)          sumpppr(i) = sumpppr(i) + lprecp
   244) 
   245)          ! For convective precipitation volume at the top interface of each layer. Neglect the current layer.
   246)          cldvcu(i,k)   = max(min(1._r8,cldv1_cu(i)/sumpppr_cu(i))*(sumppr_cu(i)/sumpppr_cu(i)),0._r8)
   247)          lprec_cu      = (pdel(i,k)/gravit)*(cmfdqr(i,k)-evapc(i,k))
   248)          lprecp_cu     = max(lprec_cu,1.e-30_r8)
   249)          cldv1_cu(i)   = cldv1_cu(i) + cldcu(i,k)*lprecp_cu
   250)          sumppr_cu(i)  = sumppr_cu(i) + lprec_cu
   251)          sumpppr_cu(i) = sumpppr_cu(i) + lprecp_cu
   252) 
   253)          ! For stratiform precipitation volume at the top interface of each layer. Neglect the current layer.
   254)          cldvst(i,k)   = max(min(1._r8,cldv1_st(i)/sumpppr_st(i))*(sumppr_st(i)/sumpppr_st(i)),0._r8)
   255)          lprec_st      = (pdel(i,k)/gravit)*(prain(i,k)-evapr(i,k))
   256)          lprecp_st     = max(lprec_st,1.e-30_r8)
   257)          cldv1_st(i)   = cldv1_st(i) + cldst(i,k)*lprecp_st
   258)          sumppr_st(i)  = sumppr_st(i) + lprec_st
   259)          sumpppr_st(i) = sumpppr_st(i) + lprecp_st
   260) 
   261)          rain(i,k) = 0._r8
   262)          if(t(i,k) .gt. tmelt) then
   263)             rho = pmid(i,k)/(rair*t(i,k))
   264)             vfall = convfw/sqrt(rho)
   265)             rain(i,k) = sumppr(i)/(rho*vfall)
   266)             if (rain(i,k).lt.1.e-14_r8) rain(i,k) = 0._r8
   267)          endif
   268)       end do
   269)    end do
   270) 
   271) end subroutine clddiag

   277) subroutine wetdepa_v2(                                  &
   278)    p, q, pdel, cldt, cldc,                              &
   279)    cmfdqr, evapc, conicw, precs, conds,                 &
   280)    evaps, cwat, tracer, deltat, scavt,                  &
   281)    iscavt, cldvcu, cldvst, dlf, fracis,                 &
   282)    sol_fact, ncol, scavcoef, is_strat_cloudborne, qqcw, &
   283)    f_act_conv, icscavt, isscavt, bcscavt, bsscavt,      &
   284)    sol_facti_in, sol_factic_in )
   285) 
   286)    !----------------------------------------------------------------------- 
   287)    !
   288)    ! scavenging code for very soluble aerosols
   289)    ! 
   290)    !-----------------------------------------------------------------------
   291) 
   292)    real(r8), intent(in) ::&
   293)       p(pcols,pver),        &! pressure
   294)       q(pcols,pver),        &! moisture
   295)       pdel(pcols,pver),     &! pressure thikness
   296)       cldt(pcols,pver),     &! total cloud fraction
   297)       cldc(pcols,pver),     &! convective cloud fraction
   298)       cmfdqr(pcols,pver),   &! rate of production of convective precip
   299)       evapc(pcols,pver),    &! Evaporation rate of convective precipitation
   300)       conicw(pcols,pver),   &! convective cloud water
   301)       cwat(pcols,pver),     &! cloud water amount 
   302)       precs(pcols,pver),    &! rate of production of stratiform precip
   303)       conds(pcols,pver),    &! rate of production of condensate
   304)       evaps(pcols,pver),    &! rate of evaporation of precip
   305)       cldvcu(pcols,pver),   &! Convective precipitation area at the top interface of each layer
   306)       cldvst(pcols,pver),   &! Stratiform precipitation area at the top interface of each layer
   307)       dlf(pcols,pver),      &! Detrainment of convective condensate [kg/kg/s]
   308)       deltat,               &! time step
   309)       tracer(pcols,pver)     ! trace species
   310) 
   311)    ! If subroutine is called with just sol_fact:
   312)    !    sol_fact is used for both in- and below-cloud scavenging
   313)    ! If subroutine is called with optional argument sol_facti_in:
   314)    !    sol_fact  is used for below cloud scavenging
   315)    !    sol_facti is used for in cloud scavenging
   316) 
   317)    real(r8), intent(in)  :: sol_fact
   318)    integer,  intent(in)  :: ncol
   319)    real(r8), intent(in)  :: scavcoef(pcols,pver) ! Dana and Hales coefficient (/mm) (0.1 if not MODAL_AERO)
   320)    real(r8), intent(out) ::&
   321)       scavt(pcols,pver),   &! scavenging tend 
   322)       iscavt(pcols,pver),  &! incloud scavenging tends
   323)       fracis(pcols,pver)    ! fraction of species not scavenged
   324) 
   325)    ! Setting is_strat_cloudborne=.true. indicates that tracer is stratiform-cloudborne aerosol.
   326)    !   This is only used by MAM code.  The optional args qqcw and f_act_conv are not referenced
   327)    !   in this case.
   328)    ! Setting is_strat_cloudborne=.false. is being used to indicate that the tracers are the
   329)    !   interstitial modal aerosols.  In this case the optional qqcw (the cloud borne mixing ratio
   330)    !   corresponding to the interstitial aerosol) must be provided, as well as the optional f_act_conv.
   331)    logical,  intent(in), optional :: is_strat_cloudborne   
   332)    real(r8), intent(in), optional :: qqcw(pcols,pver)
   333)    real(r8), intent(in), optional :: f_act_conv(pcols,pver)
   334) 
   335)    real(r8), intent(in), optional :: sol_facti_in   ! solubility factor (frac of aerosol scavenged in cloud)
   336)    real(r8), intent(in), optional :: sol_factic_in(pcols,pver)  ! sol_facti_in for convective clouds
   337)          
   338) 
   339)    real(r8), intent(out), optional :: icscavt(pcols,pver)     ! incloud, convective
   340)    real(r8), intent(out), optional :: isscavt(pcols,pver)     ! incloud, stratiform
   341)    real(r8), intent(out), optional :: bcscavt(pcols,pver)     ! below cloud, convective
   342)    real(r8), intent(out), optional :: bsscavt(pcols,pver)     ! below cloud, stratiform
   343) 
   344)    ! local variables
   345) 
   346)    integer :: i, k
   347) 
   348)    real(r8) :: omsm                 ! 1 - (a small number)
   349)    real(r8) :: clds(pcols)          ! stratiform cloud fraction
   350)    real(r8) :: fracev(pcols)        ! fraction of precip from above that is evaporating
   351)    real(r8) :: fracev_cu(pcols)     ! Fraction of convective precip from above that is evaporating
   352)    real(r8) :: fracp(pcols)         ! fraction of cloud water converted to precip
   353)    real(r8) :: pdog(pcols)          ! work variable (pdel/gravit)
   354)    real(r8) :: rpdog(pcols)         ! work variable (gravit/pdel)
   355)    real(r8) :: precabc(pcols)       ! conv precip from above (work array)
   356)    real(r8) :: precabs(pcols)       ! strat precip from above (work array)
   357)    real(r8) :: rat(pcols)           ! ratio of amount available to amount removed
   358)    real(r8) :: scavab(pcols)        ! scavenged tracer flux from above (work array)
   359)    real(r8) :: scavabc(pcols)       ! scavenged tracer flux from above (work array)
   360)    real(r8) :: srcc(pcols)          ! tend for convective rain
   361)    real(r8) :: srcs(pcols)          ! tend for stratiform rain
   362)    real(r8) :: srct(pcols)          ! work variable
   363) 
   364)    real(r8) :: fins(pcols)          ! fraction of rem. rate by strat rain
   365)    real(r8) :: finc(pcols)          ! fraction of rem. rate by conv. rain
   366)    real(r8) :: conv_scav_ic(pcols)  ! convective scavenging incloud
   367)    real(r8) :: conv_scav_bc(pcols)  ! convective scavenging below cloud
   368)    real(r8) :: st_scav_ic(pcols)    ! stratiform scavenging incloud
   369)    real(r8) :: st_scav_bc(pcols)    ! stratiform scavenging below cloud
   370) 
   371)    real(r8) :: odds(pcols)          ! limit on removal rate (proportional to prec)
   372)    real(r8) :: dblchek(pcols)
   373)    logical :: found
   374) 
   375)    real(r8) :: trac_qqcw(pcols)
   376)    real(r8) :: tracer_incu(pcols)
   377)    real(r8) :: tracer_mean(pcols)
   378) 
   379)    ! For stratiform cloud, cloudborne aerosol is treated explicitly,
   380)    !    and sol_facti is 1.0 for cloudborne, 0.0 for interstitial.
   381)    ! For convective cloud, cloudborne aerosol is not treated explicitly,
   382)    !    and sol_factic is 1.0 for both cloudborne and interstitial.
   383) 
   384)    real(r8) :: sol_facti              ! in cloud fraction of aerosol scavenged
   385)    real(r8) :: sol_factb              ! below cloud fraction of aerosol scavenged
   386)    real(r8) :: sol_factic(pcols,pver) ! in cloud fraction of aerosol scavenged for convective clouds
   387) 
   388)    real(r8) :: rdeltat
   389)    ! ------------------------------------------------------------------------
   390) 
   391)    omsm = 1._r8-2*epsilon(1._r8) ! used to prevent roundoff errors below zero
   392) 
   393)    ! default (if other sol_facts aren't in call, set all to required sol_fact)
   394)    sol_facti = sol_fact
   395)    sol_factb = sol_fact
   396) 
   397)    if ( present(sol_facti_in) )  sol_facti = sol_facti_in
   398) 
   399)    sol_factic  = sol_facti
   400)    if ( present(sol_factic_in ) )  sol_factic  = sol_factic_in
   401) 
   402)    ! this section of code is for highly soluble aerosols,
   403)    ! the assumption is that within the cloud that
   404)    ! all the tracer is in the cloud water
   405)    !
   406)    ! for both convective and stratiform clouds, 
   407)    ! the fraction of cloud water converted to precip defines
   408)    ! the amount of tracer which is pulled out.
   409) 
   410)    precabs(:ncol) = 0.0_r8
   411)    precabc(:ncol) = 0.0_r8
   412)    scavab(:ncol)  = 0.0_r8
   413)    scavabc(:ncol) = 0.0_r8
   414) 
   415)    do k = 1, pver
   416)       do i = 1, ncol
   417) 
   418)          clds(i)  = cldt(i,k) - cldc(i,k)
   419)          pdog(i)  = pdel(i,k)/gravit
   420)          rpdog(i) = gravit/pdel(i,k)
   421)          rdeltat  = 1.0_r8/deltat
   422) 
   423)          ! ****************** Evaporation **************************
   424)          ! calculate the fraction of strat precip from above 
   425)          !                 which evaporates within this layer
   426)          fracev(i) = evaps(i,k)*pdog(i) &
   427)                      /max(1.e-12_r8,precabs(i))
   428) 
   429)          ! trap to ensure reasonable ratio bounds
   430)          fracev(i) = max(0._r8,min(1._r8,fracev(i)))
   431) 
   432)          ! Same as above but convective precipitation part
   433)          fracev_cu(i) = evapc(i,k)*pdog(i)/max(1.e-12_r8,precabc(i))
   434)          fracev_cu(i) = max(0._r8,min(1._r8,fracev_cu(i)))
   435) 
   436)          ! ****************** Convection ***************************
   437)          !
   438)          ! set odds proportional to fraction of the grid box that is swept by the 
   439)          ! precipitation =precabc/rhoh20*(area of sphere projected on plane
   440)          !                                /volume of sphere)*deltat
   441)          ! assume the radius of a raindrop is 1 e-3 m from Rogers and Yau,
   442)          ! unless the fraction of the area that is cloud is less than odds, in which
   443)          ! case use the cloud fraction (assumes precabs is in kg/m2/s)
   444)          ! is really: precabs*3/4/1000./1e-3*deltat
   445)          ! here I use .1 from Balkanski
   446)          !
   447)          ! use a local rate of convective rain production for incloud scav
   448)          !
   449)          ! Fraction of convective cloud water converted to rain.  This version is used
   450)          ! in 2 of the 3 branches below before fracp is reused in the stratiform calc.
   451)          ! NB: In below formula for fracp conicw is a LWC/IWC that has already
   452)          !     precipitated out, i.e., conicw does not contain precipitation
   453) 
   454)          fracp(i) = cmfdqr(i,k)*deltat / &
   455)                     max( 1.e-12_r8, cldc(i,k)*conicw(i,k) + (cmfdqr(i,k)+dlf(i,k))*deltat )
   456)          fracp(i) = max( min( 1._r8, fracp(i)), 0._r8 )
   457) 
   458)          if ( present(is_strat_cloudborne) ) then
   459) 
   460)             if ( is_strat_cloudborne ) then
   461) 
   462)                ! convective scavenging
   463) 
   464)                conv_scav_ic(i) = 0._r8
   465) 
   466)                conv_scav_bc(i) = 0._r8
   467) 
   468)                ! stratiform scavenging
   469) 
   470)                fracp(i) = precs(i,k)*deltat / &
   471)                           max( 1.e-12_r8, cwat(i,k) + precs(i,k)*deltat )
   472)                fracp(i) = max( 0._r8, min(1._r8, fracp(i)) )
   473)                st_scav_ic(i) = sol_facti *fracp(i)*tracer(i,k)*rdeltat
   474) 
   475)                st_scav_bc(i) = 0._r8
   476) 
   477)             else
   478) 
   479)                ! convective scavenging
   480) 
   481)                trac_qqcw(i) = min(qqcw(i,k), &
   482)                                   tracer(i,k)*( clds(i)/max( 0.01_r8, 1._r8-clds(i) ) ) )
   483) 
   484)                tracer_incu(i) = f_act_conv(i,k)*(tracer(i,k) + trac_qqcw(i))
   485) 
   486)                conv_scav_ic(i) = sol_factic(i,k)*cldc(i,k)*fracp(i)*tracer_incu(i)*rdeltat
   487) 
   488)                tracer_mean(i) = tracer(i,k)*(1._r8 - cldc(i,k)*f_act_conv(i,k)) - &
   489)                                 cldc(i,k)*f_act_conv(i,k)*trac_qqcw(i)
   490)                tracer_mean(i) = max(0._r8,tracer_mean(i))
   491) 
   492)                odds(i) = precabc(i)/max(cldvcu(i,k),1.e-5_r8)*scavcoef(i,k)*deltat
   493)                odds(i) = max(min(1._r8,odds(i)),0._r8)
   494)                conv_scav_bc(i) = sol_factb *cldvcu(i,k)*odds(i)*tracer_mean(i)*rdeltat
   495) 
   496) 
   497)                ! stratiform scavenging
   498) 
   499)                st_scav_ic(i) = 0._r8
   500) 
   501)                odds(i) = precabs(i)/max(cldvst(i,k),1.e-5_r8)*scavcoef(i,k)*deltat
   502)                odds(i) = max(min(1._r8,odds(i)),0._r8)
   503)                st_scav_bc(i) = sol_factb *cldvst(i,k)*odds(i)*tracer_mean(i)*rdeltat
   504) 
   505)             end if
   506) 
   507)          else
   508) 
   509)             ! convective scavenging
   510) 
   511)             conv_scav_ic(i) = sol_factic(i,k)*cldc(i,k)*fracp(i)*tracer(i,k)*rdeltat
   512) 
   513)             odds(i) = precabc(i)/max(cldvcu(i,k), 1.e-5_r8)*scavcoef(i,k)*deltat
   514)             odds(i) = max( min(1._r8, odds(i)), 0._r8)
   515)             conv_scav_bc(i) = sol_factb*cldvcu(i,k)*odds(i)*tracer(i,k)*rdeltat
   516) 
   517)             ! stratiform scavenging
   518) 
   519)             ! fracp is the fraction of cloud water converted to precip
   520)             ! NB: In below formula for fracp cwat is a LWC/IWC that has already
   521)             !     precipitated out, i.e., cwat does not contain precipitation
   522)             fracp(i) = precs(i,k)*deltat / &
   523)                        max( 1.e-12_r8, cwat(i,k) + precs(i,k)*deltat )
   524)             fracp(i) = max( 0._r8, min( 1._r8, fracp(i) ) )
   525)             
   526)             ! assume the corresponding amnt of tracer is removed
   527)             st_scav_ic(i) = sol_facti*clds(i)*fracp(i)*tracer(i,k)*rdeltat
   528) 
   529)             odds(i) = precabs(i)/max(cldvst(i,k),1.e-5_r8)*scavcoef(i,k)*deltat
   530)             odds(i) = max(min(1._r8,odds(i)),0._r8)
   531)             st_scav_bc(i) =sol_factb*(cldvst(i,k)*odds(i)) *tracer(i,k)*rdeltat
   532) 
   533)          end if
   534) 
   535)          ! total convective scavenging
   536)          srcc(i) = conv_scav_ic(i) + conv_scav_bc(i)
   537)          finc(i) = conv_scav_ic(i)/(srcc(i) + 1.e-36_r8)
   538) 
   539)          ! total stratiform scavenging
   540)          srcs(i) = st_scav_ic(i) + st_scav_bc(i)
   541)          fins(i) = st_scav_ic(i)/(srcs(i) + 1.e-36_r8)
   542) 
   543)          ! make sure we dont take out more than is there
   544)          ! ratio of amount available to amount removed
   545)          rat(i) = tracer(i,k)/max(deltat*(srcc(i)+srcs(i)),1.e-36_r8)
   546)          if (rat(i).lt.1._r8) then
   547)             srcs(i) = srcs(i)*rat(i)
   548)             srcc(i) = srcc(i)*rat(i)
   549)          endif
   550)          srct(i) = (srcc(i)+srcs(i))*omsm
   551) 
   552)             
   553)          ! fraction that is not removed within the cloud
   554)          ! (assumed to be interstitial, and subject to convective transport)
   555)          fracp(i) = deltat*srct(i)/max(cldvst(i,k)*tracer(i,k),1.e-36_r8)  ! amount removed
   556)          fracp(i) = max(0._r8,min(1._r8,fracp(i)))
   557)          fracis(i,k) = 1._r8 - fracp(i)
   558) 
   559)          ! tend is all tracer removed by scavenging, plus all re-appearing from evaporation above
   560)          ! Sungsu added cumulus contribution in the below 3 blocks
   561)          scavt(i,k) = -srct(i) + (fracev(i)*scavab(i)+fracev_cu(i)*scavabc(i))*rpdog(i)
   562)          iscavt(i,k) = -(srcc(i)*finc(i) + srcs(i)*fins(i))*omsm
   563) 
   564)          if ( present(icscavt) ) icscavt(i,k) = -(srcc(i)*finc(i)) * omsm
   565)          if ( present(isscavt) ) isscavt(i,k) = -(srcs(i)*fins(i)) * omsm
   566)          if ( present(bcscavt) ) bcscavt(i,k) = -(srcc(i) * (1-finc(i))) * omsm +  &
   567)             fracev_cu(i)*scavabc(i)*rpdog(i)
   568) 
   569)          if ( present(bsscavt) ) bsscavt(i,k) = -(srcs(i) * (1-fins(i))) * omsm +  &
   570)             fracev(i)*scavab(i)*rpdog(i)
   571)          dblchek(i) = tracer(i,k) + deltat*scavt(i,k)
   572) 
   573)          ! now keep track of scavenged mass and precip
   574)          scavab(i) = scavab(i)*(1-fracev(i)) + srcs(i)*pdog(i)
   575)          precabs(i) = precabs(i) + (precs(i,k) - evaps(i,k))*pdog(i)
   576)          scavabc(i) = scavabc(i)*(1-fracev_cu(i)) + srcc(i)*pdog(i)
   577)          precabc(i) = precabc(i) + (cmfdqr(i,k) - evapc(i,k))*pdog(i)
   578) 
   579)       end do ! End of i = 1, ncol
   580) 
   581)       found = .false.
   582)       do i = 1,ncol
   583)          if ( dblchek(i) < 0._r8 ) then
   584)             found = .true.
   585)             exit
   586)          end if
   587)       end do
   588) 
   589)       if ( found ) then
   590)          do i = 1,ncol
   591)             if (dblchek(i) .lt. 0._r8) then
   592)                write(iulog,*) ' wetdapa: negative value ', i, k, tracer(i,k), &
   593)                   dblchek(i), scavt(i,k), srct(i), rat(i), fracev(i)
   594)             endif
   595)          end do
   596)       endif
   597) 
   598)    end do ! End of k = 1, pver
   599) 
   600) end subroutine wetdepa_v2
