   122) subroutine seq_hist_write(infodata, EClock_d, &
   123)      atm, lnd, ice, ocn, rof, glc, wav, &
   124)      fractions_ax, fractions_lx, fractions_ix, fractions_ox, fractions_rx,  &
   125)      fractions_gx, fractions_wx)
   126) 
   127)    implicit none
   128)    !
   129)    ! Arguments
   130)    type(seq_infodata_type) , intent(in)    :: infodata
   131)    type (ESMF_Clock)       , intent(in)    :: EClock_d   ! driver clock
   132)    type (component_type)   , intent(inout) :: atm(:)
   133)    type (component_type)   , intent(inout) :: lnd(:)
   134)    type (component_type)   , intent(inout) :: ice(:)
   135)    type (component_type)   , intent(inout) :: ocn(:)
   136)    type (component_type)   , intent(inout) :: rof(:)
   137)    type (component_type)   , intent(inout) :: glc(:)
   138)    type (component_type)   , intent(inout) :: wav(:)
   139)    type(mct_aVect)         , intent(inout) :: fractions_ax(:) ! Fractions on atm grid/decomp
   140)    type(mct_aVect)         , intent(inout) :: fractions_lx(:) ! Fractions on lnd grid/decomp
   141)    type(mct_aVect)         , intent(inout) :: fractions_ix(:) ! Fractions on ice grid/decomp
   142)    type(mct_aVect)         , intent(inout) :: fractions_ox(:) ! Fractions on ocn grid/decomp
   143)    type(mct_aVect)         , intent(inout) :: fractions_rx(:) ! Fractions on rof grid/decomp
   144)    type(mct_aVect)         , intent(inout) :: fractions_gx(:) ! Fractions on glc grid/decomp
   145)    type(mct_aVect)         , intent(inout) :: fractions_wx(:) ! Fractions on wav grid/decomp
   146)    !
   147)    ! Local Variables
   148)    integer(IN)   :: curr_ymd     ! Current date YYYYMMDD
   149)    integer(IN)   :: curr_tod     ! Current time-of-day (s)
   150)    integer(IN)   :: start_ymd    ! Starting date YYYYMMDD
   151)    integer(IN)   :: start_tod    ! Starting time-of-day (s)
   152)    real(r8)      :: curr_time    ! Time interval since reference time
   153)    integer(IN)   :: yy,mm,dd     ! year, month, day
   154)    integer(IN)   :: fk           ! index
   155)    character(CL) :: time_units   ! units of time variable
   156)    character(CL) :: calendar     ! calendar type
   157)    character(CL) :: case_name    ! case name
   158)    character(CL) :: hist_file    ! Local path to history filename
   159)    integer(IN)   :: lsize        ! local size of an aVect
   160)    real(r8)      :: tbnds(2)     ! CF1.0 time bounds
   161)    logical       :: whead,wdata  ! for writing restart/history cdf files
   162) 
   163)    type(mct_gsMap), pointer :: gsmap
   164)    type(mct_gGrid), pointer :: dom    ! comp domain on cpl pes
   165) !-------------------------------------------------------------------------------
   166) !
   167) !-------------------------------------------------------------------------------
   168) 
   169)    !----------------------------------------------------------------------------
   170)    ! get required infodata
   171)    !----------------------------------------------------------------------------
   172)    iamin_CPLID  = seq_comm_iamin(CPLID)
   173) 
   174)    call seq_comm_getdata(GLOID,mpicom=mpicom_GLOID,nthreads=nthreads_GLOID)
   175)    call seq_comm_getdata(CPLID,mpicom=mpicom_CPLID,nthreads=nthreads_CPLID)
   176) 
   177)    call seq_infodata_getData(infodata,       &
   178)         drv_threading=drv_threading,         &
   179)         atm_present=atm_present,             &
   180)         lnd_present=lnd_present,             &
   181)         rof_present=rof_present,             &
   182)         ice_present=ice_present,             &
   183)         ocn_present=ocn_present,             &
   184)         glc_present=glc_present,             &
   185)         wav_present=wav_present,             &
   186)         atm_prognostic=atm_prognostic,       &
   187)         lnd_prognostic=lnd_prognostic,       &
   188)         ice_prognostic=ice_prognostic,       &
   189)         ocn_prognostic=ocn_prognostic,       &
   190)         ocnrof_prognostic=ocnrof_prognostic, &
   191)         rof_prognostic=rof_prognostic,       &
   192)         glc_prognostic=glc_prognostic,       &
   193)         wav_prognostic=wav_prognostic,       &
   194)         atm_nx=atm_nx, atm_ny=atm_ny,        &
   195)         lnd_nx=lnd_nx, lnd_ny=lnd_ny,        &
   196)         rof_nx=rof_nx, rof_ny=rof_ny,        &
   197)         ice_nx=ice_nx, ice_ny=ice_ny,        &
   198)         glc_nx=glc_nx, glc_ny=glc_ny,        &
   199)         wav_nx=wav_nx, wav_ny=wav_ny,        &
   200)         ocn_nx=ocn_nx, ocn_ny=ocn_ny,        &
   201)         cpl_cdf64=cdf64,                     & 
   202)         case_name=case_name)
   203) 
   204)    !--- Get current date from clock needed to label the history pointer file ---
   205) 
   206)    call seq_timemgr_EClockGetData( EClock_d, curr_ymd=curr_ymd, curr_tod=curr_tod, &
   207)         start_ymd=start_ymd, start_tod=start_tod, curr_time=curr_time, &
   208)         calendar=calendar)
   209)    call shr_cal_date2ymd(curr_ymd,yy,mm,dd)
   210)    write(hist_file,"(2a,i4.4,a,i2.2,a,i2.2,a,i5.5,a)") &
   211)       trim(case_name), '.cpl.hi.', yy,'-',mm,'-',dd,'-',curr_tod,'.nc'
   212) 
   213)    time_units = 'days since ' &
   214)         // seq_io_date2yyyymmdd(start_ymd) // ' ' // seq_io_sec2hms(start_tod)
   215) 
   216)    if (iamin_CPLID) then
   217) 
   218)       if (drv_threading) call seq_comm_setnthreads(nthreads_CPLID)
   219)       call seq_io_wopen(hist_file,clobber=.true.,cdf64=cdf64)
   220) 
   221)       ! loop twice, first time write header, second time write data for perf
   222) 
   223)       do fk = 1,2
   224)          if (fk == 1) then
   225)             whead = .true.
   226)             wdata = .false.
   227)          elseif (fk == 2) then
   228)             whead = .false.
   229)             wdata = .true.
   230)             call seq_io_enddef(hist_file)
   231)          else
   232)             call shr_sys_abort('seq_hist_write fk illegal')
   233)          end if
   234) 
   235)          tbnds = curr_time
   236)          !------- tcx nov 2011 tbnds of same values causes problems in ferret
   237)          if (tbnds(1) >= tbnds(2)) then
   238)             call seq_io_write(hist_file,&
   239)                  time_units=time_units, time_cal=calendar, time_val=curr_time, &
   240)                  whead=whead, wdata=wdata)
   241)          else
   242)             call seq_io_write(hist_file, &
   243)                  time_units=time_units, time_cal=calendar, time_val=curr_time, &
   244)                  whead=whead, wdata=wdata, tbnds=tbnds)
   245)          endif
   246) 
   247)          if (atm_present) then
   248)             gsmap => component_get_gsmap_cx(atm(1))
   249)             dom   => component_get_dom_cx(atm(1))
   250)             call seq_io_write(hist_file, gsmap, dom%data, 'dom_ax',  &
   251)                  nx=atm_nx, ny=atm_ny, nt=1, whead=whead, wdata=wdata, pre='doma')
   252)             call seq_io_write(hist_file, gsmap, fractions_ax, 'fractions_ax',  &
   253)                  nx=atm_nx, ny=atm_ny, nt=1, whead=whead, wdata=wdata, pre='fraca')
   254)             call seq_io_write(hist_file, atm, 'x2c', 'x2a_ax', &
   255)                  nx=atm_nx, ny=atm_ny, nt=1, whead=whead, wdata=wdata, pre='x2a')
   256)             call seq_io_write(hist_file, atm, 'c2x', 'a2x_ax', &
   257)                  nx=atm_nx, ny=atm_ny, nt=1, whead=whead, wdata=wdata, pre='a2x')
   258)             !call seq_io_write(hist_file, gsmap, l2x_ax, 'l2x_ax',  &
   259)             !    nx=atm_nx, ny=atm_ny, nt=1, whead=whead, wdata=wdata, pre='l2x_ax')
   260)             !call seq_io_write(hist_file, gsmap, o2x_ax, 'o2x_ax',  &
   261)             !    nx=atm_nx, ny=atm_ny, nt=1, whead=whead, wdata=wdata, pre='o2x_ax')
   262)             !call seq_io_write(hist_file, gsmap, i2x_ax, 'i2x_ax',  &
   263)             !    nx=atm_nx, ny=atm_ny, nt=1, whead=whead, wdata=wdata, pre='i2x_ax')
   264)          endif
   265) 
   266)          if (lnd_present) then
   267)             gsmap => component_get_gsmap_cx(lnd(1))
   268)             dom   => component_get_dom_cx(lnd(1))
   269)             call seq_io_write(hist_file, gsmap, dom%data, 'dom_lx',  &
   270)                  nx=lnd_nx, ny=lnd_ny, nt=1, whead=whead, wdata=wdata, pre='doml')
   271)             call seq_io_write(hist_file, gsmap, fractions_lx, 'fractions_lx',  &
   272)                  nx=lnd_nx, ny=lnd_ny, nt=1, whead=whead, wdata=wdata, pre='fracl')
   273)             call seq_io_write(hist_file, lnd, 'c2x', 'l2x_lx', &
   274)                  nx=lnd_nx, ny=lnd_ny, nt=1, whead=whead, wdata=wdata, pre='l2x')
   275)             call seq_io_write(hist_file, lnd, 'x2c', 'x2l_lx',&
   276)                  nx=lnd_nx, ny=lnd_ny, nt=1, whead=whead, wdata=wdata, pre='x2l')
   277)          endif
   278) 
   279)          if (rof_present) then
   280)             gsmap => component_get_gsmap_cx(rof(1))
   281)             dom   => component_get_dom_cx(rof(1))
   282)             call seq_io_write(hist_file, gsmap, dom%data, 'dom_rx',  &
   283)                  nx=rof_nx, ny=rof_ny, nt=1, whead=whead, wdata=wdata, pre='domr')
   284)             call seq_io_write(hist_file, gsmap, fractions_rx, 'fractions_rx',  &
   285)                  nx=rof_nx, ny=rof_ny, nt=1, whead=whead, wdata=wdata, pre='fracr')
   286)             call seq_io_write(hist_file, rof, 'c2x', 'r2x_rx', &
   287)                  nx=rof_nx, ny=rof_ny, nt=1, whead=whead, wdata=wdata, pre='r2x')
   288)             call seq_io_write(hist_file, rof, 'x2c', 'x2r_rx', &
   289)                  nx=rof_nx, ny=rof_ny, nt=1, whead=whead, wdata=wdata, pre='x2r')
   290)          endif
   291) 
   292)          if (rof_present .and. ocnrof_prognostic) then
   293)             gsmap  => component_get_gsmap_cx(ocn(1))
   294)             r2x_ox => prep_ocn_get_r2x_ox()
   295)             call seq_io_write(hist_file, gsmap, r2x_ox, 'r2x_ox',  &
   296)                  nx=ocn_nx, ny=ocn_ny, nt=1, whead=whead, wdata=wdata, pre='r2xo')
   297)          endif
   298) 
   299)          if (ocn_present) then
   300)             gsmap => component_get_gsmap_cx(ocn(1))
   301)             dom   => component_get_dom_cx(ocn(1))
   302)             call seq_io_write(hist_file, gsmap, dom%data, 'dom_ox',  &
   303)                               nx=ocn_nx, ny=ocn_ny, nt=1, whead=whead, wdata=wdata, pre='domo')
   304)             call seq_io_write(hist_file, gsmap, fractions_ox, 'fractions_ox',  &
   305)                               nx=ocn_nx, ny=ocn_ny, nt=1, whead=whead, wdata=wdata, pre='fraco')
   306)             call seq_io_write(hist_file, ocn, 'c2x', 'o2x_ox', &
   307)                               nx=ocn_nx, ny=ocn_ny, nt=1, whead=whead, wdata=wdata, pre='o2x')
   308)             !call seq_io_write(hist_file, ocn, 'x2c', 'x2o_ox', &
   309)             !                  nx=ocn_nx, ny=ocn_ny, nt=1, whead=whead, wdata=wdata, pre='x2o')
   310) 
   311)             gsmap     => component_get_gsmap_cx(ocn(1))
   312)             x2oacc_ox => prep_ocn_get_x2oacc_ox()
   313)             call seq_io_write(hist_file, gsmap, x2oacc_ox, 'x2oacc_ox',  &
   314)                  nx=ocn_nx, ny=ocn_ny, nt=1, whead=whead, wdata=wdata, pre='x2oacc')
   315) 
   316)             gsmap         => component_get_gsmap_cx(ocn(1))
   317)             x2oacc_ox_cnt => prep_ocn_get_x2oacc_ox_cnt()
   318)             call seq_io_write(hist_file, x2oacc_ox_cnt, 'x2oacc_ox_cnt',  &
   319)                  whead=whead, wdata=wdata)
   320)             gsmap => component_get_gsmap_cx(ocn(1))
   321)             xao_ox => prep_aoflux_get_xao_ox()
   322)             call seq_io_write(hist_file, gsmap, xao_ox, 'xao_ox',  &
   323)                  nx=ocn_nx, ny=ocn_ny, nt=1, whead=whead, wdata=wdata, pre='xaoo')
   324) 
   325)             gsmap  => component_get_gsmap_cx(atm(1))
   326)             o2x_ax => prep_atm_get_o2x_ax()
   327)             call seq_io_write(hist_file, gsmap, o2x_ax, 'o2x_ax',  &
   328)                  nx=atm_nx, ny=atm_ny, nt=1, whead=whead, wdata=wdata, pre='o2xa')
   329) 
   330)             gsmap  => component_get_gsmap_cx(atm(1))
   331)             xao_ax => prep_aoflux_get_xao_ax()
   332)             call seq_io_write(hist_file, gsmap, xao_ax, 'xao_ax',  &
   333)                  nx=atm_nx, ny=atm_ny, nt=1, whead=whead, wdata=wdata, pre='xaoa')
   334)          endif
   335) 
   336)          if (ice_present) then
   337)             gsmap => component_get_gsmap_cx(ice(1))
   338)             dom   => component_get_dom_cx(ice(1))
   339)             call seq_io_write(hist_file, gsmap, dom%data, 'dom_ix',  &
   340)                  nx=ice_nx, ny=ice_ny, nt=1, whead=whead, wdata=wdata, pre='domi')
   341)             call seq_io_write(hist_file, gsmap, fractions_ix, 'fractions_ix',  &
   342)                  nx=ice_nx, ny=ice_ny, nt=1, whead=whead, wdata=wdata, pre='fraci')
   343)             call seq_io_write(hist_file, ice, 'c2x', 'i2x_ix', &
   344)                  nx=ice_nx, ny=ice_ny, nt=1, whead=whead, wdata=wdata, pre='i2x')
   345)             call seq_io_write(hist_file, ice, 'x2c', 'x2i_ix', &
   346)                  nx=ice_nx, ny=ice_ny, nt=1, whead=whead, wdata=wdata, pre='x2i')
   347)          endif
   348) 
   349)          if (glc_present) then
   350)             gsmap => component_get_gsmap_cx(glc(1))
   351)             dom   => component_get_dom_cx(glc(1))
   352)             call seq_io_write(hist_file, gsmap, dom%data, 'dom_gx',  &
   353)                  nx=glc_nx, ny=glc_ny, nt=1, whead=whead, wdata=wdata, pre='domg')
   354)             call seq_io_write(hist_file, gsmap, fractions_gx, 'fractions_gx',  &
   355)                  nx=glc_nx, ny=glc_ny, nt=1, whead=whead, wdata=wdata, pre='fracg')
   356)             call seq_io_write(hist_file, glc, 'c2x', 'g2x_gx', &
   357)                  nx=glc_nx, ny=glc_ny, nt=1, whead=whead, wdata=wdata, pre='g2x')
   358)             call seq_io_write(hist_file, glc, 'x2c', 'x2g_gx', &
   359)                  nx=glc_nx, ny=glc_ny, nt=1, whead=whead, wdata=wdata, pre='x2g')
   360)          endif
   361) 
   362)          if (wav_present) then
   363)             gsmap => component_get_gsmap_cx(wav(1))
   364)             dom   => component_get_dom_cx(wav(1))
   365)             call seq_io_write(hist_file, gsmap, dom%data, 'dom_wx',  &
   366)                  nx=wav_nx, ny=wav_ny, nt=1, whead=whead, wdata=wdata, pre='domw')

   921)    type(mct_avect), pointer :: av

   923)    character(CL)            :: case_name  ! case name

   926)    integer(IN)              :: start_ymd  ! Starting date YYYYMMDD

   928)    real(r8)                 :: curr_time  ! Time interval since reference time

   930)    real(r8)                 :: avg_time   ! Average time for time average
   931)    integer(IN)              :: yy, mm, dd ! year,  month,  day

   936)    integer(IN)              :: lsize      ! local size of an aVect

   938)    integer(IN)              :: found = -10
   939)    logical                  :: useavg
   940)    logical                  :: lwrite_now     
   941)    logical                  :: whead, wdata  ! for writing restart/history cdf files
   942)    real(r8)                 :: tbnds(2)
   943) 
   944)    integer(IN), parameter   :: maxout = 20
   945)    integer(IN)       , save :: ntout = 0
   946)    character(CS)     , save :: tname(maxout) = 'x1y2z3'
   947)    integer(IN)       , save :: ncnt(maxout)  = -10
   948)    character(CL)     , save :: hist_file(maxout)       ! local path to history filename

   951)    logical           , save :: fwrite(maxout) = .true. ! first write
   952)    real(r8)          , save :: tbnds1(maxout)          ! first time_bnds

   954)

   960) !
   961) !-------------------------------------------------------------------------------
   962)

   967)

   969)         mpicom=mpicom_GLOID, nthreads=nthreads_GLOID)
   970)    call seq_comm_getdata(CPLID, &
   971)         mpicom=mpicom_CPLID, nthreads=nthreads_CPLID)
   972) 
   973)    call seq_infodata_getData(infodata, &
   974)         drv_threading=drv_threading,   &
   975)         atm_present=atm_present,       &
   976)         lnd_present=lnd_present,       &
   977)         rof_present=rof_present,       &
   978)         ice_present=ice_present,       &
   979)         ocn_present=ocn_present,       &
   980)         glc_present=glc_present,       &
   981)         wav_present=wav_present,       &
   982)         cpl_cdf64=cdf64 )

   985)    useavg = .false.
   986)    if (present(write_now)) then
   987)       useavg = .true.
   988)       lwrite_now = write_now
   989)    endif
   990)  
   991)    call seq_timemgr_EClockGetData( EClock_d, &
   992)         curr_ymd=curr_ymd,                   &
   993)         curr_tod=curr_tod,                   &
   994)         start_ymd=start_ymd,                 &
   995)         start_tod=start_tod,                 &
   996)         curr_time=curr_time,                 &
   997)         prev_time=prev_time,                 &
   998)         calendar=calendar)
   999)

  1001)    do n = 1, ntout
  1002)       if (trim(tname(n)) == trim(aname)) then
  1003)          first_call = .false.
  1004)          found = n
  1005)       endif
  1006)    enddo
  1007) 
  1008)    if (first_call) then
  1009)       ntout = ntout + 1
  1010)       if (ntout > maxout) then
  1011)          write(logunit, *) 'write_history_writeaux maxout exceeded', ntout, maxout

  1013)       endif
  1014)       tname(ntout) = trim(aname)
  1015)       ncnt(ntout) = -10

  1017)          lsize = mct_aVect_lsize(av)
  1018)          call mct_aVect_init(avavg(ntout), av, lsize)
  1019)          call mct_aVect_zero(avavg(ntout))
  1020)          avcnt(ntout) = 0
  1021)       endif
  1022)       tbnds1(ntout) = prev_time
  1023)       found = ntout
  1024)    endif
  1025) 
  1026)    if (iamin_CPLID) then !>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
  1027) 
  1028)       dom   => component_get_dom_cx(comp)
  1029)       gsmap => component_get_gsmap_cx(comp)

  1081)       tbnds(2) = tbnds2(found)
  1082)

  1085)             whead = .true.

  1090)          else
  1091)             call shr_sys_abort('seq_hist_writeaux fk illegal')

  1095)             if (fk == fk1) then
  1096)                lsize = mct_aVect_lsize(av)

  1122)                call seq_io_write(hist_file(found),  gsmap,  avflds,  trim(aname),  &

  1125)             else
  1126)                call seq_io_write(hist_file(found),  gsmap,  avavg(found),  trim(aname),  &

  1128)                                  pre=trim(aname), tavg=.true.,  use_float=.true.)
  1129)             end if
  1130)          else if (present(flds)) then
  1131)             call mct_aVect_copy(aVin=av,  aVout=avflds)
  1132)             call seq_io_write(hist_file(found),  gsmap,  avflds,  trim(aname),  &
  1133)                               nx=nx, ny=ny, nt=ncnt(found), whead=whead, wdata=wdata, pre=trim(aname), &
