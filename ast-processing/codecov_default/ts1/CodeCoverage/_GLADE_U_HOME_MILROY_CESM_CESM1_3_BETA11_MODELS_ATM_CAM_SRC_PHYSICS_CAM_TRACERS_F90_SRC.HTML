    72) subroutine tracers_register
    73) !----------------------------------------------------------------------- 
    74) !
    75) ! Purpose: register advected tracers. Called by initindx.F90
    76) !  The registration lets the model know what the tracer names are
    77) !  and returns the index number ixtrct for the constituent array
    78) ! 
    79) ! Author: D. Bundy
    80) !-----------------------------------------------------------------------
    81) 
    82)    use physconst,    only: mwdry, cpair
    83)    use constituents, only: cnst_add, cnst_num_avail
    84)    use tracers_suite, only: get_tracer_name
    85)    
    86)    implicit none
    87) !---------------------------Local workspace-----------------------------
    88)    integer :: mm,m                                 ! dummy
    89)    character(len=8) :: name   ! constituent name
    90)    real(r8) minc
    91) 
    92) !-----------------------------------------------------------------------
    93)    if ( tracers_flag ) then 
    94)       minc = 0        ! min mixing ratio (normal setting)
    95)       minc = -1.e36_r8   ! min mixing ratio (disable qneg3)
    96)       
    97)       ! Set the number of test tracers equal to the number of slots available
    98)       ! in the constituent array
    99)       trac_ncnst = cnst_num_avail()
   100) 
   101)       do m = 1,trac_ncnst 
   102)          name = get_tracer_name(m)  ! get name from suite file
   103)          
   104)          ! add constituent name to list of advected, save index number ixtrct
   105)          call cnst_add(name, mwdry, cpair, minc, mm, &  
   106)               readiv=.false.,mixtype='dry')
   107)          if ( m .eq. 1 ) ixtrct = mm  ! save index number of first tracer
   108)          
   109)       end do
   110)    end if
   111)

   243) subroutine tracers_timestep_tend(state, ptend, cflx, landfrac, deltat)
   244) 
   245) !----------------------------------------------------------------------- 
   246) !
   247) ! Purpose: During the timestep, compute test tracer mixing ratio 
   248) ! tendencies and surface fluxes.
   249) ! 
   250) ! Author: D. Bundy
   251) !-----------------------------------------------------------------------
   252) 
   253)   use physics_types, only: physics_state, physics_ptend, physics_ptend_init
   254)   use ppgrid,        only: pcols, pver
   255)   use constituents,  only: pcnst, sflxnam, cnst_cam_outfld
   256)   use tracers_suite, only: flux_tr, tend_tr
   257)   use cam_history,   only: outfld
   258) 
   259)   implicit none
   260) 
   261)   ! Arguments
   262)    type(physics_state), intent(in)  :: state          ! state variables
   263)    type(physics_ptend), intent(out) :: ptend          ! package tendencies
   264)    real(r8),            intent(in)  :: deltat         ! timestep
   265)    real(r8),            intent(in)  :: landfrac(pcols) ! Land fraction
   266)    real(r8), intent(inout) :: cflx(pcols,pcnst) ! Surface constituent flux (kg/m^2/s)
   267) 
   268) ! Local variables
   269)    integer  :: m               ! tracer number (internal)
   270) 
   271)    logical  :: lq(pcnst)
   272) 
   273) !-----------------------------------------------------------------------
   274) 
   275)   if (.not. tracers_flag) then
   276)        call physics_ptend_init(ptend,state%psetcols,'none') !Initialize an empty ptend for use with physics_update
   277)        return
   278)   else
   279)      lq(:)      = .FALSE.
   280)      lq(ixtrct:ixtrct+trac_ncnst-1) = .TRUE.
   281)      call physics_ptend_init(ptend, state%psetcols, 'tracers', lq=lq)
   282)  
   283)      do  m = 1,trac_ncnst 
   284)         if (debug) write(iulog,*)'tracers.F90 calling for tracer ',m
   285)         
   286)         !calculate flux
   287)         call flux_tr(m,state%ncol,state%lchnk, landfrac, cflx(:,ixtrct+m-1))
   288)         
   289)         !calculate tendency
   290)         call tend_tr(m,state%ncol, state%q(:,:,ixtrct+m-1), deltat, ptend%q(:,:,ixtrct+m-1))
   291)         
   292)         !outfld calls could go here
   293)         if ( cnst_cam_outfld(ixtrct+m-1) ) then
   294)            call outfld (sflxnam(ixtrct+m-1),cflx(:,ixtrct+m-1),pcols,state%lchnk)
   295)         end if
   296)         
   297)         
   298)      end do
   299)      
   300)      if ( debug ) then 
   301)         do  m = 1,trac_ncnst 
   302)            write(iulog,*)'tracers_timestep_tend ixtrct,m,ixtrct+m-1',ixtrct,m,ixtrct+m-1
   303)            write(iulog,*)'tracers_timestep_tend min max flux',minval(cflx(:,ixtrct+m-1)),maxval(cflx(:,ixtrct+m-1))
   304)            write(iulog,*)'tracers_timestep_tend min max tend',minval(ptend%q(:,:,ixtrct+m-1)),maxval(ptend%q(:,:,ixtrct+m-1))
   305)         end do
   306)         write(iulog,*)'tracers_timestep_tend end'
   307)      endif
   308)   endif
   309)
