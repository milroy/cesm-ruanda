   129)   subroutine seq_rest_read(rest_file, infodata, &
   130)        atm, lnd, ice, ocn, rof, glc, wav, &
   131)        fractions_ax, fractions_lx, fractions_ix, fractions_ox, &
   132)        fractions_rx, fractions_gx, fractions_wx)
   133) 
   134)    implicit none
   135) 
   136)    character(*)           , intent(in) :: rest_file  ! restart file path/name
   137)    type(seq_infodata_type), intent(in) :: infodata
   138)    type (component_type) , intent(inout) :: atm(:)
   139)    type (component_type) , intent(inout) :: lnd(:)
   140)    type (component_type) , intent(inout) :: ice(:)
   141)    type (component_type) , intent(inout) :: ocn(:)
   142)    type (component_type) , intent(inout) :: rof(:)
   143)    type (component_type) , intent(inout) :: glc(:)
   144)    type (component_type) , intent(inout) :: wav(:)
   145)    type(mct_aVect)  , intent(inout) :: fractions_ax(:)   ! Fractions on atm grid/decomp
   146)    type(mct_aVect)  , intent(inout) :: fractions_lx(:)   ! Fractions on lnd grid/decomp
   147)    type(mct_aVect)  , intent(inout) :: fractions_ix(:)   ! Fractions on ice grid/decomp
   148)    type(mct_aVect)  , intent(inout) :: fractions_ox(:)   ! Fractions on ocn grid/decomp
   149)    type(mct_aVect)  , intent(inout) :: fractions_rx(:)   ! Fractions on rof grid/decomp
   150)    type(mct_aVect)  , intent(inout) :: fractions_gx(:)   ! Fractions on glc grid/decomp
   151)    type(mct_aVect)  , intent(inout) :: fractions_wx(:)   ! Fractions on wav grid/decomp
   152) 
   153)    integer(IN)          :: n,n1,n2,n3
   154)    real(r8),allocatable :: ds(:)         ! for reshaping diag data for restart file
   155)    real(r8),allocatable :: ns(:)         ! for reshaping diag data for restart file
   156)    character(CS)        :: string
   157)    integer(IN)          :: ierr          ! MPI error return
   158)    character(len=*), parameter :: subname = "(seq_rest_read) "
   159) 
   160) !-------------------------------------------------------------------------------
   161) !
   162) !-------------------------------------------------------------------------------
   163) 
   164)    !----------------------------------------------------------------------------
   165)    ! get required infodata
   166)    !----------------------------------------------------------------------------
   167)    iamin_CPLID  = seq_comm_iamin(CPLID)
   168) 
   169)    call seq_comm_getdata(GLOID,&
   170)         mpicom=mpicom_GLOID, nthreads=nthreads_GLOID)
   171)    call seq_comm_getdata(CPLID, &
   172)         mpicom=mpicom_CPLID, nthreads=nthreads_CPLID)
   173) 
   174)    call seq_infodata_getData(infodata,		&
   175)         drv_threading=drv_threading,		&
   176)         atm_present=atm_present,		&
   177)         lnd_present=lnd_present,		&
   178)         rof_present=rof_present,		&
   179)         ice_present=ice_present,		&
   180)         ocn_present=ocn_present,		&
   181)         glc_present=glc_present,		&
   182)         wav_present=wav_present,		&
   183)         atm_prognostic=atm_prognostic,		&
   184)         lnd_prognostic=lnd_prognostic,		&
   185)         ice_prognostic=ice_prognostic,		&
   186)         ocn_prognostic=ocn_prognostic,		&
   187)         rof_prognostic=rof_prognostic,		&
   188)         ocnrof_prognostic=ocnrof_prognostic,	&
   189)         glc_prognostic=glc_prognostic,		&
   190)         wav_prognostic=wav_prognostic)
   191) 
   192)    if (iamin_CPLID) then
   193)       if (drv_threading) call seq_comm_setnthreads(nthreads_CPLID)
   194)       if (atm_present) then
   195)          gsmap => component_get_gsmap_cx(atm(1))
   196)          call seq_io_read(rest_file, gsmap, fractions_ax, 'fractions_ax')
   197)          call seq_io_read(rest_file, atm, 'c2x', 'a2x_ax')
   198)       endif
   199)       if (lnd_present) then
   200)          gsmap => component_get_gsmap_cx(lnd(1))
   201)          call seq_io_read(rest_file, gsmap, fractions_lx, 'fractions_lx')
   202)       endif
   203)       if (lnd_present .and. rof_prognostic) then
   204)          gsmap         => component_get_gsmap_cx(lnd(1))
   205)          l2racc_lx     => prep_rof_get_l2racc_lx()
   206)          l2racc_lx_cnt => prep_rof_get_l2racc_lx_cnt()
   207)          call seq_io_read(rest_file, gsmap, l2racc_lx, 'l2racc_lx')
   208)          call seq_io_read(rest_file, l2racc_lx_cnt ,'l2racc_lx_cnt')
   209)       end if
   210)       if (lnd_present .and. glc_prognostic) then
   211)          gsmap         => component_get_gsmap_cx(lnd(1))
   212)          l2gacc_lx     => prep_glc_get_l2gacc_lx()
   213)          l2gacc_lx_cnt => prep_glc_get_l2gacc_lx_cnt()
   214)          call seq_io_read(rest_file, gsmap, l2gacc_lx, 'l2gacc_lx')
   215)          call seq_io_read(rest_file, l2gacc_lx_cnt ,'l2gacc_lx_cnt')
   216)       end if
   217)       if (ocn_present) then
   218)          gsmap         => component_get_gsmap_cx(ocn(1))
   219)          x2oacc_ox     => prep_ocn_get_x2oacc_ox()
   220)          x2oacc_ox_cnt => prep_ocn_get_x2oacc_ox_cnt()
   221)          xao_ox        => prep_aoflux_get_xao_ox()
   222)          call seq_io_read(rest_file, gsmap, fractions_ox, 'fractions_ox')
   223)          call seq_io_read(rest_file, gsmap, x2oacc_ox, 'x2oacc_ox')
   224)          call seq_io_read(rest_file, x2oacc_ox_cnt, 'x2oacc_ox_cnt')
   225)          call seq_io_read(rest_file, gsmap, xao_ox, 'xao_ox')
   226)          call seq_io_read(rest_file, ocn, 'c2x', 'o2x_ox')  ! get o2x_ox
   227)       endif
   228)       if (ice_present) then
   229)          gsmap => component_get_gsmap_cx(ice(1))
   230)          call seq_io_read(rest_file, gsmap, fractions_ix, 'fractions_ix')
   231)          call seq_io_read(rest_file, ice, 'c2x', 'i2x_ix') 
   232)       endif
   233)       if (rof_present) then
   234)          gsmap => component_get_gsmap_cx(rof(1))
   235)          call seq_io_read(rest_file, gsmap, fractions_rx, 'fractions_rx')
   236)          call seq_io_read(rest_file, rof, 'c2x', 'r2x_rx') 
   237)       endif
   238)       if (glc_present) then
   239)          gsmap => component_get_gsmap_cx(glc(1))
   240)          call seq_io_read(rest_file, gsmap, fractions_gx, 'fractions_gx')
   241)          call seq_io_read(rest_file, glc, 'c2x', 'g2x_gx') 
   242)       endif
   243)       if (wav_present) then
   244)          gsmap => component_get_gsmap_cx(wav(1))
   245)          call seq_io_read(rest_file, gsmap, fractions_wx, 'fractions_wx')
   246)          call seq_io_read(rest_file, wav, 'c2x', 'w2x_wx') 
   247)       endif
   248) 
   249)       n = size(budg_dataG)
   250)       allocate(ds(n),ns(n))
   251)       call seq_io_read(rest_file, ds, 'budg_dataG')
   252)       call seq_io_read(rest_file, ns, 'budg_ns')
   253) 
   254)       n = 0
   255)       do n1 = 1,size(budg_dataG,dim=1)
   256)       do n2 = 1,size(budg_dataG,dim=2)
   257)       do n3 = 1,size(budg_dataG,dim=3)
   258)          n = n + 1
   259)          budg_dataG(n1,n2,n3) = ds(n)
   260)          budg_ns   (n1,n2,n3) = ns(n)
   261)       enddo
   262)       enddo
   263)       enddo
   264) !     call shr_mpi_bcast(budg_dataG,cpl_io_root) ! not necessary, io lib does bcast
   265) 
   266)       deallocate(ds,ns)
   267) 
   268)       if (drv_threading) call seq_comm_setnthreads(nthreads_GLOID)
   269) 
   270)    endif
   271) 
   272) end subroutine seq_rest_read

   276) subroutine seq_rest_write(EClock_d, seq_SyncClock, infodata, &
   277)      atm, lnd, ice, ocn, rof, glc, wav,                      &
   278)      fractions_ax, fractions_lx, fractions_ix, fractions_ox, &
   279)      fractions_rx, fractions_gx, fractions_wx)
   280) 
   281)    implicit none
   282) 
   283)    type(ESMF_Clock)       , intent(in)    :: EClock_d      ! driver clock
   284)    type(seq_timemgr_type) , intent(inout) :: seq_SyncClock ! contains ptr to driver clock
   285)    type(seq_infodata_type), intent(in)    :: infodata
   286)    type (component_type)       , intent(inout) :: atm(:)
   287)    type (component_type)       , intent(inout) :: lnd(:)
   288)    type (component_type)       , intent(inout) :: ice(:)
   289)    type (component_type)       , intent(inout) :: ocn(:)
   290)    type (component_type)       , intent(inout) :: rof(:)
   291)    type (component_type)       , intent(inout) :: glc(:)
   292)    type (component_type)       , intent(inout) :: wav(:)
   293)    type(mct_aVect)        , intent(inout) :: fractions_ax(:)   ! Fractions on atm grid/decomp
   294)    type(mct_aVect)        , intent(inout) :: fractions_lx(:)   ! Fractions on lnd grid/decomp
   295)    type(mct_aVect)        , intent(inout) :: fractions_ix(:)   ! Fractions on ice grid/decomp
   296)    type(mct_aVect)        , intent(inout) :: fractions_ox(:)   ! Fractions on ocn grid/decomp
   297)    type(mct_aVect)        , intent(inout) :: fractions_rx(:)   ! Fractions on rof grid/decomp
   298)    type(mct_aVect)        , intent(inout) :: fractions_gx(:)   ! Fractions on glc grid/decomp
   299)    type(mct_aVect)        , intent(inout) :: fractions_wx(:)   ! Fractions on wav grid/decomp
   300) 
   301)    integer(IN)   :: n,n1,n2,n3,fk
   302)    integer(IN)   :: curr_ymd         ! Current date YYYYMMDD
   303)    integer(IN)   :: curr_tod         ! Current time-of-day (s)
   304)    integer(IN)   :: yy,mm,dd         ! year, month, day
   305)    character(CL) :: case_name        ! case name
   306)    character(CL) :: cvar             ! char variable
   307)    integer(IN)   :: ivar             ! integer variable
   308)    real(r8)      :: rvar             ! real variable
   309)    logical       :: whead,wdata      ! flags header/data writing
   310)    logical       :: cdf64            ! true => create netCDF with 64 bit addressing
   311)    logical       :: cplroot          ! root pe on cpl id
   312)    integer(IN)   :: iun              ! unit number
   313)    character(CL) :: rest_file        ! Local path to restart filename
   314)    integer(IN)   :: ierr             ! MPI error return
   315)    type(mct_gsMap),pointer :: gsmap
   316) 
   317)    real(r8),allocatable :: ds(:)     ! for reshaping diag data for restart file
   318)    real(r8),allocatable :: ns(:)     ! for reshaping diag data for restart file
   319)    character(len=*),parameter :: subname = "(seq_rest_write) "
   320) 
   321) !-------------------------------------------------------------------------------
   322) !
   323) !-------------------------------------------------------------------------------
   324) 
   325)    !----------------------------------------------------------------------------
   326)    ! get required infodata
   327)    !----------------------------------------------------------------------------
   328)    iamin_CPLID  = seq_comm_iamin(CPLID)
   329) 
   330)    call seq_comm_getdata(GLOID,&
   331)         mpicom=mpicom_GLOID, nthreads=nthreads_GLOID)
   332) 
   333)    call seq_comm_getdata(CPLID,&
   334)         mpicom=mpicom_CPLID, nthreads=nthreads_CPLID, iamroot=cplroot)
   335) 
   336)    call seq_infodata_getData(infodata,		&
   337)         drv_threading=drv_threading,		&
   338)         atm_present=atm_present,		&
   339)         lnd_present=lnd_present,		&
   340)         rof_present=rof_present,		&
   341)         ice_present=ice_present,		&
   342)         ocn_present=ocn_present,		&
   343)         glc_present=glc_present,		&
   344)         wav_present=wav_present,		&
   345)         atm_prognostic=atm_prognostic,		&
   346)         lnd_prognostic=lnd_prognostic,		&
   347)         ice_prognostic=ice_prognostic,		&
   348)         rof_prognostic=rof_prognostic,		&
   349)         ocn_prognostic=ocn_prognostic,		&
   350)         ocnrof_prognostic=ocnrof_prognostic,	&
   351)         glc_prognostic=glc_prognostic,		&
   352)         wav_prognostic=wav_prognostic,		&
   353)         cpl_cdf64=cdf64,			& 
   354)         case_name=case_name)
   355) 
   356)    ! Write out infodata and time manager data to restart file
   357) 
   358)    call seq_timemgr_EClockGetData( EClock_d, curr_ymd=curr_ymd, curr_tod=curr_tod)
   359)    call shr_cal_date2ymd(curr_ymd,yy,mm,dd)
   360)    write(rest_file,"(2a,i4.4,a,i2.2,a,i2.2,a,i5.5,a)") &
   361)       trim(case_name), '.cpl.r.', yy,'-',mm,'-',dd,'-',curr_tod,'.nc'
   362) 
   363)    ! Write driver data to restart file
   364) 
   365)    if (iamin_CPLID) then
   366) 
   367)       if (drv_threading) call seq_comm_setnthreads(nthreads_CPLID)
   368) 
   369)       ! copy budg_dataG into 1d array
   370)       n = size(budg_dataG)
   371)       allocate(ds(n),ns(n))
   372)       call shr_mpi_bcast(budg_dataG,mpicom_CPLID) ! pio requires data on all pe's?
   373) 
   374)       n = 0
   375)       do n1 = 1,size(budg_dataG,dim=1)
   376)       do n2 = 1,size(budg_dataG,dim=2)
   377)       do n3 = 1,size(budg_dataG,dim=3)
   378)          n = n + 1
   379)          ds(n) = budg_dataG(n1,n2,n3)
   380)          ns(n) = budg_ns(n1,n2,n3)
   381)       enddo
   382)       enddo
   383)       enddo
   384) 
   385)       if (cplroot) then
   386)          iun = shr_file_getUnit()
   387)          call seq_infodata_GetData(infodata,restart_pfile=cvar)
   388)          if (loglevel > 0) write(logunit,"(3A)") subname," write rpointer file ", &
   389)             trim(cvar)
   390)          open(iun, file=cvar, form='FORMATTED')
   391)          write(iun,'(a)') rest_file
   392)          close(iun)
   393)          call shr_file_freeUnit( iun )
   394)       endif
   395) 
   396)       call shr_mpi_bcast(rest_file,mpicom_CPLID)
   397)       call seq_io_wopen(rest_file,clobber=.true.,cdf64=cdf64)
   398) 
   399)       ! loop twice (for perf), first time write header, second time write data
   400)       do fk = 1,2
   401)          if (fk == 1) then
   402)             whead = .true.
   403)             wdata = .false.
   404) !!            call seq_io_redef(rest_file)
   405)          elseif (fk == 2) then
   406)             whead = .false.
   407)             wdata = .true.
   408)             call seq_io_enddef(rest_file)
   409)          else
   410)             call shr_sys_abort('driver_write_rstart fk illegal')
   411)          end if
   412)          call seq_infodata_GetData(infodata,nextsw_cday=rvar)
   413)          call seq_io_write(rest_file,rvar,'seq_infodata_nextsw_cday',whead=whead,wdata=wdata)
   414)          call seq_infodata_GetData(infodata,precip_fact=rvar)
   415)          call seq_io_write(rest_file,rvar,'seq_infodata_precip_fact',whead=whead,wdata=wdata)
   416)          call seq_infodata_GetData(infodata,case_name=cvar)
   417)          call seq_io_write(rest_file,trim(cvar),'seq_infodata_case_name',whead=whead,wdata=wdata)
   418) 
   419)          call seq_timemgr_EClockGetData( EClock_d, start_ymd=ivar)
   420)          call seq_io_write(rest_file,ivar,'seq_timemgr_start_ymd',whead=whead,wdata=wdata)
   421)          call seq_timemgr_EClockGetData( EClock_d, start_tod=ivar)
   422)          call seq_io_write(rest_file,ivar,'seq_timemgr_start_tod',whead=whead,wdata=wdata)
