    41)    ! arguments
    42)     real(r8), intent(in) :: o_den(:)           ! N density (molecules/cm^3)
    43)     real(r8), intent(in) :: o2_den(:)          ! O2 density (molecules/cm^3)
    44)     real(r8), intent(in) :: n2_den(:)          ! N2 density (molecules/cm^3)
    45)     real(r8), intent(in) :: no_den(:)          ! NO density (molecules/cm^3)
    46)     real(r8), intent(in) :: zint(:)            ! interface height (km)
    47) 
    48)     real(r8), intent(inout) :: rates(:,:)      ! photolysis rates (sec-1)
    49) 
    50)     real(r8), intent(out), optional :: qbko1_out(:) ! rate (cm-3 sec-1) of O  + hv ->  Op  + e 
    51)     real(r8), intent(out), optional :: qbko2_out(:) ! rate (cm-3 sec-1) of O2 + hv ->  O2p + e 
    52)     real(r8), intent(out), optional :: qbkn2_out(:) ! rate (cm-3 sec-1) of N2 + hv ->  N2p + e 
    53)     real(r8), intent(out), optional :: qbkn1_out(:) ! rate (cm-3 sec-1) of N  + hv ->  Np  + e

   105)     if (present(qbko2_out)) qbko2_out(:) = qbko2(:)

   107)     if (present(qbkn1_out)) qbkn1_out(:) = qbkn1(:)

   115) !
   116) ! Stan Solomon, 11/88, 11/92
   117) ! Comment updated 3/05
   118) ! New version uses updated TIE-GCM and TIME-GCM qinite.F formulation, 1/13
   119) !
   120) ! Estimates background ("nighttime") ionization rates.
   121) ! Four components are used:
   122) !    Geocoronal Lyman-beta 102.6 nm (ionizes O2 only)
   123) !    Geocoronal He I 58.4 nm
   124) !    Geocoronal He II 30.4 nm
   125) !    Geocoronal Lyman-alpha 121.6 nm (ionizes NO only)
   126) !
   127) ! Definitions:
   128) !
   129) ! zmaj    major species O, O2, N2 at each altitude
   130) ! zno     nitric oxide at each altitude
   131) ! zvcd    vertical column density for each major species above each altitude
   132) ! photoi  photoionization rates for each state, species, altitude
   133) ! f107    solar 10.7 cm radio flux activity index
   134) ! jm      number of altitude levels
   135) ! nmaj    number of major species (3)
   136) ! nst     number of states
   137) !
   138) ! al      photon flux at 102.6 nm, 58.4 nm, 30.4 nm
   139) ! flyan   photon flux at 121.6 nm
   140) ! sa      absorption cross sections for O, O2, N2 at each wavelength
   141) ! si      ionization cross sections for O, O2, N2 at each wavelength
   142) ! flyan   photon flux at 121.6 nm
   143) ! salyao2 absorption cross section for O2 at 121.6 nm
   144) ! silyano ionization cross section for NO at 121.6 nm
   145) ! bn2p    branching ratio for N2+ from ionization of N2
   146) ! bn1p    branching ratio for N+ from ionization of N2
   147) ! tau     optical depth
   148) ! qbko1    production rate of O+
   149) ! qbko2    production rate of O2+
   150) ! qbkn2    production rate of N2+
   151) ! qbkn1    production rate of N+
   152) ! qbkno    production rate of NO+
   153) !
   154) ! All units cgs.
   155) !
   156) !
   157) subroutine qback (zmaj,zno,zvcd,f107,nmaj,jm,qbko1,qbko2,qbkn2,qbkn1,qbkno)
   158) 
   159)  ! args:
   160)   integer,  intent(in)  :: nmaj,jm
   161)   real(r8), intent(in)  :: f107
   162)   real(r8), intent(in)  :: zmaj(nmaj,jm), zno(jm), zvcd(nmaj,jm)
   163)   real(r8), intent(out) :: qbko1(jm),qbko2(jm),qbkn2(jm),qbkn1(jm),qbkno(jm)
   164) 
   165)  ! local vars:
   166)   real(r8) :: al(3), sa(3,3), si(3,3)
   167)   real(r8) :: salyao2, silyano, bn2p, bn1p
   168)   real(r8) :: flyan
   169)   real(r8) :: tau
   170)   integer :: j,l
   171) 
   172)   data al /1.5e7_r8, 1.5e6_r8, 1.5e6_r8/
   173) 
   174)   data sa /      0._r8,  1.6e-18_r8,       0._r8, &
   175)            10.2e-18_r8, 22.0e-18_r8, 23.1e-18_r8, &
   176)             8.4e-18_r8, 16.0e-18_r8, 11.6e-18_r8/
   177) 
   178)   data si /      0._r8,  1.0e-18_r8,       0._r8, &
   179)            10.2e-18_r8, 22.0e-18_r8, 23.1e-18_r8, &
   180)             8.4e-18_r8, 16.0e-18_r8, 11.6e-18_r8/
   181) 
   182)   data salyao2/8.0e-21_r8/
   183)   data silyano/2.0e-18_r8/
   184)   data bn2p/0.86_r8/
   185)   data bn1p/0.14_r8/
   186) !
   187) ! Calculate Lyman-alpha 121.6 nm geocoronal flux as a function of F10.7:
   188) !
   189)   flyan = 5.E9_r8*(1._r8+0.002_r8*(f107-65._r8))
   190) !
   191) ! Loop over altitudes:
   192) !
   193)   do j=1,jm
   194) !
   195) ! Calculate optical depth and ionization rates for major species:
   196) !
   197)      do l=1,3
   198)         tau=(sa(1,l)*zvcd(1,j)+sa(2,l)*zvcd(2,j)+sa(3,l)*zvcd(3,j))
   199)         qbko1(j) =       al(l)*si(1,l)*zmaj(1,j)*exp(-tau)
   200)         qbko2(j) =       al(l)*si(2,l)*zmaj(2,j)*exp(-tau)
   201)         qbkn2(j) = bn2p*(al(l)*si(3,l)*zmaj(3,j)*exp(-tau))
   202)         qbkn1(j) = bn1p*(al(l)*si(3,l)*zmaj(3,j)*exp(-tau))
   203)      enddo
   204) !
   205) ! Calculate optical depth of Ly-alpha, and ionization rate of NO:
   206) !
   207)      tau = salyao2*zvcd(2,j)
   208)      qbkno(j) = flyan*silyano*zno(j)*exp(-tau)
   209) 
   210)   enddo
   211) 
   212)   return
   213) end subroutine qback
