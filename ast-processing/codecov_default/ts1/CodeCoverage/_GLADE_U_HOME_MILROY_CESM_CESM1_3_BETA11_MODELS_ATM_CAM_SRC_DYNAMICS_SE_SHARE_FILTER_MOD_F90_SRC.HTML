    39)   ! taylor_filter_create:
    40)   !
    41)   ! Create a Taylor filter (requires
    42)   ! communication) and use transfer
    43)   ! function T...
    44)   ! ==================================
    45)   function taylor_filter_create(Tp, mu, gquadp) result(flt)
    46)     use kinds, only : real_kind
    47)     use quadrature_mod, only : quadrature_t, legendre, quad_norm
    48)     real (kind=real_kind), dimension(:) :: Tp          ! transfer fn       
    49)     real (kind=real_kind), intent(in)   :: mu             ! filter viscosity
    50)     type (quadrature_t)  , intent(in)   :: gquadp  ! quadrature points and wts
    51) 
    52)     type (filter_t)                   :: flt
    53) 
    54)     ! Local variables 
    55) 
    56)     real (kind=real_kind), dimension(:), allocatable   :: gammaP ! normalization factor
    57)     real (kind=real_kind), dimension(:,:), allocatable :: legP    ! legendre polynomials
    58) 
    59)     real (kind=real_kind), dimension(:,:), allocatable :: AP     ! spectral-physical 
    60)     real (kind=real_kind), dimension(:,:), allocatable :: AinvP   ! physical-spectral
    61) 
    62) 
    63)     integer                 :: npts
    64)     integer                 :: n,j
    65)     logical, parameter      :: Debug = .FALSE.
    66) 
    67)     call t_startf('taylor_filter_create')
    68)     allocate(gammaP(np))
    69) 
    70)     allocate(legP(np,np))
    71)     allocate(AP(np,np))
    72) 
    73)     allocate(AinvP(np,np))
    74) 
    75)     gammaP=quad_norm(gquadP,np)
    76) 
    77)     do j=1,np
    78)        if(Debug) print *,'taylor_filter_create: gammaP(j) := ',gammaP(j)
    79)        legP(:,j)=legendre(gquadP%points(j),np-1)
    80)     end do
    81) 
    82)     ! -----------------------------------------
    83)     ! compute T(k), the spectral coefficent 
    84)     ! transfer function for the B-V filter
    85)     ! -----------------------------------------
    86) 
    87)     AP = legP
    88) 
    89)     do n=1,np
    90)        do j=1,np
    91)           AinvP(j,n)=legP(n,j)*gquadP%weights(j)/gammaP(n)
    92)        end do
    93)     end do
    94) 
    95)     call filter_matrix_create(AP,Tp,AinvP,flt%FmatP,np,mu)
    96) 
    97)     flt%type = "bv"
    98) 
    99)     deallocate(gammaP)
   100)     deallocate(legP)
   101)     deallocate(AP)
   102)     deallocate(AinvP)
   103)     call t_stopf('taylor_filter_create')
   104) 
   105)   end function taylor_filter_create

   161)   ! ==================================================
   162)   ! This routing builds a Fischer-Mullen transfer 
   163)   ! function that looks like:
   164)   !
   165)   !
   166)   !     ^
   167)   !  T  |
   168)   !     |                 |
   169)   !  1  |__________      _v_
   170)   !     |          -_     
   171)   !     |            \  wght
   172)   !     |             \  ___
   173)   !     |             |   ^
   174)   !  0  |-------------|---|>
   175)   !               ^   ^ 
   176)   !     0         kc  npts  k-->
   177)   !
   178)   !  Where kc := npts-kcut is the point below which T = 1.
   179)   ! ==================================================
   180) 
   181)   function fm_transfer(kcut,wght,npts) result(T)
   182) 
   183)     integer               :: kcut
   184)     real (kind=real_kind) :: wght
   185)     integer               :: npts
   186)     real (kind=real_kind) :: T(npts)
   187) 
   188)     ! Local variables
   189) 
   190)     integer :: k
   191)     integer :: kc
   192)     real (kind=real_kind) :: cut
   193)     real (kind=real_kind) :: amp
   194) 
   195)     call t_startf('fm_transfer')
   196)     kc  = npts-kcut
   197)     cut = kcut
   198) 
   199)     do k=1,kc
   200)        T(k)=one
   201)     end do
   202) 
   203)     do k=kc+1,npts
   204)        amp = wght*(k-kc)*(k-kc)/(cut*cut)    ! quadratic growth
   205)        T(k) = one - amp
   206)     end do
   207)     call t_stopf('fm_transfer')
   208) 
   209)   end function fm_transfer

   211)   ! ===========================================
   212)   ! bv_transfer:
   213)   ! 
   214)   ! compute Boyd-Vandeven transfer 
   215)   ! function for a spectral element 
   216)   ! with npts degrees of freedom...
   217)   ! ===========================================
   218) 
   219)   function bv_transfer(p,s,npts) result(T)
   220) 
   221)     real (kind=real_kind), intent(in) :: p     ! order of sigmoid
   222)     real (kind=real_kind), intent(in) :: s     ! scale of filter
   223)     integer,               intent(in) :: npts  ! number of points
   224)     real (kind=real_kind) :: T(npts)
   225) 
   226)     integer k
   227)     real (kind=real_kind) :: arg
   228)     real (kind=real_kind) :: rat
   229) 
   230)     call t_startf('bv_transfer')
   231)     do k=1,npts
   232)        rat = REAL(k,kind=real_kind)/REAL(npts,kind=real_kind)
   233)        if (rat<s) then
   234)           T(k)=one
   235)        else
   236)           arg = (rat-s)/(one-s)
   237)           !
   238)           !JPE: Need to assure that arg <= 1.0 
   239)           !
   240)           T(k)= bvsigma(p,min(arg,one))
   241)        end if
   242)     end do
   243)     call t_stopf('bv_transfer')
   244) 
   245)   end function bv_transfer

   247)   ! ===========================================
   248)   ! bvsigma:
   249)   !
   250)   ! Boyd - Vandeven sigma function
   251)   ! see p.98 of Taylor, Tribbia and Iskandarani
   252)   ! ===========================================
   253) 
   254)   function bvsigma(p,x) result(sigma)
   255) 
   256) #ifdef CAM
   257)     ! CAM code is required to use this function, to handle differences in
   258)     ! Fortran 2008 support and compiler extensions.
   259)     use shr_spfn_mod, only: erfc => shr_spfn_erfc
   260) #endif
   261) 
   262)     real (kind=real_kind), intent(in) :: p
   263)     real (kind=real_kind), intent(in) :: x
   264)     real (kind=real_kind)             :: sigma
   265) 
   266)     ! Local variables
   267) 
   268)     real (kind=real_kind) :: om
   269)     real (kind=real_kind) :: xfac
   270)     real (kind=real_kind) :: arg
   271) #if 0
   272)     real (kind=real_kind) :: tmp
   273) #endif
   274) 
   275)     ! For some old (pre-F2008) compilers, this is safer than "erfc", which
   276)     ! is sometimes only single precision (not generic).
   277) #ifndef CAM
   278)     real*8  :: derfc
   279) #endif
   280)     call t_startf('bvsigma')
   281) 
   282)     om=ABS(x)-0.5D0
   283) 
   284)     if (x==zero) then
   285)        sigma=one
   286)     else if (x==one) then
   287)        sigma=zero
   288)     else
   289)        xfac=one
   290)        if (om /= zero) xfac = SQRT(-LOG((one-4.0D0*om**2))/(4.0D0*om**2))
   291)        arg = 2.0D0*SQRT(p)*om*xfac
   292) #if 0
   293)        tmp = erfd(arg)
   294)        print *, "x, erf", x, tmp
   295) #endif
   296) #ifdef CAM
   297)        sigma = 0.50D0*erfc(arg)
   298) #else
   299)        sigma = 0.50D0*derfc(arg)
   300) #endif
   301) #if 0
   302)        sigma = 0.50D0*(one - erfd(arg))
   303) #endif
   304)     end if
   305)     call t_stopf('bvsigma')
   306) 
   307)   end function bvsigma

   340)   ! Apply a Filter Matrix
   341)   ! in both directions (x and y) of a 2-d domain.

   344) 
   345)   subroutine filter_P(p,flt) 
   346)     use kinds, only : int_kind
   347)     type (filter_t)                    :: flt
   348)     real(kind=real_kind),intent(inout) :: p(np,np)
   349) 
   350)     ! Local
   351) 
   352)     integer  :: i,j,l
   353)     integer(kind=int_kind), parameter :: unroll = 2
   354)     real(kind=real_kind) :: temptp(np,np) 
   355)     real(kind=real_kind) :: sumx00,sumx01
   356)     real(kind=real_kind) :: sumx10,sumx11
   357) 
   358)     integer(kind=int_kind) :: rem,n2,npr,nprp1
   359)     logical, parameter :: UseUnroll = .TRUE.
   360) 
   361)     !JMD if(MODULO(np,2) == 0 .and. UseUnroll) then 
   362)     if(UseUnroll) then 
   363)        rem = MODULO(np,unroll)
   364)        npr  = np - rem 
   365)        nprp1 = npr+1
   366)        !JMD    print *,'Filter_P: np,npr,nprp1,rem:',np,npr,nprp1,rem
   367) 
   368)        do j=1,npr,2
   369)           do l=1,npr,2
   370)              sumx00=0.0d0
   371)              sumx01=0.0d0
   372)              sumx10=0.0d0
   373)              sumx11=0.0d0
   374) 
   375)              do i=1,np
   376)                 sumx00 = sumx00 + flt%FmatP(i,l  )*p(i,j  )
   377)                 sumx01 = sumx01 + flt%FmatP(i,l+1)*p(i,j  )
   378)                 sumx10 = sumx10 + flt%FmatP(i,l  )*p(i,j+1)
   379)                 sumx11 = sumx11 + flt%FmatP(i,l+1)*p(i,j+1)
   380)              end do
   381) 
   382)              temptP(j  ,l  ) = sumx00
   383)              temptP(j  ,l+1) = sumx01
   384)              temptP(j+1,l  ) = sumx10
   385)              temptP(j+1,l+1) = sumx11
   386) 
   387)           end do
   388)        end do
   389)        if(rem>0) then 
   390)           ! =======================
   391)           ! Calculate the remainder
   392)           ! =======================
   393)           do l=1,npr
   394)              sumx00=0.0d0
   395)              sumx01=0.0d0
   396)              do i=1,np
   397)                 sumx00 = sumx00 + flt%FmatP(i,l  )*p(i,np  )
   398)                 sumx01 = sumx01 + flt%FmatP(i,np  )*p(i,l  )
   399)              enddo
   400)              temptP(np  ,l  ) = sumx00	
   401)              temptP(l  ,np  ) = sumx01	
   402)           enddo
   403)           sumx00=0.0d0
   404)           do i=1,np
   405)              sumx00 = sumx00 + flt%FmatP(i,np  )*p(i,np  )
   406)           enddo
   407)           temptP(np  ,np  ) = sumx00
   408)        endif
   409) 
   410)        do j=1,npr,2
   411)           do i=1,npr,2
   412)              sumx00=0.0d0
   413)              sumx01=0.0d0
   414)              sumx10=0.0d0
   415)              sumx11=0.0d0
   416) 
   417)              do l=1,np
   418)                 sumx00 = sumx00 +  flt%FmatP(l,j  )*temptP(l,i  )
   419)                 sumx01 = sumx01 +  flt%FmatP(l,j+1)*temptP(l,i  )
   420)                 sumx10 = sumx10 +  flt%FmatP(l,j  )*temptP(l,i+1)
   421)                 sumx11 = sumx11 +  flt%FmatP(l,j+1)*temptP(l,i+1)
   422)              end do
   423) 
   424)              p(i  ,j  ) = sumx00
   425)              p(i  ,j+1) = sumx01
   426)              p(i+1,j  ) = sumx10
   427)              p(i+1,j+1) = sumx11
   428) 
   429)           end do
   430)        end do
   431)        if(rem>0) then 
   432)           ! =======================
   433)           ! Calculate the remainder
   434)           ! =======================
   435)           do i=1,npr
   436)              sumx00=0.0d0
   437)              sumx01=0.0d0
   438)              do l=1,np
   439)                 sumx00 = sumx00 +  flt%FmatP(l,i  )*temptP(l,np  )
   440)                 sumx01 = sumx01 +  flt%FmatP(l,np  )*temptP(l,i  )
   441)              enddo
   442)              p(np  ,i  ) = sumx00
   443)              p(i  ,np  ) = sumx01
   444)           enddo
   445)           sumx00=0.0d0
   446)           do l=1,np
   447)              sumx00 = sumx00 +  flt%FmatP(l,np  )*temptP(l,np  )
   448)           enddo
   449)           p(np  ,np  ) = sumx00
   450) 
   451)        endif
   452)     else
   453)        do j=1,np
   454)           do l=1,np
   455)              sumx00=0.0d0
   456)              do i=1,np
   457) 		sumx00 = sumx00 + flt%FmatP(i,l  )*p(i,j  )
   458)              enddo
   459)              temptP(j  ,l  ) = sumx00	
   460)           enddo
   461)        enddo
   462)        do j=1,np
   463) 	  do i=1,np
   464) 	     sumx00=0.0d0
   465) 	     do l=1,np
   466) 		sumx00 = sumx00 +  flt%FmatP(l,j  )*temptP(l,i  )
   467) 	     enddo
   468) 	     p(i  ,j  ) = sumx00
   469)           enddo
   470)        enddo
   471)     endif
   472) 
   473)   end subroutine filter_P

   476)   ! =================================================================================
   477)   !
   478)   ! filter_matrix_create_inv:
   479)   !
   480)   ! This routing builds a 1D filter matrix, F, from a diagonal transfer function T 
   481)   ! and the transform matrix A that maps a field u into spectral coefficients u_c,
   482)   ! and its inverse A^-1, which is provided, via
   483)   !    
   484)   !   F = (1-mu)*I + mu*A.T.A^-1
   485)   !
   486)   !  where mu is the viscosity term, or weight of the filtering. 
   487)   !
   488)   ! ==================================================================================
   489) 
   490)   subroutine filter_matrix_create_inv(A,T,Ainv,F,npts,mu)
   491) 
   492)     integer                           :: npts      
   493)     real (kind=real_kind), intent(in) :: A(npts,npts)
   494)     real (kind=real_kind), intent(in) :: T(npts)
   495)     real (kind=real_kind), intent(in) :: Ainv(npts,npts)
   496)     real (kind=real_kind), intent(in) :: mu
   497) 
   498)     real (kind=real_kind) :: F(npts,npts)
   499) 
   500)     ! Local variables
   501) 
   502)     integer j,k,n
   503)     real (kind=real_kind) :: fsum
   504) 
   505)     do k=1,npts  
   506)        do j=1,npts
   507)           fsum=zero
   508)           do n=1,npts
   509)              fsum = fsum + A(n,j)*T(n)*Ainv(k,n)
   510)           end do
   511)           if (j == k) then
   512)              F(k,j) = (one-mu) + mu*fsum
   513)           else
   514)              F(k,j) = mu*fsum
   515)           end if
   516)        end do
   517)     end do
   518) 
   519)   end subroutine filter_matrix_create_inv

   528)   !    
   529)   !   F = (1-mu)*I + mu*A.T.A^-1
   530)   !
   531)   !  where mu is the viscosity term, or weight of the filtering.

   533)   ! =================================================================================

   535)   subroutine filter_matrix_create_noinv(A,T,F,npts,mu)
   536) 
   537)     integer                           :: npts      
   538)     real (kind=real_kind), intent(in) :: A(npts,npts)
   539)     real (kind=real_kind), intent(in) :: T(npts)
   540)     real (kind=real_kind), intent(in) :: mu
   541) 
   542)     real (kind=real_kind) :: F(npts,npts)
   543) 
   544)     ! Local variables
   545) 
   546)     integer               :: n,j,k
   547)     integer               :: ipiv(npts)
   548)     real (kind=real_kind) :: Ainv(npts,npts)

   557)           fsum=zero

   560)           end do
   561)           if (j == k) then
   562)              F(k,j) = (one-mu) + mu*fsum

   564)              F(k,j) = mu*fsum

   566)        end do

   577)   !
   578)   ! appropriated from code supplied by Paul Fischer and

   583)   !
   584)   ! ========================================================
   585) 
   586)   function gaujordf(a,m,n,ipiv) result(ierr)
   587) 
   588)     integer, intent(in)   :: m,n
   589)     real (kind=real_kind) :: a(m,n)
   590)     integer               :: ipiv(n)
   591)     integer               :: ierr
   592) 
   593)     integer :: indr(m),indc(n)
   594)     integer :: i,j,k
   595)     integer :: ir,jc
   596) 
   597)     real (kind=real_kind) :: amx,piv,tmp,work
   598)     real (kind=real_kind) :: rmult(m)
   599)     real (kind=real_kind) :: eps
   600) 
   601)     ierr = 0
   602)     eps = 1.0D-14
   603)     !
   604)     do k=1,m
   605)        ipiv(k)=0
   606)     end do
   607)     !
   608)     do k=1,m
   609)        amx=zero
   610)        !
   611)        !    Pivot search
   612)        !
   613)        do i=1,m
   614)           if (ipiv(i) /= 1) then
   615)              do j=1,m
   616)                 if (ipiv(j) == 0) then
   617)                    if (abs(a(i,j)) >= amx) then
   618)                       amx = ABS(a(i,j))
   619)                       ir  = i
   620)                       jc  = j
   621)                    end if
   622)                 else if (ipiv(j) > 1) then
   623)                    ierr = -ipiv(j)
   624)                    return
   625)                 end if
   626)              end do
   627)           end if
   628)        end do
   629) 
   630)        ipiv(jc) = ipiv(jc) + 1
   631)        !
   632)        !    Swap rows
   633)        ! 
   634)        if (ir /= jc) then
   635)           do j=1,n
   636)              tmp  = a(ir,j)
   637)              a(ir,j) = a(jc,j)
   638)              a(jc,j) = tmp
   639)           end do
   640)        end if
   641) 
   642)        indr(k)=ir
   643)        indc(k)=jc
   644) 
   645)        !     write(6 ,*) k,' Piv:',jc,a(jc,jc)
   646)        !       write(28,*) k,' Piv:',jc,a(jc,jc)
   647) 
   648)        if (abs(a(jc,jc)) < eps) then
   649)           write(6 ,*) 'Gauss Jordan Pivot too small:',jc,a(jc,jc)
   650)           !       write(28,*) 'small Gauss Jordan Piv:',jc,a(jc,jc)
   651)           ierr = jc
   652)           return
   653)        end if
   654) 
   655)        piv = one/a(jc,jc)
   656)        a(jc,jc)=one
   657) 
   658)        do j=1,n
   659)           a(jc,j) = a(jc,j)*piv
   660)        end do
   661) 
   662)        do j=1,n
   663)           work    = a(jc,j)
   664)           a(jc,j) = a(1 ,j)
   665)           a(1 ,j) = work
   666)        end do
   667) 
   668)        do i=2,m
   669)           rmult(i) = a(i,jc)
   670)           a(i,jc)  = zero
   671)        end do
   672) 
   673)        do j=1,n
   674)           do i=2,m
   675)              a(i,j) = a(i,j) - rmult(i)*a(1,j)
   676)           end do
   677)        end do
   678) 
   679)        do j=1,n
   680)           work    = a(jc,j)
   681)           a(jc,j) = a(1 ,j)
   682)           a(1 ,j) = work
   683)        end do
   684) 
   685)     end do
   686) 
   687)     !
   688)     !     Unscramble matrix
   689)     !
   690) 
   691)     do j=m,1,-1
   692)        if (indr(j) /= indc(j)) then
   693)           do i=1,m
   694)              tmp=a(i,indr(j))
   695)              a(i,indr(j))=a(i,indc(j))
   696)              a(i,indc(j))=tmp
   697)           end do
   698)        end if
   699)     end do
   700)   end function gaujordf

   839)   subroutine prim_filt_fm(ps,T,div,v,flt,D,Dinv)
   840)     use control_mod, only : integration
   841)     use dimensions_mod, only : nlev
   842)     implicit none
   843) 
   844)     real(kind=real_kind), intent(inout) :: ps(np,np)
   845)     real(kind=real_kind), intent(inout) :: T(np,np,nlev)
   846)     real(kind=real_kind), intent(inout) :: div(np,np,nlev)
   847)     real(kind=real_kind), intent(inout) :: v(np,np,2,nlev)
   848) 
   849)     type (filter_t), intent(in)      :: flt
   850)     real(kind=real_kind), intent(in) :: D(2,2,np,np)
   851)     real(kind=real_kind), intent(in) :: Dinv(2,2,np,np)
   852) 
   853)     integer :: i,j,k
   854)     real(kind=real_kind) :: v1,v2
   855)     real(kind=real_kind) :: usph(np,np)
   856)     real(kind=real_kind) :: vsph(np,np)
   857) 
   858)     call filter_P(ps(:,:),flt)
   859)     do k=1,nlev
   860)        call filter_P(T(:,:,k),flt)
   861)        !     call filter_P(div(:,:,k),flt)
   862) 
   863)        ! ======================================
   864)        ! Rotate cube velocities onto the sphere
   865)        ! ======================================
   866)        if( integration == "explicit" ) then
   867)           do j=1,np
   868)              do i=1,np
   869)                 v1 = v(i,j,1,k)
   870)                 v2 = v(i,j,2,k)
   871)                 usph(i,j)= v1*Dinv(1,1,i,j) + v2*Dinv(2,1,i,j)
   872)                 vsph(i,j)= v1*Dinv(1,2,i,j) + v2*Dinv(2,2,i,j)
   873)              end do
   874)           end do
   875)        else
   876)           do j=1,np
   877)              do i=1,np
   878)                 v1 = v(i,j,1,k)
   879)                 v2 = v(i,j,2,k)
   880)                 usph(i,j)= v1*D(1,1,i,j) + v2*D(1,2,i,j)
   881)                 vsph(i,j)= v1*D(2,1,i,j) + v2*D(2,2,i,j)
   882)              end do
   883)           end do
   884)        endif
   885)        ! ======================================
   886)        ! Filter the values on the sphere
   887)        ! ======================================
   888) 
   889)        call filter_P(usph(:,:),flt)
   890)        call filter_P(vsph(:,:),flt)
   891) 
   892)        ! ======================================
   893)        ! Rotate sphere velocities onto the cube
   894)        ! ======================================
   895)        if( integration == "explicit" )  then
   896)           do j=1,np
   897)              do i=1,np
   898)                 v1 = usph(i,j)
   899)                 v2 = vsph(i,j)
   900)                 v(i,j,1,k)= v1*D(1,1,i,j) + v2*D(2,1,i,j)
   901)                 v(i,j,2,k)= v1*D(1,2,i,j) + v2*D(2,2,i,j)
   902)              end do
   903)           end do
   904)        else
   905)           do j=1,np
   906)              do i=1,np
   907)                 v1 = usph(i,j)
   908)                 v2 = vsph(i,j)
   909)                 v(i,j,1,k)= v1*Dinv(1,1,i,j) + v2*Dinv(1,2,i,j)
   910)                 v(i,j,2,k)= v1*Dinv(2,1,i,j) + v2*Dinv(2,2,i,j)
   911)              end do
   912)           end do
   913)        endif
   914)     end do
   915) 
   916)   end subroutine prim_filt_fm
