   681) ! !IROUTINE: checkDates
   682) !
   683) ! !INTERFACE:
   684)   subroutine checkDates( )
   685) !
   686) ! !DESCRIPTION: 
   687) ! Make sure the dates are compatible. The date given to startup the model
   688) ! and the date on the restart file must be the same although years can be
   689) ! different. The dates need to be checked when the restart file is being
   690) ! read in for a startup or branch case (they are NOT allowed to be different
   691) ! for a restart case).
   692) !
   693) ! For the prognostic crop model the date of planting is tracked and growing
   694) ! degree days is tracked (with a 20 year mean) -- so shifting the start dates
   695) ! messes up these bits of saved information.
   696) !
   697) ! !USES:
   698) !
   699) ! !ARGUMENTS:
   700)     use clm_time_manager, only : get_driver_start_ymd, get_start_date
   701)     use clm_varctl      , only : iulog
   702)     use clm_varctl      , only : nsrest, nsrBranch, nsrStartup
   703) !
   704) ! !REVISION HISTORY:
   705) ! Author: Erik Kluzek
   706) !
   707) !EOP
   708) !
   709) ! !LOCAL VARIABLES:
   710)     integer :: stymd       ! Start date YYYYMMDD from driver
   711)     integer :: styr        ! Start year from driver
   712)     integer :: stmon_day   ! Start date MMDD from driver
   713)     integer :: rsmon_day   ! Restart date MMDD from restart file
   714)     integer :: rsyr        ! Restart year from restart file
   715)     integer :: rsmon       ! Restart month from restart file
   716)     integer :: rsday       ! Restart day from restart file
   717)     integer :: tod         ! Restart time of day from restart file
   718)     character(len=*), parameter :: formDate = '(A,i4.4,"/",i2.2,"/",i2.2)' ! log output format
   719)     character(len=32) :: subname = 'CropRest::checkDates'
   720)     !
   721)     ! If branch or startup make sure the startdate is compatible with the date
   722)     ! on the restart file.
   723)     !
   724)     if ( nsrest == nsrBranch .or. nsrest == nsrStartup )then
   725)        stymd       = get_driver_start_ymd()
   726)        styr        = stymd / 10000
   727)        stmon_day   = stymd - styr*10000
   728)        call get_start_date( rsyr, rsmon, rsday, tod )
   729)        rsmon_day = rsmon*100 + rsday
   730)        if ( masterproc ) &
   731)        write(iulog,formDate) 'Date on the restart file is: ', rsyr, rsmon, rsday
   732)        if ( stmon_day /= rsmon_day )then
   733)           write(iulog,formDate) 'Start date is: ', styr, stmon_day/100, &
   734)                                  (stmon_day - stmon_day/100)
   735)           call endrun( trim(subname)// &
   736)           ' ERROR: For prognostic crop to work correctly, the start date (month and day)'// &
   737)           ' and the date on the restart file needs to match (years can be different)' )
   738)        end if
   739)     end if
   740) 
   741)   end subroutine checkDates
