    75)   function rrtmg_state_create( pstate, cam_in ) result( rstate )
    76)     use physics_types,    only: physics_state
    77)     use camsrfexch,       only: cam_in_t
    78)     use physconst,        only: stebol
    79) 
    80)     implicit none
    81) 
    82)     type(physics_state), intent(in) :: pstate
    83)     type(cam_in_t),      intent(in) :: cam_in
    84) 
    85)     type(rrtmg_state_t), pointer  :: rstate
    86) 
    87)     real(r8) dy                   ! Temporary layer pressure thickness
    88)     real(r8) :: tint(pcols,pverp)    ! Model interface temperature
    89)     integer  :: ncol, i, kk, k
    90) 
    91)     allocate( rstate )
    92) 
    93)     allocate( rstate%h2ovmr(pcols,num_rrtmg_levs) )
    94)     allocate( rstate%o3vmr(pcols,num_rrtmg_levs) )
    95)     allocate( rstate%co2vmr(pcols,num_rrtmg_levs) )
    96)     allocate( rstate%ch4vmr(pcols,num_rrtmg_levs) )
    97)     allocate( rstate%o2vmr(pcols,num_rrtmg_levs) )
    98)     allocate( rstate%n2ovmr(pcols,num_rrtmg_levs) )
    99)     allocate( rstate%cfc11vmr(pcols,num_rrtmg_levs) )
   100)     allocate( rstate%cfc12vmr(pcols,num_rrtmg_levs) )
   101)     allocate( rstate%cfc22vmr(pcols,num_rrtmg_levs) )
   102)     allocate( rstate%ccl4vmr(pcols,num_rrtmg_levs) )
   103) 
   104)     allocate( rstate%pmidmb(pcols,num_rrtmg_levs) )
   105)     allocate( rstate%pintmb(pcols,num_rrtmg_levs+1) )
   106)     allocate( rstate%tlay(pcols,num_rrtmg_levs) )
   107)     allocate( rstate%tlev(pcols,num_rrtmg_levs+1) )
   108) 
   109)     ncol = pstate%ncol
   110) 
   111)     ! Calculate interface temperatures (following method
   112)     ! used in radtpl for the longwave), using surface upward flux and
   113)     ! stebol constant in mks units
   114)     do i = 1,ncol
   115)        tint(i,1) = pstate%t(i,1)
   116)        tint(i,pverp) = sqrt(sqrt(cam_in%lwup(i)/stebol))
   117)        do k = 2,pver
   118)           dy = (pstate%lnpint(i,k) - pstate%lnpmid(i,k)) / (pstate%lnpmid(i,k-1) - pstate%lnpmid(i,k))
   119)           tint(i,k) = pstate%t(i,k) - dy * (pstate%t(i,k) - pstate%t(i,k-1))
   120)        end do
   121)     end do
   122) 
   123)     do k = 1, num_rrtmg_levs
   124) 
   125)        kk = max(k + (pverp-num_rrtmg_levs)-1,1)
   126) 
   127)        rstate%pmidmb(:ncol,k) = pstate%pmid(:ncol,kk) * 1.e-2_r8
   128)        rstate%pintmb(:ncol,k) = pstate%pint(:ncol,kk) * 1.e-2_r8
   129) 
   130)        rstate%tlay(:ncol,k) = pstate%t(:ncol,kk)
   131)        rstate%tlev(:ncol,k) = tint(:ncol,kk)
   132) 
   133)     enddo
   134) 
   135)     ! bottom interface
   136)     rstate%pintmb(:ncol,num_rrtmg_levs+1) = pstate%pint(:ncol,pverp) * 1.e-2_r8 ! mbar
   137)     rstate%tlev(:ncol,num_rrtmg_levs+1) = tint(:ncol,pverp)
   138) 
   139)     ! top layer thickness
   140)     if (num_rrtmg_levs==pverp) then
   141)        rstate%pmidmb(:ncol,1) = 0.5_r8 * rstate%pintmb(:ncol,2) 
   142)        rstate%pintmb(:ncol,1) = 1.e-4_r8 ! mbar
   143)     endif
   144) 
   145)   endfunction rrtmg_state_create

   148) ! updates the concentration fields
   149) !--------------------------------------------------------------------------------
   150)   subroutine rrtmg_state_update(pstate,pbuf,icall,rstate)
   151)     use physics_types,    only: physics_state
   152)     use physics_buffer,   only: physics_buffer_desc
   153)     use rad_constituents, only: rad_cnst_get_gas
   154) 
   155)     implicit none
   156) 
   157)     type(physics_state), intent(in), target :: pstate
   158)     type(physics_buffer_desc),  pointer :: pbuf(:)
   159)     integer,             intent(in) :: icall                     ! index through climate/diagnostic radiation calls
   160)     type(rrtmg_state_t), pointer    :: rstate
   161) 
   162)     real(r8), pointer, dimension(:,:) :: sp_hum ! specific humidity
   163)     real(r8), pointer, dimension(:,:) :: n2o    ! nitrous oxide mass mixing ratio
   164)     real(r8), pointer, dimension(:,:) :: ch4    ! methane mass mixing ratio
   165)     real(r8), pointer, dimension(:,:) :: o2     ! O2 mass mixing ratio
   166)     real(r8), pointer, dimension(:,:) :: cfc11  ! cfc11 mass mixing ratio
   167)     real(r8), pointer, dimension(:,:) :: cfc12  ! cfc12 mass mixing ratio
   168)     real(r8), pointer, dimension(:,:) :: o3     ! Ozone mass mixing ratio
   169)     real(r8), pointer, dimension(:,:) :: co2    ! co2   mass mixing ratio
   170)     
   171)     integer  :: ncol, i, kk, k
   172) 
   173)     ncol = pstate%ncol
   174) 
   175)     ! Get specific humidity
   176)     call rad_cnst_get_gas(icall,'H2O', pstate, pbuf, sp_hum)
   177)     ! Get oxygen mass mixing ratio.
   178)     call rad_cnst_get_gas(icall,'O2',  pstate, pbuf, o2)
   179)     ! Get ozone mass mixing ratio.
   180)     call rad_cnst_get_gas(icall,'O3',  pstate, pbuf, o3)
   181)     ! Get CO2 mass mixing ratio
   182)     call rad_cnst_get_gas(icall,'CO2', pstate, pbuf, co2)
   183)     ! Get N2O mass mixing ratio
   184)     call rad_cnst_get_gas(icall,'N2O', pstate, pbuf, n2o)
   185)     ! Get CH4 mass mixing ratio
   186)     call rad_cnst_get_gas(icall,'CH4', pstate, pbuf, ch4)
   187)     ! Get CFC mass mixing ratios
   188)     call rad_cnst_get_gas(icall,'CFC11', pstate, pbuf, cfc11)
   189)     call rad_cnst_get_gas(icall,'CFC12', pstate, pbuf, cfc12)
   190) 
   191)     do k = 1, num_rrtmg_levs
   192) 
   193)        kk = max(k + (pverp-num_rrtmg_levs)-1,1)
   194) 
   195)        rstate%ch4vmr(:ncol,k)   = ch4(:ncol,kk) * amdm
   196)        rstate%h2ovmr(:ncol,k)   = (sp_hum(:ncol,kk) / (1._r8 - sp_hum(:ncol,kk))) * amdw
   197)        rstate%o3vmr(:ncol,k)    = o3(:ncol,kk) * amdo
   198)        rstate%co2vmr(:ncol,k)   = co2(:ncol,kk) * amdc
   199)        rstate%ch4vmr(:ncol,k)   = ch4(:ncol,kk) * amdm
   200)        rstate%o2vmr(:ncol,k)    = o2(:ncol,kk) * amdo2
   201)        rstate%n2ovmr(:ncol,k)   = n2o(:ncol,kk) * amdn
   202)        rstate%cfc11vmr(:ncol,k) = cfc11(:ncol,kk) * amdc1
   203)        rstate%cfc12vmr(:ncol,k) = cfc12(:ncol,kk) * amdc2
   204)        rstate%cfc22vmr(:ncol,k) = 0._r8
   205)        rstate%ccl4vmr(:ncol,k)  = 0._r8
   206) 
   207)     enddo
   208) 
   209)   end subroutine rrtmg_state_update
