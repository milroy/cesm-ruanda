     6) ! !DESCRIPTION:

     8) !  This module contains an I/O unit manager for tracking, assigning
     9) !  and reserving I/O unit numbers.

    11) !  There are three reserved I/O units set as parameters in this
    12) !  module.  The default units for standard input (stdin), standard

    14) !  set as units 5,6,6, respectively as that is the most commonly

    16) !  default units are conflicting with other models or if the
    17) !  vendor is using different values.
    18) !
    19) !  The maximum number of I/O units per node is currently set by
    20) !  the parameter ice\_IOMaxUnits.

   169) ! !IROUTINE: get_fileunit
   170) ! !INTERFACE:
   171) 
   172)  subroutine get_fileunit(iunit)
   173) 
   174) ! !DESCRIPTION:
   175) !  This routine returns the next available I/O unit and marks it as
   176) !  in use to prevent any later use.
   177) !  Note that {\em all} processors must call this routine even if only
   178) !  the master task is doing the I/O.  This is necessary insure that
   179) !  the units remain synchronized for other parallel I/O functions.
   180) !
   181) ! !REVISION HISTORY:
   182) !  same as module
   183) 
   184) ! !OUTPUT PARAMETERS:
   185) 
   186)    integer (kind=int_kind), intent(out) :: &
   187)       iunit                     ! next free I/O unit
   188) !EOP
   189) !BOC
   190) 
   191)    integer (kind=int_kind) :: n  ! dummy loop index
   192) 
   193)    logical (kind=log_kind) :: alreadyInUse
   194) 
   195) #ifdef CCSMCOUPLED
   196)    iunit = shr_file_getUnit()
   197) #else
   198)    srch_units: do n=ice_IOUnitsMinUnits, ice_IOUnitsMaxUnits
   199)       if (.not. ice_IOUnitsInUse(n)) then   ! I found one, I found one
   200) 
   201)          !*** make sure not in use by library or calling routines
   202)          INQUIRE (unit=n,OPENED=alreadyInUse)
   203) 
   204)          if (.not. alreadyInUse) then
   205)             iunit = n        ! return the free unit number
   206)             ice_IOUnitsInUse(n) = .true.  ! mark iunit as being in use
   207)             exit srch_units
   208)          else
   209)             !*** if inquire shows this unit in use, mark it as
   210)             !***    in use to prevent further queries
   211)             ice_IOUnitsInUse(iunit) = .true.
   212)          endif
   213)       endif
   214)    end do srch_units
   215) 
   216)    if (iunit > ice_IOUnitsMaxUnits) stop 'ice_IOUnitsGet: No free units'
   217) #endif
   218) 
   219) !EOC
   220)  end subroutine get_fileunit

   278) #ifdef CCSMCOUPLED
   279)    call shr_file_freeUnit(iunit)
