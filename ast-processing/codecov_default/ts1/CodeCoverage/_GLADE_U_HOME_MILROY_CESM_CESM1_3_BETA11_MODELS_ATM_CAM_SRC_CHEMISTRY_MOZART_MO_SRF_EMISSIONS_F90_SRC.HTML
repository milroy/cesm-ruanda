   233)   ! adds surf flux specified in file to sflx
   234)   subroutine set_srf_emissions( lchnk, ncol, sflx )
   235)     !--------------------------------------------------------
   236)     !	... form the surface fluxes for this latitude slice
   237)     !--------------------------------------------------------
   238) 
   239)     use mo_constants, only : pi
   240)     use time_manager, only : get_curr_calday
   241)     use string_utils, only : to_lower, GLC
   242)     use phys_grid,    only : get_rlat_all_p, get_rlon_all_p
   243) 
   244)     implicit none
   245) 
   246)     !--------------------------------------------------------
   247)     !	... Dummy arguments
   248)     !--------------------------------------------------------
   249)     integer,  intent(in)  :: ncol                  ! columns in chunk
   250)     integer,  intent(in)  :: lchnk                 ! chunk index
   251)     real(r8), intent(out) :: sflx(:,:) ! surface emissions ( kg/m^2/s )
   252) 
   253)     !--------------------------------------------------------
   254)     !	... local variables
   255)     !--------------------------------------------------------
   256)     integer  ::  i, m, n
   257)     real(r8) ::  factor
   258)     real(r8) ::  dayfrac            ! fration of day in light
   259)     real(r8) ::  iso_off            ! time iso flux turns off
   260)     real(r8) ::  iso_on             ! time iso flux turns on
   261) 
   262)     logical  :: polar_day,polar_night
   263)     real(r8) :: doy_loc
   264)     real(r8) :: sunon,sunoff
   265)     real(r8) :: loc_angle
   266)     real(r8) :: latitude
   267)     real(r8) :: declination
   268)     real(r8) :: tod
   269)     real(r8) :: calday
   270) 
   271)     real(r8), parameter :: dayspy = 365._r8
   272)     real(r8), parameter :: twopi = 2.0_r8 * pi
   273)     real(r8), parameter :: pid2  = 0.5_r8 * pi
   274)     real(r8), parameter :: dec_max = 23.45_r8 * pi/180._r8
   275) 
   276)     real(r8) :: flux(ncol)
   277)     real(r8) :: mfactor
   278)     integer  :: isec
   279) 
   280)     character(len=12),parameter :: mks_units(4) = (/ "kg/m2/s     ", &
   281)                                                      "kg/m2/sec   ", &
   282)                                                      "kg/m^2/s    ", &
   283)                                                      "kg/m^2/sec  " /)
   284)     character(len=12) :: units
   285) 
   286)     real(r8), dimension(ncol) :: rlats, rlons 
   287) 
   288)     sflx(:,:) = 0._r8
   289) 
   290)     !--------------------------------------------------------
   291)     !	... set non-zero emissions
   292)     !--------------------------------------------------------
   293)     emis_loop : do m = 1,n_emis_species
   294) 
   295)        n = emissions(m)%spc_ndx
   296) 
   297)        flux(:) = 0._r8
   298)        do isec = 1,emissions(m)%nsectors
   299)           flux(:ncol) = flux(:ncol) + emissions(m)%fields(isec)%data(:ncol,1,lchnk)
   300)        enddo
   301) 
   302)        units = to_lower(trim(emissions(m)%fields(1)%units(:GLC(emissions(m)%fields(1)%units))))
   303)        
   304)        if ( any( mks_units(:) == units ) ) then
   305)           sflx(:ncol,n) = flux(:ncol)
   306)        else
   307)           mfactor = amufac * emissions(m)%mw
   308)           sflx(:ncol,n) = flux(:ncol) * mfactor
   309)        endif
   310) 
   311)     end do emis_loop
   312) 
   313)     call get_rlat_all_p( lchnk, ncol, rlats )
   314)     call get_rlon_all_p( lchnk, ncol, rlons )
   315) 
   316)     calday = get_curr_calday()
   317)     doy_loc     = aint( calday )
   318)     declination = dec_max * cos((doy_loc - 172._r8)*twopi/dayspy)
   319)     tod = (calday - doy_loc) + .5_r8
   320) 
   321)     do i = 1,ncol
   322)        !
   323)        polar_day   = .false.
   324)        polar_night = .false.
   325)        !
   326)        loc_angle = tod * twopi + rlons(i)
   327)        loc_angle = mod( loc_angle,twopi )
   328)        latitude =  rlats(i)
   329)        !
   330)        !------------------------------------------------------------------
   331)        !        determine if in polar day or night
   332)        !        if not in polar day or night then
   333)        !        calculate terminator longitudes
   334)        !------------------------------------------------------------------
   335)        if( abs(latitude) >= (pid2 - abs(declination)) ) then
   336)           if( sign(1._r8,declination) == sign(1._r8,latitude) ) then
   337)              polar_day = .true.
   338)              sunoff = 2._r8*twopi
   339)              sunon  = -twopi
   340)           else
   341)              polar_night = .true.
   342)           end if
   343)        else
   344)           sunoff = acos( -tan(declination)*tan(latitude) )
   345)           sunon  = twopi - sunoff
   346)        end if
   347) 
   348)        !--------------------------------------------------------
   349)        !	... adjust alpha-pinene for diurnal variation
   350)        !--------------------------------------------------------
   351)        if( c10h16_ndx > 0 ) then
   352)           if( has_emis(c10h16_ndx) ) then
   353)              if( .not. polar_night .and. .not. polar_day ) then
   354)                 dayfrac = sunoff / pi
   355)                 sflx(i,c10h16_ndx) = sflx(i,c10h16_ndx) / (.7_r8 + .3_r8*dayfrac)
   356)                 if( loc_angle >= sunoff .and. loc_angle <= sunon ) then
   357)                    sflx(i,c10h16_ndx) = sflx(i,c10h16_ndx) * .7_r8
   358)                 endif
   359)              end if
   360)           end if
   361)        end if
   362) 
   363)        !--------------------------------------------------------
   364)        !	... adjust isoprene for diurnal variation
   365)        !--------------------------------------------------------
   366)        if( isop_ndx > 0 ) then
   367)           if( has_emis(isop_ndx) ) then
   368)              if( .not. polar_night ) then
   369)                 if( polar_day ) then
   370)                    iso_off = .8_r8 * pi
   371)                    iso_on  = 1.2_r8 * pi
   372)                 else
   373)                    iso_off = .8_r8 * sunoff
   374)                    iso_on  = 2._r8 * pi - iso_off
   375)                 end if
   376)                 if( loc_angle >= iso_off .and. loc_angle <= iso_on ) then
   377)                    sflx(i,isop_ndx) = 0._r8
   378)                 else
   379)                    factor = loc_angle - iso_on
   380)                    if( factor <= 0._r8 ) then
   381)                       factor = factor + 2._r8*pi
   382)                    end if
   383)                    factor = factor / (2._r8*iso_off + 1.e-6_r8)
   384)                    sflx(i,isop_ndx) = sflx(i,isop_ndx) * 2._r8 / iso_off * pi * (sin(pi*factor))**2
   385)                 end if
   386)              else
   387)                 sflx(i,isop_ndx) = 0._r8
   388)              end if
   389)           end if
   390)        end if
   391) 
   392)     end do
   393) 
   394)   end subroutine set_srf_emissions
