    53) ! !DESCRIPTION:
    54) ! This subroutine initializes the land surface decomposition into a clump

   121)     procinfo%ncols   = 0
   122)     procinfo%npfts   = 0

   201)     end if
   202) 
   203)     ! Assign gridcells to clumps (and thus pes) ---
   204) 
   205)     allocate(lcid(lns))
   206)     lcid(:) = 0
   207)     ng = 0
   208)     do ln = 1,lns
   209)        if (amask(ln) == 1) then
   210)           ng = ng  + 1
   211) 
   212)           !--- give to clumps in order based on nsegspc
   213)           if (seglen1) then
   214)              cid = mod(ng-1,nclumps) + 1
   215)           else
   216)              rcid = (dble(ng-1)/dble(numg))*dble(nsegspc)*dble(nclumps)
   217)              cid = mod(int(rcid),nclumps) + 1
   218)           endif
   219)           lcid(ln) = cid
   220) 
   221)           !--- give gridcell cell to pe that owns cid ---
   222)           !--- this needs to be done to subsequently use function
   223)           !--- get_proc_bounds(begg,endg) 
   224)           if (iam == clumps(cid)%owner) then
   225)              procinfo%ncells  = procinfo%ncells  + 1
   226)           endif
   227)           if (iam >  clumps(cid)%owner) then
   228)              procinfo%begg = procinfo%begg + 1
   229)           endif
   230)           if (iam >= clumps(cid)%owner) then
   231)              procinfo%endg = procinfo%endg + 1
   232)           endif
   233) 
   234)           !--- give gridcell to cid ---
   235)           !--- increment the beg and end indices ---
   236)           clumps(cid)%ncells  = clumps(cid)%ncells  + 1
   237)           do m = 1,nclumps
   238)              if ((clumps(m)%owner >  clumps(cid)%owner) .or. &
   239)                  (clumps(m)%owner == clumps(cid)%owner .and. m > cid)) then
   240)                 clumps(m)%begg = clumps(m)%begg + 1
   241)              endif
   242)              
   243)              if ((clumps(m)%owner >  clumps(cid)%owner) .or. &
   244)                  (clumps(m)%owner == clumps(cid)%owner .and. m >= cid)) then
   245)                 clumps(m)%endg = clumps(m)%endg + 1
   246)              endif
   247)           enddo
   248) 
   249)        end if
   250)     enddo
   251) 
   252)     ! Set ldecomp
   253) 
   254)     allocate(ldecomp%gdc2glo(numg), ldecomp%glo2gdc(lns), stat=ier)
   255)     if (ier /= 0) then
   256)        write(iulog,*) 'decompInit_lnd(): allocation error1 for ldecomp, etc'
   257)        call endrun()
   258)     end if
   259)     allocate(clumpcnt(nclumps),stat=ier)
   260)     if (ier /= 0) then
   261)        write(iulog,*) 'decompInit_lnd(): allocation error1 for clumpcnt'
   262)        call endrun()
   263)     end if
   264) 
   265)     ldecomp%gdc2glo(:) = 0
   266)     ldecomp%glo2gdc(:) = 0
   267) 
   268)     ! clumpcnt is the start gdc index of each clump
   269) 
   270)     clumpcnt = 0
   271)     ag = 1
   272)     do pid = 0,npes-1
   273)     do cid = 1,nclumps
   274)        if (clumps(cid)%owner == pid) then
   275)          clumpcnt(cid) = ag
   276)          ag = ag + clumps(cid)%ncells
   277)        endif
   278)     enddo
   279)     enddo
   280) 
   281)     ! now go through gridcells one at a time and increment clumpcnt
   282)     ! in order to set gdc2glo and glo2gdc
   283) 
   284)     do aj = 1,lnj
   285)     do ai = 1,lni
   286)        an = (aj-1)*lni + ai
   287)        cid = lcid(an)
   288)        if (cid > 0) then
   289)           ag = clumpcnt(cid)
   290)           ldecomp%gdc2glo(ag) = an
   291)           ldecomp%glo2gdc(an) = ag
   292)           clumpcnt(cid) = clumpcnt(cid) + 1
   293)        end if
   294)     enddo
   295)     enddo
   296) 
   297)     deallocate(clumpcnt)
   298) 
   299)     ! Set gsMap_lnd_gdc2glo
   300) 
   301)     call get_proc_bounds(beg, end)
   302)     allocate(gindex(beg:end))
   303)     do n = beg,end
   304)        gindex(n) = ldecomp%gdc2glo(n)
   305)     enddo
   306)     lsize = end-beg+1
   307)     gsize = lni * lnj
   308)     call mct_gsMap_init(gsMap_lnd_gdc2glo, gindex, mpicom, comp_id, lsize, gsize)
   309)     deallocate(gindex)
   310) 
   311)     ! Diagnostic output
   312) 
   313)     if (masterproc) then
   314)        write(iulog,*)' Surface Grid Characteristics'
   315)        write(iulog,*)'   longitude points               = ',lni
   316)        write(iulog,*)'   latitude points                = ',lnj
   317)        write(iulog,*)'   total number of land gridcells = ',numg
   318)        write(iulog,*)' Decomposition Characteristics'
   319)        write(iulog,*)'   clumps per process             = ',clump_pproc
   320)        write(iulog,*)' gsMap Characteristics'
   321)        write(iulog,*) '  lnd gsmap glo num of segs      = ',mct_gsMap_ngseg(gsMap_lnd_gdc2glo)
   322)        write(iulog,*)
   323)     end if
   324) 
   325)     call shr_sys_flush(iulog)
   326)
