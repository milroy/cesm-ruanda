   105)   function hox_prod_factor( ion_pairs, zmid )
   106) 
   107)     use ppgrid,  only : pcols, pver
   108)     ! for each level interpolate the factor table
   109) 
   110)     implicit none
   111) 
   112)     real(r8),intent(in) :: ion_pairs(pver)
   113)     real(r8),intent(in) :: zmid(pver)
   114)     real(r8) :: hox_prod_factor(pver)
   115) 
   116)     integer :: k
   117)     integer :: lastk
   118) 
   119) 
   120)     lastk = 1
   121) 
   122)     ! start at the bottom since the table goes from bottom to top
   123)     do k = pver,1,-1
   124)        hox_prod_factor(k) = interp_factor_tbl( ion_pairs(k), zmid(k), lastk )
   125)     enddo

   129) !----------------------------------------------------------------------- 
   130) ! bilinear interpolates the above table of factors
   131) !----------------------------------------------------------------------- 
   132)   function interp_factor_tbl( ionp, z, lastk )
   133) 
   134)     implicit none
   135) 
   136)     real(r8),intent(in) :: ionp
   137)     real(r8),intent(in) :: z
   138)     integer, intent(inout) :: lastk
   139)     real(r8) :: interp_factor_tbl
   140) 
   141)     integer :: i,j
   142)     real(r8) :: logp
   143)     real(r8) :: atlwgt1,atlwgt2
   144)     real(r8) :: prodwgt1,prodwgt2
   145)     real(r8) :: fact(nprods)
   146) 
   147)     if (ionp <= 0._r8 ) then 
   148)        interp_factor_tbl = 0._r8
   149)        return
   150)     endif
   151) 
   152)     ! interpolate log10 of the ionization rate since the table is 
   153)     ! on a log scale.
   154)     logp = log10(ionp)
   155) 
   156)     if ( z <= alts(1) ) then 
   157)        fact(:) = factor_tbl(1,:)
   158)     else if ( z >= alts(nalts) ) then
   159)        fact(:) = factor_tbl(nalts,:)
   160)     else
   161)        do i = lastk,nalts
   162)           if ( z > alts(i) .and. z <= alts(i+1) ) then 
   163)              atlwgt1 = (alts(i+1) - z)/(alts(i+1) - alts(i))
   164)              atlwgt2 =   (z - alts(i))/(alts(i+1) - alts(i))
   165)              fact(:) = atlwgt1*factor_tbl(i,:) + atlwgt2*factor_tbl(i+1,:)
   166)              lastk = i
   167)              exit
   168)           endif
   169)        enddo
   170)     endif
   171) 
   172)     if ( logp <= log_ion_prod(1) ) then 
   173)        interp_factor_tbl = fact(1)
   174)     else if ( logp >= log_ion_prod(nprods) ) then
   175)        interp_factor_tbl = fact(nprods)
   176)     else 
   177)        do i = 1,nprods
   178)           if ( logp > log_ion_prod(i) .and. logp <= log_ion_prod(i+1) ) then 
   179)              prodwgt1 = (log_ion_prod(i+1) - logp)/(log_ion_prod(i+1) - log_ion_prod(i))
   180)              prodwgt2 =   (logp - log_ion_prod(i))/(log_ion_prod(i+1) - log_ion_prod(i))
   181)              interp_factor_tbl = prodwgt1*fact(i) + prodwgt2*fact(i+1)
   182)              exit
   183)           endif
   184)        enddo
   185)     endif
   186) 
   187)   end function interp_factor_tbl
