     1) module ref_pres
     2) !--------------------------------------------------------------------------
     3) ! 
     4) ! Provides access to reference pressures for use by the physics
     5) ! parameterizations.  The pressures are provided by the dynamical core
     6) ! since it currently determines the grid used by the physics.
     7) ! 
     8) ! Note that the init method for this module is called before the init
     9) ! method in physpkg; therefore, most physics modules can use these
    10) ! reference pressures during their init phases.
    11) ! 
    12) !--------------------------------------------------------------------------
    13) 
    14) use shr_kind_mod, only: r8=>shr_kind_r8
    15) use ppgrid,       only: pver, pverp
    16) 
    17) implicit none
    18) public
    19) save
    20) 
    21) ! Reference pressures (Pa)
    22) real(r8), protected :: pref_edge(pverp)     ! Layer edges
    23) real(r8), protected :: pref_mid(pver)       ! Layer midpoints
    24) real(r8), protected :: pref_mid_norm(pver)  ! Layer midpoints normalized by
    25)                                             ! surface pressure ('eta' coordinate)
    26) 
    27) real(r8), protected :: ptop_ref             ! Top of model
    28) real(r8), protected :: psurf_ref            ! Surface pressure
    29) 
    30) ! Number of top levels using pure pressure representation
    31) integer, protected :: num_pr_lev
    32) 
    33) ! Pressure used to set troposphere cloud physics top (Pa)
    34) real(r8), protected :: trop_cloud_top_press = 0._r8
    35) ! Top level for troposphere cloud physics
    36) integer, protected :: trop_cloud_top_lev
    37) 
    38) ! Pressure used to set MAM process top (Pa)
    39) real(r8), protected :: clim_modal_aero_top_press = 0._r8
    40) ! Top level for MAM processes that impact climate
    41) integer, protected :: clim_modal_aero_top_lev
    42) 
    43) ! Molecular diffusion is calculated only if the model top is below this
    44) ! pressure (Pa).
    45) real(r8), protected :: do_molec_press = 0.1_r8
    46) ! Pressure used to set bottom of molecular diffusion region (Pa).
    47) real(r8), protected :: molec_diff_bot_press = 50._r8
    48) ! Flag for molecular diffusion, and molecular diffusion level indices.
    49) logical, protected :: do_molec_diff = .false.
    50) integer, protected :: ntop_molec = 1
    51) integer, protected :: nbot_molec = 0
    52)

    57) subroutine ref_pres_readnl(nlfile)
    58) 
    59)    use spmd_utils,      only: masterproc
    60)    use abortutils,      only: endrun
    61)    use namelist_utils,  only: find_group_name
    62)    use units,           only: getunit, freeunit
    63)    use mpishorthand
    64) 
    65)    character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
    66) 
    67)    ! Local variables
    68)    integer :: unitn, ierr
    69)    character(len=*), parameter :: subname = 'ref_pres_readnl'
    70) 
    71)    namelist /ref_pres_nl/ trop_cloud_top_press, clim_modal_aero_top_press,&
    72)         do_molec_press, molec_diff_bot_press
    73)    !-----------------------------------------------------------------------------
    74) 
    75)    if (masterproc) then
    76)       unitn = getunit()
    77)       open( unitn, file=trim(nlfile), status='old' )
    78)       call find_group_name(unitn, 'ref_pres_nl', status=ierr)
    79)       if (ierr == 0) then

    83)          end if
    84)       end if
    85)       close(unitn)
    86)       call freeunit(unitn)
    87) 
    88)       ! Check that top for modal aerosols is not lower than
    89)       ! top for clouds.
    90)       if (clim_modal_aero_top_press > trop_cloud_top_press) &
    91)            call endrun("ERROR: clim_modal_aero_top press must be less &
    92)            &than or equal to trop_cloud_top_press.")
    93)    end if
    94) 
    95) #ifdef SPMD
    96)    ! Broadcast namelist variables
    97)    call mpibcast(trop_cloud_top_press,            1 , mpir8,   0, mpicom)
    98)    call mpibcast(clim_modal_aero_top_press,       1 , mpir8,   0, mpicom)
    99)    call mpibcast(do_molec_press,                  1 , mpir8,   0, mpicom)
   100)    call mpibcast(molec_diff_bot_press,            1 , mpir8,   0, mpicom)
   101) #endif

   107) subroutine ref_pres_init
   108) 
   109)   use dyn_grid,     only: dyn_grid_get_pref
   110) 
   111)   ! Get reference pressures from the dynamical core.
   112)   call dyn_grid_get_pref(pref_edge, pref_mid, num_pr_lev)
   113) 
   114)   ptop_ref = pref_edge(1)
   115)   psurf_ref = pref_edge(pverp)
   116) 
   117)   pref_mid_norm = pref_mid/psurf_ref
   118) 
   119)   ! Find level corresponding to the top of troposphere clouds.
   120)   trop_cloud_top_lev = press_lim_idx(trop_cloud_top_press, &
   121)        top=.true.)
   122) 
   123)   ! Find level corresponding to the top for MAM processes.
   124)   clim_modal_aero_top_lev = press_lim_idx(clim_modal_aero_top_press, &
   125)        top=.true.)
   126) 
   127)   ! Find level corresponding to the molecular diffusion bottom.
   128)   do_molec_diff = (ptop_ref < do_molec_press)
   129)   if (do_molec_diff) then
   130)      nbot_molec = press_lim_idx(molec_diff_bot_press, &
   131)           top=.false.)
   132)   end if
   133) 
   134) end subroutine ref_pres_init

   138) ! Convert pressure limiters to the appropriate level.
   139) pure function press_lim_idx(p, top) result(k_lim)
   140)   ! Pressure
   141)   real(r8), intent(in) :: p
   142)   ! Is this a top or bottom limit?
   143)   logical,  intent(in) :: top
   144)   integer :: k_lim, k
   145) 
   146)   if (top) then
   147)      k_lim = pver+1
   148)      do k = 1, pver
   149)         if (pref_mid(k) > p) then
   150)            k_lim = k
   151)            exit
   152)         end if
   153)      end do
   154)   else
   155)      k_lim = 0
   156)      do k = pver, 1, -1
   157)         if (pref_mid(k) < p) then
   158)            k_lim = k
   159)            exit
   160)         end if
   161)      end do
   162)   end if
   163) 
   164) end function press_lim_idx
