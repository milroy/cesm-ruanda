   293)    subroutine perf_setopts(mastertask, &
   294)                            LogPrint, &
   295)                            timing_disable_in, &
   296)                            perf_timer_in, &
   297)                            timer_depth_limit_in, &
   298)                            timing_detail_limit_in, &
   299)                            timing_barrier_in, &
   300)                            perf_outpe_num_in, &
   301)                            perf_outpe_stride_in, &
   302)                            perf_single_file_in, &
   303)                            perf_global_stats_in, &
   304)                            perf_papi_enable_in )
   305) !----------------------------------------------------------------------- 
   306) ! Purpose: Set runtime options
   307) ! Author: P. Worley 
   308) !-----------------------------------------------------------------------
   309) !---------------------------Input arguments----------------------------
   310) !
   311)    ! master process?
   312)    logical, intent(in) :: mastertask
   313)    ! Print out to log file?
   314)    logical, intent(IN) :: LogPrint        
   315)    ! timers disable/enable option
   316)    logical, intent(in), optional :: timing_disable_in
   317)    ! performance timer option
   318)    integer, intent(in), optional :: perf_timer_in
   319)    ! timer depth limit option
   320)    integer, intent(in), optional :: timer_depth_limit_in
   321)    ! timer detail limit option
   322)    integer, intent(in), optional :: timing_detail_limit_in
   323)    ! timing barrier enable/disable option
   324)    logical, intent(in), optional :: timing_barrier_in
   325)    ! number of processes writing out timing data
   326)    integer, intent(in), optional :: perf_outpe_num_in
   327)    ! separation between process ids for processes that are writing out timing data
   328)    integer, intent(in), optional :: perf_outpe_stride_in
   329)    ! timing single / multple output file option
   330)    logical, intent(in), optional :: perf_single_file_in
   331)    ! collect and output global performance statistics option
   332)    logical, intent(in), optional :: perf_global_stats_in
   333)    ! calling PAPI to read HW performance counters option
   334)    logical, intent(in), optional :: perf_papi_enable_in
   335) !
   336) !---------------------------Local workspace-----------------------------
   337) !
   338)    integer  ierr                  ! error return
   339) !-----------------------------------------------------------------------
   340)    if ( .not. timing_initialized ) then
   341) 
   342)       if ( present(timing_disable_in) ) then
   343)          timing_disable = timing_disable_in
   344)          if (timing_disable) then
   345)             ierr = GPTLdisable()
   346)          else 
   347)             ierr = GPTLenable()
   348)          endif
   349)       endif
   350)       if ( present(perf_timer_in) ) then
   351)          if ((perf_timer_in .eq. GPTLgettimeofday) .or. &
   352)              (perf_timer_in .eq. GPTLnanotime) .or. &
   353)              (perf_timer_in .eq. GPTLread_real_time) .or. &
   354)              (perf_timer_in .eq. GPTLmpiwtime) .or. &
   355)              (perf_timer_in .eq. GPTLclockgettime) .or. &
   356)              (perf_timer_in .eq. GPTLpapitime)) then
   357)             perf_timer = perf_timer_in
   358)          else
   359)             if (mastertask) then
   360)                write(p_logunit,*) 'PERF_SETOPTS: illegal timer requested=',&
   361)                                   perf_timer_in, '. Request ignored.'
   362)             endif
   363)          endif
   364)       endif
   365)       if ( present(timer_depth_limit_in) ) then
   366)          timer_depth_limit = timer_depth_limit_in
   367)       endif
   368)       if ( present(timing_detail_limit_in) ) then
   369)          timing_detail_limit = timing_detail_limit_in
   370)       endif
   371)       if ( present(timing_barrier_in) ) then
   372)          timing_barrier = timing_barrier_in
   373)       endif
   374)       if ( present(perf_outpe_num_in) ) then
   375)          perf_outpe_num = perf_outpe_num_in
   376)       endif
   377)       if ( present(perf_outpe_stride_in) ) then
   378)          perf_outpe_stride = perf_outpe_stride_in
   379)       endif
   380)       if ( present(perf_single_file_in) ) then
   381)          perf_single_file = perf_single_file_in
   382)       endif
   383)       if ( present(perf_global_stats_in) ) then
   384)          perf_global_stats = perf_global_stats_in
   385)       endif
   386)       if ( present(perf_papi_enable_in) ) then
   387) #ifdef HAVE_PAPI
   388)          perf_papi_enable = perf_papi_enable_in
   389) #else
   390)          if (perf_papi_enable_in) then
   391)             if (mastertask) then
   392)                write(p_logunit,*) 'PERF_SETOPTS: PAPI library not linked in. ',&
   393)                                   'Request to enable PAPI ignored.'
   394)             endif
   395)          endif
   396)          perf_papi_enable = .false.
   397) #endif
   398)       endif
   399) !
   400)       if (mastertask .and. LogPrint) then
   401)          write(p_logunit,*) '(t_initf) Using profile_disable=', timing_disable, &             
   402)                             ' profile_timer=', perf_timer
   403)          write(p_logunit,*) '(t_initf)  profile_depth_limit=', timer_depth_limit, &    
   404)                             ' profile_detail_limit=', timing_detail_limit
   405)          write(p_logunit,*) '(t_initf)  profile_barrier=', timing_barrier, &
   406)                             ' profile_outpe_num=', perf_outpe_num
   407)          write(p_logunit,*) '(t_initf)  profile_outpe_stride=', perf_outpe_stride , &
   408)                             ' profile_single_file=', perf_single_file
   409)          write(p_logunit,*) '(t_initf)  profile_global_stats=', perf_global_stats , &
   410)                             ' profile_papi_enable=', perf_papi_enable 
   411)       endif                                                                               
   412) !
   413) #ifdef DEBUG
   414)    else
   415)       write(p_logunit,*) 'PERF_SETOPTS: timing library already initialized. Request ignored.'
   416) #endif
   417)    endif
   418) !
   419)    return
   420)    end subroutine perf_setopts

   842)    subroutine t_prf(filename, mpicom, num_outpe, stride_outpe, &
   843)                     single_file, global_stats, output_thispe)
   844) !----------------------------------------------------------------------- 
   845) ! Purpose: Write out performance timer data
   846) ! Author: P. Worley 
   847) !-----------------------------------------------------------------------
   848) !---------------------------Input arguments-----------------------------
   849) !
   850)    ! performance timer output file name
   851)    character(len=*), intent(in), optional :: filename
   852)    ! mpi communicator id
   853)    integer, intent(in), optional :: mpicom
   854)    ! maximum number of processes writing out timing data
   855)    integer, intent(in), optional :: num_outpe
   856)    ! separation between process ids for processes writing out data 
   857)    integer, intent(in), optional :: stride_outpe
   858)    ! enable/disable the writing of data to a single file
   859)    logical, intent(in), optional :: single_file
   860)    ! enable/disable the collection of global statistics
   861)    logical, intent(in), optional :: global_stats
   862)    ! output timing data for this process
   863)    logical, intent(in), optional :: output_thispe
   864) !
   865) !---------------------------Local workspace-----------------------------
   866) !
   867)    logical  one_file              ! flag indicting whether to write
   868)                                   !  all data to a single file
   869)    logical  glb_stats             ! flag indicting whether to compute
   870)                                   !  global statistics
   871)    logical  pr_write              ! flag indicating whether the current 
   872)                                   !  GPTL output mode is write
   873)    logical  write_data            ! flag indicating whether this process
   874)                                   !  should output its timing data
   875)    integer  i                     ! loop index
   876)    integer  mpicom2               ! local copy of MPI communicator
   877)    integer  me                    ! communicator local process id
   878)    integer  npes                  ! local communicator group size
   879)    integer  gme                   ! global process id
   880)    integer  ierr                  ! MPI error return
   881)    integer  outpe_num             ! max number of processes writing out
   882)                                   !  timing data (excluding output_thispe)
   883)    integer  outpe_stride          ! separation between process ids for
   884)                                   !  processes writing out timing data
   885)    integer  max_outpe             ! max process id for processes
   886)                                   !  writing out timing data
   887)    integer  signal                ! send/recv variable for single
   888)                                   ! output file logic
   889)    integer  str_length            ! string length
   890)    integer  unitn                 ! file unit number
   891)    integer cme_adj                ! length of filename suffix
   892)    integer status (MPI_STATUS_SIZE)    ! Status of message
   893)    character(len=7) cme                ! string representation of process id
   894)    character(len=SHR_KIND_CX+14) fname ! timing output filename
   895) !-----------------------------------------------------------------------
   896) !
   897)    if (.not. timing_initialized) return
   898) 
   899)    call t_startf("t_prf")
   900) !$OMP MASTER
   901)    call mpi_comm_rank(MPI_COMM_WORLD, gme, ierr)
   902)    if ( present(mpicom) ) then
   903)       mpicom2 = mpicom
   904)       call mpi_comm_size(mpicom2, npes, ierr)
   905)          if (ierr .eq. MPI_ERR_COMM) then
   906)             call shr_sys_abort('T_PRF: bad mpi communicator')
   907)          endif
   908)       call mpi_comm_rank(mpicom2, me, ierr)
   909)    else
   910)       call mpi_comm_size(MPI_COMM_WORLD, npes, ierr)
   911)       mpicom2 = MPI_COMM_WORLD
   912)       me = gme
   913)    endif
   914) 
   915)    do i=1,SHR_KIND_CX+14
   916)      fname(i:i) = " "
   917)    enddo
   918) 
   919)    unitn = shr_file_getUnit()
   920) 
   921)    ! determine what the current output mode is (append or write)
   922)    if (GPTLpr_query_write() == 1) then
   923)      pr_write = .true.
   924)      ierr = GPTLpr_set_append()
   925)    else 
   926)      pr_write=.false.
   927)    endif
   928) 
   929)    ! Determine whether to write all data to a single fie
   930)    if (present(single_file)) then
   931)       one_file = single_file
   932)    else
   933)       one_file = perf_single_file
   934)    endif
   935) 
   936)    ! Determine whether to compute global statistics
   937)    if (present(global_stats)) then
   938)       glb_stats = global_stats
   939)    else
   940)       glb_stats = perf_global_stats
   941)    endif
   942) 
   943)    ! Determine which processes are writing out timing data
   944)    write_data = .false.
   945) 
   946)    if (present(num_outpe)) then
   947)       if (num_outpe < 0) then
   948)          outpe_num = npes
   949)       else
   950)          outpe_num = num_outpe
   951)       endif
   952)    else
   953)       if (perf_outpe_num < 0) then
   954)          outpe_num = npes
   955)       else
   956)          outpe_num = perf_outpe_num
   957)       endif
   958)    endif
   959) 
   960)    if (present(stride_outpe)) then
   961)       if (stride_outpe < 1) then
   962)          outpe_stride = 1
   963)       else
   964)          outpe_stride = stride_outpe
   965)       endif
   966)    else
   967)       if (perf_outpe_stride < 1) then
   968)          outpe_stride = 1
   969)       else
   970)          outpe_stride = perf_outpe_stride
   971)       endif
   972)    endif
   973) 
   974)    max_outpe = min(outpe_num*outpe_stride, npes) - 1
   975) 
   976)    if ((mod(me, outpe_stride) .eq. 0) .and. (me .le. max_outpe)) &
   977)       write_data = .true.
   978) 
   979)    if (present(output_thispe)) then
   980)       write_data = output_thispe
   981)    endif
   982) 
   983)    ! If a single timing output file, take turns writing to it.
   984)    if (one_file) then
   985) 
   986)       if ( present(filename) ) then
   987)          str_length = min(SHR_KIND_CX,len_trim(filename))
   988)          fname(1:str_length) = filename(1:str_length)
   989)       else
   990)          fname(1:10) = "timing_all"
   991)       endif
   992) 
   993)       signal = 0
   994)       if (me .eq. 0) then
   995) 
   996)          if (glb_stats) then
   997)             open( unitn, file=trim(fname), status='UNKNOWN' )
   998)             write( unitn, 100) npes
   999)  100        format(/,"***** GLOBAL STATISTICS (",I6," MPI TASKS) *****",/)
  1000)             close( unitn )
  1001) 
  1002)             ierr = GPTLpr_summary_file(mpicom2, trim(fname))
  1003)          endif
  1004) 
  1005)          if (write_data) then
  1006)             if (glb_stats) then
  1007)                open( unitn, file=trim(fname), status='OLD', position='APPEND' )
  1008)             else
  1009)                open( unitn, file=trim(fname), status='UNKNOWN' )
  1010)             endif
  1011) 
  1012)             write( unitn, 101) me, gme
  1013)  101        format(/,"************ PROCESS ",I6," (",I6,") ************",/)
  1014)             close( unitn )
  1015) 
  1016)             ierr = GPTLpr_file(trim(fname))
  1017)          endif
  1018) 
  1019)       else
  1020) 
  1021)          if (glb_stats) then
  1022)             ierr = GPTLpr_summary_file(mpicom2, trim(fname))
  1023)          endif
  1024) 
  1025)          call mpi_recv (signal, 1, mpi_integer, me-1, me-1, mpicom2, status, ierr)
  1026)          if (ierr /= mpi_success) then
  1027)             write(p_logunit,*) 'T_PRF: mpi_recv failed ierr=',ierr
  1028)             call shr_sys_abort()
  1029)          end if
  1030) 
  1031)          if (write_data) then
  1032)             open( unitn, file=trim(fname), status='OLD', position='APPEND' )
  1033)             write( unitn, 101) me, gme
  1034)             close( unitn )
  1035) 
  1036)             ierr = GPTLpr_file(trim(fname))
  1037)          endif
  1038) 
  1039)       endif
  1040) 
  1041)       if (me+1 < npes) &
  1042)          call mpi_send (signal, 1, mpi_integer, me+1, me, mpicom2, ierr)
  1043) 
  1044)    else
  1045) 
  1046)       if (glb_stats) then
  1047)          if ( present(filename) ) then
  1048)             str_length = min(SHR_KIND_CX-6,len_trim(filename))
  1049)             fname(1:str_length) = filename(1:str_length)
  1050)          else
  1051)             str_length = 6
  1052)             fname(1:10) = "timing"
  1053)          endif
  1054)          fname(str_length+1:str_length+6) = '_stats'
  1055) 
  1056)          if (me .eq. 0) then
  1057)             open( unitn, file=trim(fname), status='UNKNOWN' )
  1058)             write( unitn, 100) npes
  1059)             close( unitn )
  1060)          endif
  1061) 
  1062)          ierr = GPTLpr_summary_file(mpicom2, trim(fname))
  1063)          fname(str_length+1:str_length+6) = '      '
  1064)       endif
  1065) 
  1066)       if (write_data) then
  1067)          if (npes .le. 10) then
  1068)             write(cme,'(i1.1)') me
  1069)             cme_adj = 2
  1070)          elseif (npes .le. 100) then
  1071)             write(cme,'(i2.2)') me
  1072)             cme_adj = 3
  1073)          elseif (npes .le. 1000) then
  1074)             write(cme,'(i3.3)') me
  1075)             cme_adj = 4
  1076)          elseif (npes .le. 10000) then
  1077)             write(cme,'(i4.4)') me
  1078)             cme_adj = 5
  1079)          elseif (npes .le. 100000) then
  1080)             write(cme,'(i5.5)') me
  1081)             cme_adj = 6
  1082)          else
  1083)             write(cme,'(i6.6)') me
  1084)             cme_adj = 7
  1085)          endif
  1086) 
  1087)          if ( present(filename) ) then
  1088)             str_length = min(SHR_KIND_CX-cme_adj,len_trim(filename))
  1089)             fname(1:str_length) = filename(1:str_length)
  1090)          else
  1091)             str_length = 6
  1092)             fname(1:10) = "timing"
  1093)          endif
  1094)          fname(str_length+1:str_length+1) = '.'
  1095)          fname(str_length+2:str_length+cme_adj) = cme
  1096) 
  1097)          open( unitn, file=trim(fname), status='UNKNOWN' )
  1098)          write( unitn, 101) me, gme
  1099)          close( unitn )
  1100) 
  1101)          ierr = GPTLpr_file(trim(fname))
  1102)       endif
  1103) 
  1104)    endif
  1105) 
  1106)    call shr_file_freeUnit( unitn )
  1107) 
  1108)    ! reset GPTL output mode
  1109)    if (pr_write) then
  1110)      ierr = GPTLpr_set_write()
  1111)    endif
  1112) 
  1113) !$OMP END MASTER
  1114)    call t_stopf("t_prf")
  1115) 
  1116)    return
  1117)    end subroutine t_prf

  1121)    subroutine t_initf(NLFilename, LogPrint, LogUnit, mpicom, MasterTask)
  1122) !----------------------------------------------------------------------- 
  1123) ! Purpose:  Set default values of runtime timing options 
  1124) !           before namelists prof_inparm and papi_inparm are read,
  1125) !           read namelists (and broadcast, if SPMD),
  1126) !           then initialize timing library.
  1127) ! Author:   P. Worley (based on shr_inputinfo_mod and runtime_opts)
  1128) !-----------------------------------------------------------------------
  1129) !---------------------------Input arguments-----------------------------
  1130) !
  1131)    character(len=*),   intent(IN) :: NLFilename      ! Name-list filename
  1132)    logical, optional,  intent(IN) :: LogPrint        ! If print out to log file
  1133)    integer, optional,  intent(IN) :: LogUnit         ! Unit number for log output
  1134)    integer, optional,  intent(IN) :: mpicom          ! MPI communicator
  1135)    logical, optional,  intent(IN) :: MasterTask      ! If MPI master task
  1136) !
  1137) !---------------------------Local workspace-----------------------------
  1138) !
  1139)    character(len=*), parameter    :: subname = '(T_INITF) '
  1140)    logical                        :: MasterTask2     ! If MPI master task
  1141)    logical                        :: LogPrint2       ! If print to log
  1142) 
  1143)    integer  me                    ! communicator local process id
  1144)    integer  ierr                  ! error return
  1145)    integer  unitn                 ! file unit number
  1146)    integer  papi_ctr1_id          ! PAPI counter id
  1147)    integer  papi_ctr2_id          ! PAPI counter id
  1148)    integer  papi_ctr3_id          ! PAPI counter id
  1149)    integer  papi_ctr4_id          ! PAPI counter id
  1150) !
  1151) !---------------------------Namelists ----------------------------------
  1152) !
  1153)    logical profile_disable
  1154)    logical profile_barrier
  1155)    logical profile_single_file
  1156)    logical profile_global_stats
  1157)    integer profile_depth_limit
  1158)    integer profile_detail_limit
  1159)    integer profile_outpe_num
  1160)    integer profile_outpe_stride
  1161)    integer profile_timer
  1162)    logical profile_papi_enable
  1163)    namelist /prof_inparm/ profile_disable, profile_barrier, &
  1164)                           profile_single_file, profile_global_stats, &
  1165)                           profile_depth_limit, &
  1166)                           profile_detail_limit, profile_outpe_num, &
  1167)                           profile_outpe_stride, profile_timer, &
  1168)                           profile_papi_enable
  1169) 
  1170)    character(len=16) papi_ctr1_str
  1171)    character(len=16) papi_ctr2_str
  1172)    character(len=16) papi_ctr3_str
  1173)    character(len=16) papi_ctr4_str
  1174)    namelist /papi_inparm/ papi_ctr1_str, papi_ctr2_str,  &
  1175)                           papi_ctr3_str, papi_ctr4_str
  1176) !-----------------------------------------------------------------------
  1177)     if ( timing_initialized ) then
  1178) #ifdef DEBUG
  1179)        write(p_logunit,*) 'T_INITF: timing library already initialized. Request ignored.'
  1180) #endif
  1181)        return
  1182)     endif
  1183) 
  1184) !$OMP MASTER
  1185)     if ( present(LogUnit) ) then
  1186)        call t_setLogUnit(LogUnit)
  1187)     else
  1188)        call t_setLogUnit(def_p_logunit)
  1189)     endif
  1190) 
  1191)     if ( present(MasterTask) .and. present(mpicom) )then
  1192)        call mpi_comm_rank(mpicom, me, ierr)
  1193)        if (ierr .eq. MPI_ERR_COMM) then
  1194)           call shr_sys_abort('T_INITF: bad mpi communicator')
  1195)        endif
  1196)        if (me .eq. 0) then
  1197)           MasterTask2 = .true.
  1198)        else
  1199)           MasterTask2 = .false.
  1200)        endif
  1201)     else
  1202)        MasterTask2 = .true.
  1203)     end if
