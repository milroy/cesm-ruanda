     4)   use pio

    12)   include 'mpif.h' !_EXTERNAL
    13) #endif
    14)   type(iosystem_desc_t) :: iosystem
    15)   integer, intent(in) :: msg
    16) 
    17)   type(file_desc_t), pointer :: file
    18)   integer fh, ierr
    19) 
    20) 
    21)   integer :: len, id, dimids(PIO_MAX_VAR_DIMS), type, info(4)
    22)   character(len=pio_max_name) :: name
    23)   type(var_desc_t) :: vardesc
    24) 
    25) 
    26)   call mpi_bcast(fh, 1, mpi_integer, iosystem%compmaster, iosystem%intercomm, ierr)
    27)   file=> lookupfile(fh)
    28)   
    29) 
    30)   select case(msg)
    31)   case (PIO_MSG_CLOSE_FILE)
    32)      call delete_from_file_list(fh)

    38)      ierr = pio_def_var(file, name, type, dimids, vardesc)
    39)   case (PIO_MSG_ENDDEF)

    42)      ierr = pio_redef(file)

    44)      ierr = pio_inq_varndims(file, id, len)
    45)   case (PIO_MSG_INQ_VARNATTS)
    46)      ierr = pio_inq_varnatts(file, id, len)
    47)   case (PIO_MSG_INQ_VARDIMID)
    48)      ierr = pio_inq_vardimid(file, id, dimids)
    49)   case (PIO_MSG_INQ_VARID)
    50)      ierr = pio_inq_varid(File, name, id)
    51)   case (PIO_MSG_INQ_VARNAME)
    52)      ierr = pio_inq_varname(file, id, name)
    53)   case (PIO_MSG_INQ_VARTYPE)
    54)      ierr = pio_inq_vartype(file, id, type)
    55)   case (PIO_MSG_INQ_DIMID)
    56)      ierr = pio_inq_dimid(file, name, id)
    57)   case (PIO_MSG_INQ_DIMLEN)
    58)      ierr = pio_inq_dimlen(file, id, len)
    59)   case (PIO_MSG_INQ_DIMNAME)
    60)      ierr = pio_inq_dimname(file, id, name)
    61)   case(PIO_MSG_INQUIRE)
    62)      ierr = pio_inquire(file, info(1), info(2), info(3), info(4))
    63)   case(PIO_MSG_INQ_ATT)
    64)      ierr = pio_inq_att(file, id, name, type, len)
    65)   case(PIO_MSG_INQ_ATTNAME)
    66)      ierr = pio_inq_attname(file, id, type, name)
    67)   case(PIO_MSG_INQ_ATTLEN)
    68)      ierr = pio_inq_attlen(file, id, name, len)
    69)   case(PIO_MSG_SYNC_FILE)
    70)      call pio_syncfile(file)
    71)   case default
    72)      print *, 'PIO Got unrecognized message ', msg, ierr
    73)      call piodie(__PIO_FILE__,__LINE__)  
    74)   end select
    75) 
    76) end subroutine pio_callback_handler

   422) subroutine att_1d_handler(ios, msg)
   423)   
   424)   use pio, only : iosystem_desc_t, file_desc_t, pio_get_att, pio_max_name, pio_put_att
   425)   use pio_kinds, only : i4, r4, r8
   426)   use pio_msg_mod, only : lookupfile, pio_msg_getatt_1d, pio_msg_putatt_1d
   427)   use pio_support, only : debugAsync, piodie
   428) #ifndef NO_MPIMOD
   429)   use mpi !_EXTERNAL
   430) #endif
   431)   implicit none
   432) #ifdef NO_MPIMOD
   433)   include 'mpif.h' !_EXTERNAL
   434) #endif
   435) 
   436)   type(iosystem_desc_t), intent(inout) :: ios
   437)   integer, intent(in) :: msg
   438)   type(file_desc_t), pointer :: file
   439)   integer :: fh, varid, ierr, itype, strlen, nlen, clen
   440)   character(len=PIO_MAX_NAME) :: name
   441) 
   442)   real(r4), allocatable :: rvar(:)
   443)   real(r8), allocatable :: dvar(:)
   444)   integer(i4), allocatable :: ivar(:)
   445)   
   446)   call mpi_bcast(fh, 1, mpi_integer, ios%compmaster, ios%intercomm, ierr)
   447)   call mpi_bcast(varid, 1, mpi_integer, ios%compmaster, ios%intercomm, ierr)

   449)   call mpi_bcast(nlen, 1, mpi_integer, ios%compmaster, ios%intercomm, ierr)
   450)   call mpi_bcast(name(1:nlen), nlen, mpi_integer, ios%compmaster, ios%intercomm, ierr)
   451)   call MPI_BCAST(clen,1,MPI_INTEGER,ios%CompMaster, ios%intercomm , ierr)
   452) 
   453)   file=> lookupfile(fh)
   454)   
   455)   select case(itype)
   456)   case (TYPEREAL)
   457)      allocate(rvar(clen))
   458)      if(msg==pio_msg_getatt_1d) then
   459)         ierr = pio_get_att(file, varid, name(1:nlen), rvar)
   460)      else
   461)         ierr = pio_put_att(file, varid, name(1:nlen), rvar)
   462)      end if
   463)      deallocate(rvar)
   464)   case (TYPEDOUBLE)
   465)      allocate(dvar(clen))
   466)      if(msg==pio_msg_getatt_1d) then
   467)         ierr = pio_get_att(file, varid, name(1:nlen), dvar)
   468)      else
   469)         ierr = pio_put_att(file, varid, name(1:nlen), dvar)
   470)      end if
   471)      deallocate(dvar)
   472)   case (TYPEINT)
   473)      allocate(ivar(clen))
   474)      if(msg==pio_msg_getatt_1d) then
   475)         ierr = pio_get_att(file, varid, name(1:nlen), ivar)
   476)      else
   477)         ierr = pio_put_att(file, varid, name(1:nlen), ivar)
   478)      end if
   479)      deallocate(ivar)
