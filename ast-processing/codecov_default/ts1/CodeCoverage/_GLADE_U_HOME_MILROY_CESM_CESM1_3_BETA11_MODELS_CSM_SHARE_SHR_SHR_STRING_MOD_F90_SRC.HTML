   622) ! !IROUTINE: shr_string_parseCFtunit -- Parse CF time unit
   623) !
   624) ! !DESCRIPTION:

   629) !     \newline
   630) !  Input string is like "days since 0001-06-15 15:20:45.5 -6:00"
   631) !    - recognizes "days", "hours", "minutes", "seconds"
   632) !    - must have at least yyyy-mm-dd, hh:mm:ss.s is optional
   633) !    - expects a "since" in the string
   634) !    - ignores time zone part
   635) !
   636) ! !REVISION HISTORY:
   637) !     2005-May-15 - T. Craig - first version
   638) !
   639) ! !INTERFACE: ------------------------------------------------------------------

   641) subroutine shr_string_parseCFtunit(string,unit,bdate,bsec,rc)
   642) 
   643)    implicit none
   644) 
   645) ! !INPUT/OUTPUT PARAMETERS:
   646) 
   647)    character(*)        ,intent(in)           :: string ! string to search
   648)    character(*)        ,intent(out)          :: unit   ! delta time unit
   649)    integer(SHR_KIND_IN),intent(out)          :: bdate  ! base date yyyymmdd
   650)    real(SHR_KIND_R8)   ,intent(out)          :: bsec   ! base seconds
   651)    integer(SHR_KIND_IN),intent(out),optional :: rc     ! return code
   652) 
   653) !EOP
   654) 
   655)    !--- local ---
   656)    integer(SHR_KIND_IN)   :: i,i1,i2          ! generic index
   657)    character(SHR_KIND_CL) :: tbase            ! baseline time
   658)    character(SHR_KIND_CL) :: lstr             ! local string
   659)    integer(SHR_KIND_IN)   :: yr,mo,da,hr,min  ! time stuff
   660)    real(SHR_KIND_R8)      :: sec              ! time stuff
   661)    integer(SHR_KIND_IN) :: t01 = 0 ! timer
   662) 
   663)    !----- formats -----
   664)    character(*),parameter :: subName =   "(shr_string_parseCFtunit) "
   665)    character(*),parameter :: F00     = "('(shr_string_parseCFtunit) ',4a)"
   666) 
   667) !-------------------------------------------------------------------------------
   668) ! Notes:
   669) ! o assume length of CF-1.0 time attribute char string  < SHR_KIND_CL
   670) !   This is a reasonable assumption.
   671) !-------------------------------------------------------------------------------
   672) 
   673)    if (debug>1 .and. t01<1) call shr_timer_get(t01,subName)
   674)    if (debug>1) call shr_timer_start(t01)
   675) 
   676)    unit = 'none'
   677)    bdate = 0
   678)    bsec = 0.0_SHR_KIND_R8
   679) 
   680)    i = shr_string_lastIndex(string,'days ')
   681)    if (i > 0) unit = 'days'
   682)    i = shr_string_lastIndex(string,'hours ')
   683)    if (i > 0) unit = 'hours'
   684)    i = shr_string_lastIndex(string,'minutes ')
   685)    if (i > 0) unit = 'minutes'
   686)    i = shr_string_lastIndex(string,'seconds ')
   687)    if (i > 0) unit = 'seconds'
   688) 
   689)    if (trim(unit) == 'none') then
   690)      write(s_logunit,F00) ' ERROR time unit unknown'
   691)      call shr_string_abort(subName//' time unit unknown')
   692)    endif
   693) 
   694)    i = shr_string_lastIndex(string,' since ')
   695)    if (i < 1) then
   696)      write(s_logunit,F00) ' ERROR since does not appear in unit attribute for time '
   697)      call shr_string_abort(subName//' no since in attr name')
   698)    endif
   699)    tbase = trim(string(i+6:))
   700)    call shr_string_leftAlign(tbase)
   701) 
   702)    if (debug > 0 .and. s_logunit > 0) then
   703)      write(s_logunit,*) trim(subName)//' '//'unit '//trim(unit)
   704)      write(s_logunit,*) trim(subName)//' '//'tbase '//trim(tbase)
   705)    endif
   706) 
   707)    yr=0; mo=0; da=0; hr=0; min=0; sec=0
   708)    i1 = 1
   709) 
   710)    i2 = index(tbase,'-') - 1
   711)    if(i2<0) goto 200
   712)    lstr = tbase(i1:i2)
   713) 
   714)    read(lstr,*,ERR=200,END=200) yr
   715)    tbase = tbase(i2+2:)
   716)    call shr_string_leftAlign(tbase)
   717) 
   718)    i2 = index(tbase,'-') - 1
   719)    if(i2<0) goto 200
   720)    lstr = tbase(i1:i2)
   721)    read(lstr,*,ERR=200,END=200) mo
   722)    tbase = tbase(i2+2:)
   723)    call shr_string_leftAlign(tbase)
   724) 
   725)    i2 = index(tbase,' ') - 1
   726)    if(i2<0) i2= len_trim(tbase)
   727)    lstr = tbase(i1:i2)
   728)    read(lstr,*,ERR=200,END=200) da
   729)    tbase = tbase(i2+2:)
   730)    call shr_string_leftAlign(tbase)
   731) 
   732)    i2 = index(tbase,':') - 1
   733)    if(i2<0) i2=len_trim(tbase)
   734)    lstr = tbase(i1:i2)
   735)    read(lstr,*,ERR=200,END=100) hr
   736)    tbase = tbase(i2+2:)
   737)    call shr_string_leftAlign(tbase)
   738) 
   739)    i2 = index(tbase,':') - 1
   740)    if(i2<0) i2=len_trim(tbase)
   741)    lstr = tbase(i1:i2)
   742)    read(lstr,*,ERR=200,END=100) min
   743)    tbase = tbase(i2+2:)
   744)    call shr_string_leftAlign(tbase)
   745) 
   746)    i2 = index(tbase,' ') - 1
   747)    if(i2<0) i2=len_trim(tbase)
   748)    lstr = tbase(i1:i2)
   749)    read(lstr,*,ERR=200,END=100) sec
   750) 
   751) 100  continue
   752)    if (debug > 0 .and. s_loglev > 0) write(s_logunit,*) trim(subName),'ymdhms:',yr,mo,da,hr,min,sec
   753) 
   754)    bdate = abs(yr)*10000 + mo*100 + da
   755)    if (yr < 0) bdate = -bdate
   756)    bsec = real(hr*3600 + min*60,SHR_KIND_R8) + sec
   757) 
   758)    if (present(rc)) rc = 0
   759) 
   760)    if (debug>1) call shr_timer_stop (t01)
   761)    return
   762) 
   763) 200  continue
   764)    write(s_logunit,F00) 'ERROR 200 on char num read '
   765)    call shr_string_abort(subName//' ERROR on char num read')
   766)    if (debug>1) call shr_timer_stop (t01)
   767)    return
   768) 
   769) end subroutine shr_string_parseCFtunit

  1378)

  1392)    integer(SHR_KIND_IN)   :: nFields          ! number of fields in a string
  1393)    integer(SHR_KIND_IN)   :: i0,i1            ! fldStr == string(i0,i1) ??
  1394)    integer(SHR_KIND_IN)   :: j0,j1            ! fldStr == string(j0,j1) ??
  1395)    logical                :: found            ! T => field found in fieldNames
  1396)    logical                :: lprint           ! local print flag
  1397)    integer(SHR_KIND_IN)   :: t01 = 0          ! timer

  1399)    !----- formats -----
  1400)    character(*),parameter :: subName =   "(shr_string_listGetIndex) "
  1401)    character(*),parameter :: F00     = "('(shr_string_listGetIndex) ',4a)"
  1402) 
  1403) !-------------------------------------------------------------------------------
  1404) ! Notes:
  1405) ! - searching from both ends of the list at the same time seems to be 20% faster
  1406) !   but I'm not sure why (B. Kauffman, Feb 2007)
  1407) ! - I commented out sanity check to a little gain speed (B. Kauffman, Mar 2007)
  1408) !-------------------------------------------------------------------------------
  1409) 
  1410)    if (debug>1 .and. t01<1) call shr_timer_get(t01,subName)

  1415)    if (present(print)) lprint = print
  1416) 
  1417)    !--- confirm proper size of input data ---
  1418)    if (len_trim(fldStr) < 1) then
  1419)       if (lprint) write(s_logunit,F00) "ERROR: input field name has 0 length"
  1420)       call shr_string_abort(subName//"invalid field name")
  1421)    end if
  1422) 
  1423)    !--- search for field name in string's list of fields ---
  1424)    found   = .false.
  1425)    kFld    = 0
  1426)    i0      =  1  ! ?? fldStr == string(i0:i1) ??
  1427)    i1      = -1

  1437)       if (n > 0) then   
  1438)          i1 = i0 + n - 2       ! *not*  the  last field name in fieldNames
  1439)       else               
  1440)          i1 = len_trim(string) ! this is the last field name in fieldNames
  1441)       endif

  1444)    !     call shr_string_abort(subName//"ERROR: wrong string%nf ?")
  1445)    !  end if

  1448)          found = .true.
  1449)          kFld = k
  1450)          exit
  1451)       endif
  1452)       i0 = i1 + 2 ! start index for next iteration
  1453)       !--------------------------------------------------------

  1455)       !--------------------------------------------------------
  1456)       !--- get start index of field number (nFields + 1 - k ) ---
  1457)       n = index(string(1:j1),listDel,back=.true.)

  1461)    !     call shr_string_abort(subName//"ERROR: wrong string%nf ?")
  1462)    !  end if

  1464)       if (trim(fldStr) == string(j0:j1)) then
  1465)          found = .true.
  1466)          kFld = nFields + 1 - k
  1467)          exit
  1468)       endif
  1469)       j1 = j0 - 2 ! end index for next iteration
  1470)       !--------------------------------------------------------
  1471)       ! exit if all field names have been checked
  1472)       !--------------------------------------------------------
  1473)       if (2*k >= nFields) exit
  1474)    end do
  1475) 
  1476)    !--- not finding a field is not a fatal error ---
