    34) ! !IROUTINE: Establishment
    35) !
    36) ! !INTERFACE:
    37)   subroutine Establishment(lbg, ubg, lbp, ubp)
    38) !
    39) ! !DESCRIPTION:
    40) ! Calculates establishment of new pfts
    41) ! Called once per year
    42) !
    43) ! !USES:
    44)     use clmtype
    45)     use clm_varpar   , only : numpft
    46)     use clm_varcon   , only : istsoil
    47)     use clm_varctl   , only : iulog
    48)     use pftvarcon    , only : noveg, nc3_arctic_grass
    49)     use shr_const_mod, only : SHR_CONST_CDAY, SHR_CONST_PI, SHR_CONST_TKFRZ
    50) !
    51) ! !ARGUMENTS:
    52)     implicit none
    53)     integer , intent(in) :: lbg, ubg         ! gridcell bounds
    54)     integer , intent(in) :: lbp, ubp         ! pft bounds
    55) !
    56) ! !CALLED FROM:
    57) ! subroutine dv in module CNDVMod
    58) !
    59) ! !REVISION HISTORY:
    60) ! Author: Sam Levis (adapted from Stephen Sitch's LPJ subr. establishment)
    61) ! 3/4/02, Peter Thornton: Migrated to new data structures.
    62) ! 10/05 , Sam Levis: adapted to work with CN
    63) ! 09/07 , Sam Levis: as 10/05 but with current CN
    64) !
    65) ! !LOCAL VARIABLES:
    66) !
    67) ! local pointers to implicit in arguments
    68) !
    69)     logical , pointer :: pftmayexist(:)   ! exclude seasonal decid pfts from tropics [1=true, 0=false]
    70)     integer , pointer :: plandunit(:)     ! landunit of corresponding pft
    71)     integer , pointer :: pgridcell(:)     ! gridcell of corresponding pft
    72)     integer , pointer :: ltype(:)         ! landunit type for corresponding pft
    73)     real(r8), pointer :: tmomin20(:)      ! 20-yr running mean of tmomin
    74)     real(r8), pointer :: agdd20(:)        ! 20-yr running mean of agdd
    75)     real(r8), pointer :: agddtw(:)        ! accumulated growing degree days above twmax
    76)     real(r8), pointer :: prec365(:)       ! 365-day running mean of tot. precipitation
    77)     real(r8), pointer :: slatop(:)    !specific leaf area at top of canopy, projected area basis [m^2/gC]
    78)     real(r8), pointer :: dsladlai(:)  !dSLA/dLAI, projected area basis [m^2/gC]
    79)     real(r8), pointer :: woody(:)         ! ecophys const - woody pft or not
    80)     real(r8), pointer :: crownarea_max(:) ! ecophys const - tree maximum crown area [m2]
    81)     real(r8), pointer :: twmax(:)         ! ecophys const - upper limit of temperature of the warmest month
    82)     real(r8), pointer :: reinickerp(:)    ! ecophys const - parameter in allometric equation
    83)     real(r8), pointer :: dwood(:)         ! ecophys const - wood density (gC/m3)
    84)     real(r8), pointer :: allom1(:)        ! ecophys const - parameter in allometric
    85)     real(r8), pointer :: tcmin(:)         ! ecophys const - minimum coldest monthly mean temperature
    86)     real(r8), pointer :: tcmax(:)         ! ecophys const - maximum coldest monthly mean temperature
    87)     real(r8), pointer :: gddmin(:)        ! ecophys const - minimum growing degree days (at or above 5 C)
    88)     real(r8), pointer :: leafcmax(:)      ! (gC/m2) ann max leaf C
    89)     real(r8), pointer :: deadstemc(:)     ! (gC/m2) dead stem C
    90)     real(r8), pointer :: annsum_npp(:)         ! annual sum NPP (gC/m2/yr)
    91)     real(r8), pointer :: annsum_litfall(:)     ! annual sum litfall (gC/m2/yr)
    92) !
    93) ! local pointers to implicit in/out arguments
    94) !
    95)     integer , pointer :: ivt(:)           ! vegetation type for this pft
    96)     logical , pointer :: present(:)       ! true=> PFT present in patch
    97)     real(r8), pointer :: nind(:)          ! number of individuals (#/m**2)
    98) !
    99) ! local pointers to implicit out arguments
   100) !
   101)     real(r8), pointer :: fpcgrid(:)       ! foliar projective cover on gridcell (fraction)
   102)     real(r8), pointer :: crownarea(:)     ! area that each individual tree takes up (m^2)
   103)     real(r8), pointer :: greffic(:)       ! lpj's growth efficiency
   104)     real(r8), pointer :: heatstress(:)
   105) !
   106) !EOP
   107) !
   108) ! !OTHER LOCAL VARIABLES:
   109) !
   110)     integer  :: g,l,p,m                        ! indices
   111)     integer  :: fn, filterg(ubg-lbg+1)         ! local gridcell filter for error check
   112) !
   113) ! gridcell level variables
   114) !
   115)     integer  :: ngrass(lbg:ubg)                ! counter
   116)     integer  :: npft_estab(lbg:ubg)            ! counter
   117)     real(r8) :: fpc_tree_total(lbg:ubg)        ! total fractional cover of trees in vegetated portion of gridcell
   118)     real(r8) :: fpc_total(lbg:ubg)             ! old-total fractional vegetated portion of gridcell (without bare ground)
   119)     real(r8) :: fpc_total_new(lbg:ubg)         ! new-total fractional vegetated portion of gridcell (without bare ground)
   120) !
   121) ! pft level variables
   122) !
   123)     logical  :: survive(lbp:ubp)               ! true=>pft survives
   124)     logical  :: estab(lbp:ubp)                 ! true=>pft is established
   125)     real(r8) :: dstemc(lbp:ubp)                ! local copy of deadstemc
   126) !
   127) ! local and temporary variables or parameters
   128) !
   129)     real(r8) :: taper ! ratio of height:radius_breast_height (tree allometry)
   130)     real(r8) :: estab_rate                     !establishment rate
   131)     real(r8) :: estab_grid                     !establishment rate on grid cell
   132)     real(r8) :: fpcgridtemp                    ! temporary
   133)     real(r8) :: stemdiam                       ! stem diameter
   134)     real(r8) :: stocking                       ! #stems / ha (stocking density)
   135)     real(r8) :: lai_ind                        ! LAI per individual
   136)     real(r8) :: lm_ind                         !leaf carbon (gC/ind)
   137)     real(r8) :: fpc_ind                        !individual foliage projective cover
   138)     real(r8):: bm_delta
   139) 
   140)     real(r8), parameter :: ramp_agddtw = 300.0
   141) 
   142) ! minimum individual density for persistence of PFT (indiv/m2)
   143) !
   144)     real(r8), parameter :: nind_min = 1.0e-10_r8
   145) !
   146) ! minimum precip. for establishment (mm/s)
   147) !
   148)     real(r8), parameter :: prec_min_estab = 100._r8/(365._r8*SHR_CONST_CDAY)
   149) !
   150) ! maximum sapling establishment rate (indiv/m2)
   151) !
   152)     real(r8), parameter :: estab_max = 0.24_r8
   153) !-----------------------------------------------------------------------
   154) 
   155)     ! Assign local pointers to derived type members (gridcell-level)
   156) 
   157)     agdd20        => gdgvs%agdd20
   158)     tmomin20      => gdgvs%tmomin20
   159) 
   160)     ! Assign local pointers to derived type members (landunit-level)
   161) 
   162)     ltype => lun%itype
   163) 
   164)     ! Assign local pointers to derived type members (pft-level)
   165) 
   166)     ivt           => pft%itype
   167)     pgridcell     => pft%gridcell
   168)     plandunit     => pft%landunit
   169)     present       => pdgvs%present
   170)     nind          => pdgvs%nind
   171)     fpcgrid       => pdgvs%fpcgrid
   172)     crownarea     => pdgvs%crownarea
   173)     greffic       => pdgvs%greffic
   174)     heatstress    => pdgvs%heatstress
   175)     annsum_npp     => pepv%annsum_npp
   176)     annsum_litfall => pepv%annsum_litfall
   177)     prec365       => pdgvs%prec365
   178)     agddtw        => pdgvs%agddtw
   179)     pftmayexist   => pdgvs%pftmayexist
   180) 
   181)     ! Assign local pointers to derived type members (vegetation types)
   182) 
   183)     crownarea_max => dgv_pftcon%crownarea_max
   184)     twmax         => dgv_pftcon%twmax
   185)     reinickerp    => dgv_pftcon%reinickerp
   186)     allom1        => dgv_pftcon%allom1
   187)     tcmax         => dgv_pftcon%tcmax
   188)     tcmin         => dgv_pftcon%tcmin
   189)     gddmin        => dgv_pftcon%gddmin
   190)     leafcmax      => pcs%leafcmax
   191)     deadstemc     => pcs%deadstemc
   192)     slatop        => pftcon%slatop
   193)     dsladlai      => pftcon%dsladlai
   194)     dwood         => pftcon%dwood
   195)     woody         => pftcon%woody
   196) 
   197)     ! **********************************************************************
   198)     ! Slevis version of LPJ's subr. bioclim
   199)     ! Limits based on 20-year running averages of coldest-month mean
   200)     ! temperature and growing degree days (5 degree base).
   201)     ! For SURVIVAL, coldest month temperature and GDD should be
   202)     ! at least as high as PFT-specific limits.
   203)     ! For REGENERATION, PFT must be able to survive AND coldest month
   204)     ! temperature should be no higher than a PFT-specific limit.
   205)     ! **********************************************************************
   206) 
   207)     taper = 200._r8 ! make a global constant as with dwood (lpj's wooddens)
   208) 
   209)     ! Initialize gridcell-level metrics
   210) 
   211)     do g = lbg, ubg
   212)        ngrass(g) = 0
   213)        npft_estab(g) = 0
   214)        fpc_tree_total(g) = 0._r8
   215)        fpc_total(g) = 0._r8
   216)        fpc_total_new(g) = 0._r8
   217)     end do
   218) 
   219)     do p = lbp, ubp
   220)        g = pgridcell(p)
   221) 
   222)        ! Set the presence of pft for this gridcell
   223) 
   224)        if (nind(p) == 0._r8) present(p) = .false.
   225)        if (.not. present(p)) then
   226)           nind(p) = 0._r8
   227)           fpcgrid(p) = 0._r8
   228)        end if
   229)        survive(p) = .false.
   230)        estab(p)   = .false.
   231)        dstemc(p)  = deadstemc(p)
   232)     end do
   233) 
   234)     ! Must go thru all 16 pfts and decide which can/cannot establish or survive
   235)     ! Determine present, survive, estab.  Note: Even if tmomin20>tcmax, crops
   236)     ! and 2nd boreal summergreen tree cannot exist (see
   237)     ! EcosystemDynini) because this model cannot simulate such pfts, yet.
   238)     ! Note - agddtw is only defined at the pft level and has now been moved
   239)     ! to an if-statement below to determine establishment of boreal trees
   240) 
   241)     do p = lbp, ubp
   242)        g = pgridcell(p)
   243)        if (tmomin20(g) >= tcmin(ivt(p)) + SHR_CONST_TKFRZ ) then
   244)           if (tmomin20(g) <= tcmax(ivt(p)) + SHR_CONST_TKFRZ  .and. agdd20(g) >= gddmin(ivt(p))) then
   245)              estab(p) = .true.
   246)           end if
   247)           survive(p) = .true.
   248)           ! seasonal decid. pfts that would have occurred in regions without
   249)           ! short winter day lengths (see CNPhenology)
   250)           if (.not. pftmayexist(p)) then
   251)              survive(p) = .false.
   252)              estab(p) = .false.
   253)              pftmayexist(p) = .true.
   254)           end if
   255)        end if
   256)     end do
   257) 
   258)     do p = lbp, ubp
   259)        g = pgridcell(p)
   260)        l = plandunit(p)
   261) 
   262)        ! Case 1 -- pft ceases to exist -kill pfts not adapted to current climate
   263) 
   264)        if (present(p) .and. (.not. survive(p) .or. nind(p)<nind_min)) then
   265)           present(p) = .false.
   266)           fpcgrid(p) = 0._r8
   267)           nind(p) = 0._r8
   268)        end if
   269) 
   270)        ! Case 2 -- pft begins to exist - introduce newly "adapted" pfts
   271) 
   272)        if (ltype(l) == istsoil) then
   273)           if (.not. present(p) .and. prec365(p) >= prec_min_estab .and. estab(p)) then
   274)              if (twmax(ivt(p)) > 999._r8 .or. agddtw(p) == 0._r8) then
   275) 
   276)                 present(p) = .true.
   277)                 nind(p) = 0._r8
   278)                 ! lpj starts with fpcgrid=0 and calculates
   279)                 ! seed fpcgrid from the carbon of saplings;
   280)                 ! with CN we need the seed fpcgrid up front
   281)                 ! to scale seed leafc to lm_ind to get fpcgrid;
   282)                 ! sounds circular; also seed fpcgrid depends on sla,
   283)                 ! so theoretically need diff value for each pft;slevis
   284)                 fpcgrid(p) = 0.000844_r8
   285)                 if (woody(ivt(p)) < 1._r8) then
   286)                    fpcgrid(p) = 0.05_r8
   287)                 end if
   288) 
   289)                 ! Seed carbon for newly established pfts
   290)                 ! Equiv. to pleaf=1 & pstor=1 set in subr pftwt_cnbal (slevis)
   291)                 ! ***Dangerous*** to hardwire leafcmax here; find alternative!
   292)                 ! Consider just assigning nind and fpcgrid for newly
   293)                 ! established pfts instead of entering the circular procedure
   294)                 ! outlined in the paragraph above
   295)                 leafcmax(p) = 1._r8
   296)                 if (dstemc(p) <= 0._r8) dstemc(p) = 0.1_r8
   297) 
   298)              end if   ! conditions required for establishment
   299)           end if   ! conditions required for establishment
   300)        end if   ! if soil
   301) 
   302)        ! Case 3 -- some pfts continue to exist (no change) and some pfts
   303)        ! continue to not exist (no change). Do nothing for this case.
   304) 
   305)     end do
   306) 
   307)     ! Sapling and grass establishment
   308)     ! Calculate total woody FPC, FPC increment and grass cover (= crown area)
   309)     ! Calculate total woody FPC and number of woody PFTs present and able to establish
   310) 
   311)     do p = lbp, ubp
   312)        g = pgridcell(p)
   313)        if (present(p)) then
   314)           if (woody(ivt(p)) == 1._r8) then
   315)              fpc_tree_total(g) = fpc_tree_total(g) + fpcgrid(p)
   316)              if (estab(p)) npft_estab(g) = npft_estab(g) + 1
   317)           else if (woody(ivt(p)) < 1._r8 .and. ivt(p) > noveg) then !grass
   318)              ngrass(g) = ngrass(g) + 1
   319)           end if
   320)        end if
   321)     end do
   322) 
   323)     ! Above grid-level establishment counters are required for the next steps.
   324) 
   325)     do p = lbp, ubp
   326)        g = pgridcell(p)
   327) 
   328)        if (present(p) .and. woody(ivt(p)) == 1._r8 .and. estab(p)) then
   329) 
   330)           ! Calculate establishment rate over available space, per tree PFT
   331)           ! Max establishment rate reduced by shading as tree FPC approaches 1
   332)           ! Total establishment rate partitioned equally among regenerating woody PFTs
   333) 
   334)           estab_rate = estab_max * (1._r8-exp(5._r8*(fpc_tree_total(g)-1._r8))) / real(npft_estab(g))
   335) 
   336)           ! Calculate grid-level establishment rate per woody PFT
   337)           ! Space available for woody PFT establishment is fraction of grid cell
   338)           ! not currently occupied by woody PFTs
   339) 
   340)           estab_grid = estab_rate * (1._r8-fpc_tree_total(g))
   341) 
   342)           ! Add new saplings to current population
   343) 
   344)           nind(p) = nind(p) + estab_grid
   345) 
   346)           !slevis: lpj's lm_ind was the max leaf mass for the year;
   347)           !now lm_ind is the max leaf mass for the year calculated in CNFire
   348)           !except when a pft is newly established (nind==0); then lm_ind
   349)           !is assigned a leafcmax above
   350)  
   351)           lm_ind = leafcmax(p) * fpcgrid(p) / nind(p) ! nind>0 for sure
   352)           if (fpcgrid(p) > 0._r8 .and. nind(p) > 0._r8) then
   353)              stocking = nind(p)/fpcgrid(p) !#ind/m2 nat veg area -> #ind/m2 pft area
   354)              ! stemdiam derived here from cn's formula for htop found in
   355)              ! CNVegStructUpdate and cn's assumption stemdiam=2*htop/taper
   356)              ! this derivation neglects upper htop limit enforced elsewhere
   357)              stemdiam = (24._r8 * dstemc(p) / (SHR_CONST_PI * stocking * dwood(ivt(p)) * taper))**(1._r8/3._r8)
   358)           else
   359)              stemdiam = 0._r8
   360)           end if
   361)           ! Eqn D (now also in Light; need here for 1st yr when pfts haven't established, yet)
   362)           crownarea(p) = min(crownarea_max(ivt(p)), allom1(ivt(p))*stemdiam**reinickerp(ivt(p)))
   363) 
   364)           ! Update LAI and FPC
   365) 
   366)           if (crownarea(p) > 0._r8) then
   367)              if (dsladlai(ivt(p)) > 0._r8) then
   368)                 ! make lai_ind >= 0.001 to avoid killing plants at this stage
   369)                 lai_ind = max(0.001_r8,((exp(lm_ind*dsladlai(ivt(p)) + log(slatop(ivt(p)))) - &
   370)                               slatop(ivt(p)))/dsladlai(ivt(p))) / crownarea(p))
   371)              else ! currently redundant because dsladlai=0 for grasses only
   372)                 lai_ind = lm_ind * slatop(ivt(p)) / crownarea(p) ! lpj's formula
   373)              end if
   374)           else
   375)              lai_ind = 0._r8
   376)           end if
   377) 
   378)           fpc_ind = 1._r8 - exp(-0.5_r8*lai_ind)
   379)           fpcgrid(p) = crownarea(p) * nind(p) * fpc_ind
   380) 
   381)        end if   ! add new saplings block
   382)        if (present(p) .and. woody(ivt(p)) == 1._r8) then
   383)           fpc_total_new(g) = fpc_total_new(g) + fpcgrid(p)
   384)        end if
   385)     end do   ! close loop to update fpc_total_new
   386) 
   387)     ! Adjustments- don't allow trees to exceed 95% of vegetated landunit
   388) 
   389)     do p = lbp, ubp
   390)        g = pgridcell(p)
   391)        if (fpc_total_new(g) > 0.95_r8) then
   392)           if (woody(ivt(p)) == 1._r8 .and. present(p)) then
   393)              nind(p) = nind(p) * 0.95_r8 / fpc_total_new(g)
   394)              fpcgrid(p) = fpcgrid(p) * 0.95_r8 / fpc_total_new(g)
   395)           end if
   396)           fpc_total(g) = 0.95_r8
   397) 
   398)        else
   399)           fpc_total(g) = fpc_total_new(g)
   400)        end if
   401)     end do
   402) 
   403)     ! Section for grasses. Grasses can establish in non-vegetated areas
   404) 
   405)     do p = lbp, ubp
   406)        g = pgridcell(p)
   407)        if (present(p) .and. woody(ivt(p)) < 1._r8) then
   408)           if (leafcmax(p) <= 0._r8 .or. fpcgrid(p) <= 0._r8 ) then
   409)              present(p) = .false.
   410)              nind(p) = 0._r8
   411)           else
   412)              nind(p) = 1._r8 ! in case these grasses just established
   413)              crownarea(p) = 1._r8
   414)              lm_ind = leafcmax(p) * fpcgrid(p) / nind(p)
   415)              if (dsladlai(ivt(p)) > 0._r8) then
   416)                 lai_ind = max(0.001_r8,((exp(lm_ind*dsladlai(ivt(p)) + log(slatop(ivt(p)))) - &
   417)                               slatop(ivt(p)))/dsladlai(ivt(p))) / crownarea(p))
   418)              else ! 'if' is currently redundant b/c dsladlai=0 for grasses only
   419)                 lai_ind = lm_ind * slatop(ivt(p)) / crownarea(p)
   420)              end if
   421)              fpc_ind = 1._r8 - exp(-0.5_r8*lai_ind)
   422)              fpcgrid(p) = crownarea(p) * nind(p) * fpc_ind
   423)              fpc_total(g) = fpc_total(g) + fpcgrid(p)
   424)           end if
   425)        end if
   426)     end do   ! end of pft-loop
   427) 
   428)     ! Adjustment of fpc_total > 1 due to grasses (ivt >= nc3_arctic_grass)
   429) 
   430)     do p = lbp, ubp
   431)        g = pgridcell(p)
   432) 
   433)        if (fpc_total(g) > 1._r8) then
   434)           if (ivt(p) >= nc3_arctic_grass .and. fpcgrid(p) > 0._r8) then
   435)              fpcgridtemp = fpcgrid(p)
   436)              fpcgrid(p) = max(0._r8, fpcgrid(p) - (fpc_total(g)-1._r8))
   437)              fpc_total(g) = fpc_total(g) - fpcgridtemp + fpcgrid(p)
   438)           end if
   439)        end if
   440) 
   441)        ! Remove tiny fpcgrid amounts
   442) 
   443)        if (fpcgrid(p) < 1.e-15_r8) then
   444)           fpc_total(g) = fpc_total(g) - fpcgrid(p)
   445)           fpcgrid(p) = 0._r8
   446)           present(p) = .false.
   447)           nind(p) = 0._r8
   448)        end if
   449) 
   450)        ! Set the fpcgrid for bare ground if there is bare ground in
   451)        ! vegetated landunit and pft is bare ground so that everything
   452)        ! can add up to one.
   453) 
   454)        if (fpc_total(g) < 1._r8 .and. ivt(p) == noveg) then
   455)           fpcgrid(p) = 1._r8 - fpc_total(g)
   456)           fpc_total(g) = fpc_total(g) + fpcgrid(p)
   457)        end if
   458) 
   459)     end do
   460) 
   461)     ! Annual calculations used hourly in GapMortality
   462)     ! Ultimately may wish to place in separate subroutine...
   463) 
   464)     do p = lbp, ubp
   465)        g = pgridcell(p)
   466) 
   467)        ! Stress mortality from lpj's subr Mortality
   468) 
   469)        if (woody(ivt(p)) == 1._r8 .and. nind(p) > 0._r8 .and. &
   470)            leafcmax(p) > 0._r8 .and. fpcgrid(p) > 0._r8) then
   471) 
   472)           if (twmax(ivt(p)) < 999._r8) then
   473)              heatstress(p) = max(0._r8, min(1._r8, agddtw(p) / ramp_agddtw))
   474)           else
   475)              heatstress(p) = 0._r8
   476)           end if

   478)           ! Net individual living biomass increment

   480)           ! lpj's sap->heartwood turnover not included in litfall (slevis)

   491)           else ! currently redundant because dsladlai=0 for grasses only
   492)              greffic(p) = bm_delta / (lm_ind * slatop(ivt(p)))
   493)           end if
   494)        else
   495)           greffic(p) = 0.
   496)           heatstress(p) = 0.
