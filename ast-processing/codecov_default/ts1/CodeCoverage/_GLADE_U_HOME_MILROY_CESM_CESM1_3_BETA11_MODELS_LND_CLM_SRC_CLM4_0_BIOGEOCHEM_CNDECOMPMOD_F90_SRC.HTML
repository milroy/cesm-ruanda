   247)    soil2n_to_soil3n      => cnf%soil2n_to_soil3n

   255)    sminn_to_denit_s1s2   => cnf%sminn_to_denit_s1s2
   256)    sminn_to_denit_s2s3   => cnf%sminn_to_denit_s2s3
   257)    sminn_to_denit_s3s4   => cnf%sminn_to_denit_s3s4

   266)    ! set time steps
   267)    dt = real( get_step_size(), r8 )
   268)    dtd = dt/secspday
   269) 
   270)    ! set soil organic matter compartment C:N ratios (from Biome-BGC v4.2.0)
   271)    cn_s1 = 12.0_r8
   272)    cn_s2 = 12.0_r8
   273)    cn_s3 = 10.0_r8
   274)    cn_s4 = 10.0_r8

   279)    rf_l2s2 = 0.55_r8
   280)    rf_l3s3 = 0.29_r8
   281)    rf_s1s2 = 0.28_r8
   282)    rf_s2s3 = 0.46_r8
   283)    rf_s3s4 = 0.55

   289)    ! set initial base rates for decomposition mass loss (1/day)
   290)    ! (from Biome-BGC v4.2.0, using three SOM pools)
   291)    ! Value inside log function is the discrete-time values for a
   292)    ! daily time step model, and the result of the log function is
   293)    ! the corresponding continuous-time decay rate (1/day), following
   294)    ! Olson, 1963.
   295)    k_l1 = -log(1.0_r8-0.7_r8)
   296)    k_l2 = -log(1.0_r8-0.07_r8)
   297)    k_l3 = -log(1.0_r8-0.014_r8)
   298)    k_s1 = -log(1.0_r8-0.07_r8)
   299)    k_s2 = -log(1.0_r8-0.014_r8)
   300)    k_s3 = -log(1.0_r8-0.0014_r8)
   301)    k_s4 = -log(1.0_r8-0.0001_r8)
   302)    k_frag = -log(1.0_r8-0.001_r8)
   303) 
   304)    ! calculate the new discrete-time decay rate for model timestep
   305)    k_l1 = 1.0_r8-exp(-k_l1*dtd)
   306)    k_l2 = 1.0_r8-exp(-k_l2*dtd)
   307)    k_l3 = 1.0_r8-exp(-k_l3*dtd)
   308)    k_s1 = 1.0_r8-exp(-k_s1*dtd)
   309)    k_s2 = 1.0_r8-exp(-k_s2*dtd)
   310)    k_s3 = 1.0_r8-exp(-k_s3*dtd)
   311)    k_s4 = 1.0_r8-exp(-k_s4*dtd)
   312)    k_frag = 1.0_r8-exp(-k_frag*dtd)
   313)    
   314)    ! The following code implements the acceleration part of the AD spinup
   315)    ! algorithm, by multiplying all of the SOM decomposition base rates by 10.0.
   316) 
   317)    if (use_ad_spinup) then
   318)       spinup_scalar = 20._r8
   319)       k_s1 = k_s1 * spinup_scalar
   320)       k_s2 = k_s2 * spinup_scalar
   321)       k_s3 = k_s3 * spinup_scalar
   322)       k_s4 = k_s4 * spinup_scalar
   323)    end if
   324) 
   325)    ! calculate function to weight the temperature and water potential scalars
   326)    ! for decomposition control.  
   327) 
   328) 
   329)    ! the following normalizes values in fr so that they
   330)    ! sum to 1.0 across top nlevdecomp levels on a column
   331)    frw(lbc:ubc) = 0._r8
   332)    nlevdecomp=5
   333)    allocate(fr(lbc:ubc,nlevdecomp))
   334)    do j=1,nlevdecomp
   335)       do fc = 1,num_soilc
   336)          c = filter_soilc(fc)
   337)          frw(c) = frw(c) + dz(c,j)
   338)       end do
   339)    end do
   340)    do j = 1,nlevdecomp
   341)       do fc = 1,num_soilc
   342)          c = filter_soilc(fc)
   343)          if (frw(c) /= 0._r8) then
   344)             fr(c,j) = dz(c,j) / frw(c)
   345)          else
   346)             fr(c,j) = 0._r8
   347)          end if
   348)       end do
   349)    end do
   350) 
   351)    ! calculate rate constant scalar for soil temperature
   352)    ! assuming that the base rate constants are assigned for non-moisture
   353)    ! limiting conditions at 25 C. 
   354)    ! Peter Thornton: 3/13/09
   355)    ! Replaced the Lloyd and Taylor function with a Q10 formula, with Q10 = 1.5
   356)    ! as part of the modifications made to improve the seasonal cycle of 
   357)    ! atmospheric CO2 concentration in global simulations. This does not impact
   358)    ! the base rates at 25 C, which are calibrated from microcosm studies.
   359)    t_scalar(:) = 0._r8
   360)    do j = 1,nlevdecomp
   361)       do fc = 1,num_soilc
   362)          c = filter_soilc(fc)
   363)          t_scalar(c)=t_scalar(c) + (1.5**((t_soisno(c,j)-(SHR_CONST_TKFRZ+25._r8))/10._r8))*fr(c,j)
   364)       end do
   365)    end do
   366) 
   367)    ! calculate the rate constant scalar for soil water content.
   368)    ! Uses the log relationship with water potential given in
   369)    ! Andren, O., and K. Paustian, 1987. Barley straw decomposition in the field:
   370)    ! a comparison of models. Ecology, 68(5):1190-1200.
   371)    ! and supported by data in
   372)    ! Orchard, V.A., and F.J. Cook, 1983. Relationship between soil respiration
   373)    ! and soil moisture. Soil Biol. Biochem., 15(4):447-453.
   374) 
   375)    minpsi = -10.0_r8;
   376)    w_scalar(:) = 0._r8
   377)    do j = 1,nlevdecomp
   378)       do fc = 1,num_soilc
   379)          c = filter_soilc(fc)
   380)          maxpsi = psisat(c,j)
   381)          psi = min(soilpsi(c,j),maxpsi)
