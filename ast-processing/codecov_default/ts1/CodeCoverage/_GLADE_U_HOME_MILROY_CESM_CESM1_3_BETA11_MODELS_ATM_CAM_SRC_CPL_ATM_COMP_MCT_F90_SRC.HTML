   100)   subroutine atm_init_mct( EClock, cdata_a, x2a_a, a2x_a, NLFilename )
   101) 
   102)     !-----------------------------------------------------------------------
   103)     !
   104)     ! Arguments
   105)     !
   106)     type(ESMF_Clock),intent(inout)              :: EClock
   107)     type(seq_cdata), intent(inout)              :: cdata_a
   108)     type(mct_aVect), intent(inout)              :: x2a_a
   109)     type(mct_aVect), intent(inout)              :: a2x_a   
   110)     character(len=*), optional,   intent(IN)    :: NLFilename ! Namelist filename
   111)     !
   112)     ! Locals
   113)     !
   114)     type(mct_gsMap), pointer   :: gsMap_atm
   115)     type(mct_gGrid), pointer   :: dom_a
   116)     integer :: ATMID
   117)     integer :: mpicom_atm
   118)     integer :: lsize
   119)     integer :: iradsw
   120)     logical :: exists           ! true if file exists
   121)     real(r8):: nextsw_cday      ! calendar of next atm shortwave
   122)     integer :: stepno           ! time step			 
   123)     integer :: dtime_sync       ! integer timestep size
   124)     integer :: currentymd       ! current year-month-day
   125)     integer :: dtime            ! time step increment (sec)
   126)     integer :: atm_cpl_dt       ! driver atm coupling time step 
   127)     integer :: nstep            ! CAM nstep
   128)     real(r8):: caldayp1         ! CAM calendar day for for next cam time step
   129)     integer :: dtime_cam        ! Time-step increment (sec)
   130)     integer :: ymd              ! CAM current date (YYYYMMDD)
   131)     integer :: yr               ! CAM current year
   132)     integer :: mon              ! CAM current month
   133)     integer :: day              ! CAM current day
   134)     integer :: tod              ! CAM current time of day (sec)
   135)     integer :: start_ymd        ! Start date (YYYYMMDD)
   136)     integer :: start_tod        ! Start time of day (sec)
   137)     integer :: ref_ymd          ! Reference date (YYYYMMDD)
   138)     integer :: ref_tod          ! Reference time of day (sec)
   139)     integer :: stop_ymd         ! Stop date (YYYYMMDD)
   140)     integer :: stop_tod         ! Stop time of day (sec)
   141)     logical :: perpetual_run    ! If in perpetual mode or not
   142)     integer :: perpetual_ymd    ! Perpetual date (YYYYMMDD)
   143)     integer :: shrlogunit,shrloglev ! old values
   144)     logical :: first_time = .true.
   145)     character(len=SHR_KIND_CS) :: calendar  ! Calendar type
   146)     character(len=SHR_KIND_CS) :: starttype ! infodata start type
   147)     integer :: lbnum
   148)     integer :: hdim1_d, hdim2_d ! dimensions of rectangular horizontal grid
   149)                                 ! data structure, If 1D data structure, then
   150)                                 ! hdim2_d == 1.
   151)     character(len=64) :: filein ! Input namelist filename
   152)     !-----------------------------------------------------------------------
   153)     !
   154)     ! Determine cdata points
   155)     !
   156) #if (defined _MEMTRACE)
   157)     if(masterproc) then
   158)       lbnum=1
   159)       call memmon_dump_fort('memmon.out','atm_init_mct:start::',lbnum)
   160)     endif                      
   161) #endif                         
   162) 
   163)     call seq_cdata_setptrs(cdata_a, ID=ATMID, mpicom=mpicom_atm, &
   164)          gsMap=gsMap_atm, dom=dom_a, infodata=infodata)
   165) 
   166)     if (first_time) then
   167)        
   168)        call cam_instance_init(ATMID)
   169) 
   170)        ! Set filename specifier for restart surface file
   171)        ! (%c=caseid, $y=year, $m=month, $d=day, $s=seconds in day)
   172)        rsfilename_spec_cam = '%c.cam' // trim(inst_suffix) // '.rs.%y-%m-%d-%s.nc' 
   173) 
   174)        ! Determine attribute vector indices
   175) 
   176)        call cam_cpl_indices_set()
   177) 
   178)        ! Redirect share output to cam log
   179)        
   180)        call spmdinit(mpicom_atm)
   181)        
   182)        if (masterproc) then
   183)           inquire(file='atm_modelio.nml'//trim(inst_suffix), exist=exists)
   184)           if (exists) then
   185)              iulog = shr_file_getUnit()
   186)              call shr_file_setIO('atm_modelio.nml'//trim(inst_suffix), iulog)
   187)           endif
   188)           write(iulog,*) "CAM atmosphere model initialization"
   189)        endif
   190)        
   191)        call shr_file_getLogUnit (shrlogunit)
   192)        call shr_file_getLogLevel(shrloglev)
   193)        call shr_file_setLogUnit (iulog)
   194)        ! 
   195)        ! Consistency check                              
   196)        !
   197)        if (co2_readFlux_ocn .and. index_x2a_Faoo_fco2_ocn /= 0) then
   198)           write(iulog,*)'error co2_readFlux_ocn and index_x2a_Faoo_fco2_ocn cannot both be active'
   199)           call shr_sys_abort()
   200)        end if
   201)        ! 
   202)        ! Get data from infodata object
   203)        !
   204)        call seq_infodata_GetData( infodata,                                           &
   205)             case_name=caseid, case_desc=ctitle,                                       &
   206)             start_type=starttype,                                                     &
   207)             atm_adiabatic=adiabatic,                                                  &
   208)             atm_ideal_phys=ideal_phys,                                                &
   209)             aqua_planet=aqua_planet,                                                  &
   210)             brnch_retain_casename=brnch_retain_casename,                              &
   211)             single_column=single_column, scmlat=scmlat, scmlon=scmlon,                &
   212)             orb_eccen=eccen, orb_mvelpp=mvelpp, orb_lambm0=lambm0, orb_obliqr=obliqr, &
   213)             lnd_present=lnd_present, ocn_present=ocn_present,                         & 
   214)             perpetual=perpetual_run, perpetual_ymd=perpetual_ymd)
   215)        !
   216)        ! Get nsrest from startup type methods
   217)        !
   218)        if (     trim(starttype) == trim(seq_infodata_start_type_start)) then
   219)           nsrest = 0
   220)        else if (trim(starttype) == trim(seq_infodata_start_type_cont) ) then
   221)           nsrest = 1
   222)        else if (trim(starttype) == trim(seq_infodata_start_type_brnch)) then
   223)           nsrest = 3
   224)        else
   225)           write(iulog,*) 'atm_comp_mct: ERROR: unknown starttype'
   226)           call shr_sys_abort()
   227)        end if
   228)        !
   229)        ! Initialize time manager.
   230)        !
   231)        call seq_timemgr_EClockGetData(EClock, &
   232)             start_ymd=start_ymd, start_tod=start_tod, &
   233)             ref_ymd=ref_ymd, ref_tod=ref_tod,         &
   234)             stop_ymd=stop_ymd, stop_tod=stop_tod,     &
   235)             calendar=calendar )
   236)        !
   237)        ! Read namelist
   238)        !
   239)        filein = "atm_in" // trim(inst_suffix)
   240)        call read_namelist(single_column_in=single_column, scmlat_in=scmlat, &
   241)             scmlon_in=scmlon, nlfilename_in=filein)
   242)        !
   243)        ! Initialize cam time manager
   244)        !
   245)        if ( nsrest == 0 )then
   246)           call timemgr_init( calendar_in=calendar, start_ymd=start_ymd, &
   247)                start_tod=start_tod, ref_ymd=ref_ymd,      &
   248)                ref_tod=ref_tod, stop_ymd=stop_ymd,        &
   249)                stop_tod=stop_tod,                         &
   250)                perpetual_run=perpetual_run,               &
   251)                perpetual_ymd=perpetual_ymd )
   252)        end if
   253)        !
   254)        ! First phase of cam initialization 
   255)        ! Initialize mpicom_atm, allocate cam_in and cam_out and determine 
   256)        ! atm decomposition (needed to initialize gsmap) 
   257)        ! for an initial run, cam_in and cam_out are allocated in cam_initial
   258)        ! for a restart/branch run, cam_in and cam_out are allocated in restart 
   259)        ! Set defaults then override with user-specified input and initialize time manager
   260)        ! Note that the following arguments are needed to cam_init for timemgr_restart only
   261)        !
   262)        call cam_init( cam_out, cam_in, mpicom_atm, &
   263)             start_ymd, start_tod, ref_ymd, ref_tod, stop_ymd, stop_tod, &
   264)             perpetual_run, perpetual_ymd, calendar)
   265)        !
   266)        ! Check consistency of restart time information with input clock
   267)        !
   268)        if (nsrest /= 0) then
   269)           dtime_cam = get_step_size()
   270)           call timemgr_check_restart( calendar, start_ymd, start_tod, ref_ymd, &
   271)                ref_tod, dtime_cam, perpetual_run, perpetual_ymd)
   272)        end if
   273)        !
   274)        ! Initialize MCT gsMap, domain and attribute vectors (and dof)
   275)        !
   276)        call atm_SetgsMap_mct( mpicom_atm, ATMID, gsMap_atm )
   277)        lsize = mct_gsMap_lsize(gsMap_atm, mpicom_atm)
   278) 
   279)        ! Set dof (module variable, needed for pio for restarts)
   280)        call mct_gsmap_orderedpoints(gsmap_atm, iam, dof)
   281)        !
   282)        ! Initialize MCT domain 
   283)        !
   284)        call atm_domain_mct( lsize, gsMap_atm, dom_a )
   285)        !
   286)        ! Initialize MCT attribute vectors
   287)        !
   288)        call mct_aVect_init(a2x_a, rList=seq_flds_a2x_fields, lsize=lsize)
   289)        call mct_aVect_zero(a2x_a)
   290)        
   291)        call mct_aVect_init(x2a_a, rList=seq_flds_x2a_fields, lsize=lsize) 
   292)        call mct_aVect_zero(x2a_a)
   293)        !
   294)        ! Create initial atm export state
   295)        !
   296)        call atm_export( cam_out, a2x_a%rattr )
   297)        !
   298)        ! Set flag to specify that an extra albedo calculation is to be done (i.e. specify active)
   299)        !
   300)        call seq_infodata_PutData(infodata, atm_prognostic=.true.)
   301)        call get_horiz_grid_dim_d(hdim1_d, hdim2_d)
   302)        call seq_infodata_PutData(infodata, atm_nx=hdim1_d, atm_ny=hdim2_d)
   303) 
   304)        ! Set flag to indicate that CAM will provide carbon and dust deposition fluxes.
   305)        ! This is now hardcoded to .true. since the ability of CICE to read these
   306)        ! fluxes from a file has been removed.
   307)        call seq_infodata_PutData(infodata, atm_aero=.true.)
   308) 
   309)        !
   310)        ! Set time step of radiation computation as the current calday
   311)        ! This will only be used on the first timestep of an initial run
   312)        !
   313)        if (nsrest == 0) then
   314)           nextsw_cday = get_curr_calday()
   315)           call seq_infodata_PutData( infodata, nextsw_cday=nextsw_cday )
   316)        end if
   317)        
   318)        ! End redirection of share output to cam log
   319)        
   320)        call shr_file_setLogUnit (shrlogunit)
   321)        call shr_file_setLogLevel(shrloglev)
   322) 
   323)        first_time = .false.
   324) 
   325)     else
   326)        
   327)        ! For initial run, run cam radiation/clouds and return
   328)        ! For restart run, read restart x2a_a
   329)        ! Note - a2x_a is computed upon the completion of the previous run - cam_run1 is called
   330)        ! only for the purposes of finishing the flux averaged calculation to compute a2x_a
   331)        ! Note - cam_run1 is called on restart only to have cam internal state consistent with the 
   332)        ! a2x_a state sent to the coupler
   333) 
   334)        ! Redirect share output to cam log
   335) 
   336)        call shr_file_getLogUnit (shrlogunit)
   337)        call shr_file_getLogLevel(shrloglev)
   338)        call shr_file_setLogUnit (iulog)
   339) 
   340)        call seq_timemgr_EClockGetData(EClock,curr_ymd=CurrentYMD, StepNo=StepNo, dtime=DTime_Sync )
   341)        if (StepNo == 0) then
   342)           call atm_import( x2a_a%rattr, cam_in )
   343)           call cam_run1 ( cam_in, cam_out ) 
   344)           call atm_export( cam_out, a2x_a%rattr )
   345)        else
   346)           call atm_read_srfrest_mct( EClock, x2a_a, a2x_a )
   347)           call atm_import( x2a_a%rattr, cam_in )
   348)           call cam_run1 ( cam_in, cam_out ) 
   349)        end if
   350) 
   351)        ! Compute time of next radiation computation, like in run method for exact restart
   352) 
   353)        call seq_timemgr_EClockGetData(Eclock,dtime=atm_cpl_dt)
   354)        dtime = get_step_size()          
   355)        nstep = get_nstep()
   356)        if (nstep < 1 .or. dtime < atm_cpl_dt) then
   357)           nextsw_cday = radiation_nextsw_cday() 
   358)        else if (dtime == atm_cpl_dt) then
   359)           caldayp1 = get_curr_calday(offset=int(dtime))
   360)           nextsw_cday = radiation_nextsw_cday() 
   361)           if (caldayp1 /= nextsw_cday) nextsw_cday = -1._r8
   362)        else
   363)           call shr_sys_abort('dtime must be less than or equal to atm_cpl_dt')
   364)        end if
   365)        call seq_infodata_PutData( infodata, nextsw_cday=nextsw_cday ) 
   366) 
   367)        ! End redirection of share output to cam log
   368)        
   369)        call shr_file_setLogUnit (shrlogunit)
   370)        call shr_file_setLogLevel(shrloglev)
   371)        
   372)     end if
   373) 
   374) #if (defined _MEMTRACE )
   375)     if(masterproc) then
   376)       lbnum=1
   377)       call memmon_dump_fort('memmon.out','atm_init_mct:end::',lbnum)
   378)       call memmon_reset_addr()
   379)     endif
   380) #endif
   381)     
   382)     call shr_sys_flush(iulog)
   383) 
   384)  end subroutine atm_init_mct

   388)  subroutine atm_run_mct( EClock, cdata_a, x2a_a, a2x_a)
   389) 
   390)     !-----------------------------------------------------------------------
   391)     use time_manager,    only: advance_timestep, get_curr_date, get_curr_calday, &
   392) 	                       get_step_size
   393)    !use iop,             only: scam_use_iop_srf
   394)     use pmgrid,          only: plev, plevp
   395)     use constituents,    only: pcnst
   396)     use shr_sys_mod,     only: shr_sys_flush
   397) 
   398)     ! 
   399)     ! Arguments
   400)     !
   401)     type(ESMF_Clock)            ,intent(inout) :: EClock
   402)     type(seq_cdata)             ,intent(inout) :: cdata_a
   403)     type(mct_aVect)             ,intent(inout) :: x2a_a
   404)     type(mct_aVect)             ,intent(inout) :: a2x_a
   405)     !
   406)     ! Local variables
   407)     !
   408)     integer :: lsize           ! size of attribute vector
   409)     integer :: StepNo          ! time step			 
   410)     integer :: DTime_Sync      ! integer timestep size
   411)     integer :: CurrentYMD      ! current year-month-day
   412)     integer :: iradsw          ! shortwave radation frequency (time steps) 
   413)     logical :: dosend          ! true => send data back to driver
   414)     integer :: dtime           ! time step increment (sec)
   415)     integer :: atm_cpl_dt      ! driver atm coupling time step 
   416)     integer :: ymd_sync        ! Sync date (YYYYMMDD)
   417)     integer :: yr_sync         ! Sync current year
   418)     integer :: mon_sync        ! Sync current month
   419)     integer :: day_sync        ! Sync current day
   420)     integer :: tod_sync        ! Sync current time of day (sec)
   421)     integer :: ymd             ! CAM current date (YYYYMMDD)
   422)     integer :: yr              ! CAM current year
   423)     integer :: mon             ! CAM current month
   424)     integer :: day             ! CAM current day
   425)     integer :: tod             ! CAM current time of day (sec)
   426)     integer :: nstep           ! CAM nstep
   427)     integer :: shrlogunit,shrloglev ! old values
   428)     real(r8):: caldayp1        ! CAM calendar day for for next cam time step
   429)     real(r8):: nextsw_cday     ! calendar of next atm shortwave
   430)     logical :: rstwr           ! .true. ==> write restart file before returning
   431)     logical :: nlend           ! Flag signaling last time-step
   432)     logical :: rstwr_sync      ! .true. ==> write restart file before returning
   433)     logical :: nlend_sync      ! Flag signaling last time-step
   434)     logical :: first_time = .true.
   435)     integer :: lbnum
   436)     character(len=*), parameter :: subname="atm_run_mct"
   437)     !-----------------------------------------------------------------------
   438) 
   439) #if (defined _MEMTRACE)
   440)     if(masterproc) then
   441)       lbnum=1
   442)       call memmon_dump_fort('memmon.out',SubName //':start::',lbnum)
   443)     endif
   444) #endif
   445) 
   446)     ! Redirect share output to cam log
   447)     
   448)     call shr_file_getLogUnit (shrlogunit)
   449)     call shr_file_getLogLevel(shrloglev)
   450)     call shr_file_setLogUnit (iulog)
   451)     
   452)     ! Note that sync clock time should match cam time at end of time step/loop not beginning
   453)     
   454)     call seq_timemgr_EClockGetData(EClock,curr_ymd=ymd_sync,curr_tod=tod_sync, &
   455)        curr_yr=yr_sync,curr_mon=mon_sync,curr_day=day_sync)
   456) 
   457)     !load orbital parameters 
   458) 
   459)     call seq_infodata_GetData( infodata,                                           &
   460)        orb_eccen=eccen, orb_mvelpp=mvelpp, orb_lambm0=lambm0, orb_obliqr=obliqr)
   461) 
   462)     nlend_sync = seq_timemgr_StopAlarmIsOn(EClock)
   463)     rstwr_sync = seq_timemgr_RestartAlarmIsOn(EClock)
   464) 
   465)     ! Map input from mct to cam data structure
   466) 
   467)     call t_startf ('CAM_import')
   468)     call atm_import( x2a_a%rattr, cam_in )
   469)     call t_stopf  ('CAM_import')
   470)     
   471)     ! Cycle over all time steps in the atm coupling interval
   472)     
   473)     dosend = .false.
   474)     do while (.not. dosend)
   475) 
   476)        ! Determine if dosend
   477)        ! When time is not updated at the beginning of the loop - then return only if
   478)        ! are in sync with clock before time is updated
   479)        
   480)        call get_curr_date( yr, mon, day, tod )
   481)        ymd = yr*10000 + mon*100 + day
   482)        tod = tod

   569)     
   570)     ! End redirection of share output to cam log

   573)     call shr_file_setLogLevel(shrloglev)
