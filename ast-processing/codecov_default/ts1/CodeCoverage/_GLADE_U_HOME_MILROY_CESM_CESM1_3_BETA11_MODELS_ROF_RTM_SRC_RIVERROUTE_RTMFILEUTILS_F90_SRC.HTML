    59)    subroutine getfil (fulpath, locfn, iflag)
    60) 
    61)      ! !DESCRIPTION:
    62)      ! Obtain local copy of file. First check current working directory,
    63)      ! Next check full pathname[fulpath] on disk
    64)      ! 
    65)      ! !ARGUMENTS:
    66)      implicit none
    67)      character(len=*), intent(in)  :: fulpath !Archival or permanent disk full pathname
    68)      character(len=*), intent(out) :: locfn   !output local file name
    69)      integer,          intent(in)  :: iflag   !0=>abort if file not found 1=>do not abort
    70) 
    71)      ! !LOCAL VARIABLES:
    72)      integer i               !loop index
    73)      integer klen            !length of fulpath character string
    74)      logical lexist          !true if local file exists
    75)      !--------------------------------------------------
    76) 
    77)      ! get local file name from full name
    78)      locfn = get_filename( fulpath )
    79)      if (len_trim(locfn) == 0) then
    80) 	if (masterproc) write(iulog,*)'(GETFIL): local filename has zero length'
    81)         call shr_sys_abort()
    82)      else
    83)         if (masterproc) write(iulog,*)'(GETFIL): attempting to find local file ',  &
    84)              trim(locfn)
    85)      endif
    86) 
    87)      ! first check if file is in current working directory.
    88)      inquire (file=locfn,exist=lexist)
    89)      if (lexist) then
    90)         if (masterproc) write(iulog,*) '(GETFIL): using ',trim(locfn), &
    91)              ' in current working directory'
    92)         RETURN
    93)      endif
    94) 
    95)      ! second check for full pathname on disk
    96)      locfn = fulpath
    97) 
    98)      inquire (file=fulpath,exist=lexist)
    99)      if (lexist) then
   100)         if (masterproc) write(iulog,*) '(GETFIL): using ',trim(fulpath)
   101)         RETURN
   102)      else
   103)         if (masterproc) write(iulog,*)'(GETFIL): failed getting file from full path: ', fulpath
   104)         if (iflag==0) then
   105)            call shr_sys_abort ('GETFIL: FAILED to get '//trim(fulpath))
   106)         else
   107)            RETURN
   108)         endif
   109)      endif
