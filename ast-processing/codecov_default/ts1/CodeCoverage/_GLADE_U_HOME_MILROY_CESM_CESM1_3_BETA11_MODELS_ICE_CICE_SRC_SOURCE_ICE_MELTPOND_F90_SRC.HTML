    84)       subroutine compute_ponds(nx_block,ny_block,          &
    85)                                ilo, ihi, jlo, jhi,         &
    86)                                meltt, melts,  frain,       &
    87)                                aicen, vicen,  vsnon,       &
    88)                                trcrn, apondn, hpondn)

    90) ! !DESCRIPTION:
    91) !
    92) ! !REVISION HISTORY:
    93) !
    94) ! same as module
    95) !
    96) ! !USES:
    97) !
    98)       use ice_state, only: nt_Tsfc, nt_volpn
    99)       use ice_calendar, only: dt
   100)       use ice_domain_size, only: max_ntrcr

   102)       integer (kind=int_kind), intent(in) :: &
   103)          nx_block, ny_block, & ! block dimensions
   104)          ilo,ihi,jlo,jhi       ! beginning and end of physical domain
   105) 
   106)       real (kind=dbl_kind), dimension(nx_block,ny_block), &
   107)          intent(in) :: &
   108)          meltt, &
   109)          melts, &
   110)          frain, &
   111)          aicen, &
   112)          vicen, &
   113)          vsnon

   117)          apondn, &
   118)          hpondn
   119) 
   120)       real (kind=dbl_kind), dimension(nx_block,ny_block,max_ntrcr), &
   121)          intent(inout) :: &
   122)          trcrn
   123) 
   124)       real (kind=dbl_kind), dimension(nx_block,ny_block) :: &
   125)          volpn, &
   126)          Tsfcn
   127) 
   128)       integer (kind=int_kind), dimension (nx_block*ny_block) :: &
   129)          indxi, indxj     ! compressed indices for cells with ice melting
   130) 
   131)       integer (kind=int_kind) :: i,j,ij,icells
   132) 
   133)       real (kind=dbl_kind) :: hi,hs,dTs,rfrac,asnow
   134) 
   135)       real (kind=dbl_kind), parameter :: &
   136)          hi_min = p1
   137) 
   138)       Tsfcn(:,:) = trcrn(:,:,nt_Tsfc)
   139)       volpn(:,:) = trcrn(:,:,nt_volpn)
   140) 
   141)       !-----------------------------------------------------------------
   142)       ! Identify grid cells where ice can melt.
   143)       !-----------------------------------------------------------------
   144) 
   145)       icells = 0
   146)       do j = jlo, jhi
   147)       do i = ilo, ihi
   148)          if (aicen(i,j) > puny) then
   149)             icells = icells + 1
   150)             indxi(icells) = i
   151)             indxj(icells) = j
   152)          endif
   153)       enddo                     ! i
   154)       enddo                     ! j
   155) 
   156)       do ij = 1, icells
   157)          i = indxi(ij)
   158)          j = indxj(ij)
   159) 
   160)          hi = vicen(i,j)/aicen(i,j)
   161)          hs = vsnon(i,j)/aicen(i,j)
   162)          dTs = Timelt - Tsfcn(i,j)
   163) 
   164) ! Make the runoff fraction proportional to ice area.
   165)          rfrac = 0.85_dbl_kind - 0.7_dbl_kind*aicen(i,j)
   166) 
   167)          volpn(i,j) = volpn(i,j) + (c1-rfrac) &
   168)             * (meltt(i,j)*(rhoi/rhofresh) + melts(i,j)*(rhos/rhofresh) &
   169)             + frain(i,j)*dt/rhofresh)
   170) 
   171) !        Use exponential decrease in pond volume DAB
   172)          if (Tsfcn(i,j) .lt. Timelt-c2) then
   173)             volpn(i,j) = volpn(i,j) &
   174)                *exp(p01*(dTs-c2)/(Timelt-c2))
   175)          endif
   176) 
   177)          volpn(i,j) = max(volpn(i,j), c0)
   178) 
   179)          apondn(i,j) = min ( sqrt ( volpn(i,j) * 1.25 ), c1)
   180)          hpondn(i,j) = 0.8 * apondn(i,j)
   181) 
   182) !        Limit pond depth to 90% of ice thickness
   183)          if (hpondn(i,j) .gt. 0.9*hi) then
   184)             hpondn(i,j) = 0.9*hi
   185)             volpn(i,j) = hpondn(i,j)*apondn(i,j)
   186) !           apondn(i,j) = min(volpn(i,j)/hpondn(i,j), 1.)
   187)          endif
   188) 
   189) !        If we are freezing, do not change albedo
   190) !        if (Tsfcn(i,j) .lt. Timelt-0.15) apondn(i,j) = c0
   191) 
   192)          ! fractional area of snow cover
   193)          ! use linear function as in delta-Eddington
   194)          if (hs >= hsmin) then
   195)             asnow = min( hs/hs0, c1 )
   196)          else
   197)             asnow = c0
   198)          endif
   199) 
   200)          ! fractional area of snow cover
   201)          ! use snowpatch as in the CCSM3 shortwave
   202) !        if (hs > puny) then
   203) !           asnow = hs / (hs + snowpatch)
   204) !        else
   205) !           asnow = c0
   206) !        endif
   207) 
   208) !        If we have snow, reduce ponds
   209)          if ( hs .gt. puny ) then
   210)             apondn(i,j) = (c1-asnow)*apondn(i,j)
   211)             hpondn(i,j) = 0.8_dbl_kind * apondn(i,j)
   212)          endif
   213) 
   214) !        remove ponds if ice becomes very thin
   215)          if (hi .lt. hi_min) then
   216)             apondn(i,j) = c0
   217)             hpondn(i,j) = c0
   218)             volpn(i,j)  = c0
   219)          endif
   220) 
   221)          trcrn(i,j,nt_volpn) = volpn(i,j)
   222) 
   223)       enddo
   224) 
   225)       end subroutine compute_ponds
