   157)  subroutine ndep_interp( )
   158) 
   159)    !-----------------------------------------------------------------------
   160)    use decompMod       , only : get_proc_bounds
   161)    use clm_time_manager, only : get_curr_date, get_days_per_year

   165)    ! Local variables

   167)    integer :: g, ig, begg, endg
   168)    integer :: year    ! year (0, ...) for nstep+1
   169)    integer :: mon     ! month (1, ..., 12) for nstep+1
   170)    integer :: day     ! day of month (1, ..., 31) for nstep+1
   171)    integer :: sec     ! seconds into current date for nstep+1
   172)    integer :: mcdate  ! Current model date (yyyymmdd)
   173)    integer :: dayspyr ! days per year
   174)    !-----------------------------------------------------------------------
   175) 
   176)    call get_curr_date(year, mon, day, sec)
   177)    mcdate = year*10000 + mon*100 + day
   178) 
   179)    call shr_strdata_advance(sdat, mcdate, sec, mpicom, 'ndepdyn')
   180) 
   181)    call get_proc_bounds(begg, endg)
   182)    ig = 0
   183)    dayspyr = get_days_per_year( )
   184)    do g = begg,endg
   185)       ig = ig+1
   186)       clm_a2l%forc_ndep(g) = sdat%avs(1)%rAttr(1,ig) / (secspday * dayspyr)
   187)    end do

   193)   subroutine clm_domain_mct( dom_clm )
   194) 
   195)     !-------------------------------------------------------------------
   196)     ! Set domain data type for internal clm grid
   197)     use clm_varcon  , only : re
   198)     use domainMod   , only : ldomain
   199)     use seq_flds_mod
   200)     implicit none
   201)     ! 
   202)     ! arguments
   203)     type(mct_ggrid), intent(out)   :: dom_clm     ! Output domain information for land model
   204)     !
   205)     ! local variables
   206)     integer :: g,i,j              ! index
   207)     integer :: begg, endg         ! beginning and ending gridcell indices
   208)     integer :: lsize              ! land model domain data size
   209)     real(r8), pointer :: data(:)  ! temporary
   210)     integer , pointer :: idata(:) ! temporary
   211)     !-------------------------------------------------------------------
   212)     !
   213)     ! Initialize mct domain type
   214)     ! lat/lon in degrees,  area in radians^2, mask is 1 (land), 0 (non-land)
   215)     ! Note that in addition land carries around landfrac for the purposes of domain checking
   216)     ! 
   217)     lsize = mct_gsMap_lsize(gsmap_lnd_gdc2glo, mpicom)
   218)     call mct_gGrid_init( GGrid=dom_clm, CoordChars=trim(seq_flds_dom_coord), &
   219)                          OtherChars=trim(seq_flds_dom_other), lsize=lsize )
   220)     !
   221)     ! Allocate memory
   222)     !
   223)     allocate(data(lsize))
   224)     !
   225)     ! Determine global gridpoint number attribute, GlobGridNum, which is set automatically by MCT
   226)     !
   227)     call mct_gsMap_orderedPoints(gsmap_lnd_gdc2glo, iam, idata)
   228)     call mct_gGrid_importIAttr(dom_clm,'GlobGridNum',idata,lsize)
   229)     !
   230)     ! Determine domain (numbering scheme is: West to East and South to North to South pole)
   231)     ! Initialize attribute vector with special value
   232)     !
   233)     data(:) = -9999.0_R8 
   234)     call mct_gGrid_importRAttr(dom_clm,"lat"  ,data,lsize) 
   235)     call mct_gGrid_importRAttr(dom_clm,"lon"  ,data,lsize) 
   236)     call mct_gGrid_importRAttr(dom_clm,"area" ,data,lsize) 
   237)     call mct_gGrid_importRAttr(dom_clm,"aream",data,lsize) 
   238)     data(:) = 0.0_R8     
   239)     call mct_gGrid_importRAttr(dom_clm,"mask" ,data,lsize) 
   240)     !
   241)     ! Determine bounds
   242)     !
   243)     call get_proc_bounds(begg, endg)
   244)     !
   245)     ! Fill in correct values for domain components
   246)     ! Note aream will be filled in in the atm-lnd mapper
   247)     !
   248)     do g = begg,endg
   249)        i = 1 + (g - begg)
   250)        data(i) = ldomain%lonc(g)
   251)     end do
   252)     call mct_gGrid_importRattr(dom_clm,"lon",data,lsize) 
   253) 
   254)     do g = begg,endg
   255)        i = 1 + (g - begg)
   256)        data(i) = ldomain%latc(g)
   257)     end do
   258)     call mct_gGrid_importRattr(dom_clm,"lat",data,lsize) 
   259) 
   260)     do g = begg,endg
   261)        i = 1 + (g - begg)
   262)        data(i) = ldomain%area(g)/(re*re)
   263)     end do
   264)     call mct_gGrid_importRattr(dom_clm,"area",data,lsize) 
   265) 
   266)     do g = begg,endg
   267)        i = 1 + (g - begg)
   268)        data(i) = real(ldomain%mask(g), r8)
   269)     end do
   270)     call mct_gGrid_importRattr(dom_clm,"mask",data,lsize) 
   271) 
   272)     do g = begg,endg
   273)        i = 1 + (g - begg)
   274)        data(i) = real(ldomain%frac(g), r8)
   275)     end do
   276)     call mct_gGrid_importRattr(dom_clm,"frac",data,lsize) 
   277) 
   278)     deallocate(data)
   279)     deallocate(idata)
   280) 
   281)   end subroutine clm_domain_mct
