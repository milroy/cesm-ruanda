    41) ! !DESCRIPTION:
    42) ! This is the main subroutine to execute the calculation of leaf temperature
    43) ! and surface fluxes. Biogeophysics2.F90 then determines soil/snow and ground
    44) ! temperatures and updates the surface fluxes for the new ground
    45) ! temperature.
    46) !
    47) ! Calling sequence is:
    48) ! Biogeophysics1:           surface biogeophysics driver
    49) !  -> QSat:                 saturated vapor pressure, specific humidity, and
    50) !                           derivatives at ground surface and derivatives at
    51) !                           leaf surface using updated leaf temperature
    52) ! Leaf temperature
    53) ! Foliage energy conservation is given by the foliage energy budget
    54) ! equation:
    55) !                Rnet - Hf - LEf = 0
    56) ! The equation is solved by Newton-Raphson iteration, in which this
    57) ! iteration includes the calculation of the photosynthesis and
    58) ! stomatal resistance, and the integration of turbulent flux profiles.
    59) ! The sensible and latent heat transfer between foliage and atmosphere
    60) ! and ground is linked by the equations:
    61) !                Ha = Hf + Hg and Ea = Ef + Eg
    62) !

   159)     real(r8), pointer :: z0mg(:)          !roughness length over ground, momentum [m]
   160)     real(r8), pointer :: z0hg(:)          !roughness length over ground, sensible heat [m]
   161)     real(r8), pointer :: z0qg(:)          !roughness length over ground, latent heat [m]
   162)     real(r8), pointer :: emv(:)           !vegetation emissivity
   163)     real(r8), pointer :: z0m(:)           !momentum roughness length (m)
   164)     real(r8), pointer :: displa(:)        !displacement height (m)

   166)     real(r8), pointer :: z0hv(:)          !roughness length over vegetation, sensible heat [m]
   167)     real(r8), pointer :: z0qv(:)          !roughness length over vegetation, latent heat [m]
   168)     real(r8), pointer :: eflx_sh_tot(:)   !total sensible heat flux (W/m**2) [+ to atm]
   169)     real(r8), pointer :: eflx_sh_tot_u(:) !urban total sensible heat flux (W/m**2) [+ to atm]
   170)     real(r8), pointer :: eflx_sh_tot_r(:) !rural total sensible heat flux (W/m**2) [+ to atm]
   171)     real(r8), pointer :: eflx_lh_tot(:)   !total latent heat flux (W/m**2)  [+ to atm]
   172)     real(r8), pointer :: eflx_lh_tot_u(:) !urban total latent heat flux (W/m**2)  [+ to atm]
   173)     real(r8), pointer :: eflx_lh_tot_r(:) !rural total latent heat flux (W/m**2)  [+ to atm]
   174)     real(r8), pointer :: eflx_sh_veg(:)   !sensible heat flux from leaves (W/m**2) [+ to atm]
   175)     real(r8), pointer :: qflx_evap_tot(:) !qflx_evap_soi + qflx_evap_can + qflx_tran_veg
   176)     real(r8), pointer :: qflx_evap_veg(:) !vegetation evaporation (mm H2O/s) (+ = to atm)
   177)     real(r8), pointer :: qflx_tran_veg(:) !vegetation transpiration (mm H2O/s) (+ = to atm)
   178)     real(r8), pointer :: cgrnd(:)         !deriv. of soil energy flux wrt to soil temp [w/m2/k]
   179)     real(r8), pointer :: cgrnds(:)        !deriv. of soil sensible heat flux wrt soil temp [w/m2/k]
   180)     real(r8), pointer :: cgrndl(:)        !deriv. of soil latent heat flux wrt soil temp [w/m**2/k]
   181)     real(r8) ,pointer :: tssbef(:,:)      !soil/snow temperature before update

   220)     npfts         => grc%npfts
   221)     pfti          => grc%pfti

   224)

   226)     z_0_town      => lun%z_0_town
   227)     z_d_town      => lun%z_d_town
   228) 
   229)     ! Assign local pointers to derived type members (column-level)
   230) 
   231)     forc_pbot     => cps%forc_pbot
   232)     forc_q        => cws%forc_q
   233)     forc_t        => ces%forc_t
   234)     forc_th       => ces%forc_th
   235) 
   236)     cgridcell     => col%gridcell
   237)     clandunit     => col%landunit
   238)     ctype         => col%itype
   239)     beta          => cps%beta
   240)     dqgdT         => cws%dqgdT
   241)     emg           => cps%emg
   242)     frac_sno      => cps%frac_sno
   243)     h2osno        => cws%h2osno
   244)     htvp          => cps%htvp
   245)     qg            => cws%qg
   246)     smpmin        => cps%smpmin
   247)     snl           => cps%snl
   248)     t_grnd        => ces%t_grnd
   249)     thv           => ces%thv
   250)     z0hg          => cps%z0hg
   251)     z0mg          => cps%z0mg
   252)     z0qg          => cps%z0qg
   253)     zii           => cps%zii
   254)     bsw           => cps%bsw
   255)     dz            => cps%dz
   256)     h2osoi_ice    => cws%h2osoi_ice
   257)     h2osoi_liq    => cws%h2osoi_liq
   258)     soilalpha     => cws%soilalpha
   259)     soilbeta      => cws%soilbeta
   260)     soilalpha_u   => cws%soilalpha_u
   261)     sucsat        => cps%sucsat
   262)     t_soisno      => ces%t_soisno
   263)     tssbef        => ces%tssbef
   264)     watsat        => cps%watsat
   265)     watfc         => cps%watfc
   266)     watdry        => cps%watdry
   267)     watopt        => cps%watopt
   268)     rootfr_road_perv => cps%rootfr_road_perv
   269)     rootr_road_perv  => cps%rootr_road_perv
   270) 
   271)     ! Assign local pointers to derived type members (pft-level)
   272) 
   273)     ivt           => pft%itype
   274)     elai          => pps%elai
   275)     esai          => pps%esai
   276)     htop          => pps%htop
   277)     emv           => pps%emv
   278)     z0m           => pps%z0m
   279)     displa        => pps%displa
   280)     z0mv          => pps%z0mv
   281)     z0hv          => pps%z0hv
   282)     z0qv          => pps%z0qv
   283)     eflx_sh_tot   => pef%eflx_sh_tot
   284)     eflx_sh_tot_u => pef%eflx_sh_tot_u
   285)     eflx_sh_tot_r => pef%eflx_sh_tot_r
   286)     eflx_lh_tot   => pef%eflx_lh_tot
   287)     eflx_lh_tot_u => pef%eflx_lh_tot_u
   288)     eflx_lh_tot_r => pef%eflx_lh_tot_r
   289)     eflx_sh_veg   => pef%eflx_sh_veg
   290)     qflx_evap_tot => pwf%qflx_evap_tot
   291)     qflx_evap_veg => pwf%qflx_evap_veg
   292)     qflx_tran_veg => pwf%qflx_tran_veg
   293)     cgrnd         => pef%cgrnd
   294)     cgrnds        => pef%cgrnds
   295)     cgrndl        => pef%cgrndl
   296)     forc_hgt_u_pft => pps%forc_hgt_u_pft
   297)     forc_hgt_t_pft => pps%forc_hgt_t_pft
   298)     forc_hgt_q_pft => pps%forc_hgt_q_pft
   299)     plandunit      => pft%landunit
   300)     frac_veg_nosno => pps%frac_veg_nosno
   301)     thm            => pes%thm
   302)     pgridcell      => pft%gridcell
   303)     pcolumn        => pft%column
   304)     pwtgcell       => pft%wtgcell
   305) 
   306)     ! Assign local pointers to derived type members (ecophysiological)
   307)

   309)     displar       => pftcon%displar
   310) 
   311)     do j = -nlevsno+1, nlevgrnd
   312)        do fc = 1,num_nolakec
   313)           c = filter_nolakec(fc)
   314)           tssbef(c,j) = t_soisno(c,j)
   315)        end do
   316)     end do
   317) 
   318)     do fc = 1,num_nolakec
   319)        c = filter_nolakec(fc)
   320)        l = clandunit(c)
   321) 
   322)        if (ctype(c) == icol_road_perv) then
   323)           hr_road_perv = 0._r8
   324)        end if
   325) 
   326)        ! begin calculations that relate only to the column level
   327)        ! Ground and soil temperatures from previous time step
   328) 
   329)        t_grnd(c) = t_soisno(c,snl(c)+1)
   330) 
   331)        ! Saturated vapor pressure, specific humidity and their derivatives
   332)        ! at ground surface
   333) 
   334)        qred = 1._r8
   335)        if (ityplun(l)/=istwet .AND. ityplun(l)/=istice  &
   336)                               .AND. ityplun(l)/=istice_mec) then
   337)           if (ityplun(l) == istsoil .or. ityplun(l) == istcrop) then
   338)              wx   = (h2osoi_liq(c,1)/denh2o+h2osoi_ice(c,1)/denice)/dz(c,1)
   339)              fac  = min(1._r8, wx/watsat(c,1))
   340)              fac  = max( fac, 0.01_r8 )
   341)              psit = -sucsat(c,1) * fac ** (-bsw(c,1))
   342)              psit = max(smpmin(c), psit)
   343)              hr   = exp(psit/roverg/t_grnd(c))
   344)              qred = (1.-frac_sno(c))*hr + frac_sno(c)
   345) 
   346)              !! Lee and Pielke 1992 beta, added by K.Sakaguchi
   347)              if (wx < watfc(c,1) ) then  !when water content of ths top layer is less than that at F.C.
   348)                 fac_fc  = min(1._r8, wx/watfc(c,1))  !eqn5.66 but divided by theta at field capacity
   349)                 fac_fc  = max( fac_fc, 0.01_r8 )
   350)                 ! modifiy soil beta by snow cover. soilbeta for snow surface is one
   351)                 soilbeta(c) = (1._r8-frac_sno(c))*0.25_r8*(1._r8 - cos(SHR_CONST_PI*fac_fc))**2._r8 &
   352)                               + frac_sno(c)
   353)              else   !when water content of ths top layer is more than that at F.C.
   354)                 soilbeta(c) = 1._r8
   355)              end if
   356) 
   357)              soilalpha(c) = qred
   358)           ! Pervious road depends on water in total soil column
   359)           else if (ctype(c) == icol_road_perv) then
   360)              do j = 1, nlevsoi
   361)                 if (t_soisno(c,j) >= tfrz) then
   362)                    vol_ice = min(watsat(c,j), h2osoi_ice(c,j)/(dz(c,j)*denice))
   363)                    eff_porosity = watsat(c,j)-vol_ice
   364)                    vol_liq = min(eff_porosity, h2osoi_liq(c,j)/(dz(c,j)*denh2o))
   365)                    fac = min( max(vol_liq-watdry(c,j),0._r8) / (watopt(c,j)-watdry(c,j)), 1._r8 )
   366)                 else
   367)                    fac = 0._r8
   368)                 end if
   369)                 rootr_road_perv(c,j) = rootfr_road_perv(c,j)*fac
   370)                 hr_road_perv = hr_road_perv + rootr_road_perv(c,j)
   371)              end do
   372)              ! Allows for sublimation of snow or dew on snow
   373)              qred = (1.-frac_sno(c))*hr_road_perv + frac_sno(c)
   374) 
   375)              ! Normalize root resistances to get layer contribution to total ET
   376)              if (hr_road_perv .gt. 0._r8) then
   377)                 do j = 1, nlevsoi
   378)                    rootr_road_perv(c,j) = rootr_road_perv(c,j)/hr_road_perv
   379)                 end do
   380)              end if
   381)              soilalpha_u(c) = qred
   382)              soilbeta(c) = 0._r8
   383)           else if (ctype(c) == icol_sunwall .or. ctype(c) == icol_shadewall) then
   384)              qred = 0._r8
   385)              soilbeta(c) = 0._r8
   386)              soilalpha_u(c) = spval
   387)           else if (ctype(c) == icol_roof .or. ctype(c) == icol_road_imperv) then
   388)              qred = 1._r8
   389)              soilbeta(c) = 0._r8
   390)              soilalpha_u(c) = spval
   391)           end if
   392)        else
   393)           soilalpha(c) = spval
   394)           soilbeta(c) =   1._r8
   395)        end if
   396) 
   397)        call QSat(t_grnd(c), forc_pbot(c), eg, degdT, qsatg, qsatgdT)
   398) 
   399)        qg(c) = qred*qsatg
   400)        dqgdT(c) = qred*qsatgdT
