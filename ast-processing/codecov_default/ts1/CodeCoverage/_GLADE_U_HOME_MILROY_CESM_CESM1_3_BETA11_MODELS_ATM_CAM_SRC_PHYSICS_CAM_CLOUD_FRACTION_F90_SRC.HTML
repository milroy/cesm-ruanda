    82) subroutine cldfrc_readnl(nlfile)
    83) 
    84)    use namelist_utils,  only: find_group_name
    85)    use units,           only: getunit, freeunit
    86)    use mpishorthand
    87) 
    88)    character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
    89) 
    90)    ! Local variables
    91)    integer :: unitn, ierr
    92)    character(len=*), parameter :: subname = 'cldfrc_readnl'
    93) 
    94)    namelist /cldfrc_nl/ cldfrc_freeze_dry,      cldfrc_ice,    cldfrc_rhminl, &
    95)                         cldfrc_rhminl_adj_land, cldfrc_rhminh, cldfrc_sh1,    &
    96)                         cldfrc_rhminp,          cldfrc_rhminp_botmb, &
    97)                         cldfrc_sh2,             cldfrc_dp1,    cldfrc_dp2,    &
    98)                         cldfrc_premit,          cldfrc_premib, cldfrc_iceopt, &
    99)                         cldfrc_icecrit
   100)    !-----------------------------------------------------------------------------
   101) 
   102)    if (masterproc) then
   103)       unitn = getunit()
   104)       open( unitn, file=trim(nlfile), status='old' )
   105)       call find_group_name(unitn, 'cldfrc_nl', status=ierr)
   106)       if (ierr == 0) then
   107)          read(unitn, cldfrc_nl, iostat=ierr)
   108)          if (ierr /= 0) then
   109)             call endrun(subname // ':: ERROR reading namelist')
   110)          end if
   111)       end if
   112)       close(unitn)
   113)       call freeunit(unitn)
   114) 
   115)       ! set local variables
   116)       rhminl = cldfrc_rhminl
   117)       rhminl_adj_land = cldfrc_rhminl_adj_land
   118)       rhminh = cldfrc_rhminh
   119)       rhminp = cldfrc_rhminp
   120)       sh1    = cldfrc_sh1
   121)       sh2    = cldfrc_sh2
   122)       dp1    = cldfrc_dp1
   123)       dp2    = cldfrc_dp2
   124)       premit = cldfrc_premit
   125)       premib  = cldfrc_premib
   126)       iceopt  = cldfrc_iceopt
   127)       icecrit = cldfrc_icecrit
   128) 
   129)    end if
   130) 
   131) #ifdef SPMD
   132)    ! Broadcast namelist variables
   133)    call mpibcast(cldfrc_freeze_dry, 1, mpilog, 0, mpicom)
   134)    call mpibcast(cldfrc_ice,        1, mpilog, 0, mpicom)
   135)    call mpibcast(rhminl,            1, mpir8,  0, mpicom)
   136)    call mpibcast(rhminl_adj_land,   1, mpir8,  0, mpicom)
   137)    call mpibcast(rhminh,            1, mpir8,  0, mpicom)
   138)    call mpibcast(rhminp,            1, mpir8,  0, mpicom)
   139)    call mpibcast(sh1   ,            1, mpir8,  0, mpicom)
   140)    call mpibcast(sh2   ,            1, mpir8,  0, mpicom)
   141)    call mpibcast(dp1   ,            1, mpir8,  0, mpicom)
   142)    call mpibcast(dp2   ,            1, mpir8,  0, mpicom)
   143)    call mpibcast(premit,            1, mpir8,  0, mpicom)
   144)    call mpibcast(premib,            1, mpir8,  0, mpicom)
   145)    call mpibcast(iceopt,            1, mpiint, 0, mpicom)
   146)    call mpibcast(icecrit,           1, mpir8,  0, mpicom)
   147) #endif
   148) 
   149) end subroutine cldfrc_readnl

   196) subroutine cldfrc_init
   197) 
   198)    ! Initialize cloud fraction run-time parameters
   199) 
   200)    use cam_history,   only:  phys_decomp, addfld
   201)    use dycore,        only:  dycore_is, get_resolution
   202)    use phys_control,  only:  phys_getopts
   203) 
   204)    ! horizontal grid specifier
   205)    character(len=32) :: hgrid
   206) 
   207)    ! query interfaces for scheme settings
   208)    character(len=16) :: shallow_scheme, eddy_scheme, macrop_scheme
   209) 
   210)    integer :: k
   211)    !-----------------------------------------------------------------------------
   212) 
   213)    call phys_getopts(shallow_scheme_out = shallow_scheme ,&
   214)                      eddy_scheme_out    = eddy_scheme    ,&
   215)                      macrop_scheme_out  = macrop_scheme  )
   216) 
   217)    ! Limit CAM5 cloud physics to below top cloud level.
   218)    if (macrop_scheme /= "rk") top_lev = trop_cloud_top_lev
   219) 
   220)    hgrid = get_resolution()
   221) 
   222)    ! Turn off inversion_cld if any UW PBL scheme is being used
   223)    if ( (eddy_scheme .eq. 'diag_TKE' ) .or. (shallow_scheme .eq.  'UW' )) then
   224)       inversion_cld_off = .true.
   225)    else
   226)       inversion_cld_off = .false.
   227)    endif
   228) 
   229)    if ( masterproc ) then 
   230)       write(iulog,*)'tuning parameters cldfrc_init: inversion_cld_off',inversion_cld_off
   231)       write(iulog,*)'tuning parameters cldfrc_init: dp1',dp1,'dp2',dp2,'sh1',sh1,'sh2',sh2
   232)       if (shallow_scheme .ne. 'UW' ) then
   233)          write(iulog,*)'tuning parameters cldfrc_init: rhminl',rhminl,'rhminl_adj_land',rhminl_adj_land, &
   234)                        'rhminh',rhminh,'premit',premit,'premib',premib
   235)          write(iulog,*)'tuning parameters cldfrc_init: iceopt',iceopt,'icecrit',icecrit
   236)       endif
   237)    endif
   238) 
   239)    if (pref_mid(top_lev) > 7.e4_r8) &
   240)         call endrun ('cldfrc_init: model levels bracketing 700 mb not found')
   241) 
   242)    ! Find vertical level nearest 700 mb.
   243)    k700 = minloc(abs(pref_mid(top_lev:pver) - 7.e4_r8), 1)
   244) 
   245)    if (masterproc) then
   246)       write(iulog,*)'cldfrc_init: model level nearest 700 mb is',k700,'which is',pref_mid(k700),'pascals'
   247)    end if
   248) 
   249)    call addfld ('SH_CLD   ', 'fraction', pver, 'A', 'Shallow convective cloud cover'                          ,phys_decomp)
   250)    call addfld ('DP_CLD   ', 'fraction', pver, 'A', 'Deep convective cloud cover'                             ,phys_decomp)
   251) 
   252) end subroutine cldfrc_init

   256) subroutine cldfrc(lchnk   ,ncol    , pbuf,  &
   257)        pmid    ,temp    ,q       ,omga    , phis, &
   258)        shfrc   ,use_shfrc, &
   259)        cloud   ,rhcloud, clc     ,pdel    , &
   260)        cmfmc   ,cmfmc2  ,landfrac,snowh   ,concld  ,cldst   , &
   261)        ts      ,sst     ,ps      ,zdu     ,ocnfrac ,&
   262)        rhu00   ,cldice  ,icecldf ,liqcldf ,relhum  ,dindex )
   263)     !----------------------------------------------------------------------- 
   264)     ! 
   265)     ! Purpose: 
   266)     ! Compute cloud fraction 
   267)     ! 
   268)     ! 
   269)     ! Method: 
   270)     ! This calculate cloud fraction using a relative humidity threshold
   271)     ! The threshold depends upon pressure, and upon the presence or absence 
   272)     ! of convection as defined by a reasonably large vertical mass flux 
   273)     ! entering that layer from below.
   274)     ! 
   275)     ! Author: Many. Last modified by Jim McCaa
   276)     ! 
   277)     !-----------------------------------------------------------------------
   278)     use cam_history,   only: outfld
   279)     use physconst,     only: cappa, gravit, rair, tmelt
   280)     use wv_saturation, only: qsat, qsat_water, svp_ice
   281)     use phys_grid,     only: get_rlat_all_p, get_rlon_all_p
   282)     use dycore,        only: dycore_is, get_resolution
   283) 
   284)    
   285) !RBN - Need this to write shallow,deep fraction to phys buffer.
   286) !PJR - we should probably make seperate modules for determining convective
   287) !      clouds and make this one just responsible for relative humidity clouds
   288)     
   289)     use physics_buffer, only: physics_buffer_desc, pbuf_get_field
   290) 
   291)     ! Arguments
   292)     integer, intent(in) :: lchnk                  ! chunk identifier
   293)     integer, intent(in) :: ncol                   ! number of atmospheric columns
   294)     integer, intent(in) :: dindex                 ! 0 or 1 to perturb rh
   295)     
   296)     type(physics_buffer_desc), pointer :: pbuf(:)
   297)     real(r8), intent(in) :: pmid(pcols,pver)      ! midpoint pressures
   298)     real(r8), intent(in) :: temp(pcols,pver)      ! temperature
   299)     real(r8), intent(in) :: q(pcols,pver)         ! specific humidity
   300)     real(r8), intent(in) :: omga(pcols,pver)      ! vertical pressure velocity
   301)     real(r8), intent(in) :: cmfmc(pcols,pverp)    ! convective mass flux--m sub c
   302)     real(r8), intent(in) :: cmfmc2(pcols,pverp)   ! shallow convective mass flux--m sub c
   303)     real(r8), intent(in) :: snowh(pcols)          ! snow depth (liquid water equivalent)
   304)     real(r8), intent(in) :: pdel(pcols,pver)      ! pressure depth of layer
   305)     real(r8), intent(in) :: landfrac(pcols)       ! Land fraction
   306)     real(r8), intent(in) :: ocnfrac(pcols)        ! Ocean fraction
   307)     real(r8), intent(in) :: ts(pcols)             ! surface temperature
   308)     real(r8), intent(in) :: sst(pcols)            ! sea surface temperature
   309)     real(r8), intent(in) :: ps(pcols)             ! surface pressure
   310)     real(r8), intent(in) :: zdu(pcols,pver)       ! detrainment rate from deep convection
   311)     real(r8), intent(in) :: phis(pcols)           ! surface geopotential
   312)     real(r8), intent(in) :: shfrc(pcols,pver)     ! cloud fraction from convect_shallow
   313)     real(r8), intent(in) :: cldice(pcols,pver)    ! cloud ice mixing ratio
   314)     logical,  intent(in)  :: use_shfrc
   315) 
   316)     ! Output arguments
   317)     real(r8), intent(out) :: cloud(pcols,pver)     ! cloud fraction
   318)     real(r8), intent(out) :: rhcloud(pcols,pver)   ! cloud fraction
   319)     real(r8), intent(out) :: clc(pcols)            ! column convective cloud amount
   320)     real(r8), intent(out) :: cldst(pcols,pver)     ! cloud fraction
   321)     real(r8), intent(out) :: rhu00(pcols,pver)     ! RH threshold for cloud
   322)     real(r8), intent(out) :: relhum(pcols,pver)    ! RH 
   323)     real(r8), intent(out) :: icecldf(pcols,pver)   ! ice cloud fraction
   324)     real(r8), intent(out) :: liqcldf(pcols,pver)   ! liquid cloud fraction (combined into cloud)
   325) 
   326)     !---------------------------Local workspace-----------------------------
   327)     !
   328)     real(r8) concld(pcols,pver)    ! convective cloud cover
   329)     real(r8) cld                   ! intermediate scratch variable (low cld)
   330)     real(r8) dthdpmn(pcols)         ! most stable lapse rate below 750 mb
   331)     real(r8) dthdp                 ! lapse rate (intermediate variable)
   332)     real(r8) es(pcols,pver)        ! saturation vapor pressure
   333)     real(r8) qs(pcols,pver)        ! saturation specific humidity
   334)     real(r8) rhwght                ! weighting function for rhlim transition
   335)     real(r8) rh(pcols,pver)        ! relative humidity
   336)     real(r8) rhdif                 ! intermediate scratch variable
   337)     real(r8) strat                 ! intermediate scratch variable
   338)     real(r8) theta(pcols,pver)     ! potential temperature
   339)     real(r8) rhlim                 ! local rel. humidity threshold estimate
   340)     real(r8) coef1                 ! coefficient to convert mass flux to mb/d
   341)     real(r8) clrsky(pcols)         ! temporary used in random overlap calc
   342)     real(r8) rpdeli(pcols,pver-1) ! 1./(pmid(k+1)-pmid(k))
   343)     real(r8) rhpert                !the specified perturbation to rh
   344) 
   345)     real(r8), pointer, dimension(:,:) :: deepcu      ! deep convection cloud fraction
   346)     real(r8), pointer, dimension(:,:) :: shallowcu   ! shallow convection cloud fraction
   347) 
   348)     logical cldbnd(pcols)          ! region below high cloud boundary
   349) 
   350)     integer i, ierror, k           ! column, level indices
   351)     integer kp1, ifld
   352)     integer kdthdp(pcols)
   353)     integer numkcld                ! number of levels in which to allow clouds
   354) 
   355)     !  In Cloud Ice Content variables
   356)     real(r8) :: a,b,c,as,bs,cs        !fit parameters
   357)     real(r8) :: Kc                    !constant for ice cloud calc (wood & field)
   358)     real(r8) :: ttmp                  !limited temperature
   359)     real(r8) :: icicval               !empirical iwc value
   360)     real(r8) :: rho                   !local air density
   361)     real(r8) :: esl(pcols,pver)       !liq sat vapor pressure
   362)     real(r8) :: esi(pcols,pver)       !ice sat vapor pressure
   363)     real(r8) :: ncf,phi               !Wilson and Ballard parameters
   364) 
   365)     real(r8) thetas(pcols)                    ! ocean surface potential temperature
   366)     real(r8) :: clat(pcols)                   ! current latitudes(radians)
   367)     real(r8) :: clon(pcols)                   ! current longitudes(radians)
   368) 
   369)     ! Statement functions
   370)     logical land
   371)     land(i) = nint(landfrac(i)) == 1
   372) 
   373)     call get_rlat_all_p(lchnk, ncol, clat)
   374)     call get_rlon_all_p(lchnk, ncol, clon)
   375) 
   376)     call pbuf_get_field(pbuf, sh_frac_idx, shallowcu )
   377)     call pbuf_get_field(pbuf, dp_frac_idx, deepcu )
   378) 
   379)     ! Initialise cloud fraction
   380)     shallowcu = 0._r8
   381)     deepcu    = 0._r8
   382) 
   383)     !==================================================================================
   384)     ! PHILOSOPHY OF PRESENT IMPLEMENTATION
   385)     !++ag ice3
   386)     ! Modification to philosophy for ice supersaturation
   387)     ! philosophy below is based on RH water only. This is 'liquid condensation'
   388)     ! or liquid cloud (even though it will freeze immediately to ice)
   389)     ! The idea is that the RH limits for condensation are strict only for
   390)     ! water saturation
   391)     !
   392)     ! Ice clouds are formed by explicit parameterization of ice nucleation. 
   393)     ! Closure for ice cloud fraction is done on available cloud ice, such that
   394)     ! the in-cloud ice content matches an empirical fit
   395)     ! thus, icecldf = min(cldice/icicval,1) where icicval = f(temp,cldice,numice)
   396)     ! for a first cut, icicval=f(temp) only.
   397)     ! Combined cloud fraction is maximum overlap  cloud=max(1,max(icecldf,liqcldf))
   398)     ! No dA/dt term for ice?
   399)     !--ag
   400)     !
   401)     ! There are three co-existing cloud types: convective, inversion related low-level
   402)     ! stratocumulus, and layered cloud (based on relative humidity).  Layered and 
   403)     ! stratocumulus clouds do not compete with convective cloud for which one creates 
   404)     ! the most cloud.  They contribute collectively to the total grid-box average cloud 
   405)     ! amount.  This is reflected in the way in which the total cloud amount is evaluated 
   406)     ! (a sum as opposed to a logical "or" operation)
   407)     !
   408)     !==================================================================================
   409)     ! set defaults for rhu00
   410)     rhu00(:,:) = 2.0_r8
   411)     ! define rh perturbation in order to estimate rhdfda
   412)     rhpert = 0.01_r8 
   413) 
   414)     !set Wang and Sassen IWC paramters
   415)     a=26.87_r8
   416)     b=0.569_r8
   417)     c=0.002892_r8
   418)     !set schiller parameters
   419)     as=-68.4202_r8
   420)     bs=0.983917_r8
   421)     cs=2.81795_r8
   422)     !set wood and field paramters...
   423)     Kc=75._r8
   424) 
   425)     ! Evaluate potential temperature and relative humidity
   426)     ! If not computing ice cloud fraction then hybrid RH, if MG then water RH
   427)     if ( cldfrc_ice ) then
   428)        call qsat_water(temp(1:ncol,top_lev:pver), pmid(1:ncol,top_lev:pver), &
   429)             esl(1:ncol,top_lev:pver), qs(1:ncol,top_lev:pver))
   430) 
   431)        esi(1:ncol,top_lev:pver) = svp_ice(temp(1:ncol,top_lev:pver))
   432)     else
   433)        call qsat(temp(1:ncol,top_lev:pver), pmid(1:ncol,top_lev:pver), &
   434)             es(1:ncol,top_lev:pver), qs(1:ncol,top_lev:pver))
   435)     endif
   436) 
   437)     cloud    = 0._r8
   438)     icecldf  = 0._r8

   754)   subroutine cldfrc_fice(ncol, t, fice, fsnow)
   755) !
   756) ! Compute the fraction of the total cloud water which is in ice phase.
   757) ! The fraction depends on temperature only. 
   758) ! This is the form that was used for radiation, the code came from cldefr originally
   759) ! 
   760) ! Author: B. A. Boville Sept 10, 2002
   761) !  modified: PJR 3/13/03 (added fsnow to ascribe snow production for convection )
   762) !-----------------------------------------------------------------------
   763)     use physconst, only: tmelt

   765) ! Arguments

   767)     real(r8), intent(in)  :: t(pcols,pver)        ! temperature
   768) 
   769)     real(r8), intent(out) :: fice(pcols,pver)     ! Fractional ice content within cloud
   770)     real(r8), intent(out) :: fsnow(pcols,pver)    ! Fractional snow content for convection
   771) 
   772) ! Local variables
   773)     real(r8) :: tmax_fice                         ! max temperature for cloud ice formation
   774)     real(r8) :: tmin_fice                         ! min temperature for cloud ice formation
   775)     real(r8) :: tmax_fsnow                        ! max temperature for transition to convective snow
   776)     real(r8) :: tmin_fsnow                        ! min temperature for transition to convective snow
   777) 
   778)     integer :: i,k                                ! loop indexes
   779) 
   780) !-----------------------------------------------------------------------
   781) 
   782)     tmax_fice = tmelt - 10._r8        ! max temperature for cloud ice formation
   783)     tmin_fice = tmax_fice - 30._r8    ! min temperature for cloud ice formation
   784)     tmax_fsnow = tmelt                ! max temperature for transition to convective snow
   785)     tmin_fsnow = tmelt - 5._r8        ! min temperature for transition to convective snow
   786) 
   787)     fice(:,:top_lev-1) = 0._r8
   788)     fsnow(:,:top_lev-1) = 0._r8
   789) 
   790) ! Define fractional amount of cloud that is ice
   791)     do k=top_lev,pver
   792)        do i=1,ncol
   793) 
   794) ! If warmer than tmax then water phase
   795)           if (t(i,k) > tmax_fice) then
   796)              fice(i,k) = 0.0_r8
   797) 
   798) ! If colder than tmin then ice phase
   799)           else if (t(i,k) < tmin_fice) then
   800)              fice(i,k) = 1.0_r8
   801) 
   802) ! Otherwise mixed phase, with ice fraction decreasing linearly from tmin to tmax
   803)           else 
   804)              fice(i,k) =(tmax_fice - t(i,k)) / (tmax_fice - tmin_fice)
   805)           end if
   806) 
   807) ! snow fraction partitioning
   808) 
   809) ! If warmer than tmax then water phase
   810)           if (t(i,k) > tmax_fsnow) then
   811)              fsnow(i,k) = 0.0_r8
   812) 
   813) ! If colder than tmin then ice phase
   814)           else if (t(i,k) < tmin_fsnow) then
   815)              fsnow(i,k) = 1.0_r8
   816) 
   817) ! Otherwise mixed phase, with ice fraction decreasing linearly from tmin to tmax
   818)           else 
   819)              fsnow(i,k) =(tmax_fsnow - t(i,k)) / (tmax_fsnow - tmin_fsnow)
   820)           end if
   821) 
   822)        end do
   823)     end do

   825)   end subroutine cldfrc_fice
