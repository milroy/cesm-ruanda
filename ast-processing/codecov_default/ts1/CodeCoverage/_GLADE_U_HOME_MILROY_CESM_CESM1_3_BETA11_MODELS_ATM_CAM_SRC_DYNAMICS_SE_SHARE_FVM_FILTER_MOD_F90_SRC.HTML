     6) ! AUTHOR: CHRISTOPH ERATH, 30.November 2011                                         !
     7) ! This module contains everything  to do (ONLY) a CUBIC (3rd order) filtering       ! 
     8) ! which expect coefficient from the cubic reconstruciton                            !
     9) !-----------------------------------------------------------------------------------!
    10) module fvm_filter_mod
    11) 
    12)   use kinds, only                  : int_kind, real_kind
    13)   use dimensions_mod, only         : nc,nhc,nhe
    14)   use coordinate_systems_mod, only : cartesian2D_t,cartesian3D_t
    15)   use control_mod, only : north, south, east, west, neast, nwest, seast, swest
    16)   use parallel_mod, only : haltmp
    17)   implicit none
    18)   private
    19)   public :: monotonic_gradient_cart
    20)   public :: recons_val_cart

    22) ! ----------------------------------------------------------------------------------!
    23) !SUBROUTINE MONOTONIC_GRADIENT_CART---------------------------------------CE-for FVM!
    24) ! AUTHOR: CHRISTOPH ERATH, 30.November 2011                                         !
    25) ! DESCRIPTION: apply a monotonic filter to the calculated gradient from the cubic   !
    26) !              reconstruction                                                       !
    27) !                                                                                   !
    28) ! CALLS: monotonic_interior, monotonic_exterior, monotonic_exteriorswap             !
    29) ! INPUT: fcube  ...  tracer values incl. the halo zone                              !
    30) !        fvm  ...  structure incl. tracer values aso                                !
    31) !        desc   ...  edge descriptor, needed for reordering                         !
    32) ! INPUT/OUTPUT:                                                                     !
    33) !        recons ...  array of reconstructed coefficients, which will be scaled by   !
    34) !                    the filter                                                     !
    35) ! REMARK:                                                                           !
    36) !   This monotonizing scheme is based on the monotone scheme for unstructured       !
    37) !   grids of 'The design and application of upwind schemes on unstructed meshes',   !
    38) !   Barth and Jespersen, AIAA-89-0366, 27th Aerospace Sciences Meeting,             !
    39) !   January 9-12, 1989.                                                             !
    40) !-----------------------------------------------------------------------------------!
    41) subroutine monotonic_gradient_cart(fcube,fvm,recons,desc)
    42)   use fvm_control_volume_mod, only: fvm_struct
    43)   use edge_mod, only : edgedescriptor_t
    44)   
    45)   implicit none
    46)   real (kind=real_kind), dimension(1-nhc:nc+nhc, 1-nhc:nc+nhc),intent(in)    :: fcube
    47)   type (fvm_struct), intent(in)                                            :: fvm   
    48)   real (kind=real_kind),dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe),intent(inout) :: recons  
    49)   type (edgedescriptor_t),intent(in)                                         :: desc
    50)   
    51)   integer (kind=int_kind)          :: i, j, m, n,tmpi,tmpj
    52)   real (kind=real_kind)            :: local_min, local_max, value, phi, min_phi
    53)   real (kind=real_kind)            :: disc, mx, my
    54)   
    55)   call monotonic_int(fcube,recons,fvm%acartx,fvm%acarty,fvm%spherecentroid, &
    56)                      fvm%jx_min,fvm%jx_max,fvm%jy_min,fvm%jy_max,fvm%cubeboundary)
    57) 
    58)   ! all elements, which have a cube edge
    59)   if(fvm%cubeboundary>0)then
    60)     if(fvm%swap1) then
    61)       call monotonic_haloswap(fcube,recons,fvm%acartx1,fvm%acarty1,&
    62)                                   fvm%spherecentroid,fvm%jx_min1,fvm%jx_max1,&
    63)                                   fvm%jy_min1,fvm%jy_max1,fvm%cubeboundary,desc)
    64)     else
    65)       call monotonic_halo(fcube,recons,fvm%acartx1,fvm%acarty1,&
    66)                               fvm%spherecentroid,fvm%jx_min1,fvm%jx_max1,&
    67)                               fvm%jy_min1,fvm%jy_max1,fvm%cubeboundary,desc)
    68)     endif
    69)     ! only for corner elements 
    70)     if(fvm%cubeboundary>4)then
    71)       if(fvm%swap2) then
    72)         call monotonic_haloswap(fcube,recons,fvm%acartx2,fvm%acarty2,&
    73)                                     fvm%spherecentroid,fvm%jx_min2,fvm%jx_max2,&
    74)                                     fvm%jy_min2,fvm%jy_max2,fvm%cubeboundary,desc)
    75)       else
    76)         call monotonic_halo(fcube,recons,fvm%acartx2,fvm%acarty2,&
    77)                                 fvm%spherecentroid,fvm%jx_min2,fvm%jx_max2,&
    78)                                 fvm%jy_min2,fvm%jy_max2,fvm%cubeboundary,desc)
    79)       endif
    80)     endif
    81)   endif
    82) end subroutine monotonic_gradient_cart
    83) !END SUBROUTINE MONOTONIC_GRADIENT_CART-----------------------------------CE-for FVM!
    84) ! ----------------------------------------------------------------------------------!
    85) !SUBROUTINE MONOTONIC_INT-------------------------------------------------CE-for FVM!
    86) ! AUTHOR: CHRISTOPH ERATH, 30.November 2011                                         !
    87) ! DESCRIPTION: filtering for interior elements                                      !
    88) !              this subroutine calls an optimized minmax search algorithm, and does !
    89) !              not have to reorder the coefficients and centroids                   !
    90) !                                                                                   !
    91) ! CALLS: minmax_patch, recons_val_cart, slopelimiter_val                            !
    92) ! INPUT: fcube  ...  tracer values incl. the halo zone                              !
    93) !        fvm  ...  structure incl. tracer values aso                                !
    94) !        acartx ...  x cartesian coordinats of the arrival grid on the cube         !
    95) !        acarty ...  y cartesian coordinats of the arrival grid on the cube         !
    96) !        centroid..  x,y,x^2,y^2,xy                                                 !
    97) !        jx_min, jx_max, jy_min, jy_max                                             !
    98) !               ... cell boundaries in x and y directions                           !
    99) !        cubeboundary. 0-8, depending of an element is interior, west, south, aso   !
   100) ! INPUT/OUTPUT:                                                                     !
   101) !        recons ...  array of reconstructed coefficients, which will be scaled by   !
   102) !                    the filter                                                     !
   103) !-----------------------------------------------------------------------------------!
   104) subroutine monotonic_int(fcube,recons,acartx,acarty,centroid,jx_min,jx_max,jy_min,jy_max,&
   105)                          cubeboundary)
   106)   
   107)   implicit none
   108)   real (kind=real_kind), dimension(1-nhc:nc+nhc, 1-nhc:nc+nhc),intent(in)    :: fcube  
   109)   real (kind=real_kind),dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe),intent(inout) :: recons  
   110)   real (kind=real_kind), dimension(-nhe:nc+2+nhe), intent(in)   :: acartx, acarty
   111)   real (kind=real_kind), dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe), intent(in)  :: centroid
   112)   integer, intent(in)                                  :: cubeboundary
   113)   
   114)   integer, intent(in)                                  :: jx_min,jx_max,jy_min,jy_max
   115)   
   116)   integer (kind=int_kind)          :: i, j, m, n
   117)   real (kind=real_kind)            :: local_min, local_max, value, phi, min_phi
   118)   real (kind=real_kind)            :: disc, mx, my
   119) 
   120)   ! Apply monotone limiting, go through all coefficients
   121)   do j = jy_min, jy_max-1
   122)     do i = jx_min, jx_max-1
   123)       call minmax_patch(fcube, i, j, local_min, local_max, cubeboundary)
   124)       ! Initialize the limiter
   125)       min_phi = 1.0D0
   126)       ! Check the minima and maxima at the corner points of the element
   127)       do m = i, i+1
   128)         do n = j, j+1
   129)           ! Evaluate the function at each corner point
   130)           call recons_val_cart(fcube, acartx(m),acarty(n),centroid,recons,i,j,value)
   131)           call slopelimiter_val(value, fcube(i,j), local_min, local_max, min_phi)
   132)         enddo
   133)       enddo
   134)       ! For the third order method, the minima and maxima may occur along
   135)       ! the line segments given by du/dx = 0 and du/dy = 0.  Also check
   136)       ! for the presence of a maxima / minima of the quadratic within
   137)       ! the domain.
   138)       disc =  4.0D0 * recons(4,i,j) * recons(3,i,j) - recons(5,i,j)**2
   139)       ! Check if the quadratic is minimized within the element
   140)       ! Extrema in the interior of the element (there might be just one candidate)
   141)       ! DO NOT NEED ABS here, if disc<0 we have a saddle point (no maximum or minimum)
   142)       if (abs(disc) > 1.0D-12) then
   143)         mx = recons(5,i,j) * recons(2,i,j)        &
   144)              - 2.0D0 * recons(4,i,j) * recons(1,i,j)
   145)         my = recons(5,i,j) * recons(1,i,j)        &
   146)              - 2.0D0 * recons(3,i,j) * recons(2,i,j)
   147) 
   148)         mx = mx / disc + centroid(1,i,j)
   149)         my = my / disc + centroid(2,i,j)
   150) 
   151)         if ((mx - acartx(i)   > -1.0D-12) .and. &
   152)             (mx - acartx(i+1) <  1.0D-12) .and. &
   153)             (my - acarty(j)   > -1.0D-12) .and. &
   154)             (my - acarty(j+1) <  1.0D-12)) then       
   155)           call recons_val_cart(fcube, mx, my, centroid, recons, i, j, value)
   156)           call slopelimiter_val(value, fcube(i,j),local_min, local_max, min_phi)
   157)         endif
   158)       endif
   159)       ! Check all potential minimizer points along element boundaries
   160)       if (abs(recons(5,i,j)) > 1.0D-12) then
   161)         ! Left/right edge, intercept with du/dx = 0
   162)         do m = i, i+1
   163)           my = - recons(1,i,j) - 2.0D0 * recons(3,i,j) * &
   164)                  (acartx(m) - centroid(1,i,j))
   165)           my = my / recons(5,i,j) + centroid(2,i,j)
   166)           if ((my > acarty(j)) .and. (my < acarty(j+1))) then
   167)             call recons_val_cart(fcube, acartx(m), my, centroid, recons, i, j, value)
   168)             call slopelimiter_val(value, fcube(i,j),local_min, local_max, min_phi)
   169)           endif
   170)         enddo
   171)         ! Top/bottom edge, intercept with du/dy = 0
   172)         do n = j, j+1
   173)           mx = - recons(2,i,j) - 2.0D0 * recons(4,i,j) * &
   174)                (acarty(n) - centroid(2,i,j))
   175)           mx = mx / recons(5,i,j) + centroid(1,i,j)
   176)           if ((mx > acartx(i)) .and. (mx < acartx(i+1))) then
   177)             call recons_val_cart(fcube, mx, acarty(n),centroid,recons, i, j, value)
   178)             call slopelimiter_val(value, fcube(i,j),local_min, local_max, min_phi)
   179)           endif
   180)         enddo
   181)       endif
   182)       ! Top/bottom edge, y=const., du/dx=0
   183)       if (abs(recons(3,i,j)) > 1.0D-12) then
   184)         do n = j, j+1
   185)           mx = - recons(1,i,j) - recons(5,i,j) * &
   186)                (acarty(n) - centroid(2,i,j))
   187) 
   188)           mx = mx / (2.0D0 * recons(3,i,j)) + centroid(1,i,j)
   189)           if ((mx > acartx(i)) .and. (mx < acartx(i+1))) then
   190)             call recons_val_cart(fcube, mx, acarty(n), centroid, recons, i, j, value)
   191)             call slopelimiter_val(value, fcube(i,j),local_min, local_max, min_phi)
   192)           endif
   193)         enddo
   194)       endif
   195)         ! Left/right edge, x=const., du/dy=0
   196)       if (abs(recons(4,i,j)) > 1.0D-12) then
   197)         do m = i, i+1
   198)           my = - recons(2,i,j) - recons(5,i,j) * &
   199)                  (acartx(m) - centroid(1,i,j))
   200)           my = my / (2.0D0 * recons(4,i,j)) + centroid(2,i,j)
   201)           if ((my > acarty(j)) .and. (my < acarty(j+1))) then
   202)             call recons_val_cart(fcube, acartx(m), my, centroid, recons, i, j, value)
   203)             call slopelimiter_val(value, fcube(i,j),local_min, local_max, min_phi)
   204)           endif
   205)         enddo
   206)       endif
   207)       
   208)       if ((min_phi < -1.0D-12) .or. (min_phi > 1.0D0 + 1.0D-12)) then
   209)         write(*,*) value, fcube(i,j),local_min, local_max, min_phi
   210)         call haltmp("Fatal Error in monotonic_int: slope limiter out of range (not in [0,1])!")        
   211)         stop
   212)       endif
   213)       ! Apply monotone limiter to all reconstruction coefficients
   214)       recons(1,i,j) = min_phi * recons(1,i,j)
   215)       recons(2,i,j) = min_phi * recons(2,i,j)
   216)       recons(3,i,j) = min_phi * recons(3,i,j)
   217)       recons(4,i,j) = min_phi * recons(4,i,j)
   218)       recons(5,i,j) = min_phi * recons(5,i,j)
   219)     enddo
   220)   enddo
   221) end subroutine monotonic_int
   222) !END SUBROUTINE MONOTONIC_INT---------------------------------------------CE-for FVM!
   223) ! ----------------------------------------------------------------------------------!
   224) !SUBROUTINE MONOTONIC_HALO------------------------------------------------CE-for FVM!
   225) ! AUTHOR: CHRISTOPH ERATH, 30.November 2011                                         !
   226) ! DESCRIPTION: filtering for elements with an edge on the cube boundary             !
   227) !             for certain elements we have to reorder the coefficients and centroids!
   228) !                                                                                   !
   229) ! CALLS: minmax_patch_halo, recons_val_cart, slopelimiter_val                       !
   230) ! INPUT: fcube  ...  tracer values incl. the halo zone                              !
   231) !        fvm  ...  structure incl. tracer values aso                                !
   232) !        acartx ...  x cartesian coordinats of the arrival grid on the cube         !
   233) !        acarty ...  y cartesian coordinats of the arrival grid on the cube         !
   234) !        centroid..  x,y,x^2,y^2,xy                                                 !
   235) !        jx_min, jx_max, jy_min, jy_max                                             !
   236) !               ... cell boundaries in x and y directions                           !
   237) !        cubeboundary. 0-8, depending of an element is interior, west, south, aso   !
   238) !        desc   ... edge descriptor, needed for reordering                          !
   239) ! INPUT/OUTPUT:                                                                     !
   240) !        recons ...  array of reconstructed coefficients, which will be scaled by   !
   241) !                    the filter                                                     !
   242) !-----------------------------------------------------------------------------------!
   243) subroutine monotonic_halo(fcube,recons,acartx,acarty,centroid,&
   244)                               jx_min,jx_max,jy_min,jy_max,cubeboundary,desc)
   245)   use edge_mod, only : edgedescriptor_t
   246)   
   247)   implicit none
   248)   real (kind=real_kind), dimension(1-nhc:nc+nhc, 1-nhc:nc+nhc),intent(in)    :: fcube  
   249)   real (kind=real_kind),dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe),intent(inout) :: recons  
   250)   real (kind=real_kind), dimension(-nhe:nc+2+nhe), intent(in)   :: acartx, acarty
   251)   real (kind=real_kind), dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe), intent(in)  :: centroid
   252)   integer, intent(in)                                  :: jx_min,jx_max,jy_min,jy_max
   253)   integer, intent(in)                                  :: cubeboundary
   254)   type (edgedescriptor_t),intent(in)                   :: desc
   255)   
   256)   integer (kind=int_kind)          :: i, j, m, n, tmpi, tmpj
   257)   real (kind=real_kind)            :: local_min, local_max, value, phi, min_phi
   258)   real (kind=real_kind)            :: disc, mx, my
   259) 
   260)   do j = jy_min, jy_max-1
   261)     do i = jx_min, jx_max-1
   262)       if(desc%reverse(west) .or. desc%reverse(east)) then
   263)         tmpj=jy_max-1+jy_min-j
   264)       else
   265)         tmpj=j
   266)       endif
   267)       if(desc%reverse(north) .or. desc%reverse(south)) then
   268)         tmpi=jx_max-1+jx_min-i
   269)       else
   270)         tmpi=i
   271)       endif
   272)       call minmax_patch_halo(fcube, tmpi, tmpj, cubeboundary,local_min, local_max)
   273)       ! Initialize the limiter
   274)       min_phi = 1.0D0
   275)       ! For the second-order calculation, the minima and maxima will occur
   276)       ! at the corner points of the element
   277)       do m = i, i+1
   278)         do n = j, j+1
   279)         ! Evaluate the function at each corner point
   280)           call recons_val_cart(fcube, acartx(m),acarty(n),centroid,recons,tmpi,tmpj,value)
   281)           call slopelimiter_val(value, fcube(tmpi,tmpj), local_min, local_max, min_phi) 
   282)         enddo
   283)       enddo 
   284) 
   285)       disc = 4.0D0 * recons(4,tmpi,tmpj) * recons(3,tmpi,tmpj) - recons(5,tmpi,tmpj)**2
   286)       ! Check if the quadratic is minimized within the element
   287)       if (abs(disc) > 1.0D-12) then
   288)         mx =  recons(5,tmpi,tmpj) * recons(2,tmpi,tmpj)        &
   289)              - 2.0D0 * recons(4,tmpi,tmpj) * recons(1,tmpi,tmpj)
   290)         my =   recons(5,tmpi,tmpj) * recons(1,tmpi,tmpj)        &
   291)              - 2.0D0 * recons(3,tmpi,tmpj) * recons(2,tmpi,tmpj)
   292) 
   293)         mx = mx / disc + centroid(1,tmpi,tmpj)
   294)         my = my / disc + centroid(2,tmpi,tmpj)
   295)         if ((mx - acartx(i)   > -1.0D-12) .and. &
   296)             (mx - acartx(i+1) <  1.0D-12) .and. &
   297)             (my - acarty(j)   > -1.0D-12) .and. &
   298)             (my - acarty(j+1) <  1.0D-12)) then       
   299)           call recons_val_cart(fcube, mx, my, centroid, recons, tmpi, tmpj, value)
   300)           call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   301)         endif
   302)       endif
   303)       ! Check all potential minimizer points along element boundaries
   304)       if (abs(recons(5,tmpi,tmpj)) > 1.0D-12) then
   305)         ! Left/right edge, intercept with du/dx = 0
   306)         do m = i, i+1
   307)           my = - recons(1,tmpi,tmpj) - 2.0D0 * recons(3,tmpi,tmpj) * &
   308)                 (acartx(m) - centroid(1,tmpi,tmpj))
   309)           my = my / recons(5,tmpi,tmpj) + centroid(2,tmpi,tmpj)
   310)           if ((my > acarty(j)) .and. (my < acarty(j+1))) then
   311)             call recons_val_cart(fcube, acartx(m), my, centroid, recons, tmpi, tmpj, value)
   312)             call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   313)           endif
   314)         enddo
   315)         ! Top/bottom edge, intercept with du/dy = 0
   316)         do n = j, j+1
   317)           mx = - recons(2,tmpi,tmpj) - 2.0D0 * recons(4,tmpi,tmpj) * &
   318)               (acarty(n) - centroid(2,tmpi,tmpj))
   319)           mx = mx / recons(5,tmpi,tmpj) + centroid(1,tmpi,tmpj)
   320)           if ((mx > acartx(i)) .and. (mx < acartx(i+1))) then
   321)             call recons_val_cart(fcube, mx, acarty(n),centroid,recons, tmpi, tmpj, value)
   322)             call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   323)           endif
   324)         enddo
   325)       endif
   326)       ! Top/bottom edge, intercept with du/dx = 0
   327)       if (abs(recons(3,tmpi,tmpj)) > 1.0D-12) then
   328)         do n = j, j+1
   329)           mx = - recons(1,tmpi,tmpj) - recons(5,tmpi,tmpj) * &
   330)               (acarty(n) - centroid(2,tmpi,tmpj))
   331) 
   332)           mx = mx / (2.0D0 * recons(3,tmpi,tmpj)) + centroid(1,tmpi,tmpj)
   333)           if ((mx > acartx(i)) .and. (mx < acartx(i+1))) then
   334)             call recons_val_cart(fcube, mx, acarty(n), centroid, recons, tmpi, tmpj, value)
   335)             call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   336)           endif
   337)         enddo
   338)       endif
   339)        ! Left/right edge, intercept with du/dy = 0
   340)       if (abs(recons(4,tmpi,tmpj)) > 1.0D-12) then
   341)         do m = i, i+1
   342)           my = - recons(2,tmpi,tmpj) - recons(5,tmpi,tmpj) * &
   343)                 (acartx(m) - centroid(1,tmpi,tmpj))
   344)           my = my / (2.0D0 * recons(4,tmpi,tmpj)) + centroid(2,tmpi,tmpj)
   345)           if ((my > acarty(j)) .and. (my < acarty(j+1))) then
   346)             call recons_val_cart(fcube, acartx(m), my, centroid, recons, tmpi, tmpj, value)
   347)             call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   348)           endif
   349)         enddo
   350)       endif
   351)       if ((min_phi < -1.0D-12) .OR. (min_phi > 1.0D0 + 1.0D-12)) then
   352)         call haltmp("Fatal Error in monotonic_halo: slope limiter out of range (no in [0,1])!")        
   353)         stop
   354)       endif
   355)       recons(1,tmpi,tmpj) = min_phi * recons(1,tmpi,tmpj)
   356)       recons(2,tmpi,tmpj) = min_phi * recons(2,tmpi,tmpj)
   357)       recons(3,tmpi,tmpj) = min_phi * recons(3,tmpi,tmpj)
   358)       recons(4,tmpi,tmpj) = min_phi * recons(4,tmpi,tmpj)
   359)       recons(5,tmpi,tmpj) = min_phi * recons(5,tmpi,tmpj)
   360)     enddo
   361)   enddo
   362) end subroutine monotonic_halo
   363) !END SUBROUTINE MONOTONIC_HALO--------------------------------------------CE-for FVM!

   365) !SUBROUTINE MONOTONIC_HALOSWAP--------------------------------------------CE-for FVM!
   366) ! AUTHOR: CHRISTOPH ERATH, 30.November 2011                                         !
   367) ! DESCRIPTION: filtering for elements with an edge on the cube boundary             !
   368) !              use this if we have to swap x and y coordinates                      !
   369) !             for certain elements we have to reorder the coefficients and centroids!
   370) !                                                                                   !
   371) ! CALLS: minmax_patch_halo, recons_val_cart, slopelimiter_val                       !

   373) !        fvm  ...  structure incl. tracer values aso                                !
   374) !        acartx ...  x cartesian coordinats of the arrival grid on the cube         !
   375) !        acarty ...  y cartesian coordinats of the arrival grid on the cube         !
   376) !        centroid..  x,y,x^2,y^2,xy                                                 !
   377) !        jx_min, jx_max, jy_min, jy_max                                             !

   381) ! INPUT/OUTPUT:                                                                     !
   382) !        recons ...  array of reconstructed coefficients, which will be scaled by   !
   383) !                    the filter                                                     !

   385) subroutine monotonic_haloswap(fcube,recons,acartx,acarty,centroid,&
   386)                               jx_min,jx_max,jy_min,jy_max,cubeboundary,desc)
   387)   use edge_mod, only : edgedescriptor_t
   388)   
   389)   implicit none
   390)   real (kind=real_kind), dimension(1-nhc:nc+nhc, 1-nhc:nc+nhc),intent(in)    :: fcube  
   391)   real (kind=real_kind),dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe),intent(inout) :: recons  
   392)   real (kind=real_kind), dimension(-nhe:nc+2+nhe), intent(in)   :: acartx, acarty
   393)   real (kind=real_kind), dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe), intent(in)  :: centroid
   394)   integer, intent(in)                                  :: jx_min,jx_max,jy_min,jy_max
   395)   integer, intent(in)                                  :: cubeboundary
   396)   type (edgedescriptor_t),intent(in)                   :: desc
   397)   
   398)   integer (kind=int_kind)          :: i, j, m, n, tmpi, tmpj
   399)   real (kind=real_kind)            :: local_min, local_max, value, phi, min_phi
   400)   real (kind=real_kind)            :: disc, mx, my
   401) 
   402)   do j = jy_min, jy_max-1
   403)     do i = jx_min, jx_max-1
   404)       if(desc%reverse(west) .or. desc%reverse(east)) then
   405)         tmpj=jy_max-1+jy_min-j
   406)       else
   407)         tmpj=j
   408)       endif
   409)       if(desc%reverse(north) .or. desc%reverse(south)) then
   410)         tmpi=jx_max-1+jx_min-i
   411)       else
   412)         tmpi=i
   413)       endif
   414)      
   415)       call minmax_patch_halo(fcube, tmpi, tmpj, cubeboundary,local_min, local_max)
   416)      ! Initialize the limiter
   417)       min_phi = 1.0D0
   418)      ! For the second-order calculation, the minima and maxima will occur
   419)      ! at the corner points of the element
   420)       do m = i, i+1
   421)         do n = j, j+1
   422)         ! Evaluate the function at each corner point
   423)           call recons_val_cart(fcube, acarty(n),acartx(m),centroid,recons,tmpi,tmpj,value)
   424)           call slopelimiter_val(value, fcube(tmpi,tmpj), local_min, local_max, min_phi) 
   425)         enddo
   426)       enddo 
   427)       disc = 4.0D0 * recons(4,tmpi,tmpj) * recons(3,tmpi,tmpj) - recons(5,tmpi,tmpj)**2 
   428)       ! Check if the quadratic is minimized within the element
   429)       if (abs(disc) > 1.0D-12) then
   430)         mx =   recons(5,tmpi,tmpj) * recons(2,tmpi,tmpj)        &
   431)              - 2.0D0 * recons(4,tmpi,tmpj) * recons(1,tmpi,tmpj)
   432)         my =   recons(5,tmpi,tmpj) * recons(1,tmpi,tmpj)        &
   433)              - 2.0D0 * recons(3,tmpi,tmpj) * recons(2,tmpi,tmpj)
   434)         mx = mx / disc + centroid(1,tmpi,tmpj)
   435)         my = my / disc + centroid(2,tmpi,tmpj)
   436) 
   437)         if ((mx - acarty(j)   > -1.0D-12) .and. &
   438)             (mx - acarty(j+1) <  1.0D-12) .and. &
   439)             (my - acartx(i)   > -1.0D-12) .and. &
   440)             (my - acartx(i+1) <  1.0D-12)) then       
   441)           call recons_val_cart(fcube, mx, my, centroid, recons, tmpi, tmpj, value)
   442)           call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   443)         endif
   444)       endif
   445)       ! Check all potential minimizer points along element boundaries
   446)       if (abs(recons(5,tmpi,tmpj)) > 1.0D-12) then
   447)         ! Left/right edge, intercept with du/dx = 0
   448)         do m = j, j+1
   449)           my = - recons(1,tmpi,tmpj) - 2.0D0 * recons(3,tmpi,tmpj) * &
   450)                 (acarty(m) - centroid(1,tmpi,tmpj))
   451)           my = my / recons(5,tmpi,tmpj) + centroid(2,tmpi,tmpj)
   452)           if ((my > acartx(i)) .and. (my < acartx(i+1))) then
   453)             call recons_val_cart(fcube, acarty(m), my, centroid, recons, tmpi, tmpj, value)
   454)             call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   455)           endif
   456)         enddo
   457)         ! Top/bottom edge, intercept with du/dy = 0
   458)         do n = i, i+1
   459)           mx = - recons(2,tmpi,tmpj) - 2.0D0 * recons(4,tmpi,tmpj) * &
   460)                 (acartx(n) - centroid(2,tmpi,tmpj))
   461)           mx = mx / recons(5,tmpi,tmpj) + centroid(1,tmpi,tmpj)
   462)           if ((mx > acarty(j)) .and. (mx < acarty(j+1))) then
   463)             call recons_val_cart(fcube, mx, acartx(n),centroid,recons, tmpi, tmpj, value)
   464)             call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   465)           endif
   466)         enddo
   467)       endif
   468)       ! Top/bottom edge, intercept with du/dx = 0
   469)       if (abs(recons(3,tmpi,tmpj)) > 1.0D-12) then
   470)         do n = i, i+1
   471)           mx = - recons(1,tmpi,tmpj) - recons(5,tmpi,tmpj) * &
   472)                (acartx(n) - centroid(2,tmpi,tmpj))
   473) 
   474)           mx = mx / (2.0D0 * recons(3,tmpi,tmpj)) + centroid(1,tmpi,tmpj)
   475)           if ((mx > acarty(j)) .and. (mx < acarty(j+1))) then
   476)             call recons_val_cart(fcube, mx, acartx(n), centroid, recons, tmpi, tmpj, value)
   477)             call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   478)           endif
   479)         enddo
   480)       endif
   481)       ! Left/right edge, intercept with du/dy = 0
   482)       if (abs(recons(4,tmpi,tmpj)) > 1.0D-12) then
   483)         do m = j, j+1
   484)           my = - recons(2,tmpi,tmpj) - recons(5,tmpi,tmpj) * &
   485)                  (acarty(m) - centroid(1,tmpi,tmpj))
   486)           my = my / (2.0D0 * recons(4,tmpi,tmpj)) + centroid(2,tmpi,tmpj)
   487)           if ((my > acartx(i)) .and. (my < acartx(i+1))) then
   488)             call recons_val_cart(fcube, acarty(m), my, centroid, recons, tmpi, tmpj, value)
   489)             call slopelimiter_val(value, fcube(tmpi,tmpj),local_min, local_max, min_phi)
   490)           endif
   491)         enddo
   492)       endif
   493)       if ((min_phi < -1.0D-12) .or. (min_phi > 1.0D0 + 1.0D-12)) then
   494)         call haltmp("Fatal Error in monotonic_haloswap: slope limiter out of range (not in [0,1])!")                
   495)         stop
   496)       endif
   497)       recons(1,tmpi,tmpj) = min_phi * recons(1,tmpi,tmpj)
   498)       recons(2,tmpi,tmpj) = min_phi * recons(2,tmpi,tmpj)
   499)       recons(3,tmpi,tmpj) = min_phi * recons(3,tmpi,tmpj)
   500)       recons(4,tmpi,tmpj) = min_phi * recons(4,tmpi,tmpj)
   501)       recons(5,tmpi,tmpj) = min_phi * recons(5,tmpi,tmpj)
   502)     enddo
   503)   enddo
   504) end subroutine monotonic_haloswap

   647) !SUBROUTINE RECONS_VAL_CART-----------------------------------------------CE-for FVM!
   648) ! AUTHOR: CHRISTOPH ERATH, 30.November 2011                                         !
   649) ! DESCRIPTION: returns the value from the reconstruction (3rd order Taylor polynom) !
   650) !              at the point (cartx,carty) -> in cube CARTESIAN coordinates          !
   651) !                                                                                   !
   652) ! INPUT: fcube  ...  tracer values incl. the halo zone                              !
   653) !        cartx ...  x cartesian coordinate of the evaluation point                  !
   654) !        carty ...  y cartesian coordinate of the evaluation point                  !
   655) !        centroid..  x,y,x^2,y^2,xy                                                 !
   656) !        recons ...  array of reconstructed coefficients                            !
   657) !        a,b   ...   index of the cell, we do the evaluation                        !
   658) ! OUTPUT: value ... evaluation at a given point                                     !
   659) !-----------------------------------------------------------------------------------!
   660) subroutine recons_val_cart(fcube, cartx, carty, centroid, recons, a, b, value)
   661) 
   662)   implicit none
   663)   real (kind=real_kind), dimension(1-nhc:nc+nhc, 1-nhc:nc+nhc), intent(in) :: fcube
   664)   real (kind=real_kind), intent(in)                                   :: cartx, carty 
   665)   real (kind=real_kind), dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe),intent(in) :: centroid
   666)   real (kind=real_kind), dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe), intent(in):: recons
   667)   integer (kind=int_kind), intent(in)                                      :: a, b
   668)   real    (kind=real_kind), intent(out)                                    :: value
   669) 
   670)   ! Evaluate constant order terms
   671)   value = fcube(a,b) + &
   672)   ! Evaluate linear order terms
   673)           recons(1,a,b) * (cartx - centroid(1,a,b)) + &
   674)           recons(2,a,b) * (carty - centroid(2,a,b)) + &
   675)   ! Evaluate second order terms
   676)           recons(3,a,b) * (centroid(1,a,b)**2 - centroid(3,a,b)) + &
   677)           recons(4,a,b) * (centroid(2,a,b)**2 - centroid(4,a,b)) + &
   678)           recons(5,a,b) * (centroid(1,a,b) * centroid(2,a,b) - centroid(5,a,b)) + &
   679) 
   680)           recons(3,a,b) * (cartx - centroid(1,a,b))**2 + &
   681)           recons(4,a,b) * (carty - centroid(2,a,b))**2 + &
   682)           recons(5,a,b) * (cartx - centroid(1,a,b)) * (carty - centroid(2,a,b))
   683) end subroutine recons_val_cart

   686) !SUBROUTINE SLOPELIMITER_VAL----------------------------------------------CE-for FVM!
   687) ! AUTHOR: CHRISTOPH ERATH, 30.November 2011                                         !
   688) ! DESCRIPTION: returns the value from the reconstruction (3rd order Taylor polynom) !
   689) !              at the point (cartx,carty) -> in cube CARTESIAN coordinates          !
   690) !                                                                                   !
   691) ! INPUT: value  ...  point value (calculated here by recons_val_cart)               !
   692) !        cell_value ...  tracer value (in the cell center) of the cell              !
   693) !        local_min ...  minmal value in the patch                                   !
   694) !        local_max ...  maximal value in the patch                                  !
   695) ! INPUT/OUTPUT: min_phi ... slope limiter, inout because we go through any possible !
   696) !                           extrema on the cell                                     !
   697) !-----------------------------------------------------------------------------------!
   698) subroutine slopelimiter_val(value, cell_value, local_min, local_max, min_phi)
   699) 
   700)   implicit none
   701)   real (kind=real_kind), intent(in)    :: value, cell_value
   702)   real (kind=real_kind), intent(in)    :: local_min, local_max
   703)   real (kind=real_kind), intent(inout) :: min_phi
   704) 
   705)   real (kind=real_kind) :: phi 
   706)   
   707)   phi= 0.0D0
   708)   ! Check against the minimum bound on the reconstruction
   709)   if (value - cell_value > 1.0D-12 * value) then
   710)     phi = (local_max - cell_value) / (value - cell_value)
   711)     if (phi < min_phi) then
   712)       min_phi = phi
   713)     endif
   714)   ! Check against the maximum bound on the reconstruction
   715)   elseif (value - cell_value < -1.0D-12 * value) then
   716)     phi = (local_min - cell_value) / (value - cell_value)    
   717)     if(phi < min_phi) then
   718)       min_phi = phi
   719)     endif
   720)   endif
   721) 
   722) end subroutine slopelimiter_val
