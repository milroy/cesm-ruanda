   154)   subroutine trcdata_init( specifier, filename, filelist, datapath, flds, file, &
   155)                            rmv_file, data_cycle_yr, data_fixed_ymd, data_fixed_tod, data_type )

   157)     use mo_constants,    only : d2r
   158)     use cam_control_mod, only : nsrest
   159)     use dyn_grid,        only : get_dyn_grid_parm
   160)     use string_utils,    only : to_upper
   161)     use horizontal_interpolate, only : xy_interp_init
   162) #if ( defined SPMD )
   163)     use mpishorthand,    only: mpicom, mpir8, mpiint
   164) #endif
   165) 
   166)     implicit none
   167) 
   168)     character(len=*),    intent(in)    :: specifier(:)
   169)     character(len=*),    intent(in)    :: filename
   170)     character(len=*),    intent(in)    :: filelist
   171)     character(len=*),    intent(in)    :: datapath
   172)     type(trfld), dimension(:), pointer :: flds
   173)     type(trfile),        intent(inout) :: file
   174)     logical,             intent(in)    :: rmv_file
   175)     integer,             intent(in)    :: data_cycle_yr
   176)     integer,             intent(in)    :: data_fixed_ymd
   177)     integer,             intent(in)    :: data_fixed_tod
   178)     character(len=*),    intent(in)    :: data_type
   179) 
   180)     integer :: f, mxnflds, astat
   181)     integer :: str_yr, str_mon, str_day
   182)     integer :: lon_dimid, lat_dimid, lev_dimid, tim_dimid, old_dimid
   183)     integer :: dimids(4), did
   184)     type(var_desc_t) :: varid
   185)     integer :: idx
   186)     integer :: ierr
   187)     integer :: errcode
   188)     real(r8) :: start_time, time1, time2
   189)     integer :: i1,i2,j1,j2
   190)     integer :: nvardims, vardimids(4)
   191) 
   192)     character(len=80) :: data_units
   193) 
   194)     call specify_fields( specifier, flds )
   195) 
   196)     file%datatimep=-1.e36_r8
   197)     file%datatimem=-1.e36_r8
   198) 
   199)     mxnflds = 0 
   200)     if (associated(flds)) mxnflds = size( flds )
   201) 
   202)     if (mxnflds < 1) return
   203)     
   204)     file%remove_trc_file = rmv_file
   205)     file%pathname = trim(datapath)
   206)     file%filenames_list = trim(filelist)
   207) 
   208)     file%fill_in_months = .false.
   209)     file%cyclical = .false.  
   210)     file%cyclical_list = .false.  
   211) 
   212) ! does not work when compiled with pathf90
   213) !    select case ( to_upper(data_type) )
   214)     select case ( data_type )
   215)     case( 'FIXED' )
   216)        file%fixed = .true.
   217)     case( 'INTERP_MISSING_MONTHS' )
   218)        file%fill_in_months = .true.
   219)     case( 'CYCLICAL' )
   220)        file%cyclical = .true.
   221)        file%cyc_yr = data_cycle_yr
   222)     case( 'CYCLICAL_LIST' )
   223)        file%cyclical_list = .true.
   224)        file%cyc_yr = data_cycle_yr
   225)     case( 'SERIAL' )
   226)     case default 
   227)        write(iulog,*) 'trcdata_init: invalid data type: '//trim(data_type)//' file: '//trim(filename)
   228)        write(iulog,*) 'trcdata_init: valid data types: SERIAL | CYCLICAL | CYCLICAL_LIST | FIXED | INTERP_MISSING_MONTHS '
   229)        call endrun('trcdata_init: invalid data type: '//trim(data_type)//' file: '//trim(filename))
   230)     endselect
   231) 
   232)     if ( (.not.file%fixed) .and. ((data_fixed_ymd>0._r8) .or.(data_fixed_tod>0._r8))) then
   233)        call endrun('trcdata_init: Cannot specify data_fixed_ymd or data_fixed_tod if data type is not FIXED')
   234)     endif
   235)     if ( (.not.file%cyclical) .and. (data_cycle_yr>0._r8) ) then
   236)        call endrun('trcdata_init: Cannot specify data_cycle_yr if data type is not CYCLICAL')
   237)     endif
   238) 
   239)     if (masterproc) then
   240)        write(iulog,*) 'trcdata_init: data type: '//trim(data_type)//' file: '//trim(filename)
   241)     endif
   242) 
   243)     ! if there is no list of files (len_trim(file%filenames_list)<1) then
   244)     !  -> set curr_filename from namelist rather from restart data
   245)     if ( len_trim(file%curr_filename)<1 .or. len_trim(file%filenames_list)<1 .or. file%fixed ) then ! initial run
   246)        file%curr_filename = trim(filename)
   247) 
   248)        call get_model_time(file)
   249) 
   250)        if ( file%fixed ) then
   251)           str_yr = data_fixed_ymd/10000
   252)           str_mon = (data_fixed_ymd - str_yr*10000)/100
   253)           str_day = data_fixed_ymd - str_yr*10000 - str_mon*100
   254)           call set_time_float_from_date( start_time, str_yr, str_mon, str_day, data_fixed_tod )
   255)           file%offset_time = start_time - file%curr_mod_time
   256)        else
   257)           file%offset_time = 0
   258)        endif
   259)     endif
   260) 
   261)     call set_time_float_from_date( time2, 2, 1, 1, 0 )
   262)     call set_time_float_from_date( time1, 1, 1, 1, 0 )
   263)     file%one_yr = time2-time1
   264) 
   265)     if ( file%cyclical .or. file%cyclical_list) then
   266)        file%cyc_ndx_beg = -1
   267)        file%cyc_ndx_end = -1
   268)        if ( file%cyc_yr /= 0 ) then
   269)           call set_time_float_from_date( time1, file%cyc_yr  , 1, 1, 0 )
   270)           call set_time_float_from_date( time2, file%cyc_yr+1, 1, 1, 0 )
   271)           file%one_yr = time2-time1
   272)        endif
   273) 
   274)        call open_trc_datafile( file%curr_filename, file%pathname, file%curr_fileid, file%curr_data_times, &
   275)             cyc_ndx_beg=file%cyc_ndx_beg, cyc_ndx_end=file%cyc_ndx_end, cyc_yr=file%cyc_yr )
   276)     else
   277)        call open_trc_datafile( file%curr_filename, file%pathname, file%curr_fileid, file%curr_data_times )
   278)        file%curr_data_times = file%curr_data_times - file%offset_time
   279)     endif
   280) 
   281)     call pio_seterrorhandling(File%curr_fileid, PIO_BCAST_ERROR)
   282)     ierr = pio_inq_dimid( file%curr_fileid, 'lon', idx )
   283)     call pio_seterrorhandling(File%curr_fileid, PIO_INTERNAL_ERROR)
   284) 
   285)     file%zonal_ave = (ierr/=PIO_NOERR)
   286) 
   287)     plon = get_dyn_grid_parm('plon')
   288)     plat = get_dyn_grid_parm('plat')
   289) 
   290)     if ( file%zonal_ave ) then
   291) 
   292)        file%nlon = 1
   293) 
   294)     else
   295) 
   296)        call get_dimension( file%curr_fileid, 'lon', file%nlon, dimid=old_dimid, data=file%lons )
   297) 
   298)        file%lons =  file%lons * d2r
   299) 
   300)        lon_dimid = old_dimid
   301) 
   302)     endif
   303) 
   304)     ierr = pio_inq_dimid( file%curr_fileid, 'time', old_dimid)
   305) 
   306)     ! Hack to work with weird netCDF and old gcc or NAG bug.
   307)     tim_dimid = old_dimid
   308) 
   309)     call get_dimension( file%curr_fileid, 'lat', file%nlat, dimid=old_dimid, data=file%lats )
   310)     file%lats =  file%lats * d2r
   311) 
   312)     lat_dimid = old_dimid
   313) 
   314)     allocate( file%ps(file%nlon,file%nlat), stat=astat )
   315)     if( astat /= 0 ) then
   316)        write(iulog,*) 'trcdata_init: file%ps allocation error = ',astat
   317)        call endrun('trcdata_init: failed to allocate x array')
   318)     end if
   319) 
   320)     call pio_seterrorhandling(File%curr_fileid, PIO_BCAST_ERROR)
   321)     ierr = pio_inq_varid( file%curr_fileid, 'PS', file%ps_id )
   322)     file%has_ps = (ierr==PIO_NOERR)
   323)     ierr = pio_inq_dimid( file%curr_fileid, 'altitude', idx )
   324)     file%alt_data = (ierr==PIO_NOERR)
   325) 
   326)     call pio_seterrorhandling(File%curr_fileid, PIO_INTERNAL_ERROR)
   327) 
   328)     if ( file%has_ps) then
   329)        if ( file%zonal_ave ) then
   330)           ierr = pio_inq_vardimid (file%curr_fileid, file%ps_id, dimids(1:2))
   331)           do did = 1,2
   332)              if      ( dimids(did) == lat_dimid ) then
   333)                 file%ps_coords(LATDIM) = did
   334)                 file%ps_order(did) = LATDIM
   335)              else if ( dimids(did) == tim_dimid ) then
   336)                 file%ps_coords(PS_TIMDIM) = did
   337)                 file%ps_order(did) = PS_TIMDIM
   338)              endif
   339)           enddo
   340)        else
   341)           ierr = pio_inq_vardimid (file%curr_fileid, file%ps_id, dimids(1:3))
   342)           do did = 1,3
   343)              if      ( dimids(did) == lon_dimid ) then
   344)                 file%ps_coords(LONDIM) = did
   345)                 file%ps_order(did) = LONDIM
   346)              else if ( dimids(did) == lat_dimid ) then
   347)                 file%ps_coords(LATDIM) = did
   348)                 file%ps_order(did) = LATDIM
   349)              else if ( dimids(did) == tim_dimid ) then
   350)                 file%ps_coords(PS_TIMDIM) = did
   351)                 file%ps_order(did) = PS_TIMDIM
   352)              endif
   353)           enddo
   354)        end if
   355)     endif
   356) 
   357)     if (masterproc) then 
   358)        write(iulog,*) 'trcdata_init: file%has_ps = ' , file%has_ps 
   359)     endif ! masterproc
   360) 
   361)     if (file%alt_data) then
   362)        call get_dimension( file%curr_fileid, 'altitude_int', file%nilev,  data=file%ilevs  )
   363)        call get_dimension( file%curr_fileid, 'altitude',     file%nlev, dimid=old_dimid, data=file%levs  )
   364)     else
   365)        call get_dimension( file%curr_fileid, 'lev', file%nlev, dimid=old_dimid, data=file%levs  )
   366)        if (old_dimid>0) then
   367)           file%levs =  file%levs*100._r8 ! mbar->pascals
   368)        endif
   369)     endif
   370) 
   371)     ! For some bizarre reason, netCDF with older gcc is keeping a pointer to the dimid, and overwriting it later!
   372)     ! Hackish workaround is to make a copy...
   373)     lev_dimid = old_dimid
   374) 
   375)     if (file%has_ps) then
   376) 
   377)        allocate( file%hyam(file%nlev),  file%hybm(file%nlev), stat=astat )
   378)        if( astat /= 0 ) then
   379)           write(iulog,*) 'trcdata_init: file%hyam,file%hybm allocation error = ',astat
   380)           call endrun('trcdata_init: failed to allocate file%hyam and file%hybm arrays')
   381)        end if
   382) 
   383)        allocate( file%hyai(file%nlev+1),  file%hybi(file%nlev+1), stat=astat )
   384)        if( astat /= 0 ) then
   385)           write(iulog,*) 'trcdata_init: file%hyai,file%hybi allocation error = ',astat
   386)           call endrun('trcdata_init: failed to allocate file%hyai and file%hybi arrays')
   387)        end if
   388) 
   389)        call pio_seterrorhandling(File%curr_fileid, PIO_BCAST_ERROR)
   390)        ierr = pio_inq_varid( file%curr_fileid, 'P0', varid)
   391)        call pio_seterrorhandling(File%curr_fileid, PIO_INTERNAL_ERROR)
   392) 
   393)        if ( ierr == PIO_NOERR ) then
   394)           ierr = pio_get_var( file%curr_fileid, varid, file%p0 )
   395)        else
   396)           file%p0 = 100000._r8
   397)        endif
   398)        ierr = pio_inq_varid( file%curr_fileid, 'hyam', varid )
   399)        ierr = pio_get_var( file%curr_fileid, varid, file%hyam )
   400)        ierr = pio_inq_varid( file%curr_fileid, 'hybm', varid )
   401)        ierr = pio_get_var( file%curr_fileid, varid, file%hybm )
   402)        if (file%conserve_column) then
   403)           ierr = pio_inq_varid( file%curr_fileid, 'hyai', varid )
   404)           ierr = pio_get_var( file%curr_fileid, varid, file%hyai )
   405)           ierr = pio_inq_varid( file%curr_fileid, 'hybi', varid )
   406)           ierr = pio_get_var( file%curr_fileid, varid, file%hybi )
   407)        endif
   408) 
   409)        allocate( file         %ps  (pcols,begchunk:endchunk), stat=astat   )
   410)        if( astat/= 0 ) then
   411)           write(iulog,*) 'trcdata_init: failed to allocate file%ps array; error = ',astat
   412)           call endrun
   413)        end if
   414)        allocate( file%ps_in(1)%data(pcols,begchunk:endchunk), stat=astat   )
   415)        if( astat/= 0 ) then
   416)           write(iulog,*) 'trcdata_init: failed to allocate file%ps_in(1)%data array; error = ',astat
   417)           call endrun
   418)        end if
   419)        allocate( file%ps_in(2)%data(pcols,begchunk:endchunk), stat=astat   )
   420)        if( astat/= 0 ) then
   421)           write(iulog,*) 'trcdata_init: failed to allocate file%ps_in(2)%data array; error = ',astat
   422)           call endrun
   423)        end if
   424)        if( file%fill_in_months ) then
   425)           allocate( file%ps_in(3)%data(pcols,begchunk:endchunk), stat=astat   )
   426)           if( astat/= 0 ) then
   427)              write(iulog,*) 'trcdata_init: failed to allocate file%ps_in(3)%data array; error = ',astat
   428)              call endrun
   429)           end if
   430)           allocate( file%ps_in(4)%data(pcols,begchunk:endchunk), stat=astat   )
   431)           if( astat/= 0 ) then
   432)              write(iulog,*) 'trcdata_init: failed to allocate file%ps_in(4)%data array; error = ',astat
   433)              call endrun
   434)           end if
   435)        end if
   436)     endif
   437) 
   438) 
   439)     flds_loop: do f = 1,mxnflds
   440) 
   441)        ! get netcdf variable id for the field
   442)        ierr = pio_inq_varid( file%curr_fileid, flds(f)%srcnam, flds(f)%var_id )
   443) 
   444)        ! determine if the field has a vertical dimension
   445) 
   446)        if (lev_dimid>0) then
   447)           ierr = pio_inquire_variable(  file%curr_fileid, flds(f)%var_id, ndims=nvardims )
   448)           ierr = pio_inquire_variable(  file%curr_fileid, flds(f)%var_id, dimids=vardimids(:nvardims) )
   449)           flds(f)%srf_fld = .not.any(vardimids(:nvardims)==lev_dimid)
   450)        else
   451)           flds(f)%srf_fld = .true.
   452)        endif
   453) 
   454)        ! allocate memory only if not already in pbuf2d

   750)   subroutine check_files( file, fids, itms, times_found)
   751) 
   752)     implicit none
   753) 
   754)     type(trfile),      intent(inout) :: file
   755)     type(file_desc_t), intent(out)   :: fids(2) ! ids of files that contains these recs
   756)     integer, optional, intent(out)   :: itms(2)
   757)     logical, optional, intent(inout) :: times_found
   758) 
   759)     !-----------------------------------------------------------------------
   760)     ! 	... local variables
   761)     !-----------------------------------------------------------------------
   762)     logical            :: list_cycled
   763) 
   764)     list_cycled = .false.
   765) 
   766)    !-----------------------------------------------------------------------
   767)    !        If next time beyond the end of the time list,
   768)    !        then increment the filename and move on to the next file
   769)    !-----------------------------------------------------------------------
   770)     if ((file%next_mod_time > file%curr_data_times(size(file%curr_data_times))).or.file%cyclical_list) then
   771)        if (file%cyclical_list) then
   772)           if ( associated(file%next_data_times) ) then
   773)               if ((file%curr_mod_time > file%datatimep)) then
   774) 
   775)                call advance_file(file)
   776)      
   777)             endif
   778)          endif
   779) 
   780)        endif
   781)        if ( .not. associated(file%next_data_times) ) then
   782)           ! open next file if not already opened...
   783)           if (file%cyclical_list) then
   784)               file%next_filename = incr_filename( file%curr_filename, filenames_list=file%filenames_list, datapath=file%pathname ,&
   785)                                                   cyclical_list=file%cyclical_list, list_cycled=list_cycled)
   786)           else
   787)               file%next_filename = incr_filename( file%curr_filename, filenames_list=file%filenames_list, datapath=file%pathname)
   788)           endif
   789)           call open_trc_datafile( file%next_filename, file%pathname, file%next_fileid, file%next_data_times )
   790)           file%next_data_times = file%next_data_times - file%offset_time
   791)        endif
   792)     endif
   793)     
   794)     !-----------------------------------------------------------------------
   795)     !        If using next_data_times and the current is greater than or equal to the next, then
   796)     !        close the current file, and set up for next file.
   797)     !-----------------------------------------------------------------------
   798)     if ( associated(file%next_data_times) ) then
   799)        if (file%cyclical_list .and. list_cycled) then    ! special case - list cycled 
   800) 
   801)           file%datatimem = file%curr_data_times(size(file%curr_data_times)) 
   802)           itms(1)=size(file%curr_data_times)
   803)           fids(1)=file%curr_fileid
   804) 
   805)           file%datatimep = file%next_data_times(1)
   806)           itms(2)=1
   807)           fids(2) = file%next_fileid
   808) 
   809)           times_found = .true.
   810) 
   811)        else if (file%curr_mod_time >= file%next_data_times(1)) then
   812) 
   813)           call advance_file(file)
   814) 
   815)        endif
   816)     endif
   817) 
   818)   end subroutine check_files

   820) !-----------------------------------------------------------------------
   821) !-----------------------------------------------------------------------
   822)   function incr_filename( filename, filenames_list, datapath, cyclical_list, list_cycled, abort )
   823) 
   824)     !-----------------------------------------------------------------------
   825)     ! 	... Increment or decrement a date string withing a filename
   826)     !           the filename date section is assumed to be of the form
   827)     !           yyyy-dd-mm
   828)     !-----------------------------------------------------------------------
   829) 
   830)     use string_utils, only : incstr
   831)     use shr_file_mod, only : shr_file_getunit, shr_file_freeunit
   832) 
   833)     implicit none
   834) 
   835)     character(len=*),           intent(in)    :: filename ! present dynamical dataset filename
   836)     character(len=*), optional, intent(in)    :: filenames_list 
   837)     character(len=*), optional, intent(in)    :: datapath
   838)     logical         , optional, intent(in)    :: cyclical_list  ! If true, allow list to cycle
   839)     logical         , optional, intent(out)   :: list_cycled
   840)     logical         , optional, intent(in)    :: abort
   841) 
   842)     character(len=shr_kind_cl)                :: incr_filename         ! next filename in the sequence
   843) 
   844)     ! set new next_filename ...
   845) 
   846)     !-----------------------------------------------------------------------
   847)     !	... local variables
   848)     !-----------------------------------------------------------------------
   849)     integer :: pos, pos1, istat
   850)     character(len=shr_kind_cl) :: fn_new, line, filepath
   851)     character(len=6)   :: seconds
   852)     character(len=5)   :: num
   853)     integer :: ios,unitnumber
   854)     logical :: abort_run
   855)  
   856)     if (present(abort)) then 
   857)        abort_run = abort
   858)     else
   859)        abort_run = .true.
   860)     endif
   861) 
   862)     if (present(list_cycled)) list_cycled = .false.
   863) 
   864)     if (( .not. present(filenames_list)) .or.(len_trim(filenames_list) == 0)) then
   865)        !-----------------------------------------------------------------------
   866)        !	... ccm type filename
   867)        !-----------------------------------------------------------------------
   868)        pos = len_trim( filename )
   869)        fn_new = filename(:pos)
   870)        if ( masterproc ) write(iulog,*) 'incr_flnm: old filename = ',trim(fn_new)
   871)        if( fn_new(pos-2:) == '.nc' ) then
   872)           pos = pos - 3
   873)        end if
   874)        istat = incstr( fn_new(:pos), 1 )
   875)        if( istat /= 0 ) then
   876)           write(iulog,*) 'incr_flnm: incstr returned ', istat
   877)           write(iulog,*) '           while trying to decrement ',trim( fn_new )
   878)           call endrun
   879)        end if
   880) 
   881)     else
   882) 
   883)        !-------------------------------------------------------------------
   884)        !  ... open filenames_list
   885)        !-------------------------------------------------------------------
   886)        if ( masterproc ) write(iulog,*) 'incr_flnm: old filename = ',trim(filename)
   887)        if ( masterproc ) write(iulog,*) 'incr_flnm: open filenames_list : ',trim(filenames_list)
   888)        unitnumber = shr_file_getUnit()
   889)        if ( present(datapath) ) then
   890)          filepath = trim(datapath) //'/'// trim(filenames_list)
   891)        else
   892)          filepath = trim(filenames_list)
   893)        endif
   894) 
   895)        open( unit=unitnumber, file=filepath, iostat=ios, status="OLD")
   896)        if (ios /= 0) then
   897)           call endrun('not able to open filenames_list file: '//trim(filepath))
   898)        endif
   899) 
   900)        !-------------------------------------------------------------------
   901)        !  ...  read file names
   902)        !-------------------------------------------------------------------
   903)        read( unit=unitnumber, fmt='(A)', iostat=ios ) line 
   904)        if (ios /= 0) then
   905)           if (abort_run) then
   906)              call endrun('not able to increment file name from filenames_list file: '//trim(filenames_list))
   907)           else 
   908)              fn_new = 'NOT_FOUND'
   909)              incr_filename = trim(fn_new)
   910)              return
   911)           endif
   912)        endif
   913) 
   914)        !-------------------------------------------------------------------
   915)        !      If current filename is '', then initialize with the first filename read in
   916)        !      and skip this section.
   917)        !-------------------------------------------------------------------
   918)        if (filename /= '') then 
   919) 
   920)           !-------------------------------------------------------------------
   921)           !       otherwise read until find current filename
   922)           !-------------------------------------------------------------------
   923)           do while( trim(line) /= trim(filename) )
   924)              read( unit=unitnumber, fmt='(A)', iostat=ios ) line 
   925)              if (ios /= 0) then
   926)                 if (abort_run) then
   927)                    call endrun('not able to increment file name from filenames_list file: '//trim(filenames_list))
   928)                 else 
   929)                    fn_new = 'NOT_FOUND'
   930)                    incr_filename = trim(fn_new)
   931)                    return
   932)                 endif
   933)              endif
   934)           enddo
   935)    
   936)           !-------------------------------------------------------------------
   937)           !      Read next filename
   938)           !-------------------------------------------------------------------
   939)           read( unit=unitnumber, fmt='(A)', iostat=ios ) line 
   940) 
   941)           !---------------------------------------------------------------------------------
   942)           !       If cyclical_list, then an end of file is not an error, but rather 
   943)           !       a signal to rewind and start over
   944)           !---------------------------------------------------------------------------------
   945) 
   946)           if (ios /= 0) then
   947)              if (present(cyclical_list)) then
   948)                 if (cyclical_list) then
   949)                    list_cycled=.true.
   950)                    rewind(unitnumber)
   951)                    read( unit=unitnumber, fmt='(A)', iostat=ios ) line 
   952)                      ! Error here should never happen, but check just in case
   953)                    if (ios /= 0) then
   954)                       call endrun('not able to increment file name from filenames_list file: '//trim(filenames_list))
   955)                    endif
   956)                 else
   957)                    call endrun('not able to increment file name from filenames_list file: '//trim(filenames_list))
   958)                 endif
   959)              else
   960)                 if (abort_run) then
   961)                    call endrun('not able to increment file name from filenames_list file: '//trim(filenames_list))
   962)                 else 
   963)                    fn_new = 'NOT_FOUND'

  1564)

  1566)     real(r8) :: deltat
  1567)     integer :: f,nflds,c,ncol, i,k
  1568)     real(r8) :: ps(pcols)
  1569)     real(r8) :: datain(pcols,file%nlev)
  1570)     real(r8) :: pin(pcols,file%nlev)

  1576)     real(r8) :: data_col(pver)

  1578)     nflds = size(flds)
  1579) 
  1580)     if ( file%interp_recs == 4 ) then
  1581)        deltat = file%datatimes(3) - file%datatimes(1)
  1582)        fact1 = (file%datatimes(3) - file%datatimem)/deltat
  1583)        fact2 = 1._r8-fact1

  1586)           ncol = state(c)%ncol

  1596)        fact1 = (file%datatimes(4) - file%datatimep)/deltat
  1597)        fact2 = 1._r8-fact1

  1600)        do c = begchunk,endchunk
  1601)           ncol = state(c)%ncol
  1602)           if ( file%has_ps ) then

  1607)           enddo
  1608)        enddo
  1609) 
  1610)     endif
  1611)     !-------------------------------------------------------------------------
  1612)     !       If file%interp_recs=1 then no time interpolation -- set
  1613)     !       fact1=1 and fact2=0 and will just use first value unmodified

  1631)        else

  1641)

  1665)              do i = 1,ncol

  1670)                 call rebin( file%nlev, pver, file%ilevs, model_z, datain(i,:), data_col(:) )
  1671)                 data_out(i,:) = data_col(pver:1:-1)
  1672)              enddo
  1673) 
  1674)           else
  1675) 
  1676)              if ( file%nlev>1 ) then
  1677)                 if ( file%has_ps ) then
  1678)                    if (fact2 == 0) then  ! This needed as %data is not set if fact2=0 (and lahey compiler core dumps)

  1680)                    else
  1681)                       ps(:ncol) = fact1*file%ps_in(nm)%data(:ncol,c) + fact2*file%ps_in(np)%data(:ncol,c)

  1683)                    do i = 1,ncol

  1687)                    enddo
  1688)                 else

  1690)                       pin(:,k) = file%levs(k)

  1693)              endif
  1694)

  1696)                 do i = 1,ncol
  1697)                    if (fact2 == 0) then  ! This needed as %data is not set if fact2=0 (and lahey compiler core dumps)

  1699)                            fact1*flds(f)%input(nm)%data(i,1,c)
  1700)                    else
  1701)                       data_out(i,1) = &
  1702)                            fact1*flds(f)%input(nm)%data(i,1,c) + fact2*flds(f)%input(np)%data(i,1,c)

  1704)                 enddo

  1706)                 if (fact2 == 0) then  ! This needed as %data is not set if fact2=0 (and lahey compiler core dumps)
  1707)                    datain(:ncol,:) = fact1*flds(f)%input(nm)%data(:ncol,:,c)

  1711)                 if ( file%top_bndry ) then
  1712)                    call vert_interp_ub(ncol, file%nlev, file%levs,  datain(:ncol,:), data_out(:ncol,:) )
  1713)                 else if(file%conserve_column) then

  1715)                         datain, data_out(:,:), &
  1716)                         file%p0,ps,file%hyai,file%hybi)
  1717)                 else
  1718)                    call vert_interp(ncol, file%nlev, pin, state(c)%pmid, datain, data_out(:,:) )
  1719)                 endif
  1720)              endif
  1721) 
  1722)           endif
  1723)        enddo
  1724) 
  1725)     enddo fld_loop
