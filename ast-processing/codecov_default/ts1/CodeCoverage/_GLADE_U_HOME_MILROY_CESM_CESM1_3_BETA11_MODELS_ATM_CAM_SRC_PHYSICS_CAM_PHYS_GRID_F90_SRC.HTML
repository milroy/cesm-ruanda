   311)   subroutine phys_grid_init( )
   312)     !----------------------------------------------------------------------- 
   313)     ! 
   314)     ! Purpose: Physics mapping initialization routine:  
   315)     ! 
   316)     ! Method: 
   317)     ! 
   318)     ! Author: John Drake and Patrick Worley
   319)     ! 
   320)     !-----------------------------------------------------------------------
   321)     use pmgrid, only: plev
   322)     use dyn_grid, only: get_block_bounds_d, &
   323)          get_block_gcol_d, get_block_gcol_cnt_d, &
   324)          get_block_levels_d, get_block_lvl_cnt_d, &
   325)          get_block_owner_d, &
   326)          get_gcol_block_d, get_gcol_block_cnt_d, &
   327)          get_horiz_grid_dim_d, get_horiz_grid_d
   328)        use spmd_utils, only: pair, ceil2
   329)     !
   330)     !------------------------------Arguments--------------------------------
   331)     !
   332)     !
   333)     !---------------------------Local workspace-----------------------------
   334)     !
   335)     integer :: i, j, jb, k, p             ! loop indices
   336)     integer :: pre_i                      ! earlier index in loop iteration
   337)     integer :: clat_p_dex, clon_p_dex     ! indices into unique lat. and lon. arrays
   338)     integer :: maxblksiz                  ! maximum number of columns in a dynamics block
   339)     integer :: beg_dex, end_dex           ! index range
   340)     integer :: cid, lcid                  ! global and local chunk ids
   341)     integer :: max_ncols                  ! upper bound on number of columns in a block
   342)     integer :: ncols                      ! number of columns in current chunk
   343)     integer :: curgcol, curgcol_d         ! current global column index
   344)     integer :: firstblock, lastblock      ! global block indices
   345)     integer :: blksiz                     ! current block size
   346)     integer :: glbcnt, curcnt             ! running grid point counts
   347)     integer :: curp                       ! current process id
   348)     integer :: block_cnt                  ! number of blocks containing data
   349)     ! for a given vertical column
   350)     integer :: numlvl                     ! number of vertical levels in block 
   351)     ! column
   352)     integer :: levels(plev+1)             ! vertical level indices
   353)     integer :: owner_d                    ! process owning given block column
   354)     integer :: owner_p                    ! process owning given chunk column
   355)     integer :: blockids(plev+1)           ! block indices
   356)     integer :: bcids(plev+1)              ! block column indices
   357)     integer :: glon, glat                 ! global (lon,lat) indices
   358)     integer :: ntmp1, ntmp2               ! work variables
   359) 
   360)     logical :: clon_wrap                  ! flag used in initializing lat_p, lon_p
   361) 
   362)     ! column surface area (from dynamics)
   363)     real(r8), dimension(:), allocatable :: area_d 
   364) 
   365)     ! column integration weight (from dynamics)
   366)     real(r8), dimension(:), allocatable :: wght_d 
   367) 
   368)     ! chunk global ordering
   369)     integer, dimension(:), allocatable :: pchunkid                   
   370) 
   371)     ! permutation array used in physics column sorting;
   372)     ! reused later as work space in (lbal_opt == -1) logic
   373)     integer, dimension(:), allocatable :: cdex
   374) 
   375)     ! latitudes and longitudes and column area for dynamics columns
   376)     real(r8), dimension(:), allocatable :: clat_d
   377)     real(r8), dimension(:), allocatable :: clon_d
   378)     real(r8) :: clat_p_tmp
   379)     real(r8) :: clon_p_tmp
   380) 
   381)     integer lons(2), lats(2)
   382) 
   383)     call t_adj_detailf(-2)
   384)     call t_startf("phys_grid_init")
   385) 
   386)     !-----------------------------------------------------------------------
   387)     !
   388)     ! Initialize physics grid, using dynamics grid
   389)     ! a) column coordinates
   390) 
   391)     call get_horiz_grid_dim_d(hdim1_d,hdim2_d)
   392)     ngcols = hdim1_d*hdim2_d
   393)     allocate( clat_d(1:ngcols) )
   394)     allocate( clon_d(1:ngcols) )
   395)     allocate( cdex(1:ngcols) )
   396)     clat_d = 100000.0_r8
   397)     clon_d = 100000.0_r8
   398)     call get_horiz_grid_d(ngcols, clat_d_out=clat_d, clon_d_out=clon_d)
   399) 
   400)     ! count number of "real" column indices
   401)     ngcols_p = 0
   402)     do i=1,ngcols
   403)        if (clon_d(i) < 100000.0_r8) then
   404)           ngcols_p = ngcols_p + 1
   405)        endif
   406)     enddo
   407) 
   408)     ! sort over longitude and identify unique longitude coordinates
   409)     call IndexSet(ngcols,cdex)
   410)     call IndexSort(ngcols,cdex,clon_d,descend=.false.)
   411)     clon_p_tmp = clon_d(cdex(1))
   412)     clon_p_tot = 1
   413) 
   414)     do i=2,ngcols_p
   415)        if (clon_d(cdex(i)) > clon_p_tmp) then
   416)           clon_p_tot = clon_p_tot + 1
   417)           clon_p_tmp = clon_d(cdex(i))
   418)        endif
   419)     enddo
   420) 
   421)     allocate( clon_p(1:clon_p_tot) )
   422)     allocate( clon_p_cnt(1:clon_p_tot) )
   423) 
   424)     pre_i = 1
   425)     clon_p_tot = 1
   426)     clon_p(1) = clon_d(cdex(1))
   427)     do i=2,ngcols_p
   428)        if (clon_d(cdex(i)) > clon_p(clon_p_tot)) then
   429)           clon_p_cnt(clon_p_tot) = i-pre_i
   430)           pre_i = i
   431)           clon_p_tot = clon_p_tot + 1
   432)           clon_p(clon_p_tot) = clon_d(cdex(i))
   433)        endif
   434)     enddo
   435)     clon_p_cnt(clon_p_tot) = (ngcols_p+1)-pre_i
   436) 
   437)     ! sort over latitude and identify unique latitude coordinates
   438)     call IndexSet(ngcols,cdex)
   439)     call IndexSort(ngcols,cdex,clat_d,descend=.false.)
   440)     clat_p_tmp = clat_d(cdex(1))
   441)     clat_p_tot = 1
   442)     do i=2,ngcols_p
   443)        if (clat_d(cdex(i)) > clat_p_tmp) then
   444)           clat_p_tot = clat_p_tot + 1
   445)           clat_p_tmp = clat_d(cdex(i))
   446)        endif
   447)     enddo
   448) 
   449)     allocate( clat_p(1:clat_p_tot) )
   450)     allocate( clat_p_cnt(1:clat_p_tot) )
   451)     allocate( clat_p_idx(1:clat_p_tot) )
   452) 
   453)     pre_i = 1
   454)     clat_p_tot = 1
   455)     clat_p(1) = clat_d(cdex(1))
   456)     do i=2,ngcols_p
   457)        if (clat_d(cdex(i)) > clat_p(clat_p_tot)) then
   458)           clat_p_cnt(clat_p_tot) = i-pre_i
   459)           pre_i = i
   460)           clat_p_tot = clat_p_tot + 1
   461)           clat_p(clat_p_tot) = clat_d(cdex(i))
   462)        endif
   463)     enddo
   464)     clat_p_cnt(clat_p_tot) = (ngcols_p+1)-pre_i
   465) 
   466)     clat_p_idx(1) = 1
   467)     do j=2,clat_p_tot
   468)        clat_p_idx(j) = clat_p_idx(j-1) + clat_p_cnt(j-1)
   469)     enddo
   470) 
   471)     ! sort by longitude within latitudes
   472)     end_dex = 0
   473)     do j=1,clat_p_tot
   474)        beg_dex = end_dex + 1
   475)        end_dex = end_dex + clat_p_cnt(j)
   476)        call IndexSort(cdex(beg_dex:end_dex),clon_d,descend=.false.)
   477)     enddo
   478) 
   479)     ! Early clean-up, to minimize memory high water mark
   480)     ! (not executing find_partner or find_twin)
   481)     if (((twin_alg .ne. 1) .and. (lbal_opt .ne. 3)) .or. &
   482)         (lbal_opt .eq. -1)) deallocate( clat_p_cnt)
   483) 
   484)     ! save "longitude within latitude" column ordering
   485)     ! and determine mapping from unsorted global column index to 
   486)     ! unique latitude/longitude indices
   487)     allocate( lat_p(1:ngcols) )
   488)     allocate( lon_p(1:ngcols) )
   489)     allocate( dyn_to_latlon_gcol_map(1:ngcols) )
   490)     if (lbal_opt .ne. -1) allocate( latlon_to_dyn_gcol_map(1:ngcols_p) )
   491) 
   492)     clat_p_dex = 1
   493)     lat_p = -1
   494)     dyn_to_latlon_gcol_map = -1
   495)     do i=1,ngcols_p
   496)        if (lbal_opt .ne. -1) latlon_to_dyn_gcol_map(i) = cdex(i)
   497)        dyn_to_latlon_gcol_map(cdex(i)) = i
   498) 
   499)        do while ((clat_p(clat_p_dex) < clat_d(cdex(i))) .and. &
   500)                  (clat_p_dex < clat_p_tot))
   501)           clat_p_dex = clat_p_dex + 1
   502)        enddo
   503)        lat_p(cdex(i)) = clat_p_dex
   504)     enddo
   505) 
   506)     ! sort by latitude within longitudes
   507)     call IndexSet(ngcols,cdex)
   508)     call IndexSort(ngcols,cdex,clon_d,descend=.false.)
   509)     end_dex = 0
   510)     do i=1,clon_p_tot
   511)        beg_dex = end_dex + 1
   512)        end_dex = end_dex + clon_p_cnt(i)
   513)        call IndexSort(cdex(beg_dex:end_dex),clat_d,descend=.false.)
   514)     enddo
   515) 
   516)     ! Early clean-up, to minimize memory high water mark
   517)     ! (not executing find_twin)
   518)     if ((twin_alg .ne. 1) .or. (lbal_opt .eq. -1)) deallocate( clon_p_cnt )
   519) 
   520)     ! save "latitude within longitude" column ordering
   521)     ! (only need in find_twin)
   522)     if ((twin_alg .eq. 1) .and. (lbal_opt .ne. -1)) &
   523)        allocate( lonlat_to_dyn_gcol_map(1:ngcols_p) )
   524) 
   525)     clon_p_dex = 1
   526)     lon_p = -1
   527)     do i=1,ngcols_p
   528)        if ((twin_alg .eq. 1) .and. (lbal_opt .ne. -1)) &
   529)          lonlat_to_dyn_gcol_map(i) = cdex(i)
   530)        do while ((clon_p(clon_p_dex) < clon_d(cdex(i))) .and. &
   531)                  (clon_p_dex < clon_p_tot))
   532)           clon_p_dex = clon_p_dex + 1
   533)        enddo
   534)        lon_p(cdex(i)) = clon_p_dex
   535)     enddo
   536) 
   537)     ! Clean-up
   538)     deallocate( clat_d )
   539)     deallocate( clon_d )
   540)     deallocate( cdex )
   541) 
   542)     !
   543)     ! Determine block index bounds
   544)     !
   545)     call get_block_bounds_d(firstblock,lastblock)
   546) 
   547)     ! Allocate storage to save number of chunks and columns assigned to each
   548)     ! process during chunk creation and assignment
   549)     !
   550)     allocate( npchunks(0:npes-1) )
   551)     allocate( gs_col_num(0:npes-1) )
   552)     npchunks(:) = 0
   553)     gs_col_num(:) = 0
   554) 
   555)     !
   556)     ! Option -1: each dynamics block is a single chunk
   557)     !            
   558)     if (lbal_opt == -1) then
   559)        !
   560)        ! Check that pcols >= maxblksiz
   561)        !
   562)        maxblksiz = 0
   563)        do jb=firstblock,lastblock
   564)           maxblksiz = max(maxblksiz,get_block_gcol_cnt_d(jb))
   565)        enddo
   566)        if (pcols < maxblksiz) then
   567) 	  write(iulog,*) 'pcols = ',pcols, ' maxblksiz=',maxblksiz
   568)           call endrun ('PHYS_GRID_INIT error: phys_loadbalance -1 specified but PCOLS < MAXBLKSIZ')
   569)        endif
   570) 
   571)        !
   572)        ! Determine total number of chunks
   573)        !
   574)        nchunks = (lastblock-firstblock+1)
   575) 
   576)        !
   577)        ! Set max virtual SMP node size
   578)        !
   579)        max_nproc_smpx = 1
   580) 
   581)        !
   582)        ! Allocate and initialize chunks data structure
   583)        !
   584)        allocate( cdex(1:maxblksiz) )
   585)        allocate( chunks(1:nchunks) )
   586) 
   587)        do cid=1,nchunks
   588)           ! get number of global column indices in block
   589)           max_ncols = get_block_gcol_cnt_d(cid+firstblock-1)
   590)           ! fill cdex array with global indices from current block
   591)           call get_block_gcol_d(cid+firstblock-1,max_ncols,cdex)
   592) 
   593)           ncols = 0
   594)           do i=1,max_ncols
   595)              ! check whether global index is for a column that dynamics
   596)              ! intends to pass to the physics
   597)              curgcol_d = cdex(i)
   598)              if (dyn_to_latlon_gcol_map(curgcol_d) .ne. -1) then
   599)                 ! yes - then save the information
   600)                 ncols = ncols + 1
   601)                 chunks(cid)%gcol(ncols) = curgcol_d
   602)                 chunks(cid)%lat(ncols) = lat_p(curgcol_d)
   603)                 chunks(cid)%lon(ncols) = lon_p(curgcol_d)
   604)              endif
   605)           enddo
   606)           chunks(cid)%ncols = ncols
   607)        enddo
   608) 
   609)        ! Clean-up
   610)        deallocate( cdex )
   611)        deallocate( lat_p )
   612)        deallocate( lon_p )
   613) 
   614)        !
   615)        ! Specify parallel decomposition 
   616)        !
   617)        do cid=1,nchunks
   618) #if (defined SPMD)
   619)           p = get_block_owner_d(cid+firstblock-1)
   620) #else
   621)           p = 0
   622) #endif
   623)           chunks(cid)%owner = p
   624)           npchunks(p)       = npchunks(p) + 1
   625)           gs_col_num(p)     = gs_col_num(p) + chunks(cid)%ncols
   626)        enddo
   627)        !
   628)        ! Set flag indicating columns in physics and dynamics 
   629)        ! decompositions reside on the same processes
   630)        !
   631)        local_dp_map = .true. 
   632)        !
   633)     else
   634)        !
   635)        ! Option == 0: split local blocks into chunks,
   636)        !               while attempting to create load-balanced chunks.
   637)        !               Does not work with vertically decomposed blocks.
   638)        !               (default)
   639)        ! Option == 1: split SMP-local blocks into chunks,
   640)        !               while attempting to create load-balanced chunks.
   641)        !               Does not work with vertically decomposed blocks.
   642)        ! Option == 2: load balance chunks with respect to diurnal and
   643)        !               seaonsal cycles and wth respect to latitude, 
   644)        !               and assign chunks to processes
   645)        !               in a way that attempts to minimize communication costs
   646)        ! Option == 3: divide processes into pairs and split 
   647)        !               blocks assigned to these pairs into 
   648)        !               chunks, attempting to create load-balanced chunks.
   649)        !               The process pairs are chosen to maximize load balancing
   650)        !               opportunities.
   651)        !               Does not work with vertically decomposed blocks.
   652)        ! Option == 4: concatenate local blocks, then
   653)        !               divide into chunks.
   654)        !               Does not work with vertically decomposed blocks.
   655)        ! Option == 5: split indiviudal blocks into chunks,
   656)        !               assigning columns using block ordering
   657)        !
   658)        !
   659)        ! Allocate and initialize chunks data structure, then
   660)        ! assign chunks to processes.
   661)        !
   662)        call create_chunks(lbal_opt, chunks_per_thread)
   663) 
   664)        ! Early clean-up, to minimize memory high water mark
   665)        deallocate( lat_p )
   666)        deallocate( lon_p )
   667)        deallocate( latlon_to_dyn_gcol_map )
   668)        if  (twin_alg .eq. 1) deallocate( lonlat_to_dyn_gcol_map )
   669)        if  (twin_alg .eq. 1) deallocate( clon_p_cnt )
   670)        if ((twin_alg .eq. 1) .or. (lbal_opt .eq. 3)) deallocate( clat_p_cnt )
   671) 
   672)        !
   673)        ! Determine whether dynamics and physics decompositions
   674)        ! are colocated, not requiring any interprocess communication
   675)        ! in the coupling.
   676)        local_dp_map = .true.   
   677)        do cid=1,nchunks
   678)           do i=1,chunks(cid)%ncols
   679)              curgcol_d = chunks(cid)%gcol(i)
   680)              block_cnt = get_gcol_block_cnt_d(curgcol_d)
   681)              call get_gcol_block_d(curgcol_d,block_cnt,blockids,bcids)
   682)              do jb=1,block_cnt
   683)                 owner_d = get_block_owner_d(blockids(jb)) 
   684)                 if (owner_d .ne. chunks(cid)%owner) then
   685)                    local_dp_map = .false.   
   686)                 endif
   687)              enddo
   688)           enddo
   689)        enddo
   690)     endif
   691)     !
   692)     ! Allocate and initialize data structures for gather/scatter
   693)     !  
   694)     allocate( pgcols(1:ngcols_p) )
   695)     allocate( gs_col_offset(0:npes) )
   696)     allocate( pchunkid(0:npes) )
   697) 
   698)     ! Initialize pchunkid and gs_col_offset by summing 
   699)     ! number of chunks and columns per process, respectively
   700)     pchunkid(0) = 0
   701)     gs_col_offset(0) = 0
   702)     do p=1,npes-1
   703)        pchunkid(p)      = pchunkid(p-1)      + npchunks(p-1)
   704)        gs_col_offset(p) = gs_col_offset(p-1) + gs_col_num(p-1)
   705)     enddo
   706)     
   707)     ! Determine local ordering via "process id" bin sort
   708)     do cid=1,nchunks
   709)        p = chunks(cid)%owner
   710)        pchunkid(p) = pchunkid(p) + 1
   711) 
   712)        chunks(cid)%lcid = pchunkid(p) + lastblock
   713) 
   714)        curgcol = gs_col_offset(p)
   715)        do i=1,chunks(cid)%ncols
   716)           curgcol = curgcol + 1
   717)           pgcols(curgcol)%chunk = cid
   718)           pgcols(curgcol)%ccol = i
   719)        enddo
   720)        gs_col_offset(p) = curgcol
   721)     enddo
   722) 
   723)     ! Reinitialize pchunkid and gs_col_offset (for real)
   724)     pchunkid(0) = 1
   725)     gs_col_offset(0) = 1
   726)     do p=1,npes-1
   727)        pchunkid(p)      = pchunkid(p-1)      + npchunks(p-1)
   728)        gs_col_offset(p) = gs_col_offset(p-1) + gs_col_num(p-1)
   729)     enddo
   730)     pchunkid(npes)      = pchunkid(npes-1)      + npchunks(npes-1)
   731)     gs_col_offset(npes) = gs_col_offset(npes-1) + gs_col_num(npes-1)
   732) 
   733)     ! Save local information
   734)     ! (Local chunk index range chosen so that it does not overlap 
   735)     !  {begblock,...,endblock})
   736)     ! 
   737)     nlcols   = gs_col_num(iam)
   738)     nlchunks = npchunks(iam)
   739)     begchunk = pchunkid(iam)   + lastblock
   740)     endchunk = pchunkid(iam+1) + lastblock - 1
   741)     !
   742)     allocate( lchunks(begchunk:endchunk) )
   743)     do cid=1,nchunks
   744)        if (chunks(cid)%owner == iam) then
   745)           lcid = chunks(cid)%lcid
   746)           lchunks(lcid)%ncols = chunks(cid)%ncols
   747)           lchunks(lcid)%cid   = cid
   748)           do i=1,chunks(cid)%ncols
   749)              lchunks(lcid)%gcol(i) = chunks(cid)%gcol(i)
   750)           enddo
   751)        endif
   752)     enddo
   753) 
   754)     deallocate( pchunkid )
   755)     deallocate( npchunks )
   756)     !
   757)     !-----------------------------------------------------------------------
   758)     !
   759)     ! Initialize physics grid, using dynamics grid
   760)     ! b) column area and integration weight
   761) 
   762)     allocate( area_d(1:ngcols) )
   763)     allocate( wght_d(1:ngcols) )
   764)     area_d = 0.0_r8
   765)     wght_d = 0.0_r8
   766) 
   767)     call get_horiz_grid_d(ngcols, area_d_out=area_d, wght_d_out=wght_d)
   768) 
   769) 
   770)     if ( abs(sum(area_d) - 4.0_r8*pi) > 1.e-10_r8 ) then
   771)        write(iulog,*) ' ERROR: sum of areas on globe does not equal 4*pi'
   772)        write(iulog,*) ' sum of areas = ', sum(area_d), sum(area_d)-4.0_r8*pi
   773)        call endrun('phys_grid')
   774)     end if
   775) 
   776)     if ( abs(sum(wght_d) - 4.0_r8*pi) > 1.e-10_r8 ) then
   777)        write(iulog,*) ' ERROR: sum of integration weights on globe does not equal 4*pi'
   778)        write(iulog,*) ' sum of weights = ', sum(wght_d), sum(wght_d)-4.0_r8*pi
   779)        call endrun('phys_grid')
   780)     end if
   781) 
   782)     do lcid=begchunk,endchunk
   783)        do i=1,lchunks(lcid)%ncols
   784)           lchunks(lcid)%area(i) = area_d(lchunks(lcid)%gcol(i))
   785)           lchunks(lcid)%wght(i) = wght_d(lchunks(lcid)%gcol(i))
   786)        enddo
   787)     enddo
   788) 
   789)     deallocate( area_d )
   790)     deallocate( wght_d )
   791) 
   792)     if (.not. local_dp_map) then
   793)        !
   794)        ! allocate and initialize data structures for transposes
   795)        !  
   796)        allocate( btofc_blk_num(0:npes-1) )
   797)        btofc_blk_num = 0
   798)        allocate( btofc_blk_offset(firstblock:lastblock) )
   799)        do jb = firstblock,lastblock
   800)           nullify( btofc_blk_offset(jb)%pter )
   801)        enddo
   802)        !
   803)        glbcnt = 0
   804)        curcnt = 0
   805)        curp = 0
   806)        do curgcol=1,ngcols_p
   807)           cid = pgcols(curgcol)%chunk
   808)           i   = pgcols(curgcol)%ccol
   809)           owner_p   = chunks(cid)%owner
   810)           do while (curp < owner_p)
   811)              btofc_blk_num(curp) = curcnt
   812)              curcnt = 0
   813)              curp = curp + 1
   814)           enddo
   815)           curgcol_d = chunks(cid)%gcol(i)
   816)           block_cnt = get_gcol_block_cnt_d(curgcol_d)
   817)           call get_gcol_block_d(curgcol_d,block_cnt,blockids,bcids)
   818)           do jb = 1,block_cnt
   819)              owner_d = get_block_owner_d(blockids(jb))
   820)              if (iam == owner_d) then
   821)                 if (.not. associated(btofc_blk_offset(blockids(jb))%pter)) then
   822)                    blksiz = get_block_gcol_cnt_d(blockids(jb))
   823)                    numlvl = get_block_lvl_cnt_d(blockids(jb),bcids(jb))
   824)                    btofc_blk_offset(blockids(jb))%ncols = blksiz
   825)                    btofc_blk_offset(blockids(jb))%nlvls = numlvl
   826)                    allocate( btofc_blk_offset(blockids(jb))%pter(blksiz,numlvl) )
   827)                 endif
   828)                 do k=1,btofc_blk_offset(blockids(jb))%nlvls
   829)                    btofc_blk_offset(blockids(jb))%pter(bcids(jb),k) = glbcnt
   830)                    curcnt = curcnt + 1
   831)                    glbcnt = glbcnt + 1
   832)                 enddo
   833)              endif
   834)           enddo
   835)        enddo
   836)        btofc_blk_num(curp) = curcnt
   837)        block_buf_nrecs = glbcnt
   838)        !  
   839)        allocate( btofc_chk_num(0:npes-1) )
   840)        btofc_chk_num = 0
   841)        allocate( btofc_chk_offset(begchunk:endchunk) )
   842)        do lcid=begchunk,endchunk
   843)           ncols = lchunks(lcid)%ncols
   844)           btofc_chk_offset(lcid)%ncols = ncols
   845)           btofc_chk_offset(lcid)%nlvls = pver+1
   846)           allocate( btofc_chk_offset(lcid)%pter(ncols,pver+1) )
   847)        enddo
   848)        !
   849)        curcnt = 0
   850)        glbcnt = 0
   851)        do p=0,npes-1
   852)           do curgcol=gs_col_offset(iam),gs_col_offset(iam+1)-1
   853)              cid  = pgcols(curgcol)%chunk
   854)              owner_p  = chunks(cid)%owner
   855)              if (iam == owner_p) then
   856)                 i    = pgcols(curgcol)%ccol
   857)                 lcid = chunks(cid)%lcid
   858)                 curgcol_d = chunks(cid)%gcol(i)
   859)                 block_cnt = get_gcol_block_cnt_d(curgcol_d)
   860)                 call get_gcol_block_d(curgcol_d,block_cnt,blockids,bcids)
   861)                 do jb = 1,block_cnt
   862)                    owner_d = get_block_owner_d(blockids(jb))
   863)                    if (p == owner_d) then
   864)                       numlvl = get_block_lvl_cnt_d(blockids(jb),bcids(jb))
   865)                       call get_block_levels_d(blockids(jb),bcids(jb),numlvl,levels)
   866)                       do k=1,numlvl
   867)                          btofc_chk_offset(lcid)%pter(i,levels(k)+1) = glbcnt
   868)                          curcnt = curcnt + 1
   869)                          glbcnt = glbcnt + 1
   870)                       enddo
   871)                    endif
   872)                 enddo
   873)              endif
   874)           enddo
   875)           btofc_chk_num(p) = curcnt
   876)           curcnt = 0
   877)        enddo
   878)        chunk_buf_nrecs = glbcnt
   879)        !
   880)        ! Precompute swap partners and number of steps in point-to-point
   881)        ! implementations of alltoall algorithm.
   882)        ! First, determine number of swaps.
   883)        !
   884)        dp_coup_steps = 0
   885)        do i=1,ceil2(npes)-1
   886)           p = pair(npes,i,iam)
   887)           if (p >= 0) then
   888)              if ((btofc_blk_num(p) > 0 .or. btofc_chk_num(p) > 0)) then
   889)                 dp_coup_steps = dp_coup_steps + 1
   890)              end if
   891)           end if
   892)        end do
   893)        !
   894)        ! Second, determine swap partners.
   895)        !
   896)        allocate( dp_coup_proc(dp_coup_steps) )
   897)        dp_coup_steps = 0
   898)        do i=1,ceil2(npes)-1
   899)           p = pair(npes,i,iam)
   900)           if (p >= 0) then
   901)              if ((btofc_blk_num(p) > 0 .or. btofc_chk_num(p) > 0)) then
   902)                 dp_coup_steps = dp_coup_steps + 1
   903)                 dp_coup_proc(dp_coup_steps) = p
   904)              end if
   905)           end if
   906)        end do
   907)        !
   908)     endif
   909) 
   910)     ! Final clean-up
   911)     deallocate( gs_col_offset )
   912)     ! (if eliminate get_lon_xxx, can also deallocate
   913)     !  clat_p_idx, and grid_latlon?))
   914) 
   915)     !
   916)     physgrid_set = .true.   ! Set flag indicating physics grid is now set
   917)     !
   918)     if (masterproc) then
   919)        write(iulog,*) 'PHYS_GRID_INIT:  Using PCOLS=',pcols,     &
   920)             '  phys_loadbalance=',lbal_opt,            &
   921)             '  phys_twin_algorithm=',twin_alg,         &
   922)             '  phys_alltoall=',phys_alltoall,          &
   923)             '  chunks_per_thread=',chunks_per_thread
   924)     endif
   925)     !
   926) 
   927)     call t_stopf("phys_grid_init")
   928)     call t_adj_detailf(+2)
   929)     return
   930)   end subroutine phys_grid_init

   934) subroutine phys_grid_find_col(lat, lon, owner, lcid, icol)
   935) 
   936)    !----------------------------------------------------------------------- 
   937)    ! 
   938)    ! Purpose: Find the global column closest to the point specified by lat
   939)    !          and lon.  Return indices of owning process, local chunk, and 
   940)    !          column.
   941)    ! 
   942)    ! Authors: Phil Rasch / Patrick Worley / B. Eaton
   943)    ! 
   944)    !-----------------------------------------------------------------------
   945) 
   946)    real(r8), intent(in) :: lat, lon    ! requested location in degrees
   947)    integer, intent(out) :: owner       ! rank of chunk owner
   948)    integer, intent(out) :: lcid      ! local chunk index
   949)    integer, intent(out) :: icol        ! column index within the chunk
   950) 
   951)    ! local
   952)    real(r8) dist2           ! the distance (in radians**2 from lat, lon)
   953)    real(r8) distmin         ! the distance (in radians**2 from closest column)
   954)    real(r8) latr, lonr      ! lat, lon (in radians) of requested location
   955)    real(r8) clat, clon      ! lat, lon (in radians) of column being tested
   956)    real(r8) const
   957) 
   958)    integer i
   959)    integer cid
   960)    !-----------------------------------------------------------------------
   961) 
   962)    ! Check that input lat and lon are in valid range
   963)    if (lon < 0.0_r8 .or. lon >= 360._r8 .or. &
   964)        lat < -90._r8 .or. lat > 90._r8) then
   965)       if (masterproc) then
   966)          write(iulog,*) &
   967)             'phys_grid_find_col: ERROR: lon must satisfy 0.<=lon<360. and lat must satisfy -90<=lat<=90.'
   968)          write(iulog,*) &
   969)             'input lon=', lon, '  input lat=', lat
   970)       endif
   971)       call endrun('phys_grid_find_col: input ERROR')
   972)    end if
   973) 
   974)    const = 180._r8/pi            ! degrees per radian
   975)    latr = lat/const              ! to radians
   976)    lonr = lon/const              ! to radians
   977) 
   978)    owner   = -999
   979)    lcid  = -999
   980)    icol    = -999
   981)    distmin = 1.e10_r8
   982) 
   983)    ! scan all chunks for closest point to lat, lon
   984)    do cid = 1, nchunks
   985)       do i = 1, chunks(cid)%ncols
   986)          clat = clat_p(chunks(cid)%lat(i))
   987)          clon = clon_p(chunks(cid)%lon(i))
   988)          dist2 = (clat-latr)**2 + (clon-lonr)**2
   989)          if (dist2 < distmin ) then
   990)             distmin = dist2
   991)             owner = chunks(cid)%owner
   992)             lcid = chunks(cid)%lcid
   993)             icol = i
   994)          endif
   995)       enddo
   996)    end do
   997) 
   998) end subroutine phys_grid_find_col

  1002) subroutine phys_grid_find_cols(lat, lon, nclosest, owner, lcid, icol, distmin, mlats, mlons)
  1003) 
  1004)    !----------------------------------------------------------------------- 
  1005)    ! 
  1006)    ! Purpose: Find the global columns closest to the point specified by lat
  1007)    !          and lon.  Return indices of owning process, local chunk, and 
  1008)    !          column.
  1009)    ! 
  1010)    ! Authors: Phil Rasch / Patrick Worley / B. Eaton
  1011)    ! 
  1012)    !-----------------------------------------------------------------------
  1013)    use physconst,    only : rearth
  1014)    
  1015)    real(r8), intent(in) :: lat, lon            ! requested location in degrees
  1016)    integer, intent(in)  :: nclosest            ! number of closest points to find
  1017)    integer, intent(out) :: owner(nclosest)     ! rank of chunk owner
  1018)    integer, intent(out) :: lcid(nclosest)      ! local chunk index
  1019)    integer, intent(out) :: icol(nclosest)      ! column index within the chunk
  1020)    real(r8),intent(out) :: distmin(nclosest)   ! the distance (m) of the closest column(s)
  1021)    real(r8),intent(out) :: mlats(nclosest)     ! the latitude of the closest column(s)
  1022)    real(r8),intent(out) :: mlons(nclosest)     ! the longitude of the closest column(s)
  1023) 
  1024)    ! local
  1025)    real(r8) dist2           ! the distance (in radians**2 from lat, lon)
  1026)    real(r8) latr, lonr      ! lat, lon (in radians) of requested location
  1027)    real(r8) clat, clon      ! lat, lon (in radians) of column being tested
  1028)    real(r8) const
  1029) 
  1030)    integer i, j
  1031)    integer cid
  1032)    !-----------------------------------------------------------------------
  1033) 
  1034)    ! Check that input lat and lon are in valid range
  1035)    if (lon < 0.0_r8 .or. lon >= 360._r8 .or. &
  1036)        lat < -90._r8 .or. lat > 90._r8) then
  1037)       if (masterproc) then
  1038)          write(iulog,*) &
  1039)             'phys_grid_find_cols: ERROR: lon must satisfy 0.<=lon<360. and lat must satisfy -90<=lat<=90.'
  1040)          write(iulog,*) &
  1041)             'input lon=', lon, '  input lat=', lat
  1042)       endif
  1043)       call endrun('phys_grid_find_cols: input ERROR')
  1044)    end if
  1045) 
  1046)    const = 180._r8/pi            ! degrees per radian
  1047)    latr = lat/const              ! to radians
  1048)    lonr = lon/const              ! to radians
  1049) 
  1050)    owner(:)   = -999
  1051)    lcid(:)    = -999
  1052)    icol(:)    = -999
  1053)    mlats(:)   = -999
  1054)    mlons(:)   = -999
  1055)    distmin(:) = 1.e10_r8
  1056) 
  1057)    ! scan all chunks for closest point to lat, lon
  1058)    do cid = 1, nchunks
  1059)       do i = 1, chunks(cid)%ncols
  1060)          clat = clat_p(chunks(cid)%lat(i))
  1061)          clon = clon_p(chunks(cid)%lon(i))
  1062)          dist2 = acos(sin(latr) * sin(clat) + cos(latr) * cos(clat) * cos(clon - lonr)) * rearth       
  1063)          
  1064)          do j = nclosest, 1, -1
  1065)             if (dist2 < distmin(j)) then
  1066)             
  1067)                if (j < nclosest) then
  1068)                  distmin(j+1) = distmin(j)
  1069)                  owner(j+1)   = owner(j)
  1070)                  lcid(j+1)    = lcid(j)
  1071)                  icol(j+1)    = icol(j)
  1072)                  mlats(j+1)   = mlats(j)
  1073)                  mlons(j+1)    = mlons(j)
  1074)                end if
  1075)              
  1076)                distmin(j) = dist2
  1077)                owner(j)   = chunks(cid)%owner
  1078)                lcid(j)    = chunks(cid)%lcid
  1079)                icol(j)    = i
  1080)                mlats(j)   = clat * const
  1081)                mlons(j)   = clon * const
  1082)             else
  1083)                exit
  1084)             end if
  1085)          enddo
  1086)       enddo
  1087)    end do
  1088)    
  1089) end subroutine phys_grid_find_cols

  1989) !

  1991) ! 
  1992) 
  1993) !------------------------------Arguments--------------------------------
  1994)    integer, intent(in) :: fdim      ! declared length of first dimension
  1995)    integer, intent(in) :: mdim      ! declared length of middle dimension

  2000)                                     ! global field

  2009)    integer :: lcid                       ! local chunk id
  2010)    integer :: lid                        ! local column index
  2011)    integer :: gcol                       ! global column index

  2013)    integer :: h2                         ! second horizontal dimension index

  2102) 
  2103)    do l=1,ldim
  2104) !DIR$ PREFERVECTOR
  2105) !DIR$ PREFERSTREAM
  2106) !DIR$ CONCURRENT
  2107)       do i=1,ngcols_p

  2109)          lcid = chunks(cid)%lcid

  3912) 
  3913)    elseif (opt == 2) then
  3914)

  3916)       do p=0,npes-1

  3919)

  3927) !     (wrap map)
  3928)       nsmpy = 0
  3929)       do p=0,npes-1
  3930)          if (proc_smp_mapx(p) .eq. -1) then
  3931)             proc_smp_mapx(p) = nsmpy
  3932)             nsmpy = mod(nsmpy+1,nsmpx)
  3933)          endif
  3934)       enddo
  3935) 
  3936)    else
  3937) 
  3938)       nsmpx = npes
  3939)       do p=0,npes-1
  3940)          proc_smp_mapx(p) = p
  3941)       enddo
  3942) 
  3943)    endif
  3944) !
  3945)    deallocate( proc_busy_d )

  3949) ! virtual SMP node
  3950) !

  3952) !
  3953)    nsmpprocs(:) = 0
  3954)    do p=0,npes-1
  3955)       smp = proc_smp_mapx(p)
  3956)       nsmpprocs(smp) = nsmpprocs(smp) + 1
  3957)    enddo

  3959) !
  3960)    deallocate( nsmpprocs )
  3961) 
  3962) !
  3963) ! Determine number of columns assigned to each
  3964) ! virtual SMP in block decomposition
  3965) 
  3966)    allocate( col_smp_mapx(ngcols) )
  3967) !
  3968)    col_smp_mapx(:) = -1
  3969)    error = .false.
  3970)    do i=1,ngcols_p
  3971)       curgcol = latlon_to_dyn_gcol_map(i)
  3972)       block_cnt = get_gcol_block_cnt_d(curgcol)

  3975)          p = get_block_owner_d(blockids(jb)) 
  3976)          if (col_smp_mapx(i) .eq. -1) then
  3977)             col_smp_mapx(i) = proc_smp_mapx(p)
  3978)          elseif (col_smp_mapx(i) .ne. proc_smp_mapx(p)) then
  3979)             error = .true.
  3980)          endif
  3981)       enddo
  3982)    end do
  3983)    if (error) then

  3985)                "but vertical decomposition not limited to virtual SMP"
  3986)       call endrun()
  3987)    endif
  3988) !
  3989)    allocate( nsmpcolumns(0:nsmpx-1) )
  3990)    nsmpcolumns(:) = 0
  3991)    do i=1,ngcols_p
  3992)       curgcol = latlon_to_dyn_gcol_map(i)
  3993)       smp = col_smp_mapx(curgcol)
  3994)       nsmpcolumns(smp) = nsmpcolumns(smp) + 1

  4004)    allocate( maxcol_chk (0:nsmpx-1) )
  4005)    allocate( maxcol_chks(0:nsmpx-1) )

  4007)    allocate( local_cid  (0:nsmpx-1) )
  4008)    allocate( cols(1:maxblksiz) )
  4009) !
  4010) ! Options 0-3: split local dynamics blocks into chunks,
  4011) !              using wrap-map assignment of columns and
  4012) !              day/night and north/south column pairs
  4013) !              to chunks to improve load balance

  4015) !  Option 1: local is subset of`processes assigned to same SMP node
  4016) !  Option 2: local is global
  4017) !  Option 3: local is pair of processes chosen to maximize load-balance

  4030)          nsmpthreads(smp) = nsmpthreads(smp) + npthreads(p)

  4065)                maxcol_chks(smp) = nsmpchunks(smp)

  4118)                      cid = cid_offset(smp) + local_cid(smp)
  4119)                   enddo
  4120)                endif

  4122)                if (chunks(cid)%ncols .eq. maxcol_chk(smp)) &
  4123)                   maxcol_chks(smp) = maxcol_chks(smp) - 1

  4128)                chunks(cid)%lat(i)  = lat_p(curgcol)

  4136)                        (maxcol_chks(smp) > 0) .and. (twin_alg > 0)) then

  4139)                                     proc_smp_mapx, twingcol)

  4144)                            maxcol_chks(smp) = maxcol_chks(smp) - 1

  4154)                   endif

  4159)                endif
  4160) !
  4161)             endif
  4162)          enddo
  4163)       enddo
  4164) !
  4165)    else

  4182)          p = get_block_owner_d(j) 
  4183)          nsmpchunks(p) = nsmpchunks(p) + nlchunks
  4184)       enddo
  4185) !
  4186) ! Determine chunk id ranges for each SMP
  4187) !
  4188)       cid_offset(0) = 1
  4189)       local_cid(0) = 0
  4190)       do smp=1,nsmpx-1
  4191)          cid_offset(smp) = cid_offset(smp-1) + nsmpchunks(smp-1)
  4192)          local_cid(smp) = 0
  4193)       enddo
  4194) !
  4195) ! Allocate chunks and knuhcs data structures
  4196) !
  4197)       allocate( chunks(1:nchunks) )
  4198)       allocate( knuhcs(1:ngcols) )
  4199) !
  4200) ! Initialize chunks and knuhcs data structures
  4201) !
  4202)       knuhcs(:)%chunkid = -1
  4203)       knuhcs(:)%col = -1
  4204)       cid = 0

  4206)          p = get_block_owner_d(jb)
  4207)          smp = proc_smp_mapx(p)
  4208)          blksiz = get_block_gcol_cnt_d(jb)

  4212)          do while (ib < blksiz)
  4213) 
  4214)             cid = cid_offset(smp) + local_cid(smp)
  4215)             max_ncols = min(pcols,blksiz-ib)
  4216) 
  4217)             ncols = 0
  4218)             do i=1,max_ncols
  4219)                ib = ib + 1
  4220)                ! check whether global index is for a column that dynamics
  4221)                ! intends to pass to the physics
  4222)                curgcol = cols(ib)
  4223)                if (dyn_to_latlon_gcol_map(curgcol) .ne. -1) then
  4224)                   ! yes - then save the information
  4225)                   ncols = ncols + 1
  4226)                   chunks(cid)%gcol(ncols) = curgcol
  4227)                   chunks(cid)%lon(ncols)  = lon_p(curgcol)
  4228)                   chunks(cid)%lat(ncols)  = lat_p(curgcol)
  4229)                   knuhcs(curgcol)%chunkid = cid
  4230)                   knuhcs(curgcol)%col = ncols
  4231)                endif
  4232)             enddo
  4233)             chunks(cid)%ncols = ncols
  4234) 
  4235)             local_cid(smp) = local_cid(smp) + 1
  4236)          enddo
  4237)       enddo
  4238) !
  4239) ! Set number of threads available in each "SMP node". 
  4240) !
  4241)       do p=0,npes-1
  4242)          nsmpthreads(p) = npthreads(p)
  4243)       enddo
  4244) !
  4245)    endif
  4246) !
  4247) ! Assign chunks to processes.
  4248) !
  4249)    call assign_chunks(npthreads, nsmpx, proc_smp_mapx, &
  4250)                       nsmpthreads, nsmpchunks)
  4251) !
  4252) ! Clean up
  4253) !
  4254)    deallocate( nsmpcolumns )
  4255)    deallocate( nsmpthreads )
  4256)    deallocate( nsmpchunks  )
  4257)    deallocate( maxcol_chk  )
  4258)    deallocate( maxcol_chks )
  4259)    deallocate( cid_offset  )
  4260)    deallocate( local_cid   )
  4261)    deallocate( cols )
  4262)    deallocate( knuhcs )
  4263) 
  4264)    return
  4265)    end subroutine create_chunks

  4269)    subroutine find_partners(opt, proc_busy_d, nsmpx, proc_smp_mapx)
  4270) !----------------------------------------------------------------------- 
  4271) ! 
  4272) ! Purpose: Divide processes into pairs, attempting to maximize the
  4273) !          the number of columns in one process whose twins are in the 
  4274) !          other process.
  4275) ! 
  4276) ! Method: The day/night and north/south hemisphere complement is defined
  4277) !         to be the column twin.
  4278) ! 
  4279) ! Author: Patrick Worley
  4280) ! 
  4281) !-----------------------------------------------------------------------
  4282)    use dyn_grid, only: get_gcol_block_cnt_d, get_gcol_block_d, &
  4283)                        get_block_owner_d
  4284)    use pmgrid, only: plev
  4285) !------------------------------Arguments--------------------------------
  4286)    integer, intent(in)  :: opt           ! chunking option
  4287)    logical, intent(in)  :: proc_busy_d(0:npes-1)
  4288)                                          ! active/idle dynamics process flags
  4289)    integer, intent(out) :: nsmpx         ! calculated number of virtual 
  4290)                                          !  SMP nodes
  4291)    integer, intent(out) :: proc_smp_mapx(0:npes-1)
  4292)                                          ! process/virtual smp map
  4293) !---------------------------Local workspace-----------------------------
  4294)    integer :: gcol_latlon                ! physics column index (latlon sorted)
  4295)    integer :: twingcol_latlon            ! physics column index (latlon sorted)
  4296)    integer :: gcol, twingcol             ! physics column indices
  4297)    integer :: lon, lat, twinlat          ! longitude and latitude indices
  4298)    integer :: twinlon_off                ! estimate as to offset of twinlon
  4299)                                          ! on a latitude line
  4300)    integer :: block_cnt                  ! number of blocks containing data
  4301)                                          ! for a given vertical column
  4302)    integer :: blockids(plev+1)           ! block indices
  4303)    integer :: bcids(plev+1)              ! block column indices
  4304)    integer :: jb                         ! block index
  4305)    integer :: p, twp                     ! process indices
  4306)    integer :: col_proc_mapx(ngcols)      ! location of columns in 
  4307)                                          !  dynamics decomposition
  4308)    integer :: twin_proc_mapx(ngcols)     ! location of column twins in 
  4309)                                          !  dynamics decomposition
  4310)    integer :: twin_cnt(0:npes-1)         ! for each process, number of twins 
  4311)                                          !  in each of the other processes
  4312)    logical :: assigned(0:npes-1)         ! flag indicating whether process
  4313)                                          !  assigned to an SMP node yet
  4314)    integer :: maxpartner, maxcnt         ! process with maximum number of 
  4315)                                          !  twins and this count
  4316) 
  4317)    logical :: error                      ! error flag 
  4318) !-----------------------------------------------------------------------
  4319) !
  4320) ! Determine process location of column and its twin in dynamics decomposition
  4321) !
  4322)    col_proc_mapx(:) = -1
  4323)    twin_proc_mapx(:) = -1
  4324) 
  4325)    error = .false.
  4326)    do gcol_latlon=1,ngcols_p
  4327) 
  4328)       ! Assume latitude and longitude symmetries and that index manipulations
  4329)       ! are sufficient to find partners. (Will be true for lon/lat grids.)
  4330)       gcol = latlon_to_dyn_gcol_map(gcol_latlon)
  4331)       lat = lat_p(gcol)
  4332)       twinlat = clat_p_tot+1-lat
  4333)       lon = lon_p(gcol)
  4334)       twinlon_off = mod((lon-1)+(clat_p_cnt(twinlat)/2), clat_p_cnt(twinlat))
  4335)       twingcol_latlon = clat_p_idx(twinlat) + twinlon_off
  4336)       twingcol = latlon_to_dyn_gcol_map(twingcol_latlon)
  4337) 
  4338)       block_cnt = get_gcol_block_cnt_d(gcol)
  4339)       call get_gcol_block_d(gcol,block_cnt,blockids,bcids)
  4340)       do jb=1,block_cnt
  4341)          p = get_block_owner_d(blockids(jb)) 
  4342)          if (col_proc_mapx(gcol) .eq. -1) then
  4343)             col_proc_mapx(gcol) = p
  4344)          elseif (col_proc_mapx(gcol) .ne. p) then
  4345)             error = .true.
  4346)          endif
  4347)       enddo
  4348) 
  4349)       block_cnt = get_gcol_block_cnt_d(twingcol)
  4350)       call get_gcol_block_d(twingcol,block_cnt,blockids,bcids)
  4351)       do jb=1,block_cnt
  4352)          p = get_block_owner_d(blockids(jb)) 
  4353)          if (twin_proc_mapx(gcol) .eq. -1) then
  4354)             twin_proc_mapx(gcol) = p
  4355)          elseif (twin_proc_mapx(gcol) .ne. p) then
  4356)             error = .true.
  4357)          endif
  4358)       enddo
  4359) 
  4360)    end do
  4361) 
  4362)    if (error) then
  4363)       if (masterproc) then
  4364)          write(iulog,*) "PHYS_GRID_INIT error: opt", opt, "specified, ", &
  4365)             "but vertical decomposition not limited to single process"
  4366)       endif
  4367)       call endrun()
  4368)    endif
  4369) 
  4370) !
  4371) ! Assign process pairs to SMPs, attempting to maximize the number of column,twin
  4372) ! pairs in same SMP.
  4373) !
  4374)    assigned(:) = .false.
  4375)    twin_cnt(:) = 0
  4376)    nsmpx = 0
  4377)    do p=0,npes-1
  4378)       if ((.not. assigned(p)) .and. (proc_busy_d(p))) then
  4379) !
  4380) ! For each process, determine number of twins in each of the other processes
  4381) ! (running over all columns multiple times to minimize memory requirements).
  4382) !
  4383)          do gcol_latlon=1,ngcols_p
  4384)             gcol = latlon_to_dyn_gcol_map(gcol_latlon)
  4385)             if (col_proc_mapx(gcol) .eq. p) then
  4386)                twin_cnt(twin_proc_mapx(gcol)) = &
  4387)                   twin_cnt(twin_proc_mapx(gcol)) + 1
  4388)             endif
  4389)          enddo
  4390) !
  4391) ! Find process with maximum number of twins that has not yet been designated
  4392) ! a partner.
  4393) !
  4394)          maxpartner = -1
  4395)          maxcnt = 0
  4396)          do twp=0,npes-1
  4397)             if ((.not. assigned(twp)) .and. (twp .ne. p)) then
  4398)                if (twin_cnt(twp) >= maxcnt) then
  4399)                   maxcnt = twin_cnt(twp)
  4400)                   maxpartner = twp
  4401)                endif
  4402)             endif
  4403)          enddo
  4404) !
  4405) ! Assign p and twp to the same SMP node
  4406) !
  4407)          if (maxpartner .ne. -1) then
  4408)             assigned(p) = .true.
  4409)             assigned(maxpartner) = .true.
  4410)             proc_smp_mapx(p) = nsmpx
  4411)             proc_smp_mapx(maxpartner) = nsmpx
  4412)             nsmpx = nsmpx + 1
  4413)          else
  4414)             if (masterproc) then
  4415)                write(iulog,*) "PHYS_GRID_INIT error: opt", opt, "specified, ", &
  4416)                   "but could not divide processes into pairs."
  4417)             endif
  4418)             call endrun()
  4419)          endif
  4420) !
  4421)       endif
  4422) !      
  4423)    enddo
  4424) !
  4425)    return
  4426)    end subroutine find_partners

  4430)    subroutine find_twin(gcol, smp, proc_smp_mapx, twingcol_f)
  4431) !----------------------------------------------------------------------- 
  4432) ! 
  4433) ! Purpose: Find column that when paired with gcol in a chunk
  4434) !          balances the load. A column is a candidate to be paired with
  4435) !          gcol if it is in the same SMP node as gcol as defined
  4436) !          by proc_smp_mapx.
  4437) ! 
  4438) ! Method: The day/night and north/south hemisphere complement is
  4439) !         tried first. If it is not a candidate or if it has already been
  4440) !         assigned, then the day/night complement is tried next. If that
  4441) !         also is not available, then nothing is returned.
  4442) ! 
  4443) ! Author: Patrick Worley
  4444) ! 
  4445) !-----------------------------------------------------------------------
  4446)    use dyn_grid, only: get_gcol_block_d, get_block_owner_d
  4447) 
  4448) !------------------------------Arguments--------------------------------
  4449)    integer, intent(in)  :: gcol          ! global column index for column
  4450)                                          ! seeking a twin for
  4451)    integer, intent(in)  :: smp           ! index of SMP node 
  4452)                                          ! currently assigned to
  4453)    integer, intent(in)  :: proc_smp_mapx(0:npes-1)
  4454)                                          ! process/virtual smp map
  4455)    integer, intent(out) :: twingcol_f
  4456)                                          ! global column index for twin
  4457) !---------------------------Local workspace-----------------------------
  4458)    integer :: lon, lat                   ! global lon/lat indices for column
  4459)                                          ! seeking a twin for
  4460)    integer :: twinlon, twinlat           ! lon/lat indices of twin candidate
  4461)    integer :: twinlon_off                ! estimate as to offset of twinlon
  4462)                                          ! on a latitude line
  4463)    logical :: found                      ! found flag
  4464)    integer :: i                          ! loop index
  4465)    integer :: upper, lower               ! search temporaries
  4466)    integer :: twingcol_latlon            ! global physics column index (latlon sorted)
  4467)    integer :: twingcol_lonlat            ! global physics column index (lonlat sorted)
  4468)    integer :: twingcol                   ! global physics column indes
  4469)    integer :: diff, min_diff, min_i      ! search temporaries
  4470)    integer :: jbtwin(npes)               ! global block indices
  4471)    integer :: ibtwin(npes)               ! global column indices
  4472)    integer :: twinproc, twinsmp          ! process and smp ids
  4473) 
  4474)    integer :: clon_p_idx(clon_p_tot)     ! index in lonlat ordering for first 
  4475)                                          !  occurrence of longitude corresponding to 
  4476)                                          !  given latitude index
  4477) 
  4478)    real(r8):: twopi                      ! 2*pi
  4479)    real(r8):: clat, twinclat             ! latitude and twin
  4480)    real(r8):: clon, twinclon             ! longitude and twin
  4481) 
  4482) !-----------------------------------------------------------------------
  4483)    twingcol_f = -1
  4484) 
  4485)    ! precompute clon_p_idx
  4486)    clon_p_idx(1) = 1
  4487)    do i=2,clon_p_tot
  4488)       clon_p_idx(i) = clon_p_idx(i-1) + clon_p_cnt(i-1)
  4489)    enddo
  4490) !
  4491) ! Try day/night and north/south hemisphere complement first
  4492) !
  4493)    ! determine twin latitude
  4494)    lat = lat_p(gcol)
  4495)    clat = clat_p(lat)
  4496)    twinclat = -clat
  4497)    twinlat = clat_p_tot+1-lat
  4498)    if (clat_p(twinlat) .eq. twinclat) then
  4499)       found = .true.
  4500)    else
  4501)       found = .false.
  4502)       upper = twinlat
  4503)       lower = twinlat
  4504)       if (upper < clat_p_tot) upper = twinlat + 1
  4505)       if (lower > 1) lower = twinlat - 1
  4506)    endif
  4507)    do while (.not. found)
  4508)       if      ((abs(clat_p(upper)-twinclat) < abs(clat_p(twinlat)-twinclat)) .and. &
  4509)                (upper .ne. twinlat)) then
  4510)          twinlat = upper
  4511)          if (upper < clat_p_tot) then
  4512)             upper = twinlat + 1
  4513)          else
  4514)             found = .true.
  4515)          endif
  4516)       else if ((abs(clat_p(lower)-twinclat) < abs(clat_p(twinlat)-twinclat)) .and. &
  4517)                (lower .ne. twinlat))    then
  4518)          twinlat = lower
  4519)          if (lower > 1) then
  4520)             lower = twinlat - 1
  4521)          else
  4522)             found = .true.
  4523)          endif
  4524)       else
  4525)          found = .true.
  4526)       endif
  4527)     enddo
  4528) 
  4529)    ! determine twin longitude
  4530)    twopi = 2.0_r8*pi
  4531)    lon = lon_p(gcol)
  4532)    clon = clon_p(lon)
  4533)    twinclon = mod(clon+pi,twopi)
  4534)    twinlon = mod((lon-1)+(clon_p_tot/2), clon_p_tot) + 1
  4535)    if (clon_p(twinlon) .eq. twinclon) then
  4536)       found = .true.
  4537)    else
  4538)       found = .false.
  4539)       upper = twinlon
  4540)       lower = twinlon
  4541)       if (upper < clon_p_tot) upper = twinlon + 1
  4542)       if (lower > 1) lower = twinlon - 1
  4543)    endif
  4544)    do while (.not. found)
  4545)       if      ((abs(clon_p(upper)-twinclon) < abs(clon_p(twinlon)-twinclon)) .and. &
  4546)                (upper .ne. twinlon)) then
  4547)          twinlon = upper
  4548)          if (upper < clon_p_tot) then
  4549)             upper = twinlon + 1
  4550)          else
  4551)             found = .true.
  4552)          endif
  4553)       else if ((abs(clon_p(lower)-twinclon) < abs(clon_p(twinlon)-twinclon)) .and. &
  4554)                (lower .ne. twinlon))    then
  4555)          twinlon = lower
  4556)          if (lower > 1) then
  4557)             lower = twinlon - 1
  4558)          else
  4559)             found = .true.
  4560)          endif
  4561)       else
  4562)          found = .true.
  4563)       endif
  4564)    enddo
  4565) 
  4566)    ! first, look for an exact match (assuming latitude and longitude symmetries)
  4567)    twinlon_off = mod((lon-1)+(clat_p_cnt(twinlat)/2), clat_p_cnt(twinlat))
  4568)    twingcol_latlon = clat_p_idx(twinlat) + twinlon_off
  4569)    twingcol = latlon_to_dyn_gcol_map(twingcol_latlon)
  4570) 
  4571)    ! otherwise, look around for an approximate match using lonlat sorted indices
  4572)    if ((lon_p(twingcol) .ne. twinlon) .or. (lat_p(twingcol) .ne. twinlat)) then
  4573)       twingcol_lonlat = clon_p_idx(twinlon)
  4574)       twingcol = lonlat_to_dyn_gcol_map(twingcol_lonlat)
  4575)       min_diff = abs(lat_p(twingcol) - twinlat)
  4576)       min_i = 0
  4577)       do i = 1, clon_p_cnt(twinlon)-1
  4578)          twingcol_lonlat = clon_p_idx(twinlon)+i
  4579)          twingcol = lonlat_to_dyn_gcol_map(twingcol_lonlat)
  4580)          diff = abs(lat_p(twingcol) - twinlat)
  4581)          if (diff < min_diff) then
  4582)             min_diff = diff
  4583)             min_i = i
  4584)          endif
  4585)       enddo
  4586)       twingcol_lonlat = clon_p_idx(twinlon) + min_i
  4587)       twingcol = lonlat_to_dyn_gcol_map(twingcol_lonlat)
  4588)    endif
  4589) 
  4590)    ! Check whether twin and original are in same smp
  4591)    found = .false.
  4592)    call get_gcol_block_d(twingcol,npes,jbtwin,ibtwin)
  4593)    twinproc = get_block_owner_d(jbtwin(1))
  4594)    twinsmp  = proc_smp_mapx(twinproc)
  4595) !
  4596)    if ((twinsmp .eq. smp) .and. &
  4597)        (knuhcs(twingcol)%chunkid == -1)) then
  4598)       found = .true.
  4599)       twingcol_f = twingcol
  4600)    endif
  4601) !
  4602) ! Try day/night complement next
  4603)    if (.not. found) then
  4604) 
  4605)       ! first, look for an exact match (assuming longitude symmetries)
  4606)       twinlon_off = mod((lon-1)+(clat_p_cnt(lat)/2), clat_p_cnt(lat))
  4607)       twingcol_latlon = clat_p_idx(lat) + twinlon_off
  4608)       twingcol = latlon_to_dyn_gcol_map(twingcol_latlon)
  4609) 
  4610)       ! otherwise, look around for an approximate match using lonlat
  4611)       ! column ordering
  4612)       if ((lon_p(twingcol) .ne. twinlon) .or. &
  4613)           (lat_p(twingcol) .ne. lat)) then
  4614)          twingcol_lonlat = clon_p_idx(twinlon)
  4615)          twingcol = lonlat_to_dyn_gcol_map(twingcol_lonlat)
  4616)          min_diff = abs(lat_p(twingcol) - lat)
  4617)          min_i = 0
  4618)          do i = 1, clon_p_cnt(twinlon)-1
  4619)             twingcol_lonlat = clon_p_idx(twinlon)+i
  4620)             twingcol = lonlat_to_dyn_gcol_map(twingcol_lonlat)
  4621)             diff = abs(lat_p(twingcol) - lat)
  4622)             if (diff < min_diff) then
  4623)                min_diff = diff
  4624)                min_i = i
  4625)             endif
  4626)          enddo
  4627)          twingcol_lonlat = clon_p_idx(twinlon) + min_i
  4628)          twingcol = lonlat_to_dyn_gcol_map(twingcol_lonlat)
  4629)       endif
  4630) !
  4631)       call get_gcol_block_d(twingcol,npes,jbtwin,ibtwin)
  4632)       twinproc = get_block_owner_d(jbtwin(1))
  4633)       twinsmp  = proc_smp_mapx(twinproc)
  4634) !
  4635)       if ((twinsmp .eq. smp) .and. &
  4636)           (knuhcs(twingcol)%chunkid == -1)) then
  4637)          found = .true.
  4638)          twingcol_f = twingcol
  4639)       endif
  4640) !
  4641)    endif
  4642) !
  4643)    return
  4644)    end subroutine find_twin

  4648)    subroutine assign_chunks(npthreads, nsmpx, proc_smp_mapx, &
  4649)                             nsmpthreads, nsmpchunks)
  4650) !----------------------------------------------------------------------- 
  4651) ! 
  4652) ! Purpose: Assign chunks to processes, balancing the number of
  4653) !          chunks per thread and minimizing the communication costs
  4654) !          in dp_coupling subject to the restraint that columns
  4655) !          do not migrate outside of the current SMP node.
  4656) ! 
  4657) ! Method: 
  4658) ! 
  4659) ! Author: Patrick Worley
  4660) ! 
  4661) !-----------------------------------------------------------------------
  4662)    use pmgrid, only: plev
  4663)    use dyn_grid, only: get_gcol_block_cnt_d, get_gcol_block_d,&
  4664)                        get_block_owner_d 
  4665) !------------------------------Arguments--------------------------------
  4666)    integer, intent(in)  :: npthreads(0:npes-1)
  4667)                                          ! number of OpenMP threads per process
  4668)    integer, intent(in)  :: nsmpx         ! virtual smp count
  4669)    integer, intent(in)  :: proc_smp_mapx(0:npes-1)
  4670)                                          ! process/virtual smp map
  4671)    integer, intent(in)  :: nsmpthreads(0:nsmpx-1)
  4672)                                          ! number of OpenMP threads 
  4673)                                          ! per virtual SMP
  4674)    integer, intent(in)  :: nsmpchunks(0:nsmpx-1)
  4675)                                          ! number of chunks assigned 
  4676)                                          ! to a given virtual SMP
  4677) !---------------------------Local workspace-----------------------------
  4678)    integer :: i, jb, p                   ! loop indices
  4679)    integer :: cid                        ! chunk id
  4680)    integer :: smp                        ! SMP index
  4681)    integer :: curgcol                    ! global column index
  4682)    integer :: block_cnt                  ! number of blocks containing data
  4683)                                          ! for a given vertical column
  4684)    integer :: blockids(plev+1)           ! block indices
  4685)    integer :: bcids(plev+1)              ! block column indices
  4686)    integer :: ntsks_smpx(0:nsmpx-1)      ! number of processes per virtual SMP
  4687)    integer :: smp_proc_mapx(0:nsmpx-1,max_nproc_smpx)   
  4688)                                          ! virtual smp to process id map
  4689)    integer :: cid_offset(0:nsmpx)        ! chunk id virtual smp offset
  4690)    integer :: ntmp1_smp(0:nsmpx-1)       ! minimum number of chunks per thread
  4691)                                          !  in a virtual SMP
  4692)    integer :: ntmp2_smp(0:nsmpx-1)       ! number of extra chunks to be assigned
  4693)                                          !  in a virtual SMP
  4694)    integer :: ntmp3_smp(0:nsmpx-1)       ! number of processes in a virtual
  4695)                                          !  SMP that get more extra chunks
  4696)                                          !  than the others
  4697)    integer :: ntmp4_smp(0:nsmpx-1)       ! number of extra chunks per process
  4698)                                          !  in a virtual SMP
  4699)    integer :: ntmp1, ntmp2               ! work variables
  4700) !  integer :: npchunks(0:npes-1)         ! number of chunks to be assigned to
  4701) !                                        !  a given process
  4702)    integer :: cur_npchunks(0:npes-1)     ! current number of chunks assigned 
  4703)                                          !  to a given process
  4704)    integer :: column_count(0:npes-1)     ! number of columns from current chunk
  4705)                                          !  assigned to each process in dynamics
  4706)                                          !  decomposition
  4707) !-----------------------------------------------------------------------
  4708) !
  4709) ! Count number of processes per virtual SMP and determine virtual SMP
  4710) ! to process id map
  4711) !
  4712)    ntsks_smpx(:) = 0
  4713)    smp_proc_mapx(:,:) = -1
  4714)    do p=0,npes-1
  4715)       smp = proc_smp_mapx(p)
  4716)       ntsks_smpx(smp) = ntsks_smpx(smp) + 1
  4717)       smp_proc_mapx(smp,ntsks_smpx(smp)) = p
  4718)    enddo
  4719) !
  4720) ! Determine chunk id ranges for each virtual SMP
  4721) !
  4722)    cid_offset(0) = 1
  4723)    do smp=1,nsmpx
  4724)       cid_offset(smp) = cid_offset(smp-1) + nsmpchunks(smp-1)
  4725)    enddo
  4726) !
  4727) ! Determine number of chunks to assign to each process
  4728) !
  4729)    do smp=0,nsmpx-1
  4730) !
  4731) ! Minimum number of chunks per thread
  4732)       ntmp1_smp(smp) = nsmpchunks(smp)/nsmpthreads(smp)
  4733) 
  4734) ! Number of extra chunks to be assigned
  4735)       ntmp2_smp(smp) = mod(nsmpchunks(smp),nsmpthreads(smp))
  4736) 
  4737) ! Number of processes that get more extra chunks than the others
  4738)       ntmp3_smp(smp) = mod(ntmp2_smp(smp),ntsks_smpx(smp))
  4739) 
  4740) ! Number of extra chunks per process
  4741)       ntmp4_smp(smp) = ntmp2_smp(smp)/ntsks_smpx(smp)
  4742)       if (ntmp3_smp(smp) > 0) then
  4743)          ntmp4_smp(smp) = ntmp4_smp(smp) + 1
  4744)       endif
  4745)    enddo
  4746) 
  4747)    do p=0,npes-1
  4748)       smp = proc_smp_mapx(p)
  4749) 
  4750) ! Update number of extra chunks
  4751)       if (ntmp2_smp(smp) > ntmp4_smp(smp)) then
  4752)          ntmp2_smp(smp) = ntmp2_smp(smp) - ntmp4_smp(smp)
  4753)       else
  4754)          ntmp4_smp(smp) = ntmp2_smp(smp)
  4755)          ntmp2_smp(smp) = 0
  4756)          ntmp3_smp(smp) = 0
  4757)       endif
