   112) ! !IROUTINE: dumpfile - dumps all fields required for restart
   113) !
   114) ! !INTERFACE:
   115) !
   116)     subroutine dumpfile_bin()
   117) !
   118) ! !DESCRIPTION:
   119) !
   120) ! Dumps all values needed for a restart
   121) !
   122) ! !REVISION HISTORY:
   123) !
   124) ! author Elizabeth C. Hunke, LANL
   125) !
   126) ! !USES:
   127) !
   128)       use ice_domain_size
   129)       use ice_flux
   130)       use ice_grid
   131)       use ice_state
   132)       use ice_dyn_evp
   133)       use ice_blocks, only : block, get_block, nx_block, ny_block
   134) !
   135) !EOP
   136) !
   137)       integer (kind=int_kind) :: &
   138)           i, j, k, n, it, iblk, & ! counting indices
   139)           iyear, imonth, iday     ! year, month, day
   140) 
   141)       logical (kind=log_kind) :: diag
   142) 
   143)       character(len=char_len_long) :: filename
   144) 
   145)       iyear = nyr + year_init - 1
   146)       imonth = month
   147)       iday = mday
   148)       
   149)       write(filename,'(a,a,a,i4.4,a,i2.2,a,i2.2,a,i5.5)') &
   150)             restart_dir(1:lenstr(restart_dir)), &
   151)             restart_file(1:lenstr(restart_file)),'.', &
   152)             iyear,'-',month,'-',mday,'-',sec
   153) 
   154)       call ice_open(nu_dump,filename,0)
   155) 
   156)       if (my_task == master_task) then
   157)          write(nu_dump) istep1,time,time_forc
   158)          write(nu_diag,*) 'Writing ',filename(1:lenstr(filename))
   159)          write(nu_diag,*) 'Restart written ',istep1,time,time_forc
   160)       endif
   161) 
   162)       diag = .true.
   163) 
   164)       !-----------------------------------------------------------------
   165)       ! state variables
   166)       !-----------------------------------------------------------------
   167) 
   168)       do n=1,ncat
   169)          call ice_write(nu_dump,0,aicen(:,:,n,:),'ruf8',diag)
   170)          call ice_write(nu_dump,0,vicen(:,:,n,:),'ruf8',diag)
   171)          call ice_write(nu_dump,0,vsnon(:,:,n,:),'ruf8',diag)
   172)          call ice_write(nu_dump,0,trcrn(:,:,nt_Tsfc,n,:),'ruf8',diag)
   173)       enddo
   174) 
   175)       do k=1,ntilyr
   176)          call ice_write(nu_dump,0,eicen(:,:,k,:),'ruf8',diag)
   177)       enddo
   178) 
   179)       do k=1,ntslyr
   180)          call ice_write(nu_dump,0,esnon(:,:,k,:),'ruf8',diag)
   181)       enddo
   182) 
   183)       !-----------------------------------------------------------------
   184)       ! velocity
   185)       !-----------------------------------------------------------------
   186)       call ice_write(nu_dump,0,uvel,'ruf8',diag)
   187)       call ice_write(nu_dump,0,vvel,'ruf8',diag)
   188) 
   189)       !-----------------------------------------------------------------
   190)       ! radiation fields
   191)       !-----------------------------------------------------------------
   192)       call ice_write(nu_dump,0,coszen,'ruf8',diag)
   193)       call ice_write(nu_dump,0,scale_factor,'ruf8',diag)
   194)       call ice_write(nu_dump,0,swvdr,'ruf8',diag)
   195)       call ice_write(nu_dump,0,swvdf,'ruf8',diag)
   196)       call ice_write(nu_dump,0,swidr,'ruf8',diag)
   197)       call ice_write(nu_dump,0,swidf,'ruf8',diag)
   198) 
   199)       !-----------------------------------------------------------------
   200)       ! ocean stress (for bottom heat flux in thermo)
   201)       !-----------------------------------------------------------------
   202)       call ice_write(nu_dump,0,strocnxT,'ruf8',diag)
   203)       call ice_write(nu_dump,0,strocnyT,'ruf8',diag)
   204) 
   205)       !-----------------------------------------------------------------
   206)       ! internal stress
   207)       !-----------------------------------------------------------------
   208)       call ice_write(nu_dump,0,stressp_1,'ruf8',diag)
   209)       call ice_write(nu_dump,0,stressp_3,'ruf8',diag)
   210)       call ice_write(nu_dump,0,stressp_2,'ruf8',diag)
   211)       call ice_write(nu_dump,0,stressp_4,'ruf8',diag)
   212) 
   213)       call ice_write(nu_dump,0,stressm_1,'ruf8',diag)
   214)       call ice_write(nu_dump,0,stressm_3,'ruf8',diag)
   215)       call ice_write(nu_dump,0,stressm_2,'ruf8',diag)
   216)       call ice_write(nu_dump,0,stressm_4,'ruf8',diag)
   217) 
   218)       call ice_write(nu_dump,0,stress12_1,'ruf8',diag)
   219)       call ice_write(nu_dump,0,stress12_3,'ruf8',diag)
   220)       call ice_write(nu_dump,0,stress12_2,'ruf8',diag)
   221)       call ice_write(nu_dump,0,stress12_4,'ruf8',diag)
   222) 
   223)       !-----------------------------------------------------------------
   224)       ! ice mask for dynamics
   225)       !-----------------------------------------------------------------
   226)       
   227)       !$OMP PARALLEL DO PRIVATE(iblk,j,i)
   228)       do iblk = 1, nblocks
   229)          do j = 1, ny_block
   230)          do i = 1, nx_block
   231)             work1(i,j,iblk) = c0
   232)             if (iceumask(i,j,iblk)) work1(i,j,iblk) = c1
   233)          enddo
   234)          enddo
   235)       enddo
   236)       !$OMP END PARALLEL DO
   237)       call ice_write(nu_dump,0,work1,'ruf8',diag)
   238) 
   239)       if (my_task == master_task) then
   240)          write(nu_dump) filename_volpn
   241)          write(nu_dump) filename_aero
   242)          write(nu_dump) filename_iage
   243)          write(nu_dump) filename_FY
   244)          write(nu_dump) filename_lvl
   245) 
   246)          close(nu_dump)
   247)       endif
   248) 
   249)     end subroutine dumpfile_bin

   255) ! !IROUTINE: dumpfile_pio - dumps all fields required for restart
   256) !
   257) ! !INTERFACE:
   258) !
   259)       subroutine dumpfile_pio(filename_spec)
   260) !
   261) ! !DESCRIPTION:
   262) !
   263) ! Dumps all values needed for a restart
   264) !
   265) ! !REVISION HISTORY:
   266) !
   267) ! author Elizabeth C. Hunke, LANL
   268) !
   269) ! !USES:
   270) !
   271)       use ice_domain_size
   272)       use ice_flux
   273)       use ice_grid
   274)       use ice_state
   275)       use ice_dyn_evp
   276)       use ice_blocks, only : block, get_block, nx_block, ny_block
   277)       use ice_pio	
   278)       use pio
   279) !
   280) ! !INPUT/OUTPUT PARAMETERS:
   281) !
   282)       character(len=char_len_long), intent(in) :: filename_spec
   283) !
   284) !EOP
   285) !
   286)       integer (kind=int_kind) :: &
   287)           i, j, k, n, it, iblk, & ! counting indices
   288)           ilo, ihi, jlo, jhi,   & ! counting indices
   289)           lon, lat                ! global indices
   290) 
   291)       character(len=char_len_long) :: filename
   292) 
   293)       logical (kind=log_kind) :: diag
   294) 
   295)       integer (kind=int_kind) :: dimid_ni, dimid_nj, dimid_ncat, &
   296)                                  dimid_ntilyr, dimid_ntslyr
   297) 
   298)       integer (kind=int_kind), allocatable :: dims(:)
   299) 
   300)       type(file_desc_t)     :: File
   301)       type(io_desc_t)       :: iodesc2d
   302)       type(io_desc_t)       :: iodesc3d_ncat
   303)       type(io_desc_t)       :: iodesc3d_ntilyr
   304)       type(io_desc_t)       :: iodesc3d_ntslyr
   305)       type(var_desc_t)      :: varid
   306) 
   307)       type(block) :: this_block 
   308) 
   309)       ! construct path/file
   310)       filename = trim(filename_spec) // ".nc"
   311)          
   312)       ! write pointer (path/file)
   313)       if (my_task == master_task) then
   314)         open(nu_rst_pointer,file=pointer_file)
   315)         write(nu_rst_pointer,'(a)') filename
   316)         close(nu_rst_pointer)
   317)       endif
   318) 
   319)       ! begin writing restart data
   320) 
   321)       File%fh=-1
   322)       call ice_pio_init(mode='write',filename=trim(filename), File=File, &
   323)            clobber=.true., cdf64=lcdf64 )
   324) 
   325) 
   326)       status = pio_put_att(File,pio_global,'istep1',istep1)
   327)       status = pio_put_att(File,pio_global,'time',time)
   328)       status = pio_put_att(File,pio_global,'time_forc',time_forc)
   329)       status = pio_put_att(File,pio_global,'nyr',nyr)
   330)       status = pio_put_att(File,pio_global,'month',month)
   331)       status = pio_put_att(File,pio_global,'mday',mday)
   332)       status = pio_put_att(File,pio_global,'sec',sec)
   333) 
   334)       status = pio_def_dim(File,'ni',nx_global,dimid_ni)
   335)       status = pio_def_dim(File,'nj',ny_global,dimid_nj)
   336)       status = pio_def_dim(File,'ncat',ncat,dimid_ncat)
   337)       status = pio_def_dim(File,'ntilyr',ntilyr,dimid_ntilyr)
   338)       status = pio_def_dim(File,'ntslyr',ntslyr,dimid_ntslyr)
   339)       if (my_task == master_task) then
   340)          write(nu_diag,*) 'Writing ',filename(1:lenstr(filename))
   341)       endif
   342)       diag = .true.
   343) 
   344)       allocate(dims(3))
   345) 
   346)       dims(1) = dimid_ni
   347)       dims(2) = dimid_nj
   348)       dims(3) = dimid_ncat
   349) 
   350)       call define_rest_field(File,'aicen',dims)
   351)       call define_rest_field(File,'vicen',dims)
   352)       call define_rest_field(File,'vsnon',dims)
   353)       call define_rest_field(File,'Tsfcn',dims)
   354) 
   355)       if (tr_aero) then
   356)          do k=1,n_aero
   357)             write(nchar,'(i1.1)') k
   358)             call define_rest_field(File,'aerosnossl'//nchar, dims)
   359)             call define_rest_field(File,'aerosnoint'//nchar, dims)
   360)             call define_rest_field(File,'aeroicessl'//nchar, dims)
   361)             call define_rest_field(File,'aeroiceint'//nchar, dims)
   362)          enddo
   363)       endif
   364) 
   365)       if (tr_iage) then
   366)          call define_rest_field(File,'iage',dims)
   367)       end if
   368) 
   369)       if (tr_FY) then
   370)          call define_rest_field(File,'FY',dims)
   371)       end if
   372) 
   373)       if (tr_lvl) then
   374)          call define_rest_field(File,'alvl',dims)
   375)          call define_rest_field(File,'vlvl',dims)
   376)       end if
   377) 
   378)       if (tr_pond) then
   379)          call define_rest_field(File,'volpn' ,dims)
   380)          call define_rest_field(File,'apondn',dims)
   381)          call define_rest_field(File,'hpondn',dims)
   382)       end if
   383) 
   384)       dims(3) = dimid_ntilyr
   385)       call define_rest_field(File,'eicen',dims)
   386)       dims(3) = dimid_ntslyr
   387)       call define_rest_field(File,'esnon',dims)
   388) 
   389)       deallocate(dims)
   390) 
   391)       allocate(dims(2))
   392)       dims(1) = dimid_ni
   393)       dims(2) = dimid_nj
   394) 
   395)       call define_rest_field(File,'uvel',dims)
   396)       call define_rest_field(File,'vvel',dims)
   397) 
   398)       call define_rest_field(File,'coszen',dims)
   399)       call define_rest_field(File,'scale_factor',dims)
   400)       call define_rest_field(File,'swvdr',dims)
   401)       call define_rest_field(File,'swvdf',dims)
   402)       call define_rest_field(File,'swidr',dims)
   403)       call define_rest_field(File,'swidf',dims)
   404) 
   405)       call define_rest_field(File,'strocnxT',dims)
   406)       call define_rest_field(File,'strocnyT',dims)
   407) 
   408)       call define_rest_field(File,'stressp_1',dims)
   409)       call define_rest_field(File,'stressp_2',dims)
   410)       call define_rest_field(File,'stressp_3',dims)
   411)       call define_rest_field(File,'stressp_4',dims)
   412) 
   413)       call define_rest_field(File,'stressm_1',dims)
   414)       call define_rest_field(File,'stressm_2',dims)
   415)       call define_rest_field(File,'stressm_3',dims)
   416)       call define_rest_field(File,'stressm_4',dims)
   417) 
   418)       call define_rest_field(File,'stress12_1',dims)
   419)       call define_rest_field(File,'stress12_2',dims)
   420)       call define_rest_field(File,'stress12_3',dims)
   421)       call define_rest_field(File,'stress12_4',dims)
   422) 
   423)       call define_rest_field(File,'iceumask',dims)
   424) 
   425)       status = pio_enddef(File)
   426) 
   427)       deallocate(dims)
   428) 
   429)       !-----------------------------------------------------------------
   430)       ! state variables
   431)       !-----------------------------------------------------------------
   432)       call ice_pio_initdecomp(ndim3=ncat  , iodesc=iodesc3d_ncat)
   433) 
   434)       status = pio_inq_varid(File,'aicen',varid)
   435)       call pio_write_darray(File, varid, iodesc3d_ncat, aicen(:,:,:,:), status, fillval=c0)
   436)       
   437)       status = pio_inq_varid(File,'vicen',varid)
   438)       call pio_write_darray(File, varid, iodesc3d_ncat, vicen, status, fillval=c0)
   439)       
   440)       status = pio_inq_varid(File,'vsnon',varid)
   441)       call pio_write_darray(File, varid, iodesc3d_ncat, vsnon, status, fillval=c0)
   442) 
   443)       status = pio_inq_varid(File,'Tsfcn',varid)
   444)       call pio_write_darray(File, varid, iodesc3d_ncat, trcrn(:,:,nt_Tsfc,:,:), status, fillval=c0)
   445) 
   446)       call ice_pio_initdecomp(ndim3=ntilyr, iodesc=iodesc3d_ntilyr)
   447)       status = pio_inq_varid(File,'eicen',varid)
   448)       call pio_write_darray(File, varid, iodesc3d_ntilyr, eicen, status, fillval=c0)
   449)       call PIO_freeDecomp(File,iodesc3d_ntilyr)
   450) 
   451)       call ice_pio_initdecomp(ndim3=ntslyr, iodesc=iodesc3d_ntslyr)
   452)       status = pio_inq_varid(File,'esnon',varid)
   453)       call pio_write_darray(File, varid, iodesc3d_ntslyr, esnon, status, fillval=c0)
   454) 
   455)       call PIO_freeDecomp(File,iodesc3d_ntslyr)
   456) 
   457)       call ice_pio_initdecomp(iodesc=iodesc2d)
   458)       !-----------------------------------------------------------------
   459)       ! velocity
   460)       !-----------------------------------------------------------------
   461)       status = pio_inq_varid(File,'uvel',varid)
   462)       call pio_write_darray(File, varid, iodesc2d, uvel, status, fillval=c0)
   463) 
   464)       status = pio_inq_varid(File,'vvel',varid)
   465)       call pio_write_darray(File, varid, iodesc2d, vvel, status, fillval=c0)
   466) 
   467)       !-----------------------------------------------------------------
   468)       ! radiation fields
   469)       !-----------------------------------------------------------------
   470)       status = pio_inq_varid(File,'coszen',varid)
   471)       call pio_write_darray(File, varid, iodesc2d, coszen, status, fillval=c0)
   472) 
   473)       status = pio_inq_varid(File,'scale_factor',varid)
   474)       call pio_write_darray(File, varid, iodesc2d, scale_factor, status, fillval=c0)
   475) 
   476)       status = pio_inq_varid(File,'swvdr',varid)
   477)       call pio_write_darray(File, varid, iodesc2d, swvdr, status, fillval=c0)
   478) 
   479)       status = pio_inq_varid(File,'swvdf',varid)
   480)       call pio_write_darray(File, varid, iodesc2d, swvdf, status, fillval=c0)
   481) 
   482)       status = pio_inq_varid(File,'swidr',varid)
   483)       call pio_write_darray(File, varid, iodesc2d, swidr, status, fillval=c0)
   484) 
   485)       status = pio_inq_varid(File,'swidf',varid)
   486)       call pio_write_darray(File, varid, iodesc2d, swidf, status, fillval=c0)
   487) 
   488)       !-----------------------------------------------------------------
   489)       ! ocean stress (for bottom heat flux in thermo)
   490)       !-----------------------------------------------------------------
   491)       status = pio_inq_varid(File,'strocnxT',varid)
   492)       call pio_write_darray(File, varid, iodesc2d, strocnxT, status, fillval=c0)
   493) 
   494)       status = pio_inq_varid(File,'strocnyT',varid)
   495)       call pio_write_darray(File, varid, iodesc2d, strocnyT, status, fillval=c0)
   496) 
   497)       !-----------------------------------------------------------------
   498)       ! internal stress
   499)       !-----------------------------------------------------------------
   500) 
   501)       status = pio_inq_varid(File,'stressp_1',varid)
   502)       call pio_write_darray(File, varid, iodesc2d, stressp_1, status, fillval=c0)
   503) 
   504)       status = pio_inq_varid(File,'stressp_2',varid)
   505)       call pio_write_darray(File, varid, iodesc2d, stressp_2, status, fillval=c0)
   506) 
   507)       status = pio_inq_varid(File,'stressp_3',varid)
   508)       call pio_write_darray(File, varid, iodesc2d, stressp_3, status, fillval=c0)
   509) 
   510)       status = pio_inq_varid(File,'stressp_4',varid)
   511)       call pio_write_darray(File, varid, iodesc2d, stressp_4, status, fillval=c0)
   512) 
   513)       status = pio_inq_varid(File,'stressm_1',varid)
   514)       call pio_write_darray(File, varid, iodesc2d, stressm_1, status, fillval=c0)
   515) 
   516)       status = pio_inq_varid(File,'stressm_2',varid)
   517)       call pio_write_darray(File, varid, iodesc2d, stressm_2, status, fillval=c0)
