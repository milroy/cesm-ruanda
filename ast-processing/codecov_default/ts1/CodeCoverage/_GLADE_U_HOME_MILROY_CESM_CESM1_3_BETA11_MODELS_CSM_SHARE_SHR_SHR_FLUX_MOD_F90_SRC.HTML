    63) ! !IROUTINE: shr_flux_atmOcn -- internal atm/ocn flux calculation
    64) !
    65) ! !DESCRIPTION:
    66) !
    67) !     Internal atm/ocn flux calculation
    68) !     
    69) ! !REVISION HISTORY:
    70) !     2002-Jun-10 - B. Kauffman - code migrated from cpl5 to cpl6
    71) !     2003-Apr-02 - B. Kauffman - taux & tauy now utilize ocn velocity
    72) !     2003-Apr-02 - B. Kauffman - tref,qref,duu10n mods as per Bill Large
    73) !     2006-Nov-07 - B. Kauffman - code migrated from cpl6 to share
    74) !

    77) SUBROUTINE shr_flux_atmOcn(nMax  ,zbot  ,ubot  ,vbot  ,thbot ,   & 
    78)            &               qbot  ,rbot  ,tbot  ,us    ,vs    ,   &
    79)            &               ts    ,mask  ,sen   ,lat   ,lwup  ,   &
    80)            &               evap  ,taux  ,tauy  ,tref  ,qref  ,   &
    81)            &               duu10n,  ustar_sv   ,re_sv ,ssq_sv,   &
    82)            &               missval    )
    83) 
    84) ! !USES:
    85) 
    86)    implicit none
    87) 
    88) ! !INPUT/OUTPUT PARAMETERS:
    89) 
    90)    !--- input arguments --------------------------------
    91)    integer(IN),intent(in) ::       nMax  ! data vector length
    92)    integer(IN),intent(in) :: mask (nMax) ! ocn domain mask       0 <=> out of domain
    93)    real(R8)   ,intent(in) :: zbot (nMax) ! atm level height      (m)
    94)    real(R8)   ,intent(in) :: ubot (nMax) ! atm u wind            (m/s)
    95)    real(R8)   ,intent(in) :: vbot (nMax) ! atm v wind            (m/s)
    96)    real(R8)   ,intent(in) :: thbot(nMax) ! atm potential T       (K)
    97)    real(R8)   ,intent(in) :: qbot (nMax) ! atm specific humidity (kg/kg)
    98)    real(R8)   ,intent(in) :: rbot (nMax) ! atm air density       (kg/m^3)
    99)    real(R8)   ,intent(in) :: tbot (nMax) ! atm T                 (K) 
   100)    real(R8)   ,intent(in) :: us   (nMax) ! ocn u-velocity        (m/s)
   101)    real(R8)   ,intent(in) :: vs   (nMax) ! ocn v-velocity        (m/s)
   102)    real(R8)   ,intent(in) :: ts   (nMax) ! ocn temperature       (K)
   103) 
   104)    !--- output arguments -------------------------------
   105)    real(R8),intent(out)  ::  sen  (nMax) ! heat flux: sensible    (W/m^2)
   106)    real(R8),intent(out)  ::  lat  (nMax) ! heat flux: latent      (W/m^2)
   107)    real(R8),intent(out)  ::  lwup (nMax) ! heat flux: lw upward   (W/m^2)
   108)    real(R8),intent(out)  ::  evap (nMax) ! water flux: evap  ((kg/s)/m^2)
   109)    real(R8),intent(out)  ::  taux (nMax) ! surface stress, zonal      (N)
   110)    real(R8),intent(out)  ::  tauy (nMax) ! surface stress, maridional (N)
   111)    real(R8),intent(out)  ::  tref (nMax) ! diag:  2m ref height T     (K)
   112)    real(R8),intent(out)  ::  qref (nMax) ! diag:  2m ref humidity (kg/kg)
   113)    real(R8),intent(out)  :: duu10n(nMax) ! diag: 10m wind speed squared (m/s)^2
   114) 
   115)    real(R8),intent(out),optional :: ustar_sv(nMax) ! diag: ustar
   116)    real(R8),intent(out),optional :: re_sv   (nMax) ! diag: sqrt of exchange coefficient (water)
   117)    real(R8),intent(out),optional :: ssq_sv  (nMax) ! diag: sea surface humidity  (kg/kg)
   118)  
   119)    real(R8),intent(in) ,optional :: missval        ! masked value
   120) 
   121) ! !EOP
   122) 
   123)    !--- local constants --------------------------------
   124)    real(R8),parameter :: umin  =  0.5_R8 ! minimum wind speed       (m/s)
   125)    real(R8),parameter :: zref  = 10.0_R8 ! reference height           (m)
   126)    real(R8),parameter :: ztref =  2.0_R8 ! reference height for air T (m)
   127) 
   128)    !--- local variables --------------------------------
   129)    integer(IN) :: n      ! vector loop index
   130)    real(R8)    :: vmag   ! surface wind magnitude   (m/s)
   131)    real(R8)    :: thvbot ! virtual temperature      (K)
   132)    real(R8)    :: ssq    ! sea surface humidity     (kg/kg)
   133)    real(R8)    :: delt   ! potential T difference   (K)
   134)    real(R8)    :: delq   ! humidity difference      (kg/kg)
   135)    real(R8)    :: stable ! stability factor
   136)    real(R8)    :: rdn    ! sqrt of neutral exchange coeff (momentum) 
   137)    real(R8)    :: rhn    ! sqrt of neutral exchange coeff (heat)     
   138)    real(R8)    :: ren    ! sqrt of neutral exchange coeff (water)    
   139)    real(R8)    :: rd     ! sqrt of exchange coefficient (momentum)         
   140)    real(R8)    :: rh     ! sqrt of exchange coefficient (heat)             
   141)    real(R8)    :: re     ! sqrt of exchange coefficient (water)            
   142)    real(R8)    :: ustar  ! ustar             
   143)    real(R8)    :: qstar  ! qstar             
   144)    real(R8)    :: tstar  ! tstar             
   145)    real(R8)    :: hol    ! H (at zbot) over L
   146)    real(R8)    :: xsq    ! ?
   147)    real(R8)    :: xqq    ! ?
   148)    real(R8)    :: psimh  ! stability function at zbot (momentum)
   149)    real(R8)    :: psixh  ! stability function at zbot (heat and water)
   150)    real(R8)    :: psix2  ! stability function at ztref reference height
   151)    real(R8)    :: alz    ! ln(zbot/zref)
   152)    real(R8)    :: al2    ! ln(zref/ztref)
   153)    real(R8)    :: u10n   ! 10m neutral wind 
   154)    real(R8)    :: tau    ! stress at zbot
   155)    real(R8)    :: cp     ! specific heat of moist air
   156)    real(R8)    :: bn     ! exchange coef funct for interpolation
   157)    real(R8)    :: bh     ! exchange coef funct for interpolation
   158)    real(R8)    :: fac    ! vertical interpolation factor
   159)    real(R8)    :: spval  ! local missing value
   160) 
   161)    !--- local functions --------------------------------
   162)    real(R8)    :: qsat   ! function: the saturation humididty of air (kg/m^3)
   163)    real(R8)    :: cdn    ! function: neutral drag coeff at 10m
   164)    real(R8)    :: psimhu ! function: unstable part of psimh
   165)    real(R8)    :: psixhu ! function: unstable part of psimx
   166)    real(R8)    :: Umps   ! dummy arg ~ wind velocity (m/s)
   167)    real(R8)    :: Tk     ! dummy arg ~ temperature (K)
   168)    real(R8)    :: xd     ! dummy arg ~ ?
   169)  
   170)    qsat(Tk)   = 640380.0_R8 / exp(5107.4_R8/Tk)
   171)    cdn(Umps)  =   0.0027_R8 / Umps + 0.000142_R8 + 0.0000764_R8 * Umps
   172)    psimhu(xd) = log((1.0_R8+xd*(2.0_R8+xd))*(1.0_R8+xd*xd)/8.0_R8) - 2.0_R8*atan(xd) + 1.571_R8
   173)    psixhu(xd) = 2.0_R8 * log((1.0_R8 + xd*xd)/2.0_R8)
   174)  
   175)    !--- formats ----------------------------------------
   176)    character(*),parameter :: subName = '(shr_flux_atmOcn) '
   177)    character(*),parameter ::   F00 = "('(shr_flux_atmOcn) ',4a)"
   178) 
   179) !-------------------------------------------------------------------------------
   180) ! PURPOSE:
   181) !   computes atm/ocn surface fluxes
   182) !
   183) ! NOTES: 
   184) !   o all fluxes are positive downward
   185) !   o net heat flux = net sw + lw up + lw down + sen + lat
   186) !   o here, tstar = <WT>/U*, and qstar = <WQ>/U*.
   187) !   o wind speeds should all be above a minimum speed (eg. 1.0 m/s)
   188) ! 
   189) ! ASSUMPTIONS:
   190) !   o Neutral 10m drag coeff: cdn = .0027/U10 + .000142 + .0000764 U10
   191) !   o Neutral 10m stanton number: ctn = .0327 sqrt(cdn), unstable
   192) !                                 ctn = .0180 sqrt(cdn), stable
   193) !   o Neutral 10m dalton number:  cen = .0346 sqrt(cdn)
   194) !   o The saturation humidity of air at T(K): qsat(T)  (kg/m^3)
   195) !-------------------------------------------------------------------------------
   196) 
   197)    if (debug > 0 .and. s_loglev > 0) write(s_logunit,F00) "enter"
   198) 
   199)    if (present(missval)) then
   200)       spval = missval
   201)    else
   202)       spval = shr_const_spval
   203)    endif
   204)  
   205)    al2 = log(zref/ztref)
   206) 
   207)    DO n=1,nMax
   208)      if (mask(n) /= 0) then
   209)     
   210)         !--- compute some needed quantities ---
   211)         vmag   = max(umin, sqrt( (ubot(n)-us(n))**2 + (vbot(n)-vs(n))**2) )
   212)         thvbot = thbot(n) * (1.0_R8 + shr_const_zvir * qbot(n)) ! virtual temp (K)
   213)         ssq    = 0.98_R8 * qsat(ts(n)) / rbot(n)   ! sea surf hum (kg/kg)
   214)         delt   = thbot(n) - ts(n)                  ! pot temp diff (K)
   215)         delq   = qbot(n) - ssq                     ! spec hum dif (kg/kg)
   216)         alz    = log(zbot(n)/zref) 
   217)         cp     = shr_const_cpdair*(1.0_R8 + shr_const_cpvir*ssq) 
   218)    
   219)         !------------------------------------------------------------
   220)         ! first estimate of Z/L and ustar, tstar and qstar
   221)         !------------------------------------------------------------
   222)    
   223)         !--- neutral coefficients, z/L = 0.0 ---
   224)         stable = 0.5_R8 + sign(0.5_R8 , delt)
   225)         rdn    = sqrt(cdn(vmag))
   226)         rhn    = (1.0_R8-stable) * 0.0327_R8 + stable * 0.018_R8 
   227)         ren    = 0.0346_R8 
   228)    
   229)         !--- ustar, tstar, qstar ---
   230)         ustar = rdn * vmag
   231)         tstar = rhn * delt  
   232)         qstar = ren * delq  
   233)    
   234)         !--- compute stability & evaluate all stability functions ---
   235)         hol  = shr_const_karman*shr_const_g*zbot(n)*  &
   236)                (tstar/thbot(n)+qstar/(1.0_R8/shr_const_zvir+qbot(n)))/ustar**2
   237)         hol  = sign( min(abs(hol),10.0_R8), hol )
   238)         stable = 0.5_R8 + sign(0.5_R8 , hol)
   239)         xsq    = max(sqrt(abs(1.0_R8 - 16.0_R8*hol)) , 1.0_R8)
   240)         xqq    = sqrt(xsq)
   241)         psimh  = -5.0_R8*hol*stable + (1.0_R8-stable)*psimhu(xqq)
   242)         psixh  = -5.0_R8*hol*stable + (1.0_R8-stable)*psixhu(xqq)
   243)    
   244)         !--- shift wind speed using old coefficient ---
   245)         rd   = rdn / (1.0_R8 + rdn/shr_const_karman*(alz-psimh))
   246)         u10n = vmag * rd / rdn 
   247)    
   248)         !--- update transfer coeffs at 10m and neutral stability ---
   249)         rdn = sqrt(cdn(u10n))
   250)         ren = 0.0346_R8
   251)         rhn = (1.0_R8-stable)*0.0327_R8 + stable * 0.018_R8 
   252)     
   253)         !--- shift all coeffs to measurement height and stability ---
   254)         rd = rdn / (1.0_R8 + rdn/shr_const_karman*(alz-psimh)) 
   255)         rh = rhn / (1.0_R8 + rhn/shr_const_karman*(alz-psixh)) 
   256)         re = ren / (1.0_R8 + ren/shr_const_karman*(alz-psixh)) 
   257)    
   258)         !--- update ustar, tstar, qstar using updated, shifted coeffs --
   259)         ustar = rd * vmag 
   260)         tstar = rh * delt 
   261)         qstar = re * delq 
   262)     
   263)         !------------------------------------------------------------
   264)         ! iterate to converge on Z/L, ustar, tstar and qstar
   265)         !------------------------------------------------------------
   266)     
   267)         !--- compute stability & evaluate all stability functions ---
   268)         hol  = shr_const_karman*shr_const_g*zbot(n)* &
   269)                (tstar/thbot(n)+qstar/(1.0_R8/shr_const_zvir+qbot(n)))/ustar**2
   270)         hol  = sign( min(abs(hol),10.0_R8), hol )
   271)         stable = 0.5_R8 + sign(0.5_R8 , hol)
   272)         xsq    = max(sqrt(abs(1.0_R8 - 16.0_R8*hol)) , 1.0_R8)
   273)         xqq    = sqrt(xsq)
   274)         psimh  = -5.0_R8*hol*stable + (1.0_R8-stable)*psimhu(xqq)
   275)         psixh  = -5.0_R8*hol*stable + (1.0_R8-stable)*psixhu(xqq)
   276)     
   277)         !--- shift wind speed using old coeffs ---
   278)         rd   = rdn / (1.0_R8 + rdn/shr_const_karman*(alz-psimh))
   279)         u10n = vmag * rd/rdn 
   280)     
   281)         !--- update transfer coeffs at 10m and neutral stability ---
   282)         rdn = sqrt(cdn(u10n))
   283)         ren = 0.0346_R8
   284)         rhn = (1.0_R8 - stable)*0.0327_R8 + stable * 0.018_R8 
   285)    
   286)         !--- shift all coeffs to measurement height and stability ---
   287)         rd = rdn / (1.0_R8 + rdn/shr_const_karman*(alz-psimh)) 
   288)         rh = rhn / (1.0_R8 + rhn/shr_const_karman*(alz-psixh)) 
   289)         re = ren / (1.0_R8 + ren/shr_const_karman*(alz-psixh)) 
   290)     
   291)         !--- update ustar, tstar, qstar using updated, shifted coeffs ---
   292)         ustar = rd * vmag 
   293)         tstar = rh * delt 
   294)         qstar = re * delq 
   295)     
   296)         !------------------------------------------------------------
   297)         ! compute the fluxes
   298)         !------------------------------------------------------------
   299)     
   300)         tau = rbot(n) * ustar * ustar 
   301)        
   302)         !--- momentum flux ---
   303)         taux(n) = tau * (ubot(n)-us(n)) / vmag 
   304)         tauy(n) = tau * (vbot(n)-vs(n)) / vmag 
   305)         
   306)         !--- heat flux ---
   307)         sen (n) =                cp * tau * tstar / ustar 
   308)         lat (n) =  shr_const_latvap * tau * qstar / ustar
   309)         lwup(n) = -shr_const_stebol * ts(n)**4 
   310)       
   311)         !--- water flux ---
   312)         evap(n) = lat(n)/shr_const_latvap 
   313)     
   314)         !------------------------------------------------------------
   315)         ! compute diagnositcs: 2m ref T & Q, 10m wind speed squared
   316)         !------------------------------------------------------------
   317)         hol = hol*ztref/zbot(n)
   318)         xsq = max( 1.0_R8, sqrt(abs(1.0_R8-16.0_R8*hol)) )
   319)         xqq = sqrt(xsq)
   320)         psix2   = -5.0_R8*hol*stable + (1.0_R8-stable)*psixhu(xqq)
   321)         fac     = (rh/shr_const_karman) * (alz + al2 - psixh + psix2 )
   322)         tref(n) = thbot(n) - delt*fac 
   323)         tref(n) = tref(n) - 0.01_R8*ztref   ! pot temp to temp correction
   324)         fac     = (re/shr_const_karman) * (alz + al2 - psixh + psix2 )
   325)         qref(n) =  qbot(n) - delq*fac
   326)     
   327)         duu10n(n) = u10n*u10n ! 10m wind speed squared
   328) 
   329)         !------------------------------------------------------------
   330)         ! optional diagnostics, needed for water tracer fluxes (dcn)
   331)         !------------------------------------------------------------
   332)         if (present(ustar_sv)) ustar_sv(n) = ustar
   333)         if (present(re_sv   )) re_sv(n)    = re
   334)         if (present(ssq_sv  )) ssq_sv(n)   = ssq
   335) 
   336)      else
   337)         !------------------------------------------------------------
   338)         ! no valid data here -- out of domain
   339)         !------------------------------------------------------------
   340)         sen   (n) = spval  ! sensible         heat flux  (W/m^2)
   341)         lat   (n) = spval  ! latent           heat flux  (W/m^2)
   342)         lwup  (n) = spval  ! long-wave upward heat flux  (W/m^2)
   343)         evap  (n) = spval  ! evaporative water flux ((kg/s)/m^2)
   344)         taux  (n) = spval  ! x surface stress (N)
   345)         tauy  (n) = spval  ! y surface stress (N)
   346)         tref  (n) = spval  !  2m reference height temperature (K)
   347)         qref  (n) = spval  !  2m reference height humidity (kg/kg)
   348)         duu10n(n) = spval  ! 10m wind speed squared (m/s)^2
   349) 
   350)         if (present(ustar_sv)) ustar_sv(n) = spval
   351)         if (present(re_sv   )) re_sv   (n) = spval
   352)         if (present(ssq_sv  )) ssq_sv  (n) = spval
   353)      endif
   354)    ENDDO 
   355) 
   356) END subroutine shr_flux_atmOcn

   369) 
   370) subroutine shr_flux_atmIce(mask  ,zbot  ,ubot  ,vbot  ,thbot  &
   371)                &          ,qbot  ,rbot  ,tbot  ,ts    ,sen    &
   372)                &          ,lat   ,lwup  ,evap  ,taux  ,tauy   &
   373)                &          ,tref  ,qref                        )
   374) 
   375)    implicit none
   376) 
   377) ! !INPUT/OUTPUT PARAMETERS:
   378) 
   379)    !--- input arguments --------------------------------
   380)    integer(IN),intent(in)  :: mask (:)    ! 0 <=> cell NOT in model domain
   381)    real(R8)   ,intent(in)  :: zbot (:)    ! atm level height  (m)
   382)    real(R8)   ,intent(in)  :: ubot (:)    ! atm u wind     (m/s)
   383)    real(R8)   ,intent(in)  :: vbot (:)    ! atm v wind     (m/s)
   384)    real(R8)   ,intent(in)  :: thbot(:)    ! atm potential T   (K)
   385)    real(R8)   ,intent(in)  :: qbot (:)    ! atm specific humidity (kg/kg)
   386)    real(R8)   ,intent(in)  :: rbot (:)    ! atm air density   (kg/m^3)
   387)    real(R8)   ,intent(in)  :: tbot (:)    ! atm T       (K) 
   388)    real(R8)   ,intent(in)  :: ts   (:)    ! surface temperature
   389) 
   390)    !--- output arguments -------------------------------
   391)    real(R8)   ,intent(out) :: sen  (:)    ! sensible      heat flux  (W/m^2)
   392)    real(R8)   ,intent(out) :: lat  (:)    ! latent        heat flux  (W/m^2)
   393)    real(R8)   ,intent(out) :: lwup (:)    ! long-wave upward heat flux  (W/m^2)
   394)    real(R8)   ,intent(out) :: evap (:)    ! evaporative water flux ((kg/s)/m^2)
   395)    real(R8)   ,intent(out) :: taux (:)    ! x surface stress (N)
   396)    real(R8)   ,intent(out) :: tauy (:)    ! y surface stress (N)
   397)    real(R8)   ,intent(out) :: tref (:)    ! 2m reference height temperature
   398)    real(R8)   ,intent(out) :: qref (:)    ! 2m reference height humidity
   399)  
   400) !EOP
   401) 
   402)    !--- local constants --------------------------------
   403)    real(R8),parameter :: umin   =  1.0_R8            ! minimum wind speed (m/s)
   404)    real(R8),parameter :: zref   = 10.0_R8            ! ref height           ~ m
   405)    real(R8),parameter :: ztref  =  2.0_R8            ! ref height for air T ~ m
   406)    real(R8),parameter :: spval  = shr_const_spval    ! special value
   407)    real(R8),parameter :: g      = shr_const_g        ! gravity
   408)    real(R8),parameter :: cpdair = shr_const_cpdair   ! spec heat of dry air
   409)    real(R8),parameter :: cpvir  = shr_const_cpvir    ! cpwv/cpdair - 1.0
   410)    real(R8),parameter :: zvir   = shr_const_zvir     ! rh2o/rair   - 1.0
   411)    real(R8),parameter :: latvap = shr_const_latvap   ! latent heat of evap
   412)    real(R8),parameter :: latice = shr_const_latice   ! latent heat of fusion
   413)    real(R8),parameter :: stebol = shr_const_stebol   ! Stefan-Boltzmann
   414)    real(R8),parameter :: karman = shr_const_karman   ! Von Karman constant
   415)    real(R8),parameter :: zzsice = 0.0005_R8          ! ice surface roughness
   416) 
   417)    !--- local variables --------------------------------
   418)    integer(IN) :: lsize  ! array dimensions
   419)    integer(IN) :: n      ! array indicies
   420)    real(R8)    :: vmag   ! surface wind magnitude   (m/s)
   421)    real(R8)    :: thvbot ! virtual temperature      (K)
   422)    real(R8)    :: ssq    ! sea surface humidity     (kg/kg)
   423)    real(R8)    :: dssqdt ! derivative of ssq wrt Ts (kg/kg/K)
   424)    real(R8)    :: delt   ! potential T difference   (K)
   425)    real(R8)    :: delq   ! humidity difference      (kg/kg)
   426)    real(R8)    :: stable ! stability factor
   427)    real(R8)    :: rdn    ! sqrt of neutral exchange coefficient (momentum)
   428)    real(R8)    :: rhn    ! sqrt of neutral exchange coefficient (heat)
   429)    real(R8)    :: ren    ! sqrt of neutral exchange coefficient (water)
   430)    real(R8)    :: rd     ! sqrt of exchange coefficient (momentum)
   431)    real(R8)    :: rh     ! sqrt of exchange coefficient (heat)
   432)    real(R8)    :: re     ! sqrt of exchange coefficient (water)      
   433)    real(R8)    :: ustar  ! ustar
   434)    real(R8)    :: qstar  ! qstar
   435)    real(R8)    :: tstar  ! tstar
   436)    real(R8)    :: hol    ! H (at zbot) over L
   437)    real(R8)    :: xsq    ! temporary variable
   438)    real(R8)    :: xqq    ! temporary variable
   439)    real(R8)    :: psimh  ! stability function at zbot (momentum)
   440)    real(R8)    :: psixh  ! stability function at zbot (heat and water)
   441)    real(R8)    :: alz    ! ln(zbot/z10)
   442)    real(R8)    :: ltheat ! latent heat for surface
   443)    real(R8)    :: tau    ! stress at zbot
   444)    real(R8)    :: cp     ! specific heat of moist air
   445) 
   446)    real(R8)    :: bn     ! exchange coef funct for interpolation
   447)    real(R8)    :: bh     ! exchange coef funct for interpolation
   448)    real(R8)    :: fac    ! interpolation factor
   449)    real(R8)    :: ln0    ! log factor for interpolation
   450)    real(R8)    :: ln3    ! log factor for interpolation
   451) 
   452)    !--- local functions --------------------------------
   453)    real(R8)   :: Tk      ! temperature (K)
   454)    real(R8)   :: qsat    ! the saturation humididty of air (kg/m^3)
   455)    real(R8)   :: dqsatdt ! derivivative of qsat wrt surface temperature
   456)    real(R8)   :: xd      ! dummy argument  
   457)    real(R8)   :: psimhu  ! unstable part of psimh
   458)    real(R8)   :: psixhu  ! unstable part of psimx
   459) 
   460)    qsat(Tk)    = 627572.4_R8 / exp(5107.4_R8/Tk)
   461)    dqsatdt(Tk) = (5107.4_R8 / Tk**2) * 627572.4_R8 / exp(5107.4_R8/Tk)
   462)    psimhu(xd)  = log((1.0_R8+xd*(2.0_R8+xd))*(1.0_R8+xd*xd)/8.0_R8) - 2.0_R8*atan(xd) + 1.571_R8
   463)    psixhu(xd)  =  2.0_R8 * log((1.0_R8 + xd*xd)/2.0_R8)
   464) 
   465)    !--- formats ----------------------------------------
   466)    character(*),parameter :: subName =  "(shr_flux_atmIce) "
   467) 
   468) !-------------------------------------------------------------------------------
   469) ! PURPOSE:
   470) !   using atm & ice state variables, compute atm/ice fluxes 
   471) !   and diagnostic 10m air temperature and humidity
   472) !
   473) ! NOTE: 
   474) !   o all fluxes are positive downward
   475) !   o net heat flux = net sw + lw up + lw down + sen + lat
   476) !   o here, tstar = <WT>/U*, and qstar = <WQ>/U*.
   477) !   o wind speeds should all be above a minimum speed (eg. 1.0 m/s)
   478) ! 
   479) ! ASSUME:
   480) !   o The saturation humidity of air at T(K): qsat(T)  (kg/m^3)
   481) !-------------------------------------------------------------------------------
   482) 
   483)    lsize = size(tbot)
   484) 
   485)    do n = 1,lsize
   486) 
   487)      if (mask(n) == 0) then
   488)         sen  (n) = spval
   489)         lat  (n) = spval
   490)         lwup (n) = spval
   491)         evap (n) = spval
   492)         taux (n) = spval
   493)         tauy (n) = spval
   494)         tref (n) = spval
   495)         qref (n) = spval
   496)      else
   497)         !--- define some needed variables ---
   498)         vmag   = max(umin, sqrt(ubot(n)**2+vbot(n)**2))
   499)         thvbot = thbot(n)*(1.0_R8 + zvir * qbot(n)) ! virtual pot temp (K)
   500)          ssq   =  qsat  (ts(n)) / rbot(n)           ! sea surf hum (kg/kg)
   501)         dssqdt = dqsatdt(ts(n)) / rbot(n)           ! deriv of ssq wrt Ts 
   502)         delt   = thbot(n) - ts(n)                   ! pot temp diff (K)
   503)         delq   = qbot(n) - ssq                        ! spec hum dif (kg/kg)
   504)         alz    = log(zbot(n)/zref) 
   505)         cp     = cpdair*(1.0_R8 + cpvir*ssq) 
   506)         ltheat = latvap + latice
   507) 
   508)         !----------------------------------------------------------
   509)         ! first estimate of Z/L and ustar, tstar and qstar
   510)         !----------------------------------------------------------
   511) 
   512)         !--- neutral coefficients, z/L = 0.0 ---
   513)         rdn = karman/log(zref/zzsice)
   514)         rhn = rdn
   515)         ren = rdn
   516) 
   517)         !--- ustar,tstar,qstar ----
   518)         ustar = rdn * vmag
   519)         tstar = rhn * delt  
   520)         qstar = ren * delq  
   521) 
   522)         !--- compute stability & evaluate all stability functions ---
   523)         hol    = karman * g * zbot(n) &
   524)         &     * (tstar/thvbot+qstar/(1.0_R8/zvir+qbot(n))) / ustar**2
   525)         hol    = sign( min(abs(hol),10.0_R8), hol )
   526)         stable = 0.5_R8 + sign(0.5_R8 , hol)
   527)         xsq    = max(sqrt(abs(1.0_R8 - 16.0_R8*hol)) , 1.0_R8)
   528)         xqq    = sqrt(xsq)
   529)         psimh  = -5.0_R8*hol*stable + (1.0_R8-stable)*psimhu(xqq)
   530)         psixh  = -5.0_R8*hol*stable + (1.0_R8-stable)*psixhu(xqq)
   531) 
   532)         !--- shift all coeffs to measurement height and stability ---
   533)         rd = rdn / (1.0_R8+rdn/karman*(alz-psimh))
   534)         rh = rhn / (1.0_R8+rhn/karman*(alz-psixh))
   535)         re = ren / (1.0_R8+ren/karman*(alz-psixh))
   536) 
   537)         !--- update ustar, tstar, qstar w/ updated, shifted coeffs --
   538)         ustar = rd * vmag 
   539)         tstar = rh * delt 
   540)         qstar = re * delq 
   541) 
   542)         !----------------------------------------------------------
   543)         ! iterate to converge on Z/L, ustar, tstar and qstar
   544)         !----------------------------------------------------------
   545) 
   546)         !--- compute stability & evaluate all stability functions ---
   547)         hol    = karman * g * zbot(n) &
   548)         &      * (tstar/thvbot+qstar/(1.0_R8/zvir+qbot(n))) / ustar**2
   549)         hol    = sign( min(abs(hol),10.0_R8), hol )
   550)         stable = 0.5_R8 + sign(0.5_R8 , hol)
   551)         xsq    = max(sqrt(abs(1.0_R8 - 16.0_R8*hol)) , 1.0_R8)
   552)         xqq    = sqrt(xsq)
   553)         psimh  = -5.0_R8*hol*stable + (1.0_R8-stable)*psimhu(xqq)
   554)         psixh  = -5.0_R8*hol*stable + (1.0_R8-stable)*psixhu(xqq)
   555) 
   556)         !--- shift all coeffs to measurement height and stability ---
   557)         rd = rdn / (1.0_R8+rdn/karman*(alz-psimh)) 
   558)         rh = rhn / (1.0_R8+rhn/karman*(alz-psixh)) 
   559)         re = ren / (1.0_R8+ren/karman*(alz-psixh)) 
   560) 
   561)         !--- update ustar, tstar, qstar w/ updated, shifted coeffs --
   562)         ustar = rd * vmag 
   563)         tstar = rh * delt 
   564)         qstar = re * delq 
   565) 
   566)         !----------------------------------------------------------
   567)         ! compute the fluxes
   568)         !----------------------------------------------------------
   569) 
   570)         tau = rbot(n) * ustar * ustar 
   571)     
   572)         !--- momentum flux ---
   573)         taux(n) = tau * ubot(n) / vmag 
   574)         tauy(n) = tau * vbot(n) / vmag 
   575)      
   576)         !--- heat flux ---
   577)         sen (n) =   cp * tau * tstar / ustar 
   578)         lat (n) =  ltheat * tau * qstar / ustar
   579)         lwup(n) = -stebol * ts(n)**4 
   580)      
   581)         !--- water flux ---
   582)         evap(n) = lat(n)/ltheat 
   583) 
   584)         !----------------------------------------------------------
   585)         ! compute diagnostic: 2m reference height temperature
   586)         !----------------------------------------------------------
   587) 
   588)         !--- Compute function of exchange coefficients. Assume that 
   589)         !--- cn = rdn*rdn, cm=rd*rd and ch=rh*rd, and therefore 
   590)         !--- 1/sqrt(cn(n))=1/rdn and sqrt(cm(n))/ch(n)=1/rh 
   591)         bn = karman/rdn
   592)         bh = karman/rh
   593) 
   594)         !--- Interpolation factor for stable and unstable cases
   595)         ln0 = log(1.0_R8 + (ztref/zbot(n))*(exp(bn) - 1.0_R8))
   596)         ln3 = log(1.0_R8 + (ztref/zbot(n))*(exp(bn - bh) - 1.0_R8))
   597)         fac = (ln0 - ztref/zbot(n)*(bn - bh))/bh * stable &
   598)         &   + (ln0 - ln3)/bh * (1.0_R8-stable)
   599)         fac = min(max(fac,0.0_R8),1.0_R8)
   600) 
   601)         !--- actual interpolation
   602)         tref(n) = ts(n) + (tbot(n) - ts(n))*fac
   603)         qref(n) = qbot(n) - delq*fac
   604) 
   605)      endif
   606)    enddo

   631)

   636)    real(R8)   ,intent(inout)        :: arg2   ! scales: zkB   (in)  funct: phim (out)

   646)    real(R8)           :: zkB   ! (height)*(von Karman)*(surface bouyancy flux)
   647)    real(R8)           :: phim  ! momentum    gradient function or scale
   648)    real(R8)           :: phis  ! temperature gradient function or scale
   649)    real(R8)           :: psim  ! momentum    integral function or scale

   654)    real(R8),parameter :: uStarMin = 0.001_R8 ! lower bound on uStar
   655)    real(R8),parameter :: a = 1.000_R8  ! constant from Holtslag & de Bruin, equation 12
   656)    real(R8),parameter :: b = 0.667_R8  ! constant from Holtslag & de Bruin, equation 12
   657)    real(R8),parameter :: c = 5.000_R8  ! constant from Holtslag & de Bruin, equation 12
   658)    real(R8),parameter :: d = 0.350_R8  ! constant from Holtslag & de Bruin, equation 12
   659) 
   660)    !----- local variables, unstable case -----
   661)    real(R8),parameter :: a2 = 3.0_R8   ! constant from Wilson, equation 10 
   662) 
   663)    !----- formats -----
   664)    character(*),parameter :: subName = '(shr_flux_MOstability) '
   665)    character(*),parameter ::   F00 = "('(shr_flux_MOstability) ',4a)"
   666)    character(*),parameter ::   F01 = "('(shr_flux_MOstability) ',a,i5)"
   667) 
   668) !-------------------------------------------------------------------------------
   669) ! Notes::
   670) !   o this could be two routines, but are one to help keep them aligned
   671) !   o the stable calculation is taken from...
   672) !     A.A.M. HoltSlag and H.A.R. de Bruin, 1988:
   673) !     "Applied Modeling of the Nighttime Surface Energy Balance over Land",
   674) !     Journal of Applied Meteorology, Vol. 27, No. 6, June 1988, 659-704
   675) !   o the unstable calculation is taken from...
   676) !     D. Keith Wilson, 2001: "An Alternative Function for the Wind and 
   677) !     Temperature Gradients in Unstable Surface Layers", 
   678) !     Boundary-Layer Meteorology, 99 (2001), 151-158
   679) !-------------------------------------------------------------------------------
   680) 
   681)    !----- check for consistancy between option and arguments ------------------
   682)    if (debug > 1 .and. s_loglev > 0) then
   683)       if (debug > 2) write(s_logunit,F01) "enter, option = ",option
   684)       if ( option == shr_flux_MOwScales .and. present(arg5) ) then
   685)          write(s_logunit,F01) "ERROR: option1 must have four arguments"
   686)          call shr_sys_abort(subName//"option inconsistant with arguments")
   687)       else if ( option == shr_flux_MOfunctions .and. .not. present(arg5) ) then
   688)          write(s_logunit,F01) "ERROR: option2 must have five arguments"
   689)          call shr_sys_abort(subName//"option inconsistant with arguments")
   690)       else 
   691)          write(s_logunit,F01) "invalid option = ",option
   692)          call shr_sys_abort(subName//"invalid option")
   693)       end if
   694)    end if
   695) 
   696)    !------ velocity scales option ----------------------------------------------
   697)    if (option == shr_flux_MOwScales) then

   699)       !--- input ---
   700)       uStar = arg1
   701)       zkB   = arg2
   702) 
   703)       if (zkB >= 0.0_R8) then ! ----- stable -----
   704)          zeta = zkB/(max(uStar,uStarMin)**3)
   705)          temp = exp(-d*zeta)
   706)          phim = uStar/(1.0_R8 + zeta*(a + b*(1.0_R8 + c - d*zeta)*temp))
   707)          phis = phim
   708)       else                    ! ----- unstable -----

   710)          phim = sqrt(uStar**2 + shr_flux_MOgammaM*temp)
   711)          phis = sqrt(uStar**2 + shr_flux_MOgammaS*temp)

   713) 
   714)       !--- output ---
   715)       arg3 = phim
   716)       arg4 = phis
   717)    !  arg5 = <unused>
   718) 
   719)    !------ stability function option -------------------------------------------
   720)    else if (option == shr_flux_MOfunctions) then
   721) 
   722)       !--- input ---
   723)       zeta  = arg1
   724) 
   725)       if (zeta >= 0.0_R8) then ! ----- stable -----

   727)          phim =        1.0_R8 + zeta*(a + b*(1.0_R8 + c - d*zeta)*temp)
   728)          phis = phim
   729)          psim = -a*zeta - b*(zeta - c/d)*temp - b*c/d
   730)          psis = psim 
   731)       else                    ! ----- unstable ----
   732)          temp = (zeta*zeta)**(1.0_R8/a2)   ! note: zeta < 0, zeta*zeta > 0
   733)          phim = 1.0_R8/sqrt(1.0_R8 + shr_flux_MOgammaM*temp)
   734)          phis = 1.0_R8/sqrt(1.0_R8 + shr_flux_MOgammaS*temp)
   735)          psim = a2*log(0.5_R8 + 0.5_R8/phim)
   736)          psis = a2*log(0.5_R8 + 0.5_R8/phis)
   737)       end if
   738) 
   739)       !--- output ---
   740)       arg2 = phim
   741)       arg3 = phis
   742)       arg4 = psim
   743)       arg5 = psis
   744)    !----------------------------------------------------------------------------
   745)    else 
   746)       write(s_logunit,F01) "invalid option = ",option
   747)       call shr_sys_abort(subName//"invalid option")
   748)    endif
