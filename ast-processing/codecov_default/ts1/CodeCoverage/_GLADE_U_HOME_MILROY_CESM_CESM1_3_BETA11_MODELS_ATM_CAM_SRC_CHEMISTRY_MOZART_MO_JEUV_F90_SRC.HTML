   325)         subroutine jeuv( nlev, zen, occ, o2cc, n2cc, &
   326)                          zkm, euv_prates)
   327) !==============================================================================
   328) !   Purpose:
   329) !      Calculate euv photolysis/ionization rates (in s-1)
   330) !==============================================================================
   331) !   Arguments:
   332) !       nlev: number of model vertical levels
   333) !       zen:  solar zenith angle in degree
   334) !       occ:  atmoic oxygen number density (#/cm3)
   335) !       o2cc: molecular oxygen number density (#/cm3)
   336) !       n2cc: molecular nitrogen number density (#/cm3)
   337) !       zkm:  altitude of model levels in KM
   338) !       euv_prates: array for EUV photolysis/ionization rates
   339) !==============================================================================
   340) !   Approach:
   341) !       call sphers 
   342) !            input: zenith angle
   343) !            output: dsdh and nid used in slant column routine
   344) !       call slant_col
   345) !            input: dsdh and nid, etc
   346) !            output: slant column density
   347) !       calculate photon production rates and photoelectron production rates
   348) !==============================================================================
   349) !  Photolysis/ionization considered in EUV calculation
   350) !
   351) !  O + hv --> O+ (4S) + e*                        ! J1
   352) !  O + hv --> O+ (2D) + e*                        ! J2
   353) !  O + hv --> O+ (2P) + e*                        ! J3
   354) !  N (4S) + hv --> N+ + e*                        ! J4
   355) !  O2 + hv --> O2+ + e*                           ! J5
   356) !  N2 + hv --> N2+ + e*                           ! J6
   357) !  O2 + hv --> O + O+(4S) + e*                    ! J7
   358) !  O2 + hv --> O + O+(2D) + e*                    ! J8
   359) !  O2 + hv --> O + O+(2P) + e*                    ! J9
   360) !  N2 + hv --> N (4S) + N+ + e*                   ! J10
   361) !  N2 + hv --> N (2D) + N+ + e*                   ! J11
   362) !  O2 + hv --> O (3P) + O (3P)                    ! J12
   363) !  N2 + hv --> N (4S) + N (2D)                    ! J13
   364) !
   365) !  O + e* --> O+ (4S) + e* + e                    ! J14
   366) !  O + e* --> O+ (2D) + e*+ e                     ! J15
   367) !  O + e* --> O+ (2P) + e*+ e                     ! J16
   368) !  O2 + e* --> O2+ + e*+ e                        ! J17
   369) !  N2 + e*--> N2+ + e*+ e                         ! J18
   370) !  O2 + e* --> O + O+(4S) + e*+ e                 ! J19
   371) !  O2 + e*--> O + O+(2D) + e*+ e                  ! J20
   372) !  O2 + e* --> O + O+(2P) + e*+ e                 ! J21
   373) !  N2 + e* --> N (4S) + N+ + e*+ e                ! J22
   374) !  N2 + e* --> N (2D) + N+ + e*+ e                ! J23
   375) !  O2 + e* --> O (3P) + O (3P) + e*               ! J24
   376) !  N2 + e* --> N (4S) + N (2D) + e*               ! J25
   377) !==============================================================================
   378) 
   379)         use euvac,       only : euvac_etf
   380)         use mo_jshort,   only : sphers, slant_col
   381)         use cam_history, only : outfld
   382) 
   383) 	implicit none
   384) 
   385) !------------------------------------------------------------------------------
   386) !       dummy arguments
   387) !------------------------------------------------------------------------------
   388) 	integer, intent(in)     :: nlev                        ! model vertical levels
   389) 	real(r8), intent(in)    :: zen	                       ! Zenith  angle in degree
   390)         real(r8), intent(in)    :: occ(nlev)                   ! atmic oxygen number density (#/cm3) 
   391) 	real(r8), intent(in)    :: o2cc(nlev)		       ! Molecular oxygen number density (#/cm3) 
   392) 	real(r8), intent(in)    :: n2cc(nlev)		       ! molecular nitrogen number density(#/cm3) 
   393) 	real(r8), intent(in)    :: zkm(nlev)		       ! Altitude, km,from top to bottom
   394) 	real(r8), intent(out)   :: euv_prates(:,:)             ! EUV photolysis/ionization rates (s-1)   
   395) 
   396) !------------------------------------------------------------------------------
   397) !       local variables
   398) !------------------------------------------------------------------------------
   399)         real(r8), parameter :: km2cm = 1.e5_r8
   400)         integer  :: l, k, m, n              ! loop indecies
   401)         real(r8) :: tau(lmax)               ! wavelength dependant optical depth
   402)         real(r8) :: delz(nlev)              ! layer thickness (cm)
   403) 	real(r8) :: scol(nlev,nmaj)         ! major species' (O,O2,N2) Slant Column Density
   404) 	real(r8) :: nflux(nlev,lmax)
   405) 	real(r8) :: wrk(nmaj)               ! temporary array for photoabsorption rate
   406) 	real(r8) :: absorp(nlev,lmax)       ! temporary array for photoabsorption rate
   407) 	real(r8) :: ioniz(nlev,lmax)        ! temporary array for photoionization rate
   408) 	real(r8) :: dsdh(0:nlev,nlev)       ! Slant path of direct beam through each layer 
   409) 	                                    ! crossed  when travelling from the top of the atmosphere 
   410) 				            ! to layer i; dsdh(i,j), i = 0..NZ-1, j = 1..NZ-1
   411)         integer :: nid(0:nlev)              ! Number of layers crossed by the direct
   412) 	                                    ! beam when travelling from the top of the 
   413)                                             ! atmosphere to layer i; NID(i), i = 0..NZ-1
   414)         real(r8) :: p_photon(nlev,nstat,nspecies)   !  photoionization/dissociation rates(s-1) (O,O2,N2,N)
   415)         real(r8) :: p_electron(nlev,nstat,nmaj) !  photoelectron ionization/dissociation rates(s-1) (O,O2,N2)
   416) 
   417)         real(r8) :: prates(nlev,neuv)
   418) !------------------------------------------------------------------------------
   419) ! zero arrays
   420) !------------------------------------------------------------------------------
   421)         p_photon(:,:,:)   = 0._r8
   422)         p_electron(:,:,:) = 0._r8
   423) 
   424)         call sphers( nlev, zkm, zen, dsdh, nid )
   425) 	delz(1:nlev-1) = km2cm*(zkm(1:nlev-1) - zkm(2:nlev))
   426) 	call slant_col( nlev, delz, dsdh, nid, occ, scol )
   427) 	call slant_col( nlev, delz, dsdh, nid, o2cc, scol(1,2) )
   428) 	call slant_col( nlev, delz, dsdh, nid, n2cc, scol(1,3) )
   429) 
   430) !------------------------------------------------------------------------------
   431) ! The calculation is in the order from model bottom to model top
   432) ! because scol array is in this order.
   433) !------------------------------------------------------------------------------
   434)         do k = 1,nlev
   435)            wrk(:) = scol(k,:)
   436)            tau(:) = matmul( sigabs(:,:nmaj),wrk )
   437)            where( tau(:) < 20._r8 )
   438) 	      nflux(k,:) = euvac_etf(:) * exp( -tau(:) )
   439)            elsewhere
   440) 	      nflux(k,:) = 0._r8
   441)            endwhere
   442)         end do
   443) 
   444) !------------------------------------------------------------------------------
   445) ! remember occ,o2cc and n2cc is from top to bottom
   446) !------------------------------------------------------------------------------
   447)         do m = 1,nspecies
   448)            do l = 1,lmax
   449)               absorp(:,l) = sigabs(l,m) * nflux(:,l)
   450)            end do
   451) 	   if( m <= nmaj ) then
   452)               do l = 1,lmax
   453)                  ioniz(:,l) = absorp(:,l) * branch_p(l,1,m)
   454)               end do
   455) 	      do n = 1,nstat
   456) 	         p_photon(:,n,m)   = matmul( absorp,branch_p(:,n,m) )
   457) 	         p_electron(:,n,m) = matmul( ioniz,branch_e(:,n,m) )
   458)               end do
   459)            else
   460)               p_photon(:,1,m) = matmul( nflux,sigabs(:,m) )
   461)            end if
   462)         end do
   463) 
   464) !------------------------------------------------------------------------------
   465) ! set photolysis/ionization rate for each reaction
   466) !------------------------------------------------------------------------------
   467)        prates(:,1)  = p_photon(:,2,1)  
   468)        prates(:,2)  = p_photon(:,3,1)
   469)        prates(:,3)  = p_photon(:,4,1)
   470)        prates(:,4)  = p_photon(:,1,4)
   471)        prates(:,5)  = p_photon(:,2,2) + p_photon(:,3,2)

   502) 			zkm, euv_hrates, kbot )
   503) !==============================================================================
   504) !   Purpose:
   505) !      Calculate euv photolysis heating rates
   506) !==============================================================================
   507) !   Arguments:
   508) !       nlev: number of model vertical levels
   509) !       zen:  solar zenith angle in degree
   510) !       occ:  atmoic oxygen number density (#/cm3)
   511) !       o2cc: molecular oxygen number density (#/cm3)
   512) !       n2cc: molecular nitrogen number density (#/cm3)

   522) !            output: slant column density
   523) !       calculate photon production rates and photoelectron production rates
   524) !==============================================================================
   525) !  Photolysis/ionization considered in EUV heating rate calculation
   526) !  O + hv --> O+ (4S) + e*                        ! J1
   527) !  O + hv --> O+ (2D) + e*                        ! J2
   528) !  O + hv --> O+ (2P) + e*                        ! J3
   529) !  N (4S) + hv --> N+ + e*                        ! J4
   530) !  O2 + hv --> O2+ + e*                           ! J5
   531) !  N2 + hv --> N2+ + e*                           ! J6
   532) !  O2 + hv --> O + O+(4S) + e*                    ! J7
   533) !  O2 + hv --> O + O+(2D) + e*                    ! J8
   534) !  O2 + hv --> O + O+(2P) + e*                    ! J9
   535) !  N2 + hv --> N (4S) + N+ + e*                   ! J10
   536) !  N2 + hv --> N (2D) + N+ + e*                   ! J11
   537) !  O2 + hv --> O (3P) + O (3P)                    ! J12
   538) !  N2 + hv --> N (4S) + N (2D)                    ! J13
   539) !==============================================================================
   540) 
   541)         use euvac,         only : euvac_etf
   542)         use mo_jshort,     only : sphers, slant_col
   543)         use physconst,     only : avogad
   544) 
   545) 	implicit none
   546) 
   547) !------------------------------------------------------------------------------
   548) !       dummy arguments
   549) !------------------------------------------------------------------------------
   550) 	integer, intent(in)     :: nlev                        ! model vertical levels
   551) 	integer, intent(in)     :: kbot                        ! heating vertical levels
   552) 	real(r8), intent(in)    :: zen	                       ! zenith  angle (degrees)
   553)         real(r8), intent(in)    :: occ(nlev)                   ! atomic oxygen number density (molecules/cm^3)
   554) 	real(r8), intent(in)    :: o2cc(nlev)		       ! molecular oxygen concentration (molecules/cm^3)
   555) 	real(r8), intent(in)    :: n2cc(nlev)		       ! molecular nitrogen concentration (molecules/cm^3)
   556)         real(r8), intent(in)    :: o_vmr(nlev)                 ! atomic oxygen concentration (mol/mol)
   557) 	real(r8), intent(in)    :: o2_vmr(nlev)		       ! molecular oxygen concentration (mol/mol)
   558) 	real(r8), intent(in)    :: n2_vmr(nlev)		       ! molecular nitrogen concentration (mol/mol)
   559) 	real(r8), intent(in)    :: zkm(nlev)		       ! midpoint geopotential (km)
   560) 	real(r8), intent(in)    :: cparg(nlev)		       ! specific heat capacity
   561) 	real(r8), intent(in)    :: mw(nlev)		       ! atm mean mass
   562) 	real(r8), intent(out)   :: euv_hrates(:)               ! euv heating rates
   563) 
   564) !------------------------------------------------------------------------------
   565) !       local variables
   566) !------------------------------------------------------------------------------
   567)         real(r8), parameter :: km2cm = 1.e5_r8
   568)         integer  :: l, k, k1, m, n          ! indicies
   569)         real(r8) :: tau(lmax)               ! wavelength dependant optical depth
   570)         real(r8) :: delz(nlev)              ! layer thickness (cm)
   571)         real(r8) :: hfactor(kbot)
   572) 	real(r8) :: scol(nlev,nmaj)         ! major species' (O,O2,N2) Slant Column Density
   573) 	real(r8) :: nflux(kbot,lmax)

   575) 	real(r8) :: wrk(nmaj)               ! temporary array for photoabsorption rate
   576) 	real(r8) :: absorp(kbot,lmax)       ! temporary array for photoabsorption rate
   577) 	real(r8) :: ioniz(kbot,lmax)        ! temporary array for photoionization rate
   578) 	real(r8) :: dsdh(0:nlev,nlev)       ! Slant path of direct beam through each layer 
   579) 	                                    ! crossed  when travelling from the top of the atmosphere 
   580) 				            ! to layer i; dsdh(i,j), i = 0..NZ-1, j = 1..NZ-1
   581)         integer :: nid(0:nlev)              ! Number of layers crossed by the direct
   582) 	                                    ! beam when travelling from the top of the 
   583)                                             ! atmosphere to layer i; NID(i), i = 0..NZ-1

   585)

   593) 	call slant_col( nlev, delz, dsdh, nid, occ, scol )
   594) 	call slant_col( nlev, delz, dsdh, nid, o2cc, scol(1,2) )

   596)

   599) ! because scol array is in this order.
   600) !------------------------------------------------------------------------------

   602)            k1 = nlev - kbot + k

   605)            where( tau(:) < 20._r8 )
   606) 	      nflux(k,:) = energy(:) * euvac_etf(:) * exp( -tau(:) )
   607)            elsewhere
   608) 	      nflux(k,:) = 0._r8
   609)            endwhere
   610)         end do

   617)               absorp(:,l) = sigabs(l,m) * nflux(:,l)
   618)            end do

   621) 	         p_photon(:,n,m) = matmul( absorp,branch_p(:,n,m) )
   622)               end do
   623)            else

   625)            end if
   626)         end do

   630) !------------------------------------------------------------------------------
   631)        prates(:,1)  = p_photon(:,2,1)
   632)        prates(:,2)  = p_photon(:,3,1)
   633)        prates(:,3)  = p_photon(:,4,1)

   637)        prates(:,8)  = .24_r8*p_photon(:,4,2)

   639)        prates(:,10) = .2_r8*p_photon(:,4,3)
   640)        prates(:,11) = .8_r8*p_photon(:,4,3)
   641)        prates(:,12) = p_photon(:,5,2)
   642)        prates(:,13) = p_photon(:,5,3)
   643)        hfactor(:)   = avogad/(cparg(:kbot)*mw(:kbot))
   644)        euv_hrates(kbot+1:nlev) = 0._r8
   645)        euv_hrates(:kbot) = ((prates(kbot:1:-1,1) + prates(kbot:1:-1,2) + prates(kbot:1:-1,3))*o_vmr(:kbot) &
   646) 		     + (prates(kbot:1:-1,5) + prates(kbot:1:-1,7) + prates(kbot:1:-1,8) &
   647) 			+ prates(kbot:1:-1,9) + prates(kbot:1:-1,12))*o2_vmr(:kbot) &
