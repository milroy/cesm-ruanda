    86)   subroutine radiation_register
    87) !-----------------------------------------------------------------------
    88) ! 
    89) ! Register radiation fields in the physics buffer
    90) !
    91) !-----------------------------------------------------------------------
    92) 
    93)     use physics_buffer, only: pbuf_add_field, dtype_r8

    96)     call pbuf_add_field('QRS' , 'global',dtype_r8,(/pcols,pver/), qrs_idx) ! shortwave radiative heating rate 
    97)     call pbuf_add_field('QRL' , 'global',dtype_r8,(/pcols,pver/), qrl_idx) ! longwave  radiative heating rate

   101)     if (spectralflux) then
   102)       call pbuf_add_field('SU'  , 'global',dtype_r8,(/pcols,pverp,nswbands/), su_idx) ! shortwave upward flux (per band)
   103)       call pbuf_add_field('SD'  , 'global',dtype_r8,(/pcols,pverp,nswbands/), sd_idx) ! shortwave downward flux (per band)
   104)       call pbuf_add_field('LU'  , 'global',dtype_r8,(/pcols,pverp,nlwbands/), lu_idx) ! longwave upward flux (per band)
   105)       call pbuf_add_field('LD'  , 'global',dtype_r8,(/pcols,pverp,nlwbands/), ld_idx) ! longwave downward flux (per band)

   218) function radiation_do(op, timestep)
   219) !----------------------------------------------------------------------- 
   220) ! Purpose: Returns true if the specified operation is done this timestep.
   221) !-----------------------------------------------------------------------
   222) 
   223)    character(len=*), intent(in) :: op             ! name of operation
   224)    integer, intent(in), optional:: timestep
   225)    logical                      :: radiation_do   ! return value
   226) 
   227)    ! Local variables
   228)    integer :: nstep             ! current timestep number
   229)    !-----------------------------------------------------------------------
   230) 
   231)    if (present(timestep)) then
   232)       nstep = timestep
   233)    else
   234)       nstep = get_nstep()
   235)    end if
   236) 
   237)    select case (op)
   238) 
   239)    case ('sw') ! do a shortwave heating calc this timestep?
   240)       radiation_do = nstep == 0  .or.  iradsw == 1                     &
   241)                     .or. (mod(nstep-1,iradsw) == 0  .and.  nstep /= 1) &
   242)                     .or. nstep <= irad_always
   243) 
   244)    case ('lw') ! do a longwave heating calc this timestep?
   245)       radiation_do = nstep == 0  .or.  iradlw == 1                     &
   246)                     .or. (mod(nstep-1,iradlw) == 0  .and.  nstep /= 1) &
   247)                     .or. nstep <= irad_always
   248) 
   249)    case ('aeres') ! write absorptivity/emissivity to restart file this timestep?
   250)       ! for RRTMG there is no abs/ems restart file
   251)       radiation_do = .false.
   252)          
   253)    case default
   254)       call endrun('radiation_do: unknown operation:'//op)
   255) 
   256)    end select
   257) end function radiation_do

   261) real(r8) function radiation_nextsw_cday()
   262)   
   263) !----------------------------------------------------------------------- 
   264) ! Purpose: Returns calendar day of next sw radiation calculation
   265) !-----------------------------------------------------------------------
   266) 
   267)    use time_manager, only: get_curr_calday, get_nstep, get_step_size
   268) 
   269)    ! Local variables
   270)    integer :: nstep      ! timestep counter
   271)    logical :: dosw       ! true => do shosrtwave calc   
   272)    integer :: offset     ! offset for calendar day calculation
   273)    integer :: dTime      ! integer timestep size
   274)    real(r8):: calday     ! calendar day of 
   275)    !-----------------------------------------------------------------------
   276) 
   277)    radiation_nextsw_cday = -1._r8
   278)    dosw   = .false.
   279)    nstep  = get_nstep()
   280)    dtime  = get_step_size()
   281)    offset = 0
   282)    do while (.not. dosw)
   283)       nstep = nstep + 1
   284)       offset = offset + dtime
   285)       if (radiation_do('sw', nstep)) then
   286)          radiation_nextsw_cday = get_curr_calday(offset=offset) 
   287)          dosw = .true.
   288)       end if
   289)    end do
   290)    if(radiation_nextsw_cday == -1._r8) then
   291)       call endrun('error in radiation_nextsw_cday')
   292)    end if
   293)         
   294) end function radiation_nextsw_cday
