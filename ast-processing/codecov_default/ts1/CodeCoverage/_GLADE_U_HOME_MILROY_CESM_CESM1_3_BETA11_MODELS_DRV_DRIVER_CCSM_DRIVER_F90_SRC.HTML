     1) program ccsm_driver
     2) 
     3) !-------------------------------------------------------------------------------
     4) !
     5) ! Purpose: Main program for NCAR CCSM4/cpl7. Can have different
     6) !          land, sea-ice, and ocean models plugged in at compile-time.
     7) !          These models can be either: stub, dead, data, or active
     8) !          components or some combination of the above.
     9) !
    10) !               stub -------- Do nothing.
    11) !               dead -------- Send analytic data back.
    12) !               data -------- Send data back interpolated from input files.
    13) !               active ------ Prognostically simulate the given component.
    14) !
    15) ! Method: Call appropriate initialization, run (time-stepping), and 
    16) !         finalization routines.
    17) ! 
    18) !-------------------------------------------------------------------------------
    19) 
    20)    !----------------------------------------------------------------------------
    21)    ! share code & libs
    22)    !----------------------------------------------------------------------------
    23)    use shr_sys_mod,       only: shr_sys_abort
    24)    use perf_mod
    25)    use ESMF
    26)    use ccsm_comp_mod
    27) 
    28)    implicit none
    29) 
    30)    !--------------------------------------------------------------------------
    31)    ! Local Variables
    32)    !--------------------------------------------------------------------------
    33)    integer                    :: localrc
    34) #ifdef USE_ESMF_LIB
    35)    character(len=ESMF_MAXSTR) :: compName
    36)    type(ESMF_CplComp)         :: drvcomp
    37) #endif
    38) 
    39) 
    40)    !--------------------------------------------------------------------------
    41)    ! Setup and initialize the communications and logging.  
    42)    !--------------------------------------------------------------------------
    43)    call ccsm_pre_init1()
    44) 
    45)    !--------------------------------------------------------------------------
    46)    ! Initialize ESMF.  This is done outside of the ESMF_INTERFACE ifdef
    47)    ! because it is needed for the time manager, even if the ESMF_INTERFACE
    48)    ! is not used.
    49)    !--------------------------------------------------------------------------
    50)    call ESMF_Initialize()
    51) 
    52)    !--------------------------------------------------------------------------
    53)    ! Read in the configuration information and initialize the time manager.
    54)    !--------------------------------------------------------------------------
    55)    call ccsm_pre_init2()
    56) 
    57) #ifdef USE_ESMF_LIB
    58) 
    59)    !--------------------------------------------------------------------------
    60)    ! Create the "Cap" component and set the services using the register
    61)    ! routine.  Setting the services is where the initialize, run and 
    62)    ! finalize routines for this component are set.
    63)    !--------------------------------------------------------------------------
    64)    compName = "CESM_Component"
    65)    drvcomp = ESMF_CplCompCreate(name=compName, rc=localrc)
    66)    if (localrc /= 0) call shr_sys_abort('failed to create CESM Component')
    67) 
    68)    call ESMF_CplCompSetServices(drvcomp, userRoutine=ccsm_comp_register, &
    69)         rc=localrc)
    70)    if (localrc /= 0) call shr_sys_abort('failed to set services for CESM Comp')
    71) 
    72)    !--------------------------------------------------------------------------
    73)    ! Call the initialize, run and finalize routines registered with the
    74)    ! cap component.
    75)    !--------------------------------------------------------------------------
    76) 
    77)    call ESMF_CplCompInitialize(drvcomp, rc=localrc)
    78)    if (localrc /= 0) call shr_sys_abort('failed to esmf initialize')
    79) 
    80)    call ESMF_CplCompRun(drvcomp, rc=localrc)
    81)    if (localrc /= 0) call shr_sys_abort('failed to esmf run')
    82) 
    83)    call ESMF_CplCompFinalize(drvcomp, rc=localrc)
    84)    if (localrc /= 0) call shr_sys_abort('failed to esmf finalize')
    85) 
    86) #else
    87) 
    88)    !--------------------------------------------------------------------------
    89)    ! If ESMF is not defined, then just call the initialize, run and finalize
    90)    ! routines directly.
    91)    !--------------------------------------------------------------------------
    92)    call ccsm_init()
    93)    call ccsm_run()
    94)    call ccsm_final()
    95) 
    96) #endif
    97) 
    98)    !--------------------------------------------------------------------------
    99)    ! Clean-up
   100)    !--------------------------------------------------------------------------
   101)    call ESMF_Finalize( )
   102) 
   103) 
   104) end program ccsm_driver
