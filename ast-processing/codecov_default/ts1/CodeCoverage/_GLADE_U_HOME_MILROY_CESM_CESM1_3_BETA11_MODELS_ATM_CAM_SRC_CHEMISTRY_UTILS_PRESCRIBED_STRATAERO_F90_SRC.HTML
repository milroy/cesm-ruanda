   206)   subroutine prescribed_strataero_adv( state, pbuf2d)
   207) 
   208)     use tracer_data,  only : advance_trcdata
   209)     use physics_types,only : physics_state
   210)     use ppgrid,       only : begchunk, endchunk
   211)     use ppgrid,       only : pcols, pver
   212)     use string_utils, only : to_lower, GLC
   213)     use cam_history,  only : outfld
   214)     use physconst,    only : mwdry                ! molecular weight dry air ~ kg/kmole
   215)     use physconst,    only : boltz, gravit        ! J/K/molecule
   216)     use tropopause,   only : tropopause_find, TROP_ALG_TWMO, TROP_ALG_CLIMATE
   217)     
   218)     use physics_buffer, only : physics_buffer_desc, pbuf_get_field, pbuf_get_chunk
   219)     
   220)     implicit none
   221) 
   222)     type(physics_state), intent(in)    :: state(begchunk:endchunk)                 
   223)     
   224)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   225) 
   226)     type(physics_buffer_desc), pointer :: pbuf_chnk(:)
   227) 
   228)     integer :: c,ncol,i,k
   229)     real(r8) :: to_mmr(pcols,pver)
   230)     real(r8), parameter :: molmass = 4._r8/3._r8*98.0_r8 !convert dry mass to wet mass of h2so4 
   231)     real(r8) :: ptrop
   232)     real(r8) :: concvolc ! micrograms of wetted aerosol per cubic centimeter
   233)     real(r8) :: volcmass(pcols,pver)
   234)     real(r8) :: columnmass(pcols)
   235)     real(r8) :: mmrvolc
   236)     integer  :: tropLev(pcols)
   237)     real(r8) :: area_fact
   238) 
   239)     real(r8) :: outdata(pcols,pver)
   240)     real(r8), pointer :: mass(:,:)
   241)     real(r8), pointer :: area(:,:)
   242)     real(r8), pointer :: radius(:,:)
   243) 
   244)     !WACCM-derived relation between mass concentration and wet aerosol radius in meters
   245)     real(r8),parameter :: radius_conversion = 1.9e-4_r8
   246) 
   247)     if( .not. has_prescribed_strataero ) return
   248) 
   249)     call advance_trcdata( fields, file, state, pbuf2d )
   250) 
   251)     ! copy prescribed tracer fields into state svariable with the correct units
   252)     do c = begchunk,endchunk
   253) 
   254)        pbuf_chnk => pbuf_get_chunk(pbuf2d, c)
   255) 
   256)        ncol = state(c)%ncol
   257) 
   258)        select case ( to_lower(trim(fields(1)%units(:GLC(fields(1)%units)))) )
   259)        case ("molecules/cm3air", "molec/cm3","/cm3","molecules/cm3","cm^-3","cm**-3")
   260)           to_mmr(:ncol,:) = (molmass*1.e6_r8*boltz*state(c)%t(:ncol,:))/(mwdry*state(c)%pmiddry(:ncol,:))
   261)        case ('kg/kg','mmr','kg kg-1')
   262)           to_mmr(:ncol,:) = 1._r8
   263)        case ('mol/mol','mole/mole','vmr','fraction')
   264)           to_mmr(:ncol,:) = molmass/mwdry
   265)        case default
   266)           write(iulog,*) 'prescribed_strataero_adv: mass units = ',trim(fields(1)%units) ,' are not recognized'
   267)           call endrun('prescribed_strataero_adv: mass units are not recognized')
   268)        end select
   269) 
   270)        call pbuf_get_field(pbuf_chnk, mmr_ndx, mass)
   271) 
   272)        call outfld( dens_name,         mass(:,:),     pcols, state(c)%lchnk)
   273) 
   274)        mass(:ncol,:) = to_mmr(:ncol,:) * mass(:ncol,:) ! mmr
   275) 
   276)        call pbuf_get_field(pbuf_chnk, rad_ndx, radius)
   277)        call pbuf_get_field(pbuf_chnk, sad_ndx, area)
   278) 
   279)        select case ( to_lower(trim(fields(3)%units(:7))) )
   280)        case ("um2/cm3")
   281)           area_fact = 1.e-8_r8
   282)        case default
   283)           write(iulog,*) 'prescribed_strataero_adv: surface area density units = ',trim(fields(3)%units) ,' are not recognized'
   284)           call endrun('prescribed_strataero_adv: surface area density are not recognized')
   285)        end select
   286)        area(:ncol,:) = area_fact*area(:ncol,:)
   287) 
   288)        ! this definition of tropopause is consistent with what is used in chemistry
   289)        call tropopause_find(state(c), tropLev, primary=TROP_ALG_TWMO, backup=TROP_ALG_CLIMATE)
   290) 
   291)        do i = 1,ncol
   292)           do k = 1,pver
   293)              ! set to zero below tropopause
   294)              if ( k >= tropLev(i) ) then
   295)                 mass(i,k) = 0._r8
   296)                 radius(i,k) = 0._r8
   297)                 area(i,k) = 0._r8
   298)              endif
   299)           enddo
   300)        enddo
   301) 
   302)        volcmass(:ncol,:) = mass(:ncol,:)*state(c)%pdel(:ncol,:)/gravit
   303)        columnmass(:ncol) = sum(volcmass(:ncol,:), 2)
   304) 
   305)        call outfld( mmr_name,         mass(:,:),     pcols, state(c)%lchnk)
   306)        call outfld( mass_name,        volcmass(:,:), pcols, state(c)%lchnk)
   307)        call outfld( mass_column_name, columnmass(:), pcols, state(c)%lchnk)
   308)        call outfld( rad_name,         radius(:,:),   pcols, state(c)%lchnk)
   309)        call outfld( sad_name,         area(:,:),     pcols, state(c)%lchnk)
   310) 
   311)     enddo
   312) 
   313)   end subroutine prescribed_strataero_adv
