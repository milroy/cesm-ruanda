   240)   !

   242)   ! on unit sphere. Note: spherical longitude is [0,2\pi]

   247)   pure function cart_to_spherical(cart) result (sphere)
   248)     use physical_constants, only : dd_pi
   249)     implicit none
   250)     type(cartesian3D_t), intent(in) :: cart         
   251)     type(spherical_polar_t)         :: sphere
   252) 
   253)     sphere%r=distance(cart)
   254)     sphere%lat=ASIN(cart%z/sphere%r)
   255)     sphere%lon=0
   256) 
   257)     ! ==========================================================
   258)     ! enforce three facts:
   259)     !
   260)     ! 1) lon at poles is defined to be zero
   261)     ! 
   262)     ! 2) Grid points must be separated by about .01 Meter (on earth)
   263)     !    from pole to be considered "not the pole".
   264)     !
   265)     ! 3) range of lon is { 0<= lon < 2*pi }
   266)     !
   267)     ! ==========================================================
   268) 
   269) !   if point is away from the POLE.  distance(cart) = distance from center of earth,
   270) !   so this was a bug:
   271) !    if (distance(cart) >= DIST_THRESHOLD) then 
   272) 
   273)     if ( abs(abs(sphere%lat)-DD_PI/2)  >= DIST_THRESHOLD ) then
   274)        sphere%lon=ATAN2(cart%y,cart%x)
   275)        if (sphere%lon<0) then
   276)           sphere%lon=sphere%lon + 2*DD_PI
   277)        end if
   278)     end if
   279)

   307)   function unit_face_based_cube_to_unit_sphere(cart, face_no) result(sphere)
   308) 
   309) ! Note: Output spherical longitude is [-pi,pi]
   310) 
   311) ! Project from a UNIT cube to a UNIT sphere.  ie, the lenght of the cube edge is 2.
   312) ! Face 1 of the cube touches the sphere at longitude, latitude (0,0). The negative 
   313) ! x axis is negative longitude (ie. going west is negative), the positive x axis
   314) ! is increasing longitude.  Face 1 maps the Face 1 to the lat,lon on the sphere:
   315) !    [-1,1] x [-1,1] => [-\pi/4,\pi/4] x [-\pi/4, \pi/4]
   316) 
   317) ! Face 2 continues with increasing longitude (ie to the east of Face 1).  
   318) ! The left edge of Face 2 (negative x) is the right edge of Face 1 (positive x)
   319) ! The latitude is the same as Face 1, but the longitude increases:
   320) !    [-1,1] x [-1,1] => [\pi/4, 3\pi/4] x [-\pi/4, \pi/4]
   321) 
   322) ! Face 3 continues with increasing longitude (ie to the east of Face 2).  
   323) ! Face 3 is like Face 1, but the x coordinates are reversed, ie. decreasing x 
   324) ! is increasing longitude:
   325) !    [-1,1] x [-1,1]  =    [-1,0] x [-1,1] U  [0,1] x [-1,1] =>
   326) !            [3\pi/4,\pi] x [-\pi, -3\pi/4]
   327) 
   328) ! Face 4 finally connects Face 3 to Face 1.  Like Face 2, but wtih opposite x
   329) !    [-1,1] x [-1,1] => [-3\pi/4, -\pi/4] x [-\pi/4, \pi/4]
   330) 
   331) ! Face 5 is along the bottom edges of Faces 1,2,3,and 4 so the latitude goes from
   332) ! -\pi/4 to -\pi/2.  The tricky part is lining up the longitude.  The zero longitude
   333) ! must line up with the center of Face 1. ATAN2(x,1) = 0 => x = 0.  
   334) ! So the (0,1) point on Face 5 is the zero longitude on the sphere.  The top edge of 
   335) ! Face 5 is the bottom edge of Face 1. 
   336) ! ATAN(x,0) = \pi/2 => x = 1, so the right edge of Face 5 is the bottom of Face 2.
   337) ! Continueing, the bottom edge of 5 is the bottom of 3.  Left of 5 is bottom of 4.
   338) 
   339) ! Face 6 is along the top edges of Faces 1,2,3 and 4 so the latitude goes from  
   340) ! \pi/4 to \pi/2.   The zero longitude must line up with the center of Face 1.  
   341) ! This is just like Face 5, but the y axis is reversed.  So the bottom edge of Face 6
   342) ! is the top edge of Face 1.  The right edge of Face 6 is the top of Face 2.  The
   343) ! top of 6 the top of 3 and the left of 6 the top of 4.
   344) 
   345)     use physical_constants, only : DD_PI
   346)     use parallel_mod,       only : abortmp
   347)     implicit none
   348)     type (cartesian2d_t), intent(in)     :: cart   ! On face_no of a unit cube
   349)     integer,              intent(in)     :: face_no 
   350)  
   351)     type (spherical_polar_t)             :: sphere
   352) 
   353)     integer i,j
   354)     real(kind=real_kind) :: r!, l_inf
   355) 
   356) ! MNL: removing check that points are on the unit cube because we allow
   357) ! spherical grids to map beyond the extent of the cube (though we probably
   358) ! should still have an upper bound for how far past the edge the element lies)
   359) !    l_inf = MAX(ABS(cart%x), ABS(cart%y)) 
   360) !    if (1.01 < l_inf) then
   361) !      call abortmp('unit_face_based_cube_to_unit_sphere: Input not on unit cube.')
   362) !    end if
   363) 
   364)     sphere%r=one
   365)     r = SQRT( one + (cart%x)**2 + (cart%y)**2)
   366)     select case (face_no)
   367)     case (1) 
   368)        sphere%lat=ASIN((cart%y)/r)
   369)        sphere%lon=ATAN2(cart%x,one)
   370)     case (2) 
   371)        sphere%lat=ASIN((cart%y)/r)
   372)        sphere%lon=ATAN2(one,-cart%x)
   373)     case (3) 
   374)        sphere%lat=ASIN((cart%y)/r)
   375)        sphere%lon=ATAN2(-cart%x,-one)
   376)     case (4) 
   377)        sphere%lat=ASIN((cart%y)/r)
   378)        sphere%lon=ATAN2(-one,cart%x)
   379)     case (5) 
   380)        if (ABS(cart%y) > DIST_THRESHOLD .or. ABS(cart%x) > DIST_THRESHOLD ) then
   381)           sphere%lon=ATAN2(cart%x, cart%y )
   382)        else
   383)           sphere%lon= 0.0D0     ! longitude is meaningless at south pole set to 0.0
   384)        end if
   385)        sphere%lat=ASIN(-one/r)
   386)     case (6) 
   387)        if (ABS(cart%y) > DIST_THRESHOLD .or. ABS(cart%x) > DIST_THRESHOLD ) then
   388)           sphere%lon = ATAN2(cart%x, -cart%y)
   389)        else
   390)           sphere%lon= 0.0D0     ! longitude is meaningless at north pole set to 0.0
   391)        end if
   392)        sphere%lat=ASIN(one/r)
   393)     case default
   394)        call abortmp('unit_face_based_cube_to_unit_sphere: Face number not 1 to 6.')
   395)     end select
   396) 
   397)     if (sphere%lon < 0.0D0) then
   398)        sphere%lon=sphere%lon + two*DD_PI
   399)     end if
   400) 
   401)   end function unit_face_based_cube_to_unit_sphere

   403)   function cart2spherical(x,y, face_no) result(sphere)
   404) ! IMPORTANT: INPUT ARE the REAL cartesian from the cube sphere
   405) ! Note: Output spherical longitude is [-pi,pi]
   406) 
   407) ! Project from a UNIT cube to a UNIT sphere.  ie, the lenght of the cube edge is 2.
   408) ! Face 1 of the cube touches the sphere at longitude, latitude (0,0). The negative 
   409) ! x axis is negative longitude (ie. going west is negative), the positive x axis
   410) ! is increasing longitude.  Face 1 maps the Face 1 to the lat,lon on the sphere:
   411) !    [-1,1] x [-1,1] => [-\pi/4,\pi/4] x [-\pi/4, \pi/4]
   412) 
   413) ! Face 2 continues with increasing longitude (ie to the east of Face 1).  
   414) ! The left edge of Face 2 (negative x) is the right edge of Face 1 (positive x)
   415) ! The latitude is the same as Face 1, but the longitude increases:
   416) !    [-1,1] x [-1,1] => [\pi/4, 3\pi/4] x [-\pi/4, \pi/4]
   417) 
   418) ! Face 3 continues with increasing longitude (ie to the east of Face 2).  
   419) ! Face 3 is like Face 1, but the x coordinates are reversed, ie. decreasing x 
   420) ! is increasing longitude:
   421) !    [-1,1] x [-1,1]  =    [-1,0] x [-1,1] U  [0,1] x [-1,1] =>
   422) !            [3\pi/4,\pi] x [-\pi, -3\pi/4]
   423) 
   424) ! Face 4 finally connects Face 3 to Face 1.  Like Face 2, but wtih opposite x
   425) !    [-1,1] x [-1,1] => [-3\pi/4, -\pi/4] x [-\pi/4, \pi/4]
   426) 
   427) ! Face 5 is along the bottom edges of Faces 1,2,3,and 4 so the latitude goes from
   428) ! -\pi/4 to -\pi/2.  The tricky part is lining up the longitude.  The zero longitude
   429) ! must line up with the center of Face 1. ATAN2(x,1) = 0 => x = 0.  
   430) ! So the (0,1) point on Face 5 is the zero longitude on the sphere.  The top edge of 
   431) ! Face 5 is the bottom edge of Face 1. 
   432) ! ATAN(x,0) = \pi/2 => x = 1, so the right edge of Face 5 is the bottom of Face 2.
   433) ! Continueing, the bottom edge of 5 is the bottom of 3.  Left of 5 is bottom of 4.
   434) 
   435) ! Face 6 is along the top edges of Faces 1,2,3 and 4 so the latitude goes from  
   436) ! \pi/4 to \pi/2.   The zero longitude must line up with the center of Face 1.  
   437) ! This is just like Face 5, but the y axis is reversed.  So the bottom edge of Face 6
   438) ! is the top edge of Face 1.  The right edge of Face 6 is the top of Face 2.  The
   439) ! top of 6 the top of 3 and the left of 6 the top of 4.
   440) 
   441)     use physical_constants, only : DD_PI
   442)     use parallel_mod,       only : abortmp
   443)     implicit none
   444)     real(kind=real_kind), intent(in)     :: x,y   ! On face_no of a unit cube
   445)     integer,              intent(in)     :: face_no 
   446)  
   447)     type (spherical_polar_t)             :: sphere
   448) 
   449)     integer i,j
   450)     real(kind=real_kind) :: r!, l_inf
   451) 
   452) ! MNL: removing check that points are on the unit cube because we allow
   453) ! spherical grids to map beyond the extent of the cube (though we probably
   454) ! should still have an upper bound for how far past the edge the element lies)
   455) !    l_inf = MAX(ABS(cart%x), ABS(cart%y)) 
   456) !    if (1.01 < l_inf) then
   457) !      call abortmp('unit_face_based_cube_to_unit_sphere: Input not on unit cube.')
   458) !    end if
   459) 
   460)     sphere%r=one
   461)     r = SQRT( one + x**2 + y**2)
   462)     select case (face_no)
   463)     case (1) 
   464)        sphere%lat=ASIN(y/r)
   465)        sphere%lon=ATAN2(x,one)
   466)     case (2) 
   467)        sphere%lat=ASIN(y/r)
   468)        sphere%lon=ATAN2(one,-x)
   469)     case (3) 
   470)        sphere%lat=ASIN(y/r)
   471)        sphere%lon=ATAN2(-x,-one)
   472)     case (4) 
   473)        sphere%lat=ASIN(y/r)
   474)        sphere%lon=ATAN2(-one,x)
   475)     case (5) 
   476)        if (ABS(y) > DIST_THRESHOLD .or. ABS(x) > DIST_THRESHOLD ) then
   477)           sphere%lon=ATAN2(x, y )
   478)        else
   479)           sphere%lon= 0.0D0     ! longitude is meaningless at south pole set to 0.0
   480)        end if
   481)        sphere%lat=ASIN(-one/r)
   482)     case (6) 
   483)        if (ABS(y) > DIST_THRESHOLD .or. ABS(x) > DIST_THRESHOLD ) then
   484)           sphere%lon = ATAN2(x, -y)
   485)        else
   486)           sphere%lon= 0.0D0     ! longitude is meaningless at north pole set to 0.0
   487)        end if
   488)        sphere%lat=ASIN(one/r)
   489)     case default
   490)        call abortmp('unit_face_based_cube_to_unit_sphere: Face number not 1 to 6.')
   491)     end select
   492) 
   493)     if (sphere%lon < 0.0D0) then

   652) ! faces of the cube.  It should work for any (x,y,z) coordinate

   680)       endif
   681)    else if (y>x .and. y<-x) then  ! x<0

   684)       else if (z<x) then

   686)       else

   688)       endif
   689)    else if (y>x .and. y>-x) then  ! y>0

   692)       else if (z<-y) then

   694)       else

   696)       endif
   697)    else if (y<x .and. y<-x) then  ! y<0

   700)       else if (z<y) then
   701)          face_no=5 ! south pole
   702)       else 
   703)          face_no=4
   704)       endif
   705)    else
   706)       ! abs(y) = abs(x).  point is on cube edge, or on face 5 or 6:
   707)       if (abs(x)<z) then

   709)       else if (z<-abs(x)) then

   713)       else if (x < 0 .and. 0 < y) then

   771)   subroutine sphere_tri_area( v1, v2, v3, area )
   772)   !  input: v1(3),v2(3),v3(3)  cartesian coordinates of triangle
   773)   !  output: area
   774)   !  based on formulas in STRI_QUAD:
   775)   !  http://people.sc.fsu.edu/~burkardt/f_src/stri_quad/stri_quad.html
   776)   use physical_constants, only : dd_pi
   777)   implicit none
   778)   real(kind=real_kind) area
   779)   real(kind=real_kind) a,b,c,al,bl,cl,sina,sinb,sinc,sins,a1,b1,c1
   780)   type (cartesian3D_t) v1,v2,v3
   781)   
   782)   ! compute great circle lengths
   783)   al = acos( v2%x * v3%x + v2%y * v3%y + v2%z * v3%z )
   784)   bl = acos( v3%x * v1%x + v3%y * v1%y + v3%z * v1%z )
   785)   cl = acos( v1%x * v2%x + v1%y * v2%y + v1%z * v2%z )
   786) 
   787)   ! compute angles
   788)   sina = sin( (bl+cl-al)/2 )  ! sin(sl-al)
   789)   sinb = sin( (al+cl-bl)/2 )  ! sin(sl-bl)
   790)   sinc = sin( (al+bl-cl)/2 ) 
   791)   sins = sin( (al+bl+cl)/2 )
   792) 
   793) 
   794) #if 0
   795)   a = 2*atan2(sqrt(sinb*sinc), sqrt(sins*sina)  )
   796)   b = 2*atan2(sqrt(sina*sinc), sqrt(sins*sinb) )
   797)   c = 2*atan2(sqrt(sina*sinb), sqrt(sins*sinc) )
   798)   ! apply Girard's theorem
   799)   area = a+b+c - dd_pi
   800) #endif
   801) 
   802)   ! for small areas, formula above looses precision.  
   803)   ! 2atan(x) + 2atan(1/x) = pi      
   804)   ! 2atan(x) - pi = -2atan(1/x)
   805)   a = sqrt( (sinb*sinc) / (sins*sina) ) 
   806)   b = sqrt( (sina*sinc) / (sins*sinb) ) 
   807)   c = sqrt( (sina*sinb) / (sins*sinc) ) 
   808) 
   809) 
   810)   a1 = 2*atan(a)
   811)   b1 = 2*atan(b)
   812)   c1 = 2*atan(c)
   813) 
   814)   if (a.gt.b.and.a.gt.c) then
   815)      a1 = -2*atan(1/a)
   816)   else if (b.gt.c) then
   817)      b1 = -2*atan(1/b)
   818)   else 
   819)      c1 = -2*atan(1/c)
   820)   endif
   821)   ! apply Girard's theorem
   822)   area = a1+b1+c1  
   823) 
   824) 
   825)   end subroutine sphere_tri_area
