    35) ! !INTERFACE:
    36) subroutine CStateUpdate0(num_soilp, filter_soilp)
    37) !
    38) ! !DESCRIPTION:
    39) ! On the radiation time step, update cpool carbon state
    40) !
    41) ! !USES:
    42)    use clmtype
    43)    use clm_time_manager, only: get_step_size
    44) !
    45) ! !ARGUMENTS:
    46)    implicit none
    47)    integer, intent(in) :: num_soilp       ! number of soil pfts in filter
    48)    integer, intent(in) :: filter_soilp(:) ! filter for soil pfts
    49) !
    50) ! !CALLED FROM:
    51) ! subroutine CNEcosystemDyn
    52) !
    53) ! !REVISION HISTORY:
    54) ! 7/1/05: Created by Peter Thornton
    55) !
    56) ! !LOCAL VARIABLES:
    57) ! local pointers to implicit in arrays
    58)    real(r8), pointer :: psnshade_to_cpool(:)
    59)    real(r8), pointer :: psnsun_to_cpool(:)
    60) !
    61) ! local pointers to implicit in/out arrays
    62)    real(r8), pointer :: cpool(:)              ! (gC/m2) temporary photosynthate C pool
    63) ! !OTHER LOCAL VARIABLES:
    64)    integer :: p     ! indices
    65)    integer :: fp   ! lake filter indices
    66)    real(r8):: dt      ! radiation time step (seconds)
    67) !
    68) !EOP
    69) !-----------------------------------------------------------------------
    70)     ! assign local pointers at the pft level
    71)     cpool                          => pcs%cpool
    72)     psnshade_to_cpool              => pcf%psnshade_to_cpool
    73)     psnsun_to_cpool                => pcf%psnsun_to_cpool
    74) 
    75)     ! set time steps
    76)     dt = real( get_step_size(), r8 )
    77) 
    78)     ! pft loop
    79)     do fp = 1,num_soilp
    80)        p = filter_soilp(fp)
    81)        ! gross photosynthesis fluxes
    82)        cpool(p) = cpool(p) + psnsun_to_cpool(p)*dt
    83)        cpool(p) = cpool(p) + psnshade_to_cpool(p)*dt
    84)     end do
    85) 
    86) end subroutine CStateUpdate0

   432) 
   433)     ! column loop
   434)     do fc = 1,num_soilc
   435)        c = filter_soilc(fc)
   436)  
   437)        ! column level fluxes
   438)  
   439)        ! plant to litter fluxes
   440)        ! leaf litter
   441)        litr1c(c) = litr1c(c) + leafc_to_litr1c(c)*dt
   442)        litr2c(c) = litr2c(c) + leafc_to_litr2c(c)*dt
   443)        litr3c(c) = litr3c(c) + leafc_to_litr3c(c)*dt
   444)        ! fine root litter
   445)        litr1c(c) = litr1c(c) + frootc_to_litr1c(c)*dt
   446)        litr2c(c) = litr2c(c) + frootc_to_litr2c(c)*dt
   447)        litr3c(c) = litr3c(c) + frootc_to_litr3c(c)*dt
   448)        if ( crop_prog )then
   449)           ! livestem litter
   450)           litr1c(c) = litr1c(c) + livestemc_to_litr1c(c)*dt
   451)           litr2c(c) = litr2c(c) + livestemc_to_litr2c(c)*dt
   452)           litr3c(c) = litr3c(c) + livestemc_to_litr3c(c)*dt
   453)           ! grain litter
   454)           litr1c(c) = litr1c(c) + grainc_to_litr1c(c)*dt
   455)           litr2c(c) = litr2c(c) + grainc_to_litr2c(c)*dt
   456)           litr3c(c) = litr3c(c) + grainc_to_litr3c(c)*dt
   457)        end if
   458)        
   459)        ! seeding fluxes, from dynamic landcover
   460)        seedc(c) = seedc(c) - dwt_seedc_to_leaf(c) * dt
   461)        seedc(c) = seedc(c) - dwt_seedc_to_deadstem(c) * dt
   462)    
   463)        ! fluxes into litter and CWD, from dynamic landcover
   464)        litr1c(c) = litr1c(c) + dwt_frootc_to_litr1c(c)*dt
   465)        litr2c(c) = litr2c(c) + dwt_frootc_to_litr2c(c)*dt
   466)        litr3c(c) = litr3c(c) + dwt_frootc_to_litr3c(c)*dt
   467)        cwdc(c)   = cwdc(c)   + dwt_livecrootc_to_cwdc(c)*dt
   468)        cwdc(c)   = cwdc(c)   + dwt_deadcrootc_to_cwdc(c)*dt
   469)        
   470)        ! litter and SOM HR fluxes
   471)        litr1c(c) = litr1c(c) - litr1_hr(c)*dt
   472)        litr2c(c) = litr2c(c) - litr2_hr(c)*dt
   473)        litr3c(c) = litr3c(c) - litr3_hr(c)*dt
   474)        soil1c(c) = soil1c(c) - soil1_hr(c)*dt
   475)        soil2c(c) = soil2c(c) - soil2_hr(c)*dt
   476)        soil3c(c) = soil3c(c) - soil3_hr(c)*dt
   477)        soil4c(c) = soil4c(c) - soil4_hr(c)*dt
   478)  
   479)        ! CWD to litter fluxes
   480)        cwdc(c)   = cwdc(c)   - cwdc_to_litr2c(c)*dt
   481)        litr2c(c) = litr2c(c) + cwdc_to_litr2c(c)*dt
   482)        cwdc(c)   = cwdc(c)   - cwdc_to_litr3c(c)*dt
   483)        litr3c(c) = litr3c(c) + cwdc_to_litr3c(c)*dt
   484)

   486)        litr1c(c) = litr1c(c) - litr1c_to_soil1c(c)*dt
   487)        soil1c(c) = soil1c(c) + litr1c_to_soil1c(c)*dt
   488)        litr2c(c) = litr2c(c) - litr2c_to_soil2c(c)*dt
   489)        soil2c(c) = soil2c(c) + litr2c_to_soil2c(c)*dt
   490)        litr3c(c) = litr3c(c) - litr3c_to_soil3c(c)*dt
   491)        soil3c(c) = soil3c(c) + litr3c_to_soil3c(c)*dt
   492)  
   493)        ! SOM to SOM fluxes
   494)        soil1c(c) = soil1c(c) - soil1c_to_soil2c(c)*dt
   495)        soil2c(c) = soil2c(c) + soil1c_to_soil2c(c)*dt
   496)        soil2c(c) = soil2c(c) - soil2c_to_soil3c(c)*dt
   497)        soil3c(c) = soil3c(c) + soil2c_to_soil3c(c)*dt
   498)        soil3c(c) = soil3c(c) - soil3c_to_soil4c(c)*dt
   499)        soil4c(c) = soil4c(c) + soil3c_to_soil4c(c)*dt
   500)  
   501)     end do ! end of columns loop
   502)  
   503)     ! pft loop
   504)     do fp = 1,num_soilp
   505)        p = filter_soilp(fp)
   506)  
   507)        ! phenology: transfer growth fluxes
   508)        leafc(p)           = leafc(p)       + leafc_xfer_to_leafc(p)*dt
   509)        leafc_xfer(p)      = leafc_xfer(p)  - leafc_xfer_to_leafc(p)*dt
   510)        frootc(p)          = frootc(p)      + frootc_xfer_to_frootc(p)*dt
   511)        frootc_xfer(p)     = frootc_xfer(p) - frootc_xfer_to_frootc(p)*dt
   512)        if (woody(ivt(p)) == 1._r8) then
   513)            livestemc(p)       = livestemc(p)           + livestemc_xfer_to_livestemc(p)*dt
   514)            livestemc_xfer(p)  = livestemc_xfer(p)  - livestemc_xfer_to_livestemc(p)*dt
   515)            deadstemc(p)       = deadstemc(p)           + deadstemc_xfer_to_deadstemc(p)*dt
   516)            deadstemc_xfer(p)  = deadstemc_xfer(p)  - deadstemc_xfer_to_deadstemc(p)*dt
   517)            livecrootc(p)      = livecrootc(p)          + livecrootc_xfer_to_livecrootc(p)*dt
   518)            livecrootc_xfer(p) = livecrootc_xfer(p) - livecrootc_xfer_to_livecrootc(p)*dt
   519)            deadcrootc(p)      = deadcrootc(p)          + deadcrootc_xfer_to_deadcrootc(p)*dt
   520)            deadcrootc_xfer(p) = deadcrootc_xfer(p) - deadcrootc_xfer_to_deadcrootc(p)*dt
   521)        end if
   522)        if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   523)            ! lines here for consistency; the transfer terms are zero
   524)            livestemc(p)       = livestemc(p)      + livestemc_xfer_to_livestemc(p)*dt
   525)            livestemc_xfer(p)  = livestemc_xfer(p) - livestemc_xfer_to_livestemc(p)*dt

   527)            grainc_xfer(p)     = grainc_xfer(p)    - grainc_xfer_to_grainc(p)*dt
   528)        end if
   529)  
   530)        ! phenology: litterfall fluxes
   531)        leafc(p) = leafc(p) - leafc_to_litter(p)*dt
   532)        frootc(p) = frootc(p) - frootc_to_litter(p)*dt
   533)  
   534)        ! livewood turnover fluxes
   535)        if (woody(ivt(p)) == 1._r8) then
   536)            livestemc(p)  = livestemc(p)  - livestemc_to_deadstemc(p)*dt
   537)            deadstemc(p)  = deadstemc(p)  + livestemc_to_deadstemc(p)*dt
   538)            livecrootc(p) = livecrootc(p) - livecrootc_to_deadcrootc(p)*dt
   539)            deadcrootc(p) = deadcrootc(p) + livecrootc_to_deadcrootc(p)*dt
   540)        end if
   541)        if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   542)            livestemc(p)  = livestemc(p)  - livestemc_to_litter(p)*dt
   543)            grainc(p)     = grainc(p)     - grainc_to_food(p)*dt
   544)        end if
   545)  
   546)        ! maintenance respiration fluxes from cpool
   547)        cpool(p) = cpool(p) - cpool_to_xsmrpool(p)*dt
   548)        cpool(p) = cpool(p) - leaf_curmr(p)*dt
   549)        cpool(p) = cpool(p) - froot_curmr(p)*dt
   550)        if (woody(ivt(p)) == 1._r8) then
   551)            cpool(p) = cpool(p) - livestem_curmr(p)*dt
   552)            cpool(p) = cpool(p) - livecroot_curmr(p)*dt
   553)        end if
   554)        if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   555)            cpool(p) = cpool(p) - livestem_curmr(p)*dt
   556)        end if
   557) 
   558)        ! maintenance respiration fluxes from xsmrpool
   559)        xsmrpool(p) = xsmrpool(p) + cpool_to_xsmrpool(p)*dt
   560)        xsmrpool(p) = xsmrpool(p) - leaf_xsmr(p)*dt
   561)        xsmrpool(p) = xsmrpool(p) - froot_xsmr(p)*dt
   562)        if (woody(ivt(p)) == 1._r8) then
   563)            xsmrpool(p) = xsmrpool(p) - livestem_xsmr(p)*dt
   564)            xsmrpool(p) = xsmrpool(p) - livecroot_xsmr(p)*dt
   565)        end if
   566)        if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   567)            xsmrpool(p) = xsmrpool(p) - livestem_xsmr(p)*dt

   569)               xsmrpool_to_atm(p) = xsmrpool_to_atm(p) + xsmrpool(p)/dt
   570)               xsmrpool(p) = xsmrpool(p) - xsmrpool_to_atm(p)*dt
   571)            end if
   572)        end if
   573)  
   574)        ! allocation fluxes
   575)        cpool(p)           = cpool(p)          - cpool_to_leafc(p)*dt
   576)        leafc(p)           = leafc(p)          + cpool_to_leafc(p)*dt
   577)        cpool(p)           = cpool(p)          - cpool_to_leafc_storage(p)*dt
   578)        leafc_storage(p)   = leafc_storage(p)  + cpool_to_leafc_storage(p)*dt
   579)        cpool(p)           = cpool(p)          - cpool_to_frootc(p)*dt
   580)        frootc(p)          = frootc(p)         + cpool_to_frootc(p)*dt
   581)        cpool(p)           = cpool(p)          - cpool_to_frootc_storage(p)*dt
   582)        frootc_storage(p)  = frootc_storage(p) + cpool_to_frootc_storage(p)*dt
   583)        if (woody(ivt(p)) == 1._r8) then
   584)            cpool(p)               = cpool(p)              - cpool_to_livestemc(p)*dt
   585)            livestemc(p)           = livestemc(p)          + cpool_to_livestemc(p)*dt
   586)            cpool(p)               = cpool(p)              - cpool_to_livestemc_storage(p)*dt
   587)            livestemc_storage(p)   = livestemc_storage(p)  + cpool_to_livestemc_storage(p)*dt
   588)            cpool(p)               = cpool(p)              - cpool_to_deadstemc(p)*dt
   589)            deadstemc(p)           = deadstemc(p)          + cpool_to_deadstemc(p)*dt
   590)            cpool(p)               = cpool(p)              - cpool_to_deadstemc_storage(p)*dt
   591)            deadstemc_storage(p)   = deadstemc_storage(p)  + cpool_to_deadstemc_storage(p)*dt

   595)            livecrootc_storage(p)  = livecrootc_storage(p) + cpool_to_livecrootc_storage(p)*dt
   596)            cpool(p)               = cpool(p)              - cpool_to_deadcrootc(p)*dt
   597)            deadcrootc(p)          = deadcrootc(p)         + cpool_to_deadcrootc(p)*dt
   598)            cpool(p)               = cpool(p)              - cpool_to_deadcrootc_storage(p)*dt
   599)            deadcrootc_storage(p)  = deadcrootc_storage(p) + cpool_to_deadcrootc_storage(p)*dt
   600)        end if
   601)        if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   602)            cpool(p)               = cpool(p)              - cpool_to_livestemc(p)*dt
   603)            livestemc(p)           = livestemc(p)          + cpool_to_livestemc(p)*dt
   604)            cpool(p)               = cpool(p)              - cpool_to_livestemc_storage(p)*dt
   605)            livestemc_storage(p)   = livestemc_storage(p)  + cpool_to_livestemc_storage(p)*dt
   606)            cpool(p)               = cpool(p)              - cpool_to_grainc(p)*dt
   607)            grainc(p)              = grainc(p)             + cpool_to_grainc(p)*dt
   608)            cpool(p)               = cpool(p)              - cpool_to_grainc_storage(p)*dt
   609)            grainc_storage(p)      = grainc_storage(p)     + cpool_to_grainc_storage(p)*dt
   610)        end if
   611)  
   612)        ! growth respiration fluxes for current growth
   613)        cpool(p) = cpool(p) - cpool_leaf_gr(p)*dt
   614)        cpool(p) = cpool(p) - cpool_froot_gr(p)*dt
   615)        if (woody(ivt(p)) == 1._r8) then
   616)            cpool(p) = cpool(p) - cpool_livestem_gr(p)*dt
   617)            cpool(p) = cpool(p) - cpool_deadstem_gr(p)*dt
   618)            cpool(p) = cpool(p) - cpool_livecroot_gr(p)*dt
   619)            cpool(p) = cpool(p) - cpool_deadcroot_gr(p)*dt
   620)        end if
   621)        if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   622)            cpool(p) = cpool(p) - cpool_livestem_gr(p)*dt
   623)            cpool(p) = cpool(p) - cpool_grain_gr(p)*dt
   624)        end if
   625)  
   626)        ! growth respiration for transfer growth
   627)        gresp_xfer(p) = gresp_xfer(p) - transfer_leaf_gr(p)*dt
   628)        gresp_xfer(p) = gresp_xfer(p) - transfer_froot_gr(p)*dt
   629)        if (woody(ivt(p)) == 1._r8) then
   630)            gresp_xfer(p) = gresp_xfer(p) - transfer_livestem_gr(p)*dt
   631)            gresp_xfer(p) = gresp_xfer(p) - transfer_deadstem_gr(p)*dt
   632)            gresp_xfer(p) = gresp_xfer(p) - transfer_livecroot_gr(p)*dt
   633)            gresp_xfer(p) = gresp_xfer(p) - transfer_deadcroot_gr(p)*dt
   634)        end if
   635)        if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   636)            gresp_xfer(p) = gresp_xfer(p) - transfer_livestem_gr(p)*dt
   637)            gresp_xfer(p) = gresp_xfer(p) - transfer_grain_gr(p)*dt
   638)        end if
   639)  
   640)        ! growth respiration at time of storage
   641)        cpool(p) = cpool(p) - cpool_leaf_storage_gr(p)*dt
   642)        cpool(p) = cpool(p) - cpool_froot_storage_gr(p)*dt
   643)        if (woody(ivt(p)) == 1._r8) then
   644)            cpool(p) = cpool(p) - cpool_livestem_storage_gr(p)*dt
   645)            cpool(p) = cpool(p) - cpool_deadstem_storage_gr(p)*dt
   646)            cpool(p) = cpool(p) - cpool_livecroot_storage_gr(p)*dt
   647)            cpool(p) = cpool(p) - cpool_deadcroot_storage_gr(p)*dt
   648)        end if
   649)        if (ivt(p) >= npcropmin) then ! skip 2 generic crops
   650)            cpool(p) = cpool(p) - cpool_livestem_storage_gr(p)*dt
   651)            cpool(p) = cpool(p) - cpool_grain_storage_gr(p)*dt
   652)        end if
   653)  
   654)        ! growth respiration stored for release during transfer growth
   655)        cpool(p)         = cpool(p)         - cpool_to_gresp_storage(p)*dt
   656)        gresp_storage(p) = gresp_storage(p) + cpool_to_gresp_storage(p)*dt
   657)  
   658)        ! move storage pools into transfer pools
   659)        leafc_storage(p)   = leafc_storage(p)   - leafc_storage_to_xfer(p)*dt
   660)        leafc_xfer(p)  = leafc_xfer(p)  + leafc_storage_to_xfer(p)*dt
   661)        frootc_storage(p)  = frootc_storage(p)  - frootc_storage_to_xfer(p)*dt
   662)        frootc_xfer(p) = frootc_xfer(p) + frootc_storage_to_xfer(p)*dt
   663)        if (woody(ivt(p)) == 1._r8) then
