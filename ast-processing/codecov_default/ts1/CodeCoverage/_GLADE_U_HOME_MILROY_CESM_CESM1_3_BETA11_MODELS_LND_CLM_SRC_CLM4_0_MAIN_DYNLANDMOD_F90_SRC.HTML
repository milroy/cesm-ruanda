    43)    subroutine dynland_hwcontent(begg,endg,gcell_liq,gcell_ice,gcell_heat)
    44)  
    45) ! !DESCRIPTION:
    46) !    Compute grid-level heat and water content
    47) !
    48) ! !REVISION HISTORY:
    49) !    2009-feb-20 B. Kauffman, created by
    50) !
    51) ! !USES:
    52) 
    53)    use clm_varcon, only : istsoil,istice,istwet,istdlak,istslak,isturb,istice_mec
    54)    use clm_varcon, only : istcrop
    55)    use clm_varcon, only : icol_road_perv,icol_road_imperv,icol_roof
    56)    use clm_varcon, only : icol_sunwall,icol_shadewall
    57)    use clm_varcon, only : cpice,  cpliq
    58)    use clm_varpar, only : nlevsno, nlevgrnd
    59) 
    60)    implicit none
    61) 
    62) ! !ARGUMENTS:
    63) 
    64)    integer , intent(in)  :: begg, endg              ! proc beg & end gridcell indices
    65)    real(r8), intent(out) :: gcell_liq(begg:endg)
    66)    real(r8), intent(out) :: gcell_ice  (begg:endg)
    67)    real(r8), intent(out) :: gcell_heat (begg:endg)
    68)  
    69) ! !LOCAL VARIABLES:
    70) !EOP
    71) 
    72)    integer  :: li,lf         ! loop initial/final indicies
    73)    integer  :: ci,cf         ! loop initial/final indicies
    74)    integer  :: pi,pf         ! loop initial/final indicies
    75) 
    76)    integer  :: g,l,c,p,k     ! loop indicies (grid,lunit,column,pft,vertical level)
    77) 
    78)    real(r8) :: wtgcell       ! weight relative to grid cell
    79)    real(r8) :: wtcol         ! weight relative to column
    80)    real(r8) :: liq           ! sum of liquid water at column level
    81)    real(r8) :: ice           ! sum of frozen water at column level
    82)    real(r8) :: heat          ! sum of heat content at column level
    83)    real(r8) :: cv            ! heat capacity [J/(m^2 K)]
    84) 
    85)    integer ,pointer :: ltype(:)          ! landunit type index
    86)    integer ,pointer :: ctype(:)          ! column   type index
    87)    integer ,pointer :: ptype(:)          ! pft      type index
    88) 
    89)    integer,  pointer :: nlev_improad(:)  ! number of impervious road layers
    90)    real(r8), pointer :: cv_wall(:,:)     ! thermal conductivity of urban wall
    91)    real(r8), pointer :: cv_roof(:,:)     ! thermal conductivity of urban roof
    92)    real(r8), pointer :: cv_improad(:,:)  ! thermal conductivity of urban impervious road
    93) 
    94)    integer , pointer :: snl(:)           ! number of snow layers
    95)    real(r8), pointer :: t_soisno(:,:)    ! soil temperature (Kelvin)
    96)    real(r8), pointer :: h2osno(:)        ! snow water (mm H2O)
    97)    real(r8), pointer :: h2osoi_liq(:,:)  ! liquid water (kg/m2)
    98)    real(r8), pointer :: h2osoi_ice(:,:)  ! frozen water (kg/m2)
    99)    real(r8), pointer :: watsat(:,:)      ! volumetric soil water at saturation (porosity)
   100)    real(r8), pointer :: csol(:,:)        ! heat capacity, soil solids (J/m**3/Kelvin)
   101)    real(r8), pointer :: dz(:,:)          ! layer depth (m)
   102)    real(r8), pointer :: wa(:,:)          ! h2o in underground aquifer
   103) 
   104)    type(gridcell_type), pointer :: gptr  ! pointer to gridcell derived subtype
   105)    type(landunit_type), pointer :: lptr  ! pointer to landunit derived subtype
   106)    type(column_type)  , pointer :: cptr  ! pointer to column derived subtype
   107)    type(pft_type)     , pointer :: pptr  ! pointer to pft derived subtype
   108) 
   109) !-------------------------------------------------------------------------------
   110) ! Note: this routine does not compute heat or water content of lakes.
   111) !
   112) !-------------------------------------------------------------------------------
   113) 
   114)    ! Set pointers into derived type
   115) 
   116)    gptr => grc
   117)    lptr => lun
   118)    cptr => col
   119)    pptr => pft
   120) 
   121)    ltype => lun%itype
   122)    ctype => col%itype
   123)    ptype => pft%itype
   124) 
   125)    nlev_improad => lps%nlev_improad
   126)    cv_wall      => lps%cv_wall
   127)    cv_roof      => lps%cv_roof
   128)    cv_improad   => lps%cv_improad
   129) 
   130)    snl          => cps%snl
   131)    watsat       => cps%watsat
   132)    csol         => cps%csol
   133)    dz           => cps%dz
   134)    t_soisno     => ces%t_soisno
   135)    h2osoi_liq   => cws%h2osoi_liq
   136)    h2osoi_ice   => cws%h2osoi_ice
   137)    h2osno       => cws%h2osno
   138) 
   139)    ! Get relevant sizes
   140) 
   141)    do g = begg,endg ! loop over grid cells
   142) 
   143)       gcell_liq  (g) = 0.0_r8   ! sum for one grid cell
   144)       gcell_ice  (g) = 0.0_r8   ! sum for one grid cell
   145)       gcell_heat (g) = 0.0_r8   ! sum for one grid cell
   146) 
   147)       li = gptr%luni(g)
   148)       lf = gptr%lunf(g)
   149)       do l = li,lf   ! loop over land units  
   150) 
   151)          ci = lptr%coli(l)
   152)          cf = lptr%colf(l)
   153)          do c = ci,cf   ! loop over columns
   154) 
   155)             liq   = 0.0_r8 ! sum for one column
   156)             ice   = 0.0_r8
   157)             heat  = 0.0_r8
   158) 
   159)             !--- water & ice, above ground only ---
   160)             if ( (ltype(l) == istsoil .or. ltype(l) == istcrop         )  &
   161)             .or. (ltype(l) == istwet                                   )  &
   162)             .or. (ltype(l) == istice                                   )  &
   163)             .or. (ltype(l) == istice_mec                               )  &           
   164)             .or. (ltype(l) == isturb .and. ctype(c) == icol_roof       )  &
   165)             .or. (ltype(l) == isturb .and. ctype(c) == icol_road_imperv)  &
   166)             .or. (ltype(l) == isturb .and. ctype(c) == icol_road_perv  )) then
   167) 
   168)                if ( snl(c) < 0 ) then
   169)                   do k = snl(c)+1,0 ! loop over snow layers
   170)                      liq   = liq   + cws%h2osoi_liq(c,k)
   171)                      ice   = ice   + cws%h2osoi_ice(c,k)
   172)                   end do
   173)                else                 ! no snow layers exist
   174)                   ice = ice + cws%h2osno(c)
   175)                end if
   176)             end if
   177) 
   178)             !--- water & ice, below ground only ---
   179)             if ( (ltype(l) == istsoil .or. ltype(l) == istcrop         )  &
   180)             .or. (ltype(l) == istwet                                   )  &
   181)             .or. (ltype(l) == istice                                   )  &
   182)             .or. (ltype(l) == istice_mec                               )  &           
   183)             .or. (ltype(l) == isturb .and. ctype(c) == icol_road_perv  )) then
   184)                do k = 1,nlevgrnd
   185)                   liq   = liq   + cws%h2osoi_liq(c,k)
   186)                   ice   = ice   + cws%h2osoi_ice(c,k)
   187)                end do
   188)             end if
   189) 
   190)             !--- water in aquifer ---
   191)             if ( (ltype(l) == istsoil .or. ltype(l) == istcrop         )  &
   192)             .or. (ltype(l) == istwet                                   )  &
   193)             .or. (ltype(l) == istice                                   )  &
   194)             .or. (ltype(l) == istice_mec                               )  &           
   195)             .or. (ltype(l) == isturb .and. ctype(c) == icol_road_perv  )) then
   196)                liq = liq + cws%wa(c)
   197)             end if
   198) 
   199)             !--- water in canopy (at pft level) ---
   200)             if (ltype(l) == istsoil .or. ltype(l) == istcrop) then   ! note: soil specified at LU level
   201)                pi = cptr%pfti(c)
   202)                pf = cptr%pftf(c)
   203)                do p = pi,pf ! loop over pfts
   204)                   wtcol = pptr%wtcol(p)
   205)                   liq = liq + pws%h2ocan(p) * wtcol
   206)                end do
   207)             end if
   208) 
   209)             if ( (ltype(l) /= istslak) .and. ltype(l) /= istdlak) then
   210) 
   211)                !--- heat content, below ground only ---
   212)                do k = 1,nlevgrnd
   213)                   if (ctype(c)==icol_sunwall .OR. ctype(c)==icol_shadewall) then
   214)                       cv = cv_wall(l,k) * dz(c,k)
   215)                    else if (ctype(c) == icol_roof) then
   216)                       cv = cv_roof(l,k) * dz(c,k)
   217)                    else if (ctype(c) == icol_road_imperv .and. k >= 1 .and. k <= nlev_improad(l)) then
   218)                       cv = cv_improad(l,k) * dz(c,k)
   219)                    else if (ltype(l) /= istwet .AND. ltype(l) /= istice .AND. ltype(l) /= istice_mec) then
   220)                       cv = csol(c,k)*(1-watsat(c,k))*dz(c,k) + (h2osoi_ice(c,k)*cpice + h2osoi_liq(c,k)*cpliq)
   221)                    else
   222)                       cv = (h2osoi_ice(c,k)*cpice + h2osoi_liq(c,k)*cpliq)
   223)                    endif
   224)                    heat = heat + cv*t_soisno(c,k) / 1.e6_r8 
   225)                 end do
   226) 
   227)                !--- heat content, above ground only ---
   228)                if ( snl(c) < 0 ) then
   229)                   do k = snl(c)+1,0 ! loop over snow layers
   230)                      cv = cpliq*h2osoi_liq(c,k) + cpice*h2osoi_ice(c,k)
   231)                      heat = heat + cv*t_soisno(c,k) / 1.e6_r8
   232)                   end do
   233)                else if ( h2osno(c) > 0.0_r8) then
   234)                   k = 1
   235)                   cv = cpice*h2osno(c)
   236)                   heat = heat + cv*t_soisno(c,k) / 1.e6_r8
   237)                end if
   238) 
   239)             end if
   240) 
   241)             !--- scale x/m^2 column-level values into x/m^2 gridcell-level values ---
   242)             wtgcell = cptr%wtgcell(c)
   243)             gcell_liq  (g) = gcell_liq  (g) + liq   * wtgcell
   244)             gcell_ice  (g) = gcell_ice  (g) + ice   * wtgcell
   245)             gcell_heat (g) = gcell_heat (g) + heat  * wtgcell
   246) 
   247)          end do ! column loop      
   248)       end do ! landunit loop
   249)    end do ! grid cell loop
   250) 
   251)    end subroutine dynland_hwcontent
