    96) subroutine ndrop_init
    97) 
    98)    integer  :: ii, l, lptr, m, mm
    99)    integer  :: nspec_max            ! max number of species in a mode
   100)    character(len=32)   :: tmpname
   101)    character(len=32)   :: tmpname_cw
   102)    character(len=128)  :: long_name
   103)    character(len=8)    :: unit
   104)    logical :: history_amwg         ! output the variables used by the AMWG diag package
   105) 
   106)    !-------------------------------------------------------------------------------
   107) 
   108)    ! get indices into state%q and pbuf structures
   109)    call cnst_get_ind('NUMLIQ', numliq_idx)
   110) 
   111)    kvh_idx      = pbuf_get_index('kvh')
   112) 
   113)    zero     = 0._r8
   114)    third    = 1._r8/3._r8
   115)    twothird = 2._r8*third
   116)    sixth    = 1._r8/6._r8
   117)    sq2      = sqrt(2._r8)
   118)    sqpi     = sqrt(pi)
   119) 
   120)    t0       = 273._r8
   121)    surften  = 0.076_r8
   122)    aten     = 2._r8*mwh2o*surften/(r_universal*t0*rhoh2o)
   123)    alogaten = log(aten)
   124)    alog2    = log(2._r8)
   125)    alog3    = log(3._r8)
   126) 
   127)    ! get info about the modal aerosols
   128)    ! get ntot_amode
   129)    call rad_cnst_get_info(0, nmodes=ntot_amode)
   130) 
   131)    allocate( &
   132)       nspec_amode(ntot_amode),  &
   133)       sigmag_amode(ntot_amode), &
   134)       dgnumlo_amode(ntot_amode), &
   135)       dgnumhi_amode(ntot_amode), &
   136)       alogsig(ntot_amode),      &
   137)       exp45logsig(ntot_amode),  &
   138)       f1(ntot_amode),           &
   139)       f2(ntot_amode),           &
   140)       voltonumblo_amode(ntot_amode), &
   141)       voltonumbhi_amode(ntot_amode)  )
   142) 
   143)    do m = 1, ntot_amode
   144)       ! use only if width of size distribution is prescribed
   145) 
   146)       ! get mode info
   147)       call rad_cnst_get_info(0, m, nspec=nspec_amode(m))
   148) 
   149)       ! get mode properties
   150)       call rad_cnst_get_mode_props(0, m, sigmag=sigmag_amode(m),  &
   151)          dgnumhi=dgnumhi_amode(m), dgnumlo=dgnumlo_amode(m))
   152) 
   153)       alogsig(m)     = log(sigmag_amode(m))
   154)       exp45logsig(m) = exp(4.5_r8*alogsig(m)*alogsig(m))
   155)       f1(m)          = 0.5_r8*exp(2.5_r8*alogsig(m)*alogsig(m))
   156)       f2(m)          = 1._r8 + 0.25_r8*alogsig(m)
   157) 
   158)       voltonumblo_amode(m) = 1._r8 / ( (pi/6._r8)*                          &
   159)                              (dgnumlo_amode(m)**3._r8)*exp(4.5_r8*alogsig(m)**2._r8) )
   160)       voltonumbhi_amode(m) = 1._r8 / ( (pi/6._r8)*                          &
   161)                              (dgnumhi_amode(m)**3._r8)*exp(4.5_r8*alogsig(m)**2._r8) )
   162)    end do
   163)       
   164)    ! Init the table for local indexing of mam number conc and mmr.
   165)    ! This table uses species index 0 for the number conc.
   166) 
   167)    ! Find max number of species in all the modes, and the total
   168)    ! number of mode number concentrations + mode species
   169)    nspec_max = nspec_amode(1)
   170)    ncnst_tot = nspec_amode(1) + 1
   171)    do m = 2, ntot_amode
   172)       nspec_max = max(nspec_max, nspec_amode(m))
   173)       ncnst_tot = ncnst_tot + nspec_amode(m) + 1
   174)    end do
   175) 
   176)    allocate( &
   177)       mam_idx(ntot_amode,0:nspec_max),      &
   178)       mam_cnst_idx(ntot_amode,0:nspec_max), &
   179)       fieldname(ncnst_tot),                 &
   180)       fieldname_cw(ncnst_tot)               )
   181) 
   182)    ! Local indexing compresses the mode and number/mass indicies into one index.
   183)    ! This indexing is used by the pointer arrays used to reference state and pbuf
   184)    ! fields.
   185)    ii = 0
   186)    do m = 1, ntot_amode
   187)       do l = 0, nspec_amode(m)
   188)          ii = ii + 1
   189)          mam_idx(m,l) = ii
   190)       end do
   191)    end do
   192) 
   193)    ! Add dropmixnuc tendencies for all modal aerosol species
   194) 
   195)    call phys_getopts(history_amwg_out = history_amwg, &
   196)                      history_aerosol_out = history_aerosol, &
   197)                      prog_modal_aero_out=prog_modal_aero)
   198) 
   199) 
   200)    do m = 1, ntot_amode
   201)       do l = 0, nspec_amode(m)   ! loop over number + chem constituents
   202) 
   203)          mm = mam_idx(m,l)
   204) 
   205)          unit = 'kg/m2/s'
   206)          if (l == 0) then   ! number
   207)             unit = '#/m2/s'
   208)          end if
   209) 
   210)          if (l == 0) then   ! number
   211)             call rad_cnst_get_info(0, m, num_name=tmpname, num_name_cw=tmpname_cw)
   212)          else
   213)             call rad_cnst_get_info(0, m, l, spec_name=tmpname, spec_name_cw=tmpname_cw)
   214)          end if
   215) 
   216)          fieldname(mm)    = trim(tmpname) // '_mixnuc1'
   217)          fieldname_cw(mm) = trim(tmpname_cw) // '_mixnuc1'
   218) 
   219)          if (prog_modal_aero) then
   220) 
   221)             ! To set tendencies in the ptend object need to get the constituent indices
   222)             ! for the prognostic species
   223)             if (l == 0) then   ! number
   224)                call rad_cnst_get_mode_num_idx(m, lptr)
   225)             else
   226)                call rad_cnst_get_mam_mmr_idx(m, l, lptr)
   227)             end if
   228)             mam_cnst_idx(m,l) = lptr
   229)             lq(lptr)          = .true.
   230) 
   231)             ! Add tendency fields to the history only when prognostic MAM is enabled.
   232)             long_name = trim(tmpname) // ' dropmixnuc mixnuc column tendency'
   233)             call addfld(fieldname(mm), unit, 1, 'A', long_name, phys_decomp)
   234) 
   235)             long_name = trim(tmpname_cw) // ' dropmixnuc mixnuc column tendency'
   236)             call addfld(fieldname_cw(mm), unit, 1, 'A', long_name, phys_decomp)
   237) 
   238)             if (history_aerosol) then
   239)                call add_default(fieldname(mm), 1, ' ')
   240)                call add_default(fieldname_cw(mm), 1, ' ')
   241)             end if
   242) 
   243) 
   244) 
   245)          end if
   246)             
   247)       end do
   248)    end do
   249) 
   250)    call addfld('CCN1    ','#/cm3   ',pver, 'A','CCN concentration at S=0.02%',phys_decomp)
   251)    call addfld('CCN2    ','#/cm3   ',pver, 'A','CCN concentration at S=0.05%',phys_decomp)
   252)    call addfld('CCN3    ','#/cm3   ',pver, 'A','CCN concentration at S=0.1%',phys_decomp)
   253)    call addfld('CCN4    ','#/cm3   ',pver, 'A','CCN concentration at S=0.2%',phys_decomp)
   254)    call addfld('CCN5    ','#/cm3   ',pver, 'A','CCN concentration at S=0.5%',phys_decomp)
   255)    call addfld('CCN6    ','#/cm3   ',pver, 'A','CCN concentration at S=1.0%',phys_decomp)
   256) 
   257) 
   258)    call addfld('WTKE     ', 'm/s     ', pver, 'A', 'Standard deviation of updraft velocity', phys_decomp)
   259)    call addfld('NDROPMIX ', '#/kg/s  ', pver, 'A', 'Droplet number mixing',                  phys_decomp)
   260)    call addfld('NDROPSRC ', '#/kg/s  ', pver, 'A', 'Droplet number source',                  phys_decomp)
   261)    call addfld('NDROPSNK ', '#/kg/s  ', pver, 'A', 'Droplet number loss by microphysics',    phys_decomp)
   262)    call addfld('NDROPCOL ', '#/m2    ', 1,    'A', 'Column droplet number',                  phys_decomp)
   263) 
   264)    ! set the add_default fields  
   265)    if (history_amwg) then
   266)       call add_default('CCN3', 1, ' ')
   267)    endif
   268) 
   269)    if (history_aerosol .and. prog_modal_aero) then
   270)      do m = 1, ntot_amode
   271)         do l = 0, nspec_amode(m)   ! loop over number + chem constituents
   272)            mm = mam_idx(m,l)
   273)            if (l == 0) then   ! number
   274)               call rad_cnst_get_info(0, m, num_name=tmpname, num_name_cw=tmpname_cw)
   275)            else
   276)               call rad_cnst_get_info(0, m, l, spec_name=tmpname, spec_name_cw=tmpname_cw)
   277)            end if
   278)            fieldname(mm)    = trim(tmpname) // '_mixnuc1'
   279)            fieldname_cw(mm) = trim(tmpname_cw) // '_mixnuc1'
   280)         end do
   281)      end do
   282)    endif
   283) 
   284) 
   285) 
   286) end subroutine ndrop_init

   290) subroutine dropmixnuc( &
   291)    state, ptend, dtmicro, pbuf, wsub, &
   292)    cldn, cldo, tendnd)
   293) 
   294)    ! vertical diffusion and nucleation of cloud droplets
   295)    ! assume cloud presence controlled by cloud fraction
   296)    ! doesn't distinguish between warm, cold clouds
   297) 
   298)    ! arguments
   299)    type(physics_state), target, intent(in)    :: state
   300)    type(physics_ptend),         intent(out)   :: ptend
   301)    real(r8),                    intent(in)    :: dtmicro     ! time step for microphysics (s)
   302) 
   303)    type(physics_buffer_desc), pointer :: pbuf(:)
   304) 
   305)    ! arguments
   306)    real(r8), intent(in) :: wsub(pcols,pver)    ! subgrid vertical velocity
   307)    real(r8), intent(in) :: cldn(pcols,pver)    ! cloud fraction
   308)    real(r8), intent(in) :: cldo(pcols,pver)    ! cloud fraction on previous time step
   309) 
   310)    ! output arguments
   311)    real(r8), intent(out) :: tendnd(pcols,pver) ! change in droplet number concentration (#/kg/s)
   312) 
   313)    !--------------------Local storage-------------------------------------
   314) 
   315)    integer  :: lchnk               ! chunk identifier
   316)    integer  :: ncol                ! number of columns
   317) 
   318)    real(r8), pointer :: ncldwtr(:,:) ! droplet number concentration (#/kg)
   319)    real(r8), pointer :: temp(:,:)    ! temperature (K)
   320)    real(r8), pointer :: omega(:,:)   ! vertical velocity (Pa/s)
   321)    real(r8), pointer :: pmid(:,:)    ! mid-level pressure (Pa)
   322)    real(r8), pointer :: pint(:,:)    ! pressure at layer interfaces (Pa)
   323)    real(r8), pointer :: pdel(:,:)    ! pressure thickess of layer (Pa)
   324)    real(r8), pointer :: rpdel(:,:)   ! inverse of pressure thickess of layer (/Pa)
   325)    real(r8), pointer :: zm(:,:)      ! geopotential height of level (m)
   326) 
   327)    real(r8), pointer :: kvh(:,:)     ! vertical diffusivity (m2/s)
   328) 
   329)    type(ptr2d_t), allocatable :: raer(:)     ! aerosol mass, number mixing ratios
   330)    type(ptr2d_t), allocatable :: qqcw(:)
   331)    real(r8) :: raertend(pver)  ! tendency of aerosol mass, number mixing ratios
   332)    real(r8) :: qqcwtend(pver)  ! tendency of cloudborne aerosol mass, number mixing ratios
   333) 
   334) 
   335)    real(r8), parameter :: zkmin = 0.01_r8, zkmax = 100._r8
   336)    real(r8), parameter :: wmixmin = 0.1_r8        ! minimum turbulence vertical velocity (m/s)
   337)    real(r8) :: sq2pi
   338) 
   339)    integer  :: i, k, l, m, mm, n
   340)    integer  :: km1, kp1
   341)    integer  :: nnew, nsav, ntemp
   342)    integer  :: lptr
   343)    integer  :: nsubmix, nsubmix_bnd
   344)    integer, save :: count_submix(100)
   345)    integer  :: phase ! phase of aerosol
   346) 
   347)    real(r8) :: arg
   348)    real(r8) :: dtinv
   349)    real(r8) :: dtmin, tinv, dtt
   350) 
   351)    real(r8) :: zs(pver) ! inverse of distance between levels (m)
   352)    real(r8) :: qcld(pver) ! cloud droplet number mixing ratio (#/kg)
   353)    real(r8) :: qncld(pver)     ! droplet number nucleated on cloud boundaries
   354)    real(r8) :: srcn(pver)       ! droplet source rate (/s)
   355)    real(r8) :: cs(pcols,pver)      ! air density (kg/m3)
   356)    real(r8) :: csbot(pver)       ! air density at bottom (interface) of layer (kg/m3)
   357)    real(r8) :: csbot_cscen(pver) ! csbot(i)/cs(i,k)
   358)    real(r8) :: dz(pcols,pver)      ! geometric thickness of layers (m)
   359) 
   360)    real(r8) :: wtke(pcols,pver)     ! turbulent vertical velocity at base of layer k (m/s)
   361)    real(r8) :: wtke_cen(pcols,pver) ! turbulent vertical velocity at center of layer k (m/s)
   362)    real(r8) :: wbar, wmix, wmin, wmax
   363) 
   364)    real(r8) :: zn(pver)   ! g/pdel (m2/g) for layer
   365)    real(r8) :: flxconv    ! convergence of flux into lowest layer
   366) 
   367)    real(r8) :: wdiab           ! diabatic vertical velocity
   368)    real(r8) :: ekd(pver)       ! diffusivity for droplets (m2/s)
   369)    real(r8) :: ekk(0:pver)     ! density*diffusivity for droplets (kg/m3 m2/s)
   370)    real(r8) :: ekkp(pver)      ! zn*zs*density*diffusivity
   371)    real(r8) :: ekkm(pver)      ! zn*zs*density*diffusivity
   372) 
   373)    real(r8) :: dum, dumc
   374)    real(r8) :: tmpa
   375)    real(r8) :: dact
   376)    real(r8) :: fluxntot         ! (#/cm2/s)
   377)    real(r8) :: dtmix
   378)    real(r8) :: alogarg
   379)    real(r8) :: overlapp(pver), overlapm(pver) ! cloud overlap
   380) 
   381)    real(r8) :: nsource(pcols,pver)            ! droplet number source (#/kg/s)
   382)    real(r8) :: ndropmix(pcols,pver)           ! droplet number mixing (#/kg/s)
   383)    real(r8) :: ndropcol(pcols)               ! column droplet number (#/m2)
   384)    real(r8) :: cldo_tmp, cldn_tmp
   385)    real(r8) :: tau_cld_regenerate
   386)    real(r8) :: taumix_internal_pver_inv ! 1/(internal mixing time scale for k=pver) (1/s)
   387) 
   388) 
   389)    real(r8), allocatable :: nact(:,:)  ! fractional aero. number  activation rate (/s)
   390)    real(r8), allocatable :: mact(:,:)  ! fractional aero. mass    activation rate (/s)
   391) 
   392)    real(r8), allocatable :: raercol(:,:,:)    ! single column of aerosol mass, number mixing ratios
   393)    real(r8), allocatable :: raercol_cw(:,:,:) ! same as raercol but for cloud-borne phase
   394) 
   395) 
   396)    real(r8) :: na(pcols), va(pcols), hy(pcols)
   397)    real(r8), allocatable :: naermod(:)  ! (1/m3)
   398)    real(r8), allocatable :: hygro(:)    ! hygroscopicity of aerosol mode
   399)    real(r8), allocatable :: vaerosol(:) ! interstit+activated aerosol volume conc (cm3/cm3)
   400) 
   401)    real(r8) :: source(pver)
   402) 
   403)    real(r8), allocatable :: fn(:)              ! activation fraction for aerosol number
   404)    real(r8), allocatable :: fm(:)              ! activation fraction for aerosol mass
   405) 
   406)    real(r8), allocatable :: fluxn(:)           ! number  activation fraction flux (cm/s)
   407)    real(r8), allocatable :: fluxm(:)           ! mass    activation fraction flux (cm/s)
   408)    real(r8)              :: flux_fullact(pver) ! 100%    activation fraction flux (cm/s)
   409)    !     note:  activation fraction fluxes are defined as 
   410)    !     fluxn = [flux of activated aero. number into cloud (#/cm2/s)]
   411)    !           / [aero. number conc. in updraft, just below cloudbase (#/cm3)]
   412) 
   413) 
   414)    real(r8), allocatable :: coltend(:,:)       ! column tendency for diagnostic output
   415)    real(r8), allocatable :: coltend_cw(:,:)    ! column tendency
   416)    real(r8) :: ccn(pcols,pver,psat)    ! number conc of aerosols activated at supersat
   417) 
   418)    !-------------------------------------------------------------------------------
   419) 
   420)    sq2pi = sqrt(2._r8*pi)
   421) 
   422)    lchnk = state%lchnk
   423)    ncol  = state%ncol
   424) 
   425)    ncldwtr  => state%q(:,:,numliq_idx)
   426)    temp     => state%t
   427)    omega    => state%omega
   428)    pmid     => state%pmid
   429)    pint     => state%pint
   430)    pdel     => state%pdel
   431)    rpdel    => state%rpdel
   432)    zm       => state%zm
   433) 
   434)    call pbuf_get_field(pbuf, kvh_idx, kvh)
   435) 
   436) 
   437) 
   438)    arg = 1.0_r8
   439)    if (abs(0.8427_r8 - erf(arg))/0.8427_r8 > 0.001_r8) then
   440)       write(iulog,*) 'erf(1.0) = ',ERF(arg)
   441)       call endrun('dropmixnuc: Error function error')
   442)    endif
   443)    arg = 0.0_r8
   444)    if (erf(arg) /= 0.0_r8) then
   445)       write(iulog,*) 'erf(0.0) = ',erf(arg)
   446)       write(iulog,*) 'dropmixnuc: Error function error'
   447)       call endrun('dropmixnuc: Error function error')
   448)    endif
   449) 
   450)    dtinv = 1._r8/dtmicro
   451) 
   452)    allocate( &
   453)       nact(pver,ntot_amode),          &

   904)          do m = 1, ntot_amode
   905)             mm = mam_idx(m,0)
   906) 
   907)             ! update droplet source
   908)             ! rce-comment- activation source in layer k involves particles from k+1
   909)             !	       srcn(:)=srcn(:)+nact(:,m)*(raercol(:,mm,nsav))
   910)             srcn(top_lev:pver-1) = srcn(top_lev:pver-1) + nact(top_lev:pver-1,m)*(raercol(top_lev+1:pver,mm,nsav))
   911) 
   912)             ! rce-comment- new formulation for k=pver
   913)             !              srcn(  pver  )=srcn(  pver  )+nact(  pver  ,m)*(raercol(  pver,mm,nsav))
   914)             tmpa = raercol(pver,mm,nsav)*nact(pver,m) &
   915)                  + raercol_cw(pver,mm,nsav)*(nact(pver,m) - taumix_internal_pver_inv)
   916)             srcn(pver) = srcn(pver) + max(0.0_r8,tmpa)
   917)          end do
   918)          call explmix(  &
   919)             qcld, srcn, ekkp, ekkm, overlapp,  &
   920)             overlapm, qncld, zero, zero, pver, &
   921)             dtmix, .false.)
   922) 
   923)          ! rce-comment
   924)          !    the interstitial particle mixratio is different in clear/cloudy portions
   925)          !    of a layer, and generally higher in the clear portion.  (we have/had
   926)          !    a method for diagnosing the the clear/cloudy mixratios.)  the activation
   927)          !    source terms involve clear air (from below) moving into cloudy air (above).
   928)          !    in theory, the clear-portion mixratio should be used when calculating 
   929)          !    source terms
   930)          do m = 1, ntot_amode
   931)             mm = mam_idx(m,0)
   932)             ! rce-comment -   activation source in layer k involves particles from k+1
   933)             !	              source(:)= nact(:,m)*(raercol(:,mm,nsav))
   934)             source(top_lev:pver-1) = nact(top_lev:pver-1,m)*(raercol(top_lev+1:pver,mm,nsav))
   935)             ! rce-comment - new formulation for k=pver
   936)             !               source(  pver  )= nact(  pver,  m)*(raercol(  pver,mm,nsav))
   937)             tmpa = raercol(pver,mm,nsav)*nact(pver,m) &
   938)                  + raercol_cw(pver,mm,nsav)*(nact(pver,m) - taumix_internal_pver_inv)
   939)             source(pver) = max(0.0_r8, tmpa)
   940)             flxconv = 0._r8
   941) 
   942)             call explmix( &
   943)                raercol_cw(:,mm,nnew), source, ekkp, ekkm, overlapp, &
   944)                overlapm, raercol_cw(:,mm,nsav), zero, zero, pver,   &
   945)                dtmix, .false.)
   946) 
   947)             call explmix( &
   948)                raercol(:,mm,nnew), source, ekkp, ekkm, overlapp,  &
   949)                overlapm, raercol(:,mm,nsav), zero, flxconv, pver, &
   950)                dtmix, .true., raercol_cw(:,mm,nsav))
   951) 
   952)             do l = 1, nspec_amode(m)
   953)                mm = mam_idx(m,l)
   954)                ! rce-comment -   activation source in layer k involves particles from k+1
   955)                !	          source(:)= mact(:,m)*(raercol(:,mm,nsav))
   956)                source(top_lev:pver-1) = mact(top_lev:pver-1,m)*(raercol(top_lev+1:pver,mm,nsav))
   957)                ! rce-comment- new formulation for k=pver
   958)                !                 source(  pver  )= mact(  pver  ,m)*(raercol(  pver,mm,nsav))
   959)                tmpa = raercol(pver,mm,nsav)*mact(pver,m) &
   960)                     + raercol_cw(pver,mm,nsav)*(mact(pver,m) - taumix_internal_pver_inv)
   961)                source(pver) = max(0.0_r8, tmpa)
   962)                flxconv = 0._r8
   963) 
   964)                call explmix( &
   965)                   raercol_cw(:,mm,nnew), source, ekkp, ekkm, overlapp, &
   966)                   overlapm, raercol_cw(:,mm,nsav), zero, zero, pver,   &
   967)                   dtmix, .false.)
   968) 
   969)                call explmix( &
   970)                   raercol(:,mm,nnew), source, ekkp, ekkm, overlapp,  &
   971)                   overlapm, raercol(:,mm,nsav), zero, flxconv, pver, &
   972)                   dtmix, .true., raercol_cw(:,mm,nsav))
   973) 
   974)             end do
   975)          end do
   976) 
   977)       end do ! old_cloud_nsubmix_loop
   978) 
   979)       ! evaporate particles again if no cloud
   980) 
   981)       do k = top_lev, pver
   982)          if (cldn(i,k) == 0._r8) then
   983)             ! no cloud
   984)             qcld(k)=0._r8
   985) 
   986)             ! convert activated aerosol to interstitial in decaying cloud
   987)             do m = 1, ntot_amode
   988)                mm = mam_idx(m,0)
   989)                raercol(k,mm,nnew)    = raercol(k,mm,nnew) + raercol_cw(k,mm,nnew)
   990)                raercol_cw(k,mm,nnew) = 0._r8
   991) 
   992)                do l = 1, nspec_amode(m)
   993)                   mm = mam_idx(m,l)
   994)                   raercol(k,mm,nnew)    = raercol(k,mm,nnew) + raercol_cw(k,mm,nnew)
   995)                   raercol_cw(k,mm,nnew) = 0._r8
   996)                end do
   997)             end do
   998)          end if
   999)       end do
  1000) 
  1001)       ! droplet number
  1002) 
  1003)       ndropcol(i) = 0._r8
  1004)       do k = top_lev, pver
  1005)          ndropmix(i,k) = (qcld(k) - ncldwtr(i,k))*dtinv - nsource(i,k)
  1006)          tendnd(i,k)   = (max(qcld(k), 1.e-6_r8) - ncldwtr(i,k))*dtinv
  1007)          ndropcol(i)   = ndropcol(i) + ncldwtr(i,k)*pdel(i,k)
  1008)       end do
  1009)       ndropcol(i) = ndropcol(i)/gravit
  1010) 
  1011)       if (prog_modal_aero) then
  1012) 
  1013)          raertend = 0._r8
  1014)          qqcwtend = 0._r8
  1015) 
  1016)          do m = 1, ntot_amode
  1017)             do l = 0, nspec_amode(m)
  1018) 
  1019)                mm   = mam_idx(m,l)
  1020)                lptr = mam_cnst_idx(m,l)
  1021) 
  1022)                raertend(top_lev:pver) = (raercol(top_lev:pver,mm,nnew) - raer(mm)%fld(i,top_lev:pver))*dtinv
  1023)                qqcwtend(top_lev:pver) = (raercol_cw(top_lev:pver,mm,nnew) - qqcw(mm)%fld(i,top_lev:pver))*dtinv
  1024) 
  1025)                coltend(i,mm)    = sum( pdel(i,:)*raertend )/gravit
  1026)                coltend_cw(i,mm) = sum( pdel(i,:)*qqcwtend )/gravit
  1027) 
  1028)                ptend%q(i,:,lptr) = 0.0_r8
  1029)                ptend%q(i,top_lev:pver,lptr) = raertend(top_lev:pver)           ! set tendencies for interstitial aerosol
  1030)                qqcw(mm)%fld(i,:) = 0.0_r8
  1031)                qqcw(mm)%fld(i,top_lev:pver) = raercol_cw(top_lev:pver,mm,nnew) ! update cloud-borne aerosol
  1032)             end do
  1033)          end do
  1034) 
  1035)       end if
  1036) 
  1037)    end do  ! overall_main_i_loop
  1038)    ! end of main loop over i/longitude ....................................
  1039) 
  1040)    call outfld('NDROPCOL', ndropcol, pcols, lchnk)
  1041)    call outfld('NDROPSRC', nsource,  pcols, lchnk)
  1042)    call outfld('NDROPMIX', ndropmix, pcols, lchnk)
  1043)    call outfld('WTKE    ', wtke,     pcols, lchnk)
  1044) 
  1045)    call ccncalc(state, pbuf, cs, ccn)
  1046)    do l = 1, psat
  1047)       call outfld(ccn_name(l), ccn(1,1,l), pcols, lchnk)
  1048)    enddo
  1049) 
  1050)    ! do column tendencies
  1051)    if (prog_modal_aero) then
  1052)       do m = 1, ntot_amode
  1053)          do l = 0, nspec_amode(m)
  1054)             mm = mam_idx(m,l)
  1055)             call outfld(fieldname(mm),    coltend(:,mm),    pcols, lchnk)
  1056)             call outfld(fieldname_cw(mm), coltend_cw(:,mm), pcols, lchnk)
  1057)          end do
  1058)       end do
  1059)    end if
  1060) 
  1061)    deallocate( &
  1062)       nact,       &
  1063)       mact,       &
  1064)       raer,       &
  1065)       qqcw,       &
  1066)       raercol,    &
  1067)       raercol_cw, &
  1068)       coltend,    &
  1069)       coltend_cw, &
  1070)       naermod,    &
  1071)       hygro,      &
  1072)       vaerosol,   &
  1073)       fn,         &
  1074)       fm,         &
  1075)       fluxn,      &
  1076)       fluxm       )
  1077) 
  1078) end subroutine dropmixnuc

  1082) subroutine explmix( q, src, ekkp, ekkm, overlapp, overlapm, &
  1083)    qold, surfrate, flxconv, pver, dt, is_unact, qactold )
  1084) 
  1085)    !  explicit integration of droplet/aerosol mixing
  1086)    !     with source due to activation/nucleation
  1087) 
  1088) 
  1089)    integer, intent(in) :: pver ! number of levels
  1090)    real(r8), intent(out) :: q(pver) ! mixing ratio to be updated
  1091)    real(r8), intent(in) :: qold(pver) ! mixing ratio from previous time step
  1092)    real(r8), intent(in) :: src(pver) ! source due to activation/nucleation (/s)
  1093)    real(r8), intent(in) :: ekkp(pver) ! zn*zs*density*diffusivity (kg/m3 m2/s) at interface
  1094)    ! below layer k  (k,k+1 interface)
  1095)    real(r8), intent(in) :: ekkm(pver) ! zn*zs*density*diffusivity (kg/m3 m2/s) at interface
  1096)    ! above layer k  (k,k+1 interface)
  1097)    real(r8), intent(in) :: overlapp(pver) ! cloud overlap below
  1098)    real(r8), intent(in) :: overlapm(pver) ! cloud overlap above
  1099)    real(r8), intent(in) :: surfrate ! surface exchange rate (/s)
  1100)    real(r8), intent(in) :: flxconv ! convergence of flux from surface
  1101)    real(r8), intent(in) :: dt ! time step (s)
  1102)    logical, intent(in) :: is_unact ! true if this is an unactivated species
  1103)    real(r8), intent(in),optional :: qactold(pver)
  1104)    ! mixing ratio of ACTIVATED species from previous step
  1105)    ! *** this should only be present
  1106)    !     if the current species is unactivated number/sfc/mass
  1107) 
  1108)    integer k,kp1,km1
  1109) 
  1110)    if ( is_unact ) then
  1111)       !     the qactold*(1-overlap) terms are resuspension of activated material
  1112)       do k=top_lev,pver
  1113)          kp1=min(k+1,pver)
  1114)          km1=max(k-1,top_lev)
  1115)          q(k) = qold(k) + dt*( - src(k) + ekkp(k)*(qold(kp1) - qold(k) +       &
  1116)             qactold(kp1)*(1.0_r8-overlapp(k)))               &
  1117)             + ekkm(k)*(qold(km1) - qold(k) +     &
  1118)             qactold(km1)*(1.0_r8-overlapm(k))) )
  1119)          !        force to non-negative
  1120)          !        if(q(k)<-1.e-30)then
  1121)          !           write(iulog,*)'q=',q(k),' in explmix'
  1122)          q(k)=max(q(k),0._r8)
  1123)          !        endif
  1124)       end do
  1125) 
  1126)       !     diffusion loss at base of lowest layer
  1127)       q(pver)=q(pver)-surfrate*qold(pver)*dt+flxconv*dt
  1128)       !        force to non-negative
  1129)       !        if(q(pver)<-1.e-30)then
  1130)       !           write(iulog,*)'q=',q(pver),' in explmix'
  1131)       q(pver)=max(q(pver),0._r8)
  1132)       !        endif
  1133)    else
  1134)       do k=top_lev,pver
  1135)          kp1=min(k+1,pver)
  1136)          km1=max(k-1,top_lev)
  1137)          q(k) = qold(k) + dt*(src(k) + ekkp(k)*(overlapp(k)*qold(kp1)-qold(k)) +      &
  1138)             ekkm(k)*(overlapm(k)*qold(km1)-qold(k)) )
  1139)          !        force to non-negative
  1140)          !        if(q(k)<-1.e-30)then
  1141)          !           write(iulog,*)'q=',q(k),' in explmix'
  1142)          q(k)=max(q(k),0._r8)
  1143)          !        endif
  1144)       end do
  1145)       !     diffusion loss at base of lowest layer
  1146)       q(pver)=q(pver)-surfrate*qold(pver)*dt+flxconv*dt
  1147)       !        force to non-negative
  1148)       !        if(q(pver)<-1.e-30)then
  1149)       !           write(iulog,*)'q=',q(pver),' in explmix'
  1150)       q(pver)=max(q(pver),0._r8)
  1151) 
  1152)    end if
  1153) 
  1154) end subroutine explmix

  1158) subroutine activate_modal(wbar, sigw, wdiab, wminf, wmaxf, tair, rhoair,  &
  1159)    na, nmode, volume, hygro, &
  1160)    fn, fm, fluxn, fluxm, flux_fullact )
  1161) 
  1162)    !      calculates number, surface, and mass fraction of aerosols activated as CCN
  1163)    !      calculates flux of cloud droplets, surface area, and aerosol mass into cloud
  1164)    !      assumes an internal mixture within each of up to nmode multiple aerosol modes
  1165)    !      a gaussiam spectrum of updrafts can be treated.
  1166) 
  1167)    !      mks units
  1168) 
  1169)    !      Abdul-Razzak and Ghan, A parameterization of aerosol activation.
  1170)    !      2. Multiple aerosol types. J. Geophys. Res., 105, 6837-6844.
  1171) 
  1172) 
  1173)    !      input
  1174) 
  1175)    real(r8) :: wbar          ! grid cell mean vertical velocity (m/s)
  1176)    real(r8) :: sigw          ! subgrid standard deviation of vertical vel (m/s)
  1177)    real(r8) :: wdiab         ! diabatic vertical velocity (0 if adiabatic)
  1178)    real(r8) :: wminf         ! minimum updraft velocity for integration (m/s)
  1179)    real(r8) :: wmaxf         ! maximum updraft velocity for integration (m/s)
  1180)    real(r8) :: tair          ! air temperature (K)
  1181)    real(r8) :: rhoair        ! air density (kg/m3)
  1182)    real(r8) :: na(:)      ! aerosol number concentration (/m3)
  1183)    integer  :: nmode      ! number of aerosol modes
  1184)    real(r8) :: volume(:)  ! aerosol volume concentration (m3/m3)
  1185)    real(r8) :: hygro(:)   ! hygroscopicity of aerosol mode
  1186) 
  1187)    !      output
  1188) 
  1189)    real(r8) :: fn(:)      ! number fraction of aerosols activated
  1190)    real(r8) :: fm(:)      ! mass fraction of aerosols activated
  1191)    real(r8) :: fluxn(:)   ! flux of activated aerosol number fraction into cloud (cm/s)
  1192)    real(r8) :: fluxm(:)   ! flux of activated aerosol mass fraction into cloud (cm/s)
  1193)    real(r8) :: flux_fullact   ! flux of activated aerosol fraction assuming 100% activation (cm/s)
  1194)    !    rce-comment
  1195)    !    used for consistency check -- this should match (ekd(k)*zs(k))
  1196)    !    also, fluxm/flux_fullact gives fraction of aerosol mass flux
  1197)    !       that is activated
  1198) 
  1199)    !      local
  1200) 
  1201)    integer, parameter:: nx=200
  1202)    integer iquasisect_option, isectional
  1203)    real(r8) integ,integf
  1204)    real(r8), parameter :: p0 = 1013.25e2_r8    ! reference pressure (Pa)
  1205)    real(r8) xmin(nmode),xmax(nmode) ! ln(r) at section interfaces
  1206)    real(r8) volmin(nmode),volmax(nmode) ! volume at interfaces
  1207)    real(r8) tmass ! total aerosol mass concentration (g/cm3)
  1208)    real(r8) sign(nmode)    ! geometric standard deviation of size distribution
  1209)    real(r8) rm ! number mode radius of aerosol at max supersat (cm)
  1210)    real(r8) pres ! pressure (Pa)
  1211)    real(r8) path ! mean free path (m)
  1212)    real(r8) diff ! diffusivity (m2/s)
  1213)    real(r8) conduct ! thermal conductivity (Joule/m/sec/deg)
  1214)    real(r8) diff0,conduct0
  1215)    real(r8) es ! saturation vapor pressure
  1216)    real(r8) qs ! water vapor saturation mixing ratio
  1217)    real(r8) dqsdt ! change in qs with temperature
  1218)    real(r8) dqsdp ! change in qs with pressure
  1219)    real(r8) g ! thermodynamic function (m2/s)
  1220)    real(r8) zeta(nmode), eta(nmode)
  1221)    real(r8) lnsmax ! ln(smax)
  1222)    real(r8) alpha
  1223)    real(r8) gamma
  1224)    real(r8) beta
  1225)    real(r8) sqrtg(nmode)
  1226)    real(r8) :: amcube(nmode) ! cube of dry mode radius (m)
  1227)    real(r8) :: smcrit(nmode) ! critical supersatuation for activation
  1228)    real(r8) :: lnsm(nmode) ! ln(smcrit)
  1229)    real(r8) smc(nmode) ! critical supersaturation for number mode radius
  1230)    real(r8) sumflx_fullact
  1231)    real(r8) sumflxn(nmode)
  1232)    real(r8) sumflxm(nmode)
  1233)    real(r8) sumfn(nmode)
  1234)    real(r8) sumfm(nmode)
  1235)    real(r8) fnold(nmode)   ! number fraction activated
  1236)    real(r8) fmold(nmode)   ! mass fraction activated
  1237)    real(r8) wold,gold
  1238)    real(r8) alogam
  1239)    real(r8) rlo,rhi,xint1,xint2,xint3,xint4
  1240)    real(r8) wmin,wmax,w,dw,dwmax,dwmin,wnuc,dwnew,wb
  1241)    real(r8) dfmin,dfmax,fnew,fold,fnmin,fnbar,fsbar,fmbar
  1242)    real(r8) alw,sqrtalw
  1243)    real(r8) smax
  1244)    real(r8) x,arg
  1245)    real(r8) xmincoeff,xcut,volcut,surfcut
  1246)    real(r8) z,z1,z2,wf1,wf2,zf1,zf2,gf1,gf2,gf
  1247)    real(r8) etafactor1,etafactor2(nmode),etafactor2max
  1248)    integer m,n
  1249)    !      numerical integration parameters
  1250)    real(r8), parameter :: eps=0.3_r8,fmax=0.99_r8,sds=3._r8
  1251) 
  1252)    real(r8), parameter :: namin=1.e6_r8   ! minimum aerosol number concentration (/m3)
  1253) 
  1254)    integer ndist(nx)  ! accumulates frequency distribution of integration bins required
  1255)    data ndist/nx*0/
  1256)    save ndist
  1257) 
  1258)    fn(:)=0._r8
  1259)    fm(:)=0._r8
  1260)    fluxn(:)=0._r8
  1261)    fluxm(:)=0._r8
  1262)    flux_fullact=0._r8
  1263) 
  1264)    if(nmode.eq.1.and.na(1).lt.1.e-20_r8)return
  1265) 
  1266)    if(sigw.le.1.e-5_r8.and.wbar.le.0._r8)return
  1267) 
  1268)    pres=rair*rhoair*tair
  1269)    diff0=0.211e-4_r8*(p0/pres)*(tair/t0)**1.94_r8
  1270)    conduct0=(5.69_r8+0.017_r8*(tair-t0))*4.186e2_r8*1.e-5_r8 ! convert to J/m/s/deg
  1271)    call qsat(tair, pres, es, qs)
  1272)    dqsdt=latvap/(rh2o*tair*tair)*qs
  1273)    alpha=gravit*(latvap/(cpair*rh2o*tair*tair)-1._r8/(rair*tair))
  1274)    gamma=(1+latvap/cpair*dqsdt)/(rhoair*qs)
  1275)    etafactor2max=1.e10_r8/(alpha*wmaxf)**1.5_r8 ! this should make eta big if na is very small.

  1278)       if(volume(m).gt.1.e-39_r8.and.na(m).gt.1.e-39_r8)then
  1279)          !            number mode radius (m)
  1280)          !           write(iulog,*)'alogsig,volc,na=',alogsig(m),volc(m),na(m)
  1281)          amcube(m)=(3._r8*volume(m)/(4._r8*pi*exp45logsig(m)*na(m)))  ! only if variable size dist
  1282)          !           growth coefficent Abdul-Razzak & Ghan 1998 eqn 16
  1283)          !           should depend on mean radius of mode to account for gas kinetic effects
  1284)          !           see Fountoukis and Nenes, JGR2005 and Meskhidze et al., JGR2006
  1285)          !           for approriate size to use for effective diffusivity.
  1286)          g=1._r8/(rhoh2o/(diff0*rhoair*qs)                                    &
  1287)             +latvap*rhoh2o/(conduct0*tair)*(latvap/(rh2o*tair)-1._r8))
  1288)          sqrtg(m)=sqrt(g)
  1289)          beta=2._r8*pi*rhoh2o*g*gamma
  1290)          etafactor2(m)=1._r8/(na(m)*beta*sqrtg(m))
  1291)          if(hygro(m).gt.1.e-10_r8)then
  1292)             smc(m)=2._r8*aten*sqrt(aten/(27._r8*hygro(m)*amcube(m))) ! only if variable size dist
  1293)          else
  1294)             smc(m)=100._r8
  1295)          endif
  1296)          !	    write(iulog,*)'sm,hygro,amcube=',smcrit(m),hygro(m),amcube(m)
  1297)       else
  1298)          g=1._r8/(rhoh2o/(diff0*rhoair*qs)                                    &
  1299)             +latvap*rhoh2o/(conduct0*tair)*(latvap/(rh2o*tair)-1._r8))

  1537) subroutine maxsat(zeta,eta,nmode,smc,smax)
  1538) 
  1539)    !      calculates maximum supersaturation for multiple
  1540)    !      competing aerosol modes.
  1541) 
  1542)    !      Abdul-Razzak and Ghan, A parameterization of aerosol activation.
  1543)    !      2. Multiple aerosol types. J. Geophys. Res., 105, 6837-6844.
  1544) 
  1545)    integer  :: nmode ! number of modes
  1546)    real(r8) :: smc(nmode) ! critical supersaturation for number mode radius
  1547)    real(r8) :: zeta(nmode)
  1548)    real(r8) :: eta(nmode)
  1549)    real(r8) :: smax ! maximum supersaturation
  1550)    integer  :: m  ! mode index
  1551)    real(r8) :: sum, g1, g2, g1sqrt, g2sqrt
  1552) 
  1553)    do m=1,nmode
  1554)       if(zeta(m).gt.1.e5_r8*eta(m).or.smc(m)*smc(m).gt.1.e5_r8*eta(m))then
  1555)          !            weak forcing. essentially none activated
  1556)          smax=1.e-20_r8
  1557)       else
  1558)          !            significant activation of this mode. calc activation all modes.
  1559)          go to 1
  1560)       endif
  1561)    enddo
  1562) 
  1563)    return
  1564) 
  1565) 1  continue
  1566) 
  1567)    sum=0
  1568)    do m=1,nmode
  1569)       if(eta(m).gt.1.e-20_r8)then
  1570)          g1=zeta(m)/eta(m)
  1571)          g1sqrt=sqrt(g1)
  1572)          g1=g1sqrt*g1
  1573)          g2=smc(m)/sqrt(eta(m)+3._r8*zeta(m))
  1574)          g2sqrt=sqrt(g2)
  1575)          g2=g2sqrt*g2
  1576)          sum=sum+(f1(m)*g1+f2(m)*g2)/(smc(m)*smc(m))
  1577)       else
  1578)          sum=1.e20_r8
  1579)       endif
  1580)    enddo
  1581) 
  1582)    smax=1._r8/sqrt(sum)
  1583) 
  1584) end subroutine maxsat

  1588) subroutine ccncalc(state, pbuf, cs, ccn)
  1589) 
  1590)    ! calculates number concentration of aerosols activated as CCN at
  1591)    ! supersaturation supersat.
  1592)    ! assumes an internal mixture of a multiple externally-mixed aerosol modes
  1593)    ! cgs units
  1594) 
  1595)    ! Ghan et al., Atmos. Res., 1993, 198-221.
  1596) 
  1597)    ! arguments
  1598) 
  1599)    type(physics_state), target, intent(in)    :: state
  1600)    type(physics_buffer_desc),   pointer       :: pbuf(:)
  1601) 
  1602) 
  1603)    real(r8), intent(in)  :: cs(pcols,pver)       ! air density (kg/m3)
  1604)    real(r8), intent(out) :: ccn(pcols,pver,psat) ! number conc of aerosols activated at supersat (#/m3)
  1605) 
  1606)    ! local
  1607) 
  1608)    integer :: lchnk ! chunk index
  1609)    integer :: ncol  ! number of columns
  1610)    real(r8), pointer :: tair(:,:)     ! air temperature (K)
  1611) 
  1612)    real(r8) naerosol(pcols) ! interstit+activated aerosol number conc (/m3)
  1613)    real(r8) vaerosol(pcols) ! interstit+activated aerosol volume conc (m3/m3)
  1614) 
  1615)    real(r8) amcube(pcols)
  1616)    real(r8) super(psat) ! supersaturation
  1617)    real(r8), allocatable :: amcubecoef(:)
  1618)    real(r8), allocatable :: argfactor(:)
  1619)    real(r8) :: surften       ! surface tension of water w/respect to air (N/m)
  1620)    real(r8) surften_coef
  1621)    real(r8) a(pcols) ! surface tension parameter
  1622)    real(r8) hygro(pcols)  ! aerosol hygroscopicity
  1623)    real(r8) sm(pcols)  ! critical supersaturation at mode radius
  1624)    real(r8) arg(pcols)
  1625)    !     mathematical constants
  1626)    real(r8) twothird,sq2
  1627)    integer l,m,n,i,k
  1628)    real(r8) log,cc
  1629)    real(r8) smcoefcoef,smcoef(pcols)
  1630)    integer phase ! phase of aerosol
  1631)    !-------------------------------------------------------------------------------
  1632) 
  1633)    lchnk = state%lchnk
  1634)    ncol  = state%ncol
  1635)    tair  => state%t
  1636) 
  1637)    allocate( &
  1638)       amcubecoef(ntot_amode), &
  1639)       argfactor(ntot_amode)   )
  1640) 
  1641)    super(:)=supersat(:)*0.01_r8
  1642)    sq2=sqrt(2._r8)
  1643)    twothird=2._r8/3._r8
  1644)    surften=0.076_r8
  1645)    surften_coef=2._r8*mwh2o*surften/(r_universal*rhoh2o)
  1646)    smcoefcoef=2._r8/sqrt(27._r8)
  1647) 
  1648)    do m=1,ntot_amode
  1649)       amcubecoef(m)=3._r8/(4._r8*pi*exp45logsig(m))
  1650)       argfactor(m)=twothird/(sq2*alogsig(m))
  1651)    end do
  1652) 
  1653)    ccn = 0._r8
  1654)    do k=top_lev,pver
  1655) 
  1656)       do i=1,ncol
  1657)          a(i)=surften_coef/tair(i,k)
  1658)          smcoef(i)=smcoefcoef*a(i)*sqrt(a(i))
  1659)       end do
  1660) 
  1661)       do m=1,ntot_amode
  1662) 
  1663)          phase=3 ! interstitial+cloudborne
  1664) 
  1665)          call loadaer( &
  1666)             state, pbuf, 1, ncol, k, &
  1667)             m, cs, phase, naerosol, vaerosol, &
  1668)             hygro)
  1669) 
  1670)          where(naerosol(:ncol)>1.e-3_r8)
  1671)             amcube(:ncol)=amcubecoef(m)*vaerosol(:ncol)/naerosol(:ncol)
  1672)             sm(:ncol)=smcoef(:ncol)/sqrt(hygro(:ncol)*amcube(:ncol)) ! critical supersaturation
  1673)          elsewhere
  1674)             sm(:ncol)=1._r8 ! value shouldn't matter much since naerosol is small
  1675)          endwhere
  1676)          do l=1,psat
  1677)             do i=1,ncol
  1678)                arg(i)=argfactor(m)*log(sm(i)/super(l))
  1679)                ccn(i,k,l)=ccn(i,k,l)+naerosol(i)*0.5_r8*(1._r8-erf(arg(i)))
  1680)             enddo
  1681)          enddo
  1682)       enddo
  1683)    enddo
  1684)    ccn(:ncol,:,:)=ccn(:ncol,:,:)*1.e-6_r8 ! convert from #/m3 to #/cm3
  1685) 
  1686)    deallocate( &
  1687)       amcubecoef, &
  1688)       argfactor   )
  1689)

  1694) subroutine loadaer( &
  1695)    state, pbuf, istart, istop, k, &
  1696)    m, cs, phase, naerosol, &
  1697)    vaerosol, hygro)
  1698) 
  1699)    ! return aerosol number, volume concentrations, and bulk hygroscopicity
  1700) 
  1701)    ! input arguments
  1702)    type(physics_state), target, intent(in) :: state
  1703)    type(physics_buffer_desc),   pointer    :: pbuf(:)
  1704) 
  1705)    integer,  intent(in) :: istart      ! start column index (1 <= istart <= istop <= pcols)
  1706)    integer,  intent(in) :: istop       ! stop column index  
  1707)    integer,  intent(in) :: m           ! mode index
  1708)    integer,  intent(in) :: k           ! level index
  1709)    real(r8), intent(in) :: cs(:,:)     ! air density (kg/m3)
  1710)    integer,  intent(in) :: phase       ! phase of aerosol: 1 for interstitial, 2 for cloud-borne, 3 for sum
  1711) 
  1712)    ! output arguments
  1713)    real(r8), intent(out) :: naerosol(:)  ! number conc (1/m3)
  1714)    real(r8), intent(out) :: vaerosol(:)  ! volume conc (m3/m3)

  1716) 
  1717)    ! internal
  1718)    integer  :: lchnk               ! chunk identifier
  1719) 
  1720)    real(r8), pointer :: raer(:,:) ! interstitial aerosol mass, number mixing ratios
  1721)    real(r8), pointer :: qqcw(:,:) ! cloud-borne aerosol mass, number mixing ratios
  1722)    real(r8) :: specdens, spechygro

  1724)    real(r8) :: vol(pcols) ! aerosol volume mixing ratio
  1725)    integer  :: i, l
  1726)    !-------------------------------------------------------------------------------

  1728)    lchnk = state%lchnk
  1729) 
  1730)    do i = istart, istop
  1731)       vaerosol(i) = 0._r8
  1732)       hygro(i)    = 0._r8

  1736)

  1738)       call rad_cnst_get_aer_mmr(0, m, l, 'c', state, pbuf, qqcw)

  1740)

  1748)          end do
  1749)       else if (phase == 1) then
  1750)          do i = istart, istop
  1751)             vol(i) = max(raer(i,k), 0._r8)/specdens
  1752)          end do
  1753)       else

  1757)

  1759)          vaerosol(i) = vaerosol(i) + vol(i)

  1761)       end do
  1762) 
  1763)    end do
  1764)

  1766)       if (vaerosol(i) > 1.0e-30_r8) then   ! +++xl add 8/2/2007
  1767)          hygro(i)    = hygro(i)/(vaerosol(i))
  1768)          vaerosol(i) = vaerosol(i)*cs(i,k)
  1769)       else

  1773)    end do
