   108)     concld_idx, &       		! Convective cloud fraction

   118)     kvm_idx, &				! CLUBB eddy diffusivity on mom levels
   119)     pblh_idx, &                         ! PBL pbuf

   121)     tke_idx, &                          ! turbulent kinetic energy

   139)   subroutine clubb_register_cam( )
   140) !-------------------------------------------------------------------------------
   141) ! Description:
   142) !   Register the constituents and fields in the physics buffer
   143) ! Author: P. Bogenschutz, C. Craig, A. Gettelman
   144) !
   145) !-------------------------------------------------------------------------------
   146) #ifdef CLUBB_SGS
   147) 
   148)     !------------------------------------------------ !
   149)     ! Register physics buffer fields and constituents !
   150)     !------------------------------------------------ !
   151) 
   152)     !  Add CLUBB fields to pbuf 
   153)     use physics_buffer,  only: pbuf_add_field, dtype_r8, dyn_time_lvls
   154)     use ppgrid,          only: pver, pverp, pcols
   155)     
   156)     call phys_getopts( eddy_scheme_out = eddy_scheme, &
   157)                        do_tms_out      = do_tms)	      
   158) 
   159)     !  put pbuf_add calls here (see macrop_driver.F90 for sample) use indicies defined at top
   160)     call pbuf_add_field('pblh',       'global', dtype_r8, (/pcols/),                    pblh_idx)
   161)     call pbuf_add_field('tke',        'global', dtype_r8, (/pcols, pverp/),             tke_idx)
   162)     call pbuf_add_field('kvh',        'global', dtype_r8, (/pcols, pverp/),             kvh_idx)
   163)     call pbuf_add_field('kvm',        'global', dtype_r8, (/pcols, pverp/),             kvm_idx)
   164)     call pbuf_add_field('tpert',      'global', dtype_r8, (/pcols/),                    tpert_idx)
   165)     call pbuf_add_field('AST',        'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),    ast_idx)
   166)     call pbuf_add_field('AIST',       'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),    aist_idx)
   167)     call pbuf_add_field('ALST',       'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),    alst_idx)
   168)     call pbuf_add_field('QIST',       'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),    qist_idx)
   169)     call pbuf_add_field('QLST',       'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),    qlst_idx)
   170)     call pbuf_add_field('CONCLD',     'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),    concld_idx)
   171)     call pbuf_add_field('CLD',        'global', dtype_r8, (/pcols,pver,dyn_time_lvls/),    cld_idx)
   172)     call pbuf_add_field('FICE',       'physpkg',dtype_r8, (/pcols,pver/),                  fice_idx)
   173)     call pbuf_add_field('CMELIQ',     'physpkg',dtype_r8, (/pcols,pver/),                  cmeliq_idx)
   174)     
   175)     call pbuf_add_field('WP2',        'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), wp2_idx)
   176)     call pbuf_add_field('WP3',        'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), wp3_idx)
   177)     call pbuf_add_field('WPTHLP',     'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), wpthlp_idx)
   178)     call pbuf_add_field('WPRTP',      'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), wprtp_idx)
   179)     call pbuf_add_field('RTPTHLP',    'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), rtpthlp_idx)
   180)     call pbuf_add_field('RTP2',       'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), rtp2_idx)
   181)     call pbuf_add_field('THLP2',      'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), thlp2_idx)
   182)     call pbuf_add_field('UP2',        'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), up2_idx)
   183)     call pbuf_add_field('VP2',        'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), vp2_idx)
   184)     call pbuf_add_field('UPWP',       'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), upwp_idx)
   185)     call pbuf_add_field('VPWP',       'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), vpwp_idx)
   186)     call pbuf_add_field('THLM',       'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), thlm_idx)
   187)     call pbuf_add_field('RTM',        'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), rtm_idx)
   188)     call pbuf_add_field('UM',         'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), um_idx)
   189)     call pbuf_add_field('VM',         'global', dtype_r8, (/pcols,pverp,dyn_time_lvls/), vm_idx)
   190) 
   191) #endif 
   192) 
   193)   end subroutine clubb_register_cam

   199)   subroutine clubb_ini_cam(pbuf2d)
   200) !-------------------------------------------------------------------------------
   201) ! Description:
   202) !   Initialize UWM CLUBB.
   203) ! Author: Cheryl Craig March 2011
   204) ! Modifications: Pete Bogenschutz 2011 March and onward
   205) ! Origin: Based heavily on UWM clubb_init.F90
   206) ! References:
   207) !   None
   208) !-------------------------------------------------------------------------------
   209) 
   210) 
   211) 
   212) #ifdef CLUBB_SGS
   213) 
   214)     !  From CAM libraries
   215)     use physics_types, 		only: physics_state, physics_ptend
   216)     use cam_history,     	only: addfld, add_default, phys_decomp
   217)     use ppgrid, 		only: pver, pverp, pcols
   218)     use ref_pres,               only: pref_mid
   219)     use hb_diff,                only: init_hb_diff
   220)     use trb_mtn_stress,         only: init_tms
   221)     use rad_constituents,       only: rad_cnst_get_info, rad_cnst_get_mode_num_idx, rad_cnst_get_mam_mmr_idx
   222) 
   223)     !  From the CLUBB libraries
   224)     use clubb_core, 	        only: setup_clubb_core
   225)     use clubb_precision, 	only: time_precision
   226)     use error_code, 		only: set_clubb_debug_level ! Subroutines
   227)     use parameter_indices, 	only: nparams ! Constant
   228)     use parameters_tunable, 	only: read_parameters ! Subroutine
   229)     use stats_variables,        only: l_stats, l_stats_samp, l_grads, l_output_rad_files, &
   230)                                       zt, zm, sfc, rad_zt, rad_zm
   231)     use namelist_utils,         only: find_group_name
   232)     use units,                  only: getunit, freeunit
   233)     use abortutils,             only: endrun
   234)     use error_messages,         only: handle_errmsg
   235)     use time_manager,         	only: is_first_step
   236)     use constants_clubb,        only: em_min, w_tol_sqd, rt_tol, thl_tol  
   237) 
   238) 
   239)     !  These are only needed if we're using a passive scalar
   240)     use array_index, 		only: iisclr_rt, iisclr_thl, iisclr_CO2, &    ! [kg/kg]/[K]/[1e6 mol/mol]
   241)       				      iiedsclr_rt, iiedsclr_thl, iiedsclr_CO2 ! "    "
   242)     use constituents,           only: cnst_get_ind
   243)     use phys_control,           only: phys_getopts
   244) 
   245) #endif
   246) 
   247)     use physics_buffer,         only: pbuf_get_index, pbuf_set_field, physics_buffer_desc
   248)     implicit none
   249)     !  Input Variables
   250)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   251) 
   252) #ifdef CLUBB_SGS
   253) 
   254)     real(kind=time_precision) :: dum1, dum2, dum3
   255)     
   256)     real(8), dimension(nparams)  :: clubb_params    ! These adjustable CLUBB parameters (C1, C2 ...)
   257) 
   258)     logical :: clubb_history, clubb_rad_history ! Stats enabled (T/F)
   259) 
   260)     ! The similar name to clubb_history is unfortunate...
   261)     logical :: history_amwg, history_clubb
   262) 
   263)     character(len=128) :: errstring             ! error status for CLUBB init
   264) 
   265)     integer :: err_code, iunit 			! Code for when CLUBB fails
   266)     integer :: i, j, k, l                       ! Indices
   267)     integer :: read_status      		! Length of a string
   268)     integer :: ntop_eddy    			! Top    interface level to which eddy vertical diffusion is applied ( = 1 )
   269)     integer :: nbot_eddy   			! Bottom interface level to which eddy vertical diffusion is applied ( = pver )
   270)     integer :: nmodes, nspec, pmam_ncnst, m
   271)     integer :: ixnumliq
   272)     integer :: lptr
   273) 
   274)     real(r8)  :: zt_g(pverp)                        ! Height dummy array
   275)     real(r8)  :: zi_g(pverp)                        ! Height dummy array
   276) 
   277)     namelist /clubb_his_nl/ clubb_history, clubb_rad_history
   278) 
   279)     !----- Begin Code -----
   280) 
   281)     ! ----------------------------------------------------------------- !
   282)     ! Determine how many constituents CLUBB will transport.  Note that  
   283)     ! CLUBB does not transport aerosol consituents.  Therefore, need to 
   284)     ! determine how many aerosols constituents there are and subtract that
   285)     ! off of pcnst (the total consituents) 
   286)     ! ----------------------------------------------------------------- !
   287) 
   288)     call phys_getopts(prog_modal_aero_out=prog_modal_aero, &
   289)                       history_amwg_out=history_amwg, &
   290)                       history_clubb_out=history_clubb)
   291) 
   292)    !  Select variables to apply tendencies back to CAM
   293)  
   294)    ! Initialize all consituents to true to start
   295)    lq(1:pcnst) = .true.
   296)    edsclr_dim  = pcnst
   297) 
   298)    if (prog_modal_aero) then
   299)       ! Turn off modal aerosols and decrement edsclr_dim accordingly
   300)       call rad_cnst_get_info(0, nmodes=nmodes)
   301) 
   302)       do m = 1, nmodes
   303)          call rad_cnst_get_mode_num_idx(m, lptr)
   304)          lq(lptr)=.false.
   305)          edsclr_dim = edsclr_dim-1
   306) 
   307)          call rad_cnst_get_info(0, m, nspec=nspec)
   308)          do l = 1, nspec
   309)             call rad_cnst_get_mam_mmr_idx(m, l, lptr)
   310)             lq(lptr)=.false.
   311)             edsclr_dim = edsclr_dim-1
   312)          end do
   313)       end do
   314) 
   315) 
   316)       !  In addition, if running with MAM, droplet number is transported
   317)       !  in dropmixnuc, therefore we do NOT want CLUBB to apply transport
   318)       !  tendencies to avoid double counted.  Else, we apply tendencies.
   319)       call cnst_get_ind('NUMLIQ',ixnumliq)
   320)       lq(ixnumliq)	= .false.
   321)       edsclr_dim = edsclr_dim-1
   322)    endif
   323) 
   324)     ! ----------------------------------------------------------------- !
   325)     ! Set the debug level.  Level 2 has additional computational expense since
   326)     ! it checks the array variables in CLUBB for invalid values.
   327)     ! ----------------------------------------------------------------- !
   328)     call set_clubb_debug_level( 0 )
   329) 
   330)     ! ----------------------------------------------------------------- !
   331)     ! use pbuf_get_fld_idx to get existing physics buffer fields from other
   332)     ! physics packages (e.g. tke)
   333)     ! ----------------------------------------------------------------- !
   334) 
   335)     !  Determine if we want clubb_history to be output  
   336)     clubb_history      = .false.   ! Initialize to false
   337)     l_stats            = .false.   ! Initialize to false
   338)     l_output_rad_files = .false.   ! Initialize to false
   339) 
   340)     !  Read namelist to determine if CLUBB history should be called
   341)     if (masterproc) then
   342)       iunit= getunit()
   343)       open(unit=iunit,file="atm_in",status='old')
   344)       call find_group_name(iunit, 'clubb_his_nl', status=read_status)
   345)       if (read_status == 0) then
   346)          read(unit=iunit, nml=clubb_his_nl, iostat=read_status)
   347)          if (read_status /= 0) then
   348)             call endrun('clubb_tend_cam:  error reading namelist')
   349)          end if
   350)       end if
   351)       close(unit=iunit) 
   352)       call freeunit(iunit)
   353)     end if
   354) 
   355) #ifdef SPMD
   356)       ! Broadcast namelist variables
   357)       call mpibcast(clubb_history,      1,   mpilog,   0, mpicom)
   358)       call mpibcast(clubb_rad_history,  1,   mpilog,   0, mpicom)
   359) #endif
   360) 
   361)     !  Overwrite defaults if they are true
   362)     if (clubb_history) l_stats = .true.
   363)     if (clubb_rad_history) l_output_rad_files = .true. 
   364) 
   365)     !  Defaults
   366)     l_stats_samp = .false.
   367)     l_grads = .false.
   368) 
   369)     !  Overwrite defaults if needbe     
   370)     if (l_stats) l_stats_samp = .true.
   371) 
   372)     !  Define physics buffers indexes
   373)     cld_idx     = pbuf_get_index('CLD')         ! Cloud fraction
   374)     concld_idx  = pbuf_get_index('CONCLD')      ! Convective cloud cover
   375)     ast_idx     = pbuf_get_index('AST')         ! Stratiform cloud fraction
   376)     alst_idx    = pbuf_get_index('ALST')        ! Liquid stratiform cloud fraction
   377)     aist_idx    = pbuf_get_index('AIST')        ! Ice stratiform cloud fraction
   378)     qlst_idx    = pbuf_get_index('QLST')        ! Physical in-stratus LWC 
   379)     qist_idx    = pbuf_get_index('QIST')        ! Physical in-stratus IWC
   380)     dp_frac_idx = pbuf_get_index('DP_FRAC')     ! Deep convection cloud fraction
   381)     icwmrdp_idx = pbuf_get_index('ICWMRDP')     ! In-cloud deep convective mixing ratio
   382)     sh_frac_idx = pbuf_get_index('SH_FRAC')     ! Shallow convection cloud fraction
   383)     relvar_idx  = pbuf_get_index('RELVAR')      ! Relative cloud water variance
   384)     accre_enhan_idx = pbuf_get_index('ACCRE_ENHAN') ! accretion enhancement for MG
   385) 
   386)     iisclr_rt  = -1
   387)     iisclr_thl = -1
   388)     iisclr_CO2 = -1
   389) 
   390)     iiedsclr_rt  = -1
   391)     iiedsclr_thl = -1
   392)     iiedsclr_CO2 = -1
   393)     
   394)     ! ----------------------------------------------------------------- !
   395)     ! Setup CLUBB core
   396)     ! ----------------------------------------------------------------- !
   397)     
   398)     !  Read in parameters for CLUBB.  Just read in default values 
   399) !$OMP PARALLEL
   400)       call read_parameters( -99, "", clubb_params )
   401) !$OMP END PARALLEL
   402)       
   403)     !  Fill in dummy arrays for height.  Note that these are overwrote
   404)     !  at every CLUBB step to physical values.    
   405)     do k=1,pverp
   406)       zt_g(k) = ((k-1)*1000._r8)-500._r8  !  this is dummy garbage
   407)       zi_g(k) = (k-1)*1000._r8            !  this is dummy garbage
   408)     enddo
   409)    
   410)     !  Set up CLUBB core.  Note that some of these inputs are overwrote
   411)     !  when clubb_tend_cam is called.  The reason is that heights can change
   412)     !  at each time step, which is why dummy arrays are read in here for heights
   413)     !  as they are immediately overwrote.     
   414) !$OMP PARALLEL
   415)     call setup_clubb_core     &
   416)          ( pverp, theta0, ts_nudge, &                                 ! In
   417)            hydromet_dim,  sclr_dim, &                                 ! In
   418)            sclr_tol, edsclr_dim, clubb_params, &                      ! In
   419)            l_host_applies_sfc_fluxes, &                               ! In
   420)            l_uv_nudge, saturation_equation,  &                        ! In
   421)            l_implemented, grid_type, zi_g(2), zi_g(1), zi_g(pverp), & ! In
   422)            zi_g(1:pverp), zt_g(1:pverp), &                            ! In
   423)            host_dx, host_dy, zi_g(1), &                               ! In
   424)            err_code )
   425) !$OMP END PARALLEL
   426) 	   
   427)     ! ----------------------------------------------------------------- !
   428)     ! Set-up HB diffusion.  Only initialized to diagnose PBL depth      !
   429)     ! ----------------------------------------------------------------- !
   430) 
   431)     ! Initialize eddy diffusivity module
   432)     
   433)     ntop_eddy = 1    ! if >1, must be <= nbot_molec
   434)     nbot_eddy = pver ! currently always pver
   435)     
   436)     call init_hb_diff( gravit, cpair, ntop_eddy, nbot_eddy, pref_mid, karman, eddy_scheme )
   437)     
   438)     ! ----------------------------------------------------------------- !
   439)     ! Initialize turbulent mountain stress module                       !
   440)     ! ------------------------------------------------------------------!
   441)

  1470)    ! ------------------------------------------------------------ ! 
  1471)    ! ------------------------------------------------------------ !
  1472)    ! The rest of the code deals with diagnosing variables         !
  1473)    ! for microphysics/radiation computation and macrophysics      !
  1474)    ! ------------------------------------------------------------ !
  1475)    ! ------------------------------------------------------------ !
  1476)    ! ------------------------------------------------------------ !
  1477) 
  1478)    ! ------------------------------------------------- !
  1479)    ! Diagnose relative cloud water variance	       !
  1480)    ! ------------------------------------------------- !
  1481) 
  1482)    
  1483)    relvar(:,:) = 1.0_r8  ! default
  1484)    
  1485)    where (rcm(:ncol,:pver) /= 0 .and. qclvar(:ncol,:pver) /= 0) &
  1486)          relvar(:ncol,:pver) = min(1.0_r8,max(0.001_r8,rcm(:ncol,:pver)**2/qclvar(:ncol,:pver)))
  1487)    
  1488)    ! ------------------------------------------------- !
  1489)    ! Optional Accretion enhancement factor	       !
  1490)    ! ------------------------------------------------- !   
  1491)    
  1492)    accre_enhan(:ncol,:pver) = 1._r8+0.65_r8*(1.0_r8/relvar(:ncol,:pver))
  1493)    
  1494)    ! ------------------------------------------------- !
  1495)    ! Diagnose some output variables		       !
  1496)    ! ------------------------------------------------- !
  1497) 
  1498)    !  density
  1499)    rho(:ncol,1:pver) = state1%pmid(:ncol,1:pver)/(rair*state1%t(:ncol,1:pver))
  1500)    rho(:ncol,pverp)  = state1%ps(:ncol)/(rair*state1%t(:ncol,pver))
  1501) 
  1502)    eps = rair/rh2o
  1503)    wpthvp(:,:) = 0.0_r8
  1504)    do k=1,pver
  1505)      do i=1,ncol
  1506)         !  buoyancy flux
  1507)         wpthvp(i,k) = wpthlp(i,k)+((1._r8-eps)/eps)*theta0*wprtp(i,k)+((latvap/cpair)* &
  1508)           state1%exner(i,k)-(1._r8/eps)*theta0)*wprcp(i,k)
  1509) 
  1510)         !  total water mixing ratio
  1511)         qt_output(i,k) = state1%q(i,k,ixq)+state1%q(i,k,ixcldliq)+state1%q(i,k,ixcldice)
  1512)         !  liquid water potential temperature
  1513)         thetal_output(i,k) = (state1%t(i,k)*state1%exner(i,k))-(latvap/cpair)*state1%q(i,k,ixcldliq)
  1514)         !  liquid water static energy
  1515)         sl_output(i,k) = cpair*state1%t(i,k)+gravit*state1%zm(i,k)-latvap*state1%q(i,k,ixcldliq)
  1516)      enddo
  1517)    enddo
  1518)    
  1519)    do k=1,pverp
  1520)      do i=1,ncol
  1521)         !  liquid water potential temperature flux
  1522)         wpthlp_output(i,k) = wpthlp(i,k)*rho(i,k)*cpair
  1523)         !  total water mixig ratio flux
  1524)         wprtp_output(i,k)  = wprtp(i,k)*rho(i,k)*latvap
  1525) 	!  turbulent kinetic energy
  1526) 	tke(i,k) = 0.5_r8*(up2(i,k)+vp2(i,k)+wp2(i,k))
  1527)      enddo
  1528)    enddo
  1529)    
  1530)    ! --------------------------------------------------------------------------------- ! 
  1531)    !  Diagnose some quantities that are computed in macrop_tend here.                  !
  1532)    !  These are inputs required for the microphysics calculation.                      !
  1533)    !                              						       !	
  1534)    !  FIRST PART COMPUTES THE STRATIFORM CLOUD FRACTION FROM CLUBB CLOUD FRACTION      !
  1535)    ! --------------------------------------------------------------------------------- ! 
  1536)  
  1537)    !  initialize variables 
  1538)    alst(:,:) = 0.0_r8
  1539)    qlst(:,:) = 0.0_r8 
  1540)  
  1541)    do k=1,pver
  1542)      do i=1,ncol
  1543)        alst(i,k) = cloud_frac(i,k)   
  1544)        qlst(i,k) = rcm(i,k)/max(0.01_r8,alst(i,k))  ! Incloud stratus condensate mixing ratio
  1545)      enddo
  1546)    enddo
  1547)  
  1548)    ! --------------------------------------------------------------------------------- !  
  1549)    !  THIS PART COMPUTES CONVECTIVE AND DEEP CONVECTIVE CLOUD FRACTION 		       !
  1550)    ! --------------------------------------------------------------------------------- ! 
  1551)  
  1552)    deepcu(:,pver) = 0.0_r8
  1553)    shalcu(:,pver) = 0.0_r8
  1554)  
  1555)    do k=1,pver-1
  1556)      do i=1,ncol
  1557)        !  diagnose the deep convective cloud fraction, as done in macrophysics based on the 
  1558)        !  deep convective mass flux, read in from pbuf.  Since shallow convection is never 
  1559)        !  called, the shallow convective mass flux will ALWAYS be zero, ensuring that this cloud
  1560)        !  fraction is purely from deep convection scheme.  
  1561)        deepcu(i,k) = max(0.0_r8,min(0.1_r8*log(1.0_r8+500.0_r8*(cmfmc(i,k+1)-cmfmc2(i,k+1))),0.6_r8))
  1562)        shalcu(i,k) = 0._r8
  1563)        
  1564)        if (deepcu(i,k) <= frac_limit .or. dp_icwmr(i,k) < ic_limit) then
  1565)          deepcu(i,k) = 0._r8
  1566)        endif
  1567)              
  1568)        !  using the deep convective cloud fraction, and CLUBB cloud fraction (variable 
  1569)        !  "cloud_frac"), compute the convective cloud fraction.  This follows the formulation
  1570)        !  found in macrophysics code.  Assumes that convective cloud is all nonstratiform cloud 
  1571)        !  from CLUBB plus the deep convective cloud fraction
  1572)        concld(i,k) = min(cloud_frac(i,k)-alst(i,k)+deepcu(i,k),0.80_r8)
  1573)      enddo
  1574)    enddo
  1575)    
  1576)    if (single_column) then
  1577)      if (trim(scm_clubb_iop_name) .eq. 'ATEX_48hr' .or. trim(scm_clubb_iop_name) .eq. 'BOMEX_5day' .or. &
  1578)          trim(scm_clubb_iop_name) .eq. 'DYCOMSrf01_4day' .or. &
  1579) 	 trim(scm_clubb_iop_name) .eq. 'DYCOMSrf02_06hr' .or. &
  1580) 	 trim(scm_clubb_iop_name) .eq. 'RICO_3day' .or. &
  1581) 	 trim(scm_clubb_iop_name) .eq. 'ARM_CC') then
  1582)        
  1583)            deepcu(:,:) = 0.0_r8
  1584)            concld(:,:) = 0.0_r8
  1585)        
  1586)      endif       
  1587)    endif
  1588)    
  1589)    ! --------------------------------------------------------------------------------- !  
  1590)    !  COMPUTE THE ICE CLOUD FRACTION PORTION					       !
  1591)    !  use the aist_vector function to compute the ice cloud fraction                   !
  1592)    ! --------------------------------------------------------------------------------- !  
  1593) 
  1594)     do k=1,pver
  1595)       call aist_vector(state1%q(:,k,ixq),state1%t(:,k),state1%pmid(:,k),state1%q(:,k,ixcldice), &
  1596)            cam_in%landfrac(:),cam_in%snowhland(:),aist(:,k),ncol)
  1597)     enddo
  1598)   
  1599)    ! --------------------------------------------------------------------------------- !  
  1600)    !  THIS PART COMPUTES THE LIQUID STRATUS FRACTION				       !
  1601)    !										       !
  1602)    !  For now leave the computation of ice stratus fraction from macrop_driver intact  !
  1603)    !  because CLUBB does nothing with ice.  Here I simply overwrite the liquid stratus ! 
  1604)    !  fraction that was coded in macrop_driver 					       !	
  1605)    ! --------------------------------------------------------------------------------- !  
  1606)  
  1607)    !  Recompute net stratus fraction using maximum over-lapping assumption, as done
  1608)    !  in macrophysics code, using alst computed above and aist read in from physics buffer
  1609)    
  1610)    cldthresh=1.e-18_r8
  1611) 
  1612)    do k=1,pver
  1613)      do i=1,ncol
  1614)        ast(i,k) = 0._r8  ! init AST
  1615) 
  1616)        if (newfice(i,k) .le. 0.5_r8 .and. state1%q(i,k,2) .gt. cldthresh) then
  1617)           ast(i,k) = alst(i,k)
  1618)        else if ((newfice(i,k) .gt. 0.5_r8 .and. state1%q(i,k,3) .gt. cldthresh) .or. &
  1619)                 (newfice(i,k) .le. 0.5_r8 .and. state1%q(i,k,2) .lt. cldthresh &
  1620)                                           .and. state1%q(i,k,3) .gt. cldthresh)) then
  1621)           ast(i,k) = aist(i,k)
  1622)        end if
  1623) 	 
  1624)        qist(i,k) = state1%q(i,k,ixcldice)/max(0.01_r8,aist(i,k)) 
  1625)      enddo
  1626)    enddo
  1627)    
  1628)    !  Probably need to add deepcu cloud fraction to the cloud fraction array, else would just 
  1629)    !  be outputting the shallow convective cloud fraction 
  1630) 
  1631)    do k=1,pver
  1632)      do i=1,ncol
  1633)        cloud_frac(i,k) = min(ast(i,k)+deepcu(i,k),1.0_r8)
  1634)      enddo
  1635)    enddo
  1636)    
  1637)    ! --------------------------------------------------------------------------------- !  
  1638)    !  DIAGNOSE THE PBL DEPTH					                       !
  1639)    !  this is needed for aerosol code                                                  !
  1640)    ! --------------------------------------------------------------------------------- ! 
  1641)     
  1642)    do i=1,ncol
  1643)      do k=1,pver
  1644)        th(i,k) = state1%t(i,k)*state1%exner(i,k)
  1645)        thv(i,k) = th(i,k)*(1.0_r8+zvir*state1%q(i,k,ixq))
  1646)      enddo
  1647)    enddo
  1648)  
  1649)    ! diagnose surface friction and obukhov length (inputs to diagnose PBL depth)
  1650)    do i=1,ncol
  1651)        call calc_ustar( state1%t(i,pver), state1%pmid(i,pver), cam_in%wsx(i), cam_in%wsy(i), &
  1652)                         rrho, ustar2(i) )
  1653)        call calc_obklen( th(i,pver), thv(i,pver), cam_in%lhf(i)/latvap, cam_in%shf(i), rrho, ustar2(i), &
  1654)                         kinheat(i), kinwat(i), kbfs(i), obklen(i) )  		
  1655)    enddo
  1656)    
  1657)    dummy2(:) = 0._r8
  1658)    dummy3(:) = 0._r8
  1659) 
  1660)    !  Compute PBL depth according to Holtslag-Boville Scheme
  1661)    call pblintd(ncol, thv, state1%zm, state1%u, state1%v, &
  1662)                 ustar2, obklen, kinheat, pblh, dummy2, &
  1663)         	state1%zi, cloud_frac(:,1:pver), 1._r8-cam_in%landfrac, dummy3)
  1664) 		 
  1665)    !  Output the PBL depth
  1666)    call outfld('PBLH', pblh, pcols, lchnk)
  1667)  
  1668)    ! Assign the first pver levels of cloud_frac back to cld
  1669)    cld(:,1:pver) = cloud_frac(:,1:pver)
  1670) 
  1671)    ! --------------------------------------------------------------------------------- !   
  1672)    !  END CLOUD FRACTION DIAGNOSIS, begin to store variables back into buffer          !
  1673)    ! --------------------------------------------------------------------------------- !  
  1674)  
  1675)    !  Output calls of variables goes here
  1676)    call outfld( 'RHO_CLUBB',        rho,                   pcols, lchnk )
  1677)    call outfld( 'WP2_CLUBB',        wp2,                   pcols, lchnk )
  1678)    call outfld( 'UP2_CLUBB',        up2,                   pcols, lchnk )
  1679)    call outfld( 'VP2_CLUBB',        vp2,                   pcols, lchnk )
  1680)    call outfld( 'WP3_CLUBB',        wp3,                   pcols, lchnk )
  1681)    call outfld( 'UPWP_CLUBB',       upwp,                  pcols, lchnk )
  1682)    call outfld( 'VPWP_CLUBB',       vpwp,                  pcols, lchnk )
  1683)    call outfld( 'WPTHLP_CLUBB',     wpthlp_output,         pcols, lchnk )
  1684)    call outfld( 'WPRTP_CLUBB',      wprtp_output,          pcols, lchnk )
  1685)    call outfld( 'RTP2_CLUBB',       rtp2*1000._r8,         pcols, lchnk )
  1686)    call outfld( 'THLP2_CLUBB',      thlp2,                 pcols, lchnk )
  1687)    call outfld( 'RTPTHLP_CLUBB',    rtpthlp*1000._r8,      pcols, lchnk )
  1688)    call outfld( 'RCM_CLUBB',        rcm*1000._r8,          pcols, lchnk )
  1689)    call outfld( 'WPRCP_CLUBB',      wprcp*latvap,          pcols, lchnk )
  1690)    call outfld( 'CLOUDFRAC_CLUBB',  alst,                  pcols, lchnk )
  1691)    call outfld( 'RCMINLAYER_CLUBB', rcm_in_layer*1000._r8, pcols, lchnk )
  1692)    call outfld( 'CLOUDCOVER_CLUBB', cloud_frac,            pcols, lchnk )
  1693)    call outfld( 'WPTHVP_CLUBB',     wpthvp*cpair,          pcols, lchnk )
  1694)    call outfld( 'ZT_CLUBB',         1._r8*zt_out,          pcols, lchnk )
  1695)    call outfld( 'ZM_CLUBB',         1._r8*zi_out,          pcols, lchnk )
  1696)    call outfld( 'UM_CLUBB',         um,                    pcols, lchnk )
  1697)    call outfld( 'VM_CLUBB',         vm,                    pcols, lchnk )
  1698)    call outfld( 'THETAL',           thetal_output,         pcols, lchnk )
  1699)    call outfld( 'QT',               qt_output,             pcols, lchnk )
  1700)    call outfld( 'SL',               sl_output,             pcols, lchnk )
  1701) 
  1702)    !  Output CLUBB history here
  1703)    if (l_stats) then 
  1704)       
  1705)    do i=1,zt%nn
  1706)    
  1707)      temp1 = trim(zt%f%var(i)%name)
  1708)      sub   = temp1
  1709)      if (len(temp1) .gt. 16) sub = temp1(1:16)
  1710)    
  1711)      call outfld(trim(sub), out_zt(:,:,i), pcols, lchnk )
  1712)    enddo
  1713)    
  1714)    do i=1,zm%nn
  1715)    
  1716)      temp1 = trim(zm%f%var(i)%name)
  1717)      sub   = temp1
  1718)      if (len(temp1) .gt. 16) sub = temp1(1:16)
  1719)    
  1720)      call outfld(trim(sub),out_zm(:,:,i), pcols, lchnk)
  1721)    enddo
  1722) 
  1723)    if (l_output_rad_files) then  
  1724)      do i=1,rad_zt%nn
  1725)        call outfld(trim(rad_zt%f%var(i)%name), out_radzt(:,:,i), pcols, lchnk)
  1726)      enddo
  1727)    
  1728)      do i=1,rad_zm%nn
  1729)        call outfld(trim(rad_zm%f%var(i)%name), out_radzm(:,:,i), pcols, lchnk)
  1730)      enddo
  1731)    endif
  1732)    
  1733)    do i=1,sfc%nn
  1734)      call outfld(trim(sfc%f%var(i)%name), out_sfc(:,:,i), pcols, lchnk)
  1735)    enddo
  1736)    
  1737)    endif
  1738)    
  1739)    return
  1740) #endif
  1741)   end subroutine clubb_tend_cam

  1747)     subroutine clubb_surface ( &
  1748)     			     state, ptend, ztodt, cam_in, ustar, obklen)
  1749)     
  1750) !-------------------------------------------------------------------------------
  1751) ! Description: Provide the obukov length and the surface friction velocity 
  1752) !              for the dry deposition code in routine tphysac.  Since University
  1753) !              of Washington Moist Turbulence (UWMT) scheme is not called when 
  1754) !              CLUBB is turned on the obukov length and ustar are never initialized
  1755) !              nor computed (sometimes never updated from NaN).  In addition, surface
  1756) !              fluxes are applied to the constituents.  
  1757) !   
  1758) ! Author: Peter Bogenschutz, August 2011
  1759) ! Origin: Based heavily on UWMT code (eddy_diff.F90)
  1760) ! References:
  1761) !   None
  1762) !-------------------------------------------------------------------------------
  1763) 
  1764)     use physics_types,		only: physics_state, physics_ptend, physics_ptend_init
  1765)     use physconst, 		only: gravit, zvir, latvap
  1766)     use ppgrid, 	        only: pver, pcols
  1767)     use constituents, 	        only: pcnst, cnst_get_ind
  1768)     use camsrfexch,             only: cam_in_t
  1769)     
  1770)     implicit none
  1771)     
  1772)     ! --------------- !
  1773)     ! Input Auguments !
  1774)     ! --------------- !
  1775) 
  1776)     type(physics_state), intent(in)    	:: state		! Physics state variables
  1777)     type(cam_in_t),      intent(in)     :: cam_in
  1778)     
  1779)     real(r8),  		 intent(in)     :: ztodt		! 2 delta-t        [ s ] 
  1780) 
  1781)     ! ---------------- !
  1782)     ! Output Auguments !
  1783)     ! ---------------- !
  1784)     
  1785)     type(physics_ptend), intent(out)    :: ptend		! Individual parameterization tendencies
  1786)     real(r8), 		 intent(out)  	:: obklen(pcols)    	! Obukhov length [ m ]
  1787)     real(r8), 		 intent(out)	:: ustar(pcols)		! Surface friction velocity [ m/s ]
  1788)     
  1789) #ifdef CLUBB_SGS 
  1790) 
  1791)     ! --------------- !
  1792)     ! Local Variables !
  1793)     ! --------------- !
  1794)     
  1795)     integer :: i   						! indicees					
  1796)     integer :: ncol						! # of atmospheric columns
  1797)     
  1798)     real(r8) :: th(pcols)                                       ! surface potential temperature
  1799)     real(r8) :: thv(pcols)                                      ! surface virtual potential temperature
  1800)     real(r8) :: kinheat						! kinematic surface heat flux
  1801)     real(r8) :: kinwat						! kinematic surface vapor flux
  1802)     real(r8) :: kbfs						! kinematic surface buoyancy flux
  1803)     real(r8) :: tmp1(pcols)
  1804)     real(r8) :: rztodt                                          ! 1./ztodt
  1805)     integer  :: m
  1806)     integer  :: ixq
  1807)     real(r8) :: rrho						! Inverse air density
  1808)     
  1809)     logical  :: lq(pcnst)
  1810) 
  1811) #endif
  1812)     obklen(pcols) = 0.0_r8
  1813)     ustar(pcols)  = 0.0_r8
  1814) #ifdef CLUBB_SGS
  1815) 
  1816)     ! ----------------------- !
  1817)     ! Main Computation Begins !
  1818)     ! ----------------------- !
  1819) 
  1820) 
  1821)     call cnst_get_ind('Q',ixq)
  1822)     
  1823)     lq(:) = .TRUE.
  1824)     call physics_ptend_init(ptend, state%psetcols, 'clubb', lq=lq)
  1825)     
  1826)     ncol = state%ncol
  1827)     
  1828)     ! Compute the surface friction velocity and obukov length
  1829)     
  1830)     do i = 1, ncol
  1831)       th(i) = state%t(i,pver)*state%exner(i,pver)         ! diagnose potential temperature
  1832)       thv(i) = th(i)*(1._r8+zvir*state%q(i,pver,ixq))  ! diagnose virtual potential temperature
  1833)     enddo
  1834)     
  1835)     do i = 1, ncol
  1836)       call calc_ustar( state%t(i,pver), state%pmid(i,pver), cam_in%wsx(i), cam_in%wsy(i), &
  1837)                        rrho, ustar(i) )
  1838)       call calc_obklen( th(i), thv(i), cam_in%lhf(i)/latvap, cam_in%shf(i), rrho, ustar(i), &
  1839)                        kinheat, kinwat, kbfs, obklen(i) )
  1840)     enddo
  1841) 
  1842)     rztodt = 1._r8/ztodt
  1843)     ptend%q(:ncol,:pver,:) = state%q(:ncol,:pver,:)
  1844)     tmp1(:ncol) = ztodt * gravit * state%rpdel(:ncol,pver)
  1845)     do m = 2, pcnst
  1846)       ptend%q(:ncol,pver,m) = ptend%q(:ncol,pver,m) + tmp1(:ncol) * cam_in%cflx(:ncol,m)
  1847)     enddo
  1848)     
  1849)     ptend%q(:ncol,:pver,:) = (ptend%q(:ncol,:pver,:) - state%q(:ncol,:pver,:)) * rztodt
  1850)     
  1851)     return
  1852) 
  1853) #endif
  1854) 
  1855)     end subroutine clubb_surface

  1864) !
  1865) ! Function diag_ustar:  returns value of ustar using the below
  1866) ! similarity functions and a specified buoyancy flux (bflx) given in
  1867) ! kinematic units
  1868) !
  1869) ! phi_m (zeta > 0) =  (1 + am * zeta)
  1870) ! phi_m (zeta < 0) =  (1 - bm * zeta)^(-1/4)
  1871) !
  1872) ! where zeta = z/lmo and lmo = (theta_rev/g*vonk) * (ustar^2/tstar)
  1873) !
  1874) ! Ref: Businger, 1973, Turbulent Transfer in the Atmospheric Surface
  1875) ! Layer, in Workshop on Micormeteorology, pages 67-100.
  1876) !
  1877) ! Code writen March, 1999 by Bjorn Stevens
  1878) !
  1879) 
  1880) real(r8) function diag_ustar( z, bflx, wnd, z0 ) 
  1881) 
  1882) use shr_const_mod, only : shr_const_karman, shr_const_pi, shr_const_g
  1883) 
  1884) implicit none
  1885) 
  1886) real(r8), parameter      :: am   =  4.8_r8   !   "          "         "
  1887) real(r8), parameter      :: bm   = 19.3_r8  !   "          "         "
  1888) 
  1889) real(r8), parameter      :: grav = shr_const_g
  1890) real(r8), parameter      :: vonk = shr_const_karman
  1891) real(r8), parameter      :: pi   = shr_const_pi
  1892) 
  1893) real(r8), intent (in)    :: z             ! height where u locates
  1894) real(r8), intent (in)    :: bflx          ! surface buoyancy flux (m^2/s^3)
  1895) real(r8), intent (in)    :: wnd           ! wind speed at z
  1896) real(r8), intent (in)    :: z0            ! momentum roughness height
  1897) 
  1898) 
  1899) integer :: iterate
  1900) real(r8)    :: lnz, klnz, c1, x, psi1, zeta, lmo, ustar

  1921)     endif
  1922) 
  1923)   end do
