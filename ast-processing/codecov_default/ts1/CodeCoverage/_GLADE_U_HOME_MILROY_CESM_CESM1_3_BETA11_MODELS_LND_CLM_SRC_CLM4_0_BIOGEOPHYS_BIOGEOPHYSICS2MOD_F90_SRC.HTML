    33) ! !IROUTINE: Biogeophysics2
    34) !
    35) ! !INTERFACE:
    36)   subroutine Biogeophysics2 (lbl, ubl, lbc, ubc, lbp, ubp, &
    37)              num_urbanl, filter_urbanl, num_nolakec, filter_nolakec, &
    38)              num_nolakep, filter_nolakep)
    39) !
    40) ! !DESCRIPTION:
    41) ! This is the main subroutine to execute the calculation of soil/snow and
    42) ! ground temperatures and update surface fluxes based on the new ground
    43) ! temperature
    44) !
    45) ! Calling sequence is:
    46) ! Biogeophysics2:             surface biogeophysics driver
    47) !    -> SoilTemperature:      soil/snow and ground temperatures
    48) !          -> SoilTermProp    thermal conductivities and heat capacities
    49) !          -> Tridiagonal     tridiagonal matrix solution
    50) !          -> PhaseChange     phase change of liquid/ice contents
    51) !
    52) ! (1) Snow and soil temperatures
    53) !     o The volumetric heat capacity is calculated as a linear combination
    54) !       in terms of the volumetric fraction of the constituent phases.
    55) !     o The thermal conductivity of soil is computed from
    56) !       the algorithm of Johansen (as reported by Farouki 1981), and the
    57) !       conductivity of snow is from the formulation used in
    58) !       SNTHERM (Jordan 1991).
    59) !     o Boundary conditions:
    60) !       F = Rnet - Hg - LEg (top),  F= 0 (base of the soil column).
    61) !     o Soil / snow temperature is predicted from heat conduction
    62) !       in 10 soil layers and up to 5 snow layers.
    63) !       The thermal conductivities at the interfaces between two
    64) !       neighboring layers (j, j+1) are derived from an assumption that
    65) !       the flux across the interface is equal to that from the node j
    66) !       to the interface and the flux from the interface to the node j+1.
    67) !       The equation is solved using the Crank-Nicholson method and
    68) !       results in a tridiagonal system equation.
    69) !
    70) ! (2) Phase change (see PhaseChange.F90)
    71) !
    72) ! !USES:
    73)     use clmtype
    74)     use clm_atmlnd        , only : clm_a2l
    75)     use clm_time_manager  , only : get_step_size
    76)     use clm_varcon        , only : hvap, cpair, grav, vkc, tfrz, sb, &
    77)                                    isturb, icol_roof, icol_sunwall, icol_shadewall, istsoil, &
    78)                                    istice_mec
    79)     use clm_varcon        , only : istcrop
    80)     use clm_varpar        , only : nlevsno, nlevgrnd, max_pft_per_col
    81)     use SoilTemperatureMod, only : SoilTemperature
    82)     use subgridAveMod     , only : p2c
    83) !
    84) ! !ARGUMENTS:
    85)     implicit none
    86)     integer, intent(in) :: lbp, ubp                    ! pft bounds
    87)     integer, intent(in) :: lbc, ubc                    ! column bounds
    88)     integer, intent(in) :: lbl, ubl                    ! landunit bounds
    89)     integer, intent(in) :: num_nolakec                 ! number of column non-lake points in column filter
    90)     integer, intent(in) :: filter_nolakec(ubc-lbc+1)   ! column filter for non-lake points
    91)     integer, intent(in) :: num_urbanl                  ! number of urban landunits in clump
    92)     integer, intent(in) :: filter_urbanl(ubl-lbl+1)    ! urban landunit filter
    93)     integer, intent(in) :: num_nolakep                 ! number of column non-lake points in pft filter
    94)     integer, intent(in) :: filter_nolakep(ubp-lbp+1)   ! pft filter for non-lake points
    95) !
    96) ! !CALLED FROM:
    97) ! subroutine clm_driver1
    98) !
    99) ! !REVISION HISTORY:
   100) ! 15 September 1999: Yongjiu Dai; Initial code
   101) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
   102) ! Migrated to clm2.0 by Keith Oleson and Mariana Vertenstein
   103) ! Migrated to clm2.1 new data structures by Peter Thornton and M. Vertenstein
   104) !
   105) ! !LOCAL VARIABLES:
   106) !
   107) ! local pointers to implicit in arguments
   108) !
   109)     integer , pointer :: ctype(:)                ! column type
   110)     integer , pointer :: clandunit(:)            ! column's landunit index
   111)     integer , pointer :: ltype(:)                ! landunit type
   112)     integer , pointer :: pcolumn(:)         ! pft's column index
   113)     integer , pointer :: plandunit(:)            ! pft's landunit index
   114)     integer , pointer :: pgridcell(:)       ! pft's gridcell index
   115)     real(r8), pointer :: pwtgcell(:)        ! pft's weight relative to corresponding column
   116)     integer , pointer :: npfts(:)           ! column's number of pfts 
   117)     integer , pointer :: pfti(:)            ! column's beginning pft index 
   118)     integer , pointer :: snl(:)             ! number of snow layers
   119)     logical , pointer :: do_capsnow(:)      ! true => do snow capping
   120)     real(r8), pointer :: forc_lwrad(:)      ! downward infrared (longwave) radiation (W/m**2)
   121)     real(r8), pointer :: emg(:)             ! ground emissivity
   122)     real(r8), pointer :: htvp(:)            ! latent heat of vapor of water (or sublimation) [j/kg]
   123)     real(r8), pointer :: t_grnd(:)          ! ground temperature (Kelvin)
   124)     integer , pointer :: frac_veg_nosno(:)  ! fraction of vegetation not covered by snow (0 OR 1 now) [-]
   125)     real(r8), pointer :: cgrnds(:)          ! deriv, of soil sensible heat flux wrt soil temp [w/m2/k]
   126)     real(r8), pointer :: cgrndl(:)          ! deriv of soil latent heat flux wrt soil temp [w/m**2/k]
   127)     real(r8), pointer :: sabg(:)            ! solar radiation absorbed by ground (W/m**2)
   128)     real(r8), pointer :: dlrad(:)           ! downward longwave radiation below the canopy [W/m2]
   129)     real(r8), pointer :: ulrad(:)           ! upward longwave radiation above the canopy [W/m2]
   130)     real(r8), pointer :: eflx_sh_veg(:)     ! sensible heat flux from leaves (W/m**2) [+ to atm]
   131)     real(r8), pointer :: qflx_evap_veg(:)   ! vegetation evaporation (mm H2O/s) (+ = to atm)
   132)     real(r8), pointer :: qflx_tran_veg(:)   ! vegetation transpiration (mm H2O/s) (+ = to atm)
   133)     real(r8), pointer :: qflx_evap_can(:)   ! evaporation from leaves and stems (mm H2O/s) (+ = to atm)
   134)     real(r8), pointer :: wtcol(:)           ! pft weight relative to column
   135)     real(r8), pointer :: tssbef(:,:)        ! soil/snow temperature before update
   136)     real(r8), pointer :: t_soisno(:,:)      ! soil temperature (Kelvin)
   137)     real(r8), pointer :: h2osoi_ice(:,:)    ! ice lens (kg/m2) (new)
   138)     real(r8), pointer :: h2osoi_liq(:,:)    ! liquid water (kg/m2) (new)
   139)     real(r8), pointer :: eflx_building_heat(:)   ! heat flux from urban building interior to walls, roof
   140)     real(r8), pointer :: eflx_traffic_pft(:)    ! traffic sensible heat flux (W/m**2)
   141)     real(r8), pointer :: eflx_wasteheat_pft(:)  ! sensible heat flux from urban heating/cooling sources of waste heat (W/m**2)
   142)     real(r8), pointer :: eflx_heat_from_ac_pft(:) ! sensible heat flux put back into canyon due to removal by AC (W/m**2)
   143)     real(r8), pointer :: canyon_hwr(:)      ! ratio of building height to street width (-)
   144)  
   145) ! local pointers to implicit inout arguments
   146) !
   147)     real(r8), pointer :: eflx_sh_grnd(:)    ! sensible heat flux from ground (W/m**2) [+ to atm]
   148)     real(r8), pointer :: qflx_evap_soi(:)   ! soil evaporation (mm H2O/s) (+ = to atm)
   149)     real(r8), pointer :: qflx_snwcp_liq(:)  ! excess rainfall due to snow capping (mm H2O /s)
   150)     real(r8), pointer :: qflx_snwcp_ice(:)  ! excess snowfall due to snow capping (mm H2O /s)
   151) ! 
   152) ! local pointers to implicit out arguments
   153) ! 
   154)     real(r8), pointer :: dt_grnd(:)         ! change in t_grnd, last iteration (Kelvin)
   155)     real(r8), pointer :: eflx_soil_grnd(:)  ! soil heat flux (W/m**2) [+ = into soil]
   156)     real(r8), pointer :: eflx_soil_grnd_u(:)! urban soil heat flux (W/m**2) [+ = into soil]
   157)     real(r8), pointer :: eflx_soil_grnd_r(:)! rural soil heat flux (W/m**2) [+ = into soil]
   158)     real(r8), pointer :: eflx_sh_tot(:)     ! total sensible heat flux (W/m**2) [+ to atm]
   159)     real(r8), pointer :: eflx_sh_tot_u(:)   ! urban total sensible heat flux (W/m**2) [+ to atm]
   160)     real(r8), pointer :: eflx_sh_tot_r(:)   ! rural total sensible heat flux (W/m**2) [+ to atm]
   161)     real(r8), pointer :: qflx_evap_tot(:)   ! qflx_evap_soi + qflx_evap_can + qflx_tran_veg
   162)     real(r8), pointer :: eflx_lh_tot(:)     ! total latent heat flux (W/m**2)  [+ to atm]
   163)     real(r8), pointer :: eflx_lh_tot_u(:)   ! urban total latent heat flux (W/m**2)  [+ to atm]
   164)     real(r8), pointer :: eflx_lh_tot_r(:)   ! rural total latent heat flux (W/m**2)  [+ to atm]
   165)     real(r8), pointer :: qflx_evap_grnd(:)  ! ground surface evaporation rate (mm H2O/s) [+]
   166)     real(r8), pointer :: qflx_sub_snow(:)   ! sublimation rate from snow pack (mm H2O /s) [+]
   167)     real(r8), pointer :: qflx_dew_snow(:)   ! surface dew added to snow pack (mm H2O /s) [+]
   168)     real(r8), pointer :: qflx_dew_grnd(:)   ! ground surface dew formation (mm H2O /s) [+]
   169)     real(r8), pointer :: eflx_lwrad_out(:)  ! emitted infrared (longwave) radiation (W/m**2)
   170)     real(r8), pointer :: eflx_lwrad_net(:)  ! net infrared (longwave) rad (W/m**2) [+ = to atm]
   171)     real(r8), pointer :: eflx_lwrad_net_u(:)  ! urban net infrared (longwave) rad (W/m**2) [+ = to atm]
   172)     real(r8), pointer :: eflx_lwrad_net_r(:)  ! rural net infrared (longwave) rad (W/m**2) [+ = to atm]
   173)     real(r8), pointer :: eflx_lh_vege(:)    ! veg evaporation heat flux (W/m**2) [+ to atm]
   174)     real(r8), pointer :: eflx_lh_vegt(:)    ! veg transpiration heat flux (W/m**2) [+ to atm]
   175)     real(r8), pointer :: eflx_lh_grnd(:)    ! ground evaporation heat flux (W/m**2) [+ to atm]
   176)     real(r8), pointer :: errsoi_pft(:)      ! pft-level soil/lake energy conservation error (W/m**2)
   177)     real(r8), pointer :: errsoi_col(:)      ! column-level soil/lake energy conservation error (W/m**2)
   178) !
   179) !
   180) ! !OTHER LOCAL VARIABLES:
   181) !EOP
   182) !
   183)     integer  :: p,c,g,j,pi,l         ! indices
   184)     integer  :: fc,fp                ! lake filtered column and pft indices
   185)     real(r8) :: dtime                ! land model time step (sec)
   186)     real(r8) :: egsmax(lbc:ubc)      ! max. evaporation which soil can provide at one time step
   187)     real(r8) :: egirat(lbc:ubc)      ! ratio of topsoil_evap_tot : egsmax
   188)     real(r8) :: tinc(lbc:ubc)        ! temperature difference of two time step
   189)     real(r8) :: xmf(lbc:ubc)         ! total latent heat of phase change of ground water
   190)     real(r8) :: sumwt(lbc:ubc)       ! temporary
   191)     real(r8) :: evaprat(lbp:ubp)     ! ratio of qflx_evap_soi/topsoil_evap_tot
   192)     real(r8) :: save_qflx_evap_soi   ! temporary storage for qflx_evap_soi
   193)     real(r8) :: topsoil_evap_tot(lbc:ubc)          ! column-level total evaporation from top soil layer
   194)     real(r8) :: fact(lbc:ubc, -nlevsno+1:nlevgrnd)  ! used in computing tridiagonal matrix
   195)     real(r8) :: eflx_lwrad_del(lbp:ubp)            ! update due to eflx_lwrad
   196) !-----------------------------------------------------------------------
   197) 
   198)     ! Assign local pointers to derived subtypes components (gridcell-level)
   199) 
   200)     forc_lwrad => clm_a2l%forc_lwrad
   201) 
   202)     ! Assign local pointers to derived subtypes components (landunit-level)
   203) 
   204)     ltype      => lun%itype
   205)     canyon_hwr     => lun%canyon_hwr
   206) 
   207)     ! Assign local pointers to derived subtypes components (column-level)
   208) 
   209)     ctype      => col%itype
   210)     clandunit  => col%landunit
   211)     npfts      => col%npfts
   212)     pfti       => col%pfti
   213)     snl        => cps%snl
   214)     do_capsnow => cps%do_capsnow
   215)     htvp       => cps%htvp
   216)     emg        => cps%emg
   217)     t_grnd     => ces%t_grnd
   218)     dt_grnd    => ces%dt_grnd
   219)     t_soisno   => ces%t_soisno
   220)     tssbef     => ces%tssbef
   221)     h2osoi_ice => cws%h2osoi_ice
   222)     h2osoi_liq => cws%h2osoi_liq
   223)     errsoi_col => cebal%errsoi
   224)     eflx_building_heat => cef%eflx_building_heat
   225) 
   226)     ! Assign local pointers to derived subtypes components (pft-level)
   227) 
   228)     pcolumn        => pft%column
   229)     plandunit      => pft%landunit
   230)     pgridcell      => pft%gridcell
   231)     pwtgcell       => pft%wtgcell
   232)     frac_veg_nosno => pps%frac_veg_nosno
   233)     sabg           => pef%sabg
   234)     dlrad          => pef%dlrad
   235)     ulrad          => pef%ulrad
   236)     eflx_sh_grnd   => pef%eflx_sh_grnd
   237)     eflx_sh_veg    => pef%eflx_sh_veg
   238)     qflx_evap_soi  => pwf%qflx_evap_soi
   239)     qflx_evap_veg  => pwf%qflx_evap_veg
   240)     qflx_tran_veg  => pwf%qflx_tran_veg
   241)     qflx_evap_can  => pwf%qflx_evap_can
   242)     qflx_snwcp_liq => pwf%qflx_snwcp_liq
   243)     qflx_snwcp_ice => pwf%qflx_snwcp_ice
   244)     qflx_evap_tot  => pwf%qflx_evap_tot
   245)     qflx_evap_grnd => pwf%qflx_evap_grnd
   246)     qflx_sub_snow  => pwf%qflx_sub_snow
   247)     qflx_dew_snow  => pwf%qflx_dew_snow
   248)     qflx_dew_grnd  => pwf%qflx_dew_grnd
   249)     eflx_soil_grnd => pef%eflx_soil_grnd
   250)     eflx_soil_grnd_u => pef%eflx_soil_grnd_u
   251)     eflx_soil_grnd_r => pef%eflx_soil_grnd_r
   252)     eflx_sh_tot    => pef%eflx_sh_tot
   253)     eflx_sh_tot_u  => pef%eflx_sh_tot_u
   254)     eflx_sh_tot_r  => pef%eflx_sh_tot_r
   255)     eflx_lh_tot    => pef%eflx_lh_tot
   256)     eflx_lh_tot_u    => pef%eflx_lh_tot_u
   257)     eflx_lh_tot_r    => pef%eflx_lh_tot_r
   258)     eflx_lwrad_out => pef%eflx_lwrad_out
   259)     eflx_lwrad_net => pef%eflx_lwrad_net
   260)     eflx_lwrad_net_u => pef%eflx_lwrad_net_u
   261)     eflx_lwrad_net_r => pef%eflx_lwrad_net_r
   262)     eflx_lh_vege   => pef%eflx_lh_vege
   263)     eflx_lh_vegt   => pef%eflx_lh_vegt
   264)     eflx_lh_grnd   => pef%eflx_lh_grnd
   265)     cgrnds         => pef%cgrnds
   266)     cgrndl         => pef%cgrndl
   267)     eflx_sh_grnd   => pef%eflx_sh_grnd
   268)     qflx_evap_soi  => pwf%qflx_evap_soi
   269)     errsoi_pft     => pebal%errsoi
   270)     wtcol          => pft%wtcol
   271)     eflx_wasteheat_pft => pef%eflx_wasteheat_pft
   272)     eflx_heat_from_ac_pft => pef%eflx_heat_from_ac_pft
   273)     eflx_traffic_pft => pef%eflx_traffic_pft
   274) 
   275)     ! Get step size
   276) 
   277)     dtime = get_step_size()
   278) 
   279)     ! Determine soil temperatures including surface soil temperature
   280) 
   281)     call SoilTemperature(lbl, ubl, lbc, ubc, num_urbanl, filter_urbanl, &
   282)                          num_nolakec, filter_nolakec, xmf , fact)
   283) 
   284)     do fc = 1,num_nolakec
   285)        c = filter_nolakec(fc)
   286)        j = snl(c)+1
   287) 
   288)        ! Calculate difference in soil temperature from last time step, for
   289)        ! flux corrections
   290) 
   291)        tinc(c) = t_soisno(c,j) - tssbef(c,j)
   292) 
   293)        ! Determine ratio of topsoil_evap_tot
   294) 
   295)        egsmax(c) = (h2osoi_ice(c,j)+h2osoi_liq(c,j)) / dtime
   296) 
   297)        ! added to trap very small negative soil water,ice
   298) 
   299)        if (egsmax(c) < 0._r8) then
   300)           egsmax(c) = 0._r8
   301)        end if
   302)     end do
   303) 
   304)     ! A preliminary pft loop to determine if corrections are required for
   305)     ! excess evaporation from the top soil layer... Includes new logic
   306)     ! to distribute the corrections between pfts on the basis of their
   307)     ! evaporative demands.
   308)     ! egirat holds the ratio of demand to availability if demand is
   309)     ! greater than availability, or 1.0 otherwise.
   310)     ! Correct fluxes to present soil temperature
   311) 
   312)     do fp = 1,num_nolakep
   313)        p = filter_nolakep(fp)
   314)        c = pcolumn(p)
   315)        eflx_sh_grnd(p) = eflx_sh_grnd(p) + tinc(c)*cgrnds(p)
   316)        qflx_evap_soi(p) = qflx_evap_soi(p) + tinc(c)*cgrndl(p)
   317)     end do
   318) 
   319)     ! Set the column-average qflx_evap_soi as the weighted average over all pfts
   320)     ! but only count the pfts that are evaporating
   321) 
   322)     do fc = 1,num_nolakec
   323)        c = filter_nolakec(fc)
   324)        topsoil_evap_tot(c) = 0._r8
   325)        sumwt(c) = 0._r8
   326)     end do
   327) 
   328)     do pi = 1,max_pft_per_col
   329)        do fc = 1,num_nolakec
   330)           c = filter_nolakec(fc)
   331)           l = clandunit(c)
   332)           if ( pi <= npfts(c) ) then
   333)              p = pfti(c) + pi - 1
   334)              ! Note: some glacier_mec pfts may have zero weight
   335)              if (pwtgcell(p)>0._r8 .or. ltype(l)==istice_mec) then
   336)                 topsoil_evap_tot(c) = topsoil_evap_tot(c) + qflx_evap_soi(p) * wtcol(p)
   337)              end if
   338)           end if
   339)        end do
   340)     end do
   341) 
   342)     ! Calculate ratio for rescaling pft-level fluxes to meet availability
   343) 
   344)     do fc = 1,num_nolakec
   345)        c = filter_nolakec(fc)
   346)        if (topsoil_evap_tot(c) > egsmax(c)) then
   347)           egirat(c) = (egsmax(c)/topsoil_evap_tot(c))
   348)        else
   349)           egirat(c) = 1.0_r8
   350)        end if
   351)     end do
   352) 
   353)     do fp = 1,num_nolakep
   354)        p = filter_nolakep(fp)
   355)        c = pcolumn(p)
   356)        l = plandunit(p)
   357)        g = pgridcell(p)
   358)        j = snl(c)+1
   359) 
   360)        ! Correct soil fluxes for possible evaporation in excess of top layer water
   361)        ! excess energy is added to the sensible heat flux from soil
   362) 
   363)        if (egirat(c) < 1.0_r8) then
   364)           save_qflx_evap_soi = qflx_evap_soi(p)
   365)           qflx_evap_soi(p) = qflx_evap_soi(p) * egirat(c)
   366)           eflx_sh_grnd(p) = eflx_sh_grnd(p) + (save_qflx_evap_soi - qflx_evap_soi(p))*htvp(c)
   367)        end if
   368) 
   369)        ! Ground heat flux
   370) 
   371)        if (ltype(l) /= isturb) then
   372)           eflx_soil_grnd(p) = sabg(p) + dlrad(p) &
   373)                               + (1-frac_veg_nosno(p))*emg(c)*forc_lwrad(g) &
   374)                               - emg(c)*sb*tssbef(c,j)**3*(tssbef(c,j) + 4._r8*tinc(c)) &
   375)                               - (eflx_sh_grnd(p) + qflx_evap_soi(p)*htvp(c))
   376)           if (ltype(l) == istsoil .or. ltype(l) == istcrop) then
   377)             eflx_soil_grnd_r(p) = eflx_soil_grnd(p)
   378)           end if
   379)        else
   380)           ! For all urban columns we use the net longwave radiation (eflx_lwrad_net) since
   381)           ! the term (emg*sb*tssbef(snl+1)**4) is not the upward longwave flux because of 
   382)           ! interactions between urban columns.
   383) 
   384)           eflx_lwrad_del(p) = 4._r8*emg(c)*sb*tssbef(c,j)**3*tinc(c)
   385)           ! Include transpiration term because needed for pervious road
   386)           ! and wasteheat and traffic flux
   387)           eflx_soil_grnd(p) = sabg(p) + dlrad(p) &
   388)                               - eflx_lwrad_net(p) - eflx_lwrad_del(p) &
