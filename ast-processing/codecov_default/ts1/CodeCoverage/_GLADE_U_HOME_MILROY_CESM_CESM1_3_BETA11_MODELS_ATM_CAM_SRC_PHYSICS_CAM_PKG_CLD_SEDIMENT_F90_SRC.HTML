    30)   real (r8), parameter :: vland  = 1.5_r8            ! liquid fall velocity over land  (cm/s)
    31)   real (r8), parameter :: vocean = 2.8_r8            ! liquid fall velocity over ocean (cm/s)
    32)   real (r8), parameter :: mxsedfac   = 0.99_r8       ! maximum sedimentation flux factor

    34)   logical,   parameter :: stokes = .true.         ! use Stokes velocity instead of McFarquhar and Heymsfield

    40)   real (r8), parameter :: eta =  1.7e-5_r8           ! viscosity of air (kg m / s)
    41)   real (r8), parameter :: r40 =  40._r8              !  40 micron radius
    42)   real (r8), parameter :: r400= 400._r8              ! 400 micron radius
    43)   real (r8), parameter :: v400= 1.00_r8              ! fall velocity of 400 micron sphere (m/s)
    44)   real (r8)            :: v40 ! = (2._r8/9._r8) * rhoh2o * gravit/eta * r40**2 * 1.e-12_r8  
    45)                                                      ! Stokes fall velocity of 40 micron sphere (m/s)
    46)   real (r8)            :: vslope !  = (v400 - v40)/(r400 -r40) ! linear slope for large particles m/s/micron
    47) 
    48)   ! namelist variables
    49)   real(r8) :: cldsed_ice_stokes_fac = huge(1._r8)    ! factor applied to the ice fall velocity computed from 
    50)                                                      ! stokes terminal velocity

    97)   subroutine cld_sediment_vel (ncol,                               &
    98)        icefrac , landfrac, ocnfrac , pmid    , pdel    , t       , &
    99)        cloud   , cldliq  , cldice  , pvliq   , pvice   , landm, snowh)
   100) 
   101) !----------------------------------------------------------------------
   102) 
   103) ! Compute gravitational sedimentation velocities for cloud liquid water
   104) ! and ice, based on Lawrence and Crutzen (1998).
   105) 
   106) ! LIQUID
   107) 
   108) ! The fall velocities assume that droplets have a gamma distribution
   109) ! with effective radii for land and ocean as assessed by Han et al.;
   110) ! see Lawrence and Crutzen (1998) for a derivation.
   111) 
   112) ! ICE
   113) 
   114) ! The fall velocities are based on data from McFarquhar and Heymsfield
   115) ! or on Stokes terminal velocity for spheres and the effective radius.
   116) 
   117) ! NEED TO BE CAREFUL - VELOCITIES SHOULD BE AT THE *LOWER* INTERFACE
   118) ! (THAT IS, FOR K+1), FLUXES ARE ALSO AT THE LOWER INTERFACE (K+1), 
   119) ! BUT MIXING RATIOS ARE AT THE MIDPOINTS (K)...
   120) 
   121) ! NOTE THAT PVEL IS ON PVERP (INTERFACES), WHEREAS VFALL IS FOR THE CELL
   122) ! AVERAGES (I.E., MIDPOINTS); ASSUME THE FALL VELOCITY APPLICABLE TO THE 
   123) ! LOWER INTERFACE (K+1) IS THE SAME AS THAT APPLICABLE FOR THE CELL (V(K))
   124) 
   125) !-----------------------------------------------------------------------
   126) !     MATCH-MPIC version 2.0, Author: mgl, March 1998
   127) ! adapted by P. J. Rasch
   128) !            B. A. Boville, September 19, 2002
   129) !            P. J. Rasch    May 22, 2003 (added stokes flow calc for liquid
   130) !                                         drops based on effect radii)
   131) !-----------------------------------------------------------------------
   132) 
   133) 
   134) ! Arguments
   135)     integer, intent(in) :: ncol                     ! number of colums to process
   136) 
   137)     real(r8), intent(in)  :: icefrac (pcols)        ! sea ice fraction (fraction)
   138)     real(r8), intent(in)  :: landfrac(pcols)        ! land fraction (fraction)
   139)     real(r8), intent(in)  :: ocnfrac (pcols)        ! ocean fraction (fraction)
   140)     real(r8), intent(in)  :: pmid  (pcols,pver)     ! pressure of midpoint levels (Pa)
   141)     real(r8), intent(in)  :: pdel  (pcols,pver)     ! pressure diff across layer (Pa)
   142)     real(r8), intent(in)  :: cloud (pcols,pver)     ! cloud fraction (fraction)
   143)     real(r8), intent(in)  :: t     (pcols,pver)     ! temperature (K)
   144)     real(r8), intent(in)  :: cldliq(pcols,pver)     ! cloud water, liquid (kg/kg)
   145)     real(r8), intent(in)  :: cldice(pcols,pver)     ! cloud water, ice    (kg/kg)
   146)     real(r8), intent(in) :: snowh(pcols)         ! Snow depth over land, water equivalent (m)
   147) 
   148)     real(r8), intent(out) :: pvliq (pcols,pverp)    ! vertical velocity of cloud liquid drops (Pa/s)
   149)     real(r8), intent(out) :: pvice (pcols,pverp)    ! vertical velocity of cloud ice particles (Pa/s)
   150)     real(r8), intent(in) :: landm(pcols)            ! land fraction ramped over water
   151) ! -> note that pvel is at the interfaces (loss from cell is based on pvel(k+1))
   152) 
   153) ! Local variables
   154)     real (r8) :: rho(pcols,pver)                    ! air density in kg/m3
   155)     real (r8) :: vfall                              ! settling velocity of cloud particles (m/s)
   156)     real (r8) :: icice                              ! in cloud ice water content (kg/kg)
   157)     real (r8) :: iciwc                              ! in cloud ice water content in g/m3
   158)     real (r8) :: icefac
   159)     real (r8) :: logiwc
   160) 
   161)     real (r8) :: rei(pcols,pver)                    ! effective radius of ice particles (microns)
   162)     real (r8) :: rel(pcols,pver)                    ! effective radius of liq particles (microns)
   163)     real(r8)  pvliq2 (pcols,pverp)    ! vertical velocity of cloud liquid drops (Pa/s)
   164) 
   165)     integer i,k
   166) 
   167)     real (r8) :: lbound, ac, bc, cc
   168) 
   169) !-----------------------------------------------------------------------
   170) !------- initialize linear ramp variables for fall velocity ------------
   171) !-----------------------------------------------------------------------
   172) 
   173)   v40 = (2._r8/9._r8) * rhoh2o * gravit/eta * r40**2 * 1.e-12_r8  
   174)   vslope = (v400 - v40)/(r400 -r40)
   175) 
   176) !-----------------------------------------------------------------------
   177) !--------------------- liquid fall velocity ----------------------------
   178) !-----------------------------------------------------------------------
   179) 
   180)     ! compute air density
   181)     rho(:ncol,:) = pmid(:ncol,:) / (rair * t(:ncol,:))
   182) 
   183)     pvliq(:ncol,:) = 0._r8
   184) 
   185)     ! get effective radius of liquid drop
   186)     call reltab(ncol, t, landfrac, landm, icefrac, rel, snowh)
   187) 
   188)     do k = 1,pver
   189)        do i = 1,ncol
   190)           if (cloud(i,k) > 0._r8 .and. cldliq(i,k) > 0._r8) then
   191) 
   192) #ifdef OLDLIQSED
   193) ! oldway
   194)              ! merge the liquid fall velocities for land and ocean (cm/s)
   195)              ! SHOULD ALSO ACCOUNT FOR ICEFRAC
   196)              vfall = vland*landfrac(i) + vocean*(1._r8-landfrac(i))
   197) !!$          vfall = vland*landfrac(i) + vocean*ocnfrac(i) + vseaice*icefrac(i)
   198) 
   199)              ! convert the fall speed to pressure units, but do not apply the traditional
   200)              ! negative convention for pvel.
   201)              pvliq(i,k+1) = vfall     &
   202)                   * 0.01_r8                 & ! cm to meters
   203)                   * rho(i,k)*gravit        ! meters/sec to pascals/sec
   204) #else
   205) 
   206) ! newway
   207)              if (rel(i,k) < 40._r8 ) then
   208)                 vfall = 2._r8/9._r8 * rhoh2o * gravit * rel(i,k)**2 / eta  * 1.e-12_r8  ! micons^2 -> m^2
   209)              else
   210)                 vfall = v40 + vslope * (rel(i,k)-r40)      ! linear above 40 microns
   211)              end if
   212)              ! convert the fall speed to pressure units
   213)              ! but do not apply the traditional
   214)              ! negative convention for pvel.
   215) !             pvliq2(i,k+1) = vfall * rho(i,k)*gravit        ! meters/sec to pascals/sec
   216)              pvliq(i,k+1) = vfall * rho(i,k)*gravit        ! meters/sec to pascals/sec
   217) #endif
   218)           end if
   219)        end do
   220)     end do
   221) 
   222)     !-----------------------------------------------------------------------
   223)     !--------------------- ice fall velocity -------------------------------
   224)     !-----------------------------------------------------------------------
   225) 
   226)     pvice(:ncol,:) = 0._r8
   227) 
   228)     if (stokes) then
   229) 
   230)        !-----------------------------------------------------------------------
   231)        !--------------------- stokes terminal velocity < 40 microns -----------
   232)        !-----------------------------------------------------------------------
   233) 
   234)        ! get effective radius
   235)        call reitab(ncol, t, rei)
   236) 
   237)        do k = 1,pver
   238)           do i = 1,ncol
   239)              if (cloud(i,k) > 0._r8 .and. cldice(i,k) > 0._r8) then
   240)                 if (rei(i,k) < 40._r8 ) then
   241)                    vfall = 2._r8/9._r8 * rhoh2o * gravit * rei(i,k)**2 / eta  * 1.e-12_r8  ! micons^2 -> m^2
   242)                    vfall = vfall * cldsed_ice_stokes_fac
   243)                 else
   244)                    vfall = v40 + vslope * (rei(i,k)-r40)      ! linear above 40 microns
   245)                 end if
   246) 
   247)                 ! convert the fall speed to pressure units, but do not apply the traditional
   248)                 ! negative convention for pvel.
   249)                 pvice(i,k+1) = vfall * rho(i,k)*gravit        ! meters/sec to pascals/sec
   250)              end if
   251)           end do
   252)        end do
   253) 
   254)     else
   255) 
   256)        !-----------------------------------------------------------------------
   257)        !--------------------- McFarquhar and Heymsfield > icritc --------------
   258)        !-----------------------------------------------------------------------
   259) 
   260)        ! lower bound for iciwc
   261) 
   262)        cc = 128.64_r8 
   263)        bc = 53.242_r8
   264)        ac = 5.4795_r8
   265)        lbound = (-bc + sqrt(bc*bc-4*ac*cc))/(2*ac)
   266)        lbound = 10._r8**lbound
   267) 
   268)        do k = 1,pver
   269)           do i = 1,ncol
   270)              if (cloud(i,k) > 0._r8 .and. cldice(i,k) > 0._r8) then
   271) 
   272)                 ! compute the in-cloud ice concentration (kg/kg)
   273)                 icice = cldice(i,k) / cloud(i,k)
   274) 
   275)                 ! compute the ice water content in g/m3
   276)                 iciwc = icice * rho(i,k) * 1.e3_r8
   277) 
   278)                 ! set the fall velocity (cm/s) to depend on the ice water content in g/m3,
   279)                 if (iciwc > lbound) then ! need this because of log10
   280)                    logiwc = log10(iciwc)
   281)                    !          Median - 
   282)                    vfall = 128.64_r8 + 53.242_r8*logiwc + 5.4795_r8*logiwc**2
   283)                    !          Average - 
   284)                 !!$             vfall = 122.63 + 44.111*logiwc + 4.2144*logiwc**2
   285)                 else
   286)                    vfall = 0._r8
   287)                 end if
   288) 
   289)                 ! set ice velocity to 1 cm/s if ice mixing ratio < icritc, ramp to value
   290)                 ! calculated above at 2*icritc
   291)                 if (icice <= icritc) then
   292)                    vfall = vice_small
   293)                 else if(icice < 2*icritc) then
   294)                    icefac = (icice-icritc)/icritc
   295)                    vfall = vice_small * (1._r8-icefac) + vfall * icefac
   296)                 end if
   297) 
   298)                 ! bound the terminal velocity of ice particles at high concentration
   299)                 vfall = min(100.0_r8, vfall)
   300) 
   301)                 ! convert the fall speed to pressure units, but do not apply the traditional
   302)                 ! negative convention for pvel.
   303)                 pvice(i,k+1) = vfall     &
   304)                    * 0.01_r8                 & ! cm to meters
   305)                    * rho(i,k)*gravit        ! meters/sec to pascals/sec
   306)              end if
   307)           end do
   308)        end do
   309) 
   310)     end if
   311) 
   312)  end subroutine cld_sediment_vel

   316)   subroutine cld_sediment_tend (ncol, dtime  ,               &
   317)        pint   , pmid   , pdel   , t      ,                   &
   318)        cloud  , cldliq , cldice , pvliq  , pvice  ,          &
   319)        liqtend, icetend, wvtend , htend  , sfliq  , sfice   )
   320) 
   321) !----------------------------------------------------------------------
   322) !     Apply Cloud Particle Gravitational Sedimentation to Condensate
   323) !----------------------------------------------------------------------
   324) 
   325) 
   326) ! Arguments
   327)     integer,  intent(in)  :: ncol                      ! number of colums to process
   328) 
   329)     real(r8), intent(in)  :: dtime                     ! time step
   330)     real(r8), intent(in)  :: pint  (pcols,pverp)       ! interfaces pressure (Pa)
   331)     real(r8), intent(in)  :: pmid  (pcols,pver)        ! midpoint pressures (Pa)
   332)     real(r8), intent(in)  :: pdel  (pcols,pver)        ! pressure diff across layer (Pa)
   333)     real(r8), intent(in)  :: cloud (pcols,pver)        ! cloud fraction (fraction)
   334)     real(r8), intent(in)  :: t     (pcols,pver)        ! temperature (K)
   335)     real(r8), intent(in)  :: cldliq(pcols,pver)        ! cloud liquid water (kg/kg)
   336)     real(r8), intent(in)  :: cldice(pcols,pver)        ! cloud ice water    (kg/kg)
   337)     real(r8), intent(in)  :: pvliq (pcols,pverp)       ! vertical velocity of liquid drops  (Pa/s)
   338)     real(r8), intent(in)  :: pvice (pcols,pverp)       ! vertical velocity of ice particles (Pa/s)
   339) ! -> note that pvel is at the interfaces (loss from cell is based on pvel(k+1))
   340) 
   341)     real(r8), intent(out) :: liqtend(pcols,pver)       ! liquid condensate tend
   342)     real(r8), intent(out) :: icetend(pcols,pver)       ! ice condensate tend
   343)     real(r8), intent(out) :: wvtend (pcols,pver)       ! water vapor tend
   344)     real(r8), intent(out) :: htend  (pcols,pver)       ! heating rate
   345)     real(r8), intent(out) :: sfliq  (pcols)            ! surface flux of liquid (rain, kg/m/s)
   346)     real(r8), intent(out) :: sfice  (pcols)            ! surface flux of ice    (snow, kg/m/s)
   347) 
   348) ! Local variables
   349)     real(r8) :: fxliq(pcols,pverp)                     ! fluxes at the interfaces, liquid (positive = down)
   350)     real(r8) :: fxice(pcols,pverp)                     ! fluxes at the interfaces, ice    (positive = down)
   351)     real(r8) :: cldab(pcols)                           ! cloud in layer above
   352)     real(r8) :: evapliq                                ! evaporation of cloud liquid into environment
   353)     real(r8) :: evapice                                ! evaporation of cloud ice into environment
   354)     real(r8) :: cldovrl                                ! cloud overlap factor
   355) 
   356)     integer :: i,k
   357) !----------------------------------------------------------------------
   358) 
   359) ! initialize variables
   360)     fxliq  (:ncol,:) = 0._r8 ! flux at interfaces (liquid)
   361)     fxice  (:ncol,:) = 0._r8 ! flux at interfaces (ice)
   362)     liqtend(:ncol,:) = 0._r8 ! condensate tend (liquid)
   363)     icetend(:ncol,:) = 0._r8 ! condensate tend (ice)
   364)     wvtend(:ncol,:)  = 0._r8 ! environmental moistening
   365)     htend(:ncol,:)   = 0._r8 ! evaporative cooling
   366)     sfliq(:ncol)     = 0._r8 ! condensate sedimentation flux out bot of column (liquid)
   367)     sfice(:ncol)     = 0._r8 ! condensate sedimentation flux out bot of column (ice)
   368) 
   369) ! fluxes at interior points
   370)     call getflx(ncol, pint, cldliq, pvliq, dtime, fxliq)
   371)     call getflx(ncol, pint, cldice, pvice, dtime, fxice)
   372) 
   373) ! calculate fluxes at boundaries
   374)     do i = 1,ncol
   375)        fxliq(i,1) = 0._r8
   376)        fxice(i,1) = 0._r8
   377) ! surface flux by upstream scheme
   378)        fxliq(i,pverp) = cldliq(i,pver) * pvliq(i,pverp) * dtime
   379)        fxice(i,pverp) = cldice(i,pver) * pvice(i,pverp) * dtime
   380)     end do
   381) 
   382) ! filter out any negative fluxes from the getflx routine
   383) ! (typical fluxes are of order > 1e-3 when clouds are present)
   384)     do k = 2,pver
   385)        fxliq(:ncol,k) = max(0._r8, fxliq(:ncol,k))
   386)        fxice(:ncol,k) = max(0._r8, fxice(:ncol,k))
   387)     end do
   388) 
   389) ! Limit the flux out of the bottom of each cell to the water content in each phase.
   390) ! Apply mxsedfac to prevent generating very small negative cloud water/ice
   391) ! NOTE, REMOVED CLOUD FACTOR FROM AVAILABLE WATER. ALL CLOUD WATER IS IN CLOUDS.
   392) ! ***Should we include the flux in the top, to allow for thin surface layers?
   393) ! ***Requires simple treatment of cloud overlap, already included below.
   394)     do k = 1,pver
   395)        do i = 1,ncol
   396)           fxliq(i,k+1) = min( fxliq(i,k+1), mxsedfac * cldliq(i,k) * pdel(i,k) )
   397)           fxice(i,k+1) = min( fxice(i,k+1), mxsedfac * cldice(i,k) * pdel(i,k) )
   398) !!$        fxliq(i,k+1) = min( fxliq(i,k+1), cldliq(i,k) * pdel(i,k) + fxliq(i,k))
   399) !!$        fxice(i,k+1) = min( fxice(i,k+1), cldice(i,k) * pdel(i,k) + fxice(i,k))
   400) !!$        fxliq(i,k+1) = min( fxliq(i,k+1), cloud(i,k) * cldliq(i,k) * pdel(i,k) )
   401) !!$        fxice(i,k+1) = min( fxice(i,k+1), cloud(i,k) * cldice(i,k) * pdel(i,k) )
   402)        end do
   403)     end do
   404) 
   405) ! Now calculate the tendencies assuming that condensate evaporates when
   406) ! it falls into environment, and does not when it falls into cloud.
   407) ! All flux out of the layer comes from the cloudy part.
   408) ! Assume maximum overlap for stratiform clouds
   409) !  if cloud above < cloud,  all water falls into cloud below
   410) !  if cloud above > cloud,  water split between cloud  and environment
   411)     do k = 1,pver
   412)        cldab(:ncol) = 0._r8
   413)        do i = 1,ncol
   414) ! cloud overlap cloud factor
   415)           cldovrl  = min( cloud(i,k) / (cldab(i)+.0001_r8), 1._r8 )
   416)           cldab(i) = cloud(i,k)
   417) ! evaporation into environment cause moistening and cooling
   418)           evapliq = fxliq(i,k) * (1._r8-cldovrl) / (dtime * pdel(i,k))  ! into env (kg/kg/s)
   419)           evapice = fxice(i,k) * (1._r8-cldovrl) / (dtime * pdel(i,k))  ! into env (kg/kg/s)
   420)           wvtend(i,k) = evapliq + evapice                          ! evaporation into environment (kg/kg/s)
   421)           htend (i,k) = -latvap*evapliq -(latvap+latice)*evapice   ! evaporation (W/kg)
   422) ! net flux into cloud changes cloud liquid/ice (all flux is out of cloud)
   423)           liqtend(i,k)  = (fxliq(i,k)*cldovrl - fxliq(i,k+1)) / (dtime * pdel(i,k))
   424)           icetend(i,k)  = (fxice(i,k)*cldovrl - fxice(i,k+1)) / (dtime * pdel(i,k))
   425)        end do
   426)     end do
   427) 
   428) ! convert flux out the bottom to mass units Pa -> kg/m2/s
   429)     sfliq(:ncol) = fxliq(:ncol,pverp) / (dtime*gravit)
   430)     sfice(:ncol) = fxice(:ncol,pverp) / (dtime*gravit)
   431) 
   432)     return
   433)   end subroutine cld_sediment_tend

   436)   subroutine getflx(ncol, xw, phi, vel, deltat, flux)
   437) 
   438) !.....xw1.......xw2.......xw3.......xw4.......xw5.......xw6
   439) !....psiw1.....psiw2.....psiw3.....psiw4.....psiw5.....psiw6
   440) !....velw1.....velw2.....velw3.....velw4.....velw5.....velw6
   441) !.........phi1......phi2.......phi3.....phi4.......phi5.......
   442) 
   443) 
   444) 
   445)     integer, intent(in) :: ncol                      ! number of colums to process
   446) 
   447)     integer i
   448)     integer k
   449)

   619)   subroutine cfdotmc_pro (ncol, x, f, fdot)
   620) 
   621) !     prototype version; eventually replace with final SPITFIRE scheme
   622) 
   623) !     calculate the derivative for the interpolating polynomial
   624) !     multi column version
   625) 
   626) 
   627) 
   628) ! input
   629)     integer ncol                      ! number of colums to process
   630) 
   631)     real (r8) x(pcols, pverp)
   632)     real (r8) f(pcols, pverp)
   633) ! output
   634)     real (r8) fdot(pcols, pverp)          ! derivative at nodes
   635) 
   636) ! assumed variable distribution
   637) !     x1.......x2.......x3.......x4.......x5.......x6     1,pverp points
   638) !     f1.......f2.......f3.......f4.......f5.......f6     1,pverp points
   639) !     ...sh1.......sh2......sh3......sh4......sh5....     1,pver points
   640) !     .........d2.......d3.......d4.......d5.........     2,pver points
   641) !     .........s2.......s3.......s4.......s5.........     2,pver points
   642) !     .............dh2......dh3......dh4.............     2,pver-1 points
   643) !     .............eh2......eh3......eh4.............     2,pver-1 points
   644) !     ..................e3.......e4..................     3,pver-1 points
   645) !     .................ppl3......ppl4................     3,pver-1 points
   646) !     .................ppr3......ppr4................     3,pver-1 points
   647) !     .................t3........t4..................     3,pver-1 points
   648) !     ................fdot3.....fdot4................     3,pver-1 points
   649) 
   650) 
   651) ! work variables
   652) 
   653) 
   654)     integer i
   655)     integer k
   656) 
   657)     real (r8) a                    ! work var
   658)     real (r8) b                    ! work var
   659)     real (r8) c                    ! work var
   660)     real (r8) s(pcols,pverp)             ! first divided differences at nodes
   661)     real (r8) sh(pcols,pverp)            ! first divided differences between nodes
   662)     real (r8) d(pcols,pverp)             ! second divided differences at nodes
   663)     real (r8) dh(pcols,pverp)            ! second divided differences between nodes
   664)     real (r8) e(pcols,pverp)             ! third divided differences at nodes
   665)     real (r8) eh(pcols,pverp)            ! third divided differences between nodes
   666)     real (r8) pp                   ! p prime
   667)     real (r8) ppl(pcols,pverp)           ! p prime on left
   668)     real (r8) ppr(pcols,pverp)           ! p prime on right
   669)     real (r8) qpl
   670)     real (r8) qpr
   671)     real (r8) ttt
   672)     real (r8) t
   673)     real (r8) tmin
   674)     real (r8) tmax
   675)     real (r8) delxh(pcols,pverp)
   676) 
   677) 
   678) !     the minmod function 
   679)     real (r8) minmod
   680)     real (r8) medan
   681)     minmod(a,b) = 0.5_r8*(sign(1._r8,a) + sign(1._r8,b))*min(abs(a),abs(b))
   682)     medan(a,b,c) = a + minmod(b-a,c-a)
   683) 
   684)     do k = 1,pver
   685) 
   686) 
   687) !        first divided differences between nodes
   688)        do i = 1, ncol
   689)           delxh(i,k) = (x(i,k+1)-x(i,k))
   690)           sh(i,k) = (f(i,k+1)-f(i,k))/delxh(i,k)
   691)        end do
   692) 
   693) !        first and second divided differences at nodes
   694)        if (k.ge.2) then
   695)           do i = 1,ncol
   696)              d(i,k) = (sh(i,k)-sh(i,k-1))/(x(i,k+1)-x(i,k-1))
   697)              s(i,k) = minmod(sh(i,k),sh(i,k-1))
   698)           end do
   699)        endif
   700)     end do
   701) 
   702) !     second and third divided diffs between nodes
   703)     do k = 2,pver-1
   704)        do i = 1, ncol
   705)           eh(i,k) = (d(i,k+1)-d(i,k))/(x(i,k+2)-x(i,k-1))
   706)           dh(i,k) = minmod(d(i,k),d(i,k+1))
   707)        end do
   708)     end do
   709) 
   710) !     treat the boundaries
   711)     do i = 1,ncol
   712)        e(i,2) = eh(i,2)
   713)        e(i,pver) = eh(i,pver-1)
   714) !        outside level
   715)        fdot(i,1) = sh(i,1) - d(i,2)*delxh(i,1)  &
   716)             - eh(i,2)*delxh(i,1)*(x(i,1)-x(i,3))
   717)        fdot(i,1) = minmod(fdot(i,1),3*sh(i,1))
   718)        fdot(i,pverp) = sh(i,pver) + d(i,pver)*delxh(i,pver)  &
   719)             + eh(i,pver-1)*delxh(i,pver)*(x(i,pverp)-x(i,pver-1))
   720)        fdot(i,pverp) = minmod(fdot(i,pverp),3*sh(i,pver))
   721) !        one in from boundary
   722)        fdot(i,2) = sh(i,1) + d(i,2)*delxh(i,1) - eh(i,2)*delxh(i,1)*delxh(i,2)
   723)        fdot(i,2) = minmod(fdot(i,2),3*s(i,2))
   724)        fdot(i,pver) = sh(i,pver) - d(i,pver)*delxh(i,pver)   &
   725)             - eh(i,pver-1)*delxh(i,pver)*delxh(i,pver-1)
   726)        fdot(i,pver) = minmod(fdot(i,pver),3*s(i,pver))
   727)     end do
   728) 
   729) 
   730)     do k = 3,pver-1
   731)        do i = 1,ncol
   732)           e(i,k) = minmod(eh(i,k),eh(i,k-1))
   733)        end do
   734)     end do
   735) 
   736) 
   737) 
   738)     do k = 3,pver-1
   739) 
   740)        do i = 1,ncol
   741) 
   742) !           p prime at k-0.5
   743)           ppl(i,k)=sh(i,k-1) + dh(i,k-1)*delxh(i,k-1)  
   744) !           p prime at k+0.5
   745)           ppr(i,k)=sh(i,k)   - dh(i,k)  *delxh(i,k)
   746) 
   747)           t = minmod(ppl(i,k),ppr(i,k))
   748) 
   749) !           derivate from parabola thru f(i,k-1), f(i,k), and f(i,k+1)
   750)           pp = sh(i,k-1) + d(i,k)*delxh(i,k-1) 
   751) 
   752) !           quartic estimate of fdot
   753)           fdot(i,k) = pp                            &
   754)                - delxh(i,k-1)*delxh(i,k)            &
   755)                *(  eh(i,k-1)*(x(i,k+2)-x(i,k  ))    &
   756)                + eh(i,k  )*(x(i,k  )-x(i,k-2))      &
   757)                )/(x(i,k+2)-x(i,k-2))
   758) 
   759) !           now limit it
   760)           qpl = sh(i,k-1)       &
   761)                + delxh(i,k-1)*minmod(d(i,k-1)+e(i,k-1)*(x(i,k)-x(i,k-2)), &
   762)                d(i,k)  -e(i,k)*delxh(i,k))
   763)           qpr = sh(i,k)         &
   764)                + delxh(i,k  )*minmod(d(i,k)  +e(i,k)*delxh(i,k-1),        &
   765)                d(i,k+1)+e(i,k+1)*(x(i,k)-x(i,k+2)))
   766) 
   767)           fdot(i,k) = medan(fdot(i,k), qpl, qpr)
   768) 
   769)           ttt = minmod(qpl, qpr)
   770)           tmin = min(0._r8,3*s(i,k),1.5_r8*t,ttt)
   771)           tmax = max(0._r8,3*s(i,k),1.5_r8*t,ttt)
   772) 
   773)           fdot(i,k) = fdot(i,k) + minmod(tmin-fdot(i,k), tmax-fdot(i,k))
   774) 
   775)        end do
   776) 
   777)     end do
   778) 
   779)     return
