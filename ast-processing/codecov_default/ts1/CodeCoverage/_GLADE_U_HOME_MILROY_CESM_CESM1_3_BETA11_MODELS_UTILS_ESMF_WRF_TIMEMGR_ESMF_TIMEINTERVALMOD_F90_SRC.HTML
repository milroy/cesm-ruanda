   309)

   311)       type(ESMF_TimeInterval), intent(in) :: timeinterval
   312)       type(ESMF_Time), optional, intent(in) :: StartTimeIn
   313)       integer, intent(out), optional :: yy

   318)       integer, intent(out), optional :: S
   319)       integer, intent(out), optional :: Sn

   325) ! !DESCRIPTION:
   326) !     Get the value of the {\tt ESMF\_TimeInterval} in units specified by the
   327) !     user via F90 optional arguments.
   328) !
   329) !     Time manager represents and manipulates time internally with integers 
   330) !     to maintain precision.  Hence, user-specified floating point values are
   331) !     converted internally from integers.
   332) !
   333) !     See {\tt ../include/ESMC\_BaseTime.h} and
   334) !     {\tt ../include/ESMC\_TimeInterval.h} for complete description.
   335) !     
   336) !     The arguments are:
   337) !     \begin{description}
   338) !     \item[timeinterval]
   339) !          The object instance to query
   340) !     \item[{[YY]}]
   341) !          Integer years (>= 32-bit)
   342) !     \item[{[YYl]}]
   343) !          Integer years (large, >= 64-bit)
   344) !     \item[{[MO]}]
   345) !          Integer months (>= 32-bit)
   346) !     \item[{[MOl]}]
   347) !          Integer months (large, >= 64-bit)
   348) !     \item[{[D]}]
   349) !          Integer days (>= 32-bit)
   350) !     \item[{[Dl]}]
   351) !          Integer days (large, >= 64-bit)
   352) !     \item[{[H]}]
   353) !          Integer hours
   354) !     \item[{[M]}]
   355) !          Integer minutes
   356) !     \item[{[S]}]
   357) !          Integer seconds (>= 32-bit)
   358) !     \item[{[Sl]}]
   359) !          Integer seconds (large, >= 64-bit)
   360) !     \item[{[MS]}]
   361) !          Integer milliseconds
   362) !     \item[{[US]}]
   363) !          Integer microseconds
   364) !     \item[{[NS]}]
   365) !          Integer nanoseconds
   366) !     \item[{[d\_]}]
   367) !          Double precision days
   368) !     \item[{[h\_]}]
   369) !          Double precision hours
   370) !     \item[{[m\_]}]
   371) !          Double precision minutes
   372) !     \item[{[s\_]}]
   373) !          Double precision seconds
   374) !     \item[{[ms\_]}]
   375) !          Double precision milliseconds
   376) !     \item[{[us\_]}]
   377) !          Double precision microseconds
   378) !     \item[{[ns\_]}]
   379) !          Double precision nanoseconds
   380) !     \item[{[Sn]}]
   381) !          Integer fractional seconds - numerator
   382) !     \item[{[Sd]}]
   383) !          Integer fractional seconds - denominator
   384) !     \item[{[rc]}]
   385) !          Return code; equals {\tt ESMF\_SUCCESS} if there are no errors.
   386) !     \end{description}
   387) !
   388) ! !REQUIREMENTS:
   389) !     TMG1.1
   390) !EOP
   391)       type(ESMF_Time) :: lstarttime
   392)       logical :: lstarttime_set
   393)       logical :: doyear
   394)       INTEGER(ESMF_KIND_I8) :: seconds, secondsym, years
   395)       INTEGER :: ierr
   396)       INTEGER :: mpyi4, iyr,imo,mmon,nmon,mstart,ndays
   397) 
   398)       ierr = ESMF_FAILURE
   399) 
   400)       if (present(StartTimeIn)) then
   401)          lstarttime_set = .true.
   402)          lstarttime = StartTimeIn
   403)       else
   404)          lstarttime_set = timeinterval%StartTime_set
   405)          lstarttime = timeinterval%StartTime
   406)       endif
   407) 
   408) 
   409)       CALL timeintchecknormalized( timeinterval,                &
   410)                                    'ESMF_TimeIntervalGet arg1', &
   411)                                     relative_interval=.true. )
   412)       seconds = timeinterval%basetime%S
   413)       years = timeinterval%YR
   414) 
   415)       secondsym = 0
   416) 
   417)       IF ( PRESENT( YY ) )THEN
   418)         YY = years + timeinterval%MM / MONTHS_PER_YEAR
   419) !        seconds = seconds - years * ( 365_ESMF_KIND_I8 * SECONDS_PER_DAY )
   420)         IF ( PRESENT( MM ) )THEN
   421)            mpyi4 = MONTHS_PER_YEAR
   422)            MM = MOD( timeinterval%MM, mpyi4)
   423)         else
   424)            call wrf_error_fatal("ESMF_TimeIntervalGet: requires MM with YY")
   425)         END IF
   426)       ELSE IF ( PRESENT( MM ) )THEN
   427)         MM = timeinterval%MM + years*12
   428)       else if (lstarttime_set) then
   429)         ! convert years and months to days carefully
   430) 
   431)         mpyi4 = MONTHS_PER_YEAR
   432)         mmon = timeinterval%mm + timeinterval%yr*mpyi4
   433)         mstart = nmonthinyearsec(lstarttime%yr,lstarttime%basetime,lstarttime%calendar%type)
   434) !        write(6,*) 'tcxti1 ',mmon,lstarttime%yr,mstart,lstarttime%basetime%s
   435) 
   436)         iyr = lstarttime%yr
   437)         if (mmon > 0) then
   438)            imo = mstart-1  ! if adding months, start with this month after adding first +1
   439)         else
   440)            imo = mstart    ! if going backwards, start with last month after first -1
   441)         endif
   442)         nmon = 1
   443) !        do nmon = 1,abs(mmon)
   444)         do while (nmon <= abs(mmon))
   445)            if (mmon > 0) then
   446)               if (imo == 12 .and. (abs(mmon) - nmon) > 12) then
   447)                  iyr = iyr + 1
   448)                  nmon = nmon + 12
   449)                  doyear = .true.
   450)               else
   451)                  imo = imo + 1
   452)                  nmon = nmon + 1
   453)                  doyear = .false.
   454)               endif
   455)            else
   456)               if (imo == 1 .and. (abs(mmon) - nmon) > 12) then
   457)                  iyr = iyr - 1
   458)                  nmon = nmon + 12
   459)                  doyear = .true.
   460)               else
   461)                  imo = imo - 1
   462)                  nmon = nmon + 1
   463)                  doyear = .false.
   464)               endif
   465)            endif
   466) 
   467)            do while (imo > 12)
   468)               imo = imo - 12
   469)               iyr = iyr + 1
   470)            enddo
   471)            do while (imo < 1)
   472)               imo = imo + 12
   473)               iyr = iyr - 1
   474)            enddo
   475) 
   476)            if (doyear) then
   477)               ndays = ndaysinyear(iyr,lstarttime%calendar%type)
   478)            else
   479)               ndays = ndaysinmonth(iyr,imo,lstarttime%calendar%type)
   480)            endif
   481)            secondsym = secondsym + (ndays * SECONDS_PER_DAY)
   482) !           write(6,*) 'tcxti2 ',nmon,iyr,imo,ndays
   483)         enddo
   484)         if (mmon < 0) then
   485)            secondsym = -secondsym
   486)         endif
   487) !        write(6,*) 'tcxti3 ',mmon,iyr,imo,secondsym
   488)       elseif (PRESENT(D) .or. PRESENT(d_r8) .or. present(S) .or. present(S_i8)) then
   489)         IF (timeinterval%MM /= 0) then
   490)           CALL wrf_error_fatal("ESMF_TimeIntervalGet:  Need MM with D,d_r8,S,or S_i8")
   491)         endif
   492)         if (timeinterval%YR /= 0) then
   493)           CALL wrf_error_fatal("ESMF_TimeIntervalGet:  Need YY or MM with D,d_r8,S,or S_i8")
   494)         endif
   495)       END IF
   496) 
   497)       seconds = seconds+secondsym
   498) 
   499)       IF ( PRESENT( D ) )THEN
   500)         D = seconds / SECONDS_PER_DAY
   501)         IF ( PRESENT(S) )   S    = mod( seconds, SECONDS_PER_DAY )
   502)         IF ( PRESENT(S_i8)) S_i8 = mod( seconds, SECONDS_PER_DAY )
   503)       ELSE 
   504)         IF ( PRESENT(S) )   S    = seconds
   505)         IF ( PRESENT(S_i8)) S_i8 = seconds
   506)       END IF
   507) 
   508)       IF ( PRESENT( d_r8 ) )THEN
   509)         D_r8 = REAL( seconds, ESMF_KIND_R8 ) / &
   510)                REAL( SECONDS_PER_DAY, ESMF_KIND_R8 )
   511)       END IF
   512) 
   513)       ! If d_r8 present and sec present
   514)       IF ( PRESENT( d_r8 ) )THEN
   515)         IF ( PRESENT( S ) .or. present(s_i8) )THEN
   516)           CALL wrf_error_fatal( &
   517)             "ESMF_TimeIntervalGet:  Can not specify d_r8 and S S_i8 values" )
   518)         END IF
   519)       END IF
   520) 
   521)       ierr = ESMF_SUCCESS
   522) 
   523)       IF ( PRESENT( timeString ) ) THEN
   524)         CALL ESMFold_TimeIntervalGetString( timeinterval, timeString, rc=ierr )
   525)       ENDIF
   526)       
   527)       IF ( PRESENT(Sn) ) THEN
   528)         Sn = timeinterval%basetime%Sn
   529)       ENDIF
   530)       IF ( PRESENT(Sd) ) THEN
   531)         Sd = timeinterval%basetime%Sd
   532)       ENDIF
   533)       
   534)       IF ( PRESENT(rc) ) rc = ierr
   535)     
   536)       end subroutine ESMF_TimeIntervalGet
