     1) module camsrfexch

    43)   type cam_out_t

    51)      real(r8) :: pbot(pcols)         ! bot level pressure
    52)      real(r8) :: rho(pcols)          ! bot level density	
    53)      real(r8) :: netsw(pcols)        !	
    54)      real(r8) :: flwds(pcols)        ! 
    55)      real(r8) :: precsc(pcols)       !
    56)      real(r8) :: precsl(pcols)       !
    57)      real(r8) :: precc(pcols)        ! 
    58)      real(r8) :: precl(pcols)        ! 
    59)      real(r8) :: soll(pcols)         ! 
    60)      real(r8) :: sols(pcols)         ! 
    61)      real(r8) :: solld(pcols)        !
    62)      real(r8) :: solsd(pcols)        !
    63)      real(r8) :: thbot(pcols)        ! 
    64)      real(r8) :: co2prog(pcols)      ! prognostic co2
    65)      real(r8) :: co2diag(pcols)      ! diagnostic co2
    66)      real(r8) :: psl(pcols)
    67)      real(r8) :: bcphiwet(pcols)     ! wet deposition of hydrophilic black carbon
    68)      real(r8) :: bcphidry(pcols)     ! dry deposition of hydrophilic black carbon
    69)      real(r8) :: bcphodry(pcols)     ! dry deposition of hydrophobic black carbon
    70)      real(r8) :: ocphiwet(pcols)     ! wet deposition of hydrophilic organic carbon
    71)      real(r8) :: ocphidry(pcols)     ! dry deposition of hydrophilic organic carbon
    72)      real(r8) :: ocphodry(pcols)     ! dry deposition of hydrophobic organic carbon

    74)      real(r8) :: dstdry1(pcols)      ! dry deposition of dust (bin1)
    75)      real(r8) :: dstwet2(pcols)      ! wet deposition of dust (bin2)
    76)      real(r8) :: dstdry2(pcols)      ! dry deposition of dust (bin2)
    77)      real(r8) :: dstwet3(pcols)      ! wet deposition of dust (bin3)
    78)      real(r8) :: dstdry3(pcols)      ! dry deposition of dust (bin3)
    79)      real(r8) :: dstwet4(pcols)      ! wet deposition of dust (bin4)
    80)      real(r8) :: dstdry4(pcols)      ! dry deposition of dust (bin4)

    83) !---------------------------------------------------------------------------
    84) ! This is the merged state of sea-ice, land and ocean surface parameterizations
    85) !---------------------------------------------------------------------------
    86) 
    87)   type cam_in_t    
    88)      integer  :: lchnk                   ! chunk index
    89)      integer  :: ncol                    ! number of active columns
    90)      real(r8) :: asdir(pcols)            ! albedo: shortwave, direct
    91)      real(r8) :: asdif(pcols)            ! albedo: shortwave, diffuse
    92)      real(r8) :: aldir(pcols)            ! albedo: longwave, direct
    93)      real(r8) :: aldif(pcols)            ! albedo: longwave, diffuse
    94)      real(r8) :: lwup(pcols)             ! longwave up radiative flux
    95)      real(r8) :: lhf(pcols)              ! latent heat flux
    96)      real(r8) :: shf(pcols)              ! sensible heat flux
    97)      real(r8) :: wsx(pcols)              ! surface u-stress (N)
    98)      real(r8) :: wsy(pcols)              ! surface v-stress (N)
    99)      real(r8) :: tref(pcols)             ! ref height surface air temp
   100)      real(r8) :: qref(pcols)             ! ref height specific humidity 
   101)      real(r8) :: u10(pcols)              ! 10m wind speed
   102)      real(r8) :: ts(pcols)               ! merged surface temp 
   103)      real(r8) :: sst(pcols)              ! sea surface temp
   104)      real(r8) :: snowhland(pcols)        ! snow depth (liquid water equivalent) over land 
   105)      real(r8) :: snowhice(pcols)         ! snow depth over ice
   106)      real(r8) :: fco2_lnd(pcols)         ! co2 flux from lnd
   107)      real(r8) :: fco2_ocn(pcols)         ! co2 flux from ocn
   108)      real(r8) :: fdms(pcols)             ! dms flux
   109)      real(r8) :: landfrac(pcols)         ! land area fraction
   110)      real(r8) :: icefrac(pcols)          ! sea-ice areal fraction
   111)      real(r8) :: ocnfrac(pcols)          ! ocean areal fraction
   112)      real(r8), pointer, dimension(:) :: ram1  !aerodynamical resistance (s/m) (pcols)
   113)      real(r8), pointer, dimension(:) :: fv    !friction velocity (m/s) (pcols)
   114)      real(r8), pointer, dimension(:) :: soilw !volumetric soil water (m3/m3)
   115)      real(r8) :: cflx(pcols,pcnst)       ! constituent flux (emissions)
   116)      real(r8) :: ustar(pcols)            ! atm/ocn saved version of ustar
   117)      real(r8) :: re(pcols)               ! atm/ocn saved version of re
   118)      real(r8) :: ssq(pcols)              ! atm/ocn saved version of ssq
   119)      real(r8), pointer, dimension(:,:) :: depvel ! deposition velocities
   120)      real(r8), pointer, dimension(:,:) :: dstflx ! dust fluxes
   121)      real(r8), pointer, dimension(:,:) :: meganflx ! MEGAN fluxes
   122)   end type cam_in_t

   134) ! !DESCRIPTION:

   143)   subroutine hub2atm_alloc( cam_in )
   144)     use seq_drydep_mod,  only: lnd_drydep, n_drydep
   145)     use cam_cpl_indices, only: index_x2a_Sl_ram1, index_x2a_Sl_fv, index_x2a_Sl_soilw, index_x2a_Fall_flxdst1
   146)     use cam_cpl_indices, only: index_x2a_Fall_flxvoc
   147)     use shr_megan_mod,   only: shr_megan_mechcomps_n
   148) 
   149) !
   150) !!ARGUMENTS:
   151) !
   152)    type(cam_in_t), pointer ::  cam_in(:)     ! Merged surface state
   153) !
   154) !!LOCAL VARIABLES:
   155) !
   156)     integer :: c        ! chunk index
   157)     integer :: ierror   ! Error code
   158) !----------------------------------------------------------------------- 
   159) ! 
   160) ! EOP
   161) !
   162)     if ( .not. phys_grid_initialized() ) call endrun( "HUB2ATM_ALLOC error: phys_grid not called yet" )
   163)     allocate (cam_in(begchunk:endchunk), stat=ierror)
   164)     if ( ierror /= 0 )then
   165)       write(iulog,*) 'Allocation error: ', ierror
   166)       call endrun('HUB2ATM_ALLOC error: allocation error')
   167)     end if
   168) 
   169)     do c = begchunk,endchunk
   170)        nullify(cam_in(c)%ram1)
   171)        nullify(cam_in(c)%fv)
   172)        nullify(cam_in(c)%soilw)
   173)        nullify(cam_in(c)%depvel)
   174)        nullify(cam_in(c)%dstflx)
   175)        nullify(cam_in(c)%meganflx)
   176)     enddo  
   177)     do c = begchunk,endchunk 
   178)        if (index_x2a_Sl_ram1>0) then
   179)           allocate (cam_in(c)%ram1(pcols), stat=ierror)
   180)           if ( ierror /= 0 ) call endrun('HUB2ATM_ALLOC error: allocation error ram1')
   181)        endif
   182)        if (index_x2a_Sl_fv>0) then
   183)           allocate (cam_in(c)%fv(pcols), stat=ierror)
   184)           if ( ierror /= 0 ) call endrun('HUB2ATM_ALLOC error: allocation error fv')
   185)        endif
   186)        if (index_x2a_Sl_soilw /= 0) then
   187)           allocate (cam_in(c)%soilw(pcols), stat=ierror)
   188)           if ( ierror /= 0 ) call endrun('HUB2ATM_ALLOC error: allocation error soilw')
   189)        end if
   190)        if (index_x2a_Fall_flxdst1>0) then
   191)           ! Assume 4 bins from surface model ....
   192)           allocate (cam_in(c)%dstflx(pcols,4), stat=ierror)
   193)           if ( ierror /= 0 ) call endrun('HUB2ATM_ALLOC error: allocation error dstflx')
   194)        endif
   195)        if ( index_x2a_Fall_flxvoc>0 .and. shr_megan_mechcomps_n>0 ) then
   196)           allocate (cam_in(c)%meganflx(pcols,shr_megan_mechcomps_n), stat=ierror)
   197)           if ( ierror /= 0 ) call endrun('HUB2ATM_ALLOC error: allocation error meganflx')
   198)        endif
   199)     end do
   200) 
   201)     if (lnd_drydep .and. n_drydep>0) then
   202)        do c = begchunk,endchunk 
   203)           allocate (cam_in(c)%depvel(pcols,n_drydep), stat=ierror)
   204)           if ( ierror /= 0 ) call endrun('HUB2ATM_ALLOC error: allocation error depvel')
   205)        end do
   206)     endif
   207) 
   208)     do c = begchunk,endchunk
   209)        cam_in(c)%lchnk = c
   210)        cam_in(c)%ncol  = get_ncols_p(c)
   211)        cam_in(c)%asdir    (:) = 0._r8
   212)        cam_in(c)%asdif    (:) = 0._r8
   213)        cam_in(c)%aldir    (:) = 0._r8
   214)        cam_in(c)%aldif    (:) = 0._r8
   215)        cam_in(c)%lwup     (:) = 0._r8
   216)        cam_in(c)%lhf      (:) = 0._r8
   217)        cam_in(c)%shf      (:) = 0._r8
   218)        cam_in(c)%wsx      (:) = 0._r8
   219)        cam_in(c)%wsy      (:) = 0._r8
   220)        cam_in(c)%tref     (:) = 0._r8
   221)        cam_in(c)%qref     (:) = 0._r8
   222)        cam_in(c)%u10      (:) = 0._r8
   223)        cam_in(c)%ts       (:) = 0._r8
   224)        cam_in(c)%sst      (:) = 0._r8
   225)        cam_in(c)%snowhland(:) = 0._r8
   226)        cam_in(c)%snowhice (:) = 0._r8
   227)        cam_in(c)%fco2_lnd (:) = 0._r8
   228)        cam_in(c)%fco2_ocn (:) = 0._r8
   229)        cam_in(c)%fdms     (:) = 0._r8
   230)        cam_in(c)%landfrac (:) = posinf
   231)        cam_in(c)%icefrac  (:) = posinf
   232)        cam_in(c)%ocnfrac  (:) = posinf
   233) 
   234)        if (associated(cam_in(c)%ram1)) &
   235)             cam_in(c)%ram1  (:) = 0.1_r8
   236)        if (associated(cam_in(c)%fv)) &
   237)             cam_in(c)%fv    (:) = 0.1_r8
   238)        if (associated(cam_in(c)%soilw)) &
   239)             cam_in(c)%soilw (:) = 0.0_r8
   240)        if (associated(cam_in(c)%dstflx)) &
   241)             cam_in(c)%dstflx(:,:) = 0.0_r8
   242)        if (associated(cam_in(c)%meganflx)) &
   243)             cam_in(c)%meganflx(:,:) = 0.0_r8
   244) 
   245)        cam_in(c)%cflx   (:,:) = 0._r8
   246)        cam_in(c)%ustar    (:) = 0._r8
   247)        cam_in(c)%re       (:) = 0._r8
   248)        cam_in(c)%ssq      (:) = 0._r8
   249)        if (lnd_drydep .and. n_drydep>0) then
   250)           cam_in(c)%depvel (:,:) = 0._r8
   251)        endif
   252)     end do
   253) 
   254)   end subroutine hub2atm_alloc

   377)           
   378)        enddo

   389) subroutine cam_export(state,cam_out,pbuf)
   390) 
   391) !----------------------------------------------------------------------- 
   392) ! 
   393) ! Purpose: 
   394) ! Transfer atmospheric fields into necessary surface data structures
   395) ! 
   396) ! Author: L. Bath  CMS Contact: M. Vertenstein
   397) ! 
   398) !-----------------------------------------------------------------------
   399)    use physics_types,    only: physics_state
   400)    use ppgrid,           only: pver
   401)    use cam_history,      only: outfld
   402)    use comsrf,           only: psm1, srfrpdel, prcsnw
   403)    use chem_surfvals,    only: chem_surfvals_get
   404)    use co2_cycle,        only: co2_transport, c_i
   405)    use physconst,        only: mwdry, mwco2
   406)    use constituents,     only: pcnst
   407)    use cam_control_mod,  only: rair
   408)    use physics_buffer,   only: pbuf_get_index, pbuf_get_field, physics_buffer_desc
   409)    implicit none
   410) 
   411)    !------------------------------Arguments--------------------------------
   412)    !
   413)    ! Input arguments
   414)    !
   415)    type(physics_state),  intent(in)    :: state
   416)    type (cam_out_t),     intent(inout) :: cam_out
   417)    type(physics_buffer_desc), pointer  :: pbuf(:)
   418) 
   419)    !
   420)    !---------------------------Local variables-----------------------------
   421)    !
   422)    integer :: i              ! Longitude index
   423)    integer :: m              ! constituent index
   424)    integer :: lchnk          ! Chunk index
   425)    integer :: ncol
   426)    integer :: prec_dp_idx, snow_dp_idx, prec_sh_idx, snow_sh_idx
   427)    integer :: prec_sed_idx,snow_sed_idx,prec_pcw_idx,snow_pcw_idx
   428) 
   429)    real(r8), pointer :: prec_dp(:)                 ! total precipitation   from ZM convection
   430)    real(r8), pointer :: snow_dp(:)                 ! snow from ZM   convection
   431)    real(r8), pointer :: prec_sh(:)                 ! total precipitation   from Hack convection
   432)    real(r8), pointer :: snow_sh(:)                 ! snow from   Hack   convection
   433)    real(r8), pointer :: prec_sed(:)                ! total precipitation   from ZM convection
   434)    real(r8), pointer :: snow_sed(:)                ! snow from ZM   convection
   435)    real(r8), pointer :: prec_pcw(:)                ! total precipitation   from Hack convection
   436)    real(r8), pointer :: snow_pcw(:)                ! snow from Hack   convection
   437)    !-----------------------------------------------------------------------
   438) 
   439)    lchnk = state%lchnk
   440)    ncol  = state%ncol
   441) 
   442)    prec_dp_idx = pbuf_get_index('PREC_DP')
   443)    snow_dp_idx = pbuf_get_index('SNOW_DP')
   444)    prec_sh_idx = pbuf_get_index('PREC_SH')
   445)    snow_sh_idx = pbuf_get_index('SNOW_SH')
   446)    prec_sed_idx = pbuf_get_index('PREC_SED')
   447)    snow_sed_idx = pbuf_get_index('SNOW_SED')
   448)    prec_pcw_idx = pbuf_get_index('PREC_PCW')
   449)    snow_pcw_idx = pbuf_get_index('SNOW_PCW')
   450) 
   451)    call pbuf_get_field(pbuf, prec_dp_idx, prec_dp)
   452)    call pbuf_get_field(pbuf, snow_dp_idx, snow_dp)
   453)    call pbuf_get_field(pbuf, prec_sh_idx, prec_sh)
   454)    call pbuf_get_field(pbuf, snow_sh_idx, snow_sh)
   455)    call pbuf_get_field(pbuf, prec_sed_idx, prec_sed)
