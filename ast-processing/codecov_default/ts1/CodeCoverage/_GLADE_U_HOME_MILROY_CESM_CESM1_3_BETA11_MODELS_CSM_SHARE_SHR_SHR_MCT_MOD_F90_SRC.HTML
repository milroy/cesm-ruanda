    56) ! !IROUTINE: shr_mct_sMatReadnc - read all mapping data from a NetCDF SCRIP file
    57) !                              in to a full SparseMatrix
    58) !
    59) ! !DESCRIPTION:
    60) !   Read in mapping matrix data from a SCRIP netCDF data file so a sMat.
    61) !
    62) ! !REMARKS:
    63) !   Based on cpl_map_read
    64) !
    65) ! !REVISION HISTORY:
    66) !     2006 Nov 27: R. Jacob
    67) !
    68) ! !INTERFACE: ------------------------------------------------------------------
    69) 
    70) subroutine shr_mct_sMatReadnc(sMat,fileName)
    71) 
    72) #include <netcdf.inc>
    73) 
    74) ! !INPUT/OUTPUT PARAMETERS:
    75) 
    76)    type(mct_sMat),intent(inout)  :: sMat
    77)    character(*),intent(in)  :: filename  ! netCDF file to read
    78) 
    79) !EOP
    80) 
    81)    !--- local ---
    82)    integer(IN)           :: n       ! generic loop indicies
    83)    integer(IN)           :: na      ! size of source domain
    84)    integer(IN)           :: nb      ! size of destination domain
    85)    integer(IN)           :: ns      ! number of non-zero elements in matrix
    86)    integer(IN)           :: ni,nj   ! number of row and col in the matrix
    87)    integer(IN)           :: igrow   ! aVect index for matrix row
    88)    integer(IN)           :: igcol   ! aVect index for matrix column
    89)    integer(IN)           :: iwgt    ! aVect index for matrix element
    90) 
    91)    real(R8)   ,allocatable :: rtemp(:)  ! reals
    92)    integer(IN),allocatable :: itemp(:)  ! ints
    93) 
    94)    integer(IN)           :: rcode   ! netCDF routine return code
    95)    integer(IN)           :: fid     ! netCDF file      ID
    96)    integer(IN)           :: vid     ! netCDF variable  ID
    97)    integer(IN)           :: did     ! netCDF dimension ID
    98) 
    99)    character(*),parameter :: subName = '(shr_mct_sMatReadnc) '
   100)    character(*),parameter :: F00 = "('(shr_mct_sMatReadnc) ',4a)"
   101)    character(*),parameter :: F01 = '("(shr_mct_sMatReadnc) ",2(a,i9))'
   102) 
   103)    if (s_loglev > 0) write(s_logunit,F00) "reading mapping matrix data..."
   104) 
   105)    !----------------------------------------------------------------------------
   106)    ! open & read the file
   107)    !----------------------------------------------------------------------------
   108)    if (s_loglev > 0) write(s_logunit,F00) "* file name                  : ",trim(fileName)
   109)    rcode = nf_open(filename,NF_NOWRITE,fid)
   110)    if (rcode /= NF_NOERR) then
   111)       write(s_logunit,F00) nf_strerror(rcode)
   112)       call mct_die(subName,"error opening Netcdf file")
   113)    endif
   114) 
   115)    !--- allocate memory & get matrix data ----------
   116)    rcode = nf_inq_dimid (fid, 'n_s', did)  ! size of sparse matrix
   117)    rcode = nf_inq_dimlen(fid, did  , ns)
   118)    rcode = nf_inq_dimid (fid, 'n_a', did)  ! size of  input vector
   119)    rcode = nf_inq_dimlen(fid, did  , na)
   120)    rcode = nf_inq_dimid (fid, 'n_b', did)  ! size of output vector
   121)    rcode = nf_inq_dimlen(fid, did  , nb)
   122) 
   123)    if (s_loglev > 0) write(s_logunit,F01) "* matrix dimensions src x dst: ",na,' x',nb
   124)    if (s_loglev > 0) write(s_logunit,F01) "* number of non-zero elements: ",ns
   125) 
   126)    !----------------------------------------------------------------------------
   127)    ! init the mct sMat data type
   128)    !----------------------------------------------------------------------------
   129)    ! mct_sMat_init must be given the number of rows and columns that
   130)    ! would be in the full matrix.  Nrows= size of output vector=nb.
   131)    ! Ncols = size of input vector = na.
   132)    call mct_sMat_init(sMat, nb, na, ns)
   133) 
   134)    igrow = mct_sMat_indexIA(sMat,'grow')
   135)    igcol = mct_sMat_indexIA(sMat,'gcol')
   136)    iwgt  = mct_sMat_indexRA(sMat,'weight')
   137) 
   138)    !!!!!!!!!!!!!!!!!!!!!!!!!!
   139)    ! read and load matrix weights
   140)    allocate(rtemp(ns),stat=rcode)
   141)    if (rcode /= 0) &
   142)      call mct_die(subName,':: allocate weights',rcode)
   143) 
   144)    rcode = nf_inq_varid     (fid,'S'  ,vid)
   145)    rcode = nf_get_var_double(fid,vid  ,rtemp  )
   146)    if (rcode /= NF_NOERR .and. s_loglev > 0) write(s_logunit,F00) nf_strerror(rcode)
   147) 
   148)    sMat%data%rAttr(iwgt ,:) =   rtemp(:)
   149) 
   150)    deallocate(rtemp, stat=rcode)
   151)    if (rcode /= 0) call mct_perr_die(subName,':: deallocate weights',rcode)
   152) 
   153)    !!!!!!!!!!!!!!!!!!!!!!!!!!
   154)    ! read and load rows
   155)    allocate(itemp(ns),stat=rcode)
   156)    if (rcode /= 0) call mct_perr_die(subName,':: allocate rows',rcode)
   157) 
   158)    rcode = nf_inq_varid     (fid,'row',vid)
   159)    rcode = nf_get_var_int   (fid,vid  ,itemp)
   160)    if (rcode /= NF_NOERR .and. s_loglev > 0) write(s_logunit,F00) nf_strerror(rcode)
   161) 
   162)    sMat%data%iAttr(igrow,:) = itemp(:)
   163) 
   164) 
   165)    !!!!!!!!!!!!!!!!!!!!!!!!!!
   166)    ! read and load columns
   167)    itemp(:) = 0
   168) 
   169)    rcode = nf_inq_varid     (fid,'col',vid)
   170)    rcode = nf_get_var_int   (fid,vid  ,itemp)
   171)    if (rcode /= NF_NOERR .and. s_loglev > 0) write(s_logunit,F00) nf_strerror(rcode)
   172) 
   173)    sMat%data%iAttr(igcol,:) = itemp(:)
   174) 
   175)    deallocate(itemp, stat=rcode)
   176)    if (rcode /= 0) call mct_perr_die(subName,':: deallocate cols',rcode)
   177) 
   178)    rcode = nf_close(fid)
   179) 
   180)    if (s_loglev > 0) write(s_logunit,F00) "... done reading file"
   181)    call shr_sys_flush(s_logunit)
   182) 
   183) end subroutine shr_mct_sMatReadnc

   201)            Label1,Value1,Label2,Value2,Label3,Value3)

   206)    character(len=*), intent(in)  :: Label1
   207)    character(len=*), intent(out) :: Value1

   209)    character(len=*), intent(out),optional :: Value2

   234)

   236) 
   237)       call i90_label(trim(Label2),iret)
   238)       if(iret /= 0) then
   239)          write(s_logunit,*) trim(subname),"Cant find label ",Label2
   240)          call shr_sys_abort(trim(subname)//' Label2 Not Found')

   242)

   245)          write(s_logunit,*)"Error reading token ",Value2
   246)          call shr_sys_abort(trim(subname)//' Error on read value2')
   247)       endif
   248) 
   249)    endif

   261)          write(s_logunit,*)"Error reading token ",Value3
   262)          call shr_sys_abort(trim(subname)//' Error on read value3')

   264)

   276) ! !DESCRIPTION:

   287) subroutine shr_mct_sMatPInitnc_mapfile(sMatP, gsMapX, gsMapY, &
   288)                                        filename, maptype, mpicom, &
   289)                                        ni_i, nj_i, ni_o, nj_o, &
   290)                                        areasrc, areadst)
   291) 
   292) ! !INPUT/OUTPUT PARAMETERS:
   293) 
   294)    type(mct_sMatP),intent(inout)         :: sMatP
   295)    type(mct_gsMap),intent(in)            :: gsMapX
   296)    type(mct_gsMap),intent(in)            :: gsMapY
   297)    character(*)   ,intent(in)            :: filename        ! scrip map file to read
   298)    character(*)   ,intent(in)            :: maptype         ! map type
   299)    integer        ,intent(in)            :: mpicom
   300)    integer        ,intent(out), optional :: ni_i            ! number of longitudes on input grid
   301)    integer        ,intent(out), optional :: nj_i            ! number of latitudes  on input grid
   302)    integer        ,intent(out), optional :: ni_o            ! number of longitudes on output grid
   303)    integer        ,intent(out), optional :: nj_o            ! number of latitudes  on output grid
   304)    type(mct_Avect),intent(out), optional :: areasrc         ! area of src grid from mapping file
   305)    type(mct_Avect),intent(out), optional :: areadst         ! area of src grid from mapping file
   306) 
   307) !EOP
   308)    type(mct_sMat ) :: sMati    ! initial sMat from read (either root or decomp)
   309)    type(mct_Avect) :: areasrc_map ! area of src grid from mapping file
   310)    type(mct_Avect) :: areadst_map ! area of dst grid from mapping file
   311) 
   312)    integer          :: lsize
   313)    integer          :: iret
   314)    integer          :: pe_loc
   315)    logical          :: usevector 
   316)    character(len=3) :: Smaptype
   317)    character(*),parameter :: areaAV_field = 'aream'
   318)    character(*),parameter :: F00 = "('(shr_mct_sMatPInitnc) ',4a)"
   319)    character(*),parameter :: F01 = "('(shr_mct_sMatPInitnc) ',a,i10)"
   320) 
   321)    call shr_mpi_commrank(mpicom,pe_loc)
   322) 
   323)    if (s_loglev > 0) write(s_logunit,*) " "
   324)    if (s_loglev > 0) write(s_logunit,F00) "Initializing SparseMatrixPlus"
   325)    if (s_loglev > 0) write(s_logunit,F00) "SmatP mapname ",trim(filename)
   326)    if (s_loglev > 0) write(s_logunit,F00) "SmatP maptype ",trim(maptype)
   327) 
   328)    if (maptype == "X") then
   329)       Smaptype = "src"
   330)    else if(maptype == "Y") then
   331)       Smaptype = "dst"
   332)    end if
   333) 
   334)    call shr_mpi_commrank(mpicom, pe_loc)
   335) 
   336)    lsize = mct_gsMap_lsize(gsMapX, mpicom)
   337)    call mct_aVect_init(areasrc_map, rList=areaAV_field, lsize=lsize)
   338) 
   339)    lsize = mct_gsMap_lsize(gsMapY, mpicom)
   340)    call mct_aVect_init(areadst_map, rList=areaAV_field, lsize=lsize)
   341) 
   342)    if (present(ni_i) .and. present(nj_i) .and. present(ni_o) .and. present(nj_o)) then
   343)       call shr_mct_sMatReaddnc(sMati, gsMapX, gsMapY, Smaptype, areasrc_map, areadst_map, &
   344)            fileName, pe_loc, mpicom, ni_i, nj_i, ni_o, nj_o)
   345)    else
   346)       call shr_mct_sMatReaddnc(sMati, gsMapX, gsMapY, Smaptype, areasrc_map, areadst_map, &
   347)            fileName, pe_loc, mpicom)
   348)    end if
   349)    call mct_sMatP_Init(sMatP, sMati, gsMapX, gsMapY, 0, mpicom, gsMapX%comp_id)
   350) 
   351) #ifdef CPP_VECTOR
   352)    !--- initialize the vector parts of the sMat ---
   353)    call mct_sMatP_Vecinit(sMatP)
   354) #endif
   355) 
   356)    lsize = mct_smat_gNumEl(sMatP%Matrix,mpicom)
   357)    if (s_loglev > 0) write(s_logunit,F01) "Done initializing SmatP, nElements = ",lsize
   358) 
   359) #ifdef CPP_VECTOR
   360)    usevector = .true.
   361) #else
   362)    usevector = .false.
   363) #endif
   364)    if (present(areasrc)) then
   365)       call mct_aVect_copy(aVin=areasrc_map, aVout=areasrc, vector=usevector)
   366)    end if
   367)    if (present(areadst)) then
   368)       call mct_aVect_copy(aVin=areadst_map, aVout=areadst, vector=usevector)
   369)    end if
   370) 
   371)    call mct_aVect_clean(areasrc_map)
   372)    call mct_aVect_clean(areadst_map)
   373) 
   374)    call mct_sMat_Clean(sMati)
   375) 
   376) end subroutine shr_mct_sMatPInitnc_mapfile

   380) ! !IROUTINE:  shr_mct_sMatReaddnc - Do a distributed read of a NetCDF SCRIP file and
   381) !                                return weights in a distributed SparseMatrix
   382) !
   383) ! !DESCRIPTION: 
   384) !     Read in mapping matrix data from a SCRIP netCDF data file using
   385) !     a low memory method and then scatter to all pes.
   386) !
   387) ! !REMARKS:
   388) !   This routine leverages gsmaps to determine scatter pattern
   389) !   The scatter is implemented as a bcast of all weights then a local
   390) !     computation on each pe to determine with weights to keep based
   391) !     on gsmap information.
   392) !   The algorithm to determine whether a weight belongs on a pe involves
   393) !     creating a couple local arrays (lsstart and lscount) which are
   394) !     the local values of start and length from the gsmap.  these are
   395) !     sorted via a bubble sort and then searched via a binary search
   396) !     to check whether a global index is on the local pe.
   397) !   The local buffer sizes are estimated up front based on ngridcell/npes
   398) !     plus 20% (see 1.2 below).  If the local buffer size fills up, then
   399) !     the buffer is reallocated 50% large (see 1.5 below) and the fill
   400) !     continues.  The idea is to trade off memory reallocation and copy
   401) !     with memory usage.  1.2 and 1.5 are arbitary, other values may
   402) !     result in better performance.
   403) !   Once all the matrix weights have been read, the sMat is initialized,
   404) !     the values from the buffers are copied in, and everything is deallocated.
   405) 
   406) ! !SEE ALSO: 
   407) !    mct/m_SparseMatrix.F90 (MCT source code)
   408) !
   409) ! !REVISION HISTORY: 
   410) !    2007-Jan-18 - T. Craig -- first version
   411) !    2007-Mar-20 - R. Jacob -- rename to shr_mct_sMatReaddnc.  Remove use of cpl_
   412) !                  variables and move to shr_mct_mod
   413) ! 
   414) ! !INTERFACE:  -----------------------------------------------------------------
   415) 
   416) subroutine shr_mct_sMatReaddnc(sMat,SgsMap,DgsMap,newdom,areasrc,areadst, &
   417)                             fileName,mytask, mpicom, ni_i,nj_i,ni_o,nj_o )
   418) 
   419) ! !USES:
   420) 
   421) #include <netcdf.inc>
   422) 
   423) ! !INPUT/OUTPUT PARAMETERS:
   424) 
   425)    type(mct_sMat)  ,intent(out)           :: sMat    ! mapping data
   426)    type(mct_gsMap) ,intent(in) ,target    :: SgsMap  ! src gsmap
   427)    type(mct_gSMap) ,intent(in) ,target    :: DgsMap  ! dst gsmap
   428)    character(*)    ,intent(in)            :: newdom  ! type of sMat (src or dst)
   429)    type(mct_Avect) ,intent(out), optional :: areasrc ! area of src grid from mapping file
   430)    type(mct_Avect) ,intent(out), optional :: areadst ! area of dst grid from mapping file
   431)    character(*)    ,intent(in)            :: filename! netCDF file to read
   432)    integer(IN)     ,intent(in)            :: mytask   ! processor id
   433)    integer(IN)     ,intent(in)            :: mpicom  ! communicator
   434)    integer(IN)     ,intent(out), optional :: ni_i    ! number of lons on input grid   
   435)    integer(IN)     ,intent(out), optional :: nj_i    ! number of lats on input grid   
   436)    integer(IN)     ,intent(out), optional :: ni_o    ! number of lons on output grid   
   437)    integer(IN)     ,intent(out), optional :: nj_o    ! number of lats on output grid   
   438) 
   439) ! !EOP
   440) 
   441)    !--- local ---
   442)    integer(IN)           :: n,m     ! generic loop indicies
   443)    integer(IN)           :: na      ! size of source domain
   444)    integer(IN)           :: nb      ! size of destination domain
   445)    integer(IN)           :: ns      ! number of non-zero elements in matrix
   446)    integer(IN)           :: ni,nj   ! number of row and col in the matrix
   447)    integer(IN)           :: igrow   ! aVect index for matrix row
   448)    integer(IN)           :: igcol   ! aVect index for matrix column
   449)    integer(IN)           :: iwgt    ! aVect index for matrix element
   450)    integer(IN)           :: iarea   ! aVect index for area
   451)    integer(IN)           :: rsize   ! size of read buffer
   452)    integer(IN)           :: cnt     ! local num of wgts
   453)    integer(IN)           :: cntold  ! local num of wgts, previous read
   454)    integer(IN)           :: start(1)! netcdf read
   455)    integer(IN)           :: count(1)! netcdf read
   456)    integer(IN)           :: bsize   ! buffer size
   457)    integer(IN)           :: nread   ! number of reads 
   458)    logical               :: mywt    ! does this weight belong on my pe
   459) 
   460)    !--- buffers for i/o ---
   461)    real(R8)   ,allocatable :: rtemp(:) ! real temporary
   462)    real(R8)   ,allocatable :: Sbuf(:)  ! real weights
   463)    integer(IN),allocatable :: Rbuf(:)  ! ints rows
   464)    integer(IN),allocatable :: Cbuf(:)  ! ints cols
   465) 
   466)    !--- variables associated with local computation of global indices
   467)    integer(IN)             :: lsize     ! size of local seg map
   468)    integer(IN)             :: commsize  ! size of local communicator
   469)    integer(IN),allocatable :: lsstart(:) ! local seg map info
   470)    integer(IN),allocatable :: lscount(:) ! local seg map info
   471)    type(mct_gsMap),pointer :: mygsmap ! pointer to one of the gsmaps
   472)    integer(IN)             :: l1,l2     ! generice indices for sort
   473)    logical                 :: found     ! for sort
   474) 
   475)    !--- variable assocaited with local data buffers and reallocation
   476)    real(R8)   ,allocatable :: Snew(:),Sold(:)  ! reals
   477)    integer(IN),allocatable :: Rnew(:),Rold(:)  ! ints
   478)    integer(IN),allocatable :: Cnew(:),Cold(:)  ! ints
