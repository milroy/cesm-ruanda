     1) module shr_reprosum_mod
     2) !----------------------------------------------------------------------- 
     3) ! 
     4) ! Purpose: 
     5) ! Compute reproducible global sums of a set of arrays across an MPI 
     6) ! subcommunicator
     7) !
     8) ! Methods: 
     9) ! Compute using either or both a scalable, reproducible algorithm and a
    10) ! scalable, nonreproducible algorithm:
    11) ! * Reproducible (scalable): 
    12) !    Convert to fixed point (integer vector representation) to enable
    13) !    reproducibility when using MPI_Allreduce
    14) ! * Alternative usually reproducible (scalable):
    15) !    Use parallel double-double algorithm due to Helen He and 
    16) !    Chris Ding, based on David Bailey's/Don Knuth's DDPDD algorithm
    17) ! * Nonreproducible (scalable): 
    18) !    Floating point and MPI_Allreduce based.
    19) ! If computing both reproducible and nonreproducible sums, compare 
    20) ! these and report relative difference (if absolute difference
    21) ! less than sum) or absolute difference back to calling routine.
    22) !
    23) ! Author: P. Worley (based on suggestions from J. White for fixed 
    24) !                    point algorithm and on He/Ding paper for ddpdd 
    25) !                    algorithm)

   186)    subroutine shr_reprosum_calc (arr, arr_gsum, nsummands, dsummands,     &

   188)                                  arr_gbl_max, arr_gbl_max_out,            &
   189)                                  arr_max_levels, arr_max_levels_out,      &
   190)                                  gbl_max_nsummands, gbl_max_nsummands_out,&
   191)                                  gbl_count, repro_sum_validate,           &
   192)                                  repro_sum_stats, rel_diff, commid        )
   193) !----------------------------------------------------------------------
   194) ! 
   195) ! Purpose: 
   196) ! Compute the global sum of each field in "arr" using the indicated 
   197) ! communicator with a reproducible yet scalable implementation based 
   198) ! on a fixed point algorithm. An alternative is to use an "almost
   199) ! always reproducible" floating point algorithm, as described below.
   200) !
   201) ! The accuracy of the fixed point algorithm is controlled by the
   202) ! number of "levels" of integer expansion. The algorithm will calculate
   203) ! the number of levels that is required for the sum to be essentially
   204) ! exact. The optional parameter arr_max_levels can be used to override
   205) ! the calculated value. The optional parameter arr_max_levels_out can be 
   206) ! used to return the values used.
   207) !
   208) ! The algorithm also requires an upper bound on 
   209) ! the maximum summand (in absolute value) for each field, and will 
   210) ! calculate this internally. However, if the optional parameters
   211) ! arr_max_levels and arr_gbl_max are both set, then the algorithm will
   212) ! use the values in arr_gbl_max for the upper bounds instead. If these
   213) ! are not upper bounds, or if the upper bounds are not tight enough
   214) ! to achieve the requisite accuracy, and if the optional parameter
   215) ! repro_sum_validate is NOT set to .false., the algorithm will repeat the
   216) ! computation with appropriate upper bounds. If only arr_gbl_max is present, 
   217) ! then the maxima are computed internally (and the specified values are
   218) ! ignored). The optional parameter arr_gbl_max_out can be 
   219) ! used to return the values used.
   220) !
   221) ! Finally, the algorithm requires an upper bound on the number of
   222) ! local summands across all processes. This will be calculated internally, 
   223) ! using an MPI collective, but the value in the optional argument 
   224) ! gbl_max_nsummands will be used instead if (1) it is present, (2)
   225) ! it is > 0, and (3) the maximum value and required number of levels 
   226) ! are also specified. (If the maximum value is calculated, the same 
   227) ! MPI collective is used to determine the maximum number of local 
   228) ! summands.) The accuracy of the user-specified value is not checked. 
   229) ! However, if set to < 1, the value will instead be calculated. If the 
   230) ! optional parameter gbl_max_nsummands_out is present, then the value 
   231) ! used (gbl_max_nsummands if >= 1; calculated otherwise) will be 
   232) ! returned.
   233) !
   234) ! If requested (by setting shr_reprosum_reldiffmax >= 0.0 and passing in
   235) ! the optional rel_diff parameter), results are compared with a 
   236) ! nonreproducible floating point algorithm. 
   237) !
   238) ! Note that the cost of the algorithm is not strongly correlated with 
   239) ! the number of levels, which primarily shows up as a (modest) increase
   240) ! in cost of the MPI_Allreduce as a function of vector length. Rather the 
   241) ! cost is more a function of (a) the number of integers required to 
   242) ! represent an individual summand and (b) the number of MPI_Allreduce
   243) ! calls. The number of integers required to represent an individual 
   244) ! summand is 1 or 2 when using 8-byte integers for 8-byte real summands 
   245) ! when the number of local summands is not too large. As the number of 
   246) ! local summands increases, the number of integers required increases. 
   247) ! The number of MPI_Allreduce calls is either 2 (specifying nothing) or
   248) ! 1 (specifying gbl_max_nsummands, arr_max_levels, and arr_gbl_max 
   249) ! correctly). When specifying arr_max_levels and arr_gbl_max 
   250) ! incorrectly, 3 or 4 MPI_Allreduce calls will be required.
   251) !
   252) ! The alternative algorithm is a minor modification of a parallel
   253) ! implementation of David Bailey's routine DDPDD by Helen He
   254) ! and Chris Ding. Bailey uses the Knuth trick to implement quadruple
   255) ! precision summation of double precision values with 10 double 
   256) ! precision operations. The advantage of this algorithm is that 
   257) ! it requires a single MPI_Allreduce and is less expensive per summand
   258) ! than is the fixed precision algorithm. The disadvantage is that it 
   259) ! is not guaranteed to be reproducible (though it is reproducible 
   260) ! much more often than is the standard algorithm). This alternative 
   261) ! is used when the optional parameter ddpdd_sum is set to .true. It is 
   262) ! also used if the fixed precision algorithm radix assumption does not 
   263) ! hold.

   547)             allocate(arr_tlmin_exp(nflds,omp_nthreads))
   548)             allocate(isum_beg(omp_nthreads))
   549)             allocate(isum_end(omp_nthreads))
   550) 
   551) ! split summand index range over OpenMP threads
   552)             call split_indices(nsummands, omp_nthreads, isum_beg, isum_end)
   553) 
   554) !$omp parallel do      &
   555) !$omp default(shared)  &
   556) !$omp private(ithread, ifld, isum, arr_exp, arr_exp_tlmin, arr_exp_tlmax)
   557)             do ithread=1,omp_nthreads
   558)                call t_startf('repro_sum_loopa')
   559)                do ifld=1,nflds
   560)                   arr_exp_tlmin = MAXEXPONENT(1._r8)
   561)                   arr_exp_tlmax = MINEXPONENT(1._r8)
   562)                   do isum=isum_beg(ithread),isum_end(ithread)
   563)                      if (arr(isum,ifld) .ne. 0.0_r8) then
   564)                         arr_exp = exponent(arr(isum,ifld))
   565)                         arr_exp_tlmin = min(arr_exp,arr_exp_tlmin)
   566)                         arr_exp_tlmax = max(arr_exp,arr_exp_tlmax)
   567)                      endif
   568)                   end do
   569)                   arr_tlmin_exp(ifld,ithread) = arr_exp_tlmin
   570)                   arr_tlmax_exp(ifld,ithread) = arr_exp_tlmax
   571)                end do
   572)                call t_stopf('repro_sum_loopa')
   573)             end do
   574) 
   575)             do ifld=1,nflds
   576)                arr_lmax_exp(ifld) = maxval(arr_tlmax_exp(ifld,:))
   577)                arr_lmin_exp(ifld) = minval(arr_tlmin_exp(ifld,:))
   578)             end do
   579)             deallocate(arr_tlmin_exp,arr_tlmax_exp,isum_beg,isum_end)
   580) 
   581)             arr_lextremes(0,:) = -nsummands
   582)             arr_lextremes(1:nflds,1) = -arr_lmax_exp(:)
   583)             arr_lextremes(1:nflds,2) = arr_lmin_exp(:)
   584)             call t_startf("repro_sum_allr_minmax")
   585)             call mpi_allreduce (arr_lextremes, arr_gextremes, 2*(nflds+1), &
   586)                                 MPI_INTEGER, MPI_MIN, mpi_comm, ierr)
   587)             call t_stopf("repro_sum_allr_minmax")
   588)             max_nsummands   = -arr_gextremes(0,1)
   589)             arr_gmax_exp(:) = -arr_gextremes(1:nflds,1)
   590)             arr_gmin_exp(:) =  arr_gextremes(1:nflds,2)
   591) 
   592) ! if a field is identically zero, arr_gmin_exp still equals MAXEXPONENT
   593) !   and arr_gmax_exp still equals MINEXPONENT. In this case, set 
   594) !   arr_gmin_exp = arr_gmax_exp = MINEXPONENT
   595)             do ifld=1,nflds
   596)                arr_gmin_exp(ifld) = min(arr_gmax_exp(ifld),arr_gmin_exp(ifld))
   597)             enddo
   598) 
   599) ! if requested, return upper bounds on observed maxima 
   600)             if ( present(arr_gbl_max_out) ) then
   601)                do ifld=1,nflds
   602)                   arr_gbl_max_out(ifld) = scale(1.0_r8,arr_gmax_exp(ifld))
   603)                enddo
   604)             endif
   605) 
   606) ! if requested, return max_nsummands before it is redefined
   607)             if ( present( gbl_max_nsummands_out) ) then
   608)                gbl_max_nsummands_out = max_nsummands
   609)             endif
   610) 
   611) ! determine maximum shift (same as in previous branch, but with calculated
   612) !  max_nsummands). Shift needs to be small enough that summation does not 
   613) !  exceed maximum number of digits in i8.
   614) 
   615) ! summing within each thread first
   616)             max_nsummands = (max_nsummands/omp_nthreads) + 1
   617) ! then over threads and tasks
   618)             max_nsummands = max(max_nsummands, tasks*omp_nthreads)
   619) 
   620)             xmax_nsummands = dble(max_nsummands)
   621)             arr_max_shift = digits(0_i8) - (exponent(xmax_nsummands) + 1)
   622)             if (arr_max_shift < 2) then
   623)                call shr_sys_abort('repro_sum failed: number of summands too '// &
   624)                                   'large for fixed precision algorithm' )
   625)             endif
   626) 
   627) ! determine maximum number of levels required for each field
   628) ! ((digits(0_i8) + (arr_gmax_exp(ifld)-arr_gmin_exp(ifld))) / arr_max_shift)
   629) ! + 1 because first truncation probably does not involve a maximal shift
   630) ! + 1 to guarantee that the integer division rounds up (not down)
   631) ! (setting lower bound on max_level*nflds to be 64 to improve OpenMP
   632) !  performance for loopb in shr_reprosum_int)
   633)             max_level = (64/nflds) + 1
   634)             do ifld=1,nflds
   635)                max_levels(ifld) = 2 + &
   636)                 ((digits(0_i8) + (arr_gmax_exp(ifld)-arr_gmin_exp(ifld))) &
   637)                 / arr_max_shift)
   638)                if ( present(arr_max_levels) .and. (.not. validate) ) then
   639) ! if validate true, then computation with arr_max_levels failed
   640) !  previously
   641)                   if ( arr_max_levels(ifld) > 0 ) then
   642)                      max_levels(ifld) = &
   643)                         min(arr_max_levels(ifld),max_levels(ifld))
   644)                   endif
   645)                endif
   646)                if (max_level < max_levels(ifld)) &
   647)                   max_level = max_levels(ifld)
   648)             enddo
   649) 
   650) ! if requested, return calculated levels
   651)             if ( present(arr_max_levels_out) ) then
   652)                do ifld=1,nflds
   653)                   arr_max_levels_out(ifld) = max_levels(ifld)
   654)                enddo
   655)             endif
   656) 
   657) ! calculate sum
   658)             validate = .false.
   659)             call shr_reprosum_int(arr, arr_gsum, nsummands, dsummands, nflds, &
   660)                                   arr_max_shift, arr_gmax_exp, max_levels, &
   661)                                   max_level, validate, recompute, &
   662)                                   omp_nthreads, mpi_comm)
   663) 
   664)          endif
   665) 
   666)          call t_stopf('shr_reprosum_int')
   667) 
   668)       endif
   669) 
   670) ! compare fixed and floating point results
   671)       if ( present(rel_diff) ) then
   672)          if (shr_reprosum_reldiffmax >= 0.0_r8) then
   673) 
   674)             call t_barrierf('sync_nonrepro_sum',mpi_comm)
   675)             call t_startf('nonrepro_sum')
   676) ! record statistic
   677)             nonrepro_sum = 1
   678) ! compute nonreproducible sum
   679)             arr_lsum(:) = 0._r8
   680) !$omp parallel do      &
   681) !$omp default(shared)  &
   682) !$omp private(ifld, isum)
   683)             do ifld=1,nflds
   684)                do isum=1,nsummands
   685)                   arr_lsum(ifld) = arr(isum,ifld) + arr_lsum(ifld)
   686)                end do
   687)             end do
   688) 
   689)             call mpi_allreduce (arr_lsum, arr_gsum_fast, nflds, &
   690)                                 MPI_REAL8, MPI_SUM, mpi_comm, ierr)
   691) 
   692)             call t_stopf('nonrepro_sum')
   693) 
   694) ! determine differences
   695) !$omp parallel do      &
   696) !$omp default(shared)  &
   697) !$omp private(ifld, abs_diff)
   698)             do ifld=1,nflds
   699)                abs_diff = abs(arr_gsum_fast(ifld)-arr_gsum(ifld))
   700)                if (abs(arr_gsum(ifld)) > abs_diff) then
   701)                   rel_diff(1,ifld) = abs_diff/abs(arr_gsum(ifld))
   702)                else
   703)                   rel_diff(1,ifld) = abs_diff
   704)                endif
   705)                rel_diff(2,ifld) = abs_diff
   706)             enddo
   707)          else
   708)             rel_diff(:,:) = 0.0_r8
   709)          endif
   710)       endif
   711) 
   712) ! return statistics
   713)       if ( present(repro_sum_stats) ) then
   714)          repro_sum_stats(1) = repro_sum_stats(1) + repro_sum_fast
   715)          repro_sum_stats(2) = repro_sum_stats(2) + repro_sum_slow
   716)          repro_sum_stats(3) = repro_sum_stats(3) + repro_sum_both
   717)          repro_sum_stats(4) = repro_sum_stats(4) + nonrepro_sum
   718)          repro_sum_stats(5) = repro_sum_stats(5) + gbl_max_red
   719)       endif
   720) 
   721) 
   722)    end subroutine shr_reprosum_calc
   723)

   728)    subroutine shr_reprosum_int (arr, arr_gsum, nsummands, dsummands, nflds, &
   729)                                 arr_max_shift, arr_gmax_exp, max_levels,    &
   730)                                 max_level, validate, recompute,             &
   731)                                 omp_nthreads, mpi_comm                      )
   732) !----------------------------------------------------------------------
   733) ! 
   734) ! Purpose: 
   735) ! Compute the global sum of each field in "arr" using the indicated 
   736) ! communicator with a reproducible yet scalable implementation based 
   737) ! on a fixed point algorithm. The accuracy of the fixed point algorithm
   738) ! is controlled by the number of "levels" of integer expansion, the
   739) ! maximum value of which is specified by max_level. 
   740) ! 
   741) !----------------------------------------------------------------------
   742) !
   743) ! Arguments
   744) !
   745)       integer,  intent(in) :: nsummands     ! number of local summands
   746)       integer,  intent(in) :: dsummands     ! declared first dimension
   747)       integer,  intent(in) :: nflds         ! number of fields
   748)       integer,  intent(in) :: arr_max_shift ! maximum safe exponent for 
   749)                                             !  value < 1 (so that sum 
   750)                                             !  does not overflow)
   751)       integer,  intent(in) :: arr_gmax_exp(nflds)  
   752)                                             ! exponents of global maxima
   753)       integer,  intent(in) :: max_levels(nflds) 
   754)                                             ! maximum number of levels 
   755)                                             !  of integer expansion
   756)       integer,  intent(in) :: max_level     ! maximum value in 
   757)                                             !  max_levels
   758)       integer,  intent(in) :: omp_nthreads  ! number of OpenMP threads
   759)       integer,  intent(in) :: mpi_comm      ! MPI subcommunicator
   760) 
   761)       real(r8), intent(in) :: arr(dsummands,nflds) 
   762)                                              ! input array
   763) 
   764)       logical,  intent(in):: validate
   765)          ! flag indicating that accuracy of solution generated from
   766)          ! arr_gmax_exp and max_levels should be tested
   767) 
   768)       logical,  intent(out):: recompute            
   769)          ! flag indicating that either the upper bounds are inaccurate,
   770)          !  or max_levels and arr_gmax_exp do not generate accurate 
   771)          !  enough sums
   772) 
   773)       real(r8), intent(out):: arr_gsum(nflds)      ! global means
   774) !
   775) ! Local workspace
   776) !
   777)       integer, parameter  :: max_jlevel = &
   778)                                 1 + (digits(0_i8)/digits(0.0_r8))
   779) 
   780)       integer(i8) :: i8_arr_tlsum_level(0:max_level,nflds,omp_nthreads) 
   781)                                    ! integer vector representing local 
   782)                                    !  sum (per thread, per field)
   783)       integer(i8) :: i8_arr_lsum_level((max_level+3)*nflds) 
   784)                                    ! integer vector representing local 
   785)                                    !  sum
   786)       integer(i8) :: i8_arr_level  ! integer part of summand for current
   787)                                    !  expansion level
   788)       integer(i8) :: i8_arr_gsum_level((max_level+3)*nflds) 
   789)                                    ! integer vector representing global
   790)                                    !  sum
   791)       integer(i8) :: IX_8          ! integer representation of current 
   792)                                    !  jlevels of X_8 ('part' of 
   793)                                    !  i8_arr_gsum_level)
   794)       integer(i8) :: i8_sign       ! sign global sum
   795)       integer(i8) :: i8_radix      ! radix for i8 variables
   796) 
   797)       integer :: max_error(nflds,omp_nthreads)  
   798)                                    ! accurate upper bound on data?
   799)       integer :: not_exact(nflds,omp_nthreads)  
   800)                                    ! max_levels sufficient to 
   801)                                    !  capture all digits?
   802)       integer :: isum_beg(omp_nthreads), isum_end(omp_nthreads)
   803)                                    ! range of summand indices for each
   804)                                    !  OpenMP thread
   805)       integer :: ifld, isum, ithread
   806)                                    ! loop variables
   807)       integer :: arr_exp           ! exponent of summand
   808)       integer :: arr_shift         ! exponent used to generate integer 
   809)                                    !  for current expansion level
   810)       integer :: ilevel            ! current integer expansion level
   811)       integer :: offset(nflds)     ! beginning location in 
   812)                                    !  i8_arr_{g,l}sum_level for integer 
   813)                                    !  expansion of current ifld
   814)       integer :: voffset           ! modification to offset used to 
   815)                                    !  include validation metrics 
   816)       integer :: ioffset           ! offset(ifld)
   817)       integer :: jlevel            ! number of floating point 'pieces' 
   818)                                    !  extracted from a given i8 integer
   819)       integer :: ierr              ! MPI error return
   820)       integer :: LX(max_jlevel)    ! exponent of X_8 (see below)
   821)       integer :: veclth            ! total length of i8_arr_lsum_level
   822)       integer :: sum_digits        ! lower bound on number of significant
   823)                                    !  in integer expansion of sum
   824)       integer :: curr_exp          ! exponent of partial sum during
   825)                                    !  reconstruction from integer vector
   826)       integer :: corr_exp          ! exponent of current summand in
   827)                                    !  reconstruction from integer vector
   828) 
   829)       real(r8) :: arr_frac         ! fraction of summand
   830)       real(r8) :: arr_remainder    ! part of summand remaining after 
   831)                                    !  current level of integer expansion
   832)       real(r8) :: X_8(max_jlevel)  ! r8 vector representation of current 
   833)                                    !  i8_arr_gsum_level
   834)       real(r8) :: RX_8             ! r8 representation of difference 
   835)                                    !  between current i8_arr_gsum_level
   836)                                    !  and current jlevels of X_8 
   837)                                    !  (== IX_8). Also used in final
   838)                                    !  scaling step
   839) 
   840)       logical :: first             ! flag used to indicate that just
   841)                                    !  beginning reconstruction of sum
   842)                                    !  from integer vector
   843) 
   844) !
   845) !-----------------------------------------------------------------------
   846) ! Save radix of i8 variables in an i8 variable
   847)       i8_radix = radix(IX_8)
   848) 
   849) ! If validating upper bounds, reserve space for validation metrics
   850) ! In both cases, reserve an extra level for overflow from the top level
   851)       if (validate) then
   852)         voffset = 3
   853)       else
   854)         voffset = 1
   855)       endif
   856) 
   857) ! compute offsets for each field
   858)       offset(1) = voffset
   859)       do ifld=2,nflds
   860)          offset(ifld) = offset(ifld-1) &
   861)                         + (max_levels(ifld-1) + voffset)
   862)       enddo
   863)       veclth = offset(nflds) + max_levels(nflds)
   864) 
   865) ! split summand index range over OpenMP threads
   866)       call split_indices(nsummands, omp_nthreads, isum_beg, isum_end)
   867) 
   868) ! convert local summands to vector of integers and sum
   869) ! (Using scale instead of set_exponent because arr_remainder may not be
   870) !  "normal" after level 1 calculation)
   871)       i8_arr_lsum_level(:) = 0_i8
   872) 
   873) !$omp parallel do      &
   874) !$omp default(shared)  &
   875) !$omp private(ithread, ifld, ioffset, isum, arr_frac, arr_exp, &
   876) !$omp         arr_shift, ilevel, i8_arr_level, arr_remainder, RX_8, IX_8)
   877)       do ithread=1,omp_nthreads
   878)        call t_startf('repro_sum_loopb')
   879)        do ifld=1,nflds
   880)           ioffset = offset(ifld)
   881) 
   882)           max_error(ifld,ithread) = 0
   883)           not_exact(ifld,ithread) = 0
   884) 
   885)           i8_arr_tlsum_level(:,ifld,ithread) = 0_i8
   886)           do isum=isum_beg(ithread),isum_end(ithread)
   887)             arr_remainder = 0.0_r8
   888) 
   889)             if (arr(isum,ifld) .ne. 0.0_r8) then
   890)                arr_exp   = exponent(arr(isum,ifld))
   891)                arr_frac  = fraction(arr(isum,ifld))
   892) 
   893) ! test that global maximum upper bound is an upper bound
   894)                if (arr_exp > arr_gmax_exp(ifld)) then
   895)                   max_error(ifld,ithread) = 1
   896)                   exit
   897)                endif
   898) 
   899) ! calculate first shift
   900)                arr_shift = arr_max_shift - (arr_gmax_exp(ifld)-arr_exp)
   901) 
   902) ! determine first (probably) nonzero level (assuming initial fraction is 
   903) !  'normal' - algorithm still works if this is not true)
   904) !  NOTE: this is critical; scale will set to zero if min exponent is too small.
   905)                if (arr_shift < 1) then
   906)                   ilevel = (1 + (arr_gmax_exp(ifld)-arr_exp))/arr_max_shift
   907)                   arr_shift = ilevel*arr_max_shift - (arr_gmax_exp(ifld)-arr_exp)
   908) 
   909)                   do while (arr_shift < 1)
   910)                      arr_shift = arr_shift + arr_max_shift
   911)                      ilevel = ilevel + 1
   912)                   enddo
   913)                else
   914)                   ilevel = 1
   915)                endif
   916) 
   917)                if (ilevel .le. max_levels(ifld)) then
   918) ! apply first shift/truncate, add it to the relevant running 
   919) ! sum, and calculate the remainder.
   920)                   arr_remainder = scale(arr_frac,arr_shift)
   921)                   i8_arr_level = int(arr_remainder,i8)
   922)                   i8_arr_tlsum_level(ilevel,ifld,ithread) = &
   923)                      i8_arr_tlsum_level(ilevel,ifld,ithread) + i8_arr_level
   924)                   arr_remainder = arr_remainder - i8_arr_level
   925) 
   926) ! while the remainder is non-zero, continue to shift, truncate, 
   927) ! sum, and calculate new remainder
   928)                   do while ((arr_remainder .ne. 0.0_r8) &
   929)                      .and. (ilevel < max_levels(ifld)))
   930)                      ilevel = ilevel + 1
   931)                      arr_remainder = scale(arr_remainder,arr_max_shift)
   932)                      i8_arr_level = int(arr_remainder,i8)
   933)                      i8_arr_tlsum_level(ilevel,ifld,ithread) = &
   934)                         i8_arr_tlsum_level(ilevel,ifld,ithread) + i8_arr_level
   935)                      arr_remainder = arr_remainder - i8_arr_level
   936)                   enddo
   937) 
   938)                endif
   939)             endif
   940) 
   941)             if (arr_remainder .ne. 0.0_r8) then
   942)                not_exact(ifld,ithread) = 1
   943)             endif
   944) 
   945)           enddo
   946) 
   947) ! postprocess integer vector to eliminate potential for overlap in the following
   948) ! sums over threads and processes: if value larger than or equal to 
   949) ! (radix(IX_8)**arr_max_shift), add this 'overlap' to next larger integer in 
   950) ! vector, resulting in nonoverlapping ranges for each component. Note that 
   951) ! "ilevel-1==0" corresponds to an extra level used to guarantee that the sums
   952) ! over threads and processes do not overflow for ilevel==1.
   953)           do ilevel=max_levels(ifld),1,-1
   954)              RX_8 = i8_arr_tlsum_level(ilevel,ifld,ithread)
   955)              IX_8 = int(scale(RX_8,-arr_max_shift),i8)
   956)              if (IX_8 .ne. 0_i8) then
   957)                 i8_arr_tlsum_level(ilevel-1,ifld,ithread) = &
   958)                    i8_arr_tlsum_level(ilevel-1,ifld,ithread) + IX_8
   959)                 IX_8 = IX_8*(i8_radix**arr_max_shift)
   960)                 i8_arr_tlsum_level(ilevel,ifld,ithread)   = &
   961)                    i8_arr_tlsum_level(ilevel,ifld,ithread) - IX_8
   962)              endif
   963)           enddo
   964)        enddo
   965)        call t_stopf('repro_sum_loopb')
   966)       enddo
   967) 
   968) ! sum contributions from different threads
   969)       do ifld=1,nflds
   970)          ioffset = offset(ifld)
   971)          do ithread = 1,omp_nthreads
   972)             do ilevel = 0,max_levels(ifld)
   973)                i8_arr_lsum_level(ioffset+ilevel) = &
   974)                   i8_arr_lsum_level(ioffset+ilevel) &
   975)                   + i8_arr_tlsum_level(ilevel,ifld,ithread)
   976)             enddo
   977)          enddo
   978)       enddo
   979) 
   980) ! record if upper bound was inaccurate or if level expansion stopped
   981) ! before full accuracy was achieved
   982)       if (validate) then
   983)          do ifld=1,nflds
   984)             ioffset = offset(ifld)
   985)             i8_arr_lsum_level(ioffset-voffset+1) = maxval(max_error(ifld,:))
   986)             i8_arr_lsum_level(ioffset-voffset+2) = maxval(not_exact(ifld,:))
   987)          enddo
   988)       endif
   989) 
   990) ! sum integer vector element-wise
   991) #if ( defined noI8 )
   992)      ! Workaround for when shr_kind_i8 is not supported.
   993)       call t_startf("repro_sum_allr_i4")
   994)       call mpi_allreduce (i8_arr_lsum_level, i8_arr_gsum_level, & 
   995)                           veclth, MPI_INTEGER, MPI_SUM, mpi_comm, ierr)
   996)       call t_stopf("repro_sum_allr_i4")
   997) #else
   998)       call t_startf("repro_sum_allr_i8")
   999)       call mpi_allreduce (i8_arr_lsum_level, i8_arr_gsum_level, & 
  1000)                           veclth, MPI_INTEGER8, MPI_SUM, mpi_comm, ierr)
  1001)       call t_stopf("repro_sum_allr_i8")
  1002) #endif
  1003) 
  1004) ! Construct global sum from integer vector representation:
  1005) !  1) arr_max_shift is the shift applied to fraction(arr_gmax) .
  1006) !   When shifting back, need to "add back in" true arr_gmax exponent. This was
  1007) !   removed implicitly by working only with the fraction .
  1008) !  2) want to add levels into sum in reverse order (smallest to largest). However,
  1009) !   even this can generate floating point rounding errors if signs of integers
  1010) !   alternate. To avoid this, do some arithmetic with integer vectors so that all
  1011) !   components have the same sign. This should keep relative difference between
  1012) !   using different integer sizes (e.g. i8 and i4) to machine epsilon
  1013) !  3) assignment to X_8 will usually lose accuracy since maximum integer 
  1014) !   size is greater than the max number of 'digits' in r8 value (if xmax_nsummands 
  1015) !   correction not very large). Calculate remainder and add in first (since 
  1016) !   smaller). One correction is sufficient for r8 (53 digits) and i8 (63 digits). 
  1017) !   For r4 (24 digits) may need to correct twice. Code is written in a general
  1018) !   fashion, to work no matter how many corrections are necessary (assuming
  1019) !   max_jlevel parameter calculation is correct).
