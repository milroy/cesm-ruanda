   440)   subroutine seq_drydep_read(NLFilename, seq_drydep_fields)
   441) 
   442)     !========================================================================
   443)     ! reads drydep_inparm namelist and sets up CCSM driver list of fields for 
   444)     ! land-atmosphere communications.
   445)     ! 
   446)     ! !REVISION HISTORY:
   447)     !  2009-Feb-20 - E. Kluzek - Separate out as subroutine from previous input_init
   448)     !========================================================================
   449)     
   450)     use shr_file_mod,only : shr_file_getUnit, shr_file_freeUnit
   451)     use shr_log_mod, only : s_logunit => shr_log_Unit
   452) 
   453)     implicit none
   454) 
   455)     character(len=*), intent(in)  :: NLFilename ! Namelist filename
   456)     character(len=*), intent(out) :: seq_drydep_fields	
   457) 
   458)     !----- local -----
   459)     integer :: i                ! Indices
   460)     integer :: unitn            ! namelist unit number
   461)     integer :: ierr             ! error code
   462)     logical :: exists           ! if file exists or not
   463)     character(len=8) :: token   ! dry dep field name to add
   464) 
   465)     !----- formats -----
   466)     character(*),parameter :: subName = '(seq_drydep_read) ' 
   467)     character(*),parameter :: F00   = "('(seq_drydep_read) ',8a)" 
   468)     character(*),parameter :: FI1   = "('(seq_drydep_init) ',a,I2)" 
   469) 
   470)     namelist /drydep_inparm/ drydep_list, drydep_method
   471) 
   472)     !-----------------------------------------------------------------------------
   473)     ! Read namelist and figure out the drydep field list to pass
   474)     ! First check if file exists and if not, n_drydep will be zero
   475)     !-----------------------------------------------------------------------------
   476)  
   477)     !--- Open and read namelist ---
   478)     if ( len_trim(NLFilename) == 0  )then
   479)        call shr_sys_abort( subName//'ERROR: nlfilename not set' )
   480)     end if
   481)     inquire( file=trim(NLFileName), exist=exists)
   482)     if ( exists ) then
   483)        unitn = shr_file_getUnit()
   484)        open( unitn, file=trim(NLFilename), status='old' )
   485)        if ( s_loglev > 0 ) write(s_logunit,F00) &
   486)             'Read in drydep_inparm namelist from: ', trim(NLFilename)
   487)        ierr = 1
   488)        do while ( ierr /= 0 )
   489)           read(unitn, drydep_inparm, iostat=ierr)
   490)           if (ierr < 0) then
   491)              call shr_sys_abort( subName//'ERROR: encountered end-of-file on namelist read' )
   492)           endif
   493)        end do
   494)        close( unitn )
   495)        call shr_file_freeUnit( unitn )
   496)     end if
   497) 
   498)     n_drydep = 0
   499) 
   500)     !--- Loop over species to fill list of fields to communicate for drydep ---
   501)     seq_drydep_fields = ' '	
   502)     do i=1,maxspc
   503)        if ( len_trim(drydep_list(i))==0 ) exit
   504)        write(token,333) i
   505)        seq_drydep_fields = trim(seq_drydep_fields)//':'//trim(token)                 
   506)        if ( i == 1 ) then
   507)           seq_drydep_fields = trim(token)                 
   508)           drydep_fields_token = trim(token)
   509)        endif
   510)        n_drydep = n_drydep+1
   511)     enddo
   512) 
   513)     !--- Make sure method is valid and determine if land is passing drydep fields ---
   514)     lnd_drydep = n_drydep>0 .and. drydep_method == DD_XLND
   515) 
   516)     if ( s_loglev > 0 ) then
   517)        write(s_logunit,*) 'seq_drydep_read: drydep_method: ', trim(drydep_method)
   518)        if ( n_drydep == 0 )then
   519)           write(s_logunit,F00) 'No dry deposition fields will be transfered'
   520)        else
   521)           write(s_logunit,FI1) 'Number of dry deposition fields transfered is ', &
   522)                                 n_drydep
   523)        end if
   524)     end if
   525)     
   526)     if ( trim(drydep_method)/=trim(DD_XATM) .and. &
   527)          trim(drydep_method)/=trim(DD_XLND) .and. &
   528)          trim(drydep_method)/=trim(DD_TABL) ) then
   529)        if ( s_loglev > 0 ) then
   530)           write(s_logunit,*) 'seq_drydep_read: drydep_method : ', trim(drydep_method)
   531)           write(s_logunit,*) 'seq_drydep_read: drydep_method must be set to : ', &
   532)                DD_XATM,', ', DD_XLND,', or ', DD_TABL
   533)        end if
   534)        call shr_sys_abort('seq_drydep_read: incorrect dry deposition method specification')
   535)     endif
   536) 
   537)     ! Need to explicitly add Sl_ based on naming convention
   538) 333 format ('Sl_dd',i3.3)
   539) 
   540)   end subroutine seq_drydep_read

   544)   subroutine seq_drydep_init( )
   545) 
   546)     !========================================================================
   547)     ! Initialization of dry deposition fields
   548)     ! reads drydep_inparm namelist and sets up CCSM driver list of fields for 
   549)     ! land-atmosphere communications.
   550)     ! !REVISION HISTORY:
   551)     !  2008-Nov-12 - F. Vitt - first version
   552)     !  2009-Feb-20 - E. Kluzek - Check for existance of file if not return, set n_drydep=0
   553)     !  2009-Feb-20 - E. Kluzek - Move namelist read to separate subroutine
   554)     !========================================================================
   555) 
   556)     use shr_log_mod, only : s_logunit => shr_log_Unit
   557)     use shr_infnan_mod, only: shr_infnan_posinf, assignment(=)
   558) 
   559)     implicit none
   560) 
   561)     !----- local -----
   562)     integer :: i, l                      ! Indices
   563)     character(len=32) :: test_name       ! field test name
   564)     !----- formats -----
   565)     character(*),parameter :: subName = '(seq_drydep_init) ' 
   566)     character(*),parameter :: F00   = "('(seq_drydep_init) ',8a)" 
   567) 
   568)     !-----------------------------------------------------------------------------
   569)     ! Allocate and fill foxd, drat and mapping as well as species indices
   570)     !-----------------------------------------------------------------------------
   571) 
   572)     if ( n_drydep > 0 ) then
   573) 
   574)        allocate( foxd(n_drydep) )
   575)        allocate( drat(n_drydep) )
   576)        allocate( mapping(n_drydep) )
   577) 
   578)        ! This initializes these variables to infinity.
   579)        foxd = shr_infnan_posinf
   580)        drat = shr_infnan_posinf
   581) 
   582)        mapping(:) = 0
   583) 
   584)     end if
   585) 
   586)     h2_ndx=-1; ch4_ndx=-1; co_ndx=-1; mpan_ndx = -1; pan_ndx = -1; so2_ndx=-1; o3_ndx=-1; xpan_ndx=-1
   587) 
   588)     !--- Loop over drydep species that need to be worked with ---
   589)     do i=1,n_drydep
   590)        if ( len_trim(drydep_list(i))==0 ) exit
   591) 
   592)        test_name = drydep_list(i)
   593) 
   594)        if( trim(test_name) == 'O3' ) then
   595)           test_name = 'OX'
   596)        end if
   597) 
   598)        !--- Figure out if species maps to a species in the species table ---
   599)        do l = 1,n_species_table
   600)           if(  trim( test_name ) == trim( species_name_table(l) ) ) then
   601)              mapping(i)  = l
   602)              exit
   603)           end if
   604)        end do
   605) 
   606)        !--- If it doesn't map to a species in the species table find species close enough ---
   607)        if( mapping(i) < 1 ) then
   608)           select case( trim(test_name) )
   609)           case( 'H2' )
   610)              test_name = 'CO'
   611)           case( 'HYAC', 'CH3COOH', 'EOOH' )
   612)              test_name = 'CH2O'
   613)           case( 'O3S', 'O3INERT', 'MPAN' )
   614)              test_name = 'OX'
   615)           case( 'ISOPOOH', 'MACROOH', 'Pb', 'XOOH', 'H2SO4' )
   616)              test_name = 'HNO3'
   617)           case( 'ALKOOH', 'MEKOOH', 'TOLOOH', 'BENOOH', 'XYLOOH', 'TERPOOH','SOGM','SOGI','SOGT','SOGB','SOGX' )
   618)                 test_name = 'CH3OOH'
   619)           case( 'SOA', 'SO4', 'CB1', 'CB2', 'OC1', 'OC2', 'NH3', 'NH4', 'SA1', 'SA2', 'SA3', 'SA4','HCN','CH3CN','HCOOH' )
   620)              test_name = 'OX'  ! this is just a place holder. values are explicitly set below
   621)           case( 'SOAM', 'SOAI', 'SOAT', 'SOAB', 'SOAX' )
   622)              test_name = 'OX'  ! this is just a place holder. values are explicitly set below
   623)           case( 'O3A', 'XMPAN' )
   624)              test_name = 'OX'
   625)           case( 'XPAN' )
   626)              test_name = 'PAN'
   627)           case( 'XNO' )
   628)              test_name = 'NO'
   629)           case( 'XNO2' )
   630)              test_name = 'NO2'
   631)           case( 'XHNO3' )
   632)              test_name = 'HNO3'
   633)           case( 'XONIT' )
   634)              test_name = 'ONIT'
   635)           case( 'XONITR' )
   636)              test_name = 'ONITR'
   637)           case( 'XHO2NO2')
   638)              test_name = 'HO2NO2'
   639)           case( 'XNH4NO3' )
   640)              test_name = 'HNO3'
   641) 
   642)           case( 'COhc','COme')
   643)                 test_name = 'CO'  ! this is just a place holder. values are set in drydep_fromlnd
   644)           case( 'CO01','CO02','CO03','CO04','CO05','CO06','CO07','CO08','CO09','CO10' )
   645)                 test_name = 'CO'  ! this is just a place holder. values are set in drydep_fromlnd
   646)           case( 'CO11','CO12','CO13','CO14','CO15','CO16','CO17','CO18','CO19','CO20' )
   647)                 test_name = 'CO'  ! this is just a place holder. values are set in drydep_fromlnd
   648)           case( 'CO21','CO22','CO23','CO24','CO25','CO26','CO27','CO28','CO29','CO30' )
   649)                 test_name = 'CO'  ! this is just a place holder. values are set in drydep_fromlnd
   650)           case( 'CO31','CO32','CO33','CO34','CO35','CO36','CO37','CO38','CO39','CO40' )
   651)                 test_name = 'CO'  ! this is just a place holder. values are set in drydep_fromlnd
   652)           case( 'CO41','CO42','CO43','CO44','CO45','CO46','CO47','CO48','CO49','CO50' )
   653)                 test_name = 'CO'  ! this is just a place holder. values are set in drydep_fromlnd
   654) 
   655)           case( 'NH4NO3' )
   656)              test_name = 'HNO3'
   657)           case default
   658)              test_name = 'blank'
   659)           end select
   660) 
   661)           !--- If found a match check the species table again ---
   662)           if( trim(test_name) /= 'blank' ) then
   663)              do l = 1,n_species_table
   664)                 if( trim( test_name ) == trim( species_name_table(l) ) ) then
   665)                    mapping(i)  = l
   666)                    exit
   667)                 end if
   668)              end do
   669)           else
   670)              if ( s_loglev > 0 ) write(s_logunit,F00) trim(drydep_list(i)), &
   671)                                  ' not in tables; will have dep vel = 0'
   672)              call shr_sys_abort( subName//': '//trim(drydep_list(i))//' is not in tables' )
   673)           end if
   674)        end if
   675) 
   676)        !--- Figure out the specific species indices ---
   677)        if ( trim(drydep_list(i)) == 'H2' )   h2_ndx   = i
   678)        if ( trim(drydep_list(i)) == 'CO' )   co_ndx   = i
   679)        if ( trim(drydep_list(i)) == 'CH4' )  ch4_ndx  = i
   680)        if ( trim(drydep_list(i)) == 'MPAN' ) mpan_ndx = i
   681)        if ( trim(drydep_list(i)) == 'PAN' )  pan_ndx  = i
   682)        if ( trim(drydep_list(i)) == 'SO2' )  so2_ndx  = i
   683)        if ( trim(drydep_list(i)) == 'OX' .or. trim(drydep_list(i)) == 'O3' ) o3_ndx  = i 
   684)        if ( trim(drydep_list(i)) == 'O3A' ) o3a_ndx  = i 
   685)        if ( trim(drydep_list(i)) == 'XPAN' ) xpan_ndx = i
   686) 
   687)        if( mapping(i) > 0) then
   688)          l = mapping(i)
   689)          foxd(i) = dfoxd(l)
   690)          drat(i) = sqrt(mol_wgts(l)/wh2o)
   691)        endif
   692) 
   693)     enddo
   694) 
   695)     where( rgss < 1._r8 )
   696)        rgss = 1._r8
   697)     endwhere
   698)

   733)   subroutine set_hcoeff_vector( ncol, sfc_temp, heff )
   734) 
   735)     !========================================================================
   736)     ! Interface to seq_drydep_setHCoeff when input is vector
   737)     ! sets dry depositions coefficients -- used by both land and atmosphere models
   738)     ! !REVISION HISTORY:
   739)     !  2008-Nov-12 - F. Vitt - first version
   740)     !========================================================================
   741)     
   742)     use shr_log_mod, only : s_logunit => shr_log_Unit
   743) 
   744)     implicit none
   745) 
   746)     integer, intent(in)      :: ncol                  ! Input size of surface-temp vector
   747)     real(r8), intent(in)     :: sfc_temp(ncol)        ! Surface temperature
   748)     real(r8), intent(out)    :: heff(ncol,n_drydep)   ! Henry's law coefficients
   749) 
   750)     !----- local -----
   751)     real(r8), parameter :: t0     = 298._r8    ! Standard Temperature
   752)     real(r8), parameter :: ph_inv = 1._r8/ph   ! Inverse of PH
   753)     integer  :: m, l, id       ! indices
   754)     real(r8) :: e298           ! Henry's law coefficient @ standard temperature (298K)
   755)     real(r8) :: dhr            ! temperature dependence of Henry's law coefficient
   756)     real(r8) :: dk1s(ncol)     ! DK Work array 1
   757)     real(r8) :: dk2s(ncol)     ! DK Work array 2
   758)     real(r8) :: wrk(ncol)      ! Work array
   759) 
   760)     !----- formats -----
   761)     character(*),parameter :: subName = '(seq_drydep_set_hcoeff) ' 
   762)     character(*),parameter :: F00   = "('(seq_drydep_set_hcoeff) ',8a)" 
   763) 
   764)     !-------------------------------------------------------------------------------
   765)     ! notes: 
   766)     !-------------------------------------------------------------------------------
   767) 
   768)     wrk(:) = (t0 - sfc_temp(:))/(t0*sfc_temp(:))
   769)     do m = 1,n_drydep
   770)        l    = mapping(m)
   771)        id   = 6*(l - 1)
   772)        e298 = dheff(id+1)
   773)        dhr  = dheff(id+2)
   774)        heff(:,m) = e298*exp( dhr*wrk(:) )
   775)        !--- Calculate coefficients based on the drydep tables ---
   776)        if( dheff(id+3) /= 0._r8 .and. dheff(id+5) == 0._r8 ) then
   777)           e298 = dheff(id+3)
   778)           dhr  = dheff(id+4)
   779)           dk1s(:) = e298*exp( dhr*wrk(:) )
   780)           where( heff(:,m) /= 0._r8 )
   781)              heff(:,m) = heff(:,m)*(1._r8 + dk1s(:)*ph_inv)
   782)           elsewhere
   783)              heff(:,m) = dk1s(:)*ph_inv
   784)           endwhere
   785)        end if
   786)        !--- For coefficients that are non-zero AND CO2 or NH3 handle things this way ---
   787)        if( dheff(id+5) /= 0._r8 ) then
   788)           if( trim( drydep_list(m) ) == 'CO2' .or. trim( drydep_list(m) ) == 'NH3' ) then
   789)              e298 = dheff(id+3)
   790)              dhr  = dheff(id+4)
   791)              dk1s(:) = e298*exp( dhr*wrk(:) )
   792)              e298 = dheff(id+5)
   793)              dhr  = dheff(id+6)
   794)              dk2s(:) = e298*exp( dhr*wrk(:) )
   795)              !--- For Carbon dioxide ---
   796)              if( trim(drydep_list(m)) == 'CO2' ) then
   797)                 heff(:,m) = heff(:,m)*(1._r8 + dk1s(:)*ph_inv)*(1._r8 + dk2s(:)*ph_inv)
   798)              !--- For NH3 ---
   799)              else if( trim( drydep_list(m) ) == 'NH3' ) then
   800)                 heff(:,m) = heff(:,m)*(1._r8 + dk1s(:)*ph/dk2s(:))
   801)              !--- This can't happen ---
   802)              else
   803)                 write(s_logunit,F00) 'Bad species ',drydep_list(m)
   804)                 call shr_sys_abort( subName//'ERROR: in assigning coefficients' )
   805)              end if
   806)           end if
   807)        end if
   808)     end do
   809) 
   810)   end subroutine set_hcoeff_vector
