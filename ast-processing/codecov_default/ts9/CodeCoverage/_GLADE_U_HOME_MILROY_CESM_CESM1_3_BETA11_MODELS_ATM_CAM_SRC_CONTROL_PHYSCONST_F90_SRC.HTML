   237)   subroutine physconst_update(mmr, t, cnst_mw_o, cnst_mw_o2, cnst_mw_h, cnst_mw_n, ixo, ixo2, ixh, ncnst, lchnk, ncol)
   238)   
   239) !-----------------------------------------------------------------------
   240) ! Update the physics "constants" that vary
   241) !-----------------------------------------------------------------------
   242) 
   243) !------------------------------Arguments--------------------------------------------------------------
   244) 
   245)     integer, intent(in)  :: ncnst           ! number of constituents in mmr array
   246)     real(r8), intent(in) :: mmr(pcols,pver,ncnst) ! constituents q array from state structure
   247)     real(r8), intent(in) :: t(pcols,pver)   ! temperature t array from state structure
   248) 
   249)     real(r8), intent(in) :: cnst_mw_o, cnst_mw_o2, cnst_mw_h, cnst_mw_n  ! Molecular weight of O, O2, H, and N
   250) 
   251)     integer, intent(in)  :: ixo, ixo2, ixh  ! indices for O, O2, and H
   252)     integer, intent(in)  :: lchnk           ! Chunk number
   253)     integer, intent(in)  :: ncol            ! number of columns
   254) !
   255) !---------------------------Local storage-------------------------------------------------------------
   256)     integer :: i,k                                 ! column,level,constituent indices
   257) 
   258)     real(r8):: mmro, mmro2, mmrh, mmrn2            ! Mass mixing ratios of O, O2, H, and N
   259)     real(r8):: mbarvi, tint                        ! Mean mass, temperature, and specific heat on interface levels
   260)     real(r8):: dof1, dof2                          ! Degress of freedom for cpairv calculation
   261)     real(r8):: kv1, kv2, kv3, kv4                  ! Coefficients for kmvis calculation
   262)     real(r8):: kc1, kc2, kc3, kc4                  ! Coefficients for kmcnd calculation
   263)     !--------------------------------------------
   264)     ! Set constants needed for updates
   265)     !--------------------------------------------
   266)     dof1 = 5._r8 
   267)     dof2 = 7._r8
   268)     kv1  = 4.03_r8
   269)     kv2  = 3.42_r8
   270)     kv3  = 3.9_r8
   271)     kv4  = 0.69_r8
   272)     kc1  = 56._r8
   273)     kc2  = 56._r8
   274)     kc3  = 75.9_r8
   275)     kc4  = 0.69_r8
   276) 
   277)      !--------------------------------------------
   278)      ! update cpairv, rairv, mbarv, and cappav
   279)      !--------------------------------------------
   280)      do k=1,pver
   281)         do i=1,ncol
   282)            mmro = mmr(i,k, ixo)
   283)            mmro2 = mmr(i,k, ixo2)
   284)            mmrh = mmr(i,k, ixh)
   285)            mmrn2 = 1._r8-mmro-mmro2-mmrh
   286)            mbarv(i,k,lchnk) = 1._r8/(mmro/cnst_mw_o +      &
   287)                                      mmro2/cnst_mw_o2 +    &
   288)                                      mmrn2/cnst_mw_n/2._r8 + &
   289)                                      mmrh/cnst_mw_h)
   290)            rairv(i,k,lchnk) = shr_const_rgas / mbarv(i,k,lchnk)
   291)            cpairv(i,k,lchnk) = 0.5_r8*shr_const_rgas*    &
   292)                                  (dof1*mmro/cnst_mw_o+        &
   293)                                   dof2*mmro2/cnst_mw_o2+     &
   294)                                   dof2*mmrn2/cnst_mw_n/2._r8+ &
   295)                                      dof1*mmrh/cnst_mw_h)
   296)            cappav(i,k,lchnk) = rairv(i,k,lchnk)/cpairv(i,k,lchnk)
   297)         enddo
   298)      enddo
   299) 
   300)      do k=2,pver
   301)         do i=1,ncol
   302)            mmro = .5_r8*(mmr(i,k-1, ixo)+mmr(i,k,ixo))
   303)            mmro2 = .5_r8*(mmr(i,k-1, ixo2)+mmr(i,k,ixo2))
   304)            mmrn2 = 1._r8-mmro-mmro2
   305)            mbarvi = .5_r8*(mbarv(i,k-1,lchnk)+mbarv(i,k,lchnk))
   306)            tint = .5_r8*(t(i,k-1)+t(i,k))
   307)  
   308)            kmvis(i,k,lchnk) = (kv1*mmro2/cnst_mw_o2+             &
   309)                                kv2*mmrn2/cnst_mw_n/2._r8+        &
   310)                                kv3*mmro/cnst_mw_o)*mbarvi*       &
   311)                                tint**kv4 * 1.e-7_r8
   312)            kmcnd(i,k,lchnk) = (kc1*mmro2/cnst_mw_o2+             &
   313)                                kc2*mmrn2/cnst_mw_n/2._r8+        &
   314)                                kc3*mmro/cnst_mw_o)*mbarvi*   &
   315)                                tint**kc4 * 1.e-5_r8
   316)         enddo
   317)      enddo
   318)      do i=1,ncol
   319)         kmvis(i,1,lchnk) = 1.5_r8*kmvis(i,2,lchnk)-.5_r8*kmvis(i,3,lchnk)
   320)         kmcnd(i,1,lchnk) = 1.5_r8*kmcnd(i,2,lchnk)-.5_r8*kmcnd(i,3,lchnk)
   321)         kmvis(i,pverp,lchnk) = kmvis(i,pver,lchnk)
   322)         kmcnd(i,pverp,lchnk) = kmcnd(i,pver,lchnk)
   323)      enddo
   324) 
   325)    end subroutine physconst_update
