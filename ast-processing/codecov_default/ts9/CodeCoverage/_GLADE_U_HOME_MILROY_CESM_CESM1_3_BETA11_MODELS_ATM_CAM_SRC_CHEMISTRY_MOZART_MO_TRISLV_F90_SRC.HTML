     8)       subroutine tridec( syscnt, order, lower, main, upper )
     9) !--------------------------------------------------------------------
    10) ! dimension of           lower(syscnt,order), main(syscnt,order), upper(syscnt,order)
    11) ! arguments
    12) !
    13) ! latest revision        june 1995
    14) !
    15) ! purpose                trdi computes the solution of the tridiagonal
    16) !                        linear system,
    17) !                            b(1)*x(1)+c(1)*x(2)               = y(1)
    18) !                            a(i)*x(i-1)+b(i)*x(i)+c(i)*x(i+1) = y(i)
    19) !                                i=2,3,...,n-1
    20) !                            a(n)*x(n-1)+b(n)*x(n)             = y(n)
    21) !
    22) ! usage                  call trdi (n, a, b, c, x )
    23) !
    24) ! arguments
    25) !
    26) ! on input               n
    27) !                          the number of unknowns.  this subroutine
    28) !                          requires that  n  be greater than  2.
    29) !
    30) !                        a
    31) !                          the subdiagonal of the matrix is stored in
    32) !                          locations a(2) through a(n).
    33) !
    34) !                        b
    35) !                          the diagonal of the matrix is stored in
    36) !                          locations b(1) through b(n).
    37) !
    38) !                        c
    39) !                          the super-diagonal of the matrix is stored in
    40) !                          locations c(1) through c(n-1).
    41) !
    42) !                        x
    43) !                          the right-hand side of the equations is
    44) !                          stored in y(1) through y(n).
    45) !
    46) ! on output              x
    47) !                          an array which contains the solution to the
    48) !                          system of equations.
    49) !
    50) ! special conditions     this subroutine executes satisfactorily
    51) !                        if the input matrix is diagonally dominant
    52) !                        and non-singular.  the diagonal elements
    53) !                        are used to pivot, and no tests are made to
    54) !                        determine singularity.  if a singular, or
    55) !                        nearly singular, matrix is used as input,
    56) !                        a divide by zero or floating point overflow
    57) !                        may result.
    58) !
    59) ! precision              compiler dependent
    60) !
    61) ! language               fortran
    62) !
    63) ! history                originally written by nancy werner at ncar
    64) !                        in october, 1973. modified by s. walters at
    65) !                        ncar in june 1989 to functionally replace
    66) !                        the cal routine tridla.
    67) !
    68) ! portability            fortran 90
    69) !
    70) ! algorithm              an lu-decomposition is obtained using the
    71) !                        algorithm described in the reference below.
    72) !
    73) !                        to avoid unnecessary divisions, the alpha
    74) !                        values used in the routine are the
    75) !                        reciprocals of the alpha values described
    76) !                        in the reference below.
    77) !
    78) ! accuracy               every component of the residual of the linear
    79) !                        system (i.e. the difference between  y  and
    80) !                        the matrix applied to x) should be less in
    81) !                        magnitude than ten times the machine precision
    82) !                        times the matrix order times the maximum
    83) !                        absolute component of the solution vector
    84) !                        times the largest absolute row sum of the
    85) !                        input matrix.
    86) !
    87) ! timing                 the timing is roughly proportional to the
    88) !                        order n of the linear system.
    89) !
    90) ! references             analysis of numerical methods by
    91) !                        e. isaacson and h. keller
    92) !                        (john wiley and sons, 1966) pp. 55-58.

   100)       integer, intent(in) ::  syscnt, order

   109) !----------------------------------------------------------------------
   110) !     	... lu-decomposition

   112)       main(:,1)  = 1._r8 / main(:,1)
   113)       upper(:,1) = upper(:,1)*main(:,1)

   116)          upper(:,i) = upper(:,i)*main(:,i)

   121)       subroutine trislv( syscnt, order, lower, main, upper, x )
   122) 
   123)       implicit none
   124) 
   125) !----------------------------------------------------------------------
   126) !     	... dummy args
   127) !----------------------------------------------------------------------
   128)       integer, intent(in) ::  syscnt, order
   129)       real(r8), intent(in), dimension(syscnt,order)    ::  lower, &
   130)                                                        main, &
   131)                                                        upper
   132)       real(r8), intent(inout), dimension(syscnt,order) ::  x
   133)       
   134) !----------------------------------------------------------------------
   135) !     	... local variables
   136) !----------------------------------------------------------------------
   137)       integer :: i, im1, j, n, nm1
   138) 
   139)       nm1 = order - 1
   140)       n   = order
   141) !----------------------------------------------------------------------
   142) !     	... solve the system
   143) !----------------------------------------------------------------------
   144)       x(:,1) = x(:,1)*main(:,1)
   145)       do i = 2,nm1
   146)          x(:,i) = (x(:,i) - lower(:,i)*x(:,i-1))*main(:,i)
   147)       end do
   148) 
   149)       x(:,n) = (x(:,n) - lower(:,n)*x(:,nm1))/(main(:,n) - lower(:,n)*upper(:,nm1))
   150)       do i = nm1,1,-1
   151)          x(:,i) = x(:,i) - upper(:,i)*x(:,i+1)
   152)       end do
   153) 
   154)       end subroutine trislv
