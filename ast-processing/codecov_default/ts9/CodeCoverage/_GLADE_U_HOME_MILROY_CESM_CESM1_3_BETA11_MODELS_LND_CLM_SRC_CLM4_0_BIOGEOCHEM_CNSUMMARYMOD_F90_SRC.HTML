   952)          prod10c(c) + &

   956)       totecosysc(c) = &
   957)          cwdc(c) + &
   958)          totlitc(c) + &
   959)          totsomc(c) + &
   960) 	      totprodc(c) + &
   961)          col_totvegc(c)
   962) 
   963)       ! total column carbon, including veg and cpool (TOTCOLC)

   969)          totsomc(c) + &
   970) 	      totprodc(c) + &
   971) 		   seedc(c) + &
   972) 		   col_ctrunc(c)
   973) 
   974)       ! new summary variables for CLAMP
   975)       
   976)       ! (CWDC_HR) - coarse woody debris heterotrophic respiration
   977)       cwdc_hr(c) = 0._r8
   978)

   980)       cwdc_loss(c) = & 
   981)         m_cwdc_to_fire(c) + &
   982)         cwdc_to_litr2c(c) + &
   983)         cwdc_to_litr3c(c)
   984)       
   985)       ! (LITTERC_LOSS) - litter C loss
   986)       litterc_loss(c) = &
   987)         lithr(c)            + &
   988)         m_litr1c_to_fire(c) + &
   989)         m_litr2c_to_fire(c) + &
   990)         m_litr3c_to_fire(c) + &
   991)         litr1c_to_soil1c(c) + &
   992)         litr2c_to_soil2c(c) + &
   993)         litr3c_to_soil3c(c)
   994)       
   995)    end do ! end of columns loop
   996) 
   997) 
   998) end subroutine CSummary
