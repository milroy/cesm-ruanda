   307)   subroutine check_file_type(File, filename) 
   308) #ifndef NO_MPIMOD
   309)     use mpi !_EXTERNAL
   310) #else
   311)     include 'mpif.h'      ! _EXTERNAL
   312) #endif
   313) 
   314)     type (File_desc_t), intent(inout) :: File
   315)     character(len=*), intent(in) :: filename
   316)     character(len=4) :: magic
   317)     integer :: fh, mpierr, reclength=4, i, eof 
   318)     logical :: UNITOK, UNITOP
   319) 
   320) !   Check format of existing files opened to read.
   321) 
   322)     inquire(file=filename, exist=UNITOK) 
   323)     if(.not. UNITOK) return
   324) 
   325)     magic='fail'
   326)    
   327)     if(File%iosystem%ioproc) then      
   328)        if(File%iosystem%io_rank==0) then
   329) !  Find a unique unit number to open the file
   330)           do fh=12,99
   331)              inquire (unit=fh,exist=UNITOK,opened=UNITOP)
   332)              if (UNITOK .and. .not. UNITOP) then
   333)                 open (unit = fh,File=filename,access='direct',recl=reclength,&
   334)                      FORM='UNFORMATTED',STATUS='OLD',err=100)
   335) ! Read the first 4 bytes and look for the CDF or HDF stamp
   336)                 read (fh,rec=1,err=101) magic
   337)                 close(fh)
   338)                 exit
   339)              endif
   340)           end do
   341)           if(magic(1:3) .eq. 'CDF') then
   342)              ! No need to do anything here
   343)           else if(magic(2:4).eq.'HDF') then
   344) #ifdef _NETCDF4
   345)              if(File%iotype /= PIO_IOTYPE_NETCDF4C .and. &
   346)                   File%iotype /= PIO_IOTYPE_NETCDF4P) then
   347)                 print *,'Changing file type to netcdf4p'
   348)                 File%iotype=pio_iotype_netcdf4c
   349)              end if
   350) #else
   351)              call piodie(__PIO_FILE__,__LINE__,'You must link with the netcdf4 ',0,&
   352)                   'library built with hdf5 support to read this file',0,filename)
   353) #endif       
   354)           else 
   355)              ! The HDF identifier could be offset further into the file.
   356)              
   357)              open (unit = fh,file=filename,access='direct',recl=reclength,&
   358)                   form='UNFORMATTED',STATUS='OLD',err=100)
   359) 
   360)              i=128
   361)              eof=0
   362)              do while(eof>=0)
   363)                 read (fh,rec=i, iostat=eof, err=101) magic
   364) 
   365)                 if(magic(2:4).eq.'HDF') then
   366)                    if(debug) print *,'Changing file type to netcdf4p'
   367)                    File%iotype=pio_iotype_netcdf4p
   368)                    exit
   369)                 end if
   370)                 i=i*2
   371)              end do
   372)              close(fh)
   373)              if(eof<0) call piodie(__PIO_FILE__,__LINE__,'Unrecognized file format ',0,filename)             
   374)           end if
   375) 
   376)        end if
   377)        
   378)        call mpi_bcast(file%iotype,1,mpi_integer, 0, file%iosystem%io_comm, mpierr)
   379)        call CheckMPIReturn('nf_mod',mpierr)
   380)     end if
   381)     return
   382) 100 call piodie(__PIO_FILE__,__LINE__,'File open error ',0,filename)
   383) 101 call piodie(__PIO_FILE__,__LINE__,'File read error ',0,filename)
   384)
