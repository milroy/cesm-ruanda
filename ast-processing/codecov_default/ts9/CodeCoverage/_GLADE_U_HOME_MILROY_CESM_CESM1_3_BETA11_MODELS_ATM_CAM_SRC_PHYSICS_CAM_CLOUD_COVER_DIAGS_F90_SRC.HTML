    78) subroutine cldsav(lchnk   ,ncol    , &

    83) ! Purpose: 
    84) ! Compute total & 3 levels of cloud fraction assuming maximum-random overlap.
    85) ! Pressure ranges for the 3 cloud levels are specified.
    86) ! 
    87) ! Method: 
    88) ! <Describe the algorithm(s) used in the routine.>

    94) 
    95)    implicit none
    96) !
    97) !------------------------------Arguments--------------------------------
    98) !
    99) ! Input arguments
   100) !
   101)    integer, intent(in) :: lchnk                ! chunk identifier
   102)    integer, intent(in) :: ncol                 ! number of atmospheric columns
   103) 
   104)    real(r8), intent(in) :: cld(pcols,pver)     ! Cloud fraction
   105)    real(r8), intent(in) :: pmid(pcols,pver)    ! Level pressures
   106)    real(r8), intent(in) :: pmxrgn(pcols,pverp) ! Maximum values of pressure for each
   107) !    maximally overlapped region.
   108) !    0->pmxrgn(i,1) is range of pressure for
   109) !    1st region,pmxrgn(i,1)->pmxrgn(i,2) for
   110) !    2nd region, etc
   111) 
   112)    integer, intent(in) :: nmxrgn(pcols)        ! Number of maximally overlapped regions
   113) !
   114) ! Output arguments
   115) !
   116)    real(r8), intent(out) :: cldtot(pcols)       ! Total random overlap cloud cover
   117)    real(r8), intent(out) :: cldlow(pcols)       ! Low random overlap cloud cover
   118)    real(r8), intent(out) :: cldmed(pcols)       ! Middle random overlap cloud cover
   119)    real(r8), intent(out) :: cldhgh(pcols)       ! High random overlap cloud cover
   120) 
   121) !
   122) !---------------------------Local workspace-----------------------------
   123) !
   124)    integer i,k                  ! Longitude,level indices
   125)    integer irgn(pcols)          ! Max-overlap region index
   126)    integer max_nmxrgn           ! maximum value of nmxrgn over columns
   127)    integer ityp                 ! Type counter
   128)    real(r8) clrsky(pcols)       ! Max-random clear sky fraction
   129)    real(r8) clrskymax(pcols)    ! Maximum overlap clear sky fraction
   130) !------------------------------Parameters-------------------------------
   131)    real(r8) plowmax             ! Max prs for low cloud cover range
   132)    real(r8) plowmin             ! Min prs for low cloud cover range
   133)    real(r8) pmedmax             ! Max prs for mid cloud cover range
   134)    real(r8) pmedmin             ! Min prs for mid cloud cover range
   135)    real(r8) phghmax             ! Max prs for hgh cloud cover range
   136)    real(r8) phghmin             ! Min prs for hgh cloud cover range
   137) !
   138)    parameter (plowmax = 120000._r8,plowmin = 70000._r8, &
   139)               pmedmax =  70000._r8,pmedmin = 40000._r8, &
   140)               phghmax =  40000._r8,phghmin =  5000._r8)
   141) 
   142)    real(r8) ptypmin(4)
   143)    real(r8) ptypmax(4)
   144) 
   145)    data ptypmin /phghmin, plowmin, pmedmin, phghmin/
   146)    data ptypmax /plowmax, plowmax, pmedmax, phghmax/
   147) !
   148) !-----------------------------------------------------------------------
   149) !
   150) ! Initialize region number
   151) !
   152)    max_nmxrgn = -1
   153)    do i=1,ncol
   154)       max_nmxrgn = max(max_nmxrgn,nmxrgn(i))
   155)    end do
   156) 
   157)    do ityp = 1, 4
   158)       irgn(1:ncol) = 1
   159)       do k =1,max_nmxrgn-1
   160)          do i=1,ncol
   161)             if (pmxrgn(i,irgn(i)) < ptypmin(ityp) .and. irgn(i) < nmxrgn(i)) then
   162)                irgn(i) = irgn(i) + 1
   163)             end if
   164)          end do
   165)       end do
   166) !
   167) ! Compute cloud amount by estimating clear-sky amounts
   168) !
   169)       clrsky(1:ncol)    = 1.0_r8
   170)       clrskymax(1:ncol) = 1.0_r8
   171)       do k = 1, pver
   172)          do i=1,ncol
   173)             if (pmid(i,k) >= ptypmin(ityp) .and. pmid(i,k) <= ptypmax(ityp)) then
   174)                if (pmxrgn(i,irgn(i)) < pmid(i,k) .and. irgn(i) < nmxrgn(i)) then
   175)                   irgn(i) = irgn(i) + 1
   176)                   clrsky(i) = clrsky(i) * clrskymax(i)
   177)                   clrskymax(i) = 1.0_r8
   178)                endif
   179)                clrskymax(i) = min(clrskymax(i),1.0_r8-cld(i,k))
   180)             endif
   181)          end do
   182)       end do

   184)       if (ityp == 2) cldlow(1:ncol) = 1.0_r8 - (clrsky(1:ncol) * clrskymax(1:ncol))
   185)       if (ityp == 3) cldmed(1:ncol) = 1.0_r8 - (clrsky(1:ncol) * clrskymax(1:ncol))
   186)       if (ityp == 4) cldhgh(1:ncol) = 1.0_r8 - (clrsky(1:ncol) * clrskymax(1:ncol))
   187)    end do
