    18) ! ------- Modules -------
    19) 
    20)       use shr_kind_mod, only: r8 => shr_kind_r8
    21) 
    22) !      use parkind, only : jpim, jprb
    23)       use parrrsw, only : ngptsw, jpband, jpb1, jpb2
    24)       use rrsw_cld, only : extliq1, ssaliq1, asyliq1, &
    25)                            extice2, ssaice2, asyice2, &
    26)                            extice3, ssaice3, asyice3, fdlice3, &
    27)                            abari, bbari, cbari, dbari, ebari, fbari
    28)       use rrsw_wvn, only : wavenum1, wavenum2, ngb
    29)       use rrsw_vsn, only : hvrclc, hnamclc
    30) 
    31)       implicit none
    32) 
    33)       contains
    34) 
    35) ! ----------------------------------------------------------------------------
    36)       subroutine cldprmc_sw(nlayers, inflag, iceflag, liqflag, cldfmc, &
    37)                             ciwpmc, clwpmc, reicmc, dgesmc, relqmc, &
    38)                             taormc, taucmc, ssacmc, asmcmc, fsfcmc)
    39) ! ----------------------------------------------------------------------------
    40) 
    41) ! Purpose: Compute the cloud optical properties for each cloudy layer
    42) ! and g-point interval for use by the McICA method.  
    43) ! Note: Only inflag = 0 and inflag=2/liqflag=1/iceflag=2,3 are available;
    44) ! (Hu & Stamnes, Key, and Fu) are implemented. 
    45) 
    46) ! ------- Input -------
    47) 
    48)       integer, intent(in) :: nlayers         ! total number of layers
    49)       integer, intent(in) :: inflag          ! see definitions
    50)       integer, intent(in) :: iceflag         ! see definitions
    51)       integer, intent(in) :: liqflag         ! see definitions
    52) 
    53)       real(kind=r8), intent(in) :: cldfmc(:,:)        ! cloud fraction [mcica]
    54)                                                         !    Dimensions: (ngptsw,nlayers)
    55)       real(kind=r8), intent(in) :: ciwpmc(:,:)        ! cloud ice water path [mcica]
    56)                                                         !    Dimensions: (ngptsw,nlayers)
    57)       real(kind=r8), intent(in) :: clwpmc(:,:)        ! cloud liquid water path [mcica]
    58)                                                         !    Dimensions: (ngptsw,nlayers)
    59)       real(kind=r8), intent(in) :: relqmc(:)          ! cloud liquid particle effective radius (microns)
    60)                                                         !    Dimensions: (nlayers)
    61)       real(kind=r8), intent(in) :: reicmc(:)          ! cloud ice particle effective radius (microns)
    62)                                                         !    Dimensions: (nlayers)
    63)       real(kind=r8), intent(in) :: dgesmc(:)          ! cloud ice particle generalized effective size (microns)
    64)                                                         !    Dimensions: (nlayers)
    65)       real(kind=r8), intent(in) :: fsfcmc(:,:)        ! cloud forward scattering fraction 
    66)                                                         !    Dimensions: (ngptsw,nlayers)
    67) 
    68) ! ------- Output -------
    69) 
    70)       real(kind=r8), intent(inout) :: taucmc(:,:)     ! cloud optical depth (delta scaled)
    71)                                                         !    Dimensions: (ngptsw,nlayers)
    72)       real(kind=r8), intent(inout) :: ssacmc(:,:)     ! single scattering albedo (delta scaled)
    73)                                                         !    Dimensions: (ngptsw,nlayers)
    74)       real(kind=r8), intent(inout) :: asmcmc(:,:)     ! asymmetry parameter (delta scaled)
    75)                                                         !    Dimensions: (ngptsw,nlayers)
    76)       real(kind=r8), intent(out) :: taormc(:,:)       ! cloud optical depth (non-delta scaled)
    77)                                                         !    Dimensions: (ngptsw,nlayers)
    78) 
    79) ! ------- Local -------
    80) 
    81) !      integer :: ncbands
    82)       integer :: ib, lay, istr, index, icx, ig
    83) 
    84)       real(kind=r8), parameter :: eps = 1.e-06_r8     ! epsilon
    85)       real(kind=r8), parameter :: cldmin = 1.e-80_r8  ! minimum value for cloud quantities
    86)       real(kind=r8) :: cwp                            ! total cloud water path
    87)       real(kind=r8) :: radliq                         ! cloud liquid droplet radius (microns)
    88)       real(kind=r8) :: radice                         ! cloud ice effective radius (microns)
    89)       real(kind=r8) :: dgeice                         ! cloud ice generalized effective size (microns)
    90)       real(kind=r8) :: factor
    91)       real(kind=r8) :: fint
    92) 
    93)       real(kind=r8) :: taucldorig_a, taucloud_a, ssacloud_a, ffp, ffp1, ffpssa
    94)       real(kind=r8) :: tauiceorig, scatice, ssaice, tauice, tauliqorig, scatliq, ssaliq, tauliq
    95) 
    96)       real(kind=r8) :: fdelta(ngptsw)
    97)       real(kind=r8) :: extcoice(ngptsw), gice(ngptsw)
    98)       real(kind=r8) :: ssacoice(ngptsw), forwice(ngptsw)
    99)       real(kind=r8) :: extcoliq(ngptsw), gliq(ngptsw)
   100)       real(kind=r8) :: ssacoliq(ngptsw), forwliq(ngptsw)
   101) 
   102) ! Initialize
   103) 
   104)       hvrclc = '$Revision: 1.4 $'
   105) 
   106) ! Initialize
   107) 
   108) ! Some of these initializations are done in rrtmg_sw.f90.
   109)       do lay = 1, nlayers
   110)          do ig = 1, ngptsw
   111)             taormc(ig,lay) = taucmc(ig,lay)
   112) !            taucmc(ig,lay) = 0.0_r8
   113) !            ssacmc(ig,lay) = 1.0_r8
   114) !            asmcmc(ig,lay) = 0.0_r8
   115)          enddo
   116)       enddo
   117) 
   118) ! Main layer loop
   119)       do lay = 1, nlayers
   120) 
   121) ! Main g-point interval loop
   122)          do ig = 1, ngptsw 
   123)             cwp = ciwpmc(ig,lay) + clwpmc(ig,lay)
   124)             if (cldfmc(ig,lay) .ge. cldmin .and. &
   125)                (cwp .ge. cldmin .or. taucmc(ig,lay) .ge. cldmin)) then
   126) 
   127) ! (inflag=0): Cloud optical properties input directly
   128)                if (inflag .eq. 0) then
   129) ! Cloud optical properties already defined in taucmc, ssacmc, asmcmc are unscaled;
   130) ! Apply delta-M scaling here (using Henyey-Greenstein approximation)
   131)                   taucldorig_a = taucmc(ig,lay)
   132)                   ffp = fsfcmc(ig,lay)
   133)                   ffp1 = 1.0_r8 - ffp
   134)                   ffpssa = 1.0_r8 - ffp * ssacmc(ig,lay)
   135)                   ssacloud_a = ffp1 * ssacmc(ig,lay) / ffpssa
   136)                   taucloud_a = ffpssa * taucldorig_a
   137) 
   138)                   taormc(ig,lay) = taucldorig_a
   139)                   ssacmc(ig,lay) = ssacloud_a
   140)                   taucmc(ig,lay) = taucloud_a
   141)                   asmcmc(ig,lay) = (asmcmc(ig,lay) - ffp) / (ffp1)
   142) 
   143)                elseif (inflag .eq. 1) then 
   144)                   stop 'INFLAG = 1 OPTION NOT AVAILABLE WITH MCICA'
   145) 
   146) ! (inflag=2): Separate treatement of ice clouds and water clouds.
   147)                elseif (inflag .eq. 2) then       
   148)                   radice = reicmc(lay)
   149) 
   150) ! Calculation of absorption coefficients due to ice clouds.
   151)                   if (ciwpmc(ig,lay) .eq. 0.0) then
   152)                      extcoice(ig) = 0.0_r8
   153)                      ssacoice(ig) = 0.0_r8
   154)                      gice(ig)     = 0.0_r8
   155)                      forwice(ig)  = 0.0_r8
   156) 
   157) ! (iceflag = 1): 
   158) ! Note: This option uses Ebert and Curry approach for all particle sizes similar to
   159) ! CAM3 implementation, though this is somewhat unjustified for large ice particles
   160)                   elseif (iceflag .eq. 1) then
   161)                      ib = ngb(ig)
   162)                      if (wavenum2(ib) .gt. 1.43e04_r8) then
   163)                         icx = 1
   164)                      elseif (wavenum2(ib) .gt. 7.7e03_r8) then
   165)                         icx = 2
   166)                      elseif (wavenum2(ib) .gt. 5.3e03_r8) then
   167)                         icx = 3
   168)                      elseif (wavenum2(ib) .gt. 4.0e03_r8) then
   169)                         icx = 4
   170)                      elseif (wavenum2(ib) .ge. 2.5e03_r8) then
   171)                         icx = 5
   172)                      endif
   173)                      extcoice(ig) = (abari(icx) + bbari(icx)/radice)
   174)                      ssacoice(ig) = 1._r8 - cbari(icx) - dbari(icx) * radice
   175)                      gice(ig) = ebari(icx) + fbari(icx) * radice
   176) ! Check to ensure upper limit of gice is within physical limits for large particles
   177)                      if (gice(ig).ge.1._r8) gice(ig) = 1._r8 - eps
   178)                      forwice(ig) = gice(ig)*gice(ig)
   179) ! Check to ensure all calculated quantities are within physical limits.
   180)                      if (extcoice(ig) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   181)                      if (ssacoice(ig) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   182)                      if (ssacoice(ig) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   183)                      if (gice(ig) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   184)                      if (gice(ig) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   185) 
   186) ! For iceflag=2 option, combine with iceflag=0 option to handle large particle sizes.
   187) ! Use iceflag=2 option for ice particle effective radii from 5.0 to 131.0 microns
   188) ! and use iceflag=0 option for ice particles greater than 131.0 microns.
   189) ! *** NOTE: Transition between two methods has not been smoothed.
   190) 
   191)                   elseif (iceflag .eq. 2) then
   192)                      if (radice .lt. 5.0_r8) stop 'ICE RADIUS OUT OF BOUNDS'
   193)                      if (radice .ge. 5.0_r8 .and. radice .le. 131._r8) then
   194)                         factor = (radice - 2._r8)/3._r8
   195)                         index = int(factor)
   196)                         if (index .eq. 43) index = 42
   197)                         fint = factor - float(index)
   198)                         ib = ngb(ig)
   199)                         extcoice(ig) = extice2(index,ib) + fint * &
   200)                                       (extice2(index+1,ib) -  extice2(index,ib))
   201)                         ssacoice(ig) = ssaice2(index,ib) + fint * &
   202)                                       (ssaice2(index+1,ib) -  ssaice2(index,ib))
   203)                         gice(ig) = asyice2(index,ib) + fint * &
   204)                                       (asyice2(index+1,ib) -  asyice2(index,ib))
   205)                         forwice(ig) = gice(ig)*gice(ig)
   206) ! Check to ensure all calculated quantities are within physical limits.
   207)                         if (extcoice(ig) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   208)                         if (ssacoice(ig) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   209)                         if (ssacoice(ig) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   210)                         if (gice(ig) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   211)                         if (gice(ig) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   212)                      elseif (radice .gt. 131._r8) then
   213)                         ib = ngb(ig)
   214)                         if (wavenum2(ib) .gt. 1.43e04_r8) then
   215)                            icx = 1
   216)                         elseif (wavenum2(ib) .gt. 7.7e03_r8) then
   217)                            icx = 2
   218)                         elseif (wavenum2(ib) .gt. 5.3e03_r8) then
   219)                            icx = 3
   220)                         elseif (wavenum2(ib) .gt. 4.0e03_r8) then
   221)                            icx = 4
   222)                         elseif (wavenum2(ib) .ge. 2.5e03_r8) then
   223)                            icx = 5
   224)                         endif
   225)                         extcoice(ig) = (abari(icx) + bbari(icx)/radice)
   226)                         ssacoice(ig) = 1._r8 - cbari(icx) - dbari(icx) * radice
   227)                         gice(ig) = ebari(icx) + fbari(icx) * radice
   228) ! Check to ensure upper limit of gice is within physical limits for large particles
   229)                         if (gice(ig).ge.1.0_r8) gice(ig) = 1.0_r8-eps
   230)                         forwice(ig) = gice(ig)*gice(ig)
   231) ! Check to ensure all calculated quantities are within physical limits.
   232)                         if (extcoice(ig) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   233)                         if (ssacoice(ig) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   234)                         if (ssacoice(ig) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   235)                         if (gice(ig) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   236)                         if (gice(ig) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   237)                      endif
   238) 
   239) ! For iceflag=3 option, combine with iceflag=0 option to handle large particle sizes
   240) ! Use iceflag=3 option for ice particle effective radii from 3.2 to 91.0 microns
   241) ! (generalized effective size, dge, from 5 to 140 microns), and use iceflag=0 option
   242) ! for ice particle effective radii greater than 91.0 microns (dge = 140 microns).
   243) ! *** NOTE: Fu parameterization requires particle size in generalized effective size.
   244) ! *** NOTE: Transition between two methods has not been smoothed. 
   245) 
   246)                   elseif (iceflag .eq. 3) then
   247)                      dgeice = dgesmc(lay)
   248)                      if (dgeice .lt. 5.0_r8) stop 'ICE GENERALIZED EFFECTIVE SIZE OUT OF BOUNDS'
   249)                      if (dgeice .ge. 5.0_r8 .and. dgeice .le. 140._r8) then
   250)                         factor = (dgeice - 2._r8)/3._r8
   251)                         index = int(factor)
   252)                         if (index .eq. 46) index = 45
   253)                         fint = factor - float(index)
   254)                         ib = ngb(ig)
   255)                         extcoice(ig) = extice3(index,ib) + fint * &
   256)                                       (extice3(index+1,ib) - extice3(index,ib))
   257)                         ssacoice(ig) = ssaice3(index,ib) + fint * &
   258)                                       (ssaice3(index+1,ib) - ssaice3(index,ib))
   259)                         gice(ig) = asyice3(index,ib) + fint * &
   260)                                   (asyice3(index+1,ib) - asyice3(index,ib))
   261)                         fdelta(ig) = fdlice3(index,ib) + fint * &
   262)                                     (fdlice3(index+1,ib) - fdlice3(index,ib))
   263)                         if (fdelta(ig) .lt. 0.0_r8) stop 'FDELTA LESS THAN 0.0'
   264)                         if (fdelta(ig) .gt. 1.0_r8) stop 'FDELTA GT THAN 1.0'
   265)                         forwice(ig) = fdelta(ig) + 0.5_r8 / ssacoice(ig)
   266) ! See Fu 1996 p. 2067 
   267)                         if (forwice(ig) .gt. gice(ig)) forwice(ig) = gice(ig)
   268) ! Check to ensure all calculated quantities are within physical limits.  
   269)                         if (extcoice(ig) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   270)                         if (ssacoice(ig) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   271)                         if (ssacoice(ig) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   272)                         if (gice(ig) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   273)                         if (gice(ig) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   274)                      elseif (dgeice .gt. 140._r8) then
   275)                         ib = ngb(ig)
   276)                         if (wavenum2(ib) .gt. 1.43e04_r8) then
   277)                            icx = 1
   278)                         elseif (wavenum2(ib) .gt. 7.7e03_r8) then
   279)                            icx = 2
   280)                         elseif (wavenum2(ib) .gt. 5.3e03_r8) then
   281)                            icx = 3
   282)                         elseif (wavenum2(ib) .gt. 4.0e03_r8) then
   283)                            icx = 4
   284)                         elseif (wavenum2(ib) .ge. 2.5e03_r8) then
   285)                            icx = 5
   286)                         endif
   287)                         extcoice(ig) = (abari(icx) + bbari(icx)/radice)
   288)                         ssacoice(ig) = 1._r8 - cbari(icx) - dbari(icx) * radice
   289)                         gice(ig) = ebari(icx) + fbari(icx) * radice
   290) ! Check to ensure upper limit of gice is within physical limits for large particles
   291)                         if (gice(ig).ge.1._r8) gice(ig) = 1._r8 - eps
   292)                         forwice(ig) = gice(ig)*gice(ig)
   293) ! Check to ensure all calculated quantities are within physical limits.
   294)                         if (extcoice(ig) .lt. 0.0_r8) stop 'ICE EXTINCTION LESS THAN 0.0'
   295)                         if (ssacoice(ig) .gt. 1.0_r8) stop 'ICE SSA GRTR THAN 1.0'
   296)                         if (ssacoice(ig) .lt. 0.0_r8) stop 'ICE SSA LESS THAN 0.0'
   297)                         if (gice(ig) .gt. 1.0_r8) stop 'ICE ASYM GRTR THAN 1.0'
   298)                         if (gice(ig) .lt. 0.0_r8) stop 'ICE ASYM LESS THAN 0.0'
   299)                      endif
   300)                   endif
   301) 
   302) ! Calculation of absorption coefficients due to water clouds.
   303)                   if (clwpmc(ig,lay) .eq. 0.0_r8) then
   304)                      extcoliq(ig) = 0.0_r8
   305)                      ssacoliq(ig) = 0.0_r8
   306)                      gliq(ig) = 0.0_r8
   307)                      forwliq(ig) = 0.0_r8
   308) 
   309)                   elseif (liqflag .eq. 1) then
   310)                      radliq = relqmc(lay)
   311)                      if (radliq .lt. 1.5_r8 .or. radliq .gt. 60._r8) stop &
   312)                         'liquid effective radius out of bounds'
   313)                      index = int(radliq - 1.5_r8)
   314)                      if (index .eq. 0) index = 1
   315)                      if (index .eq. 58) index = 57
   316)                      fint = radliq - 1.5_r8 - float(index)
   317)                      ib = ngb(ig)
   318)                      extcoliq(ig) = extliq1(index,ib) + fint * &
   319)                                    (extliq1(index+1,ib) - extliq1(index,ib))
   320)                      ssacoliq(ig) = ssaliq1(index,ib) + fint * &
   321)                                    (ssaliq1(index+1,ib) - ssaliq1(index,ib))
   322)                      if (fint .lt. 0._r8 .and. ssacoliq(ig) .gt. 1._r8) &
   323)                                     ssacoliq(ig) = ssaliq1(index,ib)
   324)                      gliq(ig) = asyliq1(index,ib) + fint * &
   325)                                (asyliq1(index+1,ib) - asyliq1(index,ib))
   326)                      forwliq(ig) = gliq(ig)*gliq(ig)
   327) ! Check to ensure all calculated quantities are within physical limits.
   328)                      if (extcoliq(ig) .lt. 0.0_r8) stop 'LIQUID EXTINCTION LESS THAN 0.0'
   329)                      if (ssacoliq(ig) .gt. 1.0_r8) stop 'LIQUID SSA GRTR THAN 1.0'
   330)                      if (ssacoliq(ig) .lt. 0.0_r8) stop 'LIQUID SSA LESS THAN 0.0'
   331)                      if (gliq(ig) .gt. 1.0_r8) stop 'LIQUID ASYM GRTR THAN 1.0'
   332)                      if (gliq(ig) .lt. 0.0_r8) stop 'LIQUID ASYM LESS THAN 0.0'
   333)                   endif
   334)    
   335)                   tauliqorig = clwpmc(ig,lay) * extcoliq(ig)
   336)                   tauiceorig = ciwpmc(ig,lay) * extcoice(ig)
   337)                   taormc(ig,lay) = tauliqorig + tauiceorig
   338) 
   339)                   ssaliq = ssacoliq(ig) * (1._r8 - forwliq(ig)) / &
   340)                           (1._r8 - forwliq(ig) * ssacoliq(ig))
   341)                   tauliq = (1._r8 - forwliq(ig) * ssacoliq(ig)) * tauliqorig
   342)                   ssaice = ssacoice(ig) * (1._r8 - forwice(ig)) / &
   343)                           (1._r8 - forwice(ig) * ssacoice(ig))
   344)                   tauice = (1._r8 - forwice(ig) * ssacoice(ig)) * tauiceorig
   345) 
   346)                   scatliq = ssaliq * tauliq
   347)                   scatice = ssaice * tauice
   348)                   taucmc(ig,lay) = tauliq + tauice
   349) 
   350) ! Ensure non-zero taucmc and scatice
   351)                   if(taucmc(ig,lay).eq.0.) taucmc(ig,lay) = cldmin
   352)                   if(scatice.eq.0.) scatice = cldmin
   353) 
   354)                   ssacmc(ig,lay) = (scatliq + scatice) / taucmc(ig,lay)
   355) 
   356)                   if (iceflag .eq. 3) then
   357) ! In accordance with the 1996 Fu paper, equation A.3, 
   358) ! the moments for ice were calculated depending on whether using spheres
   359) ! or hexagonal ice crystals.
   360) ! Set asymetry parameter to first moment (istr=1)
   361)                      istr = 1
   362)                      asmcmc(ig,lay) = (1.0_r8/(scatliq+scatice))* &
   363)                         (scatliq*(gliq(ig)**istr - forwliq(ig)) / &
   364)                         (1.0_r8 - forwliq(ig)) + scatice * ((gice(ig)-forwice(ig))/ &
   365)                         (1.0_r8 - forwice(ig)))**istr)
   366) 
   367)                   else 
   368) ! This code is the standard method for delta-m scaling. 
   369) ! Set asymetry parameter to first moment (istr=1)
   370)                      istr = 1
   371)                      asmcmc(ig,lay) = (scatliq *  &
   372)                         (gliq(ig)**istr - forwliq(ig)) / &
   373)                         (1.0_r8 - forwliq(ig)) + scatice * (gice(ig)**istr - forwice(ig)) / &
   374)                         (1.0_r8 - forwice(ig)))/(scatliq + scatice)
   375)                   endif 
   376) 
   377)                endif
   378) 
   379)             endif
   380) 
   381) ! End g-point interval loop
   382)          enddo
   383) 
   384) ! End layer loop
   385)       enddo
   386) 
   387)       end subroutine cldprmc_sw
   388)
