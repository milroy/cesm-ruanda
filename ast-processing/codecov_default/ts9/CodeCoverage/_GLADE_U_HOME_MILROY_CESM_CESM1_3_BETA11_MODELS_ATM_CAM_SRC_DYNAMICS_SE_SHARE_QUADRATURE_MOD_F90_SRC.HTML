    99)   function gauss_pts(np1) result(pts)
   100)     use physical_constants, only : qq_pi
   101) 
   102)     integer, intent(in)     :: np1        ! Number of velocity grid points
   103)     real (kind=longdouble_kind) :: pts(np1)
   104) 
   105)     ! Local variables
   106) 
   107)     real (kind=longdouble_kind) :: alpha,beta
   108)     real (kind=longdouble_kind) :: xjac(0:np1-1)
   109)     real (kind=longdouble_kind) :: jac(0:np1)
   110)     real (kind=longdouble_kind) :: djac(0:np1)
   111) 
   112)     integer  prec                    ! number of mantissa bits
   113)     real (kind=longdouble_kind) eps      ! machine epsilon
   114)     real (kind=longdouble_kind), parameter :: convthresh = 10  ! convergence threshold relative\
   115) 
   116)     ! to machine epsilon
   117)     integer, parameter :: kstop = 30 ! max iterations for polynomial deflation
   118) 
   119)     real (kind=longdouble_kind) :: poly
   120)     real (kind=longdouble_kind) :: pder
   121)     real (kind=longdouble_kind) :: recsum,thresh
   122)     real (kind=longdouble_kind) :: dth
   123) 
   124)     real (kind=longdouble_kind) :: x
   125)     real (kind=longdouble_kind) :: delx
   126)     real (kind=longdouble_kind) :: c0,c1,c2,c10
   127) 
   128)     integer i,j,k
   129)     integer n, nh
   130) 
   131)     n  = np1 - 1
   132)     c0 = 0.0_longdouble_kind
   133)     c1 = 1.0_longdouble_kind
   134)     c2 = 2.0_longdouble_kind
   135)     c10 = 10.0_longdouble_kind
   136)     alpha = c0
   137)     beta  = c0
   138) 
   139)     ! =========================================================
   140)     ! compute machine precision and set the convergence
   141)     ! threshold thresh to 10 times that level
   142)     ! =========================================================
   143) 
   144)     prec   = precision(c10)
   145)     eps    = c10**(-prec)
   146)     thresh = convthresh*eps
   147) 
   148)     ! ============================================================
   149)     ! Compute first half of the roots by "polynomial deflation".
   150)     ! ============================================================
   151) 
   152)     dth = QQ_PI/(2*n+2)
   153) 
   154)     nh  = (n+1)/2
   155) 
   156)     do j=0,nh-1
   157)        x=COS((c2*j+1)*dth)          ! first guess at root
   158)        k=0
   159)        delx=c1
   160)        do while(k<kstop .and. ABS(delx) > thresh)
   161)           call jacobi(n+1,x,alpha,beta,jac(0:n+1),djac(0:n+1))
   162)           poly =  jac(n+1)
   163)           pder = djac(n+1)
   164)           recsum=c0
   165)           do i=0,j-1
   166)              recsum = recsum + c1/(x-xjac(i))
   167)           end do
   168)           delx = -poly/(pder-recsum*poly)
   169)           x = x + delx
   170)           k = k + 1
   171)        end do
   172) 
   173)        xjac(j)=x
   174) 
   175)     end do
   176) 
   177)     ! ================================================
   178)     ! compute the second half of the roots by symmetry
   179)     ! ================================================
   180) 
   181)     do j=0,nh
   182)        xjac(n-j) = -xjac(j)
   183)     end do
   184) 
   185)     if (MODULO(n,2)==0) xjac(nh)=c0
   186) 
   187)     ! ====================================================
   188)     ! Reverse the sign of everything so that indexing
   189)     ! increases with position
   190)     ! ====================================================
   191) 
   192)     do j=0,n
   193)        pts(j+1) = -xjac(j)
   194)     end do
   195) 
   196)   end function gauss_pts

   293)   ! gausslobatto_pts:
   294)   !
   295)   ! Compute the Gauss-Lobatto Collocation points 
   296)   ! for Jacobi Polynomials
   297)   !
   298)   ! ==============================================================
   299) 
   300)   function gausslobatto_pts(np1) result(pts)
   301)     use physical_constants, only : QQ_PI
   302) 
   303)     integer, intent(in)     :: np1        ! Number of velocity grid points
   304)     real (kind=longdouble_kind) :: pts(np1)
   305) 
   306)     ! Local variables
   307) 
   308)     real (kind=longdouble_kind) :: alpha,beta
   309)     real (kind=longdouble_kind) :: xjac(0:np1-1)
   310)     real (kind=longdouble_kind) :: jac(0:np1)
   311)     real (kind=longdouble_kind) :: jacm1(0:np1)
   312)     real (kind=longdouble_kind) :: djac(0:np1)
   313) 
   314)     integer  prec                    ! number of mantissa bits 
   315)     real (kind=longdouble_kind) eps      ! machine epsilon
   316)     real (kind=longdouble_kind), parameter :: convthresh = 10  ! convergence threshold relative 
   317)     ! to machine epsilon 
   318)     integer, parameter :: kstop = 30 ! max iterations for polynomial deflation
   319) 
   320)     real (kind=longdouble_kind) :: a,b,det
   321)     real (kind=longdouble_kind) :: poly
   322)     real (kind=longdouble_kind) :: pder
   323)     real (kind=longdouble_kind) :: recsum,thresh
   324)     real (kind=longdouble_kind) :: dth,cd,sd,cs,ss,cstmp
   325) 
   326)     real (kind=longdouble_kind) :: x
   327)     real (kind=longdouble_kind) :: delx
   328)     real (kind=longdouble_kind) :: c0,c1,c2,c10
   329) 
   330)     integer i,j,k
   331)     integer n, nh
   332) 
   333)     n  = np1 - 1
   334)     c0 = 0.0_longdouble_kind
   335)     c1 = 1.0_longdouble_kind
   336)     c2 = 2.0_longdouble_kind
   337)     c10 = 10.0_longdouble_kind
   338) 
   339)     alpha = c0
   340)     beta  = c0
   341) 
   342)     ! =========================================================
   343)     ! compute machine precision and set the convergence
   344)     ! threshold thresh to 10 times that level 
   345)     ! =========================================================
   346) 
   347)     prec   = PRECISION(c10)
   348)     eps    = c10**(-prec)
   349)     thresh = convthresh*eps
   350) 
   351)     ! =====================================================
   352)     ! initialize the end points
   353)     ! =====================================================
   354) 
   355)     xjac(0) =  c1
   356)     xjac(n) = -c1
   357) 
   358)     ! ============================================================
   359)     ! Compute first half of the roots by "polynomial deflation".
   360)     ! ============================================================
   361) 
   362)     ! ============================================================
   363)     ! compute the parameters in the polynomial whose 
   364)     ! roots are desired...
   365)     ! ============================================================
   366) 
   367)     call jacobi(n+1, c1,alpha,beta,jac(0:n+1),djac(0:n+1))
   368)     call jacobi(n+1,-c1,alpha,beta,jacm1(0:n+1),djac(0:n+1))
   369) 
   370)     det =   jac(n  )*jacm1(n-1)-jacm1(n  )*jac(n-1)
   371)     a   = -(jac(n+1)*jacm1(n-1)-jacm1(n+1)*jac(n-1))/det
   372)     b   = -(jac(n  )*jacm1(n+1)-jacm1(n  )*jac(n+1))/det
   373) 
   374)     dth = QQ_PI/(2*n+1)
   375)     cd  = COS(c2*dth)
   376)     sd  = SIN(c2*dth)
   377)     cs  = COS(dth)
   378)     ss  = SIN(dth)
   379) 
   380)     nh  = (n+1)/2
   381) 
   382)     do j=1,nh-1
   383)        x=cs          ! first guess at root 
   384)        k=0
   385)        delx=c1
   386)        do while(k<kstop .and. ABS(delx) > thresh)
   387)           call jacobi(n+1,x,alpha,beta,jac(0:n+1),djac(0:n+1))
   388)           poly =  jac(n+1)+a* jac(n)+b* jac(n-1)
   389)           pder = djac(n+1)+a*djac(n)+b*djac(n-1)
   390)           recsum=c0
   391)           do i=0,j-1
   392)              recsum = recsum + c1/(x-xjac(i))
   393)           end do
   394)           delx = -poly/(pder-recsum*poly)
   395)           x = x + delx
   396)           k = k + 1
   397)        end do
   398) 
   399)        xjac(j)=x
   400) 
   401)        ! =====================================================
   402)        !  compute the guesses for the roots
   403)        !  for the next points, i.e :
   404)        !
   405)        !  ss = sn(theta) => sin(theta+2*dth)
   406)        !  cs = cs(theta) => cs(theta+2*dth)
   407)        ! =====================================================
   408) 
   409)        cstmp=cs*cd-ss*sd      
   410)        ss=cs*sd+ss*cd    
   411)        cs=cstmp          
   412)     end do
   413) 
   414)     ! ================================================
   415)     ! compute the second half of the roots by symmetry
   416)     ! ================================================
   417) 
   418)     do j=1,nh 
   419)        xjac(n-j) = -xjac(j)
   420)     end do
   421) 
   422)     if (MODULO(n,2)==0) xjac(nh)=c0
   423) 
   424)     ! ====================================================
   425)     ! Reverse the sign of everything so that indexing
   426)     ! increases with position          
   427)     ! ====================================================
   428) 
   429)     do j=0,n
   430)        pts(j+1) = -xjac(j)
   431)     end do
   432) 
   433)   end function gausslobatto_pts

   503)   ! subroutine jacobi:
   504)   !
   505)   !  Computes the Jacobi Polynomials (jac) and their
   506)   !  first derivatives up to and including degree n 
   507)   !  at point x on the interval (-1,1).
   508)   !
   509)   !    See for example the recurrence relations 
   510)   !    in equation 2.5.4 (page 70) in 
   511)   !
   512)   !    "Spectral Methods in Fluid Dynamics",
   513)   !    by C. Canuto, M.Y. Hussaini, A. Quarteroni, T.A.Zang
   514)   !    Springer-Verlag, 1988.
   515)   ! ================================================
   516) 
   517)   subroutine jacobi(n, x, alpha, beta, jac, djac)
   518) 
   519)     integer, intent(in)                 :: n
   520)     real (kind=longdouble_kind), intent(in) :: x
   521)     real (kind=longdouble_kind), intent(in) :: alpha
   522)     real (kind=longdouble_kind), intent(in) :: beta
   523)     real (kind=longdouble_kind)             :: jac(0:n)
   524)     real (kind=longdouble_kind)             :: djac(0:n)
   525) 
   526)     ! Local variables
   527) 
   528)     real (kind=longdouble_kind) :: a1k
   529)     real (kind=longdouble_kind) :: a2k
   530)     real (kind=longdouble_kind) :: a3k
   531)     real (kind=longdouble_kind) :: da2kdx
   532) 
   533)     real (kind=longdouble_kind) :: c2,c1,c0
   534) 
   535)     integer ::  k
   536) 
   537)     c0 = 0.0_longdouble_kind
   538)     c1 = 1.0_longdouble_kind
   539)     c2 = 2.0_longdouble_kind
   540) 
   541)     jac(0)=c1
   542)     jac(1)=(c1 + alpha)*x
   543) 
   544)     djac(0)=c0
   545)     djac(1)=(c1 + alpha)
   546) 
   547)     do k=1,n-1
   548)        a1k      = c2*( k + c1 )*( k + alpha + beta + c1 )*( c2*k + alpha + beta )
   549)        da2kdx   = ( c2*( k + c1 ) + alpha + beta )*( c2*k + alpha + beta + c1 )*( c2*k + alpha + beta )
   550)        a2k      = ( c2*k + alpha + beta + c1 )*( alpha*alpha - beta*beta ) + x*da2kdx
   551)        a3k      = c2*(k + alpha)*( k + beta )*( c2*k + alpha + beta + c2 )
   552)        jac(k+1) = ( a2k*jac(k)-a3k*jac(k-1) )/a1k
   553)        djac(k+1)= ( a2k*djac(k) + da2kdx*jac(k) - a3k*djac(k-1) )/a1k          
   554)     end do
   555) 
   556)   end subroutine jacobi
