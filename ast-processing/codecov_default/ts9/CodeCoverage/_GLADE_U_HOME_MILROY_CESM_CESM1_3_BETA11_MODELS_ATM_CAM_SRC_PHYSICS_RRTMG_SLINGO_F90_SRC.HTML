   185) subroutine slingo_liq_optics_sw(state, pbuf, liq_tau, liq_tau_w, liq_tau_w_g, liq_tau_w_f, oldliqwp)
   186) 
   187)    use physconst, only: gravit
   188) 
   189)    type(physics_state), intent(in) :: state
   190)    type(physics_buffer_desc), pointer :: pbuf(:)
   191) 
   192)    real(r8),intent(out) :: liq_tau    (nswbands,pcols,pver) ! extinction optical depth
   193)    real(r8),intent(out) :: liq_tau_w  (nswbands,pcols,pver) ! single scattering albedo * tau
   194)    real(r8),intent(out) :: liq_tau_w_g(nswbands,pcols,pver) ! assymetry parameter * tau * w
   195)    real(r8),intent(out) :: liq_tau_w_f(nswbands,pcols,pver) ! forward scattered fraction * tau * w
   196)    logical, intent(in) :: oldliqwp
   197) 
   198)    real(r8), pointer, dimension(:,:) :: rel
   199)    real(r8), pointer, dimension(:,:) :: cldn
   200)    real(r8), pointer, dimension(:,:) :: tmpptr
   201)    real(r8), dimension(pcols,pver) :: cliqwp
   202)    real(r8), dimension(nswbands) :: wavmin
   203)    real(r8), dimension(nswbands) :: wavmax
   204) 
   205)    ! Minimum cloud amount (as a fraction of the grid-box area) to 
   206)    ! distinguish from clear sky
   207)    real(r8), parameter :: cldmin = 1.0e-80_r8
   208) 
   209)    ! Decimal precision of cloud amount (0 -> preserve full resolution;
   210)    ! 10^-n -> preserve n digits of cloud amount)
   211)    real(r8), parameter :: cldeps = 0.0_r8
   212) 
   213)    ! A. Slingo's data for cloud particle radiative properties (from 'A GCM
   214)    ! Parameterization for the Shortwave Properties of Water Clouds' JAS
   215)    ! vol. 46 may 1989 pp 1419-1427)
   216)    real(r8) :: abarl(4) = &  ! A coefficient for extinction optical depth
   217)       (/ 2.817e-02_r8, 2.682e-02_r8,2.264e-02_r8,1.281e-02_r8/)
   218)    real(r8) :: bbarl(4) = &  ! B coefficient for extinction optical depth
   219)       (/ 1.305_r8    , 1.346_r8    ,1.454_r8    ,1.641_r8    /)
   220)    real(r8) :: cbarl(4) = &  ! C coefficient for single scat albedo
   221)       (/-5.62e-08_r8 ,-6.94e-06_r8 ,4.64e-04_r8 ,0.201_r8    /)
   222)    real(r8) :: dbarl(4) = &  ! D coefficient for single  scat albedo
   223)       (/ 1.63e-07_r8 , 2.35e-05_r8 ,1.24e-03_r8 ,7.56e-03_r8 /)
   224)    real(r8) :: ebarl(4) = &  ! E coefficient for asymmetry parameter
   225)       (/ 0.829_r8    , 0.794_r8    ,0.754_r8    ,0.826_r8    /)
   226)    real(r8) :: fbarl(4) = &  ! F coefficient for asymmetry parameter
   227)       (/ 2.482e-03_r8, 4.226e-03_r8,6.560e-03_r8,4.353e-03_r8/)
   228) 
   229)    real(r8) :: abarli        ! A coefficient for current spectral band
   230)    real(r8) :: bbarli        ! B coefficient for current spectral band
   231)    real(r8) :: cbarli        ! C coefficient for current spectral band
   232)    real(r8) :: dbarli        ! D coefficient for current spectral band
   233)    real(r8) :: ebarli        ! E coefficient for current spectral band
   234)    real(r8) :: fbarli        ! F coefficient for current spectral band
   235) 
   236)    ! Caution... A. Slingo recommends no less than 4.0 micro-meters nor
   237)    ! greater than 20 micro-meters
   238) 
   239)    integer :: ns, i, k, indxsl, Nday
   240)    integer :: i_rel, lchnk, icld, itim_old
   241)    real(r8) :: tmp1l, tmp2l, tmp3l, g
   242)    real(r8) :: kext(pcols,pver)
   243)    real(r8), pointer, dimension(:,:) :: iclwpth
   244) 
   245)    Nday = state%ncol
   246)    lchnk = state%lchnk
   247) 
   248)    itim_old = pbuf_old_tim_idx()
   249)    call pbuf_get_field(pbuf, cld_idx, cldn, start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
   250)    call pbuf_get_field(pbuf, rel_idx, rel)
   251) 
   252)    if (oldliqwp) then
   253)      do k=1,pver
   254)         do i = 1,Nday
   255)            cliqwp(i,k) = state%q(i,k,ixcldliq)*state%pdel(i,k)/(gravit*max(0.01_r8,cldn(i,k)))
   256)         end do
   257)      end do
   258)    else
   259)      if (iclwp_idx<=0) then 
   260)         call endrun('slingo_liq_optics_sw: oldliqwp must be set to true since ICLWP was not found in pbuf')
   261)      endif
   262)      ! The following is the eventual target specification for in cloud liquid water path.
   263)      call pbuf_get_field(pbuf, iclwp_idx, tmpptr)
   264)      cliqwp = tmpptr
   265)    endif
   266)    
   267)    call get_sw_spectral_boundaries(wavmin,wavmax,'microns')
   268)   
   269)    do ns = 1, nswbands
   270)       ! Set index for cloud particle properties based on the wavelength,
   271)       ! according to A. Slingo (1989) equations 1-3:
   272)       ! Use index 1 (0.25 to 0.69 micrometers) for visible
   273)       ! Use index 2 (0.69 - 1.19 micrometers) for near-infrared
   274)       ! Use index 3 (1.19 to 2.38 micrometers) for near-infrared
   275)       ! Use index 4 (2.38 to 4.00 micrometers) for near-infrared
   276)       if(wavmax(ns) <= 0.7_r8) then
   277)          indxsl = 1
   278)       else if(wavmax(ns) <= 1.25_r8) then
   279)          indxsl = 2
   280)       else if(wavmax(ns) <= 2.38_r8) then
   281)          indxsl = 3
   282)       else if(wavmax(ns) > 2.38_r8) then
   283)          indxsl = 4
   284)       end if
   285) 
   286)       ! Set cloud extinction optical depth, single scatter albedo,
   287)       ! asymmetry parameter, and forward scattered fraction:
   288)       abarli = abarl(indxsl)
   289)       bbarli = bbarl(indxsl)
   290)       cbarli = cbarl(indxsl)
   291)       dbarli = dbarl(indxsl)
   292)       ebarli = ebarl(indxsl)
   293)       fbarli = fbarl(indxsl)
   294) 
   295)       do k=1,pver
   296)          do i=1,Nday
   297) 
   298)             ! note that optical properties for liquid valid only
   299)             ! in range of 4.2 > rel > 16 micron (Slingo 89)
   300)             if (cldn(i,k) >= cldmin .and. cldn(i,k) >= cldeps) then
   301)                tmp1l = abarli + bbarli/min(max(4.2_r8,rel(i,k)),16._r8)
   302)                liq_tau(ns,i,k) = 1000._r8*cliqwp(i,k)*tmp1l
   303)             else
   304)                liq_tau(ns,i,k) = 0.0_r8
   305)             endif
   306) 
   307)             tmp2l = 1._r8 - cbarli - dbarli*min(max(4.2_r8,rel(i,k)),16._r8)
   308)             tmp3l = fbarli*min(max(4.2_r8,rel(i,k)),16._r8)
   309)             ! Do not let single scatter albedo be 1.  Delta-eddington solution
   310)             ! for non-conservative case has different analytic form from solution
   311)             ! for conservative case, and raddedmx is written for non-conservative case.
   312)             liq_tau_w(ns,i,k) = liq_tau(ns,i,k) * min(tmp2l,.999999_r8)
   313)             g = ebarli + tmp3l
   314)             liq_tau_w_g(ns,i,k) = liq_tau_w(ns,i,k) * g
   315)             liq_tau_w_f(ns,i,k) = liq_tau_w(ns,i,k) * g * g
   316) 
   317)          end do ! End do i=1,Nday
   318)       end do    ! End do k=1,pver
   319)    end do ! nswbands
   320) 
   321)    !call outfld('CL_OD_SW_OLD',liq_tau(idx_sw_diag,:,:), pcols, lchnk)
   322)    !call outfld('REL_OLD',rel(:,:), pcols, lchnk)
   323)    !call outfld('CLWPTH_OLD',cliqwp(:,:), pcols, lchnk)
   324)    !call outfld('KEXT_OLD',kext(:,:), pcols, lchnk)
   325) 
   326) 
   327) end subroutine slingo_liq_optics_sw

   329) subroutine slingo_liq_get_rad_props_lw(state, pbuf, abs_od, oldliqwp)
   330)    use physconst, only: gravit
   331) 
   332)    type(physics_state), intent(in)  :: state
   333)    type(physics_buffer_desc),pointer  :: pbuf(:)
   334)    real(r8), intent(out) :: abs_od(nlwbands,pcols,pver)
   335)    logical, intent(in) :: oldliqwp
   336) 
   337)    real(r8) :: gicewp(pcols,pver)
   338)    real(r8) :: gliqwp(pcols,pver)
   339)    real(r8) :: cicewp(pcols,pver)
   340)    real(r8) :: cliqwp(pcols,pver)
   341)    real(r8) :: ficemr(pcols,pver)
   342)    real(r8) :: cwp(pcols,pver)
   343)    real(r8) :: cldtau(pcols,pver)
   344) 
   345)    real(r8), pointer, dimension(:,:) :: cldn
   346)    real(r8), pointer, dimension(:,:) :: rei
   347)    integer :: ncol, icld, itim_old, i_rei, lwband, i, k, lchnk 
   348) 
   349)     real(r8) :: kabs, kabsi
   350)     real(r8) kabsl                  ! longwave liquid absorption coeff (m**2/g)
   351)     parameter (kabsl = 0.090361_r8)
   352) 
   353)    real(r8), pointer, dimension(:,:) :: iclwpth, iciwpth
   354) 
   355)     ncol=state%ncol
   356)    lchnk = state%lchnk
   357) 
   358)    itim_old  =  pbuf_old_tim_idx()
   359)    call pbuf_get_field(pbuf, rei_idx,   rei)
   360)    call pbuf_get_field(pbuf, cld_idx,   cldn, start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
   361) 
   362)    if (oldliqwp) then
   363)      do k=1,pver
   364)          do i = 1,ncol
   365)             gicewp(i,k) = state%q(i,k,ixcldice)*state%pdel(i,k)/gravit*1000.0_r8  ! Grid box ice water path.
   366)             gliqwp(i,k) = state%q(i,k,ixcldliq)*state%pdel(i,k)/gravit*1000.0_r8  ! Grid box liquid water path.
   367)             cicewp(i,k) = gicewp(i,k) / max(0.01_r8,cldn(i,k))                 ! In-cloud ice water path.
   368)             cliqwp(i,k) = gliqwp(i,k) / max(0.01_r8,cldn(i,k))                 ! In-cloud liquid water path.
   369)             ficemr(i,k) = state%q(i,k,ixcldice) /                 &
   370)                  max(1.e-10_r8,(state%q(i,k,ixcldice)+state%q(i,k,ixcldliq)))
   371)          end do
   372)      end do
   373)      cwp(:ncol,:pver) = cicewp(:ncol,:pver) + cliqwp(:ncol,:pver)
   374)    else
   375)      if (iclwp_idx<=0 .or. iciwp_idx<=0) then 
   376)         call endrun('slingo_liq_get_rad_props_lw: oldliqwp must be set to true since ICIWP and/or ICLWP were not found in pbuf')
   377)      endif
   378)      call pbuf_get_field(pbuf, iclwp_idx, iclwpth)
   379)      call pbuf_get_field(pbuf, iciwp_idx, iciwpth)
   380)      do k=1,pver
   381)          do i = 1,ncol
   382)               cwp   (i,k) = 1000.0_r8 * iclwpth(i,k) + 1000.0_r8 * iciwpth(i, k)
   383)               ficemr(i,k) = 1000.0_r8 * iciwpth(i,k)/(max(1.e-18_r8, cwp(i,k)))
   384)          end do
   385)      end do
   386)    endif
   387) 
   388) 
   389)    do k=1,pver
   390)        do i=1,ncol
   391) 
   392)           ! Note from Andrew Conley:
   393)           !  Optics for RK no longer supported, This is constructed to get
   394)           !  close to bit for bit.  Otherwise we could simply use liquid water path
   395)           !note that optical properties for ice valid only
   396)           !in range of 13 > rei > 130 micron (Ebert and Curry 92)
   397)           kabs = kabsl*(1._r8-ficemr(i,k))
   398)           cldtau(i,k) = kabs*cwp(i,k)
   399)        end do
   400)    end do
   401) !
   402)    do lwband = 1,nlwbands
   403)       abs_od(lwband,1:ncol,1:pver)=cldtau(1:ncol,1:pver)
   404)    enddo
   405) 
   406) 
   407) end subroutine slingo_liq_get_rad_props_lw
