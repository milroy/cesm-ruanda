     1) module cam_history
     2) !-------------------------------------------------------------------------------------------

     4) ! The cam_history module provides the user interface for CAM's history output capabilities.
     5) ! It maintains the lists of fields that are written to each history file, and the associated

   179) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   180) !
   181) !  Hashing.
   182) !
   183) !  Accelerate outfld processing by using a hash function of the field name
   184) !  to index masterlist and determine whehter the particular field is to
   185) !  be written to any history tape.
   186) !
   187) !
   188) !  Note: the outfld hashing logic will fail if any of the following are true:
   189) !
   190) !         1) The lower bound on the dimension of 'masterlist' is less than 1.
   191) !
   192) !         2) 'outfld' is called with field names that are not defined on
   193) !            masterlist.  This applies to both initial/branch and restart
   194) !            runs.
   195) !
   196) !         3) An inconsistency between a field's tape active flag
   197) !            'masterlist(ff)%actflag(t)' and active fields read from
   198) !            restart files.
   199) !
   200) !         4) Invoking function 'gen_hash_key' before the primary and secondary
   201) !            hash tables have been created (routine bld_outfld_hash_tbls).
   202) !
   203) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   204) 
   205) !
   206) !  User definable constants for hash and overflow tables.
   207) !  Define size of primary hash table (specified as 2**size).
   208) !
   209)    integer, parameter :: tbl_hash_pri_sz_lg2 = 16
   210) !
   211) !  Define size of overflow hash table % of primary hash table.
   212) !
   213)    integer, parameter :: tbl_hash_oflow_percent = 20
   214) !
   215) !  Do *not* modify the parameters below.
   216) !
   217)    integer, parameter :: tbl_hash_pri_sz = 2**tbl_hash_pri_sz_lg2
   218)    integer, parameter :: tbl_hash_oflow_sz = tbl_hash_pri_sz * (tbl_hash_oflow_percent/100.0_r8) 
   219) !
   220) !  The primary and overflow tables are organized to mimimize space (read:
   221) !  try to maximimze cache line usage).
   222) !
   223) !  gen_hash_key(fieldname) will return an index on the interval
   224) !  [0 ... tbl_hash_pri_sz-1].
   225) !
   226) !
   227) !  Primary:
   228) !  gen_hash_key(fieldname)-------+     +----------+
   229) !                                |     |   -ii    | 1 ------>tbl_hash_oflow(ii)
   230) !                                |     +----------+
   231) !                                +-->  |    ff    | 2 ------>masterlist(ff)
   232) !                                      +----------+
   233) !                                      |          | ...
   234) !                                      +----------+
   235) !                                      |          | tbl_hash_pri_sz
   236) !                                      +----------+
   237) !
   238) !  Overflow (if tbl_hash_pri() < 0):
   239) !  tbl_hash_pri(gen_hash_key(fieldname))
   240) !                         |
   241) !                         |            +----------+
   242) !                         |            |     1    | 1  (one entry on O.F. chain)
   243) !                         |            +----------+
   244) !                         |            |    ff_m  | 2
   245) !                         |            +----------+
   246) !                         +--------->  |     3    | 3  (three entries on chain)
   247) !                                      +----------+
   248) !                                      |    ff_x  | 4
   249) !                                      +----------+
   250) !                                      |    ff_y  | 5
   251) !                                      +----------+
   252) !                                      |    ff_z  | 6
   253) !                                      +----------+
   254) !                                      |          | ...
   255) !                                      +----------+
   256) !                                      |          | tbl_hash_oflow_sz
   257) !                                      +----------+
   258) !
   259) !
   260)    integer, dimension(0:tbl_hash_pri_sz-1) :: tbl_hash_pri ! Primary hash table

   344)   subroutine intht ()
   345) !
   346) !----------------------------------------------------------------------- 
   347) ! 
   348) ! Purpose: Initialize history file handler for initial or continuation run.
   349) !          For example, on an initial run, this routine initializes "ptapes"
   350) !          history files.  On a restart or regeneration  run, this routine 
   351) !          only initializes history files declared beyond what existed on the 
   352) !          previous run.  Files which already existed on the previous run have 
   353) !          already been initialized (i.e. named and opened) in routine RESTRT.
   354) ! 
   355) ! Method: Loop over tapes and fields per tape setting appropriate variables and
   356) !         calling appropriate routines
   357) ! 
   358) ! Author: CCM Core Group
   359) ! 
   360) !-----------------------------------------------------------------------
   361)     use ioFileMod
   362)     use shr_sys_mod,     only: shr_sys_getenv
   363)     use time_manager,    only: get_prev_time, get_curr_time
   364)     use cam_control_mod, only: nsrest
   365)     use dycore,          only: dycore_is
   366)     use sat_hist,        only: sat_hist_init
   367) #if (defined SPMD)
   368)     use spmd_utils, only : mpichar, mpicom
   369) #endif
   370) !
   371) !-----------------------------------------------------------------------
   372) !
   373) ! Local workspace
   374) !
   375)     integer :: t, f              ! tape, field indices
   376)     integer :: begdim1           ! on-node dim1 start index
   377)     integer :: enddim1           ! on-node dim1 end index
   378)     integer :: begdim2           ! on-node dim2 start index
   379)     integer :: enddim2           ! on-node dim2 end index
   380)     integer :: begdim3           ! on-node chunk or lat start index
   381)     integer :: enddim3           ! on-node chunk or lat end index
   382)     type (dim_index_3d) :: dimind  ! 3-D dimension index
   383)     integer :: day, sec          ! day and seconds from base date
   384)     integer :: i, ii, ff         ! index
   385)     integer :: rcode             ! shr_sys_getenv return code
   386)     type(master_entry), pointer :: listentry
   387)     character(len=32) :: fldname ! temp variable used to produce a left justified field name
   388)                                    ! in the formatted logfile output
   389) 
   390) !
   391) ! Print master field list
   392) !
   393) 
   394)       if (masterproc) then
   395)          write(iulog,*) ' '
   396)          write(iulog,*)' ******* MASTER FIELD LIST *******'
   397)       end if
   398)       listentry=>masterlinkedlist
   399)       f=0
   400)       do while(associated(listentry))
   401)          f=f+1
   402)          if(masterproc) then
   403)             fldname = listentry%field%name
   404)             write(iulog,9000) f, fldname, listentry%field%units, listentry%field%numlev, &
   405)                              listentry%avgflag(1), trim(listentry%field%long_name)
   406) 9000        format(i5, 1x, a32, 1x, a16, 1x, i4, 1x, a1, 2x, a)
   407)          end if
   408)          listentry=>listentry%next_entry
   409)       end do
   410)       nfmaster = f
   411)       if(masterproc) write(iulog,*)'BLDFLD:nfmaster=',nfmaster
   412) 
   413) !
   414) !  Now that masterlinkedlist is defined and we are performing a restart run
   415) !  (after all addfld calls), construct primary and secondary hashing tables.
   416) !
   417)       if ( nsrest==1 ) then
   418)          call bld_outfld_hash_tbls()
   419)          call bld_htapefld_indices()
   420)          return
   421)       end if
   422) !
   423) ! Get users logname and machine hostname
   424) !
   425)       if ( masterproc )then
   426)          logname = ' '
   427)          call shr_sys_getenv ('LOGNAME',logname,rcode)
   428)          host = ' '
   429)          call shr_sys_getenv ('HOST',host,rcode)
   430)       end if
   431) #ifdef SPMD
   432) ! PIO requires netcdf attributes have consistant values on all tasks
   433)       call mpibcast(logname, len(logname), mpichar, 0, mpicom)
   434)       call mpibcast(host, len(host), mpichar, 0, mpicom)
   435) #endif
   436) !
   437) ! Override averaging flag for all fields on a particular tape if namelist input so specifies
   438) !
   439)       do t=1,ptapes
   440)          if (avgflag_pertape(t) /= ' ') then
   441)             call h_override (t)
   442)          end if
   443)       end do
   444) !
   445) ! Define field list information for all history files.  
   446) !
   447)       call fldlst ()
   448) !
   449) ! Loop over max. no. of history files permitted  
   450) !
   451)       if ( nsrest .eq. 3 ) then
   452)          call get_prev_time(day, sec)  ! elapased time since reference date
   453)       else
   454)          call get_curr_time(day, sec)  ! elapased time since reference date
   455)       end if
   456)       do t=1,ptapes
   457)          nfils(t) = 0            ! no. of time samples in hist. file no. t
   458) 
   459) ! Time at beginning of current averaging interval.
   460)          
   461)          beg_time(t) = day + sec/86400._r8
   462)       end do
   463) 
   464) !
   465) ! Initialize history variables
   466) !
   467)       do t=1,ptapes
   468)          do f=1,nflds(t)
   469)             begdim1  = tape(t)%hlist(f)%field%begdim1
   470)             enddim1  = tape(t)%hlist(f)%field%enddim1
   471)             begdim2  = tape(t)%hlist(f)%field%begdim2
   472)             enddim2  = tape(t)%hlist(f)%field%enddim2
   473)             begdim3  = tape(t)%hlist(f)%field%begdim3
   474)             enddim3  = tape(t)%hlist(f)%field%enddim3
   475)             allocate(tape(t)%hlist(f)%hbuf(begdim1:enddim1,begdim2:enddim2,begdim3:enddim3))
   476)             tape(t)%hlist(f)%hbuf = 0._r8
   477)             if(tape(t)%hlist(f)%field%flag_xyfill .or. (avgflag_pertape(t) .eq. 'L')) then
   478)                allocate (tape(t)%hlist(f)%nacs(begdim1:enddim1,begdim3:enddim3))
   479)             else
   480)                allocate (tape(t)%hlist(f)%nacs(1,begdim3:enddim3))
   481)             end if
   482)             tape(t)%hlist(f)%nacs(:,:)=0           
   483)             tape(t)%hlist(f)%field%vector_compliment=0            
   484)          end do
   485)       end do
   486)       ! Setup vector pairs for unstructured grid interpolation
   487)       call setup_interpolation_and_define_vector_compliments()
   488)       !  Initialize the sat following history subsystem
   489)       call sat_hist_init()
   490)       
   491)       return
   492)    end subroutine intht

  1548) !---------------------------Local variables-----------------------------
  1549) !

  1551)       integer ff                     ! index into include, exclude and fprec list
  1552)       character(len=fieldname_len) :: name ! field name portion of fincl (i.e. no avgflag separator)

  1572)             name = getname (fincl(f,t))

  1582)

  1586)             listentry => get_entry_by_name(masterlinkedlist, fexcl(f,t))

  1618)       if(is_initfile()) then

  1623)          mfilt (ptapes) = 1

  1647)                   nflds(t) = nflds(t)+1

  1775)                end do

  1778)

  1792)                end if
  1793)

  1795) 
  1796)                prec_str = 'double'
  1797)                if (ndens(t) == 2) prec_str = 'single'
  1798)                write(iulog,*) ' Output precision:                ', prec_str
  1799)                write(iulog,*) ' Number of time samples per file: ', mfilt(t)
  1800)

  1824)

  1826) 
  1827) 9000  format(i5, 1x, a32, 1x, a16, 1x, i4, 1x, a1, 2x, 256a)

  2350)             call endrun
  2351)          endif
  2352) !
  2353) ! Find closest lat index for each substring
  2354) !
  2355)          latind(i)=mlat
  2356)          do j = 1, mlat-1
  2357)             if ( abs(latdeg(j)-latdegree(i)) .lt.  &
  2358)                  abs(latdeg(j+1)-latdegree(i))) then
  2359)                latind(i)=j
  2360)                exit
  2361)             endif
  2362)          enddo
  2363)       end do
  2364) !
  2365) ! output begining and ending latitude indices. If just a column is specified then beginning latitude index will be the same as the
  2366) ! ending latitude index  
  2367) 
  2368) !      
  2369)       if (latind(1) .le. latind(2) ) then
  2370)          beglatind =latind(1)
  2371)          endlatind =latind(2)
  2372)       else 
  2373)          beglatind = latind(2)
  2374)          endlatind = latind(1)
  2375)       end if
  2376)       if (beglatind .eq. endlatind) then
  2377)          latname = 'LAT_'//trim(tmpstr(1))
  2378)       else
  2379)          if (latind(1) .le. latind(2) ) then
  2380)             latname = 'LAT_'//trim(tmpstr(1)) // "_to_" // trim(tmpstr(2))
  2381)          else 
  2382)             latname = 'LAT_'//trim(tmpstr(2)) // "_to_" // trim(tmpstr(1))
  2383)          end if
  2384)       end if
  2385)       ! one last sanity check
  2386)       if(scan(latname,' ') > 0) then
  2387)          latname(scan(latname,' '):) = ' '
  2388)       end if
  2389) 
  2390)       return
  2391)    end subroutine getlatind

  2394)    subroutine getlonind (inname,beglonind,endlonind,lonname,londeg,mlon)
  2395) !
  2396) !----------------------------------------------------------------------- 
  2397) ! 
  2398) ! Purpose: retrieve closes model lat index given north south latitude string
  2399) !          
  2400) ! Method:  If an averaging flag separater character is present (":") in inname, 
  2401) !          lop it off
  2402) ! 
  2403) !-------------------------------------------------------------------------------
  2404) !
  2405) ! Arguments
  2406) !
  2407)       character(len=*) inname,lonname
  2408)       integer :: beglonind,endlonind
  2409)       integer :: mlon
  2410)       real(r8) :: londeg(mlon)
  2411) !
  2412) ! Local workspace
  2413) !
  2414)       character(len=max_chars) str1,str2,tmpstr(4)
  2415)       integer :: i,j,marker,lonind(2),tmplen(4)
  2416)       real(r8) :: londegree(2)
  2417)       real(r8) :: min, tmp
  2418) !
  2419) !-------------------------------------------------------------------------------
  2420) !
  2421)       str1=' '
  2422)       str2=' '
  2423)       tmpstr(:)= ' '
  2424) !
  2425) ! make sure _ separator is present
  2426) !      
  2427)       if (scan(inname,'_').eq.0) then
  2428)          write(iulog,*)'GETLONIND: Improperly formatted column string.  Missing underscore character (xxxE_yyyS) ',&
  2429)               inname,scan(inname,'_')
  2430)          call endrun
  2431)       end if
  2432) !
  2433) ! split string in to lat lon strings
  2434) !      
  2435)       marker=scan(inname,'_')
  2436)       str1=inname(:marker-1)
  2437)       str2=inname(marker+1:)
  2438) 
  2439) !
  2440) ! split ranges of lats( or lons) into seperate substrings. Substrings 1,2 will contain lats(lons)
  2441) ! from portion of string before underscore
  2442) ! if a single column and not a range is specified substrings 1 and 2 will be the same
  2443) ! and substrings 3,4 will contain lats(lons) from portion of main string after underscore character
  2444) ! if a single column and not a range is specified substrings 3 and 4 will be the same
  2445) !
  2446)       if (scan(str1,':') .ne. 0) then
  2447)          marker=scan(str1,':')
  2448)          tmpstr(1)=str1(:marker-1)
  2449)          tmpstr(2)=str1(marker+1:)
  2450)       else
  2451)          tmpstr(1)=str1
  2452)          tmpstr(2)=str1
  2453)       end if
  2454) 
  2455)       if (scan(str2,':') .ne. 0) then
  2456)          marker=scan(str2,':')
  2457)          tmpstr(3)=str2(:marker-1)
  2458)          tmpstr(4)=str2(marker+1:)
  2459)       else
  2460)          tmpstr(3)=str2
  2461)          tmpstr(4)=str2
  2462)       end if
  2463) 
  2464) ! check format of substrings - (number followed by single character north/south east/west designation)
  2465) 
  2466)       do i = 1,4
  2467)          tmplen(i)=len_trim(tmpstr(i))
  2468)          if (verify(tmpstr(i),"0123456789.").ne.tmplen(i) .or. verify(tmpstr(i)(tmplen(i):tmplen(i)),"eEwWnNsS") .ne. 0) then
  2469)             write(iulog,*)'GETLONIND (2): Improperly formatted column string. ', &
  2470)                  inname,verify(tmpstr(i),"0123456789."),tmplen(i), &
  2471)                  verify(tmpstr(i)(tmplen(i):tmplen(i)),"eEwWnNsS")
  2472)             call endrun
  2473)          end if
  2474)       end do
  2475) 
  2476) ! find longitude substrings and put into temporary work space tmpstr(1:2)
  2477) 
  2478)       if (verify(tmpstr(1)(tmplen(1):tmplen(1)),"eEwW").eq.0  .and. verify(tmpstr(2)(tmplen(2):tmplen(2)),"eEwW").eq.0 ) then
  2479) 
  2480)       else if (verify(tmpstr(3)(tmplen(3):tmplen(3)),"eEwW").eq.0 .and. verify(tmpstr(3)(tmplen(3):tmplen(3)),"eEwW").eq.0) then
  2481)          tmpstr(1) = tmpstr(3)
  2482)          tmplen(1)=tmplen(3)
  2483)          tmpstr(2) = tmpstr(4)
  2484)          tmplen(2)=tmplen(4)
  2485)       else
  2486)          call endrun ('GETLONIND (3): Improperly formatted column string. '//inname)
  2487)       end if
  2488) !
  2489) ! convert lon substrings to real and make sure its degrees east
  2490) !
  2491)       do i = 1,2
  2492)          read(tmpstr(i)(1:tmplen(i)-1),*) londegree(i)
  2493)          if (verify(tmpstr(i)(tmplen(i):tmplen(i)),'wW').eq.0) then
  2494)             londegree(i) = 360._r8 - londegree(i)
  2495)          end if
  2496) !
  2497) ! Make sure specified longitudes are in bounds
  2498) !
  2499)          if (londegree(i) .lt. 0._r8 .or. londegree(i) .gt. 360._r8) then
  2500)             write(iulog,*)'GETLONIND: longitude for column namelist is out of range (0 .. 360) value=',londegree(i)
  2501)             call endrun
  2502)          endif
  2503) !
  2504) ! Find closest lon index for each substring.  If just a column is specified then beginning longitude index will be the same as the
  2505) ! ending longitude index  
  2506) !
  2507)          min  = 1.e+36_r8
  2508)          do j = 1, mlon
  2509)             tmp = abs(londeg(j)-londegree(i))
  2510)             if ( tmp .le. min) then
  2511)                min       = tmp
  2512)                lonind(i) = j
  2513)             endif
  2514)          end do
  2515)       end do
  2516) !
  2517) ! output begining and ending longitude indicies. If just a column is specified then beginning longitude index
  2518) ! will be the same as the ending longitude index  
  2519) !      
  2520)       if (lonind(1) .le. lonind(2) ) then
  2521)          beglonind =lonind(1)
  2522)          endlonind =lonind(2)
  2523)       else 
  2524)          beglonind = lonind(2)
  2525)          endlonind = lonind(1)
  2526)       end if
  2527) 
  2528) 
  2529)       if (beglonind .eq. endlonind) then
  2530)          lonname = 'LON_'//trim(tmpstr(1))
  2531)       else
  2532)          if (lonind(1) .le. lonind(2) ) then
  2533)             lonname = 'LON_'//trim(tmpstr(1)) // "_to_" // trim(tmpstr(2))
  2534)          else 
  2535)             lonname = 'LON_'//trim(tmpstr(2)) // "_to_" // trim(tmpstr(1))
  2536)          end if
  2537)       end if
  2538) 
  2539) 
  2540)       return
  2541)     end subroutine getlonind

  2619)    recursive subroutine outfld (fname, field, idim, c, avg_subcol_field)
  2620)       use cam_history_buffers, only: hbuf_accum_inst, hbuf_accum_add,         &
  2621)            hbuf_accum_add00z, hbuf_accum_max, hbuf_accum_min,                 &
  2622)            hbuf_accum_addlcltime, dim_index_2d
  2623)       use subcol_utils,        only: subcol_unpack
  2624) 
  2625)       interface
  2626)          subroutine subcol_field_avg_handler(idim, field_in, c, field_out)
  2627)             use shr_kind_mod, only: r8 => shr_kind_r8
  2628)             integer,  intent(in)  :: idim
  2629)             real(r8), intent(in)  :: field_in(idim, *)
  2630)             integer,  intent(in)  :: c
  2631)             real(r8), intent(out) :: field_out(:,:)
  2632)           end subroutine subcol_field_avg_handler
  2633)       end interface
  2634) 
  2635) !
  2636) !----------------------------------------------------------------------- 
  2637) ! 
  2638) ! Purpose: Accumulate (or take min, max, etc. as appropriate) input field
  2639) !          into its history buffer for appropriate tapes
  2640) ! 
  2641) ! Method: Check 'masterlist' whether the requested field 'fname' is active
  2642) !         on one or more history tapes, and if so do the accumulation.
  2643) !         If not found, return silently.
  2644) !         subcol_field_avg_handler:
  2645) !            An interface into subcol_field_avg without creating a dependency as
  2646) !            this would cause a dependency loop. See subcol.F90
  2647) ! Note: We cannot know a priori if field is a grid average field or a subcolumn
  2648) !       field because many fields passed to outfld are defined on ncol rather
  2649) !       than pcols or psetcols. Therefore, we use the avg_subcol_field input
  2650) !       to determine whether to average the field input before accumulation.
  2651) !       NB: If output is on a subcolumn grid (requested in addfle), it is
  2652) !           an error to use avg_subcol_field. A subcolumn field is assumed and
  2653) !           subcol_unpack is called before accumulation.
  2654) ! 
  2655) ! Author: CCM Core Group
  2656) ! 
  2657) !-----------------------------------------------------------------------
  2658) !
  2659) ! Arguments
  2660) !
  2661)       character(len=*),  intent(in) :: fname ! Field name--should be 8 chars
  2662) 
  2663)       ! For structured grids, idim is the local longitude dimension.
  2664)       ! For unstructured grids, idim is the local column dimension
  2665)       ! For phys_decomp, it should be pcols or pcols*psubcols
  2666)       integer,           intent(in) :: idim
  2667)       real(r8),          intent(in) :: field(idim,*) ! Field values
  2668)       integer,           intent(in) :: c     ! chunk (physics) or latitude (dynamics) index
  2669)       logical, optional, intent(in) :: avg_subcol_field
  2670) 
  2671) !
  2672) ! Local variables
  2673) !
  2674)       integer               :: t, f           ! tape, field indices
  2675)       integer               :: begdim1        ! on-node dim1 start index
  2676)       integer               :: enddim1        ! on-node dim1 end index
  2677)       integer               :: begdim2        ! on-node dim2 start index
  2678)       integer               :: enddim2        ! on-node dim2 end index
  2679)       integer               :: endi           ! ending longitude index (reduced grid)
  2680)       integer               :: i, j           ! Loop indices
  2681) 
  2682)       character*1           :: avgflag        ! averaging flag
  2683)       
  2684)       real(r8), pointer     :: hbuf(:,:)      ! history buffer
  2685)       integer,  pointer     :: nacs(:)        ! accumulation counter
  2686)       type (dim_index_2d)   :: dimind         ! 2-D dimension index
  2687)       logical               :: flag_xyfill    ! non-applicable xy points flagged with fillvalue
  2688)       real(r8)              :: fillvalue
  2689)       real(r8), allocatable :: afield(:,:)    ! Averaged field values
  2690)       real(r8), allocatable :: ufield(:,:,:)  ! Unpacked field values
  2691)       integer               :: ff             ! masterlist index pointer
  2692)       integer               :: ierr, ncol
  2693)       logical               :: found
  2694)       logical               :: avg_subcols    ! average subcols before accum
  2695) !-----------------------------------------------------------------------
  2696) 
  2697)       call get_field_properties (fname, found, tape_out=tape, ff_out=ff,      &
  2698)            flag_xyfill_out=flag_xyfill, fillvalue_out=fillvalue,              &
  2699)            begdim1_out=begdim1, enddim1_out=enddim1,                          &
  2700)            begdim2_out=begdim2, enddim2_out=enddim2)
  2701) 
  2702)       ! If this field is not active, return now
  2703)       if (.not. found) then
  2704)         return
  2705)       end if
  2706) 
  2707) !
  2708) ! Note, the field may be on any or all of the history files (primary
  2709) ! and auxiliary).
  2710) !
  2711) !      write(iulog,*)'fname_loc=',fname_loc
  2712)       do 40 t = 1, ptapes
  2713)          if ( .not. masterlist(ff)%thisentry%actflag(t)) cycle
  2714)          f = masterlist(ff)%thisentry%htapeindx(t)
  2715) !
  2716) ! Update history buffer
  2717) !
  2718)          avgflag = tape(t)%hlist(f)%avgflag
  2719) 
  2720)          nacs   => tape(t)%hlist(f)%nacs(:,c)
  2721) 
  2722)          
  2723)          hbuf => tape(t)%hlist(f)%hbuf(:,:,c)
  2724) 
  2725)          if(associated(tape(t)%hlist(f)%field%colperdim3)) then         
  2726)             endi    = tape(t)%hlist(f)%field%colperdim3(c)
  2727)          else
  2728)             endi = enddim1 - begdim1 + 1
  2729)          end if
  2730)          dimind = dim_index_2d (1,endi,begdim2,enddim2)
  2731) 
  2732)          ! See notes above about validity of avg_subcol_field
  2733)          if (tape(t)%hlist(f)%field%is_subcol) then
  2734)             if (present(avg_subcol_field)) then
  2735)               call endrun('OUTFLD: Cannot average '//trim(fname)//', subcolumn output was requested in addfld')
  2736)             end if
  2737)             avg_subcols = .false.
  2738)          else if (tape(t)%hlist(f)%field%decomp_type == phys_decomp) then
  2739)             if (present(avg_subcol_field)) then
  2740)                avg_subcols = avg_subcol_field
  2741)             else
  2742)                avg_subcols = .false.
  2743)             end if
  2744)          else ! Any dynamics decomposition
  2745)             if (present(avg_subcol_field)) then
  2746)                call endrun('OUTFLD: avg_subcol_field only valid for phys_decomp')
  2747)             else
  2748)                avg_subcols = .false.
  2749)             end if
  2750)          end if
  2751) 
  2752)          if (avg_subcols) then
  2753)             allocate(afield(pcols, begdim2:enddim2))
  2754)             call subcol_field_avg_handler(idim, field, c, afield)
  2755)             ! Hack! Avoid duplicating select statement below
  2756)             call outfld(fname, afield, pcols, c)
  2757)             deallocate(afield)
  2758)          else if (tape(t)%hlist(f)%field%is_subcol) then
  2759)             ! We have to assume that using mdimnames (e.g., psubcols) is
  2760)             ! incompatible with the begdimx, enddimx usage (checked in addfld)
  2761)             ! Since psubcols is included in levels, take that out
  2762)             endi = (enddim2 - begdim2 + 1) / psubcols
  2763)             allocate(ufield(pcols, psubcols, endi))
  2764)             allocate(afield(pcols*psubcols, endi))
  2765)             do j = 1, endi
  2766)                do i = 1, idim
  2767)                   afield(i, j) = field(i, j)
  2768)                end do
  2769)             end do
  2770)             ! Initialize unused aray locations.
  2771)             if (idim < pcols*psubcols) then
  2772)               if (flag_xyfill) then
  2773)                 afield(idim+1:pcols*psubcols, :) = fillvalue
  2774)               else
  2775)                 afield(idim+1:pcols*psubcols, :) = 0.0_r8
  2776)               end if
  2777)             end if
  2778)             if (flag_xyfill) then
  2779)               call subcol_unpack(c, afield, ufield, fillvalue)
  2780)             else
  2781)               call subcol_unpack(c, afield, ufield)
  2782)             end if
  2783)             deallocate(afield)
  2784)             select case (avgflag)
  2785) 
  2786)             case ('I') ! Instantaneous
  2787)                call hbuf_accum_inst(hbuf, ufield, nacs, dimind, pcols,        &
  2788)                     flag_xyfill, fillvalue)
  2789) 
  2790)             case ('A') ! Time average
  2791)                call hbuf_accum_add(hbuf, ufield, nacs, dimind, pcols,         &
  2792)                     flag_xyfill, fillvalue)
  2793) 
  2794)             case ('B') ! Time average only 00z values
  2795)                call hbuf_accum_add00z(hbuf, ufield, nacs, dimind, pcols,      &
  2796)                     flag_xyfill, fillvalue)
  2797) 
  2798)             case ('X') ! Maximum over time
  2799)                call hbuf_accum_max (hbuf, ufield, nacs, dimind, pcols,        &
  2800)                     flag_xyfill, fillvalue)
  2801) 
  2802)             case ('M') ! Minimum over time
  2803)                call hbuf_accum_min(hbuf, ufield, nacs, dimind, pcols,         &
  2804)                     flag_xyfill, fillvalue)
  2805) 
  2806)             case ('L')
  2807)                call hbuf_accum_addlcltime(hbuf, field, nacs, dimind, pcols,   &
  2808)                     flag_xyfill, fillvalue, c,                                &
  2809)                     tape(t)%hlist(f)%field%decomp_type,                       &
  2810)                     lcltod_start(t), lcltod_stop(t))
  2811) 
  2812)             case default
  2813)                call endrun ('OUTFLD: invalid avgflag='//avgflag)
  2814) 
  2815)             end select
  2816)             deallocate(ufield)
  2817)          else
  2818)             select case (avgflag)
  2819) 
  2820)             case ('I') ! Instantaneous
  2821)                call hbuf_accum_inst(hbuf, field, nacs, dimind, idim,          &
  2822)                     flag_xyfill, fillvalue)
  2823) 
  2824)             case ('A') ! Time average
  2825)                call hbuf_accum_add(hbuf, field, nacs, dimind, idim,           &
  2826)                     flag_xyfill, fillvalue)
  2827) 
  2828)             case ('B') ! Time average only 00z values
  2829)                call hbuf_accum_add00z(hbuf, field, nacs, dimind, idim,        &
  2830)                     flag_xyfill, fillvalue)
  2831) 
  2832)             case ('X') ! Maximum over time
  2833)                call hbuf_accum_max (hbuf, field, nacs, dimind, idim,          &
  2834)                     flag_xyfill, fillvalue)
  2835) 
  2836)             case ('M') ! Minimum over time
  2837)                call hbuf_accum_min(hbuf, field, nacs, dimind, idim,           &
  2838)                     flag_xyfill, fillvalue)
  2839) 
  2840)             case ('L')
  2841)                call hbuf_accum_addlcltime(hbuf, field, nacs, dimind, idim,    &
  2842)                     flag_xyfill, fillvalue, c,                                &
  2843)                     tape(t)%hlist(f)%field%decomp_type,                       &
  2844)                     lcltod_start(t), lcltod_stop(t))
  2845) 
  2846)             case default
  2847)                call endrun ('OUTFLD: invalid avgflag='//avgflag)
  2848) 
  2849)             end select
  2850)          end if
  2851) 40    continue
  2852) 
  2853)       return
  2854)    end subroutine outfld

  4191)                   where (tape(t)%hlist(f)%nacs(ib:ie,c) /= 0)
  4192)                      tape(t)%hlist(f)%hbuf(ib:ie,k,c) = &
  4193)                           tape(t)%hlist(f)%hbuf(ib:ie,k,c) &
  4194)                           / tape(t)%hlist(f)%nacs(ib:ie,c)

  4200)                        tape(t)%hlist(f)%hbuf(ib:ie,k,c) &
  4201)                        / tape(t)%hlist(f)%nacs(1,c)

  4278) !

  4280) !

  5078)    subroutine wrapup (rstwr, nlend)
  5079) !
  5080) !-----------------------------------------------------------------------
  5081) !
  5082) ! Purpose: 
  5083) ! Close history files.
  5084) ! 
  5085) ! Method: 
  5086) ! This routine will close any full hist. files
  5087) ! or any hist. file that has data on it when restart files are being 
  5088) ! written.
  5089) ! If a partially full history file was disposed (for restart 
  5090) ! purposes), then wrapup will open that unit back up and position 
  5091) ! it for appending new data. 
  5092) !
  5093) ! Original version: CCM2
  5094) !
  5095) !-----------------------------------------------------------------------
  5096) !
  5097)       use pio, only : pio_file_is_open
  5098)       use shr_kind_mod,  only: r8 => shr_kind_r8

  5101)       use time_manager,  only: get_nstep, get_curr_date, get_curr_time
  5102)       use cam_pio_utils, only: cam_pio_openfile
  5103) 
  5104) !
  5105) ! Input arguments
  5106) !
  5107)       logical, intent(in) :: rstwr   ! true => restart files are written this timestep
  5108)       logical, intent(in) :: nlend   ! Flag if time to end
  5109) 
  5110) !
  5111) ! Local workspace
  5112) !
  5113)       integer :: nstep          ! current timestep number
  5114)       integer :: ncdate         ! current date in integer format [yyyymmdd]
  5115)       integer :: ncsec          ! time of day relative to current date [seconds]
  5116)       integer :: ndcur          ! days component of current time
  5117)       integer :: nscur          ! seconds component of current time
  5118)       integer :: yr, mon, day   ! year, month, day components of a date
  5119)

  5121)       logical hdispose(ptapes)  ! Primary file disposed
  5122)       logical lhdisp            ! true => history file is disposed
  5123)       logical lhfill            ! true => history file is full
  5124) 
  5125)       integer t                 ! History file number
  5126)       integer f
  5127)       integer :: ierr
  5128)       real(r8) tday             ! Model day number for printout
  5129) !-----------------------------------------------------------------------
  5130) 
  5131)       tape => history_tape
  5132) 
  5133)       nstep = get_nstep()
  5134)       call get_curr_date(yr, mon, day, ncsec)
  5135)       ncdate = yr*10000 + mon*100 + day
  5136)       call get_curr_time(ndcur, nscur)
  5137) !
  5138) !-----------------------------------------------------------------------
  5139) ! Dispose history files.
  5140) !-----------------------------------------------------------------------
  5141) !
  5142) ! Begin loop over ptapes (the no. of declared history files - primary
  5143) ! and auxiliary).  This loop disposes a history file to Mass Store
  5144) ! when appropriate.
  5145) !
  5146)       do t=1,ptapes
  5147)          if (nflds(t) == 0) cycle
  5148) 
  5149)          hdispose(t) = .false.
  5150)          lfill(t) = .false.
  5151) !
  5152) ! Find out if file is full
  5153) !
  5154)          if (hstwr(t) .and. nfils(t) >= mfilt(t)) then
  5155)             lfill(t) = .true.
  5156)          endif
  5157) !
  5158) ! Dispose history file if 
  5159) !    1) file is filled or 
  5160) !    2) this is the end of run and file has data on it or
  5161) !    3) restarts are being put out and history file has data on it
  5162) !
  5163)          if (lfill(t) .or. (nlend .and. nfils(t) >= 1) .or. (rstwr .and. nfils(t) >= 1)) then
  5164) !
  5165) ! Dispose history file
  5166) !
  5167)             hdispose(t) = .true.
  5168) !
  5169) ! Is this the 0 timestep data of a monthly run?
  5170) ! If so, just close primary unit do not dispose. 
  5171) !
  5172)             if (masterproc) write(iulog,*)'WRAPUP: nf_close(',t,')=',trim(nhfil(t))

  5175)                   do f=1,nflds(t)
  5176)                      if (associated(tape(t)%hlist(f)%varid)) then
  5177)                         deallocate(tape(t)%hlist(f)%varid)
  5178)                         nullify(tape(t)%hlist(f)%varid)
  5179)                      end if
  5180)                   end do
  5181)                end if
  5182)                call PIO_CloseFile (tape(t)%File)
  5183)             end if
  5184)             if (nhtfrq(t) /= 0 .or. nstep > 0) then
  5185) 
  5186) ! 
  5187) ! Print information concerning model output.
  5188) ! Model day number = iteration number of history file data * delta-t / (seconds per day)
  5189) ! 
  5190)                tday = ndcur + nscur/86400._r8
  5191)                if(masterproc) then
  5192)                   if (t==1) then
  5193)                      write(iulog,*)'   Primary history file'
  5194)                   else
  5195)                      write(iulog,*)'   Auxiliary history file number ', t-1
  5196)                   end if
  5197)                   write(iulog,9003)nstep,nfils(t),tday
  5198)                   write(iulog,9004)
  5199)                end if
  5200)                   !                      
  5201) ! Auxilary files may have been closed and saved off without being full. 
  5202) ! We must reopen the files and position them for more data.
  5203) ! Must position auxiliary files if not full
  5204) !              
  5205)                if (.not.nlend .and. .not.lfill(t)) then
  5206)                   call cam_PIO_openfile (tape(t)%File, nhfil(t), PIO_WRITE)
  5207)                   call h_inquire(t)
  5208)                end if
  5209)             endif                 ! if 0 timestep of montly run****
  5210)          end if                      ! if time dispose history fiels***   
  5211)       end do                         ! do ptapes
  5212) !
  5213) ! Reset number of files on each history tape
  5214) !
  5215)       do t=1,ptapes
  5216)          if (nflds(t) == 0) cycle
  5217)          lhfill = hstwr(t) .and. nfils(t) >= mfilt(t)
  5218)          lhdisp = lhfill .or. (nlend .and. nfils(t) >= 1) .or. &
  5219)               (rstwr .and. nfils(t) >= 1)
  5220)          if (lhfill.and.lhdisp) then
  5221)             nfils(t) = 0
  5222)          endif
  5223)       end do
  5224)       return
  5225) 9003  format('    Output at NSTEP     = ',i10,/, &
  5226)              '    Number of time samples on this file = ',i10,/, &
  5227)              '    Model Day           = ',f10.2)
  5228) 9004  format('---------------------------------------')
  5229)    end subroutine wrapup

  5347) !-----------------------------------------------------------------------
  5348) !
  5349) ! Purpose: Build primary and overflow hash tables for outfld processing.
  5350) !
  5351) ! Steps:
