   453)   pure logical function pbuf_do_deallocate(hdr, persistence, col_type) result(rval)
   454)     type(physics_buffer_hdr), pointer            :: hdr
   455)     character(len=*),                 intent(in) :: persistence
   456)     integer,                          intent(in) :: col_type
   457) 
   458)     if (persistence .eq. 'physpkg') then
   459)       if (hdr%is_copy(col_type)) then
   460)         ! If this is a copied field, it is always deallocated
   461)         rval = .true.
   462)       else if (global_allocate_all) then
   463)         ! This is an all-global run so no deallocation
   464)         rval = .false.
   465)       else if(hdr%persistence == persistence_physpkg) then
   466)         ! This pbuf is a physpkg type, deallocate
   467)         rval = .true.
   468)       else
   469)         ! This pbuf is global, do not deallocate
   470)         rval = .false.
   471)       end if
   472)     else
   473)       ! We are doing a global deallocate, everything must go!
   474)       rval = .true.
   475)     end if

   508)   ! Find a pbuf header pointer based on the name input
   509)   ! Automatically tacks on an extra header if <name> is not found.
   510)   subroutine find_pbuf_header(name, index, bufptr)
   511)     character(len=*),                 intent(in)  :: name
   512)     type(physics_buffer_hdr), pointer             :: bufptr
   513)     integer,                          intent(out) :: index
   514) 
   515)     ! Local Variables
   516)     logical                                       :: buf_found
   517) 
   518)     if(.not. associated(hdrbuffertop)) then
   519)       ! This is the very first pbuf, allocate and set
   520)       allocate(hdrbuffertop)
   521)       currentpbufflds = 1
   522)       hdrbuffertop%name = name
   523)     end if
   524) 
   525)     bufptr=>hdrbuffertop
   526)     buf_found = .false.
   527)     index = 1
   528)     do while(.not. buf_found)
   529)       if(trim(bufptr%name) == trim(name)) then
   530)         buf_found = .true.
   531)       else
   532)         if (associated(bufptr%nexthdr)) then
   533)           bufptr=>bufptr%nexthdr
   534)           index = index + 1
   535)         else
   536)           ! We ran off the end of the buffers, make a new one for <name>
   537)           ! Sanity check, we should have checked exactly currentpbufflds
   538)           if (index /= currentpbufflds) then
   539)             call endrun("find_pbuf_header: currentpbufflds indexing off")
   540)           end if
   541)           currentpbufflds = currentpbufflds + 1
   542)           index = currentpbufflds
   543)           allocate(bufptr%nexthdr)
   544)           bufptr=>bufptr%nexthdr
   545)           bufptr%name = trim(name)
   546)           buf_found = .true.
   547)         end if
   548)       end if
   549)     end do
   550)   end subroutine find_pbuf_header

   886)             call buffer_get_field_ptr(pbuf(index)%bfg(col_type_grid),           &

   894)                  gfield)

   903)         end if
   904)       end if
   905)     end if

   910)          errcode = -1
   911)        else
   912)          call endrun('pbuf_get_field: field not allocated for '//trim(pbuf(index)%hdr%name))
