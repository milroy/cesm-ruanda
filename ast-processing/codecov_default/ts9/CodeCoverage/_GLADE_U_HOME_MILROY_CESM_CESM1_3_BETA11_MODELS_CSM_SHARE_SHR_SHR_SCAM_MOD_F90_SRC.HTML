    72) ! !IROUTINE: shr_scam_getCloseLatLonNC
    73) !
    74) ! !DESCRIPTION:
    75) !    routine to search in netcdf file and return lat and lon point/index closest to target point
    76) !
    77) ! !REVISION HISTORY:
    78) !     2010 Nov 05 - E. Kluzek ---- Use is_latlon/get_close/get_latlonindices routines
    79) !     2007 Aug 29 - J. Truesdale - first version
    80) !
    81) ! !INTERFACE: ------------------------------------------------------------------
    82) 
    83) subroutine shr_scam_getCloseLatLonNC(ncid, targetLat,  targetLon, closeLat, closeLon, &
    84)                                      closeLatIdx, closeLonIdx, found, rc)
    85) ! !USES:
    86)    use netcdf
    87)    use shr_ncread_mod, only: shr_ncread_handleErr
    88)    implicit none
    89) 
    90) ! !INPUT/OUTPUT PARAMETERS:
    91) 
    92)    integer(IN),intent(in)  :: ncid         ! netcdf id
    93)    real   (R8),intent(in)  :: targetLat    ! find closest latitude to this point
    94)    real   (R8),intent(in)  :: targetLon    ! find closest longitude to this point
    95)    real   (R8),intent(out) :: closeLat     ! returned close lat
    96)    real   (R8),intent(out) :: closeLon     ! returned close lon
    97)    integer(IN),intent(out) :: closeLatIdx  ! index of returned lat point
    98)    integer(IN),intent(out) :: closeLonIdx  ! index of returned lon point
    99)    logical, optional, intent(out) :: found ! if found answer (will abort if found NOT sent and
   100)                                            ! it couldn't find the lat/lon dimensions)
   101)    integer, optional, intent(out)   :: rc  ! Return code
   102) 
   103) !EOP
   104) 
   105) 
   106)    !----- local variables -----
   107)    real   (R8),allocatable          :: lats(:),lons(:)
   108)    integer(IN)                      :: rcode   ! netCDF routine return code
   109)    integer(IN)                      ::  i
   110)    integer(IN)                      ::  len
   111)    integer(IN)                      ::  latlen
   112)    integer(IN)                      ::  lonlen
   113)    integer(IN)                      ::  ndims
   114)    integer(IN)                      ::  nlatdims
   115)    integer(IN)                      ::  nlondims
   116)    integer(IN)                      ::  nvars
   117)    integer(IN)                      ::  nvarid
   118)    integer(IN)                      ::  ndimid
   119)    integer(IN)                      ::  strt(nf90_max_var_dims),cnt(nf90_max_var_dims)
   120)    integer(IN)                      ::  nlon,nlat
   121)    integer(IN), dimension(nf90_max_var_dims) :: dimids
   122)    logical                          ::  lfound        ! local version of found
   123)    character(len=80), allocatable   ::  vars(:)
   124)    character(len=80), allocatable   ::  latdimnames(:)
   125)    character(len=80), allocatable   ::  londimnames(:)
   126)    character(*),parameter :: subname = "(shr_scam_getCloseLatLonNC) "
   127) 
   128) !-------------------------------------------------------------------------------
   129) ! Notes:
   130) !-------------------------------------------------------------------------------
   131) 
   132)    if ( present(rc) )then
   133)       rc = 0
   134)    end if
   135)    if ( present(found) )then
   136)       lfound = found
   137)    else
   138)       lfound = .false.
   139)    end if
   140)    if ( present(rc) )then
   141)       rc = 0   ! Initialize return code to something
   142)    end if
   143) 
   144)    !--- Get variable info for search ---
   145) 
   146)    rcode = nf90_inquire(ncid, nVariables=nvars)
   147)    if (rcode /= nf90_noerr) then
   148)       call shr_ncread_handleErr( rcode, subname//"ERROR from nf90_inquire" )
   149)       if ( present(rc) )then
   150)          rc = rcode
   151)          return
   152)       end if
   153)    endif
   154) 
   155)    allocate( vars(nvars) )
   156)    do nvarid = 1, nvars
   157)       rcode = nf90_inquire_variable(ncid, nvarid, vars(nvarid), ndims=ndims,dimids = dimids)
   158)       if (rcode /= nf90_noerr) then
   159)          call shr_ncread_handleErr( rcode, subname//"ERROR inquiring about variable "// &
   160)                                     trim(vars(nvarid)) )
   161)          if ( present(rc) )then
   162)             rc = rcode
   163)             return
   164)          end if
   165)       endif
   166)       !-- If latitude variable ---
   167)       if ( is_latlon( vars(nvarid), latitude=.true., varnotdim=.true. ) )then
   168)          nlatdims = ndims
   169)          allocate( latdimnames(ndims) )
   170)          do ndimid =  1,ndims
   171)             rcode = nf90_inquire_dimension(ncid, dimids(ndimid), latdimnames(ndimid), len)
   172)             if (rcode /= nf90_noerr) then
   173)                call shr_ncread_handleErr( rcode, subname// &
   174)                    "ERROR: Cant read netcdf latitude variable dimension")
   175)                if ( present(rc) )then
   176)                   rc = rcode
   177)                   return
   178)                end if
   179)             endif
   180)             !--- is this a latitude dimension  ---
   181)             if ( is_latlon( latdimnames(ndimid), latitude=.true., varnotdim=.false. ) )then
   182)                latlen = len
   183)             end if
   184)          end do
   185)       end if
   186)       !-- If longitude variable ---
   187)       if ( is_latlon( vars(nvarid), latitude=.false., varnotdim=.true. ) )then
   188)          nlondims = ndims
   189)          allocate( londimnames(ndims) )
   190)          do ndimid =  1,ndims
   191)             rcode = nf90_inquire_dimension(ncid, dimids(ndimid), londimnames(ndimid), len)
   192)             call shr_ncread_handleErr( rcode, subname &
   193)                            //"ERROR: Cant read netcdf longitude variable dimension" )
   194)             if ( rcode /= nf90_noerr .and. present(rc) )then
   195)                rc = rcode
   196)                return
   197)             end if
   198)             !--- is this a longitude dimension  ---
   199)             if ( is_latlon( londimnames(ndimid), latitude=.false., varnotdim=.false. ) )then
   200)                lonlen = len
   201)             end if
   202)          end do
   203)       end if
   204)    end do
   205) 
   206)    !--- Look for/extract lat lon coordinate variables from file ---
   207) 
   208)    nlat=0
   209)    nlon=0
   210)    nvarid=0
   211) 
   212)    !--- Loop through all variables until we find lat and lon ---
   213) 
   214)    do while (nvarid < nvars .and.(nlon.eq.0 .or. nlat.eq.0))
   215)       nvarid=nvarid+1
   216) 
   217)       !--- Get latitude ---
   218) 
   219)       if ( is_latlon( vars(nvarid), latitude=.true., varnotdim=.true. ) )then
   220) 
   221)          call get_latlonindices( latitude=.true., dimnames=latdimnames, ndims=nlatdims, &
   222)                                  nlen=latlen, strt=strt, cnt=cnt )
   223)          nlat = latlen
   224)          allocate(lats(nlat))
   225)          rcode= nf90_get_var(ncid, nvarid ,lats, start = strt, count = cnt)
   226)          call shr_ncread_handleErr( rcode, subname &
   227)                            //"ERROR: Cant read netcdf latitude" )
   228)          if ( rcode /= nf90_noerr .and. present(rc) )then
   229)             rc = rcode
   230)             return
   231)          end if
   232)       end if
   233) 
   234)       !--- Get longitude ---
   235) 
   236)       if ( is_latlon( vars(nvarid), latitude=.false., varnotdim=.true. ) )then
   237)          call get_latlonindices( latitude=.false., ndims=nlondims, dimnames=londimnames, &
   238)                                  nlen=lonlen, strt=strt, cnt=cnt )
   239)          nlon = lonlen
   240)          allocate(lons(nlon))
   241)          rcode= nf90_get_var(ncid, nvarid ,lons, start = strt, count = cnt)
   242)          call shr_ncread_handleErr( rcode, subname &
   243)                            //"ERROR: Cant read netcdf longitude" )
   244)          if ( rcode /= nf90_noerr .and. present(rc) )then
   245)             rc = rcode
   246)             return
   247)          end if
   248)       end if
   249)    end do
   250)    if ( present(found) )then
   251)       if ( nlat == 0 .or. nlon == 0 ) then
   252)          write(s_logunit,*) subname//"WARNING: Cant find appropriate latitude or longitude coordinate variables"
   253)          found = .false.
   254)       else
   255)          call get_close( targetLon, targetLat, nlon, lons, nlat, lats, closelonidx, closelatidx, found )
   256)          if ( found )then
   257)             closelon=lons(closelonidx)
   258)             closelat=lats(closelatidx)
   259)          end if
   260)       end if
   261)    else
   262) 
   263)       call get_close( targetLon, targetLat, nlon, lons, nlat, lats, closelonidx, closelatidx )
   264)       closelon=lons(closelonidx)
   265)       closelat=lats(closelatidx)
   266) 
   267)    end if
   268) 
   269)    if ( allocated(lats)        ) deallocate(lats)
   270)    if ( allocated(lons)        ) deallocate(lons)
   271)    if ( allocated(latdimnames) ) deallocate(latdimnames)
   272)    if ( allocated(londimnames) ) deallocate(londimnames)
   273)    if ( allocated(vars)        ) deallocate( vars )
   274) 
   275)    return
   276) 
   277) end subroutine shr_scam_getCloseLatLonNC

   295)    use netcdf
   296)    use pio
   297)    use shr_ncread_mod, only: shr_ncread_handleErr
   298)    implicit none
   299) 
   300) ! !INPUT/OUTPUT PARAMETERS:
   301) 
   302)    type(file_desc_t), intent(inout) :: pioid        ! pio file ID
   303)    real   (R8),       intent(in)    :: targetLat    ! find closest latitude to this point
   304)    real   (R8),       intent(in)    :: targetLon    ! find closest longitude to this point
   305)    real   (R8),       intent(out)   :: closeLat     ! returned close lat
   306)    real   (R8),       intent(out)   :: closeLon     ! returned close lon
   307)    integer(IN),       intent(out)   :: closeLatIdx  ! index of returned lat point
   308)    integer(IN),       intent(out)   :: closeLonIdx  ! index of returned lon point
   309)    logical, optional, intent(out)   :: found        ! if found answer (will abort if found NOT sent and
   310)                                                     ! it couldn't find the lat/lon dimensions)
   311)    integer, optional, intent(out)   :: rc           ! Return code
   312) 
   313) !EOP
   314) 
   315) 
   316)    !----- local variables -----
   317)    real   (R8),allocatable          :: lats(:),lons(:)
   318)    integer(IN)                      :: rcode   ! netCDF routine return code
   319)    integer(IN)                      ::  i
   320)    integer(IN)                      ::  len      = 0
   321)    integer(IN)                      ::  latlen   = 0
   322)    integer(IN)                      ::  lonlen   = 0
   323)    integer(IN)                      ::  ndims    = 0
   324)    integer(IN)                      ::  nlatdims = 0
   325)    integer(IN)                      ::  nlondims = 0
   326)    integer(IN)                      ::  nvars    = 0
   327)    integer(IN)                      ::  nvarid
   328)    integer(IN)                      ::  ndimid
   329)    integer(IN)                      ::  strt(nf90_max_var_dims),cnt(nf90_max_var_dims)
   330)    integer(IN)                      ::  nlon = 0, nlat = 0
   331)    logical                          ::  lfound        ! local version of found
   332)    integer(IN), dimension(nf90_max_var_dims) :: dimids
   333)    character(len=80), allocatable   ::  vars(:)
   334)    character(len=80), allocatable   ::  latdimnames(:)
   335)    character(len=80), allocatable   ::  londimnames(:)
   336)    character(*),parameter :: subname = "(shr_scam_getCloseLatLonPIO) "
   337) 
   338) !-------------------------------------------------------------------------------
   339) ! Notes:
   340) !-------------------------------------------------------------------------------
   341) 
   342)    if ( present(found) )then
   343)       lfound = found
   344)    else
   345)       lfound = .false.
   346)    end if
   347)    if ( present(rc) )then
   348)       rc = 0   ! Initialize return code to something
   349)    end if
   350)    !--- Get variable info for search ---
   351) 
   352)    call pio_seterrorhandling(pioid,PIO_BCAST_ERROR)
   353)    rcode = pio_inquire(pioid, nVariables=nvars)
   354)    if (rcode /= PIO_noerr) then
   355)       call shr_ncread_handleErr( rcode, subname//"ERROR: from PIO_inquire ")
   356)       if ( present(rc) )then
   357)          rc = rcode
   358)          return
   359)       end if
   360)    endif
   361) 
   362)    allocate( vars(nvars) )
   363)    do nvarid = 1, nvars
   364)       rcode = pio_inquire_variable(pioid, nvarid, vars(nvarid), ndims=ndims,dimids = dimids)
   365)       if (rcode /= PIO_noerr) then
   366)          write(s_logunit,*) subname//"ERROR inquiring about variable id #", nvarid
   367)          call shr_ncread_handleErr( rcode, subname//"ERROR: inquiring about variable" )
   368)          if ( present(rc) )then
   369)             rc = rcode
   370)             return
   371)          end if
   372)       endif
   373)       !-- If latitude variable ---
   374)       if ( is_latlon( vars(nvarid), latitude=.true., varnotdim=.true. ) )then
   375)          nlatdims = ndims
   376)          allocate( latdimnames(ndims) )
   377)          do ndimid =  1,ndims
   378)             rcode = pio_inquire_dimension(pioid, dimids(ndimid), latdimnames(ndimid), len)
   379)             if (rcode /= pio_noerr) then
   380)                call shr_ncread_handleErr( rcode, subname// &
   381)                                    "ERROR: Cant read netcdf latitude variable dimension")
   382)                if ( present(rc) )then
   383)                   rc = rcode
   384)                   return
   385)                end if
   386)             endif
   387)             !--- is this a latitude dimension  ---
   388)             if ( is_latlon( latdimnames(ndimid), latitude=.true., varnotdim=.false. ) )then
   389)                latlen = len
   390)             end if
   391)          end do
   392)       end if
   393)       !-- If longitude variable ---
   394)       if ( is_latlon( vars(nvarid), latitude=.false., varnotdim=.true. ) )then
   395)          nlondims = ndims
   396)          allocate( londimnames(ndims) )
   397)          do ndimid =  1,ndims
   398)             rcode = pio_inquire_dimension(pioid, dimids(ndimid), londimnames(ndimid), len)
   399)             if (rcode /= PIO_noerr) then
   400)                call shr_ncread_handleErr( rcode, subname// &
   401)                                    "ERROR: Cant read netcdf longitude variable dimension")
   402)                if ( present(rc) )then
   403)                   rc = rcode
   404)                   return
   405)                end if
   406)             endif
   407)             !--- is this a longitude dimension  ---
   408)             if ( is_latlon( londimnames(ndimid), latitude=.false., varnotdim=.false. ) )then
   409)                lonlen = len
   410)             end if
   411)          end do
   412)       end if
   413)    end do
   414) 
   415)    !--- Look for/extract lat lon coordinate variables from file ---
   416) 
   417)    nlat=0
   418)    nlon=0
   419)    nvarid=0
   420) 
   421)    !--- Loop through all variables until we find lat and lon ---
   422) 
   423)    do while (nvarid < nvars .and.(nlon.eq.0 .or. nlat.eq.0))
   424)       nvarid=nvarid+1
   425) 
   426)       !--- Get latitude ---
   427) 
   428)       if ( is_latlon( vars(nvarid), latitude=.true., varnotdim=.true. ) )then
   429) 
   430)          call get_latlonindices( latitude=.true., ndims=nlatdims, dimnames=latdimnames, &
   431)                                  nlen=latlen, strt=strt, cnt=cnt )
   432)          nlat = latlen
   433)          allocate(lats(nlat))
   434)          rcode= pio_get_var(pioid, nvarid ,strt(:nlatdims), cnt(:nlatdims), lats)
   435)          if (rcode /= PIO_noerr) then
   436)             call shr_ncread_handleErr( rcode, subname// &
   437)                                "ERROR: Cant read netcdf latitude")
   438)             if ( present(rc) )then
   439)                rc = rcode
   440)                return
   441)             end if
   442)          endif
   443)       end if
   444) 
   445)       !--- Get longitude ---
   446) 
   447)       if ( is_latlon( vars(nvarid), latitude=.false., varnotdim=.true. ) )then
   448)          call get_latlonindices( latitude=.false., ndims=nlondims, dimnames=londimnames, &
   449)                                  nlen=lonlen, strt=strt, cnt=cnt )
   450)          nlon = lonlen

   452)          rcode= pio_get_var(pioid, nvarid ,strt(:nlondims), cnt(:nlondims), lons)

   454)             call shr_ncread_handleErr( rcode, subname// &

   457)                rc = rcode
   458)                return
   459)             end if
   460)          endif
   461)       end if
   462)    end do
   463)    call pio_seterrorhandling(pioid,PIO_INTERNAL_ERROR)
   464)

   476)    else
   477)       call get_close( targetLon, targetLat, nlon, lons, nlat, lats, closelonidx, closelatidx )
   478)       closelon=lons(closelonidx)
   479)       closelat=lats(closelatidx)
   480)    end if
   481)    if ( allocated(lats) ) deallocate(lats)
   482)    if ( allocated(lons) ) deallocate(lons)
   483)    deallocate( vars )
   484) 
   485)    return
   486) end subroutine shr_scam_getCloseLatLonPIO

   864) ! !DESCRIPTION:

   870) ! !INTERFACE: ------------------------------------------------------------------

   872)                       closelatidx, found )
   873) 
   874) ! !USES:
   875) ! !INPUT/OUTPUT PARAMETERS:
   876)    implicit none
   877)    real   (R8),intent(in)   :: targetLon    ! find closest longitude to this point
   878)    real   (R8),intent(in)   :: targetLat    ! find closest latitude to this point
   879)    integer(IN),intent(in)   :: nlon         ! Number of longitudes
   880)    real   (R8),intent(in)   :: lons(nlon)   ! Longitude array
   881)    integer(IN),intent(in)   :: nlat         ! Number of latitudes
   882)    real   (R8),intent(in)   :: lats(nlat)   ! Latitude array
   883)    integer(IN),intent(out)  :: closeLatIdx  ! index of returned lat point
   884)    integer(IN),intent(out)  :: closeLonIdx  ! index of returned lon point
   885)    logical, optional, intent(out):: found   ! if found answer (will abort if found NOT sent and
   886)                                             ! it couldn't find the lat/lon dimensions)
   887) !EOP
   888) 
   889)    !----- local variables -----
   890)    real   (R8),allocatable  :: poslons(:) 
   891)    real   (R8)              :: postargetlon
   892)    character(*),parameter :: subname = "(shr_scam_getclose) "
   893) !-------------------------------------------------------------------------------
   894) ! Notes:
   895) !-------------------------------------------------------------------------------
   896)    if ( present(found) )then
   897)       found = .true.
   898)    end if
   899)

   901) 
   902)    if (nlon == 0) then
   903)       write(s_logunit,*) subname//"ERROR: Coudnt find longitude coordinate"
   904)       if ( present(found) )then
   905)          found = .false.
   906)          return
   907)       else
   908)          call shr_sys_abort( subname//"ERROR: Couldnt find a longitude coordinate variable")
   909)       end if
   910)    end if
   911)    if (nlat == 0) then
   912)       write(s_logunit,*) subname//"ERROR: Coudnt find latitude coordinate"
   913)       if ( present(found) )then
   914)          found = .false.
   915)          return
   916)       else
   917)          call shr_sys_abort( subname//"ERROR: Couldnt find a latitude coordinate variable")
   918)       end if
   919)    end if
   920)    !--- Convert target latitude to within 0-360 ---
   921)    postargetlon=mod(targetlon+360._r8,360._r8)
   922) 
   923)    !--- Make sure target latitude within globe ---
   924)    if ( targetlat < -90.0_r8 .or. targetlat > 90.0_r8 )then
   925)       write(s_logunit,*) subname//"ERROR: target latitude out of range = ", targetlat

   931)       end if
   932)    end if
   933) 
   934)    !--- convert lons array and targetlon to 0,360 ---
   935) 
   936)    allocate(poslons(nlon))

   942)    closelatidx=(MINLOC(abs(lats-targetlat),dim=1))
   943)

   945) 
   946)    deallocate(poslons)
