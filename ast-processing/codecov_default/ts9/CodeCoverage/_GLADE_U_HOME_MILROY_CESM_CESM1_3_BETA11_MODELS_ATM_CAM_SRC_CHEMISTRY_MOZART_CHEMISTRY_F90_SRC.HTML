   181)     integer  :: idx                                 ! Index for adding fields to pbuf physics buffer

   193)     integer  :: bry_ndx,br_ndx,bro_ndx,hbr_ndx,brono2_ndx,brcl_ndx,hobr_ndx

   212)     cfc12_ndx = get_spc_ndx('CFC12')

   214)     o2_1d_ndx = get_spc_ndx('O2_1D')

   221)     e_ndx     = get_spc_ndx('e')
   222)     np_ndx    = get_spc_ndx('Np')
   223)     op_ndx    = get_spc_ndx('Op')

   241)     br_ndx    = get_spc_ndx('BR')
   242)     bro_ndx   = get_spc_ndx('BRO')
   243)     hbr_ndx   = get_spc_ndx('HBR')
   244)     brono2_ndx= get_spc_ndx('BRONO2')
   245)     brcl_ndx  = get_spc_ndx('BRCL')
   246)     hobr_ndx  = get_spc_ndx('HOBR')

   249)     hf_ndx    = get_spc_ndx('HF')

   251)     !-----------------------------------------------------------------------
   252)     ! Set names of diffused variable tendencies and declare them as history variables
   253)     !-----------------------------------------------------------------------
   254)     !-----------------------------------------------------------------------
   255)     ! BAB: 2004-09-01 kludge to define a fixed ubc for water vapor
   256)     !      required because water vapor is not declared by chemistry but
   257)     !      has a fixed ubc only if chemistry is running.
   258)     !-----------------------------------------------------------------------
   259)     if (chem_is('waccm_mozart') .or. chem_is('waccm_mozart_mam3')) then
   260) 
   261)     !-----------------------------------------------------------------------------------------
   262)     !For WACCM-X, change variable cnst_fixed_ubc(1) from .true. to .false. which eliminates 
   263)     !the constant fixed upper boundary condition for all species
   264)     !-----------------------------------------------------------------------------------------
   265)       if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   266)         cnst_fixed_ubc(1) = .false.
   267)       else
   268)         cnst_fixed_ubc(1) = .true.
   269)       endif
   270)       
   271)     endif

   281)      do m = 1,gas_pcnst

   289)

   291)           qmin = 1.e-5_r8
   292)        else if ( m == o3_ndx ) then
   293)           qmin = 1.e-12_r8
   294)        else if ( m == ch4_ndx ) then
   295)           qmin = 1.e-12_r8
   296)           if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   297)             has_fixed_ubc = .false.   ! diffusive equilibrium at UB
   298)           else
   299)             has_fixed_ubc = .true.
   300)           endif
   301)        else if ( m == n2o_ndx ) then
   302)           qmin = 1.e-15_r8
   303)        else if( m == cfc11_ndx .or. m == cfc12_ndx ) then
   304)           qmin = 1.e-20_r8
   305)        else if( m == o2_1s_ndx .or. m == o2_1d_ndx ) then
   306)           ic_from_cam2 = .false.
   307)           if( m == o2_1d_ndx ) then
   308)              lng_name = 'O2(1-delta)'
   309)           else
   310)              lng_name = 'O2(1-sigma)'
   311)           end if
   312)        else if ( m==o2_ndx .or. m==n_ndx .or. m==no_ndx .or. m==h_ndx .or. m==h2_ndx .or. m==o_ndx .or. m==hf_ndx &
   313)                .or. m==f_ndx ) then
   314)          if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   315)            has_fixed_ubc = .false.   ! diffusive equilibrium at UB
   316)            if ( m == h_ndx ) has_fixed_ubflx = .true. ! fixed flux value for H at UB
   317)            if ( m == o2_ndx .or. m == o_ndx ) molectype = 'major'
   318)          else
   319)            has_fixed_ubc = .true.
   320)          endif
   321)        else if( m == e_ndx ) then
   322)           lng_name = 'electron concentration'
   323)        else if( m == np_ndx ) then
   324)           lng_name = 'N+'
   325)        else if( m == op_ndx ) then
   326)           lng_name = 'O+'
   327)        else if( m == o1d_ndx ) then
   328)           lng_name = 'O(1D)'
   329)        else if( m == n2d_ndx ) then
   330)           lng_name = 'N(2D)'
   331)        else if( m == o2p_ndx ) then
   332)           lng_name = 'O2+'
   333)        else if( m == n2p_ndx ) then
   334)           lng_name = 'N2+'
   335)        else if( m == nop_ndx ) then
   336)           lng_name = 'NO+'
   337)        else if( m == h2o_ndx ) then
   338)           map2chm(1) = m
   339)           cycle
   340)        endif
   341) 
   342)        cam_outfld=.false.
   343)        is_active = .true.
   344) 
   345)        if ( chem_is('waccm_mozart').or.chem_is('waccm_mozart_mam3').or. &

   347)           mixtype='wet'
   348)        else
   349)           mixtype='dry'
   350)        endif
   351) 
   352)        islvd = slvd_index(solsym(m))
   353) 
   354)        if ( islvd > 0 ) then
   355)           short_lived_map(islvd) = m
   356)        else
   357)           call cnst_add( solsym(m), adv_mass(m), cptmp, qmin, n, readiv=ic_from_cam2, cam_outfld=cam_outfld, &
   358)                          mixtype=mixtype, molectype=molectype, fixed_ubc=has_fixed_ubc, fixed_ubflx=has_fixed_ubflx, &
   359)                          longname=trim(lng_name) )
   360) 
   361)           if( imozart == -1 ) then
   362)              imozart = n
   363)           end if
   364)           map2chm(n) = m
   365)        endif
   366) 
   367)     end do

   369)     call register_short_lived_species()
   370)     call register_cfc11star()

   374)        call aurora_register()
   375)     endif
   376)

   379)
