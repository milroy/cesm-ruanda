   120) !

   123) subroutine docn_comp_init( EClock, cdata, x2o, o2x, NLFilename )
   124)     use shr_pio_mod, only : shr_pio_getiosys, shr_pio_getiotype
   125)     implicit none
   126) 
   127) ! !INPUT/OUTPUT PARAMETERS:
   128) 
   129)     type(ESMF_Clock)            , intent(in)    :: EClock
   130)     type(seq_cdata)             , intent(inout) :: cdata
   131)     type(mct_aVect)             , intent(inout) :: x2o, o2x
   132)     character(len=*), optional  , intent(in)    :: NLFilename ! Namelist filename
   133) 
   134) !EOP
   135) 
   136)     !--- local variables ---
   137)     integer(IN)   :: n,k         ! generic counters
   138)     integer(IN)   :: ierr        ! error code
   139)     integer(IN)   :: COMPID      ! comp id
   140)     integer(IN)   :: gsize       ! global size
   141)     integer(IN)   :: lsize     ! local size
   142)     integer(IN)   :: shrlogunit, shrloglev ! original log unit and level
   143)     integer(IN)   :: nunit       ! unit number
   144)     integer(IN)   :: kmask       ! field reference
   145)     logical       :: ocn_present    ! flag
   146)     logical       :: ocn_prognostic ! flag
   147)     logical       :: ocnrof_prognostic  ! flag
   148)     character(CL) :: calendar    ! model calendar
   149) 
   150)     type(seq_infodata_type), pointer :: infodata
   151)     type(mct_gsMap)        , pointer :: gsmap
   152)     type(mct_gGrid)        , pointer :: ggrid
   153) 
   154)     character(CL) :: filePath    ! generic file path
   155)     character(CL) :: fileName    ! generic file name
   156)     character(CS) :: timeName    ! domain file: time variable name
   157)     character(CS) ::  lonName    ! domain file: lon  variable name
   158)     character(CS) ::  latName    ! domain file: lat  variable name
   159)     character(CS) :: maskName    ! domain file: mask variable name
   160)     character(CS) :: areaName    ! domain file: area variable name
   161) 
   162)     integer(IN)   :: yearFirst   ! first year to use in data stream
   163)     integer(IN)   :: yearLast    ! last  year to use in data stream
   164)     integer(IN)   :: yearAlign   ! data year that aligns with yearFirst
   165) 
   166)     character(CL) :: ocn_in      ! dshr ocn namelist
   167)     character(CL) :: decomp      ! decomp strategy
   168)     character(CL) :: rest_file   ! restart filename
   169)     character(CL) :: rest_file_strm   ! restart filename for stream
   170)     character(CL) :: restfilm    ! restart filename for namelist
   171)     character(CL) :: restfils    ! restart filename for stream for namelist
   172)     logical       :: force_prognostic_true ! if true set prognostic true
   173) 
   174)     logical       :: exists      ! file existance
   175)     integer(IN)   :: nu          ! unit number
   176) 
   177)     !----- define namelist -----
   178)     namelist / docn_nml / &
   179)         ocn_in, decomp, restfilm, restfils, &
   180)         force_prognostic_true
   181) 
   182)     !--- formats ---
   183)     character(*), parameter :: F00   = "('(docn_comp_init) ',8a)"
   184)     character(*), parameter :: F0L   = "('(docn_comp_init) ',a, l2)"
   185)     character(*), parameter :: F01   = "('(docn_comp_init) ',a,5i8)"
   186)     character(*), parameter :: F02   = "('(docn_comp_init) ',a,4es13.6)"
   187)     character(*), parameter :: F03   = "('(docn_comp_init) ',a,i8,a)"
   188)     character(*), parameter :: F04   = "('(docn_comp_init) ',2a,2i8,'s')"
   189)     character(*), parameter :: F05   = "('(docn_comp_init) ',a,2f10.4)"
   190)     character(*), parameter :: F90   = "('(docn_comp_init) ',73('='))"
   191)     character(*), parameter :: F91   = "('(docn_comp_init) ',73('-'))"
   192)     character(*), parameter :: subName = "(docn_comp_init) "
   193) !-------------------------------------------------------------------------------
   194) 
   195) 
   196)     call t_startf('DOCN_INIT')
   197) 
   198)     firstcall = .true.
   199) 
   200)     ! Set cdata pointers
   201) 
   202)     call seq_cdata_setptrs(cdata, ID=COMPID, mpicom=mpicom, &
   203)          gsMap=gsmap, dom=ggrid, infodata=infodata)
   204) 
   205)     ! Determine communicator groups and sizes
   206) 
   207)     call mpi_comm_rank(mpicom, my_task, ierr)
   208)     call mpi_comm_size(mpicom, npes, ierr)
   209) 
   210)     inst_name   = seq_comm_name(COMPID)
   211)     inst_index  = seq_comm_inst(COMPID)
   212)     inst_suffix = seq_comm_suffix(COMPID)
   213) 
   214)     !--- open log file ---
   215)     logUnit = 6
   216)     if (my_task == master_task) then
   217)        inquire(FILE='ocn_modelio.nml'//trim(inst_suffix), exist=exists)
   218)        if (exists) then
   219)          logUnit = shr_file_getUnit()
   220)          call shr_file_setIO('ocn_modelio.nml'//trim(inst_suffix),logUnit)
   221)        end if
   222)     endif
   223) 
   224)     !----------------------------------------------------------------------------
   225)     ! Reset shr logging to my log file
   226)     !----------------------------------------------------------------------------
   227)     call shr_file_getLogUnit (shrlogunit)
   228)     call shr_file_getLogLevel(shrloglev)
   229)     call shr_file_setLogUnit (logUnit)
   230) 
   231)     !----------------------------------------------------------------------------
   232)     ! Set a Few Defaults
   233)     !----------------------------------------------------------------------------
   234) 
   235)     call seq_infodata_getData(infodata,single_column=scmMode, &
   236)    &                          scmlat=scmlat, scmlon=scmLon)
   237) 
   238)     ocn_present = .false.
   239)     ocn_prognostic = .false.
   240)     ocnrof_prognostic = .false.
   241)     call seq_infodata_GetData(infodata,read_restart=read_restart)
   242) 
   243)     !----------------------------------------------------------------------------
   244)     ! Read docn_in
   245)     !----------------------------------------------------------------------------
   246) 
   247)     call t_startf('docn_readnml')
   248) 
   249)     filename = "docn_in"//trim(inst_suffix)
   250)     ocn_in = "unset"
   251)     decomp = "1d"
   252)     restfilm = trim(nullstr)
   253)     restfils = trim(nullstr)
   254)     force_prognostic_true = .false.
   255)     if (my_task == master_task) then
   256)        nunit = shr_file_getUnit() ! get unused unit number
   257)        open (nunit,file=trim(filename),status="old",action="read")
   258)        read (nunit,nml=docn_nml,iostat=ierr)
   259)        close(nunit)
   260)        call shr_file_freeUnit(nunit)
   261)        if (ierr > 0) then
   262)           write(logunit,F01) 'ERROR: reading input namelist, '//trim(filename)//' iostat=',ierr
   263)           call shr_sys_abort(subName//': namelist read error '//trim(filename))
   264)        end if
   265)        write(logunit,F00)' ocn_in = ',trim(ocn_in)
   266)        write(logunit,F00)' decomp = ',trim(decomp)
   267)        write(logunit,F00)' restfilm = ',trim(restfilm)
   268)        write(logunit,F00)' restfils = ',trim(restfils)
   269)        write(logunit,F0L)' force_prognostic_true = ',force_prognostic_true
   270)     endif
   271)     call shr_mpi_bcast(ocn_in,mpicom,'ocn_in')
   272)     call shr_mpi_bcast(decomp,mpicom,'decomp')
   273)     call shr_mpi_bcast(restfilm,mpicom,'restfilm')
   274)     call shr_mpi_bcast(restfils,mpicom,'restfils')
   275)     call shr_mpi_bcast(force_prognostic_true,mpicom,'force_prognostic_true')
   276)  
   277)     rest_file = trim(restfilm)
   278)     rest_file_strm = trim(restfils)
   279)     if (force_prognostic_true) then
   280)        ocn_present    = .true.
   281)        ocn_prognostic = .true.
   282)     endif
   283) 
   284)     !----------------------------------------------------------------------------
   285)     ! Read dshr namelist
   286)     !----------------------------------------------------------------------------
   287) 
   288)     call shr_strdata_readnml(SDOCN,trim(ocn_in),mpicom=mpicom)
   289) 
   290)     !----------------------------------------------------------------------------
   291)     ! Validate mode
   292)     !----------------------------------------------------------------------------
   293) 
   294)     ocn_mode = trim(SDOCN%dataMode)
   295) 
   296)     ! check that we know how to handle the mode
   297) 
   298)     if (trim(ocn_mode) == 'NULL' .or. &
   299)         trim(ocn_mode) == 'SSTDATA' .or. &
   300)         trim(ocn_mode) == 'COPYALL' .or. &
   301)         trim(ocn_mode) == 'SOM') then
   302)       if (my_task == master_task) &
   303)          write(logunit,F00) ' ocn mode = ',trim(ocn_mode)
   304)     else
   305)       write(logunit,F00) ' ERROR illegal ocn mode = ',trim(ocn_mode)
   306)       call shr_sys_abort()
   307)     endif
   308) 
   309)     call t_stopf('docn_readnml')
   310) 
   311)     !----------------------------------------------------------------------------
   312)     ! Initialize datasets
   313)     !----------------------------------------------------------------------------
   314) 
   315)     call t_startf('docn_strdata_init')
   316) 
   317)     if (trim(ocn_mode) /= 'NULL') then
   318)        ocn_present = .true.
   319)        call seq_timemgr_EClockGetData( EClock, calendar=calendar )
   320)        iosystem => shr_pio_getiosys(trim(inst_name))
   321)        
   322)        call shr_strdata_pioinit(SDOCN, iosystem, shr_pio_getiotype(trim(inst_name)))
   323) 
   324)        if (scmmode) then
   325)           if (my_task == master_task) &
   326)              write(logunit,F05) ' scm lon lat = ',scmlon,scmlat
   327)           call shr_strdata_init(SDOCN,mpicom,compid,name='ocn', &
   328)                       scmmode=scmmode,scmlon=scmlon,scmlat=scmlat, &
   329)                       calendar=calendar)
   330)        else
   331)           call shr_strdata_init(SDOCN,mpicom,compid,name='ocn', &
   332)                       calendar=calendar)
   333)        endif
   334)     endif
   335) 
   336)     if (trim(ocn_mode) == 'SOM') then
   337)        ocn_prognostic = .true.
   338)     endif
   339) 
   340)     if (my_task == master_task) then
   341)        call shr_strdata_print(SDOCN,'SDOCN data')
   342)     endif
   343) 
   344)     call t_stopf('docn_strdata_init')
   345) 
   346)     !----------------------------------------------------------------------------
   347)     ! Set flag to specify data components
   348)     !----------------------------------------------------------------------------
   349) 
   350)     call seq_infodata_PutData(infodata, ocnrof_prognostic=ocnrof_prognostic, &
   351)       ocn_present=ocn_present, ocn_prognostic=ocn_prognostic, &
   352)       ocn_nx=SDOCN%nxg, ocn_ny=SDOCN%nyg )
   353) 
   354)     !----------------------------------------------------------------------------
   355)     ! Initialize MCT global seg map, 1d decomp
   356)     !----------------------------------------------------------------------------
   357) 
   358)     call t_startf('docn_initgsmaps')
   359)     if (my_task == master_task) write(logunit,F00) ' initialize gsmaps'
   360)     call shr_sys_flush(logunit)
   361) 
   362)     call shr_dmodel_gsmapcreate(gsmap,SDOCN%nxg*SDOCN%nyg,compid,mpicom,decomp)
   363)     lsize = mct_gsmap_lsize(gsmap,mpicom)
   364) 
   365)     if (ocn_present) then
   366)        call mct_rearr_init(SDOCN%gsmap,gsmap,mpicom,rearr)
   367)     endif
   368) 
   369)     call t_stopf('docn_initgsmaps')
   370) 
   371)     !----------------------------------------------------------------------------
   372)     ! Initialize MCT domain
   373)     !----------------------------------------------------------------------------
   374) 
   375)     call t_startf('docn_initmctdom')
   376)     if (my_task == master_task) write(logunit,F00) 'copy domains'
   377)     call shr_sys_flush(logunit)
   378) 
   379)     if (ocn_present) call shr_dmodel_rearrGGrid(SDOCN%grid, ggrid, gsmap, rearr, mpicom)
   380) 
   381)     call t_stopf('docn_initmctdom')
   382) 
   383)     !----------------------------------------------------------------------------
   384)     ! Initialize MCT attribute vectors
   385)     !----------------------------------------------------------------------------
   386) 
   387)     call t_startf('docn_initmctavs')
   388)     if (my_task == master_task) write(logunit,F00) 'allocate AVs'
   389)     call shr_sys_flush(logunit)
   390) 
   391)     call mct_aVect_init(o2x, rList=seq_flds_o2x_fields, lsize=lsize)
   392)     call mct_aVect_zero(o2x)
   393) 
   394)     kt    = mct_aVect_indexRA(o2x,'So_t')
   395)     ks    = mct_aVect_indexRA(o2x,'So_s')
   396)     ku    = mct_aVect_indexRA(o2x,'So_u')
   397)     kv    = mct_aVect_indexRA(o2x,'So_v')
   398)     kdhdx = mct_aVect_indexRA(o2x,'So_dhdx')
   399)     kdhdy = mct_aVect_indexRA(o2x,'So_dhdy')
   400)     kq    = mct_aVect_indexRA(o2x,'Fioo_q')
   401) 
   402)     call mct_aVect_init(x2o, rList=seq_flds_x2o_fields, lsize=lsize)
   403)     call mct_aVect_zero(x2o)
   404) 
   405)     kswnet = mct_aVect_indexRA(x2o,'Foxx_swnet')
   406)     klwup  = mct_aVect_indexRA(x2o,'Foxx_lwup')
   407)     klwdn  = mct_aVect_indexRA(x2o,'Faxa_lwdn')
   408)     ksen   = mct_aVect_indexRA(x2o,'Foxx_sen')
   409)     klat   = mct_aVect_indexRA(x2o,'Foxx_lat')
   410)     kmelth = mct_aVect_indexRA(x2o,'Fioi_melth')
   411)     ksnow  = mct_aVect_indexRA(x2o,'Faxa_snow')
   412)     krofi  = mct_aVect_indexRA(x2o,'Foxx_rofi')
   413) 
   414)     call mct_aVect_init(avstrm, rList=flds_strm, lsize=lsize)
   415)     call mct_aVect_zero(avstrm)
   416) 
   417)     kh    = mct_aVect_indexRA(avstrm,'strm_h')
   418)     kqbot = mct_aVect_indexRA(avstrm,'strm_qbot')
   419) 
   420)     allocate(somtp(lsize))
   421)     allocate(imask(lsize))
   422) 
   423)     kmask = mct_aVect_indexRA(ggrid%data,'mask')
   424)     imask(:) = nint(ggrid%data%rAttr(kmask,:))
   425) 
   426)     call t_stopf('docn_initmctavs')
   427) 
   428)     !----------------------------------------------------------------------------
   429)     ! Read restart
   430)     !----------------------------------------------------------------------------
   431) 
   432)     if (read_restart) then
   433)        if (trim(rest_file) == trim(nullstr) .and. &
   434)            trim(rest_file_strm) == trim(nullstr)) then
   435)           if (my_task == master_task) then
   436)              write(logunit,F00) ' restart filenames from rpointer'
   437)              call shr_sys_flush(logunit)
   438)              inquire(file=trim(rpfile)//trim(inst_suffix),exist=exists)
   439)              if (.not.exists) then
   440)                 write(logunit,F00) ' ERROR: rpointer file does not exist'
   441)                 call shr_sys_abort(trim(subname)//' ERROR: rpointer file missing')
   442)              endif
   443)              nu = shr_file_getUnit()
   444)              open(nu,file=trim(rpfile)//trim(inst_suffix),form='formatted')
   445)              read(nu,'(a)') rest_file
   446)              read(nu,'(a)') rest_file_strm
   447)              close(nu)
   448)              call shr_file_freeUnit(nu)
   449)              inquire(file=trim(rest_file_strm),exist=exists)
   450)           endif
   451)           call shr_mpi_bcast(rest_file,mpicom,'rest_file')
   452)           call shr_mpi_bcast(rest_file_strm,mpicom,'rest_file_strm')
   453)        else
   454)           ! use namelist already read
   455)           if (my_task == master_task) then
   456)              write(logunit,F00) ' restart filenames from namelist '
   457)              call shr_sys_flush(logunit)
   458)              inquire(file=trim(rest_file_strm),exist=exists)
   459)           endif
   460)        endif
   461)        call shr_mpi_bcast(exists,mpicom,'exists')
   462)        if (trim(ocn_mode) == 'SOM') then
   463)           if (my_task == master_task) write(logunit,F00) ' reading ',trim(rest_file)
   464)           call shr_pcdf_readwrite('read',iosystem,SDOCN%io_type,trim(rest_file),mpicom,gsmap,rf1=somtp,rf1n='somtp')
   465)        endif
   466)        if (exists) then
   467)           if (my_task == master_task) write(logunit,F00) ' reading ',trim(rest_file_strm)
