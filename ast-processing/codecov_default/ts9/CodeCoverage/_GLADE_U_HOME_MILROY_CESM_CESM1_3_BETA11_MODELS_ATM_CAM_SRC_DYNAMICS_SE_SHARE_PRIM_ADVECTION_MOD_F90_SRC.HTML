     9)       SEM 2D RK2 + monotone remap + hyper viscosity
    10)       SEM 2D RK2 can use sign-preserving or monotone reconstruction

    12) Notes on Lagrange+REMAP advection
    13) dynamics will compute mean fluxes, so that (i.e. for qsplit=3)
    14) 
    15)     dp(t+3)-dp(t) = -3dt div(Udp_sum/3) - 3dt d(eta_dot_dpdn_sum/3)  + 3dt D(dpdiss_sum/3)
    16) 
    17) Where the floating lagrangian component:
    18)     dp_star(t+3) = dp(t)  -3dt div(Udp_sum/3)  + 3dt D(dpdiss_sum/3)
    19) OR:
    20)     dp_star(t+3) = dp(t+1) + 3dt d( eta_dot_dpdn_ave(t) ) 
    21) 
    22) 
    23) For RK2 advection of Q:  (example of 2 stage RK for tracers):   dtq = qsplit*dt
    24) For consistency, if Q=1
    25)   dp1  = dp(t)- dtq div[ U1 dp(t)]     
    26)   dp2  = dp1  - dtq div[ U2 dp1  ]  + 2*dtq D( dpdiss_ave )   
    27)   dp*  = (dp(t) + dp2 )/2
    28)        =  dp(t) - dtq  div[ U1 dp(t) + U2 dp1 ]/2   + dtq D( dpdiss_ave )
    29) 
    30) so we require:
    31)   U1 = Udp_ave / dp(t)
    32)   U2 = Udp_ave / dp1
    33) 
    34) For tracer advection:
    35)   Qdp1  = Qdp(t)- dtq div[ U1 Qdp(t)]     
    36)   Qdp2  = Qdp1  - dtq div[ U2 Qdp1  ]  + 2*dtq D( Q dpdiss_ave )   
    37)   Qdp*  = (Qdp(t) + Qdp2 )/2
    38)        =  Qdp(t) - dtq  div[ U1 Qdp(t) + U2 Qdp1 ]   + dtq D( Q dpdiss_ave )
    39) 
    40) Qdp1:  limit Q, with Q = Qdp1-before-DSS/(dp1-before-DSS)      with dp1 as computed above
    41) Qdp2:  limit Q, with Q = Qdp2-before-DSS/(dp2-before-DSS)      with dp2 as computed above
    42) 
    43) For dissipation: Q = Qdp1-after-DSS / dp1-after-DSS
    44) 
    45) 
    46) last step:
    47)   remap Qdp* to Qdp(t+1)   [ dp_star(t+1) -> dp(t+1) ]

    93) !remap_calc_grids computes the vertical pressures and pressure differences for one vertical column for the reference grid
    94) !and for the deformed Lagrangian grid. This was pulled out of each routine since it was a repeated task.
    95) subroutine remap_calc_grids( hvcoord , ps , dt , eta_dot_dpdn , p_lag , p_ref , dp_lag , dp_ref )
    96)   implicit none
    97)   type(hvcoord_t)      , intent(in   ) :: hvcoord               !Derived type to hold vertical sigma grid parameters
    98)   real(kind=real_kind) , intent(in   ) :: ps                    !Surface pressure for this column
    99)   real(kind=real_kind) , intent(in   ) :: dt                    !Time step
   100)   real(kind=real_kind) , intent(in   ) :: eta_dot_dpdn(nlev+1)  !Looks like a vertical pressure flux
   101)                                                                 !to compute deformed grid spacing
   102)   real(kind=real_kind) , intent(  out) :: p_lag(nlev+1)         !Pressures at interfaces of the Lagrangian deformed grid
   103)   real(kind=real_kind) , intent(  out) :: p_ref(nlev+1)         !Pressures at interfaces of the reference grid
   104)   real(kind=real_kind) , intent(  out) :: dp_lag(nlev)          !Pressure differences on Lagrangian deformed grid
   105)   real(kind=real_kind) , intent(  out) :: dp_ref(nlev)          !Pressure differences on reference grid
   106)   integer :: k                                                  !Iterator
   107)   p_ref(1) = 0  !Both grids have a model top pressure of zero
   108)   p_lag(1) = 0  !Both grids have a model top pressure of zero
   109)   do k = 1 , nlev
   110)     dp_ref(k) = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) ) * hvcoord%ps0 + &
   111)          ( hvcoord%hybi(k+1) - hvcoord%hybi(k) ) * ps  !Reference pressure difference
   112)     ! Lagrangian pressure difference (flux in - flux out over the time step)
   113)     dp_lag(k) = dp_ref(k) + dt * ( eta_dot_dpdn(k+1) - eta_dot_dpdn(k) )
   114)     p_ref(k+1) = p_ref(k) + dp_ref(k) !Pressure at interfaces accumulated using difference over each cell
   115)     p_lag(k+1) = p_lag(k) + dp_lag(k) !Pressure at interfaces accumulated using difference over each cell
   116)   enddo
   117) end subroutine remap_calc_grids

   123) subroutine remap1(Qdp,nx,qsize,dp1,dp2)
   124)   ! remap 1 field
   125)   ! input:  Qdp   field to be remapped (NOTE: MASS, not MIXING RATIO)
   126)   !         dp1   layer thickness (source)
   127)   !         dp2   layer thickness (target)
   128)   !
   129)   ! output: remaped Qdp, conserving mass, monotone on Q=Qdp/dp
   130)   !
   131)   implicit none
   132)   integer, intent(in) :: nx,qsize 
   133)   real (kind=real_kind), intent(inout) :: Qdp(nx,nx,nlev,qsize)
   134)   real (kind=real_kind), intent(in) :: dp1(nx,nx,nlev),dp2(nx,nx,nlev)
   135)   ! ========================
   136)   ! Local Variables
   137)   ! ========================
   138) 
   139)   real (kind=real_kind), dimension(nlev+1)    :: rhs,lower_diag,diag,upper_diag,q_diag,zgam,z1c,z2c,zv
   140)   real (kind=real_kind), dimension(nlev)      :: h,Qcol,dy,za0,za1,za2,zarg,zhdp,dp_star,dp_np1
   141)   real (kind=real_kind)  :: f_xm,level1,level2,level3,level4,level5, &
   142)                             peaks_min,peaks_max,tmp_cal,xm,xm_d,zv1,zv2, &
   143)                             zero = 0,one = 1,tiny = 1e-12,qmax = 1d50
   144)   integer(kind=int_kind) :: zkr(nlev+1),filter_code(nlev),peaks,im1,im2,im3,ip1,ip2, & 
   145)                             lt1,lt2,lt3,t0,t1,t2,t3,t4,tm,tp,ie,i,ilev,j,jk,k,q
   146)   logical :: abort=.false.
   147)   call t_startf('remap1')
   148) 
   149)   if (vert_remap_q_alg == 1 .or. vert_remap_q_alg == 2) then
   150)      call remap_Q_ppm(qdp,nx,qsize,dp1,dp2)
   151)      call t_stopf('remap1')
   152)      return
   153)   endif
   154) 
   155) #if (defined ELEMENT_OPENMP)
   156) !$omp parallel do private(qsize,i,j,z1c,z2c,zv,k,dp_np1,dp_star,Qcol,zkr,ilev) &
   157) !$omp    private(jk,zgam,zhdp,h,zarg,rhs,lower_diag,diag,upper_diag,q_diag,tmp_cal,filter_code) &
   158) !$omp    private(dy,im1,im2,im3,ip1,t1,t2,t3,za0,za1,za2,xm_d,xm,f_xm,t4,tm,tp,peaks,peaks_min) &
   159) !$omp    private(peaks_max,ip2,level1,level2,level3,level4,level5,lt1,lt2,lt3,zv1,zv2)
   160) #endif
   161)   do q=1,qsize
   162)   do i=1,nx
   163)     do j=1,nx
   164) 
   165)       z1c(1)=0 ! source grid
   166)       z2c(1)=0 ! target grid
   167)       do k=1,nlev
   168)          z1c(k+1)=z1c(k)+dp1(i,j,k)
   169)          z2c(k+1)=z2c(k)+dp2(i,j,k)
   170)       enddo
   171) 
   172)       zv(1)=0
   173)       do k=1,nlev
   174)         Qcol(k)=Qdp(i,j,k,q)!  *(z1c(k+1)-z1c(k)) input is mass
   175)         zv(k+1) = zv(k)+Qcol(k)
   176)       enddo
   177)       
   178)       if (ABS(z2c(nlev+1)-z1c(nlev+1)).GE.0.000001) then
   179)         write(6,*) 'SURFACE PRESSURE IMPLIED BY ADVECTION SCHEME'
   180)         write(6,*) 'NOT CORRESPONDING TO SURFACE PRESSURE IN    '
   181)         write(6,*) 'DATA FOR MODEL LEVELS'
   182)         write(6,*) 'PLEVMODEL=',z2c(nlev+1)
   183)         write(6,*) 'PLEV     =',z1c(nlev+1)
   184)         write(6,*) 'DIFF     =',z2c(nlev+1)-z1c(nlev+1)
   185)         abort=.true.
   186)       endif
   187)       
   188)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   189)       !! quadratic splies with UK met office monotonicity constraints  !!
   190)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   191)       
   192)       zkr  = 99
   193)       ilev = 2
   194)       zkr(1) = 1
   195)       zkr(nlev+1) = nlev
   196)       kloop: do k = 2,nlev
   197)         do jk = ilev,nlev+1
   198)           if (z1c(jk).ge.z2c(k)) then
   199)             ilev      = jk
   200)             zkr(k)   = jk-1
   201)             cycle kloop
   202)           endif
   203)         enddo
   204)       enddo kloop
   205)       
   206)       zgam  = (z2c(1:nlev+1)-z1c(zkr)) / (z1c(zkr+1)-z1c(zkr))
   207)       zgam(1)      = 0.0
   208)       zgam(nlev+1) = 1.0
   209)       zhdp = z1c(2:nlev+1)-z1c(1:nlev)
   210)       
   211)       
   212)       h = 1/zhdp
   213)       zarg = Qcol * h
   214)       rhs = 0
   215)       lower_diag = 0
   216)       diag = 0
   217)       upper_diag = 0
   218)       
   219)       rhs(1)=3*zarg(1)
   220)       rhs(2:nlev) = 3*(zarg(2:nlev)*h(2:nlev) + zarg(1:nlev-1)*h(1:nlev-1)) 
   221)       rhs(nlev+1)=3*zarg(nlev)
   222)       
   223)       lower_diag(1)=1
   224)       lower_diag(2:nlev) = h(1:nlev-1)
   225)       lower_diag(nlev+1)=1
   226)       
   227)       diag(1)=2
   228)       diag(2:nlev) = 2*(h(2:nlev) + h(1:nlev-1))
   229)       diag(nlev+1)=2
   230)       
   231)       upper_diag(1)=1
   232)       upper_diag(2:nlev) = h(2:nlev)
   233)       upper_diag(nlev+1)=0
   234)       
   235)       q_diag(1)=-upper_diag(1)/diag(1)
   236)       rhs(1)= rhs(1)/diag(1)
   237)       
   238)       do k=2,nlev+1
   239)         tmp_cal    =  1/(diag(k)+lower_diag(k)*q_diag(k-1))
   240)         q_diag(k) = -upper_diag(k)*tmp_cal
   241)         rhs(k) =  (rhs(k)-lower_diag(k)*rhs(k-1))*tmp_cal
   242)       enddo
   243)       do k=nlev,1,-1
   244)         rhs(k)=rhs(k)+q_diag(k)*rhs(k+1)
   245)       enddo
   246)       
   247)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   248)       !!  monotonicity modifications  !!
   249)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!            
   250)       
   251)       filter_code = 0
   252)       dy(1:nlev-1) = zarg(2:nlev)-zarg(1:nlev-1)
   253)       dy(nlev) = dy(nlev-1)                          
   254)       
   255)       dy = merge(zero, dy, abs(dy) < tiny )
   256)       
   257)       do k=1,nlev
   258)         im1=MAX(1,k-1)
   259)         im2=MAX(1,k-2)
   260)         im3=MAX(1,k-3)
   261)         ip1=MIN(nlev,k+1)
   262)         t1 = merge(1,0,(zarg(k)-rhs(k))*(rhs(k)-zarg(im1)) >= 0)
   263)         t2 = merge(1,0,dy(im2)*(rhs(k)-zarg(im1)) > 0 .AND. dy(im2)*dy(im3) > 0 &
   264)              .AND. dy(k)*dy(ip1) > 0 .AND. dy(im2)*dy(k) < 0 )
   265)         t3 = merge(1,0,ABS(rhs(k)-zarg(im1)) > ABS(rhs(k)-zarg(k)))	
   266)         
   267)         filter_code(k) = merge(0,1,t1+t2 > 0) 
   268)         rhs(k) = (1-filter_code(k))*rhs(k)+filter_code(k)*(t3*zarg(k)+(1-t3)*zarg(im1))
   269)         filter_code(im1) = MAX(filter_code(im1),filter_code(k))
   270)       enddo
   271)       
   272)       rhs = merge(qmax,rhs,rhs > qmax)
   273)       rhs = merge(zero,rhs,rhs < zero)
   274)       
   275)       za0 = rhs(1:nlev) 
   276)       za1 = -4*rhs(1:nlev) - 2*rhs(2:nlev+1) + 6*zarg  
   277)       za2 =  3*rhs(1:nlev) + 3*rhs(2:nlev+1) - 6*zarg 
   278)       
   279)       dy(1:nlev) = rhs(2:nlev+1)-rhs(1:nlev)                             
   280)       dy = merge(zero, dy, abs(dy) < tiny )
   281)       
   282)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!     
   283)       !! Compute the 3 quadratic spline coeffients {za0, za1, za2}				   !!
   284)       !! knowing the quadratic spline parameters {rho_left,rho_right,zarg}		   !!
   285)       !! Zerroukat et.al., Q.J.R. Meteorol. Soc., Vol. 128, pp. 2801-2820 (2002).   !!
   286)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!! 
   287)       
   288)       
   289)       h = rhs(2:nlev+1)  
   290)       
   291)       do k=1,nlev
   292)         xm_d = merge(one,2*za2(k),abs(za2(k)) < tiny)
   293)         xm = merge(zero,-za1(k)/xm_d, abs(za2(k)) < tiny)            
   294)         f_xm = za0(k) + za1(k)*xm + za2(k)*xm**2
   295)         
   296)         t1 = merge(1,0,ABS(za2(k)) > tiny)
   297)         t2 = merge(1,0,xm <= zero .OR. xm >= 1)
   298)         t3 = merge(1,0,za2(k) > zero)
   299)         t4 = merge(1,0,za2(k) < zero)
   300)         tm = merge(1,0,t1*((1-t2)+t3) .EQ. 2)
   301)         tp = merge(1,0,t1*((1-t2)+(1-t3)+t4) .EQ. 3)
   302)         
   303)         peaks=0
   304)         peaks = merge(-1,peaks,tm .EQ. 1)
   305)         peaks = merge(+1,peaks,tp .EQ. 1)
   306)         peaks_min = merge(f_xm,MIN(za0(k),za0(k)+za1(k)+za2(k)),tm .EQ. 1)
   307)         peaks_max = merge(f_xm,MAX(za0(k),za0(k)+za1(k)+za2(k)),tp .EQ. 1)
   308)         
   309)         im1=MAX(1,k-1)
   310)         im2=MAX(1,k-2)
   311)         ip1=MIN(nlev,k+1)
   312)         ip2=MIN(nlev,k+2)
   313)         
   314)         t1 = merge(abs(peaks),0,(dy(im2)*dy(im1) <= tiny) .OR. &
   315)              (dy(ip1)*dy(ip2) <= tiny) .OR. (dy(im1)*dy(ip1) >= tiny) .OR. &
   316)              (dy(im1)*float(peaks) <= tiny))
   317)         
   318)         filter_code(k) = merge(1,t1+(1-t1)*filter_code(k),(rhs(k) >= qmax) .OR. & 
   319)              (rhs(k) <= zero) .OR. (peaks_max > qmax) .OR. (peaks_min < tiny))
   320)         
   321)         if (filter_code(k) > 0) then
   322)           level1 = rhs(k)
   323)           level2 = (2*rhs(k)+h(k))/3
   324)           level3 = 0.5*(rhs(k)+h(k)) 
   325)           level4 = (1/3d0)*rhs(k)+2*(1/3d0)*h(k)
   326)           level5 = h(k)
   327)           
   328)           t1 = merge(1,0,h(k) >= rhs(k))
   329)           t2 = merge(1,0,zarg(k) <= level1 .OR.  zarg(k) >= level5)
   330)           t3 = merge(1,0,zarg(k) >  level1 .AND. zarg(k) <  level2)
   331)           t4 = merge(1,0,zarg(k) >  level4 .AND. zarg(k) <  level5)
   332)           
   333)           lt1 = t1*t2
   334)           lt2 = t1*(1-t2+t3)
   335)           lt3 = t1*(1-t2+1-t3+t4)
   336)           
   337)           za0(k) = merge(zarg(k),za0(k),lt1 .EQ. 1)
   338)           za1(k) = merge(zero,za1(k),lt1 .EQ. 1)
   339)           za2(k) = merge(zero,za2(k),lt1 .EQ. 1)
   340)           
   341)           za0(k) = merge(rhs(k),za0(k),lt2 .EQ. 2)
   342)           za1(k) = merge(zero,za1(k),lt2 .EQ. 2)
   343)           za2(k) = merge(3*(zarg(k)-rhs(k)),za2(k),lt2 .EQ. 2)
   344)           
   345)           za0(k) = merge(-2*h(k)+3*zarg(k),za0(k),lt3 .EQ. 3)
   346)           za1(k) = merge(+6*h(k)-6*zarg(k),za1(k),lt3 .EQ. 3)
   347)           za2(k) = merge(-3*h(k)+3*zarg(k),za2(k),lt3 .EQ. 3)
   348)           
   349)           t2 = merge(1,0,zarg(k) >= level1 .OR.  zarg(k) <= level5)
   350)           t3 = merge(1,0,zarg(k) <  level1 .AND. zarg(k) >  level2)
   351)           t4 = merge(1,0,zarg(k) <  level4 .AND. zarg(k) >  level5)
   352)           
   353)           lt1 = (1-t1)*t2
   354)           lt2 = (1-t1)*(1-t2+t3)
   355)           lt3 = (1-t1)*(1-t2+1-t3+t4)
   356)           
   357)           za0(k) = merge(zarg(k),za0(k),lt1 .EQ. 1)
   358)           za1(k) = merge(zero,za1(k),lt1 .EQ. 1)
   359)           za2(k) = merge(zero,za2(k),lt1 .EQ. 1)
   360)           
   361)           za0(k) = merge(rhs(k),za0(k),lt2 .EQ. 2)
   362)           za1(k) = merge(zero,za1(k),lt2 .EQ. 2)
   363)           za2(k) = merge(3*(zarg(k)-rhs(k)),za2(k),lt2 .EQ. 2)
   364)           
   365)           za0(k) = merge(-2*h(k)+3*zarg(k),za0(k),lt3 .EQ. 3)
   366)           za1(k) = merge(+6*h(k)-6*zarg(k),za1(k),lt3 .EQ. 3)
   367)           za2(k) = merge(-3*h(k)+3*zarg(k),za2(k),lt3 .EQ. 3)
   368)         endif
   369)       enddo
   370)       
   371)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   372)       !! start iteration from top to bottom of atmosphere !! 
   373)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   374)       
   375)       zv1 = 0
   376)       do k=1,nlev
   377)         if (zgam(k+1)>1d0) then
   378)           WRITE(*,*) 'r not in [0:1]', zgam(k+1)
   379)           abort=.true.
   380)         endif
   381)         zv2 = zv(zkr(k+1))+(za0(zkr(k+1))*zgam(k+1)+(za1(zkr(k+1))/2)*(zgam(k+1)**2)+ &
   382)              (za2(zkr(k+1))/3)*(zgam(k+1)**3))*zhdp(zkr(k+1))
   383)         Qdp(i,j,k,q) = (zv2 - zv1) ! / (z2c(k+1)-z2c(k) ) dont convert back to mixing ratio
   384)         zv1 = zv2
   385)       enddo
   386)     enddo
   387)   enddo
   388)   enddo ! q loop 
   389)   if (abort) call abortmp('Bad levels in remap1.  usually CFL violatioin')
   390)   call t_stopf('remap1')
   391) end subroutine remap1

   396)   !         dp1   layer thickness (source)
   397)   !         dp2   layer thickness (target)
   398)   !
   399)   ! output: remaped Qdp, conserving mass
   400)   !
   401)   implicit none
   402)   integer, intent(in) :: nx,qsize 
   403)   real (kind=real_kind), intent(inout) :: Qdp(nx,nx,nlev,qsize)
   404)   real (kind=real_kind), intent(in) :: dp1(nx,nx,nlev),dp2(nx,nx,nlev)
   405)   ! ========================
   406)   ! Local Variables
   407)   ! ========================
   408) 
   409)   real (kind=real_kind), dimension(nlev+1)    :: rhs,lower_diag,diag,upper_diag,q_diag,zgam,z1c,z2c,zv
   410)   real (kind=real_kind), dimension(nlev)      :: h,Qcol,dy,za0,za1,za2,zarg,zhdp,dp_star,dp_np1
   411)   real (kind=real_kind)  :: f_xm,level1,level2,level3,level4,level5, &
   412)                             peaks_min,peaks_max,tmp_cal,xm,xm_d,zv1,zv2, &
   413)                             zero = 0,one = 1,tiny = 1e-12,qmax = 1d50
   414)   integer(kind=int_kind) :: zkr(nlev+1),filter_code(nlev),peaks,im1,im2,im3,ip1,ip2, & 
   415)                             lt1,lt2,lt3,t0,t1,t2,t3,t4,tm,tp,ie,i,ilev,j,jk,k,q
   416)   logical :: abort=.false.  
   417) !   call t_startf('remap1_nofilter')
   418) 
   419) #if (defined ELEMENT_OPENMP)
   420) !$omp parallel do private(qsize,i,j,z1c,z2c,zv,k,dp_np1,dp_star,Qcol,zkr,ilev) &
   421) !$omp    private(jk,zgam,zhdp,h,zarg,rhs,lower_diag,diag,upper_diag,q_diag,tmp_cal,filter_code) &
   422) !$omp    private(dy,im1,im2,im3,ip1,t1,t2,t3,za0,za1,za2,xm_d,xm,f_xm,t4,tm,tp,peaks,peaks_min) &
   423) !$omp    private(peaks_max,ip2,level1,level2,level3,level4,level5,lt1,lt2,lt3,zv1,zv2)
   424) #endif
   425)   do q=1,qsize
   426)   do i=1,nx
   427)     do j=1,nx
   428) 
   429)       z1c(1)=0 ! source grid
   430)       z2c(1)=0 ! target grid
   431)       do k=1,nlev
   432)          z1c(k+1)=z1c(k)+dp1(i,j,k)
   433)          z2c(k+1)=z2c(k)+dp2(i,j,k)
   434)       enddo
   435) 
   436)       zv(1)=0
   437)       do k=1,nlev
   438)         Qcol(k)=Qdp(i,j,k,q)!  *(z1c(k+1)-z1c(k)) input is mass
   439)         zv(k+1) = zv(k)+Qcol(k)
   440)       enddo
   441)       
   442)       if (ABS(z2c(nlev+1)-z1c(nlev+1)).GE.0.000001) then
   443)         write(6,*) 'SURFACE PRESSURE IMPLIED BY ADVECTION SCHEME'
   444)         write(6,*) 'NOT CORRESPONDING TO SURFACE PRESSURE IN    '
   445)         write(6,*) 'DATA FOR MODEL LEVELS'
   446)         write(6,*) 'PLEVMODEL=',z2c(nlev+1)
   447)         write(6,*) 'PLEV     =',z1c(nlev+1)
   448)         write(6,*) 'DIFF     =',z2c(nlev+1)-z1c(nlev+1)
   449)         abort=.true.
   450)       endif
   451)       
   452)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   453)       !! quadratic splies with UK met office monotonicity constraints  !!
   454)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   455)       
   456)       zkr  = 99
   457)       ilev = 2
   458)       zkr(1) = 1
   459)       zkr(nlev+1) = nlev
   460)       kloop: do k = 2,nlev
   461)         do jk = ilev,nlev+1
   462)           if (z1c(jk).ge.z2c(k)) then
   463)             ilev      = jk
   464)             zkr(k)   = jk-1
   465)             cycle kloop
   466)           endif
   467)         enddo
   468)       enddo kloop
   469)       
   470)       zgam  = (z2c(1:nlev+1)-z1c(zkr)) / (z1c(zkr+1)-z1c(zkr))
   471)       zgam(1)      = 0.0
   472)       zgam(nlev+1) = 1.0
   473)       zhdp = z1c(2:nlev+1)-z1c(1:nlev)
   474)       
   475)       
   476)       h = 1/zhdp
   477)       zarg = Qcol * h
   478)       rhs = 0
   479)       lower_diag = 0
   480)       diag = 0
   481)       upper_diag = 0
   482)       
   483)       rhs(1)=3*zarg(1)
   484)       rhs(2:nlev) = 3*(zarg(2:nlev)*h(2:nlev) + zarg(1:nlev-1)*h(1:nlev-1)) 
   485)       rhs(nlev+1)=3*zarg(nlev)
   486)       
   487)       lower_diag(1)=1
   488)       lower_diag(2:nlev) = h(1:nlev-1)
   489)       lower_diag(nlev+1)=1
   490)       
   491)       diag(1)=2
   492)       diag(2:nlev) = 2*(h(2:nlev) + h(1:nlev-1))
   493)       diag(nlev+1)=2
   494)       
   495)       upper_diag(1)=1
   496)       upper_diag(2:nlev) = h(2:nlev)
   497)       upper_diag(nlev+1)=0
   498)       
   499)       q_diag(1)=-upper_diag(1)/diag(1)
   500)       rhs(1)= rhs(1)/diag(1)
   501)       
   502)       do k=2,nlev+1
   503)         tmp_cal    =  1/(diag(k)+lower_diag(k)*q_diag(k-1))
   504)         q_diag(k) = -upper_diag(k)*tmp_cal
   505)         rhs(k) =  (rhs(k)-lower_diag(k)*rhs(k-1))*tmp_cal
   506)       enddo
   507)       do k=nlev,1,-1
   508)         rhs(k)=rhs(k)+q_diag(k)*rhs(k+1)
   509)       enddo
   510)       
   511)       za0 = rhs(1:nlev) 
   512)       za1 = -4*rhs(1:nlev) - 2*rhs(2:nlev+1) + 6*zarg  
   513)       za2 =  3*rhs(1:nlev) + 3*rhs(2:nlev+1) - 6*zarg 
   514)       
   515)       
   516)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   517)       !! start iteration from top to bottom of atmosphere !! 
   518)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   519)       
   520)       zv1 = 0
   521)       do k=1,nlev
   522)         if (zgam(k+1)>1d0) then
   523)           WRITE(*,*) 'r not in [0:1]', zgam(k+1)
   524)           abort=.true.
   525)         endif
   526)         zv2 = zv(zkr(k+1))+(za0(zkr(k+1))*zgam(k+1)+(za1(zkr(k+1))/2)*(zgam(k+1)**2)+ &
   527)              (za2(zkr(k+1))/3)*(zgam(k+1)**3))*zhdp(zkr(k+1))
   528)         Qdp(i,j,k,q) = (zv2 - zv1) ! / (z2c(k+1)-z2c(k) ) dont convert back to mixing ratio
   529)         zv1 = zv2
   530)       enddo
   531)     enddo
   532)   enddo
   533)   enddo ! q loop 
   534)   if (abort) call abortmp('Bad levels in remap1_nofilter.  usually CFL violatioin')
   535) !   call t_stopf('remap1_nofilter')

   541) !This uses the exact same model and reference grids and data as remap_Q, but it interpolates
   542) !using PPM instead of splines.
   543) subroutine remap_Q_ppm(Qdp,nx,qsize,dp1,dp2)
   544)   ! remap 1 field
   545)   ! input:  Qdp   field to be remapped (NOTE: MASS, not MIXING RATIO)
   546)   !         dp1   layer thickness (source)
   547)   !         dp2   layer thickness (target)
   548)   !
   549)   ! output: remaped Qdp, conserving mass
   550)   !
   551)   use control_mod, only        : prescribed_wind, vert_remap_q_alg
   552)   implicit none
   553)   integer,intent(in) :: nx,qsize
   554)   real (kind=real_kind), intent(inout) :: Qdp(nx,nx,nlev,qsize)
   555)   real (kind=real_kind), intent(in) :: dp1(nx,nx,nlev),dp2(nx,nx,nlev)
   556)   ! Local Variables
   557)   integer, parameter :: gs = 2                              !Number of cells to place in the ghost region
   558)   real(kind=real_kind), dimension(       nlev+2 ) :: pio    !Pressure at interfaces for old grid
   559)   real(kind=real_kind), dimension(       nlev+1 ) :: pin    !Pressure at interfaces for new grid
   560)   real(kind=real_kind), dimension(       nlev+1 ) :: masso  !Accumulate mass up to each interface
   561)   real(kind=real_kind), dimension(  1-gs:nlev+gs) :: ao     !Tracer value on old grid
   562)   real(kind=real_kind), dimension(  1-gs:nlev+gs) :: dpo    !change in pressure over a cell for old grid
   563)   real(kind=real_kind), dimension(  1-gs:nlev+gs) :: dpn    !change in pressure over a cell for old grid
   564)   real(kind=real_kind), dimension(3,     nlev   ) :: coefs  !PPM coefficients within each cell
   565)   real(kind=real_kind), dimension(       nlev   ) :: z1, z2
   566)   real(kind=real_kind) :: ppmdx(10,0:nlev+1)  !grid spacings
   567)   real(kind=real_kind) :: mymass, massn1, massn2
   568)   integer :: i, j, k, q, kk, kid(nlev)
   569) 
   570)   call t_startf('remap_Q_ppm')
   571)   do j = 1 , nx
   572)     do i = 1 , nx
   573)       
   574)       pin(1)=0
   575)       pio(1)=0
   576)       do k=1,nlev
   577)          dpn(k)=dp2(i,j,k)
   578)          dpo(k)=dp1(i,j,k)
   579)          pin(k+1)=pin(k)+dpn(k)
   580)          pio(k+1)=pio(k)+dpo(k)
   581)       enddo
   582) 
   583) 
   584) 
   585)       pio(nlev+2) = pio(nlev+1) + 1.  !This is here to allow an entire block of k threads to run in the remapping phase.
   586)                                       !It makes sure there's an old interface value below the domain that is larger.
   587)       pin(nlev+1) = pio(nlev+1)       !The total mass in a column does not change.
   588)                                       !Therefore, the pressure of that mass cannot either.
   589)       !Fill in the ghost regions with mirrored values. if vert_remap_q_alg is defined, this is of no consequence.
   590)       do k = 1 , gs
   591)         dpo(1   -k) = dpo(       k)
   592)         dpo(nlev+k) = dpo(nlev+1-k)
   593)       enddo
   594) 
   595)       !Compute remapping intervals once for all tracers. Find the old grid cell index in which the
   596)       !k-th new cell interface resides. Then integrate from the bottom of that old cell to the new
   597)       !interface location. In practice, the grid never deforms past one cell, so the search can be
   598)       !simplified by this. Also, the interval of integration is usually of magnitude close to zero
   599)       !or close to dpo because of minimial deformation.
   600)       !Numerous tests confirmed that the bottom and top of the grids match to machine precision, so
   601)       !I set them equal to each other.
   602)       do k = 1 , nlev
   603)         kk = k  !Keep from an order n^2 search operation by assuming the old cell index is close.
   604)         !Find the index of the old grid cell in which this new cell's bottom interface resides.
   605)         do while ( pio(kk) <= pin(k+1) )
   606)           kk = kk + 1
   607)         enddo
   608)         kk = kk - 1                   !kk is now the cell index we're integrating over.
   609)         if (kk == nlev+1) kk = nlev   !This is to keep the indices in bounds.
   610)                                       !Top bounds match anyway, so doesn't matter what coefficients are used
   611)         kid(k) = kk                   !Save for reuse
   612)         z1(k) = -0.5D0                !This remapping assumes we're starting from the left interface of an old grid cell
   613)                                       !In fact, we're usually integrating very little or almost all of the cell in question
   614)         z2(k) = ( pin(k+1) - ( pio(kk) + pio(kk+1) ) * 0.5 ) / dpo(kk)  !PPM interpolants are normalized to an independent
   615)                                                                         !coordinate domain [-0.5,0.5].
   616)       enddo
   617) 
   618)       !This turned out a big optimization, remembering that only parts of the PPM algorithm depends on the data, namely the
   619)       !limiting. So anything that depends only on the grid is pre-computed outside the tracer loop.
   620)       ppmdx(:,:) = compute_ppm_grids( dpo )
   621) 
   622)       !From here, we loop over tracers for only those portions which depend on tracer data, which includes PPM limiting and
   623)       !mass accumulation
   624)       do q = 1 , qsize
   625)         !Accumulate the old mass up to old grid cell interface locations to simplify integration
   626)         !during remapping. Also, divide out the grid spacing so we're working with actual tracer
   627)         !values and can conserve mass. The option for ifndef ZEROHORZ I believe is there to ensure
   628)         !tracer consistency for an initially uniform field. I copied it from the old remap routine.
   629)         masso(1) = 0.
   630)         do k = 1 , nlev
   631)           ao(k) = Qdp(i,j,k,q)
   632)           masso(k+1) = masso(k) + ao(k) !Accumulate the old mass. This will simplify the remapping
   633)           ao(k) = ao(k) / dpo(k)        !Divide out the old grid spacing because we want the tracer mixing ratio, not mass.
   634)         enddo
   635)         !Fill in ghost values. Ignored if vert_remap_q_alg == 2
   636)         do k = 1 , gs
   637)           ao(1   -k) = ao(       k)
   638)           ao(nlev+k) = ao(nlev+1-k)
   639)         enddo
   640)         !Compute monotonic and conservative PPM reconstruction over every cell
   641)         coefs(:,:) = compute_ppm( ao , ppmdx )
   642)         !Compute tracer values on the new grid by integrating from the old cell bottom to the new
   643)         !cell interface to form a new grid mass accumulation. Taking the difference between
   644)         !accumulation at successive interfaces gives the mass inside each cell. Since Qdp is
   645)         !supposed to hold the full mass this needs no normalization.
   646)         massn1 = 0.
   647)         do k = 1 , nlev
   648)           kk = kid(k)
   649)           massn2 = masso(kk) + integrate_parabola( coefs(:,kk) , z1(k) , z2(k) ) * dpo(kk)
   650)           Qdp(i,j,k,q) = massn2 - massn1
   651)           massn1 = massn2
   652)         enddo
   653)       enddo
   654)     enddo
   655)   enddo
   656)   call t_stopf('remap_Q_ppm')
   657) end subroutine remap_Q_ppm

   663) !THis compute grid-based coefficients from Collela & Woodward 1984.
   664) function compute_ppm_grids( dx )   result(rslt)
   665)   use control_mod, only: vert_remap_q_alg
   666)   implicit none
   667)   real(kind=real_kind), intent(in) :: dx(-1:nlev+2)  !grid spacings
   668)   real(kind=real_kind)             :: rslt(10,0:nlev+1)  !grid spacings
   669)   integer :: j
   670)   integer :: indB, indE
   671) 
   672)   !Calculate grid-based coefficients for stage 1 of compute_ppm
   673)   if (vert_remap_q_alg == 2) then
   674)     indB = 2
   675)     indE = nlev-1
   676)   else
   677)     indB = 0
   678)     indE = nlev+1
   679)   endif
   680)   do j = indB , indE
   681)     rslt( 1,j) = dx(j) / ( dx(j-1) + dx(j) + dx(j+1) )
   682)     rslt( 2,j) = ( 2.*dx(j-1) + dx(j) ) / ( dx(j+1) + dx(j) )
   683)     rslt( 3,j) = ( dx(j) + 2.*dx(j+1) ) / ( dx(j-1) + dx(j) )
   684)   enddo
   685) 
   686)   !Caculate grid-based coefficients for stage 2 of compute_ppm
   687)   if (vert_remap_q_alg == 2) then
   688)     indB = 2
   689)     indE = nlev-2
   690)   else
   691)     indB = 0
   692)     indE = nlev
   693)   endif
   694)   do j = indB , indE
   695)     rslt( 4,j) = dx(j) / ( dx(j) + dx(j+1) )
   696)     rslt( 5,j) = 1. / sum( dx(j-1:j+2) )
   697)     rslt( 6,j) = ( 2. * dx(j+1) * dx(j) ) / ( dx(j) + dx(j+1 ) )
   698)     rslt( 7,j) = ( dx(j-1) + dx(j  ) ) / ( 2. * dx(j  ) + dx(j+1) )
   699)     rslt( 8,j) = ( dx(j+2) + dx(j+1) ) / ( 2. * dx(j+1) + dx(j  ) )
   700)     rslt( 9,j) = dx(j  ) * ( dx(j-1) + dx(j  ) ) / ( 2.*dx(j  ) +    dx(j+1) )
   701)     rslt(10,j) = dx(j+1) * ( dx(j+1) + dx(j+2) ) / (    dx(j  ) + 2.*dx(j+1) )
   702)   enddo
   703) end function compute_ppm_grids

   709) !This computes a limited parabolic interpolant using a net 5-cell stencil, but the stages of computation are broken up into 3 stages
   710) function compute_ppm( a , dx )    result(coefs)
   711)   use control_mod, only: vert_remap_q_alg
   712)   implicit none
   713)   real(kind=real_kind), intent(in) :: a    (    -1:nlev+2)  !Cell-mean values
   714)   real(kind=real_kind), intent(in) :: dx   (10,  0:nlev+1)  !grid spacings
   715)   real(kind=real_kind) ::             coefs(0:2,   nlev  )  !PPM coefficients (for parabola)
   716)   real(kind=real_kind) :: ai (0:nlev  )                     !fourth-order accurate, then limited interface values
   717)   real(kind=real_kind) :: dma(0:nlev+1)                     !An expression from Collela's '84 publication
   718)   real(kind=real_kind) :: da                                !Ditto
   719)   ! Hold expressions based on the grid (which are cumbersome).
   720)   real(kind=real_kind) :: dx1, dx2, dx3, dx4, dx5, dx6, dx7, dx8, dx9, dx10
   721)   real(kind=real_kind) :: al, ar                            !Left and right interface values for cell-local limiting
   722)   integer :: j
   723)   integer :: indB, indE
   724) 
   725)   ! Stage 1: Compute dma for each cell, allowing a 1-cell ghost stencil below and above the domain
   726)   if (vert_remap_q_alg == 2) then
   727)     indB = 2
   728)     indE = nlev-1
   729)   else
   730)     indB = 0
   731)     indE = nlev+1
   732)   endif
   733)   do j = indB , indE
   734)     da = dx(1,j) * ( dx(2,j) * ( a(j+1) - a(j) ) + dx(3,j) * ( a(j) - a(j-1) ) )
   735)     dma(j) = minval( (/ abs(da) , 2. * abs( a(j) - a(j-1) ) , 2. * abs( a(j+1) - a(j) ) /) ) * sign(1.D0,da)
   736)     if ( ( a(j+1) - a(j) ) * ( a(j) - a(j-1) ) <= 0. ) dma(j) = 0.
   737)   enddo
   738) 
   739)   ! Stage 2: Compute ai for each cell interface in the physical domain (dimension nlev+1)
   740)   if (vert_remap_q_alg == 2) then
   741)     indB = 2
   742)     indE = nlev-2
   743)   else
   744)     indB = 0
   745)     indE = nlev
   746)   endif
   747)   do j = indB , indE
   748)     ai(j) = a(j) + dx(4,j) * ( a(j+1) - a(j) ) + dx(5,j) * ( dx(6,j) * ( dx(7,j) - dx(8,j) ) &
   749)          * ( a(j+1) - a(j) ) - dx(9,j) * dma(j+1) + dx(10,j) * dma(j) )
   750)   enddo
   751) 
   752)   ! Stage 3: Compute limited PPM interpolant over each cell in the physical domain
   753)   ! (dimension nlev) using ai on either side and ao within the cell.
   754)   if (vert_remap_q_alg == 2) then
   755)     indB = 3
   756)     indE = nlev-2
   757)   else
   758)     indB = 1
   759)     indE = nlev
   760)   endif
   761)   do j = indB , indE
   762)     al = ai(j-1)
   763)     ar = ai(j  )
   764)     if ( (ar - a(j)) * (a(j) - al) <= 0. ) then
   765)       al = a(j)
   766)       ar = a(j)
   767)     endif
   768)     if ( (ar - al) * (a(j) - (al + ar)/2.) >  (ar - al)**2/6. ) al = 3.*a(j) - 2. * ar
   769)     if ( (ar - al) * (a(j) - (al + ar)/2.) < -(ar - al)**2/6. ) ar = 3.*a(j) - 2. * al
   770)     !Computed these coefficients from the edge values and cell mean in Maple. Assumes normalized coordinates: xi=(x-x0)/dx
   771)     coefs(0,j) = 1.5 * a(j) - ( al + ar ) / 4.
   772)     coefs(1,j) = ar - al
   773)     coefs(2,j) = -6. * a(j) + 3. * ( al + ar )
   774)   enddo
   775) 
   776)   !If we're not using a mirrored boundary condition, then make the two cells bordering the top and bottom
   777)   !material boundaries piecewise constant. Zeroing out the first and second moments, and setting the zeroth
   778)   !moment to the cell mean is sufficient to maintain conservation.
   779)   if (vert_remap_q_alg == 2) then
   780)     coefs(0,1:2) = a(1:2)
   781)     coefs(1:2,1:2) = 0.
   782)     coefs(0,nlev-1:nlev) = a(nlev-1:nlev)
   783)     coefs(1:2,nlev-1:nlev) = 0.D0
   784)   endif
   785) end function compute_ppm

   790) !Simple function computes the definite integral of a parabola in normalized coordinates, xi=(x-x0)/dx,
   791) !given two bounds. Make sure this gets inlined during compilation.
   792) function integrate_parabola( a , x1 , x2 )    result(mass)
   793)   implicit none
   794)   real(kind=real_kind), intent(in) :: a(0:2)  !Coefficients of the parabola
   795)   real(kind=real_kind), intent(in) :: x1      !lower domain bound for integration
   796)   real(kind=real_kind), intent(in) :: x2      !upper domain bound for integration
   797)   real(kind=real_kind)             :: mass
   798)   mass = a(0) * (x2 - x1) + a(1) * (x2 ** 2 - x1 ** 2) / 0.2D1 + a(2) * (x2 ** 3 - x1 ** 3) / 0.3D1
   799) end function integrate_parabola

   820) !
   821) ! two formulations.  both are conservative
   822) ! u grad Q formulation:
   823) !
   824) !    d/dt[ Q] +  U grad Q  +  eta_dot dp/dn dQ/dp  = 0
   825) !                            ( eta_dot dQ/dn )
   826) !
   827) !    d/dt[ dp/dn ] = div( dp/dn U ) + d/dn ( eta_dot dp/dn )

  1138)   subroutine Prim_Advec_Tracers_remap_rk2( elem , deriv , hvcoord , flt , hybrid , dt , tl , nets , nete )
  1139)     use perf_mod      , only : t_startf, t_stopf            ! _EXTERNAL
  1140)     use derivative_mod, only : divergence_sphere
  1141)     use control_mod   , only : vert_remap_q_alg, qsplit

  1151)     integer              , intent(in   ) :: nete
  1152) 
  1153)     real (kind=real_kind), dimension(np,np,2     ) :: gradQ
  1154)     real (kind=real_kind), dimension(np,np  ,nlev) :: dp_star
  1155)     real (kind=real_kind), dimension(np,np  ,nlev) :: dp_np1
  1156)     integer :: i,j,k,l,ie,q,nmin
  1157)     integer :: nfilt,rkstage,rhs_multiplier
  1158)     integer :: n0_qdp, np1_qdp

  1161)     call t_startf('prim_advec_tracers_remap_rk2')
  1162)     call TimeLevel_Qdp( tl, qsplit, n0_qdp, np1_qdp) !time levels for qdp are not the same
  1163)     rkstage = 3 !   3 stage RKSSP scheme, with optimal SSP CFL
  1164) 
  1165)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1166)     ! RK2 2D advection step
  1167)     ! note: stage 3 we take the oppertunity to DSS omega
  1168)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1169)     ! use these for consistent advection (preserve Q=1)
  1170)     ! derived%vdp_ave        =  mean horiz. flux:   U*dp
  1171)     ! derived%eta_dot_dpdn    =  mean vertical velocity (used for remap)
  1172)     ! derived%omega_p         =  advection code will DSS this for the physics, but otherwise 
  1173)     !                            it is not needed 
  1174)     ! Also: save a copy of div(U dp) in derived%div(:,:,:,1), which will be DSS'd 
  1175)     !       and a DSS'ed version stored in derived%div(:,:,:,2)
  1176)     do ie=nets,nete
  1177) #if (defined ELEMENT_OPENMP)
  1178) !$omp parallel do private(k, gradQ)
  1179) #endif
  1180)       do k=1,nlev
  1181)         ! div( U dp Q), 
  1182)         gradQ(:,:,1)=elem(ie)%derived%vn0(:,:,1,k)
  1183)         gradQ(:,:,2)=elem(ie)%derived%vn0(:,:,2,k)
  1184)         elem(ie)%derived%divdp(:,:,k) = divergence_sphere(gradQ,deriv,elem(ie))
  1185)       enddo
  1186)       elem(ie)%derived%divdp_proj(:,:,:) = elem(ie)%derived%divdp(:,:,:)
  1187)     enddo
  1188) 
  1189)     !rhs_multiplier is for obtaining dp_tracers at each stage:
  1190)     !dp_tracers(stage) = dp - rhs_multiplier*dt*divdp_proj
  1191)     rhs_multiplier = 0
  1192)     call euler_step( np1_qdp , n0_qdp  , dt/2 , elem , hvcoord , hybrid , deriv , nets , nete , DSSdiv_vdp_ave , rhs_multiplier )
  1193) 
  1194)     rhs_multiplier = 1
  1195)     call euler_step( np1_qdp , np1_qdp , dt/2 , elem , hvcoord , hybrid , deriv , nets , nete , DSSeta         , rhs_multiplier )
  1196) 
  1197)     rhs_multiplier = 2
  1198)     call euler_step( np1_qdp , np1_qdp , dt/2 , elem , hvcoord , hybrid , deriv , nets , nete , DSSomega       , rhs_multiplier )
  1199) 
  1200)     !to finish the 2D advection step, we need to average the t and t+2 results to get a second order estimate for t+1.  
  1201)     call qdp_time_avg( elem , rkstage , n0_qdp , np1_qdp , limiter_option , nu_p , nets , nete )
  1202) 
  1203)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1204)     !  Dissipation
  1205)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1206)     if ( limiter_option == 8  ) then
  1207)       ! dissipation was applied in RHS.  
  1208)     else
  1209)       call advance_hypervis_scalar(edgeadv,elem,hvcoord,hybrid,deriv,tl%np1,np1_qdp,nets,nete,dt)
  1210)     endif
  1211) 
  1212)     call t_stopf('prim_advec_tracers_remap_rk2')

  1233)                    ( elem(ie)%state%Qdp(:,:,:,1:qsize,n0_qdp) + &
  1234)                      (rkstage-1)*elem(ie)%state%Qdp(:,:,:,1:qsize,np1_qdp) ) / rkstage
  1235)     enddo

  1241)   subroutine euler_step( np1_qdp , n0_qdp , dt , elem , hvcoord , hybrid , deriv , nets , nete , DSSopt , rhs_multiplier )
  1242)   ! ===================================
  1243)   ! This routine is the basic foward
  1244)   ! euler component used to construct RK SSP methods
  1245)   !
  1246)   !           u(np1) = u(n0) + dt2*DSS[ RHS(u(n0)) ]
  1247)   !
  1248)   ! n0 can be the same as np1.  
  1249)   !
  1250)   ! DSSopt = DSSeta or DSSomega:   also DSS eta_dot_dpdn or omega
  1251)   !
  1252)   ! ===================================
  1253)   use kinds          , only : real_kind
  1254)   use dimensions_mod , only : np, npdg, nlev
  1255)   use hybrid_mod     , only : hybrid_t
  1256)   use element_mod    , only : element_t
  1257)   use derivative_mod , only : derivative_t, divergence_sphere, gradient_sphere, vorticity_sphere
  1258)   use edge_mod       , only : edgevpack, edgevunpack
  1259)   use bndry_mod      , only : bndry_exchangev
  1260)   use hybvcoord_mod  , only : hvcoord_t
  1261) #if USE_CUDA_FORTRAN
  1262)   use cuda_mod, only: euler_step_cuda
  1263) #endif
  1264)   implicit none
  1265)   integer              , intent(in   )         :: np1_qdp, n0_qdp
  1266)   real (kind=real_kind), intent(in   )         :: dt
  1267)   type (element_t)     , intent(inout), target :: elem(:)
  1268)   type (hvcoord_t)     , intent(in   )         :: hvcoord
  1269)   type (hybrid_t)      , intent(in   )         :: hybrid
  1270)   type (derivative_t)  , intent(in   )         :: deriv
  1271)   integer              , intent(in   )         :: nets
  1272)   integer              , intent(in   )         :: nete
  1273)   integer              , intent(in   )         :: DSSopt
  1274)   integer              , intent(in   )         :: rhs_multiplier
  1275) 
  1276)   ! local
  1277)   real(kind=real_kind), dimension(np,np                       ) :: divdp, dpdiss
  1278)   real(kind=real_kind), dimension(np,np,2                     ) :: gradQ
  1279)   real(kind=real_kind), dimension(np,np,2,nlev                ) :: Vstar
  1280)   real(kind=real_kind), dimension(np,np  ,nlev                ) :: Qtens
  1281)   real(kind=real_kind), dimension(np,np  ,nlev                ) :: dp,dp_star
  1282)   real(kind=real_kind), dimension(np,np  ,nlev,qsize,nets:nete) :: Qtens_biharmonic
  1283)   real(kind=real_kind), pointer, dimension(:,:,:)               :: DSSvar
  1284)   real(kind=real_kind) :: dp0
  1285)   integer :: ie,q,i,j,k
  1286)   integer :: rhs_viss = 0
  1287) 
  1288)   if ( npdg > 0 ) then
  1289)     call euler_step_dg( np1_qdp , n0_qdp , dt , elem , hvcoord , hybrid , deriv , nets , nete , DSSopt , rhs_multiplier )
  1290)     return
  1291)   endif
  1292) #if USE_CUDA_FORTRAN
  1293)   call euler_step_cuda( np1_qdp , n0_qdp , dt , elem , hvcoord , hybrid , deriv , nets , nete , DSSopt , rhs_multiplier )
  1294)   return
  1295) #endif
  1296) ! call t_barrierf('sync_euler_step', hybrid%par%comm)
  1297) !   call t_startf('euler_step')
  1298) 
  1299)   !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1300)   !   compute Q min/max values for lim8
  1301)   !   compute biharmonic mixing term f
  1302)   !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1303)   rhs_viss = 0
  1304)   if ( limiter_option == 8  ) then
  1305)     ! when running lim8, we also need to limit the biharmonic, so that term needs
  1306)     ! to be included in each euler step.  three possible algorithms here:
  1307)     ! 1) most expensive:
  1308)     !     compute biharmonic (which also computes qmin/qmax) during all 3 stages
  1309)     !     be sure to set rhs_viss=1
  1310)     !     cost:  3 biharmonic steps with 3 DSS
  1311)     !
  1312)     ! 2) cheapest:
  1313)     !     compute biharmonic (which also computes qmin/qmax) only on first stage
  1314)     !     be sure to set rhs_viss=3
  1315)     !     reuse qmin/qmax for all following stages (but update based on local qmin/qmax)
  1316)     !     cost:  1 biharmonic steps with 1 DSS
  1317)     !     main concern:  viscosity 
  1318)     !     
  1319)     ! 3)  compromise:
  1320)     !     compute biharmonic (which also computes qmin/qmax) only on last stage
  1321)     !     be sure to set rhs_viss=3
  1322)     !     compute qmin/qmax directly on first stage
  1323)     !     reuse qmin/qmax for 2nd stage stage (but update based on local qmin/qmax)
  1324)     !     cost:  1 biharmonic steps, 2 DSS
  1325)     !
  1326)     !  NOTE  when nu_p=0 (no dissipation applied in dynamics to dp equation), we should
  1327)     !        apply dissipation to Q (not Qdp) to preserve Q=1
  1328)     !        i.e.  laplace(Qdp) ~  dp0 laplace(Q)                
  1329)     !        for nu_p=nu_q>0, we need to apply dissipation to Q * diffusion_dp
  1330)     !
  1331)     ! initialize dp, and compute Q from Qdp (and store Q in Qtens_biharmonic)
  1332)     do ie = nets , nete
  1333)       ! add hyperviscosity to RHS.  apply to Q at timelevel n0, Qdp(n0)/dp
  1334) #if (defined ELEMENT_OPENMP)
  1335) !$omp parallel do private(k, q)
  1336) #endif
  1337)       do k = 1 , nlev    !  Loop index added with implicit inversion (AAM)
  1338)         dp(:,:,k) = elem(ie)%derived%dp(:,:,k) - rhs_multiplier*dt*elem(ie)%derived%divdp_proj(:,:,k) 
  1339)         do q = 1 , qsize
  1340)           Qtens_biharmonic(:,:,k,q,ie) = elem(ie)%state%Qdp(:,:,k,q,n0_qdp)/dp(:,:,k)
  1341)         enddo
  1342)       enddo
  1343)     enddo
  1344) 
  1345)     ! compute element qmin/qmax
  1346)     if ( rhs_multiplier == 0 ) then
  1347)       do ie = nets , nete
  1348)         do k = 1 , nlev    
  1349)           do q = 1 , qsize
  1350)             qmin(k,q,ie)=minval(Qtens_biharmonic(:,:,k,q,ie))
  1351)             qmax(k,q,ie)=maxval(Qtens_biharmonic(:,:,k,q,ie))
  1352)             qmin(k,q,ie)=max(qmin(k,q,ie),0d0)
  1353)           enddo
  1354)         enddo
  1355)       enddo
  1356)       ! update qmin/qmax based on neighbor data for lim8
  1357)       call neighbor_minmax(elem,hybrid,edgeAdvQ2,nets,nete,qmin(:,:,nets:nete),qmax(:,:,nets:nete))
  1358)     endif
  1359) 
  1360)     ! lets just reuse the old neighbor min/max, but update based on local data
  1361)     if ( rhs_multiplier == 1 ) then
  1362)       do ie = nets , nete
  1363)         do k = 1 , nlev    !  Loop index added with implicit inversion (AAM)
  1364)           do q = 1 , qsize
  1365)             qmin(k,q,ie)=min(qmin(k,q,ie),minval(Qtens_biharmonic(:,:,k,q,ie)))
  1366)             qmin(k,q,ie)=max(qmin(k,q,ie),0d0)
  1367)             qmax(k,q,ie)=max(qmax(k,q,ie),maxval(Qtens_biharmonic(:,:,k,q,ie)))
  1368)           enddo
  1369)         enddo
  1370)       enddo
  1371)     endif
  1372) 
  1373)     ! get niew min/max values, and also compute biharmonic mixing term
  1374)     if ( rhs_multiplier == 2 ) then
  1375)       rhs_viss = 3
  1376)       ! compute element qmin/qmax  
  1377)       do ie = nets , nete
  1378)         do k = 1  ,nlev    
  1379)           do q = 1 , qsize
  1380)             qmin(k,q,ie)=minval(Qtens_biharmonic(:,:,k,q,ie))
  1381)             qmax(k,q,ie)=maxval(Qtens_biharmonic(:,:,k,q,ie))
  1382)             qmin(k,q,ie)=max(qmin(k,q,ie),0d0)
  1383)           enddo
  1384)         enddo
  1385)       enddo
  1386)       ! two scalings depending on nu_p:
  1387)       ! nu_p=0:    qtens_biharmonic *= dp0                   (apply viscsoity only to q)
  1388)       ! nu_p>0):   qtens_biharmonc *= elem()%psdiss_ave      (for consistency, if nu_p=nu_q)
  1389)       if ( nu_p > 0 ) then
  1390)         do ie = nets , nete
  1391) #if (defined ELEMENT_OPENMP)
  1392)           !$omp parallel do private(k, q, dp0, dpdiss)
  1393) #endif
  1394)           do k = 1 , nlev    
  1395)             dp0 = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
  1396)                   ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*hvcoord%ps0
  1397) #if 0
  1398)             dpdiss(:,:) = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
  1399)                           ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*elem(ie)%derived%psdiss_ave(:,:)
  1400) #else
  1401)             dpdiss(:,:) = elem(ie)%derived%dpdiss_ave(:,:,k)
  1402) #endif
  1403)             do q = 1 , qsize
  1404)               ! NOTE: divide by dp0 since we multiply by dp0 below
  1405)               Qtens_biharmonic(:,:,k,q,ie)=Qtens_biharmonic(:,:,k,q,ie)*dpdiss(:,:)/dp0
  1406)             enddo
  1407)           enddo
  1408)         enddo
  1409)       endif
  1410)       call biharmonic_wk_scalar_minmax( elem , qtens_biharmonic , deriv , edgeAdvQ3 , hybrid , &
  1411)            nets , nete , qmin(:,:,nets:nete) , qmax(:,:,nets:nete) )
  1412)       do ie = nets , nete
  1413) #if (defined ELEMENT_OPENMP)
  1414)         !$omp parallel do private(k, q, dp0)
  1415) #endif
  1416)         do k = 1 , nlev    !  Loop inversion (AAM)
  1417)           dp0 = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
  1418)                 ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*hvcoord%ps0
  1419)           do q = 1 , qsize
  1420)             ! note: biharmonic_wk() output has mass matrix already applied. Un-apply since we apply again below:
  1421)             qtens_biharmonic(:,:,k,q,ie) = &
  1422)                      -rhs_viss*dt*nu_q*dp0*Qtens_biharmonic(:,:,k,q,ie) / elem(ie)%spheremp(:,:)
  1423)           enddo
  1424)         enddo
  1425)       enddo
  1426)     endif
  1427)   endif  ! compute biharmonic mixing term and qmin/qmax
  1428) 
  1429) 
  1430)   !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1431)   !   2D Advection step
  1432)   !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1433)   do ie = nets , nete
  1434)     ! note: eta_dot_dpdn is actually dimension nlev+1, but nlev+1 data is
  1435)     ! all zero so we only have to DSS 1:nlev
  1436)     if ( DSSopt == DSSeta         ) DSSvar => elem(ie)%derived%eta_dot_dpdn(:,:,:)
  1437)     if ( DSSopt == DSSomega       ) DSSvar => elem(ie)%derived%omega_p(:,:,:)
  1438)     if ( DSSopt == DSSdiv_vdp_ave ) DSSvar => elem(ie)%derived%divdp_proj(:,:,:)
  1439) 
  1440)     ! Compute velocity used to advance Qdp 
  1441) #if (defined ELEMENT_OPENMP)
  1442) !$omp parallel do private(k)
  1443) #endif
  1444)     do k = 1 , nlev    !  Loop index added (AAM)
  1445)       ! derived variable divdp_proj() (DSS'd version of divdp) will only be correct on 2nd and 3rd stage
  1446)       ! but that's ok because rhs_multiplier=0 on the first stage:
  1447)       dp(:,:,k) = elem(ie)%derived%dp(:,:,k) - rhs_multiplier * dt * elem(ie)%derived%divdp_proj(:,:,k) 
  1448)       Vstar(:,:,1,k) = elem(ie)%derived%vn0(:,:,1,k) / dp(:,:,k)
  1449)       Vstar(:,:,2,k) = elem(ie)%derived%vn0(:,:,2,k) / dp(:,:,k)
  1450)     enddo
  1451) 
  1452)     ! advance Qdp
  1453) #if (defined ELEMENT_OPENMP)
  1454) !$omp parallel do private(q,k,gradQ,dp_star,qtens,dpdiss)
  1455) #endif
  1456)     do q = 1 , qsize
  1457)       do k = 1 , nlev  !  dp_star used as temporary instead of divdp (AAM)
  1458)         ! div( U dp Q), 
  1459)         gradQ(:,:,1) = Vstar(:,:,1,k) * elem(ie)%state%Qdp(:,:,k,q,n0_qdp)
  1460)         gradQ(:,:,2) = Vstar(:,:,2,k) * elem(ie)%state%Qdp(:,:,k,q,n0_qdp)
  1461)         dp_star(:,:,k) = divergence_sphere( gradQ , deriv , elem(ie) )
  1462)         Qtens(:,:,k) = elem(ie)%state%Qdp(:,:,k,q,n0_qdp) - dt * dp_star(:,:,k)
  1463)         ! optionally add in hyperviscosity computed above:
  1464)         if ( rhs_viss /= 0 ) Qtens(:,:,k) = Qtens(:,:,k) + Qtens_biharmonic(:,:,k,q,ie)
  1465)       enddo
  1466)          
  1467)       if ( limiter_option == 8 ) then
  1468)         do k = 1 , nlev  ! Loop index added (AAM)
  1469)           ! UN-DSS'ed dp at timelevel n0+1:  
  1470)           dp_star(:,:,k) = dp(:,:,k) - dt * elem(ie)%derived%divdp(:,:,k)  
  1471)           if ( nu_p > 0 .and. rhs_viss /= 0 ) then
  1472)             ! add contribution from UN-DSS'ed PS dissipation
  1473) !            dpdiss(:,:) = ( hvcoord%hybi(k+1) - hvcoord%hybi(k) ) * elem(ie)%derived%psdiss_biharmonic(:,:)
  1474)             dpdiss(:,:) = elem(ie)%derived%dpdiss_biharmonic(:,:,k)
  1475)             dp_star(:,:,k) = dp_star(:,:,k) - rhs_viss * dt * nu_q * dpdiss(:,:) / elem(ie)%spheremp(:,:)
  1476)           endif
  1477)         enddo
  1478)         ! apply limiter to Q = Qtens / dp_star 
  1479)         call limiter_optim_iter_full( Qtens(:,:,:) , elem(ie)%spheremp(:,:) , qmin(:,q,ie) , &
  1480)                                       qmax(:,q,ie) , dp_star(:,:,:) )
  1481)       endif
  1482) 
  1483)       ! apply mass matrix, overwrite np1 with solution:
  1484)       ! dont do this earlier, since we allow np1_qdp == n0_qdp 
  1485)       ! and we dont want to overwrite n0_qdp until we are done using it
  1486)       do k = 1 , nlev
  1487)         elem(ie)%state%Qdp(:,:,k,q,np1_qdp) = elem(ie)%spheremp(:,:) * Qtens(:,:,k) 
  1488)       enddo
  1489) 
  1490)       if ( limiter_option == 4 ) then
  1491)         !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!1
  1492)         ! sign-preserving limiter, applied after mass matrix
  1493)         !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!1
  1494)         call limiter2d_zero( elem(ie)%state%Qdp(:,:,:,q,np1_qdp) , hvcoord ) 
  1495)       endif
  1496)     enddo
  1497) 
  1498)     if ( DSSopt == DSSno_var ) then
  1499)       call edgeVpack(edgeAdv    , elem(ie)%state%Qdp(:,:,:,:,np1_qdp) , nlev*qsize , 0 , elem(ie)%desc )
  1500)     else
  1501)       call edgeVpack(edgeAdv_p1 , elem(ie)%state%Qdp(:,:,:,:,np1_qdp) , nlev*qsize , 0 , elem(ie)%desc )
  1502)       ! also DSS extra field
  1503) #if (defined ELEMENT_OPENMP)
  1504) !$omp parallel do private(k)
  1505) #endif
  1506)       do k = 1 , nlev
  1507)         DSSvar(:,:,k) = elem(ie)%spheremp(:,:) * DSSvar(:,:,k) 
  1508)       enddo
  1509)       call edgeVpack( edgeAdv_p1 , DSSvar(:,:,1:nlev) , nlev , nlev*qsize , elem(ie)%desc )
  1510)     endif
  1511)   enddo
  1512) 
  1513)   if ( DSSopt == DSSno_var ) then
  1514)     call bndry_exchangeV( hybrid , edgeAdv    )
  1515)   else
  1516)     call bndry_exchangeV( hybrid , edgeAdv_p1 )
  1517)   endif
  1518) 
  1519)   do ie = nets , nete
  1520)     if ( DSSopt == DSSeta         ) DSSvar => elem(ie)%derived%eta_dot_dpdn(:,:,:)
  1521)     if ( DSSopt == DSSomega       ) DSSvar => elem(ie)%derived%omega_p(:,:,:)
  1522)     if ( DSSopt == DSSdiv_vdp_ave ) DSSvar => elem(ie)%derived%divdp_proj(:,:,:)
  1523) 
  1524)     if ( DSSopt == DSSno_var ) then
  1525)       call edgeVunpack( edgeAdv    , elem(ie)%state%Qdp(:,:,:,:,np1_qdp) , nlev*qsize , 0 , elem(ie)%desc )
  1526) #if (defined ELEMENT_OPENMP)
  1527) !$omp parallel do private(k,q)
  1528) #endif
  1529)       do q = 1 , qsize
  1530)         do k = 1 , nlev    !  Potential loop inversion (AAM)
  1531)           elem(ie)%state%Qdp(:,:,k,q,np1_qdp) = elem(ie)%rspheremp(:,:) * elem(ie)%state%Qdp(:,:,k,q,np1_qdp)
  1532)         enddo
  1533)       enddo
  1534)     else
  1535)       call edgeVunpack( edgeAdv_p1 , elem(ie)%state%Qdp(:,:,:,:,np1_qdp) , nlev*qsize , 0 , elem(ie)%desc )
  1536) #if (defined ELEMENT_OPENMP)
  1537)       !$omp parallel do private(k,q)
  1538) #endif
  1539)       do q = 1 , qsize
  1540)         do k = 1 , nlev    !  Potential loop inversion (AAM)
  1541)           elem(ie)%state%Qdp(:,:,k,q,np1_qdp) = elem(ie)%rspheremp(:,:) * elem(ie)%state%Qdp(:,:,k,q,np1_qdp)
  1542)         enddo
  1543)       enddo
  1544)       call edgeVunpack( edgeAdv_p1 , DSSvar(:,:,1:nlev) , nlev , qsize*nlev , elem(ie)%desc )
  1545)        
  1546)       do k = 1 , nlev
  1547)         DSSvar(:,:,k) = DSSvar(:,:,k) * elem(ie)%rspheremp(:,:)
  1548)       enddo
  1549)     endif
  1550)   enddo
  1551) #ifdef DEBUGOMP
  1552) #if (! defined ELEMENT_OPENMP)
  1553) !$OMP BARRIER
  1554) #endif
  1555) #endif
  1556) !   call t_stopf('euler_step')
  1557)   end subroutine euler_step

  1632) 
  1633)      ! note: eta_dot_dpdn is actually dimension nlev+1, but nlev+1 data is
  1634)      ! all zero so we only have to DSS 1:nlev
  1635)      if ( DSSopt == DSSeta) DSSvar => elem(ie)%derived%eta_dot_dpdn(:,:,:)
  1636)      if ( DSSopt == DSSomega) DSSvar => elem(ie)%derived%omega_p(:,:,:)
  1637)      if ( DSSopt == DSSdiv_vdp_ave) DSSvar => elem(ie)%derived%divdp_proj(:,:,:)
  1638) 
  1639)      if(DSSopt==DSSno_var)then
  1640) 	call edgeVpack(edgeAdv,elem(ie)%state%Qdp(:,:,:,:,n0_qdp),nlev*qsize,0,elem(ie)%desc)
  1641)      else
  1642) 	call edgeVpack(edgeAdv_p1,elem(ie)%state%Qdp(:,:,:,:,n0_qdp),nlev*qsize,0,elem(ie)%desc)
  1643) 	! also DSS extra field
  1644) #if (defined ELEMENT_OPENMP)
  1645) !$omp parallel do private(k)
  1646) #endif
  1647) 	do k=1,nlev
  1648) 	    DSSvar(:,:,k) = elem(ie)%spheremp(:,:)*DSSvar(:,:,k) 
  1649) 	enddo
  1650) 	call edgeVpack(edgeAdv_p1,DSSvar(:,:,1:nlev),nlev,nlev*qsize,elem(ie)%desc)
  1651)      endif
  1652) 
  1653)   end do
  1654) 
  1655)   if(DSSopt==DSSno_var)then
  1656)      call bndry_exchangeV(hybrid,edgeAdv)
  1657)   else
  1658)      call bndry_exchangeV(hybrid,edgeAdv_p1)
  1659)   endif
  1660) 
  1661)   do ie=nets,nete
  1662) 
  1663)      if ( DSSopt == DSSeta) DSSvar => elem(ie)%derived%eta_dot_dpdn(:,:,:)
  1664)      if ( DSSopt == DSSomega) DSSvar => elem(ie)%derived%omega_p(:,:,:)
  1665)      if ( DSSopt == DSSdiv_vdp_ave) DSSvar => elem(ie)%derived%divdp_proj(:,:,:)
  1666) 
  1667)      if(DSSopt==DSSno_var)then
  1668) 	call edgeDGVunpack(edgeAdv,qedges,nlev*qsize,0,elem(ie)%desc)
  1669)      else
  1670) 	call edgeDGVunpack(edgeAdv_p1,qedges,nlev*qsize,0,elem(ie)%desc)
  1671) 	call edgeVunpack(edgeAdv_p1,DSSvar(:,:,1:nlev),nlev,qsize*nlev,elem(ie)%desc)
  1672) 	do k=1,nlev
  1673) 	  DSSvar(:,:,k)=DSSvar(:,:,k)*elem(ie)%rspheremp(:,:)
  1674) 	enddo
  1675)      endif
  1676) 
  1677)      ! compute flux and advection term
  1678) #if (defined ELEMENT_OPENMP)
  1679) !$omp parallel do private(k)
  1680) #endif
  1681)      do k=1,nlev
  1682)         dp(:,:,k) = elem(ie)%derived%dp(:,:,k) - &
  1683)              rhs_multiplier*dt*elem(ie)%derived%divdp_proj(:,:,k) 
  1684)         Vstar(:,:,1,k) = elem(ie)%derived%vn0(:,:,1,k)/dp(:,:,k)
  1685)         Vstar(:,:,2,k) = elem(ie)%derived%vn0(:,:,2,k)/dp(:,:,k)
  1686)      enddo
  1687) 
  1688) #if (defined ELEMENT_OPENMP)
  1689) !$omp parallel do private(k,q,vtemp,divdp,pshat,i,j)
  1690) #endif
  1691)      do q=1,qsize
  1692)         do k=1,nlev
  1693)            vtemp(:,:,1)=elem(ie)%state%Qdp(:,:,k,q,n0_qdp)*Vstar(:,:,1,k)
  1694)            vtemp(:,:,2)=elem(ie)%state%Qdp(:,:,k,q,n0_qdp)*Vstar(:,:,2,k)
  1695)            
  1696)            divdp = divergence_sphere_wk(vtemp,deriv,elem(ie)) + &
  1697)                 edge_flux_u_cg( Vstar(:,:,:,k), elem(ie)%state%Qdp(:,:,k,q,n0_qdp),qedges(:,:,k,q),&
  1698)                 deriv, elem(ie), u_is_contra=.false.)
  1699) 
  1700)            ! advance in time. GLL quadrature, cardinal function basis, under-integrated.  
  1701)            ! local mass matrix is diagonal, with entries elem(ie)%spheremp(),
  1702)            ! so we divide through by elem(ie)%spheremp().
  1703)            elem(ie)%state%Qdp(:,:,k,q,np1_qdp)=elem(ie)%state%Qdp(:,:,k,q,n0_qdp) - dt*divdp/elem(ie)%spheremp
  1704)            
  1705)            if (npdg<np) then
  1706)               ! modal timestep, with exact integration.  using prognostic variable: p*metdet
  1707)               ! local mass matrix is diagonal assuming npdg<np so that GLL quadrature is exact)
  1708)               ! (note: GLL/modal conversion comutes with time-stepping)
  1709)               
  1710)               ! compute modal coefficients of p*metdet
  1711)               ! (spherical inner-product of Legendre polynomial and p)
  1712)               pshat = gll_to_dgmodal(elem(ie)%state%Qdp(:,:,k,q,np1_qdp)*elem(ie)%metdet(:,:),deriv)
  1713) 
  1714)               ! modal based limiter goes here
  1715)               ! apply a little dissipation to last mode:
  1716)               do j=1,npdg
  1717)               do i=1,npdg
  1718)                  !if ( (i-1)+(j-1) == 4) pshat(i,j)=pshat(i,j)*.75
  1719)                  !if ( (i-1)+(j-1) == 3) pshat(i,j)=pshat(i,j)*.90
  1720)                  if ( i==npdg) pshat(i,j)=pshat(i,j)*.90
  1721)                  if ( j==npdg) pshat(i,j)=pshat(i,j)*.90
  1722)               enddo
  1723)               enddo
  1724) 
  1725) 
  1726)               ! evalute modal expanion of p*metdet on GLL points
  1727)               divdp=dgmodal_to_gll(pshat,deriv)  
  1728) 
  1729)               ! convert from p*metdet back to p:
  1730)               elem(ie)%state%Qdp(:,:,k,q,np1_qdp)=divdp/elem(ie)%metdet(:,:)
  1731)            endif
  1732)         enddo
  1733)         if(limiter_option == 4)then
  1734)            ! reuse CG limiter, which wants Qdp*spheremp:
  1735)            do k=1,nlev
  1736)               elem(ie)%state%Qdp(:,:,k,q,np1_qdp)=elem(ie)%state%Qdp(:,:,k,q,np1_qdp)*elem(ie)%spheremp(:,:)
  1737)            enddo
  1738)            call limiter2d_zero(elem(ie)%state%Qdp(:,:,:,q,np1_qdp),hvcoord)
  1739)            do k=1,nlev
  1740)               elem(ie)%state%Qdp(:,:,k,q,np1_qdp)=elem(ie)%state%Qdp(:,:,k,q,np1_qdp)/elem(ie)%spheremp(:,:)
  1741)            enddo
  1742)         endif
  1743)      end do
  1744)   end do
  1745) #ifdef DEBUGOMP
  1746) #if (! defined ELEMENT_OPENMP)
  1747) !$OMP BARRIER
  1748) #endif
  1749) #endif
  1750)   call t_stopf('euler_step_dg')
  1751)   
  1752)   end subroutine euler_step_dg

  1758)   subroutine limiter_optim_iter_full(ptens,sphweights,minp,maxp,dpmass)
  1759)     !THIS IS A NEW VERSION OF LIM8, POTENTIALLY FASTER BECAUSE INCORPORATES KNOWLEDGE FROM
  1760)     !PREVIOUS ITERATIONS
  1761)     
  1762)     !The idea here is the following: We need to find a grid field which is closest
  1763)     !to the initial field (in terms of weighted sum), but satisfies the min/max constraints.
  1764)     !So, first we find values which do not satisfy constraints and bring these values
  1765)     !to a closest constraint. This way we introduce some mass change (addmass),
  1766)     !so, we redistribute addmass in the way that l2 error is smallest. 
  1767)     !This redistribution might violate constraints thus, we do a few iterations. 
  1768)     use kinds         , only : real_kind
  1769)     use dimensions_mod, only : np, np, nlev
  1770)     real (kind=real_kind), dimension(np*np,nlev), intent(inout)            :: ptens
  1771)     real (kind=real_kind), dimension(np*np     ), intent(in   )            :: sphweights
  1772)     real (kind=real_kind), dimension(      nlev), intent(inout)            :: minp
  1773)     real (kind=real_kind), dimension(      nlev), intent(inout)            :: maxp
  1774)     real (kind=real_kind), dimension(np*np,nlev), intent(in   ), optional  :: dpmass
  1775)  
  1776)     real (kind=real_kind), dimension(np*np,nlev) :: weights
  1777)     integer  k1, k, i, j, iter, i1, i2
  1778)     integer :: whois_neg(np*np), whois_pos(np*np), neg_counter, pos_counter
  1779)     real (kind=real_kind) :: addmass, weightssum, mass
  1780)     real (kind=real_kind) :: x(np*np),c(np*np)
  1781)     real (kind=real_kind) :: al_neg(np*np), al_pos(np*np), howmuch
  1782)     real (kind=real_kind) :: tol_limiter = 1e-15
  1783)     integer, parameter :: maxiter = 5
  1784) 
  1785)     do k = 1 , nlev
  1786)       weights(:,k) = sphweights(:) * dpmass(:,k)
  1787)       ptens(:,k) = ptens(:,k) / dpmass(:,k)
  1788)     enddo
  1789) 
  1790)     do k = 1 , nlev
  1791)       c = weights(:,k)
  1792)       x = ptens(:,k)
  1793) 
  1794)       mass = sum(c*x)
  1795) 
  1796)       ! relax constraints to ensure limiter has a solution:
  1797)       ! This is only needed if runnign with the SSP CFL>1 or 
  1798)       ! due to roundoff errors
  1799)       if( (mass / sum(c)) < minp(k) ) then
  1800)         minp(k) = mass / sum(c)
  1801)       endif
  1802)       if( (mass / sum(c)) > maxp(k) ) then
  1803)         maxp(k) = mass / sum(c)
  1804)       endif
  1805) 
  1806)       addmass = 0.0d0
  1807)       pos_counter = 0;
  1808)       neg_counter = 0;
  1809)       
  1810)       ! apply constraints, compute change in mass caused by constraints 
  1811)       do k1 = 1 , np*np
  1812)         if ( ( x(k1) >= maxp(k) ) ) then
  1813)           addmass = addmass + ( x(k1) - maxp(k) ) * c(k1)
  1814)           x(k1) = maxp(k)
  1815)           whois_pos(k1) = -1
  1816)         else
  1817)           pos_counter = pos_counter+1;
  1818)           whois_pos(pos_counter) = k1;
  1819)         endif
  1820)         if ( ( x(k1) <= minp(k) ) ) then
  1821)           addmass = addmass - ( minp(k) - x(k1) ) * c(k1)
  1822)           x(k1) = minp(k)
  1823)           whois_neg(k1) = -1
  1824)         else
  1825)           neg_counter = neg_counter+1;
  1826)           whois_neg(neg_counter) = k1;
  1827)         endif
  1828)       enddo
  1829)       
  1830)       ! iterate to find field that satifies constraints and is l2-norm closest to original 
  1831)       weightssum = 0.0d0
  1832)       if ( addmass > 0 ) then
  1833)         do i2 = 1 , maxIter
  1834)           weightssum = 0.0
  1835)           do k1 = 1 , pos_counter
  1836)             i1 = whois_pos(k1)
  1837)             weightssum = weightssum + c(i1)
  1838)             al_pos(i1) = maxp(k) - x(i1)
  1839)           enddo
  1840)           
  1841)           if( ( pos_counter > 0 ) .and. ( addmass > tol_limiter * abs(mass) ) ) then
  1842)             do k1 = 1 , pos_counter
  1843)               i1 = whois_pos(k1)
  1844)               howmuch = addmass / weightssum
  1845)               if ( howmuch > al_pos(i1) ) then
  1846)                 howmuch = al_pos(i1)
  1847)                 whois_pos(k1) = -1
  1848)               endif
  1849)               addmass = addmass - howmuch * c(i1)
  1850)               weightssum = weightssum - c(i1)
  1851)               x(i1) = x(i1) + howmuch
  1852)             enddo
  1853)             !now sort whois_pos and get a new number for pos_counter
  1854)             !here neg_counter and whois_neg serve as temp vars
  1855)             neg_counter = pos_counter
  1856)             whois_neg = whois_pos
  1857)             whois_pos = -1
  1858)             pos_counter = 0
  1859)             do k1 = 1 , neg_counter
  1860)               if ( whois_neg(k1) .ne. -1 ) then
  1861)                 pos_counter = pos_counter+1
  1862)                 whois_pos(pos_counter) = whois_neg(k1)
  1863)               endif
  1864)             enddo
  1865)           else
  1866)             exit
  1867)           endif
  1868)         enddo
  1869)       else
  1870)          do i2 = 1 , maxIter
  1871)            weightssum = 0.0
  1872)            do k1 = 1 , neg_counter
  1873)              i1 = whois_neg(k1)
  1874)              weightssum = weightssum + c(i1)
  1875)              al_neg(i1) = x(i1) - minp(k)
  1876)            enddo
  1877)            
  1878)            if ( ( neg_counter > 0 ) .and. ( (-addmass) > tol_limiter * abs(mass) ) ) then
  1879)              do k1 = 1 , neg_counter
  1880)                i1 = whois_neg(k1)
  1881)                howmuch = -addmass / weightssum
  1882)                if ( howmuch > al_neg(i1) ) then
  1883)                  howmuch = al_neg(i1)
  1884)                  whois_neg(k1) = -1
  1885)                endif
  1886)                addmass = addmass + howmuch * c(i1)
  1887)                weightssum = weightssum - c(i1)
  1888)                x(i1) = x(i1) - howmuch
  1889)              enddo
  1890)              !now sort whois_pos and get a new number for pos_counter
  1891)              !here pos_counter and whois_pos serve as temp vars
  1892)              pos_counter = neg_counter
  1893)              whois_pos = whois_neg
  1894)              whois_neg = -1
  1895)              neg_counter = 0
  1896)              do k1 = 1 , pos_counter
  1897)                if ( whois_pos(k1) .ne. -1 ) then
  1898)                  neg_counter = neg_counter+1
  1899)                  whois_neg(neg_counter) = whois_pos(k1)
  1900)                endif
  1901)              enddo
  1902)            else
  1903)              exit
  1904)            endif
  1905)          enddo
  1906)       endif
  1907)       
  1908)       ptens(:,k) = x
  1909)     enddo
  1910)     
  1911)     do k = 1 , nlev
  1912)       ptens(:,k) = ptens(:,k) * dpmass(:,k)
  1913)     enddo
  1914)   end subroutine limiter_optim_iter_full

  1920)   subroutine limiter2d_minmax(Q,dp,hvcoord,spheremp,qmin,qmax)
  1921) !
  1922) ! mass conserving limiter (2D only).  to be called just before DSS
  1923) !
  1924) ! in pure 2D advection, the element mass will not be negative before DSS
  1925) ! this routine will redistribute to remove negative values (conservative)
  1926) !
  1927) ! if used in 3D, should be applied with 2D/vertical split advection
  1928) ! 
  1929) ! call with Qdp and assocated dp
  1930) !
  1931) !
  1932)   implicit none
  1933)   real (kind=real_kind), intent(inout) :: Q(np,np,nlev)
  1934)   real (kind=real_kind), intent(in   ) :: spheremp(np,np)
  1935)   real (kind=real_kind), intent(in   ) ::  dp(np,np,nlev)
  1936)   type (hvcoord_t)     , intent(in   ) :: hvcoord
  1937) 
  1938)   ! local
  1939)   integer i,j,k
  1940)   real (kind=real_kind) :: mass,mass_new,area,qmin(nlev),qmax(nlev),mass2
  1941) 
  1942) #if (defined ELEMENT_OPENMP)
  1943) !$omp parallel do private(k,mass,area,mass2,mass_new,i,j)
  1944) #endif
  1945)   do k = 1 , nlev
  1946)     mass = sum( Q(:,:,k)*spheremp(:,:) )
  1947)     area = sum( dp(:,:,k)*spheremp(:,:) )
  1948) 
  1949)     Q(:,:,k) = Q(:,:,k) / dp(:,:,k)  ! % convert to concentration
  1950) 
  1951) !   if (mass>0) print *,k,mass/area,qmin(k),qmax(k)
  1952)     ! max limiter
  1953)     if ( maxval(Q(:,:,k)) > qmax(k) ) then
  1954)       Q(:,:,k) = qmax(k) - Q(:,:,k)      ! some of these will be negative
  1955)       mass2 = area * qmax(k) - mass
  1956) 
  1957)       if (mass2 < 0) Q(:,:,k) = -Q(:,:,k) 
  1958)       mass_new = 0
  1959)       do j = 1 , np
  1960)         do i = 1 , np
  1961)           if ( Q(i,j,k) < 0 ) then
  1962)             Q(i,j,k) = 0
  1963)           else
  1964)             mass_new = mass_new + Q(i,j,k) * dp(i,j,k) * spheremp(i,j)
  1965)           endif
  1966)         enddo
  1967)       enddo
  1968)     
  1969)       ! now scale the all positive values to restore mass
  1970)       if ( mass_new > 0 ) Q(:,:,k) = Q(:,:,k) * abs(mass2) / mass_new
  1971)       if ( mass2    < 0 ) Q(:,:,k) = -Q(:,:,k) 
  1972)       
  1973)       Q(:,:,k) = qmax(k) - Q(:,:,k)
  1974)     endif
  1975) 
  1976)     ! min limiter
  1977)     if ( minval(Q(:,:,k)) < qmin(k) ) then
  1978)       Q(:,:,k) = Q(:,:,k) - qmin(k)
  1979)       mass2 = mass - area * qmin(k)
  1980)       ! negative mass.  so reduce all postive values to zero 
  1981)       ! then increase negative values as much as possible
  1982)       if ( mass2 < 0 ) Q(:,:,k) = -Q(:,:,k) 
  1983)       mass_new = 0
  1984)       do j = 1 , np
  1985)         do i = 1 , np
  1986)           if ( Q(i,j,k) < 0 ) then
  1987)             Q(i,j,k) = 0
  1988)           else
  1989)             mass_new = mass_new + Q(i,j,k) * dp(i,j,k) * spheremp(i,j)
  1990)           endif
  1991)         enddo
  1992)       enddo
  1993)       
  1994)       ! now scale the all positive values to restore mass
  1995)       if ( mass_new > 0 ) Q(:,:,k) = Q(:,:,k) * abs(mass2) / mass_new
  1996)       if ( mass2    < 0 ) Q(:,:,k) = -Q(:,:,k) 
  1997) 
  1998)       Q(:,:,k) = Q(:,:,k) + qmin(k)
  1999)     endif
  2000) 
  2001)     Q(:,:,k) = Q(:,:,k) * dp(:,:,k)
  2002)   enddo
  2003)   end subroutine limiter2d_minmax

  2008)   subroutine limiter2d_zero(Q,hvcoord)
  2009)   ! mass conserving zero limiter (2D only).  to be called just before DSS
  2010)   !
  2011)   ! this routine is called inside a DSS loop, and so Q had already
  2012)   ! been multiplied by the mass matrix.  Thus dont include the mass
  2013)   ! matrix when computing the mass = integral of Q over the element
  2014)   !
  2015)   ! ps is only used when advecting Q instead of Qdp
  2016)   ! so ps should be at one timelevel behind Q
  2017)   implicit none
  2018)   real (kind=real_kind), intent(inout) :: Q(np,np,nlev)
  2019)   type (hvcoord_t)     , intent(in   ) :: hvcoord
  2020) 
  2021)   ! local
  2022)   real (kind=real_kind) :: dp(np,np)
  2023)   real (kind=real_kind) :: mass,mass_new,ml
  2024)   integer i,j,k
  2025) 
  2026)   do k = nlev , 1 , -1
  2027)     mass = 0
  2028)     do j = 1 , np
  2029)       do i = 1 , np
  2030)         !ml = Q(i,j,k)*dp(i,j)*spheremp(i,j)  ! see above
  2031)         ml = Q(i,j,k)
  2032)         mass = mass + ml
  2033)       enddo
  2034)     enddo
  2035) 
  2036)     ! negative mass.  so reduce all postive values to zero 
  2037)     ! then increase negative values as much as possible
  2038)     if ( mass < 0 ) Q(:,:,k) = -Q(:,:,k) 
  2039)     mass_new = 0
  2040)     do j = 1 , np
  2041)       do i = 1 , np
  2042)         if ( Q(i,j,k) < 0 ) then
  2043)           Q(i,j,k) = 0
  2044)         else
  2045)           ml = Q(i,j,k)
  2046)           mass_new = mass_new + ml
  2047)         endif
  2048)       enddo
  2049)     enddo
  2050) 
  2051)     ! now scale the all positive values to restore mass
  2052)     if ( mass_new > 0 ) Q(:,:,k) = Q(:,:,k) * abs(mass) / mass_new
  2053)     if ( mass     < 0 ) Q(:,:,k) = -Q(:,:,k) 
  2054)   enddo
  2055)   end subroutine limiter2d_zero

  2060)   subroutine advance_hypervis_scalar( edgeAdv , elem , hvcoord , hybrid , deriv , nt , nt_qdp , nets , nete , dt2 )
  2061)   !  hyperviscsoity operator for foward-in-time scheme
  2062)   !  take one timestep of:  
  2063)   !          Q(:,:,:,np) = Q(:,:,:,np) +  dt2*nu*laplacian**order ( Q )
  2064)   !
  2065)   !  For correct scaling, dt2 should be the same 'dt2' used in the leapfrog advace
  2066) #if USE_CUDA_FORTRAN
  2067)   use cuda_mod       , only : advance_hypervis_scalar_cuda
  2068) #endif
  2069)   use kinds          , only : real_kind
  2070)   use dimensions_mod , only : np, nlev
  2071)   use hybrid_mod     , only : hybrid_t
  2072)   use element_mod    , only : element_t
  2073)   use derivative_mod , only : derivative_t
  2074)   use edge_mod       , only : EdgeBuffer_t, edgevpack, edgevunpack
  2075)   use bndry_mod      , only : bndry_exchangev
  2076)   use perf_mod       , only : t_startf, t_stopf                          ! _EXTERNAL
  2077)   implicit none
  2078)   type (EdgeBuffer_t)  , intent(inout)         :: edgeAdv
  2079)   type (element_t)     , intent(inout), target :: elem(:)
  2080)   type (hvcoord_t)     , intent(in   )         :: hvcoord
  2081)   type (hybrid_t)      , intent(in   )         :: hybrid
  2082)   type (derivative_t)  , intent(in   )         :: deriv
  2083)   integer              , intent(in   )         :: nt
  2084)   integer              , intent(in   )         :: nt_qdp
  2085)   integer              , intent(in   )         :: nets
