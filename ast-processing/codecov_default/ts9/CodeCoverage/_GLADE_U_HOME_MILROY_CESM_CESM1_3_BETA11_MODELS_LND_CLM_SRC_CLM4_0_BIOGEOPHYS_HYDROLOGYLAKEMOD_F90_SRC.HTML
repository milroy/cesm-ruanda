    74)     real(r8), pointer :: qflx_evap_tot(:) !qflx_evap_soi + qflx_evap_can + qflx_tran_veg

    84)     real(r8), pointer :: qflx_floodc(:)   ! column flux of flood water from RTM

    93)     real(r8), pointer :: qflx_snow_grnd_col(:)!snow on ground after interception (mm H2O/s) [+]
    94)     real(r8), pointer :: eflx_snomelt(:)  !snow melt heat flux (W/m**2)
    95)     real(r8), pointer :: qflx_infl(:)     !infiltration (mm H2O /s)
    96)     real(r8), pointer :: qflx_snomelt(:)  !snow melt (mm H2O /s)
    97)     real(r8), pointer :: qflx_surf(:)     !surface runoff (mm H2O /s)
    98)     real(r8), pointer :: qflx_drain(:)    !sub-surface runoff (mm H2O /s)
    99)     real(r8), pointer :: qflx_irrig(:)    !irrigation flux (mm H2O /s)
   100)     real(r8), pointer :: qflx_qrgwl(:)    !qflx_surf at glaciers, wetlands, lakes
   101)     real(r8), pointer :: qflx_runoff(:)   !total runoff (qflx_drain+qflx_surf+qflx_qrgwl) (mm H2O /s)
   102)     real(r8), pointer :: qflx_snwcp_ice(:)!excess snowfall due to snow capping (mm H2O /s) [+]`
   103)     real(r8), pointer :: qflx_snwcp_ice_col(:)!excess snowfall due to snow capping (mm H2O /s) [+]`

   105)     real(r8), pointer :: qflx_evap_grnd(:)! ground surface evaporation rate (mm H2O/s) [+]
   106)     real(r8), pointer :: qflx_evap_grnd_col(:)! ground surface evaporation rate (mm H2O/s) [+]
   107)     real(r8), pointer :: qflx_sub_snow(:) ! sublimation rate from snow pack (mm H2O /s) [+]
   108)     real(r8), pointer :: qflx_sub_snow_col(:) ! sublimation rate from snow pack (mm H2O /s) [+]
   109)     real(r8), pointer :: qflx_dew_snow(:) ! surface dew added to snow pack (mm H2O /s) [+]
   110)     real(r8), pointer :: qflx_dew_snow_col(:) ! surface dew added to snow pack (mm H2O /s) [+]

   112)     real(r8), pointer :: qflx_dew_grnd_col(:) ! ground surface dew formation (mm H2O /s) [+]
   113)     real(r8) ,pointer :: soilalpha(:)     !factor that reduces ground saturated specific humidity (-)
   114)     real(r8), pointer :: zwt(:)           !water table depth
   115)     real(r8), pointer :: fcov(:)          !fractional impermeable area
   116)     real(r8), pointer :: fsat(:)          !fractional area with water table at surface
   117)     real(r8), pointer :: qcharge(:)       !aquifer recharge rate (mm/s)
   118)     real(r8), pointer :: qflx_top_soil(:)      ! net water input into soil from top (mm/s)
   119)     real(r8), pointer :: qflx_prec_grnd(:)     ! water onto ground including canopy runoff [kg/(m2 s)]
   120)     real(r8), pointer :: qflx_prec_grnd_col(:) ! water onto ground including canopy runoff [kg/(m2 s)]

   124)     real(r8), pointer :: rootr_column(:,:) !effective fraction of roots in each soil layer
   125)     real(r8), pointer :: h2osoi_vol(:,:)   !volumetric soil water (0<=h2osoi_vol<=watsat) [m3/m3]

   128)     real(r8), pointer :: qflx_snofrz_col(:)!column-integrated snow freezing rate (kg m-2 s-1) [+]
   129) !
   130) !
   131) ! !OTHER LOCAL VARIABLES:
   132) !EOP
   133)     real(r8), parameter :: snow_bd = 250._r8  !constant snow bulk density
   134)     integer  :: fp, p, c, g    ! indices
   135)     real(r8) :: dtime          ! land model time step (sec)
   136) !-----------------------------------------------------------------------
   137) 
   138)     ! Assign local pointers to derived type gridcell members
   139) 
   140)     forc_snow    => clm_a2l%forc_snow
   141)     forc_rain    => clm_a2l%forc_rain

   143)     ! Assign local pointers to derived type column members

   147)     qflx_floodc      => cwf%qflx_floodc
   148) !rtm_flood
   149)     begwb          => cwbal%begwb
   150)     endwb          => cwbal%endwb
   151)     do_capsnow     => cps%do_capsnow
   152)     snowdp         => cps%snowdp
   153)     t_grnd         => ces%t_grnd
   154)     h2osno         => cws%h2osno
   155)     snowice        => cws%snowice
   156)     snowliq        => cws%snowliq
   157)     eflx_snomelt   => cef%eflx_snomelt
   158)     qmelt          => cwf%qmelt
   159)     qflx_snomelt   => cwf%qflx_snomelt
   160)     qflx_surf      => cwf%qflx_surf
   161)     qflx_qrgwl     => cwf%qflx_qrgwl
   162)     qflx_runoff    => cwf%qflx_runoff
   163)     qflx_snwcp_ice_col => pwf_a%qflx_snwcp_ice
   164)     qflx_drain     => cwf%qflx_drain
   165)     qflx_irrig     => cwf%qflx_irrig
   166)     qflx_infl      => cwf%qflx_infl
   167)     rootr_column   => cps%rootr_column
   168)     h2osoi_vol     => cws%h2osoi_vol
   169)     h2osoi_ice     => cws%h2osoi_ice
   170)     h2osoi_liq     => cws%h2osoi_liq
   171)     qflx_evap_tot_col => pwf_a%qflx_evap_tot
   172)     soilalpha      => cws%soilalpha
   173)     zwt            => cws%zwt
   174)     fcov           => cws%fcov
   175)     fsat           => cws%fsat
   176)     qcharge        => cws%qcharge
   177)     qflx_snofrz_col => cwf%qflx_snofrz_col

   183)     qflx_sub_snow_col  => pwf_a%qflx_sub_snow
   184)     qflx_rain_grnd_col => pwf_a%qflx_rain_grnd
   185)     qflx_snow_grnd_col => pwf_a%qflx_snow_grnd

   190)     pgridcell     => pft%gridcell
   191)     qflx_evap_soi => pwf%qflx_evap_soi
   192)     qflx_evap_tot => pwf%qflx_evap_tot
   193)     qflx_evap_grnd => pwf%qflx_evap_grnd
   194)     qflx_sub_snow  => pwf%qflx_sub_snow
   195)     qflx_dew_snow  => pwf%qflx_dew_snow
   196)     qflx_dew_grnd  => pwf%qflx_dew_grnd
   197)     qflx_rain_grnd => pwf%qflx_rain_grnd
   198)     qflx_snow_grnd => pwf%qflx_snow_grnd
   199)     qflx_prec_grnd => pwf%qflx_prec_grnd
   200)     qflx_snwcp_ice => pwf%qflx_snwcp_ice
   201) 
   202)     ! Determine step size
   203) 
   204)     dtime = get_step_size()
   205) 
   206)     do fp = 1, num_lakep
   207)        p = filter_lakep(fp)
   208)        c = pcolumn(p)
   209)        g = pgridcell(p)
   210) 
   211)        ! Snow on the lake ice
   212) 
   213)        qflx_evap_grnd(p) = 0._r8
   214)        qflx_sub_snow(p) = 0._r8
   215)        qflx_dew_snow(p) = 0._r8
   216)        qflx_dew_grnd(p) = 0._r8
   217) 
   218)        if (qflx_evap_soi(p) >= 0._r8) then
   219) 
   220)           ! Sublimation: do not allow for more sublimation than there is snow
   221)           ! after melt.  Remaining surface evaporation used for infiltration.
   222) 
   223)           qflx_sub_snow(p) = min(qflx_evap_soi(p), h2osno(c)/dtime-qmelt(c))
   224)           ! Liquid water evaporation from snow or "ground" is implicitly treated as a term in qrgwl 
   225)           qflx_evap_grnd(p) = 0._r8
   226) 
   227)        else
   228) 
   229)           if (t_grnd(c) < tfrz-0.1_r8) then
   230)              qflx_dew_snow(p) = abs(qflx_evap_soi(p))
   231)           else
   232)              ! Liquid dew on snow or "ground" is implicitly treated as a term in qrgwl
   233)              qflx_dew_grnd(p) = 0._r8
   234)           end if
   235) 
   236)        end if
   237) 
   238)        ! Update snow pack
   239)        
   240)        ! WJS (8-26-11): For consistency with non-lake columns, I am setting the values of
   241)        ! qflx_rain_grnd and qflx_snow_grnd dependent on do_capsnow. For qflx_snow_grnd,
   242)        ! this makes sense: as with non-lake columns, this gives the amount of snowfall
   243)        ! that is added to the snowpack as opposed to running off due to snow capping. For
   244)        ! qflx_rain_grnd, the definition over lakes is less well defined, since (I believe)
   245)        ! all rain runs off over lakes (and qflx_snwcp_liq is always 0 over
   246)        ! lakes). Nevertheless, I am trying to be consistent with the definition of
   247)        ! qflx_rain_grnd elsewhere, which is: the amount of rainfall reaching the ground,
   248)        ! but 0 if there is snow capping.
   249) 
   250)        if (do_capsnow(c)) then
   251)           qflx_rain_grnd(p) = 0._r8
   252)           qflx_snow_grnd(p) = 0._r8
   253)           h2osno(c) = h2osno(c) - (qmelt(c) + qflx_sub_snow(p))*dtime
   254)           qflx_snwcp_ice(p) = forc_snow(g) + qflx_dew_snow(p)
   255)        else
   256)           qflx_rain_grnd(p) = forc_rain(g)
   257)           qflx_snow_grnd(p) = forc_snow(g)
   258)           h2osno(c) = h2osno(c) + (forc_snow(g)-qmelt(c)-qflx_sub_snow(p)+qflx_dew_snow(p))*dtime
   259)           qflx_snwcp_ice(p) = 0._r8
   260)        end if

   265)        if (t_grnd(c) > tfrz) h2osno(c) = 0._r8
   266) 
   267)        ! Snow depth
   268) 
   269)        snowdp(c) = h2osno(c)/snow_bd !Assume a constant snow bulk density = 250.
   270) 
   271)        ! Determine ending water balance
   272) 
   273)        endwb(c) = h2osno(c)
   274) 
   275)        ! The following are needed for global average on history tape.
   276)        ! Note that components that are not displayed over lake on history tape
   277)        ! must be set to spval here
   278) 
   279)        eflx_snomelt(c)   = qmelt(c)*hfus
   280)        qflx_infl(c)      = 0._r8
   281)        qflx_snomelt(c)   = qmelt(c)
   282)        qflx_surf(c)      = 0._r8
   283)        qflx_drain(c)     = 0._r8
   284)        qflx_irrig(c)     = 0._r8
   285)        rootr_column(c,:) = spval
   286)        snowice(c)        = spval
   287)        snowliq(c)        = spval
   288)        soilalpha(c)      = spval
   289)        zwt(c)            = spval
   290)        fcov(c)           = spval
   291)        fsat(c)           = spval
   292)        qcharge(c)        = spval
   293)        h2osoi_vol(c,:)   = spval
   294)        h2osoi_ice(c,:)   = spval
   295)        h2osoi_liq(c,:)   = spval
   296)        qflx_snofrz_col(c) = spval
   297)        qflx_qrgwl(c)     = forc_rain(g) + forc_snow(g) + qflx_floodg(g) - qflx_evap_tot(p) - qflx_snwcp_ice(p) - &
   298)                            (endwb(c)-begwb(c))/dtime
   299)        qflx_floodc(c) = qflx_floodg(g)
   300)        qflx_runoff(c)    = qflx_drain(c) + qflx_surf(c) + qflx_qrgwl(c)
   301)        qflx_top_soil(c)      = forc_rain(g) + qflx_snomelt(c)
   302)        qflx_prec_grnd(p)     = forc_rain(g) + forc_snow(g)
   303) 
   304)        ! pft averages must be done here for output to history tape and other uses
   305)        ! (note that pft2col is called before HydrologyLake, so we can't use that routine
   306)        ! to do these column -> pft averages)
   307) 
   308)        qflx_evap_tot_col(c) = qflx_evap_tot(p)
   309)        qflx_prec_grnd_col(c) = qflx_prec_grnd(p)
   310)        qflx_evap_grnd_col(c) = qflx_evap_grnd(p)
   311)        qflx_dew_grnd_col(c) = qflx_dew_grnd(p)
   312)        qflx_dew_snow_col(c) = qflx_dew_snow(p)
   313)        qflx_sub_snow_col(c) = qflx_sub_snow(p)
   314)        qflx_rain_grnd_col(c) = qflx_rain_grnd(p)
   315)        qflx_snow_grnd_col(c) = qflx_snow_grnd(p)
   316)        qflx_snwcp_ice_col(c) = qflx_snwcp_ice(p)
