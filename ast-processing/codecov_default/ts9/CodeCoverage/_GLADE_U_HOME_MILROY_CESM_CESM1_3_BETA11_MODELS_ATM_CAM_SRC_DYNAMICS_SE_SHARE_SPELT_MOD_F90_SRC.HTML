   960) !SUBROUTINE CHECK_DEPARTURECELL----------------------------------------CE-for  SPELT!
   961) ! AUTHOR: CHRISTOPH ERATH, 17.October 2011                                          !
   962) ! DESCRIPTION: Check if CFL number > 1.0D0                                          !

   965) ! INPUT:  spelt  ... spelt structur, dsphere ... departure points, klev ... level   ! 
   966) !-----------------------------------------------------------------------------------!
   967) subroutine check_departurecell(spelt,dsphere,klev)
   968)   use coordinate_systems_mod, only: cart2cubedsphere, spherical_to_cart, cartesian3D_t
   969)   implicit none
   970) 
   971)   type (spelt_struct), intent(inout)   :: spelt
   972)   type (spherical_polar_t)             :: dsphere(1:nep,1:nep)  
   973)   integer, intent(in)                  :: klev
   974)   
   975)   type (cartesian2D_t)                 :: dcart(nep,nep), acart(nep,nep)
   976)   type (cartesian3D_t)                 :: tmp
   977)   real (kind=real_kind)                :: cflx, cfly  
   978)   integer                              :: i,j
   979)             
   980)   ! calculate xy Cartesian on the cube of departure points on the corresponding face  
   981)   ! the calculation of dcart is also done in fvm_line_integrals_mod.F90, one could 
   982)   ! save the points here...
   983)   ! BUT: this subroutine should only be used for test reasons
   984)   cflx=0.0D0

   987)   spelt%maxcfl(2,klev)=cfly

   994)       ! equidistant mesh in alpha/beta coordinates
   995)         cflx=abs(dcart(i,j)%x-acart(i,j)%x)/spelt%dab((i-1)/2+1)
   996)         cfly=abs(dcart(i,j)%y-acart(i,j)%y)/spelt%dab((j-1)/2+1)
   997)         if(cflx>spelt%maxcfl(1,klev)) then
   998)           spelt%maxcfl(1,klev)=cflx

  1000)         if(cfly>spelt%maxcfl(2,klev)) then
  1001)            spelt%maxcfl(2,klev)=cfly
  1002)          endif
  1003)      end do

  1005) 
  1006)   if (spelt%maxcfl(1,klev) > 1.0D0 .OR. spelt%maxcfl(2,klev) > 1.0D0) then

  2807) !SUBROUTINE FVM_DEP_FROM_GLL----------------------------------------------CE-for FVM!
  2808) ! AUTHOR: CHRISTOPH ERATH, MARK TAYLOR 14. December 2011                            !
  2809) ! DESCRIPTION: calculates the deparute grid for spelt coming from the gll points    !

  2816) subroutine spelt_dep_from_gll(elem, deriv, asphere,dsphere,dt,tl,klev)
  2817)   use physical_constants, only : DD_PI, rearth
  2818)   use coordinate_systems_mod, only : cartesian3D_t, change_coordinates
  2819)   use derivative_mod, only : derivative_t, interpolate_gll2spelt_points
  2820) 
  2821)   implicit none
  2822)   type (element_t), intent(in)          :: elem
  2823)   type (derivative_t)  , intent(in)     :: deriv
  2824)   type (spherical_polar_t),intent(in)   :: asphere(nep,nep)
  2825)   type (spherical_polar_t),intent(out)  :: dsphere(nep,nep)
  2826)   type (timelevel_t),intent(in)         :: tl
  2827)   real (kind=real_kind),intent(in)      :: dt
  2828)   integer,intent(in)                    :: klev
  2829)   
  2830)   real (kind=real_kind)                 :: uxyz_gll(np,np,3),uxyz(nep,nep,3)
  2831)   real (kind=real_kind)                 :: un,ue,ur,clon,clat,slon,slat
  2832)   type(cartesian3D_t)                   :: acart
  2833)   integer                               :: i,j
  2834)   
  2835)   real (kind=real_kind)  :: vstar(np,np,2)
  2836)   
  2837)   ! convert velocity from lat/lon to cartesian 3D
  2838)   vstar = elem%derived%vstar(:,:,:,klev)
  2839)     
  2840)   do i=1,np
  2841)     do j=1,np
  2842)       clon = cos(elem%spherep(i,j)%lon)
  2843)       slon = sin(elem%spherep(i,j)%lon)
  2844)       clat = cos(elem%spherep(i,j)%lat)
  2845)       slat = sin(elem%spherep(i,j)%lat)
  2846) 
  2847)       ur = 0
  2848)       ue = vstar(i,j,1) 
  2849)       un = vstar(i,j,2)
  2850)       uxyz_gll(i,j,1)= clon*clat*ur - clon*slat*un - slon*ue
  2851)       uxyz_gll(i,j,2)= slon*clon*ur - slon*slat*un + clon*ue
  2852)       uxyz_gll(i,j,3)= slat          *ur + clat          *un    
  2853)     enddo
  2854)   enddo
  2855)   ! interpolate velocity to spelt nodes
  2856)   do i=1,3
  2857)     uxyz(:,:,i)=interpolate_gll2spelt_points(uxyz_gll(:,:,i),deriv)
  2858)   end do 
  2859)   ! compute departure point 
  2860)   ! crude, 1st order accurate approximation.  to be improved
  2861)   do i=1,nep
  2862)     do j=1,nep
  2863)       ! note: asphere()%r=1, so we need to convert velocity to radians/sec:
  2864)       acart = change_coordinates(asphere(i,j))  
  2865)       acart%x = acart%x - dt*uxyz(i,j,1)/rearth
  2866)       acart%y = acart%y - dt*uxyz(i,j,2)/rearth
  2867)       acart%z = acart%z - dt*uxyz(i,j,3)/rearth
  2868)       dsphere(i,j) = change_coordinates(acart)
  2869)       dsphere(i,j)%r = asphere(i,j)%r
  2870)     enddo
  2871)   enddo
  2872) end subroutine spelt_dep_from_gll

  2875) function metric_term(alphabeta) result(sg) 
  2876)   implicit none
  2877)   type (cartesian2D_t), intent(in)      :: alphabeta
  2878)   real (kind=real_kind)                 :: sg 
  2879) 
  2880)   real (kind=real_kind) :: rr, ca,cb,ta,tb, cos2ab, rho2, rho3
  2881)   real (kind=real_kind) :: ta2, tb2                
  2882) 
  2883)   !Square root of G-determinent  [sqrt(|g_ij|) = g = R^2/(rho^3*cab)]
  2884) 
  2885)   rr = 1.0D0   !*sqrt(3)        !radius^2 for unit cubed-sphere
  2886)   ca = cos(alphabeta%x)
  2887)   cb = cos(alphabeta%y)
  2888)   ta = tan(alphabeta%x)
  2889)   tb = tan(alphabeta%y)
  2890) 
  2891)   ta2 = ta*ta 
  2892)   tb2 = tb*tb 
  2893)   rho2 = (1.0D0 + ta2 + tb2)
  2894) 
  2895)   cos2ab = (ca*cb)**2
  2896)   rho3 = (rho2)**(1.50D0)
  2897)   sg =  rr / (rho3 * cos2ab)
  2898)
