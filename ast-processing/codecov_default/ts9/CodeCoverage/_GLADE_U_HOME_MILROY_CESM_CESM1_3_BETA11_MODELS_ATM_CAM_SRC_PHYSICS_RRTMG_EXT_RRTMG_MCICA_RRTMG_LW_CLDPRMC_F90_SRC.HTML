    34)       subroutine cldprmc(nlayers, inflag, iceflag, liqflag, cldfmc, &
    35)                          ciwpmc, clwpmc, reicmc, dgesmc, relqmc, ncbands, taucmc)

    37) 
    38) ! Purpose:  Compute the cloud optical depth(s) for each cloudy layer.
    39) 
    40) ! ------- Input -------
    41) 
    42)       integer, intent(in) :: nlayers                    ! total number of layers
    43)       integer, intent(in) :: inflag                     ! see definitions
    44)       integer, intent(in) :: iceflag                    ! see definitions
    45)       integer, intent(in) :: liqflag                    ! see definitions
    46) 
    47)       real(kind=r8), intent(in) :: cldfmc(:,:)          ! cloud fraction [mcica]
    48)                                                         !    Dimensions: (ngptlw,nlayers)
    49)       real(kind=r8), intent(in) :: ciwpmc(:,:)          ! cloud ice water path [mcica]
    50)                                                         !    Dimensions: (ngptlw,nlayers)
    51)       real(kind=r8), intent(in) :: clwpmc(:,:)          ! cloud liquid water path [mcica]
    52)                                                         !    Dimensions: (ngptlw,nlayers)
    53)       real(kind=r8), intent(in) :: relqmc(:)            ! liquid particle effective radius (microns)
    54)                                                         !    Dimensions: (nlayers)
    55)       real(kind=r8), intent(in) :: reicmc(:)            ! ice particle effective radius (microns)
    56)                                                         !    Dimensions: (nlayers)
    57)       real(kind=r8), intent(in) :: dgesmc(:)            ! ice particle generalized effective size (microns)
    58)                                                         !    Dimensions: (nlayers)
    59) 
    60) ! ------- Output -------
    61) 
    62)       integer, intent(out) :: ncbands                   ! number of cloud spectral bands
    63)       real(kind=r8), intent(inout) :: taucmc(:,:)       ! cloud optical depth [mcica]
    64)                                                         !    Dimensions: (ngptlw,nlayers)
    65) 
    66) ! ------- Local -------
    67) 
    68)       integer :: lay                            ! Layer index
    69)       integer :: ib                             ! spectral band index
    70)       integer :: ig                             ! g-point interval index
    71)       integer :: index 
    72) 
    73)       real(kind=r8) :: abscoice(ngptlw)         ! ice absorption coefficients
    74)       real(kind=r8) :: abscoliq(ngptlw)         ! liquid absorption coefficients
    75)       real(kind=r8) :: cwp                      ! cloud water path
    76)       real(kind=r8) :: radice                   ! cloud ice effective radius (microns)
    77)       real(kind=r8) :: dgeice                   ! cloud ice generalized effective size
    78)       real(kind=r8) :: factor                   ! 
    79)       real(kind=r8) :: fint                     ! 
    80)       real(kind=r8) :: radliq                   ! cloud liquid droplet radius (microns)
    81)       real(kind=r8), parameter :: eps = 1.e-6_r8     ! epsilon
    82)       real(kind=r8), parameter :: cldmin = 1.e-80_r8 ! minimum value for cloud quantities
    83) 
    84) ! ------- Definitions -------
    85) 
    86) !     Explanation of the method for each value of INFLAG.  Values of
    87) !     0 or 1 for INFLAG do not distingish being liquid and ice clouds.
    88) !     INFLAG = 2 does distinguish between liquid and ice clouds, and
    89) !     requires further user input to specify the method to be used to 
    90) !     compute the aborption due to each.
    91) !     INFLAG = 0:  For each cloudy layer, the cloud fraction and (gray)
    92) !                  optical depth are input.  
    93) !     INFLAG = 1:  For each cloudy layer, the cloud fraction and cloud
    94) !                  water path (g/m2) are input.  The (gray) cloud optical 
    95) !                  depth is computed as in CAM3.
    96) !     INFLAG = 2:  For each cloudy layer, the cloud fraction, cloud 
    97) !                  water path (g/m2), and cloud ice fraction are input.
    98) !       ICEFLAG = 0:  The ice effective radius (microns) is input and the
    99) !                     optical depths due to ice clouds are computed as in CAM3.
   100) !       ICEFLAG = 1:  The ice effective radius (microns) is input and the
   101) !                     optical depths due to ice clouds are computed as in 
   102) !                     Ebert and Curry, JGR, 97, 3831-3836 (1992).  The 
   103) !                     spectral regions in this work have been matched with
   104) !                     the spectral bands in RRTM to as great an extent 
   105) !                     as possible:  
   106) !                     E&C 1      IB = 5      RRTM bands 9-16
   107) !                     E&C 2      IB = 4      RRTM bands 6-8
   108) !                     E&C 3      IB = 3      RRTM bands 3-5
   109) !                     E&C 4      IB = 2      RRTM band 2
   110) !                     E&C 5      IB = 1      RRTM band 1
   111) !       ICEFLAG = 2:  The ice effective radius (microns) is input and the
   112) !                     optical properties due to ice clouds are computed from
   113) !                     the optical properties stored in the RT code,
   114) !                     STREAMER v3.0 (Reference: Key. J., Streamer 
   115) !                     User's Guide, Cooperative Institute for
   116) !                     Meteorological Satellite Studies, 2001, 96 pp.).
   117) !                     Valid range of values for re are between 5.0 and
   118) !                     131.0 micron.
   119) !       ICEFLAG = 3: The ice generalized effective size (dge) is input
   120) !                    and the optical properties, are calculated as in
   121) !                    Q. Fu, J. Climate, (1998). Q. Fu provided high resolution
   122) !                    tables which were appropriately averaged for the
   123) !                    bands in RRTM_LW.  Linear interpolation is used to
   124) !                    get the coefficients from the stored tables.
   125) !                    Valid range of values for dge are between 5.0 and
   126) !                    140.0 micron.
   127) !       LIQFLAG = 0:  The optical depths due to water clouds are computed as
   128) !                     in CAM3.
   129) !       LIQFLAG = 1:  The water droplet effective radius (microns) is input 
   130) !                     and the optical depths due to water clouds are computed 
   131) !                     as in Hu and Stamnes, J., Clim., 6, 728-742, (1993).
   132) !                     The values for absorption coefficients appropriate for
   133) !                     the spectral bands in RRTM have been obtained for a 
   134) !                     range of effective radii by an averaging procedure 
   135) !                     based on the work of J. Pinto (private communication).
   136) !                     Linear interpolation is used to get the absorption 
   137) !                     coefficients for the input effective radius.
   138) 
   139)       hvrclc = '$Revision: 1.5 $'
   140) 
   141)       ncbands = 1
   142) 
   143) ! This initialization is done in rrtmg_lw_subcol.F90.
   144) !      do lay = 1, nlayers
   145) !         do ig = 1, ngptlw
   146) !            taucmc(ig,lay) = 0.0_r8
   147) !         enddo
   148) !      enddo
   149) 
   150) ! Main layer loop
   151)       do lay = 1, nlayers
   152) 
   153)         do ig = 1, ngptlw
   154)           cwp = ciwpmc(ig,lay) + clwpmc(ig,lay)
   155)           if (cldfmc(ig,lay) .ge. cldmin .and. &
   156)              (cwp .ge. cldmin .or. taucmc(ig,lay) .ge. cldmin)) then
   157) 
   158) ! Ice clouds and water clouds combined.
   159)             if (inflag .eq. 0) then
   160) ! Cloud optical depth already defined in taucmc, return to main program
   161)                return
   162) 
   163)             elseif(inflag .eq. 1) then 
   164)                 stop 'INFLAG = 1 OPTION NOT AVAILABLE WITH MCICA'
   165) !               cwp = ciwpmc(ig,lay) + clwpmc(ig,lay)
   166) !               taucmc(ig,lay) = abscld1 * cwp
   167) 
   168) ! Separate treatement of ice clouds and water clouds.
   169)             elseif(inflag .eq. 2) then
   170)                radice = reicmc(lay)
   171) 
   172) ! Calculation of absorption coefficients due to ice clouds.
   173)                if (ciwpmc(ig,lay) .eq. 0.0_r8) then
   174)                   abscoice(ig) = 0.0_r8
   175) 
   176)                elseif (iceflag .eq. 0) then
   177)                   if (radice .lt. 10.0_r8) stop 'ICE RADIUS TOO SMALL'
   178)                   abscoice(ig) = absice0(1) + absice0(2)/radice
   179) 
   180)                elseif (iceflag .eq. 1) then
   181) ! mji - turn off limits to mimic CAM3
   182) !                  if (radice .lt. 13.0_r8 .or. radice .gt. 130._r8) stop &
   183) !                      'ICE RADIUS OUT OF BOUNDS'
   184)                   ncbands = 5
   185)                   ib = ngb(ig)
   186)                   abscoice(ig) = absice1(1,ib) + absice1(2,ib)/radice
   187) 
   188) ! For iceflag=2 option, combine with iceflag=0 option to handle out of bounds 
   189) ! particle sizes.
   190) ! Use iceflag=2 option for ice particle effective radii from 5.0 and 131.0 microns
   191) ! and use iceflag=0 option for ice particles greater than 131.0 microns.
   192) ! *** NOTE: Transition between two methods has not been smoothed. 
   193) 
   194)                elseif (iceflag .eq. 2) then
   195)                   if (radice .lt. 5.0_r8) stop 'ICE RADIUS OUT OF BOUNDS'
   196)                   if (radice .ge. 5.0_r8 .and. radice .le. 131._r8) then
   197)                      ncbands = 16
   198)                      factor = (radice - 2._r8)/3._r8
   199)                      index = int(factor)
   200)                      if (index .eq. 43) index = 42
   201)                      fint = factor - float(index)
   202)                      ib = ngb(ig)
   203)                      abscoice(ig) = &
   204)                          absice2(index,ib) + fint * &
   205)                          (absice2(index+1,ib) - (absice2(index,ib))) 
   206)                   elseif (radice .gt. 131._r8) then
   207)                      abscoice(ig) = absice0(1) + absice0(2)/radice
   208)                   endif
   209)                
   210) ! For iceflag=3 option, combine with iceflag=0 option to handle large particle sizes.
   211) ! Use iceflag=3 option for ice particle effective radii from 3.2 and 91.0 microns
   212) ! (generalized effective size, dge, from 5 to 140 microns), and use iceflag=0 option
   213) ! for ice particle effective radii greater than 91.0 microns (dge = 140 microns).
   214) ! *** NOTE: Fu parameterization requires particle size in generalized effective size.
   215) ! *** NOTE: Transition between two methods has not been smoothed. 
   216) 
   217)                elseif (iceflag .eq. 3) then
   218)                   dgeice = dgesmc(lay)
   219)                   if (dgeice .lt. 5.0_r8) stop 'ICE GENERALIZED EFFECTIVE SIZE OUT OF BOUNDS'
   220)                   if (dgeice .ge. 5.0_r8 .and. dgeice .le. 140._r8) then
   221)                      ncbands = 16
   222)                      factor = (dgeice - 2._r8)/3._r8
   223)                      index = int(factor)
   224)                      if (index .eq. 46) index = 45
   225)                      fint = factor - float(index)
   226)                      ib = ngb(ig)
   227)                      abscoice(ig) = &
   228)                          absice3(index,ib) + fint * &
   229)                          (absice3(index+1,ib) - (absice3(index,ib)))
   230)                   elseif (dgeice .gt. 140._r8) then
   231)                      abscoice(ig) = absice0(1) + absice0(2)/radice
   232)                   endif
   233)    
   234)                endif
   235)                   
   236) ! Calculation of absorption coefficients due to water clouds.
   237)                if (clwpmc(ig,lay) .eq. 0.0_r8) then
   238)                   abscoliq(ig) = 0.0_r8
   239) 
   240)                elseif (liqflag .eq. 0) then
   241)                    abscoliq(ig) = absliq0
   242) 
   243)                elseif (liqflag .eq. 1) then
   244)                   radliq = relqmc(lay)
   245)                   if (radliq .lt. 1.5_r8 .or. radliq .gt. 60._r8) stop &
   246)                        'LIQUID EFFECTIVE RADIUS OUT OF BOUNDS'
   247)                   index = radliq - 1.5_r8
   248)                   if (index .eq. 58) index = 57
   249)                   if (index .eq. 0) index = 1
   250)                   fint = radliq - 1.5_r8 - index
   251)                   ib = ngb(ig)
   252)                   abscoliq(ig) = &
   253)                         absliq1(index,ib) + fint * &
   254)                         (absliq1(index+1,ib) - (absliq1(index,ib)))
   255)                endif
   256) 
   257)                taucmc(ig,lay) = ciwpmc(ig,lay) * abscoice(ig) + &
   258)                                 clwpmc(ig,lay) * abscoliq(ig)
   259) 
   260)             endif
   261)          endif
   262)          enddo
   263)       enddo
   264) 
   265)       end subroutine cldprmc
