   198)   subroutine check_energy_timestep_init(state, tend, pbuf, col_type)
   199)     use physics_buffer, only : physics_buffer_desc, pbuf_set_field
   200) !-----------------------------------------------------------------------
   201) ! Compute initial values of energy and water integrals, 
   202) ! zero cumulative tendencies
   203) !-----------------------------------------------------------------------
   204) !------------------------------Arguments--------------------------------
   205) 
   206)     type(physics_state),   intent(inout)    :: state
   207)     type(physics_tend ),   intent(inout)    :: tend
   208)     type(physics_buffer_desc), pointer      :: pbuf(:)
   209)     integer, optional                       :: col_type  ! Flag inidicating whether using grid or subcolumns
   210) !---------------------------Local storage-------------------------------
   211) 
   212)     real(r8) :: ke(state%ncol)                     ! vertical integral of kinetic energy
   213)     real(r8) :: se(state%ncol)                     ! vertical integral of static energy
   214)     real(r8) :: wv(state%ncol)                     ! vertical integral of water (vapor)
   215)     real(r8) :: wl(state%ncol)                     ! vertical integral of water (liquid)
   216)     real(r8) :: wi(state%ncol)                     ! vertical integral of water (ice)
   217) 
   218)     integer ncol                                   ! number of atmospheric columns
   219)     integer  i,k                                   ! column, level indices
   220)     integer :: ixcldice, ixcldliq                  ! CLDICE and CLDLIQ indices
   221) !-----------------------------------------------------------------------
   222) 
   223)     ncol  = state%ncol
   224)     call cnst_get_ind('CLDICE', ixcldice, abort=.false.)
   225)     call cnst_get_ind('CLDLIQ', ixcldliq, abort=.false.)
   226) 
   227) ! Compute vertical integrals of dry static energy and water (vapor, liquid, ice)
   228)     ke = 0._r8
   229)     se = 0._r8
   230)     wv = 0._r8
   231)     wl = 0._r8
   232)     wi = 0._r8
   233)     do k = 1, pver
   234)        do i = 1, ncol
   235)           ke(i) = ke(i) + 0.5_r8*(state%u(i,k)**2 + state%v(i,k)**2)*state%pdel(i,k)/gravit
   236)           se(i) = se(i) + state%s(i,k         )*state%pdel(i,k)/gravit
   237)           wv(i) = wv(i) + state%q(i,k,1       )*state%pdel(i,k)/gravit
   238)        end do
   239)     end do
   240)     ! Don't require cloud liq/ice to be present.  Allows for adiabatic/ideal phys.
   241)     if (ixcldliq > 1  .and.  ixcldice > 1) then
   242)        do k = 1, pver
   243)           do i = 1, ncol
   244)              wl(i) = wl(i) + state%q(i,k,ixcldliq)*state%pdel(i,k)/gravit
   245)              wi(i) = wi(i) + state%q(i,k,ixcldice)*state%pdel(i,k)/gravit
   246)           end do
   247)        end do
   248)     end if
   249) 
   250) ! Compute vertical integrals of frozen static energy and total water.
   251)     do i = 1, ncol
   252)        state%te_ini(i) = se(i) + ke(i) + (latvap+latice)*wv(i) + latice*wl(i)
   253)        state%tw_ini(i) = wv(i) + wl(i) + wi(i)
   254) 
   255)        state%te_cur(i) = state%te_ini(i)
   256)        state%tw_cur(i) = state%tw_ini(i)
   257)     end do
   258) 
   259) ! zero cummulative boundary fluxes 
   260)     tend%te_tnd(:ncol) = 0._r8
   261)     tend%tw_tnd(:ncol) = 0._r8
   262) 
   263)     state%count = 0
   264) 
   265) ! initialize physics buffer
   266)     if (is_first_step()) then
   267)        call pbuf_set_field(pbuf, teout_idx, state%te_ini, col_type=col_type)
   268)     end if
   269) 
   270)   end subroutine check_energy_timestep_init

   274)   subroutine check_energy_chng(state, tend, name, nstep, ztodt,        &
   275)        flx_vap, flx_cnd, flx_ice, flx_sen)
   276) 
   277) !-----------------------------------------------------------------------
   278) ! Check that the energy and water change matches the boundary fluxes
   279) !-----------------------------------------------------------------------
   280) !------------------------------Arguments--------------------------------
   281) 
   282)     type(physics_state)    , intent(inout) :: state
   283)     type(physics_tend )    , intent(inout) :: tend
   284)     character*(*),intent(in) :: name               ! parameterization name for fluxes
   285)     integer , intent(in   ) :: nstep               ! current timestep number
   286)     real(r8), intent(in   ) :: ztodt               ! 2 delta t (model time increment)
   287)     real(r8), intent(in   ) :: flx_vap(:)          ! (pcols) - boundary flux of vapor         (kg/m2/s)
   288)     real(r8), intent(in   ) :: flx_cnd(:)          ! (pcols) -boundary flux of liquid+ice    (m/s) (precip?)
   289)     real(r8), intent(in   ) :: flx_ice(:)          ! (pcols) -boundary flux of ice           (m/s) (snow?)
   290)     real(r8), intent(in   ) :: flx_sen(:)          ! (pcols) -boundary flux of sensible heat (w/m2)
   291) 
   292) !******************** BAB ******************************************************
   293) !******* Note that the precip and ice fluxes are in precip units (m/s). ********
   294) !******* I would prefer to have kg/m2/s.                                ********
   295) !******* I would also prefer liquid (not total) and ice fluxes          ********
   296) !*******************************************************************************
   297) 
   298) !---------------------------Local storage-------------------------------
   299) 
   300)     real(r8) :: te_xpd(state%ncol)                 ! expected value (f0 + dt*boundary_flux)
   301)     real(r8) :: te_dif(state%ncol)                 ! energy of input state - original energy
   302)     real(r8) :: te_tnd(state%ncol)                 ! tendency from last process
   303)     real(r8) :: te_rer(state%ncol)                 ! relative error in energy column
   304) 
   305)     real(r8) :: tw_xpd(state%ncol)                 ! expected value (w0 + dt*boundary_flux)
   306)     real(r8) :: tw_dif(state%ncol)                 ! tw_inp - original water
   307)     real(r8) :: tw_tnd(state%ncol)                 ! tendency from last process
   308)     real(r8) :: tw_rer(state%ncol)                 ! relative error in water column
   309) 
   310)     real(r8) :: ke(state%ncol)                     ! vertical integral of kinetic energy
   311)     real(r8) :: se(state%ncol)                     ! vertical integral of static energy
   312)     real(r8) :: wv(state%ncol)                     ! vertical integral of water (vapor)
   313)     real(r8) :: wl(state%ncol)                     ! vertical integral of water (liquid)
   314)     real(r8) :: wi(state%ncol)                     ! vertical integral of water (ice)
   315) 
   316)     real(r8) :: te(state%ncol)                     ! vertical integral of total energy
   317)     real(r8) :: tw(state%ncol)                     ! vertical integral of total water
   318) 
   319)     integer lchnk                                  ! chunk identifier
   320)     integer ncol                                   ! number of atmospheric columns
   321)     integer  i,k                                   ! column, level indices
   322)     integer :: ixcldice, ixcldliq                  ! CLDICE and CLDLIQ indices
   323) !-----------------------------------------------------------------------
   324) 
   325)     lchnk = state%lchnk
   326)     ncol  = state%ncol
   327)     call cnst_get_ind('CLDICE', ixcldice, abort=.false.)
   328)     call cnst_get_ind('CLDLIQ', ixcldliq, abort=.false.)
   329) 
   330)     ! Compute vertical integrals of dry static energy and water (vapor, liquid, ice)
   331)     ke = 0._r8
   332)     se = 0._r8
   333)     wv = 0._r8
   334)     wl = 0._r8
   335)     wi = 0._r8
   336)     do k = 1, pver
   337)        do i = 1, ncol
   338)           ke(i) = ke(i) + 0.5_r8*(state%u(i,k)**2 + state%v(i,k)**2)*state%pdel(i,k)/gravit
   339)           se(i) = se(i) + state%s(i,k         )*state%pdel(i,k)/gravit
   340)           wv(i) = wv(i) + state%q(i,k,1       )*state%pdel(i,k)/gravit
   341)        end do
   342)     end do
   343)     ! Don't require cloud liq/ice to be present.  Allows for adiabatic/ideal phys.
   344)     if (ixcldliq > 1  .and.  ixcldice > 1) then
   345)        do k = 1, pver
   346)           do i = 1, ncol
   347)              wl(i) = wl(i) + state%q(i,k,ixcldliq)*state%pdel(i,k)/gravit
   348)              wi(i) = wi(i) + state%q(i,k,ixcldice)*state%pdel(i,k)/gravit
   349)           end do
   350)        end do
   351)     end if
   352) 
   353)     ! Compute vertical integrals of frozen static energy and total water.
   354)     do i = 1, ncol
   355)        te(i) = se(i) + ke(i) + (latvap+latice)*wv(i) + latice*wl(i)
   356)        tw(i) = wv(i) + wl(i) + wi(i)
   357)     end do
   358) 
   359)     ! compute expected values and tendencies
   360)     do i = 1, ncol
   361)        ! change in static energy and total water
   362)        te_dif(i) = te(i) - state%te_cur(i)
   363)        tw_dif(i) = tw(i) - state%tw_cur(i)
   364) 
   365)        ! expected tendencies from boundary fluxes for last process
   366)        te_tnd(i) = flx_vap(i)*(latvap+latice) - (flx_cnd(i) - flx_ice(i))*1000._r8*latice + flx_sen(i)
   367)        tw_tnd(i) = flx_vap(i) - flx_cnd(i) *1000._r8
   368) 
   369)        ! cummulative tendencies from boundary fluxes
   370)        tend%te_tnd(i) = tend%te_tnd(i) + te_tnd(i)
   371)        tend%tw_tnd(i) = tend%tw_tnd(i) + tw_tnd(i)
   372) 
   373)        ! expected new values from previous state plus boundary fluxes
   374)        te_xpd(i) = state%te_cur(i) + te_tnd(i)*ztodt
   375)        tw_xpd(i) = state%tw_cur(i) + tw_tnd(i)*ztodt
   376) 
   377)        ! relative error, expected value - input state / previous state 
   378)        te_rer(i) = (te_xpd(i) - te(i)) / state%te_cur(i)
   379)     end do
   380) 
   381)     ! relative error for total water (allow for dry atmosphere)
   382)     tw_rer = 0._r8
   383)     where (state%tw_cur(:ncol) > 0._r8) 
   384)        tw_rer(:ncol) = (tw_xpd(:ncol) - tw(:ncol)) / state%tw_cur(:ncol)
   385)     end where
   386) 
   387)     ! error checking
   388)     if (print_energy_errors) then
   389)        if (any(abs(te_rer(1:ncol)) > 1.E-14_r8 .or. abs(tw_rer(1:ncol)) > 1.E-10_r8)) then
   390)           do i = 1, ncol
   391)              ! the relative error threshold for the water budget has been reduced to 1.e-10
   392)              ! to avoid messages generated by QNEG3 calls
   393)              ! PJR- change to identify if error in energy or water 
   394)              if (abs(te_rer(i)) > 1.E-14_r8 ) then 
   395)                 state%count = state%count + 1
   396)                 write(iulog,*) "significant energy conservation error after ", name,        &
   397)                       " count", state%count, " nstep", nstep, "chunk", lchnk, "col", i
   398)                 write(iulog,*) te(i),te_xpd(i),te_dif(i),tend%te_tnd(i)*ztodt,  &
   399)                       te_tnd(i)*ztodt,te_rer(i)
   400)              endif
   401)              if ( abs(tw_rer(i)) > 1.E-10_r8) then
   402)                 state%count = state%count + 1
   403)                 write(iulog,*) "significant water conservation error after ", name,        &
   404)                       " count", state%count, " nstep", nstep, "chunk", lchnk, "col", i
   405)                 write(iulog,*) tw(i),tw_xpd(i),tw_dif(i),tend%tw_tnd(i)*ztodt,  &
   406)                       tw_tnd(i)*ztodt,tw_rer(i)
   407)              end if
   408)           end do
   409)        end if
   410)     end if
   411) 
   412)     ! copy new value to state
   413)     do i = 1, ncol
   414)        state%te_cur(i) = te(i)
   415)        state%tw_cur(i) = tw(i)
   416)     end do
   417) 
   418)   end subroutine check_energy_chng

   422)   subroutine check_energy_gmean(state, pbuf2d, dtime, nstep)
   423) 
   424)     use physics_buffer, only : physics_buffer_desc, pbuf_get_field, pbuf_get_chunk
   425)     
   426) !-----------------------------------------------------------------------
   427) ! Compute global mean total energy of physics input and output states
   428) !-----------------------------------------------------------------------
   429) !------------------------------Arguments--------------------------------
   430) 
   431)     type(physics_state), intent(in   ), dimension(begchunk:endchunk) :: state
   432)     type(physics_buffer_desc),    pointer    :: pbuf2d(:,:)
   433) 
   434)     real(r8), intent(in) :: dtime        ! physics time step
   435)     integer , intent(in) :: nstep        ! current timestep number
   436) 
   437) !---------------------------Local storage-------------------------------
   438)     integer :: ncol                      ! number of active columns
   439)     integer :: lchnk                     ! chunk index
   440) 
   441)     real(r8) :: te(pcols,begchunk:endchunk,3)   
   442)                                          ! total energy of input/output states (copy)
   443)     real(r8) :: te_glob(3)               ! global means of total energy
   444)     real(r8), pointer :: teout(:)
   445) !-----------------------------------------------------------------------
   446) 
   447)     ! Copy total energy out of input and output states
   448) !DIR$ CONCURRENT
   449)     do lchnk = begchunk, endchunk
   450)        ncol = state(lchnk)%ncol
   451)        ! input energy
   452)        te(:ncol,lchnk,1) = state(lchnk)%te_ini(:ncol)
   453)        ! output energy
   454)        call pbuf_get_field(pbuf_get_chunk(pbuf2d,lchnk),teout_idx, teout)
   455) 
   456)        te(:ncol,lchnk,2) = teout(1:ncol)
   457)        ! surface pressure for heating rate
   458)        te(:ncol,lchnk,3) = state(lchnk)%pint(:ncol,pver+1)
   459)     end do
   460) 
   461)     ! Compute global means of input and output energies and of
   462)     ! surface pressure for heating rate (assume uniform ptop)
   463)     call gmean(te, te_glob, 3)
   464) 
   465)     if (begchunk .le. endchunk) then
   466)        teinp_glob = te_glob(1)
   467)        teout_glob = te_glob(2)
   468)        psurf_glob = te_glob(3)
   469)        ptopb_glob = state(begchunk)%pint(1,1)
   470) 
   471)        ! Global mean total energy difference
   472)        tedif_glob =  teinp_glob - teout_glob
   473)        heat_glob  = -tedif_glob/dtime * gravit / (psurf_glob - ptopb_glob)
   474) 
   475)        if (masterproc) then
   476)           write(iulog,'(1x,a9,1x,i8,4(1x,e25.17))') "nstep, te", nstep, teinp_glob, teout_glob, heat_glob, psurf_glob
   477)        end if
   478)     else
   479)        heat_glob = 0._r8
   480)     end if  !  (begchunk .le. endchunk)
   481)     
   482)   end subroutine check_energy_gmean

   485)   subroutine check_energy_fix(state, ptend, nstep, eshflx)
   486) 
   487) !-----------------------------------------------------------------------
   488) ! Add heating rate required for global mean total energy conservation
   489) !-----------------------------------------------------------------------
   490) !------------------------------Arguments--------------------------------
   491) 
   492)     type(physics_state), intent(in   ) :: state
   493)     type(physics_ptend), intent(out)   :: ptend
   494) 
   495)     integer , intent(in   ) :: nstep          ! time step number
