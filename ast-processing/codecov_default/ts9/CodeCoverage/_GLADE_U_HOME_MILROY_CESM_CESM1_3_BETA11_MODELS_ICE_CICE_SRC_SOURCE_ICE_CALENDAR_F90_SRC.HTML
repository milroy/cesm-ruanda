   122)       subroutine init_calendar
   123) !
   124) ! !DESCRIPTION:
   125) !
   126) ! Initialize calendar variables
   127) !
   128) ! !REVISION HISTORY:
   129) !
   130) ! authors: Elizabeth C. Hunke, LANL
   131) !          Tony Craig, NCAR
   132) !
   133) ! !USES:

   136) !

   140)          k                          , &
   141) 
   142)       istep = 0         ! local timestep number
   143)       time=istep0*dt    ! s
   144) #ifdef CCSMCOUPLED
   145)       time_forc = c0    ! for coupled runs
   146) #endif
   147)       yday=c0           ! absolute day number
   148)       mday=0            ! day of the month
   149)       month=0           ! month
   150)       nyr=0             ! year
   151)       idate=00000101    ! date

   156)       dt_thm = dt       ! convenience copy of thermodynamic timestep
   157)       dt_dyn = dt/xndt_dyn ! dynamics et al timestep

   159)       dayyr = real(days_per_year, kind=dbl_kind)  ! days_per_year set by ice_init
   160) 
   161) 
   162)       ! determine initial date (assumes namelist year_init, istep0 unchanged)     
   163)       sec = mod(time,secday)            ! elapsed seconds into date at
   164)                                         ! end of dt
   165)       tday = (time-sec)/secday + c1     ! absolute day number
   166)

   168)          nyr = int((tday-c1)/dayyr) + 1    ! year number
   169)       endif
   170) 
   171)       ! reset the number of leap days: this is necessary to add one one
   172)       ! the year turns from a leap-year to a non-leap year
   173)       nleaps = leap_year_count(nyr+year_init-1)

   198)       subroutine calendar(ttime)
   199) !
   200) ! !DESCRIPTION:
   201) !
   202) ! Determine the date at the end of the time step
   203) !
   204) ! !REVISION HISTORY:
   205) !
   206) ! authors: Elizabeth C. Hunke, LANL
   207) !          Tony Craig, NCAR

   221)          k, ns                      , &
   222)          nyrp,mdayp,hourp           , & ! previous year, day, hour

   229)       monthp=month
   230)       mdayp=mday
   231)       hourp=hour
   232)       if (calendar_type /= "GREGORIAN") new_year=.false.
   233)       new_month=.false.
   234)       new_day=.false.
   235)       new_hour=.false.
   236)       write_history(:)=.false.
   237)       write_restart=0
   238) 
   239)       sec = mod(ttime,secday)           ! elapsed seconds into date at
   240)                                         ! end of dt
   241)       tday = (ttime-sec)/secday + c1    ! absolute day number
   242)       
   243)       if (calendar_type /= "GREGORIAN") then 	
   244)          nyr = int((tday-c1)/dayyr) + 1    ! year number
   245)          if (nyr   /= nyrp)   new_year = .true.
   246)       endif
   247) 
   248)       ! reset the number of leap days: this is necessary to add one one
   249)       ! the year turns from a leap-year to a non-leap year
   250)       nleaps = leap_year_count(nyr+year_init-1)
   251) 
   252)       ! get the daycal variable, depending on calendar and days_per_year
   253)       call get_daycal(year=nyr+year_init-1,days_per_year_in=days_per_year,&
   254)            daycal_out=daycal) 
   255) 
   256)       ! subtract the number of days in prior years from the number of days to get the number of days this year
   257)       yday = tday-real(nleaps,kind=dbl_kind)-real(nyr-1,kind=dbl_kind)*dayyr    ! days that have passed this year
   258) 
   259)       do k = 1, 12
   260)         if (yday > real(daycal(k),kind=dbl_kind)) month = k
   261)       enddo
   262)       mday = int(yday) - daycal(month)  ! day of the month 
   263) 
   264)       hour = int((ttime-dt)/c3600) + c1 ! hour  
   265) 
   266)       elapsed_months = (nyr - 1)*12 + month - 1
   267)       elapsed_days = int(tday) - 1 
   268)       elapsed_hours = int(ttime/3600)
   269) 
   270)       idate = (nyr+year_init-1)*10000 + month*100 + mday ! date (yyyymmdd) 
   271) 
   272) #ifndef CCSMCOUPLED
   273)       if (istep >= npt+1)  stop_now = 1
   274) #endif
   275) 
   276)       if (month /= monthp) new_month = .true.
   277)       if (mday  /= mdayp)  new_day = .true.
   278)       if (hour  /= hourp)  new_hour = .true.
   279) 
   280) 
   281)       do ns = 1, nstreams
   282)          if (histfreq(ns)=='1' .and. histfreq_n(ns)/=0) then
   283)              if (mod(istep1, histfreq_n(ns))==0) &
   284)                 write_history(ns)=.true.
   285)          endif
   286)       enddo
   287) 
   288)       if (istep > 1) then
   289) 
   290)         do ns = 1, nstreams
   291) 
   292)            select case (histfreq(ns))
   293)            case ("y", "Y")
   294)              if (new_year  .and. histfreq_n(ns)/=0) then
   295)                 if (mod(nyr, histfreq_n(ns))==0) &
   296)                    write_history(ns) = .true.
   297)              endif
   298)            case ("m", "M")
   299)              if (new_month .and. histfreq_n(ns)/=0) then
   300)                 if (mod(elapsed_months,histfreq_n(ns))==0) &
   301)                    write_history(ns) = .true.
   302)              endif
   303)            case ("d", "D")
   304)              if (new_day  .and. histfreq_n(ns)/=0) then
   305)                 if (mod(elapsed_days,histfreq_n(ns))==0) &
   306)                    write_history(ns) = .true.
   307)              endif
   308)            case ("h", "H")
   309)              if (new_hour  .and. histfreq_n(ns)/=0) then
   310)                 if (mod(elapsed_hours,histfreq_n(ns))==0) &
   311)                    write_history(ns) = .true.
   312)              endif
   313)            end select
   314) 
   315)         enddo ! nstreams
   316) 
   317)         select case (dumpfreq)
   318)         case ("y", "Y")
   319)           if (new_year  .and. mod(nyr, dumpfreq_n)==0) &
   320)                 write_restart = 1
   321)         case ("m", "M")
   322)           if (new_month .and. mod(elapsed_months,dumpfreq_n)==0) &
   323)                 write_restart=1
   324)         case ("d", "D")
   325)           if (new_day   .and. mod(elapsed_days, dumpfreq_n)==0) &
   326)                 write_restart = 1
   327)         case default
   328)           call abort_ice('ice_calendar: Invalid dumpfreq: '//dumpfreq)
   329)         end select
   330)       endif
   331) 
   332)       if (my_task == master_task .and. mod(istep,diagfreq) == 0 &
   333)                                  .and. stop_now /= 1) then
   334)         write(nu_diag,*) ' '
   335)         write(nu_diag,'(a7,i10,4x,a6,i10,4x,a4,i10)') &
   336)              'istep1:', istep1, 'idate:', idate, 'sec:', sec
   337)       endif
   338) 
   339)       end subroutine calendar

   432)       integer function leap_year_count(Y)
   433)         ! counts the number of leap years since year 1
   434) 
   435)         ! Input/output paramters
   436)         integer, intent(in) :: Y
   437) 
   438) 
   439)         if (calendar_type == "GREGORIAN") then 	
   440)            ! count the number of leap years before Y
   441)            if ( Y .lt. 0 ) then
   442)               leap_year_count = 0
   443)               write(6,*) 'WARNING: leap_year_count for year ',Y,&
   444)                    'assumes no leap years before year 0'
   445)            else
   446)               leap_year_count  = ( (Y-1)/4 - (Y-1)/100 + (Y-1)/400 ) + 1
   447)            endif
   448)         else
   449)            leap_year_count = 0
   450)         endif
   451) 
   452)         ! set module variable
   453)         nleaps = leap_year_count
   454) 
   455)         return
