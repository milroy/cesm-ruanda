     6)       SUBROUTINE APXMKA (MSGUN, EPOCH, GPLAT,GPLON,GPALT,NLAT,NLON, &
     7)                          NALT,WK,LWK, IST)
     8) !
     9) !-----------------------------------------------------------------------
    10) !          This contains ENTRYs that determine quantities related to Apex
    11) !          coordinates.  They are designed to speed up coordinate determination
    12) !          by substituting interpolation from previously calculated tables
    13) !          for direct calculation.  They also permit calculating quantities
    14) !	   involving gradients of Apex coordinates, such as base vectors in
    15) !	   the Modified-Apex and Quasi-Dipole systems.  Options allow table 
    16) !	   creation, storage, and read-back in addition to interpolation.
    17) !
    18) !          Tables (arrays) must be prepared prior to coordinate determination.
    19) !          Tables may be created for one date and held only in memory by calling
    20) !
    21) !            APXMKA - make magnetic arrays.
    22) !
    23) !          Or, they may be created and written, then read back later by
    24) !
    25) !            APXWRA - make and write arrays
    26) !            APXRDA - read stored arrays.
    27) !
    28) !          Tables may be created for multiple times (e.g., the DGRF dates)
    29) !          and written to a file using APXWRA.  However, tables are held in
    30) !          memory for only one time due to the potential to exceed memory
    31) !          capacity.  Thus, when reading back multiples times, APXRDA
    32) !          interpolates/extrapolates to a single time.
    33) !
    34) !          Alternatively, If APXWRA is called to make and write tables for one
    35) !          time only, then it is not necessary to call APXRDA because tables
    36) !          have already been initialized for that time.
    37) !
    38) !          Creation of global lookup tables can be time consuming, thus, it
    39) !          is expected that this is done rarely; it might be considered an
    40) !          installation step.  Once the tables are established, programs may
    41) !          reference them by calling APXRDA.  In this case, the grid coordinates
    42) !          (originally specified during the table creation) may be ascertained
    43) !          by calling
    44) !
    45) !            APXGGC - get grid coordinates.
    46) !
    47) !          After the tables have been loaded in memory for a given date, it
    48) !          is possible to interpolate in space to determinine various magnetic
    49) !          parameters using the following calls:
    50) !
    51) !            APXALL  - get Apex coordinates (radius, lat, lon).
    52) !            APXMALL - get Modified Apex coordinates, Quasi-Dipole
    53) !                      coordinates and associated base vectors.
    54) !            APXQ2G  - convert quasi-dipole to geodetic coordinates.
    55) !
    56) !          Details for each ENTRY introduced above are provided below.
    57) !          Following the ENTRY summaries are comments describing EXTERNALS,
    58) !          INSTALLATION SPECIFICS for different computers, and a HISTORY.
    59) !
    60) !          REFERENCE:
    61) !          Richmond, A. D., Ionospheric Electrodynamics Using Magnetic Apex
    62) !          Coordinates, J. Geomag. Geoelectr., 47, 191-212, 1995.
    63) !
    64) !          ALGORITHM:
    65) !          When arrays are created, APXMKA calls subroutine MAKEXYZV, which
    66) !          in turn calls subroutine APEX at each grid point to get the apex
    67) !          radius A, the apex longitude PHIA, and the magnetic potential
    68) !          VMP.  The cosine (CLP) and sine (SLP) of the quasi-dipole
    69) !          latitude are computed from A.  From these can be computed
    70) !          preliminary quantites defined as
    71) !
    72) !            x = cos(quasi-dipole latitude)*cos(apex longitude)
    73) !            y = cos(quasi-dipole latitude)*sin(apex longitude)
    74) !            z = sin(quasi-dipole latitude)
    75) !            v = (VMP/VP)*((RE+ALT)/RE)**2
    76) !
    77) !          where VP is the magnitude of the magnetic potential of the
    78) !          geomagnetic dipole at a radius of RE; ALT is altitude; and RE is
    79) !          the mean Earth radius.  Note that all of these quantities vary
    80) !          smoothly across the apex poles, unlike apex latitude or
    81) !          quasi-dipole latitude, so that they can be linearly interpolated
    82) !          near these poles.  Corresponding values of x,y,z,v for a dipole
    83) !          field on a spherical Earth are computed analytically and
    84) !          subtracted from the above quantities, and the results are put
    85) !          into the 3D arrays X,Y,Z,V.  When APXALL or APXMALL is called,
    86) !          trilinear interpolations (in latitude, longitude, and inverse
    87) !          altitude) are carried out between the grid points.  Gradients are
    88) !          calculated for APXMALL in order to determine the base vectors.
    89) !          Analytic formulas appropriate for a dipole field on a spherical
    90) !          Earth are used to determine the previously removed dipole
    91) !          components of x,y,z,v, and their gradients and these are added
    92) !          back to the interpolated values obtained for the non-dipole
    93) !          components.  Finally, the apex-based coordinates and their base
    94) !          vectors are calculated from x,y,z,v and their gradients.
    95) !
    96) !------------------------------------------------------------------------------
    97) !
    98) !       ENTRY APXMKA and APXWRA:
    99) !          These create gridded arrays of quantities that can later be
   100) !          linearly interpolated to any desired geographic point within
   101) !          their range by APXALL, APXMALL, and APXQ2G.  The spatial extent
   102) !          and resolution of the arrays in latitude, longitude, and altitude
   103) !          can be tailored for the application at hand.  In one extreme,
   104) !          very high interpolation accuracy can be achieved by creating a
   105) !          2x2x2 array with very small spatial increments surrounding the
   106) !          point of interest. (However, since vector quantities are obtained
   107) !          by taking differences of the quantities at adjacent points,
   108) !          computer roundoff errors will limit the benefits of making the
   109) !          increments too small.) In another extreme, a global array from
   110) !          the ground to an altitude of several Earth radii can be created.
   111) !          (In this case, the spatial resolution is limited by the need to
   112) !          maintain reasonable array sizes.) For most purposes, we have
   113) !          found it adequate to use a global grid with dimensions 91,151,6
   114) !          (latitude,longitude,altitude) and altitude from ground to 1274 km
   115) !          as created by GGRID with NVERT = 30 (see file test.f).
   116) !
   117) !          WARNING: execution time to create these look-up tables can be
   118) !          substantial.  It is dependent on the grid dimensions and the
   119) !          number of Epochs to be computed.  For eight epochs and dimensions
   120) !          (91,151,6), it took 76 minutes on a Sun SPARCstation IPX and the
   121) !          resulting file is 15.8 Mbytes.
   122) !
   123) !          Use APXMKA to create the interpolation tables for a single time
   124) !          without writing them.  Otherwise, use APXWRA create the tables
   125) !          and save them in a file.
   126) !
   127) !              CALL APXMKA (MSGUN, EPOCH, GPLAT,GPLON,GPALT,NLAT,NLON,NALT,
   128) !             +            WK,LWK, IST)
   129) !
   130) !              CALL APXWRA (MSGUN, FILNAM,IUN, EPOCH,NEPOCH,
   131) !             +            GPLAT,GPLON,GPALT,NLAT,NLON,NALT, WK,LWK, IST)
   132) !
   133) !          INPUTS:
   134) !            MSGUN  = Fortran unit number to write diagnostic messages.
   135) !            EPOCH  = Time formatted as UT year and fraction; e.g., 1990.0
   136) !                     is 0 UT 1 Jan 1990.  For APXMKA EPOCH is single valued;
   137) !                     for APXWRA EPOCH contains NEPOCH values.
   138) !            GPLAT,GPLON,GPALT = Grid point latitudes, longitudes, and
   139) !                     altitudes.  NLAT values in GPLAT must be in the range
   140) !                     -90. to +90. degrees North.  NLON values in GPLON
   141) !                     must be in the range -180. to 180. degrees East.  NALT
   142) !                     values in GPALT are km MSL.  Each array must be in
   143) !                     ascending numerical order but they may have arbitrary
   144) !                     spacing (as desired for model grid resolution).
   145) !                     Subroutine GGRID can be used to set up these arrays.
   146) !            NLAT,NLON,NALT = Number of latitudes, longitudes and altitudes
   147) !                     are the single dimensions of the arrays GPLAT,GPLON,
   148) !                     and GPALT.  Subroutine GGRID can be used to select
   149) !                     appropriate values.  In general, increasing the number
   150) !                     of grid points, increases the model resolution.  While
   151) !                     array values at the grid points are exact, values at
   152) !                     locations between grid points are estimated by linear
   153) !                     interpolation.  One can increase the grid resolution to
   154) !                     the point of degraded accuracy very close to the poles
   155) !                     of quantities involving east-west gradients, viz.,
   156) !                     B(1), G, H, W, Bhat(1), D1(1), D2(1), D3, E1, E2, E3,
   157) !                     F1(2), and F2(2).  This is evident with dimensions
   158) !                     301x501x15 for a global grid, 0-1000 km.
   159) !            WK    =  Work array is used internally, i.e., there is no need
   160) !                     to access the contents.  WK should not be altered
   161) !                     between initialization (by APXMKA, APXWRA or APXRDA)
   162) !                     and use (by APXGGC, APXALL, APXMALL, or APXQ2G).
   163) !            LWK   =  Dimension of WK >=  NLAT*NLON*NALT*5 + NLAT+NLON+NALT
   164) !
   165) !          Additional INPUTS for APXWRA only:
   166) !            FILNAM = file name where arrays are stored.
   167) !            IUN    = Fortran unit number to be associated with FILNAM.
   168) !            NEPOCH = Number of times in EPOCH.
   169) !
   170) !          RETURNS:
   171) !            IST = Return status: = 0  okay
   172) !                                 > 0  failed
   173) !
   174) !          Declarations for APXMKA formal arguments:
   175) !            DIMENSION GPLAT(NLAT), GPLON(NLON), GPALT(NALT), WK(LWK)
   176) !
   177) !          Additional declarations for APXWRA formal arguments:
   178) !            DIMENSION EPOCH(NEPOCH)
   179) !            CHARACTER*(*) FILNAM
   180) !
   181) !------------------------------------------------------------------------------
   182) !
   183) !       ENTRY APXRDA:
   184) !          Read back tables (previously created by calling APXWRA) in
   185) !          preparation for magnetic coordinate determination (using APXALL,
   186) !          APXMALL, APXQ2G).  If multiple dates were stored, interpolate
   187) !          in time.
   188) !
   189) !              CALL APXRDA (MSGUN, FILNAM,IUN, DATE, WK,LWK, IST)
   190) !
   191) !          INPUTS:
   192) !            MSGUN  = Fortran unit number to write diagnostic messages.
   193) !            FILNAM = file name where arrays are stored.
   194) !            IUN    = Fortran unit number to be associated with FILNAM.
   195) !            DATE   = Time formatted as UT year and fraction; e.g., 1990.0
   196) !                     is 0 UT 1 Jan 1990.
   197) !            WK,LWK = Same as APXMKA
   198) !
   199) !          RETURNS:
   200) !            IST = Return status: = 0  okay
   201) !                                 = -1 non-fatal date problem
   202) !                                 > 0  failed
   203) !
   204) !          Declarations for APXRDA formal arguments:
   205) !            CHARACTER*(*) FILNAM
   206) !            DIMENSION WK(LWK)
   207) !
   208) !------------------------------------------------------------------------------
   209) !
   210) !       ENTRY APXGGC:
   211) !          Obtain grid coordinates from tables read back using APXRDA.
   212) !
   213) !              CALL APXGGC (MSGUN, WK,LWK, GPLAT,GPLON,GPALT,NLAT,NLON,NALT,
   214) !             +            IST)
   215) !
   216) !          INPUTS:
   217) !            MSGUN  = Fortran unit number to write diagnostic messages.
   218) !            WK,LWK = Same as APXMKA
   219) !
   220) !          RETURNS:
   221) !            GPLAT,GPLON,GPALT = Grid point latitudes, longitudes, and
   222) !                                altitudes.  Units are degrees and kilometers.
   223) !            NLAT,NLON,NALT    = Number of latitudes, longitudes and altitudes.
   224) !            IST               = Return status, where IST=0 implies okay.
   225) !
   226) !          Declarations for APXGGC formal arguments:
   227) !            DIMENSION GPLAT(NLAT), GPLON(NLON), GPALT(NALT), WK(LWK)
   228) !
   229) !
   230) !------------------------------------------------------------------------------
   231) !
   232) !
   233) !       ENTRY APXALL:
   234) !          Determine Apex coordinates by interpolation from precalculated
   235) !          arrays.  First call APXMKA, APXWRA, or APXRDA to load look-up
   236) !          tables in memory.
   237) !
   238) !              CALL APXALL (GLAT,GLON,ALT, WK, A,ALAT,ALON, IST)
   239) !
   240) !          INPUTS:
   241) !            GLAT = Geographic (geodetic) latitude, degrees, must be within
   242) !                   the grid domain (GPLAT(1) <= GLAT <= GPLAT(NLAT)).
   243) !            GLON = Geographic (geodetic) longitude, degrees, must be within
   244) !                   one revolution of the grid domain:
   245) !                     GPLON(1) <= GLON-360.,GLON, or GLON+360. <= GPLON(NLON))
   246) !            ALT  = Altitude, km
   247) !            WK   = same as entry APXMKA
   248) !          RETURNS:
   249) !            A    = Apex radius, normalized by Req
   250) !            ALAT = Apex latitude, degrees
   251) !            ALON = Apex longitude, degrees
   252) !            IST  = Return status:  okay (0); or failure (1).
   253) !
   254) !          Dimensions of non-scalar arguments to APXALL:
   255) !            WK(LWK)
   256) !
   257) !
   258) !------------------------------------------------------------------------------
   259) !
   260) !       ENTRY APXMALL:
   261) !          Compute Modified Apex coordinates, quasi-dipole coordinates,
   262) !          base vectors and other parameters by interpolation from
   263) !          precalculated arrays.  First call APXMKA, APXWRA, or APXRDA
   264) !          to load look-up tables in memory.
   265) !
   266) !              CALL APXMALL (GLAT,GLON,ALT,HR, WK,
   267) !             +             B,BHAT,BMAG,SI,
   268) !             +             ALON,
   269) !             +             XLATM,VMP,W,D,BE3,SIM,D1,D2,D3,E1,E2,E3,
   270) !             +             XLATQD,F,F1,F2 , IST)
   271) !
   272) !          Reference:  Richmond, A. D., Ionospheric Electrodynamics Using
   273) !          Magnetic Apex Coordinates, J. Geomag. Geoelectr., 47, 191-212, 1995.
   274) !
   275) !          INPUTS:
   276) !            GLAT = Geographic (geodetic) latitude, degrees, must be within
   277) !                   the grid domain (GPLAT(1) <= GLAT <= GPLAT(NLAT)).
   278) !            GLON = Geographic (geodetic) longitude, degrees, must be within
   279) !                   one revolution of the grid domain:
   280) !                     GPLON(1) <= GLON-360.,GLON, or GLON+360. <= GPLON(NLON))
   281) !            ALT  = Altitude, km
   282) !            HR   = Reference altitude, km (used only for Modified Apex coords)
   283) !            WK   = same as entry APXMKA
   284) !          RETURNS:
   285) !            B      = magnetic field components (east, north, up), in nT
   286) !            BHAT   = components (east, north, up) of unit vector along
   287) !                     geomagnetic field direction
   288) !            BMAG   = magnitude of magnetic field, in nT
   289) !            SI     = sin(I)
   290) !            ALON   = apex longitude = modified apex longitude = quasi-dipole
   291) !                     longitude, degrees
   292) !            XLATM  = Modified Apex latitude, degrees
   293) !            VMP    = magnetic potential, in T.m
   294) !            W      = W of reference above, in km**2 /nT  (i.e., 10**15 m**2 /T)
   295) !            D      = D of reference above
   296) !            BE3    = B_e3 of reference above (= Bmag/D), in nT
   297) !            SIM    = sin(I_m) described in reference above
   298) !            D1,D2,D3,E1,E2,E3 = components (east, north, up) of base vectors
   299) !                     described in reference above
   300) !            XLATQD = quasi-dipole latitude, degrees
   301) !            F      = F described in reference above for quasi-dipole
   302) !                     coordinates
   303) !            F1,F2  = components (east, north) of base vectors described in
   304) !                     reference above for quasi-dipole coordinates
   305) !            IST    = Return status:  okay (0); or failure (1).
   306) !
   307) !          Dimensions of non-scalar arguments to APXMALL:
   308) !            GPLAT(NLAT),GPLON(NLON),GPALT(NALT),WK(LWK),
   309) !            B(3),BHAT(3),D1(3),D2(3),D3(3), E1(3),E2(3),E3(3), F1(2),F2(2)
   310) !
   311) !
   312) !------------------------------------------------------------------------------
   313) !
   314) !       ENTRY APXQ2G:
   315) !          Convert from quasi-dipole to geodetic coordinates, APXQ2G
   316) !          (input magnetic, output geodetic) is the functional inverse
   317) !          of APXALL or APXMALL (input geodetic, output magnetic).  First
   318) !          call APXMKA, APXWRA, or APXRDA to load look-up tables in memory.
   319) !
   320) !              CALL APXQ2G (QDLAT,QDLON,ALT, WK, GDLAT,GDLON, IST)
   321) !
   322) !          INPUTS:
   323) !            QDLAT = quasi-dipole latitude in degrees
   324) !            QDLON = quasi-dipole longitude in degrees
   325) !            ALT   = altitude in km
   326) !            WK    = same as entry APXMKA
   327) !          RETURNS:
   328) !            GDLAT = geodetic latitude in degrees
   329) !            GDLON = geodetic longitude in degrees
   330) !            IST   = Return status: =  0 okay
   331) !                                   = -1 non-fatal, results are not as
   332) !                                        close as PRECISE
   333) !                                   >  0 failure
   334) !
   335) !          Dimensions of non-scalar arguments to APXQ2G:
   336) !            WK(LWK)
   337) !

   421) !-------------------------------Commons---------------------------------
   422) !
   423) !          Common APXDIPL is assigned in MAKEXYZV but computed in DYPOL
   424) !            COLAT = geocentric colatitude (degrees) of north geomagnetic pole
   425) !            ELON  = geocentric east longitude (degrees) of north geomagnetic
   426) !                    pole
   427) !            VP    = magnetic potential at 1 RE, south geomagnetic pole
   428) !            CTP   = cos(colat*dtor)
   429) !            STP   = sin(colat*dtor)
   430) 
   431)       real(r8) colat, elon, vp, ctp, stp
   432)       COMMON /APXDIPL/  COLAT,ELON,VP,CTP,STP
   433) 
   434) !          Common APXCON is assigned here
   435) !            RTOD, DTOR = radians to degrees (180/pi) and inverse
   436) !            RE, REQ    = 6371.2, 6378.160 (Mean and equatorial Earth radius)
   437) !            MSGU       = MSGUN to be passed to subroutines
   438) !            POLA       = Pole angle (deg); when the geographic latitude is
   439) !                         poleward of POLA, X,Y,Z,V are forced to be constant.
   440) !                         for all longitudes at each altitude
   441) !
   442) ! bf: separate ints from reals in commons:
   443)       integer msgu
   444)       COMMON /APXCON_int/ MSGU
   445) 
   446)       real(r8) rtod, dtor, re, req, pola
   447)       COMMON /APXCON/ RTOD,DTOR,RE,REQ,POLA
   448) !
   449) !------------------------------Arguments--------------------------------
   450) !
   451)       integer msgun, nlat, nlon, nalt, lwk, ist
   452)       real(r8) GPLAT(NLAT),GPLON(NLON),GPALT(NALT), EPOCH(*), WK(LWK)
   453) !
   454) !-----------------------------Parameters------------------------------
   455) !
   456) !          XMISS   = value used to fill returned variables when requested
   457) !                    point is outside array boundaries
   458) !          GLATLIM = Extreme polar latitude allowed before changing east-west
   459) !                    gradient calculation to avoid possible underflow at
   460) !                    poles.  GLATLIM is superseded when the second to last
   461) !                    grid point value is closer to a pole.

  1665) !          Computes apex latitude, longitude.
  1666) !          990309 A. D. Richmond
  1667) !          INPUT:
  1668) !            ALT      = altitude
  1669) !            FX,FY,FZ = interpolated values of x,y,z, plus pseudodipole
  1670) !                       component
  1671) !          OUTPUT:
  1672) !            A    = apex radius, normalized by req
  1673) !            ALAT = apex latitude, degrees
  1674) !            ALON = apex longitude, degrees
  1675) !
  1676) !          Mod (Mar 99):  Lines 19-30 are changed from the original in order
  1677) !          to avoid a truncation error near the magnetic equator.  What was
  1678) !          done is to make use of the identity
  1679) !
  1680) !                  SIN(ALAT/RTOD)**2 + COS(ALAT/RTOD)**2 = 1,
  1681) !
  1682) !          so that since
  1683) !
  1684) !                  COS(ALAT/RTOD)**2 = 1/A (Eq. 1),
  1685) !
  1686) !          then
  1687) !
  1688) !                  SIN(ALAT/RTOD)**2 = (A-1)/A (Eq. 2)
  1689) !
  1690) !          Below AM1 = A-1.  If AM1 is less than 1, use Eq. 1;
  1691) !          otherwise use Eq. 2.  Mathematically, both equations should
  1692) !          give identical results, but numerically it is better to use
  1693) !          that function ASIN or ACOS that has the smaller argument.
  1694) !          The jump from one equation to the other occurs at ALAT = 45.
  1695)

  1697)       ALON  = ATAN2(FY,FX)*RTOD
  1698)       SLP2  = FZ*FZ

  1700)       XNORM = SLP2 + X2PY2
  1701)       SLP2  = SLP2/XNORM
  1702)       CLP2  = X2PY2/XNORM
  1703)       AM1   = (RE*SLP2 + ALT)/(REQ*CLP2)
  1704)       A = 1._r8 + AM1
  1705) 
  1706)       IF (AM1.LT.0._r8) THEN
  1707)         IERR = 1
  1708)         write(iulog,*) 'Missing alat returned because point lies below field line that peaks at Earth surface.'
  1709)         RETURN
  1710)       ELSEIF (AM1.LT.1._r8) THEN
  1711)         ALAT = RTOD*ASIN(SQRT(AM1/A))
  1712)       ELSE
  1713)         ALAT = RTOD*ACOS(1._r8/SQRT(A))
  1714)       ENDIF
  1715) !  If southern magnetic hemisphere, reverse sign of alat
  1716)       IF (FZ.LT.0._r8) ALAT = - ALAT
  1717)       RETURN

  1745) !            F       = F of article
  1746) !            D       = D of article
  1747) !            W       = W of article
  1748) !            BHAT    = unit vector along geomagnetic field direction
  1749) !            D1...F2 = base vectors of article
  1750) !-----------------------------------------------------------------------

  1753)       implicit none
  1754) !

  1757) ! bf: separate ints from reals in commons:

  1759)       COMMON /APXCON_int/ MSGU
  1760) 
  1761)       real(r8) rtod, dtor, re, req, pola
  1762)       COMMON /APXCON/ RTOD,DTOR,RE,REQ,POLA
  1763) !

  1766)       real(r8) hr, xlatm
  1767)       real(r8) GRCLM(3),CLMGRP(3),RGRLP(3),B(3)
  1768)       real(r8) clm, r3_2, bmag, sim, si, f, d, w
  1769)       real(r8) BHAT(3),D1(3),D2(3),D3(3),E1(3),E2(3),E3(3),F1(2),F2(2)
  1770) !
  1771) !---------------------------Local variables-----------------------------
  1772) !

  1774)

  1780)       SIMOSLM = 2._r8/SQRT(4._r8 - 3._r8*CLM*CLM)
  1781)       SIM = SIMOSLM*SIN(XLATM*DTOR)
  1782)       BMAG = SQRT(B(1)*B(1) + B(2)*B(2) + B(3)*B(3))
  1783)       D1DB = 0._r8
  1784)       D2DB = 0._r8
  1785)       DO 10 I=1,3
  1786)         BHAT(I) = B(I)/BMAG
  1787)         D1(I) = RR*CLMGRP(I)
  1788)         D1DB = D1DB + D1(I)*BHAT(I)
  1789)         D2(I) = RR*SIMOSLM*GRCLM(I)
  1790)    10   D2DB = D2DB + D2(I)*BHAT(I)

  1792)       DO 15 I=1,3

  1794)    15   D2(I) = D2(I) - D2DB*BHAT(I)

  1796)       E3(2) = D1(3)*D2(1) - D1(1)*D2(3)
  1797)       E3(3) = D1(1)*D2(2) - D1(2)*D2(1)
  1798)       D = BHAT(1)*E3(1) + BHAT(2)*E3(2) + BHAT(3)*E3(3)
  1799)       DO 20 I=1,3
  1800)         D3(I) = BHAT(I)/D
  1801) ! Following step may be unnecessary, but it ensures that e3 lies along bhat.
  1802)         E3(I) = BHAT(I)*D
  1803)    20   CONTINUE
  1804)       E1(1) = D2(2)*D3(3) - D2(3)*D3(2)
  1805)       E1(2) = D2(3)*D3(1) - D2(1)*D3(3)
  1806)       E1(3) = D2(1)*D3(2) - D2(2)*D3(1)
  1807)       E2(1) = D3(2)*D1(3) - D3(3)*D1(2)
  1808)       E2(2) = D3(3)*D1(1) - D3(1)*D1(3)
  1809)       E2(3) = D3(1)*D1(2) - D3(2)*D1(1)
  1810)       W = RR*RR*CLM*ABS(SIM)/(BMAG*D)
  1811)       SI = -BHAT(3)
  1812)       F1(1) =  RGRLP(2) 
  1813)       F1(2) = -RGRLP(1)
  1814)       F2(1) = -D1(2)*R3_2
  1815)       F2(2) =  D1(1)*R3_2
  1816)       F = F1(1)*F2(2) - F1(2)*F2(1)
  1817)       RETURN

  1899)       SUBROUTINE MAKEXYZV (EPOCH,NLAT,NLON,NALT,GPLAT,GPLON,GPALT, &
  1900)                            X,Y,Z,V)
  1901) !
  1902) !-----------------------------------------------------------------------
  1903) !          Sets up grid arrays for later interpolation
  1904) !          940822 A. D. Richmond, NCAR
  1905) !          INPUT:
  1906) !            EPOCH = year and fraction (e.g., 1994.50 for 1994 July 2)
  1907) !            NLAT,NLON,NALT = triple dimensions of X,Y,Z,V and respective
  1908) !                    single dimensions of GP___ arrays
  1909) !            GPLAT,GPLON,GPALT = grid point latitudes, longitudes and altitudes
  1910) !          OUTPUT:
  1911) !            X = array containing cos(lambda')cos(phi_a) less pseudodipole
  1912) !                component
  1913) !            Y = array containing cos(lambda')sin(phi_a) less pseudodipole
  1914) !                component
  1915) !            Z = array containing sin(lambda') less pseudodipole component
  1916) !            V = array containing ((magnetic potential)/vp)*((re+height)/re)**2,
  1917) !                less pseudodipole component
  1918) !
  1919) !          Modification (99 Mar):  Make X,Y,Z,V constant near the poles
  1920) !          for all GPLON(j) at each height.  Add POLA to APXCON
  1921) !            POLA       = Pole angle (deg); when the geographic latitude is
  1922) !                         poleward of POLA, X,Y,Z,V are forced to be constant.
  1923) !                         for all longitudes at each altitude.  POLA is defined
  1924) !                         in APXMKA (POLA = 90. - SQRT (PRECISE) * RTOD),
  1925) !                         which currently makes POLA = 89.995
  1926) !-----------------------------------------------------------------------
  1927) !
  1928)       use shr_kind_mod, only: r8 => shr_kind_r8
  1929)       implicit none
  1930) !
  1931) !-------------------------------Commons---------------------------------
  1932) !
  1933) ! bf: separate ints from reals in commons:
  1934)       integer msgu
  1935)       COMMON /APXCON_int/ MSGU
  1936) 
  1937)       real(r8) rtod, dtor, re, req, pola
  1938)       COMMON /APXCON/ RTOD,DTOR,RE,REQ,POLA
  1939) 
  1940)       real(r8) colat, elon, vp, ctp, stp
  1941)       COMMON /APXDIPL/ COLAT,ELON,VP,CTP,STP
  1942) !
  1943) !------------------------------Arguments--------------------------------
  1944) !
  1945)       integer nlat, nlon, nalt
  1946) 
  1947)       real(r8) epoch
  1948)       real(r8) X(NLAT,NLON,NALT), Y(NLAT,NLON,NALT), Z(NLAT,NLON,NALT), &
  1949)            V(NLAT,NLON,NALT), GPLAT(NLAT), GPLON(NLON), GPALT(NALT)
  1950) !
  1951) !---------------------------Local variables-----------------------------
  1952) !
  1953)       integer kpol, i, j, k
  1954) 
  1955)       real(r8) xmag, ymag, zdown, alat, phia, bmag, vmp, slp, clp, phiar
  1956)       real(r8)  vnor, rp, reqam1, a, ct, st, reqore, rqorm1, ctm, stmcpm
  1957)       real(r8)  stmspm, cp, sp
  1958) !
  1959) !-----------------------------------------------------------------------
  1960) !
  1961)       CALL COFRM (EPOCH)
  1962)       CALL DYPOL (COLAT,ELON,VP)
  1963)       CTP = COS (COLAT*DTOR)
  1964)       STP = SIN (COLAT*DTOR)
  1965)       REQORE = REQ/RE
  1966)       RQORM1 = REQORE-1._r8
  1967) 
  1968)       DO 100 I=1,NLAT
  1969)       CT = SIN (GPLAT(I)*DTOR)
  1970)       ST = COS (GPLAT(I)*DTOR)
  1971)       KPOL = 0
  1972)       IF (ABS (GPLAT(I)) .GT. POLA) KPOL = 1
  1973) 
  1974)       DO 100 J=1,NLON
  1975)       IF (KPOL .EQ. 0) GO TO 20
  1976)       IF (J    .EQ. 1) GO TO 20
  1977) !          KPOL = 1 (poleward of POLA) and first lon's XYZV are defined
  1978)       DO 10 K=1,NALT
  1979)       V(I,J,K) = V(I,1,K)
  1980)       X(I,J,K) = X(I,1,K)
  1981)       Y(I,J,K) = Y(I,1,K)
  1982)    10 Z(I,J,K) = Z(I,1,K)
  1983)       GO TO 100
  1984) 
  1985)    20 CP  = COS ((GPLON(J)-ELON)*DTOR)
  1986)       SP  = SIN ((GPLON(J)-ELON)*DTOR)
  1987) !           ctm   is pseudodipole component of z
  1988) !          -ctm   is pseudodipole component of v
  1989) !          stmcpm is pseudodipole component of x
  1990) !          stmspm is pseudodipole component of y
  1991)       CTM    = CTP*CT + STP*ST*CP
  1992)       STMCPM = ST*CTP*CP - CT*STP
  1993)       STMSPM = ST*SP
  1994) 
  1995)       DO 30 K=1,NALT
  1996)       CALL APEX (EPOCH,GPLAT(I),GPLON(J),GPALT(K), &
  1997)                  A,ALAT,PHIA,BMAG,XMAG,YMAG,ZDOWN,VMP)
  1998)       VNOR = VMP/VP
  1999)       RP = 1._r8 + GPALT(K)/RE
  2000)       V(I,J,K) = VNOR*RP*RP + CTM
  2001)       REQAM1 = REQ*(A-1._r8)
  2002)       SLP = SQRT(max(REQAM1-GPALT(K),0._r8)/(REQAM1+RE))
  2003) !          Reverse sign of slp in southern magnetic hemisphere
  2004)       IF (ZDOWN.LT.0._r8) SLP = -SLP
  2005)       CLP = SQRT (RP/(REQORE*A-RQORM1))
  2006)       PHIAR = PHIA*DTOR
  2007)       X(I,J,K) = CLP*COS (PHIAR) - STMCPM
  2008)       Y(I,J,K) = CLP*SIN (PHIAR) - STMSPM
  2009)       Z(I,J,K) = SLP - CTM
  2010)    30 CONTINUE
  2011) 
  2012)   100 CONTINUE
  2013) 
  2014)       RETURN
  2015)       end subroutine makexyzv

  2221) !***DESCRIPTION                                                                 
  2222) !     The method used is as follow:                                             
  2223) !       1. Calculates step size as a function of the geomagnetic                
  2224) !          dipole coordinates of the starting point.                            
  2225) !       2. Determine direction of trace                                         
  2226) !       3. Convert the geodetic coordinates of the starting point               
  2227) !          to the cartesian coordinates for tracing.                            
  2228) !       Loop:                                                                   
  2229) !       i)   Increment step count, if count > 200,                              
  2230) !            assume it is dipole field, call DIPAPX to                          
  2231) !            determine Apex coordinates else continue.                          
  2232) !       ii)  Get field components in X, Y and Z direction                       
  2233) !       iii) Call ITRACE to trace field line.                                   
  2234) !       iv)  Test if Apex passed call FNDAPX to determine Apex coordinates      
  2235) !            else loop:                                                         
  2236) !                                                                               
  2237) !   INPUT                                                                       
  2238) !     GDLAT  Latitude of starting point (Deg)                                   
  2239) !     GLON   Longitude (East=+) of starting point (Deg)                         
  2240) !     ALT    Ht. of starting point (Km)                                         
  2241) !                                                                               
  2242) !  OUTPUT
  2243) !          A  (Apex height + REQ)/REQ, where REQ = equatorial Earth radius.
  2244) !             A is analogous to the L value in invariant coordinates.
  2245) !       ALAT  Apex Lat. (deg)                                                   
  2246) !       ALON  Apex Lon. (deg)                                                   
  2247) !       XMAG  North component of magnetic field at starting point
  2248) !       YMAG  East component of magnetic field at starting point
  2249) !       ZMAG  Down component of magnetic field at starting point
  2250) !          F  Magnetic field magnitude at starting point
  2251) !
  2252) !     COMMON Blocks Used                                                        
  2253) !     /APXIN/ YAPX(3,3)                                                         
  2254) !       YAPX    Matrix of cartesian coordinates (loaded columnwise)             
  2255) !               of the 3 points about APEX. Set in subprogram ITRACE.           
  2256) !                                                                               
  2257) !   /DIPOLE/COLAT,ELON,VP,CTP,STP                           
  2258) !     COLAT   Geographic colatitude of the north pole of the                    
  2259) !             earth-centered dipole (Deg).                                      
  2260) !     ELON    Geographic longitude of the north pole of the                     
  2261) !             earth-centered dipole (Deg).                                      
  2262) !     VP      Magnetic potential magnitude at geomagnetic pole (T.m)
  2263) !     CTP     cos(COLAT*DTOR)
  2264) !     STP     sin(COLAT*DTOR)
  2265) !                                                                               
  2266) !     /ITRA/ NSTP, Y(3), YOLD(3), SGN, DS                                       
  2267) !     NSTP      Step count. Incremented in subprogram LINAPX.                   
  2268) !     Y         Array containing current tracing point cartesian coordinates.   
  2269) !     YOLD      Array containing previous tracing point cartesian coordinates.
  2270) !     SGN       Determines direction of trace. Set in subprogram LINAPX         
  2271) !     DS        Step size (Km) Computed in subprogram LINAPX.                   
  2272) !                                                                               
  2273) !     /FLDCOMD/ BX, BY, BZ, BB                                                  
  2274) !     BX        X comp. of field vector at the current tracing point (Gauss)    
  2275) !     BY        Y comp. of field vector at the current tracing point (Gauss)    
  2276) !     BZ        Z comp. of field vector at the current tracing point (Gauss)    
  2277) !     BB        Magnitude of field vector at the current tracing point (Gauss)  
  2278) !                                                                               
  2279) !***REFERENCES  Stassinopoulos E. G. , Mead Gilbert D., X-841-72-17             
  2280) !                 (1971) GSFC, Greenbelt, Maryland                              
  2281) !                                                                               
  2282) !***ROUTINES CALLED  GD2CART,CONVRT,FELDG,
  2283) !                    ITRACE,DIPAPX,FNDAPX                                       
  2284) !                                                                               
  2285) !***END PROLOGUE  LINAPX                                                        
  2286) !-----------------------------------------------------------------------
  2287) !
  2288)       use shr_kind_mod, only: r8 => shr_kind_r8
  2289)       implicit none
  2290) !
  2291) !-------------------------------Commons---------------------------------
  2292) !
  2293)       real(r8) bx, by, bz, bb
  2294)       COMMON /FLDCOMD/ BX, BY, BZ, BB
  2295) 
  2296)       real(r8) yapx
  2297)       COMMON /APXIN/   YAPX(3,3)
  2298) 
  2299)       real(r8) colat, elon, vp, ctp, stp
  2300)       COMMON /DIPOLE/  COLAT,ELON,VP,CTP,STP
  2301) !
  2302) ! bf: separate ints and reals into 2 different commons:
  2303) !     COMMON /ITRA/    NSTP, Y(3), YP(3),  SGN, DS
  2304) 
  2305)       integer nstp
  2306)       COMMON /ITRA_int/    NSTP
  2307) 
  2308)       real(r8) y, yp, sqn, ds
  2309)       COMMON /ITRA/  Y(3), YP(3),  SGN, DS
  2310) !
  2311) !------------------------------Arguments--------------------------------
  2312) !
  2313)       real(r8) gdlat, glon, alt, a, alat, alon, xmag, ymag, zmag, f
  2314) !
  2315) !-----------------------------Parameters------------------------------
  2316) !
  2317)       integer maxs
  2318)       real(r8) rtod, dtor, re, req
  2319)       PARAMETER (MAXS = 200, &
  2320)                  RTOD=5.72957795130823E1_r8, DTOR=1.745329251994330E-2_r8, &
  2321)                  RE=6371.2_r8, REQ=6378.160_r8)
  2322) !
  2323) !---------------------------Local variables-----------------------------
  2324) !
  2325)       integer ientry, i, j
  2326) 
  2327)       real(r8) rho, bnrth, xlon, ht, beast, iapx, babs, bdown, xlat, r
  2328)       real(r8)  gclat, sgn, singml, cgml2
  2329) !
  2330) !-----------------------------------------------------------------------
  2331) !
  2332) !          Determine step size as a function of geomagnetic dipole
  2333) !          coordinates of the starting point
  2334)       CALL CONVRT (2,GDLAT,ALT,GCLAT,R)
  2335) !     SINGML = .98*SIN(GCLAT*DTOR) + .199*COS(GCLAT*DTOR)*
  2336)       SINGML = CTP*SIN(GCLAT*DTOR) +  STP*COS(GCLAT*DTOR)* &
  2337)                                                    COS((GLON-ELON)*DTOR)
  2338) 
  2339) !          May 99: avoid possible divide by zero (when SINGML = 1.)
  2340)       CGML2 = max (0.25_r8,1._r8-SINGML*SINGML)
  2341)       DS = .06_r8*R/CGML2 - 370._r8
  2342) !old:      Limit DS to its value at 60 deg gm latitude.
  2343) !old: DS = .06*R/(1.-SINGML*SINGML) - 370.
  2344) !old: IF (DS .GT. 1186.) DS = 1186.
  2345) 
  2346) !          Initialize YAPX array
  2347)       DO 4 J=1,3
  2348)       DO 4 I=1,3
  2349)     4 YAPX(I,J) = 0._r8
  2350) 
  2351) !          Convert from geodetic to earth centered cartesian coordinates
  2352)       CALL GD2CART (GDLAT,GLON,ALT,Y(1),Y(2),Y(3))
  2353)       NSTP = 0
  2354) 
  2355) !          Get magnetic field components to determine the direction for
  2356) !          tracing the field line
  2357)       CALL FELDG (1,GDLAT,GLON,ALT,XMAG,YMAG,ZMAG,F)
  2358)       SGN = SIGN (1._r8,-ZMAG)
  2359) 
  2360) !          Use cartesian coordinates to get magnetic field components
  2361) !          (from which gradients steer the tracing)
  2362)    10 IENTRY=2
  2363) 
  2364)       CALL FELDG (IENTRY,Y(1)/RE,Y(2)/RE,Y(3)/RE,BX,BY,BZ,BB)
  2365) 
  2366)       NSTP = NSTP + 1
  2367) 
  2368) !          Quit if too many steps
  2369)       IF (NSTP .GE. MAXS) THEN                                                 
  2370)         RHO = SQRT(Y(1)*Y(1) + Y(2)*Y(2))
  2371)         CALL CONVRT(3,XLAT,HT,RHO,Y(3))
  2372)         XLON = RTOD*ATAN2(Y(2),Y(1))
  2373) 	CALL FELDG (1,XLAT,XLON,HT,BNRTH,BEAST,BDOWN,BABS)
  2374) 	CALL DIPAPX (XLAT,XLON,HT,BNRTH,BEAST,BDOWN,A,ALON)
  2375)         ALAT = -SGN*RTOD*ACOS(SQRT(1._r8/A))
  2376)         RETURN                                                                
  2377)       END IF                                                                   
  2378) 
  2379) !          Find next point using adams algorithm after 7 points
  2380)        CALL ITRACE (IAPX)
  2381)        IF (IAPX .EQ. 1) GO TO 10
  2382) 
  2383) !          Maximum radius just passed.  Find apex coords
  2384)       CALL FNDAPX (ALT,ZMAG,A,ALAT,ALON)
  2385)       RETURN
  2386)       end subroutine linapx

  2390)       SUBROUTINE ITRACE (IAPX)
  2391) !
  2392) !-----------------------------------------------------------------------
  2393) !***BEGIN PROLOGUE  ITRACE                                                      
  2394) !***DATE WRITTEN   731029   (YYMMDD)                                            
  2395) !***AUTHOR  CLARK, W. N.O.A.A. ERL LAB.                                         
  2396) !***REVISION DATE  880201, H. Passi 
  2397) !***PURPOSE  Field line integration routine.                                    
  2398) !***DESCRIPTION                                                                 
  2399) !     It uses 4-point ADAMS formula after initialization.                       
  2400) !     First 7 iterations advance point by 3 steps.                              
  2401) !                                                                               
  2402) !   INPUT                                                                       
  2403) !            Passed in through the common blocks ITRA, FLDCOMD.                 
  2404) !            See the description below.                                         
  2405) !   OUTPUT                                                                      
  2406) !    IAPX    Flag set to 2 when APEX passed, otherwise set to 1.                
  2407) !                                                                               
  2408) !            Passed out through the common block APXIN.                         
  2409) !            See the description below.                                         
  2410) !                                                                               
  2411) !     COMMON Blocks Used                                                        
  2412) !     /APXIN/ YAPX(3,3)                                                         
  2413) !     YAPX      Matrix of cartesian coordinates (loaded columnwise)             
  2414) !               of the 3 points about APEX. Set in subprogram ITRACE.           
  2415) !                                                                               
  2416) !     /FLDCOMD/ BX, BY, BZ, BB                                                  
  2417) !     BX        X comp. of field vector at the current tracing point (Gauss)    
  2418) !     BY        Y comp. of field vector at the current tracing point (Gauss)    
  2419) !     BZ        Z comp. of field vector at the current tracing point (Gauss)    
  2420) !     BB        Magnitude of field vector at the current tracing point (Gauss)  
  2421) !                                                                               
  2422) !     /ITRA/ NSTP, Y(3), YOLD(3), SGN, DS                                       
  2423) !     NSTP      Step count for line integration.                                
  2424) !               Incremented in subprogram LINAPX.                               
  2425) !     Y         Array containing current tracing point cartesian coordinates.   
  2426) !     YOLD      Array containing previous tracing point cartesian coordinates.  
  2427) !     SGN       Determines direction of trace. Set in subprogram LINAPX         
  2428) !     DS        Integration step size (arc length Km)                           
  2429) !               Computed in subprogram LINAPX.                                  
  2430) !                                                                               
  2431) !***REFERENCES  reference 1                                                     
  2432) !                                                                               
  2433) !***ROUTINES CALLED  None                                                       
  2434) !***COMMON BLOCKS    APXIN,FLDCOMD,ITRA                                         
  2435) !***END PROLOGUE  ITRACE                                                        
  2436) ! **                                                                            
  2437) !-----------------------------------------------------------------------
  2438) !
  2439)       use shr_kind_mod, only: r8 => shr_kind_r8
  2440)       implicit none
  2441) !
  2442) !-------------------------------Commons---------------------------------
  2443) !
  2444) ! bf: separate ints and reals into 2 different commons:
  2445) !     COMMON /ITRA/    NSTP, Y(3), YOLD(3), SGN, DS
  2446) 
  2447)       integer nstp
  2448)       COMMON /ITRA_int/    NSTP
  2449) 
  2450)       real(r8) y, yold, sgn, ds
  2451)       COMMON /ITRA/ Y(3), YOLD(3), SGN, DS
  2452) 
  2453)       real(r8) bx, by, bz, bb
  2454)       COMMON /FLDCOMD/ BX, BY, BZ, BB
  2455) 
  2456)       real(r8) yapx
  2457)       COMMON /APXIN/   YAPX(3,3)
  2458) !
  2459) !------------------------------Arguments--------------------------------
  2460) !
  2461)       real(r8) iapx
  2462) !
  2463) !---------------------------Local variables-----------------------------
  2464) !
  2465)       integer i, j
  2466) 
  2467)       real(r8)  YP(3, 4)
  2468)       real(r8) rc, d24, rp, term, d2, d12, d6
  2469) 
  2470)       SAVE
  2471) !
  2472) !-------------------------Statement Functions----------------------------
  2473) !
  2474)       real(r8) rdus, d, e, f
  2475)       RDUS(D,E,F) = SQRT( D**2 + E**2 + F**2 )
  2476) !
  2477) !-----------------------------------------------------------------------
  2478) !
  2479)       IAPX = 1
  2480) !          Field line is defined by the following differential equations
  2481) !          in cartesian coordinates:
  2482)       YP(1, 4) = SGN*BX/BB
  2483)       YP(2, 4) = SGN*BY/BB
  2484)       YP(3, 4) = SGN*BZ/BB
  2485)       IF (NSTP .GT. 7) GO TO 90
  2486) 
  2487) !          First seven steps use this block
  2488)       DO 80 I = 1, 3
  2489)       GO TO (10, 20, 30, 40, 50, 60, 70) NSTP
  2490)    10 D2  = DS/2._r8
  2491)       D6  = DS/6._r8
  2492)       D12 = DS/12._r8
  2493)       D24 = DS/24._r8
  2494)       YP(I,1)   = YP(I,4)
  2495)       YOLD(I)   = Y(I)
  2496)       YAPX(I,1) = Y(I)
  2497)       Y(I) = YOLD(I) + DS*YP(I, 1)
  2498)       GO TO 80
  2499) 
  2500)    20 YP(I, 2) = YP(I,4)
  2501)       Y(I) = YOLD(I) + D2*(YP(I,2)+YP(I,1))
  2502)       GO TO 80
  2503) 
  2504)    30 Y(I) = YOLD(I) + D6*(2._r8*YP(I,4)+YP(I,2)+3._r8*YP(I,1))
  2505)       GO TO 80
  2506) 
  2507)    40 YP(I,2)  = YP(I,4)
  2508)       YAPX(I,2)= Y(I)
  2509)       YOLD(I)  = Y(I)
  2510)       Y(I)     = YOLD(I) + D2*(3._r8*YP(I,2)-YP(I,1))
  2511)       GO TO 80
  2512) 
  2513)    50 Y(I) = YOLD(I) + D12*(5._r8*YP(I,4)+8._r8*YP(I,2)-YP(I,1))
  2514)       GO TO 80
  2515) 
  2516)    60 YP(I,3)  = YP(I,4)
  2517)       YOLD(I)  = Y(I)
  2518)       YAPX(I,3)= Y(I)
  2519)       Y(I)     = YOLD(I) + D12*(23._r8*YP(I,3)-16._r8*YP(I,2)+5._r8*YP(I,1))
  2520)       GO TO 80
  2521) 
  2522)    70 YAPX(I, 1) = YAPX(I, 2)
  2523)       YAPX(I, 2) = YAPX(I, 3)
  2524)       Y(I) = YOLD(I) + D24*(9._r8*YP(I,4)+19._r8*YP(I,3)-5._r8*YP(I,2)+YP(I,1))
  2525)       YAPX(I, 3) = Y(I)
  2526)    80 CONTINUE
  2527) 
  2528) !          Signal if apex passed
  2529)       IF ( NSTP .EQ. 6 .OR. NSTP .EQ. 7) THEN
  2530) 	RC = RDUS( YAPX(1,3), YAPX(2,3), YAPX(3,3))
  2531)         RP = RDUS( YAPX(1,2), YAPX(2,2), YAPX(3,2))                             
  2532) 	IF (RC .LT. RP) IAPX=2
  2533)       ENDIF
  2534)       RETURN
  2535) 
  2536) !          Stepping block for NSTP .gt. 7
  2537)    90 DO 110 I = 1, 3                                                           
  2538)       YAPX(I, 1) = YAPX(I, 2)
  2539)       YAPX(I, 2) = Y(I)
  2540)       YOLD(I) = Y(I)
  2541)       TERM = 55._r8*YP(I, 4) - 59._r8*YP(I, 3) + 37._r8*YP(I, 2) - 9._r8*YP(I, 1)
  2542)       Y(I) = YOLD(I) + D24*TERM
  2543)       YAPX(I, 3) = Y(I)
  2544) 
  2545)       DO 100 J = 1, 3
  2546)       YP(I, J) = YP(I, J+1)
  2547)   100 CONTINUE
  2548)   110 CONTINUE                                                                  
  2549) 
  2550)       RC = RDUS (   Y(1),    Y(2),    Y(3))
  2551)       RP = RDUS (YOLD(1), YOLD(2), YOLD(3))
  2552)       IF (RC .LT. RP) IAPX=2
  2553) 
  2554)       RETURN
  2555)       END subroutine itrace

  2559)        SUBROUTINE FNDAPX (ALT,ZMAG,A,ALAT,ALON)
  2560) !
  2561) !-----------------------------------------------------------------------
  2562) !***BEGIN PROLOGUE  FNDAPX                                                      
  2563) !***DATE WRITTEN   731023   (YYMMDD)                                            
  2564) !***AUTHOR  CLARK, W., NOAA BOULDER                                             
  2565) !***REVISION DATE  940803, A. D. Richmond, NCAR 
  2566) !***PURPOSE  Finds apex coords once tracing has signalled that the apex         
  2567) !            has been passed.  
  2568) !***DESCRIPTION                                                                 
  2569) !                                                                               
  2570) !     It uses second degree interpolation routine, FINT, to find                
  2571) !     apex latitude and apex longtitude.                                        
  2572) !   INPUT                                                                       
  2573) !     ALT    Altitude of starting point
  2574) !     ZMAG   Downward component of geomagnetic field at starting point
  2575) !     NMAX   Order of IGRF coefficients being used
  2576) !     G      Array of coefficients from COFRM
  2577) !   OUTPUT                                                                      
  2578) !          A  (Apex height + REQ)/REQ, where REQ = equatorial Earth radius.
  2579) !             A is analogous to the L value in invariant coordinates.
  2580) !       ALAT  Apex Lat. (deg)                                                   
  2581) !       ALON  Apex Lon. (deg)                                                   
  2582) !                                                                               
  2583) !***LONG DESCRIPTION                                                            
  2584) !                                                                               
  2585) !     COMMON Blocks Used                                                        
  2586) !     /APXIN/ YAPX(3,3)                                                         
  2587) !     YAPX      Matrix of cartesian coordinates (loaded columnwise)             
  2588) !               of the 3 points about APEX. Set in subprogram ITRACE.           
  2589) !                                                                               
  2590) !   /DIPOLE/COLAT,ELON,VP,CTP,STP                           
  2591) !     COLAT   Geocentric colatitude of the north pole of the                    
  2592) !             earth-centered dipole (Deg).                                      
  2593) !     ELON    Geographic longitude of the north pole of the                     
  2594) !             earth-centered dipole (Deg).                                      
  2595) !     CTP     cos(COLAT*DTOR)
  2596) !     STP     sin(COLAT*DTOR)
  2597) !                                                                               
  2598) !***ROUTINES CALLED  FINT                                                       
  2599) !***COMMON BLOCKS    APXIN,DIPOLE                                         
  2600) !***END PROLOGUE  FNDAPX                                                        
  2601) !-----------------------------------------------------------------------
  2602) !
  2603)       use shr_kind_mod, only: r8 => shr_kind_r8
  2604)       use abortutils,   only: endrun
  2605)       use cam_logfile,  only: iulog
  2606)       implicit none
  2607) !
  2608) !-------------------------------Commons---------------------------------
  2609) !
  2610)       real(r8) yapx
  2611)       COMMON /APXIN/  YAPX(3,3)
  2612) 
  2613)       real(r8) colat, elon, vp, ctp, stp
  2614)       COMMON /DIPOLE/ COLAT,ELON,VP,CTP,STP
  2615) !
  2616) !------------------------------Arguments--------------------------------
  2617) !
  2618)       real(r8) alt, zmag, a, alat, alon
  2619) !
  2620) !-----------------------------Parameters------------------------------
  2621) !
  2622)       real(r8) rtod, dtor, re, req
  2623)       PARAMETER (RTOD=5.72957795130823E1_r8,DTOR=1.745329251994330E-2_r8)
  2624)       PARAMETER (RE=6371.2_r8,REQ=6378.160_r8)
  2625) !
  2626) !---------------------------Local variables-----------------------------
  2627) !
  2628)       integer i, ientry
  2629) 
  2630)       real(r8) Z(3), HT(3), Y(3)
  2631)       real(r8) xlon, ang, cang, f, xinter, rasq, ste, stfcpa, stfspa
  2632)       real(r8) sang, r, cte, gdln, x, ydum, gdlt, rho
  2633) !
  2634) !-----------------------------------------------------------------------
  2635) !
  2636) !       ****
  2637) !       ****     GET GEODETIC FIELD COMPONENTS
  2638) !       ****
  2639)       IENTRY = 1
  2640)       DO 2 I = 1,3
  2641) 	RHO = SQRT(YAPX(1,I)**2+YAPX(2,I)**2)
  2642) 	CALL CONVRT (3,GDLT,HT(I),RHO,YAPX(3,I))
  2643) 	GDLN = RTOD*ATAN2 (YAPX(2,I),YAPX(1,I))
  2644) 	CALL FELDG (IENTRY,GDLT,GDLN,HT(I),X,YDUM,Z(I),F)
  2645)     2 CONTINUE
  2646) !     ****
  2647) !     ****     FIND CARTESIAN COORDINATES AT DIP EQUATOR BY INTERPOLATION
  2648) !     ****
  2649)       DO 3 I = 1,3
  2650) 	CALL FINT(Z(1),Z(2),Z(3),YAPX(I,1),YAPX(I,2),YAPX(I,3),0._r8,Y(I))
  2651)     3 CONTINUE
  2652) !     ****
  2653) !     ****     FIND APEX HEIGHT BY INTERPOLATION
  2654) !     ****
  2655)       CALL FINT(Z(1),Z(2),Z(3),HT(1),HT(2),HT(3),0._r8,XINTER)
  2656) !          Ensure that XINTER is not less than original starting altitude (ALT)
  2657)       XINTER = max(ALT,XINTER)
  2658)       A = (REQ+XINTER)/(REQ)
  2659) !     ****
  2660) !     ****     FIND APEX COORDINATES , GIVING ALAT SIGN OF DIP AT
  2661) !     ****       STARTING POINT.  ALON IS THE VALUE OF THE GEOMAGNETIC
  2662) !     ****       LONGITUDE AT THE APEX.
  2663) !     ****
  2664)       IF (A.LT.1._r8) THEN
  2665) 	write(iulog,20)
  2666)   20    FORMAT (' BOMBED! THIS MAKES A LESS THAN ONE')
  2667) 	call endrun
  2668)       ENDIF
  2669)       RASQ = RTOD*ACOS(SQRT(1._r8/A))
  2670)       ALAT = SIGN(RASQ,ZMAG)
  2671) 
  2672) ! Algorithm for ALON:
  2673) !   Use spherical coordinates.
  2674) !   Let GP be geographic pole.
  2675) !   Let GM be geomagnetic pole (colatitude COLAT, east longitude ELON).
  2676) !   Let XLON be longitude of apex.
  2677) !   Let TE be colatitude of apex.
  2678) !   Let ANG be longitude angle from GM to apex.
  2679) !   Let TP be colatitude of GM.
  2680) !   Let TF be arc length between GM and apex.
  2681) !   Let PA = ALON be geomagnetic longitude, i.e., Pi minus angle measured 
  2682) !     counterclockwise from arc GM-apex to arc GM-GP.
  2683) !   Then, using notation C=cos, S=sin, spherical-trigonometry formulas 
  2684) !     for the functions of the angles are as shown below.  Note: STFCPA,
  2685) !     STFSPA are sin(TF) times cos(PA), sin(PA), respectively.
  2686) 
  2687)       XLON = ATAN2(Y(2),Y(1))
  2688)       ANG  = XLON-ELON*DTOR
  2689)       CANG = COS(ANG)
  2690)       SANG = SIN(ANG)
  2691)       R    = SQRT(Y(1)**2+Y(2)**2+Y(3)**2)
  2692)       CTE  = Y(3)/R
  2693)       STE  = SQRT(1._r8-CTE*CTE)
  2694)       STFCPA = STE*CTP*CANG - CTE*STP
  2695)       STFSPA = SANG*STE
  2696)       ALON = ATAN2(STFSPA,STFCPA)*RTOD
  2697)       RETURN
  2698)       END subroutine fndapx

  2702)       SUBROUTINE DIPAPX(GDLAT,GDLON,ALT,BNORTH,BEAST,BDOWN,A,ALON)
  2703) !
  2704) !-----------------------------------------------------------------------
  2705) ! Compute a, alon from local magnetic field using dipole and spherical approx.
  2706) ! 940501 A. D. Richmond
  2707) ! Input:
  2708) !   GDLAT  = geodetic latitude, degrees
  2709) !   GDLON  = geodetic longitude, degrees
  2710) !   ALT    = altitude, km
  2711) !   BNORTH = geodetic northward magnetic field component (any units)
  2712) !   BEAST  = eastward magnetic field component
  2713) !   BDOWN  = geodetic downward magnetic field component
  2714) ! Output:
  2715) !   A      = apex radius, 1 + h_A/R_eq
  2716) !   ALON   = apex longitude, degrees
  2717) !
  2718) ! Algorithm:
  2719) !   Use spherical coordinates.
  2720) !   Let GP be geographic pole.
  2721) !   Let GM be geomagnetic pole (colatitude COLAT, east longitude ELON).
  2722) !   Let G be point at GDLAT,GDLON.
  2723) !   Let E be point on sphere below apex of dipolar field line passing through G.
  2724) !   Let TD be dipole colatitude of point G, found by applying dipole formula
  2725) !     for dip angle to actual dip angle.
  2726) !   Let B be Pi plus local declination angle.  B is in the direction 
  2727) !     from G to E.
  2728) !   Let TG be colatitude of G.
  2729) !   Let ANG be longitude angle from GM to G.
  2730) !   Let TE be colatitude of E.
  2731) !   Let TP be colatitude of GM.
  2732) !   Let A be longitude angle from G to E.
  2733) !   Let APANG = A + ANG
  2734) !   Let PA be geomagnetic longitude, i.e., Pi minus angle measured 
  2735) !     counterclockwise from arc GM-E to arc GM-GP.
  2736) !   Let TF be arc length between GM and E.
  2737) !   Then, using notation C=cos, S=sin, COT=cot, spherical-trigonometry formulas 
  2738) !     for the functions of the angles are as shown below.  Note: STFCPA,
  2739) !     STFSPA are sin(TF) times cos(PA), sin(PA), respectively.
  2740) !-----------------------------------------------------------------------
  2741) !
  2742)       use shr_kind_mod, only: r8 => shr_kind_r8
  2743)       implicit none
  2744) !
  2745) !-------------------------------Commons---------------------------------
  2746) !
  2747)       real(r8) colat, elon, vp, ctp, stp
  2748)       COMMON/DIPOLE/COLAT,ELON,VP,CTP,STP
  2749) !
  2750) !------------------------------Arguments--------------------------------
  2751) !
  2752)       real(r8) gdlat, gdlon, alt, bnorth, beast, bdown, a, alon
  2753) !
  2754) !-----------------------------Parameters------------------------------
  2755) !
  2756)       real(r8) rtod, dtor, re, req
  2757)       PARAMETER (RTOD=5.72957795130823E1_r8,DTOR=1.745329251994330E-2_r8)
  2758)       PARAMETER (RE=6371.2_r8,REQ=6378.160_r8)
  2759) !
  2760) !---------------------------Local variables-----------------------------
  2761) !
  2762)       real(r8) bhor, ca, capang, sa, cte, ste, r, ha, stfspa, sapang
  2763)       real(r8) stfcpa, sb, cb, ctd, cottd, std, sang, cang, ang, ctg
  2764)       real(r8) stg
  2765) !
  2766) !-----------------------------------------------------------------------
  2767) !
  2768)       BHOR = SQRT(BNORTH*BNORTH + BEAST*BEAST)
  2769)       IF (BHOR.EQ.0._r8) THEN
  2770) 	ALON = 0._r8
  2771) 	A = 1.E34_r8
  2772) 	RETURN
  2773)       ENDIF
  2774)       COTTD = BDOWN*.5_r8/BHOR
  2775)       STD = 1._r8/SQRT(1._r8+COTTD*COTTD)
  2776)       CTD = COTTD*STD
  2777)       SB = -BEAST/BHOR
  2778)       CB = -BNORTH/BHOR
  2779)       CTG = SIN(GDLAT*DTOR) 
  2780)       STG = COS(GDLAT*DTOR)
  2781)       ANG = (GDLON-ELON)*DTOR 
  2782)       SANG = SIN(ANG)
  2783)       CANG = COS(ANG)
  2784)       CTE = CTG*STD + STG*CTD*CB
  2785)       STE = SQRT(1._r8 - CTE*CTE)
  2786)       SA = SB*CTD/STE
  2787)       CA = (STD*STG - CTD*CTG*CB)/STE
  2788)       CAPANG = CA*CANG - SA*SANG
  2789)       SAPANG = CA*SANG + SA*CANG
  2790)       STFCPA = STE*CTP*CAPANG - CTE*STP
  2791)       STFSPA = SAPANG*STE
  2792)       ALON = ATAN2(STFSPA,STFCPA)*RTOD
  2793)       R = ALT + RE
  2794)       HA = ALT + R*COTTD*COTTD
  2795)       A = 1._r8 + HA/REQ
  2796)       RETURN
  2797)       end subroutine dipapx

  3631)                  RE=6371.2_r8, REQ=6378.160_r8)
  3632) !
  3633) !---------------------------Local variables-----------------------------
  3634) !
  3635)       real(r8) gpl, stp, ctp
  3636) !
  3637) !-----------------------------------------------------------------------
  3638) !
  3639) !          Compute geographic colatitude and longitude of the north pole of
  3640) !          earth centered dipole
  3641)       GPL = SQRT( G(2  )**2+ G(3  )**2+ G(4  )**2)
  3642)       CTP = G(2  )/GPL
  3643)       STP = SQRT(1._r8 - CTP*CTP)
  3644)       COLAT = (ACOS(CTP))*RTOD
  3645)       ELON = ATAN2( G(4  ), G(3  ))*RTOD
  3646)  
  3647) !          Compute magnitude of magnetic potential at pole, radius Re.
  3648)       VP = .2_r8*GPL*RE
  3649) !          .2 = 2*(10**-4 T/gauss)*(1000 m/km) (2 comes through F0 in COFRM).
  3650)  
  3651)       RETURN
  3652)       END SUBROUTINE DYPOL

  3656)       SUBROUTINE FELDG (IENTY,GLAT,GLON,ALT, BNRTH,BEAST,BDOWN,BABS)
  3657) !
  3658) !-----------------------------------------------------------------------
  3659) !          Compute the DGRF/IGRF field components at the point GLAT,GLON,ALT.
  3660) !          COFRM must be called to establish coefficients for correct date
  3661) !          prior to calling FELDG.
  3662) !
  3663) !          IENTY is an input flag controlling the meaning and direction of the
  3664) !                remaining formal arguments:
  3665) !          IENTY = 1
  3666) !            INPUTS:
  3667) !              GLAT = Latitude of point (deg)
  3668) !              GLON = Longitude (east=+) of point (deg)
  3669) !              ALT  = Ht of point (km)
  3670) !            RETURNS:
  3671) !              BNRTH  north component of field vector (Gauss)
  3672) !              BEAST  east component of field vector  (Gauss)
  3673) !              BDOWN  downward component of field vector (Gauss)
  3674) !              BABS   magnitude of field vector (Gauss)
  3675) !
  3676) !          IENTY = 2
  3677) !            INPUTS:
  3678) !              GLAT = X coordinate (in units of earth radii 6371.2 km )
  3679) !              GLON = Y coordinate (in units of earth radii 6371.2 km )
  3680) !              ALT  = Z coordinate (in units of earth radii 6371.2 km )
  3681) !            RETURNS:
  3682) !              BNRTH = X component of field vector (Gauss)
  3683) !              BEAST = Y component of field vector (Gauss)
  3684) !              BDOWN = Z component of field vector (Gauss)
  3685) !              BABS  = Magnitude of field vector (Gauss)
  3686) !          IENTY = 3
  3687) !            INPUTS:
  3688) !              GLAT = X coordinate (in units of earth radii 6371.2 km )
  3689) !              GLON = Y coordinate (in units of earth radii 6371.2 km )
  3690) !              ALT  = Z coordinate (in units of earth radii 6371.2 km )
  3691) !            RETURNS:
  3692) !              BNRTH = Dummy variable
  3693) !              BEAST = Dummy variable
  3694) !              BDOWN = Dummy variable
  3695) !              BABS  = Magnetic potential (T.m)
  3696) !
  3697) !          INPUT from COFRM through COMMON /MAGCOF/ NMAX,GB(144),GV(144),ICHG
  3698) !            NMAX = Maximum order of spherical harmonic coefficients used
  3699) !            GB   = Coefficients for magnetic field calculation
  3700) !            GV   = Coefficients for magnetic potential calculation
  3701) !            ICHG = Flag indicating when GB,GV have been changed
  3702) !
  3703) !          HISTORY:
  3704) !          COFRM and FELDG originated 15 Apr 83 by Vincent B. Wickwar
  3705) !          (formerly at SRI. Int., currently at Utah State).
  3706) !
  3707) !          May 94 (A.D. Richmond): Added magnetic potential calculation
  3708) !
  3709) !          Oct 95 (Barnes): Added ICHG
  3710) !-----------------------------------------------------------------------
  3711) !
  3712)       use shr_kind_mod, only: r8 => shr_kind_r8
  3713)       implicit none
  3714) !
  3715) !-------------------------------Commons---------------------------------
  3716) !
  3717) !     COMMON /MAGCOF/ NMAX,GB(144),GV(144),ICHG
  3718) 
  3719)       integer nmax, ichg
  3720)       COMMON /MAGCOF_int/ NMAX,ICHG
  3721) 
  3722)       real(r8) gb, gv
  3723)       COMMON /MAGCOF_real/ GB(144),GV(144)
  3724) !
  3725) !------------------------------Arguments--------------------------------
  3726) !
  3727)       integer ienty
  3728) 
  3729)       real(r8) glat, glon, alt, bnrth, beast, bdown, babs
  3730) !
  3731) !-----------------------------Parameters------------------------------
  3732) !
  3733)       real(r8) rtod, dtor, re, req
  3734)       PARAMETER (RTOD=57.2957795130823_r8, DTOR=0.01745329251994330_r8, &
  3735)                  RE=6371.2_r8, REQ=6378.160_r8)
  3736) !
  3737) !---------------------------Local variables-----------------------------
  3738) !
  3739)       integer m, ih, k, il, ihm, ilm, is, imax, last, mk, i, ihmax
  3740) 
  3741)       real(r8) y, x, z, f, byyy, bxxx, brho, bzzz, t, s, cp, rlon, xxx
  3742)       real(r8) sp, rlat, st, ct, zzz, yyy, rq
  3743)       real(r8) XI(3),H(144),G(144)
  3744)       SAVE G
  3745) 
  3746)       integer ientyp
  3747)       SAVE IENTYP
  3748)       DATA IENTYP/-10000/
  3749) !
  3750) !-----------------------------------------------------------------------
  3751) !
  3752)       IF (IENTY .EQ. 1) THEN
  3753)         IS=1
  3754)         RLAT = GLAT*DTOR
  3755)         CT   = SIN(RLAT)
  3756)         ST   = COS(RLAT)
  3757)         RLON = GLON*DTOR
  3758)         CP   = COS(RLON)
  3759)         SP   = SIN(RLON)
  3760)         CALL GD2CART (GLAT,GLON,ALT,XXX,YYY,ZZZ)
  3761)         XXX = XXX/RE
  3762)         YYY = YYY/RE
  3763)         ZZZ = ZZZ/RE
  3764)       ELSE
  3765)         IS   = 2
  3766)         XXX  = GLAT
  3767)         YYY  = GLON
  3768)         ZZZ  = ALT
  3769)       ENDIF
  3770)       RQ    = 1._r8/(XXX**2+YYY**2+ZZZ**2)
  3771)       XI(1) = XXX*RQ
  3772)       XI(2) = YYY*RQ
  3773)       XI(3) = ZZZ*RQ
  3774)       IHMAX = NMAX*NMAX+1
  3775)       LAST  = IHMAX+NMAX+NMAX
  3776)       IMAX  = NMAX+NMAX-1
  3777)  
  3778)       IF (IENTY .NE. IENTYP .OR. ICHG .EQ. 1) THEN
  3779)         IENTYP = IENTY
  3780) 	ICHG = 0
  3781)         IF (IENTY .NE. 3) THEN
  3782) 	  DO 10 I=1,LAST
  3783)    10     G(I) = GB(I)
  3784)         ELSE
  3785) 	  DO 20 I=1,LAST
  3786)    20     G(I) = GV(I)
  3787)         ENDIF
  3788)       ENDIF
  3789)  
  3790)       DO 30 I=IHMAX,LAST
  3791)    30 H(I) = G(I)
  3792) 
  3793)       MK = 3
  3794)       IF (IMAX .EQ. 1) MK=1
  3795) 
  3796)       DO 100 K=1,MK,2
  3797)       I  = IMAX
  3798)       IH = IHMAX
  3799) 
  3800)    60 IL = IH-I
  3801)       F = 2._r8/real(I-K+2,r8)
  3802)       X = XI(1)*F
  3803)       Y = XI(2)*F
  3804)       Z = XI(3)*(F+F)
  3805) 
  3806)       I = I-2
  3807)       IF (I .LT. 1) GO TO 90
  3808)       IF (I .EQ. 1) GO TO 80
  3809) 
  3810)       DO 70 M=3,I,2
  3811)       IHM = IH+M
  3812)       ILM = IL+M
  3813)       H(ILM+1) = G(ILM+1)+ Z*H(IHM+1) + X*(H(IHM+3)-H(IHM-1)) &
  3814)                                               -Y*(H(IHM+2)+H(IHM-2))
  3815)    70 H(ILM)   = G(ILM)  + Z*H(IHM)   + X*(H(IHM+2)-H(IHM-2)) &
  3816)                                               +Y*(H(IHM+3)+H(IHM-1))
  3817) 
  3818)    80 H(IL+2) = G(IL+2) + Z*H(IH+2) + X*H(IH+4) - Y*(H(IH+3)+H(IH))
  3819)       H(IL+1) = G(IL+1) + Z*H(IH+1) + Y*H(IH+4) + X*(H(IH+3)-H(IH))
  3820) 
  3821)    90 H(IL)   = G(IL)   + Z*H(IH)   + 2._r8*(X*H(IH+1)+Y*H(IH+2))
  3822)       IH = IL
  3823)       IF (I .GE. K) GO TO 60
  3824)   100 CONTINUE
  3825)  
  3826)       S = .5_r8*H(1)+2._r8*(H(2)*XI(3)+H(3)*XI(1)+H(4)*XI(2))
  3827)       T = (RQ+RQ)*SQRT(RQ)
  3828)       BXXX = T*(H(3)-S*XXX)
  3829)       BYYY = T*(H(4)-S*YYY)
  3830)       BZZZ = T*(H(2)-S*ZZZ)
  3831)       BABS = SQRT(BXXX**2+BYYY**2+BZZZ**2)
  3832)       IF (IS .EQ. 1) THEN            ! (convert back to geodetic)
  3833)         BEAST = BYYY*CP-BXXX*SP
  3834)         BRHO  = BYYY*SP+BXXX*CP
  3835)         BNRTH =  BZZZ*ST-BRHO*CT
  3836)         BDOWN = -BZZZ*CT-BRHO*ST
  3837)       ELSEIF (IS .EQ. 2) THEN        ! (leave in earth centered cartesian)
  3838)         BNRTH = BXXX
  3839)         BEAST = BYYY
  3840)         BDOWN = BZZZ
  3841)       ENDIF
  3842)  
  3843) !          Magnetic potential computation makes use of the fact that the
  3844) !          calculation of V is identical to that for r*Br, if coefficients
  3845) !          in the latter calculation have been divided by (n+1) (coefficients
  3846) !          GV).  Factor .1 converts km to m and gauss to tesla.
  3847)       IF (IENTY.EQ.3) BABS = (BXXX*XXX + BYYY*YYY + BZZZ*ZZZ)*RE*.1_r8
  3848)  
  3849)       RETURN
  3850)       END SUBROUTINE FELDG

  3888)       SUBROUTINE CONVRT (I,GDLAT,ALT,X1,X2)
  3889) !
  3890) !-----------------------------------------------------------------------
  3891) !          Convert space point from geodetic to geocentric or vice versa.
  3892) !
  3893) !          I is an input flag controlling the meaning and direction of the
  3894) !            remaining formal arguments:
  3895) !
  3896) !          I = 1  (convert from geodetic to cylindrical)
  3897) !            INPUTS:
  3898) !              GDLAT = Geodetic latitude (deg)
  3899) !              ALT   = Altitude above reference ellipsoid (km)
  3900) !            RETURNS:
  3901) !              X1    = Distance from Earth's rotation axis (km)
  3902) !              X2    = Distance above (north of) Earth's equatorial plane (km)
  3903) !
  3904) !          I = 2  (convert from geodetic to geocentric spherical)
  3905) !            INPUTS:
  3906) !              GDLAT = Geodetic latitude (deg)
  3907) !              ALT   = Altitude above reference ellipsoid (km)
  3908) !            RETURNS:
  3909) !              X1    = Geocentric latitude (deg)
  3910) !              X2    = Geocentric distance (km)
  3911) !
  3912) !          I = 3  (convert from cylindrical to geodetic)
  3913) !            INPUTS:
  3914) !              X1    = Distance from Earth's rotation axis (km)
  3915) !              X2    = Distance from Earth's equatorial plane (km)
  3916) !            RETURNS:
  3917) !              GDLAT = Geodetic latitude (deg)
  3918) !              ALT   = Altitude above reference ellipsoid (km)
  3919) !
  3920) !          I = 4  (convert from geocentric spherical to geodetic)
  3921) !            INPUTS:
  3922) !              X1    = Geocentric latitude (deg)
  3923) !              X2    = Geocentric distance (km)
  3924) !            RETURNS:
  3925) !              GDLAT = Geodetic latitude (deg)
  3926) !              ALT   = Altitude above reference ellipsoid (km)
  3927) !
  3928) !
  3929) !          HISTORY:
  3930) !          940503 (A. D. Richmond):  Based on a routine originally written
  3931) !          by V. B. Wickwar.
  3932) !
  3933) !          REFERENCE:  ASTRON. J. VOL. 66, p. 15-16, 1961.
  3934) !-----------------------------------------------------------------------
  3935) !
  3936)       use shr_kind_mod, only: r8 => shr_kind_r8
  3937)       implicit none
  3938) !
  3939) !------------------------------Arguments--------------------------------
  3940) !
  3941)       integer i
  3942) 
  3943)       real(r8) gdlat, alt, x1, x2
  3944) !
  3945) !-----------------------------Parameters------------------------------
  3946) !
  3947)       real(r8) rtod, dtor, re, req, fltnvrs, e2, e4, e6, e8, ome2req
  3948)       real(r8) a21, a22, a23, a41, a42, a43, a44, a61, a62, a63, a81, a82
  3949)       real(r8) a83, a84
  3950)       PARAMETER (RTOD=57.2957795130823_r8, DTOR=0.01745329251994330_r8 , &
  3951)         RE=6371.2_r8 , REQ=6378.160_r8 , FLTNVRS=298.25_r8 , &
  3952)         E2=(2._r8-1._r8/FLTNVRS)/FLTNVRS , E4=E2*E2 , E6=E4*E2 , E8=E4*E4 , &
  3953)         OME2REQ = (1._r8-E2)*REQ , &
  3954)            A21 =     (512._r8*E2 + 128._r8*E4 + 60._r8*E6 + 35._r8*E8)/1024._r8 , &
  3955)            A22 =     (                        E6 +     E8)/  32._r8 , &
  3956)            A23 = -3._r8*(                     4._r8*E6 +  3._r8*E8)/ 256._r8 , &
  3957)            A41 =    -(           64._r8*E4 + 48._r8*E6 + 35._r8*E8)/1024._r8 , &
  3958)            A42 =     (            4._r8*E4 +  2._r8*E6 +     E8)/  16._r8 , &
  3959)            A43 =                                   15._r8*E8 / 256._r8 , &
  3960)            A44 =                                      -E8 /  16._r8 , &
  3961)            A61 =  3._r8*(                     4._r8*E6 +  5._r8*E8)/1024._r8 , &
  3962)            A62 = -3._r8*(                        E6 +     E8)/  32._r8 , &
  3963)            A63 = 35._r8*(                     4._r8*E6 +  3._r8*E8)/ 768._r8 , &
  3964)            A81 =                                   -5._r8*E8 /2048._r8 , &
  3965)            A82 =                                   64._r8*E8 /2048._r8 , &
  3966)            A83 =                                 -252._r8*E8 /2048._r8 , &
  3967)            A84 =                                  320._r8*E8 /2048._r8 )
  3968) !          E2 = Square of eccentricity
  3969) !
  3970) !---------------------------Local variables-----------------------------
  3971) !
  3972)       real(r8) a2, a4, a6, a8, ccl, c2cl, c4cl, dltcl, gclat, gclatm, &
  3973)            coslat, sinlat, s2cl, s6cl, sgl, s4cl, s8cl, ri, &
  3974)            rkm, scl, d, z, rho
  3975) !
  3976) !-----------------------------------------------------------------------
  3977) !
  3978)       IF (I .GE. 3) GO TO 300
  3979)  
  3980) !          Geodetic to geocentric
  3981)  
  3982) !          Compute RHO,Z
  3983)       SINLAT = SIN(GDLAT*DTOR)
  3984)       COSLAT = SQRT(1._r8-SINLAT*SINLAT)
  3985)       D      = SQRT(1._r8-E2*SINLAT*SINLAT)
  3986)       Z      = (ALT+OME2REQ/D)*SINLAT
  3987)       RHO    = (ALT+REQ/D)*COSLAT
  3988)       X1 = RHO
  3989)       X2 = Z
  3990)       IF (I .EQ. 1) RETURN
  3991)  
  3992) !          Compute GCLAT,RKM
  3993)       RKM   = SQRT(Z*Z + RHO*RHO)
  3994)       GCLAT = RTOD*ATAN2(Z,RHO)
  3995)       X1 = GCLAT
  3996)       X2 = RKM
  3997)       RETURN
  3998)  
  3999) !          Geocentric to geodetic
  4000)   300 IF (I .EQ. 3) THEN
  4001)          RHO = X1
  4002)          Z = X2
  4003)          RKM = SQRT(Z*Z+RHO*RHO)
  4004)          SCL = Z/RKM
  4005)          GCLAT = ASIN(SCL)*RTOD
  4006)       ELSEIF (I .EQ. 4) THEN
  4007)          GCLAT = X1
  4008)          RKM = X2
  4009)          SCL = SIN(GCLAT*DTOR)
  4010)       ELSE
  4011)          RETURN
  4012)       ENDIF
  4013)  
  4014)       RI = REQ/RKM
  4015)       A2 = RI*(A21+RI*(A22+RI* A23))
  4016)       A4 = RI*(A41+RI*(A42+RI*(A43+RI*A44)))
  4017)       A6 = RI*(A61+RI*(A62+RI* A63))
  4018)       A8 = RI*(A81+RI*(A82+RI*(A83+RI*A84)))
  4019)       CCL = SQRT(1._r8-SCL*SCL)
  4020)       S2CL = 2._r8*SCL*CCL
  4021)       C2CL = 2._r8*CCL*CCL-1._r8
  4022)       S4CL = 2._r8*S2CL*C2CL
  4023)       C4CL = 2._r8*C2CL*C2CL-1._r8
  4024)       S8CL = 2._r8*S4CL*C4CL
  4025)       S6CL = S2CL*C4CL+C2CL*S4CL
  4026)       DLTCL = S2CL*A2+S4CL*A4+S6CL*A6+S8CL*A8
  4027)       GDLAT = DLTCL*RTOD+GCLAT
  4028)       SGL = SIN(GDLAT*DTOR)
  4029)       ALT = RKM*COS(DLTCL)-REQ*SQRT(1._r8-E2*SGL*SGL)
  4030)       RETURN
  4031)       END SUBROUTINE CONVRT

  4066)       SUBROUTINE MAGLOCTM (ALON,SBSLLAT,SBSLLON,CLATP,POLON,MLT)
  4067) !
  4068) !-----------------------------------------------------------------------
  4069) !  Computes magnetic local time from magnetic longitude, subsolar coordinates,
  4070) !   and geomagnetic pole coordinates.
  4071) !  950302 A. D. Richmond, NCAR
  4072) !  Algorithm:  MLT is calculated from the difference of the apex longitude,
  4073) !   alon, and the geomagnetic dipole longitude of the subsolar point.
  4074) !
  4075) !   Inputs:
  4076) !    ALON    = apex magnetic longitude of the point (deg)
  4077) !    SBSLLAT = geographic latitude of subsolar point (degrees)
  4078) !    SBSLLON = geographic longitude of subsolar point (degrees)
  4079) !    CLATP   = Geocentric colatitude of geomagnetic dipole north pole (deg)
  4080) !    POLON   = East longitude of geomagnetic dipole north pole (deg)
  4081) !
  4082) !   Output:
  4083) !    mlt (real) = magnetic local time for the apex longitude alon (hours)
  4084) !
  4085) !-----------------------------------------------------------------------
  4086) !
  4087)         use shr_kind_mod, only: r8 => shr_kind_r8
  4088)         implicit none
  4089) !
  4090) !------------------------------Arguments--------------------------------
  4091) !
  4092) 	REAL(r8) alon, sbsllat, sbsllon, clatp, polon, MLT
  4093) !
  4094) !---------------------------Local variables-----------------------------
  4095) !
  4096)         real(r8) smlon
  4097) !
  4098) !-----------------------------------------------------------------------
  4099) !
  4100) 	CALL SOLGMLON (SBSLLAT,SBSLLON,CLATP,POLON,SMLON)
  4101) 	MLT = (ALON - SMLON)/15.0_r8 + 12.0_r8
  4102) 	IF (MLT .GE. 24.0_r8) MLT = MLT - 24.0_r8
  4103) 	IF (MLT .LT.   0._r8) MLT = MLT + 24.0_r8
  4104) 	RETURN
  4105)         END SUBROUTINE MAGLOCTM

  4109)       SUBROUTINE SOLGMLON (XLAT,XLON,COLAT,ELON,MLON)
  4110) !

  4112) ! Computes geomagnetic longitude of the point with geocentric spherical
  4113) !  latitude and longitude of XLAT and XLON, respectively.
  4114) ! 940719 A. D. Richmond, NCAR
  4115) ! Inputs:
  4116) !   XLAT  = geocentric spherical latitude (deg)
  4117) !   XLON  = geocentric spherical longitude (deg)
  4118) !   COLAT = Geocentric colatitude of geomagnetic dipole north pole (deg)
  4119) !   ELON  = East longitude of geomagnetic dipole north pole (deg)
  4120) ! Output:
  4121) !   MLON  = Geomagnetic dipole longitude of the point (deg, -180. to 180.)
  4122) !-----------------------------------------------------------------------
  4123) !
  4124)       use shr_kind_mod, only: r8 => shr_kind_r8
  4125)       implicit none
  4126) !
  4127) !------------------------------Arguments--------------------------------
  4128) !
  4129)       REAL(r8) xlat, xlon, colat, elon, MLON
  4130) !
  4131) !-----------------------------Parameters------------------------------
  4132) !
  4133)       real(r8) rtod, dtor
  4134)       PARAMETER (RTOD=5.72957795130823E1_r8,DTOR=1.745329251994330E-2_r8)

  4142) ! Algorithm:
  4143) !   Use spherical coordinates.
  4144) !   Let GP be geographic pole.
  4145) !   Let GM be geomagnetic pole (colatitude COLAT, east longitude ELON).
  4146) !   Let XLON be longitude of point P.
  4147) !   Let TE be colatitude of point P.
  4148) !   Let ANG be longitude angle from GM to P.
  4149) !   Let TP be colatitude of GM.
  4150) !   Let TF be arc length between GM and P.
  4151) !   Let PA = MLON be geomagnetic longitude, i.e., Pi minus angle measured
  4152) !     counterclockwise from arc GM-P to arc GM-GP.
  4153) !   Then, using notation C=cos, S=sin, spherical-trigonometry formulas
  4154) !     for the functions of the angles are as shown below.  Note: STFCPA,
  4155) !     STFSPA are sin(TF) times cos(PA), sin(PA), respectively.
  4156) 
  4157)       CTP = COS(COLAT*DTOR)
  4158)       STP = SQRT(1._r8 - CTP*CTP)
  4159)       ANG = (XLON-ELON)*DTOR
  4160)       CANG = COS(ANG)

  4172)       SUBROUTINE SUBSOL (IYR,IDAY,IHR,IMN,SEC,SBSLLAT,SBSLLON)
  4173) !
  4174) !-----------------------------------------------------------------------
  4175) !          Find subsolar geographic latitude and longitude given the
  4176) !          date and time (Universal Time).
  4177) !
  4178) !          This is based on formulas in Astronomical Almanac for the
  4179) !          year 1996, p.  C24. (U.S.  Government Printing Office,
  4180) !          1994).  According to the Almanac, results are good to at
  4181) !          least 0.01 degree latitude and 0.025 degree longitude
  4182) !          between years 1950 and 2050.  Accuracy for other years has
  4183) !          not been tested although the algorithm has been designed to
  4184) !          accept input dates from 1601 to 2100.  Every day is assumed
  4185) !          to have exactly 86400 seconds; thus leap seconds that
  4186) !          sometimes occur on June 30 and December 31 are ignored:
  4187) !          their effect is below the accuracy threshold of the algorithm.
  4188) !
  4189) !          961026 A. D. Richmond, NCAR
  4190) !
  4191) !          INPUTS:
  4192) !            IYR  = Year (e.g., 1994). IYR must be in the range: 1601 to 2100.
  4193) !            IDAY = Day number of year (e.g., IDAY = 32 for Feb 1)
  4194) !            IHR  = Hour of day    (e.g., 13 for 13:49)
  4195) !            IMN  = Minute of hour (e.g., 49 for 13:49)
  4196) !            SEC  = Second and fraction after the hour/minute.
  4197) !          Note:  While IYR is bounds tested, there is no constraint
  4198) !                 placed on values: IDAY,IHR,IMN,SEC; e.g., IHR=25 is
  4199) !                 properly interpreted.
  4200) !          RETURNS:
  4201) !            SBSLLAT = geographic latitude of subsolar point (degrees)
  4202) !            SBSLLON = geographic longitude of subsolar point (degrees,
  4203) !                      between -180 and +180)
  4204) !-----------------------------------------------------------------------
  4205) !
  4206)       use shr_kind_mod, only: r8 => shr_kind_r8
  4207)       implicit none
  4208) !
  4209) !------------------------------Arguments--------------------------------
  4210) !
  4211)       INTEGER IYR,IDAY,IHR,IMN
  4212) 
  4213)       REAL(r8) SEC,SBSLLAT,SBSLLON
  4214) !
  4215) !-----------------------------Parameters------------------------------
  4216) !
  4217)       integer msgun
  4218)       PARAMETER (MSGUN=6)
  4219) 
  4220)       real(r8) d2r, r2d
  4221)       PARAMETER (D2R=0.0174532925199432957692369076847_r8 , &
  4222)                  R2D=57.2957795130823208767981548147_r8)
  4223) !
  4224) !---------------------------Local variables-----------------------------
  4225) !
  4226)       INTEGER YR,NLEAP,NCENT,NROT
  4227) 
  4228)       REAL(r8) L0,G0,DF,LF,GF,L,G,LAMBDA,EPSILON,N &
  4229)           ,GRAD,LAMRAD,SINLAM,EPSRAD,DELTA,UT,ETDEG
  4230)       real(r8) aptime, alpha
  4231) !
  4232) !-----------------------------------------------------------------------
  4233) !
  4234) ! Number of years from 2000 to IYR (negative if IYR < 2000):
  4235)       YR = IYR - 2000
  4236) !
  4237) ! NLEAP (final) = number of leap days from (2000 January 1) to (IYR January 1)
  4238) !                 (negative if IYR is before 1997)
  4239)       NLEAP = (IYR-1601)/4
  4240)       NLEAP = NLEAP - 99
  4241)       IF (IYR.LE.1900) THEN
  4242) 	IF (IYR.LE.1600) THEN
  4243) 	 WRITE(MSGUN,*) 'SUBSOLR INVALID BEFORE 1601: INPUT YEAR = ',IYR       
  4244) 	 STOP
  4245) 	ENDIF
  4246) 	NCENT = (IYR-1601)/100
  4247) 	NCENT = 3 - NCENT 
  4248) 	NLEAP = NLEAP + NCENT
  4249)       ENDIF
  4250)       IF (IYR.GE.2101) THEN
  4251) 	WRITE(MSGUN,*) 'SUBSOLR INVALID AFTER 2100:  INPUT YEAR = ',IYR
  4252) 	STOP
  4253)       ENDIF
  4254) !
  4255) ! L0 = Mean longitude of Sun at 12 UT on January 1 of IYR:
  4256) !     L0 = 280.461 + .9856474*(365*(YR-NLEAP) + 366*NLEAP) 
  4257) !	   - (ARBITRARY INTEGER)*360.
  4258) !        = 280.461 + .9856474*(365*(YR-4*NLEAP) + (366+365*3)*NLEAP) 
  4259) !	   - (ARBITRARY INTEGER)*360.
  4260) !        = (280.461 - 360.) + (.9856474*365 - 360.)*(YR-4*NLEAP) 
  4261) !	   + (.9856474*(366+365*3) - 4*360.)*NLEAP,
  4262) !  where ARBITRARY INTEGER = YR+1.  This gives:
  4263)       L0 = -79.549_r8 + (-.238699_r8*(YR-4*NLEAP) + 3.08514E-2_r8*NLEAP)
  4264) !
  4265) ! G0 = Mean anomaly at 12 UT on January 1 of IYR:
  4266) !     G0 = 357.528 + .9856003*(365*(YR-NLEAP) + 366*NLEAP) 
  4267) !	   - (ARBITRARY INTEGER)*360.
  4268) !        = 357.528 + .9856003*(365*(YR-4*NLEAP) + (366+365*3)*NLEAP) 
  4269) !	   - (ARBITRARY INTEGER)*360.
  4270) !        = (357.528 - 360.) + (.9856003*365 - 360.)*(YR-4*NLEAP) 
  4271) !	   + (.9856003*(366+365*3) - 4*360.)*NLEAP,
  4272) !  where ARBITRARY INTEGER = YR+1.  This gives:
  4273)       G0 = -2.472_r8 + (-.2558905_r8*(YR-4*NLEAP) - 3.79617E-2_r8*NLEAP)
  4274) !
  4275) ! Universal time in seconds:
  4276)       UT = real(IHR*3600 + IMN*60,r8) + SEC
  4277) !
  4278) ! Days (including fraction) since 12 UT on January 1 of IYR:
  4279)       DF = (UT/86400._r8 - 1.5_r8) + IDAY
  4280) !
  4281) ! Addition to Mean longitude of Sun since January 1 of IYR:
  4282)       LF = .9856474_r8*DF
  4283) !
  4284) ! Addition to Mean anomaly since January 1 of IYR:
  4285)       GF = .9856003_r8*DF
  4286) !
  4287) ! Mean longitude of Sun:
  4288)       L = L0 + LF
  4289) !
  4290) ! Mean anomaly:
  4291)       G = G0 + GF
  4292)       GRAD = G*D2R
  4293) !
  4294) ! Ecliptic longitude:
  4295)       LAMBDA = L + 1.915_r8*SIN(GRAD) + .020_r8*SIN(2._r8*GRAD)
  4296)       LAMRAD = LAMBDA*D2R
  4297)       SINLAM = SIN(LAMRAD)
  4298) !
  4299) ! Days (including fraction) since 12 UT on January 1 of 2000:
  4300)       N = DF + real(365*YR + NLEAP,r8)
  4301) !
  4302) ! Obliquity of ecliptic:
  4303)       EPSILON = 23.439_r8 - 4.E-7_r8*N
  4304)       EPSRAD = EPSILON*D2R
  4305) !
  4306) ! Right ascension:
  4307)       ALPHA = ATAN2(COS(EPSRAD)*SINLAM,COS(LAMRAD))*R2D
  4308) !
  4309) ! Declination:
  4310)       DELTA = ASIN(SIN(EPSRAD)*SINLAM)*R2D
  4311) !
  4312) ! Subsolar latitude:
  4313)       SBSLLAT = DELTA
  4314) !
  4315) ! Equation of time (degrees):
  4316)       ETDEG = L - ALPHA
  4317)       NROT = NINT(ETDEG/360._r8)
  4318)       ETDEG = ETDEG - real(360*NROT,r8)
  4319) !
  4320) ! Apparent time (degrees):
  4321)       APTIME = UT/240._r8 + ETDEG
  4322) !          Earth rotates one degree every 240 s.
  4323) !
  4324) ! Subsolar longitude:
  4325)       SBSLLON = 180._r8 - APTIME
  4326)       NROT = NINT(SBSLLON/360._r8)
  4327)       SBSLLON = SBSLLON - real(360*NROT,r8)
  4328) !
  4329)       RETURN
  4330)       END SUBROUTINE SUBSOL
