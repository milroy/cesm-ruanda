   170)       subroutine schu( o2col, secchi, dto2, xscho2 )
   171) !-----------------------------------------------------------------------------
   172) !   purpose:
   173) !   calculate the equivalent absorption cross section of o2 in the sr bands.
   174) !   the algorithm is based on:  g.kockarts, penetration of solar radiation
   175) !   in the schumann-runge bands of molecular oxygen:  a robust approximation,
   176) !   annales geophysicae, v12, n12, pp. 1207ff, dec 1994.  calculation is
   177) !   done on the wavelength grid used by kockarts (1994).  final values do
   178) !   include effects from the herzberg continuum.
   179) !-----------------------------------------------------------------------------
   180) !   parameters:
   181) !   nz      - integer, number of specified altitude levels in the working (i)
   182) !             grid
   183) !   o2col   - real, slant overhead o2 column (molec/cc) at each specified (i)
   184) !             altitude
   185) !   dto2    - real, optical depth due to o2 absorption at each specified  (o)
   186) !             vertical layer at each specified wavelength
   187) !   xscho2  - real, molecular absorption cross section in sr bands at     (o)
   188) !             each specified wavelength.  includes herzberg continuum
   189) !-----------------------------------------------------------------------------
   190) 
   191) !!$      use mo_grid, only : plev, plevp
   192)       use ppgrid, only : pver,pverp
   193) 
   194)       implicit none
   195) 
   196) !-----------------------------------------------------------------------------
   197) !	... dummy arguments
   198) !-----------------------------------------------------------------------------
   199)       real(r8), intent(inout) :: dto2(pver,ngast-1)
   200)       real(r8), intent(inout) :: xscho2(pverp,ngast-1)
   201)       real(r8), intent(in)    :: o2col(pverp)
   202)       real(r8), intent(in)    :: secchi(pverp)
   203) 
   204) !-----------------------------------------------------------------------------
   205) !	... local variables
   206) !-----------------------------------------------------------------------------
   207)       integer :: i, iw, j, j1, jp1, k, ki, kp1
   208)       integer :: index(pverp)
   209)       integer :: minind(1), maxind(1)
   210)       real(r8)    :: a0, a1, b0, b1
   211)       real(r8), dimension(6)        :: ac, bc
   212)       real(r8), dimension(pverp,6)  :: aa, bb
   213)       real(r8), dimension(pverp)    :: rjm, rjo2
   214)       real(r8), dimension(pverp)    :: rjmi, rjo2i, lo2col, dels
   215) 
   216) 
   217) !-----------------------------------------------------------------------------
   218) ! 	... initialize r(m)
   219) !-----------------------------------------------------------------------------
   220)       rjm(1:pverp)  = 0._r8
   221)       rjo2(1:pverp) = 0._r8
   222) 
   223) !-----------------------------------------------------------------------------
   224) ! 	... initialize the table interpolation
   225) !-----------------------------------------------------------------------------
   226)       where( o2col(:) /= 0 )
   227)          lo2col(:) = log10( o2col(:) )
   228)       endwhere
   229)       do ki = 1,pverp
   230) 	 if( o2col(ki) /= 0._r8 ) then
   231) 	    if( lo2col(ki) <= o2_table(1) ) then
   232) 	       dels(ki)  = 0._r8
   233) 	       index(ki) = 1
   234) 	    else if( lo2col(ki) >= o2_table(tdim) ) then
   235) 	       dels(ki)  = 1._r8
   236) 	       index(ki) = tdim-1
   237) 	    else
   238)                do k = 2,tdim
   239) 	          if( lo2col(ki) <= o2_table(k) ) then
   240) 	             dels(ki)  = t_fac*(lo2col(ki) - o2_table(k-1))
   241) 	             index(ki) = k-1
   242) 		     exit
   243) 		  end if
   244)                end do
   245) 	    end if
   246) 	 else
   247) 	    index(ki) = 0
   248) 	    dels(ki)  = 0._r8
   249) 	 end if
   250)       end do
   251) 
   252) !-----------------------------------------------------------------------------
   253) ! 	... calculate sum of exponentials (eqs 7 and 8 of kockarts 1994)
   254) !-----------------------------------------------------------------------------
   255)       do iw = 1,ngast-1
   256)          do k = 1,pverp
   257) 	    ki = index(k)
   258) 	    rjm(k)  = x_table(ki,iw) + dels(k)*(x_table(ki+1,iw) - x_table(ki,iw))
   259) 	    rjo2(k) = d_table(ki,iw) + dels(k)*(d_table(ki+1,iw) - d_table(ki,iw))
   260) 	 end do
   261) 
   262)          do k = 1,pver
   263)             if( rjm(k) > 1.e-100_r8 ) then
   264)                kp1 = k + 1
   265)                if( rjm(kp1) > 0._r8 ) then
   266)                   dto2(k,iw) = log( rjm(kp1) ) / secchi(kp1) - log( rjm(k) ) * secchi(k)
   267)                else
   268)                   dto2(k,iw) = 1000._r8
   269)                end if
   270)             else
   271)                dto2(k,iw)   = 1000._r8
   272)             end if
   273)          end do
   274)          do k = 1,pverp
   275)             if( rjm(k) > 1.e-100_r8 ) then
   276)                if( rjo2(k) > 1.e-100_r8 ) then
   277)                   xscho2(k,iw) = rjo2(k)/rjm(k)
   278)                else
   279)                   xscho2(k,iw) = 0._r8
   280)                end if
   281)             else
   282)                xscho2(k,iw) = 0._r8
   283)             end if
   284)          end do
