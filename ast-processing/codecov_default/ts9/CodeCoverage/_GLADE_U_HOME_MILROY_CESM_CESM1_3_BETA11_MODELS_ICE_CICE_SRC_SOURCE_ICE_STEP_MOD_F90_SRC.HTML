   242)             j = indxj(ij)
   243)

   245)             fswintn(i,j,n,iblk) = scale_factor(i,j,iblk)*fswintn (i,j,n,iblk)

   250)                     scale_factor(i,j,iblk)*Iswabsn(i,j,il1:il2,iblk)

   252)       enddo                  ! ncat

   275) !
   276) ! !USES:
   277) !
   278) ! !INPUT/OUTPUT PARAMETERS:
   279) !
   280)       real (kind=dbl_kind), intent(in) :: &
   281)          dt      ! time step
   282) !
   283) !EOP
   284) !
   285)       integer (kind=int_kind) :: &
   286)          iblk,  &  ! block index
   287)          i, j
   288) 
   289) !      call t_barrierf('cice_step2_therm_BARRIER',MPI_COMM_ICE)
   290)       call t_startf('cice_step2_therm')
   291)       call ice_timer_start(timer_column)  ! column physics
   292)       call ice_timer_start(timer_thermo)  ! thermodynamics
   293) !      call ice_timer_start(timer_tmp)  ! temporary timer
   294) 
   295)       !$OMP PARALLEL DO PRIVATE(iblk)
   296)       do iblk = 1, nblocks
   297)          call step_therm2_iblk(dt, iblk)
   298)       end do
   299)       !$OMP END PARALLEL DO
   300) 
   301)       !-------------------------------------------------------------------
   302)       ! Ghost cell updates for state variables.
   303)       !-------------------------------------------------------------------
   304) 
   305)       call ice_timer_start(timer_bound)
   306)       call bound_state (aicen, trcrn, &
   307)                         vicen, vsnon, &
   308)                         eicen, esnon)
   309)       call ice_timer_stop(timer_bound)
   310) 
   311)       !$OMP PARALLEL DO PRIVATE(iblk,i,j)
   312)       do iblk = 1, nblocks
   313) 
   314)       !-----------------------------------------------------------------
   315)       ! Aggregate the updated state variables. 
   316)       !----------------------------------------------------------------- 
   317)  
   318)          call aggregate (nx_block,          ny_block,             &
   319)                          aicen(:,:,:,iblk), trcrn(:,:,:,:,iblk),  &
   320)                          vicen(:,:,:,iblk), vsnon(:,:,  :,iblk),  &
   321)                          eicen(:,:,:,iblk), esnon(:,:,  :,iblk),  &
   322)                          aice (:,:,  iblk), trcr (:,:,:,  iblk),  &
   323)                          vice (:,:,  iblk), vsno (:,:,    iblk),  &
   324)                          eice (:,:,  iblk), esno (:,:,    iblk),  &
   325)                          aice0(:,:,  iblk), tmask(:,:,    iblk),  &
   326)                          ntrcr,             trcr_depend) 
   327) 
   328) 
   329)       !-----------------------------------------------------------------
   330)       ! Compute thermodynamic area and volume tendencies.
   331)       !-----------------------------------------------------------------
   332) 
   333)          do j = 1, ny_block
   334)          do i = 1, nx_block
   335)             daidtt(i,j,iblk) = (aice(i,j,iblk) - daidtt(i,j,iblk)) / dt
   336)             dvidtt(i,j,iblk) = (vice(i,j,iblk) - dvidtt(i,j,iblk)) / dt
   337)          enddo
   338)          enddo
   339) 
   340)       enddo                     ! iblk
   341)       !$OMP END PARALLEL DO
   342) 
   343)       call t_stopf('cice_step2_therm')
   344) !      call ice_timer_stop(timer_tmp)  ! temporary timer
   345)       call ice_timer_stop(timer_thermo)  ! column physics
   346)       call ice_timer_stop(timer_column)  ! column physics
   347) 
   348)       end subroutine step_therm2
