   197) subroutine cam_run1(cam_in, cam_out)
   198) !-----------------------------------------------------------------------
   199) !
   200) ! Purpose:   First phase of atmosphere model run method.
   201) !            Runs first phase of dynamics and first phase of
   202) !            physics (before surface model updates).
   203) !
   204) !-----------------------------------------------------------------------
   205)    
   206)    use physpkg,          only: phys_run1
   207)    use stepon,           only: stepon_run1
   208) #if ( defined SPMD )
   209)    use mpishorthand,     only: mpicom
   210) #endif
   211)    use time_manager,     only: get_nstep
   212) 
   213)    type(cam_in_t)  :: cam_in(begchunk:endchunk)
   214)    type(cam_out_t) :: cam_out(begchunk:endchunk)
   215) 
   216) #if ( defined SPMD )
   217)    real(r8) :: mpi_wtime
   218) #endif
   219) !-----------------------------------------------------------------------
   220) 
   221) #if ( defined SPMD )
   222)    if (stepon_time_beg == -1.0_r8) stepon_time_beg = mpi_wtime()
   223)    if (nstep_beg == -1) nstep_beg = get_nstep()
   224) #endif
   225)    if (masterproc .and. print_step_cost) then
   226)       call t_stampf (wcstart, usrstart, sysstart)
   227)    end if
   228) 
   229)    if (offline_driver_dorun) return
   230) 
   231)    !----------------------------------------------------------
   232)    ! First phase of dynamics (at least couple from dynamics to physics)
   233)    ! Return time-step for physics from dynamics.
   234)    !----------------------------------------------------------
   235)    call t_barrierf ('sync_stepon_run1', mpicom)
   236)    call t_startf ('stepon_run1')
   237)    call stepon_run1( dtime, phys_state, phys_tend, pbuf2d, dyn_in, dyn_out )
   238)    call t_stopf  ('stepon_run1')
   239) 
   240)    !
   241)    !----------------------------------------------------------
   242)    ! PHYS_RUN Call the Physics package
   243)    !----------------------------------------------------------
   244)    !
   245)    call t_barrierf ('sync_phys_run1', mpicom)
   246)    call t_startf ('phys_run1')
   247)    call phys_run1(phys_state, dtime, phys_tend, pbuf2d,  cam_in, cam_out)
   248)    call t_stopf  ('phys_run1')
   249) 
   250) end subroutine cam_run1
   251)

   256) subroutine cam_run2( cam_out, cam_in )
   257) !-----------------------------------------------------------------------
   258) !
   259) ! Purpose:   Second phase of atmosphere model run method.
   260) !            Run the second phase physics, run methods that
   261) !            require the surface model updates.  And run the
   262) !            second phase of dynamics that at least couples
   263) !            between physics to dynamics.
   264) !
   265) !-----------------------------------------------------------------------
   266)    
   267)    use physpkg,          only: phys_run2
   268)    use stepon,           only: stepon_run2
   269)    use time_manager,     only: is_first_step, is_first_restart_step
   270) #if ( defined SPMD )
   271)    use mpishorthand,     only: mpicom
   272) #endif
   273) 
   274)    type(cam_out_t), intent(inout) :: cam_out(begchunk:endchunk)
   275)    type(cam_in_t),  intent(inout) :: cam_in(begchunk:endchunk)
   276) 
   277)    if (offline_driver_dorun) then
   278)       call offline_driver_run( phys_state, pbuf2d, cam_out, cam_in )
   279)       return
   280)    endif
   281) 
   282)    !
   283)    ! Second phase of physics (after surface model update)
   284)    !
   285)    call t_barrierf ('sync_phys_run2', mpicom)
   286)    call t_startf ('phys_run2')
   287)    call phys_run2(phys_state, dtime, phys_tend, pbuf2d,  cam_out, cam_in )
   288)    call t_stopf  ('phys_run2')
   289) 
   290)    !
   291)    ! Second phase of dynamics (at least couple from physics to dynamics)
   292)    !
   293)    call t_barrierf ('sync_stepon_run2', mpicom)
   294)    call t_startf ('stepon_run2')
   295)    call stepon_run2( phys_state, phys_tend, dyn_in, dyn_out )
   296) 
   297)    call t_stopf  ('stepon_run2')
   298) 
   299)    if (is_first_step() .or. is_first_restart_step()) then
   300)       call t_startf ('cam_run2_memusage')
   301)       call t_stopf  ('cam_run2_memusage')
   302)    end if
   303) end subroutine cam_run2
   304)

   309) subroutine cam_run3( cam_out )
   310) !-----------------------------------------------------------------------
   311) !
   312) ! Purpose:  Third phase of atmosphere model run method. This consists
   313) !           of the third phase of the dynamics. For some dycores
   314) !           this will be the actual dynamics run, for others the
   315) !           dynamics happens before physics in phase 1.
   316) !
   317) !-----------------------------------------------------------------------
   318)    use stepon,           only: stepon_run3
   319)    use time_manager,     only: is_first_step, is_first_restart_step
   320) #if ( defined SPMD )
   321)    use mpishorthand,     only: mpicom
   322) #endif
   323) 
   324)    type(cam_out_t), intent(inout) :: cam_out(begchunk:endchunk)
   325) !-----------------------------------------------------------------------
   326) 
   327)    if (offline_driver_dorun) return
   328) 
   329)    !
   330)    ! Third phase of dynamics
   331)    !
   332)    call t_barrierf ('sync_stepon_run3', mpicom)
   333)    call t_startf ('stepon_run3')
   334)    call stepon_run3( dtime, etamid, cam_out, phys_state, dyn_in, dyn_out )
   335) 
   336)    call t_stopf  ('stepon_run3')
   337) 
   338)    if (is_first_step() .or. is_first_restart_step()) then
   339)       call t_startf ('cam_run3_memusage')
   340)       call t_stopf  ('cam_run3_memusage')
   341)    end if
   342) end subroutine cam_run3

   348) subroutine cam_run4( cam_out, cam_in, rstwr, nlend, &
   349)                      yr_spec, mon_spec, day_spec, sec_spec )
   350) 
   351) !-----------------------------------------------------------------------
   352) !
   353) ! Purpose:  Final phase of atmosphere model run method. This consists
   354) !           of all the restart output, history writes, and other
   355) !           file output.
   356) !
   357) !-----------------------------------------------------------------------
   358)    use cam_history,      only: wshist, wrapup
   359)    use cam_restart,      only: cam_write_restart
   360)    use dycore,           only: dycore_is
   361) #if ( defined SPMD )
   362)    use mpishorthand,     only: mpicom
   363) #endif
   364) 
   365)    type(cam_out_t), intent(inout)        :: cam_out(begchunk:endchunk)
   366)    type(cam_in_t) , intent(inout)        :: cam_in(begchunk:endchunk)
   367)    logical            , intent(in)           :: rstwr           ! true => write restart file
   368)    logical            , intent(in)           :: nlend           ! true => this is final timestep
   369)    integer            , intent(in), optional :: yr_spec         ! Simulation year
   370)    integer            , intent(in), optional :: mon_spec        ! Simulation month
   371)    integer            , intent(in), optional :: day_spec        ! Simulation day
   372)    integer            , intent(in), optional :: sec_spec        ! Seconds into current simulation day
   373) 
   374) #if ( defined SPMD )
   375)    real(r8) :: mpi_wtime
   376) #endif
   377) !-----------------------------------------------------------------------
   378) ! print_step_cost
   379) 
   380)    !
   381)    !----------------------------------------------------------
   382)    ! History and restart logic: Write and/or dispose history tapes if required
   383)    !----------------------------------------------------------
   384)    !
   385)    call t_barrierf ('sync_wshist', mpicom)
   386)    call t_startf ('wshist')
   387)    call wshist ()
   388)    call t_stopf  ('wshist')
   389) 
   390) #if ( defined SPMD )
   391)    stepon_time_end = mpi_wtime()
   392) #endif
   393)    !
   394)    ! Write restart files
   395)    !
   396)    if (rstwr) then
   397)       call t_startf ('cam_write_restart')
   398)       if (present(yr_spec).and.present(mon_spec).and.present(day_spec).and.present(sec_spec)) then
   399)          call cam_write_restart(cam_in, cam_out, dyn_out, pbuf2d, &
   400)               yr_spec=yr_spec, mon_spec=mon_spec, day_spec=day_spec, sec_spec= sec_spec )
   401)       else
   402)          call cam_write_restart(cam_in, cam_out, dyn_out, pbuf2d )
   403)       end if
   404)       call t_stopf  ('cam_write_restart')
   405)    end if
   406) 
   407)    call t_startf ('cam_run4_wrapup')
   408)    call wrapup(rstwr, nlend)
   409)    call t_stopf  ('cam_run4_wrapup')
   410) 
   411)    if (masterproc .and. print_step_cost) then
   412)       call t_startf ('cam_run4_print')
   413)       call t_stampf (wcend, usrend, sysend)
   414)       write(iulog,'(a,3f8.3,a)')'Prv timestep wallclock, usr, sys=', &
   415)                             wcend-wcstart, usrend-usrstart, sysend-sysstart, &
   416)                             ' seconds'
   417)       call t_stopf  ('cam_run4_print')
   418)    end if
   419) 
   420) #ifndef UNICOSMP
   421)    call t_startf ('cam_run4_flush')
   422)    call shr_sys_flush(iulog)
   423)    call t_stopf  ('cam_run4_flush')
   424) #endif
   425) 
   426) end subroutine cam_run4
   427)

   432) subroutine cam_final( cam_out, cam_in )
   433) !-----------------------------------------------------------------------
   434) !
   435) ! Purpose:  CAM finalization.
   436) !
   437) !-----------------------------------------------------------------------
   438)    use units,            only: getunit
   439)    use time_manager,     only: get_nstep, get_step_size
   440) #if ( defined SPMD )
   441)    use mpishorthand,     only: mpicom, mpiint, &
   442)                                nsend, nrecv, nwsend, nwrecv
   443)    use spmd_utils,       only: iam, npes
   444) #endif
   445)    use stepon,           only: stepon_final
   446)    use physpkg,          only: phys_final
   447)    use cam_initfiles,    only: cam_initfiles_close
   448)    use camsrfexch,       only: atm2hub_deallocate, hub2atm_deallocate
   449)    !
   450)    ! Arguments
   451)    !
   452)    type(cam_out_t), pointer :: cam_out(:) ! Output from CAM to surface
   453)    type(cam_in_t), pointer :: cam_in(:)   ! Input from merged surface to CAM
   454) !-----------------------------------------------------------------------
   455)    !
   456)    ! Local variables
   457)    !
   458)    integer :: nstep           ! Current timestep number.
   459) 
   460) #if ( defined SPMD )   
   461)    integer :: iu              ! SPMD Statistics output unit number
   462)    character*24 :: filenam    ! SPMD Stats output filename
   463)    integer :: signal          ! MPI message buffer
   464) !------------------------------Externals--------------------------------
   465)    real(r8) :: mpi_wtime
   466) #endif
   467) 
   468)    call phys_final( phys_state, phys_tend , pbuf2d)
   469)    call stepon_final(dyn_in, dyn_out)
   470) 
   471)    if(nsrest==0) then
   472)       call cam_initfiles_close()
   473)    end if
   474) 
   475)    call hub2atm_deallocate(cam_in)
   476)    call atm2hub_deallocate(cam_out)
   477) 
   478) #if ( defined SPMD )
   479)    if (.false.) then
   480)       write(iulog,*)'The following stats are exclusive of initialization/boundary datasets'
   481)       write(iulog,*)'Number of messages sent by proc ',iam,' is ',nsend
   482)       write(iulog,*)'Number of messages recv by proc ',iam,' is ',nrecv
   483)    end if
   484) #endif
   485) 
   486)    ! This flush attempts to ensure that asynchronous diagnostic prints from all 
   487)    ! processes do not get mixed up with the "END OF MODEL RUN" message printed 
   488)    ! by masterproc below.  The test-model script searches for this message in the 
   489)    ! output log to figure out if CAM completed successfully.  This problem has 
   490)    ! only been observed with the Linux Lahey compiler (lf95) which does not 
   491)    ! support line-buffered output.  
   492) #ifndef UNICOSMP
   493)    call shr_sys_flush( 0 )   ! Flush all output to standard error
   494)    call shr_sys_flush( iulog )   ! Flush all output to standard output
   495) #endif
   496) 
   497)    if (masterproc) then
   498) #if ( defined SPMD )
   499)       cam_time_end = mpi_wtime()
   500) #endif
   501)       nstep = get_nstep()
   502)       write(iulog,9300) nstep-1,nstep
   503) 9300  format (//'Number of completed timesteps:',i6,/,'Time step ',i6, &
   504)                 ' partially done to provide convectively adjusted and ', &
   505)                 'time filtered values for history tape.')
   506)       write(iulog,*)'------------------------------------------------------------'
   507) #if ( defined SPMD )
   508)       write(iulog,*)
   509)       write(iulog,*)' Total run time (sec) : ', cam_time_end-cam_time_beg
   510)       write(iulog,*)' Time Step Loop run time(sec) : ', stepon_time_end-stepon_time_beg
   511)       if (((nstep-1)-nstep_beg) > 0) then
   512)          write(iulog,*)' SYPD : ',  &
   513)          236.55_r8/((86400._r8/(dtime*((nstep-1)-nstep_beg)))*(stepon_time_end-stepon_time_beg))
   514)       endif
   515)       write(iulog,*)
   516) #endif
   517)       write(iulog,*)'******* END OF MODEL RUN *******'
   518)    end if
   519) 
   520) 
   521) #if ( defined SPMDSTATS )
   522)    if (t_single_filef()) then
   523)       write(filenam,'(a17)') 'spmdstats_cam.all'
   524)       iu = getunit ()
   525)       if (iam .eq. 0) then
   526)          open (unit=iu, file=filenam, form='formatted', status='replace')
   527)          signal = 1
   528)       else
   529)          call mpirecv(signal, 1, mpiint, iam-1, iam, mpicom) 
   530)          open (unit=iu, file=filenam, form='formatted', status='old', position='append')
   531)       endif
   532)       write (iu,*)'************ PROCESS ',iam,' ************'
   533)       write (iu,*)'iam ',iam,' msgs  sent =',nsend
   534)       write (iu,*)'iam ',iam,' msgs  recvd=',nrecv
   535)       write (iu,*)'iam ',iam,' words sent =',nwsend
   536)       write (iu,*)'iam ',iam,' words recvd=',nwrecv
   537)       write (iu,*)
   538)       close(iu)
   539)       if (iam+1 < npes) then
   540)          call mpisend(signal, 1, mpiint, iam+1, iam+1, mpicom)
   541)       endif
   542)    else
