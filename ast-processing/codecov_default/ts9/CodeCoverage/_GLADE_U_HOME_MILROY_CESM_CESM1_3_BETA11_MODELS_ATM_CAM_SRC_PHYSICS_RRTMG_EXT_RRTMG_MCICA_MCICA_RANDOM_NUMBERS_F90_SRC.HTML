    87)   ! Algorithm parameters
    88)   ! -------
    89)   ! Period parameters
    90)   integer, parameter :: blockSize = 624,         &
    91)                         M         = 397,         &
    92)                         MATRIX_A  = -1727483681, & ! constant vector a         (0x9908b0dfUL)
    93)                         LMASK     =  2147483647, & ! least significant r bits (0x7fffffffUL)
    94)                         UMASK     = (-LMASK) - 1   ! most significant w-r bits (0x80000000UL)
    95)   ! Tempering parameters
    96)   integer, parameter :: TMASKB= -1658038656, & ! (0x9d2c5680UL)
    97)                         TMASKC= -272236544     ! (0xefc60000UL)
    98)   ! -------
    99) 
   100)   ! The type containing the state variable  
   101)   type randomNumberSequence
   102)     integer                            :: currentElement ! = blockSize
   103)     integer, dimension(0:blockSize -1) :: state ! = 0
   104)   end type randomNumberSequence

   118)   function mixbits(u, v)
   119)     integer, intent( in) :: u, v
   120)     integer              :: mixbits
   121)     
   122)     mixbits = ior(iand(u, UMASK), iand(v, LMASK))
   123)   end function mixbits
   124)   ! ---------------------------
   125)   function twist(u, v)
   126)     integer, intent( in) :: u, v
   127)     integer              :: twist
   128) 
   129)     ! Local variable
   130)     integer, parameter, dimension(0:1) :: t_matrix = (/ 0, MATRIX_A /)
   131)     
   132)     twist = ieor(ishft(mixbits(u, v), -1), t_matrix(iand(v, 1)))
   133)     twist = ieor(ishft(mixbits(u, v), -1), t_matrix(iand(v, 1)))
   134)   end function twist

   136)   subroutine nextState(twister)
   137)     type(randomNumberSequence), intent(inout) :: twister
   138)     
   139)     ! Local variables
   140)     integer :: k
   141)     
   142)     do k = 0, blockSize - M - 1
   143)       twister%state(k) = ieor(twister%state(k + M), &
   144)                               twist(twister%state(k), twister%state(k + 1)))
   145)     end do 
   146)     do k = blockSize - M, blockSize - 2
   147)       twister%state(k) = ieor(twister%state(k + M - blockSize), &
   148)                               twist(twister%state(k), twister%state(k + 1)))
   149)     end do 
   150)     twister%state(blockSize - 1) = ieor(twister%state(M - 1), &
   151)                                         twist(twister%state(blockSize - 1), twister%state(0)))
   152)     twister%currentElement = 0
   153) 
   154)   end subroutine nextState

   156)   elemental function temper(y)
   157)     integer, intent(in) :: y
   158)     integer             :: temper
   159)     
   160)     integer :: x
   161)     
   162)     ! Tempering
   163)     x      = ieor(y, ishft(y, -11))
   164)     x      = ieor(x, iand(ishft(x,  7), TMASKB))
   165)     x      = ieor(x, iand(ishft(x, 15), TMASKC))
   166)     temper = ieor(x, ishft(x, -18))
   167)   end function temper

   171)   function initialize_scalar(seed) result(twister)
   172)     integer,       intent(in   ) :: seed
   173)     type(randomNumberSequence)                :: twister 
   174)     
   175)     integer :: i
   176)     ! See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. In the previous versions, 
   177)     !   MSBs of the seed affect only MSBs of the array state[].                       
   178)     !   2002/01/09 modified by Makoto Matsumoto            
   179)     
   180)     twister%state(0) = iand(seed, -1)
   181)     do i = 1,  blockSize - 1 ! ubound(twister%state)
   182)        twister%state(i) = 1812433253 * ieor(twister%state(i-1), &
   183)                                             ishft(twister%state(i-1), -30)) + i
   184)        twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
   185)     end do
   186)     twister%currentElement = blockSize
   187)   end function initialize_scalar
   188)   ! -------------------------------------------------------------
   189)   function initialize_vector(seed) result(twister)
   190)     integer, dimension(0:), intent(in) :: seed
   191)     type(randomNumberSequence)                      :: twister 
   192)     
   193)     integer :: i, j, k, nFirstLoop, nWraps
   194)     
   195)     nWraps  = 0
   196)     twister = initialize_scalar(19650218)
   197)     
   198)     nFirstLoop = max(blockSize, size(seed))
   199)     do k = 1, nFirstLoop
   200)        i = mod(k + nWraps, blockSize)
   201)        j = mod(k - 1,      size(seed))
   202)        if(i == 0) then
   203)          twister%state(i) = twister%state(blockSize - 1)
   204)          twister%state(1) = ieor(twister%state(1),                                 &
   205)                                  ieor(twister%state(1-1),                          & 
   206)                                       ishft(twister%state(1-1), -30)) * 1664525) + & 
   207)                             seed(j) + j ! Non-linear
   208)          twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
   209)          nWraps = nWraps + 1
   210)        else
   211)          twister%state(i) = ieor(twister%state(i),                                 &
   212)                                  ieor(twister%state(i-1),                          & 
   213)                                       ishft(twister%state(i-1), -30)) * 1664525) + & 
   214)                             seed(j) + j ! Non-linear
   215)          twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
   216)       end if
   217)     end do
   218)     
   219)     !
   220)     ! Walk through the state array, beginning where we left off in the block above
   221)     ! 
   222)     do i = mod(nFirstLoop, blockSize) + nWraps + 1, blockSize - 1
   223)       twister%state(i) = ieor(twister%state(i),                                 &
   224)                               ieor(twister%state(i-1),                          & 
   225)                                    ishft(twister%state(i-1), -30)) * 1566083941) - i ! Non-linear
   226)       twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
   227)     end do
   228)     
   229)     twister%state(0) = twister%state(blockSize - 1) 
   230)     
   231)     do i = 1, mod(nFirstLoop, blockSize) + nWraps
   232)       twister%state(i) = ieor(twister%state(i),                                 &
   233)                               ieor(twister%state(i-1),                          & 
   234)                                    ishft(twister%state(i-1), -30)) * 1566083941) - i ! Non-linear
   235)       twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
   236)     end do
   237)     
   238)     twister%state(0) = UMASK 
   239)     twister%currentElement = blockSize
   240)     
   241)   end function initialize_vector

   245)   function getRandomInt(twister)
   246)     type(randomNumberSequence), intent(inout) :: twister
   247)     integer                        :: getRandomInt
   248)     ! Generate a random integer on the interval [0,0xffffffff]
   249)     !   Equivalent to genrand_int32 in the C code. 
   250)     !   Fortran doesn't have a type that's unsigned like C does, 
   251)     !   so this is integers in the range -2**31 - 2**31
   252)     ! All functions for getting random numbers call this one, 
   253)     !   then manipulate the result
   254)     
   255)     if(twister%currentElement >= blockSize) call nextState(twister)
   256)       
   257)     getRandomInt = temper(twister%state(twister%currentElement))
   258)     twister%currentElement = twister%currentElement + 1
   259)   
   260)   end function getRandomInt

   282)     ! Generate a random number on [0,1]
