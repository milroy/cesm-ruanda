    73)     use error_messages, only: handle_err

    91)     call trcdata_init( fld_names, filename, filelist, datapath, fields, file, &
    92)                        rmv_file, cycle_yr, fixed_ymd, fixed_tod, datatype)
    93)         
    94)     number_flds = 0
    95)     if (associated(fields)) number_flds = size( fields )
    96) 
    97)     if( number_flds < 1 ) then
    98)        if ( masterproc ) then
    99)           write(iulog,*) 'linoz_data_init: There are no linoz data'
   100)           write(iulog,*) ' '
   101)        endif
   102)        return
   103)     end if
   104) 
   105)     do i = 1,number_flds
   106)        ndx = get_ndx( fields(i)%fldnam )
   107)        index_map(i) = ndx
   108) 
   109)        if (ndx < 1) then
   110)           call endrun('linoz_data_init: '//trim(fields(i)%fldnam)//' is not one of the named linoz data fields ')
   111)        endif
   112)        call addfld(fld_names(i), fld_units(i), pver, 'I', 'linoz data', phys_decomp )
   113)     enddo
