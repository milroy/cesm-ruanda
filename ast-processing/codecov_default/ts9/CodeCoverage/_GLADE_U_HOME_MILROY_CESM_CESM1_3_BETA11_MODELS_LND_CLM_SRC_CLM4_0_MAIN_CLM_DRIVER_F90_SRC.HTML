     6) ! !MODULE: clm_driver
     7) !
     8) ! !DESCRIPTION:
     9) ! This module provides the main CLM driver physics calling sequence.  Most
    10) ! computations occurs over ``clumps'' of gridcells (and associated subgrid
    11) ! scale entities) assigned to each MPI process. Computation is further
    12) ! parallelized by looping over clumps on each process using shared memory OpenMP.
    13) !
    14) ! The main CLM driver physics calling sequence for clm_driver1 is as follows:
    15) ! \begin{verbatim}
    16) !
    17) !     + interpMonthlyVeg      interpolate monthly vegetation data        [! CN or ! CNDV]
    18) !       + readMonthlyVegetation read vegetation data for two months      [! CN or ! CNDV]
    19) !
    20) ! ==== Begin Loop over clumps ====
    21) !  -> dynland_hwcontent   Get initial heat, water content
    22) !     + pftdyn_interp                                                    [pftdyn]
    23) !     + dynland_hwcontent   Get new heat, water content                  [pftdyn]
    24) ! ==== End Loop over clumps  ====
    25) !
    26) ! ==== Begin Loop over clumps ====
    27) !  -> clm_driverInit      save of variables from previous time step
    28) !  -> Hydrology1          canopy interception and precip on ground
    29) !     -> FracWet          fraction of wet vegetated surface and dry elai
    30) !  -> SurfaceRadiation    surface solar radiation
    31) !  -> UrbanRadiation      surface solar and longwave radiation for Urban landunits
    32) !  -> Biogeophysics1      leaf temperature and surface fluxes
    33) !  -> BareGroundFluxes    surface fluxes for bare soil or snow-covered
    34) !                         vegetation patches
    35) !  -> UrbanFluxes         surface fluxes for urban landunits
    36) !     -> MoninObukIni     first-guess Monin-Obukhov length and wind speed
    37) !     -> FrictionVelocity friction velocity and potential temperature and
    38) !                         humidity profiles
    39) !  -> CanopyFluxes        leaf temperature and surface fluxes for vegetated
    40) !                         patches
    41) !     -> QSat             saturated vapor pressure, specific humidity, &
    42) !                         derivatives at leaf surface
    43) !     -> MoninObukIni     first-guess Monin-Obukhov length and wind speed
    44) !     -> FrictionVelocity friction velocity and potential temperature and
    45) !                         humidity profiles
    46) !     -> Stomata          stomatal resistance and photosynthesis for
    47) !                         sunlit leaves
    48) !     -> Stomata          stomatal resistance and photosynthesis for
    49) !                         shaded leaves
    50) !     -> QSat             recalculation of saturated vapor pressure,
    51) !                         specific humidity, & derivatives at leaf surface
    52) !   + DustEmission        Dust mobilization
    53) !   + DustDryDep          Dust dry deposition
    54) !  -> Biogeophysics_Lake  lake temperature and surface fluxes
    55) !   + VOCEmission         compute VOC emission                          [VOC]
    56) !  -> Biogeophysics2      soil/snow & ground temp and update surface fluxes
    57) !  -> pft2col             Average from PFT level to column level
    58) !  -> Hydrology2          surface and soil hydrology
    59) !  -> Hydrology_Lake      lake hydrology
    60) !  -> SnowAge_grain       update snow effective grain size for snow radiative transfer
    61) !   + CNEcosystemDyn      Carbon Nitrogen model ecosystem dynamics:     [CN]
    62) !                         vegetation phenology and soil carbon  
    63) !   + EcosystemDyn        "static" ecosystem dynamics:                  [! CN ]
    64) !                         vegetation phenology and soil carbon  
    65) !  -> BalanceCheck        check for errors in energy and water balances
    66) !  -> SurfaceAlbedo       albedos for next time step
    67) !  -> UrbanAlbedo         Urban landunit albedos for next time step
    68) !  ====  End Loop over clumps  ====
    69) !
    70) ! Second phase of the clm main driver, for handling history and restart file output.
    71) !
    72) !  -> write_diagnostic    output diagnostic if appropriate
    73) !  -> updateAccFlds       update accumulated fields
    74) !  -> hist_update_hbuf    accumulate history fields for time interval
    75) !  -> htapes_wrapup       write history tapes if appropriate
    76) !  -> restFile_write      write restart file if appropriate
    77) ! \end{verbatim}
    78) !
    79) ! Optional subroutines are denoted by an plus (+) with the associated
    80) ! CPP token or variable in brackets at the end of the line.
    81) !
    82) ! !USES:
    83)   use shr_kind_mod        , only : r8 => shr_kind_r8
    84)   use clmtype
    85)   use clm_varctl          , only : wrtdia, flanduse_timeseries, iulog, create_glacier_mec_landunit, &
    86)                                    use_cn, use_cndv, use_exit_spinup
    87)   use spmdMod             , only : masterproc,mpicom
    88)   use decompMod           , only : get_proc_clumps, get_clump_bounds, get_proc_bounds
    89)   use filterMod           , only : filter, setFilters
    90)   use CNDVMod             , only : dv, histCNDV
    91)   use pftdynMod           , only : pftwt_interp
    92)   use pftdynMod           , only : pftdyn_interp, pftdyn_wbal_init, pftdyn_wbal
    93)   use pftdynMod           , only : pftdyn_cnbal
    94)   use dynlandMod          , only : dynland_hwcontent
    95)   use clm_varcon          , only : zlnd, isturb
    96)   use clm_time_manager    , only : get_step_size,get_curr_date,get_ref_date,get_nstep
    97)   use CropRestMod         , only : CropRestIncYear
    98)   use histFileMod         , only : hist_update_hbuf, hist_htapes_wrapup
    99)   use restFileMod         , only : restFile_write, restFile_filename
   100)   use accFldsMod          , only : updateAccFlds
   101)   use clm_driverInitMod   , only : clm_driverInit
   102)   use BalanceCheckMod     , only : BeginWaterBalance, BalanceCheck
   103)   use SurfaceRadiationMod , only : SurfaceRadiation
   104)   use Hydrology1Mod       , only : Hydrology1
   105)   use Hydrology2Mod       , only : Hydrology2
   106)   use HydrologyLakeMod    , only : HydrologyLake
   107)   use Biogeophysics1Mod   , only : Biogeophysics1
   108)   use BareGroundFluxesMod , only : BareGroundFluxes
   109)   use CanopyFluxesMod     , only : CanopyFluxes
   110)   use Biogeophysics2Mod   , only : Biogeophysics2
   111)   use BiogeophysicsLakeMod, only : BiogeophysicsLake
   112)   use SurfaceAlbedoMod    , only : SurfaceAlbedo
   113)   use pft2colMod          , only : pft2col
   114)   use CNSetValueMod       , only : CNZeroFluxes_dwt
   115)   use CNEcosystemDynMod   , only : CNEcosystemDyn
   116)   use CNAnnualUpdateMod   , only : CNAnnualUpdate
   117)   use CNBalanceCheckMod   , only : BeginCBalance, BeginNBalance, &
   118)                                    CBalanceCheck, NBalanceCheck
   119)   use ndepStreamMod       , only : ndep_interp
   120)   use STATICEcosysDynMod  , only : EcosystemDyn
   121)   use DUSTMod             , only : DustDryDep, DustEmission
   122)   use VOCEmissionMod      , only : VOCEmission
   123)   use seq_drydep_mod      , only : n_drydep, drydep_method, DD_XLND
   124)   use STATICEcosysDynMod  , only : interpMonthlyVeg
   125)   use DryDepVelocity      , only : depvel_compute
   126)   use abortutils          , only : endrun
   127)   use UrbanMod            , only : UrbanAlbedo, UrbanRadiation, UrbanFluxes 
   128)   use SNICARMod           , only : SnowAge_grain
   129)   use clm_atmlnd          , only : clm_map2gcell
   130)   use clm_glclnd          , only : update_clm_s2x
   131)   use perf_mod
   132) !

   134)   implicit none
   135) !

   137)   public :: clm_drv                 ! clm physics,history, restart writes

   149) ! !ROUTINE: clm_drv
   150) !
   151) ! !INTERFACE:
   152) subroutine clm_drv(doalb, nextsw_cday, declinp1, declin, rstwr, nlend, rdate)
   153) !
   154) ! !DESCRIPTION:
   155) !
   156) ! First phase of the clm driver calling the clm physics. An outline of
   157) ! the calling tree is given in the description of this module.
   158) !
   159) ! !USES:
   160) 
   161) ! !ARGUMENTS:
   162)   implicit none
   163)   logical,         intent(in) :: doalb       ! true if time for surface albedo calc
   164)   real(r8),        intent(in) :: nextsw_cday ! calendar day for nstep+1
   165)   real(r8),        intent(in) :: declinp1    ! declination angle for next time step
   166)   real(r8),        intent(in) :: declin      ! declination angle for current time step
   167)   logical,         intent(in) :: rstwr       ! true => write restart file this step
   168)   logical,         intent(in) :: nlend       ! true => end of run on this step
   169)   character(len=*),intent(in) :: rdate       ! restart file time stamp for name
   170) !
   171) ! !REVISION HISTORY:
   172) ! 2002.10.01  Mariana Vertenstein latest update to new data structures
   173) ! 11/26/03, Peter Thornton: Added new call for SurfaceRadiationSunShade when
   174) !  cpp directive SUNSHA is set, for sunlit/shaded canopy radiation.
   175) ! 4/25/05, Peter Thornton: Made the sun/shade routine the default, no longer
   176) !  need to have SUNSHA defined.  
   177) ! Oct/05 & Jul/07 Sam Levis: Starting dates of CNDV and crop model work
   178) ! 2/29/08, Dave Lawrence: Revised snow cover fraction according to Niu and Yang, 2007
   179) ! 3/6/09, Peter Thornton: Added declin as new argument, for daylength control on Vcmax
   180) ! 2008.11.12  B. Kauffman: morph routine casa() in casa_ecosytemDyn(), so casa
   181) !    is more similar to CN & DGVM
   182) ! 2/25/2012 M. Vertenstein: Removed CASA references 
   183) !
   184) !EOP
   185) !
   186) ! !LOCAL VARIABLES:
   187) !
   188) ! local pointers to implicit in arguments
   189) !
   190)   integer , pointer :: clandunit(:) ! landunit index associated with each column
   191)   integer , pointer :: itypelun(:)  ! landunit type
   192) !
   193) ! !OTHER LOCAL VARIABLES:
   194)   integer  :: nstep                    ! time step number
   195)   real(r8) :: dtime                    ! land model time step (sec)
   196)   real(r8) :: t1, t2, t3               ! temporary for mass balance checks
   197)   integer  :: nc, fc, c, fp, p, l, g   ! indices
   198)   integer  :: nclumps                  ! number of clumps on this processor
   199)   integer  :: begg, endg               ! clump beginning and ending gridcell indices
   200)   integer  :: begl, endl               ! clump beginning and ending landunit indices
   201)   integer  :: begc, endc               ! clump beginning and ending column indices
   202)   integer  :: begp, endp               ! clump beginning and ending pft indices
   203)   integer  :: begg_proc, endg_proc     ! proc beginning and ending gridcell indices
   204)   integer  :: begl_proc, endl_proc     ! proc beginning and ending landunit indices
   205)   integer  :: begc_proc, endc_proc     ! proc beginning and ending column indices
   206)   integer  :: begp_proc, endp_proc     ! proc beginning and ending pft indices
   207)   type(column_type), pointer :: cptr   ! pointer to column derived subtype
   208)   integer  :: yrp1                     ! year (0, ...) for nstep+1
   209)   integer  :: monp1                    ! month (1, ..., 12) for nstep+1
   210)   integer  :: dayp1                    ! day of month (1, ..., 31) for nstep+1
   211)   integer  :: secp1                    ! seconds into current date for nstep+1
   212)   integer  :: yr                       ! year (0, ...)
   213)   integer  :: mon                      ! month (1, ..., 12)
   214)   integer  :: day                      ! day of month (1, ..., 31)
   215)   integer  :: sec                      ! seconds of the day
   216)   integer  :: ncdate                   ! current date
   217)   integer  :: nbdate                   ! base date (reference date)
   218)   integer  :: kyr                      ! thousand years, equals 2 at end of first year
   219)   character(len=256) :: filer          ! restart file name
   220)   integer :: ier                       ! error code
   221) !-----------------------------------------------------------------------
   222) 
   223)   ! Assign local pointers to derived subtypes components (landunit-level)
   224) 
   225)   itypelun            => lun%itype
   226) 
   227)   ! Assign local pointers to derived subtypes components (column-level)
   228) 
   229)   clandunit           => col%landunit
   230) 
   231)   ! Set pointers into derived type
   232) 
   233)   cptr => col
   234) 
   235)   ! Update time-related info
   236) 
   237)   call CropRestIncYear()
   238) 
   239) 
   240)   if (use_cn) then
   241)      ! For dry-deposition need to call CLMSP so that mlaidiff is obtained
   242)      if ( n_drydep > 0 .and. drydep_method == DD_XLND ) then
   243)         call t_startf('interpMonthlyVeg')
   244)         call interpMonthlyVeg()
   245)         call t_stopf('interpMonthlyVeg')
   246)      endif
   247)   else
   248)      ! Determine weights for time interpolation of monthly vegetation data.
   249)      ! This also determines whether it is time to read new monthly vegetation and
   250)      ! obtain updated leaf area index [mlai1,mlai2], stem area index [msai1,msai2],
   251)      ! vegetation top [mhvt1,mhvt2] and vegetation bottom [mhvb1,mhvb2]. The
   252)      ! weights obtained here are used in subroutine ecosystemdyn to obtain time
   253)      ! interpolated values.
   254)      if (doalb .or. ( n_drydep > 0 .and. drydep_method == DD_XLND )) then
   255)         call t_startf('interpMonthlyVeg')
   256)         call interpMonthlyVeg()
   257)         call t_stopf('interpMonthlyVeg')
   258)      end if
   259)   end if
   260) 
   261)   ! ============================================================================
   262)   ! Loop over clumps
   263)   ! ============================================================================
   264) 
   265)   nclumps = get_proc_clumps()
   266) 
   267)   !$OMP PARALLEL DO PRIVATE (nc,g,begg,endg,begl,endl,begc,endc,begp,endp)
   268)   do nc = 1,nclumps
   269) 
   270)      ! ============================================================================
   271)      ! Determine clump boundaries
   272)      ! ============================================================================
   273) 
   274)      call get_clump_bounds(nc, begg, endg, begl, endl, begc, endc, begp, endp)
   275) 
   276)      ! ============================================================================
   277)      ! change pft weights and compute associated heat & water fluxes
   278)      ! ============================================================================
   279) 
   280)      ! initialize heat and water content and dynamic balance fields to zero
   281)      do g = begg,endg
   282)         gwf%qflx_liq_dynbal(g) = 0._r8
   283)         gws%gc_liq2(g)         = 0._r8
   284)         gws%gc_liq1(g)         = 0._r8
   285)         gwf%qflx_ice_dynbal(g) = 0._r8
   286)         gws%gc_ice2(g)         = 0._r8 
   287)         gws%gc_ice1(g)         = 0._r8
   288)         gef%eflx_dynbal(g)     = 0._r8
   289)         ges%gc_heat2(g)        = 0._r8
   290)         ges%gc_heat1(g)        = 0._r8
   291)      enddo
   292) 
   293)      !--- get initial heat,water content ---
   294)       call dynland_hwcontent( begg, endg, gws%gc_liq1(begg:endg), &
   295)                               gws%gc_ice1(begg:endg), ges%gc_heat1(begg:endg) )
   296)    end do
   297)    !$OMP END PARALLEL DO
   298) 
   299)    if (.not. use_cndv) then
   300)       if (flanduse_timeseries /= ' ') then
   301)          call pftdyn_interp  ! change the pft weights
   302)       
   303)          !$OMP PARALLEL DO PRIVATE (nc,g,begg,endg,begl,endl,begc,endc,begp,endp)
   304)          do nc = 1,nclumps
   305)             call get_clump_bounds(nc, begg, endg, begl, endl, begc, endc, begp, endp)
   306)             
   307)             !--- get new heat,water content: (new-old)/dt = flux into lnd model ---
   308)             call dynland_hwcontent( begg, endg, gws%gc_liq2(begg:endg), &
   309)                  gws%gc_ice2(begg:endg), ges%gc_heat2(begg:endg) )
   310)             dtime = get_step_size()
   311)             do g = begg,endg
   312)                gwf%qflx_liq_dynbal(g) = (gws%gc_liq2 (g) - gws%gc_liq1 (g))/dtime
   313)                gwf%qflx_ice_dynbal(g) = (gws%gc_ice2 (g) - gws%gc_ice1 (g))/dtime
   314)                gef%eflx_dynbal    (g) = (ges%gc_heat2(g) - ges%gc_heat1(g))/dtime
   315)             enddo
   316)          end do
   317)          !$OMP END PARALLEL DO
   318)       end if
   319)    end if
   320)       
   321)    !$OMP PARALLEL DO PRIVATE (nc,g,begg,endg,begl,endl,begc,endc,begp,endp)
   322)    do nc = 1,nclumps
   323)      call get_clump_bounds(nc, begg, endg, begl, endl, begc, endc, begp, endp)
   324) 
   325)      ! ============================================================================
   326)      ! Initialize the mass balance checks: water, carbon, and nitrogen
   327)      ! ============================================================================
   328) 
   329)      call t_startf('begwbal')
   330)      call BeginWaterBalance(begc, endc, begp, endp, &
   331)           filter(nc)%num_nolakec, filter(nc)%nolakec, filter(nc)%num_lakec, filter(nc)%lakec, &
   332)           filter(nc)%num_hydrologyc, filter(nc)%hydrologyc)
   333)      call t_stopf('begwbal')
   334) 
   335)      if (use_cn) then
   336)         call t_startf('begcnbal')
   337)         call BeginCBalance(begc, endc, filter(nc)%num_soilc, filter(nc)%soilc)
   338)         call BeginNBalance(begc, endc, filter(nc)%num_soilc, filter(nc)%soilc)
   339)         call t_stopf('begcnbal')
   340)      end if
   341) 
   342)   end do
   343)   !$OMP END PARALLEL DO
   344) 
   345)   ! ============================================================================
   346)   ! Initialize h2ocan_loss to zero
   347)   ! ============================================================================
   348) 
   349)   call t_startf('pftdynwts')
   350) 
   351)   !$OMP PARALLEL DO PRIVATE (nc,begg,endg,begl,endl,begc,endc,begp,endp)
   352)   do nc = 1,nclumps
   353)      call get_clump_bounds(nc, begg, endg, begl, endl, begc, endc, begp, endp)
   354)      call pftdyn_wbal_init( begc, endc )
   355) 
   356)      if (use_cndv) then
   357)         ! NOTE: Currently CNDV and flanduse_timeseries /= ' ' are incompatible
   358)         call CNZeroFluxes_dwt( begc, endc, begp, endp )
   359)         call pftwt_interp( begp, endp )
   360)         call pftdyn_wbal( begg, endg, begc, endc, begp, endp )
   361)         call pftdyn_cnbal( begc, endc, begp, endp )
   362)         call setFilters(nc)
   363)      else
   364)         ! ============================================================================
   365)         ! Update weights and reset filters if dynamic land use
   366)         ! This needs to be done outside the clumps loop, but after BeginWaterBalance()
   367)         ! The call to CNZeroFluxes_dwt() is needed regardless of flanduse_timeseries
   368)         ! ============================================================================
   369)         if (use_cn) then
   370)            call CNZeroFluxes_dwt( begc, endc, begp, endp )
   371)         end if
   372)         if (flanduse_timeseries /= ' ') then
   373)            if (use_cn) then
   374)               call pftdyn_cnbal( begc, endc, begp, endp )
   375)            end if
   376)            call setFilters(nc)
   377)         end if
   378)      end if
   379) 
   380)   end do
   381)   !$OMP END PARALLEL DO
   382) 
   383) 
   384)   if (use_cn) then
   385)      ! ============================================================================
   386)      ! Update dynamic N deposition field, on albedo timestep
   387)      ! currently being done outside clumps loop, but no reason why it couldn't be
   388)      ! re-written to go inside.
   389)      ! ============================================================================
   390)      ! PET: switching CN timestep
   391)      call ndep_interp()
   392)   end if
   393)   call t_stopf('pftdynwts')
   394) 
   395)   !$OMP PARALLEL DO PRIVATE (nc,l,c,begg,endg,begl,endl,begc,endc,begp,endp)
   396)   do nc = 1,nclumps
   397) 
   398)      ! ============================================================================
   399)      ! Determine clump boundaries
   400)      ! ============================================================================
   401) 
   402)      call get_clump_bounds(nc, begg, endg, begl, endl, begc, endc, begp, endp)
   403) 
   404)      ! ============================================================================
   405)      ! Initialize variables from previous time step and
   406)      ! Determine canopy interception and precipitation onto ground surface.
   407)      ! Determine the fraction of foliage covered by water and the fraction
   408)      ! of foliage that is dry and transpiring. Initialize snow layer if the
   409)      ! snow accumulation exceeds 10 mm.
   410)      ! ============================================================================
   411)      
   412)      ! initialize intracellular CO2 (Pa) parameters each timestep for use in VOCEmission
   413)      pps%cisun(begp:endp) = -999._r8
   414)      pps%cisha(begp:endp) = -999._r8
   415) 
   416)      ! initialize declination for current timestep
   417)      do c = begc,endc
   418)         cps%decl(c) = declin
   419)      end do
   420)      
   421)      call t_startf('drvinit')
   422)      call clm_driverInit(begc, endc, begp, endp, &
   423)           filter(nc)%num_nolakec, filter(nc)%nolakec, filter(nc)%num_lakec, filter(nc)%lakec)
   424)      call t_stopf('drvinit')
   425) 
   426)      ! ============================================================================
   427)      ! Hydrology1
   428)      ! ============================================================================
   429) 
   430)      call t_startf('hydro1')
   431)      call Hydrology1(begc, endc, begp, endp, &
   432)                      filter(nc)%num_nolakec, filter(nc)%nolakec, &
   433)                      filter(nc)%num_nolakep, filter(nc)%nolakep)
   434)      call t_stopf('hydro1')
   435) 
   436)      ! ============================================================================
   437)      ! Surface Radiation
   438)      ! ============================================================================
   439) 
   440)      call t_startf('surfrad')
   441) 
   442)      ! Surface Radiation for non-urban columns
   443) 
   444)      call SurfaceRadiation(begp, endp, &
   445)                            filter(nc)%num_nourbanp, filter(nc)%nourbanp)
   446) 
   447)      ! Surface Radiation for urban columns
   448) 
   449)      call UrbanRadiation(nc, begl, endl, begc, endc, begp, endp, &
   450)                          filter(nc)%num_nourbanl, filter(nc)%nourbanl, &
   451)                          filter(nc)%num_urbanl, filter(nc)%urbanl, &
   452)                          filter(nc)%num_urbanc, filter(nc)%urbanc, &
   453)                          filter(nc)%num_urbanp, filter(nc)%urbanp)
   454) 
   455)      call t_stopf('surfrad')
   456) 
   457)      ! ============================================================================
   458)      ! Determine leaf temperature and surface fluxes based on ground
   459)      ! temperature from previous time step.
   460)      ! ============================================================================
   461) 
   462)      call t_startf('bgp1')
   463)      call Biogeophysics1(begg, endg, begc, endc, begp, endp, &
   464)                          filter(nc)%num_nolakec, filter(nc)%nolakec, &
   465)                          filter(nc)%num_nolakep, filter(nc)%nolakep)
   466)      call t_stopf('bgp1')
   467) 
   468)      ! ============================================================================
   469)      ! Determine bare soil or snow-covered vegetation surface temperature and fluxes
   470)      ! Calculate Ground fluxes (frac_veg_nosno is either 1 or 0)
   471)      ! ============================================================================
   472) 
   473)      call t_startf('bgflux')
   474) 
   475)      ! BareGroundFluxes for all pfts except lakes and urban landunits
   476) 
   477)      call BareGroundFluxes(begp, endp, &
   478)                            filter(nc)%num_nolakeurbanp, filter(nc)%nolakeurbanp)

   808) ! !INTERFACE:
   809) subroutine write_diagnostic (wrtdia, nstep)
   810) !
   811) ! !DESCRIPTION:
   812) ! Write diagnostic surface temperature output each timestep.  Written to
   813) ! be fast but not bit-for-bit because order of summations can change each
   814) ! timestep.
   815) !
   816) ! !USES:
   817)   use clm_atmlnd , only : clm_l2a
   818)   use decompMod  , only : get_proc_bounds, get_proc_global
   819)   use spmdMod    , only : masterproc, npes, MPI_REAL8, MPI_ANY_SOURCE, &
   820)                           MPI_STATUS_SIZE, mpicom, MPI_SUM
   821)   use shr_sys_mod, only : shr_sys_flush
   822)   use abortutils , only : endrun
   823) !
   824) ! !ARGUMENTS:
   825)   implicit none
   826)   logical, intent(in) :: wrtdia     !true => write diagnostic
   827)   integer, intent(in) :: nstep      !model time step
   828) !
   829) ! !REVISION HISTORY:
   830) ! Created by Mariana Vertenstein
   831) !
   832) !EOP
   833) !
   834) ! !LOCAL VARIABLES:
   835)   integer :: p                       ! loop index
   836)   integer :: begp, endp              ! per-proc beginning and ending pft indices
   837)   integer :: begc, endc              ! per-proc beginning and ending column indices
   838)   integer :: begl, endl              ! per-proc beginning and ending landunit indices
   839)   integer :: begg, endg              ! per-proc gridcell ending gridcell indices
   840)   integer :: numg                    ! total number of gridcells across all processors
   841)   integer :: numl                    ! total number of landunits across all processors
   842)   integer :: numc                    ! total number of columns across all processors
   843)   integer :: nump                    ! total number of pfts across all processors
   844)   integer :: ier                     ! error status
   845)   real(r8):: psum                    ! partial sum of ts
   846)   real(r8):: tsum                    ! sum of ts
   847)   real(r8):: tsxyav                  ! average ts for diagnostic output
   848)   integer :: status(MPI_STATUS_SIZE) ! mpi status
   849)   logical,parameter :: old_sendrecv = .false.  ! Flag if should use old send/receive method rather than MPI reduce
   850) !------------------------------------------------------------------------
   851) 
   852)   call get_proc_bounds(begg, endg, begl, endl, begc, endc, begp, endp)
   853)   call get_proc_global(numg, numl, numc, nump)
   854) 
   855)   if (wrtdia) then
   856) 
   857)      call t_barrierf('sync_write_diag', mpicom)
   858)      psum = sum(clm_l2a%t_rad(begg:endg))
   859)      if (old_sendrecv) then
   860)         if (masterproc) then
   861)            tsum = psum
   862)            do p = 1, npes-1
   863)               call mpi_recv(psum, 1, MPI_REAL8, p, 999, mpicom, status, ier)
   864)               if (ier/=0) then
   865)                  write(iulog,*) 'write_diagnostic: Error in mpi_recv()'
   866)                  call endrun
   867)               end if
   868)               tsum = tsum + psum
   869)            end do
   870)         else
   871)            call mpi_send(psum, 1, MPI_REAL8, 0, 999, mpicom, ier)
   872)            if (ier/=0) then
   873)               write(iulog,*) 'write_diagnostic: Error in mpi_send()'
   874)               call endrun
   875)            end if
   876)         end if
   877)      else
   878)         call mpi_reduce(psum, tsum, 1, MPI_REAL8, MPI_SUM, 0, mpicom, ier)
   879)         if (ier/=0) then
   880)            write(iulog,*) 'write_diagnostic: Error in mpi_reduce()'
   881)            call endrun
   882)         end if
   883)      endif
   884)      if (masterproc) then
   885)         tsxyav = tsum / numg
   886)         write(iulog,1000) nstep, tsxyav
   887)         call shr_sys_flush(iulog)
   888)      end if
   889) 
   890)   else
   891) 
   892)      if (masterproc) then
   893)         write(iulog,*)'clm2: completed timestep ',nstep
   894)         call shr_sys_flush(iulog)
   895)      end if
   896) 
   897)   endif
   898) 
   899) 1000 format (1x,'nstep = ',i10,'   TS = ',f21.15)
