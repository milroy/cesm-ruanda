   260)   subroutine find_iodesc(dimcnt, dimlens, dtype, decomptype, iodesc, found)
   261)     use dycore, only: dycore_is
   262)     integer, intent(in) :: dimcnt
   263)     integer, intent(in) :: dimlens(dimcnt)
   264)     integer, intent(in) :: dtype
   265)     integer, intent(in) :: decomptype
   266)     type(io_desc_t), pointer :: iodesc
   267)     logical,intent(out) :: found
   268)     type(iodesc_list), pointer :: this, prev
   269)     integer :: i
   270)     integer(i8) :: tag, j
   271)     integer :: multiplier = 1000
   272) 
   273)     found = .false.
   274)     this => iodesc_list_top
   275) 
   276)     if(dycore_is('UNSTRUCTURED')) multiplier = 10000
   277) 
   278)     j=1
   279)     tag = 0
   280)     do i=1,dimcnt
   281)        tag = tag+dimlens(i)*j
   282)        j=j*multiplier
   283)     end do
   284)     tag = tag+j*int((dimcnt*1000+dtype*100+decomptype),i8)
   285) 
   286) 
   287)     do while(associated(this) .and. .not. found)
   288)        if(tag==this%tag) then
   289)           found=.true.
   290)           iodesc => this%iodesc
   291)        else
   292)           prev=>this
   293)           this=>this%next
   294)        end if
   295)     end do
   296)     if(.not.found) then
   297)        this=>prev
   298)        if(associated(this%iodesc)) then
   299)           allocate(this%next)
   300)           this=>this%next
   301)        end if
   302)        allocate(this%iodesc)
   303)        this%tag = tag
   304)        iodesc=>this%iodesc
   305)        if(masterproc) write(iulog,*) 'Creating new decomp: ',this%tag
   306)        nullify(this%next)
   307)     end if
   308) !    if(masterproc) write(iulog,*) 'Using decomp: ',this%tag
   309)     
   310)   end subroutine find_iodesc
