   242)   subroutine clmvarctl_init( masterproc, dtime )
   243) !
   244) ! !DESCRIPTION:
   245) !      Check that values are correct, and finish setting variables based on other variables.

   247) ! !USES:

   251) ! !ARGUMENTS:

   253)   integer, intent(IN) :: dtime       ! timestep in seconds
   254) !
   255) ! !LOCAL VARIABLES:
   256)    character(len=32) :: subname = 'clmvarctl_init'  ! subroutine name

   259) ! Author: Erik Kluzek

   261) !EOP
   262) !-----------------------------------------------------------------------
   263) 
   264)     ! landunit generation
   265) 
   266)     if (maxpatch_pft == numpft+1) then
   267)        allocate_all_vegpfts = .true.
   268)     else
   269)        allocate_all_vegpfts = .false.
   270)        if (use_crop) then
   271)           write(iulog,*)'maxpatch_pft = ',maxpatch_pft,&
   272)                ' does NOT equal numpft+1 = ',numpft+1
   273)           call shr_sys_abort( subname//' ERROR:: Can NOT turn CROP on without all PFTs' )
   274)        end if
   275)     end if
   276) 
   277)     if (masterproc) then
   278) 
   279)        ! Consistency settings for co2 type
   280) 
   281)        if (co2_type /= 'constant' .and. co2_type /= 'prognostic' .and. co2_type /= 'diagnostic') then
   282)           write(iulog,*)'co2_type = ',co2_type,' is not supported'
   283)           call shr_sys_abort( subname//' ERROR:: choices are constant, prognostic or diagnostic' )
   284)        end if
   285) 
   286)        ! Consistency settings for dynamic land use, etc.
   287) 
   288)        if (flanduse_timeseries /= ' ' .and. create_crop_landunit) &
   289)           call shr_sys_abort( subname//' ERROR:: dynamic landuse is currently not supported with create_crop_landunit option' )
   290)        if (create_crop_landunit .and. .not.allocate_all_vegpfts) &
   291)           call shr_sys_abort( subname//' ERROR:: maxpft<numpft+1 is currently not supported with create_crop_landunit option' )
   292)        if (flanduse_timeseries /= ' ') then
   293)           if (use_cndv) then
   294)              call shr_sys_abort( subname//' ERROR:: dynamic landuse is currently not supported with CNDV option' )
   295)           end if
   296)        end if
   297) 
   298)        ! Check on run type
   299)        if (nsrest == iundef) call shr_sys_abort( subname//' ERROR:: must set nsrest' )
   300)        if (nsrest == nsrBranch .and. nrevsn == ' ') &
   301)           call shr_sys_abort( subname//' ERROR: need to set restart data file name' )
   302) 
   303)        ! Model physics
   304) 
   305)        if ( (co2_ppmv <= 0.0_r8) .or. (co2_ppmv > 3000.0_r8) ) &
   306)           call shr_sys_abort( subname//' ERROR: co2_ppmv is out of a reasonable range' )
   307) 
   308)        if (nsrest == nsrStartup ) nrevsn = ' '
   309)        if (nsrest == nsrContinue) nrevsn = 'set by restart pointer file file'
   310)        if (nsrest /= nsrStartup .and. nsrest /= nsrContinue .and. nsrest /= nsrBranch ) &
   311)           call shr_sys_abort( subname//' ERROR: nsrest NOT set to a valid value' )
   312) 
   313)        if ( single_column .and. (scmlat == rundef  .or. scmlon == rundef ) ) &
   314)           call shr_sys_abort( subname//' ERROR:: single column mode on -- but scmlat and scmlon are NOT set' )
   315) 
   316)     endif   ! end of if-masterproc if-block
