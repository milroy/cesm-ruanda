    12)       subroutine lymana( o2col, secchi, dto2la, xso2la )
    13) !-----------------------------------------------------------------------------
    14) !   purpose:
    15) !   calculate the effective absorption cross section of o2 in the lyman-alpha
    16) !   bands and an effective o2 optical depth at all altitudes.  parameterized
    17) !   after:  chabrillat, s., and g. kockarts, simple parameterization of the
    18) !   absorption of the solar lyman-alpha line, geophysical research letters,
    19) !   vol.24, no.21, pp 2659-2662, 1997.
    20) !-----------------------------------------------------------------------------
    21) !   parameters:
    22) !   nz      - integer, number of specified altitude levels in the working (i)
    23) !             grid
    24) !   o2col   - real, slant overhead o2 column (molec/cc) at each specified (i)
    25) !             altitude
    26) !   dto2la  - real, optical depth due to o2 absorption at each specified  (o)
    27) !             vertical layer
    28) !   xso2la  - real, molecular absorption cross section in la bands        (o)
    29) !-----------------------------------------------------------------------------
    30) 
    31)       use mo_params
    32)       use ppgrid, only: pver, pverp
    33) 
    34)       implicit none
    35) 
    36) !-----------------------------------------------------------------------------
    37) !	... dummy arguments
    38) !-----------------------------------------------------------------------------
    39)       real(r8), intent(in)  :: o2col(pverp)
    40)       real(r8), intent(in)  :: secchi(pverp)
    41)       real(r8), intent(out) :: dto2la(pver,nla-1)
    42)       real(r8), intent(out) :: xso2la(pverp,nla-1)
    43) 
    44) !-----------------------------------------------------------------------------
    45) !	... local variables
    46) !-----------------------------------------------------------------------------
    47)       integer :: i, k, kp1
    48)       real(r8), dimension(pverp) :: rm, ro2
    49)       real(r8), save :: b(3), c(3), d(3), e(3)
    50) 
    51)       data b / 6.8431e-01_r8, 2.29841e-01_r8,  8.65412e-02_r8/, &
    52)            c /8.22114e-21_r8, 1.77556e-20_r8,  8.22112e-21_r8/, &
    53)            d / 6.0073e-21_r8, 4.28569e-21_r8,  1.28059e-20_r8/, &
    54)            e /8.21666e-21_r8, 1.63296e-20_r8,  4.85121e-17_r8/
    55) 
    56) !-----------------------------------------------------------------------------
    57) ! 	... calculate reduction factors at every altitude
    58) !-----------------------------------------------------------------------------
    59)       rm(:)  = 0._r8
    60)       ro2(:) = 0._r8
    61)       do k = 1,pverp
    62)         do i = 1,3
    63)           rm(k)  = rm(k) + b(i) * exp( -c(i)*o2col(k) )
    64)           ro2(k) = ro2(k) + d(i) * exp( -e(i)*o2col(k) )
    65)         end do
    66)       end do
    67) 
    68) !-----------------------------------------------------------------------------
    69) ! 	... calculate effective o2 optical depths and effective o2 cross sections
    70) !-----------------------------------------------------------------------------
    71)       do k = 1,pver
    72)          if( rm(k) > 1.e-100_r8 ) then
    73)             kp1 = k + 1
    74)             if( rm(kp1) > 0._r8 ) then
    75)                dto2la(k,1) = log( rm(kp1) )/secchi(kp1) - log( rm(k) )/secchi(k)
    76)             else
    77)                dto2la(k,1) = 1000._r8
    78)             end if
    79)          else
    80)             dto2la(k,1) = 1000._r8
    81)          end if
    82)       end do
    83)       do k = 1,pverp
    84)          if( rm(k) > 1.e-100_r8 ) then
    85)             if( ro2(k) > 1.e-100_r8 ) then
    86)                xso2la(k,1) = ro2(k)/rm(k)
    87)             else
    88)                xso2la(k,1) = 0._r8          
    89)             end if
    90)          else
    91)             xso2la(k,1) = 0._r8
    92)          end if
    93)       end do
    94) 
    95)       end subroutine lymana
    96) 
    97)       end module mo_lymana
