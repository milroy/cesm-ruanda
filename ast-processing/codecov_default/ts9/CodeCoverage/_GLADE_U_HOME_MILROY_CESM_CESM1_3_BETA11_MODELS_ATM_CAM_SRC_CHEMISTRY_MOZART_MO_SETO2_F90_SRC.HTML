    28)       subroutine o2_xsect_inti( o2_xsect_file )
    29) !-----------------------------------------------------------------------------
    30) !   purpose:
    31) !   compute equivalent optical depths for o2 absorption, parameterized in
    32) !   the sr bands and the lyman-alpha line.
    33) !-----------------------------------------------------------------------------
    34) !   parameters:
    35) !   nz      - integer, number of specified altitude levels in the working (i)
    36) !             grid
    37) !   z       - real(r8), specified altitude working grid (km)                  (i)
    38) !   nw      - integer, number of specified intervals + 1 in working       (i)
    39) !             wavelength grid
    40) !   wl      - real(r8), vector of lower limits of wavelength intervals in     (i)
    41) !             working wavelength grid
    42) !   cz      - real(r8), number of air molecules per cm^2 at each specified    (i)
    43) !             altitude layer
    44) !   zen     - real(r8), solar zenith angle                                    (i)
    45) !   dto2    - real(r8), optical depth due to o2 absorption at each specified  (o)
    46) !             vertical layer at each specified wavelength
    47) !   xso2    - real(r8), molecular absorption cross section in sr bands at     (o)
    48) !             each specified altitude and wavelength.  includes herzberg
    49) !             continuum.
    50) !-----------------------------------------------------------------------------
    51) 
    52)       use mo_params,     only : deltax
    53)       use mo_inter,      only : inter2
    54)       use mo_inter,      only : inter_inti
    55)       use mo_wavelen,    only : nw, wl
    56)       use ioFileMod,     only : getfil
    57)       use cam_pio_utils, only : cam_pio_openfile
    58)       use pio, only : file_desc_t, pio_inq_dimid, pio_inq_dimlen, pio_inq_varid, &
    59)            pio_get_var, pio_closefile, pio_nowrite
    60) 
    61)       implicit none
    62) 
    63) !-----------------------------------------------------------------------------
    64) !	... dummy arguments
    65) !-----------------------------------------------------------------------------
    66)       character(len=*), intent(in) :: o2_xsect_file
    67) 
    68) !-----------------------------------------------------------------------------
    69) !	... local variables
    70) !-----------------------------------------------------------------------------
    71)       type(file_desc_t) :: ncid
    72)       integer :: dimid
    73)       integer :: vid
    74)       integer :: astat
    75)       integer :: iret
    76)       integer :: i, wn, n
    77)       integer :: wrk_ind(4)
    78)       real(r8), allocatable :: x1(:)
    79)       real(r8), allocatable :: y1(:)
    80)       character(len=256) :: filespec
    81)       character(len=256) :: locfn
    82)       integer :: ierr
    83) !-----------------------------------------------------------------------------
    84) ! 	... cross section data for use outside the sr-bands (combined from
    85) !           brasseur and solomon and the jpl 1994 recommendation)
    86) !-----------------------------------------------------------------------------
    87) 
    88) !-----------------------------------------------------------------------------
    89) ! 	... read o2 cross section data outside sr-bands
    90) !-----------------------------------------------------------------------------
    91) !	... o2 absorption cross sections:
    92) !           from 116 nm to 245 nm, including schumann-runge continumm
    93) !           from brasseur and solomon 1986.
    94) !-----------------------------------------------------------------------------
    95)       filespec = trim( o2_xsect_file )
    96)       call getfil( filespec, locfn, 0 )
    97)       call cam_pio_openfile( ncid, trim( locfn ), PIO_NOWRITE )
    98) !---------------------------------------------------------------------------
    99) ! 	... get the dimensions
   100) !---------------------------------------------------------------------------
   101)       ierr = pio_inq_dimid( ncid, 'nosr', dimid )
   102)       ierr = pio_inq_dimlen( ncid, dimid, nsrc )
   103)       ierr = pio_inq_dimid( ncid, 'ngast', dimid )
   104)       ierr = pio_inq_dimlen( ncid, dimid, ngast )
   105)       ierr = pio_inq_dimid( ncid, 'nla', dimid )
   106)       ierr = pio_inq_dimlen( ncid, dimid, nla )
   107) !---------------------------------------------------------------------------
   108) ! 	... allocate arrays
   109) !---------------------------------------------------------------------------
   110)       allocate( wlint(nsrc), xso2int(nsrc), x1(nsrc), y1(nsrc), stat=astat )
   111)       if( astat /= 0 ) then
   112)          write(iulog,*) 'o2_xsect_inti: failed to allocate wlint ... y1; error = ',astat
   113)          call endrun
   114)       end if
   115)       allocate( wlgast(ngast), wlla(nla), stat=astat )
   116)       if( astat /= 0 ) then
   117)          write(iulog,*) 'o2_xsect_inti: failed to allocate wlgast, wlla; error = ',astat
   118)          call endrun
   119)       end if
   120) !---------------------------------------------------------------------------
   121) ! 	... read the wave bin coordinates
   122) !---------------------------------------------------------------------------
   123)       ierr = pio_inq_varid( ncid, 'wl_src', vid )
   124)       ierr = pio_get_var( ncid, vid, x1 )
   125)       ierr = pio_inq_varid( ncid, 'xs_src', vid )
   126)       ierr = pio_get_var( ncid, vid, y1 )
   127)       ierr = pio_inq_varid( ncid, 'wl_gast', vid )
   128)       ierr = pio_get_var( ncid, vid, wlgast )
   129)       ierr = pio_inq_varid( ncid, 'wl_lym', vid )
   130)       ierr = pio_get_var( ncid, vid, wlla )
   131)       call pio_closefile(  ncid )
   132) !-----------------------------------------------------------------------------
   133) ! 	... put together the internal grid by "pasting" the lyman-alpha grid and 
   134) !           kockarts grid into the combination of brasseur/solomon and jpl grid
   135) !-----------------------------------------------------------------------------
   136)       wlint(1:9) = x1(1:9)
   137)       nwint = 9
   138)       wlint(nwint+1:nwint+2) = wlla(1:2)
   139)       nwint = 11
   140)       wlint(nwint+1:nwint+36) = x1(12:47)
   141)       nwint = 47
   142)       wlint(nwint+1:nwint+ngast) = wlgast(1:ngast)
   143)       nwint = nwint + ngast
   144)       wlint(nwint+1:nwint+41) = x1(65:105)
   145)       nwint = nwint + 41
   146)       wrk_ind(1:4) = (/ nsrc, ngast, nla, nwint /)
   147) !-----------------------------------------------------------------------------
   148) ! 	... initialize interpolation module
   149) !-----------------------------------------------------------------------------
   150)       call inter_inti( nw+1, wl, nsrc, wlint )
   151) !-----------------------------------------------------------------------------
   152) ! 	... interpolate brasseur/solomon and jpl data onto internal grid
   153) !-----------------------------------------------------------------------------
   154)       call inter2( nsrc, wlint, xso2int, nsrc, x1, y1, iret )
   155)       deallocate( x1, y1 )
   156) 
   157) 
   158)       end subroutine o2_xsect_inti

   160)       subroutine set_o2_xsect( z, nw, wl, cz, &
   161)                                vcol, scol, dto2, xso2 )
   162) !-----------------------------------------------------------------------------
   163) !   purpose:
   164) !   compute equivalent optical depths for o2 absorption, parameterized in
   165) !   the sr bands and the lyman-alpha line.
   166) !-----------------------------------------------------------------------------
   167) !   parameters:
   168) !   nz      - integer, number of specified altitude levels in the working (i)
   169) !             grid
   170) !   z       - real(r8), specified altitude working grid (km)
   171) !   nw      - integer, number of specified intervals + 1 in working
   172) !             wavelength grid
   173) !   wl      - real(r8), vector of lower limits of wavelength intervals in
   174) !             working wavelength grid
   175) !   cz      - real(r8), number of air molecules per cm^2 at each specified
   176) !             altitude layer
   177) !   zen     - real(r8), solar zenith angle
   178) !   dto2    - real(r8), optical depth due to o2 absorption at each specified  (o)
   179) !             vertical layer at each specified wavelength
   180) !   xso2    - real(r8), molecular absorption cross section in sr bands at     (o)
   181) !             each specified altitude and wavelength.  includes herzberg
   182) !             continuum.
   183) !-----------------------------------------------------------------------------
   184) !   edit history:
   185) !   02/98  included lyman-alpha parameterization
   186) !   03/97  fix dto2 problem at top level (nz)
   187) !   02/97  changed offset for grid-end interpolation to relative number
   188) !          (x * (1 +- deltax))
   189) !   08/96  modified for early exit, no redundant read of data and smaller
   190) !          internal grid if possible;  internal grid uses user grid points
   191) !          whenever possible
   192) !   07/96  modified to work on internal grid and interpolate final values
   193) !          onto the user-defined grid
   194) !-----------------------------------------------------------------------------
   195) 
   196)       use mo_params,  only : kw
   197)       use mo_wavelen, only : delw_bin
   198)       use mo_inter,   only : inter3
   199)       use mo_schu,    only : schu
   200)       use mo_lymana,  only : lymana
   201)       use ppgrid,     only : pver, pverp
   202) 
   203)       implicit none
   204) 
   205) !-----------------------------------------------------------------------------
   206) !	... dummy arguments
   207) !-----------------------------------------------------------------------------
   208)       integer, intent(in)    :: nw
   209)       real(r8), intent(in)   :: wl(kw)
   210)       real(r8), intent(in)   :: cz(pverp)
   211)       real(r8), intent(in)   :: z(pverp)
   212)       real(r8), intent(in)   :: vcol(pverp)
   213)       real(r8), intent(in)   :: scol(pverp)
   214)       real(r8), intent(out)  :: dto2(pver,nw)
   215)       real(r8), intent(out)  :: xso2(nw,pverp)
   216) 
   217) !-----------------------------------------------------------------------------
   218) !	... local variables
   219) !-----------------------------------------------------------------------------
   220)       integer  :: wn, k, igast
   221)       integer  :: astat
   222)       real(r8) :: secchi(pverp)
   223) 
   224) !-----------------------------------------------------------------------------
   225) ! 	... o2 optical depth and equivalent cross section on kockarts grid
   226) !-----------------------------------------------------------------------------
   227)       real(r8), allocatable :: dto2k(:,:)
   228)       real(r8), allocatable :: xso2k(:,:)
   229) !-----------------------------------------------------------------------------
   230) ! 	... o2 optical depth and equivalent cross section in the lyman-alpha region
   231) !-----------------------------------------------------------------------------
   232)       real(r8), allocatable :: dto2la(:,:)
   233)       real(r8), allocatable :: xso2la(:,:)
   234) !-----------------------------------------------------------------------------
   235) ! 	... temporary one-dimensional storage for optical depth and cross section values
   236) !           xxtmp  - on internal grid
   237) !           xxuser - on user defined grid
   238) !-----------------------------------------------------------------------------
   239)       real(r8), dimension(2*kw) :: dttmp, xstmp
   240)       real(r8) :: dtuser(kw)
   241)       real(r8) :: xsuser(kw)
   242)       real(r8) :: o2col(pverp)
   243) 
   244)       real(r8) :: x, y
   245)       real(r8) :: delo2
   246) 
   247) !-----------------------------------------------------------------------------
   248) !	... allocate local variables
   249) !-----------------------------------------------------------------------------
   250)       allocate( dto2k(pver,ngast-1), xso2k(pverp,ngast-1), stat=astat )
   251)       if( astat /= 0 ) then
   252)          write(iulog,*) 'set_o2_xsect: failed to allocate dto2k,xso2k; error = ',astat
   253)          call endrun
   254)       end if
   255)       allocate( dto2la(pver,nla-1), xso2la(pverp,nla-1), stat=astat )
   256)       if( astat /= 0 ) then
   257)          write(iulog,*) 'set_o2_xsect: failed to allocate dto2k,xso2k; error = ',astat
   258)          call endrun
   259)       end if
   260) !-----------------------------------------------------------------------------
   261) ! 	... check, whether user grid is in the o2 absorption band at all...
   262) !           if not, set cross section and optical depth values to zero and return
   263) !-----------------------------------------------------------------------------
   264)       dto2(:pver,:nw)  = 0._r8
   265)       xso2(:nw,:pverp) = 0._r8
   266)       if( wl(1) > 243._r8 ) then
   267)          return
   268)       end if
   269) 
   270) !-----------------------------------------------------------------------------
   271) ! 	... sec xhi or chapman calculation
   272) !           for zen > 95 degrees, use zen = 95.  (this is only to compute effective o2
   273) !           cross sections. still, better than setting dto2 = 0. as was done up to 
   274) !           version 4.0) sm 1/2000
   275) !           in future could replace with mu2(iz) (but mu2 is also wavelength-depenedent)
   276) !           or imporved chapman function 
   277) !-----------------------------------------------------------------------------
   278) 
   279) !-----------------------------------------------------------------------------
   280) ! 	... slant o2 column 
   281) !-----------------------------------------------------------------------------
   282)       o2col(1:pverp) = 0.2095_r8 * scol(1:pverp)
   283) 
   284) !-----------------------------------------------------------------------------
   285) ! 	... effective secant of solar zenith angle.  use 2.0 if no direct sun. 
   286) !           for nz, use value at nz-1
   287) !-----------------------------------------------------------------------------
   288)       secchi(1:pver) = scol(1:pver)/vcol(1:pver)
   289)       where( secchi(1:pver) == 0._r8 )
   290)          secchi(1:pver) = 2._r8
   291)       endwhere
   292)       secchi(pverp) = secchi(pver)
   293) 
   294) !-----------------------------------------------------------------------------
   295) ! 	... if necessary:
   296) !           kockarts parameterization of the sr bands, output values of o2
   297) !           optical depth and o2 equivalent cross section are on his grid
   298) !-----------------------------------------------------------------------------
   299)       if( wl(1) < wlgast(ngast) .and. wl(nw+1) > wlgast(1) ) then
   300)            call schu( o2col, secchi, dto2k, xso2k )
   301)       else
   302)          dto2k(:,:) = 0._r8
   303)          xso2k(:,:) = 0._r8
   304)       end if
   305) 
   306) !-----------------------------------------------------------------------------
   307) ! 	... lyman-alpha parameterization, output values of o2 opticaldepth
   308) !           and o2 effective (equivalent) cross section
   309) !-----------------------------------------------------------------------------
   310)       if( wl(1) <= wlla(nla) .and. wl(nw+1) >= wlla(1) ) then
   311)          call lymana( o2col, secchi, dto2la, xso2la )
   312)       else
   313)          dto2la(:,:) = 0._r8
   314)          xso2la(:,:) = 0._r8
   315)       end if
   316) 
   317) !-----------------------------------------------------------------------------
   318) ! 	... loop through the altitude levels
   319) !-----------------------------------------------------------------------------
   320) level_loop : &
   321)       do k = 1,pverp
   322)          igast = 0
   323) !-----------------------------------------------------------------------------
   324) ! 	... loop through the internal wavelength grid
   325) !-----------------------------------------------------------------------------
   326)          do wn = 1,nwint-1
   327) !-----------------------------------------------------------------------------
   328) ! 	... if outside kockarts grid and outside lyman-alpha, use the 
   329) !           jpl/brasseur+solomon data, if inside
   330) !           kockarts grid, use the parameterized values from the call to schu,
   331) !           if inside lyman-alpha, use the paraemterized values from call to lymana
   332) !-----------------------------------------------------------------------------
   333)             if( wlint(wn+1) <= wlgast(1) .or. wlint(wn) >= wlgast(ngast) ) then
   334)               if( wlint(wn+1) <= wlla(1) .or. wlint(wn) >= wlla(nla) ) then
   335)                  xstmp(wn) = xso2int(wn)
   336)               else
   337)                  xstmp(wn) = xso2la(k,1)
   338)               end if
   339)             else
   340)                igast = igast + 1
   341)                xstmp(wn) = xso2k(k,igast)
   342)             end if
   343) !-----------------------------------------------------------------------------
   344) ! 	... compute the area in each bin (for correct interpolation purposes only!)
   345) !-----------------------------------------------------------------------------
   346)             xstmp(wn) = xstmp(wn) * (wlint(wn+1) - wlint(wn))
   347)          end do
   348) !-----------------------------------------------------------------------------
   349) ! 	... interpolate o2 cross section from the internal grid onto the user grid
   350) !-----------------------------------------------------------------------------
   351)          call inter3( nw+1, wl, xsuser, nwint, wlint, xstmp )
   352)          xso2(:nw,k) = xsuser(:nw) * delw_bin(:nw)
   353)       end do level_loop
   354) 
   355)       do k = 1,pver
   356)          igast = 0
   357)          delo2 = .2095_r8 * cz(k)    ! vertical o2 column
   358) !-----------------------------------------------------------------------------
   359) ! 	... loop through the internal wavelength grid
   360) !-----------------------------------------------------------------------------
   361)          do wn = 1,nwint-1
   362) !-----------------------------------------------------------------------------
   363) ! 	... if outside kockarts grid and outside lyman-alpha, use the 
   364) !           jpl/brasseur+solomon data, if inside
   365) !           kockarts grid, use the parameterized values from the call to schu,
   366) !           if inside lyman-alpha, use the paraemterized values from call to lymana
   367) !-----------------------------------------------------------------------------
   368)             if( wlint(wn+1) <= wlgast(1) .or. wlint(wn) >= wlgast(ngast) ) then
   369)               if( wlint(wn+1) <= wlla(1) .or. wlint(wn) >= wlla(nla) ) then
   370)                  dttmp(wn) = xso2int(wn) * delo2
   371)               else
   372)                  dttmp(wn) = dto2la(k,1)
   373)               end if
   374)             else
   375)                igast = igast + 1
   376)                dttmp(wn) = dto2k(k,igast)
   377)             end if
   378) !-----------------------------------------------------------------------------
   379) ! 	... compute the area in each bin (for correct interpolation purposes only!)
   380) !-----------------------------------------------------------------------------
   381)             dttmp(wn) = dttmp(wn) * (wlint(wn+1) - wlint(wn))
   382)          end do
   383) !-----------------------------------------------------------------------------
   384) ! 	... interpolate o2 optical depth from the internal grid onto the user grid
   385) !-----------------------------------------------------------------------------
   386)          call inter3( nw+1, wl, dtuser, nwint, wlint, dttmp )
   387)          dto2(k,:nw) = dtuser(:nw) * delw_bin(:nw)
   388)       end do
   389) 
   390)       deallocate( dto2k, xso2k, dto2la, xso2la )
   391) 
   392)       end subroutine set_o2_xsect
