    62)   subroutine setinv( invariants, tfld, h2ovmr, vmr, pmid, ncol, lchnk, pbuf )
    63)     !-----------------------------------------------------------------
    64)     !        ... set the invariant densities (molecules/cm**3)
    65)     !-----------------------------------------------------------------
    66) 
    67)     use mo_constants,  only : boltz_cgs
    68)     use tracer_cnst,   only : num_tracer_cnst, tracer_cnst_flds, get_cnst_data
    69)     use mo_chem_utls,  only : get_inv_ndx
    70)     use physics_buffer, only : physics_buffer_desc
    71) 
    72)     implicit none
    73) 
    74)     !-----------------------------------------------------------------
    75)     !        ... dummy arguments
    76)     !-----------------------------------------------------------------
    77)     integer,  intent(in)  ::      ncol                      ! chunk column count
    78)     real(r8), intent(in)  ::      tfld(pcols,pver)          ! temperature
    79)     real(r8), intent(in)  ::      h2ovmr(ncol,pver)         ! water vapor vmr
    80)     real(r8), intent(in)  ::      pmid(pcols,pver)          ! pressure
    81)     integer,  intent(in)  ::      lchnk                     ! chunk number
    82)     real(r8), intent(in)  ::      vmr(ncol,pver,gas_pcnst)  ! vmr
    83)     real(r8), intent(out) ::      invariants(ncol,pver,nfs) ! invariant array
    84)     type(physics_buffer_desc), pointer :: pbuf(:)
    85) 
    86) 
    87)     real(r8) :: cnst_offline( ncol, pver )
    88) 
    89)     !-----------------------------------------------------------------
    90)     !        .. local variables
    91)     !-----------------------------------------------------------------
    92)     integer :: k, i, ndx
    93)     real(r8), parameter ::  Pa_xfac = 10._r8                 ! Pascals to dyne/cm^2
    94)     real(r8) :: sum1(ncol)
    95)     real(r8) :: tmp_out(ncol,pver)
    96) 
    97)     !-----------------------------------------------------------------
    98)     !        note: invariants are in cgs density units.
    99)     !              the pmid array is in pascals and must be
   100)     !	       mutiplied by 10. to yield dynes/cm**2.
   101)     !-----------------------------------------------------------------
   102)     invariants(:,:,:) = 0._r8
   103)     !-----------------------------------------------------------------
   104)     !	... set m, n2, o2, and h2o densities
   105)     !-----------------------------------------------------------------
   106)     do k = 1,pver
   107)        invariants(:ncol,k,m_ndx) = Pa_xfac * pmid(:ncol,k) / (boltz_cgs*tfld(:ncol,k))
   108)     end do
   109) 
   110)     if( has_n2 ) then
   111)        if ( has_var_o2 ) then
   112)           do k = 1,pver
   113)              sum1(:ncol) = (vmr(:ncol,k,id_o) + vmr(:ncol,k,id_o2) + vmr(:ncol,k,id_h))
   114)              invariants(:ncol,k,n2_ndx) = (1._r8 - sum1(:)) * invariants(:ncol,k,m_ndx)
   115)           end do
   116)        else
   117)           do k = 1,pver
   118)              invariants(:ncol,k,n2_ndx) = .79_r8 * invariants(:ncol,k,m_ndx)
   119)           end do
   120)        endif
   121)     end if
   122)     if( has_o2 ) then
   123)        do k = 1,pver
   124)           invariants(:ncol,k,o2_ndx) = .21_r8 * invariants(:ncol,k,m_ndx)
   125)        end do
   126)     end if
   127)     if( has_h2o ) then
   128)        do k = 1,pver
   129)           invariants(:ncol,k,h2o_ndx) = h2ovmr(:ncol,k) * invariants(:ncol,k,m_ndx)
   130)        end do
   131)     end if
   132) 
   133)     do i = 1,num_tracer_cnst
   134) 
   135)        call get_cnst_data( tracer_cnst_flds(i), cnst_offline,  ncol, lchnk, pbuf )
   136)        ndx =  get_inv_ndx( tracer_cnst_flds(i) )
   137) 
   138)        do k = 1,pver
   139)           invariants(:ncol,k,ndx) = cnst_offline(:ncol,k)*invariants(:ncol,k,m_ndx)
   140)        enddo
   141) 
   142)     enddo
   143) 
   144)     do i = 1,nfs
   145)       tmp_out(:ncol,:) =  invariants(:ncol,:,i) 
   146)       call outfld( trim(inv_lst(i))//'_dens', tmp_out(:ncol,:), ncol, lchnk )
   147)       tmp_out(:ncol,:) =  invariants(:ncol,:,i) / invariants(:ncol,:,m_ndx)
   148)       call outfld( trim(inv_lst(i))//'_vmr',  tmp_out(:ncol,:), ncol, lchnk )
   149)     enddo
   150) 
   151)   end subroutine setinv
