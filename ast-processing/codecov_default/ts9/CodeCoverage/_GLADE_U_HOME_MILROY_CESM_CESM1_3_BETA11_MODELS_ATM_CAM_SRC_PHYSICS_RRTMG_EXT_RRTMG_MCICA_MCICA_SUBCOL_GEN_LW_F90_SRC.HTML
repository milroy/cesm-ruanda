    19) ! Purpose: Create McICA stochastic arrays for cloud physical or optical properties.
    20) ! Two options are possible:
    21) ! 1) Input cloud physical properties: cloud fraction, ice and liquid water
    22) !    paths, ice fraction, and particle sizes.  Output will be stochastic
    23) !    arrays of these variables.  (inflag = 1)
    24) ! 2) Input cloud optical properties directly: cloud optical depth, single
    25) !    scattering albedo and asymmetry parameter.  Output will be stochastic
    26) !    arrays of these variables.  (inflag = 0; longwave scattering is not
    27) !    yet available, ssac and asmc are for future expansion)
    28)

    42)

    52)       subroutine mcica_subcol_lw(lchnk, ncol, nlay, icld, permuteseed, play, &
    53)                        cldfrac, ciwp, clwp, rei, rel, tauc, cldfmcl, &
    54)                        ciwpmcl, clwpmcl, reicmcl, relqmcl, taucmcl)
    55) 
    56) ! ----- Input -----
    57) ! Control
    58)       integer, intent(in) :: lchnk           ! chunk identifier
    59)       integer, intent(in) :: ncol            ! number of columns
    60)       integer, intent(in) :: nlay            ! number of model layers
    61)       integer, intent(in) :: icld            ! clear/cloud, cloud overlap flag
    62)       integer, intent(in) :: permuteseed     ! if the cloud generator is called multiple times, 
    63)                                                         ! permute the seed between each call.
    64)                                                         ! between calls for LW and SW, recommended
    65)                                                         ! permuteseed differes by 'ngpt'
    66) 
    67) ! Atmosphere
    68)       real(kind=r8), intent(in) :: play(:,:)          ! layer pressures (mb) 
    69)                                                         !    Dimensions: (ncol,nlay)
    70) 
    71) ! Atmosphere/clouds - cldprop
    72)       real(kind=r8), intent(in) :: cldfrac(:,:)       ! layer cloud fraction
    73)                                                         !    Dimensions: (ncol,nlay)
    74)       real(kind=r8), intent(in) :: tauc(:,:,:)        ! cloud optical depth
    75)                                                         !    Dimensions: (nbndlw,ncol,nlay)
    76) !      real(kind=r8), intent(in) :: ssac(:,:,:)       ! cloud single scattering albedo
    77)                                                         !    Dimensions: (nbndlw,ncol,nlay)
    78) !      real(kind=r8), intent(in) :: asmc(:,:,:)       ! cloud asymmetry parameter
    79)                                                         !    Dimensions: (nbndlw,ncol,nlay)
    80)       real(kind=r8), intent(in) :: ciwp(:,:)          ! cloud ice water path
    81)                                                         !    Dimensions: (ncol,nlay)
    82)       real(kind=r8), intent(in) :: clwp(:,:)          ! cloud liquid water path
    83)                                                         !    Dimensions: (ncol,nlay)
    84)       real(kind=r8), intent(in) :: rei(:,:)           ! cloud ice particle size
    85)                                                         !    Dimensions: (ncol,nlay)
    86)       real(kind=r8), intent(in) :: rel(:,:)           ! cloud liquid particle size
    87)                                                         !    Dimensions: (ncol,nlay)
    88) 
    89) ! ----- Output -----
    90) ! Atmosphere/clouds - cldprmc [mcica]
    91)       real(kind=r8), intent(out) :: cldfmcl(:,:,:)    ! cloud fraction [mcica]
    92)                                                         !    Dimensions: (ngptlw,ncol,nlay)
    93)       real(kind=r8), intent(out) :: ciwpmcl(:,:,:)    ! cloud ice water path [mcica]
    94)                                                         !    Dimensions: (ngptlw,ncol,nlay)
    95)       real(kind=r8), intent(out) :: clwpmcl(:,:,:)    ! cloud liquid water path [mcica]
    96)                                                         !    Dimensions: (ngptlw,ncol,nlay)
    97)       real(kind=r8), intent(out) :: relqmcl(:,:)      ! liquid particle size (microns)
    98)                                                         !    Dimensions: (ncol,nlay)
    99)       real(kind=r8), intent(out) :: reicmcl(:,:)      ! ice partcle size (microns)
   100)                                                         !    Dimensions: (ncol,nlay)
   101)       real(kind=r8), intent(out) :: taucmcl(:,:,:)    ! cloud optical depth [mcica]
   102)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   103) !      real(kind=r8), intent(out) :: ssacmcl(:,:,:)   ! cloud single scattering albedo [mcica]
   104)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   105) !      real(kind=r8), intent(out) :: asmcmcl(:,:,:)   ! cloud asymmetry parameter [mcica]
   106)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   107) 
   108) ! ----- Local -----
   109) 
   110) ! Stochastic cloud generator variables [mcica]
   111)       integer, parameter :: nsubclw = ngptlw ! number of sub-columns (g-point intervals)
   112)       integer :: km, im, nm                  ! loop indices
   113) 
   114)       real(kind=r8) :: pmid(ncol, nlay)               ! layer pressures (Pa) 
   115) !      real(kind=r8) :: pdel(ncol, nlay)              ! layer pressure thickness (Pa) 
   116) !      real(kind=r8) :: qi(ncol, nlay)                ! ice water (specific humidity)
   117) !      real(kind=r8) :: ql(ncol, nlay)                ! liq water (specific humidity)
   118) 
   119) 
   120) ! Return if clear sky; or stop if icld out of range
   121)       if (icld.eq.0) return
   122)       if (icld.lt.0.or.icld.gt.3) then 
   123)          call endrun('MCICA_SUBCOL: INVALID ICLD')
   124)       endif 
   125) 
   126) ! NOTE: For GCM mode, permuteseed must be offset between LW and SW by at least the number of subcolumns
   127) 
   128) 
   129) ! Pass particle sizes to new arrays, no subcolumns for these properties yet
   130) ! Convert pressures from mb to Pa
   131) 
   132)       reicmcl(:ncol,:nlay) = rei(:ncol,:nlay)
   133)       relqmcl(:ncol,:nlay) = rel(:ncol,:nlay)
   134)       pmid(:ncol,:nlay)    = play(:ncol,:nlay)*1.e2_r8
   135) 
   136) ! Convert input ice and liquid cloud water paths to specific humidity ice and liquid components 
   137) 
   138) !      cwp =  (q * pdel * 1000.) / gravit)
   139) !           = (kg/kg * kg m-1 s-2 *1000.) / m s-2
   140) !           = (g m-2)
   141) !
   142) !      q  = (cwp * gravit) / (pdel *1000.)
   143) !         = (g m-2 * m s-2) / (kg m-1 s-2 * 1000.)
   144) !         =  kg/kg
   145) 
   146) !      do km = 1, nlay
   147) !         qi(km) = (ciwp(km) * grav) / (pdel(km) * 1000._r8)
   148) !         ql(km) = (clwp(km) * grav) / (pdel(km) * 1000._r8)
   149) !      enddo
   150) 
   151) !  Generate the stochastic subcolumns of cloud optical properties for the longwave;
   152)       call generate_stochastic_clouds (ncol, nlay, nsubclw, icld, pmid, cldfrac, clwp, ciwp, tauc, &
   153)                                cldfmcl, clwpmcl, ciwpmcl, taucmcl, permuteseed)
   154) 
   155)       end subroutine mcica_subcol_lw

   158) !-------------------------------------------------------------------------------------------------
   159)       subroutine generate_stochastic_clouds(ncol, nlay, nsubcol, icld, pmid, cld, clwp, ciwp, tauc, &
   160)                                    cld_stoch, clwp_stoch, ciwp_stoch, tauc_stoch, changeSeed) 
   161) !-------------------------------------------------------------------------------------------------
   162) 
   163)   !----------------------------------------------------------------------------------------------------------------
   164)   ! ---------------------
   165)   ! Contact: Cecile Hannay (hannay@ucar.edu)
   166)   ! 
   167)   ! Original code: Based on Raisanen et al., QJRMS, 2004.
   168)   ! 
   169)   ! Modifications: Generalized for use with RRTMG and added Mersenne Twister as the default
   170)   !   random number generator, which can be changed to the optional kissvec random number generator
   171)   !   with flag 'irnd' below. Some extra functionality has been commented or removed.  
   172)   !   Michael J. Iacono, AER, Inc., February 2007
   173)   !
   174)   ! Given a profile of cloud fraction, cloud water and cloud ice, we produce a set of subcolumns.
   175)   ! Each layer within each subcolumn is homogeneous, with cloud fraction equal to zero or one 
   176)   ! and uniform cloud liquid and cloud ice concentration.
   177)   ! The ensemble as a whole reproduces the probability function of cloud liquid and ice within each layer 
   178)   ! and obeys an overlap assumption in the vertical.   
   179)   ! 
   180)   ! Overlap assumption:
   181)   !  The cloud are consistent with 4 overlap assumptions: random, maximum, maximum-random and exponential. 
   182)   !  The default option is maximum-random (option 3)
   183)   !  The options are: 1=random overlap, 2=max/random, 3=maximum overlap, 4=exponential overlap
   184)   !  This is set with the variable "overlap" 
   185)   !mji - Exponential overlap option (overlap=4) has been deactivated in this version
   186)   !  The exponential overlap uses also a length scale, Zo. (real,    parameter  :: Zo = 2500. ) 
   187)   ! 
   188)   ! Seed:
   189)   !  If the stochastic cloud generator is called several times during the same timestep, 
   190)   !  one should change the seed between the call to insure that the subcolumns are different.
   191)   !  This is done by changing the argument 'changeSeed'
   192)   !  For example, if one wants to create a set of columns for the shortwave and another set for the longwave ,
   193)   !  use 'changeSeed = 1' for the first call and'changeSeed = 2' for the second call 
   194)   !
   195)   ! PDF assumption:
   196)   !  We can use arbitrary complicated PDFS. 
   197)   !  In the present version, we produce homogeneuous clouds (the simplest case).  
   198)   !  Future developments include using the PDF scheme of Ben Johnson. 
   199)   !
   200)   ! History file:
   201)   !  Option to add diagnostics variables in the history file. (using FINCL in the namelist)
   202)   !  nsubcol = number of subcolumns
   203)   !  overlap = overlap type (1-3)
   204)   !  Zo = length scale 
   205)   !  CLOUD_S = mean of the subcolumn cloud fraction ('_S" means Stochastic)
   206)   !  CLDLIQ_S = mean of the subcolumn cloud water
   207)   !  CLDICE_S = mean of the subcolumn cloud ice 
   208)   !
   209)   ! Note:
   210)   !   Here: we force that the cloud condensate to be consistent with the cloud fraction 
   211)   !   i.e we only have cloud condensate when the cell is cloudy. 
   212)   !   In CAM: The cloud condensate and the cloud fraction are obtained from 2 different equations 
   213)   !   and the 2 quantities can be inconsistent (i.e. CAM can produce cloud fraction 
   214)   !   without cloud condensate or the opposite).
   215)   !---------------------------------------------------------------------------------------------------------------
   216) 
   217)       use mcica_random_numbers
   218) ! The Mersenne Twister random number engine
   219)       use MersenneTwister, only: randomNumberSequence, &   
   220)                                  new_RandomNumberSequence, getRandomReal
   221) 
   222)       type(randomNumberSequence) :: randomNumbers
   223) 
   224) ! -- Arguments
   225) 
   226)       integer, intent(in) :: ncol            ! number of columns
   227)       integer, intent(in) :: nlay            ! number of layers
   228)       integer, intent(in) :: icld            ! clear/cloud, cloud overlap flag
   229)       integer, intent(in) :: nsubcol         ! number of sub-columns (g-point intervals)
   230)       integer, optional, intent(in) :: changeSeed     ! allows permuting seed
   231) 
   232) ! Column state (cloud fraction, cloud water, cloud ice) + variables needed to read physics state 
   233)       real(kind=r8), intent(in) :: pmid(:,:)          ! layer pressure (Pa)
   234)                                                         !    Dimensions: (ncol,nlay)
   235)       real(kind=r8), intent(in) :: cld(:,:)           ! cloud fraction 
   236)                                                         !    Dimensions: (ncol,nlay)
   237)       real(kind=r8), intent(in) :: clwp(:,:)          ! cloud liquid water path
   238)                                                         !    Dimensions: (ncol,nlay)
   239)       real(kind=r8), intent(in) :: ciwp(:,:)          ! cloud ice water path
   240)                                                         !    Dimensions: (ncol,nlay)
   241)       real(kind=r8), intent(in) :: tauc(:,:,:)        ! cloud optical depth
   242)                                                         !    Dimensions: (nbndlw,ncol,nlay)
   243) !      real(kind=r8), intent(in) :: ssac(:,:,:)       ! cloud single scattering albedo
   244)                                                         !    Dimensions: (nbndlw,ncol,nlay)
   245)                                                         !   inactive - for future expansion
   246) !      real(kind=r8), intent(in) :: asmc(:,:,:)       ! cloud asymmetry parameter
   247)                                                         !    Dimensions: (nbndlw,ncol,nlay)
   248)                                                         !   inactive - for future expansion
   249) 
   250)       real(kind=r8), intent(out) :: cld_stoch(:,:,:)  ! subcolumn cloud fraction 
   251)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   252)       real(kind=r8), intent(out) :: clwp_stoch(:,:,:) ! subcolumn cloud liquid water path
   253)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   254)       real(kind=r8), intent(out) :: ciwp_stoch(:,:,:) ! subcolumn cloud ice water path
   255)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   256)       real(kind=r8), intent(out) :: tauc_stoch(:,:,:) ! subcolumn cloud optical depth
   257)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   258) !      real(kind=r8), intent(out) :: ssac_stoch(:,:,:)! subcolumn cloud single scattering albedo
   259)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   260)                                                         !   inactive - for future expansion
   261) !      real(kind=r8), intent(out) :: asmc_stoch(:,:,:)! subcolumn cloud asymmetry parameter
   262)                                                         !    Dimensions: (ngptlw,ncol,nlay)
   263)                                                         !   inactive - for future expansion
   264) 
   265) ! -- Local variables
   266)       real(kind=r8) :: cldf(ncol,nlay)                ! cloud fraction 
   267)     
   268) ! Mean over the subcolumns (cloud fraction, cloud water , cloud ice) - inactive
   269) !      real(kind=r8) :: mean_cld_stoch(ncol, nlay)    ! cloud fraction 
   270) !      real(kind=r8) :: mean_clwp_stoch(ncol, nlay)   ! cloud water
   271) !      real(kind=r8) :: mean_ciwp_stoch(ncol, nlay)   ! cloud ice
   272) !      real(kind=r8) :: mean_tauc_stoch(ncol, nlay)   ! cloud optical depth
   273) !      real(kind=r8) :: mean_ssac_stoch(ncol, nlay)   ! cloud single scattering albedo
   274) !      real(kind=r8) :: mean_asmc_stoch(ncol, nlay)   ! cloud asymmetry parameter
   275) 
   276) ! Set overlap
   277)       integer :: overlap                     ! 1 = random overlap, 2 = maximum/random,
   278)                                                         ! 3 = maximum overlap, 
   279) !      real(kind=r8), parameter  :: Zo = 2500._r8   ! length scale (m) 
   280) !      real(kind=r8) :: zm(ncol,nlay)                 ! Height of midpoints (above surface)
   281) !      real(kind=r8), dimension(nlay) :: alpha=0.0_r8    ! overlap parameter  
   282) 
   283) ! Constants (min value for cloud fraction and cloud water and ice)
   284)       real(kind=r8), parameter :: cldmin = 1.0e-80_r8     ! min cloud fraction
   285) !      real(kind=r8), parameter :: qmin   = 1.0e-10_r8   ! min cloud water and cloud ice (not used)
   286) 
   287) ! Variables related to random number and seed 
   288)       integer :: irnd                        ! flag for random number generator
   289)                                                         !  0 = kissvec
   290)                                                         !  1 = Mersenne Twister
   291) 
   292)       real(kind=r8), dimension(nsubcol, ncol, nlay) :: CDF, CDF2      ! random numbers
   293)       integer, dimension(ncol) :: seed1, seed2, seed3, seed4 ! seed to create random number (kissvec)
   294)       real(kind=r8), dimension(ncol) :: rand_num      ! random number (kissvec)
   295)       integer :: iseed                       ! seed to create random number (Mersenne Teister)
   296)       real(kind=r8) :: rand_num_mt                    ! random number (Mersenne Twister)
   297) 
   298) ! Flag to identify cloud fraction in subcolumns
   299)       logical,  dimension(nsubcol, ncol, nlay) :: iscloudy   ! flag that says whether a gridbox is cloudy
   300) 
   301) ! Indices
   302)       integer :: ilev, isubcol, i, n         ! indices
   303) 
   304) !------------------------------------------------------------------------------------------ 
   305) 
   306) ! Set randum number generator to use (0 = kissvec; 1 = mersennetwister)
   307)       irnd = 0
   308) !      irnd = 1
   309) 
   310) ! Pass input cloud overlap setting to local variable
   311)       overlap = icld
   312) 
   313) ! ensure that cloud fractions are in bounds 
   314)       cldf(:,:) = cld(:ncol,:nlay)
   315)       where (cldf(:,:) < cldmin)
   316)           cldf(:,:) = 0._r8
   317)       end where
   318) 
   319) ! ----- Create seed  --------
   320)    
   321) ! Advance randum number generator by changeseed values
   322)       if (irnd.eq.0) then   
   323) ! For kissvec, create a seed that depends on the state of the columns. Maybe not the best way, but it works.  
   324) ! Must use pmid from bottom four layers. 
   325)          do i=1,ncol
   326)             if (pmid(i,nlay).lt.pmid(i,nlay-1)) then 
   327)                call endrun('MCICA_SUBCOL: KISSVEC SEED GENERATOR REQUIRES PMID FROM BOTTOM FOUR LAYERS.')
   328)             endif 
   329)             seed1(i) = (pmid(i,nlay) - int(pmid(i,nlay)))  * 1000000000
   330)             seed2(i) = (pmid(i,nlay-1) - int(pmid(i,nlay-1)))  * 1000000000
   331)             seed3(i) = (pmid(i,nlay-2) - int(pmid(i,nlay-2)))  * 1000000000
   332)             seed4(i) = (pmid(i,nlay-3) - int(pmid(i,nlay-3)))  * 1000000000
   333)           enddo
   334)          do i=1,changeSeed
   335)             call kissvec(seed1, seed2, seed3, seed4, rand_num)
   336)          enddo
   337)       elseif (irnd.eq.1) then
   338)          randomNumbers = new_RandomNumberSequence(seed = changeSeed)
   339)       endif 
   340) 
   341) 
   342) ! ------ Apply overlap assumption --------
   343) 
   344) ! generate the random numbers  
   345) 
   346)       select case (overlap)
   347) 
   348)       case(1) 
   349) ! Random overlap
   350) ! i) pick a random value at every level
   351)   
   352)          if (irnd.eq.0) then 
   353)             do isubcol = 1,nsubcol
   354)                do ilev = 1,nlay
   355)                   call kissvec(seed1, seed2, seed3, seed4, rand_num)  ! we get different random number for each level
   356)                   CDF(isubcol,:,ilev) = rand_num
   357)                enddo
   358)             enddo
   359)          elseif (irnd.eq.1) then
   360)             do isubcol = 1, nsubcol
   361)                do i = 1, ncol
   362)                   do ilev = 1, nlay
   363)                      rand_num_mt = getRandomReal(randomNumbers)
   364)                      CDF(isubcol,i,ilev) = rand_num_mt
   365)                   enddo
   366)                enddo
   367)              enddo
   368)          endif
   369) 
   370)       case(2) 
   371) ! Maximum-Random overlap
   372) ! i) pick a random number for top layer.
   373) ! ii) walk down the column: 
   374) !    - if the layer above is cloudy, we use the same random number than in the layer above
   375) !    - if the layer above is clear, we use a new random number 
   376) 
   377)          if (irnd.eq.0) then 
   378)             do isubcol = 1,nsubcol
   379)                do ilev = 1,nlay
   380)                   call kissvec(seed1, seed2, seed3, seed4, rand_num) 
   381)                   CDF(isubcol,:,ilev) = rand_num
   382)                enddo
   383)             enddo
   384)          elseif (irnd.eq.1) then
   385)             do isubcol = 1, nsubcol
   386)                do i = 1, ncol
   387)                   do ilev = 1, nlay
   388)                      rand_num_mt = getRandomReal(randomNumbers)
   389)                      CDF(isubcol,i,ilev) = rand_num_mt
   390)                   enddo
   391)                enddo
   392)              enddo
   393)          endif
   394) 
   395)          do ilev = 2,nlay
   396)             do i = 1, ncol
   397)                do isubcol = 1, nsubcol
   398)                   if (CDF(isubcol, i, ilev-1) > 1._r8 - cldf(i,ilev-1) ) then
   399)                      CDF(isubcol,i,ilev) = CDF(isubcol,i,ilev-1) 
   400)                   else
   401)                      CDF(isubcol,i,ilev) = CDF(isubcol,i,ilev) * (1._r8 - cldf(i,ilev-1))
   402)                   end if
   403)                end do
   404)             end do
   405)          enddo
   406)        
   407)       case(3) 
   408) ! Maximum overlap
   409) ! i) pick the same random numebr at every level  
   410) 
   411)          if (irnd.eq.0) then 
   412)             do isubcol = 1,nsubcol
   413)                call kissvec(seed1, seed2, seed3, seed4, rand_num)
   414)                do ilev = 1,nlay
   415)                   CDF(isubcol,:,ilev) = rand_num
   416)                enddo
   417)             enddo
   418)          elseif (irnd.eq.1) then
   419)             do isubcol = 1, nsubcol
   420)                do i = 1, ncol
   421)                   rand_num_mt = getRandomReal(randomNumbers)
   422)                   do ilev = 1, nlay
   423)                      CDF(isubcol,i,ilev) = rand_num_mt
   424)                   enddo
   425)                enddo
   426)              enddo
   427)          endif
   428) 
   429) !    case(4) - inactive
   430) !       ! Exponential overlap: weighting between maximum and random overlap increases with the distance. 
   431) !       ! The random numbers for exponential overlap verify:
   432) !       ! j=1   RAN(j)=RND1
   433) !       ! j>1   if RND1 < alpha(j,j-1) => RAN(j) = RAN(j-1)
   434) !       !                                 RAN(j) = RND2
   435) !       ! alpha is obtained from the equation
   436) !       ! alpha = exp(- (Zi-Zj-1)/Zo) where Zo is a characteristic length scale    
   437) 
   438) 
   439) !       ! compute alpha
   440) !       zm    = state%zm     
   441) !       alpha(:, 1) = 0.
   442) !       do ilev = 2,nlay
   443) !          alpha(:, ilev) = exp( -( zm (:, ilev-1) -  zm (:, ilev)) / Zo)
   444) !       end do
   445)        
   446) !       ! generate 2 streams of random numbers
   447) !       do isubcol = 1,nsubcol

   577)       do i = 1, size(ran_arr)

   579)          seed1(i) = low_byte(kiss)
   580)          seed2(i) = m (m (m (seed2(i), 13), - 17), 5)
