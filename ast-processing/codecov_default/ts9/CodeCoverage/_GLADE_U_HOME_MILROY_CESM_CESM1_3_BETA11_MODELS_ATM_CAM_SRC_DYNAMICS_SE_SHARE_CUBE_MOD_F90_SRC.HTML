   210)   ! metric_atomic:
   211)   !
   212)   ! Initialize cube-sphere metric terms:
   213)   ! equal angular elements (atomic)
   214)   ! initialize:  
   215)   !         metdet, rmetdet  (analytic)    = detD, 1/detD
   216)   !         met                (analytic)    D^t D     (symmetric)
   217)   !         metdet             (analytic)    = detD
   218)   !         metinv             (analytic)    Dinv Dinv^t  (symmetic)
   219)   !         D     (from subroutine vmap)
   220)   !         Dinv  (computed directly from D)
   221)   ! 
   222)   ! ucontra = Dinv * u  =  metinv * ucov   
   223)   ! ucov    = D^t * u   =  met * ucontra
   224)   !
   225)   ! we also compute DE = D*E, where 
   226)   ! E = eigenvectors of metinv as a basis      metinv = E LAMBDA E^t
   227)   !   
   228)   ! ueig = E^t ucov  = E^t D^t u =  (DE)^t u  
   229)   !  
   230)   !
   231)   ! so if we want to tweak the mapping by a factor alpha (so he weights add up to 4pi, for example)
   232)   ! we take:
   233)   !    NEW       OLD     
   234)   !       D = sqrt(alpha) D  and then rederive all quantities.  
   235)   !    detD = alpha detD
   236)   !    
   237)   ! where alpha = 4pi/SEMarea, SEMarea = global sum elem(ie)%mv(i,j)*elem(ie)%metdet(i,j)
   238)   ! 
   239)   ! =========================================
   240) 
   241)   subroutine metric_atomic(elem,gll_points,alpha)
   242)     use element_mod, only : element_t
   243)     use dimensions_mod, only : np
   244)     use physical_constants, only : rrearth
   245) 
   246)     type (element_t) :: elem
   247)     real(kind=real_kind) :: alpha
   248)     real (kind=longdouble_kind)      :: gll_points(np)
   249)     ! Local variables
   250)     integer ii
   251)     integer i,j,nn
   252)     integer iptr
   253) 
   254)     real (kind=real_kind) :: r         ! distance from origin for point on cube tangent to unit sphere
   255) 
   256)     real (kind=real_kind) :: const, norm
   257)     real (kind=real_kind) :: detD      ! determinant of vector field mapping matrix.  
   258) 
   259)     real (kind=real_kind) :: x1        ! 1st cube face coordinate
   260)     real (kind=real_kind) :: x2        ! 2nd cube face coordinate
   261)     real (kind=real_kind) :: tmpD(2,2)
   262)     real (kind=real_kind) :: M(2,2),E(2,2),eig(2),DE(2,2),DEL(2,2),V(2,2), nu1, nu2, lamStar1, lamStar2
   263)     integer :: imaxM(2)
   264)     real (kind=real_kind) :: l1, l2, sc     ! eigen values of met
   265) 
   266)     real (kind=real_kind) :: roundoff_err = 1e-11 !!! OG: this is a temporal fix
   267)     
   268)     ! ==============================================
   269)     ! Initialize differential mapping operator
   270)     ! to and from vector fields on the sphere to 
   271)     ! contravariant vector fields on the cube
   272)     ! i.e. dM/dx^i in Sadourney (1972) and it's 
   273)     ! inverse
   274)     ! ==============================================
   275) 
   276)     ! MNL: Calculate Jacobians of bilinear map from cubed-sphere to ref element
   277)     if (cubed_sphere_map==0) then
   278)        call elem_jacobians(elem%cartp, elem%u2qmap)
   279)     endif
   280) 
   281)     elem%max_eig = 0.0d0
   282)     elem%min_eig = 1d99
   283)     elem%max_eig_ratio = 0d0
   284)     do j=1,np
   285)        do i=1,np
   286)           x1=gll_points(i)
   287)           x2=gll_points(j)
   288)           call Dmap(elem%D(:,:,i,j),elem,x1,x2)
   289) 
   290) 
   291)           ! Numerical metric tensor based on analytic D: met = D^T times D
   292)           ! (D maps between sphere and reference element)
   293)           elem%met(1,1,i,j) = elem%D(1,1,i,j)*elem%D(1,1,i,j) + &
   294)                               elem%D(2,1,i,j)*elem%D(2,1,i,j)
   295)           elem%met(1,2,i,j) = elem%D(1,1,i,j)*elem%D(1,2,i,j) + &
   296)                               elem%D(2,1,i,j)*elem%D(2,2,i,j)
   297)           elem%met(2,1,i,j) = elem%D(1,1,i,j)*elem%D(1,2,i,j) + &
   298)                               elem%D(2,1,i,j)*elem%D(2,2,i,j)
   299)           elem%met(2,2,i,j) = elem%D(1,2,i,j)*elem%D(1,2,i,j) + &
   300)                               elem%D(2,2,i,j)*elem%D(2,2,i,j)
   301) 
   302)           ! compute D^-1...
   303)           ! compute determinant of D mapping matrix... if not zero compute inverse
   304) 
   305)           detD = elem%D(1,1,i,j)*elem%D(2,2,i,j) - elem%D(1,2,i,j)*elem%D(2,1,i,j)      
   306) 
   307)           elem%Dinv(1,1,i,j) =  elem%D(2,2,i,j)/detD
   308)           elem%Dinv(1,2,i,j) = -elem%D(1,2,i,j)/detD
   309)           elem%Dinv(2,1,i,j) = -elem%D(2,1,i,j)/detD
   310)           elem%Dinv(2,2,i,j) =  elem%D(1,1,i,j)/detD
   311) 
   312)           ! L2 norm = sqrt max eigenvalue of metinv
   313)           !         = 1/sqrt(min eigenvalue of met)
   314)           ! l1 and l2 are eigenvalues of met
   315)           ! (should both be positive, l1 > l2)
   316)           l1 = (elem%met(1,1,i,j) + elem%met(2,2,i,j) + sqrt(4.0d0*elem%met(1,2,i,j)*elem%met(2,1,i,j) + &
   317)               (elem%met(1,1,i,j) - elem%met(2,2,i,j))**2))/2.0d0
   318)           l2 = (elem%met(1,1,i,j) + elem%met(2,2,i,j) - sqrt(4.0d0*elem%met(1,2,i,j)*elem%met(2,1,i,j) + &
   319)               (elem%met(1,1,i,j) - elem%met(2,2,i,j))**2))/2.0d0
   320)           ! Max L2 norm of Dinv is sqrt of max eigenvalue of metinv
   321)           ! max eigenvalue of metinv is 1/min eigenvalue of met
   322)           norm = 1.0d0/sqrt(min(abs(l1),abs(l2)))
   323)           elem%max_eig = max(norm, elem%max_eig)
   324)           ! Min L2 norm of Dinv is sqrt of min eigenvalue of metinv
   325)           ! min eigenvalue of metinv is 1/max eigenvalue of met
   326)           norm = 1.0d0/sqrt(max(abs(l1),abs(l2)))
   327)           elem%min_eig = min(norm, elem%min_eig)
   328) 
   329)           ! Need inverse of met if not calculated analytically
   330)           elem%metdet(i,j) = abs(detD)
   331)           elem%rmetdet(i,j) = 1.0D0/abs(detD)
   332) 
   333)           elem%metinv(1,1,i,j) =  elem%met(2,2,i,j)/(detD*detD)
   334)           elem%metinv(1,2,i,j) = -elem%met(1,2,i,j)/(detD*detD)
   335)           elem%metinv(2,1,i,j) = -elem%met(2,1,i,j)/(detD*detD)
   336)           elem%metinv(2,2,i,j) =  elem%met(1,1,i,j)/(detD*detD)
   337) 
   338)           ! matricies for tensor hyper-viscosity
   339)           ! compute eigenvectors of metinv (probably same as computed above)
   340)           M = elem%metinv(:,:,i,j)
   341) 
   342)           eig(1) = (M(1,1) + M(2,2) + sqrt(4.0d0*M(1,2)*M(2,1) + &
   343)               (M(1,1) - M(2,2))**2))/2.0d0
   344)           eig(2) = (M(1,1) + M(2,2) - sqrt(4.0d0*M(1,2)*M(2,1) + &
   345)               (M(1,1) - M(2,2))**2))/2.0d0
   346)           
   347)           ! use DE to store M - Lambda, to compute eigenvectors
   348)           DE=M
   349)           DE(1,1)=DE(1,1)-eig(1)
   350)           DE(2,2)=DE(2,2)-eig(1)
   351) 
   352)           imaxM = maxloc(abs(DE))
   353)           if (maxval(abs(DE))==0) then
   354)              E(1,1)=1; E(2,1)=0;
   355)           elseif ( imaxM(1)==1 .and. imaxM(2)==1 ) then
   356)              E(2,1)=1; E(1,1) = -DE(2,1)/DE(1,1)
   357)           else   if ( imaxM(1)==1 .and. imaxM(2)==2 ) then
   358)              E(2,1)=1; E(1,1) = -DE(2,2)/DE(1,2)
   359)           else   if ( imaxM(1)==2 .and. imaxM(2)==1 ) then
   360)              E(1,1)=1; E(2,1) = -DE(1,1)/DE(2,1)
   361)           else   if ( imaxM(1)==2 .and. imaxM(2)==2 ) then
   362)              E(1,1)=1; E(2,1) = -DE(1,2)/DE(2,2)
   363)           else
   364)              call abortmp('Impossible error in cube_mod.F90::metric_atomic()')
   365)           endif
   366) 
   367)           ! the other eigenvector is orthgonal:
   368)           E(1,2)=-E(2,1)
   369)           E(2,2)= E(1,1)
   370) 
   371) !normalize columns
   372) 	  E(:,1)=E(:,1)/sqrt(sum(E(:,1)*E(:,1))); 
   373) 	  E(:,2)=E(:,2)/sqrt(sum(E(:,2)*E(:,2)));

   376) ! OBTAINING TENSOR FOR HV:
   377) 
   378) ! Instead of the traditional scalar Laplace operator \grad \cdot \grad
   379) ! we introduce \grad \cdot V \grad
   380) ! where V = D E LAM LAM^* E^T D^T. 
   381) ! Recall (metric_tensor)^{-1}=(D^T D)^{-1} = E LAM E^T.
   382) ! Here, LAM = diag( 4/((np-1)dx)^2 , 4/((np-1)dy)^2 ) = diag(  4/(dx_elem)^2, 4/(dy_elem)^2 )
   383) ! Note that metric tensors and LAM correspondingly are quantities on a unit sphere.
   384) 
   385) ! This motivates us to use V = D E LAM LAM^* E^T D^T
   386) ! where LAM^* = diag( nu1, nu2 ) where nu1, nu2 are HV coefficients scaled like (dx)^{hv_scaling/2}, (dy)^{hv_scaling/2}.
   387) ! (Halves in powers come from the fact that HV consists of two Laplace iterations.)
   388) 
   389) ! Originally, we took LAM^* = diag(
   390) !  1/(eig(1)**(hypervis_scaling/4.0d0))*(rearth**(hypervis_scaling/2.0d0))
   391) !  1/(eig(2)**(hypervis_scaling/4.0d0))*(rearth**(hypervis_scaling/2.0d0)) ) = 
   392) !  = diag( lamStar1, lamStar2)
   393) !  \simeq ((np-1)*dx_sphere / 2 )^hv_scaling/2 = SQRT(OPERATOR_HV)
   394) ! because 1/eig(...) \simeq (dx_on_unit_sphere)^2 .
   395) ! Introducing the notation OPERATOR = lamStar^2 is useful for conversion formulas.
   396) 
   397) ! This leads to the following conversion formula: nu_const is nu used for traditional HV on uniform grids
   398) ! nu_tensor = nu_const * OPERATOR_HV^{-1}, so
   399) ! nu_tensor = nu_const *((np-1)*dx_sphere / 2 )^{ - hv_scaling} or
   400) ! nu_tensor = nu_const *(2/( (np-1) * dx_sphere) )^{hv_scaling} .
   401) ! dx_sphere = 2\pi *rearth/(np-1)/4/NE
   402) ! [nu_tensor] = [meter]^{4-hp_scaling}/[sec]
   403) 
   404) ! (1) Later developments:
   405) ! Apply tensor V only at the second Laplace iteration. Thus, LAM^* should be scaled as (dx)^{hv_scaling}, (dy)^{hv_scaling},
   406) ! see this code below:
   407) !          DEL(1:2,1) = (lamStar1**2) *eig(1)*DE(1:2,1)
   408) !          DEL(1:2,2) = (lamStar2**2) *eig(2)*DE(1:2,2)
   409) 
   410) ! (2) Later developments:
   411) ! Bringing [nu_tensor] to 1/[sec]:
   412) !	  lamStar1=1/(eig(1)**(hypervis_scaling/4.0d0)) *(rearth**2.0d0)
   413) !	  lamStar2=1/(eig(2)**(hypervis_scaling/4.0d0)) *(rearth**2.0d0)
   414) ! OPERATOR_HV = ( (np-1)*dx_unif_sphere / 2 )^{hv_scaling} * rearth^4
   415) ! Conversion formula:
   416) ! nu_tensor = nu_const * OPERATOR_HV^{-1}, so
   417) ! nu_tensor = nu_const *( 2*rearth /((np-1)*dx))^{hv_scaling} * rearth^{-4.0}.
   418) 
   419) ! For the baseline coefficient nu=1e15 for NE30, 
   420) ! nu_tensor=7e-8 (BUT RUN TWICE AS SMALL VALUE FOR NOW) for hv_scaling=3.2
   421) ! and 
   422) ! nu_tensor=1.3e-6 for hv_scaling=4.0.
   423)

   426)           DE(1,1)=sum(elem%D(1,:,i,j)*E(:,1))

   428)           DE(2,1)=sum(elem%D(2,:,i,j)*E(:,1))
   429)           DE(2,2)=sum(elem%D(2,:,i,j)*E(:,2))
