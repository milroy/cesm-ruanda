     3)   !====================================================================
     4)   !   New standardized naming convention
     5)   !====================================================================
     6)   !  
     7)   !  ---------
     8)   !  definitions:
     9)   !  ---------
    10)   !  state-prefix
    11)   !    first 3 characters: Sx_, Sa_, Si_, Sl_, So_ 
    12)   !    one letter indices: x,a,l,i,o,g,r 
    13)   !    x => coupler (mapping, merging, atm/ocn flux calc done on coupler procs)
    14)   !    a => atm
    15)   !    l => lnd
    16)   !    i => ice
    17)   !    o => ocn
    18)   !    g => glc
    19)   !    r => rof
    20)   !    w => wav
    21)   !
    22)   !  state-name 
    23)   !    what follows state prefix
    24)   !
    25)   !  flux-prefix
    26)   !    first 5 characters: Flmn__ 
    27)   !    lm => between components l and m
    28)   !    n  => computed by component n
    29)   !    example: Fioi => ice/ocn flux computed by ice
    30)   !    example: Fall => atm/lnd flux computed by lnd
    31)   !    If flux prefix has first letter of P (so first five characters are PFlmn_)
    32)   !    then flux is passed straight through without scaling by the corresponding fraction)
    33)   !    
    34)   !  flux-name
    35)   !    what follows flux-prefix
    36)   !
    37)   !  ---------
    38)   !  rules:
    39)   !  ---------
    40)   !  1) states: 
    41)   !     a) atm attributes fields that HAVE a state-prefix of Sx_ in seq_flds_x2a_states
    42)   !        rule: will merge all identical values of the state-names from
    43)   !           seq_flds_i2x_states 
    44)   !           seq_flds_l2x_states 
    45)   !           seq_flds_o2x_states 
    46)   !           seq_flds_xao_states
    47)   !         to obtain output state-name in seq_flds_x2a_states
    48)   !  
    49)   !        rule: to merge input states that originate in the 
    50)   !           lnd (l2x_a) will be scaled by the lndfrac
    51)   !           ice (i2x_a) will be scaled by the icefrac
    52)   !           cpl (xao_a) will be scaled by the ocnfrac
    53)   !           ocn (o2x_a) will be scaled by the ocnfrac
    54)   !  
    55)   !        example: 
    56)   !           seq_flds_l2x_states = "Sl_t"
    57)   !           seq_flds_i2x_states = "Si_t"
    58)   !           seq_flds_o2x_states = "So_t"
    59)   !           seq_flds_x2a_states = "Sx_t" 
    60)   !           attribute fields Sl_t, Si_t, So_t, in 
    61)   !           attribute vectors l2x_a, i2x_a, o2x_a will be
    62)   !           merged to obtain attribute Sx_t in attribute vector x2a_a
    63)   !  
    64)   !     b) atm attribute fields that DO NOT HAVE a state-prefix of Sx_ in seq_flds_x2a_states
    65)   !        rule: copy directly all variables that identical state-prefix 
    66)   !               AND state-name in
    67)   !           seq_flds_i2x_states and seq_flds_x2a_states
    68)   !           seq_flds_l2x_states and seq_flds_x2a_states
    69)   !           seq_flds_o2x_states and seq_flds_x2a_states
    70)   !           seq_flds_xao_states and seq_flds_x2a_states
    71)   !  
    72)   !        example 
    73)   !           seq_flds_i2x_states = ":Si_snowh"
    74)   !           seq_flds_x2a_states = ":Si_snowh"
    75)   !           attribute field of Si_snowh in i2x_a will be copied to 
    76)   !           attribute field Si_snowh in x2a_a
    77)   !  
    78)   !  2) fluxes: 
    79)   !     rule: will merge all identical values of the flux-names from
    80)   !         seq_flds_i2x_states 
    81)   !         seq_flds_l2x_states 
    82)   !         seq_flds_o2x_states 
    83)   !         seq_flds_xao_states
    84)   !       to obtain output state-name in seq_flds_x2a_states
    85)   !  
    86)   !     rule: input flux fields that originate in the 
    87)   !         lnd (l2x_a) will be scaled by the lndfrac
    88)   !         ice (i2x_a) will be scaled by the icefrac
    89)   !            - ignore all fluxes that are ice/ocn fluxes (e.g. Fioi_)
    90)   !         cpl (xao_a) will be scaled by the ocnfrac
    91)   !         ocn (o2x_a) will be scaled by the ocnfrac+icefrac
    92)   !
    93)   !====================================================================
    94)   !
    95)   !   New user specified fields
    96)   ! 
    97)   !====================================================================
    98)   ! New fields that are user specidied can be added as namelist variables
    99)   ! by the user in the cpl namelist seq_flds_user using the namelist variable
   100)   ! array cplflds_customs. The user specified new fields must follow the
   101)   ! above naming convention.
   102)   ! As an example, say you want to add a new state 'foo' that is passed 
   103)   ! from the land to the atm - you would do this as follows 
   104)   !    &seq_flds_user
   105)   !       cplflds_custom = 'Sa_foo->a2x', 'Sa_foo->x2a'
   106)   !    /
   107)   ! This would add the field 'Sa_foo' to the character strings defining the 
   108)   ! attribute vectors a2x and x2a. It is assumed that code would need to be 
   109)   ! introduced in the atm and land components to deal with this new attribute 
   110)   ! vector field.
   111)   ! Currently, the only way to add this is to edit $CASEROOT/user_nl_cpl
   112)   !====================================================================
   113)   !
   114)   !   Coupler fields use cases
   115)   !
   116)   !====================================================================
   117)   ! Previously, new fields that were needed to be passed between components
   118)   ! for certain compsets were specified by cpp-variables. This has been 
   119)   ! modified to now be use cases. The use cases are specified in the 
   120)   ! namelist cpl_flds_inparm and are currently triggered by the xml 
   121)   ! variables CCSM_VOC, CCSM_BGC and GLC_NEC.  
   122)   !====================================================================

   242)    subroutine seq_flds_set(nmlfile, ID)

   249) ! !INPUT/OUTPUT PARAMETERS:
   250)      character(len=*), intent(in) :: nmlfile   ! Name-list filename
   251)      integer         , intent(in) :: ID        ! seq_comm ID
   252) 
   253)      !----- local -----
   254)      integer :: mpicom             ! MPI communicator
   255)      integer :: ierr               ! I/O error code
   256)      integer :: unitn              ! Namelist unit number to read
   257) 
   258)      character(len=CSS) :: attname
   259)      character(len=CSS) :: units
   260)      character(len=CSS) :: longname
   261)      character(len=CSS) :: stdname
   262)      integer            :: num
   263)      character(len=  2) :: cnum
   264)      character(len=CSS) :: name
   265) 
   266)      character(CXX) :: dom_coord  = ''
   267)      character(CXX) :: dom_other  = ''
   268) 
   269)      character(CXX) :: a2x_states = ''
   270)      character(CXX) :: a2x_fluxes = ''
   271)      character(CXX) :: x2a_states = ''
   272)      character(CXX) :: x2a_fluxes = ''
   273)      character(CXX) :: i2x_states = ''
   274)      character(CXX) :: i2x_fluxes = ''
   275)      character(CXX) :: x2i_states = ''
   276)      character(CXX) :: x2i_fluxes = ''
   277)      character(CXX) :: l2x_states = ''
   278)      character(CXX) :: l2x_fluxes = ''
   279)      character(CXX) :: x2l_states = ''
   280)      character(CXX) :: x2l_fluxes = ''
   281)      character(CXX) :: o2x_states = ''
   282)      character(CXX) :: o2x_fluxes = ''
   283)      character(CXX) :: x2o_states = ''
   284)      character(CXX) :: x2o_fluxes = ''
   285)      character(CXX) :: g2x_states = ''
   286)      character(CXX) :: g2x_fluxes = ''
   287)      character(CXX) :: x2g_states = ''
   288)      character(CXX) :: x2g_fluxes = ''
   289)      character(CXX) :: xao_albedo = ''
   290)      character(CXX) :: xao_states = ''
   291)      character(CXX) :: xao_fluxes = ''
   292)      character(CXX) :: r2x_states = ''
   293)      character(CXX) :: r2x_fluxes = ''
   294)      character(CXX) :: x2r_states = ''
   295)      character(CXX) :: x2r_fluxes = ''
   296)      character(CXX) :: w2x_states = ''
   297)      character(CXX) :: w2x_fluxes = ''
   298)      character(CXX) :: x2w_states = ''
   299)      character(CXX) :: x2w_fluxes = ''
   300) 
   301)      character(CXX) :: stringtmp  = ''
   302) 
   303)      !------ namelist -----
   304)      character(len=CSS)  :: fldname, fldflow
   305)      logical :: is_state, is_flux
   306)      integer :: i,n
   307) 
   308)      ! use cases namelists
   309)      logical :: flds_co2a 
   310)      logical :: flds_co2b 
   311)      logical :: flds_co2c 
   312)      logical :: flds_co2_dmsa 
   313)      integer :: glc_nec
   314) 
   315)      namelist /seq_cplflds_inparm/  &
   316)           flds_co2a, flds_co2b, flds_co2c, flds_co2_dmsa, glc_nec
   317) 
   318)      ! user specified new fields
   319)      integer,  parameter :: nfldmax = 200
   320)      character(len=CLL)  :: cplflds_custom(nfldmax) = ''
   321) 
   322)      namelist /seq_cplflds_userspec/ &          
   323)           cplflds_custom
   324) 
   325)      character(len=*),parameter :: subname = '(seq_flds_set) '
   326) 
   327) !-------------------------------------------------------------------------------
   328) 
   329)      call seq_comm_setptrs(ID,mpicom=mpicom)
   330) 
   331)      !---------------------------------------------------------------------------
   332)      ! Read in namelist for use cases
   333)      !---------------------------------------------------------------------------
   334)      ! TODO: permit duplicates to occur - then check for this in seq_flds_add
   335)      ! TODO: add entries for lookup entry table for custom fields 
   336)      !---------------------------------------------------------------------------
   337)      
   338)      if (seq_comm_iamroot(ID)) then
   339)         flds_co2a = .false.
   340)         flds_co2b = .false.
   341)         flds_co2c = .false.
   342)         flds_co2_dmsa = .false.
   343)         glc_nec   = 0
   344) 
   345)         unitn = shr_file_getUnit()
   346)         write(logunit,"(A)") subname//': read seq_cplflds_inparm namelist from: '&
   347)              //trim(nmlfile)
   348)         open( unitn, file=trim(nmlfile), status='old' )
   349)         ierr = 1
   350)         do while( ierr /= 0 )
   351)            read(unitn,nml=seq_cplflds_inparm,iostat=ierr)
   352)            if (ierr < 0) then
   353)               call shr_sys_abort( &
   354)                    subname//"ERROR: namelist read returns an EOF or EOR condition" )
   355)            end if
   356)         end do
   357)         close(unitn)
   358)         call shr_file_freeUnit( unitn )
   359)      end if
   360)      call shr_mpi_bcast(flds_co2a    , mpicom)
   361)      call shr_mpi_bcast(flds_co2b    , mpicom)
   362)      call shr_mpi_bcast(flds_co2c    , mpicom)
   363)      call shr_mpi_bcast(flds_co2_dmsa, mpicom)
   364)      call shr_mpi_bcast(glc_nec      , mpicom)
   365)      seq_flds_glc_nec = glc_nec
   366) 
   367)      !---------------------------------------------------------------------------
   368)      ! Read in namelists for user specified new fields
   369)      !---------------------------------------------------------------------------
   370)      ! TODO: permit duplicates to occur - then check for this in seq_flds_add
   371)      ! TODO: add entries for lookup entry table for custom fields 
   372)      !---------------------------------------------------------------------------
   373)

   375)         cplflds_custom(:) = ' '
   376) 
   377)         unitn = shr_file_getUnit()
   378)         write(logunit,"(A)") subname//': read seq_cplflds_userspec namelist from: '&
   379)              //trim(nmlfile)
   380)         open( unitn, file=trim(nmlfile), status='old' )
   381)         ierr = 1
   382)         do while( ierr /= 0 )
   383)            read(unitn,nml=seq_cplflds_userspec,iostat=ierr)
   384)            if (ierr < 0) then
   385)               call shr_sys_abort( &
   386)                    subname//"ERROR: namelist read returns an EOF or EOR condition" )
   387)            end if
   388)         end do
   389)         close(unitn)
   390)         call shr_file_freeUnit( unitn )
   391)      end if
   392)      do n = 1, nfldmax
   393)         call shr_mpi_bcast(cplflds_custom(n), mpicom)
   394)      end do
   395)         
   396)      ! add customized fields through coupler
   397) 
   398)      do n = 1,nfldmax
   399)         if (cplflds_custom(n) /= ' ') then
   400)            i = scan(cplflds_custom(n),'->')
   401)            fldname = trim(adjustl(cplflds_custom(n)(:i-1)))
   402)            fldflow = trim(adjustl(cplflds_custom(n)(i+2:)))
   403) 
   404)            if (fldname(1:1) == 'S') then
   405)               is_state = .true.
   406)               is_flux  = .false.
   407)            else if (fldname (1:1) == 'F')  then
   408)               is_state = .false.
   409)               is_flux  = .true.
   410)            else if (fldname (1:2) == 'PF') then
   411)               is_state = .false.
   412)               is_flux  = .true.
   413)            else
   414)               write(logunit,*) subname//'ERROR: fldname must start with S,F,P, not ',trim(fldname)
   415)               call shr_sys_abort(subname//"ERROR: fldname must start with S, F, or P")
   416)            end if
   417) 
   418)            select case (trim(fldflow))
   419)            case('a2x')
   420)               if (is_state) call seq_flds_add(a2x_states,trim(fldname))
   421)               if (is_flux ) call seq_flds_add(a2x_fluxes,trim(fldname))
   422)            case('x2a')
   423)               if (is_state) call seq_flds_add(x2a_states,trim(fldname))
   424)               if (is_flux ) call seq_flds_add(x2a_fluxes,trim(fldname))
   425)            case('l2x')
   426)               if (is_state) call seq_flds_add(l2x_states,trim(fldname))
   427)               if (is_flux ) call seq_flds_add(l2x_fluxes,trim(fldname))
   428)            case('x2l')
   429)               if (is_state) call seq_flds_add(x2l_states,trim(fldname))
   430)               if (is_flux ) call seq_flds_add(x2l_fluxes,trim(fldname))
   431)            case('r2x')
   432)               if (is_state) call seq_flds_add(r2x_states,trim(fldname))
   433)               if (is_flux ) call seq_flds_add(r2x_fluxes,trim(fldname))
   434)            case('x2r')
   435)               if (is_state) call seq_flds_add(x2r_states,trim(fldname))
   436)               if (is_flux ) call seq_flds_add(x2r_fluxes,trim(fldname))
   437)            case('i2x')
   438)               if (is_state) call seq_flds_add(i2x_states,trim(fldname))
   439)               if (is_flux ) call seq_flds_add(i2x_fluxes,trim(fldname))
   440)            case('x2i')
   441)               if (is_state) call seq_flds_add(x2i_states,trim(fldname))
   442)               if (is_flux ) call seq_flds_add(x2i_fluxes,trim(fldname))
   443)            case('o2x')
   444)               if (is_state) call seq_flds_add(o2x_states,trim(fldname))
   445)               if (is_flux ) call seq_flds_add(o2x_fluxes,trim(fldname))
   446)            case('x2o')
   447)               if (is_state) call seq_flds_add(x2o_states,trim(fldname))
   448)               if (is_flux ) call seq_flds_add(x2o_fluxes,trim(fldname))
   449)            case('g2x')
   450)               if (is_state) call seq_flds_add(g2x_states,trim(fldname))
   451)               if (is_flux ) call seq_flds_add(g2x_fluxes,trim(fldname))
   452)            case('x2g')
   453)               if (is_state) call seq_flds_add(x2g_states,trim(fldname))
   454)               if (is_flux ) call seq_flds_add(x2g_fluxes,trim(fldname))
   455)            case default
   456)               write(logunit,*) subname//'ERROR: ',trim(cplflds_custom(n)),&
   457)                    ' not a recognized value'
   458)               call shr_sys_abort()
   459)            end select
   460)         else
   461)            exit
   462)         end if
   463)      end do
   464) 
   465)      !----------------------------------------------------------
   466)      ! domain coordinates
   467)      !----------------------------------------------------------
   468) 
   469)      call seq_flds_add(dom_coord,'lat')
   470)      longname = ''
   471)      stdname  = 'latitude'
   472)      units    = 'degrees north'
