   401)            orig_start =  Svertex(i)%nbrs_ptr(j+1);

   412)      if(Debug) write(iulog,*)'CreateSubGridGraph: point #7'

   542)   subroutine initgridedge(GridEdge,GridVertex)
   543)   use parallel_mod, only : abortmp
   544)   use dimensions_mod, only : max_corner_elem
   545)   implicit none
   546) 
   547)   type (GridEdge_t), intent(inout)       :: GridEdge(:)
   548)   type (GridVertex_t), intent(in),target :: GridVertex(:)
   549) 
   550)   integer                                :: i,j,k,iptr,m,n,wgtV,wgtP
   551)   integer                                :: nelem,nelem_edge,inbr
   552)   logical                                :: Verbose=.FALSE.
   553)   integer                                :: mynbr_cnt, cnt, mystart, start
   554) 
   555)   nelem      = SIZE(GridVertex)
   556)   nelem_edge = SIZE(GridEdge)
   557) 
   558)   GridEdge(:)%reverse=.FALSE.
   559) 
   560)   iptr=1
   561)   do j=1,nelem
   562)      do i=1,num_neighbors    
   563)         mynbr_cnt = GridVertex(j)%nbrs_ptr(i+1) - GridVertex(j)%nbrs_ptr(i) !length of neighbor location  
   564)         mystart = GridVertex(j)%nbrs_ptr(i) 
   565)         do m=0,mynbr_cnt-1
   566)            if((GridVertex(j)%nbrs_wgt(mystart + m) .gt. 0)) then    ! Do this only if has a non-zero weight
   567)               if (nelem_edge<iptr) call abortmp('Error in initgridedge: Number of edges greater than expected.')
   568)               GridEdge(iptr)%tail      => GridVertex(j)
   569)               GridEdge(iptr)%tail_face =  mystart + m ! needs to be mystart + m (location in array)
   570)               GridEdge(iptr)%tail_dir = i*max_corner_elem + m !conversion needed for setcycle
   571)               inbr                     =  GridVertex(j)%nbrs(mystart+m)
   572)               GridEdge(iptr)%head      => GridVertex(inbr)
   573) 
   574)               ! ===========================================
   575)               ! Need this awful piece of code to determine
   576)               ! which "face" of the neighbor element the
   577)               ! edge links (i.e. the "head_face")
   578)               ! ===========================================
   579)               do k=1,num_neighbors
