   162)    subroutine conv_water_4rad(state, pbuf, totg_liq, totg_ice)
   163) 
   164)    ! --------------------------------------------------------------------- ! 
   165)    ! Purpose:                                                              !
   166)    ! Computes grid-box average liquid (and ice) from stratus and cumulus   !
   167)    ! Just for the purposes of radiation.                                   !
   168)    !                                                                       ! 
   169)    ! Method:                                                               !
   170)    ! Extract information about deep+shallow liquid and cloud fraction from !
   171)    ! the physics buffer.                                                   !
   172)    !                                                                       !
   173)    ! Author: Rich Neale, August 2006                                       !
   174)    !         October 2006: Allow averaging of liquid to give a linear      !
   175)    !                       average in emissivity.                          !
   176)    !                                                                       !
   177)    !---------------------------------------------------------------------- !
   178) 
   179)    
   180)    use physics_buffer, only : physics_buffer_desc, pbuf_get_field, pbuf_old_tim_idx
   181) 
   182)    use physics_types,   only: physics_state
   183)    use cam_history,     only: outfld
   184)    use phys_control,    only: phys_getopts
   185)    
   186)    implicit none
   187) 
   188)    ! ---------------------- !
   189)    ! Input-Output Arguments !
   190)    ! ---------------------- !
   191) 
   192)    
   193)    type(physics_state), target, intent(in) :: state        ! state variables
   194)    type(physics_buffer_desc),   pointer    :: pbuf(:)
   195) 
   196)    real(r8), intent(out):: totg_ice(pcols,pver)   ! Total GBA in-cloud ice
   197)    real(r8), intent(out):: totg_liq(pcols,pver)   ! Total GBA in-cloud liquid
   198) 
   199)    ! --------------- !
   200)    ! Local Workspace !
   201)    ! --------------- !
   202) 
   203)    real(r8), pointer, dimension(:,:) ::  pdel     ! Moist pressure difference across layer
   204)    real(r8), pointer, dimension(:,:) ::  ls_liq   ! Large-scale contributions to GBA cloud liq
   205)    real(r8), pointer, dimension(:,:) ::  ls_ice   ! Large-scale contributions to GBA cloud ice
   206) 
   207)    ! Physics buffer fields
   208)    real(r8), pointer, dimension(:,:) ::  ast      ! Physical liquid+ice stratus cloud fraction
   209)    real(r8), pointer, dimension(:,:) ::  sh_frac  ! Shallow convective cloud fraction
   210)    real(r8), pointer, dimension(:,:) ::  dp_frac  ! Deep convective cloud fraction
   211)    real(r8), pointer, dimension(:,:) ::  rei      ! Ice effective drop size (microns)
   212) 
   213)    real(r8), pointer, dimension(:,:) ::  dp_icwmr ! Deep conv. cloud water
   214)    real(r8), pointer, dimension(:,:) ::  sh_icwmr ! Shallow conv. cloud water
   215)    real(r8), pointer, dimension(:,:) ::  fice     ! Ice partitioning ratio
   216)    real(r8), pointer, dimension(:,:) ::  sh_cldliq ! shallow convection gbx liq cld mixing ratio for COSP
   217)    real(r8), pointer, dimension(:,:) ::  sh_cldice ! shallow convection gbx ice cld mixing ratio for COSP
   218) 
   219)    real(r8) :: conv_ice(pcols,pver)               ! Convective contributions to IC cloud ice
   220)    real(r8) :: conv_liq(pcols,pver)               ! Convective contributions to IC cloud liquid
   221)    real(r8) :: tot_ice(pcols,pver)                ! Total IC ice
   222)    real(r8) :: tot_liq(pcols,pver)                ! Total IC liquid
   223) 
   224)    integer  :: i,k,itim_old                       ! Lon, lev indices buff stuff.
   225)    real(r8) :: cu_icwmr                           ! Convective  water for this grid-box.   
   226)    real(r8) :: ls_icwmr                           ! Large-scale water for this grid-box. 
   227)    real(r8) :: tot_icwmr                          ! Large-scale water for this grid-box.  
   228)    real(r8) :: ls_frac                            ! Large-scale cloud frac for this grid-box. 
   229)    real(r8) :: tot0_frac, cu0_frac, dp0_frac, sh0_frac 
   230)    real(r8) :: kabs, kabsi, kabsl, alpha, dp0, sh0, ic_limit
   231)    real(r8) :: wrk1         
   232) 
   233)    integer :: lchnk
   234)    integer :: ncol
   235) 
   236)    ! --------- !
   237)    ! Parameter !
   238)    ! --------- !
   239) 
   240)    parameter( kabsl = 0.090361_r8, ic_limit = 1.e-12_r8 )
   241)    character(len=16) :: microp_scheme 
   242) 
   243)    ncol  = state%ncol
   244)    lchnk = state%lchnk
   245)    pdel   => state%pdel
   246)    ls_liq => state%q(:,:,ixcldliq)
   247)    ls_ice => state%q(:,:,ixcldice)
   248) 
   249)  ! Get microphysics option
   250)    call phys_getopts( microp_scheme_out = microp_scheme )
   251) 
   252)  ! Get convective in-cloud water and ice/water temperature partitioning.
   253) 
   254)    call pbuf_get_field(pbuf, icwmrsh_idx, sh_icwmr )
   255)    call pbuf_get_field(pbuf, icwmrdp_idx, dp_icwmr )
   256)    call pbuf_get_field(pbuf, fice_idx,    fice )
   257) 
   258)  ! Get convective in-cloud fraction    
   259) 
   260)    call pbuf_get_field(pbuf, sh_frac_idx,  sh_frac )
   261)    call pbuf_get_field(pbuf, dp_frac_idx,  dp_frac )
   262)    call pbuf_get_field(pbuf, rei_idx,      rei )
   263) 
   264)    itim_old = pbuf_old_tim_idx()
   265)    call pbuf_get_field(pbuf, ast_idx,  ast,  start=(/1,1,itim_old/), kount=(/pcols,pver,1/) ) 
   266) 
   267)    ! --------------------------------------------------------------- !
   268)    ! Loop through grid-boxes and determine:                          !
   269)    ! 1. Effective mean in-cloud convective ice/liquid (deep+shallow) !
   270)    ! 2. Effective mean in-cloud total ice/liquid (ls+convective)     !
   271)    ! --------------------------------------------------------------- !
   272) 
   273)    do k = 1, pver
   274)    do i = 1, ncol
   275) 
   276)       if( sh_frac(i,k) <= frac_limit .or. sh_icwmr(i,k) <= ic_limit ) then
   277)           sh0_frac = 0._r8
   278)       else
   279)           sh0_frac = sh_frac(i,k)
   280)       endif
   281)       if( dp_frac(i,k) <= frac_limit .or. dp_icwmr(i,k) <= ic_limit ) then
   282)           dp0_frac = 0._r8
   283)       else
   284)           dp0_frac = dp_frac(i,k)
   285)       endif
   286)       cu0_frac = sh0_frac + dp0_frac
   287) 
   288)     ! For the moment calculate the emissivity based upon the ls clouds ice fraction
   289) 
   290)       wrk1 = min(1._r8,max(0._r8, ls_ice(i,k)/(ls_ice(i,k)+ls_liq(i,k)+1.e-36_r8)))
   291) 
   292)       if( ( cu0_frac < frac_limit ) .or. ( ( sh_icwmr(i,k) + dp_icwmr(i,k) ) < ic_limit ) ) then
   293) 
   294)             cu0_frac = 0._r8
   295)             cu_icwmr = 0._r8
   296)          
   297)             ls_frac = ast(i,k)
   298)             if( ls_frac < frac_limit ) then
   299)                 ls_frac  = 0._r8
   300)                 ls_icwmr = 0._r8
   301)             else
   302)                 ls_icwmr = ( ls_liq(i,k) + ls_ice(i,k) )/max(frac_limit,ls_frac) ! Convert to IC value.
   303)             end if
   304) 
   305)             tot0_frac = ls_frac
   306)             tot_icwmr = ls_icwmr
   307)            
   308)       else
   309) 
   310)           ! Select radiation constants (effective radii) for emissivity averaging.
   311)             
   312)             if( microp_scheme == 'RK' ) then
   313)                kabsi = 0.005_r8 + 1._r8/rei(i,k)
   314)             else
   315)                kabsi = 0.005_r8 + 1._r8/min(max(13._r8,rei(i,k)),130._r8)
   316)             endif
   317)             kabs  = kabsl * ( 1._r8 - wrk1 ) + kabsi * wrk1
   318)             alpha = -1.66_r8*kabs*pdel(i,k)/gravit*1000.0_r8
   319) 
   320)           ! Selecting cumulus in-cloud water.            
   321) 
   322)             select case (conv_water_mode) ! Type of average
   323)             case (1) ! Area weighted arithmetic average
   324)                cu_icwmr = ( sh0_frac * sh_icwmr(i,k) + dp0_frac*dp_icwmr(i,k))/max(frac_limit,cu0_frac)
   325)             case (2)
   326)                sh0 = exp(alpha*sh_icwmr(i,k))
   327)                dp0 = exp(alpha*dp_icwmr(i,k))               
   328)                cu_icwmr = log((sh0_frac*sh0+dp0_frac*dp0)/max(frac_limit,cu0_frac))
   329)                cu_icwmr = cu_icwmr/alpha
   330)             case default ! Area weighted 'arithmetic in emissivity' average.
   331) !               call endrun ('CONV_WATER_4_RAD: Unknown option for conv_water_in_rad - exiting')
   332)             end select
   333) 
   334)           ! Selecting total in-cloud water. 
   335)           ! Attribute large-scale/convective area fraction differently from default.
   336) 
   337)             ls_frac   = ast(i,k) 
   338)             ls_icwmr  = (ls_liq(i,k) + ls_ice(i,k))/max(frac_limit,ls_frac) ! Convert to IC value.
   339)             tot0_frac = (ls_frac + cu0_frac) 
   340) 
   341)             select case (conv_water_mode) ! Type of average
   342)             case (1) ! Area weighted 'arithmetic in emissivity' average
   343)                tot_icwmr = (ls_frac*ls_icwmr + cu0_frac*cu_icwmr)/max(frac_limit,tot0_frac)
   344)             case (2)
   345)                tot_icwmr = log((ls_frac*exp(alpha*ls_icwmr)+cu0_frac*exp(alpha*cu_icwmr))/max(frac_limit,tot0_frac))
   346)                tot_icwmr = tot_icwmr/alpha
   347)             case default ! Area weighted 'arithmetic in emissivity' average.
   348) !               call endrun ('CONV_WATER_4_RAD: Unknown option for conv_water_in_rad - exiting')
   349)             end select
   350) 
   351)       end if
   352) 
   353)     ! Repartition convective cloud water into liquid and ice phase.
   354)     ! Currently, this partition is made using the ice fraction of stratus condensate.
   355)     ! In future, we should use ice fraction explicitly computed from the convection scheme.
   356) 
   357)       conv_ice(i,k) = cu_icwmr * wrk1
   358)       conv_liq(i,k) = cu_icwmr * (1._r8-wrk1)
   359) 
   360)       tot_ice(i,k)  = tot_icwmr * wrk1
   361)       tot_liq(i,k)  = tot_icwmr * (1._r8-wrk1)
   362) 
   363)       totg_ice(i,k) = tot0_frac * tot_icwmr * wrk1
   364)       totg_liq(i,k) = tot0_frac * tot_icwmr * (1._r8-wrk1)
   365) 
   366)    end do
   367)    end do
   368) 
   369) !add pbuff calls for COSP
   370)    call pbuf_get_field(pbuf, sh_cldliq1_idx, sh_cldliq  )
   371)    call pbuf_get_field(pbuf, sh_cldice1_idx, sh_cldice  )
   372) 
   373)    sh_cldliq(:ncol,:pver)=sh_icwmr(:ncol,:pver)*(1-fice(:ncol,:pver))*sh_frac(:ncol,:pver)
   374)    sh_cldice(:ncol,:pver)=sh_icwmr(:ncol,:pver)*fice(:ncol,:pver)*sh_frac(:ncol,:pver)
   375) 
   376)   ! Output convective IC WMRs
   377)    
   378)    call outfld( 'ICLMRCU ', conv_liq  , pcols, lchnk )
   379)    call outfld( 'ICIMRCU ', conv_ice  , pcols, lchnk )
   380)    call outfld( 'ICLMRTOT', tot_liq   , pcols, lchnk )
   381)    call outfld( 'ICIMRTOT', tot_ice   , pcols, lchnk )
   382) 
   383)   end subroutine conv_water_4rad
   384)
