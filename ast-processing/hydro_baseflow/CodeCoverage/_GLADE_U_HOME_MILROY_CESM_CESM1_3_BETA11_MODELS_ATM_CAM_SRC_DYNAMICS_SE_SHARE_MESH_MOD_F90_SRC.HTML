   405) ! subroutine create_index_table()
   406) !
   407) ! this is needed to detremine side and corner neighbors
   408) !======================================================================
   409) 
   410)   subroutine create_index_table(index_table, element_nodes)
   411) 
   412)     use dimensions_mod, only : max_elements_attached_to_node
   413)     use parallel_mod, only : abortmp
   414) 
   415)     integer, allocatable, intent(inout)  :: index_table(:,:) 
   416)     integer             ,  intent(in)    :: element_nodes(p_number_elements, 4)
   417)     integer                              :: cnt, cnt_index, node
   418)     integer                              :: k, ll 
   419) 
   420)     !Create an index table so that we can find neighbors on O(n)
   421)     ! so for each node, we want to know which elements it is part of
   422)     allocate(index_table(p_number_nodes, max_elements_attached_to_node + 1))
   423)    
   424)     !the last column in the index table is a count of the number of elements
   425)     index_table = 0
   426) 
   427)     cnt_index =  max_elements_attached_to_node + 1
   428)      
   429)     do k=1,p_number_elements
   430)         do ll=1,4 
   431)            node = element_nodes(k, ll) !the node
   432)            cnt = index_table(node, cnt_index)  !how many elements for that node already in table
   433)            cnt = cnt + 1 !increment since we are adding an element
   434)            if (cnt >  max_elements_attached_to_node) then
   435)               call abortmp('Found a node in too many elements.')
   436)            endif
   437)            index_table(node, cnt_index) = cnt  
   438)            index_table(node, cnt) = k !put the element in the indextable
   439)         enddo
   440)     enddo
   441) 
   442)   end subroutine create_index_table

   445) ! subroutine find_side_neighbors()
   446) !
   447) ! find the element neighbors to the n,s,e,w and put them in GridVertex_t
   448) !  (only 1 neighbor to the n,s,e,w)
   449) !======================================================================
   450)   subroutine find_side_neighbors (GridVertex, normal_to_homme_ordering, element_nodes, edge_wgt, index_table)
   451)     use coordinate_systems_mod, only : cartesian3D_t
   452)     use gridgraph_mod, only   : GridVertex_t
   453)     use parallel_mod, only : abortmp
   454)     use dimensions_mod, only : max_elements_attached_to_node
   455) 
   456)     implicit none
   457)     integer             ,  intent(in)    :: normal_to_homme_ordering(8)
   458)     integer             ,  intent(in)    :: element_nodes(p_number_elements, 4)
   459)     integer             ,  intent(in)    :: edge_wgt
   460)     integer             ,  intent(in)     :: index_table(:,:) 
   461)     type (GridVertex_t) ,  intent(inout) :: GridVertex(:)
   462) 
   463)     integer                              :: i_node(2), my_node(2) 
   464)     integer                              :: neighbor, direction, init_size
   465)     integer                              :: j,k,ll,i, m
   466)     integer                              :: i_elem, jump, end_i
   467)     integer                              :: loc, node, cnt_index, cnt, a_count(2)
   468)     logical                              :: found
   469)     if (0 == p_number_blocks)  call abortmp('find_side_neighbors called before MeshOpen')
   470) 
   471)    
   472)     !the last column in the index table is a count of the number of elements
   473)     cnt_index =  max_elements_attached_to_node + 1
   474)      
   475)     !use index table to find neighbors
   476)     do k=1,p_number_elements  ! for each element k
   477)        !set the side weights
   478)        GridVertex(k)%nbrs_wgt(1:4) = edge_wgt
   479)        do ll=1,4 !loop through the four sides
   480) 
   481)           jump = normal_to_homme_ordering(ll)
   482)           loc = GridVertex(k)%nbrs_ptr(jump)
   483) 
   484)           if (GridVertex(k)%nbrs(loc) == 0) then  !if side is not set yet, then
   485)              !look for side element
   486)              found = .false.
   487)              neighbor = 0
   488) 
   489)              my_node(1) = element_nodes(k, ll)
   490)              a_count(1) = index_table(my_node(1), cnt_index)
   491)              my_node(2) = element_nodes(k, mod(ll,4)+1)
   492)              a_count(2) = index_table(my_node(2), cnt_index)
   493) 
   494)              !loop through the elements that are in the index table for each node
   495)              !and find the element number and direction of the side neighbor
   496)              do m = 1,2
   497)                 if (found) exit
   498)                 end_i = a_count(m)
   499)                 do i = 1, end_i
   500)                    if (found) exit
   501)                    i_elem = index_table(my_node(m),i)
   502)                    if (i_elem /= k) then !k is the element we are setting sides for
   503)                       do j=1,4 !loop through each of i_elem's four sides
   504)                          i_node(1) = element_nodes(i_elem, j)
   505)                          i_node(2) = element_nodes(i_elem, mod(j,4)+1)
   506)                          if ( (i_node(1) == my_node(2) .and. i_node(2) == my_node(1)) .or. &
   507)                               (i_node(1) == my_node(1) .and. i_node(2) == my_node(2)) ) then
   508)                             neighbor = i_elem
   509)                             direction = j
   510)                             found = .true.
   511)                             !found a match
   512)                             exit
   513)                          end if
   514)                       end do ! j loop
   515)                    end if
   516)                 enddo ! i loop
   517)              enddo !m loop
   518) 
   519)              if (neighbor == 0) call abortmp('find_side_neighbor: Neighbor not found! Every side should have a neighbor.') 
   520) 
   521)              GridVertex(k)%nbrs(loc) = neighbor         
   522)              jump = normal_to_homme_ordering(direction)
   523)              loc = GridVertex(neighbor)%nbrs_ptr(jump)
   524)              GridVertex(neighbor)%nbrs(loc)= k         
   525)           endif

   670) ! subroutine initialize_space_filling_curve
   671) !======================================================================
   672)   subroutine initialize_space_filling_curve(GridVertex, element_nodes)
   673)     use gridgraph_mod, only   : GridVertex_t
   674)     use parallel_mod,  only   : abortmp
   675)     use spacecurve_mod, only  : GenspaceCurve
   676)     
   677)     implicit none
   678)     type (GridVertex_t), intent(inout) :: GridVertex(:)
   679)     integer            , intent(in)    :: element_nodes(:,:)
   680)     
   681)     integer,allocatable                :: Mesh2(:,:),Mesh2_map(:,:),sfcij(:,:)
   682)     
   683)     real                               :: centroids(p_number_elements,2)
   684)     integer                            :: face_numbers(p_number_elements)
   685)     real                               :: x, y, h
   686)     integer                            :: i, j, i2, j2, ne, ne2
   687)     integer                            :: sfc_index, face, nelem
   688)     
   689)     if (SIZE(GridVertex) /= p_number_elements) then
   690)        call abortmp('initialize_space_filling_curve:Element count check failed &
   691)             &in exodus_mesh. Vertex array length not equal to number of elements.')
   692)     end if
   693)     if (SIZE(element_nodes,dim=1) /= p_number_elements) then
   694)        call abortmp('initialize_space_filling_curve:Element count check failed &
   695)             &in exodus_mesh. Connectivity array length not equal to number of elements.')
   696)     end if
   697)     
   698)     face_numbers(:) = GridVertex(:)%face_number
   699)     h = smallest_diameter_element    (                         element_nodes)
   700) 
   701)     call cube_face_element_centroids (centroids, face_numbers, element_nodes)
   702)     
   703)     if (h<.00001) call abortmp('initialize_space_filling_curve: Unreasonably small element found. less than .00001')
   704)     
   705)     ne = CEILING(0.5*DD_PI/(h/2));
   706)     
   707)     ! find the smallest ne2 which is a power of 2 and ne2>ne
   708)     ne2=2**ceiling( log(real(ne))/log(2d0) )
   709)     if (ne2<ne) call abortmp('initialize_space_filling_curve: Fatel SFC error')
   710)     
   711)     allocate(Mesh2(ne2,ne2))
   712)     allocate(Mesh2_map(ne2,ne2))
   713)     allocate(sfcij(0:ne2*ne2,2))
   714)     
   715)     ! create a reverse index array for Mesh2
   716)     ! j = Mesh2(i,j) 
   717)     ! (i,j) = (sfcij(j,1),sfci(j,2)) 
   718)     call GenspaceCurve(Mesh2)  ! SFC partition for ne2
   719)     do j2=1,ne2
   720)        do i2=1,ne2
   721)           j=Mesh2(i2,j2)
   722)           sfcij(j,1)=i2
   723)           sfcij(j,2)=j2
   724)        enddo
   725)     enddo
   726)     
   727)     
   728)     GridVertex(:)%SpaceCurve=-1
   729)     sfc_index   = 0
   730)     do face = 1,nfaces
   731)        ! associate every element on the ne x ne mesh (Mesh)
   732)        ! with its closest element on the ne2 x ne2 mesh (Mesh2)
   733)        ! Store this as a map from Mesh2 -> Mesh in Mesh2_map.
   734)        ! elements in Mesh2 which are not mapped get assigned a value of 0
   735)        Mesh2_map=0
   736)        do i=1,p_number_elements
   737)           if (face_numbers(i) == face ) then
   738)              x = centroids(i,1)
   739)              y = centroids(i,2)
   740)              ! map this element to an (i2,j2) element
   741)              ! [ -DD_PI/4, DD_PI/4 ]  -> [ 0, ne2 ]
   742)              i2=nint( (0.5 + 2.0*x/DD_PI)*ne2 + .5 )
   743)              j2=nint( (0.5 + 2.0*y/DD_PI)*ne2 + .5 )
   744)              if (face == 4 .or. face == 6 )               i2 = ne2-i2+1
   745)              if (face == 1 .or. face == 2 .or. face == 6) j2 = ne2-j2+1
   746)              if (i2<1  ) i2=1
   747)              if (i2>ne2) i2=ne2
   748)              if (j2<1  ) j2=1
   749)              if (j2>ne2) j2=ne2
   750)              Mesh2_map(i2,j2)=i
   751)           end if
   752)        end do
   753)        
   754)        ! generate a SFC for Mesh with the same ordering as the 
   755)        ! elements in Mesh2 which map to Mesh.
   756)        do j=0,ne2*ne2-1
   757)           i2=sfcij(j,1)
   758)           j2=sfcij(j,2)
   759)           i=Mesh2_map(i2,j2)
   760)           if (i/=0) then
   761)              ! (i2,j2) element maps to element
   762)              GridVertex(i)%SpaceCurve=sfc_index
   763)              sfc_index=sfc_index+1
   764)           endif
   765)        enddo
   766)     enddo
   767)     deallocate(Mesh2)
   768)     deallocate(Mesh2_map)
   769)     deallocate(sfcij)
   770)     
   771)     if (minval(GridVertex(:)%SpaceCurve) == -1) then
   772)        do i=1,p_number_elements
   773)           if (-1==GridVertex(i)%SpaceCurve) then
   774)              write (*,*) " Error in projecting element ",i," to space filling curve."
   775)              write (*,*) " Face:",face_numbers(i)
   776)              write (*,*) " Centroid:",centroids(i,:)
   777)           end if
   778)        end do
   779)        call abortmp('initialize_space_filling_curve: Vertex not on SpaceCurve')
   780)     end if
   781)     
   782)   end subroutine initialize_space_filling_curve

   785) ! subroutine find_corner_neighbors
   786) !======================================================================
   787) 
   788)   subroutine find_corner_neighbors  (GridVertex, normal_to_homme_ordering, element_nodes, corner_wgt, index_table)
   789)     use parallel_mod,           only : abortmp
   790)     use gridgraph_mod,          only : GridVertex_t
   791)     use dimensions_mod,         only : max_elements_attached_to_node, max_corner_elem
   792)     use control_mod, only: north, south, east, west, neast,seast, nwest,swest
   793)     implicit none
   794)     
   795)     type (GridVertex_t), intent(inout) :: GridVertex(:)
   796)     integer            , intent(in)    :: normal_to_homme_ordering(8)
   797)     integer            , intent(in)    :: element_nodes(p_number_elements, 4)
   798)     integer            , intent(in)    :: corner_wgt
   799)     integer            , intent(in)    :: index_table(:,:) 
   800) 
   801)     integer                          :: node_elements (2*max_elements_attached_to_node)
   802)     integer                          :: elem_neighbor (4*max_elements_attached_to_node)
   803)     integer                          :: nbr_cnt(4)
   804)     integer                          :: elem_nbr_start, start
   805)     integer                          :: i, j, k, ll, jj, kk
   806)     integer                          :: node, loc, cnt, cnt_index
   807)     integer                          :: corner_array(max_corner_elem), orig_pos(max_corner_elem)
   808)     integer                          :: face_array(max_corner_elem), a_corner_elems(max_corner_elem)
   809)     integer                          :: corner_sides(2)
   810)     integer                          :: side_elem, corner_elem, tmp_s 
   811) 
   812)     !the last column in the index table is a count of the number of elements
   813)     cnt_index =  max_elements_attached_to_node + 1
   814) 
   815)     do i=1, p_number_elements  !loop through all elements
   816)        node_elements(:) = 0
   817)        elem_neighbor(:) = 0
   818)        elem_nbr_start = 0
   819)        nbr_cnt(:) = 0
   820) 
   821)        do j=1,4  !check each of the 4 nodes at the element corners
   822)           node = element_nodes(i,j)
   823)           cnt = index_table(node, cnt_index)
   824)           if (cnt < 3 .or. max_elements_attached_to_node < cnt) then
   825)              call abortmp('find_corner_neighbors: Number of elements attached to node less than 3 or greater than maximum.')
   826)           endif
   827) 
   828)           node_elements(1:cnt) = index_table(node, 1:cnt)
   829) 
   830)           !now node_elements contains the element neighbors to that node - so grab the 
   831)           ! corner neighbors - these are the ones that are not already side neighbors (or myself)
   832)           k = 0
   833)           do ll=1,cnt 
   834)              if ( i /= node_elements(ll) .and. & !not me
   835)                   GridVertex(i)%nbrs(1) /= node_elements(ll) .and. & !not side 1
   836)                   GridVertex(i)%nbrs(2) /= node_elements(ll) .and. & ! etc ... 
   837)                   GridVertex(i)%nbrs(3) /= node_elements(ll) .and. &
   838)                   GridVertex(i)%nbrs(4) /= node_elements(ll)) then   
   839)                 k = k + 1
   840)                 elem_neighbor(elem_nbr_start + k) = node_elements(ll)
   841)              end if
   842)           end do ! end of ll loop for multiplicity
   843) 
   844)           !keep track of where we are starting in elem_neighbor for each corner j
   845)           elem_nbr_start = elem_nbr_start + k
   846)           nbr_cnt(j) = k !how many neighbors in this corner
   847)        end do ! end of j loop through 4 nodes
   848) 
   849) 
   850)        ! now that we have done the 4 corners we can populate nbrs and nbrs_ptr 
   851)        ! with the corners in the proper order (clockwise) in neighbors
   852)        ! also we can add the corner weight
   853) 
   854)        do j=5,8 !loop through 4 corners
   855)           elem_nbr_start = 1
   856)           !easiest to do the corner in ascending order - find loc
   857)           do jj = 5,8
   858)              ll = normal_to_homme_ordering(jj)
   859)              if (j == ll) then
   860)                 loc = jj
   861)                 exit
   862)              end if
   863)              elem_nbr_start = elem_nbr_start + nbr_cnt(jj-4)          
   864)           end do
   865) 
   866)           start =  GridVertex(i)%nbrs_ptr(j)
   867)           cnt = nbr_cnt(loc - 4)
   868)           GridVertex(i)%nbrs_ptr(j+1) = start + cnt
   869)           
   870)           if (cnt > 0) then
   871)              GridVertex(i)%nbrs(start : start + cnt-1) = &
   872)                   elem_neighbor(elem_nbr_start : elem_nbr_start + cnt -1)
   873)              GridVertex(i)%nbrs_face(start : start + cnt - 1) = &
   874)                   GridVertex(elem_neighbor(elem_nbr_start : elem_nbr_start + cnt -1))%face_number
   875)              GridVertex(i)%nbrs_wgt(start : start + cnt-1)  = corner_wgt
   876) 
   877)           end if
   878) 
   879)           ! within each corner neighbor, lets list the corners in clockwise order
   880)           if (cnt > 1) then !cnt is the number of neighbors in this corner j
   881)                             !there can be at most max_corner element of these
   882)              
   883)              a_corner_elems = 0
   884)              a_corner_elems = elem_neighbor(elem_nbr_start : elem_nbr_start + cnt -1)
   885)              !corner-sides(2) is clockwise of corner_side(1)
   886)              corner_array= 0
   887)              orig_pos = 0
   888)              select case (j)
   889)              case(neast)
   890)                 corner_sides(1) = north
   891)                 corner_sides(2) = east
   892)              case(seast)
   893)                 corner_sides(1) = east
   894)                 corner_sides(2) = south
   895)              case(swest)
   896)                 corner_sides(1) = south
   897)                 corner_sides(2) = west
   898)              case(nwest)
   899)                 corner_sides(1) = west
   900)                 corner_sides(2) = north
   901)              end select
   902)              
   903)              !so the first element to list touches  corner_sides(1) element
   904)              side_elem = GridVertex(i)%nbrs(corner_sides(1))
   905)              
   906)              !loop though the corner elements and see if any have a side neighbor 
   907)              !that = side_elem
   908)              do k = 1,cnt !number of corner elements
   909)                 corner_elem = a_corner_elems(k)
   910)                 do kk = 1,4 !number of sides to check
   911)                    loc = GridVertex(corner_elem)%nbrs_ptr(kk)
   912)                    tmp_s = GridVertex(corner_elem)%nbrs(loc)
   913)                    if (tmp_s == side_elem) then
   914)                       corner_array(1) = corner_elem
   915)                       orig_pos(1) = k
   916)                       exit
   917)                    endif
   918)                 enddo
   919)                 if  (corner_array(1)> 0) exit
   920)              enddo
   921)              if (corner_array(1)==0) then
   922)                 print *, i, cnt
   923)                 call abortmp('find_corner_neighbors (1) : mistake finding corner neighbor order')
   924)              endif
   925) 
   926)              !if cnt == 2, we are done (we know the order of neighbors)
   927)              if (cnt ==2) then
   928)                 if (corner_array(1) ==  a_corner_elems(1)) then
   929)                    corner_array(2) =  a_corner_elems(2)
   930)                    orig_pos(2) = 2
   931)                 else
   932)                    corner_array(2) =  a_corner_elems(1)
   933)                    orig_pos(2) = 1
   934)                 end if
   935)              else !cnt = 3 or 4
   936)                 !find which corner element borders corner_sides(2)
   937)                 side_elem = GridVertex(i)%nbrs(corner_sides(2))
   938)                 do k = 1,cnt
   939)                    corner_elem = a_corner_elems(k)
   940)                    do kk = 1,4
   941)                       loc = GridVertex(corner_elem)%nbrs_ptr(kk)
   942)                       tmp_s = GridVertex(corner_elem)%nbrs(loc)
   943)                       if (tmp_s == side_elem) then
   944)                          corner_array(4) = corner_elem
   945)                          orig_pos(4) = k
   946)                          exit
   947)                       endif
   948)                    enddo
   949)                    if  (corner_array(4)> 0) exit
   950)                 enddo
   951)                 if (corner_array(4)==0 .or. corner_array(4) == corner_array(1)) then
   952)                    print *, i, cnt
   953)                    call abortmp('find_corner_neighbors (2) : mistake finding corner neighbor order')
   954)                 endif
   955)                 
   956)                 !now if cnt = 3 then we are done
   957)                 if (cnt ==3) then
   958)                    corner_array(3) = corner_array(4)
   959)                    orig_pos(3) = orig_pos(4) 
   960)                    
   961)                    do k = 1,cnt !find the "middle" element
   962)                       if (k /= orig_pos(1) .and. k /= orig_pos(3)) then
   963)                          orig_pos(2) = k
   964)                          corner_array(2) = a_corner_elems(k)
   965)                          exit
   966)                       endif
   967)                    enddo
   968)                 else  !cnt = 4 
   969)                    !which of the two unassigned elements borders the element in
   970)                    !corner_array(1) => put in corner_array(2)
   971)                    side_elem = corner_array(1)
   972)                    
   973)                    do k = 1,cnt
   974)                       corner_elem = a_corner_elems(k)
   975)                       if (corner_elem == corner_array(4) .or. corner_elem == corner_array(1)) then
   976)                          cycle
   977)                       else
   978)                          do kk = 1,4 !check each side
   979)                             loc = GridVertex(corner_elem)%nbrs_ptr(kk)
   980)                             tmp_s = GridVertex(corner_elem)%nbrs(loc)
   981)                             if (tmp_s == side_elem) then
   982)                                corner_array(2) = corner_elem
   983)                                orig_pos(2) = k
   984)                                exit
   985)                             endif
   986)                          enddo
   987)                       endif
   988)                       if  (corner_array(2)> 0) exit
   989)                    enddo
   990)                    !now put the remaining one in pos 3
   991)                    do k = 1,cnt
   992)                       corner_elem = a_corner_elems(k)
   993)                       if (corner_elem /= corner_array(4) .and. corner_elem /= &
   994)                            corner_array(2) .and. corner_elem /= corner_array(1)) then
   995)                          corner_array(3) = corner_elem
   996)                          orig_pos(3) = k
   997)                          exit
   998)                       endif
   999)                    enddo
  1000)                 endif ! end of cnt=4
  1001)              endif! end of not cnt=2
  1002) 
  1003)              !now re-set the elements in this corner
  1004)              GridVertex(i)%nbrs(start : start + cnt-1) = corner_array(1:cnt)
  1005)              !nbrs_wgt are the same - nothing to do
  1006)              !fix neighbors face
  1007)              do k = 1,cnt
