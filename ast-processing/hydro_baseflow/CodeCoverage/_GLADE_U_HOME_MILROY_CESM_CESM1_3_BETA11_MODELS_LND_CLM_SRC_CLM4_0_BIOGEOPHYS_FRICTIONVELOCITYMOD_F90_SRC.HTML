    38) ! !IROUTINE: FrictionVelocity

    40) ! !INTERFACE:
    41)   subroutine FrictionVelocity(lbn, ubn, fn, filtern, &
    42)                               displa, z0m, z0h, z0q, &
    43)                               obu, iter, ur, um, ustar, &
    44)                               temp1, temp2, temp12m, temp22m, fm, landunit_index)
    45) !
    46) ! !DESCRIPTION:
    47) ! Calculation of the friction velocity, relation for potential
    48) ! temperature and humidity profiles of surface boundary layer.
    49) ! The scheme is based on the work of Zeng et al. (1998):
    50) ! Intercomparison of bulk aerodynamic algorithms for the computation
    51) ! of sea surface fluxes using TOGA CORE and TAO data. J. Climate,
    52) ! Vol. 11, 2628-2644.
    53) !
    54) ! !USES:
    55)    use clmtype
    56)    use clm_atmlnd, only : clm_a2l
    57)    use clm_varcon, only : vkc
    58)    use clm_varctl, only : iulog
    59) !
    60) ! !ARGUMENTS:
    61)    implicit none
    62)    integer , intent(in)  :: lbn, ubn         ! pft/landunit array bounds
    63)    integer , intent(in)  :: fn               ! number of filtered pft/landunit elements
    64)    integer , intent(in)  :: filtern(fn)      ! pft/landunit filter
    65)    real(r8), intent(in)  :: displa(lbn:ubn)  ! displacement height (m)
    66)    real(r8), intent(in)  :: z0m(lbn:ubn)     ! roughness length over vegetation, momentum [m]
    67)    real(r8), intent(in)  :: z0h(lbn:ubn)     ! roughness length over vegetation, sensible heat [m]
    68)    real(r8), intent(in)  :: z0q(lbn:ubn)     ! roughness length over vegetation, latent heat [m]
    69)    real(r8), intent(in)  :: obu(lbn:ubn)     ! monin-obukhov length (m)
    70)    integer,  intent(in)  :: iter             ! iteration number
    71)    real(r8), intent(in)  :: ur(lbn:ubn)      ! wind speed at reference height [m/s]
    72)    real(r8), intent(in)  :: um(lbn:ubn)      ! wind speed including the stablity effect [m/s]
    73)    logical,  optional, intent(in)  :: landunit_index  ! optional argument that defines landunit or pft level
    74)    real(r8), intent(out) :: ustar(lbn:ubn)   ! friction velocity [m/s]
    75)    real(r8), intent(out) :: temp1(lbn:ubn)   ! relation for potential temperature profile
    76)    real(r8), intent(out) :: temp12m(lbn:ubn) ! relation for potential temperature profile applied at 2-m
    77)    real(r8), intent(out) :: temp2(lbn:ubn)   ! relation for specific humidity profile
    78)    real(r8), intent(out) :: temp22m(lbn:ubn) ! relation for specific humidity profile applied at 2-m
    79)    real(r8), intent(inout) :: fm(lbn:ubn)    ! diagnose 10m wind (DUST only)
    80) !
    81) ! !CALLED FROM:
    82) !
    83) ! !REVISION HISTORY:
    84) ! 15 September 1999: Yongjiu Dai; Initial code
    85) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
    86) ! 12/19/01, Peter Thornton
    87) ! Added arguments to eliminate passing clm derived type into this function.
    88) ! Created by Mariana Vertenstein
    89) !
    90) ! !LOCAL VARIABLES:
    91) !
    92) ! local pointers to implicit in arguments
    93) !
    94)    integer , pointer :: ngridcell(:)      !pft/landunit gridcell index
    95)    real(r8), pointer :: forc_hgt_u_pft(:) !observational height of wind at pft level [m]
    96)    real(r8), pointer :: forc_hgt_t_pft(:) !observational height of temperature at pft level [m]
    97)    real(r8), pointer :: forc_hgt_q_pft(:) !observational height of specific humidity at pft level [m]
    98)    integer , pointer :: pfti(:)           !beginning pfti index for landunit
    99)    integer , pointer :: pftf(:)           !final pft index for landunit
   100) !
   101) ! local pointers to implicit out arguments
   102) !
   103)    real(r8), pointer :: u10(:)         ! 10-m wind (m/s) (for dust model)
   104)    real(r8), pointer :: fv(:)          ! friction velocity (m/s) (for dust model)
   105)    real(r8), pointer :: vds(:)         ! dry deposition velocity term (m/s) (for SO4 NH4NO3)
   106)    real(r8), pointer :: u10_clm(:)     ! 10-m wind (m/s)
   107)    real(r8), pointer :: va(:)          ! atmospheric wind speed plus convective velocity (m/s)
   108) !
   109) !
   110) ! !OTHER LOCAL VARIABLES:
   111) !EOP
   112) !
   113)    real(r8), parameter :: zetam = 1.574_r8 ! transition point of flux-gradient relation (wind profile)
   114)    real(r8), parameter :: zetat = 0.465_r8 ! transition point of flux-gradient relation (temp. profile)
   115)    integer :: f                            ! pft/landunit filter index
   116)    integer :: n                            ! pft/landunit index
   117)    integer :: g                            ! gridcell index
   118)    integer :: pp                           ! pfti,pftf index
   119)    real(r8):: zldis(lbn:ubn)               ! reference height "minus" zero displacement heght [m]
   120)    real(r8):: zeta(lbn:ubn)                ! dimensionless height used in Monin-Obukhov theory
   121)    real(r8) :: tmp1,tmp2,tmp3,tmp4         ! Used to diagnose the 10 meter wind
   122)    real(r8) :: fmnew                       ! Used to diagnose the 10 meter wind
   123)    real(r8) :: fm10                        ! Used to diagnose the 10 meter wind
   124)    real(r8) :: zeta10                      ! Used to diagnose the 10 meter wind
   125)    real(r8) :: vds_tmp                     ! Temporary for dry deposition velocity
   126) !------------------------------------------------------------------------------
   127) 
   128)    ! Assign local pointers to derived type members (gridcell-level)
   129) 
   130)    if (present(landunit_index)) then
   131)      ngridcell  => lun%gridcell
   132)    else
   133)      ngridcell  => pft%gridcell
   134)    end if
   135) 
   136)    vds        => pps%vds
   137)    u10        => pps%u10
   138)    u10_clm    => pps%u10_clm
   139)    va         => pps%va
   140)    fv         => pps%fv
   141) 
   142)    ! Assign local pointers to derived type members (pft or landunit-level)
   143) 
   144)    pfti             => lun%pfti
   145)    pftf             => lun%pftf
   146) 
   147)    ! Assign local pointers to derived type members (pft-level)
   148) 
   149)    forc_hgt_u_pft => pps%forc_hgt_u_pft
   150)    forc_hgt_t_pft => pps%forc_hgt_t_pft
   151)    forc_hgt_q_pft => pps%forc_hgt_q_pft
   152) 
   153)    ! Adjustment factors for unstable (moz < 0) or stable (moz > 0) conditions.
   154) 
   155)    do f = 1, fn
   156)       n = filtern(f)
   157)       g = ngridcell(n)
   158) 
   159)       ! Wind profile
   160) 
   161)       if (present(landunit_index)) then
   162)         zldis(n) = forc_hgt_u_pft(pfti(n))-displa(n)
   163)       else
   164)         zldis(n) = forc_hgt_u_pft(n)-displa(n)
   165)       end if
   166)       zeta(n) = zldis(n)/obu(n)
   167)       if (zeta(n) < -zetam) then
   168)          ustar(n) = vkc*um(n)/(log(-zetam*obu(n)/z0m(n))&
   169)               - StabilityFunc1(-zetam) &
   170)               + StabilityFunc1(z0m(n)/obu(n)) &
   171)               + 1.14_r8*((-zeta(n))**0.333_r8-(zetam)**0.333_r8))
   172)       else if (zeta(n) < 0._r8) then
   173)          ustar(n) = vkc*um(n)/(log(zldis(n)/z0m(n))&
   174)               - StabilityFunc1(zeta(n))&
   175)               + StabilityFunc1(z0m(n)/obu(n)))
   176)       else if (zeta(n) <=  1._r8) then
   177)          ustar(n) = vkc*um(n)/(log(zldis(n)/z0m(n)) + 5._r8*zeta(n) -5._r8*z0m(n)/obu(n))
   178)       else
   179)          ustar(n) = vkc*um(n)/(log(obu(n)/z0m(n))+5._r8-5._r8*z0m(n)/obu(n) &
   180)               +(5._r8*log(zeta(n))+zeta(n)-1._r8))
   181)       end if
   182)       
   183)       if (zeta(n) < 0._r8) then
   184)          vds_tmp = 2.e-3_r8*ustar(n) * ( 1._r8 + (300._r8/(-obu(n)))**0.666_r8)
   185)       else
   186)          vds_tmp = 2.e-3_r8*ustar(n)
   187)       endif
   188) 
   189)       if (present(landunit_index)) then
   190)          do pp = pfti(n),pftf(n)
   191)             vds(pp) = vds_tmp
   192)          end do
   193)       else
   194)          vds(n) = vds_tmp
   195)       end if
   196) 
   197) ! Calculate a 10-m wind (10m + z0m + d)
   198) ! For now, this will not be the same as the 10-m wind calculated for the dust 
   199) ! model because the CLM stability functions are used here, not the LSM stability
   200) ! functions used in the dust model. We will eventually change the dust model to be 
   201) ! consistent with the following formulation.
   202) ! Note that the 10-m wind calculated this way could actually be larger than the
   203) ! atmospheric forcing wind because 1) this includes the convective velocity, 2)
   204) ! this includes the 1 m/s minimum wind threshold
   205) 
   206) ! If forcing height is less than or equal to 10m, then set 10-m wind to um
   207)       if (present(landunit_index)) then
   208)         do pp = pfti(n),pftf(n)
   209)           if (zldis(n)-z0m(n) .le. 10._r8) then
   210)             u10_clm(pp) = um(n)
   211)           else
   212)             if (zeta(n) < -zetam) then
   213)               u10_clm(pp) = um(n) - ( ustar(n)/vkc*(log(-zetam*obu(n)/(10._r8+z0m(n)))      &
   214)                                       - StabilityFunc1(-zetam)                              &
   215)                                       + StabilityFunc1((10._r8+z0m(n))/obu(n))              &
   216)                                       + 1.14_r8*((-zeta(n))**0.333_r8-(zetam)**0.333_r8)) )
   217)             else if (zeta(n) < 0._r8) then
   218)               u10_clm(pp) = um(n) - ( ustar(n)/vkc*(log(zldis(n)/(10._r8+z0m(n)))           &
   219)                                       - StabilityFunc1(zeta(n))                             &
   220)                                       + StabilityFunc1((10._r8+z0m(n))/obu(n))) )
   221)             else if (zeta(n) <=  1._r8) then
   222)               u10_clm(pp) = um(n) - ( ustar(n)/vkc*(log(zldis(n)/(10._r8+z0m(n)))           &
   223)                                       + 5._r8*zeta(n) - 5._r8*(10._r8+z0m(n))/obu(n)) )
   224)             else
   225)               u10_clm(pp) = um(n) - ( ustar(n)/vkc*(log(obu(n)/(10._r8+z0m(n)))             &
   226)                                       + 5._r8 - 5._r8*(10._r8+z0m(n))/obu(n)                &
   227)                                       + (5._r8*log(zeta(n))+zeta(n)-1._r8)) )
   228) 
   229)             end if
   230)           end if
   231)           va(pp) = um(n)
   232)         end do
   233)       else
   234)         if (zldis(n)-z0m(n) .le. 10._r8) then
   235)           u10_clm(n) = um(n)
   236)         else
   237)           if (zeta(n) < -zetam) then
   238)             u10_clm(n) = um(n) - ( ustar(n)/vkc*(log(-zetam*obu(n)/(10._r8+z0m(n)))         &
   239)                                    - StabilityFunc1(-zetam)                                 &
   240)                                    + StabilityFunc1((10._r8+z0m(n))/obu(n))                 &
   241)                                    + 1.14_r8*((-zeta(n))**0.333_r8-(zetam)**0.333_r8)) )
   242)           else if (zeta(n) < 0._r8) then
   243)             u10_clm(n) = um(n) - ( ustar(n)/vkc*(log(zldis(n)/(10._r8+z0m(n)))              &
   244)                                    - StabilityFunc1(zeta(n))                                &
   245)                                    + StabilityFunc1((10._r8+z0m(n))/obu(n))) )
   246)           else if (zeta(n) <=  1._r8) then
   247)             u10_clm(n) = um(n) - ( ustar(n)/vkc*(log(zldis(n)/(10._r8+z0m(n)))              &
   248)                                    + 5._r8*zeta(n) - 5._r8*(10._r8+z0m(n))/obu(n)) )
   249)           else
   250)             u10_clm(n) = um(n) - ( ustar(n)/vkc*(log(obu(n)/(10._r8+z0m(n)))                &
   251)                                    + 5._r8 - 5._r8*(10._r8+z0m(n))/obu(n)                   &
   252)                                    + (5._r8*log(zeta(n))+zeta(n)-1._r8)) )
   253)           end if
   254)         end if
   255)         va(n) = um(n)
   256)       end if
   257) 
   258)       ! Temperature profile
   259) 
   260)       if (present(landunit_index)) then
   261)         zldis(n) = forc_hgt_t_pft(pfti(n))-displa(n)
   262)       else
   263)         zldis(n) = forc_hgt_t_pft(n)-displa(n)
   264)       end if
   265)       zeta(n) = zldis(n)/obu(n)
   266)       if (zeta(n) < -zetat) then
   267)          temp1(n) = vkc/(log(-zetat*obu(n)/z0h(n))&
   268)               - StabilityFunc2(-zetat) &
   269)               + StabilityFunc2(z0h(n)/obu(n)) &
   270)               + 0.8_r8*((zetat)**(-0.333_r8)-(-zeta(n))**(-0.333_r8)))
   271)       else if (zeta(n) < 0._r8) then
   272)          temp1(n) = vkc/(log(zldis(n)/z0h(n)) &
   273)               - StabilityFunc2(zeta(n)) &
   274)               + StabilityFunc2(z0h(n)/obu(n)))
   275)       else if (zeta(n) <=  1._r8) then
   276)          temp1(n) = vkc/(log(zldis(n)/z0h(n)) + 5._r8*zeta(n) - 5._r8*z0h(n)/obu(n))
   277)       else
   278)          temp1(n) = vkc/(log(obu(n)/z0h(n)) + 5._r8 - 5._r8*z0h(n)/obu(n) &
   279)               + (5._r8*log(zeta(n))+zeta(n)-1._r8))
   280)       end if
   281) 
   282)       ! Humidity profile
   283) 
   284)       if (present(landunit_index)) then
   285)        if (forc_hgt_q_pft(pfti(n)) == forc_hgt_t_pft(pfti(n)) .and. z0q(n) == z0h(n)) then
   286)          temp2(n) = temp1(n)
   287)        else
   288)          zldis(n) = forc_hgt_q_pft(pfti(n))-displa(n)
   289)          zeta(n) = zldis(n)/obu(n)
   290)          if (zeta(n) < -zetat) then
   291)             temp2(n) = vkc/(log(-zetat*obu(n)/z0q(n)) &
   292)                  - StabilityFunc2(-zetat) &
   293)                  + StabilityFunc2(z0q(n)/obu(n)) &
   294)                  + 0.8_r8*((zetat)**(-0.333_r8)-(-zeta(n))**(-0.333_r8)))
   295)          else if (zeta(n) < 0._r8) then
   296)             temp2(n) = vkc/(log(zldis(n)/z0q(n)) &
   297)                  - StabilityFunc2(zeta(n)) &
   298)                  + StabilityFunc2(z0q(n)/obu(n)))
   299)          else if (zeta(n) <=  1._r8) then
   300)             temp2(n) = vkc/(log(zldis(n)/z0q(n)) + 5._r8*zeta(n)-5._r8*z0q(n)/obu(n))
   301)          else
   302)             temp2(n) = vkc/(log(obu(n)/z0q(n)) + 5._r8 - 5._r8*z0q(n)/obu(n) &
   303)                  + (5._r8*log(zeta(n))+zeta(n)-1._r8))
   304)          end if
   305)        end if
   306)       else
   307)        if (forc_hgt_q_pft(n) == forc_hgt_t_pft(n) .and. z0q(n) == z0h(n)) then
   308)          temp2(n) = temp1(n)
   309)        else
   310)          zldis(n) = forc_hgt_q_pft(n)-displa(n)
   311)          zeta(n) = zldis(n)/obu(n)
   312)          if (zeta(n) < -zetat) then
   313)             temp2(n) = vkc/(log(-zetat*obu(n)/z0q(n)) &
   314)                  - StabilityFunc2(-zetat) &
   315)                  + StabilityFunc2(z0q(n)/obu(n)) &
   316)                  + 0.8_r8*((zetat)**(-0.333_r8)-(-zeta(n))**(-0.333_r8)))
   317)          else if (zeta(n) < 0._r8) then
   318)             temp2(n) = vkc/(log(zldis(n)/z0q(n)) &
   319)                  - StabilityFunc2(zeta(n)) &
   320)                  + StabilityFunc2(z0q(n)/obu(n)))
   321)          else if (zeta(n) <=  1._r8) then
   322)             temp2(n) = vkc/(log(zldis(n)/z0q(n)) + 5._r8*zeta(n)-5._r8*z0q(n)/obu(n))
   323)          else
   324)             temp2(n) = vkc/(log(obu(n)/z0q(n)) + 5._r8 - 5._r8*z0q(n)/obu(n) &
   325)                  + (5._r8*log(zeta(n))+zeta(n)-1._r8))
   326)          end if
   327)        endif
   328)       endif
   329) 
   330)       ! Temperature profile applied at 2-m
   331) 
   332)       zldis(n) = 2.0_r8 + z0h(n)
   333)       zeta(n) = zldis(n)/obu(n)
   334)       if (zeta(n) < -zetat) then
   335)          temp12m(n) = vkc/(log(-zetat*obu(n)/z0h(n))&
   336)               - StabilityFunc2(-zetat) &
   337)               + StabilityFunc2(z0h(n)/obu(n)) &
   338)               + 0.8_r8*((zetat)**(-0.333_r8)-(-zeta(n))**(-0.333_r8)))
   339)       else if (zeta(n) < 0._r8) then
   340)          temp12m(n) = vkc/(log(zldis(n)/z0h(n)) &
   341)               - StabilityFunc2(zeta(n))  &
   342)               + StabilityFunc2(z0h(n)/obu(n)))
   343)       else if (zeta(n) <=  1._r8) then
   344)          temp12m(n) = vkc/(log(zldis(n)/z0h(n)) + 5._r8*zeta(n) - 5._r8*z0h(n)/obu(n))
   345)       else
   346)          temp12m(n) = vkc/(log(obu(n)/z0h(n)) + 5._r8 - 5._r8*z0h(n)/obu(n) &
   347)               + (5._r8*log(zeta(n))+zeta(n)-1._r8))
   348)       end if
   349) 
   350)       ! Humidity profile applied at 2-m
   351) 
   352)       if (z0q(n) == z0h(n)) then
   353)          temp22m(n) = temp12m(n)
   354)       else
   355)          zldis(n) = 2.0_r8 + z0q(n)
   356)          zeta(n) = zldis(n)/obu(n)
   357)          if (zeta(n) < -zetat) then
   358)             temp22m(n) = vkc/(log(-zetat*obu(n)/z0q(n)) - &
   359)                  StabilityFunc2(-zetat) + StabilityFunc2(z0q(n)/obu(n)) &
   360)                  + 0.8_r8*((zetat)**(-0.333_r8)-(-zeta(n))**(-0.333_r8)))
   361)          else if (zeta(n) < 0._r8) then
   362)             temp22m(n) = vkc/(log(zldis(n)/z0q(n)) - &
   363)                  StabilityFunc2(zeta(n))+StabilityFunc2(z0q(n)/obu(n)))
   364)          else if (zeta(n) <=  1._r8) then
   365)             temp22m(n) = vkc/(log(zldis(n)/z0q(n)) + 5._r8*zeta(n)-5._r8*z0q(n)/obu(n))
   366)          else
   367)             temp22m(n) = vkc/(log(obu(n)/z0q(n)) + 5._r8 - 5._r8*z0q(n)/obu(n) &
   368)                  + (5._r8*log(zeta(n))+zeta(n)-1._r8))
   369)          end if
   370)       end if
   371) 
   372)       ! diagnose 10-m wind for dust model (dstmbl.F)
   373)       ! Notes from C. Zender's dst.F:
   374)       ! According to Bon96 p. 62, the displacement height d (here displa) is
   375)       ! 0.0 <= d <= 0.34 m in dust source regions (i.e., regions w/o trees).
   376)       ! Therefore d <= 0.034*z1 and may safely be neglected.
   377)       ! Code from LSM routine SurfaceTemperature was used to obtain u10
   378) 
   379)       if (present(landunit_index)) then
   380)         zldis(n) = forc_hgt_u_pft(pfti(n))-displa(n)
   381)       else
   382)         zldis(n) = forc_hgt_u_pft(n)-displa(n)
   383)       end if
   384)       zeta(n) = zldis(n)/obu(n)
   385)       if (min(zeta(n), 1._r8) < 0._r8) then
   386)          tmp1 = (1._r8 - 16._r8*min(zeta(n),1._r8))**0.25_r8
   387)          tmp2 = log((1._r8+tmp1*tmp1)/2._r8)
   388)          tmp3 = log((1._r8+tmp1)/2._r8)
   389)          fmnew = 2._r8*tmp3 + tmp2 - 2._r8*atan(tmp1) + 1.5707963_r8
   390)       else
   391)          fmnew = -5._r8*min(zeta(n),1._r8)
   392)       endif
   393)       if (iter == 1) then
   394)          fm(n) = fmnew
   395)       else
   396)          fm(n) = 0.5_r8 * (fm(n)+fmnew)
   397)       end if
   398)       zeta10 = min(10._r8/obu(n), 1._r8)
   399)       if (zeta(n) == 0._r8) zeta10 = 0._r8
