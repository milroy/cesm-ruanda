   198)   subroutine timemgr_init( )
   199) 
   200)     !---------------------------------------------------------------------------------
   201)     ! Initialize the ESMF time manager from the sync clock
   202)     ! 
   203)     ! Arguments
   204)     !
   205)     character(len=*), parameter :: sub = 'clm::timemgr_init'
   206)     integer :: rc                            ! return code
   207)     integer :: yr, mon, day, tod             ! Year, month, day, and second as integers
   208)     type(ESMF_Time) :: start_date            ! start date for run
   209)     type(ESMF_Time) :: stop_date             ! stop date for run
   210)     type(ESMF_Time) :: curr_date             ! temporary date used in logic
   211)     type(ESMF_Time) :: ref_date              ! reference date for time coordinate
   212)     logical :: run_length_specified = .false.
   213)     type(ESMF_Time) :: current               ! current date (from clock)
   214)     type(ESMF_TimeInterval) :: day_step_size ! day step size
   215)     type(ESMF_TimeInterval) :: step_size     ! timestep size
   216)     !---------------------------------------------------------------------------------
   217)     call timemgr_spmdbcast( )
   218) 
   219)     ! Initalize calendar 
   220) 
   221)     call init_calendar()
   222) 
   223)     ! Initalize start date.
   224) 
   225)     if ( start_ymd == uninit_int ) then
   226)        write(iulog,*)sub,': start_ymd must be specified '
   227)        call shr_sys_abort
   228)     end if
   229)     if ( start_tod == uninit_int ) then
   230)        write(iulog,*)sub,': start_tod must be specified '
   231)        call shr_sys_abort
   232)     end if
   233)     start_date = TimeSetymd( start_ymd, start_tod, "start_date" )
   234) 
   235)     ! Initialize current date
   236) 
   237)     curr_date = start_date
   238) 
   239)     ! Initalize stop date.
   240) 
   241)     stop_date = TimeSetymd( 99991231, stop_tod, "stop_date" )
   242) 
   243)     call ESMF_TimeIntervalSet( step_size, s=dtime, rc=rc )
   244)     call chkrc(rc, sub//': error return from ESMF_TimeIntervalSet: setting step_size')
   245) 
   246)     call ESMF_TimeIntervalSet( day_step_size, d=1, rc=rc )
   247)     call chkrc(rc, sub//': error return from ESMF_TimeIntervalSet: setting day_step_size')
   248) 
   249)     if ( stop_ymd /= uninit_int ) then
   250)        current = TimeSetymd( stop_ymd, stop_tod, "stop_date" )
   251)        if ( current < stop_date ) stop_date = current
   252)        run_length_specified = .true.
   253)     end if
   254)     if ( nelapse /= uninit_int ) then
   255)        if ( nelapse >= 0 ) then
   256)           current = curr_date + step_size*nelapse
   257)        else
   258)           current = curr_date - day_step_size*nelapse
   259)        end if
   260)        if ( current < stop_date ) stop_date = current
   261)        run_length_specified = .true.
   262)     end if
   263)     if ( .not. run_length_specified ) then
   264)        call shr_sys_abort (sub//': Must specify stop_ymd or nelapse')
   265)     end if
   266) 
   267)     ! Error check 
   268) 
   269)     if ( stop_date <= start_date ) then
   270)        write(iulog,*)sub, ': stop date must be specified later than start date: '
   271)        call ESMF_TimeGet( start_date, yy=yr, mm=mon, dd=day, s=tod )
   272)        write(iulog,*) ' Start date (yr, mon, day, tod): ', yr, mon, day, tod
   273)        call ESMF_TimeGet( stop_date, yy=yr, mm=mon, dd=day, s=tod )
   274)        write(iulog,*) ' Stop date  (yr, mon, day, tod): ', yr, mon, day, tod
   275)        call shr_sys_abort
   276)     end if
   277)     if ( curr_date >= stop_date ) then
   278)        write(iulog,*)sub, ': stop date must be specified later than current date: '
   279)        call ESMF_TimeGet( curr_date, yy=yr, mm=mon, dd=day, s=tod )
   280)        write(iulog,*) ' Current date (yr, mon, day, tod): ', yr, mon, day, tod
   281)        call ESMF_TimeGet( stop_date, yy=yr, mm=mon, dd=day, s=tod )
   282)        write(iulog,*) ' Stop date    (yr, mon, day, tod): ', yr, mon, day, tod
   283)        call shr_sys_abort
   284)     end if
   285) 
   286)     ! Initalize reference date for time coordinate.
   287) 
   288)     if ( ref_ymd /= uninit_int ) then
   289)        ref_date = TimeSetymd( ref_ymd, ref_tod, "ref_date" )
   290)     else
   291)        ref_date = start_date
   292)     end if
   293) 
   294)     ! Initialize clock
   295) 
   296)     call init_clock( start_date, ref_date, curr_date, stop_date )
   297) 
   298)     ! Initialize date used for perpetual calendar day calculation.
   299) 
   300)     if (tm_perp_calendar) then
   301)        tm_perp_date = TimeSetymd( perpetual_ymd, 0, "tm_perp_date" )
   302)     end if
   303) 
   304)     ! Print configuration summary to log file (stdout).
   305) 
   306)     if (masterproc) call timemgr_print()
   307) 
   308)     timemgr_set = .true.
   309) 
   310)   end subroutine timemgr_init

   314)   subroutine init_clock( start_date, ref_date, curr_date, stop_date )
   315) 
   316)     !---------------------------------------------------------------------------------
   317)     ! Purpose: Initialize the clock based on the start_date, ref_date, and curr_date
   318)     ! as well as the settings from the namelist specifying the time to stop
   319)     !
   320)     type(ESMF_Time), intent(in) :: start_date  ! start date for run
   321)     type(ESMF_Time), intent(in) :: ref_date    ! reference date for time coordinate
   322)     type(ESMF_Time), intent(in) :: curr_date   ! current date (equal to start_date)
   323)     type(ESMF_Time), intent(in) :: stop_date   ! stop date for run
   324)     !
   325)     character(len=*), parameter :: sub = 'clm::init_clock'
   326)     type(ESMF_TimeInterval)     :: step_size         ! timestep size
   327)     type(ESMF_Time)             :: current           ! current date (from clock)
   328)     integer                     :: rc                ! return code
   329)     !---------------------------------------------------------------------------------
   330) 
   331)     call ESMF_TimeIntervalSet( step_size, s=dtime, rc=rc )
   332)     call chkrc(rc, sub//': error return from ESMF_TimeIntervalSet: setting step_size')
   333) 
   334)     ! Initialize the clock
   335) 
   336)     tm_clock = ESMF_ClockCreate(name="CLM Time-manager clock", timeStep=step_size, startTime=start_date, &
   337)          stopTime=stop_date, refTime=ref_date, rc=rc)
   338)     call chkrc(rc, sub//': error return from ESMF_ClockSetup')
   339) 
   340)     ! Advance clock to the current time (in case of a restart)
   341) 
   342)     call ESMF_ClockGet(tm_clock, currTime=current, rc=rc )
   343)     call chkrc(rc, sub//': error return from ESMF_ClockGet')
   344)     do while( curr_date > current )
   345)        call ESMF_ClockAdvance( tm_clock, rc=rc )
   346)        call chkrc(rc, sub//': error return from ESMF_ClockAdvance')
   347)        call ESMF_ClockGet(tm_clock, currTime=current )
   348)        call chkrc(rc, sub//': error return from ESMF_ClockGet')
   349)     end do
   350)   end subroutine init_clock

   354)   function TimeSetymd( ymd, tod, desc )
   355)     !---------------------------------------------------------------------------------
   356)     !
   357)     ! Set the time by an integer as YYYYMMDD and integer seconds in the day
   358)     !
   359)     integer, intent(in) :: ymd            ! Year, month, day YYYYMMDD
   360)     integer, intent(in) :: tod            ! Time of day in seconds
   361)     character(len=*), intent(in) :: desc  ! Description of time to set
   362) 
   363)     type(ESMF_Time) :: TimeSetymd    ! Return value

   365)     character(len=*), parameter :: sub = 'clm::TimeSetymd'

   367)     integer :: rc                    ! return code

   370)     if ( (ymd < 0) .or. (tod < 0) .or. (tod > isecspday) )then

   372)             ymd, tod
   373)        call shr_sys_abort
   374)     end if
   375)     yr  = ymd / 10000
   376)     mon = (ymd - yr*10000) / 100
   377)     day =  ymd - yr*10000 - mon*100
   378)     call ESMF_TimeSet( TimeSetymd, yy=yr, mm=mon, dd=day, s=tod, &
   379)          calendar=tm_cal, rc=rc)
   380)     call chkrc(rc, sub//': error return from ESMF_TimeSet: setting '//trim(desc))

   528)   subroutine timemgr_restart( )
   529) 
   530)     !---------------------------------------------------------------------------------
   531)     ! Restart the ESMF time manager using the synclock for ending date.
   532)     !
   533)     character(len=*), parameter :: sub = 'clm::timemgr_restart'
   534)     integer :: rc                            ! return code
   535)     integer :: yr, mon, day, tod             ! Year, month, day, and second as integers
   536)     type(ESMF_Time) :: start_date            ! start date for run
   537)     type(ESMF_Time) :: ref_date              ! reference date for run
   538)     type(ESMF_Time) :: curr_date             ! date of data in restart file
   539)     type(ESMF_Time) :: stop_date             ! stop date for run
   540)     type(ESMF_Time) :: current               ! current date (from clock)
   541)     type(ESMF_TimeInterval) :: day_step_size ! day step size
   542)     type(ESMF_TimeInterval) :: step_size     ! timestep size
   543)     logical :: run_length_specified = .false.
   544)     !---------------------------------------------------------------------------------
   545)     call timemgr_spmdbcast( )
   546) 
   547)     ! Initialize calendar from restart info
   548) 
   549)     call init_calendar()
   550) 
   551)     ! Initialize the timestep from restart info
   552) 
   553)     dtime = rst_step_sec
   554) 
   555)     ! Initialize start date from restart info
   556) 
   557)     start_date = TimeSetymd( rst_start_ymd, rst_start_tod, "start_date" )
   558) 
   559)     ! Initialize current date from restart info
   560) 
   561)     curr_date = TimeSetymd( rst_curr_ymd, rst_curr_tod, "curr_date" )
   562) 
   563)     ! Initialize stop date from sync clock or namelist input
   564) 
   565)     stop_date = TimeSetymd( 99991231, stop_tod, "stop_date" )
   566) 
   567)     call ESMF_TimeIntervalSet( step_size, s=dtime, rc=rc )
   568)     call chkrc(rc, sub//': error return from ESMF_TimeIntervalSet: setting step_size')
   569) 
   570)     call ESMF_TimeIntervalSet( day_step_size, d=1, rc=rc )
   571)     call chkrc(rc, sub//': error return from ESMF_TimeIntervalSet: setting day_step_size')
   572) 
   573)     if    ( stop_ymd /= uninit_int ) then
   574)        current = TimeSetymd( stop_ymd, stop_tod, "stop_date" )
   575)        if ( current < stop_date ) stop_date = current
   576)        run_length_specified = .true.
   577)     else if ( nelapse /= uninit_int ) then
   578)        if ( nelapse >= 0 ) then
   579)           current = curr_date + step_size*nelapse
   580)        else
   581)           current = curr_date - day_step_size*nelapse
   582)        end if
   583)        if ( current < stop_date ) stop_date = current
   584)        run_length_specified = .true.
   585)     end if
   586)     if ( .not. run_length_specified ) then
   587)        call shr_sys_abort (sub//': Must specify stop_ymd or nelapse')
   588)     end if
   589) 
   590)     ! Error check
   591) 
   592)     if ( stop_date <= start_date ) then
   593)        write(iulog,*)sub, ': stop date must be specified later than start date: '
   594)        call ESMF_TimeGet( start_date, yy=yr, mm=mon, dd=day, s=tod )
   595)        write(iulog,*) ' Start date (yr, mon, day, tod): ', yr, mon, day, tod
   596)        call ESMF_TimeGet( stop_date, yy=yr, mm=mon, dd=day, s=tod )
   597)        write(iulog,*) ' Stop date  (yr, mon, day, tod): ', yr, mon, day, tod
   598)        call shr_sys_abort
   599)     end if
   600)     if ( curr_date >= stop_date ) then
   601)        write(iulog,*)sub, ': stop date must be specified later than current date: '
   602)        call ESMF_TimeGet( curr_date, yy=yr, mm=mon, dd=day, s=tod )
   603)        write(iulog,*) ' Current date (yr, mon, day, tod): ', yr, mon, day, tod
   604)        call ESMF_TimeGet( stop_date, yy=yr, mm=mon, dd=day, s=tod )
   605)        write(iulog,*) ' Stop date    (yr, mon, day, tod): ', yr, mon, day, tod
   606)        call shr_sys_abort
   607)     end if
   608) 
   609)     ! Initialize nstep_rad_prev from restart info
   610) 
   611)     nstep_rad_prev = rst_nstep_rad_prev
   612) 
   613)     ! Initialize ref date from restart info
   614) 
   615)     ref_date = TimeSetymd( rst_ref_ymd, rst_ref_tod, "ref_date" )
   616) 
   617)     ! Initialize clock 
   618) 
   619)     call init_clock( start_date, ref_date, curr_date, stop_date )
   620) 
   621)     ! Advance the timestep.  
   622)     ! Data from the restart file corresponds to the last timestep of the previous run.
   623) 
   624)     call advance_timestep()
   625) 
   626)     ! Set flag that this is the first timestep of the restart run.
   627) 
   628)     tm_first_restart_step = .true.
   629) 
   630)     ! Calculate ending time step
   631) 
   632)     call calc_nestep( )
   633) 
   634)     ! Print configuration summary to log file (stdout).
   635) 
   636)     if (masterproc) call timemgr_print()
   637) 
   638)     timemgr_set = .true.
   639) 
   640)   end subroutine timemgr_restart

   644)   subroutine calc_nestep()
   645)     !---------------------------------------------------------------------------------
   646)     !
   647)     ! Calculate ending timestep number
   648)     ! Calculation of ending timestep number (nestep) assumes a constant stepsize.
   649)     !
   650)     character(len=*), parameter :: sub = 'clm::calc_nestep'
   651)     integer :: ntspday               ! Number of time-steps per day
   652)     type(ESMF_TimeInterval) :: diff  !
   653)     type(ESMF_Time) :: start_date    ! start date for run
   654)     type(ESMF_Time) :: stop_date     ! stop date for run
   655)     integer :: ndays, nsecs          ! Number of days, seconds to ending time
   656)     integer :: rc                    ! return code
   657)     !---------------------------------------------------------------------------------
   658) 
   659)     call ESMF_ClockGet( tm_clock, stopTime=stop_date, startTime=start_date, rc=rc )
   660)     call chkrc(rc, sub//': error return from ESMF_ClockGet')
   661)     ntspday = isecspday/dtime
   662)     diff = stop_date - start_date
   663)     call ESMF_TimeIntervalGet( diff, d=ndays, s=nsecs, rc=rc )
   664)     call chkrc(rc, sub//': error return from ESMF_TimeIntervalGet calculating nestep')
   665)     nestep = ntspday*ndays + nsecs/dtime
   666)     if ( mod(nsecs,dtime) /= 0 ) nestep = nestep + 1
   667)   end subroutine calc_nestep
