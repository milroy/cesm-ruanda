    54) ! !INTERFACE:
    55)   subroutine VOCEmission (lbp, ubp, num_soilp, filter_soilp )
    56) !
    57) ! ! NEW DESCRIPTION
    58) ! Volatile organic compound emission
    59) ! This code simulates volatile organic compound emissions following
    60) ! MEGAN (Model of Emissions of Gases and Aerosols from Nature) v2.1 
    61) ! for 20 compound classes. The original description of this
    62) ! algorithm (for isoprene only) can be found in Guenther et al., 2006
    63) ! (we follow equations 2-9, 16-17, 20 for explicit canopy).
    64) ! The model scheme came be described as:
    65) !    E= epsilon * gamma * rho
    66) ! VOC flux (E) [ug m-2 h-1] is calculated from baseline emission
    67) ! factors (epsilon) [ug m-2 h-1] which are specified for each of the 16
    68) ! CLM PFTs (in input file) OR in the case of isoprene, from
    69) ! mapped EFs for each PFT which reflect species divergence of emissions,
    70) ! particularly in North America. 
    71) ! The emission activity factor (gamma) [unitless] for includes 
    72) ! dependence on PPFT, temperature, LAI, leaf age and soil moisture.
    73) ! For isoprene only we also include the effect of CO2 inhibition as
    74) ! described by Heald et al., 2009. 
    75) ! The canopy environment constant was calculated offline for CLM+CAM at 
    76) ! standard conditions.
    77) ! We assume that the escape efficiency (rho) here is unity following
    78) ! Guenther et al., 2006.
    79) ! A manuscript describing MEGAN 2.1 and the implementation in CLM is
    80) ! in preparation: Guenther, Heald et al., 2012
    81) ! Subroutine written to operate at the patch level.
    82) !
    83) ! Input: <filename> to be read in with EFs and some parameters.  
    84) !        Currently these are set in procedure init_EF_params
    85) ! Output: vocflx(shr_megan_mechcomps_n) !VOC flux [moles/m2/sec]
    86) !
    87) !
    88) ! !USES:
    89)     use shr_kind_mod , only : r8 => shr_kind_r8
    90)     use clm_atmlnd   , only : clm_a2l
    91)     use clmtype
    92)     use domainMod,     only : ldomain 
    93)     use clm_varcon   , only : spval
    94) !
    95) ! !ARGUMENTS:
    96)     implicit none
    97)     integer, intent(in) :: lbp, ubp                    ! pft bounds
    98)     integer, intent(in) :: num_soilp                   ! number of columns in soil pft filter
    99)     integer, intent(in) :: filter_soilp(num_soilp)     ! pft filter for soil
   100) !
   101) ! !CALLED FROM:
   102) !
   103) ! !REVISION HISTORY:
   104) ! Author: Sam Levis
   105) ! 2/1/02: Peter Thornton: migration to new data structure
   106) ! 4/15/06: Colette L. Heald: modify for updated MEGAN model (Guenther et al., 2006)
   107) ! 4/29/11: Colette L. Heald: expand MEGAN to 20 compound classes
   108) ! 7 Feb 2012: Francis Vitt: Implemented capability to specify MEGAN emissions in namelist
   109) !                           and read in MEGAN factors from file.
   110) !
   111) ! !LOCAL VARIABLES:
   112) !
   113) ! local pointers to implicit in arguments
   114) !
   115)     integer , pointer :: pgridcell(:)     ! gridcell index of corresponding pft
   116)     integer , pointer :: pcolumn(:)       ! column index of corresponding pft
   117)     integer , pointer :: ivt(:)           ! pft vegetation type for current
   118)     real(r8), pointer :: t_veg(:)         ! pft vegetation temperature (Kelvin)
   119)     real(r8), pointer :: fsun(:)          ! sunlit fraction of canopy
   120)     real(r8), pointer :: elai(:)          ! one-sided leaf area index with burying by snow
   121)     real(r8), pointer :: clayfrac(:)      ! fraction of soil that is clay
   122)     real(r8), pointer :: sandfrac(:)      ! fraction of soil that is sand
   123)     real(r8), pointer :: forc_solad(:,:)  ! direct beam radiation (visible only)
   124)     real(r8), pointer :: forc_solai(:,:)  ! diffuse radiation     (visible only)
   125)     real(r8), pointer :: elai_p(:)        ! one-sided leaf area index from previous timestep
   126)     real(r8), pointer :: t_veg24(:)       ! avg pft vegetation temperature for last 24 hrs
   127)     real(r8), pointer :: t_veg240(:)      ! avg pft vegetation temperature for last 240 hrs
   128)     real(r8), pointer :: fsun24(:)        ! sunlit fraction of canopy last 24 hrs
   129)     real(r8), pointer :: fsun240(:)       ! sunlit fraction of canopy last 240 hrs
   130)     real(r8), pointer :: forc_solad24(:)  ! direct beam radiation last 24hrs  (visible only)
   131)     real(r8), pointer :: forc_solai24(:)  ! diffuse radiation  last 24hrs     (visible only)
   132)     real(r8), pointer :: forc_solad240(:) ! direct beam radiation last 240hrs (visible only)
   133)     real(r8), pointer :: forc_solai240(:) ! diffuse radiation  last 240hrs    (visible only)
   134)     real(r8), pointer :: h2osoi_vol(:,:)  ! volumetric soil water (m3/m3)
   135)     real(r8), pointer :: h2osoi_ice(:,:)  ! ice soil content (kg/m3)
   136)     real(r8), pointer :: dz(:,:)          ! depth of layer (m)
   137)     real(r8), pointer :: bsw(:,:)         ! Clapp and Hornberger "b" (nlevgrnd)
   138)     real(r8), pointer :: watsat(:,:)      ! volumetric soil water at saturation (porosity) (nlevgrnd)
   139)     real(r8), pointer :: sucsat(:,:)      ! minimum soil suction (mm) (nlevgrnd)
   140)     real(r8), pointer :: cisun(:)         ! sunlit intracellular CO2 (Pa)
   141)     real(r8), pointer :: cisha(:)         ! shaded intracellular CO2 (Pa)
   142)     real(r8), pointer :: forc_pbot(:)     ! atmospheric pressure (Pa)
   143) !
   144) ! local pointers to original implicit out arrays
   145) !
   146)     real(r8), pointer :: vocflx(:,:)      ! VOC flux [moles/m2/sec]
   147)     real(r8), pointer :: vocflx_tot(:)    ! VOC flux [moles/m2/sec]
   148) 
   149)     type(megan_out_type), pointer :: meg_out(:) ! fluxes for CLM history 
   150) 
   151)     real(r8), pointer :: gamma_out(:)
   152)     real(r8), pointer :: gammaT_out(:)
   153)     real(r8), pointer :: gammaP_out(:)
   154)     real(r8), pointer :: gammaL_out(:)
   155)     real(r8), pointer :: gammaA_out(:)
   156)     real(r8), pointer :: gammaS_out(:)
   157)     real(r8), pointer :: gammaC_out(:)
   158) 
   159)     real(r8), pointer :: Eopt_out(:)     
   160)     real(r8), pointer :: topt_out(:)
   161)     real(r8), pointer :: alpha_out(:)
   162)     real(r8), pointer :: cp_out(:)
   163)     real(r8), pointer :: paru_out(:)
   164)     real(r8), pointer :: par24u_out(:)
   165)     real(r8), pointer :: par240u_out(:)
   166)     real(r8), pointer :: para_out(:)
   167)     real(r8), pointer :: par24a_out(:)
   168)     real(r8), pointer :: par240a_out(:)
   169) 
   170) !
   171) !
   172) ! !OTHER LOCAL VARIABLES:
   173) !
   174)     integer  :: fp,p,g,c                ! indices
   175)     real(r8) :: epsilon                 ! emission factor [ug m-2 h-1]
   176)     real(r8) :: par_sun                 ! temporary
   177)     real(r8) :: par24_sun               ! temporary
   178)     real(r8) :: par240_sun              ! temporary
   179)     real(r8) :: par_sha                 ! temporary
   180)     real(r8) :: par24_sha               ! temporary
   181)     real(r8) :: par240_sha              ! temporary
   182)     real(r8) :: gamma                   ! activity factor (accounting for light, T, age, LAI conditions)
   183)     real(r8) :: gamma_p                 ! activity factor for PPFD
   184)     real(r8) :: gamma_l                 ! activity factor for PPFD & LAI
   185)     real(r8) :: gamma_t                 ! activity factor for temperature
   186)     real(r8) :: gamma_a                 ! activity factor for leaf age
   187)     real(r8) :: gamma_sm                ! activity factor for soil moisture
   188)     real(r8) :: gamma_c                 ! activity factor for CO2 (only isoprene)
   189)     
   190)     integer :: class_num, n_meg_comps, imech, imeg, ii
   191)     character(len=16) :: mech_name
   192) 
   193)     real(r8) :: vocflx_meg(shr_megan_megcomps_n)
   194)     type(shr_megan_megcomp_t), pointer :: meg_cmp
   195) 
   196)     real(r8) :: cp, alpha,  Eopt, topt  ! for history output
   197) 
   198)     ! factor used convert MEGAN units [micro-grams/m2/hr] to CAM srf emis units [g/m2/sec]
   199)     real(r8), parameter :: megemis_units_factor = 1._r8/3600._r8/1.e6_r8
   200) 
   201) !    real(r8) :: root_depth(0:numpft)    ! Root depth [m]
   202) !
   203) !!-----------------------------------------------------------------------
   204) !
   205) !    ! root depth (m) (defined based on Zeng et al., 2001, cf Guenther 2006)
   206) !    root_depth(noveg)                                     = 0._r8   ! bare-soil
   207) !    root_depth(ndllf_evr_tmp_tree:ndllf_evr_brl_tree)     = 1.8_r8  ! evergreen tree
   208) !    root_depth(ndllf_dcd_brl_tree)                        = 2.0_r8  ! needleleaf deciduous boreal tree
   209) !    root_depth(nbrdlf_evr_trp_tree:nbrdlf_evr_tmp_tree)   = 3.0_r8  ! broadleaf evergreen tree
   210) !    root_depth(nbrdlf_dcd_trp_tree:nbrdlf_dcd_brl_tree)   = 2.0_r8  ! broadleaf deciduous tree
   211) !    root_depth(nbrdlf_evr_shrub:nbrdlf_dcd_brl_shrub)     = 2.5_r8  ! shrub
   212) !    root_depth(nc3_arctic_grass:numpft)                   = 1.5_r8  ! grass/crop
   213) !
   214) !-----------------------------------------------------------------------
   215)     if ( shr_megan_mechcomps_n < 1) return
   216) 
   217)     ! Assign local pointers to derived type members (gridcell-level)
   218)     forc_solad => clm_a2l%forc_solad
   219)     forc_solai => clm_a2l%forc_solai
   220)     forc_pbot  => clm_a2l%forc_pbot
   221) 
   222)     ! Assign local pointers to derived subtypes components (column-level)
   223)     h2osoi_vol       => cws%h2osoi_vol
   224)     h2osoi_ice       => cws%h2osoi_ice
   225)     dz               => cps%dz
   226)     bsw              => cps%bsw
   227)     watsat           => cps%watsat
   228)     sucsat           => cps%sucsat
   229) 
   230)     ! Assign local pointers to derived subtypes components (pft-level)
   231) 
   232)     pgridcell        => pft%gridcell
   233)     pcolumn          => pft%column
   234)     ivt              => pft%itype
   235)     t_veg            => pes%t_veg
   236)     fsun             => pps%fsun
   237)     elai             => pps%elai
   238)     clayfrac         => pps%clayfrac
   239)     sandfrac         => pps%sandfrac
   240) 
   241)     cisun            => pps%cisun
   242)     cisha            => pps%cisha
   243) 
   244)     vocflx           => pvf%vocflx
   245)     vocflx_tot       => pvf%vocflx_tot
   246)     meg_out          => pvf%meg
   247) 
   248)     gammaL_out       => pvf%gammaL_out
   249)     gammaT_out       => pvf%gammaT_out
   250)     gammaP_out       => pvf%gammaP_out
   251)     gammaA_out       => pvf%gammaA_out
   252)     gammaS_out       => pvf%gammaS_out
   253)     gammaC_out       => pvf%gammaC_out
   254)     gamma_out        => pvf%gamma_out
   255) 
   256)     Eopt_out         => pvf%Eopt_out
   257)     topt_out         => pvf%topt_out
   258)     alpha_out        => pvf%alpha_out
   259)     cp_out           => pvf%cp_out
   260)     paru_out         => pvf%paru_out
   261)     par24u_out       => pvf%par24u_out
   262)     par240u_out      => pvf%par240u_out
   263)     para_out         => pvf%para_out
   264)     par24a_out       => pvf%par24a_out
   265)     par240a_out      => pvf%par240a_out
   266) 
   267)     t_veg24          => pvs%t_veg24
   268)     t_veg240         => pvs%t_veg240
   269)     forc_solad24     => pvs%fsd24
   270)     forc_solad240    => pvs%fsd240
   271)     forc_solai24     => pvs%fsi24
   272)     forc_solai240    => pvs%fsi240
   273)     fsun24           => pvs%fsun24
   274)     fsun240          => pvs%fsun240
   275)     elai_p           => pvs%elai_p
   276) 
   277)     ! initialize variables which get passed to the atmosphere
   278)     vocflx(lbp:ubp,:) = 0._r8
   279)     vocflx_tot(lbp:ubp) = 0._r8
   280) 
   281)     do imeg=1,shr_megan_megcomps_n
   282)       meg_out(imeg)%flux_out(lbp:ubp) = 0._r8
   283)     enddo
   284)     
   285)     gamma_out(lbp:ubp) = spval
   286)     gammaP_out(lbp:ubp) = spval
   287)     gammaT_out(lbp:ubp) = spval
   288)     gammaA_out(lbp:ubp) = spval
   289)     gammaS_out(lbp:ubp) = spval
   290)     gammaL_out(lbp:ubp) = spval
   291)     gammaC_out(lbp:ubp) = spval
   292) 
   293)     paru_out(lbp:ubp) = spval
   294)     par24u_out(lbp:ubp) = spval
   295)     par240u_out(lbp:ubp) = spval
   296) 
   297)     para_out(lbp:ubp) = spval
   298)     par24a_out(lbp:ubp) = spval
   299)     par240a_out(lbp:ubp) = spval
   300) 
   301)     alpha_out(lbp:ubp) = spval
   302)     cp_out(lbp:ubp) = spval
   303) 
   304)     topt_out(lbp:ubp) = spval
   305)     Eopt_out(lbp:ubp) = spval
   306) 
   307)     ! Begin loop over points
   308)     !_______________________________________________________________________________
   309)     do fp = 1,num_soilp
   310)        p = filter_soilp(fp)
   311)        g = pgridcell(p)
   312)        c = pcolumn(p)
   313) 
   314)        ! initialize EF
   315)        epsilon=0._r8
   316)        
   317)        ! initalize to zero since this might not alway get set
   318)        ! this needs to be within the fp loop ... 
   319)        vocflx_meg(:) = 0._r8
   320) 
   321)        ! calculate VOC emissions for non-bare ground PFTs
   322)        if (ivt(p) > 0) then 
   323)           gamma=0._r8
   324) 
   325)           ! Calculate PAR: multiply w/m2 by 4.6 to get umol/m2/s for par (added 8/14/02)
   326)           !------------------------
   327)           ! SUN:
   328)           par_sun = (forc_solad(g,1) + fsun(p) * forc_solai(g,1)) * 4.6_r8
   329)           par24_sun = (forc_solad24(p) + fsun24(p) * forc_solai24(p)) * 4.6_r8
   330)           par240_sun = (forc_solad240(p) + fsun240(p) * forc_solai240(p)) * 4.6_r8
   331)           ! SHADE:
   332)           par_sha = ((1._r8 - fsun(p)) * forc_solai(g,1)) * 4.6_r8
   333)           par24_sha = ((1._r8 - fsun24(p)) * forc_solai24(p)) * 4.6_r8
   334)           par240_sha = ((1._r8 - fsun240(p)) * forc_solai240(p)) * 4.6_r8
   335) 
   336)           ! Activity factor for LAI (Guenther et al., 2006): all species
   337)           gamma_l = get_gamma_L(fsun240(p), elai(p))
   338) 
   339)           ! Activity factor for soil moisture: all species (commented out for now)
   340) !          gamma_sm = get_gamma_SM(clayfrac(p), sandfrac(p), h2osoi_vol(c,:), h2osoi_ice(c,:), &
   341) !               dz(c,:), bsw(c,:), watsat(c,:), sucsat(c,:), root_depth(ivt(p)))
   342)           gamma_sm = 1.0_r8
   343) 
   344)           ! Loop through VOCs for light, temperature and leaf age activity factor & apply
   345)           ! all final activity factors to baseline emission factors
   346)           !_______________________________________________________________________________
   347) 
   348)           ! loop over megan compounds
   349)           meg_cmp => shr_megan_linkedlist
   350)           meg_cmp_loop: do while(associated(meg_cmp))
   351)              imeg = meg_cmp%index
   352) 
   353)              ! set emis factor
   354)              ! if specified, set EF for isoprene with mapped values
   355)              if ( trim(meg_cmp%name) == 'isoprene' .and. shr_megan_mapped_emisfctrs) then
   356)                 epsilon = get_map_EF(ivt(p),g)
   357)              else
   358)                 epsilon = meg_cmp%emis_factors(ivt(p))
   359)              end if
   360) 
   361)              class_num = meg_cmp%class_number
   362) 
   363)              ! Activity factor for PPFD
   364)              gamma_p = get_gamma_P(par_sun, par24_sun, par240_sun, par_sha, par24_sha, par240_sha, &
   365)                   fsun(p), fsun240(p), forc_solad240(p),forc_solai240(p), LDF(class_num), cp, alpha)
   366) 
   367)              ! Activity factor for T
   368)              gamma_t = get_gamma_T(t_veg240(p), t_veg24(p),t_veg(p), ct1(class_num), ct2(class_num),&
   369)                                    betaT(class_num),LDF(class_num), Ceo(class_num), Eopt, topt)
   370) 
   371)              ! Activity factor for Leaf Age
   372)              gamma_a = get_gamma_A(ivt(p), elai_p(p),elai(p),class_num)
   373) 
   374)              ! Activity factor for CO2 (only for isoprene)
   375)              if (trim(meg_cmp%name) == 'isoprene') then 
   376)                 gamma_c = get_gamma_C(cisun(p),cisha(p),forc_pbot(g),fsun(p))
   377)              else
   378)                 gamma_c = 1._r8
   379)              end if
   380) 
   381)              ! Calculate total scaling factor
   382)              gamma = gamma_l * gamma_sm * gamma_a * gamma_p * gamma_T * gamma_c
   383) 
   384)              if ( (gamma >=0.0_r8) .and. (gamma< 100._r8) ) then
   385) 
   386)                 vocflx_meg(imeg) = epsilon * gamma * megemis_units_factor / meg_cmp%molec_weight ! moles/m2/sec
   387) 
   388)                 ! assign to arrays for history file output (not weighted by landfrac)
   389)                 meg_out(imeg)%flux_out(p) = meg_out(imeg)%flux_out(p) &
   390)                                           + epsilon * gamma * megemis_units_factor*1.e-3_r8 ! Kg/m2/sec
   391) 
   392)                 if (imeg==1) then 
   393)                    ! 
   394)                    gamma_out(p)=gamma
   395)                    gammaP_out(p)=gamma_p
   396)                    gammaT_out(p)=gamma_t
   397)                    gammaA_out(p)=gamma_a
   398)                    gammaS_out(p)=gamma_sm
   399)                    gammaL_out(p)=gamma_l
   400)                    gammaC_out(p)=gamma_c
   401) 
   402)                    paru_out(p)=par_sun
   403)                    par24u_out(p)=par24_sun
   404)                    par240u_out(p)=par240_sun
   405) 
   406)                    para_out(p)=par_sha
   407)                    par24a_out(p)=par24_sha
   408)                    par240a_out(p)=par240_sha
   409) 
   410)                    alpha_out(p)=alpha
   411)                    cp_out(p)=cp
   412) 
   413)                    topt_out(p)=topt
   414)                    Eopt_out(p)=Eopt
   415) 
   416)                 end if
   417)              endif
   418) 
   419)              if (debug .and. gamma > 0.0_r8) then
   420)                 write(iulog,*) 'MEGAN: n, megan name, epsilon, gamma, vocflx: ', &
   421)                      imeg, meg_cmp%name, epsilon, gamma, vocflx_meg(imeg), gamma_p,gamma_t,gamma_a,gamma_sm,gamma_l
   422)              endif
   423) 
   424)              meg_cmp => meg_cmp%next_megcomp
   425)           enddo meg_cmp_loop
   426) 
   427)           ! sum up the megan compound fluxes for the fluxes of chem mechanism compounds 
   428)           do imech = 1,shr_megan_mechcomps_n
   429)              n_meg_comps = shr_megan_mechcomps(imech)%n_megan_comps
   430)              do imeg = 1,n_meg_comps ! loop over number of megan compounds that make up the nth mechanism compoud
   431)                 ii = shr_megan_mechcomps(imech)%megan_comps(imeg)%ptr%index
   432)                 vocflx(p,imech) = vocflx(p,imech) + vocflx_meg(ii)
   433)              enddo
   434)              vocflx_tot(p) = vocflx_tot(p) + vocflx(p,imech) ! moles/m2/sec
   435)           enddo
   436) 
   437)        end if ! ivt(1:15 only)
   438) 
   439)     enddo ! fp 
   440) 
   441)   end subroutine VOCEmission

   445) ! !IROUTINE: init_EF_params
   446) !
   447) ! !INTERFACE:
   448)   subroutine VOCEmission_init(  )
   449)     ! Interface to set all input parameters for 20 VOC compound classes.
   450)     ! including EFs for 16(+1 bare ground) PFTs.

   555) ! FUNCTION: get_gamma_P
   556) !
   557) ! !INTERFACE:
   558)   function get_gamma_P(par_sun_in, par24_sun_in, par240_sun_in, par_sha_in, par24_sha_in, par240_sha_in, &
   559)        fsun_in, fsun240_in, forc_solad240_in,forc_solai240_in, LDF_in, cp, alpha) 
   560)   
   561) ! Activity factor for PPFD (Guenther et al., 2006): all light dependent species
   562) !-------------------------
   563) ! With distinction between sunlit and shaded leafs, weight scalings by
   564) ! fsun and fshade 
   565) ! Scale total incident par by fraction of sunlit leaves (added on 1/2002)
   566)     
   567) ! fvitt -- forc_solad240, forc_solai240 can be zero when CLM finidat is specified
   568) !          which will cause par240 to be zero and produce NaNs via log(par240)
   569) ! dml   -- fsun240 can be equal to or greater than one before 10 day averages are
   570) !           set on startup or if a new pft comes online during land cover change.
   571) !           Avoid this problem by only doing calculations with fsun240 when fsun240 is
   572) !           between 0 and 1
   573) !
   574) ! !CALLED FROM: VOCEmission
   575) !
   576) ! !REVISION HISTORY:
   577) ! Author: Colette L. Heald (4/27/11)
   578) !
   579) !
   580) ! !ARGUMENTS:
   581)     implicit none
   582) !
   583) ! !LOCAL VARIABLES:
   584) 
   585)     ! varibles in
   586)     real(r8),intent(in) :: par_sun_in
   587)     real(r8),intent(in) :: par24_sun_in
   588)     real(r8),intent(in) :: par240_sun_in
   589)     real(r8),intent(in) :: par_sha_in
   590)     real(r8),intent(in) :: par24_sha_in
   591)     real(r8),intent(in) :: par240_sha_in
   592)     real(r8),intent(in) :: fsun_in
   593)     real(r8),intent(in) :: fsun240_in
   594)     real(r8),intent(in) :: forc_solad240_in
   595)     real(r8),intent(in) :: forc_solai240_in
   596)     real(r8),intent(in) :: LDF_in
   597) 
   598)     real(r8),intent(out) :: cp                      ! temporary
   599)     real(r8),intent(out) :: alpha                   ! temporary
   600)     real(r8) :: gamma_p_LDF             ! activity factor for PPFD
   601)     real(r8) :: get_gamma_P             ! return value
   602) 
   603)     real(r8), parameter :: ca1 = 0.004_r8                  ! empirical coefficent for alpha
   604)     real(r8), parameter :: ca2 = 0.0005_r8                 ! empirical coefficent for alpha
   605)     real(r8), parameter :: ca3 = 0.0468_r8                 ! empirical coefficent for cp
   606)     real(r8), parameter :: par0_sun = 200._r8              ! std conditions for past 24 hrs [umol/m2/s]
   607)     real(r8), parameter :: par0_shade = 50._r8             ! std conditions for past 24 hrs [umol/m2/s]
   608)     real(r8), parameter :: alpha_fix = 0.001_r8            ! empirical coefficient
   609)     real(r8), parameter :: cp_fix = 1.21_r8                ! empirical coefficient
   610) !
   611) ! local pointers to implicit in arguments
   612) !
   613) !-----------------------------------------------------------------------
   614) 
   615)   if ( (fsun240_in > 0._r8) .and. (fsun240_in < 1._r8) .and.  (forc_solad240_in > 0._r8) &
   616)        .and. (forc_solai240_in > 0._r8)) then
   617)      ! With alpha and cp calculated based on eq 6 and 7:
   618)      ! Note indexing for accumulated variables is all at pft level
   619)      ! SUN:
   620)      alpha = ca1 - ca2 * log(par240_sun_in)
   621)      cp = ca3 * exp(ca2 * (par24_sun_in-par0_sun))*par240_sun_in**(0.6_r8)
   622)      gamma_p_LDF = fsun_in * ( cp * alpha * par_sun_in * (1._r8 + alpha*alpha*par_sun_in*par_sun_in)**(-0.5_r8) )
   623)      ! SHADE:
   624)      alpha = ca1 - ca2 * log(par240_sha_in)
   625)      cp = ca3 * exp(ca2 * (par_sha_in-par0_shade))*par240_sha_in**(0.6_r8)
   626)      gamma_p_LDF = gamma_p_LDF + (1._r8-fsun_in) * (cp*alpha*par_sha_in*(1._r8 + alpha*alpha*par_sha_in*par_sha_in)**(-0.5_r8))
   627)   else
   628)      ! With fixed alpha and cp (from MEGAN User's Guide):
   629)      ! SUN: direct + diffuse  
   630)      alpha = alpha_fix
   631)      cp = cp_fix
   632)      gamma_p_LDF = fsun_in * ( cp * alpha*par_sun_in * (1._r8 + alpha*alpha*par_sun_in*par_sun_in)**(-0.5_r8) )
   633)      ! SHADE: diffuse 
   634)      gamma_p_LDF = gamma_p_LDF + (1._r8-fsun_in) * (cp*alpha*par_sha_in*(1._r8 + alpha*alpha*par_sha_in*par_sha_in)**(-0.5_r8))
   635)   end if
   636)   
   637)   ! Calculate total activity factor for PPFD accounting for light-dependent fraction
   638)   get_gamma_P = (1._r8 - LDF_in) + LDF_in * gamma_p_LDF
   639) 
   640) end function get_gamma_P

   688) !
   689) ! !INTERFACE:
   690) function get_gamma_SM(clayfrac_in, sandfrac_in, h2osoi_vol_in, h2osoi_ice_in, dz_in, &
   691)      bsw_in, watsat_in, sucsat_in, root_depth_in)
   692) !
   693) ! Activity factor for soil moisture (Guenther et al., 2006): all species
   694) !----------------------------------
   695) ! Calculate the mean scaling factor throughout the root depth.
   696) ! wilting point potential is in units of matric potential (mm) 
   697) ! (1 J/Kg = 0.001 MPa, approx = 0.1 m)
   698) ! convert to volumetric soil water using equation 7.118 of the CLM4 Technical Note
   699) !
   700) ! !CALLED FROM: VOCEmission
   701) !
   702) ! !REVISION HISTORY:
   703) ! Author: Colette L. Heald (4/27/11)
   704) !
   705) 
   706) ! !USES:
   707)     use clm_varcon   , only : denice
   708)     use clm_varpar   , only : nlevsoi
   709) !
   710) ! !ARGUMENTS:
   711)     implicit none
   712) !
   713) ! !LOCAL VARIABLES:
   714) 
   715)     ! varibles in
   716)     real(r8),intent(in) :: clayfrac_in
   717)     real(r8),intent(in) :: sandfrac_in
   718)     real(r8),intent(in) :: h2osoi_vol_in(nlevsoi)
   719)     real(r8),intent(in) :: h2osoi_ice_in(nlevsoi)
   720)     real(r8),intent(in) :: dz_in(nlevsoi)
   721)     real(r8),intent(in) :: bsw_in(nlevsoi)
   722)     real(r8),intent(in) :: watsat_in(nlevsoi)
   723)     real(r8),intent(in) :: sucsat_in(nlevsoi)
   724)     real(r8),intent(in) :: root_depth_in
   725)  
   726)     real(r8)            :: get_gamma_SM
   727) 
   728)     ! local variables
   729)     integer  :: j
   730)     real(r8) :: nl                      ! temporary number of soil levels
   731)     real(r8) :: theta_ice               ! water content in ice in m3/m3
   732)     real(r8) :: wilt                    ! wilting point in m3/m3
   733)     real(r8) :: theta1                  ! temporary
   734) 
   735)     ! parameters
   736)     real(r8), parameter :: deltheta1=0.06_r8               ! empirical coefficient
   737)     real(r8), parameter :: smpmax = 2.57e5_r8              ! maximum soil matrix potential
   738) 
   739)     if ((clayfrac_in > 0) .and. (sandfrac_in > 0)) then 
   740)        get_gamma_SM = 0._r8
   741)        nl=0._r8
   742)        
   743)        do j = 1,nlevsoi
   744)           if  (sum(dz_in(1:j)) < root_depth_in)  then
   745)              theta_ice = h2osoi_ice_in(j)/(dz_in(j)*denice)
   746)              wilt = ((smpmax/sucsat_in(j))**(-1._r8/bsw_in(j))) * (watsat_in(j) - theta_ice)
   747)              theta1 = wilt + deltheta1
   748)              if (h2osoi_vol_in(j) >= theta1) then 
   749)                 get_gamma_SM = get_gamma_SM + 1._r8
   750)              else if ( (h2osoi_vol_in(j) > wilt) .and. (h2osoi_vol_in(j) < theta1) ) then
   751)                 get_gamma_SM = get_gamma_SM + ( h2osoi_vol_in(j) - wilt ) / deltheta1
   752)              else
   753)                 get_gamma_SM = get_gamma_SM + 0._r8
   754)              end if
   755)              nl=nl+1._r8
   756)           end if
   757)        end do
   758)        
   759)        if (nl > 0._r8) then
   760)           get_gamma_SM = get_gamma_SM/nl
   761)        endif
   762) 
   763)        if (get_gamma_SM > 1.0_r8) then 
   764)           write(iulog,*) 'healdSM > 1: gamma_SM, nl', get_gamma_SM, nl
   765)           get_gamma_SM=1.0_r8
   766)        endif
   767) 
   768)     else
   769)        get_gamma_SM = 1.0_r8
   770)     end if
   771) 
   772) 
   773) end function get_gamma_SM

   779) ! !INTERFACE:
   780) function get_gamma_T(t_veg240_in, t_veg24_in,t_veg_in, ct1_in, ct2_in, betaT_in, LDF_in, Ceo_in, Eopt, topt)
   781) 
   782) ! Activity factor for temperature 
   783) !--------------------------------
   784) ! Calculate both a light-dependent fraction as in Guenther et al., 2006 for isoprene
   785) ! of a max saturation type form. Also caculate a light-independent fraction of the
   786) ! form of an exponential. Final activity factor depends on light dependent fraction
   787) ! of compound type.
   788) !
   789) ! !CALLED FROM: VOCEmission
   790) !
   791) ! !REVISION HISTORY:
   792) ! Author: Colette L. Heald (4/27/11)
   793) !
   794) ! !ARGUMENTS:
   795)     implicit none
   796) !
   797) ! !LOCAL VARIABLES:
   798) 
   799)     ! varibles in
   800)     real(r8),intent(in) :: t_veg240_in
   801)     real(r8),intent(in) :: t_veg24_in
   802)     real(r8),intent(in) :: t_veg_in
   803)     real(r8),intent(in) :: ct1_in
   804)     real(r8),intent(in) :: ct2_in
   805)     real(r8),intent(in) :: betaT_in
   806)     real(r8),intent(in) :: LDF_in
   807)     real(r8),intent(in) :: Ceo_in
   808)     real(r8),intent(out) :: Eopt                    ! temporary 
   809)     real(r8),intent(out) :: topt                    ! temporary 
   810) 
   811)     ! local variables
   812)     real(r8) :: get_gamma_T
   813)     real(r8) :: gamma_t_LDF             ! activity factor for temperature
   814)     real(r8) :: gamma_t_LIF             ! activity factor for temperature
   815)     real(r8) :: x                       ! temporary 
   816) 
   817)     ! parameters
   818)     real(r8), parameter :: co1 = 313._r8                   ! empirical coefficient
   819)     real(r8), parameter :: co2 = 0.6_r8                    ! empirical coefficient
   820)     real(r8), parameter :: co4 = 0.05_r8                   ! empirical coefficient
   821)     real(r8), parameter :: tstd0 = 297_r8                  ! std temperature [K]
   822)     real(r8), parameter :: topt_fix = 317._r8              ! std temperature [K]
   823)     real(r8), parameter :: Eopt_fix = 2.26_r8              ! empirical coefficient
   824)     real(r8), parameter :: ct3 = 0.00831_r8                ! empirical coefficient (0.0083 in User's Guide)
   825)     real(r8), parameter :: tstd = 303.15_r8                ! std temperature [K]
   826)     real(r8), parameter :: bet = 0.09_r8                   ! beta empirical coefficient [K-1]
   827) !-----------------------------------------------------------------------
   828) 
   829)     ! Light dependent fraction (Guenther et al., 2006)
   830)     if ( (t_veg240_in > 0.0_r8) .and. (t_veg240_in < 1.e30_r8) ) then 
   831)        ! topt and Eopt from eq 8 and 9:
   832)        topt = co1 + (co2 * (t_veg240_in-tstd0))
   833)        Eopt = Ceo_in * exp (co4 * (t_veg24_in-tstd0)) * exp(co4 * (t_veg240_in -tstd0))
   834)     else
   835)        topt = topt_fix
   836)        Eopt = Eopt_fix
   837)     endif
   838)     x = ( (1._r8/topt) - (1._r8/(t_veg_in)) ) / ct3
   839)     gamma_t_LDF = Eopt * ( ct2_in * exp(ct1_in * x)/(ct2_in - ct1_in * (1._r8 - exp(ct2_in * x))) )
   840)     
   841)     
   842)     ! Light independent fraction (of exp(beta T) form)
   843)     gamma_t_LIF = exp(betaT_in * (t_veg_in - tstd))
   844)     
   845)     ! Calculate total activity factor for light as a function of light-dependent fraction
   846)     !--------------------------------
   847)     get_gamma_T = (1-LDF_in)*gamma_T_LIF + LDF_in*gamma_T_LDF 
   848) 
   849) end function get_gamma_T

   853) ! FUNCTION: get_gamma_A
   854) !
   855) ! !INTERFACE:
   856) function get_gamma_A(ivt_in, elai_p_in,elai_in,nclass_in)
   857) 
   858) ! Activity factor for leaf age (Guenther et al., 2006)
   859) !-----------------------------
   860) ! If not CNDV elai is constant therefore gamma_a=1.0
   861) ! gamma_a set to unity for evergreens (PFTs 1, 2, 4, 5)
   862) ! Note that we assume here that the time step is shorter than the number of 
   863) !days after budbreak required to induce isoprene emissions (ti=12 days) and 
   864) ! the number of days after budbreak to reach peak emission (tm=28 days)
   865) !
   866) ! !CALLED FROM: VOCEmission
   867) !
   868) ! !REVISION HISTORY:
   869) ! Author: Colette L. Heald (4/27/11)
   870) !
   871) ! !ARGUMENTS:
   872)     implicit none
   873) ! !LOCAL VARIABLES:
   874) 
   875)     ! varibles in
   876)     integer,intent(in)  :: ivt_in
   877)     integer,intent(in)  :: nclass_in
   878)     real(r8),intent(in) :: elai_p_in
   879)     real(r8),intent(in) :: elai_in
   880) 
   881)     real(r8)            :: get_gamma_A
   882) 
   883)     ! local variables
   884)     real(r8) :: elai_prev               ! lai for previous timestep
   885)     real(r8) :: fnew, fgro, fmat, fold  ! fractions of leaves at different phenological stages
   886)     !-----------------------------------------------------------------------
   887)     if ( (ivt_in == ndllf_dcd_brl_tree) .or. (ivt_in >= nbrdlf_dcd_trp_tree) ) then  ! non-evergreen
   888)        
   889)        if ( (elai_p_in > 0.0_r8) .and. (elai_p_in < 1.e30_r8) )then 
   890)           elai_prev = 2._r8*elai_p_in-elai_in  ! have accumulated average lai over last timestep
   891)           if (elai_prev == elai_in) then
   892)              fnew = 0.0_r8
   893)              fgro = 0.0_r8
   894)              fmat = 1.0_r8
   895)              fold = 0.0_r8
   896)           else if (elai_prev > elai_in) then
   897)              fnew = 0.0_r8
   898)              fgro = 0.0_r8
   899)              fmat = 1.0_r8 - (elai_prev - elai_in)/elai_prev
   900)              fold = (elai_prev - elai_in)/elai_prev
   901)           else if (elai_prev < elai_in) then
   902)              fnew = 1 - (elai_prev / elai_in)
   903)              fgro = 0.0_r8
   904)              fmat = (elai_prev / elai_in)
   905)              fold = 0.0_r8
   906)           end if
   907)           
   908)           get_gamma_A = fnew*Anew(nclass_in) + fgro*Agro(nclass_in) + fmat*Amat(nclass_in) + fold*Aold(nclass_in)
   909) 
   910)        else
   911)           get_gamma_A = 1.0_r8
   912)        end if
   913)        
   914)     else
   915)        get_gamma_A = 1.0_r8
   916)     end if
   917)     
   918) 
   919)   end function get_gamma_A

   930) ! With distinction between sunlit and shaded leafs, weight scalings by
