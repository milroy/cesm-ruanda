    80)   subroutine aircraft_emit_register()
    81) 
    82) !------------------------------------------------------------------
    83) ! **** Add the aircraft aerosol data to the physics buffer ****
    84) !------------------------------------------------------------------
    85)     use ppgrid,         only: pver,pcols
    86)     use physics_buffer, only : pbuf_add_field, dtype_r8

    89) 
    90)     integer :: i,idx, mm, ind, n
    91)     character(len=16) :: spc_name
    92)     character(len=256) :: filelist, curr_filename
    93)     character(len=128) :: long_name
    94)     logical            :: has_fixed_ubc=.false.
    95)     logical            :: read_iv=.false.
    96) 
    97)     !------------------------------------------------------------------
    98)     ! Return if air_specifier is blank (no aircraft data to process)
    99)     !------------------------------------------------------------------
   100)     if (air_specifier(1) == "") return
   101) 
   102) ! count aircraft emission species used in the simulation
   103)     count_emis: do n=1,N_AERO
   104) 	
   105) 	if( len_trim(air_specifier(n) ) == 0 ) then
   106) 	    exit count_emis
   107)         endif
   108) 
   109)         i = scan(air_specifier(n),'->')
   110)         spc_name = trim(adjustl(air_specifier(n)(:i-1)))
   111)         filelist = trim(adjustl(air_specifier(n)(i+2:)))
   112) 
   113)         mm = get_aircraft_ndx(spc_name)
   114)         if( mm < 1 ) then
   115) 	 call endrun('aircraft_emit_register: '//trim(spc_name)//' is not in the aircraft emission dataset')
   116)         endif
   117) 
   118)         aircraft_cnt = aircraft_cnt + 1
   119)         call pbuf_add_field(aero_names(mm),'physpkg',dtype_r8,(/pcols,pver/),idx)
   120) 
   121)         spc_flist(aircraft_cnt) = filelist
   122)         spc_name_list(aircraft_cnt) = spc_name
   123)         index_map(aircraft_cnt) = mm
   124) 
   125)         curr_filename=''	
   126)         datapath=''
   127)         spc_fname(aircraft_cnt) = incr_filename( curr_filename, filenames_list=spc_flist(aircraft_cnt), datapath=datapath)
   128) 
   129)         if( advective_tracer(mm) ) then
   130)           long_name = 'aircraft_'//trim(spc_name)
   131)           call cnst_add(aero_names(mm),molmass(mm),cptmp,qmin,ind,longname=long_name,readiv=read_iv, &
   132)                         mixtype=mixtype(mm),cam_outfld=cam_outfld,fixed_ubc=has_fixed_ubc)
   133)         endif
   134) 
   135)     enddo count_emis
   136) ! count aircraft emission species used in the simulation
   137)

   264)   subroutine aircraft_emit_adv( state, pbuf2d)
   265) !-------------------------------------------------------------------
   266) ! **** Advance to the next aircraft data ****
   267) !-------------------------------------------------------------------
   268) 
   269)     use tracer_data,  only : advance_trcdata
   270)     use physics_types,only : physics_state
   271)     use ppgrid,       only : begchunk, endchunk
   272)     use ppgrid,       only : pcols, pver
   273)     use string_utils, only : to_lower, GLC
   274)     use cam_history,  only : outfld
   275)     use physconst,    only : mwdry       ! molecular weight dry air ~ kg/kmole
   276)     use physconst,    only : boltz                ! J/K/molecule
   277) ! C.-C. Chen
   278)     use physconst,    only : gravit, rearth
   279)     use phys_grid,    only : get_wght_all_p
   280)     
   281)     use physics_buffer, only : physics_buffer_desc, pbuf_set_field, pbuf_get_field, pbuf_get_chunk
   282)   
   283)     implicit none
   284) 
   285)     type(physics_state), intent(in)    :: state(begchunk:endchunk)                 
   286)     
   287)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   288) 
   289)     type(physics_buffer_desc), pointer :: pbuf_chnk(:)
   290)     integer :: ind,c,ncol,i,caseid,m,n
   291)     real(r8) :: to_mmr(pcols,pver)
   292)     real(r8),pointer :: tmpptr(:,:)
   293)    
   294) ! C.-C. Chen
   295)     real(r8) :: wght(pcols)
   296) 
   297)    !------------------------------------------------------------------
   298)    ! Return if aircraft_cnt is zero (no aircraft data to process)
   299)    !------------------------------------------------------------------
   300)     if (aircraft_cnt == 0 ) return
   301)     call t_startf('All_aircraft_emit_adv')
   302) 
   303)    !-------------------------------------------------------------------
   304)    !    For each field, read more data if needed and interpolate it to the current model time
   305)    !-------------------------------------------------------------------
   306)     do m = 1, aircraft_cnt
   307)        call advance_trcdata( forcings_air(m)%fields, forcings_air(m)%file, state, pbuf2d)
   308)     
   309)    !-------------------------------------------------------------------
   310)    !    set the tracer fields with the correct units
   311)    !-------------------------------------------------------------------
   312)        do i = 1,number_flds
   313) 
   314) ! C.-C. Chen, adding case 4  for kg/sec
   315)           select case ( to_lower(trim(forcings_air(m)%fields(i)%units(:GLC(forcings_air(m)%fields(i)%units)))) )
   316)           case ("molec/cm3","/cm3","molecules/cm3","cm^-3","cm**-3")
   317)              caseid = 1
   318)           case ('kg/kg','mmr')
   319)              caseid = 2
   320)           case ('mol/mol','mole/mole','vmr','fraction')
   321)              caseid = 3
   322)           case ('kg/kg/sec')
   323)               caseid = 4
   324)           case default
   325)              print*, 'aircraft_emit_adv: units = ',trim(forcings_air(m)%fields(i)%units) ,' are not recognized'
   326)              call endrun('aircraft_emit_adv: units are not recognized')
   327)           end select
   328) 
   329)           ind = index_map(i)
   330) 
   331) !$OMP PARALLEL DO PRIVATE (C, NCOL, TO_MMR, tmpptr, pbuf_chnk)
   332)           do c = begchunk,endchunk
   333)              ncol = state(c)%ncol
   334) 
   335) ! C.-C. Chen, turning emission data to mixing ratio
   336)              call get_wght_all_p(c,ncol,wght(:ncol))
   337) 
   338)              if (caseid == 1) then
   339)                 to_mmr(:ncol,:) = (molmass(ind)*1.e6_r8*boltz*state(c)%t(:ncol,:))/(mwdry*state(c)%pmiddry(:ncol,:))
   340)              elseif (caseid == 2) then
   341)                 to_mmr(:ncol,:) = 1._r8
   342)              elseif (caseid == 4) then
   343) !                do n=1,ncol
   344) !                  to_mmr(n,:) = 1.0_r8/(rearth*rearth*wght(n)*state(c)%pdel(n,:)/gravit)
   345) !                end do
   346)                 to_mmr(:ncol,:) = 1.0_r8
   347)              else
   348)                 to_mmr(:ncol,:) = molmass(ind)/mwdry
   349)              endif
   350)              pbuf_chnk => pbuf_get_chunk(pbuf2d, c)
   351)              call pbuf_get_field(pbuf_chnk, forcings_air(m)%fields(i)%pbuf_ndx, tmpptr )
   352)    
   353)              tmpptr(:ncol,:) = tmpptr(:ncol,:)*to_mmr(:ncol,:)
   354) 
   355)              call outfld( forcings_air(m)%fields(i)%fldnam, &
   356)                   tmpptr, ncol, state(c)%lchnk )
   357) 
   358)           enddo
   359)        enddo
   360)     enddo
   361) 
   362)     call t_stopf('All_aircraft_emit_adv')
   363)   end subroutine aircraft_emit_adv
