   112) subroutine physprop_accum_unique_files(radname, type)
   113) 
   114)    ! Count number of aerosols in input radname array.  Aerosols are identified
   115)    ! as strings with a ".nc" suffix.
   116)    ! Construct a cumulative list of unique filenames containing physical property data.
   117) 
   118)    character(len=*), intent(in)  :: radname(:)
   119)    character(len=1), intent(in)  :: type(:)
   120) 
   121)    integer :: ncnst, i
   122)    character(len=*), parameter :: subname = 'physprop_accum_unique_files'
   123)    !------------------------------------------------------------------------------------
   124) 
   125)    ! Initial guess for number of files we need.
   126)    if (.not. allocated(uniquefilenames)) allocate(uniquefilenames(50))
   127) 
   128)    ncnst = ubound(radname, 1)
   129) 
   130)    do i = 1, ncnst
   131) 
   132)       ! check if radname is either a bulk aerosol or a mode
   133)       if (type(i) == 'A' .or. type(i) == 'M') then
   134) 
   135)          ! check if this filename has been used by another aerosol.  If not
   136)          ! then add it to the list of unique names.
   137)          if (physprop_get_id(radname(i)) < 0) then
   138)             numphysprops = numphysprops + 1
   139)             if (numphysprops > size(uniquefilenames)) then
   140)                call double_capacity(uniquefilenames)
   141)             end if
   142)             uniquefilenames(numphysprops) = trim(radname(i))
   143)          endif
   144) 
   145)       endif
   146)    enddo
   147) 
   148)  contains
   149) 
   150)    ! Simple routine to re-allocate an array with twice the size, but with
   151)    ! the inital values being preserved.
   152)    subroutine double_capacity(array)
   153)      character(len=256), intent(inout), allocatable :: array(:)
   154) 
   155)      character(len=256), allocatable :: tmp(:)
   156)      integer :: ierr
   157) 
   158)      allocate(tmp(size(array)*2), stat=ierr)
   159)      if ( ierr /= 0 ) then
   160)         call endrun('physprop_accum_unique_files: Allocation error.')
   161)      end if
   162) 
   163)      tmp(:size(array)) = array
   164) 
   165)      deallocate(array, stat=ierr)
   166)      if ( ierr /= 0 ) then
   167)         call endrun('physprop_accum_unique_files: Deallocation error.')
   168)      end if
   169) 
   170)      call move_alloc(tmp, array)
   171) 
   172)    end subroutine double_capacity
   173) 
   174) end subroutine physprop_accum_unique_files

   178) subroutine physprop_init()
   179) 
   180)    ! Read properties from the aerosol data files.
   181) 
   182)    ! ***N.B.*** The calls to physprop_accum_unique_files must be made before calling
   183)    !            this init routine.  physprop_accum_unique_files is responsible for building
   184)    !            the list of files to be read here.
   185) 
   186)    ! Local variables
   187)    integer            :: fileindex

   189)    character(len=256) :: locfn ! path to actual file used
   190)    character(len=32)  :: aername_str ! string read from netCDF file -- may contain trailing
   191)                                      ! nulls which aren't dealt with by trim()
   192)    
   193)    integer :: ierr ! error codes from mpi
   194) 
   195)    !------------------------------------------------------------------------------------
   196) 
   197)    allocate(physprop(numphysprops))
   198) 
   199)    do fileindex = 1, numphysprops
   200)       nullify(physprop(fileindex)%sw_hygro_ext)
   201)       nullify(physprop(fileindex)%sw_hygro_ssa)
   202)       nullify(physprop(fileindex)%sw_hygro_asm)
   203)       nullify(physprop(fileindex)%lw_hygro_abs)
   204) 
   205)       nullify(physprop(fileindex)%sw_nonhygro_ext)
   206)       nullify(physprop(fileindex)%sw_nonhygro_ssa)
   207)       nullify(physprop(fileindex)%sw_nonhygro_asm)
   208)       nullify(physprop(fileindex)%sw_nonhygro_scat)
   209)       nullify(physprop(fileindex)%sw_nonhygro_ascat)
   210)       nullify(physprop(fileindex)%lw_abs)
   211) 
   212)       nullify(physprop(fileindex)%refindex_aer_sw)
   213)       nullify(physprop(fileindex)%refindex_aer_lw)
   214) 
   215)       nullify(physprop(fileindex)%r_sw_ext)
   216)       nullify(physprop(fileindex)%r_sw_scat)
   217)       nullify(physprop(fileindex)%r_sw_ascat)
   218)       nullify(physprop(fileindex)%r_lw_abs)
   219)       nullify(physprop(fileindex)%mu)
   220) 
   221)       nullify(physprop(fileindex)%extpsw)
   222)       nullify(physprop(fileindex)%abspsw)
   223)       nullify(physprop(fileindex)%asmpsw)
   224)       nullify(physprop(fileindex)%absplw)
   225)       nullify(physprop(fileindex)%refrtabsw)
   226)       nullify(physprop(fileindex)%refitabsw)
   227)       nullify(physprop(fileindex)%refrtablw)
   228)       nullify(physprop(fileindex)%refitablw)
   229) 
   230)       call getfil(uniquefilenames(fileindex), locfn, 0)
   231)       physprop(fileindex)%sourcefile = locfn
   232) 
   233)       ! Open the physprop file
   234)       call cam_pio_openfile(nc_id, locfn, PIO_NOWRITE)
   235) 
   236)       call aerosol_optics_init(physprop(fileindex), nc_id)
   237) 
   238)       ! Close the physprop file
   239)       call pio_closefile(nc_id)
   240) 
   241)    end do
   242) end subroutine physprop_init

   246) integer function physprop_get_id(filename)
   247) 
   248)    ! Look for filename in the global list of unique filenames (module data uniquefilenames).
   249)    ! If found, return it's index in the list.  Otherwise return -1.
   250) 
   251)    character(len=*), intent(in) :: filename

   388) subroutine aerosol_optics_init(phys_prop, nc_id)
   389) 
   390)    ! Determine the opticstype, then call the 
   391)    ! appropriate routine to read the data.
   392) 
   393)    type(physprop_type), intent(inout) :: phys_prop  ! data after interp onto cam rh mesh
   394)    type(file_desc_t),   intent(inout) :: nc_id      ! indentifier for netcdf file
   395) 
   396)    integer :: opticslength_id, opticslength
   397)    type(var_desc_t) :: op_type_id
   398)    integer :: ierr ! mpi error codes
   399)    character(len=ot_length)  :: opticstype_str ! string read from netCDF file -- may contain trailing
   400)                                         ! nulls which aren't dealt with by trim()
   401)    !------------------------------------------------------------------------------------
   402) 
   403)    ierr = pio_inq_dimid(nc_id, 'opticsmethod_len', opticslength_id)
   404)    ierr = pio_inq_dimlen(nc_id, opticslength_id, opticslength)
   405)    if ( opticslength .gt. ot_length ) then
   406)       call endrun(" optics type length in "//phys_prop%sourcefile//" excedes maximum length of 32")
   407)    endif
   408)    ierr = pio_inq_varid(nc_id, 'opticsmethod', op_type_id)
   409)    ierr = pio_get_var(nc_id, op_type_id,phys_prop%opticsmethod )
   410) 
   411)    select case (phys_prop%opticsmethod)
   412)    case ('zero')
   413)       call zero_optics_init(phys_prop, nc_id)
   414) 
   415)    case ('hygro')
   416)       call hygro_optics_init(phys_prop, nc_id)
   417) 
   418)    case ('hygroscopic')
   419)       call hygroscopic_optics_init(phys_prop, nc_id)
   420) 
   421)    case ('nonhygro')
   422)       call nonhygro_optics_init(phys_prop, nc_id)
   423)         
   424)    case ('insoluble')
   425)       call insoluble_optics_init(phys_prop, nc_id)
   426)         
   427)    case ('volcanic_radius')
   428)       call volcanic_radius_optics_init(phys_prop, nc_id)
   429) 
   430)    case ('volcanic')
   431)       call volcanic_optics_init(phys_prop, nc_id)
   432)         
   433)    case ('modal')
   434)       call modal_optics_init(phys_prop, nc_id)
   435)         
   436)    ! other types of optics can be added here
   437) 
   438)    case default

  1150) function exp_interpol(x, f, y) result(g)
  1151) ! Purpose:
  1152) !   interpolates f(x) to point y
  1153) !   assuming f(x) = f(x0) exp a(x - x0)
  1154) !   where a = ( ln f(x1) - ln f(x0) ) / (x1 - x0)
  1155) !   x0 <= x <= x1
  1156) !   assumes x is monotonically increasing
  1157) ! Author: D. Fillmore
  1158) 
  1159)    implicit none
  1160) 
  1161)    real(r8), intent(in), dimension(:) :: x  ! grid points
  1162)    real(r8), intent(in), dimension(:) :: f  ! grid function values
  1163)    real(r8), intent(in) :: y                ! interpolation point
  1164)    real(r8) :: g                            ! interpolated function value
  1165) 
  1166)    integer :: k  ! interpolation point index
  1167)    integer :: n  ! length of x
  1168)    real(r8) :: a
  1169) 
  1170)    n = size(x)
  1171) 
  1172)    ! find k such that x(k) < y =< x(k+1)
  1173)    ! set k = 1 if y <= x(1)  and  k = n-1 if y > x(n)
  1174) 
  1175)    if (y <= x(1)) then
  1176)      k = 1
  1177)    else if (y >= x(n)) then
  1178)      k = n - 1
  1179)    else
  1180)      k = 1
  1181)      do while (y > x(k+1) .and. k < n)
  1182)        k = k + 1
  1183)      end do
  1184)    end if
  1185) 
  1186)    ! interpolate
  1187)    a = (  log( f(k+1) / f(k) )  ) / ( x(k+1) - x(k) )
  1188)    g = f(k) * exp( a * (y - x(k)) )
  1189)    return
