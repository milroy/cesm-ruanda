    53)     endif
    54)     !I have to reorder it, note that the corner is already included (southwest in 
    55)     ! south and west), note, no element can have a west and a east or a north and a
    56)     ! south, we do not allow just one element per face
    57)     if(desc%reverse(west) .or. desc%reverse(east)) then
    58)       do i=fvm%jx_min1,fvm%jx_max1-1
    59)         tmpval=fvm%spherecentroid(:,i,:)
    60)         tmpvalarea=fvm%area_sphere(i,:)
    61)         do j=fvm%jy_min1,fvm%jy_max1-1
    62)           tmpj=fvm%jy_max1-1+fvm%jy_min1-j
    63)           fvm%spherecentroid(:,i,tmpj)=tmpval(:,j)
    64)           fvm%area_sphere(i,tmpj)=tmpvalarea(j)
    65)         enddo
    66)       enddo
    67)     endif
    68)     if(desc%reverse(north) .or. desc%reverse(south)) then
    69)       do j=fvm%jy_min1,fvm%jy_max1-1
    70)         tmpval=fvm%spherecentroid(:,:,j)
    71)         tmpvalarea=fvm%area_sphere(:,j)
    72)         do i=fvm%jx_min1,fvm%jx_max1-1
    73)           tmpi=fvm%jx_max1-1+fvm%jx_min1-i
    74)           fvm%spherecentroid(:,tmpi,j)=tmpval(:,i)
    75)           fvm%area_sphere(tmpi,j)=tmpvalarea(i)
    76)         enddo
    77)       enddo
    78)     endif
    79)

    81)       if(fvm%swap2) then
    82)         call moment_onsphereswap(fvm%area_sphere,fvm%spherecentroid, &
    83)                            fvm%acartx2,fvm%acarty2,fvm%jx_min2,fvm%jx_max2, &
    84)                            fvm%jy_min2,fvm%jy_max2)
    85)       else
    86)         call moment_onsphere(fvm%area_sphere,fvm%spherecentroid,fvm%acartx2,&
    87)                fvm%acarty2,fvm%jx_min2,fvm%jx_max2,fvm%jy_min2,fvm%jy_max2)
    88)       endif
    89) 
    90)       if(desc%reverse(west) .or. desc%reverse(east)) then
    91)         do i=fvm%jx_min2,fvm%jx_max2-1
    92)           tmpval=fvm%spherecentroid(:,i,:)
    93)           tmpvalarea=fvm%area_sphere(i,:)
    94)           do j=fvm%jy_min2,fvm%jy_max2-1
    95)             tmpj=fvm%jy_max2-1+fvm%jy_min2-j
    96)             fvm%spherecentroid(:,i,tmpj)=tmpval(:,j)
    97)             fvm%area_sphere(i,tmpj)=tmpvalarea(j)

    99)         enddo
   100)       endif
   101) 
   102)       if(desc%reverse(north) .or. desc%reverse(south)) then
   103)         do j=fvm%jy_min2,fvm%jy_max2-1
   104)           tmpval=fvm%spherecentroid(:,:,j)
   105)           tmpvalarea=fvm%area_sphere(:,j)
   106)           do i=fvm%jx_min2,fvm%jx_max2-1
   107)             tmpi=fvm%jx_max2-1+fvm%jx_min2-i
   108)             fvm%spherecentroid(:,tmpi,j)=tmpval(:,i)
   109)             fvm%area_sphere(tmpi,j)=tmpvalarea(i)
   110)           enddo
   111)         enddo
   112)       endif
   113)     endif
   114)   endif

   119) !SUBROUTINE MOMENT_ONSPHERE-----------------------------------------------CE-for FVM!
   120) ! AUTHOR: CHRISTOPH ERATH, 20.July 2011                                             !
   121) ! DESCRIPTION: Compute area and centroids/moments via line integrals                !
   122) !                                                                                   !
   123) ! INPUT:  x  ...  x cartesian coordinats of the arrival grid on the cube            !
   124) !         y  ...  y cartesian coordinats of the arrival grid on the cube            !
   125) !         jx_min, jx_max, jy_min, jy_max                                            !
   126) !            ... cell boundaries in x and y directions                              !
   127) ! INPUT/OUTPUT:                                                                     !
   128) !         area      ... area of cells on the sphere                                 ! 
   129) !         centroid  ... x,y,x^2,y^2,xy                                              !
   130) !-----------------------------------------------------------------------------------!
   131) subroutine moment_onsphere(area,centroid,x,y,jx_min,jx_max,jy_min,jy_max)
   132)   use coordinate_systems_mod, only : surfareaxy
   133)   implicit none
   134)   real (kind=real_kind), dimension(1-nhe:nc+nhe,1-nhe:nc+nhe), intent(inout) :: area
   135)   real (kind=real_kind), dimension(5,1-nhe:nc+nhe,1-nhe:nc+nhe), intent(inout) :: centroid
   136)   real (kind=real_kind), dimension(-nhe:nc+2+nhe), intent(in)   :: x, y
   137)   integer, intent(in)                                  :: jx_min,jx_max,jy_min,jy_max
   138)   integer                                              :: i,j
   139) 
   140)   do j=jy_min,jy_max-1
   141)     do i=jx_min,jx_max-1
   142) !       area(i,j) = surfareaxy(x(i),x(i+1),y(j),y(j+1)) 
   143)       area(i,j) = (I_00(x(i+1),y(j+1)) - I_00(x(i),y(j+1)) + &
   144)                          I_00(x(i),y(j)) - I_00(x(i+1),y(j)))
   145)       ! Compute centroids via line integrals
   146)       centroid(1,i,j) = (I_10(x(i+1),y(j+1)) - I_10(x(i),y(j+1)) + &
   147)                          I_10(x(i),y(j)) - I_10(x(i+1),y(j))) / area(i,j)
   148)       centroid(2,i,j) = (I_01(x(i+1),y(j+1)) - I_01(x(i),y(j+1)) + &
   149)                          I_01(x(i),y(j)) - I_01(x(i+1),y(j))) / area(i,j)
   150)       !ADD PHL START
   151)      ! TAN(alpha)^2 component
   152)       centroid(3,i,j) = (I_20(x(i+1),y(j+1)) - I_20(x(i),y(j+1)) + &
   153)                          I_20(x(i),y(j)) - I_20(x(i+1),y(j))) / area(i,j)
   154)      ! TAN(beta)^2 component
   155)       centroid(4,i,j) = (I_02(x(i+1),y(j+1)) - I_02(x(i),y(j+1)) + &
   156)                          I_02(x(i),y(j)) - I_02(x(i+1),y(j))) / area(i,j)
   157)      ! TAN(alpha) TAN(beta) component
   158)       centroid(5,i,j) = (I_11(x(i+1),y(j+1)) - I_11(x(i),y(j+1)) + &
   159)                          I_11(x(i),y(j)) - I_11(x(i+1),y(j))) / area(i,j)
   160)      !ADD PHL END
   161)     end do
   162)   end do  
   163) end subroutine moment_onsphere

   224) function I_00(x,y)

   226)   real (kind=real_kind)                 :: I_00
   227)   real (kind=real_kind), intent(in)     :: x,y

   229)   I_00 = ATAN(x*y/SQRT(1.0D0+x*x+y*y))

   232) function I_10(x,y)
   233)   implicit none
   234)   real (kind=real_kind)                 :: I_10
   235)   real (kind=real_kind), intent(in)     :: x,y
   236)   real (kind=real_kind)                 :: tmp

   238) !   tmp = ATAN(x)
   239) !   I_10 = -ASINH(y*COS(tmp))
   240)   tmp = y*COS(ATAN(x))
   241)   I_10 = -log(tmp+sqrt(tmp*tmp+1))

   245) function I_01(x,y)
   246)   implicit none
   247)   real (kind=real_kind)                 :: I_01

   249)   real (kind=real_kind)                 :: tmp

   251) !   I_01 = -ASINH(x/SQRT(1+y*y))

   253)   I_01 = -log(tmp+sqrt(tmp*tmp+1))

   256) function I_20(x,y)

   262)   tmp = 1.0D0+y*y
   263)   tmp1=x/SQRT(tmp)
   264)   I_20 = y*log(tmp1+sqrt(tmp1*tmp1+1))+ACOS(x*y/(SQRT((1.0D0+x*x)*tmp)))

   276)   tmp1=y/SQRT(tmp) 
   277)   
   278)   I_02 = x*log(tmp1+sqrt(tmp1*tmp1+1))+ACOS(x*y/SQRT(tmp*(1+y*y)))
