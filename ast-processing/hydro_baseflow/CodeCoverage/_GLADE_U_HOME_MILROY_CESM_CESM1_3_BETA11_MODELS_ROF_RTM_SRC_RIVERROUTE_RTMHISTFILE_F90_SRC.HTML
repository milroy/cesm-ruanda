   228)   subroutine RtmHistHtapesBuild ()
   229) 
   230)     ! !DESCRIPTION:
   231)     ! Initialize ntapes history file for initial or branch run.  
   232) 
   233)     ! !ARGUMENTS:
   234)     implicit none
   235) 
   236)     ! !LOCAL VARIABLES:
   237)     integer :: i                   ! index
   238)     integer :: ier                 ! error code
   239)     integer :: t, f                ! tape, field indices
   240)     integer :: day, sec            ! day and seconds from base date
   241)     character(len=1) :: avgflag    ! lcl equiv of rtmhist_avgflag_pertape(t)
   242)     character(len=*),parameter :: subname = 'hist_htapes_build'
   243)     !----------------------------------------------------------
   244) 
   245)     if (masterproc) then
   246)        write(iulog,*)  trim(subname),' Initializing RTM history files'
   247)        write(iulog,'(72a1)') ("-",i=1,60)
   248)        call shr_sys_flush(iulog)
   249)     endif
   250) 
   251)     ! Override averaging flag for all fields on a particular tape
   252)     ! if namelist input so specifies
   253) 
   254)     do t=1,max_tapes
   255)        if (rtmhist_avgflag_pertape(t) /= ' ') then
   256)           avgflag = rtmhist_avgflag_pertape(t)
   257)           do f = 1,nfmaster
   258)              select case (avgflag)
   259)              case ('A')
   260)                 masterlist(f)%avgflag(t) = avgflag
   261)              case ('I')
   262)                 masterlist(f)%avgflag(t) = avgflag
   263)              case ('X')
   264)                 masterlist(f)%avgflag(t) = avgflag
   265)              case ('M')
   266)                 masterlist(f)%avgflag(t) = avgflag
   267)              case default
   268)                 write(iulog,*) trim(subname),' ERROR: unknown avgflag=',avgflag
   269)                 call shr_sys_abort ()
   270)              end select
   271)           end do
   272)        end if
   273)     end do
   274) 
   275)     fincl(:,1) = rtmhist_fincl1(:)
   276)     fincl(:,2) = rtmhist_fincl2(:)
   277)     fincl(:,3) = rtmhist_fincl3(:)
   278) 
   279)     fexcl(:,1) = rtmhist_fexcl1(:)
   280)     fexcl(:,2) = rtmhist_fexcl2(:)
   281)     fexcl(:,3) = rtmhist_fexcl3(:)
   282) 
   283)     ! Define field list information for all history files.
   284)     ! Update ntapes to reflect number of active history files
   285)     ! Note - branch runs can have additional auxiliary history files declared
   286) 
   287)     call htapes_fieldlist()
   288) 
   289)     ! Set number of time samples in each history file and
   290)     ! Note - the following entries will be overwritten by history restart
   291)     ! Note - with netcdf, only 1 (ncd_double) and 2 (ncd_float) are allowed
   292) 
   293)     do t=1,ntapes
   294)        tape(t)%ntimes = 0
   295)        tape(t)%nhtfrq = rtmhist_nhtfrq(t)
   296)        tape(t)%mfilt = rtmhist_mfilt(t)
   297)        if (rtmhist_ndens(t) == 1) then
   298)           tape(t)%ncprec = ncd_double
   299)        else
   300)           tape(t)%ncprec = ncd_float
   301)        endif
   302)     end do
   303) 
   304)     ! Set time of beginning of current averaging interval
   305)     ! First etermine elapased time since reference date
   306)     call get_prev_time(day, sec)
   307)     do t=1,ntapes
   308)        tape(t)%begtime = day + sec/secspday
   309)     end do
   310) 
   311)     if (masterproc) then
   312)        write(iulog,*)  trim(subname),' Successfully initialized RTM history files'
   313)        write(iulog,'(72a1)') ("-",i=1,60)
   314)        call shr_sys_flush(iulog)
   315)     endif
   316) 
   317)   end subroutine RtmHistHtapesBuild

   321)   subroutine htapes_fieldlist()
   322) 
   323)     ! !DESCRIPTION:
   324)     ! Define the contents of each history file based on namelist
   325)     ! input for initial or branch run, and restart data if a restart run.
   326)     ! Use arrays fincl and fexcl to modify default history tape contents.
   327)     ! Then sort the result alphanumerically.
   328) 
   329)     ! !ARGUMENTS:
   330)     implicit none
   331)     !
   332)     ! !LOCAL VARIABLES:
   333)     integer :: t, f                ! tape, field indices
   334)     integer :: ff                  ! index into include, exclude and fprec list
   335)     character(len=max_namlen) :: name ! field name portion of fincl (i.e. no avgflag separator)
   336)     character(len=max_namlen) :: mastername ! name from masterlist field
   337)     character(len=1)  :: avgflag    ! averaging flag
   338)     character(len=1)  :: prec_acc   ! history buffer precision flag
   339)     character(len=1)  :: prec_wrt   ! history buffer write precision flag
   340)     type (history_entry) :: tmp     ! temporary used for swapping
   341)     character(len=*),parameter :: subname = 'htapes_fieldlist'
   342)     !---------------------------------------------------------
   343) 
   344)     ! First ensure contents of fincl and fexcl are valid names
   345)     do t = 1,max_tapes
   346)        f = 1
   347)        do while (f < max_flds .and. fincl(f,t) /= ' ')
   348)           name = getname (fincl(f,t)) !namelist
   349)           do ff = 1,nfmaster
   350)              mastername = masterlist(ff)%field%name
   351)              if (name == mastername) exit
   352)           end do
   353)           if (name /= mastername) then
   354)              write(iulog,*) trim(subname),' ERROR: ', trim(name), ' in fincl(', f, ') ',&
   355)                   'for history tape ',t,' not found'
   356)              call shr_sys_abort()
   357)           end if
   358)           f = f + 1
   359)        end do
   360) 
   361)        f = 1
   362)        do while (f < max_flds .and. fexcl(f,t) /= ' ')
   363)           do ff = 1,nfmaster
   364)              mastername = masterlist(ff)%field%name
   365)              if (fexcl(f,t) == mastername) exit
   366)           end do
   367)           if (fexcl(f,t) /= mastername) then
   368)              write(iulog,*) trim(subname),' ERROR: ', fexcl(f,t), ' in fexcl(', f, ') ', &
   369)                   'for history tape ',t,' not found'
   370)              call shr_sys_abort()
   371)           end if
   372)           f = f + 1
   373)        end do
   374)     end do
   375) 
   376)     tape(:)%nflds = 0
   377)     do t = 1,max_tapes
   378) 
   379)        ! Loop through the masterlist set of field names and determine if any of those
   380)        ! are in the FINCL or FEXCL arrays
   381)        ! The call to list_index determines the index in the FINCL or FEXCL arrays
   382)        ! that the masterlist field corresponds to
   383)        ! Add the field to the tape if specified via namelist (FINCL[1-max_tapes]),
   384)        ! or if it is on by default and was not excluded via namelist (FEXCL[1-max_tapes]).
   385) 
   386)        do f = 1,nfmaster
   387)           mastername = masterlist(f)%field%name
   388)           call list_index (fincl(1,t), mastername, ff)
   389)           if (ff > 0) then
   390)              ! if field is in include list, ff > 0 and htape_addfld
   391)              ! will not be called for field
   392)              avgflag = getflag (fincl(ff,t))
   393)              call htape_addfld (t, f, avgflag)
   394)           else 
   395)              ! find index of field in exclude list
   396)              call list_index (fexcl(1,t), mastername, ff)
   397) 
   398)              ! if field is in exclude list, ff > 0 and htape_addfld
   399)              ! will not be called for field
   400)              ! if field is not in exclude list, ff =0 and htape_addfld
   401)              ! will be called for field (note that htape_addfld will be
   402)              ! called below only if field is not in exclude list OR in
   403)              ! include list
   404)              if (ff == 0 .and. masterlist(f)%actflag(t)) then
   405)                 call htape_addfld (t, f, ' ')
   406)              end if
   407)           end if
   408)        end do
   409) 
   410)        ! Specification of tape contents now complete.
   411)        ! Sort each list of active entries
   412) 
   413)        do f = tape(t)%nflds-1,1,-1
   414)           do ff = 1,f
   415)              if (tape(t)%hlist(ff)%field%name > tape(t)%hlist(ff+1)%field%name) then
   416)                 tmp = tape(t)%hlist(ff)
   417)                 tape(t)%hlist(ff  ) = tape(t)%hlist(ff+1)
   418)                 tape(t)%hlist(ff+1) = tmp
   419)              else if (tape(t)%hlist(ff)%field%name == tape(t)%hlist(ff+1)%field%name) then
   420)                 write(iulog,*) trim(subname),' ERROR: Duplicate field ', &
   421)                    tape(t)%hlist(ff)%field%name, &
   422)                    't,ff,name=',t,ff,tape(t)%hlist(ff+1)%field%name
   423)                 call shr_sys_abort()
   424)              end if
   425)           end do
   426)        end do
   427) 
   428)        if (masterproc) then
   429)           if (tape(t)%nflds > 0) then
   430)              write(iulog,*) trim(subname),' : Included fields tape ',t,'=',tape(t)%nflds
   431)           end if
   432)           do f = 1,tape(t)%nflds
   433)              write(iulog,*) f,' ',tape(t)%hlist(f)%field%name,' ',tape(t)%hlist(f)%avgflag
   434)           end do
   435)           call shr_sys_flush(iulog)
   436)        end if
   437)     end do
   438) 
   439)     ! Determine total number of active history tapes
   440) 
   441)     ntapes = 0
   442)     do t = max_tapes,1,-1
   443)        if (tape(t)%nflds > 0) then
   444)           ntapes = t
   445)           exit
   446)        end if
   447)     end do
   448) 
   449)     ! Ensure there are no "holes" in tape specification, i.e. empty tapes.
   450)     ! Enabling holes should not be difficult if necessary.
   451) 
   452)     do t = 1,ntapes
   453)        if (tape(t)%nflds  ==  0) then
   454)           write(iulog,*) trim(subname),' ERROR: Tape ',t,' is empty'
   455)           call shr_sys_abort()
   456)        end if
   457)     end do
   458) 
   459)     ! Check that the number of history files declared does not exceed
   460)     ! the maximum allowed.
   461) 
   462)     if (ntapes > max_tapes) then
   463)        write(iulog,*) trim(subname),' ERROR: Too many history files declared, max_tapes=',max_tapes
   464)        call shr_sys_abort()
   465)     end if
   466) 
   467)     if (masterproc) then
   468)        write(iulog,*) 'There will be a total of ',ntapes,'RTM  history tapes'
   469)        do t=1,ntapes
   470)           write(iulog,*)
   471)           if (rtmhist_nhtfrq(t) == 0) then
   472)              write(iulog,*)'RTM History tape ',t,' write frequency is MONTHLY'
   473)           else
   474)              write(iulog,*)'RTM History tape ',t,' write frequency = ',rtmhist_nhtfrq(t)
   475)           endif
   476)           write(iulog,*)'Number of time samples on RTM history tape ',t,' is ',rtmhist_mfilt(t)
   477)           write(iulog,*)'Output precision on RTM history tape ',t,'=',rtmhist_ndens(t)
   478)           write(iulog,*)
   479)        end do
   480)        call shr_sys_flush(iulog)
   481)     end if
   482) 
   483)     ! Set flag indicating h-tape contents are now defined 
   484) 
   485)     htapes_defined = .true.
   486) 
   487)   end subroutine htapes_fieldlist

   491)   subroutine htape_addfld (t, f, avgflag)
   492) 
   493)     ! !DESCRIPTION:
   494)     ! Add a field to the active list for a history tape. Copy the data from
   495)     ! the master field list to the active list for the tape.
   496) 
   497)     ! !ARGUMENTS:
   498)     implicit none
   499)     integer, intent(in) :: t                 ! history tape index
   500)     integer, intent(in) :: f                 ! field index from master field list
   501)     character(len=1), intent(in) :: avgflag  ! time averaging flag
   502) 
   503)     ! !LOCAL VARIABLES:
   504)     integer :: n               ! field index on defined tape
   505)     integer :: begrof          ! per-proc beginning land runoff index
   506)     integer :: endrof          ! per-proc ending land runoff index
   507)     integer :: numrtm          ! total number of rtm cells on all procs
   508)     character(len=*),parameter :: subname = 'htape_addfld'
   509)     !-------------------------------------------------------
   510) 
   511)     ! Ensure that it is not to late to add a field to the history tape
   512)     if (htapes_defined) then
   513)        write(iulog,*) trim(subname),' ERROR: attempt to add field ', &
   514)             masterlist(f)%field%name, ' after history files are set'
   515)        call shr_sys_abort()
   516)     end if
   517) 
   518)     ! Determine bounds
   519)     begrof = runoff%begr
   520)     endrof = runoff%endr
   521)     numrtm = runoff%numr
   522) 
   523)     tape(t)%nflds = tape(t)%nflds + 1
   524)     n = tape(t)%nflds

   550)   subroutine RtmHistUpdateHbuf()
   551) 
   552)     ! !DESCRIPTION:
   553)     ! Accumulate (or take min, max, etc. as appropriate) input field
   554)     ! into its history buffer for appropriate tapes.
   555) 
   556)     ! !ARGUMENTS:
   557)     implicit none
   558) 
   559)     ! !LOCAL VARIABLES:
   560)     integer :: t                   ! tape index
   561)     integer :: f                   ! field index
   562)     integer :: k                   ! index
   563)     integer :: hpindex             ! history pointer index
   564)     integer :: begrof,endrof       ! beginning and ending indices
   565)     character(len=1)  :: avgflag   ! time averaging flag
   566)     real(r8), pointer :: hbuf(:)   ! history buffer
   567)     integer , pointer :: nacs(:)   ! accumulation counter
   568)     real(r8), pointer :: field(:)  ! 1d pointer field
   569)     integer j
   570)     character(len=*),parameter :: subname = 'RtmHistUpdateHbuf'
   571)     !----------------------------------------------------------
   572) 
   573)     begrof = runoff%begr
   574)     endrof = runoff%endr
   575) 
   576)     do t = 1,ntapes
   577)        do f = 1,tape(t)%nflds
   578)           avgflag  =  tape(t)%hlist(f)%avgflag
   579)           nacs     => tape(t)%hlist(f)%nacs
   580)           hbuf     => tape(t)%hlist(f)%hbuf
   581)           hpindex  =  tape(t)%hlist(f)%field%hpindex
   582)           field    => rtmptr(hpindex)%ptr
   583) 
   584)           select case (avgflag)
   585)           case ('I') ! Instantaneous
   586)              do k = begrof,endrof
   587)                 if (field(k) /= spval) then
   588)                    hbuf(k) = field(k)
   589)                 else
   590)                    hbuf(k) = spval
   591)                 end if
   592)                 nacs(k) = 1
   593)              end do
   594)           case ('A') ! Time average
   595)              do k = begrof,endrof
   596)                 if (field(k) /= spval) then
   597)                    if (nacs(k) == 0) hbuf(k) = 0._r8
   598)                    hbuf(k) = hbuf(k) + field(k)
   599)                    nacs(k) = nacs(k) + 1
   600)                 else
   601)                    if (nacs(k) == 0) hbuf(k) = spval
   602)                 end if
   603)              end do
   604)           case ('X') ! Maximum over time
   605)              do k = begrof,endrof
   606)                 if (field(k) /= spval) then
   607)                    if (nacs(k) == 0) hbuf(k) = -1.e50_r8
   608)                    hbuf(k) = max( hbuf(k), field(k) )
   609)                 else
   610)                    if (nacs(k) == 0) hbuf(k) = spval
   611)                 end if
   612)                 nacs(k) = 1
   613)              end do
   614)           case ('M') ! Minimum over time
   615)              do k = begrof,endrof
   616)                 if (field(k) /= spval) then
   617)                    if (nacs(k) == 0) hbuf(k) = +1.e50_r8
   618)                    hbuf(k) = min( hbuf(k), field(k) )
   619)                 else
   620)                    if (nacs(k) == 0) hbuf(k) = spval
   621)                 end if
   622)                 nacs(k) = 1
   623)              end do
   624)           case default
   625)              write(iulog,*) trim(subname),' ERROR: invalid time averaging flag ', avgflag
   626)              call shr_sys_abort()
   627)           end select
   628)        end do
   629)     end do
   630) 
   631)   end subroutine RtmHistUpdateHbuf

   891)   subroutine RtmHistHtapesWrapup( rstwr, nlend )
   892) 
   893)     ! DESCRIPTION:
   894)     ! Write history tape(s)
   895)     ! Determine if next time step is beginning of history interval and if so:
   896)     !   increment the current time sample counter, open a new history file
   897)     !   and if needed (i.e., when ntim = 1), write history data to current
   898)     !   history file, reset field accumulation counters to zero.
   899)     ! If primary history file is full or at the last time step of the simulation,
   900)     !   write restart dataset and close all history fiels.
   901)     ! If history file is full or at the last time step of the simulation:
   902)     !   close history file
   903)     !   and reset time sample counter to zero if file is full.
   904)     ! Daily-averaged data for the first day in September are written on
   905)     !   date = 00/09/02 with mscur = 0.
   906)     ! Daily-averaged data for the first day in month mm are written on
   907)     !   date = yyyy/mm/02 with mscur = 0.
   908)     ! Daily-averaged data for the 30th day (last day in September) are written
   909)     !   on date = 0000/10/01 mscur = 0.
   910)     ! Daily-averaged data for the last day in month mm are written on
   911)     !   date = yyyy/mm+1/01 with mscur = 0.
   912) 
   913) 
   914)     ! !ARGUMENTS:
   915)     implicit none
   916)     logical, intent(in) :: rstwr   ! true => write restart file this step
   917)     logical, intent(in) :: nlend   ! true => end of run on this step
   918) 
   919)     ! !LOCAL VARIABLES:
   920)     integer :: begrof, endrof      ! beg and end rof indices
   921)     integer :: t,f,k,nt            ! indices
   922)     integer :: nstep               ! current step
   923)     integer :: day                 ! current day (1 -> 31)
   924)     integer :: mon                 ! current month (1 -> 12)
   925)     integer :: yr                  ! current year (0 -> ...)
   926)     integer :: mdcur               ! current day
   927)     integer :: mscur               ! seconds of current day
   928)     integer :: mcsec               ! current time of day [seconds]
   929)     integer :: daym1               ! nstep-1 day (1 -> 31)
   930)     integer :: monm1               ! nstep-1 month (1 -> 12)
   931)     integer :: yrm1                ! nstep-1 year (0 -> ...)
   932)     integer :: mcsecm1             ! nstep-1 time of day [seconds]
   933)     real(r8):: time                ! current time
   934)     character(len=256):: str       ! global attribute string
   935)     character(len=1)  :: avgflag   ! averaging flag
   936)     real(r8), pointer :: histo(:)  ! temporary
   937)     real(r8), pointer :: hbuf(:)   ! history buffer
   938)     integer , pointer :: nacs(:)   ! accumulation counter
   939)     character(len=32) :: avgstr    ! time averaging type
   940)     character(len=max_chars) :: long_name ! long name
   941)     character(len=max_chars) :: units     ! units
   942)     character(len=max_namlen):: varname   ! variable name
   943)     character(len=*),parameter :: subname = 'hist_htapes_wrapup'
   944)     !-----------------------------------------------------------
   945) 
   946)     begrof = runoff%begr
   947)     endrof = runoff%endr
   948) 
   949)     ! get current step
   950)     nstep = get_nstep()
   951) 
   952)     ! Set calendar for current time step
   953)     call get_curr_date (yr, mon, day, mcsec)
   954)     call get_curr_time (mdcur, mscur)
   955)     time = mdcur + mscur/secspday
   956) 
   957)     ! Set calendar for current for previous time step
   958)     call get_prev_date (yrm1, monm1, daym1, mcsecm1)
   959) 
   960)     ! Loop over active history tapes, create new history files if necessary
   961)     ! and write data to history files if end of history interval.
   962)     do t = 1, ntapes
   963) 
   964)        ! Skip nstep=0 if monthly average
   965)        if (nstep==0 .and. tape(t)%nhtfrq==0) cycle
   966) 
   967)        ! Determine if end of history interval
   968)        tape(t)%is_endhist = .false.
   969)        if (tape(t)%nhtfrq==0) then   !monthly average
   970)           if (mon /= monm1) then
   971)              tape(t)%is_endhist = .true.
   972)           end if
   973)        else
   974)           if (mod(nstep,tape(t)%nhtfrq) == 0) then
   975)              tape(t)%is_endhist = .true.
   976)           end if
   977)        end if
   978) 
   979)        ! If end of history interval
   980)        if (tape(t)%is_endhist) then
   981)
