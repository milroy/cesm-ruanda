     7) ! !MODULE: shr_file_mod.F90 --- Module to handle various file utilily functions.
     8) !
     9) ! !DESCRIPTION:
    10) !
    11) ! Miscilaneous methods to handle file and directory utilities as well as FORTRAN
    12) ! unit control. Also put/get local files into/from archival location
    13) !
    14) ! File utilites used with CCSM Message passing:
    15) !
    16) ! shr_file_stdio is the main example here, it changes the working directory, 
    17) !                changes stdin and stdout to a given filename.
    18) !
    19) ! This is needed because some implementations of MPI with MPMD so that
    20) ! each executable can run in a different working directory and redirect
    21) ! output to different files.
    22) !
    23) ! File name archival convention, eg.
    24) !    call shr_file_put(rcode,"foo","mss:/USER/foo",rtpd=3650)
    25) ! is extensible -- the existence of the option file name prefix, eg. "mss:", 
    26) ! and optional arguments, eg. rtpd-3650 can be used to access site-specific 
    27) ! storage devices.  Based on CCM (atmosphere) getfile & putfile routines, but
    28) ! intended to be a more extensible, shared code.
    29) ! 
    30) ! !REVISION HISTORY:
    31) !   2006-05-08 E. Kluzek, Add in shr_file_mod and getUnit, freeUnif methods.
    32) !   2000-??-?? B. Kauffman, original version circa 2000
    33) !

    96) ! !IROUTINE: shr_file_put -- Put a file to an archival location.
    97) !
    98) ! !DESCRIPTION:
    99) !    a generic, extensible put-local-file-into-archive routine
   100) ! USAGE:
   101) !    call shr_file_put(rcode,"foo","/home/user/foo")
   102) !    if ( rcode /= 0 ) call shr_sys_abort( "error copying foo" )
   103) !    call shr_file_put(rcode,"foo","cp:/home/user/foo",remove=.true.)
   104) !    if ( rcode /= 0 ) call shr_sys_abort( "error copying foo" )
   105) !    call shr_file_put(rcode,"foo","mss:/USER/foo",rtpd=3650)
   106) !    if ( rcode /= 0 ) call shr_sys_abort( "error archiving foo to MSS" )
   107) !
   108) ! !INTERFACE: ------------------------------------------------------------------  
   109) 
   110) SUBROUTINE shr_file_put(rcode,loc_fn,rem_fn,passwd,rtpd,async,remove)
   111) 
   112)    implicit none
   113) 
   114) ! !INPUT/OUTPUT PARAMETERS:
   115) 
   116)    integer(SHR_KIND_IN),intent(out)         :: rcode  ! return code (non-zero -- error)
   117)    character(*),        intent(in)          :: loc_fn ! local filename
   118)    character(*),        intent(in)          :: rem_fn ! remote filename
   119)    character(*),        intent(in),optional :: passwd ! password
   120)    integer(SHR_KIND_IN),intent(in),optional :: rtpd   ! MSS retention period
   121)    logical,             intent(in),optional :: async  ! true <=> asynchronous put
   122)    logical,             intent(in),optional :: remove ! true <=> rm after put
   123) 
   124) !EOP
   125) 
   126)    !----- local -----  
   127)    integer(SHR_KIND_IN)   :: rtpd2    ! MSS retention period
   128)    logical                :: remove2  ! true <=> rm after put
   129)    logical                :: async2   ! true <=> asynchronous put
   130)    character(SHR_KIND_CL) :: passwd2  ! password
   131)    character(SHR_KIND_CL) :: rfn      ! rem_fn without the destination prefix
   132)    character(SHR_KIND_CL) :: cmd      ! command sent to system call
   133)    integer(SHR_KIND_IN)   :: prefix   ! remote file prefix type
   134) 
   135)    !----- formats -----
   136)    character(*),parameter :: subName = '(shr_file_put) '
   137)    character(*),parameter :: F00 =   "('(shr_file_put) ',4a)"
   138)    character(*),parameter :: F01 =   "('(shr_file_put) ',a,i3,2a)"
   139)    character(*),parameter :: F02 =   "(a,i4)"
   140) 
   141) !-------------------------------------------------------------------------------
   142) ! Notes: 
   143) ! - On some machines the system call will not return a valid error code
   144) ! - when things are sent asynchronously, there probably won't be a error code
   145) !   returned.
   146) !-------------------------------------------------------------------------------
   147) 
   148)    remove2 =.false. ; if ( PRESENT(remove )) remove2 = remove
   149)    async2  =.true.  ; if ( PRESENT(async  )) async2  = async
   150)    passwd2 = " "    ; if ( PRESENT(passwd )) passwd2 = passwd
   151)    rtpd2   = 365    ; if ( PRESENT(rtpd   )) rtpd2   = rtpd
   152)    rcode   = 0
   153)    prefix = shr_file_queryPrefix( rem_fn )
   154) 
   155)    if ( trim(rem_fn) == trim(loc_fn) ) then
   156)       !------------------------------------------------------
   157)       ! (remote file name) == (local file name) => do nothing
   158)       !------------------------------------------------------
   159)       cmd = 'do nothing: remote file = local file = '//trim(loc_fn)
   160)       rcode = 0
   161)    else if ( prefix == shr_file_cpPrefix .or. prefix == shr_file_noPrefix )then
   162)       !------------------------------------------------------
   163)       ! put via unix cp
   164)       !------------------------------------------------------
   165)       rfn = rem_fn
   166)       if ( rem_fn(1:3) == "cp:") rfn = rem_fn(4:len_trim(rem_fn))
   167) #if defined(CATAMOUNT)
   168)       call shr_jlcp(trim(loc_fn),len_trim(loc_fn),trim(rfn),len_trim(rfn),rcode)
   169)       if (remove2) call unlink(trim(loc_fn))
   170)       if (async2 .and. s_loglev > 0) write(s_logunit,F00) 'Error: asynchronous copy not supported.'
   171)       cmd = 'shr_jlcp -f '//trim(loc_fn)//' '//trim(rfn)
   172)       rcode = 0
   173) #else
   174)       cmd = '/bin/cp -f '//trim(loc_fn)//' '//trim(rfn)
   175)       if (remove2) cmd = trim(cmd)//' && /bin/rm -f '//trim(loc_fn)
   176)       if (async2 ) cmd = trim(cmd)//' & '
   177)       call shr_sys_system(trim(cmd),rcode)
   178) #endif
   179)    else if ( prefix == shr_file_mssPrefix )then
   180)       !------------------------------------------------------
   181)       ! put onto NCAR's MSS
   182)       !------------------------------------------------------
   183)       if (rtpd2 > 9999) rtpd2 = 9999
   184)       write(cmd,F02) '/usr/local/bin/msrcp -period ',rtpd2
   185)       if (async2 .and. (.not. remove2) ) cmd = trim(cmd)//' -async '
   186)       if (len_trim(passwd2) > 0 ) cmd = trim(cmd)//' -wpwd '//trim(passwd)
   187)       cmd = trim(cmd)//' '//trim(loc_fn)//' '//trim(rem_fn)
   188)       if (remove2) cmd = trim(cmd)//' && /bin/rm -f '//trim(loc_fn)
   189)       if (async2 .and. remove2 ) cmd = trim(cmd)//' & '
   190)       call shr_sys_system(trim(cmd),rcode)
   191)    else if ( prefix == shr_file_hpssPrefix )then
   192)       !------------------------------------------------------
   193)       ! put onto LANL's hpss
   194)       !------------------------------------------------------
   195)       rcode = -1
   196)       cmd = 'rem_fn='//trim(rem_fn)//' loc_fn='//trim(loc_fn)
   197)       write(s_logunit,F00) 'ERROR: hpss option not yet implemented'
   198)       call shr_sys_abort( subName//'ERROR: hpss option not yet implemented' )
   199)    else if ( prefix == shr_file_nullPrefix )then
   200)       ! do nothing
   201)       cmd = "null prefix => no file archival, do nothing"
   202)       rcode = 0
   203)    end if
   204) 
   205)    if (s_loglev > 0) write(s_logunit,F01) 'rcode =',rcode,' cmd = ', trim(cmd)
   206) 
   207) END SUBROUTINE shr_file_put

   212) ! !IROUTINE: shr_file_get -- Get a file from archival location.
   213) !
   214) ! !DESCRIPTION:
   215) !    a generic, extensible get-local-file-from-archive routine
   216) !
   217) ! USAGE:
   218) !    call shr_file_get(rcode,"foo","/home/user/foo")
   219) !    if ( rcode /= 0 ) call shr_sys_abort( "error getting file foo" )
   220) !    call shr_file_get(rcode,"foo","cp:/home/user/foo",remove=.true.)
   221) !    if ( rcode /= 0 ) call shr_sys_abort( "error getting file foo" )
   222) !    call shr_file_get(rcode,"foo","mss:/USER/foo",clobber=.true.)
   223) !    if ( rcode /= 0 ) call shr_sys_abort( "error getting file foo from MSS" )
   224) !
   225) ! !INTERFACE: ------------------------------------------------------------------  
   226) 
   227) SUBROUTINE shr_file_get(rcode,loc_fn,rem_fn,passwd,async,clobber)
   228) 
   229)    implicit none
   230) 
   231) ! !INPUT/OUTPUT PARAMETERS:
   232) 
   233)    integer(SHR_KIND_IN),intent(out)         :: rcode   ! return code (non-zero means error)
   234)    character(*)        ,intent(in)          :: loc_fn  ! local filename
   235)    character(*)        ,intent(in)          :: rem_fn  ! remote filename
   236)    character(*)        ,intent(in),optional :: passwd  ! password
   237)    logical             ,intent(in),optional :: async   ! true <=> asynchronous get
   238)    logical             ,intent(in),optional :: clobber ! true <=> clobber existing file
   239) 
   240) !EOP
   241) 
   242)    !----- local -----
   243)    logical                :: async2   ! true <=> asynchronous get
   244)    logical                :: clobber2 ! true <=> clobber existing file
   245)    logical                :: exists   ! true <=> local file a ready exists
   246)    character(SHR_KIND_CL) :: passwd2  ! password
   247)    character(SHR_KIND_CL) :: rfn      ! rem_fn without the destination prefix
   248)    character(SHR_KIND_CL) :: cmd      ! command sent to system call
   249)    integer(SHR_KIND_IN)   :: prefix   ! remote file prefix type
   250) 
   251)    !----- formats -----
   252)    character(*),parameter :: subName = '(shr_file_get) '
   253)    character(*),parameter :: F00 =   "('(shr_file_get) ',4a)"
   254)    character(*),parameter :: F01 =   "('(shr_file_get) ',a,i3,2a)"
   255) 
   256) !-------------------------------------------------------------------------------
   257) ! Notes: 
   258) ! - On some machines the system call will not return a valid error code
   259) ! - When things are sent asynchronously, there probably won't be a error code
   260) !   returned.
   261) !-------------------------------------------------------------------------------
   262) 
   263)    passwd2  = " "     ; if (PRESENT(passwd )) passwd2  = passwd
   264)    async2   = .false. ; if (PRESENT(async  )) async2   = async
   265)    clobber2 = .false. ; if (PRESENT(clobber)) clobber2 = clobber
   266)    rcode    = 0
   267) 
   268)    inquire(file=trim(loc_fn),exist=exists)
   269)    prefix = shr_file_queryPrefix( rem_fn )
   270) 
   271)    if ( exists .and. .not. clobber2 ) then
   272)       !------------------------------------------------------
   273)       ! (file exists) and (don't clobber) => do nothing
   274)       !------------------------------------------------------
   275)       cmd = 'do nothing: file exists & no-clobber for '//trim(loc_fn)
   276)       rcode = 0
   277)    else if ( trim(rem_fn) == trim(loc_fn) ) then
   278)       !------------------------------------------------------
   279)       ! (remote file name) == (local file name) => do nothing
   280)       !------------------------------------------------------
   281)       cmd = 'do nothing: remote file = local file for '//trim(loc_fn)
   282)       rcode = 0
   283)    else if ( prefix == shr_file_cpPrefix .or. prefix == shr_file_noPrefix )then
   284)       !------------------------------------------------------
   285)       ! get via unix cp
   286)       !------------------------------------------------------
   287)       rfn = rem_fn  ! remove prefix from this temp file name
   288)       if (rem_fn(1:3) == "cp:") rfn = rem_fn(4:len_trim(rem_fn))
   289) #if defined(CATAMOUNT)
   290)       call shr_jlcp(trim(rfn),len(trim(rfn)),trim(loc_fn),len(trim(loc_fn)),rcode)
   291)       if (async2.and.s_loglev>0) write(s_logunit,F00) 'Error: asynchronous copy not supported.'
   292)       cmd = 'shr_jlcp -f '//trim(rfn)//' '//trim(loc_fn)
   293)       rcode = 0
   294) #else
   295)       cmd = '/bin/cp -f '//trim(rfn)//' '//trim(loc_fn)
   296)       if (async2) cmd = trim(cmd)//' & '
   297)       call shr_sys_system(trim(cmd),rcode)
   298) #endif
   299)    else if ( prefix == shr_file_mssPrefix )then
   300)       !------------------------------------------------------
   301)       ! get from NCAR's MSS
   302)       !------------------------------------------------------
   303)       cmd = '/usr/local/bin/msrcp '
   304)       if (async2)   cmd = trim(cmd)//' -async '
   305)       cmd = trim(cmd)//' '//trim(rem_fn)//' '//trim(loc_fn)
   306)       call shr_sys_system(trim(cmd),rcode)
   307)    else if ( prefix == shr_file_hpssPrefix )then
   308)       !------------------------------------------------------
   309)       ! get from LANL's hpss
   310)       !------------------------------------------------------
   311)       rcode = -1
   312)       cmd = 'rem_fn='//trim(rem_fn)//' loc_fn='//trim(loc_fn)
   313)       write(s_logunit,F00) 'ERROR: hpss option not yet implemented'
   314)       call shr_sys_abort( subName//'ERROR: hpss option not yet implemented' )
   315)    else if ( prefix == shr_file_nullPrefix )then
   316)       ! do nothing
   317)       cmd = "null prefix => no file retrieval, do nothing"
   318)       rcode = 0
   319)    end if
   320) 
   321)    if (s_loglev > 0) write(s_logunit,F01) 'rcode =',rcode,' cmd = ', trim(cmd)
   322) 
   323) END SUBROUTINE shr_file_get

   331) !    
   332) ! !INTERFACE: ------------------------------------------------------------------

   334) integer(SHR_KIND_IN) FUNCTION shr_file_queryPrefix( filepath, prefix )
   335) 
   336)    implicit none
   337) 
   338) ! !INPUT/OUTPUT PARAMETERS:
   339) 
   340)    character(*), intent(in)            :: filepath ! Input filepath
   341)    character(*), intent(out), optional :: prefix   ! Output prefix description
   342) 
   343) !EOP
   344) 
   345)    !----- local -----
   346) 
   347) !-------------------------------------------------------------------------------
   348) ! Notes:

   351)    if (     filepath(1:5) == "null:" )then
   352)       shr_file_queryPrefix = shr_file_nullPrefix
   353)       if ( present(prefix) ) prefix = "null:"

   358)       shr_file_queryPrefix = shr_file_mssPrefix
   359)       if ( present(prefix) ) prefix = "mss:"
   360)    else if( filepath(1:5) == "hpss:" )then
   361)       shr_file_queryPrefix = shr_file_hpssPrefix
   362)       if ( present(prefix) ) prefix = "hpss:"
   363)    else
   364)       shr_file_queryPrefix = shr_file_noPrefix
   365)       if ( present(prefix) ) prefix = ""
   366)    end if

   373) ! !IROUTINE: shr_file_getUnit -- Get a free FORTRAN unit number
   374) !
   375) ! !DESCRIPTION: Get the next free FORTRAN unit number.
   376) !
   377) ! !REVISION HISTORY:
   378) !     2005-Dec-14 - E. Kluzek - creation
   379) !
   380) ! !INTERFACE: ------------------------------------------------------------------  
   381) 
   382) INTEGER FUNCTION shr_file_getUnit ( unit )
   383) 
   384)    implicit none
   385) 
   386) ! !INPUT/OUTPUT PARAMETERS:
   387) 
   388)    integer(SHR_KIND_IN),intent(in),optional :: unit ! desired unit number
   389) 
   390) !EOP
   391) 
   392)    !----- local -----
   393)    integer(SHR_KIND_IN)   :: n      ! loop index
   394)    logical                :: opened ! If unit opened or not
   395) 
   396)    !----- formats -----
   397)    character(*),parameter :: subName = '(shr_file_getUnit) '
   398)    character(*),parameter :: F00   = "('(shr_file_getUnit) ',A,I4,A)"
   399) 
   400) !-------------------------------------------------------------------------------
   401) ! Notes:
   402) !-------------------------------------------------------------------------------
   403) 
   404)    if (present (unit)) then
   405)       inquire( unit, opened=opened )
   406)       if (unit < 0 .or. unit > shr_file_maxUnit) then
   407)          write(s_logunit,F00) 'invalid unit number request:', unit
   408)          call shr_sys_abort( 'ERROR: bad input unit number' )
   409)       else if (opened .or. UnitTag(unit) .or. unit == 0 .or. unit == 5 &
   410)                .or. unit == 6) then
   411)          write(s_logunit,F00) 'unit number ', unit, ' is already in use'
   412)          call shr_sys_abort( 'ERROR: Input unit number already in use' )
   413)       else
   414)          shr_file_getUnit = unit
   415)          UnitTag (unit)   = .true.
   416)          return
   417)       end if
   418) 
   419)    else
   420)       ! --- Choose first available unit other than 0, 5, or 6  ------
   421)       do n=shr_file_maxUnit, shr_file_minUnit, -1
   422)          inquire( n, opened=opened )
   423)          if (n == 5 .or. n == 6 .or. opened) then
   424)             cycle
   425)          end if
   426)          if ( .not. UnitTag(n) ) then
   427)             shr_file_getUnit = n
   428)             UnitTag(n)       = .true.
   429)             return
   430)          end if
   431)       end do
   432)    end if
   433) 
   434)    call shr_sys_abort( subName//': Error: no available units found' )
   435) 
   436) END FUNCTION shr_file_getUnit

   455)

   463)    character(*), parameter :: subName = '(shr_file_freeUnit) '

   487) ! !IROUTINE: shr_file_stdio -- Change working directory, and redirect stdin/stdout
   488) !
   489) ! !DESCRIPTION:
   490) !   1) change the cwd (current working directory) and 
   491) !   2) redirect stdin & stdout (units 5 & 6) to named files,
   492) !   where the desired cwd & files are specified by namelist file.
   493) !
   494) !   Normally this is done to work around limitations in the execution syntax
   495) !   of common MPI implementations.  For example, SGI's mpirun syntax is not 
   496) !   flexible enough to allow MPMD models to select different execution
   497) !   directories or to redirect stdin & stdout on the command line.  
   498) !   Such functionality is highly desireable for CCSM purposes.  
   499) !   ie. mpirun can't handle this:
   500) !   unix> cd /usr/tmp/jdoe/csm/case01/atm ; atm < atm.parm > atm.log &
