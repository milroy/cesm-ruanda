    78) ! !IROUTINE: init_transport - initializations for horizontal transport
    79) !
    80) ! !INTERFACE:
    81) !
    82)       subroutine init_transport
    83) !
    84) ! !DESCRIPTION:
    85) !
    86) ! This subroutine is a wrapper for init_remap, which initializes the
    87) ! remapping transport scheme.  If the model is run with upwind
    88) ! transport, no initializations are necessary.
    89) !
    90) ! !REVISION HISTORY:
    91) !
    92) ! authors William H. Lipscomb, LANL
    93) !
    94) ! !USES:
    95) !
    96)       use ice_state, only: trcr_depend, ntrcr
    97)       use ice_exit
    98)       use ice_timers
    99)       use ice_transport_remap, only: init_remap
   100) !
   101) !EOP
   102) !
   103)       integer (kind=int_kind) ::       &
   104)          k, nt, nt1     ! tracer indices
   105) 
   106)       call ice_timer_start(timer_advect)  ! advection 
   107) 
   108)       ntrace = 2+ntrcr+nilyr+nslyr  ! hice,hsno,qice,qsno,trcr
   109) 
   110)       allocate (tracer_type   (ntrace), &
   111)                 depend        (ntrace), &
   112)                 has_dependents(ntrace))
   113) 
   114)       if (trim(advection)=='remap') then
   115) 
   116) !lipscomb - two branches for now; consolidate later
   117) 
   118)          ! define tracer dependency arrays
   119)          ! see comments in remapping routine
   120) 
   121)           depend(1:2)         = 0 ! hice, hsno
   122)           tracer_type(1:2)    = 1 ! no dependency
   123)       
   124)           k = 2
   125) 
   126)           do nt = 1, ntrcr
   127)              depend(k+nt) = trcr_depend(nt) ! 0 for ice area tracers
   128)                                             ! 1 for ice volume tracers
   129)                                             ! 2 for snow volume tracers
   130)              if (trcr_depend(nt) == 0) then
   131)                 tracer_type(k+nt) = 1
   132)              else               ! trcr_depend = 1 or 2
   133)                 tracer_type(k+nt) = 2
   134)              endif
   135)           enddo
   136) 
   137)           k = k + ntrcr
   138)           
   139)           depend(k+1:k+nilyr) = 1 ! qice depends on hice
   140)           tracer_type(k+1:k+nilyr) = 2 
   141) 
   142)           k = k + nilyr
   143) 
   144)           depend(k+1:k+nslyr) = 2 ! qsno depends on hsno
   145)           tracer_type(k+1:k+nslyr) = 2 
   146) 
   147)           has_dependents = .false.
   148)           do nt = 1, ntrace
   149)              if (depend(nt) > 0) then
   150)                 nt1 = depend(nt)
   151)                 has_dependents(nt1) = .true.
   152)                 if (nt1 > nt) then
   153)                    write(nu_diag,*)     &
   154)                       'Tracer nt2 =',nt,' depends on tracer nt1 =',nt1
   155)                    call abort_ice       &
   156)                       ('ice: remap transport: Must have nt2 > nt1')
   157)                 endif
   158)              endif
   159)           enddo                 ! ntrace
   160) 
   161)           call init_remap    ! grid quantities
   162) 
   163)       else   ! upwind
   164) 
   165)          continue
   166) 
   167)       endif
   168) 
   169)       call ice_timer_stop(timer_advect)  ! advection 
   170) 
   171)       end subroutine init_transport

   176) ! !IROUTINE: transport_remap - wrapper for remapping transport scheme

   180)       subroutine transport_remap (dt)
   181) !
   182) ! !DESCRIPTION:
   183) !
   184) ! This subroutine solves the transport equations for one timestep
   185) ! using the conservative remapping scheme developed by John Dukowicz
   186) ! and John Baumgardner (DB) and modified for sea ice by William
   187) ! Lipscomb and Elizabeth Hunke.
   188) !
   189) ! This scheme preserves monotonicity of ice area and tracers.  That is,
   190) ! it does not produce new extrema.  It is second-order accurate in space,
   191) ! except where gradients are limited to preserve monotonicity. 
   192) !
   193) ! !REVISION HISTORY:
   194) !
   195) ! authors William H. Lipscomb, LANL
   196) !
   197) ! !USES:
   198) !
   199)       use ice_boundary
   200)       use ice_global_reductions
   201)       use ice_domain
   202)       use ice_blocks
   203)       use ice_state
   204)       use ice_grid, only: tarea, HTE, HTN
   205)       use ice_exit
   206)       use ice_calendar, only: istep1, istep, diagfreq
   207)       use ice_timers
   208)       use ice_transport_remap, only: horizontal_remap, make_masks
   209) !
   210) ! !INPUT/OUTPUT PARAMETERS:
   211) !
   212)       real (kind=dbl_kind), intent(in) ::     &
   213)          dt      ! time step
   214) !
   215) !EOP
   216) !
   217)       ! local variables
   218) 
   219)       integer (kind=int_kind) ::     &
   220)          i, j           ,&! horizontal indices
   221)          iblk           ,&! block index
   222)          ilo,ihi,jlo,jhi,&! beginning and end of physical domain
   223)          n              ,&! ice category index
   224)          nt, nt1, nt2     ! tracer indices
   225) 
   226)       real (kind=dbl_kind),      &
   227)          dimension (nx_block,ny_block,0:ncat,max_blocks) ::     &
   228)          aim            ,&! mean ice category areas in each grid cell
   229)          aimask           ! = 1. if ice is present, = 0. otherwise
   230) 
   231)       real (kind=dbl_kind),      &
   232)          dimension (nx_block,ny_block,max_ntrace,ncat,max_blocks) ::     &
   233)          trm            ,&! mean tracer values in each grid cell
   234)          trmask           ! = 1. if tracer is present, = 0. otherwise
   235) 
   236)       logical (kind=log_kind) ::     &
   237)          l_stop           ! if true, abort the model
   238) 
   239)       integer (kind=int_kind) ::     &
   240)          istop, jstop     ! indices of grid cell where model aborts 
   241) 
   242)       integer (kind=int_kind), dimension(0:ncat,max_blocks) ::     &
   243)          icellsnc         ! number of cells with ice
   244) 
   245)       integer (kind=int_kind),      &
   246)          dimension(nx_block*ny_block,0:ncat,max_blocks) ::     &
   247)          indxinc, indxjnc   ! compressed i/j indices
   248) 
   249)       type (block) :: &
   250)          this_block           ! block information for current block
   251)       
   252)     !-------------------------------------------------------------------
   253)     ! If l_fixed_area is true, the area of each departure region is
   254)     !  computed in advance (e.g., by taking the divergence of the 
   255)     !  velocity field and passed to locate_triangles.  The departure 
   256)     !  regions are adjusted to obtain the desired area.
   257)     ! If false, edgearea is computed in locate_triangles and passed out.
   258)     !-------------------------------------------------------------------
   259) 
   260)       real (kind=dbl_kind), dimension(nx_block,ny_block,max_blocks) ::   &
   261)          edgearea_e     ,&! area of departure regions for east edges
   262)          edgearea_n       ! area of departure regions for north edges
   263) 
   264)       ! variables related to optional bug checks
   265) 
   266)       logical (kind=log_kind), parameter ::     &
   267)          l_conservation_check = .true. ,&! if true, check conservation
   268)          l_monotonicity_check = .true.   ! if true, check monotonicity
   269) 
   270)       real (kind=dbl_kind), dimension(0:ncat) ::     &
   271)          asum_init      ,&! initial global ice area
   272)          asum_final       ! final global ice area
   273) 
   274)       real (kind=dbl_kind), dimension(max_ntrace,ncat) ::     &
   275)          atsum_init     ,&! initial global ice area*tracer
   276)          atsum_final      ! final global ice area*tracer
   277) 
   278)       real (kind=dbl_kind), dimension (:,:,:,:,:), allocatable ::     &
   279)          tmin         ,&! local min tracer
   280)          tmax           ! local max tracer
   281) 
   282)       integer (kind=int_kind) :: alloc_error
   283) 
   284)       real (kind=dbl_kind), dimension(nx_block,ny_block,max_blocks) :: &
   285)          work1
   286) 
   287)       call ice_timer_start(timer_advect)  ! advection 
   288) 
   289) !---!-------------------------------------------------------------------
   290) !---! Prepare for remapping.
   291) !---! Initialize, update ghost cells, fill tracer arrays.
   292) !---!-------------------------------------------------------------------
   293) 
   294)       l_stop = .false.
   295)       istop = 0
   296)       jstop = 0
   297) 
   298)     !-------------------------------------------------------------------
   299)     ! Compute open water area in each grid cell.
   300)     ! Note: An aggregate_area call is needed only if the open
   301)     !       water area has changed since the previous call.
   302)     !       Here we assume that aice0 is up to date.
   303)     !-------------------------------------------------------------------
   304) 
   305) !      do iblk = 1, nblocks
   306) !         call aggregate_area (nx_block, ny_block,
   307) !                              iblk,     &
   308) !                              aicen(:,:,:,iblk),     &
   309) !                              aice (:,:,  iblk),     &
   310) !                              aice0(:,:,  iblk)) 
   311) !      enddo
   312) 
   313)     !-------------------------------------------------------------------
   314)     ! Ghost cell updates for state variables.
   315)     ! Commented out because ghost cells are updated after cleanup_itd.
   316)     !-------------------------------------------------------------------
   317) !      call ice_timer_start(timer_bound)
   318) 
   319) !      call ice_HaloUpdate (aice0,            halo_info,     &
   320) !                           field_loc_center, field_type_scalar)
   321) 
   322) !      call bound_state (aicen, trcrn,     &
   323) !                        vicen, vsnon,      &
   324) !                        eicen, esnon)
   325) 
   326) !      call ice_timer_stop(timer_bound)
   327) 
   328)     !-------------------------------------------------------------------
   329)     ! Ghost cell updates for ice velocity.
   330)     ! Commented out because ghost cell velocities are computed
   331)     !  in ice_dyn_evp.
   332)     !-------------------------------------------------------------------
   333) 
   334) !      call ice_timer_start(timer_bound)
   335) !      call ice_HaloUpdate (uvel,               halo_info,     &
   336) !                           field_loc_NEcorner, field_type_vector)
   337) !      call ice_HaloUpdate (vvel,               halo_info,     &
   338) !                           field_loc_NEcorner, field_type_vector)
   339) !      call ice_timer_stop(timer_bound)
   340) 
   341) 
   342)       call t_barrierf('cice_remap_s2t_BARRIER',MPI_COMM_ICE)
   343)       call t_startf  ('cice_remap_s2t')
   344) 
   345)       !$OMP PARALLEL DO PRIVATE(iblk)
   346)       do iblk = 1, nblocks
   347) 
   348)     !-------------------------------------------------------------------
   349)     ! Fill arrays with fields to be remapped.
   350)     !-------------------------------------------------------------------
   351) 
   352)          call state_to_tracers(nx_block,          ny_block,             &
   353)                                ntrcr,             ntrace,               &
   354)                                aice0(:,:,  iblk),                       &
   355)                                aicen(:,:,:,iblk), trcrn(:,:,:,:,iblk),  &
   356)                                vicen(:,:,:,iblk), vsnon(:,:,  :,iblk),  &
   357)                                eicen(:,:,:,iblk), esnon(:,:,  :,iblk),  &
   358)                                aim  (:,:,:,iblk), trm  (:,:,:,:,iblk))
   359) 
   360)       enddo
   361)       !$OMP END PARALLEL DO
   362) 
   363)       call t_stopf   ('cice_remap_s2t')
   364)       call t_barrierf('cice_remap_check1_BARRIER',MPI_COMM_ICE)
   365)       call t_startf  ('cice_remap_check1')
   366) 
   367) !---!-------------------------------------------------------------------
   368) !---! Optional conservation and monotonicity checks.
   369) !---!-------------------------------------------------------------------
   370) 
   371)       if (l_conservation_check .and. mod(istep,diagfreq) == 0) then
   372) 
   373)     !-------------------------------------------------------------------
   374)     ! Compute initial values of globally conserved quantities.
   375)     !-------------------------------------------------------------------
   376) 
   377)          do n = 0, ncat
   378)             asum_init(n) = global_sum(aim(:,:,n,:),     distrb_info,       &
   379)                                       field_loc_center, tarea)
   380)          enddo
   381) 
   382)          do n = 1, ncat
   383)             do nt = 1, ntrace
   384)                if (tracer_type(nt)==1) then ! does not depend on another tracer
   385)                   atsum_init(nt,n) =      &
   386)                       global_sum_prod(trm(:,:,nt,n,:), aim(:,:,n,:),       &
   387)                                       distrb_info,     field_loc_center,   &
   388)                                       tarea)
   389)                elseif (tracer_type(nt)==2) then ! depends on another tracer
   390)                   nt1 = depend(nt)
   391)                   do iblk = 1, nblocks
   392)                      do j= 1,ny_block  
   393)                         do i = 1,nx_block
   394)                            work1(i,j,iblk) = trm(i,j,nt,n,iblk)*trm(i,j,nt1,n,iblk)
   395)                         end do
   396)                      end do
   397)                   end do
   398)                   atsum_init(nt,n) =     &
   399)                       global_sum_prod(work1(:,:,:), aim(:,:,n,:),          &
   400)                                       distrb_info,  field_loc_center,      &
   401)                                       tarea)
   402)                elseif (tracer_type(nt)==3) then ! depends on two tracers
   403)                   nt1 = depend(nt)
   404)                   nt2 = depend(nt1)
   405)                   do iblk = 1, nblocks
   406)                      do j= 1,ny_block  
   407)                         do i = 1,nx_block
   408)                            work1(i,j,iblk) = trm(i,j,nt,n,iblk)*trm(i,j,nt1,n,iblk) &
   409) 	                                                       *trm(i,j,nt2,n,iblk)
   410)                         end do
   411)                      end do
   412)                   end do
   413)                   atsum_init(nt,n) =     &
   414)                       global_sum_prod(work1(:,:,:), aim(:,:,n,:),          &
   415)                                       distrb_info,  field_loc_center,      &
   416)                                       tarea)
   417)                endif            ! tracer_type
   418)             enddo               ! nt
   419)          enddo                  ! n
   420) 
   421)       endif                     ! l_conservation_check
   422)       
   423)       if (l_monotonicity_check .and. mod(istep,diagfreq) == 0) then
   424) 
   425)          allocate(tmin(nx_block,ny_block,ntrace,ncat,max_blocks),     &
   426)                   tmax(nx_block,ny_block,ntrace,ncat,max_blocks),     &
   427)                   STAT=alloc_error)
   428) 
   429)          if (alloc_error /= 0)      &
   430)               call abort_ice ('ice: allocation error')
   431) 
   432)          tmin(:,:,:,:,:) = c0
   433)          tmax(:,:,:,:,:) = c0
   434) 
   435)          !$OMP PARALLEL DO PRIVATE(iblk,this_block,ilo,ihi,jlo,jhi,n)
   436)          do iblk = 1, nblocks
   437)             this_block = get_block(blocks_ice(iblk),iblk)         
   438)             ilo = this_block%ilo
   439)             ihi = this_block%ihi
   440)             jlo = this_block%jlo
   441)             jhi = this_block%jhi
   442) 
   443)     !------------------------------------------------------------------- 
   444)     ! Compute masks.
   445)     ! Masks are used to prevent tracer values in cells without ice
   446)     !  from being used in the monotonicity check.
   447)     !------------------------------------------------------------------- 
   448) 
   449)             call make_masks (nx_block,          ny_block,              &
   450)                              ilo, ihi,          jlo, jhi,              &
   451)                              nghost,            ntrace,                &
   452)                              has_dependents,    icellsnc(:,iblk),      &
   453)                              indxinc(:,:,iblk), indxjnc(:,:,iblk),     &
   454)                              aim(:,:,:,iblk),   aimask(:,:,:,iblk),    &
   455)                              trm(:,:,:,:,iblk), trmask(:,:,:,:,iblk))
   456) 
   457)     !-------------------------------------------------------------------
   458)     ! Compute local max and min of tracer fields.
   459)     !-------------------------------------------------------------------
   460) 
   461)             do n = 1, ncat
   462)                call local_max_min                                      &  
   463)                             (nx_block,           ny_block,             &
   464)                              ilo, ihi,           jlo, jhi,             &
   465)                              trm (:,:,:,n,iblk),                       &
   466)                              tmin(:,:,:,n,iblk), tmax  (:,:,:,n,iblk), &
   467)                              aimask(:,:,n,iblk), trmask(:,:,:,n,iblk))
   468)             enddo
   469)          enddo
   470)          !$OMP END PARALLEL DO
   471) 
   472)          call ice_timer_start(timer_bound)
   473)          call ice_HaloUpdate (tmin,             halo_info,     &
   474)                               field_loc_center, field_type_scalar)
   475)          call ice_HaloUpdate (tmax,             halo_info,     &
   476)                               field_loc_center, field_type_scalar)
   477)          call ice_timer_stop(timer_bound)
   478) 
   479)          !$OMP PARALLEL DO PRIVATE(iblk,this_block,ilo,ihi,jlo,jhi,n)
   480)          do iblk = 1, nblocks
   481)             this_block = get_block(blocks_ice(iblk),iblk)         
   482)             ilo = this_block%ilo
   483)             ihi = this_block%ihi
   484)             jlo = this_block%jlo
   485)             jhi = this_block%jhi
   486) 
   487)             do n = 1, ncat
   488)                call quasilocal_max_min (nx_block, ny_block,     &
   489)                                         ilo, ihi, jlo, jhi,     &
   490)                                         tmin(:,:,:,n,iblk),      &
   491)                                         tmax(:,:,:,n,iblk))
   492)             enddo
   493)          enddo
   494)          !$OMP END PARALLEL DO
   495) 
   496)       endif                     ! l_monotonicity_check
   497) 
   498)       call t_stopf   ('cice_remap_check1')
   499)       call t_barrierf('cice_remap_horz_BARRIER',MPI_COMM_ICE)
   500)       call t_startf  ('cice_remap_horz')
   501) 
   502)     !-------------------------------------------------------------------
   503)     ! Main remapping routine: Step ice area and tracers forward in time.
   504)     !-------------------------------------------------------------------
   505)     ! If l_fixed_area is true, compute edgearea by taking the divergence
   506)     !  of the velocity field.  Otherwise, initialize edgearea.
   507)     !-------------------------------------------------------------------
   508) 
   509)          !$OMP PARALLEL DO PRIVATE(iblk,i,j)
   510)          do iblk = 1, nblocks

   542)                enddo
   543) 
   544)             enddo  ! iblk
   545)             !$OMP END PARALLEL DO
   546) 
   547)          endif
   548) 
   549)          call horizontal_remap (dt,                ntrace,             &
   550)                                 uvel      (:,:,:), vvel      (:,:,:),  &

   557) 
   558)       call t_stopf   ('cice_remap_horz')
   559)       call t_barrierf('cice_remap_t2s_BARRIER',MPI_COMM_ICE)
   560)       call t_startf  ('cice_remap_t2s')
   561) 
   562)     !-------------------------------------------------------------------
   563)     ! Given new fields, recompute state variables.
   564)     !-------------------------------------------------------------------
   565) 
   566)       !$OMP PARALLEL DO PRIVATE(iblk)
   567)       do iblk = 1, nblocks
   568) 
   569)          call tracers_to_state (nx_block,          ny_block,            &
   570)                                 ntrcr,             ntrace,              &
   571)                                 aim  (:,:,:,iblk), trm  (:,:,:,:,iblk), &
   572)                                 aice0(:,:,  iblk),                      &
   573)                                 aicen(:,:,:,iblk), trcrn(:,:,:,:,iblk), &
   574)                                 vicen(:,:,:,iblk), vsnon(:,:,  :,iblk), &
   575)                                 eicen(:,:,:,iblk), esnon(:,:,  :,iblk)) 
   576) 
   577)       enddo                     ! iblk
   578)       !$OMP END PARALLEL DO
   579) 
   580)       call t_stopf   ('cice_remap_t2s')
   581)       call t_barrierf('cice_remap_bound1_BARRIER',MPI_COMM_ICE)
   582)       call t_startf  ('cice_remap_bound1')
   583) 
   584)     !-------------------------------------------------------------------
   585)     ! Ghost cell updates for state variables.
   586)     !-------------------------------------------------------------------
   587) 
   588)       call ice_timer_start(timer_bound)
   589) 
   590)       call bound_state (aicen, trcrn,     &
   591)                         vicen, vsnon,      &
   592)                         eicen, esnon)
   593) 
   594)       call ice_timer_stop(timer_bound)
   595)       call t_stopf   ('cice_remap_bound1')
   596)       call t_barrierf('cice_remap_check2_BARRIER',MPI_COMM_ICE)
   597)       call t_startf  ('cice_remap_check2')
   598) 
   599) !---!-------------------------------------------------------------------
   600) !---! Optional conservation and monotonicity checks
   601) !---!-------------------------------------------------------------------
   602) 
   603)     !-------------------------------------------------------------------
   604)     ! Compute final values of globally conserved quantities.
   605)     ! Check global conservation of area and area*tracers.  (Optional)
   606)     !-------------------------------------------------------------------
   607) 
   608)       if (l_conservation_check .and. mod(istep,diagfreq) == 0) then
   609) 
   610)          do n = 0, ncat
   611)             asum_final(n) = global_sum(aim(:,:,n,:),     distrb_info,      &
   612)                                        field_loc_center, tarea)
   613)          enddo
   614) 
   615)          do n = 1, ncat
   616)             do nt = 1, ntrace
   617)                if (tracer_type(nt)==1) then ! does not depend on another tracer
   618)                   atsum_final(nt,n) =      &
   619)                       global_sum_prod(trm(:,:,nt,n,:), aim(:,:,n,:),       &
   620)                                       distrb_info,     field_loc_center,   &
   621)                                       tarea)
   622)                elseif (tracer_type(nt)==2) then ! depends on another tracer
   623)                   nt1 = depend(nt)
   624)                   do iblk = 1, nblocks
   625)                      do j= 1,ny_block  
   626)                         do i = 1,nx_block
   627)                            work1(i,j,iblk) = trm(i,j,nt,n,iblk)*trm(i,j,nt1,n,iblk)
   628)                         end do
   629)                      end do
   630)                   end do
   631)                   atsum_final(nt,n) =     &
   632)                       global_sum_prod(work1(:,:,:), aim(:,:,n,:),          &
   633)                                       distrb_info,  field_loc_center,      &
   634)                                       tarea)
   635)                elseif (tracer_type(nt)==3) then ! depends on two tracers
   636)                   nt1 = depend(nt)
   637)                   nt2 = depend(nt1)
   638)                   do iblk = 1, nblocks
   639)                      do j= 1,ny_block  
   640)                         do i = 1,nx_block
   641)                            work1(i,j,iblk) = trm(i,j,nt,n,iblk)*trm(i,j,nt1,n,iblk) &
   642) 	                                                       *trm(i,j,nt2,n,iblk)
   643)                         end do
   644)                      end do
   645)                   end do
   646)                   atsum_final(nt,n) =     &
   647)                       global_sum_prod(work1(:,:,:), aim(:,:,n,:),          &
   648)                                       distrb_info,  field_loc_center,      &
   649)                                       tarea)
   650)                endif            ! tracer_type
   651)             enddo               ! nt
   652)          enddo                  ! n
   653) 
   654) 
   655)          if (my_task == master_task) then
   656)             call global_conservation (l_stop,     &
   657)                                       asum_init(0), asum_final(0))
   658) 
   659)             if (l_stop) then
   660)                write (nu_diag,*) 'istep1 =', istep1
   661)                write (nu_diag,*) 'transport: conservation error, cat 0'
   662)                l_stop = .false.
   663)                call abort_ice('ice remap transport: conservation error')
   664)             endif
   665) 
   666)             do n = 1, ncat               
   667)                call global_conservation                                 &
   668)                                      (l_stop,                           &
   669)                                       asum_init(n),    asum_final(n),   &
   670)                                       atsum_init(:,n), atsum_final(:,n))
   671) 
   672)                if (l_stop) then
   673)                   write (nu_diag,*) 'istep1, cat =',     &
   674)                                      istep1, n
   675)                   write (nu_diag,*) 'transport: conservation error, cat ',n
   676)                   l_stop = .false.
   677)                   call abort_ice     &
   678)                        ('ice remap transport: conservation error')
   679)                endif
   680)             enddo               ! n
   681) 
   682)          endif                  ! my_task = master_task
   683) 
   684)       endif                     ! l_conservation_check
   685) 
   686)     !-------------------------------------------------------------------
   687)     ! Check tracer monotonicity.  (Optional)
   688)     !-------------------------------------------------------------------
   689) 
   690)       if (l_monotonicity_check .and. mod(istep,diagfreq) == 0) then
   691)          do iblk = 1, nblocks
   692)             this_block = get_block(blocks_ice(iblk),iblk)         
   693)             ilo = this_block%ilo
   694)             ihi = this_block%ihi
   695)             jlo = this_block%jlo
   696)             jhi = this_block%jhi
   697) 
   698)             do n = 1, ncat
   699)                call check_monotonicity      &
   700)                                (nx_block,           ny_block,     &
   701)                                 ilo, ihi, jlo, jhi,     &
   702)                                 iblk,     &
   703)                                 tmin(:,:,:,n,iblk), tmax(:,:,:,n,iblk),  &
   704)                                 aim (:,:,  n,iblk), trm (:,:,:,n,iblk),  &
   705)                                 l_stop,     &
   706)                                 istop,              jstop)
   707) 
   708)                if (l_stop) then
   709)                   write (nu_diag,*) 'istep1, my_task, iblk, cat =',     &
   710)                                      istep1, my_task, iblk, n
   711)                   write (nu_diag,*) 'i_glob, j_glob',this_block%i_glob(istop), &
   712)                                                      this_block%j_glob(jstop)
   713)                   call abort_ice('ice remap transport: monotonicity error')
   714)                endif
   715) 
   716)             enddo               ! n
   717) 
   718)          enddo                  ! iblk
   719) 
   720)          deallocate(tmin, tmax, STAT=alloc_error)
   721)          if (alloc_error /= 0) call abort_ice ('deallocation error')
   722) 
   723)       endif                     ! l_monotonicity_check
   724) 
   725)       call ice_timer_stop(timer_advect)  ! advection 
   726)       call t_stopf   ('cice_remap_check2')
   727) 
   728)       end subroutine transport_remap

   733) ! !IROUTINE: transport_upwind - upwind transport
   734) !
   735) ! !INTERFACE:
   736) !
   737)       subroutine transport_upwind (dt)
   738) !
   739) ! !DESCRIPTION:
   740) !
   741) ! Computes the transport equations for one timestep using upwind. Sets
   742) ! several fields into a work array and passes it to upwind routine.
   743) !
   744) ! !REVISION HISTORY:
   745) !
   746) ! same as module
   747) !
   748) ! !USES:
   749) !
   750)       use ice_boundary
   751)       use ice_blocks
   752)       use ice_domain
   753)       use ice_state
   754)       use ice_grid, only: HTE, HTN, tarea
   755)       use ice_timers
   756) !
   757) ! !INPUT/OUTPUT PARAMETERS:
   758) !
   759)       real (kind=dbl_kind), intent(in) ::     &
   760)          dt      ! time step
   761) !
   762) !EOP
   763) !
   764)       integer (kind=int_kind) ::     &
   765)          narr               ! number of state variable arrays
   766)                             ! not including eicen, esnon
   767) 
   768)       integer (kind=int_kind) ::     &
   769)          i, j, iblk       ,&! horizontal indices
   770)          ilo,ihi,jlo,jhi    ! beginning and end of physical domain
   771) 
   772)       real (kind=dbl_kind), dimension (nx_block,ny_block,nblocks) ::     &
   773)          uee, vnn           ! cell edge velocities
   774) 
   775)       real (kind=dbl_kind),     &
   776)          dimension (:,:,:,:), allocatable ::      &
   777)          works              ! work array
   778) 
   779)       type (block) ::     &
   780)          this_block           ! block information for current block
   781) 
   782)       call ice_timer_start(timer_advect)  ! advection 
   783) 
   784)       narr = 1 + ncat*(3+ntrcr) ! max number of state variable arrays
   785)                                 ! not including eicen, esnon
   786) 
   787)       allocate (works(nx_block,ny_block,narr,max_blocks))
   788) 
   789)     !-------------------------------------------------------------------
   790)     ! Get ghost cell values of state variables.
   791)     ! (Assume velocities are already known for ghost cells, also.)
   792)     !-------------------------------------------------------------------
   793) !      call bound_state (aicen, trcrn,     &
   794) !                        vicen, vsnon,     &
   795) !                        eicen, esnon)
   796) 
   797)       uee(:,:,:) = c0
   798)       vnn(:,:,:) = c0
   799) 
   800)     !-------------------------------------------------------------------
   801)     ! Average corner velocities to edges.
   802)     !-------------------------------------------------------------------
   803)       
   804)       !$OMP PARALLEL DO PRIVATE(iblk,this_block,i,j,ilo,ihi,jlo,jhi)
   805)       do iblk = 1, nblocks
   806)          this_block = get_block(blocks_ice(iblk),iblk)         
   807)          ilo = this_block%ilo
   808)          ihi = this_block%ihi
   809)          jlo = this_block%jlo
   810)          jhi = this_block%jhi
   811) 
   812)          do j = jlo, jhi
   813)          do i = ilo, ihi
   814)             uee(i,j,iblk) = p5*(uvel(i,j,iblk) + uvel(i,j-1,iblk))
   815)             vnn(i,j,iblk) = p5*(vvel(i,j,iblk) + vvel(i-1,j,iblk))
   816)          enddo
   817)          enddo
   818)       enddo
   819)       !$OMP END PARALLEL DO
   820) 
   821)       call ice_timer_start(timer_bound)
   822)       call ice_HaloUpdate (uee,             halo_info,     &
   823)                            field_loc_Eface, field_type_vector)
   824)       call ice_HaloUpdate (vnn,             halo_info,     &
   825)                            field_loc_Nface, field_type_vector)
   826)       call ice_timer_stop(timer_bound)
   827) 
   828)       !$OMP PARALLEL DO PRIVATE(iblk,this_block,ilo,ihi,jlo,jhi)
   829)       do iblk = 1, nblocks
   830)          this_block = get_block(blocks_ice(iblk),iblk)         
   831)          ilo = this_block%ilo
   832)          ihi = this_block%ihi
   833)          jlo = this_block%jlo
   834)          jhi = this_block%jhi
   835) 
   836) 
   837)       !-----------------------------------------------------------------
   838)       ! fill work arrays with fields to be advected
   839)       !-----------------------------------------------------------------
   840) 
   841)          call state_to_work (nx_block,             ny_block,             &
   842)                              ntrcr,                                      &
   843)                              narr,                 trcr_depend,          &
   844)                              aicen (:,:,  :,iblk), trcrn (:,:,:,:,iblk), &
   845)                              vicen (:,:,  :,iblk), vsnon (:,:,  :,iblk), &
   846)                              aice0 (:,:,    iblk), works (:,:,  :,iblk))
   847) 
   848)       !-----------------------------------------------------------------
   849)       ! advect
   850)       !-----------------------------------------------------------------
   851) 
   852)          call upwind_field (nx_block,       ny_block,               &
   853)                             ilo, ihi,       jlo, jhi,               &
   854)                             dt,                                     &
   855)                             narr,           works(:,:,:,iblk),      &
   856)                             uee(:,:,iblk),  vnn    (:,:,iblk),      &
   857)                             HTE(:,:,iblk),  HTN    (:,:,iblk),      &
   858)                             tarea(:,:,iblk))
   859) 
   860)          call upwind_field (nx_block,       ny_block,               &
   861)                             ilo, ihi,       jlo, jhi,               &
   862)                             dt,                                     &
   863)                             ntilyr,         eicen(:,:,:,iblk),      &
   864)                             uee(:,:,iblk),  vnn    (:,:,iblk),      &
   865)                             HTE(:,:,iblk),  HTN    (:,:,iblk),      &
   866)                             tarea(:,:,iblk))
   867) 
   868)          call upwind_field (nx_block,       ny_block,               &
   869)                             ilo, ihi,       jlo, jhi,               &
   870)                             dt,                                     &
   871)                             ntslyr,         esnon(:,:,:,iblk),      &
   872)                             uee(:,:,iblk),  vnn    (:,:,iblk),      &
   873)                             HTE(:,:,iblk),  HTN    (:,:,iblk),      &
   874)                             tarea(:,:,iblk))
   875) 
   876)       !-----------------------------------------------------------------
   877)       ! convert work arrays back to state variables
   878)       !-----------------------------------------------------------------
   879) 
   880)          call work_to_state (nx_block,            ny_block,              &
   881)                              ntrcr,                                      &
   882)                              narr,                trcr_depend,           &
   883)                              aicen(:,:,  :,iblk), trcrn (:,:,:,:,iblk),  &
   884)                              vicen(:,:,  :,iblk), vsnon (:,:,  :,iblk),  &
   885)                              aice0(:,:,    iblk), works (:,:,  :,iblk)) 
   886) 
   887)       enddo                     ! iblk
   888)       !$OMP END PARALLEL DO
   889)  
   890)     !-------------------------------------------------------------------
   891)     ! Ghost cell updates for state variables.
   892)     !-------------------------------------------------------------------
   893) 
   894)       call ice_timer_start(timer_bound)
   895) 
   896)       call bound_state (aicen, trcrn,     &
   897)                         vicen, vsnon,      &
   898)                         eicen, esnon)
   899) 
   900)       call ice_timer_stop(timer_bound)
   901) 
   902)       call ice_timer_stop(timer_advect)  ! advection 
   903) 
   904)       deallocate(works)
   905) 
   906)       end subroutine transport_upwind

   915) ! !IROUTINE: state_to_tracers -fill ice area and tracer arrays
   916) !
   917) ! !INTERFACE:
   918) !
   919)       subroutine state_to_tracers (nx_block, ny_block,   &
   920)                                    ntrcr,    ntrace,     &
   921)                                    aice0,                &
   922)                                    aicen,    trcrn,      &
   923)                                    vicen,    vsnon,      &
   924)                                    eicen,    esnon,      &
   925)                                    aim,      trm)
   926) !
   927) ! !DESCRIPTION:
   928) !
   929) ! Fill ice area and tracer arrays.
   930) ! Assume that the advected tracers are hicen, hsnon, trcrn, 
   931) !  qicen(1:nilyr), and qsnon(1:nslyr).
   932) ! This subroutine must be modified if a different set of tracers
   933) !   is to be transported.  The rule for ordering tracers
   934) !   is that a dependent tracer (such as qice) must have a larger
   935) !   tracer index than the tracer it depends on (i.e., hice).
   936) !
   937) ! !REVISION HISTORY:
   938) !
   939) ! author William H. Lipscomb, LANL
   940) !
   941) ! !USES:
   942) !
   943)       use ice_itd, only: ilyr1, slyr1
   944) !
   945) ! !INPUT/OUTPUT PARAMETERS:
   946) !
   947)       integer (kind=int_kind), intent(in) ::     &
   948)            nx_block, ny_block, &  ! block dimensions
   949)            ntrcr             , & ! number of tracers in use
   950)            ntrace                ! number of tracers in use incl. hi, hs
   951) 
   952)       real (kind=dbl_kind), dimension (nx_block,ny_block),     &
   953)            intent(in) ::     &
   954)            aice0     ! fractional open water area
   955) 
   956)       real (kind=dbl_kind), dimension (nx_block,ny_block,ncat),     &
   957)            intent(in) ::     &
   958)            aicen   ,&! fractional ice area
   959)            vicen   ,&! volume per unit area of ice          (m)
   960)            vsnon     ! volume per unit area of snow         (m)
   961) 
   962)       real (kind=dbl_kind), dimension (nx_block,ny_block,max_ntrcr,ncat),     &
   963)            intent(in) ::     &
   964)            trcrn     ! ice area tracers
   965) 
   966)       real (kind=dbl_kind), dimension (nx_block,ny_block,ntilyr),     &
   967)            intent(in) ::     &
   968)            eicen     ! energy of melting for each ice layer (J/m^2)
   969) 
   970)       real (kind=dbl_kind), dimension (nx_block,ny_block,ntslyr),     &
   971)            intent(in) ::     &
   972)            esnon     ! energy of melting for each snow layer (J/m^2)
   973) 
   974)       real (kind=dbl_kind), dimension (nx_block,ny_block,0:ncat),     &
   975)             intent(out)::     &
   976)            aim       ! mean ice area in each grid cell
   977) 
   978)       real (kind=dbl_kind), dimension (nx_block,ny_block,max_ntrace,ncat),  &
   979)            intent(out) ::     &
   980)            trm       ! mean tracer values in each grid cell
   981) 
   982)       real (kind=dbl_kind), dimension (nx_block,ny_block) :: &
   983)          worka, &
   984)          workb
   985) !
   986) !EOP
   987) !
   988)       integer (kind=int_kind) ::     &
   989)            i, j, k, n   ,&! standard indices
   990)            it, kt       ,&! tracer indices
   991)            ij             ! combined i/j index
   992) 
   993)       real (kind=dbl_kind) ::     &
   994)            w1             ! work variable
   995) 
   996)       integer (kind=int_kind), dimension(nx_block*ny_block,0:ncat) ::  &
   997)            indxi        ,&! compressed i/j indices
   998)            indxj
   999) 
  1000)       integer (kind=int_kind), dimension(0:ncat) ::     &
  1001)            icells         ! number of cells with ice
  1002) 
  1003)       worka(:,:) = c0
  1004)       workb(:,:) = c0
  1005) 
  1006)       aim(:,:,0) = aice0(:,:)
  1007) 
  1008)       do n = 1, ncat
  1009) 
  1010)          trm(:,:,:,n) = c0
  1011) 
  1012)     !-------------------------------------------------------------------
  1013)     ! Find grid cells where ice is present and fill area array.
  1014)     !-------------------------------------------------------------------
  1015) 
  1016)          icells(n) = 0
  1017)          do j = 1, ny_block
  1018)          do i = 1, nx_block
  1019)             aim(i,j,n) = aicen(i,j,n)
  1020)             if (aim(i,j,n) > puny) then
  1021)                icells(n) = icells(n) + 1
  1022)                ij = icells(n)
  1023)                indxi(ij,n) = i
  1024)                indxj(ij,n) = j
  1025)             endif               ! aim > puny
  1026)          enddo
  1027)          enddo
  1028)       
  1029)     !-------------------------------------------------------------------
  1030)     ! Fill tracer array
  1031)     ! Note: If aice > 0, then hice > 0, but we can have hsno = 0.
  1032)     ! Alse note: We transport qice*nilyr rather than qice, so as to
  1033)     !  avoid extra operations here and in tracers_to_state.
  1034)     !-------------------------------------------------------------------
  1035) 
  1036)          do ij = 1, icells(n)
  1037)             i = indxi(ij,n)
  1038)             j = indxj(ij,n)
  1039)             w1 = c1 / aim(i,j,n)
  1040)             worka(i,j) = c1 / vicen(i,j,n)
  1041)             trm(i,j,1,n) = vicen(i,j,n) * w1 ! hice
  1042)             trm(i,j,2,n) = vsnon(i,j,n) * w1 ! hsno
  1043)             if (trm(i,j,2,n) > puny) then
  1044)                workb(i,j) = c1 / vsnon(i,j,n)
  1045)             else
  1046)                workb(i,j) = c0
  1047)             endif
  1048)          enddo
  1049)          kt = 2
  1050) 
  1051)          do it = 1, ntrcr
  1052)             do ij = 1, icells(n)
  1053)                i = indxi(ij,n)
  1054)                j = indxj(ij,n)
  1055)                trm(i,j,kt+it,n) = trcrn(i,j,it,n) ! ice area tracers
  1056)             enddo
  1057)          enddo
  1058)          kt = kt + ntrcr
  1059) 
  1060)          do k =1, nilyr
  1061)             do ij = 1, icells(n)
  1062)                i = indxi(ij,n)
  1063)                j = indxj(ij,n)
  1064)                trm(i,j,kt+k,n) = eicen(i,j,ilyr1(n)+k-1)*worka(i,j) ! qice
  1065)             enddo               ! ij
  1066)          enddo                  ! ilyr
  1067)          kt = kt + nilyr
  1068) 
  1069)          do k = 1, nslyr
  1070)             do ij = 1, icells(n)
  1071)                i = indxi(ij,n)
  1072)                j = indxj(ij,n)
  1073)                if (trm(i,j,2,n) > puny)    &    ! hsno > puny
  1074)                  trm(i,j,kt+k,n) = esnon(i,j,slyr1(n)+k-1)*workb(i,j) & ! qsno
  1075)                                  + rhos*Lfresh
  1076)             enddo               ! ij
  1077)          enddo                  ! nslyr
  1078) 
  1079)       enddo                     ! ncat
  1080)  
  1081)       end subroutine state_to_tracers

  1301) ! !IROUTINE: local_max_min - compute local max and min of a scalar field
  1302) !
  1303) ! !INTERFACE:
  1304) !
  1305)       subroutine local_max_min (nx_block, ny_block,     &
  1306)                                 ilo, ihi, jlo, jhi,     &
  1307)                                 trm,                    &
  1308)                                 tmin,     tmax,         &
  1309)                                 aimask,   trmask)
  1310) !
  1311) ! !DESCRIPTION:
  1312) !

  1314) ! field phi: i.e., the max and min values in the nine-cell region
  1315) ! consisting of the home cell and its eight neighbors.
  1316) ! 
  1317) ! To extend to the neighbors of the neighbors (25 cells in all),
  1318) ! follow this call with a call to quasilocal_max_min.
  1319) !
  1320) ! !REVISION HISTORY:
  1321) !
  1322) ! author William H. Lipscomb, LANL
  1323) !
  1324) ! !USES:
  1325) !
  1326) ! !INPUT/OUTPUT PARAMETERS:
  1327) !
  1328)       integer (kind=int_kind), intent(in) ::     &
  1329)            nx_block, ny_block,&! block dimensions
  1330)            ilo,ihi,jlo,jhi     ! beginning and end of physical domain
  1331) 
  1332)       real (kind=dbl_kind), intent(in),        &
  1333)            dimension(nx_block,ny_block) ::     &
  1334)            aimask         ! ice area mask
  1335) 
  1336)       real (kind=dbl_kind), intent(in),               &
  1337)            dimension (nx_block,ny_block,max_ntrace) ::    &
  1338)            trm          ,&! tracer fields
  1339)            trmask         ! tracer mask
  1340) 
  1341)       real (kind=dbl_kind), intent(out),              &
  1342)            dimension (nx_block,ny_block,max_ntrace) ::    &
  1343)            tmin         ,&! local min tracer
  1344)            tmax           ! local max tracer
  1345) !
  1346) !EOP
  1347) !
  1348)       integer (kind=int_kind) ::     &
  1349)            i, j         ,&! horizontal indices
  1350)            nt, nt1        ! tracer indices
  1351) 
  1352)       real (kind=dbl_kind), dimension(nx_block,ny_block) ::     &
  1353)            phimask        ! aimask or trmask, as appropriate
  1354) 
  1355)       real (kind=dbl_kind) ::     &
  1356)            phi_nw, phi_n, phi_ne ,&! field values in 8 neighbor cells
  1357)            phi_w, phi_e          ,&
  1358)            phi_sw, phi_s, phi_se
  1359) 
  1360)       do nt = 1, ntrace
  1361) 
  1362)          if (tracer_type(nt)==1) then  ! does not depend on another tracer
  1363) 
  1364)             do j = 1, ny_block
  1365)             do i = 1, nx_block
  1366)                phimask(i,j) = aimask(i,j)
  1367)             enddo
  1368)             enddo
  1369) 
  1370)          else   ! depends on another tracer
  1371) 
  1372)             nt1 = depend(nt)
  1373)             do j = 1, ny_block
  1374)             do i = 1, nx_block
  1375)                phimask(i,j) = trmask(i,j,nt1)
  1376)             enddo
  1377)             enddo
  1378) 
  1379)          endif

  1384) !  so that non-physical values of phi do not contribute to the gradient.
  1385) !-----------------------------------------------------------------------
  1386) 
  1387)          do j = jlo, jhi
  1388)             do i = ilo, ihi
  1389) 
  1390)                phi_nw = phimask(i-1,j+1) * trm(i-1,j+1,nt)     &
  1391)                   + (c1-phimask(i-1,j+1))* trm(i,  j,  nt)

  1393)                   + (c1-phimask(i,  j+1))* trm(i,  j,  nt)
  1394)                phi_ne = phimask(i+1,j+1) * trm(i+1,j+1,nt)     &
  1395)                   + (c1-phimask(i+1,j+1))* trm(i,  j,  nt)
  1396)                phi_w  = phimask(i-1,j)   * trm(i-1,j,  nt)     &
  1397)                   + (c1-phimask(i-1,j))  * trm(i,  j,  nt)
  1398)                phi_e  = phimask(i+1,j)   * trm(i+1,j,  nt)     &
  1399)                   + (c1-phimask(i+1,j))  * trm(i,  j,  nt)
  1400)                phi_sw = phimask(i-1,j-1) * trm(i-1,j-1,nt)     &
  1401)                   + (c1-phimask(i-1,j-1))* trm(i,  j,  nt)

  1403)                   + (c1-phimask(i,  j-1))* trm(i,  j,  nt)
  1404)                phi_se = phimask(i+1,j-1) * trm(i+1,j-1,nt)     &
  1405)                   + (c1-phimask(i+1,j-1))* trm(i,  j,  nt)
  1406) 
  1407) !-----------------------------------------------------------------------
  1408) !     Compute the minimum and maximum among the nine local cells.
  1409) !-----------------------------------------------------------------------
  1410) 
  1411)                tmax(i,j,nt) = max (phi_nw, phi_n,  phi_ne, phi_w,     &
  1412)                       trm(i,j,nt), phi_e,  phi_sw, phi_s,  phi_se)
  1413) 
  1414)                tmin(i,j,nt) = min (phi_nw, phi_n,  phi_ne, phi_w,     &
  1415)                       trm(i,j,nt), phi_e,  phi_sw, phi_s,  phi_se)
  1416) 
  1417)             enddo               ! i
  1418)          enddo                  ! j
  1419) 
  1420)       enddo                     ! nt

  1512) !
  1513) ! !INPUT/OUTPUT PARAMETERS:

  1541)       integer (kind=int_kind) ::     &
  1542)            i, j           ,&! horizontal indices

  1544) 
  1545)       real (kind=dbl_kind) ::     &

  1547)

  1550)

  1558) 
  1559)             do j = jlo, jhi

  1562)                   l_check(i,j) = .true.

  1586)             do j = jlo, jhi

  1589)                    abs(trm(i,j,nt2)) > puny .and. aim(i,j) > puny) then
  1590)                   l_check(i,j) = .true.
  1591)                else
  1592)                   l_check(i,j) = .false.
  1593)                endif
  1594)             enddo
  1595)             enddo

  1599)     ! Make sure new values lie between tmin and tmax
  1600)     !-------------------------------------------------------------------
