    39)   subroutine lin_strat_chem_inti(phys_state)
    40)     !
    41)     ! initialize linearized stratospheric chemistry by reading
    42)     ! input parameters from netcdf file and interpolate to
    43)     ! present model grid
    44)     !
    45)     use linoz_data,   only : linoz_data_init, has_linoz_data
    46)     use ppgrid,       only : pver
    47)     use mo_chem_utls, only : get_spc_ndx
    48)     use cam_history,  only : addfld, phys_decomp, add_default
    49)     use physics_buffer, only : physics_buffer_desc
    50)     use phys_control, only : phys_getopts
    51)

    53) 
    54) 
    55)     type(physics_state), intent(in) :: phys_state(begchunk:endchunk)
    56) 
    57)     logical :: history_chemistry

    78)     endif

    83)

    87)     call addfld( 'LINOZ_DO3_PSC', '/s'     , pver, 'A', 'ozone vmr loss by PSCs using Carille et al. (1990)', phys_decomp )
    88)     call addfld( 'LINOZ_SSO3'   , 'kg'     , pver, 'A', 'steady state ozone in LINOZ'                       , phys_decomp )

    91)     call addfld( 'LINOZ_SZA'    , 'degrees',    1, 'A', 'solar zenith angle in LINOZ'                       , phys_decomp )

   108)   subroutine lin_strat_chem_solve( ncol, lchnk, o3_vmr, o3col, temp, sza, pmid, delta_t, rlats, ltrop )
   109)  
   110)     use chlorine_loading_data, only: chlorine_loading
   111) 
   112)     !
   113)     ! this subroutine updates the ozone mixing ratio in the stratosphere
   114)     ! using linearized chemistry 
   115)     !
   116) 
   117)     use ppgrid,        only : pcols, pver
   118)     use physconst,     only : pi, &
   119)                               grav => gravit, &
   120)                               mw_air => mwdry
   121)     use cam_history,   only : outfld
   122)     use linoz_data,    only : fields, o3_clim_ndx,t_clim_ndx,o3col_clim_ndx,PmL_clim_ndx,dPmL_dO3_ndx,&
   123)                                       dPmL_dT_ndx,dPmL_dO3col_ndx,cariolle_pscs_ndx
   124)     !
   125)     ! dummy arguments
   126)     !
   127)     integer,  intent(in)                           :: ncol                ! number of columns in chunk
   128)     integer,  intent(in)                           :: lchnk               ! chunk index
   129)     real(r8), intent(inout), dimension(ncol ,pver) :: o3_vmr              ! ozone volume mixing ratio
   130)     real(r8), intent(in)   , dimension(ncol ,pver) :: o3col               ! ozone column above box (mol/cm^2)
   131)     real(r8), intent(in)   , dimension(pcols,pver) :: temp                ! temperature (K)
   132)     real(r8), intent(in)   , dimension(ncol )      :: sza                 ! local solar zenith angle
   133)     real(r8), intent(in)   , dimension(pcols,pver) :: pmid                ! midpoint pressure (Pa)
   134)     real(r8), intent(in)                           :: delta_t             ! timestep size (secs)
   135)     real(r8), intent(in)                           :: rlats(ncol)         ! column latitudes (radians)
   136)     integer,  intent(in)   , dimension(pcols)      :: ltrop               ! chunk index
   137)     !
   138)     ! local
   139)     !
   140)     integer :: i,k,n !,index_lat,index_month
   141)     real(r8) :: o3col_du,delta_temp,delta_o3col,o3_old,o3_new,delta_o3
   142)     real(r8) :: max_sza, psc_loss
   143)     real(r8) :: o3_clim
   144)     real(r8), dimension(ncol) :: lats
   145)     real(r8), dimension(ncol,pver) :: do3_linoz,do3_linoz_psc,ss_o3,o3col_du_diag,o3clim_linoz_diag
   146) 
   147)     real(r8), dimension(:,:), pointer :: linoz_o3_clim
   148)     real(r8), dimension(:,:), pointer :: linoz_t_clim
   149)     real(r8), dimension(:,:), pointer :: linoz_o3col_clim
   150)     real(r8), dimension(:,:), pointer :: linoz_PmL_clim
   151)     real(r8), dimension(:,:), pointer :: linoz_dPmL_dO3
   152)     real(r8), dimension(:,:), pointer :: linoz_dPmL_dT
   153)     real(r8), dimension(:,:), pointer :: linoz_dPmL_dO3col
   154)     real(r8), dimension(:,:), pointer :: linoz_cariolle_psc
   155) 
   156)     !
   157)     ! parameters
   158)     !
   159)     real(r8), parameter :: convert_to_du = 1._r8/(2.687e16_r8)      ! convert ozone column from mol/cm^2 to DU
   160)     real(r8), parameter :: degrees_to_radians = pi/180._r8          ! conversion factors
   161)     real(r8), parameter :: radians_to_degrees = 180._r8/pi
   162)     real(r8), parameter :: temp_activation_cariolle = 193._r8       ! O3 loss freq when T below temp_activation_cariolle (K)
   163)     real(r8), parameter :: chlorine_loading_1987    = 2.5977_r8     ! EESC value (ppbv)
   164)     real(r8), parameter :: chlorine_loading_bgnd    = 0.0000_r8     ! EESC value (ppbv) for background conditions
   165)     real(r8), parameter :: pressure_threshold       = 210.e+2_r8    ! {PJC} for diagnostics only
   166) 
   167)     !
   168)     ! skip if no ozone field available
   169)     !
   170)     if ( .not. do_lin_strat_chem ) return
   171) 
   172)     ! 
   173)     ! associate the field pointers
   174)     !
   175)     linoz_o3_clim      => fields(o3_clim_ndx)      %data(:,:,lchnk )
   176)     linoz_t_clim       => fields(t_clim_ndx)       %data(:,:,lchnk )
   177)     linoz_o3col_clim   => fields(o3col_clim_ndx)   %data(:,:,lchnk )
   178)     linoz_PmL_clim     => fields(PmL_clim_ndx)     %data(:,:,lchnk )
   179)     linoz_dPmL_dO3     => fields(dPmL_dO3_ndx)     %data(:,:,lchnk )
   180)     linoz_dPmL_dT      => fields(dPmL_dT_ndx)      %data(:,:,lchnk )
   181)     linoz_dPmL_dO3col  => fields(dPmL_dO3col_ndx)  %data(:,:,lchnk )
   182)     linoz_cariolle_psc => fields(cariolle_pscs_ndx)%data(:,:,lchnk )
   183) 
   184)     !
   185)     ! initialize output arrays
   186)     !
   187)     do3_linoz         = 0._r8
   188)     do3_linoz_psc     = 0._r8
   189)     o3col_du_diag     = 0._r8
   190)     o3clim_linoz_diag = 0._r8
   191)     ss_o3             = 0._r8
   192)     !
   193)     ! convert lats from radians to degrees
   194)     !
   195)     lats = rlats * radians_to_degrees
   196) 
   197)     LOOP_COL: do i=1,ncol
   198)        LOOP_LEV: do k=1,ltrop(i)
   199)           !
   200)           ! climatological ozone
   201)           !
   202)           o3_clim = linoz_o3_clim(i,k)
   203)           !
   204)           ! skip if not in the stratosphere
   205)           !
   206)           if ( pmid(i,k) > pressure_threshold ) THEN   ! PJC diagnostic
   207)              WRITE(iulog,*)'LINOZ WARNING: Exceeded PRESSURE threshold (i,k,p_threshold,pmid,o3)=',&
   208)                i,k,nint(pressure_threshold/100._r8),'mb',nint(pmid(i,k)/100._r8),'mb',nint(o3_vmr(i,k)*1e9_r8),'ppb'   !PJC
   209) !             cycle LOOP_LEV
   210)           endif
   211)           !
   212)           ! diagnostic for output
   213)           !
   214)           o3clim_linoz_diag(i,k) = o3_clim
   215)           !
   216)           ! old ozone mixing ratio
   217)           !
   218)           o3_old = o3_vmr(i,k)
   219)           !
   220)           ! convert o3col from mol/cm2
   221)           !
   222)           o3col_du = o3col(i,k) * convert_to_du
   223)           o3col_du_diag(i,k) = o3col_du
   224)           !
   225)           ! compute differences from climatology
   226)           !
   227)           delta_temp  = temp(i,k) - linoz_t_clim    (i,k)
   228)           delta_o3col = o3col_du  - linoz_o3col_clim(i,k)
   229) 
   230) 
   231)           !
   232)           ! steady state ozone
   233)           !
   234)           ss_o3(i,k) = o3_clim - (               linoz_PmL_clim   (i,k)   &
   235)                                  + delta_o3col * linoz_dPmL_dO3col(i,k)   &
   236)                                  + delta_temp  * linoz_dPmL_dT    (i,k)   &
   237)                                              ) / linoz_dPmL_dO3   (i,k)
   238) 
   239) 
   240)           !
   241)           ! ozone change
   242)           !
   243)           delta_o3 = (ss_o3(i,k)-o3_old) * (1._r8 - exp(linoz_dPmL_dO3(i,k)*delta_t))
   244)           !
   245)           ! define new ozone mixing ratio
   246)           !
   247)           o3_new = o3_old + delta_o3
   248)           !
   249)           ! output diagnostic
   250)           !
   251)           do3_linoz(i,k) = delta_o3/delta_t
   252)           !
   253)           ! PSC activation (follows Cariolle et al 1990.)
   254)           !
   255)           ! use only if abs(latitude) > 40.
   256)           !
   257)           if ( abs(lats(i)) > 40._r8 ) then   
   258)              if ( (chlorine_loading-chlorine_loading_bgnd) > 0._r8 ) then
   259)                 if ( temp(i,k) <= temp_activation_cariolle ) then
   260)                    !
   261)                    ! define maximum SZA for PSC loss (= tangent height at sunset)
   262)                    !
   263)                    max_sza = (90._r8 + sqrt( max( 16._r8*log10(100000._r8/pmid(i,k)),0._r8)))
   264) #ifdef DEBUG
   265)                    write(iulog,*)sza(i),max_sza
   266) #endif
   267)                    if ( (sza(i)*radians_to_degrees) <= max_sza ) then
   268) 
   269)                       psc_loss = exp(-linoz_cariolle_psc(i,k) &
   270)                            * (chlorine_loading/chlorine_loading_1987)**2 &
   271)                            * delta_t )
   272) 
   273)                       o3_new = o3_old * psc_loss
   274)                       !
   275)                       ! output diagnostic
   276)                       !
   277)                       do3_linoz_psc(i,k) = (o3_new-o3_old)/delta_t
   278)                       !
   279)                    end if
   280)                 end if
   281)              end if
   282)           end if
   283)           !
   284)           ! update ozone vmr
   285)           !
   286)           o3_vmr(i,k) = o3_new
   287) 
   288)        end do LOOP_LEV
   289)     end do LOOP_COL
   290)     !
   291)     ! output
   292)     !
   293)     call outfld( 'LINOZ_DO3'    , do3_linoz              , ncol, lchnk )
   294)     call outfld( 'LINOZ_DO3_PSC', do3_linoz_psc          , ncol, lchnk )
   295)     call outfld( 'LINOZ_SSO3'   , ss_o3                  , ncol, lchnk )
   296)     call outfld( 'LINOZ_O3COL'  , o3col_du_diag          , ncol, lchnk )
   297)     call outfld( 'LINOZ_O3CLIM' , o3clim_linoz_diag      , ncol, lchnk )
   298)     call outfld( 'LINOZ_SZA'    ,(sza*radians_to_degrees), ncol, lchnk )
   299) 
   300)     return
   301)   end subroutine lin_strat_chem_solve
