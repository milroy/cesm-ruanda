     1) ! FFT99F
     2) !
     3) ! PURPOSE      PERFORMS MULTIPLE FAST FOURIER TRANSFORMS.  THIS PACKAGE
     4) !              WILL PERFORM A NUMBER OF SIMULTANEOUS REAL/HALF-COMPLEX
     5) !              PERIODIC FOURIER TRANSFORMS OR CORRESPONDING INVERSE
     6) !              TRANSFORMS, I.E.  GIVEN A SET OF REAL DATA VECTORS, THE
     7) !              PACKAGE RETURNS A SET OF 'HALF-COMPLEX' FOURIER
     8) !              COEFFICIENT VECTORS, OR VICE VERSA.  THE LENGTH OF THE
     9) !              TRANSFORMS MUST BE AN EVEN NUMBER GREATER THAN 4 THAT HAS
    10) !              NO OTHER FACTORS EXCEPT POSSIBLY POWERS OF 2, 3, AND 5.
    11) !              THIS IS AN ALL FORTRAN VERSION OF THE CRAYLIB PACKAGE
    12) !              THAT IS MOSTLY WRITTEN IN CAL.
    13) !
    14) !              THE PACKAGE FFT99F CONTAINS SEVERAL USER-LEVEL ROUTINES:
    15) !
    16) !            SUBROUTINE SET99
    17) !                AN INITIALIZATION ROUTINE THAT MUST BE CALLED ONCE
    18) !                BEFORE A SEQUENCE OF CALLS TO THE FFT ROUTINES
    19) !                (PROVIDED THAT N IS NOT CHANGED).
    20) !
    21) !            SUBROUTINES FFT99 AND FFT991
    22) !                TWO FFT ROUTINES THAT RETURN SLIGHTLY DIFFERENT
    23) !                ARRANGEMENTS OF THE DATA IN GRIDPOINT SPACE.
    24) !
    25) !
    26) ! ACCESS       THIS FORTRAN VERSION MAY BE ACCESSED WITH
    27) !
    28) !                   *FORTRAN,P=XLIB,SN=FFT99F
    29) !
    30) !              TO ACCESS THE CRAY OBJECT CODE, CALLING THE USER ENTRY
    31) !              POINTS FROM A CRAY PROGRAM IS SUFFICIENT.  THE SOURCE
    32) !              FORTRAN AND CAL CODE FOR THE CRAYLIB VERSION MAY BE
    33) !              ACCESSED USING
    34) !
    35) !                   FETCH P=CRAYLIB,SN=FFT99
    36) !                   FETCH P=CRAYLIB,SN=CAL99
    37) !
    38) ! USAGE        LET N BE OF THE FORM 2**P * 3**Q * 5**R, WHERE P .GE. 1,
    39) !              Q .GE. 0, AND R .GE. 0.  THEN A TYPICAL SEQUENCE OF
    40) !              CALLS TO TRANSFORM A GIVEN SET OF REAL VECTORS OF LENGTH
    41) !              N TO A SET OF 'HALF-COMPLEX' FOURIER COEFFICIENT VECTORS
    42) !              OF LENGTH N IS
    43) !
    44) !                   DIMENSION IFAX(13),TRIGS(3*N/2+1),A(M*(N+2)),
    45) !                  +          WORK(M*(N+1))
    46) !
    47) !                   CALL SET99 (TRIGS, IFAX, N)
    48) !                   CALL FFT99 (A,WORK,TRIGS,IFAX,INC,JUMP,N,M,ISIGN)
    49) !
    50) !              SEE THE INDIVIDUAL WRITE-UPS FOR SET99, FFT99, AND
    51) !              FFT991 BELOW, FOR A DETAILED DESCRIPTION OF THE
    52) !              ARGUMENTS.
    53) !
    54) ! HISTORY      THE PACKAGE WAS WRITTEN BY CLIVE TEMPERTON AT ECMWF IN
    55) !              NOVEMBER, 1978.  IT WAS MODIFIED, DOCUMENTED, AND TESTED
    56) !              FOR NCAR BY RUSS REW IN SEPTEMBER, 1980.
    57) !
    58) !-----------------------------------------------------------------------
    59) !
    60) ! SUBROUTINE SET99 (TRIGS, IFAX, N)
    61) !
    62) ! PURPOSE      A SET-UP ROUTINE FOR FFT99 AND FFT991.  IT NEED ONLY BE
    63) !              CALLED ONCE BEFORE A SEQUENCE OF CALLS TO THE FFT
    64) !              ROUTINES (PROVIDED THAT N IS NOT CHANGED).
    65) !
    66) ! ARGUMENT     IFAX(13),TRIGS(3*N/2+1)
    67) ! DIMENSIONS
    68) !
    69) ! ARGUMENTS
    70) !
    71) ! ON INPUT     TRIGS
    72) !               A FLOATING POINT ARRAY OF DIMENSION 3*N/2 IF N/2 IS
    73) !               EVEN, OR 3*N/2+1 IF N/2 IS ODD.
    74) !
    75) !              IFAX
    76) !               AN INTEGER ARRAY.  THE NUMBER OF ELEMENTS ACTUALLY USED
    77) !               WILL DEPEND ON THE FACTORIZATION OF N.  DIMENSIONING
    78) !               IFAX FOR 13 SUFFICES FOR ALL N LESS THAN A MILLION.
    79) !
    80) !              N
    81) !               AN EVEN NUMBER GREATER THAN 4 THAT HAS NO PRIME FACTOR
    82) !               GREATER THAN 5.  N IS THE LENGTH OF THE TRANSFORMS (SEE
    83) !               THE DOCUMENTATION FOR FFT99 AND FFT991 FOR THE
    84) !               DEFINITIONS OF THE TRANSFORMS).
    85) !
    86) ! ON OUTPUT    IFAX
    87) !               CONTAINS THE FACTORIZATION OF N/2.  IFAX(1) IS THE
    88) !               NUMBER OF FACTORS, AND THE FACTORS THEMSELVES ARE STORED
    89) !               IN IFAX(2),IFAX(3),...  IF SET99 IS CALLED WITH N ODD,
    90) !               OR IF N HAS ANY PRIME FACTORS GREATER THAN 5, IFAX(1)
    91) !               IS SET TO -99.
    92) !
    93) !              TRIGS
    94) !               AN ARRAY OF TRIGONOMETRIC FUNCTION VALUES SUBSEQUENTLY
    95) !               USED BY THE FFT ROUTINES.
    96) !
    97) !-----------------------------------------------------------------------
    98) !
    99) ! SUBROUTINE FFT991 (A,WORK,TRIGS,IFAX,INC,JUMP,N,M,ISIGN)
   100) !                       AND
   101) ! SUBROUTINE FFT99 (A,WORK,TRIGS,IFAX,INC,JUMP,N,M,ISIGN)
   102) !
   103) ! PURPOSE      PERFORM A NUMBER OF SIMULTANEOUS REAL/HALF-COMPLEX
   104) !              PERIODIC FOURIER TRANSFORMS OR CORRESPONDING INVERSE
   105) !              TRANSFORMS, USING ORDINARY SPATIAL ORDER OF GRIDPOINT
   106) !              VALUES (FFT991) OR EXPLICIT CYCLIC CONTINUITY IN THE
   107) !              GRIDPOINT VALUES (FFT99).  GIVEN A SET
   108) !              OF REAL DATA VECTORS, THE PACKAGE RETURNS A SET OF
   109) !              'HALF-COMPLEX' FOURIER COEFFICIENT VECTORS, OR VICE
   110) !              VERSA.  THE LENGTH OF THE TRANSFORMS MUST BE AN EVEN
   111) !              NUMBER THAT HAS NO OTHER FACTORS EXCEPT POSSIBLY POWERS
   112) !              OF 2, 3, AND 5.  THESE VERSION OF FFT991 AND FFT99 ARE
   113) !              OPTIMIZED FOR USE ON THE CRAY-1.
   114) !
   115) ! ARGUMENT     A(M*(N+2)), WORK(M*(N+1)), TRIGS(3*N/2+1), IFAX(13)
   116) ! DIMENSIONS
   117) !
   118) ! ARGUMENTS
   119) !
   120) ! ON INPUT     A
   121) !               AN ARRAY OF LENGTH M*(N+2) CONTAINING THE INPUT DATA
   122) !               OR COEFFICIENT VECTORS.  THIS ARRAY IS OVERWRITTEN BY
   123) !               THE RESULTS.
   124) !
   125) !              WORK
   126) !               A WORK ARRAY OF DIMENSION M*(N+1)
   127) !
   128) !              TRIGS
   129) !               AN ARRAY SET UP BY SET99, WHICH MUST BE CALLED FIRST.
   130) !
   131) !              IFAX
   132) !               AN ARRAY SET UP BY SET99, WHICH MUST BE CALLED FIRST.
   133) !
   134) !              INC
   135) !               THE INCREMENT (IN WORDS) BETWEEN SUCCESSIVE ELEMENTS OF
   136) !               EACH DATA OR COEFFICIENT VECTOR (E.G.  INC=1 FOR
   137) !               CONSECUTIVELY STORED DATA).
   138) !
   139) !              JUMP
   140) !               THE INCREMENT (IN WORDS) BETWEEN THE FIRST ELEMENTS OF
   141) !               SUCCESSIVE DATA OR COEFFICIENT VECTORS.  ON THE CRAY-1,
   142) !               TRY TO ARRANGE DATA SO THAT JUMP IS NOT A MULTIPLE OF 8
   143) !               (TO AVOID MEMORY BANK CONFLICTS).  FOR CLARIFICATION OF
   144) !               INC AND JUMP, SEE THE EXAMPLES BELOW.
   145) !
   146) !              N
   147) !               THE LENGTH OF EACH TRANSFORM (SEE DEFINITION OF
   148) !               TRANSFORMS, BELOW).
   149) !
   150) !              M
   151) !               THE NUMBER OF TRANSFORMS TO BE DONE SIMULTANEOUSLY.
   152) !
   153) !              ISIGN
   154) !               = +1 FOR A TRANSFORM FROM FOURIER COEFFICIENTS TO
   155) !                    GRIDPOINT VALUES.
   156) !               = -1 FOR A TRANSFORM FROM GRIDPOINT VALUES TO FOURIER
   157) !                    COEFFICIENTS.
   158) !
   159) ! ON OUTPUT    A
   160) !               IF ISIGN = +1, AND M COEFFICIENT VECTORS ARE SUPPLIED
   161) !               EACH CONTAINING THE SEQUENCE:
   162) !
   163) !               A(0),B(0),A(1),B(1),...,A(N/2),B(N/2)  (N+2 VALUES)
   164) !
   165) !               THEN THE RESULT CONSISTS OF M DATA VECTORS EACH
   166) !               CONTAINING THE CORRESPONDING N+2 GRIDPOINT VALUES:
   167) !
   168) !               FOR FFT991, X(0), X(1), X(2),...,X(N-1),0,0.
   169) !               FOR FFT99, X(N-1),X(0),X(1),X(2),...,X(N-1),X(0).
   170) !                   (EXPLICIT CYCLIC CONTINUITY)
   171) !
   172) !               WHEN ISIGN = +1, THE TRANSFORM IS DEFINED BY:
   173) !                 X(J)=SUM(K=0,...,N-1)(C(K)*EXP(2*I*J*K*PI/N))
   174) !                 WHERE C(K)=A(K)+I*B(K) AND C(N-K)=A(K)-I*B(K)
   175) !                 AND I=SQRT (-1)
   176) !
   177) !               IF ISIGN = -1, AND M DATA VECTORS ARE SUPPLIED EACH
   178) !               CONTAINING A SEQUENCE OF GRIDPOINT VALUES X(J) AS
   179) !               DEFINED ABOVE, THEN THE RESULT CONSISTS OF M VECTORS
   180) !               EACH CONTAINING THE CORRESPONDING FOURIER COFFICIENTS
   181) !               A(K), B(K), 0 .LE. K .LE N/2.
   182) !
   183) !               WHEN ISIGN = -1, THE INVERSE TRANSFORM IS DEFINED BY:
   184) !                 C(K)=(1/N)*SUM(J=0,...,N-1)(X(J)*EXP(-2*I*J*K*PI/N))
   185) !                 WHERE C(K)=A(K)+I*B(K) AND I=SQRT(-1)
   186) !
   187) !               A CALL WITH ISIGN=+1 FOLLOWED BY A CALL WITH ISIGN=-1
   188) !               (OR VICE VERSA) RETURNS THE ORIGINAL DATA.
   189) !
   190) !               NOTE: THE FACT THAT THE GRIDPOINT VALUES X(J) ARE REAL
   191) !               IMPLIES THAT B(0)=B(N/2)=0.  FOR A CALL WITH ISIGN=+1,
   192) !               IT IS NOT ACTUALLY NECESSARY TO SUPPLY THESE ZEROS.
   193) !
   194) ! EXAMPLES      GIVEN 19 DATA VECTORS EACH OF LENGTH 64 (+2 FOR EXPLICIT
   195) !               CYCLIC CONTINUITY), COMPUTE THE CORRESPONDING VECTORS OF
   196) !               FOURIER COEFFICIENTS.  THE DATA MAY, FOR EXAMPLE, BE
   197) !               ARRANGED LIKE THIS:
   198) !
   199) ! FIRST DATA   A(1)=    . . .                A(66)=             A(70)
   200) ! VECTOR       X(63) X(0) X(1) X(2) ... X(63) X(0)  (4 EMPTY LOCATIONS)
   201) !
   202) ! SECOND DATA  A(71)=   . . .                                  A(140)
   203) ! VECTOR       X(63) X(0) X(1) X(2) ... X(63) X(0)  (4 EMPTY LOCATIONS)
   204) !
   205) !               AND SO ON.  HERE INC=1, JUMP=70, N=64, M=19, ISIGN=-1,
   206) !               AND FFT99 SHOULD BE USED (BECAUSE OF THE EXPLICIT CYCLIC
   207) !               CONTINUITY).
   208) !
   209) !               ALTERNATIVELY THE DATA MAY BE ARRANGED LIKE THIS:
   210) !
   211) !                FIRST         SECOND                          LAST
   212) !                DATA          DATA                            DATA
   213) !                VECTOR        VECTOR                          VECTOR
   214) !
   215) !                 A(1)=         A(2)=                           A(19)=
   216) !
   217) !                 X(63)         X(63)       . . .               X(63)
   218) !        A(20)=   X(0)          X(0)        . . .               X(0)
   219) !        A(39)=   X(1)          X(1)        . . .               X(1)
   220) !                  .             .                               .
   221) !                  .             .                               .
   222) !                  .             .                               .
   223) !
   224) !               IN WHICH CASE WE HAVE INC=19, JUMP=1, AND THE REMAINING
   225) !               PARAMETERS ARE THE SAME AS BEFORE.  IN EITHER CASE, EACH
   226) !               COEFFICIENT VECTOR OVERWRITES THE CORRESPONDING INPUT
   227) !               DATA VECTOR.
   228) !-----------------------------------------------------------------------
   229) !

   235)       SUBROUTINE FFT99(A,WORK,TRIGS,IFAX,INC,JUMP,N,LOT,ISIGN)
   236) !
   237) !-----------------------------------------------------------------------
   238) !     SUBROUTINE "FFT99" - MULTIPLE FAST REAL PERIODIC TRANSFORM
   239) !     CORRESPONDING TO OLD SCALAR ROUTINE FFT9
   240) !     PROCEDURE USED TO CONVERT TO HALF-LENGTH COMPLEX TRANSFORM
   241) !     IS GIVEN BY COOLEY, LEWIS AND WELCH (J. SOUND VIB., VOL. 12
   242) !     (1970), 315-337)
   243) !
   244) !     A IS THE ARRAY CONTAINING INPUT AND OUTPUT DATA
   245) !     WORK IS AN AREA OF SIZE (N+1)*LOT
   246) !     TRIGS IS A PREVIOUSLY PREPARED LIST OF TRIG FUNCTION VALUES
   247) !     IFAX IS A PREVIOUSLY PREPARED LIST OF FACTORS OF N/2
   248) !     INC IS THE INCREMENT WITHIN EACH DATA 'VECTOR'
   249) !         (E.G. INC=1 FOR CONSECUTIVELY STORED DATA)
   250) !     JUMP IS THE INCREMENT BETWEEN THE START OF EACH DATA VECTOR
   251) !     N IS THE LENGTH OF THE DATA VECTORS
   252) !     LOT IS THE NUMBER OF DATA VECTORS
   253) !     ISIGN = +1 FOR TRANSFORM FROM SPECTRAL TO GRIDPOINT
   254) !           = -1 FOR TRANSFORM FROM GRIDPOINT TO SPECTRAL
   255) !
   256) !     ORDERING OF COEFFICIENTS:
   257) !         A(0),B(0),A(1),B(1),A(2),B(2),...,A(N/2),B(N/2)
   258) !         WHERE B(0)=B(N/2)=0; (N+2) LOCATIONS REQUIRED
   259) !
   260) !     ORDERING OF DATA:
   261) !         X(N-1),X(0),X(1),X(2),...,X(N),X(0)
   262) !         I.E. EXPLICIT CYCLIC CONTINUITY; (N+2) LOCATIONS REQUIRED
   263) !
   264) !     VECTORIZATION IS ACHIEVED ON CRAY BY DOING THE TRANSFORMS IN
   265) !     PARALLEL
   266) !
   267) !     *** N.B. N IS ASSUMED TO BE AN EVEN NUMBER
   268) !
   269) !     DEFINITION OF TRANSFORMS:
   270) !     -------------------------
   271) !
   272) !     ISIGN=+1: X(J)=SUM(K=0,...,N-1)(C(K)*EXP(2*I*J*K*PI/N))
   273) !         WHERE C(K)=A(K)+I*B(K) AND C(N-K)=A(K)-I*B(K)
   274) !
   275) !     ISIGN=-1: A(K)=(1/N)*SUM(J=0,...,N-1)(X(J)*COS(2*J*K*PI/N))
   276) !               B(K)=-(1/N)*SUM(J=0,...,N-1)(X(J)*SIN(2*J*K*PI/N))
   277) !
   278) !-----------------------------------------------------------------------
   279) !
   280)       use shr_kind_mod,    only: r8 => shr_kind_r8
   281)       implicit none
   282) !
   283) !------------------------------Arguments--------------------------------
   284) !
   285)       integer IFAX(13), inc, jump, n, lot, isign
   286)       real(r8) A(LOT* (N+2) ), WORK(LOT*(N+1)), TRIGS(3*N/2+1)
   287) !
   288) !---------------------------Local variables-----------------------------
   289) !
   290)       integer i, j, k, l, m, ia, ib, la, ink, nh, nx, nfax
   291)       integer ibase, jbase, igo
   292) !
   293) !-----------------------------------------------------------------------
   294) !
   295)       NFAX=IFAX(1)
   296)       NX=N+1
   297)       NH=N/2
   298)       INK=INC+INC
   299)       IF (ISIGN.EQ.+1) GO TO 30
   300) !
   301) !     IF NECESSARY, TRANSFER DATA TO WORK AREA
   302)       IGO=50
   303)       IF (MOD(NFAX,2).EQ.1) GOTO 40
   304)       IBASE=INC+1
   305)       JBASE=1
   306)       DO 20 L=1,LOT
   307)       I=IBASE
   308)       J=JBASE
   309) !cdir nodep
   310) !DIR$ CONCURRENT
   311)       DO 10 M=1,N
   312)       WORK(J)=A(I)
   313)       I=I+INC
   314)       J=J+1
   315)    10 CONTINUE
   316)       IBASE=IBASE+JUMP
   317)       JBASE=JBASE+NX
   318)    20 CONTINUE
   319) !
   320)       IGO=60
   321)       GO TO 40
   322) !
   323) !     PREPROCESSING (ISIGN=+1)
   324) !     ------------------------
   325) !
   326)    30 CONTINUE
   327)       CALL FFT99A(A,WORK,TRIGS,INC,JUMP,N,LOT)
   328)       IGO=60
   329) !
   330) !     COMPLEX TRANSFORM
   331) !     -----------------
   332) !
   333)    40 CONTINUE
   334)       IA=INC+1
   335)       LA=1
   336)       DO 80 K=1,NFAX
   337)       IF (IGO.EQ.60) GO TO 60
   338)    50 CONTINUE
   339)       CALL VPASSM(A(IA),A(IA+INC),WORK(1),WORK(2),TRIGS, &
   340)          INK,2,JUMP,NX,LOT,NH,IFAX(K+1),LA)
   341)       IGO=60
   342)       GO TO 70
   343)    60 CONTINUE
   344)       CALL VPASSM(WORK(1),WORK(2),A(IA),A(IA+INC),TRIGS, &
   345)           2,INK,NX,JUMP,LOT,NH,IFAX(K+1),LA)
   346)       IGO=50
   347)    70 CONTINUE
   348)       LA=LA*IFAX(K+1)
   349)    80 CONTINUE
   350) !
   351)       IF (ISIGN.EQ.-1) GO TO 130
   352) !
   353) !     IF NECESSARY, TRANSFER DATA FROM WORK AREA
   354)       IF (MOD(NFAX,2).EQ.1) GO TO 110
   355)       IBASE=1
   356)       JBASE=IA
   357)       DO 100 L=1,LOT
   358)       I=IBASE
   359)       J=JBASE
   360) !cdir nodep
   361) !DIR$ CONCURRENT
   362)       DO 90 M=1,N
   363)       A(J)=WORK(I)
   364)       I=I+1
   365)       J=J+INC
   366)    90 CONTINUE
   367)       IBASE=IBASE+NX
   368)       JBASE=JBASE+JUMP
   369)   100 CONTINUE
   370) !
   371) !     FILL IN CYCLIC BOUNDARY POINTS
   372)   110 CONTINUE
   373)       IA=1
   374)       IB=N*INC+1
   375) !cdir nodep
   376) !DIR$ CONCURRENT
   377)       DO 120 L=1,LOT
   378)       A(IA)=A(IB)
   379)       A(IB+INC)=A(IA+INC)
   380)       IA=IA+JUMP
   381)       IB=IB+JUMP
   382)   120 CONTINUE
   383)       GO TO 140
   384) !
   385) !     POSTPROCESSING (ISIGN=-1):
   386) !     --------------------------
   387) !
   388)   130 CONTINUE
   389)       CALL FFT99B(WORK,A,TRIGS,INC,JUMP,N,LOT)
   390) !
   391)   140 CONTINUE
   392)       RETURN
   393)       END SUBROUTINE FFT99

   397)       SUBROUTINE FFT99A(A,WORK,TRIGS,INC,JUMP,N,LOT)
   398) !
   399) !-----------------------------------------------------------------------
   400) !     SUBROUTINE FFT99A - PREPROCESSING STEP FOR FFT99, ISIGN=+1
   401) !     (SPECTRAL TO GRIDPOINT TRANSFORM)
   402) !-----------------------------------------------------------------------
   403) !
   404)       use shr_kind_mod,    only: r8 => shr_kind_r8
   405)       implicit none
   406) !
   407) !------------------------------Arguments--------------------------------
   408) !
   409)       integer inc, jump, n, lot
   410)       real(r8) A(*), WORK(*), TRIGS(*)
   411) !
   412) !---------------------------Local variables-----------------------------
   413) !
   414)       integer iabase, ibbase, jabase, jbbase, ia, ib, ink
   415)       integer ja, jb, k, l, nh, nx
   416)       real(r8) c, s
   417) !
   418) !-----------------------------------------------------------------------
   419) !
   420)       NH=N/2
   421)       NX=N+1
   422)       INK=INC+INC
   423) !
   424) !     A(0) AND A(N/2)
   425)       IA=1
   426)       IB=N*INC+1
   427)       JA=1
   428)       JB=2
   429) !cdir nodep
   430) !DIR$ CONCURRENT
   431)       DO 10 L=1,LOT
   432)       WORK(JA)=A(IA)+A(IB)
   433)       WORK(JB)=A(IA)-A(IB)
   434)       IA=IA+JUMP
   435)       IB=IB+JUMP
   436)       JA=JA+NX
   437)       JB=JB+NX
   438)    10 CONTINUE
   439) !
   440) !     REMAINING WAVENUMBERS
   441)       IABASE=2*INC+1
   442)       IBBASE=(N-2)*INC+1
   443)       JABASE=3
   444)       JBBASE=N-1
   445) !
   446)       DO 30 K=3,NH,2
   447)       IA=IABASE
   448)       IB=IBBASE
   449)       JA=JABASE
   450)       JB=JBBASE
   451)       C=TRIGS(N+K)
   452)       S=TRIGS(N+K+1)
   453) !cdir nodep
   454) !DIR$ CONCURRENT
   455)       DO 20 L=1,LOT
   456)       WORK(JA)=(A(IA)+A(IB))- &
   457)           (S*(A(IA)-A(IB))+C*(A(IA+INC)+A(IB+INC)))
   458)       WORK(JB)=(A(IA)+A(IB))+ &
   459)           (S*(A(IA)-A(IB))+C*(A(IA+INC)+A(IB+INC)))
   460)       WORK(JA+1)=(C*(A(IA)-A(IB))-S*(A(IA+INC)+A(IB+INC)))+ &
   461)           (A(IA+INC)-A(IB+INC))
   462)       WORK(JB+1)=(C*(A(IA)-A(IB))-S*(A(IA+INC)+A(IB+INC)))- &
   463)           (A(IA+INC)-A(IB+INC))
   464)       IA=IA+JUMP
   465)       IB=IB+JUMP
   466)       JA=JA+NX
   467)       JB=JB+NX
   468)    20 CONTINUE
   469)       IABASE=IABASE+INK
   470)       IBBASE=IBBASE-INK
   471)       JABASE=JABASE+2
   472)       JBBASE=JBBASE-2
   473)    30 CONTINUE
   474) !
   475)       IF (IABASE.NE.IBBASE) GO TO 50
   476) !     WAVENUMBER N/4 (IF IT EXISTS)
   477)       IA=IABASE
   478)       JA=JABASE
   479) !cdir nodep
   480) !DIR$ CONCURRENT
   481)       DO 40 L=1,LOT
   482)       WORK(JA)=2.0_r8*A(IA)
   483)       WORK(JA+1)=-2.0_r8*A(IA+INC)
   484)       IA=IA+JUMP
   485)       JA=JA+NX
   486)    40 CONTINUE
   487) !
   488)    50 CONTINUE
   489)       RETURN
   490)       END SUBROUTINE FFT99A

   494)       SUBROUTINE FFT99B(WORK,A,TRIGS,INC,JUMP,N,LOT)
   495) !
   496) !-----------------------------------------------------------------------
   497) !     SUBROUTINE FFT99B - POSTPROCESSING STEP FOR FFT99, ISIGN=-1
   498) !     (GRIDPOINT TO SPECTRAL TRANSFORM)
   499) !-----------------------------------------------------------------------
   500) !
   501)       use shr_kind_mod,    only: r8 => shr_kind_r8
   502)       implicit none
   503) !
   504) !------------------------------Arguments--------------------------------
   505) !
   506)       integer inc, jump, n, lot
   507)       real(r8) WORK(*), A(*), TRIGS(*)
   508) !
   509) !---------------------------Local variables-----------------------------
   510) !
   511)       integer iabase, ibbase, jabase, jbbase, ia, ib, ink
   512)       integer ja, jb, k, l, nh, nx
   513)       real(r8) scale, s, c
   514) !
   515) !-----------------------------------------------------------------------
   516) !
   517)       NH=N/2
   518)       NX=N+1
   519)       INK=INC+INC
   520) !
   521) !     A(0) AND A(N/2)
   522)       SCALE=1.0_r8/real(N,r8)
   523)       IA=1
   524)       IB=2
   525)       JA=1
   526)       JB=N*INC+1
   527) !cdir nodep
   528) !DIR$ CONCURRENT
   529)       DO 10 L=1,LOT
   530)       A(JA)=SCALE*(WORK(IA)+WORK(IB))
   531)       A(JB)=SCALE*(WORK(IA)-WORK(IB))
   532)       A(JA+INC)=0.0_r8
   533)       A(JB+INC)=0.0_r8
   534)       IA=IA+NX
   535)       IB=IB+NX
   536)       JA=JA+JUMP
   537)       JB=JB+JUMP
   538)    10 CONTINUE
   539) !
   540) !     REMAINING WAVENUMBERS
   541)       SCALE=0.5_r8*SCALE
   542)       IABASE=3
   543)       IBBASE=N-1
   544)       JABASE=2*INC+1
   545)       JBBASE=(N-2)*INC+1
   546) !
   547)       DO 30 K=3,NH,2
   548)       IA=IABASE
   549)       IB=IBBASE
   550)       JA=JABASE
   551)       JB=JBBASE
   552)       C=TRIGS(N+K)
   553)       S=TRIGS(N+K+1)
   554) !cdir nodep
   555) !DIR$ CONCURRENT
   556)       DO 20 L=1,LOT
   557)       A(JA)=SCALE*((WORK(IA)+WORK(IB)) &
   558)          +(C*(WORK(IA+1)+WORK(IB+1))+S*(WORK(IA)-WORK(IB))))
   559)       A(JB)=SCALE*((WORK(IA)+WORK(IB)) &
   560)          -(C*(WORK(IA+1)+WORK(IB+1))+S*(WORK(IA)-WORK(IB))))
   561)       A(JA+INC)=SCALE*((C*(WORK(IA)-WORK(IB))-S*(WORK(IA+1)+WORK(IB+1))) &
   562)           +(WORK(IB+1)-WORK(IA+1)))
   563)       A(JB+INC)=SCALE*((C*(WORK(IA)-WORK(IB))-S*(WORK(IA+1)+WORK(IB+1))) &
   564)           -(WORK(IB+1)-WORK(IA+1)))
   565)       IA=IA+NX
   566)       IB=IB+NX
   567)       JA=JA+JUMP
   568)       JB=JB+JUMP
   569)    20 CONTINUE
   570)       IABASE=IABASE+2
   571)       IBBASE=IBBASE-2
   572)       JABASE=JABASE+INK
   573)       JBBASE=JBBASE-INK
   574)    30 CONTINUE
   575) !
   576)       IF (IABASE.NE.IBBASE) GO TO 50
   577) !     WAVENUMBER N/4 (IF IT EXISTS)
   578)       IA=IABASE
   579)       JA=JABASE
   580)       SCALE=2.0_r8*SCALE
   581) !cdir nodep
   582) !DIR$ CONCURRENT
   583)       DO 40 L=1,LOT
   584)       A(JA)=SCALE*WORK(IA)
   585)       A(JA+INC)=-SCALE*WORK(IA+1)
   586)       IA=IA+NX
   587)       JA=JA+JUMP
   588)    40 CONTINUE
   589) !
   590)    50 CONTINUE
   591)       RETURN
   592)       END SUBROUTINE FFT99B

   596)       SUBROUTINE FFT991(A,WORK,TRIGS,IFAX,INC,JUMP,N,LOT,ISIGN)
   597) !
   598) !-----------------------------------------------------------------------
   599) !     SUBROUTINE "FFT991" - MULTIPLE REAL/HALF-COMPLEX PERIODIC
   600) !     FAST FOURIER TRANSFORM
   601) !
   602) !     SAME AS FFT99 EXCEPT THAT ORDERING OF DATA CORRESPONDS TO
   603) !     THAT IN MRFFT2
   604) !
   605) !     PROCEDURE USED TO CONVERT TO HALF-LENGTH COMPLEX TRANSFORM
   606) !     IS GIVEN BY COOLEY, LEWIS AND WELCH (J. SOUND VIB., VOL. 12
   607) !     (1970), 315-337)
   608) !
   609) !     A IS THE ARRAY CONTAINING INPUT AND OUTPUT DATA
   610) !     WORK IS AN AREA OF SIZE (N+1)*LOT
   611) !     TRIGS IS A PREVIOUSLY PREPARED LIST OF TRIG FUNCTION VALUES
   612) !     IFAX IS A PREVIOUSLY PREPARED LIST OF FACTORS OF N/2
   613) !     INC IS THE INCREMENT WITHIN EACH DATA 'VECTOR'
   614) !         (E.G. INC=1 FOR CONSECUTIVELY STORED DATA)
   615) !     JUMP IS THE INCREMENT BETWEEN THE START OF EACH DATA VECTOR
   616) !     N IS THE LENGTH OF THE DATA VECTORS
   617) !     LOT IS THE NUMBER OF DATA VECTORS
   618) !     ISIGN = +1 FOR TRANSFORM FROM SPECTRAL TO GRIDPOINT
   619) !           = -1 FOR TRANSFORM FROM GRIDPOINT TO SPECTRAL
   620) !
   621) !     ORDERING OF COEFFICIENTS:
   622) !         A(0),B(0),A(1),B(1),A(2),B(2),...,A(N/2),B(N/2)
   623) !         WHERE B(0)=B(N/2)=0; (N+2) LOCATIONS REQUIRED
   624) !
   625) !     ORDERING OF DATA:
   626) !         X(0),X(1),X(2),...,X(N-1)
   627) !
   628) !     VECTORIZATION IS ACHIEVED ON CRAY BY DOING THE TRANSFORMS IN
   629) !     PARALLEL
   630) !
   631) !     *** N.B. N IS ASSUMED TO BE AN EVEN NUMBER
   632) !
   633) !     DEFINITION OF TRANSFORMS:
   634) !     -------------------------
   635) !
   636) !     ISIGN=+1: X(J)=SUM(K=0,...,N-1)(C(K)*EXP(2*I*J*K*PI/N))
   637) !         WHERE C(K)=A(K)+I*B(K) AND C(N-K)=A(K)-I*B(K)
   638) !
   639) !     ISIGN=-1: A(K)=(1/N)*SUM(J=0,...,N-1)(X(J)*COS(2*J*K*PI/N))
   640) !               B(K)=-(1/N)*SUM(J=0,...,N-1)(X(J)*SIN(2*J*K*PI/N))
   641) !-----------------------------------------------------------------------
   642) !
   643)       use shr_kind_mod,    only: r8 => shr_kind_r8
   644)       implicit none
   645) !
   646) !------------------------------Arguments--------------------------------
   647) !
   648)       integer IFAX(13), inc, jump, n, lot, isign
   649)       real(r8) A(*), WORK(*), TRIGS(*)
   650) !
   651) !---------------------------Local variables-----------------------------
   652) !
   653)       integer ibase, jbase, i, j, ia, ib, igo, ink, k
   654)       integer l, la, m, nh, nfax, nx
   655) !
   656) !-----------------------------------------------------------------------
   657) !
   658)       NFAX=IFAX(1)
   659)       NX=N+1
   660)       NH=N/2
   661)       INK=INC+INC
   662)       IF (ISIGN.EQ.+1) GO TO 30
   663) !
   664) !     IF NECESSARY, TRANSFER DATA TO WORK AREA
   665)       IGO=50
   666)       IF (MOD(NFAX,2).EQ.1) GOTO 40
   667)       IBASE=1
   668)       JBASE=1
   669)       DO 20 L=1,LOT
   670)       I=IBASE
   671)       J=JBASE
   672) !cdir nodep
   673) !DIR$ CONCURRENT
   674)       DO 10 M=1,N
   675)       WORK(J)=A(I)
   676)       I=I+INC
   677)       J=J+1
   678)    10 CONTINUE
   679)       IBASE=IBASE+JUMP
   680)       JBASE=JBASE+NX
   681)    20 CONTINUE
   682) !
   683)       IGO=60
   684)       GO TO 40
   685) !
   686) !     PREPROCESSING (ISIGN=+1)
   687) !     ------------------------
   688) !
   689)    30 CONTINUE
   690)       CALL FFT99A(A,WORK,TRIGS,INC,JUMP,N,LOT)
   691)       IGO=60
   692) !
   693) !     COMPLEX TRANSFORM
   694) !     -----------------
   695) !
   696)    40 CONTINUE
   697)       IA=1
   698)       LA=1
   699)       DO 80 K=1,NFAX
   700)       IF (IGO.EQ.60) GO TO 60
   701)    50 CONTINUE
   702)       CALL VPASSM(A(IA),A(IA+INC),WORK(1),WORK(2),TRIGS, &
   703)          INK,2,JUMP,NX,LOT,NH,IFAX(K+1),LA)
   704)       IGO=60
   705)       GO TO 70
   706)    60 CONTINUE
   707)       CALL VPASSM(WORK(1),WORK(2),A(IA),A(IA+INC),TRIGS, &
   708)           2,INK,NX,JUMP,LOT,NH,IFAX(K+1),LA)
   709)       IGO=50
   710)    70 CONTINUE
   711)       LA=LA*IFAX(K+1)
   712)    80 CONTINUE
   713) !
   714)       IF (ISIGN.EQ.-1) GO TO 130
   715) !
   716) !     IF NECESSARY, TRANSFER DATA FROM WORK AREA
   717)       IF (MOD(NFAX,2).EQ.1) GO TO 110
   718)       IBASE=1
   719)       JBASE=1
   720)       DO 100 L=1,LOT
   721)       I=IBASE
   722)       J=JBASE
   723) !cdir nodep
   724) !DIR$ CONCURRENT
   725)       DO 90 M=1,N
   726)       A(J)=WORK(I)
   727)       I=I+1
   728)       J=J+INC
   729)    90 CONTINUE
   730)       IBASE=IBASE+NX
   731)       JBASE=JBASE+JUMP
   732)   100 CONTINUE
   733) !
   734) !     FILL IN ZEROS AT END
   735)   110 CONTINUE
   736)       IB=N*INC+1
   737) !cdir nodep
   738) !DIR$ CONCURRENT
   739)       DO 120 L=1,LOT
   740)       A(IB)=0.0_r8
   741)       A(IB+INC)=0.0_r8
   742)       IB=IB+JUMP
   743)   120 CONTINUE
   744)       GO TO 140
   745) !
   746) !     POSTPROCESSING (ISIGN=-1):
   747) !     --------------------------
   748) !
   749)   130 CONTINUE
   750)       CALL FFT99B(WORK,A,TRIGS,INC,JUMP,N,LOT)
   751) !
   752)   140 CONTINUE
   753)       RETURN
   754)       END SUBROUTINE FFT991

   876)       SUBROUTINE FFTRIG(TRIGS,N,MODE)
   877) !
   878) !-----------------------------------------------------------------------
   879) !
   880)       use shr_kind_mod,    only: r8 => shr_kind_r8
   881)       implicit none
   882) !
   883) !------------------------------Arguments--------------------------------
   884) !
   885)       integer n, mode
   886)       real(r8) TRIGS(*)
   887) !
   888) !---------------------------Local variables-----------------------------
   889) !
   890)       integer i, l, la, nh, imode, nn
   891)       real(r8) del, pi, angle
   892) !
   893) !-----------------------------------------------------------------------
   894) !
   895)       PI=2.0_r8*ASIN(1.0_r8)
   896)       IMODE=IABS(MODE)
   897)       NN=N
   898)       IF (IMODE.GT.1.AND.IMODE.LT.6) NN=N/2
   899)       DEL=(PI+PI)/real(NN,r8)
   900)       L=NN+NN
   901)       DO 10 I=1,L,2
   902)       ANGLE=0.5_r8*real(I-1,r8)*DEL
   903)       TRIGS(I)=COS(ANGLE)
   904)       TRIGS(I+1)=SIN(ANGLE)
   905)    10 CONTINUE
   906)       IF (IMODE.EQ.1) RETURN
   907)       IF (IMODE.EQ.8) RETURN
   908)       DEL=0.5_r8*DEL
   909)       NH=(NN+1)/2
   910)       L=NH+NH
   911)       LA=NN+NN
   912)       DO 20 I=1,L,2
   913)       ANGLE=0.5_r8*real(I-1,r8)*DEL
   914)       TRIGS(LA+I)=COS(ANGLE)
   915)       TRIGS(LA+I+1)=SIN(ANGLE)
   916)    20 CONTINUE
   917)       IF (IMODE.LE.3) RETURN
   918)       DEL=0.5_r8*DEL
   919)       LA=LA+NN
   920)       IF (MODE.EQ.5) GO TO 40
   921)       DO 30 I=2,NN
   922)       ANGLE=real(I-1,r8)*DEL
   923)       TRIGS(LA+I)=2.0_r8*SIN(ANGLE)
   924)    30 CONTINUE
   925)       RETURN
   926)    40 CONTINUE
   927)       DEL=0.5_r8*DEL
   928)       DO 50 I=2,N
   929)       ANGLE=real(I-1,r8)*DEL
   930)       TRIGS(LA+I)=SIN(ANGLE)
   931)    50 CONTINUE
   932)       RETURN
   933)       END SUBROUTINE FFTRIG

   937)       SUBROUTINE VPASSM(A,B,C,D,TRIGS,INC1,INC2,INC3,INC4,LOT,N,IFAC,LA)
   938) !
   939) !-----------------------------------------------------------------------
   940) !     SUBROUTINE "VPASSM" - MULTIPLE VERSION OF "VPASSA"
   941) !     PERFORMS ONE PASS THROUGH DATA
   942) !     AS PART OF MULTIPLE COMPLEX FFT ROUTINE
   943) !     A IS FIRST REAL INPUT VECTOR
   944) !     B IS FIRST IMAGINARY INPUT VECTOR
   945) !     C IS FIRST REAL OUTPUT VECTOR
   946) !     D IS FIRST IMAGINARY OUTPUT VECTOR
   947) !     TRIGS IS PRECALCULATED TABLE OF SINES " COSINES
   948) !     INC1 IS ADDRESSING INCREMENT FOR A AND B
   949) !     INC2 IS ADDRESSING INCREMENT FOR C AND D
   950) !     INC3 IS ADDRESSING INCREMENT BETWEEN A"S & B"S
   951) !     INC4 IS ADDRESSING INCREMENT BETWEEN C"S & D"S
   952) !     LOT IS THE NUMBER OF VECTORS
   953) !     N IS LENGTH OF VECTORS
   954) !     IFAC IS CURRENT FACTOR OF N
   955) !     LA IS PRODUCT OF PREVIOUS FACTORS
   956) !-----------------------------------------------------------------------
   957) !
   958)       use shr_kind_mod,    only: r8 => shr_kind_r8
   959)       implicit none
   960) !
   961) !------------------------------Arguments--------------------------------
   962) !
   963)       integer inc1, inc2, inc3, inc4, lot, n, ifac, la
   964)       real(r8) A(*),B(*),C(*),D(*),TRIGS(*)
   965) !
   966) !---------------------------Local variables-----------------------------
   967) !
   968)       integer ie, je, ke, kd, ib, ja, ia, i, l, jb, igo, jink 
   969)       integer iink, m, jbase, ibase, jump, j, kc, jc, jd, id
   970)       integer ic, k, la1, ijk, kb
   971) 
   972)       real(r8) s3, c3, s4, c4, c2, s2, s1, c1
   973) 
   974)       real(r8) sin36, cos36, sin72, cos72, sin60
   975)       DATA SIN36/0.587785252292473_r8/,COS36/0.809016994374947_r8/, &
   976)            SIN72/0.951056516295154_r8/,COS72/0.309016994374947_r8/, &
   977)            SIN60/0.866025403784437_r8/
   978) !
   979) !-----------------------------------------------------------------------
   980) !
   981)       M=N/IFAC
   982)       IINK=M*INC1
   983)       JINK=LA*INC2
   984)       JUMP=(IFAC-1)*JINK
   985)       IBASE=0
   986)       JBASE=0
   987)       IGO=IFAC-1
   988)       IF (IGO.GT.4) RETURN
   989)       GO TO (10,50,90,130),IGO
   990) !
   991) !     CODING FOR FACTOR 2
   992) !
   993)    10 IA=1
   994)       JA=1
   995)       IB=IA+IINK
   996)       JB=JA+JINK
   997)       DO 20 L=1,LA
   998)       I=IBASE
   999)       J=JBASE
  1000) !cdir nodep
  1001) !DIR$ CONCURRENT
  1002)       DO 15 IJK=1,LOT
  1003)       C(JA+J)=A(IA+I)+A(IB+I)
  1004)       D(JA+J)=B(IA+I)+B(IB+I)
  1005)       C(JB+J)=A(IA+I)-A(IB+I)
  1006)       D(JB+J)=B(IA+I)-B(IB+I)
  1007)       I=I+INC3
  1008)       J=J+INC4
  1009)    15 CONTINUE
  1010)       IBASE=IBASE+INC1
  1011)       JBASE=JBASE+INC2
  1012)    20 CONTINUE
  1013)       IF (LA.EQ.M) RETURN
  1014)       LA1=LA+1
  1015)       JBASE=JBASE+JUMP
  1016)       DO 40 K=LA1,M,LA
  1017)       KB=K+K-2
  1018)       C1=TRIGS(KB+1)
  1019)       S1=TRIGS(KB+2)
  1020)       DO 30 L=1,LA
  1021)       I=IBASE
  1022)       J=JBASE
  1023) !cdir nodep
  1024) !DIR$ CONCURRENT
  1025)       DO 25 IJK=1,LOT
  1026)       C(JA+J)=A(IA+I)+A(IB+I)
  1027)       D(JA+J)=B(IA+I)+B(IB+I)
  1028)       C(JB+J)=C1*(A(IA+I)-A(IB+I))-S1*(B(IA+I)-B(IB+I))
  1029)       D(JB+J)=S1*(A(IA+I)-A(IB+I))+C1*(B(IA+I)-B(IB+I))
  1030)       I=I+INC3
  1031)       J=J+INC4
  1032)    25 CONTINUE
  1033)       IBASE=IBASE+INC1
  1034)       JBASE=JBASE+INC2
  1035)    30 CONTINUE
  1036)       JBASE=JBASE+JUMP
  1037)    40 CONTINUE
  1038)       RETURN
  1039) !
  1040) !     CODING FOR FACTOR 3
  1041) !
  1042)    50 IA=1
  1043)       JA=1
  1044)       IB=IA+IINK
  1045)       JB=JA+JINK
  1046)       IC=IB+IINK
  1047)       JC=JB+JINK
  1048)       DO 60 L=1,LA
  1049)       I=IBASE
  1050)       J=JBASE
  1051) !cdir nodep
  1052) !DIR$ CONCURRENT
  1053)       DO 55 IJK=1,LOT
