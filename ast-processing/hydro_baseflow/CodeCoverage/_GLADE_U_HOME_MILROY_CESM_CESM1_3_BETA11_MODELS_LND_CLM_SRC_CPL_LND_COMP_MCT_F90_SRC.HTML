    35)   subroutine lnd_init_mct( EClock, cdata_l, x2l_l, l2x_l, NLFilename )
    36)     !
    37)     ! !DESCRIPTION:
    38)     ! Initialize land surface model and obtain relevant atmospheric model arrays
    39)     ! back from (i.e. albedos, surface temperature and snow cover over land).
    40)     !
    41)     ! !USES:
    42)     use abortutils       , only : endrun
    43)     use clm_time_manager , only : get_nstep, get_step_size, set_timemgr_init, &
    44)                                   set_nextsw_cday
    45)     use clm_atmlnd       , only : clm_l2a
    46)     use clm_glclnd       , only : clm_s2x
    47)     use clm_initializeMod, only : initialize1, initialize2
    48)     use clm_varctl       , only : finidat,single_column, clm_varctl_set, iulog, noland, &
    49)                                   inst_index, inst_suffix, inst_name
    50)     use clm_varorb       , only : eccen, obliqr, lambm0, mvelpp
    51)     use controlMod       , only : control_setNL
    52)     use decompMod        , only : get_proc_bounds
    53)     use domainMod        , only : ldomain
    54)     use shr_file_mod     , only : shr_file_setLogUnit, shr_file_setLogLevel, &
    55)                                   shr_file_getLogUnit, shr_file_getLogLevel, &
    56)                                   shr_file_getUnit, shr_file_setIO
    57)     use seq_cdata_mod    , only : seq_cdata, seq_cdata_setptrs
    58)     use seq_timemgr_mod  , only : seq_timemgr_EClockGetData
    59)     use seq_infodata_mod , only : seq_infodata_type, seq_infodata_GetData, seq_infodata_PutData, &
    60)                                   seq_infodata_start_type_start, seq_infodata_start_type_cont,   &
    61)                                   seq_infodata_start_type_brnch
    62)     use seq_comm_mct     , only : seq_comm_suffix, seq_comm_inst, seq_comm_name
    63)     use seq_flds_mod     , only : seq_flds_x2l_fields, seq_flds_l2x_fields
    64)     use spmdMod          , only : masterproc, spmd_init
    65)     use clm_varctl       , only : nsrStartup, nsrContinue, nsrBranch
    66)     use clm_cpl_indices  , only : clm_cpl_indices_set
    67)     use mct_mod
    68)     use ESMF
    69)     !
    70)     ! !ARGUMENTS:
    71)     type(ESMF_Clock),           intent(inout) :: EClock           ! Input synchronization clock
    72)     type(seq_cdata),            intent(inout) :: cdata_l          ! Input land-model driver data
    73)     type(mct_aVect),            intent(inout) :: x2l_l, l2x_l     ! land model import and export states
    74)     character(len=*), optional, intent(in)    :: NLFilename       ! Namelist filename to read
    75)     !
    76)     ! !LOCAL VARIABLES:
    77)     integer                          :: LNDID	     ! Land identifyer
    78)     integer                          :: mpicom_lnd   ! MPI communicator
    79)     type(mct_gsMap),         pointer :: GSMap_lnd    ! Land model MCT GS map
    80)     type(mct_gGrid),         pointer :: dom_l        ! Land model domain
    81)     type(seq_infodata_type), pointer :: infodata     ! CESM driver level info data
    82)     integer  :: lsize                                ! size of attribute vector
    83)     integer  :: g,i,j                                ! indices
    84)     integer  :: dtime_sync                           ! coupling time-step from the input synchronization clock
    85)     integer  :: dtime_clm                            ! clm time-step
    86)     logical  :: exists                               ! true if file exists
    87)     logical  :: atm_aero                             ! Flag if aerosol data sent from atm model
    88)     real(r8) :: scmlat                               ! single-column latitude
    89)     real(r8) :: scmlon                               ! single-column longitude
    90)     real(r8) :: nextsw_cday                          ! calday from clock of next radiation computation
    91)     character(len=SHR_KIND_CL) :: caseid             ! case identifier name
    92)     character(len=SHR_KIND_CL) :: ctitle             ! case description title
    93)     character(len=SHR_KIND_CL) :: starttype          ! start-type (startup, continue, branch, hybrid)
    94)     character(len=SHR_KIND_CL) :: calendar           ! calendar type name
    95)     character(len=SHR_KIND_CL) :: hostname           ! hostname of machine running on
    96)     character(len=SHR_KIND_CL) :: version            ! Model version
    97)     character(len=SHR_KIND_CL) :: username           ! user running the model
    98)     integer :: nsrest                                ! clm restart type
    99)     integer :: ref_ymd                               ! reference date (YYYYMMDD)
   100)     integer :: ref_tod                               ! reference time of day (sec)
   101)     integer :: start_ymd                             ! start date (YYYYMMDD)
   102)     integer :: start_tod                             ! start time of day (sec)
   103)     integer :: stop_ymd                              ! stop date (YYYYMMDD)
   104)     integer :: stop_tod                              ! stop time of day (sec)
   105)     logical :: brnch_retain_casename                 ! flag if should retain the case name on a branch start type
   106)     integer :: lbnum                                 ! input to memory diagnostic
   107)     integer :: shrlogunit,shrloglev                  ! old values for log unit and log level
   108)     type(bounds_type) :: bounds                      ! bounds
   109)     character(len=32), parameter :: sub = 'lnd_init_mct'
   110)     character(len=*),  parameter :: format = "('("//trim(sub)//") :',A)"
   111)     !-----------------------------------------------------------------------
   112) 
   113)     ! Set cdata data
   114) 
   115)     call seq_cdata_setptrs(cdata_l, ID=LNDID, mpicom=mpicom_lnd, &
   116)          gsMap=GSMap_lnd, dom=dom_l, infodata=infodata)
   117) 
   118)     ! Determine attriute vector indices
   119) 
   120)     call clm_cpl_indices_set()
   121) 
   122)     ! Initialize clm MPI communicator 
   123) 
   124)     call spmd_init( mpicom_lnd, LNDID )
   125) 
   126) #if (defined _MEMTRACE)
   127)     if(masterproc) then
   128)        lbnum=1
   129)        call memmon_dump_fort('memmon.out','lnd_init_mct:start::',lbnum)
   130)     endif
   131) #endif                      
   132) 
   133)     inst_name   = seq_comm_name(LNDID)
   134)     inst_index  = seq_comm_inst(LNDID)
   135)     inst_suffix = seq_comm_suffix(LNDID)
   136) 
   137)     ! Initialize io log unit
   138) 
   139)     call shr_file_getLogUnit (shrlogunit)
   140)     if (masterproc) then
   141)        inquire(file='lnd_modelio.nml'//trim(inst_suffix),exist=exists)
   142)        if (exists) then
   143)           iulog = shr_file_getUnit()
   144)           call shr_file_setIO('lnd_modelio.nml'//trim(inst_suffix),iulog)
   145)        end if
   146)        write(iulog,format) "CLM land model initialization"
   147)     else
   148)        iulog = shrlogunit
   149)     end if
   150) 
   151)     call shr_file_getLogLevel(shrloglev)
   152)     call shr_file_setLogUnit (iulog)
   153)     
   154)     ! Use infodata to set orbital values
   155) 
   156)     call seq_infodata_GetData( infodata, orb_eccen=eccen, orb_mvelpp=mvelpp, &
   157)          orb_lambm0=lambm0, orb_obliqr=obliqr )
   158) 
   159)     ! Consistency check on namelist filename	
   160) 
   161)     call control_setNL("lnd_in"//trim(inst_suffix))
   162) 
   163)     ! Initialize clm
   164)     ! initialize1 reads namelist, grid and surface data (need this to initialize gsmap) 
   165)     ! initialize2 performs rest of initialization	
   166) 
   167)     call seq_timemgr_EClockGetData(EClock,                               &
   168)                                    start_ymd=start_ymd,                  &
   169)                                    start_tod=start_tod, ref_ymd=ref_ymd, &
   170)                                    ref_tod=ref_tod, stop_ymd=stop_ymd,   &
   171)                                    stop_tod=stop_tod,                    &
   172)                                    calendar=calendar )
   173)     call seq_infodata_GetData(infodata, case_name=caseid,    &
   174)                               case_desc=ctitle, single_column=single_column,    &
   175)                               scmlat=scmlat, scmlon=scmlon,                     &
   176)                               brnch_retain_casename=brnch_retain_casename,      &
   177)                               start_type=starttype, model_version=version,      &
   178)                               hostname=hostname, username=username )
   179)     call set_timemgr_init( calendar_in=calendar, start_ymd_in=start_ymd, start_tod_in=start_tod, &
   180)                            ref_ymd_in=ref_ymd, ref_tod_in=ref_tod, stop_ymd_in=stop_ymd,         &
   181)                            stop_tod_in=stop_tod)
   182)     if (     trim(starttype) == trim(seq_infodata_start_type_start)) then
   183)        nsrest = nsrStartup
   184)     else if (trim(starttype) == trim(seq_infodata_start_type_cont) ) then
   185)        nsrest = nsrContinue
   186)     else if (trim(starttype) == trim(seq_infodata_start_type_brnch)) then
   187)        nsrest = nsrBranch
   188)     else
   189)        call endrun( sub//' ERROR: unknown starttype' )
   190)     end if
   191) 
   192)     call clm_varctl_set(caseid_in=caseid, ctitle_in=ctitle,                     &
   193)                         brnch_retain_casename_in=brnch_retain_casename,         &
   194)                         single_column_in=single_column, scmlat_in=scmlat,       &
   195)                         scmlon_in=scmlon, nsrest_in=nsrest, version_in=version, &
   196)                         hostname_in=hostname, username_in=username)
   197) 
   198)     ! Read namelist, grid and surface data
   199) 
   200)     call initialize1( )
   201) 
   202)     ! If no land then exit out of initialization
   203) 
   204)     if ( noland ) then
   205)        call seq_infodata_PutData( infodata, lnd_present   =.false.)
   206)        call seq_infodata_PutData( infodata, lnd_prognostic=.false.)
   207)        return
   208)     end if
   209) 
   210)     ! Determine if aerosol and dust deposition come from atmosphere component
   211) 
   212)     call seq_infodata_GetData(infodata, atm_aero=atm_aero )
   213)     if ( .not. atm_aero )then
   214)        call endrun( sub//' ERROR: atmosphere model MUST send aerosols to CLM' )
   215)     end if
   216) 
   217)     ! Initialize clm gsMap, clm domain and clm attribute vectors
   218) 
   219)     call get_proc_bounds( bounds )
   220) 
   221)     call lnd_SetgsMap_mct( bounds, mpicom_lnd, LNDID, gsMap_lnd ) 	
   222)     lsize = mct_gsMap_lsize(gsMap_lnd, mpicom_lnd)
   223) 
   224)     call lnd_domain_mct( bounds, lsize, gsMap_lnd, dom_l )
   225) 
   226)     call mct_aVect_init(x2l_l, rList=seq_flds_x2l_fields, lsize=lsize)
   227)     call mct_aVect_zero(x2l_l)
   228) 
   229)     call mct_aVect_init(l2x_l, rList=seq_flds_l2x_fields, lsize=lsize)
   230)     call mct_aVect_zero(l2x_l)
   231) 
   232)     ! Finish initializing clm
   233) 
   234)     call initialize2()
   235) 
   236)     ! Check that clm internal dtime aligns with clm coupling interval
   237) 
   238)     call seq_timemgr_EClockGetData(EClock, dtime=dtime_sync )
   239)     dtime_clm = get_step_size()
   240)     if (masterproc) then
   241)        write(iulog,*)'dtime_sync= ',dtime_sync,&
   242)             ' dtime_clm= ',dtime_clm,' mod = ',mod(dtime_sync,dtime_clm)
   243)     end if
   244)     if (mod(dtime_sync,dtime_clm) /= 0) then
   245)        write(iulog,*)'clm dtime ',dtime_clm,' and Eclock dtime ',&
   246)             dtime_sync,' never align'
   247)        call endrun( sub//' ERROR: time out of sync' )
   248)     end if
   249) 
   250)     ! Create land export state 
   251) 
   252)     call lnd_export(bounds, clm_l2a, clm_s2x, l2x_l%rattr)
   253) 
   254)     ! Fill in infodata settings
   255) 
   256)     call seq_infodata_PutData(infodata, lnd_prognostic=.true.)
   257)     call seq_infodata_PutData(infodata, lnd_nx=ldomain%ni, lnd_ny=ldomain%nj)
   258) 
   259)     ! Get infodata info
   260) 
   261)     call seq_infodata_GetData(infodata, nextsw_cday=nextsw_cday )
   262)     call set_nextsw_cday(nextsw_cday)
   263) 
   264)     ! Reset shr logging to original values
   265) 
   266)     call shr_file_setLogUnit (shrlogunit)
   267)     call shr_file_setLogLevel(shrloglev)
   268) 
   269) #if (defined _MEMTRACE)
   270)     if(masterproc) then
   271)        write(iulog,*) TRIM(Sub) // ':end::'
   272)        lbnum=1
   273)        call memmon_dump_fort('memmon.out','lnd_int_mct:end::',lbnum)
   274)        call memmon_reset_addr()
   275)     endif
   276) #endif
   277) 
   278)   end subroutine lnd_init_mct
   279)

   282)   subroutine lnd_run_mct(EClock, cdata_l, x2l_l, l2x_l)
   283)     !
   284)     ! !DESCRIPTION:
   285)     ! Run clm model
   286)     !
   287)     ! !USES:
   288)     use shr_kind_mod    ,only : r8 => shr_kind_r8
   289)     use clmtype
   290)     use clm_atmlnd      ,only : clm_l2a, clm_a2l, a2l_not_downscaled_gcell
   291)     use clm_glclnd      ,only : clm_s2x, clm_x2s
   292)     use clm_driver      ,only : clm_drv
   293)     use clm_time_manager,only : get_curr_date, get_nstep, get_curr_calday, get_step_size, &
   294)                                 advance_timestep, set_nextsw_cday,update_rad_dtime
   295)     use decompMod       ,only : get_proc_bounds
   296)     use abortutils      ,only : endrun
   297)     use clm_varctl      ,only : iulog
   298)     use clm_varorb      ,only : eccen, obliqr, lambm0, mvelpp
   299)     use shr_file_mod    ,only : shr_file_setLogUnit, shr_file_setLogLevel, &
   300)                                 shr_file_getLogUnit, shr_file_getLogLevel
   301)     use seq_cdata_mod   ,only : seq_cdata, seq_cdata_setptrs
   302)     use seq_timemgr_mod ,only : seq_timemgr_EClockGetData, seq_timemgr_StopAlarmIsOn, &
   303)                                 seq_timemgr_RestartAlarmIsOn, seq_timemgr_EClockDateInSync
   304)     use seq_infodata_mod,only : seq_infodata_type, seq_infodata_GetData
   305)     use spmdMod         ,only : masterproc, mpicom
   306)     use perf_mod        ,only : t_startf, t_stopf, t_barrierf
   307)     use shr_orb_mod     ,only : shr_orb_decl
   308)     use mct_mod
   309)     use ESMF
   310)     !
   311)     ! !ARGUMENTS:
   312)     type(ESMF_Clock) , intent(inout) :: EClock    ! Input synchronization clock from driver
   313)     type(seq_cdata)  , intent(inout) :: cdata_l   ! Input driver data for land model
   314)     type(mct_aVect)  , intent(inout) :: x2l_l     ! Import state to land model
   315)     type(mct_aVect)  , intent(inout) :: l2x_l     ! Export state from land model
   316)     !
   317)     ! !LOCAL VARIABLES:
   318)     integer      :: ymd_sync             ! Sync date (YYYYMMDD)
   319)     integer      :: yr_sync              ! Sync current year
   320)     integer      :: mon_sync             ! Sync current month
   321)     integer      :: day_sync             ! Sync current day
   322)     integer      :: tod_sync             ! Sync current time of day (sec)
   323)     integer      :: ymd                  ! CLM current date (YYYYMMDD)
   324)     integer      :: yr                   ! CLM current year
   325)     integer      :: mon                  ! CLM current month
   326)     integer      :: day                  ! CLM current day
   327)     integer      :: tod                  ! CLM current time of day (sec)
   328)     integer      :: dtime                ! time step increment (sec)
   329)     integer      :: nstep                ! time step index
   330)     logical      :: rstwr_sync           ! .true. ==> write restart file before returning
   331)     logical      :: rstwr                ! .true. ==> write restart file before returning
   332)     logical      :: nlend_sync           ! Flag signaling last time-step
   333)     logical      :: nlend                ! .true. ==> last time-step
   334)     logical      :: dosend               ! true => send data back to driver
   335)     logical      :: doalb                ! .true. ==> do albedo calculation on this time step
   336)     real(r8)     :: nextsw_cday          ! calday from clock of next radiation computation
   337)     real(r8)     :: caldayp1             ! clm calday plus dtime offset
   338)     integer      :: shrlogunit,shrloglev ! old values for share log unit and log level
   339)     integer      :: lbnum                ! input to memory diagnostic
   340)     integer      :: g,i,lsize            ! counters
   341)     real(r8)     :: calday               ! calendar day for nstep
   342)     real(r8)     :: declin               ! solar declination angle in radians for nstep
   343)     real(r8)     :: declinp1             ! solar declination angle in radians for nstep+1
   344)     real(r8)     :: eccf                 ! earth orbit eccentricity factor
   345)     real(r8)     :: recip                ! reciprical
   346)     logical,save :: first_call = .true.  ! first call work
   347)     type(seq_infodata_type),pointer :: infodata             ! CESM information from the driver
   348)     type(mct_gGrid),        pointer :: dom_l                ! Land model domain data
   349)     type(bounds_type)               :: bounds               ! bounds
   350)     character(len=32)               :: rdate                ! date char string for restart file names
   351)     character(len=32), parameter    :: sub = "lnd_run_mct"
   352)     !---------------------------------------------------------------------------
   353) 
   354)     ! Determine processor bounds
   355) 
   356)     call get_proc_bounds(bounds)
   357) 
   358) #if (defined _MEMTRACE)
   359)     if(masterproc) then
   360)        lbnum=1
   361)        call memmon_dump_fort('memmon.out','lnd_run_mct:start::',lbnum)
   362)     endif
   363) #endif
   364) 
   365)     ! Reset shr logging to my log file
   366)     call shr_file_getLogUnit (shrlogunit)
   367)     call shr_file_getLogLevel(shrloglev)
   368)     call shr_file_setLogUnit (iulog)
   369) 
   370)     ! Determine time of next atmospheric shortwave calculation
   371)     call seq_cdata_setptrs(cdata_l, infodata=infodata, dom=dom_l)
   372)     call seq_timemgr_EClockGetData(EClock, &
   373)          curr_ymd=ymd, curr_tod=tod_sync,  &
   374)          curr_yr=yr_sync, curr_mon=mon_sync, curr_day=day_sync)
   375)     call seq_infodata_GetData(infodata, nextsw_cday=nextsw_cday )
   376) 
   377)     call set_nextsw_cday( nextsw_cday )
   378)     dtime = get_step_size()
   379) 
   380)     write(rdate,'(i4.4,"-",i2.2,"-",i2.2,"-",i5.5)') yr_sync,mon_sync,day_sync,tod_sync
   381)     nlend_sync = seq_timemgr_StopAlarmIsOn( EClock )
   382)     rstwr_sync = seq_timemgr_RestartAlarmIsOn( EClock )
   383) 
   384)     ! Map MCT to land data type
   385)     ! Perform downscaling if appropriate
   386) 
   387)     
   388)     ! Map to clm (only when state and/or fluxes need to be updated)
   389) 
   390)     call t_startf ('lc_lnd_import')
   391)     call lnd_import( bounds, x2l_l%rattr, clm_a2l, a2l_not_downscaled_gcell, clm_x2s)
   392)     call t_stopf ('lc_lnd_import')
   393) 
   394)     ! Use infodata to set orbital values if updated mid-run
   395) 
   396)     call seq_infodata_GetData( infodata, orb_eccen=eccen, orb_mvelpp=mvelpp, &
   397)          orb_lambm0=lambm0, orb_obliqr=obliqr )
   398) 
   399)     ! Loop over time steps in coupling interval
   400) 
   401)     dosend = .false.
   402)     do while(.not. dosend)
   403) 
   404)        ! Determine if dosend
   405)        ! When time is not updated at the beginning of the loop - then return only if
   406)        ! are in sync with clock before time is updated
   407) 
   408)        call get_curr_date( yr, mon, day, tod )
   409)        ymd = yr*10000 + mon*100 + day
   410)        tod = tod
   411)        dosend = (seq_timemgr_EClockDateInSync( EClock, ymd, tod))
   412) 
   413)        ! Determine doalb based on nextsw_cday sent from atm model
   414) 
   415)        nstep = get_nstep()
   416)        caldayp1 = get_curr_calday(offset=dtime)
   417)        if (nstep == 0) then
   418) 	  doalb = .false. 	
   419)        else if (nstep == 1) then 
   420)           doalb = (abs(nextsw_cday- caldayp1) < 1.e-10_r8) 
   421)        else
   422)           doalb = (nextsw_cday >= -0.5_r8) 
   423)        end if
   424)        call update_rad_dtime(doalb)
   425) 
   426)        ! Determine if time to write cam restart and stop
   427) 
   428)        rstwr = .false.
   429)        if (rstwr_sync .and. dosend) rstwr = .true.
   430)        nlend = .false.
   431)        if (nlend_sync .and. dosend) nlend = .true.
   432) 
   433)        ! Run clm 
   434) 
   435)        call t_barrierf('sync_clm_run1', mpicom)
   436)        call t_startf ('clm_run')
   437)        call t_startf ('shr_orb_decl')
   438)        calday = get_curr_calday()
   439)        call shr_orb_decl( calday     , eccen, mvelpp, lambm0, obliqr, declin  , eccf )
   440)        call shr_orb_decl( nextsw_cday, eccen, mvelpp, lambm0, obliqr, declinp1, eccf )
   441)        call t_stopf ('shr_orb_decl')
   442)        call clm_drv(doalb, nextsw_cday, declinp1, declin, rstwr, nlend, rdate)

   559)   subroutine lnd_domain_mct( bounds, lsize, gsMap_l, dom_l )
   560)     !
   561)     ! !DESCRIPTION:
   562)     ! Send the land model domain information to the coupler
   563)     !
   564)     ! !USES:
   565)     use clm_varcon  , only: re
   566)     use domainMod   , only: ldomain
   567)     use spmdMod     , only: iam
   568)     use mct_mod     , only: mct_gsMap, mct_gGrid, mct_gGrid_importIAttr
   569)     use mct_mod     , only: mct_gGrid_importRAttr, mct_gGrid_init, mct_gsMap_orderedPoints
   570)     use seq_flds_mod, only: seq_flds_dom_coord, seq_flds_dom_other
   571)     !
   572)     ! !ARGUMENTS: 
   573)     type(bounds_type), intent(in)  :: bounds  ! bounds
   574)     integer        , intent(in)    :: lsize   ! land model domain data size
   575)     type(mct_gsMap), intent(inout) :: gsMap_l ! Output land model MCT GS map
   576)     type(mct_ggrid), intent(out)   :: dom_l   ! Output domain information for land model
   577)     !
   578)     ! Local Variables
   579)     integer :: g,i,j              ! index
   580)     real(r8), pointer :: data(:)  ! temporary
   581)     integer , pointer :: idata(:) ! temporary
   582)     !---------------------------------------------------------------------------
   583)     !
   584)     ! Initialize mct domain type
   585)     ! lat/lon in degrees,  area in radians^2, mask is 1 (land), 0 (non-land)
   586)     ! Note that in addition land carries around landfrac for the purposes of domain checking
   587)     ! 
   588)     call mct_gGrid_init( GGrid=dom_l, CoordChars=trim(seq_flds_dom_coord), &
   589)        OtherChars=trim(seq_flds_dom_other), lsize=lsize )
   590)     !
   591)     ! Allocate memory
   592)     !
   593)     allocate(data(lsize))
   594)     !
   595)     ! Determine global gridpoint number attribute, GlobGridNum, which is set automatically by MCT
   596)     !
   597)     call mct_gsMap_orderedPoints(gsMap_l, iam, idata)
   598)     call mct_gGrid_importIAttr(dom_l,'GlobGridNum',idata,lsize)
   599)     !
   600)     ! Determine domain (numbering scheme is: West to East and South to North to South pole)
   601)     ! Initialize attribute vector with special value
   602)     !
   603)     data(:) = -9999.0_R8 
   604)     call mct_gGrid_importRAttr(dom_l,"lat"  ,data,lsize) 
   605)     call mct_gGrid_importRAttr(dom_l,"lon"  ,data,lsize) 
   606)     call mct_gGrid_importRAttr(dom_l,"area" ,data,lsize) 
   607)     call mct_gGrid_importRAttr(dom_l,"aream",data,lsize) 
   608)     data(:) = 0.0_R8     
   609)     call mct_gGrid_importRAttr(dom_l,"mask" ,data,lsize) 
   610)     !
   611)     ! Fill in correct values for domain components
   612)     ! Note aream will be filled in in the atm-lnd mapper
   613)     !
   614)     do g = bounds%begg,bounds%endg
   615)        i = 1 + (g - bounds%begg)
   616)        data(i) = ldomain%lonc(g)
   617)     end do
   618)     call mct_gGrid_importRattr(dom_l,"lon",data,lsize) 
   619) 
   620)     do g = bounds%begg,bounds%endg
   621)        i = 1 + (g - bounds%begg)
   622)        data(i) = ldomain%latc(g)
   623)     end do
   624)     call mct_gGrid_importRattr(dom_l,"lat",data,lsize) 
   625) 
   626)     do g = bounds%begg,bounds%endg
   627)        i = 1 + (g - bounds%begg)
   628)        data(i) = ldomain%area(g)/(re*re)
   629)     end do
   630)     call mct_gGrid_importRattr(dom_l,"area",data,lsize) 
   631) 
   632)     do g = bounds%begg,bounds%endg
   633)        i = 1 + (g - bounds%begg)
   634)        data(i) = real(ldomain%mask(g), r8)
   635)     end do
   636)     call mct_gGrid_importRattr(dom_l,"mask",data,lsize) 
   637) 
   638)     do g = bounds%begg,bounds%endg
   639)        i = 1 + (g - bounds%begg)
   640)        data(i) = real(ldomain%frac(g), r8)
   641)     end do
   642)     call mct_gGrid_importRattr(dom_l,"frac",data,lsize) 
   643) 
   644)     deallocate(data)
   645)     deallocate(idata)
   646) 
   647)   end subroutine lnd_domain_mct
