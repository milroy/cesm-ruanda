   123)   subroutine derivinit(deriv,fvm_corners, fvm_points, spelt_refnep)

   127)     real (kind=longdouble_kind),optional :: fvm_points(nc)
   128)     real (kind=longdouble_kind),optional :: spelt_refnep(nep)

   130)     ! Local variables
   131)     type (quadrature_t) :: gp   ! Quadrature points and weights on pressure grid
   132)     
   133)     real (kind=longdouble_kind) :: dmat(np,np)
   134)     real (kind=longdouble_kind) :: dpv(np,np)
   135)     real (kind=longdouble_kind) :: v2p(np,np)

   137)     real (kind=longdouble_kind) :: dvv(np,np)
   138)     real (kind=longdouble_kind) :: dvv_diag(np,np)
   139)     real (kind=longdouble_kind) :: v2v(np,np)
   140)     real (kind=longdouble_kind) :: xnorm
   141)     integer i,j
   142) 
   143)     ! ============================================
   144)     ! initialize matrices in longdouble_kind precision
   145)     ! and transfer results into real_kind
   146)     ! floating point precision
   147)     ! ============================================
   148) 
   149)     gp=gausslobatto(np)
   150) 
   151)     ! Legendre polynomials of degree npdg-1, on the np GLL grid:
   152)     if (npdg>np) call abortmp( 'FATAL ERROR: npdg>np')
   153)     if (npdg>0 .and. npdg<np) then
   154)        ! in this case, we use a DG basis of Legendre polynomials
   155)        ! stored at the GLL points.  
   156)        do i=1,np
   157)           deriv%legdg(1:npdg,i) = legendre(gp%points(i),npdg-1)
   158)        end do
   159)        ! normalize
   160)        do j=1,npdg
   161)           xnorm=sqrt(sum(deriv%legdg(j,:)*deriv%legdg(j,:)*gp%weights(:)))
   162)           deriv%legdg(j,:)=deriv%legdg(j,:)/xnorm
   163)        enddo
   164)     endif
   165)     call dvvinit(dvv,gp)
   166)     deriv%Dvv(:,:)   = dvv(:,:)
   167) 
   168)     do i=1,np
   169)        do j=1,np
   170)           if (i.eq.j) then
   171)              deriv%dvv_diag(i,j)   = dvv(i,j)
   172)           else
   173)              deriv%dvv_diag(i,j) = 0.0D0
   174)           endif 
   175)         end do
   176)      end do
   177) 
   178) 
   179) 
   180) 
   181) 
   182) 
   183) 
   184)     v2v = 0.0D0
   185)     do i=1,np
   186)        v2v(i,i) = gp%weights(i)
   187)     end do
   188) 
   189)     do i=1,np
   190)        do j=1,np
   191)           dvv(j,i) = dvv(j,i)*gp%weights(i)
   192)        end do
   193)     end do
   194) 
   195)     deriv%Dvv_twt = TRANSPOSE(dvv)
   196)     deriv%Mvv_twt = v2v
   197)     if (present(fvm_corners)) &
   198)          call v2pinit(deriv%Mfvm,gp%points,fvm_corners,np,nc+1)
   199) 
   200)     if (present(fvm_points)) &
   201)          call v2pinit(deriv%Cfvm,gp%points,fvm_points,np,nc)
   202)          
   203)     if (present(spelt_refnep)) &     
   204)       call v2pinit(deriv%Sfvm,gp%points,spelt_refnep,np,nep)
   205)          
   206)     ! notice we deallocate this memory here even though it was allocated 
   207)     ! by the call to gausslobatto.
   208)     deallocate(gp%points)
   209)     deallocate(gp%weights)

   239)   subroutine dmatinit(dmat)
   240) 
   241)     real (kind=longdouble_kind) :: dmat(np,np)
   242) 
   243)     ! Local variables
   244) 
   245)     type (quadrature_t) :: gll
   246)     type (quadrature_t) :: gs
   247) 
   248)     integer i,j
   249)     real(kind=longdouble_kind)  fact,f1,f2
   250)     real(kind=longdouble_kind)  func0,func1
   251)     real(kind=longdouble_kind)  dis,c0,c1
   252) 
   253)     real(kind=longdouble_kind)  :: leg(np,np)
   254)     real(kind=longdouble_kind)  ::  jac(0:np-1)
   255)     real(kind=longdouble_kind)  :: djac(0:np-1)
   256) 
   257)     c0 = 0.0_longdouble_kind
   258)     c1 = 1.0_longdouble_kind
   259) 
   260)     gll= gausslobatto(np)
   261)     gs = gauss(np)
   262) 
   263)     ! =============================================================
   264)     ! Compute Legendre polynomials on Gauss-Lobatto grid (velocity)
   265)     ! =============================================================
   266) 
   267)     do i=1,np
   268)        leg(:,i) = legendre(gll%points(i),np-1)
   269)     end do
   270) 
   271)     ! ================================================================
   272)     !  Derivatives of velocity cardinal functions on pressure grid
   273)     !  d(i,j) = D(j,i) = D' (D-transpose) since D(i,j) = dh_j(x_i)/dx
   274)     ! ================================================================
   275) 
   276)     fact = np*(np-1)
   277) 
   278)     do j=1,np
   279)        call jacobi(np-1,gs%points(j),c0,c0,jac(0:np-1),djac(0:np-1))
   280)        func0 =  jac(np-1)
   281)        func1 = djac(np-1)
   282)        f1 = fact*func0
   283)        f2 = (c1 - gs%points(j))*(c1 + gs%points(j)) * func1
   284)        do i = 1, np
   285)           if ( gs%points(j) /= gll%points(i) ) then
   286)              dis = gs%points(j) - gll%points(i)
   287)              dmat(i,j) = func0 / ( leg(np,i)*dis ) + f2 / (fact*leg(np,i)*dis*dis)
   288) !!! OTHER             dmat(i,j) = (1.0D0/(fact*leg(np,i)*dis*dis))* (func0*fact*dis + f2)
   289)           else
   290)              dmat(i,j) = c0
   291)           endif
   292)        end do
   293)     end do
   294) 
   295)     deallocate(gll%points)
   296)     deallocate(gll%weights)
   297) 
   298) 	deallocate(gs%points)
   299) 	deallocate(gs%weights)
   300) 
   301) end subroutine dmatinit

   373)     integer :: n1,n2
   374)     real(kind=real_kind)  ::  v2p(n1,n2)

   389)     real(kind=longdouble_kind)  :: leg_out(n1,n2)

   397)     ! ==============================================================

   399)     fact = -n1*(n1-1)
   400)     do i=1,n1
   401)        leg(:,i) = legendre(gll(i),n1-1)
   402)        leg(n1,i) = fact * leg(n1,i)
   403)     end do
   404) 
   405)     ! ===================================================
   406)     !  Velocity cardinal functions on pressure grid
   407)     ! ===================================================

   412)        func1 = djac(n1-1)

  1978)   function curl_sphere(s,deriv,elem) result(ds)
  1979) !
  1980) !   input s:  scalar  (assumed to be  s khat)
  1981) !   output  curl(s khat) vector in lat-lon coordinates
  1982) ! 
  1983) !   This subroutine can be used to compute divergence free velocity fields,
  1984) !   since div(ds)=0
  1985) !
  1986) !    first compute:  
  1987) !    curl(s khat) = (1/jacobian) ( ds/dy, -ds/dx ) in contra-variant coordinates
  1988) !    then map to lat-lon
  1989) !
  1990)     type (derivative_t)              :: deriv
  1991)     type (element_t)                 :: elem
  1992)     real(kind=real_kind), intent(in) :: s(np,np)
  1993) 
  1994)     real(kind=real_kind) :: ds(np,np,2)
  1995) 
  1996)     integer i
  1997)     integer j
  1998)     integer l
  1999) 
  2000)     real(kind=real_kind) ::  dsdx00
  2001)     real(kind=real_kind) ::  dsdy00
  2002)     real(kind=real_kind) ::  v1(np,np),v2(np,np)
  2003)     
  2004)     do j=1,np
  2005)        do l=1,np
  2006)           dsdx00=0.0d0
  2007)           dsdy00=0.0d0
  2008)           do i=1,np
  2009)              dsdx00 = dsdx00 + deriv%Dvv(i,l  )*s(i,j  )
  2010)              dsdy00 = dsdy00 + deriv%Dvv(i,l  )*s(j  ,i)
  2011)           end do
  2012)           v2(l  ,j  ) = -dsdx00*rrearth
  2013)           v1(j  ,l  ) =  dsdy00*rrearth
  2014)        end do
  2015)     end do
  2016)     ! convert contra -> latlon *and* divide by jacobian
  2017)     do j=1,np
  2018)        do i=1,np
  2019)           ds(i,j,1)=(elem%D(1,1,i,j)*v1(i,j) + elem%D(1,2,i,j)*v2(i,j))/elem%metdet(i,j)
  2020)           ds(i,j,2)= (elem%D(2,1,i,j)*v1(i,j) + elem%D(2,2,i,j)*v2(i,j))/elem%metdet(i,j)
  2021)        enddo
  2022)     enddo
