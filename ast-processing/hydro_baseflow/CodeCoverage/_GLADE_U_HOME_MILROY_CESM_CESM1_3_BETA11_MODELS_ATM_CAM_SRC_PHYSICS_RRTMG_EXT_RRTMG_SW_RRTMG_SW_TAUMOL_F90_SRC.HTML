    69) ! *    TAUMOL                                                                  *
    70) ! *                                                                            *
    71) ! *    This file contains the subroutines TAUGBn (where n goes from            *
    72) ! *    1 to 28).  TAUGBn calculates the optical depths and Planck fractions    *
    73) ! *    per g-value and layer for band n.                                       *
    74) ! *                                                                            *
    75) ! * Output:  optical depths (unitless)                                         *
    76) ! *          fractions needed to compute Planck functions at every layer       *
    77) ! *              and g-value                                                   *
    78) ! *                                                                            *
    79) ! *    COMMON /TAUGCOM/  TAUG(MXLAY,MG)                                        *
    80) ! *    COMMON /PLANKG/   FRACS(MXLAY,MG)                                       *
    81) ! *                                                                            *
    82) ! * Input                                                                      *
    83) ! *                                                                            *
    84) ! *    PARAMETER (MG=16, MXLAY=203, NBANDS=14)                                 *
    85) ! *                                                                            *
    86) ! *    COMMON /FEATURES/ NG(NBANDS),NSPA(NBANDS),NSPB(NBANDS)                  *
    87) ! *    COMMON /PRECISE/  ONEMINUS                                              *
    88) ! *    COMMON /PROFILE/  NLAYERS,PAVEL(MXLAY),TAVEL(MXLAY),                    *
    89) ! *   &                  PZ(0:MXLAY),TZ(0:MXLAY),TBOUND                        *
    90) ! *    COMMON /PROFDATA/ LAYTROP,LAYSWTCH,LAYLOW,                              *
    91) ! *   &                  COLH2O(MXLAY),COLCO2(MXLAY),                          *
    92) ! *   &                  COLO3(MXLAY),COLN2O(MXLAY),COLCH4(MXLAY),             *
    93) ! *   &                  COLO2(MXLAY),CO2MULT(MXLAY)                           *
    94) ! *    COMMON /INTFAC/   FAC00(MXLAY),FAC01(MXLAY),                            *
    95) ! *   &                  FAC10(MXLAY),FAC11(MXLAY)                             *
    96) ! *    COMMON /INTIND/   JP(MXLAY),JT(MXLAY),JT1(MXLAY)                        *
    97) ! *    COMMON /SELF/     SELFFAC(MXLAY), SELFFRAC(MXLAY), INDSELF(MXLAY)       *
    98) ! *                                                                            *
    99) ! *    Description:                                                            *
   100) ! *    NG(IBAND) - number of g-values in band IBAND                            *
   101) ! *    NSPA(IBAND) - for the lower atmosphere, the number of reference         *
   102) ! *                  atmospheres that are stored for band IBAND per            *
   103) ! *                  pressure level and temperature.  Each of these            *
   104) ! *                  atmospheres has different relative amounts of the         *
   105) ! *                  key species for the band (i.e. different binary           *
   106) ! *                  species parameters).                                      *
   107) ! *    NSPB(IBAND) - same for upper atmosphere                                 *
   108) ! *    ONEMINUS - since problems are caused in some cases by interpolation     *
   109) ! *               parameters equal to or greater than 1, for these cases       *
   110) ! *               these parameters are set to this value, slightly < 1.        *
   111) ! *    PAVEL - layer pressures (mb)                                            *
   112) ! *    TAVEL - layer temperatures (degrees K)                                  *
   113) ! *    PZ - level pressures (mb)                                               *
   114) ! *    TZ - level temperatures (degrees K)                                     *
   115) ! *    LAYTROP - layer at which switch is made from one combination of         *
   116) ! *              key species to another                                        *
   117) ! *    COLH2O, COLCO2, COLO3, COLN2O, COLCH4 - column amounts of water         *
   118) ! *              vapor,carbon dioxide, ozone, nitrous ozide, methane,          *
   119) ! *              respectively (molecules/cm**2)                                *
   120) ! *    CO2MULT - for bands in which carbon dioxide is implemented as a         *
   121) ! *              trace species, this is the factor used to multiply the        *
   122) ! *              band's average CO2 absorption coefficient to get the added    *
   123) ! *              contribution to the optical depth relative to 355 ppm.        *
   124) ! *    FACij(LAY) - for layer LAY, these are factors that are needed to        *
   125) ! *                 compute the interpolation factors that multiply the        *
   126) ! *                 appropriate reference k-values.  A value of 0 (1) for      *
   127) ! *                 i,j indicates that the corresponding factor multiplies     *
   128) ! *                 reference k-value for the lower (higher) of the two        *
   129) ! *                 appropriate temperatures, and altitudes, respectively.     *
   130) ! *    JP - the index of the lower (in altitude) of the two appropriate        *
   131) ! *         reference pressure levels needed for interpolation                 *
   132) ! *    JT, JT1 - the indices of the lower of the two appropriate reference     *
   133) ! *              temperatures needed for interpolation (for pressure           *
   134) ! *              levels JP and JP+1, respectively)                             *
   135) ! *    SELFFAC - scale factor needed to water vapor self-continuum, equals     *
   136) ! *              (water vapor density)/(atmospheric density at 296K and        *
   137) ! *              1013 mb)                                                      *
   138) ! *    SELFFRAC - factor needed for temperature interpolation of reference     *
   139) ! *               water vapor self-continuum data                              *
   140) ! *    INDSELF - index of the lower of the two appropriate reference           *
   141) ! *              temperatures needed for the self-continuum interpolation      *
   142) ! *                                                                            *
   143) ! * Data input                                                                 *
   144) ! *    COMMON /Kn/ KA(NSPA(n),5,13,MG), KB(NSPB(n),5,13:59,MG), SELFREF(10,MG) *
   145) ! *       (note:  n is the band number)                                        *
   146) ! *                                                                            *
   147) ! *    Description:                                                            *
   148) ! *    KA - k-values for low reference atmospheres (no water vapor             *

   150) ! *    KB - k-values for high reference atmospheres (all sources)              *
   151) ! *         (units: cm**2/molecule)                                            *
   152) ! *    SELFREF - k-values for water vapor self-continuum for reference         *

   245)       subroutine taumol16
   246) !----------------------------------------------------------------------------
   247) !
   248) !     band 16:  2600-3250 cm-1 (low - h2o,ch4; high - ch4)
   249) !
   250) !----------------------------------------------------------------------------
   251) 
   252) ! ------- Modules -------
   253) 
   254)       use parrrsw, only : ng16
   255)       use rrsw_kg16, only : absa, absb, forref, selfref, &
   256)                             sfluxref, rayl, layreffr, strrat1
   257) 
   258) ! ------- Declarations -------
   259) 
   260) ! Local
   261) 
   262)       integer :: ig, ind0, ind1, inds, indf, js, lay, laysolfr
   263)       real(kind=r8) :: fac000, fac001, fac010, fac011, fac100, fac101, &
   264)                          fac110, fac111, fs, speccomb, specmult, specparm, &
   265)                          tauray
   266) 
   267) ! Compute the optical depth by interpolating in ln(pressure), 
   268) ! temperature, and appropriate species.  Below LAYTROP, the water
   269) ! vapor self-continuum is interpolated (in temperature) separately.  
   270) 
   271) ! Lower atmosphere loop
   272)       do lay = 1, laytrop
   273)          speccomb = colh2o(lay) + strrat1*colch4(lay)
   274)          specparm = colh2o(lay)/speccomb 
   275)          if (specparm .ge. oneminus) specparm = oneminus
   276)          specmult = 8._r8*(specparm)
   277)          js = 1 + int(specmult)
   278)          fs = mod(specmult, 1._r8 )
   279)          fac000 = (1._r8 - fs) * fac00(lay)
   280)          fac010 = (1._r8 - fs) * fac10(lay)
   281)          fac100 = fs * fac00(lay)
   282)          fac110 = fs * fac10(lay)
   283)          fac001 = (1._r8 - fs) * fac01(lay)
   284)          fac011 = (1._r8 - fs) * fac11(lay)
   285)          fac101 = fs * fac01(lay)
   286)          fac111 = fs * fac11(lay)
   287)          ind0 = ((jp(lay)-1)*5+(jt(lay)-1))*nspa(16) + js
   288)          ind1 = (jp(lay)*5+(jt1(lay)-1))*nspa(16) + js
   289)          inds = indself(lay)
   290)          indf = indfor(lay)
   291)          tauray = colmol(lay) * rayl
   292) 
   293)          do ig = 1, ng16
   294)             taug(lay,ig) = speccomb * &
   295)                 (fac000 * absa(ind0   ,ig) + &
   296)                  fac100 * absa(ind0 +1,ig) + &
   297)                  fac010 * absa(ind0 +9,ig) + &
   298)                  fac110 * absa(ind0+10,ig) + &
   299)                  fac001 * absa(ind1   ,ig) + &
   300)                  fac101 * absa(ind1 +1,ig) + &
   301)                  fac011 * absa(ind1 +9,ig) + &
   302)                  fac111 * absa(ind1+10,ig)) + &
   303)                  colh2o(lay) * &
   304)                  (selffac(lay) * (selfref(inds,ig) + &
   305)                  selffrac(lay) * &
   306)                  (selfref(inds+1,ig) - selfref(inds,ig))) + &
   307)                  forfac(lay) * (forref(indf,ig) + &
   308)                  forfrac(lay) * &
   309)                  (forref(indf+1,ig) - forref(indf,ig)))) 
   310) !            ssa(lay,ig) = tauray/taug(lay,ig)
   311)             taur(lay,ig) = tauray
   312)          enddo
   313)       enddo
   314) 
   315)       laysolfr = nlayers
   316) 
   317) ! Upper atmosphere loop
   318)       do lay = laytrop+1, nlayers
   319)          if (jp(lay-1) .lt. layreffr .and. jp(lay) .ge. layreffr) &
   320)             laysolfr = lay
   321)          ind0 = ((jp(lay)-13)*5+(jt(lay)-1))*nspb(16) + 1
   322)          ind1 = ((jp(lay)-12)*5+(jt1(lay)-1))*nspb(16) + 1
   323)          tauray = colmol(lay) * rayl
   324) 
   325)          do ig = 1, ng16
   326)             taug(lay,ig) = colch4(lay) * &
   327)                 (fac00(lay) * absb(ind0  ,ig) + &
   328)                  fac10(lay) * absb(ind0+1,ig) + &
   329)                  fac01(lay) * absb(ind1  ,ig) + &
   330)                  fac11(lay) * absb(ind1+1,ig)) 
   331) !            ssa(lay,ig) = tauray/taug(lay,ig)
   332)             if (lay .eq. laysolfr) sfluxzen(ig) = sfluxref(ig) 
   333)             taur(lay,ig) = tauray  
   334)          enddo
   335)       enddo
   336) 
   337)       end subroutine taumol16

   354) 
   355) ! Local
   356) 
   357)       integer :: ig, ind0, ind1, inds, indf, js, lay, laysolfr
   358)       real(kind=r8) :: fac000, fac001, fac010, fac011, fac100, fac101, &
   359)                          fac110, fac111, fs, speccomb, specmult, specparm, &

   390)                 (fac000 * absa(ind0,ig) + &
   391)                  fac100 * absa(ind0+1,ig) + &
   392)                  fac010 * absa(ind0+9,ig) + &
   393)                  fac110 * absa(ind0+10,ig) + &
   394)                  fac001 * absa(ind1,ig) + &
   395)                  fac101 * absa(ind1+1,ig) + &
   396)                  fac011 * absa(ind1+9,ig) + &
   397)                  fac111 * absa(ind1+10,ig)) + &
   398)                  colh2o(lay) * &
   399)                  (selffac(lay) * (selfref(inds,ig) + &
   400)                  selffrac(lay) * &
   401)                  (selfref(inds+1,ig) - selfref(inds,ig))) + &
   402)                  forfac(lay) * (forref(indf,ig) + &
   403)                  forfrac(lay) * &
   404)                  (forref(indf+1,ig) - forref(indf,ig)))) 
   405) !             ssa(lay,ngs16+ig) = tauray/taug(lay,ngs16+ig)
   406)             taur(lay,ngs16+ig) = tauray
   407)          enddo
   408)       enddo
   409) 
   410)       laysolfr = nlayers
   411) 
   412) ! Upper atmosphere loop
   413)       do lay = laytrop+1, nlayers
   414)          if (jp(lay-1) .lt. layreffr .and. jp(lay) .ge. layreffr) &
   415)             laysolfr = lay
   416)          speccomb = colh2o(lay) + strrat*colco2(lay)
   417)          specparm = colh2o(lay)/speccomb 
   418)          if (specparm .ge. oneminus) specparm = oneminus
   419)          specmult = 4._r8*(specparm)
   420)          js = 1 + int(specmult)
   421)          fs = mod(specmult, 1._r8 )
   422)          fac000 = (1._r8 - fs) * fac00(lay)
   423)          fac010 = (1._r8 - fs) * fac10(lay)
   424)          fac100 = fs * fac00(lay)
   425)          fac110 = fs * fac10(lay)
   426)          fac001 = (1._r8 - fs) * fac01(lay)
   427)          fac011 = (1._r8 - fs) * fac11(lay)
   428)          fac101 = fs * fac01(lay)
   429)          fac111 = fs * fac11(lay)
   430)          ind0 = ((jp(lay)-13)*5+(jt(lay)-1))*nspb(17) + js
   431)          ind1 = ((jp(lay)-12)*5+(jt1(lay)-1))*nspb(17) + js
   432)          indf = indfor(lay)
   433)          tauray = colmol(lay) * rayl
   434) 
   435)          do ig = 1, ng17
   436)             taug(lay,ngs16+ig) = speccomb * &
   437)                 (fac000 * absb(ind0,ig) + &
   438)                  fac100 * absb(ind0+1,ig) + &
   439)                  fac010 * absb(ind0+5,ig) + &
   440)                  fac110 * absb(ind0+6,ig) + &
   441)                  fac001 * absb(ind1,ig) + &
   442)                  fac101 * absb(ind1+1,ig) + &
   443)                  fac011 * absb(ind1+5,ig) + &
   444)                  fac111 * absb(ind1+6,ig)) + &
   445)                  colh2o(lay) * &
   446)                  forfac(lay) * (forref(indf,ig) + &
   447)                  forfrac(lay) * &
   448)                  (forref(indf+1,ig) - forref(indf,ig)))
