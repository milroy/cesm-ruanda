    98)     ! Energy equation:
    99)     ! KE_t  = T1 + T2  + D1   + Err   +  vertical & horizontal advection terms
   100)     ! IE_t  = S1 + D2                 +  vertical & horizontal advection terms
   101)     ! PE_t  = S2        
   102)     !
   103)     ! KEvert*  =  KE net vertical advection    (should be zero)
   104)     ! KEhoriz* =  KE net horizonatl advection  (should be zero)
   105)     ! IEvert*  =  IE net vertical advection    (should be zero)
   106)     ! IEhoriz* =  IE net horizonatl advection  (should be zero)
   107)     !
   108)     ! With leapfrog, energy equations are all exact except KE

   365)   function element_coordinates(start,end,points) result(cart)
   366) 
   367)     ! Initialize 2D rectilinear element colocation points
   368) 
   369)     use kinds, only : longdouble_kind
   370)     type (cartesian2D_t), intent(in) :: start
   371)     type (cartesian2D_t), intent(in) :: end
   372)     real (kind=longdouble_kind), intent(in) :: points(:)
   373) 
   374)     type (cartesian2D_t) :: cart(SIZE(points),SIZE(points))
   375)     type (cartesian2D_t) :: length, centroid
   376)     real (kind=longdouble_kind) :: y 
   377)     integer i,j
   378) 
   379)     length%x   = 0.50D0*(end%x-start%x)
   380)     length%y   = 0.50D0*(end%y-start%y)
   381)     centroid%x = 0.50D0*(end%x+start%x) 
   382)     centroid%y = 0.50D0*(end%y+start%y) 
   383)     do j=1,SIZE(points)
   384)        y = centroid%y + length%y*points(j)
   385)        do i=1,SIZE(points)
   386)           cart(i,j)%x = centroid%x + length%x*points(i)
   387)           cart(i,j)%y = y
   388)        end do
   389)     end do
   390)   end function element_coordinates
