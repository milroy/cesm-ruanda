    92)  function create_distribution(dist_type, nprocs, minBlock, maxBlock, work_per_block, prob_per_block, blockType, bStats, &
    93)       FixMaxBlock, maxDil)
    94) 
    95) ! !DESCRIPTION:
    96) !  This routine determines the distribution of blocks across processors
    97) !  by call the appropriate subroutine based on distribution type
    98) !  requested.  Currently three distributions are supported:
    99) !  2-d Cartesian distribution (cartesian), a load-balanced
   100) !  distribution using a rake algorithm based on an input amount of work 
   101) !  per block, and a space-filling-curve algorithm.
   102) !
   103) ! !REVISION HISTORY:
   104) !  same as module
   105) 
   106) ! !INPUT PARAMETERS:
   107) 
   108)    character (*), intent(in) :: &
   109)       dist_type             ! method for distributing blocks
   110)                             !  either cartesian or rake
   111) 
   112)    integer (int_kind), intent(in) :: &
   113)       nprocs                ! number of processors in this distribution
   114) 
   115)    integer (int_kind), intent(in) :: minBlock ! minimum number of blocks to use
   116)    integer (int_kind), intent(in) :: maxBlock ! maximum number of blocks to use
   117) 
   118)    integer (int_kind), dimension(:), intent(in) :: &
   119)       work_per_block        ! amount of work per block
   120) 
   121)    real (dbl_kind), dimension(:), intent(in) :: &
   122)       prob_per_block        ! probability that block contains sea-ice
   123) 
   124)    integer (int_kind), dimension(:), intent(in) :: &
   125)       blockType             ! type of block
   126) 
   127)    real (dbl_kind), dimension(:,:), intent(in)  :: bStats  ! block statistics 
   128) 
   129)    logical, intent(in) :: FixMaxBlock
   130)    real (real_kind), intent(in) :: maxDil
   131) 
   132) ! !OUTPUT PARAMETERS:
   133) 
   134)    type (distrb) :: &
   135)       create_distribution   ! resulting structure describing
   136)                             !  distribution of blocks
   137) 
   138) !EOP
   139) 
   140) !BOC
   141) !-----------------------------------------------------------------------
   142) !
   143) !  local variables
   144) !
   145) !-----------------------------------------------------------------------
   146) 
   147)    integer (int_kind) :: &
   148)       n, nb, nc        ! dummy counters
   149) 
   150) !----------------------------------------------------------------------
   151) !
   152) !  select the appropriate distribution type
   153) !
   154) !----------------------------------------------------------------------
   155) 
   156)    select case (trim(dist_type))
   157) 
   158)    case('roundrobin')
   159) 
   160)       create_distribution = create_distrb_roundrobin(nprocs, work_per_block)
   161) 
   162)    case('blkrobin')
   163) 
   164)       create_distribution = create_distrb_blkrobin(nprocs, work_per_block)
   165) 
   166)    case('blkcart')
   167) 
   168)       create_distribution = create_distrb_blkcart(nprocs, work_per_block)
   169) 
   170)    case('cartesian')
   171) 
   172)       create_distribution = create_distrb_cart(nprocs, work_per_block)
   173) 
   174)    case('rake')
   175) 
   176)       create_distribution = create_distrb_rake(nprocs, work_per_block)
   177) 
   178)    case('spacecurve')
   179) 
   180) !DBG      write(nu_diag,*) 'before call to create_distrb_spacecurve'
   181)       create_distribution = create_distrb_spacecurve(nprocs, minBlock, maxBlock, &
   182)                             work_per_block,prob_per_block,blockType,bStats, FixMaxBlock, maxDil )
   183) !DBG    stop 'create_distribution: after call to create_distrb_spacecurve'
   184) 
   185)    case default
   186) 
   187)       call abort_ice('ice distribution: unknown distribution type')
   188) 
   189)    end select
   190) 
   191)    if(my_task == master_task) then 
   192)       write(nu_diag,*) ' '
   193)       nc = 0
   194)       do n = 1,nprocs
   195)          nb = create_distribution%blockcnt(n)
   196)          if (nb > 0) then
   197)             nc = nc + 1
   198) !            write(nu_diag,*) ' Blocks on proc : ',n,nb,':', &
   199) !               create_distribution%blockindex(n,1:nb)
   200)          else
   201) !            write(nu_diag,*) ' Blocks on proc : ',n,nb
   202)          endif
   203)       enddo
   204)       write(nu_diag,*) ' Active processors: ',nc,MAXVAL(create_distribution%blockLocation)
   205)       write(nu_diag,*) ' '
   206)    endif
   207) !DBG write(nu_diag,*) 'end of create_distribution'
   208) !-----------------------------------------------------------------------
   209) !EOC
   210) 
   211)  end function create_distribution

   218)  subroutine create_local_block_ids(block_ids, distribution)
   219) 
   220) ! !DESCRIPTION:
   221) !  This routine determines which blocks in an input distribution are
   222) !  located on the local processor and creates an array of block ids
   223) !  for all local blocks.
   224) !
   225) ! !REVISION HISTORY:
   226) !  same as module
   227) 
   228) ! !INPUT PARAMETERS:
   229) 
   230)    type (distrb), intent(in) :: &
   231)       distribution           ! input distribution for which local
   232)                              !  blocks required
   233) 
   234) ! !OUTPUT PARAMETERS:
   235) 
   236)    integer (int_kind), dimension(:), pointer :: &
   237)       block_ids              ! array of block ids for every block
   238)                              ! that resides on the local processor
   239) !EOP
   240) !BOC
   241) !-----------------------------------------------------------------------
   242) !
   243) !  local variables
   244) !
   245) !-----------------------------------------------------------------------
   246) 
   247)    integer (int_kind) :: &
   248)       n, bid, bcount        ! dummy counters
   249) 
   250)    logical (log_kind) :: dbug
   251) 
   252) !-----------------------------------------------------------------------
   253) !
   254) !  first determine number of local blocks to allocate array
   255) !
   256) !-----------------------------------------------------------------------
   257) 
   258)    bcount = 0
   259)    do n=1,size(distribution%blockLocation)
   260)       if (distribution%blockLocation(n) == my_task+1) bcount = bcount + 1
   261)    end do
   262) 
   263) 
   264)    if (bcount > 0) allocate(block_ids(bcount))
   265) 
   266) !-----------------------------------------------------------------------
   267) !
   268) !  now fill array with proper block ids
   269) !
   270) !-----------------------------------------------------------------------
   271) 
   272) !   dbug = .true.
   273)    dbug = .false.
   274)    if (bcount > 0) then
   275)       do n=1,size(distribution%blockLocation)
   276)          if (distribution%blockLocation(n) == my_task+1) then
   277)             block_ids(distribution%blockLocalID(n)) = n
   278) 
   279)             if (dbug) then
   280)             write(nu_diag,*) 'block id, proc, local_block: ', &
   281)                              block_ids(distribution%blockLocalID(n)), &
   282)                              distribution%blockLocation(n), &
   283)                              distribution%blockLocalID(n)
   284)             endif
   285)          endif
   286)       end do
   287)    endif
   288) 
   289) !EOC
   290) 
   291)  end subroutine create_local_block_ids

   298)  subroutine proc_decomposition(nprocs, nprocs_x, nprocs_y)
   299) 
   300) ! !DESCRIPTION:
   301) !  This subroutine attempts to find an optimal (nearly square)
   302) !  2d processor decomposition for a given number of processors.
   303) !
   304) ! !REVISION HISTORY:
   305) !  same as module
   306) !
   307) ! !USES:
   308) 
   309)    use ice_domain_size
   310) 
   311) ! !INPUT PARAMETERS:
   312) 
   313)    integer (int_kind), intent(in) :: &
   314)       nprocs                       ! total number or processors
   315) 
   316) ! !OUTPUT PARAMETERS:
   317) 
   318)    integer (int_kind), intent(out) :: &
   319)       nprocs_x, nprocs_y           ! number of procs in each dimension
   320) 
   321) !EOP
   322) !BOC
   323) !----------------------------------------------------------------------
   324) !
   325) !  local variables
   326) !
   327) !----------------------------------------------------------------------
   328) 
   329)    integer (int_kind) :: &
   330)       iguess, jguess               ! guesses for nproc_x,y
   331) 
   332)    real (real_kind) :: &
   333)       square                       ! square root of nprocs
   334) 
   335) !----------------------------------------------------------------------
   336) !
   337) !  start with an initial guess
   338) !
   339) !----------------------------------------------------------------------
   340) 
   341)    square = sqrt(real(nprocs,kind=real_kind))
   342)    nprocs_x = 0
   343)    nprocs_y = 0
   344) 
   345)    if (processor_shape == 'square-pop') then ! make as square as possible
   346)       iguess = nint(square)
   347)       jguess = nprocs/iguess
   348)    elseif (processor_shape == 'square-ice') then ! better for bipolar ice
   349)       jguess = nint(square)
   350)       iguess = nprocs/jguess
   351)    elseif (processor_shape == 'slenderX1') then ! 1 proc in y direction
   352)       jguess = 1
   353)       iguess = nprocs/jguess
   354)    elseif (processor_shape == 'slenderX2') then ! 2 proc in y direction
   355)       jguess = min(2, nprocs)
   356)       iguess = nprocs/jguess
   357)    else                                  ! abort
   358)       call abort_ice('ice: processor_shape not supported, '//trim(processor_shape))
   359)    endif
   360) 
   361) !----------------------------------------------------------------------
   362) !
   363) !  try various decompositions to find the best
   364) !
   365) !----------------------------------------------------------------------
   366) 
   367)    proc_loop: do
   368)    if (processor_shape == 'square-pop') then
   369)       jguess = nprocs/iguess
   370)    else
   371)       iguess = nprocs/jguess
   372)    endif
   373) 
   374)       if (iguess*jguess == nprocs) then ! valid decomp
   375) 
   376)          !*** if the blocks can be evenly distributed, it is a
   377)          !*** good decomposition
   378)          if (mod(nblocks_x,iguess) == 0 .and. &
   379)              mod(nblocks_y,jguess) == 0) then
   380)             nprocs_x = iguess
   381)             nprocs_y = jguess
   382)             exit proc_loop
   383) 
   384)          !*** if the blocks can be evenly distributed in a
   385)          !*** transposed direction, it is a good decomposition
   386)          else if (mod(nblocks_x,jguess) == 0 .and. &
   387)                 mod(nblocks_y,iguess) == 0) then
   388)             nprocs_x = jguess
   389)             nprocs_y = iguess
   390)             exit proc_loop
   391) 
   392)          !*** A valid decomposition, but keep searching for
   393)          !***  a better one
   394)          else
   395)             if (nprocs_x == 0) then
   396)                nprocs_x = iguess
   397)                nprocs_y = jguess
   398)             endif
   399)             if (processor_shape == 'square-pop') then
   400)                iguess = iguess - 1
   401)                if (iguess == 0) then
   402)                   exit proc_loop
   403)                else
   404)                   cycle proc_loop
   405)                endif
   406)             else
   407)                jguess = jguess - 1
   408)                if (jguess == 0) then
   409)                   exit proc_loop
   410)                else
   411)                   cycle proc_loop
   412)                endif
   413)             endif
   414)          endif
   415) 
   416)       else ! invalid decomp - keep trying
   417) 
   418)          if (processor_shape == 'square-pop') then
   419)             iguess = iguess - 1
   420)             if (iguess == 0) then
   421)                exit proc_loop
   422)             else
   423)                cycle proc_loop
   424)             endif
   425)          else
   426)             jguess = jguess - 1
   427)             if (jguess == 0) then
   428)                exit proc_loop
   429)             else
   430)                cycle proc_loop
   431)             endif
   432)          endif
   433)       endif
   434) 
   435)    end do proc_loop
   436) 
   437)    if (nprocs_x == 0) then
   438)       call abort_ice('ice: Unable to find 2d processor config')
   439)    endif
   440) 
   441)    if (my_task == master_task) then
   442)      write(nu_diag,'(a23,i4,a3,i4)') '  Processors (X x Y) = ', &
   443)                                         nprocs_x,' x ',nprocs_y
   444)    endif
   445) 
   446) !----------------------------------------------------------------------
   447) !EOC
   448) 
   449)  end subroutine proc_decomposition

   602) !  the processor and local index for the block.  A zero for both
   603) !  is returned in the case that the block has been eliminated from
   604) !  the distribution (i.e. has no active points).

  1516)  function create_distrb_rake(nprocs, workPerBlock) result(newDistrb)
  1517) 
  1518) ! !DESCRIPTION:

  1520) !  load-balanced manner based on the amount of work per block.
  1521) !  A rake algorithm is used in which the blocks are first distributed
  1522) !  in a Cartesian distribution and then a rake is applied in each

  1537)

  1540)                           ! load-balanced distribution of blocks
  1541)

  1545) !

  1547) !
  1548) !----------------------------------------------------------------------
  1549) 
  1550)    integer (int_kind) ::    &
  1551)       i,j,n              ,&! dummy loop indices

  1553)       istat              ,&! status flag for allocates

  1558)       nprocsY            ! num of procs in y for global domain

  1574)

  1578) !  only do a 1-d rake on the entire distribution

  1596)             priority(n) = 0
  1597)          endif
  1598)       end do
  1599)       end do
  1600) 
  1601)       allocate(workTmp(nblocks_tot), procTmp(nblocks_tot), stat=istat)
  1602)

  1605)          procTmp(i) = i
  1606)          do n=1,nblocks_tot
  1607)             if (dist%blockLocation(n) == i) then
  1608)                workTmp(i) = workTmp(i) + workPerBlock(n)
  1609)             endif
  1610)          end do

  1614)                                  priority, dist)

  1622) !----------------------------------------------------------------------

  1685)       do i=1,nblocks_x

  1688)             priority(n) = (maxWork + 1)*(nblocks_y + j) - &
  1689)                           workPerBlock(n)
  1690)          else
  1691)             priority(n) = 0
  1692)          endif
  1693)       end do
  1694)       end do
  1695) 
  1696)       allocate(workTmp(nprocsY), procTmp(nprocsY), stat=istat)
  1697) 
  1698)       do i=1,nprocsX
  1699) 
  1700)          workTmp(:) = 0
  1701)          do j=1,nprocsY
  1702)             pid = (j-1)*nprocsX + i
  1703)             procTmp(j) = pid
  1704)             do n=1,nblocks_tot
  1705)                if (dist%blockLocation(n) == pid) then
  1706)                   workTmp(j) = workTmp(j) + workPerBlock(n)
  1707)                endif
  1708)             end do
  1709)          end do
  1710) 
  1711)          call ice_distributionRake (workTmp, procTmp, workPerBlock, &
  1712)                                     priority, dist)
  1713) 
  1714)       end do
  1715) 
  1716)       deallocate(workTmp, procTmp, priority, stat=istat)
  1717) 
  1718)    endif  ! 1d or 2d rake
  1719)

  1721) !
  1722) !  create new distribution with info extracted from the temporary
  1723) !  distribution
  1724) !
  1725) !----------------------------------------------------------------------
  1726) 
  1727)    newDistrb%nprocs     = nprocs

  1731)             newDistrb%blockLocalID(nblocks_tot), stat=istat)

  1737)    newDistrb%blockIndex(:,:) = 0
  1738) 
  1739)    procTmp = 0
  1740)    do n=1,nblocks_tot
  1741)       pid = dist%blockLocation(n)  ! processor id

  1749)          endif
  1750)       else
  1751)          newDistrb%blockLocalID (n) = 0
