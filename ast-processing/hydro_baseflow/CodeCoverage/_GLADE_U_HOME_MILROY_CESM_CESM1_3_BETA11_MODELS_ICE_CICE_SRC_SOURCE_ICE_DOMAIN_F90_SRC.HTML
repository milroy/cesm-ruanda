   112)  subroutine init_domain_blocks
   113) 
   114) ! !DESCRIPTION:
   115) !  This routine reads in domain information and calls the routine
   116) !  to set up the block decomposition.
   117) !
   118) ! !REVISION HISTORY:
   119) !  same as module
   120) 
   121) ! !USES:
   122) !
   123)    use ice_global_reductions
   124) !
   125) !EOP
   126) !BOC
   127) !----------------------------------------------------------------------
   128) !
   129) !  local variables
   130) !
   131) !----------------------------------------------------------------------
   132) 
   133)    integer (int_kind) :: &
   134)       nml_error          ! namelist read error flag
   135) 
   136)    integer :: omp_get_num_threads
   137) !----------------------------------------------------------------------
   138) !
   139) !  input namelists
   140) !
   141) !----------------------------------------------------------------------
   142) 
   143)    namelist /domain_nml/ nprocs, &
   144)                          processor_shape,   &
   145)                          distribution_type, &
   146)                          distribution_wght, &
   147)    			 distribution_wght_file, &
   148)                          ew_boundary_type,  &
   149)                          ns_boundary_type,   &
   150) 			 minBlock,           &
   151)                          maxBlock,           &
   152)                          FixMaxBlock,        &
   153)    			 maxDil,             &
   154)                          profile_barrier
   155) 
   156) !----------------------------------------------------------------------
   157) !
   158) !  read domain information from namelist input
   159) !
   160) !----------------------------------------------------------------------
   161) 
   162)    nprocs = -1
   163)    processor_shape   = 'slenderX2'
   164)    distribution_type = 'cartesian'
   165)    distribution_wght = 'latitude'
   166)    distribution_wght_file  = 'unknown_distribution_wght_file'
   167)    ew_boundary_type  = 'cyclic'
   168)    ns_boundary_type  = 'open'
   169)    maxBlock          = max_blocks
   170) #if defined(THREADED_OMP)
   171) !$OMP PARALLEL
   172)    minBlock          = omp_get_num_threads()
   173) !$OMP END PARALLEL
   174) #else
   175)    minBlock          = 1
   176) #endif
   177)    maxDil            = 2.0
   178)    profile_barrier   = .false.
   179)    FixMaxBlock       = .false.
   180) 
   181)    call get_fileunit(nu_nml)
   182)    if (my_task == master_task) then
   183)       open (nu_nml, file=nml_filename, status='old',iostat=nml_error)
   184)       if (nml_error /= 0) then
   185)          nml_error = -1
   186)       else
   187)          nml_error =  1
   188)       endif
   189)       do while (nml_error > 0)
   190)          read(nu_nml, nml=domain_nml,iostat=nml_error)
   191)       end do
   192)       if (nml_error == 0) close(nu_nml)
   193)    endif
   194)    call release_fileunit(nu_nml)
   195) 
   196)    call broadcast_scalar(nml_error, master_task)
   197)    if (nml_error /= 0) then
   198)       call abort_ice('ice: error reading domain_nml')
   199)    endif
   200) 
   201)    call broadcast_scalar(nprocs,            master_task)
   202)    call broadcast_scalar(processor_shape,   master_task)
   203)    call broadcast_scalar(distribution_type, master_task)
   204) 
   205)    call broadcast_scalar(distribution_wght_file,  master_task)
   206)    call broadcast_scalar(distribution_wght, master_task)
   207) 
   208)    call broadcast_scalar(ew_boundary_type,  master_task)
   209)    call broadcast_scalar(ns_boundary_type,  master_task)
   210)    call broadcast_scalar(profile_barrier,   master_task)
   211)    call broadcast_scalar(maxBlock,          master_task)
   212)    call broadcast_scalar(minBlock,          master_task)
   213)    call broadcast_scalar(maxDil,          master_task)
   214) 
   215) !----------------------------------------------------------------------
   216) !
   217) !  perform some basic checks on domain
   218) !
   219) !----------------------------------------------------------------------
   220)    if(trim(distribution_type) == 'spacecurve') then 
   221)       if(trim(distribution_wght_file) == 'unknown_distribution_wght_file') then 
   222) 	distribution_wght = 'erfc'
   223)       else 
   224)         distribution_wght =  'file'
   225)       endif
   226)    endif
   227) 
   228)    if (trim(ns_boundary_type) == 'tripole') then
   229)       ltripole_grid = .true.
   230)    else
   231)       ltripole_grid = .false.
   232)    endif
   233) 
   234)    if (nx_global < 1 .or. ny_global < 1 .or. ncat < 1) then
   235)       !***
   236)       !*** domain size zero or negative
   237)       !***
   238)       call abort_ice('ice: Invalid domain: size < 1') ! no domain
   239)    else if (nprocs /= get_num_procs()) then
   240)       !***
   241)       !*** input nprocs does not match system (eg MPI) request
   242)       !***
   243) #ifdef CCSMCOUPLED
   244)       nprocs = get_num_procs()
   245) #else
   246)       call abort_ice('ice: Input nprocs not same as system request')
   247) #endif
   248)    else if (nghost < 1) then
   249)       !***
   250)       !*** must have at least 1 layer of ghost cells
   251)       !***
   252)       call abort_ice('ice: Not enough ghost cells allocated')
   253)    endif
   254) 
   255) !----------------------------------------------------------------------
   256) !  notify global_reductions whether tripole grid is being used
   257) !----------------------------------------------------------------------
   258) 
   259)    call init_global_reductions (ltripole_grid)
   260) 
   261) !----------------------------------------------------------------------
   262) !
   263) !  compute block decomposition and details
   264) !
   265) !----------------------------------------------------------------------
   266) 
   267)    call create_blocks(nx_global, ny_global, trim(ew_boundary_type), &
   268)                                             trim(ns_boundary_type))
   269) 
   270) !----------------------------------------------------------------------
   271) !
   272) !  Now we need grid info before proceeding further
   273) !  Print some domain information
   274) !
   275) !----------------------------------------------------------------------
   276) 
   277)    if (my_task == master_task) then
   278)      write(nu_diag,'(/,a18,/)')'Domain Information'
   279)      write(nu_diag,'(a26,i6)') '  Horizontal domain: nx = ',nx_global
   280)      write(nu_diag,'(a26,i6)') '                     ny = ',ny_global
   281)      write(nu_diag,'(a26,i6)') '  No. of categories: nc = ',ncat
   282)      write(nu_diag,'(a26,i6)') '  No. of ice layers: ni = ',nilyr
   283)      write(nu_diag,'(a26,i6)') '  No. of snow layers:ns = ',nslyr
   284)      write(nu_diag,'(a26,i6)') '  Processors:  total    = ',nprocs
   285)      write(nu_diag,'(a25,a10)') '  Processor shape:        ', &
   286)                                   trim(processor_shape)
   287)      write(nu_diag,'(a25,a10)') '  Distribution type:      ', &
   288)                                   trim(distribution_type)
   289) !     write(nu_diag,'(a31,a9)') '  Probability: ', &
   290) !                                  trim(probability_type)
   291)      write(nu_diag,'(a25,a10)') '  Distribution weight:    ', &
   292)                                   trim(distribution_wght)
   293)      if(trim(distribution_wght) == 'file') then 
   294)          write(nu_diag,'(a30,a80)') '  Distribution weight file:  ', &
   295)                                   trim(distribution_wght_file)
   296)      endif
   297)      write(nu_diag,'(a26,2(i6))') '  {min,max}Blocks =            ', minBlock,maxBlock
   298)      write(nu_diag,'(a26,i6,/)')'  Number of ghost cells:  ', nghost
   299)      call flush_fileunit(nu_diag)
   300)    endif
   301) 
   302) !----------------------------------------------------------------------
   303) !EOC
   304) 
   305)  end subroutine init_domain_blocks

   312)  subroutine init_domain_distribution(KMTG,ULATG,work_per_block,prob_per_block,blockType,bStats,maxDil)
   313) 
   314) ! !DESCRIPTION:
   315) !  This routine calls appropriate setup routines to distribute blocks
   316) !  across processors and defines arrays with block ids for any local
   317) !  blocks. Information about ghost cell update routines is also
   318) !  initialized here through calls to the appropriate boundary routines.
   319) !
   320) ! !REVISION HISTORY:
   321) !  same as module
   322) 
   323) ! !INPUT PARAMETERS:
   324) 
   325)    real (dbl_kind), dimension(nx_global,ny_global), intent(in) :: &
   326)       KMTG           ,&! global topography
   327)       ULATG            ! global latitude field (radians)
   328) 
   329)    integer(int_kind), intent(in), dimension(:) ::  work_per_block,blockType
   330)    
   331)    real (dbl_kind), intent(in), dimension(:) :: prob_per_block
   332)    
   333)    real (dbl_kind), intent(in), dimension(:,:)  :: bStats
   334) 
   335)    real (real_kind), intent(in) :: maxDil
   336) !EOP
   337) !BOC
   338) !----------------------------------------------------------------------
   339) !
   340) !  local variables
   341) !
   342) !----------------------------------------------------------------------
   343) 
   344)    integer (int_kind), dimension (nx_global, ny_global) :: &
   345)       flat                 ! latitude-dependent scaling factor
   346) 
   347)    character (char_len) :: outstring
   348) 
   349)    integer (int_kind), parameter :: &
   350)       max_work_unit=10    ! quantize the work into values from 1,max
   351) 
   352)    integer (int_kind) :: &
   353)       i,j,k,n            ,&! dummy loop indices
   354)       ig,jg              ,&! global indices
   355)       work_unit          ,&! size of quantized work unit
   356)       nblocks_tmp        ,&! temporary value of nblocks
   357)       nblocks_max          ! max blocks on proc
   358) 
   359)    integer (int_kind), dimension(:), allocatable :: &
   360)       nocn               ! number of ocean points per block
   361) 
   362)    type (block) :: &
   363)       this_block           ! block information for current block
   364) 
   365) !----------------------------------------------------------------------
   366) !
   367) !  check that there are at least nghost+1 rows or columns of land cells
   368) !  for closed boundary conditions (otherwise grid lengths are zero in
   369) !  cells neighboring ocean points).  
   370) !
   371) !----------------------------------------------------------------------
   372) 
   373)    if (trim(ns_boundary_type) == 'closed') then
   374)       allocate(nocn(nblocks_tot))
   375)       nocn = 0
   376)       do n=1,nblocks_tot
   377)          this_block = get_block(n,n)
   378)          if (this_block%jblock == nblocks_y) then ! north edge
   379)          do j = this_block%jhi-1, this_block%jhi
   380)             if (this_block%j_glob(j) > 0) then
   381)                do i = 1, nx_block
   382)                   if (this_block%i_glob(i) > 0) then
   383)                      ig = this_block%i_glob(i)
   384)                      jg = this_block%j_glob(j)
   385)                      if (KMTG(ig,jg) > puny) nocn(n) = nocn(n) + 1
   386)                   endif
   387)                enddo
   388)             endif
   389)          enddo
   390)          endif
   391)          if (this_block%jblock == 1) then ! south edge
   392)          do j = this_block%jlo, this_block%jlo+1
   393)             if (this_block%j_glob(j) > 0) then
   394)                do i = 1, nx_block
   395)                   if (this_block%i_glob(i) > 0) then
   396)                      ig = this_block%i_glob(i)
   397)                      jg = this_block%j_glob(j)
   398)                      if (KMTG(ig,jg) > puny) nocn(n) = nocn(n) + 1
   399)                   endif
   400)                enddo
   401)             endif
   402)          enddo
   403)          endif
   404)          if (nocn(n) > 0) then
   405)             print*, 'ice: Not enough land cells along ns edge'
   406)             call abort_ice('ice: Not enough land cells along ns edge')
   407)          endif
   408)       enddo
   409)       deallocate(nocn)
   410)    endif
   411)    if (trim(ew_boundary_type) == 'closed') then
   412)       allocate(nocn(nblocks_tot))
   413)       nocn = 0
   414)       do n=1,nblocks_tot
   415)          this_block = get_block(n,n)
   416)          if (this_block%iblock == nblocks_x) then ! east edge
   417)          do j = 1, ny_block
   418)             if (this_block%j_glob(j) > 0) then
   419)                do i = this_block%ihi-1, this_block%ihi
   420)                   if (this_block%i_glob(i) > 0) then
   421)                      ig = this_block%i_glob(i)
   422)                      jg = this_block%j_glob(j)
   423)                      if (KMTG(ig,jg) > puny) nocn(n) = nocn(n) + 1
   424)                   endif
   425)                enddo
   426)             endif
   427)          enddo
   428)          endif
   429)          if (this_block%iblock == 1) then ! west edge
   430)          do j = 1, ny_block
   431)             if (this_block%j_glob(j) > 0) then
   432)                do i = this_block%ilo, this_block%ilo+1
   433)                   if (this_block%i_glob(i) > 0) then
   434)                      ig = this_block%i_glob(i)
   435)                      jg = this_block%j_glob(j)
   436)                      if (KMTG(ig,jg) > puny) nocn(n) = nocn(n) + 1
   437)                   endif
   438)                enddo
   439)             endif
   440)          enddo
   441)          endif
   442)          if (nocn(n) > 0) then
   443)             print*, 'ice: Not enough land cells along ew edge'
   444)             call abort_ice('ice: Not enough land cells along ew edge')
   445)          endif
   446)       enddo
   447)       deallocate(nocn)
   448)    endif
   449) 
   450) !----------------------------------------------------------------------
   451) !
   452) !  estimate the amount of work per processor using the topography
   453) !  and latitude
   454) !
   455) !----------------------------------------------------------------------
   456) 
   457)    if (distribution_wght == 'latitude') then
   458)        flat = NINT(abs(ULATG*rad_to_deg), int_kind) ! linear function
   459)    else
   460)        flat = 1
   461)    endif
   462) 
   463)    allocate(nocn(nblocks_tot))
   464) 
   465)    nocn = 0
   466)    do n=1,nblocks_tot
   467)       this_block = get_block(n,n)
   468)       do j=this_block%jlo,this_block%jhi
   469)          if (this_block%j_glob(j) > 0) then
   470)             do i=this_block%ilo,this_block%ihi
   471)                if (this_block%i_glob(i) > 0) then
   472) 	          ig = this_block%i_glob(i)
   473)                   jg = this_block%j_glob(j)
   474)                   if (KMTG(ig,jg) > puny .and.                      &
   475)                      (ULATG(ig,jg) < shlat/rad_to_deg .or.          &
   476)                       ULATG(ig,jg) > nhlat/rad_to_deg) )            & 
   477)  	              nocn(n) = nocn(n) + flat(ig,jg)
   478)                endif
   479)             end do
   480)          endif
   481)       end do
   482) 
   483)       !*** with array syntax, we actually do work on non-ocean
   484)       !*** points, so where the block is not completely land,
   485)       !*** reset nocn to be the full size of the block
   486) 
   487)       ! use processor_shape = 'square-pop' and distribution_wght = 'block' 
   488)       ! to make CICE and POP decompositions/distributions identical.
   489) 
   490)       if (distribution_wght == 'block' .and. &   ! POP style
   491)           nocn(n) > 0) nocn(n) = nx_block*ny_block
   492) 
   493)    end do
   494) 
   495)    work_unit = maxval(nocn)/max_work_unit + 1
   496) 
   497)    !*** find number of work units per block
   498) 
   499)    deallocate(nocn)
   500) 
   501) !----------------------------------------------------------------------
   502) !
   503) !  determine the distribution of blocks across processors
   504) !
   505) !----------------------------------------------------------------------
   506) 
   507) !DBG   print *,'init_domain_distribution: before call to create_distribution'
   508)    distrb_info = create_distribution(distribution_type, nprocs, minBlock, maxBlock, &
   509)                      work_per_block, prob_per_block, blockType, bStats, FixMaxBlock,maxDil )
   510) !JMD   call abort_ice('init_domain_distribution: after call to create_distribution')
   511) 
   512) !DBG   print *,'after call to create_distribution'
   513) 
   514) !----------------------------------------------------------------------
   515) !
   516) !  allocate and determine block id for any local blocks
   517) !
   518) !----------------------------------------------------------------------
   519) 
   520)    call create_local_block_ids(blocks_ice, distrb_info)
   521) !DBG   print *,'after call to create_local_block_ids'
   522) 
   523)    if (associated(blocks_ice)) then
   524)       nblocks = size(blocks_ice)
   525)    else
   526)       nblocks = 0
   527)    endif
   528)    nblocks_max = 0
   529)    do n=0,distrb_info%nprocs - 1
   530)      nblocks_tmp = nblocks
   531)      call broadcast_scalar(nblocks_tmp, n)
   532)      nblocks_max = max(nblocks_max,nblocks_tmp)
   533)    end do
   534) 
   535)    if (nblocks_max > max_blocks) then
   536)      write(outstring,*) &
   537)          'ice: no. blocks exceed max: increase max to', nblocks_max
   538)      call abort_ice(trim(outstring))
   539)    else if (nblocks_max < max_blocks) then
   540)      write(outstring,*) &
   541)          'ice: no. blocks too large: decrease max to', nblocks_max
   542)      if (my_task == master_task) then
   543)         write(nu_diag,*) ' ********WARNING***********'
   544)         write(nu_diag,*) trim(outstring)
   545)         write(nu_diag,*) ' **************************'
   546)         write(nu_diag,*) ' '
   547)      endif
   548)    endif
   549) 
   550) !----------------------------------------------------------------------
   551) !
   552) !  Set up ghost cell updates for each distribution.
   553) !  Boundary types are cyclic, closed, or tripole. 
   554) !
   555) !----------------------------------------------------------------------
   556) 
   557)    ! update ghost cells on all four boundaries
   558)    halo_info = ice_HaloCreate(distrb_info,     &
   559)                         trim(ns_boundary_type),     &
   560)                         trim(ew_boundary_type),     &
   561)                         nx_global)
   562) 
   563) !----------------------------------------------------------------------
   564) !EOC
   565) 
   566)  end subroutine init_domain_distribution
