    84)   rmwf11 = mwf11/mwdry      ! ratio of molecular weight cfc11 to dry air
    85)   rmwf12 = mwf12/mwdry      ! ratio of molecular weight cfc12 to dry air

   104) subroutine trcmix(name, ncol, clat, pmid, q)
   105) !----------------------------------------------------------------------- 
   106) ! 
   107) ! Purpose: 
   108) ! Specify zonal mean mass mixing ratios of CH4, N2O, CFC11 and
   109) ! CFC12
   110) ! 
   111) ! Method: 
   112) ! Distributions assume constant mixing ratio in the troposphere
   113) ! and a decrease of mixing ratio in the stratosphere. Tropopause
   114) ! defined by ptrop. The scale height of the particular trace gas
   115) ! depends on latitude. This assumption produces a more realistic
   116) ! stratospheric distribution of the various trace gases.
   117) ! 
   118) ! Author: J. Kiehl
   119) ! 
   120) !-----------------------------------------------------------------------
   121) 
   122)    ! Arguments
   123)    character(len=*), intent(in)  :: name              ! constituent name
   124)    integer,          intent(in)  :: ncol              ! number of columns
   125)    real(r8),         intent(in)  :: clat(pcols)       ! latitude in radians for columns
   126)    real(r8),         intent(in)  :: pmid(pcols,pver)  ! model pressures
   127)    real(r8),         intent(out) :: q(pcols,pver)     ! constituent mass mixing ratio
   128) 
   129)    integer i                ! longitude loop index
   130)    integer k                ! level index
   131) 
   132)    real(r8) coslat(pcols)   ! cosine of latitude
   133)    real(r8) dlat            ! latitude in degrees
   134)    real(r8) ptrop           ! pressure level of tropopause
   135)    real(r8) pratio          ! pressure divided by ptrop
   136)    real(r8) trop_mmr        ! tropospheric mass mixing ratio
   137)    real(r8) scale           ! pressure scale height
   138) !-----------------------------------------------------------------------
   139) 
   140)    do i = 1, ncol
   141)       coslat(i) = cos(clat(i))
   142)    end do
   143) 
   144)    if (name == 'O2') then
   145) 
   146)       q = chem_surfvals_get('O2MMR')
   147) 
   148)    else if (name == 'CO2') then
   149) 
   150)       q = chem_surfvals_co2_rad()
   151) 
   152)    else if (name == 'CH4') then
   153) 
   154)       ! set tropospheric mass mixing ratios
   155)       trop_mmr = rmwch4 * chem_surfvals_get('CH4VMR')
   156) 
   157)       do k = 1,pver
   158)          do i = 1,ncol
   159)             ! set stratospheric scale height factor for gases
   160)             dlat = abs(57.2958_r8 * clat(i))
   161)             if(dlat.le.45.0_r8) then
   162)                scale = 0.2353_r8
   163)             else
   164)                scale = 0.2353_r8 + 0.0225489_r8 * (dlat - 45)
   165)             end if
   166) 
   167)             ! pressure of tropopause
   168)             ptrop = 250.0e2_r8 - 150.0e2_r8*coslat(i)**2.0_r8
   169) 
   170)             ! determine output mass mixing ratios
   171)             if (pmid(i,k) >= ptrop) then
   172)                q(i,k) = trop_mmr
   173)             else
   174)                pratio = pmid(i,k)/ptrop
   175)                q(i,k) = trop_mmr * (pratio)**scale
   176)             end if
   177)          end do
   178)       end do
   179) 
   180)    else if (name == 'N2O') then
   181) 
   182)       ! set tropospheric mass mixing ratios
   183)       trop_mmr = rmwn2o * chem_surfvals_get('N2OVMR')
   184) 
   185)       do k = 1,pver
   186)          do i = 1,ncol
   187)             ! set stratospheric scale height factor for gases
   188)             dlat = abs(57.2958_r8 * clat(i))
   189)             if(dlat.le.45.0_r8) then
   190)                scale = 0.3478_r8 + 0.00116_r8 * dlat
   191)             else
   192)                scale = 0.4000_r8 + 0.013333_r8 * (dlat - 45)
   193)             end if
   194) 
   195)             ! pressure of tropopause
   196)             ptrop = 250.0e2_r8 - 150.0e2_r8*coslat(i)**2.0_r8
   197) 
   198)             ! determine output mass mixing ratios
   199)             if (pmid(i,k) >= ptrop) then
   200)                q(i,k) = trop_mmr
   201)             else
   202)                pratio = pmid(i,k)/ptrop
   203)                q(i,k) = trop_mmr * (pratio)**scale
   204)             end if
   205)          end do
   206)       end do
   207) 
   208)    else if (name == 'CFC11') then
   209) 
   210)       ! set tropospheric mass mixing ratios
   211)       trop_mmr = rmwf11 * chem_surfvals_get('F11VMR')
   212) 
   213)       do k = 1,pver
   214)          do i = 1,ncol
   215)             ! set stratospheric scale height factor for gases
   216)             dlat = abs(57.2958_r8 * clat(i))
   217)             if(dlat.le.45.0_r8) then
   218)                scale = 0.7273_r8 + 0.00606_r8 * dlat
   219)             else
   220)                scale = 1.00_r8 + 0.013333_r8 * (dlat - 45)
   221)             end if
   222) 
   223)             ! pressure of tropopause
   224)             ptrop = 250.0e2_r8 - 150.0e2_r8*coslat(i)**2.0_r8
   225) 
   226)             ! determine output mass mixing ratios
   227)             if (pmid(i,k) >= ptrop) then
   228)                q(i,k) = trop_mmr
   229)             else
   230)                pratio = pmid(i,k)/ptrop
   231)                q(i,k) = trop_mmr * (pratio)**scale
   232)             end if
   233)          end do
   234)       end do
   235) 
   236)    else if (name == 'CFC12') then
   237) 
   238)       ! set tropospheric mass mixing ratios
   239)       trop_mmr = rmwf12 * chem_surfvals_get('F12VMR')
   240) 
   241)       do k = 1,pver
   242)          do i = 1,ncol
   243)             ! set stratospheric scale height factor for gases
   244)             dlat = abs(57.2958_r8 * clat(i))
   245)             if(dlat.le.45.0_r8) then
   246)                scale = 0.4000_r8 + 0.00222_r8 * dlat
   247)             else
   248)                scale = 0.50_r8 + 0.024444_r8 * (dlat - 45)
   249)             end if
   250) 
   251)             ! pressure of tropopause
   252)             ptrop = 250.0e2_r8 - 150.0e2_r8*coslat(i)**2.0_r8
   253) 
   254)             ! determine output mass mixing ratios
   255)             if (pmid(i,k) >= ptrop) then
   256)                q(i,k) = trop_mmr
   257)             else
   258)                pratio = pmid(i,k)/ptrop
   259)                q(i,k) = trop_mmr * (pratio)**scale
   260)             end if
   261)          end do
   262)       end do
   263) 
   264)    end if
   265) 
   266) end subroutine trcmix
