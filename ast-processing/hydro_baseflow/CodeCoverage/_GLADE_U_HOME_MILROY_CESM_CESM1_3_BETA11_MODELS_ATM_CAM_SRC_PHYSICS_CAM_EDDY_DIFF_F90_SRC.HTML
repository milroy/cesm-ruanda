   103)   real(r8)                    :: b123                           ! b1**(2/3)

   106)   real(r8),         parameter :: alph2          =  -4.3640_r8   !               These coefficients are used to calculate 
   107)   real(r8),         parameter :: alph3          = -34.6764_r8   !               'sh' and 'sm' from 'gh'.
   108)   real(r8),         parameter :: alph4          =  -6.1272_r8   !
   109)   real(r8),         parameter :: alph5          =   0.6986_r8   !

   112)   real(r8),         parameter :: ae             =   1._r8       ! TKE transport efficiency [no unit]
   113)   real(r8),         parameter :: rinc           =  -0.04_r8     ! Minimum W/<W> used for CL merging test 
   114)   real(r8),         parameter :: wpertmin       =   1.e-6_r8    ! Minimum PBL eddy vertical velocity perturbation
   115)   real(r8),         parameter :: wfac           =   1._r8       ! Ratio of 'wpert' to sqrt(tke) for CL.
   116)   real(r8),         parameter :: tfac           =   1._r8       ! Ratio of 'tpert' to (w't')/wpert for CL.
   117)                                                                 ! Same ratio also used for q
   118)   real(r8),         parameter :: fak            =   8.5_r8      ! Constant in surface temperature excess for stable STL.
   119)                                                                 ! [ no unit ]
   120)   real(r8),         parameter :: rcapmin        =   0.1_r8      ! Minimum allowable e/<e> in a CL
   121)   real(r8),         parameter :: rcapmax        =   2.0_r8      ! Maximum allowable e/<e> in a CL
   122)   real(r8),         parameter :: tkemax         =  20._r8       ! TKE is capped at tkemax [m2/s2]
   123)   real(r8),         parameter :: lambda         =   0.5_r8      ! Under-relaxation factor ( 0 < lambda =< 1 )
   124) 
   125)   logical,          parameter :: use_kvf        =  .false.      ! .true. (.false.) : initialize kvh/kvm =  kvf ( 0. )
   126)   logical,          parameter :: use_dw_surf    =  .true.       ! Used in 'zisocl'. Default is 'true'

   129)                                                                 ! For this case, 'dl2n2_surf' is only used for a merging test

   182)   subroutine init_eddy_diff( kind, pver, gravx, cpairx, rairx, zvirx, & 
   183)                              latvapx, laticex, ntop_eddy, nbot_eddy, vkx, &
   184)                              eddy_lbulk_max, eddy_leng_max, eddy_max_bot_pressure, &
   185)                              eddy_moist_entrain_a2l)
   186)     !---------------------------------------------------------------- ! 
   187)     ! Purpose:                                                        !
   188)     ! Initialize time independent constants/variables of PBL package. !
   189)     !---------------------------------------------------------------- !
   190)     use diffusion_solver, only: new_fieldlist_vdiff, vdiff_select
   191)     use cam_history,      only: outfld, addfld, phys_decomp
   192)     use ref_pres,         only: pref_mid
   193)     
   194)     ! --------- !
   195)     ! Arguments !
   196)     ! --------- !
   197)     integer,  intent(in) :: kind       ! Kind of reals being passed in
   198)     integer,  intent(in) :: pver       ! Number of vertical layers
   199)     integer,  intent(in) :: ntop_eddy  ! Top interface level to which eddy vertical diffusivity is applied ( = 1 )
   200)     integer,  intent(in) :: nbot_eddy  ! Bottom interface level to which eddy vertical diffusivity is applied ( = pver )
   201)     real(r8), intent(in) :: gravx      ! Acceleration of gravity
   202)     real(r8), intent(in) :: cpairx     ! Specific heat of dry air
   203)     real(r8), intent(in) :: rairx      ! Gas constant for dry air
   204)     real(r8), intent(in) :: zvirx      ! rh2o/rair - 1
   205)     real(r8), intent(in) :: latvapx    ! Latent heat of vaporization
   206)     real(r8), intent(in) :: laticex    ! Latent heat of fusion
   207)     real(r8), intent(in) :: vkx        ! Von Karman's constant
   208)     real(r8), intent(in) :: eddy_lbulk_max ! Maximum master length scale
   209)     real(r8), intent(in) :: eddy_leng_max  ! Maximum dissipation length scale
   210)     real(r8), intent(in) :: eddy_max_bot_pressure  ! Bottom pressure level (hPa) at which namelist leng_max and lbulk_max
   211)                                                    ! are applied
   212)     real(r8), intent(in) :: eddy_moist_entrain_a2l ! Moist entrainment enhancement param
   213) 
   214)     integer              :: k          ! Vertical loop index
   215) 
   216)     if( kind .ne. r8 ) then
   217)         write(iulog,*) 'wrong KIND of reals passed to init_diffusvity -- exiting.'
   218)         call endrun('init_eddy_diff: wrong KIND of reals passed to init_diffusvity')
   219)     endif
   220) 
   221)     ! --------------- !
   222)     ! Basic constants !
   223)     ! --------------- !
   224) 
   225)     cpair     = cpairx
   226)     rair      = rairx
   227)     g         = gravx
   228)     zvir      = zvirx
   229)     latvap    = latvapx
   230)     latice    = laticex
   231)     latsub    = latvap + latice
   232)     vk        = vkx
   233)     ccon      = fak*sffrac*vk
   234)     ntop_turb = ntop_eddy
   235)     nbot_turb = nbot_eddy
   236)     b123      = b1**(2._r8/3._r8)
   237)     a2l       = eddy_moist_entrain_a2l
   238)     
   239)     lbulk_max = eddy_lbulk_max
   240)     do k = 1,pver
   241)       if ( pref_mid(k) .le. eddy_max_bot_pressure*1.D2 ) leng_max(k)  = eddy_leng_max
   242)     end do
   243) 
   244)     if (masterproc) then
   245)        write(iulog,*)'init_eddy_diff: eddy_leng_max=',eddy_leng_max,' lbulk_max=',lbulk_max
   246)        do k = 1,pver
   247)           write(iulog,*)'init_eddy_diff:',k,pref_mid(k),'leng_max=',leng_max(k)
   248)        end do
   249)     end if
   250) 
   251)     ! Set the square of the mixing lengths. Only for CAM3 HB PBL scheme.
   252)     ! Not used for UW moist PBL. Used for free air eddy diffusivity.
   253) 
   254)     allocate(ml2(pver+1))
   255)     ml2(1:ntop_turb) = 0._r8
   256)     do k = ntop_turb + 1, nbot_turb
   257)        ml2(k) = 30.0_r8**2
   258)     end do
   259)     ml2(nbot_turb+1:pver+1) = 0._r8
   260)     
   261)     ! Get fieldlists to pass to diffusion solver.
   262)     fieldlist_wet   = new_fieldlist_vdiff(1)
   263)     fieldlist_molec = new_fieldlist_vdiff(1)
   264) 
   265)     ! Select the fields which will be diffused 
   266) 
   267)     if(vdiff_select(fieldlist_wet,'s').ne.'')   call endrun( vdiff_select(fieldlist_wet,'s') )
   268)     if(vdiff_select(fieldlist_wet,'q',1).ne.'') call endrun( vdiff_select(fieldlist_wet,'q',1) )
   269)     if(vdiff_select(fieldlist_wet,'u').ne.'')   call endrun( vdiff_select(fieldlist_wet,'u') )
   270)     if(vdiff_select(fieldlist_wet,'v').ne.'')   call endrun( vdiff_select(fieldlist_wet,'v') )
   271) 
   272)     ! ------------------------------------------------------------------- !
   273)     ! Writing outputs for detailed analysis of UW moist turbulence scheme !
   274)     ! ------------------------------------------------------------------- !
   275) 
   276)     call addfld('UW_errorPBL',      'm2/s',    1,      'A',  'Error function of UW PBL',                              phys_decomp )
   277)     call addfld('UW_n2',            's-2',     pver,   'A',  'Buoyancy Frequency, LI',                                phys_decomp )
   278)     call addfld('UW_s2',            's-2',     pver,   'A',  'Shear Frequency, LI',                                   phys_decomp )
   279)     call addfld('UW_ri',            'no',      pver,   'A',  'Interface Richardson Number, I',                        phys_decomp )
   280)     call addfld('UW_sfuh',          'no',      pver,   'A',  'Upper-Half Saturation Fraction, L',                     phys_decomp )
   281)     call addfld('UW_sflh',          'no',      pver,   'A',  'Lower-Half Saturation Fraction, L',                     phys_decomp )
   282)     call addfld('UW_sfi',           'no',      pver+1, 'A',  'Interface Saturation Fraction, I',                      phys_decomp )
   283)     call addfld('UW_cldn',          'no',      pver,   'A',  'Cloud Fraction, L',                                     phys_decomp )
   284)     call addfld('UW_qrl',           'g*W/m2',  pver,   'A',  'LW cooling rate, L',                                    phys_decomp )
   285)     call addfld('UW_ql',            'kg/kg',   pver,   'A',  'ql(LWC), L',                                            phys_decomp )
   286)     call addfld('UW_chu',           'g*kg/J',  pver+1, 'A',  'Buoyancy Coefficient, chu, I',                          phys_decomp )
   287)     call addfld('UW_chs',           'g*kg/J',  pver+1, 'A',  'Buoyancy Coefficient, chs, I',                          phys_decomp )
   288)     call addfld('UW_cmu',           'g/kg/kg', pver+1, 'A',  'Buoyancy Coefficient, cmu, I',                          phys_decomp )
   289)     call addfld('UW_cms',           'g/kg/kg', pver+1, 'A',  'Buoyancy Coefficient, cms, I',                          phys_decomp )
   290)     call addfld('UW_tke',           'm2/s2',   pver+1, 'A',  'TKE, I',                                                phys_decomp )
   291)     call addfld('UW_wcap',          'm2/s2',   pver+1, 'A',  'Wcap, I',                                               phys_decomp )
   292)     call addfld('UW_bprod',         'm2/s3',   pver+1, 'A',  'Buoyancy production, I',                                phys_decomp )
   293)     call addfld('UW_sprod',         'm2/s3',   pver+1, 'A',  'Shear production, I',                                   phys_decomp )
   294)     call addfld('UW_kvh',           'm2/s',    pver+1, 'A',  'Eddy diffusivity of heat, I',                           phys_decomp )
   295)     call addfld('UW_kvm',           'm2/s',    pver+1, 'A',  'Eddy diffusivity of uv, I',                             phys_decomp )
   296)     call addfld('UW_pblh',          'm',       1,      'A',  'PBLH, 1',                                               phys_decomp )
   297)     call addfld('UW_pblhp',         'Pa',      1,      'A',  'PBLH pressure, 1',                                      phys_decomp )
   298)     call addfld('UW_tpert',         'K',       1,      'A',  'Convective T excess, 1',                                phys_decomp )
   299)     call addfld('UW_qpert',         'kg/kg',   1,      'A',  'Convective qt excess, I',                               phys_decomp )
   300)     call addfld('UW_wpert',         'm/s',     1,      'A',  'Convective W excess, I',                                phys_decomp )
   301)     call addfld('UW_ustar',         'm/s',     1,      'A',  'Surface Frictional Velocity, 1',                        phys_decomp )
   302)     call addfld('UW_tkes',          'm2/s2',   1,      'A',  'Surface TKE, 1',                                        phys_decomp )
   303)     call addfld('UW_minpblh',       'm',       1,      'A',  'Minimum PBLH, 1',                                       phys_decomp )
   304)     call addfld('UW_turbtype',      'no',      pver+1, 'A',  'Interface Turbulence Type, I',                          phys_decomp )
   305)     call addfld('UW_kbase_o',       'no',      ncvmax, 'A',  'Initial CL Base Exterbal Interface Index, CL',          phys_decomp )
   306)     call addfld('UW_ktop_o',        'no',      ncvmax, 'A',  'Initial Top Exterbal Interface Index, CL',              phys_decomp )

   314)     call addfld('UW_wet',           'm/s',     ncvmax, 'A',  'Entrainment rate at CL top, CL',                        phys_decomp )
   315)     call addfld('UW_web',           'm/s',     ncvmax, 'A',  'Entrainment rate at CL base, CL',                       phys_decomp )
   316)     call addfld('UW_jtbu',          'm/s2',    ncvmax, 'A',  'Buoyancy jump across CL top, CL',                       phys_decomp )
   317)     call addfld('UW_jbbu',          'm/s2',    ncvmax, 'A',  'Buoyancy jump across CL base, CL',                      phys_decomp )
   318)     call addfld('UW_evhc',          'no',      ncvmax, 'A',  'Evaporative enhancement factor, CL',                    phys_decomp )
   319)     call addfld('UW_jt2slv',        'J/kg',    ncvmax, 'A',  'slv jump for evhc, CL',                                 phys_decomp )

   574)     tfd(:ncol,:)  = t(:ncol,:)

   589)                         tauy(:ncol) - ksrftms(:ncol) * vfd(:ncol,pver), & ! Meridional wind stress
   590)                         rrho(:ncol), ustar(:ncol))

   606)            sl(:ncol,:) = slfd(:ncol,:)
   607)        endif

   610) 
   611)        call austausch_atm( pcols, pver, ncol, ri, s2, kvf )

   618)            ! First iteration of first model timestep : Use free tropospheric value or zero.
   619)              if( use_kvf ) then
   620)                  kvh(:ncol,:) = kvf(:ncol,:)
   621)                  kvm(:ncol,:) = kvf(:ncol,:)
   622)              else
   623)                  kvh(:ncol,:) = 0._r8
   624)                  kvm(:ncol,:) = 0._r8
   625)              endif
   626)            else
   627)              ! First iteration on any model timestep except the first : Use value from previous timestep
   628)              kvh(:ncol,:) = kvh_in(:ncol,:)
   629)              kvm(:ncol,:) = kvm_in(:ncol,:)
   630)            endif
   631)        else
   632)         ! Not the first iteration : Use from previous iteration
   633)           kvh(:ncol,:) = kvh_out(:ncol,:)
   634)           kvm(:ncol,:) = kvm_out(:ncol,:)
   635)        endif
   636) 
   637)      ! Calculate eddy diffusivity (kvh_out,kvm_out) and (tke,bprod,sprod) using
   638)      ! a given (kvh,kvm) which are used only for initializing (bprod,sprod)  at
   639)      ! the first part of caleddy. (bprod,sprod) are fully updated at the end of
   640)      ! caleddy after calculating (kvh_out,kvm_out) 
   641) 
   642)        call caleddy( pcols     , pver      , ncol      ,                     &
   643)                      slfd      , qtfd      , qlfd      , slv      ,ufd     , &
   644)                      vfd       , pi        , z         , zi       ,          &
   645)                      qflx      , shflx     , slslope   , qtslope  ,          &
   646)                      chu       , chs       , cmu       , cms      ,sfuh    , &
   647)                      sflh      , n2        , s2        , ri       ,rrho    , &
   648)                      pblh      , ustar     ,                                 &
   649)                      kvh       , kvm       , kvh_out   , kvm_out  ,          &
   650)                      tpert     , qpert     , qrl       , kvf      , tke    , &
   651)                      wstarent  , bprod     , sprod     , minpblh  , wpert  , &
   652)                      tkes      , went      , turbtype  , sm_aw    ,          & 
   653)                      kbase_o   , ktop_o    , ncvfin_o  ,                     &
   654)                      kbase_mg  , ktop_mg   , ncvfin_mg ,                     &                  
   655)                      kbase_f   , ktop_f    , ncvfin_f  ,                     &                  
   656)                      wet       , web       , jtbu      , jbbu     ,          &
   657)                      evhc      , jt2slv    , n2ht      , n2hb     ,          & 
   658)                      lwp       , opt_depth , radinvfrac, radf     ,          &
   659)                      wstar     , wstar3fact,                                 &
   660)                      ebrk      , wbrk      , lbrk      , ricl     , ghcl   , & 
   661)                      shcl      , smcl      , ghi       , shi      , smi    , &
   662)                      rii       , lengi     , wcap      , pblhp    , cldn   , &
   663)                      ipbl      , kpblh     , wsedl     , wsed)
   664) 
   665)      ! Calculate errorPBL to check whether PBL produced convergent solutions or not.
   666) 
   667)        if( iturb .eq. nturb ) then
   668)            do i = 1, ncol
   669)               errorPBL(i) = 0._r8 
   670)               do k = 1, pver
   671)                  errorPBL(i) = errorPBL(i) + ( kvh(i,k) - kvh_out(i,k) )**2 
   672)               end do 
   673)               errorPBL(i) = sqrt(errorPBL(i)/pver)
   674)            end do
   675)        end if
   676) 
   677)      ! Eddy diffusivities which will be used for the initialization of (bprod,
   678)      ! sprod) in 'caleddy' at the next iteration step.
   679) 
   680)        if( iturb .gt. 1 .and. iturb .lt. nturb ) then
   681)            kvm_out(:ncol,:) = lambda * kvm_out(:ncol,:) + ( 1._r8 - lambda ) * kvm(:ncol,:)
   682)            kvh_out(:ncol,:) = lambda * kvh_out(:ncol,:) + ( 1._r8 - lambda ) * kvh(:ncol,:)
   683)        endif
   684) 
   685)      ! Set nonlocal terms to zero for flux diagnostics, since not used by caleddy.
   686) 
   687)        cgh(:ncol,:) = 0._r8
   688)        cgs(:ncol,:) = 0._r8      
   689) 
   690)        if( iturb .lt. nturb ) then
   691) 
   692)          ! Each time we diffuse the original state
   693) 
   694)            slfd(:ncol,:)  = sl(:ncol,:)
   695)            qtfd(:ncol,:)  = qt(:ncol,:)
   696)            ufd(:ncol,:)   = u(:ncol,:)
   697)            vfd(:ncol,:)   = v(:ncol,:)
   698) 
   699)          !------------------------------------------------------------------------ 
   700)          !  Check to see if constituent dependent gas constant needed (WACCM-X)
   701)          !------------------------------------------------------------------------
   702)          if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   703)            rairi(:ncol,1) = rairv(:ncol,1,lchnk)
   704)            do k = 2, pver
   705)              do i = 1, ncol
   706)                rairi(i,k) = 0.5_r8 * (rairv(i,k,lchnk)+rairv(i,k-1,lchnk))
   707)              end do
   708)            end do      
   709)          else
   710)            rairi(:ncol,:pver+1) = rair          
   711)          endif
   712) 
   713)          ! Diffuse initial profile of each time step using a given (kvh_out,kvm_out)
   714)          ! In the below 'compute_vdiff', (slfd,qtfd,ufd,vfd) are 'inout' variables.
   715) 
   716)          call compute_vdiff( lchnk   ,                                                  &
   717)                              pcols   , pver     , 1        , ncol         , pmid      , &
   718)                              pi      , rpdel    , t        , ztodt        , taux      , &
   719)                              tauy    , shflx    , qflx     , ntop_turb    , nbot_turb , &
   720)                              kvh_out , kvm_out  , kvh_out  , cgs          , cgh       , &
   721)                              zi      , ksrftms  , zero     , fieldlist_wet, fieldlist_molec, &
   722)                              ufd     , vfd      , qtfd     , slfd         ,             &
   723)                              jnk1d   , jnk1d    , jnk2d    , jnk1d        , errstring , &
   724)                              tauresx , tauresy  , 0        , cpairv(:,:,lchnk), rairi , .false. )
   725) 
   726)          call handle_errmsg(errstring, subname="compute_vdiff", &
   727)               extra_msg="compute_vdiff called from eddy_diff")
   728) 
   729)          ! Retrieve (tfd,qvfd,qlfd) from (slfd,qtfd) in order to 
   730)          ! use 'trbintd' at the next iteration.
   731)           
   732)           do k = 1, pver
   733)              do i = 1, ncol
   734)               ! ----------------------------------------------------- ! 
   735)               ! Compute the condensate 'qlfd' in the updated profiles !
   736)               ! ----------------------------------------------------- !  
   737)               ! Option.1 : Assume grid-mean condensate is homogeneously diffused by the moist turbulence scheme.
   738)               !            This should bs used if 'pseudodiff = .false.' in vertical_diffusion.F90.
   739)               ! Modification : Need to be check whether below is correct in the presence of ice, qi.       
   740)               !                I should understand why the variation of ice, qi is neglected during diffusion.
   741)                 templ     = ( slfd(i,k) - g*z(i,k) ) / cpair
   742)                 call qsat( templ, pmid(i,k), es, qs)
   743)                 ep2       =  .622_r8 
   744)                 temps     =   templ + ( qtfd(i,k) - qs ) / ( cpair / latvap + latvap * qs / ( rair * templ**2 ) )
   745)                 call qsat( temps, pmid(i,k), es, qs)
   746)                 qlfd(i,k) =   max( qtfd(i,k) - qi(i,k) - qs ,0._r8 )
   747)               ! Option.2 : Assume condensate is not diffused by the moist turbulence scheme. 
   748)               !            This should bs used if 'pseudodiff = .true.'  in vertical_diffusion.F90.       
   749)               ! qlfd(i,k) = ql(i,k)
   750)               ! ----------------------------- !
   751)               ! Compute the other 'qvfd, tfd' ! 
   752)               ! ----------------------------- !
   753)                 qvfd(i,k) = max( 0._r8, qtfd(i,k) - qi(i,k) - qlfd(i,k) )
   754)                 tfd(i,k)  = ( slfd(i,k) + latvap * qlfd(i,k) + latsub * qi(i,k) - g*z(i,k)) / cpair
   755)              end do
   756)           end do
   757)        endif
   758) 
   759)      ! Debug 
   760)      ! icol = phys_debug_col(lchnk) 
   761)      ! if( icol > 0 .and. get_nstep() .ge. 1 ) then
   762)      !     write(iulog,*) ' '
   763)      !     write(iulog,*) 'eddy_diff debug at the end of iteration' 
   764)      !     write(iulog,*) 't,     qv,     ql,     cld,     u,     v'
   765)      !     do k = pver-3, pver
   766)      !        write (iulog,*) k, tfd(icol,k), qvfd(icol,k), qlfd(icol,k), cldn(icol,k), ufd(icol,k), vfd(icol,k)
   767)      !     end do
   768)      ! endif
   769)      ! Debug
   770) 
   771)     end do  ! End of 'iturb' iteration
   772) 
   773)     kvq(:ncol,:) = kvh_out(:ncol,:)
   774) 
   775)   ! Compute 'wstar' within the PBL for use in the future convection scheme.
   776) 
   777)     do i = 1, ncol
   778)        if(ipbl(i) .eq. 1) then 
   779)            wstarPBL(i) = max( 0._r8, wstar(i,1) )
   780)        else
   781)            wstarPBL(i) = 0._r8
   782)        endif
   783)     end do
   784) 
   785)     ! --------------------------------------------------------------- !
   786)     ! Writing for detailed diagnostic analysis of UW moist PBL scheme !
   787)     ! --------------------------------------------------------------- !
   788) 
   789)     call outfld( 'UW_errorPBL',    errorPBL,   pcols,   lchnk )
   790) 
   791)     call outfld( 'UW_n2',          n2,         pcols,   lchnk )
   792)     call outfld( 'UW_s2',          s2,         pcols,   lchnk )
   793)     call outfld( 'UW_ri',          ri,         pcols,   lchnk )
   794) 
   795)     call outfld( 'UW_sfuh',        sfuh,       pcols,   lchnk )
   796)     call outfld( 'UW_sflh',        sflh,       pcols,   lchnk )
   797)     call outfld( 'UW_sfi',         sfi,        pcols,   lchnk )
   798) 
   799)     call outfld( 'UW_cldn',        cldn,       pcols,   lchnk )
   800)     call outfld( 'UW_qrl',         qrl,        pcols,   lchnk )
   801)     call outfld( 'UW_ql',          qlfd,       pcols,   lchnk )
   802) 
   803)     call outfld( 'UW_chu',         chu,        pcols,   lchnk )
   804)     call outfld( 'UW_chs',         chs,        pcols,   lchnk )
   805)     call outfld( 'UW_cmu',         cmu,        pcols,   lchnk )
   806)     call outfld( 'UW_cms',         cms,        pcols,   lchnk )
   807) 
   808)     call outfld( 'UW_tke',         tke,        pcols,   lchnk )
   809)     call outfld( 'UW_wcap',        wcap,       pcols,   lchnk )
   810)     call outfld( 'UW_bprod',       bprod,      pcols,   lchnk )
   811)     call outfld( 'UW_sprod',       sprod,      pcols,   lchnk )
   812) 
   813)     call outfld( 'UW_kvh',         kvh_out,    pcols,   lchnk )
   814)     call outfld( 'UW_kvm',         kvm_out,    pcols,   lchnk )
   815) 
   816)     call outfld( 'UW_pblh',        pblh,       pcols,   lchnk )
   817)     call outfld( 'UW_pblhp',       pblhp,      pcols,   lchnk )
   818)     call outfld( 'UW_tpert',       tpert,      pcols,   lchnk )
   819)     call outfld( 'UW_qpert',       qpert,      pcols,   lchnk )
   820)     call outfld( 'UW_wpert',       wpert,      pcols,   lchnk )
   821) 
   822)     call outfld( 'UW_ustar',       ustar,      pcols,   lchnk )
   823)     call outfld( 'UW_tkes',        tkes,       pcols,   lchnk )
   824)     call outfld( 'UW_minpblh',     minpblh,    pcols,   lchnk )
   825) 
   826)     call outfld( 'UW_turbtype',    real(turbtype,r8),   pcols,   lchnk )
   827)

   829)     call outfld( 'UW_ktop_o',      ktop_o,     pcols,   lchnk )

   837)     call outfld( 'UW_ktop_f',      ktop_f,     pcols,   lchnk )
   838)     call outfld( 'UW_ncvfin_f',    ncvfin_f,   pcols,   lchnk ) 
   839) 
   840)     call outfld( 'UW_wet',         wet,        pcols,   lchnk )
   841)     call outfld( 'UW_web',         web,        pcols,   lchnk )

   878)   subroutine sfdiag( pcols   , pver    , ncol    , qt      , ql      , sl      , &
   879)                      pi      , pm      , zi      , cld     , sfi     , sfuh    , &
   880)                      sflh    , slslope , qtslope )
   881)     !----------------------------------------------------------------------- ! 
   882)     !                                                                        !
   883)     ! Purpose: Interface for calculating saturation fractions  at upper and  ! 
   884)     !          lower-half layers, & interfaces for use by turbulence scheme  !
   885)     !                                                                        !
   886)     ! Method : Various but 'l' should be chosen for consistency.             !
   887)     !                                                                        ! 
   888)     ! Author : B. Stevens and C. Bretherton (August 2000)                    !
   889)     !          Sungsu Park. August 2006.                                     !
   890)     !                       May.   2008.                                     ! 
   891)     !                                                                        !  
   892)     ! S.Park : The computed saturation fractions are repeatedly              !
   893)     !          used to compute buoyancy coefficients in'trbintd' & 'caleddy'.!  
   894)     !----------------------------------------------------------------------- !
   895) 
   896)     implicit none       
   897) 
   898)     ! --------------- !
   899)     ! Input arguments !
   900)     ! --------------- !
   901) 
   902)     integer,  intent(in)  :: pcols               ! Number of atmospheric columns   
   903)     integer,  intent(in)  :: pver                ! Number of atmospheric layers   
   904)     integer,  intent(in)  :: ncol                ! Number of atmospheric columns   
   905) 
   906)     real(r8), intent(in)  :: sl(pcols,pver)      ! Liquid water static energy [ J/kg ]
   907)     real(r8), intent(in)  :: qt(pcols,pver)      ! Total water specific humidity [ kg/kg ]
   908)     real(r8), intent(in)  :: ql(pcols,pver)      ! Liquid water specific humidity [ kg/kg ]
   909)     real(r8), intent(in)  :: pi(pcols,pver+1)    ! Interface pressures [ Pa ]
   910)     real(r8), intent(in)  :: pm(pcols,pver)      ! Layer mid-point pressures [ Pa ]
   911)     real(r8), intent(in)  :: zi(pcols,pver+1)    ! Interface heights [ m ]
   912)     real(r8), intent(in)  :: cld(pcols,pver)     ! Stratiform cloud fraction [ fraction ]
   913)     real(r8), intent(in)  :: slslope(pcols,pver) ! Slope of 'sl' in each layer
   914)     real(r8), intent(in)  :: qtslope(pcols,pver) ! Slope of 'qt' in each layer
   915) 
   916)     ! ---------------- !
   917)     ! Output arguments !
   918)     ! ---------------- !
   919) 
   920)     real(r8), intent(out) :: sfi(pcols,pver+1)   ! Interfacial layer saturation fraction [ fraction ]
   921)     real(r8), intent(out) :: sfuh(pcols,pver)    ! Saturation fraction in upper half-layer [ fraction ]
   922)     real(r8), intent(out) :: sflh(pcols,pver)    ! Saturation fraction in lower half-layer [ fraction ]
   923) 
   924)     ! --------------- !
   925)     ! Local Variables !
   926)     ! --------------- !
   927) 
   928)     integer               :: i                   ! Longitude index
   929)     integer               :: k                   ! Vertical index
   930)     integer               :: km1                 ! k-1
   931)     integer               :: status              ! Status returned by function calls
   932)     real(r8)              :: sltop, slbot        ! sl at top/bot of grid layer
   933)     real(r8)              :: qttop, qtbot        ! qt at top/bot of grid layer
   934)     real(r8)              :: tltop, tlbot  ! Liquid water temperature at top/bot of grid layer
   935)     real(r8)              :: qxtop, qxbot        ! Sat excess at top/bot of grid layer
   936)     real(r8)              :: qxm                 ! Sat excess at midpoint
   937)     real(r8)              :: es               ! Saturation vapor pressure
   938)     real(r8)              :: qs               ! Saturation spec. humidity
   939)     real(r8)              :: cldeff(pcols,pver)  ! Effective Cloud Fraction [ fraction ]
   940) 
   941)     ! ----------------------- !
   942)     ! Main Computation Begins ! 
   943)     ! ----------------------- !
   944) 
   945)     sfi(1:ncol,:)    = 0._r8
   946)     sfuh(1:ncol,:)   = 0._r8
   947)     sflh(1:ncol,:)   = 0._r8
   948)     cldeff(1:ncol,:) = 0._r8
   949) 
   950)     select case (sftype)
   951)     case ('d')
   952)        ! ----------------------------------------------------------------------- !
   953)        ! Simply use the given stratus fraction ('horizontal' cloud partitioning) !
   954)        ! ----------------------------------------------------------------------- !
   955)        do k = ntop_turb + 1, nbot_turb
   956)           km1 = k - 1
   957)           do i = 1, ncol
   958)              sfuh(i,k) = cld(i,k)
   959)              sflh(i,k) = cld(i,k)
   960)              sfi(i,k)  = 0.5_r8 * ( sflh(i,km1) + min( sflh(i,km1), sfuh(i,k) ) )
   961)           end do
   962)        end do
   963)        do i = 1, ncol
   964)           sfi(i,pver+1) = sflh(i,pver) 
   965)        end do
   966)     case ('l')
   967)        ! ------------------------------------------ !
   968)        ! Use modified stratus fraction partitioning !
   969)        ! ------------------------------------------ !
   970)        do k = ntop_turb + 1, nbot_turb
   971)           km1 = k - 1
   972)           do i = 1, ncol
   973)              cldeff(i,k) = cld(i,k)
   974)              sfuh(i,k)   = cld(i,k)
   975)              sflh(i,k)   = cld(i,k)
   976)              if( ql(i,k) .lt. qmin ) then
   977)                  sfuh(i,k) = 0._r8
   978)                  sflh(i,k) = 0._r8
   979)              end if
   980)            ! Modification : The contribution of ice should be carefully considered.
   981)              if( choice_evhc .eq. 'ramp' .or. choice_radf .eq. 'ramp' ) then 
   982)                  cldeff(i,k) = cld(i,k) * min( ql(i,k) / qmin, 1._r8 )
   983)                  sfuh(i,k)   = cldeff(i,k)
   984)                  sflh(i,k)   = cldeff(i,k)
   985)              elseif( choice_evhc .eq. 'maxi' .or. choice_radf .eq. 'maxi' ) then 
   986)                  cldeff(i,k) = cld(i,k)
   987)                  sfuh(i,k)   = cldeff(i,k)
   988)                  sflh(i,k)   = cldeff(i,k)
   989)              endif
   990)            ! At the stratus top, take the minimum interfacial saturation fraction
   991)              sfi(i,k) = 0.5_r8 * ( sflh(i,km1) + min( sfuh(i,k), sflh(i,km1) ) )
   992)            ! Modification : Currently sfi at the top and surface interfaces are set to be zero.
   993)            !                Also, sfuh and sflh in the top model layer is set to be zero.
   994)            !                However, I may need to set 
   995)            !                         do i = 1, ncol
   996)            !                            sfi(i,pver+1) = sflh(i,pver) 
   997)            !                         end do
   998)            !                for treating surface-based fog. 
   999)            ! OK. I added below block similar to the other cases.
  1000)           end do
  1001)        end do
  1002)        do i = 1, ncol
  1003)           sfi(i,pver+1) = sflh(i,pver)
  1004)        end do
  1005)     case ('u')
  1006)        ! ------------------------------------------------------------------------- !
  1007)        ! Use unsaturated buoyancy - since sfi, sfuh, sflh have already been zeroed !
  1008)        ! nothing more need be done for this case.                                  !
  1009)        ! ------------------------------------------------------------------------- !
  1010)     case ('z')
  1011)        ! ------------------------------------------------------------------------- !
  1012)        ! Calculate saturation fraction based on whether the air just above or just !
  1013)        ! below the interface is saturated, i.e. with vertical cloud partitioning.  !
  1014)        ! The saturation fraction of the interfacial layer between mid-points k and !
  1015)        ! k+1 is computed by averaging the saturation fraction   of the half-layers !
  1016)        ! above and below the interface,  with a special provision   for cloud tops !
  1017)        ! (more cloud in the half-layer below than in the half-layer above).In each !
  1018)        ! half-layer, vertical partitioning of  cloud based on the slopes diagnosed !
  1019)        ! above is used.     Loop down through the layers, computing the saturation !
  1020)        ! fraction in each half-layer (sfuh for upper half, sflh for lower half).   !
  1021)        ! Once sfuh(i,k) is computed, use with sflh(i,k-1) to determine  saturation !
  1022)        ! fraction sfi(i,k) for interfacial layer k-0.5.                            !
  1023)        ! This is 'not' chosen for full consistent treatment of stratus fraction in !
  1024)        ! all physics schemes.                                                      !
  1025)        ! ------------------------------------------------------------------------- !
  1026)        do k = ntop_turb + 1, nbot_turb
  1027)           km1 = k - 1
  1028)           do i = 1, ncol
  1029)            ! Compute saturation excess at the mid-point of layer k
  1030)              sltop    = sl(i,k) + slslope(i,k) * ( pi(i,k) - pm(i,k) )      
  1031)              qttop    = qt(i,k) + qtslope(i,k) * ( pi(i,k) - pm(i,k) )
  1032)              tltop = ( sltop - g * zi(i,k) ) / cpair 
  1033)              call qsat( tltop, pi(i,k), es, qs)
  1034)              qxtop    = qttop - qs
  1035)              slbot    = sl(i,k) + slslope(i,k) * ( pi(i,k+1) - pm(i,k) )      
  1036)              qtbot    = qt(i,k) + qtslope(i,k) * ( pi(i,k+1) - pm(i,k) )
  1037)              tlbot = ( slbot - g * zi(i,k+1) ) / cpair 
  1038)              call qsat( tlbot, pi(i,k+1), es, qs)
  1039)              qxbot    = qtbot - qs
  1040)              qxm      = qxtop + ( qxbot - qxtop ) * ( pm(i,k) - pi(i,k) ) / ( pi(i,k+1) - pi(i,k) )
  1041)            ! Find the saturation fraction sfuh(i,k) of the upper half of layer k.
  1042)              if( ( qxtop .lt. 0._r8 ) .and. ( qxm .lt. 0._r8 ) ) then
  1043)                    sfuh(i,k) = 0._r8 
  1044)              else if( ( qxtop .gt. 0._r8 ) .and. ( qxm .gt. 0._r8 ) ) then
  1045)                    sfuh(i,k) = 1._r8  
  1046)              else ! Either qxm < 0 and qxtop > 0 or vice versa
  1047)                    sfuh(i,k) = max( qxtop, qxm ) / abs( qxtop - qxm )
  1048)              end if
  1049)            ! Combine with sflh(i) (still for layer k-1) to get interfac layer saturation fraction
  1050)              sfi(i,k) = 0.5_r8 * ( sflh(i,k-1) + min( sflh(i,k-1), sfuh(i,k) ) )
  1051)            ! Update sflh to be for the lower half of layer k.             
  1052)              if( ( qxbot .lt. 0._r8 ) .and. ( qxm .lt. 0._r8 ) ) then
  1053)                    sflh(i,k) = 0._r8 
  1054)              else if( ( qxbot .gt. 0._r8 ) .and. ( qxm .gt. 0._r8 ) ) then
  1055)                    sflh(i,k) = 1._r8 
  1056)              else ! Either qxm < 0 and qxbot > 0 or vice versa
  1057)                    sflh(i,k) = max( qxbot, qxm ) / abs( qxbot - qxm )
  1058)              end if
  1059)           end do  ! i
  1060)        end do ! k
  1061)        do i = 1, ncol
  1062)           sfi(i,pver+1) = sflh(i,pver)  ! Saturation fraction in the lowest half-layer. 
  1063)        end do
  1064)     end select
  1065) 
  1066)   return
  1067)   end subroutine sfdiag

  1072)  
  1073)   subroutine trbintd( pcols   , pver    , ncol    ,                               &
  1074)                       z       , u       , v       ,                               &
  1075)                       t       , pmid    ,                                         &
  1076)                       s2      , n2      , ri      ,                               &
  1077)                       zi      , pi      , cld     ,                               &
  1078)                       qt      , qv      , ql      , qi      , sfi     , sfuh    , &
  1079)                       sflh    , sl      , slv     , slslope , qtslope ,           &
  1080)                       chs     , chu     , cms     , cmu     )
  1081)     !----------------------------------------------------------------------- !
  1082)     ! Purpose: Calculate buoyancy coefficients at all interfaces including   !
  1083)     !          surface. Also, computes the profiles of ( sl,qt,n2,s2,ri ).   !
  1084)     !          Note that (n2,s2,ri) are defined at each interfaces except    !
  1085)     !          surface.                                                      !
  1086)     !                                                                        !
  1087)     ! Author: B. Stevens  ( Extracted from pbldiff, August, 2000 )           !
  1088)     !         Sungsu Park ( August 2006, May. 2008 )                         !
  1089)     !----------------------------------------------------------------------- !
  1090) 
  1091)     implicit none
  1092) 
  1093)     ! --------------- !
  1094)     ! Input arguments !
  1095)     ! --------------- !
  1096) 
  1097)     integer,  intent(in)  :: pcols                            ! Number of atmospheric columns   
  1098)     integer,  intent(in)  :: pver                             ! Number of atmospheric layers   
  1099)     integer,  intent(in)  :: ncol                             ! Number of atmospheric columns
  1100)     real(r8), intent(in)  :: z(pcols,pver)                    ! Layer mid-point height above surface [ m ]
  1101)     real(r8), intent(in)  :: u(pcols,pver)                    ! Layer mid-point u [ m/s ]
  1102)     real(r8), intent(in)  :: v(pcols,pver)                    ! Layer mid-point v [ m/s ]
  1103)     real(r8), intent(in)  :: t(pcols,pver)                    ! Layer mid-point temperature [ K ]
  1104)     real(r8), intent(in)  :: pmid(pcols,pver)                 ! Layer mid-point pressure [ Pa ]
  1105)     real(r8), intent(in)  :: zi(pcols,pver+1)                 ! Interface height [ m ]
  1106)     real(r8), intent(in)  :: pi(pcols,pver+1)                 ! Interface pressure [ Pa ]
  1107)     real(r8), intent(in)  :: cld(pcols,pver)                  ! Stratus fraction
  1108)     real(r8), intent(in)  :: qv(pcols,pver)                   ! Water vapor specific humidity [ kg/kg ]
  1109)     real(r8), intent(in)  :: ql(pcols,pver)                   ! Liquid water specific humidity [ kg/kg ]
  1110)     real(r8), intent(in)  :: qi(pcols,pver)                   ! Ice water specific humidity [ kg/kg ]
  1111) 
  1112)     ! ---------------- !
  1113)     ! Output arguments !
  1114)     ! ---------------- !
  1115) 
  1116)     real(r8), intent(out) :: s2(pcols,pver)                   ! Interfacial ( except surface ) shear squared [ s-2 ]
  1117)     real(r8), intent(out) :: n2(pcols,pver)                   ! Interfacial ( except surface ) buoyancy frequency [ s-2 ]
  1118)     real(r8), intent(out) :: ri(pcols,pver)                   ! Interfacial ( except surface ) Richardson number, 'n2/s2'
  1119)  
  1120)     real(r8), intent(out) :: qt(pcols,pver)                   ! Total specific humidity [ kg/kg ]
  1121)     real(r8), intent(out) :: sfi(pcols,pver+1)                ! Interfacial layer saturation fraction [ fraction ]
  1122)     real(r8), intent(out) :: sfuh(pcols,pver)                 ! Saturation fraction in upper half-layer [ fraction ]
  1123)     real(r8), intent(out) :: sflh(pcols,pver)                 ! Saturation fraction in lower half-layer [ fraction ]
  1124)     real(r8), intent(out) :: sl(pcols,pver)                   ! Liquid water static energy [ J/kg ] 
  1125)     real(r8), intent(out) :: slv(pcols,pver)                  ! Liquid water virtual static energy [ J/kg ]
  1126)    
  1127)     real(r8), intent(out) :: chu(pcols,pver+1)                ! Heat buoyancy coef for dry states at all interfaces, finally.
  1128)                                                               ! [ unit ? ]
  1129)     real(r8), intent(out) :: chs(pcols,pver+1)                ! heat buoyancy coef for sat states at all interfaces, finally.
  1130)                                                               ! [ unit ? ]
  1131)     real(r8), intent(out) :: cmu(pcols,pver+1)                ! Moisture buoyancy coef for dry states at all interfaces, finally.
  1132)                                                               ! [ unit ? ]
  1133)     real(r8), intent(out) :: cms(pcols,pver+1)                ! Moisture buoyancy coef for sat states at all interfaces, finally.
  1134)                                                               ! [ unit ? ]
  1135)     real(r8), intent(out) :: slslope(pcols,pver)              ! Slope of 'sl' in each layer
  1136)     real(r8), intent(out) :: qtslope(pcols,pver)              ! Slope of 'qt' in each layer
  1137)  
  1138)     ! --------------- !
  1139)     ! Local Variables !
  1140)     ! --------------- ! 
  1141) 
  1142)     integer               :: i                                ! Longitude index
  1143)     integer               :: k, km1                           ! Level index
  1144)     integer               :: status                           ! Status returned by function calls
  1145) 
  1146)     real(r8)              :: qs(pcols,pver)                   ! Saturation specific humidity
  1147)     real(r8)              :: es(pcols,pver)                   ! Saturation vapor pressure
  1148)     real(r8)              :: gam(pcols,pver)                  ! (l/cp)*(d(qs)/dT)
  1149)     real(r8)              :: rdz                              ! 1 / (delta z) between midpoints
  1150)     real(r8)              :: dsldz                            ! 'delta sl / delta z' at interface
  1151)     real(r8)              :: dqtdz                            ! 'delta qt / delta z' at interface
  1152)     real(r8)              :: ch                               ! 'sfi' weighted ch at the interface
  1153)     real(r8)              :: cm                               ! 'sfi' weighted cm at the interface
  1154)     real(r8)              :: bfact                            ! Buoyancy factor in n2 calculations
  1155)     real(r8)              :: product                          ! Intermediate vars used to find slopes
  1156)     real(r8)              :: dsldp_a, dqtdp_a                 ! Slopes across interface above 
  1157)     real(r8)              :: dsldp_b(pcols), dqtdp_b(pcols)   ! Slopes across interface below
  1158) 
  1159)     ! ----------------------- !
  1160)     ! Main Computation Begins !
  1161)     ! ----------------------- !
  1162) 
  1163)     ! Calculate conservative scalars (qt,sl,slv) and buoyancy coefficients at the layer mid-points.
  1164)     ! Note that 'ntop_turb = 1', 'nbot_turb = pver'
  1165) 
  1166)     do k = ntop_turb, nbot_turb
  1167)        call qsat( t(:ncol,k), pmid(:ncol,k), es(:ncol,k), qs(:ncol,k), gam=gam(:ncol,k))
  1168)        do i = 1, ncol
  1169)           qt(i,k)  = qv(i,k) + ql(i,k) + qi(i,k) 
  1170)           sl(i,k)  = cpair * t(i,k) + g * z(i,k) - latvap * ql(i,k) - latsub * qi(i,k)
  1171)           slv(i,k) = sl(i,k) * ( 1._r8 + zvir * qt(i,k) )
  1172)         ! Thermodynamic coefficients for buoyancy flux - in this loop these are
  1173)         ! calculated at mid-points; later,  they will be averaged to interfaces,
  1174)         ! where they will ultimately be used.  At the surface, the coefficients
  1175)         ! are taken from the lowest mid point.
  1176)           bfact    = g / ( t(i,k) * ( 1._r8 + zvir * qv(i,k) - ql(i,k) - qi(i,k) ) )
  1177)           chu(i,k) = ( 1._r8 + zvir * qt(i,k) ) * bfact / cpair
  1178)           chs(i,k) = ( ( 1._r8 + ( 1._r8 + zvir ) * gam(i,k) * cpair * t(i,k) / latvap ) / ( 1._r8 + gam(i,k) ) ) * bfact / cpair
  1179)           cmu(i,k) = zvir * bfact * t(i,k)
  1180)           cms(i,k) = latvap * chs(i,k)  -  bfact * t(i,k)
  1181)        end do
  1182)     end do
  1183) 
  1184)     do i = 1, ncol
  1185)        chu(i,pver+1) = chu(i,pver)
  1186)        chs(i,pver+1) = chs(i,pver)
  1187)        cmu(i,pver+1) = cmu(i,pver)
  1188)        cms(i,pver+1) = cms(i,pver)
  1189)     end do
  1190) 
  1191)     ! Compute slopes of conserved variables sl, qt within each layer k. 
  1192)     ! 'a' indicates the 'above' gradient from layer k-1 to layer k and 
  1193)     ! 'b' indicates the 'below' gradient from layer k   to layer k+1.
  1194)     ! We take a smaller (in absolute value)  of these gradients as the
  1195)     ! slope within layer k. If they have opposite signs,   gradient in 
  1196)     ! layer k is taken to be zero. I should re-consider whether   this
  1197)     ! profile reconstruction is the best or not.
  1198)     ! This is similar to the profile reconstruction used in the UWShCu. 
  1199) 
  1200)     do i = 1, ncol
  1201)      ! Slopes at endpoints determined by extrapolation
  1202)        slslope(i,pver) = ( sl(i,pver) - sl(i,pver-1) ) / ( pmid(i,pver) - pmid(i,pver-1) )
  1203)        qtslope(i,pver) = ( qt(i,pver) - qt(i,pver-1) ) / ( pmid(i,pver) - pmid(i,pver-1) )
  1204)        slslope(i,1)    = ( sl(i,2) - sl(i,1) ) / ( pmid(i,2) - pmid(i,1) )
  1205)        qtslope(i,1)    = ( qt(i,2) - qt(i,1) ) / ( pmid(i,2) - pmid(i,1) )
  1206)        dsldp_b(i)      = slslope(i,1)
  1207)        dqtdp_b(i)      = qtslope(i,1)
  1208)     end do
  1209) 
  1210)     do k = 2, pver - 1
  1211)        do i = 1, ncol
  1212)           dsldp_a    = dsldp_b(i)
  1213)           dqtdp_a    = dqtdp_b(i)
  1214)           dsldp_b(i) = ( sl(i,k+1) - sl(i,k) ) / ( pmid(i,k+1) - pmid(i,k) )
  1215)           dqtdp_b(i) = ( qt(i,k+1) - qt(i,k) ) / ( pmid(i,k+1) - pmid(i,k) )
  1216)           product    = dsldp_a * dsldp_b(i)
  1217)           if( product .le. 0._r8 ) then 
  1218)               slslope(i,k) = 0._r8
  1219)           else if( product .gt. 0._r8 .and. dsldp_a .lt. 0._r8 ) then 
  1220)               slslope(i,k) = max( dsldp_a, dsldp_b(i) )
  1221)           else if( product .gt. 0._r8 .and. dsldp_a .gt. 0._r8 ) then 
  1222)               slslope(i,k) = min( dsldp_a, dsldp_b(i) )
  1223)           end if
  1224)           product = dqtdp_a*dqtdp_b(i)
  1225)           if( product .le. 0._r8 ) then 
  1226)               qtslope(i,k) = 0._r8
  1227)           else if( product .gt. 0._r8 .and. dqtdp_a .lt. 0._r8 ) then 
  1228)               qtslope(i,k) = max( dqtdp_a, dqtdp_b(i) )
  1229)           else if( product .gt. 0._r8 .and. dqtdp_a .gt. 0._r8 ) then 
  1230)               qtslope(i,k) = min( dqtdp_a, dqtdp_b(i) )
  1231)           end if
  1232)        end do ! i
  1233)     end do ! k
  1234) 
  1235)     !  Compute saturation fraction at the interfacial layers for use in buoyancy
  1236)     !  flux computation.
  1237) 
  1238)     call sfdiag( pcols  , pver    , ncol    , qt      , ql      , sl      , & 
  1239)                  pi     , pmid    , zi      , cld     , sfi     , sfuh    , &
  1240)                  sflh   , slslope , qtslope )
  1241) 
  1242)     ! Calculate buoyancy coefficients at all interfaces (1:pver+1) and (n2,s2,ri) 
  1243)     ! at all interfaces except surface. Note 'nbot_turb = pver', 'ntop_turb = 1'.
  1244)     ! With the previous definition of buoyancy coefficients at the surface, the 
  1245)     ! resulting buoyancy coefficients at the top and surface interfaces becomes 
  1246)     ! identical to the buoyancy coefficients at the top and bottom layers. Note 
  1247)     ! that even though the dimension of (s2,n2,ri) is 'pver',  they are defined
  1248)     ! at interfaces ( not at the layer mid-points ) except the surface. 
  1249) 
  1250)     do k = nbot_turb, ntop_turb + 1, -1
  1251)        km1 = k - 1
  1252)        do i = 1, ncol
  1253)           rdz      = 1._r8 / ( z(i,km1) - z(i,k) )
  1254)           dsldz    = ( sl(i,km1) - sl(i,k) ) * rdz
  1255)           dqtdz    = ( qt(i,km1) - qt(i,k) ) * rdz 
  1256)           chu(i,k) = ( chu(i,km1) + chu(i,k) ) * 0.5_r8
  1257)           chs(i,k) = ( chs(i,km1) + chs(i,k) ) * 0.5_r8
  1258)           cmu(i,k) = ( cmu(i,km1) + cmu(i,k) ) * 0.5_r8
  1259)           cms(i,k) = ( cms(i,km1) + cms(i,k) ) * 0.5_r8
  1260)           ch       = chu(i,k) * ( 1._r8 - sfi(i,k) ) + chs(i,k) * sfi(i,k)
  1261)           cm       = cmu(i,k) * ( 1._r8 - sfi(i,k) ) + cms(i,k) * sfi(i,k)
  1262)           n2(i,k)  = ch * dsldz +  cm * dqtdz
  1263)           s2(i,k)  = ( ( u(i,km1) - u(i,k) )**2 + ( v(i,km1) - v(i,k) )**2) * rdz**2
  1264)           s2(i,k)  = max( ntzero, s2(i,k) )
  1265)           ri(i,k)  = n2(i,k) / s2(i,k)
  1266)        end do
  1267)     end do 
  1268)     do i = 1, ncol
  1269)        n2(i,1) = n2(i,2)
  1270)        s2(i,1) = s2(i,2)
  1271)        ri(i,1) = ri(i,2)
  1272)     end do
  1273) 
  1274)   return
  1275) 
  1276)   end subroutine trbintd

  1282)   subroutine austausch_atm( pcols, pver, ncol, ri, s2, kvf )
  1283) 
  1284)     !---------------------------------------------------------------------- ! 
  1285)     !                                                                       !
  1286)     ! Purpose: Computes exchange coefficients for free turbulent flows.     !
  1287)     !          This is not used in the UW moist turbulence scheme.          !
  1288)     !                                                                       !
  1289)     ! Method:                                                               !
  1290)     !                                                                       !
  1291)     ! The free atmosphere diffusivities are based on standard mixing length !
  1292)     ! forms for the neutral diffusivity multiplied by functns of Richardson !
  1293)     ! number. K = l^2 * |dV/dz| * f(Ri). The same functions are used for    !
  1294)     ! momentum, potential temperature, and constitutents.                   !
  1295)     !                                                                       !
  1296)     ! The stable Richardson num function (Ri>0) is taken from Holtslag and  !
  1297)     ! Beljaars (1989), ECMWF proceedings. f = 1 / (1 + 10*Ri*(1 + 8*Ri))    !
  1298)     ! The unstable Richardson number function (Ri<0) is taken from  CCM1.   !
  1299)     ! f = sqrt(1 - 18*Ri)                                                   !
  1300)     !                                                                       !
  1301)     ! Author: B. Stevens (rewrite, August 2000)                             !
  1302)     !                                                                       !
  1303)     !---------------------------------------------------------------------- !
  1304)     implicit none
  1305)     
  1306)     ! --------------- ! 
  1307)     ! Input arguments !
  1308)     ! --------------- !
  1309) 
  1310)     integer,  intent(in)  :: pcols                ! Number of atmospheric columns   
  1311)     integer,  intent(in)  :: pver                 ! Number of atmospheric layers   
  1312)     integer,  intent(in)  :: ncol                 ! Number of atmospheric columns
  1313) 
  1314)     real(r8), intent(in)  :: s2(pcols,pver)       ! Shear squared
  1315)     real(r8), intent(in)  :: ri(pcols,pver)       ! Richardson no
  1316) 
  1317)     ! ---------------- !
  1318)     ! Output arguments !
  1319)     ! ---------------- !
  1320) 
  1321)     real(r8), intent(out) :: kvf(pcols,pver+1)    ! Eddy diffusivity for heat and tracers
  1322) 
  1323)     ! --------------- !
  1324)     ! Local Variables !
  1325)     ! --------------- !
  1326) 
  1327)     real(r8)              :: fofri                ! f(ri)
  1328)     real(r8)              :: kvn                  ! Neutral Kv
  1329) 
  1330)     integer               :: i                    ! Longitude index
  1331)     integer               :: k                    ! Vertical index
  1332) 
  1333)     ! ----------------------- !
  1334)     ! Main Computation Begins !
  1335)     ! ----------------------- !
  1336) 
  1337)     kvf(:ncol,:)           = 0.0_r8
  1338)     kvf(:ncol,pver+1)      = 0.0_r8
  1339)     kvf(:ncol,1:ntop_turb) = 0.0_r8
  1340) 
  1341)     ! Compute the free atmosphere vertical diffusion coefficients: kvh = kvq = kvm. 
  1342) 
  1343)     do k = ntop_turb + 1, nbot_turb
  1344)        do i = 1, ncol
  1345)           if( ri(i,k) < 0.0_r8 ) then
  1346)               fofri = sqrt( max( 1._r8 - 18._r8 * ri(i,k), 0._r8 ) )
  1347)           else 
  1348)               fofri = 1.0_r8 / ( 1.0_r8 + 10.0_r8 * ri(i,k) * ( 1.0_r8 + 8.0_r8 * ri(i,k) ) )    
  1349)           end if
  1350)           kvn = ml2(k) * sqrt(s2(i,k))
  1351)           kvf(i,k) = max( zkmin, kvn * fofri )
  1352)        end do
  1353)     end do
  1354) 
  1355)     return
  1356) 
  1357)     end subroutine austausch_atm

  2395)                   jt2slv = slv(i,max(kt-2,1)) - slv(i,kt)
  2396)                   jt2slv = max(jt2slv, jbumin*slv(i,kt-1)/g)
  2397)                   evhc = 1._r8+max(cldeff(i,kt)-cldeff(i,kt-1),0._r8)*sedfact*a2l*a3l*latvap*ql(i,kt) / jt2slv
  2398)                   evhc = min(evhc,evhcmax)
  2399)               elseif( choice_evhc .eq. 'maxi' ) then
  2400)                   qleff  = max(ql(i,kt-1),ql(i,kt))
  2401)                   jt2slv = slv(i,max(kt-2,1)) - slv(i,kt)
  2402)                   jt2slv = max(jt2slv, jbumin*slv(i,kt-1)/g)
  2403)                   evhc = 1._r8+sedfact*a2l*a3l*latvap*qleff / jt2slv
  2404)                   evhc = min(evhc,evhcmax)
  2405)               endif
  2406)           endif
  2407) 
  2408)           ! -------------------------------------------------------------------------- !
  2409)           ! Now diagnose CL top and bottom entrainment rates (and the contribution of  !
  2410)           ! top/bottom entrainments to wstar3) using entrainment closures of the form  !
  2411)           !                                                                            !        
  2412)           !                   wet = cet*wstar3, web = ceb*wstar3                       !
  2413)           !                                                                            !
  2414)           ! where cet and ceb depend on the entrainment interface jumps, ql, etc.      !
  2415)           ! No entrainment is diagnosed unless the wstar3 > 0. Note '1/wstar3fact' is  !
  2416)           ! a factor indicating the enhancement of wstar3 due to entrainment process.  !
  2417)           ! Q : Below setting of 'wstar3fact = max(..,0.5)'might prevent the possible  !
  2418)           !     case when buoyancy consumption by entrainment is  stronger than cloud  !
  2419)           !     top radiative cooling production. Is that OK ? No.  According to bulk  !
  2420)           !     modeling study, entrainment buoyancy consumption was always a certain  !
  2421)           !     fraction of other net productions, rather than a separate sum.  Thus,  !
  2422)           !     below max limit of wstar3fact is correct.   'wstar3fact = max(.,0.5)'  !
  2423)           !     prevents unreasonable enhancement of CL entrainment rate by cloud-top  !
  2424)           !     entrainment instability, CTEI.                                         !
  2425)           ! Q : Use of the same dry entrainment coefficient, 'a1i' both at the CL  top !
  2426)           !     and base interfaces may result in too small 'wstar3' and 'ebrk' below, !
  2427)           !     as was seen in my generalized bulk modeling study. This should be re-  !
  2428)           !     considered later                                                       !
  2429)           ! -------------------------------------------------------------------------- !
  2430)           
  2431)           if( wstar3 .gt. 0._r8 ) then
  2432)               cet = a1i * evhc / ( jtbu * lbulk )
  2433)               if( kb .eq. pver + 1 ) then 
  2434)                   wstar3fact = max( 1._r8 + 2.5_r8 * cet * n2ht * jtzm * dzht, wstar3factcrit )
  2435)               else    
  2436)                   ceb = a1i / ( jbbu * lbulk )
  2437)                   wstar3fact = max( 1._r8 + 2.5_r8 * cet * n2ht * jtzm * dzht &
  2438)                                           + 2.5_r8 * ceb * n2hb * jbzm * dzhb, wstar3factcrit )
  2439)               end if
  2440)               wstar3 = wstar3 / wstar3fact       
  2441)           else ! wstar3 == 0
  2442)               wstar3fact = 0._r8 ! This is just for dianostic output
  2443)               cet        = 0._r8
  2444)               ceb        = 0._r8
  2445)           end if 
  2446) 
  2447)           ! ---------------------------------------------------------------------------- !
  2448)           ! Calculate net CL mean TKE including entrainment contribution by solving a    !
  2449)           ! canonical cubic equation. The solution of cubic equ. is 'rootp**2 = ebrk'    !
  2450)           ! where 'ebrk' originally (before solving cubic eq.) was interior CL mean TKE, !
  2451)           ! but after solving cubic equation,  it is replaced by net CL mean TKE in the  !
  2452)           ! same variable 'ebrk'.                                                        !
  2453)           ! ---------------------------------------------------------------------------- !
  2454)           ! Solve cubic equation (canonical form for analytic solution)                  !
  2455)           !   r^3 - 3*trmp*r - 2*trmq = 0,   r = sqrt<e>                                 ! 
  2456)           ! to estimate <e> for CL, derived from layer-mean TKE balance:                 !
  2457)           !                                                                              !
  2458)           !   <e>^(3/2)/(b_1*<l>) \approx <B + S>   (*)                                  !
  2459)           !   <B+S> = (<B+S>_int * l_int + <B+S>_et * dzt + <B+S>_eb * dzb)/lbulk        !
  2460)           !   <B+S>_int = <e>^(1/2)/(b_1*<l>)*<e>_int                                    !
  2461)           !   <B+S>_et  = (-vyt+vut)*wet*jtbu + radf                                     !
  2462)           !   <B+S>_eb  = (-vyb+vub)*web*jbbu                                            !
  2463)           !                                                                              !
  2464)           ! where:                                                                       !
  2465)           !   <> denotes a vertical avg (over the whole CL unless indicated)             !
  2466)           !   l_int (called lbrk below) is aggregate thickness of interior CL layers     !
  2467)           !   dzt = zi(i,kt)-z(i,kt)   is thickness of top entrainment layer             !
  2468)           !   dzb = z(i,kb-1)-zi(i,kb) is thickness of bot entrainment layer             !
  2469)           !   <e>_int (called ebrk below) is the CL-mean TKE if only interior            !
  2470)           !                               interfaces contributed.                        !
  2471)           !   wet, web                  are top. bottom entrainment rates                !
  2472)           !                                                                              !
  2473)           ! For a single-level radiatively-driven convective layer, there are no         ! 
  2474)           ! interior interfaces so 'ebrk' = 'lbrk' = 0. If the CL goes to the            !
  2475)           ! surface, 'vyb' and 'vub' are set to zero before and 'ebrk' and 'lbrk'        !
  2476)           ! have already incorporated the surface interfacial layer contribution,        !
  2477)           ! so the same formulas still apply.                                            !
  2478)           !                                                                              !
  2479)           ! In the original formulation based on TKE,                                    !
  2480)           !    wet*jtbu = a1l*evhc*<e>^3/2/leng(i,kt)                                    ! 
  2481)           !    web*jbbu = a1l*<e>^3/2/leng(i,kt)                                         !
  2482)           !                                                                              !
  2483)           ! In the wstar formulation                                                     !
  2484)           !    wet*jtbu = a1i*evhc*wstar3/lbulk                                          !
  2485)           !    web*jbbu = a1i*wstar3/lbulk,                                              !
  2486)           ! ---------------------------------------------------------------------------- !
  2487) 
  2488)           fact = ( evhc * ( -vyt + vut ) * dzht + ( -vyb + vub ) * dzhb * leng(i,kb) / leng(i,kt) ) / lbulk
  2489) 
  2490)           if( wstarent ) then
  2491) 
  2492)               ! (Option 1) 'wstar' entrainment formulation 
  2493)               ! Here trmq can have either sign, and will usually be nonzero even for non-
  2494)               ! cloud topped CLs.  If trmq > 0, there will be two positive roots r; we take 
  2495)               ! the larger one. Why ? If necessary, we limit entrainment and wstar to prevent
  2496)               ! a solution with r < ccrit*wstar ( Why ? ) where we take ccrit = 0.5. 
  2497) 
  2498)               trma = 1._r8          
  2499)               trmp = ebrk(i,ncv) * ( lbrk(i,ncv) / lbulk ) / 3._r8 + ntzero
  2500)               trmq = 0.5_r8 * b1 * ( leng(i,kt)  / lbulk ) * ( radf * dzht + a1i * fact * wstar3 )
  2501) 
  2502)               ! Check if there is an acceptable root with r > rcrit = ccrit*wstar. 
  2503)               ! To do this, first find local minimum fmin of the cubic f(r) at sqrt(p), 
  2504)               ! and value fcrit = f(rcrit).
  2505) 
  2506)               rmin  = sqrt(trmp)
  2507)               fmin  = rmin * ( rmin * rmin - 3._r8 * trmp ) - 2._r8 * trmq
  2508)               wstar = wstar3**onet
  2509)               rcrit = ccrit * wstar
  2510)               fcrit = rcrit * ( rcrit * rcrit - 3._r8 * trmp ) - 2._r8 * trmq
  2511) 
  2512)               ! No acceptable root exists (noroot = .true.) if either:
  2513)               !    1) rmin < rcrit (in which case cubic is monotone increasing for r > rcrit)
  2514)               !       and f(rcrit) > 0.
  2515)               ! or 2) rmin > rcrit (in which case min of f(r) in r > rcrit is at rmin)
  2516)               !       and f(rmin) > 0.  
  2517)               ! In this case, we reduce entrainment and wstar3 such that r/wstar = ccrit;
  2518)               ! this changes the coefficients of the cubic.   It might be informative to
  2519)               ! check when and how many 'noroot' cases occur,  since when 'noroot',   we
  2520)               ! will impose arbitrary limit on 'wstar3, wet, web, and ebrk' using ccrit.
  2521) 
  2522)               noroot = ( ( rmin .lt. rcrit ) .and. ( fcrit .gt. 0._r8 ) ) &
  2523)                   .or. ( ( rmin .ge. rcrit ) .and. ( fmin  .gt. 0._r8 ) )
  2524)               if( noroot ) then ! Solve cubic for r
  2525)                   trma = 1._r8 - b1 * ( leng(i,kt) / lbulk ) * a1i * fact / ccrit**3
  2526)                   trma = max( trma, 0.5_r8 )  ! Limit entrainment enhancement of ebrk
  2527)                   trmp = trmp / trma 
  2528)                   trmq = 0.5_r8 * b1 * ( leng(i,kt) / lbulk ) * radf * dzht / trma
  2529)               end if   ! noroot
  2530) 
  2531)               ! Solve the cubic equation
  2532) 
  2533)               qq = trmq**2 - trmp**3
  2534)               if( qq .ge. 0._r8 ) then 
  2535)                   rootp = ( trmq + sqrt(qq) )**(1._r8/3._r8) + ( max( trmq - sqrt(qq), 0._r8 ) )**(1._r8/3._r8)
  2536)               else
  2537)                   rootp = 2._r8 * sqrt(trmp) * cos( acos( trmq / sqrt(trmp**3) ) / 3._r8 )
  2538)               end if
  2539)  
  2540)               ! Adjust 'wstar3' only if there is 'noroot'. 
  2541)               ! And calculate entrainment rates at the top and base interfaces.
  2542) 
  2543)               if( noroot )  wstar3 = ( rootp / ccrit )**3     ! Adjust wstar3 
  2544)               wet = cet * wstar3                              ! Find entrainment rates
  2545)               if( kb .lt. pver + 1 ) web = ceb * wstar3       ! When 'kb.eq.pver+1', it was set to web=0. 
  2546) 
  2547)           else !
  2548) 
  2549)               ! (Option.2) wstarentr = .false. Use original entrainment formulation.
  2550)               ! trmp > 0 if there are interior interfaces in CL, trmp = 0 otherwise.
  2551)               ! trmq > 0 if there is cloudtop radiative cooling, trmq = 0 otherwise.
  2552)              
  2553)               trma = 1._r8 - b1 * a1l * fact
  2554)               trma = max( trma, 0.5_r8 )  ! Prevents runaway entrainment instability
  2555)               trmp = ebrk(i,ncv) * ( lbrk(i,ncv) / lbulk ) / ( 3._r8 * trma )
  2556)               trmq = 0.5_r8 * b1 * ( leng(i,kt)  / lbulk ) * radf * dzht / trma
  2557) 
  2558)               qq = trmq**2 - trmp**3
  2559)               if( qq .ge. 0._r8 ) then 
  2560)                   rootp = ( trmq + sqrt(qq) )**(1._r8/3._r8) + ( max( trmq - sqrt(qq), 0._r8 ) )**(1._r8/3._r8)
  2561)               else ! Also part of case 3
  2562)                   rootp = 2._r8 * sqrt(trmp) * cos( acos( trmq / sqrt(trmp**3) ) / 3._r8 )
  2563)               end if   ! qq
  2564) 
  2565)              ! Find entrainment rates and limit them by free-entrainment values a1l*sqrt(e)
  2566) 
  2567)               wet = a1l * rootp * min( evhc * rootp**2 / ( leng(i,kt) * jtbu ), 1._r8 )   
  2568)               if( kb .lt. pver + 1 ) web = a1l * rootp * min( evhc * rootp**2 / ( leng(i,kb) * jbbu ), 1._r8 )
  2569) 
  2570)           end if ! wstarentr
  2571) 
  2572)           ! ---------------------------------------------------- !
  2573)           ! Finally, get the net CL mean TKE and normalized TKE  ! 
  2574)           ! ---------------------------------------------------- !
  2575) 
  2576)           ebrk(i,ncv) = rootp**2
  2577)           ebrk(i,ncv) = min(ebrk(i,ncv),tkemax) ! Limit CL-avg TKE used for entrainment
  2578)           wbrk(i,ncv) = ebrk(i,ncv)/b1  
  2579)         
  2580)           ! The only way ebrk = 0 is for SRCL which are actually radiatively cooled 
  2581)           ! at top interface. In this case, we remove 'convective' label from the 
  2582)           ! interfaces around this layer. This case should now be impossible, so 
  2583)           ! we flag it. Q: I can't understand why this case is impossible now. Maybe,
  2584)           ! due to various limiting procedures used in solving cubic equation ? 
  2585)           ! In case of SRCL, 'ebrk' should be positive due to cloud top LW radiative
  2586)           ! cooling contribution, although 'ebrk(internal)' of SRCL before including
  2587)           ! entrainment contribution (which include LW cooling contribution also) is
  2588)           ! zero. 
  2589) 
  2590)           if( ebrk(i,ncv) .le. 0._r8 ) then
  2591)               write(iulog,*) 'CALEDDY: Warning, CL with zero TKE, i, kt, kb ', i, kt, kb
  2592)               belongcv(i,kt) = .false.
  2593)               belongcv(i,kb) = .false. 
  2594)           end if
  2595)           
  2596)           ! ----------------------------------------------------------------------- !
  2597)           ! Calculate complete TKE profiles at all CL interfaces, capped by tkemax. !
  2598)           ! We approximate TKE = <e> at entrainment interfaces. However when CL is  !
  2599)           ! based at surface, correct 'tkes' will be inserted to tke(i,pver+1).     !
  2600)           ! Note that this approximation at CL external interfaces do not influence !
  2601)           ! numerical calculation since 'e' at external interfaces are not used  in !
  2602)           ! actual numerical calculation afterward. In addition in order to extract !
  2603)           ! correct TKE averaged over the PBL in the cumulus scheme,it is necessary !
  2604)           ! to set e = <e> at the top entrainment interface.  Since net CL mean TKE !
  2605)           ! 'ebrk' obtained by solving cubic equation already includes tkes  ( tkes !
  2606)           ! is included when bflxs > 0 but not when bflxs <= 0 into internal ebrk ),!
  2607)           ! 'tkes' should be written to tke(i,pver+1)                               !
  2608)           ! ----------------------------------------------------------------------- !
  2609) 
  2610)           ! 1. At internal interfaces          
  2611)           do k = kb - 1, kt + 1, -1
  2612)              rcap = ( b1 * ae + wcap(i,k) / wbrk(i,ncv) ) / ( b1 * ae + 1._r8 )
  2613)              rcap = min( max(rcap,rcapmin), rcapmax )
  2614)              tke(i,k) = ebrk(i,ncv) * rcap
  2615)              tke(i,k) = min( tke(i,k), tkemax )
  2616)              kvh(i,k) = leng(i,k) * sqrt(tke(i,k)) * shcl(i,ncv)
  2617)              kvm(i,k) = leng(i,k) * sqrt(tke(i,k)) * smcl(i,ncv)
  2618)              bprod(i,k) = -kvh(i,k) * n2(i,k)
  2619)              sprod(i,k) =  kvm(i,k) * s2(i,k)
  2620)              turbtype(i,k) = 2                     ! CL interior interfaces.
  2621)              sm_aw(i,k) = smcl(i,ncv)/alph1        ! Diagnostic output for microphysics
  2622)           end do
  2623) 
  2624)           ! 2. At CL top entrainment interface
  2625)           kentr = wet * jtzm
  2626)           kvh(i,kt) = kentr
  2627)           kvm(i,kt) = kentr
  2628)           bprod(i,kt) = -kentr * n2ht + radf       ! I must use 'n2ht' not 'n2'
  2629)           sprod(i,kt) =  kentr * s2(i,kt)
  2630)           turbtype(i,kt) = 4                       ! CL top entrainment interface
  2631)           trmp = -b1 * ae / ( 1._r8 + b1 * ae )
  2632)           trmq = -(bprod(i,kt)+sprod(i,kt))*b1*leng(i,kt)/(1._r8+b1*ae)/(ebrk(i,ncv)**(3._r8/2._r8))
  2633)           rcap = compute_cubic(0._r8,trmp,trmq)**2._r8
  2634)           rcap = min( max(rcap,rcapmin), rcapmax )
  2635)           tke(i,kt)  = ebrk(i,ncv) * rcap
  2636)           tke(i,kt)  = min( tke(i,kt), tkemax )
  2637)           sm_aw(i,kt) = smcl(i,ncv) / alph1        ! Diagnostic output for microphysics
  2638) 
  2639)           ! 3. At CL base entrainment interface and double entraining interfaces
  2640)           ! When current CL base is also the top interface of CL regime below,
  2641)           ! simply add the two contributions for calculating eddy diffusivity
  2642)           ! and buoyancy/shear production. Below code correctly works because
  2643)           ! we (CL regime index) always go from surface upward.
  2644) 
  2645)           if( kb .lt. pver + 1 ) then 
  2646) 
  2647)               kentr = web * jbzm
  2648) 
  2649)               if( kb .ne. ktblw ) then
  2650) 
  2651)                   kvh(i,kb) = kentr
  2652)                   kvm(i,kb) = kentr
  2653)                   bprod(i,kb) = -kvh(i,kb)*n2hb     ! I must use 'n2hb' not 'n2'
  2654)                   sprod(i,kb) =  kvm(i,kb)*s2(i,kb)
  2655)                   turbtype(i,kb) = 3                ! CL base entrainment interface
  2656)                   trmp = -b1*ae/(1._r8+b1*ae)
  2657)                   trmq = -(bprod(i,kb)+sprod(i,kb))*b1*leng(i,kb)/(1._r8+b1*ae)/(ebrk(i,ncv)**(3._r8/2._r8))
  2658)                   rcap = compute_cubic(0._r8,trmp,trmq)**2._r8
  2659)                   rcap = min( max(rcap,rcapmin), rcapmax )
  2660)                   tke(i,kb)  = ebrk(i,ncv) * rcap
  2661)                   tke(i,kb)  = min( tke(i,kb),tkemax )
  2662) 
  2663)               else
  2664)                   
  2665)                   kvh(i,kb) = kvh(i,kb) + kentr 
  2666)                   kvm(i,kb) = kvm(i,kb) + kentr
  2667)                 ! dzhb5 : Half thickness of the lowest  layer of  current CL regime
  2668)                 ! dzht5 : Half thickness of the highest layer of adjacent CL regime just below current CL. 
  2669)                   dzhb5 = z(i,kb-1) - zi(i,kb)
  2670)                   dzht5 = zi(i,kb) - z(i,kb)
  2671)                   bprod(i,kb) = ( dzht5*bprod(i,kb) - dzhb5*kentr*n2hb )     / ( dzhb5 + dzht5 )
  2672)                   sprod(i,kb) = ( dzht5*sprod(i,kb) + dzhb5*kentr*s2(i,kb) ) / ( dzhb5 + dzht5 )
  2673)                   trmp = -b1*ae/(1._r8+b1*ae)
  2674)                   trmq = -kentr*(s2(i,kb)-n2hb)*b1*leng(i,kb)/(1._r8+b1*ae)/(ebrk(i,ncv)**(3._r8/2._r8))
  2675)                   rcap = compute_cubic(0._r8,trmp,trmq)**2._r8
  2676)                   rcap = min( max(rcap,rcapmin), rcapmax )
  2677)                   tke_imsi = ebrk(i,ncv) * rcap
  2678)                   tke_imsi = min( tke_imsi, tkemax )
  2679)                   tke(i,kb)  = ( dzht5*tke(i,kb) + dzhb5*tke_imsi ) / ( dzhb5 + dzht5 )               
  2680)                   tke(i,kb)  = min(tke(i,kb),tkemax)
  2681)                   turbtype(i,kb) = 5                ! CL double entraining interface      
  2682)                  
  2683)               end if
  2684) 
  2685)            else
  2686) 
  2687)              ! If CL base interface is surface, compute similarly using wcap(i,kb)=tkes/b1    
  2688)              ! Even when bflx < 0, use the same formula in order to impose consistency of
  2689)              ! tke(i,kb) at bflx = 0._r8
  2690)  
  2691)              rcap = (b1*ae + wcap(i,kb)/wbrk(i,ncv))/(b1*ae + 1._r8)
  2692)              rcap = min( max(rcap,rcapmin), rcapmax )
  2693)              tke(i,kb) = ebrk(i,ncv) * rcap
  2694)              tke(i,kb) = min( tke(i,kb),tkemax )
  2695) 
  2696)           end if
  2697) 
  2698)           ! For double entraining interface, simply use smcl(i,ncv) of the overlying CL. 
  2699)           ! Below 'sm_aw' is a diagnostic output for use in the microphysics.
  2700)           ! When 'kb' is surface, 'sm' will be over-written later below.
  2701) 
  2702)           sm_aw(i,kb) = smcl(i,ncv)/alph1             
  2703) 
  2704)           ! Calculate wcap at all interfaces of CL. Put a  minimum threshold on TKE
  2705)           ! to prevent possible division by zero.  'wcap' at CL internal interfaces
  2706)           ! are already calculated in the first part of 'do ncv' loop correctly.
  2707)           ! When 'kb.eq.pver+1', below formula produces the identical result to the
  2708)           ! 'tkes(i)/b1' if leng(i,kb) is set to vk*z(i,pver). Note  wcap(i,pver+1)
  2709)           ! is already defined as 'tkes(i)/b1' at the first part of caleddy.
  2710)           
  2711)           wcap(i,kt) = (bprod(i,kt)+sprod(i,kt))*leng(i,kt)/sqrt(max(tke(i,kt),1.e-6_r8))
  2712)           if( kb .lt. pver + 1 ) then
  2713)               wcap(i,kb) = (bprod(i,kb)+sprod(i,kb))*leng(i,kb)/sqrt(max(tke(i,kb),1.e-6_r8))
  2714)           end if
  2715) 
  2716)           ! Save the index of upper external interface of current CL-regime in order to
  2717)           ! handle the case when this interface is also the lower external interface of 
  2718)           ! CL-regime located just above. 
  2719) 
  2720)           ktblw = kt 
  2721) 
  2722)           ! Diagnostic Output
  2723) 
  2724)           wet_CL(i,ncv)        = wet
  2725)           web_CL(i,ncv)        = web
  2726)           jtbu_CL(i,ncv)       = jtbu
  2727)           jbbu_CL(i,ncv)       = jbbu
  2728)           evhc_CL(i,ncv)       = evhc
  2729)           jt2slv_CL(i,ncv)     = jt2slv
  2730)           n2ht_CL(i,ncv)       = n2ht
  2731)           n2hb_CL(i,ncv)       = n2hb          
  2732)           lwp_CL(i,ncv)        = lwp
  2733)           opt_depth_CL(i,ncv)  = opt_depth
  2734)           radinvfrac_CL(i,ncv) = radinvfrac
  2735)           radf_CL(i,ncv)       = radf
  2736)           wstar_CL(i,ncv)      = wstar          
  2737)           wstar3fact_CL(i,ncv) = wstar3fact          
  2738) 
  2739)        end do        ! ncv
  2740)  
  2741)        ! Calculate PBL height and characteristic cumulus excess for use in the
  2742)        ! cumulus convection shceme. Also define turbulence type at the surface
  2743)        ! when the lowest CL is based at the surface. These are just diagnostic
  2744)        ! outputs, not influencing numerical calculation of current PBL scheme.
  2745)        ! If the lowest CL is based at the surface, define the PBL depth as the
  2746)        ! CL top interface. The same rule is applied for all CLs including SRCL.
  2747) 
  2748)        if( ncvsurf .gt. 0 ) then
  2749) 
  2750)            ktopbl(i) = ktop(i,ncvsurf)
  2751)            pblh(i)   = zi(i, ktopbl(i))
  2752)            pblhp(i)  = pi(i, ktopbl(i))
  2753)            wpert(i)  = max(wfac*sqrt(ebrk(i,ncvsurf)),wpertmin)
  2754)            tpert(i)  = max(abs(shflx(i)*rrho(i)/cpair)*tfac/wpert(i),0._r8)
  2755)            qpert(i)  = max(abs(qflx(i)*rrho(i))*tfac/wpert(i),0._r8)
  2756) 
  2757)            if( bflxs(i) .gt. 0._r8 ) then
  2758)                turbtype(i,pver+1) = 2 ! CL interior interface
  2759)            else
  2760)                turbtype(i,pver+1) = 3 ! CL external base interface
  2761)            endif
  2762) 
  2763)            ipbl(i)  = 1
  2764)            kpblh(i) = max(ktopbl(i)-1, 1)
  2765)            went(i)  = wet_CL(i,ncvsurf)
  2766)        end if ! End of the calculationf of te properties of surface-based CL.
  2767) 
  2768)        ! -------------------------------------------- !
  2769)        ! Treatment of Stable Turbulent Regime ( STL ) !
  2770)        ! -------------------------------------------- !
  2771) 
  2772)        ! Identify top and bottom most (internal) interfaces of STL except surface.
  2773)        ! Also, calculate 'turbulent length scale (leng)' at each STL interfaces.     
  2774) 
  2775)        belongst(i,1) = .false.   ! k = 1 (top interface) is assumed non-turbulent
  2776)        do k = 2, pver            ! k is an interface index
  2777)           belongst(i,k) = ( ri(i,k) .lt. ricrit ) .and. ( .not. belongcv(i,k) )
  2778)           if( belongst(i,k) .and. ( .not. belongst(i,k-1) ) ) then
  2779)               kt = k             ! Top interface index of STL
  2780)           elseif( .not. belongst(i,k) .and. belongst(i,k-1) ) then
  2781)               kb = k - 1         ! Base interface index of STL
  2782)               lbulk = z(i,kt-1) - z(i,kb)
  2783)               lbulk = min( lbulk, lbulk_max )
  2784)               do ks = kt, kb
  2785)                  if( choice_tunl .eq. 'rampcl' ) then
  2786)                      tunlramp = tunl
  2787)                  elseif( choice_tunl .eq. 'rampsl' ) then
  2788)                     tunlramp = max( 1.e-3_r8, ctunl * tunl * exp(-log(ctunl)*ri(i,ks)/ricrit) )
  2789)                   ! tunlramp = 0.065_r8 + 0.7_r8 * exp(-20._r8*ri(i,ks))
  2790)                  else
  2791)                     tunlramp = tunl
  2792)                  endif
  2793)                  if( choice_leng .eq. 'origin' ) then
  2794)                      leng(i,ks) = ( (vk*zi(i,ks))**(-cleng) + (tunlramp*lbulk)**(-cleng) )**(-1._r8/cleng)
  2795)                    ! leng(i,ks) = vk*zi(i,ks) / (1._r8+vk*zi(i,ks)/(tunlramp*lbulk))
  2796)                  else
  2797)                      leng(i,ks) = min( vk*zi(i,ks), tunlramp*lbulk )              
  2798)                  endif
  2799)                  leng(i,ks) = min(leng_max(ks), leng(i,ks))
  2800)               end do
  2801)           end if
  2802)        end do ! k
  2803) 
  2804)        ! Now look whether STL extends to ground.  If STL extends to surface,
  2805)        ! re-define master length scale,'lbulk' including surface interfacial
  2806)        ! layer thickness, and re-calculate turbulent length scale, 'leng' at
  2807)        ! all STL interfaces again. Note that surface interface is assumed to
  2808)        ! always be STL if it is not CL.   
  2809)        
  2810)        belongst(i,pver+1) = .not. belongcv(i,pver+1)
  2811) 
  2812)        if( belongst(i,pver+1) ) then     ! kb = pver+1 (surface  STL)
  2813) 
  2814)            turbtype(i,pver+1) = 1        ! Surface is STL interface
  2815)           
  2816)            if( belongst(i,pver) ) then   ! STL includes interior
  2817)              ! 'kt' already defined above as the top interface of STL
  2818)                lbulk = z(i,kt-1)          
  2819)            else                          ! STL with no interior turbulence
  2820)                kt = pver+1
  2821)                lbulk = z(i,kt-1)
  2822)            end if
  2823)            lbulk = min( lbulk, lbulk_max )
  2824) 
  2825)            ! PBL height : Layer mid-point just above the highest STL interface
  2826)            ! Note in contrast to the surface based CL regime where  PBL height
  2827)            ! was defined at the top external interface, PBL height of  surface
  2828)            ! based STL is defined as the layer mid-point.
  2829) 
  2830)            ktopbl(i) = kt - 1
  2831)            pblh(i)   = z(i,ktopbl(i))
  2832)            pblhp(i)  = 0.5_r8 * ( pi(i,ktopbl(i)) + pi(i,ktopbl(i)+1) )          
  2833) 
  2834)            ! Re-calculate turbulent length scale including surface interfacial
  2835)            ! layer contribution to lbulk.
  2836) 
  2837)            do ks = kt, pver
  2838)               if( choice_tunl .eq. 'rampcl' ) then
  2839)                   tunlramp = tunl
  2840)               elseif( choice_tunl .eq. 'rampsl' ) then
  2841)                   tunlramp = max(1.e-3_r8,ctunl*tunl*exp(-log(ctunl)*ri(i,ks)/ricrit))
  2842)                 ! tunlramp = 0.065_r8 + 0.7_r8 * exp(-20._r8*ri(i,ks))
  2843)               else
  2844)                   tunlramp = tunl
  2845)               endif
  2846)               if( choice_leng .eq. 'origin' ) then
  2847)                   leng(i,ks) = ( (vk*zi(i,ks))**(-cleng) + (tunlramp*lbulk)**(-cleng) )**(-1._r8/cleng)
  2848)                 ! leng(i,ks) = vk*zi(i,ks) / (1._r8+vk*zi(i,ks)/(tunlramp*lbulk))
  2849)               else
  2850)                   leng(i,ks) = min( vk*zi(i,ks), tunlramp*lbulk )              
  2851)               endif
  2852)              leng(i,ks) = min(leng_max(ks), leng(i,ks))
  2853)            end do ! ks
  2854) 
  2855)            ! Characteristic cumulus excess of surface-based STL.
  2856)            ! We may be able to use ustar for wpert.
  2857) 
  2858)            wpert(i) = 0._r8 
  2859)            tpert(i) = max(shflx(i)*rrho(i)/cpair*fak/ustar(i),0._r8) ! CCM stable-layer forms
  2860)            qpert(i) = max(qflx(i)*rrho(i)*fak/ustar(i),0._r8)
  2861) 
  2862)            ipbl(i)  = 0
  2863)            kpblh(i) = ktopbl(i)
  2864) 
  2865)        end if
  2866) 
  2867)        ! Calculate stability functions and energetics at the STL interfaces
  2868)        ! except the surface. Note that tke(i,pver+1) and wcap(i,pver+1) are
  2869)        ! already calculated in the first part of 'caleddy', kvm(i,pver+1) &
  2870)        ! kvh(i,pver+1) were already initialized to be zero, bprod(i,pver+1)
  2871)        ! & sprod(i,pver+1) were direcly calculated from the bflxs and ustar.
  2872)        ! Note transport term is assumed to be negligible at STL interfaces.
  2873)            
  2874)        do k = 2, pver
  2875) 
  2876)           if( belongst(i,k) ) then
  2877) 
  2878)               turbtype(i,k) = 1    ! STL interfaces
  2879)               trma = alph3*alph4exs*ri(i,k) + 2._r8*b1*(alph2-alph4exs*alph5*ri(i,k))
  2880)               trmb = (alph3+alph4exs)*ri(i,k) + 2._r8*b1*(-alph5*ri(i,k)+alph1)
  2881)               trmc = ri(i,k)
  2882)               det = max(trmb*trmb-4._r8*trma*trmc,0._r8)
  2883)               ! Sanity Check
  2884)               if( det .lt. 0._r8 ) then
  2885)                   write(iulog,*) 'The det < 0. for the STL in UW eddy_diff'
  2886)                   call endrun('CALEDDY: The det < 0. for the STL in UW eddy_diff')
  2887)               end if                  
  2888)               gh = (-trmb + sqrt(det))/(2._r8*trma)
  2889)             ! gh = min(max(gh,-0.28_r8),0.0233_r8)
  2890)             ! gh = min(max(gh,-3.5334_r8),0.0233_r8)
  2891)               gh = min(max(gh,ghmin),0.0233_r8)
  2892)               sh = max(0._r8,alph5/(1._r8+alph3*gh))
  2893)               sm = max(0._r8,(alph1 + alph2*gh)/(1._r8+alph3*gh)/(1._r8+alph4exs*gh))
  2894) 
  2895)               tke(i,k)   = b1*(leng(i,k)**2)*(-sh*n2(i,k)+sm*s2(i,k))
  2896)               tke(i,k)   = min(tke(i,k),tkemax)
  2897)               wcap(i,k)  = tke(i,k)/b1
  2898)               kvh(i,k)   = leng(i,k) * sqrt(tke(i,k)) * sh
  2899)               kvm(i,k)   = leng(i,k) * sqrt(tke(i,k)) * sm
  2900)               bprod(i,k) = -kvh(i,k) * n2(i,k)
  2901)               sprod(i,k) =  kvm(i,k) * s2(i,k)
  2902) 
  2903)               sm_aw(i,k) = sm/alph1     ! This is diagnostic output for use in the microphysics             
  2904) 
  2905)           end if
  2906) 
  2907)        end do  ! k
  2908) 
  2909)        ! --------------------------------------------------- !
  2910)        ! End of treatment of Stable Turbulent Regime ( STL ) !
  2911)        ! --------------------------------------------------- !
  2912) 
  2913)        ! --------------------------------------------------------------- !
  2914)        ! Re-computation of eddy diffusivity at the entrainment interface !
  2915)        ! assuming that it is purely STL (0<Ri<0.19). Note even Ri>0.19,  !
  2916)        ! turbulent can exist at the entrainment interface since 'Sh,Sm'  !
  2917)        ! do not necessarily go to zero even when Ri>0.19. Since Ri can   !
  2918)        ! be fairly larger than 0.19 at the entrainment interface, I      !
  2919)        ! should set minimum value of 'tke' to be 0. in order to prevent  !
  2920)        ! sqrt(tke) from being imaginary.                                 !
  2921)        ! --------------------------------------------------------------- !
  2922) 
  2923)        ! goto 888
  2924) 
  2925)          do k = 2, pver
  2926) 
  2927)          if( ( turbtype(i,k) .eq. 3 ) .or. ( turbtype(i,k) .eq. 4 ) .or. &
  2928)              ( turbtype(i,k) .eq. 5 ) ) then
  2929) 
  2930)              trma = alph3*alph4exs*ri(i,k) + 2._r8*b1*(alph2-alph4exs*alph5*ri(i,k))
  2931)              trmb = (alph3+alph4exs)*ri(i,k) + 2._r8*b1*(-alph5*ri(i,k)+alph1)
  2932)              trmc = ri(i,k)
  2933)              det  = max(trmb*trmb-4._r8*trma*trmc,0._r8)
  2934)              gh   = (-trmb + sqrt(det))/(2._r8*trma)
  2935)            ! gh   = min(max(gh,-0.28_r8),0.0233_r8)
  2936)            ! gh   = min(max(gh,-3.5334_r8),0.0233_r8)
  2937)              gh   = min(max(gh,ghmin),0.0233_r8)
  2938)              sh   = max(0._r8,alph5/(1._r8+alph3*gh))
  2939)              sm   = max(0._r8,(alph1 + alph2*gh)/(1._r8+alph3*gh)/(1._r8+alph4exs*gh))
  2940) 
  2941)              lbulk = z(i,k-1) - z(i,k)
  2942)              lbulk = min( lbulk, lbulk_max )
  2943) 
  2944)              if( choice_tunl .eq. 'rampcl' ) then
  2945)                  tunlramp = tunl
  2946)              elseif( choice_tunl .eq. 'rampsl' ) then
  2947)                  tunlramp = max(1.e-3_r8,ctunl*tunl*exp(-log(ctunl)*ri(i,k)/ricrit))
  2948)                ! tunlramp = 0.065_r8 + 0.7_r8*exp(-20._r8*ri(i,k))
  2949)              else
  2950)                  tunlramp = tunl
  2951)              endif
  2952)              if( choice_leng .eq. 'origin' ) then
  2953)                  leng_imsi = ( (vk*zi(i,k))**(-cleng) + (tunlramp*lbulk)**(-cleng) )**(-1._r8/cleng)
  2954)                ! leng_imsi = vk*zi(i,k) / (1._r8+vk*zi(i,k)/(tunlramp*lbulk))
  2955)              else
  2956)                  leng_imsi = min( vk*zi(i,k), tunlramp*lbulk )              
  2957)              endif
  2958)              leng_imsi = min(leng_max(k), leng_imsi)
  2959) 
  2960)              tke_imsi = b1*(leng_imsi**2)*(-sh*n2(i,k)+sm*s2(i,k))
  2961)              tke_imsi = min(max(tke_imsi,0._r8),tkemax)
  2962)              kvh_imsi = leng_imsi * sqrt(tke_imsi) * sh
  2963)              kvm_imsi = leng_imsi * sqrt(tke_imsi) * sm
  2964) 
  2965)              if( kvh(i,k) .lt. kvh_imsi ) then 
  2966)                  kvh(i,k)   =  kvh_imsi
  2967)                  kvm(i,k)   =  kvm_imsi
  2968)                  leng(i,k)  = leng_imsi
  2969)                  tke(i,k)   =  tke_imsi
  2970)                  wcap(i,k)  =  tke_imsi / b1
  2971)                  bprod(i,k) = -kvh_imsi * n2(i,k)
  2972)                  sprod(i,k) =  kvm_imsi * s2(i,k)
  2973)                  sm_aw(i,k) =  sm/alph1     ! This is diagnostic output for use in the microphysics             
  2974)                  turbtype(i,k) = 1          ! This was added on Dec.10.2009 for use in microphysics.
  2975)              endif
  2976) 
  2977)          end if
  2978) 
  2979)          end do
  2980) 
  2981)  ! 888   continue 
  2982) 
  2983)        ! ------------------------------------------------------------------ !
  2984)        ! End of recomputation of eddy diffusivity at entrainment interfaces !
  2985)        ! ------------------------------------------------------------------ !
  2986) 
  2987)        ! As an option, we can impose a certain minimum back-ground diffusivity.
  2988) 
  2989)        ! do k = 1, pver+1
  2990)        !    kvh(i,k) = max(0.01_r8,kvh(i,k))
  2991)        !    kvm(i,k) = max(0.01_r8,kvm(i,k))
  2992)        ! enddo
  2993)  
  2994)        ! --------------------------------------------------------------------- !
  2995)        ! Diagnostic Output                                                     !
  2996)        ! Just for diagnostic purpose, calculate stability functions at  each   !
  2997)        ! interface including surface. Instead of assuming neutral stability,   !
  2998)        ! explicitly calculate stability functions using an reverse procedure   !
  2999)        ! starting from tkes(i) similar to the case of SRCL and SBCL in zisocl. !
  3000)        ! Note that it is possible to calculate stability functions even when   !
  3001)        ! bflxs < 0. Note that this inverse method allows us to define Ri even  !
  3002)        ! at the surface. Note also tkes(i) and sprod(i,pver+1) are always      !
  3003)        ! positive values by limiters (e.g., ustar_min = 0.01).                 !
  3004)        ! Dec.12.2006 : Also just for diagnostic output, re-set                 !
  3005)        ! 'bprod(i,pver+1)= bflxs(i)' here. Note that this setting does not     !
  3006)        ! influence numerical calculation at all - it is just for diagnostic    !
  3007)        ! output.                                                               !
  3008)        ! --------------------------------------------------------------------- !
  3009) 
  3010)        bprod(i,pver+1) = bflxs(i)
  3011)               
  3012)        gg = 0.5_r8*vk*z(i,pver)*bprod(i,pver+1)/(tkes(i)**(3._r8/2._r8))
  3013)        if( abs(alph5-gg*alph3) .le. 1.e-7_r8 ) then
  3014)          ! gh = -0.28_r8
  3015)            if( bprod(i,pver+1) .gt. 0._r8 ) then
  3016)                gh = -3.5334_r8
  3017)            else
  3018)                gh = ghmin
  3019)            endif
  3020)        else    
  3021)            gh = gg/(alph5-gg*alph3)
  3022)        end if 
  3023) 
  3024)      ! gh = min(max(gh,-0.28_r8),0.0233_r8)
  3025)        if( bprod(i,pver+1) .gt. 0._r8 ) then
  3026)            gh = min(max(gh,-3.5334_r8),0.0233_r8)
  3027)        else
  3028)            gh = min(max(gh,ghmin),0.0233_r8)
  3029)        endif
  3030) 
  3031)        gh_a(i,pver+1) = gh     
  3032)        sh_a(i,pver+1) = max(0._r8,alph5/(1._r8+alph3*gh))
  3033)        if( bprod(i,pver+1) .gt. 0._r8 ) then       
  3034)            sm_a(i,pver+1) = max(0._r8,(alph1+alph2*gh)/(1._r8+alph3*gh)/(1._r8+alph4*gh))
  3035)        else
  3036)            sm_a(i,pver+1) = max(0._r8,(alph1+alph2*gh)/(1._r8+alph3*gh)/(1._r8+alph4exs*gh))
  3037)        endif
  3038)        sm_aw(i,pver+1) = sm_a(i,pver+1)/alph1
  3039)        ri_a(i,pver+1)  = -(sm_a(i,pver+1)/sh_a(i,pver+1))*(bprod(i,pver+1)/sprod(i,pver+1))
  3040) 
  3041)        do k = 1, pver
  3042)           if( ri(i,k) .lt. 0._r8 ) then
  3043)               trma = alph3*alph4*ri(i,k) + 2._r8*b1*(alph2-alph4*alph5*ri(i,k))
  3044)               trmb = (alph3+alph4)*ri(i,k) + 2._r8*b1*(-alph5*ri(i,k)+alph1)
  3045)               trmc = ri(i,k)
  3046)               det  = max(trmb*trmb-4._r8*trma*trmc,0._r8)
  3047)               gh   = (-trmb + sqrt(det))/(2._r8*trma)
  3048)               gh   = min(max(gh,-3.5334_r8),0.0233_r8)
  3049)               gh_a(i,k) = gh
  3050)               sh_a(i,k) = max(0._r8,alph5/(1._r8+alph3*gh))
  3051)               sm_a(i,k) = max(0._r8,(alph1+alph2*gh)/(1._r8+alph3*gh)/(1._r8+alph4*gh))
  3052)               ri_a(i,k) = ri(i,k)
  3053)           else
  3054)               if( ri(i,k) .gt. ricrit ) then
  3055)                   gh_a(i,k) = ghmin
  3056)                   sh_a(i,k) = 0._r8
  3057)                   sm_a(i,k) = 0._r8
  3058)                   ri_a(i,k) = ri(i,k)
  3059)               else
  3060)                   trma = alph3*alph4exs*ri(i,k) + 2._r8*b1*(alph2-alph4exs*alph5*ri(i,k))
  3061)                   trmb = (alph3+alph4exs)*ri(i,k) + 2._r8*b1*(-alph5*ri(i,k)+alph1)
  3062)                   trmc = ri(i,k)
  3063)                   det  = max(trmb*trmb-4._r8*trma*trmc,0._r8)
  3064)                   gh   = (-trmb + sqrt(det))/(2._r8*trma)
  3065)                   gh   = min(max(gh,ghmin),0.0233_r8)
  3066)                   gh_a(i,k) = gh
  3067)                   sh_a(i,k) = max(0._r8,alph5/(1._r8+alph3*gh))
  3068)                   sm_a(i,k) = max(0._r8,(alph1+alph2*gh)/(1._r8+alph3*gh)/(1._r8+alph4exs*gh))
  3069)                   ri_a(i,k) = ri(i,k)
  3070)               endif
  3071)           endif
  3072) 
  3073)        end do
  3074) 
  3075)     end do   ! End of column index loop, i 
  3076) 
  3077)     return
  3078) 
  3079)     end subroutine caleddy

  3085)     subroutine exacol( pcols, pver, ncol, ri, bflxs, minpblh, zi, ktop, kbase, ncvfin ) 
  3086) 
  3087)     ! ---------------------------------------------------------------------------- !
  3088)     ! Object : Find unstable CL regimes and determine the indices                  !
  3089)     !          kbase, ktop which delimit these unstable layers :                   !
  3090)     !          ri(kbase) > 0 and ri(ktop) > 0, but ri(k) < 0 for ktop < k < kbase. ! 
  3091)     ! Author : Chris  Bretherton 08/2000,                                          !
  3092)     !          Sungsu Park       08/2006, 11/2008                                  !
  3093)     !----------------------------------------------------------------------------- !
  3094) 
  3095)     implicit none
  3096) 
  3097)     ! --------------- !
  3098)     ! Input variables !
  3099)     ! --------------- !
  3100) 
  3101)     integer,  intent(in) :: pcols                  ! Number of atmospheric columns   
  3102)     integer,  intent(in) :: pver                   ! Number of atmospheric vertical layers   
  3103)     integer,  intent(in) :: ncol                   ! Number of atmospheric columns   
  3104) 
  3105)     real(r8), intent(in) :: ri(pcols,pver)         ! Moist gradient Richardson no.
  3106)     real(r8), intent(in) :: bflxs(pcols)           ! Buoyancy flux at surface
  3107)     real(r8), intent(in) :: minpblh(pcols)         ! Minimum PBL height based on surface stress
  3108)     real(r8), intent(in) :: zi(pcols,pver+1)       ! Interface heights
  3109) 
  3110)     ! ---------------- !
  3111)     ! Output variables !      
  3112)     ! ---------------- !
  3113) 
  3114)     integer, intent(out) :: kbase(pcols,ncvmax)    ! External interface index of CL base
  3115)     integer, intent(out) :: ktop(pcols,ncvmax)     ! External interface index of CL top
  3116)     integer, intent(out) :: ncvfin(pcols)          ! Total number of CLs
  3117) 
  3118)     ! --------------- !
  3119)     ! Local variables !
  3120)     ! --------------- !
  3121) 
  3122)     integer              :: i
  3123)     integer              :: k
  3124)     integer              :: ncv
  3125)     real(r8)             :: rimaxentr
  3126)     real(r8)             :: riex(pver+1)           ! Column Ri profile extended to surface
  3127) 
  3128)     ! ----------------------- !
  3129)     ! Main Computation Begins !
  3130)     ! ----------------------- !
  3131) 
  3132)     do i = 1, ncol
  3133)        ncvfin(i) = 0
  3134)        do ncv = 1, ncvmax
  3135)           ktop(i,ncv)  = 0
  3136)           kbase(i,ncv) = 0
  3137)        end do
  3138)     end do
  3139) 
  3140)     ! ------------------------------------------------------ !
  3141)     ! Find CL regimes starting from the surface going upward !
  3142)     ! ------------------------------------------------------ !
  3143)     
  3144)     rimaxentr = 0._r8   
  3145)     
  3146)     do i = 1, ncol
  3147) 
  3148)        riex(2:pver) = ri(i,2:pver)
  3149) 
  3150)        ! Below allows consistent treatment of surface and other interfaces.
  3151)        ! Simply, if surface buoyancy flux is positive, Ri of surface is set to be negative.
  3152) 
  3153)        riex(pver+1) = rimaxentr - bflxs(i) 
  3154) 
  3155)        ncv = 0
  3156)        k   = pver + 1 ! Work upward from surface interface
  3157) 
  3158)        do while ( k .gt. ntop_turb + 1 )
  3159) 
  3160)         ! Below means that if 'bflxs > 0' (do not contain '=' sign), surface
  3161)         ! interface is energetically interior surface. 
  3162)        
  3163)           if( riex(k) .lt. rimaxentr ) then 
  3164) 
  3165)               ! Identify a new CL
  3166) 
  3167)               ncv = ncv + 1
  3168) 
  3169)               ! First define 'kbase' as the first interface below the lower-most unstable interface
  3170)               ! Thus, Richardson number at 'kbase' is positive.
  3171) 
  3172)               kbase(i,ncv) = min(k+1,pver+1)
  3173) 
  3174)               ! Decrement k until top unstable level
  3175) 
  3176)               do while( riex(k) .lt. rimaxentr .and. k .gt. ntop_turb + 1 )
  3177)                  k = k - 1
  3178)               end do
  3179) 
  3180)               ! ktop is the first interface above upper-most unstable interface
  3181)               ! Thus, Richardson number at 'ktop' is positive. 
  3182) 
  3183)               ktop(i,ncv) = k
  3184)              
  3185)           else
  3186) 
  3187)               ! Search upward for a CL.
  3188) 
  3189)               k = k - 1
  3190) 
  3191)           end if
  3192) 
  3193)        end do ! End of CL regime finding for each atmospheric column
  3194) 
  3195)        ncvfin(i) = ncv    
  3196) 
  3197)     end do  ! End of atmospheric column do loop
  3198) 
  3199)     return 
  3200)

  3277)     integer               :: ncv                      ! CL Index increasing with height

  3284)     integer               :: cntd                     ! Number of merged CLs during downward extension of individual CL

  3286)     integer               :: ktinc                    ! Index for incorporating  overlying CL
  3287)

  3289)     real(r8)              :: dwinc                    ! Normalized TKE of CL external interfaces
  3290)     real(r8)              :: dw_surf                  ! Normalized TKE of surface interfacial layer
  3291)     real(r8)              :: dzinc
  3292)     real(r8)              :: gh
  3293)     real(r8)              :: sh
  3294)     real(r8)              :: sm
  3295)     real(r8)              :: gh_surf                  ! Half of normalized buoyancy production in surface interfacial layer 
  3296)     real(r8)              :: sh_surf                  ! Galperin instability function in surface interfacial layer  
  3297)     real(r8)              :: sm_surf                  ! Galperin instability function in surface interfacial layer 
  3298)     real(r8)              :: l2n2                     ! Vertical integral of 'l^2N^2' over CL. Include thickness product
  3299)     real(r8)              :: l2s2                     ! Vertical integral of 'l^2S^2' over CL. Include thickness product
  3300)     real(r8)              :: dl2n2                    ! Vertical integration of 'l^2*N^2' of CL external interfaces
  3301)     real(r8)              :: dl2s2                    ! Vertical integration of 'l^2*S^2' of CL external interfaces
  3302)     real(r8)              :: dl2n2_surf               ! 'dl2n2' defined in the surface interfacial layer
  3303)     real(r8)              :: dl2s2_surf               ! 'dl2s2' defined in the surface interfacial layer  
  3304)     real(r8)              :: lint                     ! Thickness of (energetically) internal CL
  3305)     real(r8)              :: dlint                    ! Interfacial layer thickness of CL external interfaces
  3306)     real(r8)              :: dlint_surf               ! Surface interfacial layer thickness 
  3307)     real(r8)              :: lbulk                    ! Master Length Scale : Whole CL thickness from top to base external interface
  3308)     real(r8)              :: lz                       ! Turbulent length scale
  3309)     real(r8)              :: ricll                    ! Mean Richardson number of internal CL 
  3310)     real(r8)              :: trma
  3311)     real(r8)              :: trmb
  3312)     real(r8)              :: trmc
  3313)     real(r8)              :: det
  3314)     real(r8)              :: zbot                     ! Height of CL base
  3315)     real(r8)              :: l2rat                    ! Square of ratio of actual to initial CL (not used)
  3316)     real(r8)              :: gg                       ! Intermediate variable used for calculating stability functions of SBCL
  3317)     real(r8)              :: tunlramp                 ! Ramping tunl
  3318)

  3320)     ! Main Computation Begins !
  3321)     ! ----------------------- ! 
  3322) 
  3323)     i = long
  3324)

  3326)     
  3327)     do k = 1, ncvmax
  3328)        ricl(i,k) = 0._r8
  3329)        ghcl(i,k) = 0._r8
  3330)        shcl(i,k) = 0._r8
  3331)        smcl(i,k) = 0._r8
  3332)        lbrk(i,k) = 0._r8
  3333)        wbrk(i,k) = 0._r8
  3334)        ebrk(i,k) = 0._r8
  3335)     end do
  3336)     extend    = .false.
  3337)     extend_up = .false.
  3338)     extend_dn = .false.
  3339) 
  3340)     ! ----------------------------------------------------------- !
  3341)     ! Loop over each CL to see if any of them need to be extended !
  3342)     ! ----------------------------------------------------------- !
  3343) 
  3344)     ncv = 1
  3345) 
  3346)     do while( ncv .le. ncvfin(i) )
  3347) 
  3348)        ncvinit = ncv
  3349)        cntu    = 0
  3350)        cntd    = 0
  3351)        kb      = kbase(i,ncv) 
  3352)        kt      = ktop(i,ncv)
  3353)        
  3354)        ! ---------------------------------------------------------------------------- !
  3355)        ! Calculation of CL interior energetics including surface before extension     !
  3356)        ! ---------------------------------------------------------------------------- !
  3357)        ! Note that the contribution of interior interfaces (not surface) to 'wint' is !
  3358)        ! accounted by using '-sh*l2n2 + sm*l2s2' while the contribution of surface is !
  3359)        ! accounted by using 'dwsurf = tkes/b1' when bflxs > 0. This approach is fully !
  3360)        ! reasonable. Another possible alternative,  which seems to be also consistent !
  3361)        ! is to calculate 'dl2n2_surf'  and  'dl2s2_surf' of surface interfacial layer !
  3362)        ! separately, and this contribution is explicitly added by initializing 'l2n2' !
  3363)        ! 'l2s2' not by zero, but by 'dl2n2_surf' and 'ds2n2_surf' below.  At the same !
  3364)        ! time, 'dwsurf' should be excluded in 'wint' calculation below. The only diff.!
  3365)        ! between two approaches is that in case of the latter approach, contributions !
  3366)        ! of surface interfacial layer to the CL mean stability function (ri,gh,sh,sm) !
  3367)        ! are explicitly included while the first approach is not. In this sense,  the !
  3368)        ! second approach seems to be more conceptually consistent,   but currently, I !
  3369)        ! (Sungsu) will keep the first default approach. There is a switch             !
  3370)        ! 'use_dw_surf' at the first part of eddy_diff.F90 chosing one of              !
  3371)        ! these two options.                                                           !
  3372)        ! ---------------------------------------------------------------------------- !
  3373)        
  3374)        ! ------------------------------------------------------ !   
  3375)        ! Step 0: Calculate surface interfacial layer energetics !
  3376)        ! ------------------------------------------------------ !
  3377) 
  3378)        lbulk      = zi(i,kt) - zi(i,kb)
  3379)        lbulk      = min( lbulk, lbulk_max )
  3380)        dlint_surf = 0._r8
  3381)        dl2n2_surf = 0._r8
  3382)        dl2s2_surf = 0._r8
  3383)        dw_surf    = 0._r8
  3384)        if( kb .eq. pver+1 ) then
  3385) 
  3386)            if( bflxs(i) .gt. 0._r8 ) then
  3387) 
  3388)                ! Calculate stability functions of surface interfacial layer
  3389)                ! from the given 'bprod(i,pver+1)' and 'sprod(i,pver+1)' using
  3390)                ! inverse approach. Since alph5>0 and alph3<0, denominator of
  3391)                ! gg is always positive if bprod(i,pver+1)>0.

  3393)                gg    = 0.5_r8*vk*z(i,pver)*bprod(i,pver+1)/(tkes(i)**(3._r8/2._r8))
  3394)                gh    = gg/(alph5-gg*alph3)
  3395)              ! gh    = min(max(gh,-0.28_r8),0.0233_r8)
  3396)                gh    = min(max(gh,-3.5334_r8),0.0233_r8)
  3397)                sh    = alph5/(1._r8+alph3*gh)
  3398)                sm    = (alph1 + alph2*gh)/(1._r8+alph3*gh)/(1._r8+alph4*gh)
  3399)                ricll = min(-(sm/sh)*(bprod(i,pver+1)/sprod(i,pver+1)),ricrit)
  3400)

  3402)                ! energetics. By construction, 'dw_surf = -dl2n2_surf + ds2n2_surf'
  3403)                ! is exactly satisfied, which corresponds to assuming turbulent
  3404)                ! length scale of surface interfacial layer = vk * z(i,pver). Note
  3405)                ! 'dl2n2_surf','dl2s2_surf','dw_surf' include thickness product.   
  3406) 
  3407)                dlint_surf = z(i,pver)
  3408)                dl2n2_surf = -vk*(z(i,pver)**2)*bprod(i,pver+1)/(sh*sqrt(tkes(i)))
  3409)                dl2s2_surf =  vk*(z(i,pver)**2)*sprod(i,pver+1)/(sm*sqrt(tkes(i)))
  3410)                dw_surf    = (tkes(i)/b1)*z(i,pver) 
  3411) 
  3412)            else
  3413) 
  3414)                ! Note that this case can happen when surface is an external 
  3415)                ! interface of CL.
  3416)                lbulk = zi(i,kt) - z(i,pver)
  3417)                lbulk = min( lbulk, lbulk_max )
  3418) 
  3419)            end if
  3420) 
  3421)        end if
  3422)            
  3423)        ! ------------------------------------------------------ !   
  3424)        ! Step 1: Include surface interfacial layer contribution !
  3425)        ! ------------------------------------------------------ !
  3426)        
  3427)        lint = dlint_surf
  3428)        l2n2 = dl2n2_surf
  3429)        l2s2 = dl2s2_surf          
  3430)        wint = dw_surf
  3431)        if( use_dw_surf ) then
  3432)            l2n2 = 0._r8
  3433)            l2s2 = 0._r8
  3434)        else
  3435)            wint = 0._r8
  3436)        end if    
  3437)        
  3438)        ! --------------------------------------------------------------------------------- !
  3439)        ! Step 2. Include the contribution of 'pure internal interfaces' other than surface !
  3440)        ! --------------------------------------------------------------------------------- !

  3442)        if( kt .lt. kb - 1 ) then ! The case of non-SBCL.
  3443)                               
  3444)            do k = kb - 1, kt + 1, -1       
  3445)               if( choice_tunl .eq. 'rampcl' ) then
  3446)                 ! Modification : I simply used the average tunlramp between the two limits.
  3447)                   tunlramp = 0.5_r8*(1._r8+ctunl)*tunl
  3448)               elseif( choice_tunl .eq. 'rampsl' ) then
  3449)                   tunlramp = ctunl*tunl
  3450)                 ! tunlramp = 0.765_r8
  3451)               else
  3452)                   tunlramp = tunl
  3453)               endif

  3458)                   lz = min( vk*zi(i,k), tunlramp*lbulk )              
  3459)               endif
  3460)               lz = min(leng_max(k), lz)
  3461)               dzinc = z(i,k-1) - z(i,k)
  3462)               l2n2  = l2n2 + lz*lz*n2(i,k)*dzinc
  3463)               l2s2  = l2s2 + lz*lz*s2(i,k)*dzinc
  3464)               lint  = lint + dzinc
  3465)            end do
  3466) 
  3467)            ! Calculate initial CL stability functions (gh,sh,sm) and net
  3468)            ! internal energy of CL including surface contribution if any.

  3470)          ! Modification : It seems that below cannot be applied when ricrit > 0.19.

  3472) 
  3473)            ricll = min(l2n2/max(l2s2,ntzero),ricrit) ! Mean Ri of internal CL
  3474)            trma  = alph3*alph4*ricll+2._r8*b1*(alph2-alph4*alph5*ricll)

  3476)            trmc  = ricll
  3477)            det   = max(trmb*trmb-4._r8*trma*trmc,0._r8)
  3478)            gh    = (-trmb + sqrt(det))/2._r8/trma
  3479)          ! gh    = min(max(gh,-0.28_r8),0.0233_r8)
  3480)            gh    = min(max(gh,-3.5334_r8),0.0233_r8)
  3481)            sh    = alph5/(1._r8+alph3*gh)
  3482)            sm    = (alph1 + alph2*gh)/(1._r8+alph3*gh)/(1._r8+alph4*gh)
  3483)            wint  = wint - sh*l2n2 + sm*l2s2 
  3484) 
  3485)        else ! The case of SBCL
  3486)  
  3487)            ! If there is no pure internal interface, use only surface interfacial
  3488)            ! values. However, re-set surface interfacial values such  that it can
  3489)            ! be used in the merging tests (either based on 'wint' or 'l2n2')  and
  3490)            ! in such that surface interfacial energy is not double-counted.
  3491)            ! Note that regardless of the choise of 'use_dw_surf', below should be
  3492)            ! kept as it is below, for consistent merging test of extending SBCL. 
  3493)        
  3494)            lint = dlint_surf
  3495)            l2n2 = dl2n2_surf
  3496)            l2s2 = dl2s2_surf 
  3497)            wint = dw_surf
  3498) 
  3499)            ! Aug.29.2006 : Only for the purpose of merging test of extending SRCL
  3500)            ! based on 'l2n2', re-define 'l2n2' of surface interfacial layer using
  3501)            ! 'wint'. This part is designed for similar treatment of merging as in
  3502)            ! the original 'eddy_diff.F90' code,  where 'l2n2' of SBCL was defined
  3503)            ! as 'l2n2 = - wint / sh'. Note that below block is used only when (1)
  3504)            ! surface buoyancy production 'bprod(i,pver+1)' is NOT included in the
  3505)            ! calculation of surface TKE in the initialization of 'bprod(i,pver+1)'
  3506)            ! in the main subroutine ( even though bflxs > 0 ), and (2) to force 
  3507)            ! current scheme be similar to the previous scheme in the treatment of  
  3508)            ! extending-merging test of SBCL based on 'l2n2'. Otherwise below line
  3509)            ! must be commented out. Note at this stage, correct non-zero value of
  3510)            ! 'sh' has been already computed.      
  3511) 
  3512)            if( choice_tkes .eq. 'ebprod' ) then
  3513)                l2n2 = - wint / sh 
  3514)            endif
  3515)            
  3516)        endif
  3517)            
  3518)        ! Set consistent upper limits on 'l2n2' and 'l2s2'. Below limits are
  3519)        ! reasonable since l2n2 of CL interior interface is always negative.
  3520) 
  3521)        l2n2 = -min(-l2n2, tkemax*lint/(b1*sh))
  3522)        l2s2 =  min( l2s2, tkemax*lint/(b1*sm))
  3523)        
  3524)        ! Note that at this stage, ( gh, sh, sm )  are the values of surface
  3525)        ! interfacial layer if there is no pure internal interface, while if
  3526)        ! there is pure internal interface, ( gh, sh, sm ) are the values of
  3527)        ! pure CL interfaces or the values that include both the CL internal
  3528)        ! interfaces and surface interfaces, depending on the 'use_dw_surf'.       
  3529)        
  3530)        ! ----------------------------------------------------------------------- !
  3531)        ! Perform vertical extension-merging process                              !
  3532)        ! ----------------------------------------------------------------------- !
  3533)        ! During the merging process, we assumed ( lbulk, sh, sm ) of CL external !
  3534)        ! interfaces are the same as the ones of the original merging CL. This is !
  3535)        ! an inevitable approximation since we don't know  ( sh, sm ) of external !
  3536)        ! interfaces at this stage.     Note that current default merging test is !
  3537)        ! purely based on buoyancy production without including shear production, !
  3538)        ! since we used 'l2n2' instead of 'wint' as a merging parameter. However, !
  3539)        ! merging test based on 'wint' maybe conceptually more attractable.       !
  3540)        ! Downward CL merging process is identical to the upward merging process, !
  3541)        ! but when the base of extended CL reaches to the surface, surface inter  !
  3542)        ! facial layer contribution to the energetic of extended CL must be done  !
  3543)        ! carefully depending on the sign of surface buoyancy flux. The contribu  !
  3544)        ! tion of surface interfacial layer energetic is included to the internal !
  3545)        ! energetics of merging CL only when bflxs > 0.                           !
  3546)        ! ----------------------------------------------------------------------- !
  3547)        
  3548)        ! ---------------------------- !
  3549)        ! Step 1. Extend the CL upward !
  3550)        ! ---------------------------- !
  3551)        
  3552)        extend = .false.    ! This will become .true. if CL top or base is extended
  3553)
