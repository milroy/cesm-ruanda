   439)     !   	... j-val calculation
   440)     !           spherical irradiance (actinic flux)
   441)     !           as a function of altitude
   442)     !           convert to quanta s-1 nm-1 cm-2
   443)     !           (1.e-4 * (wc*1e-9) / (hc = 6.62e-34 * 2.998e8))
   444)     !----------------------------------------------------------
   445)     rate_loop : do m = 1,nlng
   446)        if( .not. z_dep(m) ) then

   454)           end do
   455)        end if
   456)        prate(1:pverp,m) = prate(1:pverp,m) * adjcoe(pverp:1:-1,m)  
   457)     end do rate_loop
   458)
