     2) subroutine dadadj (lchnk   ,ncol    , &
     3)                    pmid    ,pint    ,pdel    ,t       , &
     4)                    q       )
     5) !----------------------------------------------------------------------- 
     6) ! 
     7) ! Purpose: 
     8) ! GFDL style dry adiabatic adjustment
     9) ! 
    10) ! Method: 
    11) ! if stratification is unstable, adjustment to the dry adiabatic lapse
    12) ! rate is forced subject to the condition that enthalpy is conserved.
    13) ! 
    14) ! Author: CMS Contact J.Hack
    15) ! 
    16) !-----------------------------------------------------------------------
    17)    use shr_kind_mod,    only: r8 => shr_kind_r8
    18)    use ppgrid
    19)    use phys_grid,       only: get_lat_p, get_lon_p
    20)    use physconst,       only: cappa
    21)    use abortutils,      only: endrun
    22)    use cam_control_mod, only: nlvdry
    23)    use cam_logfile,     only: iulog
    24)    implicit none
    25) 
    26)    integer niter           ! number of iterations for convergence
    27)    parameter (niter = 15)
    28) 
    29) !
    30) ! Arguments
    31) !
    32)    integer, intent(in) :: lchnk               ! chunk identifier
    33)    integer, intent(in) :: ncol                ! number of atmospheric columns
    34) 
    35)    real(r8), intent(in) :: pmid(pcols,pver)   ! pressure at model levels
    36)    real(r8), intent(in) :: pint(pcols,pverp)  ! pressure at model interfaces
    37)    real(r8), intent(in) :: pdel(pcols,pver)   ! vertical delta-p
    38) 
    39) !
    40) ! Input/output arguments
    41) !
    42)    real(r8), intent(inout) :: t(pcols,pver)      ! temperature (K)
    43)    real(r8), intent(inout) :: q(pcols,pver)      ! specific humidity
    44) !
    45) !---------------------------Local workspace-----------------------------
    46) !
    47)    integer i,k             ! longitude, level indices
    48)    integer jiter           ! iteration index
    49) 
    50)    real(r8) c1dad(pver)        ! intermediate constant
    51)    real(r8) c2dad(pver)        ! intermediate constant
    52)    real(r8) c3dad(pver)        ! intermediate constant
    53)    real(r8) c4dad(pver)        ! intermediate constant
    54)    real(r8) gammad             ! dry adiabatic lapse rate (deg/Pa)
    55)    real(r8) zeps               ! convergence criterion (deg/Pa)
    56)    real(r8) rdenom             ! reciprocal of denominator of expression
    57)    real(r8) dtdp               ! delta-t/delta-p
    58)    real(r8) zepsdp             ! zeps*delta-p
    59)    real(r8) zgamma             ! intermediate constant
    60)    real(r8) qave               ! mean q between levels
    61) 
    62)    logical ilconv          ! .TRUE. ==> convergence was attained
    63)    logical dodad(pcols)    ! .TRUE. ==> do dry adjustment
    64) !
    65) !-----------------------------------------------------------------------
    66) !
    67)    zeps = 2.0e-5_r8           ! set convergence criteria
    68) !
    69) ! Find gridpoints with unstable stratification
    70) !
    71)    do i=1,ncol
    72)       gammad = cappa*0.5_r8*(t(i,2) + t(i,1))/pint(i,2)
    73)       dtdp = (t(i,2) - t(i,1))/(pmid(i,2) - pmid(i,1))
    74)       dodad(i) = (dtdp + zeps) .gt. gammad
    75)    end do
    76)    do k=2,nlvdry
    77)       do i=1,ncol
    78)          gammad = cappa*0.5_r8*(t(i,k+1) + t(i,k))/pint(i,k+1)
    79)          dtdp = (t(i,k+1) - t(i,k))/(pmid(i,k+1) - pmid(i,k))
    80)          dodad(i) = dodad(i) .or. (dtdp + zeps).gt.gammad
    81)       end do
    82)    end do
    83) !
    84) ! Make a dry adiabatic adjustment
    85) ! Note: nlvdry ****MUST**** be < pver
    86) !
    87)    do 80 i=1,ncol
    88)       if (dodad(i)) then
    89)          zeps = 2.0e-5_r8
    90)          do k=1,nlvdry
    91)             c1dad(k) = cappa*0.5_r8*(pmid(i,k+1)-pmid(i,k))/pint(i,k+1)
    92)             c2dad(k) = (1._r8 - c1dad(k))/(1._r8 + c1dad(k))
    93)             rdenom = 1._r8/(pdel(i,k)*c2dad(k) + pdel(i,k+1))
    94)             c3dad(k) = rdenom*pdel(i,k)
    95)             c4dad(k) = rdenom*pdel(i,k+1)
    96)          end do
    97) 50       do jiter=1,niter
    98)             ilconv = .true.
    99)             do k=1,nlvdry
   100)                zepsdp = zeps*(pmid(i,k+1) - pmid(i,k))
   101)                zgamma = c1dad(k)*(t(i,k) + t(i,k+1))
   102)                if ((t(i,k+1)-t(i,k)) >= (zgamma+zepsdp)) then
   103)                   ilconv = .false.
   104)                   t(i,k+1) = t(i,k)*c3dad(k) + t(i,k+1)*c4dad(k)
   105)                   t(i,k) = c2dad(k)*t(i,k+1)
   106)                   qave = (pdel(i,k+1)*q(i,k+1) + pdel(i,k)*q(i,k))/(pdel(i,k+1)+ pdel(i,k))
   107)                   q(i,k+1) = qave
   108)                   q(i,k) = qave
   109)                end if
   110)             end do
   111)             if (ilconv) go to 80 ! convergence => next longitude
   112)          end do
   113) !
   114) ! Double convergence criterion if no convergence in niter iterations
   115) !
   116)          zeps = zeps + zeps
   117)          if (zeps > 1.e-4_r8) then
   118)             write(iulog,*)'DADADJ: No convergence in dry adiabatic adjustment'
   119)             write(iulog,800) get_lat_p(lchnk,i),get_lon_p(lchnk,i),zeps
   120)             call endrun
   121)          else
   122)             write(iulog,810) zeps,get_lat_p(lchnk,i),get_lon_p(lchnk,i)
   123)             go to 50
   124)          end if
   125)       end if
   126) 80    continue
   127)       return
   128) !
   129) ! Formats
   130) !
   131) 800   format(' lat,lon = ',2i5,', zeps= ',e9.4)
   132) 810   format(//,'DADADJ: Convergence criterion doubled to EPS=',E9.4, &
   133)              ' for'/'        DRY CONVECTIVE ADJUSTMENT at Lat,Lon=', &
   134)              2i5)
   135) end subroutine dadadj
