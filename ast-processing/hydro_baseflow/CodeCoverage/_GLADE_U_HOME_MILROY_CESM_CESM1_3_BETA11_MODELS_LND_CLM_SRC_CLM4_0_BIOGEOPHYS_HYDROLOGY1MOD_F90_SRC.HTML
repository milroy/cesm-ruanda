    39) !
    40) ! !DESCRIPTION:
    41) ! Calculation of
    42) ! (1) water storage of intercepted precipitation
    43) ! (2) direct throughfall and canopy drainage of precipitation
    44) ! (3) the fraction of foliage covered by water and the fraction
    45) !     of foliage that is dry and transpiring.
    46) ! (4) snow layer initialization if the snow accumulation exceeds 10 mm.
    47) ! Note:  The evaporation loss is taken off after the calculation of leaf
    48) ! temperature in the subroutine clm\_leaftem.f90, not in this subroutine.
    49) !
    50) ! !USES:
    51)     use shr_kind_mod , only : r8 => shr_kind_r8
    52)     use clmtype
    53)     use clm_atmlnd   , only : clm_a2l
    54)     use clm_varcon   , only : tfrz, istice, istwet, istsoil, istice_mec, isturb, &
    55)                               icol_roof, icol_sunwall, icol_shadewall
    56)     use clm_varcon   , only : istcrop
    57)     use FracWetMod   , only : FracWet
    58)     use clm_time_manager , only : get_step_size
    59)     use subgridAveMod, only : p2c
    60)     use SNICARMod    , only : snw_rds_min
    61) 
    62) !
    63) ! !ARGUMENTS:
    64)     implicit none
    65)     integer, intent(in) :: lbp, ubp                     ! pft bounds
    66)     integer, intent(in) :: lbc, ubc                     ! column bounds
    67)     integer, intent(in) :: num_nolakec                  ! number of column non-lake points in column filter
    68)     integer, intent(in) :: filter_nolakec(ubc-lbc+1)    ! column filter for non-lake points
    69)     integer, intent(in) :: num_nolakep                  ! number of pft non-lake points in pft filter
    70)     integer, intent(in) :: filter_nolakep(ubp-lbp+1)    ! pft filter for non-lake points
    71) !
    72) ! !CALLED FROM:
    73) ! subroutine clm_driver1
    74) !
    75) ! !REVISION HISTORY:
    76) ! 15 September 1999: Yongjiu Dai; Initial code
    77) ! 15 December 1999:  Paul Houser and Jon Radakovich; F90 Revision
    78) ! 2/15/02, Peter Thornton: Migrated to new data structures. Required
    79) ! adding a PFT loop.
    80) ! 4/26/05, Peter Thornton: Made the canopy interception factor fpi max=0.25
    81) !   the default behavior
    82) !
    83) ! !LOCAL VARIABLES:
    84) !
    85) ! local pointers to original implicit in arrays
    86) !
    87)     real(r8), pointer :: qflx_floodg(:)   ! gridcell flux of flood water from RTM
    88)     real(r8), pointer :: qflx_floodc(:)   ! column flux of flood water from RTM
    89)     integer , pointer :: cgridcell(:)      ! columns's gridcell
    90)     integer , pointer :: clandunit(:)      ! columns's landunit
    91)     integer , pointer :: pgridcell(:)      ! pft's gridcell
    92)     integer , pointer :: plandunit(:)      ! pft's landunit
    93)     integer , pointer :: pcolumn(:)        ! pft's column
    94)     integer , pointer :: npfts(:)          ! number of pfts in column
    95)     integer , pointer :: pfti(:)           ! column's beginning pft index
    96)     integer , pointer :: ltype(:)          ! landunit type
    97)     integer , pointer :: ctype(:)          ! column type
    98)     real(r8), pointer :: forc_rain(:)      ! rain rate [mm/s]
    99)     real(r8), pointer :: forc_snow(:)      ! snow rate [mm/s]
   100)     real(r8), pointer :: forc_t(:)         ! atmospheric temperature (Kelvin)
   101)     logical , pointer :: do_capsnow(:)     ! true => do snow capping
   102)     real(r8), pointer :: t_grnd(:)         ! ground temperature (Kelvin)
   103)     real(r8), pointer :: dewmx(:)          ! Maximum allowed dew [mm]
   104)     integer , pointer :: frac_veg_nosno(:) ! fraction of veg not covered by snow (0/1 now) [-]
   105)     real(r8), pointer :: elai(:)           ! one-sided leaf area index with burying by snow
   106)     real(r8), pointer :: esai(:)           ! one-sided stem area index with burying by snow
   107)     real(r8), pointer :: h2ocan_loss(:)    ! canopy water mass balance term (column)
   108)     real(r8), pointer :: irrig_rate(:)     ! current irrigation rate (applied if n_irrig_steps_left > 0) [mm/s]
   109) !
   110) ! local pointers to original implicit inout arrays
   111) !
   112)     integer , pointer :: snl(:)            ! number of snow layers
   113)     real(r8), pointer :: snowdp(:)         ! snow height (m)
   114)     real(r8), pointer :: h2osno(:)         ! snow water (mm H2O)
   115)     real(r8), pointer :: h2ocan(:)         ! total canopy water (mm H2O)
   116)     real(r8), pointer :: qflx_irrig(:)          ! irrigation amount (mm/s)
   117)     integer, pointer  :: n_irrig_steps_left(:)  ! number of time steps for which we still need to irrigate today
   118) !
   119) ! local pointers to original implicit out arrays
   120) !
   121)     real(r8), pointer :: qflx_prec_intr(:)     ! interception of precipitation [mm/s]
   122)     real(r8), pointer :: qflx_prec_grnd(:)     ! water onto ground including canopy runoff [kg/(m2 s)]
   123)     real(r8), pointer :: qflx_snwcp_liq(:)     ! excess rainfall due to snow capping (mm H2O /s) [+]
   124)     real(r8), pointer :: qflx_snwcp_ice(:)     ! excess snowfall due to snow capping (mm H2O /s) [+]
   125)     real(r8), pointer :: qflx_snow_grnd_pft(:) ! snow on ground after interception (mm H2O/s) [+]
   126)     real(r8), pointer :: qflx_snow_grnd_col(:) ! snow on ground after interception (mm H2O/s) [+]
   127)     real(r8), pointer :: qflx_rain_grnd(:)     ! rain on ground after interception (mm H2O/s) [+]
   128)     real(r8), pointer :: fwet(:)               ! fraction of canopy that is wet (0 to 1)
   129)     real(r8), pointer :: fdry(:)               ! fraction of foliage that is green and dry [-] (new)
   130)     real(r8), pointer :: zi(:,:)               ! interface level below a "z" level (m)
   131)     real(r8), pointer :: dz(:,:)               ! layer depth (m)
   132)     real(r8), pointer :: z(:,:)                ! layer thickness (m)
   133)     real(r8), pointer :: t_soisno(:,:)         ! soil temperature (Kelvin)
   134)     real(r8), pointer :: h2osoi_ice(:,:)       ! ice lens (kg/m2)
   135)     real(r8), pointer :: h2osoi_liq(:,:)       ! liquid water (kg/m2)
   136)     real(r8), pointer :: frac_iceold(:,:)      ! fraction of ice relative to the tot water
   137)     real(r8), pointer :: snw_rds(:,:)          ! effective snow grain radius (col,lyr) [microns, m^-6]
   138)     real(r8), pointer :: mss_bcpho(:,:)        ! mass of hydrophobic BC in snow (col,lyr) [kg]
   139)     real(r8), pointer :: mss_bcphi(:,:)        ! mass of hydrophilic BC in snow (col,lyr) [kg]
   140)     real(r8), pointer :: mss_bctot(:,:)        ! total mass of BC in snow (col,lyr) [kg]
   141)     real(r8), pointer :: mss_bc_col(:)         ! total column mass of BC in snow (col,lyr) [kg]
   142)     real(r8), pointer :: mss_bc_top(:)         ! total top-layer mass of BC (col,lyr) [kg]
   143)     real(r8), pointer :: mss_ocpho(:,:)        ! mass of hydrophobic OC in snow (col,lyr) [kg]
   144)     real(r8), pointer :: mss_ocphi(:,:)        ! mass of hydrophilic OC in snow (col,lyr) [kg]
   145)     real(r8), pointer :: mss_octot(:,:)        ! total mass of OC in snow (col,lyr) [kg]
   146)     real(r8), pointer :: mss_oc_col(:)         ! total column mass of OC in snow (col,lyr) [kg]
   147)     real(r8), pointer :: mss_oc_top(:)         ! total top-layer mass of OC (col,lyr) [kg]
   148)     real(r8), pointer :: mss_dst1(:,:)         ! mass of dust species 1 in snow (col,lyr) [kg]
   149)     real(r8), pointer :: mss_dst2(:,:)         ! mass of dust species 2 in snow (col,lyr) [kg]
   150)     real(r8), pointer :: mss_dst3(:,:)         ! mass of dust species 3 in snow (col,lyr) [kg]
   151)     real(r8), pointer :: mss_dst4(:,:)         ! mass of dust species 4 in snow (col,lyr) [kg]
   152)     real(r8), pointer :: mss_dsttot(:,:)       ! total mass of dust in snow (col,lyr) [kg]
   153)     real(r8), pointer :: mss_dst_col(:)        ! total column mass of dust in snow (col,lyr) [kg]
   154)     real(r8), pointer :: mss_dst_top(:)        ! total top-layer mass of dust in snow (col,lyr) [kg]
   155) !
   156) !
   157) ! !OTHER LOCAL VARIABLES:
   158) !EOP
   159) !
   160)     integer  :: f                            ! filter index
   161)     integer  :: pi                           ! pft index
   162)     integer  :: p                            ! pft index
   163)     integer  :: c                            ! column index
   164)     integer  :: l                            ! landunit index
   165)     integer  :: g                            ! gridcell index
   166)     integer  :: newnode                      ! flag when new snow node is set, (1=yes, 0=no)
   167)     real(r8) :: dtime                        ! land model time step (sec)
   168)     real(r8) :: h2ocanmx                     ! maximum allowed water on canopy [mm]
   169)     real(r8) :: fpi                          ! coefficient of interception
   170)     real(r8) :: xrun                         ! excess water that exceeds the leaf capacity [mm/s]
   171)     real(r8) :: dz_snowf                     ! layer thickness rate change due to precipitation [mm/s]
   172)     real(r8) :: bifall                       ! bulk density of newly fallen dry snow [kg/m3]
   173)     real(r8) :: fracsnow(lbp:ubp)            ! frac of precipitation that is snow
   174)     real(r8) :: fracrain(lbp:ubp)            ! frac of precipitation that is rain
   175)     real(r8) :: qflx_candrip(lbp:ubp)        ! rate of canopy runoff and snow falling off canopy [mm/s]
   176)     real(r8) :: qflx_through_rain(lbp:ubp)   ! direct rain throughfall [mm/s]
   177)     real(r8) :: qflx_through_snow(lbp:ubp)   ! direct snow throughfall [mm/s]
   178)     real(r8) :: qflx_prec_grnd_snow(lbp:ubp) ! snow precipitation incident on ground [mm/s]
   179)     real(r8) :: qflx_prec_grnd_rain(lbp:ubp) ! rain precipitation incident on ground [mm/s]
   180) !-----------------------------------------------------------------------
   181) 
   182)     ! Assign local pointers to derived type members (gridcell-level)
   183) 
   184)     pgridcell          => pft%gridcell
   185)     forc_rain          => clm_a2l%forc_rain
   186)     forc_snow          => clm_a2l%forc_snow
   187) 
   188)     ! Assign local pointers to derived type members (landunit-level)
   189) 
   190)     ltype              => lun%itype
   191) 
   192)     ! Assign local pointers to derived type members (column-level)
   193) 
   194)     qflx_floodg        => clm_a2l%forc_flood
   195)     qflx_floodc        => cwf%qflx_floodc
   196)     cgridcell          => col%gridcell
   197)     clandunit          => col%landunit
   198)     ctype              => col%itype
   199)     pfti               => col%pfti
   200)     npfts              => col%npfts
   201)     do_capsnow         => cps%do_capsnow
   202)     forc_t             => ces%forc_t
   203)     t_grnd             => ces%t_grnd
   204)     snl                => cps%snl
   205)     snowdp             => cps%snowdp
   206)     h2osno             => cws%h2osno
   207)     zi                 => cps%zi
   208)     dz                 => cps%dz
   209)     z                  => cps%z
   210)     frac_iceold        => cps%frac_iceold
   211)     t_soisno           => ces%t_soisno
   212)     h2osoi_ice         => cws%h2osoi_ice
   213)     h2osoi_liq         => cws%h2osoi_liq
   214)     qflx_snow_grnd_col => pwf_a%qflx_snow_grnd
   215)     h2ocan_loss        => cwf%h2ocan_loss
   216)     snw_rds            => cps%snw_rds
   217)     mss_bcpho          => cps%mss_bcpho
   218)     mss_bcphi          => cps%mss_bcphi
   219)     mss_bctot          => cps%mss_bctot
   220)     mss_bc_col         => cps%mss_bc_col
   221)     mss_bc_top         => cps%mss_bc_top
   222)     mss_ocpho          => cps%mss_ocpho
   223)     mss_ocphi          => cps%mss_ocphi
   224)     mss_octot          => cps%mss_octot
   225)     mss_oc_col         => cps%mss_oc_col
   226)     mss_oc_top         => cps%mss_oc_top
   227)     mss_dst1           => cps%mss_dst1
   228)     mss_dst2           => cps%mss_dst2
   229)     mss_dst3           => cps%mss_dst3
   230)     mss_dst4           => cps%mss_dst4
   231)     mss_dsttot         => cps%mss_dsttot
   232)     mss_dst_col        => cps%mss_dst_col
   233)     mss_dst_top        => cps%mss_dst_top
   234) 
   235)     ! Assign local pointers to derived type members (pft-level)
   236) 
   237)     plandunit          => pft%landunit
   238)     pcolumn            => pft%column
   239)     dewmx              => pps%dewmx
   240)     frac_veg_nosno     => pps%frac_veg_nosno
   241)     elai               => pps%elai
   242)     esai               => pps%esai
   243)     h2ocan             => pws%h2ocan
   244)     qflx_prec_intr     => pwf%qflx_prec_intr
   245)     qflx_prec_grnd     => pwf%qflx_prec_grnd
   246)     qflx_snwcp_liq     => pwf%qflx_snwcp_liq
   247)     qflx_snwcp_ice     => pwf%qflx_snwcp_ice
   248)     qflx_snow_grnd_pft => pwf%qflx_snow_grnd
   249)     qflx_rain_grnd     => pwf%qflx_rain_grnd
   250)     fwet               => pps%fwet
   251)     fdry               => pps%fdry
   252)     irrig_rate         => cps%irrig_rate
   253)     n_irrig_steps_left => cps%n_irrig_steps_left
   254)     qflx_irrig         => cwf%qflx_irrig
   255) 
   256)     ! Compute time step
   257) 
   258)     dtime = get_step_size()
   259) 
   260)     ! Start pft loop
   261) 
   262)     do f = 1, num_nolakep
   263)        p = filter_nolakep(f)
   264)        g = pgridcell(p)
   265)        l = plandunit(p)
   266)        c = pcolumn(p)
   267)        
   268)        ! Canopy interception and precipitation onto ground surface
   269)        ! Add precipitation to leaf water
   270) 
   271)        if (ltype(l)==istsoil .or. ltype(l)==istwet .or. ltype(l)==isturb .or. &
   272)            ltype(l)==istcrop) then
   273)           qflx_candrip(p) = 0._r8      ! rate of canopy runoff
   274)           qflx_through_snow(p) = 0._r8 ! rain precipitation direct through canopy
   275)           qflx_through_rain(p) = 0._r8 ! snow precipitation direct through canopy
   276)           qflx_prec_intr(p) = 0._r8    ! total intercepted precipitation
   277)           fracsnow(p) = 0._r8          ! fraction of input precip that is snow
   278)           fracrain(p) = 0._r8          ! fraction of input precip that is rain
   279) 
   280)           if (ctype(c) /= icol_sunwall .and. ctype(c) /= icol_shadewall) then
   281)              if (frac_veg_nosno(p) == 1 .and. (forc_rain(g) + forc_snow(g)) > 0._r8) then
   282) 
   283)                 ! determine fraction of input precipitation that is snow and rain
   284) 
   285)                 fracsnow(p) = forc_snow(g)/(forc_snow(g) + forc_rain(g))
   286)                 fracrain(p) = forc_rain(g)/(forc_snow(g) + forc_rain(g))
   287)                 
   288)                 ! The leaf water capacities for solid and liquid are different,
   289)                 ! generally double for snow, but these are of somewhat less
   290)                 ! significance for the water budget because of lower evap. rate at
   291)                 ! lower temperature.  Hence, it is reasonable to assume that
   292)                 ! vegetation storage of solid water is the same as liquid water.
   293)                 h2ocanmx = dewmx(p) * (elai(p) + esai(p))
   294)                 
   295)                 ! Coefficient of interception
   296)                 ! set fraction of potential interception to max 0.25
   297)                 fpi = 0.25_r8*(1._r8 - exp(-0.5_r8*(elai(p) + esai(p))))
   298)                 
   299)                 ! Direct throughfall
   300)                 qflx_through_snow(p) = forc_snow(g) * (1._r8-fpi)
   301)                 qflx_through_rain(p) = forc_rain(g) * (1._r8-fpi)
   302)                 
   303)                 ! Intercepted precipitation [mm/s]
   304)                 qflx_prec_intr(p) = (forc_snow(g) + forc_rain(g)) * fpi
   305)                 
   306)                 ! Water storage of intercepted precipitation and dew
   307)                 h2ocan(p) = max(0._r8, h2ocan(p) + dtime*qflx_prec_intr(p))
   308)                 
   309)                 ! Initialize rate of canopy runoff and snow falling off canopy
   310)                 qflx_candrip(p) = 0._r8
   311)                 
   312)                 ! Excess water that exceeds the leaf capacity
   313)                 xrun = (h2ocan(p) - h2ocanmx)/dtime
   314)                 
   315)                 ! Test on maximum dew on leaf
   316)                 ! Note if xrun > 0 then h2ocan must be at least h2ocanmx
   317)                 if (xrun > 0._r8) then
   318)                    qflx_candrip(p) = xrun
   319)                    h2ocan(p) = h2ocanmx
   320)                 end if
   321)                 
   322)              end if
   323)           end if
   324) 
   325)        else if (ltype(l)==istice .or. ltype(l)==istice_mec) then
   326) 
   327)           h2ocan(p)            = 0._r8
   328)           qflx_candrip(p)      = 0._r8
   329)           qflx_through_snow(p) = 0._r8
   330)           qflx_through_rain(p) = 0._r8
   331)           qflx_prec_intr(p)    = 0._r8
   332)           fracsnow(p)          = 0._r8
   333)           fracrain(p)          = 0._r8
   334) 
   335)        end if
   336) 
   337)        ! Precipitation onto ground (kg/(m2 s))
   338)        ! PET, 1/18/2005: Added new terms for mass balance correction
   339)        ! due to dynamic pft weight shifting (column-level h2ocan_loss)
   340)        ! Because the fractionation between rain and snow is indeterminate if
   341)        ! rain + snow = 0, I am adding this very small flux only to the rain
   342)        ! components.
   343) 
   344)        if (ctype(c) /= icol_sunwall .and. ctype(c) /= icol_shadewall) then
   345)           if (frac_veg_nosno(p) == 0) then
   346)              qflx_prec_grnd_snow(p) = forc_snow(g)
   347)              qflx_prec_grnd_rain(p) = forc_rain(g) + h2ocan_loss(c)
   348)           else
   349)              qflx_prec_grnd_snow(p) = qflx_through_snow(p) + (qflx_candrip(p) * fracsnow(p))
   350)              qflx_prec_grnd_rain(p) = qflx_through_rain(p) + (qflx_candrip(p) * fracrain(p)) + h2ocan_loss(c)
   351)           end if
   352)        ! Urban sunwall and shadewall have no intercepted precipitation
   353)        else
   354)           qflx_prec_grnd_snow(p) = 0.
   355)           qflx_prec_grnd_rain(p) = 0.
   356)        end if
   357) 
   358)        ! Determine whether we're irrigating here; set qflx_irrig appropriately
   359)        if (n_irrig_steps_left(c) > 0) then
   360)           qflx_irrig(c)         = irrig_rate(c)
   361)           n_irrig_steps_left(c) = n_irrig_steps_left(c) - 1
   362)        else
   363)           qflx_irrig(c) = 0._r8
   364)        end if
   365) 
   366)        ! Add irrigation water directly onto ground (bypassing canopy interception)
   367)        ! Note that it's still possible that (some of) this irrigation water will runoff (as runoff is computed later)
   368)        qflx_prec_grnd_rain(p) = qflx_prec_grnd_rain(p) + qflx_irrig(c)
   369) 
   370)        ! Done irrigation
   371) 
   372)        qflx_prec_grnd(p) = qflx_prec_grnd_snow(p) + qflx_prec_grnd_rain(p)
   373) 
   374)        if (do_capsnow(c)) then
   375)           qflx_snwcp_liq(p) = qflx_prec_grnd_rain(p)
   376)           qflx_snwcp_ice(p) = qflx_prec_grnd_snow(p)
   377) 
   378)           qflx_snow_grnd_pft(p) = 0._r8
   379)           qflx_rain_grnd(p) = 0._r8
   380)        else
   381)           qflx_snwcp_liq(p) = 0._r8
   382)           qflx_snwcp_ice(p) = 0._r8
   383)           qflx_snow_grnd_pft(p) = qflx_prec_grnd_snow(p)           ! ice onto ground (mm/s)
   384)           qflx_rain_grnd(p)     = qflx_prec_grnd_rain(p)           ! liquid water onto ground (mm/s)
   385)        end if
   386) 
   387)     end do ! (end pft loop)
   388) 
   389)     ! Determine the fraction of foliage covered by water and the
   390)     ! fraction of foliage that is dry and transpiring.
   391) 
   392)     call FracWet(num_nolakep, filter_nolakep)
   393) 
   394)     ! Update column level state variables for snow.
   395) 
   396)     call p2c(num_nolakec, filter_nolakec, qflx_snow_grnd_pft, qflx_snow_grnd_col)
   397) 
   398) !rtm_flood: apply gridcell flood water flux to non-lake columns
   399) !                    no inputs to urban wall columns, as above with atm inputs
   400) !dir$ concurrent
   401) !cdir nodep
   402)     do f = 1, num_nolakec
   403)        c = filter_nolakec(f)
   404)        g = cgridcell(c)
   405)        if (ctype(c) /= icol_sunwall .and. ctype(c) /= icol_shadewall) then      
   406)           qflx_floodc(c) = qflx_floodg(g)
   407)        else
   408)           qflx_floodc(c) = 0._r8
   409)        endif    
   410)     enddo
   411) !rtm_flood
