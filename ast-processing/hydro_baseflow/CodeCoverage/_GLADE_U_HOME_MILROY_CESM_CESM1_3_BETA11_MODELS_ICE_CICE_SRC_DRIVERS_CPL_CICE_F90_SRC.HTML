    46) ! !DESCRIPTION:

    51) !
    52) ! !REVISION HISTORY:
    53) !
    54) ! author Elizabeth C. Hunke, LANL
    55) !
    56) ! !INTERFACE:
    57) !
    58)       subroutine debug_ice(plabeld)
    59) !
    60) ! !USES:
    61) !
    62)       use ice_kinds_mod
    63)       use ice_diagnostics
    64)       use ice_domain, only: nblocks
    65)       use ice_blocks, only: nx_block, ny_block
    66) !
    67) ! !INPUT/OUTPUT PARAMETERS:
    68) !
    69)       character (char_len), intent(in) :: plabeld
    70) !
    71) !EOP
    72) !
    73)       integer (kind=int_kind) :: i, j, iblk
    74) 
    75)       do iblk = 1, nblocks
    76)       do j = 1, ny_block
    77)       do i = 1, nx_block
    78)          if (iblk==iblkp .and. i==ip .and. j==jp .and. my_task==mtask) &
    79)               call print_state(plabeld,i,j,iblk)
    80)       enddo
    81)       enddo
