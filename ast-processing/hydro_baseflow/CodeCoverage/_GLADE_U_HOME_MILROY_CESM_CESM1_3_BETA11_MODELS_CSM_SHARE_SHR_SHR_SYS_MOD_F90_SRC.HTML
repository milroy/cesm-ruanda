    57) SUBROUTINE shr_sys_system(str,rcode)
    58) 
    59)    IMPLICIT none
    60) 
    61)    !----- arguments ---
    62)    character(*)        ,intent(in)  :: str    ! system/shell command string
    63)    integer(SHR_KIND_IN),intent(out) :: rcode  ! function return error code
    64) 
    65)    !----- functions -----
    66) #if (defined CRAY) || (defined UNICOSMP)
    67)    integer(SHR_KIND_IN),external    :: ishell ! function to envoke shell command
    68) #endif
    69) #if (defined OSF1 || defined SUNOS || (defined LINUX && !defined CPRGNU && !defined CATAMOUNT))
    70)    integer(SHR_KIND_IN),external    :: system ! function to envoke shell command
    71) #endif
    72) 
    73)    !----- local -----
    74) #if (defined CATAMOUNT)
    75)    character(2*SHR_KIND_CL) :: file1            ! one or two filenames
    76)    character(  SHR_KIND_CL) :: file2            ! 2nd file name
    77)    integer(SHR_KIND_IN)     :: iloc             ! index/location within a string
    78) #endif
    79) 
    80)    !----- formats -----
    81)    character(*),parameter :: subName =   '(shr_sys_system) '
    82)    character(*),parameter :: F00     = "('(shr_sys_system) ',4a)"
    83) 
    84) !-------------------------------------------------------------------------------
    85) ! PURPOSE: an architecture independent system call
    86) ! NOTE: 
    87) ! - for Catamount (Cray, pheonix at ORNL) there is no system call -- workarounds 
    88) !   exist only for simple "rm" and "cp" commands
    89) !-------------------------------------------------------------------------------
    90)    rcode = 0
    91) #ifdef HAVE_EXECUTE
    92)    call execute_command_line(str,exitstat=rcode)   ! Intrinsic as of F2008
    93) #else
    94) #if (defined CRAY) || (defined UNICOSMP)
    95) 
    96)    rcode=ishell(str)
    97) 
    98) #elif (defined IRIX64 || defined NEC_SX)
    99) 
   100) 
   101)    call system(str)
   102) 
   103) #elif (defined AIX)
   104) 
   105)    call system(str,rcode)
   106) 
   107) #elif (defined OSF1 || defined SUNOS || defined CPRGNU || (defined LINUX && !defined CATAMOUNT))
   108) 
   109)    rcode = system(str)
   110) 
   111) #elif (defined CATAMOUNT)
   112)    if (str(1:3) == 'rm ') then
   113)       call unlink(str(4:))
   114)       if (s_loglev > 0) write(s_logunit,F00) 'CATAMOUNT unlink ',trim(str(4:))
   115)       rcode = 0
   116)    elseif (str(1:3) == 'mv ') then
   117)       file1 = str(4:)
   118)       iloc = index(file1,' ') + 3
   119)       if (iloc < 6) then
   120)          if (s_loglev > 0) write(s_logunit,*) 'CATAMOUNT mv error ',trim(str),iloc
   121)          rcode = -1
   122)       else
   123)          file1 = str(4:iloc)
   124)          file2 = str(iloc+1:)
   125)          call rename(trim(file1),trim(file2))
   126)          if (s_loglev > 0) write(s_logunit,F00) 'CATAMOUNT rename ',trim(file1)," ",trim(file2)
   127)          rcode = 0
   128)       endif
   129)    else
   130)       rcode = -1
   131)    endif
   132) 
   133) #else
   134) 
   135)    write(s_logunit,F00) 'ERROR: no implementation of system call for this architecture'
   136)    call shr_sys_abort(subName//'no implementation of system call for this architecture')
   137) #endif
   138) #endif
   139) 
   140) END SUBROUTINE shr_sys_system

   252) SUBROUTINE shr_sys_abort(string,rc)
   253) 
   254)    IMPLICIT none
   255) 
   256)    character(*)        ,optional :: string  ! error message string
   257)    integer(SHR_KIND_IN),optional :: rc      ! error code
   258) 
   259)    !----- local -----
   260)    logical              :: flag
   261) 
   262)    !----- formats -----
   263)    character(*),parameter :: subName =   '(shr_sys_abort) '
   264)    character(*),parameter :: F00     = "('(shr_sys_abort) ',4a)"
   265) 
   266)    ! Local version of the string.
   267)    ! (Gets a default value if string is not present.)
   268)    character(len=shr_kind_cx) :: local_string
   269) 
   270) !-------------------------------------------------------------------------------
   271) ! PURPOSE: consistent stopping mechanism
   272) !-------------------------------------------------------------------------------
   273) 
   274)    if (present(string)) then
   275)       local_string = trim(string)
   276)    else
   277)       local_string = "Unknown error submitted to shr_sys_abort."
   278)    end if
   279) 
   280)    call print_error_to_logs("ERROR", local_string)
   281) 
   282)    call shr_sys_backtrace()
   283) 
   284)    call shr_mpi_initialized(flag)
   285) 
   286)    if (flag) then
   287)       if (present(rc)) then
   288)          call shr_mpi_abort(trim(local_string),rc)
   289)       else
   290)          call shr_mpi_abort(trim(local_string))
   291)       endif
   292)    endif
   293) 
   294)   ! A compiler's abort method may print a backtrace or do other nice
   295)   ! things, but in fact we can rarely leverage this, because MPI_Abort
   296)   ! usually sends SIGTERM to the process, and we don't catch that signal.
   297)    call abort()
   298)

   312)    integer(SHR_KIND_IN)      :: count

   314)    integer(SHR_KIND_IN)      :: count_max
   315)    integer(SHR_KIND_IN),save :: last_count   = -1

   317) 
   318)    !----- formats -----
   319)    character(*),parameter :: subName =   '(shr_sys_irtc) '
   320)    character(*),parameter :: F00     = "('(shr_sys_irtc) ',4a)"
   321) 
   322) !-------------------------------------------------------------------------------
   323) ! emulates Cray/SGI irtc function (returns clock tick since last reboot)
   324) !-------------------------------------------------------------------------------
   325) 
   326)    call system_clock(count=count,count_rate=count_rate, count_max=count_max)
   327)    if ( present(rate) ) rate = count_rate
   328)    shr_sys_irtc = count
   329) 
   330)    !--- adjust for clock wrap-around ---
   331)    if ( last_count /= -1 ) then
   332)      if ( count < last_count ) count_offset = count_offset + count_max
   333)    end if
   334)    shr_sys_irtc = shr_sys_irtc + count_offset
   335)    last_count   = count
   336)

   420) subroutine shr_sys_backtrace()
   421) 
   422)   ! This routine uses compiler-specific facilities to print a backtrace to
   423)   ! error_unit (standard error, usually unit 0).
   424) 
   425) #if defined(CPRIBM)
   426) 
   427)   ! This theoretically should be in xlfutility, but using it from that
   428)   ! module doesn't seem to always work.
   429)   interface
   430)      subroutine xl_trbk()
   431)      end subroutine xl_trbk
   432)   end interface
   433) 
   434)   call xl__trbk()
   435) 
   436) #elif defined(CPRGNU) && (__GNUC__ > 4 || (__GNUC__ == 4 && __GNUC_MINOR__ >= 8 ))
   437) 
   438)   ! gfortran 4.8 and later implement this intrinsic. We explicitly call it
   439)   ! out as such to make sure that it really is available, just in case the
   440)   ! CPP logic above screws up.
   441)   intrinsic :: backtrace
   442) 
   443)   call backtrace()
   444) 
   445) #elif defined(CPRINTEL)
   446) 
   447)   ! tracebackqq uses optional arguments, so *must* have an explicit
   448)   ! interface.
   449)   use ifcore, only: tracebackqq
   450) 
   451)   ! An exit code of -1 is a special value that prevents this subroutine
   452)   ! from aborting the run.
   453)   call tracebackqq(user_exit_code=-1)
   454) 
   455) #else
   456) 
   457)   ! Currently we have no means to request a backtrace from the NAG runtime,
   458)   ! even though it is capable of emitting backtraces itself, if you use the
   459)   ! "-gline" option.
   460) 
   461)   ! Similarly, PGI has a -traceback option, but no user interface for
   462)   ! requesting a backtrace to be printed.
   463) 
   464) #endif
   465) 
   466)   flush(error_unit)
   467) 
   468) end subroutine shr_sys_backtrace

   473) !
   474) ! This routine prints error messages to s_logunit (which is standard output
   475) ! for most tasks in CESM) and also to standard error if s_logunit is a
   476) ! file.
   477) !
   478) ! It also flushes these output units.
   479) !
   480) subroutine print_error_to_logs(error_type, message)
   481)   character(len=*), intent(in) :: error_type, message
   482) 
   483)   integer, allocatable :: log_units(:)
   484) 
   485)   integer :: i
   486) 
   487)   if (s_logunit == output_unit .or. s_logunit == error_unit) then
   488)      ! If the log unit number is standard output or standard error, just
   489)      ! print to that.
   490)      allocate(log_units(1), source=[s_logunit])
   491)   else
   492)      ! Otherwise print the same message to both the log unit and standard
   493)      ! error.
   494)      allocate(log_units(2), source=[error_unit, s_logunit])
   495)   end if
   496) 
   497)   do i = 1, size(log_units)
   498)      write(log_units(i),*) trim(error_type), ": ", trim(message)
   499)      flush(log_units(i))
   500)   end do
   501) 
   502) end subroutine print_error_to_logs
