     1) module vertical_diffusion
     2) 
     3)   !----------------------------------------------------------------------------------------------------- !
     4)   ! Module to compute vertical diffusion of momentum,  moisture, trace constituents                      !
     5)   ! and static energy. Separate modules compute                                                          !
     6)   !   1. stresses associated with turbulent flow over orography                                          !
     7)   !      ( turbulent mountain stress )                                                                   !
     8)   !   2. eddy diffusivities, including nonlocal tranport terms                                           !
     9)   !   3. molecular diffusivities                                                                         !
    10)   ! Lastly, a implicit diffusion solver is called, and tendencies retrieved by                           !
    11)   ! differencing the diffused and initial states.                                                        !
    12)   !                                                                                                      !
    13)   ! Calling sequence:                                                                                    !
    14)   !                                                                                                      !
    15)   !  vertical_diffusion_init      Initializes vertical diffustion constants and modules                  !
    16)   !        init_molec_diff        Initializes molecular diffusivity module                               !
    17)   !        init_eddy_diff         Initializes eddy diffusivity module (includes PBL)                     !
    18)   !        init_tms               Initializes turbulent mountain stress module                           !
    19)   !        init_vdiff             Initializes diffusion solver module                                    !
    20)   !  vertical_diffusion_ts_init   Time step initialization (only used for upper boundary condition)      !
    21)   !  vertical_diffusion_tend      Computes vertical diffusion tendencies                                 !
    22)   !        compute_tms            Computes turbulent mountain stresses                                   !
    23)   !        compute_eddy_diff      Computes eddy diffusivities and countergradient terms                  !
    24)   !        compute_vdiff          Solves vertical diffusion equations, including molecular diffusivities !
    25)   !                                                                                                      !

   142)   subroutine vd_readnl(nlfile)
   143) 
   144)     use namelist_utils,  only: find_group_name
   145)     use units,           only: getunit, freeunit
   146)     use mpishorthand
   147)     use spmd_utils,      only: masterproc
   148)   
   149)     character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
   150)   
   151)     ! Local variables
   152)     integer :: unitn, ierr
   153)     character(len=*), parameter :: subname = 'vd_readnl'
   154)   
   155)     namelist /vert_diff_nl/ kv_top_pressure, kv_top_scale, kv_freetrop_scale, eddy_lbulk_max, eddy_leng_max, &
   156)          eddy_max_bot_pressure, eddy_moist_entrain_a2l, diff_cnsrv_mass_check, do_iss
   157)     !-----------------------------------------------------------------------------
   158)   
   159)     if (masterproc) then
   160)       unitn = getunit()
   161)       open( unitn, file=trim(nlfile), status='old' )
   162)       call find_group_name(unitn, 'vert_diff_nl', status=ierr)
   163)       if (ierr == 0) then
   164)         read(unitn, vert_diff_nl, iostat=ierr)
   165)         if (ierr /= 0) then
   166)           call endrun(subname // ':: ERROR reading namelist')
   167)         end if
   168)       end if
   169)       close(unitn)
   170)       call freeunit(unitn)
   171)     end if
   172)   
   173) #ifdef SPMD
   174)     ! Broadcast namelist variables
   175)     call mpibcast(kv_top_pressure,                 1 , mpir8,   0, mpicom)
   176)     call mpibcast(kv_top_scale,                    1 , mpir8,   0, mpicom)
   177)     call mpibcast(kv_freetrop_scale,               1 , mpir8,   0, mpicom)
   178)     call mpibcast(eddy_lbulk_max,                  1 , mpir8,   0, mpicom)
   179)     call mpibcast(eddy_leng_max,                   1 , mpir8,   0, mpicom)
   180)     call mpibcast(eddy_max_bot_pressure,           1 , mpir8,   0, mpicom)
   181)     call mpibcast(eddy_moist_entrain_a2l,          1 , mpir8,   0, mpicom)
   182)     call mpibcast(diff_cnsrv_mass_check,           1 , mpilog,  0, mpicom)
   183)     call mpibcast(do_iss,                          1 , mpilog,  0, mpicom)

   241)   subroutine vertical_diffusion_init(pbuf2d)
   242) 
   243)     !------------------------------------------------------------------!
   244)     ! Initialization of time independent fields for vertical diffusion !
   245)     ! Calls initialization routines for subsidiary modules             !
   246)     !----------------------------------------------------------------- !
   247) 
   248)     use cam_history,       only : addfld, add_default, phys_decomp
   249)     use eddy_diff,         only : init_eddy_diff
   250)     use hb_diff,           only : init_hb_diff
   251)     use molec_diff,        only : init_molec_diff
   252)     use trb_mtn_stress,    only : init_tms
   253)     use diffusion_solver,  only : init_vdiff, new_fieldlist_vdiff, vdiff_select
   254)     use constituents,      only : cnst_get_ind, cnst_get_type_byind, cnst_name, cnst_get_molec_byind
   255)     use spmd_utils,        only : masterproc
   256)     use ref_pres,          only : ntop_molec, nbot_molec, press_lim_idx, pref_mid
   257)     use physics_buffer,    only : pbuf_set_field, pbuf_get_index, physics_buffer_desc
   258)     use rad_constituents,  only : rad_cnst_get_info, rad_cnst_get_mode_num_idx, &
   259)                                   rad_cnst_get_mam_mmr_idx
   260) 
   261)     type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   262)     character(128) :: errstring   ! Error status for init_vdiff
   263)     integer        :: ntop_eddy   ! Top    interface level to which eddy vertical diffusion is applied ( = 1 )
   264)     integer        :: nbot_eddy   ! Bottom interface level to which eddy vertical diffusion is applied ( = pver )
   265)     integer        :: k           ! Vertical loop index
   266) 
   267)     real(r8), parameter :: ntop_eddy_pres = 1.e-5_r8 ! Pressure below which eddy diffusion is not done in WACCM-X. (Pa)
   268) 
   269)     integer :: im, l, m, nmodes, nspec
   270) 
   271)     ! ----------------------------------------------------------------- !
   272) 
   273)     if (masterproc) then
   274)        write(iulog,*)'Initializing vertical diffusion (vertical_diffusion_init)'
   275)     end if
   276) 
   277)     ! ----------------------------------------------------------------- !
   278)     ! Get indices of cloud liquid and ice within the constituents array !
   279)     ! ----------------------------------------------------------------- !
   280) 
   281)     call cnst_get_ind( 'CLDLIQ', ixcldliq )
   282)     call cnst_get_ind( 'CLDICE', ixcldice )
   283)     if( microp_scheme == 'MG' ) then
   284)         call cnst_get_ind( 'NUMLIQ', ixnumliq )
   285)         call cnst_get_ind( 'NUMICE', ixnumice )
   286)     endif
   287) 
   288)     ! prog_modal_aero determines whether prognostic modal aerosols are present in the run.
   289)     call phys_getopts(prog_modal_aero_out=prog_modal_aero)
   290)     if (prog_modal_aero) then
   291) 
   292)        ! Get the constituent indices of the number and mass mixing ratios of the modal 
   293)        ! aerosols.
   294)        !
   295)        ! N.B. - This implementation assumes that the prognostic modal aerosols are 
   296)        !        impacting the climate calculation (i.e., can get info from list 0).
   297)        ! 
   298) 
   299)        ! First need total number of mam constituents
   300)        call rad_cnst_get_info(0, nmodes=nmodes)
   301)        do m = 1, nmodes
   302)           call rad_cnst_get_info(0, m, nspec=nspec)
   303)           pmam_ncnst = pmam_ncnst + 1 + nspec
   304)        end do
   305) 
   306)        allocate(pmam_cnst_idx(pmam_ncnst))
   307) 
   308)        ! Get the constituent indicies
   309)        im = 1
   310)        do m = 1, nmodes
   311)           call rad_cnst_get_mode_num_idx(m, pmam_cnst_idx(im))
   312)           im = im + 1
   313)           call rad_cnst_get_info(0, m, nspec=nspec)
   314)           do l = 1, nspec
   315)              call rad_cnst_get_mam_mmr_idx(m, l, pmam_cnst_idx(im))
   316)              im = im + 1
   317)           end do
   318)        end do
   319)     end if
   320) 
   321)     ! ---------------------------------------------------------------------------------------- !
   322)     ! Initialize molecular diffusivity module                                                  !
   323)     ! Note that computing molecular diffusivities is a trivial expense, but constituent        !
   324)     ! diffusivities depend on their molecular weights. Decomposing the diffusion matric        !
   325)     ! for each constituent is a needless expense unless the diffusivity is significant.        !
   326)     ! ---------------------------------------------------------------------------------------- !
   327) 
   328)     !----------------------------------------------------------------------------------------
   329)     ! Initialize molecular diffusion and get top and bottom molecular diffusion limits
   330)     !----------------------------------------------------------------------------------------
   331) 
   332)     if( do_molec_diff ) then
   333)        call init_molec_diff( r8, pcnst, rair, mwdry, avogad, gravit, &
   334)             cpair, boltz, errstring)
   335) 
   336)        call handle_errmsg(errstring, subname="init_molec_diff")
   337) 
   338)        call addfld( 'TTPXMLC', 'K/S', 1, 'A', 'Top interf. temp. flux: molec. viscosity', phys_decomp )
   339)        if( masterproc ) write(iulog,fmt='(a,i3,5x,a,i3)') 'NTOP_MOLEC =', ntop_molec, 'NBOT_MOLEC =', nbot_molec
   340)     end if
   341) 
   342)     ! ---------------------------------- !    
   343)     ! Initialize eddy diffusivity module !
   344)     ! ---------------------------------- !
   345)  
   346)     ! ntop_eddy must be 1 or <= nbot_molec
   347)     ! Currently, it is always 1 except for WACCM-X.
   348)     if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then
   349)        ntop_eddy  = press_lim_idx(ntop_eddy_pres, top=.true.)
   350)     else
   351)        ntop_eddy = 1
   352)     end if
   353)     nbot_eddy  = pver
   354) 
   355)     if (masterproc) write(iulog, fmt='(a,i3,5x,a,i3)') 'NTOP_EDDY  =', ntop_eddy, 'NBOT_EDDY  =', nbot_eddy
   356) 
   357)     select case ( eddy_scheme )
   358)     case ( 'diag_TKE' ) 
   359)         if( masterproc ) write(iulog,*) &
   360)              'vertical_diffusion_init: eddy_diffusivity scheme: UW Moist Turbulence Scheme by Bretherton and Park'
   361)         call init_eddy_diff( r8, pver, gravit, cpair, rair, zvir, latvap, latice, &
   362)                              ntop_eddy, nbot_eddy, karman, eddy_lbulk_max, eddy_leng_max, &
   363)                              eddy_max_bot_pressure, eddy_moist_entrain_a2l)
   364)         if( masterproc ) write(iulog,*) 'vertical_diffusion: nturb, ntop_eddy, nbot_eddy ', nturb, ntop_eddy, nbot_eddy
   365)     case ( 'HB', 'HBR')
   366)         if( masterproc ) write(iulog,*) 'vertical_diffusion_init: eddy_diffusivity scheme:  Holtslag and Boville'
   367)         call init_hb_diff(gravit, cpair, ntop_eddy, nbot_eddy, pref_mid, &
   368)                           karman, eddy_scheme)
   369)         call addfld('HB_ri', 'no',      pver,  'A',  'Richardson Number (HB Scheme), I',  phys_decomp )
   370)     end select
   371)     
   372)     ! The vertical diffusion solver must operate 
   373)     ! over the full range of molecular and eddy diffusion
   374) 
   375)     ntop = min(ntop_molec,ntop_eddy)
   376)     nbot = max(nbot_molec,nbot_eddy)
   377)     
   378)     ! ------------------------------------------- !
   379)     ! Initialize turbulent mountain stress module !
   380)     ! ------------------------------------------- !
   381) 
   382)     if( do_tms ) then
   383)        call init_tms( r8, tms_orocnst, tms_z0fac, karman, gravit, rair, errstring )
   384) 
   385)        call handle_errmsg(errstring, subname="init_tms")
   386) 
   387)        call addfld( 'TAUTMSX' ,'N/m2  ',  1,  'A',  'Zonal      turbulent mountain surface stress',  phys_decomp )
   388)        call addfld( 'TAUTMSY' ,'N/m2  ',  1,  'A',  'Meridional turbulent mountain surface stress',  phys_decomp )
   389) 
   390)        if (history_amwg) then
   391)           call add_default('TAUTMSX ', 1, ' ')
   392)           call add_default('TAUTMSY ', 1, ' ')
   393)        end if
   394) 
   395)        if (masterproc) then
   396)           write(iulog,*)'Using turbulent mountain stress module'
   397)           write(iulog,*)'  tms_orocnst = ',tms_orocnst
   398)           write(iulog,*)'  tms_z0fac = ',tms_z0fac
   399)        end if
   400)     endif
   401)     
   402)     ! ---------------------------------- !
   403)     ! Initialize diffusion solver module !
   404)     ! ---------------------------------- !
   405) 
   406)     call init_vdiff( r8, iulog, rair, gravit, do_iss, errstring )
   407)     call handle_errmsg(errstring, subname="init_vdiff")
   408) 
   409)     ! Use fieldlist_wet to select the fields which will be diffused using moist mixing ratios ( all by default )
   410)     ! Use fieldlist_dry to select the fields which will be diffused using dry   mixing ratios.
   411) 
   412)     fieldlist_wet = new_fieldlist_vdiff( pcnst)
   413)     fieldlist_dry = new_fieldlist_vdiff( pcnst)
   414)     fieldlist_molec = new_fieldlist_vdiff( pcnst)
   415) 
   416)     if( vdiff_select( fieldlist_wet, 'u' ) .ne. '' ) call endrun( vdiff_select( fieldlist_wet, 'u' ) )
   417)     if( vdiff_select( fieldlist_wet, 'v' ) .ne. '' ) call endrun( vdiff_select( fieldlist_wet, 'v' ) )
   418)     if( vdiff_select( fieldlist_wet, 's' ) .ne. '' ) call endrun( vdiff_select( fieldlist_wet, 's' ) )
   419) 
   420)     constit_loop: do k = 1, pcnst
   421) 
   422)        if (prog_modal_aero) then
   423)      ! Do not diffuse droplet number - treated in dropmixnuc
   424)           if (k == ixnumliq) cycle constit_loop
   425)           ! Don't diffuse modal aerosol - treated in dropmixnuc
   426)           do m = 1, pmam_ncnst
   427)              if (k == pmam_cnst_idx(m)) cycle constit_loop
   428)        enddo

   758)     real(r8) :: t_aftPBL(pcols,pver)                                ! Temperature after PBL diffusion
   759)     real(r8) :: tten(pcols,pver)                                    ! Temperature tendency by PBL diffusion
   760)     real(r8) :: rhten(pcols,pver)                                   ! RH tendency by PBL diffusion
   761)     real(r8) :: qv_aft_PBL(pcols,pver)                              ! qv after PBL diffusion
   762)     real(r8) :: ql_aft_PBL(pcols,pver)                              ! ql after PBL diffusion
   763)     real(r8) :: qi_aft_PBL(pcols,pver)                              ! qi after PBL diffusion
   764)     real(r8) :: s_aft_PBL(pcols,pver)                               ! s after PBL diffusion
   765)     real(r8) :: u_aft_PBL(pcols,pver)                               ! u after PBL diffusion
   766)     real(r8) :: v_aft_PBL(pcols,pver)                               ! v after PBL diffusion
   767)     real(r8) :: qv_pro(pcols,pver) 
   768)     real(r8) :: ql_pro(pcols,pver)
   769)     real(r8) :: qi_pro(pcols,pver)
   770)     real(r8) :: s_pro(pcols,pver)
   771)     real(r8) :: t_pro(pcols,pver)
   772)     real(r8), pointer :: tauresx(:)                                      ! Residual stress to be added in vdiff to correct
   773)     real(r8), pointer :: tauresy(:)                                      ! for turb stress mismatch between sfc and atm accumulated.
   774)     integer(i4), pointer :: ipbl(:)
   775)     integer(i4), pointer :: kpblh(:)
   776)     real(r8), pointer :: wstarPBL(:)
   777)     real(r8), pointer :: tkes(:)
   778)     real(r8), pointer :: went(:)
   779)     real(r8) :: tpertPBL(pcols)
   780)     real(r8) :: qpertPBL(pcols)
   781)     real(r8) :: rairi(pcols,pver+1)                                 ! interface gas constant needed for compute_vdiff
   782) 
   783)     real(r8), pointer :: tpert(:)
   784)     real(r8), pointer :: qpert(:)
   785)     real(r8), pointer :: pblh(:)
   786) 
   787)     real(r8) :: tmp1(pcols)                                         ! Temporary storage
   788) 
   789)     integer  :: nstep
   790)     real(r8) :: sum1, sum2, sum3, pdelx 
   791)     real(r8) :: sflx
   792) 
   793)     ! Copy state so we can pass to intent(inout) routines that return
   794)     ! new state instead of a tendency.
   795)     real(r8) :: s_tmp(pcols,pver)
   796)     real(r8) :: u_tmp(pcols,pver)
   797)     real(r8) :: v_tmp(pcols,pver)
   798)     real(r8) :: q_tmp(pcols,pver,pcnst)
   799) 
   800)     logical  :: lq(pcnst)
   801) 
   802)     ! ----------------------- !
   803)     ! Main Computation Begins !
   804)     ! ----------------------- !
   805) 
   806)     rztodt = 1._r8 / ztodt
   807)     lchnk  = state%lchnk
   808)     ncol   = state%ncol
   809) 
   810)     call pbuf_get_field(pbuf, tauresx_idx,  tauresx)
   811)     call pbuf_get_field(pbuf, tauresy_idx,  tauresy)
   812)     call pbuf_get_field(pbuf, tpert_idx,    tpert)
   813)     call pbuf_get_field(pbuf, qpert_idx,    qpert)
   814)     call pbuf_get_field(pbuf, pblh_idx,     pblh)
   815)     call pbuf_get_field(pbuf, turbtype_idx, turbtype)
   816) 
   817)     ! ---------------------------------------- !
   818)     ! Computation of turbulent mountain stress !
   819)     ! ---------------------------------------- !
   820)    
   821)     ! Consistent with the computation of 'normal' drag coefficient, we are using 
   822)     ! the raw input (u,v) to compute 'ksrftms', not the provisionally-marched 'u,v' 
   823)     ! within the iteration loop of the PBL scheme. 
   824) 
   825)     if( do_tms ) then
   826)         call compute_tms( pcols      , pver     , ncol    ,              &
   827)                           state%u    , state%v  , state%t , state%pmid , & 
   828)                           state%exner, state%zm , sgh     , ksrftms    , & 
   829)                           tautmsx    , tautmsy  , landfrac )
   830)       ! Here, both 'taux, tautmsx' are explicit surface stresses.        
   831)       ! Note that this 'tautotx, tautoty' are different from the total stress
   832)       ! that has been actually added into the atmosphere. This is because both
   833)       ! taux and tautmsx are fully implicitly treated within compute_vdiff.
   834)       ! However, 'tautotx, tautoty' are not used in the actual numerical
   835)       ! computation in this module.   
   836)         tautotx(:ncol) = taux(:ncol) + tautmsx(:ncol)
   837)         tautoty(:ncol) = tauy(:ncol) + tautmsy(:ncol)
   838)     else
   839)         ksrftms(:ncol) = 0._r8
   840)         tautotx(:ncol) = taux(:ncol)
   841)         tautoty(:ncol) = tauy(:ncol)
   842)     endif
   843) 
   844)     !----------------------------------------------------------------------- !
   845)     !   Computation of eddy diffusivities - Select appropriate PBL scheme    !
   846)     !----------------------------------------------------------------------- !
   847)     call pbuf_get_field(pbuf, kvm_idx,  kvm_in)
   848)     call pbuf_get_field(pbuf, kvh_idx,  kvh_in)
   849)     call pbuf_get_field(pbuf, smaw_idx, smaw)
   850)     call pbuf_get_field(pbuf, tke_idx,  tke)
   851) 
   852) 
   853)     select case (eddy_scheme)
   854)     case ( 'diag_TKE' ) 
   855) 
   856)        ! ---------------------------------------------------------------- !
   857)        ! At first time step, have eddy_diff.F90:caleddy() use kvh=kvm=kvf !
   858)        ! This has to be done in compute_eddy_diff after kvf is calculated !
   859)        ! ---------------------------------------------------------------- !
   860) 
   861)        if( is_first_step() ) then
   862)            kvinit = .true.
   863)        else
   864)            kvinit = .false.
   865)        endif
   866) 
   867)        ! ---------------------------------------------- !
   868)        ! Get LW radiative heating out of physics buffer !
   869)        ! ---------------------------------------------- !
   870)        call pbuf_get_field(pbuf, qrl_idx,   qrl)
   871) 
   872)        call pbuf_get_field(pbuf, wsedl_idx, wsedl)
   873) 
   874)        ! These fields are put into the pbuf for UNICON only.
   875)        if (trim(shallow_scheme) == 'UNICON') then
   876)           call pbuf_get_field(pbuf, bprod_idx,    bprod)
   877)           call pbuf_get_field(pbuf, ipbl_idx,     ipbl)
   878)           call pbuf_get_field(pbuf, kpblh_idx,    kpblh)
   879)           call pbuf_get_field(pbuf, wstarPBL_idx, wstarPBL)
   880)           call pbuf_get_field(pbuf, tkes_idx,     tkes)
   881)           call pbuf_get_field(pbuf, went_idx,     went)
   882)        else
   883)           allocate(bprod(pcols,pverp), ipbl(pcols), kpblh(pcols), wstarPBL(pcols), tkes(pcols), went(pcols))
   884)        end if
   885) 
   886)        call compute_eddy_diff( lchnk    ,                                                                    &
   887)                                pcols    , pver        , ncol       , state%t    , state%q(:,:,1) , ztodt   , &
   888)                                state%q(:,:,ixcldliq)  , state%q(:,:,ixcldice)   ,                            &
   889)                                state%s  , state%rpdel , cldn       , qrl        , wsedl          ,           &
   890)                                state%zm , state%zi    , state%pmid , state%pint , state%u        , state%v , &
   891)                                taux     , tauy        , shflx      , cflx(:,1)  , wstarent       , nturb   , &
   892)                                rrho     , ustar       , pblh       , kvm_in     , kvh_in         , kvm     , &
   893)                                kvh      , kvq         , cgh        ,                                         &
   894)                                cgs      , tpert       , qpert      , wpert      , tke            , bprod   , &
   895)                                sprod    , sfi         , kvinit     ,                                         &
   896)                                tauresx  , tauresy     , ksrftms    ,                                         &
   897)                                ipbl     , kpblh       , wstarPBL   , tkes       , went           , turbtype, &
   898)                                smaw )
   899) 
   900)        ! The diag_TKE scheme does not calculate the Monin-Obukhov length, which is used in dry deposition calculations.
   901)        ! Use the routines from pbl_utils to accomplish this. Assumes ustar and rrho have been set.
   902)        th(:ncol,pver) = state%t(:ncol,pver) * state%exner(:ncol,pver)
   903)        thvs(:ncol) = virtem(th(:ncol,pver),state%q(:ncol,pver,1))
   904)        call calc_obklen(th(:ncol,pver), thvs(:ncol), cflx(:ncol,1), shflx(:ncol), rrho(:ncol), ustar(:ncol), &
   905)                         khfs(:ncol),    kqfs(:ncol), kbfs(:ncol),   obklen(:ncol))
   906) 
   907)        ! ----------------------------------------------- !       
   908)        ! Store TKE in pbuf for use by shallow convection !
   909)        ! ----------------------------------------------- !   
   910) 
   911)        tpertPBL(:ncol) = tpert(:ncol)
   912)        qpertPBL(:ncol) = qpert(:ncol)
   913) 
   914) 
   915)        ! The diffusivities from diag_TKE can be much larger than from HB in the free
   916)        ! troposphere and upper atmosphere. These seem to be larger than observations,
   917)        ! and in WACCM the gw_drag code is already applying an eddy diffusivity in the
   918)        ! upper atmosphere. Optionally, adjust the diffusivities in the free troposphere
   919)        ! or the upper atmosphere.
   920)        !
   921)        ! NOTE: Further investigation should be done as to why the diffusivities are
   922)        ! larger in diag_TKE.
   923)        if ((kv_freetrop_scale /= 1._r8) .or. ((kv_top_scale /= 1._r8) .and. (kv_top_pressure > 0._r8))) then
   924)          do i = 1, ncol
   925)            do k = 1, pverp
   926)            
   927)              ! Outside of the boundary layer?
   928)              if (state%zi(i,k) > pblh(i)) then
   929) 
   930)                ! In the upper atmosphere?
   931)                if (state%pint(i,k) <= kv_top_pressure) then
   932)                  kvh(i,k) = kvh(i,k) * kv_top_scale
   933)                  kvm(i,k) = kvm(i,k) * kv_top_scale
   934)                  kvq(i,k) = kvq(i,k) * kv_top_scale
   935)                else
   936)                  kvh(i,k) = kvh(i,k) * kv_freetrop_scale
   937)                  kvm(i,k) = kvm(i,k) * kv_freetrop_scale
   938)                  kvq(i,k) = kvq(i,k) * kv_freetrop_scale
   939)                end if
   940)              else
   941)                exit
   942)              end if
   943)            end do
   944)          end do
   945)        end if
   946) 
   947)        ! Write out fields that are only used by this scheme
   948) 
   949)        call outfld( 'BPROD   ', bprod, pcols, lchnk )
   950)        call outfld( 'SPROD   ', sprod, pcols, lchnk )
   951)        call outfld( 'SFI     ', sfi,   pcols, lchnk )
   952) 
   953)        if (trim(shallow_scheme) /= 'UNICON') then
   954)           deallocate(bprod, ipbl, kpblh, wstarPBL, tkes, went)
   955)        end if
   956) 
   957)     case ( 'HB', 'HBR' )
   958) 
   959)        ! Modification : We may need to use 'taux' instead of 'tautotx' here, for
   960)        !                consistency with the previous HB scheme.
   961) 
   962)        th(:ncol,:pver) = state%t(:ncol,:pver) * state%exner(:ncol,:pver)
   963) 
   964)        call compute_hb_diff( lchnk     , ncol     ,                                &
   965)                              th        , state%t  , state%q , state%zm , state%zi, &
   966)                              state%pmid, state%u  , state%v , tautotx  , tautoty , &
   967)                              shflx     , cflx(:,1), obklen  , ustar    , pblh    , &
   968)                              kvm       , kvh      , kvq     , cgh      , cgs     , &
   969)                              tpert     , qpert    , cldn    , ocnfrac  , tke     , &
   970)                              ri        , &
   971)                              eddy_scheme )
   972)     
   973)        call outfld( 'HB_ri',          ri,         pcols,   lchnk )
   974) 
   975)        wpert = 0._r8  
   976) 
   977)     end select
   978) 
   979)     call outfld( 'ustar',   ustar(:), pcols, lchnk )
   980)     call outfld( 'obklen', obklen(:), pcols, lchnk )
   981) 
   982)     ! kvh (in pbuf) is used by other physics parameterizations, and as an initial guess in compute_eddy_diff
   983)     ! on the next timestep.  It is not updated by the compute_vdiff call below.
   984)     call pbuf_set_field(pbuf, kvh_idx, kvh)
   985) 
   986)     ! kvm (in pbuf) is only used as an initial guess in compute_eddy_diff on the next timestep.
   987)     ! The contributions for molecular diffusion made to kvm by the call to compute_vdiff below 
   988)     ! are not included in the pbuf as these are not needed in the initial guess by compute_eddy_diff.
   989)     call pbuf_set_field(pbuf, kvm_idx, kvm)
   990) 
   991)     call outfld( 'WGUSTD' , wpert, pcols, lchnk )
   992) 
   993)     !------------------------------------ ! 
   994)     !    Application of diffusivities     !
   995)     !------------------------------------ !
   996) 
   997)     ! Set arrays from input state.
   998)     q_tmp(:ncol,:,:) = state%q(:ncol,:,:)
   999)     s_tmp(:ncol,:) = state%s(:ncol,:)
  1000)     u_tmp(:ncol,:) = state%u(:ncol,:)
  1001)     v_tmp(:ncol,:) = state%v(:ncol,:)
  1002) 
  1003)     !------------------------------------------------------ !
  1004)     ! Write profile output before applying diffusion scheme !
  1005)     !------------------------------------------------------ !
  1006) 
  1007)     sl_prePBL(:ncol,:pver)  = s_tmp(:ncol,:) -   latvap * q_tmp(:ncol,:,ixcldliq) &
  1008)                                              - ( latvap + latice) * q_tmp(:ncol,:,ixcldice)
  1009)     qt_prePBL(:ncol,:pver)  = q_tmp(:ncol,:,1) + q_tmp(:ncol,:,ixcldliq) &
  1010)                                                + q_tmp(:ncol,:,ixcldice)
  1011)     slv_prePBL(:ncol,:pver) = sl_prePBL(:ncol,:pver) * ( 1._r8 + zvir*qt_prePBL(:ncol,:pver) ) 
  1012) 
  1013)     call qsat(state%t(:ncol,:), state%pmid(:ncol,:), &
  1014)          tem2(:ncol,:), ftem(:ncol,:))
  1015)     ftem_prePBL(:ncol,:) = state%q(:ncol,:,1)/ftem(:ncol,:)*100._r8
  1016) 
  1017)     call outfld( 'qt_pre_PBL   ', qt_prePBL,                 pcols, lchnk )
  1018)     call outfld( 'sl_pre_PBL   ', sl_prePBL,                 pcols, lchnk )
  1019)     call outfld( 'slv_pre_PBL  ', slv_prePBL,                pcols, lchnk )
  1020)     call outfld( 'u_pre_PBL    ', state%u,                   pcols, lchnk )
  1021)     call outfld( 'v_pre_PBL    ', state%v,                   pcols, lchnk )
  1022)     call outfld( 'qv_pre_PBL   ', state%q(:ncol,:,1),        pcols, lchnk )
  1023)     call outfld( 'ql_pre_PBL   ', state%q(:ncol,:,ixcldliq), pcols, lchnk )
  1024)     call outfld( 'qi_pre_PBL   ', state%q(:ncol,:,ixcldice), pcols, lchnk )
  1025)     call outfld( 't_pre_PBL    ', state%t,                   pcols, lchnk )
  1026)     call outfld( 'rh_pre_PBL   ', ftem_prePBL,               pcols, lchnk )
  1027) 
  1028)     ! --------------------------------------------------------------------------------- !
  1029)     ! Call the diffusivity solver and solve diffusion equation                          !
  1030)     ! The final two arguments are optional function references to                       !
  1031)     ! constituent-independent and constituent-dependent moleculuar diffusivity routines !
  1032)     ! --------------------------------------------------------------------------------- !
  1033) 
  1034)     !------------------------------------------------------------------------ 
  1035)     !  Check to see if constituent dependent gas constant needed (WACCM-X) 
  1036)     !------------------------------------------------------------------------
  1037)     if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
  1038)       rairi(:ncol,1) = rairv(:ncol,1,lchnk)
  1039)       do k = 2, pver
  1040)         do i = 1, ncol
  1041)           rairi(i,k) = 0.5_r8 * (rairv(i,k,lchnk)+rairv(i,k-1,lchnk))
  1042)         end do
  1043)       end do
  1044)     else
  1045)       rairi(:ncol,:pver+1) = rair 
  1046)     endif
  1047) 
  1048)     ! Modification : We may need to output 'tautotx_im,tautoty_im' from below 'compute_vdiff' and
  1049)     !                separately print out as diagnostic output, because these are different from
  1050)     !                the explicit 'tautotx, tautoty' computed above. 
  1051)     ! Note that the output 'tauresx,tauresy' from below subroutines are fully implicit ones.
  1052) 
  1053)     call pbuf_get_field(pbuf, kvt_idx, kvt)
  1054) 
  1055)     if( any(fieldlist_wet) ) then
  1056) 
  1057)         call compute_vdiff( state%lchnk   ,                                                                     &
  1058)                             pcols         , pver               , pcnst        , ncol          , state%pmid    , &
  1059)                             state%pint    , state%rpdel        , state%t      , ztodt         , taux          , &
  1060)                             tauy          , shflx              , cflx         , ntop          , nbot          , &
  1061)                             kvh           , kvm                , kvq          , cgs           , cgh           , &
  1062)                             state%zi      , ksrftms            , qmincg       , fieldlist_wet , fieldlist_molec,&
  1063)                             u_tmp         , v_tmp              , q_tmp        , s_tmp         ,                 &
  1064)                             tautmsx       , tautmsy            , dtk          , topflx        , errstring     , &
  1065)                             tauresx       , tauresy            , 1            , cpairv(:,:,state%lchnk), rairi, &
  1066)                             do_molec_diff , compute_molec_diff , vd_lu_qdecomp, kvt )
  1067) 
  1068)         call handle_errmsg(errstring, subname="compute_vdiff", &
  1069)              extra_msg="Error in fieldlist_wet call from vertical_diffusion.")
  1070) 
  1071)     end if
  1072)  
  1073)     if( any( fieldlist_dry ) ) then
  1074) 
  1075)         if( do_molec_diff ) then
  1076)             errstring = "Design flaw: dry vdiff not currently supported with molecular diffusion"
  1077)             call endrun(errstring)
  1078)         end if
  1079) 
  1080)         call compute_vdiff( state%lchnk   ,                                                                     &
  1081)                             pcols         , pver               , pcnst        , ncol          , state%pmiddry , &
  1082)                             state%pintdry , state%rpdeldry     , state%t      , ztodt         , taux          , &       
  1083)                             tauy          , shflx              , cflx         , ntop          , nbot          , &       
  1084)                             kvh           , kvm                , kvq          , cgs           , cgh           , &   
  1085)                             state%zi      , ksrftms            , qmincg       , fieldlist_dry , fieldlist_molec,&
  1086)                             u_tmp         , v_tmp              , q_tmp        , s_tmp         ,                 &
  1087)                             tautmsx       , tautmsy            , dtk          , topflx        , errstring     , &
  1088)                             tauresx       , tauresy            , 1            , cpairv(:,:,state%lchnk), rairi, &
  1089)                             do_molec_diff , compute_molec_diff , vd_lu_qdecomp )
  1090) 
  1091)         call handle_errmsg(errstring, subname="compute_vdiff", &
  1092)              extra_msg="Error in fieldlist_dry call from vertical_diffusion.")
  1093) 
  1094)     end if
  1095) 
  1096)     if (prog_modal_aero) then
  1097) 
  1098)        ! Modal aerosol species not diffused, so just add the explicit surface fluxes to the
  1099)        ! lowest layer
  1100) 
  1101)     tmp1(:ncol) = ztodt * gravit * state%rpdel(:ncol,pver)
  1102)        do m = 1, pmam_ncnst
  1103)           l = pmam_cnst_idx(m)
  1104)           q_tmp(:ncol,pver,l) = q_tmp(:ncol,pver,l) + tmp1(:ncol) * cflx(:ncol,l)
  1105)        enddo
  1106)     end if
  1107) 
  1108)     ! -------------------------------------------------------- !
  1109)     ! Diagnostics and output writing after applying PBL scheme !
  1110)     ! -------------------------------------------------------- !
  1111) 
  1112)     sl(:ncol,:pver)  = s_tmp(:ncol,:) -   latvap           * q_tmp(:ncol,:,ixcldliq) &
  1113)                                       - ( latvap + latice) * q_tmp(:ncol,:,ixcldice)
  1114)     qt(:ncol,:pver)  = q_tmp(:ncol,:,1) + q_tmp(:ncol,:,ixcldliq) &
  1115)                                     + q_tmp(:ncol,:,ixcldice)
  1116)     slv(:ncol,:pver) = sl(:ncol,:pver) * ( 1._r8 + zvir*qt(:ncol,:pver) ) 
  1117) 
  1118)     slflx(:ncol,1) = 0._r8
  1119)     qtflx(:ncol,1) = 0._r8
  1120)     uflx(:ncol,1)  = 0._r8
  1121)     vflx(:ncol,1)  = 0._r8
  1122) 
  1123)     slflx_cg(:ncol,1) = 0._r8
  1124)     qtflx_cg(:ncol,1) = 0._r8
  1125)     uflx_cg(:ncol,1)  = 0._r8
  1126)     vflx_cg(:ncol,1)  = 0._r8
  1127) 
  1128)     do k = 2, pver
  1129)        do i = 1, ncol
  1130)           rhoair     = state%pint(i,k) / ( rair * ( ( 0.5_r8*(slv(i,k)+slv(i,k-1)) - gravit*state%zi(i,k))/cpair ) )
  1131)           slflx(i,k) = kvh(i,k) * &
  1132)                           ( - rhoair*(sl(i,k-1)-sl(i,k))/(state%zm(i,k-1)-state%zm(i,k)) &
  1133)                             + cgh(i,k) ) 
  1134)           qtflx(i,k) = kvh(i,k) * &
  1135)                           ( - rhoair*(qt(i,k-1)-qt(i,k))/(state%zm(i,k-1)-state%zm(i,k)) &
  1136)                             + rhoair*(cflx(i,1)+cflx(i,ixcldliq)+cflx(i,ixcldice))*cgs(i,k) )
  1137)           uflx(i,k)  = kvm(i,k) * &
  1138)                           ( - rhoair*(u_tmp(i,k-1)-u_tmp(i,k))/(state%zm(i,k-1)-state%zm(i,k)))
  1139)           vflx(i,k)  = kvm(i,k) * &

  1358)   subroutine positive_moisture( cp, xlv, xls, ncol, mkx, dt, qvmin, qlmin, qimin, &

  1361)   ! If any 'ql < qlmin, qi < qimin, qv < qvmin' are developed in any layer,         !
  1362)   ! force them to be larger than minimum value by (1) condensating water vapor      !

  1364)   ! layer. '2._r8' is multiplied to the minimum values for safety.                  !

  1368)   ! input tendencies.                                                               !

  1371)     implicit none
  1372)     integer,  intent(in)     :: ncol, mkx
  1373)     real(r8), intent(in)     :: cp, xlv, xls
  1374)     real(r8), intent(in)     :: dt, qvmin, qlmin, qimin
  1375)     real(r8), intent(in)     :: dp(ncol,mkx)

  1382)   !                shallow convection and cloud macrophysics.

  1393)           qi(i,k)    = qi(i,k) +  dqi
  1394)           qv(i,k)    = qv(i,k) -  dql - dqi

  1405)           ql(i,k) = max(ql(i,k),qlmin)
  1406)           qi(i,k) = max(qi(i,k),qimin)
