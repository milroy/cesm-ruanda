    49) ! 1. Calculates the leaf temperature:
    50) ! 2. Calculates the leaf fluxes, transpiration, photosynthesis and
    51) !    updates the dew accumulation due to evaporation.
    52) !
    53) ! Method:
    54) ! Use the Newton-Raphson iteration to solve for the foliage
    55) ! temperature that balances the surface energy budget:
    56) !
    57) ! f(t_veg) = Net radiation - Sensible - Latent = 0
    58) ! f(t_veg) + d(f)/d(t_veg) * dt_veg = 0     (*)
    59) !
    60) ! Note:
    61) ! (1) In solving for t_veg, t_grnd is given from the previous timestep.
    62) ! (2) The partial derivatives of aerodynamical resistances, which cannot
    63) !     be determined analytically, are ignored for d(H)/dT and d(LE)/dT
    64) ! (3) The weighted stomatal resistance of sunlit and shaded foliage is used
    65) ! (4) Canopy air temperature and humidity are derived from => Hc + Hg = Ha
    66) !                                                          => Ec + Eg = Ea
    67) ! (5) Energy loss is due to: numerical truncation of energy budget equation
    68) !     (*); and "ecidif" (see the code) which is dropped into the sensible
    69) !     heat
    70) ! (6) The convergence criteria: the difference, del = t_veg(n+1)-t_veg(n)
    71) !     and del2 = t_veg(n)-t_veg(n-1) less than 0.01 K, and the difference
    72) !     of water flux from the leaf between the iteration step (n+1) and (n)
    73) !     less than 0.1 W/m2; or the iterative steps over 40.

   781) 
   782)          tlbef(p) = t_veg(p)
   783)          del2(p) = del(p)
   784) 
   785)          ! Determine aerodynamic resistances
   786) 
   787)          ram1(p)  = 1._r8/(ustar(p)*ustar(p)/um(p))
   788)          rah(p,1) = 1._r8/(temp1(p)*ustar(p))
   789)          raw(p,1) = 1._r8/(temp2(p)*ustar(p))
   790) 
   791)          ! Bulk boundary layer resistance of leaves
   792) 
   793)          uaf(p) = um(p)*sqrt( 1._r8/(ram1(p)*um(p)) )
   794)          cf  = 0.01_r8/(sqrt(uaf(p))*sqrt(dleaf(ivt(p))))
   795)          rb(p)  = 1._r8/(cf*uaf(p))
   796)          rb1(p) = rb(p)
   797)   
   798)          ! Parameterization for variation of csoilc with canopy density from
   799)          ! X. Zeng, University of Arizona
   800) 
   801)          w = exp(-(elai(p)+esai(p)))
   802) 
   803)          ! changed by K.Sakaguchi from here
   804)          ! transfer coefficient over bare soil is changed to a local variable
   805)          ! just for readability of the code (from line 680)
   806)          csoilb = (vkc/(0.13_r8*(z0mg(c)*uaf(p)/1.5e-5_r8)**0.45_r8))
   807) 
   808)          !compute the stability parameter for ricsoilc  ("S" in Sakaguchi&Zeng,2008)
   809) 
   810)          ri = ( grav*htop(p) * (taf(p) - t_grnd(c)) ) / (taf(p) * uaf(p) **2.00_r8)
   811) 
   812)          !! modify csoilc value (0.004) if the under-canopy is in stable condition
   813) 
   814)          if ( (taf(p) - t_grnd(c) ) > 0._r8) then
   815)                ! decrease the value of csoilc by dividing it with (1+gamma*min(S, 10.0))
   816)                ! ria ("gmanna" in Sakaguchi&Zeng, 2008) is a constant (=0.5)
   817)                ricsoilc = csoilc / (1.00_r8 + ria*min( ri, 10.0_r8) )
   818)                csoilcn = csoilb*w + ricsoilc*(1._r8-w)
   819)          else
   820)               csoilcn = csoilb*w + csoilc*(1._r8-w)
   821)          end if
   822) 
   823)          !! Sakaguchi changes for stability formulation ends here
   824) 
   825)          rah(p,2) = 1._r8/(csoilcn*uaf(p))
   826)          raw(p,2) = rah(p,2)
   827) 
   828)          ! Stomatal resistances for sunlit and shaded fractions of canopy.
   829)          ! Done each iteration to account for differences in eah, tv.
   830) 
   831)          svpts(p) = el(p)                         ! pa
   832)          eah(p) = forc_pbot(g) * qaf(p) / 0.622_r8   ! pa
   833)       end do
   834) 
   835)       ! 4/25/05, PET: Now calling the sun/shade version of Stomata by default
   836)       call Stomata (fn, filterp, lbp, ubp, svpts, eah, o2, co2, rb, dayl_factor, phase='sun')
   837)       call Stomata (fn, filterp, lbp, ubp, svpts, eah, o2, co2, rb, dayl_factor, phase='sha')
   838) 
   839)       do f = 1, fn
   840)          p = filterp(f)
   841)          c = pcolumn(p)
   842)          g = pgridcell(p)
   843) 
   844)          ! Sensible heat conductance for air, leaf and ground
   845)          ! Moved the original subroutine in-line...
   846) 
   847)          wta    = 1._r8/rah(p,1)             ! air
   848)          wtl    = (elai(p)+esai(p))/rb(p)    ! leaf
   849)          wtg(p) = 1._r8/rah(p,2)             ! ground
   850)          wtshi  = 1._r8/(wta+wtl+wtg(p))
   851) 
   852)          wtl0(p) = wtl*wtshi         ! leaf
   853)          wtg0    = wtg(p)*wtshi      ! ground
   854)          wta0(p) = wta*wtshi         ! air
   855) 
   856)          wtga    = wta0(p)+wtg0      ! ground + air
   857)          wtal(p) = wta0(p)+wtl0(p)   ! air + leaf
   858) 
   859)          ! Fraction of potential evaporation from leaf
   860) 
   861)          if (fdry(p) .gt. 0._r8) then
   862)             rppdry  = fdry(p)*rb(p)*(laisun(p)/(rb(p)+rssun(p)) + &
   863)                                      laisha(p)/(rb(p)+rssha(p)))/elai(p)
   864)          else
   865)             rppdry = 0._r8
   866)          end if
   867)          efpot = forc_rho(g)*wtl*(qsatl(p)-qaf(p))
   868) 
   869)          if (efpot > 0._r8) then
   870)             if (btran(p) > btran0) then
   871)                qflx_tran_veg(p) = efpot*rppdry
   872)                rpp = rppdry + fwet(p)
   873)             else
   874)                !No transpiration if btran below 1.e-10
   875)                rpp = fwet(p)
   876)                qflx_tran_veg(p) = 0._r8
   877)             end if
   878)             !Check total evapotranspiration from leaves
   879)             rpp = min(rpp, (qflx_tran_veg(p)+h2ocan(p)/dtime)/efpot)
   880)          else
   881)             !No transpiration if potential evaporation less than zero
   882)             rpp = 1._r8
   883)             qflx_tran_veg(p) = 0._r8
   884)          end if
   885) 
   886)          ! Update conductances for changes in rpp
   887)          ! Latent heat conductances for ground and leaf.
   888)          ! Air has same conductance for both sensible and latent heat.
   889)          ! Moved the original subroutine in-line...
   890) 
   891)          wtaq    = frac_veg_nosno(p)/raw(p,1)                        ! air
   892)          wtlq    = frac_veg_nosno(p)*(elai(p)+esai(p))/rb(p) * rpp   ! leaf
   893) 
   894)          !Litter layer resistance. Added by K.Sakaguchi
   895)          snowdp_c = z_dl ! critical depth for 100% litter burial by snow (=litter thickness)
   896)          fsno_dl = snowdp(c)/snowdp_c    ! effective snow cover for (dry)plant litter
   897)          elai_dl = lai_dl*(1._r8 - min(fsno_dl,1._r8)) ! exposed (dry)litter area index
   898)          rdl = ( 1._r8 - exp(-elai_dl) ) / ( 0.004_r8*uaf(p)) ! dry litter layer resistance
   899) 
   900)          ! add litter resistance and Lee and Pielke 1992 beta
   901)          if (delq(p) .lt. 0._r8) then  !dew. Do not apply beta for negative flux (follow old rsoil)
   902)             wtgq(p) = frac_veg_nosno(p)/(raw(p,2)+rdl)
   903)          else
   904)             wtgq(p) = soilbeta(c)*frac_veg_nosno(p)/(raw(p,2)+rdl)
   905)          end if
   906) 
   907)          wtsqi   = 1._r8/(wtaq+wtlq+wtgq(p))
   908) 
   909)          wtgq0    = wtgq(p)*wtsqi      ! ground
   910)          wtlq0(p) = wtlq*wtsqi         ! leaf
   911)          wtaq0(p) = wtaq*wtsqi         ! air
   912) 
   913)          wtgaq    = wtaq0(p)+wtgq0     ! air + ground
   914)          wtalq(p) = wtaq0(p)+wtlq0(p)  ! air + leaf
   915) 
   916)          dc1 = forc_rho(g)*cpair*wtl
   917)          dc2 = hvap*forc_rho(g)*wtlq
   918) 
   919)          efsh   = dc1*(wtga*t_veg(p)-wtg0*t_grnd(c)-wta0(p)*thm(p))
   920)          efe(p) = dc2*(wtgaq*qsatl(p)-wtgq0*qg(c)-wtaq0(p)*forc_q(g))
   921) 
   922)          ! Evaporation flux from foliage
   923) 
   924)          erre = 0._r8
   925)          if (efe(p)*efeb(p) < 0._r8) then
   926)             efeold = efe(p)
   927)             efe(p)  = 0.1_r8*efeold
   928)             erre = efe(p) - efeold
   929)          end if
   930)          dt_veg(p) = (sabv(p) + air(p) + bir(p)*t_veg(p)**4 + &
   931)               cir(p)*t_grnd(c)**4 - efsh - efe(p)) / &
   932)               (- 4._r8*bir(p)*t_veg(p)**3 +dc1*wtga +dc2*wtgaq*qsatldT(p))
   933)          t_veg(p) = tlbef(p) + dt_veg(p)
   934)          dels = dt_veg(p)
   935)          del(p)  = abs(dels)
   936)          err(p) = 0._r8
   937)          if (del(p) > delmax) then
   938)             dt_veg(p) = delmax*dels/del(p)
   939)             t_veg(p) = tlbef(p) + dt_veg(p)
   940)             err(p) = sabv(p) + air(p) + bir(p)*tlbef(p)**3*(tlbef(p) + &
   941)                  4._r8*dt_veg(p)) + cir(p)*t_grnd(c)**4 - &
   942)                  (efsh + dc1*wtga*dt_veg(p)) - (efe(p) + &
   943)                  dc2*wtgaq*qsatldT(p)*dt_veg(p))
   944)          end if
   945) 
   946)          ! Fluxes from leaves to canopy space
   947)          ! "efe" was limited as its sign changes frequently.  This limit may
   948)          ! result in an imbalance in "hvap*qflx_evap_veg" and
   949)          ! "efe + dc2*wtgaq*qsatdt_veg"
   950) 
   951)          efpot = forc_rho(g)*wtl*(wtgaq*(qsatl(p)+qsatldT(p)*dt_veg(p)) &
   952)             -wtgq0*qg(c)-wtaq0(p)*forc_q(g))
   953)          qflx_evap_veg(p) = rpp*efpot
   954)          
   955)          ! Calculation of evaporative potentials (efpot) and
   956)          ! interception losses; flux in kg m**-2 s-1.  ecidif
   957)          ! holds the excess energy if all intercepted water is evaporated
   958)          ! during the timestep.  This energy is later added to the
   959)          ! sensible heat flux.
   960) 
   961)          ecidif = 0._r8
   962)          if (efpot > 0._r8 .and. btran(p) > btran0) then
   963)             qflx_tran_veg(p) = efpot*rppdry
   964)          else
   965)             qflx_tran_veg(p) = 0._r8
   966)          end if
   967)          ecidif = max(0._r8, qflx_evap_veg(p)-qflx_tran_veg(p)-h2ocan(p)/dtime)
   968)          qflx_evap_veg(p) = min(qflx_evap_veg(p),qflx_tran_veg(p)+h2ocan(p)/dtime)
   969) 
   970)          ! The energy loss due to above two limits is added to
   971)          ! the sensible heat flux.
   972) 
   973)          eflx_sh_veg(p) = efsh + dc1*wtga*dt_veg(p) + err(p) + erre + hvap*ecidif
   974) 
   975)          ! Re-calculate saturated vapor pressure, specific humidity, and their
   976)          ! derivatives at the leaf surface
   977) 
   978)          call QSat(t_veg(p), forc_pbot(g), el(p), deldT, qsatl(p), qsatldT(p))
   979) 
   980)          ! Update vegetation/ground surface temperature, canopy air
   981)          ! temperature, canopy vapor pressure, aerodynamic temperature, and
   982)          ! Monin-Obukhov stability parameter for next iteration.
   983) 
   984)          taf(p) = wtg0*t_grnd(c) + wta0(p)*thm(p) + wtl0(p)*t_veg(p)
   985)          qaf(p) = wtlq0(p)*qsatl(p) + wtgq0*qg(c) + forc_q(g)*wtaq0(p)
   986) 
   987)          ! Update Monin-Obukhov length and wind speed including the
   988)          ! stability effect
   989) 
   990)          dth(p) = thm(p)-taf(p)
   991)          dqh(p) = forc_q(g)-qaf(p)
   992)          delq(p) = wtalq(p)*qg(c)-wtlq0(p)*qsatl(p)-wtaq0(p)*forc_q(g)
   993) 
   994)          tstar = temp1(p)*dth(p)
   995)          qstar = temp2(p)*dqh(p)
   996) 
   997)          thvstar = tstar*(1._r8+0.61_r8*forc_q(g)) + 0.61_r8*forc_th(g)*qstar
   998)          zeta = zldis(p)*vkc*grav*thvstar/(ustar(p)**2*thv(c))
   999) 
  1000)          if (zeta >= 0._r8) then     !stable
  1001)             zeta = min(2._r8,max(zeta,0.01_r8))
  1002)             um(p) = max(ur(p),0.1_r8)
  1003)          else                     !unstable
  1004)             zeta = max(-100._r8,min(zeta,-0.01_r8))
  1005)             wc = beta*(-grav*ustar(p)*thvstar*zii/thv(c))**0.333_r8
  1006)             um(p) = sqrt(ur(p)*ur(p)+wc*wc)
  1007)          end if
  1008)          obu(p) = zldis(p)/zeta
  1009) 
  1010)          if (obuold(p)*obu(p) < 0._r8) nmozsgn(p) = nmozsgn(p)+1
  1011)          if (nmozsgn(p) >= 4) obu(p) = zldis(p)/(-0.01_r8)
  1012)          obuold(p) = obu(p)
  1013) 
  1014)       end do   ! end of filtered pft loop
  1015) 
  1016)       ! Test for convergence
  1017) 
  1018)       itlef = itlef+1
  1019)       if (itlef > itmin) then
  1020)          do f = 1, fn
  1021)             p = filterp(f)
  1022)             dele(p) = abs(efe(p)-efeb(p))
  1023)             efeb(p) = efe(p)
  1024)             det(p)  = max(del(p),del2(p))
  1025)          end do
  1026)          fnold = fn
  1027)          fn = 0
  1028)          do f = 1, fnold
  1029)             p = filterp(f)
  1030)             if (.not. (det(p) < dtmin .and. dele(p) < dlemin)) then
  1031)                fn = fn + 1
  1032)                filterp(fn) = p
  1033)             end if
  1034)          end do
  1035)       end if
  1036) 
  1037)    end do ITERATION     ! End stability iteration
  1038) 
  1039)    fn = fnorig
  1040)    filterp(1:fn) = fporig(1:fn)
  1041) 
  1042)    do f = 1, fn
  1043)       p = filterp(f)
  1044)       c = pcolumn(p)
  1045)       g = pgridcell(p)
  1046) 
  1047)       ! Energy balance check in canopy
  1048) 
  1049)       err(p) = sabv(p) + air(p) + bir(p)*tlbef(p)**3*(tlbef(p) + 4._r8*dt_veg(p)) &
  1050)          + cir(p)*t_grnd(c)**4 - eflx_sh_veg(p) - hvap*qflx_evap_veg(p)
  1051) 
  1052)       ! Fluxes from ground to canopy space
  1053) 
  1054)       delt    = wtal(p)*t_grnd(c)-wtl0(p)*t_veg(p)-wta0(p)*thm(p)
  1055)       taux(p) = -forc_rho(g)*forc_u(g)/ram1(p)
  1056)       tauy(p) = -forc_rho(g)*forc_v(g)/ram1(p)
  1057)       eflx_sh_grnd(p) = cpair*forc_rho(g)*wtg(p)*delt
  1058)       qflx_evap_soi(p) = forc_rho(g)*wtgq(p)*delq(p)
  1059) 
  1060)       ! 2 m height air temperature
  1061) 
  1062)       t_ref2m(p) = thm(p) + temp1(p)*dth(p)*(1._r8/temp12m(p) - 1._r8/temp1(p))
  1063)       t_ref2m_r(p) = t_ref2m(p)
  1064) 
  1065)       ! 2 m height specific humidity
  1066) 
  1067)       q_ref2m(p) = forc_q(g) + temp2(p)*dqh(p)*(1._r8/temp22m(p) - 1._r8/temp2(p))
  1068) 
  1069)       ! 2 m height relative humidity
  1070) 
  1071)       call QSat(t_ref2m(p), forc_pbot(g), e_ref2m, de2mdT, qsat_ref2m, dqsat2mdT)
  1072)       rh_ref2m(p) = min(100._r8, q_ref2m(p) / qsat_ref2m * 100._r8)
  1073)       rh_ref2m_r(p) = rh_ref2m(p)
  1074) 
  1075)       ! Downward longwave radiation below the canopy
  1076) 
  1077)       dlrad(p) = (1._r8-emv(p))*emg(c)*forc_lwrad(g) + &
  1078)          emv(p)*emg(c)*sb*tlbef(p)**3*(tlbef(p) + 4._r8*dt_veg(p))
  1079) 
  1080)       ! Upward longwave radiation above the canopy
  1081) 
  1082)       ulrad(p) = ((1._r8-emg(c))*(1._r8-emv(p))*(1._r8-emv(p))*forc_lwrad(g) &
  1083)          + emv(p)*(1._r8+(1._r8-emg(c))*(1._r8-emv(p)))*sb*tlbef(p)**3*(tlbef(p) + &
  1084)          4._r8*dt_veg(p)) + emg(c)*(1._r8-emv(p))*sb*t_grnd(c)**4)
  1085) 
  1086)       ! Derivative of soil energy flux with respect to soil temperature
  1087) 
  1088)       cgrnds(p) = cgrnds(p) + cpair*forc_rho(g)*wtg(p)*wtal(p)
  1089)       cgrndl(p) = cgrndl(p) + forc_rho(g)*wtgq(p)*wtalq(p)*dqgdT(c)
  1090)       cgrnd(p)  = cgrnds(p) + cgrndl(p)*htvp(c)
  1091) 
  1092)       ! Update dew accumulation (kg/m2)
  1093)       
  1094)       h2ocan(p) = max(0._r8,h2ocan(p)+(qflx_tran_veg(p)-qflx_evap_veg(p))*dtime)
  1095)       
  1096)       ! total photosynthesis
  1097) 
  1098)       fpsn(p) = psnsun(p)*laisun(p) + psnsha(p)*laisha(p)
  1099)       
  1100)       if (use_cn) then
  1101)          if (use_c13) then
  1102)             ! 4/14/05: PET
  1103)             ! Adding isotope code
  1104)             rc13_canair(p) = c13o2(p)/(co2(p)-c13o2(p))
  1105)             rc13_psnsun(p) = rc13_canair(p)/alphapsnsun(p)
  1106)             rc13_psnsha(p) = rc13_canair(p)/alphapsnsha(p)
  1107)             c13_psnsun(p) = psnsun(p) * (rc13_psnsun(p)/(1._r8+rc13_psnsun(p)))
  1108)             c13_psnsha(p) = psnsha(p) * (rc13_psnsha(p)/(1._r8+rc13_psnsha(p)))
  1109)             !write(iulog,*) p,ivt(p),btran(p),psnsun(p),psnsha(p),alphapsnsun(p),alphapsnsha(p)
  1110)          end if
  1111)       end if
  1112)       
  1113)    end do
  1114) 
  1115)    ! Filter out pfts which have small energy balance errors; report others
  1116) 
  1117)    fnold = fn
  1118)    fn = 0
  1119)    do f = 1, fnold
  1120)       p = filterp(f)
  1121)       if (abs(err(p)) > 0.1_r8) then
  1122)          fn = fn + 1
  1123)          filterp(fn) = p
  1124)       end if
  1125)    end do
  1126) 
  1127)    do f = 1, fn
  1128)       p = filterp(f)
  1129)       write(iulog,*) 'energy balance in canopy ',p,', err=',err(p)
  1130)    end do
  1131) 
  1132)    end subroutine CanopyFluxes

  1140)    subroutine Stomata (fn, filterp, lbp, ubp, ei, ea, o2, co2, rb, dayl_factor, phase)
  1141) !
  1142) ! !DESCRIPTION: 
  1143) ! Leaf stomatal resistance and leaf photosynthesis. Modifications for CN code.
  1144) 
  1145) ! !REVISION HISTORY:
  1146) ! 22 January 2004: Created by Peter Thornton
  1147) ! 4/14/05: Peter Thornton: Converted Ci from local variable to pps struct member
  1148) !    now returns cisun or cisha per pft as implicit output argument.
  1149) !    Also sets alphapsnsun and alphapsnsha.
  1150) ! 4/25/05, Peter Thornton: Adopted as the default code for CLM, together with
  1151) !   modifications for sun/shade canopy.  Renamed from StomataCN to Stomata,
  1152) !   and eliminating the older Stomata subroutine
  1153) ! 3/6/09: Peter Thornton; added dayl_factor control on Vcmax, from Bill Bauerle
  1154) 
  1155) ! !USES:
  1156)      use shr_kind_mod , only : r8 => shr_kind_r8
  1157)      use shr_const_mod, only : SHR_CONST_TKFRZ, SHR_CONST_RGAS
  1158)      use clmtype
  1159)      use clm_atmlnd   , only : clm_a2l
  1160)      use spmdMod      , only: masterproc
  1161)      use pftvarcon    , only : nbrdlf_dcd_tmp_shrub
  1162)      use pftvarcon    , only : nsoybean, npcropmin
  1163) !
  1164) ! !ARGUMENTS:
  1165)      implicit none
  1166)      integer , intent(in)    :: fn                 ! size of pft filter
  1167)      integer , intent(in)    :: filterp(fn)        ! pft filter

  1169)      real(r8), intent(in)    :: ei(lbp:ubp)        ! vapor pressure inside leaf (sat vapor press at tl) (pa)
  1170)      real(r8), intent(in)    :: ea(lbp:ubp)        ! vapor pressure of canopy air (pa)
  1171)      real(r8), intent(in)    :: o2(lbp:ubp)        ! atmospheric o2 concentration (pa)
  1172)      real(r8), intent(in)    :: co2(lbp:ubp)       ! atmospheric co2 concentration (pa)
  1173)      real(r8), intent(inout) :: rb(lbp:ubp)        ! boundary layer resistance (s/m)
  1174)      real(r8), intent(in)    :: dayl_factor(lbp:ubp) ! scalar (0-1) for daylength
  1175)      character(len=*), intent(in) :: phase         ! 'sun' or 'sha'
  1176) !
  1177) ! !CALLED FROM:
  1178) ! subroutine CanopyFluxes in this module
  1179) !
  1180) ! !LOCAL VARIABLES:
  1181) !
  1182) ! local pointers to implicit in variables
  1183) ! new ecophys variables (leafcn, flnr) added 1/26/04
  1184) !
  1185)      integer , pointer :: pcolumn(:)     ! pft's column index
  1186)      integer , pointer :: pgridcell(:)   ! pft's gridcell index
  1187)      integer , pointer :: ivt(:)         ! pft vegetation type
