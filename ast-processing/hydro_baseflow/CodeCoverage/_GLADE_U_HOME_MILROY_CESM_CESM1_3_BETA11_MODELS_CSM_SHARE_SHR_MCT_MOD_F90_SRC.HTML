   200) ! !IROUTINE: mct_aVect_info - print out aVect info for debugging
   201) !
   202) ! !DESCRIPTION:
   203) !     Print out information about the input MCT {\it AttributeVector}
   204) !     {\tt aVect} to stdout. {\tt flag} sets the level of information:
   205) !     \begin{enumerate}
   206) !     \item  print out names of attributes in {\tt aVect}.
   207) !     \item  also print out local max and min of data in {\tt aVect}.
   208) !     \item  also print out global max and min of data in {\tt aVect}.
   209) !     \item  Same as 3 but include name of this routine.
   210) !     \end{enumerate}
   211) !     If {\tt flag} is 3 or higher, then optional argument {\tt comm}
   212) !     must be provided.
   213) !     If optional argument {\tt fld} is present, only information for
   214) !     that field will be printed.
   215) !     If optional argument {\tt istr} is present, it will be output
   216) !     before any of the information.
   217) !
   218) !
   219) ! !REVISION HISTORY:
   220) !     2003 Jul 01 - B. Kauffman, T. Craig - first version
   221) !
   222) ! !INTERFACE: ------------------------------------------------------------------
   223) 
   224) subroutine mct_aVect_info(flag,aVect,comm,pe,fld,istr)
   225) 
   226) ! !USES:
   227)   
   228) ! !INPUT/OUTPUT PARAMETERS:
   229) 
   230)    integer(IN)    ,intent(in)           :: flag  ! info level flag
   231)    type(mct_aVect),intent(in)           :: aVect ! Attribute vector
   232)    integer(IN)    ,intent(in),optional  :: comm  ! MPI communicator
   233)    integer(IN)    ,intent(in),optional  :: pe    ! processor number
   234)    character(*)   ,intent(in),optional  :: fld   ! fld
   235)    character(*)   ,intent(in),optional  :: istr  ! string for print
   236) 
   237) !EOP
   238) 
   239)    !--- local ---
   240)    integer(IN)          :: i,j,k,n      ! generic indicies
   241)    integer(IN)          :: ks,ke        ! start and stop k indices
   242)    integer(IN)          :: nflds        ! number of flds in AV to diagnose
   243)    integer(IN)          :: nsize        ! grid point size of AV
   244)    type(mct_string)     :: item         ! mct string
   245)    character(CL)        :: itemc        ! item converted to char
   246)    integer(IN)          :: comm_loc     ! local variable for comm
   247)    integer(IN)          :: pe_loc       ! local variable for pe
   248)    logical              :: commOK       ! is comm available
   249)    logical              :: peOK         ! is pe available
   250)    real(R8),allocatable :: minl(:)      ! local  min
   251)    real(R8),allocatable :: ming(:)      ! global min
   252)    real(R8),allocatable :: maxl(:)      ! local  max
   253)    real(R8),allocatable :: maxg(:)      ! global max
   254) 
   255)    !--- formats ---
   256)    character(*),parameter :: subName = '(mct_aVect_info) '
   257)    character(*),parameter :: F00 = "('(mct_aVect_info) ',8a)"
   258)    character(*),parameter :: F01 = "('(mct_aVect_info) ',a,i9)"
   259)    character(*),parameter :: F02 = "('(mct_aVect_info) ',240a)"
   260)    character(*),parameter :: F03 = "('(mct_aVect_info) ',a,2es11.3,i4,2x,a)"
   261) 
   262) !-------------------------------------------------------------------------------
   263) ! NOTE: has hard-coded knowledge/assumptions about mct aVect data type internals
   264) !-------------------------------------------------------------------------------
   265) 
   266)    commOK = .false.
   267)    peOK   = .false.
   268) 
   269)    if (present(pe)) then
   270)      peOK = .true.
   271)      pe_loc = pe
   272)    endif
   273)    if (present(comm)) then
   274)      commOK = .true.
   275)      comm_loc = comm
   276)      if (.not.PEOK) then
   277)        call shr_mpi_commrank(comm,pe_loc,subName)
   278)        peOK = .true.
   279)      endif
   280)    endif
   281) 
   282)    nsize = mct_aVect_lsize(aVect)
   283) 
   284)    if (present(fld)) then
   285)      nflds = 1
   286)      ks = mct_aVect_indexRA(aVect,fld,perrWith=subName)
   287)      ke = ks
   288)    else
   289)      nflds = mct_aVect_nRAttr(aVect)
   290)      ks = 1
   291)      ke = nflds
   292)    endif
   293) 
   294)    if (flag >= 1) then
   295)      if (present(istr)) then
   296)         if (s_loglev > 0) write(s_logunit,*) trim(istr)
   297)      endif
   298)      if (s_loglev > 0) write(s_logunit,F01) "local size =",nsize
   299)      if (associated(aVect%iList%bf)) then
   300)         if (s_loglev > 0) write(s_logunit,F02) "iList = ",aVect%iList%bf
   301)      endif
   302)      if (associated(aVect%rList%bf)) then
   303)         if (s_loglev > 0) write(s_logunit,F02) "rList = ",aVect%rList%bf
   304)      endif
   305)    endif
   306) 
   307)    if (flag >= 2) then
   308) 
   309)      allocate(minl(nflds))
   310)      allocate(maxl(nflds))
   311) 
   312)      do k=ks,ke
   313)        minl(k) = minval(aVect%rAttr(k,:))
   314)        maxl(k) = maxval(aVect%rAttr(k,:))
   315)      enddo
   316) 
   317)      if (flag >= 4 .and. commOK) then
   318)        allocate(ming(nflds))
   319)        allocate(maxg(nflds))
   320)        ming = 0._R8
   321)        maxg = 0._R8
   322)        call shr_mpi_min(minl,ming,comm,subName)
   323)        call shr_mpi_max(maxl,maxg,comm,subName)
   324)      endif
   325) 
   326)      do k=ks,ke
   327)        call mct_aVect_getRList(item,k,aVect)
   328)        itemc = mct_string_toChar(item)
   329)        call mct_string_clean(item)
   330)        if (s_loglev > 0) write(s_logunit,F03) 'l min/max ',minl(k),maxl(k),k,trim(itemc)
   331)        if (flag >= 3 .and. commOK) then
   332)          if ((peOK .and. pe_loc == 0) .or. .not.peOK) then
   333)            if (s_loglev > 0) write(s_logunit,F03) 'g min/max ',ming(k),maxg(k),k,trim(itemc)
   334)          endif
   335)        endif
   336)        if (flag >= 4 .and. commOK) then
   337)          if ((peOK .and. pe_loc == 0) .or. .not.peOK) then
   338)            if (s_loglev > 0) write(s_logunit,*) trim(subName),'g min/max ',ming(k),maxg(k),k,trim(itemc)
   339)          endif
   340)        endif
   341)      enddo
   342) 
   343)       deallocate(minl)
   344)       deallocate(maxl)
   345)       if (flag >= 4 .and. commOK) then
   346)          deallocate(ming)
   347)          deallocate(maxg)
   348)       endif
   349) 
   350)    endif
   351) 
   352)    call shr_sys_flush(s_logunit)
   353) 
   354) end subroutine mct_aVect_info

   402) ! !DESCRIPTION:

   405) ! !REMARKS:

   407) ! !REVISION HISTORY:
   408) !     2013 Jul 17 - T. Craig - first version
   409) !
   410) ! !INTERFACE: ------------------------------------------------------------------
   411) 
   412) subroutine mct_aVect_sharedFields(aVect1, aVect2, rlistout, ilistout)
   413) 
   414) ! !INPUT/OUTPUT PARAMETERS:
   415) 
   416)    type(mct_aVect),intent(in)  :: aVect1   ! an Attribute vector
   417)    type(mct_aVect),intent(in)  :: aVect2   ! an Attribute vector
   418)    character(*)   ,intent(inout),optional  :: rlistout      ! field name string
   419)    character(*)   ,intent(inout),optional  :: ilistout      ! field name string
   420) 
   421) !EOP
   422) 
   423)    !--- local ---
   424)    integer(IN) :: nflds1,nflds2
   425)    character(len=CXX) :: list1,list2
   426) 
   427)    !--- formats ---
   428)    character(*),parameter :: subName = "(mct_aVect_sharedFields) "
   429)    character(*),parameter :: F00 = "('(mct_aVect_sharedFields) ',8a)"
   430) 
   431) !-------------------------------------------------------------------------------
   432) !
   433) !-------------------------------------------------------------------------------
   434) 
   435)    if (present(rlistout)) then
   436)       nflds1 = mct_aVect_nRAttr(aVect1)
   437)       nflds2 = mct_aVect_nRAttr(aVect2)
   438)       rlistout = ''
   439)       list1 = ''
   440)       list2 = ''
   441)       if (nflds1 > 0 .and. nflds2 > 0) then
   442)          list1 = mct_aVect_exportRList2c(aVect1)
   443)          list2 = mct_aVect_exportRlist2c(aVect2)
   444)          call shr_string_listIntersect(list1,list2,rlistout)
   445)       endif
   446)    endif

   642) ! !IROUTINE: mct_aVect_accum - accumulate attributes from one aVect to another
   643) !
   644) ! !DESCRIPTION:
   645) ! This routine accumulates from input argment {\tt aVin} into the output 
   646) ! {\it AttrVect} argument {\tt aVout} the real and integer attributes specified in 
   647) ! input {\tt CHARACTER} argument {\tt iList} and {\tt rList}. The attributes can
   648) ! be listed in any order.  If neither {\tt iList} nor {\tt rList} are provided, 
   649) ! all attributes shared between {\tt aVin} and {\tt aVout} will be copied.
   650) !
   651) ! If any attributes in {\tt aVout} have different names but represent the
   652) ! the same quantity and should still be copied, you must provide a translation
   653) ! argument {\tt TrList} and/or {\tt TiList}.  The translation arguments should
   654) ! be identical to the {\tt rList} or {\tt iList} but with the correct {\tt aVout}
   655) ! name subsititued at the appropriate place.
   656) !
   657) ! This routine leverages the mct copy routines directly
   658) !
   659) ! {\bf N.B.:}  This routine will fail if the {\tt aVout} is not initialized or
   660) ! if any of the specified attributes are not present in either {\tt aVout} or {\tt aVin}.
   661) !
   662) ! !REVISION HISTORY:
   663) !    2002 Sep 15 - ? - initial version.
   664) !     2013-Jul-20 - T. Craig -- updated
   665) !
   666) ! !INTERFACE: ------------------------------------------------------------------
   667) 
   668)  subroutine mct_avect_accum(aVin, aVout, rList, TrList, iList, TiList, vector, sharedIndices,counter)
   669) 
   670)       implicit none
   671) 
   672) ! !INPUT PARAMETERS: 
   673) 
   674)       type(mct_avect),            intent(in)    :: aVin
   675)       character(len=*), optional, intent(in)    :: iList
   676)       character(len=*), optional, intent(in)    :: rList
   677)       character(len=*), optional, intent(in)    :: TiList
   678)       character(len=*), optional, intent(in)    :: TrList
   679)       logical, optional,          intent(in)    :: vector 
   680)       type(mct_avect_SharedIndices), optional, intent(in) :: sharedIndices
   681) 
   682) ! !OUTPUT PARAMETERS: 
   683) 
   684)       type(mct_avect),         intent(inout) :: aVout
   685)       integer, optional,       intent(inout) :: counter
   686) 
   687) 
   688) ! !REVISION HISTORY:
   689) 
   690) !EOP ___________________________________________________________________
   691) 
   692)    !--- local ---
   693)   logical :: usevector
   694)   integer(IN) :: lsize,nflds,npts,i,j
   695)   type(mct_avect) :: avotmp  ! temporary aVout copy
   696)   character(*),parameter :: subName = '(mct_aVect_accum) '
   697) 
   698) !-----------------------------------------------------------------
   699) 
   700)   usevector = .false.
   701)   if (present(vector)) then
   702)      usevector = vector
   703)   endif
   704) 
   705)   if (present(counter)) then
   706)      counter = counter + 1
   707)   endif
   708) 
   709)   ! --- allocate avotmp, a duplciate of aVout
   710) 
   711)   lsize = mct_aVect_lsize(aVout)
   712)   call mct_avect_init(avotmp,aVout,lsize)
   713)   call mct_avect_zero(avotmp)
   714) 
   715)   ! --- copy aVin fields into avotmp
   716) 
   717)   if (present(sharedIndices)) then
   718) 
   719)      if (present(rList) .and. present(iList)) then
   720)         if (present(trList) .and. present(tilist)) then
   721)            call mct_avect_copy(aVin, avotmp, rList, TrList, iList, tiList, vector = usevector, sharedIndices=sharedIndices)
   722)         elseif (present(trList)) then
   723)            call mct_avect_copy(aVin, avotmp, rList, TrList, iList, vector = usevector, sharedIndices=sharedIndices)
   724)         elseif (present(tiList)) then
   725)            call mct_avect_copy(aVin, avotmp, rList, iList=iList, tiList=tiList, vector = usevector, sharedIndices=sharedIndices)
   726)         else
   727)            call mct_avect_copy(aVin, avotmp, rList=rList, iList=iList, vector = usevector, sharedIndices=sharedIndices)
   728)         endif
   729)      else if (present(rList)) then
   730)         if (present(trList)) then
   731)            call mct_avect_copy(aVin, avotmp, rList, TrList, vector = usevector, sharedIndices=sharedIndices)
   732)         else
   733)            call mct_avect_copy(aVin, avotmp, rList, vector = usevector, sharedIndices=sharedIndices)
   734)         endif
   735) 
   736)      else if (present(iList)) then
   737)         if (present(tiList)) then
   738)            call mct_avect_copy(aVin, avotmp, ilist=iList, tiList=tiList, vector = usevector, sharedIndices=sharedIndices)
   739)         else
   740)            call mct_avect_copy(aVin, avotmp, ilist=iList, vector = usevector, sharedIndices=sharedIndices)
   741)         endif
   742) 
   743)      else
   744)         call mct_avect_copy(aVin, avotmp, vector=usevector, sharedIndices=sharedIndices)
   745) 
   746)      endif
   747) 
   748)   else   ! sharedIndices
   749) 
   750)      if (present(rList) .and. present(iList)) then
   751)         if (present(trList) .and. present(tilist)) then
   752)            call mct_avect_copy(aVin, avotmp, rList, TrList, iList, tiList, vector = usevector)
   753)         elseif (present(trList)) then
   754)            call mct_avect_copy(aVin, avotmp, rList, TrList, iList, vector = usevector)
   755)         elseif (present(tiList)) then
   756)            call mct_avect_copy(aVin, avotmp, rList, iList=iList, tiList=tiList, vector = usevector)
   757)         else
   758)            call mct_avect_copy(aVin, avotmp, rList=rList, iList=iList, vector = usevector)
   759)         endif
   760)      else if (present(rList)) then
   761)         if (present(trList)) then
   762)            call mct_avect_copy(aVin, avotmp, rList, TrList, vector = usevector)
   763)         else
   764)            call mct_avect_copy(aVin, avotmp, rList, vector = usevector)
   765)         endif
   766) 
   767)      else if (present(iList)) then
   768)         if (present(tiList)) then
   769)            call mct_avect_copy(aVin, avotmp, ilist=iList, tiList=tiList, vector = usevector)
   770)         else
   771)            call mct_avect_copy(aVin, avotmp, ilist=iList, vector = usevector)
   772)         endif
   773) 
   774)      else
   775)         call mct_avect_copy(aVin, avotmp, vector=usevector)
   776) 
   777)      endif
   778) 
   779)   endif ! shared indices
   780) 
   781)   ! --- accumulate avotmp into avout
   782) 
   783)   nflds = mct_aVect_nRAttr(aVout)
   784)   npts  = mct_aVect_lsize (aVout)
   785) !DIR$ CONCURRENT
   786) !DIR$ PREFERVECTOR
   787)   do i=1,npts 
   788)   do j=1,nflds
   789)      aVout%rattr(j,i) = aVout%rattr(j,i) + avotmp%rattr(j,i)
   790)   enddo
   791)   enddo
   792) 
   793)   ! --- clean avotmp
   794) 
   795)   call mct_avect_clean(avotmp)
   796) 
   797)  end subroutine mct_avect_accum

   866) !     is replaced by the product of the data in {\tt initav} and {\tt fld1}
   867) !     from {\tt av1}. NOTE:  this assume {\tt initav} has the exact same
   868) !     attributes in the same order as {\tt av}.
   869) !
   870) !
   871) ! !REVISION HISTORY:
   872) !     2007-Jun-11 - M. Vertenstein -- initial version
   873) !
   874) ! !INTERFACE: ------------------------------------------------------------------
   875) 
   876) subroutine mct_avect_mult(av,av1,fld1,avlist)
   877) 
   878) ! !USES:
   879) 
   880) ! !INPUT/OUTPUT PARAMETERS:
   881) 
   882)    type(mct_aVect)      ,intent(inout) :: av       ! attribute vector output
   883)    type(mct_aVect)      ,intent(in)    :: av1      ! attribute vector input
   884)    character(*)         ,intent(in)    :: fld1     ! av1 field name
   885)    character(*),optional,intent(in)    :: avlist   ! sublist of field in av
   886) 
   887) !EOP
   888) 
   889)    !--- local ---
   890)    integer(IN) :: n,m            ! generic indicies
   891)    integer(IN) :: npts           ! number of points (local) in an aVect field
   892)    integer(IN) :: nfld           ! number of fields (local) in an aVect field
   893)    integer(IN) :: nfldi          ! number of fields (local) in an aVect field
   894)    integer(IN) :: nptsx          ! number of points (local) in an aVect field
   895)    integer(IN) :: nptsi          ! number of points (local) in an aVect field
   896)    integer(IN) :: kfld           ! field number of fld1 in av1
   897)    integer(IN),dimension(:),allocatable :: kfldin   ! field numbers of avlist in av
   898)    type(mct_list)   :: blist     ! avlist as a List
   899)    type(mct_string) :: tattr     ! an attribute
   900) 
   901)    !--- formats ---
   902)    character(*),parameter :: subName = '(mct_aVect_mult) '
   903) 
   904) !-------------------------------------------------------------------------------
   905) !
   906) !-------------------------------------------------------------------------------
   907) 
   908)    nptsx = mct_aVect_lsize(av1)
   909)    npts  = mct_aVect_lsize(av)
   910)    if (nptsx /= npts .and. s_loglev > 0) write(s_logunit,*) subName,' ERROR: npts error1 ',npts,nptsx
   911) 
   912)    kfld  = mct_aVect_indexRA(av1,fld1,perrWith=subName)
   913) 
   914)    if (present(avlist)) then
   915) 
   916)      call mct_list_init(blist,avlist)
   917) 
   918)      nfld=mct_list_nitem(blist)
   919) 
   920)      allocate(kfldin(nfld))
   921)      do m=1,nfld
   922)        call mct_list_get(tattr,m,blist)
   923)        kfldin(m) = mct_aVect_indexRA(av,mct_string_toChar(tattr))
   924)        call mct_string_clean(tattr)
   925)      enddo
   926)      call mct_list_clean(blist)
   927) 
   928) #ifdef CPP_VECTOR
   929)      do m=1,nfld
   930) !CDIR SELECT(VECTOR)
   931) !DIR$ CONCURRENT
   932)      do n=1,npts
   933) #else
   934)      do n=1,npts
   935)      do m=1,nfld
   936) #endif
   937)         av%rAttr(kfldin(m),n) = av%rAttr(kfldin(m),n)*av1%rAttr(kfld,n)
   938)      enddo
   939)      enddo
   940) 
   941)      deallocate(kfldin)
   942) 
   943)    else
   944) 
   945)      nfld  = mct_aVect_nRAttr(av)
   946) 
   947) #ifdef CPP_VECTOR
   948)      do m=1,nfld
   949) !CDIR SELECT(VECTOR)
   950) !DIR$ CONCURRENT
   951)      do n=1,npts
   952) #else
   953)      do n=1,npts
   954)      do m=1,nfld
   955) #endif
   956)         av%rAttr(m,n) = av%rAttr(m,n)*av1%rAttr(kfld,n)
   957)      enddo
   958)      enddo
   959) 
   960)    endif
   961)
