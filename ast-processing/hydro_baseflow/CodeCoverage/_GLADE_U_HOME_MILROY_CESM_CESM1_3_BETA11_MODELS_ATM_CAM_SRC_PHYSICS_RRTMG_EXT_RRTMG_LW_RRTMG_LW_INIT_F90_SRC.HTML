    29)       subroutine rrtmg_lw_ini
    30) ! **************************************************************************
    31) !
    32) !  Original version:       Michael J. Iacono; July, 1998
    33) !  First revision for NCAR CCM:   September, 1998
    34) !  Second revision for RRTM_V3.0:  September, 2002
    35) !
    36) !  This subroutine performs calculations necessary for the initialization
    37) !  of the longwave model.  Lookup tables are computed for use in the LW
    38) !  radiative transfer, and input absorption coefficient data for each
    39) !  spectral band are reduced from 256 g-point intervals to 140.
    40) ! **************************************************************************
    41) 
    42)       use parrrtm, only : mg, nbndlw, ngptlw
    43)       use rrlw_tbl, only: ntbl, tblint, pade, bpade, tau_tbl, exp_tbl, tfn_tbl
    44)       use rrlw_vsn, only: hvrini, hnamini
    45) 
    46) ! ------- Local -------
    47) 
    48)       integer :: itr, ibnd, igc, ig, ind, ipr 
    49)       integer :: igcsm, iprsm
    50) 
    51)       real(kind=r8) :: wtsum, wtsm(mg)        !
    52)       real(kind=r8) :: tfn                    !
    53) 
    54) ! ------- Definitions -------
    55) !     Arrays for 10000-point look-up tables:
    56) !     TAU_TBL Clear-sky optical depth (used in cloudy radiative transfer)
    57) !     EXP_TBL Exponential lookup table for ransmittance
    58) !     TFN_TBL Tau transition function; i.e. the transition of the Planck
    59) !             function from that for the mean layer temperature to that for
    60) !             the layer boundary temperature as a function of optical depth.
    61) !             The "linear in tau" method is used to make the table.
    62) !     PADE    Pade approximation constant (= 0.278)
    63) !     BPADE   Inverse of the Pade approximation constant
    64) !
    65) 
    66)       hvrini = '$Revision: 1.2 $'
    67) 
    68) ! Initialize model data
    69)       call lwdatinit
    70)       call lwcmbdat               ! g-point interval reduction data
    71)       call lwcldpr                ! cloud optical properties
    72)       call lwatmref               ! reference MLS profile
    73)       call lwavplank              ! Planck function 
    74)       call lw_kgb01               ! molecular absorption coefficients
    75)       call lw_kgb02
    76)       call lw_kgb03
    77)       call lw_kgb04
    78)       call lw_kgb05
    79)       call lw_kgb06
    80)       call lw_kgb07
    81)       call lw_kgb08
    82)       call lw_kgb09
    83)       call lw_kgb10
    84)       call lw_kgb11
    85)       call lw_kgb12
    86)       call lw_kgb13
    87)       call lw_kgb14
    88)       call lw_kgb15
    89)       call lw_kgb16
    90) 
    91) ! Compute lookup tables for transmittance, tau transition function,
    92) ! and clear sky tau (for the cloudy sky radiative transfer).  Tau is 
    93) ! computed as a function of the tau transition function, transmittance 
    94) ! is calculated as a function of tau, and the tau transition function 
    95) ! is calculated using the linear in tau formulation at values of tau 
    96) ! above 0.01.  TF is approximated as tau/6 for tau < 0.01.  All tables 
    97) ! are computed at intervals of 0.001.  The inverse of the constant used
    98) ! in the Pade approximation to the tau transition function is set to b.
    99) 
   100)       tau_tbl(0) = 0.0_r8
   101)       tau_tbl(ntbl) = 1.e10_r8
   102)       exp_tbl(0) = 1.0_r8
   103)       exp_tbl(ntbl) = 0.0_r8
   104)       tfn_tbl(0) = 0.0_r8
   105)       tfn_tbl(ntbl) = 1.0_r8
   106)       bpade = 1.0_r8 / pade
   107)       do itr = 1, ntbl-1
   108)          tfn = float(itr) / float(ntbl)
   109)          tau_tbl(itr) = bpade * tfn / (1._r8 - tfn)
   110)          exp_tbl(itr) = exp(-tau_tbl(itr))
   111)          if (tau_tbl(itr) .lt. 0.06_r8) then
   112)             tfn_tbl(itr) = tau_tbl(itr)/6._r8
   113)          else
   114)             tfn_tbl(itr) = 1._r8-2._r8*((1._r8/tau_tbl(itr))-(exp_tbl(itr)/(1.-exp_tbl(itr))))
   115)          endif
   116)       enddo
   117) 
   118) ! Perform g-point reduction from 16 per band (256 total points) to
   119) ! a band dependant number (140 total points) for all absorption
   120) ! coefficient input data and Planck fraction input data.
   121) ! Compute relative weighting for new g-point combinations.
   122) 
   123)       igcsm = 0
   124)       do ibnd = 1,nbndlw
   125)          iprsm = 0
   126)          if (ngc(ibnd).lt.mg) then
   127)             do igc = 1,ngc(ibnd) 
   128)                igcsm = igcsm + 1
   129)                wtsum = 0._r8
   130)                do ipr = 1, ngn(igcsm)
   131)                   iprsm = iprsm + 1
   132)                   wtsum = wtsum + wt(iprsm)
   133)                enddo
   134)                wtsm(igc) = wtsum
   135)             enddo
   136)             do ig = 1, ng(ibnd)
   137)                ind = (ibnd-1)*mg + ig
   138)                rwgt(ind) = wt(ig)/wtsm(ngm(ind))
   139)             enddo
   140)          else
   141)             do ig = 1, ng(ibnd)
   142)                igcsm = igcsm + 1
   143)                ind = (ibnd-1)*mg + ig
   144)                rwgt(ind) = 1.0_r8
   145)             enddo
   146)          endif
   147)       enddo
   148) 
   149) ! Reduce g-points for absorption coefficient data in each LW spectral band.
   150) 
   151)       call cmbgb1
   152)       call cmbgb2
   153)       call cmbgb3
   154)       call cmbgb4
   155)       call cmbgb5
   156)       call cmbgb6
   157)       call cmbgb7
   158)       call cmbgb8
   159)       call cmbgb9
   160)       call cmbgb10
   161)       call cmbgb11
   162)       call cmbgb12
   163)       call cmbgb13
   164)       call cmbgb14
   165)       call cmbgb15
   166)       call cmbgb16
   167) 
   168)       end subroutine rrtmg_lw_ini

   338) !***************************************************************************
   339)       subroutine cmbgb1
   340) !***************************************************************************
   341) !
   342) !  Original version:    MJIacono; July 1998
   343) !  Revision for GCMs:   MJIacono; September 1998
   344) !  Revision for RRTMG:  MJIacono, September 2002
   345) !  Revision for F90 reformatting:  MJIacono, June 2006
   346) !
   347) !  The subroutines CMBGB1->CMBGB16 input the absorption coefficient
   348) !  data for each band, which are defined for 16 g-points and 16 spectral
   349) !  bands. The data are combined with appropriate weighting following the
   350) !  g-point mapping arrays specified in RRTMINIT.  Plank fraction data
   351) !  in arrays FRACREFA and FRACREFB are combined without weighting.  All
   352) !  g-point reduced data are put into new arrays for use in RRTM.
   353) !
   354) !  band 1:  10-350 cm-1 (low key - h2o; low minor - n2)
   355) !                       (high key - h2o; high minor - n2)
   356) !  note: previous versions of rrtm band 1: 
   357) !        10-250 cm-1 (low - h2o; high - h2o)
   358) !***************************************************************************
   359) 
   360)       use parrrtm, only : mg, nbndlw, ngptlw, ng1
   361)       use rrlw_kg01, only: fracrefao, fracrefbo, kao, kbo, kao_mn2, kbo_mn2, &
   362)                            selfrefo, forrefo, &
   363)                            fracrefa, fracrefb, ka, kb, ka_mn2, kb_mn2, &
   364)                            selfref, forref
   365) 
   366) ! ------- Local -------
   367)       integer :: jt, jp, igc, ipr, iprsm 
   368)       real(kind=r8) :: sumk, sumk1, sumk2, sumf1, sumf2
   369) 
   370) 
   371)       do jt = 1,5
   372)          do jp = 1,13
   373)             iprsm = 0
   374)             do igc = 1,ngc(1)
   375)                sumk = 0.
   376)                do ipr = 1, ngn(igc)
   377)                   iprsm = iprsm + 1
   378)                   sumk = sumk + kao(jt,jp,iprsm)*rwgt(iprsm)
   379)                enddo
   380)                ka(jt,jp,igc) = sumk
   381)             enddo
   382)          enddo
   383)          do jp = 13,59
   384)             iprsm = 0
   385)             do igc = 1,ngc(1)
   386)                sumk = 0.
   387)                do ipr = 1, ngn(igc)
   388)                   iprsm = iprsm + 1
   389)                   sumk = sumk + kbo(jt,jp,iprsm)*rwgt(iprsm)
   390)                enddo
   391)                kb(jt,jp,igc) = sumk
   392)             enddo
   393)          enddo
   394)       enddo
   395) 
   396)       do jt = 1,10
   397)          iprsm = 0
   398)          do igc = 1,ngc(1)
   399)             sumk = 0.
   400)             do ipr = 1, ngn(igc)
   401)                iprsm = iprsm + 1
   402)                sumk = sumk + selfrefo(jt,iprsm)*rwgt(iprsm)
   403)             enddo
   404)             selfref(jt,igc) = sumk
   405)          enddo
   406)       enddo
   407) 
   408)       do jt = 1,4
   409)          iprsm = 0
   410)          do igc = 1,ngc(1)
   411)             sumk = 0.
   412)             do ipr = 1, ngn(igc)
   413)                iprsm = iprsm + 1
   414)                sumk = sumk + forrefo(jt,iprsm)*rwgt(iprsm)
   415)             enddo
   416)             forref(jt,igc) = sumk
   417)          enddo
   418)       enddo
   419) 
   420)       do jt = 1,19
   421)          iprsm = 0
   422)          do igc = 1,ngc(1)
   423)             sumk1 = 0.
   424)             sumk2 = 0.
   425)             do ipr = 1, ngn(igc)
   426)                iprsm = iprsm + 1
   427)                sumk1 = sumk1 + kao_mn2(jt,iprsm)*rwgt(iprsm)
   428)                sumk2 = sumk2 + kbo_mn2(jt,iprsm)*rwgt(iprsm)
   429)             enddo
   430)             ka_mn2(jt,igc) = sumk1
   431)             kb_mn2(jt,igc) = sumk2
   432)          enddo
   433)       enddo
   434) 
   435)       iprsm = 0
   436)       do igc = 1,ngc(1)
   437)          sumf1 = 0.
   438)          sumf2 = 0.
   439)          do ipr = 1, ngn(igc)
   440)             iprsm = iprsm + 1
   441)             sumf1= sumf1+ fracrefao(iprsm)
   442)             sumf2= sumf2+ fracrefbo(iprsm)
   443)          enddo
   444)          fracrefa(igc) = sumf1
   445)          fracrefb(igc) = sumf2
   446)       enddo
   447) 
   448)       end subroutine cmbgb1
