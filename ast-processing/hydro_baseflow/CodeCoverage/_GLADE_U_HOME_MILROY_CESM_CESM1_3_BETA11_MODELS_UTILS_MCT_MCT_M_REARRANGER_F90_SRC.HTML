     8) ! !MODULE: m_Rearranger -- Remaps an AttrVect within a group of processes
     9) !
    10) ! !DESCRIPTION:
    11) ! This module provides routines and datatypes for rearranging data
    12) ! between two {\tt Attribute Vectors} defined on the same grid but
    13) ! with two different {\tt GlobalSegMaps}.  ''Rearrange'' is a
    14) ! generalized form of a parallel matrix transpose.
    15) ! A parallel matrix transpose can take advantage of symmetry in the
    16) ! data movement algorithm.  An MCT Rearranger makes no assumptions
    17) ! about symmetry.
    18) !
    19) ! When data needs to move between two components and the components
    20) ! share any processors, use m\_Rearranger.  If the components are on
    21) ! distinct sets of processors, use m\_Transfer.

    96) ! !IROUTINE: Init_ - Initialize a Rearranger
    97) !
    98) ! !DESCRIPTION:
    99) ! This routine takes two {\tt GlobalSegMap} inputs, {\tt SourceGSMap}
   100) ! and {\tt TargetGSMap} and build a Rearranger {\tt OutRearranger}
   101) ! between them. {\tt myComm} is used for the internal communication.
   102) !
   103) ! {\bf N.B.} The two {\tt GlolbalSegMap} inputs must be initialized so
   104) ! that the index values on a processor are in ascending order.
   105) !
   106) ! !INTERFACE:
   107) 
   108)  subroutine init_(SourceGSMap,TargetGSMap,myComm,OutRearranger)
   109) 
   110) !
   111) ! !USES:
   112) !
   113) 
   114)    use m_MCTWorld,     only : ThisMCTWorld
   115)    use m_GlobalSegMap, only : GlobalSegMap
   116)    use m_GlobalSegMap, only : GSMap_lsize => lsize
   117)    use m_GlobalSegMap, only : GSMap_increasing => increasing
   118)    use m_Router,       only : Router     
   119)    use m_Router,       only : Router_init => init
   120)    use m_mpif90
   121)    use m_die
   122)    use m_stdio
   123) 
   124)    implicit none
   125)   
   126) ! !INPUT PARAMETERS:
   127) !
   128)    type(GlobalSegMap), intent(in)            :: SourceGSMap, TargetGSMap
   129)    integer,            intent(in)            :: myComm
   130) 
   131) ! !OUTPUT PARAMETERS:
   132) !
   133)    type(Rearranger),   intent(out)           :: OutRearranger
   134) 
   135) ! !REVISION HISTORY:
   136) ! 31Jan02 - E.T. Ong <eong@mcs.anl.gov> - initial prototype
   137) ! 20Mar02 - E.T. Ong <eong@mcs.anl.gov> - working code
   138) ! 05Jun02 - E.T. Ong <eong@mcs.anl.gov> - Use LocalPack
   139) ! 30Mar06 - P. Worley <worleyph@ornl.gov> - added max_nprocs,
   140) !           used in communication optimizations in rearrange
   141) !EOP ___________________________________________________________________
   142)    character(len=*),parameter :: myname_=myname//'::init_'
   143)    integer,dimension(:,:),pointer :: temp_seg_starts,temp_seg_lengths
   144)    integer,dimension(:),pointer :: temp_pe_list,temp_numsegs,temp_locsize
   145)    integer :: temp_maxsize,temp_nprocs,maxsegcount
   146)    integer :: procindex,nprocs,nseg,len,myPid
   147)    integer :: src_seg_start,src_seg_length,trg_seg_start,trg_seg_length
   148)    integer :: i,j,k,l,m,n,ier
   149)    logical :: SendingToMyself,ReceivingFromMyself
   150) 
   151) 
   152)    ! Initialize Router component of Rearranger
   153)    call Router_init(SourceGSMap,TargetGSMap,myComm,OutRearranger%SendRouter)
   154)    call Router_init(TargetGSMap,SourceGSMap,myComm,OutRearranger%RecvRouter)
   155) 
   156)    call MP_comm_size(MP_COMM_WORLD,max_nprocs,ier)
   157)    if(ier/=0) call MP_perr_die(myname_,'MP_comm_size',ier)
   158) 
   159)    ! SANITY CHECK: Make sure that if SendRouter is sending to self, then,
   160)    ! by definition, RecvRouter is also receiving from self. If this is not 
   161)    ! true, then write to stderr and die. 
   162) 
   163)    call MP_comm_rank(ThisMCTWorld%MCT_comm,myPid,ier)

   165) 
   166)    SendingToMyself = .false.
   167)    ReceivingFromMyself = .false.
   168) 
   169)    do i=1,OutRearranger%SendRouter%nprocs
   170)       if(OutRearranger%SendRouter%pe_list(i) == myPid) then
   171) 	 SendingToMyself = .true.
   172)       endif
   173)    enddo
   174) 
   175)    do i=1,OutRearranger%RecvRouter%nprocs
   176)       if(OutRearranger%RecvRouter%pe_list(i) == myPid) then
   177) 	 ReceivingFromMyself = .true.
   178)       endif
   179)    enddo
   180) 
   181)    if( SendingToMyself.or.ReceivingFromMyself ) then
   182)       if( .not. (SendingToMyself.and.ReceivingFromMyself) ) then
   183) 	 call die(myname_,"SendRouter is not compatible with RecvRouter")
   184)       endif
   185)    endif
   186) 
   187) 
   188)    ! If not sending to nor receiving from own processor then initialize 
   189)    ! the rearranger so that no local copy can be made. Then end the routine. 
   190) 
   191)    if( .not. (SendingToMyself.or.ReceivingFromMyself) ) then
   192)       nullify(OutRearranger%LocalPack)
   193)       allocate(OutRearranger%LocalPack(0,0),stat=ier)
   194)       if(ier/=0) call die(myname_,'allocate(OutRearranger%LocalPack(0,0))',ier)
   195)       OutRearranger%LocalSize=0
   196)    endif
   197) 
   198) 
   199)    ! Start the process of Router modification: Router information for 
   200)    ! the local processor is extracted out and put into the local copy 
   201)    ! structure- Rearranger%LocalPack. Router structures are then reassigned 
   202)    ! to exclude the local copy information.
   203) 
   204) 
   205)    ! Operate on SendRouter and create local copy structures.
   206) 
   207)    if( SendingToMyself.and.ReceivingFromMyself ) then
   208) 
   209)    temp_nprocs = OutRearranger%SendRouter%nprocs-1
   210)    maxsegcount = SIZE(OutRearranger%SendRouter%seg_starts,2)
   211) 
   212)    ! Allocate temporary Router structures to be used for modifying SendRouter 
   213)    nullify(temp_seg_starts,temp_seg_lengths,temp_pe_list, &
   214)            temp_numsegs,temp_locsize)
   215)    allocate(temp_seg_starts(temp_nprocs,maxsegcount), &
   216)             temp_seg_lengths(temp_nprocs,maxsegcount), &
   217) 	    temp_pe_list(temp_nprocs), &
   218)             temp_numsegs(temp_nprocs), &
   219)             temp_locsize(temp_nprocs), stat=ier)
   220)    if(ier/=0) call die(myname_,'allocate(temp_seg_starts...)',ier)
   221) 
   222)    temp_maxsize=0
   223)    procindex=0
   224)    nullify(OutRearranger%LocalPack)
   225) 
   226)    ! Start assigning Rearranger copy structures and  
   227)    ! non-local Router components
   228)    do i=1,OutRearranger%SendRouter%nprocs
   229) 
   230)       ! Gather local copy information 
   231)       if(OutRearranger%SendRouter%pe_list(i) == myPid) then
   232) 
   233) 	 ! Allocate Rearranger copy structure
   234) 	 allocate(OutRearranger%LocalPack(2, &
   235)                   OutRearranger%SendRouter%locsize(i)),stat=ier)
   236) 	 if(ier/=0) call die(myname_,'allocate(OutRearranger%LocalPack)',ier)
   237) 	 OutRearranger%LocalPack = 0
   238) 
   239) 	 m=0
   240) 	 do nseg = 1,OutRearranger%SendRouter%num_segs(i)
   241) 	    src_seg_start = OutRearranger%SendRouter%seg_starts(i,nseg)
   242) 	    src_seg_length = OutRearranger%SendRouter%seg_lengths(i,nseg)-1
   243) 	    do len=0,src_seg_length
   244) 	       m=m+1
   245) 	       OutRearranger%LocalPack(2,m) = src_seg_start+len
   246) 	    enddo
   247) 	 enddo
   248) 
   249)       else
   250) 
   251) 	 ! Gather non-local Router information
   252) 	 procindex = procindex+1
   253) 	 temp_seg_starts(procindex,1:maxsegcount) = &
   254)             OutRearranger%SendRouter%seg_starts(i,1:maxsegcount)
   255) 	 temp_seg_lengths(procindex,1:maxsegcount) = &
   256)             OutRearranger%SendRouter%seg_lengths(i,1:maxsegcount)
   257) 	 temp_pe_list(procindex) = OutRearranger%SendRouter%pe_list(i)
   258) 	 temp_numsegs(procindex) = OutRearranger%SendRouter%num_segs(i)
   259) 	 temp_locsize(procindex) = OutRearranger%SendRouter%locsize(i)
   260) 	 temp_maxsize = max(temp_locsize(procindex),temp_maxsize)
   261) 
   262)       endif
   263) 
   264)    enddo
   265) 
   266)    ! Copy SendRouter components back in
   267) 
   268)    ! Deallocate existing SendRouter components
   269)    deallocate(OutRearranger%SendRouter%seg_starts,&
   270)               OutRearranger%SendRouter%seg_lengths, &
   271) 	      OutRearranger%SendRouter%pe_list, &
   272)               OutRearranger%SendRouter%num_segs, &
   273)               OutRearranger%SendRouter%locsize,stat=ier)
   274)    if(ier/=0) call die(myname_, &
   275)                   'deallocate(OutRearranger%SendRouter%seg_starts...)',ier)
   276) 
   277)    ! Re-allocate SendRouter components
   278)    allocate(OutRearranger%SendRouter%seg_starts(temp_nprocs,maxsegcount), &
   279)             OutRearranger%SendRouter%seg_lengths(temp_nprocs,maxsegcount), &
   280) 	    OutRearranger%SendRouter%pe_list(temp_nprocs), &
   281) 	    OutRearranger%SendRouter%num_segs(temp_nprocs), &
   282)             OutRearranger%SendRouter%locsize(temp_nprocs),stat=ier)
   283)    if(ier/=0) call die(myname_, &
   284)                    'allocate(OutRearranger%SendRouter%seg_starts...)',ier)      
   285) 
   286)    ! Copy back in the spliced router information
   287)    OutRearranger%SendRouter%nprocs = temp_nprocs
   288)    OutRearranger%SendRouter%seg_starts(1:temp_nprocs,1:maxsegcount) = &
   289)       temp_seg_starts(1:temp_nprocs,1:maxsegcount)
   290)    OutRearranger%SendRouter%seg_lengths(1:temp_nprocs,1:maxsegcount) = &
   291)       temp_seg_lengths(1:temp_nprocs,1:maxsegcount)
   292)    OutRearranger%SendRouter%pe_list(1:temp_nprocs) = &
   293)       temp_pe_list(1:temp_nprocs)
   294)    OutRearranger%SendRouter%num_segs(1:temp_nprocs) = &
   295)       temp_numsegs(1:temp_nprocs)
   296)    OutRearranger%SendRouter%locsize(1:temp_nprocs) = &
   297)       temp_locsize(1:temp_nprocs)
   298)    OutRearranger%SendRouter%maxsize = temp_maxsize
   299) 
   300)    deallocate(temp_seg_starts,temp_seg_lengths,temp_pe_list, &
   301)               temp_numsegs,temp_locsize,stat=ier)
   302)    if(ier/=0) call die(myname_,'deallocate(temp_seg_starts...)',ier)      
   303) 
   304) 
   305)    ! :::::::::::::::::::::::::::::::::::::::::::::::::::::::
   306) 
   307) 
   308)    ! Operate on RecvRouter and create local copy structures.
   309) 
   310)    temp_nprocs = OutRearranger%RecvRouter%nprocs-1
   311)    maxsegcount = SIZE(OutRearranger%RecvRouter%seg_starts,2)
   312) 
   313)   ! Allocate temporary Router structures to be used for modifying RecvRouter
   314)    nullify(temp_seg_starts,temp_seg_lengths,temp_pe_list, &
   315)            temp_numsegs,temp_locsize)
   316)    allocate(temp_seg_starts(temp_nprocs,maxsegcount), &
   317)             temp_seg_lengths(temp_nprocs,maxsegcount), &
   318) 	    temp_pe_list(temp_nprocs),temp_numsegs(temp_nprocs), &
   319)             temp_locsize(temp_nprocs),stat=ier)
   320)    if(ier/=0) call die(myname_,'allocate(temp_seg_starts...)',ier)
   321) 
   322)    temp_maxsize=0
   323)    procindex = 0
   324) 
   325)    ! Start assigning Rearranger copy structures and  
   326)    ! non-local Router components
   327)    do i=1,OutRearranger%RecvRouter%nprocs
   328) 
   329)       ! Gather local copy information 
   330)       if(OutRearranger%RecvRouter%pe_list(i) == myPid) then
   331) 
   332) 	 ! Senity Check for Router%locsize
   333) 	 if( (SIZE(OutRearranger%LocalPack,2) /= &
   334)               OutRearranger%RecvRouter%locsize(i)) ) then
   335) 	    call die(myname_, &
   336)                      'Router Error: Local RecvRouter%locsize(myPid) /=  &
   337)                      & Local SendRouter%locsize(myPid)')
   338) 	 endif
   339) 
   340) 	 OutRearranger%LocalSize = OutRearranger%RecvRouter%locsize(i)
   341) 
   342) 	 m=0
   343) 	 do nseg = 1,OutRearranger%RecvRouter%num_segs(i)
   344) 	    trg_seg_start = OutRearranger%RecvRouter%seg_starts(i,nseg)
   345) 	    trg_seg_length = OutRearranger%RecvRouter%seg_lengths(i,nseg)-1
   346) 	    do len=0,trg_seg_length
   347) 	       m=m+1
   348) 	       OutRearranger%LocalPack(1,m) = trg_seg_start+len
   349) 	    enddo
   350) 	 enddo
   351) 	 
   352)       else
   353) 
   354) 	 ! Gather non-local Router information
   355) 	 procindex = procindex+1
   356) 	 temp_seg_starts(procindex,1:maxsegcount) = &
   357) 	    OutRearranger%RecvRouter%seg_starts(i,1:maxsegcount)
   358) 	 temp_seg_lengths(procindex,1:maxsegcount) = &
   359)             OutRearranger%RecvRouter%seg_lengths(i,1:maxsegcount)
   360) 	 temp_pe_list(procindex) = OutRearranger%RecvRouter%pe_list(i)
   361) 	 temp_numsegs(procindex) = OutRearranger%RecvRouter%num_segs(i)
   362) 	 temp_locsize(procindex) = OutRearranger%RecvRouter%locsize(i)
   363) 	 temp_maxsize = max(temp_locsize(procindex),temp_maxsize)
   364) 
   365)       endif
   366) 
   367)    enddo
   368) 
   369)    ! Copy RecvRouter components back in
   370) 
   371)    ! Deallocate existing SendRouter components
   372)    deallocate(OutRearranger%RecvRouter%seg_starts, &
   373)               OutRearranger%RecvRouter%seg_lengths, &
   374) 	      OutRearranger%RecvRouter%pe_list, &
   375)               OutRearranger%RecvRouter%num_segs, &
   376)               OutRearranger%RecvRouter%locsize,stat=ier)
   377)    if(ier/=0) call die(myname_, &
   378)                    'deallocate(OutRearranger%RecvRouter%seg_starts...)',ier)
   379) 
   380)    ! Re-allocate RecvRouter components
   381)    allocate(OutRearranger%RecvRouter%seg_starts(temp_nprocs,maxsegcount), &
   382)             OutRearranger%RecvRouter%seg_lengths(temp_nprocs,maxsegcount), &
   383) 	    OutRearranger%RecvRouter%pe_list(temp_nprocs), &
   384) 	    OutRearranger%RecvRouter%num_segs(temp_nprocs), &
   385)             OutRearranger%RecvRouter%locsize(temp_nprocs),stat=ier)
   386)    if(ier/=0) call die(myname_, &
   387)                    'allocate(OutRearranger%RecvRouter%seg_starts...)',ier)      
   388) 
   389)     ! Copy back in the spliced router information
   390)    OutRearranger%RecvRouter%nprocs = temp_nprocs
   391)    OutRearranger%RecvRouter%seg_starts(1:temp_nprocs,1:maxsegcount) =  &
   392)       temp_seg_starts(1:temp_nprocs,1:maxsegcount)
   393)    OutRearranger%RecvRouter%seg_lengths(1:temp_nprocs,1:maxsegcount) = &
   394)       temp_seg_lengths(1:temp_nprocs,1:maxsegcount)
   395)    OutRearranger%RecvRouter%pe_list(1:temp_nprocs) = &
   396)       temp_pe_list(1:temp_nprocs)
   397)    OutRearranger%RecvRouter%num_segs(1:temp_nprocs) = &
   398)       temp_numsegs(1:temp_nprocs)
   399)    OutRearranger%RecvRouter%locsize(1:temp_nprocs) = &
   400)       temp_locsize(1:temp_nprocs)
   401)    OutRearranger%RecvRouter%maxsize = temp_maxsize
   402) 
   403)    deallocate(temp_seg_starts,temp_seg_lengths,temp_pe_list, &
   404)               temp_numsegs,temp_locsize,stat=ier)
   405)    if(ier/=0) call die(myname_,'deallocate(temp_seg_starts...)',ier)
   406)    
   407)    endif

   415) ! !IROUTINE: clean_ - Clean a Rearranger
   416) !
   417) ! !DESCRIPTION:

   419) ! input/output {\tt Rearranger} argument {\tt ReArr}.  The success

   423) ! !INTERFACE:

   425)  subroutine clean_(ReArr, status)
   426) 
   427) !
   428) ! !USES:
   429) !
   430)    use m_Router,only : Router     
   431)    use m_Router,only : Router_clean => clean
   432)    use m_mpif90
   433)    use m_die

   441) 
   442) ! !OUTPUT PARAMETERS:

   444)    integer, optional,   intent(out)             :: status

   446) ! !REVISION HISTORY:
   447) ! 31Jan02 - E.T. Ong <eong@mcs.anl.gov> - initial prototype

   469)    endif

   473)       if(present(status)) then
   474) 	 status = ier

   477)          write(stderr,'(2a,i8)') myname_, &

   479)       endif
   480)    endif
   481) 
   482)        ! Clean up Local on-PE copy buffer:
   483)

   485)       deallocate(ReArr%LocalPack, stat=ier)
   486)       if(ier /= 0) then
   487) 	 if(present(status)) then
   488) 	    status=ier
   489) 	 else
   490) 	    write(stderr,'(2a,i8)') myname_, &
   491) 	      ':: ERROR--deallocate(ReArr%LocalPack) failed with stat=',ier
   492) 	 endif
   493)       endif
   494)    endif
   495) 
   496)  end subroutine clean_

   502) ! !IROUTINE: rearrange_ - Rearrange data between two Attribute Vectors
   503) !
   504) ! !DESCRIPTION: 
   505) ! This subroutine will take data in the {\tt SourceAv} Attribute
   506) ! Vector and rearrange it to match the GlobalSegMap used to define
   507) ! the {\tt TargetAv} Attribute Vector using the Rearrnger
   508) ! {\tt InRearranger}.
   509) !
   510) ! The optional argument {\tt Tag} can be used to set the tag value used in 
   511) ! the rearrangement.  DefaultTag will be used otherwise.
   512) !
   513) ! If the optional argument {\tt Sum} is present and true, data for the same
   514) ! physical point coming from two or more processes will be summed.
   515) ! Otherwise, data is overwritten.
   516) !
   517) ! If the optional argument {\tt Vector} is present and true,
   518) ! vector architecture-friendly parts of this routine will be invoked.
   519) !
   520) ! If the optional argument {\tt AlltoAll} is present and true,
   521) ! the communication will be done with an alltoall call instead of
   522) ! individual sends and receives.
   523) !
   524) ! The size of the {\tt SourceAv} and {\tt TargetAv}
   525) ! argument must match those stored in the {\tt InRearranger} or
   526) ! and error will result.
   527) !
   528) ! {\bf N.B.:} {\tt SourceAv} and {\tt TargetAv} are
   529) ! assumed to have exactly the same attributes
   530) ! in exactly the same order.
   531) !
   532) ! !INTERFACE:
   533) 
   534)  subroutine rearrange_(SourceAVin,TargetAV,InRearranger,Tag,Sum,Vector,AlltoAll)
   535) 
   536) !
   537) ! !USES:
   538) !
   539) 
   540)    use m_MCTWorld,only :MCTWorld
   541)    use m_MCTWorld,only :ThisMCTWorld
   542)    use m_AttrVect,  only : AttrVect
   543)    use m_AttrVect,  only : AttrVect_init => init
   544)    use m_AttrVect,  only : AttrVect_lsize => lsize
   545)    use m_AttrVect,  only : AttrVect_copy => copy
   546)    use m_AttrVect,  only : AttrVect_clean => clean
   547)    use m_AttrVect,  only : AttrVect_zero => zero
   548)    use m_AttrVect,  only : nIAttr,nRAttr
   549)    use m_AttrVect,  only : Permute,Unpermute
   550)    use m_Router,    only : Router     
   551)    use m_realkinds, only : FP
   552)    use m_mpif90
   553)    use m_die
   554)    use m_stdio
   555) 
   556)    implicit none
   557)   
   558) ! !INPUT/OUTPUT PARAMETERS:
   559) !
   560)    type(AttrVect),             intent(inout)   :: TargetAV
   561)    
   562) ! !INPUT PARAMETERS:
   563) !
   564)    type(AttrVect),   target,   intent(in)      :: SourceAVin
   565)    type(Rearranger), target,   intent(in)      :: InRearranger
   566)    integer,          optional, intent(in)      :: Tag
   567)    logical,          optional, intent(in)      :: Sum
   568)    logical,          optional, intent(in)      :: Vector
   569)    logical,          optional, intent(in)      :: AlltoAll
   570) 
   571) ! !REVISION HISTORY:
   572) ! 31Jan02 - E.T. Ong <eong@mcs.anl.gov> - initial prototype
   573) ! 20Mar02 - E.T. Ong <eong@mcs.anl.gov> - working code
   574) ! 08Jul02 - E.T. Ong <eong@mcs.anl.gov> - change intent of Target,Source
   575) ! 29Oct03 - R. Jacob <jacob@mcs.anl.gov> - add optional argument vector
   576) !           to control use of vector-friendly mods provided by Fujitsu.
   577) ! 30Mar06 - P. Worley <worleyph@ornl.gov> - added alltoall option and
   578) !           reordered send/receive order to improve communication 
   579) !           performance.  Also remove replace allocated arrays with
   580) !           automatic.
   581) ! 14Oct06 - R. Jacob <jacob@mcs.anl.gov> - check value of Sum argument.
   582) ! 25Jan08 - R. Jacob <jacob@mcs.anl.gov> - Permute/unpermute if the internal
   583) !           routers permarr is defined.
   584) !EOP ___________________________________________________________________
   585) 
   586)   character(len=*),parameter :: myname_=myname//'::Rearrange_'
   587)   integer ::	numi,numr,i,j,k,ier
   588)   integer ::    VectIndex,AttrIndex,seg_start,seg_end
   589)   integer ::    localindex,SrcVectIndex,TrgVectIndex,IAttrIndex,RAttrIndex
   590)   integer ::    proc,numprocs,nseg,pe,pe_shift,max_pe,myPid
   591)   integer ::    mp_Type_rp
   592)   integer ::    mytag
   593)   integer ::    ISendSize, RSendSize, IRecvSize, RRecvSize
   594)   logical ::    usevector, usealltoall
   595)   logical ::    DoSum
   596)   logical ::    Sendunordered
   597)   logical ::    Recvunordered
   598)   real(FP) ::  realtyp
   599) !-----------------------------------------------------------------------
   600) 
   601)    ! DECLARE STRUCTURES FOR MPI ARGUMENTS.
   602) 
   603)    ! declare arrays mapping from all processes to those sending to
   604)    ! or receiving from
   605)    integer :: SendList(0:max_nprocs-1)
   606)    integer :: RecvList(0:max_nprocs-1)
   607) 
   608)    ! declare arrays to hold count and locations where data is to be sent from
   609)    integer :: ISendLoc(max_nprocs)
   610)    integer :: RSendLoc(max_nprocs)
   611) 
   612)    integer :: ISendCnts(0:max_nprocs-1)
   613)    integer :: RSendCnts(0:max_nprocs-1)
   614) 
   615)    integer :: ISdispls(0:max_nprocs-1)
   616)    integer :: RSdispls(0:max_nprocs-1)
   617) 
   618)    ! declare arrays to hold data to be sent
   619)    integer,dimension(:),allocatable  :: ISendBuf
   620)    real(FP),dimension(:),allocatable :: RSendBuf
   621) 
   622)    ! declare arrays to hold count and locations where data is to be received into
   623)    integer :: IRecvLoc(max_nprocs)
   624)    integer :: RRecvLoc(max_nprocs)
   625) 
   626)    integer :: IRecvCnts(0:max_nprocs-1)
   627)    integer :: RRecvCnts(0:max_nprocs-1)
   628) 
   629)    integer :: IRdispls(0:max_nprocs-1)
   630)    integer :: RRdispls(0:max_nprocs-1)
   631) 
   632)    ! declare arrays to hold data to be received
   633)    integer,dimension(:),allocatable  :: IRecvBuf
   634)    real(FP),dimension(:),allocatable :: RRecvBuf
   635) 
   636)    ! Structure to hold MPI request information for sends
   637)    integer :: send_ireqs(max_nprocs)
   638)    integer :: send_rreqs(max_nprocs)
   639) 
   640)    ! Structure to hold MPI request information for sends
   641)    integer :: recv_ireqs(max_nprocs)
   642)    integer :: recv_rreqs(max_nprocs)
   643) 
   644)    ! Structure to hold MPI status information for sends 
   645)    integer :: send_istatus(MP_STATUS_SIZE,max_nprocs)
   646)    integer :: send_rstatus(MP_STATUS_SIZE,max_nprocs)
   647) 
   648)    ! Structure to hold MPI status information for sends 
   649)    integer :: recv_istatus(MP_STATUS_SIZE,max_nprocs)
   650)    integer :: recv_rstatus(MP_STATUS_SIZE,max_nprocs)
   651) 
   652)    ! Pointer structure to make Router access simpler
   653)    type(Router), pointer :: SendRout, RecvRout
   654)    type(AttrVect),pointer :: SourceAv
   655)    type(AttrVect),target  :: SourceAvtmp
   656) 
   657) !::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
   658) 
   659)    Sendunordered=associated(InRearranger%SendRouter%permarr)
   660)    Recvunordered=associated(InRearranger%RecvRouter%permarr)
   661) 
   662)    if(Sendunordered) then 
   663)       call AttrVect_init(SourceAvtmp,SourceAvin,AttrVect_lsize(SourceAvin))    
   664)       call AttrVect_copy(SourceAvin, SourceAvtmp)
   665)       call Permute(SourceAvtmp,InRearranger%SendRouter%permarr)
   666)       SourceAv => SourceAvtmp
   667)    else
   668)       SourceAv => SourceAvin
   669)    endif
   670)    
   671)    if(Recvunordered) call Permute(TargetAv,InRearranger%RecvRouter%permarr)
   672) 
   673)    ! CHECK ARGUMENTS 
   674) 
   675)    ! Check the size of the Source AttrVect
   676)    if(InRearranger%SendRouter%lAvsize /= AttrVect_lsize(SourceAV)) then
   677)       call warn(myname_,"SourceAV size is not appropriate for this Rearranger")
   678)       call die(myname_,"InRearranger%SendRouter%lAvsize",InRearranger%SendRouter%lAvsize, &
   679) 	        "AttrVect_lsize(SourceAV)", AttrVect_lsize(SourceAV))
   680)    endif
   681) 
   682)    ! Check the size of the Target AttrVect
   683)    if(InRearranger%RecvRouter%lAvsize /= AttrVect_lsize(TargetAV)) then
   684)       call warn(myname_,"TargetAV size is not appropriate for this Rearranger")
   685)       call die(myname_,"InRearranger%RecvRouter%lAvsize",InRearranger%RecvRouter%lAvsize, &
   686) 	        "AttrVect_lsize(TargetAV)", AttrVect_lsize(TargetAV))
   687)    endif
   688) 
   689)    ! Check the number of integer attributes 
   690)    if(nIAttr(SourceAV) /= nIAttr(TargetAV)) then
   691)       call warn(myname_, &
   692)                 "Number of attributes in SourceAV and TargetAV do not match")
   693)       call die(myname_,"nIAttr(SourceAV)", nIAttr(SourceAV), &
   694)                         "nIAttr(TargetAV)", nIAttr(TargetAV))
   695)    endif
   696) 
   697)    ! Check the number of real attributes
   698)    if(nRAttr(SourceAV) /= nRAttr(TargetAV)) then
   699)       call warn(myname_, &
   700)       "Number of attributes in SourceAV and TargetAV do not match")
   701)       call die(myname_,"nRAttr(SourceAV)", nRAttr(SourceAV), &
   702)                         "nRAttr(TargetAV)", nRAttr(TargetAV))
   703)    endif
   704) 
   705)    usevector=.false.
   706)    if(present(Vector)) then
   707)     if(Vector) usevector=.true.
   708)    endif
   709) 
   710)    usealltoall=.false.
   711)    if(present(Alltoall)) then
   712)     if(Alltoall) usealltoall=.true.
   713)    endif
   714) 
   715)    DoSum=.false.
   716)    if(present(Sum)) then
   717)     if(Sum) DoSum=.true.
   718)    endif
   719) 
   720)    ! ASSIGN VARIABLES
   721) 
   722) 
   723)    ! Get the number of integer and real attributes
   724)    numi = nIAttr(SourceAV)
   725)    numr = nRAttr(SourceAV)
   726) 
   727)    ! Assign the pointers
   728)    nullify(SendRout,RecvRout)
   729)    SendRout => InRearranger%SendRouter
   730)    RecvRout => InRearranger%RecvRouter
   731) 
   732)    mp_Type_rp=MP_Type(realtyp)
   733) 
   734) !::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
   735) 
   736)   ! ALLOCATE DATA STRUCTURES !
   737) 
   738)   ! IF SENDING DATA
   739)   if(SendRout%nprocs > 0) then
   740) 
   741)      ! IF SENDING INTEGER DATA
   742)      if(numi .ge. 1) then
   743) 
   744) 	! allocate buffer to hold all outgoing data
   745)         ISendSize = 1
   746) 	do proc=1,SendRout%nprocs
   747)            ISendLoc(proc) = ISendSize
   748)            ISendSize = ISendSize + SendRout%locsize(proc)*numi
   749) 	enddo
   750)         ISendSize = ISendSize - 1        
   751) 	allocate(ISendBuf(ISendSize),stat=ier)
   752) 	if(ier/=0) call die(myname_,'allocate(ISendBuf)',ier)
   753) 
   754)      endif
   755) 
   756)      ! IF SENDING REAL DATA
   757)      if(numr .ge. 1) then
   758) 
   759) 	! allocate buffer to hold all outgoing data
   760)         RSendSize = 1
   761) 	do proc=1,SendRout%nprocs
   762)            RSendLoc(proc) = RSendSize
   763)            RSendSize = RSendSize + SendRout%locsize(proc)*numr
   764) 	enddo
   765)         RSendSize = RSendSize - 1        
   766) 	allocate(RSendBuf(RSendSize),stat=ier)
   767) 	if(ier/=0) call die(myname_,'allocate(RSendBuf)',ier)
   768) 
   769) 
   770)      endif
   771) 
   772)   endif
   773) 
   774)   ! IF RECEVING DATA
   775)   if(RecvRout%nprocs > 0) then
   776) 
   777)      ! IF RECEIVING INTEGER DATA
   778)      if(numi .ge. 1) then
   779) 
   780) 	! allocate buffer to hold all outgoing data
   781)         IRecvSize = 1
   782) 	do proc=1,RecvRout%nprocs
   783)            IRecvLoc(proc) = IRecvSize
   784)            IRecvSize = IRecvSize + RecvRout%locsize(proc)*numi
   785) 	enddo
   786)         IRecvSize = IRecvSize - 1        
   787) 	allocate(IRecvBuf(IRecvSize),stat=ier)
   788) 	if(ier/=0) call die(myname_,'allocate(IRecvBuf)',ier)
   789) 
   790)      endif
   791) 
   792)      ! IF RECEIVING REAL DATA
   793)      if(numr .ge. 1) then
   794) 
   795) 	! allocate buffer to hold all outgoing data
   796)         RRecvSize = 1
   797) 	do proc=1,RecvRout%nprocs
   798)            RRecvLoc(proc) = RRecvSize
   799)            RRecvSize = RRecvSize + RecvRout%locsize(proc)*numr
   800) 	enddo
   801)         RRecvSize = RRecvSize - 1        
   802) 	allocate(RRecvBuf(RRecvSize),stat=ier)
   803) 	if(ier/=0) call die(myname_,'allocate(RRecvBuf)',ier)
   804) 
   805) 
   806)      endif
   807) 
   808)   endif
   809) 
   810) !::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
   811) 
   812)   ! INVERT PE LIST !
   813)   call MP_comm_rank(ThisMCTWorld%MCT_comm,myPid,ier)
   814)   if(ier/=0) call MP_perr_die(myname_,'MP_comm_rank',ier)
   815) 
   816)   call MP_comm_size(ThisMCTWorld%MCT_comm, max_pe, ier)
   817)   if(ier/=0) call MP_perr_die(myname_,'MP_comm_size',ier)
   818) 
   819)   SendList(:) = -1
   820)   do proc = 1,SendRout%nprocs
   821)      SendList(SendRout%pe_list(proc)) = proc
   822)   enddo
   823) 
   824)   RecvList(:) = -1
   825)   do proc = 1,RecvRout%nprocs
   826)      RecvList(RecvRout%pe_list(proc)) = proc
   827)   enddo
   828) 
   829)   if (usealltoall) then
   830)      ! CONSTRUCT CNTS AND DISPLS FOR ALLTOALLV !
   831)      ISendCnts(:) = 0
   832)      ISdispls(:)  = 0
   833)      RSendCnts(:) = 0
   834)      RSdispls(:)  = 0
   835)      IRecvCnts(:) = 0
   836)      IRdispls(:)  = 0
   837)      RRecvCnts(:) = 0
   838)      RRdispls(:)  = 0
   839)      do pe = 0,max_pe-1
   840)         proc = SendList(pe)
   841)         if (proc .ne. -1) then
   842)            ISendCnts(pe) = SendRout%locsize(proc)*numi
   843)            ISdispls(pe)  = ISendLoc(proc) - 1
   844) 
   845)            RSendCnts(pe) = SendRout%locsize(proc)*numr
   846)            RSdispls(pe)  = RSendLoc(proc) - 1
   847)         endif
   848) 
   849)         proc = RecvList(pe)
   850)         if (proc .ne. -1) then
   851)            IRecvCnts(pe) = RecvRout%locsize(proc)*numi
   852)            IRdispls(pe)  = IRecvLoc(proc) - 1
   853) 
   854)            RRecvCnts(pe) = RecvRout%locsize(proc)*numr
   855)            RRdispls(pe)  = RRecvLoc(proc) - 1
   856)         endif
   857)      enddo
   858)   endif
   859)   
   860) !::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
   861) if (usealltoall) then
   862) 
   863)   ! Load data going to each processor
   864)   do proc = 1,SendRout%nprocs
   865)      j=0
   866)      k=0
   867) 
   868)      ! load the correct pieces of the integer and real vectors
   869)      do nseg = 1,SendRout%num_segs(proc)
   870)         seg_start = SendRout%seg_starts(proc,nseg)
   871)         seg_end = seg_start + SendRout%seg_lengths(proc,nseg)-1
   872)         do VectIndex = seg_start,seg_end
   873)            do AttrIndex = 1,numi
   874)               ISendBuf(ISendLoc(proc)+j) = SourceAV%iAttr(AttrIndex,VectIndex)
   875)               j=j+1
   876)            enddo
   877)            do AttrIndex = 1,numr
   878)               RSendBuf(RSendLoc(proc)+k) = SourceAV%rAttr(AttrIndex,VectIndex)
   879)               k=k+1
   880)            enddo
   881)         enddo
   882)      enddo
   883)   enddo
   884) 
   885) else
   886)   ! POST MPI_IRECV
   887) 
   888)   ! Load data coming from each processor
   889)   do pe_shift = 1,max_pe
   890)    proc = RecvList(mod(myPid+pe_shift,max_pe))
   891)     if (proc .ne. -1) then
   892)     
   893)      ! receive the integer data
   894)      if(numi .ge. 1) then
   895) 
   896)         ! set tag
   897)         mytag = DefaultTag
   898)         if(present(Tag)) mytag=Tag
   899) 
   900) 	if( (RecvRout%num_segs(proc) > 1) .or. DoSum ) then
   901) 
   902) 	   call MPI_IRECV(IRecvBuf(IRecvLoc(proc)),                 &
   903) 		          RecvRout%locsize(proc)*numi,MP_INTEGER,   &
   904) 			  RecvRout%pe_list(proc),mytag,             &
   905) 			  ThisMCTWorld%MCT_comm,recv_ireqs(proc),ier)
   906) 
   907) 	else
   908) 
   909) 	   call MPI_IRECV(TargetAV%iAttr(1,RecvRout%seg_starts(proc,1)), &
   910) 		          RecvRout%locsize(proc)*numi,MP_INTEGER,        &
   911)                           RecvRout%pe_list(proc),mytag,                  &
   912)                           ThisMCTWorld%MCT_comm,recv_ireqs(proc),ier)
   913) 
   914) 	endif
   915) 
   916) 	if(ier /= 0) call MP_perr_die(myname_,'MPI_IRECV(ints)',ier)
   917) 
   918)      endif
   919) 
   920)      ! receive the real data
   921)      if(numr .ge. 1) then
   922) 
   923)         ! set tag
   924)         mytag = DefaultTag + 1
   925)         if(present(Tag)) mytag=Tag +1
   926) 
   927) 	if( (RecvRout%num_segs(proc) > 1) .or. DoSum ) then
   928) 
   929) 	   call MPI_IRECV(RRecvBuf(RRecvLoc(proc)),                 &
   930) 		          RecvRout%locsize(proc)*numr,mp_Type_rp,   &
   931) 			  RecvRout%pe_list(proc),mytag,             &
   932) 			  ThisMCTWorld%MCT_comm,recv_rreqs(proc),ier)
   933) 
   934) 	else
   935) 
   936) 	   call MPI_IRECV(TargetAV%rAttr(1,RecvRout%seg_starts(proc,1)), &
   937) 		          RecvRout%locsize(proc)*numr,mp_Type_rp,        &
   938) 			  RecvRout%pe_list(proc),mytag,                  &
   939) 			  ThisMCTWorld%MCT_comm,recv_rreqs(proc),ier)
   940) 
   941) 	endif
   942) 
   943) 	if(ier /= 0) call MP_perr_die(myname_,'MPI_IRECV(reals)',ier)
   944) 
   945)      endif
   946)     endif
   947)   enddo
   948) 
   949) !::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
   950) 
   951)   ! POST MPI_ISEND
   952) 
   953)   ! Load data going to each processor
   954)   do pe_shift = max_pe,1,-1
   955)    proc = SendList(mod(myPid+pe_shift,max_pe))
   956)     if (proc .ne. -1) then
   957)     
   958)      if( SendRout%num_segs(proc) > 1 ) then
   959) 
   960) 	j=0
   961) 	k=0
   962) 
   963) 	! load the correct pieces of the integer and real vectors
   964) 	do nseg = 1,SendRout%num_segs(proc)
   965) 	   seg_start = SendRout%seg_starts(proc,nseg)
   966) 	   seg_end = seg_start + SendRout%seg_lengths(proc,nseg)-1
   967) 	   do VectIndex = seg_start,seg_end
   968) 	      do AttrIndex = 1,numi
   969) 		 ISendBuf(ISendLoc(proc)+j) = SourceAV%iAttr(AttrIndex,VectIndex)
   970) 		 j=j+1
   971) 	      enddo
   972) 	      do AttrIndex = 1,numr
   973) 		 RSendBuf(RSendLoc(proc)+k) = SourceAV%rAttr(AttrIndex,VectIndex)
   974) 		 k=k+1
   975) 	      enddo
   976) 	   enddo
   977) 	enddo
   978) 
   979)      endif
   980) 
   981)      ! send the integer data
   982)      if(numi .ge. 1) then
   983) 
   984)         ! set tag
   985)         mytag = DefaultTag
   986)         if(present(Tag)) mytag=Tag
   987) 
   988) 	if( SendRout%num_segs(proc) > 1 ) then
   989) 
   990) 	   call MPI_ISEND(ISendBuf(ISendLoc(proc)),                 &
   991) 		          SendRout%locsize(proc)*numi,MP_INTEGER,   &
   992) 			  SendRout%pe_list(proc),mytag,             &
   993) 			  ThisMCTWorld%MCT_comm,send_ireqs(proc),ier)
   994) 
   995) 	else
   996) 
   997) 	   call MPI_ISEND(SourceAV%iAttr(1,SendRout%seg_starts(proc,1)), &
   998) 		          SendRout%locsize(proc)*numi,MP_INTEGER,        &
   999)                           SendRout%pe_list(proc),mytag,                  &
  1000)                           ThisMCTWorld%MCT_comm,send_ireqs(proc),ier)
  1001) 
  1002) 	endif
  1003) 
  1004) 	if(ier /= 0) call MP_perr_die(myname_,'MPI_ISEND(ints)',ier)
  1005) 
  1006)      endif
  1007) 
  1008)      ! send the real data
  1009)      if(numr .ge. 1) then
  1010) 
  1011)         ! set tag
  1012)         mytag = DefaultTag +1
  1013)         if(present(Tag)) mytag=Tag +1
  1014) 
  1015) 	if( SendRout%num_segs(proc) > 1 ) then
  1016) 
  1017) 	   call MPI_ISEND(RSendBuf(RSendLoc(proc)),                 &
  1018) 		          SendRout%locsize(proc)*numr,mp_Type_rp,   &
  1019) 			  SendRout%pe_list(proc),mytag,             &
  1020) 			  ThisMCTWorld%MCT_comm,send_rreqs(proc),ier)
  1021) 
  1022) 	else
  1023) 
  1024) 	   call MPI_ISEND(SourceAV%rAttr(1,SendRout%seg_starts(proc,1)), &
  1025) 		          SendRout%locsize(proc)*numr,mp_Type_rp,        &
  1026) 			  SendRout%pe_list(proc),mytag,                    &
  1027) 			  ThisMCTWorld%MCT_comm,send_rreqs(proc),ier)
  1028) 
  1029) 	endif
  1030) 
  1031) 	if(ier /= 0) call MP_perr_die(myname_,'MPI_ISEND(reals)',ier)
  1032) 
  1033)      endif
  1034)     endif
  1035)   enddo
  1036) endif  ! end of else for if(usealltoall)
  1037) !:::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
  1038) 
  1039)   ! ZERO TARGETAV WHILE WAITING FOR MESSAGES TO COMPLETE
  1040)
