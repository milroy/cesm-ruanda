   269)   real(r8), intent(in), optional :: lambda_bounds(2), min_mean_mass

   293) ! Calculate correction due to latent heat for evaporation/sublimation
   294) elemental function calc_ab(t, qv, xxl) result(ab)
   295)   real(r8), intent(in) :: t     ! Temperature
   296)   real(r8), intent(in) :: qv    ! Saturation vapor pressure
   297)   real(r8), intent(in) :: xxl   ! Latent heat
   298) 
   299)   real(r8) :: ab
   300) 
   301)   real(r8) :: dqsdt
   302) 
   303)   dqsdt = xxl*qv / (rv * t**2)
   304)   ab = 1._r8 + dqsdt*xxl/cpp
   305) 
   306) end function calc_ab

   308) ! get cloud droplet size distribution parameters
   309) elemental subroutine size_dist_param_liq(props, qcic, ncic, rho, pgam, lamc)
   310)   type(MGHydrometeorProps), intent(in) :: props
   311)   real(r8), intent(in) :: qcic
   312)   real(r8), intent(inout) :: ncic
   313)   real(r8), intent(in) :: rho
   314) 
   315)   real(r8), intent(out) :: pgam
   316)   real(r8), intent(out) :: lamc
   317) 
   318)   type(MGHydrometeorProps) :: props_loc
   319) 
   320)   if (qcic > qsmall) then
   321) 
   322)      ! Local copy of properties that can be modified.
   323)      ! (Elemental routines that operate on arrays can't modify scalar
   324)      ! arguments.)
   325)      props_loc = props
   326) 
   327)      ! Get pgam from fit to observations of martin et al. 1994
   328)      pgam = 0.0005714_r8*(ncic/1.e6_r8*rho) + 0.2714_r8
   329)      pgam = 1._r8/(pgam**2) - 1._r8
   330)      pgam = max(pgam, 2._r8)
   331)      pgam = min(pgam, 15._r8)
   332) 
   333)      ! Set coefficient for use in size_dist_param_basic.
   334)      props_loc%shape_coef = pi * props_loc%rho / 6._r8 * &
   335)           rising_factorial(pgam+1._r8, props_loc%eff_dim)
   336) 
   337)      ! Limit to between 2 and 50 microns mean size.
   338)      props_loc%lambda_bounds = (pgam+1._r8)*1._r8/[50.e-6_r8, 2.e-6_r8]
   339) 
   340)      call size_dist_param_basic(props_loc, qcic, ncic, lamc)
   341) 
   342)   else
   343)      ! pgam not calculated in this case, so set it to a value likely to
   344)      ! cause an error if it is accidentally used
   345)      ! (gamma function undefined for negative integers)
   346)      pgam = -100._r8
   347)      lamc = 0._r8
   348)   end if

   350) contains

   352)   ! Use gamma function to implement rising factorial extended to the reals.
   353)   pure function rising_factorial(x, n)
   354)     real(r8), intent(in) :: x, n
   355)     real(r8) :: rising_factorial
   356) 
   357)     rising_factorial = gamma(x+n)/gamma(x)
   358) 
   359)   end function rising_factorial
   360) 
   361) end subroutine size_dist_param_liq

   363) ! Basic routine for getting size distribution parameters.
   364) elemental subroutine size_dist_param_basic(props, qic, nic, lam, n0)
   365)   type(MGHydrometeorProps), intent(in) :: props
   366)   real(r8), intent(in) :: qic
   367)   real(r8), intent(inout) :: nic
   368) 
   369)   real(r8), intent(out) :: lam
   370)   real(r8), intent(out), optional :: n0
   371) 
   372)   if (qic > qsmall) then
   373) 
   374)      ! add upper limit to in-cloud number concentration to prevent
   375)      ! numerical error
   376)      if (limiter_is_on(props%min_mean_mass)) then
   377)         nic = min(nic, qic / props%min_mean_mass)
   378)      end if
   379) 
   380)      ! lambda = (c n/q)^(1/d)
   381)      lam = (props%shape_coef * nic/qic)**(1._r8/props%eff_dim)
   382) 
   383)      ! check for slope
   384)      ! adjust vars
   385)      if (lam < props%lambda_bounds(1)) then
   386)         lam = props%lambda_bounds(1)
   387)         nic = lam**(props%eff_dim) * qic/props%shape_coef
   388)      else if (lam > props%lambda_bounds(2)) then
   389)         lam = props%lambda_bounds(2)
   390)         nic = lam**(props%eff_dim) * qic/props%shape_coef
   391)      end if
   392) 
   393)   else
   394)      lam = 0._r8
   395)   end if
   396) 
   397)   if (present(n0)) n0 = nic * lam
   398) 
   399) end subroutine size_dist_param_basic

   401) real(r8) elemental function avg_diameter(q, n, rho_air, rho_sub)
   402)   ! Finds the average diameter of particles given their density, and
   403)   ! mass/number concentrations in the air.
   404)   real(r8), intent(in) :: q         ! mass mixing ratio
   405)   real(r8), intent(in) :: n         ! number concentration
   406)   real(r8), intent(in) :: rho_air   ! local density of the air
   407)   real(r8), intent(in) :: rho_sub   ! density of the particle substance
   408) 
   409)   avg_diameter = (pi * rho_sub * n/(q*rho_air))**(-1._r8/3._r8)
   410) 
   411) end function avg_diameter

   413) real(r8) elemental function var_coef(relvar, a)
   414)   ! Finds a coefficient for process rates based on the relative variance
   415)   ! of cloud water.
   416)   real(r8), intent(in) :: relvar
   417)   real(r8), intent(in) :: a
   418) 
   419)   var_coef = gamma(relvar + a) / (gamma(relvar) * relvar**a)
   420) 
   421) end function var_coef

   428) ! Initial ice deposition and sublimation loop.
   429) ! Run before the main loop
   430) ! This subroutine written by Peter Caldwell
   431) 
   432) elemental subroutine ice_deposition_sublimation(t, qv, qi, ni, &
   433)                                                 icldm, rho, dv,qvl, qvi, &
   434)                                                 berg, vap_dep, ice_sublim)
   435) 
   436)   !INPUT VARS:
   437)   !===============================================
   438)   real(r8), intent(in) :: t
   439)   real(r8), intent(in) :: qv
   440)   real(r8), intent(in) :: qi
   441)   real(r8), intent(in) :: ni
   442)   real(r8), intent(in) :: icldm
   443)   real(r8), intent(in) :: rho
   444)   real(r8), intent(in) :: dv
   445)   real(r8), intent(in) :: qvl
   446)   real(r8), intent(in) :: qvi
   447) 
   448)   !OUTPUT VARS:
   449)   !===============================================
   450)   real(r8), intent(out) :: vap_dep !ice deposition (cell-ave value)
   451)   real(r8), intent(out) :: ice_sublim !ice sublimation (cell-ave value)
   452)   real(r8), intent(out) :: berg !bergeron enhancement (cell-ave value)
   453) 
   454)   !INTERNAL VARS:
   455)   !===============================================
   456)   real(r8) :: ab
   457)   real(r8) :: epsi
   458)   real(r8) :: qiic
   459)   real(r8) :: niic
   460)   real(r8) :: lami
   461)   real(r8) :: n0i
   462) 
   463)   if (qi>=qsmall) then
   464) 
   465)      !GET IN-CLOUD qi, ni
   466)      !===============================================
   467)      qiic = qi/icldm
   468)      niic = ni/icldm
   469) 
   470)      !Compute linearized condensational heating correction
   471)      ab=calc_ab(t, qvi, xxls)
   472)      !Get slope and intercept of gamma distn for ice.
   473)      call size_dist_param_basic(mg_ice_props, qiic, niic, lami, n0i)
   474)      !Get depletion timescale=1/eps
   475)      epsi = 2._r8*pi*n0i*rho*Dv/(lami*lami)
   476) 
   477)      !Compute deposition/sublimation
   478)      vap_dep = epsi/ab*(qv - qvi)
   479) 
   480)      !Make this a grid-averaged quantity
   481)      vap_dep=vap_dep*icldm
   482) 
   483)      !Split into deposition or sublimation.
   484)      if (t < tmelt .and. vap_dep>0._r8) then
   485)         ice_sublim=0._r8
   486)      else
   487)      !hm, make ice_sublim negative for consistency with other evap/sub processes
   488)         ice_sublim=min(vap_dep,0._r8)
   489)         vap_dep=0._r8
   490)      end if
   491) 
   492)      !sublimation occurs @ any T. Not so for berg.
   493)      if (t < tmelt) then
   494) 
   495)         !Compute bergeron rate assuming cloud for whole step.
   496)         berg = max(epsi/ab*(qvl - qvi), 0._r8)
   497)      else !T>frz
   498)         berg=0._r8
   499)      end if !T<frz
   500) 
   501)   else !where qi<qsmall
   502)      berg=0._r8
   503)      vap_dep=0._r8
   504)      ice_sublim=0._r8
   505)   end if !qi>qsmall
   506) 
   507) end subroutine ice_deposition_sublimation

   510) ! autoconversion of cloud liquid water to rain
   511) ! formula from Khrouditnov and Kogan (2000), modified for sub-grid distribution of qc
   512) ! minimum qc of 1 x 10^-8 prevents floating point error
   513) 
   514) elemental subroutine kk2000_liq_autoconversion(microp_uniform, qcic, &
   515)      ncic, rho, relvar, prc, nprc, nprc1)
   516) 
   517)   logical, intent(in) :: microp_uniform
   518) 
   519)   real(r8), intent(in) :: qcic
   520)   real(r8), intent(in) :: ncic
   521)   real(r8), intent(in) :: rho
   522) 
   523)   real(r8), intent(in) :: relvar
   524) 
   525)   real(r8), intent(out) :: prc
   526)   real(r8), intent(out) :: nprc
   527)   real(r8), intent(out) :: nprc1
   528) 
   529)   real(r8) :: prc_coef
   530) 
   531)   ! Take variance into account, or use uniform value.
   532)   if (.not. microp_uniform) then
   533)      prc_coef = var_coef(relvar, 2.47_r8)
   534)   else
   535)      prc_coef = 1._r8
   536)   end if
   537) 
   538)   if (qcic >= icsmall) then
   539) 
   540)      ! nprc is increase in rain number conc due to autoconversion
   541)      ! nprc1 is decrease in cloud droplet conc due to autoconversion
   542) 
   543)      ! assume exponential sub-grid distribution of qc, resulting in additional
   544)      ! factor related to qcvar below
   545)      ! hm switch for sub-columns, don't include sub-grid qc
   546) 
   547)      prc = prc_coef * &

   649) ! contact freezing (-40<T<-3 C) (Young, 1974) with hooks into simulated dust
   650) !===================================================================
   651) ! dust size and number in multiple bins are read in from companion routine
   652) 
   653) pure subroutine contact_freezing (microp_uniform, t, p, rndst, nacon, &
   654)      pgam, lamc, cdist1, qcic, relvar, mnucct, nnucct)
   655) 
   656)   logical, intent(in) :: microp_uniform
   657) 
   658)   real(r8), intent(in) :: t(:)            ! Temperature
   659)   real(r8), intent(in) :: p(:)            ! Pressure
   660)   real(r8), intent(in) :: rndst(:,:)      ! Radius (for multiple dust bins)
   661)   real(r8), intent(in) :: nacon(:,:)      ! Number (for multiple dust bins)
   662) 
   663)   ! Size distribution parameters for cloud droplets
   664)   real(r8), intent(in) :: pgam(:)
   665)   real(r8), intent(in) :: lamc(:)
   666)   real(r8), intent(in) :: cdist1(:)
   667) 
   668)   ! MMR of in-cloud liquid water
   669)   real(r8), intent(in) :: qcic(:)
   670) 
   671)   ! Relative cloud water variance
   672)   real(r8), intent(in) :: relvar(:)
   673) 
   674)   ! Output tendencies
   675)   real(r8), intent(out) :: mnucct(:) ! MMR
   676)   real(r8), intent(out) :: nnucct(:) ! Number
   677) 
   678)   real(r8) :: tcnt                  ! scaled relative temperature
   679)   real(r8) :: viscosity             ! temperature-specific viscosity (kg/m/s)
   680)   real(r8) :: mfp                   ! temperature-specific mean free path (m)
   681) 
   682)   ! Dimension these according to number of dust bins, inferred from rndst size
   683)   real(r8) :: nslip(size(rndst,2))  ! slip correction factors
   684)   real(r8) :: ndfaer(size(rndst,2)) ! aerosol diffusivities (m^2/sec)
   685) 
   686)   ! Coefficients not used for subcolumns
   687)   real(r8) :: dum, dum1
   688) 
   689)   integer  :: i
   690) 
   691)   do i = 1,size(t)
   692) 
   693)      if (qcic(i) >= qsmall .and. t(i) < 269.15_r8) then
   694) 
   695)         if (.not. microp_uniform) then
   696)            dum = var_coef(relvar(i), 4._r8/3._r8)
   697)            dum1 = var_coef(relvar(i), 1._r8/3._r8)
   698)         else
   699)            dum = 1._r8
   700)            dum1 = 1._r8
   701)         endif
   702) 
   703)         tcnt=(270.16_r8-t(i))**1.3_r8
   704)         viscosity = 1.8e-5_r8*(t(i)/298.0_r8)**0.85_r8    ! Viscosity (kg/m/s)
   705)         mfp = 2.0_r8*viscosity/ &                         ! Mean free path (m)
   706)                      (p(i)*sqrt( 8.0_r8*28.96e-3_r8/(pi*8.314409_r8*t(i)) ))
   707) 
   708)         ! Note that these two are vectors.
   709)         nslip = 1.0_r8+(mfp/rndst(i,:))*(1.257_r8+(0.4_r8*exp(-(1.1_r8*rndst(i,:)/mfp))))! Slip correction factor
   710) 
   711)         ndfaer = 1.381e-23_r8*t(i)*nslip/(6._r8*pi*viscosity*rndst(i,:))  ! aerosol diffusivity (m2/s)
   712) 
   713)         mnucct(i) = dum *  &
   714)              dot_product(ndfaer,nacon(i,:)*tcnt)*pi*pi/3._r8*rhow* &
   715)              cdist1(i)*gamma(pgam(i)+5._r8)/lamc(i)**4
   716) 
   717)         nnucct(i) =  dum1 *  &
   718)              dot_product(ndfaer,nacon(i,:)*tcnt)*2._r8*pi*  &
   719)              cdist1(i)*gamma(pgam(i)+2._r8)/lamc(i)
   720) 
   721)      else
   722) 
   723)         mnucct(i)=0._r8
   724)         nnucct(i)=0._r8
   725) 
   726)      end if ! qcic > qsmall and t < 4 deg C
   727)   end do
   728) 
   729) end subroutine contact_freezing

   761) ! accretion of cloud droplets onto snow/graupel
   762) !===================================================================
   763) ! here use continuous collection equation with
   764) ! simple gravitational collection kernel
   765) ! ignore collisions between droplets/cloud ice
   766) ! since minimum size ice particle for accretion is 50 - 150 micron
   767) 
   768) elemental subroutine accrete_cloud_water_snow(t, rho, asn, uns, mu, qcic, ncic, qsic, &
   769)      pgam, lamc, lams, n0s, psacws, npsacws)
   770) 
   771)   real(r8), intent(in) :: t   ! Temperature
   772)   real(r8), intent(in) :: rho ! Density
   773)   real(r8), intent(in) :: asn ! Fallspeed parameter (snow)
   774)   real(r8), intent(in) :: uns ! Current fallspeed   (snow)
   775)   real(r8), intent(in) :: mu  ! Viscosity
   776) 
   777)   ! In-cloud liquid water
   778)   real(r8), intent(in) :: qcic ! MMR
   779)   real(r8), intent(in) :: ncic ! Number
   780) 
   781)   ! In-cloud snow
   782)   real(r8), intent(in) :: qsic ! MMR
   783) 
   784)   ! Cloud droplet size parameters
   785)   real(r8), intent(in) :: pgam
   786)   real(r8), intent(in) :: lamc
   787) 
   788)   ! Snow size parameters
   789)   real(r8), intent(in) :: lams
   790)   real(r8), intent(in) :: n0s
   791) 
   792)   ! Output tendencies
   793)   real(r8), intent(out) :: psacws  ! Mass mixing ratio
   794)   real(r8), intent(out) :: npsacws ! Number concentration
   795) 
   796)   real(r8) :: dc0 ! Provisional mean droplet size
   797)   real(r8) :: dum
   798)   real(r8) :: eci ! collection efficiency for riming of snow by droplets
   799) 
   800)   ! ignore collision of snow with droplets above freezing
   801) 
   802)   if (qsic >= qsmall .and. t <= tmelt .and. qcic >= qsmall) then
   803) 
   804)      ! put in size dependent collection efficiency
   805)      ! mean diameter of snow is area-weighted, since
   806)      ! accretion is function of crystal geometric area
   807)      ! collection efficiency is approximation based on stoke's law (Thompson et al. 2004)
   808) 
   809)      dc0 = (pgam+1._r8)/lamc
   810)      dum = dc0*dc0*uns*rhow/(9._r8*mu*(1._r8/lams))
   811)      eci = dum*dum/((dum+0.4_r8)*(dum+0.4_r8))
   812) 
   813)      eci = max(eci,0._r8)
   814)      eci = min(eci,1._r8)
   815) 
   816)      ! no impact of sub-grid distribution of qc since psacws
   817)      ! is linear in qc
   818) 
   819)      psacws = pi/4._r8*asn*qcic*rho*n0s*eci*gamma_bs_plus3 / lams**(bs+3._r8)
   820)      npsacws = pi/4._r8*asn*ncic*rho*n0s*eci*gamma_bs_plus3 / lams**(bs+3._r8)
   821)   else
   822)      psacws = 0._r8
   823)      npsacws = 0._r8
   824)   end if
   825) 
   826) end subroutine accrete_cloud_water_snow

   828) ! add secondary ice production due to accretion of droplets by snow
   829) !===================================================================
   830) ! (Hallet-Mossop process) (from Cotton et al., 1986)
   831) 
   832) elemental subroutine secondary_ice_production(t, psacws, msacwi, nsacwi)
   833)   real(r8), intent(in) :: t ! Temperature
   834) 
   835)   ! Accretion of cloud water to snow tendencies
   836)   real(r8), intent(inout) :: psacws ! MMR
   837) 
   838)   ! Output (ice) tendencies
   839)   real(r8), intent(out) :: msacwi ! MMR
   840)   real(r8), intent(out) :: nsacwi ! Number
   841) 
   842)   if((t < 270.16_r8) .and. (t >= 268.16_r8)) then
   843)      nsacwi = 3.5e8_r8*(270.16_r8-t)/2.0_r8*psacws
   844)      msacwi = min(nsacwi*mi0, psacws)
   845)   else if((t < 268.16_r8) .and. (t >= 265.16_r8)) then
   846)      nsacwi = 3.5e8_r8*(t-265.16_r8)/3.0_r8*psacws
   847)      msacwi = min(nsacwi*mi0, psacws)
   848)   else
   849)      nsacwi = 0.0_r8
   850)      msacwi = 0.0_r8
   851)   endif
   852) 
   853)   psacws = max(0.0_r8,psacws - nsacwi*mi0)

   859) ! formula from ikawa and saito, 1991, used by reisner et al., 1998
   860) 
   861) elemental subroutine accrete_rain_snow(t, rho, umr, ums, unr, uns, qric, qsic, &
   862)      lamr, n0r, lams, n0s, pracs, npracs )
   863) 
   864)   real(r8), intent(in) :: t   ! Temperature
   865)   real(r8), intent(in) :: rho ! Density
   866) 
   867)   ! Fallspeeds
   868)   ! mass-weighted
   869)   real(r8), intent(in) :: umr ! rain
   870)   real(r8), intent(in) :: ums ! snow
   871)   ! number-weighted
   872)   real(r8), intent(in) :: unr ! rain

   881)   real(r8), intent(in) :: lamr

   884)   real(r8), intent(in) :: lams
   885)   real(r8), intent(in) :: n0s

   888)   real(r8), intent(out) :: pracs  ! MMR

   907)           1._r8/(lamr**2 * lams**2)+ &

   909) 
   910)   else

   912)      npracs = 0._r8
   913)   end if
