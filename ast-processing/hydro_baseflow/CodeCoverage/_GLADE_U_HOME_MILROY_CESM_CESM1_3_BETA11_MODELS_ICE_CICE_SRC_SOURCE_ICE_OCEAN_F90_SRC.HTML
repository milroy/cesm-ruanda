    51) ! !ROUTINE: ocean_mixed_layer - compute SST and freeze/melt potential
    52) !
    53) ! !DESCRIPTION:
    54) !
    55) ! Compute the mixed layer heat balance and update the SST.
    56) ! Compute the energy available to freeze or melt ice.
    57) ! NOTE: SST changes due to fluxes through the ice are computed in
    58) !       ice_therm_vertical.
    59) !
    60) ! !REVISION HISTORY: same as module
    61) !
    62) ! !INTERFACE:
    63) !
    64)       subroutine ocean_mixed_layer (dt)
    65) !
    66) ! !USES:
    67) !
    68)       use ice_blocks
    69)       use ice_domain
    70)       use ice_state, only: aice, uvel, vvel
    71)       use ice_flux
    72)       use ice_grid, only: tmask
    73)       use ice_atmo, only: atmo_boundary_layer, atmbndy, atmo_boundary_const
    74) !
    75) ! !INPUT/OUTPUT PARAMETERS:
    76) !
    77)       real (kind=dbl_kind), intent(in) :: &
    78)          dt      ! time step
    79) !
    80) !EOP
    81) !
    82)       real (kind=dbl_kind) :: &
    83)          TsfK , & ! surface temperature (K)
    84)          swabs    ! surface absorbed shortwave heat flux (W/m^2)
    85) 
    86)       real (kind=dbl_kind), parameter :: &
    87)          frzmlt_max = c1000   ! max magnitude of frzmlt (W/m^2)
    88) 
    89)       integer (kind=int_kind) :: &
    90)          i, j           , & ! horizontal indices
    91)          ij             , & ! combined ij index
    92)          iblk           , & ! block index
    93)          ilo,ihi,jlo,jhi    ! beginning and end of physical domain
    94) 
    95)       real (kind=dbl_kind), dimension(nx_block,ny_block) :: &
    96)          delt  , & ! potential temperature difference   (K)
    97)          delq  , & ! specific humidity difference   (kg/kg)
    98)          shcoef, & ! transfer coefficient for sensible heat
    99)          lhcoef    ! transfer coefficient for latent heat
   100) 
   101)       integer (kind=int_kind), save :: &
   102)          icells    ! number of ocean cells
   103) 
   104)       integer (kind=int_kind), dimension(nx_block*ny_block), save :: &
   105)          indxi, indxj    ! compressed indices for ocean cells
   106) 
   107)       type (block) :: &
   108)          this_block           ! block information for current block
   109) 
   110)       do iblk = 1, nblocks
   111) 
   112)       !-----------------------------------------------------------------
   113)       ! Identify ocean cells.
   114)       ! Set fluxes to zero in land cells.
   115)       !-----------------------------------------------------------------
   116) 
   117)          icells = 0
   118)          do j = 1, ny_block
   119)          do i = 1, nx_block
   120)             if (tmask(i,j,iblk)) then
   121)                icells = icells + 1
   122)                indxi(icells) = i
   123)                indxj(icells) = j
   124)             else
   125)                sst       (i,j,iblk) = c0
   126)                frzmlt    (i,j,iblk) = c0
   127)                flwout_ocn(i,j,iblk) = c0
   128)                fsens_ocn (i,j,iblk) = c0
   129)                flat_ocn  (i,j,iblk) = c0
   130)                evap_ocn  (i,j,iblk) = c0
   131)             endif
   132)          enddo                  ! i
   133)          enddo                  ! j
   134) 
   135)       !-----------------------------------------------------------------
   136)       ! Compute boundary layer quantities
   137)       !-----------------------------------------------------------------
   138) 
   139)          if (trim(atmbndy) == 'constant') then
   140)             call atmo_boundary_const (nx_block,  ny_block,   &
   141)                                       'ice',     icells,     &
   142)                                       indxi,     indxj,      &
   143)                                       uatm       (:,:,iblk), &   
   144)                                       vatm       (:,:,iblk), &   
   145)                                       wind       (:,:,iblk), &   
   146)                                       rhoa       (:,:,iblk), &
   147)                                       strairx_ocn(:,:,iblk), & 
   148)                                       strairy_ocn(:,:,iblk), & 
   149)                                       lhcoef     (:,:),      &
   150)                                       shcoef     (:,:) )
   151)          else ! default
   152)             call atmo_boundary_layer (nx_block,  ny_block,   &
   153)                                       'ocn',     icells,     &
   154)                                       indxi,     indxj,      &
   155)                                       sst        (:,:,iblk), &    
   156)                                       potT       (:,:,iblk), &
   157)                                       uatm       (:,:,iblk), &   
   158)                                       vatm       (:,:,iblk), &   
   159)                                       uvel       (:,:,iblk), &   
   160)                                       vvel       (:,:,iblk), &   
   161)                                       wind       (:,:,iblk), &   
   162)                                       zlvl       (:,:,iblk), &   
   163)                                       Qa         (:,:,iblk), &     
   164)                                       rhoa       (:,:,iblk), &
   165)                                       strairx_ocn(:,:,iblk), & 
   166)                                       strairy_ocn(:,:,iblk), & 
   167)                                       Uref_ocn   (:,:,iblk), &
   168)                                       Tref_ocn   (:,:,iblk), & 
   169)                                       Qref_ocn   (:,:,iblk), & 
   170)                                       delt       (:,:),      &    
   171)                                       delq       (:,:),      &
   172)                                       lhcoef     (:,:),      &
   173)                                       shcoef     (:,:) )
   174)          endif
   175) 
   176)       !-----------------------------------------------------------------
   177)       ! Ocean  albedo
   178)       ! For now, assume albedo = albocn in each spectral band.
   179)       !-----------------------------------------------------------------
   180) 
   181)          alvdr_ocn(:,:,iblk) = albocn
   182)          alidr_ocn(:,:,iblk) = albocn
   183)          alvdf_ocn(:,:,iblk) = albocn
   184)          alidf_ocn(:,:,iblk) = albocn
   185) 
   186)       !-----------------------------------------------------------------
   187)       ! Compute ocean fluxes and update SST
   188)       !-----------------------------------------------------------------
   189) 
   190) !DIR$ CONCURRENT !Cray
   191) !cdir nodep      !NEC
   192) !ocl novrec      !Fujitsu
   193)       do ij = 1, icells
   194)          i = indxi(ij)
   195)          j = indxj(ij)
   196) 
   197)          ! shortwave radiative flux
   198)          swabs = (c1-alvdr_ocn(i,j,iblk)) * swvdr(i,j,iblk) &
   199)                + (c1-alidr_ocn(i,j,iblk)) * swidr(i,j,iblk) &
   200)                + (c1-alvdf_ocn(i,j,iblk)) * swvdf(i,j,iblk) &
   201)                + (c1-alidf_ocn(i,j,iblk)) * swidf(i,j,iblk) 
   202) 
   203)          ! ocean surface temperature in Kelvin
   204)          TsfK = sst(i,j,iblk) + Tffresh
   205) 
   206)          ! longwave radiative flux
   207)          flwout_ocn(i,j,iblk) = -stefan_boltzmann * TsfK**4
   208) 
   209)          ! downward latent and sensible heat fluxes
   210)          fsens_ocn(i,j,iblk) =  shcoef(i,j) * delt(i,j)
   211)          flat_ocn (i,j,iblk) =  lhcoef(i,j) * delq(i,j)
   212)          evap_ocn (i,j,iblk) = -flat_ocn(i,j,iblk) / Lvap
   213) 
   214)          ! Compute sst change due to exchange with atm/ice above
   215)          sst(i,j,iblk) = sst(i,j,iblk) + dt * ( &
   216)               (fsens_ocn(i,j,iblk) + flat_ocn(i,j,iblk) + flwout_ocn(i,j,iblk) &
   217)              + flw(i,j,iblk) + swabs) * (c1-aice(i,j,iblk)) &
   218)              + fhocn(i,j,iblk) + fswthru(i,j,iblk))         &  ! these are *aice
   219)              / (cprho*hmix(i,j,iblk))
   220) 
   221)          ! adjust qdp if cooling of mixed layer would occur when sst <= Tf
   222)          if (sst(i,j,iblk) <= Tf(i,j,iblk) .and. qdp(i,j,iblk) > c0) qdp(i,j,iblk) = c0
   223) 
   224)          ! computed T change due to exchange with deep layers:
   225)          sst(i,j,iblk) = sst(i,j,iblk) - qdp(i,j,iblk)*dt/(cprho*hmix(i,j,iblk))
   226) 
   227)          ! compute potential to freeze or melt ice
   228)          frzmlt(i,j,iblk) = (Tf(i,j,iblk)-sst(i,j,iblk))*cprho*hmix(i,j,iblk)/dt
   229)          frzmlt(i,j,iblk) = min(max(frzmlt(i,j,iblk),-frzmlt_max),frzmlt_max)
   230) 
   231)          ! if sst is below freezing, reset sst to Tf
   232)          if (sst(i,j,iblk) <= Tf(i,j,iblk)) sst(i,j,iblk) = Tf(i,j,iblk)
   233) 
   234)       enddo                     ! ij
   235)       enddo                     ! iblk
   236) 
   237)       end subroutine ocean_mixed_layer
