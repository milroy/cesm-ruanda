   321) ! Reads in the ETF data for the current date.

   323)   subroutine solar_data_advance( )
   324) 
   325)     use physconst,   only : cday
   326) 
   327)     integer  :: year, month, day, sec
   328)     integer  :: index, i, nt
   329)     integer  :: offset(2), count(2)
   330)     logical  :: do_adv, read_data
   331)     real(r8) :: time, delt
   332)     real(r8) :: data(nbins)
   333)     integer  :: ierr
   334) 
   335)     if ( fixed_scon ) return
   336)     if ( fixed_solar .and. initialized ) return
   337) 
   338)     index = -1
   339)     call get_model_time( time, year=year, month=month, day=day, seconds=sec )
   340)     read_data = time > data_times(last_index) .or. .not.initialized
   341) 
   342)     if ( read_data ) then
   343) 
   344)        find_ndx: do i = last_index, ntimes
   345)           if ( data_times(i) > time ) then
   346)              index = i-1
   347)              exit find_ndx
   348)           endif
   349)        enddo find_ndx
   350) 
   351)        last_index = index+1
   352) 
   353)        nt = 2
   354) 
   355)        if ( index < 1 ) then
   356)           write(iulog,102) year,month,day,sec
   357)           call endrun('solar_data_advance: failed to read data from '//trim(solar_data_file))
   358)        endif
   359) 
   360)        ! get the surrounding time slices
   361)        offset = (/ 1, index /)
   362)        count =  (/ nbins, nt /)
   363) 
   364)    
   365)        if (has_spectrum) then
   366)           ierr = pio_get_var( file_id, ssi_vid, offset, count, irradi )
   367)        endif
   368)        if (has_tsi .and. (.not.do_spctrl_scaling)) then
   369)           ierr = pio_get_var( file_id, tsi_vid, (/index/), (/nt/), itsi )
   370)           if ( any(itsi(:nt) < 0._r8) ) then
   371)              call endrun( 'solar_data_advance: invalid or missing tsi data  ' )
   372)           endif
   373)        endif
   374)     else
   375)        index = last_index - 1
   376)     endif
   377) 
   378)     delt = ( time - data_times(index) ) / ( data_times(index+1) - data_times(index) )
   379) 
   380)     ! this assures that FIXED data are b4b on restarts
   381)     if ( fixed_solar ) then
   382)        delt = dble(int(delt*cday+.5_r8))/dble(cday)
   383)     endif
   384) 
   385)     if (has_spectrum) then
   386)        data(:) = irradi(:,1) + delt*( irradi(:,2) - irradi(:,1) )
   387)        do i = 1,nbins
   388)           sol_irrad(i) = data(i)*irrad_fac(i) ! W/m2/nm
   389)           sol_etf(i)   = data(i)*etf_fac(i)   ! photons/cm2/sec/nm 
   390)        enddo
   391)     endif
   392)     if (has_tsi .and. (.not.do_spctrl_scaling)) then
   393)        sol_tsi = itsi(1) + delt*( itsi(2) - itsi(1) )
   394)        if ( masterproc ) then
   395)           if (day == 1 .and. sec == 0) then
   396)              write(iulog,101) year, month, day, sec, sol_tsi
   397)           end if
   398)        endif
   399)     endif
   400) 
   401)  101 FORMAT('solar_data_advance: date, TSI : ',i4.4,'-',i2.2,'-',i2.2,'-',i5.5,',  ',f12.6)
   402)  102 FORMAT('solar_data_advance: not able to find data for : ',i4.4,'-',i2.2,'-',i2.2,'-',i5.5)
   403) 
   404)   end subroutine solar_data_advance
