    34)     synoz_ndx = get_extfrc_ndx( 'SYNOZ' )
    35)     xno_ndx   = get_extfrc_ndx( 'XNO' )
    36)     aoa_nh_ndx = get_extfrc_ndx('AOA_NH')

    39)     o2p_ndx  = get_extfrc_ndx( 'O2p' )
    40)     np_ndx   = get_extfrc_ndx( 'Np' )
    41)     n2p_ndx  = get_extfrc_ndx( 'N2p' )
    42)     n2d_ndx  = get_extfrc_ndx( 'N2D' )

    44)     e_ndx    = get_extfrc_ndx( 'e' )
    45)     oh_ndx   = get_extfrc_ndx( 'OH' )
    46) 
    47)     has_ions = op_ndx > 0 .and. o2p_ndx > 0 .and. np_ndx > 0 .and. n2p_ndx > 0 .and. e_ndx > 0
    48) 
    49)     if (masterproc) then
    50)        write(iulog,*) ' '
    51)        write(iulog,*) 'setext_inti: diagnostics: co_ndx, no_ndx, synoz_ndx, xno_ndx'
    52)        write(iulog,'(10i5)') co_ndx, no_ndx, synoz_ndx, xno_ndx
    53)     endif
    54) 
    55)     call addfld( 'NO_Lightning','molec/cm3/s', pver, 'A', 'lightning NO source', phys_decomp )
    56)     call addfld( 'NO_Aircraft', 'molec/cm3/s', pver, 'A', 'aircraft NO source',  phys_decomp )
    57)     call addfld( 'CO_Aircraft', 'molec/cm3/s', pver, 'A', 'aircraft CO source',  phys_decomp )
    58)     
    59)     if ( n2d_ndx > 0 .and. n_ndx>0 ) then
    60)        call addfld( 'N4S_SPE', 'molec/cm3/s', pver, 'I', 'solar proton event N(4S) source',phys_decomp )
    61)        call addfld( 'N2D_SPE', 'molec/cm3/s', pver, 'I', 'solar proton event N(2D) source',phys_decomp )
    62)        call addfld( 'N4S_GCR', 'molec/cm3/s', pver, 'I', 'galactic cosmic ray N(4S) source',phys_decomp )
    63)        call addfld( 'N2D_GCR', 'molec/cm3/s', pver, 'I', 'galactic cosmic ray N(2S) source',phys_decomp )
    64)     elseif ( no_ndx > 0 .and. n_ndx>0 ) then
    65)        call addfld( 'N4S_SPE', 'molec/cm3/s', pver, 'I', 'solar proton event N(4S) source',phys_decomp )
    66)        call addfld( 'NO_SPE',  'molec/cm3/s', pver, 'I', 'solar proton event NO source',phys_decomp )
    67)        call addfld( 'N4S_GCR', 'molec/cm3/s', pver, 'I', 'galactic cosmic ray N(4S) source',phys_decomp )
    68)        call addfld( 'NO_GCR',  'molec/cm3/s', pver, 'I', 'galactic cosmic ray NO source',phys_decomp )
    69)     endif
    70)     if ( oh_ndx > 0 ) then
    71)        call addfld( 'OH_SPE',  'molec/cm3/s', pver, 'I', 'solar proton event OH source',  phys_decomp )
    72)        call addfld( 'OH_GCR',  'molec/cm3/s', pver, 'I', 'galactic cosmic ray OH source',  phys_decomp )
    73)     endif
    74) 
    75)     if ( has_ions ) then
    76)        call addfld( 'P_Op',   '/s ', pver, 'I', 'production o+', phys_decomp )
    77)        call addfld( 'P_O2p',  '/s ', pver, 'I', 'production o2+', phys_decomp )
    78)        call addfld( 'P_N2p',  '/s ', pver, 'I', 'production n2+', phys_decomp )
    79)        call addfld( 'P_Np',   '/s ', pver, 'I', 'production n+', phys_decomp )
    80)        call addfld( 'P_IONS', '/s ', pver, 'I', 'total ion production', phys_decomp )
    81)     endif
    82) 
    83)     if ( aoa_nh_ndx > 0 ) then
    84)        call addfld('AOA_NH_XFRC', 'molec/cm3/s', pver, 'A',  'external forcing for AOA_NH',   phys_decomp )
    85)     endif
    86)

    89)   subroutine setext( extfrc, zint_abs, zint_rel, cldtop, &
    90)        zmid, lchnk, tfld, o2mmr, ommr, &
    91)        pmid, mbar, rlats, calday, ncol, rlons, pbuf )
    92)     !--------------------------------------------------------
    93)     !     ... for this latitude slice:
    94)     !         - form the production from datasets
    95)     !         - form the nox (xnox) production from lighing
    96)     !         - form the nox (xnox) production from airplanes
    97)     !         - form the co production from airplanes
    98)     !--------------------------------------------------------
    99) 
   100)     use cam_history,  only : outfld
   101)     use ppgrid,       only : pver, pcols
   102)     use mo_airplane,  only : airpl_set
   103)     use chem_mods,    only : extcnt
   104)     use mo_lightning, only : prod_no
   105) 
   106)     use mo_extfrc,    only : extfrc_set
   107)     use chem_mods,    only : extcnt
   108)     use tracer_srcs,  only : num_tracer_srcs, tracer_src_flds, get_srcs_data
   109)     use mo_chem_utls, only : get_extfrc_ndx
   110)     use mo_synoz,     only : po3
   111) 
   112)     use mo_aurora,      only : aurora
   113)     use mo_solarproton, only : spe_prod 
   114)     use gcr_ionization, only : gcr_ionization_noxhox
   115) 
   116)     use physics_buffer, only : physics_buffer_desc
   117) 
   118)     implicit none
   119) 
   120)     !--------------------------------------------------------
   121)     !     ... dummy arguments
   122)     !--------------------------------------------------------
   123)     !--------------------------------------------------------
   124)     integer,  intent(in)  ::   lchnk                       ! chunk id
   125)     integer,  intent(in)  ::   ncol                        ! columns in chunk
   126)     real(r8), intent(in)  ::   zint_abs(ncol,pver+1)           ! interface geopot height ( km )
   127)     real(r8), intent(in)  ::   zint_rel(ncol,pver+1)           ! interface geopot height ( km )
   128)     real(r8), intent(in)  ::   cldtop(ncol)                ! cloud top index
   129)     real(r8), intent(out) ::   extfrc(ncol,pver,extcnt)    ! the "extraneous" forcing
   130) 
   131)     real(r8), intent(in)  ::   calday                      ! calendar day of year
   132)     real(r8), intent(in)  ::   rlats(ncol)                 ! column latitudes (radians)
   133)     real(r8), intent(in)  ::   rlons(ncol)                 ! column longitudes (radians)
   134)     real(r8), intent(in)  ::   zmid(ncol,pver)             ! midpoint geopot height ( km )
   135)     real(r8), intent(in)  ::   pmid(pcols,pver)            ! midpoint pressure (Pa)
   136)     real(r8), intent(in)  ::   tfld(pcols,pver)            ! midpoint temperature (K)
   137)     real(r8), intent(in)  ::   mbar(ncol,pver)             ! mean molecular mass (g/mole)
   138)     real(r8), intent(in)  ::   o2mmr(ncol,pver)            ! o2 concentration (kg/kg)
   139)     real(r8), intent(in)  ::   ommr(ncol,pver)             ! o concentration (kg/kg)
   140) 
   141)     type(physics_buffer_desc),pointer :: pbuf(:)
   142) 
   143)     !--------------------------------------------------------
   144)     !     ... local variables
   145)     !--------------------------------------------------------
   146)     integer :: i, k, k1, kk, cldind, ii, jj, nlev
   147)     real(r8) :: srcs_offline( ncol, pver )
   148)     integer :: ndx
   149) 
   150)     real(r8), dimension(ncol,pver) :: no_lgt
   151) 
   152)     real(r8) :: spe_nox(ncol,pver)        ! Solar Proton Event NO production
   153)     real(r8) :: spe_hox(ncol,pver)        ! Solar Proton Event HOx production
   154)     real(r8) :: gcr_nox(ncol,pver)        ! Galactic comsic rays NO production
   155)     real(r8) :: gcr_hox(ncol,pver)        ! Galactic comsic rays HOx production
   156) 
   157)     real(r8), parameter :: rad2deg = 180._r8/pi                ! radians to degrees conversion factor
   158)     real(r8) :: xlat
   159) 
   160)     extfrc(:,:,:) = 0._r8
   161) 
   162)     no_lgt(:,:) = 0._r8
   163) 
   164)     !--------------------------------------------------------
   165)     !     ... set frcing from datasets
   166)     !--------------------------------------------------------
   167)     call extfrc_set( lchnk, zint_rel, extfrc, ncol )
   168)     
   169)     !--------------------------------------------------------
   170)     !     ... set nox production from lighting
   171)     !         note: from ground to cloud top production is c shaped
   172)     !--------------------------------------------------------
   173)     if ( no_ndx > 0 ) then
   174)        do i = 1,ncol
   175)           cldind = nint( cldtop(i) )
   176)           if( cldind < pver .and. cldind > 0  ) then
   177)              extfrc(i,cldind:pver,no_ndx) = extfrc(i,cldind:pver,no_ndx) &
   178)                   + prod_no(i,cldind:pver,lchnk)
   179)              no_lgt(i,cldind:pver) = prod_no(i,cldind:pver,lchnk)
   180)           end if
   181)        end do
   182)     endif
   183)     if ( xno_ndx > 0 ) then
   184)        do i = 1,ncol
   185)           cldind = nint( cldtop(i) )
   186)           if( cldind < pver .and. cldind > 0  ) then
   187)              extfrc(i,cldind:pver,xno_ndx) = extfrc(i,cldind:pver,xno_ndx) &
   188)                   + prod_no(i,cldind:pver,lchnk)
   189)           end if
   190)        end do
   191)     endif
   192) 
   193)     call outfld( 'NO_Lightning', no_lgt(:ncol,:), ncol, lchnk )
   194) 
   195)     call airpl_set( lchnk, ncol, no_ndx, co_ndx, xno_ndx, cldtop, zint_abs, extfrc)
   196) 
   197)     !---------------------------------------------------------------------
   198)     ! 	... synoz production
   199)     !---------------------------------------------------------------------
   200)     if( synoz_ndx > 0 ) then
   201)        do k = 1,pver
   202)           extfrc(:ncol,k,synoz_ndx) = extfrc(:ncol,k,synoz_ndx) + po3(:ncol,k,lchnk)
   203)        end do
   204)     end if
   205) 
   206)     do i = 1,num_tracer_srcs
   207) 
   208)        ndx =  get_extfrc_ndx( tracer_src_flds(i) )
   209)        call get_srcs_data( tracer_src_flds(i), srcs_offline,  ncol, lchnk, pbuf )
   210)        do k = 1,pver
   211)           extfrc(:ncol,k,ndx) = extfrc(:ncol,k,ndx) + srcs_offline(:ncol,k)
   212)        enddo
   213) 
   214)     enddo
   215) 
   216) !
   217) ! CCMI : external forcing for AOA_NH
   218) ! set to 100 ppb per year
   219) !
   220)     if ( aoa_nh_ndx > 0 ) then
   221)        extfrc(:ncol,:,aoa_nh_ndx) = 1.e-7_r8/(86400._r8*365._r8)
   222)        do i=1,ncol
   223)          xlat = rlats(i)*rad2deg              ! convert to degrees
   224)          if ( xlat >= 30._r8 .and. xlat <= 50._r8 ) then
   225)             extfrc(i,pver,aoa_nh_ndx) = 0._r8
   226)          end if
   227)        end do
   228)        call outfld('AOA_NH_XFRC',extfrc(:ncol,:,aoa_nh_ndx), ncol, lchnk )
   229)     end if
   230) 
   231)     if ( has_ions ) then
   232)        !---------------------------------------------------------------------
   233)        !     ... set ion auroral production
   234)        !---------------------------------------------------------------------
   235) 
   236)        call aurora( tfld, o2mmr, ommr, mbar, rlats, &
   237)             extfrc(:,:,o2p_ndx), extfrc(:,:,op_ndx), extfrc(:,:,n2p_ndx), extfrc(:,:,np_ndx), pmid, &
   238)             lchnk, calday, ncol, rlons, pbuf )
   239)        !---------------------------------------------------------------------
   240)        !     ... set n(2d) and n(4s) auroral production
   241)        !         Stan Solomon HAO
   242)        !         include production of N by secondary auroral hot electrons (e_s*):
   243)        !         this is not a "real" reaction; instead, the production is parameterized in terms
   244)        !         of the production rate of N2+ by primary electrons, QN2P (which is in the model),
   245)        !         as follows:
   246)        !---------------------------------------------------------------------
   247)        do k = 1,pver
   248)           extfrc(:,k,n2d_ndx) = 1.57_r8*.6_r8*extfrc(:,k,n2p_ndx)
   249)           extfrc(:,k,n_ndx) = 1.57_r8*.4_r8*extfrc(:,k,n2p_ndx)
   250)        end do
   251)        !---------------------------------------------------------------------
   252)        !     ... set electron auroral production
   253)        !---------------------------------------------------------------------
   254)        do k = 1,pver
   255)           extfrc(:,k,e_ndx) = extfrc(:,k,op_ndx) + extfrc(:,k,o2p_ndx) &
   256)                + extfrc(:,k,np_ndx) + extfrc(:,k,n2p_ndx)
   257)        end do
   258) 
   259)        call outfld( 'P_Op',  extfrc(:,:,op_ndx), ncol, lchnk )
   260)        call outfld( 'P_O2p', extfrc(:,:,o2p_ndx), ncol, lchnk )
   261)        call outfld( 'P_Np',  extfrc(:,:,np_ndx), ncol, lchnk )
   262)        call outfld( 'P_N2p', extfrc(:,:,n2p_ndx), ncol, lchnk )
   263)        call outfld( 'P_IONS',extfrc(:,:,n2d_ndx), ncol, lchnk )
   264) 
   265)     endif
   266) 
   267)     !---------------------------------------------------------------------
   268)     !     ... set SPE NOx and HOx production
   269)     !     Jackman et al., JGR, 2005 
   270)     !     production of 1.25 Nitrogen atoms/ion pair with branching ratios 
   271)     !     of 0.55 N(4S) and 0.7 N(2D).
   272)     !---------------------------------------------------------------------
   273)     call spe_prod( spe_nox, spe_hox, pmid, zmid, lchnk, ncol)
   274) 
   275)     !---------------------------------------------------------------------
   276)     ! NOx and HOx production due to Galactic Cosmic Ray Ionization 
   277)     !---------------------------------------------------------------------
   278)     call gcr_ionization_noxhox( ncol, lchnk, zmid, gcr_nox, gcr_hox )
   279) 
   280)     if ( n2d_ndx > 0 .and. n_ndx>0 ) then
   281)        extfrc(:ncol,:pver,n2d_ndx) = extfrc(:ncol,:pver,n2d_ndx) +  0.7_r8*(spe_nox(:ncol,:pver) + gcr_nox(:ncol,:pver))
   282)        extfrc(:ncol,:pver,  n_ndx) = extfrc(:ncol,:pver,  n_ndx) + 0.55_r8*(spe_nox(:ncol,:pver) + gcr_nox(:ncol,:pver))
   283)        call outfld( 'N2D_SPE', 0.7_r8*spe_nox, ncol, lchnk ) ! N(2D) produciton (molec/cm3/s)
   284)        call outfld( 'N4S_SPE',0.55_r8*spe_nox, ncol, lchnk ) ! N(4S) produciton (molec/cm3/s)
   285)        call outfld( 'N2D_GCR', 0.7_r8*gcr_nox, ncol, lchnk ) ! N(2D) produciton (molec/cm3/s)
   286)        call outfld( 'N4S_GCR',0.55_r8*gcr_nox, ncol, lchnk ) ! N(4S) produciton (molec/cm3/s)
   287)     elseif ( no_ndx > 0 .and. n_ndx>0 ) then
   288)        ! for mechanisms that do not include N2D the SPEs produce NO 
   289)        extfrc(:ncol,:pver, no_ndx) = extfrc(:ncol,:pver, no_ndx) +  0.7_r8*(spe_nox(:ncol,:pver) + gcr_nox(:ncol,:pver))
   290)        extfrc(:ncol,:pver,  n_ndx) = extfrc(:ncol,:pver,  n_ndx) + 0.55_r8*(spe_nox(:ncol,:pver) + gcr_nox(:ncol,:pver))
   291)        call outfld( 'NO_SPE',  0.7_r8*spe_nox, ncol, lchnk ) ! NO produciton (molec/cm3/s)
   292)        call outfld( 'N4S_SPE',0.55_r8*spe_nox, ncol, lchnk ) ! N(4S) produciton (molec/cm3/s)
   293)        call outfld( 'NO_GCR',  0.7_r8*gcr_nox, ncol, lchnk ) ! NO produciton (molec/cm3/s)
   294)        call outfld( 'N4S_GCR',0.55_r8*gcr_nox, ncol, lchnk ) ! N(4S) produciton (molec/cm3/s)
   295)     endif
   296)     if ( oh_ndx > 0 ) then
   297)        extfrc(:ncol,:pver, oh_ndx) = extfrc(:ncol,:pver, oh_ndx)         + spe_hox(:ncol,:pver) + gcr_hox(:ncol,:pver)
   298)        call outfld( 'OH_SPE' ,        spe_hox, ncol, lchnk ) ! HOX produciton (molec/cm3/s)
   299)        call outfld( 'OH_GCR' ,        gcr_hox, ncol, lchnk ) ! HOX produciton (molec/cm3/s)
   300)     endif
   301) 
   302)   end subroutine setext
