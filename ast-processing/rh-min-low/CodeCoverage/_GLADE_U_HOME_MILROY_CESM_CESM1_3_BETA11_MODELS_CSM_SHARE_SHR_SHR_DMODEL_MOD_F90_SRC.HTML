    69) subroutine shr_dmodel_gsmapCreate(gsmap,gsize,compid,mpicom,decomp)
    70) 
    71)     implicit none
    72) 
    73)     type(mct_gsMap), intent(inout) :: gsmap
    74)     integer(IN)    , intent(in)    :: gsize
    75)     integer(IN)    , intent(in)    :: compid
    76)     integer(IN)    , intent(in)    :: mpicom
    77)     character(len=*),intent(in)    :: decomp
    78) 
    79)     ! local
    80) 
    81)     integer(IN) :: n,npes,ierr
    82)     integer(IN), pointer :: start(:)     ! for gsmap initialization
    83)     integer(IN), pointer :: length(:)    ! for gsmap initialization
    84)     integer(IN), pointer :: pe_loc(:)    ! for gsmap initialization
    85)     character(*), parameter :: subname = '(shr_dmodel_gsmapCreate) '
    86)     character(*), parameter :: F00   = "('(shr_dmodel_gsmapCreate) ',8a)"
    87)     character(*), parameter :: F01   = "('(shr_dmodel_gsmapCreate) ',a,5i8)"
    88) 
    89)     ! ---------------------------------------------
    90) 
    91)     if (gsize > 0) then
    92)        call mpi_comm_size(mpicom,npes,ierr)
    93)        allocate(start(npes),length(npes),pe_loc(npes))
    94) 
    95)        start = 0
    96)        length = 0
    97)        do n = 1,npes
    98)           if (trim(decomp) == '1d') then
    99)              length(n)  = gsize/npes
   100)              if (n <= mod(gsize,npes)) length(n) = length(n) + 1
   101)           elseif (trim(decomp) == 'root') then
   102)              length = 0
   103)              length(1) = gsize
   104)           else
   105)              write(logunit,F00) ' ERROR: decomp not allowed, ',trim(decomp)
   106)              call shr_sys_abort(subname//' ERROR decomp')
   107)           endif
   108)           if (n == 1) then
   109)              start(n) = 1
   110)           else
   111)              start(n) = start(n-1) + length(n-1)
   112)           endif
   113)           pe_loc(n) = n-1
   114)        enddo
   115)        call mct_gsMap_init( gsMap, COMPID, npes, gsize, start, length, pe_loc)
   116)        deallocate(start,length,pe_loc)
   117)     endif
   118) 
   119) end subroutine shr_dmodel_gsmapCreate

   122) subroutine shr_dmodel_readgrid( gGrid, gsMap, nxgo, nygo, filename, compid, mpicom, &
   123)   decomp, lonname, latname, maskname, areaname, fracname, readfrac, &
   124)   scmmode, scmlon, scmlat)
   125) 
   126)   use seq_flds_mod, only : seq_flds_dom_coord, seq_flds_dom_other
   127)   use shr_file_mod, only : shr_file_noprefix, shr_file_queryprefix, shr_file_get
   128)   use shr_string_mod, only : shr_string_lastindex
   129)   use shr_ncread_mod, only : shr_ncread_domain, shr_ncread_vardimsizes, &
   130)                              shr_ncread_varexists, shr_ncread_vardimnum
   131)   implicit none
   132) 
   133)   !----- arguments -----
   134)   type(mct_gGrid), intent(inout) :: gGrid
   135)   type(mct_gsMap), intent(inout) :: gsMap
   136)   integer(IN)    , intent(out)   :: nxgo
   137)   integer(IN)    , intent(out)   :: nygo
   138)   character(len=*),intent(in)    :: filename
   139)   integer(IN)    , intent(in)    :: compid
   140)   integer(IN)    , intent(in)    :: mpicom
   141)   character(len=*),optional,intent(in)    ::   decomp ! decomp strategy for gsmap
   142)   character(len=*),optional,intent(in)    ::  lonname ! name of  lon variable in file
   143)   character(len=*),optional,intent(in)    ::  latname ! name of  lat variable in file
   144)   character(len=*),optional,intent(in)    :: maskname ! name of mask variable in file
   145)   character(len=*),optional,intent(in)    :: areaname ! name of area variable in file
   146)   character(len=*),optional,intent(in)    :: fracname ! name of frac variable in file
   147)   logical         ,optional,intent(in)    :: readfrac ! T <=> also read frac  in file
   148)   logical         ,optional,intent(in)    :: scmmode  ! single column mode
   149)   real(R8)        ,optional,intent(in)    :: scmlon   ! single column lon
   150)   real(R8)        ,optional,intent(in)    :: scmlat   ! single column lat
   151) 
   152)   !----- local -----
   153)   integer(IN)   :: n,k,j,i    ! indices  
   154)   integer(IN)   :: lsize      ! lsize
   155)   integer(IN)   :: gsize      ! gsize
   156)   integer(IN)   :: my_task, master_task
   157)   integer(IN)   :: ierr       ! error code
   158)   logical       :: fileexists ! 
   159)   integer(IN)   :: rCode      ! return code
   160)   character(CL) :: remoteFn   ! input file name (possibly at an archival location)
   161)   character(CL) :: localFn    ! file name to be opened (possibly a local copy)
   162)   character(CS) :: prefix     ! file prefix
   163)   character(CS) ::   ldecomp  ! decomp strategy
   164)   character(CS) ::  llatname  ! name of  lat variable
   165)   character(CS) ::  llonname  ! name of  lon variable
   166)   character(CS) :: lmaskname  ! name of mask variable
   167)   character(CS) :: lareaname  ! name of area variable
   168)   character(CS) :: lfracname  ! name of area variable
   169)   logical       :: lreadfrac  ! read fraction
   170)   logical       :: maskexists ! is mask on dataset
   171)   integer(IN)   :: nxg,nyg    ! size of input fields
   172)   integer(IN)   :: ndims      ! number of dims
   173)   integer(IN)   :: nlon,nlat,narea,nmask,nfrac
   174)   logical       :: lscmmode   ! local scm mode
   175)   real(R8)      :: dist,mind  ! scmmode point search
   176)   integer(IN)   :: ni,nj      ! scmmode point search
   177)   real(R8)      :: lscmlon    ! local copy of scmlon
   178) 
   179)   real   (R8),allocatable ::  lon(:,:) ! temp array for domain lon  info
   180)   real   (R8),allocatable ::  lat(:,:) ! temp array for domain lat  info
   181)   integer(IN),allocatable :: mask(:,:) ! temp array for domain mask info
   182)   real   (R8),allocatable :: area(:,:) ! temp array for domain area info
   183)   real   (R8),allocatable :: frac(:,:) ! temp array for domain frac info
   184) 
   185)   integer(IN), pointer :: idata(:)   ! temporary
   186)   type(mct_ggrid)      :: gGridRoot       ! global mct ggrid
   187) 
   188)   character(*), parameter :: subname = '(shr_dmodel_readgrid) '
   189)   character(*), parameter :: F00   = "('(shr_dmodel_readgrid) ',8a)"
   190)   character(*), parameter :: F01   = "('(shr_dmodel_readgrid) ',a,5i8)"
   191) 
   192) !-------------------------------------------------------------------------------
   193) ! PURPOSE:  Read MCT ggrid and set gsmap
   194) !----------------------------------------------------------------------------
   195) ! Notes:
   196) ! o as per shr_file_get(), the file name format is expected to be
   197) !   remoteFn = [location:][directory path]localFn
   198) !   eg. "foobar.nc"  "/home/user/foobar.nc"  "mss:/USER/fobar.nc"
   199) ! o assumes a very specific netCDF domain file format wrt var names, etc.
   200) !
   201) ! TO DO: have the calling routine select/input the domain's file name
   202) !----------------------------------------------------------------------------
   203) 
   204)   call MPI_COMM_RANK(mpicom,my_task,ierr)
   205)   master_task = 0
   206) 
   207)   lscmmode = .false.
   208)   if (present(scmmode)) then
   209)      lscmmode = scmmode
   210)      if (lscmmode) then
   211)         if (.not.present(scmlon) .or. .not.present(scmlat)) then
   212)            write(logunit,*) subname,' ERROR: scmmode must supply scmlon and scmlat'
   213)            call shr_sys_abort(subname//' ERROR: scmmode1 lon lat')
   214)         endif
   215)         if (my_task > 0) then
   216)            write(logunit,*) subname,' ERROR: scmmode must be run on one pe'
   217)            call shr_sys_abort(subname//' ERROR: scmmode2 tasks')
   218)         endif
   219)      endif
   220)   endif
   221) 
   222)   if (my_task == master_task) then
   223)      if ( shr_file_queryPrefix(fileName,prefix=prefix) /= shr_file_noPrefix ) then
   224)         n        = max(len_trim(prefix),shr_string_lastIndex(fileName,"/"))
   225)         remoteFn = fileName
   226)         localFn  = fileName(n+1: len_trim(fileName) )
   227)         call shr_file_get(rCode,localFn,remoteFn)
   228)      else
   229)         remoteFn = "undefined" ! this isn't needed
   230)         localFn  = fileName    ! file to open
   231)      end if
   232)      inquire(file=trim(localFn),exist=fileExists)
   233)      if (.not. fileExists) then
   234)          write(logunit,F00) "ERROR: file does not exist: ", trim(localFn)
   235)          call shr_sys_abort(subName//"ERROR: file does not exist: "//trim(localFn))
   236)      end if
   237)   endif
   238) 
   239)   lreadfrac = .false.
   240)   ldecomp   = "1d"
   241)   llonname  = "xc"  ! default values / standard data model domain file format
   242)   llatname  = "yc"
   243)   lmaskname = "mask"
   244)   lareaname = "area"
   245)   lfracname = "frac"
   246)   if (present(  decomp))   ldecomp =   decomp
   247)   if (present(readfrac)) lreadfrac = readfrac
   248)   if (present( lonname))  llonname =  lonname
   249)   if (present( latname))  llatname =  latname
   250)   if (present(maskname)) lmaskname = maskname
   251)   if (present(areaname)) lareaname = areaname
   252)   if (present(fracname)) lfracname = fracname
   253) 
   254)   ! Initialize mct domain type
   255)   ! lat/lon in degrees,  area in radians^2, mask is 1 (ocean), 0 (non-ocean)
   256) 
   257)   if (my_task == master_task) then
   258)      if (shr_ncread_varexists(localFn,lmaskname)) then
   259)         maskexists = .true.
   260)         call shr_ncread_varDimSizes(localFn,lmaskname,nxg,nyg)
   261)      else
   262)         maskexists = .false.
   263)         call shr_ncread_varDimNum(localFn,llonName,ndims)
   264)         if (ndims == 1) then
   265)            call shr_ncread_varDimSizes(localFn,llonName,nxg)
   266)            call shr_ncread_varDimSizes(localFn,llatName,nyg)
   267)         else
   268)            call shr_ncread_varDimSizes(localFn,llonName,nxg,nyg)
   269)         endif
   270)      endif
   271)   endif
   272)   call shr_mpi_bcast(nxg,mpicom)
   273)   call shr_mpi_bcast(nyg,mpicom)
   274)   if (lscmmode) then
   275)      nxgo = 1
   276)      nygo = 1
   277)      gsize = 1
   278)   else
   279)      nxgo = nxg
   280)      nygo = nyg
   281)      gsize = nxg*nyg
   282)      if (gsize < 1) return
   283)   endif
   284) 
   285)   call shr_dmodel_gsmapCreate(gsMap,gsize,compid,mpicom,trim(ldecomp))
   286)   lsize = mct_gsMap_lsize(gsMap, mpicom)
   287)   call mct_gGrid_init( GGrid=gGrid, CoordChars=trim(seq_flds_dom_coord), &
   288)      OtherChars=trim(seq_flds_dom_other), lsize=lsize )
   289) 
   290)   ! Determine global gridpoint number attribute, GlobGridNum, automatically in ggrid
   291) 
   292)   call mct_gsMap_orderedPoints(gsMap, my_task, idata)
   293)   call mct_gGrid_importIAttr(gGrid,'GlobGridNum',idata,lsize)
   294)   deallocate(idata)
   295) 
   296)   ! Initialize attribute vector with special value
   297) 
   298)   gGrid%data%rAttr = -9999.0_R8
   299) 
   300)   ! Load file data into domG then scatter to ggrid
   301) 
   302)   if (my_task == master_task) then
   303) 
   304)      allocate(lon(nxg,nyg))
   305)      allocate(lat(nxg,nyg))
   306)      allocate(area(nxg,nyg))
   307)      allocate(mask(nxg,nyg))
   308)      allocate(frac(nxg,nyg))
   309) 
   310)      if (.not.maskexists) then
   311)         call shr_ncread_domain(localFn,llonName,lon,llatName,lat)
   312)         mask = 1
   313)         frac = 1.0_R8
   314)         area = 1.0e36_R8
   315)      else
   316)         if (lreadfrac) then
   317)            call shr_ncread_domain(localFn,llonName,lon,llatName,lat, &
   318)                 lmaskName,mask,lareaName,area,lfracName,frac)
   319)         else ! assume frac = 1.0
   320)            call shr_ncread_domain(localFn,llonName,lon,llatName,lat, &
   321)                 lmaskName,mask,lareaName,area)
   322)            where (mask == 0)
   323)               frac = 0.0_R8
   324)            elsewhere
   325)               frac = 1.0_R8
   326)            end where
   327)         endif
   328)      endif
   329) 
   330)      call mct_gGrid_init(gGridRoot,gGrid,gsize)
   331) 
   332) ! initialize gGridRoot to avoid errors when using strict compiler checks
   333)      gGridRoot%data%rAttr = -9999.0_R8
   334) 
   335)      nlon  = mct_aVect_indexRA(gGridRoot%data,'lon')
   336)      nlat  = mct_aVect_indexRA(gGridRoot%data,'lat')
   337)      narea = mct_aVect_indexRA(gGridRoot%data,'area')
   338)      nmask = mct_aVect_indexRA(gGridRoot%data,'mask')
   339)      nfrac = mct_aVect_indexRA(gGridRoot%data,'frac')
   340) 
   341)      if (lscmmode) then
   342)         !--- assumes regular 2d grid for compatability with shr_scam_getCloseLatLon ---
   343)         !--- want lon values between 0 and 360, assume 1440 is enough ---
   344)         lscmlon = mod(scmlon+1440.0_r8,360.0_r8)
   345)         lon     = mod(lon   +1440.0_r8,360.0_r8)
   346) 
   347)         !--- start with wraparound ---
   348)         ni = 1
   349)         mind = abs(lscmlon - (lon(1,1)+360.0_r8))
   350)         do i=1,nxg
   351)            dist = abs(lscmlon - lon(i,1))
   352)            if (dist < mind) then
   353)               mind = dist
   354)               ni = i
   355)            endif
   356)         enddo        
   357) 
   358)         nj = -1
   359)         mind = 1.0e20
   360)         do j=1,nyg
   361)            dist = abs(scmlat - lat(1,j))
   362)            if (dist < mind) then
   363)               mind = dist
   364)               nj = j
   365)            endif
   366)         enddo
   367) 
   368)         n = 1
   369)         i = ni
   370)         j = nj
   371)         gGridRoot%data%rAttr(nlat ,n) = lat(i,j)
   372)         gGridRoot%data%rAttr(nlon ,n) = lon(i,j)
   373)         gGridRoot%data%rAttr(narea,n) = area(i,j)
   374)         gGridRoot%data%rAttr(nmask,n) = real(mask(i,j),R8)
   375)         gGridRoot%data%rAttr(nfrac,n) = frac(i,j)
   376)      else
   377)         n=0                                                                                    
   378)         do j=1,nyg
   379)         do i=1,nxg
   380)            n=n+1                                                                               
   381)            gGridRoot%data%rAttr(nlat ,n) = lat(i,j)
   382)            gGridRoot%data%rAttr(nlon ,n) = lon(i,j)
   383)            gGridRoot%data%rAttr(narea,n) = area(i,j)
   384)            gGridRoot%data%rAttr(nmask,n) = real(mask(i,j),R8)
   385)            gGridRoot%data%rAttr(nfrac,n) = frac(i,j)
   386)         enddo
   387)         enddo
   388)      endif
   389)      deallocate(lon)
   390)      deallocate(lat)
   391)      deallocate(area)
   392)      deallocate(mask)
   393)      deallocate(frac)
   394)   endif
   395) 
   396)   call mct_gGrid_scatter(gGridRoot, gGrid, gsMap, master_task, mpicom)
   397)   if (my_task == master_task) call mct_gGrid_clean(gGridRoot)
   398) 
   399) end subroutine shr_dmodel_readgrid

   725) ! !IROUTINE: shr_dmodel_gGridCompare -- returns TRUE if two domains are the same.
   726) !
   727) ! !DESCRIPTION:
   728) !    Returns TRUE if two domains are the the same (within tolerance).
   729) !
   730) ! !REVISION HISTORY:
   731) !    2005-May-03 - B. Kauffman - added mulitiple methods
   732) !    2005-May-27 - T. Craig - first version
   733) !
   734) ! !INTERFACE: ------------------------------------------------------------------
   735) 
   736) logical function shr_dmodel_gGridCompare(ggrid1,gsmap1,ggrid2,gsmap2,method,mpicom,eps)
   737) 
   738) ! !USES:
   739) 
   740)    implicit none
   741) 
   742) ! !INPUT/OUTPUT PARAMETERS:
   743) 
   744)    type(mct_gGrid)     ,intent(in)  :: ggrid1   ! 1st ggrid
   745)    type(mct_gsmap)     ,intent(in)  :: gsmap1   ! 1st gsmap
   746)    type(mct_gGrid)     ,intent(in)  :: ggrid2   ! 2nd ggrid
   747)    type(mct_gsmap)     ,intent(in)  :: gsmap2   ! 2nd gsmap
   748)    integer(IN)         ,intent(in)  :: method   ! selects what to compare
   749)    integer(IN)         ,intent(in)  :: mpicom   ! mpicom
   750)    real(R8)   ,optional,intent(in)  :: eps      ! epsilon compare value
   751) 
   752) !EOP
   753) 
   754)    !--- local ---
   755)    real(R8)    :: leps         ! local epsilon
   756)    integer(IN) :: n            ! counters
   757)    integer(IN) :: my_task,master_task
   758)    integer(IN) :: gsize
   759)    integer(IN) :: ierr
   760)    integer(IN) :: nlon1, nlon2, nlat1, nlat2, nmask1, nmask2  ! av field indices
   761)    logical     :: compare      ! local compare logical
   762)    real(R8)    :: lon1,lon2    ! longitudes to compare
   763)    real(R8)    :: lat1,lat2    ! latitudes to compare
   764)    real(R8)    :: msk1,msk2    ! masks to compare
   765)    integer(IN) :: nx,ni1,ni2   ! i grid size, i offset for 1 vs 2 and 2 vs 1
   766)    integer(IN) :: n1,n2,i,j    ! local indices
   767)    type(mct_aVect) :: avG1     ! global av
   768)    type(mct_aVect) :: avG2     ! global av
   769)    integer(IN) :: lmethod      ! local method
   770)    logical     :: maskmethod, maskpoint ! masking on method
   771)  
   772)    !--- formats ---
   773)    character(*),parameter :: subName = '(shr_dmodel_gGridCompare) '
   774)    character(*),parameter :: F01     = "('(shr_dmodel_gGridCompare) ',4a)" 
   775) 
   776) !-------------------------------------------------------------------------------
   777) !
   778) !-------------------------------------------------------------------------------
   779) 
   780)    call MPI_COMM_RANK(mpicom,my_task,ierr)
   781)    master_task = 0
   782) 
   783)    leps = 1.0e-6_R8
   784)    if (present(eps)) leps = eps
   785) 
   786)    lmethod = mod(method,100)
   787)    if (method > 100) then
   788)       maskmethod=.true.
   789)    else
   790)       maskmethod=.false.
   791)    endif
   792) 
   793)    call mct_aVect_gather(gGrid1%data,avG1,gsmap1,master_task,mpicom)
   794)    call mct_aVect_gather(gGrid2%data,avG2,gsmap2,master_task,mpicom)
   795) 
   796)    if (my_task == master_task) then
   797) 
   798)       compare = .true. 
   799)       gsize = mct_aVect_lsize(avG1)
   800)       if (gsize /= mct_aVect_lsize(avG2)) then
   801)          compare = .false.
   802)       endif
   803) 
   804)       if (.not. compare ) then
   805)          !--- already failed the comparison test, check no futher ---
   806)       else 
   807)          nlon1 = mct_aVect_indexRA(avG1,'lon')
   808)          nlat1 = mct_aVect_indexRA(avG1,'lat')
   809)          nlon2 = mct_aVect_indexRA(avG2,'lon')
   810)          nlat2 = mct_aVect_indexRA(avG2,'lat')
   811)          nmask1 = mct_aVect_indexRA(avG1,'mask')
   812)          nmask2 = mct_aVect_indexRA(avG2,'mask')
   813) 
   814)          ! To compare, want to be able to treat longitude wraparound generally.
   815)          ! So we need to compute i index offset and we need to compute the size of the nx dimension
   816)          ! First adjust the lon so it's in the range [0,360), add 1440 to lon to take into
   817)          ! accounts lons less than 1440.  if any lon is less than -1440, abort.  1440 is arbitrary
   818)          ! Next, comute ni1 and ni2.  These are the offsets of grid1 relative to grid2 and
   819)          ! grid2 relative to grid1.  The sum of those offsets is nx.  Use ni1 to offset grid2
   820)          ! in comparison and compute new grid2 index from ni1 and nx.  If ni1 is zero, then
   821)          ! there is no offset, don't need to compute ni2, and nx can be anything > 0.
   822) 
   823)          !--- compute offset of grid2 compared to pt 1 of grid 1
   824)          lon1 = minval(avG1%rAttr(nlon1,:))
   825)          lon2 = minval(avG2%rAttr(nlon2,:))
   826)          if ((lon1 < -1440.0_R8) .or. (lon2 < -1440.0_R8)) then
   827)             write(logunit,*) subname,' ERROR: lon1 lon2 lt -1440 ',lon1,lon2
   828)             call shr_sys_abort(subname//' ERROR: lon1 lon2 lt -1440')
   829)          endif
   830) 
   831)          lon1 = mod(avG1%rAttr(nlon1,1)+1440.0_R8,360.0_R8)
   832)          lat1 = avG1%rAttr(nlat1,1)
   833)          ni1 = -1
   834)          do n = 1,gsize
   835)             lon2 = mod(avG2%rAttr(nlon2,n)+1440.0_R8,360.0_R8)
   836)             lat2 = avG2%rAttr(nlat2,n)
   837)             if ((ni1 < 0) .and. abs(lon1-lon2) <= leps .and. abs(lat1-lat2) <= leps) then
   838)                ni1 = n - 1  ! offset, compare to first gridcell in grid 1
   839)             endif
   840)          enddo
   841) 
   842)          if (ni1 < 0) then        ! no match for grid point 1, so fails without going further
   843)             compare = .false.
   844)          elseif (ni1 == 0) then   ! no offset, set nx to anything > 0
   845)             nx = 1
   846)          else                     ! now compute ni2
   847)             ! compute offset of grid1 compared to pt 1 of grid 2
   848)             lon2 = mod(avG2%rAttr(nlon2,1)+1440.0_R8,360.0_R8)
   849)             lat2 = avG2%rAttr(nlat2,1)
   850)             ni2 = -1
   851)             do n = 1,gsize
   852)                lon1 = mod(avG1%rAttr(nlon1,n)+1440.0_R8,360.0_R8)
   853)                lat1 = avG1%rAttr(nlat1,n)
   854)                if ((ni2 < 0) .and. abs(lon1-lon2) <= leps .and. abs(lat1-lat2) <= leps) then
   855)                   ni2 = n - 1  ! offset, compare to first gridcell in grid 1
   856)                endif
   857)             enddo
   858)             if (ni2 < 0) then
   859)                write(logunit,*) subname,' ERROR in ni2 ',ni1,ni2
   860)                call shr_sys_abort(subname//' ERROR in ni2')
   861)             endif
   862)             nx = ni1 + ni2
   863)          endif
   864) 
   865)          if (compare) then
   866)             do n = 1,gsize
   867)                j = ((n-1)/nx) + 1
   868)                i = mod(n-1,nx) + 1
   869)                n1 = (j-1)*nx + mod(n-1,nx) + 1
   870)                n2 = (j-1)*nx + mod(n-1+ni1,nx) + 1
   871)                if (n1 /= n) then    ! sanity check, could be commented out
   872)                   write(logunit,*) subname,' ERROR in n1 n2 ',n,i,j,n1,n2
   873)                   call shr_sys_abort(subname//' ERROR in n1 n2')
   874)                endif
   875)                lon1 = mod(avG1%rAttr(nlon1,n1)+1440.0_R8,360.0_R8)
   876)                lat1 = avG1%rAttr(nlat1,n1)
   877)                lon2 = mod(avG2%rAttr(nlon2,n2)+1440.0_R8,360.0_R8)
   878)                lat2 = avG2%rAttr(nlat2,n2)
   879)                msk1 = avG1%rAttr(nmask1,n1)
   880)                msk2 = avG2%rAttr(nmask2,n2)
   881) 
   882)                maskpoint = .true.
   883)                if (maskmethod .and. (msk1 == 0 .or. msk2 == 0)) then
   884)                   maskpoint = .false.
   885)                endif
   886) 
   887)                if (maskpoint) then
   888)                if (lmethod == shr_dmodel_gGridCompareXYabs      ) then
   889)                   if (abs(lon1 - lon2) > leps) compare = .false.
   890)                   if (abs(lat1 - lat2) > leps) compare = .false.
   891)                else if (lmethod == shr_dmodel_gGridCompareXYrel      ) then
   892)                   if (rdiff(lon1,lon2) > leps) compare = .false.
   893)                   if (rdiff(lat1,lat2) > leps) compare = .false.
   894)                else if (lmethod == shr_dmodel_gGridCompareMaskIdent  ) then
   895)                   if (msk1 /= msk2)compare = .false.
   896)                else if (lmethod == shr_dmodel_gGridCompareMaskZeros  ) then
   897)                   if (msk1 == 0 .and. msk2 /= 0) compare = .false.
   898)                   if (msk1 /= 0 .and. msk2 == 0) compare = .false.
   899)                else if (lmethod == shr_dmodel_gGridCompareMaskSubset ) then
   900)                   if (msk1 /= 0 .and. msk2 == 0) compare = .false.
   901)                else
   902)                   write(logunit,F01) "ERROR: compare method not recognized, method = ",method
   903)                   call shr_sys_abort(subName//"ERROR: compare method not recognized")
   904)                endif  ! lmethod
   905)                endif  ! maskpoint
   906)             enddo ! gsize
   907)          endif  ! compare
   908)       endif   ! compare
   909)    endif   ! master_task
   910) 
   911)    if (my_task == master_task) call mct_avect_clean(avG1)
   912)    if (my_task == master_task) call mct_avect_clean(avG2)
   913) 
   914)    call shr_mpi_bcast(compare,mpicom)
