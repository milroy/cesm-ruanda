  1027)           do k=1,nlev    !  Loop inversion (AAM)
  1028)              do t=tl%n0,tl%n0
  1029)                 do q=1,qsize       
  1030)                    do i=1,np
  1031)                       do j=1,np          
  1032)                          dp = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
  1033)                               ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*elem(ie)%state%ps_v(i,j,t)
  1034)                          elem(ie)%state%Q(i,j,k,q)=elem(ie)%state%Qdp(i,j,k,q, n0_qdp)/dp 
  1035)                       enddo
  1036)                    enddo
  1037)                 enddo
  1038)              enddo
  1039)           enddo
  1040)        enddo
  1041)     endif
  1042) 
  1043) 
  1044)     ! timesteps to use for advective stability:  tstep*qsplit and tstep
  1045)     call print_cfl(elem,hybrid,nets,nete,dtnu)

  1048)        ! CAM has set tstep based on dtime before calling prim_init2(), 
  1049)        ! so only now does HOMME learn the timstep.  print them out:
  1050)        write(iulog,'(a,2f9.2)') "dt_remap: (0=disabled)   ",tstep*qsplit*rsplit
  1051)        write(iulog,'(a,2f9.2)') "dt_tracer, per RK stage: ",tstep*qsplit,(tstep*qsplit)/(rk_stage_user-1)
  1052)        write(iulog,'(a,2f9.2)') "dt_dyn:                  ",tstep
  1053)        write(iulog,'(a,2f9.2)') "dt_dyn (viscosity):      ",dt_dyn_vis
  1054)        write(iulog,'(a,2f9.2)') "dt_tracer (viscosity):   ",dt_tracer_vis
  1055) 
  1056)  
  1057) #ifdef CAM
  1058)        if (phys_tscale/=0) then
  1059)           write(iulog,'(a,2f9.2)') "CAM physics timescale:       ",phys_tscale
  1060)        endif
  1061)        write(iulog,'(a,2f9.2)') "CAM dtime (dt_phys):         ",tstep*nsplit*qsplit*max(rsplit,1)
  1062) #endif
  1063)     end if

  1067)     !Inside this routine, we enforce an OMP BARRIER and an OMP MASTER. It's left out of here because it's ugly
  1068)     call cuda_mod_init(elem,deriv(hybrid%ithr),hvcoord)
  1069) #endif
  1070)     if (hybrid%masterthread) write(iulog,*) "initial state:"
  1071)     call prim_printstate(elem, tl, hybrid,hvcoord,nets,nete, fvm)

  1078)   subroutine leapfrog_bootstrap(elem, hybrid,nets,nete,tstep,tl,hvcoord)
  1079) 
  1080)   !
  1081)   ! leapfrog bootstrap code.  
  1082)   !
  1083)   ! take the equivilent of one timestep, but do it with a 
  1084)   ! dt/2 euler and a dt/2 leapfrog step  
  1085)   !
  1086)   use hybvcoord_mod, only : hvcoord_t
  1087)   use time_mod, only : TimeLevel_t
  1088) 
  1089)   type (element_t) , intent(inout)        :: elem(:)
  1090)   type (hybrid_t), intent(in)           :: hybrid  ! distributed parallel structure (shared)
  1091)   type (hvcoord_t), intent(in)      :: hvcoord         ! hybrid vertical coordinate struct
  1092)   integer, intent(in)                     :: nets  ! starting thread element number (private)
  1093)   integer, intent(in)                     :: nete  ! ending thread element number   (private)
  1094)   real(kind=real_kind), intent(in)        :: tstep          ! "timestep dependent" timestep
  1095)   type (TimeLevel_t), intent(inout)       :: tl
  1096) 
  1097) 
  1098)   ! local
  1099)   real(kind=real_kind) :: tstep_tmp,tstep_dyn
  1100)   integer :: i,ie
  1101) 
  1102) 
  1103)   tstep_dyn = tstep
  1104)   ! forward euler to get to tstep_dyn/2 (keep t=0 in nm1 timelevel) 
  1105)   ! (note: leapfrog tstep_dyn/4 with nm1=n0 is Euler with tstep_dyn/2 )
  1106)   tstep_tmp=tstep_dyn/4        
  1107) 
  1108)   call prim_run(elem, hybrid,nets,nete, tstep_tmp, tl, hvcoord, "forward")
  1109)   
  1110)   ! leapfrog with tstep_dyn/2 to get to tstep_dyn (keep t=0 in nm1 timelevel)
  1111)   tstep_tmp=tstep_dyn/2
  1112)   call prim_run(elem, hybrid,nets,nete, tstep_tmp, tl, hvcoord, "forward")
  1113) 
  1114) 
  1115)   tl%nstep=tl%nstep-1        ! count all of that as 1 timestep
  1116) 
  1117)   end subroutine leapfrog_bootstrap

  1124)     use hybvcoord_mod, only : hvcoord_t
  1125)     use time_mod, only : TimeLevel_t, timelevel_update, smooth
  1126)     use control_mod, only: statefreq, integration, ftype, qsplit
  1127)     use prim_advance_mod, only : prim_advance_exp, prim_advance_si, preq_robert3
  1128)     use prim_state_mod, only : prim_printstate, prim_diag_scalars, prim_energy_halftimes
  1129)     use parallel_mod, only : abortmp
  1130) #ifndef CAM
  1131)     use column_model_mod, only : ApplyColumnModel
  1132) #endif
  1133) 
  1134)     type (element_t) , intent(inout)        :: elem(:)
  1135)     type (hybrid_t), intent(in)           :: hybrid  ! distributed parallel structure (shared)
  1136) 
  1137)     type (hvcoord_t), intent(in)      :: hvcoord         ! hybrid vertical coordinate struct
  1138) 
  1139)     integer, intent(in)                     :: nets  ! starting thread element number (private)
  1140)     integer, intent(in)                     :: nete  ! ending thread element number   (private)
  1141)     real(kind=real_kind), intent(in)        :: dt              ! "timestep dependent" timestep
  1142)     type (TimeLevel_t), intent(inout)       :: tl
  1143)     character(len=*), intent(in) :: advance_name
  1144)     real(kind=real_kind) :: st, st1, dp
  1145)     integer :: ie, t, q,k,i,j
  1146) 
  1147) 
  1148)     logical :: compute_diagnostics
  1149) 
  1150)     ! ===================================
  1151)     ! Main timestepping loop
  1152)     ! ===================================
  1153) 
  1154)     ! compute diagnostics and energy for STDOUT 
  1155)     ! compute energy if we are using an energy fixer
  1156) 
  1157)     compute_diagnostics=.false.
  1158)     if (MODULO(tl%nstep+1,statefreq)==0 .or. tl%nstep+1==tl%nstep0) then
  1159)        compute_diagnostics=.true.  
  1160)     endif
  1161)     tot_iter=0.0       
  1162) 
  1163) 
  1164)     ! Forcing options for testing CAM-HOMME energy balance:
  1165)     if (ftype == -1) then
  1166)        ! disable all forcing, but allow moisture:
  1167)        do ie=nets,nete
  1168)           elem(ie)%derived%FQ = 0
  1169)           elem(ie)%derived%FM = 0
  1170)           elem(ie)%derived%FT = 0
  1171)        enddo
  1172)     endif
  1173)     if (ftype == -2) then
  1174)        ! disable moisture, but allow dynamics forcing
  1175)        do ie=nets,nete
  1176)           elem(ie)%state%Q = 0
  1177)           elem(ie)%state%Qdp = 0
  1178)           elem(ie)%derived%FQ = 0
  1179)        enddo
  1180)     endif
  1181)     if (ftype == -3) then
  1182)        ! disable forcing & moisture
  1183)        do ie=nets,nete
  1184)           elem(ie)%state%Q = 0
  1185)           elem(ie)%state%Qdp = 0
  1186)           elem(ie)%derived%FQ = 0
  1187)           elem(ie)%derived%FM = 0
  1188)           elem(ie)%derived%FT = 0
  1189)        enddo
  1190)     endif
  1191) 
  1192)     ! =================================
  1193)     ! energy, dissipation rate diagnostics.  Uses data at t-1,t 
  1194)     ! to compute diagnostics at t - 0.5.  
  1195)     ! small error in the t+.5 terms because at this
  1196)     ! point only state variables at t-1 has been Robert filtered.  
  1197)     ! =================================
  1198)     if (compute_diagnostics) then
  1199)        call prim_energy_halftimes(elem,hvcoord,tl,1,.true.,nets,nete)
  1200)        call prim_diag_scalars(elem,hvcoord,tl,1,.true.,nets,nete)
  1201)     endif
  1202) 
  1203)     ! ===============
  1204)     ! initialize mean flux accumulation variables
  1205)     ! ===============
  1206)     do ie=nets,nete
  1207)        elem(ie)%derived%eta_dot_dpdn=0
  1208)        elem(ie)%derived%omega_p=0
  1209)     enddo
  1210) 
  1211)     ! ===============
  1212)     ! Dynamical Step  uses Q at tl%n0
  1213)     ! ===============
  1214) #if (! defined ELEMENT_OPENMP)
  1215) !$OMP BARRIER
  1216) #endif
  1217)     if (integration == "semi_imp") then
  1218)        call prim_advance_si(elem, nets, nete, cg(hybrid%ithr), blkjac, red, &
  1219)             refstate, hvcoord, deriv(hybrid%ithr), flt, hybrid, tl, dt)
  1220)        tot_iter=tot_iter+cg(hybrid%ithr)%iter
  1221)     else if (integration == "full_imp") then
  1222)        call abortmp('full_imp integration requires tstep_type > 0')
  1223)     else 
  1224)        call prim_advance_exp(elem, deriv(hybrid%ithr), hvcoord,   &
  1225)             hybrid, dt, tl, nets, nete, compute_diagnostics)
  1226) 
  1227)        ! keep lnps up to date (we should get rid of this requirement)
  1228)        do ie=nets,nete
  1229)           elem(ie)%state%lnps(:,:,tl%np1)= LOG(elem(ie)%state%ps_v(:,:,tl%np1))
  1230)        enddo
  1231)     end if
  1232) 
  1233)     ! =================================
  1234)     ! energy, dissipation rate diagnostics.  Uses data at t and t+1
  1235)     ! to compute diagnostics at t + 0.5.
  1236)     ! =================================
  1237)     if (compute_diagnostics) then
  1238)        call prim_energy_halftimes(elem,hvcoord,tl,2,.false.,nets,nete)
  1239)        call prim_diag_scalars(elem,hvcoord,tl,2,.false.,nets,nete)
  1240)     endif
  1241) 
  1242)     ! =================================== 
  1243)     ! Compute Forcing Tendencies from nm1 data (for PROCESS SPLIT)
  1244)     ! or np1 data (for TIMESPLIT) and add tendencies into soluiton at timelevel np1
  1245)     ! ===================================       
  1246) #ifdef CAM
  1247)     call abortmp('CAM-HOMME-SE requires RK timestepping option turned on')
  1248) #else
  1249)     call ApplyColumnModel(elem, hybrid, hvcoord, cm(hybrid%ithr),dt)
  1250) #endif
  1251)     ! measure the effects of forcing
  1252)     if (compute_diagnostics) then
  1253)        call prim_energy_halftimes(elem,hvcoord,tl,3,.false.,nets,nete)
  1254)        call prim_diag_scalars(elem,hvcoord,tl,3,.false.,nets,nete)
  1255)     endif
  1256) 
  1257) 
  1258)     ! =================================
  1259)     ! timestep is complete.  
  1260)     ! =================================
  1261) 
  1262) 
  1263)     !Now apply robert filter to all prognostic variables
  1264)     if (smooth/=0) &
  1265)        call preq_robert3(tl%nm1,tl%n0,tl%np1,elem,hvcoord,nets,nete)
  1266)     ! measure the effects of Robert filter
  1267)     if (compute_diagnostics) then
  1268)        call prim_energy_halftimes(elem,hvcoord,tl,4,.false.,nets,nete)
  1269)        call prim_diag_scalars(elem,hvcoord,tl,4,.false.,nets,nete)
  1270)     endif
  1271) 
  1272) 
  1273)     ! =================================
  1274)     ! update dynamics time level pointers 
  1275)     ! =================================
  1276)     call TimeLevel_update(tl,advance_name)
  1277) 
  1278)   ! ============================================================
  1279)     ! Print some diagnostic information 
  1280)     ! ============================================================
  1281) 
  1282)     if (compute_diagnostics) then
  1283)        if (hybrid%masterthread) then 
  1284)           if (integration == "semi_imp") write(iulog,*) "cg its=",cg(0)%iter
  1285)        end if
  1286)        call prim_printstate(elem, tl, hybrid,hvcoord,nets,nete)
  1287)     end if
  1288)   end subroutine prim_run

  1293)   subroutine prim_run_subcycle(elem, fvm, hybrid,nets,nete, dt, tl, hvcoord,nsubstep)
  1294) !
  1295) !   advance all variables (u,v,T,ps,Q,C) from time t to t + dt_q
  1296) !     
  1297) !   input: 
  1298) !       tl%nm1   not used
  1299) !       tl%n0    data at time t
  1300) !       tl%np1   new values at t+dt_q
  1301) !   
  1302) !   then we update timelevel pointers:
  1303) !       tl%nm1 = tl%n0
  1304) !       tl%n0  = tl%np1
  1305) !   so that:
  1306) !       tl%nm1   tracers:  t    dynamics:  t+(qsplit-1)*dt
  1307) !       tl%n0    time t + dt_q
  1308) !   
  1309) !
  1310)     use hybvcoord_mod, only : hvcoord_t
  1311)     use time_mod, only : TimeLevel_t, timelevel_update, timelevel_qdp, nsplit
  1312)     use control_mod, only: statefreq,&
  1313)            energy_fixer, ftype, qsplit, rsplit, test_cfldep
  1314)     use prim_advance_mod, only : applycamforcing, &
  1315)                                  applycamforcing_dynamics
  1316)     use prim_state_mod, only : prim_printstate, prim_diag_scalars, prim_energy_halftimes
  1317)     use parallel_mod, only : abortmp
  1318)     use reduction_mod, only : parallelmax
  1319)     use prim_advection_mod, only : vertical_remap
  1320) #if USE_CUDA_FORTRAN
  1321)     use cuda_mod, only: copy_qdp_h2d, copy_qdp_d2h
  1322) #endif
  1323)     
  1324) 
  1325)     type (element_t) , intent(inout)        :: elem(:)
  1326)     
  1327) #if defined(_SPELT)
  1328)       type(spelt_struct), intent(inout) :: fvm(:)
  1329) #else
  1330)       type(fvm_struct), intent(inout) :: fvm(:)
  1331) #endif
  1332)     type (hybrid_t), intent(in)           :: hybrid  ! distributed parallel structure (shared)
  1333) 
  1334)     type (hvcoord_t), intent(in)      :: hvcoord         ! hybrid vertical coordinate struct
  1335) 
  1336)     integer, intent(in)                     :: nets  ! starting thread element number (private)
  1337)     integer, intent(in)                     :: nete  ! ending thread element number   (private)
  1338)     real(kind=real_kind), intent(in)        :: dt  ! "timestep dependent" timestep
  1339)     type (TimeLevel_t), intent(inout)       :: tl
  1340)     integer, intent(in)                     :: nsubstep  ! nsubstep = 1 .. nsplit
  1341)     real(kind=real_kind) :: st, st1, dp, dt_q, dt_remap
  1342)     integer :: ie, t, q,k,i,j,n, n_Q
  1343)     integer :: n0_qdp,np1_qdp,r, nstep_end
  1344) 
  1345)     real (kind=real_kind)                          :: maxcflx, maxcfly  
  1346)     real (kind=real_kind) :: dp_np1(np,np)
  1347)     logical :: compute_diagnostics, compute_energy
  1348) 
  1349) 
  1350)     ! ===================================
  1351)     ! Main timestepping loop
  1352)     ! ===================================
  1353)     dt_q = dt*qsplit
  1354)     dt_remap = dt_q
  1355)     nstep_end = tl%nstep + qsplit
  1356)     if (rsplit>0) then
  1357)        dt_remap=dt_q*rsplit   ! rsplit=0 means use eulerian code, not vert. lagrange
  1358)        nstep_end = tl%nstep + qsplit*rsplit  ! nstep at end of this routine
  1359)     endif
  1360) 
  1361) 
  1362) 
  1363)     ! compute diagnostics and energy for STDOUT 
  1364)     ! compute energy if we are using an energy fixer
  1365)     compute_diagnostics=.false.
  1366)     compute_energy=energy_fixer > 0
  1367) 
  1368)     if (MODULO(nstep_end,statefreq)==0 .or. nstep_end==tl%nstep0) then
  1369)        compute_diagnostics=.true.  
  1370)        compute_energy = .true.
  1371)     endif
  1372)     if (compute_diagnostics) &
  1373)        call prim_diag_scalars(elem,hvcoord,tl,4,.true.,nets,nete)
  1374) 
  1375) 
  1376) 
  1377) #ifdef CAM
  1378)     ! ftype=2  Q was adjusted by physics, but apply u,T forcing here
  1379)     ! ftype=1  forcing was applied time-split in CAM coupling layer
  1380)     ! ftype=0 means forcing apply here
  1381)     ! ftype=-1 do not apply forcing
  1382)     call TimeLevel_Qdp( tl, qsplit, n0_qdp)
  1383)     if (ftype==0) call ApplyCAMForcing(elem, hvcoord,tl%n0,n0_qdp, dt_remap,nets,nete)
  1384)     if (ftype==2) call ApplyCAMForcing_dynamics(elem, hvcoord,tl%n0,dt_remap,nets,nete)
  1385) #endif
  1386) 
  1387)     ! E(1) Energy after CAM forcing
  1388)     if (compute_energy) call prim_energy_halftimes(elem,hvcoord,tl,1,.true.,nets,nete)
  1389) 
  1390)     ! qmass and variance, using Q(n0),Qdp(n0)
  1391)     if (compute_diagnostics) call prim_diag_scalars(elem,hvcoord,tl,1,.true.,nets,nete)
  1392) 
  1393)     ! initialize dp3d from ps
  1394)     if (rsplit>0) then
  1395)     do ie=nets,nete
  1396)        do k=1,nlev
  1397)           elem(ie)%state%dp3d(:,:,k,tl%n0)=&
  1398)                ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
  1399)                ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*elem(ie)%state%ps_v(:,:,tl%n0)
  1400)        enddo
  1401)        ! DEBUGDP step: ps_v should not be used for rsplit>0 code during prim_step
  1402)        ! vertical_remap.  so to this for debugging:
  1403)        elem(ie)%state%ps_v(:,:,tl%n0)=-9e9
  1404)     enddo
  1405)     endif
  1406) 
  1407) #if USE_CUDA_FORTRAN
  1408)     call TimeLevel_Qdp( tl, qsplit, n0_qdp, np1_qdp) 
  1409)     call copy_qdp_h2d( elem , n0_qdp )
  1410) #endif
  1411) 
  1412)     ! loop over rsplit vertically lagrangian timesteps
  1413)     call prim_step(elem, fvm, hybrid,nets,nete, dt, tl, hvcoord,compute_diagnostics)
  1414)     do r=2,rsplit   
  1415)        call TimeLevel_update(tl,"leapfrog")
  1416)        call prim_step(elem, fvm, hybrid,nets,nete, dt, tl, hvcoord,.false.)
  1417)     enddo
  1418)     ! defer final timelevel update until after remap and diagnostics
  1419) 
  1420) 
  1421)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1422)     !
  1423)     !  apply vertical remap
  1424)     !  always for tracers  
  1425)     !  if rsplit>0:  also remap dynamics and compute reference level ps_v
  1426)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1427)     !compute timelevels for tracers (no longer the same as dynamics)
  1428)     call TimeLevel_Qdp( tl, qsplit, n0_qdp, np1_qdp) 
  1429)     call vertical_remap(elem,fvm,hvcoord,dt_remap,tl%np1,np1_qdp,nets,nete)
  1430) 
  1431) #if USE_CUDA_FORTRAN
  1432)     call TimeLevel_Qdp( tl, qsplit, n0_qdp, np1_qdp) 
  1433)     call copy_qdp_d2h( elem , np1_qdp )
  1434) #endif
  1435) 
  1436)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  1437)     ! time step is complete.  update some diagnostic variables:
  1438)     ! lnps (we should get rid of this)
  1439)     ! Q    (mixing ratio)
  1440)     !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!    
  1441)     do ie=nets,nete
  1442)        elem(ie)%state%lnps(:,:,tl%np1)= LOG(elem(ie)%state%ps_v(:,:,tl%np1))
  1443) #if (defined ELEMENT_OPENMP)
  1444)        !$omp parallel do private(k,q)
  1445) #endif
  1446)        do k=1,nlev    !  Loop inversion (AAM)
  1447)           dp_np1(:,:) = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
  1448)                ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*elem(ie)%state%ps_v(:,:,tl%np1)
  1449)           do q=1,qsize
  1450)              elem(ie)%state%Q(:,:,k,q)=elem(ie)%state%Qdp(:,:,k,q,np1_qdp)/dp_np1(:,:) 
  1451)           enddo
  1452)        enddo
  1453)     enddo
  1454)     
  1455) 
  1456) 
  1457)    
  1458) 
  1459)     ! now we have:
  1460)     !   u(nm1)   dynamics at  t+dt_remap - 2*dt 
  1461)     !   u(n0)    dynamics at  t+dt_remap - dt    
  1462)     !   u(np1)   dynamics at  t+dt_remap
  1463)     !
  1464)     !   Q(1)   Q at t+dt_remap
  1465)     if (compute_diagnostics) call prim_diag_scalars(elem,hvcoord,tl,2,.false.,nets,nete)
  1466)     if (compute_energy) call prim_energy_halftimes(elem,hvcoord,tl,2,.false.,nets,nete)
  1467) 
  1468)     if (energy_fixer > 0) then
  1469)        call prim_energy_fixer(elem,hvcoord,hybrid,tl,nets,nete,nsubstep)
  1470)     endif
  1471) 
  1472)     if (compute_diagnostics) then
  1473)        call prim_diag_scalars(elem,hvcoord,tl,3,.false.,nets,nete)
  1474)        call prim_energy_halftimes(elem,hvcoord,tl,3,.false.,nets,nete)
  1475)      endif
  1476) 
  1477)     ! =================================
  1478)     ! update dynamics time level pointers 
  1479)     ! =================================
  1480)     call TimeLevel_update(tl,"leapfrog")
  1481) 
  1482)     ! now we have:
  1483)     !   u(nm1)   dynamics at  t+dt_remap - dt       (Robert-filtered)
  1484)     !   u(n0)    dynamics at  t+dt_remap 
  1485)     !   u(np1)   undefined
  1486)  
  1487) 
  1488)     ! ============================================================
  1489)     ! Print some diagnostic information 
  1490)     ! ============================================================
  1491)     if (compute_diagnostics) then

  1743)     ! slightly different (about 0.1 W/m^2) then expected by the physics
