    71) ! fincl1 = 'field1', 'field2',...
    72) !                      List of fields to add to the primary history file.
    73) ! fincl1lonlat = 'longitude by latitude','longitude by latitude',...
    74) !                      List of columns ('longitude_latitude') or contiguous 
    75) !                      columns ('longitude:longitude_latitude:latitude') at 
    76) !                      which the fincl1 fields will be output. Individual 
    77) !                      columns are specified as a string using a longitude
    78) !                      degree (greater or equal to 0.) followed by a single 
    79) !                      character (e)ast/(w)est identifer, an
    80) !                      underscore '_' , and a latitude degree followed by a 
    81) !                      single character (n)orth/(s)outh identifier.
    82) !                      example '10e_20n' would pick the model column closest
    83) !                      to 10 degrees east longitude by 20 degrees north 
    84) !                      latitude.  A group of contiguous columns can be 
    85) !                      specified by using lon lat ranges with their single
    86) !                      character east/west or north/south identifiers
    87) !                      example '10e:20e_15n:20n'.  Would outfield all 
    88) !                      fincl1 fields at the model columns which fall
    89) !                      with in the longitude range from 10 east to 20 east
    90) !                      and the latitude range from 15 north to 20 north
    91) !
    92) ! fincl[2..6] = 'field1', 'field2',...
    93) !                      List of fields to add to the auxiliary history file.
    94) !
    95) ! fincl2..6]lonlat = 'longitude by latitude','longitude by latitude',...
    96) !                      List of columns ('longitude_latitude') or contiguous 
    97) !                      columns ('longitude:longitude_latitude:latitude') at 
    98) !                      which the fincl[2..6] fields will be output. Individual 
    99) !                      columns are specified as a string using a longitude
   100) !                      degree (greater or equal to 0.) followed by a single 
   101) !                      character (e)ast/(w)est identifer, an
   102) !                      underscore '_' , and a latitude degree followed by a 
   103) !                      singel character (n)orth/(s)outh identifier.
   104) !                      example '10e_20n' would pick the model column closest
   105) !                      to 10 degrees east longitude by 20 degrees north 
   106) !                      latitude.  A group of contiguous columns can be 
   107) !                      specified by using lon lat ranges with their single
   108) !                      character east/west or north/south identifiers
   109) !                      example '10e:20e_15n:20n'.  Would outfield all 
   110) !                      fincl[2..6] fields at the model columns which fall
   111) !                      with in the longitude range from 10 east to 20 east
   112) !                      and the latitude range from 15 north to 20 north
   113) !
   114) ! fexcl1 = 'field1','field2',... 
   115) !                      List of field names to exclude from default
   116) !                      primary history file (default fields on the 
   117) !                      Master Field List).
   118) ! 
   119) ! fexcl[2..6] = 'field1','field2',... 
   120) !                      List of field names to exclude from
   121) !                      auxiliary history files.
   122) ! 
   123) ! lcltod_start = nn,nn,nn,...
   124) !                      Array containing the starting time of day for local time history
   125) !                      averaging. Used in conjuction with lcltod_stop. If lcltod_stop
   126) !                      is less than lcltod_start, then the time range wraps around
   127) !                      24 hours. The start time is included in the interval. Time is
   128) !                      in seconds and defaults to 39600 (11:00 AM).
   129) !                      The first value applies to the primary hist. file,
   130) !                      the second to the first aux. hist. file, etc.
   131) ! 
   132) ! lcltod_stop = nn,nn,nn,...
   133) !                      Array containing the stopping time of day for local time history
   134) !                      averaging. Used in conjuction with lcltod_start. If lcltod_stop
   135) !                      is less than lcltod_start, then the time range wraps around
   136) !                      24 hours. The stop time is not included in the interval. Time is
   137) !                      in seconds and defaults to 0 (midnight).
   138) !                      The first value applies to the primary hist. file,
   139) !                      the second to the first aux. hist. file, etc.
   140) ! 
   141) ! lcltod_start = nn,nn,nn,...
   142) !                      Array containing the starting time of day for local time history
   143) !                      averaging. Used in conjuction with lcltod_stop. If lcltod_stop
   144) !                      is less than lcltod_start, then the time range wraps around
   145) !                      24 hours. The start time is included in the interval. Time is
   146) !                      in seconds and defaults to 39600 (11:00 AM).
   147) !                      The first value applies to the primary hist. file,
   148) !                      the second to the first aux. hist. file, etc.
   149) ! 
   150) ! lcltod_stop = nn,nn,nn,...
   151) !                      Array containing the stopping time of day for local time history
   152) !                      averaging. Used in conjuction with lcltod_start. If lcltod_stop
   153) !                      is less than lcltod_start, then the time range wraps around
   154) !                      24 hours. The stop time is not included in the interval. Time is
   155) !                      in seconds and defaults to 0 (midnight).
   156) !                      The first value applies to the primary hist. file,
   157) !                      the second to the first aux. hist. file, etc.
   158) ! 
   159) ! mfilt = nn,nn,nn     Array containing the maximum number of time 
   160) !                      samples per disk history file. Defaults to 5.
   161) !                      The first value applies to the primary hist. file,
   162) !                      the second to the first aux. hist. file, etc.
   163) ! 
   164) ! ncdata               Path and filename of initial condition dataset.
