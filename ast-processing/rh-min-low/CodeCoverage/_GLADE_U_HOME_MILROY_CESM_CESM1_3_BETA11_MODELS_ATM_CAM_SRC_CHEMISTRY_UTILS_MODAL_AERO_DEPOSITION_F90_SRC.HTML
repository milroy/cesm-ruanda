   116) subroutine set_srf_wetdep(aerdepwetis, aerdepwetcw, cam_out)
   117) 
   118) ! Set surface wet deposition fluxes passed to coupler.
   119) 
   120)    ! Arguments:
   121)    real(r8), intent(in) :: aerdepwetis(:,:)  ! aerosol wet deposition (interstitial)
   122)    real(r8), intent(in) :: aerdepwetcw(:,:)  ! aerosol wet deposition (cloud water)
   123)    type(cam_out_t), intent(inout) :: cam_out     ! cam export state
   124) 
   125)    ! Local variables:
   126)    integer :: i
   127)    integer :: ncol                      ! number of columns
   128)    !----------------------------------------------------------------------------
   129) 
   130)    if (.not.bin_fluxes) return
   131) 
   132)    ncol = cam_out%ncol
   133) 
   134)    ! derive cam_out variables from deposition fluxes
   135)    !  note: wet deposition fluxes are negative into surface, 
   136)    !        dry deposition fluxes are positive into surface.
   137)    !        CLM wants positive definite fluxes.
   138)    do i = 1, ncol
   139)       ! black carbon fluxes
   140)       cam_out%bcphiwet(i) = -(aerdepwetis(i,idx_bc1)+aerdepwetcw(i,idx_bc1))
   141) 
   142)       ! organic carbon fluxes
   143)       cam_out%ocphiwet(i) = -(aerdepwetis(i,idx_pom1)+aerdepwetis(i,idx_soa1)+aerdepwetcw(i,idx_pom1)+aerdepwetcw(i,idx_soa1))
   144) 
   145)       ! dust fluxes
   146)       !
   147)       ! bulk bin1 (fine) dust deposition equals accumulation mode deposition:
   148)       cam_out%dstwet1(i) = -(aerdepwetis(i,idx_dst1)+aerdepwetcw(i,idx_dst1))
   149)       
   150)       !  A. Simple: Assign all coarse-mode dust to bulk size bin 3:
   151)       cam_out%dstwet2(i) = 0._r8
   152)       cam_out%dstwet3(i) = -(aerdepwetis(i,idx_dst3)+aerdepwetcw(i,idx_dst3))
   153)       cam_out%dstwet4(i) = 0._r8
   154) 
   155)       ! in rare cases, integrated deposition tendency is upward
   156)       if (cam_out%bcphiwet(i) .lt. 0._r8) cam_out%bcphiwet(i) = 0._r8
   157)       if (cam_out%ocphiwet(i) .lt. 0._r8) cam_out%ocphiwet(i) = 0._r8
   158)       if (cam_out%dstwet1(i)  .lt. 0._r8) cam_out%dstwet1(i)  = 0._r8
   159)       if (cam_out%dstwet3(i)  .lt. 0._r8) cam_out%dstwet3(i)  = 0._r8
   160)    enddo
   161) 
   162) end subroutine set_srf_wetdep

   166) subroutine set_srf_drydep(aerdepdryis, aerdepdrycw, cam_out)
   167) 
   168) ! Set surface dry deposition fluxes passed to coupler.
   169)    
   170)    ! Arguments:
   171)    real(r8), intent(in) :: aerdepdryis(:,:)  ! aerosol dry deposition (interstitial)
   172)    real(r8), intent(in) :: aerdepdrycw(:,:)  ! aerosol dry deposition (cloud water)
   173)    type(cam_out_t), intent(inout) :: cam_out     ! cam export state
   174) 
   175)    ! Local variables:
   176)    integer :: i
   177)    integer :: ncol                      ! number of columns
   178)    !----------------------------------------------------------------------------
   179)    if (.not.bin_fluxes) return
   180) 
   181)    ncol = cam_out%ncol
   182) 
   183)    ! derive cam_out variables from deposition fluxes
   184)    !  note: wet deposition fluxes are negative into surface, 
   185)    !        dry deposition fluxes are positive into surface.
   186)    !        CLM wants positive definite fluxes.
   187)    do i = 1, ncol
   188)       ! black carbon fluxes
   189)       cam_out%bcphidry(i) = aerdepdryis(i,idx_bc1)+aerdepdrycw(i,idx_bc1)
   190)       cam_out%bcphodry(i) = 0._r8
   191) 
   192)       ! organic carbon fluxes
   193)       cam_out%ocphidry(i) = aerdepdryis(i,idx_pom1)+aerdepdryis(i,idx_soa1)+aerdepdrycw(i,idx_pom1)+aerdepdrycw(i,idx_soa1)
   194)       cam_out%ocphodry(i) = aerdepdryis(i,idx_soa2)+aerdepdrycw(i,idx_soa2)
   195) 
   196)       ! dust fluxes
   197)       !
   198)       ! bulk bin1 (fine) dust deposition equals accumulation mode deposition:
   199)       cam_out%dstdry1(i) = aerdepdryis(i,idx_dst1)+aerdepdrycw(i,idx_dst1)
   200)       
   201)       ! Two options for partitioning deposition into bins 2-4:
   202)       !  A. Simple: Assign all coarse-mode dust to bulk size bin 3:
   203)       cam_out%dstdry2(i) = 0._r8
   204)       cam_out%dstdry3(i) = aerdepdryis(i,idx_dst3)+aerdepdrycw(i,idx_dst3)
   205)       cam_out%dstdry4(i) = 0._r8
   206) 
   207)       ! in rare cases, integrated deposition tendency is upward
   208)       if (cam_out%bcphidry(i) .lt. 0._r8) cam_out%bcphidry(i) = 0._r8
   209)       if (cam_out%bcphodry(i) .lt. 0._r8) cam_out%bcphodry(i) = 0._r8
   210)       if (cam_out%ocphidry(i) .lt. 0._r8) cam_out%ocphidry(i) = 0._r8
   211)       if (cam_out%ocphodry(i) .lt. 0._r8) cam_out%ocphodry(i) = 0._r8
   212)       if (cam_out%dstdry1(i)  .lt. 0._r8) cam_out%dstdry1(i)  = 0._r8
   213)       if (cam_out%dstdry3(i)  .lt. 0._r8) cam_out%dstdry3(i)  = 0._r8
   214)    enddo
   215) 
   216) end subroutine set_srf_drydep
