    52)   subroutine lininterp_init(yin, nin, yout, nout, extrap_method, interp_wgts, &
    53)        cyclicmin, cyclicmax)
    54) !
    55) ! Description:
    56) !   Initialize a variable of type(interp_type) with weights for linear interpolation.
    57) !       this variable can then be used in calls to lininterp1d and lininterp2d.
    58) !   yin is a 1d array of length nin of locations to interpolate from - this array must
    59) !       be monotonic but can be increasing or decreasing
    60) !   yout is a 1d array of length nout of locations to interpolate to, this array need
    61) !       not be ordered
    62) !   extrap_method determines how to handle yout points beyond the bounds of yin
    63) !       if 0 set values outside output grid to 0
    64) !       if 1 set to boundary value
    65) !       if 2 set to cyclic boundaries
    66) !         optional values cyclicmin and cyclicmax can be used to set the bounds of the
    67) !         cyclic mapping - these default to 0 and 360.
    68) !
    69) 
    70)     integer, intent(in) :: nin
    71)     integer, intent(in) :: nout
    72)     real(r8), intent(in) :: yin(:)           ! input mesh
    73)     real(r8), intent(in) :: yout(:)         ! output mesh
    74)     integer, intent(in) :: extrap_method       ! if 0 set values outside output grid to 0
    75)                                                ! if 1 set to boundary value
    76)                                                ! if 2 set to cyclic boundaries
    77)     real(r8), intent(in), optional :: cyclicmin, cyclicmax
    78) 
    79)     type (interp_type), intent(out) :: interp_wgts
    80)     real(r8) :: cmin, cmax
    81)     real(r8) :: extrap
    82)     real(r8) :: dyinwrap
    83)     real(r8) :: ratio
    84)     real(r8) :: avgdyin
    85)     integer :: i, j, icount
    86)     integer :: jj
    87)     real(r8), pointer :: wgts(:)
    88)     real(r8), pointer :: wgtn(:)
    89)     integer, pointer :: jjm(:)
    90)     integer, pointer :: jjp(:)
    91)     logical :: increasing
    92)     !
    93)     ! Check validity of input coordinate arrays: must be monotonically increasing,
    94)     ! and have a total of at least 2 elements
    95)     !
    96)     if (nin.lt.2) then
    97)        call endrun('LININTERP: Must have at least 2 input points for interpolation')
    98)     end if
    99)     if(present(cyclicmin)) then
   100)        cmin=cyclicmin
   101)     else
   102)        cmin=0_r8
   103)     end if
   104)     if(present(cyclicmax)) then
   105)        cmax=cyclicmax
   106)     else
   107)        cmax=360_r8
   108)     end if
   109)     if(cmax<=cmin) then
   110)        call endrun('LININTERP: cyclic min value must be < max value')
   111)     end if
   112)     increasing=.true.
   113)     icount = 0
   114)     do j=1,nin-1
   115)        if (yin(j).gt.yin(j+1)) icount = icount + 1
   116)     end do
   117)     if(icount.eq.nin-1) then
   118)        increasing = .false.
   119)        icount=0
   120)     endif
   121)     if (icount.gt.0) then
   122)        call endrun('LININTERP: Non-monotonic input coordinate array found')
   123)     end if
   124)     allocate(interp_wgts%jjm(nout), &
   125)          interp_wgts%jjp(nout), &
   126)          interp_wgts%wgts(nout), &
   127)          interp_wgts%wgtn(nout))
   128) 
   129)     jjm => interp_wgts%jjm
   130)     jjp => interp_wgts%jjp
   131)     wgts =>  interp_wgts%wgts
   132)     wgtn =>  interp_wgts%wgtn
   133) 
   134)     !
   135)     ! Initialize index arrays for later checking
   136)     !
   137)     jjm = 0
   138)     jjp = 0
   139) 
   140)     extrap = 0._r8
   141)     select case (extrap_method)
   142)     case (extrap_method_zero)
   143)        !
   144)        ! For values which extend beyond boundaries, set weights
   145)        ! such that values will be 0.
   146)        !
   147)        do j=1,nout
   148)           if(increasing) then
   149)              if (yout(j).lt.yin(1)) then
   150)                 jjm(j) = 1
   151)                 jjp(j) = 1
   152)                 wgts(j) = 0._r8
   153)                 wgtn(j) = 0._r8
   154)                 extrap = extrap + 1._r8
   155)              else if (yout(j).gt.yin(nin)) then
   156)                 jjm(j) = nin
   157)                 jjp(j) = nin
   158)                 wgts(j) = 0._r8
   159)                 wgtn(j) = 0._r8
   160)                 extrap = extrap + 1._r8
   161)              end if
   162)           else
   163)              if (yout(j).gt.yin(1)) then
   164)                 jjm(j) = 1
   165)                 jjp(j) = 1
   166)                 wgts(j) = 0._r8
   167)                 wgtn(j) = 0._r8
   168)                 extrap = extrap + 1._r8
   169)              else if (yout(j).lt.yin(nin)) then
   170)                 jjm(j) = nin
   171)                 jjp(j) = nin
   172)                 wgts(j) = 0._r8
   173)                 wgtn(j) = 0._r8
   174)                 extrap = extrap + 1._r8
   175)              end if
   176)           end if
   177)        end do
   178)     case (extrap_method_bndry)
   179)        !
   180)        ! For values which extend beyond boundaries, set weights
   181)        ! such that values will just be copied.
   182)        !
   183)        do j=1,nout
   184)           if(increasing) then
   185)              if (yout(j).le.yin(1)) then
   186)                 jjm(j) = 1
   187)                 jjp(j) = 1
   188)                 wgts(j) = 1._r8
   189)                 wgtn(j) = 0._r8
   190)                 extrap = extrap + 1._r8
   191)              else if (yout(j).gt.yin(nin)) then
   192)                 jjm(j) = nin
   193)                 jjp(j) = nin
   194)                 wgts(j) = 1._r8
   195)                 wgtn(j) = 0._r8
   196)                 extrap = extrap + 1._r8
   197)              end if
   198)           else
   199)              if (yout(j).gt.yin(1)) then
   200)                 jjm(j) = 1
   201)                 jjp(j) = 1
   202)                 wgts(j) = 1._r8
   203)                 wgtn(j) = 0._r8
   204)                 extrap = extrap + 1._r8
   205)              else if (yout(j).le.yin(nin)) then
   206)                 jjm(j) = nin
   207)                 jjp(j) = nin
   208)                 wgts(j) = 1._r8
   209)                 wgtn(j) = 0._r8
   210)                 extrap = extrap + 1._r8
   211)              end if
   212)           end if
   213)        end do
   214)     case (extrap_method_cycle)
   215)        !
   216)        ! For values which extend beyond boundaries, set weights
   217)        ! for circular boundaries
   218)        !
   219)        dyinwrap = yin(1) + (cmax-cmin) - yin(nin)
   220)        avgdyin = abs(yin(nin)-yin(1))/(nin-1._r8)
   221)        ratio = dyinwrap/avgdyin
   222)        if (ratio < 0.9_r8 .or. ratio > 1.1_r8) then
   223)           write(iulog,*) 'Lininterp: Bad dyinwrap value =',dyinwrap,&
   224)                ' avg=', avgdyin, yin(1),yin(nin)
   225)           call endrun('interpolate_data')
   226)        end if
   227) 
   228)        do j=1,nout
   229)           if(increasing) then
   230)              if (yout(j) <= yin(1)) then
   231)                 jjm(j) = nin
   232)                 jjp(j) = 1
   233)                 wgts(j) = (yin(1)-yout(j))/dyinwrap
   234)                 wgtn(j) = (yout(j)+(cmax-cmin) - yin(nin))/dyinwrap
   235)              else if (yout(j) > yin(nin)) then
   236)                 jjm(j) = nin
   237)                 jjp(j) = 1
   238)                 wgts(j) = (yin(1)+(cmax-cmin)-yout(j))/dyinwrap
   239)                 wgtn(j) = (yout(j)-yin(nin))/dyinwrap
   240)              end if
   241)           else
   242)              if (yout(j) > yin(1)) then
   243)                 jjm(j) = nin
   244)                 jjp(j) = 1
   245)                 wgts(j) = (yin(1)-yout(j))/dyinwrap
   246)                 wgtn(j) = (yout(j)+(cmax-cmin) - yin(nin))/dyinwrap
   247)              else if (yout(j) <= yin(nin)) then
   248)                 jjm(j) = nin
   249)                 jjp(j) = 1
   250)                 wgts(j) = (yin(1)+(cmax-cmin)-yout(j))/dyinwrap
   251)                 wgtn(j) = (yout(j)+(cmax-cmin)-yin(nin))/dyinwrap
   252)              end if
   253) 
   254)           endif
   255)        end do
   256)     end select
   257) 
   258)     !
   259)     ! Loop though output indices finding input indices and weights
   260)     !
   261)     if(increasing) then
   262)        do j=1,nout
   263)           do jj=1,nin-1
   264)              if (yout(j).gt.yin(jj) .and. yout(j).le.yin(jj+1)) then
   265)                 jjm(j) = jj
   266)                 jjp(j) = jj + 1
   267)                 wgts(j) = (yin(jj+1)-yout(j))/(yin(jj+1)-yin(jj))
   268)                 wgtn(j) = (yout(j)-yin(jj))/(yin(jj+1)-yin(jj))
   269)                 exit
   270)              end if
   271)           end do
   272)        end do
   273)     else
   274)        do j=1,nout
   275)           do jj=1,nin-1
   276)              if (yout(j).le.yin(jj) .and. yout(j).gt.yin(jj+1)) then
   277)                 jjm(j) = jj
   278)                 jjp(j) = jj + 1
   279)                 wgts(j) = (yin(jj+1)-yout(j))/(yin(jj+1)-yin(jj))
   280)                 wgtn(j) = (yout(j)-yin(jj))/(yin(jj+1)-yin(jj))
   281)                 exit
   282)              end if
   283)           end do
   284)        end do
   285)     end if
   286) 
   287)     !
   288)     ! Check that interp/extrap points have been found for all outputs
   289)     !
   290)     icount = 0
   291)     do j=1,nout
   292)        if (jjm(j).eq.0 .or. jjp(j).eq.0) icount = icount + 1
   293)        ratio=wgts(j)+wgtn(j)
   294)        if((ratio<0.9_r8.or.ratio>1.1_r8).and.extrap_method.ne.0) then
   295)           write(iulog,*) j, wgts(j),wgtn(j),jjm(j),jjp(j), increasing,extrap_method
   296)           call endrun('Bad weight computed in LININTERP_init')
   297)        end if
   298)     end do
   299)     if (icount.gt.0) then
   300)        call endrun('LININTERP: Point found without interp indices')
   301)     end if
   302) 
   303)   end subroutine lininterp_init

   305)   subroutine lininterp1d (arrin, n1, arrout, m1, interp_wgts)
   306)     !-----------------------------------------------------------------------
   307)     !
   308)     ! Purpose: Do a linear interpolation from input mesh to output
   309)     !          mesh with weights as set in lininterp_init.
   310)     !
   311)     !
   312)     ! Author: Jim Edwards
   313)     !
   314)     !-----------------------------------------------------------------------
   315)     !-----------------------------------------------------------------------
   316)     implicit none
   317)     !-----------------------------------------------------------------------
   318)     !
   319)     ! Arguments
   320)     !
   321)     integer, intent(in) :: n1                 ! number of input latitudes
   322)     integer, intent(in) :: m1                ! number of output latitudes
   323) 
   324)     real(r8), intent(in) :: arrin(n1)    ! input array of values to interpolate
   325)     type(interp_type), intent(in) :: interp_wgts
   326)     real(r8), intent(out) :: arrout(m1) ! interpolated array
   327) 
   328)     !
   329)     ! Local workspace
   330)     !
   331)     integer j                ! latitude indices
   332)     integer, pointer :: jjm(:)
   333)     integer, pointer :: jjp(:)
   334) 
   335)     real(r8), pointer :: wgts(:)
   336)     real(r8), pointer :: wgtn(:)
   337) 
   338) 
   339)     jjm => interp_wgts%jjm
   340)     jjp => interp_wgts%jjp
   341)     wgts =>  interp_wgts%wgts
   342)     wgtn =>  interp_wgts%wgtn
   343) 
   344)     !
   345)     ! Do the interpolation
   346)     !
   347)     do j=1,m1
   348)       arrout(j) = arrin(jjm(j))*wgts(j) + arrin(jjp(j))*wgtn(j)
   349)     end do
   350) 
   351)     return
   352)   end subroutine lininterp1d

   495)   subroutine lininterp_original (arrin, yin, nlev, nlatin, arrout, &
   496)        yout, nlatout)
   497)     !-----------------------------------------------------------------------
   498)     !
   499)     ! Purpose: Do a linear interpolation from input mesh defined by yin to output
   500)     !          mesh defined by yout.  Where extrapolation is necessary, values will
   501)     !          be copied from the extreme edge of the input grid.  Vectorization is over
   502)     !          the vertical (nlev) dimension.
   503)     !
   504)     ! Method: Check validity of input, then determine weights, then do the N-S interpolation.
   505)     !
   506)     ! Author: Jim Rosinski
   507)     ! Modified: Jim Edwards so that there is no requirement of monotonacity on the yout array
   508)     !
   509)     !-----------------------------------------------------------------------
   510)     implicit none
   511)     !-----------------------------------------------------------------------
   512)     !
   513)     ! Arguments
   514)     !
   515)     integer, intent(in) :: nlev                   ! number of vertical levels
   516)     integer, intent(in) :: nlatin                 ! number of input latitudes
   517)     integer, intent(in) :: nlatout                ! number of output latitudes
   518) 
   519)     real(r8), intent(in) :: arrin(nlev,nlatin)    ! input array of values to interpolate
   520)     real(r8), intent(in) :: yin(nlatin)           ! input mesh
   521)     real(r8), intent(in) :: yout(nlatout)         ! output mesh
   522) 
   523)     real(r8), intent(out) :: arrout(nlev,nlatout) ! interpolated array
   524)     !
   525)     ! Local workspace
   526)     !
   527)     integer j, jj              ! latitude indices
   528)     integer jjprev             ! latitude indices
   529)     integer k                  ! level index
   530)     integer icount             ! number of values
   531) 
   532)     real(r8) extrap            ! percent grid non-overlap
   533)     !
   534)     ! Dynamic
   535)     !
   536)     integer :: jjm(nlatout)
   537)     integer :: jjp(nlatout)
   538) 
   539)     real(r8) :: wgts(nlatout)
   540)     real(r8) :: wgtn(nlatout)
   541)     !
   542)     ! Check validity of input coordinate arrays: must be monotonically increasing,
   543)     ! and have a total of at least 2 elements
   544)     !
   545)     if (nlatin.lt.2) then
   546)        call endrun('LININTERP: Must have at least 2 input points for interpolation')
   547)     end if
   548) 
   549)     icount = 0
   550)     do j=1,nlatin-1
   551)        if (yin(j).gt.yin(j+1)) icount = icount + 1
   552)     end do
   553) 
   554) 
   555)     if (icount.gt.0) then
   556)        call endrun('LININTERP: Non-monotonic coordinate array(s) found')
   557)     end if
   558)     !
   559)     ! Initialize index arrays for later checking
   560)     !
   561)     do j=1,nlatout
   562)        jjm(j) = 0
   563)        jjp(j) = 0
   564)     end do
   565)     !
   566)     ! For values which extend beyond N and S boundaries, set weights
   567)     ! such that values will just be copied.
   568)     !
   569)     extrap = 0._r8
   570) 
   571)     do j=1,nlatout
   572)        if (yout(j).le.yin(1)) then
   573)           jjm(j) = 1
   574)           jjp(j) = 1
   575)           wgts(j) = 1._r8
   576)           wgtn(j) = 0._r8
   577)           extrap=extrap+1._r8
   578)        else if (yout(j).gt.yin(nlatin)) then
   579)           jjm(j) = nlatin
   580)           jjp(j) = nlatin
   581)           wgts(j) = 1._r8
   582)           wgtn(j) = 0._r8
   583)           extrap=extrap+1._r8
   584)        endif
   585)     end do
   586) 
   587)     !
   588)     ! Loop though output indices finding input indices and weights
   589)     !
   590)     do j=1,nlatout
   591)        do jj=1,nlatin-1
   592)           if (yout(j).gt.yin(jj) .and. yout(j).le.yin(jj+1)) then
   593)              jjm(j) = jj
   594)              jjp(j) = jj + 1
   595)              wgts(j) = (yin(jj+1)-yout(j))/(yin(jj+1)-yin(jj))
   596)              wgtn(j) = (yout(j)-yin(jj))/(yin(jj+1)-yin(jj))
   597)              exit
   598)           end if
   599)        end do
   600)     end do
   601)     !
   602)     ! Check that interp/extrap points have been found for all outputs
   603)     !
   604)     icount = 0
   605)     do j=1,nlatout
   606)        if (jjm(j).eq.0 .or. jjp(j).eq.0) then
   607)           icount = icount + 1
   608)        end if
   609)     end do
   610)     if (icount.gt.0) then
   611)        call endrun('LININTERP: Point found without interp indices')
   612)     end if
   613)     !
   614)     ! Do the interpolation
   615)     !
   616)     do j=1,nlatout
   617)        do k=1,nlev
   618)           arrout(k,j) = arrin(k,jjm(j))*wgts(j) + arrin(k,jjp(j))*wgtn(j)
   619)        end do
   620)     end do
   621) 
   622)     return
   623)   end subroutine lininterp_original
   624)

   626)   subroutine bilin (arrin, xin, yin, nlondin, nlonin, &
   627)        nlevdin, nlev, nlatin, arrout, xout, &
   628)        yout, nlondout, nlonout, nlevdout, nlatout)
   629)     !-----------------------------------------------------------------------
   630)     !
   631)     ! Purpose:
   632)     !
   633)     ! Do a bilinear interpolation from input mesh defined by xin, yin to output
   634)     ! mesh defined by xout, yout.  Circularity is assumed in the x-direction so
   635)     ! input x-grid must be in degrees east and must start from Greenwich.  When
   636)     ! extrapolation is necessary in the N-S direction, values will be copied
   637)     ! from the extreme edge of the input grid.  Vectorization is over the
   638)     ! longitude dimension.  Input grid is assumed rectangular. Output grid
   639)     ! is assumed ragged, i.e. xout is a 2-d mesh.
   640)     !
   641)     ! Method: Interpolate first in longitude, then in latitude.
   642)     !
   643)     ! Author: Jim Rosinski
   644)     !
   645)     !-----------------------------------------------------------------------
   646)     use shr_kind_mod, only: r8 => shr_kind_r8
   647)     use abortutils,   only: endrun
   648)     !-----------------------------------------------------------------------
   649)     implicit none
   650)     !-----------------------------------------------------------------------
   651)     !
   652)     ! Input arguments
   653)     !
   654)     integer, intent(in) :: nlondin                        ! longitude dimension of input grid
   655)     integer, intent(in) :: nlonin                         ! number of real longitudes (input)
   656)     integer, intent(in) :: nlevdin                        ! vertical dimension of input grid
   657)     integer, intent(in) :: nlev                           ! number of vertical levels
   658)     integer, intent(in) :: nlatin                         ! number of input latitudes
   659)     integer, intent(in) :: nlatout                        ! number of output latitudes
   660)     integer, intent(in) :: nlondout                       ! longitude dimension of output grid
   661)     integer, intent(in) :: nlonout(nlatout)               ! number of output longitudes per lat
   662)     integer, intent(in) :: nlevdout                       ! vertical dimension of output grid
   663) 
   664)     real(r8), intent(in) :: arrin(nlondin,nlevdin,nlatin) ! input array of values to interpolate
   665)     real(r8), intent(in) :: xin(nlondin)                  ! input x mesh
   666)     real(r8), intent(in) :: yin(nlatin)                   ! input y mesh
   667)     real(r8), intent(in) :: xout(nlondout,nlatout)        ! output x mesh
   668)     real(r8), intent(in) :: yout(nlatout)                 ! output y mesh
   669)     !
   670)     ! Output arguments
   671)     !
   672)     real(r8), intent(out) :: arrout(nlondout,nlevdout,nlatout) ! interpolated array
   673)     !
   674)     ! Local workspace
   675)     !
   676)     integer :: i, ii, iw, ie, iiprev ! longitude indices
   677)     integer :: j, jj, js, jn, jjprev ! latitude indices
   678)     integer :: k                     ! level index
   679)     integer :: icount                ! number of bad values
   680) 
   681)     real(r8) :: extrap               ! percent grid non-overlap
   682)     real(r8) :: dxinwrap             ! delta-x on input grid for 2-pi
   683)     real(r8) :: avgdxin              ! avg input delta-x
   684)     real(r8) :: ratio                ! compare dxinwrap to avgdxin
   685)     real(r8) :: sum                  ! sum of weights (used for testing)
   686)     !
   687)     ! Dynamic
   688)     !
   689)     integer :: iim(nlondout)         ! interpolation index to the left
   690)     integer :: iip(nlondout)         ! interpolation index to the right
   691)     integer :: jjm(nlatout)          ! interpolation index to the south
   692)     integer :: jjp(nlatout)          ! interpolation index to the north
   693) 
   694)     real(r8) :: wgts(nlatout)        ! interpolation weight to the north
   695)     real(r8) :: wgtn(nlatout)        ! interpolation weight to the north
   696)     real(r8) :: wgte(nlondout)       ! interpolation weight to the north
   697)     real(r8) :: wgtw(nlondout)       ! interpolation weight to the north
   698)     real(r8) :: igrid(nlonin)        ! interpolation weight to the north
   699)     !
   700)     ! Check validity of input coordinate arrays: must be monotonically increasing,
   701)     ! and have a total of at least 2 elements
   702)     !
   703)     if (nlonin < 2 .or. nlatin < 2) then
   704)        call endrun ('BILIN: Must have at least 2 input points for interpolation')
   705)     end if
   706) 
   707)     if (xin(1) < 0._r8 .or. xin(nlonin) > 360._r8) then
   708)        call endrun ('BILIN: Input x-grid must be between 0 and 360')
   709)     end if
   710) 
   711)     icount = 0
   712)     do i=1,nlonin-1
   713)        if (xin(i) >= xin(i+1)) icount = icount + 1
   714)     end do
   715) 
   716)     do j=1,nlatin-1
   717)        if (yin(j) >= yin(j+1)) icount = icount + 1
   718)     end do
   719) 
   720)     do j=1,nlatout-1
   721)        if (yout(j) >= yout(j+1)) icount = icount + 1
   722)     end do
   723) 
   724)     do j=1,nlatout
   725)        do i=1,nlonout(j)-1
   726)           if (xout(i,j) >= xout(i+1,j)) icount = icount + 1
   727)        end do
   728)     end do
   729) 
   730)     if (icount > 0) then
   731)        call endrun ('BILIN: Non-monotonic coordinate array(s) found')
   732)     end if
   733) 
   734)     if (yout(nlatout) <= yin(1) .or. yout(1) >= yin(nlatin)) then
   735)        call endrun ('BILIN: No overlap in y-coordinates')
   736)     end if
   737) 
   738)     do j=1,nlatout
   739)        if (xout(1,j) < 0._r8 .or. xout(nlonout(j),j) > 360._r8) then
   740)           call endrun ('BILIN: Output x-grid must be between 0 and 360')
   741)        end if
   742) 
   743)        if (xout(nlonout(j),j) <= xin(1) .or.  &
   744)             xout(1,j)          >= xin(nlonin)) then
   745)           call endrun ('BILIN: No overlap in x-coordinates')
   746)        end if
   747)     end do
   748)     !
   749)     ! Initialize index arrays for later checking
   750)     !
   751)     do j=1,nlatout
   752)        jjm(j) = 0
   753)        jjp(j) = 0
   754)     end do
   755)     !
   756)     ! For values which extend beyond N and S boundaries, set weights
   757)     ! such that values will just be copied.
   758)     !
   759)     do js=1,nlatout
   760)        if (yout(js) > yin(1)) exit
   761)        jjm(js) = 1
   762)        jjp(js) = 1
   763)        wgts(js) = 1._r8
   764)        wgtn(js) = 0._r8
   765)     end do
   766) 
   767)     do jn=nlatout,1,-1
   768)        if (yout(jn) <= yin(nlatin)) exit
   769)        jjm(jn) = nlatin
   770)        jjp(jn) = nlatin
   771)        wgts(jn) = 1._r8
   772)        wgtn(jn) = 0._r8
   773)     end do
   774)     !
   775)     ! Loop though output indices finding input indices and weights
   776)     !
   777)     jjprev = 1
   778)     do j=js,jn
   779)        do jj=jjprev,nlatin-1
   780)           if (yout(j) > yin(jj) .and. yout(j) <= yin(jj+1)) then
   781)              jjm(j) = jj
   782)              jjp(j) = jj + 1
   783)              wgts(j) = (yin(jj+1) - yout(j)) / (yin(jj+1) - yin(jj))
   784)              wgtn(j) = (yout(j)   - yin(jj)) / (yin(jj+1) - yin(jj))
   785)              goto 30
   786)           end if
   787)        end do
   788)        call endrun ('BILIN: Failed to find interp values')
   789) 30     jjprev = jj
   790)     end do
   791) 
   792)     dxinwrap = xin(1) + 360._r8 - xin(nlonin)
   793)     !
   794)     ! Check for sane dxinwrap values.  Allow to differ no more than 10% from avg
   795)     !
   796)     avgdxin = (xin(nlonin)-xin(1))/(nlonin-1._r8)
   797)     ratio = dxinwrap/avgdxin
   798)     if (ratio < 0.9_r8 .or. ratio > 1.1_r8) then
   799)        write(iulog,*)'BILIN: Insane dxinwrap value =',dxinwrap,' avg=', avgdxin
   800)        call endrun
   801)     end if
   802)     !
   803)     ! Check that interp/extrap points have been found for all outputs, and that
   804)     ! they are reasonable (i.e. within 32-bit roundoff)
   805)     !
   806)     icount = 0
   807)     do j=1,nlatout
   808)        if (jjm(j) == 0 .or. jjp(j) == 0) icount = icount + 1
   809)        sum = wgts(j) + wgtn(j)
   810)        if (sum < 0.99999_r8 .or. sum > 1.00001_r8) icount = icount + 1
   811)        if (wgts(j) < 0._r8 .or. wgts(j) > 1._r8) icount = icount + 1
   812)        if (wgtn(j) < 0._r8 .or. wgtn(j) > 1._r8) icount = icount + 1
   813)     end do
   814) 
   815)     if (icount > 0) then
   816)        call endrun ('BILIN: something bad in latitude indices or weights')
   817)     end if
   818)     !
   819)     ! Do the bilinear interpolation
   820)     !
   821)     do j=1,nlatout
   822)        !
   823)        ! Initialize index arrays for later checking
   824)        !
   825)        do i=1,nlondout
   826)           iim(i) = 0
   827)           iip(i) = 0
   828)        end do
   829)        !
   830)        ! For values which extend beyond E and W boundaries, set weights such that
   831)        ! values will be interpolated between E and W edges of input grid.
   832)        !
   833)        do iw=1,nlonout(j)
   834)           if (xout(iw,j) > xin(1)) exit
   835)           iim(iw) = nlonin
   836)           iip(iw) = 1
   837)           wgtw(iw) = (xin(1)        - xout(iw,j))   /dxinwrap
   838)           wgte(iw) = (xout(iw,j)+360._r8 - xin(nlonin))/dxinwrap
   839)        end do
   840) 
   841)        do ie=nlonout(j),1,-1
   842)           if (xout(ie,j) <= xin(nlonin)) exit
   843)           iim(ie) = nlonin
   844)           iip(ie) = 1
   845)           wgtw(ie) = (xin(1)+360._r8 - xout(ie,j))   /dxinwrap
   846)           wgte(ie) = (xout(ie,j)    - xin(nlonin))/dxinwrap
   847)        end do
   848)        !
   849)        ! Loop though output indices finding input indices and weights
   850)        !
   851)        iiprev = 1
   852)        do i=iw,ie
   853)           do ii=iiprev,nlonin-1
   854)              if (xout(i,j) > xin(ii) .and. xout(i,j) <= xin(ii+1)) then
   855)                 iim(i) = ii
   856)                 iip(i) = ii + 1
   857)                 wgtw(i) = (xin(ii+1) - xout(i,j)) / (xin(ii+1) - xin(ii))
   858)                 wgte(i) = (xout(i,j) - xin(ii))   / (xin(ii+1) - xin(ii))
   859)                 goto 60
   860)              end if
   861)           end do
   862)           call endrun ('BILIN: Failed to find interp values')
   863) 60        iiprev = ii
   864)        end do
   865) 
   866)        icount = 0
   867)        do i=1,nlonout(j)
   868)           if (iim(i) == 0 .or. iip(i) == 0) icount = icount + 1
   869)           sum = wgtw(i) + wgte(i)
   870)           if (sum < 0.99999_r8 .or. sum > 1.00001_r8) icount = icount + 1
   871)           if (wgtw(i) < 0._r8 .or. wgtw(i) > 1._r8) icount = icount + 1
   872)           if (wgte(i) < 0._r8 .or. wgte(i) > 1._r8) icount = icount + 1
   873)        end do
   874) 
   875)        if (icount > 0) then
   876)           write(iulog,*)'BILIN: j=',j,' Something bad in longitude indices or weights'
   877)           call endrun
   878)        end if
   879)        !
   880)        ! Do the interpolation, 1st in longitude then latitude
   881)        !
   882)        do k=1,nlev
   883)           do i=1,nlonin
   884)              igrid(i) = arrin(i,k,jjm(j))*wgts(j) + arrin(i,k,jjp(j))*wgtn(j)
   885)           end do
   886) 
   887)           do i=1,nlonout(j)
   888)              arrout(i,k,j) = igrid(iim(i))*wgtw(i) + igrid(iip(i))*wgte(i)
   889)           end do
   890)        end do
   891)     end do
   892) 
   893) 
   894)     return
   895)   end subroutine bilin

   897)   subroutine vertinterp(ncol, ncold, nlev, pmid, pout, arrin, arrout)
   898) 
   899)     !-----------------------------------------------------------------------
   900)     !
   901)     ! Purpose:
   902)     ! Vertically interpolate input array to output pressure level
   903)     ! Copy values at boundaries.
   904)     !
   905)     ! Method:
   906)     !
   907)     ! Author:
   908)     !
   909)     !-----------------------------------------------------------------------
   910) 
   911)     implicit none
   912) 
   913)     !------------------------------Arguments--------------------------------
   914)     integer , intent(in)  :: ncol              ! column dimension
   915)     integer , intent(in)  :: ncold             ! declared column dimension
   916)     integer , intent(in)  :: nlev              ! vertical dimension
   917)     real(r8), intent(in)  :: pmid(ncold,nlev)  ! input level pressure levels
   918)     real(r8), intent(in)  :: pout              ! output pressure level
   919)     real(r8), intent(in)  :: arrin(ncold,nlev) ! input  array
   920)     real(r8), intent(out) :: arrout(ncold)     ! output array (interpolated)
   921)     !--------------------------------------------------------------------------
   922) 
   923)     !---------------------------Local variables-----------------------------
   924)     integer i,k               ! indices
   925)     integer kupper(ncold)     ! Level indices for interpolation
   926)     real(r8) dpu              ! upper level pressure difference
   927)     real(r8) dpl              ! lower level pressure difference
   928)     logical found(ncold)      ! true if input levels found
   929)     logical error             ! error flag
   930)     !-----------------------------------------------------------------
   931)     !
   932)     ! Initialize index array and logical flags
   933)     !
   934)     do i=1,ncol
   935)        found(i)  = .false.
   936)        kupper(i) = 1
   937)     end do
   938)     error = .false.
   939)     !
   940)     ! Store level indices for interpolation.
   941)     ! If all indices for this level have been found,
   942)     ! do the interpolation
   943)     !
   944)     do k=1,nlev-1
   945)        do i=1,ncol
   946)           if ((.not. found(i)) .and. pmid(i,k)<pout .and. pout<=pmid(i,k+1)) then
   947)              found(i) = .true.
   948)              kupper(i) = k
   949)           end if
   950)        end do
   951)     end do
   952)     !
   953)     ! If we've fallen through the k=1,nlev-1 loop, we cannot interpolate and
   954)     ! must extrapolate from the bottom or top data level for at least some
   955)     ! of the longitude points.
   956)     !
   957)     do i=1,ncol
   958)        if (pout <= pmid(i,1)) then
   959)           arrout(i) = arrin(i,1)
   960)        else if (pout >= pmid(i,nlev)) then
   961)           arrout(i) = arrin(i,nlev)
   962)        else if (found(i)) then
   963)           dpu = pout - pmid(i,kupper(i))
   964)           dpl = pmid(i,kupper(i)+1) - pout
   965)           arrout(i) = (arrin(i,kupper(i)  )*dpl + arrin(i,kupper(i)+1)*dpu)/(dpl + dpu)
   966)        else
   967)           error = .true.
   968)        end if
   969)     end do
   970)     !
   971)     ! Error check
   972)     !
   973)     if (error) then
   974)        call endrun ('VERTINTERP: ERROR FLAG')
   975)     end if
   976) 
   977)     return
   978)   end subroutine vertinterp

   980)   subroutine get_timeinterp_factors (cycflag, np1, cdayminus, cdayplus, cday, &
   981)        fact1, fact2, str)
   982)     !---------------------------------------------------------------------------
   983)     !
   984)     ! Purpose: Determine time interpolation factors (normally for a boundary dataset)
   985)     !          for linear interpolation.
   986)     !
   987)     ! Method:  Assume 365 days per year.  Output variable fact1 will be the weight to
   988)     !          apply to data at calendar time "cdayminus", and fact2 the weight to apply
   989)     !          to data at time "cdayplus".  Combining these values will produce a result
   990)     !          valid at time "cday".  Output arguments fact1 and fact2 will be between
   991)     !          0 and 1, and fact1 + fact2 = 1 to roundoff.
   992)     !
   993)     ! Author:  Jim Rosinski
   994)     !
   995)     !---------------------------------------------------------------------------
   996)     implicit none
   997)     !
   998)     ! Arguments
   999)     !
  1000)     logical, intent(in) :: cycflag             ! flag indicates whether dataset is being cycled yearly
  1001) 
  1002)     integer, intent(in) :: np1                 ! index points to forward time slice matching cdayplus
  1003) 
  1004)     real(r8), intent(in) :: cdayminus          ! calendar day of rearward time slice
  1005)     real(r8), intent(in) :: cdayplus           ! calendar day of forward time slice
  1006)     real(r8), intent(in) :: cday               ! calenar day to be interpolated to
  1007)     real(r8), intent(out) :: fact1             ! time interpolation factor to apply to rearward time slice
  1008)     real(r8), intent(out) :: fact2             ! time interpolation factor to apply to forward time slice
  1009) 
  1010)     character(len=*), intent(in) :: str        ! string to be added to print in case of error (normally the callers name)
  1011)     !
  1012)     ! Local workspace
  1013)     !
  1014)     real(r8) :: deltat                         ! time difference (days) between cdayminus and cdayplus
  1015)     real(r8), parameter :: daysperyear = 365._r8  ! number of days in a year
  1016)     !
  1017)     ! Initial sanity checks
  1018)     !
  1019)     if (np1 == 1 .and. .not. cycflag) then
  1020)        call endrun ('GETFACTORS:'//str//' cycflag false and forward month index = Jan. not allowed')
  1021)     end if
  1022) 
  1023)     if (np1 < 1) then
  1024)        call endrun ('GETFACTORS:'//str//' input arg np1 must be > 0')
  1025)     end if
  1026) 
  1027)     if (cycflag) then
  1028)        if ((cday < 1._r8) .or. (cday > (daysperyear+1._r8))) then
  1029)           write(iulog,*) 'GETFACTORS:', str, ' bad cday=',cday
  1030)           call endrun ()
  1031)        end if
  1032)     else
  1033)        if (cday < 1._r8) then
  1034)           write(iulog,*) 'GETFACTORS:', str, ' bad cday=',cday
  1035)           call endrun ()
  1036)        end if
  1037)     end if
  1038)     !
  1039)     ! Determine time interpolation factors.  Account for December-January
  1040)     ! interpolation if dataset is being cycled yearly.
  1041)     !
  1042)     if (cycflag .and. np1 == 1) then                     ! Dec-Jan interpolation
  1043)        deltat = cdayplus + daysperyear - cdayminus
  1044)        if (cday > cdayplus) then                         ! We are in December
  1045)           fact1 = (cdayplus + daysperyear - cday)/deltat
  1046)           fact2 = (cday - cdayminus)/deltat
  1047)        else                                              ! We are in January
  1048)           fact1 = (cdayplus - cday)/deltat
  1049)           fact2 = (cday + daysperyear - cdayminus)/deltat
  1050)        end if
  1051)     else
