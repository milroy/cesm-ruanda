   201)   subroutine physics_update(state, ptend, dt, tend)
   202) !-----------------------------------------------------------------------
   203) ! Update the state and or tendency structure with the parameterization tendencies
   204) !-----------------------------------------------------------------------
   205)     use shr_sys_mod,  only: shr_sys_flush
   206)     use geopotential, only: geopotential_dse
   207)     use constituents, only: cnst_get_ind, cnst_mw
   208)     use scamMod,      only: scm_crm_mode, single_column
   209)     use phys_control, only: phys_getopts
   210)     use physconst,    only: physconst_update ! Routine which updates physconst variables (WACCM-X)
   211)     use ppgrid,       only: begchunk, endchunk
   212) 
   213) !------------------------------Arguments--------------------------------
   214)     type(physics_ptend), intent(inout)  :: ptend   ! Parameterization tendencies
   215) 
   216)     type(physics_state), intent(inout)  :: state   ! Physics state variables
   217) 
   218)     real(r8), intent(in) :: dt                     ! time step
   219) 
   220)     type(physics_tend ), intent(inout), optional  :: tend  ! Physics tendencies over timestep
   221)                     ! This is usually only needed by calls from physpkg.
   222) !
   223) !---------------------------Local storage-------------------------------
   224)     integer :: i,k,m                               ! column,level,constituent indices
   225)     integer :: ixcldice, ixcldliq                  ! indices for CLDICE and CLDLIQ
   226)     integer :: ixnumice, ixnumliq
   227)     integer :: ncol                                ! number of columns
   228)     character*40 :: name    ! param and tracer name for qneg3
   229) 
   230)     integer :: ixo, ixo2, ixh, ixh2, ixn    ! indices for O, O2, H2, and N
   231) 
   232)     real(r8) :: zvirv(state%psetcols,pver)  ! Local zvir array pointer
   233) 
   234)     real(r8),allocatable :: cpairv_loc(:,:,:)
   235)     real(r8),allocatable :: rairv_loc(:,:,:)
   236) 
   237)     ! PERGRO limits cldliq/ice for macro/microphysics:
   238)     character(len=24), parameter :: pergro_cldlim_names(4) = &
   239)          (/ "stratiform", "cldwat    ", "micro_mg  ", "macro_park" /)
   240) 
   241)     ! cldliq/ice limits that are always on.
   242)     character(len=24), parameter :: cldlim_names(2) = &
   243)          (/ "convect_deep", "zm_conv_tend" /)
   244) 
   245)     ! Whether to do validation of state on each call.
   246)     logical :: state_debug_checks
   247) 
   248)     !-----------------------------------------------------------------------
   249) 
   250)     ! The column radiation model does not update the state
   251)     if(single_column.and.scm_crm_mode) return
   252) 
   253) 
   254)     !-----------------------------------------------------------------------
   255)     ! If no fields are set, then return
   256)     if (.not. (any(ptend%lq(:)) .or. ptend%ls .or. ptend%lu .or. ptend%lv)) then
   257)        ptend%name  = "none"
   258)        ptend%psetcols = 0
   259)        return
   260)     end if
   261) 
   262)     !-----------------------------------------------------------------------
   263)     ! Check that the state/tend/ptend are all dimensioned with the same number of columns
   264)     if (state%psetcols /= ptend%psetcols) then
   265)        call endrun('ERROR in physics_update with ptend%name='//trim(ptend%name) &
   266)             //': state and ptend must have the same number of psetcols.')
   267)     end if
   268) 
   269)     if (present(tend)) then
   270)        if (state%psetcols /= tend%psetcols) then
   271)           call endrun('ERROR in physics_update with ptend%name='//trim(ptend%name) &
   272)                //': state and tend must have the same number of psetcols.')
   273)        end if
   274)     end if
   275) 
   276)     !-----------------------------------------------------------------------
   277)     ! cpairv_loc and rairv_loc need to be allocated to a size which matches state and ptend
   278)     ! If psetcols == pcols, the cpairv is the correct size and just copy
   279)     ! If psetcols > pcols and all cpairv match cpair, then assign the constant cpair
   280)     if (state%psetcols == pcols) then
   281)        allocate (cpairv_loc(state%psetcols,pver,begchunk:endchunk))
   282)        cpairv_loc(:,:,:) = cpairv(:,:,:)
   283)     else if (state%psetcols > pcols .and. all(cpairv(:,:,:) == cpair)) then
   284)        allocate(cpairv_loc(state%psetcols,pver,begchunk:endchunk))
   285)        cpairv_loc(:,:,:) = cpair
   286)     else
   287)        call endrun('physics_update: cpairv is not allowed to vary when subcolumns are turned on')
   288)     end if
   289)     if (state%psetcols == pcols) then
   290)        allocate (rairv_loc(state%psetcols,pver,begchunk:endchunk))
   291)        rairv_loc(:,:,:) = rairv(:,:,:)
   292)     else if (state%psetcols > pcols .and. all(rairv(:,:,:) == rair)) then
   293)        allocate(rairv_loc(state%psetcols,pver,begchunk:endchunk))
   294)        rairv_loc(:,:,:) = rair
   295)     else
   296)        call endrun('physics_update: rairv_loc is not allowed to vary when subcolumns are turned on')
   297)     end if
   298) 
   299)     !-----------------------------------------------------------------------
   300)     call phys_getopts(state_debug_checks_out=state_debug_checks)
   301) 
   302)     ncol = state%ncol
   303) 
   304)     ! Update u,v fields
   305)     if(ptend%lu) then
   306)        do k = ptend%top_level, ptend%bot_level
   307)           state%u  (:ncol,k) = state%u  (:ncol,k) + ptend%u(:ncol,k) * dt
   308)           if (present(tend)) &
   309)                tend%dudt(:ncol,k) = tend%dudt(:ncol,k) + ptend%u(:ncol,k)
   310)        end do
   311)     end if
   312) 
   313)     if(ptend%lv) then
   314)        do k = ptend%top_level, ptend%bot_level
   315)           state%v  (:ncol,k) = state%v  (:ncol,k) + ptend%v(:ncol,k) * dt
   316)           if (present(tend)) &
   317)                tend%dvdt(:ncol,k) = tend%dvdt(:ncol,k) + ptend%v(:ncol,k)
   318)        end do
   319)     end if
   320) 
   321)    ! Update constituents, all schemes use time split q: no tendency kept
   322)     call cnst_get_ind('CLDICE', ixcldice, abort=.false.)
   323)     call cnst_get_ind('CLDLIQ', ixcldliq, abort=.false.)
   324)     ! Check for number concentration of cloud liquid and cloud ice (if not present
   325)     ! the indices will be set to -1)
   326)     call cnst_get_ind('NUMICE', ixnumice, abort=.false.)
   327)     call cnst_get_ind('NUMLIQ', ixnumliq, abort=.false.)
   328)   
   329)     do m = 1, pcnst
   330)        if(ptend%lq(m)) then
   331)           do k = ptend%top_level, ptend%bot_level
   332)              state%q(:ncol,k,m) = state%q(:ncol,k,m) + ptend%q(:ncol,k,m) * dt
   333)           end do
   334) 
   335)           ! now test for mixing ratios which are too small
   336)           ! don't call qneg3 for number concentration variables
   337)           if (m /= ixnumice  .and.  m /= ixnumliq) then
   338)              name = trim(ptend%name) // '/' // trim(cnst_name(m))
   339)              call qneg3(trim(name), state%lchnk, ncol, state%psetcols, pver, m, m, qmin(m), state%q(1,1,m))
   340)           else
   341)              do k = ptend%top_level, ptend%bot_level
   342)                 ! checks for number concentration
   343)                 state%q(:ncol,k,m) = max(1.e-12_r8,state%q(:ncol,k,m))
   344)                 state%q(:ncol,k,m) = min(1.e10_r8,state%q(:ncol,k,m))
   345)              end do
   346)           end if
   347) 
   348)        end if
   349) 
   350)     end do
   351) 
   352)     !------------------------------------------------------------------------
   353)     ! This is a temporary fix for the large H, H2 in WACCM-X
   354)     ! Well, it was supposed to be temporary, but it has been here
   355)     ! for a while now.
   356)     !------------------------------------------------------------------------
   357)     if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then
   358)        call cnst_get_ind('H', ixh)
   359)        do k = ptend%top_level, ptend%bot_level
   360)           state%q(:ncol,k,ixh) = min(state%q(:ncol,k,ixh), 0.01_r8)
   361)        end do
   362) 
   363)        call cnst_get_ind('H2', ixh2)
   364)        do k = ptend%top_level, ptend%bot_level
   365)           state%q(:ncol,k,ixh2) = min(state%q(:ncol,k,ixh2), 6.e-5_r8)
   366)        end do
   367)     endif
   368) 
   369)     ! Special tests for cloud liquid and ice:
   370)     ! Enforce a minimum non-zero value.
   371)     if (ixcldliq > 1) then
   372)        if(ptend%lq(ixcldliq)) then
   373) #ifdef PERGRO
   374)           if ( any(ptend%name == pergro_cldlim_names) ) &
   375)                call state_cnst_min_nz(1.e-12_r8, ixcldliq, ixnumliq)
   376) #endif
   377)           if ( any(ptend%name == cldlim_names) ) &
   378)                call state_cnst_min_nz(1.e-36_r8, ixcldliq, ixnumliq)
   379)        end if
   380)     end if
   381) 
   382)     if (ixcldice > 1) then
   383)        if(ptend%lq(ixcldice)) then
   384) #ifdef PERGRO
   385)           if ( any(ptend%name == pergro_cldlim_names) ) &
   386)                call state_cnst_min_nz(1.e-12_r8, ixcldice, ixnumice)
   387) #endif
   388)           if ( any(ptend%name == cldlim_names) ) &
   389)                call state_cnst_min_nz(1.e-36_r8, ixcldice, ixnumice)
   390)        end if
   391)     end if
   392) 
   393)     !------------------------------------------------------------------------
   394)     ! Get indices for molecular weights and call WACCM-X physconst_update
   395)     !------------------------------------------------------------------------
   396)     if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   397)       call cnst_get_ind('O', ixo)
   398)       call cnst_get_ind('O2', ixo2)
   399)       call cnst_get_ind('N', ixn)             
   400) 
   401)       call physconst_update(state%q, state%t, &
   402) 	         cnst_mw(ixo), cnst_mw(ixo2), cnst_mw(ixh), cnst_mw(ixn), &
   403) 	                              ixo, ixo2, ixh, pcnst, state%lchnk, ncol)
   404)     endif	  
   405)    
   406)     if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
   407)       zvirv(:,:) = shr_const_rwv / rairv_loc(:,:,state%lchnk) - 1._r8
   408)     else
   409)       zvirv(:,:) = zvir    
   410)     endif
   411) 
   412)     !-------------------------------------------------------------------------------------------
   413)     ! Update dry static energy(moved from above for WACCM-X so updating after cpairv_loc update)
   414)     !-------------------------------------------------------------------------------------------
   415)     if(ptend%ls) then
   416)        do k = ptend%top_level, ptend%bot_level
   417)           state%s(:ncol,k)   = state%s(:ncol,k)   + ptend%s(:ncol,k) * dt
   418)           if (present(tend)) &
   419)                tend%dtdt(:ncol,k) = tend%dtdt(:ncol,k) + ptend%s(:ncol,k)/cpairv_loc(:ncol,k,state%lchnk)
   420)        end do
   421)     end if
   422) 
   423)     ! Derive new temperature and geopotential fields if heating or water tendency not 0.
   424)     if (ptend%ls .or. ptend%lq(1)) then
   425)        call geopotential_dse(  &
   426)             state%lnpint, state%lnpmid, state%pint  , state%pmid  , state%pdel  , state%rpdel  , &
   427)             state%s     , state%q(:,:,1),state%phis , rairv_loc(:,:,state%lchnk), gravit  , cpairv_loc(:,:,state%lchnk), &
   428)             zvirv    , state%t     , state%zi    , state%zm    , ncol         )
   429)     end if
   430) 
   431)     ! Good idea to do this regularly.
   432)     call shr_sys_flush(iulog)
   433) 
   434)     if (state_debug_checks) call physics_state_check(state, ptend%name)
   435) 
   436)     deallocate(cpairv_loc, rairv_loc)
   437) 
   438)     ! Deallocate ptend
   439)     call physics_ptend_dealloc(ptend)
   440) 
   441)     ptend%name  = "none"
   442)     ptend%lq(:) = .false.
   443)     ptend%ls    = .false.
   444)     ptend%lu    = .false.
   445)     ptend%lv    = .false.
   446)     ptend%psetcols = 0
   447) 
   448)   contains

   450)     subroutine state_cnst_min_nz(lim, qix, numix)
   451)       ! Small utility function for setting minimum nonzero
   452)       ! constituent concentrations.
   453) 
   454)       ! Lower limit and constituent index
   455)       real(r8), intent(in) :: lim
   456)       integer,  intent(in) :: qix
   457)       ! Number concentration that goes with qix.
   458)       ! Ignored if <= 0 (and therefore constituent is not present).
   459)       integer,  intent(in) :: numix
   460) 
   461)       if (numix > 0) then
   462)          ! Where q is too small, zero mass and number
   463)          ! concentration.
   464)          where (state%q(:ncol,:,qix) < lim)
   465)             state%q(:ncol,:,qix) = 0._r8
   466)             state%q(:ncol,:,numix) = 0._r8
   467)          end where
   468)       else
   469)          ! If no number index, just do mass.
   470)           where (state%q(:ncol,:,qix) < lim)
   471)              state%q(:ncol,:,qix) = 0._r8
   472)           end where
   473)       end if
   474) 
   475)     end subroutine state_cnst_min_nz

   672)   subroutine physics_ptend_sum(ptend, ptend_sum, ncol)
   673) !-----------------------------------------------------------------------
   674) ! Add ptend fields to ptend_sum for ptend logical flags = .true.
   675) ! Where ptend logical flags = .false, don't change ptend_sum
   676) !-----------------------------------------------------------------------
   677) 
   678) !------------------------------Arguments--------------------------------
   679)     type(physics_ptend), intent(in)     :: ptend   ! New parameterization tendencies
   680)     type(physics_ptend), intent(inout)  :: ptend_sum   ! Sum of incoming ptend_sum and ptend
   681)     integer, intent(in)                 :: ncol    ! number of columns
   682) 
   683) !---------------------------Local storage-------------------------------
   684)     integer :: i,k,m                               ! column,level,constituent indices
   685)     integer :: psetcols                            ! maximum number of columns
   686)     integer :: ierr = 0
   687) 
   688) !-----------------------------------------------------------------------
   689)     if (ptend%psetcols /= ptend_sum%psetcols) then
   690)        call endrun('physics_ptend_sum error: ptend and ptend_sum must have the same value for psetcols')
   691)     end if
   692)       
   693)     if (ncol > ptend_sum%psetcols) then
   694)        call endrun('physics_ptend_sum error: ncol must be less than or equal to psetcols')
   695)     end if
   696)     
   697)     psetcols = ptend_sum%psetcols
   698)       
   699)     ptend_sum%top_level = ptend%top_level
   700)     ptend_sum%bot_level = ptend%bot_level
   701) 
   702) ! Update u,v fields
   703)     if(ptend%lu) then
   704)        if (.not. allocated(ptend_sum%u)) then 
   705)           allocate(ptend_sum%u(psetcols,pver), stat=ierr)
   706)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%u')
   707)           ptend_sum%u=0.0_r8
   708) 
   709)           allocate(ptend_sum%taux_srf(psetcols), stat=ierr)
   710)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%taux_srf')
   711)           ptend_sum%taux_srf=0.0_r8
   712) 
   713)           allocate(ptend_sum%taux_top(psetcols), stat=ierr)
   714)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%taux_top')
   715)           ptend_sum%taux_top=0.0_r8
   716)        end if
   717)        ptend_sum%lu = .true.
   718) 
   719)        do k = ptend%top_level, ptend%bot_level
   720)           do i = 1, ncol
   721)              ptend_sum%u(i,k) = ptend_sum%u(i,k) + ptend%u(i,k)
   722)           end do
   723)        end do
   724)        do i = 1, ncol
   725)           ptend_sum%taux_srf(i) = ptend_sum%taux_srf(i) + ptend%taux_srf(i)
   726)           ptend_sum%taux_top(i) = ptend_sum%taux_top(i) + ptend%taux_top(i)
   727)        end do
   728)     end if
   729) 
   730)     if(ptend%lv) then
   731)        if (.not. allocated(ptend_sum%v)) then 
   732)           allocate(ptend_sum%v(psetcols,pver), stat=ierr)
   733)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%v')
   734)           ptend_sum%v=0.0_r8
   735) 
   736)           allocate(ptend_sum%tauy_srf(psetcols), stat=ierr)
   737)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%tauy_srf')
   738)           ptend_sum%tauy_srf=0.0_r8
   739) 
   740)           allocate(ptend_sum%tauy_top(psetcols), stat=ierr)
   741)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%tauy_top')
   742)           ptend_sum%tauy_top=0.0_r8
   743)        end if
   744)        ptend_sum%lv = .true.
   745) 
   746)        do k = ptend%top_level, ptend%bot_level
   747)           do i = 1, ncol
   748)              ptend_sum%v(i,k) = ptend_sum%v(i,k) + ptend%v(i,k)
   749)           end do
   750)        end do
   751)        do i = 1, ncol
   752)           ptend_sum%tauy_srf(i) = ptend_sum%tauy_srf(i) + ptend%tauy_srf(i)
   753)           ptend_sum%tauy_top(i) = ptend_sum%tauy_top(i) + ptend%tauy_top(i)
   754)        end do
   755)     end if
   756) 
   757) 
   758)     if(ptend%ls) then
   759)        if (.not. allocated(ptend_sum%s)) then 
   760)           allocate(ptend_sum%s(psetcols,pver), stat=ierr)
   761)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%s')
   762)           ptend_sum%s=0.0_r8
   763) 
   764)           allocate(ptend_sum%hflux_srf(psetcols), stat=ierr)
   765)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%hflux_srf')
   766)           ptend_sum%hflux_srf=0.0_r8
   767) 
   768)           allocate(ptend_sum%hflux_top(psetcols), stat=ierr)
   769)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%hflux_top')
   770)           ptend_sum%hflux_top=0.0_r8
   771)        end if
   772)        ptend_sum%ls = .true.
   773) 
   774)        do k = ptend%top_level, ptend%bot_level
   775)           do i = 1, ncol
   776)              ptend_sum%s(i,k) = ptend_sum%s(i,k) + ptend%s(i,k)
   777)           end do
   778)        end do
   779)        do i = 1, ncol
   780)           ptend_sum%hflux_srf(i) = ptend_sum%hflux_srf(i) + ptend%hflux_srf(i)
   781)           ptend_sum%hflux_top(i) = ptend_sum%hflux_top(i) + ptend%hflux_top(i)
   782)        end do
   783)     end if
   784) 
   785)     if (any(ptend%lq(:))) then
   786) 
   787)        if (.not. allocated(ptend_sum%q)) then
   788)           allocate(ptend_sum%q(psetcols,pver,pcnst), stat=ierr)
   789)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%q')
   790)           ptend_sum%q=0.0_r8
   791) 
   792)           allocate(ptend_sum%cflx_srf(psetcols,pcnst), stat=ierr)
   793)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%cflx_srf')
   794)           ptend_sum%cflx_srf=0.0_r8
   795)   
   796)           allocate(ptend_sum%cflx_top(psetcols,pcnst), stat=ierr)
   797)           if ( ierr /= 0 ) call endrun('physics_ptend_sum error: allocation error for ptend_sum%cflx_top')
   798)           ptend_sum%cflx_top=0.0_r8
   799)        end if
   800) 
   801)        do m = 1, pcnst
   802)           if(ptend%lq(m)) then
   803)              ptend_sum%lq(m) = .true.
   804)              do k = ptend%top_level, ptend%bot_level
   805)                 do i = 1,ncol
   806)                    ptend_sum%q(i,k,m) = ptend_sum%q(i,k,m) + ptend%q(i,k,m)
   807)                 end do
   808)              end do
   809)              do i = 1,ncol
   810)                 ptend_sum%cflx_srf(i,m) = ptend_sum%cflx_srf(i,m) + ptend%cflx_srf(i,m)
   811)                 ptend_sum%cflx_top(i,m) = ptend_sum%cflx_top(i,m) + ptend%cflx_top(i,m)
   812)              end do
   813)           end if
   814)        end do
   815) 
   816)     end if
   817) 
   818)   end subroutine physics_ptend_sum

  1070)   subroutine physics_dme_adjust(state, tend, qini, dt)
  1071)     !----------------------------------------------------------------------- 
  1072)     ! 
  1073)     ! Purpose: Adjust the dry mass in each layer back to the value of physics input state
  1074)     ! 
  1075)     ! Method: Conserve the integrated mass, momentum and total energy in each layer
  1076)     !         by scaling the specific mass of consituents, specific momentum (velocity)
  1077)     !         and specific total energy by the relative change in layer mass. Solve for
  1078)     !         the new temperature by subtracting the new kinetic energy from total energy
  1079)     !         and inverting the hydrostatic equation
  1080)     !
  1081)     !         The mass in each layer is modified, changing the relationship of the layer 
  1082)     !         interfaces and midpoints to the surface pressure. The result is no longer in 
  1083)     !         the original hybrid coordinate. 
  1084)     !
  1085)     !         This procedure cannot be applied to the "eul" or "sld" dycores because they
  1086)     !         require the hybrid coordinate.
  1087)     ! 
  1088)     ! Author: Byron Boville
  1089) 
  1090)     ! !REVISION HISTORY:
  1091)     !   03.03.28  Boville    Created, partly from code by Lin in p_d_adjust
  1092)     ! 
  1093)     !-----------------------------------------------------------------------
  1094) 
  1095)     use constituents, only : cnst_get_type_byind
  1096) 
  1097)     implicit none
  1098)     !
  1099)     ! Arguments
  1100)     !
  1101)     type(physics_state), intent(inout) :: state
  1102)     type(physics_tend ), intent(inout) :: tend
  1103)     real(r8),            intent(in   ) :: qini(pcols,pver)    ! initial specific humidity
  1104)     real(r8),            intent(in   ) :: dt                  ! model physics timestep
  1105)     !
  1106)     !---------------------------Local workspace-----------------------------
  1107)     !
  1108)     integer  :: lchnk         ! chunk identifier
  1109)     integer  :: ncol          ! number of atmospheric columns
  1110)     integer  :: i,k,m         ! Longitude, level indices
  1111)     real(r8) :: fdq(pcols)    ! mass adjustment factor
  1112)     real(r8) :: te(pcols)     ! total energy in a layer
  1113)     real(r8) :: utmp(pcols)   ! temp variable for recalculating the initial u values
  1114)     real(r8) :: vtmp(pcols)   ! temp variable for recalculating the initial v values
  1115) 
  1116)     real(r8) :: zvirv(pcols,pver)    ! Local zvir array pointer
  1117)     !
  1118)     !-----------------------------------------------------------------------
  1119)     ! verify that the dycore is FV
  1120)     if (.not. dycore_is('LR') ) return
  1121) 
  1122)     if (state%psetcols .ne. pcols) then
  1123)        call endrun('physics_dme_adjust: cannot pass in a state which has sub-columns')
  1124)     end if
  1125) 
  1126)     lchnk = state%lchnk
  1127)     ncol  = state%ncol
  1128) 
  1129)     ! adjust dry mass in each layer back to input value, while conserving
  1130)     ! constituents, momentum, and total energy
  1131)     do k = 1, pver
  1132) 
  1133)        ! adjusment factor is just change in water vapor
  1134)        fdq(:ncol) = 1._r8 + state%q(:ncol,k,1) - qini(:ncol,k)
  1135) 
  1136)        ! adjust constituents to conserve mass in each layer
  1137)        do m = 1, pcnst
  1138)           state%q(:ncol,k,m) = state%q(:ncol,k,m) / fdq(:ncol)
  1139)        end do
  1140) 
  1141)        if (adjust_te) then
  1142)           ! compute specific total energy of unadjusted state (J/kg)
  1143)           te(:ncol) = state%s(:ncol,k) + 0.5_r8*(state%u(:ncol,k)**2 + state%v(:ncol,k)**2) 
  1144) 
  1145)           ! recompute initial u,v from the new values and the tendencies
  1146)           utmp(:ncol) = state%u(:ncol,k) - dt * tend%dudt(:ncol,k)
  1147)           vtmp(:ncol) = state%v(:ncol,k) - dt * tend%dvdt(:ncol,k)
  1148)           ! adjust specific total energy and specific momentum (velocity) to conserve each
  1149)           te     (:ncol)   = te     (:ncol)     / fdq(:ncol)
  1150)           state%u(:ncol,k) = state%u(:ncol,k  ) / fdq(:ncol)
  1151)           state%v(:ncol,k) = state%v(:ncol,k  ) / fdq(:ncol)
  1152)           ! compute adjusted u,v tendencies
  1153)           tend%dudt(:ncol,k) = (state%u(:ncol,k) - utmp(:ncol)) / dt
  1154)           tend%dvdt(:ncol,k) = (state%v(:ncol,k) - vtmp(:ncol)) / dt
  1155) 
  1156)           ! compute adjusted static energy
  1157)           state%s(:ncol,k) = te(:ncol) - 0.5_r8*(state%u(:ncol,k)**2 + state%v(:ncol,k)**2)
  1158)        end if
  1159) 
  1160) ! compute new total pressure variables
  1161)        state%pdel  (:ncol,k  ) = state%pdel(:ncol,k  ) * fdq(:ncol)
  1162)        state%pint  (:ncol,k+1) = state%pint(:ncol,k  ) + state%pdel(:ncol,k)
  1163)        state%lnpint(:ncol,k+1) = log(state%pint(:ncol,k+1))
  1164)        state%rpdel (:ncol,k  ) = 1._r8/ state%pdel(:ncol,k  )
  1165)     end do
  1166) 
  1167)     if ( waccmx_is('ionosphere') .or. waccmx_is('neutral') ) then 
  1168)       zvirv(:,:) = shr_const_rwv / rairv(:,:,state%lchnk) - 1._r8
  1169)     else
  1170)       zvirv(:,:) = zvir    
  1171)     endif
  1172) 
  1173) ! compute new T,z from new s,q,dp
  1174)     if (adjust_te) then
  1175)        call geopotential_dse(state%lnpint, state%lnpmid, state%pint,  &
  1176)             state%pmid  , state%pdel    , state%rpdel,  &
  1177)             state%s     , state%q(:,:,1), state%phis , rairv(:,:,state%lchnk), &
  1178) 	    gravit, cpairv(:,:,state%lchnk), zvirv, &
  1179)             state%t     , state%zi      , state%zm   , ncol)
  1180)     end if
  1181) 
  1182)   end subroutine physics_dme_adjust
