   147) 
   148) ! !ARGUMENTS:
   149)       type(ESMF_ClockInt), intent(out) :: clockint
   150)       type(ESMF_TimeInterval), intent(in), optional :: TimeStep
   151)       type(ESMF_Time), intent(in) :: StartTime

   154)       integer, intent(out), optional :: rc

   158) ! !DESCRIPTION:
   159) !     Initialize an {\tt ESMF\_Clock}
   160) !     
   161) !     The arguments are:
   162) !     \begin{description}
   163) !     \item[clockint]
   164) !          The object instance to initialize
   165) !     \item[{[TimeStep]}]
   166) !          The {\tt ESMF\_Clock}'s time step interval
   167) !     \item[StartTime]
   168) !          The {\tt ESMF\_Clock}'s starting time
   169) !     \item[StopTime]
   170) !          The {\tt ESMF\_Clock}'s stopping time
   171) !     \item[{[RefTime]}]
   172) !          The {\tt ESMF\_Clock}'s reference time
   173) !     \item[{[rc]}]
   174) !          Return code; equals {\tt ESMF\_SUCCESS} if there are no errors.
   175) !     \end{description}
   176) !     
   177) ! !REQUIREMENTS:
   178) !     TMG3.1, TMG3.4.4
   179) !EOP
   180)       IF ( PRESENT(TimeStep) ) clockint%TimeStep = TimeStep
   181)       IF ( PRESENT(RefTime) )THEN
   182)          clockint%RefTime = RefTime
   183)       ELSE
   184)          clockint%RefTime = StartTime
   185)       END IF
   186)       clockint%CurrTime = StartTime
   187)       clockint%StartTime = StartTime
   188)       clockint%StopTime = StopTime
   189)       clockint%NumAlarms = 0
   190)       clockint%AdvanceCount = 0
   191)       ALLOCATE(clockint%AlarmList(MAX_ALARMS))
   192)       ! TBH:  This incredible hack can be removed once ESMF_*Validate() 
   193)       ! TBH:  can tell if a deep ESMF_* was created or not.  
   194)       DO i = 1, MAX_ALARMS
   195)         NULLIFY( clockint%AlarmList( i )%alarmint )
   196)       ENDDO
   197)       IF ( PRESENT( rc ) ) rc = ESMF_SUCCESS
   198)     
   199)       end subroutine ESMF_ClockSetOLD

   906) ! !IROUTINE: ESMF_ClockAdvance - Advance a clock's current time by one time step
   907) 
   908) ! !INTERFACE:
   909)       subroutine ESMF_ClockAdvance(clock, RingingAlarmList, &
   910)                                    NumRingingAlarms, rc)
   911) 
   912) use ESMF_TimeMod
   913) 
   914) ! !ARGUMENTS:
   915)       type(ESMF_Clock), intent(inout) :: clock
   916)       type(ESMF_Alarm), dimension(MAX_ALARMS), intent(out), optional :: &
   917)                                         RingingAlarmList
   918)       integer, intent(out), optional :: NumRingingAlarms
   919)       integer, intent(out), optional :: rc
   920) ! Local
   921)       logical pred1, pred2, pred3
   922)       integer i, n
   923)       type(ESMF_Alarm) :: alarm
   924) !   
   925) ! !DESCRIPTION:
   926) !     Advance an {\tt ESMF\_Clock}'s current time by one time step
   927) !  
   928) !     The arguments are:
   929) !     \begin{description}
   930) !     \item[clock]
   931) !          The object instance to advance
   932) !     \item[{[RingingAlarmList]}]
   933) !          Return a list of any ringing alarms after the time step
   934) !     \item[{[NumRingingAlarms]}]
   935) !          The number of ringing alarms returned
   936) !     \item[{[rc]}]
   937) !          Return code; equals {\tt ESMF\_SUCCESS} if there are no errors.
   938) !     \end{description}
   939) !  
   940) ! !REQUIREMENTS:
   941) !     TMG3.4.1
   942) !EOP
   943)       clock%clockint%CurrTime = clock%clockint%CurrTime + &
   944)                                 clock%clockint%TimeStep
   945) 
   946)       IF ( Present(NumRingingAlarms) ) NumRingingAlarms = 0
   947)       clock%clockint%AdvanceCount = clock%clockint%AdvanceCount + 1
   948)       DO i = 1, MAX_ALARMS
   949)         alarm = clock%clockint%AlarmList(i)
   950)         ! TBH:  This is really dangerous.  We need to be able to NULLIFY 
   951)         ! TBH:  alarmint at compile-time (F95 synax) to make this safe.  
   952) !$$$TBH:  see if F95 compile-time pointer-nullification is supported by all 
   953) !$$$TBH:  compilers we support
   954)         IF ( ASSOCIATED( alarm%alarmint ) ) THEN
   955)           IF ( alarm%alarmint%Enabled ) THEN
   956)             IF ( alarm%alarmint%RingIntervalSet ) THEN
   957)               pred1 = .FALSE. ; pred2 = .FALSE. ; pred3 = .FALSE.
   958)               IF ( alarm%alarmint%StopTimeSet ) THEN
   959)                 PRED1 = clock%clockint%CurrTime > alarm%alarmint%StopTime
   960)               ENDIF
   961)               IF ( alarm%alarmint%RingTimeSet ) THEN
   962)                  PRED2 = ( alarm%alarmint%RingTime <= clock%clockint%CurrTime     &
   963)                         .AND. clock%clockint%CurrTime < alarm%alarmint%RingTime + &
   964)                               clock%clockint%TimeStep )
   965)               ENDIF
   966)               IF ( alarm%alarmint%RingIntervalSet ) THEN
   967)                  PRED3 = ( alarm%alarmint%PrevRingTime + alarm%alarmint%RingInterval <= &
   968)                            clock%clockint%CurrTime )
   969)               ENDIF
   970)               IF ( ( .NOT. ( pred1 ) ) .AND. &
   971)                    ( ( pred2 ) .OR. ( pred3 ) ) ) THEN
   972)                  alarm%alarmint%Ringing = .TRUE.
   973)                  IF ( PRED3) alarm%alarmint%PrevRingTime = alarm%alarmint%PrevRingTime + &
   974)                                                   alarm%alarmint%RingInterval
   975)                  IF ( PRESENT( RingingAlarmList ) .AND. &
   976)                       PRESENT ( NumRingingAlarms ) ) THEN
   977)                    NumRingingAlarms = NumRingingAlarms + 1
   978)                    RingingAlarmList( NumRingingAlarms ) = alarm
   979)                  ENDIF
   980)               ENDIF
   981)             ELSE IF ( alarm%alarmint%RingTimeSet ) THEN
   982)               IF ( alarm%alarmint%RingTime <= clock%clockint%CurrTime ) THEN
   983)                  alarm%alarmint%Ringing = .TRUE.
   984)                  IF ( PRESENT( RingingAlarmList ) .AND. &
   985)                       PRESENT ( NumRingingAlarms ) ) THEN
   986)                    NumRingingAlarms = NumRingingAlarms + 1
   987)                    RingingAlarmList( NumRingingAlarms ) = alarm
   988)                  ENDIF
   989)               ENDIF
   990)             ENDIF
   991)             IF ( alarm%alarmint%StopTimeSet ) THEN
   992)             ENDIF
   993)           ENDIF
   994)         ENDIF
   995)         clock%clockint%AlarmList(i) = alarm
   996)       ENDDO
   997)       IF ( PRESENT( rc ) ) rc = ESMF_SUCCESS
   998)     
   999)       end subroutine ESMF_ClockAdvance
