   116)   subroutine ghg_chem_set_rates( rxn_rates, latmapback, zen_angle, ncol, lchnk )
   117)     use chem_mods,    only : rxntot
   118)     use chem_mods,    only : gas_pcnst
   119)     use cam_history,  only : outfld
   120)     use mo_constants, only : pi
   121) 
   122)     implicit none

   124)     integer,  intent(in)    :: ncol                           ! number columns in chunk
   125)     real(r8), intent(inout) :: rxn_rates(ncol,pver,rxntot)   ! ghg loss rates
   126)     integer,  intent(in)    :: latmapback(pcols)
   127)     real(r8), intent(in)    :: zen_angle(ncol)
   128)     integer,  intent(in)    :: lchnk                          ! chunk index
   129) 
   130)     integer :: i,k
   131)     real(r8), parameter :: half_pi = pi/2._r8 
   132) 
   133)     do k=1,pver-2
   134)        do i=1,ncol
   135)           rxn_rates(i,k,n2o_rxt)   = chemdata%datainst(latmapback(i),k,lchnk,1)
   136)           rxn_rates(i,k,ch4_rxt)   = chemdata%datainst(latmapback(i),k,lchnk,2)
   137)           rxn_rates(i,k,cfc11_rxt) = chemdata%datainst(latmapback(i),k,lchnk,3)
   138)           rxn_rates(i,k,cfc12_rxt) = chemdata%datainst(latmapback(i),k,lchnk,4)
   139)        enddo
   140)     enddo
   141) 
   142)     rxn_rates(:ncol,pver-1:pver,n2o_rxt)   = 0._r8
   143)     rxn_rates(:ncol,pver-1:pver,ch4_rxt)   = 0._r8
   144)     rxn_rates(:ncol,pver-1:pver,cfc11_rxt) = 0._r8
   145)     rxn_rates(:ncol,pver-1:pver,cfc12_rxt) = 0._r8
   146) 
   147)     call outfld( 'GHG_CFC11_R', rxn_rates(:ncol,:,cfc11_rxt), ncol, lchnk )
   148)     call outfld( 'GHG_CFC12_R', rxn_rates(:ncol,:,cfc12_rxt), ncol, lchnk )
   149)     call outfld( 'GHG_N2O_R',   rxn_rates(:ncol,:,n2o_rxt),   ncol, lchnk )
   150)     call outfld( 'GHG_CH4_R',   rxn_rates(:ncol,:,ch4_rxt),   ncol, lchnk )
   151)     
   152)     if (lyman_alpha_rxt > 0) then
   153)        rxn_rates(:ncol,:,lyman_alpha_rxt) = 0._r8 
   154)     endif
   155) 
   156)     if (lyman_alpha) then
   157)        do i=1,ncol
   158)           if (zen_angle(i) < half_pi) then
   159)              rxn_rates(i,:,lyman_alpha_rxt) = h2orate_data%datainst(latmapback(i),:,lchnk,1)
   160)           endif
   161)        enddo
   162) 
   163)        call outfld( 'GHG_H2O_R', rxn_rates(:ncol,:,lyman_alpha_rxt), ncol, lchnk )
   164)     endif
