    20)   subroutine xy_interp_init(im1,jm1,lon0,lat0,im2,jm2,weight_x,weight_y)
    21) !------------------------------------------------------------------------------------------------------------
    22) ! This program computes weighting functions to map a variable of (im1,jm1) resolution to (im2,jm2) resolution
    23) ! weight_x(im2,im1) is the weighting function for zonal interpolation
    24) ! weight_y(jm2,jm1) is the weighting function for meridional interpolation
    25) ! 
    26) ! Author: Chih-Chieh (Jack) Chen  -- May 2010
    27) !
    28) !------------------------------------------------------------------------------------------------------------
    29)   implicit none
    30)   integer,  intent(in)  :: im1, jm1, im2, jm2
    31)   real(r8), intent(in)  :: lon0(im1), lat0(jm1)
    32)   real(r8), intent(out) :: weight_x(im2,im1), weight_y(jm2,jm1)
    33) 
    34)   real(r8) :: lon1(im1), lat1(jm1)
    35)   real(r8) :: lon2(im2), lat2(jm2)
    36)   real(r8) :: slon1(im1+1), slon2(im2+1), slat1(jm1+1), slat2(jm2+1)  
    37)   real(r8) :: x1_west, x1_east, x2_west, x2_east
    38)   real(r8) :: y1_south, y1_north, y2_south, y2_north
    39)   integer  :: i1, j1, i2, j2
    40) 
    41)   weight_x(:,:) = 0.0_r8
    42)   weight_y(:,:) = 0.0_r8
    43) 
    44) ! lon0 & lat0 are longitude & latitude on the source mesh in radians
    45) ! convert lon1, lat1 from radians to degrees
    46)   lon1(:) = lon0(:)/SHR_CONST_PI*180.0_r8
    47)   lat1(:) = lat0(:)/SHR_CONST_PI*180.0_r8
    48) 
    49) ! set up lon2, lat2 (target mesh), in CAM convention
    50)   do i2=1,im2
    51)    lon2(i2) = (float(i2)-1.0_r8)*360.0_r8/float(im2)
    52)   enddo
    53)   do j2=1,jm2
    54)    lat2(j2) = -90.0_r8+(float(j2)-1.0_r8)*180.0_r8/(float(jm2)-1.0_r8)
    55)   enddo
    56) 
    57) 
    58) ! set up staggered longitudes (cell edges in x)
    59)   do i1=2,im1
    60) 	slon1(i1) = (lon1(i1-1)+lon1(i1))/2.0_r8
    61)   enddo
    62)   slon1(1) = lon1(1)-(lon1(2)-lon1(1))/2.0_r8
    63)   slon1(im1+1) = lon1(im1)+(lon1(im1)-lon1(im1-1))/2.0_r8
    64) 
    65)   do i2=2,im2
    66) 	slon2(i2) = (lon2(i2-1)+lon2(i2))/2.0_r8
    67)   enddo
    68)   slon2(1) = lon2(1)-(lon2(2)-lon2(1))/2.0_r8
    69)   slon2(im2+1) = lon2(im2)+(lon2(im2)-lon2(im2-1))/2.0_r8
    70) 
    71) ! set up staggered lattiudes (cell edges in y)
    72)   slat1(1)=-90.0_r8
    73)   do j1=2,jm1
    74) 	slat1(j1) = (lat1(j1-1)+lat1(j1))/2.0_r8
    75)   enddo
    76)   slat1(jm1+1)=90.0_r8
    77) 
    78)   slat2(1)=-90.0_r8
    79)   do j2=2,jm2
    80) 	slat2(j2)=(lat2(j2-1)+lat2(j2))/2.0_r8
    81)   enddo	
    82)   slat2(jm2+1)=90.0_r8
    83) 
    84) ! compute Guassian weight for two meshes (discrete form of cos(lat).)
    85)   do j1=1,jm1
    86)    	gw1(j1) = sin(slat1(j1+1)/180.0_r8*SHR_CONST_PI)-sin(slat1(j1)/180.0_r8*SHR_CONST_PI)
    87)   enddo
    88) 
    89)   do j2=1,jm2
    90) 	gw2(j2) = sin(slat2(j2+1)/180.0_r8*SHR_CONST_PI)-sin(slat2(j2)/180.0_r8*SHR_CONST_PI)
    91)   enddo
    92) 
    93) 
    94) ! add 360 to slon1 and slon2 
    95)   slon1(:) = slon1(:)+360.0_r8
    96)   slon2(:) = slon2(:)+360.0_r8
    97) 
    98)  do i2=1,im2
    99) 
   100) ! target grid east-west boundaries
   101)   x2_west=slon2(i2)
   102)   x2_east=slon2(i2+1)
   103) 
   104)   do i1=1,im1
   105) 
   106) ! source grid east-west boundaries
   107)    x1_west=slon1(i1)
   108)    x1_east=slon1(i1+1)
   109) 
   110) ! check if there is any overlap between the source grid and the target grid
   111) ! if no overlap, then weighting is zero
   112) ! there are three scenarios overlaps can take place 
   113)    if( (x1_west.ge.x2_west).and.(x1_east.le.x2_east) ) then
   114) ! case 1: 
   115) !                x1_west             x1_east
   116) !                  |-------------------|
   117) !            |---------------------------------|
   118) !          x2_west                           x2_east
   119)      weight_x(i2,i1) =  (x1_east-x1_west)/(x2_east-x2_west)
   120)    elseif ( (x1_west.ge.x2_west).and.(x1_west.lt.x2_east) ) then
   121) ! case 2: 
   122) !                x1_west                          x1_east
   123) !                  |--------------------------------|
   124) !            |---------------------------------|
   125) !          x2_west                           x2_east
   126)      weight_x(i2,i1) = (x2_east-x1_west)/(x2_east-x2_west)
   127)    elseif ( (x1_east>x2_west).and.(x1_east.le.x2_east) ) then
   128) ! case 3: 
   129) !       x1_west                          x1_east
   130) !         |--------------------------------|
   131) !                |---------------------------------|
   132) !              x2_west                           x2_east
   133)      weight_x(i2,i1) = (x1_east-x2_west)/(x2_east-x2_west)
   134)    endif
   135) 
   136)    enddo	
   137)   enddo
   138) 
   139) 
   140) ! consider end points
   141)       if(slon1(im1+1).gt.slon2(im2+1)) then
   142) ! case 1:
   143) !           slon1(im1)                slon1(im1+1) <--- end point
   144) !              |-------------------------|
   145) !           |----------------|......................|
   146) !        slon2(im2)         slon2(im2+1)        slon2(2)  (note: slon2(im2+1) = slon2(1))
   147)      	weight_x(1,im1)= weight_x(1,im1)+(slon1(im1+1)-slon2(im2+1))/(slon2(2)-slon2(1))
   148)       endif	
   149) 
   150)       if(slon1(im1+1).lt.slon2(im2+1)) then
   151) ! case 1:
   152) !           slon1(im1)                slon1(im1+1)                  slon1(2)    (note: slon1(im1+1) = slon1(1))
   153) !              |-------------------------|.............................|
   154) !                   |-------------------------------|
   155) !               slon2(im2)                        slon2(im2+1) <--- end point
   156)         weight_x(im2,1) = weight_x(im2,1)+(slon2(1)-slon1(1))/(slon2(2)-slon2(1)) 
   157)       endif
   158) 
   159) 
   160) 
   161)       do j2=1,jm2
   162) ! target grid north-south boundaries
   163)          y2_south=slat2(j2)
   164)          y2_north=slat2(j2+1)
   165) 
   166)          do j1=1,jm1
   167) 
   168) ! source grid north-south boundaries
   169)             y1_south=slat1(j1)
   170)             y1_north=slat1(j1+1)
   171)  
   172) ! check if there is any overlap between the source grid and the target grid
   173) ! if no overlap, then weighting is zero
   174) ! there are three scenarios overlaps can take place 
   175) ! note: there is Guassian weight to consider in the meridional direction!
   176) 
   177)             if( (y1_south.ge.y2_south).and.(y1_north.le.y2_north) ) then
   178) ! case 1: 
   179) !                y1_south             y1_north
   180) !                  |-------------------|
   181) !            |---------------------------------|
   182) !          y2_south                           y2_north
   183)                  weight_y(j2,j1) =  gw1(j1)/gw2(j2)
   184)             elseif ( (y1_south.ge.y2_south).and.(y1_south.lt.y2_north) ) then
   185) ! case 2: 
   186) !                y1_south                          y1_north
   187) !                  |--------------------------------|
   188) !            |---------------------------------|
   189) !          y2_south                           y2_north
   190)                  weight_y(j2,j1) = (y2_north-y1_south)/(y1_north-y1_south)*gw1(j1)/gw2(j2)
   191)             elseif ( (y1_north.gt.y2_south).and.(y1_north.le.y2_north) ) then
   192) ! case 3: 
   193) !       y1_south                          y1_north
   194) !         |--------------------------------|
   195) !                |---------------------------------|
   196) !              y2_south                           y2_north
   197)                  weight_y(j2,j1) = (y1_north-y2_south)/(y1_north-y1_south)*gw1(j1)/gw2(j2)
   198)             endif
   199) 
   200)           enddo
   201)         enddo
   202) 
   203)   end subroutine xy_interp_init

   205)   subroutine xy_interp(im1,jm1,km1,im2,jm2,pcols,ncols,weight_x,weight_y,var_src,var_trg,lons,lats,count_x,count_y,index_x,index_y)
   206) !-------------------------------------------------------------------------------------------------------------
   207) ! This program interpolates var_src(im1,jm1,km1) to var_trg(im2,jm2,km1) based on weighting functions weight_x & weight_y.
   208) !-------------------------------------------------------------------------------------------------------------
   209)   implicit none
   210)   integer,  intent(in)  :: im1   ! source number of longitudes
   211)   integer,  intent(in)  :: jm1   ! source number of latitudes
   212)   integer,  intent(in)  :: km1   ! source/target number of levels
   213)   integer,  intent(in)  :: im2   ! target number of longitudes
   214)   integer,  intent(in)  :: jm2   ! target number of latitudes
   215)   integer,  intent(in)  :: pcols
   216)   integer,  intent(in)  :: ncols
   217)   real(r8), intent(in)  :: weight_x(im2,im1), weight_y(jm2,jm1)
   218)   real(r8), intent(in)  :: var_src(im1,jm1,km1)
   219)   integer,  intent(in)  :: lons(pcols), lats(pcols)
   220)   integer,  intent(in)  :: count_x(im2), count_y(jm2)
   221)   integer,  intent(in)  :: index_x(im2,im1), index_y(jm2,jm1)
   222)   real(r8), intent(out) :: var_trg(pcols,km1)
   223)   integer  :: n, i1, j1, k1, i2, j2, ii, jj
   224)   real(r8) :: sum_x
   225) 
   226)   var_trg(:,:) = 0.0_r8
   227)  
   228) 
   229)   do k1=1,km1
   230)      do n=1,ncols
   231) ! interpolate in x
   232)         do jj=1,count_y(lats(n))
   233)            sum_x = 0.0_r8
   234)            do ii=1,count_x(lons(n))
   235)               sum_x = sum_x + var_src(index_x(lons(n),ii),index_y(lats(n),jj),k1)* &
   236)                              weight_x(lons(n),index_x(lons(n),ii))
   237)            enddo
   238)            var_trg(n,k1) = var_trg(n,k1)+sum_x*weight_y(lats(n),index_y(lats(n),jj))
   239)         enddo
   240)      enddo
   241)   enddo
   242) 
   243)   end subroutine xy_interp
