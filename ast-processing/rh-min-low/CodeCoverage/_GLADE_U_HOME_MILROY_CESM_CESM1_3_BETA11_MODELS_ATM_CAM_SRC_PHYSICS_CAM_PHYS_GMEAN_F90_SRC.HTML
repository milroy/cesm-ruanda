     4) ! Purpose: 
     5) ! Perform mixed layer global calculations for energy conservation checks.
     6) !
     7) ! Methods: 
     8) ! Reproducible (nonscalable): 
     9) !    Gather to a master processor who does all the work.
    10) ! Reproducible (scalable): 
    11) !    Convert to fixed point (integer representation) to enable
    12) !    reproducibility when using MPI collectives. Results compared with
    13) !    a nonreproducible (but scalable) algorithm using floating point
    14) !    and MPI_Allreduce to verify the results are good enough.
    15) !
    16) ! Author: Byron Boville from SOM code by Jim Rosinski/Bruce Briegleb
    17) ! Modified: P. Worley to aggregate calculations (4/04)
    18) ! Modified: J. White/P. Worley to introduce scalable algorithms;
    19) !           B. Eaton to remove dycore-specific dependencies and to 
    20) !           introduce gmean_mass (10/07)
    21) ! Modified: P. Worley to replace in-place implementation with call
    22) !           to repro_sum.
    23) !

    56)    subroutine gmean_arr (arr, arr_gmean, nflds)
    57) !----------------------------------------------------------------------- 
    58) ! 
    59) ! Purpose: 
    60) ! Compute the global mean of each field in "arr" in the physics 
    61) ! chunked decomposition
    62) ! 
    63) !-----------------------------------------------------------------------
    64) !
    65) ! Arguments
    66) !
    67)       integer, intent(in)  :: nflds                 ! number of fields
    68)       real(r8), intent(in) :: arr(pcols,begchunk:endchunk,nflds) 
    69)                                                     ! Input array, chunked
    70)       real(r8), intent(out):: arr_gmean(nflds)      ! global means
    71) !
    72) ! Local workspace
    73) !
    74)       real(r8) :: rel_diff(2,nflds)                 ! relative differences between 
    75)                                                     !  'fast' reproducible and 
    76)                                                     !  nonreproducible means
    77)       integer  :: ifld                              ! field index
    78)       logical  :: write_warning
    79) !
    80) !-----------------------------------------------------------------------
    81) !
    82)       call t_startf ('gmean_fixed_repro')
    83)       call gmean_fixed_repro(arr, arr_gmean, rel_diff, nflds)
    84)       call t_stopf ('gmean_fixed_repro')
    85) 
    86)       ! check that "fast" reproducible sum is accurate enough. If not, calculate
    87)       ! using old method
    88)       write_warning = masterproc
    89)       if ( shr_reprosum_tolExceeded('gmean', nflds, write_warning, &
    90)                                   iulog, rel_diff) ) then
    91)          if ( shr_reprosum_recompute ) then
    92)             do ifld=1,nflds
    93)                if ( rel_diff(1,ifld) > shr_reprosum_reldiffmax ) then
    94)                   call t_startf ('gmean_float_repro')
    95)                   call gmean_float_repro(arr(:,:,ifld), arr_gmean(ifld), 1)
    96)                   call t_stopf ('gmean_float_repro')
    97)                endif
    98)             enddo
    99)          endif
   100)       endif
   101) 
   102)       return
   103)    end subroutine gmean_arr

   146)    subroutine gmean_mass(title, state)
   147) !-----------------------------------------------------------------------
   148) !
   149) ! Purpose:
   150) ! Computes global mean mass, max and min mmr, of constituents on the 
   151) ! physics decomposition. Prints diagnostics to log file.
   152) !
   153) ! Author: B. Eaton (based on gavglook)
   154) !
   155) !-----------------------------------------------------------------------
   156)       use ppgrid,         only: pver
   157)       use physconst,      only: gravit
   158)       use phys_grid,      only: get_ncols_p
   159)       use physics_types,  only: physics_state
   160)       use constituents,   only: pcnst, cnst_name
   161) !
   162) ! Arguments
   163) !
   164)       character(len=*),    intent(in) :: title    ! location of this call
   165)       type(physics_state), intent(in) :: state(begchunk:endchunk)
   166) !
   167) ! Local workspace
   168) !
   169)       character(len=*), parameter :: sub_name='gmean_mass: '
   170) 
   171)       integer :: c, i, k, m
   172)       integer :: ierr
   173)       integer :: ncols
   174) 
   175)       real(r8), pointer :: mass_wet(:,:,:) ! constituent masses assuming moist mmr
   176)       real(r8), pointer :: mass_dry(:,:,:) ! constituent masses assuming dry mmr
   177)       real(r8) :: mass_wet_mean(pcnst)     ! global mean constituent masses assuming moist mmr
   178)       real(r8) :: mass_dry_mean(pcnst)     ! global mean constituent masses assuming dry mmr
   179)       real(r8) :: mmr_max(pcnst)           ! maximum constituent mmr in this process
   180)       real(r8) :: mmr_min(pcnst)           ! minimum constituent mmr in this process
   181)       real(r8) :: mmr_max_glob(pcnst)      ! global maximum constituent mmr
   182)       real(r8) :: mmr_min_glob(pcnst)      ! global minimum constituent mmr
   183) !
   184) !-----------------------------------------------------------------------
   185) !
   186)       allocate(mass_wet(pcols,begchunk:endchunk,pcnst), stat=ierr)
   187)       if (ierr /= 0) write(iulog,*) sub_name // 'FAIL to allocate mass_wet'
   188) 
   189)       allocate(mass_dry(pcols,begchunk:endchunk,pcnst), stat=ierr)
   190)       if (ierr /= 0) write(iulog,*) sub_name // 'FAIL to allocate mass_wet'
   191) 
   192)       mmr_max(:) = -1.e36_r8
   193)       mmr_min(:) =  1.e36_r8
   194)       do m = 1, pcnst
   195)          do c = begchunk, endchunk
   196)             ncols = get_ncols_p(c)
   197)             do i = 1, ncols
   198) 
   199)                ! Compute column masses assuming both dry and wet mixing ratios
   200) 
   201)                mass_wet(i,c,m) = 0.0_r8
   202)                do k = 1, pver
   203)                   mass_wet(i,c,m) = mass_wet(i,c,m) + &
   204)                                     state(c)%pdel(i,k)*state(c)%q(i,k,m)
   205)                   mmr_max(m) = max(mmr_max(m), state(c)%q(i,k,m))
   206)                   mmr_min(m) = min(mmr_min(m), state(c)%q(i,k,m))
   207)                end do
   208)                mass_wet(i,c,m) = mass_wet(i,c,m)/gravit
   209) 
   210)                mass_dry(i,c,m) = 0.0_r8
   211)                do k = 1, pver
   212)                   mass_dry(i,c,m) = mass_dry(i,c,m) + &
   213)                                     state(c)%pdeldry(i,k)*state(c)%q(i,k,m)
   214)                end do
   215)                mass_dry(i,c,m) = mass_dry(i,c,m)/gravit
   216) 
   217)             end do
   218)          end do
   219)       end do
   220) 
   221)       ! compute global mean mass
   222)       call gmean(mass_wet, mass_wet_mean, pcnst)
   223)       call gmean(mass_dry, mass_dry_mean, pcnst)
   224) 
   225)       ! global min/max mmr
   226) #ifdef SPMD
   227)       call mpi_reduce(mmr_max, mmr_max_glob, pcnst, mpir8, MPI_MAX, 0, mpicom, ierr)
   228)       call mpi_reduce(mmr_min, mmr_min_glob, pcnst, mpir8, MPI_MIN, 0, mpicom, ierr)
   229) #else
   230)       mmr_max_glob(:) = mmr_max(:)
   231)       mmr_min_glob(:) = mmr_min(:)
   232) #endif
   233) 
   234)       ! report to log file
   235)       if (masterproc) then
   236) 
   237)          do m = 1, pcnst
   238)                write (6,66) trim(title)//' m=',m, &
   239)                   'name='//trim(cnst_name(m))//' gavg dry, wet, min, max ', &
   240)                   mass_dry_mean(m), mass_wet_mean(m), mmr_min_glob(m), mmr_max_glob(m)
   241) 66             format (a24,i2,a36,1p,4e25.13)
   242)          end do
   243) 
   244)       endif
   245) 
   246)       deallocate(mass_wet)
   247)       deallocate(mass_dry)
   248) 
   249)    end subroutine gmean_mass

   255)    subroutine gmean_float_repro (arr, arr_gmean, nflds)
   256) !----------------------------------------------------------------------- 
   257) ! 
   258) ! Purpose: 
   259) ! Compute the global mean of each field in "arr" in the physics 
   260) ! chunked decomposition - all work is done on the masterproc to avoid
   261) ! order of operations differences and assure bfb reproducibility.
   262) ! 
   263) !-----------------------------------------------------------------------
   264) 
   265)       use rgrid,        only: nlon
   266)       use dycore,       only: dycore_is
   267)       use phys_grid,    only: gather_chunk_to_field
   268)       use dyn_grid,     only: get_horiz_grid_dim_d, get_horiz_grid_d, get_dyn_grid_parm_real1d
   269) !
   270) ! Arguments
   271) !
   272)       integer, intent(in)  :: nflds               ! number of fields
   273)       real(r8), intent(in) :: &
   274)          arr(pcols,begchunk:endchunk,nflds)       ! Input array, chunked
   275)       real(r8), intent(out):: arr_gmean(nflds)    ! global means
   276) !
   277) ! Local workspace
   278) !
   279)       real(r8), pointer :: w(:)
   280)       real(r8) :: zmean                       ! zonal mean value
   281)       real(r8) :: tmean                       ! temp global mean value
   282)       integer :: i, j, ifld, n                ! longitude, latitude, field, 
   283)                                               !  and global column indices
   284)       integer :: hdim1, hdim2                 ! dimensions of rectangular horizontal 
   285)                                               !  grid data structure, If 1D data 
   286)                                               !  structure, then hdim2_d == 1.
   287)       integer :: ngcols                       ! global column count (all)
   288) 
   289)       ! rectangular version of arr
   290)       real(r8), allocatable :: arr_field(:,:,:)   
   291) 
   292)       ! column integration weight (from dynamics)
   293)       real(r8), dimension(:), allocatable :: wght_d 
   294) 
   295) !
   296) !-----------------------------------------------------------------------
   297) !
   298)       call get_horiz_grid_dim_d(hdim1, hdim2)
   299)       allocate(arr_field(hdim1,hdim2,nflds))
   300) 
   301)       arr_field(:,:,:) = 0.0_r8
   302)       call gather_chunk_to_field (1, 1, nflds, hdim1, arr, arr_field)
   303) 
   304)       if (masterproc) then
   305) 
   306)          if (dycore_is('UNSTRUCTURED')) then
   307) 
   308)             ngcols = hdim1*hdim2
   309)             allocate ( wght_d(1:ngcols) )
   310) 
   311)             wght_d = 0.0_r8
   312)             call get_horiz_grid_d(ngcols, wght_d_out=wght_d)
   313) 
   314)             do ifld=1,nflds
   315)                arr_gmean(ifld) = 0._r8
   316)                do j=1,hdim2
   317)                   do i=1,hdim1
   318)                      n = (j-1)*hdim1 + i
   319)                      arr_gmean(ifld) = arr_gmean(ifld) + &
   320)                                        arr_field(i,j,ifld)*wght_d(n)
   321)                   end do
   322)                end do
   323)                arr_gmean(ifld) = arr_gmean(ifld) / (4.0_r8 * pi)
   324)             end do
   325) 
   326)             deallocate ( wght_d )
   327) 
   328)          else
   329)             w => get_dyn_grid_parm_real1d('w')
   330)             do ifld=1,nflds
   331)                tmean = 0._r8
   332)                do j=1,hdim2
   333)                   zmean = 0._r8
   334)                   do i=1,hdim1
   335)                      zmean = zmean + arr_field(i,j,ifld)
   336)                   end do
   337)                   tmean = tmean + zmean * 0.5_r8*w(j)/nlon(j)
   338)                end do
   339)                arr_gmean(ifld) = tmean
   340)             end do
   341) 
   342)          end if
   343) 
   344)       end if
   345) 
   346) #if ( defined SPMD )
   347)       call mpibcast (arr_gmean, nflds, mpir8, 0, mpicom)
   348) #endif
   349)       deallocate(arr_field)
   350) 
   351)       return
   352) 
   353)    end subroutine gmean_float_repro

   358)    subroutine gmean_fixed_repro (arr, arr_gmean, rel_diff, nflds)
   359) !----------------------------------------------------------------------- 
   360) ! 
   361) ! Purpose: 
   362) ! Compute the global mean of each field in "arr" in the physics 
   363) ! chunked decomposition with a reproducible yet scalable implementation
   364) ! based on a fixed-point algorithm.
   365) !
   366) !-----------------------------------------------------------------------
   367)       use phys_grid, only    : get_ncols_p, get_wght_all_p, ngcols_p, &
   368)                                get_nlcols_p
   369) !
   370) ! Arguments
   371) !
   372)       integer, intent(in)  :: nflds             ! number of fields
   373)       real(r8), intent(in) :: &
   374)          arr(pcols,begchunk:endchunk,nflds)     ! Input array, chunked
   375)       real(r8), intent(out):: arr_gmean(nflds)  ! global means
   376)       real(r8), intent(out):: rel_diff(2,nflds) ! relative and absolute
   377)                                                 !  differences between 
   378)                                                 !  reproducible and nonreproducible
   379)                                                 !  means
   380) !
   381) ! Local workspace
   382) !
   383)       integer :: lchnk, i, ifld                 ! chunk, column, field indices
   384)       integer :: ncols                          ! number of columns in current chunk
   385)       integer :: count                          ! summand count
   386)       integer :: ierr                           ! MPI error return
   387) #if (! defined SPMD)
   388)       integer  :: mpicom = 0
   389) #endif
   390)       
   391)       real(r8) :: wght(pcols)                   ! column for integration weights
   392)       real(r8), allocatable :: xfld(:,:)        ! weighted summands
   393)       integer :: nlcols
   394) !
   395) !-----------------------------------------------------------------------
   396) !
   397)       nlcols = get_nlcols_p()
   398)       allocate(xfld(nlcols, nflds))
   399) 
   400) ! pre-weight summands
   401)       do ifld=1,nflds
   402)          count = 0
   403)          do lchnk=begchunk,endchunk
   404)             ncols = get_ncols_p(lchnk)
   405)             call get_wght_all_p(lchnk, ncols, wght)
   406)             do i=1,ncols
   407)                count = count + 1
   408)                xfld(count,ifld) = arr(i,lchnk,ifld)*wght(i)
   409)             end do
   410)          end do
   411)       end do
   412) 
   413) ! call fixed-point algorithm
   414)       call shr_reprosum_calc (xfld, arr_gmean, count, nlcols, nflds, &
   415)                       gbl_count=ngcols_p, commid=mpicom, rel_diff=rel_diff) 
   416) 
   417)       deallocate(xfld)
   418) ! final normalization
   419)       arr_gmean(:) = arr_gmean(:) / (4.0_r8 * pi)
   420) 
   421)       return
   422) 
   423)    end subroutine gmean_fixed_repro
