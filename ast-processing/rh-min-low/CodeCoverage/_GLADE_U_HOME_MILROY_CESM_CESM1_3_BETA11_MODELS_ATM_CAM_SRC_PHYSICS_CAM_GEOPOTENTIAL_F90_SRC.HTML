     4) !---------------------------------------------------------------------------------
     5) ! Compute geopotential from temperature or
     6) ! compute geopotential and temperature from dry static energy.
     7) !
     8) ! The hydrostatic matrix elements must be consistent with the dynamics algorithm.
     9) ! The diagonal element is the itegration weight from interface k+1 to midpoint k.
    10) ! The offdiagonal element is the weight between interfaces.
    11) ! 
    12) ! Author: B.Boville, Feb 2001 from earlier code by Boville and S.J. Lin
    13) !---------------------------------------------------------------------------------
    14) 
    15)   use shr_kind_mod, only: r8 => shr_kind_r8
    16)   use ppgrid,       only: pver, pverp
    17)   use dycore,       only: dycore_is
    18) 
    19)   implicit none
    20)   private
    21)   save
    22) 
    23)   public geopotential_dse
    24)   public geopotential_t

    28)   subroutine geopotential_dse(                                &
    29)        piln   , pmln   , pint   , pmid   , pdel   , rpdel  ,  &
    30)        dse    , q      , phis   , rair   , gravit , cpair  ,  &
    31)        zvir   , t      , zi     , zm     , ncol             )
    32) !----------------------------------------------------------------------- 
    33) ! 
    34) ! Purpose: 
    35) ! Compute the temperature  and geopotential height (above the surface) at the
    36) ! midpoints and interfaces from the input dry static energy and pressures.
    37) !
    38) !-----------------------------------------------------------------------
    39) !------------------------------Arguments--------------------------------
    40) !
    41) ! Input arguments
    42)     integer, intent(in) :: ncol                  ! Number of longitudes
    43) 
    44)     ! rair, and cpair are passed in as slices of rank 3 arrays allocated
    45)     ! at runtime. Don't specify size to avoid temporary copy.
    46)     real(r8), intent(in) :: piln (:,:)    ! (pcols,pverp) - Log interface pressures
    47)     real(r8), intent(in) :: pmln (:,:)    ! (pcols,pver)  - Log midpoint pressures
    48)     real(r8), intent(in) :: pint (:,:)    ! (pcols,pverp) - Interface pressures
    49)     real(r8), intent(in) :: pmid (:,:)    ! (pcols,pver)  - Midpoint pressures
    50)     real(r8), intent(in) :: pdel (:,:)    ! (pcols,pver)  - layer thickness
    51)     real(r8), intent(in) :: rpdel(:,:)    ! (pcols,pver)  - inverse of layer thickness
    52)     real(r8), intent(in) :: dse  (:,:)    ! (pcols,pver)  - dry static energy
    53)     real(r8), intent(in) :: q    (:,:)    ! (pcols,pver)  - specific humidity
    54)     real(r8), intent(in) :: phis (:)      ! (pcols)       - surface geopotential
    55)     real(r8), intent(in) :: rair (:,:)    !               - Gas constant for dry air
    56)     real(r8), intent(in) :: gravit        !               - Acceleration of gravity
    57)     real(r8), intent(in) :: cpair(:,:)    !               - specific heat at constant p for dry air
    58)     real(r8), intent(in) :: zvir (:,:)    ! (pcols,pver)  - rh2o/rair - 1
    59) 
    60) ! Output arguments
    61) 
    62)     real(r8), intent(out) :: t(:,:)       ! (pcols,pver)  - temperature
    63)     real(r8), intent(out) :: zi(:,:)      ! (pcols,pverp) - Height above surface at interfaces
    64)     real(r8), intent(out) :: zm(:,:)      ! (pcols,pver)  - Geopotential height at mid level
    65) !
    66) !---------------------------Local variables-----------------------------------------
    67) !
    68)     logical  :: fvdyn                   ! finite volume dynamics
    69)     integer  :: i,k                     ! Lon, level, level indices
    70)     real(r8) :: hkk(ncol)               ! diagonal element of hydrostatic matrix
    71)     real(r8) :: hkl(ncol)               ! off-diagonal element
    72)     real(r8) :: rog(ncol,pver)          ! Rair / gravit
    73)     real(r8) :: tv                      ! virtual temperature
    74)     real(r8) :: tvfac                   ! Tv/T
    75) !
    76) !----------------------------------------------------------------------------------
    77)     rog(:ncol,:) = rair(:ncol,:) / gravit
    78) 
    79) ! Set dynamics flag
    80)     fvdyn = dycore_is ('LR')
    81) 
    82) ! The surface height is zero by definition.
    83)     do i = 1,ncol
    84)        zi(i,pverp) = 0.0_r8
    85)     end do
    86) 
    87) ! Compute the virtual temperature, zi, zm from bottom up
    88) ! Note, zi(i,k) is the interface above zm(i,k)
    89)     do k = pver, 1, -1
    90) 
    91) ! First set hydrostatic elements consistent with dynamics
    92)        if (fvdyn) then
    93)           do i = 1,ncol
    94)              hkl(i) = piln(i,k+1) - piln(i,k)
    95)              hkk(i) = 1._r8 - pint(i,k) * hkl(i) * rpdel(i,k)
    96)           end do
    97)        else
    98)           do i = 1,ncol
    99)              hkl(i) = pdel(i,k) / pmid(i,k)
   100)              hkk(i) = 0.5_r8 * hkl(i)
   101)           end do
   102)        end if
   103) 
   104) ! Now compute tv, t, zm, zi
   105)        do i = 1,ncol
   106)           tvfac   = 1._r8 + zvir(i,k) * q(i,k)
   107)           tv      = (dse(i,k) - phis(i) - gravit*zi(i,k+1)) / ((cpair(i,k) / tvfac) + &
   108) 	                                                               rair(i,k)*hkk(i))
   109) 
   110)           t (i,k) = tv / tvfac
   111) 
   112)           zm(i,k) = zi(i,k+1) + rog(i,k) * tv * hkk(i)
   113)           zi(i,k) = zi(i,k+1) + rog(i,k) * tv * hkl(i)
   114)        end do
   115)     end do
   116) 
   117)     return
   118)   end subroutine geopotential_dse

   127) ! 
   128) ! Purpose: 
   129) ! Compute the geopotential height (above the surface) at the midpoints and 
   130) ! interfaces using the input temperatures and pressures.
   131) !
   132) !-----------------------------------------------------------------------
   133) 
   134) use ppgrid, only : pcols
   135) 
   136) !------------------------------Arguments--------------------------------
   137) !
   138) ! Input arguments
   139) !
   140)     integer, intent(in) :: ncol                  ! Number of longitudes
   141) 
   142)     real(r8), intent(in) :: piln (:,:)    ! (pcols,pverp) - Log interface pressures
   143)     real(r8), intent(in) :: pmln (:,:)    ! (pcols,pver)  - Log midpoint pressures
   144)     real(r8), intent(in) :: pint (:,:)    ! (pcols,pverp) - Interface pressures
   145)     real(r8), intent(in) :: pmid (:,:)    ! (pcols,pver)  - Midpoint pressures
   146)     real(r8), intent(in) :: pdel (:,:)    ! (pcols,pver)  - layer thickness
   147)     real(r8), intent(in) :: rpdel(:,:)    ! (pcols,pver)  - inverse of layer thickness
   148)     real(r8), intent(in) :: t    (:,:)    ! (pcols,pver)  - temperature
   149)     real(r8), intent(in) :: q    (:,:)    ! (pcols,pver)  - specific humidity
   150)     real(r8), intent(in) :: rair (:,:)    ! (pcols,pver)  - Gas constant for dry air
   151)     real(r8), intent(in) :: gravit        !               - Acceleration of gravity
   152)     real(r8), intent(in) :: zvir (:,:)    ! (pcols,pver)  - rh2o/rair - 1
   153) 
   154) ! Output arguments
   155) 
   156)     real(r8), intent(out) :: zi(:,:)      ! (pcols,pverp) - Height above surface at interfaces
   157)     real(r8), intent(out) :: zm(:,:)      ! (pcols,pver)  - Geopotential height at mid level
   158) !
   159) !---------------------------Local variables-----------------------------
   160) !
   161)     logical  :: fvdyn                   ! finite volume dynamics
   162)     integer  :: i,k                     ! Lon, level indices
   163)     real(r8) :: hkk(ncol)               ! diagonal element of hydrostatic matrix
   164)     real(r8) :: hkl(ncol)               ! off-diagonal element
   165)     real(r8) :: rog(ncol,pver)          ! Rair / gravit
   166)     real(r8) :: tv                      ! virtual temperature
   167)     real(r8) :: tvfac                   ! Tv/T
   168) !
   169) !-----------------------------------------------------------------------
   170) !
   171)     rog(:ncol,:) = rair(:ncol,:) / gravit
   172) 
   173) ! Set dynamics flag
   174) 
   175)     fvdyn = dycore_is ('LR')
   176) 
   177) ! The surface height is zero by definition.
   178) 
   179)     do i = 1,ncol
   180)        zi(i,pverp) = 0.0_r8
   181)     end do
   182) 
   183) ! Compute zi, zm from bottom up. 
   184) ! Note, zi(i,k) is the interface above zm(i,k)
   185) 
   186)     do k = pver, 1, -1
   187) 
   188) ! First set hydrostatic elements consistent with dynamics
   189) 
   190)        if (fvdyn) then
   191)           do i = 1,ncol
   192)              hkl(i) = piln(i,k+1) - piln(i,k)
   193)              hkk(i) = 1._r8 - pint(i,k) * hkl(i) * rpdel(i,k)
   194)           end do
   195)        else
   196)           do i = 1,ncol
   197)              hkl(i) = pdel(i,k) / pmid(i,k)
   198)              hkk(i) = 0.5_r8 * hkl(i)
   199)           end do
   200)        end if
   201) 
   202) ! Now compute tv, zm, zi
   203) 
   204)        do i = 1,ncol
   205)           tvfac   = 1._r8 + zvir(i,k) * q(i,k)
   206)           tv      = t(i,k) * tvfac
   207) 
   208)           zm(i,k) = zi(i,k+1) + rog(i,k) * tv * hkk(i)
   209)           zi(i,k) = zi(i,k+1) + rog(i,k) * tv * hkl(i)
   210)        end do
   211)     end do
   212) 
   213)     return
