   712)    ! ----------------------------------------------------------------------------- !
   713) 
   714)    T_05(:ncol,top_lev:)  =  T_0(:ncol,top_lev:) + (  A_T(:ncol,top_lev:) +  C_T(:ncol,top_lev:) ) * dt
   715)    qv_05(:ncol,top_lev:) = qv_0(:ncol,top_lev:) + ( A_qv(:ncol,top_lev:) + C_qv(:ncol,top_lev:) ) * dt
   716)    ql_05(:ncol,top_lev:) = ql_0(:ncol,top_lev:) + ( A_ql(:ncol,top_lev:) + C_ql(:ncol,top_lev:) ) * dt
   717)    qi_05(:ncol,top_lev:) = qi_0(:ncol,top_lev:) + ( A_qi(:ncol,top_lev:) + C_qi(:ncol,top_lev:) ) * dt 
   718)    nl_05(:ncol,top_lev:) = max(0._r8, nl_0(:ncol,top_lev:) + ( A_nl(:ncol,top_lev:) + C_nl(:ncol,top_lev:) ) * dt )
   719)    ni_05(:ncol,top_lev:) = max(0._r8, ni_0(:ncol,top_lev:) + ( A_ni(:ncol,top_lev:) + C_ni(:ncol,top_lev:) ) * dt )
   720) 
   721)    call positive_moisture( ncol, dt, qmin1, qmin2, qmin3, dp, & 
   722)                            qv_05, ql_05, qi_05, T_05, A_qv_adj, &
   723)                            A_ql_adj, A_qi_adj, A_T_adj, do_cldice)
   724) 
   725)    ! -------------------------------------------------------------- !
   726)    ! Define reference state at the first iteration. This will be    !
   727)    ! continuously updated within the iteration loop below.          !
   728)    ! While equlibrium state properties are already output from the  !
   729)    ! 'instratus_condensate', they will be re-computed within the    !
   730)    ! each iteration process. At the first iteration, they will      !
   731)    ! produce exactly identical results. Note that except at the     !
   732)    ! very first iteration iter = 1, we must use updated cumulus     !
   733)    ! properties at all the other iteration processes. Even at the   !
   734)    ! first iteration, we should use updated cumulus properties      !
   735)    ! when computing limiters for (Q,P,E).                           !
   736)    ! -------------------------------------------------------------- !
   737) 
   738)    ! -------------------------------------------------------------- !
   739)    ! Define variables at the reference state of the first iteration !
   740)    ! -------------------------------------------------------------- !
   741) 
   742)    T(:ncol,top_lev:)     = T_0(:ncol,top_lev:)
   743)    qv(:ncol,top_lev:)    = qv_0(:ncol,top_lev:)
   744)    ql(:ncol,top_lev:)    = ql_0(:ncol,top_lev:)
   745)    qi(:ncol,top_lev:)    = qi_0(:ncol,top_lev:)
   746)    al_st(:ncol,top_lev:) = al_st_0(:ncol,top_lev:)
   747)    ai_st(:ncol,top_lev:) = ai_st_0(:ncol,top_lev:)
   748)    a_st(:ncol,top_lev:)  = a_st_0(:ncol,top_lev:)
   749)    ql_st(:ncol,top_lev:) = ql_st_0(:ncol,top_lev:)
   750)    qi_st(:ncol,top_lev:) = qi_st_0(:ncol,top_lev:)
   751)    nl(:ncol,top_lev:)    = nl_0(:ncol,top_lev:)
   752)    ni(:ncol,top_lev:)    = ni_0(:ncol,top_lev:)
   753) 
   754)    ! -------------------------- !
   755)    ! Main iterative computation !
   756)    ! -------------------------- !
   757) 
   758)    do iter = 1, niter
   759) 
   760)       ! ------------------------------------------ !
   761)       ! Initialize array within the iteration loop !
   762)       ! ------------------------------------------ !
   763) 
   764)       QQ(:,:)         = 0._r8
   765)       QQw(:,:)        = 0._r8
   766)       QQi(:,:)        = 0._r8
   767)       QQnl(:,:)       = 0._r8
   768)       QQni(:,:)       = 0._r8 
   769)       QQw2(:,:)       = 0._r8
   770)       QQi2(:,:)       = 0._r8
   771)       QQnl2(:,:)      = 0._r8
   772)       QQni2(:,:)      = 0._r8
   773)       nlten_pwi2(:,:) = 0._r8
   774)       niten_pwi2(:,:) = 0._r8
   775)       ACnl(:,:)       = 0._r8
   776)       ACni(:,:)       = 0._r8 
   777)       aa(:,:)         = 0._r8
   778)       bb(:,:)         = 0._r8
   779) 
   780)       do k = top_lev, pver
   781) 
   782)       ! "False" means that ice will not be taken into account.
   783)       call findsp_vc(qv_05(:ncol,k),T_05(:ncol,k),p(:ncol,k), .false., &
   784)            Twb_aw(:ncol,k),qvwb_aw(:ncol,k))
   785) 
   786)       call qsat_water(T_05(1:ncol,k), p(1:ncol,k), &
   787)            esat_a(1:ncol), qsat_a(1:ncol), dqsdt=dqsdT_a(1:ncol))
   788)       call qsat_water(T(1:ncol,k), p(1:ncol,k), &
   789)            esat_b(1:ncol), qsat_b(1:ncol), dqsdt=dqsdT_b(1:ncol))
   790) 
   791)       if( iter .eq. 1 ) then
   792)           a_cu(:ncol,k) = a_cud(:ncol,k)
   793)       else
   794)           a_cu(:ncol,k) = a_cu0(:ncol,k)
   795)       endif
   796)       do i = 1, ncol
   797)          U(i,k)    =  qv(i,k)/qsat_b(i)
   798)          U_nc(i,k) =  U(i,k)
   799)       enddo
   800)       if( CAMstfrac ) then
   801)           call astG_RHU(U_nc(:,k),p(:,k),qv(:,k),landfrac(:),snowh(:),al_st_nc(:,k),G_nc(:,k),ncol)
   802)       else
   803)           call astG_PDF(U_nc(:,k),p(:,k),qv(:,k),landfrac(:),snowh(:),al_st_nc(:,k),G_nc(:,k),ncol)
   804)       endif
   805)       call aist_vector(qv(:,k),T(:,k),p(:,k),qi(:,k),landfrac(:),snowh(:),ai_st_nc(:,k),ncol)
   806)       ai_st(:ncol,k)  =  (1._r8-a_cu(:ncol,k))*ai_st_nc(:ncol,k)
   807)       al_st(:ncol,k)  =  (1._r8-a_cu(:ncol,k))*al_st_nc(:ncol,k)
   808)       a_st(:ncol,k)   =  max(al_st(:ncol,k),ai_st(:ncol,k))  
   809) 
   810)       do i = 1, ncol
   811) 
   812)          ! -------------------------------------------------------- !
   813)          ! Compute basic thermodynamic coefficients for computing Q !
   814)          ! -------------------------------------------------------- !
   815) 
   816)          alpha  =  1._r8/qsat_b(i)
   817)          beta   =  dqsdt_b(i)*(qv(i,k)/qsat_b(i)**2)
   818)          betast =  alpha*dqsdT_b(i) 
   819)          gammal =  alpha + (latvap/cpair)*beta
   820)          gammai =  alpha + ((latvap+latice)/cpair)*beta
   821)          gammaQ =  alpha + (latvap/cpair)*beta
   822)          deltal =  1._r8 + a_st(i,k)*(latvap/cpair)*(betast/alpha)
   823)          deltai =  1._r8 + a_st(i,k)*((latvap+latice)/cpair)*(betast/alpha)
   824)          A_Tc   =  A_T(i,k)+A_T_adj(i,k)-(latvap/cpair)*(A_ql(i,k)+A_ql_adj(i,k))-((latvap+latice)/cpair)*(A_qi(i,k)+A_qi_adj(i,k))
   825)          A_qt   =  A_qv(i,k) + A_qv_adj(i,k) + A_ql(i,k) + A_ql_adj(i,k) + A_qi(i,k) + A_qi_adj(i,k)
   826)          C_Tc   =  C_T(i,k) - (latvap/cpair)*C_ql(i,k) - ((latvap+latice)/cpair)*C_qi(i,k)
   827)          C_qt   =  C_qv(i,k) + C_ql(i,k) + C_qi(i,k)
   828)          dTcdt  =  A_Tc + C_Tc
   829)          dqtdt  =  A_qt + C_qt
   830)        ! dqtstldt = A_qt + C_ql(i,k)/max(1.e-2_r8,al_st(i,k))                             ! Original  
   831)        ! dqtstldt = A_qt - A_qi(i,k) - A_qi_adj(i,k) + C_ql(i,k)/max(1.e-2_r8,al_st(i,k)) ! New 1 on Dec.30.2009.
   832)          dqtstldt = A_qt - A_qi(i,k) - A_qi_adj(i,k) + C_qlst(i,k)                        ! New 2 on Dec.30.2009.
   833)        ! dqtstldt = A_qt + C_qt                                                           ! Original Conservative treatment
   834)        ! dqtstldt = A_qt - A_qi(i,k) - A_qi_adj(i,k) + C_qt - C_qi(i,k)            ! New Conservative treatment on Dec.30.2009
   835)          dqidt = A_qi(i,k) + A_qi_adj(i,k) + C_qi(i,k) 
   836) 
   837)          anic    = max(1.e-8_r8,(1._r8-a_cu(i,k)))
   838)          GG      = G_nc(i,k)/anic
   839)          aa(1,1) = gammal*al_st(i,k)
   840)          aa(1,2) = GG + gammal*cc*ql_st(i,k)          
   841)          aa(2,1) = alpha + (latvap/cpair)*betast*al_st(i,k)
   842)          aa(2,2) = (latvap/cpair)*betast*cc*ql_st(i,k) 
   843)          bb(1,1) = alpha*dqtdt - beta*dTcdt - gammai*dqidt - GG*al_st_nc(i,k)*dacudt(i,k) + F_nc(i,k) 
   844)          bb(2,1) = alpha*dqtstldt - betast*(dTcdt + ((latvap+latice)/cpair)*dqidt) 
   845)          call gaussj(aa(1:2,1:2),2,2,bb(1:2,1),1,1)
   846)          dqlstdt = bb(1,1)
   847)          dalstdt = bb(2,1)
   848)          QQ(i,k) = al_st(i,k)*dqlstdt + cc*ql_st(i,k)*dalstdt - ( A_ql(i,k) + A_ql_adj(i,k) + C_ql(i,k) )
   849) 
   850)        ! ------------------------------------------------------------ !
   851)        ! Limiter for QQ                                               !
   852)        ! Here, 'fice' should be from the reference equilibrium state  !
   853)        ! since QQ itself is computed from the reference state.        !
   854)        ! From the assumption used for derivation of QQ(i), it must be !
   855)        ! that QQw(i) = QQ(i)*(1._r8-fice(i)), QQi(i) = QQ(i)*fice(i)  !  
   856)        ! ------------------------------------------------------------ !
   857) 
   858)          if( QQ(i,k) .ge. 0._r8 ) then
   859)              QQmax    = (qv_05(i,k) - qmin(1))/dt ! For ghost cumulus & semi-ghost ice stratus
   860)              QQmax    = max(0._r8,QQmax) 
   861)              QQ(i,k)  = min(QQ(i,k),QQmax)
   862)              QQw(i,k) = QQ(i,k)
   863)              QQi(i,k) = 0._r8 
   864)          else
   865)              QQmin  = 0._r8
   866)              if( qv_05(i,k) .lt. qsat_a(i) ) QQmin = min(0._r8,cone*(qv_05(i,k)-qvwb_aw(i,k))/dt)
   867)              QQ(i,k)  = max(QQ(i,k),QQmin)
   868)              QQw(i,k) = QQ(i,k)
   869)              QQi(i,k) = 0._r8
   870)              QQwmin   = min(0._r8,-cone*ql_05(i,k)/dt)
   871)              QQimin   = min(0._r8,-cone*qi_05(i,k)/dt)
   872)              QQw(i,k) = min(0._r8,max(QQw(i,k),QQwmin))
   873)              QQi(i,k) = min(0._r8,max(QQi(i,k),QQimin))
   874)          endif
   875) 
   876)        ! -------------------------------------------------- !
   877)        ! Reduce droplet concentration if evaporation occurs !
   878)        ! Note 'QQnl1,QQni1' are computed from the reference !
   879)        ! equilibrium state but limiter is from 'nl_05'.     !
   880)        ! -------------------------------------------------- !
   881) 
   882)          if( QQw(i,k) .lt. 0._r8 ) then
   883)              if( ql_05(i,k) .gt. qsmall ) then
   884)                  QQnl(i,k) = QQw(i,k)*nl_05(i,k)/ql_05(i,k)
   885)                  QQnl(i,k) = min(0._r8,cone*max(QQnl(i,k),-nl_05(i,k)/dt))
   886)              else
   887)                  QQnl(i,k) = 0._r8
   888)              endif  
   889)          endif 
   890) 
   891)          if( QQi(i,k) .lt. 0._r8 ) then
   892)              if( qi_05(i,k) .gt. qsmall ) then
   893)                  QQni(i,k) = QQi(i,k)*ni_05(i,k)/qi_05(i,k)
   894)                  QQni(i,k) = min(0._r8,cone*max(QQni(i,k),-ni_05(i,k)/dt))
   895)              else
   896)                  QQni(i,k) = 0._r8
   897)              endif  
   898)          endif 
   899) 
   900)       enddo
   901)       enddo
   902) 
   903)     ! -------------------------------------------------------------------- !
   904)     ! Until now, we have finished computing all necessary tendencies       ! 
   905)     ! from the equilibrium input state (T_0).                              !
   906)     ! If ramda = 0 : fully explicit scheme                                 !
   907)     !    ramda = 1 : fully implicit scheme                                 !
   908)     ! Note that 'ramda = 0.5 with niter = 2' can mimic                     !
   909)     ! -------------------------------------------------------------------- !
   910) 
   911)       if( iter .eq. 1 ) then
   912)           QQw_prev(:ncol,top_lev:)  = QQw(:ncol,top_lev:)       
   913)           QQi_prev(:ncol,top_lev:)  = QQi(:ncol,top_lev:)   
   914)           QQnl_prev(:ncol,top_lev:) = QQnl(:ncol,top_lev:)       
   915)           QQni_prev(:ncol,top_lev:) = QQni(:ncol,top_lev:)   
   916)       endif
   917) 
   918)       QQw_prog(:ncol,top_lev:)   = ramda*QQw(:ncol,top_lev:)   + (1._r8-ramda)*QQw_prev(:ncol,top_lev:)
   919)       QQi_prog(:ncol,top_lev:)   = ramda*QQi(:ncol,top_lev:)   + (1._r8-ramda)*QQi_prev(:ncol,top_lev:)
   920)       QQnl_prog(:ncol,top_lev:)  = ramda*QQnl(:ncol,top_lev:)  + (1._r8-ramda)*QQnl_prev(:ncol,top_lev:)
   921)       QQni_prog(:ncol,top_lev:)  = ramda*QQni(:ncol,top_lev:)  + (1._r8-ramda)*QQni_prev(:ncol,top_lev:)
   922) 
   923)       QQw_prev(:ncol,top_lev:)   = QQw_prog(:ncol,top_lev:)
   924)       QQi_prev(:ncol,top_lev:)   = QQi_prog(:ncol,top_lev:)
   925)       QQnl_prev(:ncol,top_lev:)  = QQnl_prog(:ncol,top_lev:)
   926)       QQni_prev(:ncol,top_lev:)  = QQni_prog(:ncol,top_lev:)
   927) 
   928)     ! -------------------------------------------------------- !
   929)     ! Compute final prognostic state on which final diagnostic !
   930)     ! in-stratus condensate adjustment is applied in the below.!
   931)     ! Important : I must check whether there are any external  !  
   932)     !             advective forcings of 'A_nl(i,k),A_ni(i,k)'. !
   933)     !             Even they are (i.e., advection of aerosol),  !
   934)     !             actual droplet activation will be performd   !
   935)     !             in microphysics, so it will be completely    !
   936)     !             reasonable to 'A_nl(i,k)=A_ni(i,k)=0'.       !
   937)     ! -------------------------------------------------------- !
   938) 
   939)     do k = top_lev, pver
   940)     do i = 1, ncol
   941)        T_prime0(i,k)  = T_0(i,k)  + dt*( A_T(i,k)  +  A_T_adj(i,k) +  C_T(i,k) + &
   942)             (latvap*QQw_prog(i,k)+(latvap+latice)*QQi_prog(i,k))/cpair )
   943)        qv_prime0(i,k) = qv_0(i,k) + dt*( A_qv(i,k) + A_qv_adj(i,k) + C_qv(i,k) - QQw_prog(i,k) - QQi_prog(i,k) )
   944)        ql_prime0(i,k) = ql_0(i,k) + dt*( A_ql(i,k) + A_ql_adj(i,k) + C_ql(i,k) + QQw_prog(i,k) )
   945)        qi_prime0(i,k) = qi_0(i,k) + dt*( A_qi(i,k) + A_qi_adj(i,k) + C_qi(i,k) + QQi_prog(i,k) )
   946)        nl_prime0(i,k) = max(0._r8,nl_0(i,k) + dt*( A_nl(i,k) + C_nl(i,k) + QQnl_prog(i,k) ))
   947)        ni_prime0(i,k) = max(0._r8,ni_0(i,k) + dt*( A_ni(i,k) + C_ni(i,k) + QQni_prog(i,k) ))
   948)        if( ql_prime0(i,k) .lt. qsmall ) nl_prime0(i,k) = 0._r8
   949)        if( qi_prime0(i,k) .lt. qsmall ) ni_prime0(i,k) = 0._r8
   950)     enddo
   951)     enddo
   952) 
   953)    ! -------------------------------------------------- !
   954)    ! Perform diagnostic 'positive_moisture' constraint. !
   955)    ! -------------------------------------------------- !
   956) 
   957)    T_dprime(:ncol,top_lev:)  =  T_prime0(:ncol,top_lev:) 
   958)    qv_dprime(:ncol,top_lev:) = qv_prime0(:ncol,top_lev:) 
   959)    ql_dprime(:ncol,top_lev:) = ql_prime0(:ncol,top_lev:) 
   960)    qi_dprime(:ncol,top_lev:) = qi_prime0(:ncol,top_lev:) 
   961)    nl_dprime(:ncol,top_lev:) = nl_prime0(:ncol,top_lev:) 
   962)    ni_dprime(:ncol,top_lev:) = ni_prime0(:ncol,top_lev:) 
   963) 
   964)    call positive_moisture( ncol, dt, qmin1, qmin2, qmin3, dp,          & 
   965)                            qv_dprime, ql_dprime, qi_dprime, T_dprime,  &
   966)                            qvten_pwi2, qlten_pwi2, qiten_pwi2, Tten_pwi2, do_cldice)
   967) 
   968)    do k = top_lev, pver
   969)    do i = 1, ncol
   970)       if( ql_dprime(i,k) .lt. qsmall ) then
   971)           nlten_pwi2(i,k) = -nl_dprime(i,k)/dt
   972)           nl_dprime(i,k)   = 0._r8
   973)       endif 
   974)       if( qi_dprime(i,k) .lt. qsmall ) then
   975)           niten_pwi2(i,k) = -ni_dprime(i,k)/dt
   976)           ni_dprime(i,k)   = 0._r8
   977)       endif 
   978)    enddo
   979)    enddo
   980) 
   981)    ! -------------------------------------------------------------- !
   982)    ! Add tendency associated with detrainment of cumulus condensate !
   983)    ! This tendency is not used in computing Q                       !
   984)    ! Since D_ql,D_qi,D_nl,D_ni > 0, don't need to worry about       !
   985)    ! negative scalar.                                               !
   986)    ! This tendency is not reflected into Fzs2, which is OK.         !
   987)    ! -------------------------------------------------------------- !
   988) 
   989)    T_dprime(:ncol,top_lev:)   =  T_dprime(:ncol,top_lev:)  + D_T(:ncol,top_lev:) * dt 
   990)    qv_dprime(:ncol,top_lev:)  = qv_dprime(:ncol,top_lev:) + D_qv(:ncol,top_lev:) * dt 
   991)    ql_dprime(:ncol,top_lev:)  = ql_dprime(:ncol,top_lev:) + D_ql(:ncol,top_lev:) * dt
   992)    qi_dprime(:ncol,top_lev:)  = qi_dprime(:ncol,top_lev:) + D_qi(:ncol,top_lev:) * dt
   993)    nl_dprime(:ncol,top_lev:)  = nl_dprime(:ncol,top_lev:) + D_nl(:ncol,top_lev:) * dt 
   994)    ni_dprime(:ncol,top_lev:)  = ni_dprime(:ncol,top_lev:) + D_ni(:ncol,top_lev:) * dt
   995) 
   996)    ! ---------------------------------------------------------- !
   997)    ! Impose diagnostic upper and lower limits on the in-stratus !
   998)    ! condensate amount. This produces a final equilibrium state !
   999)    ! at the end of each iterative process.                      !
  1000)    ! ---------------------------------------------------------- !
  1001) 
  1002)    do k = top_lev, pver
  1003)       call instratus_condensate( lchnk          , ncol           , k              , p(:,k)        , &
  1004)                                  T_dprime(:,k)  , qv_dprime(:,k) , ql_dprime(:,k) , qi_dprime(:,k), &
  1005)                                  a_cu0(:,k)     , zeros(:,k)     , zeros(:,k)     ,                 & 
  1006)                                  zeros(:,k)     , zeros(:,k)     , zeros(:,k)     ,                 &
  1007)                                  landfrac       , snowh          ,                                  &
  1008)                                  T_star(:,k)    , qv_star(:,k)   , ql_star(:,k)   , qi_star(:,k)  , & 
  1009)                                  al_st_star(:,k), ai_st_star(:,k), ql_st_star(:,k), qi_st_star(:,k) )
  1010)       a_st_star(:ncol,k)  = max(al_st_star(:ncol,k),ai_st_star(:ncol,k))
  1011)       QQw2(:ncol,k) = (ql_star(:ncol,k) - ql_dprime(:ncol,k))/dt
  1012)       QQi2(:ncol,k) = (qi_star(:ncol,k) - qi_dprime(:ncol,k))/dt
  1013)       ! -------------------------------------------------- !
  1014)       ! Reduce droplet concentration if evaporation occurs !
  1015)       ! -------------------------------------------------- !
  1016)       do i = 1, ncol
  1017)          if( QQw2(i,k) .le. 0._r8 ) then
  1018)              if( ql_dprime(i,k) .ge. qsmall ) then
  1019)                  QQnl2(i,k) = QQw2(i,k)*nl_dprime(i,k)/ql_dprime(i,k)
  1020)                  QQnl2(i,k) = min(0._r8,cone*max(QQnl2(i,k),-nl_dprime(i,k)/dt))
  1021)              else
  1022)                  QQnl2(i,k) = 0._r8
  1023)              endif  
  1024)          endif 
  1025)          if( QQi2(i,k) .le. 0._r8 ) then
  1026)              if( qi_dprime(i,k) .gt. qsmall ) then
  1027)                  QQni2(i,k) = QQi2(i,k)*ni_dprime(i,k)/qi_dprime(i,k)
  1028)                  QQni2(i,k) = min(0._r8,cone*max(QQni2(i,k),-ni_dprime(i,k)/dt))
  1029)              else
  1030)                  QQni2(i,k) = 0._r8
  1031)              endif  
  1032)          endif 
  1033)       enddo
  1034)    enddo
  1035)    nl_star(:ncol,top_lev:) = max(0._r8,nl_dprime(:ncol,top_lev:)+QQnl2(:ncol,top_lev:)*dt) 
  1036)    ni_star(:ncol,top_lev:) = max(0._r8,ni_dprime(:ncol,top_lev:)+QQni2(:ncol,top_lev:)*dt)
  1037) 
  1038)    ! ------------------------------------------ !
  1039)    ! Final adjustment of droplet concentration. !
  1040)    ! Set # to zero if there is no cloud.        !
  1041)    ! ------------------------------------------ !
  1042) 
  1043)    do k = top_lev, pver
  1044)    do i = 1, ncol 
  1045)       if( ql_star(i,k) .lt. qsmall ) then
  1046)           ACnl(i,k) = - nl_star(i,k)/dt
  1047)           nl_star(i,k) = 0._r8
  1048)       endif

  1050)           ACni(i,k) = - ni_star(i,k)/dt
  1051)           ni_star(i,k) = 0._r8
  1052)       endif
  1053)    enddo
  1054)    enddo
  1055) 
  1056)    ! ----------------------------------------------------- !
  1057)    ! Define equilibrium reference state for next iteration !
  1058)    ! ----------------------------------------------------- !
  1059) 
  1060)    T(:ncol,top_lev:)     = T_star(:ncol,top_lev:)
  1061)    qv(:ncol,top_lev:)    = qv_star(:ncol,top_lev:)
  1062)    ql(:ncol,top_lev:)    = ql_star(:ncol,top_lev:)
  1063)    qi(:ncol,top_lev:)    = qi_star(:ncol,top_lev:)
  1064)    al_st(:ncol,top_lev:) = al_st_star(:ncol,top_lev:)
  1065)    ai_st(:ncol,top_lev:) = ai_st_star(:ncol,top_lev:)
  1066)    a_st(:ncol,top_lev:)  = a_st_star(:ncol,top_lev:)
  1067)    ql_st(:ncol,top_lev:) = ql_st_star(:ncol,top_lev:)
  1068)    qi_st(:ncol,top_lev:) = qi_st_star(:ncol,top_lev:)
  1069)    nl(:ncol,top_lev:)    = nl_star(:ncol,top_lev:)
  1070)    ni(:ncol,top_lev:)    = ni_star(:ncol,top_lev:)
  1071) 
  1072)    enddo ! End of 'iter' prognostic iterative computation
  1073) 
  1074)    ! ------------------------------------------------------------------------ !
  1075)    ! Compute final tendencies of main output variables and diagnostic outputs !
  1076)    ! Note that the very input state [T0,qv0,ql0,qi0] are                      !
  1077)    ! marched to [T_star,qv_star,ql_star,qi_star] with equilibrium             !
  1078)    ! stratus informations of [a_st_star,ql_st_star,qi_st_star] by             !
  1079)    ! below final tendencies and [A_T,A_qv,A_ql,A_qi]                          !
  1080)    ! ------------------------------------------------------------------------ !
  1081) 
  1082)    ! ------------------ !
  1083)    ! Process tendencies !
  1084)    ! ------------------ !
  1085) 
  1086)    QQw_final(:ncol,top_lev:)  = QQw_prog(:ncol,top_lev:)
  1087)    QQi_final(:ncol,top_lev:)  = QQi_prog(:ncol,top_lev:)
  1088)    QQ_final(:ncol,top_lev:)   = QQw_final(:ncol,top_lev:) + QQi_final(:ncol,top_lev:)
  1089)    QQw_all(:ncol,top_lev:)    = QQw_prog(:ncol,top_lev:)  + QQw1(:ncol,top_lev:) + QQw2(:ncol,top_lev:) + &
  1090)         qlten_pwi1(:ncol,top_lev:) + qlten_pwi2(:ncol,top_lev:) + A_ql_adj(:ncol,top_lev:)
  1091)    QQi_all(:ncol,top_lev:)    = QQi_prog(:ncol,top_lev:)  + QQi1(:ncol,top_lev:) + QQi2(:ncol,top_lev:) + &
  1092)         qiten_pwi1(:ncol,top_lev:) + qiten_pwi2(:ncol,top_lev:) + A_qi_adj(:ncol,top_lev:)
  1093)    QQ_all(:ncol,top_lev:)     = QQw_all(:ncol,top_lev:)   + QQi_all(:ncol,top_lev:)
  1094)    QQnl_final(:ncol,top_lev:) = QQnl_prog(:ncol,top_lev:)
  1095)    QQni_final(:ncol,top_lev:) = QQni_prog(:ncol,top_lev:)
  1096)    QQn_final(:ncol,top_lev:)  = QQnl_final(:ncol,top_lev:) + QQni_final(:ncol,top_lev:)
  1097)    QQnl_all(:ncol,top_lev:)   = QQnl_prog(:ncol,top_lev:)  + QQnl1(:ncol,top_lev:) + QQnl2(:ncol,top_lev:) + &
  1098)         nlten_pwi1(:ncol,top_lev:) + nlten_pwi2(:ncol,top_lev:) + ACnl(:ncol,top_lev:) + A_nl_adj(:ncol,top_lev:)
  1099)    QQni_all(:ncol,top_lev:)   = QQni_prog(:ncol,top_lev:)  + QQni1(:ncol,top_lev:) + QQni2(:ncol,top_lev:) + &
  1100)         niten_pwi1(:ncol,top_lev:) + niten_pwi2(:ncol,top_lev:) + ACni(:ncol,top_lev:) + A_ni_adj(:ncol,top_lev:)
  1101)    QQn_all(:ncol,top_lev:)    = QQnl_all(:ncol,top_lev:)   + QQni_all(:ncol,top_lev:)
  1102)    qme(:ncol,top_lev:)        = QQ_final(:ncol,top_lev:)   
  1103)    qvadj(:ncol,top_lev:)      = qvten_pwi1(:ncol,top_lev:) + qvten_pwi2(:ncol,top_lev:) + A_qv_adj(:ncol,top_lev:)
  1104)    qladj(:ncol,top_lev:)      = qlten_pwi1(:ncol,top_lev:) + qlten_pwi2(:ncol,top_lev:) + A_ql_adj(:ncol,top_lev:)
  1105)    qiadj(:ncol,top_lev:)      = qiten_pwi1(:ncol,top_lev:) + qiten_pwi2(:ncol,top_lev:) + A_qi_adj(:ncol,top_lev:)
  1106)    qllim(:ncol,top_lev:)      = QQw1      (:ncol,top_lev:) + QQw2      (:ncol,top_lev:)
  1107)    qilim(:ncol,top_lev:)      = QQi1      (:ncol,top_lev:) + QQi2      (:ncol,top_lev:)
  1108) 
  1109)    ! ----------------- !
  1110)    ! Output tendencies !
  1111)    ! ----------------- !
  1112) 
  1113)    s_tendout(:ncol,top_lev:)  = cpair*( T_star(:ncol,top_lev:)  -  T0(:ncol,top_lev:) )/dt - &
  1114)         cpair*(A_T(:ncol,top_lev:)+C_T(:ncol,top_lev:))
  1115)    qv_tendout(:ncol,top_lev:) =    ( qv_star(:ncol,top_lev:) - qv0(:ncol,top_lev:) )/dt - &
  1116)         (A_qv(:ncol,top_lev:)+C_qv(:ncol,top_lev:))
  1117)    ql_tendout(:ncol,top_lev:) =    ( ql_star(:ncol,top_lev:) - ql0(:ncol,top_lev:) )/dt - &
  1118)         (A_ql(:ncol,top_lev:)+C_ql(:ncol,top_lev:))
  1119)    qi_tendout(:ncol,top_lev:) =    ( qi_star(:ncol,top_lev:) - qi0(:ncol,top_lev:) )/dt - &
  1120)         (A_qi(:ncol,top_lev:)+C_qi(:ncol,top_lev:))
  1121)    nl_tendout(:ncol,top_lev:) =    ( nl_star(:ncol,top_lev:) - nl0(:ncol,top_lev:) )/dt - &
  1122)         (A_nl(:ncol,top_lev:)+C_nl(:ncol,top_lev:))
  1123)    ni_tendout(:ncol,top_lev:) =    ( ni_star(:ncol,top_lev:) - ni0(:ncol,top_lev:) )/dt - &
  1124)         (A_ni(:ncol,top_lev:)+C_ni(:ncol,top_lev:))
  1125) 
  1126)    if (.not. do_cldice) then
  1127)       do k = top_lev, pver
  1128)          do i = 1, ncol
  1129) 
  1130)             ! Don't want either qi or ni tendencies, but the code above is somewhat convoluted and
  1131)             ! is trying to adjust both (small numbers). Just force it to zero here.
  1132)             qi_tendout(i,k) = 0._r8
  1133)             ni_tendout(i,k) = 0._r8
  1134)           end do
  1135)       end do
  1136)    end if
  1137) 
  1138)    ! ------------------ !
  1139)    ! Net cloud fraction !
  1140)    ! ------------------ !
  1141) 
  1142)    cld(:ncol,top_lev:) = a_st_star(:ncol,top_lev:) + a_cu0(:ncol,top_lev:)
  1143) 
  1144)    ! --------------------------------- !
  1145)    ! Updated grid-mean state variables !
  1146)    ! --------------------------------- !
  1147) 
  1148)    T0(:ncol,top_lev:)  = T_star(:ncol,top_lev:)
  1149)    qv0(:ncol,top_lev:) = qv_star(:ncol,top_lev:)
  1150)    ql0(:ncol,top_lev:) = ql_star(:ncol,top_lev:)
  1151)    qi0(:ncol,top_lev:) = qi_star(:ncol,top_lev:)
  1152)    nl0(:ncol,top_lev:) = nl_star(:ncol,top_lev:)
  1153)    ni0(:ncol,top_lev:) = ni_star(:ncol,top_lev:)
  1154) 
  1155)    return
  1156)    end subroutine mmacro_pcond

  1159)    ! End of subroutine !

  1162)    subroutine instratus_condensate( lchnk, ncol, k,                      &

  1164)                                     a_dc_in, ql_dc_in, qi_dc_in,         &
  1165)                                     a_sc_in, ql_sc_in, qi_sc_in,         & 
  1166)                                     landfrac, snowh,                     &
  1167)                                     T_out, qv_out, ql_out, qi_out,       &
  1168)                                     al_st_out, ai_st_out, ql_st_out, qi_st_out )
  1169)

  1179)

  1198)    real(r8), intent(in)  :: snowh(pcols)         ! Snow depth (liquid water equivalent)
  1199)

  1201)    real(r8), intent(out) :: qv_out(pcols)        ! Grid-mean water vapor [kg/kg]

  1206)    real(r8), intent(out) :: ai_st_out(pcols)     ! Ice stratus fraction
  1207)    real(r8), intent(out) :: ql_st_out(pcols)     ! In-stratus LWC [kg/kg]

  1211)

  1213) 
  1214)    real(r8) p    
  1215)    real(r8) T0   
  1216)    real(r8) qv0    
  1217)    real(r8) ql0    
  1218)    real(r8) qi0    
  1219)    real(r8) a_dc   
  1220)    real(r8) ql_dc  
  1221)    real(r8) qi_dc  
  1222)    real(r8) a_sc   
  1223)    real(r8) ql_sc  
  1224)    real(r8) qi_sc  
  1225)    real(r8) esat0  
  1226)    real(r8) qsat0

  1228)    real(r8) U0_nc  
  1229)    real(r8) G0_nc
  1230)    real(r8) al0_st_nc            
  1231)    real(r8) al0_st
  1232)    real(r8) ai0_st_nc            
  1233)    real(r8) ai0_st               
  1234)    real(r8) a0_st               
  1235)    real(r8) ql0_nc
  1236)    real(r8) qi0_nc
  1237)    real(r8) qc0_nc
  1238)    real(r8) ql0_st
  1239)    real(r8) qi0_st
  1240)    real(r8) qc0_st

  1242)    real(r8) qv    
  1243)    real(r8) ql    
  1244)    real(r8) qi
  1245)    real(r8) ql_st
  1246)    real(r8) qi_st
  1247)    real(r8) es  
  1248)    real(r8) qs  
  1249)    real(r8) esat_in(pcols)  
  1250)    real(r8) qsat_in(pcols)  
  1251)    real(r8) U0_in(pcols)  
  1252)    real(r8) al0_st_nc_in(pcols)
  1253)    real(r8) ai0_st_nc_in(pcols)
  1254)    real(r8) G0_nc_in(pcols)
  1255)    integer  idxmod 
  1256)    real(r8) U
  1257)    real(r8) U_nc
  1258)    real(r8) al_st_nc
  1259)    real(r8) ai_st_nc
  1260)    real(r8) G_nc
  1261)    real(r8) a_st
  1262)    real(r8) al_st
  1263)    real(r8) ai_st
  1264)    real(r8) Tmin0
  1265)    real(r8) Tmax0
  1266)    real(r8) Tmin
  1267)    real(r8) Tmax
  1268)    integer caseid
  1269) 
  1270)    ! ---------------- !
  1271)    ! Main Computation ! 
  1272)    ! ---------------- !
  1273) 
  1274)    call qsat_water(T0_in(1:ncol), p_in(1:ncol), &
  1275)         esat_in(1:ncol), qsat_in(1:ncol))
  1276)    U0_in(:ncol) = qv0_in(:ncol)/qsat_in(:ncol)
  1277)    if( CAMstfrac ) then
  1278)        call astG_RHU(U0_in(:),p_in(:),qv0_in(:),landfrac(:),snowh(:),al0_st_nc_in(:),G0_nc_in(:),ncol)
  1279)    else
  1280)        call astG_PDF(U0_in(:),p_in(:),qv0_in(:),landfrac(:),snowh(:),al0_st_nc_in(:),G0_nc_in(:),ncol)
  1281)    endif
  1282)    call aist_vector(qv0_in(:),T0_in(:),p_in(:),qi0_in(:),landfrac(:),snowh(:),ai0_st_nc_in(:),ncol)
  1283) 
  1284)    do i = 1, ncol
  1285) 
  1286)       ! ---------------------- !
  1287)       ! Define local variables !
  1288)       ! ---------------------- !
  1289) 
  1290)       p   = p_in(i)
  1291)

  1294)       ql0 = ql0_in(i)
  1295)       qi0 = qi0_in(i)

  1298)       ql_dc = ql_dc_in(i)
  1299)       qi_dc = qi_dc_in(i)
  1300) 
  1301)       a_sc  = a_sc_in(i)
  1302)       ql_sc = ql_sc_in(i)
  1303)       qi_sc = qi_sc_in(i)
  1304)

  1306)       qi_dc = 0._r8
  1307)       ql_sc = 0._r8
  1308)       qi_sc = 0._r8
  1309)

  1311)       qs  = qsat_in(i) 
  1312) 
  1313)       idxmod = 0
  1314)       caseid = -1
  1315) 
  1316)       ! ------------------------------------------------------------ !
  1317)       ! Force the grid-mean RH to be smaller than 1 if oversaturated !
  1318)       ! In order to be compatible with reduced 3x3 QQ, condensation  !
  1319)       ! should occur only into the liquid in gridmean_RH.            !
  1320)       ! ------------------------------------------------------------ !
  1321) 
  1322)       if( qv0 .gt. qs ) then

  1324)                             a_dc, ql_dc, qi_dc, a_sc, ql_sc, qi_sc, &

  1361)       if( ( ql0_nc .ge. qlst_min*al0_st ) .and. ( ql0_nc .le. qlst_max*al0_st ) ) then

  1397)                  caseid = 1

  1400)                  ! ------------------------------------------- !

  1402)                  ! ------------------------------------------- !
  1403)                  Tmin = Tmin0
  1404)                  Tmax = T0 
  1405)                  call instratus_core( lchnk, i, k, p,                              &
  1406)                                       T0, qv0, ql0, 0._r8,                         &
  1407)                                       a_dc, ql_dc, qi_dc,                          &
  1408)                                       a_sc, ql_sc, qi_sc, ai0_st,                  &
  1409)                                       qlst_max, Tmin, Tmax, landfrac(i), snowh(i), &
  1410)                                       T, qv, ql, qi )   
  1411)                  idxmod = 1
  1412)                  caseid = 2
  1413)                  goto 10
  1414)              endif
  1415)          ! ------------------------------ !
  1416)          ! This is case III : Empty Cloud !
  1417)          ! ------------------------------ !  
  1418)          elseif( al0_st .gt. 0._r8 .and. ql0_nc .eq. 0._r8 ) then
  1419)               ! ------------------------------------------ ! 
  1420)               ! Condense until qc_st increases to qlst_min !
  1421)               ! ------------------------------------------ !
  1422)               Tmin = Tmin0
  1423)               Tmax = Tmax0  
  1424)               call instratus_core( lchnk, i, k, p,                              &
  1425)                                    T0, qv0, ql0, 0._r8,                         &
  1426)                                    a_dc, ql_dc, qi_dc,                          &
  1427)                                    a_sc, ql_sc, qi_sc, ai0_st,                  &
  1428)                                    qlst_min, Tmin, Tmax, landfrac(i), snowh(i), &
  1429)                                    T, qv, ql, qi )   
  1430)               idxmod = 1 
  1431)               caseid = 3
  1432)               goto 10
  1433)          ! --------------- !
  1434)          ! This is case IV !
  1435)          ! --------------- !   
  1436)          elseif( al0_st .gt. 0._r8 .and. ql0_nc .gt. 0._r8 ) then
  1437) 
  1438)              if( ql0_nc .gt. qlst_max*al0_st ) then
  1439)                  ! --------------------------------------- !
  1440)                  ! Evaporate until qc_st drops to qlst_max !
  1441)                  ! --------------------------------------- !
  1442)                  Tmin = Tmin0
  1443)                  Tmax = Tmax0
  1444)                  call instratus_core( lchnk, i, k, p,                              &
  1445)                                       T0, qv0, ql0, 0._r8,                         &
  1446)                                       a_dc, ql_dc, qi_dc,                          &
  1447)                                       a_sc, ql_sc, qi_sc, ai0_st,                  &
  1448)                                       qlst_max, Tmin, Tmax, landfrac(i), snowh(i), &
  1449)                                       T, qv, ql, qi )   
  1450)                  idxmod = 1
  1451)                  caseid = 4
  1452)                  goto 10
  1453)              elseif( ql0_nc .lt. qlst_min*al0_st ) then
  1454)                  ! -------------------------------------------- !
  1455)                  ! Condensate until qc_st increases to qlst_min !
  1456)                  ! -------------------------------------------- !
  1457)                  Tmin = Tmin0
  1458)                  Tmax = Tmax0 
  1459)                  call instratus_core( lchnk, i, k, p,                              &
  1460)                                       T0, qv0, ql0, 0._r8,                         &
  1461)                                       a_dc, ql_dc, qi_dc,                          &
  1462)                                       a_sc, ql_sc, qi_sc, ai0_st,                  &
  1463)                                       qlst_min, Tmin, Tmax, landfrac(i), snowh(i), & 
  1464)                                       T, qv, ql, qi )   
  1465)                  idxmod = 1
  1466)                  caseid = 5
  1467)                  goto 10
  1468)              else
  1469)                  ! ------------------------------------------------ !
  1470)                  ! This case should not happen. Issue error message !
  1471)                  ! ------------------------------------------------ !
  1472)                  write(iulog,*) 'Impossible case1 in instratus_condensate' 
  1473)                  call endrun
  1474)              endif
  1475)          ! ------------------------------------------------ !                   
  1476)          ! This case should not happen. Issue error message !
  1477)          ! ------------------------------------------------ !    
  1478)          else
  1479)              write(iulog,*) 'Impossible case2 in instratus_condensate' 
  1480)              write(iulog,*)  al0_st, a_sc, a_dc
  1481)              write(iulog,*)  1000*ql0_nc, 1000*(ql0+qi0)
  1482)              call endrun
  1483)          endif
  1484)       endif
  1485) 
  1486) 10 continue   
  1487) 
  1488)    ! -------------------------------------------------- !
  1489)    ! Force final energy-moisture conserving consistency !
  1490)    ! -------------------------------------------------- !
  1491) 
  1492)      qi = qi0
  1493) 
  1494)      if( idxmod .eq. 1 ) then
  1495)          call aist_single(qv,T,p,qi,landfrac(i),snowh(i),ai_st_nc)
  1496)          ai_st = (1._r8-a_dc-a_sc)*ai_st_nc
  1497)          call qsat_water(T, p, es, qs)
  1498)          U     = (qv/qs)
  1499)          U_nc  =  U
  1500)          if( CAMstfrac ) then
  1501)              call astG_RHU_single(U_nc,p,qv,landfrac(i),snowh(i),al_st_nc,G_nc)
  1502)          else
  1503)              call astG_PDF_single(U_nc,p,qv,landfrac(i),snowh(i),al_st_nc,G_nc)
  1504)          endif
  1505)          al_st = (1._r8-a_dc-a_sc)*al_st_nc
  1506)      else
  1507)          ai_st  = (1._r8-a_dc-a_sc)*ai0_st_nc_in(i)
  1508)          al_st  = (1._r8-a_dc-a_sc)*al0_st_nc_in(i)
  1509)      endif
  1510) 
  1511)      a_st  = max(ai_st,al_st)
  1512) 
  1513)      if( al_st .eq. 0._r8 ) then
  1514)          ql_st = 0._r8
  1515)      else
  1516)          ql_st = ql/al_st
  1517)          ql_st = min(qlst_max,max(qlst_min,ql_st)) ! PJR
  1518)      endif
  1519)      if( ai_st .eq. 0._r8 ) then
  1520)          qi_st = 0._r8
  1521)      else
  1522)          qi_st = qi/ai_st
  1523)      endif
  1524) 
  1525)      qi    = ai_st*qi_st
  1526)      ql    = al_st*ql_st
  1527) 
  1528)      T     = T0 - (latvap/cpair)*(ql0-ql) - ((latvap+latice)/cpair)*(qi0-qi)
  1529)      qv    = qv0 + ql0 - ql + qi0 - qi
  1530) 
  1531)    ! -------------- !
  1532)    ! Send to output !
  1533)    ! -------------- !
  1534) 
  1535)    T_out(i)  = T
  1536)    qv_out(i) = qv
  1537)    ql_out(i) = ql
  1538)    qi_out(i) = qi
  1539)    al_st_out(i) = al_st
  1540)    ai_st_out(i) = ai_st
  1541)    ql_st_out(i) = ql_st
  1542)    qi_st_out(i) = qi_st
  1543) 
  1544)    enddo 
  1545) 
  1546)    return
  1547)    end subroutine instratus_condensate

  1560)    ! ------------------------------------------------------ !
  1561)    ! Subroutine to find saturation equilibrium state using  ! 
  1562)    ! a Newton iteration method, so that 'qc_st = qcst_crit' !
  1563)    ! is satisfied.                                          !
  1564)    ! ------------------------------------------------------ !
  1565) 
  1566)    use time_manager,  only: is_first_step, get_nstep
  1567) 
  1568)    implicit none
  1569) 
  1570)    integer,  intent(in)  :: lchnk      ! Chunk identifier
  1571)    integer,  intent(in)  :: icol       ! Number of atmospheric columns
  1572)    integer,  intent(in)  :: k          ! Layer index
  1573) 
  1574)    real(r8), intent(in)  :: p          ! Pressure [Pa]
  1575)    real(r8), intent(in)  :: T0         ! Temperature [K]
  1576)    real(r8), intent(in)  :: qv0        ! Grid-mean water vapor [kg/kg]
  1577)    real(r8), intent(in)  :: ql0        ! Grid-mean LWC [kg/kg]
  1578)    real(r8), intent(in)  :: qi0        ! Grid-mean IWC [kg/kg]
  1579) 
  1580)    real(r8), intent(in)  :: a_dc       ! Deep cumulus cloud fraction
  1581)    real(r8), intent(in)  :: ql_dc      ! In-deep cumulus LWC [kg/kg]
  1582)    real(r8), intent(in)  :: qi_dc      ! In-deep cumulus IWC [kg/kg]
  1583)    real(r8), intent(in)  :: a_sc       ! Shallow cumulus cloud fraction
  1584)    real(r8), intent(in)  :: ql_sc      ! In-shallow cumulus LWC [kg/kg]
  1585)    real(r8), intent(in)  :: qi_sc      ! In-shallow cumulus IWC [kg/kg]
  1586) 
  1587)    real(r8), intent(in)  :: ai_st      ! Ice stratus fraction (fixed)
  1588) 
  1589)    real(r8), intent(in)  :: Tmin       ! Minimum temperature system can have [K]
  1590)    real(r8), intent(in)  :: Tmax       ! Maximum temperature system can have [K]
  1591)    real(r8), intent(in)  :: qcst_crit  ! Critical in-stratus condensate [kg/kg]
  1592)    real(r8), intent(in)  :: landfrac   ! Land fraction
  1593)    real(r8), intent(in)  :: snowh      ! Snow depth (liquid water equivalent)
  1594) 
  1595)    real(r8), intent(out) :: T          ! Temperature [K]
  1596)    real(r8), intent(out) :: qv         ! Grid-mean water vapor [kg/kg]
  1597)    real(r8), intent(out) :: ql         ! Grid-mean LWC [kg/kg]
  1598)    real(r8), intent(out) :: qi         ! Grid-mean IWC [kg/kg]
  1599) 
  1600)    ! Local variables
  1601) 
  1602)    integer i                           ! Iteration index
  1603) 
  1604)    real(r8) muQ0, muQ
  1605)    real(r8) ql_nc0, qi_nc0, qc_nc0, qc_nc    
  1606)    real(r8) fice0, fice    
  1607)    real(r8) ficeg0, ficeg   
  1608)    real(r8) esat0
  1609)    real(r8) qsat0
  1610)    real(r8) dqcncdt, dastdt, dUdt
  1611)    real(r8) alpha, beta
  1612)    real(r8) U, U_nc
  1613)    real(r8) al_st_nc, G_nc
  1614)    real(r8) al_st
  1615) 
  1616)    ! Variables for root-finding algorithm
  1617) 
  1618)    integer j                          
  1619)    real(r8)  x1, x2
  1620)    real(r8)  rtsafe
  1621)    real(r8)  df, dx, dxold, f, fh, fl, temp, xh, xl
  1622)    real(r8), parameter :: xacc = 1.e-3_r8
  1623) 
  1624)    ! ---------------- !
  1625)    ! Main computation !
  1626)    ! ---------------- !
  1627) 
  1628)    ql_nc0 = max(0._r8,ql0-a_dc*ql_dc-a_sc*ql_sc)
  1629)    qi_nc0 = max(0._r8,qi0-a_dc*qi_dc-a_sc*qi_sc)
  1630)    qc_nc0 = max(0._r8,ql0+qi0-a_dc*(ql_dc+qi_dc)-a_sc*(ql_sc+qi_sc))
  1631)    fice0  = 0._r8
  1632)    ficeg0 = 0._r8
  1633)    muQ0   = 1._r8
  1634) 
  1635)    ! ------------ !
  1636)    ! Root finding !
  1637)    ! ------------ !
  1638) 
  1639)    x1 = Tmin
  1640)    x2 = Tmax
  1641)    call funcd_instratus( x1, p, T0, qv0, ql0, qi0, fice0, muQ0, qc_nc0, &
  1642)                          a_dc, ql_dc, qi_dc, a_sc, ql_sc, qi_sc, ai_st, &
  1643)                          qcst_crit, landfrac, snowh,                    &
  1644)                          fl, df, qc_nc, fice, al_st )
  1645)    call funcd_instratus( x2, p, T0, qv0, ql0, qi0, fice0, muQ0, qc_nc0, &
  1646)                          a_dc, ql_dc, qi_dc, a_sc, ql_sc, qi_sc, ai_st, &
  1647)                          qcst_crit, landfrac, snowh,                    &
  1648)                          fh, df, qc_nc, fice, al_st )
  1649)    if((fl > 0._r8 .and. fh > 0._r8) .or. (fl < 0._r8 .and. fh < 0._r8)) then
  1650)        call funcd_instratus( T0, p, T0, qv0, ql0, qi0, fice0, muQ0, qc_nc0, &
  1651)                              a_dc, ql_dc, qi_dc, a_sc, ql_sc, qi_sc, ai_st, &
  1652)                              qcst_crit, landfrac, snowh,                    &
  1653)                              fl, df, qc_nc, fice, al_st )
  1654)        rtsafe = T0 
  1655)        goto 10       
  1656)    endif
  1657)    if( fl == 0._r8) then
  1658)            rtsafe = x1
  1659)            goto 10
  1660)    elseif( fh == 0._r8) then
  1661)            rtsafe = x2
  1662)            goto 10
  1663)    elseif( fl < 0._r8) then
  1664)            xl = x1
  1665)            xh = x2
  1666)    else
  1667)            xh = x1
  1668)            xl = x2
  1669)    end if
  1670)    rtsafe = 0.5_r8*(x1+x2)
  1671)    dxold = abs(x2-x1)
  1672)    dx = dxold
  1673)    call funcd_instratus( rtsafe, p, T0, qv0, ql0, qi0, fice0, muQ0, qc_nc0, &
  1674)                          a_dc, ql_dc, qi_dc, a_sc, ql_sc, qi_sc, ai_st,     &
  1675)                          qcst_crit, landfrac, snowh,                        &
  1676)                          f, df, qc_nc, fice, al_st )
  1677)    do j = 1, 20
  1678)       if(((rtsafe-xh)*df-f)*((rtsafe-xl)*df-f) > 0._r8 .or. abs(2.0_r8*f) > abs(dxold*df) ) then
  1679)            dxold = dx
  1680)            dx = 0.5_r8*(xh-xl)
  1681)            rtsafe = xl + dx
  1682)            if(xl == rtsafe) goto 10
  1683)       else
  1684)            dxold = dx
  1685)            dx = f/df
  1686)            temp = rtsafe
  1687)            rtsafe = rtsafe - dx
  1688)            if (temp == rtsafe) goto 10
  1689)       end if
  1690)     ! if(abs(dx) < xacc) goto 10
  1691)       call funcd_instratus( rtsafe, p, T0, qv0, ql0, qi0, fice0, muQ0, qc_nc0, &
  1692)                             a_dc, ql_dc, qi_dc, a_sc, ql_sc, qi_sc, ai_st,     &
  1693)                             qcst_crit, landfrac, snowh,                        &
  1694)                             f, df, qc_nc, fice, al_st )
  1695)     ! Sep.21.2010. Sungsu modified to enhance convergence and guarantee 'qlst_min <  qlst < qlst_max'.
  1696)       if( qcst_crit < 0.5_r8 * ( qlst_min + qlst_max ) ) then
  1697)           if( ( qc_nc*(1._r8-fice) .gt.          qlst_min*al_st .and. &
  1698)                 qc_nc*(1._r8-fice) .lt. 1.1_r8 * qlst_min*al_st ) ) goto 10
  1699)       else
  1700)           if( ( qc_nc*(1._r8-fice) .gt. 0.9_r8 * qlst_max*al_st .and. &
  1701)                 qc_nc*(1._r8-fice) .lt.          qlst_max*al_st ) ) goto 10
  1702)       endif
  1703)       if(f < 0._r8) then
  1704)           xl = rtsafe
  1705)       else
  1706)           xh = rtsafe
  1707)       endif
  1708) 
  1709)    enddo
  1710) 
  1711) 10 continue
  1712) 
  1713)    ! ------------------------------------------- !
  1714)    ! Final safety check before sending to output !
  1715)    ! ------------------------------------------- !
  1716) 
  1717)    qc_nc = max(0._r8,qc_nc)
  1718) 
  1719)    T  = rtsafe
  1720)    ql = qc_nc*(1._r8-fice) + a_dc*ql_dc + a_sc*ql_sc
  1721)    qi = qc_nc*fice + a_dc*qi_dc + a_sc*qi_sc
  1722)    qv = qv0 + ql0 + qi0 - (qc_nc + a_dc*(ql_dc+qi_dc) + a_sc*(ql_sc+qi_sc))
  1723)    qv = max(qv,1.e-12_r8) 
  1724) 
  1725)    return
  1726)    end subroutine instratus_core

  1739)    ! --------------------------------------------------- !

  1742)

  1747)    real(r8), intent(in)  :: qv0        ! Grid-mean water vapor [kg/kg]

  1755)    real(r8), intent(in)  :: ql_dc      ! In-deep cumulus LWC [kg/kg]

  1768)    real(r8), intent(out) :: fg         ! Gradient of minimization function 
  1769)    real(r8), intent(out) :: qc_nc      !
  1770)    real(r8), intent(out) :: al_st      !
  1771)    real(r8), intent(out) :: fice       !

  1776)    real(r8) qs

  1778)    real(r8) dqcncdt

  1780)    real(r8) beta
  1781)    real(r8) U
  1782)    real(r8) U_nc
  1783)    real(r8) al_st_nc
  1784)    real(r8) G_nc

  1791)    ! ---------------- !

  1794) 
  1795)    fice    = fice0

  1797)    dqcncdt = (cpair/latvap) 
  1798)    qv      = (qv0 + ql0 + qi0 - (qc_nc + a_dc*(ql_dc+qi_dc) + a_sc*(ql_sc+qi_sc)))
  1799)    alpha   = (1._r8/qs)
  1800)    beta    = (qv/qs**2._r8)*dqsdT 
  1801) 
  1802)    U      =  (qv/qs)
  1803)    U_nc   =   U
  1804)    if( CAMstfrac ) then
  1805)        call astG_RHU_single(U_nc,p,qv,landfrac,snowh,al_st_nc,G_nc)
  1806)    else

  1811)    dalstdt =  (1._r8/G_nc)*dUdt
  1812)    if( U_nc .eq. 1._r8 ) dalstdt = 0._r8
  1813) 
  1814)    f  = qc_nc   - qcst_crit*al_st
  1815)    fg = dqcncdt - qcst_crit*dalstdt
  1816) 
  1817)    return

  1865)    real(r8)  Tc, qt, qc, dqcdt, qc_nc    
  1866)    real(r8)  es, qs, dqsdT
  1867)    real(r8)  al_st_nc, G_nc
  1868)    real(r8)  f, fg
  1869)    real(r8), parameter :: xacc = 1.e-3_r8
  1870) 
  1871)    ! ---------------- !
  1872)    ! Main computation !
  1873)    ! ---------------- !
  1874) 
  1875)    ql_nc0 = max(0._r8,ql-a_dc*ql_dc-a_sc*ql_sc)
  1876)    qi_nc0 = max(0._r8,qi-a_dc*qi_dc-a_sc*qi_sc)
  1877)    qc_nc0 = max(0._r8,ql+qi-a_dc*(ql_dc+qi_dc)-a_sc*(ql_sc+qi_sc))
  1878)    Tc    = T - (latvap/cpair)*ql
  1879)    qt    = qv + ql
  1880) 
  1881)    do m = 1, 20
  1882)       call qsat_water(T, p, es, qs, dqsdt=dqsdT)
  1883)       Tscale = latvap/cpair
  1884)       qc     = (T-Tc)/Tscale
  1885)       dqcdt  = 1._r8/Tscale
  1886)       f      = qs + qc - qt 
  1887)       fg     = dqsdT + dqcdt
  1888)       fg     = sign(1._r8,fg)*max(1.e-10_r8,abs(fg))
  1889)     ! Sungsu modified convergence criteria to speed up convergence and guarantee RH <= 1.
  1890)       if( qc .ge. 0._r8 .and. ( qt - qc ) .ge. 0.999_r8*qs .and. ( qt - qc ) .le. 1._r8*qs ) then
  1891)           goto 10
  1892)       endif
  1893)       T = T - f/fg
  1894)    enddo
  1895)  ! write(iulog,*) 'Convergence in gridmean_RH is not reached. RH = ', ( qt - qc ) / qs
  1896) 10 continue
  1897) 
  1898)    call qsat_water(T, p, es, qs)
  1899)  ! Sungsu modified 'qv = qs' in consistent with the modified convergence criteria above.
  1900)    qv = min(qt,qs) ! Modified
  1901)    ql = qt - qv
  1902)    T  = Tc + (latvap/cpair)*ql
  1903)

  1911)    subroutine positive_moisture( ncol, dt, qvmin, qlmin, qimin, dp, &
  1912)                                  qv,   ql, qi,    t,     qvten, &
  1913)                                  qlten,    qiten, tten,  do_cldice)
  1914) 
  1915)    ! ------------------------------------------------------------------------------- !
  1916)    ! If any 'ql < qlmin, qi < qimin, qv < qvmin' are developed in any layer,         !
  1917)    ! force them to be larger than minimum value by (1) condensating water vapor      !
  1918)    ! into liquid or ice, and (2) by transporting water vapor from the very lower     !
  1919)    ! layer. '2._r8' is multiplied to the minimum values for safety.                  !
  1920)    ! Update final state variables and tendencies associated with this correction.    !
  1921)    ! If any condensation happens, update (s,t) too.                                  !
  1922)    ! Note that (qv,ql,qi,t,s) are final state variables after applying corresponding !
  1923)    ! input tendencies.                                                               !
  1924)    ! Be careful the order of k : '1': top layer, 'pver' : near-surface layer         ! 
  1925)    ! ------------------------------------------------------------------------------- !
  1926) 
  1927)    implicit none
  1928)    integer,  intent(in)     :: ncol
  1929)    real(r8), intent(in)     :: dt
  1930)    real(r8), intent(in)     :: dp(pcols,pver), qvmin(pcols,pver), qlmin(pcols,pver), qimin(pcols,pver)
  1931)    real(r8), intent(inout)  :: qv(pcols,pver), ql(pcols,pver), qi(pcols,pver), t(pcols,pver)
  1932)    real(r8), intent(out)    :: qvten(pcols,pver), qlten(pcols,pver), qiten(pcols,pver), tten(pcols,pver)
  1933)    logical, intent(in)      :: do_cldice
  1934)    integer   i, k
  1935)    real(r8)  dql, dqi, dqv, sum, aa, dum 
  1936) 
  1937)    tten(:ncol,:pver)  = 0._r8
  1938)    qvten(:ncol,:pver) = 0._r8
  1939)    qlten(:ncol,:pver) = 0._r8
  1940)    qiten(:ncol,:pver) = 0._r8
  1941) 
  1942)    do i = 1, ncol
  1943)       do k = top_lev, pver
  1944)          if( qv(i,k) .lt. qvmin(i,k) .or. ql(i,k) .lt. qlmin(i,k) .or. qi(i,k) .lt. qimin(i,k) ) then
  1945)              goto 10
  1946)          endif
  1947)       enddo
  1948)       goto 11
  1949)    10 continue
  1950)       do k = top_lev, pver    ! From the top to the 1st (lowest) layer from the surface
  1951)          dql = max(0._r8,1._r8*qlmin(i,k)-ql(i,k))
  1952) 
  1953)          if (do_cldice) then
  1954)          dqi = max(0._r8,1._r8*qimin(i,k)-qi(i,k))
  1955)          else
  1956)            dqi = 0._r8
  1957)          end if
  1958) 
  1959)          qlten(i,k) = qlten(i,k) +  dql/dt
  1960)          qiten(i,k) = qiten(i,k) +  dqi/dt
  1961)          qvten(i,k) = qvten(i,k) - (dql+dqi)/dt
  1962)          tten(i,k)  = tten(i,k)  + (latvap/cpair)*(dql/dt) + ((latvap+latice)/cpair)*(dqi/dt)
  1963)          ql(i,k)    = ql(i,k) + dql
  1964)          qi(i,k)    = qi(i,k) + dqi
  1965)          qv(i,k)    = qv(i,k) - dql - dqi
  1966)          t(i,k)     = t(i,k)  + (latvap * dql + (latvap+latice) * dqi)/cpair
  1967)          dqv        = max(0._r8,1._r8*qvmin(i,k)-qv(i,k))
  1968)          qvten(i,k) = qvten(i,k) + dqv/dt
  1969)          qv(i,k)    = qv(i,k)    + dqv
  1970)          if( k .ne. pver ) then 
  1971)              qv(i,k+1)    = qv(i,k+1)    - dqv*dp(i,k)/dp(i,k+1)
  1972)              qvten(i,k+1) = qvten(i,k+1) - dqv*dp(i,k)/dp(i,k+1)/dt
  1973)          endif
  1974)          qv(i,k) = max(qv(i,k),qvmin(i,k))
  1975)          ql(i,k) = max(ql(i,k),qlmin(i,k))
  1976)          qi(i,k) = max(qi(i,k),qimin(i,k))
  1977)       end do
  1978)       ! Extra moisture used to satisfy 'qv(i,pver)=qvmin' is proportionally 
  1979)       ! extracted from all the layers that has 'qv > 2*qvmin'. This fully
  1980)       ! preserves column moisture. 
  1981)       if( dqv .gt. 1.e-20_r8 ) then
  1982)           sum = 0._r8
  1983)           do k = top_lev, pver
  1984)              if( qv(i,k) .gt. 2._r8*qvmin(i,k) ) sum = sum + qv(i,k)*dp(i,k)
  1985)           enddo
  1986)           aa = dqv*dp(i,pver)/max(1.e-20_r8,sum)
  1987)           if( aa .lt. 0.5_r8 ) then
  1988)               do k = top_lev, pver
  1989)                  if( qv(i,k) .gt. 2._r8*qvmin(i,k) ) then
  1990)                      dum        = aa*qv(i,k)
  1991)                      qv(i,k)    = qv(i,k) - dum
  1992)                      qvten(i,k) = qvten(i,k) - dum/dt
  1993)                  endif
  1994)               enddo 
  1995)           else 
  1996)               write(iulog,*) 'Full positive_moisture is impossible in Park Macro'
  1997)           endif
  1998)       endif 
  1999) 11 continue
  2000)    enddo
  2001)    return
  2002) 
  2003)    end subroutine positive_moisture
