     3) 
     4) !---------------------------------------------------------------
     5) !
     6) ! Implements artificial suite of tracers
     7) !    1) low tracer with unit mixing ratio at about 800 mb
     8) !    2) med tracer with unit mixing ratio at about 500 mb
     9) !    3) high tracer with unit mixing ratio at about 200 mb
    10) !    4) reverse med tracer with unit mixing ratio everywhere except about 500 mb
    11) !    5) unit tracer with unit mixing ratio everywhere
    12) !
    13) !  D Bundy June 2003
    14) !  modified Feb 2004 to include TT_UN and smoothing
    15) !
    16) !  A Mirin and B Eaton, August 2007
    17) !  Modified to create up to 1000 distinct copies of the 5 basic tracers
    18) !  by appending up to a 3 digit number to the base tracer name.
    19) !  RESTRICTION - trac_ncnstmx cannot exceed 5000 unless the algorithm for
    20) !                constructing new tracer names is extended.
    21) !
    22) ! This is a module that contains a suite of tracers that is interfaced
    23) ! by tracers.F90. The details of the suite are contained entirely
    24) ! in this file, so the public routines are all very generic. 
    25) !
    26) !  ------------  calling tree --------------
    27) !  Initialize the tracer mixing ratio field
    28) !  	-> tracers.F90: tracers_init_cnst 
    29) !  		-> tracers_suite.F90:init_cnst_tr
    30) !  			-> init_cnst_lw
    31) !  			-> init_cnst_md
    32) !  			-> init_cnst_hi
    33) !  			-> init_cnst_un
    34) !
    35) !  Initialize data set, things that need to be done at the beginning of a 
    36) !  run (whether restart or initial)
    37) !  	-> tracers.F90: tracers_init
    38) !  		-> tracers_suite.F90:init_tr
    39) !
    40) !  Timestepping:
    41) !  	-> tracers_timestep_init
    42) !  		-> tracers_suite.F90:timestep_init_tr
    43) !
    44) !  	-> tracers_timestep_tend
    45) !  		-> tracers_suite.F90:flux_tr
    46) !  			-> flux_lw  
    47) !  			-> flux_md
    48) !  			-> flux_hi
    49) !  			-> flux_un
    50) !
    51) !  		-> tracers_suite.F90:tend_tr
    52) !  			-> tend_lw
    53) !  			-> tend_md
    54) !  			-> tend_hi
    55) !  			-> tend_un

    87) !======================================================================
    88) function get_tracer_name(n)
    89) 
    90) !------------------------------------------------------------------------
    91) ! Purpose:
    92) !
    93) ! The tracer names are only defined in this module. This function is for
    94) ! outside programs to grab the name for each tracer number. 
    95) !    If n > trac_ncst calls endrun.
    96) !
    97) !------------------------------------------------------------------------
    98) 
    99) ! -----------------------------Arguments---------------------------------
   100) 
   101)   integer, intent(in) :: n
   102)   character(len=8) :: get_tracer_name  ! return value
   103) 
   104) ! ----------------------------- Local ---------------------------------
   105)   character(len=5), dimension(trac_names), parameter :: & ! constituent names
   106)        tracer_names  =  (/ 'TT_LW', 'TT_MD', 'TT_HI', 'TTRMD' , 'TT_UN'/)
   107)   
   108) !-----------------------------------------------------------------------
   109) 
   110)   integer :: nbase  ! Corresponding base tracer index
   111)   integer :: ncopy  ! No. of copies of base tracers
   112)   character(len=1) :: c1
   113)   character(len=2) :: c2
   114)   character(len=3) :: c3
   115) 
   116)   if ( n > trac_ncnstmx ) then
   117)      write(iulog,*) 'tracers_suite:get_tracer_name()','requested tracer',n
   118)      write(iulog,*) 'only ',trac_ncnstmx,' tracers available'
   119)      call endrun
   120)   else
   121)      nbase = mod(n-1, trac_names) + 1
   122)      ncopy = (n-1)/trac_names
   123)      if ( ncopy == 0 ) then
   124)         get_tracer_name = tracer_names(nbase)
   125)      else if ( ncopy >= 1  .and.  ncopy <= 9 ) then
   126)         write (c1,'(i1)') ncopy
   127)         get_tracer_name = tracer_names(nbase) // c1
   128)      else if ( ncopy >= 10  .and.  ncopy <= 99 ) then
   129)         write (c2,'(i2)') ncopy
   130)         get_tracer_name = tracer_names(nbase) // c2
   131)      else if ( ncopy >= 100  .and.  ncopy <= 999 ) then
   132)         write (c3,'(i3)') ncopy
   133)         get_tracer_name = tracer_names(nbase) // c3
   134)      end if
   135)   endif
   136) 
   137)   return
   138) 
   139) end function get_tracer_name

   144) subroutine init_cnst_tr(m,q, gcid)
   145) 
   146) !----------------------------------------------------------------------- 
   147) !
   148) ! Purpose:
   149) ! calls initialization routine for tracer m, returns mixing ratio in q
   150) !
   151) ! This routine must be consistent with trac_ncnstmx.
   152) !
   153) !----------------------------------------------------------------------- 
   154) 
   155)    implicit none
   156) 
   157)    real(r8), intent(out) :: q(:,:)    ! kg tracer/kg dry air (gcol, plev)
   158)    integer ,intent(in) :: m           ! index of tracer
   159)    integer, intent(in) :: gcid(:)     ! global column id   
   160) 
   161)    integer nbase ! Corresponding base tracer index
   162) 
   163)    if ( m > trac_ncnstmx ) then
   164)       write(iulog,*) 'tracers_suite:init_cnst_tr()'
   165)       write(iulog,*) ' asked to initialize tracer number ',m
   166)       write(iulog,*) ' but there are only trac_ncnstmx = ',trac_ncnstmx,' tracers'
   167)       call endrun
   168)    endif
   169) 
   170)    nbase = mod(m-1,trac_names)+1
   171) 
   172)    if ( nbase == 1 ) then
   173)       call init_cnst_lw(q, gcid)
   174)    else if ( nbase == 2 ) then
   175)       call init_cnst_md(q, gcid)
   176)    else if ( nbase == 3 ) then
   177)       call init_cnst_hi(q, gcid)
   178)    else if ( nbase == 4 ) then
   179)       call init_cnst_md(q, gcid, rev_in=1)
   180)    else if ( nbase == 5 ) then
   181)       call init_cnst_un(q, gcid)
   182)    else
   183)       write(iulog,*) 'tracers_suite:init_cnst_tr()'
   184)       write(iulog,*) 'no initialization routine specified for tracer',nbase
   185)       call endrun
   186)    endif
   187)       
   188) end subroutine init_cnst_tr

   365) subroutine setsmoothtr(indx,q,width,rev_in)
   366)   use ref_pres, only : pref_mid
   367)   implicit none
   368) 
   369) 
   370)   !Arguments
   371)   integer, intent(in)     :: indx               ! k index of pressure level
   372)   real(r8), intent(inout) :: q(:,:)  ! kg tracer/kg dry air
   373)   real(r8), intent(in)    :: width              ! eta difference from unit level where q = 0.1
   374)   integer,  intent(in), optional :: rev_in      ! reverse the mixing ratio
   375) 
   376) 
   377)   !Local variables
   378)   integer k
   379)   real(r8) alpha ! guassian width, determined by width, T
   380)   real(r8) pdist ! pressure distance (eta.e4) from k=indx
   381)   real(r8) T     ! desired m.r. in level specified by pdiff from k=indx
   382)   integer rev  ! = 1 then reverse (q = 1, q(k=indx) = 0 )
   383) 
   384) 
   385) 
   386) 
   387)   rev = 0
   388)   if (present(rev_in)) then
   389)      if (rev_in == 1) then
   390)         rev = 1
   391)      endif
   392)   endif
   393) 
   394) 
   395) !  write(iulog,*)'TR SMOOTH indx pref_mid(indx)',indx,pref_mid(indx)
   396) !  write(iulog,67)'TR SMOOTH ','k','pref_mid(k)','pdist','-a*(pd^2)','e^-a*(pd^2)'
   397) 
   398)   T = 0.1_r8
   399)   alpha = -log(T)/(width*1.e4_r8)**2  ! s.t. in level width from indx, mr = T
   400) 
   401) !  alpha = 3.e-8  ! m.r. ~ 0.1 in adjacent levels, where change eta ~ 0.08
   402) 
   403)   do k=1,pver
   404)      pdist = pref_mid(k) - pref_mid(indx)
   405) 
   406)      if ( rev == 1 ) then
   407)         q(:,k) = 1.0_r8 - exp(-alpha*(pdist**2))
   408)      else
   409)         q(:,k) =       exp(-alpha*(pdist**2))
   410)   endif
   411) !     write(iulog,66)'TR SMOOTH ',k,pref_mid(k),pdist,-alpha*pdist**2,q(1,k)
   412)   end do
   413)   
   414) 66 format (a15,i4,3f15.2,g15.6)
   415) 67 format (a15,a4,4a15)
   416)   
   417) 
   418) end subroutine setsmoothtr

   423) integer function setpindxtr(pmb)
   424) 
   425)   ! find the index of layer nearest pmb
   426) 
   427)   use ref_pres, only : pref_mid
   428) 
   429)   implicit none
   430)   
   431)   
   432)   real(r8) pmb
   433)   real(r8) pmin, pdist
   434)   integer indx, k
   435)   
   436)   indx = 0
   437)   pmin = 1.e36_r8
   438)   pdist = 1.e36_r8
   439)   do k=1,pver
   440)      pdist = abs(pref_mid(k) - pmb*100._r8)
   441)      if (pdist < pmin) then
   442)         indx = k
   443)         pmin = pdist
   444)      end if
   445)   end do
   446)   !write(iulog,*) ' index near', pmb, ' is ', indx
   447)   setpindxtr = indx
   448) 
   449) end function setpindxtr

   454) subroutine tend_tr(m,ncol, q, deltat, snk)
   455) !----------------------------------------------------------------------- 
   456) !
   457) ! Purpose: Calculate tendency (decay) of radon tracer
   458) ! 
   459) ! Method:
   460) !-------------------------Code History----------------------------------
   461) !
   462) ! tracers.F90:rndecay()
   463) ! Original version:  B. Eaton, 1995
   464) ! Standardized:      T. Acker, Feb 1996
   465) !
   466) !
   467) !-----------------------------------------------------------------------
   468)    implicit none
   469) !-------------------------Arguments--------------------------------------
   470) !
   471)    integer,  intent(in)  :: m                 ! tracer number
   472)    integer,  intent(in)  :: ncol              ! number of atmospheric columns
   473)    real(r8), intent(in)  :: q(pcols,pver)    ! radon mixing ratio (kg/(kg moist air))
   474)    real(r8), intent(in)  :: deltat            ! time step
   475)    real(r8), intent(out) :: snk(pcols,pver) ! conversion rate
   476)                                               ! (kg tracer /(s kg moist air))
   477) !--------------------------Local Variables------------------------------
   478) 
   479)   integer nbase
   480) 
   481)   if ( m > trac_ncnstmx ) then
   482)       write(iulog,*) 'tracers_suite:tend_tr()'
   483)       write(iulog,*) ' asked to calculate tendency for tracer number ',m
   484)       write(iulog,*) ' but there are only trac_ncnstmx = ',trac_ncnstmx,' tracers'
   485)       call endrun('tracers_suite.F90:tend_tr() L457')
   486)    endif
   487) 
   488)    nbase = mod(m-1,trac_names) + 1
   489) 
   490)    if ( nbase == 1 ) then
   491)       call tend_lw(ncol, q, deltat, snk)
   492)    else if ( nbase == 2 ) then
   493)       call tend_md(ncol, q, deltat, snk)
   494)    else if ( nbase == 3 ) then
   495)       call tend_hi(ncol, q, deltat, snk)
   496)    else if ( nbase == 4 ) then
   497)       call tend_md(ncol, q, deltat, snk)
   498)    else if ( nbase == 5 ) then
   499)       call tend_un(ncol, q, deltat, snk)
   500) 
   501) 
   502)    else
   503)       write(iulog,*) 'tracers_suite:tend_tr()'
   504)       write(iulog,*) 'no tendency routine specified for tracer',nbase
   505)       call endrun
   506)    endif
   507)       
   508) 
   509) 
   510) end subroutine tend_tr
