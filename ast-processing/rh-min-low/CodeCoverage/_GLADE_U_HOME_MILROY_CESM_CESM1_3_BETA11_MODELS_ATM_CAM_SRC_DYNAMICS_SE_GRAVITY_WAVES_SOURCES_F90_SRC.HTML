    87)   subroutine compute_frontogenesis(frontgf,frontga,tl,elem,ederiv,hybrid,nets,nete)
    88)   !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
    89)   ! compute frontogenesis function F
    90)   !   F =  -gradth dot C
    91)   ! with:
    92)   !   theta  = potential temperature
    93)   !   gradth = grad(theta)
    94)   !   C = ( gradth dot grad ) U
    95)   ! 
    96)   ! Original by Mark Taylor, July 2011
    97)   ! Change by Santos, 10 Aug 2011:
    98)   ! Integrated into gravity_waves_sources module, several arguments made global
    99)   !  to prevent repeated allocation/initialization
   100)   ! 
   101)   !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   102)     use physical_constants, only : kappa
   103)     use derivative_mod, only : gradient_sphere, ugradv_sphere
   104)     use edge_mod, only : edgevpack, edgevunpack
   105)     use bndry_mod, only : bndry_exchangev
   106)     use dyn_comp, only : hvcoord
   107)     implicit none
   108)     type (hybrid_t)      , intent(in) :: hybrid
   109)     type (element_t)     , intent(inout), target :: elem(:)
   110)     type (derivative_t)  , intent(in) :: ederiv
   111)     integer, intent(in) :: nets,nete
   112)     integer, intent(in) :: tl ! timelevel to use
   113)     real(kind=real_kind), intent(out) ::  frontgf(np,np,nlev,nets:nete)
   114)     real(kind=real_kind), intent(out) ::  frontga(np,np,nlev,nets:nete)
   115)   
   116)     ! local
   117)     integer :: k,kptr,i,j,ie,component
   118)     real(kind=real_kind)  ::  gradth(np,np,2,nlev,nets:nete)  ! grad(theta)
   119)     real(kind=real_kind)  :: p(np,np)        ! pressure at mid points
   120)     real(kind=real_kind)  :: theta(np,np)    ! potential temperature at mid points
   121)     real(kind=real_kind)  ::  C(np,np,2)     
   122)     
   123)     do ie=nets,nete
   124)        do k=1,nlev
   125) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!1
   126)         ! potential temperature: theta = T (p/p0)^kappa
   127) !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!1
   128)           
   129)           ! pressure at mid points
   130)           p(:,:)   = hvcoord%hyam(k)*hvcoord%ps0 + hvcoord%hybm(k)*elem(ie)%state%ps_v(:,:,tl)
   131)           theta(:,:) = elem(ie)%state%T(:,:,k,tl)*(psurf_ref / p(:,:))**kappa
   132)           gradth(:,:,:,k,ie) = gradient_sphere(theta,ederiv,elem(ie)%Dinv)
   133)           
   134)           ! compute C = (grad(theta) dot grad ) u
   135)           C(:,:,:) = ugradv_sphere(gradth(:,:,:,k,ie), elem(ie)%state%v(:,:,:,k,tl),ederiv,elem(ie))
   136)           
   137)           ! gradth dot C
   138)           frontgf(:,:,k,ie) = -( C(:,:,1)*gradth(:,:,1,k,ie) +  C(:,:,2)*gradth(:,:,2,k,ie)  )
   139)           
   140)           ! apply mass matrix
   141)           gradth(:,:,1,k,ie)=gradth(:,:,1,k,ie)*elem(ie)%spheremp(:,:)
   142)           gradth(:,:,2,k,ie)=gradth(:,:,2,k,ie)*elem(ie)%spheremp(:,:)
   143)           frontgf(:,:,k,ie)=frontgf(:,:,k,ie)*elem(ie)%spheremp(:,:)
   144)           
   145)        enddo
   146)        ! pack
   147)        call edgeVpack(edge3, frontgf(:,:,:,ie),nlev,0,elem(ie)%desc)
   148)        call edgeVpack(edge3, gradth(:,:,:,:,ie),2*nlev,nlev,elem(ie)%desc)
   149)     enddo
   150)     call bndry_exchangeV(hybrid,edge3)
   151)     do ie=nets,nete
   152)        call edgeVunpack(edge3, frontgf(:,:,:,ie),nlev,0,elem(ie)%desc)
   153)        call edgeVunpack(edge3, gradth(:,:,:,:,ie),2*nlev,nlev,elem(ie)%desc)
   154)        ! apply inverse mass matrix,
   155)        do k=1,nlev
   156)           gradth(:,:,1,k,ie)=gradth(:,:,1,k,ie)*elem(ie)%rspheremp(:,:)
   157)           gradth(:,:,2,k,ie)=gradth(:,:,2,k,ie)*elem(ie)%rspheremp(:,:)
   158)           frontgf(:,:,k,ie)=frontgf(:,:,k,ie)*elem(ie)%rspheremp(:,:)
   159)           
   160)           ! Frontogenesis angle
   161)           frontga(:,:,k,ie) = atan2 ( gradth(:,:,2,k,ie) , gradth(:,:,1,k,ie) + 1.e-10_real_kind )
   162)        enddo
   163)     enddo
   164)   end subroutine compute_frontogenesis
