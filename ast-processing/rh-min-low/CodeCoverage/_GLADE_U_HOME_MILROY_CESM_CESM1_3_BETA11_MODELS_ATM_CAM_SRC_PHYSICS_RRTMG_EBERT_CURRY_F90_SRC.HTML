   103) subroutine cloud_rad_props_get_sw(state, pbuf, &
   104)                                   tau, tau_w, tau_w_g, tau_w_f,&
   105)                                   diagnosticindex, oldliq, oldice)
   106) 
   107) ! return totaled (across all species) layer tau, omega, g, f 
   108) ! for all spectral interval for aerosols affecting the climate
   109) 
   110)    ! Arguments
   111)    type(physics_state), intent(in) :: state
   112)    type(physics_buffer_desc),  pointer :: pbuf(:)

   114) 
   115)    real(r8), intent(out) :: tau    (nswbands,pcols,pver) ! aerosol extinction optical depth
   116)    real(r8), intent(out) :: tau_w  (nswbands,pcols,pver) ! aerosol single scattering albedo * tau
   117)    real(r8), intent(out) :: tau_w_g(nswbands,pcols,pver) ! aerosol assymetry parameter * tau * w
   118)    real(r8), intent(out) :: tau_w_f(nswbands,pcols,pver) ! aerosol forward scattered fraction * tau * w
   119) 
   120)    logical, optional, intent(in) :: oldliq,oldice
   121) 
   122)    ! Local variables
   123) 
   124)    integer :: ncol
   125)    integer :: lchnk

   128) 
   129)    !-----------------------------------------------------------------------------
   130) 
   131)    ncol  = state%ncol
   132)    lchnk = state%lchnk

   136)    tau_w   (:,:,:) = -100._r8
   137)    tau_w_g (:,:,:) = -100._r8
   138)    tau_w_f (:,:,:) = -100._r8

   193) subroutine ec_ice_optics_sw   (state, pbuf, ice_tau, ice_tau_w, ice_tau_w_g, ice_tau_w_f, oldicewp)
   194) 
   195)    use physconst,      only: gravit
   196) 
   197)    type(physics_state), intent(in) :: state
   198)    type(physics_buffer_desc), pointer :: pbuf(:)
   199) 
   200)    real(r8),intent(out) :: ice_tau    (nswbands,pcols,pver) ! extinction optical depth
   201)    real(r8),intent(out) :: ice_tau_w  (nswbands,pcols,pver) ! single scattering albedo * tau
   202)    real(r8),intent(out) :: ice_tau_w_g(nswbands,pcols,pver) ! assymetry parameter * tau * w
   203)    real(r8),intent(out) :: ice_tau_w_f(nswbands,pcols,pver) ! forward scattered fraction * tau * w
   204)    logical, intent(in) :: oldicewp
   205) 
   206)    real(r8), pointer, dimension(:,:) :: rei
   207)    real(r8), pointer, dimension(:,:) :: cldn
   208)    real(r8), pointer, dimension(:,:) :: tmpptr
   209)    real(r8), dimension(pcols,pver) :: cicewp
   210)    real(r8), dimension(nswbands) :: wavmin
   211)    real(r8), dimension(nswbands) :: wavmax
   212)    !
   213)    ! ice water coefficients (Ebert and Curry,1992, JGR, 97, 3831-3836)
   214)    real(r8) :: abari(4) = &     ! a coefficient for extinction optical depth
   215)       (/ 3.448e-03_r8, 3.448e-03_r8,3.448e-03_r8,3.448e-03_r8/)
   216)    real(r8) :: bbari(4) = &     ! b coefficient for extinction optical depth
   217)       (/ 2.431_r8    , 2.431_r8    ,2.431_r8    ,2.431_r8    /)
   218)    real(r8) :: cbari(4) = &     ! c coefficient for single scat albedo
   219)       (/ 1.00e-05_r8 , 1.10e-04_r8 ,1.861e-02_r8,.46658_r8   /)
   220)    real(r8) :: dbari(4) = &     ! d coefficient for single scat albedo
   221)       (/ 0.0_r8      , 1.405e-05_r8,8.328e-04_r8,2.05e-05_r8 /)
   222)    real(r8) :: ebari(4) = &     ! e coefficient for asymmetry parameter
   223)       (/ 0.7661_r8   , 0.7730_r8   ,0.794_r8    ,0.9595_r8   /)
   224)    real(r8) :: fbari(4) = &     ! f coefficient for asymmetry parameter
   225)       (/ 5.851e-04_r8, 5.665e-04_r8,7.267e-04_r8,1.076e-04_r8/)
   226) 
   227)    real(r8) :: abarii           ! A coefficient for current spectral band
   228)    real(r8) :: bbarii           ! B coefficient for current spectral band
   229)    real(r8) :: cbarii           ! C coefficient for current spectral band
   230)    real(r8) :: dbarii           ! D coefficient for current spectral band
   231)    real(r8) :: ebarii           ! E coefficient for current spectral band
   232)    real(r8) :: fbarii           ! F coefficient for current spectral band
   233) 
   234)    ! Minimum cloud amount (as a fraction of the grid-box area) to 
   235)    ! distinguish from clear sky
   236)    real(r8), parameter :: cldmin = 1.0e-80_r8
   237) 
   238)    ! Decimal precision of cloud amount (0 -> preserve full resolution;
   239)    ! 10^-n -> preserve n digits of cloud amount)
   240)    real(r8), parameter :: cldeps = 0.0_r8
   241) 
   242)    integer :: ns, i, k, indxsl, lchnk, Nday
   243)    integer :: itim_old
   244)    real(r8) :: tmp1i, tmp2i, tmp3i, g
   245) 
   246)    Nday = state%ncol
   247)    lchnk = state%lchnk
   248) 
   249)    itim_old = pbuf_old_tim_idx()
   250)    call pbuf_get_field(pbuf, cld_idx,cldn, start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
   251)    call pbuf_get_field(pbuf, rei_idx,rei)
   252) 
   253)    if(oldicewp) then
   254)      do k=1,pver
   255)         do i = 1,Nday
   256)            cicewp(i,k) = 1000.0_r8*state%q(i,k,ixcldice)*state%pdel(i,k) /(gravit* max(0.01_r8,cldn(i,k)))
   257)         end do
   258)      end do
   259)    else
   260)      if (iciwp_idx<=0) then 
   261)         call endrun('ec_ice_optics_sw: oldicewp must be set to true since ICIWP was not found in pbuf')
   262)      endif
   263)      call pbuf_get_field(pbuf, iciwp_idx, tmpptr)
   264)      cicewp(1:pcols,1:pver) =  1000.0_r8*tmpptr(1:pcols,1:pver)
   265)    endif
   266)    
   267)    call get_sw_spectral_boundaries(wavmin,wavmax,'microns')
   268) 
   269)    do ns = 1, nswbands
   270) 
   271)       if(wavmax(ns) <= 0.7_r8) then
   272)          indxsl = 1
   273)       else if(wavmax(ns) <= 1.25_r8) then
   274)          indxsl = 2
   275)       else if(wavmax(ns) <= 2.38_r8) then
   276)          indxsl = 3
   277)       else if(wavmax(ns) > 2.38_r8) then
   278)          indxsl = 4
   279)       end if
   280) 
   281)       abarii = abari(indxsl)
   282)       bbarii = bbari(indxsl)
   283)       cbarii = cbari(indxsl)
   284)       dbarii = dbari(indxsl)
   285)       ebarii = ebari(indxsl)
   286)       fbarii = fbari(indxsl)
   287) 
   288)       do k=1,pver
   289)          do i=1,Nday
   290) 
   291)             ! note that optical properties for ice valid only
   292)             ! in range of 13 > rei > 130 micron (Ebert and Curry 92)
   293)             if (cldn(i,k) >= cldmin .and. cldn(i,k) >= cldeps) then
   294)                tmp1i = abarii + bbarii/max(13._r8,min(scalefactor*rei(i,k),130._r8))
   295)                ice_tau(ns,i,k) = cicewp(i,k)*tmp1i
   296)             else
   297)                ice_tau(ns,i,k) = 0.0_r8
   298)             endif
   299) 
   300)             tmp2i = 1._r8 - cbarii - dbarii*min(max(13._r8,scalefactor*rei(i,k)),130._r8)
   301)             tmp3i = fbarii*min(max(13._r8,scalefactor*rei(i,k)),130._r8)
   302)             ! Do not let single scatter albedo be 1.  Delta-eddington solution
   303)             ! for non-conservative case has different analytic form from solution
   304)             ! for conservative case, and raddedmx is written for non-conservative case.
   305)             ice_tau_w(ns,i,k) = ice_tau(ns,i,k) * min(tmp2i,.999999_r8)
   306)             g = ebarii + tmp3i
   307)             ice_tau_w_g(ns,i,k) = ice_tau_w(ns,i,k) * g
   308)             ice_tau_w_f(ns,i,k) = ice_tau_w(ns,i,k) * g * g
   309) 
   310)          end do ! End do i=1,Nday
   311)       end do    ! End do k=1,pver
   312)    end do ! nswbands
   313) 
   314) end subroutine ec_ice_optics_sw

   317) subroutine ec_ice_get_rad_props_lw(state, pbuf, abs_od, oldicewp)
   318)    use physconst,      only: gravit
   319) 
   320)    type(physics_state), intent(in)   :: state
   321)    type(physics_buffer_desc),pointer :: pbuf(:)
   322)     real(r8), intent(out) :: abs_od(nlwbands,pcols,pver)
   323)    logical, intent(in) :: oldicewp
   324) 
   325)    real(r8) :: gicewp(pcols,pver)
   326)    real(r8) :: gliqwp(pcols,pver)
   327)    real(r8) :: cicewp(pcols,pver)
   328)    real(r8) :: cliqwp(pcols,pver)
   329)    real(r8) :: ficemr(pcols,pver)
   330)    real(r8) :: cwp(pcols,pver)
   331)    real(r8) :: cldtau(pcols,pver)
   332) 
   333)    real(r8), pointer, dimension(:,:) :: cldn
   334)    real(r8), pointer, dimension(:,:) :: rei
   335)    integer :: ncol, itim_old, lwband, i, k, lchnk
   336) 
   337)     real(r8) :: kabs, kabsi
   338) 
   339)     real(r8) kabsl                  ! longwave liquid absorption coeff (m**2/g)
   340)     parameter (kabsl = 0.090361_r8)
   341) 
   342)    real(r8), pointer, dimension(:,:) :: iclwpth, iciwpth
   343) 
   344) 
   345)    ncol = state%ncol
   346)    lchnk = state%lchnk
   347) 
   348)    itim_old  =  pbuf_old_tim_idx()
   349)    call pbuf_get_field(pbuf, rei_idx,   rei)
   350)    call pbuf_get_field(pbuf, cld_idx,   cldn, start=(/1,1,itim_old/), kount=(/pcols,pver,1/))
   351) 
   352) 
   353)    if(oldicewp) then
   354)      do k=1,pver
   355)          do i = 1,ncol
   356)             gicewp(i,k) = state%q(i,k,ixcldice)*state%pdel(i,k)/gravit*1000.0_r8  ! Grid box ice water path.
   357)             gliqwp(i,k) = state%q(i,k,ixcldliq)*state%pdel(i,k)/gravit*1000.0_r8  ! Grid box liquid water path.
   358)             cicewp(i,k) = gicewp(i,k) / max(0.01_r8,cldn(i,k))                 ! In-cloud ice water path.
   359)             cliqwp(i,k) = gliqwp(i,k) / max(0.01_r8,cldn(i,k))                 ! In-cloud liquid water path.
   360)             ficemr(i,k) = state%q(i,k,ixcldice) /                 &
   361)                  max(1.e-10_r8,(state%q(i,k,ixcldice)+state%q(i,k,ixcldliq)))
   362)          end do
   363)      end do
   364)      cwp(:ncol,:pver) = cicewp(:ncol,:pver) + cliqwp(:ncol,:pver)
   365)    else
   366)       if (iclwp_idx<=0 .or. iciwp_idx<=0) then 
   367)          call endrun('ec_ice_get_rad_props_lw: oldicewp must be set to true since ICIWP and/or ICLWP were not found in pbuf')
   368)       endif
   369)       call pbuf_get_field(pbuf, iclwp_idx, iclwpth)
   370)       call pbuf_get_field(pbuf, iciwp_idx, iciwpth)
   371)       do k=1,pver
   372)          do i = 1,ncol
   373)             cwp(i,k) = 1000.0_r8 *iciwpth(i,k) + 1000.0_r8 *iclwpth(i,k)
   374)             ficemr(i,k) = 1000.0_r8*iciwpth(i,k)/(max(1.e-18_r8,cwp(i,k)))
   375)          end do
   376)       end do
   377)    endif
   378) 
   379)    do k=1,pver
   380)        do i=1,ncol
   381) 
   382)           ! Note from Andrew Conley:
   383)           !  Optics for RK no longer supported, This is constructed to get
   384)           !  close to bit for bit.  Otherwise we could simply use ice water path
   385)           !note that optical properties for ice valid only
   386)           !in range of 13 > rei > 130 micron (Ebert and Curry 92)
   387)           kabsi = 0.005_r8 + 1._r8/min(max(13._r8,scalefactor*rei(i,k)),130._r8)
   388)           kabs =  kabsi*ficemr(i,k) ! kabsl*(1._r8-ficemr(i,k)) + kabsi*ficemr(i,k)
   389)           !emis(i,k) = 1._r8 - exp(-1.66_r8*kabs*clwp(i,k))
   390)           cldtau(i,k) = kabs*cwp(i,k)
   391)        end do
   392)    end do
   393) !
   394)    do lwband = 1,nlwbands
   395)       abs_od(lwband,1:ncol,1:pver)=cldtau(1:ncol,1:pver)
   396)    enddo
   397) 
   398)    !if(oldicewp) then
   399)    !  call outfld('CIWPTH_OLD',cicewp(:,:)/1000,pcols,lchnk)
   400)    !else
   401)    !  call outfld('CIWPTH_OLD',iciwpth(:,:),pcols,lchnk)
   402)    !endif
   403)    !call outfld('CI_OD_LW_OLD',cldtau(:,:),pcols,lchnk)
   404) 
   405) end subroutine ec_ice_get_rad_props_lw
