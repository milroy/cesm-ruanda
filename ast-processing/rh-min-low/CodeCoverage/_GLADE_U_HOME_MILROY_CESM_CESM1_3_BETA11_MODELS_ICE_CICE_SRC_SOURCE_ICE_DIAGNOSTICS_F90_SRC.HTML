   508)          sfreshn = global_sum(fresh_gbm, distrb_info, &
   509)                               field_loc_center, tarean)
   510)          sfreshs = global_sum(fresh_gbm, distrb_info, &
   511)                               field_loc_center, tareas)
   512)          sfreshn = sfreshn*dt
   513)          sfreshs = sfreshs*dt
   514) 
   515)          ! ocean heat
   516)          ! Note: fswthru not included because it does not heat ice
   517)          fhocnn = global_sum(fhocn_gbm, distrb_info, &
   518)                              field_loc_center, tarean)
   519)          fhocns = global_sum(fhocn_gbm, distrb_info, &
   520)                              field_loc_center, tareas)
   521) 
   522)          ! latent heat
   523)          ! You may be wondering, where is the latent heat flux?
   524)          ! It is not included here because it cancels with
   525)          ! the evaporative flux times the enthalpy of the
   526)          ! ice/snow that evaporated.
   527) 
   528)          ! atmo heat flux
   529)          ! Note: flwout includes the reflected longwave down, needed by the
   530)          !  atmosphere as an upwards radiative boundary condition.
   531)          ! Also note: fswabs includes solar radiation absorbed in ocean,
   532)          !  which must be subtracted here.
   533) 
   534)          if (calc_Tsfc) then
   535) 
   536)             !$OMP PARALLEL DO PRIVATE(iblk,i,j)
   537)             do iblk = 1, nblocks
   538)                do j = 1, ny_block
   539)                do i = 1, nx_block
   540)                   work1(i,j,iblk) = &
   541)                                (fswabs(i,j,iblk) - fswthru(i,j,iblk) &
   542)                               + flwout(i,j,iblk)                          &
   543)                               + fsens (i,j,iblk)) * aice(i,j,iblk)        &
   544)                               + flw   (i,j,iblk)  * aice_init(i,j,iblk)
   545)                enddo
   546)                enddo
   547)             enddo
   548)             !$OMP END PARALLEL DO
   549) 
   550)          else   ! fsurf is computed by atmosphere model
   551) 
   552)             !$OMP PARALLEL DO PRIVATE(iblk,i,j)
   553)             do iblk = 1, nblocks
   554)                do j = 1, ny_block
   555)                do i = 1, nx_block
   556)                   work1(i,j,iblk) = &
   557)                              (fsurf(i,j,iblk) - flat(i,j,iblk)) &
   558)                               * aice(i,j,iblk)
   559)                enddo
   560)                enddo
   561)             enddo
   562)             !$OMP END PARALLEL DO
   563) 
   564)          endif     ! calc_Tsfc
   565) 
   566)          fhatmn = global_sum(work1, distrb_info, &
   567)                              field_loc_center, tarean)
   568)          fhatms = global_sum(work1, distrb_info, &
   569)                              field_loc_center, tareas)
   570)   
   571)          ! freezing potential
   572)          !$OMP PARALLEL DO PRIVATE(iblk,i,j)
   573)          do iblk = 1, nblocks
   574)             do j = 1, ny_block
   575)             do i = 1, nx_block
   576)                work1(i,j,iblk) = max(c0,frzmlt(i,j,iblk))
   577)             enddo
   578)             enddo
   579)          enddo
   580)          !$OMP END PARALLEL DO
   581)          fhfrzn = global_sum(work1, distrb_info, &
   582)                              field_loc_center, tarean)
   583)          fhfrzs = global_sum(work1, distrb_info, &
   584)                              field_loc_center, tareas)
   585) 
   586)          ! rain
   587)          rnn = global_sum_prod(frain, aice_init, distrb_info, &
   588)                                field_loc_center, tarean)
   589)          rns = global_sum_prod(frain, aice_init, distrb_info, &
   590)                                field_loc_center, tareas)
   591)          rnn = rnn*dt
   592)          rns = rns*dt
   593) 
   594)          ! snow
   595)          snn = global_sum_prod(fsnow, aice_init, distrb_info, &
   596)                                field_loc_center, tarean)
   597)          sns = global_sum_prod(fsnow, aice_init, distrb_info, &
   598)                                field_loc_center, tareas)
   599)          snn = snn*dt
   600)          sns = sns*dt
   601) 
   602)          ! frazil ice growth !! should not be multiplied by aice
   603)          ! m/step->kg/m^2/s
   604)          work1(:,:,:) = frazil(:,:,:)*rhoi/dt
   605)          frzn = global_sum(work1, distrb_info, &
   606)                            field_loc_center, tarean)
   607)          frzs = global_sum(work1, distrb_info, field_loc_center, &
   608)                            tareas)
   609)          frzn = frzn*dt
   610)          frzs = frzs*dt
   611) 
   612)          ! ice and snow mass
   613)          micen = rhoi*shmaxn
   614)          msnwn = rhos*snwmxn
   615)          mices = rhoi*shmaxs
   616)          msnws = rhos*snwmxs
   617) 
   618)          mtotn = micen + msnwn
   619)          mtots = mices + msnws
   620)   
   621)          ! mass change since beginning of time step
   622)          delmin = mtotn - totmn
   623)          delmis = mtots - totms
   624) 
   625)          ! ice mass change including frazil ice formation
   626)          delmxn = micen - totmin
   627)          delmxs = mices - totmis
   628)          if (.not. update_ocn_f) then
   629)            ! ice mass change excluding frazil ice formation
   630)            delmxn = delmxn - frzn
   631)            delmxs = delmxs - frzs
   632)          endif
   633) 
   634)          ! total water flux
   635)          fluxn  = c0
   636)          fluxs  = c0
   637)          if( arean > c0) then
   638)            ! water associated with frazil ice included in fresh
   639)            fluxn = rnn + snn + evpn - sfreshn
   640)            if (.not. update_ocn_f) then
   641)              fluxn = fluxn + frzn
   642)            endif
   643)          endif
   644)          if( areas > c0) then
   645)            ! water associated with frazil ice included in fresh
   646)            fluxs = rns + sns + evps - sfreshs
   647)            if (.not. update_ocn_f) then
   648)              fluxs = fluxs + frzs
   649)            endif
   650)          endif
   651) 
   652)          werrn = (fluxn-delmin)/(mtotn+c1)
   653)          werrs = (fluxs-delmis)/(mtots+c1)
   654) 
   655)          ! energy change
   656)          delein = etotn - toten
   657)          deleis = etots - totes
   658) 
   659)          fhatmn = fhatmn + ( - snn * Lfresh + evpn * Lvap ) / dt
   660)          fhatms = fhatms + ( - sns * Lfresh + evps * Lvap ) / dt
   661) 
   662)          hnetn = (fhatmn - fhocnn - fhfrzn) * dt
   663)          hnets = (fhatms - fhocns - fhfrzs) * dt
   664) 
   665)          herrn = (hnetn - delein) / (etotn - c1)
   666)          herrs = (hnets - deleis) / (etots - c1)
   667) 
   668)          ! salt mass
   669)          msltn = micen*ice_ref_salinity*p001
   670)          mslts = mices*ice_ref_salinity*p001
   671) 
   672)          ! change in salt mass
   673)          delmsltn = delmxn*ice_ref_salinity*p001
   674)          delmslts = delmxs*ice_ref_salinity*p001
   675) 
   676)          ! salt error
   677)          serrn = (sfsaltn + delmsltn) / (msltn + c1)
   678)          serrs = (sfsalts + delmslts) / (mslts + c1)
   679) 
   680)       endif                     ! print_global
   681)

   700)                pflw(n) = flw(i,j,iblk)             ! longwave radiation

   703)                hiavg(n) = c0                       ! avg snow/ice thickness

   707)                   hsavg(n) = vsno(i,j,iblk)/paice(n)
   708)                endif
   709)                pTsfc(n) = trcr(i,j,nt_Tsfc,iblk)   ! ice/snow sfc temperature
   710)                pevap(n) = evap(i,j,iblk)*dt/rhoi   ! sublimation/condensation
   711)                pfswabs(n) = fswabs(i,j,iblk)       ! absorbed solar flux
   712)                pflwout(n) = flwout(i,j,iblk)       ! outward longwave flux
   713)                pflat(n) = flat(i,j,iblk)           ! latent heat flux
   714)                pfsens(n) = fsens(i,j,iblk)         ! sensible heat flux
   715)                pfsurf(n) = fsurf(i,j,iblk)         ! total sfc heat flux
   716)                pfcondtop(n) = fcondtop(i,j,iblk)   ! top sfc cond flux
   717)                pmeltt(n) = meltt(i,j,iblk)         ! top melt
   718)                pmeltb(n) = meltb(i,j,iblk)         ! bottom melt
   719)                pmeltl(n) = meltl(i,j,iblk)         ! lateral melt
   720)                psnoice(n) = snoice(i,j,iblk)       ! snow ice
   721)                pfrazil(n) = frazil(i,j,iblk)       ! frazil ice
   722)                pcongel(n) = congel(i,j,iblk)       ! congelation ice
   723)                pdhi(n) = vice(i,j,iblk) - pdhi(n)  ! ice thickness change
   724)                pdhs(n) = vsno(i,j,iblk) - pdhs(n)  ! snow thickness change
   725)                pde(n) = -(eice(i,j,iblk) &         ! ice/snow energy change 
   726)                         + esno(i,j,iblk) - pde(n)) / dt
   727)                psst(n) = sst(i,j,iblk)             ! sea surface temperature
   728)                pTf(n) = Tf(i,j,iblk)               ! freezing temperature
   729)                pfhocn(n) = -fhocn(i,j,iblk)   ! ocean heat used by ice
   730) 
   731)             endif  ! my_task = pmloc
   732) 
   733)             call broadcast_scalar(pTair    (n), pmloc(n))             
   734)             call broadcast_scalar(pQa      (n), pmloc(n))             
   735)             call broadcast_scalar(pfsnow   (n), pmloc(n))             
   736)             call broadcast_scalar(pfrain   (n), pmloc(n))             
   737)             call broadcast_scalar(pfsw     (n), pmloc(n))             
   738)             call broadcast_scalar(pflw     (n), pmloc(n))             
   739)             call broadcast_scalar(paice    (n), pmloc(n))             
   740)             call broadcast_scalar(hsavg    (n), pmloc(n))

   742)             call broadcast_scalar(pTsfc    (n), pmloc(n))             
   743)             call broadcast_scalar(pevap    (n), pmloc(n))             
   744)             call broadcast_scalar(pfswabs  (n), pmloc(n)) 
   745)             call broadcast_scalar(pflwout  (n), pmloc(n)) 
   746)             call broadcast_scalar(pflat    (n), pmloc(n)) 
   747)             call broadcast_scalar(pfsens   (n), pmloc(n)) 
   748)             call broadcast_scalar(pfsurf   (n), pmloc(n))
   749)             call broadcast_scalar(pfcondtop(n), pmloc(n))
   750)             call broadcast_scalar(pmeltt   (n), pmloc(n)) 
   751)             call broadcast_scalar(pmeltb   (n), pmloc(n)) 
   752)             call broadcast_scalar(pmeltl   (n), pmloc(n)) 
   753)             call broadcast_scalar(psnoice  (n), pmloc(n)) 
   754)             call broadcast_scalar(pfrazil  (n), pmloc(n)) 
   755)             call broadcast_scalar(pcongel  (n), pmloc(n)) 
   756)             call broadcast_scalar(pdhi     (n), pmloc(n)) 
   757)             call broadcast_scalar(pdhs     (n), pmloc(n)) 
   758)             call broadcast_scalar(pde      (n), pmloc(n)) 
   759)             call broadcast_scalar(psst     (n), pmloc(n)) 
   760)             call broadcast_scalar(pTf      (n), pmloc(n)) 
   761)             call broadcast_scalar(pfhocn   (n), pmloc(n))
   762)             
   763)          enddo                  ! npnt
   764)       endif                     ! print_points
