   235)       call get_curr_date (iyear,imo,iday_m,tod)  ! year, time of day [sec]

   274)       subroutine GlobalElPotential
   275) !-----------------------------------------------------------------------
   276) ! Purpose: calculates the global electric potential field on the
   277) !          geomagnetic grid (MLT in deg) 
   278) !
   279) ! Method: rewritten code from Luedger Scherliess (11/20/99 LS)
   280) !     routine to calculate the global electric potential in magnetic
   281) !     Apex coordinates (Latitude and MLT).
   282) !     High Latitude Model is Weimer 1996.
   283) !     Midlatitude model is Scherliess 1999.
   284) !     Interpolation in a transition region at about 60 degree 
   285) !     magnetic apex lat
   286) !
   287) ! Author: A. Maute Dec 2003  am 12/17/03 
   288) !-----------------------------------------------------------------------
   289) 
   290) !-----------------------------------------------------------------------
   291) ! local variables
   292) !-----------------------------------------------------------------------
   293)       integer  :: ilon, ilat, idlat
   294)       integer  :: ihlat_bnd(0:nmlon)                  ! high latitude boundary
   295)       integer  :: itrans_width(0:nmlon)               ! width of transition zone
   296)       real(r8) :: mlt, mlon, mlat, mlat_90, pot
   297)       real(r8) :: pot_midlat(0:nmlon,0:nmlat)         ! potential from L. Scherliess model
   298)       real(r8) :: pot_highlat(0:nmlon,0:nmlat)        ! potential from Weimer model
   299)       real(r8) :: pot_highlats(0:nmlon,0:nmlat)	      ! smoothed potential from Weimer model
   300) 
   301) !-----------------------------------------------------------------------
   302) ! Externals
   303) !-----------------------------------------------------------------------
   304)       real(r8),external :: EpotVal        ! in wei96.f
   305) 
   306) !-----------------------------------------------------------------------
   307) ! convert to date and day	
   308) !-----------------------------------------------------------------------
   309)       day  = iday + ut/24._r8
   310)       date = iyear + day/dy2yr
   311) 
   312) !-----------------------------------------------------------------------
   313) ! low/midlatitude electric potential - empirical model Scherliess 1999  
   314) !-----------------------------------------------------------------------
   315) !$omp parallel do private(ilat, ilon, mlat, pot)
   316)       do ilat = 0,nmlath                        ! Calculate only for one magn. hemisphere
   317) 	mlat = ylatm(ilat)                      ! mag. latitude
   318)         do ilon = 0,nmlon	                ! lon. loop
   319)           call efield_mid( mlat, ylonm(ilon), pot )
   320) 	  pot_midlat(ilon,ilat+nmlath) = pot	! SH/NH symmetry 
   321) 	  pot_midlat(ilon,nmlath-ilat) = pot
   322)         end do
   323)       end do
   324) 
   325) !-----------------------------------------------------------------------
   326) ! hight latitude potential from Weimer model
   327) ! at the poles Weimer potential is not longitudinal dependent
   328) !-----------------------------------------------------------------------
   329)       call prep_weimer                             ! calculate IMF angle & magnitude, tilt
   330) 
   331) !$omp parallel do private(ilat, ilon, mlat_90, pot)
   332)       do ilat = 0,nmlat_wei 	                   ! Calculate only for one magn. hemisphere
   333)         mlat_90 = 90._r8 - ylatm(ilat)             ! mag. latitude
   334)         do ilon = 0,nmlon
   335)     	  pot  = 1000._r8*EpotVal( mlat_90, ylonm(ilon)*deg2mlt )     ! calculate potential (kv -> v)
   336) !-----------------------------------------------------------------------
   337) ! NH/SH symmetry
   338) !-----------------------------------------------------------------------
   339)     	  pot_highlat(ilon,ilat)        = pot
   340)     	  pot_highlat(ilon,nmlat-ilat)  = pot
   341)     	  pot_highlats(ilon,ilat)       = pot
   342)     	  pot_highlats(ilon,nmlat-ilat) = pot
   343)         end do
   344)       end do     
   345) 
   346) !-----------------------------------------------------------------------
   347) ! weighted smoothing of high latitude potential
   348) !-----------------------------------------------------------------------
   349)       idlat = 2              ! smooth over -2:2 = 5 grid points
   350)       call pot_latsmo( pot_highlats, idlat )
   351) !-----------------------------------------------------------------------
   352) ! calculate the height latitude bounday ihl_bnd
   353) ! 1. calculate E field from weimar model
   354) !    boundary is set where the total electric field exceeds
   355) !    0.015 V/m (corresp. approx. to 300 m/s)
   356) ! 2. moved halfways to 54 deg 
   357) ! output : index 0-pole nmlath-equator
   358) !-----------------------------------------------------------------------
   359)       call highlat_getbnd( ihlat_bnd )
   360) !-----------------------------------------------------------------------
   361) ! 3. adjust high latitude boundary sinusoidally
   362) !    calculate width of transition zone
   363) !-----------------------------------------------------------------------
   364)       call bnd_sinus( ihlat_bnd, itrans_width ) 
   365) !-----------------------------------------------------------------------
   366) ! 4. ajust high latitude potential to low latitude potential      
   367) !-----------------------------------------------------------------------
   368)       call highlat_adjust( pot_highlats, pot_highlat, pot_midlat, ihlat_bnd )
   369) !-----------------------------------------------------------------------
   370) ! interpolation of high and low/midlatitude potential in the
   371) ! transition zone and put it into global potent array
   372) !-----------------------------------------------------------------------
   373)       call interp_poten( pot_highlats, pot_highlat, pot_midlat, ihlat_bnd, itrans_width) 
   374) !-----------------------------------------------------------------------
   375) ! potential weighted smoothing in latitude
   376) !-----------------------------------------------------------------------
   377)       idlat = 2                 ! smooth over -2:2 = 5 grid points
   378)       call pot_latsmo2( potent, idlat )
   379) !-----------------------------------------------------------------------
   380) ! potential smoothing in longitude
   381) !-----------------------------------------------------------------------
   382)       idlat = nmlon/48          ! smooth over -idlat:idlat grid points
   383)       call pot_lonsmo( potent, idlat )
   384) 
   385) !-----------------------------------------------------------------------
   386) ! output
   387) !-----------------------------------------------------------------------
   388) ! output ( change later to netcdf file)
   389) !      do ilat=0,nmlat
   390) !       do ilon=0,nmlon
   391) !         write(iulog,'(4(x,f12.5))') ylatm(ilat),ylonm(ilon), &
   392) !           potent(ilon,ilat),potent(ilon,nmlat-ilat)
   393) !         write(iulog,'(4(x,f12.5))') ylatm(ilat),ylonm(ilon), &
   394) !           potent(ilon,ilat),potent(ilon,nmlat-ilat)
   395) !	write(iulog,'(f10.3)') potent(ilon,ilat)
   396) !       end do
   397) !      end do
   398) 
   399)       end subroutine GlobalElPotential

   401)       subroutine ff( ph, mt, f )                                                    
   402) !-----------------------------------------------------------------------
   403) ! Purpose: calculate F for normalized associated Legendre polynomial P_n^m
   404) !          Ref.: Richmond J.Atm.Ter.Phys. 1974
   405) !
   406) ! Method:  f_m(phi) = sqrt(2) sin(m phi) m > 0
   407) !                   = 1                  m = 0
   408) !                   = sqrt(2) cos(m phi) m < 0
   409) !
   410) ! Author: A. Maute Nov 2003  am 11/18/03
   411) !-----------------------------------------------------------------------
   412) 
   413)       implicit none
   414) 
   415) !-----------------------------------------------------------------------
   416) ! dummy arguments
   417) !-----------------------------------------------------------------------
   418)       integer,intent(in)   :: mt
   419)       real(r8),intent(in)  :: ph	! geo. longitude of 0SLT (ut*15)
   420)       real(r8),intent(out) :: f(-mt:mt)
   421) 
   422) !-----------------------------------------------------------------------
   423) ! local variables
   424) !-----------------------------------------------------------------------
   425)       integer  :: m, i, j, mmo
   426)       real(r8) :: sp, cp    
   427) 
   428)       sp   = sin( ph/rtd )
   429)       cp   = cos( ph/rtd )
   430)       f(0) = 1.e0_r8
   431)                                                                 
   432)       f(-1) = sqr2*cp
   433)       f(1)  = sqr2*sp      								 
   434)       do m = 2,mt
   435)         mmo   = m - 1  
   436)         f(m)  = f(-mmo)*sp + cp*f(mmo)
   437)         f(-m) = f(-mmo)*cp - sp*f(mmo)
   438)       end do      
   439) 
   440)       end subroutine ff

   442)       subroutine pnm( ct, p )
   443) !----------------------------------------------------------------------------                                                                   
   444) ! Purpose: normalized associated Legendre polynomial P_n^m
   445) !          Ref.: Richmond J.Atm.Ter.Phys. 1974
   446) ! Method:
   447) !   P_m^m    = sqrt(1+1/2m)*si*P_m-1^m-1                  m>0
   448) !   P_n^m    = [cos*P_n-1^m - R_n-1^m*P_n-2^m ]/R_n^m     n>m>=0
   449) !   dP/d phi = n*cos*P_n^m/sin-(2*n+1)*R_n^m*P_n-1^m/sin  n>=m>=0
   450) !   R_n^m    = sqrt[ (n^2-m^2)/(4n^2-1) ]
   451) !
   452) ! Author: A. Maute Nov 2003  am 11/18/03
   453) !----------------------------------------------------------------------------                                                                   
   454) 
   455)       implicit none
   456) 
   457) !-----------------------------------------------------------------------
   458) ! dummy arguments
   459) !-----------------------------------------------------------------------
   460)       real(r8), intent(inout) :: ct ! cos(colat)                 
   461)       real(r8), intent(inout) :: p(0:nm,0:mm)
   462) 
   463) !-----------------------------------------------------------------------
   464) ! local variables
   465) !-----------------------------------------------------------------------
   466)       integer  :: mp, m, n, np
   467)       real(r8) :: pm2, st
   468) 
   469) !      ct = min( ct,.99_r8 )		! cos(colat)
   470)       st = sqrt( 1._r8 - ct*ct ) 	! sin(colat)
   471) 
   472)       p(0,0) = 1._r8  
   473)       do mp = 1,mmp  ! m+1=1,mm+1
   474)         m = mp - 1
   475) 	if( m >= 1 ) then
   476)            p(m,m) = pmopmmo(m)*p(m-1,m-1)*st 			
   477) 	end if
   478) 	pm2 = 0._r8                                                                  
   479) 	do n = mp,nm                    ! n=m+1,N
   480) 	   np     = n + 1
   481) 	   p(n,m) = (ct*p(n-1,m) - r(n-1,m)*pm2)/r(n,m)
   482) 	   pm2    = p(n-1,m)
   483)         end do
   484)       end do
   485) 
   486)       end subroutine pnm

   488)       subroutine prep_pnm
   489) !----------------------------------------------------------------------------                                                                   
   490) ! Purpose: constant factors for normalized associated Legendre polynomial P_n^m
   491) !          Ref.: Richmond J.Atm.Ter.Phys. 1974
   492) !
   493) ! Method:
   494) !   PmoPmmo(m) = sqrt(1+1/2m)
   495) !   R_n^m      = sqrt[ (n^2-m^2)/(4n^2-1) ]
   496) !
   497) ! Author: A. Maute Nov 2003  am 11/18/03
   498) !----------------------------------------------------------------------------                                                                   
   499) 
   500)       implicit none                
   501) 
   502) !-----------------------------------------------------------------------
   503) ! local variables
   504) !-----------------------------------------------------------------------
   505)       integer  :: mp, m, n
   506)       real(r8) :: xms, xns, den
   507) 
   508)       do mp = 1, mmp            ! m+1 = 1,mm+1                                     
   509) 	m = mp - 1                                               
   510) 	xms = m*m                                                
   511) 	if( mp /= 1 ) then
   512)            pmopmmo(m) = sqrt( 1._r8 + .5_r8/M )
   513) 	end if
   514) 	do n = m,nm      ! n = m,N                                     
   515) 	  xns    = n*n                                       
   516) 	  den    = max(4._r8*xns - 1._r8,1._r8)
   517) 	  r(n,m) = sqrt( (xns  - xms)/den )
   518) 	end do                 
   519)       end do 
   520) 
   521)       end subroutine prep_pnm

   525) ! Purpose: set up index for factors f_m(mlt),f_l(UT),f_-k(d) to
   526) !    describe the electric potential Phi for the empirical model   
   527) !
   528) ! Method:
   529) !    Phi = sum_k sum_l sum_m sum_n [ A_klmn * P_n^m *f_m(mlt)*f_l(UT)*f_-k(d)]
   530) !    - since the electric potential is symmetric about the equator
   531) !      n+m odd terms are set zero resp. not used
   532) !    - in the summation for calculation Phi the index have the following
   533) !      range n=1,12 and m=-n,n, k=0,2 l=-2,2

   539)

   542) !----------------------------------------------------------------------------

   544) 
   545)       i = 0 	! initialize
   546)       j = 1 
   547)       do k = 2,0,-1
   548)         do l = -2,2
   549)           if( k == 2 .and. abs(l) == 2 ) then
   550)              cycle
   551)           end if
   552)           do n = 1,12
   553)             do m = -18,18 
   554)               if( abs(m) <= n ) then		    !  |m| < n

   559)              	    nf(i) = n
   560)              	    mf(i) = m
   561)              	    jf(i) = j
   562)              	    i	  = i + 1	 ! counter

   564)                 end if
   565)               end if
   566)             end do ! m
   567)           end do ! n
   568)         end do ! l
   569)       end do ! k
   570) 
   571)       imax = i - 1  
   572)       if(imax /= ni ) then    ! check if imax == ni 
   573)         write(iulog,'(a19,i5,a18,i5)') 'index_quiet: imax= ',imax, &

   718)       subroutine constants
   719) !----------------------------------------------------------------------------                                                                   
   720) ! Purpose: set up constant values (e.g. magnetic grid, convertion
   721) !      constants etc)
   722) !
   723) ! Method:
   724) !
   725) ! Author: A. Maute Nov 2003  am 11/19/03
   726) !----------------------------------------------------------------------------                                                                   
   727) 
   728) !----------------------------------------------------------------------------                                                                   
   729) ! local variables
   730) !----------------------------------------------------------------------------                                                                   
   731)       integer  :: i,j
   732)       real(r8) :: fac,lat
   733) 
   734)       rtd     = 180._r8/pi	        ! radians -> deg
   735)       dtr     = pi/180._r8	        ! deg -> radians
   736)       sqr2    = sqrt(2.e0_r8)
   737)       hr2rd   = pi/12._r8	        ! pi/12 hrs
   738)       dy2rd   = 2._r8*pi/dy2yr          ! 2*pi/365.24  average year
   739)       deg2mlt = 24._r8/360._r8          ! convert degrees to MLT hours

   751)       do j = 0,nmlat
   752)         ylatm(j) = j*dlatm

   754) 	fac = cos(lat)                   ! sinIm = 2*sin(lam_m)/sqrt[4-3*cos^2(lam_m)]
   755) 	fac = 4._r8 - 3._r8*fac*fac
   756) 	fac = 2._r8/sqrt( fac )
   757) 	sinIm_mag(j) = fac*sin( lat )

   791)       do j = 0,nmlat

   794) 	  jmin = j

   891)       subroutine efield_mid( mlat, mlon, pot )
   892) !----------------------------------------------------------------------------                                                                   
   893) ! Purpose: calculate the electric potential for low and 
   894) !      midlatitudes from an empirical model (Scherliess 1999)
   895) !
   896) ! Method:
   897) !
   898) ! Author: A. Maute Nov 2003  am 11/20/03
   899) !----------------------------------------------------------------------------                                                                   
   900) 
   901) !----------------------------------------------------------------------------                                                                   
   902) !	... dummy arguments
   903) !----------------------------------------------------------------------------                                                                   
   904)       real(r8), intent(in)  :: mlat, mlon
   905)       real(r8), intent(out) :: pot               ! electric potential (V)
   906) 
   907) !----------------------------------------------------------------------------                                                                   
   908) ! local variables
   909) !----------------------------------------------------------------------------                                                                   
   910)       integer  :: i, mp, np, nn
   911)       real(r8) :: mod_mlat, ct, x
   912)       real(r8) :: fk(0:2)      	    ! f_-k(day) 
   913)       real(r8) :: fl(-2:2)          ! f_l(ut)  
   914)       real(r8) :: fs(2)	            ! f_s(f10.7) 
   915)       real(r8) :: f(-18:18)
   916)       real(r8) :: p(0:nm,0:mm)      ! P_n^m	 spherical harmonics
   917) 
   918)       pot = 0._r8 ! initialize                                        
   919) 
   920)       mod_mlat = mlat
   921)       if( abs(mlat) <= 0.5_r8 ) then
   922)          mod_mlat = 0.5_r8                     ! avoid geomag.equator
   923)       end if
   924)

   927) !----------------------------------------------------------------------------                                                                   
   928)       call set_fkflfs( fk, fl, fs )

   931) ! spherical harmonics 
   932) !----------------------------------------------------------------------------                                                                   
   933)       ct = cos( (90._r8 - mod_mlat)*dtr )  ! magnetic colatitude 
   934)       call pnm( ct, p )	                   ! calculate P_n^m

   940)         nn  = abs(mp)                      !   P_n^m = P_n^-m  
   941)         x   = a_klnm(i)* fl(lf(i)) * fk(kf(i)) * fs(jf(i))

  1251) ! 1.adjust boundary sinusoidally
  1252) !   max. wave number to be represented nmax_sin
  1253) !   RHS(mi) = Sum_phi Sum_(mi=-nmax_sin)^_(mi=nmax_sin) f_mi(phi)*hlat_bnd(phi) 
  1254) !   U(mi,mk)   = Sum_phi Sum_(mi=-nmax_sin)^_(mi=nmax_sin) f_mi(phi) *
  1255) !                Sum_(mk=-nmax_sin)^_(mk=nmax_sin) f_mk(phi)
  1256) !   single values decomposition of U
  1257) !   solving U*LSG = RHS

  1260) !

  1262) !    potential

  1298) 
  1299)       do ilon = 0,nmlon                  ! long.

  1304)           rhs(i1) = rhs(i1) + f(i,ilon) * bnd

  1307)             u(i1,j+ishf) = u(i1,j+ishf) + f(i,ilon)*f(j,ilon)
  1308) !	    write(iulog,*) 'u ',ilon,i1,j+ishf,u(i+ishf,j+ishf)

  1310)         end do

  1318) !

  1320) !       sum = 0._r8
  1321) 	sum = dot_product( lsg(-nmax_sin+ishf:nmax_sin+ishf),f(-nmax_sin:nmax_sin,ilon) )
  1322) !       do i = -nmax_sin,nmax_sin

  1324) !       end do
  1325)         ihlat_bnd(ilon)    = nmlath - int( sum + .5_r8 )                                ! closest point
  1326)         itrans_width(ilon) = int( 8._r8 - 2._r8*cos( ylonm(ilon)*dtr ) + .5_r8 )/dlatm  ! 6 to 10 deg.

  1333)       subroutine highlat_adjust( pot_highlats, pot_highlat, pot_midlat, ihlat_bnd )

  1336) !          potential such that there are continous across the mid to high

  1339) ! Method:
  1340) ! 1. integrate Phi_low/mid(phi,bnd) along the boundary mid to high latitude
  1341) ! 2. integrate Phi_high(phi,bnd) along the boundary mid to high latitude
  1342) ! 3. adjust Phi_high by delta =
  1343) !    Int_phi Phi_high(phi,bnd) d phi/360. - Int_phi Phi_low/mid(phi,bnd) d phi/360.
  1344) !
  1345) ! Author: A. Maute Nov 2003  am 11/21/03
  1346) !----------------------------------------------------------------------------                                                                   
  1347) 
  1348) !----------------------------------------------------------------------------                                                                   
  1349) !	... dummy arguments
  1350) !----------------------------------------------------------------------------

  1353)       real(r8), intent(inout) :: pot_highlat(0:nmlon,0:nmlat)             ! high_lat potential

  1367)       pot_hl = 0._r8
  1368)       do ilon = 1,nmlon  ! long.           ! bnd -> eq to pole 0:90
  1369)     	ibnd60  = nmlat - ihlat_bnd(ilon)   ! 0:180 pole to pole
  1370)     	ibnd_hl = ihlat_bnd(ilon)         ! colatitude
  1371)         pot60   = pot60 + pot_midlat(ilon,ibnd60)
  1372)         pot_hl  = pot_hl + pot_highlats(ilon,ibnd_hl)

  1399)       subroutine interp_poten( pot_highlats, pot_highlat, pot_midlat, &
  1400)                                ihlat_bnd, itrans_width ) 
  1401) !----------------------------------------------------------------------------                                                                   
  1402) ! Purpose: construct a smooth global electric potential field 
  1403) !
  1404) ! Method: construct one global potential field
  1405) ! 1. low/mid latitude: |lam| < bnd-trans_width
  1406) !   Phi(phi,lam) = Phi_low(phi,lam)
  1407) ! 2. high latitude: |lam| > bnd+trans_width
  1408) !   Phi(phi,lam) = Phi_hl(phi,lam)
  1409) ! 3. transition zone: bnd-trans_width <= lam <= bnd+trans_width 
  1410) ! a. interpolate between high and low/midlatitude potential
  1411) !   Phi*(phi,lam) = 1/15*[ 5/(2*trans_width) * {Phi_low(phi,bnd-trans_width)*
  1412) !   [-lam+bnd+trans_width] + Phi_hl(phi,bnd+trans_width)*
  1413) !   [lam-bnd+trans_width]} + 10/(2*trans_width) {Phi_low(phi,lam)*
  1414) !   [-lam+bnd+trans_width] + Phi_hl(phi,lam)*
  1415) !   [lam-bnd+trans_width]}]
  1416) ! b.  Interpolate between just calculated Potential and the high latitude
  1417) !    potential in a 3 degree zone poleward of the boundary:
  1418) !    bnd+trans_width < lam <= bnd+trans_width+ 3 deg 
  1419) !   Phi(phi,lam) = 1/3 { [3-(lam-bnd-trans_width)]* Phi*(phi,lam) +
  1420) !   [lam-bnd-trans_width)]* Phi_hl*(phi,lam) }
  1421) !
  1422) ! Author: A. Maute Nov 2003  am 11/21/03      
  1423) !----------------------------------------------------------------------------                                                                   
  1424) 
  1425) !----------------------------------------------------------------------------                                                                   
  1426) !	... dummy arguments
  1427) !----------------------------------------------------------------------------                                                                   
  1428)       integer, intent(in)  :: ihlat_bnd(0:nmlon)
  1429)       integer, intent(in)  :: itrans_width(0:nmlon)
  1430)       real(r8), intent(in) :: pot_highlats(0:nmlon,0:nmlat)
  1431)       real(r8), intent(in) :: pot_highlat(0:nmlon,0:nmlat)
  1432)       real(r8), intent(in) :: pot_midlat(0:nmlon,0:nmlat)
  1433) 
  1434) !----------------------------------------------------------------------------                                                                   
  1435) ! local variables
  1436) !----------------------------------------------------------------------------                                                                   
  1437)       real(r8), parameter :: fac = 1._r8/3._r8
  1438)       integer  :: ilon, ilat
  1439)       integer  :: ibnd, tw, hb1, hb2, lat_ind
  1440)       integer  :: j1, j2
  1441)       real(r8) :: a, b, lat, b1, b2
  1442)       real(r8) :: wrk1, wrk2
  1443) 
  1444) !$omp parallel do private(ilat,ilon,ibnd,tw)
  1445)       do ilon = 0,nmlon
  1446)         ibnd = ihlat_bnd(ilon)     ! high latitude boundary index
  1447) 	tw   = itrans_width(ilon)  ! width of transition zone (index)
  1448) !----------------------------------------------------------------------------                                                                   
  1449) ! 1. low/mid latitude: |lam| < bnd-trans_width
  1450) !   Phi(phi,lam) = Phi_low(phi,lam)
  1451) !----------------------------------------------------------------------------                                                                   
  1452)         do ilat = 0,nmlath-(ibnd+tw+1)
  1453)           potent(ilon,nmlath+ilat) = pot_midlat(ilon,nmlath+ilat)
  1454)           potent(ilon,nmlath-ilat) = pot_midlat(ilon,nmlath+ilat)
  1455)         end do
  1456) !----------------------------------------------------------------------------                                                                   
  1457) ! 2. high latitude: |lam| > bnd+trans_width
  1458) !   Phi(phi,lam) = Phi_hl(phi,lam)
  1459) !----------------------------------------------------------------------------                                                                   
  1460)         do ilat = 0,ibnd-tw-1
  1461)           potent(ilon,ilat)       = pot_highlats(ilon,nmlat-ilat)
  1462)           potent(ilon,nmlat-ilat) = pot_highlats(ilon,nmlat-ilat)
  1463)         end do
  1464)       end do
  1465) !----------------------------------------------------------------------------                                                                   
  1466) ! 3. transition zone: bnd-trans_width <= lam <= bnd+trans_width 
  1467) !----------------------------------------------------------------------------                                                                   
  1468) ! a. interpolate between high and low/midlatitude potential
  1469) ! update only southern hemisphere (northern hemisphere is copied
  1470) ! after smoothing)
  1471) !----------------------------------------------------------------------------                                                                   
  1472) !$omp parallel do private(ilat,ilon,ibnd,tw,a,b,b1,b2,hb1,hb2,lat_ind,j1,j2,wrk1,wrk2)
  1473)       do ilon = 0,nmlon

  1512) ! Method:  Richmond [1995] eqn 5.9-5.10

  1516) ! used in the TIEGCM code

  1518) ! Author: A. Maute Dec 2003  am 12/16/03
  1519) !-----------------------------------------------------------------------

  1521)       integer  :: i, j, ip1f, ip2f, ip3f
  1522)       real(r8) :: coslm, r, fac, wrk
  1523)       real(r8) :: wrk1d(0:nmlon)
  1524) 
  1525)       r = r_e + h_r  ! earth radius + reference height [m]
  1526) !-----------------------------------------------------------------------
  1527) ! ed2= Ed2 is the equatorward/downward component of the electric field, at all 
  1528) ! geomagnetic grid points (central differencing)
  1529) !-----------------------------------------------------------------------
  1530)       fac = .5_r8/(dlatm*dtr*r)

  1535)           ed2(i,j) = (potent(i,j+1) - potent(i,j-1))*wrk
  1536)         end do

  1538) 
  1539) !$omp parallel do private(j, i, wrk )
  1540)       do j = nmlath+1,nmlat-1	! northern hemisphere

  1544)         end do
  1545)       end do

  1549) !-----------------------------------------------------------------------

  1551)       do j = jmin+1,jmax-1
  1552)         fac = (ylatm(j) - ylatm(jmin))/(ylatm(jmax) - ylatm(jmin))
  1553)         do i = 0,nmlon
  1554) 	    ed2(i,j) = ed2(i,jmin) + fac*wrk1d(i)

  1560) ! geomagnetic grid points (central differencing)

  1570)         end do

  1572) 	ed1(i,j)     = -(potent(i+1,j) - potent(nmlon-1,j))*wrk
  1573) 	ed1(nmlon,j) = ed1(i,j)

  1577) ! Poles:

  1591)         end if

  1593)         ed1(i,nmlat) = .25_r8*(ed1(i,nmlat-1) - ed1(ip2f,nmlat-1) &
