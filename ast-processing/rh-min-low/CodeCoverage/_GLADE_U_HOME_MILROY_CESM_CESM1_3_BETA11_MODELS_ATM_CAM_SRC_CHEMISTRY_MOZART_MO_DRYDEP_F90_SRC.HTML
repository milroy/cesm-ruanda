   209)     !-------------------------------------------------------------------------------------
   210)     ! combines the deposition velocities provided by the land model with deposition 
   211)     ! velocities over ocean and sea ice

   223)     !-------------------------------------------------------------------------------------
   224)     ! 	... dummy arguments

   226) 
   227)     real(r8), intent(in)    :: icefrac(pcols)            
   228)     real(r8), intent(in)    :: ocnfrac(pcols)            
   229) 
   230)     integer, intent(in)   :: ncol
   231)     integer, intent(in)   :: ncdate                   ! present date (yyyymmdd)
   232)     real(r8), intent(in)      :: sfc_temp(pcols)          ! surface temperature (K)
   233)     real(r8), intent(in)      :: pressure_sfc(pcols)      ! surface pressure (Pa)

   242)     real(r8), intent(in)      :: solar_flux(pcols)        ! direct shortwave radiation at surface (W/m^2)
   243)     real(r8), intent(in)      :: tv(pcols)                ! potential temperature

   246)     real(r8), intent(inout)   :: dflx(pcols,gas_pcnst)        ! deposition flux (/cm^2/s)

   249)     integer, intent(in)     ::   lonndx(pcols)           ! chunk longitude indicies

   264) #endif
   265) 
   266)     lndfrac(:ncol) = 1._r8 - ocnfrac(:ncol) - icefrac(:ncol)
   267) 
   268)     where( lndfrac(:ncol) < 0._r8 ) 
   269)        lndfrac(:ncol) = 0._r8 
   270)     endwhere
   271) 
   272) #if (defined OFFLINE_DYN)

   287)                          wind_speed, spec_hum, air_temp, pressure_10m, rain, &

   336)     endif

   786)     ierr = pio_inq_varid( piofile, 'lon', vid )
   787)     ierr = pio_get_var( piofile, vid, dvel_lons )
   788) 
   789)     !---------------------------------------------------------------------------
   790)     !       ... Set the transform from inputs lats to simulation lats
   791)     !---------------------------------------------------------------------------
   792)     dvel_lats(:nlat) = d2r * dvel_lats(:nlat)
   793)     dvel_lons(:nlon) = d2r * dvel_lons(:nlon)
   794) 
   795)     !---------------------------------------------------------------------------
   796)     !     	... Allocate dvel and read data from file
   797)     !---------------------------------------------------------------------------
   798)     allocate( dvel_in(nlon, nlat ,nspecies, nmonth), stat=ierr )
   799)     if( ierr /= 0 ) then
   800)        write(iulog,*) 'dvel_inti: Failed to allocate dvel_in'
   801)        call endrun
   802)     end if
   803)     start = (/ 1, 1, 1, 1 /)
   804)     count = (/ nlon, nlat, nspecies, nmonth /)
   805) 
   806)     ierr = pio_get_var( piofile, vid_dvel, start, count, dvel_in )
   807) 
   808) 
   809)     !---------------------------------------------------------------------------
   810)     !     	... Check units of deposition velocity. If necessary, convert to cm/s.
   811)     !---------------------------------------------------------------------------
   812)     units(:) = ' '
   813)     ierr = pio_get_att( piofile, vid_dvel, 'units', units )
   814)     if( to_lower(trim(units(:GLC(units)))) == 'm/s' ) then
   815) #ifdef DEBUG
   816)        if(masterproc)  write(iulog,*) 'dvel_inti: depvel units = m/s. Converting to cm/s'
   817) #endif
   818)        scale_factor = 100._r8
   819)     elseif( to_lower(trim(units(:GLC(units)))) == 'cm/s' ) then
   820) #ifdef DEBUG
   821)        if(masterproc)  write(iulog,*) 'dvel_inti: depvel units = cm/s'
   822) #endif
   823)        scale_factor = 1._r8
   824)     else
   825) #ifdef DEBUG
   826)        if(masterproc) then
   827)           write(iulog,*) 'dvel_inti: Warning! depvel units unknown = ', to_lower(trim(units)) 
   828)           write(iulog,*) '           ...      proceeding with scale_factor=1'
   829)        end if
   830) #endif
   831)        scale_factor = 1._r8
   832)     end if
   833) 
   834)     dvel_in(:,:,:,:) = scale_factor*dvel_in(:,:,:,:)
   835) 
   836)     !---------------------------------------------------------------------------
   837)     !     	... Regrid deposition velocities
   838)     !---------------------------------------------------------------------------
   839)     allocate( dvel(pcols,begchunk:endchunk,nspecies,nmonth),stat=ierr )
   840)     if( ierr /= 0 ) then
   841)        write(iulog,*) 'dvel_inti: Failed to allocate dvel'
   842)        call endrun
   843)     end if
   844) 
   845)     do c=begchunk,endchunk
   846)        ncols = get_ncols_p(c)
   847)        call get_rlat_all_p(c, pcols, to_lats)
   848)        call get_rlon_all_p(c, pcols, to_lons)
   849)        call lininterp_init(dvel_lons, nlon, to_lons, ncols, 2, lon_wgts, zero, twopi)
   850)        call lininterp_init(dvel_lats, nlat, to_lats, ncols, 1, lat_wgts)
   851) 
   852)        do ispecies = 1,nspecies
   853)           do m = 1,12
   854)              call lininterp( dvel_in( :,:,ispecies,m ), nlon, nlat, dvel(:,c,ispecies,m), ncols,lon_wgts,lat_wgts)
   855)           end do
   856)        end do
   857) 
   858)        call lininterp_finish(lat_wgts)
   859)        call lininterp_finish(lon_wgts)
   860)     end do
   861) 
   862)     deallocate( dvel_in )
   863)     deallocate( dvel_lats, dvel_lons )
   864) 
   865)     !---------------------------------------------------------------------------
   866)     !     	... Read in species names and determine mapping to tracer numbers
   867)     !---------------------------------------------------------------------------
   868)     allocate( species_names(nspecies), stat=ierr )
   869)     if( ierr /= 0 ) then
   870)        write(iulog,*) 'dvel_inti: species_names allocation error = ',ierr
   871)        call endrun
   872)     end if
   873)     ierr = pio_inq_varid( piofile, 'species_name', vid )
   874)     ierr = pio_inq_varndims( piofile, vid, ndims )
   875) 
   876)     ierr = pio_inq_vardimid( piofile, vid, dimid )
   877) 
   878)     ierr = pio_inq_dimlen( piofile, dimid(1), nchar )
   879)     map(:) = 0
   880)     do ispecies = 1,nspecies
   881)        start(:2) = (/ 1, ispecies /)
   882)        count(:2) = (/ nchar, 1 /)
   883)        species_names(ispecies)(:) = ' '
   884)        ierr = pio_get_var( piofile, vid, start(1:2), count(1:2), species_names(ispecies:ispecies) )
   885)        if( species_names(ispecies) == 'O3' ) then
   886)           o3_in_tab  = .true.
   887)           o3_tab_ndx = ispecies
   888)        else if( species_names(ispecies) == 'H2O2' ) then
   889)           h2o2_in_tab  = .true.
   890)           h2o2_tab_ndx = ispecies
   891)        else if( species_names(ispecies) == 'CH3OOH' ) then
   892)           ch3ooh_in_tab  = .true.
   893)           ch3ooh_tab_ndx = ispecies
   894)        else if( species_names(ispecies) == 'CO' ) then
   895)           co_in_tab  = .true.
   896)           co_tab_ndx = ispecies
   897)        else if( species_names(ispecies) == 'CH3CHO' ) then
   898)           ch3cho_in_tab  = .true.
   899)           ch3cho_tab_ndx = ispecies
   900)        end if
   901)        found = .false.
   902)        do m = 1,gas_pcnst
   903)           if( species_names(ispecies) == solsym(m) .or. &
   904)                (species_names(ispecies) == 'O3' .and. solsym(m) == 'OX') .or. &
   905)                (species_names(ispecies) == 'HNO4' .and. solsym(m) == 'HO2NO2') ) then
   906)              if ( has_drydep( solsym(m) ) ) then
   907)                 map(m) = ispecies
   908)                 found = .true.
   909) #ifdef DEBUG
   910)                 if( masterproc ) then
   911)                    write(iulog,*) 'dvel_inti: ispecies, m, tracnam = ',ispecies,m,trim(solsym(m))
   912)                 end if
   913) #endif
   914)                 exit
   915)              end if
   916)           end if
   917)        end do
   918)        if( .not. found ) then
   919)           write(iulog,*) 'dvel_inti: Warning! DVEL species ',trim(species_names(ispecies)),' not found'
   920)        endif
   921)     end do
   922)     deallocate( species_names )
   923) 
   924)     call pio_closefile( piofile )
   925) 
   926)     !---------------------------------------------------------------------------
   927)     !     	... Allocate dvel_interp array
   928)     !---------------------------------------------------------------------------
   929)     allocate( dvel_interp(pcols,begchunk:endchunk,nspecies),stat=ierr )
   930)     if( ierr /= 0 ) then
   931)        write(iulog,*) 'dvel_inti: Failed to allocate dvel_interp; error = ',ierr
   932)        call endrun
   933)     end if
   934) 
   935)   end subroutine dvel_inti_table

   939)   subroutine interpdvel( calday, ncol, lchnk )
   940)     !---------------------------------------------------------------------------
   941)     ! 	... Interpolate the fields whose values are required at the
   942)     !           begining of a timestep.
   943)     !---------------------------------------------------------------------------
   944) 
   945)     use time_manager,  only : get_calday
   946) 
   947)     implicit none
   948) 
   949)     !---------------------------------------------------------------------------
   950)     ! 	... Dummy arguments
   951)     !---------------------------------------------------------------------------
   952)     real(r8), intent(in) :: calday   ! Interpolate the input data to calday
   953)     integer, intent(in) :: ncol, lchnk
   954) 
   955)     !---------------------------------------------------------------------------
   956)     ! 	... Local variables
   957)     !---------------------------------------------------------------------------
   958)     integer :: m, last, next
   959)     integer  ::  dates(12) = (/ 116, 214, 316, 415,  516,  615, &
   960)                                 716, 816, 915, 1016, 1115, 1216 /)
   961)     real(r8) :: calday_loc, last_days, next_days
   962)     real(r8), save ::  dys(12)
   963)     logical, save  ::  entered = .false.
   964) 
   965)     if( .not. entered ) then
   966)        do m = 1,12
   967)           dys(m) = get_calday( dates(m), 0 )
   968)        end do
   969)        entered = .true.
   970)     end if
   971) 
   972)     if( calday < dys(1) ) then
   973)        next = 1
   974)        last = 12
   975)     else if( calday >= dys(12) ) then
   976)        next = 1
   977)        last = 12
   978)     else
   979)        do m = 11,1,-1
   980)           if( calday >= dys(m) ) then
   981)              exit
   982)           end if
   983)        end do
   984)        last = m
   985)        next = m + 1
   986)     end if
   987) 
   988)     last_days  = dys( last )
   989)     next_days  = dys( next )
   990)     calday_loc = calday
   991) 
   992)     if( next_days < last_days ) then
   993)        next_days = next_days + 365._r8
   994)     end if
   995)     if( calday_loc < last_days ) then
   996)        calday_loc = calday_loc + 365._r8
   997)     end if
   998) 
   999)     do m = 1,nspecies
  1000)        call intp2d( last_days, next_days, calday_loc, ncol, lchnk, &
  1001)                     dvel(:,lchnk,m,last), &
  1002)                     dvel(:,lchnk,m,next), &
  1003)                     dvel_interp(:,lchnk,m) )
  1004)     end do
  1005) 
  1006)   end subroutine interpdvel

  1010)   subroutine intp2d( t1, t2, tint, ncol, lchnk, f1, f2, fint )
  1011)     !-----------------------------------------------------------------------
  1012)     ! 	... Linearly interpolate between f1(t1) and f2(t2) to fint(tint).
  1013)     !-----------------------------------------------------------------------
  1014) 
  1015)     implicit none
  1016) 
  1017)     !-----------------------------------------------------------------------
  1018)     ! 	... Dummy arguments
  1019)     !-----------------------------------------------------------------------
  1020)     real(r8), intent(in) :: &
  1021)          t1, &            ! time level of f1
  1022)          t2, &            ! time level of f2
  1023)          tint             ! interpolant time
  1024)     real(r8), dimension(pcols), intent(in) :: &
  1025)          f1, &            ! field at time t1
  1026)          f2               ! field at time t2
  1027) 
  1028)     integer, intent(in) :: ncol, lchnk
  1029) 
  1030)     real(r8), intent(out) :: &
  1031)          fint(pcols) ! field at time tint
  1032) 
  1033) 
  1034)     !-----------------------------------------------------------------------
  1035)     ! 	... Local variables
  1036)     !-----------------------------------------------------------------------
  1037)     integer  :: j, plat
  1038)     real(r8) :: factor
  1039) 
  1040)     factor = (tint - t1)/(t2 - t1)
  1041) 
  1042)     fint(:ncol) = f1(:ncol) + (f2(:ncol) - f1(:ncol))*factor
  1043) 
  1044)   end subroutine intp2d

  1048)   subroutine drydep_table( calday, tsurf, zen_angle, &
  1049)                            depvel, dflx, q, p, &
  1050)                            tv, ncol, icefrac, ocnfrac, lchnk )
  1051)     !--------------------------------------------------------
  1052)     !       ... Form the deposition velocities for this
  1053)     !           latitude slice
  1054)     !--------------------------------------------------------
  1055) 
  1056)     use physconst,     only : rair,pi
  1057)     use dycore,        only : dycore_is
  1058) 
  1059)     implicit none
  1060) 
  1061)     !--------------------------------------------------------
  1062)     !       ... Dummy arguments
  1063)     !--------------------------------------------------------
  1064)     integer, intent(in)     ::   ncol                    ! columns in chunk
  1065)     real(r8), intent(in)    ::  q(pcols,plev,gas_pcnst) ! tracer mmr (kg/kg)
  1066)     real(r8), intent(in)    ::  p(pcols)            ! midpoint pressure in surface layer (Pa)
  1067)     real(r8), intent(in)    ::  tv(pcols)           ! virtual temperature in surface layer (K)
  1068)     real(r8), intent(in)    ::  calday              ! time of year in days
  1069)     real(r8), intent(in)    ::  tsurf(pcols)        ! surface temperature (K)
  1070)     real(r8), intent(in)    ::  zen_angle(ncol)    ! zenith angle (radians)
  1071)     real(r8), intent(inout) ::  dflx(pcols,gas_pcnst)   ! flux due to dry deposition (kg/m^2/sec)
  1072)     real(r8), intent(out)   ::  depvel(ncol,gas_pcnst) ! deposition vel (cm/s)
  1073) 
  1074)     real(r8), intent(in) :: icefrac(pcols)          ! sea-ice areal fraction
  1075)     real(r8), intent(in) :: ocnfrac(pcols)          ! ocean areal fraction
  1076)     
  1077)     integer, intent(in)     :: lchnk
  1078)     !-----------------------------------------------------------------------
  1079)     ! 	... Local variables
  1080)     !-----------------------------------------------------------------------
  1081)     integer :: m, spc_ndx, tmp_ndx, i
  1082)     real(r8), dimension(ncol) :: vel, glace, temp_fac, wrk, tmp
  1083)     real(r8), dimension(ncol) :: o3_tab_dvel
  1084)     real(r8), dimension(ncol) :: ocean 
  1085) 
  1086)     real(r8), parameter :: pid2 = .5_r8 * pi
  1087) 
  1088)     if(dycore_is('UNSTRUCTURED')) then
  1089)        call endrun( 'Option not supported for unstructured atmosphere grids ')
  1090)     end if
  1091) 
  1092)     !-----------------------------------------------------------------------
  1093)     !       ... Note the factor 1.e-2 in the wrk array calculation is
  1094)     !           to transform the incoming dep vel from cm/s to m/s
  1095)     !-----------------------------------------------------------------------
  1096)     wrk(:ncol) =  1.e-2_r8 * p(:ncol) / (rair * tv(:ncol))
  1097) 
  1098)     !--------------------------------------------------------
  1099)     !       ... Initialize all deposition velocities to zero
  1100)     !--------------------------------------------------------
  1101)     depvel(:,:) = 0._r8
  1102) 
  1103)     !--------------------------------------------------------
  1104)     !       ... Time interpolate primary depvel array
  1105)     !           (also seaice and npp)
  1106)     !--------------------------------------------------------
  1107)     call interpdvel( calday, ncol, lchnk )
  1108) 
  1109)     if( o3_in_tab ) then
  1110)        do i=1,ncol
  1111)           o3_tab_dvel(i) = dvel_interp(i,lchnk,o3_tab_ndx)
  1112)        enddo
  1113)     end if
  1114) 
  1115)     !--------------------------------------------------------
  1116)     !       ... Set deposition velocities
  1117)     !--------------------------------------------------------
  1118)     do m = 1,gas_pcnst
  1119)        if( map(m) /= 0 ) then
  1120)           do i = 1,ncol
  1121)              depvel(i,m) = dvel_interp(i,lchnk,map(m))
  1122)              dflx(i,m)   = wrk(i) * depvel(i,m) * q(i,plev,m)
  1123)           enddo
  1124)        end if
  1125)     end do
  1126) 
  1127)     !--------------------------------------------------------
  1128)     !       ... Set some variables needed for some dvel calculations
  1129)     !--------------------------------------------------------
  1130)     temp_fac(:ncol)   = min( 1._r8, max( 0._r8, (tsurf(:ncol) - 268._r8) / 5._r8 ) )
  1131)     ocean(:ncol)  = icefrac(:ncol)+ocnfrac(:ncol)
  1132)     glace(:ncol)  = icefrac(:ncol) + (1._r8 - ocean(:ncol)) * (1._r8 - temp_fac(:ncol))
  1133)     glace(:ncol)  = min( 1._r8,glace(:ncol) )
  1134) 
  1135)     !--------------------------------------------------------
  1136)     !       ... Set pan & mpan
  1137)     !--------------------------------------------------------
  1138)     if( o3_in_tab ) then
  1139)        tmp(:ncol) = o3_tab_dvel(:ncol) / 3._r8
  1140)     else
  1141)        tmp(:) = 0._r8
  1142)     end if
  1143)     if( pan_dd ) then
  1144)        if( map(pan_ndx) == 0 ) then
  1145)           depvel(:ncol,pan_ndx) = tmp(:ncol)
  1146)           dflx(:ncol,pan_ndx)   = wrk(:ncol) * tmp(:ncol) * q(:ncol,plev,pan_ndx)
  1147)        end if
  1148)     end if
  1149)     if( mpan_dd ) then
  1150)        if( map(mpan_ndx) == 0 ) then
  1151)           depvel(:ncol,mpan_ndx) = tmp(:ncol)
  1152)           dflx(:ncol,mpan_ndx)   = wrk(:ncol) * tmp(:ncol) * q(:ncol,plev,mpan_ndx)
  1153)        end if
  1154)     end if
  1155) 
  1156)     !--------------------------------------------------------
  1157)     !       ... Set no2 dvel
  1158)     !--------------------------------------------------------
  1159)     if( no2_dd ) then
  1160)        if( map(no2_ndx) == 0 .and. o3_in_tab ) then
  1161)           depvel(:ncol,no2_ndx) = (.6_r8*o3_tab_dvel(:ncol) + .055_r8*ocean(:ncol)) * .9_r8
  1162)           dflx(:ncol,no2_ndx)   = wrk(:) * depvel(:ncol,no2_ndx) * q(:ncol,plev,no2_ndx)
  1163)        end if
  1164)     end if
  1165) 
  1166)     !--------------------------------------------------------
  1167)     !       ... Set hno3 dvel
  1168)     !--------------------------------------------------------
  1169)     tmp(:ncol) = (2._r8 - ocnfrac(:ncol)) * (1._r8 - glace(:ncol)) + .05_r8 * glace(:ncol)
  1170)     if( hno3_dd ) then
  1171)        if( map(hno3_ndx) == 0 ) then
  1172)           depvel(:ncol,hno3_ndx) = tmp(:ncol)
  1173)           dflx(:ncol,hno3_ndx)   = wrk(:ncol) * tmp(:ncol) * q(:ncol,plev,hno3_ndx)
  1174)        else
  1175)           tmp(:ncol) = depvel(:ncol,hno3_ndx)
  1176)        end if
  1177)     end if
  1178)     if( onitr_dd ) then
  1179)        if( map(onitr_ndx) == 0 ) then
  1180)           depvel(:ncol,onitr_ndx) = tmp(:ncol)
  1181)           dflx(:ncol,onitr_ndx)   = wrk(:ncol) * tmp(:ncol) * q(:ncol,plev,onitr_ndx)
  1182)        end if
  1183)     end if
  1184)     if( isopooh_dd ) then
  1185)        if( map(isopooh_ndx) == 0 ) then
  1186)           depvel(:ncol,isopooh_ndx) = tmp(:ncol)
  1187)           dflx(:ncol,isopooh_ndx)   = wrk(:ncol) * tmp(:ncol) * q(:ncol,plev,isopooh_ndx)
  1188)        end if
  1189)     end if
  1190) 
  1191)     !--------------------------------------------------------
  1192)     !       ... Set h2o2 dvel
  1193)     !--------------------------------------------------------
  1194)     if( .not. h2o2_in_tab ) then
  1195)        if( o3_in_tab ) then
  1196)           tmp(:ncol) = .05_r8*glace(:ncol) + ocean(:ncol) - icefrac(:ncol) &
  1197)                + (1._r8 - (glace(:) + ocean(:ncol)) + icefrac(:ncol)) &
  1198)                *max( 1._r8,1._r8/(.5_r8 + 1._r8/(6._r8*o3_tab_dvel(:ncol))) )
  1199)        else
  1200)           tmp(:ncol) = 0._r8
  1201)        end if
  1202)     else
  1203)        do i=1,ncol
  1204)           tmp(i) = dvel_interp(i,lchnk,h2o2_tab_ndx)
  1205)        enddo
  1206)     end if
  1207)     if( h2o2_dd ) then
  1208)        if( map(h2o2_ndx) == 0 ) then
  1209)           depvel(:ncol,h2o2_ndx) = tmp(:ncol)
  1210)           dflx(:ncol,h2o2_ndx)   = wrk(:ncol) * tmp(:ncol) * q(:ncol,plev,h2o2_ndx)
  1211)        end if
  1212)     end if
  1213)     !--------------------------------------------------------
  1214)     !       ... Set hcn dvel
  1215)     !--------------------------------------------------------
  1216)     if( hcn_dd ) then
  1217)        if( map(hcn_ndx) == 0 ) then
  1218)           depvel(:ncol,hcn_ndx) = ocnfrac(:ncol)*0.2_r8
  1219)        endif
  1220)     endif
  1221)     !--------------------------------------------------------
  1222)     !       ... Set ch3cn dvel
  1223)     !--------------------------------------------------------
  1224)     if( ch3cn_dd ) then
  1225)        if( map(ch3cn_ndx) == 0 ) then
  1226)           depvel(:,ch3cn_ndx) = ocnfrac(:ncol)*0.2_r8
  1227)        endif
  1228)     endif
  1229)     !--------------------------------------------------------
  1230)     !       ... Set onit
  1231)     !--------------------------------------------------------
  1232)     if( onit_dd ) then
  1233)        if( map(onit_ndx) == 0 ) then
  1234)           depvel(:ncol,onit_ndx) = tmp(:ncol)
  1235)           dflx(:ncol,onit_ndx)   = wrk(:ncol) * tmp(:ncol) * q(:ncol,plev,onit_ndx)
  1236)        end if
  1237)     end if
  1238)     if( ch3cocho_dd ) then
  1239)        if( map(ch3cocho_ndx) == 0 ) then
  1240)           depvel(:ncol,ch3cocho_ndx) = tmp(:ncol)
  1241)           dflx(:ncol,ch3cocho_ndx)   = wrk(:ncol) * tmp(:ncol) * q(:ncol,plev,ch3cocho_ndx)
  1242)        end if
  1243)     end if
  1244)     if( ch3ooh_in_tab ) then
  1245)        do i=1,ncol
  1246)           tmp(i) = dvel_interp(i,lchnk,ch3ooh_tab_ndx)
  1247)        enddo
  1248)     else
  1249)        tmp(:ncol) = .5_r8 * tmp(:ncol)
  1250)     end if
  1251)     if( ch3ooh_dd ) then
  1252)        if( map(ch3ooh_ndx) == 0 ) then
  1253)           depvel(:ncol,ch3ooh_ndx) = tmp(:ncol)
  1254)           dflx(:ncol,ch3ooh_ndx)   = wrk(:ncol) * tmp(:ncol) * q(:ncol,plev,ch3ooh_ndx)
  1255)        end if
  1256)     end if
  1257)     if( pooh_dd ) then
