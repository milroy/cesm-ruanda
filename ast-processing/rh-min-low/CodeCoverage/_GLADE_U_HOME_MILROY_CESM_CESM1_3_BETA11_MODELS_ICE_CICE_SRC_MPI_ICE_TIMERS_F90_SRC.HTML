   143)  subroutine init_ice_timers
   144) 
   145) ! !DESCRIPTION:
   146) !  This routine initializes machine parameters and timer structures
   147) !  for computing cpu time from F90 intrinsic timer functions.

   149) ! !REVISION HISTORY:
   150) !  same as module

   165) !  initialize timer structures

   167) !-----------------------------------------------------------------------

   169)    return

   172)

   242)  subroutine get_ice_timer(timer_id, name_choice, num_blocks, num_nodes)
   243) 
   244) ! !DESCRIPTION:
   245) !  This routine initializes a timer with a given name and returns a 
   246) !  timer id.
   247) !
   248) ! !REVISION HISTORY:
   249) !  same as module
   250) 
   251) ! !INPUT PARAMETERS:
   252) 
   253)    character (*), intent(in) :: &
   254)       name_choice               ! input name for this timer
   255) 
   256)    integer (int_kind), intent(in) :: &
   257)       num_nodes,               &! number of nodes(tasks) using this timer
   258)       num_blocks                ! number of blocks using this timer
   259)                                 ! (can be =1 if timer called outside
   260)                                 !  threaded region)
   261) 
   262) ! !OUTPUT PARAMETERS:
   263) 
   264)    integer (int_kind), intent(out) :: &
   265)       timer_id           ! timer number assigned to this timer 
   266) 
   267) !EOP
   268) !BOC
   269) !-----------------------------------------------------------------------
   270) !
   271) !  local variables
   272) !
   273) !-----------------------------------------------------------------------
   274) 
   275)    integer (int_kind) :: &
   276)       n,                 &! dummy loop index
   277)       srch_error          ! error flag for search
   278) 
   279) !-----------------------------------------------------------------------
   280) !
   281) !  search for next free timer
   282) !
   283) !-----------------------------------------------------------------------
   284) 
   285)    return
   286)    srch_error = 1
   287) 
   288)    srch_loop: do n=1,max_timers
   289)       if (.not. all_timers(n)%in_use) then
   290)          srch_error = 0
   291)          timer_id = n
   292) 
   293)          all_timers(n)%name       = ' '
   294)          all_timers(n)%name       = name_choice
   295)          all_timers(n)%in_use     = .true.
   296)          all_timers(n)%num_blocks = num_blocks
   297)          all_timers(n)%num_nodes  = num_nodes 
   298) 
   299)          allocate(all_timers(n)%block_started   (num_blocks), &
   300)                   all_timers(n)%block_cycles1   (num_blocks), &
   301)                   all_timers(n)%block_cycles2   (num_blocks), &
   302)                   all_timers(n)%block_accum_time(num_blocks))
   303) 
   304)          all_timers(n)%block_started    = .false.
   305)          all_timers(n)%block_cycles1    = c0
   306)          all_timers(n)%block_cycles2    = c0
   307)          all_timers(n)%block_accum_time = c0
   308) 
   309)          exit srch_loop
   310)       endif
   311)    end do srch_loop
   312) 
   313)    if (srch_error /= 0) &
   314)       call abort_ice('get_ice_timer: Exceeded maximum number of timers')
   315)                     
   316) 
   317) !-----------------------------------------------------------------------
   318) !EOC
   319) 
   320)  end subroutine get_ice_timer

   381)  subroutine ice_timer_start(timer_id, block_id)
   382)  use perf_mod
   383) 
   384) ! !DESCRIPTION:
   385) !  This routine starts a given node timer if it has not already
   386) !  been started by another thread.  If block information is available,
   387) !  the appropriate block timer is also started.
   388) !
   389) ! !REVISION HISTORY:
   390) !  same as module
   391) 
   392) ! !INPUT PARAMETERS:
   393) 
   394)    integer (int_kind), intent(in) :: &
   395)       timer_id                   ! timer number
   396) 
   397)    integer (int_kind), intent(in), optional :: &
   398)       block_id                 ! optional block id for this block
   399)                                ! this must be the actual local address
   400)                                ! of the block in the distribution
   401)                                ! from which it is called
   402)                                ! (if timer called outside of block
   403)                                ! region, no block info required)
   404) 
   405)    double precision MPI_WTIME
   406)    external MPI_WTIME
   407) 
   408) !EOP
   409) !BOC
   410) !-----------------------------------------------------------------------
   411) !
   412) !  if timer is defined, start it up
   413) !
   414) !-----------------------------------------------------------------------
   415) 
   416)    return
   417) 
   418)    if (all_timers(timer_id)%in_use) then
   419) 
   420)       call t_startf(trim(all_timers(timer_id)%name))
   421) 
   422)       !***
   423)       !*** if called from within a block loop, start block timers
   424)       !***
   425) 
   426)       if (present(block_id)) then
   427) 
   428)          !*** if block timer already started, stop it first
   429) 
   430)          if (all_timers(timer_id)%block_started(block_id)) &
   431)             call ice_timer_stop(timer_id, block_id)
   432) 
   433)          !*** start block timer
   434) 
   435)          all_timers(timer_id)%block_started(block_id) = .true.
   436)          all_timers(timer_id)%block_cycles1(block_id) = MPI_WTIME()
   437) 
   438)          !*** start node timer if not already started by
   439)          !*** another thread.  if already started, keep track
   440)          !*** of number of start requests in order to match
   441)          !*** start and stop requests
   442)  
   443)          !$OMP CRITICAL
   444) 
   445)          if (.not. all_timers(timer_id)%node_started) then
   446)             all_timers(timer_id)%node_started = .true.
   447)             all_timers(timer_id)%num_starts   = 1
   448)             all_timers(timer_id)%num_stops    = 0
   449)             all_timers(timer_id)%node_cycles1 = MPI_WTIME()
   450)          else
   451)             all_timers(timer_id)%num_starts = &
   452)             all_timers(timer_id)%num_starts + 1
   453)          endif
   454) 
   455)          !$OMP END CRITICAL
   456) 
   457)       !***
   458)       !*** if called from outside a block loop, start node timer
   459)       !***
   460) 
   461)       else
   462) 
   463)          !*** stop timer if already started
   464)          if (all_timers(timer_id)%node_started)  &
   465)                                         call ice_timer_stop(timer_id)
   466) 
   467)          !*** start node timer
   468) 
   469)          all_timers(timer_id)%node_started = .true.
   470)          all_timers(timer_id)%node_cycles1 = MPI_WTIME()
   471) 
   472)       endif
   473)    else
   474)       call abort_ice &
   475)                  ('ice_timer_start: attempt to start undefined timer')
   476)                     
   477)    endif
   478) 
   479) !-----------------------------------------------------------------------
   480) !EOC
   481) 
   482)  end subroutine ice_timer_start

   487) ! !INTERFACE:
   488) 
   489)  subroutine ice_timer_stop(timer_id, block_id)
   490)  use perf_mod
   491) 
   492) ! !DESCRIPTION:
   493) !  This routine stops a given node timer if appropriate.  If block 
   494) !  information is available the appropriate block timer is also stopped.
   495) !
   496) ! !REVISION HISTORY:
   497) !  same as module
   498) 
   499) ! !INPUT PARAMETERS:
   500) 
   501)    integer (int_kind), intent(in) :: &
   502)       timer_id                   ! timer number
   503) 
   504)    integer (int_kind), intent(in), optional :: &
   505)       block_id                 ! optional block id for this block
   506)                                ! this must be the actual local address
   507)                                ! of the block in the distribution
   508)                                ! from which it is called
   509)                                ! (if timer called outside of block
   510)                                ! region, no block info required)
   511) 
   512)    double precision MPI_WTIME
   513)    external MPI_WTIME
   514) 
   515) !EOP
   516) !BOC
   517) !-----------------------------------------------------------------------
   518) !
   519) !  local variables
   520) !
   521) !-----------------------------------------------------------------------
   522) 
   523)    real (dbl_kind) :: &
   524)       cycles1, cycles2   ! temps to hold cycle info before correction
   525) 
   526) !-----------------------------------------------------------------------
   527) !
   528) !  get end cycles
   529) !
   530) !-----------------------------------------------------------------------
   531) 
   532)    return
   533)    cycles2 = MPI_WTIME()
   534) 
   535) !-----------------------------------------------------------------------
   536) !
   537) !  if timer is defined, stop it
   538) !
   539) !-----------------------------------------------------------------------
   540) 
   541)    if (all_timers(timer_id)%in_use) then
   542) 
   543)       !***
   544)       !*** if called from within a block loop, stop block timer
   545)       !***
   546) 
   547)       if (present(block_id)) then
   548) 
   549)          all_timers(timer_id)%block_started(block_id) = .false.
   550) 
   551)          cycles1 = all_timers(timer_id)%block_cycles1(block_id)
   552)          all_timers(timer_id)%block_accum_time(block_id) = &
   553)          all_timers(timer_id)%block_accum_time(block_id) + &
   554)             clock_rate*(cycles2 - cycles1)
   555) 
   556)          !*** stop node timer if number of requested stops
   557)          !*** matches the number of starts (to avoid stopping
   558)          !*** a node timer started by multiple threads)
   559)  
   560)          cycles1 = all_timers(timer_id)%node_cycles1
   561) 
   562)          !$OMP CRITICAL
   563) 
   564)          all_timers(timer_id)%num_stops = &
   565)          all_timers(timer_id)%num_stops + 1
   566) 
   567)          if (all_timers(timer_id)%num_starts == &
   568)              all_timers(timer_id)%num_stops) then
   569) 
   570)             all_timers(timer_id)%node_started = .false.
   571)             all_timers(timer_id)%node_accum_time = &
   572)             all_timers(timer_id)%node_accum_time + &
   573)                clock_rate*(cycles2 - cycles1)
   574) 
   575)             all_timers(timer_id)%num_starts   = 0
   576)             all_timers(timer_id)%num_stops    = 0
   577) 
   578)             all_ltime(timer_id) = all_timers(timer_id)%node_accum_time
   579) 
   580)          endif
   581) 
   582)          !$OMP END CRITICAL
   583) 
   584)       !***
   585)       !*** if called from outside a block loop, stop node timer
   586)       !***
   587) 
   588)       else
   589) 
   590)          all_timers(timer_id)%node_started = .false.
   591)          cycles1 = all_timers(timer_id)%node_cycles1
   592) 
   593)          all_timers(timer_id)%node_accum_time = &
   594)          all_timers(timer_id)%node_accum_time + &
   595)             clock_rate*(cycles2 - cycles1)
   596) 
   597)          all_ltime(timer_id) = all_timers(timer_id)%node_accum_time
   598) 
   599)       endif
   600) 
   601)       call t_stopf(trim(all_timers(timer_id)%name))
   602)    else
   603)       call abort_ice &
   604)                  ('ice_timer_stop: attempt to stop undefined timer')
   605)                     
   606)    endif
   607) 
   608) !-----------------------------------------------------------------------
   609) !EOC
   610) 
   611)  end subroutine ice_timer_stop
