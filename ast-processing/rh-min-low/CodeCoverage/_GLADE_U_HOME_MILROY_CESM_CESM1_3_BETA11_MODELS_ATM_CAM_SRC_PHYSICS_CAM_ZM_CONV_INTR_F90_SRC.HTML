    74) !----------------------------------------

    91)    call pbuf_add_field('DP_CLDLIQ','global',dtype_r8,(/pcols,pver/), dp_cldliq_idx)

    93) ! deep gbm cloud liquid water (kg/kg)

    95)

   100) subroutine zm_conv_init(pref_edge)
   101) 
   102) !----------------------------------------
   103) ! Purpose:  declare output fields, initialize variables needed by convection
   104) !----------------------------------------
   105) 
   106)   use cam_history,    only: outfld, addfld, add_default, phys_decomp
   107)   use ppgrid,         only: pcols, pver
   108)   use zm_conv,        only: zm_convi
   109)   use pmgrid,         only: plev,plevp
   110)   use spmd_utils,     only: masterproc
   111)   use error_messages, only: alloc_err	
   112)   use phys_control,   only: phys_deepconv_pbl, phys_getopts, cam_physpkg_is
   113)   use physics_buffer, only: pbuf_get_index
   114) 
   115)   implicit none
   116) 
   117)   real(r8),intent(in) :: pref_edge(plevp)        ! reference pressures at interfaces
   118) 
   119) 
   120)   logical :: no_deep_pbl    ! if true, no deep convection in PBL
   121)   integer  limcnv           ! top interface level limit for convection
   122)   integer k, istat
   123)   logical :: history_budget ! output tendencies and state variables for CAM4
   124)                             ! temperature, water vapor, cloud ice and cloud
   125)                             ! liquid budgets.
   126)   integer :: history_budget_histfile_num ! output history file number for budget fields
   127) 
   128) !
   129) ! Allocate space for arrays private to this module
   130) !
   131)      allocate( mu(pcols,pver,begchunk:endchunk), stat=istat )
   132)       call alloc_err( istat, 'zm_conv_tend', 'mu', &
   133)                       pcols*pver*((endchunk-begchunk)+1) )
   134)      allocate( eu(pcols,pver,begchunk:endchunk), stat=istat )
   135)       call alloc_err( istat, 'zm_conv_tend', 'eu', &
   136)                       pcols*pver*((endchunk-begchunk)+1) )
   137)      allocate( du(pcols,pver,begchunk:endchunk), stat=istat )
   138)       call alloc_err( istat, 'zm_conv_tend', 'du', &
   139)                       pcols*pver*((endchunk-begchunk)+1) )
   140)      allocate( md(pcols,pver,begchunk:endchunk), stat=istat )
   141)       call alloc_err( istat, 'zm_conv_tend', 'md', &
   142)                       pcols*pver*((endchunk-begchunk)+1) )
   143)      allocate( ed(pcols,pver,begchunk:endchunk), stat=istat )
   144)       call alloc_err( istat, 'zm_conv_tend', 'ed', &
   145)                       pcols*pver*((endchunk-begchunk)+1) )
   146)      allocate( dp(pcols,pver,begchunk:endchunk), stat=istat )
   147)       call alloc_err( istat, 'zm_conv_tend', 'dp', &
   148)                       pcols*pver*((endchunk-begchunk)+1) )
   149)      allocate( dsubcld(pcols,begchunk:endchunk), stat=istat )
   150)       call alloc_err( istat, 'zm_conv_tend', 'dsubcld', &
   151)                       pcols*((endchunk-begchunk)+1) )
   152)      allocate( jt(pcols,begchunk:endchunk), stat=istat )
   153)       call alloc_err( istat, 'zm_conv_tend', 'jt', &
   154)                       pcols*((endchunk-begchunk)+1) )
   155)      allocate( maxg(pcols,begchunk:endchunk), stat=istat )
   156)       call alloc_err( istat, 'zm_conv_tend', 'maxg', &
   157)                       pcols*((endchunk-begchunk)+1) )
   158)      allocate( ideep(pcols,begchunk:endchunk), stat=istat )
   159)       call alloc_err( istat, 'zm_conv_tend', 'ideep', &
   160)                       pcols*((endchunk-begchunk)+1) )
   161)      allocate( lengath(begchunk:endchunk), stat=istat )
   162)       call alloc_err( istat, 'zm_conv_tend', 'lengath', &
   163)                       ((endchunk-begchunk)+1) )
   164) 
   165) 
   166) ! 
   167) ! Register fields with the output buffer
   168) !
   169) 
   170) 
   171)     call addfld ('PRECZ   ','m/s     ',1,    'A','total precipitation from ZM convection',        phys_decomp)
   172)     call addfld ('ZMDT    ','K/s     ',pver, 'A','T tendency - Zhang-McFarlane moist convection', phys_decomp)
   173)     call addfld ('ZMDQ    ','kg/kg/s ',pver, 'A','Q tendency - Zhang-McFarlane moist convection', phys_decomp)
   174)     call addfld ('ZMDICE ','kg/kg/s ',pver, 'A','Cloud ice tendency - Zhang-McFarlane convection',phys_decomp)
   175)     call addfld ('ZMDLIQ ','kg/kg/s ',pver, 'A','Cloud liq tendency - Zhang-McFarlane convection',phys_decomp)
   176)     call addfld ('EVAPTZM ','K/s     ',pver, 'A','T tendency - Evaporation/snow prod from Zhang convection',phys_decomp)
   177)     call addfld ('FZSNTZM ','K/s     ',pver, 'A','T tendency - Rain to snow conversion from Zhang convection',phys_decomp)
   178)     call addfld ('EVSNTZM ','K/s     ',pver, 'A','T tendency - Snow to rain prod from Zhang convection',phys_decomp)
   179)     call addfld ('EVAPQZM ','kg/kg/s ',pver, 'A','Q tendency - Evaporation from Zhang-McFarlane moist convection',phys_decomp)
   180)     
   181)     call addfld ('ZMFLXPRC','kg/m2/s ',pverp, 'A','Flux of precipitation from ZM convection'       ,phys_decomp)
   182)     call addfld ('ZMFLXSNW','kg/m2/s ',pverp, 'A','Flux of snow from ZM convection'                ,phys_decomp)
   183)     call addfld ('ZMNTPRPD','kg/kg/s ',pver , 'A','Net precipitation production from ZM convection',phys_decomp)
   184)     call addfld ('ZMNTSNPD','kg/kg/s ',pver , 'A','Net snow production from ZM convection'         ,phys_decomp)
   185)     call addfld ('ZMEIHEAT','W/kg'    ,pver , 'A','Heating by ice and evaporation in ZM convection',phys_decomp)
   186)     
   187)     call addfld ('CMFMCDZM','kg/m2/s ',pverp,'A','Convection mass flux from ZM deep ',phys_decomp)
   188)     call addfld ('PRECCDZM','m/s     ',1,    'A','Convective precipitation rate from ZM deep',phys_decomp)
   189) 
   190)     call addfld ('PCONVB','Pa'    ,1 , 'A','convection base pressure',phys_decomp)
   191)     call addfld ('PCONVT','Pa'    ,1 , 'A','convection top  pressure',phys_decomp)
   192) 
   193)     call addfld ('CAPE',   'J/kg',       1, 'A', 'Convectively available potential energy', phys_decomp)
   194)     call addfld ('FREQZM ','fraction  ',1  ,'A', 'Fractional occurance of ZM convection',phys_decomp) 
   195) 
   196)     call addfld ('ZMMTT ', 'K/s',     pver, 'A', 'T tendency - ZM convective momentum transport',phys_decomp)
   197)     call addfld ('ZMMTU',  'm/s2',    pver, 'A', 'U tendency - ZM convective momentum transport',  phys_decomp)
   198)     call addfld ('ZMMTV',  'm/s2',    pver, 'A', 'V tendency - ZM convective momentum transport',  phys_decomp)
   199) 
   200)     call addfld ('ZMMU',   'kg/m2/s', pver, 'A', 'ZM convection updraft mass flux',   phys_decomp)
   201)     call addfld ('ZMMD',   'kg/m2/s', pver, 'A', 'ZM convection downdraft mass flux', phys_decomp)
   202) 
   203)     call addfld ('ZMUPGU', 'm/s2',    pver, 'A', 'zonal force from ZM updraft pressure gradient term',       phys_decomp)
   204)     call addfld ('ZMUPGD', 'm/s2',    pver, 'A', 'zonal force from ZM downdraft pressure gradient term',     phys_decomp)
   205)     call addfld ('ZMVPGU', 'm/s2',    pver, 'A', 'meridional force from ZM updraft pressure gradient term',  phys_decomp)
   206)     call addfld ('ZMVPGD', 'm/s2',    pver, 'A', 'merdional force from ZM downdraft pressure gradient term', phys_decomp)
   207) 
   208)     call addfld ('ZMICUU', 'm/s',     pver, 'A', 'ZM in-cloud U updrafts',      phys_decomp)
   209)     call addfld ('ZMICUD', 'm/s',     pver, 'A', 'ZM in-cloud U downdrafts',    phys_decomp)
   210)     call addfld ('ZMICVU', 'm/s',     pver, 'A', 'ZM in-cloud V updrafts',      phys_decomp)
   211)     call addfld ('ZMICVD', 'm/s',     pver, 'A', 'ZM in-cloud V downdrafts',    phys_decomp)
   212)     
   213)     call phys_getopts( history_budget_out = history_budget, &
   214)                        history_budget_histfile_num_out = history_budget_histfile_num)
   215) 
   216)     if ( history_budget ) then
   217)        call add_default('EVAPTZM  ', history_budget_histfile_num, ' ')
   218)        call add_default('EVAPQZM  ', history_budget_histfile_num, ' ')
   219)        call add_default('ZMDT     ', history_budget_histfile_num, ' ')
   220)        call add_default('ZMDQ     ', history_budget_histfile_num, ' ')
   221)        call add_default('ZMDLIQ   ', history_budget_histfile_num, ' ')
   222)        call add_default('ZMDICE   ', history_budget_histfile_num, ' ')
   223) 
   224)        if( cam_physpkg_is('cam4') .or. cam_physpkg_is('cam5') ) then
   225)           call add_default('ZMMTT    ', history_budget_histfile_num, ' ')
   226)        end if
   227) 
   228)     end if
   229) !
   230) ! Limit deep convection to regions below 40 mb
   231) ! Note this calculation is repeated in the shallow convection interface
   232) !
   233)     limcnv = 0   ! null value to check against below
   234)     if (pref_edge(1) >= 4.e3_r8) then
   235)        limcnv = 1
   236)     else
   237)        do k=1,plev
   238)           if (pref_edge(k) < 4.e3_r8 .and. pref_edge(k+1) >= 4.e3_r8) then
   239)              limcnv = k
   240)              exit
   241)           end if
   242)        end do
   243)        if ( limcnv == 0 ) limcnv = plevp
   244)     end if
   245)     
   246)     if (masterproc) then
   247)        write(iulog,*)'ZM_CONV_INIT: Deep convection will be capped at intfc ',limcnv, &
   248)             ' which is ',pref_edge(limcnv),' pascals'
   249)     end if
   250)         
   251)     no_deep_pbl = phys_deepconv_pbl()
   252)     call zm_convi(limcnv,no_deep_pbl_in = no_deep_pbl)
   253) 
   254)     cld_idx         = pbuf_get_index('CLD')
   255)     icwmrdp_idx     = pbuf_get_index('ICWMRDP')
   256)     rprddp_idx      = pbuf_get_index('RPRDDP')
   257)     fracis_idx      = pbuf_get_index('FRACIS')
   258)     nevapr_dpcu_idx = pbuf_get_index('NEVAPR_DPCU')
   259)     prec_dp_idx     = pbuf_get_index('PREC_DP')
   260)     snow_dp_idx     = pbuf_get_index('SNOW_DP')
   261) 
   262) end subroutine zm_conv_init

   266) subroutine zm_conv_tend(pblh    ,mcon    ,cme     , &
   267)      tpert   ,dlf     ,pflx    ,zdu      , &
   268)      rliq    , &
   269)      ztodt   , &
   270)      jctop   ,jcbot , &
   271)      state   ,ptend_all   ,landfrac,  pbuf)
   272)   
   273) 
   274)    use cam_history,   only: outfld
   275)    use physics_types, only: physics_state, physics_ptend
   276)    use physics_types, only: physics_ptend_init, physics_update
   277)    use physics_types, only: physics_state_copy, physics_state_dealloc
   278)    use physics_types, only: physics_ptend_sum, physics_ptend_dealloc
   279) 
   280)    use phys_grid,     only: get_lat_p, get_lon_p
   281)    use time_manager,  only: get_nstep, is_first_step
   282)    use physics_buffer, only : pbuf_get_field, physics_buffer_desc, pbuf_old_tim_idx
   283)    use constituents,  only: pcnst, cnst_get_ind, cnst_is_convtran1
   284)    use check_energy,  only: check_energy_chng
   285)    use physconst,     only: gravit
   286)    use phys_control,  only: cam_physpkg_is
   287) 
   288)    ! Arguments
   289) 
   290)    type(physics_state), intent(in )   :: state          ! Physics state variables
   291)    type(physics_ptend), intent(out)   :: ptend_all      ! individual parameterization tendencies
   292)    type(physics_buffer_desc), pointer :: pbuf(:)
   293) 
   294)    real(r8), intent(in) :: ztodt                       ! 2 delta t (model time increment)
   295)    real(r8), intent(in) :: pblh(pcols)                 ! Planetary boundary layer height
   296)    real(r8), intent(in) :: tpert(pcols)                ! Thermal temperature excess
   297)    real(r8), intent(in) :: landfrac(pcols)             ! RBN - Landfrac 
   298) 
   299)    real(r8), intent(out) :: mcon(pcols,pverp)  ! Convective mass flux--m sub c
   300)    real(r8), intent(out) :: dlf(pcols,pver)    ! scattrd version of the detraining cld h2o tend
   301)    real(r8), intent(out) :: pflx(pcols,pverp)  ! scattered precip flux at each level
   302)    real(r8), intent(out) :: cme(pcols,pver)    ! cmf condensation - evaporation
   303)    real(r8), intent(out) :: zdu(pcols,pver)    ! detraining mass flux
   304) 
   305)    real(r8), intent(out) :: rliq(pcols) ! reserved liquid (not yet in cldliq) for energy integrals
   306) 
   307) 
   308)    ! Local variables
   309) 
   310)    integer :: i,k,m
   311)    integer :: ilon                      ! global longitude index of a column
   312)    integer :: ilat                      ! global latitude index of a column
   313)    integer :: nstep
   314)    integer :: ixcldice, ixcldliq      ! constituent indices for cloud liquid and ice water.
   315)    integer :: lchnk                   ! chunk identifier
   316)    integer :: ncol                    ! number of atmospheric columns
   317)    integer :: itim_old                ! for physics buffer fields
   318) 
   319)    real(r8) :: ftem(pcols,pver)              ! Temporary workspace for outfld variables
   320)    real(r8) :: ntprprd(pcols,pver)    ! evap outfld: net precip production in layer
   321)    real(r8) :: ntsnprd(pcols,pver)    ! evap outfld: net snow production in layer
   322)    real(r8) :: tend_s_snwprd  (pcols,pver) ! Heating rate of snow production
   323)    real(r8) :: tend_s_snwevmlt(pcols,pver) ! Heating rate of evap/melting of snow
   324)    real(r8) :: fake_dpdry(pcols,pver) ! used in convtran call
   325) 
   326)    ! physics types
   327)    type(physics_state) :: state1        ! locally modify for evaporation to use, not returned
   328)    type(physics_ptend) :: ptend_loc     ! package tendencies
   329) 
   330)    ! physics buffer fields
   331)    real(r8), pointer, dimension(:)   :: prec         ! total precipitation
   332)    real(r8), pointer, dimension(:)   :: snow         ! snow from ZM convection 
   333)    real(r8), pointer, dimension(:,:) :: cld
   334)    real(r8), pointer, dimension(:,:) :: ql           ! wg grid slice of cloud liquid water.
   335)    real(r8), pointer, dimension(:,:) :: rprd         ! rain production rate
   336)    real(r8), pointer, dimension(:,:,:) :: fracis  ! fraction of transported species that are insoluble
   337)    real(r8), pointer, dimension(:,:) :: evapcdp      ! Evaporation of deep convective precipitation
   338)    real(r8), pointer, dimension(:,:) :: flxprec      ! Convective-scale flux of precip at interfaces (kg/m2/s)
   339)    real(r8), pointer, dimension(:,:) :: flxsnow      ! Convective-scale flux of snow   at interfaces (kg/m2/s)
   340)    real(r8), pointer, dimension(:,:) :: dp_cldliq
   341)    real(r8), pointer, dimension(:,:) :: dp_cldice
   342)    real(r8) :: jctop(pcols)  ! o row of top-of-deep-convection indices passed out.
   343)    real(r8) :: jcbot(pcols)  ! o row of base of cloud indices passed out.
   344) 
   345)    real(r8) :: pcont(pcols), pconb(pcols), freqzm(pcols)
   346) 
   347)    ! history output fields
   348)    real(r8) :: cape(pcols)        ! w  convective available potential energy.
   349)    real(r8) :: mu_out(pcols,pver)
   350)    real(r8) :: md_out(pcols,pver)
   351) 
   352)    ! used in momentum transport calculation
   353)    real(r8) :: winds(pcols, pver, 2)
   354)    real(r8) :: wind_tends(pcols, pver, 2)
   355)    real(r8) :: pguall(pcols, pver, 2)
   356)    real(r8) :: pgdall(pcols, pver, 2)
   357)    real(r8) :: icwu(pcols,pver, 2)
   358)    real(r8) :: icwd(pcols,pver, 2)
   359)    real(r8) :: seten(pcols, pver)
   360)    logical  :: l_windt(2)
   361)    real(r8) :: tfinal1, tfinal2
   362)    integer  :: ii
   363) 
   364)    logical  :: lq(pcnst)
   365) 
   366)    !----------------------------------------------------------------------
   367) 
   368)    ! initialize
   369)    lchnk = state%lchnk
   370)    ncol  = state%ncol
   371)    nstep = get_nstep()
   372) 
   373)    ftem = 0._r8   
   374)    mu_out(:,:) = 0._r8
   375)    md_out(:,:) = 0._r8
   376)    wind_tends(:ncol,:pver,:) = 0.0_r8
   377) 
   378)    call physics_state_copy(state,state1)             ! copy state to local state1.
   379) 
   380)    lq(:) = .FALSE.
   381)    lq(1) = .TRUE.
   382)    call physics_ptend_init(ptend_loc, state%psetcols, 'zm_convr', ls=.true., lq=lq)! initialize local ptend type
   383) 
   384) !
   385) ! Associate pointers with physics buffer fields
   386) !
   387)    itim_old = pbuf_old_tim_idx()
   388)    call pbuf_get_field(pbuf, cld_idx,         cld,    start=(/1,1,itim_old/), kount=(/pcols,pver,1/) )
   389) 
   390)    call pbuf_get_field(pbuf, icwmrdp_idx,     ql )
   391)    call pbuf_get_field(pbuf, rprddp_idx,      rprd )
   392)    call pbuf_get_field(pbuf, fracis_idx,      fracis, start=(/1,1,1/),    kount=(/pcols, pver, pcnst/) )
   393)    call pbuf_get_field(pbuf, nevapr_dpcu_idx, evapcdp )
   394)    call pbuf_get_field(pbuf, prec_dp_idx,     prec )
   395)    call pbuf_get_field(pbuf, snow_dp_idx,     snow )
   396) 
   397) !
   398) ! Begin with Zhang-McFarlane (1996) convection parameterization
   399) !
   400)    call t_startf ('zm_convr')
   401) 
   402)    call zm_convr(   lchnk   ,ncol    , &
   403)                     state%t       ,state%q(:,:,1)     ,prec    ,jctop   ,jcbot   , &
