    23)       use parrrtm, only : mg, nbndlw, ngptlw
    24)       use rrlw_con, only: fluxfac, heatfac

    33) !-----------------------------------------------------------------------------
    34)       subroutine rtrnmc(nlayers, istart, iend, iout, pz, semiss, ncbands, &
    35)                         cldfmc, taucmc, planklay, planklev, plankbnd, &
    36)                         pwvcm, fracs, taut, &
    37)                         totuflux, totdflux, fnet, htr, &
    38)                         totuclfl, totdclfl, fnetc, htrc, totufluxs, totdfluxs ) 
    39) !-----------------------------------------------------------------------------
    40) !
    41) !  Original version:   E. J. Mlawer, et al. RRTM_V3.0
    42) !  Revision for GCMs:  Michael J. Iacono; October, 2002
    43) !  Revision for F90:  Michael J. Iacono; June, 2006
    44) !
    45) !  This program calculates the upward fluxes, downward fluxes, and
    46) !  heating rates for an arbitrary clear or cloudy atmosphere.  The input
    47) !  to this program is the atmospheric profile, all Planck function
    48) !  information, and the cloud fraction by layer.  A variable diffusivity 
    49) !  angle (SECDIFF) is used for the angle integration.  Bands 2-3 and 5-9 
    50) !  use a value for SECDIFF that varies from 1.50 to 1.80 as a function of 
    51) !  the column water vapor, and other bands use a value of 1.66.  The Gaussian 
    52) !  weight appropriate to this angle (WTDIFF=0.5) is applied here.  Note that 
    53) !  use of the emissivity angle for the flux integration can cause errors of 
    54) !  1 to 4 W/m2 within cloudy layers.  
    55) !  Clouds are treated with the McICA stochastic approach and maximum-random
    56) !  cloud overlap. 
    57) !***************************************************************************
    58) 
    59) ! ------- Declarations -------
    60) 
    61) ! ----- Input -----
    62)       integer, intent(in) :: nlayers                    ! total number of layers
    63)       integer, intent(in) :: istart                     ! beginning band of calculation
    64)       integer, intent(in) :: iend                       ! ending band of calculation
    65)       integer, intent(in) :: iout                       ! output option flag
    66) 
    67) ! Atmosphere
    68)       real(kind=r8), intent(in) :: pz(0:)               ! level (interface) pressures (hPa, mb)
    69)                                                         !    Dimensions: (0:nlayers)
    70)       real(kind=r8), intent(in) :: pwvcm                ! precipitable water vapor (cm)
    71)       real(kind=r8), intent(in) :: semiss(:)            ! lw surface emissivity
    72)                                                         !    Dimensions: (nbndlw)
    73)       real(kind=r8), intent(in) :: planklay(:,:)        ! 
    74)                                                         !    Dimensions: (nlayers,nbndlw)
    75)       real(kind=r8), intent(in) :: planklev(0:,:)       ! 
    76)                                                         !    Dimensions: (0:nlayers,nbndlw)
    77)       real(kind=r8), intent(in) :: plankbnd(:)          ! 
    78)                                                         !    Dimensions: (nbndlw)
    79)       real(kind=r8), intent(in) :: fracs(:,:)           ! 
    80)                                                         !    Dimensions: (nlayers,ngptw)
    81)       real(kind=r8), intent(in) :: taut(:,:)            ! gaseous + aerosol optical depths
    82)                                                         !    Dimensions: (nlayers,ngptlw)
    83) 
    84) ! Clouds
    85)       integer, intent(in) :: ncbands                    ! number of cloud spectral bands
    86)       real(kind=r8), intent(in) :: cldfmc(:,:)          ! layer cloud fraction [mcica]
    87)                                                         !    Dimensions: (ngptlw,nlayers)
    88)       real(kind=r8), intent(in) :: taucmc(:,:)          ! layer cloud optical depth [mcica]
    89)                                                         !    Dimensions: (ngptlw,nlayers)
    90) 
    91) ! ----- Output -----
    92)       real(kind=r8), intent(out) :: totuflux(0:)        ! upward longwave flux (w/m2)
    93)                                                         !    Dimensions: (0:nlayers)
    94)       real(kind=r8), intent(out) :: totdflux(0:)        ! downward longwave flux (w/m2)
    95)                                                         !    Dimensions: (0:nlayers)
    96)       real(kind=r8), intent(out) :: fnet(0:)            ! net longwave flux (w/m2)
    97)                                                         !    Dimensions: (0:nlayers)
    98)       real(kind=r8), intent(out) :: htr(0:)             ! longwave heating rate (k/day)
    99)                                                         !    Dimensions: (0:nlayers)
   100)       real(kind=r8), intent(out) :: totuclfl(0:)        ! clear sky upward longwave flux (w/m2)
   101)                                                         !    Dimensions: (0:nlayers)
   102)       real(kind=r8), intent(out) :: totdclfl(0:)        ! clear sky downward longwave flux (w/m2)
   103)                                                         !    Dimensions: (0:nlayers)
   104)       real(kind=r8), intent(out) :: fnetc(0:)           ! clear sky net longwave flux (w/m2)
   105)                                                         !    Dimensions: (0:nlayers)
   106)       real(kind=r8), intent(out) :: htrc(0:)            ! clear sky longwave heating rate (k/day)
   107)                                                         !    Dimensions: (0:nlayers)
   108)       real(kind=r8), intent(out) :: totufluxs(:,0:)     ! upward longwave flux spectral (w/m2)
   109)                                                         !    Dimensions: (nbndlw, 0:nlayers)
   110)       real(kind=r8), intent(out) :: totdfluxs(:,0:)     ! downward longwave flux spectral (w/m2)
   111)                                                         !    Dimensions: (nbndlw, 0:nlayers)
   112) 
   113) ! ----- Local -----
   114) ! Declarations for radiative transfer
   115)       real(kind=r8) :: abscld(nlayers,ngptlw)
   116)       real(kind=r8) :: atot(nlayers)
   117)       real(kind=r8) :: atrans(nlayers)
   118)       real(kind=r8) :: bbugas(nlayers)
   119)       real(kind=r8) :: bbutot(nlayers)
   120)       real(kind=r8) :: clrurad(0:nlayers)
   121)       real(kind=r8) :: clrdrad(0:nlayers)
   122)       real(kind=r8) :: efclfrac(nlayers,ngptlw)
   123)       real(kind=r8) :: uflux(0:nlayers)
   124)       real(kind=r8) :: dflux(0:nlayers)
   125)       real(kind=r8) :: urad(0:nlayers)
   126)       real(kind=r8) :: drad(0:nlayers)
   127)       real(kind=r8) :: uclfl(0:nlayers)
   128)       real(kind=r8) :: dclfl(0:nlayers)
   129)       real(kind=r8) :: odcld(nlayers,ngptlw)
   130) 
   131) 
   132)       real(kind=r8) :: secdiff(nbndlw)                   ! secant of diffusivity angle
   133)       real(kind=r8) :: a0(nbndlw),a1(nbndlw),a2(nbndlw)  ! diffusivity angle adjustment coefficients
   134)       real(kind=r8) :: wtdiff, rec_6
   135)       real(kind=r8) :: transcld, radld, radclrd, plfrac, blay, dplankup, dplankdn
   136)       real(kind=r8) :: odepth, odtot, odepth_rec, odtot_rec, gassrc
   137)       real(kind=r8) :: tblind, tfactot, bbd, bbdtot, tfacgas, transc, tausfac
   138)       real(kind=r8) :: rad0, reflect, radlu, radclru
   139) 
   140)       integer :: icldlyr(nlayers)                        ! flag for cloud in layer
   141)       integer :: ibnd, ib, iband, lay, lev, l, ig        ! loop indices
   142)       integer :: igc                                     ! g-point interval counter
   143)       integer :: iclddn                                  ! flag for cloud in down path
   144)       integer :: ittot, itgas, itr                       ! lookup table indices
   145) 
   146) ! ------- Definitions -------
   147) ! input
   148) !    nlayers                      ! number of model layers
   149) !    ngptlw                       ! total number of g-point subintervals
   150) !    nbndlw                       ! number of longwave spectral bands
   151) !    ncbands                      ! number of spectral bands for clouds
   152) !    secdiff                      ! diffusivity angle
   153) !    wtdiff                       ! weight for radiance to flux conversion
   154) !    pavel                        ! layer pressures (mb)
   155) !    pz                           ! level (interface) pressures (mb)
   156) !    tavel                        ! layer temperatures (k)
   157) !    tz                           ! level (interface) temperatures(mb)
   158) !    tbound                       ! surface temperature (k)
   159) !    cldfrac                      ! layer cloud fraction
   160) !    taucloud                     ! layer cloud optical depth
   161) !    itr                          ! integer look-up table index
   162) !    icldlyr                      ! flag for cloudy layers
   163) !    iclddn                       ! flag for cloud in column at any layer
   164) !    semiss                       ! surface emissivities for each band
   165) !    reflect                      ! surface reflectance
   166) !    bpade                        ! 1/(pade constant)
   167) !    tau_tbl                      ! clear sky optical depth look-up table
   168) !    exp_tbl                      ! exponential look-up table for transmittance
   169) !    tfn_tbl                      ! tau transition function look-up table
   170) 
   171) ! local
   172) !    atrans                       ! gaseous absorptivity
   173) !    abscld                       ! cloud absorptivity
   174) !    atot                         ! combined gaseous and cloud absorptivity
   175) !    odclr                        ! clear sky (gaseous) optical depth
   176) !    odcld                        ! cloud optical depth
   177) !    odtot                        ! optical depth of gas and cloud
   178) !    tfacgas                      ! gas-only pade factor, used for planck fn
   179) !    tfactot                      ! gas and cloud pade factor, used for planck fn
   180) !    bbdgas                       ! gas-only planck function for downward rt
   181) !    bbugas                       ! gas-only planck function for upward rt
   182) !    bbdtot                       ! gas and cloud planck function for downward rt
   183) !    bbutot                       ! gas and cloud planck function for upward calc.
   184) !    gassrc                       ! source radiance due to gas only
   185) !    efclfrac                     ! effective cloud fraction
   186) !    radlu                        ! spectrally summed upward radiance 
   187) !    radclru                      ! spectrally summed clear sky upward radiance 
   188) !    urad                         ! upward radiance by layer
   189) !    clrurad                      ! clear sky upward radiance by layer
   190) !    radld                        ! spectrally summed downward radiance 
   191) !    radclrd                      ! spectrally summed clear sky downward radiance 
   192) !    drad                         ! downward radiance by layer
   193) !    clrdrad                      ! clear sky downward radiance by layer
   194) 
   195) ! output
   196) !    totuflux                     ! upward longwave flux (w/m2)
   197) !    totdflux                     ! downward longwave flux (w/m2)
   198) !    fnet                         ! net longwave flux (w/m2)
   199) !    htr                          ! longwave heating rate (k/day)
   200) !    totuclfl                     ! clear sky upward longwave flux (w/m2)
   201) !    totdclfl                     ! clear sky downward longwave flux (w/m2)
   202) !    fnetc                        ! clear sky net longwave flux (w/m2)
   203) !    htrc                         ! clear sky longwave heating rate (k/day)
   204) 
   205) 
   206) ! This secant and weight corresponds to the standard diffusivity 
   207) ! angle.  This initial value is redefined below for some bands.
   208)       data wtdiff /0.5_r8/
   209)       data rec_6 /0.166667_r8/
   210) 
   211) ! Reset diffusivity angle for Bands 2-3 and 5-9 to vary (between 1.50
   212) ! and 1.80) as a function of total column water vapor.  The function
   213) ! has been defined to minimize flux and cooling rate errors in these bands
   214) ! over a wide range of precipitable water values.
   215)       data a0 / 1.66_r8,  1.55_r8,  1.58_r8,  1.66_r8, &
   216)                 1.54_r8, 1.454_r8,  1.89_r8,  1.33_r8, &
   217)                1.668_r8,  1.66_r8,  1.66_r8,  1.66_r8, &
   218)                 1.66_r8,  1.66_r8,  1.66_r8,  1.66_r8 /
   219)       data a1 / 0.00_r8,  0.25_r8,  0.22_r8,  0.00_r8, &
   220)                 0.13_r8, 0.446_r8, -0.10_r8,  0.40_r8, &
   221)               -0.006_r8,  0.00_r8,  0.00_r8,  0.00_r8, &
   222)                 0.00_r8,  0.00_r8,  0.00_r8,  0.00_r8 /
   223)       data a2 / 0.00_r8, -12.0_r8, -11.7_r8,  0.00_r8, &
   224)                -0.72_r8,-0.243_r8,  0.19_r8,-0.062_r8, &
   225)                0.414_r8,  0.00_r8,  0.00_r8,  0.00_r8, &
   226)                 0.00_r8,  0.00_r8,  0.00_r8,  0.00_r8 /
   227) 
   228)       hvrrtc = '$Revision: 1.3 $'
   229) 
   230)       do ibnd = 1,nbndlw
   231)          if (ibnd.eq.1 .or. ibnd.eq.4 .or. ibnd.ge.10) then
   232)            secdiff(ibnd) = 1.66_r8
   233)          else
   234)            secdiff(ibnd) = a0(ibnd) + a1(ibnd)*exp(a2(ibnd)*pwvcm)
   235)          endif
   236)       enddo
   237)       if (pwvcm.lt.1.0) secdiff(6) = 1.80_r8
   238)       if (pwvcm.gt.7.1) secdiff(7) = 1.50_r8
   239) 
   240)       urad(0) = 0.0_r8
   241)       drad(0) = 0.0_r8
   242)       totuflux(0) = 0.0_r8
   243)       totdflux(0) = 0.0_r8
   244)       clrurad(0) = 0.0_r8
   245)       clrdrad(0) = 0.0_r8
   246)       totuclfl(0) = 0.0_r8
   247)       totdclfl(0) = 0.0_r8
   248) 
   249)       do lay = 1, nlayers
   250)          urad(lay) = 0.0_r8
   251)          drad(lay) = 0.0_r8
   252)          totuflux(lay) = 0.0_r8
   253)          totdflux(lay) = 0.0_r8
   254)          clrurad(lay) = 0.0_r8
   255)          clrdrad(lay) = 0.0_r8
   256)          totuclfl(lay) = 0.0_r8
   257)          totdclfl(lay) = 0.0_r8
   258)          icldlyr(lay) = 0
   259) 
   260) ! Change to band loop?
   261)          do ig = 1, ngptlw
   262)             if (cldfmc(ig,lay) .eq. 1._r8) then
   263)                ib = ngb(ig)
   264)                odcld(lay,ig) = secdiff(ib) * taucmc(ig,lay)
   265)                transcld = exp(-odcld(lay,ig))
   266)                abscld(lay,ig) = 1._r8 - transcld
   267)                efclfrac(lay,ig) = abscld(lay,ig) * cldfmc(ig,lay)
   268)                icldlyr(lay) = 1
   269)             else
   270)                odcld(lay,ig) = 0.0_r8
   271)                abscld(lay,ig) = 0.0_r8
   272)                efclfrac(lay,ig) = 0.0_r8
   273)             endif
   274)          enddo
   275) 
   276)       enddo
   277) 
   278)       igc = 1
   279) ! Loop over frequency bands.
   280)       do iband = istart, iend
   281) 
   282) ! Reinitialize g-point counter for each band if output for each band is requested.
   283)          if (iout.gt.0.and.iband.ge.2) igc = ngs(iband-1)+1
   284) 
   285) ! Loop over g-channels.
   286)  1000    continue
   287) 
   288) ! Radiative transfer starts here.
   289)          radld = 0._r8
   290)          radclrd = 0._r8
   291)          iclddn = 0
   292) 
   293) ! Downward radiative transfer loop.  
   294) 
   295)          do lev = nlayers, 1, -1
   296)                plfrac = fracs(lev,igc)
   297)                blay = planklay(lev,iband)
   298)                dplankup = planklev(lev,iband) - blay
   299)                dplankdn = planklev(lev-1,iband) - blay
   300)                odepth = secdiff(iband) * taut(lev,igc)
   301)                if (odepth .lt. 0.0_r8) odepth = 0.0_r8
   302) !  Cloudy layer
   303)                if (icldlyr(lev).eq.1) then
   304)                   iclddn = 1
   305)                   odtot = odepth + odcld(lev,igc)
   306)                   if (odtot .lt. 0.06_r8) then
   307)                      atrans(lev) = odepth - 0.5_r8*odepth*odepth
   308)                      odepth_rec = rec_6*odepth
   309)                      gassrc = plfrac*(blay+dplankdn*odepth_rec)*atrans(lev)
   310) 
   311)                      atot(lev) =  odtot - 0.5_r8*odtot*odtot
   312)                      odtot_rec = rec_6*odtot
   313)                      bbdtot =  plfrac * (blay+dplankdn*odtot_rec)
   314)                      bbd = plfrac*(blay+dplankdn*odepth_rec)
   315)                      radld = radld - radld * (atrans(lev) + &
   316)                          efclfrac(lev,igc) * (1. - atrans(lev))) + &
   317)                          gassrc + cldfmc(igc,lev) * &
   318)                          (bbdtot * atot(lev) - gassrc)
   319)                      drad(lev-1) = drad(lev-1) + radld
   320)                   
   321)                      bbugas(lev) =  plfrac * (blay+dplankup*odepth_rec)
   322)                      bbutot(lev) =  plfrac * (blay+dplankup*odtot_rec)
   323) 
   324)                   elseif (odepth .le. 0.06_r8) then
   325)                      atrans(lev) = odepth - 0.5_r8*odepth*odepth
   326)                      odepth_rec = rec_6*odepth
   327)                      gassrc = plfrac*(blay+dplankdn*odepth_rec)*atrans(lev)
   328) 
   329)                      odtot = odepth + odcld(lev,igc)
   330)                      tblind = odtot/(bpade+odtot)
   331)                      ittot = tblint*tblind + 0.5_r8
   332)                      tfactot = tfn_tbl(ittot)
   333)                      bbdtot = plfrac * (blay + tfactot*dplankdn)
   334)                      bbd = plfrac*(blay+dplankdn*odepth_rec)
   335)                      atot(lev) = 1. - exp_tbl(ittot)
   336) 
   337)                      radld = radld - radld * (atrans(lev) + &
   338)                          efclfrac(lev,igc) * (1._r8 - atrans(lev))) + &
   339)                          gassrc + cldfmc(igc,lev) * &
   340)                          (bbdtot * atot(lev) - gassrc)
   341)                      drad(lev-1) = drad(lev-1) + radld
   342) 
   343)                      bbugas(lev) = plfrac * (blay + dplankup*odepth_rec)
   344)                      bbutot(lev) = plfrac * (blay + tfactot * dplankup)
   345) 
   346)                   else
   347) 
   348)                      tblind = odepth/(bpade+odepth)
   349)                      itgas = tblint*tblind+0.5_r8
   350)                      odepth = tau_tbl(itgas)
   351)                      atrans(lev) = 1._r8 - exp_tbl(itgas)
   352)                      tfacgas = tfn_tbl(itgas)
   353)                      gassrc = atrans(lev) * plfrac * (blay + tfacgas*dplankdn)
   354) 
   355)                      odtot = odepth + odcld(lev,igc)
   356)                      tblind = odtot/(bpade+odtot)
   357)                      ittot = tblint*tblind + 0.5_r8
   358)                      tfactot = tfn_tbl(ittot)
   359)                      bbdtot = plfrac * (blay + tfactot*dplankdn)
   360)                      bbd = plfrac*(blay+tfacgas*dplankdn)
   361)                      atot(lev) = 1._r8 - exp_tbl(ittot)
   362) 
   363)                   radld = radld - radld * (atrans(lev) + &
   364)                     efclfrac(lev,igc) * (1._r8 - atrans(lev))) + &
   365)                     gassrc + cldfmc(igc,lev) * &
   366)                     (bbdtot * atot(lev) - gassrc)
   367)                   drad(lev-1) = drad(lev-1) + radld
   368)                   bbugas(lev) = plfrac * (blay + tfacgas * dplankup)
   369)                   bbutot(lev) = plfrac * (blay + tfactot * dplankup)
   370)                   endif
   371) !  Clear layer
   372)                else
   373)                   if (odepth .le. 0.06_r8) then
   374)                      atrans(lev) = odepth-0.5_r8*odepth*odepth
   375)                      odepth = rec_6*odepth
   376)                      bbd = plfrac*(blay+dplankdn*odepth)
   377)                      bbugas(lev) = plfrac*(blay+dplankup*odepth)
   378)                   else
   379)                      tblind = odepth/(bpade+odepth)
   380)                      itr = tblint*tblind+0.5_r8
   381)                      transc = exp_tbl(itr)
   382)                      atrans(lev) = 1._r8-transc
   383)                      tausfac = tfn_tbl(itr)
   384)                      bbd = plfrac*(blay+tausfac*dplankdn)
   385)                      bbugas(lev) = plfrac * (blay + tausfac * dplankup)
   386)                   endif   
   387)                   radld = radld + (bbd-radld)*atrans(lev)
   388)                   drad(lev-1) = drad(lev-1) + radld
   389)                endif
   390) !  Set clear sky stream to total sky stream as long as layers
   391) !  remain clear.  Streams diverge when a cloud is reached (iclddn=1),
   392) !  and clear sky stream must be computed separately from that point.
   393)                   if (iclddn.eq.1) then
   394)                      radclrd = radclrd + (bbd-radclrd) * atrans(lev) 
   395)                      clrdrad(lev-1) = clrdrad(lev-1) + radclrd
   396)                   else
   397)                      radclrd = radld
   398)                      clrdrad(lev-1) = drad(lev-1)
   399)                   endif
   400)             enddo
   401) 
   402) ! Spectral emissivity & reflectance
   403) !  Include the contribution of spectrally varying longwave emissivity
   404) !  and reflection from the surface to the upward radiative transfer.
   405) !  Note: Spectral and Lambertian reflection are identical for the
   406) !  diffusivity angle flux integration used here.
   407) 
   408)          rad0 = fracs(1,igc) * plankbnd(iband)
   409) !  Add in specular reflection of surface downward radiance.
   410)          reflect = 1._r8 - semiss(iband)
   411)          radlu = rad0 + reflect * radld
   412)          radclru = rad0 + reflect * radclrd
   413) 
   414) 
   415) ! Upward radiative transfer loop.
   416)          urad(0) = urad(0) + radlu
   417)          clrurad(0) = clrurad(0) + radclru
   418) 
   419)          do lev = 1, nlayers
   420) !  Cloudy layer
