    67) ! !ROUTINE:  stepon_init --- Time stepping initialization
    68) !
    69) ! !INTERFACE:
    70) subroutine stepon_init( gw, etamid, dyn_in, dyn_out )
    71) ! !USES:
    72)   use dimensions_mod, only: nlev, nelemd, npsq
    73)   use hycoef,         only: hyam, hybm
    74)   use cam_history,    only: phys_decomp, addfld, add_default, dyn_decomp
    75)   use control_mod,    only: smooth_phis_numcycle
    76)   use gravity_waves_sources, only: gws_init
    77)   use phys_control,   only: use_gw_front
    78) 
    79) ! !OUTPUT PARAMETERS
    80) !
    81)   real(r8), intent(out) :: gw(plat)           ! Gaussian weights
    82)   real(r8), intent(out) :: etamid(plev)       ! vertical coords at midpoints
    83)   type (dyn_import_t), intent(inout) :: dyn_in  ! Dynamics import container
    84)   type (dyn_export_t), intent(inout) :: dyn_out ! Dynamics export container
    85) 
    86)   integer :: m
    87) ! !DESCRIPTION:
    88) !
    89) ! Allocate data, initialize values, setup grid locations and other
    90) ! work needed to prepare the dynamics to run. Return weights and 
    91) ! vertical coords to atmosphere calling program.
    92) !
    93) !EOP
    94) !-----------------------------------------------------------------------
    95) !BOC
    96) 
    97)   ! This is not done in dyn_init due to a circular dependency issue.
    98)   if(iam < par%nprocs) then
    99)      call initEdgeBuffer(edgebuf, (3+pcnst)*nlev)
   100)      if (use_gw_front) call gws_init
   101)   end if
   102) 
   103)   etamid(:) = hyam(:) + hybm(:)
   104) 
   105) 
   106)   ! fields that are written by the dycore
   107)   ! these calls cant be done in dyn_init() because physics grid
   108)   ! is not initialized at that point if making a restart runs
   109)   !
   110)   ! Forcing from physics
   111)   ! FU, FV, other dycores, doc, says "m/s" but I think that is m/s^2
   112)   call addfld ('FU      ','m/s2    ',nlev, 'A','Zonal wind forcing term',dyn_decomp, &
   113)        begdim1=1,enddim1=npsq,begdim3=1,enddim3=nelemd)
   114)   call addfld ('FV      ','m/s2    ',nlev, 'A','Meridional wind forcing term',dyn_decomp, &
   115)        begdim1=1,enddim1=npsq,begdim3=1,enddim3=nelemd)
   116)   call addfld ('VOR      ','1/s     ',nlev, 'A','Vorticity',dyn_decomp, &
   117)        begdim1=1,enddim1=npsq,begdim3=1,enddim3=nelemd)
   118)   call addfld ('DIV      ','1/s     ',nlev, 'A','Divergence',dyn_decomp, &
   119)        begdim1=1,enddim1=npsq,begdim3=1,enddim3=nelemd)
   120) 
   121)   if (smooth_phis_numcycle>0) then
   122)      call addfld ('PHIS_SM ','m2/s2      ',1,    'I','Surface geopotential (smoothed)',dyn_decomp,&
   123)        begdim1=1,enddim1=npsq,begdim3=1,enddim3=nelemd)
   124)      call addfld ('SGH_SM  ','m       ',1,    'I','Standard deviation of orography (smoothed)',dyn_decomp,&
   125)        begdim1=1,enddim1=npsq,begdim3=1,enddim3=nelemd)
   126)      call addfld ('SGH30_SM','m       ',1,    'I','Standard deviation of 30s orography (smoothed)',dyn_decomp,&
   127)        begdim1=1,enddim1=npsq,begdim3=1,enddim3=nelemd)
   128)   endif
   129) 
   130)   call addfld ('CONVU   ','m/s2    ',nlev, 'A','Zonal component IE->KE conversion term',phys_decomp)
   131)   call addfld ('CONVV   ','m/s2    ',nlev, 'A','Meridional component IE->KE conversion term',phys_decomp)
   132)   call addfld ('DIFFU   ','m/s2    ',nlev, 'A','U horizontal diffusion',phys_decomp)
   133)   call addfld ('DIFFV   ','m/s2    ',nlev, 'A','V horizontal diffusion',phys_decomp)
   134)   
   135)   call addfld ('ETADOT  ','1/s ',plevp,'A','Vertical (eta) velocity',phys_decomp)
   136)   call addfld ('U&IC    ','m/s ',plev, 'I','Zonal wind'        ,phys_decomp )
   137)   call addfld ('V&IC    ','m/s ',plev, 'I','Meridional wind'    ,phys_decomp )
   138)   call add_default ('U&IC       ',0, 'I')
   139)   call add_default ('V&IC       ',0, 'I')
   140) 
   141)   call addfld ('PS&IC      ','Pa      ',1,    'I','Surface pressure'    ,phys_decomp)
   142)   call addfld ('T&IC       ','K       ',plev, 'I','Temperature'         ,phys_decomp)
   143) 
   144)   call add_default ('PS&IC      ',0, 'I')
   145)   call add_default ('T&IC       ',0, 'I')
   146)   do m = 1,pcnst
   147)      call addfld (trim(cnst_name(m))//'&IC','kg/kg   ',plev, 'I',cnst_longname(m), phys_decomp)
   148)   end do
   149)   do m = 1,pcnst
   150)      call add_default(trim(cnst_name(m))//'&IC',0, 'I')
   151)   end do
   152) 
   153) 
   154) end subroutine stepon_init

   159) ! !ROUTINE:  stepon_run1 -- Phase 1 of dynamics run method.

   161) ! !INTERFACE:
   162) subroutine stepon_run1( dtime_out, phys_state, phys_tend,               &
   163)                         pbuf2d, dyn_in, dyn_out )
   164)   
   165)   use dp_coupling, only: d_p_coupling
   166)   use time_mod,    only: tstep, phys_tscale       ! dynamics timestep
   167)   use time_manager, only: dtime
   168)   use control_mod, only: ftype
   169)   use physics_buffer, only : physics_buffer_desc
   170)   implicit none
   171) !
   172) ! !OUTPUT PARAMETERS:
   173) !
   174) 
   175)    real(r8), intent(out) :: dtime_out   ! Time-step
   176)    type(physics_state), intent(inout) :: phys_state(begchunk:endchunk)
   177)    type(physics_tend), intent(inout) :: phys_tend(begchunk:endchunk)
   178)    type (dyn_import_t), intent(inout) :: dyn_in  ! Dynamics import container
   179)    type (dyn_export_t), intent(inout) :: dyn_out ! Dynamics export container
   180)    type (physics_buffer_desc), pointer :: pbuf2d(:,:)
   181) 
   182) !-----------------------------------------------------------------------
   183) 
   184)    ! NOTE: dtime_out computed here must match formula below
   185)    dtime_out=dtime
   186)    if (phys_tscale/=0) then
   187)       dtime_out=phys_tscale  ! set by user in namelist
   188)    endif
   189) 
   190)    if(iam < par%nprocs) then
   191)       if(tstep <= 0)  call endrun( 'bad tstep')
   192)       if(dtime <= 0)  call endrun( 'bad dtime')
   193)    end if
   194)    !----------------------------------------------------------
   195)    ! Move data into phys_state structure.
   196)    !----------------------------------------------------------
   197)    
   198)    call t_barrierf('sync_d_p_coupling', mpicom)
   199)    call t_startf('d_p_coupling')
   200)    call d_p_coupling (phys_state, phys_tend,  pbuf2d, dyn_out )
   201)    call t_stopf('d_p_coupling')
   202)    
   203) end subroutine stepon_run1
   204) 
   205) subroutine stepon_run2(phys_state, phys_tend, dyn_in, dyn_out )
   206)    use bndry_mod,      only: bndry_exchangeV
   207)    use dimensions_mod, only: nlev, nelemd, np, npsq
   208)    use dp_coupling,    only: p_d_coupling
   209)    use parallel_mod,   only: par
   210)    use dyn_comp,       only: TimeLevel
   211)    
   212)    use time_manager,    only: dtime  ! namelist timestep
   213)    use time_mod,        only: tstep, phys_tscale, TimeLevel_Qdp   !  dynamics typestep
   214)    use control_mod,     only: ftype, qsplit, smooth_phis_numcycle
   215)    use hycoef,          only: hyam, hybm, hyai, hybi, ps0
   216)    use cam_history,     only: outfld, hist_fld_active
   217)    use nctopo_util_mod, only: phisdyn,sghdyn,sgh30dyn
   218) 
   219) 
   220)    type(physics_state), intent(inout) :: phys_state(begchunk:endchunk)
   221)    type(physics_tend), intent(inout) :: phys_tend(begchunk:endchunk)
   222)    type (dyn_import_t), intent(inout) :: dyn_in  ! Dynamics import container
   223)    type (dyn_export_t), intent(inout) :: dyn_out ! Dynamics export container
   224)    integer :: kptr, ie, ic, i, j, k, tl_f, tl_fQdp
   225)    real(r8) :: rec2dt, dyn_ps0
   226)    real(r8) :: dp(np,np,nlev),dp_tmp,fq,fq0,qn0, ftmp(npsq,nlev,2)
   227) 
   228)    ! copy from phys structures -> dynamics structures
   229)    call t_barrierf('sync_p_d_coupling', mpicom)
   230)    call t_startf('p_d_coupling')
   231)    call p_d_coupling(phys_state, phys_tend,  dyn_in)
   232)    call t_stopf('p_d_coupling')
   233) 
   234)    if(iam >= par%nprocs) return
   235) 
   236)    call t_startf('bndry_exchange')
   237)    ! do boundary exchange
   238)    do ie=1,nelemd
   239)       kptr=0
   240)       call edgeVpack(edgebuf,dyn_in%elem(ie)%derived%FM(:,:,:,:,1),2*nlev,kptr,dyn_in%elem(ie)%desc)
   241)       kptr=kptr+2*nlev
   242) 
   243)       call edgeVpack(edgebuf,dyn_in%elem(ie)%derived%FT(:,:,:,1),nlev,kptr,dyn_in%elem(ie)%desc)
   244)       kptr=kptr+nlev
   245)       call edgeVpack(edgebuf,dyn_in%elem(ie)%derived%FQ(:,:,:,:,1),nlev*pcnst,kptr,dyn_in%elem(ie)%desc)
   246)    end do
   247) 
   248)    call bndry_exchangeV(par, edgebuf)
   249) 
   250)    ! NOTE: rec2dt MUST be 1/dtime_out as computed above
   251) !  rec2dt = 1./real(dtime,r8)
   252) !  if (phys_tscale/=0) then
   253) !     rec2dt = 1_r8/real(phys_tscale,r8)
   254) !  endif
   255) 
   256)    rec2dt = 1._r8/dtime
   257)    if (phys_tscale/=0) then
   258)       rec2dt = 1._r8/phys_tscale
   259)    endif
   260) 
   261) 
   262)    do ie=1,nelemd
   263)       kptr=0
   264) 
   265)       call edgeVunpack(edgebuf,dyn_in%elem(ie)%derived%FM(:,:,:,:,1),2*nlev,kptr,dyn_in%elem(ie)%desc)
   266)       kptr=kptr+2*nlev
   267) 
   268)       call edgeVunpack(edgebuf,dyn_in%elem(ie)%derived%FT(:,:,:,1),nlev,kptr,dyn_in%elem(ie)%desc)
   269)       kptr=kptr+nlev
   270) 
   271)       call edgeVunpack(edgebuf,dyn_in%elem(ie)%derived%FQ(:,:,:,:,1),nlev*pcnst,kptr,dyn_in%elem(ie)%desc)
   272) 
   273)       tl_f = TimeLevel%n0   ! timelevel which was adjusted by physics
   274) 
   275)       call TimeLevel_Qdp(TimeLevel, qsplit, tl_fQdp)
   276) 
   277)       dyn_ps0=ps0
   278) 
   279)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   280)       ! ftype=2:  apply forcing to Q,ps.  Return dynamics tendencies
   281)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   282)       if (ftype==2) then
   283)          ! apply forcing to states tl_f 
   284)          ! requires forward-in-time timestepping, checked in namelist_mod.F90
   285) !$omp parallel do private(k)
   286)          do k=1,nlev
   287)             dp(:,:,k) = ( hyai(k+1) - hyai(k) )*dyn_ps0 + &
   288)                  ( hybi(k+1) - hybi(k) )*dyn_in%elem(ie)%state%ps_v(:,:,tl_f)
   289)          enddo
   290)          do k=1,nlev
   291)             do j=1,np
   292)                do i=1,np
   293) 
   294)                   do ic=1,pcnst
   295)                      ! back out tendency: Qdp*dtime 
   296)                      fq = dp(i,j,k)*(  dyn_in%elem(ie)%derived%FQ(i,j,k,ic,1) - &
   297)                           dyn_in%elem(ie)%state%Q(i,j,k,ic))
   298)                      
   299)                      ! apply forcing to Qdp
   300) !                     dyn_in%elem(ie)%state%Qdp(i,j,k,ic,tl_fQdp) = &
   301) !                          dyn_in%elem(ie)%state%Qdp(i,j,k,ic,tl_fQdp) + fq 
   302)                      dyn_in%elem(ie)%state%Qdp(i,j,k,ic,tl_fQdp) = &
   303)                           dp(i,j,k)*dyn_in%elem(ie)%derived%FQ(i,j,k,ic,1) 
   304) 
   305) ! BEWARE critical region if using OpenMP over k (AAM)
   306)                      if (ic==1) then
   307)                         ! force ps_v to conserve mass:  
   308)                         dyn_in%elem(ie)%state%ps_v(i,j,tl_f)= &
   309)                              dyn_in%elem(ie)%state%ps_v(i,j,tl_f) + fq
   310)                      endif
   311)                   enddo
   312)                end do
   313)             end do
   314)          end do
   315) 
   316) !$omp parallel do private(k, j, i, ic, dp_tmp)
   317)          do k=1,nlev
   318)           do ic=1,pcnst
   319)             do j=1,np
   320)                do i=1,np
   321)                   ! make Q consistent now that we have updated ps_v above
   322)                   ! recompute dp, since ps_v was changed above
   323)                   dp_tmp = ( hyai(k+1) - hyai(k) )*dyn_ps0 + &
   324)                        ( hybi(k+1) - hybi(k) )*dyn_in%elem(ie)%state%ps_v(i,j,tl_f)
   325)                   dyn_in%elem(ie)%state%Q(i,j,k,ic)= &
   326)                        dyn_in%elem(ie)%state%Qdp(i,j,k,ic,tl_fQdp)/dp_tmp
   327) 
   328)                end do
   329)             end do
   330)           end do
   331)          end do
   332)       endif
   333) 
   334)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   335)       ! ftype=1:  apply all forcings as an adjustment
   336)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   337)       if (ftype==1) then
   338)          ! apply forcing to state tl_f
   339)          ! requires forward-in-time timestepping, checked in namelist_mod.F90
   340) !$omp parallel do private(k)
   341)          do k=1,nlev
   342)             dp(:,:,k) = ( hyai(k+1) - hyai(k) )*dyn_ps0 + &
   343)                  ( hybi(k+1) - hybi(k) )*dyn_in%elem(ie)%state%ps_v(:,:,tl_f)
   344)          enddo
   345)          do k=1,nlev
   346)             do j=1,np
   347)                do i=1,np
   348) 
   349)                   do ic=1,pcnst
   350)                      ! back out tendency: Qdp*dtime 
   351)                      fq = dp(i,j,k)*(  dyn_in%elem(ie)%derived%FQ(i,j,k,ic,1) - &
   352)                           dyn_in%elem(ie)%state%Q(i,j,k,ic))
   353)                      
   354)                      ! apply forcing to Qdp
   355)                      dyn_in%elem(ie)%state%Qdp(i,j,k,ic,tl_fQdp) = &
   356)                           dyn_in%elem(ie)%state%Qdp(i,j,k,ic,tl_fQdp) + fq 
   357) 
   358)                      ! BEWARE critical region if using OpenMP over k (AAM)
   359)                      if (ic==1) then
   360)                         ! force ps_v to conserve mass:  
   361)                         dyn_in%elem(ie)%state%ps_v(i,j,tl_f)= &
   362)                              dyn_in%elem(ie)%state%ps_v(i,j,tl_f) + fq
   363)                      endif
   364)                   enddo
   365) 
   366)                   ! force V, T, both timelevels
   367)                   dyn_in%elem(ie)%state%v(i,j,:,k,tl_f)= &
   368)                        dyn_in%elem(ie)%state%v(i,j,:,k,tl_f) +  &
   369)                        dtime*dyn_in%elem(ie)%derived%FM(i,j,:,k,1)
   370)                   
   371)                   dyn_in%elem(ie)%state%T(i,j,k,tl_f)= &
   372)                        dyn_in%elem(ie)%state%T(i,j,k,tl_f) + &
   373)                        dtime*dyn_in%elem(ie)%derived%FT(i,j,k,1)
   374)                   
   375)                end do
   376)             end do
   377)          end do
   378) 
   379) !$omp parallel do private(k, j, i, ic, dp_tmp)
   380)          do k=1,nlev
   381)             do ic=1,pcnst
   382)                do j=1,np
   383)                   do i=1,np
   384)                      ! make Q consistent now that we have updated ps_v above
   385)                      dp_tmp = ( hyai(k+1) - hyai(k) )*dyn_ps0 + &
   386)                           ( hybi(k+1) - hybi(k) )*dyn_in%elem(ie)%state%ps_v(i,j,tl_f)
   387)                      dyn_in%elem(ie)%state%Q(i,j,k,ic)= &
   388)                           dyn_in%elem(ie)%state%Qdp(i,j,k,ic,tl_fQdp)/dp_tmp
   389)                   end do
   390)                end do
   391)             end do
   392)          end do
   393)       endif
   394) 
   395)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   396)       ! ftype=0 and ftype<0 (debugging options):  just return tendencies to dynamics
   397)       !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
   398)       if (ftype<=0) then
   399) 
   400)          do ic=1,pcnst
   401)             ! Q  =  data used for forcing, at timelevel nm1   t
   402)             ! FQ =  adjusted Q returned by forcing,  at time  t+dt
   403)             ! tendency = (FQ*dp - Q*dp) / dt 
   404)             ! Convert this to a tendency on Qdp:  CAM physics does not change ps
   405)             ! so use ps_v at t.  (or, if physics worked with Qdp
   406)             ! and returned FQdp, tendency = (FQdp-Qdp)/2dt and since physics
   407)             ! did not change dp, dp would be the same in both terms)
   408) !$omp parallel do private(k, j, i, dp_tmp)
   409)             do k=1,nlev
   410)                do j=1,np
   411)                   do i=1,np
   412)                      dp_tmp = ( hyai(k+1) - hyai(k) )*dyn_ps0 + &
   413)                           ( hybi(k+1) - hybi(k) )*dyn_in%elem(ie)%state%ps_v(i,j,tl_f)
   414)                      
   415)                      dyn_in%elem(ie)%derived%FQ(i,j,k,ic,1)=&
   416)                           (  dyn_in%elem(ie)%derived%FQ(i,j,k,ic,1) - &
   417)                           dyn_in%elem(ie)%state%Q(i,j,k,ic))*rec2dt*dp_tmp
   418)                   end do
   419)                end do
   420)             end do
   421)          end do
   422)       endif
   423)    end do
   424)    call t_stopf('bndry_exchange')
   425) 
   426) 
   427) 
   428)    ! Most output is done by physics.  We pass to the physics state variables
   429)    ! at timelevel "tl_f".  
   430)    ! we will output dycore variables here to ensure they are always at the same
   431)    ! time as what the physics is writing.  
   432) #if 0
   433)    if (hist_fld_active('VOR')) then
   434)       call compute_zeta_C0(tmp_dyn,elem,hybrid,1,nelemd,tl_f,k)
   435)       do ie=1,nelemd
   436)          do j=1,np
   437)             do i=1,np
   438)                ftmp(i+(j-1)*np,1:pver,1) = tmp_dyn(i,j,1:pver)
   439)             end do
   440)          end do
   441)          call outfld('VOR',ftmp(:,:,1),npsq,ie)
   442)       enddo
   443)    endif
   444)    if (hist_fld_active('DIV')) then
   445)       call compute_div_C0(tmp_dyn,elem,hybrid,1,nelemd,tl_f,k)
   446)       do ie=1,nelemd
   447)          do j=1,np
   448)             do i=1,np
   449)                ftmp(i+(j-1)*np,1:pver,1) = tmp_dyn(i,j,1:pver)
   450)             end do
   451)          end do
   452)          call outfld('DIV',ftmp(:,:,1),npsq,ie)
   453)       enddo
   454)    endif
   455) #endif
   456)    if (smooth_phis_numcycle>0) then
   457)       if (hist_fld_active('PHIS_SM')) then
   458)          do ie=1,nelemd
   459)             do j=1,np
   460)                do i=1,np
   461)                   ftmp(i+(j-1)*np,1,1) = phisdyn(i,j,ie)
