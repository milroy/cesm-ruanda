   149)           ierr = pio_def_var(File, 'CO2DIAG', pio_double, hdimids, co2diag_desc)

   160)           ierr = pio_def_var(File, 'CFLX'//num,  pio_double, hdimids, cflx_desc(i))

   170)        else
   171)           dimids(hdimcnt+1) = pverp_id
   172)        end if

   178)           write(pname,'(a,i3.3)') 'NAL_absorp',i
   179)           ierr = pio_def_var(File, trim(pname), pio_double, dimids(1:hdimcnt+1), abstot_desc(i))
   180)        end do
   181) 	
   182)        dimids(hdimcnt+1) = pverp_id
   183)        ierr = pio_def_var(File, 'Emissivity', pio_double, dimids(1:hdimcnt+1), emstot_desc)
   184) 
   185)        dimids(hdimcnt+1) = pver_id
   186)        do i=1,4
   187)           write(pname,'(a,i3.3)') 'NN_absorp',i
   188)           ierr = pio_def_var(File, pname, pio_double, dimids(1:hdimcnt+1), absnxt_desc(i))
   189)        end do
   190) 
   191) 
   192)     end if
   193)     if (docosp) then
   194)       ierr = pio_def_var(File, 'cosp_cnt_init', pio_int, cospcnt_desc)
   195)     end if
   196) 
   197)     if (is_subcol_on()) then
   198)       call subcol_init_restart(file, hdimids)
   199)     end if
   200)

   203)   subroutine write_restart_physics (File, cam_in, cam_out, pbuf2d)

   210)       use prescribed_ghg,     only: write_prescribed_ghg_restart
   211)       use prescribed_aero,    only: write_prescribed_aero_restart
   212)       use prescribed_volcaero,only: write_prescribed_volcaero_restart
   213)       use radiation,          only: radiation_do

   219)       ! Input arguments

   240)          call subcol_write_restart(File)

   242)

   244)

   246)

   249) 
   250)          call write_prescribed_ozone_restart(File)
   251)          call write_prescribed_ghg_restart(File)

   253)          call write_prescribed_volcaero_restart(File)

   257)             if(ncol<pcols) then
   258)                fsnt(ncol+1:pcols,i) = fillvalue
   259)                fsns(ncol+1:pcols,i) = fillvalue
   260)                fsds(ncol+1:pcols,i) = fillvalue
   261)                flnt(ncol+1:pcols,i) = fillvalue
   262)                flns(ncol+1:pcols,i) = fillvalue
   263)                landm(ncol+1:pcols,i) = fillvalue
   264)                sgh(ncol+1:pcols,i) = fillvalue
   265)                sgh30(ncol+1:pcols,i) = fillvalue

   267)                trefmxav(ncol+1:pcols,i) = fillvalue

   277)             ! Comsrf module variables (can following coup_csm definitions be removed?)

   279)          call pio_write_darray(File, fsns_desc, iodesc, fsns, ierr)

   281)          call pio_write_darray(File, flnt_desc, iodesc, flnt, ierr)
   282)          
   283)          call pio_write_darray(File, flns_desc,  iodesc,  flns, ierr)
   284)          call pio_write_darray(File, landm_desc, iodesc, landm, ierr)
   285)          call pio_write_darray(File, sgh_desc,   iodesc,   sgh, ierr)
   286)          call pio_write_darray(File, sgh30_desc, iodesc, sgh30, ierr)
   287)          
   288)          call pio_write_darray(File, trefmxav_desc, iodesc, trefmxav, ierr)
   289)          call pio_write_darray(File, trefmnav_desc, iodesc, trefmnav, ierr)
   290)

   292)          tmpfield(:) = fillvalue
   293)          do i=begchunk,endchunk

   295)             do j=1,pcols

   299)          end do

   302) 	 ii=0
   303)          do i=begchunk,endchunk
   304)             ncol = cam_out(i)%ncol
   305)             do j=1,pcols
   306)                ii=ii+1
   307)                if(j<=ncol) tmpfield(ii) = cam_out(i)%sols(j)
   308)             end do
   309)          end do
   310)          call pio_write_darray(File, sols_desc, iodesc, tmpfield, ierr)
   311) 
   312) 	 ii=0
   313)          do i=begchunk,endchunk
   314)             ncol = cam_out(i)%ncol
   315)             do j=1,pcols
   316)                ii=ii+1
   317)                if(j<=ncol) tmpfield(ii) = cam_out(i)%soll(j)
   318)             end do
   319)          end do
   320)          call pio_write_darray(File, soll_desc, iodesc, tmpfield, ierr)

   322) 	 ii=0
   323)          do i=begchunk,endchunk
   324)             ncol = cam_out(i)%ncol
   325)             do j=1,pcols
   326)                ii=ii+1
   327)                if(j<=ncol) tmpfield(ii) = cam_out(i)%solsd(j)
   328)             end do
   329)          end do
   330)          call pio_write_darray(File, solsd_desc, iodesc, tmpfield, ierr)
   331) 
   332) 	 ii=0
   333)          do i=begchunk,endchunk
   334)             ncol = cam_out(i)%ncol
   335)             do j=1,pcols
   336)                ii=ii+1
   337)                if(j<=ncol) tmpfield(ii) = cam_out(i)%solld(j)
   338)             end do
   339)          end do
   340)          call pio_write_darray(File, solld_desc, iodesc, tmpfield, ierr)

   343)          do i=begchunk,endchunk
   344)             ncol = cam_out(i)%ncol
   345)             do j=1,pcols
   346)                ii=ii+1
   347)                if(j<=ncol) tmpfield(ii) = cam_out(i)%bcphidry(j)
   348)             end do
   349)          end do
   350)          call pio_write_darray(File, bcphidry_desc, iodesc, tmpfield, ierr)
   351) 
   352) 	 ii=0
   353)          do i=begchunk,endchunk
   354)             ncol = cam_out(i)%ncol
   355)             do j=1,pcols
   356)                ii=ii+1
   357)                if(j<=ncol) tmpfield(ii) = cam_out(i)%bcphodry(j)
   358)             end do
   359)          end do
   360)          call pio_write_darray(File, bcphodry_desc, iodesc, tmpfield, ierr)
   361) 
   362) 	 ii=0
   363)          do i=begchunk,endchunk
   364)             ncol = cam_out(i)%ncol
   365)             do j=1,pcols
   366)                ii=ii+1
   367)                if(j<=ncol) tmpfield(ii) = cam_out(i)%ocphidry(j)

   369)          end do
   370)          call pio_write_darray(File, ocphidry_desc, iodesc, tmpfield, ierr)
   371) 
   372) 	 ii=0
   373)          do i=begchunk,endchunk
   374)             ncol = cam_out(i)%ncol
   375)             do j=1,pcols
   376)                ii=ii+1
   377)                if(j<=ncol) tmpfield(ii) = cam_out(i)%ocphodry(j)
   378)             end do
   379)          end do
   380)          call pio_write_darray(File, ocphodry_desc, iodesc, tmpfield, ierr)
   381) 
   382) 	 ii=0
   383)          do i=begchunk,endchunk
   384)             ncol = cam_out(i)%ncol
   385)             do j=1,pcols
   386)                ii=ii+1
   387)                if(j<=ncol) tmpfield(ii) = cam_out(i)%dstdry1(j)
   388)             end do
   389)          end do
   390)          call pio_write_darray(File, dstdry1_desc, iodesc, tmpfield, ierr)
   391) 
   392) 	 ii=0
   393)          do i=begchunk,endchunk
   394)             ncol = cam_out(i)%ncol
   395)             do j=1,pcols
   396)                ii=ii+1
   397)                if(j<=ncol) tmpfield(ii) = cam_out(i)%dstdry2(j)
   398)             end do
   399)          end do
   400)          call pio_write_darray(File, dstdry2_desc, iodesc, tmpfield, ierr)

   404)             ncol = cam_out(i)%ncol

   417)                if(j<=ncol) tmpfield(ii) = cam_out(i)%dstdry4(j)
   418)             end do
   419)          end do
   420)          call pio_write_darray(File, dstdry4_desc, iodesc, tmpfield, ierr)

   429)                end do
   430)             end do
   431)             call pio_write_darray(File, co2prog_desc, iodesc, tmpfield, ierr)
   432)             ii=0
   433)             do i=begchunk,endchunk
   434)                ncol = cam_out(i)%ncol
   435)                do j=1,pcols
   436)                   ii=ii+1
   437)                   if(j<=ncol) tmpfield(ii) = cam_out(i)%co2diag(j)
   438)                end do
   439)             end do
   440)             call pio_write_darray(File, co2diag_desc, iodesc, tmpfield, ierr)

   442) 
   443)          ! cam_in components
   444)          do m = 1, pcnst
   445)             ii = 0
   446)             do i = begchunk, endchunk
   447)                ncol = cam_in(i)%ncol
