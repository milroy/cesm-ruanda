    52) !
    53) ! Input arguments
    54) !
    55)    integer, intent(in) :: lchnk                 ! chunk identifier
    56)    integer, intent(in) :: ncol                  ! number of atmospheric columns
    57)    integer, intent(in) :: rrtmg_levs            ! number of levels rad is applied

    67) 
    68)    real(r8), intent(in) :: cld(pcols,pver)      ! Cloud cover
    69)    real(r8), intent(in) :: tauc_lw(nbndlw,pcols,pver)   ! Cloud longwave optical depth by band
    70) 
    71) !
    72) ! Output arguments
    73) !
    74)    real(r8), intent(out) :: qrl (pcols,pver)     ! Longwave heating rate
    75)    real(r8), intent(out) :: qrlc(pcols,pver)     ! Clearsky longwave heating rate
    76)    real(r8), intent(out) :: flns(pcols)          ! Surface cooling flux
    77)    real(r8), intent(out) :: flnt(pcols)          ! Net outgoing flux
    78)    real(r8), intent(out) :: flut(pcols)          ! Upward flux at top of model
    79)    real(r8), intent(out) :: flnsc(pcols)         ! Clear sky surface cooing
    80)    real(r8), intent(out) :: flntc(pcols)         ! Net clear sky outgoing flux
    81)    real(r8), intent(out) :: flutc(pcols)         ! Upward clear-sky flux at top of model
    82)    real(r8), intent(out) :: flwds(pcols)         ! Down longwave flux at surface
    83)    real(r8), intent(out) :: fldsc(pcols)         ! Down longwave clear flux at surface
    84)    real(r8), intent(out) :: fcnl(pcols,pverp)    ! clear sky net flux at interfaces
    85)    real(r8), intent(out) :: fnl(pcols,pverp)     ! net flux at interfaces
    86) 
    87)    real(r8), pointer, dimension(:,:,:) :: lu ! longwave spectral flux up
    88)    real(r8), pointer, dimension(:,:,:) :: ld ! longwave spectral flux down
    89)    
    90) !
    91) !---------------------------Local variables-----------------------------
    92) !
    93)    integer :: i, k, kk, nbnd         ! indices
    94) 
    95)    real(r8) :: ful(pcols,pverp)     ! Total upwards longwave flux
    96)    real(r8) :: fsul(pcols,pverp)    ! Clear sky upwards longwave flux
    97)    real(r8) :: fdl(pcols,pverp)     ! Total downwards longwave flux
    98)    real(r8) :: fsdl(pcols,pverp)    ! Clear sky downwards longwv flux
    99) 
   100)    integer :: inflglw               ! Flag for cloud parameterization method
   101)    integer :: iceflglw              ! Flag for ice cloud param method
   102)    integer :: liqflglw              ! Flag for liquid cloud param method
   103)    integer :: icld                  ! Flag for cloud overlap method
   104)                                  ! 0=clear, 1=random, 2=maximum/random, 3=maximum
   105) 
   106)    real(r8) :: tsfc(pcols)          ! surface temperature
   107)    real(r8) :: emis(pcols,nbndlw)   ! surface emissivity
   108) 
   109)    real(r8) :: taua_lw(pcols,rrtmg_levs-1,nbndlw)     ! aerosol optical depth by band
   110) 
   111)    real(r8), parameter :: dps = 1._r8/86400._r8 ! Inverse of seconds per day
   112) 
   113)    ! Cloud arrays for McICA 
   114)    integer, parameter :: nsubclw = ngptlw       ! rrtmg_lw g-point (quadrature point) dimension
   115)    integer :: permuteseed                       ! permute seed for sub-column generator
   116) 
   117)    real(r8) :: cicewp(pcols,rrtmg_levs-1)   ! in-cloud cloud ice water path
   118)    real(r8) :: cliqwp(pcols,rrtmg_levs-1)   ! in-cloud cloud liquid water path
   119)    real(r8) :: rei(pcols,rrtmg_levs-1)      ! ice particle effective radius (microns)
   120)    real(r8) :: rel(pcols,rrtmg_levs-1)      ! liquid particle radius (micron)
   121) 
   122)    real(r8) :: cld_stolw(nsubclw, pcols, rrtmg_levs-1)     ! cloud fraction (mcica)
   123)    real(r8) :: cicewp_stolw(nsubclw, pcols, rrtmg_levs-1)  ! cloud ice water path (mcica)
   124)    real(r8) :: cliqwp_stolw(nsubclw, pcols, rrtmg_levs-1)  ! cloud liquid water path (mcica)
   125)    real(r8) :: rei_stolw(pcols,rrtmg_levs-1)               ! ice particle size (mcica)
   126)    real(r8) :: rel_stolw(pcols,rrtmg_levs-1)               ! liquid particle size (mcica)
   127)    real(r8) :: tauc_stolw(nsubclw, pcols, rrtmg_levs-1)    ! cloud optical depth (mcica - optional)
   128) 
   129)    ! Includes extra layer above model top
   130)    real(r8) :: uflx(pcols,rrtmg_levs+1)  ! Total upwards longwave flux
   131)    real(r8) :: uflxc(pcols,rrtmg_levs+1) ! Clear sky upwards longwave flux
   132)    real(r8) :: dflx(pcols,rrtmg_levs+1)  ! Total downwards longwave flux
   133)    real(r8) :: dflxc(pcols,rrtmg_levs+1) ! Clear sky downwards longwv flux
   134)    real(r8) :: hr(pcols,rrtmg_levs)      ! Longwave heating rate (K/d)
   135)    real(r8) :: hrc(pcols,rrtmg_levs)     ! Clear sky longwave heating rate (K/d)
   136)    real(r8) lwuflxs(nbndlw,pcols,pverp+1)  ! Longwave spectral flux up
   137)    real(r8) lwdflxs(nbndlw,pcols,pverp+1)  ! Longwave spectral flux down
   138)    !-----------------------------------------------------------------------
   139) 
   140)    ! mji/rrtmg
   141) 
   142)    ! Calculate cloud optical properties here if using CAM method, or if using one of the
   143)    ! methods in RRTMG_LW, then pass in cloud physical properties and zero out cloud optical 
   144)    ! properties here
   145)    
   146)    ! Zero optional cloud optical depth input array tauc_lw, 
   147)    ! if inputting cloud physical properties into RRTMG_LW
   148)    !          tauc_lw(:,:,:) = 0.
   149)    ! Or, pass in CAM cloud longwave optical depth to RRTMG_LW
   150)    ! do nbnd = 1, nbndlw
   151)    !    tauc_lw(nbnd,:ncol,:pver) = cldtau(:ncol,:pver)
   152)    ! end do
   153) 
   154)    ! Call mcica sub-column generator for RRTMG_LW
   155) 
   156)    ! Call sub-column generator for McICA in radiation
   157)    call t_startf('mcica_subcol_lw')
   158) 
   159)    ! Select cloud overlap approach (1=random, 2=maximum-random, 3=maximum)
   160)    icld = 2
   161)    ! Set permute seed (must be offset between LW and SW by at least 140 to insure 
   162)    ! effective randomization)
   163)    permuteseed = 150
   164) 
   165)    ! These fields are no longer supplied by CAM.
   166)    cicewp = 0.0_r8
   167)    cliqwp = 0.0_r8
   168)    rei = 0.0_r8
   169)    rel = 0.0_r8
   170) 
   171)    call mcica_subcol_lw(lchnk, ncol, rrtmg_levs-1, icld, permuteseed, pmid(:, pverp-rrtmg_levs+1:pverp-1), &
   172)       cld(:, pverp-rrtmg_levs+1:pverp-1), cicewp, cliqwp, rei, rel, tauc_lw(:, :ncol, pverp-rrtmg_levs+1:pverp-1), &
   173)       cld_stolw, cicewp_stolw, cliqwp_stolw, rei_stolw, rel_stolw, tauc_stolw)
   174) 
   175)    call t_stopf('mcica_subcol_lw')
   176) 
   177)    
   178)    call t_startf('rrtmg_lw')
   179) 
   180)    !
   181)    ! Call RRTMG_LW model
   182)    !
   183)    ! Set input flags for cloud parameterizations
   184)    ! Use separate specification of ice and liquid cloud optical depth.
   185)    ! Use either Ebert and Curry ice parameterization (iceflglw = 0 or 1), 
   186)    ! or use Key (Streamer) approach (iceflglw = 2), or use Fu method
   187)    ! (iceflglw = 3), and Hu/Stamnes for liquid (liqflglw = 1).
   188)    ! For use in Fu method (iceflglw = 3), rei is converted in RRTMG_LW
   189)    ! from effective radius to generalized effective size using the
   190)    ! conversion of D. Mitchell, JAS, 2002.  For ice particles outside
   191)    ! the effective range of either the Key or Fu approaches, the 
   192)    ! Ebert and Curry method is applied. 
   193) 
   194)    ! Input CAM cloud optical depth directly
   195)    inflglw = 0
   196)    iceflglw = 0
   197)    liqflglw = 0
   198)    ! Use E&C approach for ice to mimic CAM3
   199)    !   inflglw = 2
   200)    !   iceflglw = 1
   201)    !   liqflglw = 1
   202)    ! Use merged Fu and E&C params for ice
   203)    !   inflglw = 2
   204)    !   iceflglw = 3
   205)    !   liqflglw = 1
   206) 
   207)    ! Convert incoming water amounts from specific humidity to vmr as needed;
   208)    ! Convert other incoming molecular amounts from mmr to vmr as needed;
   209)    ! Convert pressures from Pa to hPa;
   210)    ! Set surface emissivity to 1.0 here, this is treated in land surface model;
   211)    ! Set surface temperature
   212)    ! Set aerosol optical depth to zero for now
   213) 
   214)    emis(:ncol,:nbndlw) = 1._r8
   215)    tsfc(:ncol) = r_state%tlev(:ncol,rrtmg_levs+1)
   216)    taua_lw(:ncol, 1:rrtmg_levs-1, :nbndlw) = aer_lw_abs(:ncol,pverp-rrtmg_levs+1:pverp-1,:nbndlw)
   217) 
   218)    if (associated(lu)) lu(1:ncol,:,:) = 0.0_r8
   219)    if (associated(ld)) ld(1:ncol,:,:) = 0.0_r8
   220) 
   221)    call rrtmg_lw(lchnk  ,ncol ,rrtmg_levs    ,icld    ,                 &
   222)         r_state%pmidmb  ,r_state%pintmb  ,r_state%tlay    ,r_state%tlev    ,tsfc    ,r_state%h2ovmr, &
   223)         r_state%o3vmr   ,r_state%co2vmr  ,r_state%ch4vmr  ,r_state%o2vmr   ,r_state%n2ovmr  ,r_state%cfc11vmr,r_state%cfc12vmr, &
   224)         r_state%cfc22vmr,r_state%ccl4vmr ,emis    ,inflglw ,iceflglw,liqflglw, &
   225)         cld_stolw,tauc_stolw,cicewp_stolw,cliqwp_stolw ,rei, rel, &
   226)         taua_lw, &
   227)         uflx    ,dflx    ,hr      ,uflxc   ,dflxc   ,hrc, &
   228)         lwuflxs, lwdflxs)
   229) 
   230)    !
   231)    !----------------------------------------------------------------------
   232)    ! All longitudes: store history tape quantities
   233)    ! Flux units are in W/m2 on output from rrtmg_lw and contain output for
   234)    ! extra layer above model top with vertical indexing from bottom to top.
   235)    ! Heating units are in K/d on output from RRTMG and contain output for
   236)    ! extra layer above model top with vertical indexing from bottom to top.
   237)    ! Heating units are converted to J/kg/s below for use in CAM. 
   238) 
   239)    flwds(:ncol) = dflx (:ncol,1)
   240)    fldsc(:ncol) = dflxc(:ncol,1)
   241)    flns(:ncol)  = uflx (:ncol,1) - dflx (:ncol,1)
   242)    flnsc(:ncol) = uflxc(:ncol,1) - dflxc(:ncol,1)
   243)    flnt(:ncol)  = uflx (:ncol,rrtmg_levs) - dflx (:ncol,rrtmg_levs)
   244)    flntc(:ncol) = uflxc(:ncol,rrtmg_levs) - dflxc(:ncol,rrtmg_levs)
   245)    flut(:ncol)  = uflx (:ncol,rrtmg_levs)
   246)    flutc(:ncol) = uflxc(:ncol,rrtmg_levs)
   247) 
   248)    !
   249)    ! Reverse vertical indexing here for CAM arrays to go from top to bottom.
   250)    !
   251)    ful = 0._r8
   252)    fdl = 0._r8
   253)    fsul = 0._r8
   254)    fsdl = 0._r8
   255)    ful (:ncol,pverp-rrtmg_levs+1:pverp)= uflx(:ncol,rrtmg_levs:1:-1)
   256)    fdl (:ncol,pverp-rrtmg_levs+1:pverp)= dflx(:ncol,rrtmg_levs:1:-1)
   257)    fsul(:ncol,pverp-rrtmg_levs+1:pverp)=uflxc(:ncol,rrtmg_levs:1:-1)
   258)    fsdl(:ncol,pverp-rrtmg_levs+1:pverp)=dflxc(:ncol,rrtmg_levs:1:-1)
   259) 
   260)    if (single_column.and.scm_crm_mode) then
   261)       call outfld('FUL     ',ful,pcols,lchnk)
   262)       call outfld('FDL     ',fdl,pcols,lchnk)
   263)       call outfld('FULC    ',fsul,pcols,lchnk)
   264)       call outfld('FDLC    ',fsdl,pcols,lchnk)
   265)    endif
   266)    
   267)    fnl(:ncol,:) = ful(:ncol,:) - fdl(:ncol,:)
   268)    ! mji/ cam excluded this?
   269)    fcnl(:ncol,:) = fsul(:ncol,:) - fsdl(:ncol,:)
   270) 
   271)    ! Pass longwave heating to CAM arrays and convert from K/d to J/kg/s
   272)    qrl = 0._r8
   273)    qrlc = 0._r8
   274)    qrl (:ncol,pverp-rrtmg_levs+1:pver)=hr (:ncol,rrtmg_levs-1:1:-1)*cpair*dps
   275)    qrlc(:ncol,pverp-rrtmg_levs+1:pver)=hrc(:ncol,rrtmg_levs-1:1:-1)*cpair*dps
   276) 
   277)    ! Return 0 above solution domain
   278)    if ( ntoplw > 1 )then
   279)       qrl(:ncol,:ntoplw-1) = 0._r8
   280)       qrlc(:ncol,:ntoplw-1) = 0._r8
   281)    end if
   282) 
   283)    ! Pass spectral fluxes, reverse layering
   284)    ! order=(/3,1,2/) maps the first index of lwuflxs to the third index of lu.
   285)    if (associated(lu)) then
   286)       lu(:ncol,pverp-rrtmg_levs+1:pverp,:) = reshape(lwuflxs(:,:ncol,rrtmg_levs:1:-1), &
   287)            (/ncol,rrtmg_levs,nbndlw/), order=(/3,1,2/))
   288)    end if
   289)    
   290)    if (associated(ld)) then
   291)       ld(:ncol,pverp-rrtmg_levs+1:pverp,:) = reshape(lwdflxs(:,:ncol,rrtmg_levs:1:-1), &
   292)            (/ncol,rrtmg_levs,nbndlw/), order=(/3,1,2/))
   293)    end if
   294)    
   295)    call t_stopf('rrtmg_lw')
   296) 
   297) end subroutine rad_rrtmg_lw

   301) subroutine radlw_init()
   302) !----------------------------------------------------------------------- 
   303) ! 
   304) ! Purpose: 
   305) ! Initialize various constants for radiation scheme.

   307) !-----------------------------------------------------------------------
   308)

   310) 
   311)    integer :: k
   312)

   320)       ntoplw  = 1

   322)    if (masterproc) then
