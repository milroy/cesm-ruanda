   143)    subroutine cam_write_restart( cam_in, cam_out, dyn_out, pbuf2d, &
   144) 	                         yr_spec, mon_spec, day_spec, sec_spec )
   145) 
   146) !----------------------------------------------------------------------- 
   147) ! 
   148) ! Purpose: 
   149) ! Write the primary, secondary, and history buffer regeneration files.
   150) ! 
   151) ! Method: 
   152) ! The cpp SPMD definition provides for the funnelling of all program i/o
   153) ! through the master processor. Processor 0 either reads restart/history
   154) ! data from the disk and distributes it to all processors, or collects
   155) ! data from all processors and writes it to disk.
   156) ! 
   157) ! Author: 
   158) ! 
   159) !----------------------------------------------------------------------- 
   160)      use physics_buffer,            only: physics_buffer_desc
   161)       use cam_history,      only: write_restart_history, init_restart_history
   162)       use radiation,        only: radiation_do
   163)       use time_manager,     only: timemgr_write_restart
   164)       use filenames,        only: caseid, interpret_filename_spec
   165)       use dycore,           only: dycore_is
   166) 
   167)       use time_manager,     only: timemgr_write_restart, timemgr_init_restart
   168)       use restart_dynamics, only: write_restart_dynamics, init_restart_dynamics
   169)       use restart_physics,  only: write_restart_physics, init_restart_physics
   170)       use cam_pio_utils,    only: cam_pio_createfile
   171)       use spmd_utils,       only: iam, mpicom
   172)       !
   173)       ! Arguments
   174)       !
   175)       type(cam_in_t),      intent(in) :: cam_in(begchunk:endchunk)
   176)       type(cam_out_t),     intent(in) :: cam_out(begchunk:endchunk)
   177)       
   178)       type(dyn_export_t),  intent(in) :: dyn_out
   179)       
   180)       type(physics_buffer_desc), pointer  :: pbuf2d(:,:)
   181) 
   182)       integer            , intent(in), optional :: yr_spec         ! Simulation year
   183)       integer            , intent(in), optional :: mon_spec        ! Simulation month
   184)       integer            , intent(in), optional :: day_spec        ! Simulation day
   185)       integer            , intent(in), optional :: sec_spec        ! Seconds into current simulation day
   186)       !
   187)       ! Local workspace
   188)       !
   189)       integer ioerr                 ! write error status
   190)       character(len=nlen) :: fname  ! Restart filename
   191)       integer :: aeres_int = 0
   192)       integer :: ierr
   193)       type(file_desc_t) :: File
   194)       integer, pointer :: hdimids(:)
   195) 
   196) 
   197)       !-----------------------------------------------------------------------
   198)       ! Write the primary restart datasets
   199)       !-----------------------------------------------------------------------
   200) #ifdef DEBUG
   201)       write(iulog,*)'Entered CAM_WRITE_RESTART: writing nstep+1=',nstep+1, ' data to restart file'
   202) #endif
   203)       aeres = radiation_do('aeres')
   204)       if ( aeres ) aeres_int = 1
   205) 
   206)       !-----------------------------------------------------------------------
   207)       ! Write the master restart dataset
   208)       !-----------------------------------------------------------------------
   209) 
   210)       if (present(yr_spec).and.present(mon_spec).and.present(day_spec).and.present(sec_spec)) then
   211)          fname = interpret_filename_spec( rfilename_spec, &
   212)               yr_spec=yr_spec, mon_spec=mon_spec, day_spec=day_spec, sec_spec= sec_spec )
   213)       else
   214)          fname = interpret_filename_spec( rfilename_spec )
   215)       end if
   216) 
   217)       call cam_pio_createfile(File, trim(fname), 0)
   218)       call timemgr_init_restart(File)
   219)       call init_restart_dynamics(File, hdimids, dyn_out)
   220)       call init_restart_physics(File, pbuf2d, hdimids)
   221)       call init_restart_history(File)
   222)       deallocate(hdimids)
   223) 
   224)       ierr = PIO_Put_att(File, PIO_GLOBAL, 'caseid', caseid)
   225)       ierr = PIO_Put_att(File, PIO_GLOBAL, 'aeres', aeres_int)
   226)       if(.not.fullgrid) then
   227)          ierr = PIO_Put_att(File, PIO_GLOBAL, 'NLON', nlon)
   228)          ierr = PIO_Put_att(File, PIO_GLOBAL, 'WNUMMAX', wnummax)
   229)       end if
   230)       ierr = pio_enddef(File)
   231) 
   232) 
   233)       !-----------------------------------------------------------------------
   234)       ! Dynamics, physics, History
   235)       !-----------------------------------------------------------------------
   236)       call timemgr_write_restart(File)
   237)       call write_restart_dynamics(File, dyn_out)
   238)       call write_restart_physics(File, cam_in, cam_out, pbuf2d)
   239) 
   240)       if (present(yr_spec).and.present(mon_spec).and.&
   241)            present(day_spec).and.present(sec_spec)) then
   242)          call write_restart_history ( File, &
   243)               yr_spec=yr_spec, mon_spec=mon_spec, day_spec=day_spec, sec_spec= sec_spec )
   244)       else
   245)          call write_restart_history( File )
   246)       end if
   247)       call pio_closefile(File)
   248)       !-----------------------------------------------------------------------
   249)       ! Close the master restart file
   250)       !-----------------------------------------------------------------------
   251) 
   252)       if (masterproc) then
   253) 	 pname = fname
   254)          call write_rest_pfile()
   255)       end if
   256) 
   257) 
   258) 
   259)    end subroutine cam_write_restart

   263)    subroutine cam_read_restart(cam_in, cam_out, dyn_in, dyn_out, pbuf2d, stop_ymd, stop_tod, NLFileName )
   264) 
   265) !----------------------------------------------------------------------- 
   266) ! 
   267) ! Purpose: 
   268) ! Acquire and position the restart, master, primary and secondary
   269) ! datasets for a continuation run
   270) ! 
   271) ! Method: 
   272) ! 
   273) ! Author: 
   274) ! 
   275) !-----------------------------------------------------------------------
   276)      use physics_buffer, only: physics_buffer_desc
   277)       use restart_physics,  only: read_restart_physics
   278)       use restart_dynamics, only: read_restart_dynamics
   279)       use chem_surfvals,    only: chem_surfvals_init
   280)       use phys_grid,        only: phys_grid_init
   281)       use camsrfexch,       only: atm2hub_alloc, hub2atm_alloc
   282) #if (defined SPMD)
   283)       use spmd_dyn,         only: spmdbuf
   284) #endif
   285)       use cam_history,      only: read_restart_history
   286)       use dycore,           only: dycore_is
   287) 
   288)       use cam_pio_utils,    only: cam_pio_openfile, clean_iodesc_list
   289)       use spmd_utils,       only: iam, mpicom
   290)       use time_manager,     only: timemgr_read_restart, timemgr_restart
   291)       use filenames,        only: caseid, brnch_retain_casename
   292)       use ref_pres,         only: ref_pres_init
   293) 
   294) !
   295) !-----------------------------------------------------------------------
   296) !
   297) ! Arguments
   298) !
   299)    type(cam_in_t),     pointer     :: cam_in(:)
   300)    type(cam_out_t),    pointer     :: cam_out(:)
   301)    type(dyn_import_t), intent(inout) :: dyn_in
   302)    type(dyn_export_t), intent(inout) :: dyn_out
   303)    type(physics_buffer_desc), pointer :: pbuf2d(:,:)
   304)    character(len=*),   intent(in)  :: NLFileName
   305)    integer,            intent(IN)  :: stop_ymd       ! Stop date (YYYYMMDD)
   306)    integer,            intent(IN)  :: stop_tod       ! Stop time of day (sec)
   307) !
   308) ! Local workspace
   309) !
   310)    character(len=nlen) :: locfn          ! Local filename
   311)    character(len=nlen+40) :: errstr
   312)    real(r8) :: tmp_rgrid(plat)
   313)    integer :: ierr, aeres_int, slen, xtype
   314)    type(file_desc_t) :: File
   315)    logical :: filefound
   316) 
   317)       ! lbrnch is false for a restart run (nsrest=1), and true for a 
   318)       ! branch run (nsrest=3).  Only read the restart pointer file for
   319)       ! a restart run.
   320)    aeres = .false.
   321)    if (.not.lbrnch) then
   322)       call read_rest_pfile
   323)    endif
   324)   
   325) !
   326) !------------------------------------------------------------------------
   327) ! Obtain and read the master restart dataset 
   328) !------------------------------------------------------------------------
   329) !
   330) #ifdef DEBUG
   331)    if(masterproc) write(iulog,*)'READ_RESTART_MASTER: Reading master resart dataset'
   332) #endif
   333) 
   334) !
   335) ! Obtain master restart dataset
   336) !
   337)    call getfil (pname, locfn)
   338)    !-----------------------------------------------------------------------
   339)    ! Master restart dataset
   340)    !-----------------------------------------------------------------------
   341)    inquire(FILE=trim(locfn), exist=filefound)
   342) 
   343)    if(.not.filefound) then
   344)       write(errstr,*) 'Could not find restart file ', trim(locfn)
   345)       call endrun(errstr)
   346)    end if
   347) 
   348)    call cam_pio_openfile(File, trim(locfn), 0)
   349)    ierr = pio_inq_att(File, pio_global, 'caseid', xtype, slen)
   350)    ierr = PIO_Get_att(File, PIO_GLOBAL, 'caseid', tcase)
   351)    tcase(slen+1:len(tcase))=''
   352)    ierr = PIO_Get_att(File, PIO_GLOBAL, 'aeres', aeres_int)
   353)    if(aeres_int==1) aeres=.true.
   354)       
   355) ! If these variables are not in the file the call will give an error 
   356) !       which we can safely ignore
   357)    call PIO_SetErrorHandling(File, PIO_BCAST_ERROR)
   358)    ierr = pio_get_att(File, PIO_GLOBAL, 'NLON', tmp_rgrid)
   359)    if(ierr==PIO_NOERR) nlon=tmp_rgrid
   360)    ierr = pio_get_att(File, PIO_GLOBAL, 'WNUMMAX', tmp_rgrid)
   361)    if(ierr==PIO_NOERR) wnummax=tmp_rgrid
   362)    call PIO_SetErrorHandling(File, PIO_INTERNAL_ERROR)
   363)    call timemgr_read_restart(File)
   364) 
   365) 
   366)    if(masterproc) then
   367) 
   368)       if (lbrnch .and. tcase==caseid .and. .not.brnch_retain_casename) then
   369)          write(iulog,*) 'READ_RESTART_MASTER: Must change case name on branch run'
   370)          write(iulog,*) 'Prev case = ',tcase,' current case = ',caseid
   371)          call endrun
   372)       end if
   373) 
   374)       write(iulog,*) 'Files for restart:'
   375) 
   376)    endif  ! end of if-masterproc
   377) 
   378)    ! Restart the time manager.
   379)    call timemgr_restart( stop_ymd=stop_ymd, stop_tod=stop_tod )
   380) 
   381)       !-----------------------------------------------------------------------
   382)       ! Dynamics, physics, History
   383)       !-----------------------------------------------------------------------
   384) 
   385)    call read_restart_dynamics(File, dyn_in, dyn_out, NLFileName)   
   386) 
   387)    call initcom ()
   388)    call phys_grid_init
   389) 
   390)    call hub2atm_alloc(cam_in)
   391)    call atm2hub_alloc(cam_out)
   392) 
   393) 
   394)    ! Initialize physics grid reference pressures (needed by initialize_radbuffer)
   395)    call ref_pres_init()
   396) 
   397)    call read_restart_physics(File, cam_in, cam_out, pbuf2d)
   398) 
   399)    if (nlres .and. .not.lbrnch) then
   400)       call read_restart_history ( File )
   401)    end if
   402) 
   403)    call pio_closefile(File)
   404)    
   405)    !-----------------------------------------------------------------------
   406)    ! Allocate communication buffers for collective communications
   407)    ! between physics and dynamics, if necessary
   408)    !-----------------------------------------------------------------------
   409) 
   410) #if (defined SPMD)
   411)    call spmdbuf ()
   412) #endif
   413) 
   414)    ! Initialize ghg surface values.
   415) 
   416)    call chem_surfvals_init()
   417)    call clean_iodesc_list()
   418) 
   419)  end subroutine cam_read_restart

   423)    subroutine write_rest_pfile
   424) !----------------------------------------------------------------------- 
   425) ! 
   426) ! Purpose: 
   427) !
   428) ! Write out the restart pointer file
   429) !
   430) !----------------------------------------------------------------------- 
   431)    use restart_physics, only: get_abs_restart_filepath
   432)    use cam_history,     only: get_ptapes, get_hist_restart_filepath, &
   433)                               hstwr, get_hfilepath, nfils, mfilt
   434) !-----------------------------------------------------------------------
   435)    integer t      ! Tape number
   436)    integer mtapes ! Number of tapes that are active
   437) 
   438)    if ( nsds == -1 ) nsds = getunit()
   439)    call opnfil(rest_pfile, nsds, 'f')
   440)    rewind nsds
   441)    write (nsds,'(a)') trim(pname)
   442)    write (nsds,'(//a,a)') '# The following lists the other files needed for restarts', &
   443)                         ' (cam only reads the first line of this file).'
   444)    write (nsds,'(a,a)') '# The files below refer to the files needed for the master restart file:', &
   445)                        trim(pname)
   446)    if ( aeres )then
   447)       write (nsds,'(a,a)') '# ', trim(get_abs_restart_filepath())
   448)    end if
   449) !
   450) ! History files: Need restart history files when not a time-step to write history info
   451) ! Need: history files if they are not full
   452) !
   453)    mtapes = get_ptapes( )
   454)    do t=1,mtapes
   455)       if ( .not. hstwr(t) ) then
   456)          write (nsds,'(a,a)') '# ', trim(get_hist_restart_filepath( t ))
   457)       end if
   458)       if ( nfils(t) > 0 .and. nfils(t) < mfilt(t) ) then
   459)          write (nsds,'(a,a)') '# ', trim(get_hfilepath( t ))
   460)       end if
   461)    end do
   462)    close (nsds)
   463)    write(iulog,*)'(WRITE_REST_PFILE): successfully wrote local restart pointer file ',trim(rest_pfile)
   464)    write(iulog,'("---------------------------------------")')
   465)    end subroutine write_rest_pfile
