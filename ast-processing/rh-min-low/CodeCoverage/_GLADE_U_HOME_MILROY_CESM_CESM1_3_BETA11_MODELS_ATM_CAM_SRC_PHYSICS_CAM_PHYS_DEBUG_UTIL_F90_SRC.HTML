     1) module phys_debug_util

     3) !----------------------------------------------------------------------------------------
     4) 
     5) ! Module to facilitate debugging of physics parameterizations.
     6) !
     7) ! The user requests a location for debugging in lat/lon coordinates
     8) ! (degrees).  The initialization routine does a global search to find the
     9) ! column in the physics grid closest to the requested location.  The local
    10) ! indices of that column in the physics decomposition are stored as module
    11) ! data.  The user code then passes the local chunk index of the chunked
    12) ! data into the subroutine that will write diagnostic information for the
    13) ! column.  The function phys_debug_col returns the local column index if
    14) ! the column of interest is contained in the chunk, and zero otherwise.
    15) ! Printing is done only if a column index >0 is returned.
    16) !
    17) ! Phil Rasch, B. Eaton, Feb 2008
    18) !----------------------------------------------------------------------------------------
    19)

    21) use phys_grid,     only: phys_grid_find_col, get_rlat_p, get_rlon_p

    24) use abortutils,    only: endrun

    49) subroutine phys_debug_readnl(nlfile)
    50)

    52)    use units,           only: getunit, freeunit
    53)    use mpishorthand
    54) 
    55)    character(len=*), intent(in) :: nlfile  ! filepath for file containing namelist input
    56) 
    57)    ! Local variables
    58)    integer :: unitn, ierr
    59)    character(len=*), parameter :: subname = 'phys_debug_readnl'
    60) 
    61)    namelist /phys_debug_nl/ phys_debug_lat, phys_debug_lon
    62)    !-----------------------------------------------------------------------------
    63) 
    64)    if (masterproc) then
    65)       unitn = getunit()
    66)       open( unitn, file=trim(nlfile), status='old' )
    67)       call find_group_name(unitn, 'phys_debug_nl', status=ierr)
    68)       if (ierr == 0) then
    69)          read(unitn, phys_debug_nl, iostat=ierr)
    70)          if (ierr /= 0) then
    71)             call endrun(subname // ':: ERROR reading namelist')
    72)          end if
    73)       end if
    74)       close(unitn)
    75)       call freeunit(unitn)
    76)    end if
    77) 
    78) #ifdef SPMD
    79)    ! Broadcast namelist variables
    80)    call mpibcast(phys_debug_lat, 1, mpir8, 0, mpicom)
    81)    call mpibcast(phys_debug_lon, 1, mpir8, 0, mpicom)
    82) #endif
    83)

    88) subroutine phys_debug_init()
    89) 
    90)    integer  :: owner, lchunk, icol
    91)    real(r8) :: deblat, deblon
    92)    !-----------------------------------------------------------------------------
    93) 
    94)    ! If no debug column specified then do nothing
    95)    if (phys_debug_lat == uninit_r8 .or. phys_debug_lon == uninit_r8) return
    96) 
    97)    ! User has specified a column location for debugging.  Find the closest
    98)    ! column in the physics grid.
    99)    call phys_grid_find_col(phys_debug_lat, phys_debug_lon, owner, lchunk, icol)
   100) 
   101)    ! If the column is owned by this process then save its local indices
   102)    if (iam == owner) then
   103)       debchunk         = lchunk
   104)       debcol           = icol
   105)       deblat           = get_rlat_p(lchunk, icol)*57.296_r8  ! approximate conversion for log output only
   106)       deblon           = get_rlon_p(lchunk, icol)*57.296_r8
   107)       write(iulog,*) 'phys_debug_init: debugging column at lat=', deblat, '  lon=', deblon
   108)    end if
   109) 
   110) end subroutine phys_debug_init
