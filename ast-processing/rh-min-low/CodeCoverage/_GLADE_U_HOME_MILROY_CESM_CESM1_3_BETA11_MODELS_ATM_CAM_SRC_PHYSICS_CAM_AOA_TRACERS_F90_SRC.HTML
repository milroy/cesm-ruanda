    50) 
    51)   ! 15-days used for diagnostic of transport circulation and K-tensors  
    52)   ! relaxation (in the original papers PM-1987 and YSGD-2000) => Zonal Mean 
    53)   ! to evaluate eddy-fluxes for 2D-diagnostics, here relaxation to the GLOBAL MEAN  IC
    54)   ! it may help to keep gradients but will rule-out 2D-transport diagnostics
    55)   ! in km  to avoid negative values of  vertical tracers
    56)   ! VERT(k) = -7._r8*alog(hyam(k)+hybm(k)) + vert_offset
    57)   
    58)   ! PM-1987:
    59)   ! Plumb, R. A., and J. D. Mahlman (1987), The zonally averaged transport
    60)   ! characteristics of the GFDL general circulation/transport model,
    61)   ! J. Atmos.Sci.,44, 298–327
    62) 
    63)   ! YSGD-2000:
    64)   ! Yudin, Valery A., Sergey P. Smyshlyaev, Marvin A. Geller, Victor L. Dvortsov, 2000: 
    65)   ! Transport Diagnostics of GCMs and Implications for 2D Chemistry-Transport Model of 
    66)   ! Troposphere and Stratosphere. J. Atmos. Sci., 57, 673–699.
    67)   ! doi: http://dx.doi.org/10.1175/1520-0469(2000)057<0673:TDOGAI>2.0.CO;2
    68) 
    69)   real(r8) :: qrel_vert(pver)  ! = -7._r8*log(pref_mid_norm(k)) + vert_offset

   236)   subroutine aoa_tracers_timestep_init( phys_state )
   237)     !-----------------------------------------------------------------------
   238)     ! Provides a place to reinitialize diagnostic constituents HORZ and VERT
   239)     !-----------------------------------------------------------------------
   240) 
   241)     use time_manager,   only: get_curr_date
   242)     use ppgrid,         only: begchunk, endchunk
   243)     use physics_types,  only: physics_state
   244) 
   245)     type(physics_state), intent(inout), dimension(begchunk:endchunk), optional :: phys_state    
   246) 
   247) 
   248)     integer c, i, k, ncol
   249)     integer yr, mon, day, tod
   250)     !--------------------------------------------------------------------------
   251) 
   252)     if (.not. aoa_tracers_flag) return
   253) 
   254)     call get_curr_date (yr,mon,day,tod)
   255) 
   256)     if ( day == 1 .and. tod == 0) then
   257)        if (masterproc) then
   258)          write(iulog,*) 'AGE_OF_AIR_CONSTITUENTS: RE-INITIALIZING HORZ/VERT CONSTITUENTS'
   259)        endif
   260) 
   261)        do c = begchunk, endchunk
   262)           ncol = phys_state(c)%ncol
   263)           do k = 1, pver
   264)              do i = 1, ncol
   265)                 phys_state(c)%q(i,k,ixht) = 2._r8 + sin(phys_state(c)%lat(i))
   266)                 phys_state(c)%q(i,k,ixvt) = qrel_vert(k)
   267)              end do
   268)           end do
   269)        end do
   270) 
   271)     end if

   277)   subroutine aoa_tracers_timestep_tend(state, ptend, cflx, landfrac, dt)
   278) 
   279)     use physics_types, only: physics_state, physics_ptend, physics_ptend_init
   280)     use phys_grid,     only: get_rlat_all_p , get_lat_all_p
   281)     use cam_history,   only: outfld
   282)     use time_manager,  only: get_nstep
   283) 
   284)     ! Arguments
   285)     type(physics_state), intent(in)    :: state              ! state variables
   286)     type(physics_ptend), intent(out)   :: ptend              ! package tendencies
   287)     real(r8),            intent(inout) :: cflx(pcols,pcnst)  ! Surface constituent flux (kg/m^2/s)
   288)     real(r8),            intent(in)    :: landfrac(pcols)    ! Land fraction
   289)     real(r8),            intent(in)    :: dt                 ! timestep
   290) 
   291)     !----------------- Local workspace-------------------------------
   292) 
   293)     integer :: i, k
   294)     integer :: lchnk                          ! chunk identifier
   295)     integer :: ncol                           ! no. of column in chunk
   296)     integer :: nstep                          ! current timestep number
   297)     real(r8) :: qrel                          ! value to be relaxed to
   298)     real(r8) :: xhorz                         ! updated value of HORZ
   299)     real(r8) :: xvert                         ! updated value of VERT
   300)     logical  :: lq(pcnst)
   301)     real(r8) :: teul                          ! relaxation in  1/sec*dt/2 = k*dt/2
   302)     real(r8) :: wimp                          !     1./(1.+ k*dt/2)
   303)     real(r8) :: wsrc                          !  teul*wimp    
   304)     !------------------------------------------------------------------
   305) 
   306)     teul = .5_r8*dt/(86400._r8 * treldays)   ! 1/2 for the semi-implicit scheme if dt=time step
   307)     wimp = 1._r8/(1._r8 +teul)
   308)     wsrc = teul*wimp
   309) 
   310)     if (.not. aoa_tracers_flag) then
   311)        call physics_ptend_init(ptend,state%psetcols,'none') !Initialize an empty ptend for use with physics_update
   312)        return
   313)     end if
   314) 
   315)     lq(:)      = .FALSE.
   316)     lq(ixaoa1) = .TRUE.
   317)     lq(ixaoa2) = .TRUE.
   318)     lq(ixht)   = .TRUE.
   319)     lq(ixvt)   = .TRUE.
   320)     call physics_ptend_init(ptend,state%psetcols, 'aoa_tracers', lq=lq)
   321) 
   322)     nstep = get_nstep()
   323)     lchnk = state%lchnk
   324)     ncol  = state%ncol
   325) 
   326)     do k = 1, pver
   327)        do i = 1, ncol
   328) 
   329)           ! AOA1
   330)           ptend%q(i,k,ixaoa1) = 0.0_r8
   331) 
   332)           ! AOA2
   333)           ptend%q(i,k,ixaoa2) = 0.0_r8
   334) 
   335)           ! HORZ
   336)           qrel              = 2._r8 + sin(state%lat(i))          ! qrel  should zonal mean
   337)           xhorz             = state%q(i,k,ixht)*wimp + wsrc*qrel ! Xnew = weight*3D-tracer + (1.-weight)*1D-tracer
   338)           ptend%q(i,k,ixht) = (xhorz - state%q(i,k,ixht)) / dt   ! Xnew = weight*3D-tracer + (1.-weight)*2D-tracer  zonal mean
   339)                                                                  !  Can be still used .... to diagnose fluxes OT-tracers
   340)           ! VERT
   341)           qrel              = qrel_vert(k)                       ! qrel  should zonal mean
   342)           xvert             = wimp*state%q(i,k,ixvt) + wsrc*qrel
   343)           ptend%q(i,k,ixvt) = (xvert - state%q(i,k,ixvt)) / dt
   344) 
   345)        end do
   346)     end do
   347) 
   348)     ! record tendencies on history files
   349)     call outfld (src_names(1), ptend%q(:,:,ixaoa1), pcols, lchnk)
   350)     call outfld (src_names(2), ptend%q(:,:,ixaoa2), pcols, lchnk)
   351)     call outfld (src_names(3), ptend%q(:,:,ixht),   pcols, lchnk)
   352)     call outfld (src_names(4), ptend%q(:,:,ixvt),   pcols, lchnk)
   353) 
   354)     ! Set tracer fluxes
   355)     do i = 1, ncol
   356) 
   357)        ! AOA1
   358)        cflx(i,ixaoa1) = 1.e-6_r8
   359) 
   360)        ! AOA2
   361)        if (landfrac(i) .eq. 1._r8  .and.  state%lat(i) .gt. 0.35_r8) then
   362)           cflx(i,ixaoa2) = 1.e-6_r8 + 1e-6_r8*0.0434_r8*real(nstep,r8)*dt/(86400._r8*365._r8)
   363)        else
   364)           cflx(i,ixaoa2) = 0._r8
   365)        endif
   366) 
   367)        ! HORZ
   368)        cflx(i,ixht) = 0._r8
   369) 
   370)        ! VERT
   371)        cflx(i,ixvt) = 0._r8
   372) 
   373)     end do
   374) 
   375)   end subroutine aoa_tracers_timestep_tend

   379)   subroutine init_cnst_3d(m, q, gcid)
   380) 
   381)     use dyn_grid, only : get_horiz_grid_d, get_horiz_grid_dim_d
   382)     use dycore,   only : dycore_is
   383) 
   384)     integer,  intent(in)  :: m       ! global constituent index
   385)     real(r8), intent(out) :: q(:,:)  ! kg tracer/kg dry air (gcol,plev)
   386)     integer,  intent(in)  :: gcid(:) ! global column id
   387) 
   388)     real(r8), allocatable :: lat(:)
   389)     integer :: plon, plat, ngcols
   390)     integer :: j, k, gsize
   391)     !-----------------------------------------------------------------------
   392) 
   393)     if (masterproc) write(iulog,*) 'AGE-OF-AIR CONSTITUENTS: INITIALIZING ',cnst_name(m),m
   394) 
   395)     if (m == ixaoa1) then
   396) 
   397)        q(:,:) = 0.0_r8
   398) 
   399)     else if (m == ixaoa2) then
   400) 
   401)        q(:,:) = 0.0_r8
   402) 
   403)     else if (m == ixht) then
   404) 
   405)        call get_horiz_grid_dim_d( plon, plat )
   406)        ngcols = plon*plat
   407)        gsize = size(gcid)
   408)        allocate(lat(ngcols))
   409)        call get_horiz_grid_d(ngcols,clat_d_out=lat)
   410)        do j = 1, gsize
   411)           q(j,:) = 2._r8 + sin(lat(gcid(j)))
   412)        end do
   413)        deallocate(lat)
   414) 
   415)     else if (m == ixvt) then
   416) 
   417)        do k = 1, pver
   418)           do j = 1, size(q,1)
   419)              q(j,k) = qrel_vert(k)
   420)           end do
   421)        end do
   422) 
   423)     end if
   424) 
   425)   end subroutine init_cnst_3d
