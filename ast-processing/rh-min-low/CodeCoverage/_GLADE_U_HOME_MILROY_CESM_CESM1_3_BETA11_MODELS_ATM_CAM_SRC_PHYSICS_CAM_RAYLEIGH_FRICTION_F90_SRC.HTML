     2) module rayleigh_friction

     4)   !---------------------------------------------------------------------------------
     5)   ! Module to apply rayleigh friction in region of model top.
     6)   ! We specify a decay rate profile that is largest at the model top and
     7)   ! drops off vertically using a hyperbolic tangent profile.
     8)   ! We compute the tendencies in u and v using an Euler backward scheme.
     9)   ! We then apply the negative of the kinetic energy tendency to "s", the dry
    10)   ! static energy.
    11)   !
    12)   ! calling sequence:
    13)   !
    14)   !  rayleigh_friction_init          initializes rayleigh friction constants
    15)   !  rayleigh_friction_tend          computes rayleigh friction tendencies
    16)   !
    17)   !---------------------------Code history--------------------------------
    18)   ! This is a new routine written by Art Mirin in collaboration with Phil Rasch.
    19)   ! Initial coding for this version:  Art Mirin, May 2007.
    20)   !---------------------------------------------------------------------------------
    21) 
    22)   use shr_kind_mod,     only: r8 => shr_kind_r8
    23)   use ppgrid,           only: pver
    24)   use spmd_utils,       only: masterproc
    25)   use cam_logfile,      only: iulog

    27)   implicit none
    28)   private          ! Make default type private to the module
    29)   save
    30)   
    31)   !
    32)   ! Public interfaces
    33)   !
    34)   public rayleigh_friction_init          ! Initialization
    35)   public rayleigh_friction_tend          ! Computation of tendencies

    37)   !
    38)   ! Public data
    39)   !
    40)   integer, public   :: rayk0 = 2           ! vertical level at which rayleigh friction term is centered
    41)   real (r8), public :: raykrange = 0._r8   ! range of rayleigh friction profile 
    42)                                            ! if 0, range is set to satisfy x=2 (see below)
    43)   real (r8), public :: raytau0 = 0._r8     ! approximate value of decay time at model top (days)
    44)                                            ! if 0., no rayleigh friction is applied
    45) 
    46)   ! 
    47)   ! Private data
    48)   !
    49)   real (r8) :: krange         ! range of rayleigh friction profile 
    50)   real (r8) :: tau0           ! approximate value of decay time at model top
    51)   real (r8) :: otau0          ! inverse of tau0
    52)   real (r8) :: otau(pver)     ! inverse decay time versus vertical level
    53) 
    54)   ! We apply a profile of the form otau0 * [1 + tanh (x)] / 2 , where
    55)   ! x = (k0 - k) / krange. The default is for x to equal 2 at k=1, meaning
    56)   ! krange = (k0 - 1) / 2. The default is applied when raykrange is set to 0.
    57)   ! If otau0 = 0, no term is applied.
    58) 
    59) contains

    61)   !===============================================================================
    62)   subroutine rayleigh_friction_init()
    63)     !------------------------------Arguments--------------------------------
    64) 
    65)     !---------------------------Local storage-------------------------------
    66)     real (r8) x
    67)     integer k
    68) 
    69)     !-----------------------------------------------------------------------
    70)     ! Compute tau array
    71)     !-----------------------------------------------------------------------
    72) 
    73)     krange = raykrange
    74)     if (raykrange .eq. 0._r8) krange = (rayk0 - 1) / 2._r8
    75) 
    76)     tau0 = (86400._r8) * raytau0   ! convert to seconds
    77)     otau0 = 0._r8
    78)     if (tau0 .ne. 0._r8) otau0 = 1._r8/tau0
    79) 
    80)     do k = 1, pver
    81)        x = (rayk0 - k) / krange
    82)        otau(k) = otau0 * (1 + tanh(x)) / (2._r8)
    83)     enddo
    84) 
    85)     if (masterproc) then
    86)        write (iulog,*) 'Rayleigh friction - rayk0 = ', rayk0
    87)        write (iulog,*) 'Rayleigh friction - raykrange = ', raykrange
    88)        write (iulog,*) 'Rayleigh friction - raytau0 = ', raytau0
    89)        write (iulog,*) 'Rayleigh friction - krange = ', krange
    90)        write (iulog,*) 'Rayleigh friction - otau0 = ', otau0
    91)        write (iulog,*) 'Rayleigh friction decay rate profile'
    92)        do k = 1, pver
    93)           write (iulog,*) '   k = ', k, '   otau = ', otau(k)
    94)        enddo
    95)     end if
    96) 
    97)     return
    98) 
    99)   end subroutine rayleigh_friction_init

   101) !=========================================================================================
   102)   subroutine rayleigh_friction_tend(                                     &
   103)        ztodt    ,state    ,ptend    )
   104)     !-----------------------------------------------------------------------
   105)     ! interface routine for rayleigh friction
   106)     !-----------------------------------------------------------------------
   107)     use physics_types, only: physics_state, physics_ptend, physics_ptend_init
   108) 
   109) 
   110)     !------------------------------Arguments--------------------------------
   111)     real(r8), intent(in) :: ztodt                  ! physics timestep
   112)     type(physics_state), intent(in)  :: state      ! physics state variables
   113)     
   114)     type(physics_ptend), intent(out) :: ptend      ! individual parameterization tendencies
   115)     !
   116)     !---------------------------Local storage-------------------------------
   117)     integer :: ncol                                ! number of atmospheric columns
   118)     integer :: k                                   ! level
   119)     real(r8) :: rztodt                             ! 1./ztodt
   120)     real(r8) :: c1, c2, c3                         ! temporary variables
   121)     !-----------------------------------------------------------------------
   122) 
   123)     call physics_ptend_init(ptend, state%psetcols, 'rayleigh friction', ls=.true., lu=.true., lv=.true.)
   124) 
   125)     if (otau0 .eq. 0._r8) return
   126) 
   127)     rztodt = 1._r8/ztodt
   128)     ncol  = state%ncol
   129) 
   130)     ! u, v and s are modified by rayleigh friction
   131) 
   132)     do k = 1, pver
   133)        c2 = 1._r8 / (1._r8 + otau(k)*ztodt)
   134)        c1 = -otau(k) * c2
   135)        c3 = 0.5_r8 * (1._r8 - c2*c2) * rztodt
   136)        ptend%u(:ncol,k) = c1 * state%u(:ncol,k)
   137)        ptend%v(:ncol,k) = c1 * state%v(:ncol,k)
   138)        ptend%s(:ncol,k) = c3 * (state%u(:ncol,k)**2 + state%v(:ncol,k)**2)
   139)     enddo
   140) 
   141)     return
   142)   end subroutine rayleigh_friction_tend
