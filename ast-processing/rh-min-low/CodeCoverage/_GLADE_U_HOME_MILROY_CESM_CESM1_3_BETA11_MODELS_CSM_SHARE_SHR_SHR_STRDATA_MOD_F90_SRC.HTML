   149)   subroutine shr_strdata_init(SDAT,mpicom,compid,name,scmmode,scmlon,scmlat, &
   150)                               gsmap,ggrid,nxg,nyg,calendar)
   151) 
   152)   implicit none
   153) 
   154)   type(shr_strdata_type),intent(inout) :: SDAT
   155)   integer(IN)           ,intent(in)    :: mpicom
   156)   integer(IN)           ,intent(in)    :: compid
   157)   character(len=*)      ,intent(in),optional :: name
   158)   logical               ,intent(in),optional :: scmmode
   159)   real(R8)              ,intent(in),optional :: scmlon
   160)   real(R8)              ,intent(in),optional :: scmlat
   161)   type(mct_gsmap)       ,intent(in),optional :: gsmap
   162)   type(mct_ggrid)       ,intent(in),optional :: ggrid
   163)   integer(IN)           ,intent(in),optional :: nxg
   164)   integer(IN)           ,intent(in),optional :: nyg
   165)   character(len=*)      ,intent(in),optional :: calendar
   166) 
   167)   integer(IN) :: n,m,k         ! generic index
   168)   integer(IN) :: nu,nv         ! u,v index
   169)   integer(IN) :: my_task,npes  ! my task, total pes
   170)   integer(IN),parameter :: master_task = 0
   171)   character(CS) :: lname       ! local name
   172)   character(CL) :: filePath    ! generic file path
   173)   character(CL) :: fileName    ! generic file name
   174)   character(CS) :: timeName    ! domain file: time variable name
   175)   character(CS) ::  lonName    ! domain file: lon  variable name
   176)   character(CS) ::  latName    ! domain file: lat  variable name
   177)   character(CS) :: maskName    ! domain file: mask variable name
   178)   character(CS) :: areaName    ! domain file: area variable name
   179)   character(CS) :: uname       ! u vector field name
   180)   character(CS) :: vname       ! v vector field name
   181)   character(CXX):: fldList     ! list of fields
   182)   integer(IN)   :: lsize
   183)   integer(IN)   :: nfiles
   184)   integer(IN)   :: ierr
   185)   integer(IN)   :: method
   186)   integer(IN), pointer :: dof(:)
   187)   type(mct_sMat):: sMati
   188)   logical       :: lscmmode
   189) 
   190)   character(len=*),parameter :: subname = "(shr_strdata_init) "
   191)   character(*),parameter ::   F00 = "('(shr_strdata_init) ',8a)" 
   192) 
   193) !-------------------------------------------------------------------------------
   194) !
   195) !-------------------------------------------------------------------------------
   196) 
   197)   call MPI_COMM_SIZE(mpicom,npes,ierr)
   198)   call MPI_COMM_RANK(mpicom,my_task,ierr)
   199)   !--- Count streams again in case user made changes ---
   200)   if (my_task == master_task) then
   201)      do n=1,nStrMax
   202)         !--- check if a streams string is defined in strdata
   203)         if (trim(SDAT%streams(n)) /= trim(shr_strdata_nullstr)) SDAT%nstreams = max(SDAT%nstreams,n)
   204)         !--- check if a filename is defined in the stream
   205)         call shr_stream_getNFiles(SDAT%stream(n),nfiles)
   206)         if (nfiles > 0) SDAT%nstreams = max(SDAT%nstreams,n)
   207) 
   208)         if (trim(SDAT%taxMode(n)) == trim(shr_stream_taxis_extend)) SDAT%dtlimit(n) = 1.0e30
   209)      end do
   210)      SDAT%nvectors = 0
   211)      do n=1,nVecMax
   212)         if (trim(SDAT%vectors(n)) /= trim(shr_strdata_nullstr)) SDAT%nvectors = n
   213)      end do
   214)   endif
   215) 
   216)   call shr_mpi_bcast(SDAT%nstreams  ,mpicom,'nstreams')
   217)   call shr_mpi_bcast(SDAT%nvectors  ,mpicom,'nvectors')
   218)   call shr_mpi_bcast(SDAT%dtlimit   ,mpicom,'dtlimit')
   219) 
   220)   n = 0
   221)   if (present(gsmap)) then
   222)      n = n + 1
   223)   endif
   224)   if (present(ggrid)) then
   225)      n = n + 1
   226)   endif
   227)   if (present(nxg)) then
   228)      n = n + 1
   229)   endif
   230)   if (present(nyg)) then
   231)      n = n + 1
   232)   endif
   233) 
   234)   if ( n == 0 .or. n == 4) then
   235)      ! either all set or none set, this is OK
   236)   else
   237)      write(logunit,*) subname,' ERROR: gsmap, ggrid, nxg, and nyg  must be specified together'
   238)      call shr_sys_abort(subname//' ERROR: gsmap, ggrid, nxg, nyg  set together')
   239)   endif
   240) 
   241)   lscmmode = .false.
   242)   if (present(scmmode)) then
   243)      lscmmode = scmmode
   244)      if (lscmmode) then
   245)         if (.not.present(scmlon) .or. .not.present(scmlat)) then
   246)            write(logunit,*) subname,' ERROR: scmmode requires scmlon and scmlat'
   247)            call shr_sys_abort(subname//' ERROR: scmmode1 lon lat')
   248)         endif
   249)      endif
   250)   endif
   251) 
   252)   lname = ""
   253)   if (present(name)) then
   254)      lname = "_"//trim(name)
   255)   endif
   256) 
   257)   if (present(calendar)) then
   258)      SDAT%calendar = trim(shr_cal_calendarName(trim(calendar)))
   259)   endif
   260) 
   261)   ! --- initialize streams and stream domains ---
   262) 
   263) 
   264)   do n = 1,SDAT%nstreams
   265)      if (my_task == master_task) then
   266)         call shr_stream_getDomainInfo(SDAT%stream(n),filePath,fileName,timeName,lonName, &
   267)              latName,maskName,areaName)
   268)         call shr_stream_getFile(filePath,fileName)
   269)       endif
   270)       call shr_mpi_bcast(fileName,mpicom)
   271)       call shr_mpi_bcast(lonName,mpicom)
   272)       call shr_mpi_bcast(latName,mpicom)
   273)       call shr_mpi_bcast(maskName,mpicom)
   274)       call shr_mpi_bcast(areaName,mpicom)
   275)       call shr_dmodel_readgrid(SDAT%gridR(n),SDAT%gsmapR(n),SDAT%strnxg(n),SDAT%strnyg(n), &
   276)            fileName, compid, mpicom, '1d', lonName, latName, maskName, areaName)
   277)       SDAT%lsizeR(n) = mct_gsmap_lsize(SDAT%gsmapR(n),mpicom)
   278)       call mct_gsmap_OrderedPoints(SDAT%gsmapR(n), my_task, dof)
   279)       call pio_initdecomp(SDAT%pio_subsystem, pio_double, &
   280)            (/SDAT%strnxg(n),SDAT%strnyg(n)/), dof, SDAT%pio_iodesc(n))
   281)       deallocate(dof)
   282) 
   283)       call shr_mpi_bcast(SDAT%stream(n)%calendar,mpicom)
   284)   enddo
   285) 
   286)   ! --- initialize model domain ---
   287) 
   288)   if (present(gsmap)) then
   289)      SDAT%nxg = nxg
   290)      SDAT%nyg = nyg
   291)      lsize = mct_gsmap_lsize(gsmap,mpicom)
   292)      call mct_gsmap_Copy(gsmap,SDAT%gsmap)
   293)      call mct_ggrid_init(SDAT%grid, ggrid, lsize)    
   294)      call mct_aVect_copy(ggrid%data, SDAT%grid%data)
   295)   else
   296)      if (trim(SDAT%domainfile) == trim(shr_strdata_nullstr)) then
   297)         if (SDAT%nstreams > 0) then
   298)            if (my_task == master_task) then
   299)               call shr_stream_getDomainInfo(SDAT%stream(1),filePath,fileName,timeName,lonName, &
   300)                       latName,maskName,areaName)
   301)               call shr_stream_getFile(filePath,fileName)
   302)            endif
   303)            call shr_mpi_bcast(fileName,mpicom)
   304)            call shr_mpi_bcast(lonName,mpicom)
   305)            call shr_mpi_bcast(latName,mpicom)
   306)            call shr_mpi_bcast(maskName,mpicom)
   307)            call shr_mpi_bcast(areaName,mpicom)
   308)            if (lscmmode) then
   309)               call shr_dmodel_readgrid(SDAT%grid,SDAT%gsmap,SDAT%nxg,SDAT%nyg, &
   310)                    fileName, compid, mpicom, '1d', lonName, latName, maskName, areaName, &
   311)                    scmmode=lscmmode,scmlon=scmlon,scmlat=scmlat)
   312)            else
   313)               call shr_dmodel_readgrid(SDAT%grid,SDAT%gsmap,SDAT%nxg,SDAT%nyg, &
   314)                    fileName, compid, mpicom, '1d', lonName, latName, maskName, areaName)
   315)           endif
   316)         endif
   317)      else
   318)          if (lscmmode) then
   319)             call shr_dmodel_readgrid(SDAT%grid,SDAT%gsmap,SDAT%nxg,SDAT%nyg, &
   320)                  SDAT%domainfile, compid, mpicom, '1d', readfrac=.true., &
   321)                  scmmode=lscmmode,scmlon=scmlon,scmlat=scmlat)
   322)          else
   323)             call shr_dmodel_readgrid(SDAT%grid,SDAT%gsmap,SDAT%nxg,SDAT%nyg, &
   324)                  SDAT%domainfile, compid, mpicom, '1d', readfrac=.true.)
   325)          endif
   326)      endif
   327)   endif
   328)   SDAT%lsize = mct_gsmap_lsize(SDAT%gsmap,mpicom)
   329) 
   330)   ! --- setup mapping ---
   331) 
   332)   do n = 1,SDAT%nstreams
   333)      if (shr_dmodel_gGridCompare(SDAT%gridR(n),SDAT%gsmapR(n),SDAT%grid,SDAT%gsmap, &
   334)          shr_dmodel_gGridCompareMaskSubset,mpicom) .or. trim(SDAT%fillalgo(n))=='none') then
   335)         SDAT%dofill(n) = .false.
   336)      else
   337)         SDAT%dofill(n) = .true.
   338)      endif
   339)      if (trim(SDAT%mapmask(n)) == 'dstmask') then
   340)         method = shr_dmodel_gGridCompareXYabsMask
   341)      else
   342)         method = shr_dmodel_gGridCompareXYabs
   343)      endif
   344)      if (shr_dmodel_gGridCompare(SDAT%gridR(n),SDAT%gsmapR(n),SDAT%grid,SDAT%gsmap, &
   345)          method,mpicom,0.01_r8) .or. trim(SDAT%mapalgo(n))=='none') then
   346)         SDAT%domaps(n) = .false.
   347)      else
   348)         SDAT%domaps(n) = .true.
   349)      endif
   350) 
   351)      if (SDAT%dofill(n)) then
   352)         if (trim(SDAT%fillread(n)) == trim(shr_strdata_unset)) then
   353)            if (my_task == master_task) then
   354)              write(logunit,F00) ' calling shr_dmodel_mapSet for fill'
   355)              call shr_sys_flush(logunit)
   356)            endif
   357)            call shr_dmodel_mapSet(SDAT%sMatPf(n), &
   358)              SDAT%gridR(n),SDAT%gsmapR(n),SDAT%strnxg(n),SDAT%strnyg(n), &
   359)              SDAT%gridR(n),SDAT%gsmapR(n),SDAT%strnxg(n),SDAT%strnyg(n), &
   360)              name='mapFill', type='cfill', &
   361)              algo=trim(SDAT%fillalgo(n)),mask=trim(SDAT%fillmask(n)),vect='scalar', &
   362)              compid=compid,mpicom=mpicom)
   363)            if (trim(SDAT%fillwrit(n)) /= trim(shr_strdata_unset)) then
   364)              if (my_task == master_task) then
   365)                write(logunit,F00) ' writing ',trim(SDAT%fillwrit(n))
   366)                call shr_sys_flush(logunit)
   367)              endif
   368)              call shr_mct_sMatWritednc(SDAT%sMatPf(n)%Matrix,SDAT%pio_subsystem,sdat%io_type,  SDAT%fillwrit(n),compid,mpicom)
   369)            endif
   370)         else
   371)            if (my_task == master_task) then
   372)              write(logunit,F00) ' reading ',trim(SDAT%fillread(n))
   373)              call shr_sys_flush(logunit)
   374)            endif
   375)            call shr_mct_sMatReaddnc(sMati,SDAT%gsmapR(n),SDAT%gsmapR(n),'src', &
   376)              filename=trim(SDAT%fillread(n)),mytask=my_task,mpicom=mpicom)
   377)            call mct_sMatP_Init(SDAT%sMatPf(n),sMati,SDAT%gsMapR(n),SDAT%gsmapR(n),0, mpicom, compid)
   378)            call mct_sMat_Clean(sMati)
   379)         endif
   380)      endif
   381)      if (SDAT%domaps(n)) then
   382)         if (trim(SDAT%mapread(n)) == trim(shr_strdata_unset)) then
   383)            if (my_task == master_task) then
   384)              write(logunit,F00) ' calling shr_dmodel_mapSet for remap'
   385)              call shr_sys_flush(logunit)
   386)            endif
   387)            call shr_dmodel_mapSet(SDAT%sMatPs(n), &
   388)              SDAT%gridR(n),SDAT%gsmapR(n),SDAT%strnxg(n),SDAT%strnyg(n), &
   389)              SDAT%grid ,SDAT%gsmap ,SDAT%nxg      ,SDAT%nyg, &
   390)              name='mapScalar', type='remap', &
   391)              algo=trim(SDAT%mapalgo(n)),mask=trim(SDAT%mapmask(n)), vect='scalar', &
   392)              compid=compid,mpicom=mpicom)
   393)            if (trim(SDAT%mapwrit(n)) /= trim(shr_strdata_unset)) then
   394)              if (my_task == master_task) then
   395)                write(logunit,F00) ' writing ',trim(SDAT%mapwrit(n))
   396)                call shr_sys_flush(logunit)
   397)              endif
   398)              call shr_mct_sMatWritednc(SDAT%sMatPs(n)%Matrix,sdat%pio_subsystem,sdat%io_type,SDAT%mapwrit(n),compid,mpicom)
   399)            endif
   400)         else
   401)            if (my_task == master_task) then
   402)              write(logunit,F00) ' reading ',trim(SDAT%mapread(n))
   403)              call shr_sys_flush(logunit)
   404)            endif
   405)            call shr_mct_sMatReaddnc(sMati,SDAT%gsmapR(n),SDAT%gsmap,'src', &
   406)              filename=trim(SDAT%mapread(n)),mytask=my_task,mpicom=mpicom)
   407)            call mct_sMatP_Init(SDAT%sMatPs(n),sMati,SDAT%gsMapR(n),SDAT%gsmap,0, mpicom, compid)
   408)            call mct_sMat_Clean(sMati)
   409)         endif
   410)      else
   411)         call mct_rearr_init(SDAT%gsmapR(n), SDAT%gsmap, mpicom, SDAT%rearrR(n))
   412)      endif
   413)   enddo
   414) 
   415)   ! --- setup datatypes ---
   416) 
   417)   do n = 1,SDAT%nstreams
   418)      if (my_task == master_task) then
   419)         call shr_stream_getModelFieldList(SDAT%stream(n),fldList)
   420)      endif
   421)      call shr_mpi_bcast(fldList,mpicom)
   422)      call mct_aVect_init(SDAT%avs(n)  ,rlist=fldList,lsize=SDAT%lsize)
   423)      call mct_aVect_init(SDAT%avFLB(n),rlist=fldList,lsize=SDAT%lsize)
   424)      call mct_aVect_init(SDAT%avFUB(n),rlist=fldList,lsize=SDAT%lsize)
   425)      call mct_aVect_init(SDAT%avRLB(n),rlist=fldList,lsize=SDAT%lsizeR(n))
   426)      call mct_aVect_init(SDAT%avRUB(n),rlist=fldList,lsize=SDAT%lsizeR(n))
   427)      if (trim(SDAT%tintalgo(n)) == 'coszen') then
   428)         call mct_aVect_init(SDAT%avCoszen(n),rlist="tavCosz",lsize=SDAT%lsize)
   429)      endif
   430)   enddo
   431) 
   432)   ! --- check vectors and compute ustrm,vstrm ---
   433) 
   434)   do m = 1,SDAT%nvectors
   435)      if (.not. shr_string_listIsValid(SDAT%vectors(m))) then
   436)         write(logunit,*) trim(subname),' vec fldlist invalid m=',m,trim(SDAT%vectors(m))
   437)         call shr_sys_abort(subname//': vec fldlist invalid:'//trim(SDAT%vectors(m)))
   438)      endif
   439)      if (shr_string_listGetNum(SDAT%vectors(m)) /= 2) then
   440)         write(logunit,*) trim(subname),' vec fldlist ne 2 m=',m,trim(SDAT%vectors(m))
   441)         call shr_sys_abort(subname//': vec fldlist ne 2:'//trim(SDAT%vectors(m)))
   442)      endif
   443)      call shr_string_listGetName(SDAT%vectors(m),1,uname)
   444)      call shr_string_listGetName(SDAT%vectors(m),2,vname)
   445)      nu = 0
   446)      nv = 0
   447)      do n = 1,SDAT%nstreams
   448)         k = mct_aVect_indexRA(SDAT%avRLB(n),trim(uname),perrWith='quiet')
   449)         if (k > 0) nu = n
   450)         k = mct_aVect_indexRA(SDAT%avRLB(n),trim(vname),perrWith='quiet')
   451)         if (k > 0) nv = n
   452)      enddo
   453)      if (nu == 0  .or. nv == 0) then
   454)         write(logunit,*) trim(subname),' vec flds not found  m=',m,trim(SDAT%vectors(m))
   455)         call shr_sys_abort(subname//': vec flds not found:'//trim(SDAT%vectors(m)))
   456)      endif
   457)      if (nu /= nv) then
   458)         if ((.not. shr_dmodel_gGridCompare(SDAT%gridR(nu),SDAT%gsmapR(nu), &
   459)                                          SDAT%gridR(nv),SDAT%gsmapR(nv), &
   460)                                          shr_dmodel_gGridCompareXYabs,mpicom,0.01_r8)) .or. &
   461)             (.not. shr_dmodel_gGridCompare(SDAT%gridR(nu),SDAT%gsmapR(nu), &
   462)                                          SDAT%gridR(nv),SDAT%gsmapR(nv), &
   463)                                          shr_dmodel_gGridCompareMaskZeros,mpicom))) then
   464)            write(logunit,*) trim(subname),' vec fld doms not same m=',m,trim(SDAT%vectors(m))
   465)            call shr_sys_abort(subname//': vec fld doms not same:'//trim(SDAT%vectors(m)))
   466)         endif
   467)      endif
   468)      SDAT%ustrm(m) = nu
   469)      SDAT%vstrm(m) = nv
   470)   enddo
   471) 
   472)   end subroutine shr_strdata_init
   473) 
   474) !===============================================================================
   475) 
   476)   subroutine shr_strdata_advance(SDAT,ymd,tod,mpicom,istr,timers)
   477) 
   478)   implicit none
   479) 
   480)   type(shr_strdata_type),intent(inout) :: SDAT
   481)   integer(IN),intent(in) :: ymd    ! current model date
   482)   integer(IN),intent(in) :: tod    ! current model date
   483)   integer(IN),intent(in) :: mpicom
   484)   character(len=*),intent(in),optional :: istr
   485)   logical         ,intent(in),optional :: timers
   486) 
   487)   integer(IN) :: n,m,i,k,l,kf           ! generic index
   488)   integer(IN) :: my_task,npes
   489)   integer(IN),parameter :: master_task = 0
   490)   logical :: mssrmlf
   491)   logical,allocatable :: newData(:)
   492)   integer(IN) :: ierr
   493)   integer(IN) :: nu,nv
   494)   integer(IN) :: lsize,lsizeR,lsizeF
   495)   integer(IN),allocatable :: ymdmod(:)   ! modified model dates to handle Feb 29
   496)   integer(IN) :: todmod                  ! modified model dates to handle Feb 29
   497)   type(mct_avect) :: avRtmp
   498)   type(mct_avect) :: avRV,avFV
   499)   character(len=32) :: lstr
   500)   logical  :: ltimers
   501)   real(R8) :: flb,fub            ! factor for lb and ub
   502) 
   503)   !--- for cosz method ---
   504)   real(R8) :: calday             ! julian day of year
   505)   real(R8) :: declin             ! solar declination (radians)
   506)   real(R8) :: eccf               ! earth sun distance factor
   507)   real(R8),pointer :: lonr(:)    ! lon radians
   508)   real(R8),pointer :: latr(:)    ! lat radians
   509)   real(R8),pointer :: cosz(:)    ! cosz 
   510)   real(R8),pointer :: tavCosz(:) ! cosz, time avg over [LB,UB]
   511)   real(R8),pointer :: xlon(:),ylon(:)
   512)   real(R8),parameter :: solZenMin    = 0.001_R8 ! minimum solar zenith angle
   513) 
   514)   type(ESMF_Time) :: timeLB, timeUB  ! lb and ub times
   515)   type(ESMF_TimeInterval) :: timeint ! delta time
   516)   integer(IN)   :: dday          ! delta days
   517)   real(R8)      :: dtime         ! delta time
   518)   integer(IN)   :: uvar,vvar
   519)   logical       :: someNewData ! newData test
   520)   character(CS) :: uname       ! u vector field name
   521)   character(CS) :: vname       ! v vector field name
   522)   integer(IN)   :: year,month,day  ! date year month day
   523)   character(len=*),parameter :: timname = "_strd_adv"
   524)   integer(IN),parameter :: tadj = 2
   525) 
   526)    !----- formats -----
   527)   character(*),parameter :: subname = "(shr_strdata_advance) "
   528) 
   529) !-------------------------------------------------------------------------------
   530) !
   531) !-------------------------------------------------------------------------------
   532) 
   533)   if (SDAT%nstreams < 1) return
   534) 
   535)   lstr = ''
   536)   if (present(istr)) then
   537)     lstr = trim(istr)
   538)   endif
   539) 
   540)   ltimers = .true.
   541)   if (present(timers)) then
   542)     ltimers = timers
   543)   endif
   544) 
   545)   if (.not.ltimers) call t_adj_detailf(tadj)
   546) 
   547)   call MPI_COMM_SIZE(mpicom,npes,ierr)
   548)   call MPI_COMM_RANK(mpicom,my_task,ierr)
   549) 
   550)   mssrmlf = .false.
   551) 
   552)   SDAT%ymd = ymd
   553)   SDAT%tod = tod
   554) 
   555)   if (SDAT%nstreams > 0) then
   556)   allocate(newData(SDAT%nstreams))
   557)   allocate(ymdmod(SDAT%nstreams))
   558) 
   559)   do n = 1,SDAT%nstreams
   560)      ! ------------------------------------------------------- !
   561)      ! tcraig, Oct 11 2010.  Mismatching calendars: 4 cases    ! 
   562)      ! ------------------------------------------------------- !
   563)      ! ymdmod and todmod are the ymd and tod to time           !
   564)      ! interpolate to.  Generally, these are just the model    !
   565)      ! date and time.  Also, always use the stream calendar    !
   566)      ! for time interpolation for reasons described below.     !
   567)      ! When there is a calendar mismatch, support Feb 29 in a  !
   568)      ! special way as needed to get reasonable values.         !
   569)      ! Note that when Feb 29 needs to be treated specially,    !
   570)      ! a discontinuity will be introduced.  The size of that   !
   571)      ! discontinuity will depend on the time series input data.!
   572)      ! ------------------------------------------------------- !
   573)      ! (0) The stream calendar and model calendar are          !
   574)      ! identical.  Proceed in the standard way.                !
   575)      ! ------------------------------------------------------- !
   576)      ! (1) If the stream is a no leap calendar and the model   !
   577)      ! is gregorian, then time interpolate on the noleap       !
   578)      ! calendar.  Then if the model date is Feb 29, compute    !
   579)      ! stream data for Feb 28 by setting ymdmod and todmod to  !
   580)      ! Feb 28.  This results in duplicate stream data on       !
   581)      ! Feb 28 and Feb 29 and a discontinuity at the start of   !
   582)      ! Feb 29.                                                 !
   583)      ! This could be potentially updated by using the gregorian!
   584)      ! calendar for time interpolation when the input data is  !
   585)      ! relatively infrequent (say greater than daily) with the !
   586)      ! following concerns.
   587)      !   - The forcing will not be reproduced identically on   !
   588)      !     the same day with climatological inputs data        !
   589)      !   - Input data with variable input frequency might      !
   590)      !     behave funny
   591)      !   - An arbitrary discontinuity will be introduced in    !
   592)      !     the time interpolation method based upon the        !
   593)      !     logic chosen to transition from reproducing Feb 28  !
   594)      !     on Feb 29 and interpolating to Feb 29.              !
   595)      !   - The time gradient of data will change by adding a   !
   596)      !     day arbitrarily.
   597)      ! ------------------------------------------------------- !
   598)      ! (2) If the stream is a gregorian calendar and the model !
   599)      ! is a noleap calendar, then just time interpolate on the !
   600)      ! gregorian calendar.  The causes Feb 29 stream data      !
   601)      ! to be skipped and will lead to a discontinuity at the   !
   602)      ! start of March 1.                                       !
   603)      ! ------------------------------------------------------- !
   604)      ! (3) If the calendars mismatch and neither of the three  !
   605)      ! cases above are recognized, then abort.                 !
   606)      ! ------------------------------------------------------- !
   607) 
   608)      ! case(0)
   609)      ymdmod(n) = ymd
   610)      todmod    = tod
   611)      if (trim(SDAT%calendar) /= trim(SDAT%stream(n)%calendar)) then
   612)         if ((trim(SDAT%calendar) == trim(shr_cal_gregorian)) .and. &
   613)             (trim(SDAT%stream(n)%calendar) == trim(shr_cal_noleap))) then
   614)             ! case (1), set feb 29 = feb 28
   615)             call shr_cal_date2ymd (ymd,year,month,day)
   616)             if (month == 2 .and. day == 29) then
   617)                call shr_cal_ymd2date(year,2,28,ymdmod(n))
   618)             endif
   619)         else if ((trim(SDAT%calendar) == trim(shr_cal_noleap)) .and. &
   620)             (trim(SDAT%stream(n)%calendar) == trim(shr_cal_gregorian))) then
   621)             ! case (2), feb 29 input data will be skipped automatically
   622)         else
   623)             ! case (3), abort
   624)             write(logunit,*) trim(subname),' ERROR: mismatch calendar ', &
   625)                trim(SDAT%calendar),':',trim(SDAT%stream(n)%calendar)
   626)             call shr_sys_abort(trim(subname)//' ERROR: mismatch calendar ')
   627)         endif
   628)      endif
   629) 
   630)      call t_barrierf(trim(lstr)//trim(timname)//'_readLBUB_BARRIER',mpicom)
   631)      call t_startf(trim(lstr)//trim(timname)//'_readLBUB')
   632) 
   633)      call shr_dmodel_readLBUB(SDAT%stream(n),SDAT%pio_subsystem,SDAT%io_type,SDAT%pio_iodesc(n), &
   634)           ymdmod(n),todmod,mpicom,SDAT%gsmapR(n),&
   635)           SDAT%avRLB(n),SDAT%ymdLB(n),SDAT%todLB(n), &
   636)           SDAT%avRUB(n),SDAT%ymdUB(n),SDAT%todUB(n),newData(n), &
   637)           istr=trim(lstr)//'_readLBUB')
   638)      if (newData(n)) then
   639)         call shr_cal_date2ymd(SDAT%ymdLB(n),year,month,day)
   640)         call shr_cal_timeSet(timeLB,SDAT%ymdLB(n),0,SDAT%stream(n)%calendar)
   641)         call shr_cal_timeSet(timeUB,SDAT%ymdUB(n),0,SDAT%stream(n)%calendar)
   642)         timeint = timeUB-timeLB
   643)         call ESMF_TimeIntervalGet(timeint,StartTimeIn=timeLB,d=dday)
   644)         dtime = abs(real(dday,R8) + real(SDAT%todUB(n)-SDAT%todLB(n),R8)/shr_const_cDay)
   645) 
   646)         SDAT%dtmin(n) = min(SDAT%dtmin(n),dtime)
   647)         SDAT%dtmax(n) = max(SDAT%dtmax(n),dtime)
   648)         if ((SDAT%dtmax(n)/SDAT%dtmin(n)) > SDAT%dtlimit(n)) then
   649)            write(logunit,*) trim(subName),' ERROR: dt limit1 ',SDAT%dtmax(n),SDAT%dtmin(n),SDAT%dtlimit(n)
   650)            write(logunit,*) trim(subName),' ERROR: dt limit2 ',SDAT%ymdLB(n),SDAT%todLB(n), &
   651)                                                                SDAT%ymdUB(n),SDAT%todUB(n)
   652)            call shr_sys_abort(trim(subName)//' ERROR dt limit')
   653)         endif
   654)      endif
   655)      call t_stopf(trim(lstr)//trim(timname)//'_readLBUB')
   656)   enddo
   657) 
   658)   do n = 1,SDAT%nstreams
   659)      if (newData(n)) then
   660) 
   661)         if (SDAT%doFill(n)) then
   662)            call t_startf(trim(lstr)//trim(timname)//'_fill')
   663)            lsize = mct_aVect_lsize(SDAT%avRLB(n))
   664)            call mct_aVect_init(avRtmp,SDAT%avRLB(n),lsize)
   665)            call mct_aVect_copy(SDAT%avRLB(n),avRtmp)
   666)            call mct_sMat_avMult(avRtmp,SDAT%sMatPf(n),SDAT%avRLB(n))
   667)            call mct_aVect_copy(SDAT%avRUB(n),avRtmp)
   668)            call mct_sMat_avMult(avRtmp,SDAT%sMatPf(n),SDAT%avRUB(n))
   669)            call mct_aVect_clean(avRtmp)
   670)            call t_stopf(trim(lstr)//trim(timname)//'_fill')
   671)         endif
   672) 
   673)         if (SDAT%domaps(n)) then
   674)            call t_startf(trim(lstr)//trim(timname)//'_map')
   675)            call mct_sMat_avMult(SDAT%avRLB(n),SDAT%sMatPs(n),SDAT%avFLB(n))
   676)            call mct_sMat_avMult(SDAT%avRUB(n),SDAT%sMatPs(n),SDAT%avFUB(n))
   677)            call t_stopf(trim(lstr)//trim(timname)//'_map')
   678)         else
   679)            call t_startf(trim(lstr)//trim(timname)//'_rearr')
   680)            call mct_rearr_rearrange(SDAT%avRLB(n),SDAT%avFLB(n),SDAT%rearrR(n))
   681)            call mct_rearr_rearrange(SDAT%avRUB(n),SDAT%avFUB(n),SDAT%rearrR(n))
   682)            call t_stopf(trim(lstr)//trim(timname)//'_rearr')
   683)         endif
   684) 
   685)      endif
   686)   enddo
   687) 
   688)   do m = 1,SDAT%nvectors
   689)      nu = SDAT%ustrm(m)
   690)      nv = SDAT%vstrm(m)
   691)      if ((SDAT%domaps(nu) .or. SDAT%domaps(nv)) .and. &
   692)          (newdata(nu) .or. newdata(nv))) then
   693) 
   694)         call t_startf(trim(lstr)//trim(timname)//'_vect')
   695)         call shr_string_listGetName(SDAT%vectors(m),1,uname)
   696)         call shr_string_listGetName(SDAT%vectors(m),2,vname)
   697)         lsizeR = mct_aVect_lsize(SDAT%avRLB(nu))
   698)         lsizeF = mct_aVect_lsize(SDAT%avFLB(nu))
   699)         call mct_aVect_init(avRV,rlist=SDAT%vectors(m),lsize=lsizeR)
   700)         call mct_aVect_init(avFV,rlist=SDAT%vectors(m),lsize=lsizeF)
   701)         allocate(xlon(lsizeR))
   702)         allocate(ylon(lsizeF))
   703)         call mct_aVect_exportRattr(SDAT%gridR(nu)%data,'lon',xlon)
   704)         call mct_aVect_exportRattr(SDAT%grid     %data,'lon',ylon)
   705)         xlon = xlon * deg2rad
   706)         ylon = ylon * deg2rad
   707) 
   708)         !--- map LB ---
   709) 
   710)         uvar = mct_aVect_indexRA(SDAT%avRLB(nu),trim(uname))
   711)         vvar = mct_aVect_indexRA(SDAT%avRLB(nv),trim(vname))
   712)         do i = 1,lsizeR
   713)            avRV%rAttr(1,i) =  SDAT%avRLB(nu)%rAttr(uvar,i) * cos(xlon(i))  &
   714)                              -SDAT%avRLB(nv)%rAttr(vvar,i) * sin(xlon(i))
   715)            avRV%rAttr(2,i) =  SDAT%avRLB(nu)%rAttr(uvar,i) * sin(xlon(i))  &
   716)                              +SDAT%avRLB(nv)%rAttr(vvar,i) * cos(xlon(i))
   717)         enddo
   718)         call mct_sMat_avMult(avRV,SDAT%sMatPs(nu),avFV)
   719) ! ---   don't need to recompute uvar and vvar, should be the same
   720) !       uvar = mct_aVect_indexRA(SDAT%avFLB(nu),trim(uname))
   721) !       vvar = mct_aVect_indexRA(SDAT%avFLB(nv),trim(vname))
   722)         do i = 1,lsizeF
   723)            SDAT%avFLB(nu)%rAttr(uvar,i) =  avFV%rAttr(1,i) * cos(ylon(i))  &
   724)                                           +avFV%rAttr(2,i) * sin(ylon(i))
   725)            SDAT%avFLB(nv)%rAttr(vvar,i) = -avFV%rAttr(1,i) * sin(ylon(i))  &
   726)                                           +avFV%rAttr(2,i) * cos(ylon(i))
   727)         enddo
   728) 
   729)         !--- map UB ---
   730) 
   731)         uvar = mct_aVect_indexRA(SDAT%avRUB(nu),trim(uname))
   732)         vvar = mct_aVect_indexRA(SDAT%avRUB(nv),trim(vname))
   733)         do i = 1,lsizeR
   734)            avRV%rAttr(1,i) =  SDAT%avRUB(nu)%rAttr(uvar,i) * cos(xlon(i))  &
   735)                              -SDAT%avRUB(nv)%rAttr(vvar,i) * sin(xlon(i))
   736)            avRV%rAttr(2,i) =  SDAT%avRUB(nu)%rAttr(uvar,i) * sin(xlon(i))  &
   737)                              +SDAT%avRUB(nv)%rAttr(vvar,i) * cos(xlon(i))
   738)         enddo
   739)         call mct_sMat_avMult(avRV,SDAT%sMatPs(nu),avFV)
   740) ! ---   don't need to recompute uvar and vvar, should be the same
   741) !       uvar = mct_aVect_indexRA(SDAT%avFUB(nu),trim(uname))
   742) !       vvar = mct_aVect_indexRA(SDAT%avFUB(nv),trim(vname))
   743)         do i = 1,lsizeF
   744)            SDAT%avFUB(nu)%rAttr(uvar,i) =  avFV%rAttr(1,i) * cos(ylon(i))  &
   745)                                           +avFV%rAttr(2,i) * sin(ylon(i))
   746)            SDAT%avFUB(nv)%rAttr(vvar,i) = -avFV%rAttr(1,i) * sin(ylon(i))  &
   747)                                           +avFV%rAttr(2,i) * cos(ylon(i))
   748)         enddo
   749) 
   750)         call mct_aVect_clean(avRV)
   751)         call mct_aVect_clean(avFV)
   752)         deallocate(xlon,ylon)
   753) 
   754)         call t_stopf(trim(lstr)//trim(timname)//'_vect')
   755)      endif
   756)   enddo
   757) 
   758)   do n = 1,SDAT%nstreams
   759) 
   760)      !--- method: coszen -------------------------------------------------------
   761)      if (trim(SDAT%tintalgo(n)) == 'coszen') then
   762)         call t_startf(trim(lstr)//trim(timname)//'_coszen')
   763) 
   764)         !--- make sure orb info has been set ---
   765)         if (SDAT%eccen == SHR_ORB_UNDEF_REAL) then
   766)            call shr_sys_abort(subname//' ERROR in orb params for coszen tinterp')
   767)         else if (SDAT%modeldt < 1) then
   768)            call shr_sys_abort(subname//' ERROR: model dt < 1 for coszen tinterp')
   769)         endif
   770) 
   771)         !--- allocate avg cosz array ---
   772)         lsizeF = mct_aVect_lsize(SDAT%avFLB(n))
   773)         allocate(tavCosz(lsizeF),cosz(lsizeF),lonr(lsizeF),latr(lsizeF))
   774) 
   775)         !--- get lat/lon data ---
   776)         kf = mct_aVect_indexRA(SDAT%grid%data,'lat')
   777)         latr(:) = SDAT%grid%data%rAttr(kf,:) * deg2rad
   778)         kf = mct_aVect_indexRA(SDAT%grid%data,'lon')
   779)         lonr(:) = SDAT%grid%data%rAttr(kf,:) * deg2rad
   780) 
   781)         call t_startf(trim(lstr)//trim(timname)//'_coszenC')
   782)         cosz = 0.0_r8
   783)         call shr_tInterp_getCosz(cosz,lonr,latr,ymdmod(n),todmod, &
   784)              SDAT%eccen,SDAT%mvelpp,SDAT%lambm0,SDAT%obliqr,SDAT%stream(n)%calendar)
   785)         call t_stopf(trim(lstr)//trim(timname)//'_coszenC')
   786) 
   787)         if (newdata(n)) then
   788)            !--- compute a new avg cosz ---
   789)            call t_startf(trim(lstr)//trim(timname)//'_coszenN')
   790)            call shr_tInterp_getAvgCosz(tavCosz,lonr,latr, &
   791)                 SDAT%ymdLB(n),SDAT%todLB(n), SDAT%ymdUB(n),SDAT%todUB(n), &
   792)                 SDAT%eccen,SDAT%mvelpp,SDAT%lambm0,SDAT%obliqr,SDAT%modeldt,&
   793) 	        SDAT%stream(n)%calendar) 
   794)            call mct_avect_importRAttr(SDAT%avCoszen(n),'tavCosz',tavCosz,lsizeF)
   795)            call t_stopf(trim(lstr)//trim(timname)//'_coszenN')
   796)         else
   797)            !--- reuse existing avg cosz ---
   798)            call mct_avect_exportRAttr(SDAT%avCoszen(n),'tavCosz',tavCosz)
   799)         endif
   800) 
   801)         !--- t-interp is LB data normalized with this factor: cosz/tavCosz ---
   802)         do i = 1,lsizeF
   803)            if (cosz(i) > solZenMin) then
   804)               SDAT%avs(n)%rAttr(:,i) = SDAT%avFLB(n)%rAttr(:,i)*cosz(i)/tavCosz(i) 
   805)            else
   806)               SDAT%avs(n)%rAttr(:,i) =  0._r8
   807)            endif
   808)         enddo
   809)         deallocate(tavCosz,cosz,lonr,latr)
   810)         call t_stopf(trim(lstr)//trim(timname)//'_coszen')
   811) 
   812)      !--- method: not coszen ---------------------------------------------------
   813)      elseif (trim(SDAT%tintalgo(n)) /= trim(shr_strdata_nullstr)) then
   814) 
   815)         call t_startf(trim(lstr)//trim(timname)//'_tint')
   816)         call shr_tInterp_getFactors(SDAT%ymdlb(n),SDAT%todlb(n),SDAT%ymdub(n),SDAT%todub(n), &
   817)                                     ymdmod(n),todmod,flb,fub, &
   818)                                     calendar=SDAT%stream(n)%calendar,algo=trim(SDAT%tintalgo(n)))
   819)         SDAT%avs(n)%rAttr(:,:) = SDAT%avFLB(n)%rAttr(:,:)*flb + SDAT%avFUB(n)%rAttr(:,:)*fub
   820)         call t_stopf(trim(lstr)//trim(timname)//'_tint')
   821) 
   822)      else
   823)         call t_startf(trim(lstr)//trim(timname)//'_zero')
   824)         call mct_avect_zero(SDAT%avs(n))
   825)         call t_stopf(trim(lstr)//trim(timname)//'_zero')
   826)      endif
   827)   enddo
   828)
