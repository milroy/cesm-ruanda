   304)   subroutine get_gcol_block_d(gcol,cnt,blockid,bcid,localblockid)
   305)     use dimensions_mod, only : nelemd, nelem
   306)     use kinds, only: int_kind
   307)     use abortutils, only: endrun
   308)     !----------------------------------------------------------------------- 
   309)     ! 
   310)     !                          
   311)     ! Purpose: Return global block index and local column index
   312)     !          for global column index
   313)     !           element array is naturally ordered
   314)     ! 
   315)     ! Method: 
   316)     ! 
   317)     ! Author: Jim Edwards
   318)     ! Modified: replaced linear search with best guess and binary search   
   319)     !           (Pat Worley, 7/2/09)
   320)     ! 
   321)     !-----------------------------------------------------------------------
   322)     implicit none
   323)     !------------------------------Arguments--------------------------------
   324)     integer, intent(in) :: gcol     ! global column index
   325)     integer, intent(in) :: cnt      ! size of blockid and bcid arrays
   326) 
   327)     integer, intent(out) :: blockid(cnt) ! block index
   328)     integer, intent(out) :: bcid(cnt)    ! column index within block
   329)     integer, intent(out), optional :: localblockid(cnt)
   330)     integer :: sb, eb, ie, high, low
   331)     logical :: found
   332)     integer, save :: iedex_save = 1
   333)     !---------------------------Local workspace-----------------------------
   334)     !
   335)     if(gblocks_need_initialized) call gblocks_init()
   336)     found = .false.
   337)     low = 1
   338)     high = nelem
   339) 
   340)     ! check whether previous found element is the same here
   341)     if (.not. found) then
   342)        ie = iedex_save
   343)        sb = gblocks(ie)%UniquePtOffset
   344)        if (gcol >= sb) then
   345)           eb = sb+gblocks(ie)%NumUniqueP
   346)           if (gcol < eb) then
   347)              found = .true.
   348)           else
   349)              low = ie
   350)           endif
   351)        else
   352)           high = ie
   353)        endif
   354)     endif
   355) 
   356)     ! check whether next element  is the one wanted
   357)     if ((.not. found) .and. &
   358)         ((low .eq. iedex_save) .or. (iedex_save .eq. nelem))) then
   359)        ie = iedex_save + 1
   360)        if (ie > nelem) ie = 1
   361) 
   362)        sb = gblocks(ie)%UniquePtOffset
   363)        if (gcol >= sb) then
   364)           eb = sb+gblocks(ie)%NumUniqueP
   365)           if (gcol < eb) then
   366)              found = .true.
   367)           else
   368)              low = ie
   369)           endif
   370)        else
   371)           high = ie
   372)        endif
   373)     endif
   374) 
   375)     ! otherwise, use a binary search to find element
   376)     if (.not. found) then
   377)        ! (start with a sanity check)
   378)        ie = low
   379)        sb = gblocks(ie)%UniquePtOffset
   380) 
   381)        ie = high
   382)        eb = gblocks(ie)%UniquePtOffset + gblocks(ie)%NumUniqueP
   383) 
   384)        if ((gcol < sb) .or.  (gcol >= eb)) then
   385)           do ie=1,nelemd
   386)              write(iulog,*) __LINE__,ie,elem(ie)%idxP%UniquePtOffset,elem(ie)%idxP%NumUniquePts
   387)           end do
   388)           call endrun()
   389)        end if
   390) 
   391)        do while (.not. found)
   392)           ie = low + (high-low)/2;
   393) 
   394)           sb = gblocks(ie)%UniquePtOffset
   395)           if (gcol >= sb) then
   396)              eb = sb+gblocks(ie)%NumUniqueP
   397)              if (gcol < eb) then
   398)                 found = .true.
   399)              else
   400)                 low = ie+1
   401)              endif
   402)           else
   403)              high = ie-1
   404)           endif
   405)        enddo
   406)     endif
   407) 
   408)     blockid(1)=ie
   409)     if(present(localblockid)) localblockid(1)=gblocks(ie)%LocalID
   410)     bcid(1)=gcol-sb+1
   411)     iedex_save = ie
   412) 
   413)  return
   414) end subroutine get_gcol_block_d

   524) subroutine get_horiz_grid_d(nxy,clat_d_out,clon_d_out,area_d_out, &
   525)     wght_d_out)
   526) 
   527)  use shr_kind_mod, only: r8 => shr_kind_r8
   528)  use physconst, only : pi
   529)  use dof_mod, only : UniqueCoords, UniquePoints
   530)  use dimensions_mod, only: nelem, nelemd, nelemdmax, np
   531)  use spmd_utils, only : iam, npes, mpi_integer, mpir8, mpicom
   532) 
   533)  use abortutils, only: endrun
   534)  !----------------------------------------------------------------------- 
   535)  ! 
   536)  !                          
   537)  ! Purpose: Return latitude and longitude (in radians), column surface
   538)  !          area (in radians squared) and surface integration weights
   539)  !          for global column indices that will be passed to/from physics
   540)  ! 
   541)  ! Method: 
   542)  ! 
   543)  ! Author: Jim Edwards
   544)  ! 
   545)  !-----------------------------------------------------------------------
   546)  integer, intent(in)   :: nxy                     ! array sizes
   547) 
   548)  real(r8), intent(out), optional :: clat_d_out(:) ! column latitudes
   549)  real(r8), intent(out), optional :: clon_d_out(:) ! column longitudes
   550)  real(r8), intent(out), target, optional :: area_d_out(:) ! column surface 
   551)  !  area
   552)  real(r8), intent(out), target, optional :: wght_d_out(:) ! column integration
   553)  !  weight
   554) 
   555)  integer :: ie, i, sb, eb, p, j
   556) 
   557)  real (r8), dimension(np,np) :: areaw
   558)  real(r8), pointer :: area_d(:)
   559)  real(r8) :: scale_factor
   560)  integer :: ierr
   561)  integer :: rdispls(npes), recvcounts(npes), gid(npes)
   562)  integer :: ibuf
   563)  real(r8) :: rbuf(nxy)
   564)  logical :: compute_area 
   565) 
   566)  if (nxy < ngcols_d) then
   567)     write(iulog,*)'GET_HORIZ_GRID_D: arrays not large enough (', &
   568)          nxy,' < ',ngcols_d,' ) '
   569)     call endrun
   570)  endif
   571) 
   572)  compute_area = .false.
   573)  if ( present(area_d_out) ) then
   574)     if (size(area_d_out) .ne. nxy) then
   575)        call endrun('bad area_d_out array size in dyn_grid')
   576)     end if
   577)     area_d => area_d_out
   578)     compute_area = .true.
   579)  else if ( present(wght_d_out) ) then
   580)     if (size(wght_d_out) .ne. nxy) then
   581)        call endrun('bad wght_d_out array size in dyn_grid')
   582)     end if
   583)     area_d => wght_d_out
   584)     compute_area = .true.
   585)  endif
   586) 
   587)  if(gblocks_need_initialized) call gblocks_init()
   588)  ! could improve this
   589)  if(.not. associated(clat)) then	
   590)     allocate(clat(nxy), clon(nxy,1))
   591)  end if
   592) 
   593)  clon(:,1) = -iam 
   594)  do ie=1,nelemdmax        
   595)     if(ie<=nelemd) then
   596)        rdispls(iam+1)=elem(ie)%idxp%UniquePtOffset-1
   597)        eb = rdispls(iam+1) + elem(ie)%idxp%NumUniquePts
   598)        recvcounts(iam+1)=elem(ie)%idxP%NumUniquePts
   599)        call UniqueCoords(elem(ie)%idxP, elem(ie)%spherep, &
   600)             clat(rdispls(iam+1)+1:eb), &
   601)             clon(rdispls(iam+1)+1:eb,1)) 
   602) 
   603)        if (compute_area) then
   604)           areaw = 1.0_r8/elem(ie)%rspheremp(:,:)         
   605)           call UniquePoints(elem(ie)%idxP, areaw, area_d(rdispls(iam+1)+1:eb))
   606)        endif
   607)     else     
   608)        rdispls(iam+1) = 0
   609)        recvcounts(iam+1) = 0
   610)     endif
   611)     ibuf=rdispls(iam+1)
   612)     call mpi_allgather(ibuf, 1, mpi_integer, rdispls, &
   613)          1, mpi_integer, mpicom, ierr)
   614) 
   615)     ibuf=recvcounts(iam+1)
   616)     call mpi_allgather(ibuf, 1, mpi_integer, recvcounts, &
   617)          1, mpi_integer, mpicom, ierr)
   618) 
   619)     sb=rdispls(iam+1)+1
   620)     eb =rdispls(iam+1) + recvcounts(iam+1)
   621) 
   622)     rbuf(1:recvcounts(iam+1)) = clat(sb:eb)  ! whats going to happen if end=0?

   629) 
   630)     if (compute_area) then
   631)        rbuf(1:recvcounts(iam+1)) = area_d(sb:eb)
   632)        call mpi_allgatherv(rbuf,recvcounts(iam+1),mpir8, area_d, &
   633)          recvcounts(:), rdispls(:), mpir8, mpicom, ierr)
   634)     endif
   635)  end do
   636)  call mpi_barrier(mpicom,ierr)
   637) 
   638)  if ( present(clat_d_out) ) then
   639)     if (size(clat_d_out) .ne. nxy) then
   640)        call endrun('bad clat_d_out array size in dyn_grid')
   641)     end if
   642)     clat_d_out(:) = clat(:)
   643)  endif

   647)        call endrun('bad clon_d_out array size in dyn_grid')
   648)     end if
   649)     clon_d_out(:) = clon(:,1)
   650)  endif
   651) 
   652)  !if one of area_d_out  or wght_d_out was present, than it was computed
   653)  !above.  if they were *both* present, then do this: 
   654)  if ( present(area_d_out) .and. present(wght_d_out) ) then
   655)     wght_d_out(:) = area_d_out(:)
   656)  endif

   659)  return

   866) ! This returns the lat/lon information (and corresponding MPI task numbers (owners)) 
   867) ! of the global model grid columns nearest to the input satellite coordinate (lat,lon)
   868) !-------------------------------------------------------------------------------
   869) subroutine dyn_grid_find_gcols( lat, lon, nclosest, owners, col, lbk, rlat, rlon, idyn_dists ) 
   870)   use spmd_utils,     only: masterproc, iam
   871)   use dimensions_mod, only: np
   872)   use abortutils,     only: endrun
   873)   use shr_const_mod,  only: SHR_CONST_PI, SHR_CONST_REARTH
   874) 
   875)   real(r8), intent(in) :: lat
   876)   real(r8), intent(in) :: lon
   877)   integer, intent(in)  :: nclosest
   878)   integer, intent(out) :: owners(nclosest)
   879)   integer, intent(out) :: col(nclosest)
   880)   integer, intent(out) :: lbk(nclosest)
   881) 
   882)   real(r8),optional, intent(out) :: rlon(nclosest)
   883)   real(r8),optional, intent(out) :: rlat(nclosest)
   884)   real(r8),optional, intent(out) :: idyn_dists(nclosest)
   885) 
   886)   real(r8) :: dist            ! the distance (in radians**2 from lat, lon)
   887)   real(r8) :: latr, lonr      ! lat, lon inputs converted to radians
   888)   integer  :: blockid(1), bcid(1), lclblockid(1)
   889)   integer  :: i,j,k, ii
   890)   real(r8), allocatable :: clat_d(:), clon_d(:), distmin(:)
   891)   integer, allocatable :: igcol(:)
   892)   real(r8), parameter :: rad2deg = 180._r8/SHR_CONST_PI
   893) 
   894)   latr = lat/rad2deg
   895)   lonr = lon/rad2deg
   896) 
   897)   allocate( clat_d(1:ngcols_d) )
   898)   allocate( clon_d(1:ngcols_d) )
   899)   allocate( igcol(nclosest) )
   900)   allocate( distmin(nclosest) )
   901) 
   902)   call get_horiz_grid_d(ngcols_d, clat_d_out=clat_d, clon_d_out=clon_d)
   903)   
   904)   igcol(:)    = -999
   905)   distmin(:) = 1.e10_r8
   906) 
   907)   do i = 1,ngcols_d
   908)      
   909)      ! Use the Spherical Law of Cosines to find the great-circle distance.
   910)      dist = acos(sin(latr) * sin(clat_d(i)) + cos(latr) * cos(clat_d(i)) * cos(clon_d(i) - lonr)) * SHR_CONST_REARTH
   911)      do j = nclosest, 1, -1
   912)         if (dist < distmin(j)) then
   913)            
   914)            if (j < nclosest) then
   915)               distmin(j+1) = distmin(j)
   916)               igcol(j+1)    = igcol(j)
   917)            end if
   918)            
   919)            distmin(j) = dist
   920)            igcol(j)    = i
   921)         else
   922)            exit
   923)         end if
   924)      enddo
   925)      
   926)   enddo
   927) 
   928)   do i = 1,nclosest
   929) 
   930)      call  get_gcol_block_d( igcol(i), 1, blockid, bcid, lclblockid )
   931)      owners(i) = get_block_owner_d(blockid(1))
   932)      
   933)      if (owners(i)==iam) then
   934)         lbk(i) = lclblockid(1)
   935)         ii = igcol(i)-elem(lbk(i))%idxp%UniquePtoffset+1
   936)         k=elem(lbk(i))%idxp%ia(ii)
   937)         j=elem(lbk(i))%idxp%ja(ii)
   938)         col(i) = k+(j-1)*np
   939)      else
   940)         lbk(i) = -1
   941)         col(i) = -1
   942)      endif
   943)      if ( present(rlat) ) rlat(i) = clat_d(igcol(i)) * rad2deg
   944)      if ( present(rlon) ) rlon(i) = clon_d(igcol(i)) * rad2deg
   945)      
   946)      if (present(idyn_dists)) then
   947)         idyn_dists(i) = distmin(i)
   948)      end if
   949)      
   950)   end do
   951) 
   952)   deallocate( clat_d )
   953)   deallocate( clon_d )
   954)   deallocate( igcol )
   955)   deallocate( distmin )
   956) 
   957)  endsubroutine dyn_grid_find_gcols
