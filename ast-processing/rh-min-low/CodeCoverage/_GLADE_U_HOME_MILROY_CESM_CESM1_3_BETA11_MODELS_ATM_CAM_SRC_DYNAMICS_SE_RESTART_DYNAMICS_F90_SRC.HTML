    33)          pio_put_att, pio_def_var, pio_initdecomp, &
    34)          pio_setdebuglevel, pio_inq_dimid
    35)     use cam_pio_utils, only : pio_subsystem
    36)     use dimensions_mod, only : np, ne, nlev, qsize_d, nlevp, nelem , nelemd

    93)   subroutine write_restart_dynamics(File, dyn_out)
    94)     use pio, only : pio_offset, io_desc_t, pio_double, pio_write_darray, &
    95)          pio_setframe, pio_put_var, pio_initdecomp, pio_setframe, pio_def_dim, &
    96)          pio_freedecomp, pio_enddef
    97)     use cam_pio_utils, only : pio_subsystem
    98)     use dyn_comp, only : timelevel
    99)     use control_mod, only: qsplit
   100)     use time_mod, only : tstep, TimeLevel_Qdp
   101)     use element_mod, only : element_t
   102)     use fvm_control_volume_mod, only : fvm_struct
   103)     use dimensions_mod, only : nlev, qsize_d, nlevp, np, ne, nelemd, nelem
   104)     use dof_mod, only : UniquePoints
   105)     use time_manager, only: get_curr_time
   106)     use parallel_mod, only: par
   107)     use hycoef, only: write_restart_hycoef
   108) 
   109) 
   110)     implicit none
   111)     type(file_desc_t) :: File
   112)     type(dyn_export_t), intent(in)  :: dyn_out
   113) 
   114)     type(io_desc_t) :: iodesc2d, iodesc3d
   115)     integer :: st, ie, k, en, tl, tlQdp, ierr, vsize3d, vsize2d, q
   116)     integer(kind=PIO_OFFSET), parameter :: t = 1
   117) 
   118)     real(kind=r8),pointer :: vartmp(:,:,:), var3d(:,:,:,:), var2d(:,:,:)
   119)     integer :: i, j
   120) 
   121)     type(element_t), pointer :: elem(:)
   122)     type(fvm_struct), pointer :: fvm(:)
   123)     real(kind=r8) :: time
   124)     integer :: ndcur, nscur
   125)     integer, pointer :: ldof(:)
   126) 
   127)     call write_restart_hycoef(File)
   128) 
   129)    if (iam .lt. par%nprocs) then
   130)     elem => dyn_out%elem
   131)     fvm => dyn_out%fvm
   132)    else
   133)     allocate (elem(0), fvm(0))
   134)    endif
   135) 
   136)     ldof => get_restart_decomp(elem, 1)
   137)     call PIO_InitDecomp(pio_subsystem, pio_double, (/nelem*np*np/) , ldof , iodesc2d)
   138)     deallocate(ldof)
   139) 
   140)     ldof => get_restart_decomp(elem, nlev)       
   141)     call PIO_InitDecomp(pio_subsystem, pio_double, (/nelem*np*np,nlev/),ldof , iodesc3d)
   142)     deallocate(ldof)
   143)     initialized = .true.
   144) 
   145) 
   146)     call get_curr_time(ndcur, nscur)
   147) 
   148)     allocate(vartmp(np,np,nlev))
   149)     en=0
   150)     do ie=1,nelemd
   151)        en=en+elem(ie)%idxp%NumUniquePts
   152)     end do
   153) 
   154)     allocate(var2d(np,np,nelemd ))
   155)     vsize2d=np*np*nelemd
   156) 
   157)     allocate(var3d(np,np,nelemd,nlev))
   158)     vsize3d=vsize2d*nlev
   159) 
   160) !$omp parallel do private(ie, j, i)
   161)     do ie=1,nelemd
   162)        do j=1,np
   163)           do i=1,np
   164)              var2d(i,j,ie) = elem(ie)%state%phis(i,j)
   165)           end do
   166)        end do
   167)     end do
   168)     call PIO_Write_Darray(File,Phisdesc,iodesc2d, var2d,ierr)
   169) 
   170) !$omp parallel do private(ie, k, j, i)
   171)     do ie=1,nelemd
   172)        do k=1,nlev
   173)           do j=1,np
   174)              do i=1,np
   175)                 var3d(i,j,ie,k) = elem(ie)%derived%omega_p(i,j,k)
   176)              end do
   177)           end do
   178)        end do
   179)     end do
   180)     call PIO_Write_Darray(File,omegadesc,iodesc3d, var3d,ierr)
   181) 
   182) 
   183)     tl = timelevel%n0
   184)     call TimeLevel_Qdp(timelevel, qsplit, tlQdp)
   185)     time = ndcur+real(nscur,kind=r8)/86400_r8
   186) 
   187)     ierr = pio_put_var(File,timedesc%varid, (/int(t)/), time)
   188) 
   189) !$omp parallel do private(ie, j, i)
   190)     do ie=1,nelemd
   191)        do j=1,np
   192)           do i=1,np
   193)              var2d(i,j,ie) = elem(ie)%state%ps_v(i,j,tl)
   194)           end do
   195)        end do
   196)     end do
   197)     call PIO_SetFrame(PSdesc, t)
   198)     call PIO_Write_Darray(File,PSdesc,iodesc2d, var2d,ierr)
   199) 
   200)     ! Write the U component of Velocity
   201) !$omp parallel do private(ie, k, j, i)
   202)     do ie=1,nelemd
   203)        do k=1,nlev
   204)           do j=1,np
   205)              do i=1,np
   206)                 var3d(i,j,ie,k) = elem(ie)%state%V(i,j,1,k,tl)
   207)              end do
   208)           end do
   209)        end do
   210)     end do
   211) 
   212) 
   213)     call PIO_SetFrame(Udesc, t)
   214)     call PIO_Write_Darray(File,Udesc,iodesc3d,var3d,ierr)
   215) 
   216) 
   217)        ! Write the V component of Velocity
   218) !$omp parallel do private(ie, k, j, i)
   219)     do ie=1,nelemd
   220)        do k=1,nlev
   221)           do j=1,np
   222)              do i=1,np
   223)                 var3d(i,j,ie,k) = elem(ie)%state%V(i,j,2,k,tl)
   224)              end do
   225)           end do
   226)        end do
   227)     end do
   228) 
   229)     call PIO_SetFrame(Vdesc, t)
   230)     call PIO_Write_Darray(File,Vdesc,iodesc3d,var3d,ierr)
   231) 
   232)     ! Write T
   233) !$omp parallel do private(ie, k, j, i)
   234)     do ie=1,nelemd
   235)        do k=1,nlev
   236)           do j=1,np
   237)              do i=1,np
   238)                 var3d(i,j,ie,k) = elem(ie)%state%T(i,j,k,tl)
   239)              end do
   240)           end do
   241)        end do
   242)     end do
   243) 
   244)     call PIO_SetFrame(Tdesc, t)
   245)     call PIO_Write_Darray(File,Tdesc,iodesc3d,var3d,ierr)
   246) 
   247) 
   248)     do q=1,qsize_d
   249) 
   250)        ! Write Q
   251) !$omp parallel do private(ie, k, j, i)
   252)        do ie=1,nelemd
   253)           do k=1,nlev
   254)              do j=1,np
   255)                 do i=1,np
   256)                    var3d(i,j,ie,k) = elem(ie)%state%Q(i,j,k,q)
   257)                 end do
   258)              end do
   259)           end do
   260)        end do
   261)        call PIO_SetFrame(Qdesc(q), t)
   262)        call PIO_Write_Darray(File,Qdesc(q),iodesc3d,var3d,ierr)
   263) 
   264)        ! Write Q
   265) !$omp parallel do private(ie, k, j, i)
   266)        do ie=1,nelemd
   267)           do k=1,nlev
   268)              do j=1,np
   269)                 do i=1,np
   270)                    var3d(i,j,ie,k) = elem(ie)%state%Qdp(i,j,k,q,tlQdp)
   271)                 end do
   272)              end do
   273)           end do
   274)        end do
   275)        call PIO_SetFrame(Qdesc_dp(q), t)
   276)        call PIO_Write_Darray(File,Qdesc_dp(q),iodesc3d,var3d,ierr)
   277) 
   278)     end do
   279) 
   280) 
   281)     deallocate(var3d)
   282)     deallocate(var2d)
   283)     deallocate(qdesc, qdesc_dp)
   284)     call pio_freedecomp(File, iodesc2d)
   285)     call pio_freedecomp(File, iodesc3d)
   286) 
   287)    if (iam .lt. par%nprocs) then
   288)    else
   289)     deallocate(elem, fvm)
   290)    endif  !!  iam .lt. par%nprocs
   291) 
   292)   end subroutine write_restart_dynamics

   294)   !
   295)   ! Get the integer mapping of a variable in the dynamics decomp in memory.  
   296)   ! The canonical ordering is as on the file. A 0 value indicates that the
   297)   ! variable is not on the file (eg halo or boundary values)
   298)   !
   299) 
   300)   function get_restart_decomp(elem, lev) result(ldof)
   301)     use element_mod, only : element_t
   302)     use dimensions_mod, only : np, nelemd, nelem
   303)     type(element_t), intent(in) :: elem(:)
   304)     integer, pointer :: ldof(:)
   305)     integer, intent(in) :: lev
   306) 
   307)     integer ::  i, j, k, ie
   308) 
   309)     j=1
   310) 
   311)     allocate(ldof(nelemd*np*np*lev))
   312)     do k=1,lev
   313)        do ie=1,nelemd
   314)           do i=1,np*np
   315)              ldof(j) = (elem(ie)%GlobalID-1)*np*np+(k-1)*nelem*np*np+i
   316)              j=j+1
   317)           end do
   318)        end do
   319)     end do
   320) 
   321) 
   322)   end function get_restart_decomp

   325)   subroutine read_restart_dynamics (File, dyn_in, dyn_out, NLFileName)
   326)     ! for restart and initial condition, timelevel == timelevel_dyn
   327)     ! so we wont update this routine to use both  
   328)     use dyn_comp, only : timelevel
   329)     use parallel_mod, only : initmp, par
   330)     use element_mod, only : element_t
   331)     use pio, only : file_desc_t, pio_global, pio_double, pio_offset, &
   332)          pio_get_att, pio_inq_dimid, pio_inq_dimlen, pio_initdecomp, pio_inq_varid, &
   333)          pio_read_darray, pio_setframe, file_desc_t, io_desc_t, pio_double
   334)     use dyn_comp, only : dyn_init1, dyn_init2
   335)     use dimensions_mod, only : nlev, np, ne, nelemd, qsize_d
   336)     use abortutils,   only: endrun
   337)     use namelist_mod, only: readnl
   338)     use constituents, only : cnst_name
   339)     use bndry_mod,   only: bndry_exchangeV
   340)     use edge_mod,       only: EdgeBuffer_t, initEdgeBuffer, FreeEdgeBuffer, &
   341)          edgeVpack, edgeVunpack
   342)     use cam_pio_utils, only : pio_subsystem
   343)     use spmd_dyn, only: spmd_readnl
   344)     use fvm_control_volume_mod, only: fvm_struct
   345)     use control_mod, only: qsplit
   346)     use time_mod, only: TimeLevel_Qdp
   347)     !
   348)     ! Input arguments
   349)     !
   350)     type(File_desc_t), intent(inout) :: File
   351)     type(dyn_import_t), intent(inout)  :: dyn_in
   352)     type(dyn_export_t), intent(inout)  :: dyn_out
   353)     character(len=*), intent(in) :: NLFileName
   354) 
   355)     type(io_desc_t) :: iodesc2d, iodesc3d
   356)     real(r8), allocatable :: var3d(:), var2d(:)
   357)     integer :: ie, ierr, fne, fnp, fnlev
   358)     integer :: ncols
   359)     integer, pointer :: ldof(:)
   360)     type(element_t), pointer :: elem(:)               ! pointer to dyn_in element array
   361)     type(fvm_struct), pointer :: fvm(:)
   362)     integer(kind=PIO_OFFSET), parameter :: t = 1
   363)     integer :: i, k, cnt, st, en, tl, tlQdp, ii, jj, s2d, q, j
   364)     type(EdgeBuffer_T) :: edgebuf
   365)     integer :: timelevel_dimid, timelevel_chk
   366)     integer :: npes_se
   367) !    type(file_desc_t) :: ncid
   368) !    integer :: ncid
   369) 
   370)     call dyn_init1(file, NLFileName, dyn_in, dyn_out)
   371) 
   372)    if (iam .lt. par%nprocs) then
   373)     elem=>dyn_in%elem
   374)    else
   375)     allocate (elem(0))
   376)    endif
   377) 
   378)     ierr = PIO_Get_Att(File, PIO_GLOBAL, 'ne', fne)
   379)     ierr = PIO_Get_Att(File, PIO_GLOBAL, 'np', fnp)
   380) 
   381)     if(ne/=fne .or. np/=fnp) then
   382)        write(iulog,*) 'Restart file np or ne does not match model. np (file, model):',fnp,np,&
   383)             ' ne (file, model) ', fne, ne
   384)        call endrun()
   385)     end if
   386) 
   387)     ierr = PIO_Inq_DimID(File, 'lev', nlev_dimid)
   388)     ierr = PIO_Inq_dimlen(File, nlev_dimid, fnlev)
   389)     if(ne/=fne .or. np/=fnp) then
   390)        write(iulog,*) 'Restart file nlev does not match model. nlev (file, namelist):',fnlev, nlev
   391)        call endrun()
   392)     end if
   393) 
   394)     ierr = PIO_Inq_DimID(File, 'timelevels', timelevel_dimid)
   395)     ierr = PIO_Inq_dimlen(File, timelevel_dimid, timelevel_chk)
   396) 
   397) 
   398)     ierr = PIO_Inq_DimID(File, 'ncol_d', ncol_dimid)
   399) 
   400)     ierr = PIO_Inq_dimlen(File, ncol_dimid, ncols)
   401) 
   402) 
   403)     ldof => get_restart_decomp(elem, 1)
   404)     s2d=size(ldof)
   405)     allocate(var3d(s2d*nlev), var2d(s2d))
   406)     var2d = 0._r8
   407)     var3d = 0._r8
   408)     call PIO_InitDecomp(pio_subsystem, pio_double, (/ncols/) , ldof , iodesc2d)
   409)     deallocate(ldof)
   410) 
   411)     ldof => get_restart_decomp(elem, nlev)
   412)     call PIO_InitDecomp(pio_subsystem, pio_double, (/ncols,nlev/),ldof , iodesc3d)
   413)     deallocate(ldof)
   414) 
   415)     initialized = .true.
   416) 
   417)     ierr = PIO_Inq_varid(File, 'U', udesc)
   418) 
   419)     ierr = PIO_Inq_varid(File, 'V', Vdesc)
   420) 
   421)     ierr = PIO_Inq_varid(File, 'T', tdesc)
   422) 
   423)     ierr = PIO_Inq_varid(File, 'OMEGA', OMEGAdesc)
   424) 
   425)     ierr = PIO_Inq_varid(File, 'PS', psdesc)
   426) 
   427)     ierr = PIO_Inq_varid(File, 'PHIS', phisdesc)
   428) 
   429) 
   430)     allocate(qdesc(qsize_d), qdesc_dp(qsize_d))
   431) 
   432)     do q=1,qsize_d
   433)        ierr = PIO_Inq_varid(File, cnst_name(q) ,Qdesc(q))
   434)        ierr = PIO_Inq_varid(File, "dp"//cnst_name(q) ,Qdesc_dp(q))
   435)     end do
   436) 
   437) 
   438)     call pio_setframe(phisdesc, int(1,kind=PIO_OFFSET))
   439) 
   440)     call pio_read_darray(File, phisdesc, iodesc2d, var2d, ierr)
   441) 
   442)     cnt=0
   443)     do ie=1,nelemd
   444)        do j=1,np
   445)           do i=1,np
   446)              cnt=cnt+1
   447)              elem(ie)%state%phis(i,j) = var2d(cnt)
   448)           end do
   449)        end do
   450)     end do
   451) 
   452) 
   453)     do ie=1,nelemd
   454)        elem(ie)%derived%fM = 0._r8
   455)        elem(ie)%derived%fT = 0._r8
   456)        elem(ie)%derived%fQ = 0._r8
   457)        elem(ie)%state%Q = 0._r8
   458)     end do
   459) 
   460)     call pio_read_darray(File, omegadesc, iodesc3d, var3d, ierr)
   461)     cnt=0
   462) 
   463)     do k=1,nlev
   464)        do ie=1,nelemd
   465)           do j=1,np
   466)              do i=1,np
   467)                 cnt=cnt+1
   468)                 elem(ie)%derived%omega_p(i,j,k) = var3d(cnt)
   469)              end do
   470)           end do
   471)        end do
   472)     end do
   473) 
   474) 
   475)     tl = timelevel%n0
   476)     call TimeLevel_Qdp(timelevel, qsplit, tlQdp)
   477) 
   478)     call pio_setframe(psdesc, t)
   479)     call pio_read_darray(File, psdesc, iodesc2d, var2d, ierr)
   480) 
   481)     cnt=0
   482)     do ie=1,nelemd
   483)        do j=1,np
   484)           do i=1,np
   485)              cnt=cnt+1
   486)              elem(ie)%state%ps_v(i,j,tl) = var2d(cnt)
   487)              elem(ie)%state%lnps(i,j,tl) = log(var2d(cnt))
   488)           end do
   489)        end do
   490)     end do
   491) 
   492)     call pio_setframe(udesc, t)
   493)     call pio_read_darray(File, udesc, iodesc3d, var3d, ierr)
   494) 
   495)     cnt=0
   496)     do k=1,nlev
   497)        do ie=1,nelemd
   498)           do j=1,np
   499)              do i=1,np
   500)                 cnt=cnt+1
   501)                 elem(ie)%state%v(i,j,1,k,tl) = var3d(cnt)
   502)              end do
   503)           end do
   504)        end do
   505)     end do
