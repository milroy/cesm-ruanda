   180)   subroutine noy_ubc_set( lchnk, ncol, vmr )
   181)     use cam_history,  only : outfld
   182) 
   183)     implicit none
   184) 
   185)     !--------------------------------------------------------------------
   186)     !	... dummy args
   187)     !--------------------------------------------------------------------
   188)     integer,  intent(in)    :: lchnk            ! chunk id
   189)     integer,  intent(in)    :: ncol              ! columns in chunk
   190)     real(r8), intent(inout) :: vmr(:,:,:)
   191) 
   192)     integer  :: m,n,m1,m2,i
   193)     real(r8) :: xno,xno2,xnox,rno,dtime
   194)     real(r8) :: yno,yno2,ynox
   195) 
   196)     if (.not.has_noy_ubc) return
   197) !
   198) ! only update model top layer (index=1)
   199) !
   200)     if (ub_nspecies > 0) then
   201)        do m = 1,ub_nspecies
   202)           if ( trim(fields(m)%fldnam) == 'NO' .or. trim(fields(m)%fldnam) == 'NO2' ) cycle
   203)           n = map(m)
   204)           vmr(:ncol,1,n) = fields(m)%data(:ncol,1,lchnk)
   205)           call outfld( ubc_name(m), vmr(:ncol,1,n), ncol, lchnk )
   206)        enddo
   207)     endif
   208) !
   209) ! special case for NO & NO2
   210) !
   211)     m1 = -99
   212)     m2 = -99
   213)     do m=1,ub_nspecies
   214)        if ( trim(fields(m)%fldnam) == 'NO'  ) m1 = m
   215)        if ( trim(fields(m)%fldnam) == 'NO2' ) m2 = m
   216)     end do
   217)     if ( m1 > 0 .and. m2 > 0 ) then
   218) !
   219)        do i=1,ncol
   220) !
   221)           xno  = vmr(i,1,map(m1))
   222)           xno2 = vmr(i,1,map(m2))
   223)           xnox = xno + xno2
   224)           rno  = xno/xnox
   225) !
   226)           yno  = fields(m1)%data(i,1,lchnk)
   227)           yno2 = fields(m2)%data(i,1,lchnk)
   228)           ynox = yno + yno2
   229) !
   230) ! relax model NOx towards the specified values
   231) !
   232)           xnox = xnox + (ynox - xnox) * fac_relax
   233) !
   234) ! use original ratio to redistribute updated NOx between NO and NO2
   235) !
   236)           vmr(i,1,map(m1)) = rno * xnox
   237)           vmr(i,1,map(m2)) = (1._r8-rno) * xnox
   238) !
   239)        end do
   240) !
   241)        call outfld( ubc_name(m1), vmr(:ncol,1,map(m1)), ncol, lchnk )
   242)        call outfld( ubc_name(m2), vmr(:ncol,1,map(m2)), ncol, lchnk )
   243)     end if
   244) !
   245)   end subroutine noy_ubc_set
