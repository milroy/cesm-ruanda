   193)     real(r8) :: alb_roof_dif_s(num_urbanl,numrad)    ! diffuse roof albedo with snow effects

   195)     real(r8) :: alb_perroad_dir_s(num_urbanl,numrad) ! direct pervious road albedo with snow effects
   196)     real(r8) :: alb_improad_dif_s(num_urbanl,numrad) ! diffuse impervious road albedo with snow effects
   197)     real(r8) :: alb_perroad_dif_s(num_urbanl,numrad) ! diffuse pervious road albedo with snow effects

   208)     real(r8), pointer :: canyon_hwr(:)             ! ratio of building height to street width

   211)     real(r8), pointer :: alb_roof_dif(:,:)         ! diffuse roof albedo

   214)     real(r8), pointer :: alb_improad_dif(:,:)      ! diffuse imprevious road albedo
   215)     real(r8), pointer :: alb_perroad_dif(:,:)      ! diffuse pervious road albedo

   217)     real(r8), pointer :: alb_wall_dif(:,:)         ! diffuse wall albedo

   227)     alb_improad_dif    => urban_clump(nc)%alb_improad_dif

   231)     alb_wall_dif       => urban_clump(nc)%alb_wall_dif  
   232) 
   233)     ! Assign gridcell level pointers
   234) 
   235)     lat                => grc%lat
   236)     lon                => grc%lon
   237)

   239) 
   240)     lgridcell          => lun%gridcell
   241)     coli               => lun%coli
   242)     colf               => lun%colf

   244)     vf_wr              => lps%vf_wr
   245)     vf_sw              => lps%vf_sw
   246)     vf_rw              => lps%vf_rw
   247)     vf_ww              => lps%vf_ww
   248)     sabs_roof_dir      => lps%sabs_roof_dir
   249)     sabs_roof_dif      => lps%sabs_roof_dif
   250)     sabs_sunwall_dir   => lps%sabs_sunwall_dir
   251)     sabs_sunwall_dif   => lps%sabs_sunwall_dif
   252)     sabs_shadewall_dir => lps%sabs_shadewall_dir
   253)     sabs_shadewall_dif => lps%sabs_shadewall_dif
   254)     sabs_improad_dir   => lps%sabs_improad_dir
   255)     sabs_improad_dif   => lps%sabs_improad_dif
   256)     sabs_perroad_dir   => lps%sabs_perroad_dir
   257)     sabs_perroad_dif   => lps%sabs_perroad_dif
   258)

   260)

   263)     albgri             => cps%albgri

   266)     cgridcell          => col%gridcell

   277)     ftdd               => pps%ftdd

   289)     do fl = 1,num_urbanl
   290)        l = filter_urbanl(fl)
   291)        g = lgridcell(l)
   292)        coszen(fl) = czen(coli(l))               ! Assumes coszen for each column are the same
   293)        zen(fl)    = acos(coszen(fl))
   294)     end do
   295) 
   296)     do fp = 1,num_urbanp
   297)        p = filter_urbanp(fp)
   298)        g = pgridcell(p)
   299)        c = pcolumn(p)
   300)        coszen_pft(fp) = czen(c)
   301)     end do
   302) 
   303)     ! ----------------------------------------------------------------------------
   304)     ! Initialize clmtype output since solar radiation is only done if coszen > 0
   305)     ! ----------------------------------------------------------------------------
   306)     
   307)     do ib = 1,numrad
   308)        do fc = 1,num_urbanc
   309)           c = filter_urbanc(fc)
   310) 
   311)           albgrd(c,ib) = 0._r8
   312)           albgri(c,ib) = 0._r8
   313)        end do
   314) 
   315)        do fp = 1,num_urbanp
   316)           p = filter_urbanp(fp)
   317)           g = pgridcell(p)
   318)           albd(p,ib)    = 1._r8
   319)           albi(p,ib)    = 1._r8
   320)           fabd(p,ib)    = 0._r8
   321)           fabi(p,ib)    = 0._r8
   322)           if (coszen_pft(fp) > 0._r8) then
   323)              ftdd(p,ib) = 1._r8
   324)           else
   325)              ftdd(p,ib) = 0._r8
   326)           end if
   327)           ftid(p,ib)    = 0._r8
   328)           if (coszen_pft(fp) > 0._r8) then
   329)              ftii(p,ib) = 1._r8
   330)           else
   331)              ftii(p,ib) = 0._r8
   332)           end if
   333)           omega(p,ib)   = 0._r8
   334)           if (ib == 1) then
   335)              gdir(p)    = 0._r8
   336)              fsun(p)    = 0._r8
   337)           end if
   338)        end do
   339)     end do
   340) 
   341)     ! ----------------------------------------------------------------------------
   342)     ! Urban Code
   343)     ! ----------------------------------------------------------------------------
   344) 
   345)     num_solar = 0
   346)     do fl = 1,num_urbanl
   347)        if (coszen(fl) > 0._r8) num_solar = num_solar + 1
   348)     end do
   349) 
   350)     ! Initialize urban clump components
   351) 
   352)     do ib = 1,numrad
   353)        do fl = 1,num_urbanl
   354)           l = filter_urbanl(fl)
   355)           sabs_roof_dir(l,ib)      = 0._r8
   356)           sabs_roof_dif(l,ib)      = 0._r8
   357)           sabs_sunwall_dir(l,ib)   = 0._r8
   358)           sabs_sunwall_dif(l,ib)   = 0._r8
   359)           sabs_shadewall_dir(l,ib) = 0._r8
   360)           sabs_shadewall_dif(l,ib) = 0._r8
   361)           sabs_improad_dir(l,ib)   = 0._r8
   362)           sabs_improad_dif(l,ib)   = 0._r8
   363)           sabs_perroad_dir(l,ib)   = 0._r8
   364)           sabs_perroad_dif(l,ib)   = 0._r8
   365)           sref_roof_dir(fl,ib)      = 1._r8
   366)           sref_roof_dif(fl,ib)      = 1._r8
   367)           sref_sunwall_dir(fl,ib)   = 1._r8
   368)           sref_sunwall_dif(fl,ib)   = 1._r8
   369)           sref_shadewall_dir(fl,ib) = 1._r8
   370)           sref_shadewall_dif(fl,ib) = 1._r8
   371)           sref_improad_dir(fl,ib)   = 1._r8
   372)           sref_improad_dif(fl,ib)   = 1._r8
   373)           sref_perroad_dir(fl,ib)   = 1._r8
   374)           sref_perroad_dif(fl,ib)   = 1._r8
   375)        end do
   376)     end do
   377) 
   378)     ! View factors for road and one wall in urban canyon (depends only on canyon_hwr)
   379)     
   380)     if (num_urbanl .gt. 0) then
   381)        call view_factor (lbl, ubl, num_urbanl, filter_urbanl, canyon_hwr)
   382)     end if
   383) 
   384)     ! ----------------------------------------------------------------------------
   385)     ! Only do the rest if all coszen are positive 
   386)     ! ----------------------------------------------------------------------------
   387) 
   388)     if (num_solar > 0)then
   389) 
   390)        ! Set constants - solar fluxes are per unit incoming flux
   391) 
   392)        do ib = 1,numrad
   393)           do fl = 1,num_urbanl
   394)              sdir(fl,ib) = 1._r8
   395)              sdif(fl,ib) = 1._r8
   396)           end do
   397)        end do
   398) 
   399)        ! Incident direct beam radiation for 
   400)        ! (a) roof and (b) road and both walls in urban canyon
   401)           
   402)        if (num_urbanl .gt. 0) then
   403)           call incident_direct (lbl, ubl, num_urbanl, canyon_hwr, coszen, zen, sdir, sdir_road, sdir_sunwall, sdir_shadewall)
   404)        end if
   405)        
   406)        ! Incident diffuse radiation for 
   407)        ! (a) roof and (b) road and both walls in urban canyon.
   408)        
   409)        if (num_urbanl .gt. 0) then
   410)           call incident_diffuse (lbl, ubl, num_urbanl, filter_urbanl, canyon_hwr, sdif, sdif_road, &
   411)                                  sdif_sunwall, sdif_shadewall)
   412)        end if
   413) 
   414)        ! Get snow albedos for roof and impervious and pervious road
   415)        if (num_urbanl .gt. 0) then
   416)           ic = 0; call UrbanSnowAlbedo(lbl, ubl, num_urbanl, filter_urbanl, coszen, ic, albsnd_roof, albsnd_improad, albsnd_perroad)
   417)           ic = 1; call UrbanSnowAlbedo(lbl, ubl, num_urbanl, filter_urbanl, coszen, ic, albsni_roof, albsni_improad, albsni_perroad)
   418)        end if
   419) 
   420)        ! Combine snow-free and snow albedos
   421)        do ib = 1,numrad
   422)           do fl = 1,num_urbanl
   423)              l = filter_urbanl(fl)
   424)              do c = coli(l),colf(l)
   425)                 if (ctype(c) == icol_roof) then    
   426)                    alb_roof_dir_s(fl,ib) = alb_roof_dir(fl,ib)*(1._r8-frac_sno(c))  &
   427)                                         + albsnd_roof(fl,ib)*frac_sno(c)
   428)                    alb_roof_dif_s(fl,ib) = alb_roof_dif(fl,ib)*(1._r8-frac_sno(c))  &
   429)                                         + albsni_roof(fl,ib)*frac_sno(c)
   430)                 else if (ctype(c) == icol_road_imperv) then    
   431)                    alb_improad_dir_s(fl,ib) = alb_improad_dir(fl,ib)*(1._r8-frac_sno(c))  &
   432)                                         + albsnd_improad(fl,ib)*frac_sno(c)
   433)                    alb_improad_dif_s(fl,ib) = alb_improad_dif(fl,ib)*(1._r8-frac_sno(c))  &
   434)                                         + albsni_improad(fl,ib)*frac_sno(c)
   435)                 else if (ctype(c) == icol_road_perv) then    
   436)                    alb_perroad_dir_s(fl,ib) = alb_perroad_dir(fl,ib)*(1._r8-frac_sno(c))  &
   437)                                         + albsnd_perroad(fl,ib)*frac_sno(c)
   438)                    alb_perroad_dif_s(fl,ib) = alb_perroad_dif(fl,ib)*(1._r8-frac_sno(c))  &
   439)                                         + albsni_perroad(fl,ib)*frac_sno(c)
   440)                 end if
   441)              end do
   442)           end do
   443)        end do
   444)        
   445)        ! Reflected and absorbed solar radiation per unit incident radiation 
   446)        ! for road and both walls in urban canyon allowing for multiple reflection
   447)        ! Reflected and absorbed solar radiation per unit incident radiation for roof
   448)        
   449)        if (num_urbanl .gt. 0) then
   450)           call net_solar (lbl, ubl, num_urbanl, filter_urbanl, coszen, canyon_hwr, wtroad_perv, sdir, sdif, &
   451)                           alb_improad_dir_s, alb_perroad_dir_s, alb_wall_dir, alb_roof_dir_s, &
   452)                           alb_improad_dif_s, alb_perroad_dif_s, alb_wall_dif, alb_roof_dif_s, &
   453)                           sdir_road, sdir_sunwall, sdir_shadewall,  &
   454)                           sdif_road, sdif_sunwall, sdif_shadewall,  &
   455)                           sref_improad_dir, sref_perroad_dir, sref_sunwall_dir, sref_shadewall_dir, sref_roof_dir, &
   456)                           sref_improad_dif, sref_perroad_dif, sref_sunwall_dif, sref_shadewall_dif, sref_roof_dif)
   457)        end if
   458) 
   459)        ! ----------------------------------------------------------------------------
   460)        ! Map urban output to clmtype components 
   461)        ! ----------------------------------------------------------------------------
   462) 
   463)        !  Set albgrd and albgri (ground albedos) and albd and albi (surface albedos)
   464) 
   465)        do ib = 1,numrad
   466)           do fl = 1,num_urbanl
   467)              l = filter_urbanl(fl)
   468)              do c = coli(l),colf(l)
   469)                 if (ctype(c) == icol_roof) then    
   470)                    albgrd(c,ib) = sref_roof_dir(fl,ib) 
   471)                    albgri(c,ib) = sref_roof_dif(fl,ib) 
   472)                 else if (ctype(c) == icol_sunwall) then   
   473)                    albgrd(c,ib) = sref_sunwall_dir(fl,ib)
   474)                    albgri(c,ib) = sref_sunwall_dif(fl,ib)
   475)                 else if (ctype(c) == icol_shadewall) then 
   476)                    albgrd(c,ib) = sref_shadewall_dir(fl,ib)
   477)                    albgri(c,ib) = sref_shadewall_dif(fl,ib)
   478)                 else if (ctype(c) == icol_road_perv) then
   479)                    albgrd(c,ib) = sref_perroad_dir(fl,ib)
   480)                    albgri(c,ib) = sref_perroad_dif(fl,ib)
   481)                 else if (ctype(c) == icol_road_imperv) then
   482)                    albgrd(c,ib) = sref_improad_dir(fl,ib)
   483)                    albgri(c,ib) = sref_improad_dif(fl,ib)
   484)                 endif
   485)              end do
   486)           end do
   487)           do fp = 1,num_urbanp
   488)              p = filter_urbanp(fp)
   489)              c = pcolumn(p)
   490)              albd(p,ib) = albgrd(c,ib)
   491)              albi(p,ib) = albgri(c,ib)
   492)           end do
   493)        end do
   494)     end if
   495)

  1020)   subroutine view_factor (lbl, ubl, num_urbanl, filter_urbanl, canyon_hwr)
  1021) 
  1022) !
  1023) ! !DESCRIPTION: 
  1024) ! View factors for road and one wall
  1025) !                                                        WALL    |
  1026) !                  ROAD                                          |
  1027) !                                                         wall   |
  1028) !          -----\          /-----   -             -  |\----------/
  1029) !              | \  vsr   / |       |         r   |  | \  vww   /   s
  1030) !              |  \      /  |       h         o   w  |  \      /    k
  1031) !        wall  |   \    /   | wall  |         a   |  |   \    /     y
  1032) !              |vwr \  / vwr|       |         d   |  |vrw \  / vsw 
  1033) !              ------\/------       -             -  |-----\/-----
  1034) !                   road                                  wall   |
  1035) !              <----- w ---->                                    |
  1036) !                                                    <---- h --->|
  1037) !
  1038) !    vsr = view factor of sky for road          vrw = view factor of road for wall
  1039) !    vwr = view factor of one wall for road     vww = view factor of opposing wall for wall
  1040) !                                               vsw = view factor of sky for wall
  1041) !    vsr + vwr + vwr = 1                        vrw + vww + vsw = 1
  1042) !
  1043) ! Source: Masson, V. (2000) A physically-based scheme for the urban energy budget in 
  1044) ! atmospheric models. Boundary-Layer Meteorology 94:357-397
  1045) !
  1046) ! !USES:
  1047)     use shr_kind_mod, only: r8 => shr_kind_r8
  1048)     use clmtype
  1049) !
  1050) ! !ARGUMENTS:
  1051)     implicit none
  1052)     integer,  intent(in) :: lbl, ubl                  ! landunit-index bounds
  1053)     integer , intent(in)  :: num_urbanl               ! number of urban landunits
  1054)     integer , intent(in)  :: filter_urbanl(ubl-lbl+1) ! urban landunit filter
  1055)     real(r8), intent(in)  :: canyon_hwr(num_urbanl)   ! ratio of building height to street width
  1056) !
  1057) ! local pointers to original implicit out arguments (clmtype)
  1058) !
  1059)     real(r8), pointer :: vf_sr(:)     ! view factor of sky for road
  1060)     real(r8), pointer :: vf_wr(:)     ! view factor of one wall for road
  1061)     real(r8), pointer :: vf_sw(:)     ! view factor of sky for one wall
  1062)     real(r8), pointer :: vf_rw(:)     ! view factor of road for one wall
  1063)     real(r8), pointer :: vf_ww(:)     ! view factor of opposing wall for one wall
  1064) !
  1065) ! !CALLED FROM:
  1066) ! subroutine UrbanAlbedo in this module
  1067) !
  1068) ! !REVISION HISTORY:
  1069) ! Author: Gordon Bonan
  1070) ! 03/2003, Mariana Vertenstein: Migrated to clm2.2 
  1071) ! 01/2008, Erik Kluzek:         Migrated to clm3.5.15
  1072) !
  1073) !
  1074) ! !LOCAL VARIABLES:
  1075) !EOP
  1076)     integer :: l, fl   ! indices
  1077)     real(r8) :: sum    ! sum of view factors for wall or road
  1078) !-----------------------------------------------------------------------
  1079) 
  1080)     ! Assign landunit level pointer
  1081) 
  1082)     vf_sr  => lps%vf_sr
  1083)     vf_wr  => lps%vf_wr
  1084)     vf_sw  => lps%vf_sw
  1085)     vf_rw  => lps%vf_rw
  1086)     vf_ww  => lps%vf_ww
  1087) 
  1088)     do fl = 1,num_urbanl
  1089)        l = filter_urbanl(fl)
  1090) 
  1091)        ! road -- sky view factor -> 1 as building height -> 0 
  1092)        ! and -> 0 as building height -> infinity
  1093) 
  1094)        vf_sr(l) = sqrt(canyon_hwr(fl)**2 + 1._r8) - canyon_hwr(fl)
  1095)        vf_wr(l) = 0.5_r8 * (1._r8 - vf_sr(l))
  1096) 
  1097)        ! one wall -- sky view factor -> 0.5 as building height -> 0 
  1098)        ! and -> 0 as building height -> infinity
  1099) 
  1100)        vf_sw(l) = 0.5_r8 * (canyon_hwr(fl) + 1._r8 - sqrt(canyon_hwr(fl)**2+1._r8)) / canyon_hwr(fl)
  1101)        vf_rw(l) = vf_sw(l)
  1102)        vf_ww(l) = 1._r8 - vf_sw(l) - vf_rw(l)
  1103) 
  1104)     end do
  1105) 
  1106) 
  1107)     ! error check -- make sure view factor sums to one for road and wall
  1108) 
  1109)     do fl = 1,num_urbanl
  1110)        l = filter_urbanl(fl)
  1111) 
  1112)        sum = vf_sr(l) + 2._r8*vf_wr(l)
  1113)        if (abs(sum-1._r8) > 1.e-06_r8 ) then
  1114)           write (iulog,*) 'urban road view factor error',sum
  1115)           write (iulog,*) 'clm model is stopping'
  1116)           call endrun()
  1117)        endif
  1118)        sum = vf_sw(l) + vf_rw(l) + vf_ww(l)
  1119)        if (abs(sum-1._r8) > 1.e-06_r8 ) then
  1120)           write (iulog,*) 'urban wall view factor error',sum
  1121)           write (iulog,*) 'clm model is stopping'
  1122)           call endrun()
  1123)        endif
  1124) 
  1125)     end do
  1126) 
  1127)   end subroutine view_factor

  1135)   subroutine incident_direct (lbl, ubl, num_urbanl, canyon_hwr, coszen, zen, sdir, sdir_road, sdir_sunwall, sdir_shadewall)
  1136) !
  1137) ! !DESCRIPTION: 
  1138) ! Direct beam solar radiation incident on walls and road in urban canyon
  1139) !
  1140) !                           Sun
  1141) !                            /
  1142) !             roof          /
  1143) !            ------        /---            -
  1144) !                 |       / |              |
  1145) !    sunlit wall  |      /  | shaded wall  h
  1146) !                 |     /   |              |
  1147) !                 -----/-----              -
  1148) !                    road
  1149) !                 <--- w --->
  1150) !
  1151) ! Method:
  1152) ! Road          = Horizontal surface. Account for shading by wall. Integrate over all canyon orientations
  1153) ! Wall (sunlit) = Adjust horizontal radiation for 90 degree surface. Account for shading by opposing wall.
  1154) !                 Integrate over all canyon orientations
  1155) ! Wall (shaded) = 0
  1156) !
  1157) ! Conservation check: Total incoming direct beam (sdir) = sdir_road + (sdir_shadewall + sdir_sunwall)*canyon_hwr
  1158) ! Multiplication by canyon_hwr scales wall fluxes (per unit wall area) to per unit ground area
  1159) !
  1160) ! Source: Masson, V. (2000) A physically-based scheme for the urban energy budget in 
  1161) ! atmospheric models. Boundary-Layer Meteorology 94:357-397
  1162) !
  1163) ! This analytical solution from Masson (2000) agrees with the numerical solution to
  1164) ! within 0.6 W/m**2 for sdir = 1000 W/m**2 and for all H/W from 0.1 to 10 by 0.1
  1165) ! and all solar zenith angles from 1 to 90 deg by 1
  1166) !
  1167) ! !USES:
  1168)     use shr_kind_mod  , only : r8 => shr_kind_r8
  1169)     use clm_varcon    , only : rpi
  1170)     implicit none
  1171) !
  1172) ! !ARGUMENTS:
  1173)     integer,  intent(in)  :: lbl, ubl                           ! landunit-index bounds
  1174)     integer , intent(in)  :: num_urbanl                         ! number of urban landunits
  1175)     real(r8), intent(in)  :: canyon_hwr(num_urbanl)             ! ratio of building height to street width
  1176)     real(r8), intent(in)  :: coszen(num_urbanl)                 ! cosine solar zenith angle
  1177)     real(r8), intent(in)  :: zen(num_urbanl)                    ! solar zenith angle (radians)
  1178)     real(r8), intent(in)  :: sdir(num_urbanl, numrad)           ! direct beam solar radiation incident on horizontal surface
  1179)     real(r8), intent(out) :: sdir_road(num_urbanl, numrad)      ! direct beam solar radiation incident on road per unit incident flux
  1180)     real(r8), intent(out) :: sdir_sunwall(num_urbanl, numrad)   ! direct beam solar radiation (per unit wall area) incident on sunlit wall per unit incident flux
  1181)     real(r8), intent(out) :: sdir_shadewall(num_urbanl, numrad) ! direct beam solar radiation (per unit wall area) incident on shaded wall per unit incident flux
  1182) !
  1183) ! !CALLED FROM:

  1522)

  1525)

  1529)

  1533)     real(r8) :: err                                 ! energy conservation error

  1536)     real(r8) :: sabs_road                           ! temporary for absorption over road

  1546)     vf_rw              => lps%vf_rw
  1547)     vf_ww              => lps%vf_ww

  1549)     sabs_roof_dif      => lps%sabs_roof_dif

  1551)     sabs_sunwall_dif   => lps%sabs_sunwall_dif

  1553)     sabs_shadewall_dif => lps%sabs_shadewall_dif

  1555)     sabs_improad_dif   => lps%sabs_improad_dif

  1557)     sabs_perroad_dif   => lps%sabs_perroad_dif

  1559)     ! Calculate impervious road
  1560) 
  1561)     do l = 1,num_urbanl 
  1562)        wtroad_imperv(l) = 1._r8 - wtroad_perv(l)
  1563)     end do
  1564) 
  1565)     do ib = 1,numrad
  1566)        do fl = 1,num_urbanl
  1567)         if (coszen(fl) .gt. 0._r8) then
  1568)           l = filter_urbanl(fl)
  1569) 
  1570)           ! initial absorption and reflection for road and both walls. 
  1571)           ! distribute reflected radiation to sky, road, and walls 
  1572)           ! according to appropriate view factor. radiation reflected to 
  1573)           ! road and walls will undergo multiple reflections within the canyon. 
  1574)           ! do separately for direct beam and diffuse radiation.
  1575)           
  1576)           ! direct beam
  1577)

  1579)           road_r_dir(fl)              = 0.0_r8

  1601)           road_r_sunwall_dir(fl)      = road_r_dir(fl) * vf_wr(l)

  1605)           sunwall_r_dir(fl)           =     alb_wall_dir(fl,ib)  * sdir_sunwall(fl,ib)

  1611)           shadewall_r_dir(fl)         =     alb_wall_dir(fl,ib)  * sdir_shadewall(fl,ib)

  1615)

  1618)           road_a_dif(fl)              = 0.0_r8
  1619)           road_r_dif(fl)              = 0.0_r8
  1620)           if ( wtroad_imperv(fl) > 0.0_r8 ) then
  1621)              improad_a_dif(fl)           = (1._r8-alb_improad_dif(fl,ib)) * sdif_road(fl,ib)

  1625)              improad_r_shadewall_dif(fl) = improad_r_dif(fl) * vf_wr(l)

  1627)              road_r_dif(fl)              = road_r_dif(fl) + improad_r_dif(fl)*wtroad_imperv(fl)
  1628)           end if
  1629) 
  1630)           if ( wtroad_perv(fl)   > 0.0_r8 ) then
  1631)              perroad_a_dif(fl)           = (1._r8-alb_perroad_dif(fl,ib)) * sdif_road(fl,ib) 
  1632)              perroad_r_dif(fl)           =     alb_perroad_dif(fl,ib)  * sdif_road(fl,ib) 
  1633)              perroad_r_sky_dif(fl)       = perroad_r_dif(fl) * vf_sr(l)
  1634)              perroad_r_sunwall_dif(fl)   = perroad_r_dif(fl) * vf_wr(l)
  1635)              perroad_r_shadewall_dif(fl) = perroad_r_dif(fl) * vf_wr(l)

  1638)           end if
  1639) 
  1640)           road_r_sky_dif(fl)          = road_r_dif(fl) * vf_sr(l)
  1641)           road_r_sunwall_dif(fl)      = road_r_dif(fl) * vf_wr(l)
  1642)           road_r_shadewall_dif(fl)    = road_r_dif(fl) * vf_wr(l)
  1643)

  1645)           sunwall_r_dif(fl)           =     alb_wall_dif(fl,ib)  * sdif_sunwall(fl,ib)
  1646)           sunwall_r_sky_dif(fl)       = sunwall_r_dif(fl) * vf_sw(l)
  1647)           sunwall_r_road_dif(fl)      = sunwall_r_dif(fl) * vf_rw(l)
  1648)           sunwall_r_shadewall_dif(fl) = sunwall_r_dif(fl) * vf_ww(l)
  1649) 
  1650)           shadewall_a_dif(fl)         = (1._r8-alb_wall_dif(fl,ib)) * sdif_shadewall(fl,ib)
  1651)           shadewall_r_dif(fl)         =     alb_wall_dif(fl,ib)  * sdif_shadewall(fl,ib)
  1652)           shadewall_r_sky_dif(fl)     = shadewall_r_dif(fl) * vf_sw(l)
  1653)           shadewall_r_road_dif(fl)    = shadewall_r_dif(fl) * vf_rw(l) 
  1654)           shadewall_r_sunwall_dif(fl) = shadewall_r_dif(fl) * vf_ww(l) 
  1655) 
  1656)           ! initialize sum of direct and diffuse solar absorption and reflection for road and both walls
  1657) 
  1658)           if ( wtroad_imperv(fl) > 0.0_r8 ) sabs_improad_dir(l,ib)   = improad_a_dir(fl)
  1659)           if ( wtroad_perv(fl)   > 0.0_r8 ) sabs_perroad_dir(l,ib)   = perroad_a_dir(fl)
  1660)           sabs_sunwall_dir(l,ib)   = sunwall_a_dir(fl)
  1661)           sabs_shadewall_dir(l,ib) = shadewall_a_dir(fl)
  1662) 
  1663)           if ( wtroad_imperv(fl) > 0.0_r8 ) sabs_improad_dif(l,ib)   = improad_a_dif(fl)
  1664)           if ( wtroad_perv(fl)   > 0.0_r8 ) sabs_perroad_dif(l,ib)   = perroad_a_dif(fl)
  1665)           sabs_sunwall_dif(l,ib)   = sunwall_a_dif(fl)
  1666)           sabs_shadewall_dif(l,ib) = shadewall_a_dif(fl)
  1667) 
  1668)           if ( wtroad_imperv(fl) > 0.0_r8 ) sref_improad_dir(fl,ib)   = improad_r_sky_dir(fl) 
  1669)           if ( wtroad_perv(fl)   > 0.0_r8 ) sref_perroad_dir(fl,ib)   = perroad_r_sky_dir(fl) 
  1670)           sref_sunwall_dir(fl,ib)   = sunwall_r_sky_dir(fl) 
  1671)           sref_shadewall_dir(fl,ib) = shadewall_r_sky_dir(fl) 
  1672) 
  1673)           if ( wtroad_imperv(fl) > 0.0_r8 ) sref_improad_dif(fl,ib)   = improad_r_sky_dif(fl)
  1674)           if ( wtroad_perv(fl)   > 0.0_r8 ) sref_perroad_dif(fl,ib)   = perroad_r_sky_dif(fl)
  1675)           sref_sunwall_dif(fl,ib)   = sunwall_r_sky_dif(fl)
  1676)           sref_shadewall_dif(fl,ib) = shadewall_r_sky_dif(fl)
  1677)         endif
  1678) 
  1679)        end do

  1683)        ! subsequent scattering)
  1684)        !
  1685)        ! (1) absorption and reflection of scattered solar radiation
  1686)        !     road: reflected fluxes from walls need to be projected to ground area
  1687)        !     wall: reflected flux from road needs to be projected to wall area
  1688)        !
  1689)        ! (2) add absorbed radiation for ith reflection to total absorbed
  1690)        !
  1691)        ! (3) distribute reflected radiation to sky, road, and walls according to view factors
  1692)        !
  1693)        ! (4) add solar reflection to sky for ith reflection to total reflection
  1694)        !
  1695)        ! (5) stop iteration when absorption for ith reflection is less than some nominal amount. 
  1696)        !     small convergence criteria is required to ensure solar radiation is conserved
  1697)        !
  1698)        ! do separately for direct beam and diffuse
  1699)        
  1700)        do fl = 1,num_urbanl
  1701)         if (coszen(fl) .gt. 0._r8) then
  1702)           l = filter_urbanl(fl)
  1703) 
  1704)           ! reflected direct beam
  1705) 
  1706)           do iter_dir = 1, n
  1707)              ! step (1)
  1708) 
  1709)              stot(fl) = (sunwall_r_road_dir(fl) + shadewall_r_road_dir(fl))*canyon_hwr(fl)
  1710) 
  1711)              road_a_dir(fl) = 0.0_r8
  1712)              road_r_dir(fl) = 0.0_r8
  1713)              if ( wtroad_imperv(fl) > 0.0_r8 ) then
  1714)                 improad_a_dir(fl) = (1._r8-alb_improad_dir(fl,ib)) * stot(fl) 
  1715)                 improad_r_dir(fl) =     alb_improad_dir(fl,ib)  * stot(fl) 
  1716)                 road_a_dir(fl)    = road_a_dir(fl) + improad_a_dir(fl)*wtroad_imperv(fl)
  1717)                 road_r_dir(fl)    = road_r_dir(fl) + improad_r_dir(fl)*wtroad_imperv(fl)
  1718)              end if
  1719)              if ( wtroad_perv(fl)   > 0.0_r8 ) then
  1720)                 perroad_a_dir(fl) = (1._r8-alb_perroad_dir(fl,ib)) * stot(fl) 
  1721)                 perroad_r_dir(fl) =     alb_perroad_dir(fl,ib)  * stot(fl) 
  1722)                 road_a_dir(fl)    = road_a_dir(fl) + perroad_a_dir(fl)*wtroad_perv(fl)
  1723)                 road_r_dir(fl)    = road_r_dir(fl) + perroad_r_dir(fl)*wtroad_perv(fl)
  1724)              end if
  1725) 
  1726)              stot(fl) = road_r_sunwall_dir(fl)/canyon_hwr(fl) + shadewall_r_sunwall_dir(fl)
  1727)              sunwall_a_dir(fl) = (1._r8-alb_wall_dir(fl,ib)) * stot(fl)
  1728)              sunwall_r_dir(fl) =     alb_wall_dir(fl,ib)  * stot(fl)
  1729) 
  1730)              stot(fl) = road_r_shadewall_dir(fl)/canyon_hwr(fl) + sunwall_r_shadewall_dir(fl)
  1731)              shadewall_a_dir(fl) = (1._r8-alb_wall_dir(fl,ib)) * stot(fl)
  1732)              shadewall_r_dir(fl) =     alb_wall_dir(fl,ib)  * stot(fl)
  1733) 
  1734)              ! step (2)
  1735) 
  1736)              if ( wtroad_imperv(fl) > 0.0_r8 ) sabs_improad_dir(l,ib)   = sabs_improad_dir(l,ib)   + improad_a_dir(fl)
  1737)              if ( wtroad_perv(fl)   > 0.0_r8 ) sabs_perroad_dir(l,ib)   = sabs_perroad_dir(l,ib)   + perroad_a_dir(fl)
  1738)              sabs_sunwall_dir(l,ib)   = sabs_sunwall_dir(l,ib)   + sunwall_a_dir(fl)
  1739)              sabs_shadewall_dir(l,ib) = sabs_shadewall_dir(l,ib) + shadewall_a_dir(fl)
  1740) 
  1741)              ! step (3)
  1742) 
  1743)              if ( wtroad_imperv(fl) > 0.0_r8 ) then
  1744)                 improad_r_sky_dir(fl)       = improad_r_dir(fl) * vf_sr(l)
  1745)                 improad_r_sunwall_dir(fl)   = improad_r_dir(fl) * vf_wr(l)
  1746)                 improad_r_shadewall_dir(fl) = improad_r_dir(fl) * vf_wr(l)
  1747)              end if
  1748) 
  1749)              if ( wtroad_perv(fl)   > 0.0_r8 ) then
  1750)                 perroad_r_sky_dir(fl)       = perroad_r_dir(fl) * vf_sr(l)
  1751)                 perroad_r_sunwall_dir(fl)   = perroad_r_dir(fl) * vf_wr(l)
  1752)                 perroad_r_shadewall_dir(fl) = perroad_r_dir(fl) * vf_wr(l)
  1753)              end if
  1754) 
  1755)              road_r_sky_dir(fl)          = road_r_dir(fl) * vf_sr(l)
  1756)              road_r_sunwall_dir(fl)      = road_r_dir(fl) * vf_wr(l)
  1757)              road_r_shadewall_dir(fl)    = road_r_dir(fl) * vf_wr(l)
  1758) 
  1759)              sunwall_r_sky_dir(fl)       = sunwall_r_dir(fl) * vf_sw(l)
  1760)              sunwall_r_road_dir(fl)      = sunwall_r_dir(fl) * vf_rw(l)
  1761)              sunwall_r_shadewall_dir(fl) = sunwall_r_dir(fl) * vf_ww(l)
  1762) 
  1763)              shadewall_r_sky_dir(fl)     = shadewall_r_dir(fl) * vf_sw(l)
  1764)              shadewall_r_road_dir(fl)    = shadewall_r_dir(fl) * vf_rw(l)
  1765)              shadewall_r_sunwall_dir(fl) = shadewall_r_dir(fl) * vf_ww(l)
  1766) 
  1767)              ! step (4)
  1768) 
  1769)              if ( wtroad_imperv(fl) > 0.0_r8 ) sref_improad_dir(fl,ib)   = sref_improad_dir(fl,ib) + improad_r_sky_dir(fl)
  1770)              if ( wtroad_perv(fl)   > 0.0_r8 ) sref_perroad_dir(fl,ib)   = sref_perroad_dir(fl,ib) + perroad_r_sky_dir(fl)
  1771)              sref_sunwall_dir(fl,ib)   = sref_sunwall_dir(fl,ib) + sunwall_r_sky_dir(fl)
  1772)              sref_shadewall_dir(fl,ib) = sref_shadewall_dir(fl,ib) + shadewall_r_sky_dir(fl)
  1773) 
  1774)              ! step (5)
  1775) 
  1776)              crit = max(road_a_dir(fl), sunwall_a_dir(fl), shadewall_a_dir(fl))
  1777)              if (crit < errcrit) exit
  1778)           end do

  1780)              write (iulog,*) 'urban net solar radiation error: no convergence, direct beam'
  1781)              write (iulog,*) 'clm model is stopping'
  1782)              call endrun
  1783)           endif
  1784)  
  1785)           ! reflected diffuse
  1786)        
  1787)           do iter_dif = 1, n
  1788)              ! step (1)
  1789) 
  1790)              stot(fl) = (sunwall_r_road_dif(fl) + shadewall_r_road_dif(fl))*canyon_hwr(fl)
  1791)              road_a_dif(fl)    = 0.0_r8
  1792)              road_r_dif(fl)    = 0.0_r8
  1793)              if ( wtroad_imperv(fl) > 0.0_r8 ) then

  1795)                 improad_r_dif(fl) =     alb_improad_dif(fl,ib)  * stot(fl) 
  1796)                 road_a_dif(fl)    = road_a_dif(fl) + improad_a_dif(fl)*wtroad_imperv(fl)
  1797)                 road_r_dif(fl)    = road_r_dif(fl) + improad_r_dif(fl)*wtroad_imperv(fl)
  1798)              end if
  1799)              if ( wtroad_perv(fl)   > 0.0_r8 ) then
  1800)                 perroad_a_dif(fl) = (1._r8-alb_perroad_dif(fl,ib)) * stot(fl) 
  1801)                 perroad_r_dif(fl) =     alb_perroad_dif(fl,ib)  * stot(fl) 
  1802)                 road_a_dif(fl)    = road_a_dif(fl) + perroad_a_dif(fl)*wtroad_perv(fl)
  1803)                 road_r_dif(fl)    = road_r_dif(fl) + perroad_r_dif(fl)*wtroad_perv(fl)
  1804)              end if
  1805) 
  1806)              stot(fl) = road_r_sunwall_dif(fl)/canyon_hwr(fl) + shadewall_r_sunwall_dif(fl)
  1807)              sunwall_a_dif(fl) = (1._r8-alb_wall_dif(fl,ib)) * stot(fl)
  1808)              sunwall_r_dif(fl) =     alb_wall_dif(fl,ib)  * stot(fl)
  1809) 
  1810)              stot(fl) = road_r_shadewall_dif(fl)/canyon_hwr(fl) + sunwall_r_shadewall_dif(fl)
  1811)              shadewall_a_dif(fl) = (1._r8-alb_wall_dif(fl,ib)) * stot(fl)
  1812)              shadewall_r_dif(fl) =     alb_wall_dif(fl,ib)  * stot(fl)
  1813) 
  1814)              ! step (2)
  1815) 
  1816)              if ( wtroad_imperv(fl) > 0.0_r8 ) sabs_improad_dif(l,ib)   = sabs_improad_dif(l,ib)   + improad_a_dif(fl)
  1817)              if ( wtroad_perv(fl)   > 0.0_r8 ) sabs_perroad_dif(l,ib)   = sabs_perroad_dif(l,ib)   + perroad_a_dif(fl)
  1818)              sabs_sunwall_dif(l,ib)   = sabs_sunwall_dif(l,ib)   + sunwall_a_dif(fl)
  1819)              sabs_shadewall_dif(l,ib) = sabs_shadewall_dif(l,ib) + shadewall_a_dif(fl)
  1820) 
  1821)              ! step (3)
  1822) 
  1823)              if ( wtroad_imperv(fl) > 0.0_r8 ) then
  1824)                 improad_r_sky_dif(fl)       = improad_r_dif(fl) * vf_sr(l)
  1825)                 improad_r_sunwall_dif(fl)   = improad_r_dif(fl) * vf_wr(l)
  1826)                 improad_r_shadewall_dif(fl) = improad_r_dif(fl) * vf_wr(l)
  1827)              end if
  1828) 
  1829)              if ( wtroad_perv(fl)   > 0.0_r8 ) then
  1830)                 perroad_r_sky_dif(fl)       = perroad_r_dif(fl) * vf_sr(l)
  1831)                 perroad_r_sunwall_dif(fl)   = perroad_r_dif(fl) * vf_wr(l)
  1832)                 perroad_r_shadewall_dif(fl) = perroad_r_dif(fl) * vf_wr(l)
  1833)              end if
  1834) 
  1835)              road_r_sky_dif(fl)          = road_r_dif(fl) * vf_sr(l)
  1836)              road_r_sunwall_dif(fl)      = road_r_dif(fl) * vf_wr(l)
  1837)              road_r_shadewall_dif(fl)    = road_r_dif(fl) * vf_wr(l)
  1838) 
  1839)              sunwall_r_sky_dif(fl)       = sunwall_r_dif(fl) * vf_sw(l)
  1840)              sunwall_r_road_dif(fl)      = sunwall_r_dif(fl) * vf_rw(l)
  1841)              sunwall_r_shadewall_dif(fl) = sunwall_r_dif(fl) * vf_ww(l)
  1842) 
  1843)              shadewall_r_sky_dif(fl)     = shadewall_r_dif(fl) * vf_sw(l)
  1844)              shadewall_r_road_dif(fl)    = shadewall_r_dif(fl) * vf_rw(l)
  1845)              shadewall_r_sunwall_dif(fl) = shadewall_r_dif(fl) * vf_ww(l)
  1846) 
  1847)              ! step (4)
  1848) 
  1849)              if ( wtroad_imperv(fl) > 0.0_r8 ) sref_improad_dif(fl,ib)   = sref_improad_dif(fl,ib)   + improad_r_sky_dif(fl)
  1850)              if ( wtroad_perv(fl)   > 0.0_r8 ) sref_perroad_dif(fl,ib)   = sref_perroad_dif(fl,ib)   + perroad_r_sky_dif(fl)
  1851)              sref_sunwall_dif(fl,ib)   = sref_sunwall_dif(fl,ib)   + sunwall_r_sky_dif(fl)
  1852)              sref_shadewall_dif(fl,ib) = sref_shadewall_dif(fl,ib) + shadewall_r_sky_dif(fl)
  1853) 
  1854)              ! step (5)
  1855) 
  1856)              crit = max(road_a_dif(fl), sunwall_a_dif(fl), shadewall_a_dif(fl))
  1857)              if (crit < errcrit) exit
  1858)           end do
  1859)           if (iter_dif >= n) then
  1860)              write (iulog,*) 'urban net solar radiation error: no convergence, diffuse'
  1861)              write (iulog,*) 'clm model is stopping'
  1862)              call endrun()
  1863)           endif
  1864) 
  1865)           ! total reflected by canyon - sum of solar reflection to sky from canyon.
  1866)           ! project wall fluxes to horizontal surface
  1867)           
  1868)           sref_canyon_dir(fl) = 0.0_r8
  1869)           sref_canyon_dif(fl) = 0.0_r8
  1870)           if ( wtroad_imperv(fl) > 0.0_r8 ) then
  1871)              sref_canyon_dir(fl) = sref_canyon_dir(fl) + sref_improad_dir(fl,ib)*wtroad_imperv(fl)
  1872)              sref_canyon_dif(fl) = sref_canyon_dif(fl) + sref_improad_dif(fl,ib)*wtroad_imperv(fl)
  1873)           end if
  1874)           if ( wtroad_perv(fl)   > 0.0_r8 ) then
  1875)              sref_canyon_dir(fl) = sref_canyon_dir(fl) + sref_perroad_dir(fl,ib)*wtroad_perv(fl)
  1876)              sref_canyon_dif(fl) = sref_canyon_dif(fl) + sref_perroad_dif(fl,ib)*wtroad_perv(fl)
  1877)           end if
  1878)           sref_canyon_dir(fl) = sref_canyon_dir(fl) + (sref_sunwall_dir(fl,ib) + sref_shadewall_dir(fl,ib))*canyon_hwr(fl)
  1879)           sref_canyon_dif(fl) = sref_canyon_dif(fl) + (sref_sunwall_dif(fl,ib) + sref_shadewall_dif(fl,ib))*canyon_hwr(fl)
  1880) 
  1881)           ! total absorbed by canyon. project wall fluxes to horizontal surface
  1882) 
  1883)           sabs_canyon_dir(fl) = 0.0_r8
  1884)           sabs_canyon_dif(fl) = 0.0_r8
  1885)           if ( wtroad_imperv(fl) > 0.0_r8 ) then
  1886)              sabs_canyon_dir(fl) = sabs_canyon_dir(fl) + sabs_improad_dir(l,ib)*wtroad_imperv(fl)
  1887)              sabs_canyon_dif(fl) = sabs_canyon_dif(fl) + sabs_improad_dif(l,ib)*wtroad_imperv(fl)
  1888)           end if
  1889)           if ( wtroad_perv(fl)   > 0.0_r8 ) then
  1890)              sabs_canyon_dir(fl) = sabs_canyon_dir(fl) + sabs_perroad_dir(l,ib)*wtroad_perv(fl)
  1891)              sabs_canyon_dif(fl) = sabs_canyon_dif(fl) + sabs_perroad_dif(l,ib)*wtroad_perv(fl)
  1892)           end if
  1893)           sabs_canyon_dir(fl) = sabs_canyon_dir(fl) + (sabs_sunwall_dir(l,ib) + sabs_shadewall_dir(l,ib))*canyon_hwr(fl)
  1894)           sabs_canyon_dif(fl) = sabs_canyon_dif(fl) + (sabs_sunwall_dif(l,ib) + sabs_shadewall_dif(l,ib))*canyon_hwr(fl)
  1895) 
  1896)           ! conservation check. note: previous conservation checks confirm partioning of total direct
  1897)           ! beam and diffuse radiation from atmosphere to road and walls is conserved as
  1898)           !    sdir (from atmosphere) = sdir_road + (sdir_sunwall + sdir_shadewall)*canyon_hwr
  1899)           !    sdif (from atmosphere) = sdif_road + (sdif_sunwall + sdif_shadewall)*canyon_hwr
  1900) 
  1901)           stot_dir(fl) = sdir_road(fl,ib) + (sdir_sunwall(fl,ib) + sdir_shadewall(fl,ib))*canyon_hwr(fl)
  1902)           stot_dif(fl) = sdif_road(fl,ib) + (sdif_sunwall(fl,ib) + sdif_shadewall(fl,ib))*canyon_hwr(fl)
  1903) 
  1904)           err = stot_dir(fl) + stot_dif(fl) &
  1905)                 - (sabs_canyon_dir(fl) + sabs_canyon_dif(fl) + sref_canyon_dir(fl) + sref_canyon_dif(fl))
  1906)           if (abs(err) > 0.001_r8 ) then
  1907)              write(iulog,*)'urban net solar radiation balance error for ib=',ib,' err= ',err
  1908)              write(iulog,*)' l= ',l,' ib= ',ib 
  1909)              write(iulog,*)' stot_dir        = ',stot_dir(fl)
  1910)              write(iulog,*)' stot_dif        = ',stot_dif(fl)
  1911)              write(iulog,*)' sabs_canyon_dir = ',sabs_canyon_dir(fl)
  1912)              write(iulog,*)' sabs_canyon_dif = ',sabs_canyon_dif(fl)
  1913)              write(iulog,*)' sref_canyon_dir = ',sref_canyon_dir(fl)
  1914)              write(iulog,*)' sref_canyon_dif = ',sref_canyon_dir(fl)
  1915)              write(iulog,*) 'clm model is stopping'
  1916)              call endrun()
  1917)           endif
  1918) 
  1919)           ! canyon albedo
  1920) 
  1921)           canyon_alb_dif(fl) = sref_canyon_dif(fl) / max(stot_dif(fl), 1.e-06_r8)
  1922)           canyon_alb_dir(fl) = sref_canyon_dir(fl) / max(stot_dir(fl), 1.e-06_r8)
  1923)         end if
  1924) 
  1925)        end do   ! end of landunit loop
  1926) 
  1927)        ! Refected and absorbed solar radiation per unit incident radiation for roof
  1928) 
  1929)        do fl = 1,num_urbanl
  1930)         if (coszen(fl) .gt. 0._r8) then
  1931)           l = filter_urbanl(fl)
  1932)           sref_roof_dir(fl,ib) = alb_roof_dir(fl,ib) * sdir(fl,ib)
  1933)           sref_roof_dif(fl,ib) = alb_roof_dif(fl,ib) * sdif(fl,ib)
  1934)           sabs_roof_dir(l,ib) = sdir(fl,ib) - sref_roof_dir(fl,ib)
  1935)           sabs_roof_dif(l,ib) = sdif(fl,ib) - sref_roof_dif(fl,ib)
  1936)         end if
  1937)        end do
  1938) 
  1939)     end do   ! end of radiation band loop
  1940) 
  1941)   end subroutine net_solar

  2079)     vf_wr              => lps%vf_wr

  2093)        ! check for conservation (need to convert wall fluxes to ground area).
  2094)        ! lwdown (from atmosphere) = lwdown_road + (lwdown_sunwall + lwdown_shadewall)*canyon_hwr
  2095)        
  2096)        lwdown_road(fl)      = lwdown(fl) * vf_sr(l) 
  2097)        lwdown_sunwall(fl)   = lwdown(fl) * vf_sw(l)
  2098)        lwdown_shadewall(fl) = lwdown(fl) * vf_sw(l)

  2100)        err = lwdown(fl) - (lwdown_road(fl) + (lwdown_shadewall(fl) + lwdown_sunwall(fl))*canyon_hwr(fl))
  2101)        if (abs(err) > 0.10_r8 ) then
  2102)           write (iulog,*) 'urban incident atmospheric longwave radiation balance error',err
  2103)           write (iulog,*) 'clm model is stopping'

  2107) 
  2108)     do fl = 1,num_urbanl
  2109)        l = filter_urbanl(fl)
  2110) 
  2111)        ! initial absorption, reflection, and emission for road and both walls. 
  2112)        ! distribute reflected and emitted radiation to sky, road, and walls according 
  2113)        ! to appropriate view factor. radiation reflected to road and walls will
  2114)        ! undergo multiple reflections within the canyon.
  2115) 
  2116)        road_a(fl)              = 0.0_r8
  2117)        road_r(fl)              = 0.0_r8
  2118)        road_e(fl)              = 0.0_r8
  2119)        if ( wtroad_imperv(fl) > 0.0_r8 ) then
  2120)           improad_a(fl)           =     em_improad(fl)  * lwdown_road(fl) 
  2121)           improad_r(fl)           = (1._r8-em_improad(fl)) * lwdown_road(fl) 
  2122)           improad_r_sky(fl)       = improad_r(fl) * vf_sr(l)
  2123)           improad_r_sunwall(fl)   = improad_r(fl) * vf_wr(l)
  2124)           improad_r_shadewall(fl) = improad_r(fl) * vf_wr(l)
  2125)           improad_e(fl)           = em_improad(fl) * sb * (t_improad(fl)**4) 
  2126)           improad_e_sky(fl)       = improad_e(fl) * vf_sr(l)
  2127)           improad_e_sunwall(fl)   = improad_e(fl) * vf_wr(l)
  2128)           improad_e_shadewall(fl) = improad_e(fl) * vf_wr(l)
  2129)           road_a(fl)              = road_a(fl) + improad_a(fl)*wtroad_imperv(fl)
  2130)           road_r(fl)              = road_r(fl) + improad_r(fl)*wtroad_imperv(fl)
  2131)           road_e(fl)              = road_e(fl) + improad_e(fl)*wtroad_imperv(fl)
  2132)        end if
  2133) 
  2134)        if ( wtroad_perv(fl)   > 0.0_r8 ) then
  2135)           perroad_a(fl)           =     em_perroad(fl)  * lwdown_road(fl)
  2136)           perroad_r(fl)           = (1._r8-em_perroad(fl)) * lwdown_road(fl)
  2137)           perroad_r_sky(fl)       = perroad_r(fl) * vf_sr(l)
  2138)           perroad_r_sunwall(fl)   = perroad_r(fl) * vf_wr(l)
  2139)           perroad_r_shadewall(fl) = perroad_r(fl) * vf_wr(l)
  2140)           perroad_e(fl)           = em_perroad(fl) * sb * (t_perroad(fl)**4) 
  2141)           perroad_e_sky(fl)       = perroad_e(fl) * vf_sr(l)
  2142)           perroad_e_sunwall(fl)   = perroad_e(fl) * vf_wr(l)
  2143)           perroad_e_shadewall(fl) = perroad_e(fl) * vf_wr(l)
  2144)           road_a(fl)              = road_a(fl) + perroad_a(fl)*wtroad_perv(fl)
  2145)           road_r(fl)              = road_r(fl) + perroad_r(fl)*wtroad_perv(fl)
  2146)           road_e(fl)              = road_e(fl) + perroad_e(fl)*wtroad_perv(fl)
  2147)        end if
  2148) 
  2149)        road_r_sky(fl)          = road_r(fl) * vf_sr(l)
  2150)        road_r_sunwall(fl)      = road_r(fl) * vf_wr(l)
  2151)        road_r_shadewall(fl)    = road_r(fl) * vf_wr(l)
  2152)        road_e_sky(fl)          = road_e(fl) * vf_sr(l)
  2153)        road_e_sunwall(fl)      = road_e(fl) * vf_wr(l)
  2154)        road_e_shadewall(fl)    = road_e(fl) * vf_wr(l)
  2155) 
  2156)        sunwall_a(fl)           = em_wall(fl) * lwdown_sunwall(fl)
  2157)        sunwall_r(fl)           = (1._r8-em_wall(fl)) * lwdown_sunwall(fl)
  2158)        sunwall_r_sky(fl)       = sunwall_r(fl) * vf_sw(l)
  2159)        sunwall_r_road(fl)      = sunwall_r(fl) * vf_rw(l)
  2160)        sunwall_r_shadewall(fl) = sunwall_r(fl) * vf_ww(l)
  2161)        sunwall_e(fl)           = em_wall(fl) * sb * (t_sunwall(fl)**4) 
  2162)        sunwall_e_sky(fl)       = sunwall_e(fl) * vf_sw(l)
  2163)        sunwall_e_road(fl)      = sunwall_e(fl) * vf_rw(l)
  2164)        sunwall_e_shadewall(fl) = sunwall_e(fl) * vf_ww(l)
  2165) 
  2166)        shadewall_a(fl)         = em_wall(fl) * lwdown_shadewall(fl)
  2167)        shadewall_r(fl)         = (1._r8-em_wall(fl)) * lwdown_shadewall(fl)
  2168)        shadewall_r_sky(fl)     = shadewall_r(fl) * vf_sw(l)
  2169)        shadewall_r_road(fl)    = shadewall_r(fl) * vf_rw(l)
  2170)        shadewall_r_sunwall(fl) = shadewall_r(fl) * vf_ww(l)
  2171)        shadewall_e(fl)         = em_wall(fl) * sb * (t_shadewall(fl)**4) 
  2172)        shadewall_e_sky(fl)     = shadewall_e(fl) * vf_sw(l)
  2173)        shadewall_e_road(fl)    = shadewall_e(fl) * vf_rw(l)
  2174)        shadewall_e_sunwall(fl) = shadewall_e(fl) * vf_ww(l)
  2175) 
  2176)        ! initialize sum of net and upward longwave radiation for road and both walls
  2177) 
  2178)        if ( wtroad_imperv(fl) > 0.0_r8 ) lwnet_improad(fl)   = improad_e(fl)   - improad_a(fl)
  2179)        if ( wtroad_perv(fl)   > 0.0_r8 ) lwnet_perroad(fl)   = perroad_e(fl)   - perroad_a(fl)
  2180)        lwnet_sunwall(fl)   = sunwall_e(fl)   - sunwall_a(fl)
  2181)        lwnet_shadewall(fl) = shadewall_e(fl) - shadewall_a(fl)
  2182) 
  2183)        if ( wtroad_imperv(fl) > 0.0_r8 ) lwup_improad(fl)   = improad_r_sky(fl)   + improad_e_sky(fl)
  2184)        if ( wtroad_perv(fl)   > 0.0_r8 ) lwup_perroad(fl)   = perroad_r_sky(fl)   + perroad_e_sky(fl)
  2185)        lwup_sunwall(fl)   = sunwall_r_sky(fl)   + sunwall_e_sky(fl)
  2186)        lwup_shadewall(fl) = shadewall_r_sky(fl) + shadewall_e_sky(fl)
  2187) 
  2188)     end do
  2189) 
  2190)     ! now account for absorption and reflection within canyon of fluxes from road and walls 
  2191)     ! allowing for multiple reflections
  2192)     !
  2193)     ! (1) absorption and reflection. note: emission from road and walls absorbed by walls and roads
  2194)     !     only occurs in first iteration. zero out for later iterations.
  2195)     !
  2196)     !     road: fluxes from walls need to be projected to ground area
  2197)     !     wall: fluxes from road need to be projected to wall area
  2198)     !
  2199)     ! (2) add net longwave for ith reflection to total net longwave
  2200)     !
  2201)     ! (3) distribute reflected radiation to sky, road, and walls according to view factors
  2202)     !
  2203)     ! (4) add upward longwave radiation to sky from road and walls for ith reflection to total
  2204)     !
  2205)     ! (5) stop iteration when absorption for ith reflection is less than some nominal amount. 
  2206)     !     small convergence criteria is required to ensure radiation is conserved
  2207) 
  2208)     do fl = 1,num_urbanl
  2209)        l = filter_urbanl(fl)
  2210) 
  2211)        do iter = 1, n
  2212)           ! step (1)
  2213) 
  2214)           lwtot(fl) =  (sunwall_r_road(fl) + sunwall_e_road(fl)  &
  2215)                        + shadewall_r_road(fl) + shadewall_e_road(fl))*canyon_hwr(fl)
  2216)           road_a(fl)    = 0.0_r8
  2217)           road_r(fl)    = 0.0_r8
  2218)           if ( wtroad_imperv(fl) > 0.0_r8 ) then
  2219)              improad_r(fl) = (1._r8-em_improad(fl)) * lwtot(fl)
  2220)              improad_a(fl) =     em_improad(fl)  * lwtot(fl)
  2221)              road_a(fl)    = road_a(fl) + improad_a(fl)*wtroad_imperv(fl)
  2222)              road_r(fl)    = road_r(fl) + improad_r(fl)*wtroad_imperv(fl)
  2223)           end if
  2224)           if ( wtroad_perv(fl)   > 0.0_r8 ) then
  2225)              perroad_r(fl) = (1._r8-em_perroad(fl)) * lwtot(fl) 
  2226)              perroad_a(fl) =     em_perroad(fl)  * lwtot(fl) 
  2227)              road_a(fl)    = road_a(fl) + perroad_a(fl)*wtroad_perv(fl)
  2228)              road_r(fl)    = road_r(fl) + perroad_r(fl)*wtroad_perv(fl)
  2229)           end if
  2230) 
  2231)           lwtot(fl) = (road_r_sunwall(fl) + road_e_sunwall(fl))/canyon_hwr(fl) &
  2232)                       + (shadewall_r_sunwall(fl) + shadewall_e_sunwall(fl))
  2233)           sunwall_a(fl) =     em_wall(fl)  * lwtot(fl)
  2234)           sunwall_r(fl) = (1._r8-em_wall(fl)) * lwtot(fl)
  2235) 
  2236)           lwtot(fl) = (road_r_shadewall(fl) + road_e_shadewall(fl))/canyon_hwr(fl) &
  2237)                       + (sunwall_r_shadewall(fl) + sunwall_e_shadewall(fl))
  2238)           shadewall_a(fl) =     em_wall(fl)  * lwtot(fl)
  2239)           shadewall_r(fl) = (1._r8-em_wall(fl)) * lwtot(fl)
  2240) 
  2241)           sunwall_e_road(fl)      = 0._r8
  2242)           shadewall_e_road(fl)    = 0._r8
  2243)           road_e_sunwall(fl)      = 0._r8
  2244)           shadewall_e_sunwall(fl) = 0._r8
  2245)           road_e_shadewall(fl)    = 0._r8
  2246)           sunwall_e_shadewall(fl) = 0._r8
  2247) 
  2248)           ! step (2)
  2249) 
  2250)           if ( wtroad_imperv(fl) > 0.0_r8 ) lwnet_improad(fl)   = lwnet_improad(fl)   - improad_a(fl)
  2251)           if ( wtroad_perv(fl)   > 0.0_r8 ) lwnet_perroad(fl)   = lwnet_perroad(fl)   - perroad_a(fl)
  2252)           lwnet_sunwall(fl)   = lwnet_sunwall(fl)   - sunwall_a(fl)
  2253)           lwnet_shadewall(fl) = lwnet_shadewall(fl) - shadewall_a(fl)
  2254) 
  2255)           ! step (3)
  2256) 
  2257)           if ( wtroad_imperv(fl) > 0.0_r8 ) then
  2258)              improad_r_sky(fl)       = improad_r(fl) * vf_sr(l)
  2259)              improad_r_sunwall(fl)   = improad_r(fl) * vf_wr(l)
  2260)              improad_r_shadewall(fl) = improad_r(fl) * vf_wr(l)
  2261)           end if
  2262) 
  2263)           if ( wtroad_perv(fl)   > 0.0_r8 ) then
  2264)              perroad_r_sky(fl)       = perroad_r(fl) * vf_sr(l)
  2265)              perroad_r_sunwall(fl)   = perroad_r(fl) * vf_wr(l)
  2266)              perroad_r_shadewall(fl) = perroad_r(fl) * vf_wr(l)
  2267)           end if
  2268) 
  2269)           road_r_sky(fl)          = road_r(fl) * vf_sr(l)
  2270)           road_r_sunwall(fl)      = road_r(fl) * vf_wr(l)
  2271)           road_r_shadewall(fl)    = road_r(fl) * vf_wr(l)
  2272) 
  2273)           sunwall_r_sky(fl)       = sunwall_r(fl) * vf_sw(l)
  2274)           sunwall_r_road(fl)      = sunwall_r(fl) * vf_rw(l)
  2275)           sunwall_r_shadewall(fl) = sunwall_r(fl) * vf_ww(l)
  2276) 
  2277)           shadewall_r_sky(fl)     = shadewall_r(fl) * vf_sw(l)
  2278)           shadewall_r_road(fl)    = shadewall_r(fl) * vf_rw(l)
  2279)           shadewall_r_sunwall(fl) = shadewall_r(fl) * vf_ww(l)
  2280) 
  2281)           ! step (4)
  2282) 
  2283)           if ( wtroad_imperv(fl) > 0.0_r8 ) lwup_improad(fl)   = lwup_improad(fl)   + improad_r_sky(fl)
  2284)           if ( wtroad_perv(fl)   > 0.0_r8 ) lwup_perroad(fl)   = lwup_perroad(fl)   + perroad_r_sky(fl)
  2285)           lwup_sunwall(fl)   = lwup_sunwall(fl)   + sunwall_r_sky(fl)
  2286)           lwup_shadewall(fl) = lwup_shadewall(fl) + shadewall_r_sky(fl)
  2287) 
  2288)           ! step (5)
  2289) 
  2290)           crit = max(road_a(fl), sunwall_a(fl), shadewall_a(fl))
  2291)           if (crit < .001_r8) exit
  2292)        end do
  2293)        if (iter >= n) then
  2294)           write (iulog,*) 'urban net longwave radiation error: no convergence'
  2295)           write (iulog,*) 'clm model is stopping'
  2296)           call endrun
  2297)        endif
  2298) 
  2299)        ! total net longwave radiation for canyon. project wall fluxes to horizontal surface
  2300) 
  2301)        lwnet_canyon(fl) = 0.0_r8
  2302)        if ( wtroad_imperv(fl) > 0.0_r8 ) lwnet_canyon(fl) = lwnet_canyon(fl) + lwnet_improad(fl)*wtroad_imperv(fl)
  2303)        if ( wtroad_perv(fl)   > 0.0_r8 ) lwnet_canyon(fl) = lwnet_canyon(fl) + lwnet_perroad(fl)*wtroad_perv(fl)
  2304)        lwnet_canyon(fl) = lwnet_canyon(fl) + (lwnet_sunwall(fl) + lwnet_shadewall(fl))*canyon_hwr(fl)
  2305) 
  2306)        ! total emitted longwave for canyon. project wall fluxes to horizontal
  2307) 
  2308)        lwup_canyon(fl) = 0.0_r8
  2309)        if( wtroad_imperv(fl) > 0.0_r8 ) lwup_canyon(fl) = lwup_canyon(fl) + lwup_improad(fl)*wtroad_imperv(fl)
  2310)        if( wtroad_perv(fl)   > 0.0_r8 ) lwup_canyon(fl) = lwup_canyon(fl) + lwup_perroad(fl)*wtroad_perv(fl)
  2311)        lwup_canyon(fl) = lwup_canyon(fl) + (lwup_sunwall(fl) + lwup_shadewall(fl))*canyon_hwr(fl)
  2312) 
  2313)        ! conservation check. note: previous conservation check confirms partioning of incident
  2314)        ! atmospheric longwave radiation to road and walls is conserved as
  2315)        ! lwdown (from atmosphere) = lwdown_improad + lwdown_perroad + (lwdown_sunwall + lwdown_shadewall)*canyon_hwr
  2316) 
  2317)        err = lwnet_canyon(fl) - (lwup_canyon(fl) - lwdown(fl))
  2318)        if (abs(err) > .10_r8 ) then
  2319)           write (iulog,*) 'urban net longwave radiation balance error',err
  2320)           write (iulog,*) 'clm model is stopping'
  2321)           call endrun()
  2322)        end if
  2323) 
  2324)     end do
  2325) 
  2326)     ! Net longwave radiation for roof
  2327)     
  2328)     do l = 1,num_urbanl
  2329)        lwup_roof(l) = em_roof(l)*sb*(t_roof(l)**4) + (1._r8-em_roof(l))*lwdown(l)
  2330)        lwnet_roof(l) = lwup_roof(l) - lwdown(l)
  2331)     end do
  2332) 
  2333)   end subroutine net_longwave
