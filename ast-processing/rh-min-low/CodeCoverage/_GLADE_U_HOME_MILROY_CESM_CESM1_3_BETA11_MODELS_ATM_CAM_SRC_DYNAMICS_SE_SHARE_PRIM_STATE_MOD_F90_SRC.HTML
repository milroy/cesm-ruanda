   449)        
   450)        do ie=nets,nete
   451)           tmp(:,:,ie)=elem(ie)%accum%IEner_wet(:,:,n)
   452)        enddo
   453)        IEner_wet(n) = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   454)        IEner_wet(n) = IEner_wet(n)*scale
   455)        
   456)        do ie=nets,nete
   457)           tmp(:,:,ie)=elem(ie)%accum%KEner(:,:,n)
   458)        enddo
   459)        KEner(n) = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   460)        KEner(n) = KEner(n)*scale
   461)        
   462)        do ie=nets,nete
   463)           tmp(:,:,ie)=elem(ie)%accum%PEner(:,:,n)
   464)        enddo
   465)        PEner(n) = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   466)        PEner(n) = PEner(n)*scale
   467)        TOTE(n)=IEner(n)+PEner(n)+KEner(n)
   468)        
   469)        
   470)        do q=1,qsize
   471)           do ie=nets,nete
   472)              tmp(:,:,ie)=elem(ie)%accum%Qvar(:,:,q,n)
   473)           enddo
   474)           Qvar(q,n) = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   475)           Qvar(q,n) = Qvar(q,n)*scale
   476)           
   477)           do ie=nets,nete
   478)              tmp(:,:,ie)=elem(ie)%accum%Qmass(:,:,q,n)
   479)           enddo
   480)           Qmass(q,n) = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   481)           Qmass(q,n) = Qmass(q,n)*scale
   482)           
   483)           if (n==2) then
   484)              do ie=nets,nete
   485)                 tmp(:,:,ie)=elem(ie)%accum%Q1mass(:,:,q)
   486)              enddo
   487)              Q1mass(q) = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   488)              Q1mass(q) = Q1mass(q)*scale
   489)           endif
   490)        enddo
   491)        
   492)     enddo
   493)     
   494) 
   495) 
   496)     !
   497)     !   All of these transport terms are at time-tstep = (time1+time2)/2
   498)     !   Vertical transport terms
   499) #ifdef ENERGY_DIAGNOSTICS
   500)     do ie=nets,nete
   501)        tmp(:,:,ie) = elem(ie)%accum%KEvert1 + elem(ie)%accum%KEvert2
   502)     enddo
   503)     KEvert = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   504)     KEvert = KEvert*scale
   505)     do ie=nets,nete
   506)        tmp(:,:,ie) = elem(ie)%accum%IEvert1 + elem(ie)%accum%IEvert2
   507)     enddo
   508)     IEvert = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   509)     IEvert = IEvert*scale
   510) 
   511)     do ie=nets,nete
   512)        tmp(:,:,ie) = elem(ie)%accum%KEhorz1 + elem(ie)%accum%KEhorz2
   513)     enddo
   514)     KEhorz = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   515)     KEhorz = KEhorz*scale
   516)     do ie=nets,nete
   517)        tmp(:,:,ie) = elem(ie)%accum%IEhorz1 + elem(ie)%accum%IEhorz2
   518)     enddo
   519)     IEhorz = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   520)     IEhorz = IEhorz*scale
   521) 
   522)     do ie=nets,nete
   523)        tmp(:,:,ie) = elem(ie)%accum%IEhorz1_wet + elem(ie)%accum%IEhorz2_wet
   524)     enddo
   525)     IEhorz_wet = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   526)     IEhorz_wet = IEhorz_wet*scale
   527) 
   528)     do ie=nets,nete
   529)        tmp(:,:,ie) = elem(ie)%accum%IEvert1_wet + elem(ie)%accum%IEvert2_wet
   530)     enddo
   531)     IEvert_wet = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   532)     IEvert_wet = IEvert_wet*scale
   533) 
   534)     !   KE->PE
   535)     do ie=nets,nete
   536)        tmp(:,:,ie) = elem(ie)%accum%T1
   537)     enddo
   538)     T1 = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   539)     T1 = T1*scale
   540) 
   541)     do ie=nets,nete
   542)        tmp(:,:,ie) = elem(ie)%accum%T2
   543)     enddo
   544)     T2 = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   545)     T2 = T2*scale
   546) 
   547)     do ie=nets,nete
   548)        tmp(:,:,ie) = elem(ie)%accum%T2_s
   549)     enddo
   550)     T2_s = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   551)     T2_s = T2_s*scale
   552)     T2_m = T2 - T2_s
   553) 
   554)     do ie=nets,nete
   555)        tmp(:,:,ie) = elem(ie)%accum%S1
   556)     enddo
   557)     S1 = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   558)     S1 = S1*scale
   559) 
   560)     do ie=nets,nete
   561)        tmp(:,:,ie) = elem(ie)%accum%S1_wet
   562)     enddo
   563)     S1_wet = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   564)     S1_wet = S1_wet*scale
   565) 
   566)     do ie=nets,nete
   567)        tmp(:,:,ie) = elem(ie)%accum%S2
   568)     enddo
   569)     S2 = global_integral(elem, tmp(:,:,nets:nete),hybrid,npts,nets,nete)
   570)     S2 = S2*scale
   571) 
   572) 
   573) #else
   574)     T1=0; T2=0; T2_s=0; T2_m=0; S1_wet = 0; S1=0; S2=0; KEvert=0; IEvert=0; KEhorz=0; IEhorz=0
   575) #endif
   576) 
   577) 
   578)     if(hybrid%par%masterproc .and. hybrid%ithr==0) then 
   579)        if(moisture /= "dry")then
   580)           if (qsize>=1) then
   581)              write(iulog,'(a,E23.15,a,E23.15,a)') "    dry M = ",Mass-Q1mass(1),' kg/m^2'
   582)           endif
   583)        endif
   584)        
   585)        ! note: diagnostics not yet coded for semi-implicit 
   586)        if (integration == "explicit") then
   587)           
   588)           write(iulog,'(3a25)') "**DYNAMICS**        J/m^2","   W/m^2","W/m^2    "
   589) #ifdef ENERGY_DIAGNOSTICS
   590)           ! terms computed during prim_advance, if ENERGY_DIAGNOSTICS is enabled
   591)           write(iulog,'(a,2e22.14)') 'Tot KE advection horiz, vert:      ',-KEhorz,-KEvert
   592)           write(iulog,'(a,2e22.14)') 'Tot IE dry advection horiz, vert:  ',-IEhorz,-IEvert
   593)           if (use_cpstar==1) then
   594)              write(iulog,'(a,2e22.14)') 'Tot IE wet advection horiz, vert:  ',-IEhorz_wet,-IEvert_wet
   595)           endif
   596)           
   597)           write(iulog,'(a,2e22.14)') 'Transfer:   KE->IE, KE->PE:        ',-(T1+T2_m), -T2_s
   598)           write(iulog,'(a,2e22.14)') 'Transfer:   IE->KE, PE->KE:        ',-S1,-S2
   599)           if (use_cpstar==1) then
   600)              write(iulog,'(a,2e22.14)') 'Transfer:   IE->KE(wet only):      ',-S1_wet
   601)           endif
   602)           
   603)           ddt_tot =  (KEner(2)-KEner(1))/(dt)
   604)           ddt_diss = ddt_tot -(T1+T2) 
   605)           write(iulog,'(a,3E22.14)') "KE,d/dt,diss:",KEner(2),ddt_tot,ddt_diss
   606)           
   607)           if (use_cpstar==1) then
   608)              ddt_tot =  ( (IEner(2)-IEner_wet(2)) - (IEner(1)-IEner_wet(1))  )/dt
   609)              ddt_diss = ddt_tot - (S1 -S1_wet)
   610)              write(iulog,'(a,3E22.14,a)') "IE,d/dt,diss:",IEner(2)-IEner_wet(2),ddt_tot,ddt_diss,' (dry)'
   611)              ddt_tot =  ( IEner_wet(2) - IEner_wet(1) )/dt 
   612)              ddt_diss =  ddt_tot - S1_wet
   613)              write(iulog,'(a,3E22.14,a)') "IE,d/dt,diss:",IEner_wet(2),ddt_tot,ddt_diss,' (wet)'
   614)           else
   615)              ddt_tot =  (IEner(2)-IEner(1))/(dt)
   616)              ddt_diss = ddt_tot - S1 
   617)              write(iulog,'(a,3E22.14)') "IE,d/dt,diss:",IEner(2),ddt_tot,ddt_diss
   618)           endif
   619)           
   620)           ddt_tot = (PEner(2)-PEner(1))/(dt)
   621)           ddt_diss = ddt_tot - S2
   622)           write(iulog,'(a,3E22.14)') "PE,d/dt,diss:",PEner(2),ddt_tot,ddt_diss
   623) #else
   624)           write(iulog,'(a,3E22.14)') "KE,d/dt      ",KEner(2),(KEner(2)-KEner(1))/(dt)
   625)           write(iulog,'(a,3E22.14)') "IE,d/dt      ",IEner(2),(IEner(2)-IEner(1))/(dt)
   626)           write(iulog,'(a,3E22.14)') "PE,d/dt      ",PEner(2),(PEner(2)-PEner(1))/(dt)
   627) #endif
   628)           ddt_tot = (TOTE(2)-TOTE(1))/(dt)
   629)           write(iulog,'(a,3E22.14)') " E,dE/dt     ",TOTE(2),ddt_tot
   630)           
   631)           if (tstep_type>0) then  !no longer support tracer advection with tstep_type = 0
   632)              do q=1,qsize
   633)                 write(iulog,'(a,i1,a,E22.14,a,2E15.7)') "Q",q,",Q diss, dQ^2/dt:",Qmass(q,2)," kg/m^2",&
   634)                      (Qmass(q,2)-Qmass(q,1))/dt,(Qvar(q,2)-Qvar(q,1))/dt
   635)              enddo
   636)           endif
   637) 
   638)           ! LF code diagnostics
   639)           if (tstep_type==0) then  ! leapfrog
   640)              write(iulog,'(a)') 'Robert filter, Physics (except adjustments):'
   641)              write(iulog,'(a,2e15.7)') 'dKE/dt(W/m^2): ',(KEner(4)-KEner(3))/dt,(KEner(3)-KEner(2))/dt
   642)              write(iulog,'(a,2e15.7)') 'dIE/dt(W/m^2): ',(IEner(4)-IEner(3))/dt,(IEner(3)-IEner(2))/dt
   643)              write(iulog,'(a,2e15.7)') 'dPE/dt(W/m^2): ',(PEner(4)-PEner(3))/dt,(PEner(3)-PEner(2))/dt
   644)           else
   645)              write(iulog,'(a)') 'Energy Fixer, Physics (except adjustments):'
   646)              write(iulog,'(a,2e15.7)') 'dKE/dt(W/m^2): ',(KEner(3)-KEner(2))/dt,(KEner(1)-KEner(4))/dt
   647)              write(iulog,'(a,2e15.7)') 'dIE/dt(W/m^2): ',(IEner(3)-IEner(2))/dt,(IEner(1)-IEner(4))/dt
   648)              write(iulog,'(a,2e15.7)') 'dPE/dt(W/m^2): ',(PEner(3)-PEner(2))/dt,(PEner(1)-PEner(4))/dt
   649)           endif
   650)        endif
   651) 
   652)        ! Print change in energy and tracer mass from the start of the simulation
   653) #ifdef CAM
   654)        ! for cam, this information only makes sense for short test runs with
   655)        ! physics disabled (ftype<0).  So disable this diagnostic if ftype>=0:
   656)        if (ftype>=0) TOTE0=-1  
   657) #endif       
   658)        if (TOTE0>0) then
   659)           write(iulog,100) "(E-E0)/E0    ",(TOTE(4)-TOTE0)/TOTE0
   660)           if (tstep_type>0) then  !no longer support tracer advection with tstep_type = 0
   661)              do q=1,qsize
   662)                 if(Qmass0(q)>0.0D0) then
   663)                    write(iulog,'(a,E23.15,a,i1)') "(Q-Q0)/Q0 ",(Qmass(q,2)-Qmass0(q))/Qmass0(q),"   Q",q
   664)                 end if
   665)              enddo
   666)           endif
   667)        endif
   668)     endif
   669)     
   670) 100 format (A10,3(E23.15))
   671)     
   672)     
   673)     ! initialize "E0" for printout of E-E0/E0
   674)     ! energy is computed in timestep loop.  Save the value 
   675)     ! after the first real timestep, not the half-timesteps used
   676)     ! to bootstrap leapfrog: 
   677)     if (tl%nstep >= tl%nstep0 .and. TOTE0==0 ) then  
   678)        TOTE0=TOTE(4)
   679)        do q=1,qsize
   680)           Qmass0(q)=Qmass(q,1)
   681)        enddo
   682)        time0=time1
   683)     endif
   684) 
   685)   end subroutine prim_printstate

   691) 
   692) subroutine prim_energy_halftimes(elem,hvcoord,tl,n,t_before_advance,nets,nete)
   693) ! 
   694) !  called at the end of a timestep, before timelevel update.  Solution known at
   695) !  dynamics:     nm1,  n0,  np1.  
   696) !
   697) !                                                           LF code            RK code
   698) !  This routine is called 4 times:  n=1:    t1=nm1, t2=n0   start              after forcing, before timestep
   699) !                                   n=2:    t1=n0, t2=np1   after timestep     after timestep, including remap
   700) !                                   n=3:    t1=n0, t2=np1   after forcing      after fixer 
   701) !                                   n=4:    t1=n0, t2=np1   after Robert       before forcing
   702) !
   703) ! LF case we use staggered formulas:
   704) !  compute the energies at time half way between timelevels t1 and t2,
   705) !  using timelevels t1,t2.  store in location n of our elem()%accum%* variables
   706) !    
   707) !  in the Dry case, PE (and similar for IE) will be conserved exactly by Leapfrog, 
   708) !  if it is defined as:
   709) !
   710) !   PE(n+.5)  = .5*(  T(n+1) dp(n) + T(n) dp(n+1) )
   711) !
   712) !  KE cannot be exactly conserved with Leapfrog.  but we still use:
   713) !  
   714) !   KE(n+.5) = .5*(  .5 u(n+1)^2 dp(n) +  .5 u(n)^2 dp(n+1) )
   715) !
   716) !
   717)     use kinds, only : real_kind
   718)     use dimensions_mod, only : np, np, nlev
   719)     use control_mod, only : use_cpstar
   720)     use hybvcoord_mod, only : hvcoord_t
   721)     use element_mod, only : element_t
   722)     use physical_constants, only : Cp, cpwater_vapor
   723)     use physics_mod, only : Virtual_Specific_Heat, Virtual_Temperature
   724)     use prim_si_mod, only : preq_hydrostatic
   725) 
   726)     integer :: t1,t2,n,nets,nete
   727)     type (element_t)     , intent(inout), target :: elem(:)
   728)     type (hvcoord_t)                  :: hvcoord
   729)     type (TimeLevel_t), intent(in)       :: tl
   730)     logical :: t_before_advance
   731) 
   732)     integer :: ie,k,i,j,nm_f
   733)     real (kind=real_kind), dimension(np,np,nlev)  :: dpt1,dpt2   ! delta pressure
   734)     real (kind=real_kind), dimension(np,np)  :: E
   735)     real (kind=real_kind), dimension(np,np)  :: suml,suml2,v1,v2
   736)     real (kind=real_kind), dimension(np,np,nlev)  :: sumlk, suml2k
   737)     real (kind=real_kind), dimension(np,np,nlev)  :: p,T_v,phi
   738)     real (kind=real_kind) :: cp_star1,cp_star2,qval_t1,qval_t2
   739)     real (kind=real_kind) :: Qt
   740)     logical :: wet
   741) 
   742) 
   743)     logical tstagger
   744)     integer:: t2_qdp, t1_qdp   ! the time pointers for Qdp are not the same
   745) 
   746)     nm_f = 1
   747)     if (t_before_advance) then
   748)        t1=tl%nm1
   749)        t2=tl%n0
   750)        call TimeLevel_Qdp( tl, qsplit, t2_qdp, t1_qdp) !get n0 level into t2_qdp 
   751)     else
   752)        t1=tl%n0
   753)        t2=tl%np1
   754)        call TimeLevel_Qdp(tl, qsplit, t1_qdp, t2_qdp) !get np1 into t2_qdp
   755)     endif
   756) 
   757) 
   758) 
   759) ! energy_fixer
   760) !     <0         disabled, but compute energy non-staggered in time
   761) !     0          disabled, but compute energy staggered in time (for use with leapfrog code)
   762) !    >0          Enabled.  energy fixer requires dry formulation (use_cpstar=0) and non-staggered in time
   763) !
   764)     tstagger = .false.
   765)     if (energy_fixer==0) tstagger = .true.
   766) 
   767) 
   768)     !   IE   Cp*dpdn*T  + (Cpv-Cp) Qdpdn*T
   769)     !        Cp*dpdn(n)*T(n+1) + (Cpv-Cp) Qdpdn(n)*T(n+1)
   770)     !        [Cp + (Cpv-Cp) Q(n)] *dpdn(n)*T(n+1) 
   771)     do ie=nets,nete
   772) 
   773) #if (defined ELEMENT_OPENMP)
   774) !$omp parallel do private(k)
   775) #endif
   776)        do k=1,nlev
   777)           dpt1(:,:,k) = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
   778)                ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*elem(ie)%state%ps_v(:,:,t1)
   779)           dpt2(:,:,k) = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
   780)                ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*elem(ie)%state%ps_v(:,:,t2)
   781)        enddo
   782) 
   783) #if (defined ELEMENT_OPENMP)
   784) !$omp parallel do private(k,i,j,cp_star1,cp_star2,qval_t1,qval_t2)
   785) #endif
   786)        do k=1,nlev
   787)           sumlk(:,:,k)=0
   788)           suml2k(:,:,k)=0
   789)           do i=1,np
   790)           do j=1,np
   791)              if(use_cpstar == 1)  then
   792)                 ! Cp_star = cp + (Cpwater_vapor - cp)*qval
   793)                 qval_t1 = elem(ie)%state%Qdp(i,j,k,1,t1_qdp)/dpt1(i,j,k)
   794)                 qval_t2 = elem(ie)%state%Qdp(i,j,k,1,t2_qdp)/dpt2(i,j,k)
   795)                 cp_star1= Virtual_Specific_Heat(qval_t1)
   796)                 cp_star2= Virtual_Specific_Heat(qval_t2)
   797)              else
   798)                 cp_star1=cp
   799)                 cp_star2=cp
   800)              endif
   801)              if (tstagger) then
   802)                 sumlk(i,j,k) = sumlk(i,j,k) + cp_star1*elem(ie)%state%T(i,j,k,t2) *dpt1(i,j,k)/2 
   803)                 sumlk(i,j,k) = sumlk(i,j,k) + Cp_star2*elem(ie)%state%T(i,j,k,t1) *dpt2(i,j,k)/2
   804)                 suml2k(i,j,k) = suml2k(i,j,k) + (cp_star1-cp)*elem(ie)%state%T(i,j,k,t2) *dpt1(i,j,k)/2 
   805)                 suml2k(i,j,k) = suml2k(i,j,k) + (cp_star2-cp)*elem(ie)%state%T(i,j,k,t1) *dpt2(i,j,k)/2
   806)              else
   807)                 sumlk(i,j,k) = sumlk(i,j,k) + Cp_star2*elem(ie)%state%T(i,j,k,t2) *dpt2(i,j,k)
   808)                 suml2k(i,j,k) = suml2k(i,j,k) + (cp_star2-cp)*elem(ie)%state%T(i,j,k,t2) *dpt2(i,j,k)
   809)              endif
   810)           enddo
   811)           enddo
   812)        enddo
   813)        suml=0
   814)        suml2=0
   815)        do k=1,nlev
   816)           suml(:,:) = suml(:,:) + sumlk(:,:,k)
   817)           suml2(:,:) = suml2(:,:) + suml2k(:,:,k)
   818)        enddo
   819)        elem(ie)%accum%IEner(:,:,n)=suml(:,:)
   820)        elem(ie)%accum%IEner_wet(:,:,n)=suml2(:,:)
   821) 
   822)     
   823)     !   KE   .5 dp/dn U^2
   824) #if (defined ELEMENT_OPENMP)
   825) !$omp parallel do private(k,E)
   826) #endif
   827)        do k=1,nlev
   828)           if (tstagger) then
   829)              E = (elem(ie)%state%v(:,:,1,k,t2)**2 +  &
   830)                   elem(ie)%state%v(:,:,2,k,t2)**2 ) / 2 
   831)              sumlk(:,:,k) = E*dpt1(:,:,k)/2
   832)              E = (elem(ie)%state%v(:,:,1,k,t1)**2 +  &
   833)                   elem(ie)%state%v(:,:,2,k,t1)**2 ) / 2 
   834)              sumlk(:,:,k) = sumlk(:,:,k) + E*dpt2(:,:,k)/2
   835)           else
   836)              E = (elem(ie)%state%v(:,:,1,k,t2)**2 +  &
   837)                   elem(ie)%state%v(:,:,2,k,t2)**2 ) / 2 
   838)              sumlk(:,:,k) = E*dpt2(:,:,k)
   839)           endif
   840)        enddo
   841)        suml=0
   842)        do k=1,nlev
   843)           suml(:,:) = suml(:,:) + sumlk(:,:,k)
   844)        enddo
   845)        elem(ie)%accum%KEner(:,:,n)=suml(:,:)
   846) 
   847) 
   848)     
   849)     !   PE   dp/dn PHIs
   850)        suml=0
   851)        do k=1,nlev
   852)           if (tstagger) then

   886) subroutine prim_diag_scalars(elem,hvcoord,tl,n,t_before_advance,nets,nete)
   887) ! 
   888) !  called at the end of a timestep, before timelevel update.  Solution known at
   889) !  timelevel nm1,n0,np1.  
   890) !
   891) !  This routine is called twice:  n=1:    t1=nm1, t2=n0
   892) !                                 n=2:    t1=n0, t2=np1
   893) ! 
   894) !  in all cases:
   895) !     Q1mass(n) = Qdp(t2)    (or:  Q(t2)*dp(t2) )
   896) ! 
   897) !  We also compute 
   898) !     Qmass(n)  = .5*(  Q(t2) dp(t1)  Q(t1) dp(t2) )       Leapfrog, concentration formulation
   899) !     Qmass(n)  = .5*(  Qdp(t1) + Qdp(t2) )                Leapfrog, conservation formulation
   900) !     Qmass(n)  =  Qdp(t2)                                 RK2, conservation formulation
   901) !
   902) !
   903)     use kinds, only : real_kind
   904)     use dimensions_mod, only : np, np, nlev
   905)     use hybvcoord_mod, only : hvcoord_t
   906)     use element_mod, only : element_t
   907) 
   908)     integer :: t1,t2,n,nets,nete
   909)     type (element_t)     , intent(inout), target :: elem(:)
   910)     type (hvcoord_t)                  :: hvcoord
   911)

   913)     real (kind=real_kind), dimension(np,np)  :: ps         ! pressure
   914)     real (kind=real_kind), dimension(np,np)  :: dp         ! delta pressure
   915)     real (kind=real_kind), dimension(np,np)  :: E
   916)     real (kind=real_kind), dimension(np,np)  :: suml
   917)     type (TimeLevel_t), intent(in)       :: tl
   918)     logical :: t_before_advance

   920) 
   921)     nm_f = 1
   922)     if (t_before_advance) then
   923)        t1=tl%nm1     
   924)        t2=tl%n0
   925)        call TimeLevel_Qdp( tl, qsplit, t2_qdp) !get n0 level into t2_qdp 
   926)     else

   928)        t2=tl%np1
   929)        call TimeLevel_Qdp(tl, qsplit, tmp, t2_qdp) !get np1 into t2_qdp (don't need tmp)
   930)     endif
   931) 
   932) 
   933)     !
   934)     !  RK2 forward scheme.  compute everything at t2
   935)     !  (used by CAM)
   936)     !   Q has only one time dimension
   937)     if (tstep_type>0) then
   938)

   947)              enddo

   950)        enddo
   951)

   953) #if (defined ELEMENT_OPENMP)

   955) #endif
   956)           do q=1,qsize
   957)              suml=0
   958)              do k=1,nlev
   959)                 suml = suml + elem(ie)%state%Qdp(:,:,k,q,t2_qdp)
   960)              enddo
   961)              elem(ie)%accum%Q1mass(:,:,q)=suml(:,:)
   962)              elem(ie)%accum%Qmass(:,:,q,n)=suml(:,:)
