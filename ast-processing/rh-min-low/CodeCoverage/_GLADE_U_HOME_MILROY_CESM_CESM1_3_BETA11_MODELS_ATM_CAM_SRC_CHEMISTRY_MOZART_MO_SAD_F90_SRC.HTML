    93) ! DESCRIPTION
    94) !
    95) !     This routine has the logic to derive the surface area density for
    96) !     three types of aerosols: Sulfate (LBS, STS); Nitric Acid Trihydrate (NAT);
    97) !     and Water-ICE. The surface area density is stored in sad_strat(3). The
    98) !     first, second, and third dimensions are SULFATE, NAT, and ICE SAD
    99) !     respectively. The effective radius of each particle is also stored
   100) !     in radius_strat(3).
   101) !
   102) !     NOTE1: For Sulfate and H2O ICE calculations
   103) !     The Surface Area and Volume Densities are calculated from the
   104) !     second and third moment of the LOG Normal Distribution. For an example
   105) !     see Binkowski et al., JGR, 100, 26191-26209, 1995. The Volume Density
   106) !     is substituted into the SAD equation so that the SAD is dependent on
   107) !     the # of particles cm-3, the width of the distribution (sigma), and
   108) !     the volume density of the aerosol. This approach is discussed in 
   109) !     Considine et al., 2000 and Kinnison et al., 2007.
   110) !
   111) !     NOTE2: For the ternary solution calculation
   112) !     the total sulfate mass is derived from the SAGEII SAD data. This approach
   113) !     has been previously used in Considine et al., JGR, 1999. The thermodynamic
   114) !     models used in this routine are from A. Tabazedeh et al, 1994.
   115) !
   116) !     NOTE3: Updates to the PSC scheme is discussed in Wegner et al., 2013a. 
   117) !     80% of the total HNO3 is allowed to see STS, 20% NAT. The number density of 
   118) !     the NAT and ICE particles are is set to 0.01 and 0.1 particle cm-3 respectively. 
   119) !
   120) !     NOTE4: The HCl solubility (in STS) has been added and evalutede in Wegner et al., 2013b. 
   121) !     This solubility is based on Carslaw et al., 1995.
   122) !
   123) !     REFERENCES for this PSC module:
   124) !        Considine, D. B., A. R. Douglass, P. S. Connell, D. E. Kinnison, and D. A., Rotman, 
   125) !          A polar stratospheric cloud parameterization for the three dimensional model of 
   126) !          the global modeling initiative and its response to stratospheric aircraft, 
   127) !          J. Geophys. Res., 105, 3955-3975, 2000.
   128) !
   129) !        Kinnison, D. E.,et al., Sensitivity of chemical tracers to meteorological 
   130) !          parameters in the MOZART-3 chemical transport model, J. Geophys. Res., 
   131) !          112, D20302, doi:10.1029/2006JD007879, 2007.
   132) !
   133) !        Wegner, T, D. E. Kinnison, R. R. Garcia, S. Madronich, S. Solomon, and M. von Hobe, 
   134) !          On the depletion of HCl in the Antarctic polar vortex, 
   135) !          in review J. Geophys. Res., 2013.
   136) !
   137) !        Wegner, T, D. E. Kinnison, R. R. Garcia, S. Madronich, and S. Solomon, 
   138) !          Polar Stratospheric Clouds in SD-WACCM4, in review J. Geophys. Res., 2013.
   139) !
   140) !        Tabazedeh, A., R. P. Turco, K. Drdla, M. Z. Jacobson, and O. B, Toon, A study
   141) !          of the type I polar stratosphere cloud formation, 
   142) !          Geophys. Res. Lett., 21, 1619-1622, 1994.
   143) !
   144) !        Carslaw, K. S., S. L. Clegg, and P. Brimblecombe, A thermodynamic model of the
   145) !          system HCl-HNO3-H2SO4-H2O, including solubilities of HBr, from <200 to 328K, 
   146) !          J. Phys. Chem., 99, 11,557-11,574, doi:1021/100029a039, 1995.
   147) !
   148) !
   149) ! ARGUMENTS
   150) !   INPUT:
   151) !     hno3_gas		Nitric Acid gas   phase abundance (mole fraction)
   152) !     hno3_cond(2)	Nitric Acid cond. phase abundance (mole fraction)
   153) !                       (1) in STS; (2) in NAT
   154) !     h2o_cond		Water condensed phase           (mole fraction)
   155) !     h2o_gas           Water gas-phase abundance       (mole fraction)
   156) !
   157) !     hcl_gas           HCL gas-phase abundance         (mole fraction)
   158) !     hcl_cond          HCl condensed phase (STS)       (mole fraction)
   159) !
   160) !     sage_sad		SAGEII surface area density     (cm2-aer cm-3 atm)
   161) !     m                 Airdensity                      (molecules cm-3)
   162) !     press             Pressure                        (hPa)
   163) !
   164) !
   165) !   OUTPUT:
   166) !
   167) !     hno3_gas     = Gas-phase HNO3             Used in chemical solver. 
   168) !     hno3_cond(1) = Condensed HNO3 from STS    Not used in mo_aero_settling.F90
   169) !     hno3_cond(2) = Condensed HNO3 from NAT    Used in mo_aero_settling.F90
   170) !
   171) !     hcl_gas      = Gas-phase HCL              Used in chemical solver. 
   172) !     hcl_cond     = Condensed HCl from STS     
   173) !
   174) !     SAD_strat(1) = Sulfate Aerosol... Used in mo_strato_rates.F90
   175) !     SAD_strat(2) = NAT Aerosol....    Used in mo_strato_rates.F90
   176) !     SAD_strat(3) = Water-Ice......... Used in mo_strato_rates.F90
   177) !
   178) !     RAD_strat(1) = Sulfate Aerosol... Used in mo_strato_rates.F90
   179) !     RAD_strat(2) = NAT large mode.... Used in mo_aero_settling.F90
   180) !     RAD_strat(3) = Water-Ice......... Not used in mo_aero_settling.F90
   181) !
   182) !   NOTE1: The sum of hno3_cond(1-2) will be added to hno3_gas for advection of HNO3 in
   183) !          mo_gas_phase_chemdr.F90.
   184) !
   185) !   NOTE2: The sum of hcl_cond will be added to hcl_gas for advection of HCl in
   186) !          mo_gas_phase_chemdr.F90.
   187) !
   188) !   NOTE3: This routine does not partition H2O.
   189) !
   190) !
   191) ! ROUTINES Called (in and below this routine):
   192) !
   193) !       sad_strat_calc
   194) !         nat_sat_temp ...... derives the NAT saturation temp
   195) !
   196) !         sulfate_sad_calc .. Calculates the sulfate SAD;  HNO3, H2O cond. phase
   197) !         calc_radius_lbs ... Calculates the radius for a H2SO4 Binary soln. (T>200K)
   198) !         sad_to_h2so4 ...... Derives H2SO4 abundance (micrograms m-3)
   199) !                             from SAGEII SAD.
   200) !         density............ A. Tabazedeh binary thermo model
   201) !         equil.............. A. Tabazedeh ternary thermo. model
   202) !
   203) !         nat_sad_calc....... Calculates the NAT SAD; HNO3, H2O cond. phase
   204) !         nat_cond........... Derives the NAT HNO3 gas/cond partitioning
   205) !
   206) !         ice_sad_calc....... derives the ICE SAD and H2O gas/cond partitioning
   207) !===============================================================================

   209)       subroutine sad_strat_calc( lchnk, m, press, temper, hno3_gas, &
   210)                                  hno3_cond, h2o_gas, h2o_cond, hcl_gas, hcl_cond, &
   211)                                  sad_sage, radius_strat, sad_strat, ncol, pbuf )
   212) 
   213)       use cam_history, only : outfld
   214)       use physics_buffer, only : physics_buffer_desc
   215) 
   216)       implicit none
   217) 
   218) !-------------------------------------------------------------------------------
   219) !	... dummy arguments
   220) !-------------------------------------------------------------------------------
   221)       integer, intent(in)     ::  lchnk                      ! chnk id
   222)       integer, intent(in)     ::  ncol                       ! columns in chunk
   223)       real(r8), intent(in)    ::  m           (ncol,pver)    ! Air density (molecules cm-3)
   224)       real(r8), intent(in)    ::  sad_sage    (ncol,pver)    ! SAGEII surface area density (cm2 aer. cm-3 air)
   225)       real(r8), intent(in)    ::  press       (ncol,pver)    ! Pressure, hPa
   226)       real(r8), intent(in)    ::  temper      (pcols,pver)   ! Temperature (K)
   227)       real(r8), intent(inout) ::  h2o_gas     (ncol,pver)    ! H2O gas-phase (mole fraction)
   228)       real(r8), intent(inout) ::  h2o_cond    (ncol,pver)    ! H2O condensed phase  (mole fraction)
   229)       real(r8), intent(inout) ::  hno3_gas    (ncol,pver)    ! HNO3 condensed phase (mole fraction)
   230)       real(r8), intent(inout) ::  hno3_cond   (ncol,pver,2)  ! HNO3 condensed phase (mole fraction)
   231)       real(r8), intent(inout) ::  hcl_gas     (ncol,pver)    ! HCL gas-phase        (mole fraction)
   232)       real(r8), intent(inout) ::  hcl_cond    (ncol,pver)    ! HCL condensed phase  (mole fraction)
   233)       real(r8), intent(out)   ::  radius_strat(ncol,pver,3)  ! Radius of Sulfate, NAT, and ICE (cm)
   234)       real(r8), intent(out)   ::  sad_strat   (ncol,pver,3)  ! Surface area density of Sulfate, NAT, ICE (cm2 cm-3)
   235) 
   236) !-------------------------------------------------------------------------------
   237) !	... local variables
   238) !-------------------------------------------------------------------------------
   239)       real(r8), parameter :: temp_floor = 0._r8
   240) 
   241)       integer  ::  i, k, n
   242)       integer  ::  dims(1)
   243)       real(r8) ::  hno3_total    (ncol,pver)     ! HNO3 total = gas-phase + condensed
   244)       real(r8) ::  h2o_total     (ncol,pver)     ! H2O total  = gas-phase + condensed
   245)       real(r8) ::  radius_lbs    (ncol,pver)     ! Radius of Liquid Binary Sulfate (cm)
   246)       real(r8) ::  radius_sulfate(ncol,pver)     ! Radius of Sulfate aerosol (cm)
   247)       real(r8) ::  radius_nat    (ncol,pver)     ! Radius of NAT aerosol     (cm)
   248)       real(r8) ::  radius_ice    (ncol,pver)     ! Radius of ICE aerosol     (cm)
   249)       real(r8) ::  sad_nat       (ncol,pver)     ! SAD of NAT aerosol        (cm2 cm-3)
   250)       real(r8) ::  sad_sulfate   (ncol,pver)     ! SAD of Sulfate aerosol    (cm2 cm-3)
   251)       real(r8) ::  sad_ice       (ncol,pver)     ! SAD of ICE aerosol        (cm2 cm-3)
   252)       real(r8) ::  tsat_nat      (ncol,pver)     ! Temperature for NAT saturation
   253)       real(r8) ::  h2o_avail     (ncol,pver)     ! H2O temporary arrays
   254)       real(r8) ::  hno3_avail    (ncol,pver)     ! HNO3 temporary array
   255)       real(r8) ::  hno3_gas_nat  (ncol,pver)     ! HNO3 after call to NAT routines
   256)       real(r8) ::  hno3_gas_sulf (ncol,pver)     ! HNO3 after call to STS routines
   257)       real(r8) ::  hno3_cond_nat (ncol,pver)     ! HNO3 condensed after call to NAT
   258)       real(r8) ::  hno3_cond_sulf(ncol,pver)     ! HNO3 condensed after call to STS routines
   259)       real(r8) ::  hcl_total     (ncol,pver)     ! HCl total  = gas-phase + condensed
   260)       real(r8) ::  hcl_avail     (ncol,pver)     ! HCL temporary arrays
   261)       real(r8) ::  hcl_gas_sulf  (ncol,pver)     ! HCL after call to STS routines
   262)       real(r8) ::  hcl_cond_sulf (ncol,pver)     ! HCL condensed after call to STS routines
   263)       real(r8) ::  temp          (pcols,pver)    ! wrk temperature array
   264)       real(r8) ::  h2so4m        (ncol,pver)     ! wrk array
   265) 
   266)       logical  ::  z_val(ncol)
   267) 
   268)       logical  ::  mask_lbs(ncol,pver)           ! LBS mask T: P>300hPa; P<2hPa2hPa; SAGE<1e-15; T>200K
   269)       logical  ::  mask_ice(ncol,pver)           ! ICE mask T: .not. mask_lbs; h2o_cond>0
   270)       logical  ::  mask_sts(ncol,pver)           ! STS mask T: .not. mask_sts
   271)       logical  ::  mask_nat(ncol,pver)		 ! NAT mask T: mask_sts=T; T<Tsat_nat
   272)       type(physics_buffer_desc), pointer :: pbuf(:)
   273) 
   274)       real(r8), parameter :: eighty_percent = 0.8_r8
   275)       real(r8), parameter :: twenty_percent = 0.2_r8
   276) 
   277) !----------------------------------------------------------------------
   278) !     ... initialize to zero
   279) !----------------------------------------------------------------------
   280)       do n = 1,3
   281)          do k = 1,pver
   282)             radius_strat(:,k,n) = 0._r8
   283)             sad_strat(:,k,n)    = 0._r8
   284)          end do
   285)       end do
   286) !
   287)       do n = 1,2
   288)          do k = 1,pver
   289)             hno3_cond(:,k,n) = 0._r8
   290)          end do
   291)       end do
   292) !
   293)       do k = 1,pver
   294)             h2o_total     (:,k) = 0._r8
   295) !
   296)             h2o_avail     (:,k) = 0._r8
   297)             hno3_avail    (:,k) = 0._r8
   298)             hcl_avail     (:,k) = 0._r8
   299) !
   300)             hno3_total    (:,k) = 0._r8
   301)             hno3_gas_nat  (:,k) = 0._r8
   302)             hno3_gas_sulf (:,k) = 0._r8
   303)             hno3_cond_nat (:,k) = 0._r8
   304)             hno3_cond_sulf(:,k) = 0._r8
   305) !
   306)             hcl_total     (:,k) = 0._r8
   307)             hcl_cond_sulf (:,k) = 0._r8
   308)             hcl_gas_sulf  (:,k) = 0._r8
   309)       end do       
   310) !----------------------------------------------------------------------
   311) !     ... limit temperature
   312) !----------------------------------------------------------------------
   313)       do k = 1,pver
   314)          h2so4m(:ncol,k)  = 0._r8
   315)          temp(:ncol,k)    = max( temp_floor,temper(:ncol,k) )
   316)       end do
   317) 
   318) !----------------------------------------------------------------------
   319) !     ... total HNO3 and H2O gas and condensed phases
   320) !----------------------------------------------------------------------
   321)       do k = sad_topp,pver
   322)          hno3_total(:,k) = hno3_gas(:,k) + hno3_cond(:,k,1) + hno3_cond(:,k,2) 
   323) 	 h2o_total(:,k)  = h2o_gas(:,k)  + h2o_cond(:,k)
   324)          hcl_total (:,k) = hcl_gas(:,k)  + hcl_cond(:,k)
   325)       end do
   326) 
   327) !======================================================================
   328) !======================================================================
   329) !     ... Set SAD to SAGEII if Temperature is GT 200K and Return
   330) !
   331) !     ... mask_lbs  = true  .... T > 200K or SAD_SULF <= 1e-15 or 
   332) !                                P < 2hPa or P > 300hPa
   333) !     ... mask_sts  = false .... .not. mask_lbs
   334) !     ... mask_nat  = false .... T <= TSAT_NAT
   335) !     ... mask_ice  = false .... H2O_COND > 0.0
   336) !======================================================================
   337) !======================================================================
   338) 
   339)       do k = sad_topp,pver
   340)          mask_lbs(:,k) = temp(:ncol,k) > 200._r8 .or. sad_sage(:,k) <= 1.e-15_r8 &
   341)                          .or. press(:ncol,k) < 2._r8 .or. press(:ncol,k) > 300._r8
   342)       end do
   343) 
   344) sage_sad : &
   345)       if( any( mask_lbs(:,sad_topp:pver) ) ) then
   346)          do k = sad_topp,pver
   347)             where( mask_lbs(:,k) )
   348) 	       sad_strat(:,k,1)    = sad_sage(:,k)
   349) 	       sad_strat(:,k,2)    = 0._r8
   350)                sad_strat(:,k,3)    = 0._r8
   351)             endwhere
   352)          end do
   353) !----------------------------------------------------------------------
   354) !     ... Calculate Liquid Binary Sulfate Radius
   355) !----------------------------------------------------------------------
   356)          call calc_radius_lbs( ncol, mask_lbs, sad_sage, radius_lbs )
   357)          do k = sad_topp,pver
   358)             where( mask_lbs(:,k) )
   359)                radius_strat(:,k,1)     = radius_lbs(:,k)
   360) 	       radius_strat(:,k,2)     = 0._r8
   361) 	       radius_strat(:,k,3)     = 0._r8
   362) 	       hno3_gas    (:,k)       = hno3_total(:,k)
   363)                hno3_cond   (:,k,1)     = 0._r8
   364) 	       hno3_cond   (:,k,2)     = 0._r8
   365) 	       hcl_gas     (:,k)       = hcl_total(:,k)
   366)                hcl_cond    (:,k)       = 0._r8
   367)             endwhere
   368)          end do
   369)          if( all( mask_lbs(:,sad_topp:pver) ) ) then
   370)             call outfld( 'H2SO4M_C', h2so4m(:ncol,:), ncol, lchnk )
   371) 	    return
   372)          end if
   373)       end if sage_sad
   374) 
   375) !======================================================================
   376) !======================================================================
   377) !     ... Logic for deriving ICE
   378) !         Ice formation occurs here if condensed phase H2O exists.
   379) !
   380) !         mask_lbs  = false.... T > 200K or SAD_SULF < 1e-15 or 
   381) !                               P >2hPa or P <300hPa
   382) !         mask_ice  = true .... H2O_COND > 0.0
   383) !======================================================================
   384) !======================================================================
   385)       do k = sad_topp,pver
   386) 	 do i = 1,ncol
   387) 	    if( .not. mask_lbs(i,k) ) then
   388)                mask_ice(i,k) = h2o_cond(i,k) > 0._r8
   389) 	    else
   390) 	       mask_ice(i,k) = .false.
   391) 	    end if
   392)          end do
   393)       end do
   394) 
   395) all_ice : &
   396)       if( any( mask_ice(:,sad_topp:pver) ) ) then
   397)          do k = sad_topp,pver
   398)             where( mask_ice(:,k) )
   399) 	       h2o_avail(:,k) = h2o_cond(:,k)
   400)             endwhere
   401)          end do
   402) !----------------------------------------------------------------------
   403) !        ... ICE 
   404) !----------------------------------------------------------------------
   405)          call ice_sad_calc( ncol, press, temp, m, h2o_avail, &
   406) 			    sad_ice, radius_ice, mask_ice )
   407) 
   408)          do k = sad_topp,pver
   409)             where( mask_ice(:,k) )
   410)                sad_strat   (:,k,3) = sad_ice       (:,k)
   411)                radius_strat(:,k,3) = radius_ice    (:,k)
   412)             endwhere
   413)          end do
   414)       end if all_ice
   415) 
   416) !======================================================================
   417) !======================================================================
   418) !  	... LOGIC for STS and NAT 
   419) !
   420) !           mask_lbs = false .... T > 200K or SAD_SULF <= 1e-15 or 
   421) !                                 P < 2hPa or P >300hPa
   422) !           mask_sts = true  .... not  mask_lbs
   423) !           mask_nat = true  .... T <= TSAT_NAT and  mask_sts = true
   424) !======================================================================
   425) !======================================================================
   426) 
   427)       do k = sad_topp,pver
   428) 	 do i = 1,ncol
   429) 	    if( .not. mask_lbs(i,k) ) then
   430)                mask_sts(i,k) = .true.
   431) 	    else
   432) 	       mask_sts(i,k) = .false.
   433) 	    end if
   434)          end do
   435)       end do
   436) !----------------------------------------------------------------------
   437) !  	... STS (80% of total HNO3 logic)
   438) !----------------------------------------------------------------------
   439) !        NOTE: STS only sees 80% of the total HNO3 (Wegner et al., JGR, 2013a)
   440) sts_nat_sad : &
   441)       if( any( mask_sts(:,sad_topp:pver) ) ) then
   442)          do k = sad_topp,pver
   443)             where( mask_sts(:,k) )
   444)                h2o_avail (:,k)   = h2o_gas   (:,k)
   445)                hno3_avail(:,k)   = hno3_total(:,k)*eighty_percent

   498) 	    if( any(mask_nat(:,k)) ) then

   515) ! NOTE:  Add in gas-phase from STS with gas-phase from NAT

   518)                sad_strat   (:,k,2) = sad_nat       (:,k)

   543)       subroutine nat_sat_temp( ncol, hno3_total, h2o_avail, press, tsat_nat, mask )
   544) 
   545)       implicit none
   546) 
   547) !----------------------------------------------------------------------
   548) !	... dummy arguments
   549) !----------------------------------------------------------------------
   550)       integer,  intent(in)   :: ncol
   551)       real(r8), intent(in)   :: press(ncol,pver)
   552)       real(r8), intent(in)   :: h2o_avail(ncol,pver)
   553)       real(r8), intent(in)   :: hno3_total(ncol,pver)
   554)       real(r8), intent(out)  :: tsat_nat(ncol,pver)
   555)       logical,  intent(in)   :: mask(ncol,pver)
   556) 
   557) !----------------------------------------------------------------------
   558) !	... local variables
   559) !----------------------------------------------------------------------
   560)       real(r8), parameter :: ssrNAT = 10.0_r8
   561)       real(r8), parameter :: ssrNATi = .1_r8
   562)       real(r8), parameter :: aa = -2.7836_r8, &
   563)                              bb = -0.00088_r8, &
   564)                              cc = 38.9855_r8, &
   565)                              dd = -11397.0_r8, &
   566)                              ee = 0.009179_r8
   567)       integer  :: k, i
   568)       real(r8) :: bbb                     ! temporary variable
   569)       real(r8) :: ccc                     ! temporary variable
   570)       real(r8) :: wrk                     ! temporary variable
   571)       real(r8) :: tmp                     ! temporary variable
   572)       real(r8) :: phno3                   ! hno3 partial pressure
   573)       real(r8) :: ph2o                    ! h2o  partial pressure
   574) 
   575)       tsat_nat(:,:) = 0._r8
   576) 
   577) !----------------------------------------------------------------------
   578) !     	... Derive HNO3 and H2O partial pressure (torr)
   579) !          where: 0.7501 = 760/1013.
   580) !----------------------------------------------------------------------
   581)       do k = sad_topp,pver
   582)          do i = 1,ncol
   583)             if( mask(i,k) ) then
   584) 
   585)                bbb   = press(i,k) * .7501_r8
   586)                phno3 = hno3_total(i,k) * bbb
   587)                ph2o  = h2o_avail(i,k)  * bbb
   588) 
   589)                if( phno3 > 0._r8 ) then
   590) !----------------------------------------------------------------------
   591) !     	... Calculate NAT Saturation Threshold Temperature
   592) !           Hanson and Mauersberger: GRL, Vol.15, 8, p855-858, 1988.
   593) !           Substitute m(T) and b(T) into Equation (1). Rearrange and
   594) !           solve quadratic eqation. 
   595) !----------------------------------------------------------------------
   596)                   tmp = log10( ph2o )
   597)                   wrk = 1._r8 / (ee + bb*tmp)
   598)                   bbb = (aa*tmp - log10( phno3*ssrNATi ) + cc) * wrk
   599)                   ccc = dd *wrk
   600)                   tsat_nat(i,k) = .5_r8 * (-bbb + sqrt( bbb*bbb - 4._r8*ccc ))
   601)                endif
   602)             endif
   603)          enddo
   604)       end do
   605) 
   606)       end subroutine nat_sat_temp

   648) 
   649)       implicit none
   650) 
   651) !----------------------------------------------------------------------
   652) !	... dummy arguments

   708)       subroutine sulfate_sad_calc( ncol, press, temp, h2o_avail, hno3_avail, hcl_avail, &
   709)                                    sad_sage, m, hno3_gas, hno3_cond, &
   710)                                    hcl_gas, hcl_cond, sad_sulfate, &
   711)                                    radius_sulfate, mask, lchnk, flag, h2so4m, is_chem)
   712)       implicit none
   713) 
   714) !----------------------------------------------------------------------
   715) !	... dummy arguments
   716) !----------------------------------------------------------------------
   717)       integer, intent(in)   :: ncol
   718)       integer, intent(in)   :: lchnk, flag
   719)       real(r8), intent(in)  :: temp       (pcols,pver)
   720)       real(r8), intent(in)  :: press      (ncol,pver)
   721)       real(r8), intent(in)  :: m          (ncol,pver)
   722)       real(r8), intent(in)  :: h2o_avail  (ncol,pver)
   723)       real(r8), intent(in)  :: hno3_avail (ncol,pver)
   724)       real(r8), intent(in)  :: hcl_avail  (ncol,pver)   
   725)       real(r8), intent(in)  :: sad_sage   (ncol,pver)
   726)       real(r8), intent(out) :: hno3_gas   (ncol,pver)       ! Gas-phase HNO3, mole fraction
   727)       real(r8), intent(out) :: hno3_cond  (ncol,pver)       ! Condensed phase HNO3, mole fraction
   728)       real(r8), intent(out) :: hcl_gas    (ncol,pver)       ! Gas-phase HCL, mole fraction
   729)       real(r8), intent(out) :: hcl_cond   (ncol,pver)       ! Condensed phase HCL, mole fraction
   730)       real(r8), intent(out) :: sad_sulfate(ncol,pver)   
   731)       real(r8), intent(out) :: radius_sulfate(ncol,pver)
   732)       real(r8), intent(inout) :: h2so4m   (ncol,pver)       ! mass per volume, micro grams m-3
   733)       logical, intent(in)   :: is_chem                      ! chemistry calc switch
   734)       logical, intent(in)   :: mask       (ncol,pver)
   735) 
   736) !----------------------------------------------------------------------
   737) !	... local variables
   738) !----------------------------------------------------------------------
   739)       real(r8), parameter :: t_limit           = 200._r8
   740)       real(r8), parameter :: avo_num           = 6.02214e23_r8
   741)       real(r8), parameter :: mwh2so4           = 98.076_r8
   742)       real(r8), parameter :: sigma_sulfate     = 1.6_r8
   743)       real(r8), parameter :: sulfate_part_dens = 10._r8
   744) 
   745)       integer  :: i, k
   746)       integer  :: cnt1, cnt2
   747)       real(r8) :: ratio
   748)       real(r8) :: vals(2)
   749)       real(r8) :: h2so4_aer_dens  (ncol,pver)  ! grams cm-3 solution
   750)       real(r8) :: h2so4_cond      (ncol,pver)  ! Condensed H2SO4 (moles cm-3 air)
   751)       real(r8) :: sulfate_vol_dens(ncol,pver)  ! Volume Density, cm3 aerosol  cm-3 air
   752)       real(r8) :: wtf             (ncol,pver)  ! weight fraction of H2SO4 in ternary soln
   753)       real(r8) :: wts             (ncol,pver)  ! weight percent of ternary solution
   754) 
   755) ! Carslaw HCl solubility
   756)       real(r8) :: wts0            (ncol,pver)  ! weight percent of H2SO4 is LBS
   757)       real(r8) :: wtn             (ncol,pver)  ! weight percent of HNO3 in STS
   758)       real(r8) :: ch2so4          (ncol,pver)  ! Total H2SO4 (moles / cm3 of air)
   759)       real(r8) :: molh2so4        (ncol,pver)  ! Equil molality of H2SO4 in STS
   760)       real(r8) :: molhno3         (ncol,pver)  ! Equil molality of HNO3 in STS
   761)       real(r8) :: AD              (ncol,pver)  ! air density (molecules cm-3)
   762)       real(r8) :: xmf             (ncol,pver)  ! 
   763)       real(r8) :: hhcl            (ncol,pver)  ! henry's solubility of hcl in binary
   764)       real(r8) :: phcl0           (ncol,pver)  ! partial pressure of hcl (hPa)
   765)       real(r8) :: h2so4vmr        (ncol,pver)  ! atmospheric mole fraction of H2SO4
   766)       real(r8) :: nsul            (ncol,pver)  ! moles / m3- H2SO4 pure liquid 
   767)       real(r8) :: mcl             (ncol,pver)  ! molality of hcl in ?
   768)       real(r8) :: wtcl            (ncol,pver)  ! 
   769)       real(r8) :: phcl            (ncol,pver)  ! partial pressure of hcl (over aerosol)
   770)       real(r8) :: parthcl         (ncol,pver)  ! fraction of HCl in gas-phase
   771) !
   772)       real(r8) :: packer          (ncol*pver)
   773)       logical  :: do_equil                     ! local mask
   774)       logical  :: mask_lt         (ncol,pver)  ! local temperature mask
   775)       logical  :: maskx           (ncol,pver)
   776)       logical  :: converged       (ncol,pver)  ! EQUIL convergence test
   777) 
   778)       hcl_gas (:,:) = 0.0_r8
   779)       hcl_cond(:,:) = 0.0_r8
   780)       parthcl (:,:) = 0.0_r8
   781)       phcl0   (:,:) = 0.0_r8
   782) 
   783)       do k = sad_topp,pver
   784)          mask_lt(:,k)  = mask(:,k)
   785)       end do
   786) 
   787) !----------------------------------------------------------------------
   788) !  	... derive H2SO4 (micro grams / m3) from SAGEII SAD
   789) !----------------------------------------------------------------------
   790)       call sad2h2so4( h2o_avail, press, sad_sage, temp, sulfate_vol_dens, & 
   791)                       h2so4_aer_dens, h2so4m, mask, ncol )
   792) 
   793) !----------------------------------------------------------------------
   794) !  	... limit h2so4m
   795) !----------------------------------------------------------------------
   796) 
   797)          do k = sad_topp,pver
   798)             do i = 1,ncol
   799)                if( mask(i,k) ) then
   800)                   if( h2so4m(i,k) <= 0._r8 ) then
   801)                      h2so4m(i,k) = 1.e-2_r8
   802)                   end if
   803)                end if
   804)             end do
   805)          end do
   806) 
   807)       if( is_chem ) then
   808)       else
   809)         do k = sad_topp,pver
   810)             where( mask(:ncol,k) )
   811)                mask_lt(:ncol,k) = temp(:ncol,k) < t_limit
   812)             end where
   813)          end do
   814)       end if
   815) 
   816)       if( is_chem ) then
   817)          do_equil = .true.
   818)       else
   819)          do_equil = any( mask_lt(:,sad_topp:pver) )
   820)       end if
   821) 
   822) !----------------------------------------------------------------------
   823) !     .... Calculate the ternary soln volume density
   824) !----------------------------------------------------------------------
   825)       if( do_equil ) then
   826) 
   827)          call equil( temp, h2so4m, hno3_avail, h2o_avail, press, & 
   828)                      hno3_cond, h2so4_cond, wts, wtn, wts0, molh2so4, molhno3, mask_lt, ncol, &
   829)                      lchnk, flag, is_chem, converged )       
   830)   
   831)          do k = sad_topp,pver
   832) 
   833) 	   where( ( mask_lt(:ncol,k) ) .AND. ( converged(:ncol,k) ) )
   834) 
   835) !----------------------------------------------------------------------
   836) !     .... convert h2o, hno3 from moles cm-3 air to molecules cm-3 air
   837) !----------------------------------------------------------------------
   838)                hno3_cond(:ncol,k) = min( hno3_cond(:ncol,k),hno3_avail(:ncol,k) )
   839)                hno3_gas(:ncol,k)  = hno3_avail(:ncol,k) - hno3_cond(:ncol,k)
   840) 
   841) !----------------------------------------------------------------------
   842) !     .... Derive ternary volume density (cm3 aer / cm3 air)
   843) !----------------------------------------------------------------------
   844)                wtf(:ncol,k) = .01_r8* wts(:ncol,k)
   845)                sulfate_vol_dens(:ncol,k) = h2so4_cond(:ncol,k)*mwh2so4/(wtf(:ncol,k)*h2so4_aer_dens(:ncol,k))
   846) 
   847) ! Carslaw solubility
   848) !----------------------------------------------------------------------
   849) !     .... Partition HCl (gas/condensed) *** Carslaw
   850) !----------------------------------------------------------------------
   851) !          THE SOLUBILITY OF HCL 
   852) !          HHCl (MOL/KG/ATM) taken form Shi et al., JGR 2001
   853) !
   854) 
   855) !     .... Convert weight % to weight fraction 
   856)                 wtn(:ncol,k) = wtn(:ncol,k) * 0.01_r8 
   857)                 wts0(:ncol,k) = wts0(:ncol,k) * 0.01_r8
   858) 
   859) !     .... Derive xmf (mole fraction H2SO4 in LBS )
   860)                 xmf(:ncol,k) = (wts0(:ncol,k)*100.0_r8)/((wts0(:ncol,k)*100.0_r8)+ &
   861)                            (100.0_r8-(wts0(:ncol,k)*100._r8))*98.0_r8/18.0_r8)
   862) 
   863) !     .... Derive hhcl (henry's solubility of hcl in binary)
   864)                 hhcl(:ncol,k) = (0.094_r8-0.61_r8*xmf(:ncol,k)+1.2_r8*xmf(:ncol,k)**2.0_r8) &
   865)                             *exp(-8.68_r8+(8515.0_r8-10718.0_r8*xmf(:ncol,k)**(0.7_r8))/temp(:ncol,k)) 
   866) 
   867) !     .... Derive phcl0 (partial pressure of hcl( hPa))
   868)                 phcl0(:ncol,k) = hcl_avail(:ncol,k)*press(:ncol,k) / 1013.26_r8
   869) 
   870) !     .... Derive H2SO4 vmr (h2so4_cond = mole / cm-3)
   871)                 AD(:ncol,k) = (6.022098e23_r8 * press(:ncol,k) / 1013.26_r8) &
   872)                            / (temp(:ncol,k)*8.2058e-2_r8*1000.0_r8)
   873)                 h2so4vmr(:ncol,k)  = (h2so4_cond(:ncol,k)*6.022098e23_r8) / AD(:ncol,k)
   874) 
   875) !     .... Derive nsul (moles / m3 H2SO4 pure liquid )
   876)                 nsul(:ncol,k)  = h2so4vmr(:ncol,k) * press(:ncol,k) * 100.0_r8 / 8.314_r8 / temp(:ncol,k)
   877) 
   878) !     .... Derive  mcl (molality of hcl)

   886)                 phcl(:ncol,k) = mcl(:ncol,k)/hhcl(:ncol,k) 
   887)

   889)                 where(phcl0(:ncol,k)>0._r8)

   892)                   parthcl(:ncol,k) = 0._r8
   893)                 endwhere
   894) 
   895) !     .... Partition HCl (gas/condensed)  
   896)                 hcl_gas (:ncol,k)  = hcl_avail(:ncol,k) * parthcl(:ncol,k)

   898) 
   899)             elsewhere
   900)                hno3_cond(:ncol,k) = 0.0_r8
   901)                hno3_gas(:ncol,k)  = hno3_avail(:ncol,k)
   902)                hcl_cond (:ncol,k) = 0.0_r8
   903)                hcl_gas  (:ncol,k) = hcl_avail(:ncol,k)
   904) 
   905)             endwhere

   911) 	 where( mask(:,k) )

  1020)       subroutine nat_cond( ncol, press, temp, h2o_avail, hno3_avail, &
  1021) 			   hno3_gas, hno3_cond, mask )
  1022) 
  1023)       implicit none
  1024) 
  1025) !----------------------------------------------------------------------
  1026) ! 	... dummy arguments
  1027) !----------------------------------------------------------------------
  1028)       integer, intent(in)   :: ncol
  1029)       real(r8), intent(in)  :: press(ncol,pver)
  1030)       real(r8), intent(in)  :: temp(pcols,pver)
  1031)       real(r8), intent(in)  :: h2o_avail(ncol,pver)
  1032)       real(r8), intent(in)  :: hno3_avail(ncol,pver)
  1033)       real(r8), intent(out) :: hno3_gas(ncol,pver)
  1034)       real(r8), intent(out) :: hno3_cond(ncol,pver)
  1035)       logical, intent(in)   :: mask(ncol,pver)
  1036) 
  1037) !----------------------------------------------------------------------
  1038) ! 	... local variables
  1039) !----------------------------------------------------------------------
  1040)       real(r8), parameter ::  aa = -2.7836_r8,  &
  1041)                               bb = -0.00088_r8, &
  1042)                               cc = 38.9855_r8,  &
  1043)                               dd = -11397.0_r8, &
  1044)                               ee = 0.009179_r8
  1045) 
  1046)       integer  :: i, k
  1047)       real(r8) :: bt                                 ! temporary variable
  1048)       real(r8) :: mt                                 ! temporary variable
  1049)       real(r8) :: t                                  ! temporary variable
  1050)       real(r8) :: logPhno3                           ! temporary variable
  1051)       real(r8) :: phno3                              ! hno3 partial pressure
  1052)       real(r8) :: ph2o                               ! h2o  partial pressure
  1053)       real(r8) :: phno3_eq                           ! partial pressure above NAT
  1054)       real(r8) :: wrk      
  1055)       
  1056)       do k = sad_topp,pver
  1057) 	 do i = 1,ncol
  1058) !----------------------------------------------------------------------
  1059) !     .... Derive HNO3 and H2O partial pressure (torr)
  1060) !          where: 0.7501 = 760/1013.
  1061) !----------------------------------------------------------------------        
  1062) 	    if( mask(i,k) ) then
  1063)                wrk   = press(i,k) * .7501_r8
  1064)                phno3 = hno3_avail(i,k) * wrk
  1065)                ph2o  = h2o_avail(i,k)  * wrk
  1066) !----------------------------------------------------------------------
  1067) !     Calculating the temperature coefficients for the variation of HNO3
  1068) !     and H2O vapor pressure (torr) over a trihydrate solution of HNO3/H2O
  1069) !     The coefficients are taken from Hanson and Mauersberger: 
  1070) !     GRL, Vol.15, 8, p855-858, 1988.
  1071) !----------------------------------------------------------------------
  1072) 	       t   = temp(i,k)
  1073)                bt  = cc + dd/t + ee*t
  1074)                mt  = aa + bb*t
  1075)   
  1076)                logphno3 = mt*log10( ph2o ) + bt
  1077)                phno3_eq = 10._r8**logphno3
  1078) 
  1079) 	       if( phno3 > phno3_eq ) then
  1080)                   wrk            = 1._r8 / wrk
  1081)                   hno3_cond(i,k) = (phno3 - phno3_eq) * wrk
  1082)                   hno3_gas(i,k)  = phno3_eq * wrk
  1083) 	       else
  1084)                   hno3_cond(i,k) = 0._r8
  1085)                   hno3_gas(i,k)  = hno3_avail(i,k)
  1086) 	       end if
  1087) 	    end if
  1088)          end do
  1089)       end do
  1090) 
  1091)       end subroutine nat_cond

  1125) !     	... Calculate the volume density (cm3 aerosol / cm3 air)
  1126) !----------------------------------------------------------------------
  1127)       do k = sad_topp,pver
  1128) 	 do i = 1,ncol
  1129) 	    if( mask(i,k) ) then
  1130)                lbs_vol_dens(i,k) = ((sad_sage(i,k)**1.5_r8)/3._r8)/sqrt( four_pi*lbs_part_dens ) &
  1131)                                    *exp( 1.5_r8*(log( sigma_lbs ))**2 )

  1134) !----------------------------------------------------------------------
  1135) !     	... DEK, added a minimum to temperature
  1136) !----------------------------------------------------------------------
  1137)                p   = h2o(i,k) * press(i,k) * .7501_r8

  1140) 	       do l = 1,6
  1141)                   c(l) = exp( a(1,l) + tr*(a(2,l) + tr*(a(3,l) + tr*(a(4,l) + tr*a(5,l)))) )

  1146)                wts0 = max( 0._r8,c(1) + p*(-c(2) + p*(c(3) + p*(-c(4) + p*(c(5) - p*c(6))))) )

  1157) !----------------------------------------------------------------------
  1158)                h2so4m(i,k) = lbs_vol_dens(i,k)*h2so4_aer_dens(i,k)*wts0*1.e12_r8
  1159) 	    end if
  1160)          end do
  1161)       end do
