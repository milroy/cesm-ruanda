    54) function deep_scheme_does_scav_trans()
    55) !
    56) ! Function called by tphysbc to determine if it needs to do scavenging and convective transport
    57) ! or if those have been done by the deep convection scheme. Each scheme could have its own
    58) ! identical query function for a less-knowledgable interface but for now, we know that KE 
    59) ! does scavenging & transport, and ZM doesn't
    60) !
    61) 
    62)   logical deep_scheme_does_scav_trans
    63) 
    64)   deep_scheme_does_scav_trans = .false.
    65) 
    66)   if ( deep_scheme .eq. 'KE' ) deep_scheme_does_scav_trans = .true.
    67) 
    68)   return
    69) 
    70) end function deep_scheme_does_scav_trans

   113) subroutine convect_deep_init(pref_edge)
   114) 
   115) !----------------------------------------
   116) ! Purpose:  declare output fields, initialize variables needed by convection
   117) !----------------------------------------
   118) 
   119)   use cam_history,   only: phys_decomp, addfld                          
   120)   use pmgrid,        only: plevp
   121)   use spmd_utils,    only: masterproc
   122)   use zm_conv_intr,  only: zm_conv_init
   123)   use abortutils,    only: endrun
   124)   
   125)   use physics_buffer, only: physics_buffer_desc, pbuf_get_index
   126) 
   127)   implicit none
   128) 
   129)   real(r8),intent(in) :: pref_edge(plevp)        ! reference pressures at interfaces
   130) 
   131)   select case ( deep_scheme )
   132)   case('off')
   133)      if (masterproc) write(iulog,*)'convect_deep: no deep convection selected'
   134)   case('ZM')
   135)      if (masterproc) write(iulog,*)'convect_deep initializing Zhang-McFarlane convection'
   136)      call zm_conv_init(pref_edge)
   137)   case('UNICON')
   138)      if (masterproc) write(iulog,*)'convect_deep: deep convection done by UNICON'
   139)   case default
   140)      if (masterproc) write(iulog,*)'WARNING: convect_deep: no deep convection scheme. May fail.'
   141)   end select
   142) 
   143)   cldtop_idx = pbuf_get_index('CLDTOP')
   144)   cldbot_idx = pbuf_get_index('CLDBOT')
   145)   cld_idx    = pbuf_get_index('CLD')
   146)   fracis_idx = pbuf_get_index('FRACIS')
   147) 
   148)   pblh_idx   = pbuf_get_index('pblh')
   149)   tpert_idx  = pbuf_get_index('tpert')
   150) 
   151)   call addfld ('ICWMRDP  ', 'kg/kg   ', pver, 'A', 'Deep Convection in-cloud water mixing ratio '            ,phys_decomp)
   152) 
   153) end subroutine convect_deep_init

   157) subroutine convect_deep_tend( &
   158)      mcon    ,cme     ,          &
   159)      dlf     ,pflx    ,zdu      , &
   160)      rliq    , &
   161)      ztodt   , &
   162)      state   ,ptend   ,landfrac ,pbuf)
   163) 
   164) 
   165)    use physics_types, only: physics_state, physics_ptend, physics_tend, physics_ptend_init
   166)    
   167)    use cam_history,    only: outfld
   168)    use constituents,   only: pcnst
   169)    use zm_conv_intr,   only: zm_conv_tend
   170)    use cam_history,    only: outfld
   171)    use physconst,      only: cpair
   172)    use physics_buffer, only: physics_buffer_desc, pbuf_get_field
   173) 
   174) ! Arguments
   175)    type(physics_state), intent(in ) :: state   ! Physics state variables
   176)    type(physics_ptend), intent(out) :: ptend   ! individual parameterization tendencies
   177)    
   178) 
   179)    type(physics_buffer_desc), pointer :: pbuf(:)
   180)    real(r8), intent(in) :: ztodt               ! 2 delta t (model time increment)
   181)    real(r8), intent(in) :: landfrac(pcols)     ! Land fraction
   182)       
   183) 
   184)    real(r8), intent(out) :: mcon(pcols,pverp)  ! Convective mass flux--m sub c
   185)    real(r8), intent(out) :: dlf(pcols,pver)    ! scattrd version of the detraining cld h2o tend
   186)    real(r8), intent(out) :: pflx(pcols,pverp)  ! scattered precip flux at each level
   187)    real(r8), intent(out) :: cme(pcols,pver)    ! cmf condensation - evaporation
   188)    real(r8), intent(out) :: zdu(pcols,pver)    ! detraining mass flux
   189) 
   190)    real(r8), intent(out) :: rliq(pcols) ! reserved liquid (not yet in cldliq) for energy integrals
   191) 
   192)    real(r8), pointer :: prec(:)   ! total precipitation
   193)    real(r8), pointer :: snow(:)   ! snow from ZM convection 
   194) 
   195)    real(r8), pointer, dimension(:) :: jctop
   196)    real(r8), pointer, dimension(:) :: jcbot
   197)    real(r8), pointer, dimension(:,:,:) :: cld        
   198)    real(r8), pointer, dimension(:,:) :: ql        ! wg grid slice of cloud liquid water.
   199)    real(r8), pointer, dimension(:,:) :: rprd      ! rain production rate
   200)    real(r8), pointer, dimension(:,:,:) :: fracis  ! fraction of transported species that are insoluble
   201) 
   202)    real(r8), pointer, dimension(:,:) :: evapcdp   ! Evaporation of deep convective precipitation
   203) 
   204)    real(r8), pointer :: pblh(:)                ! Planetary boundary layer height
   205)    real(r8), pointer :: tpert(:)               ! Thermal temperature excess 
   206) 
   207)   real(r8) zero(pcols, pver)
   208) 
   209)   integer i, k
   210) 
   211)    real(r8), pointer, dimension(:,:) :: ttend_dp
   212) 
   213)    call pbuf_get_field(pbuf, cldtop_idx,  jctop )
   214)    call pbuf_get_field(pbuf, cldbot_idx,  jcbot )
   215)    call pbuf_get_field(pbuf, icwmrdp_idx, ql    )
   216) 
   217)   select case ( deep_scheme )
   218)   case('off', 'UNICON') ! in UNICON case the run method is called from convect_shallow_tend
   219)     zero = 0     
   220)     mcon = 0
   221)     dlf = 0
   222)     pflx = 0
   223)     cme = 0
   224)     zdu = 0
   225)     rliq = 0
   226) 
   227)     call physics_ptend_init(ptend, state%psetcols, 'convect_deep')
   228) 
   229) !
   230) ! Associate pointers with physics buffer fields
   231) !
   232) 
   233)     call pbuf_get_field(pbuf, cld_idx,         cld,    start=(/1,1/),   kount=(/pcols,pver/) ) 
   234)     call pbuf_get_field(pbuf, rprddp_idx,      rprd )
   235)     call pbuf_get_field(pbuf, fracis_idx,      fracis, start=(/1,1,1/), kount=(/pcols, pver, pcnst/) )
   236)     call pbuf_get_field(pbuf, nevapr_dpcu_idx, evapcdp )
   237)     call pbuf_get_field(pbuf, prec_dp_idx,     prec )
   238)     call pbuf_get_field(pbuf, snow_dp_idx,     snow )
   239) 
   240)     prec=0
   241)     snow=0
   242) 
   243)     jctop = pver
   244)     jcbot = 1._r8
   245)     cld = 0
   246)     ql = 0
   247)     rprd = 0
   248)     fracis = 0
   249)     evapcdp = 0
   250) 
   251)   case('ZM') !    1 ==> Zhang-McFarlane (default)
   252)      call pbuf_get_field(pbuf, pblh_idx,  pblh)
   253)      call pbuf_get_field(pbuf, tpert_idx, tpert)
   254) 
   255)      call zm_conv_tend( pblh    ,mcon    ,cme     , &
   256)           tpert   ,dlf     ,pflx    ,zdu      , &
   257)           rliq    , &
   258)           ztodt   , &
   259)           jctop, jcbot , &
   260)           state   ,ptend   ,landfrac, pbuf)
   261) 
   262)   end select
   263) 
   264)   ! If we added this, set it.
   265)   if (ttend_dp_idx > 0) then
   266)      call pbuf_get_field(pbuf, ttend_dp_idx, ttend_dp)
   267)      ttend_dp(:state%ncol,:pver) = ptend%s(:state%ncol,:pver)/cpair
   268)   end if
   269) 
   270)   call outfld( 'ICWMRDP ', ql  , pcols, state%lchnk )
   271) 
   272) 
   273) end subroutine convect_deep_tend
