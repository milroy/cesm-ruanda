     6) ! !MODULE: accFldsMod
     7) !
     8) ! !DESCRIPTION:
     9) ! This module contains subroutines that initialize, update and extract
    10) ! the user-specified fields over user-defined intervals. Each interval
    11) ! and accumulation type is unique to each field processed.
    12) ! Subroutine [initAccumFlds] defines the fields to be processed
    13) ! and the type of accumulation. Subroutine [updateAccumFlds] does
    14) ! the actual accumulation for a given field. Fields are accumulated
    15) ! by calls to subroutine [update_accum_field]. To accumulate a field,
    16) ! it must first be defined in subroutine [initAccumFlds] and then
    17) ! accumulated by calls to [updateAccumFlds].
    18) ! Four types of accumulations are possible:
    19) !   o average over time interval
    20) !   o running mean over time interval
    21) !   o running accumulation over time interval
    22) ! Time average fields are only valid at the end of the averaging interval.
    23) ! Running means are valid once the length of the simulation exceeds the
    24) ! averaging interval. Accumulated fields are continuously accumulated.
    25) ! The trigger value "-99999." resets the accumulation to zero.
    26) !
    27) ! !USES:
    28)   use shr_kind_mod, only: r8 => shr_kind_r8
    29)   use abortutils,   only: endrun
    30)   use clm_varctl,   only: iulog, use_cndv, use_cn, use_crop

    35)   save
    36) !

    38)   public :: initAccFlds     ! Initialization accumulator fields
    39)   public :: initAccClmtype  ! Initialize clmtype variables obtained from accum fields
    40)   public :: updateAccFlds   ! Update accumulator fields
    41) !

    52) ! !IROUTINE: initAccFlds()

    54) ! !INTERFACE:
    55)   subroutine initAccFlds()
    56) !
    57) ! !DESCRIPTION:
    58) ! Initializes accumulator and sets up array of accumulated fields
    59) !
    60) ! !USES:
    61)     use accumulMod   , only : init_accum_field, print_accum_fields
    62)     use clm_time_manager , only : get_step_size
    63)     use shr_const_mod, only : SHR_CONST_CDAY, SHR_CONST_TKFRZ
    64) !
    65) ! !ARGUMENTS:
    66)     implicit none
    67) !
    68) ! !REVISION HISTORY::
    69) ! Created by M. Vertenstein 03/2003
    70) !
    71) !
    72) ! !LOCAL VARIABLES:
    73) !EOP
    74) !
    75)     integer :: dtime                     !time step size
    76)     integer, parameter :: not_used = huge(1)
    77) !------------------------------------------------------------------------
    78) 
    79)     ! Hourly average of 2m temperature.
    80) 
    81)     dtime = get_step_size()
    82)     call init_accum_field(name='TREFAV', units='K', &
    83)          desc='average over an hour of 2-m temperature', &
    84)          accum_type='timeavg', accum_period=nint(3600._r8/dtime), &
    85)          subgrid_type='pft', numlev=1, init_value=0._r8)
    86) 
    87)     ! Hourly average of Urban 2m temperature.
    88) 
    89)     call init_accum_field(name='TREFAV_U', units='K', &
    90)          desc='average over an hour of urban 2-m temperature', &
    91)          accum_type='timeavg', accum_period=nint(3600._r8/dtime), &
    92)          subgrid_type='pft', numlev=1, init_value=0._r8)
    93) 
    94)     ! Hourly average of Rural 2m temperature.
    95) 
    96)     call init_accum_field(name='TREFAV_R', units='K', &
    97)          desc='average over an hour of rural 2-m temperature', &
    98)          accum_type='timeavg', accum_period=nint(3600._r8/dtime), &
    99)          subgrid_type='pft', numlev=1, init_value=0._r8)
   100) 
   101)     ! 24hr average of vegetation temperature (heald, 04/06)
   102)     call init_accum_field (name='T_VEG24', units='K', &
   103)          desc='24hr average of vegetation temperature', &
   104)          accum_type='runmean', accum_period=-1, &
   105)          subgrid_type='pft', numlev=1, init_value=0._r8)
   106) 
   107)     ! 240hr average of vegetation temperature (heald, 04/06)
   108)     call init_accum_field (name='T_VEG240', units='K', &
   109)          desc='240hr average of vegetation temperature', &
   110)          accum_type='runmean', accum_period=-10, &
   111)          subgrid_type='pft', numlev=1, init_value=0._r8)
   112) 
   113)     ! 24hr average of direct solar radiation (heald, 04/06)
   114)     call init_accum_field (name='FSD24', units='W/m2', &
   115)          desc='24hr average of direct solar radiation', &
   116)          accum_type='runmean', accum_period=-1, &
   117)          subgrid_type='pft', numlev=1, init_value=0._r8)
   118) 
   119)     ! 240hr average of direct solar radiation (heald, 04/06)
   120)     call init_accum_field (name='FSD240', units='W/m2', &
   121)          desc='240hr average of direct solar radiation', &
   122)          accum_type='runmean', accum_period=-10, &
   123)          subgrid_type='pft', numlev=1, init_value=0._r8)
   124) 
   125)     ! 24hr average of diffuse solar radiation (heald, 04/06)
   126)     call init_accum_field (name='FSI24', units='W/m2', &
   127)          desc='24hr average of diffuse solar radiation', &
   128)          accum_type='runmean', accum_period=-1, &
   129)          subgrid_type='pft', numlev=1, init_value=0._r8)
   130) 
   131)     ! 240hr average of diffuse solar radiation (heald, 04/06)
   132)     call init_accum_field (name='FSI240', units='W/m2', &
   133)          desc='240hr average of diffuse solar radiation', &
   134)          accum_type='runmean', accum_period=-10, &
   135)          subgrid_type='pft', numlev=1, init_value=0._r8)
   136) 
   137)     ! 24hr average of fraction of canopy that is sunlit (heald, 04/06)
   138)     call init_accum_field (name='FSUN24', units='fraction', &
   139)          desc='24hr average of diffuse solar radiation', &
   140)          accum_type='runmean', accum_period=-1, &
   141)          subgrid_type='pft', numlev=1, init_value=0._r8)
   142) 
   143)     ! 240hr average of fraction of canopy that is sunlit (heald, 04/06)
   144)     call init_accum_field (name='FSUN240', units='fraction', &
   145)          desc='240hr average of diffuse solar radiation', &
   146)          accum_type='runmean', accum_period=-10, &
   147)          subgrid_type='pft', numlev=1, init_value=0._r8)
   148) 
   149)     ! Average of LAI from previous and current timestep (heald, 04/06)
   150)     call init_accum_field (name='LAIP', units='m2/m2', &
   151)          desc='leaf area index average over timestep', &
   152)          accum_type='runmean', accum_period=1, &
   153)          subgrid_type='pft', numlev=1, init_value=0._r8)
   154) 
   155)     if (use_cndv .or. use_crop) then
   156)        ! The following is a running mean.
   157)        ! The accumulation period is set to -10 for a 10-day running mean.
   158)        
   159)        call init_accum_field (name='T10', units='K', &
   160)             desc='10-day running mean of 2-m temperature', &
   161)             accum_type='runmean', accum_period=-10, &
   162)             subgrid_type='pft', numlev=1,init_value=SHR_CONST_TKFRZ+20._r8)
   163)        
   164)     end if
   165) 
   166)     if (use_cndv) then
   167)        ! 30-day average of 2m temperature.
   168)        
   169)        call init_accum_field (name='TDA', units='K', &
   170)             desc='30-day average of 2-m temperature', &
   171)             accum_type='timeavg', accum_period=-30, &
   172)             subgrid_type='pft', numlev=1, init_value=0._r8)
   173)        
   174)        ! The following are running means.
   175)        ! The accumulation period is set to -365 for a 365-day running mean.
   176)        
   177)        call init_accum_field (name='PREC365', units='MM H2O/S', &
   178)             desc='365-day running mean of total precipitation', &
   179)             accum_type='runmean', accum_period=-365, &
   180)             subgrid_type='pft', numlev=1, init_value=0._r8)
   181)        
   182)        ! The following are accumulated fields.
   183)        ! These types of fields are accumulated until a trigger value resets
   184)        ! the accumulation to zero (see subroutine update_accum_field).
   185)        ! Hence, [accper] is not valid.
   186) 
   187)        call init_accum_field (name='AGDDTW', units='K', &
   188)             desc='growing degree-days base twmax', &
   189)             accum_type='runaccum', accum_period=not_used, &
   190)             subgrid_type='pft', numlev=1, init_value=0._r8)
   191)        
   192)        call init_accum_field (name='AGDD', units='K', &
   193)             desc='growing degree-days base 5C', &
   194)             accum_type='runaccum', accum_period=not_used,  &
   195)             subgrid_type='pft', numlev=1, init_value=0._r8)
   196)     end if
   197) 
   198)     if ( crop_prog )then
   199)        ! 10-day average of min 2m temperature.
   200) 
   201)        call init_accum_field (name='TDM10', units='K', &
   202)             desc='10-day running mean of min 2-m temperature', &
   203)             accum_type='runmean', accum_period=-10, &
   204)             subgrid_type='pft', numlev=1, init_value=SHR_CONST_TKFRZ)
   205) 
   206)        ! 5-day average of min 2m temperature.
   207) 
   208)        call init_accum_field (name='TDM5', units='K', &
   209)             desc='5-day running mean of min 2-m temperature', &
   210)             accum_type='runmean', accum_period=-5, &
   211)             subgrid_type='pft', numlev=1, init_value=SHR_CONST_TKFRZ)
   212) 
   213)        ! All GDD summations are relative to the planting date
   214)        ! (Kucharik & Brye 2003)
   215) 
   216)        call init_accum_field (name='GDD0', units='K', &
   217)             desc='growing degree-days base 0C from planting', &
   218)             accum_type='runaccum', accum_period=not_used, &
   219)             subgrid_type='pft', numlev=1, init_value=0._r8)
   220) 
   221)        call init_accum_field (name='GDD8', units='K', &
   222)             desc='growing degree-days base 8C from planting', &
   223)             accum_type='runaccum', accum_period=not_used, &
   224)             subgrid_type='pft', numlev=1, init_value=0._r8)
   225) 
   226)        call init_accum_field (name='GDD10', units='K', &
   227)             desc='growing degree-days base 10C from planting', &
   228)             accum_type='runaccum', accum_period=not_used,  &
   229)             subgrid_type='pft', numlev=1, init_value=0._r8)
   230) 
   231)        call init_accum_field (name='GDDPLANT', units='K', &
   232)             desc='growing degree-days from planting', &
   233)             accum_type='runaccum', accum_period=not_used,  &
   234)             subgrid_type='pft', numlev=1, init_value=0._r8)
   235) 
   236)        call init_accum_field (name='GDDTSOI', units='K', &
   237)             desc='growing degree-days from planting (top two soil layers)', &
   238)             accum_type='runaccum', accum_period=not_used,  &
   239)             subgrid_type='pft', numlev=1, init_value=0._r8)
   240)     end if
   241) 
   242)     ! Print output of accumulated fields
   243) 
   244)     call print_accum_fields()
   245) 
   246)   end subroutine initAccFlds

   433)     elai_p           => pvs%elai_p            ! (heald 04/06)
   434)     t_veg            => pes%t_veg 	           ! (heald 04/06)
   435)     fsun             => pps%fsun 	           ! (heald 04/06)
   436)     elai             => pps%elai 	           ! (heald 04/06)
   437) 
   438)     ! Determine calendar information
   439) 
   440)     dtime = get_step_size()
   441)     nstep = get_nstep()
   442)     call get_curr_date (year, month, day, secs)
   443) 
   444)     ! Don't do any accumulation if nstep is zero
   445)     ! (only applies to coupled or cam mode)
   446) 
   447)     if (nstep == 0) return
   448) 
   449)     ! NOTE: currently only single level pft fields are used below
   450)     ! Variables are declared above that should make it easy to incorporate
   451)     ! multi-level or single-level fields of any subgrid type
   452) 
   453)     ! Allocate needed dynamic memory for single level pft field
   454) 
   455)     allocate(rbufslp(begp:endp), stat=ier)
   456)     if (ier/=0) then
   457)        write(iulog,*)'update_accum_hist allocation error for rbuf1dp'
   458)        call endrun
   459)     endif
   460) 
   461)     ! Accumulate and extract TREFAV - hourly average 2m air temperature
   462)     ! Used to compute maximum and minimum of hourly averaged 2m reference
   463)     ! temperature over a day. Note that "spval" is returned by the call to
   464)     ! accext if the time step does not correspond to the end of an
   465)     ! accumulation interval. First, initialize the necessary values for
   466)     ! an initial run at the first time step the accumulator is called
   467) 
   468)     call update_accum_field  ('TREFAV', t_ref2m, nstep)
   469)     call extract_accum_field ('TREFAV', rbufslp, nstep)
   470)     end_cd = is_end_curr_day()
   471)     do p = begp,endp
   472)        if (rbufslp(p) /= spval) then
   473)           t_ref2m_max_inst(p) = max(rbufslp(p), t_ref2m_max_inst(p))
   474)           t_ref2m_min_inst(p) = min(rbufslp(p), t_ref2m_min_inst(p))
   475)        endif
   476)        if (end_cd) then
   477)           t_ref2m_max(p) = t_ref2m_max_inst(p)
   478)           t_ref2m_min(p) = t_ref2m_min_inst(p)
   479)           t_ref2m_max_inst(p) = -spval
   480)           t_ref2m_min_inst(p) =  spval
   481)        else if (secs == int(dtime)) then
   482)           t_ref2m_max(p) = spval
   483)           t_ref2m_min(p) = spval
   484)        endif
   485)     end do
   486) 
   487)     ! Accumulate and extract TREFAV_U - hourly average urban 2m air temperature
   488)     ! Used to compute maximum and minimum of hourly averaged 2m reference
   489)     ! temperature over a day. Note that "spval" is returned by the call to
   490)     ! accext if the time step does not correspond to the end of an
   491)     ! accumulation interval. First, initialize the necessary values for
   492)     ! an initial run at the first time step the accumulator is called
   493) 
   494)     call update_accum_field  ('TREFAV_U', t_ref2m_u, nstep)
   495)     call extract_accum_field ('TREFAV_U', rbufslp, nstep)
   496)     do p = begp,endp
   497)        l = plandunit(p)
   498)        if (rbufslp(p) /= spval) then
   499)           t_ref2m_max_inst_u(p) = max(rbufslp(p), t_ref2m_max_inst_u(p))
   500)           t_ref2m_min_inst_u(p) = min(rbufslp(p), t_ref2m_min_inst_u(p))
   501)        endif
   502)        if (end_cd) then
   503)          if (urbpoi(l)) then
   504)           t_ref2m_max_u(p) = t_ref2m_max_inst_u(p)
   505)           t_ref2m_min_u(p) = t_ref2m_min_inst_u(p)
   506)           t_ref2m_max_inst_u(p) = -spval
   507)           t_ref2m_min_inst_u(p) =  spval
   508)          end if
   509)        else if (secs == int(dtime)) then
   510)           t_ref2m_max_u(p) = spval
   511)           t_ref2m_min_u(p) = spval
   512)        endif
   513)     end do
   514) 
   515)     ! Accumulate and extract TREFAV_R - hourly average rural 2m air temperature
   516)     ! Used to compute maximum and minimum of hourly averaged 2m reference
   517)     ! temperature over a day. Note that "spval" is returned by the call to
   518)     ! accext if the time step does not correspond to the end of an
   519)     ! accumulation interval. First, initialize the necessary values for
   520)     ! an initial run at the first time step the accumulator is called
   521) 
   522)     call update_accum_field  ('TREFAV_R', t_ref2m_r, nstep)
   523)     call extract_accum_field ('TREFAV_R', rbufslp, nstep)
   524)     do p = begp,endp
   525)        l = plandunit(p)
   526)        if (rbufslp(p) /= spval) then
   527)           t_ref2m_max_inst_r(p) = max(rbufslp(p), t_ref2m_max_inst_r(p))
   528)           t_ref2m_min_inst_r(p) = min(rbufslp(p), t_ref2m_min_inst_r(p))
   529)        endif
   530)        if (end_cd) then
   531)          if (.not.(ifspecial(l))) then
   532)           t_ref2m_max_r(p) = t_ref2m_max_inst_r(p)
   533)           t_ref2m_min_r(p) = t_ref2m_min_inst_r(p)
   534)           t_ref2m_max_inst_r(p) = -spval
   535)           t_ref2m_min_inst_r(p) =  spval
   536)          end if
   537)        else if (secs == int(dtime)) then
   538)           t_ref2m_max_r(p) = spval
   539)           t_ref2m_min_r(p) = spval
   540)        endif
   541)     end do
   542) 
   543)     ! Accumulate and extract T_VEG24 & T_VEG240 (heald 04/06)
   544)     do p = begp,endp
   545)        rbufslp(p) = t_veg(p)
   546)     end do
   547)     call update_accum_field  ('T_VEG24', rbufslp, nstep)
   548)     call extract_accum_field ('T_VEG24', t_veg24, nstep)
   549)     call update_accum_field  ('T_VEG240', rbufslp, nstep)
   550)     call extract_accum_field ('T_VEG240', t_veg240, nstep)
   551) 
   552)     ! Accumulate and extract forc_solad24 & forc_solad240 (heald 04/06)
   553)     do p = begp,endp
   554)        g = pgridcell(p)
   555)        rbufslp(p) = forc_solad(g,1)
   556)     end do
   557)     call update_accum_field  ('FSD240', rbufslp, nstep)
   558)     call extract_accum_field ('FSD240', fsd240, nstep)
   559)     call update_accum_field  ('FSD24', rbufslp, nstep)
   560)     call extract_accum_field ('FSD24', fsd24, nstep)
   561) 
   562)     ! Accumulate and extract forc_solai24 & forc_solai240 (heald 04/06)
   563)     do p = begp,endp
   564)        g = pgridcell(p)
   565)        rbufslp(p) = forc_solai(g,1)
   566)     end do
   567)     call update_accum_field  ('FSI24', rbufslp, nstep)
   568)     call extract_accum_field ('FSI24', fsi24, nstep)
   569)     call update_accum_field  ('FSI240', rbufslp, nstep)
   570)     call extract_accum_field ('FSI240', fsi240, nstep)
   571) 
   572)     ! Accumulate and extract fsun24 & fsun240 (heald 04/06)
   573)     do p = begp,endp
   574)        rbufslp(p) = fsun(p)
   575)     end do
   576)     call update_accum_field  ('FSUN24', rbufslp, nstep)
   577)     call extract_accum_field ('FSUN24', fsun24, nstep)
   578)     call update_accum_field  ('FSUN240', rbufslp, nstep)
   579)     call extract_accum_field ('FSUN240', fsun240, nstep)
   580) 
   581)     ! Accumulate and extract elai_p (heald 04/06)
   582)     do p = begp,endp
   583)        rbufslp(p) = elai(p)
   584)     end do
   585)     call update_accum_field  ('LAIP', rbufslp, nstep)
   586)     call extract_accum_field ('LAIP', elai_p, nstep)
   587) 
   588)     if (use_cndv .or. use_crop) then
   589)        ! Accumulate and extract T10
   590)        !(acumulates TSA as 10-day running mean)
   591)        
   592)        call update_accum_field  ('T10', t_ref2m, nstep)
   593)        call extract_accum_field ('T10', t10, nstep)
   594)     end if
   595) 
   596)     if (use_cndv) then
   597)        ! Accumulate and extract TDA
   598)        ! (accumulates TBOT as 30-day average)
   599)        ! Also determine t_mo_min
   600)        
   601)        do p = begp,endp
   602)           g = pgridcell(p)
   603)           rbufslp(p) = forc_t(g)
   604)        end do
   605)        call update_accum_field  ('TDA', rbufslp, nstep)
   606)        call extract_accum_field ('TDA', rbufslp, nstep)
   607)        do p = begp,endp
   608)           t_mo(p) = rbufslp(p)
   609)           t_mo_min(p) = min(t_mo_min(p), rbufslp(p))
   610)        end do
   611)        
   612)        ! Accumulate and extract PREC365
   613)        ! (accumulates total precipitation as 365-day running mean)
   614)        
   615)        do p = begp,endp
   616)           g = pgridcell(p)
   617)           rbufslp(p) = forc_rain(g) + forc_snow(g)
   618)        end do
   619)        call update_accum_field  ('PREC365', rbufslp, nstep)
   620)        call extract_accum_field ('PREC365', prec365, nstep)
   621)        
   622)        ! Accumulate growing degree days based on 10-day running mean temperature.
   623)        ! The trigger to reset the accumulated values to zero is -99999.
   624)        
   625)        ! Accumulate and extract AGDDTW (gdd base twmax, which is 23 deg C
   626)        ! for boreal woody pfts)
   627)        
   628)        do p = begp,endp
   629)           rbufslp(p) = max(0._r8, (t10(p) - SHR_CONST_TKFRZ - twmax(ndllf_dcd_brl_tree)) &
   630)                * dtime/SHR_CONST_CDAY)
   631)           if (month==1 .and. day==1 .and. secs==int(dtime)) rbufslp(p) = -99999._r8
   632)        end do
   633)        call update_accum_field  ('AGDDTW', rbufslp, nstep)
   634)        call extract_accum_field ('AGDDTW', agddtw, nstep)
   635)        
   636)        ! Accumulate and extract AGDD
   637)        
   638)        do p = begp,endp
   639)           rbufslp(p) = max(0.0_r8, (t_ref2m(p) - (SHR_CONST_TKFRZ + 5.0_r8)) &
   640)                * dtime/SHR_CONST_CDAY)
   641)        end do
   642)        call update_accum_field  ('AGDD', rbufslp, nstep)
   643)        call extract_accum_field ('AGDD', agdd, nstep)
   644)     end if
   645) 
   646)     if ( crop_prog )then
   647)        ! Accumulate and extract TDM10
   648) 
   649)        do p = begp,endp
   650)           rbufslp(p) = min(t_ref2m_min(p),t_ref2m_min_inst(p)) !slevis: ok choice?
   651)           if (rbufslp(p) > 1.e30_r8) rbufslp(p) = SHR_CONST_TKFRZ !and were 'min'&
   652)        end do                                         !'min_inst' not initialized?
   653)        call update_accum_field  ('TDM10', rbufslp, nstep)
   654)        call extract_accum_field ('TDM10', a10tmin, nstep)
   655) 
   656)        ! Accumulate and extract TDM5
   657) 
   658)        do p = begp,endp
   659)           rbufslp(p) = min(t_ref2m_min(p),t_ref2m_min_inst(p)) !slevis: ok choice?
   660)           if (rbufslp(p) > 1.e30_r8) rbufslp(p) = SHR_CONST_TKFRZ !and were 'min'&
   661)        end do                                         !'min_inst' not initialized?
   662)        call update_accum_field  ('TDM5', rbufslp, nstep)
   663)        call extract_accum_field ('TDM5', a5tmin, nstep)
   664) 
   665)        ! Accumulate and extract GDD0
   666) 
   667)        do p = begp,endp
   668)           itypveg = itype(p)
   669)           g = pgridcell(p)
   670)           if (month==1 .and. day==1 .and. secs==int(dtime)) then
   671)              rbufslp(p) = -99999._r8 ! reset gdd
   672)           else if (( month > 3 .and. month < 10 .and. latdeg(g) >= 0._r8) .or. &
   673)                    ((month > 9 .or.  month < 4) .and. latdeg(g) <  0._r8)     ) then
   674)              rbufslp(p) = max(0._r8, min(26._r8, t_ref2m(p)-SHR_CONST_TKFRZ)) &
   675)                           * dtime/SHR_CONST_CDAY
   676)           else
   677)              rbufslp(p) = 0._r8      ! keeps gdd unchanged at other times (eg, through Dec in NH)
   678)           end if
   679)        end do
   680)        call update_accum_field  ('GDD0', rbufslp, nstep)
   681)        call extract_accum_field ('GDD0', gdd0, nstep)
   682) 
   683)        ! Accumulate and extract GDD8
   684) 
   685)        do p = begp,endp
   686)           itypveg = itype(p)
   687)           g = pgridcell(p)
   688)           if (month==1 .and. day==1 .and. secs==int(dtime)) then
   689)              rbufslp(p) = -99999._r8 ! reset gdd
   690)           else if (( month > 3 .and. month < 10 .and. latdeg(g) >= 0._r8) .or. &
   691)                    ((month > 9 .or.  month < 4) .and. latdeg(g) <  0._r8)     ) then
   692)              rbufslp(p) = max(0._r8, min(30._r8, &
   693)                                          t_ref2m(p)-(SHR_CONST_TKFRZ + 8._r8))) &
   694)                           * dtime/SHR_CONST_CDAY
   695)           else
   696)              rbufslp(p) = 0._r8      ! keeps gdd unchanged at other times (eg, through Dec in NH)
   697)           end if
   698)        end do
   699)        call update_accum_field  ('GDD8', rbufslp, nstep)
   700)        call extract_accum_field ('GDD8', gdd8, nstep)
   701) 
   702)        ! Accumulate and extract GDD10
   703) 
   704)        do p = begp,endp
   705)           itypveg = itype(p)
   706)           g = pgridcell(p)
   707)           if (month==1 .and. day==1 .and. secs==int(dtime)) then
   708)              rbufslp(p) = -99999._r8 ! reset gdd
   709)           else if (( month > 3 .and. month < 10 .and. latdeg(g) >= 0._r8) .or. &
   710)                    ((month > 9 .or.  month < 4) .and. latdeg(g) <  0._r8)     ) then
   711)              rbufslp(p) = max(0._r8, min(30._r8, &
   712)                                          t_ref2m(p)-(SHR_CONST_TKFRZ + 10._r8))) &
   713)                           * dtime/SHR_CONST_CDAY
   714)           else
   715)              rbufslp(p) = 0._r8      ! keeps gdd unchanged at other times (eg, through Dec in NH)
   716)           end if
   717)        end do
   718)        call update_accum_field  ('GDD10', rbufslp, nstep)
   719)        call extract_accum_field ('GDD10', gdd10, nstep)
   720) 
   721)        ! Accumulate and extract GDDPLANT
   722) 
   723)        do p = begp,endp
   724)           if (croplive(p)) then ! relative to planting date
   725)              itypveg = itype(p)
   726)              rbufslp(p) = max(0._r8, min(mxtmp(itypveg), &
   727)                                          t_ref2m(p)-(SHR_CONST_TKFRZ + baset(itypveg)))) &
   728)                           * dtime/SHR_CONST_CDAY
   729)              if (itypveg == nwcereal) rbufslp(p) = rbufslp(p)*vf(p)
   730)           else
   731)              rbufslp(p) = -99999._r8
   732)           end if
   733)        end do
   734)        call update_accum_field  ('GDDPLANT', rbufslp, nstep)
   735)        call extract_accum_field ('GDDPLANT', gddplant, nstep)
   736) 
   737)        ! Accumulate and extract GDDTSOI
   738)        ! In agroibis this variable is calculated
   739)        ! to 0.05 m, so here we use the top two soil layers
   740)    
   741)        do p = begp,endp
   742)           if (croplive(p)) then ! relative to planting date
   743)              itypveg = itype(p)
   744)              c = pcolumn(p)
   745)              rbufslp(p) = max(0._r8, min(mxtmp(itypveg), &
   746)               ((t_soisno(c,1)*dz(c,1)+t_soisno(c,2)*dz(c,2))/(dz(c,1)+dz(c,2))) - &
   747)               (SHR_CONST_TKFRZ + baset(itypveg)))) * dtime/SHR_CONST_CDAY
   748)              if (itypveg == nwcereal) rbufslp(p) = rbufslp(p)*vf(p)
   749)           else
   750)              rbufslp(p) = -99999._r8
   751)           end if
   752)        end do
   753)        call update_accum_field  ('GDDTSOI', rbufslp, nstep)
   754)        call extract_accum_field ('GDDTSOI', gddtsoi, nstep)
   755) 
   756)     end if
   757) 
   758)     ! Deallocate dynamic memory
   759) 
   760)     deallocate(rbufslp)
   761) 
   762)   end subroutine updateAccFlds
